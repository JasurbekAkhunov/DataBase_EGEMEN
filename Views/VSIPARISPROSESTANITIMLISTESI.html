CREATE VIEW VSIPARISPROSESTANITIMLISTESI(
    SIPARISPROSESID,
    ADI,
    ADI2,
    TRANSFERKODU,
    FASONMUHGELIRHESAPKODU,
    SATISMUHGELIRHESAPKODU,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI)
AS
Select SiparisProses.SiparisProsesId, SiparisProses.Adi, SiparisProses.Adi2, IK.Adi, SiparisProses.TransferKodu, SiparisProses.FasonMuhGelirHesapKodu, SiparisProses.SatisMuhGelirHesapKodu, SiparisProses.InsertTarihi, SK.Adi, SiparisProses.DegisimTarihi from SiparisProses
  left join Kullanici IK on IK.KullaniciId  = SiparisProses.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = SiparisProses.KullaniciId
;
