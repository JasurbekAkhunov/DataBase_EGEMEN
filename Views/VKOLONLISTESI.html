CREATE VIEW VKOLONLISTESI(
    KOLONADI,
    TABLOADI)
AS
select cast(rdb$field_name as varchar(50)), cast(rdb$relation_name as varchar(50))  from rdb$relation_fields
where Position('$' in rdb$field_name) = 0 and Position('$' in rdb$relation_name) = 0
order by rdb$relation_name, rdb$field_position
