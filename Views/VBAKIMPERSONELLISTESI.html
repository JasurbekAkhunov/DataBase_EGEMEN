CREATE VIEW VBAKIMPERSONELLISTESI(
    PERSONELID,
    ADI,
    ORMEPERSONELI,
    IPLIKPERSONELI,
    IMALATCI,
    POSTABASI,
    BAKIMCI,
    AYARPERSONELI,
    BAKIMDAIRESIID,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI)
AS
Select Personel.PersonelId, Personel.Adi, Personel.OrmePersoneli,  Personel.IplikPersoneli,  Personel.Imalatci,  Personel.PostaBasi,  Personel.Bakimci,  Personel.AyarPersoneli,
     Personel.BakimDairesiId, IK.Adi, Personel.InsertTarihi, SK.Adi, Personel.DegisimTarihi from Personel
  left join Kullanici IK on IK.KullaniciId  = Personel.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = Personel.KullaniciId
  where Personel.Kapali = 0
;
