CREATE VIEW VRECETELISTESI(
    RECETEID,
    ADI,
    ILAVE,
    FIRMAID,
    FIRMAADI,
    RENKOKEY,
    CALISMATARIHI,
    SABLON,
    ORTAKRECETE,
    USTASAMAID,
    ASAMAADI,
    RECETETURUADI,
    SABLONRECETEID,
    NOTU,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI)
AS
Select Recete.ReceteId, Recete.Adi,
            cast (f_rTrim(case when (Ilave = 0) then ''
                               else 'Ä°lave'
                           end) as varchar (5)) Ilave   ,
            Recete.FirmaId, Firma.Adi FirmaAdi, Recete.RenkOkey, Recete.CalismaTarihi, Recete.Sablon,
            Recete.OrtakRecete, Recete.UstAsamaId, Asama.Adi, ReceteTuru.Adi, Recete.SablonReceteId, Recete.Notu,
            IK.Adi, Recete.InsertTarihi, SK.Adi, Recete.DegisimTarihi  from Recete
     left join Firma on Firma.FirmaId = Recete.FirmaId
     left join Asama on Asama.AsamaId = Recete.UstAsamaId
     left join ReceteTuru on ReceteTuru.ReceteTuruId = Recete.ReceteTuruId
    left join Kullanici IK on IK.KullaniciId  = Recete.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = Recete.KullaniciId
;
