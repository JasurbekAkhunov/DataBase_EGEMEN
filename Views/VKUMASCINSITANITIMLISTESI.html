CREATE VIEW VKUMASCINSITANITIMLISTESI(
    KUMASCINSIID,
    ADI,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI)
AS
Select KumasCinsi.KumasCinsiId, KumasCinsi.Adi, IK.Adi, KumasCinsi.InsertTarihi, SK.Adi, KumasCinsi.DegisimTarihi from KumasCinsi
  left join Kullanici IK on IK.KullaniciId  = KumasCinsi.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = KumasCinsi.KullaniciId
;
