CREATE VIEW VDESENLISTESI(
    DESENKAYITID,
    DESENID,
    DESENADI,
    FIRMAID,
    FIRMAADI,
    CALISMANO,
    TASARIMNO,
    ESKIDESENNO,
    DESENKATALOGID,
    DESENKATALOGADI,
    REFERANSDESENNO,
    REFERANSDESENADI,
    MUSTERIDESENNO1,
    MUSTERIDESENADI1,
    MUSTERIDESENNO2,
    MUSTERIDESENADI2,
    DESENDURUMU,
    DESENDURUMUSTR,
    SABLONDURUMUADI,
    FASON,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI,
    NOTU,
    KAPALI)
AS
Select Desen.DesenKayitId, Desen.DesenId, Desen.Adi, Desen.FirmaId, Firma.Adi, Desen.CalismaNo, Desen.TasarimNo, Desen.EskiDesenNo, Desen.DesenKatalogId, DesenKatalog.Adi, Desen.ReferansDesenNo, Desen.ReferansDesenAdi,
            Desen.MusteriDesenNo1, Desen.MusteriDesenAdi1, Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2,
            Desen.DesenDurumu,
            cast (f_rTrim(case when (DesenDurumu = 0) then 'Exclusive'
                               when (DesenDurumu = 1) then 'Açık'
                               else 'Coğrafi Bölge'
                           end) as varchar (20)) DesenDurumuStr,
            SablonDurumu.Adi, Desen.Fason, IK.Adi,
            Desen.InsertTarihi, SK.Adi, Desen.DegisimTarihi, Desen.Notu, Desen.Kapali from Desen
      left join Firma on Firma.FirmaID = Desen.FirmaId
      left join SablonDurumu on SablonDurumu.SablonDurumuID = Desen.SablonDurumuId
      left join DesenKatalog on DesenKatalog.DesenKatalogId = Desen.DesenKatalogId
      left join Kullanici IK on IK.KullaniciId  = Desen.InsertKullaniciId
      left join Kullanici SK on SK.KullaniciId  = Desen.KullaniciId
      where Desen.Kapali = 0
;
