CREATE VIEW VIADEICINUCIKISHAREKETLISTESI(
    STOKHAREKETID,
    IRSALIYENO,
    FIRMAID,
    FIRMAADI,
    IRSALIYETARIHI,
    RENKID,
    RENKADI,
    DEPOID,
    LOT,
    MIKTAR,
    HAMID,
    HAMADI,
    BOYASIPARISDETAYID,
    BOYASIPARISBASKIID)
AS
Select StokHareket.StokHareketID, (OrmeIrsaliye.SeriNo ||  OrmeIrsaliye.IrsaliyeNo) IrsaliyeNo, OrmeIrsaliye.FirmaID, Firma.Adi, OrmeIrsaliye.IrsaliyeTarihi, StokHareket.RenkID, Renk.Adi,
        StokHareket.DepoID, StokHareket.Lot, StokHareket.Miktar, StokHareket.HamID, Ham.Adi, StokHareket.BoyaSiparisDetayID, StokHareket.BoyaSiparisBaskiID  from StokHareket
     Left Join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeID = StokHareket.OrmeIrsaliyeID
     Left Join Ham on Ham.HamID = StokHareket.HamID
     Left Join Firma on Firma.FirmaID = OrmeIrsaliye.FirmaID
     Left Join Renk on Renk.RenkID = StokHareket.RenkID
     where StokHareket.HareketTipi = 6 and StokHareket.UretimIslemTipiID = 2 and StokHareket.BoyaTalimatKayitID is Null
;
