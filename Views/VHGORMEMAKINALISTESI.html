CREATE VIEW VHGORMEMAKINALISTESI(
    ORMEMAKINAID,
    ADI,
    MAKINAGRUPID,
    MARKASI,
    MODELI,
    YAPIMYILI,
    MAKINATIPIADI,
    UNITEADI,
    OTOMASYONADAHIL)
AS
Select OrmeMakina.OrmeMakinaId, OrmeMakina.Adi, OrmeMakina.MakinaGrupId, OrmeMakina.Markasi, OrmeMakina.Modeli,
            OrmeMakina.YapimYili, MakinaTipi.Adi MakinaTipiAdi, Unite.Adi UniteAdi,
            OrmeMakina.OtomasyonaDahil from OrmeMakina
     left join MakinaTipi on MakinaTipi.MakinaTipiId = OrmeMakina.MakinaTipiId
     left join Unite on Unite.UniteId = OrmeMakina.UniteId
     where OrmeMakina.HgMakinasi = 1 and OrmeMakina.Kapali = 0 and OrmeMakina.KesimMasasi = 0
;
