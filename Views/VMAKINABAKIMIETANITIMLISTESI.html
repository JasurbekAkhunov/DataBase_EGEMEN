CREATE VIEW VMAKINABAKIMIETANITIMLISTESI(
    MAKINABAKIMISEMRIID,
    ADI,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI,
    TAHMINIBASLANGIC,
    BOYAMAKINAID,
    ORMEMAKINAID,
    IPLIKMAKINAID,
    MAKINABAKIMTIPIADI,
    MAKINABAKIMBOLUMUADI,
    BAKIMDAIRESIADI)
AS
Select MakinaBakimIsEmri.MakinaBakimIsEmriId, MakinaBakimIsEmri.Adi, IK.Adi, MakinaBakimIsEmri.InsertTarihi, SK.Adi, MakinaBakimIsEmri.DegisimTarihi,
         MakinaBakimIsEmri.TahminiBaslangic, MakinaBakimIsEmri.MakinaID, MakinaBakimIsEmri.OrmeMakinaId, MakinaBakimIsEmri.IplikMakinaId,
         MakinaBakimTipi.Adi, MakinaBakimBolumu.Adi, BakimDairesi.adi from MakinaBakimIsEmri
  left join Kullanici IK on IK.KullaniciId  = MakinaBakimIsEmri.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = MakinaBakimIsEmri.KullaniciId
  left join MakinaBakimTipi on MakinaBakimTipi.MakinaBakimTipiId = MakinaBakimIsEmri.MakinaBakimTipiId
  left join MakinaBakimBolumu on MakinaBakimBolumu.MakinaBakimBolumuId = MakinaBakimIsEmri.MakinaBakimBolumuId
  left join BakimDairesi on BakimDairesi.BakimDairesiId = MakinaBakimIsEmri.BakimDairesiId
;
