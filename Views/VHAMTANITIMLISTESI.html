CREATE VIEW VHAMTANITIMLISTESI(
    HAMID,
    ADI,
    ADI2,
    ADI3,
    ADI4,
    REFERANSKODU,
    HAMTIPIID,
    HAMTIPIADI,
    KUMASCINSIADI,
    KAPALI,
    PUS,
    FEINE,
    GRAMAJ,
    TUPACIKENMAYLI,
    IPLIKBOYU1,
    IPLIKBOYU2,
    IPLIKBOYU3,
    IPLIKBOYU4,
    SATISTRANSFERKODU,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI,
    DOKUMADESENIADI,
    TRANSFERKODUFASON)
AS
Select Ham.HamId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Ham.ReferansKodu, Ham.HamTipiId, HamTipi.Adi, KumasCinsi.Adi, Ham.Kapali,
            Ham.Pus, Ham.Feine, Ham.Gramaj, Ham.TupAcikEnMayli, Ham.IplikBoyu1, Ham.IplikBoyu2, Ham.IplikBoyu3, Ham.IplikBoyu4, Ham.SatisTransferKodu,
            IK.Adi, Ham.InsertTarihi, SK.Adi, Ham.DegisimTarihi, DokumaDeseni.Adi DokumaDeseniAdi, Ham.FasonTransferKodu from Ham
     left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
     left join KumasCinsi on KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
     left join DokumaDeseni on DokumaDeseni.DokumaDeseniId = Ham.DokumaDeseniId
     left join Kullanici IK on IK.KullaniciId  = Ham.InsertKullaniciId
     left join Kullanici SK on SK.KullaniciId  = Ham.KullaniciId
;
