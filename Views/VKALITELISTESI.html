CREATE VIEW VKALITELISTESI(
    KALITEID,
    KALITEADI,
    PARTISINIFID,
    PARTIYIL,
    PARTIID,
    HAMADI,
    HAMTIPIID,
    DESENKODU,
    DESENADI,
    KAPALI,
    KARTELASTOKMEVCUDU,
    RECETEADI,
    RECETEKODU,
    EN,
    GRAMAJ,
    ILAVEISLEMLER,
    COMPOSITION,
    DESENVARYANTADI,
    ARTIKELNO,
    ARTIKELADI,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI,
    ACIKLAMA)
AS
Select Kalite.KaliteId, Kalite.KaliteAdi, Kalite.PartiSinifId, Kalite.PartiYil, Kalite.PartiId, Kalite.HamAdi, Ham.HamTipiId, Kalite.DesenKodu, Kalite.DesenAdi, Kalite.Kapali, Kalite.KartelaStokMevcudu,
            Kalite.ReceteAdi, Kalite.ReceteKodu, Kalite.EnIstenen, Kalite.GramajIstenen, Kalite.IlaveIslemler , Kalite.Composition, Kalite.DesenVaryantAdi, Kalite.ArtikelNo, Kalite.ArtikelAdi, IK.Adi, Kalite.InsertTarihi, SK.Adi, Kalite.DegisimTarihi, Kalite.Aciklama from Kalite
  left join Ham on Ham.HamId  = Kalite.HamId
  left join Kullanici IK on IK.KullaniciId  = Kalite.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = Kalite.KullaniciId
  where Kalite.Kapali = 0
;
