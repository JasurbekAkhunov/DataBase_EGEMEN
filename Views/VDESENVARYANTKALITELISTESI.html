CREATE VIEW VDESENVARYANTKALITELISTESI(
    KALITEID,
    KALITEADI,
    PARTISINIFID,
    PARTIYIL,
    PARTIID,
    HAMADI,
    HAMTIPIID,
    DESENKODU,
    DESENADI,
    KAPALI,
    DESENID,
    VARYANTADI,
    DESENVARYANTID,
    DESENVARYANTKALITEID)
AS
select DesenVaryantKalite.KaliteId, Kalite.KaliteAdi, Kalite.PartiSinifId, Kalite.PartiYil, Kalite.PartiId, Kalite.HamAdi,
    Ham.HamTipiId, Kalite.DesenKodu, Kalite.DesenAdi, Kalite.Kapali, DesenVaryant.DesenId,DesenVaryant.varyantadi, DesenVaryant.DesenVaryantId,DesenVaryantKalite.DesenVaryantKaliteId from DesenVaryantKalite
    left join DesenVaryant on DesenVaryant.DesenVaryantId = DesenVaryantKalite.DesenVaryantId
    left join Kalite on Kalite.KaliteId = DesenVaryantKalite.KaliteId
    left join Ham on Ham.HamId  = Kalite.HamId
;
