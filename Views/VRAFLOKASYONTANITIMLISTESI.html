CREATE VIEW VRAFLOKASYONTANITIMLISTESI(
    UYGUN,
    RAFLOKASYONID,
    ADI,
    SEPETSAYISI,
    KAPALI,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI)
AS
Select Cast(Iif(Count(Raf.SonRafLokasyonId) < RafLokasyon.SepetSayisi , 'E', 'H') as varchar(1)), RafLokasyon.RafLokasyonId, RafLokasyon.Adi, RafLokasyon.SepetSayisi,
RafLokasyon.Kapali, IK.Adi, RafLokasyon.InsertTarihi, SK.Adi, RafLokasyon.DegisimTarihi From RafLokasyon
Left Join Kullanici IK on IK.KullaniciId  = RafLokasyon.InsertKullaniciId
Left Join Kullanici SK on SK.KullaniciId  = RafLokasyon.KullaniciId
Left Join Raf On Raf.SonRafLokasyonId = RafLokasyon.RafLokasyonId
Group By RafLokasyon.RafLokasyonId, RafLokasyon.Adi, RafLokasyon.SepetSayisi, RafLokasyon.Kapali,
IK.Adi, RafLokasyon.InsertTarihi, SK.Adi, RafLokasyon.DegisimTarihi
;
