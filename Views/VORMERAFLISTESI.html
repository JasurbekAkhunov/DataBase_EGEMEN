CREATE VIEW VORMERAFLISTESI(
    RAFID,
    ADI,
    DEPOID,
    RAFGRUPID,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI)
AS
Select Raf.RafId, Raf.Adi, DepoRaf.DepoId, Raf.RafGrupId, IK.Adi, Raf.InsertTarihi, SK.Adi, Raf.DegisimTarihi from Raf
  left join DepoRaf on DepoRaf.RafId  = Raf.RafId
  left join Depo on Depo.DepoId  = DepoRaf.DepoId
  left join Unite on Unite.UniteId  = Depo.UniteId
  left join Kullanici IK on IK.KullaniciId  = Raf.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = Raf.KullaniciId
  where (Unite.OrmeUnitesimi = 1 or Unite.HGUnitesimi = 1) and Raf.Kapali = 0
;
