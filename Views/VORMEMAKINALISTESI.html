CREATE VIEW VORMEMAKINALISTESI(
    ORMEMAKINAID,
    ADI,
    MAKINAGRUPID,
    PUS,
    FEINE,
    IGNESAYISI,
    SISTEMSAYISI,
    MARKASI,
    MODELI,
    YAPIMYILI,
    MAKINATIPIADI,
    MAKINALYCRATIPI,
    DESEN,
    MAKINAPLAKATIPIID,
    UNITEADI,
    OTOMASYONADAHIL,
    MASTERNO,
    KANALNO,
    BPMAKINASI)
AS
Select OrmeMakina.OrmeMakinaId, OrmeMakina.Adi, OrmeMakina.MakinaGrupId, OrmeMakina.Pus, OrmeMakina.Feine, OrmeMakina.IgneSayisi, OrmeMakina.SistemSayisi,
            OrmeMakina.Markasi, OrmeMakina.Modeli, OrmeMakina.YapimYili, MakinaTipi.Adi MakinaTipiAdi,
            cast (f_rTrim(case when (MakinaLycraTipiId = 0) then 'Yok'
                               when (MakinaLycraTipiId = 1) then 'Yarım'
                               else 'Tam'
                           end) as varchar (5)) MakinaLycraTipi,
            cast (f_rTrim(case when (Desen = 0) then ''
                               else 'Var'
                           end) as varchar (5)) Desen,
            cast (f_rTrim(case when (MakinaPlakaTipiId = 0) then 'Tek Plaka'
                               when (MakinaPlakaTipiId = 1) then 'Çift Plaka'
                               else ''
                           end) as varchar (11)) MakinaPlakaTipi, Unite.Adi UniteAdi,
            OrmeMakina.OtomasyonaDahil, OrmeMakina.MasterNo, OrmeMakina.KanalNo, OrmeMakina.BPMakinasi from OrmeMakina
     left join MakinaTipi on MakinaTipi.MakinaTipiId = OrmeMakina.MakinaTipiId
     left join Unite on Unite.UniteId = OrmeMakina.UniteId
     where OrmeMakina.Kapali = 0 and OrmeMakina.HgMakinasi = 0
;
