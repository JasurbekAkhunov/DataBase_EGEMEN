CREATE VIEW VAMBALAJLISTESI(
    AMBALAJID,
    ADI,
    AMBALAJSEKLIID,
    AMBALAJSEKLIADI,
    BOBINTOPSAYISI,
    KONIKROLIKAGIRLIGI,
    POSETAGIRLIGI,
    CUVALAGIRLIGI,
    CUVALIPIAGIRLIGI,
    PLASTIKPLAKAAGIRLIGI,
    KOLIAGIRLIGI,
    ARAKARTONAGIRLIGI,
    PALETAGIRLIGI,
    SEPERATORSAYISI,
    SEPERATORAGIRLIGI,
    STRECHAGIRLIGI,
    AMBALAJAGIRLIGI,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI)
AS
Select Ambalaj.AmbalajId, Ambalaj.Adi, Ambalaj.AmbalajSekliId, AmbalajSekli.Adi, Ambalaj.BobinTopSayisi, Ambalaj.KoniKrolikAgirligi,
Ambalaj.PosetAgirligi, Ambalaj.CuvalAgirligi, Ambalaj.CuvalIpiAgirligi, Ambalaj.PlastikPlakaAgirligi, Ambalaj.KoliAgirligi, Ambalaj.ArakarTonAgirligi,
Ambalaj.PaletAgirligi, Ambalaj.SeperatorSayisi, Ambalaj.SeperatorAgirligi, Ambalaj.StrechAgirligi, Ambalaj.AmbalajAgirligi,
IK.Adi, Ambalaj.InsertTarihi, SK.Adi, Ambalaj.DegisimTarihi From Ambalaj
Join AmbalajSekli On AmbalajSekli.AmbalajSekliId = Ambalaj.AmbalajSekliId
Left Join Kullanici IK on IK.KullaniciId  = Ambalaj.InsertKullaniciId
Left Join Kullanici SK on SK.KullaniciId  = Ambalaj.KullaniciId
;
