Definition

CREATE VIEW VURUNLISTESI(
    URUNID,
    ADI,
    ADI2,
    URUNGRUP1,
    RAF,
    DEPOID,
    URUNTIPIID,
    KIMYASALTIPIID,
    KIMYASALTIPIADI,
    STOKURETICIADI,
    URETICIURUNKODU,
    PARCANO,
    ANAURUNID,
    KATALOGNO,
    MODELID,
    MODELPARCAID,
    TARTIMYAPILACAK,
    KULLANILDIGIFIRMAID,
    KULLANILDIGIFIRMAADI)
AS
Select Urun.UrunId, Urun.Adi, Urun.Adi2, Urun.urunGrup1, Urun.Raf, Urun.MutfakDepoNo, Urun.UrunTipiId, Urun.KimyasalTipiID, KimyasalTipi.Adi,
  StokUretici.Adi, UreticiUrunKodu, ParcaNo, AnaUrunID, KatalogNo, Model.ModelID, ModelParcaId, Urun.TartimYapilacak, Urun.KullanildigiFirmaId, Firma.Adi from Urun
  left join KimyasalTipi on KimyasalTipi.KimyasalTipiID = Urun.KimyasalTipiId
  left join StokUretici on StokUretici.StokUreticiID = Urun.StokUreticiID
  left join Model on Model.ModelKayitId = Urun.ModelKayitId
  left join Firma on Firma.FirmaId = Urun.KullanildigiFirmaId
  where Urun.Kapali = 0
;
