CREATE VIEW URUNRECETELISTESI(
    URUNRECETEURUNID,
    ADI,
    BIRIMID)
AS
Select UrunRecete.UrunReceteUrunId, Urun.Adi, UrunBirim.BirimId  from UrunRecete
     left join Urun on Urun.UrunId = UrunRecete.UrunReceteUrunId
     left join UrunBirim on UrunBirim.UrunBirimId = UrunRecete.UrunBirimId
;
