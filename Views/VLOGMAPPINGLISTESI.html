CREATE VIEW VLOGMAPPINGLISTESI(
    LOGMAPPINGID,
    ADI,
    TABLOADI,
    TIPISTR,
    TABLOKOSULALANI,
    TABLOKOSULDEGERI,
    KAPALI,
    INSERTKULLANICIADI,
    INSERTTARIHI,
    KULLANICIADI,
    DEGISIMTARIHI)
AS
Select LogMapping.LogMappingId, LogMapping.Adi, LogMapping.Tabloadi,
Case LogMapping.Tipi
    When 0 Then 'Tümü'
    When 1 Then 'Silinenler'
    When 2 Then 'Güncellenenler'
    Else ''
End, LogMapping.TabloKosulAlani, LogMapping.TabloKosulDegeri,
LogMapping.Kapali, Ik.Adi, LogMapping.InsertTarihi, Sk.Adi, LogMapping.DegisimTarihi From LogMapping
Left Join Kullanici IK on IK.KullaniciId = LogMapping.InsertKullaniciId
Left Join Kullanici SK on SK.KullaniciId = LogMapping.KullaniciId
;
