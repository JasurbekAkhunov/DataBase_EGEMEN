CREATE VIEW VHGDEPOLISTESI(
    DEPOID,
    ADI,
    HGYARIMAMUL)
AS
Select Depo.DepoId, Depo.Adi, Depo.HGYariMamul from Depo
  left join Unite on Unite.UniteId  = Depo.UniteId
  where Unite.HGUnitesimi = 1 and Depo.Kapali = 0
;
