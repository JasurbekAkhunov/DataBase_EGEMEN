CREATE VIEW VRAFLOKASYONHAREKETLISTESI(
    RAFLOKASYONHAREKETID,
    RAFID,
    RAFADI,
    RAFLOKASYONADI,
    INSERTKULLANICIADI,
    INSERTTARIHI)
AS
Select RafLokasyonHareket.RafLokasyonHareketId, Raf.RafId, Raf.Adi, RafLokasyon.Adi, Kullanici.Adi, RafLokasyonHareket.InsertTarihi From RafLokasyonHareket
Left Join Raf on RafLokasyonHareket.RafId = Raf.RafId
Left Join RafLokasyon on RafLokasyon.RafLokasyonId = RafLokasyonHareket.RafLokasyonId
Left Join Kullanici on Kullanici.KullaniciId  = RafLokasyonHareket.InsertKullaniciId
Order By RafLokasyonHareket.RafId Asc, RafLokasyonHareket.InsertTarihi Desc
;
