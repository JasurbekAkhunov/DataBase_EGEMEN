CREATE VIEW VTABLOLISTESI(
    TABLOADI)
AS
select cast(rdb$relation_name as varchar(50)) from rdb$relations where rdb$view_blr is null
and (rdb$system_flag is null or rdb$system_flag = 0) and Position('$' in rdb$relation_name) = 0
order by rdb$relation_name
