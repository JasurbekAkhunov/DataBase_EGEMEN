CREATE TRIGGER BEFOREINSERTORTRANSFERFISIDETAY FOR ORMERAFTRANSFERFISIDETAY
ACTIVE BEFORE INSERT POSITION 0
as
declare variable KalanMiktar                   Numeric(12,3);
declare variable HamId                         integer;
declare variable RenkId                        integer;
declare variable BoyaSiparisKayitId            integer;
declare variable OrmeSiparisDetayId            integer;
declare variable BoyaSiparisDetayId            integer;
declare variable BoyaSiparisBaskiId            integer;
declare variable OrmePlanlamaId                integer;
declare variable DokumaPlanlamaId              integer;
declare variable Sakat                         integer;
declare variable AyarTopu                      integer;

declare variable BoyaTalimatKayitId            integer;
declare variable BukumTalimatId                integer;
declare variable UretimPartiNo                 varchar(30);
declare variable CikisPartiNo                  varchar(30);
declare variable Lot                           varchar(150);
declare variable MelanjKodu                    varchar(20);
declare variable FlamKodu                      varchar(20);
declare variable Sertifika                     varchar(30);
declare variable UreticiFirmaId                varchar(20);
declare variable OrmeMakinaId                  varchar(10);
declare variable DokumaMakinaId                varchar(10);
declare variable OrmeciMakinaNo                varchar(10);
declare variable UreticiId                     integer;
declare variable Tamir                         integer;
declare variable EskiPartiYil                  integer;
declare variable EskiPartiNo                   integer;
declare variable FasonStok                     integer;
declare variable StokFirmaId                   varchar(20);
declare variable KaliteAciklamaTipiId          integer;
declare variable MamulProsesId                 integer;
declare variable BoyahaneFasonIade             integer;
declare variable DesenId                       varchar(20);
declare variable DesenVaryantId                integer;
declare variable StokGirisNo                   integer;
declare variable BirimMaliyet                  double Precision;
declare variable Brut                          integer;
declare variable TamirTipiId                   integer;
declare variable DisTamirCozumSekliId          varchar(10);
declare variable CozguIsEmriKayitID            integer;
declare variable HasilIsEmriKayitID            integer;
declare variable IndigoIsEmriKayitID           integer;
declare variable LeventNo                      integer;
declare variable LeventSiraNo                  integer;
declare variable DepoPartiNo                   varchar(30);
declare variable TelAdedi                      Integer;
declare variable BobinSayisi                   integer;
declare variable OrmeRafTransferFisiOtoOnayli  integer;
declare variable AtkiLotu                      varchar(30);
declare variable CozguLotu                     varchar(30);
declare variable LeventTipiId                  integer;
declare variable BarkodId                      varchar(30);
declare variable OlcuId                        integer;
declare variable Adet                          integer;
declare variable IplikUzamaDegeri              double Precision;
declare variable FisTarihi                     TimeStamp;
declare variable MiktarBirim                   integer;
declare variable TransferEdildi                EgeBoolean;
begin
  Select TransferEdildi From OrmeRafTransferFisi Where OrmeRafTransferFisiId = new.OrmeRafTransferFisiId into :TransferEdildi;
  if (TransferEdildi = 1) then exception TrasferEdilmisDegistirilmez;

  Select OrmeRafTransferFisiOtoOnayli from Sistem Where SistemID = 1 into :OrmeRafTransferFisiOtoOnayli;

  if (OrmeRafTransferFisiOtoOnayli = 1) then
  begin
    new.OnayDurumu = 1;
    new.OnaylayanKullaniciID = new.KullaniciID;
    new.OnayTarihi = current_timestamp(0);
  end


  select (NetGirisMiktari - NetCikisMiktari) from OrmeRaf where OrmeRaf.OrmeRafID = new.OrmeRafId into :KalanMiktar;
  if (KalanMiktar < new.Kilo) then exception RaftaYeterliKiloMetreYok;

  Select FisTarihi From OrmeRafTransferFisi where OrmeRafTransferFisi.OrmeRafTransferFisiId = New.OrmeRafTransferFisiId into :FisTarihi;
  select BirimMaliyet from RafFifoMaliyetGetir(New.OrmeRafId, New.Kilo, :FisTarihi, Null) into new.BirimMaliyet;

  if (new.OnayDurumu = 1) then
  begin

    Select HamId, RenkId, BoyaSiparisKayitId, OrmeSiparisDetayId, BoyaSiparisDetayId, OrmePlanlamaId, DokumaPlanlamaId, Sakat, AyarTopu, BoyaTalimatKayitId, BukumTalimatId, UretimPartiNo, CikisPartiNo,
         Lot, MelanjKodu, FlamKodu, UreticiFirmaId, OrmeMakinaId, DokumaMakinaId, OrmeciMakinaNo, UreticiId, Tamir, EskiPartiYil, EskiPartiNo, FasonStok, StokFirmaId,
         KaliteAciklamaTipiId, MamulProsesId, BoyahaneFasonIade, BoyaSiparisBaskiId, DesenId, DesenVaryantId, StokGirisNo, BirimMaliyet, Brut, TamirTipiId, DisTamirCozumSekliId,
         CozguIsEmriKayitID, HasilIsEmriKayitID, LeventNo, DepoPartiNo, TelAdedi, (GirisBobinSayisi - CikisBobinSayisi), AtkiLotu, CozguLotu, LeventTipiId, BarkodId,
         OlcuId, (GirisAdet-CikisAdet), Sertifika, IndigoIsEmriKayitID, LeventSiraNo, IplikUzamaDegeri, MiktarBirim from OrmeRaf
      where OrmeRaf.OrmeRafID = New.OrmeRafId
      into :HamId, :RenkId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :OrmePlanlamaId, :DokumaPlanlamaId, :Sakat, :AyarTopu, :BoyaTalimatKayitId, :BukumTalimatId, :UretimPartiNo, :CikisPartiNo,
         :Lot, :MelanjKodu, :FlamKodu, :UreticiFirmaId, :OrmeMakinaId, :DokumaMakinaId, :OrmeciMakinaNo, :UreticiId, :Tamir, :EskiPartiYil, :EskiPartiNo, :FasonStok, :StokFirmaId,
         :KaliteAciklamaTipiId, :MamulProsesId, :BoyahaneFasonIade, :BoyaSiparisBaskiId, :DesenId, :DesenVaryantId, :StokGirisNo, :BirimMaliyet, :Brut, :TamirTipiId, :DisTamirCozumSekliId,
         :CozguIsEmriKayitID, :HasilIsEmriKayitID, :LeventNo, :DepoPartiNo, :TelAdedi, :BobinSayisi, :AtkiLotu, :CozguLotu, :LeventTipiId, :BarkodId,
         :OlcuId, :Adet, :Sertifika, :IndigoIsEmriKayitID, :LeventSiraNo, :IplikUzamaDegeri, :MiktarBirim;

    Select OrmeRafId from OrmeRafKaydet(new.RafId, :StokGirisNo, :HamId, :RenkId, :DesenId, :DesenVaryantId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :OrmePlanlamaId, :DokumaPlanlamaId, :Sakat, :AyarTopu, :BoyaTalimatKayitId,
                                  :UretimPartiNo, :CikisPartiNo, :Lot, :MelanjKodu, :FlamKodu, :UreticiFirmaId, :OrmeMakinaId, :DokumaMakinaId, :OrmeciMakinaNo, :UreticiId, :Tamir, :EskiPartiYil,
                                  :EskiPartiNo, new.KaliteNo, :FasonStok, :StokFirmaId, :KaliteAciklamaTipiId, :MamulProsesId, :BoyahaneFasonIade, new.KapAdedi, new.BrutKilo, new.Kilo, :Brut, :MiktarBirim, :BirimMaliyet, :TamirTipiId, :DisTamirCozumSekliId,
                                  :CozguIsEmriKayitID, :HasilIsEmriKayitID, :LeventNo, :BobinSayisi, :BukumTalimatId, :DepoPartiNo, :TelAdedi, :AtkiLotu, :LeventTipiId, :CozguLotu, :BarkodId,
                                  :OlcuId, new.Adet, :Sertifika, :IndigoIsEmriKayitID, :LeventSiraNo, :IplikUzamaDegeri)
    into new.YeniOrmeRafId;
  end
  if (new.OrmeRafTransferFisiDetayID is null) then new.OrmeRafTransferFisiDetayID = Gen_Id(OrmeRafTransferFisiDetay_Id_gen, 1);
end
