CREATE TRIGGER AFTERINSERTSIPARISREZERV FOR BOYASIPARISREZERV
ACTIVE AFTER INSERT POSITION 0
as
Declare variable KayitSayisi integer;
begin
  Update HamStokDetay Set RezervEdilenKilo = RezervEdilenKilo + New.Kilo, RezervEdilenMetre = RezervEdilenMetre + New.Metre, RezervEdilenAdet = RezervEdilenAdet + New.Adet, RezervEdilenKapAdedi = RezervEdilenKapAdedi + New.KapAdedi, NeredenUpdate = 3
       where HamStokDetay.HamStokDetayId = New.HamStokDetayId;

  Select Count(*) from BoyaSiparisDetayOngoru
    where BoyaSiparisDetayOngoru.BoyaSiparisDetayId = New.BoyaSiparisDetayId and BoyaSiparisDetayOngoru.HamStokDetayId = New.HamStokDetayId
    into :KayitSayisi;

  if (KayitSayisi =  0) then
  begin
    insert into BoyaSiparisDetayOngoru(BoyaSiparisKayitId, BoyaSiparisDetayId, HamStokDetayId, InsertKullaniciId, KullaniciId)
                            values(New.BoyaSiparisKayitId, new.BoyaSiparisDetayId, New.HamStokDetayId, New.InsertKullaniciId, New.KullaniciId);
  end
end
