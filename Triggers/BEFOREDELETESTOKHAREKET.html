Definition

CREATE TRIGGER BEFOREDELETESTOKHAREKET FOR STOKHAREKET
ACTIVE BEFORE DELETE POSITION 0
as
declare variable Silinemez integer;
declare variable Entegrasyon integer;
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable uitBukum integer;
declare variable uitOrme integer;
declare variable uitBoya integer;
declare variable uitDokuma integer;
declare variable uitCozgu integer;
declare variable OrmeFisId integer;
declare variable KantarKayitId          integer;
declare variable OnayDurumu             integer;
declare variable SonDonemBitisTarihi TimeStamp;
declare variable OTalReceteCikisMiktardaLotYok SmallInt;
declare variable BTalReceteCikisMiktardaLotYok SmallInt;
declare variable DokumaUretimdeOtomatikCikis EgeBoolean;
declare variable ReceteKs integer;
declare variable LogAnahtar varchar(4000);
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
   into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
        :shtFasonUretim, :ShtUretim;

  Select OTalReceteCikisMiktardaLotYok, BTalReceteCikisMiktardaLotYok, DokumaUretimdeOtomatikCikis from Sistem Where SistemID = 1 into :OTalReceteCikisMiktardaLotYok, :BTalReceteCikisMiktardaLotYok, :DokumaUretimdeOtomatikCikis;

  Silinemez = 0;
  if (Old.OrmeFaturaId is not null and Old.NeredenGirildi <> 2) then Silinemez = 2;

  if (Old.OrmeIrsaliyeId is not null and Old.NeredenGirildi <> 1) then Silinemez = 1;

  if (Silinemez = 2) then exception BelgeFaturalastirilmisSilinemez;
  if (Silinemez = 1) then exception BelgeIrsaliyelestirilmisSilme;

  if (Old.HamStokIrsaliyeId is not null or Old.IplikIrsaliyeId is not null ) then exception KayitHatasi ' Evrak Transfer edilmi≈ü, Silinemez..';

  if (Old.OrmeIrsaliyeID is not null) then
  begin
    Select KantarKayitId from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = Old.OrmeIrsaliyeId into :KantarKayitId;

    if (KantarKayitId is not null) then
    begin
      Select OnayDurumu from Kantar where Kantar.KantarKayitID = :KantarKayitId into :OnayDurumu;
      if (OnayDurumu = 1) then exception KantarOnayiVerilmisDegisemez;
    end
  end

  if (Old.GirisCikis = 1) then
    Delete From OrmeRafDetay where OrmeRafDetay.StokHareketId = Old.StokHareketId;

  if (Old.GirisCikis = -1 or Old.GirisCikis = 0) then
  begin
    /*Update OrmeRaf Set NetCikisMiktari = NetCikisMiktari - Old.Miktar,
                       BrutCikisMiktari = BrutCikisMiktari - Old.BrutMiktar,
                       CikisKapAdedi = CikisKapAdedi - Old.KapAdedi
      where OrmeRaf.OrmeRafId = Old.OrmeRafId;*/
  end

  Select uitBukum, uitOrme, uitBoya, uitDokuma, uitCozgu from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya, :uitDokuma, :uitCozgu;


  if ((Old.HareketTipi = shtFasonUretim or Old.HareketTipi = shtUretim) and (Old.HamIade = 0)) then
  begin
    if (Old.OrmePlanlamaId is not null and Old.UretimIslemTipiId = uitOrme) then
    begin
      if (Old.Sakat = 0) then
        update OrmePlanlama Set GerceklesenMiktar = GerceklesenMiktar - Old.Miktar where OrmePlanlama.OrmePlanlamaId = Old.OrmePlanlamaId;
      else
        update OrmePlanlama Set GerceklesenMiktar = GerceklesenMiktar - Old.Miktar, ToplamSakatOrulen = ToplamSakatOrulen - Old.Miktar where OrmePlanlama.OrmePlanlamaId = Old.OrmePlanlamaId;
    end
    if (Old.DokumaPlanlamaId is not null and Old.UretimIslemTipiId = uitDokuma) then
    begin
      if (Old.Sakat = 0) then
        update DokumaPlanlama Set GerceklesenMiktar = GerceklesenMiktar - Old.Miktar where DokumaPlanlama.DokumaPlanlamaId = Old.DokumaPlanlamaId;
      else
        update DokumaPlanlama Set GerceklesenMiktar = GerceklesenMiktar - Old.Miktar, ToplamSakatDokunan = ToplamSakatDokunan - Old.Miktar where DokumaPlanlama.DokumaPlanlamaId = Old.OrmePlanlamaId;
    end
    if (Old.BukumTalimatId is not null and Old.UretimIslemTipiId = uitBukum) then
      update BukumTalimat Set GerceklesenMiktar = GerceklesenMiktar - Old.Miktar where BukumTalimat.BukumTalimatId = Old.BukumTalimatId;

  end

  if (Old.UretimIslemTipiId = uitOrme and Old.OrmePlanlamaId is not null) then
  begin
    if ((Old.HareketTipi = shtFasonUretimeCikis or Old.HareketTipi = shtUretimeCikis) and (Old.HamIade = 0)) then
    begin
      if (OTalReceteCikisMiktardaLotYok = 1) then
      begin
        update OrmePlanlamaRecete Set CikisMiktari = CikisMiktari - Old.Miktar
          where OrmePlanlamaRecete.OrmePlanlamaId = Old.OrmePlanlamaId and OrmePlanlamaRecete.HamId = Old.HamId and
                IntKarsilastir(OrmePlanlamaRecete.RenkId, Old.RenkID) = 1;
      end
      else
      begin
        update OrmePlanlamaRecete Set CikisMiktari = CikisMiktari - Old.Miktar
          where OrmePlanlamaRecete.OrmePlanlamaId = Old.OrmePlanlamaId and OrmePlanlamaRecete.HamId = Old.HamId and
                IntKarsilastir(OrmePlanlamaRecete.RenkId, Old.RenkID) = 1 and StrKarsilastir(OrmePlanlamaRecete.Lot, old.Lot) = 1;
      end
    end
    else if ((Old.HareketTipi = shtFasonUretim or Old.HareketTipi = shtUretim) and (Old.HamIade = 1)) then
    begin
      if (OTalReceteCikisMiktardaLotYok = 1) then
      begin
        update OrmePlanlamaRecete Set IadeMiktari = IadeMiktari - Old.Miktar
          where OrmePlanlamaRecete.OrmePlanlamaId = Old.OrmePlanlamaId and OrmePlanlamaRecete.HamId = Old.HamId and
                IntKarsilastir(OrmePlanlamaRecete.RenkId, Old.RenkID) = 1;
      end
      else
      begin
        update OrmePlanlamaRecete Set IadeMiktari = IadeMiktari - Old.Miktar
          where OrmePlanlamaRecete.OrmePlanlamaId = Old.OrmePlanlamaId and OrmePlanlamaRecete.HamId = Old.HamId and
                IntKarsilastir(OrmePlanlamaRecete.RenkId, Old.RenkID) = 1 and StrKarsilastir(OrmePlanlamaRecete.Lot, old.Lot) = 1;
      end
    end
  end
  if (Old.UretimIslemTipiId = uitBukum and Old.BukumTalimatId is not null) then
  begin
    if ((Old.HareketTipi = shtFasonUretimeCikis or Old.HareketTipi = shtUretimeCikis) and (Old.HamIade = 0)) then
    begin
      if (BTalReceteCikisMiktardaLotYok = 1) then
      begin
    Select count(*) from BukumTalimatRecete
           where BukumTalimatRecete.BukumTalimatId = old.BukumTalimatID and BukumTalimatRecete.HamId = old.HamId and IntKarsilastir(BukumTalimatRecete.RenkId, Old.RenkID) = 1
           into :ReceteKS;
             if (ReceteKs = 0) then ReceteKs  = 1;
        update BukumTalimatRecete Set CikisMiktari = CikisMiktari - (Old.Miktar / :ReceteKs)  where BukumTalimatRecete.BukumTalimatId = Old.BukumTalimatId and BukumTalimatRecete.HamId = Old.HamId and
                IntKarsilastir(BukumTalimatRecete.RenkId, Old.RenkID) = 1;
      end
      else
      begin
    Select count(*) from BukumTalimatRecete
            where BukumTalimatRecete.BukumTalimatId = old.BukumTalimatID and BukumTalimatRecete.HamId = old.HamId and IntKarsilastir(BukumTalimatRecete.RenkId, Old.RenkID) = 1 and StrKarsilastir(BukumTalimatRecete.Lot, Old.Lot) = 1 into :ReceteKS;
             if (ReceteKs = 0) then ReceteKs  = 1;
        update BukumTalimatRecete Set CikisMiktari = CikisMiktari - (Old.Miktar / :ReceteKs)  where BukumTalimatRecete.BukumTalimatId = Old.BukumTalimatId and BukumTalimatRecete.HamId = Old.HamId and
                IntKarsilastir(BukumTalimatRecete.RenkId, Old.RenkID) = 1 and StrKarsilastir(BukumTalimatRecete.Lot, Old.Lot) = 1;
      end
    end
    else if ((Old.HareketTipi = shtFasonUretim or Old.HareketTipi = shtUretim) and (Old.HamIade = 1)) then
    begin
      if (BTalReceteCikisMiktardaLotYok = 1) then
      begin
    Select count(*) from BukumTalimatRecete
           where BukumTalimatRecete.BukumTalimatId = old.BukumTalimatID and BukumTalimatRecete.HamId = old.HamId and IntKarsilastir(BukumTalimatRecete.RenkId, Old.RenkID) = 1
           into :ReceteKS;
             if (ReceteKs = 0) then ReceteKs  = 1;
        update BukumTalimatRecete Set IadeMiktari = IadeMiktari - (Old.Miktar / :ReceteKs) where BukumTalimatRecete.BukumTalimatId = Old.BukumTalimatId and BukumTalimatRecete.HamId = Old.HamId and
                IntKarsilastir(BukumTalimatRecete.RenkId, Old.RenkID) = 1;
      end
      else
      begin
    Select count(*) from BukumTalimatRecete
            where BukumTalimatRecete.BukumTalimatId = old.BukumTalimatID and BukumTalimatRecete.HamId = old.HamId and IntKarsilastir(BukumTalimatRecete.RenkId, Old.RenkID) = 1 and StrKarsilastir(BukumTalimatRecete.Lot, Old.Lot) = 1 into :ReceteKS;
             if (ReceteKs = 0) then ReceteKs  = 1;
        update BukumTalimatRecete Set IadeMiktari = IadeMiktari - (Old.Miktar / :ReceteKs)  where BukumTalimatRecete.BukumTalimatId = Old.BukumTalimatId and BukumTalimatRecete.HamId = Old.HamId and
                IntKarsilastir(BukumTalimatRecete.RenkId, Old.RenkID) = 1 and StrKarsilastir(BukumTalimatRecete.Lot, Old.Lot) = 1;
      end
    end
  end
  if (Old.UretimIslemTipiId = uitDokuma and Old.DokumaPlanlamaId is not null) then
  begin
    if ((Old.HareketTipi = shtFasonUretimeCikis or Old.HareketTipi = shtUretimeCikis) and (Old.HamIade = 0)) then
      update DokumaPlanlamaRecete Set CikisMiktari = CikisMiktari - Old.Miktar
        where DokumaPlanlamaRecete.DokumaPlanlamaId = Old.DokumaPlanlamaId and DokumaPlanlamaRecete.HamId = Old.HamId and
              IntKarsilastir(DokumaPlanlamaRecete.RenkId, Old.RenkID) = 1 and StrKarsilastir(DokumaPlanlamaRecete.Lot, old.Lot) = 1;
    else if ((Old.HareketTipi = shtFasonUretim or Old.HareketTipi = shtUretim) and (Old.HamIade = 1)) then
      update DokumaPlanlamaRecete Set IadeMiktari = IadeMiktari - Old.Miktar
        where DokumaPlanlamaRecete.DokumaPlanlamaId = Old.DokumaPlanlamaId and DokumaPlanlamaRecete.HamId = Old.HamId and
              IntKarsilastir(DokumaPlanlamaRecete.RenkId, Old.RenkID) = 1 and StrKarsilastir(DokumaPlanlamaRecete.Lot, old.Lot) = 1;
  end

  if (Old.BoyaSiparisDetayID is not null) then
  begin
    if (Old.UretimIslemTipiId = uitBoya and (Old.HareketTipi = shtFasonUretimeCikis or Old.HareketTipi = shtUretimeCikis)) then
       Update BoyaSiparisDetay set ToplamBoyayaCikis = ToplamBoyayaCikis - Old.Miktar, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayID = Old.BoyaSiparisDetayId;
    else if (Old.UretimIslemTipiId = uitBoya and (Old.HareketTipi = shtFasonUretim or Old.HareketTipi = shtUretim) and Old.HamIade = 1) then
       Update BoyaSiparisDetay set ToplamBoyaIade = ToplamBoyaIade - Old.Miktar, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayID = Old.BoyaSiparisDetayId;
    else if (Old.UretimIslemTipiId = uitBoya and (Old.HareketTipi = shtFasonUretim or Old.HareketTipi = shtUretim) and Old.HamIade = 0) then
       Update BoyaSiparisDetay set ToplamBoyadanGelenBrutKilo = ToplamBoyadanGelenBrutKilo - Old.BrutMiktar, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayID = Old.BoyaSiparisDetayId;
  end

  if (Old.BoyaTalimatKayitID is not null) then
  begin
    if (Old.UretimIslemTipiId = uitBoya and (Old.HareketTipi = shtFasonUretimeCikis or Old.HareketTipi = shtUretimeCikis)) then
    begin
       Update BoyaTalimat set GidenMiktar = GidenMiktar - Old.Miktar where BoyaTalimat.BoyaTalimatKayitID = Old.BoyaTalimatKayitID;
       Update BoyaTalimatRecete set KullanilanMiktar = KullanilanMiktar - Old.Miktar where BoyaTalimatRecete.BoyaTalimatKayitId = Old.BoyaTalimatKayitId and BoyaTalimatRecete.OrmeRafID = Old.OrmeRafID;
    end
    else if (Old.UretimIslemTipiId = uitBoya and (Old.HareketTipi = shtFasonUretim or Old.HareketTipi = shtUretim) and Old.HamIade = 1) then
       Update BoyaTalimat set IadeAlinanMiktar = IadeAlinanMiktar - Old.Miktar where BoyaTalimat.BoyaTalimatKayitID = Old.BoyaTalimatKayitID;
    else if (Old.UretimIslemTipiId = uitBoya and (Old.HareketTipi = shtFasonUretim or Old.HareketTipi = shtUretim) and Old.HamIade = 0) then
       Update BoyaTalimat set BrutGelenMiktar = BrutGelenMiktar - Old.BrutMiktar, NetGelenMiktar = NetGelenMiktar - Old.Miktar where BoyaTalimat.BoyaTalimatKayitID = Old.BoyaTalimatKayitID;
  end

  if (Old.UretimIslemTipiId = uitBoya and Old.GirisCikis = 1 and Old.CikisPartiNo is not null) then
     Update BoyahaneCikisPartiNo set BrutGirisMiktar = BrutGirisMiktar - Old.BrutMiktar,
                                     NetGirisMiktar = NetGirisMiktar - Old.Miktar
       where BoyahaneCikisPartiNo.CikisPartiNo = Old.CikisPartiNo;

  if (Old.UretimIslemTipiId = uitBoya and Old.GirisCikis = -1) then
    delete from BoyahaneCikisPartiNo where BoyahaneCikisPartiNo.StokHareketID = Old.StokHareketID;

  if (Old.CozguIsEmriKayitID is Not Null or Old.HasilIsEmriKayitID is Not Null or Old.IndigoIsEmriKayitID is Not Null) then
    delete from DokumaUretimHata where DokumaUretimHata.StokHareketID = Old.StokHareketID;

  Update OrmeKaliteKontrol Set OrmeIrsaliyeId = Null, SatisStokHareketId = null, LinkIslemi = 1 where OrmeKaliteKontrol.SatisStokHareketId = Old.StokHareketId;
  Update DokumaKaliteKontrol Set OrmeIrsaliyeId = Null, SatisStokHareketId = null, LinkIslemi = 1 where DokumaKaliteKontrol.SatisStokHareketId = Old.StokHareketId;
  Update DokumaKaliteKontrol Set SarfOrmeFisId = Null, SarfStokHareketId = null, LinkIslemi = 1 where DokumaKaliteKontrol.SarfStokHareketId = Old.StokHareketId;

  Delete from StokHareketDetay where StokHareketDetay.StokHareketID = Old.StokHareketId;

  select max(BitisTarihi) from DonemKapatma where Bolum = 2 and OnayKullaniciId is not Null into :SonDonemBitisTarihi;

  if (SonDonemBitisTarihi is not Null and SonDonemBitisTarihi > old.Tarih) then
    exception DonemKapaliHSilinemez;

  Delete from DokumaStokPlanlama where StokHareketId = old.StokHareketId;

  if (DokumaUretimdeOtomatikCikis = 1) then
  begin
    for select OrmeFisId from OrmeFis where OrmeFis.stokhareketid = old.stokhareketid  into :OrmeFisId do
    begin
      if (OrmeFisId is not null) then
        Execute Procedure OrmeFisSil(:OrmeFisId);
    end
    if ((old.HareketTipi = shtUretim or old.HareketTipi = shtFasonUretim) and old.DokumaHareketi = 1 and old.UretimIslemTipiId = uitCozgu) then /*cozgu uretimininde cikis*/
      delete from DokumaUretimeOtomatikCikis where DokumaUretimeOtomatikCikis.StokHareketId = old.StokHareketId;
    else if (old.HareketTipi = shtFasonUretim and old.DokumaHareketi = 1 and old.UretimIslemTipiId = uitDokuma) then /*dokuma fason uretimde cikis*/
      delete from DokumaUretimeOtomatikCikis where DokumaUretimeOtomatikCikis.StokHareketId = old.StokHareketId;
  end

  LogAnahtar = 'Tarih:'||Extract(DAY from Old.Tarih)||'.'||Extract(MONTH from Old.Tarih)||'.'||Extract(YEAR from Old.Tarih)||' OrmeFisId:'||Coalesce(Old.OrmeFisId,'')||' OrmeIrsaliyeId:'||Coalesce(Old.OrmeIrsaliyeId,'')||' OrmeFaturaId:'||Coalesce(Old.OrmeFaturaId,'');
  Execute Procedure Logla('D', 2, 'STOKHAREKET', 'STOKHAREKETID', old.StokHareketId, :LogAnahtar);
end
