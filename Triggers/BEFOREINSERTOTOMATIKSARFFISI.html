CREATE TRIGGER BEFOREINSERTOTOMATIKSARFFISI FOR OTOMATIKSARFFISI
ACTIVE BEFORE INSERT POSITION 0
as
declare Variable FHSSarfFisiYil integer;
declare Variable FHSSarfFisiNo integer;
declare Variable FHSSarfFisiId integer;
declare Variable FHSSarfFisiDetayId integer;
declare Variable FHamStokMasterId integer;
declare Variable FHamStokIrsaliyeId integer;
declare Variable FKaynakHamId integer;
declare Variable FirmaId varchar(20);
declare Variable Sinif varchar(1);
declare Variable Fason integer;
declare Variable OtomatikSarfFisSerisi varchar(2);
declare Variable IrsaliyeNo IrsaliyeNoSize;
declare Variable DovizId varchar(5);
declare Variable EskiHamStokMasterId integer;
declare Variable AracId varchar(11);
declare Variable SoforId integer;
declare Variable TransferYeri integer;
declare Variable IrsaliyeGrupId varchar(4);
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable FHamTipiId integer;
declare Variable RafId varchar(10);
begin
  select OtomatikSarfFisSerisi, DokumaKumasiKilodanTakipEt from Sistem into :OtomatikSarfFisSerisi, :DokumaKumasiKilodanTakipEt;

  if (OtomatikSarfFisSerisi is Null) then
    exception KayitHatasi ' Otomatik sarf fisleri icin seri girmemissiniz. Lutfen sistem ayarlarindan bilgiyi doldurunuz...';

  if (New.PartiSinifId is not Null and New.RafDetayId is not Null) then exception KayitHatasi ' Hem parti, hem ham stok giremezsiniz...';
  if (New.PartiSinifId is Null and New.RafDetayId is Null) then exception KayitHatasi ' Hem parti, hem ham stok bo≈ü giremezsiniz...';

  new.OtomatikSarfFisiId = gen_id(OtomatikSarfFisi_Id_Gen, 1);

  if (new.RafDetayId is not Null) then
  begin
    select RafDetay.HamId, Ham.HamTipiId from RafDetay
      left join Ham on Ham.HamId = RafDetay.HamID
      where RafDetay.RafDetayId = new.RafDetayId
      into :FKaynakHamId, :FHamTipiId;
  end
  else if (new.PartiSinifId is not Null) then
  begin
    select PartiStok.HamId, Ham.HamTipiId from PartiStok
      left join Ham on Ham.HamId = PartiStok.HamID
      where PartiStok.PartiSinifId = new.PartiSinifId
      into :FKaynakHamId, :FHamTipiId;
  end

  new.BrutMiktar = new.Miktar;
  if (FHamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
  begin
    New.Kilo = New.Miktar;
    New.Brutkilo = New.BrutMiktar;
  end
  else
  begin
    New.Metre = New.Miktar;
    New.BrutMetre = New.BrutMiktar;
  end

  select id from OtomatikSarfFisiNumaratoruAl(new.FisYil) into new.FisNo;
  update OtomatikSarfFisiNumarator set Id = new.FisNo where Yil = new.FisYil and Id < New.FisNo;
end
