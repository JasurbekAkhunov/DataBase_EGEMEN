CREATE TRIGGER AFTERINSUPDDELHARMANRECETEDETAY FOR HARMANRECETEDETAY
ACTIVE AFTER INSERT OR UPDATE OR DELETE POSITION 0
as
declare variable HarmanMiktari Numeric(12,3);
declare variable FHarmanReceteID integer;
declare variable KS integer;
begin
  if (INSERTING or UPDATING) then FHarmanReceteID = New.HarmanReceteID;
  else FHarmanReceteID = Old.HarmanReceteID;

  Select Sum(Miktar) from HarmanReceteDetay where HarmanReceteDetay.HarmanReceteID = :FHarmanReceteID into :HarmanMiktari;
  if (HarmanMiktari is null) then HarmanMiktari = 0;

  Update HarmanRecete Set Miktar = :HarmanMiktari where HarmanRecete.HarmanReceteID = :FHarmanReceteID;
end
