CREATE TRIGGER BEFOREUPDATESDKALITEKONTROL FOR STOKDETAYKALITEKONTROL
ACTIVE BEFORE UPDATE POSITION 0
as
begin
  if (Old.OnayDurumu = New.OnayDurumu and New.OnayDurumu <> 0) then
  begin
    if ( (Old.Gorunum is not null and New.Gorunum is null) or
         (Old.Gorunum is null and New.Gorunum is not null) or
         (Old.Gorunum <> New.Gorunum) or
         (Old.PH is not null and New.PH is null) or
         (Old.PH is null and New.PH is not null) or
         (Old.PH <> New.PH) or
         (Old.Yogunluk is not null and New.Yogunluk is null) or
         (Old.Yogunluk is null and New.Yogunluk is not null) or
         (Old.Yogunluk <> New.Yogunluk) or
         (Old.KatiMaddeOrani is not null and New.KatiMaddeOrani is null) or
         (Old.KatiMaddeOrani is null and New.KatiMaddeOrani is not null) or
         (Old.KatiMaddeOrani <> New.KatiMaddeOrani) or
         (Old.De is not null and New.De is null) or
         (Old.De is null and New.De is not null) or
         (Old.De <> New.De)) then exception BoyaKimyasalOnayVarDegisemez;
  end

  if (new.OnayDurumu = 1 and new.KontrolDurumu = 0) then exception BoyaKimyasalKontrolYapilmamis;

  if (new.OnayDurumu <> 0 and Old.KontrolDurumu <> new.KontrolDurumu and Old.OnayDurumu <> 0) then exception OnayVerilmisKontrolDegismez;

  if (New.KontrolDurumu <> Old.KontrolDurumu) then
  if (new.KontrolDurumu <> 0) then
  begin
    New.KontrolTarihi = current_TimeStamp(0);
  end
  else
  begin
    New.KontrolKullaniciId = Null;
    New.KontrolTarihi = Null;
  end

  if (New.OnayDurumu <> Old.OnayDurumu) then
  if (new.OnayDurumu <> 0) then
  begin
    New.OnayTarihi = current_TimeStamp(0);
  end
  else
  begin
    New.OnaylayanKullaniciId = Null;
    New.OnayTarihi = Null;
  end
end
