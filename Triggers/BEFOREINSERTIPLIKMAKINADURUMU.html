CREATE TRIGGER BEFOREINSERTIPLIKMAKINADURUMU FOR IPLIKMAKINADURUMU
ACTIVE BEFORE INSERT POSITION 0
as
declare variable FIplikMakinaDurumuId  Integer;
declare variable FAriza                SmallInt;
declare variable FCalisiyor            SmallInt;
declare variable FServisciVar          SmallInt;
declare variable FImalataAraVerildi    SmallInt;
begin
  FIplikMakinaDurumuID = Null;
  Select IplikMakinaDurumuId from IplikMakina where IplikMakina.IplikMakinaID = new.IplikMakinaId into :FIplikMakinaDurumuId;

  if (FIplikMakinaDurumuID is null) then
  begin
    FCalisiyor = 0;
    FAriza = 0;
    FServisciVar = 0;
    FImalataAraVerildi = 0;
  end
  else
  begin
    Select Calisiyor, Ariza, ServisciVar, ImalataAraVerildi from IplikMakinaDurumu where IplikMakinaDurumu.IplikMakinaDurumuID = :FIplikMakinaDurumuId
      into :FCalisiyor, :FAriza, :FServisciVar, :FImalataAraVerildi;
  end

  if (FCalisiyor is null) then FCalisiyor = 0;
  if (FAriza is null) then FAriza = 0;
  if (FServisciVar is null) then FServisciVar = 0;
  if (FImalataAraVerildi is null) then FImalataAraVerildi = 0;

  if (new.Ariza is null) then
  begin
    if (FAriza = 0 or FAriza = 2) then New.Ariza = 0;
    else if (FAriza = 1 or FAriza = 3) then New.Ariza = 3;
  end

  if (new.Calisiyor is null) then New.Calisiyor = FCalisiyor;

  if (new.ServisciGeldi is null) then new.ServisciGeldi = 0;
  if (new.ServisciVar is null) then new.ServisciVar = FServisciVar;

  if (new.ImalataAraVerildi is null) then
  begin
    if (FImalataAraVerildi = 0 or FImalataAraVerildi = 2) then New.ImalataAraVerildi = 0;
    else if (FImalataAraVerildi = 1 or FImalataAraVerildi = 3) then New.ImalataAraVerildi = 3;
  end

  if (new.IplikMakinaDurumuId is null) then new.IplikMakinaDurumuId = gen_id(IplikMakinaDurumu_id_gen, 1);

  Update IplikMakina Set IplikMakinaDurumuID = new.IplikMakinaDurumuId where IplikMakina.IplikMakinaId = new.IplikMakinaId;
end
