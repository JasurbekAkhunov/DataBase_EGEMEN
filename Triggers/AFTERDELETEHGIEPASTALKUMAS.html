Definition

CREATE TRIGGER AFTERDELETEHGIEPASTALKUMAS FOR HGISEMRIPASTALKUMAS
ACTIVE AFTER DELETE POSITION 0
as
begin
  Delete from HGISEmriPastalUrun where HGISEmriPastalUrun.HGISEmriPastalID = Old.HGISEmriPastalID and
    UrunID in (Select Urun.UrunID from HGSiparisDetayKumas
                 LEft Join Urun on Urun.ModelKumasID = HGSiparisDetayKumas.ModelKumasID
                where HGSiparisDetayKumas.HGSiparisDetayKumasID = Old.HGSiparisDetayKumasID);
end
