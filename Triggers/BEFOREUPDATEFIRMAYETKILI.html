CREATE TRIGGER BEFOREUPDATEFIRMAYETKILI FOR FIRMAYETKILI
ACTIVE BEFORE UPDATE POSITION 0
as
begin
  if (Old.DefaultYetkili <> new.DefaultYetkili and new.DefaultYetkili = 1) then
    Update FirmaYetkili set DefaultYetkili = 0 where FirmaYetkili.FirmaID = New.FirmaId and FirmaYetkili.FirmaYetkiliID <> New.FirmaYetkiliId;
end
