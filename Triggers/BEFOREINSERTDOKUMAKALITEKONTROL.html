Definition

CREATE TRIGGER BEFOREINSERTDOKUMAKALITEKONTROL FOR DOKUMAKALITEKONTROL
ACTIVE BEFORE INSERT POSITION 0
as
declare variable FOrmeFisId integer;
declare variable FStokHareketId integer;
declare variable NewStokHareketId integer;
declare variable shtUretim integer;
declare variable uitDokuma      integer;
declare variable FSeriNo        varchar(2);
declare variable FFisNo         integer;

declare variable FBoyaSiparisKayitId  integer;
declare variable FOrmeSiparisDetayId  integer;
declare variable FOrmeSiparisSekliId  integer;
declare variable FHamId           integer;
declare variable FRenkId          integer;
declare variable FUretimPartiNo   varchar(30);
declare variable FDokumaMakinaId  varchar(10);
declare variable FOrmeRafDetayId  integer;

declare variable Aktif          integer;
declare variable FUniteId       integer;
declare variable MUniteId       integer;
declare variable SistemDovizId  varchar(5);
declare variable Lot            varchar(150);
declare variable KaliteNo       integer;
declare variable DokumaKKTopNoTekilOlsun integer;
declare variable KayitSayisi integer;
declare variable KaliteKontrolMetreHesapSekli integer;
declare variable DokumaKKontrolMetreHesapSekli integer;
declare variable HacimEn integer;
declare variable Gramaj  integer;
declare variable IntEn integer;
declare variable IntGramaj  integer;
declare variable FHesap integer;
declare variable TF integer;
declare variable DokumaStokGirisNoKullanimda integer;
declare variable DokumaProjesiAktif integer;
begin
  select KaliteKontrolMetreHesapSekli, DokumaKKontrolMetreHesapSekli, DokumaStokGirisNoKullanimda, DokumaProjesiAktif from Sistem where Sistem.SistemID = 1 into :KaliteKontrolMetreHesapSekli, :DokumaKKontrolMetreHesapSekli, :DokumaStokGirisNoKullanimda, :DokumaProjesiAktif;

  Execute Procedure DokumaMakinaVardiyaOlustur(current_date, current_date - 1, 1);

  if ((DokumaKKontrolMetreHesapSekli = 6) or (KaliteKontrolMetreHesapSekli = 6) ) then /* Pakİpek */
  begin
    Select Ham.HamEnCm, Ham.HamGramaj from DokumaPlanlama
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
      left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
      where DokumaPlanlama.DokumaPlanlamaId = new.DokumaPlanlamaId
    into :HacimEn, :Gramaj;

    TF = EgemenStrToIntOlurmu(Gramaj);

    if (TF = 0) then IntGramaj = 0;
    else IntGramaj = EgemenStrToInt(Gramaj);

    TF = EgemenStrToIntOlurmu(HacimEn);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(HacimEn);

    FHesap = IntEn * IntGramaj;

    if (FHesap <> 0) then new.Kilo = (Cast(FHesap as BoyaMiktarSize) / Cast(100000 as BoyaMiktarSize)) * New.Miktar;
    else new.Kilo = 0;
  end

  if (New.Tarih is Null) then New.Tarih = Current_Date;

  Select SistemDovizId, DokumaKKTopNoTekilOlsun from Sistem where sistemId = 1 into :SistemDovizId, :DokumaKKTopNoTekilOlsun;

  if (New.RafId = '') then new.RafID = null;
  if (new.Miktar is null) then new.Miktar = 0;
  if (DokumaProjesiAktif = 2) then new.Miktar = new.Kilo;

  new.KapAdedi = 1;

  /* LinaTeks için, Kalite Kontrol Hata girişi metre sıfırken tıklanıdğından dolayı
  if (new.Miktar = 0) then exception KayitHatasi 'Metre Sıfır Olamaz..';
  */

  if (New.RafID is null) then
  begin
    Select DokumaDefaultRafId from Sistem where sistemid = 1 into New.RafID;
  end

  Select DepoRaf.DepoId from DepoRaf where DepoRaf.RafID = New.RafID into new.DepoId;

  if (new.KaliteNo <> 1) then KaliteNo = new.KaliteNo;
  else if (new.Sakat = 0) then KaliteNo = 1;
  else KaliteNo = 2;

  if (new.KontrolBitti = 1 and new.OrmeVardiyaId is not null) then
  begin
    select ShtUretim from OrmeStokHareketTipiGetir into :ShtUretim;
    select uitDokuma from UretimIslemTipiGetir into :uitDokuma;

    select OrmeVardiya.Tarih, OrmeVardiya.VardiyaNo, OrmeVardiya.VardiyaAmiriPersonelId, OrmeVardiyaPersonel.PersonelId, OrmeVardiya.Aktif, OrmeVardiya.UniteId from OrmeVardiya
     left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = new.DokumaPlanlamaId
     left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = OrmeVardiya.OrmeVardiyaId and OrmeVardiyaPersonel.DokumaMakinaId = DokumaPlanlama.DokumaMakinaId
     where OrmeVardiya.OrmeVardiyaId = new.OrmeVardiyaId
     into new.VardiyaTarihi, new.Vardiya, new.VardiyaAmiriPersonelId, new.PersonelId, :Aktif, :FUniteId;

    Select BoyaSiparis.OrmeSiparisSekliId from DokumaPlanlama
       left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
       where DokumaPlanlama.DokumaPlanlamaId = new.DokumaPlanlamaId
       into :FOrmeSiparisSekliID;

    if (Aktif = 0) then exception KayitHatasi ' Vardiya Aktif Degil';

    Select DokumaUretimSeriNo from Kullanici where KullaniciId = new.InsertKullaniciId into :FSeriNo;

    Select FisNo from FisNumaratorAl(:FSeriNo) into :FFisNO;

    if (new.PersonelId is null) then exception KayitHatasi ' Vardiya Personel Bilgisi Yok';

    Select Max(OrmeFisId) from OrmeFis
      where OrmeFis.FisTarihi = New.VardiyaTarihi and OrmeFis.HareketTipi = :shtUretim and OrmeFis.UretimIslemTipiID = :uitDokuma
            and OrmeFis.Vardiya = new.Vardiya and OrmeFis.VardiyaAmiriPersonelId = new.VardiyaAmiriPersonelId
            and OrmeFis.OrmeSiparisSekliID = :FOrmeSiparisSekliID and OrmeFis.DokumaHareketi = 1
            into :FOrmeFisId;

    if (FOrmeFisId is null) then
    begin
      FOrmeFisId = gen_id(OrmeFis_Id_Gen, 1);

      insert into OrmeFis(OrmeFisId, HareketTipi, UretimIslemTipiId, SeriNo, FisNo, FisTarihi, Vardiya, VardiyaAmiriPersonelId, DovizId, OrmeSiparisSekliID, DokumaHareketi, InsertKullaniciId, KullaniciId)
               values(:FOrmeFisId, :shtUretim, :uitDokuma, :FSeriNo, :FFisNo, new.VardiyaTarihi, new.Vardiya, new.VardiyaAmiriPersonelId, :SistemDovizId, :FOrmeSiparisSekliID, 1, new.InsertKullaniciId, New.InsertKullaniciId);
    end
    else if (DokumaStokGirisNoKullanimda = 0) then
      Select max(StokHareketId) from StokHareket
        where StokHareket.OrmeFisId = :FOrmeFisId and
              StokHareket.DokumaPlanlamaId = new.DokumaPlanlamaId and
              StokHareket.Sakat = new.Sakat and
              StokHareket.AyarTopu = new.AyarTopu and
              StokHareket.DepoId = new.DepoId and
              IntKarsilastir(StokHareket.LeventNo, new.LeventNo) = 1 and
              StrKarsilastir(StokHareket.DepoPartiNo, new.DepoPartiNo) = 1 and
              StrKarsilastir(StokHareket.Lot, new.Lot) = 1 and
              StrKarsilastir(StokHareket.AtkiLotu, new.AtkiLotu) = 1 and
              StrKarsilastir(StokHareket.CozguLotu, new.CozguLotu) = 1 and
              StokHareket.LeventBitti = new.LeventBitti and
              StokHareket.KaliteKontrolvar = 1 and
              StokHareket.DokumaHareketi = 1 and
              StokHareket.MiktarBirim = 1
              into :FStokHareketId;
    if (FStokHareketId is not null) then New.StokHareketId = FStokHareketId;
    else New.StokHareketId = gen_id(StokHareket_Id_Gen, 1);

    Select DokumaPlanlama.BoyaSiparisKayitId, DokumaPlanlama.OrmeSiparisDetayId, OrmeSiparisDetay.HamId, OrmeSiparisDetay.RenkId, DokumaPlanlama.UretimPartiNo, DokumaPlanlama.DokumaMakinaId,
           DokumaPlanlama.Lot from DokumaPlanlama
       left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
       where DokumaPlanlama.DokumaPlanlamaId = new.DokumaPlanlamaId
       into :FBoyaSiparisKayitId, :FOrmeSiparisDetayId, :FhamId, :FRenkId, :FUretimPartiNo, :FDokumaMakinaId, :Lot;

    if (FStokHareketId is not Null) then Update StokHareket Set Metre = Metre + New.Miktar, Kilo = Kilo + New.Kilo, KapAdedi = KapAdedi + New.KapAdedi, NeredenUpdate = 3 where StokHareket.StokHareketId = :FStokHareketId;
    else
    begin
      insert into StokHareket(OrmeFisId, StokHareketId, HareketTipi, UretimIslemTipiId, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, Metre, Kilo, KapAdedi, UretimPartiNo, DokumaMakinaId, DokumaPlanlamaId, Sakat, AyarTopu, KaliteNo, Lot, AtkiLotu, CozguLotu, DepoId, KaliteKontrolVar, DokumaHareketi, MiktarBirim, LeventNo, DepoPartiNo, LeventBitti)
                       values(:FOrmeFisId, new.StokHareketId, :shtUretim, :uitDokuma, :FBoyaSiparisKayitId, :FOrmeSiparisDetayId, :FHamId, :FRenkId, New.Miktar, New.Kilo, New.KapAdedi, :FUretimPartiNo, :FDokumaMakinaId, new.DokumaPlanlamaId, New.Sakat, New.AyarTopu, :KaliteNo, New.Lot, New.AtkiLotu, New.CozguLotu, New.DepoId, 1, 1, 1, New.LeventNo, New.DepoPartiNo, New.LeventBitti);
    end

    if (new.RafId is not null) then
    begin
      Select max(OrmeRafDetayId) from OrmeRafDetay where OrmeRafDetay.StokHareketId = new.StokHareketId and OrmeRafDetay.rafId = New.RafId into :FOrmeRafDetayId;

      if (FOrmeRafDetayId is not null) then
      begin
        New.OrmeRafDetayId = FOrmeRafDetayId;
        Update OrmeRafDetay set Miktar = Miktar + new.Miktar, BrutMiktar = BrutMiktar + new.Miktar, KapAdedi = KapAdedi + New.KapAdedi where OrmeRafDetay.OrmeRafDetayID = :FOrmeRafDetayId;
      end
      else
      begin
        new.OrmeRafDetayId = gen_id(OrmeRafDetay_Id_Gen, 1);

        insert into OrmeRafDetay(OrmeRafDetayId, StokHareketId, RafId, Miktar, BrutMiktar, KapAdedi, InsertKullaniciId, KullaniciId)
                                Values(new.OrmeRafDetayId, new.StokHareketId, New.RafId, New.Miktar, New.Miktar, New.KapAdedi, New.InsertKullaniciId, New.KullaniciId);
      end
    end
  end

  if (new.TopNo is Null) then
  begin
    if (DokumaKKTopNoTekilOlsun = 1) then
      Select Max(topNo) from DokumaKaliteKontrol into new.TopNo;
    else
      Select Max(topNo) from DokumaKaliteKontrol where DokumaKaliteKontrol.DokumaPlanlamaId = New.DokumaPlanlamaId into new.TopNo;

    if (new.TopNo is null) then new.TopNo = 1;
    else new.TopNo = new.TopNo + 1;
  end
  else
  begin
    if (DokumaKKTopNoTekilOlsun = 1) then
      Select Count(*) from DokumaKaliteKontrol where DokumaKaliteKontrol.TopNo = New.TopNo into :KayitSayisi;
    else
      Select Max(topNo) from DokumaKaliteKontrol where DokumaKaliteKontrol.DokumaPlanlamaId = New.DokumaPlanlamaId and DokumaKaliteKontrol.TopNo = New.TopNo into :KayitSayisi;

    if (KayitSayisi > 0) then Exception KayitHatasi 'Bu top no daha önce girilmiş. Top No tekrar edemez!';
  end

  if (new.DokumaKaliteKontrolId is null or new.DokumaKaliteKontrolId < 1) then new.DokumaKaliteKontrolId = Gen_Id(DokumaKaliteKontrol_Id_Gen,1);

  if ((Coalesce(new.HesaplananMetre, 0) = 0) and (:DokumaProjesiAktif<>2)) then
    new.HesaplananMetre = new.Miktar;
end
