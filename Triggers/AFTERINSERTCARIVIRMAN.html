CREATE TRIGGER AFTERINSERTCARIVIRMAN FOR CARIVIRMAN
ACTIVE AFTER INSERT POSITION 0
as
declare variable dovizid varchar(5);
begin
  Select DovizId from Firma where FirmaID = New.BorcluFirmaId into :DovizId;

  insert into cari(FirmaId, Tarih, Aciklama,  BorcAlacak, CariTipiId, Meblag, DovizId, ValorTarihi, CariVirmanId, DovizKuru1, DovizKuru2, Sinif, KullaniciId)
          values(new.BorcluFirmaId, new.tarih, new.Aciklama, 'B', New.CariTipiId, new.Meblag, :DovizId, new.ValorTarihi, new.CariVirmanID, 1, 1, new.Sinif, new.KullaniciId);

  insert into cari(FirmaId, Tarih, Aciklama, BorcAlacak, CariTipiId, Meblag, DovizId, ValorTarihi, CariVirmanId, DovizKuru1, DovizKuru2, Sinif, KullaniciId)
          values(new.AlacakliFirmaId, new.tarih, new.Aciklama, 'A', New.CariTipiId, new.Meblag, :DovizId, new.ValorTarihi, new.CariVirmanID, New.DovizKuru1, New.DovizKuru2, new.Sinif, new.KullaniciId);
end
