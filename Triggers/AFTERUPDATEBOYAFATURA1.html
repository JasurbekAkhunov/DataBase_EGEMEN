CREATE TRIGGER AFTERUPDATEBOYAFATURA1 FOR BOYAFATURA
ACTIVE AFTER UPDATE POSITION 1
as
declare variable BoyaFaturaLogID integer;
begin
  if (Old.FaturaSeriNo <> New.FaturaSeriNo or Old.FaturaNo <> new.FaturaNo or
      Old.FaturaTarihi <> New.FaturaTarihi or Old.Entegrasyon <> New.Entegrasyon or
      Old.Sinif <> New.Sinif or Old.TransferEdildi <> new.TransferEdildi or
      Old.TransferTarihi <> New.TransferTarihi or Old.Status <> New.Status) then
  begin
    BoyaFaturaLogID = Gen_Id(BoyaFaturaLog_Id_gen, 1);

    insert into BoyaFaturaLog(BoyaFaturaLogId, BoyaFaturaId, OldFaturaSeriNo, OldFaturaNo, OldFaturaTarihi, newFaturaSeriNo, newFaturaNo, newFaturaTarihi, OldEntegrasyon, NewEntegrasyon,
                              OldSinif, NewSinif, OldTransferEdildi, newTransferEdildi, OldTransferTarihi, NewTransferTarihi, OldTransferKullaniciId, NewTransferKullaniciId, OldStatus, NewStatus,
                              InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
      values(:BoyaFaturaLogId, New.BoyaFaturaId, Old.FaturaSeriNo, Old.FaturaNo, Old.FaturaTarihi, new.FaturaSeriNo, new.FaturaNo, new.FaturaTarihi, Old.Entegrasyon, New.Entegrasyon,
                              Old.Sinif, New.Sinif, Old.TransferEdildi, new.TransferEdildi, Old.TransferTarihi, New.TransferTarihi, Old.TransferKullaniciId, New.TransferKullaniciId,
                              Old.Status, New.Status, New.InsertKullaniciId, New.InsertTarihi, New.KullaniciId, New.DegisimTarihi);
  end
  Execute Procedure Logla('U', 2, 'BOYAFATURA', 'BOYAFATURAID', new.BoyaFaturaId, 'Fatura No :'||new.FaturaNo||' Tarih:'||Extract(DAY from new.FaturaTarihi)||'.'||Extract(MONTH from new.FaturaTarihi)||'.'||Extract(YEAR from new.FaturaTarihi)||' Firma Kodu: '||new.FirmaId);
end
