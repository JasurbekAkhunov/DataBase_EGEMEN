CREATE TRIGGER BEFOREUPDATELABRECETE FOR LABRECETE
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable LabReceteKoduTipi integer;
declare variable EskiLabReceteKoduTekrarlayamaz integer;
declare variable EskiLabReceteKayitSayisi integer;
begin
  Select LabReceteKoduTipi, EskiLabReceteKoduTekrarlayamaz from Sistem into :LabReceteKoduTipi, :EskiLabReceteKoduTekrarlayamaz;

  if (LabReceteKoduTipi <> 9 and LabReceteKoduTipi <> 16 and LabReceteKoduTipi <> 19 and LabReceteKoduTipi <> 30 and LabReceteKoduTipi <> 38 and LabReceteKoduTipi <> 71 and
      LabReceteKoduTipi <> 85 and LabReceteKoduTipi <> 142) then /*Bayteks, Tekboy, Osborn, KumasBoya, Akhan, Aycan, Aka Teks, AbmCotton*/


  if (LabReceteKoduTipi not in (2, 100) and IntKarsilastir(Old.RenkDerinligiId, New.RenkDerinligiId) = 0) then Exception KayitHatasi 'Renk Derinliği Değiştirilemez..';

  if (LabReceteKoduTipi = 25) then
  if (StrKarsilastir(Old.FirmaID, New.FirmaID) = 0) then Exception KayitHatasi 'Firma Kodu Değiştirilemez..';

  if (Old.LabReceteKodu <> New.LabReceteKodu) then exception LabReceteKoduDegistirilemez;

  if (Old.HamId <> New.HamId) then
    Update LabReceteCalisma set HamId = new.HamId where LabReceteCalisma.LabReceteId = New.LabReceteId and LabReceteCalisma.HamID = Old.HamId;

  if (Old.FirmaId <> New.FirmaId) then
  begin
    Update LabReceteCalisma set FirmaId = new.FirmaId where LabReceteCalisma.LabReceteId = New.LabReceteId;
    Update Recete set FirmaId = new.FirmaId where Recete.LabReceteId = New.LabReceteId;
  end

  if (old.Adi <> new.Adi or Old.LabRenkGrupId <> New.LabRenkGrupId or IntKarsilastir(Old.RenkDerinligiId, new.RenkDerinligiId) = 0) then
  begin
    Update Recete set Adi = new.Adi, LabRenkGrupId = New.LabRenkGrupId, RenkDerinligiId = new.RenkDerinligiId where Recete.LabReceteID = New.LabReceteId;
  end

  if (new.EskiLabReceteKodu is not Null) then
  begin
    if (EskiLabReceteKoduTekrarlayamaz = 1) then
    begin
      select count(*) from LabRecete where EskiLabReceteKodu = new.EskiLabReceteKodu and LabreceteKodu <> New.LabreceteKodu into :EskiLabReceteKayitSayisi;
      if (EskiLabReceteKayitSayisi > 0) then
      begin
        exception EskiLabReceteKoduTekrariOlustu;
      end
    end
  end

  new.DegisimTarihi = current_TimeStamp(0);
end
