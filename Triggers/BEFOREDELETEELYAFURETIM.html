CREATE TRIGGER BEFOREDELETEELYAFURETIM FOR ELYAFURETIM
ACTIVE BEFORE DELETE POSITION 0
as
declare variable FKapali integer;
begin
  if (old.ElyafUretimIsEmriKayitId is not null ) then
    Select ElyafUretimIsEmri.Kapali from ElyafUretimIsEmri where ElyafUretimIsEmri.ElyafUretimIsEmriKayitId = old.ElyafUretimIsEmriKayitId
      into :FKapali ;
  else
    Select KirpintiIsEmri.Kapali from KirpintiIsEmri where KirpintiIsEmri.KirpintiIsEmriKayitId = old.KirpintiIsEmriKayitId
      into :FKapali;

  if (FKapali = 1) then Exception KayitHatasi ' Bu Is Emri Kapalı. İşlem Yapılamaz';
end
