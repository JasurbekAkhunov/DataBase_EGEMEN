CREATE TRIGGER BEFOREINSERTHARMANKARISIM FOR HARMANKARISIM
ACTIVE BEFORE INSERT POSITION 0
as
Declare Variable NumaratorID integer;
begin
  new.HarmanKarisimId = Gen_Id(HarmanKarisim_Id_Gen,1);

  if (new.KisaKod is not Null) then
  begin
    Select ID from AnaLotNumarator where Bolum = 2 and KisaKod = New.KisaKod into :NumaratorID;
    if (NumaratorID is Null) then
      insert into AnaLotNumarator (Bolum, KisaKod) Values (2, New.KisaKod);
  end

end
