CREATE TRIGGER BEFOREUPDATEDESENVARYANT FOR DESENVARYANT
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KS integer;
declare variable KalitedePartiSinifBilgiKullan Boolean;
declare variable DesenGravurProjesiAktif Boolean;
begin
  new.DegisimTarihi = current_TimeStamp(0);

  Select KalitedePartiSinifBilgiKullan, DesenGravurProjesiAktif From Sistem Where SistemId = 1 Into :KalitedePartiSinifBilgiKullan, :DesenGravurProjesiAktif;

  if ((strKarsilastir(old.VaryantAdi ,new.VaryantAdi) = 0) and ((KalitedePartiSinifBilgiKullan = 1) or (DesenGravurProjesiAktif = 1))) then
  begin
    select count(*) from Parti where Parti.DesenVaryantId = new.DesenVaryantId into :KS;
    if (KS > 0) then
      exception DesenVaryantDegistirilemez;
  end
end
