CREATE TRIGGER AFTERINSERTPSTOKFIZIKKONTROL FOR PARTISTOKFIZIKKONTROL
ACTIVE AFTER INSERT POSITION 0
as
begin
  Execute Procedure SonFizikKontrolAsamasiIsareti(New.PartiKayitId, New.PartiSinifId);

  Insert into PartiStokFizikKontrolTest(PartiStokFizikKontrolId, OzelTestID, OzelTestStandardiID, Istenen)
       select New.PartiStokFizikKontrolId, SiparisOzelTestler.OzelTestID, SiparisOzelTestler.OzelTestStandardiID, SiparisOzelTestler.Istenen from PartiStok
          left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
          left join SiparisOzelTestler on SiparisOzelTestler.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
          where PartiStok.PartiSinifID = new.PartiSinifID and SiparisOzelTestler.OzelTestID is not null;
end
