Definition

CREATE TRIGGER BEFOREUPDATEPSKALITEKONTROL FOR PSKALITEKONTROL
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KKBitirenPersonelId integer;
declare variable FPartiSinifId       integer;
declare variable FEn             Double Precision;
declare variable FGramaj         Double Precision;
declare variable IntGramaj       integer;
declare variable IntEn           integer;
declare variable HacimEn         integer;
declare variable Gramaj          varchar(10);
declare variable TF              integer;
declare variable FTup            integer;
declare variable KaliteKontrolMetreHesapSekli integer;
declare variable MetreHesabiIcinCarpan Numeric(12,3);
declare variable FHesap  Double Precision;
declare variable FEnVerilen     varchar(10);
declare variable FGramajVerilen varchar(10);
declare variable FPSKaliteKontrolAsamaId integer;
declare variable HesaplamaCarpimDegeri Integer;
declare variable KaliteKontrolHataOrani integer;
declare variable KKDaraPaketlemedenGelsin EgeBoolean;
declare variable TupRolikAgirligi Double Precision;
declare variable KKBitinceTartiyaAktar EgeBoolean;
declare variable FEgemenFirmaId integer;
declare variable DokumaKumasTopNoTekrarEdemez EgeBoolean;
declare variable FHamTipiId integer;
declare variable KS integer;
declare variable CezaMetre BoyaMiktarSize;
declare variable OzelMusteri EgeBoolean;
declare variable OlcuGramaj Numeric(12,3);
declare variable FEnIstenen     varchar(10);
declare variable FGramajIstenen varchar(10);
begin
  if (New.PSKaliteKontrolAsamaId is null) then
  begin
    Select PSKaliteKontrolAsamaId from PsKaliteKontrolAsama where PsKaliteKontrolAsama.PartiSinifID = new.PartiSinifID and PsKaliteKontrolAsama.SonKayit = 1
      into :FPSKaliteKontrolAsamaId;
    if (FPSKaliteKontrolAsamaId is not null) then
      Update PSKaliteKontrol set PsKaliteKontrolAsamaId = :FPSKaliteKontrolAsamaId, LinkIslemi = 1 where PartiSinifID = new.PartiSinifID;
  end

  Select KaliteKontrolMetreHesapSekli, KKDaraPaketlemedenGelsin, KKBitinceTartiyaAktar, EgemenFirmaId, DokumaKumasTopNoTekrarEdemez from sistem where SistemID = 1 into :KaliteKontrolMetreHesapSekli, :KKDaraPaketlemedenGelsin, :KKBitinceTartiyaAktar, :FEgemenFirmaId, :DokumaKumasTopNoTekrarEdemez;

  if ((New.LinkIslemi <> 1) and (new.SartliKabul <> 1)) then
  begin
    Select KKBitirenPersonelId from PsKaliteKontrolAsama where PsKaliteKontrolAsamaId = new.PsKaliteKontrolAsamaId into :KKBitirenPersonelId;

    if (not (KKBitinceTartiyaAktar = 1 and Old.OnayDurumu = 1 and New.OnayDurumu = 2 and new.PartiStokTartiId is Null)) then
    if (KKBitirenPersonelId is not null) then exception PSKaliteKontrolTamamlanmis;
  end

  if (Old.PartiStokTartiId is not null and new.PartiStokTartiId is not null and Old.PartiStokTartiId <> new.PartiStokTartiId) then exception BuPaketinTartiKaydiAlinmis;

  if (Old.PartiStokTartiId is null and new.PartiStokTartiId is not null) then
  begin
    Select PartiSinifId from PartiStokTarti where PartiStokTarti.PartiStokTartiID = new.PartiStokTartiId into :FPartiSinifId;
    if (FPartiSinifId <> New.PartiSinifId) then exception TartiKaydiBuPartiyeAitDegil;
  end

  Select Firma.OzelMusteri from PartiStok
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
    left join Firma on Firma.FirmaID = Parti.FirmaID
    where PartiSinifId = new.PartiSinifId
  into :OzelMusteri;

  if (New.PSKaliteKontrolAsamaId is not null and Old.PartiStokTartiId is not null and new.PartiStokTartiId is null) then
  begin
    Update PSKaliteKontrolAsama set TartiyaAktarildi = 0 where PSKaliteKontrolAsama.PSKaliteKontrolAsamaID = New.PSKaliteKontrolAsamaId;
  end

  if (DokumaKumasTopNoTekrarEdemez = 1) then
  begin
    select PartiStok.HamTipiId from PartiStok where PartiStok.PartiSinifId = new.PartiSinifId into :FHamTipiId;
    if (FHamTipiId = 2) then
    begin
      Select count(PSKaliteKontrol.PSKaliteKontrolId) From PSKaliteKontrol /*Aynı PartiSinifId ve Aynı Top Numarası Başka satırda varmı*/
        where PSKaliteKontrol.PartiSinifId = new.PartiSinifId and PSKaliteKontrol.Topno = new.TopNo and PSKaliteKontrol.PSKaliteKontrolId <> new.PSKaliteKontrolId
        into :KS;
      if (KS > 0) then
        exception DokumaKumasAyniTopNoOlamaz;
    end
  end

  if (KKDaraPaketlemedenGelsin = 1) then
  begin
   if (new.dara=0) then
     begin
         Select Ambalaj.TupRolikAgirligi from PartiStok
           left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
           left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
           left join Ambalaj on Ambalaj.AmbalajID = PaketlemeSekli.AmbalajID
           where PartiSinifId = new.PartiSinifId
           into :TupRolikAgirligi;
         if (TupRolikAgirligi > 0) then New.Dara = TupRolikAgirligi;
      end
  end

  New.OrtalamaEn = Null;

  if (EgemenStrToIntOlurmu(New.BaslangicEn) = 1 and EgemenStrToIntOlurmu(New.OrtaEn) = 1 and EgemenStrToIntOlurmu(New.BitisEn) = 1) then
    New.OrtalamaEn = (EgemenStrToInt(New.BaslangicEn) + EgemenStrToInt(New.OrtaEn) + EgemenStrToInt(New.BitisEn)) / 3;


  if (New.LinkIslemi <> 1) then
  begin
    if (Old.Kilo <> New.Kilo or Old.Metre <> New.Metre or Old.Dara <> New.Dara) then
    begin
      if (new.PartiStokTartiId is not null) then exception TartiKaydiYapilmisKiloDegisemez;
    end
  end

  if (Old.OnayDurumu <> New.OnayDurumu) then
  begin
    if (new.OnayDurumu = 0) then
    begin
      new.OnayKullaniciID = Null;
      new.OnayTarihi = Null;
    end
    else new.OnayTarihi = current_TimeStamp(0);
  end


  if (KaliteKontrolMetreHesapSekli = 0) then
  begin
    if (new.Metre is null or new.Metre <= 0 or Old.Kilo <> New.Kilo or old.Dara <> New.Dara) then
    begin
      Select PartiStok.HacimEn, PartiStok.Gramaj, KKBitirenPersonelId, PaketlemeSekli.Tup, Firma.MetreHesabiIcinCarpan from PartiStok
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
        left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
        left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
        left join Firma on Firma.FirmaID = Parti.FirmaID
        where PartiSinifId = new.PartiSinifId
        into :HacimEn, :Gramaj, :KKBitirenPersonelId, :FTup, :MetreHesabiIcinCarpan;

      TF = EgemenStrToIntOlurmu(Gramaj);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(Gramaj);

      if (FTup is null) then FTup = 0;

      if (FTup = 0) then
        FGramaj = (IntGramaj * HacimEn) / 100000.000;
      else
        FGramaj = (IntGramaj * HacimEn * 2) / 100000.000;

      if (FGramaj is null) then FGramaj = 0;

      if (FGramaj <> 0) then New.Metre = (New.Kilo - New.Dara)/ FGramaj;
      else New.Metre = 0;
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 1) then /* tekboy */
  begin
    if (new.Metre is null or new.Metre = 0) then
    if (StrKarsilastir(Old.En, new.En) = 0 or StrKarsilastir(Old.Gramaj, new.Gramaj) = 0 or Old.Kilo <> New.Kilo or Old.Dara <> New.Dara) then
    begin
      Select Firma.MetreHesabiIcinCarpan, BoyaSiparisDetay.HesaplamaCarpimDegeri from PartiStok
        left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
        left join Firma on Firma.FirmaID = Parti.FirmaID
        where PartiSinifId = new.PartiSinifId
        into :MetreHesabiIcinCarpan, :HesaplamaCarpimDegeri;

      if (HesaplamaCarpimDegeri is not null) then
        MetreHesabiIcinCarpan = HesaplamaCarpimDegeri;

      TF = EgemenStrToIntOlurmu(New.Gramaj);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(New.Gramaj);

      TF = EgemenStrToIntOlurmu(New.En);

      if (TF = 0) then IntEn = 0;
      else IntEn = EgemenStrToInt(New.En);

      FHesap = IntEn * IntGramaj;

      if (FHesap <> 0) then
        new.Metre = (1000 / FHesap ) * MetreHesabiIcinCarpan * (New.Kilo - New.Dara);
      else
        new.Metre = 0;
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 2) then /* Kral */
  begin
    if (new.Metre is null or new.Metre = 0 or Old.Kilo <> New.Kilo) then
    begin
      Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
        where PartiStokFizikKontrol.partiSinifID = new.partiSinifID and PartiStokFizikKontrol.SonKayit = 1
        into :FEnVerilen, :FGramajVerilen;

      TF = EgemenStrToIntOlurmu(FGramajVerilen);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(FGramajVerilen);

      TF = EgemenStrToIntOlurmu(FEnVerilen);

      if (TF = 0) then IntEn = 0;
      else IntEn = EgemenStrToInt(FEnVerilen);

      FHesap = IntEn * IntGramaj;

      if (FHesap <> 0) then
        new.Metre = (1000 / FHesap ) * 100 * New.Kilo;
      else
        new.Metre = 0;
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 3) then /* NY */
  begin
    if (new.Metre is null or new.Metre = 0 or Old.Kilo <> New.Kilo) then
    begin
      Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
        where PartiStokFizikKontrol.partiSinifID = new.partiSinifID and PartiStokFizikKontrol.SonKayit = 1
        into :FEnVerilen, :FGramajVerilen;

      TF = EgemenStrToIntOlurmu(FGramajVerilen);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(FGramajVerilen);

      TF = EgemenStrToIntOlurmu(FEnVerilen);

      if (TF = 0) then IntEn = 0;
      else IntEn = EgemenStrToInt(FEnVerilen);

      FHesap = IntEn * IntGramaj;

      if (FHesap <> 0) then
        new.Metre = (1000 / FHesap ) * 100 * (New.Kilo - New.Dara);
      else
        new.Metre = 0;
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 4) then /*Ottoman Bursa*/
  begin
    if (new.Metre is null or new.Metre = 0) then
    if (Old.Kilo <> New.Kilo or old.Dara <> New.Dara) then
    begin
      Select PartiStok.HacimEn, PartiStok.Gramaj, KKBitirenPersonelId, PaketlemeSekli.Tup, Firma.MetreHesabiIcinCarpan from PartiStok
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
        left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
        left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
        left join Firma on Firma.FirmaID = Parti.FirmaID
        where PartiSinifId = new.PartiSinifId
        into :HacimEn, :Gramaj, :KKBitirenPersonelId, :FTup, :MetreHesabiIcinCarpan;

      TF = EgemenStrToIntOlurmu(Gramaj);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(Gramaj);

      if (FTup = 0) then
        FGramaj = (IntGramaj * HacimEn) / 100000.000;
      else
        FGramaj = (IntGramaj * HacimEn * 2) / 100000.000;

      if (FGramaj is null) then FGramaj = 0;

      if (FGramaj <> 0) then New.Metre = (New.Kilo - New.Dara)/ FGramaj;
      else New.Metre = 0;
    end
  end
  else  if (KaliteKontrolMetreHesapSekli = 5) then /* Emin */
  begin
    if (new.Kilo is null or new.Kilo = 0 or Old.Metre <> New.Metre) then
    begin
      TF = EgemenStrToIntOlurmu(New.Gramaj);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(New.Gramaj);

      TF = EgemenStrToIntOlurmu(substring(New.En from 1 for 3));

      if (TF = 0) then IntEn = 0;
      else IntEn = EgemenStrToInt(substring(New.En from 1 for 3));

      FHesap = IntEn * IntGramaj;

      if (FHesap <> 0) then new.Kilo = (FHesap /100000) *  (New.Metre);
      else new.Kilo = 0;
    end
  end
  else  if (KaliteKontrolMetreHesapSekli = 6) then /* Pakİpek */
  begin
    Select PartiStok.HacimEn, PartiStok.Gramaj, PaketlemeSekli.Tup from PartiStok
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
      left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
      where PartiStok.PartiSinifId = new.PartiSinifId
    into :HacimEn, :Gramaj, :FTup;

    if (FTup is null) then FTup = 0;

    TF = EgemenStrToIntOlurmu(Gramaj);

    if (TF = 0) then IntGramaj = 0;
    else IntGramaj = EgemenStrToInt(Gramaj);

    TF = EgemenStrToIntOlurmu(HacimEn);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(HacimEn);

    FHesap = IntEn * IntGramaj;

    if (FHesap <> 0) then new.Kilo = (FHesap /100000) *  (New.Metre);
    else new.Kilo = 0;
  end
  else if (KaliteKontrolMetreHesapSekli = 7) then /* Gülle */
  begin

    Select Ambalaj.TupRolikAgirligi from PartiStok
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
        left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
        left join Ambalaj on Ambalaj.AmbalajID = PaketlemeSekli.AmbalajID
      where PartiSinifId = new.PartiSinifId
      into :TupRolikAgirligi;


    Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
      left join Parti on Parti.PartiKayitId = PartiStokFizikKontrol.PartiKayitId
      where PartiStokFizikKontrol.PartiSinifID = new.PartiSinifID and PartiStokFizikKontrol.SonKayit = 1
      into :FEnVerilen, :FGramajVerilen;

    TF = EgemenStrToIntOlurmu(FGramajVerilen);

    if (TF = 0) then IntGramaj = 0;
    else IntGramaj = EgemenStrToInt(FGramajVerilen);

    TF = EgemenStrToIntOlurmu(FEnVerilen);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(FEnVerilen);

  if (TupRolikAgirligi > 0) then New.Dara = TupRolikAgirligi/1000;
           else
    New.Dara = Yuvarla((Cast(IntEn * 3.9 as Double Precision) / 1000), 100);

     New.brutmetre = (1000 / Cast(IntEn * IntGramaj as Double Precision) ) * 100 * (new.Kilo-New.Dara);
  end
  else if (KaliteKontrolMetreHesapSekli = 8) then --NS ÖRME
  begin
    if (OzelMusteri = 0) then
    begin
        if (new.Metre is null or new.Metre <= 0 or Old.Kilo <> New.Kilo or old.Dara <> New.Dara) then
        begin
          Select PartiStok.HacimEn, PartiStok.Gramaj, KKBitirenPersonelId, PaketlemeSekli.Tup, Firma.MetreHesabiIcinCarpan from PartiStok
            left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
            left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
            left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
            left join Firma on Firma.FirmaID = Parti.FirmaID
            where PartiSinifId = new.PartiSinifId
            into :HacimEn, :Gramaj, :KKBitirenPersonelId, :FTup, :MetreHesabiIcinCarpan;

          TF = EgemenStrToIntOlurmu(Gramaj);

          if (TF = 0) then IntGramaj = 0;
          else IntGramaj = EgemenStrToInt(Gramaj);

          if (FTup is null) then FTup = 0;

          if (FTup = 0) then
            FGramaj = (IntGramaj * HacimEn) / 100000.000;
          else
            FGramaj = (IntGramaj * HacimEn * 2) / 100000.000;

          if (FGramaj is null) then FGramaj = 0;

          if (FGramaj <> 0) then New.Metre = (New.Kilo - New.Dara)/ FGramaj;
          else New.Metre = 0;
        end
    end
    else
    begin
        if (StrKarsilastir(Old.En, new.En) = 0 or StrKarsilastir(Old.Gramaj, new.Gramaj) = 0 or Old.Kilo <> New.Kilo or Old.Dara <> New.Dara) then
        begin
          TF = EgemenStrToIntOlurmu(New.Gramaj);

          if (TF = 0) then IntGramaj = 0;
          else IntGramaj = EgemenStrToInt(New.Gramaj);

          TF = EgemenStrToIntOlurmu(New.En);

          if (TF = 0) then IntEn = 0;
          else IntEn = EgemenStrToInt(New.En);

          FHesap = IntEn * IntGramaj;

          if (FHesap <> 0) then
            new.Metre = (1000 / FHesap ) * 100 * (New.Kilo - New.Dara);
          else
            new.Metre = 0;
        end
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 9) then /* Özsar */
  begin
    Select BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, PaketlemeSekli.Tup from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
    where PartiStok.PartiSinifId = new.PartiSinifId
    into :FEnIstenen, :FGramajIstenen, :FTup;

    TF = EgemenStrToIntOlurmu(FGramajIstenen);

    if (TF = 0) then IntGramaj = 0;
    else IntGramaj = EgemenStrToInt(FGramajIstenen);

    TF = EgemenStrToIntOlurmu(FEnIstenen);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(FEnIstenen);

    if (FTup = 0) then
      FEn  = IntEn * 0.01;
    else
      FEn  = 2 * IntEn * 0.01;

    FGramaj =  IntGramaj * 0.001 ;

    FHesap  = (FEn) * (FGramaj);

    if (FHesap <> 0) then
      new.Metre = (New.Kilo - New.Dara) /  FHesap;
    else
      new.Metre = 0;
  end
  else if (KaliteKontrolMetreHesapSekli = 10) then  /* Feyteks*/
  begin
    TF = EgemenStrToIntOlurmu(New.En);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(New.En);

    if ((New.Metre <> 0) and (IntEn <> 0) and ((New.Kilo - New.Dara)<>0)) then
      New.Gramaj = egemenintegertostr( ((New.Kilo - New.Dara)*1000) / (cast(IntEn as Double Precision)/100 * New.Metre) );

  end


  if (OzelMusteri = 1 and KaliteKontrolMetreHesapSekli <> 8) then /* Hugo Boss için buraya if gireceğiz*/
  begin
    if (New.Metre <> Old.Metre or Coalesce(old.BrutMetre, 0) <> Coalesce(new.BrutMetre, 0) or Coalesce(old.PaketlemeHataPuani, 0) <> Coalesce(new.PaketlemeHataPuani, 0)) then
    begin
      if (New.Metre <> Old.Metre and
           (Coalesce(old.BrutMetre, 0) = Coalesce(new.BrutMetre, 0) or
            Coalesce(new.BrutMetre, 0) = 0 or
            New.Metre > Coalesce(new.BrutMetre, 0))
         ) then New.BrutMetre = New.Metre;

      if (New.BrutMetre > 0) then
      begin
        Select CezaMetre from  PSKKontrolCezaMetreHesapla(New.PartiSinifId, New.PSKaliteKontrolId) into :CezaMetre;
        New.Metre = New.BrutMetre - :CezaMetre;
      end
    end
  end

  if (old.Metre <> new.Metre or Coalesce(old.PaketlemeHataPuani, 0) <> Coalesce(new.PaketlemeHataPuani, 0)) then
  begin
    select Firma.KaliteKontrolHataOrani from Parti
    left join Firma on Firma.FirmaID = Parti.FirmaId
    where Parti.PartiKayitId = new.PartiKayitId into :KaliteKontrolHataOrani;

    if (KaliteKontrolHataOrani is not null) then
    begin
      if (new.PaketlemeHataPuani > new.Metre * (Cast(KaliteKontrolHataOrani as Double Precision)/100)) then
        new.KaliteNo = 2;
      else
        new.KaliteNo = 1;
    end
  end


  if (FEgemenFirmaId = 138) then /*Artesa*/
  begin

    TF = EgemenStrToIntOlurmu(New.En);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(New.En);

    if (New.Metre > 0 and IntEn > 0) then
    New.Gramaj = egemenintegertostr( ((New.Kilo - New.Dara)*1000) / (cast(IntEn as Double Precision)/100 * New.Metre) );

    if (new.OlcuId is not null) then
    begin
      select Max(BoyaSiparisOlcu.Gramaj) from BoyaSiparisOlcu
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisOlcu.BoyaSiparisDetayId
      left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayID
      where PartiStok.partisinifid = new.partisinifid and BoyaSiparisOlcu.OlcuId = new.OlcuId
      into :OlcuGramaj ;

      if (olcugramaj > 0) then
      new.adet = ((New.Kilo - New.Dara)*1000) / olcugramaj ;
    end

  end

  if (FEgemenFirmaId = 70) then /*CeylanTeks*/
  begin
    new.Metre = Yuvarla (new.Metre, 10) ;
  end
  new.LinkIslemi = 0;
  new.DegisimTarihi = current_TimeStamp(0);
end
