Definition

CREATE TRIGGER BEFOREDELETEORTRANSFERFISIDETAY FOR ORMERAFTRANSFERFISIDETAY
ACTIVE BEFORE DELETE POSITION 0
as
declare variable FOrmeRafTransferFisiDetayId   Integer;
declare variable TransferEdildi                EgeBoolean;
begin
  Select TransferEdildi From OrmeRafTransferFisi Where OrmeRafTransferFisiId = old.OrmeRafTransferFisiId into :TransferEdildi;
  if (TransferEdildi = 1) then exception TrasferEdilmisDegistirilmez;

  Select First 1 OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId from OrmeRafTransferFisiDetay where OrmeRafTransferFisiDetay.StokHareketDetayId = Old.StokHareketDetayId and OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId <> Old.OrmeRafTransferFisiDetayId
    Order By InsertTarihi Desc
  into :FOrmeRafTransferFisiDetayId;

  Update StokHareketDetay set OrmeRafTransferFisiDetayID = :FOrmeRafTransferFisiDetayId where StokHareketDetayId = old.StokHareketDetayId;

end
