CREATE TRIGGER AFTERINSERTOTOMATIKSARFFISI FOR OTOMATIKSARFFISI
ACTIVE AFTER INSERT POSITION 0
as
declare Variable FHSSarfFisiYil integer;
declare Variable FHSSarfFisiNo integer;
declare Variable FHSSarfFisiId integer;
declare Variable FHSSarfFisiDetayId integer;
declare Variable FHamStokMasterId integer;

declare Variable FKaynakHamId integer;

declare Variable Fason integer;

begin
  select OtomatikSarfFisSerisi from Sistem into :OtomatikSarfFisSerisi;

  if (new.RafDetayId is not Null) then
  begin
    select RafDetay.HamId from RafDetay
      where RafDetay.RafDetayId = new.RafDetayId
      into :FKaynakHamId;
  end
  else if (new.PartiSinifId is not Null) then
  begin
    select PartiStok.HamId from PartiStok
      where PartiStok.PartiSinifId = new.PartiSinifId

                                  IrsaliyeTarihi, FiiliSevkTarihi, IrsaliyeGrupId, DovizId, AracId, SoforId, TransferYeri, InsertKullaniciId, KullaniciId)
        values(:FHamStokIrsaliyeId, 0, :FirmaId, 0, :Sinif, :Fason, :OtomatikSarfFisSerisi, :IrsaliyeNo,
                                  current_date, current_timestamp(0), :IrsaliyeGrupId, :DovizId, :AracId, :SoforId, 3, new.KullaniciId, new.KullaniciId);

      FHamStokMasterId = gen_id(HamStokMaster_Id_Gen, 1);

      insert into HamStokMaster(HamStokIrsaliyeId, HamStokMasterId, StokFirmaId, HamId, Kilo,
                                Lot, MelanjKodu, OrmeStokGirisNo, OrmeciMakinaNo, TupAcikEnMayli, MusteriPartiNo, MusteriSiparisNo, ReferansSiparisNo,

                                  HamKontrolIplikUzunlugu3, HamKontrolIplikUzunlugu4, MerserizeHasligiYapildi, YikamaHasligiKabul, MerserizeHasligiKabul,
                                  CekmezlikStandardiId, DemirTestiYapildi, DemirTestiKabul, IgneSayisi, HamKontrolKullanimYeri, HamKontrolNotu,
                                  HamStokSarfFisiDetayId, OtomatikSarfHSMId, OtomatikSarfFisiId, KapAdedi, OlcuId)
        Select :FHamStokIrsaliyeId, :FHamStokMasterId, HamStokMaster.StokFirmaId, new.HedefHamId,
               new.Miktar, HamStokMaster.Lot, HamStokMaster.MelanjKodu, HamStokMaster.OrmeStokGirisNo, HamStokMaster.OrmeciMakinaNo,
               HamStokMaster.TupAcikEnMayli, HamStokMaster.MusteriPartiNo,
               HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo, HamStokMaster.RenkId, HamStokMaster.Renk,
               HamStokMaster.Pus, HamStokMaster.Feine,
               HamStokMaster.Tamir, HamStokMaster.KaliteNo, HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo, HamStokMaster.UreticiFirmaId,
               HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3, Null, HamStokMaster.OrmeGirisIrsaliyeNo, HamStokMaster.Aciklama,
               HamStokMaster.BirimFiyat, HamStokMaster.Brut, HamStokMaster.InsertKullaniciId, HamStokMaster.KullaniciId, HamStokMaster.OrmePlanlamaId,
                  HamStokMaster.NemOrani, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj,
                  HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj,
                  HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu,
                  HamStokMaster.YikamaSonrasiRaporBoyu, HamStokMaster.YikamaOncesiCPI,
                  HamStokMaster.YikamaSonrasiCPI, HamStokMaster.YikamaOncesiWPI, HamStokMaster.YikamaSonrasiWPI, HamStokMaster.YikamaHasligiYapildi,
                  HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2,
                  HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4, HamStokMaster.MerserizeHasligiYapildi,
                  HamStokMaster.YikamaHasligiKabul, HamStokMaster.MerserizeHasligiKabul,
                  HamStokMaster.CekmezlikStandardiId, HamStokMaster.DemirTestiYapildi, HamStokMaster.DemirTestiKabul, HamStokMaster.IgneSayisi,
                  HamStokMaster.HamKontrolKullanimYeri, HamStokMaster.HamKontrolNotu,
                  :FHSSarfFisiDetayId, HamStokMaster.HamStokMasterId, new.OtomatikSarfFisiId, new.KapAdedi, OlcuId from HamStokMaster
        where HamStokMaster.HamStokMasterID = :EskiHamStokMasterId;

      Insert into GeciciKabulRaf(HamStokMasterId, RafId, KapAdedi, Kilo, InsertKullaniciId, KullaniciId)

    end
  end
  else if (new.PartiSinifId is not Null) then
  begin
    for select HamStokIrsaliye.FirmaId, HamStokIrsaliye.Sinif, HamStokIrsaliye.Fason,
        HamStokIrsaliye.AracId, HamStokIrsaliye.SoforId, HamStokIrsaliye.TransferYeri, HamStokIrsaliye.IrsaliyeGrupId,
        HamStokIrsaliye.DovizId, HamStokMaster.HamStokMasterId, RafDetay.RafId from PartiStokDetay
      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
      where PartiStokDetay.PartiSinifId = new.PartiSinifId
      into :FirmaId, :Sinif, :Fason, :AracId, :SoforId, :TransferYeri, :IrsaliyeGrupId, :DovizId, :EskiHamStokMasterId, :RafId do
    begin

      IrsaliyeNo = Gen_ID(OtomatikSarfHSFisNo_Id_Gen, 1);
      FHamStokIrsaliyeId = Gen_ID(HamStokIrsaliye_Id_Gen, 1);
      insert into HamStokIrsaliye(HamStokIrsaliyeId, HamStokHareketTipiId, FirmaId, EvrakTipi, Sinif, Fason, IrsaliyeSeriNo, IrsaliyeNo,
                                  IrsaliyeTarihi, FiiliSevkTarihi, IrsaliyeGrupId, DovizId, AracId, SoforId, TransferYeri, InsertKullaniciId, KullaniciId)
        values(:FHamStokIrsaliyeId, 0, :FirmaId, 0, :Sinif, :Fason, :OtomatikSarfFisSerisi, :IrsaliyeNo,
                                  current_date, current_timestamp(0), :IrsaliyeGrupId, :DovizId, :AracId, :SoforId, 3, new.KullaniciId, new.KullaniciId);

      FHamStokMasterId = gen_id(HamStokMaster_Id_Gen, 1);

      insert into HamStokMaster(HamStokIrsaliyeId, HamStokMasterId, StokFirmaId, HamId, Kilo,
                                Lot, MelanjKodu, OrmeStokGirisNo, OrmeciMakinaNo, TupAcikEnMayli, MusteriPartiNo, MusteriSiparisNo, ReferansSiparisNo,
                                RenkId, Renk, Pus, Feine,
                                Tamir, KaliteNo, EskiPartiYil, EskiPartiNo, UreticiFirmaId, IplikUzunlugu1, IplikUzunlugu2, IplikUzunlugu3,

                                  EndenCekmezlik, BoydanCekmezlik, HamDonme, YikamaOncesiRaporBoyu, YikamaSonrasiRaporBoyu, YikamaOncesiCPI,
                                  YikamaSonrasiCPI, YikamaOncesiWPI, YikamaSonrasiWPI, YikamaHasligiYapildi, HamKontrolIplikUzunlugu1, HamKontrolIplikUzunlugu2,
                                  HamKontrolIplikUzunlugu3, HamKontrolIplikUzunlugu4, MerserizeHasligiYapildi, YikamaHasligiKabul, MerserizeHasligiKabul,
                                  CekmezlikStandardiId, DemirTestiYapildi, DemirTestiKabul, IgneSayisi, HamKontrolKullanimYeri, HamKontrolNotu,
                                  HamStokSarfFisiDetayId, OtomatikSarfHSMId, OtomatikSarfFisiId, KapAdedi, OlcuId)
        Select :FHamStokIrsaliyeId, :FHamStokMasterId, HamStokMaster.StokFirmaId, new.HedefHamId,
               new.Miktar, HamStokMaster.Lot, HamStokMaster.MelanjKodu, HamStokMaster.OrmeStokGirisNo, HamStokMaster.OrmeciMakinaNo,
               HamStokMaster.TupAcikEnMayli, HamStokMaster.MusteriPartiNo,
               HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo, HamStokMaster.RenkId, HamStokMaster.Renk,
               HamStokMaster.Pus, HamStokMaster.Feine,
               HamStokMaster.Tamir, HamStokMaster.KaliteNo, HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo, HamStokMaster.UreticiFirmaId,
               HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3, Null, HamStokMaster.OrmeGirisIrsaliyeNo, HamStokMaster.Aciklama,
               HamStokMaster.BirimFiyat, HamStokMaster.Brut, HamStokMaster.InsertKullaniciId, HamStokMaster.KullaniciId, HamStokMaster.OrmePlanlamaId,
                  HamStokMaster.NemOrani, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj,
                  HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj,
                  HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu,
                  HamStokMaster.YikamaSonrasiRaporBoyu, HamStokMaster.YikamaOncesiCPI,
                  HamStokMaster.YikamaSonrasiCPI, HamStokMaster.YikamaOncesiWPI, HamStokMaster.YikamaSonrasiWPI, HamStokMaster.YikamaHasligiYapildi,
                  HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2,
                  HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4, HamStokMaster.MerserizeHasligiYapildi,
                  HamStokMaster.YikamaHasligiKabul, HamStokMaster.MerserizeHasligiKabul,
                  HamStokMaster.CekmezlikStandardiId, HamStokMaster.DemirTestiYapildi, HamStokMaster.DemirTestiKabul, HamStokMaster.IgneSayisi,
                  HamStokMaster.HamKontrolKullanimYeri, HamStokMaster.HamKontrolNotu,
                  :FHSSarfFisiDetayId, HamStokMaster.HamStokMasterId, new.OtomatikSarfFisiId, new.KapAdedi, OlcuId from HamStokMaster

      Insert into GeciciKabulRaf(HamStokMasterId, RafId, KapAdedi, Kilo, InsertKullaniciId, KullaniciId)

  end
