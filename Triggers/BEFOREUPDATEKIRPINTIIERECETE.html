CREATE TRIGGER BEFOREUPDATEKIRPINTIIERECETE FOR KIRPINTIISEMRIRECETE
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KirpintiMiktari Numeric(12,3);
declare variable ReceteMiktari Numeric(12,3);
begin
  Select Sum(Miktar) from KirpintiIsEmriRecete where KirpintiIsEmriRecete.KirpintiIsEmriKayitID = new.KirpintiIsEmriKayitID and KirpintiIsEmriRecete.KirpintiIsEmriReceteID <> new.KirpintiIsEmriReceteID into :ReceteMiktari;
  if (ReceteMiktari is null) then ReceteMiktari = 0;
  ReceteMiktari = ReceteMiktari + New.Miktar;

  Select Sum(Miktar) from KirpintiIsEmri where KirpintiIsEmri.KirpintiIsEmriKayitID = new.KirpintiIsEmriKayitID into :KirpintiMiktari;
  if (KirpintiMiktari is null) then KirpintiMiktari = 0;

  if (ReceteMiktari > KirpintiMiktari) then Exception KayitHatasi  'Kırpıntı Reçete Toplam Miktari Kırpıntı Miktarını Aştı!';

end
