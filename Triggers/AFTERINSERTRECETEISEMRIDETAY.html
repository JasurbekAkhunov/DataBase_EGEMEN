CREATE TRIGGER AFTERINSERTRECETEISEMRIDETAY FOR RECETEISEMRIDETAY
ACTIVE AFTER INSERT POSITION 0
as
declare variable BanyoCarpani Double Precision;
declare variable BanyoCarpani2 Double Precision;
declare variable BanyoCarpani3 Double Precision;
declare variable BanyoCarpani4 Double Precision;
declare variable Koyultma     Double Precision;
declare variable HamStokFlotteCarpani Double Precision;
declare variable DurulamaSuyu integer;
declare variable BoyaSuyu     integer;
declare variable Tur          Double Precision;
declare variable PompaDegeri  integer;
declare variable Duze         integer;
declare variable BoyamaSeviyesi integer;
declare variable MakinaId  varchar(10);
declare variable TuzCarpani            Double Precision;
declare variable IslaklikOrani         Double Precision;
declare variable KuleHizi              Double Precision;
declare variable MaxKuleHizi           Double Precision;
declare variable FloteOraniYikama      Double Precision;
declare variable SuTipiArtiTuz         Double Precision;
begin
  select BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, Koyultma, DurulamaSuyu, BoyaSuyu, Tur, PompaDegeri, Duze, BoyamaSeviyesi, HamStokFlotteCarpani,
         TuzCarpani, IslaklikOrani, KuleHizi, MaxKuleHizi, FloteOraniYikama, SuTipiArtiTuz from ReceteIsEmri
    where ReceteIsEmri.ReceteIsEmriYil = new.ReceteIsEmriYil and ReceteIsEmri.ReceteIsEmriId = new.ReceteIsEmriId
    into :BanyoCarpani, :BanyoCarpani2, :BanyoCarpani3, :BanyoCarpani4, :Koyultma, :DurulamaSuyu, :BoyaSuyu, :Tur, :PompaDegeri, :Duze,  :BoyamaSeviyesi, :HamStokFlotteCarpani,
         :TuzCarpani, :IslaklikOrani, :KuleHizi, :MaxKuleHizi, :FloteOraniYikama, :SuTipiArtiTuz;

  update PartiAsamalari set BanyoCarpani = :BanyoCarpani,
                            BanyoCarpani2 = :BanyoCarpani2,
                            BanyoCarpani3 = :BanyoCarpani3,
                            BanyoCarpani4 = :BanyoCarpani4,
                            Koyultma = :Koyultma,
                            DurulamaSuyu = :DurulamaSuyu,
                            BoyamaSeviyesi = :BoyamaSeviyesi,
                            BoyaSuyu = :BoyaSuyu,
                            Tur = :Tur,
                            PompaDegeri = :PompaDegeri,
                            Duze = :Duze,
                            HamStokFlotteCarpani = :HamStokFlotteCarpani,
                            TuzCarpani = :TuzCarpani,
                            IslaklikOrani = :IslaklikOrani,
                            KuleHizi  = :KuleHizi,
                            MaxKuleHizi = :MaxKuleHizi,
                            FloteOraniYikama = :FloteOraniYikama,
                            SuTipiArtiTuz = :SuTipiArtiTuz,
                            KullaniciId = New.KullaniciId where PartiAsamalari.PartiAsamalariId = new.PartiAsamalariId;

  Select Max(UretimPlanlama.MakinaId) from ReceteIsEmriDetay
    left join BolumPlanlama on BolumPlanlama.PartiAsamalariId = ReceteIsEmriDetay.PartiAsamalariId
    left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
    left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
    where ReceteIsEmriDetay.ReceteIsEmriYil = new.ReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = new.ReceteIsEmriId and
    ReceteIsEmriDetay.PartiAsamalariId <> new.PartiAsamalariId
    into :MakinaId;

  if (MakinaId is not Null) then EXECUTE PROCEDURE ASAMAGIRISINDENPLANLAMAYAP(new.PartiAsamalariId, :MakinaId, new.InsertKullaniciId);
end
