CREATE TRIGGER BEFOREINSERTURUNURETIMTARTI FOR URUNURETIMTARTI
ACTIVE BEFORE INSERT POSITION 0
AS
declare variable StoktanDusuldu integer;
begin

  StoktanDusuldu = 0;
  select count(*) from Stok where Stok.UrunUretimId = new.UrunUretimId into :StoktanDusuldu;

  if (StoktanDusuldu > 0) then exception StoktanDusulduDegisemez;
  New.UrunUretimTartiId = gen_id(UrunUretimTarti_Id_Gen, 1);
end
