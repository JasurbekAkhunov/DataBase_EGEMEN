Definition

CREATE TRIGGER BEFOREUPDATESTOKHAREKET0 FOR STOKHAREKET
ACTIVE BEFORE UPDATE POSITION 0
as

declare variable FBoyaSiparisKayitId  integer;
declare variable FHamId               integer;
declare variable uitOrme integer;
declare variable uitBukum integer;
declare variable uitBoya  integer;
declare variable uitDokuma  integer;
declare variable uitCozgu  integer;
declare variable uitHasil  integer;
declare variable uitIndigo  integer;
declare variable BoyaKapali integer;
declare variable BCBoyaSiparisDetayZorunlu integer;
declare variable XBoyaSiparisKayitId       integer;
declare variable RafDepoId      varchar(10);
declare variable SonDonemBitisTarihi TimeStamp;
declare Variable FVerilenSiparisMiktar Double Precision;
declare Variable FVerilenSipAsilabilirMiktar Double Precision;
declare Variable FTopMiktar            Double Precision;
declare Variable VerilenSiparisMiktariAsilamaz integer;
declare variable IrsTarihFatTarihtenBuyukOlamaz integer;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable FFaturaTarihi TimeStamp;
declare variable shtFasonUretim integer;
declare variable shtUretim      integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis      integer;
declare variable shtFasonAlis         integer;
declare variable KS         integer;
declare variable Kapali     EgeBoolean;
begin
  Select shtFasonUretim, shtUretim, shtFasonUretimeCikis, shtUretimeCikis, shtFasonAlis from OrmeStokHareketTipiGetir into :shtFasonUretim, :shtUretim, :shtFasonUretimeCikis, :shtUretimeCikis, :shtFasonAlis;

  if (old.GirisKKBitirenKullaniciId is null and new.GirisKKBitirenKullaniciId is not null and new.DokumaHareketi = 1) then
    select Sum(GirisKaliteKontrol.Metre) from GirisKaliteKontrol where GirisKaliteKontrol.StokHareketId = new.StokHareketId into new.Metre;

  if (New.DokumaHareketi = 1) then /*Miktar birimi tespit edilmeli. Cozgu deposuna giren metre olacak, digerleri kilo*/
  begin
    select Zorunlu from DokumaMetreZorunlu(New.DepoId, New.HamId) into :KS;

    if (KS = 1) then
      New.MiktarBirim = 1;
    else
      New.MiktarBirim = 0;

    if (Coalesce(Old.Miktar,-1) = Coalesce(New.Miktar,-1)) then
    begin
      if (New.MiktarBirim = 1) then
        New.Miktar = New.Metre;
      else
        New.Miktar = New.Kilo;
    end
    else
    begin
      if (New.MiktarBirim = 1 and Coalesce(Old.Metre,-1) = Coalesce(New.Metre,-1)) then
        New.Metre = New.Miktar;
      else if (New.MiktarBirim = 0 and Coalesce(Old.Kilo,-1) = Coalesce(New.Kilo,-1)) then
        New.Kilo = New.Miktar;
    end
  end

  if (IntKarsilastir(old.CozguIsEmriKayitId, new.CozguIsEmriKayitId) = 0 and new.UretimIslemTipiId = uitCozgu and new.CozguIsEmriKayitId is not null) then
  begin
    select CozguIsEmri.Kapali from CozguIsEmri where CozguIsEmri.CozguIsEmriKayitId = new.CozguIsEmriKayitId into :Kapali;
    if (Kapali = 1) then Exception IsEmriKapaliHareketGirilemez;
  end

  if (IntKarsilastir(Old.BoyaSiparisDetayID, New.BoyaSiparisDetayID) = 0 and New.BoyaSiparisDetayID is not null) then
  begin
    Select BoyaSiparisKayitId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = new.BoyaSiparisDetayId  into :XBoyaSiparisKayitId;

    if (XBoyaSiparisKayitId <> New.BoyaSiparisKayitId) then exception Kayithatasi 'Sipariş ve Detay Referans Farklı.';
  end

  if (new.OrmeFisId is null and new.OrmeIrsaliyeId is null and new.OrmeFaturaId is null) then exception StokHareketBaglantisiYok;

  if (New.DokumaHareketi <> Old.DokumaHareketi) then exception ModulBilgisiDegistirilemez;

  if (new.NeredenUpdate < 0 or new.NeredenUpdate > 4) then exception NeredenUpdateBilgisiHatali;

  if (old.NeredenGirildi <> new.NeredenGirildi) then exception NeredenGirildiDegisemez;
  if (old.HareketTipi <> new.HareketTipi) then exception HareketTipiDegisemez;
  if (Old.OrmeIrsaliyeId is not null and New.OrmeIrsaliyeId is not null and Old.OrmeIrsaliyeId <> New.OrmeIrsaliyeId) then exception IrsaliyeIdDegisemez;
  if (Old.OrmeFaturaId is not null and New.OrmeFaturaId is not null and Old.OrmeFaturaId <> New.OrmeFaturaId) then exception FaturaIdDegisemez;

  if (New.NeredenUpdate <> 3 and New.NeredenUpdate <> 4) then
  if ((New.HamStokIrsaliyeId is not null or New.IplikIrsaliyeId is not null) and new.HareketTipi <> :shtFasonUretimeCikis ) then exception KayitHatasi ' Evrak Transfer edilmiş, değişemez..';

  New.GirisLinkIslemi = New.LinkIslemi;

  Select uitBukum, uitOrme, uitBoya, uitDokuma, uitCozgu, uitHasil, uitIndigo from UretimIslemTipiGetir into :UitBukum, :uitOrme, :uitBoya, :uitDokuma, :uitCozgu, :uitHasil, :uitIndigo;

  if (new.UretimIslemTipiId = uitOrme and new.OrmePlanlamaId is null) then exception UretimIcinTalimatSecilmemis;
  if (new.UretimIslemTipiId = uitDokuma and new.DokumaPlanlamaId is null) then exception UretimIcinTalimatSecilmemis;
  --if (new.UretimIslemTipiId = uitCozgu and new.CozguIsEmriKayitID is null and (new.HareketTipi = shtFasonUretim or new.HareketTipi = shtUretim)) then exception UretimIcinTalimatSecilmemis;
  if (new.UretimIslemTipiId = uitCozgu and new.CozguIsEmriReceteID is null and (new.HareketTipi = shtFasonUretimeCikis or new.HareketTipi = shtUretimeCikis)) then exception UretimIcinTalimatSecilmemis;
  if (new.UretimIslemTipiId = uitHasil and new.HasilIsEmriKayitID is null) then exception UretimIcinTalimatSecilmemis;
  if (new.UretimIslemTipiId = uitIndigo and new.IndigoIsEmriKayitID is null) then exception UretimIcinTalimatSecilmemis;

  Select BCBoyaSiparisDetayZorunlu, VerilenSiparisMiktariAsilamaz, IrsTarihFatTarihtenBuyukOlamaz from Sistem where SistemID = 1 into :BCBoyaSiparisDetayZorunlu, :VerilenSiparisMiktariAsilamaz, :IrsTarihFatTarihtenBuyukOlamaz;

  if (BCBoyaSiparisDetayZorunlu = 1 and New.UretimPartiNo <> '') then
  if (New.BoyahaneFasonIade = 0 and new.UretimIslemTipiId = uitBoya and new.BoyaSiparisDetayId is null) then exception UretimIcinSiparisNoGirilmeli;

  if (New.BoyahaneFasonIade = 0) then
  if (New.UretimPartiNo <> '' and new.UretimIslemTipiId is not null and new.UretimIslemTipiId <> :UitBukum and new.UretimIslemTipiId <> :uitHasil and new.UretimIslemTipiId <> :uitIndigo and new.UretimIslemTipiId <> :UitCozgu and New.BoyaSiparisKayitId is null) then exception UretimIcinSiparisNoGirilmeli;

  if (IntKarsilastir(Old.BoyaSiparisDetayId, New.BoyaSiparisDetayId) = 0 and new.BoyaSiparisDetayId is not null) then
  begin
    Select Kapali from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = New.BoyaSiparisDetayId into :BoyaKapali;
    if (BoyaKapali = 2) then exception SiparisIptalEdilmis;
  end

  /*if (new.FaturaKesilecek = 0) then new.BirimFiyat = 0; Palben*/

  if (new.Kdv is null) then New.Kdv = 0;
  if (new.BirimFiyat is null) then new.BirimFiyat = 0;
  if (new.BirimIskontoParasal is null) then new.BirimIskontoParasal = 0;
  if (new.BirimIskontoOrani1 is null) then new.BirimIskontoOrani1 = 0;
  if (new.BirimIskontoOrani2 is null) then new.BirimIskontoOrani2 = 0;
  if (new.BirimIskontoToplami is null) then new.BirimIskontoToplami = 0;
  if (new.KdvHaricFiyat is null) then new.KdvHaricFiyat = 0;
  if (new.NetBirimFiyat is null) then new.NetBirimFiyat = 0;
  if (new.VadeGun is null) then new.VadeGun = 0;
  if (new.Lot = '') then new.Lot = Null;
  if (new.MelanjKodu = '') then new.MelanjKodu = Null;
  if (new.UretimPartiNo = '') then new.UretimPartiNo = Null;
  if (new.MiktarBirim is null) then new.MiktarBirim = 0;

  if ( (new.VerilenSiparisDetayId is not null and new.VerilenSiparisKayitId  is null ) or
      (new.VerilenSiparisDetayId is null and new.VerilenSiparisKayitId is not null )) then exception VerilenSiparisIdHatasi;

  if (IrsTarihFatTarihtenBuyukOlamaz = 1 and new.OrmeFaturaId is not Null and new.OrmeIrsaliyeID is not Null) then
  begin
    select OrmeIrsaliye.IrsaliyeTarihi from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = new.OrmeIrsaliyeId into :FIrsaliyeTarihi;
    select OrmeFatura.FaturaTarihi from OrmeFatura where OrmeFatura.OrmeFaturaId = new.OrmeFaturaId into :FFaturaTarihi;

    if (FIrsaliyeTarihi > FFaturaTarihi) then exception IrsTarFatTardanBuyukOlamaz;
  end

  if (IntKarsilastir(Old.VerilenSiparisDetayId, New.VerilenSiparisDetayId) = 0 and new.VerilenSiparisDetayId is not null) then
  begin
    Select HamId from VerilenSiparisDetay where VerilenSiparisDetay.VerilenSiparisDetayId = new.VerilenSiparisDetayId into :FHamId;

    if (New.GirisCikis = 1 and new.HamIade = 0) then
    if (FHamId is not null and FHamId <> new.HamId) then exception VerilenSiparisUrunIdHatasi;
  end

  if ((IntKarsilastir(Old.VerilenSiparisDetayId, New.VerilenSiparisDetayId) = 0 or Old.Miktar <> New.Miktar) and new.VerilenSiparisDetayId is not null) then
  begin
    Select Miktar, AsilabilirMiktar from VerilenSiparisDetay where VerilenSiparisDetay.VerilenSiparisDetayId = new.VerilenSiparisDetayId into :FVerilenSiparisMiktar, :FVerilenSipAsilabilirMiktar;

    if (VerilenSiparisMiktariAsilamaz = 1) then
    begin
      Select Sum(Miktar) from StokHareket where StokHareket.VerilenSiparisDetayID = new.VerilenSiparisDetayId and StokHareket.StokHareketID <> New.StokHareketID into :FTopMiktar;

      if (FTopMiktar is Null) then FTopMiktar = 0;

      if (FVerilenSiparisMiktar + FVerilenSipAsilabilirMiktar < FTopMiktar + New.Miktar) then Exception VerilenSiparisMiktariAsildi;
    end
  end


  if (New.GirisCikis = -1) then
  if (Old.DepoID <> New.DepoID or Old.OrmeRafID <> New.OrmeRafId) then
  begin
    Select DepoRaf.DepoID from OrmeRaf
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      where OrmeRaf.OrmeRafID = New.OrmeRafID
      into :RafDepoId;

    if (StrKarsilastir(New.DepoID, RafDepoId) = 0) then exception kayitHatasi ' Seçilen Raf, Bu Depoya Ait değil.';
  end

  select max(BitisTarihi) from DonemKapatma where Bolum = 2 and OnayKullaniciId is not Null into :SonDonemBitisTarihi;

  if (SonDonemBitisTarihi is not Null and (SonDonemBitisTarihi > new.Tarih or SonDonemBitisTarihi > old.Tarih)) then
    exception DonemKapaliHDegisemez;
end
