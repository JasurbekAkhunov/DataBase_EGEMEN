CREATE TRIGGER AFTERUPDATESTOKHAREKET0 FOR STOKHAREKET
ACTIVE AFTER UPDATE POSITION 0
as
declare variable FRafId         varchar(10);
declare variable shtDevir       integer;
declare variable shtAlis        integer;
declare variable shtSatis       integer;
declare variable shtAlisIade    integer;
declare variable shtSatisIade   integer;
declare variable shtFasonAlis   integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim  integer;
declare variable shtUretim       integer;
declare variable uitBukum integer;
declare variable uitOrme integer;
declare variable uitBoya integer;
declare variable uitDokuma integer;
declare variable FMSG varchar(255);
declare variable FirmaId                      varchar(20);
declare variable IrsaliyeTarihi               TimeStamp;
declare variable IrsaliyeNo                   IrsaliyeNoSize;
declare variable RenkAdi                      varchar(50);
declare variable GirisYeri                    varchar(50);

declare variable HarMiktar Double Precision;

  if (IntKarsilastir(Old.OrmeFaturaId, New.OrmeFaturaId) = 1) then  /* fatura kesiliyor ise islem yapma */

      begin

          update OrmePlanlama Set GerceklesenMiktar = GerceklesenMiktar - Old.Miktar where OrmePlanlama.OrmePlanlamaId = Old.OrmePlanlamaId;
        else
          update OrmePlanlama Set GerceklesenMiktar = GerceklesenMiktar - Old.Miktar, ToplamSakatOrulen = ToplamSakatOrulen - Old.Miktar where OrmePlanlama.OrmePlanlamaId = Old.OrmePlanlamaId;

        if (New.Sakat = 0) then
          update OrmePlanlama Set GerceklesenMiktar = GerceklesenMiktar + New.Miktar where OrmePlanlama.OrmePlanlamaId = new.OrmePlanlamaId;
        else
          update OrmePlanlama Set GerceklesenMiktar = GerceklesenMiktar + New.Miktar, ToplamSakatOrulen = ToplamSakatOrulen + New.Miktar where OrmePlanlama.OrmePlanlamaId = new.OrmePlanlamaId;

      end
      if (Old.DokumaPlanlamaId is not null) then
      begin
        if (Old.Sakat = 0) then
          update DokumaPlanlama Set GerceklesenMiktar = GerceklesenMiktar - Old.Miktar where DokumaPlanlama.DokumaPlanlamaId = Old.DokumaPlanlamaId;
        else
          update DokumaPlanlama Set GerceklesenMiktar = GerceklesenMiktar - Old.Miktar, ToplamSakatDokunan = ToplamSakatDokunan - Old.Miktar where DokumaPlanlama.DokumaPlanlamaId = Old.DokumaPlanlamaId;

        if (New.Sakat = 0) then
          update DokumaPlanlama Set GerceklesenMiktar = GerceklesenMiktar + New.Miktar where DokumaPlanlama.DokumaPlanlamaId = new.DokumaPlanlamaId;
        else
          update DokumaPlanlama Set GerceklesenMiktar = GerceklesenMiktar + New.Miktar, ToplamSakatDokunan = ToplamSakatDokunan + New.Miktar where DokumaPlanlama.DokumaPlanlamaId = new.OrmePlanlamaId;
      end
      if (new.BukumTalimatId is not Null) then
      begin
          update BukumTalimat Set GerceklesenMiktar = GerceklesenMiktar - old.Miktar where BukumTalimat.BukumTalimatId = old.BukumTalimatId;

          update BukumTalimat Set GerceklesenMiktar = GerceklesenMiktar + New.Miktar where BukumTalimat.BukumTalimatId = new.BukumTalimatId;
      end
    end
  end

  if (New.GirisLinkIslemi = 0) then
  begin
    if (new.GirisCikis = -1) then
    begin
      /*if (old.OrmeRafId = New.OrmeRafId) then

                            BrutCikisMiktari = BrutCikisMiktari - Old.BrutMiktar + New.BrutMiktar,

                            BrutCikisMiktari = BrutCikisMiktari - Old.BrutMiktar,
                            CikisKapAdedi = CikisKapAdedi - Old.KapAdedi
              where OrmeRaf.OrmeRafId = Old.OrmeRafId;

         Update OrmeRaf Set NetCikisMiktari = NetCikisMiktari + New.Miktar,
                            BrutCikisMiktari = BrutCikisMiktari + New.BrutMiktar,
                            CikisKapAdedi = CikisKapAdedi + New.KapAdedi
              where OrmeRaf.OrmeRafId = New.OrmeRafId;
      end*/
      Execute Procedure HesaplanacakRafEkle(New.OrmeRafId);
      Execute Procedure HesaplanacakRafEkle(Old.OrmeRafId);
    end
  end


/* Siparis Kumulatifleri Düzenleniyor */
  if ((IntKarsilastir(Old.OrmeSiparisDetayId, New.OrmeSiparisDetayId) = 0 or Old.Miktar <> New.Miktar or Old.Sakat <> New.Sakat) and Old.BoyaSiparisKayitId is not null) then
  begin
    if ( (new.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and (new.HamIade = 0) ) then
    begin
      if (New.UretimIslemTipiId = uitOrme) then
      begin
        if (Old.Sakat = 0) then
          Update OrmeSiparisDetay Set ToplamOrulen = ToplamOrulen - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
        else
          Update OrmeSiparisDetay Set ToplamOrulen = ToplamOrulen - Old.Miktar, ToplamSakatOrulen = ToplamSakatOrulen - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;

        if (old.OrmeSiparisOlcuId is not null) then
        begin
          if (Old.Sakat = 0) then
            Update OrmeSiparisOlcu Set ToplamOrulen = ToplamOrulen - Old.Miktar where OrmeSiparisOlcu.OrmeSiparisOlcuId = Old.OrmeSiparisOlcuId;
          else
            Update OrmeSiparisOlcu Set ToplamOrulen = ToplamOrulen - Old.Miktar, ToplamSakatOrulen = ToplamSakatOrulen - Old.Miktar where OrmeSiparisOlcu.OrmeSiparisOlcuId = Old.OrmeSiparisOlcuId;
        end
      end
      if (New.UretimIslemTipiId = uitDokuma) then
      begin
        if (Old.Sakat = 0) then
          Update OrmeSiparisDetay Set ToplamDokunan = ToplamDokunan - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
        else
          Update OrmeSiparisDetay Set ToplamDokunan = ToplamDokunan - Old.Miktar, ToplamSakatDokunan = ToplamSakatDokunan - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
      end
    end
    if ( (new.HareketTipi = shtFasonUretimeCikis or New.HareketTipi = shtUretimeCikis) and (new.HamIade = 0) ) then
    begin
      if (New.UretimIslemTipiId = uitBoya) then
      begin
        Update OrmeSiparisDetay Set ToplamBoyayaCikis = ToplamBoyayaCikis - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
      end
    end
    else if (New.HareketTipi = shtSatis and Old.HamIade = 0) then
    begin
      if (Old.BoyaSiparisDetayID is null) then
        Update OrmeSiparisDetay Set ToplamSatis = ToplamSatis - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
      else
        Update OrmeSiparisDetay Set ToplamBoyaliSatis = ToplamBoyaliSatis  - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
    end
    else if (New.HareketTipi = shtSatisIade) then
    begin
      Update OrmeSiparisDetay Set ToplamSatisIade = ToplamSatisIade - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;

  end

  if (new.Miktar <> Old.Miktar) then
  begin
    HarMiktar = Old.Miktar;
    for Select KullanilanMiktar, OrmeSiparisRezervID from OrmeSiparisRezerv where OrmeSiparisDetayID = New.OrmeSiparisDetayID and OrmeRafID = New.OrmeRafID
        into :FRezervMiktar, :FOrmeSiparisRezervID do
    begin
      if (HarMiktar > FRezervMiktar) then
      begin
        Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar - :FRezervMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
        HarMiktar = HarMiktar -FRezervMiktar;
      end
      else
      begin
        Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar - :HarMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
        HarMiktar = 0;
      end
    end

    if (HarMiktar > 0) then
    begin
      for Select KullanilanMiktar, OrmeSiparisRezervID from OrmeSiparisRezerv
          where OrmeSiparisDetayID = New.OrmeSiparisDetayID and HamID = New.HamID and KullanilanMiktar > 0
          Order by OrmeSiparisRezervID
          into :FRezervMiktar, :FOrmeSiparisRezervID do
      begin
        if (HarMiktar > FRezervMiktar)  then
        begin
          Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar - :FRezervMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
          HarMiktar = HarMiktar -FRezervMiktar;
        end
        else
        begin
          Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar - :HarMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
          HarMiktar = 0;
        end
      end
    end

    HarMiktar = New.Miktar;
    for Select (Miktar - KullanilanMiktar), OrmeSiparisRezervID from OrmeSiparisRezerv where OrmeSiparisDetayID = New.OrmeSiparisDetayID and OrmeRafID = New.OrmeRafID
        into :FRezervMiktar, :FOrmeSiparisRezervID do
    begin
      if (HarMiktar > FRezervMiktar) then
      begin
        Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar + :FRezervMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
        HarMiktar = HarMiktar -FRezervMiktar;
      end
      else
      begin
        Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar + :HarMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
        HarMiktar = 0;
      end
    end

    if (HarMiktar > 0) then
    begin
      for Select (Miktar - KullanilanMiktar), OrmeSiparisRezervID from OrmeSiparisRezerv
          where OrmeSiparisDetayID = New.OrmeSiparisDetayID and HamID = New.HamID and Miktar > KullanilanMiktar
          Order by OrmeSiparisRezervID
          into :FRezervMiktar, :FOrmeSiparisRezervID do
      begin
        if (HarMiktar > FRezervMiktar)  then
        begin
          Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar + :FRezervMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
          HarMiktar = HarMiktar -FRezervMiktar;
        end
        else
        begin
          Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar + :HarMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
          HarMiktar = 0;
        end
      end
    end
  end

  if ((IntKarsilastir(Old.OrmeSiparisDetayId, New.OrmeSiparisDetayId) = 0 or Old.Miktar <> New.Miktar or Old.Sakat <> New.Sakat) and New.BoyaSiparisKayitId is not null) then
  begin
    if ( (new.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and (new.HamIade = 0) ) then
    begin
      if (New.UretimIslemTipiId = uitOrme) then
      begin
        if (New.Sakat = 0) then
          Update OrmeSiparisDetay Set ToplamOrulen = ToplamOrulen + New.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayId;
        else
          Update OrmeSiparisDetay Set ToplamOrulen = ToplamOrulen + New.Miktar, ToplamSakatOrulen = ToplamSakatOrulen + New.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayId;

        if (new.OrmeSiparisOlcuId is not null) then
        begin
          if (New.Sakat = 0) then
            Update OrmeSiparisOlcu Set ToplamOrulen = ToplamOrulen + New.Miktar where OrmeSiparisOlcu.OrmeSiparisOlcuId = new.OrmeSiparisOlcuId;
          else
            Update OrmeSiparisOlcu Set ToplamOrulen = ToplamOrulen + New.Miktar, ToplamSakatOrulen = ToplamSakatOrulen + New.Miktar where OrmeSiparisOlcu.OrmeSiparisOlcuId = new.OrmeSiparisOlcuId;
        end
      end
      if (New.UretimIslemTipiId = uitDokuma) then
      begin
        if (New.Sakat = 0) then
          Update OrmeSiparisDetay Set ToplamDokunan = ToplamDokunan + New.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayId;
        else
          Update OrmeSiparisDetay Set ToplamDokunan = ToplamDokunan + New.Miktar, ToplamSakatDokunan = ToplamSakatDokunan + New.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayId;
      end
    end
    if ( (new.HareketTipi = shtFasonUretimeCikis or New.HareketTipi = shtUretimeCikis) and (new.HamIade = 0) ) then
    begin
      if (New.UretimIslemTipiId = uitBoya) then
      begin
        Update OrmeSiparisDetay Set ToplamBoyayaCikis = ToplamBoyayaCikis + New.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayId;
      end
    end
    else if (New.HareketTipi = shtSatis and New.HamIade = 0) then
    begin
      if (Old.BoyaSiparisDetayID is null) then
        Update OrmeSiparisDetay Set ToplamSatis = ToplamSatis + New.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayId;
      else
        Update OrmeSiparisDetay Set ToplamBoyaliSatis = ToplamBoyaliSatis  + New.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = New.OrmeSiparisDetayId;
    end
    else if (New.HareketTipi = shtSatisIade) then
    begin
      Update OrmeSiparisDetay Set ToplamSatisIade = ToplamSatisIade + New.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayId;

  begin
    Update VerilenSiparisDetay Set GelenMiktar = GelenMiktar - Old.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = Old.VerilenSiparisDetayId;

    Update VerilenSiparisDetay Set IadeEdilenMiktar = IadeEdilenMiktar - Old.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = Old.VerilenSiparisDetayId;

  if (new.HareketTipi = shtAlis) then
  begin
    Update VerilenSiparisDetay Set GelenMiktar = GelenMiktar + New.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = new.VerilenSiparisDetayId;
  end
  else if (new.HareketTipi = shtAlisIade) then
  begin
    Update VerilenSiparisDetay Set IadeEdilenMiktar = IadeEdilenMiktar + New.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = new.VerilenSiparisDetayId;

  begin
    select Renk.Adi from Renk where Renk.RenkId = new.RenkId into :RenkAdi;
    if (new.OrmeFisId is not Null) then
    begin
      GirisYeri = 'Örme Fişi';
      select FirmaId, FisNo, FisTarihi from OrmeFis where OrmeFis.OrmeFisId = new.OrmeFisId into :FirmaId, :IrsaliyeNo, :IrsaliyeTarihi;

  if (OrmeAlisindanHamFiyatiGuncelle = 1 and New.HareketTipi = shtAlis and new.RenkID is Null and (new.BirimFiyat <> old.BirimFiyat or New.DovizID <> Old.DovizID)) then
  begin
    update HAM set AlisFiyati = new.BirimFiyat, AlisFiyatiDovizID = New.DovizID where Ham.HamId = new.HamId;
  end
