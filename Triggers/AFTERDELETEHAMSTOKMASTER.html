CREATE TRIGGER AFTERDELETEHAMSTOKMASTER FOR HAMSTOKMASTER
ACTIVE AFTER DELETE POSITION 0
as
begin
  if (Old.Tamir = 1) then Delete From DisTamir where DisTamir.HamStokMasterId = Old.hamStokMasterId;

  if (Old.VerilenSiparisDetayId is not null) then
  begin
    Update VerilenSiparisDetay Set GelenMiktar = GelenMiktar - Old.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = Old.VerilenSiparisDetayId;
  end

  if (Old.StokHareketId is not null) then
    Update StokHareket Set HamStokIrsaliyeId = Null, NeredenUpdate = 3 where StokHareket.StokHareketId = Old.StokHareketId;

  if (Old.IplikIrsaliyeDetayID is not null) then
    Update IplikIrsaliyeDetay Set HamStokIrsaliyeId = Null, NeredenUpdate = 3 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = Old.IplikIrsaliyeDetayId;
end
