Definition

CREATE TRIGGER BEFOREUPDATEBOYATEKLIFDETAY1 FOR BOYATEKLIFDETAY
ACTIVE BEFORE UPDATE POSITION 1
as
declare variable H1 Double Precision;
declare variable AsamaToplami Double Precision;
declare variable IlaveIslemToplami Double Precision;
declare variable OzelTestToplami Double Precision;
declare variable TeklifTabanVadeGun integer;
declare variable FVadeGun integer;
declare variable TeklifFirmaId               varchar(20);
declare variable TeklifDovizId               Varchar(5);
declare variable TeklifTarihi                TimeStamp;
declare variable TeklifFason                 integer;
declare variable TeklifVadeFarkiOrani        Double Precision;
declare variable TeklifDuzeltmeFaktoru       Double Precision;
declare variable TeklifUretimGideri          Double Precision;
declare variable TeklifYonetimGideri         Double Precision;
declare variable TeklifAmortisman            Double Precision;
declare variable FIplikMaliyeti              Double Precision;
declare variable FLycraMaliyeti              Double Precision;
declare variable hstLycra                    integer;
declare variable fFireOrani                  Double Precision;
declare variable fFasonBoyaBirimFiyat        Double Precision;
declare variable fBirimFiyat                 Double Precision;
declare variable fKullanimOrani              Double Precision;
declare variable IslemKaynagi                SmallInt;
declare variable tikBoya                     SmallInt;
declare variable tikHazirGiyim               SmallInt;
declare variable TeklifFiyatiHesaplama       EgeBoolean;
begin
  Select tikBoya, tikHazirGiyim from TeklifIslemKaynagiGetir into :tikBoya, :tikHazirGiyim;

  Select Fason, FirmaId, DovizId, IslemKaynagi, TeklifTarihi, VadeFarkiOrani, DuzeltmeFaktoru, UretimGideri, YonetimGideri, Amortisman from BoyaTeklif where BoyaTeklif.BoyaTeklifKayitID = new.BoyaTeklifKayitID
           into :TeklifFason, :TeklifFirmaId, :TeklifDovizId, :IslemKaynagi, :TeklifTarihi, :TeklifVadeFarkiOrani,
                :TeklifDuzeltmeFaktoru, :TeklifUretimGideri, :TeklifYonetimGideri, :TeklifAmortisman;

  if (IslemKaynagi = tikHazirGiyim) then
  begin
    New.HesaplananFiyat = New.HesaplananFiyat / (( 100 - New.Komisyon )/ 100);
    New.HesaplananFiyat = New.HesaplananFiyat + New.Navlun;
    Select Fiyat from FormataGoreFiyatYuvarla(New.HesaplananFiyat, 'BOYA', 'BIRIMFIYAT') into New.HesaplananFiyat;

  end
  else if (old.LinkIslemi <> 2) then
  begin
    if (new.LinkIslemi = 0) then
    begin

      Select TeklifTabanVadeGun from sistem where sistemid = 1 into :TeklifTabanVadeGun;

      Select Sum(BirimFiyat) From BoyaTeklifAsama where BoyaTeklifAsama.BoyaTeklifDetayId = new.BoyaTeklifDetayId into :AsamaToplami;
      Select Sum(BirimFiyat) From BoyaTeklifIlaveIslem where BoyaTeklifIlaveIslem.BoyaTeklifDetayId = new.BoyaTeklifDetayId into :IlaveIslemToplami;
      Select Sum(BirimFiyat) From BoyaTeklifOzelTest where BoyaTeklifOzelTest.BoyaTeklifDetayId = new.BoyaTeklifDetayId into :OzelTestToplami;

      if (AsamaToplami is null) then AsamaToplami = 0;
      if (IlaveIslemToplami is null) then IlaveIslemToplami = 0;
      if (OzelTestToplami is null) then OzelTestToplami = 0;

      FVadeGun = New.VadeGun - TeklifTabanVadeGun;

      if (FVadeGun < 0) then FVadeGun = 0;

      if (FVadeGun = 0) then H1 = 1.0;
      else H1 = (TeklifVadeFarkiOrani / 30 / 100) * FVadeGun + 1;

      if (new.Fason = 0) then
      begin
        Select hstLycra from HamTipiGetir into :hstLycra;

        fiplikmaliyeti=0;
        for Select BirimFiyat, FireOrani, FasonBoyaBirimFiyat,KullanimOrani from BoyaTeklifHammadde
          left join Ham On Ham.HamID = BoyaTeklifHammadde.HamId
          where BoyaTeklifHammadde.BoyaTeklifDetayId = new.BoyaTeklifDetayId and Ham.HamTipiID <> :hstLycra
        into :fBirimFiyat, :fFireOrani, :fFasonBoyaBirimFiyat, :fKullanimOrani
        do
        begin
          FIplikMaliyeti=FIplikMaliyeti+((((fBirimFiyat/((100-fFireOrani)/100))+fFasonBoyaBirimFiyat)*fKullanimOrani)/100) ;
        end

        if (FIplikMaliyeti is null) then FIplikMaliyeti = 0;
        else FIplikMaliyeti =  FIplikMaliyeti /((100- New.IplikFireOrani) / 100);
        /*else FIplikMaliyeti = FIplikMaliyeti + FIplikMaliyeti * New.IplikFireOrani / 100;*/

        Select Sum(BirimFiyat*KullanimOrani/100) from BoyaTeklifHammadde
          left join Ham On Ham.HamID = BoyaTeklifHammadde.HamId
          where BoyaTeklifHamMadde.BoyaTeklifDetayId = new.BoyaTeklifDetayId and Ham.HamTipiID = :hstLycra
          into :FLycraMaliyeti;

        if (FLycraMaliyeti is null) then FLycraMaliyeti = 0;
        New.HamStokBedeli = FIplikMaliyeti + FLycraMaliyeti + New.OrmeBirimMaliyet;
      end

      if (TeklifFason = 1) then
      begin
        New.HesaplananFiyat = ((new.ReceteMaliyeti * TeklifDuzeltmeFaktoru) + TeklifUretimGideri + TeklifYonetimGideri) *
                              H1  * (1 + New.UretimKarOrani / 100)  +
                              TeklifAmortisman + AsamaToplami + IlaveIslemToplami + OzelTestToplami;
      end
      else
      begin
        New.HesaplananFiyat = New.HamStokBedeli + ((new.ReceteMaliyeti * TeklifDuzeltmeFaktoru) + TeklifUretimGideri + TeklifYonetimGideri) *
                              H1  * (1 + new.UretimKarOrani / 100)  + TeklifAmortisman + AsamaToplami + IlaveIslemToplami + OzelTestToplami;

        New.HesaplananFiyat = New.HesaplananFiyat / (( 100 - New.TeklifFireOrani )/ 100);
        New.HesaplananFiyat = New.HesaplananFiyat + New.FasonBirimFiyat;
        New.HesaplananFiyat = New.HesaplananFiyat / (( 100 - New.FasonFireOrani )/ 100);

        New.HesaplananFiyat = New.HesaplananFiyat * (1 + New.SatisKarOrani / 100);

        New.HesaplananFiyat = New.HesaplananFiyat / (( 100 - New.Komisyon )/ 100);
        New.HesaplananFiyat = New.HesaplananFiyat + New.Navlun;

      end

      Select Fiyat from FormataGoreFiyatYuvarla(New.HesaplananFiyat, 'BOYA', 'BIRIMFIYAT') into New.HesaplananFiyat;

      if (new.KaliteId is not null) then
      begin
        select Kalite.TeklifFiyatiHesaplama from Kalite where Kalite.KaliteId = new.KaliteId into :TeklifFiyatiHesaplama;
        if (TeklifFiyatiHesaplama = 1) then
          new.HesaplananFiyat = 0;
      end
    end

    new.LinkIslemi = 0;
  end
end
