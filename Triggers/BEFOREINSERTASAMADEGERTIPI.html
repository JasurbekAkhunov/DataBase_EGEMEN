CREATE TRIGGER BEFOREINSERTASAMADEGERTIPI FOR ASAMADEGERTIPI
ACTIVE BEFORE INSERT POSITION 0
as
declare variable RdId integer;
declare variable ASira integer;
begin
  if (new.Sira > -1) then
  begin
    ASira = new.Sira;
    for select AsamaDegerTipiId from AsamaDegerTipi where AsamaId = new.AsamaId and Sira >= new.Sira order by Sira into :RdId do
    begin
      ASira = ASira + 1;
      update AsamaDegerTipi set Sira = :ASira where AsamaDegerTipiId = :RdId;
    end
  end
  else
  begin
    select max(sira) from AsamaDegerTipi where AsamaId = new.AsamaId into :ASira;
    if (ASira is null) then ASira = 0;
    else ASira = ASira + 1;
    new.Sira = ASira;
  end
  new.AsamaDegerTipiId = gen_id(AsamaDegerTipi_Id_Gen, 1);
end
