CREATE TRIGGER BEFOREINSERTBOYASIPARISORNEK FOR BOYASIPARISORNEKPARTI
ACTIVE BEFORE INSERT POSITION 0
as
declare variable KayitSayisi integer;
begin
  Select Count(*) from BoyaSiparisOrnekParti
     where BoyaSiparisOrnekParti.BoyaSiparisDetayId = new.BoyaSiparisDetayId and BoyaSiparisOrnekParti.PartiSinifId = new.PartiSinifId
     into :KayitSayisi;

  if (KayitSayisi > 0) then exception PartiDahaOnceOrnekPartiYapilmis;

  Select Count(*) from BoyaSiparisOrnekParti
     where BoyaSiparisOrnekParti.BoyaSiparisDetayId = new.BoyaSiparisDetayId and BoyaSiparisOrnekParti.OnayDurumu <> 2
     into :KayitSayisi;

  if (KayitSayisi > 0) then exception SiparisinOrnekPartisiVar;

  Update BoyaSiparisDetay set OrnekOnay = 0, OrnekPartiAsamalariId = new.PartiAsamalariId, OrnekPartiSinifId = new.PartiSinifId, LinkIslemi = 1
       where BoyaSiparisDetay.BoyaSiparisDetayId = new.BoyaSiparisDetayId;

  new.BoyaSiparisOrnekPartiID = gen_ID(BoyaSiparisOrnekParti_Id_Gen, 1);
end
