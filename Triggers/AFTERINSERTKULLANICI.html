CREATE TRIGGER AFTERINSERTKULLANICI FOR KULLANICI
ACTIVE AFTER INSERT POSITION 0
as
declare variable ProgramModulId integer;
declare variable KullaniciyaOtomatikYetkiVer integer;
declare variable Yetki integer;
begin
  Select KullaniciyaOtomatikYetkiVer from Sistem where SistemId = 1 into :KullaniciyaOtomatikYetkiVer;
  if (KullaniciyaOtomatikYetkiVer = 1) then Yetki = 1;
  else Yetki = 0;

  if (New.FirmaId is null) then
    For Select ProgramModulId from programModul into :ProgramModulId do
       insert into KullaniciModul(KullaniciId, ProgramModulId, Girebilir, YeniKayit, Duzeltme, Silme) values(new.KullaniciId, :ProgramModulId, :Yetki, :Yetki, :Yetki, :Yetki);

  insert into Kullanicisekme(KullaniciId, PageIndex, Adi) Values (new.KullaniciId, 0, 'Kişisel Menü');
end
