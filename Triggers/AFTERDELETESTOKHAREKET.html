Definition

CREATE TRIGGER AFTERDELETESTOKHAREKET FOR STOKHAREKET
ACTIVE AFTER DELETE POSITION 0
as
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;

declare variable uitBukum integer;
declare variable uitOrme integer;
declare variable uitBoya integer;
declare variable uitDokuma integer;
declare variable FRezervMiktar Double Precision;
declare variable HarMiktar Double Precision;
declare variable FOrmeSiparisRezervID Double Precision;

begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
   into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
        :shtFasonUretim, :ShtUretim;

  Select uitBukum, uitOrme, uitBoya, uitDokuma from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya, :uitDokuma;
  if (old.OrmeFaturaId is not null) then update OrmeFatura set DegisimTarihi = current_TimeStamp(0) where OrmeFatura.OrmeFaturaId = old.OrmeFaturaId; /* Fatura Toplamını Yeniden Hesaplaması için */

/* Siparis Kumulatifleri Düzenleniyor */
  if (Old.BoyaSiparisKayitId is not null) then
  begin
    if ( (Old.HareketTipi = shtFasonUretim or Old.HareketTipi = shtUretim) and (Old.HamIade = 0) ) then
    begin
      if (Old.UretimIslemTipiId = uitOrme) then
      begin
        if (Old.Sakat = 0) then
          Update OrmeSiparisDetay Set ToplamOrulen = ToplamOrulen - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
        else
          Update OrmeSiparisDetay Set ToplamOrulen = ToplamOrulen - Old.Miktar, ToplamSakatOrulen = ToplamSakatOrulen - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;

        if (old.OrmeSiparisOlcuId is not null) then
        begin
          if (Old.Sakat = 0) then
            Update OrmeSiparisOlcu Set ToplamOrulen = ToplamOrulen - Old.Miktar where OrmeSiparisOlcu.OrmeSiparisOlcuId = Old.OrmeSiparisOlcuId;
          else
            Update OrmeSiparisOlcu Set ToplamOrulen = ToplamOrulen - Old.Miktar, ToplamSakatOrulen = ToplamSakatOrulen - Old.Miktar where OrmeSiparisOlcu.OrmeSiparisOlcuId = Old.OrmeSiparisOlcuId;
        end
      end
      if (Old.UretimIslemTipiId = uitDokuma) then
      begin
        if (Old.Sakat = 0) then
          Update OrmeSiparisDetay Set ToplamDokunan = ToplamDokunan - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
        else
          Update OrmeSiparisDetay Set ToplamDokunan = ToplamDokunan - Old.Miktar, ToplamSakatDokunan = ToplamSakatDokunan - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
      end
    end
    if ( (Old.HareketTipi = shtFasonUretimeCikis or Old.HareketTipi = shtUretimeCikis) and (Old.HamIade = 0) ) then
    begin
      if (Old.UretimIslemTipiId = uitBoya) then
      begin
        Update OrmeSiparisDetay Set ToplamBoyayaCikis = ToplamBoyayaCikis - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
      end
    end
    else if (Old.HareketTipi = shtSatis and Old.HamIade = 0) then
    begin
      if (Old.BoyaSiparisDetayID is null) then
        Update OrmeSiparisDetay Set ToplamSatis = ToplamSatis - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
      else
        Update OrmeSiparisDetay Set ToplamBoyaliSatis = ToplamBoyaliSatis - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
    end
    else if (Old.HareketTipi = shtSatisIade) then
    begin
      Update OrmeSiparisDetay Set ToplamSatisIade = ToplamSatisIade - Old.Miktar, LinkIslemi = 1 where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;
    end
  end

  if (Old.VerilenSiparisDetayId is not null) then
  if (Old.HareketTipi = shtAlis) then
  begin
    Update VerilenSiparisDetay Set GelenMiktar = GelenMiktar - Old.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = Old.VerilenSiparisDetayId;
  end
  else if (Old.HareketTipi = shtAlisIade) then
  begin
    Update VerilenSiparisDetay Set IadeEdilenMiktar = IadeEdilenMiktar - Old.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = Old.VerilenSiparisDetayId;
  end

  if ( (Old.HareketTipi = shtFasonUretimeCikis or Old.HareketTipi = shtUretimeCikis) and (Old.HamIade = 0) and (Old.UretimIslemTipiId = uitOrme)) then
  begin
    HarMiktar = Old.Miktar;
    for Select KullanilanMiktar, OrmeSiparisRezervID from OrmeSiparisRezerv where OrmeSiparisDetayID = Old.OrmeSiparisDetayID and OrmeRafID = Old.OrmeRafID
        into :FRezervMiktar, :FOrmeSiparisRezervID do
    begin
      if (HarMiktar > FRezervMiktar) then
      begin
        Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar - :FRezervMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
        HarMiktar = HarMiktar -FRezervMiktar;
      end
      else
      begin
        Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar - :HarMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
        HarMiktar = 0;
      end
    end

    if (HarMiktar > 0) then
    begin
      for Select KullanilanMiktar, OrmeSiparisRezervID from OrmeSiparisRezerv
          where OrmeSiparisDetayID = Old.OrmeSiparisDetayID and HamID = Old.HamID and KullanilanMiktar > 0
          Order by OrmeSiparisRezervID
          into :FRezervMiktar, :FOrmeSiparisRezervID do
      begin
        if (HarMiktar > FRezervMiktar)  then
        begin
          Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar - :FRezervMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
          HarMiktar = HarMiktar -FRezervMiktar;
        end
        else
        begin
          Update OrmeSiparisRezerv set KullanilanMiktar = KullanilanMiktar - :HarMiktar where OrmeSiparisRezervID = :FOrmeSiparisRezervID;
          HarMiktar = 0;
        end
      end
    end
  end

  if (Old.BoyaIrsaliyeDetayId is not null) then
    Update BoyaIrsaliyeDetay Set OrmeIrsaliyeID = Null where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = Old.BoyaIrsaliyeDetayId;

  if (Old.IplikIrsaliyeDetayId is not null) then
    Update IplikIrsaliyeDetay Set OrmeIrsaliyeID = Null, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = Old.IplikIrsaliyeDetayId;

  if (Old.GirisCikis = -1 or Old.GirisCikis = 0) then
    Execute procedure HesaplanacakRafEkle(Old.OrmeRafId);
end
