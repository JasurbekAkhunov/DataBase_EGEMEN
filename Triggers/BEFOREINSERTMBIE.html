CREATE TRIGGER BEFOREINSERTMBIE FOR MAKINABAKIMISEMRI
ACTIVE BEFORE INSERT POSITION 0
as
declare variable BIEBitmeyenMakinaTekrarlayamaz integer;
declare variable KayitSayisi integer;
declare variable EgemenFirmaID integer;
declare variable BakimProjesiAktif integer;
begin
  if (new.MakinaId is Null and new.OrmeMakinaId is Null and new.IplikMakinaId is Null and new.DokumaMakinaid is Null) then exception KayitHatasi 'Is emri icin bir makina secmelisiniz...';
  if (new.MakinaId is not Null and (new.OrmeMakinaId is not Null or new.IplikMakinaId is not Null or new.DokumaMakinaId is not Null)) then exception KayitHatasi 'Is emri icin sadece bir makina secmelisiniz...';
  if (new.OrmeMakinaId is not Null and (new.MakinaId is not Null or new.IplikMakinaId is not Null or new.DokumaMakinaId is not Null)) then exception KayitHatasi 'Is emri icin sadece bir makina secmelisiniz...';
  if (new.IplikMakinaId is not Null and (new.MakinaId is not Null or new.OrmeMakinaId is not Null or new.DokumaMakinaId is not Null)) then exception KayitHatasi 'Is emri icin sadece bir makina secmelisiniz...';
  if (new.DokumaMakinaId is not Null and (new.MakinaId is not Null or new.OrmeMakinaId is not Null or new.IplikMakinaId is not Null)) then exception KayitHatasi 'Is emri icin sadece bir makina secmelisiniz...';

  if (new.IEStartKullaniciId is Null and new.BakimStartKullaniciId is not Null) then exception KayitHatasi ' Is emri baslatilmadan bakim baslatilamaz...';
  if (new.BakimStartKullaniciId is Null and new.BakimEndKullaniciId is not Null) then exception KayitHatasi ' Bakim baslatilmadan bitirilemez...';
  if (new.IEStartKullaniciId is Null and new.IEEndKullaniciId is not Null) then exception KayitHatasi ' Is emri baslatilmadan bitirilemez...';

  Select BIEBitmeyenMakinaTekrarlayamaz, EgemenFirmaID, BakimProjesiAktif from Sistem where Sistemid = 1 into :BIEBitmeyenMakinaTekrarlayamaz, :EgemenFirmaID, :BakimProjesiAktif;

  if (BIEBitmeyenMakinaTekrarlayamaz = 1) then
  begin
    Select Count(*) from MakinaBakimIsEmri
    where Kapali = 0 and (IEStartKullaniciID is not Null and cast(PlanlananBaslangic as Date) = current_date and IEEndKullaniciID is null) and
    (
      (new.MakinaId is not Null and MakinaID = new.MakinaID) or
      (new.OrmeMakinaId is not Null and OrmeMakinaId = new.OrmeMakinaId) or
      (new.IplikMakinaId is not Null and IplikMakinaId = new.IplikMakinaId) or
      (new.DokumaMakinaId is not Null and DokumaMakinaId = new.DokumaMakinaId)
    )
    into :KayitSayisi;

    if (KayitSayisi > 0) then exception EBIEBitmeyenMakinaTekrarlayamaz;
  end

  if (new.TahminiBaslangic is not Null) then new.PlanlananBaslangic = new.TahminiBaslangic;
  if (new.InsertTarihi is Null) then new.InsertTarihi = current_timestamp(0);


  if (EgemenFirmaID = 44 and BakimProjesiAktif = 1 and Current_Date > '01.12.2013') then /*Dilan Tekstilde 01.12.2013 ten sonra Bakım kapatılacak*/
  begin
    Update Sistem set BakimProjesiAktif = 0;
  end

end
