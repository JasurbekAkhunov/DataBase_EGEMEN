CREATE TRIGGER AFTERINSERTMODELBEDEN FOR MODELBEDEN
ACTIVE AFTER INSERT POSITION 0
as
begin
  Insert Into ModelAksesuarDetay(ModelAksesuarID, ModelBedenID, KullanilacakBedenID, InsertKullaniciID, KullaniciID)
  Select ModelAksesuarID, New.ModelBedenID, New.BedenID, new.InsertKullaniciID, new.KullaniciID from ModelAksesuar where ModelAksesuar.ModelKayitID = new.ModelKayitID and (KullanimTipi = 3 or KullanimTipi = 4);

  Execute Procedure ModelUrunleriniOrganizeEt(New.ModelKayitID, New.KullaniciID);
end
