CREATE TRIGGER BEFOREINSERTBOYAFATURADETAY FOR BOYAFATURADETAY
ACTIVE BEFORE INSERT POSITION 0
as
declare variable Entegrasyon integer;
declare variable FNetFiyatYuvarlamaCarpani Double Precision;
declare variable FSinif varchar(1);
declare variable FaturaSinif varchar(1);
declare variable BIDHareketTipiId       integer;
declare variable FPartiSinifId          integer;
declare variable FHamId                 integer;
declare variable FaturaFasonSatisKarisik integer;
declare variable FaturadaPartininVadesiGecerli integer;
declare variable Fason                  integer;
declare variable IslemGrubuId           integer;
declare variable KayitSayisi            integer;
declare variable SiparisProsesAdi       varchar(255);
declare variable PartivadeGun           integer;
declare variable FHamTipiID             integer;
declare variable FDovizId               varchar(5);
declare variable Tevkifat              integer;
declare variable FaturadaSipDovizCinsiKarisik integer;
declare variable ImalatTipiId          integer;
declare variable SonDonemBitisTarihi TimeStamp;
declare variable IrsFarkliDovizBirlestiremez integer;
declare variable PartiDovizID Varchar(5);
declare variable FEgemenFirmaID Integer;
declare variable IrsTarihFatTarihtenBuyukOlamaz integer;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable KdvHesaplanmayacakSeriNo  varchar(5);
declare variable FaturaSeriNo  SeriNoSize5;
declare variable OrmeFirmaKodumuz FirmaKoduSize;
declare variable FaturaFirmaId FirmaKoduSize;
begin
  if (Entegrasyon = 1) then exception FaturaHazirDegistiremezsiniz;

  select Tevkifat, Entegrasyon, Sinif, HareketTipiId, NetFiyatYuvarlamaCarpani, FaturaSeriNo, FirmaId from BoyaFatura where BoyaFatura.BoyaFaturaId = New.BoyaFaturaId
     into :Tevkifat, :Entegrasyon, :FaturaSinif, new.HareketTipiId, :FNetFiyatYuvarlamaCarpani, :FaturaSeriNo, :FaturaFirmaId;

  Select BoyaIrsaliyeDetay.HareketTipiId, BoyaIrsaliye.Sinif from BoyaIrsaliyeDetay
   left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = new.BoyaIrsaliyeDetayId
    into :BIDHareketTipiId, :FSinif;

  if (BIDHareketTipiId <> New.hareketTipiId) then exception StokHareketTipiUymuyor;
  if (FaturaSinif <> FSinif) then exception FaturaIlePartiSeriNoTutmuyor;

  Select FaturadaPartininVadesiGecerli, FaturaFasonSatisKarisik, FaturadaSipDovizCinsiKarisik, IrsFarkliDovizBirlestiremez, EgemenFirmaID, IrsTarihFatTarihtenBuyukOlamaz, KdvHesaplanmayacakSeriNo, Coalesce(OrmeFirmaKodumuz, '')
  from Sistem where SistemId = 1
  into :FaturadaPartininVadesiGecerli, :FaturaFasonSatisKarisik, :FaturadaSipDovizCinsiKarisik, :IrsFarkliDovizBirlestiremez, :FEgemenFirmaID, :IrsTarihFatTarihtenBuyukOlamaz, :KdvHesaplanmayacakSeriNo, :OrmeFirmaKodumuz;

  if (IrsTarihFatTarihtenBuyukOlamaz = 1) then
  begin
    for Select BoyaIrsaliye.IrsaliyeTarihi from BoyaIrsaliyeDetay
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      where  BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID =  New.BoyaIrsaliyeDetayID
    into :FIrsaliyeTarihi do
    begin
      if (FIrsaliyeTarihi > New.FaturaTarihi) then exception IrsTarFatTardanBuyukOlamaz;
    end
  end

  if (FaturaFasonSatisKarisik = 0) then
  begin
    Fason = Null;

    Select Parti.Fason from BoyaIrsaliyeDetay
      left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
      where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = New.BoyaIrsaliyeDetayId
      into :Fason;

    if (Fason is null) then Fason = 0;

    Select Count(*) from BoyaFaturaDetay
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
      left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
      where BoyaFaturaDetay.BoyaFaturaId = New.BoyaFaturaId and BoyaIrsaliyeDetay.PartiSinifId is not null and Parti.Fason <> :Fason
      into :KayitSayisi;

    if (KayitSayisi > 0) then exception FaturadaFasonSatisKarisik;
  end

  select Parti.Fason, Parti.IslemGrubuId, BoyaIrsaliyeDetay.PartiSinifId, BoyaIrsaliyeDetay.HamId, Ham.HamTipiId,
         SiparisProses.Adi, Parti.VadeGun, BoyaSiparisDetay.VadeGun, BoyaSiparisDetay.DovizId, Parti.ImalatTipiId, PartiStok.DovizID from BoyaIrsaliyeDetay
    left join Parti On Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
    left join PartiStok On PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Ham On Ham.HamId = PartiStok.HamID
    left join SiparisProses On SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = new.BoyaIrsaliyeDetayId
    into :Fason, :IslemGrubuId, :FPartiSinifId, :FHamId, :FHamTipiId, :SiparisProsesAdi, :PartiVadeGun, new.VadeGun, :FDovizId, :ImalatTipiId, PartiDovizID;

  if (FaturadaSipDovizCinsiKarisik = 0 and FDovizId is not null) then
  begin
    Select Count(*) from BoyaFaturaDetay
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
      left join PartiStok On PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      where BoyaFaturaDetay.BoyaFaturaId = New.BoyaFaturaId and BoyaIrsaliyeDetay.PartiSinifId is not null and BoyaSiparisDetay.DovizId <> :FDovizId
      into :KayitSayisi;

    if (KayitSayisi > 0) then exception FaturadaDovizCinsiKarisik;
  end

  if (IrsFarkliDovizBirlestiremez = 1 and PartiDovizID is not null) then
  begin
    Select Count(*) from BoyaFaturaDetay
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
      left join PartiStok On PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      where BoyaFaturaDetay.BoyaFaturaID = new.BoyaFaturaId and PartiStok.DovizID <> :PartiDovizID
      into :KayitSayisi;

    if (KayitSayisi > 0) then exception IrsFarkliDovizBirlestiremez;
  end

  if (FaturadaPartininVadesiGecerli = 1 or New.PartiPaketId is not null) then new.VadeGun = PartiVadeGun;

  if (new.VadeGun is null) then new.VadeGun = 0;

  if (FPartiSinifId is not null) then
  begin
    new.BirimFiyat = new.SiparisBirimFiyat / new.DovizKuru1 * new.DovizKuru2;

    if (Fason = 0 and OrmeFirmaKodumuz <> FaturaFirmaId) then Select KdvOrani from Sistem where sistemid = 1 into new.KdvOrani;
    else Select FasonKdvOrani from Sistem where sistemid = 1 into new.KdvOrani;

    if (ImalatTipiId = 0) then
    begin
      if (Fason = 1) then
      begin
        if (FHamTipiID = 0) then
        begin
          if (IslemGrubuId = 0 or IslemGrubuId = 3) then New.Aciklama = 'İPLİK PEROKSIT KASAR İŞÇİLİĞİ';
          else New.Aciklama = 'İPLİK BOYAMA İŞÇİLİĞİ';
        end
        else
        begin
          if (IslemGrubuId = 2) then New.Aciklama = 'YIKAMA İŞÇİLİĞİ';
          else if (IslemGrubuId = 0) then New.Aciklama = 'PEROKSIT KASAR İŞÇİLİĞİ';
          else if (IslemGrubuId = 4) then New.Aciklama = SiparisProsesAdi;
          else if (IslemGrubuId = 7) then New.Aciklama = 'BASKI İŞÇİLİĞİ';
          else New.Aciklama = 'BOYAMA HİZMETİ';
        end
      end
      else
      begin
        if (FHamTipiID = 0) then
        begin
          New.Aciklama = 'BOYALI İPLİK';
        end
        else
        begin
          if (IslemGrubuID = 2) then New.Aciklama = 'YIKANMIŞ KUMAŞ';
          else if (IslemGrubuID = 0 or IslemGrubuID = 3) then New.Aciklama = 'KASARLI KUMAŞ';
          else New.Aciklama = 'BOYALI KUMAŞ';
        end
      end
    end
    else if (ImalatTipiId = 1) then
    begin
      if (FEgemenFirmaID = 34) then /* Örkum*/
      begin
        Select Count(*) from BoyaIrsaliyeDetay
          left join PartiAsamalari on PartiAsamalari.PartiKayitID = BoyaIrsaliyeDetay.PartiKayitID
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
          where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = new.BoyaIrsaliyeDetayID and Asama.BaskiAsamasimi = 1
        into :KayitSayisi;
        if (KayitSayisi > 0) then
        begin
          if (Fason = 1) then New.Aciklama = 'BASKI HİZMETİ';
          else New.Aciklama = 'BASKILI KUMAŞ';
        end
        else
        begin
          if (Fason = 1) then New.Aciklama = 'BOYAMA HİZMETİ';
          else New.Aciklama = 'BOYALI KUMAŞ';
        end
      end
      else
      begin
        if (Fason = 1) then New.Aciklama = 'BASKI HİZMETİ';
        else New.Aciklama = 'BASKILI KUMAŞ';
      end
    end
    else new.Aciklama = '';
  end
  else Select Ham.KdvOrani from Ham where Ham.HamID = :FHamId into new.KdvOrani;

  if (FaturaSinif = 'A') then new.KdvOrani = 0;

  if (KdvHesaplanmayacakSeriNo <> '' and FaturaSeriNo = KdvHesaplanmayacakSeriNo) then new.KdvOrani  = 0;


  new.BirimFiyat = Yuvarla(new.BirimFiyat, :FNetFiyatYuvarlamaCarpani);

  select max(BitisTarihi) from DonemKapatma where Bolum = 1 and OnayKullaniciId is not Null into :SonDonemBitisTarihi;
  if (SonDonemBitisTarihi is not Null and SonDonemBitisTarihi > new.FaturaTarihi) then
    exception DonemKapaliHEklenemez;

  New.BoyaFaturaDetayID = gen_ID(BoyaFaturaDetay_Id_Gen, 1);
end
