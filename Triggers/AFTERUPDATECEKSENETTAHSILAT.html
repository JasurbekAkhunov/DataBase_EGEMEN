Definition

CREATE TRIGGER AFTERUPDATECEKSENETTAHSILAT FOR CEKSENETTAHSILAT
ACTIVE AFTER UPDATE POSITION 0
as
begin
  if (New.Meblag <> Old.Meblag) then
    update CekSenetDetay Set TahsilEdilenMeblag = (Select Sum(CekSenetTahsilat.Meblag) from CekSenettahsilat where CekSenettahsilat.GenelKayitId = New.GenelKayitId) where CekSenetDetay.GenelKayitId = New.GenelKayitId;

 if (New.KasaId is not null) then
   update KasaHareket set tarih = new.Tarih, Meblag = new.Meblag, KullaniciId = new.KullaniciId, LinkIslemi = 1 where CsReferansId = new.CekSenetTahsilatId;
 else
   update BankaHesapCari set Tarih = new.Tarih , Meblag = new.meblag,KullaniciId = new.KullaniciId,  LinkIslemi = 1  where CsReferansId = new.CekSenetTahsilatId;
end
