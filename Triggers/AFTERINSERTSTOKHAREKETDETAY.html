CREATE TRIGGER AFTERINSERTSTOKHAREKETDETAY FOR STOKHAREKETDETAY
ACTIVE AFTER INSERT POSITION 0
as
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim    integer;
declare variable shtKaydirmaCikis  integer;
declare variable shtKaydirmaGiris  integer;
declare variable shtReklamasyonGiris   integer;
declare variable shtReklamasyonCikis   integer;
declare variable StokHareketID         integer;
declare variable IplikStokNumarator    integer;
declare variable Kilo                  Double Precision;
declare variable BrutKilo              Double Precision;
declare variable StokHareketDetayMiktarID integer;
declare variable IK     varchar(2);
declare variable TF     integer;
declare variable PaketId integer;

begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, shtKaydirmaCikis, shtKaydirmaGiris, shtReklamasyonGiris, shtReklamasyonCikis From OrmeStokHareketTipiGetir
    into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
         :shtFasonUretim, :ShtUretim, :shtKaydirmaCikis, :shtKaydirmaGiris, :shtReklamasyonGiris, :shtReklamasyonCikis;


  if (new.HareketTipi = shtAlis or new.HareketTipi = shtSatisIade or new.HareketTipi = shtFasonAlis or new.HareketTipi = shtFasonUretim or
      new.HareketTipi = shtUretim or new.HareketTipi = shtKaydirmaGiris or new.HareketTipi = shtReklamasyonGiris ) then
  begin /*Girişler*/
    Select StokHareketDetayMiktarID from StokHareketDetayMiktar where StokHareketDetayMiktar.Barkod = New.Barkod into :StokHareketDetayMiktarID;

    if (StokHareketDetayMiktarID is not Null) then
      Update StokHareketDetayMiktar set GirisKilo = GirisKilo + New.Kilo, GirisBrutKilo = GirisBrutKilo + New.BrutKilo, GirisMetre = GirisMetre + New.Metre, GirisKapAdedi = GirisKapAdedi + New.KapAdedi where StokHareketDetayMiktar.Barkod = New.Barkod;
    else
      insert into StokHareketDetayMiktar(Barkod, GirisKilo, GirisBrutKilo, GirisMetre, GirisKapAdedi) Values (New.Barkod, New.Kilo, New.BrutKilo, New.Metre, New.KapAdedi);
  end
  else /*Çıkışlar*/
  begin
    Select StokHareketDetayMiktarID from StokHareketDetayMiktar where StokHareketDetayMiktar.Barkod = New.Barkod into :StokHareketDetayMiktarID;

    if (StokHareketDetayMiktarID is not Null) then
      Update StokHareketDetayMiktar set CikisKilo = CikisKilo + New.Kilo, CikisBrutKilo = CikisBrutKilo + New.BrutKilo, CikisMetre = CikisMetre + New.Metre, CikisKapAdedi = CikisKapAdedi + New.KapAdedi where StokHareketDetayMiktar.Barkod = New.Barkod;
    else
    begin
      IK = EgemenCopyStr(New.Barkod, 1, 2);

      if (IK = 'K-') then
      begin
        TF = EgemenStrToIntOlurmu(EgemenCopyStr(New.Barkod, 3, Char_Length(New.Barkod) - 2));
        if (TF = 0) then exception BarkodHatali1;
        PaketId = EgemenStrToInt(EgemenCopyStr(New.Barkod, 3, Char_Length(New.Barkod) - 2));

        Select first 1 StokHareket.StokHareketID, PSKaliteKontrol.Kilo, PSKaliteKontrol.Kilo + PSKaliteKontrol.Dara  from PsKaliteKontrol
        left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = PsKaliteKontrol.PartiStokTartiId
        Left Join StokHareket on StokHareket.BoyaIrsaliyeDetayID = PartiStokTarti.BoyaIrsaliyeDetayID
        where PsKaliteKontrol.PsKaliteKontrolId = :PaketId
        into :StokHareketID, :Kilo, :BrutKilo;

        insert into StokHareketDetay(StokHareketID, HareketTipi, PsKaliteKontrolId, Kilo, BrutKilo, InsertKullaniciId, KullaniciId) Values
                                    (:StokHareketID, :shtAlis, :PaketId, :Kilo, :BrutKilo, New.KullaniciId, New.KullaniciId);

      end
      else
      if (IK = 'I-') then
      begin
        TF = EgemenStrToIntOlurmu(EgemenCopyStr(New.Barkod, 3, Char_Length(New.Barkod) - 2));
        if (TF = 0) then exception BarkodHatali1;
        PaketId = EgemenStrToInt(EgemenCopyStr(New.Barkod, 3, Char_Length(New.Barkod) - 2));

        Select first 1 StokHareket.StokHareketID, IplikPaketleme.NetKilo, IplikPaketleme.BrutKilo from IplikPaketleme
        left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiId
        Left Join StokHareket on StokHareket.BoyaIrsaliyeDetayID = PartiStokTarti.BoyaIrsaliyeDetayID
        where IplikPaketleme.IplikPaketlemeID = :PaketId
        into :StokHareketID, :Kilo, :BrutKilo;

        if (StokHareketID is not Null) then
        begin
          insert into StokHareketDetay(StokHareketID, HareketTipi, IplikPaketlemeID, Kilo, BrutKilo, InsertKullaniciId, KullaniciId) Values
                                      (:StokHareketID, :shtAlis, :PaketId, :Kilo, :BrutKilo, New.KullaniciId, New.KullaniciId);
        end
      end
      else if (IK <> 'G-') then
      begin
        Select first 1 StokHareket.StokHareketID, IplikStok.Numarator, IplikStok.NetMiktar, IplikStok.Miktar from IplikStok
        Left Join StokHareket on StokHareket.IplikIrsaliyeDetayID = IplikStok.CikisIplikIrsaliyeDetayID
        where IplikStok.IplikStokID = New.Barkod and StokHareket.StokHareketID is not Null
        Order by IplikStok.Numarator Desc
        into :StokHareketID, :IplikStokNumarator, :Kilo, :BrutKilo;
        if (StokHareketID is not Null) then
        begin
          insert into StokHareketDetay(StokHareketID, HareketTipi, IplikStokID, IplikStokNumarator, Kilo, BrutKilo, InsertKullaniciId, KullaniciId) Values
                                      (:StokHareketID, :shtAlis, New.Barkod, :IplikStokNumarator, :Kilo, :BrutKilo, New.KullaniciId, New.KullaniciId);

        end
      end


      if (StokHareketID is not Null) then
      begin
        Select StokHareketDetayMiktarID from StokHareketDetayMiktar where StokHareketDetayMiktar.Barkod = New.Barkod into :StokHareketDetayMiktarID;

        if (StokHareketDetayMiktarID is not Null) then
          Update StokHareketDetayMiktar set CikisKilo = CikisKilo + New.Kilo, CikisBrutKilo = CikisBrutKilo + New.BrutKilo, CikisMetre = CikisMetre + New.Metre , CikisKapAdedi = CikisKapAdedi + New.KapAdedi where StokHareketDetayMiktar.Barkod = New.Barkod;
        else exception KayitHatasi 'Çıkışı Yapılmak İstenen Barkod un Girişi Bulunamadı';

      end
    end
  end
end
