CREATE TRIGGER BEFOREDELETEDESENISTEKLERI FOR DESENISTEKLERI
ACTIVE BEFORE DELETE POSITION 0
as
begin
  if (Old.BaslamaTarihi is not null or Old.BaslatanKullaniciID is not null) then Exception KayitHatasi 'Hareket Görmüş Silinemez..';

  delete from DesenIstekBekleme where DesenIstekBekleme.DesenIstekleriID = Old.DesenIstekleriID;

end
