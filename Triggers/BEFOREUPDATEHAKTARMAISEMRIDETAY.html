CREATE TRIGGER BEFOREUPDATEHAKTARMAISEMRIDETAY FOR HAKTARMAISEMRIDETAY
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable FHamStokMasterId        integer;
begin
  if ((Old.BobinGramaji is null and New.BobinGramaji is not null) or
      (Old.BobinGramaji is not null and New.BobinGramaji is null) or
      (Old.BobinGramaji <> new.BobinGramaji) ) then
  begin
    Select HamStokDetay.HamStokMasterId from HAktarmaFisiDetay
     left join HamStokDetay on HamStokDetay.HamStokDetayId = HAktarmaFisiDetay.HamStokDetayId
     where HAktarmaFisiDetay.HAktarmaFisiDetayId = new.HAktarmaFisiDetayId
     into :FHamStokMasterId;

     update HamStokMaster set BobinGramaji = New.BobinGramaji, NeredenUpdate = 3 where HamStokMaster.HamStokMasterId = :FHamStokMasterID;
  end

  new.DegisimTarihi = current_TimeStamp(0);
end
