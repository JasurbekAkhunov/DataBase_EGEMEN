CREATE TRIGGER BEFOREDELETEOTOMATIKSARFFISI FOR OTOMATIKSARFFISI
ACTIVE BEFORE DELETE POSITION 0
as
declare variable HamStokSarfFisiId integer;
declare variable HamStokSarfFisiDetayID integer;
declare variable HamStokIrsaliyeId integer;
declare variable HamStokMasterID integer;
declare variable KS integer;
begin
  for Select HamStokMasterId, HamStokIrsaliyeId from HamStokMaster where HamStokMaster.OtomatikSarfFisiId = Old.OtomatikSarfFisiId into :HamStokMasterId, :HamStokIrsaliyeId do
  begin
    KS = 0;
    update HamStokMaster set OtomatikSarfFisiId = Null, NeredenUpdate = 1
      where HamStokMaster.HamStokMasterId = :HamStokMasterId;
    delete from HamStokMaster where HamStokMaster.HamStokMasterId = :HamStokMasterId;
    select count(*) from HamStokMaster where HamStokIrsaliyeId = :HamStokIrsaliyeId into :KS;
    if (KS = 0) then delete from HamStokIrsaliye where HamStokIrsaliyeId = :HamStokIrsaliyeId;
  end

  for Select HamStokSarfFisiDetayId, HamStokSarfFisiId from HamStokSarfFisiDetay where HamStokSarfFisiDetay.OtomatikSarfFisiId = Old.OtomatikSarfFisiId into :HamStokSarfFisiDetayId, :HamStokSarfFisiId do
  begin
    KS = 0;
    update HamStokSarfFisiDetay set OtomatikSarfFisiId = Null
      where HamStokSarfFisiDetay.HamStokSarfFisiDetayId = :HamStokSarfFisiDetayId;
    delete from HamStokSarfFisiDetay where HamStokSarfFisiDetay.HamStokSarfFisiDetayId = :HamStokSarfFisiDetayId;
    select count(*) from HamStokSarfFisiDetay where HamStokSarfFisiId = :HamStokSarfFisiId into :KS;
    if (KS = 0) then delete from HamStokSarfFisi where HamStokSarfFisiId = :HamStokSarfFisiId;
  end
end
