CREATE TRIGGER AFTERINSERTORMEKALITEKONTROL FOR ORMEKALITEKONTROL
ACTIVE AFTER INSERT POSITION 0
as
declare variable OrmeUretimeCikisOtomatik EgeBoolean;
declare variable Aciklama Blob;
declare variable OrmeFisId integer;
declare variable EgemenFirmaID integer;
begin
  select OrmeUretimeCikisOtomatik, EgemenFirmaID from Sistem where SistemId = 1 into :OrmeUretimeCikisOtomatik, :EgemenFirmaID;
  if (OrmeUretimeCikisOtomatik = 1 and (EgemenFirmaID <> 120  or EgemenFirmaID <> 130)) then
    select * from OrmeUretimIcinCikisYap(new.OrmeKaliteKontrolId, new.InsertKullaniciId) into :OrmeFisId, :Aciklama;
end
