CREATE TRIGGER BEFOREINSERTLABRECETETALEP FOR LABRECETETALEP
ACTIVE BEFORE INSERT POSITION 0
as
declare variable LabReceteTalepTarihiOtomatik integer;
begin
  Select LabReceteTalepTarihiOtomatik from Sistem where sistemID = 1 into :LabReceteTalepTarihiOtomatik;

  if (New.HamStokSecimi = 1 and new.HamStokMasterId is null) then exception TalepHamStokGirisNoSecilmemis;
  if (New.HamStokSecimi = 3 and new.KumasGelisTarihi is null) then exception TalepKumasGelisTarihiBosOlmaz;

  if (LabReceteTalepTarihiOtomatik = 1) then new.TalepTarihi = Current_Date;

  New.TalepEdenKullaniciID = New.KullaniciId;

  new.LabReceteTalepKayitId = gen_id(LabReceteTalepKayit_Id_Gen, 1);
end
