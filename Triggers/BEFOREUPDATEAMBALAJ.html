CREATE TRIGGER BEFOREUPDATEAMBALAJ FOR AMBALAJ
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KayitSayisi integer;
begin
  if (new.BobinTopSayisi is null) then New.BobinTopSayisi = 0;
  if (new.KonikRolikAgirligi is null) then New.KonikRolikAgirligi = 0;
  if (new.PosetAgirligi is null) then New.PosetAgirligi = 0;

  if (new.CuvalAgirligi is null) then New.CuvalAgirligi = 0;
  if (new.CuvalIpiAgirligi is null) then New.CuvalIpiAgirligi = 0;
  if (new.PlastikPlakaAgirligi is null) then New.PlastikPlakaAgirligi = 0;

  if (new.KoliAgirligi is null) then New.KoliAgirligi = 0;
  if (new.AraKartonAgirligi is null) then New.AraKartonAgirligi = 0;

  if (new.PaletAgirligi is null) then New.PaletAgirligi = 0;
  if (new.SeperatorSayisi is null) then New.SeperatorSayisi = 0;
  if (new.SeperatorAgirligi is null) then New.SeperatorAgirligi = 0;
  if (new.StrechAgirligi is null) then New.StrechAgirligi = 0;

  new.AmbalajAgirligi = New.BobinTopSayisi * (New.KonikRolikAgirligi + New.PosetAgirligi);
  new.AmbalajAgirligi = new.AmbalajAgirligi + New.CuvalAgirligi + New.CuvalIpiAgirligi + New.PlastikPlakaAgirligi;
  new.AmbalajAgirligi = new.AmbalajAgirligi + new.KoliAgirligi + New.AraKartonAgirligi;
  new.AmbalajAgirligi = new.AmbalajAgirligi + New.PaletAgirligi + New.SeperatorSayisi * New.SeperatorAgirligi + New.StrechAgirligi;
  New.AmbalajAgirligi = New.AmbalajAgirligi / 1000;

/*  if (Old.AmbalajAgirligi <> New.AmbalajAgirligi) then

    if (KayitSayisi > 0) then exception HareketVarAmbalajDegisemez;
  end
*/
end
