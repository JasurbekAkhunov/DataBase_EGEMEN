CREATE TRIGGER BEFOREUPDATEHAKTARMAISEMRI FOR HAKTARMAISEMRI
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KayitSayisi   integer;
begin
  if (Old.BaslatanPersonelId is null and New.BaslatanPersonelId is not null) then
  begin
    if (New.MakinaId is null) then exception GiristeMakinaIdYok;
    New.BaslamaTarihi = current_TimeStamp(0);
  end
  else if (Old.BaslatanPersonelId is not null and New.BaslatanPersonelId is null) then
  begin
    if (New.BitirenPersonelId is not null) then exception IsEmriBitirilmis;

    New.BaslamaTarihi = Null;
    New.MakinaId = Null;
  end

  if (Old.BitirenPersonelId is null and New.BitirenPersonelId is not null) then
  begin
    if (New.BaslatanPersonelId is null) then exception IsEmriBaslamamisBitemez;

    Select Count(*) from HAktarmaIsEmriDetay
         where HAktarmaIsEmriDetay.HAktarmaIsEmriKayitId = new.HAktarmaIsEmriKayitId and (HAktarmaIsEmriDetay.BobinGramaji is null or HAktarmaIsEmriDetay.BobinGramaji = 0)
         into :KayitSayisi;

    if (KayitSayisi > 0) then exception BobinGramajiGirilmemisBitemez;

    New.BitisTarihi = current_TimeStamp(0);
  end
  else if (New.BitirenPersonelId is null) then New.BitisTarihi = Null;
  new.DegisimTarihi = current_TimeStamp(0);
end
