Description

(There is no description for trigger BEFOREUPDATEBOYASIPARISDETAY)

 

Definition

CREATE TRIGGER BEFOREUPDATEBOYASIPARISDETAY FOR BOYASIPARISDETAY
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KayitSayisi   integer;
declare variable Fason         integer;
declare variable SiparisTarihi TimeStamp;
declare variable SiparisKapatmadaRezervSifirla integer;
declare variable OrmeSiparisi  integer;
declare variable HamTipiId     integer;
declare variable HstIplik      integer;
declare variable hstElyaf      integer;
declare variable hstTelef      integer;
declare variable BoyaSiparisRenkNoKullanimda integer;
declare variable OrmeHamId                   integer;
declare variable OrmeHamTipiId               integer;
declare variable MasterSiparisTipiId         integer;
declare variable FSiparisOdemeSekliId        integer;
declare variable FLabReceteKodu              varchar(20);
declare variable FLabReceteAdi               varchar(40);
declare variable RenkAdi                     varchar(61);
declare variable SipTerminBosBirakilabilsin  integer;
declare variable SistemDovizId               varchar(5);
declare variable FDovizId                    varchar(5);
declare variable DovizKuru                   Double Precision;
declare variable SipBrutMiktarHesaplamaSekli integer;
declare variable BSatisSiparisleriNetGirilecek integer;
declare variable HerZamanSiparisProsesiKullan integer;
declare variable SiparisBrutuElleGirebilsin  integer;
declare variable FirmaSiparisRiskiAsilamaz   integer;
declare variable SiparisRiski                Double Precision;
declare variable FSiparisRiskToplami         Double Precision;
declare variable SiparisFirmaId              varchar(20);
declare variable RiskDovizId                 varchar(5);
declare variable FSiparisRiski               Double Precision;
declare variable FSiparisTarihi              TimeStamp;
declare variable ISipPaketlemeDegiPartiGuncelle integer;
declare variable SiparisiSevkOlmayandaKapatma EgeBoolean;
declare variable EgemenFirmaID type of column Sistem.EgemenFirmaID;
begin
  Select SipBrutMiktarHesaplamaSekli, BSatisSiparisleriNetGirilecek, HerZamanSiparisProsesiKullan, SiparisBrutuElleGirebilsin, BoyaSiparisRenkNoKullanimda, SipTerminBosBirakilabilsin, FirmaSiparisRiskiAsilamaz, ISipPaketlemeDegiPartiGuncelle, EgemenFirmaID from Sistem where SistemID = 1
    into :SipBrutMiktarHesaplamaSekli, :BSatisSiparisleriNetGirilecek, :HerZamanSiparisProsesiKullan, :SiparisBrutuElleGirebilsin, :BoyaSiparisRenkNoKullanimda, :SipTerminBosBirakilabilsin, :FirmaSiparisRiskiAsilamaz, :ISipPaketlemeDegiPartiGuncelle, :EgemenFirmaID;

  Select FirmaId, Fason, SiparisTarihi, OrmeSiparisi, SiparisTipiId, OdemeSekliId from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitId = new.BoyaSiparisKayitId
         into :SiparisFirmaId, :Fason, :SiparisTarihi, :OrmeSiparisi, :MasterSiparisTipiId, :FSiparisOdemeSekliId;

  Select SiparisRiski, RiskDovizId from Firma where Firma.FirmaID = :SiparisFirmaID into :SiparisRiski, :RiskDovizID;

  if (new.MaximumFireOrani >= 100) then exception KayitHatasi 'Fire Orani Hatalı';
  if (new.MinimumFireOrani < 0 or new.MinimumFireOrani >= 100) then exception KayitHatasi 'Fire Orani Hatalı';

  if (OrmeSiparisi = 1 and New.OrmeSiparisDetayId is null) then exception KayitHatasi 'Örme Sipariş DetayId boş olamaz';

  if (FirmaSiparisRiskiAsilamaz = 1) then
  begin
    FSiparisRiskToplami = 0;
    for Select Sum(BirimFiyat * SiparisKilo), SiparisTarihi, DovizID from BoyaSiparis
    left Join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitID = BoyaSiparis.BoyaSiparisKayitID
    where BoyaSiparis.FirmaID = :SiparisFirmaId and BoyaSiparisDetay.Kapali = 0 and BoyaSiparisDetayID <> new.BoyaSiparisDetayID
    Group By SiparisTarihi, DovizID
    into :FSiparisRiski, :FSiparisTarihi, :FDovizID do
    begin
      if (RiskDovizId <> FDovizID) then Select DovizKuru from EnYakinDovizKuruGetir(:FDovizID, :RiskDovizId, :FSiparisTarihi) into :DovizKuru;
      else DovizKuru = 1;
      FSiparisRiskToplami  = FSiparisRiskToplami + (FSiparisRiski * DovizKuru);
    end

    if (RiskDovizId <> new.DovizID) then Select DovizKuru from EnYakinDovizKuruGetir(new.DovizID, :RiskDovizId, :SiparisTarihi) into :DovizKuru;
    else DovizKuru = 1;

    if ((FSiparisRiskToplami + (new.BirimFiyat * new.SiparisKilo * DovizKuru)) > SiparisRiski) then exception KayitHatasi ' Firmanın Tanımlanan Sipariş Riski Aşılamaz...';
  end

  if (New.OrmeSiparisDetayId is not null) then
     Select OdemeSekliId from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayId into :FSiparisOdemeSekliId;

  if (new.OdemeSekliId is null) then new.OdemeSekliId = FSiparisOdemeSekliId;
  if (FSiparisOdemeSekliId is not null and new.OdemeSekliId is not null and new.OdemeSekliId <> FSiparisOdemeSekliId) then exception OdemeSekliFarkliOlamaz;

  if (SipTerminBosBirakilabilsin = 0 and New.TerminTarihi is null) then New.TerminTarihi = SiparisTarihi;


  if (new.RevizeTerminTarihi is null) then
     new.SonTerminTarihi = new.TerminTarihi;
   else
     new.SonTerminTarihi = new.RevizeTerminTarihi;

  if (New.IslemTipiId <> 3) then new.FaturaKesilecek = 1;
  else if (new.UcretliTamir = 1) then new.FaturaKesilecek = 1;
  else new.FaturaKesilecek = 0;

  if (new.StoktanKullanilacakKilo is null) then new.StoktanKullanilacakKilo = 0;
  if (new.StoktanKullanilacakMetre is null) then new.StoktanKullanilacakMetre = 0;

  if (new.FaturaEdilecekMiktarTipi is null) then
  if (Fason = 0) then
    Select DefaultSatisPartiMiktarTipi from Sistem where sistemid = 1 into new.FaturaEdilecekMiktarTipi;
  else
    Select DefaultFasonPartiMiktarTipi from Sistem where sistemid = 1 into new.FaturaEdilecekMiktarTipi;

  if (Old.HamStokBirimId <> New.HamStokBirimId or old.FaturaEdilecekMiktarTipi <> New.FaturaEdilecekMiktarTipi) then
  begin
    Select Count(*) from PartiStok
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
      left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
      where PartiStok.BoyaSiparisDetayId = New.BoyaSiparisDetayId and BoyaIrsaliyeDetay.BoyaFaturaId is not null
      into :KayitSayisi;

    if (KayitSayisi > 0) then exception FaturaKesilmisFiyatDegisemez;
  end

  if ((EgemenFirmaId = 3 or EgemenFirmaId = 11) and New.KKDeger7 is not Null) then
  begin
    if (StrKarsilastir(New.LabReceteKodu, Old.LabReceteKodu) = 0 or StrKarsilastir(New.DesenId, Old.DesenId) = 0) then New.KKDeger7 = Null;
  end

  Select Count(*) from PartiStok where PartiStok.BoyaSiparisDetayId = New.BoyaSiparisDetayId into :KayitSayisi;

  if (Old.OnayDurumu <> New.OnayDurumu and New.OnayDurumu <> 1) then
  if (KayitSayisi > 0 ) then exception PartiAcilmisOnayDurumuDegisemez;

  if (KayitSayisi > 0) then
  begin
    if (New.HamId <> Old.HamId) then exception SipHareketGormusHamIdDegisemez;
    if (Old.SiparisProsesID <> New.SiparisProsesID) then exception PartiAcilmisDegisemez ' (Sipariş Proses)';
  end

  if (Old.HamId <> New.HamId) then
  begin
    Select Count(*) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayId = New.BoyaSiparisDetayId into :KayitSayisi;
    if (KayitSayisi > 0) then exception SipHareketGormusHamIdDegisemez;

    Select Count(*) from IplikTalepDetay where IplikTalepDetay.BoyaSiparisDetayId = new.BoyaSiparisDetayId into :KayitSayisi;

    if (KayitSayisi > 0) then exception KayitHatasi 'İplik Talebi Girilmiş, Ürün Değişemez..';

    Select Ham.HamTipiId from Ham where Ham.HamId = new.HamId into :HamTipiId;

    Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

    if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then new.SiparisTipiID = 0;
    else new.SiparisTipiID = 1;

    if (new.OrmeSiparisDetayID is not null) then
    begin
      Select OrmeSiparisDetay.HamId, Ham.HamTipiId from OrmeSiparisDetay
        left join Ham on Ham.HamID = OrmeSiparisDetay.HamID
        where OrmeSiparisDetay.OrmeSiparisDetayId = New.OrmeSiparisDetayID
        into :OrmeHamId, :OrmeHamTipiId;

      if (HamTipiId = OrmeHamTipiId and New.HamId <> OrmeHamId) then exception KayitHatasi 'Orme Siparisi ile ürün uymuyor..';
      else if (new.SiparisTipiID = MasterSiparisTipiId and New.HamId <> OrmeHamId) then exception KayitHatasi 'Orme Siparisi ile ürün uymuyor..';
    end

  end

  if (Old.Kapali <> New.Kapali and New.Kapali <> 0) then
  begin
     if (new.Kapali = 2) then
     begin
       Select Count(*) from PartiStok where PartiStok.BoyaSiparisDetayId = New.BoyaSiparisDetayId and PartiStok.PartiStokIptalKayitId is null into :KayitSayisi;

       if (KayitSayisi = 0) then
         Select Count(*) from StokHareket where StokHareket.BoyaSiparisDetayId = New.BoyaSiparisDetayId into :KayitSayisi;

       if (KayitSayisi > 0) then exception HareketVarSiparisIptalEdilemez;
     end

     Select SiparisKapatmadaRezervSifirla, SiparisiSevkOlmayandaKapatma from sistem where sistemid = 1 into :SiparisKapatmadaRezervSifirla, :SiparisiSevkOlmayandaKapatma;

     if (SiparisiSevkOlmayandaKapatma = 1) then
     begin
       if (EgemenFirmaID <> 110 and EgemenFirmaID <> 138 and EgemenFirmaID <> 146) then /*ayyıldız, artesa, özsar*/
         Select Count(*) from PartiStok where PartiStok.BoyaSiparisDetayId = New.BoyaSiparisDetayId and PartiStok.Durumu <> 8 and PartiStok.PartiStokIptalKayitId is null into :KayitSayisi;
      else
       Select Count(*) from PartiStokTarti
       left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifID
       where PartiStok.BoyaSiparisDetayId = New.BoyaSiparisDetayID and PartiStokTarti.KaliteNo = 1 and PartiStokTarti.BoyaIrsaliyeDetayId is  null and PartiStokTarti.HamStokSarfFisiDetayId is  null into :KayitSayisi;

       if (KayitSayisi > 0) then exception SipPartiVarKapatamazsiniz;
     end

     Select Count(*) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayId = New.BoyaSiparisDetayId and BoyaSiparisRezerv.RezervVar = 1 into :KayitSayisi;

     if (KayitSayisi > 0) then
     if (New.OrmeSiparisDetayId is null and SiparisKapatmadaRezervSifirla = 1) then execute Procedure SiparisRezervleriniSifirla(new.BoyaSiparisDetayId);
     else exception SiparisRezerviVarKapatamazsiniz;
  end

  if (new.OrnekPartiSinifId is null or New.OrnekPartiAsamalariId is null) then
  begin
     New.OrnekPartiAsamalariId = null;
     new.OrnekPartiSinifId = null;
     New.OrnekOnay = 0;
  end

  if (Old.FaturaKesilecek <> New.FaturaKesilecek) then
  begin
    Update PartiStok Set FaturaKesilecek = New.FaturaKesilecek where PartiStok.BoyaSiparisDetayID = new.BoyaSiparisDetayId;
  end

  if (BoyaSiparisRenkNoKullanimda = 1 and New.OrmeSiparisDetayID is not null and new.BoyaSiparisRenkNo is null) then exception KayitHatasi ' Sipariş Renk No Boş Bırakılamaz..';

  if (ISipPaketlemeDegiPartiGuncelle = 1 and Old.PaketlemeSekliID <> New.PaketlemeSekliID) then
  begin
    Update PartiStok Set PaketlemeSekliID = New.PaketlemeSekliID where PartiStok.BoyaSiparisDetayID = new.BoyaSiparisDetayId;
  end


  if (IntKarsilastir(Old.BoyaSiparisRenkNo, New.BoyaSiparisRenkNo) = 0) then
  begin
    if (New.BoyaSiparisRenkNo is not null) then
    begin
      Select LabReceteId, LabReceteKodu, LabReceteAtisId, RenkId, RenkAdi, MusteriRenkKoduveAdi from BoyaSiparisRenk
        where BoyaSiparisRenk.BoyaSiparisKayitID = New.BoyaSiparisKayitID and BoyaSiparisRenk.BoyaSiparisRenkNO = New.BoyaSiparisRenkNo
        into new.LabReceteId, new.LabReceteKodu, new.LabReceteAtisId, New.RenkId, new.RenkAdi, new.MusteriRenkKoduveAdi;
    end
    else if (BoyaSiparisRenkNoKullanimda = 1) then
    begin
      new.LabReceteId = Null;
      new.LabReceteKodu= Null;
      new.LabReceteAtisId = Null;
      new.RenkAdi = Null;
      new.MusteriRenkKoduveAdi = Null;
    end
  end

  if (StrKarsilastir(Old.DesenId, New.DesenId) = 0) then
  begin
    Select Count(*) from BoyaSiparisBaski where BoyaSiparisBaski.BoyaSiparisDetayId = New.BoyaSiparisDetayId into :KayitSayisi;
    if (KayitSayisi > 0) then exception VaryantSecilmisDesenDegisemez;
  end

  if (Old.Kapali <> new.Kapali) then
  begin
    if (new.Kapali <> 0) then
    begin
      New.KapatmaTarihi = Current_date;
      Update BoyaTalimat Set Kapali = 1 where BoyaTalimat.BoyaSiparisDetayID = New.BoyaSiparisDetayID and BoyaTalimat.Kapali = 0;
    end
    else New.KapatmaTarihi = Null;
  end

  if (Old.Boyanacak <> New.Boyanacak and New.Boyanacak = 0) then
  begin
    Select count(*) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayId = New.BoyaSiparisDetayId into :KayitSayisi;

    if (KayitSayisi > 0) then exception BoyaIcinRezervGirilmisDegisemez;
  end

  if (New.OrmeSiparisDetayId is not null and IntKarsilastir(Old.LabReceteId, New.LabReceteId) = 0 and New.LabReceteId is not null and New.RenkId is null) then
  begin
    Select LabRecete.LabReceteKodu, LabRecete.Adi from LabRecete where LabRecete.LabReceteId = new.LabReceteId into :FLabReceteKodu, :FLabReceteAdi;

    Select Renk.RenkId from Renk where Renk.LabReceteID = New.LabReceteID into new.RenkId;

    if (new.RenkID is null) then
    begin
      new.RenkID = Gen_Id(Renk_Id_Gen, 1);

      RenkAdi = FLabReceteAdi || '-' || FLabReceteKodu;

      insert into Renk(RenkId, LabReceteId, Adi) values(new.RenkId, New.LabReceteId, EgemenCopyStr(:RenkAdi, 1, 50));
    end
  end

  if (New.KaliteID = '') then New.KaliteID = Null;

  if (IntKarsilastir(Old.OdemeSekliId, new.OdemeSekliId) = 0 and new.OdemeSekliId is not null) then
    Select VadeGun from OdemeSekli where OdemeSekli.OdemeSekliID = New.OdemeSekliId into New.VadeGun;


  if (IntKarsilastir(Old.LabReceteId, New.LabReceteId) = 0) then
  begin
    if (Old.LabReceteId is null and New.ReceteGirisTarihi is null) then
    begin
      New.ReceteGirisTarihi = current_TimeStamp(0);
      New.ReceteGirisKullaniciId = New.KullaniciId;
    end
    else
    begin
      New.ReceteDegisimTarihi = current_TimeStamp(0);
      New.ReceteDegisimKullaniciId = New.KullaniciId;
    end
  end

  if (StrKarsilastir(Old.DesenId, New.DesenId) = 0) then
  begin
    if (Old.DesenId is null and New.DesenGirisTarihi is null) then
    begin
      New.DesenGirisTarihi = current_TimeStamp(0);
      New.DesenGirisKullaniciId = New.KullaniciId;
    end
    else
    begin
      New.DesenDegisimTarihi = current_TimeStamp(0);
      New.DesenDegisimKullaniciId = New.KullaniciId;
    end
  end

  if (Old.SiparisProsesID <> New.SiparisProsesID) then
  begin
    New.ProsesID = Null;
    if (New.SiparisProsesID is not null) then
    begin
      Select BirimMaliyet, DovizId from SiparisProses where SiparisProsesID = New.SiparisProsesID into new.ProsesBirimMaliyet, :FDovizId;

      if (SistemDovizID <> FDovizId) then
        Select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :SistemDovizId, :SiparisTarihi) into :DovizKuru;
      else
        DovizKuru = 1;

      new.ProsesBirimMaliyet = new.ProsesBirimMaliyet * DovizKuru;
    end
  end

  if (Old.RenkDerinligiID <> New.RenkDerinligiID) then
  if (New.RenkDerinligiID is not null) then
  begin
    Select BirimFiyat, DovizId from RenkDerinligi where RenkDerinligiId = New.RenkDerinligiId into new.RenkDerinligiBirimMaliyet, :FDovizId;

    if (SistemDovizID <> FDovizId) then
      Select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :SistemDovizId, :SiparisTarihi) into :DovizKuru;
    else
      DovizKuru = 1;

    new.RenkDerinligiBirimMaliyet = new.RenkDerinligiBirimMaliyet * DovizKuru;
  end

  if ((New.ProsesID is Null or HerZamanSiparisProsesiKullan = 1) and (New.SiparisProsesId is not Null)) then
    select ProsesId from SiparisProses where SiparisProsesId = New.SiparisProsesId into new.ProsesId;

  if (new.SiparisBrutKilo is Null or new.SiparisBrutKilo = 0 or SiparisBrutuElleGirebilsin = 0) then
  begin
    if (new.MaximumFireOrani is not null and BSatisSiparisleriNetGirilecek = 1) then
    begin
      if (SipBrutMiktarHesaplamaSekli = 0) then
      begin
        new.SiparisBrutKilo = new.SiparisKilo / ((100 - new.MaximumFireOrani) / 100);
        Select Miktar from FormataGoreKiloYuvarla(new.SiparisBrutKilo, 'BOYA', 'SIPDETAYBRUTKILO') into new.SiparisBrutKilo;
      end
      else
      begin
        new.SiparisBrutKilo = new.SiparisKilo + new.SiparisKilo * new.MaximumFireOrani / 100;
        Select Miktar from FormataGoreKiloYuvarla(new.SiparisBrutKilo, 'BOYA', 'SIPDETAYBRUTKILO') into new.SiparisBrutKilo;
      end
    end
    else new.SiparisBrutKilo = new.SiparisKilo;

    if (old.KopyaSonuUpdateYok = 1 and SiparisBrutuElleGirebilsin = 1) then new.SiparisBrutKilo = 0;
  end

  if (new.SiparisBrutMetre is Null or new.SiparisBrutMetre = 0 or SiparisBrutuElleGirebilsin = 0) then
  begin
    if (new.MaximumFireOrani is not null and Fason = 0 and BSatisSiparisleriNetGirilecek = 1) then
    begin
      if (SipBrutMiktarHesaplamaSekli = 0) then
      begin
        new.SiparisBrutMetre = new.SiparisMetre / ((100 - new.MaximumFireOrani) / 100);
        Select Miktar from FormataGoreKiloYuvarla(new.SiparisBrutMetre, 'BOYA', 'BRUTMETRE') into new.SiparisBrutMetre;
      end
      else
      begin
        new.SiparisBrutMetre = new.SiparisMetre + new.SiparisMetre * new.MaximumFireOrani / 100;
        Select Miktar from FormataGoreKiloYuvarla(new.SiparisBrutMetre, 'BOYA', 'BRUTMETRE') into new.SiparisBrutMetre;
      end
    end
    else new.SiparisBrutMetre = new.SiparisMetre;

    if (old.KopyaSonuUpdateYok = 1 and SiparisBrutuElleGirebilsin = 1) then new.SiparisBrutMetre = 0;
  end

  if (new.FirmaSevkAdresiId<>old.FirmaSevkAdresiId) then
  begin
     Update PartiStok Set PartiStok.firmasevkadresiid = new.FirmaSevkAdresiId Where PartiStok.BoyaSiparisDetayID = new.BoyaSiparisDetayId and PartiStok.Durumu <> 8;
  end
end
