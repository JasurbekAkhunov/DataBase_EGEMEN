CREATE TRIGGER AFTERUPDATEKALITE FOR KALITE
ACTIVE AFTER UPDATE POSITION 0
As
begin
  Execute Procedure Logla('U', 2, 'KALITE', 'KALITEID', new.KaliteId, 'Kalite Adı :'||Coalesce(new.KaliteAdi,'')||' Parti Id :'||Coalesce(new.PartiId,'')|| ' Parti Yıl :'||Coalesce(new.PartiYil,''));

  if (New.PartiSinifId Is Not Null And New.PartiSinifId <> Old.PartiSinifId) then
    Execute Procedure KaliteReceteyiPartidenOlustur(New.KaliteId, New.KullaniciId);
end
