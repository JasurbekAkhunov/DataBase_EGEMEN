CREATE TRIGGER BEFOREDELETEPARTISABRECILAVE FOR PARTISABLONRECETEILAVE
ACTIVE BEFORE DELETE POSITION 0
as
declare variable StoktanDusuldu integer;
begin
  Select StoktanDusuldu from PartiSablonIlave where PartiSablonIlave.PartiSablonIlaveID = Old.PartiSablonIlaveId into :StoktanDusuldu;

  if (StoktanDusuldu = 1) then  exception StokDusulduDegisemez;

  if (Old.Pat = 0) then
    Update PartiSablonIlave Set BoyaToplami = BoyaToplami - Old.Miktar where PartiSablonIlave.PartiSablonIlaveID = Old.PartiSablonIlaveId;
end
