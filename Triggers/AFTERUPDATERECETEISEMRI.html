CREATE TRIGGER AFTERUPDATERECETEISEMRI FOR RECETEISEMRI
ACTIVE AFTER UPDATE POSITION 0
as
declare variable PartiAsamalariId integer;
begin
  if (Old.BanyoCarpani <> New.BanyoCarpani or
      Old.BanyoCarpani2 <> New.BanyoCarpani2 or
      Old.BanyoCarpani3 <> New.BanyoCarpani3 or
      Old.BanyoCarpani4 <> New.BanyoCarpani4 or
      Old.Koyultma <> New.Koyultma or
      IntKarsilastir(Old.DurulamaSuyu, New.DurulamaSuyu) = 0 or
      IntKarsilastir(Old.BoyamaSeviyesi, New.BoyamaSeviyesi) = 0 or
      IntKarsilastir(Old.PompaDegeri, New.PompaDegeri) = 0 or
      IntKarsilastir(Old.Duze, New.Duze) = 0 or
      StrKarsilastir(Format_Float('',Old.HamStokFlotteCarpani), Format_Float('',New.HamStokFlotteCarpani)) = 0 or
      StrKarsilastir(Format_Float('',Old.Tur), Format_Float('',New.Tur)) = 0 or
      StrKarsilastir(Format_Float('',Old.TuzCarpani), Format_Float('',New.TuzCarpani)) = 0 or
      Old.IslaklikOrani <> New.IslaklikOrani or
      Old.KuleHizi <> New.KuleHizi or
      Old.MaxKuleHizi <> New.MaxKuleHizi or
      Old.FloteOraniYikama <> New.FloteOraniYikama or
      Old.SuTipiArtiTuz <> New.SuTipiArtiTuz or
      IntKarsilastir(Old.BoyaSuyu, New.BoyaSuyu) = 0 or
      IntKarsilastir(Old.TupSayisi, New.TupSayisi) = 0 or
      IntKarsilastir(Old.FanGucu, New.FanGucu) = 0) then
    for select PartiAsamalariId from ReceteIsEmriDetay where ReceteIsEmriDetay.ReceteIsEmriYil = new.ReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = new.ReceteIsEmriId into :PartiAsamalariId do
      update PartiAsamalari set BanyoCarpani = New.banyoCarpani,
                                BanyoCarpani2 = New.banyoCarpani2,
                                BanyoCarpani3 = New.banyoCarpani3,
                                BanyoCarpani4 = New.banyoCarpani4,
                                Koyultma = new.Koyultma,
                                DurulamaSuyu = New.DurulamaSuyu,
                                BoyamaSeviyesi = New.BoyamaSeviyesi,
                                BoyaSuyu = New.BoyaSuyu,
                                HamStokFlotteCarpani = new.HamStokFlotteCarpani,
                                Tur = New.Tur,
                                PompaDegeri = New.PompaDegeri,
                                Duze = New.Duze,
                                TuzCarpani = New.TuzCarpani,
                                IslaklikOrani = New.IslaklikOrani,
                                KuleHizi  = New.KuleHizi,
                                MaxKuleHizi = New.MaxKuleHizi,
                                FloteOraniYikama = New.FloteOraniYikama,
                                SuTipiArtiTuz = New.SuTipiArtiTuz,
                                TupSayisi = New.TupSayisi,
                                FanGucu = New.FanGucu,
                                KullaniciId = new.KullaniciId where PartiAsamalari.PartiAsamalariId = :PartiAsamalariId;
end
