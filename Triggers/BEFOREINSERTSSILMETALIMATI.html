CREATE TRIGGER BEFOREINSERTSSILMETALIMATI FOR SABLONSILMETALIMATI
ACTIVE BEFORE INSERT POSITION 0
as
begin
  update SablonSilmeTalimatNumarator set Id = new.FisNo where Yil = new.FisYil and Id < New.FisNo;

  if  (New.BaslatanKullaniciId is not null) then
    new.BaslamaTarihi = current_TimeStamp(0);

  if  (New.BitirenKullaniciId is not null) then
    new.BitisTarihi = current_TimeStamp(0);

  new.SablonSilmeTalimatiId = gen_id(SablonSilmeTalimati_Id_Gen, 1);
end
