Definition

CREATE TRIGGER BEFOREINSERTSTOKHAREKET0 FOR STOKHAREKET
ACTIVE BEFORE INSERT POSITION 0
as

declare variable FOrmeIrsaliyeId     integer;
declare variable FOrmeFaturaId       integer;
declare variable FHareketTipi        integer;
declare variable FBoyaSiparisKayitId integer;
declare variable shtAlis             integer;
declare variable shtFasonUretim      integer;
declare variable shtUretim           integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis     integer;
declare variable uitOrme     integer;
declare variable uitBukum    integer;
declare variable uitBoya     integer;
declare variable uitDokuma   integer;
declare variable uitCozgu    integer;
declare variable uitHasil    integer;
declare variable uitIndigo   integer;
declare variable shtSarf     integer;
declare variable ELF         integer;
declare variable BoyaKapali  integer;
declare variable BCBoyaSiparisDetayZorunlu integer;
declare variable KantarKayitId       integer;
declare variable OnayDurumu          integer;
declare variable XBoyaSiparisKayitId integer;
declare variable SonDonemBitisTarihi TimeStamp;
declare Variable KayitSayisi         integer;
declare Variable FDokumaHareketi     integer;
declare variable IrsTarihFatTarihtenBuyukOlamaz integer;
declare variable FIrsaliyeTarihi    TimeStamp;
declare variable shtFasonAlis       integer;
declare variable KS                 integer;
declare variable DokumaUreCikCozguLotuGuncelle  EgeBoolean;
declare variable CozguDeposu        EgeBoolean;
declare variable HasilDeposu        EgeBoolean;
declare variable CozguIsEmriKayitId integer;
declare variable Kapali EgeBoolean;
declare variable ASira integer;
declare variable DokumaProjesiAktif integer;
declare variable SettekiLeventAdedi integer;
declare variable LeventNo integer;
declare variable FLeventNo integer;
declare variable FLeventSiraNo integer;
declare variable DokumaUretimdeOtomatikSarf integer;
begin
  if (new.Adet is null) then new.Adet = 0;
  Select shtSarf from OrmeStokHareketTipiGetir into :shtSarf;
  select sistem.DokumaProjesiAktif, DokumaUretimdeOtomatikSarf from sistem  into :DokumaProjesiAktif, :DokumaUretimdeOtomatikSarf;
  Select shtAlis, shtFasonUretim, shtUretim, shtFasonUretimeCikis, shtUretimeCikis, shtFasonAlis from OrmeStokHareketTipiGetir into :shtAlis, :shtFasonUretim, :shtUretim, :shtFasonUretimeCikis, :shtUretimeCikis, :shtFasonAlis;

  if (new.DokumaHareketi = 1 and new.HareketTipi = shtUretimeCikis and new.UretimeCikisTarihi is null) then
    new.UretimeCikisTarihi = current_timestamp(0);

  if (New.DokumaHareketi = 1) then /*Miktar birimi tespit edilmeli. Cozgu deposuna giren metre olacak, digerleri kilo*/
  begin

    select Zorunlu from DokumaMetreZorunlu(New.DepoId, New.HamId) into :KS;

    if (KS = 1) then
      New.MiktarBirim = 1;
    else
      New.MiktarBirim = 0;

    if (New.Miktar is null or New.Miktar = 0) then
    begin
      if (New.MiktarBirim = 1) then
        New.Miktar = New.Metre;
      else
        New.Miktar = New.Kilo;
    end
    else
    begin
      if (New.MiktarBirim = 1 and Coalesce(New.Metre,0) = 0) then
        New.Metre = New.Miktar;
      else if (New.MiktarBirim = 0 and Coalesce(New.Kilo,0) = 0) then
        New.Kilo = New.Miktar;
    end
  end

  if (new.UretimIslemTipiId = uitCozgu and new.CozguIsEmriKayitId is not null) then
  begin
    select CozguIsEmri.Kapali from CozguIsEmri where CozguIsEmri.CozguIsEmriKayitId = new.CozguIsEmriKayitId into :Kapali;
    if (Kapali = 1) then Exception IsEmriKapaliHareketGirilemez;
  end

  if (New.BoyaSiparisDetayID is not null) then
  begin
    Select BoyaSiparisKayitId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = new.BoyaSiparisDetayId  into :XBoyaSiparisKayitId;

  --  if (XBoyaSiparisKayitId <> New.BoyaSiparisKayitId) then exception Kayithatasi 'ALİ BURAYI KAPATTI Sipariş ve Detay Referans Farklı.';
  end

  Select uitBukum, uitOrme, uitBoya, uitDokuma, uitCozgu, uitHasil, uitIndigo from UretimIslemTipiGetir into :UitBukum, :uitOrme, :uitBoya, :uitDokuma, :uitCozgu, :uitHasil, :uitIndigo;

  if (new.OrmeFisId is null and new.OrmeIrsaliyeId is null and new.OrmeFaturaId is null) then exception StokHareketBaglantisiYok;

  if (new.OrmeFisId is not null) then new.NeredenGirildi = 0;
  else if (new.OrmeIrsaliyeId is not null) then new.NeredenGirildi = 1;
  else if (new.OrmeFaturaId is not null) then new.NeredenGirildi = 2;

  if (new.NeredenGirildi = 0 and (new.OrmeIrsaliyeId is not null or new.OrmeFaturaId is not null) ) then exception StokHareketIdHatasi;
  if (new.NeredenGirildi = 1 and (new.OrmeFisId is not null or new.OrmeFaturaId is not null) ) then exception StokHareketIdHatasi;
  if (new.NeredenGirildi = 2 and (new.OrmeFisId is not null or new.OrmeIrsaliyeId is not null) ) then exception StokHareketIdHatasi;



  Select Count(*) from StokHareket where StokHareket.OrmeFisID = New.OrmeFisID and StokHareket.OrmeIrsaliyeId = New.OrmeIrsaliyeId and StokHareket.OrmeFaturaId = New.OrmeFaturaId and StokHareket.DokumaHareketi <> New.DokumaHareketi into :KayitSayisi;
  if (KayitSayisi > 0) then exception FarkliModullerdenGirisYapilamaz;

  if (New.NeredenGirildi = 0) then
  begin
    Select FirmaId, FisTarihi, UretimIslemTipiId, OrmeIrsaliyeId, OrmeFaturaId, HareketTipi, DovizId, TabanDovizKuru1, TabanDovizKuru2, DokumaHareketi from OrmeFis where OrmeFis.OrmeFisId = new.OrmeFisId
      into new.FirmaId, new.Tarih, new.UretimIslemTipiId, :FOrmeIrsaliyeId, :FOrmeFaturaId, :FHareketTipi, new.DovizId, new.TabanDovizKuru1, new.TabanDovizKuru2, :FDokumaHareketi;

    if (new.HareketTipi <> FHareketTipi) then exception StokHareketTipiUymuyor;
    if (FOrmeIrsaliyeId is not null) then exception SIrsaliyelestirilmisDegisemez;
    if (FOrmeFaturaId is not null) then exception SFaturalastirilmisDegisemez;

    new.FiiliSevkTarihi = new.Tarih;
  end
  else if (New.NeredenGirildi = 1) then
  begin
    select FirmaId, IrsaliyeTarihi, FiiliSevkTarihi, UretimIslemTipiId, OrmeFaturaId, HareketTipi, DovizId, KantarKayitId, TabanDovizKuru1, TabanDovizKuru2, DokumaHareketi from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = new.OrmeIrsaliyeId
      into new.FirmaId, new.Tarih, new.FiiliSevkTarihi, new.UretimIslemTipiId, :FOrmeFaturaId, :FHareketTipi, new.DovizId, :KantarKayitId, New.TabanDovizKuru1, New.TabanDovizKuru2, :FDokumaHareketi;

    if (new.HareketTipi <> FHareketTipi) then exception StokHareketTipiUymuyor;
    if (FOrmeFaturaId is not null or new.OrmeFaturaID is not Null) then exception IFaturalastirilmisDegisemez;

    if (KantarKayitId is not null) then
    begin
      Select OnayDurumu from Kantar where Kantar.KantarKayitID = :KantarKayitId into :OnayDurumu;
      if (OnayDurumu = 1) then exception KantarOnayiVerilmisDegisemez;
    end

  end
  else if (New.NeredenGirildi = 2) then
  begin
    select FirmaId, FaturaTarihi, UretimIslemTipiId, HareketTipi, DovizId, TabanDovizKuru1, TabanDovizKuru2, DokumaHareketi from OrmeFatura where OrmeFatura.OrmeFaturaId = new.OrmeFaturaId
      into new.FirmaId, new.Tarih, new.UretimIslemTipiId, :FHareketTipi, new.DovizId, New.TabanDovizKuru1, New.TabanDovizKuru2, :FDokumaHareketi;

    if (new.HareketTipi <> FHareketTipi) then exception StokHareketTipiUymuyor;

    new.FiiliSevkTarihi = new.Tarih;
  end

  if (FDokumaHareketi <> New.DokumaHareketi) then exception FarkliModullerdenGirisYapilamaz;

  if (New.BoyahaneFasonIade = 0 and New.UretimPartiNo <> '') then
  if (new.UretimIslemTipiId is not null and new.UretimIslemTipiId <> :UitBukum and new.UretimIslemTipiId <> :uitHasil and new.UretimIslemTipiId <> :uitIndigo and new.UretimIslemTipiId <> :UitCozgu and New.BoyaSiparisKayitId is null) then exception UretimIcinSiparisNoGirilmeli;

  if (new.UretimIslemTipiId = uitOrme and new.OrmePlanlamaId is null) then exception UretimIcinTalimatSecilmemis;
  if (new.UretimIslemTipiId = uitDokuma and new.DokumaPlanlamaId is null) then
  begin
     if (DokumaUretimdeOtomatikSarf = 1 and new.HareketTipi <> shtSarf) then
       exception UretimIcinTalimatSecilmemis;

  end

  if (new.UretimIslemTipiId = uitHasil and new.HasilIsEmriKayitID is null) then exception UretimIcinTalimatSecilmemis;
  if (new.UretimIslemTipiId = uitIndigo and new.IndigoIsEmriKayitID is null) then exception UretimIcinTalimatSecilmemis;
  --if (new.UretimIslemTipiId = uitCozgu and new.CozguIsEmriKayitID is null and (new.HareketTipi = shtFasonUretim or new.HareketTipi = shtUretim)) then exception UretimIcinTalimatSecilmemis;
  if (new.UretimIslemTipiId = uitCozgu and new.CozguIsEmriReceteID is null and (new.HareketTipi = shtFasonUretimeCikis or new.HareketTipi = shtUretimeCikis)) then exception UretimIcinTalimatSecilmemis;

  Select BCBoyaSiparisDetayZorunlu, IrsTarihFatTarihtenBuyukOlamaz, DokumaUreCikCozguLotuGuncelle from Sistem where SistemID = 1 into :BCBoyaSiparisDetayZorunlu, :IrsTarihFatTarihtenBuyukOlamaz, :DokumaUreCikCozguLotuGuncelle;
  if (BCBoyaSiparisDetayZorunlu = 1) then
  if (new.UretimIslemTipiId = uitBoya and New.BoyahaneFasonIade = 0 and New.UretimPartiNo <> '') then
  if (new.BoyaSiparisDetayId is null) then exception UretimIcinSiparisNoGirilmeli;

  if (IrsTarihFatTarihtenBuyukOlamaz = 1 and new.OrmeFaturaId is not Null and new.OrmeIrsaliyeID is not Null) then
  begin
    select OrmeIrsaliye.IrsaliyeTarihi from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = new.OrmeIrsaliyeId into :FIrsaliyeTarihi;

    if (FIrsaliyeTarihi > New.Tarih) then exception IrsTarFatTardanBuyukOlamaz;
  end


  if (new.OrmeMakinaId = '') then New.OrmeMakinaId = null;
  if (new.DokumaMakinaId = '') then New.DokumaMakinaId = null;
  if (New.KapAdedi is null) then New.KapAdedi = 0;

  if ( (new.VerilenSiparisDetayId is not null and new.VerilenSiparisKayitId  is null ) or
       (new.VerilenSiparisDetayId is null and new.VerilenSiparisKayitId is not null ) ) then exception VerilenSiparisIdHatasi;

  if (new.HamIade = 1) then new.FaturaKesilecek = 0;

  /*if (new.FaturaKesilecek = 0) then new.BirimFiyat = 0; Palben*/

  if (new.HamIade is null) then New.HamIade = 0;

  if (new.BoyaSiparisDetayId is not null) then
  begin
    Select Kapali from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = New.BoyaSiparisDetayId into :BoyaKapali;
    if (BoyaKapali = 2) then exception SiparisIptalEdilmis;
  end

  if (New.OrmeRafID is not null) then
  begin
    Select DepoRaf.DepoID from OrmeRaf
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      where OrmeRaf.OrmeRafID = New.OrmeRafID
      into New.DepoId;
  end

  if (New.HareketTipi = shtAlis) then New.UreticiFirmaId = New.FirmaId;
  else if ((new.UretimIslemTipiId = uitOrme or new.UretimIslemTipiId = uitBoya) and New.HamIade = 0) then New.UreticiFirmaId = New.FirmaId;

  select max(BitisTarihi) from DonemKapatma where Bolum = 2 and OnayKullaniciId is not Null into :SonDonemBitisTarihi;
  if (SonDonemBitisTarihi is not Null and SonDonemBitisTarihi > new.Tarih) then
    exception DonemKapaliHEklenemez;

  if (DokumaUreCikCozguLotuGuncelle = 1) then
  if (new.HareketTipi = shtUretimeCikis and new.DokumaHareketi = 1 and new.UretimIslemTipiId = uitDokuma and new.HamIade = 0) then
  begin
    select Depo.CozguDeposu, Depo.HasilDeposu, OrmeRaf.CozguIsEmriKayitId from OrmeRaf
    left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
    left join Depo on Depo.DepoId = DepoRaf.DepoId
    where OrmeRaf.OrmeRafId = new.OrmeRafId into :CozguDeposu, :HasilDeposu, :CozguIsEmriKayitId;

    if (CozguDeposu = 1 and CozguIsEmriKayitId is null) then
    begin
      update DokumaPlanlamaRecete set DokumaPlanlamaRecete.Lot = new.Lot
      where DokumaPlanlamaRecete.DokumaPlanlamaId = new.DokumaPlanlamaId and
            DokumaPlanlamaRecete.HamId = new.HamId and
            DokumaPlanlamaRecete.AtkiCozgu = 2;
    end
    else if (CozguDeposu = 1 and CozguIsEmriKayitId is not null) then
    begin
      update DokumaPlanlamaRecete set DokumaPlanlamaRecete.Lot =
            (select CozguIsEmriRecete.Lot from CozguIsEmriRecete where CozguIsEmriRecete.CozguIsEmriKayitId = :CozguIsEmriKayitId and
            CozguIsEmriRecete.HamId = DokumaPlanlamaRecete.HamId and
            IntKarsilastir(CozguIsEmriRecete.RenkId, DokumaPlanlamaRecete.RenkId) = 1)
      where DokumaPlanlamaRecete.DokumaPlanlamaId = new.DokumaPlanlamaId and
            DokumaPlanlamaRecete.AtkiCozgu = 2;
    end
  end

  if (new.HareketTipi = shtUretim) then
  begin
    if ((new.LeventSiraNo is null) and (new.UretimIslemTipiId = uitIndigo) and (new.IndigoIsEmriKayitID is not null)) then
    begin
      select max(StokHareket.LeventSiraNo) from StokHareket
        where StokHareket.IndigoIsEmriKayitID = new.IndigoIsEmriKayitID and StokHareket.HareketTipi = :shtUretim and StokHareket.UretimIslemTipiId = :uitIndigo into :ASira;

      if (ASira is null) then ASira = 0;
      else ASira = ASira + 1;
      new.LeventSiraNo = ASira;
    end

    if (new.LeventSiraNo is not null and new.UretimIslemTipiId = uitCozgu and new.CozguIsEmriKayitID is not null and DokumaProjesiAktif = 2) then
    begin
       select count(*) from StokHareket
          where StokHareket.CozguIsEmriKayitID = new.CozguIsEmriKayitID and StokHareket.HareketTipi = :shtUretim and StokHareket.UretimIslemTipiId = :uitcozgu and
                StokHareket.LeventNo = new.leventno and StokHareket.LeventSiraNo = new.LeventSiraNo into :Ks;
       if (Ks > 0) then exception kayithatasi 'Bu Levent Kaydedilmiş';

    end
    else if (new.LeventSiraNo is null and new.UretimIslemTipiId = uitCozgu and new.CozguIsEmriKayitID is not null) then
    begin
    -- LeventNo,LEventSiraNo
      if (DokumaProjesiAktif = 2) then
      begin
       --COZGUISEMRIDETAY
        select CozguIsEmri.SettekiLeventAdedi from CozguIsEmri where CozguIsEmri.CozguIsEmriKayitId = new.CozguIsEmriKayitID into :SettekiLeventAdedi;
        select CozguIsEmriDetay.LeventAdedi from CozguIsEmriDetay where CozguIsEmriDetay.CozguIsEmriKayitId = new.CozguIsEmriKayitID into :LeventNo;

        if (SettekiLeventAdedi is null) then exception KayitHatasi 'SettekiLeventAdedi Bos olamaz';
        if (LeventNo is null) then exception KayitHatasi 'LeventAdedi Bos Olamaz';


        select max(StokHareket.LeventNo) from StokHareket
          where StokHareket.CozguIsEmriKayitID = new.CozguIsEmriKayitID and StokHareket.HareketTipi = :shtUretim and StokHareket.UretimIslemTipiId = :uitCozgu  into :FLeventNo;

       if (FLeventNo is null) then FLeventNo=0;

       select max(StokHareket.LeventSiraNo) from StokHareket
          where StokHareket.CozguIsEmriKayitID = new.CozguIsEmriKayitID and StokHareket.HareketTipi = :shtUretim and StokHareket.UretimIslemTipiId = :uitcozgu and
                coalesce(StokHareket.LeventNo, 0) = :FLeventNo into :FLeventSiraNo;

       if (FLeventSiraNo is null) then FLeventSiraNo = 0;

       if (FLeventSiraNo < SettekiLeventAdedi)  then
       begin
         FLeventSiraNo = FLeventSiraNo + 1;
         if (FLeventNo = 0) then FLeventNo = FLeventNo + 1;

       end
       else if (FLeventSiraNo = SettekiLeventAdedi ) then
       begin
         FLeventSiraNo = 1;
         FLeventNo = FLeventNo + 1;
         if (FLeventNo > LeventNo) then  exception KayitHatasi 'Sınır Aşıldı';
       end
       new.LeventSiraNo = FLeventSiraNo;
        New.LeventNo = FLeventNo;
     --COZGUISEMRIDETAY
      end
      else
      begin
        select max(StokHareket.LeventSiraNo) from StokHareket
          where StokHareket.CozguIsEmriKayitID = new.CozguIsEmriKayitID and StokHareket.HareketTipi = :shtUretim and StokHareket.UretimIslemTipiId = :uitCozgu  into :ASira;
        if (ASira is null) then ASira = 0;
        else ASira = ASira + 1;
        new.LeventSiraNo = ASira;
      end
    end

    if (new.LeventSiraNo is null and new.UretimIslemTipiId = uitHasil and new.HasilIsEmriKayitID is not null) then
    begin
      select max(StokHareket.LeventSiraNo) from StokHareket
              where StokHareket.HasilIsEmriKayitID = new.HasilIsEmriKayitID and StokHareket.HareketTipi = :shtUretim and StokHareket.UretimIslemTipiId = :uitHasil  into :ASira;
      if (ASira is null) then ASira = 0;
      else ASira = ASira + 1;
      new.LeventSiraNo = ASira;
    end

  end
end
