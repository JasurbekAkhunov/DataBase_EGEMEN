CREATE TRIGGER AFTERINSERTIPLIKIRSALIYEDETAY FOR IPLIKIRSALIYEDETAY
ACTIVE AFTER INSERT POSITION 0
as
declare variable shtAlis integer;
declare variable shtFasonAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtUretimeCikis integer;
declare variable shtUretim integer;
declare variable shtUretimIade integer;
declare variable shtTelefGirisi integer;
declare variable shtUretimBoya  integer;
declare variable shtUretimKirpinti  integer;
declare variable shtBukumeCikis integer;
declare variable shtDepoTransfer integer;
declare variable I        integer;
declare variable OrtalamaMiktar Numeric(12,3);
declare variable OrtalamaNetMiktar Numeric(12,3);
declare variable FarkMiktar Numeric(12,3);
declare variable FarkNetMiktar Numeric(12,3);
declare variable shtSarimaCikis integer;
declare variable KS integer;
declare variable FirmaId                      varchar(20);
declare variable IrsaliyeTarihi               TimeStamp;
declare variable IrsaliyeNo                   IrsaliyeNoSize;
declare variable RenkAdi                      varchar(50);
declare variable GirisYeri                    varchar(50);
Declare Variable hstHamPamuk SmallInt;
Declare Variable HamTipiId Integer;
begin
  Select shtAlis, shtFasonAlis, shtSatis, ShtAlisIade, shtSatisIade, shtUretimeCikis, ShtUretim, shtUretimIade, shtTelefGirisi, shtUretimBoya, shtSarimaCikis, shtBukumeCikis, shtDepoTransfer, shtUretimKirpinti from IplikStokHareketTipiGetir
         into :shtAlis, :shtFasonAlis, :shtSatis, :ShtAlisIade, :shtSatisIade, :shtUretimeCikis, :ShtUretim, :shtUretimIade, :shtTelefGirisi, :shtUretimBoya, :shtSarimaCikis, :shtBukumeCikis, :shtDepoTransfer, :shtUretimKirpinti;


  if (new.HareketTipi = shtAlis or new.HareketTipi = shtFasonAlis or New.HareketTipi = shtUretimBoya or New.HareketTipi = shtUretimKirpinti) then
  begin
    Select HamTipiId from Ham where Ham.HamId = New.HamId into :HamTipiId;
    Select hstHamPamuk from HamTipiGetir into :hstHamPamuk;

    OrtalamaMiktar = New.Miktar / New.KapAdedi;

    OrtalamaMiktar = Yuvarla(OrtalamaMiktar, 100);

    OrtalamaNetMiktar = New.NetMiktar / New.KapAdedi;
    OrtalamaNetMiktar = Yuvarla(OrtalamaNetMiktar, 100);

    I = 1;
    if (HamTipiId = hstHamPamuk) then
    begin
      I = New.KapAdedi;
      OrtalamaMiktar    = New.Miktar;
      OrtalamaNetMiktar = New.NetMiktar;
    end

    While (I <= New.KapAdedi) do
    begin
      if (New.HareketTipi <> shtDepoTransfer) then
      begin
        insert into IplikStok(IplikIrsaliyeId, IplikIrsaliyeDetayId, KapNo, Miktar, NetMiktar)
                     Values(New.IplikIrsaliyeId, New.IplikIrsaliyeDetayId, :I, :OrtalamaMiktar, :OrtalamaNetMiktar);
      end
      I = I + 1;
    end

    if (HamTipiId <> hstHamPamuk) then
    begin
      FarkMiktar = OrtalamaMiktar * New.KapAdedi - New.Miktar;
      FarkNetMiktar = OrtalamaNetMiktar * New.KapAdedi - New.NetMiktar;

      if (FarkMiktar <> 0 or FarkNetMiktar <> 0) then
      begin
        --if (New.HareketTipi <> shtDepoTransfer) then
          Update IplikStok Set Miktar = Miktar - :FarkMiktar, NetMiktar = NetMiktar - :FarkNetMiktar where IplikStok.IplikIrsaliyeDetayID = New.IplikIrsaliyeDetayID and IplikStok.KapNo = new.KapAdedi;
      end
    end

    if (New.VerilenSiparisDetayId is not null) then
      Update VerilenSiparisDetay Set GelenMiktar = GelenMiktar + New.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = new.VerilenSiparisDetayId;

  end
  else if (new.HareketTipi = shtAlisIade) then
  begin
    if (New.VerilenSiparisDetayId is not null) then
      Update VerilenSiparisDetay Set IadeEdilenMiktar = IadeEdilenMiktar + New.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = new.VerilenSiparisDetayId;
  end
  else if (new.HareketTipi = ShtSatis) then
  begin
    if (New.IplikTalepDetayID is not null) then
    begin
      if (New.Tamir = 0) then
         Update IplikTalepDetay Set
                TeslimEdilenMiktar = TeslimEdilenMiktar + new.Miktar,
                TeslimEdilenNetMiktar = TeslimEdilenNetMiktar + new.NetMiktar,
                TeslimEdilenBobinSayisi = TeslimEdilenBobinSayisi + new.BobinSayisi,
                LinkIslemi = 1
         where IplikTalepDetay.IplikTalepDetayID = New.IplikTalepDetayId;
      else
         Update IplikTalepDetay Set
                TamirTeslimEdilenMiktar = TamirTeslimEdilenMiktar + new.Miktar,
                TamirTeslimEdilenNetMiktar = TamirTeslimEdilenNetMiktar + new.NetMiktar,
                TamirTeslimEdilenBobinSayisi = TamirTeslimEdilenBobinSayisi + new.BobinSayisi,
                LinkIslemi = 1
         where IplikTalepDetay.IplikTalepDetayID = New.IplikTalepDetayId;
    end
  end
  else if (new.HareketTipi = ShtSatisIade) then
  begin
    if (New.IplikTalepDetayID is not null) then
      Update IplikTalepDetay Set IadeGelenMiktar = IadeGelenMiktar + new.Miktar, LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = New.IplikTalepDetayId;
  end
  else if (New.HareketTipi = shtUretimeCikis and New.IplikTelefIsEmriDetayId is not null) then
  begin
    Update IplikTelefIsEmriDetay set CikisMiktari = CikisMiktari + New.Miktar where IplikTelefIsEmriDetay.IplikTelefIsEmriDetayID = New.IplikTelefIsEmriDetayId;
  end
  else if (New.HareketTipi = shtSarimaCikis) then
  begin
    Select Count(*) from SarimStok where SarimStok.IplikLotId = New.IplikLotId into :KS;

    if (KS > 0) then
    begin
      Update SarimStok set KapAdedi = KapAdedi + new.KapAdedi, Miktar = Miktar + new.Miktar where SarimStok.IplikLotId = New.IplikLotId;
    end
    else
    begin
      insert into SarimStok(IplikLotId, KapAdedi, Miktar) Values(new.IplikLotId, new.KapAdedi, new.Miktar);
    end
  end
  else if (New.HareketTipi = shtBukumeCikis) then
  begin
    Update BukumIsEmri set CikisMiktari = CikisMiktari + New.Miktar, LinkIslemi = 1 where BukumIsEmri.BukumIsEmriKayitID = New.BukumIsEmriKayitId;
  end

  if (new.TamirGirisKayitId is not Null) then
  begin
    select Renk.Adi from Renk where Renk.RenkId = new.RenkId into :RenkAdi;
    if (new.IplikIrsaliyeId is not Null) then
    begin
      GirisYeri = 'İplik Sevk İrsaliyesi';
      select FirmaId, IrsaliyeNo, IrsaliyeTarihi from IplikIrsaliye where IplikIrsaliye.IplikIrsaliyeId = new.IplikIrsaliyeId into :FirmaId, :IrsaliyeNo, :IrsaliyeTarihi;
    end
    execute procedure TamirGirisUyariMailiGonder(:FirmaId, :IrsaliyeTarihi, :IrsaliyeNo, new.HamId, :RenkAdi, new.Miktar, :GirisYeri);
  end
end
