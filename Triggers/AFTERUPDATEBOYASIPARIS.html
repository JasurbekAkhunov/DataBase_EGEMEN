Definition

CREATE TRIGGER AFTERUPDATEBOYASIPARIS FOR BOYASIPARIS
ACTIVE AFTER UPDATE POSITION 0
as
declare variable PSKKEtiketTipiSiparisDetaydan Integer;
begin
  if (Old.Fason <> New.Fason) then
    Update BoyaSiparisDetay Set FaturaEdilecekMiktarTipi = Null, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisKayitId = New.BoyaSiparisKayitId;

  if (Old.IslemTipiId <> New.IslemTipiId) then
    Update BoyaSiparisDetay Set BoyaSiparisDetay.IslemTipiId = New.IslemTipiId, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisKayitId = New.BoyaSiparisKayitId and BoyaSiparisDetay.IslemTipiID = Old.IslemTipiId;

  if (Old.FirmaId <> New.FirmaId) then
    Update BoyaSiparisDetay Set FirmaSevkAdresiId = (Select FirmaSevkAdresi.FirmaSevkAdresiId from FirmaSevkAdresi where FirmaSevkAdresi.FirmaID = New.FirmaId and FirmaSevkAdresi.DefaultAdres = 1), LinkIslemi = 1
       where BoyaSiparisDetay.BoyaSiparisKayitId = new.BoyaSiparisKayitId;

  if (IntKarsilastir(Old.OdemeSekliId, New.OdemeSekliId) = 0) then
  begin
    if (New.OdemeSekliId is not null) then
    begin
      Update OrmeSiparisDetay set OdemeSekliID = New.OdemeSekliId, LinkIslemi = 1 where OrmeSiparisDetay.BoyaSiparisKayitId = New.BoyaSiparisKayitId;

      Update BoyaSiparisDetay set OdemeSekliID = New.OdemeSekliId, KullaniciId = New.KullaniciId, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisKayitId = New.BoyaSiparisKayitId and BoyaSiparisDetay.OrmeSiparisDetayId is null;
    end
  end

  if (old.Komisyonlu <> new.Komisyonlu or old.KomisyonOrani <> new.KomisyonOrani) then
    update BoyaSiparisDetay set LinkIslemi = 1, FiyatKomisyon = new.Komisyonlu, FiyatKomisyonOrani = iif(old.KomisyonOrani <> new.KomisyonOrani, new.KomisyonOrani, FiyatKomisyonOrani)
    where BoyaSiparisDetay.BoyaSiparisKayitId = new.BoyaSiparisKayitId;

  if (IntKarsilastir(new.IslemTipiID, Old.IslemTipiID) = 0) then
    Update Parti Set IslemTipiID = New.IslemTipiID where BoyaSiparisKayitID = New.BoyaSiparisKayitID;

  If (Coalesce(Old.PSKKEtiketTipiId,0) <> Coalesce(New.PSKKEtiketTipiId,0)) Then
  Begin
    Select PSKKEtiketTipiSiparisDetaydan From Sistem Where SistemID = 1 Into :PSKKEtiketTipiSiparisDetaydan;
    If (PSKKEtiketTipiSiparisDetaydan = 0) Then
        Update BoyaSiparisDetay Set PSKKEtiketTipiId = New.PSKKEtiketTipiId Where BoyaSiparisDetay.BoyaSiparisKayitId = New.BoyaSiparisKayitId;
  End

  Execute Procedure Logla('U', 2, 'BOYASIPARIS', 'BOYASIPARISKAYITID', new.BoyaSiparisKayitId, 'Boya Sipariş Yıl :'||new.BoyaSiparisYil||' Boya Sipariş Id :'||new.BoyaSiparisId||' Firma :'||new.FirmaId);
end
