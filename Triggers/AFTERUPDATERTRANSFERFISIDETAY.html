CREATE TRIGGER AFTERUPDATERTRANSFERFISIDETAY FOR RAFTRANSFERFISIDETAY
ACTIVE AFTER UPDATE POSITION 0
as
Declare variable RTFDetayGuncelleme integer;
begin
  RTFDetayGuncelleme = Coalesce(rdb$get_context('USER_TRANSACTION', 'RTFDetayGuncelleme'), 0);

  if (RTFDetayGuncelleme = 0) then
  begin
    Update RafDetay Set RafTransferKilo = RafTransferKilo - Old.Kilo + New.kilo,
                        RafTransferMetre = RafTransferMetre - Old.Metre + New.Metre,
                        RafTransferKapAdedi = RafTransferKapAdedi - Old.KapAdedi + New.KapAdedi
    Where RafDetay.RafDetayId = New.RafDetayId;

    Update RafDetay Set KaliteNo = New.KaliteNo, Kilo = New.Kilo, Metre = New.Metre, KapAdedi = New.KapAdedi, LinkIslemi = 1
    Where RafDetay.RafTransferFisiId = New.RafTransferFisiId and RafDetay.RafId = New.RafId;
  end
end
