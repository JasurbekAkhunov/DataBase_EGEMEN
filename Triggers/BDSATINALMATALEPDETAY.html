CREATE TRIGGER BDSATINALMATALEPDETAY FOR SATINALMATALEPDETAY
ACTIVE BEFORE DELETE POSITION 0
as
declare variable KayitSayisi integer;
declare variable SATOnayDurumu integer;
begin
  Select Count(*) from VerilenSiparisDetay where VerilenSiparisDetay.SatinAlmatalepDetayId = Old.SatinAlmatalepDetayId into :KayitSayisi;
  if (KayitSayisi > 0) then exception SATSiparisDonusmusSilinemez;

  Select Count(*) from SatinAlmaTalepOnay
     where SatinAlmaTalepOnay.SatinAlmaTalepKayitId = Old.SatinAlmaTalepKayitId and SatinAlmaTalepOnay.OnayDurumu <> 0 into :KayitSayisi;

  if (KayitSayisi = 1) then exception SATOnaylanmisDegisemez;
end
