CREATE TRIGGER BEFOREDELETEOZELALANTIPI FOR OZELALANTIPI
ACTIVE BEFORE DELETE POSITION 0
as
Declare Variable KS Integer;
begin
    Select Count(*) From OzelAlan Where OzelAlan.OzelAlanTipiId = Old.OzelAlanTipiId into :KS;
    if (KS > 0) then exception KayitHatasi 'Bu Tipe Ait Özel Alan Kaydı Var, Tip Silinemez..';

	Delete From OzelAlanTipiDetay Where OzelAlanTipiDetay.OzelAlanTipiId = Old.OzelAlanTipiId;
end
