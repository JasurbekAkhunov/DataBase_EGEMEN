Definition

CREATE TRIGGER BEFOREDELETEUNITEIRSALIYE FOR UNITEIRSALIYE
ACTIVE BEFORE DELETE POSITION 0
as
declare variable OnayDurumu integer;
begin
  if (Old.TransferEdildi = 1) then exception TrasferEdilmisDegistirilmez;

  if (Old.KantarKayitId is not null) then
  begin
    Select OnayDurumu from Kantar where Kantar.KantarKayitID = Old.KantarKayitId into :OnayDurumu;
    if (OnayDurumu = 1) then exception KantarOnayiVerilmisDegisemez;
  end
  delete from UniteIrsaliyeDetay where UniteIrsaliyeDetay.UniteIrsaliyeId = Old.UniteIrsaliyeId;
end
