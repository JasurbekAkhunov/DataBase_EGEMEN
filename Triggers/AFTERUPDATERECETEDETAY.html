CREATE TRIGGER AFTERUPDATERECETEDETAY FOR RECETEDETAY
ACTIVE AFTER UPDATE POSITION 0
as
declare variable OldUrunId  varchar(20);
declare variable NewUrunId  varchar(20);
declare variable OldReceteAsamaId integer;
declare variable NewReceteAsamaId integer;
begin
  OldUrunId = Old.UrunId;
  NewUrunId = New.UrunId;
  if (OldUrunId is null) then OldUrunId = '';
  if (NewUrunId is null) then NewUrunId = '';

  OldReceteAsamaId = Old.ReceteAsamaId;
  NewReceteAsamaId = New.ReceteAsamaId;
  if (OldReceteAsamaId is null) then OldReceteAsamaId = 0;
  if (NewReceteAsamaId is null) then NewReceteAsamaId = 0;

end
