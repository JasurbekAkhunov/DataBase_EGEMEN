CREATE TRIGGER BEFOREINSERTLABRECETE4 FOR LABRECETE
ACTIVE BEFORE INSERT POSITION 4
as
declare variable LabReceteFirmaBaslangic   integer;
declare variable LabReceteFirmaDigit       integer;
declare variable LabReceteAyraci           varchar(1);
declare variable MaxNumara                 integer;
declare variable NumaraStr                 varchar(20);
declare variable MaxLen                    integer;
declare variable FLabReceteKodu            varchar(20);
declare variable NumaraDigit               integer;
declare variable LabReceteKoduTipi         integer;
declare variable TurId                     varchar(4);
declare variable DerinlikId                varchar(2);
declare variable GrupId                    varchar(2);
declare variable TF     integer;
declare variable YilStr                    varchar(4);
declare variable LabReceteGrupStr          varchar(2);
declare variable ReceteHamTipiId           integer;
declare variable HamTipiStr                varchar(1);
declare variable IkiHaneYil                varchar(2);
declare variable I integer;
declare variable RenkDerinligiIcinX        varchar(1);
declare variable RenkDerinligiAdi          varchar(20);
declare variable EgemenFirmaID             integer;
declare variable YilAyStr                  integer;
declare variable ReceteTurId               integer;
declare variable FHaslik                   varchar(2);
declare variable OzelStr                   varchar(5);
begin
  Select LabReceteFirmaBaslangic, LabReceteFirmaDigit, LabReceteAyraci, LabReceteKoduTipi, EgemenFirmaID from Sistem into :LabReceteFirmaBaslangic, :LabReceteFirmaDigit, :LabReceteAyraci, :LabReceteKoduTipi, :EgemenFirmaID;
  if (LabReceteKoduTipi = 153) then /* LiyaTeks */
  begin
    Select GrupID from LabRenkGrup where LabRenkGrup.LabRenkGrupID = new.LabRenkGrupID into :GrupID;
    Select TurId  from ReceteTuru where  ReceteTuru.ReceteTuruId   = New.ReceteTuruId  into :TurID;

    Select max(Numara) from LabRecete where LabRecete.LabRenkGrupID = new.LabRenkGrupId and Labrecete.ReceteTuruId = New.ReceteTuruId into :MaxNumara;

    if (MaxNumara is null) then
    begin
      if      (New.LabRenkGrupID = 0) then new.Numara = 0001;
      else if (New.LabRenkGrupID = 1) then new.Numara = 1001;
      else if (New.LabRenkGrupID = 2) then new.Numara = 2001;
      else if (New.LabRenkGrupID = 3) then new.Numara = 3001;
      else if (New.LabRenkGrupID = 4) then new.Numara = 4001;
      else if (New.LabRenkGrupID = 5) then new.Numara = 5001;
      else if (New.LabRenkGrupID = 6) then new.Numara = 6001;
      else if (New.LabRenkGrupID = 7) then new.Numara = 7001;
      else if (New.LabRenkGrupID = 8) then new.Numara = 8001;
      else if (New.LabRenkGrupID = 9) then new.Numara = 9001;
    end
    else
      new.Numara = MaxNumara + 1;

  if (New.LabRenkGrupID = 0) then
      New.LabReceteKodu = 'L' || TurId || LabReceteAyraci  || EgemenIntToLzStr(new.Numara, 4);
    else
      New.LabReceteKodu = 'L' || TurId || LabReceteAyraci  || EgemenIntToLzStr(new.Numara, 3);

  end
  else if (LabReceteKoduTipi = 154) then /* Madiyo*/
  begin
    Select TurId from ReceteTuru where ReceteTuru.ReceteTuruID = new.ReceteTuruId into :TurId;
    Select DerinlikId from RenkDerinligi where RenkDerinligi.RenkDerinligiID = new.RenkDerinligiId into :DerinlikId;

    IkiHaneYil = EgemenIntToLzStr((f_Year(new.TalepTarihi) - 2000), 2);

    if (New.LabRenkGrupID < 0 and New.LabRenkGrupID > 9) then exception KayitHatasi 'Renk Grup Hatalı...';

    if (New.LabReceteKodu is null) then
    begin
      Select max(Numara) from LabRecete
        where f_Year(LabRecete.TalepTarihi) = f_Year(New.TalepTarihi) and
              LabRecete.LabRenkGrupID = new.LabRenkGrupId and
              LabRecete.ReceteTuruId =     new.ReceteTuruId and
              ((LabRecete.LabRenkGrupID = 1 and LabRecete.Numara >= 10040) or
               (LabRecete.LabRenkGrupID = 2 and LabRecete.Numara >= 20040) or
               (LabRecete.LabRenkGrupID = 3 and LabRecete.Numara >= 30040) or
               (LabRecete.LabRenkGrupID = 4 and LabRecete.Numara >= 40040) or
               (LabRecete.LabRenkGrupID = 5 and LabRecete.Numara >= 50040) or
               (LabRecete.LabRenkGrupID = 6 and LabRecete.Numara >= 60040) or
               (LabRecete.LabRenkGrupID = 7 and LabRecete.Numara >= 70040) or
               (LabRecete.LabRenkGrupID = 8 and LabRecete.Numara >= 80040) or
               (LabRecete.LabRenkGrupID = 9 and LabRecete.Numara >= 90040) or
               (LabRecete.LabRenkGrupID = 0 and LabRecete.Numara >= 00040))
        into :MaxNumara;

      if (MaxNumara is not null) then
        new.Numara = MaxNumara + 1;
      else
      begin
        if (New.LabRenkGrupID = 1) then new.Numara = 10040;
        else if (New.LabRenkGrupID = 2) then new.Numara = 20040;
        else if (New.LabRenkGrupID = 3) then new.Numara = 30040;
        else if (New.LabRenkGrupID = 4) then new.Numara = 40040;
        else if (New.LabRenkGrupID = 5) then new.Numara = 50040;
        else if (New.LabRenkGrupID = 6) then new.Numara = 60040;
        else if (New.LabRenkGrupID = 7) then new.Numara = 70040;
        else if (New.LabRenkGrupID = 8) then new.Numara = 80040;
        else if (New.LabRenkGrupID = 9) then new.Numara = 90040;
        else if (New.LabRenkGrupID = 0) then new.Numara = 00040;
      end

      New.LabReceteKodu = IkiHaneYil  || DerinlikId || LabReceteAyraci || TurId  || EgemenIntToLzStr(new.Numara, 5);

    end
    else
    begin
      NumaraStr = EgemenCopyStr(New.LabReceteKodu, Char_Length(new.LabReceteKodu)-4, 5);

      TF = EgemenStrToIntOlurmu(NumaraStr);

      if (TF = 1) then
      begin
        new.Numara = EgemenStrToInt(NumaraStr);

        if ( (new.LabRenkGrupID = 1 and (New.Numara < 10000 or New.Numara >= 10040)) or
             (new.LabRenkGrupID = 2 and (New.Numara < 20000 or New.Numara >= 20040)) or
             (new.LabRenkGrupID = 3 and (New.Numara < 30000 or New.Numara >= 30040)) or
             (new.LabRenkGrupID = 4 and (New.Numara < 40000 or New.Numara >= 40040)) or
             (new.LabRenkGrupID = 5 and (New.Numara < 50000 or New.Numara >= 50040)) or
             (new.LabRenkGrupID = 6 and (New.Numara < 60000 or New.Numara >= 60040)) or
             (new.LabRenkGrupID = 7 and (New.Numara < 70000 or New.Numara >= 70040)) or
             (new.LabRenkGrupID = 8 and (New.Numara < 80000 or New.Numara >= 80040)) or
             (new.LabRenkGrupID = 9 and (New.Numara < 90000 or New.Numara >= 90040)) or
             (new.LabRenkGrupID = 0 and (New.Numara < 0 or New.Numara >= 00040)) ) then exception ManuelGirilenLabRecKoduHatali ' Numara =' || EgemenIntegerToStr(New.Numara);
      end
      else exception ReceteNumaraOlusturulamadi;

      FLabReceteKodu = IkiHaneYil  || DerinlikId || LabReceteAyraci || TurId  || EgemenIntToLzStr(new.Numara, 5);

      if (FLabReceteKodu is null or FLabReceteKodu <> New.LabReceteKodu) then
       exception ManuelGirilenLabRecKoduHatali ' FRK =' || FLabReceteKodu || '- N.RK=' || New.LabReceteKodu;
    end
  end else if (LabReceteKoduTipi = 155) then /* ezoznur */
  begin
  --NUR-FIRMAID-TURID-LABRENKGRUP
  ---NUR-A001-C-20001

    ozelstr = 'NUR';
    select TurId from ReceteTuru where ReceteTuru.ReceteTuruID = new.ReceteTuruId into :TurId;

    if (new.LabReceteKodu is null) then
    begin
      select max(Numara) from LabRecete
        where LabRecete.FirmaID = new.FirmaId and
             LabRecete.ReceteTuruId = new.ReceteTuruId and
             LabRecete.LabRenkGrupId = new.LabRenkGrupId into :MaxNumara;

      if (MaxNumara is null) then new.Numara = 1;
      else new.Numara = MaxNumara + 1;

      new.LabReceteKodu = OzelStr || LabReceteAyraci || new.firmaid || LabReceteAyraci || TurId || EgemenIntegerToStr(new.LabRenkGrupId) || EgemenIntToLzStr(new.Numara, 3);
    end
    else
    begin
      NumaraStr = EgemenCopyStr(new.LabReceteKodu, 1, 1);

      NumaraDigit = char_length(OzelStr) + char_length(turId) + char_length(new.firmaid) + char_length(EgemenIntegerToStr(new.LabRenkGrupId)) + 2;
     -- MaxLen = char_length(new.LabReceteKodu) - NumaraDigit + 1;
     -- NumaraDigit = NumaraDigit + 1;
      NumaraStr = EgemenCopyStr(new.LabReceteKodu, NumaraDigit + 1, 3);

      TF = EgemenStrToIntOlurmu(NumaraStr);

      if (TF = 1) then new.Numara = EgemenStrToInt(NumaraStr);
      else exception ReceteNumaraOlusturulamadi;



        FLabReceteKodu = OzelStr || LabReceteAyraci || new.firmaid || LabReceteAyraci || TurId ||
                EgemenIntegerToStr(new.LabRenkGrupId) || EgemenIntToLzStr(new.Numara, 3);

      if (FLabReceteKodu is null or FLabReceteKodu <> new.LabReceteKodu) then
        exception ManuelGirilenLabRecKoduHatali ' Recete Kodu =' || FLabReceteKodu;
    end
  end else if (  LabReceteKoduTipi = 114 ) then /*Else*/
  begin
    select LabRenkGrupID from LabRenkGrup where LabRenkGrup.LabRenkGrupID = new.LabRenkGrupID into :GrupID;
    select TurId from ReceteTuru where ReceteTuru.ReceteTuruId = new.ReceteTuruId into :TurID;
    IkiHaneYil = EgemenIntToLzStr((f_Year(new.TalepTarihi) - 2000), 2);

    if (new.LabReceteKodu is null) then
    begin
      select max(Numara) from LabRecete
        where f_Year(LabRecete.TalepTarihi) = f_Year(new.TalepTarihi) and
              LabRecete.LabReceteGrupId = new.LabReceteGrupId and
              LabRecete.ReceteTuruId = new.ReceteTuruId
        into :MaxNumara;

        if (MaxNumara is null) then new.Numara = 1; else new.Numara = MaxNumara + 1;

        if (new.LabGrupId is null) then
          new.LabReceteKodu = IkiHaneYil || TurId  || GrupID || EgemenIntToLzStr(new.Numara, 5);
    end
  end
  else if (LabReceteKoduTipi = 157) then /* Sadırlar Çorlu */
  begin
     select LabRenkGrupID from LabRenkGrup where LabRenkGrup.LabRenkGrupID = new.LabRenkGrupID into :GrupID;
     select TurId from ReceteTuru where ReceteTuru.ReceteTuruId = new.ReceteTuruId into :TurID;
     IkiHaneYil = EgemenIntToLzStr((f_Year(new.TalepTarihi) - 2000), 2);
     if (new.LabReceteKodu is null) then
     begin
      select max(Numara) from LabRecete
        where f_Year(LabRecete.TalepTarihi) = f_Year(new.TalepTarihi)
        into :MaxNumara;

        if (MaxNumara is null) then new.Numara = 1; else new.Numara = MaxNumara + 1;
        new.LabReceteKodu = IkiHaneYil || TurId || GrupID  || EgemenIntToLzStr(new.Numara, 5);
     end
  end
  else if (LabReceteKoduTipi = 159) then /*Feyteks*/
  begin
    -- Yıl-5 Hane
    IkiHaneYil = EgemenIntToLzStr((f_Year(new.TalepTarihi) - 2000), 2);

    if (New.LabReceteKodu is null) then
    begin
      Select max(Numara) from LabRecete where f_Year(LabRecete.TalepTarihi) = f_Year(New.TalepTarihi)  into :MaxNumara;

      if (MaxNumara is null) then new.Numara = 1; else new.Numara = MaxNumara + 1;
      New.LabReceteKodu = IkiHaneYil || LabReceteAyraci || EgemenIntToLzStr(new.Numara, 5);
    end
    else
    begin
      FLabReceteKodu  = IkiHaneYil || LabReceteAyraci ;
      NumaraDigit     = Char_Length(FLabReceteKodu) ;
      MaxLen          = Char_Length(new.LabReceteKodu) - NumaraDigit +1;
      NumaraStr       = EgemenCopyStr(New.LabReceteKodu, NumaraDigit+1, MaxLen-1);

      TF = EgemenStrToIntOlurmu(NumaraStr);

      if (TF = 1) then new.Numara = EgemenStrToInt(NumaraStr);
      else exception ReceteNumaraOlusturulamadi;

      FLabReceteKodu = FLabReceteKodu ||   EgemenIntToLzStr(new.Numara,5)  ;

      if (FLabReceteKodu is null or FLabReceteKodu <> New.LabReceteKodu) then
        exception ManuelGirilenLabRecKoduHatali ' FRK =' || FLabReceteKodu || '- N.RK' || New.LabReceteKodu;
    end
  end
end
