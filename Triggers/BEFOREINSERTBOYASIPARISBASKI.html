CREATE TRIGGER BEFOREINSERTBOYASIPARISBASKI FOR BOYASIPARISBASKI
ACTIVE BEFORE INSERT POSITION 0
as
declare variable FDesenId varchar(20);
declare variable FKapali integer;
begin
  Select DesenId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayId = new.BoyaSiparisDetayId into New.DesenId;

  Select DesenId, Kapali from DesenVaryant where DesenVaryant.DesenVaryantID = New.DesenVaryantId into :FDesenId, :FKapali;
  if (FDesenId <> New.DesenId) then exception DesenVaryantLinkHatasi;
  if (FKapali = 1) then exception KayitHatasi 'Desen Varyant Kapali';

  new.BoyaSiparisBaskiID = Gen_Id(BoyaSiparisBaski_Id_Gen, 1);
end
