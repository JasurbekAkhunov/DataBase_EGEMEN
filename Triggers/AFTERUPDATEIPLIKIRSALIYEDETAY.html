Definition

CREATE TRIGGER AFTERUPDATEIPLIKIRSALIYEDETAY FOR IPLIKIRSALIYEDETAY
ACTIVE AFTER UPDATE POSITION 0
as
declare variable degisti integer;
declare variable shtAlis integer;
declare variable shtFasonAlis integer;
declare variable shtSatis integer;
declare variable shtSatisIade integer;
declare variable shtUretimeCikis integer;
declare variable shtUretimBoya   integer;
declare variable shtUretimKirpinti   integer;
declare variable shtSarimaCikis  integer;
declare variable shtBukumeCikis  integer;
declare variable shtAlisIade     integer;
declare variable OrtalamaMiktar Numeric(12,3);
declare variable OrtalamaNetMiktar Numeric(12,3);
declare variable I integer;
declare variable FarkMiktar Numeric(12,3);
declare variable FarkNetMiktar Numeric(12,3);
declare variable FIplikStokID integer;
declare variable FOldBobinSayisi integer;
declare variable shtDepoTransfer  integer;
declare variable FIplikIrsaliyeDetayId integer;
Declare Variable hstHamPamuk Integer;
Declare Variable HamTipiId   Integer;
declare variable FMiktar Numeric(12,3);
declare variable FNetMiktar Numeric(12,3);
begin
  FOldBobinSayisi = Old.BobinSayisi;
  if (FOldBobinSayisi is null) then FOldBobinSayisi = 0;

  Select shtAlis, shtFasonAlis, shtAlisIade, shtSatis, shtSatisIade, shtUretimeCikis, shtUretimBoya, shtSarimaCikis, shtBukumeCikis, shtDepoTransfer, shtUretimKirpinti from IplikStokHareketTipiGetir into :shtAlis, :shtFasonAlis, :shtAlisIade, shtSatis, :shtSatisIade, :shtUretimeCikis, :shtUretimBoya, :shtSarimaCikis, :shtBukumeCikis, :shtDepoTransfer, :shtUretimKirpinti;

  if (new.HareketTipi = shtAlis or new.HareketTipi = shtFasonAlis or New.HareketTipi = shtUretimBoya or New.HareketTipi = shtUretimKirpinti) then
  begin
    Select HamTipiId from Ham where Ham.HamId = New.HamId into :HamTipiId;
    Select hstHamPamuk from HamTipiGetir into :hstHamPamuk;

    if (HamTipiId = hstHamPamuk) then
    begin
        Update IplikStok Set Miktar = new.Miktar, NetMiktar = new.NetMiktar, KapNo = New.KapAdedi where IplikStok.IplikIrsaliyeDetayId = New.IplikIrsaliyeDetayId;
    end
    else
    begin
      if (old.KapAdedi > new.KapAdedi) then
         delete from IplikStok where IplikStok.IplikIrsaliyeDetayId = New.IplikIrsaliyeDetayId and IplikStok.KapNo > new.KapAdedi;

      if (old.KapAdedi <> new.KapAdedi or Old.Miktar <> New.Miktar or Old.NetMiktar <> New.NetMiktar) then
      begin
        OrtalamaMiktar = New.Miktar / New.KapAdedi;

        OrtalamaMiktar = Yuvarla(OrtalamaMiktar, 100);

        OrtalamaNetMiktar = New.NetMiktar / New.KapAdedi;

        OrtalamaNetMiktar = Yuvarla(OrtalamaNetMiktar, 100);


        I = 1;

        While (I <= New.KapAdedi) do
        begin
          if (I <= Old.KapAdedi) then
          begin
            Update IplikStok Set Miktar = :OrtalamaMiktar, NetMiktar = :OrtalamaNetMiktar
              where IplikStok.IplikIrsaliyeDetayId = New.IplikIrsaliyeDetayId and IplikStok.KapNO = :I;
          end
          else
          begin
            FIplikStokID = NULL;
            if (New.HareketTipi = shtDepoTransfer) then
            begin
              Select IplikStokID, IplikStok.Miktar, iplikstok.NetMiktar from IplikStok where IplikStok.CikisIplikIrsaliyeDetayId = New.IplikIrsaliyeDetayId and transfer=1
                into :FIplikStokID, :FMiktar, :FNetMiktar;
              if (FMiktar is null) then FMiktar = OrtalamaMiktar;
              if (FNetMiktar is null) then FNetMiktar = OrtalamaNetMiktar;

              insert into IplikStok(IplikIrsaliyeId, IplikIrsaliyeDetayId, KapNo, Miktar, NetMiktar,  iplikStokId)
                            Values(New.IplikIrsaliyeId, New.IplikIrsaliyeDetayId, :I, :FMiktar, :FNetMiktar, :FIplikStokID);
            end
            else
            begin
              insert into IplikStok(IplikIrsaliyeId, IplikIrsaliyeDetayId, KapNo, Miktar, NetMiktar)
                           Values(New.IplikIrsaliyeId, New.IplikIrsaliyeDetayId, :I, :OrtalamaMiktar, :OrtalamaNetMiktar);
             end
          end
          I = I + 1;
        end

        FarkMiktar = OrtalamaMiktar * New.KapAdedi - New.Miktar;
        FarkNetMiktar = OrtalamaNetMiktar * New.KapAdedi - New.NetMiktar;

        if (FarkMiktar <> 0 or FarkNetMiktar <> 0) then
        begin
          Select Max(IplikStokID) from IplikStok
            where IplikStok.IplikIrsaliyeDetayID = New.IplikIrsaliyeDetayID and IplikStok.CikisIplikIrsaliyeDetayID is null
            into :FIplikStokID;
          if (New.HareketTipi <> shtDepoTransfer) then
             Update IplikStok Set Miktar = Miktar - :FarkMiktar, NetMiktar = NetMiktar - :FarkNetMiktar where IplikStok.IplikStokID = :FIplikStokID;
        end
      end
    end

    if (Old.VerilenSiparisDetayId is not null) then
      Update VerilenSiparisDetay Set GelenMiktar = GelenMiktar - Old.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = Old.VerilenSiparisDetayId;

    if (New.VerilenSiparisDetayId is not null) then
      Update VerilenSiparisDetay Set GelenMiktar = GelenMiktar + New.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = New.VerilenSiparisDetayId;

  end
  else if (new.HareketTipi = shtAlisIade) then
  begin
    if (Old.VerilenSiparisDetayId is not null) then
      Update VerilenSiparisDetay Set IadeEdilenMiktar = IadeEdilenMiktar - Old.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = Old.VerilenSiparisDetayId;

    if (New.VerilenSiparisDetayId is not null) then
      Update VerilenSiparisDetay Set IadeEdilenMiktar = IadeEdilenMiktar + New.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = new.VerilenSiparisDetayId;
  end
  else if (new.HareketTipi = ShtSatis) then
  begin
    if (IntKarsilastir(Old.IplikTalepDetayID, New.IplikTalepDetayID) = 0) then
    begin
       if (Old.Tamir = 0) then
       begin
         if (Old.IplikTalepDetayID is not null) then
           Update IplikTalepDetay Set TeslimEdilenMiktar = TeslimEdilenMiktar - Old.Miktar,
                                      TeslimEdilenNetMiktar = TeslimEdilenNetMiktar - Old.NetMiktar,
                                      TeslimEdilenBobinSayisi = TeslimEdilenBobinSayisi - :FOldBobinSayisi,
                                      LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = Old.IplikTalepDetayId;

         if (new.IplikTalepDetayID is not null) then
           Update IplikTalepDetay Set TeslimEdilenMiktar = TeslimEdilenMiktar + new.Miktar,
                                      TeslimEdilenNetMiktar = TeslimEdilenNetMiktar + new.NetMiktar,
                                      TeslimEdilenBobinSayisi = TeslimEdilenBobinSayisi + new.BobinSayisi,
                                      LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = New.IplikTalepDetayId;
       end
       else
       begin
         if (Old.IplikTalepDetayID is not null) then
           Update IplikTalepDetay Set TamirTeslimEdilenMiktar = TamirTeslimEdilenMiktar - Old.Miktar,
                                      TamirTeslimEdilenNetMiktar = TamirTeslimEdilenNetMiktar - Old.NetMiktar,
                                      TamirTeslimEdilenBobinSayisi = TamirTeslimEdilenBobinSayisi - :FOldBobinSayisi,
                                      LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = Old.IplikTalepDetayId;

         if (new.IplikTalepDetayID is not null) then
           Update IplikTalepDetay Set TamirTeslimEdilenMiktar = TamirTeslimEdilenMiktar + new.Miktar,
                                      TamirTeslimEdilenNetMiktar = TamirTeslimEdilenNetMiktar + new.NetMiktar,
                                      TamirTeslimEdilenBobinSayisi = TamirTeslimEdilenBobinSayisi + new.BobinSayisi,
                                      LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = New.IplikTalepDetayId;
       end
    end
    else
    begin
      if (New.IplikTalepDetayID is not null) then
      begin
        if (new.Tamir = 0) then
        begin
           Update IplikTalepDetay Set TeslimEdilenMiktar = TeslimEdilenMiktar - Old.Miktar + new.Miktar,
                                      TeslimEdilenNetMiktar = TeslimEdilenNetMiktar - Old.NetMiktar + new.NetMiktar,
                                      TeslimEdilenBobinSayisi = TeslimEdilenBobinSayisi - :FOldBobinSayisi + new.BobinSayisi,
                                      LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = New.IplikTalepDetayId;
        end
        else
        begin
          Update IplikTalepDetay Set TamirTeslimEdilenMiktar = TamirTeslimEdilenMiktar - Old.Miktar + new.Miktar,
                                     TamirTeslimEdilenNetMiktar = TamirTeslimEdilenNetMiktar - Old.NetMiktar + new.NetMiktar,
                                     TamirTeslimEdilenBobinSayisi = TamirTeslimEdilenBobinSayisi - :FOldBobinSayisi + new.BobinSayisi,
                                     LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = New.IplikTalepDetayId;
        end
      end
    end
  end
  else if (new.HareketTipi = ShtSatisIade) then
  begin
    if (IntKarsilastir(Old.IplikTalepDetayID, New.IplikTalepDetayID) = 0) then
    begin
/* degişmiş ise eskisinden cıkar, yenisine ekle */
      if (Old.IplikTalepDetayID is not null) then
        Update IplikTalepDetay Set IadeGelenMiktar = IadeGelenMiktar - Old.Miktar, LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = Old.IplikTalepDetayId;

      if (New.IplikTalepDetayID is not null) then
        Update IplikTalepDetay Set IadeGelenMiktar = IadeGelenMiktar + new.Miktar, LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = New.IplikTalepDetayId;
    end
    else
    begin
      if (New.IplikTalepDetayID is not null) then
        Update IplikTalepDetay Set IadeGelenMiktar = IadeGelenMiktar - Old.Miktar + new.Miktar, LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = New.IplikTalepDetayId;
    end
  end
  else if (New.HareketTipi = shtUretimeCikis and New.IplikTelefIsEmriDetayId is not null) then
  begin
    Update IplikTelefIsEmriDetay set CikisMiktari = CikisMiktari - Old.Miktar + New.Miktar where IplikTelefIsEmriDetay.IplikTelefIsEmriDetayID = New.IplikTelefIsEmriDetayId;
  end
  else if (New.HareketTipi = shtUretimeCikis and New.HarmanKayitID is not null and New.HarmanKayitID <> Old.HarmanKayitID) then
  begin
    Select Min(IplikIrsaliyeDetay.IplikIrsaliyeDetayId) from IplikIrsaliyeDetay
      where IplikIrsaliyeDetay.IplikIrsaliyeId = New.IplikIrsaliyeId and IplikIrsaliyeDetay.HarmanKayitId = New.HarmanKayitId and
            IplikIrsaliyeDetay.HamID = New.HamId and IplikIrsaliyeDetay.IplikDepoId = New.IplikDepoID and IplikIrsaliyeDetay.IplikIrsaliyeDetayID <> New.IplikIrsaliyeDetayID
      into :FIplikIrsaliyeDetayId;

    if (FIplikIrsaliyeDetayId is not null) then
    Begin
      Select IplikStokID from IplikStok where IplikStok.CikisIplikIrsaliyeDetayId = New.IplikIrsaliyeDetayId into :FIplikStokID;

      Delete From IplikIrsaliyeDetay where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = New.IplikIrsaliyeDetayId;

      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi + 1, Miktar = Miktar + New.Miktar, NetMiktar = NetMiktar + New.NetMiktar, NeredenUpdate = 1, KullaniciID = New.KullaniciID where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :FIplikIrsaliyeDetayId;
      Update IplikStok set CikisIplikIrsaliyeDetayId = :FIplikIrsaliyeDetayId where IplikStok.IplikStokId = :FIplikStokID;
    end
  end
  else if (New.HareketTipi = shtSarimaCikis) then
  begin
    Update SarimStok set KapAdedi = KapAdedi - Old.KapAdedi + new.KapAdedi, Miktar = Miktar - Old.Miktar + new.Miktar where SarimStok.IplikLotId = New.IplikLotId;
  end
  else if (New.HareketTipi = shtBukumeCikis and New.IplikTelefIsEmriDetayId is not null) then
  begin
    Update BukumIsEmri set CikisMiktari = CikisMiktari - Old.Miktar + New.Miktar, LinkIslemi = 1 where BukumIsEmri.BukumIsEmriKayitID = New.BukumIsEmriKayitId;
  end

  if ((IntKarsilastir(Old.IplikIsletmeId, New.IplikIsletmeId) = 0) or Old.HamId <> New.HamID or Old.IplikDepoID <> New.IplikDepoID) then
    Update IplikStok set HamID = new.HamId, IplikDepoID = new.IplikDepoID, IplikIsletmeId = new.IplikIsletmeId where IplikStok.IplikIrsaliyeDetayId = new.IplikIrsaliyeDetayId;

  if (old.FaturaMiktar <> new.FaturaMiktar) then degisti = 1;
  else if (old.BirimFiyat <> new.BirimFiyat) then degisti = 1;
  else if (old.Kdv <> new.Kdv) then degisti = 1;
  else if (old.BirimIskontoParasal <> new.BirimIskontoParasal) then degisti = 1;
  else if (old.BirimIskontoOrani1 <> new.BirimIskontoOrani1) then degisti = 1;
  else if (old.BirimIskontoOrani2 <> new.BirimIskontoOrani2) then degisti = 1;
  else if (old.KdvHaricFiyat <> new.KdvHaricFiyat) then degisti = 1;
  else if (old.Tevkifat <> new.Tevkifat) then degisti = 1;

  if (Degisti = 1) then   /* Fatura Toplamini etkileyecek degisiklik oldu, yeniden hesaplaması icin */
  if (new.IplikFaturaId is not null) then update IplikFatura set KullaniciId = New.KullaniciId  where IplikFatura.IplikFaturaId = new.IplikFaturaId;
end
