CREATE TRIGGER AFTERINSERTGIRISKALITEKONTROL FOR GIRISKALITEKONTROL
ACTIVE AFTER INSERT POSITION 0
as
begin
  insert into StokHareketDetay(StokHareketID, HareketTipi, GirisKaliteKontrolId, Kilo, BrutKilo, Metre, KapAdedi, InsertKullaniciId, KullaniciId, LinkIslemi)
  select StokHareket.StokHareketId, StokHareket.HareketTipi, new.GirisKaliteKontrolId, new.NetKilo, new.BrutKilo, new.Metre, new.Adet, new.InsertKullaniciId, new.InsertKullaniciId, 1
  from StokHareket where StokHareket.StokHareketId = new.StokHareketId;
end
