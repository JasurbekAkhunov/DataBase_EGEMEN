Definition

CREATE TRIGGER BEFOREUPDATEPASAMASTOKDETAY1 FOR PARTIASAMASTOKDETAY
ACTIVE BEFORE UPDATE POSITION 1
as
declare variable PartiKayitId integer;
declare variable SevkKontrolEt integer;
declare variable AsamaDegerOlmadanCikisYapsin integer;
declare variable AKSonucGirmedenCikisYapsin   integer;
declare variable TamirOnayKullaniciId integer;
declare variable PartiTamirId         integer;
declare variable IslemSiraNo integer;
declare variable Durumu integer;
declare variable Miktar Numeric(12,3);
declare variable Kilo   Numeric(12,3);
declare variable Metre  Numeric(12,3);
declare variable BobinSayisi integer;
declare variable KayitSayisi integer;
declare variable CikistaKiloZorunlu    integer;
declare variable CikistaMetreZorunlu   integer;
declare variable HamTipiId             integer;
declare variable CikisKiloYuvarlamaCarpani integer;
declare variable CikisMetreYuvarlamaCarpani integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable AsamaId integer;
declare variable KS      integer;
declare variable IcKontrolMinZorunluKilo Double Precision;
declare variable AsamaCikisFireOrani Double Precision;
declare variable EgemenFirmaId integer;
declare variable CikisZorunluMinMiktar BoyaMiktarSize;
declare variable PartiStokMiktar       BoyaMiktarSize;
declare variable PartiStokNetSevkMiktar BoyaMiktarSize;
begin
  if (New.LinkIslemi < 2) then
  begin
    Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

    IslemSiraNo = 0;
    SevkKontrolEt = 0;

    Select PartiAsamalari.CikistaKiloZorunlu, PartiAsamalari.CikistaMetreZorunlu, PartiAsamalari.PartiTamirId from PartiAsamalari
      where PartiAsamalariId = new.PartiAsamalariId
      into :CikistaKiloZorunlu, :CikistaMetreZorunlu, :PartiTamirId;

    Select PartiAsamaStok.AsamaID, Asama.CikisFireOrani from PartiAsamaStok
      left join Asama on Asama.AsamaID = PartiAsamaStok.AsamaId
      where PartiAsamaStok.PartiAsamalariID = new.PartiAsamalariID and PartiAsamaStok.PartiSinifID = new.PartiSinifId
      into :AsamaID, :AsamaCikisFireOrani;

    Select PartiStok.HamTipiId, Durumu, PartiStok.Miktar, PartiStok.NetSevkMiktar from PartiStok  where PartiStok.PartiSinifId = new.PartiSinifId into :HamTipiId, :Durumu, :PartiStokMiktar, :PartiStokNetSevkMiktar;

    if (Old.GirisTarihi is null) then
    begin
      if (new.Giristarihi is not null) then
      begin
        if (new.makinaId is null) then exception GiristeMakinaIdYok;
        if (new.GirisPersonelId is null) then exception GiristePersonelIdYok;

        if (new.HareketKullaniciId is null) then new.HareketKullaniciId = 1;

        Select Makina.UretimMaliyeti, Makina.UretimMaliyetiDovizId from Makina where Makina.MakinaID = new.MakinaId into New.UretimMaliyeti, New.UretimMaliyetiDovizId;

        if (new.MakinaSarjId is null) then  /* sınıf bölünüyor ise bölündügü noktada sarj deger kazanmıs oluyor, onu bozmamak icin işlem yapmıyoruz */
        begin
          Select Max(MakinaSarjId) from PartiAsamaStokDetay
            where PartiAsamaStokDetay.MakinaId = New.MakinaId and PartiAsamaStokDetay.MakinaSarjId is not null and
                  PartiAsamaStokDetay.GirisTarihi is not null and PartiAsamaStokDetay.CikisTarihi is null
            into new.MakinaSarjId;

          if (New.MakinaSarjId is null) then
          begin
            New.MakinaSarjId = Gen_ID(MakinaSarj_Id_Gen, 1);

            insert into MakinaSarj(MakinaSarjID, MakinaId, BaslamaTarihi, BitisTarihi, PartiVar)
                            values(New.MakinaSarjId, New.MakinaId, New.GirisTarihi, null, 1);
          end
        end
      end
    end
    else /* Old Giris Tarihi is not null ise */
    if (New.GirisTarihi is null) then /* Giris iptal ediliyorsa */
    begin
      if (new.CikisTarihi is not null) then exception CikisVarGirisSilinemez;
      New.AraVermeSuresi = 0;
      new.GirisPersonelId = null;
      new.MakinaId = null;
      new.HareketKullaniciId = Null;
      SevkKontrolEt = 1;
      New.MakinaSarjId = Null;
      New.CikisKilo = 0;
      New.CikisMetre = 0;
      New.CikisMiktar = 0;
      New.CikisV1Miktar = 0;
      New.CikisV2Miktar = 0;
      New.CikisV3Miktar = 0;
    end

    if (Old.Cikistarihi is not null) then
    if (New.Cikistarihi is null) then
    begin
      SevkKontrolEt = 1;
      new.CikisPersonelId = null;
    end

    if (SevkKontrolEt = 1 ) then
    if (Durumu = 8 or PartiStokNetSevkMiktar > 0 ) then exception SevkEdilmisHareketSilinemez; /*Parçalı sevklerde durum 8 olmuyor bu nednle sevk miktarınada bakıyoruz.*/

    if (New.LinkIslemi = 0) then
    begin
      if (Old.CikisTarihi is not null and new.CikisTarihi is not null) then
        if (Old.CikisKilo <> New.CikisKilo  or Old.CikisMetre <> New.CikisMetre or StrKarsilastir(Old.CikisRafId, New.CikisRafId) = 0 ) then exception CikisVarCKiloMetreDegisemez;
    end

    New.LinkIslemi = 0;

    if (Old.CikisTarihi is null and New.CikisTarihi is not null) then
    begin
      Select IcKontrolMinZorunluKilo, AsamaDegerOlmadanCikisYapsin, AKSonucGirmedenCikisYapsin, EgemenFirmaId, CikisZorunluMinMiktar from Sistem where SistemID = 1 into :IcKontrolMinZorunluKilo, :AsamaDegerOlmadanCikisYapsin, :AKSonucGirmedenCikisYapsin, :EgemenFirmaId, :CikisZorunluMinMiktar;

      if (CikisZorunluMinMiktar >= PartiStokMiktar) then
      begin
        CikistaKiloZorunlu  = 0;
        CikistaMetreZorunlu = 0;
      end

      if (AsamaDegerOlmadanCikisYapsin = 0) then execute procedure PartiAsamaDegerKontroluYap(new.PartiAsamalariId, new.PartiSinifId);
      if (AKSonucGirmedenCikisYapsin = 0) then execute procedure AraKontrolAsamasiKontroluYap(new.PartiAsamalariId, new.PartiSinifId);

      if (New.Kilo > IcKontrolMinZorunluKilo) then
      begin
        KS = 0;
        Select Count(*) from PartiAsamaIcKontrol
          left join AsamaIcKontrolTipi on AsamaIcKontrolTipi.IcKontrolTipiId = PartiAsamaIcKontrol.IcKontrolTipiId and AsamaIcKontrolTipi.AsamaID = :AsamaId
          where PartiAsamaIcKontrol.PartiAsamalariID = New.PartiAsamalariID and PartiAsamaIcKontrol.PartiSinifID = New.PartiSinifId and
                AsamaIcKontrolTipi.SonucGirmedenCikisYapmasin = 1 and
                (PartiAsamaIcKontrol.Sonuc is null or PartiAsamaIcKontrol.Sonuc = '')
          into :KS;

        if (KS > 0) then exception AsamaIciKontrolGirilmemis;
      end

      if (CikistaKiloZorunlu = 0) then
      begin
        if (AsamaCikisFireOrani < 30) then
          New.CikisKilo = New.Kilo - New.Kilo * AsamaCikisFireOrani / 100;
        else
          New.CikisKilo = New.Kilo;

        New.CikisKilo = Yuvarla(New.CikisKilo, 100);
      end

      if (CikistaMetreZorunlu = 0) then
      begin
        New.CikisMetre = New.Metre;
      end

      if (EgemenFirmaId = 0 and New.MakinaSarjID is not Null) then /*NY*/
      begin
        Select Metre * BirPulseKacMetre from MakinaSarj
          Left Join Makina on Makina.MakinaID = MakinaSarj.MakinaID
        where MakinaSarj.MakinaSarjID = New.MakinaSarjID
        into new.CikisMetre;
      end

      if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
        New.CikisMiktar = New.CikisKilo;
      else
        New.CikisMiktar = New.CikisMetre;
    end

    if (Old.Kilo <> New.Kilo or Old.Metre <> New.Metre) then
  /*  if (Old.Miktar <> New.Miktar) then */
    begin
      CikisKiloYuvarlamaCarpani = 100;
      CikisMetreYuvarlamaCarpani = 100;

      if (Old.CikisTarihi is not null and New.CikisTarihi is not null) then
      begin
        if (New.Miktar <> 0) then
        if (Old.Miktar <> 0) then
        begin
          if (CikistaKiloZorunlu = 1) then
          begin
            New.CikisKilo = Old.CikisKilo * New.Miktar / Old.Miktar;
            new.CikisKilo = Yuvarla(new.CikisKilo, CikisKiloYuvarlamaCarpani);
          end
          else New.CikisKilo = New.Kilo;

          if (CikistaMetreZorunlu = 1) then
          begin
            New.CikisMetre = Old.CikisMetre * New.Miktar / Old.Miktar;
            new.CikisMetre = Yuvarla(new.CikisMetre, CikisMetreYuvarlamaCarpani);
          end
          else New.CikisMetre = New.Metre;

        end
        else
        begin
          New.CikisKilo = 0;
          New.CikisMetre = 0;
        end
      end
    end

    if (New.CikisKilo is null) then New.CikisKilo = 0;
    if (New.CikisMetre is null) then New.CikisMetre = 0;

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
      New.CikisMiktar = New.CikisKilo;
    else
      New.CikisMiktar = New.CikisMetre;

    if (New.GirisTarihi is not null and Old.GirisTarihi is null) then
    begin
      if (new.PartiTamirSinifId is not null) then
      begin
        Select TamirOnayKullaniciId from PartiTamirSinif
          left join PartiTamir on PartiTamir.PartiTamirId = PartiTamirSinif.PartiTamirId
          where PartiTamirSinif.PartiTamirSinifId = new.PartiTamirSinifId
          into :TamirOnayKullaniciId;

        if (TamirOnayKullaniciId is null) then exception IsletmeIciTamirOnaylanmamis;
      end
      else if (PartiTamirId is not null) then
      begin
        Select TamirOnayKullaniciId from PartiTamir where PartiTamir.PartiTamirId = :PartiTamirId into :TamirOnayKullaniciId;

        if (TamirOnayKullaniciId is null) then exception IsletmeIciTamirOnaylanmamis;
      end
    end

  /* Daha önce bölünmüş ise kiloyu toparlıyoruz*/

    if (new.GirisTarihi is null and Old.GirisTarihi is not null) then
    begin
      Select Count(*) from PartiAsamaStokDetay
          where PartiAsamaStokDetay.PartiAsamalariId = new.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = new.PartiSinifId
          into :KayitSayisi;

      if (KayitSayisi = 1) then
      begin
        Select PartiAsamaStok.GirisMiktar, PartiAsamaStok.GirisKilo, PartiAsamaStok.GirisMetre, PartiStok.BobinSayisi from PartiAsamaStok
            left join PartiStok on PartiStok.PartiSinifID = PartiAsamaStok.PartiSinifId
            where PartiAsamaStok.PartiSinifId = New.PartiSinifId and PartiAsamaStok.PartiAsamalariId = new.PartiAsamalariId
            into :Miktar, :Kilo, :Metre, :BobinSayisi;

        New.Miktar = Miktar;
        New.Kilo = Kilo;
        New.Metre = Metre;
        New.BobinSayisi = BobinSayisi;
        New.KiloDegistirenPersonelId = Null;
      end
    end
  end

  New.LinkIslemi = 0;
  new.DegisimTarihi = current_TimeStamp(0);
end
