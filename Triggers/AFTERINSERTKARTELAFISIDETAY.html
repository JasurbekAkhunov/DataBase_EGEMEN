CREATE TRIGGER AFTERINSERTKARTELAFISIDETAY FOR KARTELAFISIDETAY
ACTIVE AFTER INSERT POSITION 0
as
declare variable MevcutFisDetayId integer;
begin
  if (new.KartelaTalepDetayId is not null) then
    Update KartelaTalepDetay Set KartelaFisiDetayId = New.KartelaFisiDetayId where KartelaTalepDetay.KartelaTalepDetayId = new.KartelaTalepDetayId;

  if (New.Giris = -1) then
  begin
    if (New.Miktar = 1) then
    begin
      select first 1 KartelaFisiDetayId from KartelaFisiDetay
        where KartelaFisiDetay.KartelaFisiId = new.KartelaFisiId and
        KartelaFisiDetay.KaliteId = new.KaliteId and
        (KartelaFisiDetay.KartelaTalepDetayId = new.KartelaTalepDetayId or (KartelaFisiDetay.KartelaTalepDetayId is Null and new.KartelaTalepDetayId is Null))and
        (KartelaFisiDetay.Aciklama = new.Aciklama or (KartelaFisiDetay.Aciklama is Null and new.Aciklama is Null)) and
        KartelaFisiDetay.Giris = new.Giris and KartelaFisiDetay.KartelaFisiDetayId <> new.KartelaFisiDetayId
      into :MevcutFisDetayId;

      if ((MevcutFisDetayId is not Null) and (MevcutFisDetayId > 0)) then
      begin
        update KartelaFisiDetay set Miktar = Miktar + 1, DegisimTarihi = current_timestamp(0) where KartelaFisiDetay.KartelaFisiDetayId = :MevcutFisDetayId;
        delete from KartelaFisiDetay where KartelaFisiDetay.KartelaFisiDetayId = new.KartelaFisiDetayId;
      end
    end
  end
end
