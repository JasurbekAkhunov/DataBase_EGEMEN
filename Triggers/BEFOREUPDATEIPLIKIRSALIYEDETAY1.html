CREATE TRIGGER BEFOREUPDATEIPLIKIRSALIYEDETAY1 FOR IPLIKIRSALIYEDETAY
ACTIVE BEFORE UPDATE POSITION 1
as
declare variable Degisemez integer;
declare variable KS        integer;
declare variable Nerede integer;
declare variable FEntegrasyon integer;
begin
  Degisemez = 0;

  if (New.IplikFaturaId is not null) then Nerede = 2;
  else Nerede = 1;

  if (New.NeredenUpdate <> 4) then
  if (new.IplikFaturaId is not null) then
  begin
    if (Old.IplikFaturaId is not null) then
    begin
      select Entegrasyon from IplikFatura where IplikFatura.IplikFaturaId = new.IplikFaturaId into :FEntegrasyon;
      if (FEntegrasyon = 1) then exception FF_FaturaHazirDegistiremezsiniz;
    end
  end

  if (New.NeredenUpdate <> 3 and New.NeredenUpdate <> 4) then
  begin
    if (New.NeredenUpdate <> Nerede) then exception UstBelgeVarDegistirilemez;
    else if (new.NeredenUpdate <> New.NeredenGirildi) then
    begin
      if (Old.HamId <> New.HamId) then Degisemez = 1;
      else if (IntKarsilastir(Old.IplikTalepDetayId, New.IplikTalepDetayID) = 0) then Degisemez = 1;
      else if (old.Miktar <> New.Miktar) then Degisemez = 1;
      else if (old.NetMiktar <> New.NetMiktar) then Degisemez = 1;

      if (Degisemez = 1) then exception IAltBelgeVarDegisemez;
    end
  end

  New.NeredenUpdate = -1;
end
