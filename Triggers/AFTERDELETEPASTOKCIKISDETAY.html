Definition

CREATE TRIGGER AFTERDELETEPASTOKCIKISDETAY FOR PARTIASAMASTOKCIKISDETAY
ACTIVE AFTER DELETE POSITION 0
as
Declare variable FKilo Double Precision;
Declare variable FKiloDara Double Precision;
Declare variable FMetre Double Precision;
begin
  Select Sum(Kilo), Sum(KiloDara), Sum(Metre) from PartiAsamaStokCikisDetay where PartiAsamaStokDetayID = Old.PartiAsamaStokDetayID
    into :FKilo, :FKiloDara, :FMetre;

  Update PartiAsamaStokDetay Set CikisKilo = :FKilo, CikisKiloDara = :FKiloDara, CikisMetre = :FMetre where PartiAsamaStokDetayID = Old.PartiAsamaStokDetayID;
end
