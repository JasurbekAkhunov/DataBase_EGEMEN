CREATE TRIGGER BEFOREINSERTCEKSENETTAHSILAT FOR CEKSENETTAHSILAT
ACTIVE BEFORE INSERT POSITION 0
as
declare variable SonBordroTipi integer;
declare variable SonDonemBitisTarihi TimeStamp;
begin
  select max(BitisTarihi) from DonemKapatma where Bolum = 4 and OnayKullaniciId is not Null into :SonDonemBitisTarihi;
  if (SonDonemBitisTarihi is not Null and SonDonemBitisTarihi > new.Tarih) then
    exception DonemKapaliHEklenemez;

  select SonBordroTipi from CekSenetDetay where CekSenetDetay.GenelKayitId = new.GenelKayitId into :SonBordroTipi;

  if (SonBordroTipi = 1 or SonBordroTipi = 7 or  SonBordroTipi = 8) then exception CekSenetTahsilEdilemez;

  new.CekSenetTahsilatId = gen_id(CekSenetTahsilat_Id_Gen, 1);
end
