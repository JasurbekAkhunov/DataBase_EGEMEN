CREATE TRIGGER BEFOREUPDATEHGISEMRI FOR HGISEMRI
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable kayitSayisi Integer;
begin
  if (new.OnayDurumu = 1 and old.OnayDurumu = 1) then
    if (new.Bitti = 1 and old.Bitti = 1) then
      if (StrKarsilastir(new.OrmeMakinaId, old.OrmeMakinaId) = 1 and new.Planlandi = old.Planlandi and IntKarsilastir(new.MiktarGun, old.MiktarGun) = 1 and
          Coalesce(new.BaslamaTarihi, cast(current_date as TimeStamp)) = Coalesce(old.BaslamaTarihi, cast(current_date as TimeStamp)) and
          Coalesce(new.BitisTarihi, cast(current_date as TimeStamp)) = Coalesce(old.BitisTarihi, cast(current_date as TimeStamp))) then
        Exception OnaylanmisIsEmriGuncellenemez;

  if (new.FirmaId = '') then New.FirmaId = null;
  if (new.OrmeMakinaId = '') then New.OrmeMakinaId = null;


  if (New.FirmaId is not null and New.OrmeMakinaId is not null) then exception PlanlamaFirmaMakinaIdHatasi1;

  select count(*) from StokHareket where StokHareket.HgIsEmriKayitId = new.HgIsEmriKayitId into :KayitSayisi;

  if ((KayitSayisi > 0) and (new.Bitti = 1 and old.Bitti = 1)) then
    Exception HgIsEmriHareketGormus;
  else
  begin
    select count(*) from StokDetay where StokDetay.HgIsEmriKayitId = new.HgIsEmriKayitId into :KayitSayisi;
    if ((KayitSayisi > 0) and (new.Bitti = 1 and old.Bitti = 1)) then
      Exception HgIsEmriHareketGormus;
  end

  if (((Trim(Coalesce(new.UretimPartiNo,'')) = '') or ((StrKarsilastir(old.OrmeMakinaID, New.OrmeMakinaID) = 0 or StrKarsilastir(old.FirmaID, New.FirmaID) = 0) and New.KesimTalimati = 0)) and New.OnayDurumu = 0) then
    Select UretimPartiNo, Numarator from HGIsEmriUretimPartiNoUret(new.HGSiparisKayitId, New.FirmaId, New.OrmeMakinaId, New.KesimTalimati) into new.UretimPartiNo, New.Numarator;

  new.DegisimTarihi = current_timestamp(0);
end
