CREATE TRIGGER AFTERINSERTDKKONTROLHATAMETRE FOR DOKUMAKALITEKONTROLHATAMETRE
ACTIVE AFTER INSERT POSITION 0
as
declare variable DokumaKaliteKontrolHatalariId integer;
declare variable OrmeFisId Integer;
begin
  Select DokumaKaliteKontrolHatalariId from DokumaKaliteKontrolHatalari
     where DokumaKaliteKontrolHatalari.DokumaKaliteKontrolId = new.DokumaKaliteKontrolId and DokumaKaliteKontrolHatalari.DokumaHataTipiId = new.DokumaHataTipiId into :DokumaKaliteKontrolHatalariId;

  if (DokumaKaliteKontrolHatalariId is null) then  insert into DokumaKalitekontrolHatalari(DokumaKalitekontrolid, DokumaHataTipiId, Adet) values(new.DokumaKaliteKontrolId, new.DokumahataTipiId, 1);
  else Update DokumaKaliteKontrolHatalari Set Adet = Adet + 1 where DokumaKaliteKontrolHatalari.DokumaKaliteKontrolHatalariID = :DokumaKaliteKontrolHatalariId;
end
