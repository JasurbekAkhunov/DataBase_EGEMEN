CREATE TRIGGER BEFOREINSERTRTRANSFERFISIDETAY FOR RAFTRANSFERFISIDETAY
ACTIVE BEFORE INSERT POSITION 0
as
declare variable KalanMiktar    Numeric(12,3);
declare variable FHamTipiId integer;
declare variable DokumaKumasiKilodanTakipEt integer;
begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  select KalanMiktar from RafDetayKalanMiktarGetir(new.RafDetayId) into :KalanMiktar;

  Select Ham.HamTipiId from RafDetay
    left join Ham on Ham.HamId = RafDetay.HamID
    where RafDetay.RafDetayID = New.RafDetayId
    into :FHamTipiId;

  if (FHamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
    New.Miktar = New.kilo;
  else
    New.Miktar = New.Metre;

  if (KalanMiktar < new.Miktar) then exception RaftaYeterliKiloMetreYok;

  if (new.RafTransferFisiDetayID is null) then new.RafTransferFisiDetayID = Gen_Id(RafTransferFisiDetay_Id_gen, 1);
end
