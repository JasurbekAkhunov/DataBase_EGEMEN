CREATE TRIGGER AFTERUPDATEPROSESASAMALARI FOR PROSESASAMALARI
ACTIVE AFTER UPDATE POSITION 0
as
declare variable FAsamaKisaAdi   varchar(10);
declare variable FAdi            varchar(255);
begin
  FAdi = '';
  for select distinct Asama.AsamaKisaAdi from ProsesAsamalari
  left join Asama on Asama.Asamaid = ProsesAsamalari.Asamaid
  where ProsesAsamalari.ProsesId = new.ProsesId order by Sira into :FAsamaKisaAdi do
  begin
    if (FAsamaKisaAdi is not null) then
    if (Char_Length(FAdi) + Char_Length(FAsamaKisaAdi) < 253) then
    begin
     if (FAdi <> '' ) then FAdi = FAdi || '+';
     FAdi = FAdi || FAsamaKisaAdi;
    end
  end
  Update Proses set Adi = :FAdi where Proses.ProsesId = new.ProsesId;
end
