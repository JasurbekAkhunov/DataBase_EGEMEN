CREATE TRIGGER BEFOREINSERTBCIKISPARTINO FOR BOYAHANECIKISPARTINO
ACTIVE BEFORE INSERT POSITION 0
as
declare variable FirmaKisaAdi varchar(10);
begin
  New.Numarator = 0;
  New.Yil = f_Year(Current_date);

  if (new.CikisPartiNo is null) then
  begin
    Select Firma.KisaAdi from Firma where Firma.FirmaID = New.FirmaId into :FirmaKisaAdi;

    Select Max(Numarator) from BoyahaneCikisPartiNo where BoyahaneCikisPartiNo.FirmaID = New.FirmaId and BoyahaneCikisPartiNo.Yil = new.Yil into New.Numarator;

    if (New.Numarator is null) then New.Numarator = 1;
    else New.Numarator = New.Numarator + 1;
    new.CikisPartiNo = EgemenIntToLzStr( New.Yil - 2000, 2) || '-' || FirmaKisaAdi || '-' || EgemenIntegerToStr(New.Numarator);
  end
end
