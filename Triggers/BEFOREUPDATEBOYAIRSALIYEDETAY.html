Definition

CREATE TRIGGER BEFOREUPDATEBOYAIRSALIYEDETAY FOR BOYAIRSALIYEDETAY
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable IadeEdilecekMiktar Numeric(12,3);
declare variable PartiMiktar        Numeric(12,3);
declare variable PartiNetMiktar     Numeric(12,3);
declare variable PartiSevkMiktar    Numeric(12,3);
declare variable SevkBrutMiktar     Numeric(12,3);
declare variable ParcaliSevk integer;
declare variable TamamiTartildi integer;
declare variable AsamaKisaAdi varchar(10);
declare variable IlaveIslemlerAdi varchar(30);
declare variable biSatis                integer;
declare variable biAlisIade             integer;
declare variable KantarKayitId          integer;
declare variable OnayDurumu             integer;
declare variable Fason                  integer;
declare variable HamTipiId integer;
declare variable FStokFirmaId varchar(20);
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable SiparisDetayHamId integer;
declare variable SiparisDetayKaliteId varchar(20);
declare variable HamGirisKaliteId varchar(20);
declare variable SiparisFirmaId varchar(20);
declare variable SatisFirmaId varchar(20);
declare variable SonDonemBitisTarihi TimeStamp;
declare variable DonemKapatmadaSadeceDetaylar integer;
declare variable biSatisIade integer;
declare variable EgemenFirmaID integer;
declare variable TransferEdildi EgeBoolean;
begin
  Select TransferEdildi From BoyaIrsaliye Where BoyaIrsaliyeId = new.BoyaIrsaliyeId into :TransferEdildi;

  if (((old.TransferEdildi = 1 and new.TransferEdildi = 1) or TransferEdildi = 1 ) and IntKarsilastir(old.BoyaFaturaId, new.BoyaFaturaID) = 1) then exception TrasferEdilmisDegistirilmez;

  Select DokumaKumasiKilodanTakipEt, EgemenFirmaID from Sistem into :DokumaKumasiKilodanTakipEt, :EgemenFirmaID;
  Select biSatis, biAlisIade, biSatisIade from BoyaIrsaliyeHareketTipiGetir into :biSatis, :biAlisIade, :biSatisIade;

  if (Old.HareketTipiId <> New.HareketTipiId) then exception BoyaIrsaliyeHareketTipiDegismez;

  if (Old.RafDetayId is not null) then
  if (new.RafDetayId is null or new.RafDetayId <> Old.RafDetayId) then exception RafDetayIdDegisemez;

  if (Old.HamIaderezervId is not null) then
  if (new.HamIaderezervId is null or new.HamIaderezervId <> Old.HamIaderezervId) then exception HamIadeRezervIdDegisemez;

  if (Old.PartiSinifId is not null) then
  if (new.PartiSinifId is null or new.PartiSinifId <> Old.PartiSinifId) then exception IrsaliyePartiSinifIdDegisemez;

  Select Ham.HamTipiId from Ham where Ham.HamID = New.HamId into :HamTipiId;

  if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
  begin
    New.Miktar = New.kilo;
    New.BrutMiktar = New.Brutkilo;
  end
  else
  begin
    New.Miktar = New.Metre;
    New.BrutMiktar = New.BrutMetre;
  end

  if (new.RafDetayId is not null) then
  begin
    if (new.FasonHamIade = 1 or New.HareketTipiId = biAlisIade) then
    begin
      if (new.RafDetayId is not null and Old.Miktar <> new.Miktar) then
      begin
        select IadeEdilecekMiktar from RafDetayKalanMiktarGetir(new.RafDetayId) into :IadeEdilecekMiktar;
        if ((IadeEdilecekMiktar + Old.Miktar) < new.Miktar) then exception IadeEdilecekKiloyuAstiniz;
      end

      if (new.HamIadeRezervId is null) then exception HamIadeIcinRezervSecilmemis;

      if (Old.KapAdedi <> New.KapAdedi or Old.Kilo <> New.Kilo or Old.Metre <> New.Metre or Old.Adet <> New.Adet) then
          Update HamIadeRezerv set IadeEdilenKapAdedi = IadeEdilenKapAdedi - Old.KapAdedi + New.KapAdedi,
                                   IadeEdilenKilo = IadeEdilenKilo - Old.Kilo + New.Kilo,
                                   IadeEdilenMetre = IadeEdilenMetre - Old.Metre + New.Metre,
                                   IadeEdilenAdet = IadeEdilenAdet - Old.Adet + New.Adet
                             where HamIadeRezerv.HamIadeRezervId = New.HamIadeRezervId;

    select HamStokDetay.StokFirmaId  from RafDetay
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      where RafDetay.RafDetayId = new.RafDetayId
      into :FStokFirmaId;

    end

    Select BoyaIrsaliye.KantarKayitId from BoyaIrsaliye where BoyaIrsaliye.BoyaIrsaliyeId = new.BoyaIrsaliyeId into :KantarKayitId;

    if (Old.Kilo <> New.Kilo) then
    if (KantarKayitId is not null) then
    begin
      Select OnayDurumu from Kantar where Kantar.KantarKayitID = :KantarKayitId into :OnayDurumu;
      if (OnayDurumu = 1) then exception KantarOnayiVerilmisDegisemez;
    end

    if (new.FasonHamIade = 1) then execute procedure HamStokKumilatifGuncelle(:FStokFirmaId, 1, 3, Old.HamId, New.HamId, Old.IrsaliyeTarihi, new.IrsaliyeTarihi, Old.Kilo, Old.Metre, New.Kilo, New.Metre);
    else if (new.hareketTipiId = biAlisIade) then execute procedure HamStokKumilatifGuncelle(new.FirmaId, 0, 3, Old.HamId, New.HamId, Old.IrsaliyeTarihi, new.IrsaliyeTarihi, Old.Kilo, Old.Metre, New.Kilo, New.Metre);
    else execute procedure HamStokKumilatifGuncelle(new.FirmaId, 0, 4, Old.HamId, New.HamId, Old.IrsaliyeTarihi, new.IrsaliyeTarihi, Old.Kilo, Old.Metre, New.Kilo, New.Metre);
  end

  if (new.BoyaFaturaId is not null) then
  if (Old.BrutMiktar <> new.BrutMiktar or Old.Miktar <> new.Miktar) then exception FaturasiKesilmisKiloDegisemez;

  if (new.PartiSinifId is not null) then
  begin
    if (New.BarkodLinkIslemi = 0) then
    begin
      if (old.Kilo <> New.Kilo or Old.Metre <> New.Metre) then exception SevkMiktariniDegistiremezsiniz;
    end

    select Parti.ParcaliSevk, PartiStok.Miktar, PartiStok.NetMiktar, PartiStok.NetSevkMiktar - PartiStok.NetIadeMiktar, PartiStok.TamamiTartildi, Parti.FirmaId from PartiStok
      left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      where PartiStok.PartiKayitId = new.PartiKayitId and PartiStok.PartiSinifId = new.PartiSinifId
      into :ParcaliSevk, :PartiMiktar, :PartiNetMiktar, :PartiSevkMiktar, :TamamiTartildi, :FStokFirmaId;

    if (TamamiTartildi is null) then exception PartiStokSecilmemis;

    if (ParcaliSevk = 0) then
    begin
      if (PartiNetMiktar < new.Miktar) then exception NetKiloMetreAstiniz;
      if (PartiNetMiktar > new.Miktar) then exception ParcaliSevkEdemezsiniz;
      if (PartiMiktar <> new.BrutMiktar) then exception ParcaliSevkEdemezsiniz;

      if (new.LinkIslemi = 0) then
        update PartiStok set PartiStok.NetSevkKilo = new.Kilo,
                             PartiStok.BrutSevkKilo = new.BrutKilo,
                             PartiStok.NetSevkMetre = new.Metre,
                             PartiStok.BrutSevkMetre = new.BrutMetre,
                             PartiStok.NetSevkKapAdedi = new.KapAdedi,
                             PartiStok.NetSevkAdet = new.Adet
                             where PartiStok.PartiSinifId = new.PartiSinifId;
    end
    else
    begin
      if (new.HareketTipiID <> biSatisIade) then
      begin
        Select sum(iif(BoyaIrsaliyeDetay.HareketTipiID = :biSatisIade, BoyaIrsaliyeDetay.BrutMiktar * -1, BoyaIrsaliyeDetay.BrutMiktar)) from BoyaIrsaliyeDetay where BoyaIrsaliyeDetay.PartiSinifId = new.PartiSinifId into :SevkBrutMiktar;

        if (SevkBrutMiktar is null) then SevkBrutMiktar = 0;
        if (PartiMiktar < (SevkBrutMiktar - Old.BrutMiktar + New.BrutMiktar) and EgemenFirmaID <> 75 and EgemenFirmaID <> 9 and EgemenFirmaID <> 37 and EgemenFirmaID <> 122) then exception SevkBrutKiloyuAstiniz; /*mersu, Arta,Global, zelal*/
        if (PartiNetMiktar < (PartiSevkMiktar - Old.Miktar + New.Miktar)) then exception NetKiloMetreAstiniz;
      end

      if (new.LinkIslemi = 0 and new.HareketTipiId = biSatisIade) then
        update PartiStok set PartiStok.NetIadeKilo = PartiStok.NetIadeKilo - Old.Kilo + new.Kilo ,
                         PartiStok.NetIadeMetre = PartiStok.NetIadeMetre - Old.Metre + New.Metre,
                         PartiStok.NetIadeKapAdedi = PartiStok.NetIadeKapAdedi - Old.KapAdedi + New.KapAdedi,
                         PartiStok.BrutIadeMetre = PartiStok.BrutIadeMetre - Old.BrutMetre + new.BrutMetre,
                         PartiStok.BrutIadeKilo = PartiStok.BrutIadeKilo - Old.BrutKilo + new.BrutKilo
           where PartiStok.PartiSinifId = new.PartiSinifId;
      else if (new.LinkIslemi = 0) then
        update PartiStok set PartiStok.NetSevkKilo = PartiStok.NetSevkKilo - Old.Kilo +  new.Kilo,
                             PartiStok.BrutSevkKilo = PartiStok.BrutSevkKilo - Old.BrutKilo +  new.BrutKilo,
                             PartiStok.NetSevkMetre = PartiStok.NetSevkMetre - Old.Metre +  new.Metre,
                             PartiStok.BrutSevkMetre = PartiStok.BrutSevkMetre - Old.BrutMetre +  new.BrutMetre,
                             PartiStok.NetSevkKapAdedi = PartiStok.NetSevkKapAdedi - Old.KapAdedi +  new.KapAdedi,
                             PartiStok.NetSevkAdet = PartiStok.NetSevkAdet - Old.Adet +  new.Adet
                          where PartiStok.PartiSinifId = new.PartiSinifId;
    end

    new.YapilanIslemler = '';
    for select Coalesce(Asama.Asamakisaadi, '') from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join Recete on Recete.Receteid = PartiAsamalari.Receteid
      left join Asama on Asama.Asamaid = PartiAsamalari.Asamaid or Asama.Asamaid = Recete.UstAsamaId
      left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
      left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliId = PartiTamir.TamirCozumSekliId
      where PartiAsamaStok.PartiSinifId = new.PartiSinifId and Asamayadahil = 1 and Asama.Asamakisaadi is not null and Asama.Asamakisaadi <> '' and
            (PartiAsamalari.PartiTamirID is null or TamirCozumSekli.Ucretli = 1) and
            Asama.ParasalDegeriVar = 1
      order by PartiAsamalari.Sira into :AsamaKisaAdi do
    begin
      if (Char_Length(new.YapilanIslemler) + Char_Length(AsamaKisaAdi) < 253) then
      begin
       if (new.YapilanIslemler <> '' ) then new.YapilanIslemler = new.YapilanIslemler || cast(', ' as varchar(2) character set UTF8);
       new.YapilanIslemler = new.YapilanIslemler || AsamaKisaAdi;
      end
    end

    for Select IlaveIslemler.Adi from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
      left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
      left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliId = PartiTamir.TamirCozumSekliId
      where PartiAsamaStok.PartiSinifId = new.PartiSinifId and Asamayadahil = 1 and PartiAsamalari.SerbestAsama = 0 and IlaveIslemler.Adi is not null and
            (PartiAsamalari.PartiTamirID is null or TamirCozumSekli.Ucretli = 1)
      into :IlaveIslemlerAdi do
    begin
      if (Char_Length(new.YapilanIslemler) + Char_Length(IlaveIslemlerAdi) < 253) then
      begin
       if (new.YapilanIslemler <> '' ) then new.YapilanIslemler = new.YapilanIslemler || cast(', ' as varchar(2) character set UTF8);
       new.YapilanIslemler = new.YapilanIslemler || IlaveIslemlerAdi;
      end
    end

    select Parti.Fason from PartiStok
      left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      where PartiStok.PartiSinifId = new.PartiSinifId
      into :Fason;


    execute procedure HamStokKumilatifGuncelle(:FStokFirmaId, :Fason, 5, Old.HamId, New.HamId, Old.IrsaliyeTarihi, New.IrsaliyeTarihi, Old.BrutKilo, Old.BrutMetre, New.BrutKilo, New.BrutMetre);
  end

  if (new.BoyaSiparisDetayId is not Null) then
  begin
    /*Siparisli ham stok satisi*/
    if (new.RafDetayId is Null) then Exception SiparisliHamSatisUyumsuzluk;
    /*Ham satis degilse, boya siparis detay bilgisi bulunamaz...*/

    select BoyaSiparisDetay.HamId, BoyaSiparisDetay.KaliteId, BoyaSiparis.FirmaId from BoyaSiparisDetay
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
      where BoyaSiparisDetay.BoyaSiparisDetayId = new.BoyaSiparisDetayId
      into :SiparisDetayHamId, :SiparisDetayKaliteId, :SiparisFirmaId;

    Select HamStokMaster.KaliteId from RafDetay
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    where RafDetay.RafDetayId = new.RafDetayId
    into :HamGirisKaliteId;

    select BoyaIrsaliye.FirmaId from BoyaIrsaliye
      where BoyaIrsaliye.BoyaIrsaliyeId = new.BoyaIrsaliyeId
      into :SatisFirmaId;

    if (new.HamId <> SiparisDetayHamId) then Exception HamGirisveHamSatisUyumsuzluk;
    if (HamGirisKaliteId <> SiparisDetayKaliteId) then Exception HamGirisveHamSatisUyumsuzluk;
    if (SatisFirmaId <> SiparisFirmaId) then Exception HamGirisveHamSatisUyumsuzluk;
  end

  select DonemKapatmadaSadeceDetaylar from Sistem into :DonemKapatmadaSadeceDetaylar;

  if (DonemKapatmadaSadeceDetaylar <> 1) then
  begin
    select max(BitisTarihi) from DonemKapatma where Bolum = 1 and OnayKullaniciId is not Null into :SonDonemBitisTarihi;
    if (SonDonemBitisTarihi is not Null and (SonDonemBitisTarihi > new.IrsaliyeTarihi or SonDonemBitisTarihi > old.IrsaliyeTarihi)) then
      exception DonemKapaliHDegisemez;
  end

  new.BarkodLinkIslemi = 0;
  new.DegisimTarihi = current_TimeStamp(0);
  new.LinkIslemi = 0;

  Execute Procedure Logla('U', 1, 'BOYAIRSALIYEDETAY', 'BOYAIRSALIYEDETAYID', new.BoyaIrsaliyeDetayId, null);
end
