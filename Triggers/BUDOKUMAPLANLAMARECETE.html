CREATE TRIGGER BUDOKUMAPLANLAMARECETE FOR DOKUMAPLANLAMARECETE
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KayitSayisi integer;
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable OrmeUretimeCikistaLotTakibiVar integer;
declare variable TahminiFireOrani   Double Precision;
declare variable RecAyniUrunTekrarEdebilir integer;
declare variable DokumaReceteAtkiLotGuncelle EgeBoolean;
declare variable egemenfirmaid integer;
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim;

  select OrmeUretimeCikistaLotTakibiVar, RecAyniUrunTekrarEdebilir, EgemenFirmaId from sistem where sistemID = 1 into :OrmeUretimeCikistaLotTakibiVar, :RecAyniUrunTekrarEdebilir, :EgemenFirmaId;

  select DokumaReceteAtkiLotGuncelle from Kullanici where KullaniciId = new.KullaniciId into :DokumaReceteAtkiLotGuncelle;

  if ( Old.Fason <> New.Fason or Old.HamId <> New.HamId or IntKarsilastir(Old.RenkId, New.RenkId) = 0 or
       ( StrKarsilastir(Old.Lot, New.Lot) = 0 and OrmeUretimeCikistaLotTakibiVar = 1 and (:DokumaReceteAtkiLotGuncelle = 0 or new.AtkiCozgu <> 1)) ) then
  begin
    Select Count(*) from StokHareket
      where StokHareket.DokumaPlanlamaId = New.DokumaPlanlamaId and StokHareket.HamId = Old.HamId and IntKarsilastir(StokHareket.RenkId, Old.RenkId) = 1 and
            ( (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis)  or
              (StokHareket.HamIade = 1 and (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim))) into :KayitSayisi;
    if (KayitSayisi > 0 and EgemenFirmaId<>77 and EgemenFirmaId<>138) then exception UretimeCikisvarReceteDegisemez;
  end

  if (RecAyniUrunTekrarEdebilir = 0 and (Old.HamId <> New.HamId or IntKarsilastir(Old.RenkId, New.RenkID) = 0 or StrKarsilastir(Old.Lot, New.Lot) = 0 or
    IntKarsilastir(Old.AtkiCozgu, New.AtkiCozgu) = 0)) then
  begin
   Select Count(*) from DokumaPlanlamaRecete
     where DokumaPlanlamaRecete.DokumaPlanlamaID = new.DokumaPlanlamaId and DokumaPlanlamaRecete.HamId = new.HamId and
      IntKarsilastir(DokumaPlanlamaRecete.RenkId, New.RenkID) = 1 and
      IntKarsilastir(DokumaPlanlamaRecete.AtkiCozgu, New.AtkiCozgu) = 1 and
      StrKarsilastir(DokumaPlanlamaRecete.Lot, New.Lot) = 1 into :KayitSayisi;

   if (KayitSayisi > 0) then exception RecetedeBuUrunZatenVar;
  end

  Select OrmeSiparisDetay.TahminiFireOrani from DokumaPlanlama
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    where DokumaPlanlama.DokumaPlanlamaId = new.DokumaPlanlamaId
    into :TahminiFireOrani;

  if (new.TahminiFireOrani is Null) then new.TahminiFireOrani = :TahminiFireOrani;

  if (new.DegisimTarihi is null) then New.DegisimTarihi = current_TimeStamp(0);
end
