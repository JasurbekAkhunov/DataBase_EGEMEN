Definition

CREATE TRIGGER AFTERINSERTBUKUMTALIMAT FOR BUKUMTALIMAT
ACTIVE AFTER INSERT POSITION 0
as
declare variable FHamId           integer;
declare variable KullanimOrani    double precision;
declare variable IplikCinsiMelanj smallint;
declare variable Fason            smallint;
begin
  for Select HamStokRecete.ReceteHamId, HamStokRecete.KullanimOrani, IplikCinsi.Melanj from HamStokRecete
          left join ham on ham.hamID = HamStokRecete.ReceteHamId
          left join IplikCinsi on IplikCinsi.IplikCinsiID = Ham.IplikCinsiId
          where HamStokRecete.HamID = new.hamid
        into :FHamId, :KullanimOrani, :IplikCinsiMelanj do
        begin
			Select First 1 BoyaSiparis.Fason from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitId = new.BoyaSiparisKayitId into :Fason;
            Insert into BukumTalimatRecete(BukumTalimatid, HamId, KullanimOrani, Fason) Values (new.BukumTalimatid, :FHamId, :KullanimOrani, :Fason);
         end
end
