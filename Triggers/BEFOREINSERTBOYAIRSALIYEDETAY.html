Definition

CREATE TRIGGER BEFOREINSERTBOYAIRSALIYEDETAY FOR BOYAIRSALIYEDETAY
ACTIVE BEFORE INSERT POSITION 0
as
declare variable IadeEdilecekMiktar Double Precision;
declare variable Durumu Integer;
declare variable PartiKilo        Numeric(12,3);
declare variable PartiMetre       Numeric(12,3);
declare variable PartiMiktar      Numeric(12,3);
declare variable PartiNetMiktar   Numeric(12,3);
declare variable PartiSevkMiktar  Numeric(12,3);
declare variable SevkBrutMiktar   Numeric(12,3);
declare variable ParcaliSevk integer;
declare variable TamamiTartildi integer;
declare variable AsamaKisaAdi varchar(10);
declare variable IrsaliyeFiyatlandirmaKontrolu integer;
declare variable FaturaKesilecek integer;
declare variable IlaveIslemlerAdi varchar(30);
declare variable PartiSinif varchar(1);
declare variable HamStokSinif varchar(1);
declare variable IrsaliyeSinif varchar(1);
declare variable KayitSayisi integer;
declare variable biSatis                integer;
declare variable biAlisIade             integer;
declare variable biSatisIade            integer;
declare variable FireOnayiIstensin      integer;
declare variable FireOnayiKullaniciId   integer;
declare variable KantarKayitId          integer;
declare variable OnayDurumu             integer;
declare variable IrsaliyedeFasonSatisKarisik    integer;
declare variable IrsaliyedeTamirNormalKarisik   integer;
declare variable IrsaliyedeIplikKumasKarisik    integer;
declare variable IrsaliyedeIadePartiKarisik     integer;
declare variable Fason                  integer;
declare variable IslemTipiId            integer;
declare variable itTamir                integer;
declare variable BirimFiyat             Double Precision;
declare variable HamTipiId              integer;
declare variable FirmaSevkAdresiId      integer;
declare variable HamStokSeriNoKontrolu  integer;
Declare variable FStokFirmaId varchar(20);
Declare variable BirinciKaliteSevkEdilemez integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable FKEnGramajGirmedenSevkEdilemez integer;
declare variable IrsFarkliDovizBirlestiremez integer;
declare variable IrsFarkliImalatTBirlestiremez integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstAksesuar    integer;
declare variable hstTelef       integer;
declare variable EnVerilen      varchar(10);
declare variable GramajVerilen  varchar(10);
declare variable SiparisDetayHamId integer;
declare variable SiparisDetayKaliteId varchar(20);
declare variable HamGirisKaliteId varchar(20);
declare variable SiparisFirmaId varchar(20);
declare variable SatisFirmaId varchar(20);
declare variable PartiDovizID varchar(5);
declare variable SonDonemBitisTarihi TimeStamp;
declare variable ImalatTipiId          integer;
declare variable OrmeFirmaKodumuz varchar(20);
declare variable SevkteAdresKontroluYapilsin integer;
declare variable FaturaKesilenKayitSayisi integer;
declare variable EgemenFirmaID integer;
declare variable TransferEdildi EgeBoolean;
declare variable RiskVar integer;
declare variable SiparisRiskKontroluYap integer;
begin
  Select TransferEdildi From BoyaIrsaliye Where BoyaIrsaliyeId = new.BoyaIrsaliyeId into :TransferEdildi;
  if (TransferEdildi = 1) then exception TrasferEdilmisDegistirilmez;

  Select FKEnGramajGirmedenSevkEdilemez, DokumaKumasiKilodanTakipEt, IrsaliyeFiyatlandirmaKontrolu, IrsaliyedeFasonSatisKarisik, IrsaliyedeTamirNormalKarisik,
         IrsaliyedeIplikKumasKarisik, IrsaliyedeIadePartiKarisik, HamStokSeriNoKontrolu, IrsFarkliDovizBirlestiremez, IrsFarkliImalatTBirlestiremez,
         OrmeFirmaKodumuz, SevkteAdresKontroluYapilsin, EgemenFirmaID, SiparisRiskKontroluYap from Sistem
  into :FKEnGramajGirmedenSevkEdilemez, :DokumaKumasiKilodanTakipEt, :IrsaliyeFiyatlandirmaKontrolu, :IrsaliyedeFasonSatisKarisik, :IrsaliyedeTamirNormalKarisik,
           :IrsaliyedeIplikKumasKarisik, :IrsaliyedeIadePartiKarisik, :HamStokSeriNoKontrolu, :IrsFarkliDovizBirlestiremez, :IrsFarkliImalatTBirlestiremez,
           :OrmeFirmaKodumuz, :SevkteAdresKontroluYapilsin, :EgemenFirmaID, :SiparisRiskKontroluYap;

  if (new.BrutKilo is null) then new.BrutKilo = 0;

  Select hstIplik, hstElyaf, hstAksesuar, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstAksesuar, :hstTelef;

  Select biSatis, biAlisIade, biSatisIade from BoyaIrsaliyeHareketTipiGetir into :biSatis, :biAlisIade, :biSatisIade;

  Select BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.Sinif, BoyaIrsaliye.FirmaId, BoyaIrsaliye.HareketTipiId, KantarKayitId from BoyaIrsaliye where BoyaIrsaliye.BoyaIrsaliyeId = new.BoyaIrsaliyeId
       into New.IrsaliyeTarihi, :IrsaliyeSinif, new.FirmaId, new.HareketTipiId, :KantarKayitId;

  Select count(*) from BoyairsaliyeDetay where BoyairsaliyeDetay.boyairsaliyeid = new.boyairsaliyeid and BoyairsaliyeDetay.BoyaFaturaid is not null
  into FaturaKesilenKayitSayisi;

  if (FaturaKesilenKayitSayisi > 0) then exception FaturasiKesilmisSilinemez;

  if (KantarKayitId is not null) then
  begin
    Select OnayDurumu from Kantar where Kantar.KantarKayitID = :KantarKayitId into :OnayDurumu;
    if (OnayDurumu = 1) then exception KantarOnayiVerilmisDegisemez;
  end

  if (new.RafDetayId is not null) then
    select RafDetay.HamId, RafDetay.KaliteNo, HamStokMaster.OlcuId from RafDetay
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    where RafDetay.RafDetayId = new.RafDetayId into New.HamId, new.KaliteNo, new.OlcuId;
  else
    select BoyaSiparisDetay.HamId from PartiStok
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      where PartiStok.PartiSinifId = new.PartiSinifId into new.HamId;

  Select Ham.HamTipiId from Ham where Ham.HamID = New.HamId into :HamTipiId;

  if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
  begin
    New.Miktar = New.kilo;
    New.BrutMiktar = New.Brutkilo;
    New.HamStokBirimId = 0;
  end
  else
  begin
    New.Miktar = New.Metre;
    New.BrutMiktar = New.BrutMetre;
    New.HamStokBirimId = 1;
  end

  if (new.PartiSinifId is not null) then
  begin
    if (SevkteAdresKontroluYapilsin = 1) then
    begin
      Select BoyaIrsaliye.FirmaSevkAdresiId from BoyaIrsaliye
        where BoyaIrsaliye.BoyaIrsaliyeId = new.BoyaIrsaliyeId
        into :FirmaSevkAdresiId;

      KayitSayisi = 0;
      Select Count(*) from PartiStok
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
        where PartiStok.PartiSinifId = new.PartiSinifId and PartiStok.FirmaSevkAdresiId <> :FirmaSevkAdresiId and Parti.FirmaId = new.FirmaId
        into :KayitSayisi;

      if (KayitSayisi > 0) then exception SiparisSevkAdresiUygunDegil;
    end

    if (FKEnGramajGirmedenSevkEdilemez = 1 and (HamTipiId <> hstIplik and HamTipiId <> hstElyaf and HamTipiID <> hstAksesuar and HamTipiID <> hstTelef)) then
    begin
      Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
        where PartiStokFizikKontrol.partiSinifID = New.partiSinifID and PartiStokFizikKontrol.SonKayit = 1
        into :EnVerilen, :GramajVerilen;

      if (EnVerilen is null or EnVerilen = '' or GramajVerilen is null or GramajVerilen = '') then exception EnGramajGirmedenSevkEdilemez;
    end
  end

  if (IrsaliyedeIadePartiKarisik = 0) then
  begin
    KayitSayisi = 0;

    if (new.HamIadeRezervId is not null) then
    begin
      Select Count(*) from BoyaIrsaliyeDetay
        where BoyaIrsaliyeDetay.BoyaIrsaliyeId = New.BoyaIrsaliyeId and BoyaIrsaliyeDetay.PartiSinifId is not null
        into :KayitSayisi;
    end
    else if (new.partiSinifId is not null) then
    begin
      Select Count(*) from BoyaIrsaliyeDetay
        where BoyaIrsaliyeDetay.BoyaIrsaliyeId = New.BoyaIrsaliyeId and BoyaIrsaliyeDetay.HamIadeRezervId is not null
        into :KayitSayisi;
    end
    if (KayitSayisi > 0) then exception IrsaliyedeIadePartiKarisik;
  end

  if (new.RafDetayId is not null) then
  begin
    Select HamStokIrsaliye.Sinif, HamStokDetay.StokFirmaId  from RafDetay
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
      where RafDetay.RafDetayId = new.RafDetayId
      into :HamStokSinif, :FStokFirmaId;

    if (IrsaliyeSinif <> HamStokSinif) then exception IrsaliyeIleHamStokSinifTutmuyor;

    if (new.FasonHamIade = 1 or new.hareketTipiId = biAlisIade) then
    begin
      if (new.FasonHamIade = 1) then new.FaturaKesilecek = 0;
      if (new.HamIadeRezervId is null) then exception HamIadeIcinRezervSecilmemis;
      else
      begin
        Select count(*) from HamIadeRezerv where HamIadeRezerv.HamIadeRezervId = New.HamIadeRezervId and HamIadeRezerv.HamStokIadeSebebiId is null into :KayitSayisi;
        if (KayitSayisi > 0) then exception IadeSebebiGirilmedenIadeOlamaz;
      end

      select IadeEdilecekMiktar from RafDetayKalanMiktarGetir(new.RafDetayId) into :IadeEdilecekMiktar;

      if (IadeEdilecekMiktar is null or IadeEdilecekMiktar < new.Miktar) then exception IadeEdilecekKiloyuAstiniz;

      Update HamIadeRezerv set IadeEdilenKapAdedi = IadeEdilenKapAdedi + New.KapAdedi,
                               IadeEdilenKilo = IadeEdilenKilo + New.Kilo,
                               IadeEdilenMetre = IadeEdilenMetre + New.Metre,
                               IadeEdilenAdet = IadeEdilenAdet + New.Adet
                         where HamIadeRezerv.HamIadeRezervId = New.HamIadeRezervId;
    end

    if (new.FasonHamIade = 1) then execute procedure HamStokKumilatifGuncelle(:FStokFirmaId, 1, 3, Null, New.HamId, Null, new.IrsaliyeTarihi, Null, Null, New.Kilo, New.Metre);
    else if (new.hareketTipiId = biAlisIade) then execute procedure HamStokKumilatifGuncelle(new.FirmaId, 0, 3, Null, New.HamId, Null, new.IrsaliyeTarihi, Null, Null, New.Kilo, New.Metre);
    else execute procedure HamStokKumilatifGuncelle(new.FirmaId, 0, 4, Null, New.HamId, Null, new.IrsaliyeTarihi, Null, Null, New.Kilo, New.Metre);
  end

  if (new.PartiSinifId is not null) then
  begin
    if (OrmeFirmaKodumuz = New.FirmaID) then
    begin
      select Coalesce(BoyaSiparisDetay.FasonBirimFiyati, 0), BoyaSiparisDetay.FasonBirimFiyatDovizId
      from PartiStok
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      where PartiStok.PartiSinifId = new.PartiSinifId into new.BirimFiyat, new.DovizId;
    end

    select Parti.PartiKayitId, Parti.Sinif, Parti.FaturaKesilecek, Parti.ParcaliSevk, Parti.Fason, Parti.IslemTipiId,
           PartiStok.Durumu, PartiStok.Kilo, PartiStok.Metre, PartiStok.Miktar, PartiStok.NetMiktar, PartiStok.NetSevkMiktar - PartiStok.NetIadeMiktar, PartiStok.TamamiTartildi, PartiStok.FireOnayiIstensin, PartiStok.FireOnayiKullaniciId,
           PartiStok.BirimFiyat, Parti.FirmaId, PartiStok.BirinciKaliteSevkEdilemez, PartiStok.DovizID, Parti.ImalatTipiId from PartiStok
      left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      where PartiStok.PartiSinifId = new.PartiSinifId
      into new.PartiKayitId, :PartiSinif, new.FaturaKesilecek, :ParcaliSevk, :Fason, :IslemTipiId,
           :Durumu, :PartiKilo, :PartiMetre, :PartiMiktar, :PartiNetMiktar, :PartiSevkMiktar, :TamamiTartildi,
           :FireOnayiIstensin, :FireOnayiKullaniciId, :BirimFiyat, :FStokFirmaId, :BirinciKaliteSevkEdilemez, PartiDovizID, :ImalatTipiId;

    if (IrsaliyeFiyatlandirmaKontrolu = 1) then
    if (new.FaturaKesilecek = 1 and BirimFiyat = 0 and OrmeFirmaKodumuz <> New.FirmaID) then exception FiyatYokIrsaliyeKesilemez;

    if (Durumu is null) then exception PartiStokSecilmemis;

    if (BirinciKaliteSevkEdilemez = 1 and New.KaliteNo = 1) then exception KayitHatasi 'Bu Stok Icin Birinci Kalite Sevk Edilemez İsaretlenmis..';

    if (HamStokSeriNoKontrolu = 1) then
    if (IrsaliyeSinif <> PartiSinif) then exception IrsaliyeIlePartiSeriNoTutmuyor;

    if (IrsaliyedeFasonSatisKarisik = 0) then
    begin
      Select Count(*) from BoyaIrsaliyeDetay
        left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
        where BoyaIrsaliyeDetay.BoyaIrsaliyeId = New.BoyaIrsaliyeId and Parti.Fason <> :Fason into :KayitSayisi;

      if (KayitSayisi > 0) then exception IrsaliyedeFasonSatisKarisik;
    end

    if (IrsaliyedeTamirNormalKarisik = 0) then
    begin
      Select itTamir from IslemTipiGetir into :itTamir;

      Select Count(*) from BoyaIrsaliyeDetay
        left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
        where BoyaIrsaliyeDetay.BoyaIrsaliyeId = New.BoyaIrsaliyeId and
              (Parti.IslemTipiId <> :IslemTipiId and (:IslemTipiId = :itTamir or Parti.IslemTipiId = :itTamir))
        into :KayitSayisi;

      if (KayitSayisi > 0) then exception IrsaliyedeTamirNormalKarisik;
    end

    if (IrsaliyedeIplikKumasKarisik = 0) then
    begin
      if (HamTipiId = 0) then
      begin
        Select Count(*) from BoyaIrsaliyeDetay
          left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
          where BoyaIrsaliyeDetay.BoyaIrsaliyeId = New.BoyaIrsaliyeId and Ham.HamTipiId <> :HamTipiId into :KayitSayisi;
      end
      else
      begin
        Select Count(*) from BoyaIrsaliyeDetay
          left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
          where BoyaIrsaliyeDetay.BoyaIrsaliyeId = New.BoyaIrsaliyeId and Ham.HamTipiId = 0 into :KayitSayisi;
      end

      if (KayitSayisi > 0) then exception IrsaliyedeIplikKumasKarisik;
    end

    if (IrsFarkliImalatTBirlestiremez = 1) then
    begin
      /*imalat tipinin baskı seçilmiş olmasına ilaveten, Parti aşamalarında baskı asaması*/
      /*olanları baskı olarak kabul ediyoruz ondan ayrıntılı kontrol gerek*/

      if (ImalatTipiId = 1) then /* baski mi*/
      begin
        Select Count(*) from PartiAsamalari
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
          where PartiAsamalari.PartiKayitID = new.PartiKayitID and Asama.BaskiAsamasimi = 1
        into :KayitSayisi;
      end

      if (ImalatTipiId = 1 and KayitSayisi > 0) then /* baski ise */
      begin
        Select Count(*) from BoyaIrsaliyeDetay
          left join PartiAsamalari on PartiAsamalari.PartiKayitID = BoyaIrsaliyeDetay.PartiKayitID
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
          where BoyaIrsaliyeDetay.BoyaIrsaliyeID = new.BoyaIrsaliyeID and Asama.BaskiAsamasimi = 1
        into :KayitSayisi;

        if (KayitSayisi = 0) then
        begin
          Select Count(*) from BoyaIrsaliyeDetay where BoyaIrsaliyeDetay.BoyaIrsaliyeID = new.BoyaIrsaliyeID into :KayitSayisi;
          if (KayitSayisi > 0) then exception IrsFarkliImalatTBirlestiremez;
        end
      end
      else /* boya ise */
      begin
        Select Count(*) from BoyaIrsaliyeDetay
          left join PartiAsamalari on PartiAsamalari.PartiKayitID = BoyaIrsaliyeDetay.PartiKayitID
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
          where BoyaIrsaliyeDetay.BoyaIrsaliyeID = new.BoyaIrsaliyeID and Asama.BaskiAsamasimi = 1
        into :KayitSayisi;

        if (KayitSayisi > 0) then exception IrsFarkliImalatTBirlestiremez;
      end
    end
    if (ParcaliSevk = 0) then
    begin
      if (Durumu < 7 or TamamiTartildi = 0) then exception PartiStokSevkeHazirDegil;

      if (FireOnayiIstensin = 1 and FireOnayiKullaniciId is null) then exception FireOraniYuksekOnceOnayAliniz;

      Select count(*) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = new.PartiSinifId and PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamaStok.CikisTarihi is null into :KayitSayisi;

      if (KayitSayisi > 0) then exception PartiStokSevkeHazirDegil;

      if (PartiNetMiktar is null or PartiNetMiktar = 0) then exception TartiIslemiYapilmamis;
      if (PartiNetMiktar < new.Miktar) then exception NetKiloMetreAstiniz;
      if (PartiNetMiktar > new.Miktar) then exception ParcaliSevkEdemezsiniz;
      if (PartiMiktar <> new.BrutMiktar) then
      begin
        new.BrutKilo = PartiKilo;
        new.BrutMetre = PartiMetre;
        new.BrutMiktar = PartiMiktar;
      end
    end
    else
    begin
      if (TamamiTartildi = 1) then
      if (FireOnayiIstensin = 1 and FireOnayiKullaniciId is null) then exception FireOraniYuksekOnceOnayAliniz;

      if (new.HareketTipiID <> biSatisIade) then
      begin
        Select sum(iif(BoyaIrsaliyeDetay.HareketTipiID = :biSatisIade, BoyaIrsaliyeDetay.BrutMiktar * -1, BoyaIrsaliyeDetay.BrutMiktar)) from BoyaIrsaliyeDetay where BoyaIrsaliyeDetay.PartiSinifId = new.PartiSinifId into :SevkBrutMiktar;

        if (SevkBrutMiktar is null) then SevkBrutMiktar = 0;
        if (PartiNetMiktar < (PartiSevkMiktar + New.Miktar)) then exception NetKiloMetreAstiniz;
        if (PartiMiktar < (SevkBrutMiktar + New.BrutMiktar)  and EgemenFirmaID <> 75 and EgemenFirmaID <> 37 and EgemenFirmaID <> 110) then exception SevkBrutKiloyuAstiniz; /*mersu, Arta, Ayyıldız*/
      end
    end

    if (IrsFarkliDovizBirlestiremez = 1) then
    begin
      Select Count(*) from BoyaIrsaliyeDetay
        left Join PartiStok on PartiStok.PartiSinifID = BoyaIrsaliyeDetay.PartiSinifID
        where BoyaIrsaliyeID = new.BoyaIrsaliyeId and PartiStok.DovizID <> :PartiDovizID
        into :KayitSayisi;

      if (KayitSayisi > 0) then exception IrsFarkliDovizBirlestiremez;
    end

    new.YapilanIslemler = '';
    if (EgemenFirmaID = 37) then /* arta */
    begin
      for select Coalesce(Asama.Asamakisaadi, '') from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join Recete on Recete.Receteid = PartiAsamalari.Receteid
        left join Asama on Asama.Asamaid = PartiAsamalari.Asamaid or Asama.Asamaid = Recete.UstAsamaId
        left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
        left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliId = PartiTamir.TamirCozumSekliId
        where PartiAsamaStok.PartiSinifId = new.PartiSinifId and Asamayadahil = 1 and Asama.Asamakisaadi is not null and Asama.Asamakisaadi <> ''
        order by PartiAsamalari.Sira into :AsamaKisaAdi do
      begin
        if (Char_Length(new.YapilanIslemler) + Char_Length(AsamaKisaAdi) < 253) then
        begin
         if (new.YapilanIslemler <> '' ) then new.YapilanIslemler = new.YapilanIslemler || cast(', ' as varchar(2) character set UTF8);
         new.YapilanIslemler = new.YapilanIslemler || AsamaKisaAdi;
        end
      end

      for Select IlaveIslemler.Adi from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
        left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
        left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliId = PartiTamir.TamirCozumSekliId
        where PartiAsamaStok.PartiSinifId = new.PartiSinifId and Asamayadahil = 1 and IlaveIslemler.Adi is not null
        into :IlaveIslemlerAdi do
      begin
        if (Char_Length(new.YapilanIslemler) + Char_Length(IlaveIslemlerAdi) < 253) then
        begin
         if (new.YapilanIslemler <> '' ) then new.YapilanIslemler = new.YapilanIslemler || cast(', ' as varchar(2) character set UTF8);
         new.YapilanIslemler = new.YapilanIslemler || IlaveIslemlerAdi;
        end
      end
    end
    else
    begin
      for select Coalesce(Asama.Asamakisaadi, '') from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join Recete on Recete.Receteid = PartiAsamalari.Receteid
        left join Asama on Asama.Asamaid = PartiAsamalari.Asamaid or Asama.Asamaid = Recete.UstAsamaId
        left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
        left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliId = PartiTamir.TamirCozumSekliId
        where PartiAsamaStok.PartiSinifId = new.PartiSinifId and Asamayadahil = 1 and Asama.Asamakisaadi is not null and Asama.Asamakisaadi <> '' and
              (PartiAsamalari.PartiTamirID is null or TamirCozumSekli.Ucretli = 1) and
              Asama.ParasalDegeriVar = 1
        order by PartiAsamalari.Sira into :AsamaKisaAdi do
      begin
        if (Char_Length(new.YapilanIslemler) + Char_Length(AsamaKisaAdi) < 253) then
        begin
         if (new.YapilanIslemler <> '' ) then new.YapilanIslemler = new.YapilanIslemler || cast(', ' as varchar(2) character set UTF8);
         new.YapilanIslemler = new.YapilanIslemler || AsamaKisaAdi;
        end
      end

      for Select IlaveIslemler.Adi from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
        left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
        left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliId = PartiTamir.TamirCozumSekliId
        where PartiAsamaStok.PartiSinifId = new.PartiSinifId and Asamayadahil = 1 and IlaveIslemler.Adi is not null and
              (PartiAsamalari.PartiTamirID is null or TamirCozumSekli.Ucretli = 1)
        into :IlaveIslemlerAdi do
      begin
        if (Char_Length(new.YapilanIslemler) + Char_Length(IlaveIslemlerAdi) < 253) then
        begin
         if (new.YapilanIslemler <> '' ) then new.YapilanIslemler = new.YapilanIslemler || cast(', ' as varchar(2) character set UTF8);
         new.YapilanIslemler = new.YapilanIslemler || IlaveIslemlerAdi;
        end
      end
    end
    execute procedure HamStokKumilatifGuncelle(:FStokFirmaId, :Fason, 5, Null, New.HamId, Null, New.IrsaliyeTarihi, Null, Null, New.BrutKilo, New.BrutMetre);
  end

  if (new.BoyaSiparisDetayId is not Null) then
  begin
    if (OrmeFirmaKodumuz = New.FirmaID) then
    begin
      select Coalesce(BoyaSiparisDetay.FasonBirimFiyati, 0), BoyaSiparisDetay.FasonBirimFiyatDovizId
      from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayId = new.BoyaSiparisDetayId into new.BirimFiyat, new.DovizId;
    end

    /*Siparisli ham stok satisi*/
    if (new.RafDetayId is Null) then Exception SiparisliHamSatisUyumsuzluk 'Raf detay bilgisi eksik';
    /*Ham satis degilse, boya siparis detay bilgisi bulunamaz...*/

    select BoyaSiparisDetay.HamId, BoyaSiparisDetay.KaliteId, BoyaSiparis.FirmaId from BoyaSiparisDetay
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
      where BoyaSiparisDetay.BoyaSiparisDetayId = new.BoyaSiparisDetayId
      into :SiparisDetayHamId, :SiparisDetayKaliteId, :SiparisFirmaId;

    Select HamStokMaster.KaliteId from RafDetay
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    where RafDetay.RafDetayId = new.RafDetayId
    into :HamGirisKaliteId;

    select BoyaIrsaliye.FirmaId from BoyaIrsaliye
      where BoyaIrsaliye.BoyaIrsaliyeId = new.BoyaIrsaliyeId
      into :SatisFirmaId;

    if (new.HamId <> SiparisDetayHamId) then Exception HamGirisveHamSatisUyumsuzluk 'Ham Kodu Farklı';
    if (HamGirisKaliteId <> SiparisDetayKaliteId) then Exception HamGirisveHamSatisUyumsuzluk 'Kalite kodu farklı';
    if (SatisFirmaId <> SiparisFirmaId) then Exception HamGirisveHamSatisUyumsuzluk 'Siparişle irsaliye firma farklı';
  end

  select max(BitisTarihi) from DonemKapatma where Bolum = 1 and OnayKullaniciId is not Null into :SonDonemBitisTarihi;
  if (SonDonemBitisTarihi is not Null and SonDonemBitisTarihi > new.IrsaliyeTarihi) then
    exception DonemKapaliHEklenemez;

  if (new.HamStokBirimId = 0 and new.PartiSinifId is not null) then
  begin
    select BoyaSiparisDetay.HamStokBirimId from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiSinifId = new.PartiSinifId into new.HamStokBirimId;
  end

  if (SiparisRiskKontroluYap = 2) then /*Fayteks*/
  begin
    Select RiskVar from SiparisRiskiniGetir(new.FirmaId) into :RiskVar;
    if (RiskVar = 1) then Exception RiskLimitiAsilamaz 'Bakiye Aşımı';
  end

  if (new.BoyaIrsaliyeDetayId is null) then new.BoyaIrsaliyeDetayId = gen_id(BoyaIrsaliyeDetay_Id_Gen, 1);
end
