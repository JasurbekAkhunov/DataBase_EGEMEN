CREATE TRIGGER AFTERUPDATEORMEIRSALIYE FOR ORMEIRSALIYE
ACTIVE AFTER UPDATE POSITION 0
as
begin
  if ( (old.OrmeFaturaId is null and new.OrmeFaturaId is not null) or (old.OrmeFaturaId is not null and new.OrmeFaturaId is null) ) then
  begin
    update StokHareket Set OrmeFaturaId = new.OrmeFaturaId, NeredenUpdate = 3, KullaniciId = New.KullaniciId, LinkIslemi = 1 where StokHareket.OrmeIrsaliyeId = new.OrmeIrsaliyeId;

    update Ormefatura set KullaniciId = New.KullaniciId where OrmeFatura.OrmeFaturaId = new.OrmeFaturaId or OrmeFatura.OrmeFaturaId = old.OrmeFaturaId;
  end

  if (Old.IrsaliyeTarihi <> New.IrsaliyeTarihi or Old.FiiliSevkTarihi <> New.FiiliSevkTarihi or Old.DovizId <> New.DovizId or Old.TabanDovizKuru1 <> New.TabanDovizKuru1 or Old.TabanDovizKuru2 <> New.TabanDovizKuru2) then
    update StokHareket Set Tarih = New.IrsaliyeTarihi, FiiliSevkTarihi = new.FiiliSevkTarihi, DovizId = new.DovizId, TabanDovizKuru1 = New.TabanDovizKuru1, TabanDovizKuru2 = New.TabanDovizKuru2, NeredenUpdate = 1, LinkIslemi = 1 where StokHareket.OrmeIrsaliyeId = new.OrmeIrsaliyeId and StokHareket.NeredenGirildi = 1;

  if (Old.KdvDahil <> New.kdvDahil) then /* KdvHariFiyat yeniden hesaplansın */
    update StokHareket Set KdvHaricFiyat = 0, NeredenUpdate = 1, LinkIslemi = 1 where StokHareket.OrmeIrsaliyeId = new.OrmeIrsaliyeId;

  Execute Procedure Logla('U', 2, 'ORMEIRSALIYE', 'ORMEIRSALIYEID', new.OrmeIrsaliyeId, 'Firma Kodu:'||new.FirmaId||' İrsaliye No :'||Coalesce(new.IrsaliyeNo,'') ||' Tarih:'||Extract(DAY from new.IrsaliyeTarihi)||'.'||Extract(MONTH from new.IrsaliyeTarihi)||'.'||Extract(YEAR from new.IrsaliyeTarihi));
end
