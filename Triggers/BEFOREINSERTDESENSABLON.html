CREATE TRIGGER BEFOREINSERTDESENSABLON FOR DESENSABLON
ACTIVE BEFORE INSERT POSITION 0
as
declare variable Sira integer;
begin
  if (new.SablonAdi is null) then
  begin
    Sira = new.SablonSiraNo;
    if (Sira > 8) then Sira = Sira + 1;
    new.SablonAdi = chr(Sira+64);
  end
  new.DesenSablonId = Gen_Id(DesenSablon_Id_gen,1);
end
