CREATE TRIGGER AFTERINSERTORMEMAKINADURUMU FOR ORMEMAKINADURUMU
ACTIVE AFTER INSERT POSITION 0
as
begin
  Update MakinaDurumuOzet set OrmeMakinaDurumuID = new.OrmeMakinaDurumuId,
   MemingerKoptu = new.MemingerKoptu,
   AltMemingerKoptu = New.AltMemingerKoptu,
   StartStop = New.StartStop,
   IgneKoptu = New.IgneKoptu,
   LycraKoptu = New.LycraKoptu,
   KapakAcildi = New.KapakAcildi,
   KenarStop = New.KenarStop,
   CozguKoptu = New.CozguKoptu,
   AtkiKoptu = New.AtkiKoptu,
   LeventSonu = New.LeventSonu,
   Tur = New.Tur,
   PersonelID = New.PersonelID,
   TemizlikYapildi = New.TemizlikYapildi,
   TopKesme = New.TopKesme
   where (OrmeMakinaId = new.OrmeMakinaId or DokumaMakinaId = new.DokumaMakinaId);
end
