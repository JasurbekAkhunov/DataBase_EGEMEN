CREATE TRIGGER AFTERUPDATEPARTITAMIRSINIF FOR PARTITAMIRSINIF
ACTIVE AFTER UPDATE POSITION 0
as
declare variable Bitti integer;
begin
  Bitti = Null;
  if (old.Bitti <> New.Bitti) then
  begin
    Select Min(Bitti) from PartiTamirSinif where PartiTamirSinif.PartiTamirId = New.PartiTamirId into :Bitti;
    Update PartiTamir Set Bitti = :Bitti where PartiTamirId = New.PartiTamirId;
  end
end
