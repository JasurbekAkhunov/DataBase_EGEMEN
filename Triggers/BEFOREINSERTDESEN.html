Definition

CREATE TRIGGER BEFOREINSERTDESEN FOR DESEN
ACTIVE BEFORE INSERT POSITION 0
as
declare variable DesenSablonSayisiDigit integer;
declare variable DesenNoDigit           integer;
declare variable MaxDesenId             varchar(20);
declare variable STR                    varchar(20);
declare variable FSablonSayisi          integer;
declare variable FDesenSayisi           integer;
declare variable TF                     integer;
declare variable FDesenId               varchar(20);
declare variable DesenKoduFormatiSerbest integer;
declare variable DesenNoOtomatik        integer;
declare variable IlkKarakter            integer;
declare variable BaslangicKarakter      varChar(1);
declare variable FAdi                   varChar(50);
declare variable KS                     integer;
declare variable FEgemenFirmaId         integer;
declare variable Yil                    varchar(2);
declare variable Seri                   varchar(1);
declare variable MaxTasarimNo           Integer;
declare variable DesenUstBilgiTasarimNo EgeBoolean;
begin
  if (COALESCE(rdb$get_context('USER_TRANSACTION', 'BaypasTrigger_Desen'), 0) <> 1) then  -- NULL olabiliyor, COALESCE'yi iptal etmeyin. (UP)
  begin
    Select DesenSablonSayisiDigit, DesenNoDigit, DesenNoOtomatik, DesenKoduFormatiSerbest, EgemenFirmaId, DesenUstBilgiTasarimNo from Sistem
      into :DesenSablonSayisiDigit, :DesenNoDigit, :DesenNoOtomatik, :DesenKoduFormatiSerbest, :FEgemenFirmaId, DesenUstBilgiTasarimNo;

    if (new.BaskiYonu = 0) then New.BaskiYonuVar = 0;
    else New.BaskiYonuVar = 1;

    BaslangicKarakter = '';
    if (FEgemenFirmaId = 2) then --> Batı Basma
    begin
      if (Coalesce(New.TasarimNo, '') <> '' and New.DijitalDesen = 0) then
      if (new.SablonSayisi is null or new.SablonSayisi = 0) then exception KayitHatasi 'Sablon Sayısı girilmemis..';

      Select Cast(EgemenCopyStr(Coalesce(Max(DesenID), extract(year from Current_Date)||'-0000'),
      PosBul(Coalesce(Max(DesenID), extract(year from Current_Date)||'-0000'), '-') + 1, 4) as Integer) + 1 from Desen where DijitalDesen = new.DijitalDesen and
      Desen.DesenId like '%' || extract(year from Current_Date)||'-%' into :MaxTasarimNo;

      New.DesenID = extract(year from Current_Date) || '-' || EgemenIntToLzStr(MaxTasarimNo, 4);

      if (New.DijitalDesen = 1) then
      begin
        New.DesenID =  'D' || New.DesenID;

        if (Coalesce(New.TasarimNo, '') = '') then
        begin
          Select Max(TasarimNo) from Desen where Desen.DijitalDesen = New.DijitalDesen into :MaxDesenId;

          if (MaxDesenId is null) then FDesenSayisi = 1;
          else
          begin
            STR = EgemenCopyStr(MaxDesenId, 2, 6);

            TF = EgemenStrToIntOlurmu(STR);

            if (TF = 0) then exception ManuelGirilenDesenNoHatali;
            else FDesenSayisi = EgemenStrToInt(STR);

            FDesenSayisi = FDesenSayisi + 1;
          end
          new.TasarimNo = 'D' || EgemenIntToLzStr(FDesenSayisi, 6);
        end
        else
        begin
          STR = EgemenCopyStr(New.TasarimNo, 2, 6);
          TF = EgemenStrToIntOlurmu(STR);

          if (TF = 0) then exception ManuelGirilenDesenNoHatali;

          if (EgemenCopyStr(New.TasarimNo, 1, 1) <> 'D') then exception ManuelGirilenDesenNoHatali;
        end
      end
      else
      begin
        if (New.SablonSayisi is not null and New.SablonSayisi > 0) then
        begin
          if (Coalesce(New.TasarimNo, '') = '') then
          begin
            Select Max(TasarimNo) from Desen where Desen.SablonSayisi = new.SablonSayisi and Desen.DijitalDesen = New.DijitalDesen into :MaxDesenId;

            if (MaxDesenId is null) then FDesenSayisi = 1;
            else
            begin
              STR = EgemenCopyStr(MaxDesenId, DesenSablonSayisiDigit+1, DesenNoDigit);

              TF = EgemenStrToIntOlurmu(STR);

              if (TF = 0) then exception ManuelGirilenDesenNoHatali;
              else FDesenSayisi = EgemenStrToInt(STR);

              FDesenSayisi = FDesenSayisi + 1;
            end
            new.TasarimNo = EgemenIntToLzStr(New.SablonSayisi, DesenSablonSayisiDigit) || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit);
          end
          else
          begin
            STR = EgemenCopyStr(New.TasarimNo, 1, DesenSablonSayisiDigit);
            TF = EgemenStrToIntOlurmu(STR);

            if (TF = 0) then exception ManuelGirilenDesenNoHatali;
            else FSablonSayisi = EgemenStrToInt(STR);

            if (FSablonSayisi <> New.SablonSayisi) then exception ManuelGirilenDesenNoHatali;
          end
        end
      end
    end
    else
    if (FEgemenFirmaId = 85) then --> Aka Teks
    begin
      if (new.DesenId is null) then
      begin
        select First 1 EgemenCopyStr(DesenId, 1, 1), Coalesce(Max(EgemenCopyStr(DesenId, Char_Length(DesenId) - 2, 3)), 0)
        from Desen where (Desen.DesenId like 'A%' or Desen.DesenId like 'B%' or Desen.DesenId like 'C%' or Desen.DesenId like 'D%') and
        Desen.SablonSayisi = new.SablonSayisi
        group by EgemenCopyStr(DesenId, 1, 1)
        order by EgemenCopyStr(DesenId, 1, 1) desc into :Seri, :KS;

        if (Seri is null) then Seri = 'A';
        else if (KS = 999) then
        begin
          if (Seri = 'A') then Seri = 'B';
          else if (Seri = 'B') then Seri = 'C';
          else if (Seri = 'C') then Seri = 'D';
        end
        if (KS is null) then KS = 1;
        else if (KS = 999) then KS = 1;
        else KS = KS + 1;

        new.DesenId = Seri || new.SablonSayisi || Lpad(KS, 3, '0');
      end
    end
    else
    if (New.DesenId is null and New.Fason = 0) then
    begin
      if (New.DijitalDesen = 1) then
      begin
        if (FEgemenFirmaId <> 64) then  --> Forte (değil ise)
        begin
          Select Max(DesenId) from Desen where Desen.DijitalDesen = new.DijitalDesen into :MaxDesenId;

          if (MaxDesenId is null) then
          begin
            if (FEgemenFirmaId = 14) then
             FDesenSayisi = 1000;
            else
             FDesenSayisi = 1;
          end
          else
          begin
            if (FEgemenFirmaId = 14) then
              STR = EgemenCopyStr(MaxDesenId, 4, 7);
            else if (FEgemenFirmaId = 106) then   /* Texco */
              STR = EgemenCopyStr(MaxDesenId, 4, DesenNoDigit);
            else if (FEgemenFirmaId = 131 or FEgemenFirmaId = 133) then   /* Ünver ve BTEKS Çorlu*/
              STR = EgemenCopyStr(MaxDesenId, 6, 6);
            else
              STR = EgemenCopyStr(MaxDesenId, 2, DesenNoDigit+DesenSablonSayisiDigit);

            TF = EgemenStrToIntOlurmu(STR);

            if (TF = 0) then exception ManuelGirilenDesenNoHatali;
            else FDesenSayisi = EgemenStrToInt(STR);

            FDesenSayisi = FDesenSayisi + 1;

          end
        end

        if (FEgemenFirmaId = 4) then --> Gülle
          new.DesenId = 'T' || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit + DesenSablonSayisiDigit);
        else if (FEgemenFirmaId = 64) then --> Forte
        begin
          Yil = EgemenCopyStr(F_year(current_date), 3, 2);
          select Max(EgemenStrToInt(Replace(Desen.DesenId, new.FirmaId || '-' || :Yil ||'-', ''))) from Desen where Desen.FirmaId = new.FirmaId and
          Desen.DesenId like new.FirmaId || '-' || :Yil ||'-%' and
          EgemenStrToIntOlurmu(Replace(Desen.DesenId, new.FirmaId || '-' || :Yil ||'-', '')) = 1 into :KS;

          if (KS is null or KS = 0) then
            KS = 1;
          else
            KS = KS + 1;

          new.DesenId = new.FirmaId || '-' || Yil || '-' || KS;
        end
        else if (FEgemenFirmaId = 106) then  /* Texco */
        begin
          Yil = EgemenCopyStr(F_year(current_date), 3, 2);
          select Max(EgemenStrToInt(Replace(Desen.DesenId,:Yil ||'-', ''))) from Desen
          where Desen.DesenId like :Yil ||'-%' and
          EgemenStrToIntOlurmu(Replace(Desen.DesenId, :Yil ||'-', '')) = 1 into :KS;

          if (KS is null or KS = 0) then
            KS = 1;
          else
            KS = KS + 1;

          new.DesenId = Yil || '-' || EgemenIntToLzStr(KS,  DesenNoDigit);
        end

        else
        begin
          if (FEgemenFirmaId=14) then
            new.DesenId = 'D-0' || FDesenSayisi;
          else if (FEgemenFirmaId = 99) then  --> Megam
            new.DesenId = 'DJ' || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit + DesenSablonSayisiDigit);
          else if (FEgemenFirmaId = 119) then  --> Marmara
            new.DesenId = 'DG-' || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit + DesenSablonSayisiDigit);
          else if (FEgemenFirmaId = 131) then  --> Ünver
            new.DesenId = 'RSDJ-' || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit + DesenSablonSayisiDigit);
          else if (FEgemenFirmaId = 133) then  --> BTEKS CORLU
            new.DesenId = 'BTDJ-' || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit + DesenSablonSayisiDigit);
          else
            new.DesenId = 'D' || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit + DesenSablonSayisiDigit);
        end
      end
      else
      begin
        if (FEgemenFirmaId = 67) then --> Sateks
        begin
          Select Max(substring(DesenId from 4 for 4)) from Desen where Desen.SablonSayisi = new.SablonSayisi and Desen.DijitalDesen = new.DijitalDesen into :MaxDesenId;

          if (MaxDesenId is not null) then
          begin
            if (EgemenStrtoIntOlurmu(MaxDesenId)=1) then
              FDesenSayisi = egemenstrtoint(MaxDesenId) + 1;
          end
          else
            FDesenSayisi = 1;

          new.DesenId =  EgemenIntToLzStr(New.SablonSayisi, DesenSablonSayisiDigit) || '-'|| EgemenIntToLzStr(FDesenSayisi, DesenNoDigit);
        end
        else
        if (FEgemenFirmaid = 99) then --> Megam
        begin

          select max(DesenId) from Desen where Desen.SablonSayisi = new.SablonSayisi and Desen.DijitalDesen = new.DijitalDesen and Desen.Fason = 0 into :MaxDesenId;

          if (MaxDesenId is null) then FDesenSayisi = 1;
          else
          begin
            STR = EgemenCopyStr(MaxDesenId, 1, 1);
            IlkKarakter = ascii_val(STR);

            if (IlkKarakter < 65) then
            begin
              BaslangicKarakter = '';
              STR = EgemenCopyStr(MaxDesenId, char_length(new.sablonsayisi)+1, DesenNoDigit);
            end
            else
            begin
              BaslangicKarakter = STR;
              STR = EgemenCopyStr(MaxDesenId, char_length(new.sablonsayisi)+3, DesenNoDigit);
            end

            TF = EgemenStrToIntOlurmu(STR);

            if (TF = 0) then exception ManuelGirilenDesenNoHatali;
            else FDesenSayisi = EgemenStrToInt(STR);

            if ((FDesenSayisi = 999 and DesenNoDigit = 3) or (FDesenSayisi = 9999 and DesenNoDigit = 4)) then
            begin
              FDesenSayisi = 1;

              if (BaslangicKarakter = '') then
                BaslangicKarakter = 'A';
              else
                BaslangicKarakter = ascii_char(IlkKarakter + 1);
            end
            else FDesenSayisi = FDesenSayisi + 1;
          end
          if (BaslangicKarakter = '') then
          new.DesenId =  EgemenIntToLzStr(new.SablonSayisi, 0) || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit);
          else
          new.DesenId = BaslangicKarakter || '-' || EgemenIntToLzStr(new.SablonSayisi, 0) || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit);
        end
        else
        if (FEgemenFirmaid = 131) then --> Ünver
        begin
         Select Max(DesenId) from Desen where Desen.SablonSayisi = new.SablonSayisi and Desen.DijitalDesen = new.DijitalDesen and Desen.Fason = 0 into :MaxDesenId;

          if (MaxDesenId is null) then FDesenSayisi = 0;
          else
          begin
            STR = EgemenCopyStr(MaxDesenId, 1, 1);
            IlkKarakter = ascii_val(STR);

            if (IlkKarakter < 65) then
            begin
              BaslangicKarakter = '';
              STR = EgemenCopyStr(MaxDesenId, DesenSablonSayisiDigit + 1, DesenNoDigit);
            end
            else
            begin
              BaslangicKarakter = STR;
              STR = EgemenCopyStr(MaxDesenId, DesenSablonSayisiDigit + 4, DesenNoDigit);
            end
            TF = EgemenStrToIntOlurmu(STR);

            if (TF = 0) then exception ManuelGirilenDesenNoHatali;
            else FDesenSayisi = EgemenStrToInt(STR);

            if ((FDesenSayisi = 999 and DesenNoDigit = 3) or (FDesenSayisi = 9999 and DesenNoDigit = 4)) then
            begin
              FDesenSayisi = 0;

              if (BaslangicKarakter = '') then
                BaslangicKarakter = 'A';
              else
                BaslangicKarakter = Ascii_Char(IlkKarakter + 1);
            end
            else FDesenSayisi = FDesenSayisi + 1;
          end

           new.DesenId = 'RS-' || EgemenIntToLzStr(New.SablonSayisi, DesenSablonSayisiDigit) || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit);
        end
        else
        if (FEgemenFirmaid = 133) then --> BTEKS ÇORLU
        begin
            Select Max(DesenId) from Desen where Desen.SablonSayisi = new.SablonSayisi and Desen.DijitalDesen = new.DijitalDesen and Desen.Fason = 0 and Desen.orjBitmisUrun=0 into :MaxDesenId;

            if (MaxDesenId is null) then FDesenSayisi = 1;
            else
            begin
              STR = EgemenCopyStr(MaxDesenId, 1, 1);
              IlkKarakter = ascii_val(STR);

              if (IlkKarakter < 65) then
              begin
                BaslangicKarakter = '';
                STR = EgemenCopyStr(MaxDesenId, DesenSablonSayisiDigit + 1, DesenNoDigit);
              end
              else
              begin
                BaslangicKarakter = STR;
                STR = EgemenCopyStr(MaxDesenId, DesenSablonSayisiDigit + 4, DesenNoDigit);
              end
              TF = EgemenStrToIntOlurmu(STR);

              if (TF = 0) then exception ManuelGirilenDesenNoHatali;
              else FDesenSayisi = EgemenStrToInt(STR);

              if ((FDesenSayisi = 999 and DesenNoDigit = 3) or (FDesenSayisi = 9999 and DesenNoDigit = 4)) then
              begin
                FDesenSayisi = 0;

                if (BaslangicKarakter = '') then
                  BaslangicKarakter = 'A';
                else
                  BaslangicKarakter = Ascii_Char(IlkKarakter + 1);
              end
              else FDesenSayisi = FDesenSayisi + 1;
            end

            if (FEgemenFirmaid = 133) then --> BTEKS ÇORLU
              new.DesenId = 'BT-' || EgemenIntToLzStr(New.SablonSayisi, DesenSablonSayisiDigit) || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit);
        end
        else if (FEgemenFirmaid = 138) then   -------------- ARTESA 138------------------
        begin
          Select Max(DesenId) from Desen where Desen.SablonSayisi = new.SablonSayisi and Desen.DijitalDesen = new.DijitalDesen and Desen.Fason = 0 into :MaxDesenId;


          if (MaxDesenId is null) then FDesenSayisi = 1;
          else
          begin

            STR = EgemenCopyStr(MaxDesenId, 1, 1);

            IlkKarakter = ascii_val(STR);

            if (IlkKarakter < 65) then
            begin
              BaslangicKarakter = '';
              STR = EgemenCopyStr(MaxDesenId, DesenSablonSayisiDigit + 2, DesenNoDigit);
            end
            else
            begin
              BaslangicKarakter = STR;
              STR = EgemenCopyStr(MaxDesenId, DesenSablonSayisiDigit + 3, DesenNoDigit);
            end

            TF = EgemenStrToIntOlurmu(STR);

            if (TF = 0) then exception ManuelGirilenDesenNoHatali;
            else FDesenSayisi = EgemenStrToInt(STR);

            FDesenSayisi = FDesenSayisi + 1;
          end
          new.DesenId = BaslangicKarakter || EgemenIntToLzStr(New.SablonSayisi, DesenSablonSayisiDigit) || '-' || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit);
        end
        else
        begin
          Select Max(DesenId) from Desen where Desen.SablonSayisi = new.SablonSayisi and Desen.DijitalDesen = new.DijitalDesen and Desen.Fason = 0 into :MaxDesenId;

          if (MaxDesenId is null) then FDesenSayisi = 1;
          else
          begin


            STR = EgemenCopyStr(MaxDesenId, 1, 1);

            IlkKarakter = ascii_val(STR);

            if (IlkKarakter < 65) then
            begin
              BaslangicKarakter = '';
              STR = EgemenCopyStr(MaxDesenId, DesenSablonSayisiDigit + 1, DesenNoDigit);
            end
            else
            begin
              BaslangicKarakter = STR;
              STR = EgemenCopyStr(MaxDesenId, DesenSablonSayisiDigit + 2, DesenNoDigit);

            end

            TF = EgemenStrToIntOlurmu(STR);

            if (TF = 0) then exception ManuelGirilenDesenNoHatali;
            else FDesenSayisi = EgemenStrToInt(STR);

            if ((FDesenSayisi = 999 and DesenNoDigit = 3) or (FDesenSayisi = 9999 and DesenNoDigit = 4)) then
            begin
              FDesenSayisi = 1;

              if (BaslangicKarakter = '') then
                BaslangicKarakter = 'A';
              else
                BaslangicKarakter = Ascii_Char(IlkKarakter + 1);
            end
            else FDesenSayisi = FDesenSayisi + 1;
          end
          new.DesenId = BaslangicKarakter || EgemenIntToLzStr(New.SablonSayisi, DesenSablonSayisiDigit) || EgemenIntToLzStr(FDesenSayisi, DesenNoDigit);
        end
      end
    end
    else
    begin
      if (((DesenNoOtomatik <> 0) or (DesenKoduFormatiSerbest <> 1)) and New.DijitalDesen <> 1) then
      begin
        if (New.Fason=0) then
        begin
          STR = EgemenCopyStr(New.DesenId, 1, DesenSablonSayisiDigit);
          TF = EgemenStrToIntOlurmu(STR);

          if (TF = 0) then exception ManuelGirilenDesenNoHatali;
          else FSablonSayisi = EgemenStrToInt(STR);

          if (FSablonSayisi <> New.SablonSayisi) then exception ManuelGirilenDesenNoHatali;
        end
        else
        begin
          STR = EgemenCopyStr(New.DesenId, 1, 1);
          TF = StrKarsilastir(Str,'F');

          if (TF = 0) then exception ManuelGirilenDesenNoHatali;
        end
      end
    end

    if (DesenUstBilgiTasarimNo = 0 and new.DesenID is Null) then Exception KayitHatasi ' Desen Kodu Boş Olamaz!';

    if (new.DesenKayitId is null) then new.DesenKayitId = Gen_Id(DesenKayit_Id_Gen, 1);

    if (new.TakimParcasimi = 1 and New.DesenTakimID is null) then
    begin
      if (New.TakimParcasimi = 1) then FAdi = New.Adi;
      else FAdi = 'Ana Desen Bilinmiyor';

      New.DesenTakimId = Gen_ID(DesenTakim_Id_gen, 1);
      insert into DesenTakim(DesenTakimID, Adi, InsertKullaniciId) values(New.DesenTakimId, New.Adi, New.InsertKullaniciId);
    end
    else
    if (new.TakimParcasimi = 1 and New.DesenTakimID is not null and New.AnaDesen = 1) then
    begin
      Select Count(*) from Desen where Desen.DesenTakimID = New.DesenTakimID and Desen.AnaDesen = 1 into :KS;
      if (KS > 0) then exception TakiminBirTaneAnaDeseniOlabilir;

      Update DesenTakim set Adi = new.Adi where DesenTakim.DesenTakimID = New.DesenTakimID;
    end

    if (new.TasarimOnayDurumu = 1) then new.TasarimOnayTarihi = current_TimeStamp(0);
  end
end
