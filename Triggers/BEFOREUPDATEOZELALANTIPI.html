CREATE TRIGGER BEFOREUPDATEOZELALANTIPI FOR OZELALANTIPI
ACTIVE BEFORE UPDATE POSITION 0
as
Declare Variable KS Integer;
Declare Variable oaktListe Integer;
begin
  if (Old.OzelAlanKaynagiId <> New.OzelAlanKaynagiId) then
  begin
    Select Count(*) From OzelAlan Where OzelAlan.OzelAlanTipiId = Old.OzelAlanTipiId into :KS;
    if (KS > 0) then exception KayitHatasi 'Bu Tipe Ait Özel Alan Kaydı Var, Tip Bilgisi Değişemez..';
  end

  if (Old.OzelAlanKolonTipiId <> New.OzelAlanKolonTipiId) then
  begin
    Select OaktListe From OzelAlanKolonTipiGetir into :OaktListe;

    if (Old.OzelAlanKolonTipiId = OaktListe) then
    begin
        Select Count(*) From OzelAlanTipiDetay Where OzelAlanTipiDetay.OzelAlanTipiId = Old.OzelAlanTipiId into :KS;
        if (KS > 0) then exception KayitHatasi 'Bu Tipe Ait Özel Alan Tipi Datay Kaydı Var, Tip Bilgisi Değişemez..';
    end
  end
end
