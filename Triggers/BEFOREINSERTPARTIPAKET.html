CREATE TRIGGER BEFOREINSERTPARTIPAKET FOR PARTIPAKET
ACTIVE BEFORE INSERT POSITION 0
as
declare variable SinifBirimFiyat Double Precision;
declare variable PartiPakettenFiyatlandirildi integer;
begin
  select PartiStok.PartiPakettenFiyatlandirildi from PartiStok where PartiStok.PartiSinifId = New.PartiSinifId
    into :PartiPakettenFiyatlandirildi;

  if (PartiPakettenFiyatlandirildi = 1) then exception FiyatlandirilmisDegisemez;

  Select BirimFiyat from PartiStok where PartiStok.PartiSinifID = new.PartiSinifID into :SinifBirimFiyat;
  if (SinifBirimFiyat <> 0) then exception KayitHatasi 'Parti Sınıfının paket birim fiyat var, Aynı anda iki fiyatlandırma yapılamaz..';

  if (new.BirimFiyat is null) then New.BirimFiyat = 0;
  if (new.PartiPaketID is null) then new.PartiPaketID = gen_ID(PartiPaket_Id_Gen, 1);
end
