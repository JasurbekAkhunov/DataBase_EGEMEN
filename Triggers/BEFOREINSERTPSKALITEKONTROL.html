Definition

CREATE TRIGGER BEFOREINSERTPSKALITEKONTROL FOR PSKALITEKONTROL
ACTIVE BEFORE INSERT POSITION 0
as
declare variable KKBitirenPersonelId integer;
declare variable FGramaj         Double Precision;
declare variable IntGramaj       integer;
declare variable IntEn           integer;
declare variable HacimEn         integer;
declare variable Gramaj          varchar(10);
declare variable TF              integer;
declare variable FTup            integer;
declare variable KaliteKontrolMetreHesapSekli integer;
declare variable MetreHesabiIcinCarpan Numeric(12,3);
declare variable HesaplamaCarpimDegeri Integer;
declare variable FHesap  Double Precision;
declare variable FEnVerilen     varchar(10);
declare variable FGramajVerilen varchar(10);
declare variable KKBaslatanPersonelId integer;
declare variable MaxTopNo integer;
declare variable TopSayisi integer;
declare variable KaliteKontrolDefaultOnayDurumu integer;
declare variable BolunduguPartiSinifId integer;
declare variable KKDaraPaketlemedenGelsin EgeBoolean;
declare variable TupRolikAgirligi Double Precision;
declare variable PSKKBolunenTopNoDevam EgeBoolean;
declare variable EgemenFirmaId Integer;
declare variable CezaMetre BoyaMiktarSize;
declare variable OzelMusteri EgeBoolean;
declare variable OlcuGramaj Numeric(12,3);
declare variable FEnIstenen     varchar(10);
declare variable FGramajIstenen varchar(10);
declare variable FEn             Double Precision;
begin
  Select KaliteKontrolMetreHesapSekli, KaliteKontrolDefaultOnayDurumu, KKDaraPaketlemedenGelsin, PSKKBolunenTopNoDevam, EgemenFirmaId from sistem where SistemID = 1 into :KaliteKontrolMetreHesapSekli, :KaliteKontrolDefaultOnayDurumu, :KKDaraPaketlemedenGelsin, :PSKKBolunenTopNoDevam, :EgemenFirmaId;

  Select KKBaslatanPersonelId, KKBitirenPersonelId from PsKaliteKontrolAsama where PsKaliteKontrolAsamaId = new.PsKaliteKontrolAsamaId into :KKBaslatanPersonelId, :KKBitirenPersonelId;
  if (KKBaslatanPersonelId is null) then exception PSKaliteKontrolBaslatilmamis;

  Select PaketlemeSekli.Tup, Firma.MetreHesabiIcinCarpan, PartiStok.BolunduguPartiSinifId, BoyaSiparisDetay.HesaplamaCarpimDegeri, Firma.OzelMusteri from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
    left join Firma on Firma.FirmaID = Parti.FirmaID
    where PartiSinifId = new.PartiSinifId
    into :FTup, :MetreHesabiIcinCarpan, :BolunduguPartiSinifId, :HesaplamaCarpimDegeri, :OzelMusteri;

  if (KKDaraPaketlemedenGelsin = 1) then
  begin
    if (new.dara=0) then
    begin
      Select Ambalaj.TupRolikAgirligi from PartiStok
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
        left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
        left join Ambalaj on Ambalaj.AmbalajID = PaketlemeSekli.AmbalajID
      where PartiSinifId = new.PartiSinifId
      into :TupRolikAgirligi;
      if (TupRolikAgirligi > 0) then New.Dara = TupRolikAgirligi;
    end
  end

  if ((coalesce(new.LinkIslemi,0) <> 1) and (coalesce(new.SartliKabul,0) <> 1)) then
  begin
    if (KKBitirenPersonelId is not null) then exception PSKaliteKontrolTamamlanmis;
  end

  if (KaliteKontrolDefaultOnayDurumu = 1) then New.OnayDurumu = 1;
  else New.OnayDurumu = 0;

  if (New.OnayDurumu = 1 and New.OnayKullaniciID is Null) then
  begin
    New.OnayKullaniciID = New.InsertKullaniciId;
    New.OnayTarihi = current_TimeStamp(0);
  end

  if (FTup is null) then FTup = 0;

  if (new.TopNo is null or new.TopNo = 0) then
  begin
    if (new.HamStokMasterId is not null) then
      Select Max(topNo), Count(*) from PSKaliteKontrol where PSKaliteKontrol.PsKaliteKontrolAsamaId = New.PsKaliteKontrolAsamaId and
                                                             PSKaliteKontrol.PartiSinifId = new.PartiSinifId and
                                                             PSKaliteKontrol.HamStokMasterId = new.HamStokMasterId
        into :MaxTopNo, :TopSayisi;
    else
      Select Max(topNo), Count(*) from PSKaliteKontrol where PSKaliteKontrol.PsKaliteKontrolAsamaId = New.PsKaliteKontrolAsamaId
        into :MaxTopNo, :TopSayisi;

    if (MaxTopNo is not null and TopSayisi >= 1) then New.TopNo = MaxTopNo + 1;
    else
    begin
      Select Max(TopNo) from PSKaliteKontrol where PSKaliteKontrol.PSKaliteKontrolAsamaID = (Select Distinct PSKaliteKontrolAsama.PSKaliteKontrolAsamaID from PSKaliteKontrol
        Left Join PSKaliteKontrolAsama on PSKaliteKontrolAsama.PSKaliteKontrolAsamaID = PSKaliteKontrol.PSKaliteKontrolAsamaID
        where PSKaliteKontrol.ReddenBolunenYeniPartiSinifID = new.PartiSinifID and PSKaliteKontrolAsama.SonKayit = 1)
      into :MaxTopNo;

      if (MaxTopNo is not Null) then
        New.TopNo = MaxTopNo + 1;
      else if (BolunduguPartiSinifId is not null) then
      begin
        Select Max(TopNo) from PSKaliteKontrol where PSKaliteKontrol.PSKaliteKontrolAsamaID = (Select Distinct PSKaliteKontrolAsama.PSKaliteKontrolAsamaID from PSKaliteKontrol
          Left Join PSKaliteKontrolAsama on PSKaliteKontrolAsama.PSKaliteKontrolAsamaID = PSKaliteKontrol.PSKaliteKontrolAsamaID
          where PSKaliteKontrol.PartiSinifId = :BolunduguPartiSinifId and PSKaliteKontrolAsama.SonKayit = 1)
        into :MaxTopNo;

        if (MaxTopNo is not Null and PSKKBolunenTopNoDevam = 1) then
          New.TopNo = MaxTopNo + 1;
        else
          New.TopNo = 1;
      end
      else
        New.TopNo = 1;
    end
  end

  if (new.PaketlemeId is null) then New.Tarih = current_TimeStamp(0);

  New.OrtalamaEn = Null;

  if (EgemenStrToIntOlurmu(New.BaslangicEn) = 1 and EgemenStrToIntOlurmu(New.OrtaEn) = 1 and EgemenStrToIntOlurmu(New.BitisEn) = 1) then
    New.OrtalamaEn = (EgemenStrToInt(New.BaslangicEn) + EgemenStrToInt(New.OrtaEn) + EgemenStrToInt(New.BitisEn)) / 3;

  if (KaliteKontrolMetreHesapSekli = 0 or KaliteKontrolMetreHesapSekli = 4) then
  begin
    if (new.Metre is null or new.Metre <= 0) then
    begin
      Select PartiStok.HacimEn, PartiStok.Gramaj, PaketlemeSekli.Tup from PartiStok
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
        left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
        where PartiStok.PartiSinifId = new.PartiSinifId
        into :HacimEn, :Gramaj, :FTup;

      if (FTup is null) then FTup = 0;

      TF = EgemenStrToIntOlurmu(Gramaj);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(Gramaj);

      if (FTup = 0) then
        FGramaj = (IntGramaj * HacimEn) / 100000.000;
      else
        FGramaj = (IntGramaj * HacimEn * 2) / 100000.000;

      if (FGramaj is null) then FGramaj = 0;

      if (FGramaj <> 0) then New.Metre = (New.Kilo - New.Dara) / FGramaj;
      else New.Metre = 0;
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 1) then /* Tekboy */
  begin
    if (HesaplamaCarpimDegeri is not null) then
      MetreHesabiIcinCarpan = HesaplamaCarpimDegeri;

    if (new.Metre is null or new.Metre = 0) then
    begin
      TF = EgemenStrToIntOlurmu(New.Gramaj);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(New.Gramaj);

      TF = EgemenStrToIntOlurmu(New.En);

      if (TF = 0) then IntEn = 0;
      else IntEn = EgemenStrToInt(New.En);

      FHesap = IntEn * IntGramaj;

      if (FHesap <> 0) then
        new.Metre = (1000 / FHesap )* MetreHesabiIcinCarpan * (New.Kilo - New.Dara);
      else
        new.Metre = 0;
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 2) then /* Kral */
  begin
    if (new.Metre is null or new.Metre = 0) then
    begin
      Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
        where PartiStokFizikKontrol.partiSinifID = new.partiSinifID and PartiStokFizikKontrol.SonKayit = 1
        into :FEnVerilen, :FGramajVerilen;

      TF = EgemenStrToIntOlurmu(FGramajVerilen);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(FGramajVerilen);

      TF = EgemenStrToIntOlurmu(FEnVerilen);

      if (TF = 0) then IntEn = 0;
      else IntEn = EgemenStrToInt(FEnVerilen);

      FHesap = IntEn * IntGramaj;

      if (FHesap <> 0) then
        new.Metre = (1000 / FHesap ) * 100 * New.Kilo;
      else
        new.Metre = 0;
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 3) then /* NY */
  begin
    Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
      where PartiStokFizikKontrol.partiSinifID = new.partiSinifID and PartiStokFizikKontrol.SonKayit = 1
      into :FEnVerilen, :FGramajVerilen;

    TF = EgemenStrToIntOlurmu(FGramajVerilen);

    if (TF = 0) then IntGramaj = 0;
    else IntGramaj = EgemenStrToInt(FGramajVerilen);

    TF = EgemenStrToIntOlurmu(FEnVerilen);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(FEnVerilen);

    FHesap = IntEn * IntGramaj;

    if (FHesap <> 0) then
      new.Metre = (1000 / FHesap ) * 100 * (New.Kilo - New.Dara);
    else
      new.Metre = 0;
  end
  else  if (KaliteKontrolMetreHesapSekli = 5) then /* Emin */
  begin
    if (new.Kilo is null or new.Kilo = 0 ) then
    begin
      TF = EgemenStrToIntOlurmu(New.Gramaj);

      if (TF = 0) then IntGramaj = 0;
      else IntGramaj = EgemenStrToInt(New.Gramaj);

      TF = EgemenStrToIntOlurmu(substring(New.En from 1 for 3));

      if (TF = 0) then IntEn = 0;
      else IntEn = EgemenStrToInt(substring(New.En from 1 for 3));

      FHesap = IntEn * IntGramaj;

      if (FHesap <> 0) then new.Kilo = ((FHesap /100000) *  (New.Metre)) + 0.6;
      else new.Kilo = 0;
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 6) then /* Pakİpek */
  begin
    Select PartiStok.HacimEn, PartiStok.Gramaj, PaketlemeSekli.Tup from PartiStok
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
      left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
      where PartiStok.PartiSinifId = new.PartiSinifId
    into :HacimEn, :Gramaj, :FTup;

    if (FTup is null) then FTup = 0;

    TF = EgemenStrToIntOlurmu(Gramaj);

    if (TF = 0) then IntGramaj = 0;
    else IntGramaj = EgemenStrToInt(Gramaj);

    TF = EgemenStrToIntOlurmu(HacimEn);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(HacimEn);

    FHesap = IntEn * IntGramaj;

    if (FHesap <> 0) then new.Kilo = (FHesap /100000) *  (New.Metre);
    else new.Kilo = 0;
  end
  else if (KaliteKontrolMetreHesapSekli = 7) then /* Gülle */
  begin
  Select Ambalaj.TupRolikAgirligi from PartiStok
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
        left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
        left join Ambalaj on Ambalaj.AmbalajID = PaketlemeSekli.AmbalajID
      where PartiSinifId = new.PartiSinifId
      into :TupRolikAgirligi;

    Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
      left join parti on parti.partikayitid = partistokfizikkontrol.partikayitid
      where PartiStokFizikKontrol.partiSinifID = new.partiSinifID and PartiStokFizikKontrol.SonKayit = 1
      into :FEnVerilen, :FGramajVerilen ;

    TF = EgemenStrToIntOlurmu(FGramajVerilen);

    if (TF = 0) then IntGramaj = 0;
    else IntGramaj = EgemenStrToInt(FGramajVerilen);

    TF = EgemenStrToIntOlurmu(FEnVerilen);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(FEnVerilen);

    if (TupRolikAgirligi > 0) then New.Dara = TupRolikAgirligi/1000;
    else New.Dara = Yuvarla((Cast(IntEn * 3.9 as Double Precision) / 1000), 100);

    New.brutmetre = (1000 / Cast(IntEn * IntGramaj as Double Precision) ) * 100 * (new.Kilo-new.Dara);

  end
  else if (KaliteKontrolMetreHesapSekli = 8) then --NS ÖRME
  begin
    if (OzelMusteri = 0) then
    begin
        if (new.Metre is null or new.Metre <= 0) then
        begin
          Select PartiStok.HacimEn, PartiStok.Gramaj, PaketlemeSekli.Tup from PartiStok
            left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
            left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
            where PartiStok.PartiSinifId = new.PartiSinifId
            into :HacimEn, :Gramaj, :FTup;

          if (FTup is null) then FTup = 0;

          TF = EgemenStrToIntOlurmu(Gramaj);

          if (TF = 0) then IntGramaj = 0;
          else IntGramaj = EgemenStrToInt(Gramaj);

          if (FTup = 0) then
            FGramaj = (IntGramaj * HacimEn) / 100000.000;
          else
            FGramaj = (IntGramaj * HacimEn * 2) / 100000.000;

          if (FGramaj is null) then FGramaj = 0;

          if (FGramaj <> 0) then New.Metre = (New.Kilo - New.Dara) / FGramaj;
          else New.Metre = 0;
        end
    end
    else
    begin
          TF = EgemenStrToIntOlurmu(New.Gramaj);

          if (TF = 0) then IntGramaj = 0;
          else IntGramaj = EgemenStrToInt(New.Gramaj);

          TF = EgemenStrToIntOlurmu(New.En);

          if (TF = 0) then IntEn = 0;
          else IntEn = EgemenStrToInt(New.En);

          FHesap = IntEn * IntGramaj;

          if (FHesap <> 0) then
            new.Metre = (1000 / FHesap ) * 100 * (New.Kilo - New.Dara);
          else
            new.Metre = 0;
    end
  end
  else if (KaliteKontrolMetreHesapSekli = 9) then /* Özsar */
  begin
    Select BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, PaketlemeSekli.Tup from PartiStok
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
      left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
      where PartiStok.PartiSinifId = new.PartiSinifId
      into :FEnIstenen, :FGramajIstenen, :FTup;

    TF = EgemenStrToIntOlurmu(FGramajIstenen);

    if (TF = 0) then IntGramaj = 0;
    else IntGramaj = EgemenStrToInt(FGramajIstenen);

    TF = EgemenStrToIntOlurmu(FEnIstenen);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(FEnIstenen);

    if (FTup = 0) then
      FEn  = IntEn * 0.01;
    else
      FEn  = 2 * IntEn * 0.01;

    FGramaj =  IntGramaj * 0.001 ;
    FHesap  = (FEn) * (FGramaj);

    if (FHesap <> 0) then
      new.Metre = (New.Kilo - New.Dara) /  FHesap;
    else
      new.Metre = 0;
  end
  else if (KaliteKontrolMetreHesapSekli = 10) then  /* Feyteks*/
  begin
    TF = EgemenStrToIntOlurmu(New.En);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(New.En);

    if ((New.Metre <> 0) and (IntEn <> 0) and ((New.Kilo - New.Dara)<>0)) then
      New.Gramaj = egemenintegertostr( ((New.Kilo - New.Dara)*1000) / (cast(IntEn as Double Precision)/100 * New.Metre) );

  end



  if (OzelMusteri = 1 and KaliteKontrolMetreHesapSekli <> 8) then /* Hugo Boss Metre düşümü*/
  begin
    if (New.Metre > 0 or Coalesce(new.BrutMetre, 0) > 0 ) then
    begin
      if (New.Metre > 0 and
           (Coalesce(new.BrutMetre, 0) = 0 or
            New.Metre > Coalesce(new.BrutMetre, 0))
         ) then New.BrutMetre = New.Metre;

      if (New.BrutMetre > 0) then
      begin
        Select CezaMetre from  PSKKontrolCezaMetreHesapla(New.PartiSinifId, New.PSKaliteKontrolId) into :CezaMetre;
        New.Metre = New.BrutMetre - :CezaMetre;
      end
    end
  end

  if (EgemenFirmaId = 138) then /*Artesa*/
  begin

    TF = EgemenStrToIntOlurmu(New.En);

    if (TF = 0) then IntEn = 0;
    else IntEn = EgemenStrToInt(New.En);

    if (New.Metre > 0 and IntEn > 0) then
    New.Gramaj = egemenintegertostr( ((New.Kilo - New.Dara)*1000) / (cast(IntEn as Double Precision)/100 * New.Metre) );


    if (new.OlcuId is not null) then
    begin
      select Max(BoyaSiparisOlcu.Gramaj) from BoyaSiparisOlcu
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisOlcu.BoyaSiparisDetayId
      left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayID
      where PartiStok.partisinifid = new.partisinifid and BoyaSiparisOlcu.OlcuId = new.OlcuId
      into :OlcuGramaj ;

      if (OlcuGramaj > 0) then
      new.adet = ((New.Kilo - New.Dara)*1000) / OlcuGramaj ;
    end

  end

  if (EgemenFirmaId = 70) then /*CeylanTeks*/
  begin
    new.Metre = Yuvarla (new.Metre, 10) ;
  end

  new.LinkIslemi = 0;

  if (new.PSKaliteKontrolId is null) then  new.PSKaliteKontrolId = Gen_Id(PSKaliteKontrol_Id_Gen,1);
  if (New.TestaTopID > 32760) then Execute Statement('Alter Sequence TestaTop_Id_Gen Restart with 0');
end
