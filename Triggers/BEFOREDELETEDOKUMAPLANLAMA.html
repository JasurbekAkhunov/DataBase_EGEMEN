Definition

CREATE TRIGGER BEFOREDELETEDOKUMAPLANLAMA FOR DOKUMAPLANLAMA
ACTIVE BEFORE DELETE POSITION 0
as
declare variable KayitSayisi integer;
begin
  Select count(*) from StokHareket where StokHareket.DokumaPlanlamaId = Old.DokumaPlanlamaId into :kayitSayisi;
  if (KayitSayisi > 0 or Old.GerceklesenMiktar > 0 or Old.Bitti = 1) then exception TalimatHareketGormusSilinemez;

  Update OrmeSiparisDetay  Set PlanlananMiktar = PlanlananMiktar - Old.PlanlananMiktar, LinkIslemi = 1  where OrmeSiparisDetay.OrmeSiparisDetayId = Old.OrmeSiparisDetayId;

  Delete from DokumaPlanlamaRecete where DokumaPlanlamaRecete.DokumaPlanlamaId = Old.DokumaPlanlamaId;
end
