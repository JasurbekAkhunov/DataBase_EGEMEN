CREATE TRIGGER BEFOREINSERTIPLIKSTOK FOR IPLIKSTOK
ACTIVE BEFORE INSERT POSITION 0
as
declare variable HareketTipi integer;
declare variable shtUretim integer;
declare variable shtAlis integer;
declare variable shtFasonAlis integer;
declare variable shtTelefGirisi integer;
declare variable shtSatisIade integer;
declare variable shtDepoTransfer integer;
begin
  select shtAlis, shtFasonAlis, ShtUretim, shtTelefGirisi, shtSatisIade, shtDepoTransfer from IplikStokHareketTipiGetir into :shtAlis, :shtFasonAlis, :ShtUretim, :shtTelefGirisi, :shtSatisIade, :shtDepoTransfer;

  Select HamId, IplikDepoId, IplikIsletmeId, HareketTipi from IplikIrsaliyeDetay
    where IplikIrsaliyeDetay.IplikIrsaliyeDetayID = New.IplikIrsaliyeDetayId
    into new.HamId, new.IplikDepoId, new.IplikIsletmeId, :HareketTipi;

  if (HareketTipi <> ShtUretim and HareketTipi <> ShtAlis and HareketTipi <> ShtFasonAlis and HareketTipi <> shtTelefGirisi  and HareketTipi <> shtSatisIade and HareketTipi <> shtDepoTransfer) then New.NetMiktar = New.Miktar;

  if (HareketTipi <> shtUretim) then
    Select FiiliSevkTarihi from IplikIrsaliye where IplikIrsaliye.IplikIrsaliyeID = New.IplikIrsaliyeID into New.GirisTarihi;
  else
    New.GirisTarihi = current_TimeStamp(0);

  new.Numarator = Null;

  Select Max(IplikStok.Numarator) from IplikStok where IplikStokId = New.IplikStokId into new.Numarator;

  if (new.Numarator is null) then new.Numarator = 1;
  else new.Numarator = new.Numarator + 1;

  if (New.IplikStokId is null) then New.IplikStokId = Gen_Id(IplikStok_Id_Gen, 1);

  if (New.BolunduguIplikStokID is Not Null) then
    Update IplikStok Set Miktar = Miktar - New.Miktar, NetMiktar = NetMiktar - New.NetMiktar where IplikStok.IplikStokID = New.BolunduguIplikStokID;
end
