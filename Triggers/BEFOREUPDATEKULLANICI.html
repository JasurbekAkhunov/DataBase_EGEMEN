CREATE TRIGGER BEFOREUPDATEKULLANICI FOR KULLANICI
ACTIVE BEFORE UPDATE POSITION 0
as
begin
  if (new.KullaniciId = 1 and new.FirmaId is not null) then exception KullaniciKayitHatasi;

  if (new.FirmaId is not null) then
  begin
    if (new.sifre is null or new.Sifre = '') then exception SifreBosBirakilamaz;
    new.SirketIciListedeGorunsun   = 0;
    new.SirketDisiKullanici        = 1;
    new.SirketDisiListedeGorunsun  = 0;
    new.RaporDizayniniKaydedebilir = 0;
    new.ArizaIhbarNotuGirebilir    = 0;
    new.ArizaServisNotuGirebilir   = 0;
    new.HamKontrolKontrol          = 0;
    new.HamKontrolOnay             = 0;
    new.SonKontrolOnay             = 0;
    new.MakinaTazele               = 0;
    new.PersonelId                 = null;
    new.SatirSilOnay               = 1;
    new.PartiIkazErteledeGorunsun  = 0;

    new.PlanlamaYapabilir          = 0;
    new.RaporDizayniSilebilir      = 0;
  end

  if (Old.Aktif = 1 and new.Aktif = 0) then
    Update KullaniciModul set Girebilir = 0, YeniKayit = 0, Duzeltme = 0, Silme = 0 where KullaniciModul.KullaniciID = New.KullaniciId;
end
