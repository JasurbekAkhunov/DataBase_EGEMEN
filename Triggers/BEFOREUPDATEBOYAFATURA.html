Definition

CREATE TRIGGER BEFOREUPDATEBOYAFATURA FOR BOYAFATURA
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable AraTutar Double Precision;
declare variable B1 Double Precision;
declare variable B2 Double Precision;
declare variable B3 Double Precision;
Declare Variable T1 Double Precision;
Declare Variable T2 Double Precision;
Declare Variable T3 Double Precision;
declare variable FDate Date;
declare variable FPartiSinifId integer;
declare variable VT1 Date;
declare variable VD2 Double Precision;
declare variable VD3 Double Precision;
declare variable VadeGun integer;
declare variable KayitSayisi integer;
declare variable FKdv     Double Precision;
declare variable FMeblag  Double Precision;
declare variable StokMuhasebeYazilimTipi integer;
declare variable IrsaliyeTarihi TimeStamp;
declare variable IT3 integer;
declare variable ToplamKdv          Double Precision;
declare variable SonDonemBitisTarihi TimeStamp;
declare variable BFKurDegisirseDetayKurGunclle integer;
declare variable SistemDovizId  varchar(5);
declare variable DetayDovizId  varchar(5);
declare variable BoyaFaturaDetayID integer;
declare variable EgemenFirmaID integer;
declare variable IrsTarihFatTarihtenBuyukOlamaz integer;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable OdemeSekliId integer;
begin
  Select StokMuhasebeYazilimTipi, EgemenFirmaID, BFKurDegisirseDetayKurGunclle, SistemDovizId, IrsTarihFatTarihtenBuyukOlamaz from Sistem where Sistemid = 1 into :StokMuhasebeYazilimTipi, :EgemenFirmaID, :BFKurDegisirseDetayKurGunclle, :SistemDovizId, :IrsTarihFatTarihtenBuyukOlamaz;

  if (new.FaturaTarihi > current_date) then exception IleriTariheFaturaGirilemez;

  if (Old.HareketTipiId <> New.HareketTipiId) then exception BoyaIrsaliyeHareketTipiDegismez;

  if (IrsTarihFatTarihtenBuyukOlamaz = 1) then
  begin
    for Select BoyaIrsaliye.IrsaliyeTarihi from BoyaFaturaDetay
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = BoyaFaturaDetay.BoyaIrsaliyeDetayID
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      where BoyaFaturaDetay.BoyaFaturaID = New.BoyaFaturaID
    into :FIrsaliyeTarihi do
    begin
      if (FIrsaliyeTarihi > New.FaturaTarihi) then exception IrsTarFatTardanBuyukOlamaz;
    end
  end


  if (New.IslemDovizKuru1 = 0) then
  begin
    if (New.IslemDovizID <> New.DovizID) then
    begin
      if (EgemenFirmaID = 4) then /* gulle irsaliye tarihindeki kur gelsin istiyor */
      begin
        Select Min(BoyaIrsaliye.IrsaliyeTarihi) from BoyaFaturaDetay
          left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = BoyaFaturaDetay.BoyaIrsaliyeDetayID
          left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
          where BoyaFaturaDetay.BoyaFaturaID = New.BoyaFaturaID
        into :IrsaliyeTarihi;

        if (IrsaliyeTarihi is null) then IrsaliyeTarihi = new.FaturaTarihi;
      end
      else
        IrsaliyeTarihi = new.FaturaTarihi;
      Select DovizKuru1, DovizKuru2 from FirmaDovizKurlariniGetir(new.FirmaId, New.DovizId, New.IslemDovizId, :IrsaliyeTarihi) into New.IslemDovizKuru1, New.IslemDovizKuru2;
    end
    else
    begin
      New.IslemDovizKuru1 = 1;
      New.IslemDovizKuru2 = 1;
    end
  end

  if (new.LinkIslemi = 0) then
  if (Old.Entegrasyon = 1 and new.Entegrasyon = 1) then exception faturaHazirDegistiremezsiniz;

  if (new.LinkIslemi = 0) then
  if (Old.TransferEdildi = 1 and new.TransferEdildi = 1) then exception TrasferEdilmisDegistirilmez;

  if (Old.Entegrasyon = 1 and New.Entegrasyon = 0 and New.TransferEdildi <> 0) then
  begin
    if (StokMuhasebeYazilimTipi = 3 or StokMuhasebeYazilimTipi = 5 or StokMuhasebeYazilimTipi = 6) then exception FaturaTransferEdilmisDegisemez;
  end

  if (New.Entegrasyon = 0 and New.TransferEdildi <> 0) then
  begin
    new.TransferEdildi = 0;
    New.Status = 1;
  end

  if (Old.FirmaId <> new.FirmaId) then exception FirmaKodunuDegistiremezsiniz;
  /*if (old.IslemDovizId <> new.IslemDovizId) then exception FaturaDovizIdDegisemez;*/

  if (Old.DovizId <> new.DovizId) then
  begin
    select count(*) from BoyaFaturaDetay where BoyaFaturaDetay.BoyaFaturaId = New.BoyaFaturaId into :KayitSayisi;
    if (KayitSayisi > 0) then exception BFSatirGirilmisDovizDegisemez;
  end


  if (Old.IslemDovizKuru1 <> new.IslemDovizKuru1 or Old.IslemDovizKuru2 <> new.IslemDovizKuru2) then
  begin
    if (BFKurDegisirseDetayKurGunclle = 1) then
    begin
      for Select PartiStok.DovizID, BoyaFaturaDetay.BoyaFaturaDetayID from BoyaFaturaDetay
      Left Join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
      left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      where BoyaFaturaDetay.BoyaFaturaID = new.BoyaFaturaID into :DetayDovizId, :BoyaFaturaDetayID do
      begin
        if (DetayDovizId = new.IslemDovizID) then
          update BoyaFaturaDetay Set DovizKuru2 = New.IslemDovizKuru1, DovizKuru1 = New.IslemDovizKuru2 where BoyaFaturaDetay.BoyaFaturaDetayID = :BoyaFaturaDetayID;
      end
    end
  end

  if (new.IslemDovizId is Null) then
  begin
    New.IslemDovizId = New.DovizId;
    new.IslemDovizKuru1 = New.DovizKuru1;
    new.IslemDovizKuru2 = New.DovizKuru2;
  end

/*
  select sum(Yuvarla(BoyaFaturaDetay.Miktar * BoyaFaturaDetay.BirimFiyat, new.AltIskYuvarlamaCarpani)) from BoyaFaturaDetay where BoyaFaturaDetay.BoyaFaturaId = New.BoyaFaturaId  into new.BrutToplam;
*/

  new.BrutToplam = 0;
  new.LinkIslemi = 0;
  for select BoyaFaturaDetay.Miktar * BoyaFaturaDetay.BirimFiyat from BoyaFaturaDetay where BoyaFaturaDetay.BoyaFaturaId = New.BoyaFaturaId
    into :FMeblag do
  begin
    FMeblag = Yuvarla(FMeblag, new.NetFiyatYuvarlamaCarpani);
    new.BrutToplam  = new.BrutToplam + FMeblag;
  end

  new.BrutToplam  = Yuvarla(new.BrutToplam, new.KdvYuvarlamaCarpani);

  if (New.BrutToplam is null) then New.BrutToplam = 0;

  AraTutar = new.BrutToplam - new.ParasalIskonto;

  new.IskontoOranTutari = AraTutar * new.iskontoOrani / 100;
  new.IskontoOranTutari = Yuvarla(new.IskontoOranTutari, new.AltIskYuvarlamaCarpani);

  new.IskontoToplami = new.ParasalIskonto + new.IskontoOranTutari;

  if (new.BrutToplam <> 0) then new.AltIskontoCarpani = new.IskontoToplami / new.BrutToplam;
  else new.AltIskontoCarpani = 0;

  new.TahsilEdilecekKdv = 0;
  new.IndirilecekKdv = 0;

  if (New.IstisnaKodu is Null and New.Tevkifat = 1) then Exception TevkifatIstisnasiZorunlu;
    
  if (New.IstisnaKodu is not Null) then
    Select TevkifatOrani1, TevkifatOrani2 from IstisnaKodlari where IstisnaKodlari.IstisnaKodu = New.IstisnaKodu into New.TevkifatOrani1, New.TevkifatOrani2;
	
  for select Sum(Yuvarla(BoyaFaturaDetay.BirimFiyat * BoyaFaturaDetay.Miktar, new.NetFiyatYuvarlamaCarpani)), BoyaFaturaDetay.KdvOrani from BoyaFaturaDetay where BoyaFaturaDetay.BoyaFaturaId = New.BoyaFaturaId
  Group By BoyaFaturaDetay.KdvOrani
  into :FMeblag, :FKdv  do
  begin
    B1 = (FMeblag - FMeblag * new.AltIskontoCarpani);
    B1 = Yuvarla(B1, new.KdvYuvarlamaCarpani);


    if (New.Tevkifat = 0) then
    begin
      B2 = B1 * Fkdv / 100;
      B2 = Yuvarla(B2, new.KdvYuvarlamaCarpani);
      new.TahsilEdilecekKdv = new.TahsilEdilecekKdv + B2;
    end
    else if (New.Tevkifat = 1) then
    begin
      ToplamKdv = B1 * Fkdv / 100;
      ToplamKdv = Yuvarla(ToplamKdv, new.KdvYuvarlamaCarpani);

      B2 = ToplamKdv * new.TevkifatOrani1 / new.TevkifatOrani2;
      B2 = Yuvarla(B2, new.KdvYuvarlamaCarpani);

      new.IndirilecekKdv = new.IndirilecekKdv + B2;

      B2 = ToplamKdv - B2;
      new.TahsilEdilecekKdv = new.TahsilEdilecekKdv + B2;
    end
  end

  new.IndirilecekKdv = Yuvarla(new.IndirilecekKdv, new.KdvYuvarlamaCarpani);
  new.TahsilEdilecekKdv = Yuvarla(new.TahsilEdilecekKdv, new.KdvYuvarlamaCarpani);

  T1 = 0;
  T2 = 0;
  T3 = 0;
  if (Old.Entegrasyon = 0 and New.Entegrasyon = 1) then
  if (new.VadeTipi = 2) then
  begin
    Select Sum(BoyaFaturaDetay.Vadegun * BoyaFaturaDetay.Miktar * BoyaFaturaDetay.BirimFiyat) from BoyaFaturaDetay
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
      where BoyaFaturaDetay.BoyaFaturaId = new.BoyaFaturaId and BoyaIrsaliyeDetay.PartiSinifId is not null
      into :T2;

    if (T2 is null) then T2 = 0;

    if (New.BrutToplam <> 0) then T3 = T2 / new.BrutToplam;
    else T3 = 0;

    T3 = Yuvarla(T3, 1);

    FDate = new.faturaTarihi + T3;
    new.Odemetarihi = FDate;
  end
  else if (new.VadeTipi = 1) then
  begin
    For Select BoyaIrsaliyeDetay.IrsaliyeTarihi, BoyaFaturaDetay.Vadegun, BoyaFaturaDetay.Miktar * BoyaFaturaDetay.BirimFiyat from BoyaFaturaDetay
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
      where BoyaFaturaDetay.BoyaFaturaId = new.BoyaFaturaId and BoyaIrsaliyeDetay.PartiSinifId is not null
      into :VT1, :VadeGun, :T3 do
    begin
      T1 = T1 + T3;
      if (VadeGun is null) then VadeGun = 0;
      VadeGun = VadeGun - (TimeStampToDouble(new.FaturaTarihi) -  TimeStampToDouble(VT1));
      T2 = T2 + T3 * VadeGun;
    end
    if (T1 <> 0) then
    begin
      IT3 = T2 / T1;
      new.Odemetarihi = new.FaturaTarihi + IT3;
    end else New.OdemeTarihi = New.FaturaTarihi;
  end

  /*Sadırlar FaturaDetaydaki İlk Kaydın Ödeme Şekli Bilgisi Almak İstiyor*/
  if (Old.Entegrasyon = 0 and New.Entegrasyon = 1 and EgemenFirmaID = 120) then
  begin
    if (new.OdemeSekliId is null)  then
    begin
      select First 1 BoyaSiparis.OdemeSekliId from BoyaFaturaDetay
      left Join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      left Join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      where    BoyaFaturaDetay.BoyaFaturaId = new.BoyaFaturaId
      order by BoyaFaturaDetay.BoyaFaturaDetayId asc
      into :OdemeSekliId;

      new.OdemeSekliId = OdemeSekliId;
    end
  end


  new.IslemDovizToplam = (New.BrutToplam - New.IskontoToplami) * New.IslemDovizKuru2 / New.IslemDovizKuru1;
  new.IslemDovizToplam = Yuvarla(New.IslemDovizToplam, new.KdvYuvarlamaCarpani);

  new.IslemDovizTahsilEdilecekKdv = New.TahsilEdilecekKdv * New.IslemDovizKuru2 / New.IslemDovizKuru1;
  new.IslemDovizTahsilEdilecekKdv = Yuvarla(New.IslemDovizTahsilEdilecekKdv, new.KdvYuvarlamaCarpani);

  if (Old.TransferEdildi <> 0 and new.TransferEdildi = 0) then
  begin
     New.TransferTarihi = Null;
     New.TransferKullaniciId = Null;
  end
  else
  if (Old.TransferEdildi = 0 and new.TransferEdildi <> 0) then
  begin
    New.TransferTarihi = current_TimeStamp(0);
    New.Status = 0;
  end

  select max(BitisTarihi) from DonemKapatma where Bolum = 1 and OnayKullaniciId is not Null into :SonDonemBitisTarihi;
  if (SonDonemBitisTarihi is not Null and (SonDonemBitisTarihi > new.FaturaTarihi or SonDonemBitisTarihi > old.FaturaTarihi)) then
    exception DonemKapaliHDegisemez;

  new.DegisimTarihi = current_TimeStamp(0);
  Execute Procedure Logla('U', 1, 'BOYAFATURA', 'BOYAFATURAID', new.BoyaFaturaId, null);
end
