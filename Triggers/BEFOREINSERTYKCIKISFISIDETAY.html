CREATE TRIGGER BEFOREINSERTYKCIKISFISIDETAY FOR YKONTROLCIKISFISIDETAY
ACTIVE BEFORE INSERT POSITION 0
as
declare variable TeslimEdildi integer;
begin
  Select TeslimEdildiDurumu from YKontrolCikisFisi where YKontrolCikisFisi.YKontrolCikisFisiId = new.YKontrolCikisFisiId into :TeslimEdildi;

  if (TeslimEdildi = 1) then exception FisTeslimEdilmisDegisemez;

  new.StokVar = 1;
  new.YKontrolCikisFisiDetayId = gen_id(YKontrolCikisFisiDetay_Id_Gen, 1);
end
