CREATE TRIGGER BEFOREINSERTURUNFIYAT FOR URUNFIYAT
ACTIVE BEFORE INSERT POSITION 0
as
begin
  if (new.AlisFiyatiUrunBirimId is null) then
  begin
    Select UrunBirim.UrunBirimId from UrunBirim where UrunBirim.UrunID = new.UrunID and UrunBirim.DefaultBirim = 1 into new.AlisFiyatiUrunBirimId;

    New.SatisFiyatiUrunBirimId = new.AlisFiyatiUrunBirimId;
    New.UretimeCikisFiyatiUrunBirimId = new.AlisFiyatiUrunBirimId;
  end
end
