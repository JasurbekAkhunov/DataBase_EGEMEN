CREATE TRIGGER AFTERUPDATEFIRMA FOR FIRMA
ACTIVE AFTER UPDATE POSITION 0
as
begin
  Update FirmaSevkAdresi Set SevkAdresi1 = New.Adresi1, SevkAdresi2 = new.Adresi2,
                             SevkAdresi3 = new.Adresi3, SevkSemt = New.Semt, SevkSehir = new.Sehir,
                             SevkUlke = New.Ulke where FirmaSevkAdresi.FirmaId = new.FirmaId and FirmaSevkAdresi.FirmaAdresi = 1;
end
