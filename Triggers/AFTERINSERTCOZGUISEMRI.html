CREATE TRIGGER AFTERINSERTCOZGUISEMRI FOR COZGUISEMRI
ACTIVE AFTER INSERT POSITION 0
as
declare variable FEgemenFirmaId integer;
begin
  Select EgemenFirmaId  from sistem into :FEgemenFirmaId;
  insert into CozguIsEmriRecete(CozguIsEmriKayitID, HamId, RenkID, TelSayisi, SiraNo, KullanimOrani, FireOrani, BirimFiyat, BirimFiyatDovizID, InsertKullaniciId, KullaniciId, MaliyetFireOrani, CozguFireOrani)
  Select new.CozguIsEmriKayitID, HamStokRecete.ReceteHamId, HamStokRecete.RenkID, HamStokRecete.TelSayisi, HamStokRecete.SiraNo, HamStokRecete.KullanimOrani,
    iif(:FEgemenFirmaId=151,HamStokRecete.FireOrani,HamStokRecete.CozguFireOrani), HamStokRecete.BirimFiyat, HamStokRecete.BirimFiyatDovizID, New.InsertKullaniciId, New.KullaniciId,
    HamStokRecete.MaliyetFireOrani, HamStokRecete.CozguFireOrani
    from HamStokRecete where HamStokRecete.HamID = New.IplikHamId and HamStokRecete.AtkiCozgu = 2;
end
