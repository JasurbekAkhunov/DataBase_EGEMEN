CREATE TRIGGER BEFOREINSERTHAM FOR HAM
ACTIVE BEFORE INSERT POSITION 0
as
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstLycra        integer;
declare variable hstTelef        integer;
declare variable hstKirpinti     integer;
declare variable hstHamPamuk     integer;
declare variable hstCigit        integer;
declare variable ElyafTanitimdaSadeceAdZorunlu integer;
declare variable TelefTanitimdaSadeceAdZorunlu integer;
declare variable IplikTanitimdaSadeceAdZorunlu integer;
declare variable DokumaKumasdaHarmanCinsZorunlu integer;
declare variable DokumaTanitimKumasCinsiZorunlu integer;
declare variable OrguTanitimKumasCinsiZorunlu integer;
declare variable StokTabanDovizID DovizSize;
begin
  Select ElyafTanitimdaSadeceAdZorunlu, TelefTanitimdaSadeceAdZorunlu, IplikTanitimdaSadeceAdZorunlu, DokumaKumasdaHarmanCinsZorunlu, DokumaTanitimKumasCinsiZorunlu, OrguTanitimKumasCinsiZorunlu, StokTabanDovizID from Sistem Where SistemID = 1
    into :ElyafTanitimdaSadeceAdZorunlu, :TelefTanitimdaSadeceAdZorunlu, :IplikTanitimdaSadeceAdZorunlu, :DokumaKumasdaHarmanCinsZorunlu, :DokumaTanitimKumasCinsiZorunlu, :OrguTanitimKumasCinsiZorunlu, :StokTabanDovizID;

  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstLycra, hstTelef, hstKirpinti, hstHamPamuk, hstCigit from HamTipiGetir
    into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstLycra, :hstTelef, :hstKirpinti, :hstHamPamuk, :hstCigit;

  if (new.DokumaBirimMaliyetDovizId is Null) then new.DokumaBirimMaliyetDovizId = StokTabanDovizID;
  if (new.CozguBirimMaliyetDovizId is Null) then  new.CozguBirimMaliyetDovizId  = StokTabanDovizID;
  if (new.HasilBirimMaliyetDovizId is Null) then  new.HasilBirimMaliyetDovizId  = StokTabanDovizID;

  if (new.HamTipiId = hstIplik) then
  begin
      if (IplikTanitimdaSadeceAdZorunlu = 0 and (new.KalinlikId is null or new.IplikCinsiId is null)) then exception KalinlikIplikCinsiGirilmeli;
  end
  else if (new.HamTipiId = hstLycra) then
  begin
    if (new.KalinlikId is null or new.IplikCinsiId is null) then exception KalinlikIplikCinsiGirilmeli;
  end
  else if (new.HamTipiId = hstElyaf) then
  begin
    if (new.IplikCinsiId is null and ElyafTanitimdaSadeceAdZorunlu = 0) then exception IplikCinsiBosBirakilamaz;
  end
  else if (new.HamTipiId = hstTelef) then
  begin
    if (new.IplikCinsiId is null and TelefTanitimdaSadeceAdZorunlu = 0) then exception IplikCinsiBosBirakilamaz;
  end
  else if (new.HamTipiId = hstKirpinti) then
  begin
    /* (new.IplikCinsiId is null and TelefTanitimdaSadeceAdZorunlu = 0) then exception IplikCinsiBosBirakilamaz;*/
  end
  else if (new.HamTipiId = hstCigit or new.HamTipiId = hstHamPamuk) then
  begin
    /* (new.IplikCinsiId is null and TelefTanitimdaSadeceAdZorunlu = 0) then exception IplikCinsiBosBirakilamaz;*/
  end
  else if (new.HamTipiId = hstDokumaKumas) then
  begin
    if (new.HarmanCinsiId is null and DokumaKumasdaHarmanCinsZorunlu = 1) then exception HarmanCinsiBosBirakilamaz;
    if (new.KumasCinsiId is null and DokumaTanitimKumasCinsiZorunlu = 1) then exception KumasCinsiBosBirakilamaz;
  end
  else
  begin
    if (new.KumasCinsiId is null and OrguTanitimKumasCinsiZorunlu = 1) then exception KumasCinsiBosBirakilamaz;
  end

  if (new.MaximumFireOrani = 0) then select sistem.MaximumFireOrani from sistem where sistemid = 1 into new.MaximumFireOrani;
  if (new.MinimumFireOrani = 0) then select sistem.MinimumFireOrani from sistem where sistemid = 1 into new.MinimumFireOrani;

  if (New.AlisFiyatiDovizId is null) then
    Select SistemDovizId from Sistem where sistemid = 1 into New.AlisFiyatiDovizId;

  if (new.HamId is null) then new.HamId = gen_id(Ham_Id_Gen, 1);
end
