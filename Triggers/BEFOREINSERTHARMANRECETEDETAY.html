CREATE TRIGGER BEFOREINSERTHARMANRECETEDETAY FOR HARMANRECETEDETAY
ACTIVE BEFORE INSERT POSITION 0
as
declare variable KS integer;
declare variable HarmanReceteMaxBalyaSayisi integer;
begin
  Select HarmanReceteMaxBalyaSayisi from Sistem where SistemId = 1 into :HarmanReceteMaxBalyaSayisi;

  Select Count(*) from HarmanReceteDetay where HarmanReceteDetay.HarmanReceteID = New.HarmanReceteID into :KS;
  if (KS >= HarmanReceteMaxBalyaSayisi) then  exception KayitHatasi 'Max Balya adedine ulaşılmış, yeni balya girilemez!';

  new.HarmanReceteDetayID = gen_id(HarmanReceteDetay_Id_Gen, 1);
end
