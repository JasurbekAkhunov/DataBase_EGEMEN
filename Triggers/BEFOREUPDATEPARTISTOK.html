Definition

CREATE TRIGGER BEFOREUPDATEPARTISTOK FOR PARTISTOK
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KayitSayisi                    integer;
declare variable PartiFason                     integer;
declare variable FiyatOnayiKarariSevktenSonra   integer;
declare variable KKEnGramajiFizikKontroleKaydet integer;
declare variable ProxyPartiSevkOluncaBosaCiksin integer;
declare variable DokumaKumasiKilodanTakipEt     integer;
declare variable Kapali                         integer;
declare variable SonUniteId                     integer;
declare variable MaxUniteIrsaliyeId             integer;
declare variable EksiFireKontrolEdilecek        integer;
declare variable IptalFiresiniYeniPartiyeYansit integer;
declare variable PartiNetBruttenFazlaOlabilir   integer;
declare variable IptalPartiFireOrani            Double Precision;
declare variable IptaldenAcilanMiktar           Double Precision;
declare variable IptaldenGelenFark              Double Precision;
declare variable PartiFireOnayiIstensin         integer;
declare variable KKBitinceTartiyaAktar          integer;
declare variable KontrolEdilecekMaxFire         integer;
declare variable FireOnayiUcuncuFireden         integer;
declare variable IrsFarkliDovizBirlestiremez    integer;
declare variable SevkEdilmisPartiFiyatGuncelle  integer;
declare variable ProxyPartiSevkHazirdaBosalsin  integer;
declare variable KKBitirdeFireKontrolu          integer;
Declare Variable FKaliteKontrolKilo             Double Precision;
Declare Variable IslemTipiId integer;
Declare Variable FBoyaSiparisDetayIdler         varchar(255);
declare variable Sorgu                          varchar(1000);
declare variable SarfEdilenTopSayisi            integer;
begin
  Select DokumaKumasiKilodanTakipEt, IptalFiresiniYeniPartiyeYansit, PartiFireOnayiIstensin, KKBitinceTartiyaAktar, PartiNetBruttenFazlaOlabilir, FireOnayiUcuncuFireden, IrsFarkliDovizBirlestiremez, SevkEdilmisPartiFiyatGuncelle, KKBitirdeFireKontrolu from Sistem
    into :DokumaKumasiKilodanTakipEt, :IptalFiresiniYeniPartiyeYansit, :PartiFireOnayiIstensin, :KKBitinceTartiyaAktar, :PartiNetBruttenFazlaOlabilir, :FireOnayiUcuncuFireden, :IrsFarkliDovizBirlestiremez, :SevkEdilmisPartiFiyatGuncelle, :KKBitirdeFireKontrolu;

  if (Old.PartiStokIptalKayitId is null and New.PartiStokIptalKayitId is not null and (new.NetSevkKilo <> 0 or new.NetSevkMetre <> 0)) then
    exception SevkEdilmisIptalEdilemez;

  if (New.PartiStokIptalKayitId is not null and Old.ProxyId <> New.ProxyId and New.ProxyId is not null) then exception IptalEdilmisKartAtanamaz;

  if (Old.PartiStokIptalKayitId is not null and New.PartiStokIptalKayitId is not null and New.LinkIslemi = 0) then exception IptalEdilmisDegistirilemez;

  if ( (Old.BoyaSiparisDetayId <> New.BoyaSiparisDetayId) or
       (Old.BoyaSiparisDetayId is null and New.BoyaSiparisDetayId is not null) or
       (Old.BoyaSiparisDetayId is Not null and New.BoyaSiparisDetayId is null)) then exception PartiSiparisNoDegisemez;

  if ((Old.BoyaSiparisSarjId is null and new.BoyaSiparisSarjId is not null) or
      (Old.BoyaSiparisSarjId is not null and new.BoyaSiparisSarjId is null) or
      (Old.BoyaSiparisSarjId <> new.BoyaSiparisSarjId) ) then exception PartiStokSarjIdDegisemez;

/*  if ((Old.BoyaSiparisBaskiId is null and new.BoyaSiparisBaskiId is not null) or
      (Old.BoyaSiparisBaskiId is not null and new.BoyaSiparisBaskiId is null) or
      (Old.BoyaSiparisBaskiId <> new.BoyaSiparisBaskiId) ) then exception PartiStokBaskiIdDegisemez;
*/

/*GK Henuz hareket görmemiş son cikis miktartı giris miktarı kadardır */
  if (New.PartiAsamalariId is null) then
  begin
    New.SonCikisKilo = New.Kilo;
    New.SonCikisMetre = New.Metre;
  end

  if (new.HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
  begin
    New.Miktar = New.kilo;
    New.NetMiktar = New.NetKilo;
    New.BrutSevkMiktar = New.BrutSevkKilo;
    New.NetSevkMiktar = New.NetSevkKilo;
    New.SonCikisMiktar = New.SonCikisKilo;

    New.BrutSarfMiktar = New.BrutSarfKilo;
    New.NetSarfMiktar = New.NetSarfKilo;

    New.BrutIadeMiktar = New.BrutIadeKilo;
    New.NetIadeMiktar = New.NetIadeKilo;
  end
  else
  begin
    New.Miktar = New.Metre;
    New.NetMiktar = New.NetMetre;
    New.BrutSevkMiktar = New.BrutSevkMetre;
    New.NetSevkMiktar = New.NetSevkMetre;
    New.SonCikisMiktar = New.SonCikisMetre;

    New.BrutSarfMiktar = New.BrutSarfMetre;
    New.NetSarfMiktar = New.NetSarfMetre;

    New.BrutIadeMiktar = New.BrutIadeMetre;
    New.NetIadeMiktar = New.NetIadeMetre;
  end

  if (PartiNetBruttenFazlaOlabilir = 0 and New.NetKilo > New.Kilo) then exception BrutKiloyuAstiniz;

/*  if (New.KKBitirenPersonelId is not null and New.TamamiTartildi = 0 and KKBitinceTartiyaAktar = 1) then exception TopTopKontrolVarManuelOlmaz;  Mahmut Yeni aşamalı kalite kontrol olayında sıkıntı yaratıyor...*/

  if (IntKarsilastir(Old.ProsesID, New.ProsesId) = 0) then
  if (new.ProsesID is not null) then
  begin
    Select Kapali from Proses where Proses.ProsesID = New.ProsesID into :Kapali;
    if (Kapali = 1) then exception ProsesKapaliSecilemez;
  end
/*  if (New.TamamiTartildi = 1 and Old.TamamiTartildi = 1) then
  if (Old.NetKilo <> New.NetKilo or Old.NetMetre <> New.NetMetre) then exception TamamiTartilmisKiloDegisemez;
*/

  if (New.MaximumFireOrani > 99) then exception KayitHatasi '  Fire Oranı Hatalı..';

  if (New.MinimumFireOrani > 99 or New.MinimumFireOrani < 0 ) then exception KayitHatasi '  Fire Oranı Hatalı..';

  if (New.TartiIcinFireOrani > 99) then exception KayitHatasi ' Tartı Fire Oranı Hatalı..';

  if (FireOnayiUcuncuFireden = 1) then KontrolEdilecekMaxFire = New.TartiIcinFireOrani;
  else KontrolEdilecekMaxFire = New.MaximumFireOrani;

  if (Old.TamamiTartildi = 0 and new.Tamamitartildi = 1) then /* Fire Yuksekmi kontrol et */
  begin
    EksiFireKontrolEdilecek = 1;

    Select Max(UniteIrsaliyeId) from UniteIrsaliyeDetay where UniteIrsaliyeDetay.PartiSinifId = New.PartiSinifId into :MaxUniteIrsaliyeId;

    if (MaxUniteIrsaliyeId is not null) then
    begin
      Select Unite.EksiFireKontrolEdilecek from UniteIrsaliye
        left join Unite on Unite.UniteId = UniteIrsaliye.AliciUniteId
        where UniteIrsaliye.UniteIrsaliyeID = :MaxUniteIrsaliyeId
        into :EksiFireKontrolEdilecek;
    end

    new.FireOnayiIstensin = 0;

    IptaldenGelenFark = 0;

    if (IptalFiresiniYeniPartiyeYansit = 1) then
    begin
      Select Sum(PartiStokDetay.Miktar), Max(rafDetay.IptalPartiFireOrani) from PartiStokDetay
        left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayId
        where PartiStokDetay.PartiSinifId = New.PartiSinifId and
              RafDetay.PartiStokIptalKayitId is not null
        into :IptaldenAcilanMiktar, :IptalPartiFireOrani;

      if (IptalPartiFireOrani is null) then IptalPartiFireOrani = 0;
      if (IptaldenAcilanMiktar is null) then IptaldenAcilanMiktar = 0;

      if (IptaldenAcilanMiktar > 0) then
      begin
        if (IptalPartiFireOrani <> 100) then IptaldenGelenFark = IptaldenAcilanMiktar * 100 / (100 - IptalPartiFireOrani) - IptaldenAcilanMiktar;
      end
    end

    new.FireOrani = ((new.Miktar +  IptaldenGelenFark - New.NetMiktar) / (New.Miktar + IptaldenGelenFark)) * 100;

    new.FireOrani = Yuvarla(new.FireOrani, 100);

    if (new.NetMiktar < (new.Miktar + IptaldenGelenFark - (new.Miktar + IptaldenGelenFark) * KontrolEdilecekMaxFire / 100)) then new.FireOnayiIstensin = 1;
    else if (EksiFireKontrolEdilecek = 1 and (new.NetMiktar > ( (new.Miktar + IptaldenGelenFark) + (new.Miktar + IptaldenGelenFark) * New.MinimumFireOrani / 100)) ) then new.FireOnayiIstensin = 1;

    if (new.NetMiktar > 0 and new.NetMiktar = (new.NetSevkMiktar + new.NetSarfMiktar + new.RezervMiktar - new.NetIadeMiktar) and new.Miktar > (new.BrutSevkMiktar + new.BrutSarfMiktar + new.BrutRezervMiktar - new.BrutIadeMiktar)) then exception BrutKiloBrutSevkKiloyaEsitDegil;
  end
  else if (new.Tamamitartildi = 0) then
  begin
    new.FireOrani = 0;
    new.FireOnayiIstensin = 0;
  end

  if (new.FireOnayiIstensin = 1) then
  begin
    if (PartiFireOnayiIstensin = 0 and new.FireOnayiKullaniciId is null and Old.FireOnayiKullaniciId is null) then new.FireOnayiKullaniciId = 1; /* otomatik onay veriyoruz */

    if (Old.FireOnayiKullaniciId is null and new.FireOnayiKullaniciId is not null) then new.FireOnayTarihi = current_TimeStamp(0);
    else if (Old.FireOnayiKullaniciId is not null and new.FireOnayiKullaniciId is null) then
    begin
      if (new.NetSevkMiktar + new.NetSarfMiktar + new.RezervMiktar - new.NetIadeMiktar > 0) then exception SevkVarFireOnayIptalEdilemez;
      new.FireOnayTarihi = Null;
    end
  end
  else
  begin
    new.FireOnayiKullaniciId = null;
    new.FireOnayTarihi = Null;
  end

  if (KKBitirdeFireKontrolu = 1 and Old.KKBitirenPersonelId is Null and New.KKBitirenPersonelId is not Null ) then
  begin

    Select Sum(PSKaliteKontrol.Kilo) from PSKaliteKontrolAsama
      Left Join PSKaliteKontrol on PSKaliteKontrol.PSKaliteKontrolAsamaID = PSKaliteKontrolAsama.PSKaliteKontrolAsamaID
    where PSKaliteKontrolAsama.PartiSinifID = New.PartiSinifID and PSKaliteKontrolAsama.SonKayit = 1 into :FKaliteKontrolKilo;

    if (((New.Kilo * ((98 - New.MaximumFireOrani) / 100)) > FKaliteKontrolKilo) and (New.Kilo > 100)) then Exception KayitHatasi 'Maximum Fire Oranı Aşıldı...';
    else if ((100*(( New.Kilo-FKaliteKontrolKilo)/New.Kilo) > 20) and (New.Kilo < 100) and (New.Kilo > 50)) then Exception KayitHatasi 'Maximum Fire Oranı Aşıldı...';
  end

  if ((new.NetSevkMiktar + New.NetSarfMiktar + new.RezervMiktar - new.NetIadeMiktar) > new.NetMiktar) then
  begin
    if (new.NetMiktar < old.NetMiktar) then
      Exception PartiMiktarindanFazlaRezerv;
    else
      Exception NetKiloMetreAstiniz;
  end

  if (new.TamamiTartildi = 1) then
  begin
    Select Count(*) from PartiStokTarti where PartiSinifId = New.PartiSinifId and PartiStokTarti.HamStokSarfFisiDetayId is Not Null into :SarfEdilenTopSayisi;

    if ((new.NetSevkMiktar > 0 or new.NetSarfMiktar > 0 or SarfEdilenTopSayisi > 0) and (new.NetMiktar = (new.NetSevkMiktar  + new.NetSarfMiktar - new.NetIadeMiktar))) then
    begin
      new.Durumu = 8;
    end
  end

  if (new.Durumu = 8 and (new.NetSevkMiktar + New.NetSarfMiktar - new.NetIadeMiktar) < new.NetMiktar) then New.Durumu = 7;

  Select FiyatOnayiKarariSevktenSonra, ProxyPartiSevkOluncaBosaCiksin, ProxyPartiSevkHazirdaBosalsin from Sistem where SistemID = 1 into :FiyatOnayiKarariSevktenSonra, :ProxyPartiSevkOluncaBosaCiksin, :ProxyPartiSevkHazirdaBosalsin;

  if (ProxyPartiSevkHazirdaBosalsin = 1 and new.Durumu = 7) then new.ProxyId = null;
  else if (ProxyPartiSevkOluncaBosaCiksin = 0 and (new.Durumu >= 7 and new.Tamamitartildi = 1) ) then new.ProxyId = null;
  else if (ProxyPartiSevkOluncaBosaCiksin = 1 and new.Durumu > 7) then new.ProxyId = null;
  else if (new.Durumu = 7) then
  begin
    Select IslemTipiId from Parti where Parti.PartiKayitID = new.PartiKayitID into :IslemTipiId;
    if (IslemTipiId = 5)  then new.ProxyId = null;
  end
  else if (new.YariMamulTamamlandi = 1) then new.ProxyId = null;


  if (Old.FirmaSevkAdresiId <> New.FirmaSevkAdresiId and (New.NetSevkMiktar + New.NetSarfMiktar - new.NetIadeMiktar) <> 0) then exception SevkEdilmisSevkAdresiDegisemez;

  if (Old.Durumu <> New.Durumu and New.Durumu = 7 and New.MamulKodu is null and New.Kilo > 0) then
  begin
    select Fason from Parti where Parti.PartiKayitId = New.PartiKayitId into :PartiFason;

    if (PartiFason = 0) then Select MamulKodu from MamulKoduHesapla(New.PartiSinifId) into new.MamulKodu;
  end
  else if (Old.Durumu <> New.Durumu and Old.Durumu = 7 and New.Durumu < 7 and new.BrutSevkMiktar = 0) then New.MamulKodu = null;

  if ((Old.ProxyId is null and New.ProxyId is not null) or
      (Old.ProxyId is not null and new.ProxyId is not null and Old.ProxyId <> new.ProxyId)) then
  begin
    select count(*) from PartiStok where PartiStok.ProxyId = new.ProxyId into :KayitSayisi;
    if (KayitSayisi > 0) then exception ProxyBaskaPartideKullanimda;
    else execute Procedure ProxyOkuKaydet(new.ProxyId);
    insert into PartiProxyLog(ProxyId, Tarih, PartiKayitId, PartiSinifId) values(new.ProxyId, current_TimeStamp(0), new.PartiKayitId, new.PartiSinifId);
  end

/* ekrandan Fiyat Guncellendigi zaman dovizid yi null' e cekiyoruz ki guncelleyecegimizi anlayalım */
  if (new.DovizId is null) then Select BirimFiyat, DovizId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayId = new.BoyaSiparisDetayId into new.BirimFiyat, new.DovizId;
  if (new.FasonBirimFiyatDovizID is null) then Select FasonBirimFiyati, FasonBirimFiyatDovizID from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayId = new.BoyaSiparisDetayId into new.FasonBirimFiyati, new.FasonBirimFiyatDovizID;

  if ((new.NetSevkMiktar + New.NetSarfMiktar) <> 0 and SevkEdilmisPartiFiyatGuncelle <> 1) then
  begin
    if (Old.DovizId <> New.DovizId) then
    begin
      Select Count(*) from BoyaIrsaliyeDetay
        left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
        left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
        where BoyaIrsaliyeDetay.PartiSinifId = New.PartiSinifId and BoyaIrsaliyeDetay.KaliteNo = 1 and
              BoyaIrsaliyeDetay.BoyaFaturaId is not null
         into :KayitSayisi;

      if (KayitSayisi > 0) then exception PartiSevkEdilmisFiyatDegisemez;
    end

    if (Old.BirimFiyat <> New.BirimFiyat) then
    begin
      Select Count(*) from BoyaIrsaliyeDetay
        left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
        left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
        where BoyaIrsaliyeDetay.PartiSinifId = New.PartiSinifId and BoyaIrsaliyeDetay.KaliteNo = 1 and
              BoyaIrsaliyeDetay.BoyaFaturaId is not null and
              BoyaFaturaDetay.SiparisBirimFiyat <> New.BirimFiyat into :KayitSayisi;

      if (KayitSayisi > 0) then exception PartiSevkEdilmisFiyatDegisemez;
    end
  end

  if (Old.DovizId <> New.DovizId and IrsFarkliDovizBirlestiremez = 1) then
  begin
    /*Eger İrsaliyede 1 den fazla PartiSinif varsa kızalım*/
    Select Count(BI.BoyaIrsaliyeID) from BoyaIrsaliyeDetay
      left Join BoyaIrsaliyeDetay BI on BI.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeID
      where BoyaIrsaliyeDetay.PartiSinifID =  New.PartiSinifId into :KayitSayisi;

    if (KayitSayisi > 1) then exception IrsIcinPartiDovizDegisemez;
  end



  if (Old.PartiPakettenFiyatlandirildi = 1 and New.PartiPakettenFiyatlandirildi = 0) then
  begin
    New.FiyatOnaylayanKullaniciId = Null;
    New.FiyatOnayTarihi           = Null;
  end
  else if (new.FiyatOnaylayanKullaniciId is null) then
  begin
    New.PartiPakettenFiyatlandirildi = 0;
    New.FiyatOnayTarihi = Null;
  end

  if (FiyatOnayiKarariSevktenSonra = 1) then /* modern bu sekilde calısıyor, sevk edilince onay tipine göre onay istenecekmi diye bakıyoruz */
  begin
    if ((Old.NetSevkMiktar + Old.NetSarfMiktar) = 0 and (New.NetSevkMiktar + New.NetSarfMiktar) > 0) then
    begin
      Select OnayIstensin from FiyatOnayIstensinmi(New.PartiSinifId, New.BoyaSiparisDetayId, New.BirimFiyat, New.DovizId) into New.FiyatOnayiIstensin;
    end
    else if ((Old.NetSevkMiktar + Old.NetSarfMiktar) > 0 and (New.NetSevkMiktar + New.NetSarfMiktar) = 0) then
    begin
      if (New.PartiPakettenFiyatlandirildi = 0) then
      begin
        New.FiyatOnayiIstensin = 0;
        New.FiyatOnaylayanKullaniciId = Null;
        New.FiyatOnayTarihi           = Null;
      end
    end
  end
  else /* sevke hazır olunca fiyat onayı isteyecekmiyiz kararını veriyoruz */
  begin
    if (new.FiyatOnaylayanKullaniciId is null) then
    begin
      if (Old.Durumu < 7 and new.Durumu = 7) then
        Select OnayIstensin from FiyatOnayIstensinmi(New.PartiSinifId, New.BoyaSiparisDetayId, New.BirimFiyat, New.DovizId) into New.FiyatOnayiIstensin;
      else if (Old.Durumu = 7 and new.Durumu < 7) then
        Select OnayIstensin from FiyatOnayIstensinmi(New.PartiSinifId, New.BoyaSiparisDetayId, New.BirimFiyat, New.DovizId) into New.FiyatOnayiIstensin;
    end
  end

  if ((Old.BirimFiyat <> New.BirimFiyat or Old.DovizId <> New.DovizId) and SevkEdilmisPartiFiyatGuncelle <> 1) then
  begin
    Select Count(*) from BoyaIrsaliyeDetay
      left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
      where BoyaIrsaliyeDetay.PartiSinifId = New.PartiSinifId and BoyaIrsaliyeDetay.KaliteNo = 1 and BoyaIrsaliyeDetay.BoyaFaturaId is not null and
            BoyaFaturaDetay.SiparisBirimFiyat <> New.BirimFiyat
      into :KayitSayisi;

    if (KayitSayisi > 0) then exception FaturaKesilmisFiyatDegisemez;

    if (new.BirimFiyat <> 0) then
    begin
      Select Count(*) from PartiPaket where PartiPaket.PartiSinifID = new.PartiSinifID into :KayitSayisi;
      if (KayitSayisi > 0) then exception PartiFiyatlandirmaHatali;
    end

    if (New.PartiPakettenFiyatlandirildi = 0) then
    begin
      Select OnayIstensin from FiyatOnayIstensinmi(New.PartiSinifId, New.BoyaSiparisDetayId, New.BirimFiyat, New.DovizId) into New.FiyatOnayiIstensin;

      New.FiyatOnaylayanKullaniciId = Null;
      New.FiyatOnayTarihi           = Null;
    end
  end

  if (Old.BirimFiyat <> New.BirimFiyat or Old.DovizId <> New.DovizId or old.Kilo <> new.Kilo) then
    execute procedure Logla('U', 1, 'PARTISTOK', 'PARTISINIFID', new.PartiSinifID, null);

  if (Old.FiyatOnaylayanKullaniciId is null and New.FiyatOnaylayanKullaniciId is not null) then
  begin
    Select Count(*) from PartiPaket where PartiPaket.PartiSinifID = new.PartiSinifID into :KayitSayisi;

    if  ( (New.FiyatOnayiIstensin = 0 and New.PartiPakettenFiyatlandirildi = 0) or
          (KayitSayisi > 0 and New.PartiPakettenFiyatlandirildi = 0) or
          (KayitSayisi = 0 and New.PartiPakettenFiyatlandirildi = 1)) then exception PartiFiyatlandirmaHatali;

    New.FiyatOnayTarihi = current_TimeStamp(0);
  end

  if (Old.FiyatOnaylayanKullaniciId is Not null and New.FiyatOnaylayanKullaniciId is null) then
  begin
    Select Count(*) from BoyaIrsaliyeDetay
      left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
      where BoyaIrsaliyeDetay.PartiSinifId = New.PartiSinifId and BoyaIrsaliyeDetay.BoyaFaturaId is not null into :KayitSayisi;

    if (KayitSayisi > 0) then exception FaturaKesilmisFiyatDegisemez;

    New.FiyatOnayTarihi = Null;
  end

  if ((New.NetSevkMiktar) > 0 or (Old.NetSevkMiktar <> New.NetSevkMiktar) ) then
  begin
    Select max(IrsaliyeTarihi) from BoyaIrsaliyeDetay where BoyaIrsaliyeDetay.PartiSinifId = new.PartiSinifId into new.SevkTarihi;
    Select min(IrsaliyeTarihi) from BoyaIrsaliyeDetay where BoyaIrsaliyeDetay.PartiSinifId = new.PartiSinifId into new.IlkSevkTarihi;
  end

  if (New.HamKontrolKullaniciId is null) then New.HamKontrolDurumu = 0;
  if (New.HamKontrolOnayKullaniciId is null) then New.HamKontrolOnayDurumu = 0;

  if (new.HamKontrolOnayDurumu <> 0 and Old.HamKontrolDurumu <> new.HamKontrolDurumu and Old.HamKontrolOnayDurumu <> 0) then exception OnayVerilmisHKontrolDegismez;
  if (new.HamKontrolOnayDurumu <> 0 and new.HamKontrolDurumu = 0) then exception HamKontrolYapilmamis;

  if (Old.HamKontrolDurumu <> New.HamKontrolDurumu) then
  begin
    if (New.HamKontrolDurumu <> 0) then new.HamKontrolTarihi = current_TimeStamp(0);
    else if (New.HamKontrolDurumu = 0) then
    begin
      New.HamKontrolKullaniciId = Null;
      New.HamKontrolTarihi = Null;
    end
  end

  if (Old.HamKontrolOnayDurumu <> New.HamKontrolOnayDurumu) then
  begin
    if (New.HamKontrolOnayDurumu <> 0) then new.HamKontrolOnayTarihi = current_TimeStamp(0);
    else if (New.HamKontrolOnayDurumu = 0) then
    begin
      New.HamKontrolOnayKullaniciId = Null;
      New.HamKontrolOnayTarihi = Null;
    end

    if (new.HamKontrolOnayDurumu = 2) then Execute Procedure SonrakiAsamayiBeklemeyeAl(new.PartiSinifId, Null, Null, 0, 1); /* elimizde partiasamasi yok yapılmamıs ilk asamayi beklemeye alacak */
  end

  if (New.FaturaKesilecek <> Old.FaturaKesilecek) then
  begin
    FBoyaSiparisDetayIdler = rdb$get_context('USER_TRANSACTION', 'KombinPartiUcretliTamir');

    if (FBoyaSiparisDetayIdler is null) then
      Select Count(*) from PartiStok
        where PartiStok.PartiKayitId = new.PartiKayitId and PartiStok.PartiSinifId <> New.PartiSinifId and
             PartiStok.BoyaSiparisDetayId <> New.BoyaSiparisDetayId
      into :KayitSayisi;
     else
     begin
       Sorgu = 'Select Count(*) from PartiStok '
               ||' where PartiStok.PartiKayitId = '||new.PartiKayitId||' and PartiStok.PartiSinifId <> '||New.PartiSinifId||' and '
               ||' PartiStok.BoyaSiparisDetayId not in ('||:FBoyaSiparisDetayIdler||')';

       execute statement (Sorgu) into :KayitSayisi;
     end

    if (KayitSayisi = 0) then Update Parti Set FaturaKesilecek = new.FaturaKesilecek where Parti.PartiKayitId = new.PartiKayitId;
    else
    begin

      if (FBoyaSiparisDetayIdler is null) then
        Select Count(*) from PartiStok
          where PartiStok.PartiKayitId = new.PartiKayitId and PartiStok.PartiSinifId <> New.PartiSinifId and
                PartiStok.BoyasiparisDetayId <> New.BoyasiparisDetayId and
                PartiStok.FaturaKesilecek <> New.FaturaKesilecek
         into :KayitSayisi;
       else
       begin
         Sorgu = 'Select Count(*) from PartiStok '
               ||' where PartiStok.PartiKayitId = '||new.PartiKayitId||' and PartiStok.PartiSinifId <> '||New.PartiSinifId||' and PartiStok.FaturaKesilecek <> '||New.FaturaKesilecek||' and '
               ||' PartiStok.BoyaSiparisDetayId not in ('||:FBoyaSiparisDetayIdler||')';

         execute statement (Sorgu) into :KayitSayisi;
       end

      if (kayitSayisi > 0) then exception PSFaturaKesilecekKarisikOlamaz;

      Update Parti Set FaturaKesilecek = new.FaturaKesilecek where Parti.PartiKayitId = new.PartiKayitId;
    end
  end

  if (New.BoyaSiparisDetayId is not null) then
  if (Old.Kilo <> New.Kilo or Old.Metre <> New.Metre) then
  begin
    Select Max(BoyaSiparisDetayOngoru.OngorulenEn), Max(BoyaSiparisDetayOngoru.OngorulenGramaj), Max(HamStokMaster.NemOrani), Max(HamStokMaster.IplikNemOrani) from PartiStokDetay
      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join BoyaSiparisDetayOngoru on BoyaSiparisDetayOngoru.BoyaSiparisDetayId = new.BoyaSiparisDetayId and BoyaSiparisDetayOngoru.HamStokDetayId = RafDetay.HamStokDetayId
      where PartiStokDetay.PartiSinifId = new.PartiSinifId
     into New.OngorulenEn, new.OngorulenGramaj, New.HamNemOrani, New.HamIplikNemOrani;
  end

  if (new.NetSevkMiktar = 0) then
  begin
    new.HesaplananFiyat = null;
    new.HesaplananUcretliTamirFiyati = null;
    new.HesaplananUCretsizTamirFiyati = null;
  end

  if ((new.NetSevkMiktar > 0 and Old.NetSevkMiktar = 0) or (new.NetSevkMiktar > 0 and Old.DovizId <> New.DovizId) ) then
  begin
    Select HesaplananFiyat, HesaplananUcretliTamirFiyati, HesaplananUCretsizTamirFiyati from PartiStokFiyatHesapla(new.PartiKayitId, new.PartiSinifId, new.DovizId) into new.HesaplananFiyat, new.HesaplananUcretliTamirFiyati, new.HesaplananUCretsizTamirFiyati;
  end

  if (old.GenelKontrolKullaniciId is null and new.GenelKontrolKullaniciId is not null) then new.GenelKontroltarihi = current_TimeStamp(0);
  else if (new.GenelKontrolKullaniciId is null) then new.GenelKontroltarihi = Null;

  if (StrKarsilastir(Old.Gramaj, New.Gramaj) = 0 or IntKarsilastir(Old.HacimEn, New.HacimEn) = 0 ) then
  begin
    New.En = EgemenIntegerToStr(New.HacimEn);
    Select KKEnGramajiFizikKontroleKaydet from sistem where sistemid = 1 into :KKEnGramajiFizikKontroleKaydet;

    if (KKEnGramajiFizikKontroleKaydet = 1) then
       Update PartiStokFizikKontrol Set EnVerilen = New.En, GramajVerilen = New.Gramaj where PartiStokFizikKontrol.PartiSinifID = new.PartiSinifID and PartiStokFizikKontrol.SonKayit = 1;
  end

  if (Old.PartilemeOnayiKullaniciId is null and New.PartilemeOnayiKullaniciId is not null) then new.PartilemeOnayTarihi = Current_TimeStamp(0);
  else if (Old.PartilemeOnayiKullaniciId is not null and New.PartilemeOnayiKullaniciId is null) then new.PartilemeOnayTarihi = null;

  if (old.Durumu < 7 and new.Durumu = 7) then new.SevkeHazirTarihi = Current_TimeStamp(0);
  else if (new.Durumu < 7 and old.Durumu >= 7) then new.SevkeHazirTarihi = Null;

  if (Old.PartiStokIptalKayitID is null and new.PartiStokIptalKayitID is not null) then
    new.TasarimVaryantNo = null;


  New.LinkIslemi = 0;
  new.DegisimTarihi = current_TimeStamp(0);
end
