CREATE TRIGGER BEFOREUPDATEBEDEN FOR BEDEN
ACTIVE BEFORE UPDATE POSITION 0
as
Declare Variable KS Integer;
begin
  if (Old.KisaKodu <> New.KisaKodu) then
  begin
    Select Count(*) from ModelBeden Where ModelBeden.BedenId = New.BedenID into :KS;
    if (KS > 0) then Exception KayitHatasi ' Bu Beden Model Tanıtımında Kullanılmış, Kısa Kodu Değiştirilemez..';
  end
  new.DegisimTarihi = current_TimeStamp(0);
end
