CREATE TRIGGER BEFOREUPDATEORMEFIS1 FOR ORMEFIS
ACTIVE BEFORE UPDATE POSITION 1
as
declare variable FisKayitSayisi  integer;
begin
  Select Count(*) from StokHareket where StokHareket.OrmeFisId = new.OrmeFisId into :FisKayitSayisi;

  if (FisKayitSayisi > 0) then
  begin
    if (StrKarsilastir(Old.FirmaId, New.FirmaId) = 0) then exception DetayGirilmisFirmaDegisemez; /* değişirse boyahane cikis parti no yu yeniden hesaplamak gerekir */
    if (Old.UretimIslemTipiId <> New.UretimIslemTipiId) then exception UretimIslemTipiDegistirilemez;
  end

  if (New.OrmeUretimeCikisOtomatik = 0 and New.SarfFisiID is not null) then exception KayitHatasi ' SarfFişiID Hatası...';
end
