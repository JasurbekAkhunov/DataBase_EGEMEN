Definition

CREATE TRIGGER AFTERUPDATEPARTISTOK1 FOR PARTISTOK
ACTIVE AFTER UPDATE POSITION 1
as
declare variable SonCikisKilo       Numeric(12,3);
declare variable SonCikisMetre      Numeric(12,3);
declare variable SonCikisMiktar     Numeric(12,3);
declare variable FPartiAsamalariId  integer;
declare variable FCikisTarihi       TimeStamp;
declare variable FMiktar            Numeric(12,3);
declare variable FImalatMiktari     Numeric(12,3);
declare variable FPartiAsamaStokDetayId integer;
declare variable ReceteId           varchar(20);
declare variable LabReceteId        varchar(20);
declare variable IlkBoyananPartiKayitId integer;
declare variable LabIlkBoyananPartiKayitId integer;
declare variable HamTipiId           integer;
declare variable FKilo               numeric(12,3);
declare variable FMetre              numeric(12,3);
declare variable FpersonelID         integer;
declare variable Sonuc               Varchar(255);
declare variable DesenID             varchar(20);
declare variable KullaniciId         Integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable FFirmaAdi           Adi100;
declare variable FPartiIdStr         varchar(30);
declare variable DegistirenKullaniciAdi varchar(40);
declare variable LogAnahtar varchar(4000);
begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  SonCikisKilo  = New.Kilo;
  SonCikisMetre = New.Metre;
  SonCikisMiktar = New.Miktar;

  if (Old.Kilo <> New.Kilo or Old.Metre <> New.Metre or Old.BobinSayisi <> New.BobinSayisi) then
  begin
    for Select PartiAsamaStok.PartiAsamalariId, PartiAsamaStok.CikisTarihi from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = new.PartiSinifId
      order by PartiAsamalari.Sira
      into :FPartiAsamalariId, :FCikisTarihi do
    begin
      Update partiAsamaStok Set GirisKilo = :SonCikisKilo, GirisMetre = :SonCikisMetre where partiAsamaStok.PartiSinifId = New.PartiSinifId and partiAsamaStok.PartiAsamalariId = :FPartiAsamalariId;

      FPartiAsamaStokDetayId = null;

      Select Sum(PartiAsamaStokDetay.Miktar), Max(PartiAsamaStokDetayId) from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiSinifId = new.PartiSinifId and PartiAsamaStokDetay.PartiAsamalariId = :FPartiAsamalariId
            into :FImalatMiktari, :FPartiAsamaStokDetayId;

      if (FPartiAsamaStokDetayId is not null) then
      begin
        FMiktar = SonCikisMiktar;

        FMiktar = FMiktar - FImalatMiktari;

        Update PartiAsamaStokDetay Set Miktar = Miktar + :FMiktar where PartiAsamaStokDetay.PartiAsamaStokDetayID = :FPartiAsamaStokDetayId;
      end

      if (FCikisTarihi is not null) then
        Select Sum(CikisKilo), Sum(CikisMetre), Sum(CikisMiktar) from PartiAsamaStokDetay where partiAsamaStokDetay.PartiSinifId = New.PartiSinifId and partiAsamaStokDetay.PartiAsamalariId = :FPartiAsamalariId
              into :SonCikisKilo, :SonCikisMetre, :SonCikisMiktar;
    end
  end

  if (IntKarsilastir(Old.TasarimVaryantNo, New.TasarimVaryantNo) = 0) then
  begin
    Select BoyaSiparisDetay.DesenId from BoyaSiparisDetay
          where BoyaSiparisDetay.BoyaSiparisDetayID = New.BoyaSiparisDetayID
    into :DesenId;

    if (New.TasarimVaryantNo is not null) then
      insert into DesenTasarimVaryantNo(DesenId, TasarimVaryantNo) values(:DesenId, new.TasarimVaryantNo);

    if (Old.TasarimVaryantNo is not null) then
      Delete from DesenTasarimVaryantNo where DesenId = :DesenId and TasarimVaryantNo = Old.TasarimVaryantNo;
  end

  if (Old.Durumu <> New.Durumu and New.Durumu = 4) then
  begin
    Select Parti.ReceteId, LabRecete.LabReceteId,  Recete.IlkBoyananPartiKayitId,  LabRecete.IlkBoyananPartiKayitId from Parti
      left join Recete on Recete.ReceteID = Parti.ReceteId
      left join LabRecete on LabRecete.LabReceteID = Recete.LabReceteId
      where Parti.PartiKayitID = New.PartiKayitID and (LabRecete.EskiLabReceteKodu is null or LabRecete.EskiLabReceteKodu = '')
      into :ReceteId, :LabReceteId, :IlkBoyananPartiKayitId, :LabIlkBoyananPartiKayitId;

    if (IlkBoyananPartiKayitId is null) then
      Update Recete Set IlkBoyananPartiKayitId = New.PartiKayitID where Recete.ReceteID = :ReceteId;
    if (LabIlkBoyananPartiKayitId is null) then
      Update LabRecete Set IlkBoyananPartiKayitId = New.PartiKayitID where LAbRecete.LabReceteID = :LabReceteId;
  end

  if (Old.PartilemeOnayiKullaniciId is null and New.PartilemeOnayiKullaniciId is not null) then
  begin
     Select sum(Kilo-Dara), sum(Metre) from PartiStokPartileme where PartiStokPartileme.PartiSinifID = New.PartiSinifID into :FKilo, :FMetre;

     Select PersonelId from Kullanici where Kullanici.KullaniciID = new.PartilemeOnayiKullaniciId into :FPersonelId;

     if (new.HamTipiID = 0 or new.HamTipiID = 5) then execute procedure IplikTartiKiloGirisi(New.PartiSinifID, :FKilo, :FPersonelId);
     else if (new.HamTipiID = 2 and DokumaKumasiKilodanTakipEt = 0) then execute procedure KumasTartiMetreGirisi(New.PartiSinifID, :FMetre, :FPersonelId);
     else
     begin
       select Sonuc from KumasTartiKiloGirisi(New.PartiSinifID, :FKilo, :FPersonelId) into :Sonuc;
       if (Sonuc = 'OK') then select Sonuc from KumasTartiKiloGirisi(New.PartiSinifID, :FKilo, :FPersonelId) into :Sonuc;
     end
  end

  if (old.Miktar <> new.Miktar and old.Miktar > 0) then
    for select Kullanici.KullaniciId from Kullanici where Kullanici.PartiMiktarGuncellemeBildir = 1 into :KullaniciId do
    begin
      select Firma.Adi, Parti.PartiYil||'-'||Parti.PartiId||'.'||new.PartiSiraNo from Parti
      left join Firma on Firma.FirmaId = Parti.FirmaId
      where Parti.PartiKayitId = new.PartiKayitId into :FFirmaAdi, :FPartiIdStr;

      insert into GelenMail(KullaniciId, KimdenKullaniciId, KimdenKullaniciAdi, Konu, Mesaj) values(:KullaniciId, 1, 'SİSTEM YÖNETİCİSİ', 'Parti Miktarı Değişti',
      :FFirmaAdi||cast(' Firmasına ait ' as blob character set UTF8)|| :FPartiIdStr ||cast(' Nolu ' as blob character set UTF8) || new.HamAdiToplami || cast(' Partisinin Miktarı Değiştirildi: Eski Miktar: ' as blob character set UTF8)||old.Miktar||cast(' Yeni Miktar : ' as blob character set UTF8)||new.Miktar);
    end

  if (old.BirimFiyat <> new.BirimFiyat) then
  begin
    for select Kullanici.KullaniciId from Kullanici where Kullanici.BirimFiyatBildirilsin = 1 into :KullaniciId do
    begin
      if (FFirmaAdi is null or (FPartiIdStr is null) ) then
      begin
        select Firma.Adi, Parti.PartiYil||'-'||Parti.PartiId||'.'||new.PartiSiraNo from Parti
        left join Firma on Firma.FirmaId = Parti.FirmaId
        where Parti.PartiKayitId = new.PartiKayitId into :FFirmaAdi, :FPartiIdStr;
      end

      insert into GelenMail(KullaniciId, KimdenKullaniciId, KimdenKullaniciAdi, Konu, Mesaj)
      values(:KullaniciId, 1, 'SİSTEM YÖNETİCİSİ', 'Parti Birim Fiyat Değişti',
      :FFirmaAdi||cast(' Firmasına ait ' as blob character set UTF8)|| :FPartiIdStr ||cast(' Nolu ' as blob character set UTF8) || new.HamAdiToplami || cast(' Partisinin Birim Fiyatı Değiştirildi: Eski Birim Fiyat: ' as blob character set UTF8)||old.BirimFiyat||cast(' Yeni Birim Fiyat : ' as blob character set UTF8)||new.BirimFiyat);
    end
  end

  if (old.BirimFiyat <> new.BirimFiyat or old.DovizId <> new.DovizId) then
  begin
    select Kullanici.Adi from Kullanici where Kullanici.kullaniciid = new.KullaniciId into :DegistirenKullaniciAdi;
    LogAnahtar = cast(DegistirenKullaniciAdi||' '||substring(new.DegisimTarihi from 1 for 19)||' Eski Birim Fiyat:'||Round(old.BirimFiyat, 4)||' '||old.DovizId||' Yeni Birim Fiyat:'||Round(new.BirimFiyat, 4)||' '||new.DovizId as blob character set UTF8);
    execute procedure Logla('U', 2, 'PARTISTOK', 'PARTISINIFID', new.PartiSinifId, :LogAnahtar);
  end
  else if (old.Kilo <> new.Kilo) then
  begin
    select Kullanici.Adi from Kullanici where Kullanici.kullaniciid = new.KullaniciId into :DegistirenKullaniciAdi;
    LogAnahtar = cast(DegistirenKullaniciAdi||' '||substring(new.DegisimTarihi from 1 for 19)||' Eski Kilo:'||Round(old.Kilo, 4)||' '||' Yeni Kilo:'||Round(new.Kilo, 4) as blob character set UTF8);
    execute procedure Logla('U', 2, 'PARTISTOK', 'PARTISINIFID', new.PartiSinifId, :LogAnahtar);
  end
end
