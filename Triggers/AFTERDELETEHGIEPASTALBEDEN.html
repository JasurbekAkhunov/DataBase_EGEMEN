Definition

CREATE TRIGGER AFTERDELETEHGIEPASTALBEDEN FOR HGISEMRIPASTALBEDEN
ACTIVE AFTER DELETE POSITION 0
as
begin
  Delete from HGISEmriPastalUrun where HGISEmriPastalUrun.HGISEmriPastalID = Old.HGISEmriPastalID and UrunID in (Select Urun.UrunID from HGISEmriPastal
    Left Join HGIsEmri on HGISEmri.HGISEmriKayitID = HGISEmriPastal.HGISEmriKayitID
    Left Join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayID = HGISEmri.HGSiparisDetayID
    Left Join ModelBeden on ModelBeden.ModelKayitId = HGSiparisDetay.ModelKayitId and ModelBeden.BedenID = Old.BedenID
    Left Join Urun on Urun.ModelKayitId = HGSiparisDetay.ModelKayitId and Urun.ModelBedenID = ModelBeden.ModelBedenID
  where HGISEmriPastal.HGISEmriPastalID = Old.HGISEmriPastalID) ;
end
