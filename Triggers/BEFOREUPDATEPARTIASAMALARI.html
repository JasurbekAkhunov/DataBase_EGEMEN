Definition

CREATE TRIGGER BEFOREUPDATEPARTIASAMALARI FOR PARTIASAMALARI
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KayitSayisi integer;
declare variable oAsamaId integer;
declare variable nAsamaId integer;
declare variable oReceteId varchar(20);
declare variable nReceteId varchar(20);
declare variable OBanyoCarpani Double Precision;
declare variable NBanyoCarpani Double Precision;
declare variable OBanyoCarpani2 Double Precision;
declare variable NBanyoCarpani2 Double Precision;
declare variable OBanyoCarpani3 Double Precision;
declare variable NBanyoCarpani3 Double Precision;
declare variable OBanyoCarpani4 Double Precision;
declare variable NBanyoCarpani4 Double Precision;
declare variable FIsEmriBanyoCarpani Double Precision;
declare variable FIsEmriBanyoCarpani2 Double Precision;
declare variable FIsEmriBanyoCarpani3 Double Precision;
declare variable FIsEmriBanyoCarpani4 Double Precision;
declare variable FIsEmriKoyultma     Double Precision;
declare variable FIsEmriHamStokFlotteCarpani Double Precision;
declare variable FIsEmriBoyamaSeviyesi       integer;
declare variable IslemTipiId integer;
declare variable IslemGrubuId integer;
declare variable Okey integer;
declare variable FRptKontrolu integer;
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable ReceteStoktanDusmeSekli integer;
declare variable ReceteUstAsamaId        integer;
declare variable FIsEmriTur              Double Precision;
declare variable FIsEmriPompaDegeri      integer;
declare variable FIsEmriDuze             integer;
declare variable AsamaDepoNo varchar(10);
declare variable ReceteYazildiDegisimKontYap integer;
declare variable ReceteYazildi integer;
begin
  if (New.Koyultma is null) then new.Koyultma = 0;

  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;
  if (((new.AsamaId is null) and (new.ReceteId is null)) or ((new.AsamaId is not null) and (new.ReceteId is not null))) then exception AsamaVeyaReceteSecmelisiniz;

  oAsamaId = Old.AsamaId;
  nAsamaId = new.AsamaId;
  oReceteId = Old.ReceteId;
  nReceteId = new.ReceteId;

  OBanyoCarpani = Old.BanyoCarpani;
  NBanyoCarpani = new.BanyoCarpani;

  OBanyoCarpani2 = Old.BanyoCarpani2;
  NBanyoCarpani2 = new.BanyoCarpani2;

  OBanyoCarpani3 = Old.BanyoCarpani3;
  NBanyoCarpani3 = new.BanyoCarpani3;

  OBanyoCarpani4 = Old.BanyoCarpani4;
  NBanyoCarpani4 = new.BanyoCarpani4;

  if (oAsamaId is null) then oAsamaId = -1;
  if (nAsamaId is null) then nAsamaId = -1;
  if (oReceteId is null) then oReceteId = '';
  if (nReceteId is null) then nReceteId = '';

  if (OBanyoCarpani is null) then OBanyoCarpani = 0;
  if (NBanyoCarpani is null) then NBanyoCarpani = 0;

  if (OBanyoCarpani2 is null) then OBanyoCarpani2 = 0;
  if (NBanyoCarpani2 is null) then NBanyoCarpani2 = 0;

  if (OBanyoCarpani3 is null) then OBanyoCarpani3 = 0;
  if (NBanyoCarpani3 is null) then NBanyoCarpani3 = 0;

  if (OBanyoCarpani4 is null) then OBanyoCarpani4 = 0;
  if (NBanyoCarpani4 is null) then NBanyoCarpani4 = 0;

  if ((old.OtomasyonaDahilmi <> new.OtomasyonaDahilmi) or
      (Old.CikistaKiloZorunlu <> New.CikistaKiloZorunlu) or
      (Old.CikistaMetreZorunlu <> New.CikistaMetreZorunlu) or
      (Old.CikistaArabaZorunlu <> New.CikistaArabaZorunlu) ) then
  begin
    select count(*) from PartiAsamaStok where PartiAsamalariId = old.PartiAsamalariId and (GirisTarihi is not null or CikisTarihi is not null) into :KayitSayisi;
    if (KayitSayisi > 0) then exception HareketGormusDegistiremezsiniz;
  end

  if (New.LinkIslemi = 0 and oAsamaId <> nAsamaId) then
  begin
    select count(*) from PartiAsamaStok where PartiAsamalariId = old.PartiAsamalariId and GirisTarihi is not null into :KayitSayisi;
    if (KayitSayisi > 0) then exception HareketGormusDegistiremezsiniz;
  end

  Select ReceteStoktanDusmeSekli, ReceteYazildiDegisimKontYap from Sistem where SistemID = 1 into :ReceteStoktanDusmeSekli, :ReceteYazildiDegisimKontYap;

  if (oReceteId <> nReceteId) then
  begin
    KayitSayisi = 0;

    if (ReceteStoktanDusmeSekli = 2) then
        select count(*) from PartiAsamaStok where PartiAsamalariId = old.PartiAsamalariId and GirisTarihi is not null into :KayitSayisi;
    else
      select count(*) from PartiAsamaStok where PartiAsamalariId = old.PartiAsamalariId and CikisTarihi is not null into :KayitSayisi;

    if (KayitSayisi > 0) then exception HareketGormusDegistiremezsiniz;
  end


  if (ReceteYazildiDegisimKontYap = 1) then
  begin
    Select ReceteYazildi from Parti where Parti.PartiKayitID = new.PartiKayitID into :ReceteYazildi;

    if (ReceteYazildi = 1) then
    begin
      Select Count(*) from PartiAsamalari where PartiAsamalari.PartiKayitID = New.PartiKayitID and PartiAsamalari.ReceteID is not Null and  New.Sira < PartiAsamalari.Sira
        into :KayitSayisi;

      if (KayitSayisi > 0) then Exception KayitHatasi ' Reçetesi Yazılmış Partinin Boya Öncesi Aşamalarına Müdahale Edilemez....';
    end
  end

  if (oAsamaId <> nAsamaId or oReceteId <> nReceteId) then
  begin
    if (New.AsamaId is not null) then
    begin
      Select Asama.CikistaKiloZorunlu, Asama.CikistaMetreZorunlu, Asama.CikistaArabaZorunlu, Asama.UretimMaliyeti,
        Asama.UretimMaliyetiDovizId, Asama.UretimMaliyetiKg, Asama.UretimMaliyetiKgDovizId, Asama.IscilikUretimMaliyeti,
        Asama.IscilikUretimMaliyetiDovizId, Asama.IscilikUretimMaliyetiKg, Asama.IscilikUretimMaliyetiKgDovizId, Asama.MutfakDepoNo from Asama
        where Asama.AsamaId = new.AsamaId
        into New.CikistaKiloZorunlu, new.CikistaMetreZorunlu, New.CikistaArabaZorunlu, New.UretimMaliyeti,
          New.UretimMaliyetiDovizId, new.UretimMaliyetiKg, new.UretimMaliyetiKgDovizId, new.IscilikUretimMaliyeti,
          new.IscilikUretimMaliyetiDovizId, new.IscilikUretimMaliyetiKg, new.IscilikUretimMaliyetiKgDovizId, :AsamaDepoNo;
    end
    else
    begin
      Select Asama.CikistaKiloZorunlu, Asama.CikistaMetreZorunlu, Asama.CikistaArabaZorunlu, Asama.UretimMaliyeti,
        Asama.UretimMaliyetiDovizId, Asama.UretimMaliyetiKg, Asama.UretimMaliyetiKgDovizId, Asama.IscilikUretimMaliyeti,
        Asama.IscilikUretimMaliyetiDovizId, Asama.IscilikUretimMaliyetiKg, Asama.IscilikUretimMaliyetiKgDovizId, Asama.MutfakDepoNo from Recete
      left join Asama on Asama.AsamaId = Recete.UstAsamaId
      where Recete.ReceteId = new.ReceteId
      into New.CikistaKiloZorunlu, new.CikistaMetreZorunlu, New.CikistaArabaZorunlu, new.UretimMaliyeti,
        New.UretimMaliyetiDovizId, new.UretimMaliyetiKg, new.UretimMaliyetiKgDovizId, new.IscilikUretimMaliyeti,
        new.IscilikUretimMaliyetiDovizId, new.IscilikUretimMaliyetiKg, new.IscilikUretimMaliyetiKgDovizId, :AsamaDepoNo;
    end
  end

  if (new.StoktanDusuldu = 1 and
      (  (oReceteId <> nReceteId) or
         (OBanyoCarpani <> NBanyoCarpani) or
         (OBanyoCarpani2 <> NBanyoCarpani2) or
         (OBanyoCarpani3 <> NBanyoCarpani3) or
         (OBanyoCarpani4 <> NBanyoCarpani4) or Old.Koyultma <> New.Koyultma) ) then exception StokDusulduDegisemez;

  if (OBanyoCarpani <> NBanyoCarpani or
      OBanyoCarpani2 <> NBanyoCarpani2 or
      OBanyoCarpani3 <> NBanyoCarpani3 or
      OBanyoCarpani4 <> NBanyoCarpani4 or
      Old.Koyultma <> new.Koyultma or
      IntKarsilastir(Old.BoyamaSeviyesi, New.BoyamaSeviyesi) = 0 or
      IntKarsilastir(Old.PompaDegeri, New.PompaDegeri) = 0 or
      IntKarsilastir(Old.Duze, New.Duze) = 0 or
      StrKarsilastir(Format_Float('',Old.Tur), Format_Float('',New.Tur))= 0 or
      StrKarsilastir(Format_Float('',Old.HamStokFlotteCarpani), Format_Float('',New.HamStokFlotteCarpani))= 0) then
  begin
    select BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, Koyultma, Tur, PompaDegeri, Duze, BoyamaSeviyesi, HamStokFlotteCarpani from ReceteIsEmriDetay
    left join ReceteIsEmri on ReceteIsEmri.ReceteIsEmriYil = ReceteIsEmriDetay.ReceteIsEmriYil and ReceteIsEmri.ReceteIsEmriId = ReceteIsEmriDetay.ReceteIsEmriId
    where ReceteIsEmriDetay.PartiAsamalariId = new.PartiAsamalariId
    into :FIsEmriBanyoCarpani, :FIsEmriBanyoCarpani2, :FIsEmriBanyoCarpani3, :FIsEmriBanyoCarpani4, :FIsEmriKoyultma, :FIsEmriTur, :FIsEmriPompaDegeri, :FIsEmriDuze, :FIsEmriBoyamaSeviyesi, :FIsEmriHamStokFlotteCarpani;

    if (FIsEmriBanyoCarpani is not null and New.BanyoCarpani <> FIsEmriBanyoCarpani) then exception BanyoCarpaniFarkliOlamaz;
    if (FIsEmriBanyoCarpani2 is not null and New.BanyoCarpani2 <> FIsEmriBanyoCarpani2) then exception BanyoCarpaniFarkliOlamaz;
    if (FIsEmriBanyoCarpani3 is not null and New.BanyoCarpani3 <> FIsEmriBanyoCarpani3) then exception BanyoCarpaniFarkliOlamaz;
    if (FIsEmriBanyoCarpani4 is not null and New.BanyoCarpani4 <> FIsEmriBanyoCarpani4) then exception BanyoCarpaniFarkliOlamaz;

    if (FIsEmriKoyultma is not null and New.Koyultma <> FIsEmriKoyultma) then exception BanyoCarpaniFarkliOlamaz;
    if (FIsEmriTur is not null and StrKarsilastir(Format_Float('',FIsEmriTur), Format_Float('',New.Tur))= 0) then exception BanyoCarpaniFarkliOlamaz;
    if (FIsEmriHamStokFlotteCarpani is not null and StrKarsilastir(Format_Float('',FIsEmriHamStokFlotteCarpani), Format_Float('',New.HamStokFlotteCarpani))= 0) then exception BanyoCarpaniFarkliOlamaz;
    if (FIsEmriBanyoCarpani is not null and IntKarsilastir(FIsEmriBoyamaSeviyesi, New.BoyamaSeviyesi) = 0) then exception BanyoCarpaniFarkliOlamaz;
    if (FIsEmriPompaDegeri is not null and IntKarsilastir(FIsEmriPompaDegeri, New.PompaDegeri) = 0) then exception BanyoCarpaniFarkliOlamaz;
    if (FIsEmriDuze is not null and IntKarsilastir(FIsEmriDuze, New.Duze) = 0) then exception BanyoCarpaniFarkliOlamaz;
  end

  if (OReceteId <> nReceteId) then
  begin
    Select count(*) from ReceteIsEmriDetay where ReceteIsEmriDetay.PartiAsamalariId = new.PartiAsamalariId into :KayitSayisi;
    if (KayitSayisi > 0) then exception IsEmrindeVarReceteDegisemez;

/* Rptmi Kontrol Et */
    if (new.receteId is not null) then
    begin
      Select RptKontrolu from sistem where sistemId = 1 into :FRptKontrolu;
      if (FRptKontrolu = 1) then
      begin
        select IslemTipiId, IslemGrubuId from Parti where Parti.PartiKayitId = new.PartiKayitId into :IslemTipiId, :IslemGrubuId;
        if (IslemTipiId = itNormal and IslemGrubuId = 1) then
        begin
          select Okey from ReceteRptKontrolu(new.PartiKayitId, new.ReceteId) into :Okey;
          if (Okey = 0) then exception PartiIslemTipiRptOlmali;
        end
      end
    end
  end

  if (new.ReceteId is null) then new.PartininAsilRecetesimi = 0;
  else
  if (new.PartininAsilRecetesimi = 0) then
  begin
/* secilen recete ile siparişte secilen recete uyuyormu diye bakıyoruz, uyuyorsa ve asıl recete girilmemiş ise asıl recete yapıcaz */
    Select Recete.UstAsamaId from Recete where Recete.ReceteID = new.ReceteId into :ReceteUstAsamaId;

    Select Count(*) from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join LabRecete on LabRecete.LabReceteID = BoyaSiparisDetay.LabReceteId
    where PartiStok.PartiKayitId = new.PartiKayitId and LabRecete.LabReceteId is not null and LabRecete.UstAsamaId = :ReceteUstAsamaId
    into :KayitSayisi;

    if (KayitSayisi > 0) then
    begin
      Select Count(*) from PartiAsamalari
        where PartiAsamalari.PartiKayitId = new.PartiKayitId and PartiAsamalari.ReceteId is not null and PartiAsamalari.PartininAsilRecetesimi = 1 into :KayitSayisi;
      if (KayitSayisi = 0) then new.PartininAsilRecetesimi = 1;
    end
  end

  if (oReceteId <> nReceteId or
      OBanyoCarpani <> NBanyoCarpani or
      OBanyoCarpani2 <> NBanyoCarpani2 or
      OBanyoCarpani3 <> NBanyoCarpani3 or
      OBanyoCarpani4 <> NBanyoCarpani4 or
      Old.PartininAsilRecetesimi <> new.PartininAsilRecetesimi) then
  begin
    if (New.ReceteID is not null and new.PartininAsilRecetesimi = 1) then
    begin
      if (New.BanyoCarpani is not null and New.BanyoCarpani > 0) then
        update Parti Set ReceteTarihi = current_TimeStamp(0) where PartiKayitId = new.PartiKayitId;
    end
  end

  if (new.MutfakDepoNo <> Old.MutfakDepoNo and new.StoktanDusuldu = 1) then exception StoktanDusulmusDepoDegisemez;
  if (new.MutfakDepoNo is Null and Old.MutfakDepoNo is not Null and new.StoktanDusuldu = 1) then exception StoktanDusulmusDepoDegisemez;
  if (new.MutfakDepoNo is not Null and Old.MutfakDepoNo is Null and new.StoktanDusuldu = 1) then exception StoktanDusulmusDepoDegisemez;

  if (old.MutfakDepoNo is Null and old.StoktanDusuldu = 1 and new.StoktanDusuldu = 0 and AsamaDepoNo is Not Null) then
    new.MutfakDepoNo = AsamaDepoNo;

  new.LinkIslemi = 0;

  if (Old.OtomasyonaDahilmi = 1 and new.OtomasyonaDahilmi = 0) then delete from BolumPlanlama where BolumPlanlama.PartiAsamalariID = New.PartiAsamalariID;
  new.DegisimTarihi = current_TimeStamp(0);
end
