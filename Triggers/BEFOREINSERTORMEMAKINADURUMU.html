CREATE TRIGGER BEFOREINSERTORMEMAKINADURUMU FOR ORMEMAKINADURUMU
ACTIVE BEFORE INSERT POSITION 0
as
declare variable OldPlanlamaId         integer;
declare variable FOrmeMakinaDurumuId   integer;
declare variable FAriza                SmallInt;
declare variable FArizaVar             SmallInt;
declare variable FAyarYapiliyor        SmallInt;
declare variable FServisciVar          SmallInt;
declare variable FMemingerKoptu        SmallInt;
declare variable FAltMemingerKoptu     SmallInt;
declare variable FLycraKoptu           SmallInt;
declare variable FStartStop            SmallInt;
declare variable FIgneKoptu            SmallInt;
declare variable FKapakAcildi          SmallInt;
declare variable FKenarStop            SmallInt;
declare variable FCozguKoptu           SmallInt;
declare variable FAtkiKoptu            SmallInt;
declare variable FLeventSonu           SmallInt;
declare variable FImalataAraVerildi    SmallInt;
declare variable FTur                  Integer;
begin
  FOrmeMakinaDurumuID = Null;
  if (new.OrmeMakinaId is not null) then
      select OrmeMakinaDurumuId /*, Ariza, Ayar, Bakim, AraVerme*/ from MakinaDurumuOzet where MakinaDurumuOzet.OrmeMakinaId = new.OrmeMakinaId into :FOrmeMakinaDurumuID;
  else
      select OrmeMakinaDurumuId /*, Ariza, Ayar, Bakim, AraVerme*/ from MakinaDurumuOzet where MakinaDurumuOzet.DokumaMakinaId = new.DokumaMakinaId into :FOrmeMakinaDurumuID;

  if (FOrmeMakinaDurumuID is null) then
  begin
    FAriza = 0;
    FServisciVar = 0;
    FAyarYapiliyor = 0;
    FArizaVar = 0;
    FMemingerKoptu  = 0;
    FAltMemingerKoptu  = 0;
    FLycraKoptu  = 0;
    FStartStop = 0;
    FIgneKoptu   = 0;
    FKapakAcildi = 0;
    FKenarStop = 0;
    FCozguKoptu = 0;
    FAtkiKoptu = 0;
    FLeventSonu = 0;
    OldPlanlamaId = null;
    FImalataAraVerildi = 0;
    FTur = 0;
  end
  else
  begin
      Select Ariza, ServisciVar, AyarYapiliyor, ArizaVar,  iif(New.OrmeMakinaId is null, DokumaPlanlamaId, OrmePlanlamaId), MemingerKoptu, AltMemingerKoptu, StartStop, IgneKoptu, LycraKoptu, KapakAcildi, KenarStop, CozguKoptu, AtkiKoptu, LeventSonu, ImalataAraVerildi, Tur
      from OrmeMakinaDurumu where OrmeMakinaDurumu.OrmeMakinaDurumuID = :FOrmeMakinaDurumuId --- and OrmeMakinaDurumu.Tarih <= new.Tarih Furkan iptal 19.04.2021
      into :FAriza, :FServisciVar, :FAyarYapiliyor, :FArizaVar, :OldPlanlamaId, :FMemingerKoptu, :FAltMemingerKoptu, :FStartStop, :FIgneKoptu, :FLycraKoptu, :FKapakAcildi, :FKenarStop, :FCozguKoptu, :FAtkiKoptu, :FLeventSonu, :FImalataAraVerildi, :FTur;
   end

  if (FAriza is null) then FAriza = 0;
  if (FServisciVar is null) then FServisciVar = 0;
  if (FAyarYapiliyor is null) then FAyarYapiliyor = 0;
  if (FArizaVar is null) then FArizaVar = 0;
  if (FMemingerKoptu is null) then FMemingerKoptu = 0;
  if (FAltMemingerKoptu is null) then FAltMemingerKoptu = 0;
  if (FLycraKoptu is null) then FLycraKoptu = 0;
  if (FStartStop is null) then FStartStop = 0;
  if (FIgneKoptu is null) then FIgneKoptu = 0;
  if (FKapakAcildi is null) then FKapakAcildi = 0;
  if (FKenarStop is null) then FKenarStop = 0;
  if (FCozguKoptu is null) then FCozguKoptu = 0;
  if (FAtkiKoptu is null) then FAtkiKoptu = 0;
  if (FLeventSonu is null) then FLeventSonu = 0;
  if (FImalataAraVerildi is null) then FImalataAraVerildi = 0;
  if (FTur is null) then FTur = 0;


  if (new.Ariza is null) then
  begin
    if (FAriza = 0 or FAriza = 2) then New.Ariza = 0;
    else if (FAriza = 1 or FAriza = 3) then New.Ariza = 3;
  end

  if (new.AyarYapiliyor is null and FAyarYapiliyor = 1) then
  begin
    exception AyarYapiliyor;
  end

  if (new.ImalataAraVerildi is null) then
  begin
    if (FImalataAraVerildi = 0 or FImalataAraVerildi = 2) then New.ImalataAraVerildi = 0;
    else if (FImalataAraVerildi = 1 or FImalataAraVerildi = 3) then New.ImalataAraVerildi = 3;
  end


  if (new.ServisciGeldi is null) then new.ServisciGeldi = 0;
  if (new.ServisciVar is null) then new.ServisciVar = FServisciVar;
  if (new.AyarYapiliyor is null) then new.AyarYapiliyor = FAyarYapiliyor;
  if (new.ArizaVar is null) then new.ArizaVar = FArizaVar;
  if (New.MemingerKoptu is Null) then new.MemingerKoptu = FMemingerKoptu;
  if (New.AltMemingerKoptu is Null) then new.AltMemingerKoptu = FAltMemingerKoptu;
  if (New.LycraKoptu is Null) then new.LycraKoptu = FLycraKoptu;
  if (New.StartStop is Null and FTur > 0 and new.Tur > 0) then new.StartStop = 0;
  else if (New.StartStop is Null) then new.StartStop = FStartStop;
  if (New.IgneKoptu is Null) then new.IgneKoptu = FIgneKoptu;
  if (New.KapakAcildi is Null) then new.KapakAcildi = FKapakAcildi;
  if (New.TopKesme is Null) then new.TopKesme = 0;
  if (New.TemizlikYapildi is Null) then new.TemizlikYapildi = 0;
  if (New.KenarStop is null) then New.KenarStop = FKenarStop;
  if (New.CozguKoptu is null) then New.CozguKoptu = FCozguKoptu;
  if (New.AtkiKoptu is null) then New.AtkiKoptu = FAtkiKoptu;
  if (New.LeventSonu is null) then New.LeventSonu = FLeventSonu;


  if (New.OrmeMakinaId is not null and New.OrmePlanlamaId is null) then New.OrmePlanlamaId = OldPlanlamaId;
  if (New.DokumaMakinaId is not null and New.DokumaPlanlamaId is null) then New.DokumaPlanlamaId = OldPlanlamaId;

  if (New.OrmeMakinaId is not null and New.OrmePlanlamaId is null and (new.TopKesme = 1 or new.TemizlikYapildi = 1)) then
  begin
    Select Max(OrmePlanlama.OrmePlanlamaID) from OrmePlanlama
      where OrmePlanlama.OrmeMakinaId = New.OrmeMakinaId and
            OrmePlanlama.Bitti = 0 and
            OrmePlanlama.Sira = (Select Min(sira) from OrmePlanlama where OrmePlanlama.Bitti = 0 and OrmePlanlama.OrmeMakinaId = New.OrmeMakinaId)
    into new.OrmePlanlamaID;
  end

  if (New.OrmeMakinaId is not null) then
  begin
    Select OrmeVardiyaPersonel.PersonelId from PKartiIcinSonVardiyaIdGetir(new.OrmeMakinaId, new.DokumaMakinaId, 0)
      left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = PKartiIcinSonVardiyaIdGetir.VardiyaId
      where OrmeVardiyaPersonel.OrmeMakinaId = New.OrmeMakinaId
      into New.PersonelId;
  end
  else
  begin
    Select OrmeVardiyaPersonel.PersonelId from PKartiIcinSonVardiyaIdGetir(new.OrmeMakinaId, new.DokumaMakinaId, 0)
      left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = PKartiIcinSonVardiyaIdGetir.VardiyaId
      where OrmeVardiyaPersonel.DokumaMakinaId = New.DokumaMakinaId
      into New.PersonelId;
  end
/*
  Select OrmeVardiyaPersonel.PersonelId from OrmeSonVardiyaIdGetir(new.OrmeMakinaId)
    left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = OrmeSonVardiyaIdGetir.OrmeVardiyaId
    where OrmeVardiyaPersonel.OrmeMakinaId = New.OrmeMakinaId
    into New.PersonelId;
  */
  if (new.OrmeMakinaDurumuId is null) then new.OrmeMakinaDurumuId = gen_id(OrmeMakinaDurumu_id_gen, 1);
end
