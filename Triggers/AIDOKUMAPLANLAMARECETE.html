CREATE TRIGGER AIDOKUMAPLANLAMARECETE FOR DOKUMAPLANLAMARECETE
ACTIVE AFTER INSERT POSITION 0
as
declare variable ToplamKullanimOrani Double Precision;
declare variable MaxKullanimOrani Double Precision;
declare variable Lot varchar(30);
begin
  Select Sum(KullanimOrani), Max(KullanimOrani) from DokumaPlanlamaRecete where DokumaPlanlamaRecete.DokumaPlanlamaId = New.DokumaPlanlamaId into :ToplamKullanimOrani, :MaxKullanimOrani;
  if (ToplamKullanimOrani > 100) then exception KullanimOraniHatasi;

  /*Select Max(DokumaPlanlamaRecete.Lot) from DokumaPlanlamaRecete
    where DokumaPlanlamaRecete.DokumaPlanlamaId = New.DokumaPlanlamaId and DokumaPlanlamaRecete.KullanimOrani = :MaxKullanimOrani
    into :Lot;

  update DokumaPlanlama set Lot = :Lot where DokumaPlanlama.DokumaPlanlamaId = New.DokumaPlanlamaId;*/
  update DokumaPlanlama set Lot = (select Lot from DokumaPlanlamaLotAtkiLotuGetir(new.DokumaPlanlamaId)) where DokumaPlanlamaId = new.DokumaPlanlamaId;
end
