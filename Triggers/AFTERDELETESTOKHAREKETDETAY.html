CREATE TRIGGER AFTERDELETESTOKHAREKETDETAY FOR STOKHAREKETDETAY
ACTIVE AFTER DELETE POSITION 0
as
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim    integer;
declare variable shtKaydirmaCikis  integer;
declare variable shtKaydirmaGiris  integer;
declare variable shtReklamasyonGiris   integer;
declare variable shtReklamasyonCikis   integer;
declare variable StokHareketID         integer;
declare variable IplikStokNumarator    integer;
declare variable Kilo                  Double Precision;
declare variable BrutKilo              Double Precision;
declare variable StokHareketDetayMiktarID integer;
declare variable BolunduguStokHareketDetayId integer;
declare variable Ks integer;
declare variable EgemenFirmaId integer;
begin
  select EgemenFirmaId from sistem into :EgemenFirmaId;
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, shtKaydirmaCikis, shtKaydirmaGiris, shtReklamasyonGiris, shtReklamasyonCikis From OrmeStokHareketTipiGetir
    into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
         :shtFasonUretim, :ShtUretim, :shtKaydirmaCikis, :shtKaydirmaGiris, :shtReklamasyonGiris, :shtReklamasyonCikis;


  if (Old.HareketTipi = shtAlis or Old.HareketTipi = shtSatisIade or Old.HareketTipi = shtFasonAlis or Old.HareketTipi = shtFasonUretim or
      Old.HareketTipi = shtUretim or Old.HareketTipi = shtKaydirmaGiris or Old.HareketTipi = shtReklamasyonGiris ) then
  begin /*Girişler*/
    Select StokHareketDetayMiktarID from StokHareketDetayMiktar where StokHareketDetayMiktar.Barkod = Old.Barkod into :StokHareketDetayMiktarID;

    if (StokHareketDetayMiktarID is not Null) then
    begin
      Update StokHareketDetayMiktar set GirisKilo = GirisKilo - Old.Kilo, GirisBrutKilo = GirisBrutKilo - Old.BrutKilo, GirisMetre = GirisMetre - Old.Metre, GirisKapAdedi = GirisKapAdedi - Old.KapAdedi where StokHareketDetayMiktar.Barkod = Old.Barkod;
    end
  end
  else /*Çıkışlar*/
  begin
    Select StokHareketDetayMiktarID from StokHareketDetayMiktar where StokHareketDetayMiktar.Barkod = Old.Barkod into :StokHareketDetayMiktarID;

    if (StokHareketDetayMiktarID is not Null) then
    begin
      Update StokHareketDetayMiktar set CikisKilo = CikisKilo - Old.Kilo, CikisBrutKilo = CikisBrutKilo - Old.BrutKilo, CikisMetre = CikisMetre - Old.Metre, CikisKapAdedi = CikisKapAdedi - Old.KapAdedi where StokHareketDetayMiktar.Barkod = Old.Barkod;
    end

    Update StokHareket Set KapAdedi = KapAdedi - old.KapAdedi, Miktar = Miktar - iif(MiktarBirim = 0, Old.Kilo, Old.Metre), BrutMiktar = BrutMiktar - iif(MiktarBirim = 0, Old.BrutKilo, 0), Metre = Metre - Old.Metre, Kilo = Kilo - iif(MiktarBirim = 0, 0, Old.Kilo), NeredenUpdate = 3 where StokHareket.StokHareketId = Old.StokHareketId;
  end
  if (EgemenFirmaId = 151) then /*Lico Teks*/
  begin
    if (Old.HareketTipi = shtFasonUretimeCikis) then
    begin
      select count(*) FROM StokHareketDetay where  StokHareketDetay.barkod = old.barkod into :ks;

      if (ks = 1) then
      begin
        select StokHareketDetay.BolunduguStokHareketDetayId FROM StokHareketDetay where  StokHareketDetay.barkod = old.barkod into :BolunduguStokHareketDetayId;
        if (BolunduguStokHareketDetayId is not null) then
        begin
          Update StokHareketDetay Set Kilo = Kilo + old.Kilo, BrutKilo = BrutKilo + Old.BrutKilo, Metre = Metre + Old.Metre, KapAdedi = KapAdedi + old.KapAdedi
             where StokHareketDetayID = :BolunduguStokHareketDetayId;
          Delete From StokHareketDetayMiktar where StokHareketDetayMiktar.Barkod = Old.Barkod;
          Delete From StokHareketDetay where StokHareketDetay.Barkod = Old.Barkod;

        end
      end
    end
  end
end
