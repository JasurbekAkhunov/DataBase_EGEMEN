CREATE TRIGGER AFTERUPDATEBOLUMPLANLAMA FOR BOLUMPLANLAMA
ACTIVE AFTER UPDATE POSITION 0
as
begin
  if (Old.MakinaPlanlamaGrupId is not null and New.MakinaPlanlamaGrupId is null) then delete from GrupPlanlama where GrupPlanlama.BolumPlanlamaId = new.BolumPlanlamaId;
  else
  if ( (Old.MakinaPlanlamaGrupId is null and New.MakinaPlanlamaGrupId is not null) or
       (Old.MakinaPlanlamaGrupId is not null and New.MakinaPlanlamaGrupId is not null and Old.MakinaPlanlamaGrupId <> New.MakinaPlanlamaGrupId)) then
  begin
    if (Old.MakinaPlanlamaGrupId is not null) then delete from GrupPlanlama where GrupPlanlama.BolumPlanlamaId = new.BolumPlanlamaId;

    Insert into GrupPlanlama(MakinaPlanlamaGrupId, BolumPlanlamaId) values(new.MakinaPlanlamaGrupId, new.BolumPlanlamaId);
  end
end
