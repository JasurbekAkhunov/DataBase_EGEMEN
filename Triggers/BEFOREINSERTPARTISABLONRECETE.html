CREATE TRIGGER BEFOREINSERTPARTISABLONRECETE FOR PARTISABLONRECETE
ACTIVE BEFORE INSERT POSITION 0
as
declare variable StoktanDusuldu integer;
begin
  Select StoktanDusuldu from PartiSablon where PartiSablon.PartiSablonID = new.PartiSablonId into :StoktanDusuldu;

  if (StoktanDusuldu = 1) then exception StokDusulduDegisemez;

  if (New.Pat = 0) then
  begin
    Update PartiSablon Set BoyaToplami = BoyaToplami + New.Miktar where PartiSablon.PartiSablonID = new.PartiSablonId;
  end
  new.KullanilanMiktar = Yuvarla(new.KullanilanMiktar, 1000);

  new.PartiSablonReceteId = gen_id(PartiSablonRecete_Id_Gen, 1);
end
