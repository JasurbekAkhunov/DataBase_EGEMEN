CREATE TRIGGER BEFOREUPDATEHAM FOR HAM
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KayitSayisi integer;
declare variable Degisti     integer;
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstLycra        integer;
declare variable hstTelef        integer;
declare variable hstKirpinti     integer;
declare variable hstHamPamuk     integer;
declare variable hstCigit        integer;
declare variable ElyafTanitimdaSadeceAdZorunlu integer;
declare variable TelefTanitimdaSadeceAdZorunlu integer;
declare variable IplikTanitimdaSadeceAdZorunlu integer;
declare variable DokumaTanitimKumasCinsiZorunlu integer;
declare variable OrguTanitimKumasCinsiZorunlu integer;
declare variable EgemenFirmaId  integer;
declare variable FPus integer;
declare variable FFeine integer;
declare variable FMaliyetKatsayisi double precision;
begin
  Select ElyafTanitimdaSadeceAdZorunlu, TelefTanitimdaSadeceAdZorunlu, IplikTanitimdaSadeceAdZorunlu, DokumaTanitimKumasCinsiZorunlu, OrguTanitimKumasCinsiZorunlu, EgemenFirmaId from Sistem Where SistemID = 1
    into :ElyafTanitimdaSadeceAdZorunlu, :TelefTanitimdaSadeceAdZorunlu, :IplikTanitimdaSadeceAdZorunlu, :DokumaTanitimKumasCinsiZorunlu, :OrguTanitimKumasCinsiZorunlu, :EgemenFirmaId;

  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstLycra, hstTelef, hstKirpinti, hstHamPamuk, hstCigit from HamTipiGetir
    into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstLycra, :hstTelef, :hstKirpinti, :hstHamPamuk, :hstCigit;

  if (new.HamTipiId = hstIplik) then
  begin
    if (IplikTanitimdaSadeceAdZorunlu = 0 and (new.KalinlikId is null or new.IplikCinsiId is null)) then exception KalinlikIplikCinsiGirilmeli;
  end
  else if (new.HamTipiId = hstLycra) then
  begin
    if (new.KalinlikId is null or new.IplikCinsiId is null) then exception KalinlikIplikCinsiGirilmeli;
  end
  else if (new.HamTipiId = hstElyaf) then
  begin
    if (new.IplikCinsiId is null and ElyafTanitimdaSadeceAdZorunlu = 0) then exception IplikCinsiBosBirakilamaz;
  end
  else if (new.HamTipiId = hstTelef) then
  begin
    if (new.IplikCinsiId is null and TelefTanitimdaSadeceAdZorunlu = 0) then exception IplikCinsiBosBirakilamaz;
  end
  else if (new.HamTipiId = hstDokumaKumas) then
  begin
    if (new.KumasCinsiId is null and DokumaTanitimKumasCinsiZorunlu = 1) then exception KumasCinsiBosBirakilamaz;
  end
  else if (new.HamTipiId = hstKirpinti) then
  begin
    /* (new.IplikCinsiId is null and TelefTanitimdaSadeceAdZorunlu = 0) then exception IplikCinsiBosBirakilamaz;*/
  end
  else if (new.HamTipiId = hstCigit or new.HamTipiId = hstHamPamuk) then
  begin
    /* (new.IplikCinsiId is null and TelefTanitimdaSadeceAdZorunlu = 0) then exception IplikCinsiBosBirakilamaz;*/
  end
  else if (new.HamTipiId = hstOrmeKumas) then
  begin
    if (new.KumasCinsiId is null and OrguTanitimKumasCinsiZorunlu = 1) then exception KumasCinsiBosBirakilamaz;
  end
  else
  begin
    if (new.KumasCinsiId is null) then exception KumasCinsiBosBirakilamaz;
  end

  if (Old.HamTipiId <> New.HamTipiId) then exception HamTipiAdiDegistirilemez;

  Degisti = 0;
  if ( (Old.CozguIplikKalinlikId is null and New.CozguIplikKalinlikId is not null) or
       (Old.CozguIplikKalinlikId is Not null and New.CozguIplikKalinlikId is null) or
       (Old.CozguIplikKalinlikId <> New.CozguIplikKalinlikId)) then degisti = 1;

  if ( (Old.AtkiIplikKalinlikId is null and New.AtkiIplikKalinlikId is not null) or
       (Old.AtkiIplikKalinlikId is Not null and New.AtkiIplikKalinlikId is null) or
       (Old.AtkiIplikKalinlikId <> New.AtkiIplikKalinlikId)) then degisti = 1;

  if ( (Old.HarmanCinsiId is null and New.HarmanCinsiId is not null) or
       (Old.HarmanCinsiId is Not null and New.HarmanCinsiId is null) or
       (Old.HarmanCinsiId <> New.HarmanCinsiId)) then degisti = 1;

  if ( (Old.KalinlikId is null and New.KalinlikId is not null) or
       (Old.KalinlikId is Not null and New.KalinlikId is null) or
       (Old.KalinlikId <> New.KalinlikId)) then degisti = 1;

  if (Degisti = 0) then
  if ( (Old.IplikCinsiId is null and New.IplikCinsiId is not null) or
       (Old.IplikCinsiId is Not null and New.IplikCinsiId is null) or
       (Old.IplikCinsiId <> New.IplikCinsiId)) then degisti = 1;

  if (Degisti = 0) then
  if ( (Old.OzellikId is null and New.OzellikId is not null) or
       (Old.OzellikId is Not null and New.OzellikId is null) or
       (Old.OzellikId <> New.OzellikId)) then degisti = 1;

  if (Degisti = 0) then
  if ( (Old.LycraTipiId is null and New.LycraTipiId is not null) or
       (Old.LycraTipiId is Not null and New.LycraTipiId is null) or
       (Old.LycraTipiId <> New.LycraTipiId)) then degisti = 1;

  if (Degisti = 0) then
  if ( (Old.KumasCinsiId is null and New.KumasCinsiId is not null) or
       (Old.KumasCinsiId is Not null and New.KumasCinsiId is null) or
       (Old.KumasCinsiId <> New.KumasCinsiId)) then degisti = 1;

  if (Degisti = 0) then
  if ( (Old.HarmanCinsiId is null and New.HarmanCinsiId is not null) or
       (Old.HarmanCinsiId is Not null and New.HarmanCinsiId is null) or
       (Old.HarmanCinsiId <> New.HarmanCinsiId)) then degisti = 1;

  if (Degisti = 0) then
  if ( (Old.CozguIplikKalinlikId is null and New.CozguIplikKalinlikId is not null) or
       (Old.CozguIplikKalinlikId is Not null and New.CozguIplikKalinlikId is null) or
       (Old.CozguIplikKalinlikId <> New.CozguIplikKalinlikId)) then degisti = 1;

  if (Degisti = 0) then
  if ( (Old.AtkiIplikKalinlikId is null and New.AtkiIplikKalinlikId is not null) or
       (Old.AtkiIplikKalinlikId is Not null and New.AtkiIplikKalinlikId is null) or
       (Old.AtkiIplikKalinlikId <> New.AtkiIplikKalinlikId)) then degisti = 1;

  if (Degisti = 0) then
  if (IntKarsilastir(Old.ElyafSinifId, New.ElyafSinifId) = 0) then degisti = 1;

  if (Degisti = 0) then
  if (IntKarsilastir(Old.MenseiId, New.MenseiId) = 0) then degisti = 1;

  if (Degisti = 0) then
  if (IntKarsilastir(Old.CircirId, New.CircirId) = 0) then degisti = 1;

  if (Degisti = 1 and EgemenFirmaId <> 77 and EgemenFirmaId <> 53 and EgemenFirmaId <> 139) then /*Pak ipek, Emre Tekstil , EmnDenim*/
  begin
    Select Count(*) from HamStokDetay where HamStokDetay.HamId = New.HamId into :KayitSayisi;
    if (KayitSayisi > 0) then exception KullanilanHamIdleriDegisemez;
  end

  if (Old.Adi <> New.Adi and EgemenFirmaId <> 77 and EgemenFirmaId <> 139) then /*Pak ipek,EmnDenim*/
  begin
    select count(*) from HamStokDetay where HamStokDetay.HamId = new.HamId into :KayitSayisi;
    if (KayitSayisi > 0) then exception KullanmisHamAdiDegistirilemez;

    select count(*) from StokHareket where StokHareket.HamId = new.HamId into :KayitSayisi;
    if (KayitSayisi > 0) then exception KullanmisHamAdiDegistirilemez;

    select count(*) from IplikIrsaliyeDetay where IplikIrsaliyeDetay.HamId = new.HamId into :KayitSayisi;
    if (KayitSayisi > 0) then exception KullanmisHamAdiDegistirilemez;

    select count(*) from OrmePlanlamaRecete where OrmePlanlamaRecete.HamId = new.HamId into :KayitSayisi;
    if (KayitSayisi > 0) then exception KullanmisHamAdiDegistirilemez;
  end

  /*if ((new.HamtipiID = hstOrmeKumas) and  (strkarsilastir(old.HamGrup1, new.HamGrup1) = 0)) then
  begin
    delete from OrmeMaliyetKatsayi where OrmeMaliyetKatsayi.HamId = New.HamID;
    for select OrmeMaliyetGrupKatsayi.Pus, OrmeMaliyetGrupKatsayi.Feine, OrmeMaliyetGrupKatsayi.MaliyetKatsayisi  from OrmeMaliyetGrupKatsayi Where OrmeMaliyetGrupKatsayi.HamGrupId = New.HamGrup1
        into :FPus, :FFeine, :FMaliyetKatsayisi  do
    begin
      insert into OrmeMaliyetKatsayi(HamId, Pus, Feine, MaliyetKatsayisi) values (New.HamID, :FPus, :FFeine, :FMaliyetKatsayisi);
    end
  end*/

  if (new.MaximumFireOrani = 0) then select Sistem.MaximumFireOrani from sistem where sistemid = 1 into new.MaximumFireOrani;
  if (new.MinimumFireOrani = 0) then select Sistem.MinimumFireOrani from sistem where sistemid = 1 into new.MinimumFireOrani;

  if (Coalesce(new.AlisFiyati,0) <> Coalesce(old.AlisFiyati, 0) or StrKarsilastir(new.AlisFiyatiDovizId, old.AlisFiyatiDovizId) = 0) then
  begin
    new.AlisFiyatiGuncellemeTarihi = current_timestamp(0);
    new.AlisFiyatiGunceKullaniciId = new.KullaniciId;
  end

  new.DegisimTarihi = current_TimeStamp(0);
end
