Definition

CREATE TRIGGER BEFOREINSERTBOYASIPARISDETAY FOR BOYASIPARISDETAY
ACTIVE BEFORE INSERT POSITION 0
as
declare variable IslemTipiId integer;
declare variable HamTipiId integer;
declare variable Fason integer;
declare variable SiparisTarihi timestamp;
declare variable OrmeSiparisi integer;
declare variable hstIplik integer;
declare variable hstElyaf integer;
declare variable hstTelef integer;
declare variable BoyaSiparisRenkNoKullanimda integer;
declare variable OrmeHamId integer;
declare variable OrmeHamTipiId integer;
declare variable MasterSiparisTipiId integer;
declare variable FSiparisOdemeSekliId      integer;
declare variable FLabReceteKodu         varchar(20);
declare variable FLabReceteAdi          varchar(40);
declare variable RenkAdi                varchar(61);
declare variable SipTerminBosBirakilabilsin integer;
declare variable SiparisAcilamasin      integer;
declare variable SiparisFirmaId         varchar(20);
declare variable SistemDovizId          varchar(5);
declare variable FDovizId               varchar(5);
declare variable DovizKuru              double precision;
declare variable SipBrutMiktarHesaplamaSekli   integer;
declare variable BSatisSiparisleriNetGirilecek integer;
declare variable HerZamanSiparisProsesiKullan integer;
declare variable SiparisBrutuElleGirebilsin integer;
declare variable FirmaSiparisRiskiAsilamaz integer;
declare variable SiparisRiski double precision;
declare variable FSiparisRiskToplami double precision;
declare variable RiskDovizId varchar(5);
declare variable FSiparisRiski double precision;
declare variable FSiparisTarihi timestamp;
declare variable ASiraNo integer;
declare variable FBSipDetayID integer;
declare variable EgemenFirmaid integer;
begin
  select SistemDovizId, SipBrutMiktarHesaplamaSekli, BSatisSiparisleriNetGirilecek, HerZamanSiparisProsesiKullan, SiparisBrutuElleGirebilsin, FirmaSiparisRiskiAsilamaz, EgemenFirmaid from Sistem

    where SistemID = 1
    into :SistemDovizId, :SipBrutMiktarHesaplamaSekli, :BSatisSiparisleriNetGirilecek, :HerZamanSiparisProsesiKullan, :SiparisBrutuElleGirebilsin, :FirmaSiparisRiskiAsilamaz, :EgemenFirmaid;

  select FirmaId, IslemTipiId, Fason, SiparisTarihi, OrmeSiparisi, SiparisTipiId, OdemeSekliId, Komisyonlu, KomisyonOrani from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitId = new.BoyaSiparisKayitId
         into :SiparisFirmaId, :IslemTipiId, :Fason, :SiparisTarihi, :OrmeSiparisi, :MasterSiparisTipiId, :FSiparisOdemeSekliId, new.FiyatKomisyon, new.FiyatKomisyonOrani;

  select SiparisAcilamasin, SiparisRiski, RiskDovizId from Firma where Firma.FirmaID = :SiparisFirmaID into :SiparisAcilamasin, :SiparisRiski, :RiskDovizId;

  if (SiparisAcilamasin = 1) then exception KayitHatasi ' Bu Firmaya Sipariş Girişi Durdurulmuş..';

  if (FirmaSiparisRiskiAsilamaz = 1) then
  begin
    FSiparisRiskToplami = 0;
    for select sum(BirimFiyat * SiparisKilo), SiparisTarihi, DovizID from BoyaSiparis
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitID = BoyaSiparis.BoyaSiparisKayitID
    where BoyaSiparis.FirmaID = :SiparisFirmaId and BoyaSiparisDetay.Kapali = 0 and BoyaSiparisDetayID <> new.BoyaSiparisDetayID
    group by SiparisTarihi, DovizID
    into :FSiparisRiski, :FSiparisTarihi, :FDovizID do
    begin
      if (RiskDovizId <> FDovizID) then select DovizKuru from EnYakinDovizKuruGetir(:FDovizID, :RiskDovizId, :FSiparisTarihi) into :DovizKuru;
      else DovizKuru = 1;
      FSiparisRiskToplami  = FSiparisRiskToplami + (FSiparisRiski * DovizKuru);
    end

    if (RiskDovizId <> new.DovizID) then select DovizKuru from EnYakinDovizKuruGetir(new.DovizID, :RiskDovizId, :SiparisTarihi) into :DovizKuru;
    else DovizKuru = 1;

    if ((FSiparisRiskToplami + (new.BirimFiyat * new.SiparisKilo * DovizKuru)) > SiparisRiski) then exception KayitHatasi ' Firmanın Tanımlanan Sipariş Riski Aşılamaz...';
  end

  if (new.OrmeSiparisDetayId is not null) then
     select OdemeSekliId from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayId into :FSiparisOdemeSekliId;

  if (new.IslemTipiID is null) then new.IslemTipiID = :IslemTipiID; /* tamir ise kayıt olusurken islemtipini tamir veriyoruz, degişmemesi icin bu satır kosula baglı  */
  if (new.OdemeSekliId is null) then new.OdemeSekliId = FSiparisOdemeSekliId;

  if (FSiparisOdemeSekliId is not null and new.OdemeSekliId is not null and new.OdemeSekliId <> FSiparisOdemeSekliId) then exception OdemeSekliFarkliOlamaz;

  select hstElyaf, hstIplik, hstTelef from HamTipiGetir into :hstElyaf, :hstIplik, :hstTelef;

  select BoyaSiparisRenkNoKullanimda, SipTerminBosBirakilabilsin from sistem
    where sistemid = 1 into :BoyaSiparisRenkNoKullanimda, :SipTerminBosBirakilabilsin;

  if (OrmeSiparisi = 1 and new.OrmeSiparisDetayId is null) then exception KayitHatasi 'Örme Sipariş DetayId boş olamaz';

  if (SipTerminBosBirakilabilsin = 0 and new.TerminTarihi is null) then new.TerminTarihi = SiparisTarihi;

  if (new.IslemTipiId <> 3) then new.FaturaKesilecek = 1;
  else if (new.UcretliTamir = 1) then new.FaturaKesilecek = 1;
  else new.FaturaKesilecek = 0;

  if (new.FaturaEdilecekMiktarTipi is null) then
  if (Fason = 0) then
    select DefaultSatisPartiMiktarTipi from Sistem where sistemid = 1 into new.FaturaEdilecekMiktarTipi;
  else
    select DefaultFasonPartiMiktarTipi from Sistem where sistemid = 1 into new.FaturaEdilecekMiktarTipi;

  if (new.HamStokBirimId is null) then new.HamStokBirimId = 0;
  if (new.StoktanKullanilacakKilo is null) then new.StoktanKullanilacakKilo = 0;
  if (new.StoktanKullanilacakMetre is null) then new.StoktanKullanilacakMetre = 0;

  if (new.Kapali <> 0) then new.KapatmaTarihi = current_date;

  if (new.RevizeTerminTarihi is null) then new.SonTerminTarihi = new.TerminTarihi;
  else new.SonTerminTarihi = new.RevizeTerminTarihi;


  select Ham.HamTipiId from Ham where Ham.HamId = new.HamId into :HamTipiId;

  if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then new.SiparisTipiID = 0;
  else new.SiparisTipiID = 1;

  if (new.OrmeSiparisDetayID is not null) then
  begin
    select OrmeSiparisDetay.HamId, Ham.HamTipiId from OrmeSiparisDetay
      left join Ham on Ham.HamID = OrmeSiparisDetay.HamID
      where OrmeSiparisDetay.OrmeSiparisDetayId = new.OrmeSiparisDetayID
      into :OrmeHamId, :OrmeHamTipiId;

    if (HamTipiId = OrmeHamTipiId and new.HamId <> OrmeHamId) then exception KayitHatasi 'Orme Siparisi ile ürün uymuyor..';
    else if (new.SiparisTipiID = MasterSiparisTipiId and new.HamId <> OrmeHamId) then exception KayitHatasi 'Orme Siparisi ile ürün uymuyor..';
  end

  if (BoyaSiparisRenkNoKullanimda = 1 and new.OrmeSiparisDetayID is not null and new.BoyaSiparisRenkNo is null) then exception KayitHatasi ' Sipariş Renk No Boş Bırakılamaz..';

  if (new.BoyaSiparisRenkNo is not null) then
    select LabReceteId, LabReceteKodu, LabReceteAtisId, RenkId, RenkAdi, MusteriRenkKoduveAdi from BoyaSiparisRenk
      where BoyaSiparisRenk.BoyaSiparisKayitID = new.BoyaSiparisKayitID and BoyaSiparisRenk.BoyaSiparisRenkNO = new.BoyaSiparisRenkNo
      into new.LabReceteId, new.LabReceteKodu, new.LabReceteAtisId, new.RenkId, new.RenkAdi, new.MusteriRenkKoduveAdi;

  if (new.OrmeSiparisDetayID is not null and new.LabReceteId is not null and new.RenkId is null) then
  begin
    select LabRecete.LabReceteKodu, LabRecete.Adi from LabRecete where LabRecete.LabReceteId = new.LabReceteId into :FLabReceteKodu, :FLabReceteAdi;

    new.RenkId = null;
    select Renk.RenkId from Renk where Renk.LabReceteID = new.LabReceteID into new.RenkId;

    if (new.RenkID is null) then
    begin
      new.RenkId = gen_id(Renk_Id_Gen, 1);

      RenkAdi = FLabReceteAdi || '-' || FLabReceteKodu;

      insert into Renk(RenkId, LabReceteId, Adi) values(new.RenkId, new.LabReceteId, EgemenCopyStr(:RenkAdi, 1, 50));
    end
  end

  if (new.OdemeSekliId is not null) then
    select vadeGun from OdemeSekli where OdemeSekli.OdemeSekliID = new.OdemeSekliId into new.vadeGun;

  if (new.MaximumFireOrani >= 100) then exception KayitHatasi 'Fire Orani Hatalı';
  if (new.MinimumFireOrani < 0 or new.MinimumFireOrani >= 100) then exception KayitHatasi 'Fire Orani Hatalı';

  if (new.LabReceteId is not null) then
  begin
    new.ReceteGirisTarihi = current_timestamp(0);
    new.ReceteGirisKullaniciId = new.InsertKullaniciId;
  end

  if (new.DesenId is not null) then
  begin
    new.DesenGirisTarihi = current_timestamp(0);
    new.DesenGirisKullaniciId = new.InsertKullaniciId;
  end

  if (new.SiparisProsesID is not null) then
  begin
    select BirimMaliyet, DovizId from SiparisProses where SiparisProsesID = new.SiparisProsesID into new.ProsesBirimMaliyet, :FDovizId;

    if (SistemDovizID <> FDovizId) then
      select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :SistemDovizId, :SiparisTarihi) into :DovizKuru;
    else
      DovizKuru = 1;

    new.ProsesBirimMaliyet = new.ProsesBirimMaliyet * DovizKuru;
  end

  if (new.RenkDerinligiID is not null) then
  begin
    select BirimFiyat, DovizId from RenkDerinligi where RenkDerinligiId = new.RenkDerinligiId into new.RenkDerinligiBirimMaliyet, :FDovizId;

    if (SistemDovizID <> FDovizId) then
      select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :SistemDovizId, :SiparisTarihi) into :DovizKuru;
    else
      DovizKuru = 1;

    new.RenkDerinligiBirimMaliyet = new.RenkDerinligiBirimMaliyet * DovizKuru;
  end

  if ((new.ReferansIsEmriNo is not null) and (new.MusteriSiparisNo is null)) then new.MusteriSiparisNo = new.ReferansIsEmriNo;

  if ((new.ProsesID is null or HerZamanSiparisProsesiKullan = 1) and (new.SiparisProsesId is not null)) then
    select ProsesId from SiparisProses where SiparisProsesId = new.SiparisProsesId into new.ProsesId;


  if (new.SiparisBrutKilo is null or new.SiparisBrutKilo = 0 or SiparisBrutuElleGirebilsin = 0) then
  begin
    if (new.MaximumFireOrani is not null and BSatisSiparisleriNetGirilecek = 1) then
    begin
      if (SipBrutMiktarHesaplamaSekli = 0) then
      begin
        new.SiparisBrutKilo = new.SiparisKilo / ((100 - new.MaximumFireOrani) / 100);
        select Miktar from FormataGoreKiloYuvarla(new.SiparisBrutKilo, 'BOYA', 'SIPDETAYBRUTKILO') into new.SiparisBrutKilo;
      end
      else
      begin
        new.SiparisBrutKilo = new.SiparisKilo + new.SiparisKilo * new.MaximumFireOrani / 100;
        select Miktar from FormataGoreKiloYuvarla(new.SiparisBrutKilo, 'BOYA', 'SIPDETAYBRUTKILO') into new.SiparisBrutKilo;
      end
    end
    else new.SiparisBrutKilo = new.SiparisKilo;
  end

  if (new.SiparisBrutMetre is null or new.SiparisBrutMetre = 0 or SiparisBrutuElleGirebilsin = 0) then
  begin
    if (new.MaximumFireOrani is not null and Fason = 0 and BSatisSiparisleriNetGirilecek = 1) then
    begin
      if (SipBrutMiktarHesaplamaSekli = 0) then
      begin
        new.SiparisBrutMetre = new.SiparisMetre / ((100 - new.MaximumFireOrani) / 100);
        select Miktar from FormataGoreKiloYuvarla(new.SiparisBrutMetre, 'BOYA', 'BRUTMETRE') into new.SiparisBrutMetre;
      end
      else
      begin
        new.SiparisBrutMetre = new.SiparisMetre + new.SiparisMetre * new.MaximumFireOrani / 100;
        select Miktar from FormataGoreKiloYuvarla(new.SiparisBrutMetre, 'BOYA', 'BRUTMETRE') into new.SiparisBrutMetre;
      end
    end
    else new.SiparisBrutMetre = new.SiparisMetre;
  end

  if (new.KopyaSonuUpdateYok = 1 and SiparisBrutuElleGirebilsin = 1) then
  begin
    new.SiparisBrutKilo = 0;
    new.SiparisBrutMetre = 0;
  end

  if (new.SiraNo is not null and new.SiraNo > -1) then
  begin
    ASiraNo = new.SiraNo;
    for select BoyaSiparisDetayId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisKayitID = new.BoyaSiparisKayitID and BoyaSiparisDetay.SiraNo >= new.SiraNo order by SiraNo into :FBSipDetayId do
    begin
      ASiraNo = ASiraNo + 1;
      update BoyaSiparisDetay set SiraNo = :ASiraNo where BoyaSiparisDetayID = :FBSipDetayId;
    end
  end
  else
  begin
    select max(SiraNo) from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisKayitID = new.BoyaSiparisKayitID into :ASiraNo;

    if (ASiraNo is null) then ASiraNo = 0;
    else ASiraNo = ASiraNo + 1;

    new.SiraNo = ASiraNo;
  end

  if (new.BoyaSiparisDetayId is null) then new.BoyaSiparisDetayId = gen_id(BoyaSiparisDetay_Id_Gen, 1);
end
