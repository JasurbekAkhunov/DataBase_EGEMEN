CREATE TRIGGER AFTERINSERTPSKKONTROLHATA FOR PSKALITEKONTROLHATALARI
ACTIVE AFTER INSERT POSITION 0
as
declare variable ParcaKesme integer;
declare variable OtonomTopParcalaniyor Integer;
begin
  OtonomTopParcalaniyor = Coalesce(rdb$get_context('USER_TRANSACTION', 'OtonomTopParcalaniyor'), 0);

  if (OtonomTopParcalaniyor <> 1) then
  begin
    Select iif(Count(ParcaKesme) > 0, 1, 0) from PSKaliteKontrolHatalari
      Left Join HataTipi on HataTipi.HataTipiID = PSKaliteKontrolHatalari.HataTipiID
      where PSKaliteKontrolID = New.PSKaliteKontrolID and ParcaKesme = 1 into :ParcaKesme;

    Update PSKaliteKontrol Set ParcaYarim = :ParcaKesme where PSKaliteKontrolID = New.PSKaliteKontrolID;
  end
end
