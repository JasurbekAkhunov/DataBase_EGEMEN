CREATE TRIGGER BEFOREUPDATEORMERAFDETAY1 FOR ORMERAFDETAY
ACTIVE BEFORE UPDATE POSITION 1
as
declare variable HareketTipi                   integer;
declare variable HamIade                       integer;
declare variable UretimIslemTipiId             integer;
declare variable BoyaSiparisKayitId            integer;
declare variable OrmeSiparisDetayId            integer;
declare variable BoyaSiparisDetayId            integer;
declare variable BoyaSiparisBaskiId            integer;
declare variable OrmePlanlamaId                integer;
declare variable DokumaPlanlamaId              integer;
declare variable Sakat                         integer;
declare variable AyarTopu                      integer;
declare variable BoyaTalimatKayitId            integer;
declare variable BukumTalimatId                integer;
declare variable UretimPartiNo                 varchar(30);
declare variable CikisPartiNo                  varchar(30);
declare variable Lot                           varchar(150);
declare variable MelanjKodu                    varchar(20);
declare variable FlamKodu                      varchar(20);
declare variable Sertifika                     varchar(30);
declare variable UreticiFirmaId                varchar(20);
declare variable OrmeMakinaId                  varchar(10);
declare variable DokumaMakinaId                varchar(10);
declare variable OrmeciMakinaNo                varchar(10);
declare variable UreticiId                     integer;
declare variable Tamir                         integer;
declare variable EskiPartiYil                  integer;
declare variable EskiPartiNo                   integer;
declare variable FasonStok                     integer;
declare variable StokFirmaId                   varchar(20);
declare variable KaliteAciklamaTipiId          integer;
declare variable MamulProsesId                 integer;
declare variable BoyahaneFasonIade             integer;
declare variable DesenId                       varchar(20);
declare variable DesenVaryantId                integer;
declare variable StokGirisNo                   integer;
declare variable BirimMaliyet                  Double Precision;
declare variable BaskaGirisVar                 integer;
declare variable Brut                          integer;
declare variable MiktarBirim                   integer;
declare variable TamirTipiId                   integer;
declare variable DisTamirCozumSekliId          varchar(10);
declare variable FBirimFiyat                   Double Precision;
declare variable FStokHareketId                integer;
declare variable FBirimIskontoOrani1           Double Precision;
declare variable FBirimIskontoOrani2           Double Precision;
declare variable FBirimIskontoParasal          Double Precision;
declare variable FKdv                          Double Precision;
declare variable FTevkifat                     integer;
declare variable shtAlis                       integer;
declare variable shtSatisIade                  integer;
declare variable shtFasonUretim                integer;
declare variable shtUretim                     integer;
declare variable SHBirimFiyat                  Double Precision;
declare variable SHBirimIskontoOrani1          Double Precision;
declare variable SHBirimIskontoOrani2          Double Precision;
declare variable SHBirimIskontoParasal         Double Precision;
declare variable SHKdv                         Double Precision;
declare variable SHTevkifat                    integer;
declare variable RafGirisindeMaliyetiEsitle    integer;
declare variable ORafBirimMaliyetTakibiOlmasin integer;
declare variable SHOrmeRafId                   integer;
declare variable RafHesaplamaDevreDisi         integer;
declare variable CozguIsEmriKayitID            integer;
declare variable HasilIsEmriKayitID            integer;
declare variable IndigoIsEmriKayitID            integer;
declare variable LeventNo                      integer;
declare variable LeventSiraNo                  integer;
declare variable DepoPartiNo                   varchar(30);
declare variable TelAdedi                      Integer;
declare variable AtkiLotu                      varchar(30);
declare variable CozguLotu                     varchar(30);
declare variable LeventTipiId                  integer;
declare variable BarkodId                      varchar(30);
declare variable DokumaHareketi                EgeBoolean;
declare variable OrmeRafId                     integer;
declare variable OlcuId                        integer;
declare variable IplikUzamaDegeri              double Precision;
begin
  select RafGirisindeMaliyetiEsitle, ORafBirimMaliyetTakibiOlmasin, RafHesaplamaDevreDisi from Sistem where SistemId = 1 into :RafGirisindeMaliyetiEsitle, :ORafBirimMaliyetTakibiOlmasin, :RafHesaplamaDevreDisi;

  Select shtAlis, shtSatisIade, shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir into :shtAlis, :shtSatisIade, :shtFasonUretim, :shtUretim;

  Select HareketTipi, HamIade, UretimIslemTipiId, BoyaSiparisKayitId, OrmeSiparisDetayId, BoyaSiparisDetayId, OrmePlanlamaId, DokumaPlanlamaId, Sakat, AyarTopu, BoyaTalimatKayitId, BukumTalimatId, UretimPartiNo, CikisPartiNo,
         Lot, MelanjKodu, FlamKodu, UreticiFirmaId, OrmeMakinaId, DokumaMakinaId, OrmeciMakinaNo, UreticiId, Tamir, EskiPartiYil, EskiPartiNo, FasonStok, StokFirmaId,
         KaliteAciklamaTipiId, MamulProsesId, BoyahaneFasonIade, BoyaSiparisBaskiId, DesenId, DesenVaryantId, StokGirisNo, BirimMaliyet, Brut, MiktarBirim, TamirTipiId, DisTamirCozumSekliId,
         BirimFiyat, BirimIskontoOrani1, BirimIskontoOrani2, BirimIskontoParasal, Kdv, Tevkifat, CozguIsEmriKayitID, HasilIsEmriKayitID, LeventNo, DepoPartiNo, TelAdedi, AtkiLotu, CozguLotu, LeventTipiId,
         BarkodId, DokumaHareketi, OlcuId, Sertifika, IndigoIsEmriKayitID, LeventSiraNo, IplikUzamaDegeri from StokHareket
    where StokHareket.StokHareketID = New.StokHareketId
    into :HareketTipi, :HamIade, :UretimIslemTipiId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :OrmePlanlamaId, :DokumaPlanlamaId, :Sakat, :AyarTopu,  :BoyaTalimatKayitId, :BukumTalimatId, :UretimPartiNo, :CikisPartiNo,
         :Lot, :MelanjKodu, :FlamKodu, :UreticiFirmaId, :OrmeMakinaId, :DokumaMakinaId, :OrmeciMakinaNo, :UreticiId, :Tamir, :EskiPartiYil, :EskiPartiNo, :FasonStok, :StokFirmaId,
         :KaliteAciklamaTipiId, :MamulProsesId, :BoyahaneFasonIade, :BoyaSiparisBaskiId, :DesenId, :DesenVaryantId, :StokGirisNo, :BirimMaliyet, :Brut, :MiktarBirim, :TamirTipiId, :DisTamirCozumSekliId,
         :FBirimFiyat, :FBirimIskontoOrani1, :FBirimIskontoOrani2, :FBirimIskontoParasal, :FKdv, :FTevkifat, :CozguIsEmriKayitID, :HasilIsEmriKayitID, :LeventNo, :DepoPartiNo, :TelAdedi, :AtkiLotu, :CozguLotu, :LeventTipiId,
         :BarkodId, :DokumaHareketi, :OlcuId, :Sertifika, :IndigoIsEmriKayitID, :LeventSiraNo, :IplikUzamaDegeri;

  if (HamIade = 1) then
  begin
    OrmeRafId = Null;

    if (DokumaHareketi = 1 and BarkodId is not null) then
    begin
      select OrmeRaf.OrmeRafId, OrmeRaf.UreticiFirmaId, OrmeRaf.OrmePlanlamaId, OrmeRaf.DokumaPlanlamaId, OrmeRaf.BoyaTalimatKayitId, OrmeRaf.BukumTalimatId, OrmeRaf.UretimPartiNo, OrmeRaf.CikisPartiNo,
      OrmeRaf.BoyaSiparisDetayId, OrmeRaf.BoyaSiparisKayitId, OrmeRaf.OrmeMakinaId, OrmeRaf.DokumaMakinaId, OrmeRaf.OrmeSiparisDetayId
      from OrmeRaf where OrmeRaf.BarkodId = :BarkodId
      into :OrmeRafId, :UreticiFirmaId, :OrmePlanlamaId, :DokumaPlanlamaId, :BoyaTalimatKayitId, :BukumTalimatId, :UretimPartiNo, :CikisPartiNo,
      :BoyaSiparisDetayId, :BoyaSiparisKayitId, :OrmeMakinaId, :DokumaMakinaId, :OrmeSiparisDetayId;
    end

    if (OrmeRafId is null) then
    begin
      UreticiFirmaId = Null;
      OrmePlanlamaID = Null;
      DokumaPlanlamaID = Null;
      BoyaTalimatKayitId = Null;
      BukumTalimatId = Null;

      UretimPartiNo = Null;
      CikisPartiNo = Null;
      BoyaSiparisDetayID = Null;
      BoyaSiparisBaskiId = Null;

      if (UretimIslemTipiID <> 2) then
      begin
        OrmeMakinaID = Null;
        DokumaMakinaID = Null;
        OrmeciMakinaNo = Null;
        OrmeSiparisDetayID = Null;
        BoyaSiparisKayitID = Null;
      end
    end
  end

  if (ORafBirimMaliyetTakibiOlmasin = 0) then
  begin
    Select count(*) from OrmeRafDetay
      where OrmeRafDetay.OrmeRafID = Old.OrmeRafId and OrmeRafDetay.StokHareketID <> new.StokHareketID
      into :BaskaGirisVar;

/* bu havuza başka giriş yok ise havız fiyatı güncelle ondan sonra rafa kaydet */

    if (BaskaGirisVar = 0) then
    begin
      Update OrmeRaf Set BirimMaliyet = :BirimMaliyet where OrmeRaf.OrmeRafID = Old.OrmeRafId;
    end
    else
    begin
      if (RafGirisindeMaliyetiEsitle = 1) then
      begin
        if ((HareketTipi = shtAlis or HareketTipi = shtSatisIade or HareketTipi = shtFasonUretim or HareketTipi = shtUretim)) then
        begin
          for select OrmeRafDetay.StokHareketId, StokHareket.BirimFiyat, StokHareket.BirimIskontoOrani1, StokHareket.BirimIskontoOrani2, StokHareket.BirimIskontoParasal,
              StokHareket.Kdv, StokHareket.Tevkifat, StokHareket.OrmeRafId from OrmeRafDetay
            left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketId
            where OrmeRafDetay.OrmeRafId = Old.OrmeRafId and OrmeRafDetay.StokHareketId <> new.StokHareketId
            into :FStokHareketId, :SHBirimFiyat, :SHBirimIskontoOrani1, :SHBirimIskontoOrani2, :SHBirimIskontoParasal,
              :SHKdv, :SHTevkifat, :SHOrmeRafId do
          begin
            if (FBirimFiyat <> SHBirimFiyat or FBirimIskontoOrani1 <> SHBirimIskontoOrani1 or FBirimIskontoOrani2 <> SHBirimIskontoOrani2 or FBirimIskontoParasal <> SHBirimIskontoParasal
             or FKdv <> SHKdv or FTevkifat <> SHTevkifat) then
            begin
              update StokHareket set BirimFiyat = :FBirimFiyat, BirimIskontoOrani1 = :FBirimIskontoOrani1, BirimIskontoOrani2 = :FBirimIskontoOrani2,
                BirimIskontoParasal = :FBirimIskontoParasal, Kdv = :FKdv, Tevkifat = :FTevkifat, NeredenUpdate = 4 where StokHareketId = :FStokHareketId and OrmeFaturaID is Null;
              Execute Procedure HesaplanacakRafEkle(:SHOrmeRafId);
            end
          end
          Update OrmeRaf Set BirimMaliyet = :BirimMaliyet where OrmeRaf.OrmeRafID = Old.OrmeRafId;
        end
      end
    end
  end

  if (RafHesaplamaDevreDisi <> 1) then
  begin
    Select OrmeRafId from OrmeRafKaydet(new.RafId, :StokGirisNo, new.HamId, new.RenkId, :DesenId, :DesenVaryantId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :OrmePlanlamaId, :DokumaPlanlamaId, :Sakat, :AyarTopu, :BoyaTalimatKayitId,
                                  :UretimPartiNo, :CikisPartiNo, :Lot, :MelanjKodu, :FlamKodu, :UreticiFirmaId, :OrmeMakinaId, :DokumaMakinaId, :OrmeciMakinaNo, :UreticiId, :Tamir, :EskiPartiYil,
                                  :EskiPartiNo, new.KaliteNo, :FasonStok, :StokFirmaId, :KaliteAciklamaTipiId, :MamulProsesId, :BoyahaneFasonIade,
                                  new.KapAdedi, new.BrutMiktar, new.Miktar, :Brut, :MiktarBirim, :BirimMaliyet, :TamirTipiId, :DisTamirCozumSekliId, :CozguIsEmriKayitID, :HasilIsEmriKayitID, :LeventNo, New.BobinSayisi, :BukumTalimatId, :DepoPartiNo, :TelAdedi, :AtkiLotu,
                                  :LeventTipiId, :CozguLotu, :BarkodId, :OlcuId, new.Adet, :Sertifika, :IndigoIsEmriKayitID, :LeventSiraNo, :IplikUzamaDegeri)
           into new.OrmeRafId;
  end

  /*Update OrmeRaf Set GirisKapAdedi = GirisKapAdedi - Old.KapAdedi,
                     NetGirisMiktari = NetGirisMiktari - Old.Miktar,
                     BrutGirisMiktari = BrutGirisMiktari - Old.BrutMiktar
    where OrmeRaf.OrmeRafID = Old.OrmeRafId;*/
end
