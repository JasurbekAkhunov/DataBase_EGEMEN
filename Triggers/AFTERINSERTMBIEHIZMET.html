CREATE TRIGGER AFTERINSERTMBIEHIZMET FOR MAKINABAKIMIEHIZMET
ACTIVE AFTER INSERT POSITION 0
as
declare variable TahminiSaat double precision;
begin
  select sum(HizmetSuresi) from MakinaBakimIEHizmet where MakinaBakimIsEmriId = new.MakinaBakimIsEmriId into :TahminiSaat;
  update MakinaBakimIsEmri set TahminiSaat = :TahminiSaat where MakinaBakimIsEmriId = new.MakinaBakimIsEmriId;
end
