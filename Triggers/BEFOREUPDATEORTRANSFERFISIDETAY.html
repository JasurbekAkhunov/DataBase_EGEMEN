  Select TransferEdildi From OrmeRafTransferFisi Where OrmeRafTransferFisiId = new.OrmeRafTransferFisiId into :TransferEdildi;
  if (TransferEdildi = 1) then exception TrasferEdilmisDegistirilmez;

  select (NetGirisMiktari - NetCikisMiktari) from OrmeRaf where OrmeRaf.OrmeRafID = new.OrmeRafId into :KalanMiktar;

  if (Old.KaliteNo <> New.KaliteNo or Old.RafID <> New.RafId or (Old.OnayDurumu <> New.OnayDurumu and New.OnayDurumu = 1)) then
  begin
    Select HamId, RenkId, BoyaSiparisKayitId, OrmeSiparisDetayId, BoyaSiparisDetayId, OrmePlanlamaId, DokumaPlanlamaId, Sakat, AyarTopu, BoyaTalimatKayitId, BukumTalimatId, UretimPartiNo, CikisPartiNo,

           :OlcuId, :Adet, :Sertifika, :IndigoIsEmriKayitID, :LeventSiraNo, :IplikUzamaDegeri;

    Select OrmeRafId from OrmeRafKaydet(new.RafId, :StokGirisNo, :HamId, :RenkId, :DesenId, :DesenVaryantId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisBaskiId,

                                    new.KapAdedi, new.BrutKilo, new.Kilo, :Brut, 0, :BirimMaliyet, :TamirTipiId, :DisTamirCozumSekliId,

                                    :OlcuId, new.Adet, :Sertifika, :IndigoIsEmriKayitID, :LeventSiraNo, :IplikUzamaDegeri)
     into new.YeniOrmeRafId;

    /*Update OrmeRaf Set CikisKapAdedi = CikisKapAdedi - Old.KapAdedi,
                       NetCikisMiktari = NetCikisMiktari - Old.Kilo,
                       BrutCikisMiktari = BrutCikisMiktari - Old.BrutKilo
      where OrmeRaf.OrmeRafId = Old.YeniOrmeRafId;*/

  end
  else
  begin
    if (Old.Kilo <> New.Kilo or Old.BrutKilo <> New.BrutKilo or Old.KapAdedi <> New.KapAdedi) then
    begin
      /*Update OrmeRaf Set CikisKapAdedi = CikisKapAdedi - Old.KapAdedi + New.KapAdedi,
                         NetCikisMiktari = NetCikisMiktari - Old.Kilo + New.kilo,
                         BrutCikisMiktari = BrutCikisMiktari - Old.BrutKilo + New.Brutkilo
        where OrmeRaf.OrmeRafId = New.YeniOrmeRafId;*/
    end
  end

  if ((KalanMiktar + Old.Kilo) < new.Kilo) then exception RaftaYeterliKiloMetreYok;

  Select FisTarihi From OrmeRafTransferFisi where OrmeRafTransferFisi.OrmeRafTransferFisiId = New.OrmeRafTransferFisiId into :FisTarihi;
  select BirimMaliyet from RafFifoMaliyetGetir(New.OrmeRafId, New.Kilo, :FisTarihi, Null) into new.BirimMaliyet;

end
