CREATE TRIGGER AFTERUPDATEELYAFURETIM FOR ELYAFURETIM
ACTIVE AFTER UPDATE POSITION 0
as
declare variable KayitSayisi integer;
declare variable MaxUretimTarihi TimeStamp;
declare variable StokId integer;
declare variable StokDetayId integer;
begin
  if (Old.IplikIrsaliyeDetayID <> New.IplikIrsaliyeDetayID) then
  begin
    Select count(*) from ElyafUretim where ElyafUretim.IplikIrsaliyeDetayId = Old.IplikIrsaliyeDetayId into :KayitSayisi;

    if (kayitSayisi = 0) then Delete from IplikIrsaliyeDetay where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = Old.IplikIrsaliyeDetayId;
  end

  if (Old.KapNo <> New.KapNo) then
    Update IplikStok Set KapNo = new.KapNo where IplikStok.IplikStokID = New.IplikStokId;

end
