CREATE TRIGGER BEFOREINSERTMAKINASARJ FOR MAKINASARJ
ACTIVE BEFORE INSERT POSITION 0
as
declare variable MakinaSarjId integer;
begin
  For Select MakinaSarj.MakinaSarjID from MakinaSarj
      where MakinaSarj.BitisTarihi is null and MakinaSarj.MakinaID = New.MakinaID and MakinaSarj.PartiVar = 0
      into :MakinaSarjId do
  begin
    Update MakinaSarj Set BitisTarihi = Current_TimeStamp(0) where MakinaSarj.MakinaSarjID = :MakinaSarjID;
  end
end
