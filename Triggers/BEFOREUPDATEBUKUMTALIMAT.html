Definition

CREATE TRIGGER BEFOREUPDATEBUKUMTALIMAT FOR BUKUMTALIMAT
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable KayitSayisi    integer;
declare variable OldFirmaId     FirmaKoduSize;
declare variable OldMakinaId    varchar(10);
declare variable OldDokumaMakinaId    MakinaKoduSize;
declare variable NewDokumaMakinaId    MakinaKoduSize;
declare variable NewFirmaId     FirmaKoduSize;
declare variable NewMakinaId    varchar(10);
declare variable FinalKalinlik   double precision;
begin
  if (new.FirmaId = '') then New.FirmaId = null;
  if (new.OrmeMakinaId = '') then New.OrmeMakinaId = null;
  if (new.DokumaMakinaId = '') then New.DokumaMakinaId = null;
  OldFirmaId = Old.FirmaId;
  OldMakinaId = Old.OrmeMakinaId;
  OldDokumaMakinaId = Old.DokumaMakinaId;

  NewFirmaId = New.FirmaId;
  NewMakinaId = New.OrmeMakinaId;
  NewDokumaMakinaId = New.DokumaMakinaId;

  if (OldFirmaId is null) then OldFirmaID = '';
  if (OldMakinaId is null) then OldMakinaId = '';
  if (OldDokumaMakinaId is null) then OldDokumaMakinaId = '';
  if (NewFirmaId is null) then NewFirmaID = '';
  if (NewMakinaId is null) then NewMakinaId = '';
  if (NewDokumaMakinaId is null) then NewDokumaMakinaId = '';

  if (new.Bitti = 1) then
  begin
    if (OldFirmaId <> NewFirmaId or OldMakinaId <> NewMakinaId or OldDokumaMakinaId <> NewDokumaMakinaId) then exception BitirilmisPlanlamaDegisemez;
    if (Old.BoyaSiparisKayitId <> New.BoyaSiparisKayitId) then exception BitirilmisPlanlamaDegisemez;
    if (Old.TalimatMiktari <> New.TalimatMiktari) then exception BitirilmisPlanlamaDegisemez;
    if (Old.GerceklesenMiktar <> New.GerceklesenMiktar) then exception BitirilmisPlanlamaDegisemez;
  end
  else
  begin
    Select count(*) from StokHareket where StokHareket.BukumTalimatId = new.BukumTalimatId into :kayitSayisi;
    if (KayitSayisi > 0) then
    begin
      if (OldFirmaId <> NewFirmaId or OldMakinaId <> NewMakinaId or OldDokumaMakinaId <> NewDokumaMakinaId) then exception TalimatHareketGormusDegisemez;
      if (Old.BoyaSiparisKayitId <> New.BoyaSiparisKayitId) then exception TalimatHareketGormusDegisemez;
    end
  end

  if (new.IgHizi is not Null) then
  if (new.IgHizi <> 0) then
  if (new.TurSayisi is not Null) then
  if (new.TurSayisi <> 0) then
  begin
    new.CekimHizi = new.IgHizi / new.TurSayisi;
  end
  else
    new.CekimHizi = 0;

  select Sonuc from FinalKalinlikHesapla(new.BukumTalimatId, new.CekimHizi, new.Cekim, new.IplikKalinlikBirim) into :FinalKalinlik;

  if (FinalKalinlik is not Null) then
  if (FinalKalinlik > 0) then
  if (new.Randiman is not Null) then
  if (new.Randiman > 0) then
  if (new.MakinaDevir is not Null) then
  if (new.MakinaDevir > 0) then
  if (new.Pozisyon is not Null) then
  if (new.Pozisyon > 0) then
  begin
    new.MiktarGun = new.Randiman * new.MakinaDevir * new.Pozisyon * :FinalKalinlik;
  end
  else
    new.MiktarGun = 0;

  if (New.FirmaId is null and New.OrmeMakinaId is null and New.DokumaTalimati = 0) then exception PlanlamaFirmaMakinaIdHatasi1;
  if (New.FirmaId is not null and New.OrmeMakinaId is not null and New.DokumaTalimati = 0) then exception PlanlamaFirmaMakinaIdHatasi1;
  if (New.FirmaId is null and New.DokumaMakinaId is null and New.DokumaTalimati = 1) then exception PlanlamaFirmaMakinaIdHatasi1;
  if (New.FirmaId is not null and New.DokumaMakinaId is not null and New.DokumaTalimati = 1) then exception PlanlamaFirmaMakinaIdHatasi1;

  if (new.GerceklesenMiktar > 0) then
  if (OldFirmaId <> NewFirmaId or OldMakinaId <> NewMakinaId or OldDokumaMakinaId <> NewDokumaMakinaId) then exception UretimVarPlanlamaYerDegisemez;

  new.DegisimTarihi = current_TimeStamp(0);
end
