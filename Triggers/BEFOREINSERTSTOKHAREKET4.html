Definition

CREATE TRIGGER BEFOREINSERTSTOKHAREKET4 FOR STOKHAREKET
ACTIVE BEFORE INSERT POSITION 4
as
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable uitBukum       integer;
declare variable uitOrme        integer;
declare variable uitBoya        integer;
declare variable uitDokuma      integer;
declare variable uitCozgu    integer;
declare variable uitHasil    integer;
declare variable uitIndigo   integer;
declare variable Beklemede      integer;
declare variable FHamId         integer;
declare variable TalimatUretimeCikisM   Double Precision;
declare variable TalimatUretimM         Double Precision;
declare variable TalimatMiktari         Double Precision;
declare variable OrmeUretimCikisMKontrol      Double Precision;
declare variable OrmeUretimTalimatMKontrol    Double Precision;
declare variable KS       integer;
declare variable OTalReceteCikisMiktardaLotYok SmallInt;
declare variable BTalReceteCikisMiktardaLotYok SmallInt;
declare variable FAsilabilirMiktar Double Precision;
declare variable FLabReceteKoduTipi integer;
declare variable  ReceteKS integer;
declare variable  OrmeSiparisDetayID integer;
declare variable  BukumUretimCikisMKontrol Double Precision;
begin
/* bu trigger üretim Planlanmış ise ormeplanlamaid yi alıyoruz update ediyoruz..*/

  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim;

  Select uitBukum, uitOrme, uitBoya, uitDokuma, uitCozgu, uitHasil, uitIndigo from UretimIslemTipiGetir into :UitBukum, :uitOrme, :uitBoya, :uitDokuma, :uitCozgu, :uitHasil, :uitIndigo;

  select OrmeUretimCikisMKontrol, OrmeUretimTalimatMKontrol, OTalReceteCikisMiktardaLotYok, BTalReceteCikisMiktardaLotYok, LabReceteKoduTipi, BukumUretimCikisMKontrol from Sistem where SistemId = 1 into :OrmeUretimCikisMKontrol, :OrmeUretimTalimatMKontrol, :OTalReceteCikisMiktardaLotYok, :BTalReceteCikisMiktardaLotYok, :FLabReceteKoduTipi, :BukumUretimCikisMKontrol;

  if (new.UretimIslemTipiId = uitOrme and new.OrmePlanlamaId is not null) then
  begin
    if ((new.HareketTipi = shtFasonUretimeCikis or New.HareketTipi = shtUretimeCikis) and (new.HamIade = 0)) then
    begin
      Select Beklemede from OrmePlanlama where OrmePlanlama.OrmePlanlamaId = new.OrmePlanlamaId into :Beklemede;
      if (Beklemede = 1) then exception PlanlamaBeklemeyeAlinmis;

      if (OTalReceteCikisMiktardaLotYok = 1) then
      begin
        update OrmePlanlamaRecete Set CikisMiktari = CikisMiktari + New.Miktar
          where OrmePlanlamaRecete.OrmePlanlamaId = new.OrmePlanlamaId and OrmePlanlamaRecete.HamId = new.HamId and
                IntKarsilastir(OrmePlanlamaRecete.RenkId, New.RenkID) = 1;
      end
      else
      begin
        update OrmePlanlamaRecete Set CikisMiktari = CikisMiktari + New.Miktar
          where OrmePlanlamaRecete.OrmePlanlamaId = new.OrmePlanlamaId and OrmePlanlamaRecete.HamId = new.HamId and
                IntKarsilastir(OrmePlanlamaRecete.RenkId, New.RenkID) = 1 and StrKarsilastir(OrmePlanlamaRecete.Lot, New.Lot) = 1;
      end
    end
    else if ((new.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and (new.HamIade = 1)) then
    begin
      if (OTalReceteCikisMiktardaLotYok = 1) then
        update OrmePlanlamaRecete Set IadeMiktari = IadeMiktari + New.Miktar
          where OrmePlanlamaRecete.OrmePlanlamaId = new.OrmePlanlamaId and OrmePlanlamaRecete.HamId = new.HamId and
                IntKarsilastir(OrmePlanlamaRecete.RenkId, New.RenkID) = 1;
      else
        update OrmePlanlamaRecete Set IadeMiktari = IadeMiktari + New.Miktar
          where OrmePlanlamaRecete.OrmePlanlamaId = new.OrmePlanlamaId and OrmePlanlamaRecete.HamId = new.HamId and
                IntKarsilastir(OrmePlanlamaRecete.RenkId, New.RenkID) = 1 and StrKarsilastir(OrmePlanlamaRecete.Lot, New.Lot) = 1;
    end
    if ((OrmeUretimCikisMKontrol <> -1 or OrmeUretimTalimatMKontrol <> -1) and (new.HareketTipi = shtFasonUretim or new.HareketTipi = shtUretim)) then
    begin
      select sum(StokHareket.Miktar) from stokhareket
            where (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis) and
            stokhareket.OrmePlanlamaId = new.OrmePlanlamaId
            into :TalimatUretimeCikisM;

      select sum(StokHareket.Miktar) from stokhareket
            where (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and
            stokhareket.OrmePlanlamaId = new.OrmePlanlamaId
            into :TalimatUretimM;

      select sum(OrmePlanlama.PlanlananMiktar) from OrmePlanlama
            where OrmePlanlama.OrmePlanlamaId = new.OrmePlanlamaId
            into :TalimatMiktari;

      if (TalimatUretimeCikisM is Null) then TalimatUretimeCikisM = 0;
      if (TalimatUretimM is Null) then TalimatUretimM = 0;
      if (TalimatMiktari is Null) then TalimatMiktari = 0;

      if (FLabReceteKoduTipi = 52) then /*NazÖrme*/
        Select Coalesce(Min(AsilabilirMiktar),0) AsilabilirMiktar from NazOrmeSatinAlmaFazlaMiktarlar where (:TalimatUretimM + new.Miktar) Between AltMiktar and UstMiktar into :FAsilabilirMiktar;

      if (OrmeUretimTalimatMKontrol > -1) then
      begin
        if ((FLabReceteKoduTipi <> 52 and (TalimatUretimM + new.Miktar) > (TalimatMiktari + (TalimatMiktari / 100) * OrmeUretimTalimatMKontrol)) or
        (FLabReceteKoduTipi = 52 and (TalimatUretimM + new.Miktar) > (TalimatMiktari + FAsilabilirMiktar))) then
          exception KayitHatasi 'Üretim miktarı talimat miktarını aştı... TalimatUretimM:'||TalimatUretimM||' TalimatMiktari:'||TalimatMiktari||' FAsilabilirMiktar:'||coalesce(FAsilabilirMiktar,'null');
      end

      if (OrmeUretimCikisMKontrol > -1) then
      begin
        if ((FLabReceteKoduTipi <> 52 and (TalimatUretimM + new.Miktar) > TalimatUretimeCikisM + ((TalimatUretimeCikisM / 100) * OrmeUretimCikisMKontrol)) or
        (FLabReceteKoduTipi = 52 and (TalimatUretimM + new.Miktar) > TalimatUretimeCikisM + FAsilabilirMiktar)) then
          exception KayitHatasi 'Üretim miktarı talimatın çıkış miktarını aştı...';
      end
    end

  end
  if (new.UretimIslemTipiId = uitBukum and new.BukumTalimatId is not null) then
  begin
    /*if ((new.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and new.HamIade = 0) then
      update BukumTalimat Set GerceklesenMiktar = GerceklesenMiktar + New.Miktar where BukumTalimat.BukumTalimatId = new.BukumTalimatId;
    else eb-1118 e istinaden kaldırıldı */


    if ((new.HareketTipi = shtFasonUretimeCikis or New.HareketTipi = shtUretimeCikis) and (new.HamIade = 0)) then
    begin
      if (BTalReceteCikisMiktardaLotYok = 1) then
      begin
        Select count(*) from BukumTalimatRecete where BukumTalimatRecete.BukumTalimatId = new.BukumTalimatID and BukumTalimatRecete.HamId = new.HamId  and IntKarsilastir(BukumTalimatRecete.RenkId, New.RenkID) = 1 into :ReceteKS;
        if (ReceteKs = 0) then ReceteKs  = 1;
        update BukumTalimatRecete Set CikisMiktari = CikisMiktari + (New.Miktar / :ReceteKs)  where BukumTalimatRecete.BukumTalimatId = new.BukumTalimatId and BukumTalimatRecete.HamId = new.HamId and
                IntKarsilastir(BukumTalimatRecete.RenkId, New.RenkID) = 1;
      end
      else
      begin
        Select count(*) from BukumTalimatRecete where BukumTalimatRecete.BukumTalimatId = new.BukumTalimatID and BukumTalimatRecete.HamId = new.HamId
                and IntKarsilastir(BukumTalimatRecete.RenkId, New.RenkID) = 1 and StrKarsilastir(BukumTalimatRecete.Lot, New.Lot) = 1 into :ReceteKS;
        if (ReceteKs = 0) then ReceteKs  = 1;

        update BukumTalimatRecete Set CikisMiktari = CikisMiktari + (New.Miktar / :ReceteKs)  where BukumTalimatRecete.BukumTalimatId = new.BukumTalimatId and BukumTalimatRecete.HamId = new.HamId and
                IntKarsilastir(BukumTalimatRecete.RenkId, New.RenkID) = 1 and StrKarsilastir(BukumTalimatRecete.Lot, New.Lot) = 1;
      end
    end
    else if ((new.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and (new.HamIade = 1)) then
    begin
      if (BTalReceteCikisMiktardaLotYok = 1) then
      begin
        Select count(*) from BukumTalimatRecete where BukumTalimatRecete.BukumTalimatId = new.BukumTalimatID and BukumTalimatRecete.HamId = new.HamId  and IntKarsilastir(BukumTalimatRecete.RenkId, New.RenkID) = 1 into :ReceteKS;
        if (ReceteKs = 0) then ReceteKs  = 1;

        update BukumTalimatRecete Set IadeMiktari = IadeMiktari + (New.Miktar / :ReceteKs)  where BukumTalimatRecete.BukumTalimatId = new.BukumTalimatId and BukumTalimatRecete.HamId = new.HamId and
                IntKarsilastir(BukumTalimatRecete.RenkId, New.RenkID) = 1;
      end
      else
      begin
         Select count(*) from BukumTalimatRecete where BukumTalimatRecete.BukumTalimatId = new.BukumTalimatID and BukumTalimatRecete.HamId = new.HamId and
                IntKarsilastir(BukumTalimatRecete.RenkId, New.RenkID) = 1 and StrKarsilastir(BukumTalimatRecete.Lot, New.Lot) = 1 into :ReceteKS;
        if (ReceteKs = 0) then ReceteKs  = 1;

        update BukumTalimatRecete Set IadeMiktari = IadeMiktari + (New.Miktar / :ReceteKs)  where BukumTalimatRecete.BukumTalimatId = new.BukumTalimatId and BukumTalimatRecete.HamId = new.HamId and
                IntKarsilastir(BukumTalimatRecete.RenkId, New.RenkID) = 1 and StrKarsilastir(BukumTalimatRecete.Lot, New.Lot) = 1;
      end

  end

    if ((BukumUretimCikisMKontrol <> -1) and (new.HareketTipi = shtFasonUretim or new.HareketTipi = shtUretim)) then
    begin
      select sum(StokHareket.Miktar) from stokhareket
            where (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis) and
            stokhareket.BukumTalimatId = new.BukumTalimatId
            into :TalimatUretimeCikisM;

      select sum(StokHareket.Miktar) from stokhareket
            where (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and
            stokhareket.BukumTalimatId = new.BukumTalimatId
            into :TalimatUretimM;

      select sum(BukumTalimat.TalimatMiktari) from BukumTalimat
            where BukumTalimat.BukumTalimatId = new.BukumTalimatId
            into :TalimatMiktari;

      if (TalimatUretimeCikisM is Null) then TalimatUretimeCikisM = 0;
      if (TalimatUretimM is Null) then TalimatUretimM = 0;
      if (TalimatMiktari is Null) then TalimatMiktari = 0;

      if (BukumUretimCikisMKontrol > -1) then
      begin
        if ((TalimatUretimM + new.Miktar) > TalimatUretimeCikisM + ((TalimatUretimeCikisM / 100) * BukumUretimCikisMKontrol)) then
          exception KayitHatasi 'Üretim miktarı talimatın çıkış miktarını aştı...';
      end
    end
  end

  if (new.UretimIslemTipiId = uitDokuma and new.DokumaPlanlamaId is not null) then
  begin
    if ((new.HareketTipi = shtFasonUretimeCikis or New.HareketTipi = shtUretimeCikis) and (new.HamIade = 0)) then
    begin
      Select Beklemede from DokumaPlanlama where DokumaPlanlama.DokumaPlanlamaId = new.DokumaPlanlamaId into :Beklemede;
      /*if (Beklemede = 1) then exception PlanlamaBeklemeyeAlinmis; Otomatik çıkıştan sonra Fahri Bey kapattırdı*/

      update DokumaPlanlamaRecete Set CikisMiktari = CikisMiktari + New.Miktar
        where DokumaPlanlamaRecete.DokumaPlanlamaId = new.DokumaPlanlamaId and DokumaPlanlamaRecete.HamId = new.HamId and
              IntKarsilastir(DokumaPlanlamaRecete.RenkId, New.RenkID) = 1 and StrKarsilastir(DokumaPlanlamaRecete.Lot, New.Lot) = 1;
    end
    else if ((new.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and (new.HamIade = 1)) then
      update DokumaPlanlamaRecete Set IadeMiktari = IadeMiktari + New.Miktar
        where DokumaPlanlamaRecete.DokumaPlanlamaId = new.DokumaPlanlamaId and DokumaPlanlamaRecete.HamId = new.HamId and
              IntKarsilastir(DokumaPlanlamaRecete.RenkId, New.RenkID) = 1 and StrKarsilastir(DokumaPlanlamaRecete.Lot, New.Lot) = 1;

    if ((OrmeUretimCikisMKontrol <> -1 or OrmeUretimTalimatMKontrol <> -1) and (new.HareketTipi = shtFasonUretim or new.HareketTipi = shtUretim)) then
    begin
      select sum(StokHareket.Miktar) from stokhareket
            where (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis) and
            stokhareket.DokumaPlanlamaId = new.DokumaPlanlamaId
            into :TalimatUretimeCikisM;

      select sum(StokHareket.Miktar) from stokhareket
            where (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and
            stokhareket.DokumaPlanlamaId = new.DokumaPlanlamaId
            into :TalimatUretimM;

      select sum(DokumaPlanlama.PlanlananMiktar) from DokumaPlanlama
            where DokumaPlanlama.DokumaPlanlamaId = new.DokumaPlanlamaId
            into :TalimatMiktari;

      if (TalimatUretimeCikisM is Null) then TalimatUretimeCikisM = 0;
      if (TalimatUretimM is Null) then TalimatUretimM = 0;
      if (TalimatMiktari is Null) then TalimatMiktari = 0;

      if (OrmeUretimTalimatMKontrol > -1) then
      begin
        if ((TalimatUretimM + new.Miktar) > TalimatMiktari + ((TalimatMiktari / 100) * OrmeUretimTalimatMKontrol)) then exception KayitHatasi 'Üretim miktarı talimat miktarını aştı...';
      end

      if (OrmeUretimCikisMKontrol > -1) then
      begin
        if ((TalimatUretimM + new.Miktar) > TalimatUretimeCikisM + ((TalimatUretimeCikisM / 100) * OrmeUretimCikisMKontrol)) then exception KayitHatasi 'Üretim miktarı talimatın çıkış miktarını aştı...';
      end
    end

  end

  if (New.BoyaSiparisDetayID is not null) then
  begin
    if (new.UretimIslemTipiId = uitBoya and (New.HareketTipi = shtFasonUretimeCikis or New.HareketTipi = shtUretimeCikis)) then
      Update BoyaSiparisDetay set ToplamBoyayaCikis = ToplamBoyayaCikis + New.Miktar, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayID = New.BoyaSiparisDetayId;
    else if (New.UretimIslemTipiId = uitBoya and (New.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and New.HamIade = 1) then
       Update BoyaSiparisDetay set ToplamBoyaIade = ToplamBoyaIade + new.Miktar, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayID = New.BoyaSiparisDetayId;
    else if (New.UretimIslemTipiId = uitBoya and (New.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and New.HamIade = 0) then
       Update BoyaSiparisDetay set ToplamBoyadanGelenBrutKilo = ToplamBoyadanGelenBrutKilo + new.BrutMiktar, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayID = New.BoyaSiparisDetayId;
  end
  if (new.BoyaTalimatKayitID is not null) then
  begin
    if (new.UretimIslemTipiId = uitBoya and (New.HareketTipi = shtFasonUretimeCikis or New.HareketTipi = shtUretimeCikis)) then
    begin
      Select Count(*) from BoyaTalimat where BoyaTalimatKayitId = new.BoyaTalimatKayitID and BoyaTalimat.UretimeCikisStokHareketID is not Null into :KS;
      if (KS > 0) then Exception KayitHatasi 'Seçilen Boya Talimat Uygun Değil..';

      Update BoyaTalimat set GidenMiktar = GidenMiktar + New.Miktar where BoyaTalimat.BoyaTalimatKayitId = New.BoyaTalimatKayitId;
      Update BoyaTalimatRecete set KullanilanMiktar = KullanilanMiktar + New.Miktar where BoyaTalimatRecete.BoyaTalimatKayitId = New.BoyaTalimatKayitId and BoyaTalimatRecete.OrmeRafID = new.OrmeRafID;
    end
    else if (New.UretimIslemTipiId = uitBoya and (New.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and New.HamIade = 1) then
       Update BoyaTalimat set IadeAlinanMiktar = IadeAlinanMiktar + new.Miktar where BoyaTalimat.BoyaTalimatKayitId = New.BoyaTalimatKayitId;
    else if (New.UretimIslemTipiId = uitBoya and (New.HareketTipi = shtFasonUretim or New.HareketTipi = shtUretim) and New.HamIade = 0) then
       Update BoyaTalimat set BrutGelenMiktar = BrutGelenMiktar + new.BrutMiktar, NetGelenMiktar = NetGelenMiktar + new.Miktar  where BoyaTalimat.BoyaTalimatKayitId = New.BoyaTalimatKayitId;
  end
end
