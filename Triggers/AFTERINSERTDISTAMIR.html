CREATE TRIGGER AFTERINSERTDISTAMIR FOR DISTAMIR
ACTIVE AFTER INSERT POSITION 0
as
declare variable Konu             varchar(50);
declare variable Aciklama         varchar(1500);
declare variable EskiPartiYil     integer;
declare variable EskiPartiNo      integer;
declare variable FirmaId          varchar(20);
declare variable FirmaAdi         varchar(100);
declare variable Kilo           Double Precision;
declare variable HamAdi         varchar(255);
declare variable IrsaliyeNo     IrsaliyeNoSize;
declare variable IrsaliyeTarihi TimeStamp;
begin
  Konu = 'DIŞ TAMİR';

  Select HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo, Ham.Adi, HamStokMaster.Kilo, HamStokIrsaliye.FirmaId,
         Firma.Adi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi from HamStokMaster
    left join Ham On Ham.HamID = HamStokMaster.HamId
    left join HamStokIrsaliye On HamStokIrsaliye.HamStokIrsaliyeID = HamStokMaster.HamStokIrsaliyeId
    left join Firma On Firma.FirmaID = HamStokIrsaliye.FirmaId
    left join Kullanici On Kullanici.KullaniciID = HamStokMaster.KullaniciId
    where HamStokMaster.HamStokMasterID = New.HamStokMasterId
    into :EskiPartiYil, :EskiPartiNo, :HamAdi, :Kilo, :FirmaId, :FirmaAdi, :IrsaliyeNo, :IrsaliyeTarihi;

/*  1233 nolu 01.02.2008 tarihli irsaliye ile (assdas) egemen yazilim firmasından Dış Tamir Gelmiştir.*/

  Aciklama = Cast(IrsaliyeNo || ' Nolu ' || TimeStampToDateTimeStr(IrsaliyeTarihi) || ' tarihli irsaliye ile Dış Tamir Açıldı..' || Chr(13) as Blob character set UTF8);

  Aciklama = Aciklama || Cast(Chr(13) || 'Firma  : (' || FirmaId || ') ' || FirmaAdi as Blob character set UTF8);

  Aciklama = Aciklama || Cast(Chr(13) || 'Kilo : ' || Format_Float('#,##0.#0', Kilo) as Blob character set UTF8);

  Aciklama = Aciklama || Cast(Chr(13) || 'Ham Stok Adı : ' || HamAdi as Blob character set UTF8);

  Aciklama = Aciklama || Cast(Chr(13) || 'Ham Stok Giriş No : ' || EgemenIntegerToStr(New.HamStokMasterId) as Blob character set UTF8);

  Aciklama = Aciklama || Cast(Chr(13) || 'Eski Parti No : ( ' || EgemenIntegerToStr(EskiPartiYil) || ') ' || EgemenIntegerToStr(EskiPartiNo) as Blob character set UTF8);


  insert into GelenMail(KullaniciId, KimdenKullaniciId, KimdenKullaniciAdi, Konu, Mesaj)
    Select Kullanici.KullaniciId, 1, 'SİSTEM YÖNETİCİSİ', :Konu, :Aciklama from Kullanici where Kullanici.DisTamirBildirilsin = 1;
end
