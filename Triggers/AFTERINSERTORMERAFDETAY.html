CREATE TRIGGER AFTERINSERTORMERAFDETAY FOR ORMERAFDETAY
ACTIVE AFTER INSERT POSITION 0
as
declare variable ToplamMiktar     Numeric(12,3);
declare variable GelenMiktar      Numeric(12,3);
begin
  Select sum(Miktar) from OrmeRafDetay where OrmeRafDetay.StokHareketId = new.StokHareketId into :ToplamMiktar;

  Select sum(Miktar) from StokHareket where StokHareket.StokHareketId = new.StokHareketId into :GelenMiktar;

  if (ToplamMiktar > GelenMiktar) then exception GelenKilodanFazlasiRaflanamaz;

  Update StokHareket set RaflananMiktar = RaflananMiktar + New.Miktar, NeredenUpdate = 4, LinkIslemi = 1 where StokHareket.StokHareketID = new.StokHareketId;

  Execute Procedure HesaplanacakRafEkle(new.OrmeRafId);
end
