CREATE TRIGGER AFTERINSERTPARTIASAMALARI FOR PARTIASAMALARI
ACTIVE AFTER INSERT POSITION 0
as
declare variable AsamaId integer;
declare variable AsamaAdi varchar(40);
declare variable Kumas integer;
declare variable Aksesuar integer;
declare variable Ribana integer;
declare variable SinifId integer;
declare variable SinifHamTipiId integer;
declare variable SinifDurumu integer;
declare variable Dahil integer;
declare variable PartiReceteId varchar(20);
declare variable KayitSayisi integer;
declare variable Sayac integer;
declare variable FSinifId integer;
declare variable FPartiTamirSinifID integer;
declare variable DegerTipiId integer;
declare variable MaxPartiDurumu integer;
declare variable SonradanEklendiIseHaberVer integer;
declare variable Konu              varchar(50);
declare variable FBlob             Blob;
declare variable Aciklama          varchar(1024);
declare variable PartiYil         integer;
declare variable PartiId          integer;
declare variable IlaveProsesSebebiAdi Varchar(20);
declare variable IcTamirAsamalariOtomatikKaydet integer;
declare variable PartiKilo Double Precision;
declare variable PAReceteBoyaMiktarDevrede integer;
declare variable FPamukOrani Double Precision;
declare variable FPolyesterOrani Double Precision;
declare variable FNaylonOrani Double Precision;
declare variable PamukPartiKilo Double Precision;
declare variable PolyesterPartiKilo Double Precision;
declare variable NaylonPartiKilo Double Precision;
begin
  if (new.ReceteId is not null) then
    select UstAsamaId from Recete where Recete.ReceteId = new.ReceteId into :AsamaId;
  else AsamaId = new.AsamaId;

  Select PAReceteBoyaMiktarDevrede from Sistem into :PAReceteBoyaMiktarDevrede;

  MaxPartiDurumu = 0;

  if (new.SerbestAsama = 0) then
  if (new.PartiTamirId is null) then
  begin
    Select Kumas, Aksesuar, Ribana, SonradanEklendiIseHaberVer, Adi from Asama where Asama.AsamaId = :AsamaId into :Kumas, :Aksesuar, :Ribana, :SonradanEklendiIseHaberVer, :AsamaAdi;
    Sayac = -1;

    for select PartiSinifId, HamTipiId, Durumu from PartiStok where PartiStok.PartiKayitId = new.PartiKayitId into :SinifId, :SinifHamTipiId, :SinifDurumu do
    begin
      Dahil = 0;
      if (MaxPartiDurumu < SinifDurumu) then MaxPartiDurumu = SinifDurumu;

      if (SinifDurumu = 8) then Dahil = 0;
      else
      begin
        if (SinifHamTipiId = 0 and Kumas = 1) then Dahil = 1;
        if (SinifHamTipiId = 1 and Ribana = 1) then Dahil = 1;
        if (SinifHamTipiId = 2 and Aksesuar = 1) then Dahil = 1;

        select count(*) from PartiAsamalari
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStok.PartiSinifId = :SinifId
        where PartiAsamalari.PartiKayitId = new.PartiKayitId and PartiAsamaStok.asamayadahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1
           and PartiAsamalari.Sira > new.Sira and (PartiAsamaStok.GirisTarihi is not null or PartiAsamaStok.Cikistarihi is not null) into :kayitSayisi;

        if (KayitSayisi > 0) then Dahil = 0;
        else if (Sayac = -1) then
        begin
          Sayac = 0;
          FSinifId = SinifId;
        end
      end
      if (dahil = 1) then Sayac = 1;

      insert into PartiAsamaStok(PartiAsamalariId, PartiKayitId, PartiSinifId, AsamayaDahil, MakinaPlanlamaGrupId, GrupProsesNo, AsamaId)
                          values(new.PartiAsamalariId, new.PartiKayitId, :SinifId, :Dahil, new.MakinaPlanlamaGrupId, new.GrupProsesNo, :AsamaId);
    end

    if (sayac = -1) then exception AsamayaStokDahilEdilemedi;
    else if (Sayac = 0) then update PartiAsamaStok set AsamayaDahil = 1 where PartiAsamaStok.PartiKayitId = new.PartiKayitId and PartiAsamaStok.PartiAsamalariId = new.PartiAsamalariId and PartiAsamaStok.PartiSinifId = :FSinifId;


    if (New.IlaveProses = 1) then
    begin
      Konu = 'İLAVE PROSES';

      Select Adi from IlaveProsesSebebi where IlaveProsesSebebi.IlaveProsesSebebiID = new.IlaveProsesSebebiID into :IlaveProsesSebebiAdi;

      Select Parti.PartiYil, Parti.PartiId from Parti where Parti.PartiKayitId = new.PartiKayitId
        into :PartiYil, :PartiId;

      Aciklama = cast('(' || EgemenIntegerToStr(PartiYil) || ') ' || EgemenIntegerToStr(PartiId) || ' nolu Partiye ' || IlaveProsesSebebiAdi || ' Sebebi ile ' ||
                 AsamaAdi || ' aşaması eklendi..( Kullanıcı No = ' || EgemenIntegerToStr(New.InsertKullaniciId) || ' )' as blob character set UTF8);

      FBlob = f_StrBlob(Aciklama);

      insert into GelenMail(KullaniciId, KimdenKullaniciId, KimdenKullaniciAdi, Konu, Mesaj)
         Select KullaniciId, 1, 'SİSTEM YÖNETİCİSİ', :Konu, :FBlob from Kullanici
          where Kullanici.PartiIlaveProsesBildirilsin = 1;
    end
    else if (SonradanEklendiIseHaberVer = 1 and MaxPartiDurumu > 0) then
    begin
      Konu = 'Parti Proses';

      Select Parti.PartiYil, Parti.PartiId from Parti where Parti.PartiKayitId = new.PartiKayitId
        into :PartiYil, :PartiId;

      Aciklama = cast('Temsilcisi Oldugunuz Firmanın (' || EgemenIntegerToStr(PartiYil) || ') ' || EgemenIntegerToStr(PartiId) || ' nolu Partisine  ''' || AsamaAdi || ''' aşaması eklendi..' as blob character set UTF8);

      FBlob = f_StrBlob(Aciklama);

      insert into GelenMail(KullaniciId, KimdenKullaniciId, KimdenKullaniciAdi, Konu, Mesaj)
        Select BoyaSiparis.SiparisiAlanKullaniciId, 1, 'SİSTEM YÖNETİCİSİ', :Konu, :FBlob from Parti
          left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
          where Parti.PartiKayitId = New.PartiKayitId and BoyaSiparis.SiparisiAlanKullaniciId is not null;
    end

  end
  else
  begin
    Sayac = -1;

    Select IcTamirAsamalariOtomatikKaydet from Sistem where sistemid = 1 into :IcTamirAsamalariOtomatikKaydet;

    for select PartiStok.PartiSinifId, PartiStok.HamTipiId, PartiStok.Durumu, PartiTamirSinif.PartiTamirSinifId from PartiStok
      left join PartiTamirSinif on PartiTamirSinif.PartiTamirId = new.PartiTamirId and PartiTamirSinif.PartiSinifId = PartiStok.PartiSinifId
      where PartiStok.PartiKayitId = new.PartiKayitId
      into :SinifId, :SinifHamTipiId, :SinifDurumu, :FPartiTamirSinifId do
    begin
      Dahil = 0;
      if (FPartiTamirSinifId is not null) then
      begin
        if (SinifDurumu = 8) then exception AsamayaStokDahilEdilemedi;
        Dahil = 1;
      end

      if (Dahil = 1) then
      begin
        select count(*) from PartiAsamalari
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStok.PartiSinifId = :SinifId
        where PartiAsamalari.PartiKayitId = new.PartiKayitId and PartiAsamaStok.asamayadahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1
           and PartiAsamalari.Sira > new.Sira and (PartiAsamaStok.GirisTarihi is not null or PartiAsamaStok.Cikistarihi is not null) into :kayitSayisi;

        if (KayitSayisi > 0) then exception AsamayaStokDahilEdilemedi;
      end

      if (IcTamirAsamalariOtomatikKaydet = 1 and Dahil = 1) then
      begin
        Dahil = 0;
        Select First 1 AsamayaDahil from PartiAsamaStok
          Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          where PartiAsamaStok.AsamaID = :AsamaId and PartiAsamaStok.PartiSinifID = :SinifId and Sira < New.Sira and
                PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamaStok.AsamayaDahil = 1
          Order by Sira Desc
        into :Dahil;
      end


      if (Dahil = 1) then
      begin
        Sayac = 1;
        insert into PartiAsamaStok(PartiAsamalariId, PartiKayitId, PartiSinifId, AsamayaDahil, MakinaPlanlamaGrupId, GrupProsesNo, AsamaId)
              values(new.PartiAsamalariId, new.PartiKayitId, :SinifId, :Dahil, new.MakinaPlanlamaGrupId, new.GrupProsesNo, :AsamaId);
      end

    end

    if (sayac = -1) then exception AsamayaStokDahilEdilemedi;

  end

  if (new.ReceteId is not null and new.PartininAsilRecetesimi = 1) then
  begin
    Update PartiAsamalari Set PartininAsilRecetesimi = 0
      where PartiKayitId = new.PartiKayitId and PartiAsamalariId <> new.PartiAsamalariId and ReceteId is not null;

    if (New.BanyoCarpani is not null and New.BanyoCarpani > 0) then
      update Parti Set ReceteId = new.ReceteId, ReceteTarihi = current_TimeStamp(0) where PartiKayitId = new.PartiKayitId;
    else
      update Parti Set ReceteId = new.ReceteId where PartiKayitId = new.PartiKayitId;
   end

  if (new.ReceteId is not null and PAReceteBoyaMiktarDevrede = 1) then
  begin
    Select Sum(Miktar) from PartiStok Where PartiStok.PartiKayitID = New.PartiKayitID into :PartiKilo;

    Insert Into PartiAsamalariReceteBoyaMiktar (PartiAsamalariID, PartiKayitID, ReceteAsamaID, Miktar, InsertKullaniciId)
      Select new.PartiAsamalariId, new.PartiKayitId, ReceteAsama.ReceteAsamaID, case when new.partitamirid is null then :PartiKilo else 0 end, New.KullaniciId From ReceteDetay
      left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
      Where ReceteDetay.ReceteID = New.ReceteID and ReceteAsama.BoyaAsamasimi = 1;
  end

  /* recete asaması ise üst asamaidyi dikkate alıyoruz o yuzden new.asamaid yi kullanmıyoruz */
  if (New.SerbestAsama = 0) then
  begin
    execute Procedure PartiAsamaKontrolOlustur(new.PartiKayitId, new.PartiAsamalariId, :AsamaId);
  end

  if (new.PartiKayitId is not Null) then select PartiKayitId from PartiMaliyetiKOlustur(new.PartiKayitId) into KayitSayisi;

  select Asama.Laminasyon from Asama where Asama.AsamaId = new.AsamaId into :KayitSayisi;
  if (KayitSayisi = 1) then
    update Parti set Parti.Laminasyon = 1 where Parti.PartiKayitId = new.PartiKayitId;
end
