Definition

CREATE TRIGGER BEFOREUPDATEPARTISTOKTARTI FOR PARTISTOKTARTI
ACTIVE BEFORE UPDATE POSITION 0
as
declare variable kayitSayisi Integer;
declare variable TamamiTartildi integer;
declare variable SistemKaydi integer;
declare variable MaximumFireOrani       Double Precision;
declare variable PartiBrutKilo   Numeric(12,3);
declare variable PartiBrutMetre  Numeric(12,3);
declare variable ToplamBrutKilo   Numeric(12,3);
declare variable ToplamBrutMetre  Numeric(12,3);
declare variable OldKilo  Numeric(12,3);
declare variable BirinciKaliteSevkEdilemez  integer;
declare variable BrutuHesaplanmiyacakKaliteNolar varchar(20);
declare variable FGramaj         Double Precision;
declare variable IntGramaj       integer;
declare variable IntEn           integer;
declare variable HacimEn         integer;
declare variable Gramaj          varchar(10);
declare variable TF              integer;
declare variable FTup            integer;
declare variable KaliteKontrolMetreHesapSekli integer;
declare variable MetreHesabiIcinCarpan Numeric(12,3);
declare variable FHesap  Double Precision;
declare variable FEnVerilen     varchar(10);
declare variable FGramajVerilen varchar(10);
declare BaskaFirmayaSatilabilir smallint;
declare variable UrunFirmaId varchar(20);
declare variable SatilanFirmaId varchar(20);
declare variable OzelMusteri smallint;
begin
  if (new.LinkIslemi = 0 and old.LinkIslemi = 0 or new.LinkIslemi = 2) then
  begin
    if (Old.BoyaIrsaliyeDetayId is not null and new.BoyaIrsaliyeDetayId is not null and Old.BoyaIrsaliyeDetayId <> new.BoyaIrsaliyeDetayId) then exception BuPaketinIrsaliyesiKesilmis;
    if (Old.HamStokSarfFisiDetayId is not null and new.HamStokSarfFisiDetayId is not null and Old.HamStokSarfFisiDetayId <> new.HamStokSarfFisiDetayId) then exception BuPaketSarFisiIleCikmis;
    if (new.BoyaIrsaliyeDetayId is not null and Old.BoyaRezervFisiDetayId is not null and new.BoyaRezervFisiDetayId is null) then exception BuTartiSevkEdilmis;

    if (new.HamStokSarfFisiDetayId is not null and new.BoyaIrsaliyeDetayId is not null) then exception KayitHatasi;


  select Firma.OzelMusteri  from PartiStok
      left join Parti On Parti.PartiKayitId = PartiStok.PartiKayitId
      left join Firma on Firma.FirmaId = Parti.FirmaId
      where PartiStok.PartiSinifId = new.PartiSinifId
      into :OzelMusteri;
    if (Old.BoyaIrsaliyeDetayId is not null and new.BoyaIrsaliyeDetayId is null) then
    begin
      Select Count(*) from PartiStokTarti
        Left Join IplikPaketleme on IplikPaketleme.PartiStokTartiID = PartiStokTarti.PartiStokTartiID
        Left Join PSKaliteKontrol on PSKaliteKontrol.PartiStokTartiID = PartiStokTarti.PartiStokTartiID
        Left Join PartiStokTarti IadePartiStokTarti on IadePartiStokTarti.PartiSinifID = PartiStokTarti.PartiSinifID and IadePartiStokTarti.IadeBoyaIrsaliyeID is not null and
        ((IplikPaketleme.IplikPaketlemeID is not Null and IadePartiStokTarti.BarkodID = 'I-' || EgemenIntegerToStr(IplikPaketlemeID)) or
         (PSKaliteKontrol.PSKaliteKontrolId is not Null and IadePartiStokTarti.BarkodID = 'K-' || EgemenIntegerToStr(PSKaliteKontrolID)))
       where PartiStokTarti.PartiStokTartiID = New.PartiStokTartiID and IadePartiStokTarti.PartiStokTartiID is not Null
      into :kayitSayisi;

      if (kayitSayisi > 0) then Exception KayitHatasi ' Bu Satışa Ait İade Girişi Var!';
    end

    if (Old.BoyaIrsaliyeDetayId is null and new.BoyaIrsaliyeDetayId is not null) then -- BoyaIrsaliyeDetayId dolmus ise
    begin
     select BoyaSiparis.FirmaId, BoyaSiparis.BaskaFirmayaSatilabilir from PartiStok
       left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = PartiStok.BoyaSiparisKayitId
     where PartiStok.PartiSinifId = New.PartiSinifId
     into :UrunFirmaId, :BaskaFirmayaSatilabilir;

     if (BaskaFirmayaSatilabilir = 0) then
     begin
       select BoyaSiparis.FirmaId from BoyaIrsaliyeDetay
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaIrsaliyeDetay.BoyaSiparisDetayId
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
       where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = New.BoyaIrsaliyeDetayId
       into :SatilanFirmaId;

       if (UrunFirmaId <> SatilanFirmaId) then  Exception BaskaFirmayaSatilamaz;
     end
    end

    Select KaliteKontrolMetreHesapSekli ,BrutuHesaplanmiyacakKaliteNolar  from sistem where SistemID = 1 into :KaliteKontrolMetreHesapSekli ,:BrutuHesaplanmiyacakKaliteNolar;

    Select Miktar from FormataGoreKiloYuvarla(New.Kilo, 'BOYA', 'BRUTKILO') into New.Kilo;
    Select Miktar from FormataGoreKiloYuvarla(New.Dara, 'BOYA', 'BRUTKILO') into New.Dara;

    Select Miktar from FormataGoreKiloYuvarla(Old.Kilo, 'BOYA', 'BRUTKILO') into :OldKilo;

    if (New.BoyaIrsaliyeDetayId is not null or New.HamStokSarfFisiDetayId is not null) then
    begin
      if (OldKilo <> New.Kilo or Old.Metre <> New.Metre or Old.BobinSayisi <> New.BobinSayisi) then exception SevkEdilmisKiloDegisemez;
      if (Old.KaliteNo <> New.KaliteNo) then exception SevkEdilmisKiloDegisemez;
    end
    if (New.BoyaRezervFisiDetayId is not null) then
    begin
      if (OldKilo <> New.Kilo or Old.BrutKilo <> New.BrutKilo) then exception RezervEdilmisTartiKaydi;
      if (Old.KaliteNo <> New.KaliteNo) then exception RezervEdilmisTartiKaydi;

      if (New.BoyaRezervFisiDetayId <> Coalesce(Old.BoyaRezervFisiDetayId, 0)) then
      begin
        select count(*) from BoyaRezervFisiDetay where BoyaRezervFisiDetay.BoyaRezervFisiDetayId = New.BoyaRezervFisiDetayId and BoyaRezervFisiDetay.PartiSinifId = new.PartiSinifId into :KayitSayisi;
        if (KayitSayisi = 0) then
          Exception KayitHatasi 'Farklı parti sınıfa ait olanlar girilemez';
      end
    end

    if (new.LinkIslemi = 0) then
    if (OldKilo <> New.Kilo or Old.Metre <> New.Metre or Old.BobinSayisi <> New.BobinSayisi) then
    begin
      Select Count(*) from IplikPaketleme where IplikPaketleme.PartiStokTartiID = new.PartiStokTartiId into :KayitSayisi;
      if (KayitSayisi > 0) then exception BuradanKiloDegistiremezsiniz;

      Select Count(*) from PsKaliteKontrol where PsKaliteKontrol.PartiStokTartiID = new.PartiStokTartiId into :KayitSayisi;
      if (KayitSayisi > 0) then exception BuradanKiloDegistiremezsiniz;

      select count(*) from BoyaIrsaliyeDetay where BoyaIrsaliyeDetay.PartiSinifId = new.PartiSinifId into :KayitSayisi;
      if (KayitSayisi > 0) then exception IrsaliyeKesilmisKiloDegisemez;
    end


    if (new.LinkIslemi = 0) then
    if (OldKilo <> New.Kilo or Old.Metre <> New.Metre or Old.BrutKilo <> New.BrutKilo or Old.BrutMetre <> New.BrutMetre) then
    begin
      Select SistemKaydi from MamulDepoTransferFisi where MamulDepoTransferFisi.MamulDepoTransferFisiId = New.MamulDepoTransferFisiId into :SistemKaydi;

      if (SistemKaydi = 0) then exception SevkEdilmisKiloDegisemez;
    end


    if (OldKilo <> New.Kilo or Old.Metre <> New.Metre or Old.BrutKilo <> New.BrutKilo or Old.BrutMetre <> New.BrutMetre) then
    begin
      select PartiStok.Kilo, PartiStok.Metre, PartiStok.MaximumFireOrani from PartiStok
        where PartiStok.PartiSinifId = new.PartiSinifId
        into :PartiBrutKilo, :PartiBrutMetre, :MaximumFireOrani;

      Select Sum(PartiStokTarti.BrutKilo), Sum(PartiStokTarti.BrutMetre) from PartiStokTarti
        where PartiStokTarti.PartiSinifId = new.PartiSinifId and PartiStokTarti.PartiStokTartiID <> New.PartiStokTartiId
        into :ToplamBrutKilo, :ToplamBrutMetre;

      if (ToplamBrutKilo is null) then ToplamBrutKilo = 0;
      if (ToplamBrutMetre is null) then ToplamBrutMetre = 0;

      if (new.BrutKilo = 0) then
      begin
        New.BrutKilo = new.Kilo * 100 / (100 - MaximumFireOrani);
        Select Miktar from FormataGoreKiloYuvarla(New.BrutKilo, 'BOYA', 'BRUTKILO') into New.BrutKilo;
        if ((ToplamBrutKilo + New.BrutKilo) > PartiBrutKilo) then New.BrutKilo = PartiBrutKilo - ToplamBrutKilo;
      end

      if (KaliteKontrolMetreHesapSekli = 0) then
      begin
        if (new.Metre is null or new.Metre <= 0 or  Old.Kilo <> New.Kilo or old.Dara <> New.Dara) then
        begin
          Select PartiStok.HacimEn, PartiStok.Gramaj, PaketlemeSekli.Tup from PartiStok
            left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
            left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
            where PartiSinifId = new.PartiSinifId
            into :HacimEn, :Gramaj, :FTup;

          TF = EgemenStrToIntOlurmu(Gramaj);

          if (TF = 0) then IntGramaj = 0;
          else IntGramaj = EgemenStrToInt(Gramaj);

          if (FTup is null) then FTup = 0;

          if (FTup = 0) then
            FGramaj = (IntGramaj * HacimEn) / 100000.000;
          else
            FGramaj = (IntGramaj * HacimEn * 2) / 100000.000;

          if (FGramaj is null) then FGramaj = 0;

          if (FGramaj <> 0) then New.Metre = (New.Kilo - New.Dara)/ FGramaj;
          else New.Metre = 0;
        end
      end
      else if (KaliteKontrolMetreHesapSekli = 2) then /* Kral */
      begin
        if (new.Metre is null or new.Metre = 0 or Old.Kilo <> New.Kilo) then
        begin
          Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
            where PartiStokFizikKontrol.partiSinifID = new.partiSinifID and PartiStokFizikKontrol.SonKayit = 1
            into :FEnVerilen, :FGramajVerilen;

          TF = EgemenStrToIntOlurmu(FGramajVerilen);

          if (TF = 0) then IntGramaj = 0;
          else IntGramaj = EgemenStrToInt(FGramajVerilen);

          TF = EgemenStrToIntOlurmu(FEnVerilen);

          if (TF = 0) then IntEn = 0;
          else IntEn = EgemenStrToInt(FEnVerilen);

          FHesap = IntEn * IntGramaj;

          if (FHesap <> 0) then
            new.Metre = (1000 / FHesap ) * 100 * New.Kilo;
          else
            new.Metre = 0;
        end
      end
      else if (KaliteKontrolMetreHesapSekli = 3) then /* NY */
      begin
        if (new.Metre is null or new.Metre = 0 or Old.Kilo <> New.Kilo) then
        begin
          Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
            where PartiStokFizikKontrol.partiSinifID = new.partiSinifID and PartiStokFizikKontrol.SonKayit = 1
            into :FEnVerilen, :FGramajVerilen;

          TF = EgemenStrToIntOlurmu(FGramajVerilen);

          if (TF = 0) then IntGramaj = 0;
          else IntGramaj = EgemenStrToInt(FGramajVerilen);

          TF = EgemenStrToIntOlurmu(FEnVerilen);

          if (TF = 0) then IntEn = 0;
          else IntEn = EgemenStrToInt(FEnVerilen);

          FHesap = IntEn * IntGramaj;

          if (FHesap <> 0) then
            new.Metre = (1000 / FHesap ) * 100 * (New.Kilo - New.Dara);
          else
            new.Metre = 0;
        end
      end
      else if (KaliteKontrolMetreHesapSekli = 4) then /*Ottoman Bursa*/
      begin
        if (new.Metre is null or new.Metre = 0) then
        if (Old.Kilo <> New.Kilo or old.Dara <> New.Dara) then
        begin
          Select PartiStok.HacimEn, PartiStok.Gramaj, PaketlemeSekli.Tup from PartiStok
            left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
            left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliId
            where PartiSinifId = new.PartiSinifId
            into :HacimEn, :Gramaj, :FTup;

          TF = EgemenStrToIntOlurmu(Gramaj);

          if (TF = 0) then IntGramaj = 0;
          else IntGramaj = EgemenStrToInt(Gramaj);

          if (FTup = 0) then
            FGramaj = (IntGramaj * HacimEn) / 100000.000;
          else
            FGramaj = (IntGramaj * HacimEn * 2) / 100000.000;

          if (FGramaj is null) then FGramaj = 0;

          if (FGramaj <> 0) then New.Metre = (New.Kilo - New.Dara)/ FGramaj;
          else New.Metre = 0;
        end
      end
      else if (KaliteKontrolMetreHesapSekli = 7 and OzelMusteri = 0) then /* Gülle, Güllede PSKK da testa dan metre bilgisi alıyoruz ancak Tartiya aktarırken en-gramajdan metre istiyorlar */
      begin
        Select EnVerilen, GramajVerilen from PartiStokFizikKontrol
        where PartiStokFizikKontrol.partiSinifID = new.partiSinifID and PartiStokFizikKontrol.SonKayit = 1
        into :FEnVerilen, :FGramajVerilen;

        TF = EgemenStrToIntOlurmu(FGramajVerilen);

        if (TF = 0) then IntGramaj = 0;
        else IntGramaj = EgemenStrToInt(FGramajVerilen);

        TF = EgemenStrToIntOlurmu(FEnVerilen);

        if (TF = 0) then IntEn = 0;
        else IntEn = EgemenStrToInt(FEnVerilen);

        FHesap = cast(IntEn * IntGramaj as double precision);

        if (FHesap <> 0) then
           new.Metre = (1000 / FHesap ) * 100 * (New.Kilo);  ---new.dara
        else
          new.Metre = 0;
      end

      if (new.BrutMetre = 0) then
      begin
        New.BrutMetre = New.Metre * 100 / (100 - MaximumFireOrani);
        Select Miktar from FormataGoreKiloYuvarla(New.BrutMetre, 'BOYA', 'BRUTMERE') into New.BrutMetre;
        if ((ToplamBrutMetre + New.BrutMetre) > PartiBrutMetre) then New.BrutMetre = PartiBrutMetre - ToplamBrutMetre;
      end

      if (PartiBrutKilo > 0) then
        if ((ToplamBrutKilo + New.BrutKilo) > PartiBrutKilo + 0.005) then exception SevkBrutKiloyuAstiniz;
      if (PartiBrutMetre > 0) then
        if ((ToplamBrutMetre + New.BrutMetre) > PartiBrutMetre + 0.005) then exception SevkBrutKiloyuAstiniz;
    end

    if (Old.MamulDepoTransferFisiId <> New.MamulDepoTransferFisiId and new.HamStokMasterId is null) then
      Select TransferDepoId from MamulDepoTransferFisi where MamulDepoTransferFisi.MamulDepoTransferFisiId = New.MamulDepoTransferFisiId into new.DepoId;

    if (new.KaliteNo < 1) then exception KayitHatasi 'Kalite No Birden Kucuk Olamaz..';

    if (Old.KaliteNo <> New.KaliteNo) then
    begin
      if (New.KaliteNo = 1) then
      begin
        Select BirinciKaliteSevkEdilemez from PartiStok where PartiStok.PartiSinifID = New.PartiSinifID into :BirinciKaliteSevkEdilemez;

        if (BirinciKaliteSevkEdilemez = 1) then exception KayitHatasi 'Bu Stok Icin Birinci Kalite Sevk Edilemez İsaretlenmis..';
      end

      if (Old.KaliteNo = 1) then Select MamulProsesId from MamulProsesNoBul(new.PartiSinifId) into new.MamulProsesId;
      else if (New.KaliteNo = 1) then new.MamulProsesId = Null;
    end

    if (coalesce(BrutuHesaplanmiyacakKaliteNolar, '') <> '') then
    begin
      if (BrutuHesaplanmiyacakKaliteNolar like '%'||new.kaliteno||'%') then
      begin
        New.BrutKilo = 0;
        New.BrutMetre = 0;
      end
    end

  end

  new.DegisimTarihi = current_TimeStamp(0);
  /*new.LinkIslemi = 0; after updatee alindi*/

  /*Irsaliyeden silinme loglamasi*/
  if (old.BoyaIrsaliyeDetayId is not null and new.BoyaIrsaliyeDetayId is null) then
    execute procedure Logla('U', 1, 'PARTISTOKTARTI', 'PARTISTOKTARTIID', new.PartiStokTartiID, null);
end
