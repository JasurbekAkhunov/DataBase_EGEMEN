CREATE TRIGGER BEFOREUPDATEPARTIRPT FOR PARTIRPT
ACTIVE BEFORE UPDATE POSITION 0
as
begin
  if (New.BaslamaKullaniciId is null) then new.BaslamaTarihi = Null;
  if (New.BitisKullaniciId is null) then new.BitisTarihi = Null;

  if (Old.BaslamaKullaniciId is null and New.BaslamaKullaniciId is not null) then new.BaslamaTarihi = current_TimeStamp(0);
  if (Old.BitisKullaniciId is null and New.BitisKullaniciId is not null) then new.BitisTarihi = current_TimeStamp(0);

  if (Old.Degerlendirme <> New.Degerlendirme) then
  begin
    if (New.Degerlendirme = 0) then
    begin
      New.DegerlendirmeTarihi = Null;
      new.DegerlendirenKullaniciId = Null;
    end
    else
    begin
      New.DegerlendirmeTarihi = current_TimeStamp(0);
      new.DegerlendirenKullaniciId = New.KullaniciId;
    end
  end
  New.DegisimTarihi = current_TimeStamp(0);
end
