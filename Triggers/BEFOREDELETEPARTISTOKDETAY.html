CREATE TRIGGER BEFOREDELETEPARTISTOKDETAY FOR PARTISTOKDETAY
ACTIVE BEFORE DELETE POSITION 0
as
declare variable KayitSayisi    integer;
declare variable FFirmaId       varchar(20);
declare variable FFason         integer;
begin
  if (Old.LinkIslemi = 0) then
  begin
    select count(*) from PartiAsamaStok where AsamayaDahil = 1 and PartiAsamaStok.PartiSinifId = old.PartiSinifId into :KayitSayisi;
    if (KayitSayisi > 0) then exception StogunDahilOlduguAsamaVar;
  end

  Select FirmaId, Fason from Parti where Parti.PartiKayitId = Old.PartiKayitId into :FFirmaId, :FFason;

  if (Old.RafDetayId is not null) then execute procedure HamStokKumilatifGuncelle(:FFirmaId, :FFason, 2, Old.HamId, Null, Old.PartilemeTarihi, Null, Old.Kilo, Old.Metre, Null, Null);
end
