CREATE TRIGGER BEFOREDELETEHAMSTOKDETAY FOR HAMSTOKDETAY
ACTIVE BEFORE DELETE POSITION 0
as
declare variable PartiSinifId integer;
begin
  select HamStokMaster.PartiSinifId from HamStokMaster where HamStokMaster.HamStokMasterId = old.HamStokMasterId into :PartiSinifId;

  delete from RafDetay where HamStokDetayId = Old.HamStokDetayId;
  delete from HSDRezervBilgi where HamStokDetayId = Old.HamStokDetayId;

  if (PartiSinifId is null) then /*yari mamul degil*/
    execute procedure HamStokKumilatifGuncelle(Old.StokFirmaId, Old.Fason, 1, Old.HamId, Null, Old.Tarih, Null, Old.Kilo, Old.Metre, Null, null);
end
