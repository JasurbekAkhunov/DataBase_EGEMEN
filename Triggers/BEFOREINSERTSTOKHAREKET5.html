Definition

CREATE TRIGGER BEFOREINSERTSTOKHAREKET5 FOR STOKHAREKET
ACTIVE BEFORE INSERT POSITION 5
as
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;

declare variable uitBukum       integer;
declare variable uitOrme        integer;
declare variable uitBoya        integer;
declare variable uitDokuma      integer;
declare variable uitCozgu       integer;
declare variable uitHasil       integer;
declare variable uitIndigo       integer;

declare variable StokTabanDovizId  varchar(5);
declare variable FNetBirimFiyat Double Precision;
declare variable hstIplik        integer;
declare variable hstLycra        integer;
declare variable IplikMaliyeti   Double Precision;
declare variable LycraMaliyeti   Double Precision;
declare variable TahminiFireOrani  Double Precision;
declare variable HamMaliyeti       Double Precision;
declare variable FBirimMaliyet    Double Precision;
declare variable FDovizId         varchar(5);
declare variable DovizKuru        Double Precision;
declare variable FTutar           Double Precision;
declare variable HamTipiId        integer;
declare variable PartiSinifId     integer;
declare variable LycraKullanimOrani  Double Precision;
declare variable SiparisMaliyetTakibiVar integer;
begin
/* bu trigger Birim Maliyet Hesapliyoruz..*/
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  Select SiparisMaliyetTakibiVar, StokTabanDovizId from Sistem where SistemID = 1 into :SiparisMaliyetTakibiVar, :StokTabanDovizId;


  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim;

  Select uitBukum, uitOrme, uitBoya, uitDokuma, uitCozgu, uitHasil, uitIndigo from UretimIslemTipiGetir into :UitBukum, :uitOrme, :uitBoya, :uitDokuma, :uitCozgu, :uitHasil, :uitIndigo;

  if (SiparisMaliyetTakibiVar = 1) then
  if (New.GirisCikis = 1) then
  begin
    if (New.OrmeSiparisDetayID is not null) then
      Select TahminiFireOrani from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayID = New.OrmeSiparisDetayID into :TahminiFireOrani;
    else
      TahminiFireOrani = 0;

    if (StokTabanDovizId <> New.DovizID) then
      Select DovizKuru from EnYakinDovizKuruGetir(new.DovizId, :StokTabanDovizId, new.Tarih) into :DovizKuru;
    else
      DovizKuru = 1;

/* NetBirimFiyat i kullanacagimiz yerler icin sitem dövize cevirip elimizde hazır tutuyoruz */
    FNetBirimFiyat = New.NetBirimFiyat * DovizKuru;

    if (New.UretimIslemTipiID is null) then
    begin
      new.BirimMaliyet = FNetBirimFiyat;
    end
    else
    if (new.HamIade = 1) then
    begin
      if (New.BoyaSiparisKayitId is null) then
      begin
        New.BirimMaliyet = FNetBirimFiyat;
      end
      else
      begin
        if (new.BoyaIrsaliyeDetayId is null) then
        begin
          Select Sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / Sum(StokHareket.Miktar) from StokHareket
            where StokHareket.BoyaSiparisKayitId = New.BoyaSiparisKayitId and
                  StokHareket.UretimIslemTipiID = New.UretimIslemTipiID and
                  StokHareket.GirisCikis = -1 and
                  StokHareket.HamID = New.HamId and
                  IntKarsilastir(StokHareket.RenkId, New.RenkId) = 1 and
                  StokHareket.Miktar > 0
            into New.BirimMaliyet;

          if (New.BirimMaliyet is null) then New.BirimMaliyet = 0;
        end
        else
        begin
          New.BirimMaliyet = FNetBirimFiyat;
        end
      end
    end
    else
    begin
      if (New.BoyaSiparisKayitId is null) then
      begin
        New.BirimMaliyet = FNetBirimFiyat;
      end
      else
      begin
        if (new.UretimIslemTipiId = uitBukum) then
        begin
          Select Sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / Sum(StokHareket.Miktar) from StokHareket
            where StokHareket.BoyaSiparisKayitId = New.BoyaSiparisKayitId and
          StokHareket.OrmeSiparisDetayId = New.OrmeSiparisDetayId and
                  StokHareket.UretimIslemTipiID = New.UretimIslemTipiID and
                  StokHareket.GirisCikis = -1 and
                  StokHareket.BirimMaliyet > 0 and
                  StokHareket.Miktar > 0
            into New.BirimMaliyet;

          if (New.BirimMaliyet is null) then New.BirimMaliyet = 0;
          New.BirimMaliyet = New.BirimMaliyet + FNetBirimFiyat;
        end
        else if (new.UretimIslemTipiId = uitOrme) then
        begin
          Select Sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / Sum(StokHareket.Miktar) from StokHareket
            left join Ham On Ham.HamID = StokHareket.HamId
            where StokHareket.OrmeSiparisDetayId = New.OrmeSiparisDetayId and
                  StokHareket.UretimIslemTipiID = New.UretimIslemTipiID and
                  Ham.HamTipiId = :hstIplik and
                  StokHareket.GirisCikis = -1 and
                  StokHareket.Miktar > 0
            into :IplikMaliyeti;

          if (IplikMaliyeti is null) then IplikMaliyeti = 0;

          IplikMaliyeti = IplikMaliyeti + IplikMaliyeti * TahminiFireOrani / 100;

          if (New.OrmePlanlamaId is not null) then
          begin
            Select Sum(OrmePlanlamaRecete.KullanimOrani) from OrmePlanlamaRecete
            left join Ham On Ham.HamID = OrmePlanlamaRecete.HamId
            where OrmePlanlamaRecete.OrmePlanlamaId = New.OrmePlanlamaId and
                  Ham.HamTipiId = :hstLycra
            into :LycraKullanimOrani;

          end
          else LycraKullanimOrani = 0;

          Select Sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / Sum(StokHareket.Miktar) from StokHareket
            left join Ham On Ham.HamID = StokHareket.HamId
            where StokHareket.OrmeSiparisDetayId = New.OrmeSiparisDetayId and
                  StokHareket.UretimIslemTipiID = New.UretimIslemTipiID and
                  Ham.HamTipiId = :hstLycra and
                  StokHareket.GirisCikis = -1 and
                  StokHareket.Miktar > 0
            into :LycraMaliyeti;

          if (LycraMaliyeti is null) then LycraMaliyeti = 0;
          else LycraMaliyeti = LycraMaliyeti * LycraKullanimOrani / 100;

          New.BirimMaliyet = IplikMaliyeti + LycraMaliyeti + FNetBirimFiyat;
        end
        else if (new.UretimIslemTipiId = uitDokuma) then
        begin
          Select Sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / Sum(StokHareket.Miktar) from StokHareket
            left join Ham On Ham.HamID = StokHareket.HamId
            where StokHareket.OrmeSiparisDetayId = New.OrmeSiparisDetayId and
                  StokHareket.UretimIslemTipiID = New.UretimIslemTipiID and
                  Ham.HamTipiId = :hstIplik and
                  StokHareket.GirisCikis = -1 and
                  StokHareket.Miktar > 0
            into :IplikMaliyeti;

          if (IplikMaliyeti is null) then IplikMaliyeti = 0;

          IplikMaliyeti = IplikMaliyeti + IplikMaliyeti * TahminiFireOrani / 100;

          if (New.DokumaPlanlamaId is not null) then
          begin
            Select Sum(DokumaPlanlamaRecete.KullanimOrani) from DokumaPlanlamaRecete
            left join Ham On Ham.HamID = DokumaPlanlamaRecete.HamId
            where DokumaPlanlamaRecete.DokumaPlanlamaId = New.DokumaPlanlamaId and
                  Ham.HamTipiId = :hstLycra
            into :LycraKullanimOrani;

          end
          else LycraKullanimOrani = 0;

          Select Sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / Sum(StokHareket.Miktar) from StokHareket
            left join Ham On Ham.HamID = StokHareket.HamId
            where StokHareket.OrmeSiparisDetayId = New.OrmeSiparisDetayId and
                  StokHareket.UretimIslemTipiID = New.UretimIslemTipiID and
                  Ham.HamTipiId = :hstLycra and
                  StokHareket.GirisCikis = -1 and
                  StokHareket.Miktar > 0
            into :LycraMaliyeti;

          if (LycraMaliyeti is null) then LycraMaliyeti = 0;
          else LycraMaliyeti = LycraMaliyeti * LycraKullanimOrani / 100;

          New.BirimMaliyet = IplikMaliyeti + LycraMaliyeti + FNetBirimFiyat;
        end
        else if (new.UretimIslemTipiId = uitBoya) then
        begin
          if (new.BoyaSiparisDetayID is not null) then
          if (new.BoyaIrsaliyeDetayId is null) then
          begin
            Select HamTipiId from Ham where Ham.HamID = New.HamId into :HamTipiId;
/* iplik boya ise ham iplik maliyeti, kumas boya ise ham kumas maliyetini hesaplayacagiz */

            Select Sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / Sum(StokHareket.Miktar) from StokHareket
              left join Ham On Ham.HamID = StokHareket.HamId
              where StokHareket.OrmeSiparisDetayId = New.OrmeSiparisDetayId and
                    StokHareket.UretimIslemTipiID = New.UretimIslemTipiID and
                    Ham.HamTipiId = :HamTipiId and
                    StokHareket.GirisCikis = -1 and
                    StokHareket.Miktar > 0
              into :HamMaliyeti;

            if (HamMaliyeti is null) then HamMaliyeti = 0;

            New.BirimMaliyet = HamMaliyeti + FNetBirimFiyat;
          end
          else
          begin
            New.BirimMaliyet = 0;
            HamMaliyeti = 0;

            Select PartiSinifID from BoyaIrsaliyeDetay where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = new.BoyaIrsaliyeDetayID into :PartiSinifID;

            if (PartiSinifID is not null) then
               Select HamStokBirimMaliyeti from SiparisDetayBirimMaliyetGetir(:PartiSinifID, Null) into :HamMaliyeti;

            New.BirimMaliyet = HamMaliyeti + FNetBirimFiyat;
            /* kendi boyahaneimizden gelirken FNetBirimFiyat sipariş prosesden ve ilave işlenlerden hesaplanmıs gelecek */
          end
        end
      end
    end
  end
  else
  begin
    if (SiparisMaliyetTakibiVar = 1) then Select BirimMaliyet from RafFifoMaliyetGetir(New.OrmeRafId, New.Miktar, new.Tarih, New.StokHareketId) into new.BirimMaliyet;
    else Select OrmeRaf.BirimMaliyet from OrmeRaf where OrmeRaf.OrmeRafID = New.OrmeRafId into new.BirimMaliyet;
  end

  if (New.BirimMaliyet is null) then New.BirimMaliyet = 0;
end
