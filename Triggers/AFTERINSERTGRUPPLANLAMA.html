Definition

CREATE TRIGGER AFTERINSERTGRUPPLANLAMA FOR GRUPPLANLAMA
ACTIVE AFTER INSERT POSITION 0
as
declare variable PartiAsamalariId integer;
declare variable PartiSinifId     integer;
declare variable KS               integer;
begin
  Select PartiAsamalariId, PartiSinifId from BolumPlanlama where BolumPlanlamaId = New.BolumPlanlamaId into :PartiAsamalariId, :PartiSinifId;

  Update PartiAsamaStok set MakinaPlanlamaGrupId = new.MakinaPlanlamaGrupId
    where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId and (PartiAsamaStok.MakinaPlanlamaGrupId is null or PartiAsamaStok.MakinaPlanlamaGrupId <> New.MakinaPlanlamaGrupId);
end
