CREATE TRIGGER BEFOREUPDATERAFLOKASYON FOR RAFLOKASYON
ACTIVE BEFORE UPDATE POSITION 0
AS
Declare Variable FRafSepetSayisi Integer;
begin
  Select Count(*) From Raf Where Raf.SonRafLokasyonId = New.RafLokasyonId Into :FRafSepetSayisi;

  if (New.SepetSayisi < FRafSepetSayisi) then
    exception KayitHatasi ('Sepet sayısı '||EgemenIntegerToStr(FRafSepetSayisi)||' değerinden küçük olamaz...');
end
