CREATE TRIGGER AFTERUPDATEIPLIKIRSALIYE FOR IPLIKIRSALIYE
ACTIVE AFTER UPDATE POSITION 0
as
declare variable shtAlis               integer;
declare variable shtFasonAlis          integer;
declare variable shtSatis              integer;
declare variable shtAlisIade           integer;
declare variable shtSatisIade          integer;
declare variable shtUretimeCikis       integer;
declare variable shtUretim             integer;
declare variable shtUretimIade         integer;
declare variable ShtTelefGirisi        integer;
declare variable shtUretimCikisBoya    integer;
declare variable shtUretimBoya         integer;
declare variable IplikIrsaliyeDetayId  integer;
declare variable shtSarimaCikis        integer;
declare variable shtUretimKirpinti     integer;
begin
  if ( (old.IplikFaturaId is null and new.IplikFaturaId is not null) or (old.IplikFaturaId is not null and new.IplikFaturaId is null) ) then
  begin
    update IplikIrsaliyeDetay Set IplikFaturaId = new.IplikFaturaId, NeredenUpdate = 3, KullaniciId = New.KullaniciId, LinkIslemi = 1 where IplikIrsaliyeDetay.IplikIrsaliyeId = new.IplikIrsaliyeId;

    update IplikFatura set KullaniciId = New.KullaniciId, IslemDovizKuru1 = 0 where IplikFatura.IplikFaturaId = new.IplikFaturaId or IplikFatura.IplikFaturaId = old.IplikFaturaId;
  end

  if (Old.IrsaliyeTarihi <> New.IrsaliyeTarihi or Old.DovizId <> New.DovizId or Old.FirmaId <> New.FirmaId) then
    update IplikIrsaliyeDetay Set Tarih = New.IrsaliyeTarihi, FirmaId = New.FirmaId, NeredenUpdate = 1, LinkIslemi = 1 where IplikIrsaliyeDetay.IplikIrsaliyeId = new.IplikIrsaliyeId and IplikIrsaliyeDetay.NeredenGirildi = 1;

  if (Old.KdvDahil <> New.kdvDahil) then /* KdvHariFiyat yeniden hesaplansÄ±n */
    update IplikIrsaliyeDetay Set KdvHaricFiyat = 0, NeredenUpdate = 1, LinkIslemi = 1 where IplikIrsaliyeDetay.IplikIrsaliyeId = new.IplikIrsaliyeId;

  Select shtAlis, shtSatis, shtAlisIade, shtSatisIade, shtUretimeCikis, shtUretim, shtUretimIade, ShtTelefGirisi, shtUretimCikisBoya, shtUretimBoya, shtSarimaCikis, shtFasonAlis, shtUretimKirpinti from iplikStokHareketTipiGetir
    into :shtAlis, :shtSatis, :shtAlisIade, :shtSatisIade, :shtUretimeCikis, :shtUretim, :shtUretimIade, :ShtTelefGirisi, :shtUretimCikisBoya, :shtUretimBoya, :shtSarimaCikis, :shtFasonAlis, :shtUretimKirpinti;

  if (New.HareketTipi <> shtUretim and New.HareketTipi <> shtSarimaCikis) then
  if (Old.FiiliSevkTarihi <> New.FiiliSevkTarihi) then
  begin
    if (new.HareketTipi = shtAlis or new.HareketTipi = shtFasonAlis or new.HareketTipi = shtSatisIade or new.HareketTipi = shtUretimIade or new.HareketTipi = ShtTelefGirisi or new.HareketTipi = shtUretimBoya or new.HareketTipi = shtUretimKirpinti) then
       Update IplikStok Set GirisTarihi = New.FiiliSevkTarihi where IplikStok.IplikIrsaliyeId = New.IplikIrsaliyeId;
    else
    begin
      For Select IplikIrsaliyeDetayId from IplikIrsaliyeDetay where IplikIrsaliyeDetay.IplikIrsaliyeId = New.IplikIrsaliyeId into :IplikIrsaliyeDetayId do
      begin
        Update IplikStok Set CikisTarihi = New.FiiliSevkTarihi where IplikStok.CikisIplikIrsaliyeDetayId = :IplikIrsaliyeDetayId;
      end
    end
  end
end
