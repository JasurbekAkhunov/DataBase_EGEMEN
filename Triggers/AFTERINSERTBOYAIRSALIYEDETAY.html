CREATE TRIGGER AFTERINSERTBOYAIRSALIYEDETAY FOR BOYAIRSALIYEDETAY
ACTIVE AFTER INSERT POSITION 0
as
declare variable biSatis                integer;
declare variable biAlisIade             integer;
declare variable biSatisIade            integer;
declare variable FVerilenSiparisDetayId integer;
begin
  Select biSatis, biAlisIade, biSatisIade from BoyaIrsaliyeHareketTipiGetir into :biSatis, :biAlisIade, :biSatisIade;

  if (new.RafDetayId is not null) then
  begin
    if (new.FasonHamIade = 1 or new.HareketTipiId = biAlisIade) then
    begin
      update RafDetay set IadeEdilecekKilo = IadeEdilecekKilo - new.Kilo,
                          IadeEdilenKilo = IadeEdilenKilo + new.Kilo,
                          IadeEdilecekMetre = IadeEdilecekMetre - new.Metre,
                          IadeEdilenMetre = IadeEdilenMetre + new.Metre,
                          IadeEdilecekKapAdedi = IadeEdilecekKapAdedi - new.KapAdedi,
                          IadeEdilenKapAdedi = IadeEdilenKapAdedi + new.KapAdedi,
                          IadeEdilecekAdet = IadeEdilecekAdet - new.Adet,
                          IadeEdilenAdet = IadeEdilenAdet + new.Adet
      where RafDetay.RafDetayId = new.RafDetayId;
    end
    else
    begin
      update RafDetay set SatisKilo = SatisKilo + new.Kilo,
                          SatisMetre = SatisMetre + new.Metre
      where RafDetay.RafDetayId = new.RafDetayId;
    end
  end
  else if (new.PartiSinifId is not null and new.LinkIslemi = 0 and new.HareketTipiId = biSatis) then
  begin
    update PartiStok set PartiStok.NetSevkKilo = PartiStok.NetSevkKilo +  new.Kilo ,
                         PartiStok.NetSevkMetre = PartiStok.NetSevkMetre + new.Metre,
                         PartiStok.BrutSevkMetre = PartiStok.BrutSevkMetre + new.BrutMetre,
                         PartiStok.NetSevkKapAdedi = PartiStok.NetSevkKapAdedi +  new.KapAdedi,
                         PartiStok.BrutSevkKilo = PartiStok.BrutSevkKilo +  new.BrutKilo,
                         PartiStok.NetSevkAdet = PartiStok.NetSevkAdet +  new.Adet
           where PartiStok.PartiSinifId = new.PartiSinifId;
  end
  else if (new.HareketTipiId = biSatisIade) then
  begin
    update PartiStok set PartiStok.NetIadeKilo = PartiStok.NetIadeKilo +  new.Kilo ,
                         PartiStok.NetIadeMetre = PartiStok.NetIadeMetre + new.Metre,
                         PartiStok.NetIadeKapAdedi = PartiStok.NetIadeKapAdedi +  new.KapAdedi,
                         PartiStok.BrutIadeMetre = PartiStok.BrutIadeMetre + new.BrutMetre,
                         PartiStok.BrutIadeKilo = PartiStok.BrutIadeKilo +  new.BrutKilo
           where PartiStok.PartiSinifId = new.PartiSinifId;
  end

  if (new.HareketTipiId = :biAlisIade) then
  begin
    Select HamStokMaster.VerilenSiparisDetayId from RafDetay
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      where RafDetay.RafDetayId = new.RafDetayId and HamStokMaster.VerilenSiparisDetayId is not null
      into :FVerilenSiparisDetayId;

    if (FVerilenSiparisDetayId is not null) then
       Update VerilenSiparisDetay Set IadeEdilenMiktar = IadeEdilenMiktar + New.BrutMiktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = :FVerilenSiparisDetayID;
  end
end
