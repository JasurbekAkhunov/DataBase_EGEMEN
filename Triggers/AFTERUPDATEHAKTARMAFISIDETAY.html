CREATE TRIGGER AFTERUPDATEHAKTARMAFISIDETAY FOR HAKTARMAFISIDETAY
ACTIVE AFTER UPDATE POSITION 0
as
declare variable FMsg         varchar(255);
begin
  if (new.Kilo <> Old.Kilo or New.IadeEdilenKilo <> Old.IadeEdilenKilo) then
  begin
    Update RafDetay set AktarmayaCikanKilo = AktarmayaCikanKilo - Old.Kilo + new.Kilo,
                        AktarmadanIadeEdienKilo = AktarmadanIadeEdienKilo - Old.IadeEdilenKilo + New.IadeEdilenKilo
                         where rafDetay.RafDetayId = New.RafDetayId;

    Update RafDetay Set Kilo = Kilo - Old.Kilo + New.Kilo + Old.IadeEdilenKilo - new.IadeEdilenKilo where RafDetay.HamStokDetayId = new.HamStokDetayId and RafDetay.Aktarma = 1;
  end
end
