Definition

CREATE TRIGGER AFTERUPDATEBOYATEKLIFDETAY FOR BOYATEKLIFDETAY
ACTIVE AFTER UPDATE POSITION 0
as
declare variable AsamaId                    integer;
declare variable ADovizId                   varchar(5);
declare variable ABirimFiyat                Double Precision;
declare variable FADovizId                  varchar(5);
declare variable FABirimFiyat               Double Precision;
declare variable FDovizKuru                 Double Precision;
declare variable TeklifFirmaId              varchar(20);
declare variable TeklifDovizId              Varchar(5);
declare variable TeklifTarihi               TimeStamp;
declare variable AKullanimOrani             Double Precision;
declare variable AHamId                     integer;
declare variable tikBoya                     SmallInt;
declare variable tikHazirGiyim               SmallInt;
declare variable IslemKaynagi                SmallInt;
declare variable KaliteRecetesiniKullan      SmallInt;
begin
  Select tikBoya, tikHazirGiyim from TeklifIslemKaynagiGetir into :tikBoya, :tikHazirGiyim;

  select Sistem.KaliteRecetesiniKullan from Sistem into :KaliteRecetesiniKullan;

  Select FirmaId, IslemKaynagi, DovizId, TeklifTarihi from BoyaTeklif where BoyaTeklif.BoyaTeklifKayitID = new.BoyaTeklifKayitID
         into :TeklifFirmaId, :IslemKaynagi, :TeklifDovizId, :TeklifTarihi;

  if (Old.LinkIslemi <> 2 and IslemKaynagi = tikBoya) then
  begin

    if (Old.SiparisProsesId <> New.SiparisProsesId) then
    begin
      delete from BoyaTeklifAsama where BoyaTeklifAsama.BoyaTeklifDetayId = New.BoyaTeklifDetayId;

      for Select SiparisProsesAsamalari.AsamaId, Asama.BirimFiyat, Asama.DovizId, FirmaAsama.BirimFiyat, FirmaAsama.DovizId from SiparisProsesAsamalari
        left join Asama On Asama.AsamaId = SiparisProsesAsamalari.AsamaId
        left join FirmaAsama On FirmaAsama.FirmaId = :TeklifFirmaId and FirmaAsama.AsamaId = SiparisProsesAsamalari.AsamaId
        where SiparisProsesAsamalari.SiparisProsesId= new.SiparisProsesId order by sira
        into :AsamaId, :ABirimFiyat, :ADovizId, :FABirimFiyat, :FADovizId  do
      begin
        if (FADovizId is not null) then
        begin
          ADovizID = FADovizId;
          ABirimFiyat = FABirimFiyat;
        end

        if (TeklifDovizId <> ADovizId) then
        begin
          select DovizKuru from DovizKuruGetir(:ADovizId, :TeklifDovizId, :TeklifTarihi) into :FDovizKuru;
          if (FDovizKuru = 0) then exception DovizKuruGirisiYapilmamis;
          ABirimFiyat = ABirimFiyat * FDovizKuru;

          Select Fiyat from FormataGoreFiyatYuvarla(:ABirimFiyat, 'BOYA', 'BIRIMFIYAT') into :ABirimFiyat;
        end

        insert into BoyaTeklifAsama(BoyaTeklifKayitId, BoyaTeklifDetayId, AsamaId, BirimFiyat, InsertKullaniciId, KullaniciId)
                      values(new.BoyaTeklifKayitId, New.BoyaTeklifDetayId, :AsamaId, :ABirimFiyat, new.InsertKullaniciId, new.KullaniciId);
      end
    end

    if (new.Fason = 0) then
    begin
      if (KaliteRecetesiniKullan = 0) then
      begin
        if (Old.HamId <> New.HamId) then
        begin
           delete from BoyaTeklifHammadde where BoyaTeklifHammadde.BoyaTeklifDetayId = New.BoyaTeklifDetayId;

          for Select HamStokRecete.ReceteHamId, Ham.AlisFiyati, Ham.AlisFiyatiDovizId, HamStokRecete.KullanimOrani from HamStokRecete
            left join Ham On Ham.HamId = HamStokRecete.ReceteHamId
            where HamStokRecete.HamId = new.HamId
            into :AHamId, :ABirimFiyat, :ADovizId, :AKullanimOrani  do
          begin
            if (TeklifDovizId <> ADovizId) then
            begin
              select DovizKuru from DovizKuruGetir(:ADovizId, :TeklifDovizId, :TeklifTarihi) into :FDovizKuru;
              if (FDovizKuru = 0) then exception DovizKuruGirisiYapilmamis;
              ABirimFiyat = ABirimFiyat * FDovizKuru;

              Select Fiyat from FormataGoreFiyatYuvarla(:ABirimFiyat, 'BOYA', 'BIRIMFIYAT') into :ABirimFiyat;
            end

            insert into BoyaTeklifHammadde(BoyaTeklifKayitId, BoyaTeklifDetayId, HamId, KullanimOrani, BirimFiyat, InsertKullaniciId, KullaniciId)
                          values(new.BoyaTeklifKayitId, New.BoyaTeklifDetayId, :AHamId, :AKullanimOrani, :ABirimFiyat, new.InsertKullaniciId, new.KullaniciId);
          end
        end
      end
      else if (KaliteRecetesiniKullan = 1) then
      begin
        if (Old.KaliteId <> New.KaliteId) then
        begin
          delete from BoyaTeklifHammadde where BoyaTeklifHammadde.BoyaTeklifDetayId = New.BoyaTeklifDetayId;

          for Select KaliteRecete.ReceteHamId, Ham.AlisFiyati, Ham.AlisFiyatiDovizId, KaliteRecete.KullanimOrani from KaliteRecete
            left join Ham On Ham.HamId = KaliteRecete.ReceteHamId
            where KaliteRecete.KaliteId = new.KaliteId
             into :AHamId, :ABirimFiyat, :ADovizId, :AKullanimOrani  do
          begin
            if (TeklifDovizId <> ADovizId) then
            begin
              select DovizKuru from DovizKuruGetir(:ADovizId, :TeklifDovizId, :TeklifTarihi) into :FDovizKuru;
              if (FDovizKuru = 0) then exception DovizKuruGirisiYapilmamis;
              ABirimFiyat = ABirimFiyat * FDovizKuru;

              Select Fiyat from FormataGoreFiyatYuvarla(:ABirimFiyat, 'BOYA', 'BIRIMFIYAT') into :ABirimFiyat;
            end

            insert into BoyaTeklifHammadde(BoyaTeklifKayitId, BoyaTeklifDetayId, HamId, KullanimOrani, BirimFiyat, InsertKullaniciId, KullaniciId)
                          values(new.BoyaTeklifKayitId, New.BoyaTeklifDetayId, :AHamId, :AKullanimOrani, :ABirimFiyat, new.InsertKullaniciId, new.KullaniciId);
          end
        end
      end
    end
  end
end
