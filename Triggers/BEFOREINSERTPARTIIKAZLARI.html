CREATE TRIGGER BEFOREINSERTPARTIIKAZLARI FOR PARTIIKAZLARI
ACTIVE BEFORE INSERT POSITION 0
as
declare variable kayitsayisi integer;
begin
  if (new.PartiIkazTipi = 0) then
  begin
    select count(*) from PartiStok where PartiStok.PartiKayitId = new.PartiKayitId and Durumu < 7 into :KayitSayisi;
    if (KayitSayisi = 0) then exception PartiZatenBitmis;
  end

  select min(GirisTarihi) from PartiAsamaStok where PartiAsamaStok.PartiKayitId = new.PartiKayitId into new.PartiBaslamaTarihi;

  Select Parti.PartiYil, Parti.PartiId from Parti where Parti.PartiKayitId = new.PartiKayitId into new.PartiYil, new.PartiId;
end
