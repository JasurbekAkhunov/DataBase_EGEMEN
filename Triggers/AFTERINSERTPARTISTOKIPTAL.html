Definition

CREATE TRIGGER AFTERINSERTPARTISTOKIPTAL FOR PARTISTOKIPTAL
ACTIVE AFTER INSERT POSITION 0
as

declare variable HamId integer;
declare variable FBoyaSiparisDetayId integer;
declare variable FBoyaSiparisBaskiId integer;
declare variable FOrmeSiparisDetayId integer;
declare variable FFirmaId       varchar(20);
declare variable FFason         integer;
declare variable FKilo          Numeric(12,3);
declare variable FMetre         Numeric(12,3);
declare variable FNow           Date;
declare variable HamStokBoyaSiparisDetayId integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable IptalPartiFireOrani        Double Precision;
declare variable FPartiKilo             Numeric(12,3);
declare variable FPartiMetre            Numeric(12,3);
declare variable DepoIadeSonCikisMiktarKullan integer;
declare variable HamTipiId     integer;
declare variable FSonCikisKilo                  Numeric(12,3);
declare variable FSonCikisMetre                 Numeric(12,3);
declare variable Kilo                  Numeric(12,3);
declare variable Metre                 Numeric(12,3);
declare variable FHamStokDetayId        integer;
declare variable FHamStokIrsaliyeId     integer;
declare variable FPartiKayitId     integer;
declare variable PMiktarDegisinceRecBoyaDegis     smallint;
begin
  Delete from PartiPaket where PartiPaket.PartiSinifId = New.PartiSinifId;

  Update PartiAsamaStok set AsamayaDahil = 0 where PartiAsamaStok.PartiSinifId = New.PartiSinifId and PartiAsamaStok.GirisTarihi is null;

  Update PartiStok Set PartiStokIptalKayitId = New.PartiStokIptalKayitId, ProxyId = null where PartiStok.PartiSinifId = New.PartiSinifId;

  Select PartiStok.HamId, PartiStok.BoyaSiparisDetayId, PartiStok.BoyaSiparisBaskiId, BoyaSiparisDetay.OrmeSiparisDetayId, Parti.FirmaId, Parti.Fason, PartiStok.Kilo, PartiStok.Metre from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiSinifId = New.PartiSinifId
    into :HamId, :FBoyaSiparisDetayId, :FBoyaSiparisBaskiId, :FOrmeSiparisDetayId, :FFirmaId, :FFason, :FKilo, :FMetre;

  FNow = new.IptalTarihi;

  if (FBoyaSiparisDetayId is not null) then
  begin
    if (FBoyaSiparisBaskiId is null) then
      Update BoyaSiparisDetay Set IptalKilo = IptalKilo + :FKilo, IptalMetre = IptalMetre + :FMetre, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayId = :FBoyaSiparisDetayId;
    else
      Update BoyaSiparisDetay Set IptalKiloBaski = IptalKiloBaski + :FKilo, IptalMetreBaski = IptalMetreBaski + :FMetre, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayId = :FBoyaSiparisDetayId;
  end

  if (FBoyaSiparisBaskiId is not null) then
    Update BoyaSiparisBaski Set IptalKilo = IptalKilo - :FKilo, IptalMetre = IptalMetre - :FMetre where BoyaSiparisBaski.BoyaSiparisBaskiId = :FBoyaSiparisBaskiId;

  if (New.DesenTasarimIptal = 0) then
    execute procedure HamStokKumilatifGuncelle(:FFirmaId, :FFason, 1, Null, :HamId, Null, :FNow, Null, Null, :FKilo, :FMetre);

  if (FOrmeSiparisDetayID is not null and FBoyaSiparisDetayId is not null) then
  begin
    For Select PartiStokDetay.Kilo, HamStokMaster.BoyaSiparisDetayId from PartiStokDetay
      left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
      where PartiStokDetay.PartiSinifID = New.PartiSinifID and
            HamStokMaster.BoyaSiparisDetayId is not null and
            (HamStokMaster.BoyaSiparisDetayId <> :FBoyaSiparisDetayId)
      into :FKilo, :HamStokBoyaSiparisDetayId do
    begin
      Update BoyaSiparisDetay Set BaskaSipariseKaydirilan = BaskaSipariseKaydirilan - :FKilo, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayID = :HamStokBoyaSiparisDetayId;
      Update BoyaSiparisDetay Set BaskaSiparistenKaydirilan = BaskaSiparistenKaydirilan - :FKilo, LinkIslemi = 1 where BoyaSiparisDetay.BoyaSiparisDetayID = :FBoyaSiparisDetayId;
    end
  end

  FMetre = 0;
  FKilo = 0;
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  if (New.DesenTasarimIptal = 0) then
  if (New.TeslimAlanKullaniciId is not null) then
  begin
    Select DepoIadeSonCikisMiktarKullan from sistem where sistemid = 1 into :DepoIadeSonCikisMiktarKullan;

    if (DepoIadeSonCikisMiktarKullan = 1) then
       Select hamTipiId, Kilo, Metre, SonCikisKilo, SonCikisMetre from PartiStok where PartiStok.PartiSinifID = New.PartiSinifId
       into :HamTipiId, :FPartiKilo, :FPartiMetre, :FSonCikisKilo, :FSonCikisMetre;

    if (FSonCikisKilo is null) then FSonCikisKilo = 0;
    if (FSonCikisMetre is null) then FSonCikisMetre = 0;

    if ((HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) and FSonCikisKilo <= 0) then DepoIadeSonCikisMiktarKullan = 0;
    else if (HamTipiId = 2 and FSonCikisMetre <= 0) then DepoIadeSonCikisMiktarKullan = 0;

    For Select RafDetay.HamStokDetayId, RafDetay.HamStokIrsaliyeId, Sum(PartiStokDetay.Kilo), Sum(PartiStokDetay.Metre) from PartiStokDetay
      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = rafDetay.HamStokDetayId
      where PartiStokDetay.PartiSinifId = new.PartiSinifId
      group by RafDetay.HamStokDetayId, RafDetay.HamStokIrsaliyeId
      into :FHamStokDetayId, :FHamStokIrsaliyeId, :FKilo, :FMetre do
    begin
      if (DepoIadeSonCikisMiktarKullan = 0) then
      begin
        Kilo = FKilo;
        Metre = FMetre;
        IptalPartiFireOrani = 0;
      end
      else
      begin
        if (FPartiKilo <> 0) then
          Kilo = FSonCikisKilo * FKilo / FPartiKilo;
        else
          Kilo = 0;

        if (FPartiMetre <> 0) then
          Metre = FSonCikisMetre * FMetre / FPartiMetre;
        else
          Metre = 0;

        IptalPartiFireOrani = ((FKilo - Kilo) / FKilo ) * 100;
      end

      insert into RafDetay(HamStokIrsaliyeId, HamStokDetayId, RafId, KaliteNo, Kilo, Metre, IptalPartiFireOrani, PartiStokIptalKayitId, InsertKullaniciId)
                   values(:FHamStokIrsaliyeId, :FHamStokDetayId, new.RafId, new.KaliteNo, :Kilo, :Metre, :IptalPartiFireOrani, new.PartiStokIptalKayitId, New.TeslimAlanKullaniciId);

    end
  end


  Select PartiStok.partikayitid from PartiStok where PartiStok.PartiSinifId = New.PartiSinifId into :FPartiKayitId;

  Select PMiktarDegisinceRecBoyaDegis from Sistem where SistemID = 1 into :PMiktarDegisinceRecBoyaDegis;
  if (PMiktarDegisinceRecBoyaDegis = 1) then
  begin
    Update PartiAsamalariReceteBoyaMiktar set Miktar = 0 where PartiAsamalariReceteBoyaMiktar.PartiKayitID = :FPartiKayitId;
  end
end
