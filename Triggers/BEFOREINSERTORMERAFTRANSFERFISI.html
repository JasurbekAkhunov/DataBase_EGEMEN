CREATE TRIGGER BEFOREINSERTORMERAFTRANSFERFISI FOR ORMERAFTRANSFERFISI
ACTIVE BEFORE INSERT POSITION 0
as
begin
  update OrmeRafTransferFisiNumarator set Id = new.FisNo where Yil = new.FisYil and Id < New.FisNo;
  if (new.IrsaliyeSeriNo is Not Null and new.IrsaliyeNo is Not Null) then
  begin
    update IrsaliyeNumarator set IrsaliyeNo = new.IrsaliyeNo where SeriNo = new.IrsaliyeSeriNo;
  end

  if (new.OrmeRafTransferFisiId is null) then new.OrmeRafTransferFisiId = gen_id(OrmeRafTransferFisi_Id_Gen, 1);
end
