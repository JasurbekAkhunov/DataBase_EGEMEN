CREATE TRIGGER BEFOREUPDATEKIRPINTITASNIFDETAY FOR KIRPINTITASNIFDETAY
ACTIVE BEFORE UPDATE POSITION 0
as
Declare Variable GirisIplikIrsaliyeId Integer;
begin
  new.DegisimTarihi = current_timestamp(0);

  Delete from IplikIrsaliyeDetay where IplikIrsaliyeDetay.KirpintiTasnifDetayId = Old.KirpintiTasnifDetayId;

  Select GirisIplikIrsaliyeId from KirpintiTasnifYap(new.KirpintiTasnifKayitId, new.KirpintiTasnifDetayId, new.HamId, New.RenkId, new.IplikDepoId, New.Kilo, New.KullaniciId) into :GirisIplikIrsaliyeId;

end
