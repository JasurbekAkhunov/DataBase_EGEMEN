Definition

CREATE TRIGGER BEFOREDELETEIPLIKIRSALIYEDETAY FOR IPLIKIRSALIYEDETAY
ACTIVE BEFORE DELETE POSITION 0
as
declare variable KantarKayitId          integer;
declare variable OnayDurumu             integer;
declare variable shtSatis               integer;
declare variable shtSatisIade           integer;
declare variable shtUretimeCikis        integer;
declare variable shtAlis                integer;
declare variable shtFasonAlis           integer;
declare variable shtAlisIade            integer;
declare variable shtSarimaCikis integer;
declare variable shtBukumeCikis integer;
declare variable SonDonemBitisTarihi TimeStamp;
begin
  select shtSatis, shtSatisIade, shtUretimeCikis, shtSarimaCikis, shtBukumeCikis, shtAlis, shtFasonAlis, shtAlisIade from IplikStokHareketTipiGetir into :shtSatis, :shtSatisIade, :shtUretimeCikis, :shtSarimaCikis, :shtBukumeCikis, :shtAlis, :shtFasonAlis, :shtAlisIade;

  if (Old.IplikFaturaId is not null) then exception BelgeFaturalastirilmisSilinemez;

  Select KantarKayitId from IplikIrsaliye where IplikIrsaliye.IplikIrsaliyeId = Old.IplikIrsaliyeId into :KantarKayitId;

  if (KantarKayitId is not null) then
  begin
    Select OnayDurumu from Kantar where Kantar.KantarKayitID = :KantarKayitId into :OnayDurumu;
    if (OnayDurumu = 1) then exception KantarOnayiVerilmisDegisemez;
  end

  Update IplikStok set CikisIplikIrsaliyeDetayId = Null where IplikStok.CikisIplikIrsaliyeDetayId = Old.IplikIrsaliyeDetayId; /* cikislari iptal et*/
  delete from IplikStok where IplikStok.IplikIrsaliyeDetayId = Old.IplikIrsaliyeDetayId; /* giriş ise girişleri sil */

  if (Old.HareketTipi = ShtSatis) then
  begin
    if (Old.IplikTalepDetayID is not null) then
    if (Old.Tamir = 0) then
    begin
      Update IplikTalepDetay Set TeslimEdilenMiktar = TeslimEdilenMiktar - Old.Miktar,
                                 TeslimEdilenNetMiktar = TeslimEdilenNetMiktar - Old.NetMiktar,
                                 TeslimEdilenBobinSayisi = TeslimEdilenBobinSayisi - Old.BobinSayisi,
                                 LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = Old.IplikTalepDetayId;
    end
    else
    begin
      Update IplikTalepDetay Set TamirTeslimEdilenMiktar = TamirTeslimEdilenMiktar - Old.Miktar,
                                 TamirTeslimEdilenNetMiktar = TamirTeslimEdilenNetMiktar - Old.NetMiktar,
                                 TamirTeslimEdilenBobinSayisi = TamirTeslimEdilenBobinSayisi - Old.BobinSayisi,
                                 LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = Old.IplikTalepDetayId;
    end
  end
  if (Old.HareketTipi = ShtSatisIade) then
  begin
    if (Old.IplikTalepDetayID is not null) then
      Update IplikTalepDetay Set IadeGelenMiktar = IadeGelenMiktar - Old.Miktar, LinkIslemi = 1 where IplikTalepDetay.IplikTalepDetayID = Old.IplikTalepDetayId;
  end
  else if (Old.HareketTipi = shtUretimeCikis and Old.IplikTelefIsEmriDetayId is not null) then
  begin
    Update IplikTelefIsEmriDetay set CikisMiktari = CikisMiktari - Old.Miktar where IplikTelefIsEmriDetay.IplikTelefIsEmriDetayID = Old.IplikTelefIsEmriDetayId;
  end
  else if (Old.HareketTipi = shtSarimaCikis) then
  begin
    Update SarimStok set KapAdedi = KapAdedi - Old.KapAdedi, Miktar = Miktar - Old.Miktar where SarimStok.IplikLotId = Old.IplikLotId;
  end
  else if (Old.HareketTipi = shtBukumeCikis) then
  begin
    Update BukumIsEmri set CikisMiktari = CikisMiktari - Old.Miktar, LinkIslemi = 1 where BukumIsEmri.BukumIsEmriKayitID = Old.BukumIsEmriKayitId;
  end
  else if (Old.HareketTipi = shtAlis or Old.HareketTipi = shtFasonAlis) then
  begin
    if (Old.VerilenSiparisDetayId is not null) then
      Update VerilenSiparisDetay Set GelenMiktar = GelenMiktar - Old.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = Old.VerilenSiparisDetayId;
  end
  else if (old.HareketTipi = shtAlisIade) then
  begin
    if (Old.VerilenSiparisDetayId is not null) then
      Update VerilenSiparisDetay Set IadeEdilenMiktar = IadeEdilenMiktar - Old.Miktar, LinkIslemi = 1 where VerilenSiparisDetay.VerilenSiparisDetayId = Old.VerilenSiparisDetayId;
  end

  select max(BitisTarihi) from DonemKapatma where Bolum = 3 and OnayKullaniciId is not Null into :SonDonemBitisTarihi;
  if (SonDonemBitisTarihi is not Null and SonDonemBitisTarihi > old.Tarih) then
    exception DonemKapaliHSilinemez;
end
