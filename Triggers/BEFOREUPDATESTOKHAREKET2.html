declare Variable IFaturaKesinceFiyatGuncelle integer;
declare Variable FaturaDovizId varchar(5);
declare Variable SipDovizId varchar(5);
declare Variable SipBirimFiyat Double Precision;
declare Variable FaturaFirmaID varchar(20);
declare Variable FaturaTarihi TimeStamp;
declare Variable DovizKuru Double Precision;
declare Variable OrmeTalimattaReceteFiresiYap  SmallInt;
declare Variable OtomatikUCikislardaReceteFire SmallInt;
declare variable FTahminiFireOrani Double Precision;
declare variable FTalimatReceteFireOrani Double Precision;
declare variable UrunHamId integer;
declare variable FHamTipiId SmallInt;
declare variable CozguIsEmriStokFirmaId FirmaKoduSize;

begin
  Select OrmeFirmaKodumuz, DokumaFirmaKodumuz, FirmaKodumuz, IplikFirmaKodumuz, OrmeFatTevkifatDetayiEtkile, IFaturaKesinceFiyatGuncelle, OrmeTalimattaReceteFiresiYap, OtomatikUCikislardaReceteFire, DokumaUretimdeOtomatikCikis from Sistem
   into :OrmeFirmaKodumuz, :DokumaFirmaKodumuz, :BoyaFirmaKodumuz, :IplikFirmaKodumuz, :OrmeFatTevkifatDetayiEtkile, :IFaturaKesinceFiyatGuncelle, :OrmeTalimattaReceteFiresiYap, :OtomatikUCikislardaReceteFire, :DokumaUretimdeOtomatikCikis;

  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, shtKaydirmaCikis, shtKaydirmaGiris from OrmeStokHareketTipiGetir
   into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
        :shtFasonUretim, :ShtUretim, :shtKaydirmaCikis, :shtKaydirmaGiris;

  Select assSatis, assFasonArtiSatis, assFason from OrmeSiparisSekliGetir into :assSatis, :assFasonArtiSatis, :assFason;

  Select uitBukum, uitOrme, uitBoya, uitDokuma, uitCozgu, uitHasil, uitIndigo from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya, :uitDokuma, :uitCozgu, :uitHasil, :uitIndigo;

  if (new.HareketTipi = shtUretim and new.DokumaHareketi = 1 and new.LeventNo is not null and (new.LeventNo <> old.LeventNo or old.LeventNo is null)) then
  begin
    if (new.CozguIsEmriKayitId is not null) then
    begin
      select count(*) from OrmeRaf where OrmeRaf.LeventNo = new.LeventNo and OrmeRaf.LeventSiraNo = new.LeventSiraNo and OrmeRaf.StokVar = 1 into :KS;
      if (KS > 0) then
        Exception BuLeventteHalaStokVar;
    end
    else if (new.DokumaPlanlamaId is not null) then
    begin
      select count(*) from StokHareket where StokHareket.HareketTipi = :shtUretimeCikis and StokHareket.DokumaPlanlamaId = new.DokumaPlanlamaId and StokHareket.LeventNo = new.LeventNo into :KS;
      if (KS = 0) then
        Exception BuLeventBuPlanlamayaCikilmamis;
    end
  end

  if (New.OrmeIrsaliyeId is not null and New.LinkIslemi = 0) then
  begin
    select OrmeIrsaliye.TransferEdildi from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = New.OrmeIrsaliyeId into :FOrmeIrsaliyeTransferEdildi;
    if (FOrmeIrsaliyeTransferEdildi = 1) then exception TrasferEdilmisDegistirilmez;
  end

  if (New.DokumaHareketi = 1) then
    FFirmaKodumuz = DokumaFirmaKodumuz;
  else
    FFirmaKodumuz = OrmeFirmaKodumuz;


  if ( new.HareketTipi = shtAlisIade and New.FirmaID = :IplikFirmaKodumuz) then
  if (Old.Tamir <> new.Tamir) then
  begin
    Select Count(*) from StokHareket
      where StokHareket.OrmeIrsaliyeID = New.OrmeIrsaliyeID and StokHareket.Tamir <> New.Tamir and StokHareket.StokHareketID <> New.StokHareketID
      into :KS;

    if (KS > 0) then exception TamirNormalKarisikOlamaz;
  end


  if (Old.Miktar <> New.Miktar and New.OrmeIrsaliyeID is not null) then
  begin
    Select KantarKayitId from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = new.OrmeIrsaliyeId into :KantarKayitId;

    if (KantarKayitId is not null) then
    begin
      Select OnayDurumu from Kantar where Kantar.KantarKayitID = :KantarKayitId into :OnayDurumu;
      if (OnayDurumu = 1) then exception KantarOnayiVerilmisDegisemez;
    end
  end

  if (Old.GirisKKBitirenKullaniciId is null and New.GirisKKBitirenKullaniciId is not null) then
  begin
    New.GirisKKBitisTarihi = Current_TimeStamp(0);
  end
  else if (Old.GirisKKBitirenKullaniciId is not null and New.GirisKKBitirenKullaniciId is null) then New.GirisKKBitisTarihi = Null;


  if (New.VadeGun = 0 and New.VerilenSiparisDetayID is not Null and new.HareketTipi = shtAlis and New.NeredenGirildi = 1) then
    Select VadeGun from VerilenSiparisDetay where VerilenSiparisDetayID = New.VerilenSiparisDetayID into New.VadeGun;

  if (New.LinkIslemi = 0) then
  begin
    if ( (new.HareketTipi = shtUretimeCikis or new.HareketTipi = shtFasonUretimeCikis) and
          New.FirmaID = :BoyaFirmaKodumuz and IntKarsilastir(Old.OrmeSiparisDetayID, New.OrmeSiparisDetayID) = 0 and
          new.OrmeSiparisDetayID is not null) then
    begin
      if (New.NeredenGirildi = 0) then select Sinif from OrmeFis where OrmeFis.OrmeFisId = new.OrmeFisId into :EvrakSinif;
      else if (New.NeredenGirildi = 1) then select Sinif from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = new.OrmeIrsaliyeId into :EvrakSinif;
      else if (New.NeredenGirildi = 2) then select Sinif from OrmeFatura where OrmeFatura.OrmeFaturaId = new.OrmeFaturaId into :EvrakSinif;

      Select OrmeSiparisDetay.Sinif from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayID = New.OrmeSiparisDetayid into :SiparisSinif;

      if (SiparisSinif <> EvrakSinif) then exception IrsaliyeIleHamStokSinifTutmuyor;
    end

    if (new.OrmeMakinaId = '') then New.OrmeMakinaId = null;
    if (new.DokumaMakinaId = '') then New.DokumaMakinaId = null;


    if (IntKarsilastir(Old.BoyaSiparisKayitId, New.BoyaSiparisKayitId) = 0 and New.BoyaSiparisKayitId is not null) then
    begin
      Select BoyaSiparis.KumasTemin from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitID = new.BoyaSiparisKayitID into :FKumasTeminSiparisi;

      if (FKumasTeminSiparisi = 1) then
      begin
        if (new.HareketTipi = shtSatis or
            new.HareketTipi = shtSatisIade or
            new.HareketTipi = shtFasonAlis) then exception KumasTeminHareketTipiHatali;

        if (new.UretimIslemTipiID is not null) then
        if (new.UretimIslemTipiID <> uitBukum and new.UretimIslemTipiID <> uitOrme and new.UretimIslemTipiID <> uitCozgu and new.UretimIslemTipiID <> uitHasil and new.UretimIslemTipiID <> uitIndigo and new.UretimIslemTipiID <> uitDokuma) then exception KumasTeminHareketTipiHatali;

      end
    end

    if (New.HareketTipi = ShtSatis) then
    begin
      if (New.HamIade = 1) then
        if ((new.BrutMiktar is null) or (new.BrutMiktar = 0) ) then new.BrutMiktar = New.Miktar;
    end
    else
    begin
      if (new.HareketTipi <> shtKaydirmaCikis and new.HareketTipi <> shtKaydirmaGiris) then
        if ((new.HareketTipi <> shtFasonUretimeCikis) or (new.HareketTipi = shtFasonUretimeCikis and New.UretimIslemTipiId <> uitBoya) or new.BrutMiktar is Null) then
          if ((New.UretimIslemTipiId is null or New.UretimIslemTipiId <> uitBoya or New.HamIade = 1 or New.GirisCikis = -1) and (StrKarsilastir(New.FirmaID, :IplikFirmaKodumuz) = 0)) then
            if (old.GirisKKBitirenKullaniciId is null and new.GirisKKBitirenKullaniciId is null or new.DokumaHareketi <> 1) then
              if (new.BrutMiktar is null or new.BrutMiktar = 0) then
                new.BrutMiktar = New.Miktar;
    end

    if (new.HareketTipi = shtUretimeCikis) then
    begin
    if (DokumaUretimdeOtomatikCikis = 0) then
      begin
        if (new.OrmeMakinaId is null and new.DokumaHareketi = 0 and (New.UretimIslemTipiID = uitOrme or New.UretimIslemTipiID = uitBukum)) then exception UretimdeMakinaIdBosBirakilamaz;
        if (new.DokumaMakinaId is null and new.DokumaHareketi = 1 and New.UretimIslemTipiID = uitBukum) then exception UretimdeMakinaIdBosBirakilamaz;
        if (new.DokumaMakinaId is null and (New.UretimIslemTipiID = uitDokuma)) then exception UretimdeMakinaIdBosBirakilamaz;
    end
    end

    if (new.HareketTipi = shtUretim and new.HamIade = 0) then
    begin
      if (new.OrmeMakinaId is null and new.DokumaHareketi = 0 and (New.UretimIslemTipiID = uitOrme or New.UretimIslemTipiID = uitBukum)) then exception UretimdeMakinaIdBosBirakilamaz;
      if (new.DokumaMakinaId is null and new.DokumaHareketi = 1 and (New.UretimIslemTipiID = uitOrme or New.UretimIslemTipiID = uitBukum)) then exception UretimdeMakinaIdBosBirakilamaz;
      if (new.DokumaMakinaId is null and (New.UretimIslemTipiID = uitDokuma)) then exception UretimdeMakinaIdBosBirakilamaz;
    end

    Select FirmaId, OrmeSiparisSekliId, Fason from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitId = New.BoyaSiparisKayitId into :FSiparisFirmaId, :FOrmeSiparisSekliId, :BoyaSiparisFason;

    if (FOrmeSiparisSekliId is null) then
    begin
      if (BoyaSiparisFason = 0) then FOrmeSiparisSekliId = assSatis;
      else if (BoyaSiparisFason = 1) then FOrmeSiparisSekliId = assFason;
    end

    if (New.HareketTipi = ShtSatis or New.HareketTipi = shtSatisIade or New.HareketTipi = shtFasonUretim)  then FFaturaEdilecekBirim = New.FatEdilecekHamStokBirimID;
    else FFaturaEdilecekBirim = 0;

    if (FFaturaEdilecekBirim = 1) then
      New.FaturaMiktar = New.Metre;
    else if (FFaturaEdilecekBirim = 2) then
      New.FaturaMiktar = New.Adet;
    else
    begin
      if (New.FaturaMiktarTipi = 1) then /*Brüt miktar*/
        New.FaturaMiktar = New.BrutMiktar;
      else
        New.FaturaMiktar = New.Miktar;
    end

    if (FFaturaEdilecekBirim = 0 and New.HareketTipi = shtSatis and New.HamIade = 0 and New.BoyaSiparisDetayID is not null) then
    begin
      /*if (BoyaSiparisFason = 1 and new.OrmeSiparisDetayID is null) then New.FaturaMiktar = New.BrutMiktar;*/
      /*ORme siparise bagli olsa da brutu kullansin dedi Mustafa Bey*/
      if (BoyaSiparisFason = 1) then
      begin
        New.FaturaMiktar = New.BrutMiktar;
        New.FaturaMiktarTipi = 1;
      end
    end

    /*if (FFaturaEdilecekBirim = 0 and New.HareketTipi = shtFasonUretim and New.UretimIslemTipiId = uitBoya and New.HamIade = 0) then
    begin
      New.FaturaMiktar = New.BrutMiktar;
    end
    FaturaMiktarTipi koyunca bu istisnai durum kaldırıldı
    */

    if (new.DokumaHareketi = 0 and new.FirmaId = BoyaFirmaKodumuz and new.HareketTipi in (shtUretim, shtFasonUretim) and new.HamIade = 0) then
    begin
      /*ömer beyin ns orme isteklerinde vardi*/
      New.FaturaMiktar = New.BrutMiktar;
      New.Kdv = 20;
      if ((new.BirimFiyat = 0) or (new.BirimFiyat is null)) then
      begin
        select Coalesce(BoyaSiparisDetay.FasonBirimFiyati, 0), BoyaSiparisDetay.FasonBirimFiyatDovizId
        from BoyaSiparisDetay
        where BoyaSiparisDetay.BoyaSiparisDetayId = New.BoyaSiparisDetayID into new.BirimFiyat, new.DovizId;
      end
    end

    if (New.GirisCikis = -1) then
    if (new.OrmeRafId is null) then exception StokHareketKayitHatasi;

    if (new.UretimIslemTipiId = uitBoya and (New.HareketTipi = shtUretimeCikis or New.HareketTipi = shtFasonUretimeCikis )) then
    begin
      if (New.HareketTipi = shtUretimeCikis or new.FirmaID = BoyaFirmaKodumuz) then
      begin
        Select OrmeRaf.FasonStok from OrmeRaf where OrmeRaf.OrmeRafID = New.OrmeRafID into :FFasonStok;

        KS = 1;

        if (new.OrmeFisId is not null) then
        begin
          Select Count(*) from StokHareket
            left join OrmeRaf on OrmeRaf.OrmeRafID = StokHareket.OrmeRafID
            where StokHareket.OrmeFisID = New.OrmeFisID and StokHareket.StokHareketID <> New.StokHareketID and OrmeRaf.FasonStok <> :FFasonStok
            into :KS;
        end
        else
        begin
          Select Count(*) from StokHareket
            left join OrmeRaf on OrmeRaf.OrmeRafID = StokHareket.OrmeRafID
            where StokHareket.OrmeIrsaliyeID = New.OrmeIrsaliyeID and StokHareket.StokHareketID <> New.StokHareketID and OrmeRaf.FasonStok <> :FFasonStok
            into :KS;
        end
        if (KS > 0) then exception IrsaliyedeFasonSatisKarisik;
      end
    end

    if (New.HareketTipi = shtFasonAlis  or (new.HamIade = 1 and new.HareketTipi = shtSatis)) then
    begin
      if (New.BoyaSiparisKayitId is not null) then
      if (new.FirmaId <> FSiparisFirmaId or FSiparisFirmaId is null) then exception SiparisBuFirmayaAitDegil;
    end

    if (New.HareketTipi = shtFasonUretimeCikis or New.HareketTipi = shtUretimeCikis) then
    if (new.OrmePlanlamaId is not null) then
    begin
      Select Count(*) from OrmePlanlamaRecete where OrmePlanlamaRecete.OrmePlanlamaId = new.OrmePlanlamaId and OrmePlanlamaRecete.HamId = new.HamId into :KS;
      if (KS = 0) then exception UrunTalimatRecetedeYok;

      Select PlanlananMiktar, AsilabilirMiktar, MiktarAsilamaz from OrmePlanlama where OrmePlanlama.OrmePlanlamaId = new.OrmePlanlamaId into :PlanlamaMiktar, :AsilabilirMiktar, :MiktarAsilamaz;

      if (MiktarAsilamaz = 1) then
      begin
        Select Ham.HamTipiId from Ham where Ham.HamId = New.HamId into :FHamTipiId;
        Select Sum(KullanimOrani), Sum(CikisMiktari - IadeMiktari), Avg(TahminiFireOrani) from OrmePlanlamaRecete where OrmePlanlamaRecete.OrmePlanlamaId = new.OrmePlanlamaId and OrmePlanlamaRecete.HamId = new.HamId into :FKullanimOrani, :FCikisMiktari, :FTalimatReceteFireOrani;

        if (FKullanimOrani is Null) then FKullanimOrani = 100;
        PlanlamaMiktar = PlanlamaMiktar * FKullanimOrani / 100;

        if (FHamTipiId = 0 or FHamTipiId = 6) then
        begin
          Select OrmeSiparisDetay.TahminiFireOrani, OrmeSiparisDetay.HamId  from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayID = New.OrmeSiparisDetayid into :FTahminiFireOrani, :UrunHamId;

          if (OtomatikUCikislardaReceteFire = 1) then
            PlanlamaMiktar = PlanlamaMiktar / ((100 - FTalimatReceteFireOrani) / 100);
          else if (OrmeTalimattaReceteFiresiYap <> 1) then
          begin
            PlanlamaMiktar = PlanlamaMiktar / ((100 - FTahminiFireOrani) / 100);
          end
          else
          begin
            select first 1 FireOrani from HamStokRecete where HamId = :UrunHamId and ReceteHamId = new.HamId into :FTahminiFireOrani;
            PlanlamaMiktar = PlanlamaMiktar / ((100 - FTahminiFireOrani) / 100);
          end
        end


        PlanlamaMiktar =  PlanlamaMiktar + AsilabilirMiktar;
        FCikisMiktari = FCikisMiktari - Old.Miktar + new.Miktar;

        if (FCikisMiktari > PlanlamaMiktar) then  Exception OrmeTalimatMiktariAsildi;

      end
    end
    else if (new.BukumTalimatId is not null) then
    begin
      Select First 1 BukumTalimatReceteId from BukumTalimatRecete
        where BukumTalimatRecete.BukumTalimatId = new.BukumTalimatID and BukumTalimatRecete.HamId = new.HamId Into :FBukumTalimatReceteId;

      if (FBukumTalimatReceteId is null) then exception KayitHatasi ' Talimatta bu ürün yok..';

      PlanlamaMiktar = 0;
      AsilabilirMiktar = 0;
      MiktarAsilamaz = 0;
      Select TalimatMiktari, AsilabilirMiktar, MiktarAsilamaz from BukumTalimat where BukumTalimat.BukumTalimatID = new.BukumTalimatID into :PlanlamaMiktar, :AsilabilirMiktar, :MiktarAsilamaz;

      if (MiktarAsilamaz = 1) then
      begin
        Select Sum(KullanimOrani), Sum(CikisMiktari - IadeMiktari) from BukumTalimatRecete where BukumTalimatRecete.BukumTalimatID = new.BukumTalimatID and BukumTalimatRecete.HamId = new.HamId into :FKullanimOrani, :FCikisMiktari;

        if (FKullanimOrani is Null) then FKullanimOrani = 100;
        PlanlamaMiktar = PlanlamaMiktar * FKullanimOrani / 100;

        PlanlamaMiktar =  PlanlamaMiktar + AsilabilirMiktar;
        FCikisMiktari = FCikisMiktari - Old.Miktar + new.Miktar;

        if (FCikisMiktari > PlanlamaMiktar) then  Exception BukumTalimatMiktariAsildi;

      end
    end
    else if (new.DokumaPlanlamaId is not null) then
    begin
      Select Count(*) from DokumaPlanlamaRecete where DokumaPlanlamaRecete.DokumaPlanlamaId = new.DokumaPlanlamaId and DokumaPlanlamaRecete.HamId = new.HamId into :KS;
      if (KS = 0) then exception UrunTalimatRecetedeYok;
    end
    else if (new.CozguIsEmriKayitId is not null) then
    begin
      if (New.HareketTipi = shtFasonUretimeCikis or New.HareketTipi = shtUretimeCikis) then
        Select Count(*) from CozguIsEmriRecete where CozguIsEmriRecete.CozguIsEmriReceteID = New.CozguIsEmriReceteID and CozguIsEmriRecete.HamId = New.HamId into :KS;
      else
        Select Count(*) from CozguIsEmri where CozguIsEmri.CozguIsEmriKayitId = New.CozguIsEmriKayitId and CozguIsEmri.IplikHamId = New.HamId into :KS;
      if (KS = 0) then exception UrunTalimatRecetedeYok;
    end
    else if (new.HasilIsEmriKayitId is not null) then
    begin
      Select Count(*) from HasilIsEmri where HasilIsEmri.HasilIsEmriKayitId = New.HasilIsEmriKayitId and HasilIsEmri.IplikHamId = New.HamId into :KS;
      if (KS = 0) then exception UrunTalimatRecetedeYok;
    end
    else if (new.IndigoIsEmriKayitId is not null) then
    begin
      Select Count(*) from IndigoIsEmri where IndigoIsEmri.IndigoIsEmriKayitId = New.IndigoIsEmriKayitId and IndigoIsEmri.IplikHamId = New.HamId into :KS;
      if (KS = 0) then exception UrunTalimatRecetedeYok;
    end

    New.KendiStogumuz = 0;

    if (new.GirisCikis = 1) then
    begin
      CozguIsEmriStokFirmaId = Null;
      if (new.CozguIsEmriKayitId is not null) then
        select CozguIsEmri.StokFirmaId from CozguIsEmri where CozguIsEmri.CozguIsEmriKayitId = new.CozguIsEmriKayitId into :CozguIsEmriStokFirmaId;

      if (new.HareketTipi = shtUretim and new.UretimIslemTipiId = uitCozgu and CozguIsEmriStokFirmaId is not null) then
      begin
        new.StokFirmaId = CozguIsEmriStokFirmaId;

        if (FFirmaKodumuz = new.StokFirmaId) then
        begin
          New.FasonStok = 0;
          New.KendiStogumuz = 1;
        end
        else
        begin
          New.FasonStok = 1;
          New.KendiStogumuz = 0;
        end
      end
      else
      if (new.HareketTipi = shtSatisIade and FOrmeSiparisSekliId <> assSatis and new.Tamir = 1) then
      begin
        New.StokFirmaId = FSiparisFirmaId;
        New.FasonStok = 1;
        New.KendiStogumuz = 0;
      end
      else
      if (new.HareketTipi = shtAlis or new.HareketTipi = shtSatisIade) then
      begin
        New.StokFirmaId = FFirmaKodumuz;
        New.FasonStok = 0;
        New.KendiStogumuz = 1;
      end
      else
      if (New.BoyaSiparisKayitId is null and new.HareketTipi = shtFasonUretim and New.HamIade = 1) then
      begin
        if (New.BoyahaneFasonIade = 0) then
        begin
          New.KendiStogumuz = 1;
          New.StokFirmaId = FFirmaKodumuz;
          New.FasonStok = 0;
        end
        else New.KendiStogumuz = 0;
      end
      else
      if ( (new.HareketTipi = shtFasonUretim or new.HareketTipi = shtUretim) and FOrmeSiparisSekliId = assSatis) then
      begin
        New.StokFirmaId = FFirmaKodumuz;
        New.FasonStok = 0;
        if (FSiparisFirmaId = FFirmaKodumuz) then New.KendiStogumuz = 1;
      end
      else
      if (  (new.HareketTipi = shtFasonAlis) or
         (  (new.HareketTipi = shtFasonUretim or new.HareketTipi = shtUretim) and New.HamIade = 0 and FOrmeSiparisSekliId <> assSatis)) then
      begin
        if (new.HareketTipi = shtFasonAlis) then
           New.StokFirmaId = New.FirmaId;
        else
           New.StokFirmaId = FSiparisFirmaId;
        New.FasonStok = 1;
      end
      else
      if ((new.HareketTipi = shtFasonUretim or new.HareketTipi = shtUretim) and New.HamIade = 1) then
      begin
        if (FOrmeSiparisSekliId = assSatis) then
        begin
          New.StokFirmaId = FFirmaKodumuz;
          New.FasonStok = 0;
          New.KendiStogumuz = 1;
        end
        else if (FOrmeSiparisSekliId = assFason) then
        begin
          New.StokFirmaId = FSiparisFirmaId;
          New.FasonStok = 1;
          New.KendiStogumuz = 0;
        end
        else /* fason + satıs ise */
        begin
          FFason = null;

          if (new.UretimIslemTipiID = uitOrme) then
          begin
            Select Fason from OrmePlanlamaRecete
              where OrmePlanlamaRecete.OrmePlanlamaId = New.OrmePlanlamaId and OrmePlanlamaRecete.HamId = New.HamId and IntKarsilastir(OrmePlanlamaRecete.RenkId, New.RenkId) = 1 and
              StrKarsilastir(OrmePlanlamaRecete.Lot, New.Lot) = 1
              into :FFason;

            if (FFason is null) then exception KayitHatasi ' Talimatta bu ürün yok..';
          end
          else if (new.UretimIslemTipiID = uitDokuma) then
          begin
            Select Max(Fason) from DokumaPlanlamaRecete
              where DokumaPlanlamaRecete.DokumaPlanlamaId = New.DokumaPlanlamaId and DokumaPlanlamaRecete.HamId = New.HamId and IntKarsilastir(DokumaPlanlamaRecete.RenkId, New.RenkId) = 1 and
              StrKarsilastir(DokumaPlanlamaRecete.Lot, New.Lot) = 1
              into :FFason;

            if (FFason is null) then exception KayitHatasi ' Talimatta bu ürün yok..';
          end
          else if (new.UretimIslemTipiID = uitBoya) then
          begin
            Select BoyaSiparis.Fason from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitId = New.BoyaSiparisKayitId into :FFason;
          end
          else if (new.UretimIslemTipiID = uitCozgu) then
          begin
            select iif(CozguIsEmri.FirmaId is null, 0, 1) from CozguIsEmri where CozguIsEmri.CozguIsEmriKayitId = new.CozguIsEmriKayitId into :FFason;
          end
          else exception KayitHatasi ' İade işlem tipi hatalı..';


          if (FFason = 1) then
          begin
            New.StokFirmaId = FSiparisFirmaId;
            New.FasonStok = 1;
            New.KendiStogumuz = 0;
          end
          else
          begin
            New.StokFirmaId = FFirmaKodumuz;
            New.FasonStok = 0;
            New.KendiStogumuz = 1;
          end
        end
      end
      else if (New.HareketTipi <> ShtKaydirmaGiris) then
      begin
        New.StokFirmaId = FFirmaKodumuz;
        New.FasonStok = 0;
      end
    end
    else
    begin
      New.FasonStok = -1;
    end

    new.DegisimTarihi = current_TimeStamp(0);
  end

  if (New.CikisPartiNo = '') then New.CikisPartiNo = Null;

  if (new.UretimIslemTipiId = uitBoya and New.GirisCikis = -1) then
  begin
    if (StrKarsilastir(Old.ManuelCikisPartiNo, New.ManuelCikisPartiNo) = 0 or Old.Miktar <> New.Miktar)  then
      Update BoyahaneCikisPartiNo Set Miktar = new.Miktar, CikisPartiNO = New.ManuelCikisPartiNo where BoyahaneCikisPartiNo.StokHareketID = New.StokHareketID;
  end


  if (new.UretimIslemTipiId = uitBoya and New.GirisCikis = 1) then
  begin
    if (StrKarsilastir(Old.CikisPartiNo, New.CikisPartiNo) = 0) then
    begin
      if (Old.CikisPartiNo is not null) then
        Update BoyahaneCikisPartiNo set BrutGirisMiktar = BrutGirisMiktar - Old.BrutMiktar,
                                        NetGirisMiktar = NetGirisMiktar - Old.Miktar
        where BoyahaneCikisPartiNo.CikisPartiNo = Old.CikisPartiNo;

      if (New.CikisPartiNo is not null) then
      begin
        Select FirmaId from BoyahaneCikisPartiNo where CikisPartiNo = New.CikisPartiNo into :CikisFirmaId;

        if (CikisFirmaID <> New.FirmaID) then exception CikisSecilenFirmayaYapilmamis;

        Update BoyahaneCikisPartiNo set BrutGirisMiktar = BrutGirisMiktar + new.BrutMiktar,
                                        NetGirisMiktar = NetGirisMiktar + new.Miktar
          where BoyahaneCikisPartiNo.CikisPartiNo = new.CikisPartiNo;
      end
    end
    else if (Old.Miktar <> New.Miktar or Old.BrutMiktar <> New.BrutMiktar) then
    begin
      if (New.CikisPartiNo is not null) then
        Update BoyahaneCikisPartiNo set BrutGirisMiktar = BrutGirisMiktar - Old.BrutMiktar + New.BrutMiktar,
                                        NetGirisMiktar = NetGirisMiktar - Old.Miktar + New.Miktar
          where BoyahaneCikisPartiNo.CikisPartiNo = New.CikisPartiNo;
    end
  end
  if (OrmeFatTevkifatDetayiEtkile = 1 and old.OrmeFaturaID is Null and new.OrmeFaturaID is not null) then
    Select Tevkifat from OrmeFatura where OrmeFaturaID = new.OrmeFaturaID into new.Tevkifat;

  if (IFaturaKesinceFiyatGuncelle = 1  and New.HareketTipi = shtSatis and New.HamIade = 0 and Old.OrmeFaturaID is Null and new.OrmeFaturaID is not null and  new.OrmeIrsaliyeID is not null) then
  begin
    Select DovizID, FirmaID, FaturaTarihi from OrmeFatura where OrmeFaturaID = new.OrmeFaturaID into :FaturaDovizId, :FaturaFirmaID, :FaturaTarihi;

    if (New.BoyaSiparisDetayID is not Null) then
      Select BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizID from BoyaSiparisDetay Where BoyaSiparisDetayID = new.BoyaSiparisDetayID into :SipBirimFiyat, :SipDovizId;
    else if (New.OrmeSiparisDetayID is not Null) then
      Select OrmeSiparisDetay.SatisFiyati, OrmeSiparisDetay.DovizID from OrmeSiparisDetay Where OrmeSiparisDetayID = new.OrmeSiparisDetayID into :SipBirimFiyat, :SipDovizId;

    if (SipDovizId <> FaturaDovizId) then
      Select DovizKuru from FirmaDovizKurlariniGetir(:FaturaFirmaID, :SipDovizId, :FaturaDovizId, :FaturaTarihi) into :DovizKuru;
    else
      DovizKuru = 1;

    New.BirimFiyat = SipBirimFiyat * DovizKuru;
  end


  if (new.DokumaHareketi = 1 and new.MiktarBirim = 1 and new.HareketTipi = shtUretim) then /*Metre ise Miktar2 ye Kilo bilgilerini Yazalım*/
  begin
    if (new.UretimIslemTipiId = uitDokuma and new.Miktar <> Old.Miktar) then
    begin
      Select HamEnCm, HamGramaj From Ham Where HamId = new.HamID into :HamEnCm, :HamGramaj;
      new.Miktar2 =  ((new.Miktar * (HamEnCm / 100) * (HamGramaj / 1000)));
    end
    else if ((new.UretimIslemTipiId = uitHasil or new.UretimIslemTipiId = uitIndigo or new.UretimIslemTipiId = uitCozgu) and (new.Miktar <> Old.Miktar or new.TelAdedi <> Old.TelAdedi)) then
    begin
      Select Kalinlik.KalinlikNe From Ham
        Left Join Kalinlik on Kalinlik.KalinlikID = Ham.KalinlikID
        Where HamId = new.HamID into :KalinlikNe;

      if (KalinlikNe > 0) then
        new.Miktar2 = (cast(New.TelAdedi as Double Precision) / 1693.4/ KalinlikNe) * new.Miktar;
    end
  end
  if (new.Miktar2 is Null) then new.Miktar2 = 0;
end
