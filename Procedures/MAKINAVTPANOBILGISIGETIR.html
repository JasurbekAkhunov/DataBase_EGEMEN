CREATE PROCEDURE MAKINAVTPANOBILGISIGETIR
RETURNS (
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    MAKINABILGIID INTEGER,
    PANONO INTEGER,
    ELEKTIRIK INTEGER,
    DOGALGAZ INTEGER,
    RAMUSTBESLEME DOUBLE PRECISION,
    RAMALTBESLEME DOUBLE PRECISION,
    RAMCALISMAHIZI DOUBLE PRECISION,
    RAMGIRISEN DOUBLE PRECISION,
    RAMBIRBACAFANCALISMAYUZDESI DOUBLE PRECISION,
    RAMIKIBACAFANCALISMAYUZDESI DOUBLE PRECISION,
    RAMSOGUTMAFANIUST DOUBLE PRECISION,
    RAMSOGUTMAFANIALT DOUBLE PRECISION,
    RAMSOGUTMAFANIUSTMAN DOUBLE PRECISION,
    RAMSOGUTMAFANIALTOTO DOUBLE PRECISION,
    RAMBACANEMIOTOMATIK EGEBOOLEAN,
    RAMBACANEMI DOUBLE PRECISION,
    RAMCIKISEN DOUBLE PRECISION,
    RAMCIKISNEMDEGERI DOUBLE PRECISION,
    RAMBESLEMEHIZI DOUBLE PRECISION,
    RAMKALIPGENISLIGI DOUBLE PRECISION,
    RAMKUMASCEKMEZLIKDEGERI DOUBLE PRECISION,
    RAMKUMASMAYDEGERI DOUBLE PRECISION,
    RAMKUMASNEMI DOUBLE PRECISION,
    RAMMAKINEHIZI DOUBLE PRECISION,
    RAMMAKINESILINDIRSICAKLIGI DOUBLE PRECISION,
    TOPLAMKILO NUMERIC(12,3),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    HAMADI VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(50),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAKISAADI VARCHAR(10),
    FASON INTEGER,
    FASONSTR VARCHAR(6),
    MIKTAR NUMERIC(12,3))
AS
declare variable VardiyaBaslangicSaati   Double Precision;
declare variable PartiSinifID integer;
declare variable BaslangicTarihi         TimeStamp;
declare variable BitisTarihi             TimeStamp;
begin
  Select TimeStampToDouble(TimeStampToTime(VardiyaBaslangicSaati)) from sistem into :VardiyaBaslangicSaati;

  BaslangicTarihi = Current_date;
  BaslangicTarihi = BaslangicTarihi + VardiyaBaslangicSaati;
  BitisTarihi = BaslangicTarihi + 1;

  For Select Makina.MakinaId, Makina.Adi, Makina.PanoNo, Makina.MakinaBilgiId from Makina
    where TerminalNo > 0
    into :MakinaID, :MakinaAdi, :PanoNo, :MakinaBilgiId do
  begin
    ToplamKilo = Null;

    Select MakinaBilgi.Elektirik, MakinaBilgi.Dogalgaz, MakinaBilgi.RAMUstBesleme, MakinaBilgi.RAMAltBesleme, MakinaBilgi.RAMCalismaHizi,
           MakinaBilgi.RAMGirisEn, MakinaBilgi.RAMBirBacaFanCalismaYuzdesi, MakinaBilgi.RAMIkiBacaFanCalismaYuzdesi, MakinaBilgi.RAMSogutmaFaniUst,
           MakinaBilgi.RAMSogutmaFaniAlt, MakinaBilgi.RAMSogutmaFaniUstMan, MakinaBilgi.RAMSogutmaFaniAltOto, MakinaBilgi.RAMBacaNemiOtomatik, MakinaBilgi.RAMBacaNemi,
           MakinaBilgi.RAMCikisEn, RamCikisNemDegeri, RamBeslemeHizi, RamKalipGenisligi, RamKumasCekmezlikDegeri, RamKumasMayDegeri, RamKumasNemi, RamMakineHizi, RamMakineSilindirSicakligi from MakinaBilgi
      where MakinaBilgi.MakinaBilgiID = :MakinaBilgiId
      into :Elektirik, :Dogalgaz, :RAMUstBesleme, :RAMAltBesleme, :RAMCalismaHizi,
           :RAMGirisEn, :RAMBirBacaFanCalismaYuzdesi, :RAMIkiBacaFanCalismaYuzdesi, :RAMSogutmaFaniUst, :RAMSogutmaFaniAlt, :RAMSogutmaFaniUstMan, :RAMSogutmaFaniAltOto, :RAMBacaNemiOtomatik,
           :RAMBacaNemi, :RAMCikisEn, :RamCikisNemDegeri, :RamBeslemeHizi, :RamKalipGenisligi, :RamKumasCekmezlikDegeri, :RamKumasMayDegeri, :RamKumasNemi, :RamMakineHizi, :RamMakineSilindirSicakligi;

    Select sum(PartiAsamaStokDetay.Kilo) from PartiAsamaStokDetay
      where PartiAsamaStokDetay.CikisTarihi >= :BaslangicTarihi and PartiAsamaStokDetay.CikisTarihi < :BitisTarihi and
            PartiAsamaStokDetay.MakinaID = :MakinaID
      into :ToplamKilo;

    PartiYil  = Null;
    PartiId   = Null;
    HamAdi    = Null;
    ReceteId  = Null;
    ReceteAdi = Null;

    select First 1 Parti.PartiYil, Parti.PartiId, parti.ReceteId, recete.adi, PartiStok.HamAdiToplami,
      BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparis.FirmaId, Firma.Adi, Parti.Fason,
      PartiStok.Miktar, Firma.KisaAdi from PartiAsamaStokDetay
      left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
      left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      left join recete on recete.receteid = Parti.ReceteId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = PartiStok.BoyaSiparisKayitId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
      where PartiAsamaStokDetay.MakinaId = :MakinaId and PartiAsamaStokDetay.Cikistarihi is null and PartiAsamaStokDetay.giristarihi is not null
      into :PartiYil, :PartiId, :ReceteId, :ReceteAdi, :HamAdi,
      :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :FirmaId, :FirmaAdi, :Fason,
      :Miktar, :FirmaKisaAdi;

    if (Fason = 1) then FasonStr = ' Fason';
    else FasonStr = '';

    suspend;
  end
end
