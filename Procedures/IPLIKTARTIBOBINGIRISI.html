CREATE PROCEDURE IPLIKTARTIBOBINGIRISI(
    APARTISINIFID INTEGER,
    ABOBINSAYISI INTEGER,
    APERSONELID INTEGER)
AS
declare variable KayitSayisi         integer;
declare variable FBobinSayisi        integer;
declare variable FarkBobinSayisi     integer;
declare variable FRafDetayId         integer;
declare variable IplikTartiAsamaId   integer;
begin
  select IplikTartiAsamaId from Sistem where sistemid = 1 into :IplikTartiAsamaId;

  Select Count(*) from PartiAsamaStok where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is null and
         PartiAsamaStok.AsamaId <> :IplikTartiAsamaId into :KayitSayisi;

  if (KayitSayisi > 0) then exception PartiImalattaKiloDegisemez;

  Select BobinSayisi from PartiStok where PartiStok.PartiSinifId = :APartiSinifId into :FBobinSayisi;

  if (ABobinSayisi is null or ABobinSayisi <= 0) then exception BobinSayisiSifirOlamaz;

  if (FBobinSayisi >= ABobinSayisi) then
  begin
    FarkBobinSayisi = FBobinSayisi - ABobinSayisi;

    For Select RafDetayId, BobinSayisi from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId into :FRafDetayId, :FBobinSayisi do
    begin
      if (FarkBobinSayisi > 0) then
      if (FarkBobinSayisi < FBobinSayisi) then
      begin
        Update PartiStokDetay Set BobinSayisi = BobinSayisi - :FarkBobinSayisi, KiloDegistirenPersonelId = :APersonelId where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;
        FarkBobinSayisi = 0;
      end
      else
      begin
        Update PartiStokDetay Set BobinSayisi = 0, KiloDegistirenPersonelId = :APersonelId where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;

        FarkBobinSayisi = FarkBobinSayisi - FBobinSayisi;
      end
    end
  end else
  begin
    FarkBobinSayisi = ABobinSayisi - FBobinSayisi;

    Select Max(RafDetayId) from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId into :FRafDetayId;

    Update PartiStokDetay Set BobinSayisi = BobinSayisi + :FarkBobinSayisi, KiloDegistirenPersonelId = :APersonelId where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;
    FarkBobinSayisi = 0;
  end

  if (FarkBobinSayisi <> 0) then exception BobinSayisiDegisitirlemedi;


end
