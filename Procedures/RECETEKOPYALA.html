CREATE PROCEDURE RECETEKOPYALA(
    ERECETEID VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    AKULLANICIID INTEGER)
AS
declare variable FirmaId                  varchar(20);
declare variable UstAsamaId               integer;
declare variable ReceteTipi               integer;
declare variable ReceteTuruId             integer;
declare variable CalismaTarihi            timestamp;
declare variable RenkOkey                 integer;
declare variable ReceteKapatma            integer;
declare variable ReceteGrafikId           varchar(10);

declare variable ReceteAsamaId            integer;
declare variable ReceteAltAsamaId         integer;
declare variable ReceteAsamaSablonId      integer;
declare variable UrunId                   varchar(20);
declare variable UrunBirimId              integer;
declare variable Miktar                   Double precision;
declare variable Banyo                    Double precision;
declare variable FT1                    Double Precision;
declare variable FTarih                 TimeStamp;
declare variable LabReceteGrupId          integer;
declare variable KayitSayisi              integer;
declare variable Ilave              integer;
declare variable SokumUzeriBoya     integer;
declare variable SablonReceteId varchar(20);
declare variable Sablon         integer;
declare variable LabRenkGrupId          integer;
declare variable RenkDerinligiId        integer;
declare variable BanyoSayisi            integer;
declare variable ToplamSicaklikDegisimi integer;
declare variable AtisRenkSira           integer;
declare variable AtisRenkColor          integer;
declare variable AtisRenkKodu          varchar(50);
declare variable AtisRenkAdi           varchar(120);
begin
  Select Count(*) from Recete where Recete.ReceteId = :ReceteId into :KayitSayisi;

  if (KayitSayisi = 0) then
  begin
    select FirmaId, UstAsamaId, ReceteTipi, ReceteTuruId, LabReceteGrupId, CalismaTarihi, RenkOkey, Sablon, ReceteKapatma,
           ReceteGrafikId, Ilave, SokumUzeriBoya, LabRenkGrupId, RenkDerinligiId, BanyoSayisi, ToplamSicaklikDegisimi from Recete
    where Recete.ReceteId = :EReceteId
    into :FirmaId, :UstAsamaId, :ReceteTipi, :ReceteTuruId, :LabReceteGrupId, :CalismaTarihi, :RenkOkey, :Sablon, :ReceteKapatma,
         :ReceteGrafikId, :Ilave, :SokumUzeriBoya, :LabRenkGrupId, :RenkDerinligiId, :BanyoSayisi, :ToplamSicaklikDegisimi;

    if (FirmaId is null) then exception ReceteKayitliDegil;

    FTarih =  Current_date;

    if (Sablon = 1) then SablonReceteId = EReceteId;
    else SablonReceteId = Null;

    insert into Recete(ReceteId, Adi, FirmaId, UstAsamaId, ReceteTipi, ReceteTuruId, SablonReceteId, LabReceteGrupId, LabRenkGrupId, RenkDerinligiId, CalismaTarihi, RenkOkey, ReceteKapatma, ReceteGrafikId,
                       BanyoSayisi, ToplamSicaklikDegisimi, Ilave, SokumUzeriBoya, KullaniciId, Sablon)
                values(:ReceteId, :ReceteAdi, :FirmaId, :UstAsamaId, :ReceteTipi, :ReceteTuruId, :SablonReceteId, :LabReceteGrupId, :LabRenkGrupId, :RenkDerinligiId, :FTarih, :RenkOkey, :ReceteKapatma, :ReceteGrafikId,
                       :BanyoSayisi, :ToplamSicaklikDegisimi, :Ilave, :SokumUzeriBoya, :AKullaniciId, :Sablon);
  end

  Select Count(*) from ReceteDetay where ReceteDetay.ReceteId = :ReceteId into :KayitSayisi;

  if (KayitSayisi > 0) then exception ReceteDetayivarKopyalamadi;

  Update Recete Set OzelDurumLinkIslemi = 1 where ReceteId = :ReceteId;

  For Select ReceteDetay.ReceteAsamaId, ReceteDetay.ReceteAltAsamaId, ReceteDetay.ReceteAsamaSablonId, ReceteDetay.UrunId, UrunBirimId, Miktar, Banyo, ReceteGrafikId, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi from ReceteDetay
    where ReceteDetay.ReceteId = :EReceteId
    order by ReceteDetay.ReceteId, ReceteDetay.Sira
    into :ReceteAsamaId, :ReceteAltAsamaId, :ReceteAsamaSablonId, :UrunId, :UrunBirimId, :Miktar, :Banyo, :ReceteGrafikId, :AtisRenkSira, :AtisRenkColor, :AtisRenkKodu, :AtisRenkAdi do
  begin
    insert into ReceteDetay(ReceteId, Sira, ReceteAsamaId, ReceteAltAsamaId, ReceteAsamaSablonId, UrunId, UrunBirimId, Miktar, Banyo, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, ReceteGrafikId, KullaniciId)
                values(:ReceteId, -1, :ReceteAsamaId, :ReceteAltAsamaId, :ReceteAsamaSablonId, :UrunId, :UrunBirimId, :Miktar, :Banyo, :AtisRenkSira, :AtisRenkColor, :AtisRenkKodu, :AtisRenkAdi, :ReceteGrafikId, :AKullaniciId);
  end

  Update Recete Set OzelDurumLinkIslemi = 0 where ReceteId = :ReceteId;
end
