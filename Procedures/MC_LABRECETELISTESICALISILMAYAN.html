CREATE PROCEDURE MC_LABRECETELISTESICALISILMAYAN(
    AFIRMAID VARCHAR(20),
    ABASLANGICTALEPTARIHI TIMESTAMP,
    ABITISTALEPTARIHI TIMESTAMP,
    ALABRECETEGRUPID INTEGER,
    ALABRENKGRUPID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    ADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TEMSILCIID INTEGER,
    TEMSILCIADI VARCHAR(40),
    TALEPTARIHI TIMESTAMP,
    LABRECETEGRUPID INTEGER,
    LABRECETEGRUPADI VARCHAR(20),
    LABRENKGRUPID INTEGER,
    LABRENKGRUPADI VARCHAR(30),
    MUSTERIRENKKODU VARCHAR(30),
    RECETETURUID INTEGER,
    RECETETURUADI VARCHAR(15),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    ACIKLAMA VARCHAR(40),
    ESKILABRECETEKODU VARCHAR(20),
    PANTONEKODU VARCHAR(20))
AS
declare variable Okey     integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  for Select LabRecete.LabReceteId, LabRecete.LabReceteKodu, LabRecete.Adi, LabRecete.FirmaId, Firma.Adi, Firma.TemsilciId, Tk.Adi, LabRecete.TalepTarihi,
             LabRecete.LabReceteGrupId, LabReceteGrup.Adi, LabRecete.LabRenkGrupId, LabRenkGrup.Adi, LabRecete.ReceteTuruId, ReceteTuru.Adi,
             LabRecete.MusteriRenkKodu, LabRecete.HamId, Ham.Adi, LabRecete.Aciklama, LabRecete.EskiLabReceteKodu, LabRecete.PantoneKodu from LabRecete
  left join Firma On Firma.FirmaId = LabRecete.FirmaId
  left join Ham On Ham.HamId = LabRecete.HamId
  left join ReceteTuru On ReceteTuru.ReceteTuruId = LabRecete.ReceteTuruId
  left join LabReceteGrup On LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
  left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = LabRecete.LabRenkGrupId
  left join LabReceteCalisma On LabReceteCalisma.LabReceteId = LabRecete.LabReceteId
  Left Join Kullanici TK on TK.KullaniciId = Firma.TemsilciId
  where LabRecete.TalepTarihi >= :ABaslangicTalepTarihi and LabRecete.TalepTarihi <= :ABitisTalepTarihi and
        LabRecete.FirmaId = :AFirmaId and LabReceteCalisma.LabReceteCalismaID is null and
        (LabRecete.LabRenkGrupId = :ALabRenkGrupId or :ALabRenkGrupId is null) and LabRecete.Iptal = 0
  into :LabReceteId, :LabReceteKodu, :Adi, :FirmaId, :FirmaAdi, :TemsilciId, :TemsilciAdi, :TalepTarihi, :LabReceteGrupId, :LabReceteGrupAdi, :LabRenkGrupId,
       :LabRenkGrupAdi, :ReceteTuruId, :ReceteTuruAdi, :MusteriRenkKodu, :HamId, :HamAdi, :Aciklama, :EskiLabReceteKodu, :PantoneKodu  do
  begin
    suspend;
  end

  for Select LabRecete.LabReceteId, LabRecete.LabReceteKodu, LabRecete.Adi, LabRecete.FirmaId, Firma.Adi, Firma.TemsilciId, Tk.Adi, LabRecete.TalepTarihi,
             LabRecete.LabReceteGrupId, LabReceteGrup.Adi, LabRecete.LabRenkGrupId, LabRenkGrup.Adi, LabRecete.ReceteTuruId, ReceteTuru.Adi,
             LabRecete.MusteriRenkKodu, LabRecete.HamId, Ham.Adi, LabRecete.Aciklama, LabRecete.EskiLabReceteKodu, LabRecete.PantoneKodu from LabReceteCalisma
  left join LabRecete On LabRecete.LabReceteId = LabReceteCalisma.LabReceteId
  left join Firma On Firma.FirmaId = LabRecete.FirmaId
  left join Ham On Ham.HamId = LabRecete.HamId
  left join ReceteTuru On ReceteTuru.ReceteTuruId = LabRecete.ReceteTuruId
  left join LabReceteGrup On LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
  left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = LabRecete.LabRenkGrupId
  left join LabReceteAtis On LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
  Left Join Kullanici TK on TK.KullaniciId = Firma.TemsilciId
  where LabReceteCalisma.CalismaTarihi >= :ABaslangicTalepTarihi and LabReceteCalisma.CalismaTarihi <= :ABitisTalepTarihi and
        LabRecete.FirmaId = :AFirmaId and LabReceteAtis.LabReceteAtisID is null and
        (LabRecete.LabRenkGrupId = :ALabRenkGrupId or :ALabRenkGrupId is null) and LabRecete.Iptal = 0 and
        (LabReceteCalisma.Rptmi = 0 and LabReceteCalisma.Ilave = 0 and LabReceteCalisma.KritikCalismasi = 0  and LabReceteCalisma.SokumUZeriBoya = 0)
  group by LabRecete.LabReceteId, LabRecete.LabReceteKodu, LabRecete.Adi, LabRecete.FirmaId, Firma.Adi, Firma.TemsilciId, Tk.Adi, LabRecete.TalepTarihi,
           LabRecete.LabReceteGrupId, LabReceteGrup.Adi, LabRecete.LabRenkGrupId, LabRenkGrup.Adi, LabRecete.ReceteTuruId, ReceteTuru.Adi,
           LabRecete.MusteriRenkKodu, LabRecete.HamId, Ham.Adi, LabRecete.Aciklama, LabRecete.EskiLabReceteKodu, LabRecete.PantoneKodu
  into :LabReceteId, :LabReceteKodu, :Adi, :FirmaId, :FirmaAdi, :TemsilciId, :TemsilciAdi, :TalepTarihi, :LabReceteGrupId, :LabReceteGrupAdi, :LabRenkGrupId,
       :LabRenkGrupAdi, :ReceteTuruId, :ReceteTuruAdi,  :MusteriRenkKodu, :HamId, :HamAdi, :Aciklama, :EskiLabReceteKodu, :PantoneKodu do
  begin

    suspend;
  end
end
