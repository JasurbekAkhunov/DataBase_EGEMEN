CREATE PROCEDURE PLANLAMARAPORUGETIR(
    ABOYAHANEBOLUMID INTEGER,
    AMAKINAID VARCHAR(10),
    TUMPLANLISARJLARIGOSTER INTEGER,
    AIMALATTIPIID INTEGER)
RETURNS (
    MAKINAID VARCHAR(10),
    SEPETNO INTEGER,
    PARTIIDSTR VARCHAR(20),
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    BASLAMATARIHI TIMESTAMP,
    IMALATTA INTEGER,
    BEKLEMEDE INTEGER,
    SURE INTEGER,
    FIRMAADI VARCHAR(100),
    FIRMAKISAADI VARCHAR(10),
    KAZANSURE VARCHAR(30),
    TAMIR INTEGER,
    ONCEKIASAMADA INTEGER,
    RECETEADI VARCHAR(40),
    RECETEID VARCHAR(20),
    RECETEYAZILDI INTEGER,
    SIPARISTARIHI TIMESTAMP,
    SIPARISGECENSURE VARCHAR(30),
    CIKISRAFID RAFKODUSIZE,
    MIKTAR BOYAMIKTARSIZE,
    PROSESNOTU MEMO,
    SONRAKIASAMAADI VARCHAR(40),
    FPARTIASAMALARIID INTEGER,
    ASAMACOLOR INTEGER,
    URETIMPLANLAMAGOSTERGESISTR VARCHAR(255),
    DESENID VARCHAR(20),
    VARYANTADI VARCHAR(30),
    TAMIRTIPIADI VARCHAR(20))
AS
declare variable PartiYil integer;
declare variable PartiID integer;
declare variable FGirisTarihi TimeStamp;
declare variable SonrakiSira integer;
declare variable SonYapilanSira integer;
declare variable SiradakiPartiAsamalariId integer;
declare variable FSiraNo integer;
declare variable FFSiraNo integer;
declare variable FPartiKayitID integer;
declare variable SuspendCount integer;
declare variable UretimPlanlamaGostergesi varchar(30);
declare variable FSonrakiAsamaColor Integer;
declare variable FEgemenFirmaId Integer;
begin
  KazanSure = Null;
  SuspendCount = 0;
  OncekiAsamada = 0;
  SiparisTarihi = Null;
  SiparisGecenSure = Null;

  Select EgemenFirmaId From Sistem Where SistemId = 1 Into :FEgemenFirmaId;

  For Select First 1 PartiStokBekle.PartiKayitID, PartiStokBekle.PartiSinifID, PartiAsamalari.Sira, TamirTipi.Adi, DesenVaryant.DesenId, DesenVaryant.VaryantAdi from PartiStokBekle
      Left Join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiSinifID = PartiStokBekle.PartiSinifID
      Left Join PartiAsamalari on PartiAsamalari.PArtiAsamalariID = PartiStokBekle.PartiAsamalariID
      left join Parti on Parti.PartiKayitId = PArtiStokBekle.PartiKayitId
      left join PartiTamir on PartiTamir.PartiKayitId = Parti.PartiKayitId and PartiTamir.Bitti = 0
      left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
      left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
      where PartiStokBekle.Bitti = 0 and PartiStokBekle.OtomatikBekleme = 1 and PartiAsamaStokDetay.MakinaId = :AMakinaID and
      (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is Null)
      Order By BaslamaTarihi Desc
  into :FPartiKayitID, :PartiSinifID, :FSiraNo, :TamirTipiAdi, :Desenid, :Varyantadi do
  begin
    AsamaColor = null;
    FSonrakiAsamaColor = null;
    UretimPlanlamaGostergesiStr = '';

    Select Max(Sira) from PartiAsamalari
      left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = :PartiSinifID and PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join Asama on Asama.AsamaId = PartiAsamalari.AsamaID
      where PartiAsamalari.PartiKayitId = :FPartiKayitID and PartiAsamalari.Sira < :FSiraNo and
            PartiAsamaStok.AsamayaDahil = 1 and Asama.AraKontrolAsamasimi = 1 into :FSiraNo;

    if (FSiraNo is not Null) then
    begin
      Select First 1 max(Sira), PartiAsamalari.PartiAsamalariID from PartiAsamalari
        left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = :PartiSinifID and PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        Left Join PartiAsamaStokDetay on PArtiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaID
        where PartiAsamalari.PartiKayitId = :FPartiKayitID and PartiAsamalari.Sira < :FSiraNo and
              PartiAsamaStok.AsamayaDahil = 1 and Asama.BoyaAsamasimi = 1 and PartiAsamalari.PartiTamirID is Null and PartiAsamaStokDetay.MakinaId = :AMakinaID
        Group by PartiAsamalari.PartiAsamalariID  Order by 1 Desc
        into :FSiraNo, :FPartiAsamalariID;

      if (FPartiAsamalariID is not Null) then
      begin
        Select Parti.PartiYil, Parti.PartiId, Firma.Adi, Firma.KisaAdi, PartiStok.HamAdiToplami, PartiStok.SonCikisKilo, PartiStok.SonCikisMiktar, Recete.Adi, Parti.ReceteYazildi, Recete.ReceteId from PartiAsamalari
          Left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
          Left join Firma on Firma.FirmaId = Parti.FirmaId
          Left join PartiStok on PartiStok.PartiSinifId = :PartiSinifID
          left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
          where PartiAsamalari.PartiAsamalariID = :FPartiAsamalariID and PartiAsamalari.Sira = :FSiraNo
          into :PartiYil, :PartiId, :FirmaAdi, :FirmaKisaAdi, :HamAdiToplami, :Kilo, :Miktar, :ReceteAdi, :ReceteYazildi, :ReceteId;

        if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
          else PartiIdStr = '';

        select SonAsamaBilgisiGetir.SonrakiAsamaAdi, SonAsamaBilgisiGetir.SonrakiAsamaColor from SonAsamaBilgisiGetir(:PartiSinifId, :FPartiAsamalariId)
        left join asama on asama.asamaid  = SonAsamaBilgisiGetir.sonrakiasamaid into :SonrakiAsamaAdi, :FSonrakiAsamaColor;

        Select First 1 Asama.Color from PartiAsamalari
          left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = :PartiSinifID and PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiSinifId = :PartiSinifID and PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaID
          where PartiAsamalari.PartiKayitId = :FPartiKayitID and PartiAsamalari.Sira < :FSiraNo and PartiAsamaStok.AsamayaDahil = 1 and
          Asama.Color is not null and PartiAsamaStok.CikisTarihi is null
          Order By PartiAsamalari.Sira Desc
          into :AsamaColor;

        For Select Coalesce(Asama.UretimPlanlamaGostergesi, '') from PartiAsamaStok
              left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
              left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
            where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.Giristarihi is null and
                  PartiAsamaStok.AsamayaDahil = 1 And PartiAsamalari.OtomasyonaDahilmi = 1  and Coalesce(Asama.UretimPlanlamaGostergesi, '') <> ''
            order by PartiAsamalari.Sira
            into :UretimPlanlamaGostergesi do
        begin
          if ((Char_Length(UretimPlanlamaGostergesiStr) + Char_Length(UretimPlanlamaGostergesi) < 250)  and (UretimPlanlamaGostergesi <> '')) then
          begin
            if (UretimPlanlamaGostergesiStr <> '' ) then UretimPlanlamaGostergesiStr = UretimPlanlamaGostergesiStr || cast('+' as varchar(2) character set UTF8);
            UretimPlanlamaGostergesiStr = UretimPlanlamaGostergesiStr || UretimPlanlamaGostergesi;
          end
        end
         Beklemede = 1;
         MakinaId = :AMakinaID;
         SepetNo  = 9999;
         SuspendCount = SuspendCount + 1;

         if (FEgemenFirmaId <> 128) then
           AsamaColor = FSonrakiAsamaColor;

         suspend;
      end
    end
  end

  KazanSure = Null;
  Imalatta = 0;
  BaslamaTarihi = Null;
  Beklemede = 0;
  SiparisGecenSure = Null;
  SiparisTarihi = Null;
  AsamaColor = null;

  if (TumPlanliSarjlariGoster = 0 ) then
  begin
    For select First 5 UretimPlanlama.MakinaId, UretimPlanlama.SepetNo, PartiStok.PartiSinifID,
         Parti.PartiYil, Parti.PartiId, Firma.Adi, Firma.KisaAdi, PartiAsamaStok.GirisTarihi,
         case when coalesce(PartiStok.HamAdiToplami,'')<>'' then PartiStok.HamAdiToplami else left(substring(Cast (UretimPlanlamaDetay.RezervNotu as Blob character set UTF8) from 1 FOR 255), 255) end,
         PartiStok.SonCikisKilo, PartiStok.SonCikisMiktar, iif(PartiAsamalari.PartiTamirID is Null, 0 , 1), PartiAsamalari.Sira, PartiAsamalari.PartiKayitID, Recete.Adi,
         Parti.ReceteYazildi, BoyaSiparis.SiparisTarihi, Recete.ReceteId, PartiAsamalari.ProsesNotu, TamirTipi.Adi, DesenVaryant.DesenId, DesenVaryant.VaryantAdi from Makina
    left join MakinaPlanlamaGrup on MakinaPlanlamaGrup.MakinaPlanlamaGrupId = Makina.MakinaPlanlamaGrupId
    left join UretimPlanlama on UretimPlanlama.makinaId = Makina.MakinaId
    left join UretimPlanlamaDetay on UretimPlanlamaDetay.UretimPlanlamaId = UretimPlanlama.UretimPlanlamaId
    left join BolumPlanlama on BolumPlanlama.BolumPlanlamaId = UretimPlanlamaDetay.BolumPlanlamaId
    Left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = BolumPlanlama.PartiAsamalariId
    Left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
    Left join Firma on Firma.FirmaId = Parti.FirmaId
    Left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
    Left join PartiStok on PartiStok.PartiSinifId = BolumPlanlama.PartiSinifId
    left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
    Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = PartiStok.BoyaSiparisKayitId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiTamir on PartiTamir.PartiKayitId = Parti.PartiKayitId and PartiTamir.Bitti = 0
    left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    where MakinaPlanlamaGrup.BoyahaneBolumId = :ABoyahaneBolumId and (Makina.MakinaId = :AMakinaId or :AMakinaId is null) and
          UretimPlanlama.UretimPlanlamaID is not null  and
          (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is Null)
    Order By PartiAsamaStok.GirisTarihi Desc, Makina.MakinaId, UretimPlanlama.SepetNo
    into :MakinaId, :SepetNo, :PartiSinifID, :PartiYil, :PartiId, :FirmaAdi, :FirmaKisaAdi,
         :BaslamaTarihi, :HamAdiToplami, :Kilo, :Miktar,  :Tamir, :FSiraNo, :FPartiKayitID, :ReceteAdi, :ReceteYazildi, :SiparisTarihi, :ReceteId, :ProsesNotu,
         :TamirTipiAdi, :Desenid, :Varyantadi do
    begin
      AsamaColor = null;
      UretimPlanlamaGostergesiStr = '';

      For Select Coalesce(Asama.UretimPlanlamaGostergesi, '') from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.Giristarihi is null and
              PartiAsamaStok.AsamayaDahil = 1 And PartiAsamalari.OtomasyonaDahilmi = 1  and Coalesce(Asama.UretimPlanlamaGostergesi, '') <> ''
        order by PartiAsamalari.Sira
        into :UretimPlanlamaGostergesi do
      begin
        if ((Char_Length(UretimPlanlamaGostergesiStr) + Char_Length(UretimPlanlamaGostergesi) < 250)  and (UretimPlanlamaGostergesi <> '')) then
        begin
          if (UretimPlanlamaGostergesiStr <> '' ) then UretimPlanlamaGostergesiStr = UretimPlanlamaGostergesiStr || cast('+' as varchar(2) character set UTF8);
         UretimPlanlamaGostergesiStr = UretimPlanlamaGostergesiStr || UretimPlanlamaGostergesi;
       end
      end

      if (BaslamaTarihi is not null) then Imalatta = 1;
        else Imalatta = 0;

      if (BaslamaTarihi is not null and KazanSure is Null) then KazanSure = TarihToStr(Current_Timestamp - BaslamaTarihi);

      if (SiparisTarihi is not null) then SiparisGecenSure = TarihToStr(Current_Timestamp - SiparisTarihi);

      if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      else PartiIdStr = '';

      if (FSiraNo is not Null) then
      begin
        Select First 1 Max(Sira), iif(PartiAsamaStok.GirisTarihi is Null, 0, 1), PartiAsamaStokDetay.CikisRafId from PartiAsamalari
          left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = :PartiSinifID and PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiSinifId = :PartiSinifID and PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaID
          where PartiAsamalari.PartiKayitId = :FPartiKayitID and PartiAsamalari.Sira < :FSiraNo and PartiAsamaStok.AsamayaDahil = 1
          Group By PartiAsamaStok.GirisTarihi, PartiAsamaStokDetay.CikisRafId  Order By 1 Desc
          into :FSiraNo, :OncekiAsamada, :CikisRafId;

        if (OncekiAsamada is Null) then OncekiAsamada = 1;
        /*Mayteks Onceki Asama Color*/

        Select First 1 Asama.Color from PartiAsamalari
          left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = :PartiSinifID and PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiSinifId = :PartiSinifID and PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaID
          where PartiAsamalari.PartiKayitId = :FPartiKayitID and PartiAsamalari.Sira < :FSiraNo and PartiAsamaStok.AsamayaDahil = 1 and
          Asama.Color is not null and PartiAsamaStok.CikisTarihi is null
          Order By PartiAsamalari.Sira Desc
          into :AsamaColor;

      end
      SuspendCount = SuspendCount + 1;

      select SonAsamaBilgisiGetir.SonrakiAsamaAdi, SonAsamaBilgisiGetir.SonrakiAsamaColor from SonAsamaBilgisiGetir(:PartiSinifId, :FPartiAsamalariId) --,asama.color
      left join asama on asama.asamaid  = SonAsamaBilgisiGetir.sonrakiasamaid into :SonrakiAsamaAdi, :FSonrakiAsamaColor; -- , :AsamaColor

      if (FEgemenFirmaId <> 128) then
      begin
        if (PartiSinifID is null) then
          AsamaColor = 4438460;
        else
        begin
          if (Imalatta = 1) then
            AsamaColor = 32768;
          else
            AsamaColor = FSonrakiAsamaColor;
        end
      end

      if (SuspendCount <= 5) then suspend;
    end
  end
  else
  begin
    For select PartiStok.PartiAsamalariId, UretimPlanlama.MakinaId, UretimPlanlama.SepetNo, PartiStok.PartiSinifID,
         Parti.PartiYil, Parti.PartiId, Firma.Adi, Firma.KisaAdi, PartiAsamaStok.GirisTarihi, 
         case when coalesce(PartiStok.HamAdiToplami,'')<>'' then PartiStok.HamAdiToplami else left(substring(Cast (UretimPlanlamaDetay.RezervNotu as Blob character set UTF8) from 1 FOR 255), 255) end,
         PartiStok.SonCikisKilo, PartiStok.SonCikisMiktar, iif(PartiAsamalari.PartiTamirID is Null, 0 , 1), PartiAsamalari.Sira, PartiAsamalari.PartiKayitID, Recete.Adi,
         Parti.ReceteYazildi, BoyaSiparis.SiparisTarihi, Recete.ReceteId, PartiAsamalari.ProsesNotu, TamirTipi.Adi, DesenVaryant.DesenId, DesenVaryant.VaryantAdi from Makina
    left join MakinaPlanlamaGrup on MakinaPlanlamaGrup.MakinaPlanlamaGrupId = Makina.MakinaPlanlamaGrupId
    left join UretimPlanlama on UretimPlanlama.makinaId = Makina.MakinaId
    left join UretimPlanlamaDetay on UretimPlanlamaDetay.UretimPlanlamaId = UretimPlanlama.UretimPlanlamaId
    left join BolumPlanlama on BolumPlanlama.BolumPlanlamaId = UretimPlanlamaDetay.BolumPlanlamaId
    Left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = BolumPlanlama.PartiAsamalariId
    Left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
    Left join Firma on Firma.FirmaId = Parti.FirmaId
    Left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
    Left join PartiStok on PartiStok.PartiSinifId = BolumPlanlama.PartiSinifId
    left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
    Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = PartiStok.BoyaSiparisKayitId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiTamir on PartiTamir.PartiKayitId = Parti.PartiKayitId and PartiTamir.Bitti = 0
    left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    where MakinaPlanlamaGrup.BoyahaneBolumId = :ABoyahaneBolumId and (Makina.MakinaId = :AMakinaId or :AMakinaId is null) and
          UretimPlanlama.UretimPlanlamaID is not null and
          (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is Null)
    Order By PartiAsamaStok.GirisTarihi Desc, Makina.MakinaId, UretimPlanlama.SepetNo
    into  :FPartiAsamalariID , :MakinaId, :SepetNo, :PartiSinifID, :PartiYil, :PartiId, :FirmaAdi, :FirmaKisaAdi,
         :BaslamaTarihi, :HamAdiToplami, :Kilo, :Miktar,  :Tamir, :FSiraNo, :FPartiKayitID, :ReceteAdi, :ReceteYazildi, :SiparisTarihi, :ReceteId, :ProsesNotu,
         :TamirTipiAdi, :Desenid, :Varyantadi do
    begin
      AsamaColor = null;
      FSonrakiAsamaColor = null;

      if (BaslamaTarihi is not null) then Imalatta = 1;
        else Imalatta = 0;

      if (BaslamaTarihi is not null and KazanSure is Null) then KazanSure = TarihToStr(Current_Timestamp - BaslamaTarihi);

      if (SiparisTarihi is not null) then SiparisGecenSure = TarihToStr(Current_Timestamp - SiparisTarihi);

      if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
        else PartiIdStr = '';

      if (FPartiAsamalariId is not null) then
      begin
        select  PartiAsamalari.Sira from PartiAsamaStok
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId
        into :SonYapilanSira;
      end
      else
      begin
        SonYapilanSira = -1;
      end

      FPartiAsamalariId = null;
      SonrakiAsamaAdi = null;

      select Min(PartiAsamalari.Sira) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.Sira > :SonYapilanSira
      into :SonrakiSira;

      if (SonrakiSira is not null) then
      begin
        select Asama.Adi, PartiAsamalari.PartiAsamalariId, Asama.Color from PartiAsamaStok --, Asama.Color
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :SonrakiSira
        into :SonrakiAsamaAdi, :SiradakiPartiAsamalariId, :FSonrakiAsamaColor;  --, :AsamaColor
      end
      else
      begin
        SonrakiAsamaAdi = Null;
        SiradakiPartiAsamalariId = Null;
        FSonrakiAsamaColor = Null;
      end

      if (FSiraNo is not Null) then
      begin
        Select First 1 Max(Sira), iif(PartiAsamaStok.GirisTarihi is Null, 0, 1), PartiAsamaStokDetay.CikisRafId, PartiAsamalari.PartiAsamalariID from PartiAsamalari
          left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = :PartiSinifID and PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiSinifId = :PartiSinifID and PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaID
          where PartiAsamalari.PartiKayitId = :FPartiKayitID and PartiAsamalari.Sira < :FSiraNo and PartiAsamaStok.AsamayaDahil = 1
          Group By PartiAsamaStok.GirisTarihi, PartiAsamaStokDetay.CikisRafId, PartiAsamalari.PartiAsamalariID  Order By 1 Desc
          into :FFSiraNo, :OncekiAsamada, :CikisRafId, :FPartiAsamalariId;

        /* select First 1 Asama.Adi from PartiAsamaStok
          left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          where PartiAsamaStok.PartiSinifId = :PartiSinifID and PartiAsamalari.Sira > :FFSiraNo and PartiAsamaStok.AsamayaDahil = 1
          order by PartiAsamalari.Sira asc
          into :SonrakiAsamaAdi;*/

        if (OncekiAsamada is Null) then OncekiAsamada = 1;

         /*Mayteks Onceki Asama Color*/

        Select First 1 Asama.Color from PartiAsamalari
        left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = :PartiSinifID and PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiSinifId = :PartiSinifID and PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamalari.AsamaID
        where PartiAsamalari.PartiKayitId = :FPartiKayitID and PartiAsamalari.Sira < :FSiraNo and PartiAsamaStok.AsamayaDahil = 1 and
        Asama.Color is not null and PartiAsamaStok.CikisTarihi is null
        Order By PartiAsamalari.Sira Desc
        into :AsamaColor;
      end

      SuspendCount = SuspendCount + 1;
      UretimPlanlamaGostergesiStr = '';

      For Select Coalesce(Asama.UretimPlanlamaGostergesi, '') from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.Giristarihi is null and
              PartiAsamaStok.AsamayaDahil = 1 And PartiAsamalari.OtomasyonaDahilmi = 1  and Coalesce(Asama.UretimPlanlamaGostergesi, '') <> ''
        order by PartiAsamalari.Sira
        into :UretimPlanlamaGostergesi do
      begin
        if ((Char_Length(UretimPlanlamaGostergesiStr) + Char_Length(UretimPlanlamaGostergesi) < 250)  and (UretimPlanlamaGostergesi <> '')) then
        begin
          if (UretimPlanlamaGostergesiStr <> '' ) then UretimPlanlamaGostergesiStr = UretimPlanlamaGostergesiStr || cast('+' as varchar(2) character set UTF8);
          UretimPlanlamaGostergesiStr = UretimPlanlamaGostergesiStr || UretimPlanlamaGostergesi;
        end
      end

      if (FEgemenFirmaId <> 128) then
      begin
        if (PartiSinifID is null) then
          AsamaColor = 4438460;
        else
        begin
          if (Imalatta = 1) then
            AsamaColor = 32768;
          else
            AsamaColor = FSonrakiAsamaColor;
        end
      end

      suspend;--if (SuspendCount <= 5) then suspend;

      FPartiAsamalariId = null;
      FSiraNo = null;
    end
  end
