CREATE PROCEDURE ENVANTERICINSONALISBILGISIGETIR(
    AURUNID VARCHAR(20),
    ALOT VARCHAR(150))
RETURNS (
    SONALISTARIHI TIMESTAMP,
    SONALISFIYATI DOUBLE PRECISION,
    SONALISDOVIZID VARCHAR(5),
    SONALISURUNBIRIMID INTEGER,
    SONALISBIRIMID VARCHAR(2),
    SONALISBIRIMCARPAN DOUBLE PRECISION,
    SONALISLOT VARCHAR(150),
    SONALISMIKTARI DOUBLE PRECISION)
AS
declare variable FMaxStokDetayId integer;
declare variable FDefaultBirimCarpan double precision;
declare variable SDovizId varchar(5);
declare variable IDovizId varchar(5);
declare variable FDovizId varchar(5);
declare variable FAltIskontoCarpani double precision;
begin
  SonAlisFiyati = 0;
  select max(Tarih) from StokDetay
     where Urunid = :AUrunId and ((:ALot is null) or (Lot = :ALot)) and StokDetay.harekettipi = 1 and StokDetay.NetBirimFiyat <> 0
     PLAN (StokDetay INDEX (FK_StokDetayUrunId, StokDetayHareketTipiIDX))
     into :SonAlisTarihi;

  select max(StokDetayId) from StokDetay
     where Urunid = :AUrunId and StokDetay.HareketTipi = 1 and StokDetay.Tarih = :SonAlisTarihi
     PLAN (StokDetay INDEX (FK_StokDetayUrunId, StokDetayTarihIDX2, StokDetayHareketTipiIDX))
     into :FMaxStokDetayId;

  if (FMaxStokDetayId is not null) then
  begin
    --for
    Select First 1 StokDetay.NetBirimFiyat, StokDetay.UrunBirimId, UrunBirim.BirimId, UrunBirim.Carpan,
               Stok.DovizId, Irsaliye.DovizId, Fatura.DovizId, Fatura.AltIskontoCarpani, StokDetay.Lot,StokDetay.miktar from StokDetay
    left join Stok on Stok.StokId = StokDetay.StokId
    left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
    left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
    left join Firma On Firma.FirmaId = StokDetay.FirmaId
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    where StokDetay.StokDetayId = :FMaxStokDetayId
    into :SonAlisFiyati, :SonAlisUrunBirimId, :SonAlisBirimId, :SonAlisBirimCarpan, :SDovizId, :IDovizId,
    :FDovizId, :FAltIskontoCarpani, :SonAlisLot, :SonAlisMiktari;
    --do
    --begin
      if (FDovizId is not null) then SonAlisDovizId = FDovizId;
      else if (IDovizId is not null) then SonAlisDovizId = IDovizId;
      else SonAlisDovizId = SDovizId;

      if (FAltIskontoCarpani is not null) then SonAlisFiyati = SonAlisFiyati - SonAlisFiyati * FAltIskontoCarpani;
      suspend;
      exit;
    --end
  end
  else suspend;
end
