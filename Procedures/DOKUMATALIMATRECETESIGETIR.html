CREATE PROCEDURE DOKUMATALIMATRECETESIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    DOKUMAPLANLAMARECETEID INTEGER,
    DOKUMAPLANLAMAID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(30),
    LOT2 VARCHAR(150),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    KULLANIMORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    KULLANILMISMIKTAR DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(15),
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    CIKISMIKTARI2 DOUBLE PRECISION,
    IADEMIKTARI2 DOUBLE PRECISION,
    RECETEFIREORANI DOUBLE PRECISION,
    AKTICOZGU INTEGER,
    AKTICOZGUSTR VARCHAR(5),
    TELSAYISI INTEGER,
    SIRANO INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    AGIRLIKKILO BOYAMIKTARSIZE,
    UZUNLUKMETRE BOYAMIKTARSIZE,
    HATASTR VARCHAR(500),
    IPLIKSIKLIGICM DOUBLE PRECISION,
    DISKADI ADI100,
    DISKNO ADI100,
    DISKROTASYONU ADI255,
    DISKTEMP ADI255,
    DISKGAUGE ADI255,
    UZAMA DOUBLE PRECISION,
    TANSIYONAYARI DOUBLE PRECISION,
    MMRACK INTEGER)
AS
declare variable TahminiFireOrani Double Precision;
declare variable FTalimatReceteFireOrani Double Precision;
declare variable UrunHamId Integer;
declare variable OrmeTalimattaReceteFiresiYap Integer;
declare variable OtomatikUCikislardaReceteFire integer;
declare variable FHareketTipi integer;
declare variable FHamIade integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable FMiktar Double Precision;
declare variable HareketVar integer;
begin
  Select shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim, ShtUretim from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis,  :shtUretimeCikis, :shtFasonUretim, :ShtUretim;

  For select DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlamaRecete.DokumaPlanlamaReceteId, DokumaPlanlama.OrmeSiparisDetayId,
             DokumaPlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, Ham.Adi2, DokumaPlanlamaRecete.RenkId, Renk.Adi, DokumaPlanlamaRecete.Lot, DokumaPlanlamaRecete.UreticiId, Uretici.Adi, DokumaPlanlamaRecete.Fason,
             DokumaPlanlamaRecete.KullanimOrani, DokumaPlanlamaRecete.CikisMiktari, DokumaPlanlamaRecete.IadeMiktari,
             DokumaPlanlama.PlanlananMiktar * DokumaPlanlamaRecete.KullanimOrani / 100,
             DokumaPlanlama.GerceklesenMiktar * DokumaPlanlamaRecete.KullanimOrani / 100,
             Ik.Adi, DokumaPlanlamaRecete.InsertTarihi, Sk.Adi, DokumaPlanlamaRecete.DegisimTarihi,
             OrmeSiparisDetay.TahminiFireOrani, OrmeSiparisDetay.HamId, DokumaPlanlamaRecete.TahminiFireOrani,
             DokumaPlanlamaRecete.AtkiCozgu, decode(DokumaPlanlamaRecete.AtkiCozgu, 1, 'Atkı', 2, 'Çözgü', ''), DokumaPlanlamaRecete.TelSayisi, DokumaPlanlamaRecete.SiraNo, IplikSikligiCm,
             DiskTanitim.Adi, DiskTanitim.DiskNo, DiskTanitim.Rotasyonu, DiskTanitim.Temp, DiskTanitim.Gauge, DokumaPlanlamaRecete.Uzama, DokumaPlanlamaRecete.TansiyonAyari, DokumaPlanlamaRecete.MmRack from DokumaPlanlama
  left join DokumaPlanlamaRecete on DokumaPlanlamaRecete.DokumaPlanlamaId = DokumaPlanlama.DokumaPlanlamaId
  left join DiskTanitim On DiskTanitim.DiskTanitimId = DokumaPlanlamaRecete.DiskTanitimId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
  left join Ham on Ham.HamId = DokumaPlanlamaRecete.HamId
  left join Renk on Renk.RenkId = DokumaPlanlamaRecete.RenkId
  left join Uretici on Uretici.UreticiId = DokumaPlanlamaRecete.UreticiId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  Left Join Kullanici IK on IK.KullaniciId = DokumaPlanlamaRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = DokumaPlanlamaRecete.KullaniciId
  where DokumaPlanlama.BoyaSipariskayitid = :ABoyaSipariskayitid and DokumaPlanlamaRecete.DokumaPlanlamaId is not null
  Order BY AtkiCozgu, SiraNo
  into :DokumaPlanlamaId, :DokumaPlanlamaReceteId, :OrmeSiparisDetayId, :HamId, :HamTipiId, :HamTipiAdi,
       :HamAdi, :HamAdi2, :RenkID, :RenkAdi, :Lot, :UreticiId, :UreticiAdi, :Fason,
       :KullanimOrani, :CikisMiktari, :IadeMiktari, :KullanilacakMiktar, :KullanilmisMiktar,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :TahminiFireOrani, :UrunHamId, :FTalimatReceteFireOrani,
       :AktiCozgu, :AktiCozguStr, :TelSayisi, :SiraNo, :IplikSikligiCm, :DiskAdi, :DiskNo, :DiskRotasyonu, :DiskTemp, :DiskGauge, :Uzama, :TansiyonAyari, :MmRack do
  begin
    if (HamTipiID = 0 or HamTipiID = 6) then
    begin
      select first 1 OrmeTalimattaReceteFiresiYap, OtomatikUCikislardaReceteFire from Sistem into :OrmeTalimattaReceteFiresiYap, :OtomatikUCikislardaReceteFire;

      select first 1 FireOrani from HamStokRecete where HamId = :UrunHamId and ReceteHamId = :HamId into :ReceteFireOrani;
      if (OtomatikUCikislardaReceteFire = 1) then
        KullanilacakMiktar = KullanilacakMiktar / ((100 - FTalimatReceteFireOrani) / 100);
      else if (OrmeTalimattaReceteFiresiYap <> 1) then
        KullanilacakMiktar = KullanilacakMiktar / ((100 - TahminiFireOrani) / 100);
      else
        KullanilacakMiktar = KullanilacakMiktar / ((100 - ReceteFireOrani) / 100);
    end

    KullanilacakMiktar = Yuvarla(KullanilacakMiktar, 100);

    if (Fason = 1) then FasonStr = 'Fason';
    else if (Fason = 0) then FasonStr = '';
    else FasonStr = 'Bilinmiyor';

    HareketVar = 0;
    CikisMiktari2 = 0;
    IadeMiktari2  = 0;
    Lot2 = Null;

    select AgirlikKilo, UzunlukMetre, HataStr from DokumaIplikMiktariHesaplama(Null, :DokumaPlanlamaReceteId, Null, 1, Null, Null) into :AgirlikKilo, :UzunlukMetre, :HataStr;

    /*for Select Sum(CikisMiktari), Sum(IadeMiktari), Lot from
      (Select iif((HareketTipi = :shtFasonUretimeCikis or HareketTipi = :shtUretimeCikis) and (HamIade = 0), Sum(Miktar),0) CikisMiktari,
              iif((HareketTipi = :shtFasonUretim or HareketTipi = :shtUretim) and (HamIade = 1), Sum(Miktar), 0) IadeMiktari, Lot from StokHareket
      where StokHareket.DokumaPlanlamaId = :DokumaPlanlamaId and StokHareket.HamId = :HamId and IntKarsilastir(StokHareket.RenkId, :RenkId) = 1*/
      /*and (:Lot is Null or StokHareket.Lot = :Lot)*/
     /* Group By HareketTipi, HamIade, Lot) Group By Lot into :CikisMiktari2, :IadeMiktari2, :Lot2 do
    begin
      HareketVar = 1;
      suspend;
    end */

    if (HareketVar = 0) then suspend;
  end
end
