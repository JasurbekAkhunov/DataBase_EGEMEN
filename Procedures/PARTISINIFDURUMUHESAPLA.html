CREATE PROCEDURE PARTISINIFDURUMUHESAPLA(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER)
AS
declare variable FReceteId varchar(20);
declare variable FPartiReceteId varchar(20);
declare variable FPartiAsamalariId Integer;
declare variable FAsamaId Integer;
declare variable FGirisTarihi timestamp;
declare variable FCikisTarihi timestamp;
declare variable FReceteveyaSonrasimi integer;
declare variable FRecetelimi integer;
declare variable FSonAsamami integer;
declare variable FPartilememi integer;
declare variable SDurumu integer;
declare variable SPartiAsamalariId integer;
declare variable SonCikisKilo   Numeric(12,3);
declare variable SonCikisMetre  Numeric(12,3);
declare variable SonCikisPartiAsamalariId  integer;
declare variable PartininAsilRecetesimi    integer;
declare variable IslemGrubuId integer;
declare variable AraKontrolAsamasimi integer;
declare variable FSonAsamamiKontrol integer;
declare variable FPartiIadeRafi varchar(10);
declare variable FPartiAsamalariSonAsama integer;
declare variable CikisRafID varchar(10);
declare variable YariMamulRafla EgeBoolean;
declare variable YariMamul EgeBoolean;
declare variable YariMamulTamamlandi EgeBoolean;
begin
  SDurumu = 0;
  FReceteVeyaSonrasimi = 0;
  FRecetelimi = 0;
  SPartiAsamalariId = null;
  SonCikisPartiAsamalariId = Null;
  YariMamul = 0;
/*
  select Parti.Receteid from Parti where Parti.PartiKayitId = :APartiKayitId into :FPartiReceteid;

  if (FPartiReceteId is not null) then FRecetelimi = 1;
*/
  select Parti.IslemGrubuId from Parti where Parti.PartiKayitId = :APartiKayitId into :IslemGrubuId;

  if (IslemGrubuId <> 4) then FRecetelimi = 1;

  for Select T.SonAsamami, T.Partilememi, T.PartiAsamalariId, T.ReceteId, T.AsamaId, T.GirisTarihi, T.Cikistarihi,
             T.PartininAsilRecetesimi, T.AraKontrolAsamasimi, T.PartiIadeRafi, T.SonAsama, T.YariMamulRafla
    From
     (select Asama.SonAsamami, Asama.Partilememi, PA.PartiAsamalariId, PA.ReceteId, PA.AsamaId, PartiAsamaStok.GirisTarihi, PartiAsamaStok.Cikistarihi,
              PA.PartininAsilRecetesimi, Asama.AraKontrolAsamasimi, Asama.PartiIadeRafi,
              iif(PA.AsamaId = (Select First 1 PartiAsamaStok.AsamaId from PartiAsamaStok
                                 Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
                                 Where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.AsamayaDahil = 1
                                 Order By PartiAsamalari.Sira Desc), '1','0') SonAsama,
              Asama.YariMamulRafla, PA.Sira from PartiAsamalari PA
       left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PA.PartiAsamalariId and PartiAsamaStok.PartiSinifId = :APartiSinifId
       left join asama on asama.asamaid = PartiAsamaStok.asamaid
       where PA.PartiKayitId = :APartiKayitId and
         (PartiAsamaStok.asamayadahil = 1 or PA.PartininAsilRecetesimi = 1) and PA.OtomasyonaDahilmi = 1
     ) T
     order by T.sira Asc -- Order by yavaşlattığı için önce selectleyip sonra sıralattık
    into :FSonAsamami, :FPartilememi, :FPartiAsamalariId, :FReceteId, :FAsamaId, :FGirisTarihi, :FCikistarihi, :PartininAsilRecetesimi, :AraKontrolAsamasimi, :FPartiIadeRafi, :FPartiAsamalariSonAsama, :YariMamulRafla do
  begin
    if (FCikisTarihi is not null) then SonCikisPartiAsamalariId = FPartiAsamalariId;

    if (FGirisTarihi is not null) then SPartiAsamalariId = FPartiAsamalariId;

    if (YariMamulRafla = 1) then YariMamul = 1;

    if (AraKontrolAsamasimi = 0) then
    begin
      if (FRecetelimi = 1) then
      begin
        if (FReceteid is not null and PartininAsilRecetesimi = 1) then
        begin
          FReceteveyaSonrasimi = 1;
          if (FGirisTarihi is not null) then SDurumu = 4;

          if (FCikisTarihi is not null) then  SDurumu = 5;
        end
        else if (FReceteveyaSonrasimi = 1) then
        begin
          if (FGirisTarihi is not null) then
          begin
            SDurumu = 6;
          end
        end else if (FGirisTarihi is not null) then
        begin
          SDurumu = 2;
        end
      end
      else if (FGirisTarihi is not null) then
      begin
        SDurumu = 3;
      end

      if (FGirisTarihi is not null) then
      begin
        if ((FSonAsamami = 1) and (SDurumu <> 8)) then
        begin
          /*Yapilmamis baska son asama var mi kontrol ediliyor, varsa sevke hazira cekilmiyor */
          select count(*) from PartiAsamaStok
            left join Asama on Asama.Asamaid = PartiAsamaStok.Asamaid
            where PartiAsamaStok.PartiSinifId = :APartiSinifId and Asama.SonAsamami = 1 and
                 PartiAsamaStok.AsamayaDahil = 1 and PartiAsamaStok.GirisTarihi is Null
            into :FSonAsamamiKontrol;
          if (FSonAsamamiKontrol = 0) then
          begin
            SDurumu = 7;      /* SevkeHazir;*/
          end
        end
        if (FPartilememi = 1) then SDurumu = 1;     /*Partilendi;*/
      end
    end
  end
  /*Yarı Mamul Partilerinin durumu sevke hazıra düşmediği için partinin işi bitsede İşletmedeki stok raporu vb. yerlerde geliyor. PartiStok.Durumu alanına müdahale çok maliyetli olacağı için yeni bir alan ile çözeceğiz.*/
  if ((YariMamul = 1) and (FPartiAsamalariSonAsama = 1) and (FSonAsamami = 0)) then
  begin
   if (FCikisTarihi is not Null) then YariMamulTamamlandi = 1;
   else YariMamulTamamlandi = 0;
  end

  if (SonCikisPartiAsamalariId is not null) then
  begin
    Select Sum(PartiAsamaStokDetay.CikisKilo), Sum(PartiAsamaStokDetay.CikisMetre) from PartiAsamaStokDetay
      where PartiAsamaStokDetay.PartiAsamalariID = :SonCikisPartiAsamalariId and PartiAsamaStokDetay.PartiSinifID = :APartiSinifId
      into :SonCikisKilo, :SonCikisMetre;

    update PartiStok set Durumu = :SDurumu,
                         YariMamulTamamlandi = :YariMamulTamamlandi,
                         PartiAsamalariId = :SPartiAsamalariId,
                         SonCikisKilo = :SonCikisKilo,
                         SonCikisMetre = :SonCikisMetre
        where PartiKayitId = :APartiKayitId and PartiSinifId = :APartiSinifId and
             (Durumu <> :SDurumu or (PartiAsamalariId <> :SPartiAsamalariId or PartiAsamalariId is null) or SonCikisKilo <> :SonCikisKilo or SonCikisMetre <> :SonCikisMetre or YariMamulTamamlandi <> :YariMamulTamamlandi);

    Update PartiAsamaStok set GirisKilo = :SonCikisKilo,
                              GirisMetre = :SonCikisMetre
                              where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.CikisTarihi is null and
                                    (GirisKilo <> :SonCikisKilo or GirisMetre <> :SonCikisMetre);
  end
  else
  begin
    update PartiStok set Durumu = :SDurumu,
                         YariMamulTamamlandi = :YariMamulTamamlandi,
                         PartiAsamalariId = :SPartiAsamalariId
        where PartiKayitId = :APartiKayitId and PartiSinifId = :APartiSinifId;
  end

  /* Yarı mamüller hamstokta raflanmaya başladığı için bu kısım iptal edildi. Emre-Fahri. 18.07.2019
  if ((FPartiAsamalariSonAsama = 1) and (FSonAsamami = 0) and (FCikisTarihi is not Null)) then
  begin
    Select CikisRafId from PartiAsamaStokDetay
      Left Join Raf on Raf.RafID = PartiAsamaStokDetay.CikisRafId
      where PartiAsamaStokDetay.PartiAsamalariID = :FPartiAsamalariId and PartiAsamaStokDetay.PartiSinifID = :APartiSinifId  and Raf.YariMamul = 1
    into :CikisRafID;

    if (CikisRafID is not null) then
    begin
      insert into PartiStokIptal (PartiStokIptalYil, PartiStokIptalId, PartiSinifId, IptalEdenKullaniciId, RafId, OtomatikIade, TeslimAlanKullaniciId, TeslimAlinmaTarihi) Values
                                 (2012,-1, :APartiSinifId, 1, :CikisRafID, 1, 1, current_TimeStamp(0));
    end
    else if (FPartiIadeRafi is not null) then
    begin
      insert into PartiStokIptal (PartiStokIptalYil, PartiStokIptalId, PartiSinifId, IptalEdenKullaniciId, RafId, OtomatikIade, TeslimAlanKullaniciId, TeslimAlinmaTarihi) Values
                                 (2012,-1, :APartiSinifId, 1, :FPartiIadeRafi, 1, 1, current_TimeStamp(0));
    end
  end*/

end
