CREATE PROCEDURE FIRMANINDETAYLIHAMSTOKLISTESI_5(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AORMEGIRISIRSALIYENO IRSALIYENOSIZE,
    AEVRAKNO VARCHAR(10),
    AHAMSTOKMASTERID INTEGER,
    ATAMIR INTEGER,
    AFASON INTEGER,
    ARAPORHAMTIPI INTEGER,
    ATAMIRTIPIID INTEGER,
    ASIFIROLANLARIDOK INTEGER,
    AHAMID INTEGER,
    ALOT VARCHAR(150),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AREFERANSFIRMAID VARCHAR(20),
    AORMECIID INTEGER,
    ASINIF VARCHAR(1),
    AHAMSTOKHAREKETTIPIID INTEGER,
    AIRSALIYEGRUPID VARCHAR(4),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ASIPARISFIRMAID VARCHAR(20),
    AIPLIKTALEPYIL INTEGER,
    AIPLIKTALEPID INTEGER,
    AURETIMPARTINO VARCHAR(30),
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AMUSTERIPARTINO VARCHAR(30),
    AVERILENSIPARISKAYITID INTEGER,
    AREFERANSKODU VARCHAR(30),
    ATARIHFILTRETIPI INTEGER,
    AKALITEID VARCHAR(20))
RETURNS (
    HAMSTOKMASTERID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIKAYITID INTEGER,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ISLEMGRUBUADI VARCHAR(20),
    ISLEMTIPIADI VARCHAR(20),
    MODELNO VARCHAR(20),
    MIKTAR NUMERIC(12,3),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30))
AS
begin
/* hamstogun eski partisinin bilgileri */
  for select FirmaninDetayliHamStokListesi_1.HamStokMasterId, EskiParti.PartiYil, EskiParti.PartiId, EskiParti.PartiKayitId, EskiParti.ReceteId, Recete.Adi, IslemGrubu.Adi,
          IslemTipi.Adi, EskiParti.ModelNo, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, EskiParti.FirmaId, Firma.Adi,
          (select sum(Miktar) from Partistok where Partistok.PartiKayitId = Eskiparti.PartiKayitId) from FirmaninDetayliHamStokListesi_1(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :AIrsaliyeNo, :AOrmeGirisIrsaliyeNo, :AEvrakNo, :AHamStokMasterId, :ATamir, :AFason, :ARaporHamTipi,
                                     :ATamirTipiId, :ASifirOlanlariDok, :AHamId, :ALot, :AMusteriSiparisNo, :AReferansSiparisNo, :AReferansFirmaId, :AOrmeciId, :ASinif, :AHamStokHareketTipiId, :AIrsaliyeGrupId,
                                     :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id, :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id,
                                     :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ABoyaSiparisYil, :ABoyaSiparisId, :ASiparisFirmaId, :AIplikTalepYil, :AIplikTalepId, :AUretimPartiNo, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AMusteriPartiNo, :AVerilenSiparisKayitId, :AReferansKodu, null, null, :ATarihFiltreTipi, :AKaliteId)
  left join Parti EskiParti on EskiParti.PartiKayitId = FirmaninDetayliHamStokListesi_1.EskiPartiKayitId
  left join IslemTipi on IslemTipi.IslemTipiId = EskiParti.IslemTipiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = EskiParti.IslemGrubuId
  left join Recete on Recete.ReceteId = EskiParti.ReceteId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = EskiParti.BoyaSiparisKayitId
  left join Firma on Firma.FirmaId = EskiParti.FirmaId
  where FirmaninDetayliHamStokListesi_1.EskiPartiKayitId is not null and EskiParti.PartiKayitId is not null
  order by EskiParti.PartiKayitId
  into :HamStokMasterId, :PartiYil, :PartiId, :PartiKayitId, :ReceteId, :ReceteAdi, :IslemGrubuAdi,
        :IslemTipiAdi, :ModelNo, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :Miktar do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    suspend;
  end
end
