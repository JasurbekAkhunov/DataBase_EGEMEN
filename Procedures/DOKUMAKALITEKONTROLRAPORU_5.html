CREATE PROCEDURE DOKUMAKALITEKONTROLRAPORU_5(
    ADOKUMAMAKINAID VARCHAR(10),
    ADOKUMAPLANLAMAID INTEGER,
    ASAKAT INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKONTROLDURUMU INTEGER,
    AHAMID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    AKUMASDURUMU INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMADI ADI255,
    DOKUNANMIKTAR BOYAMIKTARSIZE,
    PLANLANANMIKTAR BOYAMIKTARSIZE,
    FASON SMALLINT,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DOKUNANKILO DOUBLE PRECISION,
    KUMASCINSIADI VARCHAR(40),
    DOKUMADESENIADI VARCHAR(20))
AS
declare variable shtFasonUretim Integer;
declare variable uitDokuma Integer;
begin
  select shtFasonUretim from OrmeStokHareketTipiGetir into :shtFasonUretim;
  select uitDokuma from UretimIslemTipiGetir into :uitDokuma;

  for select Firmaid,FirmaAdi,Fason, HamId, HamAdi, Sum(Miktar), Sum(PlanlananMiktar),KumasCinsiAdi,sum(DokunanKilo),DokumaDeseniAdi from
        (select DokumaPlanlama.FirmaId, Firma.Adi as FirmaAdi, 0 Fason, DokumaKaliteKontrolRaporu.HamId, DokumaKaliteKontrolRaporu.HamAdi, Sum(DokumaKaliteKontrolRaporu.Miktar) Miktar, Max(DokumaPlanlama.PlanlananMiktar) PlanlananMiktar,
        KumasCinsi.Adi as KumasCinsiAdi ,sum(DokumaKaliteKontrolRaporu.Kilo) As DokunanKilo ,DokumaDeseni.Adi as DokumaDeseniAdi
        from DokumaKaliteKontrolRaporu(:ADokumaMakinaId, :ADokumaPlanlamaId, :ASakat, :ABaslangicTarihi, :ABitisTarihi, :AKontrolDurumu, :AHamId, :ABoyaSiparisKayitId, :AKumasDurumu)
        left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaKalitekontrolRaporu.DokumaPlanlamaId
        left join  Firma On Firma.FirmaId = DokumaPlanlama.FirmaId
        left join Ham on Ham.HamId=DokumaKalitekontrolRaporu.HamId
        left join KumasCinsi on KumasCinsi.KumasCinsiId= Ham.KumasCinsiId
        left join DokumaDeseni on DokumaDeseni.DokumaDeseniId=Ham.DokumaDeseniId
        group by DokumaPlanlama.FirmaId,Firma.Adi, DokumaKaliteKontrolRaporu.HamId, DokumaKaliteKontrolRaporu.HamAdi, DokumaKaliteKontrolRaporu.DokumaPlanlamaId, KumasCinsi.Adi,DokumaDeseni.Adi
          union all

        select DokumaPlanlama.FirmaId, Firma.Adi, 1 Fason, StokHareket.HamId, Ham.Adi, Sum(StokHareket.Miktar), Max(DokumaPlanlama.PlanlananMiktar),KumasCinsi.Adi as KumasCinsiAdi,
        Sum(StokHareket.Kilo) as DokunanKilo,DokumaDeseni.Adi as DokumaDeseniAdi
        from StokHareket
        left join Ham on Ham.HamID = StokHAreket.HamId
        left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = StokHAreket.DokumaPlanlamaId
        left join  Firma On Firma.FirmaId = DokumaPlanlama.FirmaId
        left join KumasCinsi on KumasCinsi.KumasCinsiId= Ham.KumasCinsiId
        left join DokumaDeseni on DokumaDeseni.DokumaDeseniId=Ham.DokumaDeseniId
        where StokHareket.HareketTipi = :shtFasonUretim and
        StokHareket.HamIade = 0 and
        StokHareket.DokumaHareketi = 1 and
        (StokHareket.Tarih >= cast(:ABaslangicTarihi as date) and StokHareket.Tarih <= cast(:ABitistarihi as date)+1) and
        StokHAreket.UretimIslemTipiId = :uitDokuma and
        (StokHareket.DokumaPlanlamaId = :ADokumaPlanlamaId or :ADokumaPlanlamaId is null) and
        (StokHareket.HamId = :AHamId or :AHamId is null) and
        (StokHareket.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
        (StokHareket.Sakat = :ASakat or :ASakat is null)
        group by DokumaPlanlama.FirmaId,Firma.Adi,StokHareket.HamId, Ham.Adi, StokHAreket.DokumaPlanlamaId, KumasCinsi.Adi,DokumaDeseni.Adi)
      group by FirmaId,FirmaAdi, Fason, HamId, HamAdi,KumasCinsiAdi,DokumaDeseniAdi
      order by FirmaId,HamId
  into :FirmaId,FirmaAdi,:Fason, :HamId,  :HamAdi, :DokunanMiktar, :PlanlananMiktar,:KumasCinsiAdi,:DokunanKilo,:DokumaDeseniAdi do
  begin
    suspend;
  end
end
