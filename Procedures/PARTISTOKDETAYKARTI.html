CREATE PROCEDURE PARTISTOKDETAYKARTI(
    APARTIKAYITID INTEGER)
RETURNS (
    HAMADI ADI255,
    SINIF VARCHAR(1),
    HAMTIPIID INTEGER,
    HAMTIPIADI ADI20,
    RAFID RAFKODUSIZE,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    BOYASIPARISREZERVID INTEGER,
    RAFDETAYID INTEGER,
    HAMID INTEGER,
    KAPADEDI INTEGER,
    MIKTAR NUMERIC(12,3),
    KILO NUMERIC(12,3),
    KILODARA NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    BOBINSAYISI INTEGER,
    LINKISLEMI INTEGER,
    PARTILEMETARIHI TIMESTAMP,
    KILODEGISTIRENPERSONELID INTEGER,
    PARTISTOKBOLUNUYOR SMALLINT,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    HAMSTOKMASTERID INTEGER,
    PUS INTEGER,
    LOT VARCHAR(150),
    GRM2 INTEGER,
    FEINE INTEGER,
    TUPACIKENMAYLI INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIPARTINO VARCHAR(30),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    ESKIPARTIYIL INTEGER,
    ESKIPARTINO INTEGER,
    KILODEGISTIRENPERSONELADI ADI40,
    IPTALPARTIFIREORANI DOUBLE PRECISION,
    IPTALPARTIYIL INTEGER,
    IPTALPARTIID INTEGER,
    INSERTKULLANICIADI ADI40,
    KULLANICIADI ADI40,
    YARIMAMULPARTINOSTR VARCHAR(20),
    OLCUID INTEGER,
    OLCUADI VARCHAR(40),
    URETIMPARTINO VARCHAR(30),
    BOYASIPARISKAYITID INTEGER,
    ORMEPLANLAMAID INTEGER,
    GIRISFIRMAADI VARCHAR(100),
    GIRISFIRMAID VARCHAR(20),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    REFERANSSIPARISNO VARCHAR(30),
    TUPACIKENMAYLISTR VARCHAR(10),
    HAMSTOKHAREKETTIPIIDSTR VARCHAR(10),
    FASONSTR VARCHAR(5),
    TAMIRMI VARCHAR(10),
    TAMIRTIPIID INTEGER,
    TAMIRTIPIADI VARCHAR(20))
AS
declare variable YariMamulPartiYil integer;
declare variable YariMamulPartiId  integer;
declare variable Fason integer;
begin
  for select ham.adi hamadi, hamstokirsaliye.sinif, ham.hamtipiid, hamtipi.adi hamtipiadi, rafdetay.rafid,
         partistokdetay.partikayitid, partistokdetay.partisinifid, partistokdetay.boyasiparisrezervid,
         partistokdetay.rafdetayid, partistokdetay.hamid, partistokdetay.kapadedi, partistokdetay.miktar,
         partistokdetay.kilo, partistokdetay.kilodara, partistokdetay.metre, partistokdetay.adet,
         partistokdetay.bobinsayisi, partistokdetay.linkislemi, partistokdetay.partilemetarihi,
         partistokdetay.kilodegistirenpersonelid, partistokdetay.partistokbolunuyor, partistokdetay.insertkullaniciid,
         partistokdetay.inserttarihi, partistokdetay.kullaniciid, partistokdetay.degisimtarihi, hamstokdetay.hamstokmasterid, hamstokmaster.pus,
         hamstokmaster.lot, hamstokmaster.grm2, hamstokmaster.feine, hamstokmaster.tupacikenmayli, hamstokmaster.musterisiparisno, hamstokmaster.musteripartino,
         hamstokirsaliye.irsaliyeno, hamstokirsaliye.irsaliyetarihi, hamstokmaster.eskipartiyil, hamstokmaster.eskipartino, personel.adi kilodegistirenpersoneladi,
         rafdetay.iptalpartifireorani, iptalparti.partiyil iptalpartiyil, iptalparti.partiid iptalpartiid,
         ik.adi insertkullaniciadi, sk.adi kullaniciadi, Parti.PartiYil, Parti.PartiId, Olcu.OlcuId, Olcu.OlcuAdi,
         OrmePlanlama.UretimPartiNo, OrmePlanlama.boyasipariskayitid, OrmePlanlama.OrmePlanlamaId, Firma.Adi, Firma.FirmaId, HamStokMaster.BirimFiyat, HamStokIrsaliye.DovizId,
         HamStokMaster.ReferansSiparisNo, HamStokIrsaliye.Fason, DisTamir.TamirTipiId, TamirTipi.Adi   from partistokdetay
  left join rafdetay  on rafdetay.rafdetayid = partistokdetay.rafdetayid
  left join ham on ham.hamid = partistokdetay.hamid
  left join hamtipi on hamtipi.hamtipiid = ham.hamtipiid
  left join hamstokirsaliye on hamstokirsaliye.hamstokirsaliyeid = rafdetay.hamstokirsaliyeid
  left join hamstokdetay on hamstokdetay.hamstokdetayid = rafdetay.hamstokdetayid
  left join hamstokmaster on hamstokmaster.hamstokmasterid = hamstokdetay.hamstokmasterid
  Left Join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = HamStokMaster.OrmePlanlamaId
  Left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
  left join kullanici ik on ik.kullaniciid = partistokdetay.insertkullaniciid
  left join kullanici sk on sk.kullaniciid = partistokdetay.kullaniciid
  left join personel on personel.personelid = partistokdetay.kilodegistirenpersonelid
  left join partistok iptalpartistok on iptalpartistok.partistokiptalkayitid = rafdetay.partistokiptalkayitid
  left join parti iptalparti on iptalparti.partikayitid = iptalpartistok.partikayitid
  left join PartiStok on PartiStok.PartiSinifId = HamStokMaster.PartiSinifId
  left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
  left join Olcu on Olcu.OlcuID = HamStokMaster.OlcuId
  left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
  left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  where partistokdetay.partikayitid = :apartikayitid and partistokdetay.hamid is not null into
  :hamadi, :sinif, :hamtipiid, :hamtipiadi, :rafid, :partikayitid, :partisinifid, :boyasiparisrezervid, :rafdetayid,
  :hamid, :kapadedi, :miktar, :kilo, :kilodara, :metre, :adet, :bobinsayisi, :linkislemi, :partilemetarihi, :kilodegistirenpersonelid,
  :partistokbolunuyor, :insertkullaniciid, :inserttarihi, :kullaniciid, :degisimtarihi, :hamstokmasterid, :pus, :lot,
  :grm2, :feine, :tupacikenmayli, :musterisiparisno, :musteripartino, :irsaliyeno, :irsaliyetarihi, :eskipartiyil, :eskipartino,
  :kilodegistirenpersoneladi, :iptalpartifireorani, :iptalpartiyil, :iptalpartiid, :insertkullaniciadi, :kullaniciadi, :YariMamulPartiYil, :YariMamulPartiId,
  :OlcuId, :OlcuAdi, :UretimPartiNo, :BoyaSiparisKayitId, :OrmePlanlamaId, :GirisFirmaAdi, :GirisFirmaId, :BirimFiyat, :DovizId,
  :ReferansSiparisNo, :Fason, :TamirTipiId, :TamirTipiAdi do
  begin
    FasonStr = '';
    if (Fason = 1) then FasonStr = 'Fason';

    YariMamulPartiNoStr = '';
    if (YariMamulPartiYil is not null) then
      YariMamulPartiNoStr = '('||YariMamulPartiYil||') '||YariMamulPartiId;

    Select TupAcikEnMayliStrGetir.TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;
    select HamKontrolDegerleriGetir.HamStokHareketTipiIdStr, HamKontrolDegerleriGetir.Tamirmi from HamKontrolDegerleriGetir(:HamStokMasterId) into :HamStokHareketTipiIdStr, :Tamirmi;


    suspend;
  end
end
