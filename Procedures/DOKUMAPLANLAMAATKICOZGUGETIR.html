CREATE PROCEDURE DOKUMAPLANLAMAATKICOZGUGETIR(
    ADOKUMAPLANLAMAID INTEGER)
RETURNS (
    COZGUIPI MEMO,
    ATKIIPI MEMO)
AS
declare variable FAtkiCozgu SmallInt;
declare variable FIp        Adi255;
begin
  for select DokumaPlanlamaRecete.AtkiCozgu, Ham.Adi
    from DokumaPlanlamaRecete
    left join Ham on Ham.HamId = DokumaPlanlamaRecete.HamId
    where DokumaPlanlamaRecete.DokumaPlanlamaId = :ADokumaPlanlamaId
    order by DokumaPlanlamaRecete.AtkiCozgu, DokumaPlanlamaRecete.SiraNo
    into :FAtkiCozgu, :FIp do
  begin
    if (FIp is not null and FIp <> '')  then
    begin
      if (FAtkiCozgu = 2) then
      begin
        if (CozguIpi is null) then CozguIpi = '' ;
        if (Char_Length(CozguIpi) > 0 ) then CozguIpi = CozguIpi || ',' || FIp;
        else
        CozguIpi = FIp ;
      end
      else if (FAtkiCozgu = 1) then
      begin
        if (AtkiIpi is null) then AtkiIpi = '' ;
        if (Char_Length(AtkiIpi) > 0 ) then AtkiIpi = AtkiIpi || ',' || FIp;
        else
        AtkiIpi = FIp ;
      end
    end
  end

  suspend;
end
