CREATE PROCEDURE ORMESIPDETAYRECETEGETIR(
    AORMESIPARISDETAYID INTEGER)
RETURNS (
    ORMESIPARISDETAYRECETEID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(30),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    KULLANIMORANI DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(15),
    TAHMINIFIREORANI DOUBLE PRECISION,
    ATKICOZGU INTEGER,
    ATKICOZGUSTR VARCHAR(5),
    TELSAYISI INTEGER,
    SIRANO INTEGER,
    BIRIMMALIYET DOUBLE PRECISION,
    BIRIMMALIYETDOVIZID VARCHAR(5),
    BUKUMMALIYETI DOUBLE PRECISION,
    BUKUMMALIYETIDOVIZID VARCHAR(5),
    AKTARMAMALIYETI DOUBLE PRECISION,
    AKTARMAMALIYETIDOVIZID VARCHAR(5),
    IPLIKBOYAMALIYETI DOUBLE PRECISION,
    IPLIKBOYAMALIYETIDOVIZID VARCHAR(5),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    COZGUFIREORANI DOUBLE PRECISION,
    MALIYETFIREORANI DOUBLE PRECISION,
    IPLIKSIKLIGICM DOUBLE PRECISION,
    AGIRLIKKILO BOYAMIKTARSIZE,
    UZUNLUKMETRE BOYAMIKTARSIZE)
AS
begin

  For select OrmeSiparisDetayRecete.OrmeSiparisDetayReceteID, OrmeSiparisDetayRecete.BoyaSiparisKayitID, OrmeSiparisDetayRecete.OrmeSiparisDetayId,
             OrmeSiparisDetayRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, Ham.Adi2, OrmeSiparisDetayRecete.RenkId, Renk.Adi, OrmeSiparisDetayRecete.MelanjKodu, OrmeSiparisDetayRecete.Lot, OrmeSiparisDetayRecete.UreticiId, Uretici.Adi, OrmeSiparisDetayRecete.Fason,
             OrmeSiparisDetayRecete.KullanimOrani, Ik.Adi, OrmeSiparisDetayRecete.InsertTarihi, Sk.Adi, OrmeSiparisDetayRecete.DegisimTarihi,
             OrmeSiparisDetayRecete.TahminiFireOrani, OrmeSiparisDetayRecete.BirimMaliyet, OrmeSiparisDetayRecete.BirimMaliyetDovizId,
             OrmeSiparisDetayRecete.AtkiCozgu, decode(OrmeSiparisDetayRecete.AtkiCozgu, 1, 'Atkı', 2, 'Çözgü', ''), OrmeSiparisDetayRecete.TelSayisi, OrmeSiparisDetayRecete.SiraNo, OrmeSiparisDetayRecete.BukumMaliyeti, OrmeSiparisDetayRecete.BukumMaliyetiDovizId,
             OrmeSiparisDetayRecete.AktarmaMaliyeti, OrmeSiparisDetayRecete.AktarmaMaliyetiDovizId, OrmeSiparisDetayRecete.IplikBoyaMaliyeti, OrmeSiparisDetayRecete.IplikBoyaMaliyetiDovizId,
             OrmeSiparisDetayRecete.CozguFireOrani, OrmeSiparisDetayRecete.MaliyetFireOrani, OrmeSiparisDetayRecete.IplikSikligiCm
             from OrmeSiparisDetayRecete
  left join Ham on Ham.HamId = OrmeSiparisDetayRecete.HamId
  left join Renk on Renk.RenkId = OrmeSiparisDetayRecete.RenkId
  left join Uretici on Uretici.UreticiId = OrmeSiparisDetayRecete.UreticiId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  Left Join Kullanici IK on IK.KullaniciId = OrmeSiparisDetayRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = OrmeSiparisDetayRecete.KullaniciId
  where OrmeSiparisDetayRecete.OrmeSiparisDetayID = :AOrmeSiparisDetayID
  Order By OrmeSiparisDetayRecete.AtkiCozgu, OrmeSiparisDetayRecete.SiraNo
  into :OrmeSiparisDetayReceteID, :BoyaSiparisKayitID, :OrmeSiparisDetayId, :HamId, :HamTipiId, :HamTipiAdi,
       :HamAdi, :HamAdi2, :RenkID, :RenkAdi, :MelanjKodu,  :Lot, :UreticiId, :UreticiAdi, :Fason,
       :KullanimOrani, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :TahminiFireOrani,
       :BirimMaliyet, :BirimMaliyetDovizId, :AtkiCozgu, :AtkiCozguStr, :TelSayisi, :SiraNo, :BukumMaliyeti, :BukumMaliyetiDovizId,
       :AktarmaMaliyeti, :AktarmaMaliyetiDovizId, :IplikBoyaMaliyeti, :IplikBoyaMaliyetiDovizId, :CozguFireOrani, :MaliyetFireOrani, :IplikSikligiCm do
  begin
    if (Fason = 1) then FasonStr = 'Fason';
    else if (Fason = 0) then FasonStr = '';
    else FasonStr = 'Bilinmiyor';

    select DokumaIplikMiktariHesaplama.AgirlikKilo, DokumaIplikMiktariHesaplama.UzunlukMetre  from
      DokumaIplikMiktariHesaplama(:OrmeSiparisDetayReceteId, null, null, 0, null, null )
      into :AgirlikKilo, :UzunlukMetre;

    suspend;
  end
end
