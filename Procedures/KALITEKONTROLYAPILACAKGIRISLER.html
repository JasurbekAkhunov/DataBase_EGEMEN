    ISLEMTIPIADI VARCHAR(20),

    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),

AS
declare variable shtAlis integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretim integer;
declare variable FHataPuani     Double Precision;
begin

  select shtAlis, ShtSatisIade, ShtFasonAlis, shtFasonUretim  from OrmeStokHareketTipiGetir into :shtAlis, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretim;

    For Select StokHareket.OrmeSiparisDetayId, StokHareket.BoyaSiparisDetayId, StokHareket.StokHareketId, StokHareket.OrmeFisId, StokHareket.OrmeIrsaliyeId, StokHareket.OrmeFaturaId, StokHareket.HamId,
               StokHareket.StokGirisNo, StokHareket.HareketTipi, StokHareket.HamIade, StokHareket.Tamir, StokHareket.UretimIslemTipiId,StokHareket.Tarih, StokHareket.FiiliSevkTarihi, Ham.HamTipiId,
               Ham.Adi, OrmeSiparisDetay.SiparisSiraNo, coalesce(OrmeSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.MusteriSiparisNo), StokHareket.RenkId, SHRenk.Adi, StokHareket.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, SiparisFirma.Adi, SiparisFirma.Adi2, UretimIslemTipi.Adi, StokHareket.FirmaId, SHFirma.Adi, SHFirma.Adi2,
               coalesce(StokHareket.OrmeMakinaId, StokHareket.OrmeciMakinaNo), SHOrmeMakina.Adi, StokHareket.UretimPartiNo,
               StokHareket.CikisPartiNo, StokHareket.Lot, Uretici.Adi, OrmePlanlama.OrmePlanlamaId, StokHareket.KaliteNo, KaliteAciklamaTipi.Adi,
               StokHareket.KapAdedi, StokHareket.BrutMiktar, StokHareket.Miktar, OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo,
               OrmeIrsaliye.IrsaliyeTarihi, OrmeFatura.FaturaNo, OrmeFatura.FaturaTarihi, BoyaSiparis.OrmeSiparisSekliId, OrmeSiparisSekli.Adi, BoyaSiparis.OrmeSiparisTipiId,
               OrmeSiparisTipi.Adi, Personel.Adi, StokHareket.Aciklama, Ham.KalinlikId, Kalinlik.Adi, Ham.IplikCinsiId, IplikCinsi.Adi,
               StokHareket.StokIslemTipiId, StokIslemTipi.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, StokHareket.Notu,
               StokHareket.OrmeRafId, OrmeSiparisDetay.Kapali, StokHareket.MelanjKodu, StokHareket.FlamKodu,
               OrmeSiparisDetay.RenkId, OrmeRenk.Adi, OrmeSiparisDetay.TupAcikEnMayli, StokHareket.Sakat, StokHareket.AyarTopu, Ham.Adi2, Ham.Adi3, Ham.Adi4,
               StokHareket.DepoId, Depo.Adi, StokHareket.MusteriPartiNo, OrmePlanlama.Tarih, BoyaSiparis.SiparisTarihi,
               Stokhareket.GirisKKBitisTarihi, Stokhareket.GirisKKBitirenKullaniciId, Kullanici.Adi, Olcu.OlcuId, Olcu.OlcuAdi from StokHareket
    left join Ham on Ham.HamId = StokHareket.HamId
    left join Renk SHRenk on SHRenk.RenkId = StokHareket.RenkId
    left join StokIslemTipi on StokIslemTipi.StokIslemTipiID = StokHareket.StokIslemTipiId
    left join Firma SHFirma on SHFirma.FirmaId = StokHareket.FirmaId
    left join OrmeMakina SHOrmeMakina on SHOrmeMakina.OrmeMakinaId = StokHareket.OrmeMakinaId
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = StokHareket.OrmePlanlamaId
    left join UretimIslemTipi on UretimIslemTipi.UretimIslemTipiId  = StokHareket.UretimIslemTipiId
    left join Uretici on Uretici.UreticiId = StokHareket.UreticiId
    left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
    left join OrmeFatura on OrmeFatura.OrmeFaturaId = StokHareket.OrmeFaturaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join OrmeSiparisSekli On OrmeSiparisSekli.OrmeSiparisSekliId = BoyaSiparis.OrmeSiparisSekliId
    left join OrmeSiparisTipi On OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
    left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = StokHareket.BoyaSiparisDetayId
    left join Desen on Desen.DesenId = StokHareket.DesenId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = StokHareket.DesenVaryantId
    left join Renk OrmeRenk on OrmeRenk.RenkId = OrmeSiparisDetay.RenkId
    left join Personel on Personel.PersonelId = StokHareket.PersonelId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
    left join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId  = StokHareket.KaliteAciklamaTipiId
    Left join Depo on Depo.DepoId = StokHareket.DepoId
    left join Kullanici on Kullanici.KullaniciID = StokHareket.GirisKKBitirenKullaniciId
    left join Olcu on Olcu.OlcuID = StokHareket.OlcuID
    where StokHareket.GirisCikis = 1 and
          (StokHareket.Tarih >= :ABaslangicTarihi and StokHareket.Tarih <= :ABitisTarihi) and
          ((:ARaporTipi = 0 and StokHareket.GirisKKBitirenKullaniciId is Null) or
           (:ARaporTipi = 1 and StokHareket.GirisKKBitirenKullaniciId is not Null) or
           (:ARaporTipi  is Null )) and
          (StokHareket.UretimIslemTipiID = :AUretimIslemTipi or :AUretimIslemTipi is Null) and
          (StokHareket.StokHareketId = :AStokHareketId or :AStokHareketId is null) and
          (StokHareket.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          (BoyaSiparis.FirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
          (OrmeIrsaliye.irsaliyeno = :AIrsaliyeNo or :AIrsaliyeNo is null) and
          (StokHareket.FirmaId = :AFirmaId or :AFirmaId is null) and
          (Ham.HamTipiId in(1, 2)) -- Sadece Kumaşlar Gelsin
    order by StokHareket.Tarih, StokHareket.HamId, StokHareket.StokHareketId
    into :OrmeSiparisDetayId, :BoyaSiparisDetayId, :StokHareketId, :OrmeFisId, :OrmeIrsaliyeId, :OrmeFaturaId, :HamId,
         :StokGirisNo, :HareketTipi,  :HamIade, :Tamir, :UretimIslemTipiId, :Tarih, :FiiliSevkTarihi, :HamTipiId,
         :HamAdi, :SiparisSiraNo, :MusteriSiparisNo, :RenkId, :RenkAdi, :DesenId, :DesenAdi, :VaryantAdi,
         :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisFirmaAdi2, :UretimIslemTipiIdStr, :FirmaId, :FirmaAdi, :FirmaAdi2,
         :OrmeMakinaId, :OrmeMakinaAdi,  :UretimPartiNo, :CikisPartiNo, :Lot, :UreticiAdi, :OrmePlanlamaId, :KaliteNo, :KaliteAciklamaTipiAdi,
         :KapAdedi, :BrutMiktar, :Miktar, :Fisno, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi, :FaturaNo, :FaturaTarihi, :OrmeSiparisSekliId, :OrmeSiparisSekliIdStr, :OrmeSiparisTipiId,
         :OrmeSiparisTipiIdStr, :PersonelAdi, :Aciklama, :KalinlikId, :KalinlikAdi, :IplikCinsiId, :IplikCinsiAdi, :StokIslemTipiId, :StokIslemTipiAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :Notu,
         :OrmeRafId, :OrmeSiparisDetayKapali, :MelanjKodu, :FlamKodu, :OrmeRenkId, :OrmeRenkAdi,
         :SipTupAcikEnMayli, :Sakat, :AyarTopu, :HamAdi2, :HamAdi3, :HamAdi4, :DepoId, :DepoAdi, :MusteriPartiNo, :TalimatTarihi,
         :SiparisTarihi, :GirisKKBitisTarihi, :GirisKKBitirenKullaniciId, :GirisKKBitirenKullaniciAdi, :OlcuId, :OlcuAdi do
    begin

      BoyaSiparisIdStr = '';
      if (BoyaSiparisYil is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:SipTupAcikenMayli) into :SipTupAcikEnMayliStr;

      if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
      else AyarTopuStr = '';


      KKHataAdedi            = 0;
      KKToplamBrutKilo       = 0;
      KKToplamDara           = 0;
      KKToplamNetKilo        = 0;
      KKToplamMetre          = 0;
      KKToplamNetMetre       = 0;
      KKToplamAdet          = 0;

      Select Sum(GirisKaliteKontrolHatalari.Adet) from GirisKaliteKontrol
        left join GirisKaliteKontrolHatalari on GirisKaliteKontrolHatalari.GirisKaliteKontrolId = GirisKaliteKontrol.GirisKaliteKontrolId
        where GirisKaliteKontrol.StokHareketId = :StokHareketId and GirisKaliteKontrolHatalari.GirisKaliteKontrolId is not null
        into :KKHataAdedi;

      Select Sum(NoktasalHataHataPuani+SurekliHataHataPuani), Sum(GirisKaliteKontrol.BrutKilo), Sum(GirisKaliteKontrol.Dara), Sum(GirisKaliteKontrol.NetKilo), Sum(GirisKaliteKontrol.Metre), Sum(GirisKaliteKontrol.Adet),
        Count(*) from GirisKaliteKontrol
        where GirisKaliteKontrol.StokHareketId = :StokHareketId
        into :FHataPuani, :KKToplamBrutKilo, :KKToplamDara, :KKToplamNetKilo, :KKToplamMetre, :KKToplamAdet, :KKToplamTopSayisi;
        KKToplamNetMetre   = KKToplamNetMetre - FHataPuani;

      if (HareketTipi = shtAlis) then HareketTipiStr = 'Alış';
      else if (HareketTipi = shtSatisIade) then HareketTipiStr = 'Satış İade';
      else if (HareketTipi = shtFasonAlis) then HareketTipiStr = 'Fason İşlenmek Üzere Giriş';
      else if (HareketTipi = shtFasonUretim and HamIade = 0) then HareketTipiStr = 'Fason Üretim';
      else if (HareketTipi = shtFasonUretim and HamIade = 1) then HareketTipiStr = '(İade) Fason Üretim';
      else HareketTipiStr = 'Bilinmiyor';

      suspend;
    end
end
