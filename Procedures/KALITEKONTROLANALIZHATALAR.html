CREATE PROCEDURE KALITEKONTROLANALIZHATALAR(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    ARECETEID VARCHAR(20),
    AHAMID INTEGER,
    AKALITENO INTEGER,
    AONAYDURUMU INTEGER)
RETURNS (
    HATATIPIID INTEGER,
    ADI VARCHAR(20),
    DIGERADI1 VARCHAR(20),
    DIGERADI2 VARCHAR(20),
    NOKTASALHATA SMALLINT,
    ESLESTIRMEKODU VARCHAR(20),
    PARCAKESME SMALLINT,
    HATATIPIPUANI INTEGER,
    KILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    TOPLAMTOPSAYISI INTEGER)
AS
declare variable ABoyaSiparisKayitId integer;
begin

  ABoyaSiparisKayitId = Null;
  if (ABoyaSiparisId is not null) then
    Select BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId into :ABoyaSiparisKayitId;

  for Select HataTipi.HataTipiId, HataTipi.Adi, HataTipi.DigerAdi1, HataTipi.DigerAdi2, HataTipi.NoktasalHata, HataTipi.EslestirmeKodu, HataTipi.ParcaKesme, HataTipi.HataTipiPuani,
            Sum(PSKaliteKontrol.Kilo) BrutKilo, Sum(PSKaliteKontrol.Kilo - PSKaliteKontrol.Dara) NetKilo, Sum(PSKaliteKontrol.Metre) NetMetre,
            Sum(PSKaliteKontrol.BrutMetre) BrutMetre, Count(PSKaliteKontrol.PSKaliteKontrolId) as ToplamTopSayisi from PSKaliteKontrolAsama
    left join PartiStok on PartiStok.PartiSinifID = PSKaliteKontrolAsama.PartiSinifID
    left join Parti on Parti.PartiKayitId = PSKaliteKontrolAsama.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    Left Join PSKaliteKontrol on PSKaliteKontrol.PSKaliteKontrolAsamaID = PSKaliteKontrolAsama.PSKaliteKontrolAsamaID
    left join PSKaliteKontrolHatalari on PSKaliteKontrolHatalari.PSKaliteKontrolId = PSKaliteKontrol.PSKaliteKontrolId
    left join HataTipi on HataTipi.HataTipiId = PSKaliteKontrolHatalari.HataTipiId
    where (PSKaliteKontrolAsama.SonKayit = 1) and
          (PSKaliteKontrol.PSKaliteKontrolId is Not Null) and
          (PSKaliteKontrolHatalari.PSKaliteKontrolHatalariId is Not Null) and
          (PSKaliteKontrolAsama.KKBitisTarihi >= :ABaslangicTarihi and PSKaliteKontrolAsama.KKBitisTarihi <= :ABitisTarihi) and
          (Parti.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          (Parti.ReceteId = :AReceteId or (:AReceteId is null or :AReceteId = '')) and
          (Firma.AnaHesapFirmaId = :AFirmaId or (:AFirmaId is null or :AFirmaId = '')) and
          (PartiStok.HamId = :AHamId or :AHamId is null) and
          (PSKaliteKontrol.KaliteNo = :AKaliteNo or :AKaliteNo is null) and
           (PSKaliteKontrol.OnayDurumu = :AOnayDurumu or :AOnayDurumu is Null)
    Group By HataTipi.HataTipiId, HataTipi.Adi, HataTipi.DigerAdi1, HataTipi.DigerAdi2, HataTipi.NoktasalHata, HataTipi.EslestirmeKodu, HataTipi.ParcaKesme, HataTipi.HataTipiPuani
    into :HataTipiId, :Adi, :DigerAdi1, :DigerAdi2, :NoktasalHata, :EslestirmeKodu, :ParcaKesme, :HataTipiPuani, :Kilo, :NetKilo, :NetMetre, :BrutMetre, :ToplamTopSayisi do
    begin
       suspend;
    end
end
