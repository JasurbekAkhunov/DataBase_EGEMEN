CREATE PROCEDURE PARTIASAMALARIOLUSTUR(
    APARTIKAYITID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable PartiSinifId         integer;
declare variable AsamaId              integer;
declare variable FAsamaId             integer;
declare variable LastSiraNo           integer;
declare variable ProsesId             integer;
declare variable PartiAsamalariId     integer;
declare variable KayitSayisi          integer;
declare variable Yer                  integer;
declare variable Sira                 integer;
declare variable OrtakYerBulundu      integer;
declare variable ReceteId             varchar(20);
declare variable UstAsamaId           integer;
declare variable BoyaSiparisDetayId   integer;
declare variable MerserizeAsamaId     integer;
declare variable ContinueKasarAsamaId integer;
declare variable BuyukPartiSinifId    integer;
declare variable MaxKilo              Numeric(12,3);
declare variable MerserizeReceteId    varchar(20);
declare variable ContinueKasarReceteId varchar(20);
declare variable ReceteDisiTakibi     integer;
declare variable AsamaDefaultReceteId varchar(20);
declare variable FEgemenFirmaId       varchar(20);
declare variable FreceteID recetekodusize;
declare variable FUstAsamaID integer;
begin
  Select Max(PartiStok.Kilo) from PartiStok where PartiStok.PartiKayitId = :APartiKayitId into :MaxKilo;

  Select EgemenFirmaid from Sistem where SistemID = 1 into :FEgemenFirmaid;

  Select Max(PartiStok.PartiSinifId) from PartiStok
    where PartiStok.PartiKayitId = :APartiKayitId and PartiStok.Kilo = :MaxKilo into :BuyukPartiSinifId;

  if (BuyukPartiSinifId is not null) then
  begin
    Select Ham.MerserizeReceteId, Ham.ContinueKasarReceteId from PartiStok
      left join Ham on Ham.HamId = PartiStok.HamId
      where PartiStok.PartiSinifId = :BuyukPartiSinifId
      into :MerserizeReceteId, :ContinueKasarReceteId;

    Select MerserizeAsamaId, ContinueKasarAsamaId from Sistem where SistemID = 1 into :MerserizeAsamaId, :ContinueKasarAsamaId;
  end


  Select Count(*) from PartiAsamalari where PartiAsamalari.PartiKayitId = :APartiKayitId into :KayitSayisi;
  if (KayitSayisi > 0) then exception PartiAsamalariZatenOluturulmus;

  ReceteId = Null;
  For Select PartiStok.PartiSinifId, PartiStok.ProsesId, PartiStok.BoyaSiparisDetayId from PartiStok
    where PartiStok.PartiKayitId = :APartiKayitId into :PartiSinifId, :ProsesId, :BoyaSiparisDetayId do
  begin
    if (ReceteId is null) then
    begin
      Select Recete.ReceteId, Coalesce(LabReceteCalisma.UstAsamaId, LabRecete.UstAsamaId) from BoyaSiparisDetay
      left join LabRecete On LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
      left join Recete On Recete.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
      left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
      left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
      where BoyaSiparisDetay.BoyaSiparisDetayID = :BoyaSiparisDetayID
      into :ReceteId, :UstAsamaId;
    end
    LastSiraNo = -1;

    for Select ProsesAsamalari.Asamaid, ProsesAsamalari.Sira, Asama.ReceteDisiTakibi, ProsesAsamalari.ReceteID, Recete.UstAsamaID from ProsesAsamalari
       Left join Asama On Asama.AsamaId = ProsesAsamalari.Asamaid
        left join Recete On Recete.ReceteId = ProsesAsamalari.ReceteID
       where ProsesAsamalari.ProsesId = :ProsesId order by ProsesAsamalari.sira
       into :AsamaId, :Sira, :ReceteDisiTakibi, :FReceteID, :FUstAsamaID do
    begin
      Yer = null;
      Select first 1 Sira, PartiAsamalariId From PartiAsamalari where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.Sira > :LastSiraNo and PartiAsamalari.AsamaId = :AsamaId
      order by sira nulls last into :Yer, :PartiAsamalariId;
      --Select Min(Sira) From PartiAsamalari where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.Sira > :LastSiraNo and PartiAsamalari.AsamaId = :AsamaId into :Yer;

      if (Yer is null) then
      begin
        --OrtakYerBulundu = 0;

        Select First 1 ProsesAsamalari.Asamaid, Min(PartiAsamalari.Sira) from ProsesAsamalari
        left join PartiAsamalari on PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.Sira > :LastSiraNo and PartiAsamalari.AsamaId = ProsesAsamalari.AsamaId
        where ProsesAsamalari.ProsesId = :ProsesId and ProsesAsamalari.Sira > :Sira
        group by ProsesAsamalari.Asamaid, ProsesAsamalari.Sira
        having Min(PartiAsamalari.Sira) is not null
        order by ProsesAsamalari.Sira into :FAsamaId, :Yer;

        /*for Select ProsesAsamalari.Asamaid from ProsesAsamalari where ProsesAsamalari.ProsesId= :ProsesId and ProsesAsamalari.Sira > :Sira  order by sira into :FAsamaId do
        begin
          if (OrtakYerBulundu = 0) then
          begin
            Select Min(Sira) From PartiAsamalari where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.Sira > :LastSiraNo and PartiAsamalari.AsamaId = :FAsamaId into :Yer;
            if (Yer is Not Null) then OrtakYerBulundu = 1;
          end
        end*/
        --if (OrtakYerBulundu = 0) then Yer = -1;

        if (Yer is null) then Yer = -1;

        if (FEgemenFirmaid= 134 ) then Yer = Sira;

        PartiAsamalariId = Gen_ID(PartiAsamalari_Id_gen, 1);

        insert into PartiAsamalari(PartiAsamalariId, PartiKayitId, AsamaId, OtomasyonaDahilmi, Sira, InsertKullaniciId, KullaniciId)
        values(:PartiAsamalariId, :APartiKayitId, :AsamaId, :ReceteDisiTakibi, :Yer, :AKullaniciId, :AKullaniciId);

        Update PartiAsamaStok Set AsamayaDahil = 0 where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId;
        Update PartiAsamaStok Set AsamayaDahil = 1 where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId;

        if (FReceteId is not null and FUstAsamaId is not null) then
        begin
          Select min(PartiAsamalariId) from PartiAsamalari
            where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.AsamaId = :FUstAsamaId
            into :PartiAsamalariId;

          Update PartiAsamalari set ReceteId = :FReceteId, AsamaId = null  where PartiAsamalari.PartiAsamalariId = :PartiAsamalariId;
        end
        if (Yer = -1) then Select Max(Sira) From PartiAsamalari where PartiAsamalari.PartiKayitId = :APartiKayitId into :Yer;
      end
      else
      begin
        --Select PartiAsamalariId From PartiAsamalari where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.Sira = :Yer into :PartiAsamalariId;
        Update PartiAsamaStok Set AsamayaDahil = 1 where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId;
      end
      LastSiraNo = Yer;
    end
  end

  if (ReceteId is not null and UstAsamaId is not null) then
  begin
    Select min(PartiAsamalariId) from PartiAsamalari
      where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.AsamaId = :UstAsamaId
      into :PartiAsamalariId;

    Update PartiAsamalari set ReceteId = :ReceteId, AsamaId = null, PartininAsilRecetesimi = 1  where PartiAsamalari.PartiAsamalariId = :PartiAsamalariId;
  end

  if (MerserizeAsamaId is not null and MerserizeReceteId is not null) then
    Update PartiAsamalari Set ReceteId = :MerserizeReceteId, AsamaID = Null where PartiAsamalari.PartiKayitId = :APartiKayitId And PartiAsamalari.AsamaId = :MerserizeAsamaId;

  if (ContinueKasarAsamaId is not null and ContinueKasarReceteId is not null) then
    Update PartiAsamalari Set ReceteId = :ContinueKasarReceteId, AsamaID = Null where PartiAsamalari.PartiKayitId = :APartiKayitId And PartiAsamalari.AsamaId = :ContinueKasarAsamaId;

  For Select PartiAsamalari.PartiAsamalariID, Asama.DefaultReceteID from PartiAsamalari
    left join Asama On Asama.AsamaID = PartiAsamalari.AsamaID
    where PartiAsamalari.PartiKayitID = :APartiKayitID and Asama.DefaultReceteID is not null and Asama.DefaultReceteID <> ''
    into :PartiAsamalariId, :AsamaDefaultReceteId do
  begin
    Update PartiAsamalari Set AsamaID = NUll, ReceteID = :AsamaDefaultReceteId where PartiAsamalari.PartiAsamalariID = :PartiAsamalariId;
  end
end
