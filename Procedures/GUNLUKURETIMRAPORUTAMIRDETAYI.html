CREATE PROCEDURE GUNLUKURETIMRAPORUTAMIRDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP)
RETURNS (
    DISTAMIRCOZUMSEKLIID VARCHAR(10),
    DISTAMIRCOZUMSEKLIADI VARCHAR(50),
    KILO DOUBLE PRECISION,
    AYLIKKILO DOUBLE PRECISION)
AS
begin
  For Select DisTamirCozum.DisTamirCozumSekliId, DisTamirCozumSekli.Adi, Sum(DisTamirCozum.Kilo) from HamStokDetay
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
    left join DisTamirCozum on DisTamirCozum.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliId = DisTamirCozum.DisTamirCozumSekliID
    where HamStokDetay.Tarih >= :AAylikBaslangic and HamStokDetay.Tarih <= :AAylikBitis and HamStokMaster.Tamir = 1 and DisTamir.TDBKullaniciId is not null
    group by DisTamirCozum.DisTamirCozumSekliId, DisTamirCozumSekli.Adi
    into :DisTamirCozumSekliId, :DisTamirCozumSekliAdi, :AylikKilo do
  begin
    Select Sum(DisTamirCozum.Kilo) from HamStokDetay
      left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join DisTamirCozum on DisTamirCozum.HamStokMasterId = HamStokDetay.HamStokMasterId
      where HamStokDetay.Tarih >= :ABaslangicTarihi and HamStokDetay.Tarih <= :ABitisTarihi and HamStokMaster.Tamir = 1 and DisTamir.TDBKullaniciId is not null and
            (DisTamirCozum.DisTamirCozumSekliId = :DisTamirCozumSekliId or (:DisTamirCozumSekliId is null and DisTamirCozum.DisTamirCozumSekliId is null) )
      into :Kilo;
    suspend;
  end
end
