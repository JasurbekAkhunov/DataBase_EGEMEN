CREATE PROCEDURE SEVKIYATPLANLAMALISTESI(
    AFIRMAID VARCHAR(20),
    AHAMTIPIID INTEGER,
    ATARTIDURUMU INTEGER,
    AFASON INTEGER,
    ADEPOID VARCHAR(10),
    ASIPARISIALANKULLANICIID INTEGER)
RETURNS (
    SEVKONCELIKNO INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTITIPIID INTEGER,
    TERMINTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ACILIYETDERECESIADI VARCHAR(20),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ISLEMTIPIADI VARCHAR(20),
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    SEVKEHAZIRKAPADEDI INTEGER,
    SEVKEHAZIRBOBINSAYISI INTEGER,
    SEVKEHAZIRNETKILO NUMERIC(12,3),
    SEVKEHAZIRNETMETRE NUMERIC(12,3),
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    TAMAMITARTILDI INTEGER,
    TAMAMITARTILDISTR VARCHAR(4),
    FIRMASEVKADRESIID INTEGER,
    FIRMASEVKADRESIADI VARCHAR(50),
    SONYAPILANASAMAADI VARCHAR(40),
    SONMAKINAID VARCHAR(255),
    DEPOBILGISI VARCHAR(150),
    KALITEID VARCHAR(20),
    KALITEADI ADI255,
    HAMLOT VARCHAR(255),
    HAMSATISTRANSFERKODU VARCHAR(55),
    SATISTRANSFERKODU VARCHAR(55),
    DESENADI TYPE OF COLUMN DESEN.ADI,
    DESENID TYPE OF COLUMN DESEN.DESENID,
    DESENTASARIMNO TYPE OF COLUMN DESEN.TASARIMNO,
    VARYANTADI TYPE OF COLUMN DESENVARYANT.VARYANTADI,
    VARYANTACIKLAMA TYPE OF COLUMN DESENVARYANT.ACIKLAMA,
    HAMREFERANSKODU TYPE OF COLUMN HAM.REFERANSKODU)
AS
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
declare variable DepoKayitSayisi integer;
declare variable MaxSira  integer;
declare variable SonKontrolAsamaId integer;
declare variable IplikSonKontrolAsamaId integer;
declare variable FMakinaId      varchar(10);
declare variable FDepoId        varchar(10);
declare variable FLot           varchar(150);

begin
  Select hstIplik, hstElyaf, hstTelef, hstDokumaKumas, hstOrmeKumas, hstRibana, hstAksesuar
  from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef, :hstDokumaKumas, :hstOrmeKumas, :hstRibana, :hstAksesuar;
  Select SonKontrolAsamaId, IplikSonKontrolAsamaId from Sistem Where SistemId = 1 into :SonKontrolAsamaId, :IplikSonKontrolAsamaId;

  for select PartiStok.SevkOncelikNo, PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, Parti.FirmaId, Firma.Adi, Parti.BoyaSiparisKayitId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, IslemTipi.Adi, PartiStok.PartiSinifId, PartiStok.HamAdiToplami,
             Parti.Fason, Parti.ReceteId, Recete.Adi, Parti.TerminTarihi, AciliyetDerecesi.Adi,
             PartiStok.Kilo, PartiStok.Metre, PartiStok.NetKilo, PartiStok.NetMetre, PartiStok.TamamiTartildi, PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo,
             BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
             PartiStok.FirmaSevkAdresiId, FirmaSevkAdresi.AdresAdi,
             BoyaSiparisDetay.KaliteId, Kalite.KaliteAdi, Ham.HamSatisTransferKodu, Ham.SatisTransferKodu,
             Desen.Adi, Desen.DesenId, Desen.TasarimNo, DesenVaryant.VaryantAdi, DesenVaryant.Aciklama, Ham.ReferansKodu from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join FirmaSevkAdresi on FirmaSevkAdresi.FirmaSevkAdresiId = PartiStok.FirmaSevkAdresiId
    left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    where PartiStok.durumu = 7 and PartiStok.PartiStokIptalKayitID is null and (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
         (PartiStok.TamamiTartildi = :ATartiDurumu or :ATartiDurumu is null) and
         (Parti.Fason = :AFason or :AFason is null) and
         (BoyaSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
         ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
            (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or
            ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 2) or
            (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 3) or :AHamTipiId is null)
    order by PartiStok.durumu, PartiStok.SevkOncelikNo
    into :SevkOncelikNo, :PartiKayitId, :PartiYil, :PartiId, :PartiTipiId, :FirmaId, :FirmaAdi, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId,
         :IslemTipiAdi, :PartiSinifId, :HamAdiToplami, :Fason, :ReceteId, :ReceteAdi, :TerminTarihi, :AciliyetDerecesiAdi,
         :Kilo, :Metre, :NetKilo, :NetMetre, :TamamiTartildi, :BoyaSiparisDetayId, :MusteriSiparisNo, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
         :FirmaSevkAdresiId, FirmaSevkAdresiAdi, :KaliteId, :KaliteAdi, :HamSatisTransferKodu, :SatisTransferKodu,
         :DesenAdi, :DesenId, :DesenTasarimNo, :VaryantAdi, :VaryantAciklama, :HamReferansKodu do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    SevkeHazirBobinSayisi = Null;
    SevkeHazirKapAdedi = Null;
    SevkeHazirNetKilo = Null;
    SevkeHazirNetMetre = Null;

    Select Count(*), Sum(PartiStokTarti.KapAdedi), Sum(PartiStokTarti.BobinSayisi), Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.Metre) from PartiStok
      left join PartiStokTarti on PartiStokTarti.PartiSinifID = PartiStok.PartiSinifID
      where PartiStok.PartiSinifId = :PartiSinifId and PartiStokTarti.BoyaIrsaliyeDetayId is null and
            PartistokTarti.HamStokSarfFisiDetayId is null and (PartiStokTarti.DepoId = :ADepoId or :ADepoId is null)
       into :DepoKayitSayisi, :SevkeHazirKapAdedi, :SevkeHazirBobinSayisi, :SevkeHazirNetKilo, :SevkeHazirNetMetre;

    if (TamamiTartildi = 1) then TamamiTartildiStr = 'Evet';
    else TamamiTartildiStr = '';

    if ( (ADepoId is not null and DepoKayitSayisi > 0) or ADepoId is null or Tamamitartildi = 0) then
    begin
      select max(PartiAsamalari.Sira) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiKayitId = :PartiKayitId and PartiAsamaStok.PartiSinifId = :PartiSinifId and
           (PartiAsamalari.AsamaId <> :SonKontrolAsamaId and PartiAsamalari.AsamaId <> :IplikSonKontrolAsamaId ) and
            PartiAsamaStok.Giristarihi is not null
      into :MaxSira;

      SonMakinaId        = '';
      SonYapilanAsamaAdi = Null;

      if(MaxSira is not null) then
      begin
        For select Asama.Adi, PartiAsamaStokDetay.MakinaId from PartiAsamalari
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join Recete on Recete.Receteid = PartiAsamalari.Receteid
        left join Asama on (Asama.Asamaid = PartiAsamalari.AsamaId or Asama.Asamaid = Recete.UstAsamaId)
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.sira = :MaxSira and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
        group By Asama.Adi, PartiAsamaStokDetay.MakinaId
        into :SonYapilanAsamaAdi, :FMakinaId do
        begin
          if (Char_Length(SonMakinaId) + Char_Length(FMakinaId) < 250) then
          begin
            if (SonMakinaId <> '') then SonMakinaId = SonMakinaId || ',';

            SonMakinaId = SonMakinaId || FMakinaId;
          end
        end
      end

      DepoBilgisi = '';

      For Select DepoId from PartiStokTarti
        where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.BoyaIrsaliyeDetayId is null
        group by depoId
        into :FDepoId do
      begin
        if (Char_Length(DepoBilgisi) < 100) then
        begin
          if (DepoBilgisi <> '') then DepoBilgisi = DepoBilgisi || cast(', ' as varchar(2) character set UTF8);
          DepoBilgisi = DepoBilgisi || FDepoId;
        end
      end

      HamLot = '';
      for Select HamStokMaster.Lot from PartiStokDetay
          Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
          Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
          where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.Lot is not null and HamStokMaster.Lot <> ''
          group by HamStokMaster.Lot
          into :FLot do
      begin
        if (Char_Length(HamLot) + Char_Length(FLot) < 255) then
        begin
         if (HamLot <> '' ) then HamLot = HamLot || cast(', ' as varchar(2) character set UTF8);
         HamLot = HamLot || FLot;
        end
      end

      suspend;
    end
  end
end
