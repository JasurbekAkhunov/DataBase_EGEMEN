CREATE PROCEDURE MAKINAURETIMRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AVARDIYASAYISI INTEGER,
    AUNITEID INTEGER,
    AMAKINAGRUBU VARCHAR(4))
RETURNS (
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    MAKINAGRUPID VARCHAR(4),
    MAKINAGRUPADI VARCHAR(30),
    VARDIYA1PERSONELADI VARCHAR(255),
    VARDIYA1KILO DOUBLE PRECISION,
    VARDIYA1METRE DOUBLE PRECISION,
    VARDIYA2PERSONELADI VARCHAR(255),
    VARDIYA2KILO DOUBLE PRECISION,
    VARDIYA2METRE DOUBLE PRECISION,
    VARDIYA3PERSONELADI VARCHAR(255),
    VARDIYA3KILO DOUBLE PRECISION,
    VARDIYA3METRE DOUBLE PRECISION,
    ILAVEPROSESKILO DOUBLE PRECISION,
    UCRETSIZICTAMIRKILO DOUBLE PRECISION,
    UCRETSIZDISTAMIRKILO DOUBLE PRECISION,
    UCRETLIICTAMIRKILO DOUBLE PRECISION,
    UCRETLIDISTAMIRKILO DOUBLE PRECISION,
    GUNLUKKAPASITE DOUBLE PRECISION,
    UCRETSIZICTAMIRMETRE DOUBLE PRECISION,
    UCRETSIZDISTAMIRMETRE DOUBLE PRECISION,
    UCRETLIICTAMIRMETRE DOUBLE PRECISION,
    UCRETLIDISTAMIRMETRE DOUBLE PRECISION,
    ILAVEPROSESMETRE DOUBLE PRECISION)
AS
declare variable FMakinaId         varchar(10);
declare variable FMakinaAdi        varchar(40);
declare variable FMakinaGrupId      varchar(4);
declare variable FMakinaGrupAdi     varchar(30);
declare variable FGunlukKapasite   Double Precision;
declare variable FCikisTarihi      TimeStamp;
declare variable FPersonelAdi      varchar(40);
declare variable FPersonelKisaAdi  varchar(40);
declare variable FKilo             Double Precision;
declare variable FMetre            Double Precision;
declare variable SonMakinaId       varchar(10);
declare variable Vardiya           integer;
declare variable T1                Double Precision;
Declare variable FIslemTipiId       integer;
Declare variable FPartiTamirId integer;
Declare variable FFaturaKesilecek   integer;
Declare variable FUcretli           integer;
Declare variable FIlaveProses       integer;
begin
  SonMakinaId = '';
  for select Makina.MakinaId, Makina.Adi, Makina.GunlukKapasite, PartiAsamaStokDetay.CikisTarihi, Personel.Adi, PartiAsamaStokDetay.kilo, PartiAsamaStokDetay.metre,
             PartiAsamalari.PartiTamirId, TamirCozumSekli.Ucretli, Parti.IslemTipiId, Parti.FaturaKesilecek, PartiAsamalari.IlaveProses, Makina.MakinaGrubu, MakinaGrup.Adi from makina
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.makinaId = Makina.MakinaId and PartiAsamaStokDetay.CikisTarihi >= :ABaslangicTarihi                                                                          and PartiAsamaStokDetay.CikisTarihi <= :ABitisTarihi
    left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
    left join Personel on Personel.PersonelId = PartiAsamaStokDetay.CikisPersonelId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
    left join MakinaGrup on MakinaGrup.MakinaGrupID = Makina.MakinaGrubu
    where (Makina.UniteId = :AUniteId or :AUniteId is null) and Makina.UretimRaporlarinaDahil = 1 and
          (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
          PartiAsamaStokDetay.CikisTarihi is not null
    order by Makina.MakinaId, PartiAsamaStokDetay.CikisTarihi
    into :FMakinaId, :FMakinaAdi, :FGunlukKapasite, :FCikisTarihi, :FPersonelAdi, :FKilo, :FMetre, :FPartiTamirId, :FUcretli,
         :FIslemTipiId, :FFaturaKesilecek, :FIlaveProses, :FMakinaGrupId, :FMakinaGrupAdi  do
  begin
    if (SonMakinaId <> FMakinaId) then
    begin
      if (SonMakinaID <> '') then suspend;
      SonMakinaId = FMakinaId;
      MakinaId = FMakinaId;
      MakinaAdi = FMakinaAdi;
      MakinaGrupId  = FMakinaGrupId;
      MakinaGrupAdi = FMakinaGrupAdi;
      GunlukKapasite = FGunlukKapasite;
      Vardiya1PersonelAdi = '';
      Vardiya1Kilo = 0;
      Vardiya1Metre = 0;
      Vardiya2PersonelAdi = '';
      Vardiya2Kilo = 0;
      Vardiya2Metre = 0;
      Vardiya3PersonelAdi = '';
      Vardiya3Kilo = 0;
      Vardiya3Metre = 0;
      IlaveProsesKilo = 0;
      IlaveProsesMetre = 0;
      UcretsizIcTamirKilo = 0;
      UcretsizDisTamirKilo = 0;
      UcretsizIcTamirMetre = 0;
      UcretsizDisTamirMetre = 0;
      UcretliIcTamirKilo  = 0;
      UcretliDisTamirKilo = 0;
      UcretliIcTamirMetre = 0;
      UcretliDisTamirMetre = 0;
    end

    if (FCikisTarihi is not null) then
    begin
       T1 = FCikisTarihi - ABaslangicTarihi;
       vardiya = VardiyaHesapla(T1, AVardiyaSayisi);
    end
    else vardiya = 0;

    if (PosBul(FPersonelAdi, ' ') > 0) then FPersonelKisaAdi = Substring(FPersonelAdi From 1 For 1) || '.'|| Substring(FPersonelAdi From PosBul(FPersonelAdi, ' ') For 255);
    else FPersonelKisaAdi = FPersonelAdi;

    if (Vardiya = 1) then
    begin
      Vardiya1Kilo = Vardiya1Kilo + FKilo;
      Vardiya1Metre = Vardiya1Metre + FMetre;
      if (PosBul(Vardiya1PersonelAdi, FPersonelKisaAdi) = 0) then
      if (Char_Length(Vardiya1PersonelAdi) + Char_Length(FPersonelKisaAdi) < 250) then
      begin
       if (Vardiya1PersonelAdi <> '' ) then Vardiya1PersonelAdi = Vardiya1PersonelAdi || cast(', ' as varchar(2) character set UTF8);
       Vardiya1PersonelAdi = Vardiya1PersonelAdi || FPersonelKisaAdi;
      end
    end
    else
    if (Vardiya = 2) then
    begin
      Vardiya2Kilo = Vardiya2Kilo + FKilo;
      Vardiya2Metre = Vardiya2Metre + FMetre;
      if (PosBul(Vardiya2PersonelAdi, FPersonelKisaAdi) = 0) then
      if (Char_Length(Vardiya2PersonelAdi) + Char_Length(FPersonelKisaAdi) < 250) then
      begin
       if (Vardiya2PersonelAdi <> '' ) then Vardiya2PersonelAdi = Vardiya2PersonelAdi || cast(', ' as varchar(2) character set UTF8);
       Vardiya2PersonelAdi = Vardiya2PersonelAdi || FPersonelKisaAdi;
      end
    end
    else
    if (Vardiya = 3) then
    begin
      Vardiya3Kilo = Vardiya3Kilo + FKilo;
      Vardiya3Metre = Vardiya3Metre + FMetre;
      if (PosBul(Vardiya3PersonelAdi, FPersonelKisaAdi) = 0) then
      if (Char_Length(Vardiya3PersonelAdi) + Char_Length(FPersonelKisaAdi) < 250) then
      begin
       if (Vardiya3PersonelAdi <> '' ) then Vardiya3PersonelAdi = Vardiya3PersonelAdi || cast(', ' as varchar(2) character set UTF8);
       Vardiya3PersonelAdi = Vardiya3PersonelAdi || FPersonelKisaAdi;
      end
    end

    if (FIlaveProses = 1) then
    begin
      IlaveProsesKilo = IlaveProsesKilo + FKilo;
      IlaveProsesMetre = IlaveProsesMetre + FMetre;
    end
    else if (FPartiTamirId is not null) then
    begin
      if (FUcretli = 1) then
      begin
        UcretliIcTamirKilo = UcretliIcTamirKilo + FKilo;
        UcretliIcTamirMetre = UcretliIcTamirMetre + FMetre;
      end
      else
      begin
        UcretsizIcTamirKilo = UcretsizIcTamirKilo + Fkilo;
        UcretsizIcTamirMetre = UcretsizIcTamirMetre + FMetre;
      end
    end
    else if (FIslemTipiId = 3) then
    begin
      if (FFaturaKesilecek = 0) then
      begin
        UcretsizDisTamirKilo = UcretsizDisTamirKilo + FKilo;
        UcretsizDisTamirMetre = UcretsizDisTamirMetre + FMetre;
      end
      else
      begin
        UcretliDisTamirKilo = UcretliDisTamirKilo + FKilo;
        UcretliDisTamirMetre = UcretliDisTamirMetre + FMetre;
      end
    end
  end
  if (SonMakinaId <> '') then suspend;
end
