CREATE PROCEDURE ORMEPLANLAMASURUKLEBIRAK(
    AORMESIPARISDETAYID INTEGER,
    AORMEMAKINAID VARCHAR(10),
    AFIRMAID VARCHAR(20),
    AORMEPLANLAMAID INTEGER,
    AALTORMEPLANLAMAID INTEGER,
    AKULLANICIID INTEGER,
    AMIKTARGUN DOUBLE PRECISION)
AS
declare variable FBoyaSiparisKayitId  integer;
declare variable FSira                  integer;
declare variable FPlanlanacakMiktar     Numeric(12,3);
declare variable OPId           integer;
declare variable ASira          integer;
declare variable FTarih         Date;
declare variable FBirimFiyat    Double Precision;
declare variable FDovizId       varchar(5);
declare variable FVadeGun       integer;
declare variable FBeyazlik       integer;
declare variable OrmeTalimattaOtomatikOnayVer SmallInt;
declare variable makinaustugramaj integer;
declare variable ignesayisi       integer;
declare variable iplikboyu1       varchar(20);
declare variable iplikboyu2       varchar(20);
declare variable iplikboyu3       varchar(20);
declare variable iplikboyu4       varchar(20);
declare variable dinlenmisgramaj  integer;
declare variable maysayisi        varchar(10);
declare variable tupacikenmayli   integer;
declare variable HamEn            integer;
declare variable EgemenFirmaId    integer;
begin
  FSira = -1;
  FTarih = current_TimeStamp(0);

  Select OrmeTalimattaOtomatikOnayVer, EgemenFirmaId from Sistem where SistemID = 1 into :OrmeTalimattaOtomatikOnayVer, :EgemenFirmaId;

  if (AOrmePlanlamaId is null) then
  begin
    select BoyaSiparisKayitId, (OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.PlanlananMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar) from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId
     into :FBoyaSiparisKayitId, :FPlanlanacakMiktar;

    if (FPlanlanacakMiktar <= 0) then exception SiparisinTamamiPlanlanmis;
  end

  if (AOrmePlanlamaId is Not null) then
  begin
    if (AFirmaId is null and AOrmeMakinaId is null) then
    begin
      delete from OrmePlanlama where OrmePlanlama.OrmePlanlamaID = :AOrmePlanlamaId;
      exit;
    end
  end

  if (AAltOrmePlanlamaId is not null) then
  begin
    Select Sira from OrmePlanlama where OrmePlanlama.OrmePlanlamaId = :AAltOrmePlanlamaId into :FSira;
    if (FSira is null) then exception OrmePlanlamaYerBulunamadi;
  end

  if (AOrmePlanlamaId is null) then
  begin
    select OrmeSiparisDetay.OrmeBirimMaliyet, OrmeSiparisDetay.OrmeBirimMaliyetDovizId, OrmeSiparisDetay.VadeGun, OrmeSiparisDetay.Beyazlik from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId
     into :FBirimFiyat, :FDovizId, :FVadeGun, :FBeyazlik;

    if (AOrmeSiparisDetayId is not null) then
    begin
      select ormesiparisdetay.makinaustugramaj, ormesiparisdetay.ignesayisi, ormesiparisdetay.iplikboyu1, ormesiparisdetay.iplikboyu2, ormesiparisdetay.iplikboyu3, ormesiparisdetay.iplikboyu4,
      ormesiparisdetay.dinlenmisgramaj, ormesiparisdetay.maysayisi, ormesiparisdetay.tupacikenmayli, ormesiparisdetay.hamen from ormesiparisdetay
      where ormesiparisdetay.ormesiparisdetayid = :AOrmeSiparisDetayId into :makinaustugramaj, :ignesayisi, :iplikboyu1, :iplikboyu2, :iplikboyu3, :iplikboyu4,
      :dinlenmisgramaj, :maysayisi, :tupacikenmayli, :hamen;
    end

    if (EgemenFirmaId = 146) then /*Özsar*/
    begin
      select  first 1 BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen from OrmeSiparisDetay
        left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
        where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId into :HamEn, :DinlenmisGramaj;
    end

    if (OrmeTalimattaOtomatikOnayVer = 0) then
      insert into OrmePlanlama(Tarih, FirmaId, OrmeMakinaId, Sira, BoyaSiparisKayitId, OrmeSiparisDetayId, Beyazlik, PlanlananMiktar,
                               BirimFiyat, DovizId, VadeGun, MiktarGun, InsertKullaniciId, KullaniciId, makinaustugramaj, ignesayisi, iplikboyu1, iplikboyu2, iplikboyu3, iplikboyu4,
                               dinlenmisgramaj, maysayisi, tupacikenmayli, hamen)
                        values(:FTarih, :AFirmaId, :AOrmeMakinaId, :FSira, :FBoyaSiparisKayitId, :AOrmeSiparisDetayId, :FBeyazlik, :FPlanlanacakMiktar,
                               :FBirimFiyat, :FDovizId, :FVadeGun, :AMiktarGun, :AKullaniciId, :AKullaniciId, :makinaustugramaj, :ignesayisi, :iplikboyu1, :iplikboyu2, :iplikboyu3, :iplikboyu4,
                               :dinlenmisgramaj, :maysayisi, :tupacikenmayli, :hamen);
    else
      insert into OrmePlanlama(Tarih, FirmaId, OrmeMakinaId, Sira, BoyaSiparisKayitId, OrmeSiparisDetayId, Beyazlik, PlanlananMiktar,
                               BirimFiyat, DovizId, VadeGun, MiktarGun, OnayDurumu, OnayTarihi, OnaylayanKullaniciID, InsertKullaniciId, KullaniciId,
                               makinaustugramaj, ignesayisi, iplikboyu1, iplikboyu2, iplikboyu3, iplikboyu4,
                               dinlenmisgramaj, maysayisi, tupacikenmayli, hamen)
                        values(:FTarih, :AFirmaId, :AOrmeMakinaId, :FSira, :FBoyaSiparisKayitId, :AOrmeSiparisDetayId, :FBeyazlik, :FPlanlanacakMiktar,
                               :FBirimFiyat, :FDovizId, :FVadeGun, :AMiktarGun, 1, Current_TimeStamp(0), :AKullaniciId, :AKullaniciId, :AKullaniciId,
                               :makinaustugramaj, :ignesayisi, :iplikboyu1, :iplikboyu2, :iplikboyu3, :iplikboyu4,
                               :dinlenmisgramaj, :maysayisi, :tupacikenmayli, :hamen);

  end
  else
  begin
    if (AOrmePlanlamaId is not null) then
    begin
      if (AOrmeMakinaId is not null) then
      begin
        if (FSira > -1) then
        begin
          ASira = FSira;
          for select OrmePlanlamaId from OrmePlanlama where OrmePlanlama.OrmeMakinaId = :AOrmeMakinaId and Sira >= :FSira order by Sira into :OPId do
          begin
            ASira = ASira + 1;
            update OrmePlanlama set Sira = :ASira where OrmePlanlamaId = :OPId;
          end
        end
        else
        begin
          select max(Sira) from OrmePlanlama where OrmePlanlama.OrmeMakinaId = :AOrmeMakinaId into :ASira;
          if (ASira is null) then ASira = 0;
          else ASira = ASira + 1;
          FSira = ASira;
        end
      end
      else
      begin
        if (FSira > -1) then
        begin
          ASira = FSira;
          for select OrmePlanlamaId from OrmePlanlama where OrmePlanlama.FirmaId = :AFirmaId and Sira >= :FSira order by Sira into :OPId do
          begin
            ASira = ASira + 1;
            update OrmePlanlama set Sira = :ASira where OrmePlanlamaId = :OPId;
          end
        end
        else
        begin
          select max(Sira) from OrmePlanlama where OrmePlanlama.FirmaId = :AFirmaId into :ASira;
          if (ASira is null) then ASira = 0;
          else ASira = ASira + 1;
          FSira = ASira;
        end
      end
    end

    Update OrmePlanlama Set OrmeMakinaId = :AOrmeMakinaId, FirmaId = :AFirmaId, Sira = :FSira where OrmePlanlama.OrmePlanlamaId = :AOrmePlanlamaId;
  end
end
