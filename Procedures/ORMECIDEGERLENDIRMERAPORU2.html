CREATE PROCEDURE ORMECIDEGERLENDIRMERAPORU2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TOPLAMTALIMATMIKTARI DOUBLE PRECISION,
    TOPLAMGERCEKLESENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    MIKTARPERFORMANSI DOUBLE PRECISION,
    TERMINPERFORMANSI DOUBLE PRECISION)
AS
declare variable FFirmaId       varchar(20);
declare variable FFirmaAdi      varchar(100);
declare variable FLastFirmaId   varchar(20);
declare variable FTalimatMiktari Double precision;
declare variable FGerceklesenMiktar    Double precision;
declare variable FTermin         Double precision;
declare variable SipSayisi              Double precision;
declare variable GecikenSipSayisi       Double precision;
begin
  FLastFirmaID = '';

  For SELECT OrmeciDegerlendirmeRaporu1.FirmaId, OrmeciDegerlendirmeRaporu1.FirmaAdi, OrmeciDegerlendirmeRaporu1.TalimatMiktari,  OrmeciDegerlendirmeRaporu1.GerceklesenMiktar, OrmeciDegerlendirmeRaporu1.Termin FROM OrmeciDegerlendirmeRaporu1(:ABaslangicTarihi, :ABitisTarihi, :AFirmaId)
     order by OrmeciDegerlendirmeRaporu1.FirmaId /* degistirme düzgün calısmaz */
    into :FFirmaId, :FFirmaAdi, :FTalimatMiktari, :FGerceklesenMiktar, :FTermin do
  begin
    if (FLastFirmaId <> FFirmaId) then
    begin
      if (FLastFirmaId <> '') then
      begin
        if (SipSayisi > 0) then
        begin
          TerminPerformansi = (SipSayisi - GecikenSipSayisi) / SipSayisi * 100;
          TerminPerformansi = Yuvarla(TerminPerformansi, 100);
        end

        if (ToplamTalimatMiktari > 0) then
        begin
          MiktarPerformansi = (ToplamTalimatMiktari - KalanMiktar) / ToplamTalimatMiktari * 100;
          MiktarPerformansi = Yuvarla(MiktarPerformansi, 100);

        end
        suspend;
      end

      FLastFirmaId = FFirmaId;
      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      ToplamTalimatMiktari    = 0;
      KalanMiktar             = 0;
      ToplamGerceklesenMiktar = 0;
      MiktarPerformansi    = 0;
      TerminPerformansi    = 0;
      SipSayisi            = 0;
      GecikenSipSayisi     = 0;
    end

    ToplamTalimatMiktari = ToplamTalimatMiktari + FTalimatMiktari;

    if ((FTalimatMiktari - FGerceklesenMiktar)  > 0) then KalanMiktar = KalanMiktar + FTalimatMiktari - FGerceklesenMiktar;
    ToplamGerceklesenMiktar = ToplamGerceklesenMiktar + FGerceklesenMiktar;

    SipSayisi = SipSayisi + 1;
    if (FTermin = -1) then GecikenSipSayisi = GecikenSipSayisi + 1;
  end

  if (FLastFirmaId <> '') then
  begin
    if (SipSayisi > 0) then
    begin
      TerminPerformansi = (SipSayisi - GecikenSipSayisi) / SipSayisi * 100;
      TerminPerformansi = Yuvarla(TerminPerformansi, 100);
    end

    if (ToplamTalimatMiktari > 0) then
    begin
      MiktarPerformansi = (ToplamTalimatMiktari - KalanMiktar) / ToplamTalimatMiktari * 100;
      MiktarPerformansi = Yuvarla(MiktarPerformansi, 100);
    end
    suspend;
  end
end
