CREATE PROCEDURE PARTISINIFHAREKETBILGISI(
    APARTIKAYITID INTEGER)
RETURNS (
    ACIKLAMA VARCHAR(20),
    PARTISINIFID INTEGER,
    BOYAIRSALIYEID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    HAREKETTIPISTR VARCHAR(20),
    IRSALIYETARIHI TIMESTAMP,
    HAMSTOKSARFFISIID INTEGER,
    HAMSTOKSARFFISIDETAYID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BRUTKILO NUMERIC(12,3),
    TARTIBRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    KAPADEDI INTEGER,
    FIREORANI DOUBLE PRECISION,
    BRUTMETRE NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    KULLANICIADI VARCHAR(40),
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255)
AS
declare variable PartiKayitId  integer;
declare variable HareketTipiId integer;
begin
  for Select BoyaIrsaliyeDetay.PartiKayitId, BoyaIrsaliyeDetay.PartiSinifId, BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.FirmaId, Firma.Adi,
         BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi,
         BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.Kilo, BoyaIrsaliyeDetay.BrutMetre, BoyaIrsaliyeDetay.Metre, Kullanici.Adi,
         BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo, BoyaFatura.FaturaTarihi, BoyaIrsaliye.HareketTipiId,
         BoyaIrsaliyeDetay.KapAdedi, Parti.PartiYil, Parti.PartiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Ham.HamId, Ham.Adi from BoyaIrsaliyeDetay
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
    left join Kullanici on Kullanici.KullaniciId = BoyaIrsaliyeDetay.KullaniciId
    left join BoyaFatura On BoyaFatura.BoyaFaturaId = BoyaIrsaliyeDetay.BoyaFaturaId
    left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where BoyaIrsaliyeDetay.PartiKayitId = :APartiKayitId
    into :PartiKayitId, :PartiSinifId, :BoyaIrsaliyeId, :FirmaId, :FirmaAdi, :BoyaIrsaliyeDetayId, :IrsaliyeSeriNo, :IrsaliyeNo,
         :Irsaliyetarihi, :BrutKilo, :NetKilo, :BrutMetre, :NetMetre, :KullaniciAdi,
         :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :HareketTipiId, :KapAdedi, :PartiYil, :PartiId, :BoyaSiparisYil, :BoyaSiparisId, :HamId, :HamAdi  do
  begin
    Aciklama = 'Sevk';

    if (HareketTipiId = 0) then
    begin
      HareketTipiStr = 'Satış';
      select sum(PartiStokTarti.Kilo + PartiStokTarti.Dara) from PartiStokTarti where PartiStokTarti.BoyaIrsaliyeDetayId = :BoyaIrsaliyeDetayId into :TartiBrutKilo;
    end
    else if (HareketTipiId = 1) then
    begin
      HareketTipiStr = 'Alış İade';
      TartiBrutKilo = Null;
    end
    else if (HareketTipiId = 2) then
    begin
      HareketTipiStr = 'Satış İade';
      Aciklama = 'Satış İade';
      select sum(PartiStokTarti.Kilo + PartiStokTarti.Dara) from PartiStokTarti where PartiStokTarti.IadeBoyaIrsaliyeDetayId = :BoyaIrsaliyeDetayId into :TartiBrutKilo;
    end

    if (BrutKilo <> 0) then
      FireOrani = (BrutKilo - NetKilo) * 100 / BrutKilo;
    else
      FireOrani = 0;

    suspend;
  end

  BoyaIrsaliyeId        = Null;
  BoyaIrsaliyeDetayId   = Null;
  IrsaliyeSeriNo        = Null;
  IrsaliyeNo            = Null;
  IrsaliyeTarihi        = Null;
  FirmaId               = Null;
  FirmaAdi              = Null;
  FaturaSeriNo          = Null;
  FaturaNo              = Null;
  FaturaTarihi          = Null;
  FireOrani             = Null;
  TartiBrutKilo         = Null;

  for Select HamStokSarfFisiDetay.PartiSinifId, HamStokSarfFisi.HamStokSarfFisiId,
         HamStokSarfFisiDetay.HamStokSarfFisiDetayId, HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo, HamStokSarfFisi.FisTarihi,
         HamStokSarfFisiDetay.BrutKilo, HamStokSarfFisiDetay.Kilo, HamStokSarfFisiDetay.BrutMetre, HamStokSarfFisiDetay.Metre, Kullanici.Adi,
         HamStokSarfFisiDetay.KapAdedi, Parti.PartiYil, Parti.PartiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Ham.HamId, Ham.Adi from PartiStok
    left join HamStokSarfFisiDetay on HamStokSarfFisiDetay.PartiSinifId = PartiStok.PartiSinifId
    left join Ham on Ham.HamID = HamStokSarfFisiDetay.HamId
    left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiId = HamStokSarfFisiDetay.HamStokSarfFisiId
    left join Kullanici on Kullanici.KullaniciId = HamStokSarfFisiDetay.KullaniciId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where PartiStok.PartiKayitId = :APartiKayitId and HamStokSarfFisiDetay.PartiSinifID is not null
    into :PartiSinifId, :HamStokSarfFisiId, :HamStokSarfFisiDetayId, :FisYil, :FisNo,
         :Irsaliyetarihi, :BrutKilo, :NetKilo, :BrutMetre, :NetMetre, :KullaniciAdi, :KapAdedi, :PartiYil, :PartiId, :BoyaSiparisYil, :BoyaSiparisId, :HamId, :HamAdi do
  begin
    Aciklama = 'Sarf Fişi ile Çıkış';
    HareketTipiStr = 'Sarf Fişi ile Çıkış';
    FisIDStr = '(' || EgemenIntegerToStr(FisYil)  || ') ' || EgemenIntegerToStr(FisNo) || ' ';
    select sum(PartiStokTarti.Kilo + PartiStokTarti.Dara) from PartiStokTarti where PartiStokTarti.HamStokSarfFisiDetayId = :HamStokSarfFisiDetayId into :TartiBrutKilo;
    suspend;
  end
end
