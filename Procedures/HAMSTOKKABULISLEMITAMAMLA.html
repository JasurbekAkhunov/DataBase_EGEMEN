CREATE PROCEDURE HAMSTOKKABULISLEMITAMAMLA(
    AHAMSTOKMASTERID INTEGER)
AS
declare variable HamStokDetayId      integer;
declare variable GelenMiktar         Numeric(12,3);
declare variable GelenKilo           Numeric(12,3);
declare variable GelenMetre          Numeric(12,3);
declare variable GelenAdet           integer;
declare variable GelenKapAdedi       integer;

declare variable KabulMiktar         Numeric(12,3);
declare variable RetMiktar           Numeric(12,3);
declare variable SartliKabulMiktar   Numeric(12,3);
declare variable ZorunluKabulMiktar  Numeric(12,3);

declare variable HsktKabul           integer;
declare variable HsktKismiKabul      integer;
declare variable HsktSartliKabul     integer;
declare variable HsktZorunluKabul     integer;
declare variable HsktRet              integer;

declare variable HamTipiId           integer;
declare variable HamStokKabulTipiId  integer;

declare variable Kilo           Numeric(12,3);
declare variable Metre          Numeric(12,3);
declare variable Adet           integer;
declare variable KapAdedi       integer;

declare variable FMiktar         Numeric(12,3);
declare variable InsertSayisi       integer;
declare variable BoyaSiparisDetayId integer;
declare variable BoyaSiparisKayitId integer;
declare variable TransferYeri    integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable DepartmanOnayindaOtoRezervYap integer;
declare variable SertifikaNo Varchar(30);
declare variable SertifikaKilo           Numeric(12,3);
declare variable SertifikaMetre          Numeric(12,3);
declare variable SertifikaMiktar         Numeric(12,3);
begin
  Select DokumaKumasiKilodanTakipEt, DepartmanOnayindaOtoRezervYap from Sistem into :DokumaKumasiKilodanTakipEt, :DepartmanOnayindaOtoRezervYap;

  HsktKabul = 1;
  HsktKismiKabul = 2;
  HsktSartliKabul = 3;
  HsktZorunluKabul = 4;
  HsktRet = 5;

  InsertSayisi = 0;
  Select HamStokMaster.Miktar, HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Adet,  Ham.HamTipiId,
         HamStokMaster.BoyaSiparisKayitId, HamStokMaster.BoyaSiparisDetayId, HamStokIrsaliye.TransferYeri, HamStokMaster.SertifikaNo from HamStokMaster
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
    left join Ham on Ham.HamId = HamStokMaster.HamId
    where HamStokMaster.HamStokMasterId = :AHamStokMasterId
    into :GelenMiktar, :GelenKapAdedi, :GelenKilo, :GelenMetre, :GelenAdet, :HamTipiId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :TransferYeri, :SertifikaNo;

  Select KabulMiktar, RetMiktar, SartliKabulMiktar, ZorunluKabulMiktar from HamStokMasterKabul
    where HamStokMasterKabul.HamStokMasterId = :AHamStokMasterId and HamStokMasterKabul.SonKarar = 1
    into :KabulMiktar, :RetMiktar, :SartliKabulMiktar, :ZorunluKabulMiktar;

  if (KabulMiktar + RetMiktar + SartliKabulMiktar + ZorunluKabulMiktar <> GelenMiktar) then exception KararMiktarMiktardanFarkli;

  if (KabulMiktar > 0) then
  begin
    if (RetMiktar > 0) then HamStokKabulTipiId = HsktKismiKabul;
    else HamStokKabulTipiId = HsktKabul;

    FMiktar = KabulMiktar;

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
    begin
      Kilo = FMiktar;
      if (GelenKilo > 0) then
      begin
        Metre = GelenMetre * Kilo / GelenKilo;
        KapAdedi = GelenKapAdedi * Kilo / GelenKilo;
        Adet = GelenAdet * Kilo / GelenKilo;
      end
      else
      begin
        Metre = 0;
        KapAdedi = 0;
        Adet = 0;
      end
    end
    else
    begin
      Metre = FMiktar;
      Kilo = GelenKilo * Metre / GelenMetre;
      KapAdedi = GelenKapAdedi * Metre / GelenMetre;
      Adet = GelenAdet * Metre / GelenMetre;
    end
    InsertSayisi = InsertSayisi + 1;

    if (Coalesce(SertifikaNo,'')<>'') then
    begin
      SertifikaKilo = Kilo;
      SertifikaMetre = Metre;
      SertifikaMiktar = FMiktar;
    end
    else
    begin
      SertifikaKilo = 0;
      SertifikaMetre = 0;
      SertifikaMiktar = 0;
    end

    Insert Into HamStokDetay(HamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, HamStokKabulTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
                              KapAdedi, Kilo, Metre, Adet, FireOrani, TupAcikEnMayli,
                              En, Grm2, LycraUreticiId, Aciklama, Miktar, BirimFiyat,
                              FaturaNo, FaturaTarihi, GirisCikis, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi,
                              SertifikaGirisKilo, SertifikaGirisMetre, SertifikaCikisKilo, SertifikaCikisMetre, SertifikaGirisMiktar, SertifikaCikisMiktar)
       Select :AHamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, :HamStokKabulTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
              :KapAdedi, :Kilo, :Metre, :Adet, FireOrani, TupAcikEnMayli,
              En, Grm2, LycraUreticiId, Aciklama, Miktar, BirimFiyat,
              FaturaNo, FaturaTarihi, GirisCikis, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi,
              :SertifikaKilo, :SertifikaMetre, 0, 0, :SertifikaMiktar, 0 from HamStokMaster where HamStokMaster.HamStokMasterId = :AHamStokMasterId;
  end

  if (RetMiktar > 0) then
  begin
    HamStokKabulTipiId = HsktRet;

    FMiktar = RetMiktar;

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
    begin
      Kilo = FMiktar;

      if (GelenKilo <> 0) then
      begin
        Metre = GelenMetre * Kilo / GelenKilo;
        KapAdedi = GelenKapAdedi * Kilo / GelenKilo;
        Adet = GelenAdet * Kilo / GelenKilo;
      end
      else
      begin
        Metre = 0;
        KapAdedi = 0;
        Adet = 0;
      end
    end
    else
    begin
      Metre = FMiktar;
      Kilo = GelenKilo * Metre / GelenMetre;
      KapAdedi = GelenKapAdedi * Metre / GelenMetre;
      Adet = GelenAdet * Metre / GelenMetre;
    end

    if (Coalesce(SertifikaNo,'')<>'') then
    begin
      SertifikaKilo = Kilo;
      SertifikaMetre = Metre;
      SertifikaMiktar = FMiktar;
    end
    else
    begin
      SertifikaKilo = 0;
      SertifikaMetre = 0;
      SertifikaMiktar = 0;
    end

    InsertSayisi = InsertSayisi + 1;
    Insert Into HamStokDetay(HamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, HamStokKabulTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
                              KapAdedi, Kilo, Metre, Adet, FireOrani, TupAcikEnMayli,
                              En, Grm2, LycraUreticiId, Aciklama, Miktar, BirimFiyat,
                              FaturaNo, FaturaTarihi, GirisCikis, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi,
                              SertifikaGirisKilo, SertifikaGirisMetre, SertifikaCikisKilo, SertifikaCikisMetre, SertifikaGirisMiktar, SertifikaCikisMiktar)
       Select :AHamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, :HamStokKabulTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
              :KapAdedi, :Kilo, :Metre, :Adet, FireOrani, TupAcikEnMayli,
              En, Grm2, LycraUreticiId, Aciklama, Miktar, BirimFiyat,
              FaturaNo, FaturaTarihi, GirisCikis, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi,
              :SertifikaKilo, :SertifikaMetre, 0, 0, :SertifikaMiktar,0  from HamStokMaster where HamStokMaster.HamStokMasterId = :AHamStokMasterId;
  end

  if (SartliKabulMiktar > 0) then
  begin
    HamStokKabulTipiId = HsktSartliKabul;

    FMiktar = SartliKabulMiktar;

    if (HamTipiId <> 2) then
    begin
      Kilo = FMiktar;
      if (GelenKilo <> 0) then
      begin
        Metre = GelenMetre * Kilo / GelenKilo;
        KapAdedi = GelenKapAdedi * Kilo / GelenKilo;
        Adet = GelenAdet * Kilo / GelenKilo;
      end
      else
      begin
        Metre = 0;
        KapAdedi = 0;
        Adet = 0;
      end
    end
    else
    begin
      Metre = FMiktar;
      Kilo = GelenKilo * Metre / GelenMetre;
      KapAdedi = GelenKapAdedi * Metre / GelenMetre;
      Adet = GelenAdet * Metre / GelenMetre;
    end

    if (Coalesce(SertifikaNo,'')<>'') then
    begin
      SertifikaKilo = Kilo;
      SertifikaMetre = Metre;
      SertifikaMiktar = FMiktar;
    end
    else
    begin
      SertifikaKilo = 0;
      SertifikaMetre = 0;
      SertifikaMiktar = 0;
    end

    InsertSayisi = InsertSayisi + 1;
    Insert Into HamStokDetay(HamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, HamStokKabulTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
                              KapAdedi, Kilo, Metre, Adet, FireOrani, TupAcikEnMayli,
                              En, Grm2, LycraUreticiId, Aciklama, Miktar, BirimFiyat,
                              FaturaNo, FaturaTarihi, GirisCikis, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi,
                              SertifikaGirisKilo, SertifikaGirisMetre, SertifikaCikisKilo, SertifikaCikisMetre, SertifikaGirisMiktar, SertifikaCikisMiktar)
       Select :AHamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, :HamStokKabulTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
              :KapAdedi, :Kilo, :Metre, :Adet, FireOrani, TupAcikEnMayli,
              En, Grm2, LycraUreticiId, Aciklama, Miktar, BirimFiyat,
              FaturaNo, FaturaTarihi, GirisCikis, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi,
              :SertifikaKilo, :SertifikaMetre, 0, 0, :SertifikaMiktar, 0 from HamStokMaster where HamStokMaster.HamStokMasterId = :AHamStokMasterId;
  end

  if (ZorunluKabulMiktar > 0) then
  begin
    HamStokKabulTipiId = HsktZorunluKabul;

    FMiktar = ZorunluKabulMiktar;

    if (HamTipiId <> 2) then
    begin
      Kilo = FMiktar;
      if (GelenKilo <> 0) then
      begin
        Metre = GelenMetre * Kilo / GelenKilo;
        KapAdedi = GelenKapAdedi * Kilo / GelenKilo;
        Adet = GelenAdet * Kilo / GelenKilo;
      end
      else
      begin
        Metre = 0;
        KapAdedi = 0;
        Adet = 0;
      end
    end
    else
    begin
      Metre = FMiktar;
      Kilo = GelenKilo * Metre / GelenMetre;
      KapAdedi = GelenKapAdedi * Metre / GelenMetre;
      Adet = GelenAdet * Metre / GelenMetre;
    end

    if (Coalesce(SertifikaNo,'')<>'') then
    begin
      SertifikaKilo = Kilo;
      SertifikaMetre = Metre;
      SertifikaMiktar = FMiktar;
    end
    else
    begin
      SertifikaKilo = 0;
      SertifikaMetre = 0;
      SertifikaMiktar = 0;
    end

    InsertSayisi = InsertSayisi + 1;
    Insert Into HamStokDetay(HamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, HamStokKabulTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
                              KapAdedi, Kilo, Metre, Adet, FireOrani, TupAcikEnMayli,
                              En, Grm2, LycraUreticiId, Aciklama, Miktar, BirimFiyat,
                              FaturaNo, FaturaTarihi, GirisCikis, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi,
                              SertifikaGirisKilo, SertifikaGirisMetre, SertifikaCikisKilo, SertifikaCikisMetre, SertifikaGirisMiktar, SertifikaCikisMiktar)
       Select :AHamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, :HamStokKabulTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
              :KapAdedi, :Kilo, :Metre, :Adet, FireOrani, TupAcikEnMayli,
              En, Grm2, LycraUreticiId, Aciklama, Miktar, BirimFiyat,
              FaturaNo, FaturaTarihi, GirisCikis, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi,
              :SertifikaKilo, :SertifikaMetre, 0, 0, :SertifikaMiktar, 0 from HamStokMaster where HamStokMaster.HamStokMasterId = :AHamStokMasterId;
  end

  if (InsertSayisi = 1) then
  begin
    Select HamStokDetayId from HamStokDetay where HamStokDetay.HamStokMasterId = :AHamStokMasterId into :HamStokDetayId;

    insert into RafDetay(GeciciKabulRafId, HamStokIrsaliyeId, HamStokDetayId, RafId, KapAdedi, Kilo, Metre, Adet, InsertKullaniciId, KullaniciId)
      Select GeciciKabulRaf.GeciciKabulRafId, GeciciKabulRaf.HamStokIrsaliyeId, :HamStokDetayId, GeciciKabulRaf.RafId, GeciciKabulRaf.KapAdedi, GeciciKabulRaf.Kilo, GeciciKabulRaf.Metre, GeciciKabulRaf.Adet, GeciciKabulRaf.InsertKullaniciId, GeciciKabulRaf.InsertKullaniciId from GeciciKabulRaf
        where GeciciKabulRaf.HamStokMasterID = :AHamStokMasterId;
  end

  if (DepartmanOnayindaOtoRezervYap = 1 and HamTipiId <> 0 and BoyaSiparisDetayID is not null and BoyaSiparisKayitID is not null) then
  if (KabulMiktar > 0) then
  begin
    Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, HamStokDetayId, Kilo, Metre, Adet, KapAdedi)
                   Select :BoyaSiparisKayitId, :BoyaSiparisDetayId, HamStokDetay.HamStokDetayID, HamStokDetay.Kilo, HamStokDetay.Metre, HamStokDetay.Adet, HamStokDetay.KapAdedi from HamStokDetay
                      where HamStokDetay.HamStokMasterId = :AHamStokMasterId and
                            (HamStokDetay.HamStokKabulTipiId = :HsktKismiKabul or HamStokDetay.HamStokKabulTipiId = :HsktKabul);
  end
end
