CREATE PROCEDURE PARTISTOKREZERVET(
    ABOYAREZERVFISIID INTEGER,
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    AFIRMAID FIRMAKODUSIZE,
    ABOYASIPARISDETAYID INTEGER,
    AHAMID INTEGER,
    AKULLANICIID INTEGER,
    AGIRILENMIKTAR DOUBLE PRECISION,
    AGIRILENBRUTMIKTAR DOUBLE PRECISION,
    ARAFID RAFKODUSIZE)
RETURNS (
    EKLENENBOYAREZERVFISIID INTEGER)
AS
declare variable FFisNo integer;
declare variable FBoyaRezervFisiDetayID integer;
begin
  if (ABoyaRezervFisiId = -1) then /*Fis otomatik olusacak*/
  begin
    ABoyaRezervFisiId = gen_id(boyarezervfisi_id_gen, 1);
    Select Coalesce(Max(FisNo), 0)+1 FisNo from BoyaRezervFisi into :FFisNo;

    insert into BoyaRezervFisi (BoyaRezervFisiId, FisTarihi, FisNo, FirmaId, Notu, InsertKullaniciId, InsertTarihi)
    values (:ABoyaRezervFisiId, current_date, :FFisNo, :AFirmaId, 'Oto:', :AKullaniciId, current_timestamp(0));
  end
  else
  begin
    Select BoyaRezervFisiDetayID from BoyaRezervFisiDetay
      where BoyaRezervFisiDetay.BoyaRezervFisiId = :ABoyaRezervFisiId and BoyaRezervFisiDetay.PartiSinifId = :APartiSinifId and
      BoyaRezervFisiDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId and
      BoyaRezervFisiDetay.RafId = :ARafId
      into :FBoyaRezervFisiDetayID;
  end

  /*Rezerv detay kaydet*/
  if (FBoyaRezervFisiDetayID is Not Null) then
  begin
    Update BoyaRezervFisiDetay Set EdilecekKilo = EdilecekKilo + :AGirilenMiktar,
                                   EdilecekMiktar = EdilecekMiktar + :AGirilenBrutMiktar,
                                   EdilecekBrutKilo = EdilecekBrutKilo + :AGirilenMiktar,
                                   EdilecekBrutMiktar = EdilecekBrutMiktar + :AGirilenBrutMiktar, KullaniciID = :AKullaniciId
                                   where BoyaRezervFisiDetay.BoyaRezervFisiDetayID = :FBoyaRezervFisiDetayID;
  end
  else
  begin
    insert into BoyaRezervFisiDetay (BoyaRezervFisiDetayId, BoyaRezervFisiId, BoyaSiparisDetayId, FirmaId, PartiKayitId,
                                     PartiSinifId, FisTarihi, HamId, EdilecekKilo, EdilecekMiktar, EdilecekBrutKilo, EdilecekBrutMiktar,
                                     InsertKullaniciId, InsertTarihi, Tamamlandi, RafId)
    values (gen_ID(BoyaRezervFisiDetay_Id_gen, 1), :ABoyaRezervFisiId, :ABoyaSiparisDetayId, :AFirmaId, :APartiKayitId,
                                     :APartiSinifId, (select FisTarihi from BoyaRezervFisi where BoyaRezervFisi.BoyaRezervFisiId = :ABoyaRezervFisiId), :AHamId,
                                     :AGirilenMiktar, :AGirilenBrutMiktar, :AGirilenMiktar, :AGirilenBrutMiktar, :AKullaniciId, current_TimeStamp(0), 0, :ARafId);

  end
  EklenenBoyaRezervFisiId = ABoyaRezervFisiId;
  Suspend;
end
