    where Boyafatura.BoyaFaturaId = :ABoyaFaturaId
    into :FHareketTipiId, :FFaturaDovizId, :FNetFiyatYuvarlamaCarpani;

  select EgemenFirmaId, SistemDovizId from Sistem where SistemId = 1 into :EgemenFirmaId, :SistemDovizId;

  if (FFaturaDovizId <> SistemDovizId) then
  begin

  end
  else FDovizKuru = 1;

             BoyaSiparisDetay.HamStokBirimId, HamStokBirim.BirimId, Ham.HamTipiId,

   left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId

   left join TeslimSekli on TeslimSekli.TeslimSekliId = BoyaSiparis.TeslimSekliId
   left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
   left join Proses on Proses.ProsesId = PartiStok.ProsesId
   left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
   left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
   left join IslemGrubu SipIslemGrubu on SipIslemGrubu.IslemGrubuId = BoyaSiparisDetay.IslemGrubuId
   left join SevkFirma on SevkFirma.SevkFirmaId = BoyaIrsaliye.SevkFirmaId
   where BoyaFaturaDetay.BoyaFaturaId = :ABoyaFaturaID order by BoyaFaturaDetay.BoyaFaturaDetayId
   into :BoyaFaturaDetayId, :BoyaFaturaId, :PartiKayitId, :PartiSinifId, :HamAdiToplami, :HamAdi2, :HamAdi3, :HamAdi4, :PartiYil, :PartiId, :Fason,
        :ReceteId, :ReceteAdi, :SiparisProsesId, :SiparisProsesAdi, :SiparisProsesAdi2, :Aciklama, :PaketAciklama1, :PaketAciklama2,
        :SiparisRenkAdi, :MusteriRenkKoduVeAdi, :DesenId, :DesenAdi, :EskiDesenNo, :VaryantAdi,
        :KaliteNo, :Miktar, :FBirimFiyat, :DovizKuru1, :DovizKuru2, :VadeGun, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
        :SiparisBirimFiyat, :KdvOrani, :SiparisDovizId, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :BoyaIrsaliyeId, :BoyaIrsaliyeDetayId,
        :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
        :GirisIrsaliyeNoStr, :GirisIrsaliyeTarihiStr, :GirisFirmaId, :GirisFirmaAdi, :HamStokBirimId, :SipBirimID, :HamTipiId,
        :IslemTipiId, :IslemTipiAdi, :IslemGrubuId, :IslemGrubuAdi, :PartiPaketId, :MusteriSiparisNo, :ReferansSiparisNo,
        :SiparisLabReceteKodu, :SiparisLabReceteAdi, :FAtisNoStr, :BrutKilo, :BrutMetre, :NetKilo, :NetMetre, :KapAdedi,
        :YapilanIslemler, :OrmeSiparisTipiId, :OrmeSiparisTipiAdi, :OrmeSiparisSekliId, :OdemeSekliId, :OdemeSekliAdi,
        :TeslimSekliId, :TeslimSekliAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SipIlaveIslemler, :SipOzelApreler, :SipOzelTestler,
        :TamirBoyaSiparisDetayId, :GecikmeNotu, :KaliteId, :KaliteAdi, :EnIstenen, :GramajIstenen, :RafDetayId,
        :PartiProsesAdi, :BelgeHamAdi, :BaskiTipiId, :BaskiTipiAdi, :MelanjKodu,
        :EnVerilen, :GramajVerilen, :DonmeVerilen, :LabRecRenkDerinligiId, :LabRecRenkDerinligiAdi, :LabReceteCalismaBirimFiyat, :LabReceteCalismaDovizID,
        :MiktarBirim, :FatDetayBirimID, :SiparisIslemGrubuId, :SiparisIslemGrubuAdi, :TasarimNo, :MusteriKaliteNo,
        :KKDeger1, :KKDeger2, :KKDeger3, :KKDeger4, :KKDeger5, :KKDeger6, :KKDeger7,
        :FiyatFarki, :BoyaMaliyeti, :FaturaMiktarTipi, :SevkFirmaAdi do
   begin
     select TransferKodu from BoyaIrsaliyeFaturaTransferKodu(:BoyaIrsaliyeDetayId, Null, :EgemenFirmaId) into :TransferKodu;

     IlaveProses = '';
     For Select IlaveProsesSebebi.Adi, Asama.Adi from PartiAsamaStok
       Left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
       left join IlaveProsesSebebi on IlaveProsesSebebi.IlaveProsesSebebiId = PartiAsamalari.IlaveProsesSebebiID
       left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
       where PartiAsamalari.IlaveProses = 1 and PartiAsamaStok.PartiSinifId = :PartiSinifId
       into :FIlaveProsesSebebiAdi, :FAsamaAdi do
       begin
         if (Char_Length(IlaveProses) + Char_Length(FIlaveProsesSebebiAdi) + Char_Length(FAsamaAdi) < 1020) then
         begin
          if (IlaveProses <> '' ) then IlaveProses = IlaveProses || cast(', ' as varchar(2) character set UTF8);
          IlaveProses = IlaveProses || FAsamaAdi || '(' || FIlaveProsesSebebiAdi || ')';
         end
       end

     BoyaMaliyeti = BoyaMaliyeti * FDovizKuru;

     BirimFiyat = FBirimFiyat;
     Tutar = Miktar * BirimFiyat;
     Tutar = Yuvarla(Tutar, FNetFiyatYuvarlamaCarpani);

     SiparisLabReceteKodu = SiparisLabReceteKodu || '.' || FAtisNoStr;

     IlkIslemGrubuId = IslemGrubuId;

     if (TamirBoyaSiparisDetayId is not null) then
       Select IslemGrubuId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = :TamirBoyaSiparisDetayId into :IlkIslemGrubuId;

     if (PartiSinifId is null) then
     begin

        IlaveIslemler = '';
        Fason = 0;
        if (FHareketTipiId = 0) then PartiIdStr = 'Ham Stok Sat';
        else if (FHareketTipiId = 1) then PartiIdStr = 'Ham Stok Al ade';
        else exception Kayithatasi;
        BoyaSiparisIdStr = '';

        if (HamTipiId <> 2) then
        begin
          HamStokBirimId = 0;
          BirimId = 'Kg';
        end
        else
        begin
          HamStokBirimId = 1;
          BirimId = 'Mt';
        end

       Select HamStokMaster.Renk from RafDetay
          left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
          where RafDetay.RafDetayID = :RafDetayID /*and HamStokMaster.OrmeSiparisDetayID is not null*/
        into :HamGirisRenk;
     end
     else
     begin

       PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
       BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
       IlaveIslemler = '';

       for Select IlaveIslemler.Adi from PartiAsamaStok
         left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
         left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
         left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
         where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1 and PartiAsamalari.SerbestAsama = 0 and IlaveIslemler.Adi is not null
         into :FIlaveIslemlerAdi do
       begin
         if (Char_Length(IlaveIslemler) + Char_Length(FIlaveIslemlerAdi) < 1020) then
         begin
          if (IlaveIslemler <> '' ) then IlaveIslemler = IlaveIslemler || cast(', ' as varchar(2) character set UTF8);
          IlaveIslemler = IlaveIslemler || FIlaveIslemlerAdi;
         end
       end

        GirisIrsaliyeNoStr      = '';
        GirisIrsaliyeTarihiStr  = '';
        GirisFirmaID = '';
        GirisFirmaAdi = '';

        FGirisIrsaliyeNoStr = '';
        FGirisIrsaliyeTarihiStr = '';

        for Select HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayId
          left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = RafDetay.HamStokIrsaliyeId
          where PartiStokDetay.PartiSinifID = :PartiSinifId
          group by HamStokIrsaliye.HamStokIrsaliyeID, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi
          into :FIrsaliyeNo, :FIrsaliyeTarihi do
        begin
          Str = '@' || Format_Float('', FIrsaliyeNo) || '@';

          if (PosBul(FGirisIrsaliyeNoStr, Str) = 0) then
          if (Char_Length(GirisIrsaliyeNoStr) + Char_Length(Format_Float('', FIrsaliyeNo)) < 48) then
          begin
            if (GirisIrsaliyeNoStr <> '' ) then
            begin
              GirisIrsaliyeNoStr = left((GirisIrsaliyeNoStr || cast(', ' as varchar(2) character set UTF8)), 50);
              FGirisIrsaliyeNoStr = left((FGirisIrsaliyeNoStr || cast(', ' as varchar(2) character set UTF8)), 100);
            end
            GirisIrsaliyeNoStr = GirisIrsaliyeNoStr || Format_Float('',FIrsaliyeNo);
            FGirisIrsaliyeNoStr = FGirisIrsaliyeNoStr || Str;

            Str = '@' || TimeStampToDateStr(FIrsaliyeTarihi) || '@';
            if (Char_Length(GirisIrsaliyeTarihiStr) + Char_Length(TimeStampToDateStr(FIrsaliyeTarihi)) < 48) then
            begin
              if (GirisIrsaliyeTarihiStr <> '' ) then
              begin
                GirisIrsaliyeTarihiStr = GirisIrsaliyeTarihiStr || cast(', ' as varchar(2) character set UTF8);
                FGirisIrsaliyeTarihiStr = FGirisIrsaliyeTarihiStr || cast(', ' as varchar(2) character set UTF8);
              end
              GirisIrsaliyeTarihiStr = GirisIrsaliyeTarihiStr || TimeStampToDateStr(FIrsaliyeTarihi);
              FGirisIrsaliyeTarihiStr = FGirisIrsaliyeTarihiStr || Str;
            end
          end
        end

        for Select HamStokIrsaliye.FirmaId, Firma.Adi from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayId
          left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = RafDetay.HamStokIrsaliyeId
          left join Firma on Firma.FirmaID = HamStokIrsaliye.FirmaId
          where PartiStokDetay.PartiSinifID = :PartiSinifId
          group by HamStokIrsaliye.FirmaId, Firma.Adi
          into :FFirmaId, :FFirmaAdi do
        begin
          if (Char_Length(GirisFirmaId) + Char_Length(FFirmaId) < 90) then
          begin
            if (GirisFirmaId <> '' ) then GirisFirmaId = GirisFirmaId || cast(', ' as varchar(2) character set UTF8);

            GirisFirmaId = GirisFirmaId || FFirmaId;
          end

          if (Char_Length(GirisFirmaAdi) + Char_Length(FFirmaAdi) < 240) then
          begin
            if (GirisFirmaAdi <> '' ) then GirisFirmaAdi = GirisFirmaAdi || cast(', ' as varchar(2) character set UTF8);

            GirisFirmaAdi = GirisFirmaAdi || FFirmaAdi;
          end
        end
     end

     if (SiparisDovizId is not null and FFaturaDovizId <> SiparisDovizId) then DovizKuruStr = Format_Float('', DovizKuru1) || ' ' || SiparisDovizId || '-' || Format_Float('', DovizKuru2) || ' ' || FFaturaDovizId;
     else DovizKuruStr = '';

      PartiTamirUAsamalar = '';
      FAsamaAdi = '';
      for select PartiTamir.PartiTamirId from PartiTamirSinif
        left join PartiTamir on PartiTamir.PartiTamirID = PartiTamirSinif.PartiTamirID
        left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
        where PartiTamirSinif.PartiSinifId = :PartiSinifId and TamirCozumSekli.Ucretli = 1
        into :FPartiTamirId do
      begin
        for select Asama.Adi from PartiAsamalari
          left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
          left join Asama on Asama.AsamaId = coalesce(PartiAsamalari.AsamaId, Recete.UstAsamaID)
          where PartiAsamalari.PartiTamirId = :FPartiTamirId and Asama.ParasalDegeriVar = 1
          into :FAsamaAdi do
        begin
          if (Char_Length(PartiTamirUAsamalar) < 210) then
          begin
            if (PartiTamirUAsamalar <> '') then PartiTamirUAsamalar = PartiTamirUAsamalar || cast(', ' as varchar(2) character set UTF8);
            PartiTamirUAsamalar = PartiTamirUAsamalar || FAsamaAdi;
          end
        end
      end

      Select Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara) from PartiStokTarti
            where PartiStokTarti.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID
             into :TartiBrutKilo;

     if (FatDetayBirimID is not Null) then BirimId = FatDetayBirimID;

     suspend;
   end
end
