CREATE PROCEDURE KTEBOYALIKUMASSIPARISLERIDETAYI(
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    HAMTIPIID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    SIPARISKILO NUMERIC(12,3),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    MAXIMUMFIREORANI DOUBLE PRECISION,
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    KAPALI INTEGER,
    KAPALISTR VARCHAR(6))
AS
begin
  For Select KTEBoyaliKumasSiparisleri.BoyaSiparisKayitId, KTEBoyaliKumasSiparisleri.OrmeSiparisDetayId, BoyaSiparisDetay.BoyaSiparisDetayId,
             BoyaSiparisDetay.HamId, Ham.Adi, Ham.HamTipiId, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.LabReceteKodu,
             LabRecete.Adi, BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi,
             BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi, BoyaSiparisDetay.MelanjKodu,
             BoyaSiparisDetay.Kapali, BoyaSiparisDetay.MaximumFireOrani, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi from KTEBoyaliKumasSiparisleri(:AFirmaId, :AHamId)
  left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = KTEBoyaliKumasSiparisleri.OrmeSiparisDetayId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
  where BoyaSiparisDetay.BoyaSiparisDetayid is not null
  into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :HamId, :HamAdi, :HamTipiId, :RenkAdi, :SiparisKilo, :LabReceteKodu,
       :LabReceteAdi, :IslemTipiId, :IslemTipiAdi, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi, :MelanjKodu,
       :Kapali, :MaximumFireOrani, :TerminTarihi, :RevizeTerminTarihi do
  begin
    KapaliStr = '';
    if (Kapali = 1) then KapaliStr = 'Kapalı';
    else if (Kapali = 2) then KapaliStr = 'İptal';

    suspend;
  end
end
