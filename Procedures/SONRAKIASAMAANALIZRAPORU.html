CREATE PROCEDURE SONRAKIASAMAANALIZRAPORU(
    AFIRMAID VARCHAR(20),
    AISLEMGRUBU INTEGER,
    AISLEMTIPI INTEGER,
    AACILIYETDERECESI INTEGER,
    AASAMAID INTEGER,
    ASADECESIRADAKIASAMALAR INTEGER,
    APARTITIPIID INTEGER,
    AFASON INTEGER,
    ATEMSILCIID INTEGER,
    ABOYAASAMASIPLANLANMISLAR INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ARECETEYAZILDI EGEBOOLEAN,
    AIMALATTIPIID INTEGER)
RETURNS (
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    BOYAHANEBOLUMID INTEGER,
    BOYAHANEBOLUMADI VARCHAR(30),
    TOPLAMKILO DOUBLE PRECISION,
    TOPLAMMETRE DOUBLE PRECISION,
    TOPLAMSARJKILO DOUBLE PRECISION,
    TOPLAMSARJMETRE DOUBLE PRECISION,
    GURAPORUSIRANO INTEGER,
    ISLETMEONCELIKNO INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    PARTISIRANO INTEGER)
AS
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
declare variable hstLycra        integer;
declare variable BoyaAsamasiPartiAsamalariId integer;
declare variable PlanlamaSepetNo       integer;
declare variable FKilo  Double Precision;
declare variable FMetre Double Precision;
declare variable FAsamaID             integer;
declare variable FAsamaAdi            varchar(40);
declare variable FBoyahaneBolumID     integer;
declare variable FBoyahaneBolumAdi    varchar(30);
declare Variable FGuRaporuSiraNo      Integer;
Begin
  AsamaID = -1;
  ToplamKilo = 0;
  ToplamMetre = 0;
  ToplamSarjKilo = 0;
  ToplamSarjMetre = 0;

  for select SonrakiAsamaAnaliziDetayi.AsamaId, Asama.Adi, Asama.BoyahaneBolumId, BoyahaneBolum.Adi, SonrakiAsamaAnaliziDetayi.PlanlamaSepetNo, Asama.GuRaporuSiraNo, Sum(Kilo),  Sum(Metre) from
  SonrakiAsamaAnaliziDetayi(:AFirmaId, :AIslemGrubu, :AIslemTipi, :AAciliyetDerecesi, :AAsamaId, :ASadeceSiradakiAsamalar, :APartiTipiId, :AFason, :ATemsilciID, :ABoyaAsamasiPlanlanmislar, :ABoyaSiparisYil, :ABoyaSiparisId, :AReceteYazildi, :AImalatTipiId)
      left join Asama on Asama.AsamaId = SonrakiAsamaAnaliziDetayi.AsamaId
      left join BoyahaneBolum on BoyahaneBolum.BoyahaneBolumId = Asama.BoyahaneBolumId
      Group By SonrakiAsamaAnaliziDetayi.AsamaId, Asama.Adi, Asama.BoyahaneBolumId, BoyahaneBolum.Adi, SonrakiAsamaAnaliziDetayi.PlanlamaSepetNo , asama.guraporusirano, SonrakiAsamaAnaliziDetayi.IsletmeOncelikNo, SonrakiAsamaAnaliziDetayi.CikisTarihi,
      SonrakiAsamaAnaliziDetayi.SiparisProsesAdi, SonrakiAsamaAnaliziDetayi.PartiSiraNo
      Order By Asama.BoyahaneBolumId Desc, SonrakiAsamaAnaliziDetayi.AsamaId
      into :FAsamaID, :FAsamaAdi, :FBoyahaneBolumID, :FBoyahaneBolumAdi, :PlanlamaSepetNo, :FGuRaporuSiraNo, :FKilo, :FMetre do
  begin
    if (AsamaID <> FAsamaID and AsamaId <> -1) then
    begin
      suspend;
      ToplamKilo = 0;
      ToplamMetre = 0;
      ToplamSarjKilo = 0;
      ToplamSarjMetre = 0;
    end

    AsamaID = FAsamaID;
    AsamaAdi = FAsamaAdi;
    BoyahaneBolumID = FBoyahaneBolumID;
    BoyahaneBolumAdi = FBoyahaneBolumAdi;
    GuRaporuSiraNo = FGuRaporuSiraNo;

    if (PlanlamaSepetNo = 1 or PlanlamaSepetNo = 2 or PlanlamaSepetNo = 3) then
    begin
      ToplamSarjKilo = ToplamSarjKilo + FKilo;
      ToplamSarjMetre = ToplamSarjMetre + FMetre;
    end

    ToplamKilo = ToplamKilo + FKilo;
    ToplamMetre = ToplamMetre + FMetre;
  end
  if (AsamaId <> -1) then
  begin
    suspend;
  end
end
