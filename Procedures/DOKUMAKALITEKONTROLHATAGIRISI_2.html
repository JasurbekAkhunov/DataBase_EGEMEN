CREATE PROCEDURE DOKUMAKALITEKONTROLHATAGIRISI_2(
    ADOKUMAKALITEKONTROLHATAMETREID INTEGER,
    ADOKUMAKALITEKONTROLID INTEGER,
    ADOKUMAHATATIPIID INTEGER,
    AMETRE DOUBLE PRECISION,
    ABITIS INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    DOKUMAKALITEKONTROLHATAMETREID INTEGER)
AS
begin
  if (ABitis = 0) then
  begin
    if (ADokumaKaliteKontrolHataMetreId is not null) then Update DokumaKaliteKontrolHataMetre Set BaslangicMetre = :AMetre where DokumaKaliteKontrolHataMetre.DokumaKaliteKontrolHataMetreID = :ADokumaKaliteKontrolHataMetreID;
    else
    begin
      DokumaKaliteKontrolHataMetreID = Gen_Id(DKaliteKontrolHataMetre_Id_Gen,1);
      insert into DokumaKaliteKontrolHataMetre(DokumaKaliteKontrolHataMetreID, DokumaKaliteKontrolid, DokumaHataTipiId, BaslangicMetre, InsertKullaniciId, KullaniciId) values(:DokumaKaliteKontrolHataMetreID, :ADokumaKaliteKontrolId, :ADokumaHataTipiId, :AMetre, :AKullaniciId, :AKullaniciId);
    end
  end
  else Update DokumaKaliteKontrolHataMetre Set BitisMetre = :AMetre where DokumaKaliteKontrolHataMetre.DokumaKaliteKontrolHataMetreID = :ADokumaKaliteKontrolHataMetreID;
  suspend;
end
