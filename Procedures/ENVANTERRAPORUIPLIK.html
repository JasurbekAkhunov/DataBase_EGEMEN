CREATE PROCEDURE ENVANTERRAPORUIPLIK(
    AIPLIKDEPOID VARCHAR(10),
    ABASLANGICTARIH TIMESTAMP,
    ABITISTARIH TIMESTAMP,
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    AKALINLIKID INTEGER,
    AELYAFSINIFID INTEGER,
    ACIRCIRID INTEGER,
    AMENSEIID INTEGER,
    AIPLIKCINSIID INTEGER,
    ALOT VARCHAR(30),
    AGIRISLERFIRMADETAYLI INTEGER,
    ARENKID INTEGER)
RETURNS (
    SIRANO INTEGER,
    IPLIKDEPOID VARCHAR(10),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    LOT VARCHAR(30),
    IPLIKLOTGRUBUID INTEGER,
    IPLIKLOTGRUBUADI ADI20,
    ELYAFDEGERLENDIRME INTEGER,
    SARIMSEKLIID INTEGER,
    SARIMSEKLIIDSTR VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    KONIKRENGIID INTEGER,
    KONIKRENGIADI VARCHAR(20),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    HAMGRUP1 VARCHAR(4),
    HAMGRUP2 VARCHAR(4),
    HAMGRUP3 VARCHAR(4),
    HAMGRUP4 VARCHAR(4),
    HAMGRUP1ADI VARCHAR(30),
    HAMGRUP2ADI VARCHAR(30),
    HAMGRUP3ADI VARCHAR(30),
    HAMGRUP4ADI VARCHAR(30),
    OZELLIKID INTEGER,
    OZELLIKADI VARCHAR(40),
    ELYAFSINIFID INTEGER,
    ELYAFSINIFADI VARCHAR(20),
    ELYAFMARKAID INTEGER,
    ELYAFMARKAADI VARCHAR(20),
    CIRCIRID INTEGER,
    CIRCIRADI VARCHAR(20),
    MENSEIID INTEGER,
    MENSEIADI VARCHAR(20),
    KALINLIKID INTEGER,
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIID INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    DEVREDENKAPADEDI INTEGER,
    DEVREDENMIKTAR DOUBLE PRECISION,
    DEVREDENNETMIKTAR DOUBLE PRECISION,
    GIRISFIRMAID VARCHAR(20),
    GIRISFIRMAADI VARCHAR(100),
    GIRISKAPADEDI INTEGER,
    GIRISMIKTAR DOUBLE PRECISION,
    GIRISNETMIKTAR DOUBLE PRECISION,
    CIKISKAPADEDI INTEGER,
    CIKISMIKTAR DOUBLE PRECISION,
    CIKISNETMIKTAR DOUBLE PRECISION,
    KALANKAPADEDI INTEGER,
    KALANMIKTAR DOUBLE PRECISION,
    KALANNETMIKTAR DOUBLE PRECISION,
    SONALISFIYATI DOUBLE PRECISION,
    ORTALISFIYATI DOUBLE PRECISION,
    SONSATISFIYATI DOUBLE PRECISION,
    ORTSATISFIYATI DOUBLE PRECISION,
    SONTALEPMALIYETFIYATI DOUBLE PRECISION,
    ORTTALEPMALIYETFIYATI DOUBLE PRECISION,
    HARMANELYAFBIRIMMALIYET DOUBLE PRECISION,
    HARMANISCILIKBIRIMMALIYETI DOUBLE PRECISION,
    HARMANDIGERBIRIMMALIYET DOUBLE PRECISION,
    AMBALAJID INTEGER,
    AMBALAJADI ADI20,
    AMBALAJSEKLIID INTEGER,
    AMBALAJSEKLIADI ADI10,
    HARMANIPLIKISLETMEADI ADI50,
    HARMANIPLIKISLETMEID INTEGER,
    IPLIKTALEPNOSTR VARCHAR(30),
    BOYASIPARISNOSTR VARCHAR(30),
    BOYASIPARISFIRMAKODU FIRMAKODUSIZE,
    BOYASIPARISFIRMAADI ADI100)
AS
declare variable FToplamAlisSayisi integer;
declare variable FToplamSatisSayisi integer;
declare variable FToplamAlisBirimFiyat double precision;
declare variable FToplamSatisBirimFiyat double precision;
declare variable FAlisBirimFiyat double precision;
declare variable FSatisBirimFiyat double precision;
declare variable FTalepBirimFiyat double precision;
declare variable FToplamTalepBirimFiyat double precision;
declare variable FToplamTalepSayisi integer;
declare variable FTarih Timestamp;
declare variable FSonTarih Timestamp;
declare variable FIplikIrsaliyeID integer;
declare variable FSonIplikIrsaliyeID integer;
declare variable FHareketTipi integer;
begin
  SiraNo = 0;

  For Select EnvanterRaporuIplikHazirla.IplikDepoId, EnvanterRaporuIplikHazirla.HamId, EnvanterRaporuIplikHazirla.Lot, EnvanterRaporuIplikHazirla.ElyafDegerlendirme,
             EnvanterRaporuIplikHazirla.RenkId, EnvanterRaporuIplikHazirla.MelanjKodu, EnvanterRaporuIplikHazirla.FlamKodu, EnvanterRaporuIplikHazirla.SarimSekliId,
             Ham.Adi, Ham.HamTipiId, HamTipi.Adi, Renk.Adi,
             Ham.HamGrup1, Ham.HamGrup2, Ham.HamGrup3, Ham.HamGrup4, Hg1.Adi, Hg2.Adi, Hg3.Adi, Hg3.Adi,
             Ham.OzellikId, Ozellik.Adi, Ham.ElyafSinifId,ElyafSinif.Adi, Ham.ElyafMarkaId, ElyafMarka.Adi, Ham.CircirId, Circir.Adi,
             Ham.MenseiId, Mensei.Adi,
             Sum(DevredenKapAdedi), Sum(DevredenMiktar), Sum(GirisKapAdedi), Sum(GirisMiktar), sum(CikisKapAdedi),
             sum(CikisMiktar), Sum(KalanKapAdedi), Sum(KalanMiktar), Sum(DevredenNetMiktar), Sum(GirisNetMiktar), sum(CikisNetMiktar), Sum(KalanNetMiktar),
             EnvanterRaporuIplikHazirla.GirisFirmaID, EnvanterRaporuIplikHazirla.GirisFirmaAdi, Ham.KalinlikID, Kalinlik.Adi, Ham.IplikCinsiID, IplikCinsi.Adi,
             EnvanterRaporuIplikHazirla.HarmanElyafBirimMaliyet, EnvanterRaporuIplikHazirla.HarmanIscilikBirimMaliyeti, EnvanterRaporuIplikHazirla.HarmanDigerBirimMaliyet,
             EnvanterRaporuIplikHazirla.KonikCinsiId, KonikCinsi.Adi, EnvanterRaporuIplikHazirla.KonikRengiId, KonikRengi.Adi,
             EnvanterRaporuIplikHazirla.IplikLotGrubuId, EnvanterRaporuIplikHazirla.IplikLotGrubuAdi, EnvanterRaporuIplikHazirla.AmbalajId, EnvanterRaporuIplikHazirla.AmbalajAdi,
             EnvanterRaporuIplikHazirla.AmbalajSekliId, EnvanterRaporuIplikHazirla.AmbalajSekliAdi, EnvanterRaporuIplikHazirla.HarmanIplikIsletmeId, EnvanterRaporuIplikHazirla.HarmanIplikIsletmeAdi,
             EnvanterRaporuIplikHazirla.BoyaSiparisNoStr, EnvanterRaporuIplikHazirla.IplikTalepNoStr, EnvanterRaporuIplikHazirla.BoyaSiparisFirmaKodu, EnvanterRaporuIplikHazirla.BoyaSiparisFirmaAdi
             from EnvanterRaporuIplikHazirla(:AIplikDepoId, :ABaslangicTarih, :ABitisTarih, :AHamTipiId, :AHamId, :AKalinlikId, :AElyafSinifId, :ACircirId, :AMenseiId, :AIplikCinsiId, :ALot, :AGirislerFirmaDetayli, :ARenkId)
    left join Ham on Ham.HamID = EnvanterRaporuIplikHazirla.HamId
    Left Join HamTipi ON HamTipi.HamTipiId = Ham.HamTipiId
    Left Join Renk ON Renk.RenkId = EnvanterRaporuIplikHazirla.RenkId
    left join Mensei on Mensei.MenseiID = Ham.MenseiId
    left join ElyafSinif on ElyafSinif.ElyafSinifID = Ham.ElyafSinifId
    left join Circir on Circir.CircirID = Ham.CircirId
    left join Ozellik on Ozellik.OzellikId = Ham.OzellikId
    left join ElyafMarka on ElyafMarka.ElyafMarkaID = Ham.ElyafMarkaId
    left join Kalinlik on Kalinlik.KalinlikID = Ham.KalinlikId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join HamGrup HG1 on HG1.HamGrupId = Ham.HamGrup1
    left join HamGrup HG2 on HG2.HamGrupId = Ham.HamGrup2
    left join HamGrup HG3 on HG3.HamGrupId = Ham.HamGrup3
    left join HamGrup HG4 on HG4.HamGrupId = Ham.HamGrup4
    Left Join KonikCinsi on KonikCinsi.KonikCinsiId = EnvanterRaporuIplikHazirla.KonikCinsiId
    Left Join KonikRengi on KonikRengi.KonikRengiId = EnvanterRaporuIplikHazirla.KonikRengiId
    group by EnvanterRaporuIplikHazirla.IplikDepoId, EnvanterRaporuIplikHazirla.HamId, EnvanterRaporuIplikHazirla.Lot, EnvanterRaporuIplikHazirla.ElyafDegerlendirme,
             EnvanterRaporuIplikHazirla.RenkId, EnvanterRaporuIplikHazirla.MelanjKodu, EnvanterRaporuIplikHazirla.FlamKodu, EnvanterRaporuIplikHazirla.SarimSekliId,
             Ham.Adi, Ham.HamTipiId, HamTipi.Adi, Renk.Adi,
             Ham.HamGrup1, Ham.HamGrup2, Ham.HamGrup3, Ham.HamGrup4, Hg1.Adi, Hg2.Adi, Hg3.Adi, Hg3.Adi,
             Ham.OzellikId, Ozellik.Adi, Ham.ElyafSinifId,ElyafSinif.Adi, Ham.ElyafMarkaId, ElyafMarka.Adi, Ham.CircirId, Circir.Adi,
             Ham.MenseiId, Mensei.Adi, EnvanterRaporuIplikHazirla.GirisFirmaID, EnvanterRaporuIplikHazirla.GirisFirmaAdi, Ham.KalinlikID, Kalinlik.Adi, Ham.IplikCinsiID, IplikCinsi.Adi,
             EnvanterRaporuIplikHazirla.HarmanElyafBirimMaliyet, EnvanterRaporuIplikHazirla.HarmanIscilikBirimMaliyeti, EnvanterRaporuIplikHazirla.HarmanDigerBirimMaliyet,
             EnvanterRaporuIplikHazirla.KonikCinsiId, KonikCinsi.Adi, EnvanterRaporuIplikHazirla.KonikRengiId, KonikRengi.Adi, EnvanterRaporuIplikHazirla.IplikLotGrubuId, EnvanterRaporuIplikHazirla.IplikLotGrubuAdi,
             EnvanterRaporuIplikHazirla.AmbalajId, EnvanterRaporuIplikHazirla.AmbalajAdi, EnvanterRaporuIplikHazirla.AmbalajSekliId, EnvanterRaporuIplikHazirla.AmbalajSekliAdi,
             EnvanterRaporuIplikHazirla.HarmanIplikIsletmeId, EnvanterRaporuIplikHazirla.HarmanIplikIsletmeAdi, EnvanterRaporuIplikHazirla.BoyaSiparisNoStr, EnvanterRaporuIplikHazirla.IplikTalepNoStr, EnvanterRaporuIplikHazirla.BoyaSiparisFirmaKodu, EnvanterRaporuIplikHazirla.BoyaSiparisFirmaAdi
    order by EnvanterRaporuIplikHazirla.IplikDepoId, Ham.HamTipiId, Ham.Adi, EnvanterRaporuIplikHazirla.Lot, EnvanterRaporuIplikHazirla.MelanjKodu, EnvanterRaporuIplikHazirla.FlamKodu
    into :IplikDepoId, :HamId, :Lot, :ElyafDegerlendirme, :RenkId, :MelanjKodu, :FlamKodu, :SarimSekliId, :HamAdi, :HamTipiId, :HamTipiAdi, :RenkAdi, :HamGrup1, HamGrup2,
         :HamGrup3, :HamGrup4, :HamGrup1Adi, :HamGrup2Adi, :HamGrup3Adi, :HamGrup4Adi,
         :OzellikId, :OzellikAdi, :ElyafSinifId, :ElyafSinifAdi, :ElyafMarkaId, :ElyafMarkaAdi, :CircirId,
         :CircirAdi, :MenseiId, :MenseiAdi,
         :DevredenKapAdedi, :DevredenMiktar, :GirisKapAdedi, :GirisMiktar, :CikisKapAdedi, :CikisMiktar, :KalanKapAdedi, :KalanMiktar, :DevredenNetMiktar, :GirisNetMiktar, :CikisNetMiktar,
         :KalanNetMiktar, :GirisFirmaID, :GirisFirmaAdi, :KalinlikID, :KalinlikAdi, :IplikCinsiID, :IplikCinsiAdi, HarmanElyafBirimMaliyet, :HarmanIscilikBirimMaliyeti, :HarmanDigerBirimMaliyet,
         :KonikCinsiId, :KonikCinsiAdi, :KonikRengiId, :KonikRengiAdi, :IplikLotGrubuId, :IplikLotGrubuAdi, :AmbalajId, :AmbalajAdi, :AmbalajSekliId, :AmbalajSekliAdi, :HarmanIplikIsletmeId, :HarmanIplikIsletmeAdi,
         :BoyaSiparisNoStr, :IplikTalepNoStr, :BoyaSiparisFirmaKodu, :BoyaSiparisFirmaAdi do
    begin
      SiraNo = SiraNo + 1;
      if (SarimSekliId = 0) then SarimSekliIdStr = 'Sert Sarım';
      else SarimSekliIdStr ='Yumuşak Sarım';

      OrtAlisFiyati = Null;
      SonAlisFiyati = Null;
      OrtSatisFiyati = Null;
      SonSatisFiyati = Null;
      OrtTalepMaliyetFiyati = Null;
      SonTalepMaliyetFiyati = Null;

      FToplamAlisSayisi = 0;
      FToplamSatisSayisi = 0;
      FToplamAlisBirimFiyat = 0;
      FToplamSatisBirimFiyat = 0;
      FSonTarih = null;

      for Select iif(IplikIrsaliyeDetay.HareketTipi = 0, BirimFiyat, null),
             iif(IplikIrsaliyeDetay.HareketTipi = 1, BirimFiyat, null),
             IplikIrsaliyeDetay.Tarih,
             IplikIrsaliyeDetay.IplikIrsaliyeId
             from IplikIrsaliyeDetay
        left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
        where
           (IplikIrsaliyeDetay.Tarih >= :ABaslangicTarih and IplikIrsaliyeDetay.Tarih <= :ABitisTarih) and
            (IplikIrsaliyeDetay.HareketTipi = 0 or IplikIrsaliyeDetay.HareketTipi = 1) and
            (StrKarsilastir(FirmaID, :GirisFirmaID) = 1 or :AGirislerFirmaDetayli = 0) and
            StrKarsilastir(IplikIrsaliyeDetay.IplikDepoId , :IplikDepoId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.HamId, :HamId) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.Lot, :Lot) = 1 and
            IntKarsilastir(Ham.HamTipiID, :HamTipiID) = 1 and
            IntKarsilastir(Ham.ElyafSinifId, :ElyafSinifId) = 1 and
            IntKarsilastir(Ham.CircirId, :CircirId) = 1 and
            IntKarsilastir(Ham.MenseiId, :MenseiId) = 1
        Order By IplikIrsaliyeDetay.IplikIrsaliyeId
            into :FAlisBirimFiyat, :FSatisBirimFiyat, :FTarih, :FIplikIrsaliyeID do
      begin
        if (FAlisBirimFiyat is not null) then
        begin
          SonAlisFiyati = FAlisBirimFiyat;
          FToplamAlisSayisi = FToplamAlisSayisi + 1;
          FToplamAlisBirimFiyat = FToplamAlisBirimFiyat + FAlisBirimFiyat;
        end
        if (FSatisBirimFiyat is not null) then
        begin
          FToplamSatisSayisi = FToplamSatisSayisi + 1;
          FToplamSatisBirimFiyat = FToplamSatisBirimFiyat + FSatisBirimFiyat;
          FSonIplikIrsaliyeID = FIplikIrsaliyeID;
          SonSatisFiyati = FSatisBirimFiyat;
        end
      end

      if (FToplamAlisSayisi > 0) then OrtAlisFiyati = FToplamAlisBirimFiyat / FToplamAlisSayisi;
      if (FToplamSatisSayisi > 0) then OrtSatisFiyati = FToplamSatisBirimFiyat / FToplamSatisSayisi;

      /*Select First 1 BirimFiyat from IplikIrsaliyeDetay
        left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
        where IplikIrsaliyeDetay.HareketTipi = 1 and
            (StrKarsilastir(FirmaID, :GirisFirmaID) = 1 or :AGirislerFirmaDetayli = 0) and
            StrKarsilastir(IplikIrsaliyeDetay.IplikDepoId , :IplikDepoId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.HamId, :HamId) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.Lot, :Lot) = 1 and
            IntKarsilastir(Ham.HamTipiID, :HamTipiID) = 1 and
            IntKarsilastir(Ham.ElyafSinifId, :ElyafSinifId) = 1 and
            IntKarsilastir(Ham.CircirId, :CircirId) = 1 and
            IntKarsilastir(Ham.MenseiId, :MenseiId) = 1
        Order By Tarih Desc, IplikIrsaliyeID Desc into :SonSatisFiyati;*/

      FSonIplikIrsaliyeID = null;
      FIplikIrsaliyeId = null;
      FToplamTalepSayisi = 0;
      FToplamTalepBirimFiyat = 0;
      FSonTarih = null;

      for Select BirimMaliyet, IplikIrsaliyeDetay.Tarih, IplikIrsaliyeDetay.IplikIrsaliyeId from IplikIrsaliyeDetay
        left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayID = IplikIrsaliyeDetay.IplikTalepDetayId
        left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
        where IplikIrsaliyeDetay.IplikTalepDetayId is not Null and
            (IplikIrsaliyeDetay.Tarih >= :ABaslangicTarih and IplikIrsaliyeDetay.Tarih <= :ABitisTarih) and
            StrKarsilastir(IplikIrsaliyeDetay.IplikDepoId , :IplikDepoId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.HamId, :HamId) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.Lot, :Lot) = 1 and
            IntKarsilastir(Ham.HamTipiID, :HamTipiID) = 1 and
            IntKarsilastir(Ham.ElyafSinifId, :ElyafSinifId) = 1 and
            IntKarsilastir(Ham.CircirId, :CircirId) = 1 and
            IntKarsilastir(Ham.MenseiId, :MenseiId) = 1 into :FTalepBirimFiyat, :FTarih, :FIplikIrsaliyeID do
       begin
          if (FTalepBirimFiyat is not null) then
          begin
            FToplamTalepSayisi = FToplamTalepSayisi + 1;
            FToplamTalepBirimFiyat = FToplamTalepBirimFiyat + FTalepBirimFiyat;
          end

          if (FTalepBirimFiyat is not null and FSonTarih is null or FSonTarih < FTarih or (FSonTarih = FTarih and FSonIplikIrsaliyeID < FIplikIrsaliyeID)) then
          begin
            FSonTarih = FTarih;
            FSonIplikIrsaliyeID = FIplikIrsaliyeID;
            SonTalepMaliyetFiyati = FTalepBirimFiyat;
          end
