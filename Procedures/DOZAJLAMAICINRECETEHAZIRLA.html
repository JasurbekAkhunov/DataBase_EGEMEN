CREATE PROCEDURE DOZAJLAMAICINRECETEHAZIRLA(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER,
    ADOZAJLAMASISTEMIID INTEGER)
RETURNS (
    RECETEASAMAID INTEGER,
    SIRANO INTEGER,
    DOZAJLAMAPROGRAMNO INTEGER,
    SABLONDOZAJLAMAPROGRAMNO INTEGER,
    RECETEALTASAMAID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNBIRIMID INTEGER,
    OTOMASYONISTASYONNO INTEGER,
    DOZAJLAMAURUNID VARCHAR(15),
    DOZAJLAMAIPLIKURUNID VARCHAR(15),
    URUNTIPIID INTEGER,
    URETIMECIKISFIYATI DOUBLE PRECISION,
    URETIMECIKISFIYATIBUGUN DOUBLE PRECISION,
    URETIMECIKISFIYATIURUNBIRIMID INTEGER,
    MIKTAR DOUBLE PRECISION,
    BANYO DOUBLE PRECISION,
    RECETEMIKTAR DOUBLE PRECISION,
    RECETEBANYO DOUBLE PRECISION,
    DOZAJLAMABIRIMKODU VARCHAR(10),
    DOZAJLAMASEKLI INTEGER,
    DOZAJLAMASEKLIIPLIK INTEGER,
    MAKINAID VARCHAR(10),
    OTOMASYONMAKINANO VARCHAR(10),
    LABRECETEGRUPID INTEGER,
    ORGATEXETEKBASINAAKTARILIR INTEGER,
    BIRIMID VARCHAR(2))
AS
declare variable FReceteAsamaId         integer;
declare variable FDozajlamaProgramNo    integer;
declare variable FSablonDozajlamaProgramNo    integer;
declare variable FReceteAltAsamaId      integer;
declare variable BirimCarpan  Double Precision;
begin
  FReceteAsamaId = Null;
  FDozajlamaProgramNo = Null;
  FSablonDozajlamaProgramNo = Null;
  FReceteAltAsamaId = Null;
  SiraNo = 0;

  for select PartiRecetesiDokumu_2.ReceteAsamaId, PartiRecetesiDokumu_2.DozajlamaProgramNo, PartiRecetesiDokumu_2.SablonDozajlamaProgramNo,
             PartiRecetesiDokumu_2.ReceteAltAsamaId, PartiRecetesiDokumu_2.UrunId,
             Urun.Adi, PartiRecetesiDokumu_2.UrunBirimId, PartiRecetesiDokumu_2.UretimeCikisFiyati, PartiRecetesiDokumu_2.UretimeCikisFiyatiBugun,
             PartiRecetesiDokumu_2.UretimeCikisFiyatiUrunBirimId, PartiRecetesiDokumu_2.MakinaID, PartiRecetesiDokumu_2.OtomasyonMakinaNo,
             DozajlamaUrunId, DozajlamaIplikUrunId, PartiRecetesiDokumu_2.UrunTipiId, PartiRecetesiDokumu_2.Miktar, PartiRecetesiDokumu_2.Banyo,
             PartiRecetesiDokumu_2.ReceteMiktar, PartiRecetesiDokumu_2.ReceteBanyo,
             Birim.DozajlamaBirimKodu, DozajlamaSistemiUrun.DozajlamaSekli, DozajlamaSistemiUrun.DozajlamaSekliIplik, PartiRecetesiDokumu_2.LabReceteGrupID,
             ReceteAsama.OrgatexeTekBasinaAktarilir, Urun.OtomasyonIstasyonNo, PartiRecetesiDokumu_2.BirimId from PartiRecetesiDokumu_2(:APartiKayitId, :AReceteId, :APartiAsamalariId, 1)
    left join Urun On Urun.UrunId = PartiRecetesiDokumu_2.UrunId
    left join DozajlamaSistemiUrun On DozajlamaSistemiUrun.UrunId = Urun.UrunId and DozajlamaSistemiUrun.DozajlamaSistemiId = :ADozajlamaSistemiId
    left join Birim On Birim.BirimId = PartiRecetesiDokumu_2.BirimId
    left join ReceteAsama On ReceteAsama.ReceteAsamaId = PartiRecetesiDokumu_2.ReceteAsamaId
    where (PartiRecetesiDokumu_2.ReceteAsamaId is not null and ReceteAsamaAdi is not null) or (ReceteAltAsamaId is not null and ReceteAltAsamaAdi is not null) or (PartiRecetesiDokumu_2.UrunId is not null and DozajlamaSistemiUrun.UrunId is not null)
    into :ReceteAsamaId, :DozajlamaProgramNo, :SablonDozajlamaProgramNo, :ReceteAltAsamaId, :UrunId, :UrunAdi,
         :UrunBirimId, :UretimeCikisFiyati, :UretimeCikisFiyatiBugun, :UretimeCikisFiyatiUrunBirimId, :MakinaID, :OtomasyonMakinaNo, :DozajlamaUrunId, :DozajlamaIplikUrunId, :UrunTipiId, :Miktar, :Banyo,
         :ReceteMiktar, :ReceteBanyo, :DozajlamaBirimKodu, :DozajlamaSekli, :DozajlamaSekliIplik, :LabReceteGrupID, :OrgatexeTekBasinaAktarilir, :OtomasyonIstasyonNo, :BirimId do
  begin
    if (ReceteAsamaId is null) then
    begin
      ReceteAsamaId = FReceteAsamaId;
      DozajlamaProgramNo = FDozajlamaProgramNo;
      SablonDozajlamaProgramNo = FSablonDozajlamaProgramNo;
    end
    else SiraNo = SiraNo + 1;

    if (ReceteAsamaId is null) then ReceteAsamaId = -1;

    FReceteAsamaId = ReceteAsamaId;
    FDozajlamaProgramNo = DozajlamaProgramNo;
    FSablonDozajlamaProgramNo = SablonDozajlamaProgramNo;

    if (ReceteAltAsamaId is null) then ReceteAltAsamaId = FReceteAltAsamaId;
    if (ReceteAltAsamaId is null) then ReceteAltAsamaId = -1;
    FReceteAltAsamaId = ReceteAltAsamaId;

    if (UrunID is not null and UretimeCikisFiyatiUrunBirimId is not null) then
    begin
      if (UrunBirimId <> UretimeCikisFiyatiUrunBirimId) then
      begin
        Select Carpan from UrunBirim where UrunBirim.UrunBirimID = :UrunBirimId into :BirimCarpan;

        UretimeCikisFiyati = UretimeCikisFiyati / BirimCarpan;
        UretimeCikisFiyatiBugun = UretimeCikisFiyatiBugun / BirimCarpan;
      end
    end
    else
    begin
      UretimeCikisFiyati = 0;
      UretimeCikisFiyatiBugun = 0;
    end

    suspend;
  end
end
