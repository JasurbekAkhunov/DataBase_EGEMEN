CREATE PROCEDURE VBNBASKIURETIMPERFORMANSRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    DESENID VARCHAR(20),
    DESENADI VARCHAR(80),
    DESENVARYANTID INTEGER,
    VARYANTADI VARCHAR(40),
    VARYANTACIKLAMA VARCHAR(100),
    NUMUNEBASKITARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BASKIMETRE DOUBLE PRECISION)
AS
declare variable itDesenTasarim       integer;
begin
  Select itDesenTasarim from IslemTipiGetir into :itDesenTasarim;

  for select Desen.DesenId, Desen.Adi,
             Parti.DesenVaryantId, DesenVaryant.VaryantAdi,
             PartiAsamaStok.CikisTarihi, Parti.FirmaId, Firma.Adi from PartiAsamaStok
    left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join BoyahaneBolum on BoyahaneBolum.BoyahaneBolumId = Asama.BoyahaneBolumId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Kullanici on Kullanici.KullaniciID = BoyaSiparis.SiparisiAlanKullaniciId
    where PartiAsamaStok.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStok.Cikistarihi <= :ABitisTarihi and
          Parti.IslemTipiID = :itDesenTasarim and Asama.BoyahaneBolumId is not null and BoyahaneBolum.NumuneBaski = 1 and
          (Parti.FirmaID = :AFirmaID Or :AfirmaID is null)
    into :DesenId, :DesenAdi, :DesenVaryantId, :VaryantAdi,
         :NumuneBaskiTarihi, :FirmaId, :FirmaAdi do
  begin
    BaskiMetre = null;

    Select Sum(PartiAsamaStokDetay.CikisMetre) from DesenVaryant
      Left join BoyaSiparisBaski on BoyaSiparisBaski.DesenVaryantID = DesenVaryant.DesenVaryantId
      left join PartiStok on BoyaSiparisBaski.BoyaSiparisBaskiId is not null and PartiStok.BoyaSiparisBaskiId = BoyaSiparisBaski.BoyaSiparisBaskiId
      left join PartiAsamaStokDetay on PartiStok.PartiSinifId is not null and PartiAsamaStokDetay.PartiSinifId = PartiStok.PartiSinifId
      left join PartiAsamalari on PartiAsamaStokDetay.PartiAsamalariId is not null and PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
      left join Asama on PartiAsamalari.AsamaId is not null and Asama.AsamaId = PartiAsamalari.AsamaId
      left join Parti on PartiAsamalari.PartiKayitId is not null and Parti.PartiKayitID = PartiAsamalari.PartiKayitId
      where DesenVaryant.DesenVaryantId = :DesenVaryantId and BoyaSiparisBaski.BoyaSiparisBaskiID is not null and partiStok.BoyaSiparisBaskiId is not null and PartiAsamaStokDetay.CikisTarihi >= :ABaslangicTarihi and
            Asama.BaskiAsamasimi = 1 and Parti.IslemTipiID <> :itDesenTasarim
      into :BaskiMetre;

     Suspend;
  end
end
