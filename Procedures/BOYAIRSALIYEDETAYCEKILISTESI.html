    ABOYAIRSALIYEID INTEGER)
RETURNS (
    BOYAIRSALIYEDETAYID INTEGER,
    IPLIKPAKETLEMEID INTEGER,
    PSKALITEKONTROLID INTEGER,
    HAMSTOKMASTERTOPID INTEGER,
    BARKODID VARCHAR(20),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    KOLINO INTEGER,
    KAPNO INTEGER,
    KALITENO INTEGER,
    EN VARCHAR(10),
    GRAMAJ VARCHAR(10),
    DARA NUMERIC(12,3),
    AMBALAJAGIRLIGI NUMERIC(12,3),
    BOBINSAYISI INTEGER,
    BOBINDARASI NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    HACIMEN INTEGER,
    HACIMBOY INTEGER,
    HACIMYUKSEKLIK INTEGER,
    HACIM DOUBLE PRECISION,
    PALETNO VARCHAR(5),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIIDSTR VARCHAR(20),
    ACIKLAMA VARCHAR(100),
    PARTIRENKLOTNO INTEGER,
    IADEBOYAIRSALIYEDETAYID INTEGER,
    PARCASAYISI INTEGER,
    ONAYDURUMU INTEGER,
    HATAPUANI DOUBLE PRECISION,
    BOLUNENPARTIIDSTR VARCHAR(30),
    PAKETLEMEID INTEGER,
    PAKETLEMETARIHI TIMESTAMP,
    OLCUID INTEGER,
    OLCUADI ADI40,
    ADET INTEGER,
    COZGUINDIGOISEMRIIDSTR VARCHAR(30),
    INDIGORECETEID VARCHAR(30),
    BRUTMETRE NUMERIC(12,3))
AS
declare variable PsKaliteKontrolHamStokMasterId integer;
begin
  En = null;
  Gramaj = Null;
  HacimEn = 0;
  HacimBoy = 0;
  HacimYukseklik = 0;
  Adet = null;

  For select IplikPaketleme.IplikPaketlemeId, IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.KaliteNo, IplikPaketleme.KoliNo,
             IplikPaketleme.Dara, IplikPaketleme.BobinSayisi, IplikPaketleme.BobinDarasi,
             IplikPaketleme.AmbalajAgirligi, IplikPaketleme.BrutKilo,
             IplikPaketleme.NetKilo, PartiStokTarti.PaletNo,
             IplikPaketleme.AmbalajId, Ambalaj.Adi, IplikPaketleme.KonikCinsiId, KonikCinsi.Adi,
             LabRecete.RenkDerinligiId, RenkDerinligi.Adi, PartiStokTarti.IadeBoyaIrsaliyeDetayId,
             iif(PartiBolunen.PartiYil is null, null, '('||PartiBolunen.PartiYil||') '||PartiBolunen.PartiId), IplikPaketleme.PaketlemeId, IplikPaketleme.PaketlemeTarihi from PartiStokTarti
  left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId or
  (PartiStokTarti.BarkodID not like 'R-%' and IplikPaketleme.IplikPaketlemeID = Replace(Replace(Replace(PartiStokTarti.BarkodID, 'I-', ''), 'Y-', ''), 'R-', ''))
  Left Join Ambalaj on Ambalaj.AmbalajId = IplikPaketleme.AmbalajId
  Left Join KonikCinsi on KonikCinsi.KonikCinsiId = IplikPaketleme.KonikCinsiId
  left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
  left join PartiStok on PartiStok.PartiSinifId = IplikPaketleme.PartiSinifId
  left join PartiStok PSBolunen on PSBolunen.PartiSinifId = PartiStok.BolunduguPartiSinifId
  left join Parti PartiBolunen on PartiBolunen.PartiKayitId = PsBolunen.PartiKayitId
  where PartiStokTarti.BoyaIrsaliyeId = :ABoyaIrsaliyeId and
        IplikPaketleme.IplikPaketlemeId is not null and (PartiStokTarti.IadeBoyaIrsaliyeId is null or PartiStokTarti.BarkodID like 'I-%' or PartiStokTarti.BarkodID like 'Y-%')
  into :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :KaliteNo, :KoliNo,
       :Dara, :BobinSayisi, :BobinDarasi, :AmbalajAgirligi,
       :BrutKilo, :NetKilo, :PaletNo, :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :RenkDerinligiId, :RenkDerinligiIdStr, :IadeBoyaIrsaliyeDetayId, :BolunenPartiIdStr, :PaketlemeId, :PaketlemeTarihi do
  begin
    BarkodID = 'I-' || EgemenIntegerToStr(IplikPaketlemeId);
    Hacim = 0;
    suspend;
  end

  IplikPaketlemeId = Null;
  KoliNo = null;
  AmbalajId = Null;
  AmbalajAdi= Null;
  KonikCinsiId= Null;
  KonikCinsiAdi= Null;
  AmbalajAgirligi = Null;
  BobinSayisi   = Null;
  BobinDarasi   = Null;
  PaketlemeId   = Null;
  PaketlemeTarihi = Null;

  En = null;
  Gramaj = Null;
  HacimEn = 0;
  HacimBoy = 0;
  HacimYukseklik = 0;
  Adet = null;

  For select Null, PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.KaliteNo, Null,
             PartiStokTarti.Dara, PartiStokTarti.BobinSayisi, Null,
             Null, PartiStokTarti.BrutKilo,
             PartiStokTarti.Kilo, PartiStokTarti.PaletNo,
             Null, Null, Null, Null,
             LabRecete.RenkDerinligiId, RenkDerinligi.Adi, PartiStokTarti.BarkodId, PartiStokTarti.IadeBoyaIrsaliyeDetayId,
             iif(PartiBolunen.PartiYil is null, null, '('||PartiBolunen.PartiYil||') '||PartiBolunen.PartiId), Olcu.OlcuId, Olcu.OlcuAdi, PartiStokTarti.Metre, PartiStokTarti.Adet from PartiStokTarti
  left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
  left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
  left join PartiStok PSBolunen on PSBolunen.PartiSinifId = PartiStok.BolunduguPartiSinifId
  left join Parti PartiBolunen on PartiBolunen.PartiKayitId = PsBolunen.PartiKayitId
  left join Olcu on Olcu.OlcuId = PartiStokTarti.OlcuId
  where PartiStokTarti.BoyaIrsaliyeId = :ABoyaIrsaliyeId and PartiStokTarti.BarkodID like 'R-%'
  into :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :KaliteNo, :KoliNo,
       :Dara, :BobinSayisi, :BobinDarasi, :AmbalajAgirligi,
       :BrutKilo, :NetKilo, :PaletNo, :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :RenkDerinligiId, :RenkDerinligiIdStr, :BarkodID, :IadeBoyaIrsaliyeDetayId, :BolunenPartiIdStr,
       :OlcuId, :OlcuAdi, :Metre, :Adet do
  begin
    Hacim = 0;
    suspend;
  end

  IplikPaketlemeId = Null;
  KoliNo = null;
  AmbalajId = Null;
  AmbalajAdi= Null;
  KonikCinsiId= Null;
  KonikCinsiAdi= Null;
  AmbalajAgirligi = Null;
  BobinSayisi     = Null;
  BobinDarasi     = Null;
  Adet = null;
  BrutMetre = null;
  For select PsKaliteKontrol.PsKaliteKontrolId, PsKaliteKontrol.PartiKayitId, PsKaliteKontrol.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PartiStokTarti.BoyaIrsaliyeDetayId, PsKaliteKontrol.TopNo, PsKaliteKontrol.Kilo, PsKaliteKontrol.Dara, PsKaliteKontrol.Metre,
             PartiStokTarti.KaliteNo, PsKaliteKontrol.En, PsKaliteKontrol.Gramaj,
             PartiStok.HacimEn, PartiStok.HacimBoy, PartiStok.HacimYukseklik, PartiStokTarti.PaletNo,
             LabRecete.RenkDerinligiId, RenkDerinligi.Adi, PsKaliteKontrol.Aciklama, PsKaliteKontrol.PartiRenkLotNo, PartiStokTarti.IadeBoyaIrsaliyeDetayId,
             PsKaliteKontrol.ParcaSayisi,PsKaliteKontrol.OnayDurumu, PsKaliteKontrol.PaketlemeHataPuani,
             iif(PartiBolunen.PartiYil is null, null, '('||PartiBolunen.PartiYil||') '||PartiBolunen.PartiId), Olcu.OlcuId, Olcu.OlcuAdi, PSKaliteKontrol.Adet,
       PsKaliteKontrol.HamStokMasterId, PsKaliteKontrol.BrutMetre from PartiStokTarti
    left join PsKaliteKontrol on PsKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId or
            (PartiStokTarti.BarkodID like 'K-%' and PsKaliteKontrol.PsKaliteKontrolId = Replace(Replace(Replace(Replace(PartiStokTarti.BarkodID, 'I-', ''), 'Y-', ''), 'R-', ''),'K-',''))
    left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
    left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
    left join PartiStok PSBolunen on PSBolunen.PartiSinifId = PartiStok.BolunduguPartiSinifId
    left join Parti PartiBolunen on PartiBolunen.PartiKayitId = PsBolunen.PartiKayitId
    left join Olcu on Olcu.OlcuId = PartiStokTarti.OlcuId
  where PartiStokTarti.BoyaIrsaliyeId = :ABoyaIrsaliyeId and PsKaliteKontrol.PsKaliteKontrolId is not null and
      (PartiStokTarti.IadeBoyaIrsaliyeId is null or PartiStokTarti.BarkodID like 'K-%')
  into :PsKaliteKontrolId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :KapNo, :BrutKilo, :Dara, :Metre,
       :KaliteNo, :En, :Gramaj, :HacimEn, :HacimBoy, :HacimYukseklik, :PaletNo, :RenkDerinligiId, :RenkDerinligiIdStr, :Aciklama, :PartiRenkLotNo, :IadeBoyaIrsaliyeDetayId,
       :ParcaSayisi, :OnayDurumu, :HataPuani, :BolunenPartiIdStr, :OlcuId, :OlcuAdi, :Adet, :PsKaliteKontrolHamStokMasterId, :BrutMetre do
  begin
    NetKilo = BrutKilo - Dara;
    BarkodID = 'K-' || EgemenIntegerToStr(PSKaliteKontrolId);
    Hacim = HacimEn * HacimBoy * HacimYukseklik;
    Hacim = Hacim / 1000000;
  Select First 1  PSDokumaIsEmriBilgisiGetir.CozguIndigoIsEmriIDStr, PSDokumaIsEmriBilgisiGetir.IndigoReceteId from PSDokumaIsEmriBilgisiGetir(Null,:PsKaliteKontrolHamStokMasterId) into :CozguIndigoIsEmriIDStr, :IndigoReceteId;

    suspend;
  end

  PsKaliteKontrolId = Null;
  IplikPaketlemeId = Null;
  KoliNo = null;
  AmbalajId = Null;
  AmbalajAdi= Null;
  KonikCinsiId= Null;
  KonikCinsiAdi= Null;
  AmbalajAgirligi = Null;
  BobinSayisi     = Null;
  BobinDarasi     = Null;
  PartiKayitId = Null;
  PartiStokTartiId = Null;
  KaliteNo = Null;
  HacimEn = Null;
  HacimBoy = Null;
  HacimYukseklik = Null;
  PaletNo = Null;
  KapNo = 0;
  RenkDerinligiId = Null;
  RenkDerinligiIdStr = Null;
  Aciklama = Null;
  IadeBoyaIrsaliyeDetayId = Null;
  ParcaSayisi = Null;
  OnayDurumu = Null;
  HataPuani = Null;
  BolunenPartiIdStr = Null;
  Adet = null;
  BrutMetre = null;

  For select HamStokMasterTop.HamStokMasterTopId,
             HamStokMasterTop.BoyaIrsaliyeDetayId, HamStokMasterTop.BrutKilo, HamStokMasterTop.Kilo, HamStokMasterTop.Metre,
             HamStokMaster.En, HamStokMaster.Grm2, HamStokMasterTop.KoliNo, Olcu.OlcuId, Olcu.OlcuAdi from HamStokMasterTop
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokMasterTop.HamStokMasterId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = HamStokMasterTop.BoyaIrsaliyeDetayId
    left join Olcu on Olcu.OlcuId = HamStokMaster.OlcuId
  where BoyaIrsaliyeDetay.BoyaIrsaliyeId = :ABoyaIrsaliyeId and HamStokMasterTop.BoyaIrsaliyeDetayId is not null
  into :HamStokMasterTopId, :BoyaIrsaliyeDetayId, :BrutKilo, :NetKilo, :Metre,
       :En, :Gramaj, :KoliNo, :OlcuId, :OlcuAdi do
  begin
    BarkodID = 'H-' || EgemenIntegerToStr(HamStokMasterTopId);
    suspend;
  end
end
