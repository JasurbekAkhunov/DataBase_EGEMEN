CREATE PROCEDURE ARIZAIHBARRAPORU_1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAID VARCHAR(10),
    AARIZATIPIID INTEGER,
    ABITTI INTEGER,
    APARCABEKLIYOR INTEGER)
RETURNS (
    ARIZABAKIMID INTEGER,
    ARIZAIHBARID INTEGER,
    MAKINAID VARCHAR(10),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    GITTI SMALLINT,
    GELDIMAKINADURUMUID INTEGER,
    BASLAMATARIHI TIMESTAMP,
    GITTIMAKINADURUMUID INTEGER,
    BITISTARIHI TIMESTAMP)
AS
begin
  For Select ArizaBakim.ArizaBakimId, ArizaBakim.MakinaId, ArizaBakim.PersonelId, Personel.Adi, ArizaBakim.Gitti,
             ArizaBakim.GeldiMakinaDurumuId, GeldiMakinaDurumu.Tarih, ArizaBakim.ArizaIhbarId, ArizaBakim.GittiMakinaDurumuId,
             GittiMakinaDurumu.Tarih
             from ArizaBakim
    left join ArizaIhbar on ArizaIhbar.ArizaIhbarId = ArizaBakim.ArizaIhbarId
    left join Personel on Personel.PersonelId =  ArizaBakim.PersonelId
    left join MakinaDurumu GeldiMakinaDurumu on GeldiMakinaDurumu.MakinaDurumuId = ArizaBakim.GeldiMakinaDurumuId
    left join MakinaDurumu GittiMakinaDurumu on GittiMakinaDurumu.MakinaDurumuId = ArizaBakim.GittiMakinaDurumuId
    Where (ArizaIhbar.MakinaId = :AMakinaId or :AMakinaId is null) and
          (ArizaIhbar.ArizaTipiId = :AArizaTipiId or :AArizaTipiId is null) and
          (GeldiMakinaDurumu.Tarih >= :ABaslangicTarihi and GeldiMakinaDurumu.Tarih <= :ABitisTarihi) and
          (:ABitti is null or (ArizaIhbar.BitirenPersonelId is not null and :ABitti = 1) or
            (ArizaIhbar.BitirenPersonelId is null and :ABitti = 0) ) and
          (ArizaIhbar.ParcaBekliyor = :AParcaBekliyor or :AParcaBekliyor is null)
    into :ArizaBakimId, :MakinaId, :PersonelId, :PersonelAdi, :Gitti, :GeldiMakinaDurumuId,   :BaslamaTarihi, :ArizaIhbarId, :GittiMakinaDurumuId,
         :BitisTarihi   do
  begin
    suspend;
  end
end
