CREATE PROCEDURE HRICINIPLIKELYAFGIRISLISTESI(
    AHAMID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AFIRMAID VARCHAR(20),
    ARENKID INTEGER,
    ALOT VARCHAR(30),
    AIPLIKDEPOID VARCHAR(10))
RETURNS (
    IPLIKIRSALIYEDETAYID INTEGER,
    IPLIKSTOKGIRISNO INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    LOT VARCHAR(30),
    HAMTIPIID INTEGER,
    RENKADI VARCHAR(50),
    GIRISMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    KULLANILABILIRMIKTAR NUMERIC(12,3))
AS
Declare Variable shtFasonAlis SmallInt;
Declare Variable hstKirpinti SmallInt;
Declare Variable FKullanilan  Numeric(12,3);
begin
  Select hstKirpinti from HamTipiGetir into :hstKirpinti;
  Select shtFasonAlis from IplikStokHareketTipiGetir into :shtFasonAlis;

  for select IplikStok.IplikIrsaliyeDetayId, IplikIrsaliyeDetay.IplikStokGirisNo, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
             Renk.Adi, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.Miktar, Sum(IplikStok.Miktar), Ham.HamTipiId from IplikStok
  left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeid = IplikStok.IplikIrsaliyeId
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayid = IplikStok.IplikIrsaliyeDetayId
  left join Renk on Renk.Renkid = IplikIrsaliyeDetay.RenkId
  left join Ham on Ham.Hamid = IplikStok.HamId
  where ( IplikStok.HamID = :AHamId ) and IplikStok.CikisIplikIrsaliyeDetayId is null and
        ( ( IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarihi ) or :AIrsaliyeNo is not null) and
        ( IplikIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
        ( IplikIrsaliyeDetay.IplikDepoId = :AIplikDepoId or :AIplikDepoId is null) and
        ( IplikIrsaliyeDetay.Lot = :ALot or :ALot is null) and
        ( IplikIrsaliyeDetay.RenkId = :ARenkId or :ARenkId is null) and
        ((:AFirmaID is Null and IplikIrsaliye.HareketTipi <> :shtFasonAlis) or (:AFirmaID is not Null and IplikIrsaliye.FirmaID = :AFirmaID and IplikIrsaliye.HareketTipi = :shtFasonAlis) )
  group by IplikStok.IplikIrsaliyeDetayId, IplikIrsaliyeDetay.IplikStokGirisNo, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
           Renk.Adi, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.Miktar, Ham.HamTipiId
  into :IplikIrsaliyeDetayId, :IplikStokGirisNo, :IrsaliyeNo, :IrsaliyeTarihi, :RenkAdi, :Lot, :GirisMiktar, :KalanMiktar, :HamTipiId  do
  begin
    KullanilabilirMiktar = null;
    FKullanilan          = null;

    if (HamTipiId = hstKirpinti) then
      select sum(KirpintiIsEmriRecete.Miktar) from KirpintiIsEmriRecete where KirpintiIsEmriRecete.IplikIrsaliyeDetayId = :IplikIrsaliyeDetayId into :FKullanilan;
    else
      select sum(HarmanRecete.Miktar) from HarmanRecete where HarmanRecete.IplikIrsaliyeDetayId = :IplikIrsaliyeDetayId into :FKullanilan;

    if (FKullanilan is Null) then FKullanilan = 0;

    KullanilabilirMiktar =  GirisMiktar - FKullanilan;

    suspend;
  end
end
