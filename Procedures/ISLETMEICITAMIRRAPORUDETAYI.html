CREATE PROCEDURE ISLETMEICITAMIRRAPORUDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATAMIRTIPIGRUPID VARCHAR(4),
    ATAMIRTIPIID INTEGER,
    ARAPORTIPI INTEGER,
    ABITTI INTEGER,
    APARTITIPI INTEGER,
    AFIRMAID VARCHAR(20),
    AUCRETDURUMU INTEGER,
    ATAMIRDUZELTMETIPIID VARCHAR(10),
    ATAMIRCOZUMSEKLIID INTEGER,
    AISLEMEALINACAK INTEGER,
    APARTIGRUPID VARCHAR(4),
    ASIPARISGRUPID VARCHAR(4),
    ARECETETURUID INTEGER,
    AHATA1ASAMAID INTEGER,
    AHATA2ASAMAID INTEGER,
    AHATA3ASAMAID INTEGER,
    AONLEYICIFAALIYETTIPIID INTEGER)
RETURNS (
    PARTITAMIRID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKIPTALKAYITID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    DENYE VARCHAR(30),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    ASAMATAMIRRAPORSIRANO INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BOYASIPARISDETAYID INTEGER,
    MAKINAID VARCHAR(255),
    GIRISPERSONELADI VARCHAR(255),
    CIKISPERSONELADI VARCHAR(255),
    GIRISTARIHI TIMESTAMP,
    SIPARISENISTENEN VARCHAR(10),
    SIPARISGRAMAJISTENEN VARCHAR(10),
    SIPARISDONMEISTENEN VARCHAR(10),
    SIPARISCEKMEZLIKENISTENEN DOUBLE PRECISION,
    SIPARISCEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    MUSTERISIPARISNO VARCHAR(50),
    ORMECIADI VARCHAR(100),
    ORMECIMAKINANO VARCHAR(100),
    URETICIFIRMAADI VARCHAR(100),
    LOT VARCHAR(150),
    KALITEADI ADI255,
    HAMGRUP1ADI VARCHAR(30))
AS
declare variable FLastPartiTamirId          integer;
declare variable FLastPartiSinifId          integer;

declare variable FPartiTamirId          integer;
declare variable FPartiSinifId          integer;
declare variable FHamAdiToplami         varchar(255);
declare variable FDenye                 varchar(30);
declare variable FKilo                  Numeric(12,3);
declare variable FMetre                 Numeric(12,3);
declare variable FBoyaSiparisDetayId    integer;
declare variable FMakinaId              varchar(10);
declare variable FGirisPersonelAdi      varchar(40);
declare variable FCikisPersonelAdi      varchar(40);
declare variable FAsamaId               integer;
declare variable FAsamaAdi              varchar(40);
declare variable FAsamaTamirRaporSiraNo integer;
declare variable FDate                  Date;
declare variable FGirisTarihi           TimeStamp;
declare variable FPartiStokIptalKayitId integer;
declare variable FSiparisEnIstenen              varchar(10);
declare variable FSiparisGramajIstenen          varchar(10);
declare variable FSiparisDonmeIstenen           varchar(10);
declare variable FSiparisCekmezlikEnIstenen     double precision;
declare variable FSiparisCekmezlikBoyIstenen    double precision;
declare variable FMusteriSiparisNo               varchar(50);
begin
  FLastPartiTamirId = -1;
  FLastPartiSinifId = -1;

  For select IsletmeIciTamirRaporu.PartiTamirId, PartiTamirSinif.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.Denye, PartiStok.Kilo, PartiStok.Metre, PartiStok.BoyaSiparisDetayId,
             PartiAsamaStokDetay.MakinaId, Gp.Adi, Cp.Adi, PartiAsamaStokDetay.GirisTarihi,
             IsletmeIciTamirRaporu.AsamaId, IsletmeIciTamirRaporu.AsamaAdi, IsletmeIciTamirRaporu.AsamaTamirRaporSiraNo, PartiStok.PartiStokIptalKayitId,
             BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.DonmeIstenen,
             BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen, BoyaSiparisDetay.MusteriSiparisNo, Kalite.KaliteAdi, HamGrup.Adi from IsletmeIciTamirRaporu(:ABaslangicTarihi, :ABitisTarihi, :ATamirTipiGrupId, :ATamirTipiId, :ARaporTipi, :ABitti, :APARTITIPI, :AFirmaId, :AUcretDurumu, :ATamirDuzeltmeTipiId, :ATamirCozumSekliId, :AIslemeAlinacak, :APartiGrupId, :ASiparisGrupId, :AReceteTuruId, :AHata1AsamaId, :AHata2AsamaId, :AHata3AsamaId, :AOnleyiciFaaliyetTipiId)
    left join PartiTamirSinif on PartiTamirSinif.PartiTamirId = IsletmeIciTamirRaporu.PartiTamirId
    left join PartiStok on PartiStok.PartiSinifId = PartiTamirSinif.PartiSinifId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiSinifId = PartiTamirSinif.PartiSinifId and PartiAsamaStokDetay.PartiAsamalariId = IsletmeIciTamirRaporu.HataPartiAsamalariId
    left join Personel Gp on Gp.PersonelId = PartiAsamaStokDetay.GirisPersonelId
    left join Personel Cp on Cp.PersonelId = PartiAsamaStokDetay.CikisPersonelId
    left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join HamGrup on HamGrup.HamGrupId = Ham.HamGrup1
  into :FPartiTamirId, :FPartiSinifId, :FHamAdiToplami, :FDenye, :FKilo, :FMetre, :FBoyaSiparisDetayId, :FMakinaId, :FGirisPersonelAdi, :FCikisPersonelAdi, :FGirisTarihi,
       :FAsamaId, :FAsamaAdi, :FAsamaTamirRaporSiraNo, :FPartiStokIptalKayitId,
       :FSiparisEnIstenen, :FSiparisGramajIstenen, :FSiparisDonmeIstenen, :FSiparisCekmezlikEnIstenen, :FSiparisCekmezlikBoyIstenen, :FMusteriSiparisNo, :KaliteAdi, :HamGrup1Adi do
  begin
    if (FLastPartiTamirId <> FPartiTamirId or FLastPartiSinifId <> FPartiSinifId) then
    begin
      if (FLastPartiTamirId <> -1) then suspend;
      PartiTamirId = FPartiTamirId;
      PartiSinifId = FPartiSinifId;
      FLastPartiTamirId = FPartiTamirId;
      FLastPartiSinifId = FPartiSinifId;
      HamAdiToplami = FHamAdiToplami;
      Denye = FDenye;
      Kilo = FKilo;
      Metre = FMetre;
      BoyaSiparisDetayId = FBoyaSiparisDetayId;
      AsamaID= FAsamaId;
      AsamaAdi = FAsamaAdi;
      AsamaTamirRaporSiraNo = FAsamaTamirRaporSiraNo;
      PartiStokIptalKayitId = FPartiStokIptalKayitId;
      SiparisEnIstenen           =  FSiparisEnIstenen;
      SiparisGramajIstenen       =  FSiparisGramajIstenen;
      SiparisDonmeIstenen        =  FSiparisDonmeIstenen;
      SiparisCekmezlikEnIstenen  =  FSiparisCekmezlikEnIstenen;
      SiparisCekmezlikBoyIstenen =  FSiparisCekmezlikBoyIstenen;
      MusteriSiparisNo = FMusteriSiparisNo;

      MakinaId = '';
      GirisPersonelAdi = '';
      CikisPersonelAdi = '';
      GirisTarihi = Null;
      FDate = FGirisTarihi;
    end

    OrmeciAdi = null ;
    OrmeciMakinaNo = null;
    select PartiStokGirisBilgisiGetir.GirisOrmeciAdi, PartiStokGirisBilgisiGetir.GirisOrmeciMakinaNo, PartiStokGirisBilgisiGetir.GirisOrmeFirmaAdi, PartiStokGirisBilgisiGetir.GirisLot from PartiStokGirisBilgisiGetir(:fpartisinifid) into :OrmeciAdi, :OrmeciMakinaNo, :UreticiFirmaAdi, :Lot;

    if (FGirisTarihi < FDate) then FDate = FGirisTarihi;
    GirisTarihi = FDate;

    if (FMakinaId is not null) then
    begin
      if (PosBul(MakinaId, FMakinaId) = 0) then
      if (Char_Length(MakinaId) + Char_Length(FMakinaId) < 250) then
      begin
        if (MakinaId <> '' ) then MakinaId = MakinaId || cast(', ' as varchar(2) character set UTF8);
        MakinaId = MakinaId || FMakinaId;
      end
    end


    if (FGirisPersonelAdi is not null) then
    begin
      if (PosBul(GirisPersonelAdi, FGirisPersonelAdi) = 0) then
      if (Char_Length(GirisPersonelAdi) + Char_Length(FGirisPersonelAdi) < 250) then
      begin
        if (GirisPersonelAdi <> '' ) then GirisPersonelAdi = GirisPersonelAdi || cast(', ' as varchar(2) character set UTF8);
        GirisPersonelAdi = GirisPersonelAdi || FGirisPersonelAdi;
      end
    end

    if (FCikisPersonelAdi is not null) then
    begin
      if (PosBul(CikisPersonelAdi, FCikisPersonelAdi) = 0) then
      if (Char_Length(CikisPersonelAdi) + Char_Length(FCikisPersonelAdi) < 250) then
      begin
        if (CikisPersonelAdi <> '' ) then CikisPersonelAdi = CikisPersonelAdi || cast(', ' as varchar(2) character set UTF8);
        CikisPersonelAdi = CikisPersonelAdi || FCikisPersonelAdi;
      end
    end
  end
  if (FLastPartiTamirId <> -1) then suspend;
end
