    left join Kullanici OK on OK.KullaniciId = StokDetayKaliteKontrol.OnaylayanKullaniciId

    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join Stok on Stok.StokId = StokDetay.StokId
    left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
    left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
    left join Firma on Firma.FirmaId = StokDetay.FirmaId
    where StokDetayKaliteKontrol.OnaylayanKullaniciId is null and Urun.UrunTipiId = :AUrunTipiId

         :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi, :StokId, :IrsaliyeId, :FaturaId do
    begin
      if (KontrolDurumu = 1) then KontrolDurumuStr = 'Kabul';
      else if (KontrolDurumu = 2) then KontrolDurumuStr = 'Ret';
      else KontrolDurumuStr = '';

      if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';

      suspend;
    end
  end

  begin
    for select StokDetay.StokDetayId, StokDetay.FirmaId, Firma.Adi, Stok.FisNo, Stok.FisTarihi, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi, Fatura.FaturaNo, Fatura.FaturaTarihi, StokDetay.DepoId,

    left join StokDetayKaliteKontrol on StokDetayKaliteKontrol.StokDetayId = StokDetay.StokDetayId
    left join Kullanici KK on KK.KullaniciId = StokDetayKaliteKontrol.KontrolKullaniciId
    left join Kullanici OK on OK.KullaniciId = StokDetayKaliteKontrol.OnaylayanKullaniciId
    left join Urun on Urun.UrunId = StokDetay.UrunId

    left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
    left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
    left join Firma on Firma.FirmaId = StokDetay.FirmaId
    where StokDetay.Tarih >= :ABaslangicTarihi and StokDetay.Tarih <= :ABitisTarihi and

          Urun.UrunTipiId = :AUrunTipiId and
          StokDetayKaliteKontrol.OnaylayanKullaniciId is not null
    into :StokDetayId, :FirmaId, :FirmaAdi, :FisNo, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi, :FaturaNo, :FaturaTarihi, :DepoId, :UrunId, :UrunAdi, :BirimId,
         :Miktar, :Lot, :UrunGorunum, :UrunPH, :UrunYogunluk, :UrunKatiMaddeOrani,
         :UrunBome, :UrunSertlik, :UrunDemir, :UrunMangan, :UrunBakir, :UrunNitrit, :UrunNitrat,

      else if (KontrolDurumu = 2) then KontrolDurumuStr = 'Ret';
      else KontrolDurumuStr = '';

      if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';

      suspend;
    end
  end
  else
  begin
    for select StokDetay.StokDetayId, StokDetay.FirmaId, Firma.Adi, Stok.FisNo, Stok.FisTarihi, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi, Fatura.FaturaNo, Fatura.FaturaTarihi, StokDetay.DepoId,

               Urun.Bome, Urun.Sertlik, Urun.Demir, Urun.Mangan, Urun.Bakir, Urun.Nitrit, Urun.Nitrat,
               StokDetayKaliteKontrol.Gorunum, StokDetayKaliteKontrol.PH, StokDetayKaliteKontrol.Yogunluk, StokDetayKaliteKontrol.KatiMaddeOrani,
               StokDetayKaliteKontrol.De, StokDetayKaliteKontrol.Bome, StokDetayKaliteKontrol.Sertlik, StokDetayKaliteKontrol.Demir, StokDetayKaliteKontrol.Mangan, StokDetayKaliteKontrol.Bakir, StokDetayKaliteKontrol.Nitrit, StokDetayKaliteKontrol.Nitrat,
               StokDetayKaliteKontrol.KontrolDurumu, StokDetayKaliteKontrol.KontrolKullaniciId,
               StokDetayKaliteKontrol.KontrolTarihi, KK.Adi, StokDetayKaliteKontrol.OnayDurumu, StokDetayKaliteKontrol.OnaylayanKullaniciId,
               OK.Adi, StokDetayKaliteKontrol.OnayTarihi,
               StokDetay.StokId, StokDetay.IrsaliyeId, StokDetay.FaturaId from StokDetay
    left join StokDetayKaliteKontrol on StokDetayKaliteKontrol.StokDetayId = StokDetay.StokDetayId
    left join Kullanici KK on KK.KullaniciId = StokDetayKaliteKontrol.KontrolKullaniciId
    left join Kullanici OK on OK.KullaniciId = StokDetayKaliteKontrol.OnaylayanKullaniciId
    left join Urun on Urun.UrunId = StokDetay.UrunId
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join Stok on Stok.StokId = StokDetay.StokId
    left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
    left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
    left join Firma on Firma.FirmaId = StokDetay.FirmaId
    where StokDetay.Tarih >= :ABaslangicTarihi and StokDetay.Tarih <= :ABitisTarihi and
          StokDetay.HareketTipi = :ShtAlis and Urun.UrunTipiId = :AUrunTipiId
    into :StokDetayId, :FirmaId, :FirmaAdi, :FisNo, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi, :FaturaNo, :FaturaTarihi, :DepoId, :UrunId, :UrunAdi, :BirimId,
         :Miktar, :Lot, :UrunGorunum, :UrunPH, :UrunYogunluk, :UrunKatiMaddeOrani,
         :UrunBome, :UrunSertlik, :UrunDemir, :UrunMangan, :UrunBakir, :UrunNitrit, :UrunNitrat,
         :Gorunum, :PH, :Yogunluk, :KatiMaddeOrani, :De, :Bome, :Sertlik, :Demir, :Mangan, :Bakir, :Nitrit, :Nitrat,
         :KontrolDurumu, :KontrolKullaniciId, :KontrolTarihi, :KontrolKullaniciAdi,
         :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi, :StokId, :IrsaliyeId, :FaturaId do
    begin
      if (KontrolDurumu = 1) then KontrolDurumuStr = 'Kabul';
      else if (KontrolDurumu = 2) then KontrolDurumuStr = 'Ret';
      else KontrolDurumuStr = '';

      if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';
      suspend;
    end
  end
