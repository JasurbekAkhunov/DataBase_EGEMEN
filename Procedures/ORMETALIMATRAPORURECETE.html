CREATE PROCEDURE ORMETALIMATRAPORURECETE(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASIPARISFIRMAID VARCHAR(20),
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AORMESIPARISSEKLIID INTEGER,
    APLANLAMAFIRMAID VARCHAR(20),
    APLANLAMAMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4),
    AMAKINAPUS INTEGER,
    AMAKINAFEINE INTEGER,
    ABITTI INTEGER,
    AKDVDAHIL INTEGER,
    AONAYDURUMU INTEGER,
    AHAMID INTEGER)
RETURNS (
    ORMEPLANLAMARECETEID INTEGER,
    ORMEPLANLAMAID INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    LOT VARCHAR(150),
    RECETEFIREORANI DOUBLE PRECISION,
    KULLANIMORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    HAMGRUPADI1 VARCHAR(30),
    HAMGRUPADI2 VARCHAR(30),
    HAMGRUPADI3 VARCHAR(30),
    HAMGRUPADI4 VARCHAR(30))
AS
declare variable TahminiFireOrani Double Precision;
declare variable FTalimatReceteFireOrani Double Precision;
declare variable UrunHamId Integer;
declare variable OrmeTalimattaReceteFiresiYap Integer;
declare variable OtomatikUCikislardaReceteFire integer;
begin
  For select OrmeTalimatRaporu.OrmePlanlamaId, OrmeTalimatRaporu.TahminiFireOrani, OrmePlanlamaRecete.OrmePlanlamaReceteId, OrmePlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, OrmePlanlamaRecete.RenkId, Renk.Adi, OrmePlanlamaRecete.Lot, OrmePlanlamaRecete.KullanimOrani, OrmePlanlamaRecete.CikisMiktari, OrmePlanlamaRecete.IadeMiktari,
             OrmeTalimatRaporu.TalimatMiktari * OrmePlanlamaRecete.KullanimOrani / 100, OrmePlanlamaRecete.MelanjKodu, OrmePlanlamaRecete.UreticiId,
             Uretici.Adi, OrmePlanlamaRecete.TahminiFireOrani, Ham.Adi2, Ham.Adi3, Ham.Adi4, HG1.Adi, HG2.Adi, HG3.Adi, HG4.Adi from OrmeTalimatRaporu(:ARAPORTIPI, :ABaslangicTarihi, :ABitisTarihi, :ASiparisFirmaId, :ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ASiparisGrupId, :ASiparisGrup2Id, :AOrmeSiparisSekliId, :APlanlamaFirmaId, :APlanlamaMakinaId, :AMakinaGrupId,  :AMakinaPus, :AMakinaFeine, :ABitti, :AKdvDahil, :AOnayDurumu, :AHamId)
  left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = OrmeTalimatRaporu.OrmePlanlamaId
  left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
  left join HamGrup HG1 on HG1.HamGrupId = Ham.HamGrup1
  left join HamGrup HG2 on HG2.HamGrupId = Ham.HamGrup2
  left join HamGrup HG3 on HG3.HamGrupId = Ham.HamGrup3
  left join HamGrup HG4 on HG4.HamGrupId = Ham.HamGrup4
  left join Renk on Renk.RenkId = OrmePlanlamaRecete.RenkId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  Left Join Kullanici IK on IK.KullaniciId = OrmePlanlamaRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = OrmePlanlamaRecete.KullaniciId
  left join Uretici on Uretici.UreticiId = OrmePlanlamaRecete.UreticiId
  where OrmePlanlamaRecete.OrmePlanlamaId is not null
  order by OrmeTalimatRaporu.OrmePlanlamaId
  into :OrmePlanlamaId, :TahminiFireOrani, :OrmePlanlamaReceteId, :HamId, :hamTipiId, :HamTipiAdi,
       :HamAdi, :RenkId, :RenkAdi, :Lot, :KullanimOrani, :CikisMiktari, :IadeMiktari, :KullanilacakMiktar,
       :MelanjKodu, UreticiID, :UreticiAdi, :FTalimatReceteFireOrani, :HamAdi2, :HamAdi3, :HamAdi4,
       :HamGrupAdi1, :HamGrupAdi2, :HamGrupAdi3, :HamGrupAdi4 do
  begin

    if (HamTipiID = 0 or HamTipiID = 6) then
    begin
      select first 1 OrmeTalimattaReceteFiresiYap, OtomatikUCikislardaReceteFire from Sistem into :OrmeTalimattaReceteFiresiYap, :OtomatikUCikislardaReceteFire;
      select first 1 FireOrani from HamStokRecete where HamId = :UrunHamId and ReceteHamId = :HamId into :ReceteFireOrani;
      if (OtomatikUCikislardaReceteFire = 1) then
      begin
        KullanilacakMiktar = KullanilacakMiktar / ((100 - FTalimatReceteFireOrani) / 100);
        ReceteFireOrani    = FTalimatReceteFireOrani;
      end
      else if (OrmeTalimattaReceteFiresiYap <> 1) then
      begin
        KullanilacakMiktar = KullanilacakMiktar / ((100 - TahminiFireOrani) / 100);
        ReceteFireOrani    = TahminiFireOrani;
      end
      else
        KullanilacakMiktar = KullanilacakMiktar / ((100 - ReceteFireOrani) / 100);
    end
    suspend;
  end
end
