CREATE PROCEDURE FIRMANINDETAYLIHAMSTOKLISTESI_8(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AORMEGIRISIRSALIYENO IRSALIYENOSIZE,
    AEVRAKNO VARCHAR(10),
    AHAMSTOKMASTERID INTEGER,
    ATAMIR INTEGER,
    AFASON INTEGER,
    ARAPORHAMTIPI INTEGER,
    ATAMIRTIPIID INTEGER,
    ASIFIROLANLARIDOK INTEGER,
    AHAMID INTEGER,
    ALOT VARCHAR(150),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AREFERANSFIRMAID VARCHAR(20),
    AORMECIID INTEGER,
    ASINIF VARCHAR(1),
    AHAMSTOKHAREKETTIPIID INTEGER,
    AIRSALIYEGRUPID VARCHAR(4),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ASIPARISFIRMAID VARCHAR(20),
    AIPLIKTALEPYIL INTEGER,
    AIPLIKTALEPID INTEGER,
    AURETIMPARTINO VARCHAR(30),
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AMUSTERIPARTINO VARCHAR(30),
    AVERILENSIPARISKAYITID INTEGER,
    AREFERANSKODU VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ASERTIFIKAGOSTER INTEGER,
    ATARIHFILTRETIPI INTEGER,
    AKALITEID VARCHAR(20))
RETURNS (
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    HAMSTOKSARFFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    HAMSTOKSARFFISIDETAYID INTEGER,
    METRE BOYAMIKTARSIZE,
    PARTISTOKTARTIID INTEGER,
    PARTISINIFID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    CIKILANPARTISINIFID INTEGER,
    CIKILANPARTIIDSTR VARCHAR(20),
    CIKILANPARTIHAMADI ADI255)
AS
begin
  for SELECT HamStokSarfFisi.HamStokSarfFisiId, HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo, HamStokSarfFisi.FisTarihi, Coalesce(HamStokSarfFisiDetay.HamStokSarfFisiDetayId, HamStokSarfFisiDetay2.HamStokSarfFisiDetayId),
         Coalesce(HamStokSarfFisiDetay.Metre, HamStokSarfFisiDetay2.Metre), HSD.HamStokMasterId,
         HSD.HamStokDetayId, PartiStokTarti.PartiStokTartiId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, '('||Parti.PartiYil||') '||Parti.PartiId, Ham.Adi, PartiStokTarti.PartiSinifId, PArtiStokTarti.PartiKayitId
    FROM FirmaninDetayliHamStokListesi_1(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :AIrsaliyeNo, :AOrmeGirisIrsaliyeNo, :AEvrakNo, :AHamStokMasterId, :ATamir, :AFason, :ARaporHamTipi,
                                     :ATamirTipiId, :ASifirOlanlariDok, :AHamId, :ALot, :AMusteriSiparisNo, :AReferansSiparisNo, :AReferansFirmaId, :AOrmeciId, :ASinif, :AHamStokHareketTipiId, :AIrsaliyeGrupId,
                                     :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id, :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id,
                                     :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ABoyaSiparisYil, :ABoyaSiparisId, :ASiparisFirmaId, :AIplikTalepYil, :AIplikTalepId, :AUretimPartiNo, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AMusteriPartiNo, :AVerilenSiparisKayitId, :AReferansKodu, :ASertifikaNo, :ASertifikaGoster, :ATarihFiltreTipi, :AKaliteId) HSD
    left join RafDetay on RafDetay.HamStokDetayId = HSD.HamStokDetayId
    left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
    left join PartiStokTarti on PartiStokTarti.PartiSinifId = PartiStokDetay.PartiSinifId
    left join HamStokSarfFisiDetay on HamStokSarfFisiDetay.HamStokSarfFisiDetayId = PartiStokTarti.HamStokSarfFisiDetayId

    left join HamStokSarfFisiDetay HamStokSarfFisiDetay2 on HamStokSarfFisiDetay2.RafDetayId = RafDetay.RafDetayId
    left join HamStokSarfFisi on (HamStokSarfFisi.HamStokSarfFisiId = HamStokSarfFisiDetay.HamStokSarfFisiId or HamStokSarfFisi.HamStokSarfFisiId = HamStokSarfFisiDetay2.HamStokSarfFisiId)

    left join PartiAsamaStokDetay on (PartiAsamaStokDetay.PartiAsamaStokDetayId = HamStokSarfFisiDetay.PartiAsamaStokDetayId or PartiAsamaStokDetay.PartiAsamaStokDetayId = HamStokSarfFisiDetay2.PartiAsamaStokDetayId)
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    where (PartiStokTarti.HamStokSarfFisiDetayId is not null or HamStokSarfFisiDetay2.HamStokSarfFisiDetayId is not null) and
          (HamStokSarfFisiDetay.PartiAsamaStokDetayId is not null or HamStokSarfFisiDetay2.PartiAsamaStokDetayId is not null)
    into :HamStokSarfFisiId, :FisYil, :FisNo, :FisTarihi, :HamStokSarfFisiDetayId, :Metre, :HamStokMasterId,
         :HamStokDetayId, :PartiStokTartiId, :PartiYil, :PartiId, :CikilanPartiSinifId, :CikilanPartiIdStr, :CikilanPartiHamAdi, :PartiSinifId, :PartiKayitId do
      suspend;

end
