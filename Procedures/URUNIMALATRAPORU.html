CREATE PROCEDURE URUNIMALATRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICURUNGRUP1ID VARCHAR(4),
    ABITISURUNGRUP1ID VARCHAR(4),
    ABASLANGICURUNGRUP2ID VARCHAR(4),
    ABITISURUNGRUP2ID VARCHAR(4),
    ABASLANGICURUNGRUP3ID VARCHAR(4),
    ABITISURUNGRUP3ID VARCHAR(4),
    ABASLANGICURUNGRUP4ID VARCHAR(4),
    ABITISURUNGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER)
RETURNS (
    URUNURETIMID INTEGER,
    TARIH TIMESTAMP,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    MIKTAR DOUBLE PRECISION,
    BIRIMID VARCHAR(2),
    DEPOID VARCHAR(10),
    STOKDETAYID INTEGER,
    URETIMLOT VARCHAR(150),
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable UrunGrup1                     varchar(4);
declare variable UrunGrup2                     varchar(4);
declare variable UrunGrup3                     varchar(4);
declare variable UrunGrup4                     varchar(4);
declare variable okey integer;
declare variable shtUretim SmallInt;
begin
  select shtUretim from StokHareketTipiGetir into :shtUretim;

  For Select UrunUretim.UrunUretimId, UrunUretim.Tarih, UrunUretim.UrunReceteUrunId, Urun.Adi, UrunUretim.Miktar, UrunBirim.BirimId,
             UrunUretim.DepoId, Kullanici.Adi, UrunUretim.DegisimTarihi, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3,
             Urun.UrunGrup4 from UrunUretim
  left join Urun on Urun.UrunId = UrunUretim.UrunReceteUrunId
  left join UrunBirim on UrunBirim.UrunBirimId = UrunUretim.UrunBirimId
  left join Kullanici on Kullanici.KullaniciId = UrunUretim.KullaniciId
  where UrunUretim.Tarih >= :ABaslangicTarihi and UrunUretim.Tarih <= :ABitisTarihi and
        ((:AUrunAralikli = 1 and UrunUretim.UrunReceteUrunId >= :ABaslangicUrunId and (UrunUretim.UrunReceteUrunId <= :ABitisUrunId or :ABitisUrunId = '')) or (:AUrunAralikli = 0)) and
        ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicUrunGrup1Id  and (UrunGrup1 <= :ABitisUrunGrup1Id or :ABitisUrunGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicUrunGrup1Id = '' and :ABitisUrunGrup1Id = '')) and
        ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicUrunGrup2Id and (UrunGrup2 <= :ABitisUrunGrup2Id or :ABitisUrunGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicUrunGrup2Id = '' and :ABitisUrunGrup2Id = '')) and
        ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicUrunGrup3Id and (UrunGrup3 <= :ABitisUrunGrup3Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicUrunGrup3Id = '' and :ABitisUrunGrup3Id = '')) and
        ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicUrunGrup4Id and (UrunGrup4 <= :ABitisUrunGrup4Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicUrunGrup4Id = '' and :ABitisUrunGrup4Id = ''))
  into :UrunUretimId, :Tarih, :UrunId, :UrunAdi, :Miktar, :BirimId, :DepoId, :KullaniciAdi, :DegisimTarihi,
       :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4 do
  begin
    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup4Id, UrunGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      StokDetayId  = Null;
      UretimLot    = Null;
      Select First 1 StokDetay.StokDetayId, StokDetay.Lot from Stok
        Left Join StokDetay on StokDetay.StokId = Stok.StokId
        where Stok.UrunUretimId = :UrunUretimId and Stok.HareketTipi = :shtUretim
      into :StokDetayId, :UretimLot;
      suspend;
    end
  end
end
