CREATE PROCEDURE PARTISTOKBEKLEMERAPORU(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AOTOMATIKBEKLEME INTEGER,
    AASAMAID INTEGER,
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AFIRMAID VARCHAR(20))
RETURNS (
    PARTISTOKBEKLEID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    HAMADITOPLAMI VARCHAR(255),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    DENYE VARCHAR(30),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    PARTIRECETEID VARCHAR(20),
    PARTIRECETEADI VARCHAR(40),
    ASAMAADI VARCHAR(40),
    RECETEID VARCHAR(20),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BASLATANKULLANICIADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENKULLANICIADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    NOTU VARCHAR(1023),
    SIPARISLABRECETEKODU VARCHAR(20),
    SIPARISRENKADI VARCHAR(40),
    OTOMATIKBEKLEME INTEGER,
    OTOMATIKBEKLEMESTR VARCHAR(5))
AS
declare variable FNotu Blob;
declare variable BlobSize       integer;
declare variable APartiKayitId    integer;
begin
  APartiKayitId = null;

  if (APartiId is not null) then
  begin
    Select PartiKayitId from Parti
       where Parti.PartiYil = :APartiYil and Parti.PartiId = :APartiId into :APartiKayitId;
  end

  if (APartiKayitId is not null) then
  begin
    For Select PartiStokBekle.PartiStokBekleId, PartiStokBekle.PartiKayitId, PartiStokBekle.PartiSinifId, PartiStokBekle.PartiAsamalariId,
               Parti.PartiYil, Parti.PartiId, Parti.Fason, Parti.FirmaId, Firma.Adi, Parti.receteId, Recete.Adi,
               PartiStok.HamAdiToplami, PartiStok.Denye, PartiStok.Kilo, PartiStok.Metre, Asama.Adi, PartiAsamalari.ReceteId, BasK.Adi, BitK.Adi, PartiStokBekle.BaslamaTarihi,
               PartiStokBekle.BitisTarihi, PartiStokBekle.Notu, PartiStokBekle.OtomatikBekleme,
               BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.RenkAdi from PartiStokBekle
    Left join Parti on Parti.PartiKayitId = PartiStokBekle.PartiKayitId
    Left join PartiStok on PartiStok.PartiSinifId = PartiStokBekle.PartiSinifId
    Left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiStokBekle.PartiSinifId and PartiAsamaStok.PartiAsamalariId = PartiStokBekle.PartiAsamalariId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiStokBekle.PartiAsamalariId
    left join Asama on Asama.Asamaid = PartiAsamaStok.Asamaid
    Left join Firma on Firma.FirmaId = Parti.FirmaId
    Left join Recete on Recete.ReceteId = Parti.ReceteId
    Left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    Left Join Kullanici BasK on BasK.KullaniciId = PartiStokBekle.BaslatanKullaniciId
    Left Join Kullanici BitK on BitK.KullaniciId = PartiStokBekle.BitirenKullaniciId
    where PartiStokBekle.PartiKayitId = :APartiKayitId and PartiStok.PartiStokIptalKayitId is null
    into :PartiStokBekleId, :PartiKayitId, :PartiSinifId, :PartiAsamalariId, :PartiYil, :PartiId, :Fason, :FirmaId, :FirmaAdi,
         :PartiReceteId, :PartiReceteAdi, :HamAdiToplami, :Denye,  :Kilo, :Metre, :AsamaAdi, :ReceteId,
         :BaslatanKullaniciAdi, :BitirenKullaniciAdi, :BaslamaTarihi, :BitisTarihi, :FNotu, :OtomatikBekleme,
         :SiparisLabReceteKodu, :SiparisRenkAdi do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      BlobSize = char_length(FNotu);

      if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
      else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
      else Notu = '';

      if (OtomatikBekleme = 1) then OtomatikBeklemeStr = 'Evet';
      else OtomatikBeklemeStr = '';
      suspend;
    end
  end
  else if (ARaporTipi = 0) then
  begin
    For Select PartiStokBekle.PartiStokBekleId, PartiStokBekle.PartiKayitId, PartiStokBekle.PartiSinifId, PartiStokBekle.PartiAsamalariId,
               Parti.PartiYil, Parti.PartiId, Parti.Fason, Parti.FirmaId, Firma.Adi, Parti.receteId, Recete.Adi, PartiStok.HamAdiToplami, PartiStok.Denye,
               PartiStok.Kilo, PartiStok.Metre, Asama.Adi, PartiAsamalari.ReceteId, BasK.Adi, BitK.Adi, PartiStokBekle.BaslamaTarihi,
               PartiStokBekle.BitisTarihi,PartiStokBekle.Notu, PartiStokBekle.OtomatikBekleme,
               BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.RenkAdi from PartiStokBekle
    Left join Parti on Parti.PartiKayitId = PartiStokBekle.PartiKayitId
    Left join PartiStok on PartiStok.PartiSinifId = PartiStokBekle.PartiSinifId
    Left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiStokBekle.PartiSinifId and PartiAsamaStok.PartiAsamalariId = PartiStokBekle.PartiAsamalariId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiStokBekle.PartiAsamalariId
    left join Asama on Asama.Asamaid = PartiAsamaStok.Asamaid
    Left join Firma on Firma.FirmaId = Parti.FirmaId
    Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    Left join Recete on Recete.ReceteId = Parti.ReceteId
    Left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    Left Join Kullanici BasK on BasK.KullaniciId = PartiStokBekle.BaslatanKullaniciId
    Left Join Kullanici BitK on BitK.KullaniciId = PartiStokBekle.BitirenKullaniciId
    where PartiStokBekle.BitirenKullaniciId is null and PartiStok.PartiStokIptalKayitId is null and
         (PartiStokBekle.OtomatikBekleme = :AOtomatikBekleme or :AOtomatikBekleme is null) and
         (partiStok.Durumu <= 7) and
         (PartiAsamaStok.AsamaId = :AAsamaId or :AAsamaId is null) and
         (BoyaSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
         (Parti.FirmaId = :AFirmaId or :AFirmaId is Null)
    into :PartiStokBekleId, :PartiKayitId, :PartiSinifId, :PartiAsamalariId, :PartiYil, :PartiId, :Fason, :FirmaId, :FirmaAdi,
         :PartiReceteId, :PartiReceteAdi, :HamAdiToplami, :Denye,  :Kilo, :Metre, :AsamaAdi, :ReceteId,
         :BaslatanKullaniciAdi, :BitirenKullaniciAdi, :BaslamaTarihi, :BitisTarihi, :FNotu, :OtomatikBekleme,
         :SiparisLabReceteKodu, :SiparisRenkAdi do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      BlobSize = char_length(FNotu);

      if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
      else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
      else Notu = '';

      if (OtomatikBekleme = 1) then OtomatikBeklemeStr = 'Evet';
      else OtomatikBeklemeStr = '';
      suspend;
    end
  end
  else
  begin
    For Select PartiStokBekle.PartiStokBekleId, PartiStokBekle.PartiKayitId, PartiStokBekle.PartiSinifId, PartiStokBekle.PartiAsamalariId,
               Parti.PartiYil, Parti.PartiId, Parti.Fason, Parti.FirmaId, Firma.Adi, Parti.ReceteId, Recete.Adi, PartiStok.HamAdiToplami, PartiStok.Denye,
               PartiStok.Kilo, PartiStok.Metre, Asama.Adi, PartiAsamalari.ReceteId, BasK.Adi, BitK.Adi, PartiStokBekle.BaslamaTarihi,
               PartiStokBekle.BitisTarihi,PartiStokBekle.Notu, PartiStokBekle.OtomatikBekleme,
               BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.RenkAdi from PartiStokBekle
    Left join Parti on Parti.PartiKayitId = PartiStokBekle.PartiKayitId
    Left join PartiStok on PartiStok.PartiSinifId = PartiStokBekle.PartiSinifId
    Left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiStokBekle.PartiSinifId and PartiAsamaStok.PartiAsamalariId = PartiStokBekle.PartiAsamalariId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiStokBekle.PartiAsamalariId
    left join Asama on Asama.Asamaid = PartiAsamaStok.Asamaid
    Left join Firma on Firma.FirmaId = Parti.FirmaId
    Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    Left join Recete on Recete.ReceteId = Parti.ReceteId
    Left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    Left Join Kullanici BasK on BasK.KullaniciId = PartiStokBekle.BaslatanKullaniciId
    Left Join Kullanici BitK on BitK.KullaniciId = PartiStokBekle.BitirenKullaniciId
    where PartiStokBekle.Baslamatarihi >= :ABaslangicTarihi and PartiStokBekle.Baslamatarihi <= :ABitisTarihi and PartiStok.PartiStokIptalKayitId is null and
          (PartiStokBekle.OtomatikBekleme = :AOtomatikBekleme or :AOtomatikBekleme is null) and
          (partiStok.Durumu <= 7) and
          (:ARaporTipi is null or (:ARaporTipi = 1 and PartiStokBekle.BitirenKullaniciId is not null)) and
          (PartiAsamaStok.AsamaId = :AAsamaId or :AAsamaId is null) and
         (BoyaSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
         (Parti.FirmaId = :AFirmaId or :AFirmaId is Null)
    into :PartiStokBekleId, :PartiKayitId, :PartiSinifId, :PartiAsamalariId, :PartiYil, :PartiId, :Fason,  :FirmaId, :FirmaAdi,
         :PartiReceteId, :PartiReceteAdi, :HamAdiToplami, :Denye, :Kilo, :Metre, :AsamaAdi, :ReceteId,
         :BaslatanKullaniciAdi, :BitirenKullaniciAdi, :BaslamaTarihi, :BitisTarihi, :FNotu, :OtomatikBekleme,
         :SiparisLabReceteKodu, :SiparisRenkAdi do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      BlobSize = char_length(FNotu);

      if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
      else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
      else Notu = '';

      if (OtomatikBekleme = 1) then OtomatikBeklemeStr = 'Evet';
      else OtomatikBeklemeStr = '';
      suspend;
    end
  end
end
