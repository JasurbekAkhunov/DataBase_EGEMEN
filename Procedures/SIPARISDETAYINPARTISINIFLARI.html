CREATE PROCEDURE SIPARISDETAYINPARTISINIFLARI(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    PROSESID INTEGER,
    PROSESADI VARCHAR(255),
    DURUMU INTEGER,
    DURUMUSTR VARCHAR(20),
    SINIFBOYASIPARISDETAYID INTEGER)
AS
begin
  For Select SiparisDetayinPartileri.BoyaSiparisDetayId, SiparisDetayinPartileri.PartiKayitId, PartiStok.PartiSinifId, PartiStok.HamAdiToplami,
             PartiStok.Kilo, PartiStok.Metre, PartiStok.ProsesId, Proses.Adi, PartiStok.BoyaSiparisDetayId, PartiStok.Durumu, PartiDurum.Adi from SiparisDetayinPartileri(:ABoyaSiparisKayitId)
    left join PartiStok on PartiStok.PartiKayitId = SiparisDetayinPartileri.PartiKayitId
    left join Proses on Proses.ProsesId = PartiStok.ProsesId
    left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
    where SiparisDetayinPartileri.PartiKayitId is not null
    into :BoyaSiparisDetayId, :PartiKayitId, :PartiSinifId, :HamAdiToplami, :Kilo, :Metre, :ProsesId, :ProsesAdi,
         :SinifBoyaSiparisDetayId, :Durumu, :DurumuStr do suspend;
end
