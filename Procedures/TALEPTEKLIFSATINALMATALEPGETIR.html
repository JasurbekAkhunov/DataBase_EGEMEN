    DOVIZID VARCHAR(5),
    TERMINTARIHI TIMESTAMP,
    KULLANILACAKIPLIK1 VARCHAR(40),
    KULLANILACAKIPLIK2 VARCHAR(40),
    KULLANILACAKIPLIK3 VARCHAR(40),
    IPLIKORANI1 DOUBLE PRECISION,
    IPLIKORANI2 DOUBLE PRECISION,
    IPLIKORANI3 DOUBLE PRECISION,
    IPLIKUZUNLUGU1 VARCHAR(40),
    IPLIKUZUNLUGU2 VARCHAR(40),
    IPLIKUZUNLUGU3 VARCHAR(40),
    KURUTMASEKLIADI VARCHAR(40),
    YIKAMADERECE VARCHAR(6),
    ISTENENDONME VARCHAR(10),
    DENYE INTEGER,
    REFERANSSIPARISNO VARCHAR(30),
    REFERANSFIRMAID VARCHAR(20),
    REFERANSFIRMAADI VARCHAR(100),
    LYCRAURETICIID INTEGER,
    LYCRAURETICIADI VARCHAR(20),
    MAKINAMARKASI VARCHAR(20),
    MAKINACINSI VARCHAR(20),
    SISTEMSAYISI INTEGER,
    PUS INTEGER,
    FEINE INTEGER,
    MELANJKODU VARCHAR(20),
    TUPACIKENMAYLI INTEGER,

    AMBALAJSEKLIID INTEGER,
    AMBALAJSEKLIADI VARCHAR(10),
    URUNTALEPNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,

    TALEPNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    VERILENSIPARISIDSTR VARCHAR(100),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),

    URUNGRUP4 VARCHAR(4),

    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable FUrunAdi        varchar(80);
declare variable FHamAdi         varchar(255);
declare variable UrunUrunKategoriId varchar(4);
declare variable HamUrunKategoriId  varchar(4);
declare variable VerilenSiparisYil  integer;
declare variable VerilenSiparisId   integer;
declare variable VerilenSiparisTipiId   integer;
begin

  For Select DISTINCT SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId, SatinAlmaTalepDetay.SatinAlmaTalepDetayId, SatinAlmaTalepDetay.BoyaSiparisKayitId, SatinAlmaTalepDetay.BoyaSiparisDetayId,
             SatinAlmaTalepDetay.OrmeSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             SatinAlmaTalepDetay.UrunId, SatinAlmaTalepDetay.HamId, Urun.Adi, SatinAlmaTalepDetay.Adi, Ham.Adi,
             SatinAlmaTalepDetay.RenkId, Renk.Adi, Renk.DigerAdi, SatinAlmaTalepDetay.Miktar, SatinAlmaTalepDetay.BirimFiyat, SatinAlmaTalepDetay.KdvOrani, SatinAlmaTalepDetay.UrunBirimId,
             UrunBirim.BirimId, SatinAlmaTalepDetay.DovizId, SatinAlmaTalepDetay.TerminTarihi,
             SatinAlmaTalepDetay.KullanilacakIplik1, SatinAlmaTalepDetay.KullanilacakIplik2, SatinAlmaTalepDetay.KullanilacakIplik3,
             SatinAlmaTalepDetay.IplikOrani1, SatinAlmaTalepDetay.IplikOrani2, SatinAlmaTalepDetay.IplikOrani3,
             SatinAlmaTalepDetay.IplikUzunlugu1, SatinAlmaTalepDetay.IplikUzunlugu2, SatinAlmaTalepDetay.IplikUzunlugu3,
             KurutmaSekli.Adi, YikamaDerece.Derece, SatinAlmaTalepDetay.IstenenDonme,
             SatinAlmaTalepDetay.Denye, SatinAlmaTalepDetay.LycraUreticiId, LycraUretici.Adi, SatinAlmaTalepDetay.MakinaMarkasi, SatinAlmaTalepDetay.MakinaCinsi,
             SatinAlmaTalepDetay.SistemSayisi, SatinAlmaTalepDetay.Pus, SatinAlmaTalepDetay.Feine, SatinAlmaTalepDetay.MelanjKodu, SatinAlmaTalepDetay.TupAcikEnMayli,
             SatinAlmaTalepDetay.HamEn, SatinAlmaTalepDetay.HamGramaj, SatinAlmaTalepDetay.MamulEn, SatinAlmaTalepDetay.MamulGramaj,
             SatinAlmaTalepDetay.KontrolOrani, SatinAlmaTalepDetay.KenarKonstruksiyonTipiId, KenarKonstruksiyonTipi.Adi,
             SatinAlmaTalepDetay.AmbalajSekliId, AmbalajSekli.Adi, SatinAlmaTalepDetay.TalepNotu, SatinAlmaTalepDetay.TedarikciFirma,
             SatinAlmaTalepDetay.SatinAlmaKullanimYeriId, SatinAlmaKullanimYeri.Adi, SatinAlmaTalepDetay.SiparisiVerilenMiktar,
             BoyaSiparis.FirmaId, SipFirma.Adi, BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.RenkAdi, SatinAlmaTalepDetay.Notu,
             SatinAlmaTalepDetay.ResimDosyaAdi, SatinAlmaTalepDetay.Kapali,
             Ik.Adi, SatinAlmaTalepDetay.InsertTarihi, SK.Adi, SatinAlmaTalepDetay.DegisimTarihi,
             Urun.UrunKategoriId, HamTipi.UrunKategoriId, Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi,
             SatinAlmaTalepDetay.ReferansFirmaId, RefFirma.Adi, SatinAlmaTalepDetay.ReferansSiparisNo FROM TeklifTalepDetay
    left join SatinAlmaTalepDetay on SatinAlmaTalepDetay.SatinAlmaTalepDetayId = TeklifTalepDetay.SatinAlmaTalepDetayId
    left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
    LEFT JOIN Urun ON Urun.UrunId = SatinAlmaTalepDetay.UrunId
    LEFT JOIN Ham ON Ham.HamId = SatinAlmaTalepDetay.HamId
    left join UrunBirim on UrunBirim.UrunBirimId = SatinAlmaTalepDetay.UrunBirimId and UrunBirim.UrunId = SatinAlmaTalepDetay.UrunId
    LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = SatinAlmaTalepDetay.BoyaSiparisKayitId
    LEFT JOIN Firma SipFirma ON SipFirma.FirmaId = BoyaSiparis.FirmaId
    LEFT JOIN Firma RefFirma ON RefFirma.FirmaId = SatinAlmaTalepDetay.ReferansFirmaId
    left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
    left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
    left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
    left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
    LEFT JOIN BoyaSiparisDetay ON BoyaSiparisDetay.BoyaSiparisDetayId = SatinAlmaTalepDetay.BoyaSiparisDetayId
    left join LycraUretici on LycraUretici.LycraUreticiId = SatinAlmaTalepDetay.LycraUreticiId
    left join KurutmaSekli on KurutmaSekli.KurutmaSekliID = SatinAlmaTalepDetay.KurutmaSekliID
    left join YikamaDerece on YikamaDerece.YikamaDereceID = SatinAlmaTalepDetay.YikamaDereceID
    left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = SatinAlmaTalepDetay.AmbalajSekliId
    LEFT JOIN Renk ON Renk.RenkId = SatinAlmaTalepDetay.RenkId
    left join SatinAlmaKullanimYeri on SatinAlmaKullanimYeri.SatinAlmaKullanimYeriId = SatinAlmaTalepDetay.SatinAlmaKullanimYeriId
    left join KenarKonstruksiyonTipi on KenarKonstruksiyonTipi.KenarKonstruksiyonTipiId = SatinAlmaTalepDetay.KenarKonstruksiyonTipiId
    Left Join Kullanici IK on IK.KullaniciId = SatinAlmaTalepDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = SatinAlmaTalepDetay.KullaniciId
    LEFT JOIN HamTipi ON HamTipi.HamTipiId = Ham.HamTipiId
    WHERE TeklifTalepDetay.TeklifTalepKayitId = :ATeklifTalepKayitId
  into :SatinAlmaTalepYil, :SatinAlmaTalepId, :SatinAlmaTalepDetayId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
       :UrunId, :HamId, :UrunAdi, :FUrunAdi, :FHamAdi, :RenkId, :RenkAdi, :RenkDigerAdi, :Miktar, :BirimFiyat, :KdvOrani, :UrunBirimId, :BirimId, :DovizId, :TerminTarihi,
       :KullanilacakIplik1, :KullanilacakIplik2, :KullanilacakIplik3, :IplikOrani1, :IplikOrani2, :IplikOrani3, :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3,
       :KurutmaSekliAdi, :YikamaDerece, :IstenenDonme,
       :Denye, :LycraUreticiId, :LycraUreticiAdi, :MakinaMarkasi, :MakinaCinsi, :SistemSayisi, :Pus, :Feine, :MelanjKodu, :TupAcikEnMayli,
       :HamEn, :HamGramaj, :MamulEn, :MamulGramaj, :KontrolOrani, :KenarKonstruksiyonTipiId, :KenarKonstruksiyonTipiAdi,
       :AmbalajSekliId, :AmbalajSekliAdi, :UrunTalepNotu, :TedarikciFirma, :SatinAlmaKullanimYeriId, :SatinAlmaKullanimYeriAdi,
       :SiparisiVerilenMiktar, :BoyaSiparisFirmaId, :BoyaSiparisFirmaAdi, :BoyaSiparisLabReceteKodu, :BoyaSiparisRenkAdi, :TalepNotu,
       :ResimDosyaAdi, :Kapali, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :UrunUrunKategoriId, :HamUrunKategoriId, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi,
       :ReferansFirmaId, :ReferansFirmaAdi, :ReferansSiparisNo do
  begin
    SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId)  || ' ';

    if (UrunId is null and HamId is null) then
    begin
      UrunAdi = FUrunAdi;
      UrunKategoriId = Null;
    end
    else if (HamId is not null) then
    begin
      UrunKategoriId  = HamUrunKategoriId;
      UrunAdi = FHamAdi;
    end
    else UrunKategoriId  = UrunUrunKategoriId;

    Tutar = Miktar * BirimFiyat;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    if (Kapali = 1) then KapaliStr = 'KapalÄ±';
    else KapaliStr = '';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    VerilenSiparisIdStr = '';
    For Select VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisID from VerilenSiparisDetay
      left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitID = VerilenSiparisDetay.VerilenSiparisKayitID
      where VerilenSiparisDetay.SatinAlmaTalepDetayId = :SatinAlmaTalepDetayId
      group by VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisID
      into :VerilenSiparisYil, :VerilenSiparisID do
    begin
      if ( Char_Length(VerilenSiparisIdStr) < 70 ) then
      begin
        if (VerilenSiparisIdStr <> '') then VerilenSiparisIdStr = VerilenSiparisIdStr || cast(', ' as varchar(2) character set UTF8);
        VerilenSiparisIdStr = VerilenSiparisIdStr || '(' || EgemenIntegerToStr(VerilenSiparisYil)  || ') ' || EgemenIntegerToStr(VerilenSiparisId);
      end
    end

    suspend;
  end
end
