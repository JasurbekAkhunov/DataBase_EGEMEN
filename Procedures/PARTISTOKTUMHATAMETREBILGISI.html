CREATE PROCEDURE PARTISTOKTUMHATAMETREBILGISI(
    APARTISINIFID INTEGER,
    APARTIKAYITID INTEGER)
RETURNS (
    PSKALITEKONTROLID INTEGER,
    TOPNO INTEGER,
    BRUTKILO DOUBLE PRECISION,
    DARA DOUBLE PRECISION,
    KILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION,
    BRUTMETRE NUMERIC(12,3),
    EN VARCHAR(10),
    GRAMAJ VARCHAR(10),
    HATATIPIID INTEGER,
    HATATIPIADI VARCHAR(20),
    HATATIPIDIGERADI1 VARCHAR(20),
    HATATIPIDIGERADI2 VARCHAR(20),
    BASLANGICMETRE DOUBLE PRECISION,
    BITISMETRE DOUBLE PRECISION,
    HATATIPIPUANI INTEGER,
    SONKAYIT INTEGER,
    PSKALITEKONTROLASAMAID INTEGER,
    KALITENO INTEGER,
    HATAPUANI DOUBLE PRECISION)
AS
begin
  for Select PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.TopNo, PSKaliteKontrol.Kilo, PSKaliteKontrol.Dara, PSKaliteKontrol.Metre, PSKaliteKontrol.BrutMetre, PSKaliteKontrol.En, PSKaliteKontrol.Gramaj,
             PSKaliteKontrolHataMetre.HataTipiId, HataTipi.Adi, HataTipi.DigerAdi1, HataTipi.DigerAdi2,
             PSKaliteKontrolHataMetre.BaslangicMetre, PSKaliteKontrolHataMetre.BitisMetre, HataTipi.HataTipiPuani, PSKaliteKontrolAsama.SonKayit, PSKaliteKontrol.PSKaliteKontrolAsamaID, PSKaliteKontrol.KaliteNo,
             PSKaliteKontrolHataMetre.HataPuani from PSKaliteKontrol
  left join PSKaliteKontrolAsama on PSKaliteKontrolAsama.PSKaliteKontrolAsamaID = PSKaliteKontrol.PSKaliteKontrolAsamaID
  left join PSKaliteKontrolHataMetre on PSKaliteKontrolHataMetre.PSKaliteKontrolId = PSKaliteKontrol.PSKaliteKontrolId
  left join HataTipi on HataTipi.HataTipiId  = PSKaliteKontrolHataMetre.HataTipiId
  where PSKaliteKontrol.PartiSinifId = :APartiSinifId or PSKaliteKontrol.PartiKayitId = :APartiKayitId
  order by PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.TopNo, PSKaliteKontrolHataMetre.BaslangicMetre
  into :PSKaliteKontrolId, :TopNo, :BrutKilo, :Dara, :Metre, :BrutMetre, :En, :Gramaj, :HataTipiId, :HataTipiAdi, :HataTipiDigerAdi1, :HataTipiDigerAdi2,
       :BaslangicMetre, :BitisMetre, :HataTipiPuani, :SonKayit, :PSKaliteKontrolAsamaID, :KaliteNo, :HataPuani do
  begin
    Kilo = BrutKilo - Dara;
    suspend;
  end
end
