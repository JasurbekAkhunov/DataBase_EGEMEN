CREATE PROCEDURE ORMEBOYASIPARISDETAYLISTESI(
    ABOYASIPARISKAYITID INTEGER,
    AORMESIPARISDETAYID INTEGER)
RETURNS (
    SIPARISSIRANO INTEGER,
    BOYASIPARISDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    BOYASIPARISRENKNO INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    BOYASIPARISDETAYRENKADI VARCHAR(40),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    SIPARISKILO NUMERIC(12,3),
    MUSTERISIPARISNO VARCHAR(50),
    ARTIKELNO VARCHAR(20))
AS
begin
  for Select OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.HamId, Ham.Adi, BoyaSiparisDetay.RenkId,
             BoyaliRenk.Adi,
             BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.MusteriRenkKoduveAdi,
             BoyaSiparisDetay.SiparisKilo, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo from BoyaSiparisDetay
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join Renk BoyaliRenk on BoyaliRenk.RenkId = BoyaSiparisDetay.RenkId
  where BoyaSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and
        (BoyaSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId or :AOrmeSiparisDetayId is null) and
        BoyaSiparisDetay.BoyaSiparisDetayID is not null
  order by OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.BoyaSiparisDetayId
  into :SiparisSiraNo, :BoyaSiparisDetayId, :HamId, :HamAdi, :RenkId, :RenkAdi, :LabReceteKodu, :LabReceteAdi,
       :BoyaSiparisDetayRenkAdi, :MusteriRenkKoduveAdi, :SiparisKilo, :ArtikelNo, :MusteriSiparisNo do
  begin
    suspend;
  end
end
