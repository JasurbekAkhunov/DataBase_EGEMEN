CREATE PROCEDURE URUNFIFOBIRIMMALIYETHESAPLA(
    AURUNID URUNKODUSIZE,
    ASTOKFIFOMALIYETBASLANGICTARIHI TIMESTAMP,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AUDBID INTEGER,
    AUDBCARPAN DOUBLE PRECISION,
    ASTOKTABANDOVIZID DOVIZSIZE)
RETURNS (
    STOKDETAYID INTEGER,
    TARIH TIMESTAMP,
    URETIMECIKISTARIHI TIMESTAMP,
    URUNID URUNKODUSIZE,
    HAREKETTIPI INTEGER,
    GIRISCIKIS INTEGER,
    MIKTAR DOUBLE PRECISION,
    UBID INTEGER,
    UBCARPAN DOUBLE PRECISION,
    UDBID INTEGER,
    UDBCARPAN DOUBLE PRECISION,
    UDBMIKTAR DOUBLE PRECISION,
    UDBSTOKDETAYKALANMIKTAR DOUBLE PRECISION,
    UDBSTOKMIKTARI DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    TABANDOVIZKURU DOUBLE PRECISION,
    STOKTABANDOVIZBIRIMFIYAT DOUBLE PRECISION,
    STOKTABANDOVIZID DOVIZSIZE,
    STOKTABANDOVIZMALIYET DOUBLE PRECISION,
    STOKTABANDOVIZORTALAMAMALIYET DOUBLE PRECISION,
    STOKTABANDOVIZMALIYETSTR VARCHAR(255),
    URUNGIRISFIYATLARIKAYITID INTEGER,
    URUNGIRISFIYATLARITARIH TIMESTAMP,
    URUNGIRISMIKTAR DOUBLE PRECISION,
    URUNGIRISBIRIMFIYAT DOUBLE PRECISION,
    URUNGIRISKALANMIKTAR DOUBLE PRECISION)
AS
begin

    if (AStokFifoMaliyetBaslangicTarihi is null) then Select Sistem.StokFifoMaliyetBaslangicTarihi from sistem where sistem.sistemid = 1 into :AStokFifoMaliyetBaslangicTarihi ;
    if (AStokTabanDovizId is null) then select sistem.stoktabandovizid from sistem where sistem.sistemid = 1 into :AStokTabanDovizId ;
    if (AUDBID is null) then select urunbirim.urunbirimid from urunbirim where urunbirim.urunid = :AURUNID and urunbirim.defaultbirim = 1 into :AUDBID;
    if (AUDBCARPAN is null) then select urunbirim.carpan from urunbirim where urunbirim.urunid = :AURUNID and urunbirim.defaultbirim = 1 into :AUDBCARPAN;

    UdbID = AUdbId ;
    UdbCARPAN = AUdbCarpan ;
    UrunId = AUrunId ;

    StokTabanDovizId =  AStokTabanDovizId ;
    UdbStokMiktari = 0 ;
    StokTabanDovizMaliyet = 0 ;
    Delete from urungirisfiyatlari where urungirisfiyatlari.urunid = :UrunId ;

    for select stokdetay.stokdetayid ,stokdetay.tarih, stokdetay.uretimecikistarihi, stokdetay.harekettipi, stokdetay.giriscikis, stokdetay.miktar, stokdetay.urunbirimid,
            UB.carpan, birimfiyat, coalesce(stok.dovizid,irsaliye.dovizid,fatura.dovizid),
            StokDetay.tabandovizkuru2 / StokDetay.tabandovizkuru1
        from stokdetay
        Left Join Stok on Stok.StokId = StokDetay.StokId
        Left Join Irsaliye on Irsaliye.irsaliyeid = StokDetay.irsaliyeid
        Left Join Fatura on Fatura.FaturaID = StokDetay.FaturaId
        left join urunbirim UB on ub.urunbirimid = stokdetay.urunbirimid
        where stokdetay.urunid = :aurunid and
            (stokdetay.tarih >= :AStokFifoMaliyetBaslangicTarihi or :AStokFifoMaliyetBaslangicTarihi is null)  and
            stokdetay.tarih <= :Abitistarihi
        order by Stokdetay.tarih, StokDetay.GirisCikis desc, StokDetay.stokdetayid
        into :StokDetayId, :Tarih, :UretimeCikisTarihi, :HareketTipi, :GirisCikis,  :Miktar, :UbId, :UbCarpan, :BirimFiyat, :DovizId, :TabanDovizKuru do
    begin
        StokTabanDovizBirimFiyat = BirimFiyat ;

        if (StokTabanDovizId <> DovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :StokTabanDovizId, :Tarih) into :TabanDovizKuru;
        else TabanDovizKuru = 1;

        if (TabanDovizKuru is null) then TabanDovizKuru = 0;

        StokTabanDovizBirimFiyat = BirimFiyat * TabanDovizKuru ;

        UdbMiktar = Miktar ;
        if (UDBID<>UBID) then
        begin
          UdbMiktar = Miktar * (UDBCarpan/UbCarpan);
          StokTabanDovizBirimFiyat = StokTabanDovizBirimFiyat / (UDBCarpan/ubcarpan) ;
        end
        UdbStokDetayKalanMiktar = UdbMiktar ;
        UDBStokMiktari = UDBStokMiktari + (GirisCikis * UdbMiktar) ;

        UrunGirisFiyatlariKayitId = null ;
        UrunGirisMiktar = 0 ;
        UrunGirisBirimFiyat = 0 ;
        UrunGirisKalanMiktar = 0 ;

        StokTabanDovizMaliyet = 0 ;
        StokTabanDovizOrtalamaMaliyet = 0 ;
        StokTabanDovizMaliyetStr = '' ;
        if (GirisCikis = 1) then
        begin
            select urungirisfiyatlari.urungirisfiyatlarikayitid from urungirisfiyatlari
            where urungirisfiyatlari.tarih = :tarih and urungirisfiyatlari.urunid = :urunid and
                urungirisfiyatlari.urunbirimid = :UDBID and
                urungirisfiyatlari.birimfiyat = :stoktabandovizbirimfiyat and urungirisfiyatlari.dovizid = :stoktabandovizid
            into :UrunGirisFiyatlariKayitId ;

            if (UrunGirisFiyatlariKayitId is null) then
                insert into urungirisfiyatlari(tarih, girisharekettipiid,urunid, miktar, urunbirimid, birimfiyat, dovizid,kalanmiktar)
                values(:Tarih, :HareketTipi,:UrunId, :UdbMiktar, :UdbId, :StokTabanDovizBirimFiyat, :StokTabanDovizId, :UdbMiktar) ;

            else
                update urungirisfiyatlari
                set urungirisfiyatlari.miktar = urungirisfiyatlari.miktar + :udbmiktar ,
                    urungirisfiyatlari.kalanmiktar = urungirisfiyatlari.kalanmiktar + :udbmiktar
                where urungirisfiyatlari.urungirisfiyatlarikayitid = :UrunGirisFiyatlariKayitId;
        end
        else if ( GirisCikis = -1  ) then
        begin
            for Select urungirisfiyatlari.urungirisfiyatlarikayitid, urungirisfiyatlari.tarih , urungirisfiyatlari.miktar, urungirisfiyatlari.birimfiyat, urungirisfiyatlari.kalanmiktar
                from urungirisfiyatlari
                where urungirisfiyatlari.urunid = :urunid and urungirisfiyatlari.tarih <= :tarih and urungirisfiyatlari.urunbirimid = :UDBID and urungirisfiyatlari.kalanmiktar > 0
                order by urungirisfiyatlari.tarih asc, urungirisfiyatlari.urungirisfiyatlarikayitid asc
                into :UrunGirisFiyatlariKayitId , :UrunGirisFiyatlariTarih, :UrunGirisMiktar, :UrunGirisBirimFiyat, :UrunGirisKalanMiktar do
            begin
                if (UdbStokDetayKalanMiktar > 0 ) then
                begin
                    if ( UdbStokDetayKalanMiktar >= UrunGirisKalanMiktar ) then
                    begin
                        StokTabanDovizMaliyet = StokTabanDovizMaliyet + (UrunGirisKalanMiktar * UrunGirisBirimFiyat ) ;
                        /* TEST İÇİN GEREK
                        if (Char_Length(STOKTABANDOVIZMALIYETSTR) > 0 ) then
                            StokTabanDovizMaliyetStr = StokTabanDovizMaliyetStr || ',' ;

                        StokTabanDovizMaliyetStr = StokTabanDovizMaliyetStr || URUNGIRISFIYATLARITARIH || '-( ' || round(URUNGIRISKALANMIKTAR,5) || '*' || round(URUNGIRISBIRIMFIYAT,5) || ')';
                        */
                        UdbStokDetayKalanMiktar = UdbStokDetayKalanMiktar - UrunGirisKalanMiktar ;
                        -- girişin kendi bittiği için sıfırlıyoruz.
                        UrunGirisKalanMiktar = 0 ;
                    end
                    else if ( UrunGirisKalanMiktar >= UdbStokDetayKalanMiktar ) then
                    begin
                        StokTabanDovizMaliyet = StokTabanDovizMaliyet + ( UdbStokDetayKalanMiktar * UrunGirisBirimFiyat ) ;
                        /* TEST İÇİN GEREK
                        if (Char_Length(STOKTABANDOVIZMALIYETSTR) > 0 ) then
                            StokTabanDovizMaliyetStr = StokTabanDovizMaliyetStr || ',' ;

                        StokTabanDovizMaliyetStr = StokTabanDovizMaliyetStr || URUNGIRISFIYATLARITARIH || '-( ' || round(UDBSTOKDetayKalanMiktar,5) || '*' || round(URUNGIRISBIRIMFIYAT,5) || ')';
                        */
                        UrunGirisKalanMiktar = UrunGirisKalanMiktar - UdbStokDetayKalanMiktar ;
                        UdbStokDetayKalanMiktar = 0;

                    end

                end

                update urungirisfiyatlari set urungirisfiyatlari.kalanmiktar = :UrunGirisKalanMiktar
                where urungirisfiyatlari.urungirisfiyatlarikayitid = :urungirisfiyatlarikayitid ;
            end

            --if (UdbStokMiktari > 0 ) then StokTabanDovizOrtalamaMaliyet = (StokTabanDovizMaliyet / UdbMiktar) * (UDBCarpan/ubcarpan);
            if (StokTabanDovizMaliyet > 0 and UdbMiktar > 0) then StokTabanDovizOrtalamaMaliyet = (StokTabanDovizMaliyet / UdbMiktar) * (UDBCarpan/ubcarpan);
            if (HareketTipi = 6 and Tarih >= ABaslangicTarihi and StokTabanDovizMaliyet > 0) then update stokdetay set stokdetay.birimfiyat = :StokTabanDovizOrtalamaMaliyet where stokdetay.stokdetayid = :StokDetayId ;

        end

        -- 02.07.HareketTipi çağıran prosedüre sadece sarfları göndermek zaman kazandırıyor. test sırasında devre dışı bırakılabilir.
        if (HareketTipi = 6 and Tarih >= ABaslangicTarihi) then
        suspend;

    end
end
