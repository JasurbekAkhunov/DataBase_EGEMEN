CREATE PROCEDURE IPLIKLOTLISTESI_3(
    ABASLANGICISEMRITARIHI TIMESTAMP,
    ABITISISEMRITARIHI TIMESTAMP,
    AIPLIKISLETMEID INTEGER,
    AKAPALI INTEGER,
    AHAMID INTEGER,
    ABASKAPATMATARIHI TIMESTAMP,
    ABITKAPATMATARIHI TIMESTAMP)
RETURNS (
    IPLIKLOTID INTEGER,
    KONIKCINSIID INTEGER,
    KONIKCINSIADI ADI20)
AS
begin
  For Select IplikLotListesi.IplikLotId, KonikCinsi.KonikCinsiId, KonikCinsi.Adi KonikCinsiAdi from IplikLotListesi(:ABaslangicIsEmriTarihi, :ABitisIsEmriTarihi, :AIplikIsletmeId, :AKapali, :AHamId, :ABasKapatmaTarihi, :ABitKapatmaTarihi)
    left join IplikUretim on IplikUretim.IplikLotId = IplikLotListesi.IplikLotId
    left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikUretim.KonikCinsiId
    where KonikCinsi.KonikCinsiId is not null
    group by IplikLotListesi.IplikLotId, KonikCinsi.KonikCinsiId, KonikCinsi.Adi
    into :IplikLotId, :KonikCinsiId, :KonikCinsiAdi do
  begin
    suspend;
  end
end
