CREATE PROCEDURE HGISEMRIDETAYGETIR(
    AHGSIPARISKAYITID INTEGER,
    AHGISEMRIKAYITID INTEGER)
RETURNS (
    HGISEMRIDETAYID TYPE OF COLUMN HGISEMRIDETAY.HGISEMRIDETAYID,
    HGISEMRIKAYITID TYPE OF COLUMN HGISEMRIDETAY.HGISEMRIKAYITID,
    HGISEMRIURUNID TYPE OF COLUMN HGISEMRIDETAY.HGISEMRIURUNID,
    URUNID TYPE OF COLUMN HGISEMRIDETAY.URUNID,
    URUNADI TYPE OF COLUMN URUN.ADI,
    MIKTAR TYPE OF COLUMN HGISEMRIDETAY.MIKTAR,
    NOTU TYPE OF COLUMN HGISEMRIDETAY.NOTU,
    INSERTKULLANICIID TYPE OF COLUMN HGISEMRIDETAY.INSERTKULLANICIID,
    INSERTTARIHI TYPE OF COLUMN HGISEMRIDETAY.INSERTTARIHI,
    KULLANICIID TYPE OF COLUMN HGISEMRIDETAY.KULLANICIID,
    DEGISIMTARIHI TYPE OF COLUMN HGISEMRIDETAY.DEGISIMTARIHI,
    INSERTKULLANICIADI TYPE OF COLUMN KULLANICI.ADI,
    KULLANICIADI TYPE OF COLUMN KULLANICI.ADI)
AS
begin

  For select HGIsEmriDetay.HGIsEmriDetayId, HGIsEmriDetay.HGIsEmriKayitID, HGIsEmriDetay.HGIsEmriUrunID, HGIsEmriDetay.UrunID, Urun.Adi, HGIsEmriDetay.Miktar, HGIsEmriDetay.Notu,
       HGIsEmriDetay.InsertKullaniciId, HGIsEmriDetay.InsertTarihi, HGIsEmriDetay.KullaniciId, HGIsEmriDetay.DegisimTarihi, Ik.Adi, Sk.Adi From HGIsEmri
    Left Join HGIsEmriDetay on HGIsEmriDetay.HGIsEmriKayitID = HGIsEmri.HGIsEmriKayitID
    Left Join Urun on Urun.UrunID = HGIsEmriDetay.UrunID
    Left Join Kullanici IK on IK.KullaniciId = HGIsEmriDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = HGIsEmriDetay.KullaniciId
    where (HGIsEmri.HGSiparisKayitId = :AHGSiparisKayitId or :AHGSiparisKayitId is Null) and
          (HGIsEmriDetay.HGIsEmriKayitID = :AHGIsEmriKayitID or :AHGIsEmriKayitID is Null) and
          HGIsEmriDetay.HGIsEmriDetayID is not Null
    into :HGIsEmriDetayId, :HGIsEmriKayitID, :HGIsEmriUrunID, :UrunID, :UrunAdi, :Miktar, :Notu,
         :InsertKullaniciId, :InsertTarihi, :KullaniciId, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi do
  begin
    suspend;
  end
end
