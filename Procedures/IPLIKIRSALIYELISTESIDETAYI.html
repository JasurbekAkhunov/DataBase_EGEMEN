CREATE PROCEDURE IPLIKIRSALIYELISTESIDETAYI(
    AHAREKETTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AIRSALIYENO IRSALIYENOSIZE,
    ASINIF VARCHAR(1))
RETURNS (
    IPLIKIRSALIYEDETAYID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    IPLIKTALEPKAYITID INTEGER,
    IPLIKTALEPDETAYID INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    LOT VARCHAR(30),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    OZELTESTLER VARCHAR(255),
    ILAVEISLEMLER VARCHAR(255),
    KDV DOUBLE PRECISION,
    TEVKIFAT INTEGER,
    TEVKIFATSTR VARCHAR(3),
    KAPADEDI DOUBLE PRECISION,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    IPLIKIRSALIYEID INTEGER,
    IPLIKFATURAID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    NOTU VARCHAR(1024),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable FBoyaSiparisYil      integer;
declare variable FBoyaSiparisId       integer;

declare variable FNotu Blob;
declare variable BlobSize integer;
begin
  for select IplikIrsaliyeDetay.IplikIrsaliyeDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId, IplikTalepDetay.BoyaSiparisDetayId,
             IplikIrsaliyeDetay.HamId, Ham.HamTipiId, HamTipi.Adi, Ham.Adi, Ham.Adi2, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu, IplikIrsaliyeDetay.Lot,
             coalesce(IplikIrsaliyeDetay.RenkId,HarmanRenk.RenkID), coalesce(Renk.Adi, HarmanRenk.Adi),
             IplikIrsaliyeDetay.Kdv, IplikIrsaliyeDetay.Tevkifat, IplikIrsaliyeDetay.KapAdedi, IplikIrsaliyeDetay.AmbalajId, Ambalaj.Adi,
             IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.BirimFiyat,
             IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
             IplikIrsaliyeDetay.IplikIrsaliyeId, IplikIrsaliyeDetay.IplikFaturaId,
             IK.Adi, IplikIrsaliyeDetay.InsertTarihi, SK.Adi, IplikIrsaliyeDetay.DegisimTarihi, BoyaSiparisDetay.OzelTestler, BoyaSiparisDetay.IlaveIslemler  from IplikIrsaliye
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeId = IplikIrsaliye.IplikIrsaliyeId
  left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  Left join Renk on Renk.RenkID = IplikIrsaliyeDetay.RenkId
  left join Harman on Harman.HarmanKayitId = IplikIrsaliyeDetay.HarmanKayitId
  left join Renk HarmanRenk on HarmanRenk.RenkId = Harman.RenkId
  left join Kullanici IK on IK.KullaniciId  = IplikIrsaliyeDetay.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = IplikIrsaliyeDetay.KullaniciId
  left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikIrsaliyeDetay.IplikTalepDetayId
  left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
  left join Ambalaj on Ambalaj.AmbalajId = IplikIrsaliyeDetay.AmbalajId
  where IplikIrsaliye.HareketTipi = :AHareketTipi And IplikIrsaliyeDetay.IplikIrsaliyeId is not null and
        ( (IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarihi) or (IplikIrsaliye.IrsaliyeNo = :AIrsaliyeNo and :AIrsaliyeNo is not null)) and
        (IplikIrsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
        (IplikIrsaliye.Sinif = :ASinif or :ASinif is null)
  into :IplikIrsaliyeDetayId, :FBoyaSiparisYil, :FBoyaSiparisId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId,
       :HamId, :HamTipiId, :HamTipiAdi, :HamAdi, :HamAdi2, :MelanjKodu, :FlamKodu,  :Lot, :RenkId, :RenkAdi,
       :Kdv, :Tevkifat, :KapAdedi, :AmbalajId, :AmbalajAdi, :Miktar, :BirimFiyat, :IrsaliyeNo, :IrsaliyeTarihi,
       :IplikIrsaliyeId, :IplikFaturaId,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :OzelTestler, :IlaveIslemler do
  begin
    if (Tevkifat = 1) then TevkifatStr = 'Var';
    else TevkifatStr = '';

    if (FBoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') ' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
    else
      BoyaSiparisIdStr = '';

    BlobSize = char_length(FNotu);

    if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
    else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
    else Notu = '';

    suspend;
  end
end
