CREATE PROCEDURE ELYAFURETIMICINHAMPAMUKLISTESI(
    AIPLIKDEPOID VARCHAR(10))
RETURNS (
    IPLIKIRSALIYEDETAYID INTEGER,
    IPLIKSTOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    LOT VARCHAR(30),
    KALITENO INTEGER,
    GELISSIRASI INTEGER,
    MIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    KULLANILABILIRMIKTAR NUMERIC(12,3),
    KULLANILABILIRNETMIKTAR NUMERIC(12,3))
AS
Declare Variable shtFasonAlis SmallInt;
Declare Variable hstHamPamuk SmallInt;
Declare Variable FKullanilan  Numeric(12,3);
begin
  Select hstHamPamuk from HamTipiGetir into :hstHamPamuk;

  for select IplikStok.IplikIrsaliyeDetayId, IplikIrsaliyeDetay.IplikStokGirisNo, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
             IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.NetMiktar, IplikStok.HamId, Ham.Adi,
             Sum(IplikStok.Miktar), Sum(IplikStok.NetMiktar), IplikIrsaliyeDetay.GelisSirasi, IplikIrsaliyeDetay.KaliteNo from IplikStok
  left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeid = IplikStok.IplikIrsaliyeId
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayid = IplikStok.IplikIrsaliyeDetayId
  left join Ham on Ham.Hamid = IplikStok.HamId
  where IplikStok.CikisIplikIrsaliyeDetayId is null and IplikIrsaliyeDetay.IplikDepoId = :AIplikDepoId and Ham.HamTipiId = :hstHamPamuk
  group by IplikStok.IplikIrsaliyeDetayId, IplikIrsaliyeDetay.IplikStokGirisNo, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
           IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.NetMiktar, IplikStok.HamId, Ham.Adi,
           IplikIrsaliyeDetay.GelisSirasi, IplikIrsaliyeDetay.KaliteNo
  into :IplikIrsaliyeDetayId, :IplikStokGirisNo, :IrsaliyeNo, :IrsaliyeTarihi, :Lot, :Miktar, :NetMiktar, :HamId, :HamAdi,
       :KullanilabilirMiktar, :KullanilabilirNetMiktar, :GelisSirasi, :KaliteNo do
  begin
    select Count(*) from ElyafUretimIsEmriRecete where ElyafUretimIsEmriRecete.IplikIrsaliyeDetayId = :IplikIrsaliyeDetayId into :FKullanilan;

    if (FKullanilan is Null) then FKullanilan = 0;
    if (FKullanilan = 0) then suspend;
  end
end
