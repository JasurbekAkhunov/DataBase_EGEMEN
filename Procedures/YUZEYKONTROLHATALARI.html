CREATE PROCEDURE YUZEYKONTROLHATALARI(
    AYUZEYKONTROLTALEPKAYITID INTEGER)
RETURNS (
    HAMHATATIPIID INTEGER,
    HAMHATATIPIADI VARCHAR(20),
    NOKTASALHATA INTEGER,
    BIRIM VARCHAR(5),
    ADET INTEGER)
AS
declare variable BaslangicMetre Double Precision;
declare variable BitisMetre Double Precision;
begin
  For Select HSYuzeyKontrolHataMetre.HamHataTipiId, HamHataTipi.Adi, HamHataTipi.NoktasalHata, Sum(BaslangicMetre), Sum(BitisMetre), Count(*) from YKontrolIsEmri
    left join HSYuzeyKontrol on HSYuzeyKontrol.YKontrolIsEmriKayitId = YKontrolIsEmri.YKontrolIsEmriKayitId
    left join HSYuzeyKontrolHataMetre on HSYuzeyKontrolHataMetre.HSYuzeyKontrolId = HSYuzeyKontrol.HSYuzeyKontrolId
    left join HamHataTipi on HamHataTipi.HamHataTipiId  = HSYuzeyKontrolHataMetre.HamHataTipiId
    where YKontrolIsEmri.YuzeyKontrolTalepKayitId = :AYuzeyKontrolTalepKayitId and HSYuzeyKontrolHataMetre.HSYuzeyKontrolId is not null
    group By HSYuzeyKontrolHataMetre.HamHataTipiId, HamHataTipi.Adi, HamHataTipi.NoktasalHata
    into :HamHataTipiId, :HamHataTipiAdi, :NoktasalHata, :BaslangicMetre, :BitisMetre, :Adet do
  begin
    if (NoktasalHata = 0) then
    begin
      Birim = 'Metre';
      Adet = BitisMetre - BaslangicMetre;
    end
    else Birim = 'Adet';
    suspend;
  end
end
