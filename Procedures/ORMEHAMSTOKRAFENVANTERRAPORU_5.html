CREATE PROCEDURE ORMEHAMSTOKRAFENVANTERRAPORU_5(
    AFASON INTEGER,
    AFIRMAID VARCHAR(20),
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ARAFID VARCHAR(10),
    ADEPOID VARCHAR(10),
    ABOYALI INTEGER,
    AKUMASCINSIID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    AKALITENO INTEGER,
    ASTOKGIRISNO INTEGER,
    ALOT VARCHAR(150),
    APUS INTEGER,
    AFEINE INTEGER,
    AURETICIFIRMAID VARCHAR(20),
    ASIFIROLANLARIDADOK INTEGER,
    AFIRMAADI2YIGOREBILIR INTEGER,
    ARAPORDOVIZID VARCHAR(5),
    AALISFIYAT INTEGER,
    AOLCUID INTEGER)
RETURNS (
    ORMERAFID INTEGER,
    BARKOD VARCHAR(30),
    KILO BOYAMIKTARSIZE,
    BRUTKILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    KAPADEDI INTEGER,
    BOBINGRAMAJI DOUBLE PRECISION,
    KONIKRENGIADI VARCHAR(20),
    AMBALAJNO VARCHAR(10),
    NOTU MEMO)
AS
begin
/* Girişlerin Barkod Detayi */
  AHamTipiID = 3; /*Elindeki ipliklerin barkod bilgisini vereceğiz*/
  For Select OrmeHamStokRafEnvanterRaporu_1.OrmeRafId from OrmeHamStokRafEnvanterRaporu_1(:AFason, :AFirmaId, :ABoyaSiparisYil, :ABoyaSiparisID, :AHamTipiId,
                                 :AHamId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id,
                                 :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id, :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ARafId, :ADepoId, :ABoyali, :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId, :AKaliteNo, :AStokGirisNo, :ALot, :APus, :AFeine, :AUreticiFirmaId, :ASifirOlanlaridaDok, 0, 0, :AFirmaAdi2yiGorebilir, :ARaporDovizId, :AAlisFiyat,
                                 :AOlcuId) --where OrmeHamStokRafEnvanterRaporu_1.RafFirmaId is Null
      into :OrmeRafID do
  begin
    For Select Distinct StokHareketDetay.Barkod, StokHareketDetayMiktar.GirisKilo - StokHareketDetayMiktar.CikisKilo,
              StokHareketDetayMiktar.GirisBrutKilo - StokHareketDetayMiktar.CikisBrutKilo,
              StokHareketDetayMiktar.GirisMetre - StokHareketDetayMiktar.CikisMetre,
              StokHareketDetayMiktar.GirisKapAdedi - StokHareketDetayMiktar.CikisKapAdedi, StokHareketDetay.BobinGramaji, KonikRengi.Adi, StokhareketDetay.AmbalajNo, StokhareketDetay.Notu From StokHareketDetay
              left join KonikRengi on StokHareketDetay.KonikRengiId = KonikRengi.KonikRengiId
      Left Join StokHareketDetayMiktar on StokHareketDetayMiktar.Barkod = StokHareketDetay.Barkod
    where StokHareketDetay.OrmeRafId = :OrmeRafID and StokHareketDetay.OrmeRafTransferFisiDetayId is null and
         ((StokHareketDetayMiktar.GirisKilo - StokHareketDetayMiktar.CikisKilo > 0) or
      (StokHareketDetayMiktar.GirisMetre - StokHareketDetayMiktar.CikisMetre > 0))
    into :Barkod, :Kilo, :BrutKilo, :Metre, :KapAdedi, :BobinGramaji, :KonikRengiAdi, :Ambalajno, :Notu do
    begin
      suspend;
    end

    Barkod                = Null;
    Kilo                  = Null;
    BrutKilo              = Null;
    Metre                 = Null;
    KapAdedi              = Null;

    For Select Distinct StokHareketDetay.Barkod, StokHareketDetayMiktar.GirisKilo - StokHareketDetayMiktar.CikisKilo,
              StokHareketDetayMiktar.GirisBrutKilo - StokHareketDetayMiktar.CikisBrutKilo,
              StokHareketDetayMiktar.GirisMetre - StokHareketDetayMiktar.CikisMetre,
              StokHareketDetayMiktar.GirisKapAdedi - StokHareketDetayMiktar.CikisKapAdedi,
              StokHareketDetay.BobinGramaji, KonikRengi.Adi, StokhareketDetay.ambalajno, StokhareketDetay.Notu From OrmeRafTransferFisiDetay
      Left Join StokHareketDetay on StokHareketDetay.OrmeRafTransferFisiDetayId = OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId
      Left Join StokHareketDetayMiktar on StokHareketDetayMiktar.Barkod = StokHareketDetay.Barkod
      left join KonikRengi on StokHareketDetay.KonikRengiId = KonikRengi.KonikRengiId
    where OrmeRafTransferFisiDetay.YeniOrmeRafId = :OrmeRafID and StokHareketDetay.OrmeRafTransferFisiDetayId is Not null and StokHareketDetay.IplikStokId is Null and StokHareketDetay.Iplikpaketlemeid is Null and
          StokHareketDetay.PSKaliteKontrolId is Null and
         ((StokHareketDetayMiktar.GirisKilo - StokHareketDetayMiktar.CikisKilo > 0) or
         (StokHareketDetayMiktar.GirisMetre - StokHareketDetayMiktar.CikisMetre > 0)  )
    into :Barkod, :Kilo, :BrutKilo, :Metre, :KapAdedi, :BobinGramaji, :KonikRengiAdi, :Ambalajno, :Notu do
    begin
      suspend;
    end
  end
end
