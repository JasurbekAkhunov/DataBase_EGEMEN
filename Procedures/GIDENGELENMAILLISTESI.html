CREATE PROCEDURE GIDENGELENMAILLISTESI(
    AKULLANICIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP)
RETURNS (
    ID INTEGER,
    TIPI INTEGER,
    KIMDENKULLANICIID INTEGER,
    KULLANICIADI VARCHAR(40),
    ALICIADI VARCHAR(40),
    BILGI BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TARIH TIMESTAMP,
    KONU VARCHAR(50),
    MESAJ BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    OKUNDU INTEGER)
AS
begin
  Tipi = 1;
  for select GidenMailId, KimeKullaniciAdi, KimeKullaniciAdi, Bilgi, Tarih, Konu, Mesaj from GidenMail
     where KullaniciId = :AKullaniciId and Tarih >= :ABaslangicTarihi
     into :Id, :KullaniciAdi, :AliciAdi, :Bilgi, :Tarih, :Konu, :Mesaj do suspend;

  Tipi = 2;
  for select GelenMailId, KimdenKullaniciId, KimdenKullaniciAdi, AliciAdi, Bilgi, Tarih, Konu, Mesaj, Okundu from GelenMail
     where KullaniciId = :AKullaniciId and Tarih >= :ABaslangicTarihi
     into :Id, :KimdenKullaniciId, :KullaniciAdi, :AliciAdi, :Bilgi, :Tarih, :Konu, :Mesaj, :Okundu do suspend;
end
