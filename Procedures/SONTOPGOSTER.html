CREATE PROCEDURE SONTOPGOSTER(
    AORMEPLANLAMAID INTEGER)
RETURNS (
    ORMEKALITEKONTROLID INTEGER,
    YAZICIADI VARCHAR(150),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    URETIMPARTINO VARCHAR(30),
    HAMADI ADI255,
    ORMETOPNO INTEGER,
    FIRMAADI ADI255,
    LOT VARCHAR(255),
    MIKTAR DOUBLE PRECISION,
    SIPARISGRUP4ADI VARCHAR(10),
    SIRKETADI VARCHAR(150))
AS
begin
  Select SirketAdi from Sistem where SistemID = 1 into :SirketAdi;

  Select first 1 OrmeKaliteKontrol.OrmeKaliteKontrolId,Personel.Adi,Ham.Adi,OrmePlanlama.UretimPartiNo,
        OrmeKaliteKontrol.TopNo,Firma.Adi, OrmePlanlama.Lot, SiparisGrup4.Adi,OrmeKaliteKontrol.Miktar,
        OrmeMakina.YaziciAdi from OrmeKaliteKontrol
  Left Join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeKaliteKontrol.OrmePlanlamaId
  Left Join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
  Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
  Left Join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
  Left Join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
  Left Join Ham on Ham.HamId = OrmeSiparisDetay.HamId
  Left Join OrmeMakina on OrmeMakina.OrmeMakinaID = ormeplanlama.OrmeMakinaID
  Left Join Personel on Personel.PersonelId = OrmeKaliteKontrol.PersonelId
  where  OrmeKaliteKontrol.OrmePlanlamaId = :AOrmePlanlamaID
  order by OrmeKaliteKontrolId desc
  into :OrmeKaliteKontrolId,:PERSONELADI,:HamAdi,:UretimPartiNo,:OrmeTopNo,:FirmaAdi,
       :Lot,:SiparisGrup4Adi,Miktar,YaziciAdi;

  suspend;
end
