CREATE PROCEDURE ORMEDETAYLIHAMSTOKRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMID INTEGER)
RETURNS (
    STOKHAREKETID INTEGER,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    STOKFIRMAID VARCHAR(20),
    STOKFIRMAADI VARCHAR(100),
    STOKGIRISNO INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    RENKADI VARCHAR(50),
    URETIMPARTINO VARCHAR(30),
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    KALITENO INTEGER,
    ORMEMAKINAID VARCHAR(20),
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    MIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    HAMIADE INTEGER,
    OLCUID INTEGER,
    OLCUADI VARCHAR(40))
AS
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable shtKaydirmaGiris integer;
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, shtKaydirmaGiris from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim, :shtKaydirmaGiris;

  For Select StokHareket.StokHareketId, StokHareket.HareketTipi, StokHareket.FasonStok, StokHareket.FirmaId, SHFirma.Adi,
             StokHareket.StokFirmaId, StokFirma.Adi, OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
             StokHareket.StokGirisNo, StokHareket.HamId, Ham.Adi, Renk.Adi, StokHareket.UretimPartiNo, StokHareket.Sakat, coalesce(StokHareket.OrmeMakinaId, StokHareket.OrmeciMakinaNo), StokHareket.Lot, StokHareket.MelanjKodu, StokHareket.Miktar,
             StokHareket.BrutMiktar, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, StokHareket.BoyaSiparisDetayId,
             StokHareket.OrmeSiparisDetayId, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, StokHareket.HamIade,
             StokHareket.KaliteNo, StokHareket.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Olcu.OlcuId, Olcu.OlcuAdi from StokHareket
  left join Ham on Ham.HamId = StokHareket.HamId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
  left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
  left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
  Left join Renk on Renk.RenkID = StokHareket.RenkId
  left join Firma SHFirma on SHFirma.Firmaid = StokHareket.Firmaid
  left join Firma StokFirma on StokFirma.Firmaid = StokHareket.StokFirmaid
  left join Desen on Desen.DesenId = StokHareket.DesenId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = StokHareket.DesenVaryantId
  left join Olcu on Olcu.OlcuId = StokHareket.OlcuId
  where StokHareket.Tarih >= :ABaslangicTarihi and StokHareket.Tarih <= :ABitisTarihi and
        (StokHareket.HamID = :AHamId or :AHamId is null) and StokHareket.GirisCikis = 1
  into :StokHareketId, :HareketTipi, :Fason, :FirmaId, :FirmaAdi, :StokFirmaId, :StokFirmaAdi, :Fisno, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi,
       :StokGirisNo, :HamId, :HamAdi, :RenkAdi, :UretimPartiNo, :Sakat, :OrmeMakinaId, :Lot, :MelanjKodu,  :Miktar, :BrutMiktar, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
       :OrmeSiparisDetayId, :MusteriSiparisNo, :ReferansSiparisNo, :HamIade, :KaliteNo, :DesenId, :DesenAdi, :VaryantAdi, :HamAdi2, :HamAdi3, :HamAdi4, :OlcuId, :OlcuAdi do
  begin
    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (Sakat = 1) then SakatStr = 'Sakat';
    else SakatStr = '';

    if (HareketTipi = shtDevir) then HareketTipiStr = 'Devir..';
    else if (HareketTipi = shtAlis) then HareketTipiStr = 'Alış';
    else if (HareketTipi = shtSatisIade) then HareketTipiStr = 'Satış İade';
    else if (HareketTipi = shtFasonAlis) then HareketTipiStr = 'Fason İşlenmek Üzere Giriş';
    else if (HareketTipi = shtFasonUretim and HamIade = 0) then HareketTipiStr = 'Fason Üretim';
    else if (HareketTipi = shtFasonUretim and HamIade = 1) then HareketTipiStr = 'Fason Üretim İade';
    else if (HareketTipi = shtUretim and HamIade = 0) then HareketTipiStr = 'Üretim';
    else if (HareketTipi = shtUretim and HamIade = 1) then HareketTipiStr = 'Üretim İade';
    else if (HareketTipi = shtKaydirmaGiris) then HareketTipiStr = 'Stok Kaydırma';
    else HareketTipiStr = 'Hareket Tipi Hatalı...!!!';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    suspend;
  end
end
