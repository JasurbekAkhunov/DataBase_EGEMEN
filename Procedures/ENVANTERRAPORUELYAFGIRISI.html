CREATE PROCEDURE ENVANTERRAPORUELYAFGIRISI(
    ABASLANGICTARIH TIMESTAMP,
    ABITISTARIH TIMESTAMP)
RETURNS (
    IRSALIYETARIHI TIMESTAMP,
    ARACID VARCHAR(11),
    IRSALIYENO IRSALIYENOSIZE,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MENSEIID INTEGER,
    MENSEIADI VARCHAR(20),
    ELYAFSINIFID INTEGER,
    ELYAFSINIFADI VARCHAR(20),
    CIRCIRID INTEGER,
    CIRCIRADI VARCHAR(20),
    LOT VARCHAR(30),
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION,
    NETMIKTAR DOUBLE PRECISION,
    RUTUBET DOUBLE PRECISION,
    KANTARKILO DOUBLE PRECISION)
AS
declare variable hstElyaf     integer;
declare variable shtAlis      integer;
begin
  Select hstElyaf from HamTipiGetir into :hstElyaf;
  Select shtAlis from IplikStokHareketTipiGetir into :shtAlis;

  For Select IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.AracId, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.FirmaId, Firma.Adi,
             IplikIrsaliyeDetay.HamId, Ham.Adi, Ham.MenseiId, Mensei.Adi, Ham.ElyafSinifId, ElyafSinif.Adi,
             Ham.CircirId, Circir.Adi, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.KapAdedi,
             IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.NetMiktar, IplikIrsaliyeDetay.Rutubet, IplikIrsaliyeDetay.KantarKilo from IplikIrsaliye
    left join Firma on Firma.FirmaID = IplikIrsaliye.FirmaID
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeID = IplikIrsaliye.IplikIrsaliyeID
    left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
    left join Mensei on Mensei.MenseiID = Ham.MenseiId
    left join ElyafSinif on ElyafSinif.ElyafSinifID = Ham.ElyafSinifId
    left join Circir on Circir.CircirID = Ham.CircirId
    where IplikIrsaliye.FiiliSevkTarihi >= :ABaslangicTarih and IplikIrsaliye.FiiliSevkTarihi < :ABitisTarih and
          IplikIrsaliye.hareketTipi = :ShtAlis and IplikIrsaliyeDetay.IplikIrsaliyeDetayID is not null and
          Ham.HamTipiID = :hstElyaf
    order by IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.IrsaliyeNo
    into :IrsaliyeTarihi, :AracId, :IrsaliyeNo, :FirmaId, :FirmaAdi, :HamId, :HamAdi, :MenseiId, :MenseiAdi,
         :ElyafSinifId, :ElyafSinifAdi, :CircirId, :CircirAdi, :Lot, :KapAdedi, :Miktar, :NetMiktar,
         :Rutubet, :KantarKilo  do
  begin
    suspend;
  end
end
