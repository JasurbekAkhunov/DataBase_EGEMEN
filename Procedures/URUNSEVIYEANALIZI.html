CREATE PROCEDURE URUNSEVIYEANALIZI(
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    AMINIMUM INTEGER,
    AURUNTIPIID INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BIRIMID VARCHAR(2),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    URUNGRUP1ADI VARCHAR(30),
    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30),
    ALTSEVIYE DOUBLE PRECISION,
    ALTSEVIYENINALTINAINME INTEGER,
    STOKMEVCUDU DOUBLE PRECISION,
    BEKLENENMIKTAR DOUBLE PRECISION,
    SIPARISIVERILMEMISTALEP DOUBLE PRECISION,
    ONERILENIHTIYACMIKTARI DOUBLE PRECISION,
    URUNTIPIID INTEGER,
    URUNTIPIADI VARCHAR(20))
AS
declare variable Okey integer;
declare variable Carpan Double Precision;
declare variable VerilenSipteDetayOnayiKullan integer;
begin
  Select VerilenSipteDetayOnayiKullan from Sistem where SistemId = 1 into :VerilenSipteDetayOnayiKullan;

  for Select Urun.UrunId, Urun.Adi, Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4,
             UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi, Urun.AltSeviye, Urun.AltSeviyeninAltinaInme,
             UBD.BirimId, UBD.Carpan, Urun.UrunTipiId, UrunTipi.Adi from Urun
    left join UrunBirim UBD on UBD.UrunId = urun.Urunid and UBD.DefaultBirim = 1
    left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
    left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
    left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
    left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
    left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
    where (:AUrunAralikli = 0 or (:AUrunAralikli = 1 and Urun.UrunId >= :ABaslangicUrunId) and (Urun.UrunId <= :ABitisUrunId or :ABitisUrunId  = '')) and
           ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicGrup1Id  and (UrunGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
           ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicGrup2Id and (UrunGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
           ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicGrup3Id and (UrunGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
           ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicGrup4Id and (UrunGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = '')) and
           (Urun.AltSeviyeninAltinaInme = 1 or :AMinimum = 0) and
           (Urun.UrunTipiId = :AUrunTipiId or :AUrunTipiId is null)
     into :UrunId, :UrunAdi, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4,  :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi,
          :UrunGrup4Adi, :AltSeviye, :AltSeviyeninAltinaInme, :BirimId, :Carpan, :UrunTipiId, :UrunTipiAdi do
  begin
     if (Carpan is null) then Carpan = 1;

    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, UrunGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      StokMevcudu = 0;
      BeklenenMiktar = 0;

      Select Sum(Devir) + Sum(Alis-AlisIadesi) - Sum(Satis-SatisIadesi) + Sum(Uretim - UretimeCikis) + Sum(FasonUretim - FasonUretimeCikis + FasonUretimIade) FROM DepoStok
      WHERE DepoStok.UrunID = :UrunId
      into :StokMevcudu;

     Select Sum(VerilenSiparisDetay.Miktar - VerilenSiparisDetay.GelenMiktar) from VerilenSiparisDetay
       left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
       where VerilenSiparisDetay.UrunId = :UrunId and ((VerilenSiparis.OnayDurumu = 1 and :VerilenSipteDetayOnayiKullan = 0) or (VerilenSiparisDetay.OnayDurumu = 1 and :VerilenSipteDetayOnayiKullan = 1) ) and
             VerilenSiparisDetay.Kapali = 0 and
             VerilenSiparisDetay.Miktar > VerilenSiparisDetay.GelenMiktar
       into :BeklenenMiktar;


     Select Sum(SatinAlmaTalepDetay.Miktar - SatinAlmaTalepDetay.SiparisiVerilenMiktar) from SatinAlmaTalepDetay
       left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
       where SatinAlmaTalepDetay.UrunId = :UrunId and SatinAlmaTalep.OnayDurumu <> 2 and
             SatinAlmaTalepDetay.Kapali = 0 and SatinAlmaTalep.IstekYeri = 1 and SatinAlmaTalep.SadeceFiyatArastirmasi = 0 and
             SatinAlmaTalepDetay.Miktar > SatinAlmaTalepDetay.SiparisiVerilenMiktar
       into :SiparisiVerilmemisTalep;


      if (StokMevcudu is null) then StokMevcudu = 0;
      else StokMevcudu = StokMevcudu * Carpan;

      if (BeklenenMiktar is null) then BeklenenMiktar = 0;
      if (SiparisiVerilmemisTalep is null) then SiparisiVerilmemisTalep = 0;

      if (AltSeviye > StokMevcudu + BeklenenMiktar) then
       OnerilenIhtiyacMiktari = AltSeviye - (StokMevcudu + BeklenenMiktar);
      else OnerilenIhtiyacMiktari = 0;

      if (AMinimum = 0 or StokMevcudu < AltSeviye) then suspend;

    end

  end
end
