CREATE PROCEDURE BOYASIPRAPORUSEVKBILGISIGETIR(
    AORMESIPARISDETAYID INTEGER)
RETURNS (
    ORGUDENBOYAYAILKSEVKTARIHI TIMESTAMP,
    ORGUDENBOYAYASONSEVKTARIHI TIMESTAMP,
    ORMEYECIKANIPLIK DOUBLE PRECISION,
    ORMEDENIADEIPLIK DOUBLE PRECISION,
    ORULEN DOUBLE PRECISION)
AS
declare variable FHareketTipi integer;
declare variable FHamIade integer;
declare variable FMiktar Double Precision;
declare variable uitOrme integer;
declare variable uitBoya integer;
declare variable hstIplik integer;
declare variable hstLycra integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable FHamTipiId integer;
begin
  select uitOrme, uitBoya from UretimIslemTipiGetir into :uitOrme, :uitBoya;
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;
  select shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis,  :shtUretimeCikis, :shtFasonUretim, :ShtUretim;

  OrmeyeCikanIplik   = 0;
  OrmedenIadeIplik   = 0;
  Orulen             = 0;
  OrgudenBoyayaIlkSevkTarihi = Null;
  OrgudenBoyayaSonSevkTarihi = Null;
  for Select StokHareket.HareketTipi, StokHareket.HamIade, StokHareket.Miktar, Ham.HamTipiId from StokHareket
  left join Ham on Ham.HamId = StokHareket.HamId
  where StokHareket.OrmeSiparisDetayId = :AOrmeSiparisDetayID and
        (StokHareket.HareketTipi = :shtFasonUretimeCikis or
         StokHareket.HareketTipi = :shtFasonUretim or
         StokHareket.HareketTipi = :shtUretimeCikis or
         StokHareket.HareketTipi = :shtUretim) and
         StokHareket.UretimIslemTipiId = :uitOrme
  order by StokHareket.Tarih, StokHareket.StokHareketId
  into :FHareketTipi, :FHamIade, :FMiktar, :FHamTipiId do
  begin
    if ((FHareketTipi = shtFasonUretimeCikis or FHareketTipi = shtUretimeCikis) and (FHamTipiId = hstIplik or FHamTipiId = hstLycra)) then
      OrmeyeCikanIplik = OrmeyeCikanIplik + FMiktar;
    else
    begin
      if (FHamIade = 0) then Orulen = Orulen + FMiktar;
      else if (FHamTipiId = hstIplik) then OrmedenIadeIplik = OrmedenIadeIplik + FMiktar;
    end
  end

  Select first 1 StokHareket.Tarih from StokHareket
  left join Ham on Ham.HamId = StokHareket.HamId
  where StokHareket.OrmeSiparisDetayId = :AOrmeSiparisDetayID and
        (StokHareket.HareketTipi = :shtFasonUretimeCikis or
         StokHareket.HareketTipi = :shtUretimeCikis) and
         StokHareket.UretimIslemTipiId = :uitBoya and StokHareket.HamIade = 0
  order by StokHareket.Tarih ASC, StokHareket.StokHareketId ASC
  into :OrgudenBoyayaIlkSevkTarihi;

  Select first 1 StokHareket.Tarih from StokHareket
  left join Ham on Ham.HamId = StokHareket.HamId
  where StokHareket.OrmeSiparisDetayId = :AOrmeSiparisDetayID and
        (StokHareket.HareketTipi = :shtFasonUretimeCikis or
         StokHareket.HareketTipi = :shtUretimeCikis) and
         StokHareket.UretimIslemTipiId = :uitBoya and StokHareket.HamIade = 0
  order by StokHareket.Tarih DESC, StokHareket.StokHareketId DESC
  into :OrgudenBoyayaSonSevkTarihi;

  suspend;
end
