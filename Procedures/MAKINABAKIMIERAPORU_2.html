CREATE PROCEDURE MAKINABAKIMIERAPORU_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINABAKIMTIPIID INTEGER,
    AMAKINABAKIMBOLUMUID INTEGER,
    AMAKINAGRUPID VARCHAR(4),
    AMAKINAID VARCHAR(10),
    AORMEMAKINAID VARCHAR(10),
    AIPLIKMAKINAID VARCHAR(10),
    ADOKUMAMAKINAID VARCHAR(10),
    ABOYA INTEGER,
    AORME INTEGER,
    AIPLIK INTEGER,
    ADOKUMA INTEGER,
    AIESTARTBASLANGICTARIHI TIMESTAMP,
    AIESTARTBITISTARIHI TIMESTAMP,
    ABSTARTBASLANGICTARIHI TIMESTAMP,
    ABSTARTBITISTARIHI TIMESTAMP,
    ABENDBASLANGICTARIHI TIMESTAMP,
    ABENDBITISTARIHI TIMESTAMP,
    AIEENDBASLANGICTARIHI TIMESTAMP,
    AIEENDBITISTARIHI TIMESTAMP,
    ATAHMINIBASLANGICTARIHI TIMESTAMP,
    ATAHMINIBITISTARIHI TIMESTAMP,
    ABASLAYANLAR INTEGER,
    ABAKIMBASLAYANLAR INTEGER,
    ABAKIMBITENLER INTEGER,
    ABITENLER INTEGER,
    AARIZAISEMRI INTEGER,
    ASISTEMOLUSTURDU INTEGER,
    ABASLANGICTAHMINISAAT INTEGER,
    ABITISTAHMINISAAT INTEGER,
    AKAPALILARIDOK INTEGER,
    ABAKIMDAIRESIID INTEGER,
    AHIZMETID VARCHAR(20),
    AURUNID VARCHAR(20))
RETURNS (
    MAKINABAKIMISEMRIID INTEGER,
    MAKINABAKIMIEHIZMETID INTEGER,
    MAKINABAKIMIEURUNID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    OPSIYONEL INTEGER,
    OPSIYONELSTR VARCHAR(10),
    MIKTAR INTEGER,
    TAHMINIBASLANGIC TIMESTAMP,
    TOLERANS DOUBLE PRECISION,
    SATINALMATALEPKAYITID INTEGER,
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30),
    TALEPTARIHI TIMESTAMP,
    TALEPONAYTARIHI TIMESTAMP,
    STOKMEVCUDU DOUBLE PRECISION,
    GECMISBAKIMZORUNLUIHTIYACI DOUBLE PRECISION,
    GECMISBAKIMOPSIYONELIHTIYACI DOUBLE PRECISION,
    GUNUNBAKIMIHTIYACTOPLAMI DOUBLE PRECISION,
    DIREKTTALEPMIKTARI DOUBLE PRECISION,
    TOPLAMTALEPMIKTARI DOUBLE PRECISION,
    DIREKTSIPARISMIKTARI DOUBLE PRECISION,
    TOPLAMSIPARISMIKTARI DOUBLE PRECISION,
    SIPARISGELENMIKTAR DOUBLE PRECISION,
    SIPARISKALANMIKTAR DOUBLE PRECISION)
AS
begin
  for select MakinaBakimIERaporu.MakinaBakimIsEmriId, MakinaBakimIERaporu.TahminiBaslangic, MakinaBakimIERaporu.Tolerans,
        MakinaBakimIERaporu.SatinAlmaTalepKayitId, MakinaBakimIERaporu.SatinAlmaTalepYil, MakinaBakimIERaporu.SatinAlmaTalepId, MakinaBakimIERaporu.SatinAlmaTalepIdStr,
        SatinAlmaTalep.Tarih, SatinAlmaTalep.OnayTarihi
    from MakinaBakimIERaporu(:ABaslangicTarihi, :ABitisTarihi, :AMakinaBakimTipiId, :AMakinaBakimBolumuId, :AMakinaGrupId, :AMakinaId,
          :AOrmeMakinaId, :AIplikMakinaId, :ADokumaMakinaId, :ABoya, :AOrme, :AIplik, :ADokuma, :AIEStartBaslangicTarihi, :AIEStartBitisTarihi, :ABStartBaslangicTarihi,
          :ABStartBitisTarihi, :ABEndBaslangicTarihi, :ABEndBitisTarihi, :AIEEndBaslangicTarihi, :AIEEndBitisTarihi, :ATahminiBaslangicTarihi,
          :ATahminiBitisTarihi, :ABaslayanlar, :ABakimBaslayanlar, :ABakimBitenler, :ABitenler, :AArizaIsEmri, :ASistemOlusturdu, :ABaslangicTahminiSaat,
          :ABitisTahminiSaat, :AKapalilariDok, :ABakimDairesiId, :AHizmetID, :AUrunID)
    left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = MakinaBakimIERaporu.SatinAlmaTalepKayitId
    into :MakinaBakimIsEmriId, :TahminiBaslangic, :Tolerans, :SatinAlmaTalepKayitId, :SatinAlmaTalepYil, :SatinAlmaTalepId, :SatinAlmaTalepIdStr,
          :TalepTarihi, :TalepOnayTarihi
          do
  begin

    SatinAlmaTalepIdStr = '';
    if (SatinAlmaTalepYil is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId)  || ' ';

    for select MakinaBakimIEUrun.UrunId, Urun.Adi, MakinaBakimIEUrun.Opsiyonel, sum(Miktar)
      from MakinaBakimIEUrun
      left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
      left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
      where MakinaBakimIEHizmet.MakinaBakimIsEmriId = :MakinaBakimIsEmriId and MakinaBakimIEHizmet.Kapali = 0 and
        (MakinaBakimIEHizmet.UrunID = :AHizmetID or :AHizmetID is null) and
        (MakinaBakimIEUrun.UrunID = :AUrunID or :AUrunID is null)
      group by MakinaBakimIEUrun.UrunId, Urun.Adi, MakinaBakimIEUrun.Opsiyonel
      into :UrunId, :UrunAdi, :Opsiyonel, :Miktar do
    begin
      OpsiyonelStr = '';
      if (Opsiyonel = 1) then OpsiyonelStr = 'Opsiyonel';
      else OpsiyonelStr = 'Zorunlu';

      if (Miktar is Null) then Miktar = 0;

      Select Sum(Devir) + Sum(Alis-AlisIadesi) - Sum(Satis-SatisIadesi) + Sum(Uretim - UretimeCikis) + Sum(FasonUretim + FasonUretimIade - FasonUretimeCikis) FROM DepoStok
      WHERE DepoStok.UrunID = :UrunId
      into :StokMevcudu;
      if (StokMevcudu is Null) then StokMevcudu = 0;

      select sum(Miktar)
      from MakinaBakimIEUrun
      left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
      left join MakinaBakimIsEmri on MakinaBakimIsEmri.MakinaBakimIsEmriId = MakinaBakimIEHizmet.MakinaBakimIsEmriId
      left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
      where MakinaBakimIsEmri.TahminiBaslangic <= :TahminiBaslangic and
        MakinaBakimIEHizmet.MakinaBakimIsEmriId <> :MakinaBakimIsEmriId and
        MakinaBakimIsEmri.Kapali = 0 and
        MakinaBakimIsEmri.BakimEndKullaniciId is Null and
        MakinaBakimIEHizmet.Kapali = 0 and
        MakinaBakimIEUrun.UrunId = :UrunId and
        MakinaBakimIEUrun.Opsiyonel = 0
      into :GecmisBakimZorunluIhtiyaci;
      if (GecmisBakimZorunluIhtiyaci is Null) then GecmisBakimZorunluIhtiyaci = 0;

      select sum(Miktar)
      from MakinaBakimIEUrun
      left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
      left join MakinaBakimIsEmri on MakinaBakimIsEmri.MakinaBakimIsEmriId = MakinaBakimIEHizmet.MakinaBakimIsEmriId
      left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
      where MakinaBakimIsEmri.TahminiBaslangic <= :TahminiBaslangic and
        MakinaBakimIEHizmet.MakinaBakimIsEmriId <> :MakinaBakimIsEmriId and
        MakinaBakimIsEmri.Kapali = 0 and
        MakinaBakimIsEmri.BakimEndKullaniciId is Null and
        MakinaBakimIEHizmet.Kapali = 0 and
        MakinaBakimIEUrun.UrunId = :UrunId and
        MakinaBakimIEUrun.Opsiyonel = 1
      into :GecmisBakimOpsiyonelIhtiyaci;
      if (GecmisBakimOpsiyonelIhtiyaci is Null) then GecmisBakimOpsiyonelIhtiyaci = 0;

      GununBakimIhtiyacToplami = Miktar + GecmisBakimZorunluIhtiyaci + GecmisBakimOpsiyonelIhtiyaci;

      select sum(SatinAlmaTalepDetay.Miktar) from SatinAlmaTalepDetay
        left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
        where SatinAlmaTalepDetay.SatinAlmaTalepKayitId = :SatinAlmaTalepKayitId and
          (SatinAlmaTalepDetay.UrunId is not Null and SatinAlmaTalepDetay.UrunId = :UrunId) and
          SatinAlmaTalepDetay.Kapali = 0
        into :DirektTalepMiktari;
      if (DirektTalepMiktari is Null) then DirektTalepMiktari = 0;

      select sum(SatinAlmaTalepDetay.Miktar) from SatinAlmaTalepDetay
        left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
        where (SatinAlmaTalepDetay.UrunId is not Null and SatinAlmaTalepDetay.UrunId = :UrunId) and
          SatinAlmaTalepDetay.Kapali = 0 and
          SatinAlmaTalep.OnaylayanKullaniciId is not Null
        into :ToplamTalepMiktari;
      if (ToplamTalepMiktari is Null) then ToplamTalepMiktari = 0;

      select sum(VerilenSiparisDetay.Miktar) from SatinAlmaTalepDetay
        left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
        left join VerilenSiparisDetay on VerilenSiparisDetay.SatinAlmaTalepDetayId = SatinAlmaTalepDetay.SatinAlmaTalepDetayId
        left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
        where SatinAlmaTalepDetay.SatinAlmaTalepKayitId = :SatinAlmaTalepKayitId and
          (VerilenSiparisDetay.UrunId is not Null and VerilenSiparisDetay.UrunId = :UrunId) and
          VerilenSiparisDetay.Kapali = 0 and
          VerilenSiparis.Kapali = 0
        into :DirektSiparisMiktari;
      if (DirektSiparisMiktari is Null) then DirektSiparisMiktari = 0;

      select sum(VerilenSiparisDetay.Miktar) from SatinAlmaTalepDetay
        left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
        left join VerilenSiparisDetay on VerilenSiparisDetay.SatinAlmaTalepDetayId = SatinAlmaTalepDetay.SatinAlmaTalepDetayId
        left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
        where (VerilenSiparisDetay.UrunId is not Null and VerilenSiparisDetay.UrunId = :UrunId) and
          VerilenSiparisDetay.Kapali = 0 and
          VerilenSiparis.Kapali = 0 and
          VerilenSiparis.OnaylayanKullaniciId is not Null
        into :ToplamSiparisMiktari;
      if (ToplamSiparisMiktari is Null) then ToplamSiparisMiktari = 0;

      select sum(VerilenSiparisDetay.Miktar  - (VerilenSiparisDetay.GelenMiktar - VerilenSiparisDetay.IadeEdilenMiktar)), sum(VerilenSiparisDetay.GelenMiktar - VerilenSiparisDetay.IadeEdilenMiktar) from SatinAlmaTalepDetay
        left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
        left join VerilenSiparisDetay on VerilenSiparisDetay.SatinAlmaTalepDetayId = SatinAlmaTalepDetay.SatinAlmaTalepDetayId
        left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
        where SatinAlmaTalepDetay.SatinAlmaTalepKayitId = :SatinAlmaTalepKayitId and
          (VerilenSiparisDetay.UrunId is not Null and VerilenSiparisDetay.UrunId = :UrunId) and
          VerilenSiparisDetay.Kapali = 0 and
          VerilenSiparis.Kapali = 0
        into :SiparisKalanMiktar, :SiparisGelenMiktar;
      if (SiparisKalanMiktar is Null) then SiparisKalanMiktar = 0;
      if (SiparisGelenMiktar is Null) then SiparisGelenMiktar = 0;
      suspend;
    end
  end
end
