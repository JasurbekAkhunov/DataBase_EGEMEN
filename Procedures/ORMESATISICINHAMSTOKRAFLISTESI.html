CREATE PROCEDURE ORMESATISICINHAMSTOKRAFLISTESI(
    AHAMID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    AORMESIPARISDETAYID INTEGER,
    ABOYASIPARISDETAYID INTEGER,
    ABOYASIPARISBASKIID INTEGER,
    ARENKID INTEGER,
    ADESENID VARCHAR(20),
    ADESENVARYANTID INTEGER,
    ADEPOID VARCHAR(10),
    AFIRMAID FIRMAKODUSIZE)
RETURNS (
    ORMERAFID INTEGER,
    RAFID VARCHAR(20),
    STOKGIRISNO INTEGER,
    RENKADI VARCHAR(50),
    URETIMPARTINO VARCHAR(30),
    CIKISPARTINO VARCHAR(30),
    KALANMIKTAR NUMERIC(12,3),
    BRUTKALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI INTEGER,
    LOT VARCHAR(150),
    KALITENO INTEGER,
    MELANJKODU VARCHAR(20),
    KALITEACIKLAMATIPIADI VARCHAR(20),
    MAMULPROSESADI VARCHAR(255),
    URETICIADI VARCHAR(40),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    ATKILOTU VARCHAR(30),
    OLCUID INTEGER,
    OLCUADI VARCHAR(40),
    KALANADET INTEGER,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20))
AS
declare variable SiparisFirmaId       varchar(20);
declare variable StokFirmaId          varchar(20);
declare variable OrmeFirmaKodumuz     varchar(20);
declare variable DokumaFirmaKodumuz   varchar(20);
declare variable FFirmaKodumuz        varchar(20);
declare variable assSatis             integer;
declare variable assFasonArtiSatis    integer;
declare variable assFason             integer;
declare variable OrmeSiparisSekliId   integer;
declare variable KtBoyaSiparisKayitId integer;
declare variable DokumaSiparisi       integer;
declare variable FBoyaFirmaKodumuz    varchar(20);
declare variable KS                   integer;
begin
  Select assSatis, assFasonArtiSatis, assFason from OrmeSiparisSekliGetir into :assSatis, :assFasonArtiSatis, :assFason;

  Select OrmeFirmaKodumuz, DokumaFirmaKodumuz, FirmaKodumuz from sistem where sistemid = 1 into :OrmeFirmaKodumuz, :DokumaFirmaKodumuz, :FBoyaFirmaKodumuz;

  FFirmaKodumuz = OrmeFirmaKodumuz;

  if (ABoyaSiparisKayitId is not null) then
  begin
    Select BoyaSiparis.FirmaId, BoyaSiparis.OrmeSiparisSekliId, BoyaSiparis.DokumaSiparisi from BoyaSiparis
      where BoyaSiparis.BoyaSiparisKayitId = :ABoyaSiparisKayitId
      into :SiparisFirmaId, :OrmeSiparisSekliId, :DokumaSiparisi;

      if (DokumaSiparisi = 1) then FFirmaKodumuz = DokumaFirmaKodumuz;
      else FFirmaKodumuz = OrmeFirmaKodumuz;

      if (OrmeSiparisSekliId = assSatis) then SiparisFirmaId = Null;
      else FFirmaKodumuz = Null;
  end
  else
  begin
    SiparisFirmaId = AFirmaId;
  end

  if (AOrmeSiparisDetayId is not null) then
    Select OrmeSiparisDetay.KTBoyaSiparisKayitID from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayID = :AOrmeSiparisDetayId into :KTBoyaSiparisKayitID;


  For Select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.RafId, Renk.Adi, OrmeRaf.UretimPartiNo, OrmeRaf.CikisPartiNo, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
             OrmeRaf.BrutGirisMiktari - OrmeRaf.BrutCikisMiktari,
             OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi,  OrmeRaf.Lot, OrmeRaf.KaliteNo, Uretici.Adi,
             OrmeRaf.MelanjKodu, KaliteAciklamaTipi.Adi, MamulProses.Adi, OrmeRaf.OrmeMakinaId, OrmeMakina.Adi, OrmeRaf.DokumaMakinaId, DokumaMakina.Adi, OrmeRaf.AtkiLotu,
             OrmeRaf.OlcuId, Olcu.OlcuAdi, coalesce(OrmeRaf.GirisAdet,0) - coalesce(OrmeRaf.CikisAdet,0), OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4 from OrmeRaf
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = OrmeRaf.BoyaSiparisKayitId
    Left Join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = OrmeRaf.KaliteAciklamaTipiId
    Left Join MamulProses on MamulProses.MamulProsesId = OrmeRaf.MamulProsesId
    Left Join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
    Left Join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmeRaf.OrmeMakinaId
    Left Join DokumaMakina on DokumaMakina.DokumaMakinaId = OrmeRaf.DokumaMakinaId
    left join Olcu on Olcu.OlcuId = OrmeRaf.OlcuId
    left join OrmePlanlama On OrmePlanlama.OrmePlanlamaId = OrmeRaf.OrmePlanlamaId
    where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and OrmeRaf.BoyahaneFasonIade = 0 and
         ( (OrmeRaf.StokFirmaId = :SiparisFirmaId and OrmeRaf.FasonStok = 1 ) or
           (OrmeRaf.StokFirmaId = :FFirmaKodumuz and OrmeRaf.FasonStok = 0) ) and
         ( ( (OrmeRaf.OrmeSiparisDetayID = :AOrmeSiparisDetayID or OrmeRaf.OrmeSiparisDetayID is null) and
             (OrmeRaf.BoyaSiparisDetayID = :ABoyaSiparisDetayID or OrmeRaf.BoyaSiparisDetayID is null) and
             (OrmeRaf.BoyaSiparisBaskiID = :ABoyaSiparisBaskiID or OrmeRaf.BoyaSiparisBaskiID is null) ) or
             (OrmeRaf.OrmeSiparisDetayId is not null and BoyaSiparis.FirmaID = :FFirmaKodumuz) or
             (BoyaSiparis.BaskaFirmayaSatilabilir = 1 and BoyaSiparis.FirmaID = :FBoyaFirmaKodumuz) ) and
        ( IntKarsilastir(OrmeRaf.RenkId, :ARenkId) = 1 or OrmeRaf.BoyaSiparisDetayID = :ABoyaSiparisDetayID ) and
         (StrKarsilastir(OrmeRaf.DesenId, :ADesenId) = 1) and
         (IntKarsilastir(OrmeRaf.DesenVaryantId, :ADesenVaryantId) = 1) and
         (DepoRaf.DepoId = :ADepoID or :ADepoID is null)
    into :OrmeRafId, :StokGirisNo, :RafId, :RenkAdi, :UretimPartiNo, :CikisPartiNo, :KalanMiktar, :BrutKalanMiktar, :KalanKapAdedi, :Lot, :KaliteNo, :UreticiAdi,
         :MelanjKodu,:KaliteAciklamaTipiAdi, :MamulProsesAdi, :OrmeMakinaId, :OrmeMakinaAdi, :DokumaMakinaId, :DokumaMakinaAdi, :AtkiLotu,
         :OlcuId, :OlcuAdi, :KalanAdet, :IplikBoyu1, IplikBoyu2, :Iplikboyu3, :IplikBoyu4 do
  begin
    /*top top kayit varsa raf sectirmiyoruz, top top isleme izin verecegiz*/
    select count(*) from OrmeRafDetay
    left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketId
    left join GirisKaliteKontrol on GirisKaliteKontrol.StokHareketId = StokHareket.StokHareketId
    left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketId = StokHareket.StokHareketId
    where OrmeRafDetay.OrmeRafId = :OrmeRafId and
    StokHareket.DokumaHareketi = 1 and
    (GirisKaliteKontrol.GirisKaliteKontrolId is not null or DokumaKaliteKontrol.DokumaKaliteKontrolId is not null) into :KS;

    if (KS = 0) then
      suspend;
  end
end
