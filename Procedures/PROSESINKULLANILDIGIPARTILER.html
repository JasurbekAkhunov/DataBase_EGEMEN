CREATE PROCEDURE PROSESINKULLANILDIGIPARTILER(
    APROSESID INTEGER,
    AISLETMEDE INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    HAMADITOPLAMI VARCHAR(255),
    DURUMUSTR VARCHAR(20),
    KILO NUMERIC(12,3))
AS
begin
  for select PartiStok.PartiSinifId, PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.HamAdiToplami, PartiDurum.Adi,
        PartiStok.Kilo from PartiStok
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
  left join Proses on Proses.ProsesId = PartiStok.ProsesId
  where PartiStok.ProsesId = :AProsesId and
       ( (PartiStok.Durumu >= 7 and :AIsletmede = 0) or
         (PartiStok.Durumu < 7 and :AIsletmede = 1) or :AIsletmede is null)
  order by Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId
  into :PartiSinifId, :PartiKayitId, :PartiYil, :PartiId, :HamAdiToplami, :DurumuStr, :Kilo do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    suspend;
  end
end
