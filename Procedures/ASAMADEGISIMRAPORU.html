CREATE PROCEDURE ASAMADEGISIMRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AASAMAID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    MAKINAID VARCHAR(10),
    ASAMADEGISTIRENPERSONELADI VARCHAR(40),
    ASAMADEGISIMSEBEBIID INTEGER,
    ASAMADEGISIMSEBEBIADI VARCHAR(15),
    ASAMADEGISIMTARIHI TIMESTAMP,
    PARTIRECETEID VARCHAR(20),
    PARTIRECETEADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3))
AS
begin
  For Select PartiAsamalari.PartiKayitId, PartiAsamaStok.PartiSinifId, PartiAsamalari.PartiAsamalariId,
             Parti.PartiYil, Parti.PartiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId,
             PartiStok.HamAdiToplami, PartiAsamaStokDetay.MakinaId, Personel.Adi, PartiAsamalari.AsamaDegisimSebebiID, AsamaDegisimSebebi.Adi, PartiAsamalari.AsamaDegisimTarihi,
             Parti.ReceteId, Recete.Adi, Parti.FirmaId, Firma.Adi, PartiAsamaStok.AsamaId, Asama.Adi, PartiStok.Kilo, PartiStok.Metre from PartiAsamalari
  Left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
  Left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
  Left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
  Left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
  left join Asama on Asama.Asamaid = PartiAsamaStok.Asamaid
  Left join Firma on Firma.FirmaId = Parti.FirmaId
  Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  Left join Recete on Recete.ReceteId = Parti.ReceteId
  left join Personel on Personel.PersonelId = PartiAsamalari.AsamaDegistirenPersonelID
  left join AsamaDegisimSebebi on AsamaDegisimSebebi.AsamaDegisimSebebiId = PartiAsamalari.AsamaDegisimSebebiID
  where PartiAsamalari.AsamaDegisimTarihi >= :ABaslangicTarihi and PartiAsamalari.AsamaDegisimTarihi <= :ABitisTarihi and
       (PartiAsamalari.AsamaId = :AAsamaId or :AAsamaId is null) and PartiAsamaStok.AsamayaDahil = 1
  into :PartiKayitId, :PartiSinifId, :PartiAsamalariId, :PartiYil, :PartiId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
       :HamAdiToplami, :MakinaId, :AsamaDegistirenPersonelAdi, :AsamaDegisimSebebiID, :AsamaDegisimSebebiAdi, :AsamaDegisimTarihi, :PartiReceteId, :PartiReceteAdi,
       :FirmaId, :FirmaAdi, :AsamaId, :AsamaAdi, :Kilo, :Metre do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    suspend;
  end
end
