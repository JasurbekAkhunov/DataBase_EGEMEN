CREATE PROCEDURE TOPLARAGORESINIFPARCALA(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    APSKALITEKONTROLIDLERI BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    AKULLANICIID INTEGER,
    APSKALITEKONTROLASAMAID INTEGER)
AS
declare variable HamTipiId  integer;
declare variable TopSayisi  integer;
declare variable RetTopSayisi  integer;
declare variable Kilo       Numeric(12,3);
declare variable Metre      Numeric(12,3);
declare variable RetKilo    Numeric(12,3);
declare variable RetMetre   Numeric(12,3);
declare variable ToplamKilo    Numeric(12,3);
declare variable ToplamMetre   Numeric(12,3);
declare variable AMiktar    Numeric(12,3);
declare variable AMiktar2   Numeric(12,3);
declare variable ProxyId   varchar(8);
declare variable YeniPartiSinifID  integer;
declare variable PartiYil integer;
declare variable PartiId integer;
declare variable FirmaAdi varchar(100);
declare variable FBlob             Blob;
declare variable Aciklama          varchar(1024);
declare variable Konu              varchar(50);
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable FPSKaliteKontrolAsamaId integer;
declare variable PartiBarkodTakibi integer;
declare variable OtonomTopParcalayabilir EgeBoolean;
declare variable OtonomTopParcalaniyor EgeBoolean;
declare variable PSKKKS      integer;
declare variable TartiSevkKS integer;
declare variable FPartiStokTartiId integer;
declare variable FMamulDepoTransferFisiId integer;
declare variable KayitSayisi integer;
begin
  Select DokumaKumasiKilodanTakipEt, PartiBarkodTakibi from Sistem into :DokumaKumasiKilodanTakipEt, :PartiBarkodTakibi;
  Select OtonomTopParcalayabilir from Kullanici where Kullanici.KullaniciID = :AKullaniciId into :OtonomTopParcalayabilir;

  OtonomTopParcalaniyor = 0;
  if (OtonomTopParcalayabilir = 1) then
  begin
    Select Count(PSKaliteKontrol.PSKaliteKontrolID), Count(PartiStokTarti.BoyaIrsaliyeDetayId)+ Count(PartiStokTarti.HamStokSarfFisiDetayId) from PSKaliteKontrol
     Left Join PartiStokTarti on PartiStokTarti.PartiStokTartiID = PSKaliteKontrol.Partistoktartiid
     Where :APsKaliteKontrolIdleri like '%*'||PSKaliteKontrol.PsKaliteKontrolId||'*%' and PsKaliteKontrol.PsKaliteKontrolAsamaid = :APSKALITEKONTROLASAMAID into :PSKKKS, :TartiSevkKS;
    if (PSKKKS > 0 and TartiSevkKS = 0) then OtonomTopParcalaniyor = 1;
  end

  Select Parti.PartiYil, Parti.PartiId, Firma.Adi, PartiStok.HamTipiId, PartiStok.ProxyId, PartiStok.Kilo, PartiStok.Metre from PartiStok
  left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
  left join Firma on Firma.FirmaID = Parti.FirmaID
  where PartiStok.PartiSinifID = :APartiSinifID
  into :PartiYil, :PartiId, :FirmaAdi, :HamTipiId, :ProxyId, :Kilo, :Metre;

  Select Sum(Kilo), Sum(Metre), Count(*) from PSKaliteKontrol where :APsKaliteKontrolIdleri like '%*'||PSKaliteKontrol.PsKaliteKontrolId||'*%'
    and PsKaliteKontrol.PsKaliteKontrolAsamaid = :APSKALITEKONTROLASAMAID
      into :ToplamKilo, :ToplamMetre, :TopSayisi;

  if (TopSayisi = 0) then exit;

  if (HamTipiID <> 2 or DokumaKumasiKilodanTakipEt = 1) then
  begin
    AMiktar  = ToplamKilo;
    AMiktar2 = ToplamMetre;
  end
  else
  begin
    AMiktar  = ToplamMetre;
    AMiktar2 = ToplamKilo;
  end

  if (OtonomTopParcalaniyor = 1) then
  begin
    For Select PSKaliteKontrol.PartiStokTartiId from PSKaliteKontrol
      Where :APsKaliteKontrolIdleri like '%*'||PSKaliteKontrol.PsKaliteKontrolId||'*%' and PsKaliteKontrol.PsKaliteKontrolAsamaid = :APSKALITEKONTROLASAMAID
    into :FPartiStokTartiId do
    begin
      For Select MamulDepoTransferFisiDetay.MamulDepoTransferFisiId from MamulDepoTransferFisiDetay
        where PartiStokTartiID = :FPartiStokTartiId and MamulDepoTransferFisiDetay.Oncekirafid is Not Null Order By MamulDepoTransferFisiDetay.InsertTarihi Desc into :FMamulDepoTransferFisiId do
      begin
        Delete From MamulDepoTransferFisiDetay where PartiStokTartiID = :FPartiStokTartiId and MamulDepoTransferFisiId = :FMamulDepoTransferFisiId;

        Select Count(*) from MamulDepoTransferFisiDetay where MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = :FMamulDepoTransferFisiId into :KayitSayisi;

        if (KayitSayisi = 0) then
          Delete from MamulDepoTransferFisi where MamulDepoTransferFisi.MamulDepoTransferFisiId = :FMamulDepoTransferFisiId;
      end
      rdb$set_context('USER_TRANSACTION', 'OtonomTopParcalaniyor', 1);
      Delete From PartiStokTarti where PartiStokTarti.PartiStokTartiID = :FPartiStokTartiId;
    end
  end

  execute procedure PartininSinifiniBol(:APartiKayitId, :APartiSinifId, :AMiktar, :AKullaniciId, :APsKaliteKontrolAsamaId, :AMiktar2, Null, Null, :OtonomTopParcalaniyor);

  Select Max(PartiStok.PartiSinifID) from PartiStok where PartiStok.PartiKayitID = :APartiKayitID into :YeniPartiSinifId;
  Select PSKaliteKontrolAsamaId from PSKaliteKontrolAsama where PSKaliteKontrolAsama.PartiSinifId = :YeniPartiSinifID into :FPSKaliteKontrolAsamaId;

  /*if (PartiBarkodTakibi = 1) then  --can kapattırdı
  begin
    Update PartiStok set ProxyID = Null where PartiStok.partiSinifID = :APartiSinifID;
    Update PartiStok set ProxyID = :ProxyId where PartiStok.PartiSinifID = :YeniPartiSinifID;
  end */

  Update PSKaliteKontrol set PartiSinifID = :YeniPartiSinifID,
  PSKaliteKontrolAsamaId = :FPSKaliteKontrolAsamaId,
  LinkIslemi = 1 where :APsKaliteKontrolIdleri like '%*'||PSKaliteKontrol.PsKaliteKontrolId||'*%' ;
end
