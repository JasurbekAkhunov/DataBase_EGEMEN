CREATE PROCEDURE PSKALITEKONTROLKOPYALA(
    APSKALITEKONTROLID INTEGER,
    AMIKTAR INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable Counter integer = 0;
begin
  if (AMiktar = 0) then exit;
  if (APsKaliteKontrolId is null or APsKaliteKontrolId = 0) then exit;

  while ( Counter < AMiktar ) do
  begin
    insert into PsKaliteKontrol (PsKaliteKontrolId, PsKaliteKontrolAsamaid, PaketlemeId, Tarih, PartiKayitId, PartiSinifId, MakinaId, PersonelId,
           KaliteNo, Kilo, Dara, Metre, BrutMetre, En, Gramaj, Donme, Egrilik, BaslangicEn, OrtaEn, BitisEn,
           OrtalamaEn, Aciklama, PartiStokTartiId, HataPuani, PuanliSistem, Parcayarim, PaketlemeHataPuani,
           InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi, ParcaSayisi, PartiRenkLotNo, ReddenBolunenYeniPartiSinifId, MusteriAciklama, RaporBoyu, OlcuId)
    select gen_ID(PsKaliteKontrol_Id_Gen, 1), PsKaliteKontrolAsamaid, PaketlemeId, current_timestamp(0), PartiKayitId, PartiSinifId, MakinaId, PersonelId,
           KaliteNo, Kilo, Dara, Metre, BrutMetre, En, Gramaj, Donme, Egrilik, BaslangicEn, OrtaEn, BitisEn,
           OrtalamaEn, Aciklama, PartiStokTartiId, HataPuani, PuanliSistem, Parcayarim, PaketlemeHataPuani,
           :AKullaniciId, current_timestamp(0), :AKullaniciId, current_timestamp(0), ParcaSayisi, PartiRenkLotNo, ReddenBolunenYeniPartiSinifId, MusteriAciklama, RaporBoyu,
           OlcuId
    from PsKaliteKontrol where PsKaliteKontrol.PsKaliteKontrolId = :APsKaliteKontrolId;
    Counter = Counter + 1;
  end
end
