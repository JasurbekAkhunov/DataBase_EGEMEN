        Gorus2 = Null;
        Gorus3 = Null;

          where DisTamirGorus.HamStokMasterId = :HamStokMasterId
          order by DisTamirGorus.InsertKullaniciId
          into :FNotu do
        begin
          SY = SY + 1;
          if (Sy = 1) then Gorus1 = FNotu;
          else if (Sy = 2) then Gorus2 = FNotu;

        if (FKullanilanMiktar is null) then FKullanilanMiktar = 0;

        FIrsaliyeKalanMiktar = Miktar +  FIptalPartiMiktar - FKullanilanMiktar;

        SevkEdildi = 0;

        if (FIrsaliyeKalanMiktar > 0) then
          Stokvar = 1;
        else
          StokVar = 0;

        if (StokVar = 0) then
        begin
          MinDurumu = 0;

          select Min(PartiStok.Durumu) from HamStokDetay
            left join RafDetay on RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId
            left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
            left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
            left join Parti on Parti.PartiKayitId= PartiStok.PartiKayitId
            where HamStokDetay.HamStokMasterID = :HamStokMasterID and PartiStok.PartiSinifId is not null and PartiStok.partiStokIptalKayitId is null
            into :MinDurumu;

          if (MinDurumu > 7) then SevkEdildi = 1;
        end

        if (SevkEdildi = 1) then SevkEdildiStr = 'Sevk Edildi';
        else SevkEdildiStr = '';

        suspend;
      end
    end
  end
  else if (ARaporTipi = 1) then
  begin
    for select DisTamir.HamStokMasterId, DisTamir.DisTamirYil, DisTamir.DisTamirNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
               HamStokIrsaliye.FirmaId, IrsFirma.Adi, HamStokMaster.HamId, Ham.Adi, Ham.HamTipiId, HamStokMaster.Renk, HamStokMaster.Denye, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Miktar,

        FKullanilanMiktar = Null;

        select Sum(rafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.SatisMiktar) from HamStokDetay
        left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayID
        where HamStokDetay.HamStokMasterId = :HamStokMasterId
        into :FKullanilanMiktar;

        Select Sum(RafDetay.Miktar) from HamStokDetay
           left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
           where HamStokDetay.HamStokMasterId = :HamStokMasterId
           into :FIptalPartiMiktar;

        if (FIptalPartiMiktar is null) then FIptalPartiMiktar = 0;

        if (FKullanilanMiktar is null) then FKullanilanMiktar = 0;

        if (FIrsaliyeKalanMiktar > 0) then
          Stokvar = 1;
        else
          StokVar = 0;

        if (StokVar = 0) then
        begin
          MinDurumu = 0;

          select Min(PartiStok.Durumu) from HamStokDetay
            left join RafDetay on RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId
            left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId

        else SevkEdildiStr = '';

        IslemeAlinacakOkey = 0;
        DisTamirCozumSekliStr = '';

        For Select DisTamirCozum.DisTamirCozumSekliId, DisTamirCozumSekli.Adi, DisTamirCozumSekli.IslemeAlinacak from DisTamirCozum
          left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliID = DisTamirCozum.DisTamirCozumSekliID
          where DisTamirCozum.HamStokMasterID = :HamStokMasterID
          group by DisTamirCozum.DisTamirCozumSekliId, DisTamirCozumSekli.Adi, DisTamirCozumSekli.IslemeAlinacak
          into :FDisTamirCozumSekliId, :FDisTamirCozumSekliAdi, :FIslemeAlinacak do

          if (AIslemeAlinacak is not null and AIslemeAlinacak = FIslemeAlinacak) then IslemeAlinacakOkey = 1;

          begin
           if (DisTamirCozumSekliStr <> '' ) then DisTamirCozumSekliStr = DisTamirCozumSekliStr || cast(', ' as varchar(2) character set UTF8);
           DisTamirCozumSekliStr = DisTamirCozumSekliStr || FDisTamirCozumSekliAdi;
          end
        end

        if ( (ADisTamirCozumSekliID is null or DisTamirCozumSekliOkey = 1) and
             (AIslemeAlinacak is null or IslemeAlinacakOkey = 1) ) then suspend;
      end
    end
  end

  begin
    for select DisTamir.HamStokMasterId, DisTamir.DisTamirYil, DisTamir.DisTamirNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
               HamStokIrsaliye.FirmaId, IrsFirma.Adi, HamStokMaster.HamId, Ham.Adi, Ham.HamTipiId, HamStokMaster.Renk, HamStokMaster.Denye, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Miktar,
               DisTamir.TamirTipiId, TamirTipi.Adi, HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartino,
               DisTamir.MusteriBeyani, DisTamir.OnleyiciFaaliyeti, DisTamir.BeyanTarihi, DisTamir.TamirDegerlendirmeBitti, DisTamir.TDBKullaniciId, TDBK.Adi,

               DisTamir.SebepAsamaId, Asama.Adi, Asama.TamirRaporSiraNo, DisTamir.TamirDuzeltmeTipiId, TamirDuzeltmeTipi.Adi, HamStokMaster.InsertTarihi, HamStokMaster.Fason, EskiBoyaSiparis.SiparisGrupID, SiparisGrup.Adi,
               EskiBoyaSiparis.SiparisGrup2ID, SiparisGrup2.Adi, EskiBoyaSiparis.SiparisGrup3ID, SiparisGrup3.Adi, EskiBoyaSiparis.SiparisGrup4ID, SiparisGrup4.Adi from DisTamir
    left join HamStokMaster on HamStokMaster.HamStokMasterId = DisTamir.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
    left join Firma ReferansFirma on ReferansFirma.FirmaId = HamStokMaster.ReferansFirmaId
    left join Ham on Ham.HamId = HamStokMaster.HamId
    left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
    left join Firma IrsFirma on IrsFirma.Firmaid = HamstokIrsaliye.Firmaid
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = HamStokMaster.BoyaSiparisKayitId
    left join Firma SiparisFirma on SiparisFirma.Firmaid = BoyaSiparis.Firmaid
    Left Join TamirDuzeltmeTipi on TamirDuzeltmeTipi.TamirDuzeltmeTipiId = DisTamir.TamirDuzeltmeTipiId
    left join Asama on Asama.Asamaid = DisTamir.SebepAsamaid
    Left Join Kullanici TDBK on TDBK.KullaniciId = DisTamir.TDBKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = DisTamir.KullaniciId
    left join Parti EskiParti on EskiParti.PartiYil = HamStokMaster.EskiPartiYil and EskiParti.PartiId = HamStokMaster.EskiPartiNo
    left join BoyaSiparis EskiBoyaSiparis on EskiBoyaSiparis.BoyaSiparisKayitId = EskiParti.BoyaSiparisKayitId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = EskiBoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = EskiBoyaSiparis.SiparisGrup2Id
    left join SiparisGrup SiparisGrup3 on SiparisGrup3.SiparisGrupId = EskiBoyaSiparis.SiparisGrup3Id
    left join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = EskiBoyaSiparis.SiparisGrup4Id
    where HamStokMaster.PartiSinifId is null and DisTamir.BeyanTarihi >= :ABaslangicBeyanTarihi and DisTamir.BeyanTarihi <= :ABitisBeyanTarihi and
         ( (Ham.HamTipiId = 0 and :AHamTipi = 0) or (Ham.HamTipiId <> 0 and :AHamTipi = 1)  or :AHamTipi is null) and

         (HamStokMaster.HamId = :AHamId or :AHamId is null) and
         (DisTamir.TamirDuzeltmeTipiId = :ATamirDuzeltmeTipiId or :ATamirDuzeltmeTipiId is null) and
         (DisTamir.TamirTipiId = :ATamirTipiId or :ATamirTipiId is null) and
         (TamirTipi.TamirTipiGrupId = :ATamirTipiGrupId or :ATamirTipiGrupId is null) and
         (HamStokMaster.Fason = :ASatisTipi or :ASatisTipi is null)

         :ReferansFirmaId, :ReferansFirmaAdi, :SiparisFirmaId, :SiparisFirmaAdi, :KullaniciId, :KullaniciAdi,
         :DegisimTarihi, :SebepAsamaId, :SebepAsamaAdi, :AsamaTamirRaporSiraNo, :TamirDuzeltmeTipiId, :TamirDuzeltmeTipiAdi, :IlkKayitTarihi, :Fason, :BoyaSiparisGrupId,

      if (ReferansFirmaAdi is null) then ReferansFirmaAdi = SiparisFirmaAdi;

      else FasonStr = '';

          group by DisTamirCozum.DisTamirCozumSekliId, DisTamirCozumSekli.Adi, DisTamirCozumSekli.IslemeAlinacak
          into :FDisTamirCozumSekliId, :FDisTamirCozumSekliAdi, :FIslemeAlinacak do
        begin
          if (ADisTamirCozumSekliID is not null and ADisTamirCozumSekliId = FDisTamirCozumSekliID) then DisTamirCozumSekliOkey = 1;
          if (AIslemeAlinacak is not null and AIslemeAlinacak = FIslemeAlinacak) then IslemeAlinacakOkey = 1;

          if (Char_Length(DisTamirCozumSekliStr) + Char_Length(FDisTamirCozumSekliAdi) < 1000) then

        if ( (ADisTamirCozumSekliID is null or DisTamirCozumSekliOkey = 1) and
             (AIslemeAlinacak is null or IslemeAlinacakOkey = 1) ) then suspend;
      end
    end
