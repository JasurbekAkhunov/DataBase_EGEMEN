CREATE PROCEDURE KARTELAFISIDETAYKAYDETGUNCELLE(
    AKARTELAFISIDETAYID INTEGER,
    AKARTELAFISIID INTEGER,
    AKALITEID VARCHAR(20),
    AMIKTAR DOUBLE PRECISION,
    AACIKLAMA VARCHAR(255),
    ABIRIMFIYAT DOUBLE PRECISION,
    ADOVIZID DOVIZSIZE,
    AKULLANICIID INTEGER,
    AMETRE DOUBLE PRECISION)
RETURNS (
    DURUMMSG VARCHAR(50),
    KARTELAFISIDETAYID INTEGER)
AS
begin
  if (AKartelaFisiDetayId = -1) then AKartelaFisiDetayId = Null;
  KartelaFisiDetayId = :AKartelaFisiDetayId;

  if (AKartelaFisiDetayId is Null) then
  begin
    KartelaFisiDetayId = gen_ID(KartelaFisiDetay_Id_gen, 1);

    Insert Into KartelaFisidetay (Kartelafisidetayid, Kartelafisiid, Kaliteid, Miktar, Aciklama, Giris, Insertkullaniciid, Kullaniciid, Birimfiyat, DovizId, Metre)
                          Values (:KartelaFisiDetayId, :AKartelaFisiId, :AKaliteid, :AMiktar, :AAciklama, -1, :AKullaniciID, :AKullaniciID, :ABirimfiyat, :ADovizId, :AMetre);
  end
  else
  begin
    Update KartelaFisiDetay set Kaliteid = :AKaliteid, Miktar = :AMiktar, Aciklama = :AAciklama, Kullaniciid = :AKullaniciid, Birimfiyat = :ABirimfiyat, DovizId = :ADovizId, Metre = :AMetre Where Kartelafisidetayid = :AKartelafisidetayid;
  end

  DurumMsg = 'OK';
  Suspend;
end
