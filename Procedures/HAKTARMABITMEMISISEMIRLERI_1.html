CREATE PROCEDURE HAKTARMABITMEMISISEMIRLERI_1(
    ARAPORTIPI INTEGER)
RETURNS (
    HAKTARMAISEMRIKAYITID INTEGER,
    HAKTARMAISEMRIYIL INTEGER,
    HAKTARMAISEMRIID INTEGER,
    HAKTARMAISEMRIIDSTR VARCHAR(30),
    MAKINAID VARCHAR(10),
    PLANLAMATARIHI TIMESTAMP,
    BASLATANPERSONELID INTEGER,
    BASLATANPERSONELADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENPERSONELID INTEGER,
    BITIRENPERSONELADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select HAktarmaIsEmri.HAktarmaIsEmriKayitId, HAktarmaIsEmri.HAktarmaIsEmriYil, HAktarmaIsEmri.HAktarmaIsEmriId, HAktarmaIsEmri.MakinaId,
             HAktarmaIsEmri.PlanlamaTarihi, HAktarmaIsEmri.BaslatanPersonelId, BasP.Adi, HAktarmaIsEmri.BaslamaTarihi, HAktarmaIsEmri.BitirenPersonelId,
             BitP.Adi, HAktarmaIsEmri.BitisTarihi, HAktarmaIsEmri.Notu, Ik.Adi, HAktarmaIsEmri.InsertTarihi, Sk.Adi, HAktarmaIsEmri.DegisimTarihi from HAktarmaIsEmri
  left join Personel BasP on BasP.PersonelId = HAktarmaIsEmri.BaslatanPersonelId
  left join Personel BitP on BitP.PersonelId = HAktarmaIsEmri.BitirenPersonelId
  Left Join Kullanici IK on IK.KullaniciId = HAktarmaIsEmri.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HAktarmaIsEmri.KullaniciId
  where HAktarmaIsEmri.BitisTarihi is null and
        ( (HAktarmaIsEmri.Baslamatarihi is null and :ARaporTipi = 0) or
          (HAktarmaIsEmri.Baslamatarihi is not null and :ARaporTipi = 1) or :ARaporTipi is null)
  order by HAktarmaIsEmri.HAktarmaIsEmriYil, HAktarmaIsEmri.HAktarmaIsEmriId
  into :HAktarmaIsEmriKayitId, :HAktarmaIsEmriYil, :HAktarmaIsEmriId, :MakinaId, :PlanlamaTarihi, :BaslatanPersonelId, :BaslatanPersonelAdi,
       :BaslamaTarihi, :BitirenPersonelId, :BitirenPersonelAdi, :BitisTarihi, :Notu,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    HAktarmaIsEmriIdStr = '(' || EgemenIntegerToStr(HAktarmaIsEmriYil) || ') ' || EgemenIntegerToStr(HAktarmaIsEmriId) || ' ';
    suspend;
  end
end
