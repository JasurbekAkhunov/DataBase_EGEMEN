CREATE PROCEDURE MASRAFMERKEZIURUNKATEGORIGETIR
RETURNS (
    KAYITVAR INTEGER,
    MASRAFMERKEZIID INTEGER,
    MASRAFMERKEZIADI VARCHAR(30),
    URUNKATEGORIID VARCHAR(4),
    URUNKATEGORIADI VARCHAR(30),
    MUHASEBEHESAPKODU VARCHAR(30),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select MasrafMerkezi.MasrafMerkeziId, MasrafMerkezi.Adi, Urunkategori.UrunkategoriId, Urunkategori.Adi, MasrafMerkeziUrunKategori.MuhasebeHesapKodu,
             IK.Adi, MasrafMerkeziUrunKategori.InsertTarihi, SK.Adi, MasrafMerkeziUrunKategori.DegisimTarihi from MasrafMerkezi
  left join Urunkategori on Urunkategori.Urunkategoriid is not null
  left join MasrafMerkeziUrunKategori on MasrafMerkeziUrunKategori.MasrafMerkeziID = MasrafMerkezi.MasrafMerkeziId and MasrafMerkeziUrunKategori.UrunkategoriID = Urunkategori.UrunkategoriId
  left join Kullanici IK on IK.KullaniciId = MasrafMerkeziUrunKategori.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId = MasrafMerkeziUrunKategori.KullaniciId
  order by MasrafMerkezi.Adi, Urunkategori.UrunkategoriId
  into :MasrafMerkeziId, :MasrafMerkeziAdi, :UrunkategoriId, :UrunkategoriAdi, :MuhasebeHesapKodu,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    if (MuhasebeHesapKodu is not null) then KayitVar = 1;
    else KayitVar = 0;
    suspend;
  end
end
