    CIKISTAMETREZORUNLU INTEGER,
    CIKISTAARABAZORUNLU INTEGER,
    FASAMAID INTEGER)
AS
declare variable FIlaveIslemlerAdi      varchar(30);
declare variable Virgul                 integer;

declare variable CikisYok       integer;
declare variable FHamAdiToplami varchar(255);
declare variable FGirisTarihi timestamp;
declare variable FCikisTarihi timestamp;
declare variable FPASCikisTarihi   timestamp;
declare variable FGirisPersonelAdi varchar(40);
declare variable FCikisPersonelAdi varchar(40);
declare variable FMakinaId         varchar(10);
declare variable FMakinaAdi        varchar(40);
declare variable FGirisKilo        Numeric(12,3);
declare variable FGirisMetre       Numeric(12,3);
declare variable FKilo             Numeric(12,3);
declare variable FMetre            Numeric(12,3);
declare variable FCikisKilo        Numeric(12,3);
declare variable FCikisMetre       Numeric(12,3);
declare variable FHamTipiId        integer;

declare variable FOrnekTarihi           TimeStamp;
declare variable FOrnekKullaniciAdi varchar(40);
declare variable FOrnekOnayTarihi   TimeStamp;
declare variable FOrnekOnaylayanKullaniciAdi varchar(40);
declare variable FAsamaAciklamaTipiAdi  varchar(50);
Declare variable FReceteIsEmriYil integer;
Declare variable FReceteIsEmriId  integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable FPartiSinifId          integer;

begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  for SELECT PartiKayitId, PartiSinifId FROM BOYAIRSALIYELISTESI
    (:ABaslangicTarihi, :ABitisTarihi, :ASatisTipi, :ARaporTipi, :AHamId, :AFirmaId,
    :AFirmaYetkiliId, :ASevkFirmaId, :ASiparisFirmaId, :AReceteId, :AIrsaliyeNo, :AFiyatOnay,
    :AUniteId, :APartiGrupId, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AMusteriSiparisNo, :AReferansSiparisNo, :AIslemTipiId,
    :AIslemGrubuId, :ASiparisiAlanKullaniciId, :AKalinlikId, :AIplikCinsiId, :AOzellikId, :AKumasCinsiId,
    :ALycraTipiId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id,
    :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id, :AHamGrup1Aralikli, :AHamGrup2Aralikli,
    :AHamGrup3Aralikli, :AHamGrup4Aralikli, :AYurtDisi, :AMamulSevk, :AHamSatis, :AHamIade, :ABoyaSiparisYil,
    :ABoyaSiparisId, :ASinif, :ABaslangicPartiMiktar, :ABitisPartiMiktar,
    :AOrmeOlmayanlar, :AIplikSiparisi, :ABoyaliIplikSiparisi, :ALycraSiparisi, :AHamKumasSiparisi, :ABoyaliKumasSiparisi, :AIplikBoyaliKumasSiparisi,
    :AImalatTipiId, :AKaliteId, :AAracId, :ADesenId, :ADesenVaryantId, :AHareketTipiID, :ARaporAlanKullaniciID, :APartiKayitId, :ATemsilciId, :AReceteTuruId, :ADepoIdleri, :AMusteriKaliteNo, :ABaskiTipiId)
    into :PartiKayitId, :PartiSinifId do

  begin
    Sira = 0;

    for Select PartiAsamalari.PartiAsamalariId, PartiAsamalari.ReceteId, Recete.Adi, Asama.AsamaId, Asama.Adi, PartiAsamalari.OtomasyonaDahilmi, PartiAsamalari.SerbestAsama,
               PartiAsamalari.PartiTamirId, TamirTipi.Adi, PartiTamir.TamirDuzeltmeTipiId, TamirDuzeltmeTipi.Adi, PartiAsamalari.ProsesNotu, Asama.AraKontrolAsamasimi, PartiAsamalari.CikistaKiloZorunlu,
               PartiAsamalari.CikistaMetreZorunlu, PartiAsamalari.CikistaArabaZorunlu, Personel.Adi, AsamaDegisimSebebi.Adi,
               PartiAsamalari.IlaveProses, PartiAsamalari.IlaveProsesSebebiId, IlaveProsesSebebi.Adi from PartiAsamalari
        left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join Asama on (Asama.AsamaId = PartiAsamalari.AsamaId or Asama.AsamaId = Recete.UstAsamaId)
        left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
        left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
        Left Join TamirDuzeltmeTipi on TamirDuzeltmeTipi.TamirDuzeltmeTipiId = PartiTamir.TamirDuzeltmeTipiId
        left join Personel on Personel.PersonelId = PartiAsamalari.AsamaDegistirenPersonelID
        left join AsamaDegisimSebebi on AsamaDegisimSebebi.AsamaDegisimSebebiId = PartiAsamalari.AsamaDegisimSebebiID
        left join IlaveProsesSebebi on IlaveProsesSebebi.IlaveProsesSebebiId = PartiAsamalari.IlaveProsesSebebiID
        where PartiAsamalari.PartiKayitId = :PartiKayitId
        Order by PartiAsamalari.Sira
        into :PartiAsamalariId, :ReceteId, :ReceteAdi, :AsamaId, :AsamaAdi, :OtomasyonaDahilmi, :SerbestAsama, :PartiTamirId, :TamirTipiAdi, :TamirDuzeltmeTipiId, :TamirDuzeltmeTipiAdi,
             :ProsesNotu, :AraKontrolAsamasimi, :CikistaKilozorunlu, :CikistaMetreZorunlu, :CikistaArabaZorunlu, :AsamaDegistirenPersonelAdi, :AsamaDegisimSebebiAdi,
             :IlaveProses, :IlaveProsesSebebiId, :IlaveProsesSebebiAdi do
    begin
      Virgul = 0;

      PartiAsamaStokDetayId = null;
      /*PartiSinifId          = Null;*/
      PartiSiraNo           = Null;
      PartiStokIptalkayitId = Null;
      MakinaPlanlamaGrupveProses = Null;
      MakinaPlanlamaGrupId   = Null;
      GrupProsesNo           = Null;
      GrupProsesAdi          = Null;

      GirisOtomatik          = Null;
      CikisOtomatik          = Null;
      AsamaAciklamaTipiAdi   = '';

      HareketKullaniciAdi    = Null;
      OrnekPartiSinifmi      = Null;
      OrnekPartiSinifmiStr   = Null;
      OrnekOnay              = Null;
      OrnekPartiYil          = Null;
      OrnekPartiId           = Null;
      OrnekPartiInsertBilgisi = Null;
      OrnekPartiOnayBilgisi   = Null;
      PartiStokBekleId = Null;
      PartiStokBekleBitti = Null;

      HamAdiToplami = '';
      MakinaId = '';
      MakinaAdi = '';
      PlanlananMakinaId = '';
      GirisPersonelAdi = '';
      CikisPersonelAdi = '';
      KiloDegistirenPersonelAdi = '';
      GirisTarihi = Null;
      CikisTarihi = Null;
      FPASCikisTarihi = Null;
      PartiStokBekleId = Null;
      PartiStokBekleBitti = Null;
      Kilo = 0;
      Metre = 0;
      CikisKilo = 0;
      CikisMetre = 0;
      CikisRafID = Null;

      if (IlaveProses = 1) then
        IlaveProsesStr = 'Evet';
      else
        IlaveProsesStr = '';

      ReceteIsEmriIdStr = Null;
      BirlestirilenPartiler = '';

      if (ReceteId is not null) then
        Select ReceteIsEmriIdStr, BirlestirilenPartiler from PAReceteIsEmriBilgisiGetir(:PartikayitId, :ReceteId, :PartiAsamalariId) into :ReceteIsEmriIdStr, :BirlestirilenPartiler;

      For Select PartiStok.HamAdiToplami, Sum(PartiAsamaStok.GirisKilo), Sum(PartiAsamaStok.GirisMetre) from PartiAsamaStok
        left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
        where PartiAsamaStok.PartiKayitId = :PartiKayitId and PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
        Group BY PartiStok.PartiSinifId, PartiStok.HamAdiToplami
        into :FHamAdiToplami, :FKilo, :FMetre do
      begin
        if (Char_Length(HamAdiToplami) + Char_Length(FHamAdiToplami) < 252) then
        begin
          if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set UTF8);
          HamAdiToplami = HamAdiToplami || FHamAdiToplami;
        end
        Kilo = Kilo + FKilo;
        Metre = Metre + FMetre;
      end
      CikisYok = 0;
      MakinaID = '';
      MakinaAdi = '';
      For Select PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.CikisTarihi, GP.Adi, CP.Adi,
               PartiAsamaStokDetay.MakinaId, PartiAsamaStok.CikisTarihi, PartiAsamaStokDetay.CikisKilo,
               PartiAsamaStokDetay.CikisMetre, Makina.Adi from PartiAsamaStok
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
        left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
        left join Personel Gp on Gp.PersonelId = PartiAsamaStokDetay.GirisPersonelId
        left join Personel Cp on Cp.PersonelId = PartiAsamaStokDetay.CikisPersonelId
        left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
        where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
        order by PartiAsamaStok.PartiAsamalariId, PartiAsamaStokDetay.GirisTarihi
        into :FGirisTarihi, :FCikisTarihi, :FGirisPersonelAdi, :FCikisPersonelAdi, :FMakinaId, :FPASCikisTarihi,
        :FCikisKilo, :FCikisMetre, :FMakinaAdi do
      begin
        if (FCikisTarihi is not null) then
        begin
          CikisKilo = CikisKilo + FCikisKilo;
          CikisMetre = CikisMetre + FCikisMetre;
        end

        if (FGirisTarihi is not null) then
        if ((GirisTarihi is null) or (FGirisTarihi < GirisTarihi)) then
        begin
          GirisTarihi = FGirisTarihi;
          GirisPersonelAdi = FGirisPersonelAdi;
        end


        if (FMakinaId is not null) then
        begin
          if (PosBul(MakinaId, FMakinaId) = 0) then
          if (Char_Length(MakinaId) + Char_Length(FMakinaId) < 97) then
          begin
            if (MakinaId <> '' ) then MakinaId = MakinaId || cast(', ' as varchar(2) character set UTF8);
            MakinaId = MakinaId || FMakinaId;
          end
        end

        if (FMakinaAdi is not null) then
        begin
          if (PosBul(MakinaAdi, FMakinaAdi) = 0) then
          if (Char_Length(MakinaAdi) + Char_Length(FMakinaAdi) < 210) then
          begin
            if (MakinaAdi <> '' ) then MakinaAdi = MakinaAdi || cast(', ' as varchar(2) character set UTF8);
            MakinaAdi = MakinaAdi || FMakinaAdi;
          end
        end

        if (FPASCikisTarihi is not null) then
        begin
