CREATE PROCEDURE DOKUMARECETEGETIR(
    AORMESIPARISDETAYID INTEGER,
    AHAMID INTEGER)
RETURNS (
    ORMESIPARISDETAYRECETEID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMID INTEGER,
    RECETEHAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(30),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    KULLANIMORANI DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(15),
    TAHMINIFIREORANI DOUBLE PRECISION,
    COZGUFIREORANI DOUBLE PRECISION,
    MALIYETFIREORANI DOUBLE PRECISION,
    ATKICOZGU INTEGER,
    ATKICOZGUSTR VARCHAR(5),
    TELSAYISI INTEGER,
    SIRANO INTEGER,
    BIRIMMALIYET DOUBLE PRECISION,
    BIRIMMALIYETDOVIZID VARCHAR(5),
    BUKUMMALIYETI DOUBLE PRECISION,
    BUKUMMALIYETIDOVIZID VARCHAR(5),
    AKTARMAMALIYETI DOUBLE PRECISION,
    AKTARMAMALIYETIDOVIZID VARCHAR(5),
    IPLIKBOYAMALIYETI DOUBLE PRECISION,
    IPLIKBOYAMALIYETIDOVIZID VARCHAR(5),
    BIRIMMALIYETKDVORANI DOUBLE PRECISION,
    BUKUMMALIYETKDVORANI DOUBLE PRECISION,
    AKTARMAMALIYETKDVORANI DOUBLE PRECISION,
    IPLIKBOYAMALIYETKDVORANI DOUBLE PRECISION,
    TARAKENI DOUBLE PRECISION,
    ENDENCEKME DOUBLE PRECISION,
    BOYDANCEKME DOUBLE PRECISION,
    DENYE DOUBLE PRECISION,
    NM DOUBLE PRECISION,
    NE DOUBLE PRECISION,
    ATKIIPLIKCM DOUBLE PRECISION,
    ATKIIPLIKINCH DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    AGIRLIKKILO BOYAMIKTARSIZE,
    UZUNLUKMETRE BOYAMIKTARSIZE,
    HATASTR VARCHAR(500))
AS
begin

  For select Null, Null, Null, HamStokRecete.ReceteHamId, ReceteHam.HamTipiId, HamTipi.Adi, ReceteHam.Adi, ReceteHam.Adi2, HamStokRecete.RenkId,
             Renk.Adi, Null, Null, Null, Null, Null,
             HamStokRecete.KullanimOrani, Ik.Adi, HamStokRecete.InsertTarihi, Sk.Adi, HamStokRecete.DegisimTarihi,
             HamStokRecete.FireOrani, HamStokRecete.CozguFireOrani, Null, Null,
             HamStokRecete.AtkiCozgu, decode(HamStokRecete.AtkiCozgu, 1, 'Atkı', 2, 'Çözgü', ''), HamStokRecete.TelSayisi, HamStokRecete.SiraNo, Null, Null,
             Null, Null, Null, Null, Ham.TarakEni, Ham.EndenCekme, Ham.BoydanCekme, Kalinlik.Denye, Kalinlik.Nm, Kalinlik.KalinlikNe, iif(Ham.BolgeselSiklik = 1, HamStokRecete.IplikSikligiCm, Ham.AtkiIplikCm), Ham.AtkiIplikInch,
             HamStokRecete.HamID, Null, Null, Null, Null, HamStokRecete.MaliyetFireOrani from HamStokRecete
    left join Ham on Ham.HamId = HamStokRecete.HamId
    left join Ham ReceteHam on ReceteHam.HamId = HamStokRecete.ReceteHamId
    left join Kalinlik on Kalinlik.KalinlikID = ReceteHam.KalinlikID
    left join Renk on Renk.RenkId = HamStokRecete.RenkId
    left join HamTipi on HamTipi.HamTipiId = ReceteHam.HamTipiId
    Left Join Kullanici IK on IK.KullaniciId = HamStokRecete.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = HamStokRecete.KullaniciId
    where HamStokRecete.HamID = :AHamID
    Union
    select OrmeSiparisDetayRecete.OrmeSiparisDetayReceteID, OrmeSiparisDetayRecete.BoyaSiparisKayitID, OrmeSiparisDetayRecete.OrmeSiparisDetayId,
               OrmeSiparisDetayRecete.HamId, ReceteHam.HamTipiId, HamTipi.Adi,
               ReceteHam.Adi, ReceteHam.Adi2, OrmeSiparisDetayRecete.RenkId, Renk.Adi, OrmeSiparisDetayRecete.MelanjKodu, OrmeSiparisDetayRecete.Lot, OrmeSiparisDetayRecete.UreticiId, Uretici.Adi, OrmeSiparisDetayRecete.Fason,
               OrmeSiparisDetayRecete.KullanimOrani, Ik.Adi, OrmeSiparisDetayRecete.InsertTarihi, Sk.Adi, OrmeSiparisDetayRecete.DegisimTarihi,
               OrmeSiparisDetayRecete.TahminiFireOrani, OrmeSiparisDetayRecete.CozguFireOrani, OrmeSiparisDetayRecete.BirimMaliyet, OrmeSiparisDetayRecete.BirimMaliyetDovizId,
               OrmeSiparisDetayRecete.AtkiCozgu, decode(OrmeSiparisDetayRecete.AtkiCozgu, 1, 'Atkı', 2, 'Çözgü', ''), OrmeSiparisDetayRecete.TelSayisi, OrmeSiparisDetayRecete.SiraNo, OrmeSiparisDetayRecete.BukumMaliyeti, OrmeSiparisDetayRecete.BukumMaliyetiDovizId,
               OrmeSiparisDetayRecete.AktarmaMaliyeti, OrmeSiparisDetayRecete.AktarmaMaliyetiDovizId, OrmeSiparisDetayRecete.IplikBoyaMaliyeti, OrmeSiparisDetayRecete.IplikBoyaMaliyetiDovizId,
               Ham.TarakEni, Ham.EndenCekme, Ham.BoydanCekme, Kalinlik.Denye, Kalinlik.Nm, Kalinlik.KalinlikNe, iif(Ham.BolgeselSiklik = 1, OrmeSiparisDetayRecete.IplikSikligiCm, Ham.AtkiIplikCm), Ham.AtkiIplikInch, OrmeSiparisDetay.HamID,
               OrmeSiparisDetayRecete.BirimMaliyetKdvOrani, OrmeSiparisDetayRecete.BukumMaliyetKdvOrani, OrmeSiparisDetayRecete.AktarmaMaliyetKdvOrani, OrmeSiparisDetayRecete.IplikBoyaMaliyetKdvOrani,
               OrmeSiparisDetayRecete.MaliyetFireOrani from OrmeSiparisDetayRecete
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetayRecete.OrmeSiparisDetayId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join Ham ReceteHam on ReceteHam.HamId = OrmeSiparisDetayRecete.HamId
    left join Kalinlik on Kalinlik.KalinlikID = ReceteHam.KalinlikID
    left join Renk on Renk.RenkId = OrmeSiparisDetayRecete.RenkId
    left join Uretici on Uretici.UreticiId = OrmeSiparisDetayRecete.UreticiId
    left join HamTipi on HamTipi.HamTipiId = ReceteHam.HamTipiId
    Left Join Kullanici IK on IK.KullaniciId = OrmeSiparisDetayRecete.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = OrmeSiparisDetayRecete.KullaniciId
    where OrmeSiparisDetayRecete.OrmeSiparisDetayID = :AOrmeSiparisDetayID
    into :OrmeSiparisDetayReceteID, :BoyaSiparisKayitID, :OrmeSiparisDetayId, :HamId, :HamTipiId, :HamTipiAdi,
       :HamAdi, :HamAdi2, :RenkID, :RenkAdi, :MelanjKodu,  :Lot, :UreticiId, :UreticiAdi, :Fason,
       :KullanimOrani, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :TahminiFireOrani, :CozguFireOrani,
       :BirimMaliyet, :BirimMaliyetDovizId, :AtkiCozgu, :AtkiCozguStr, :TelSayisi, :SiraNo, :BukumMaliyeti, :BukumMaliyetiDovizId,
       :AktarmaMaliyeti, :AktarmaMaliyetiDovizId, :IplikBoyaMaliyeti, :IplikBoyaMaliyetiDovizId,
       :TarakEni, :EndenCekme, :BoydanCekme, :Denye, :Nm, :Ne, :AtkiIplikCM, :AtkiIplikInch, :ReceteHamID,
       :BirimMaliyetKdvOrani, :BukumMaliyetKdvOrani, :AktarmaMaliyetKdvOrani, :IplikBoyaMaliyetKdvOrani, :MaliyetFireOrani do
  begin
    if (Fason = 1) then FasonStr = 'Fason';
    else if (Fason = 0) then FasonStr = '';
    else FasonStr = 'Bilinmiyor';

    select   AgirlikKilo, UzunlukMetre, HataStr from DokumaIplikMiktariHesaplama(:OrmeSiparisDetayReceteID, Null, Null, 0, Null, Null) into :AgirlikKilo, :UzunlukMetre, :HataStr;

    suspend;
  end
end
