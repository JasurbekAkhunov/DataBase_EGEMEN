CREATE PROCEDURE DOKUMABARKODLUKUMASSARFYAP(
    AKALITEKONTROLID INTEGER,
    AORMEFISID INTEGER)
AS
declare variable FIRSALIYEID integer;
declare variable FUIRSALIYEID integer;
declare variable DOKUMAPLANLAMAID integer;
declare variable KKID integer;
declare variable TALBOYASIPARISKAYITID integer;
declare variable BOYASIPARISDETAYID integer;
declare variable TALORMESIPARISDETAYID integer;
declare variable TALHAMID integer;
declare variable BOYASIPARISKAYITID integer;
declare variable ORMESIPARISDETAYID integer;
declare variable HAMID integer;
declare variable STOKHAREKETID integer;
declare variable ORMERAFDETAYID integer;
declare variable RAFID varchar(10);
declare variable ORMERAFID integer;
declare variable MIKTAR numeric(12,3);
declare variable METRE numeric(12,3);
declare variable KILO numeric(12,3);
declare variable KAYITSAYISI integer;
declare variable KAYITSTOKHAREKETID integer;
declare variable SIPARISDOVIZID varchar(5);
declare variable IRSALIYEDOVIZID varchar(5);
declare variable IRSALIYETARIHI timestamp;
declare variable DOVIZKURU double precision;
declare variable SATISSTOKHAREKETID integer;
declare variable URETIMECIKISSTOKHAREKETID integer;
declare variable SAKAT integer;
declare variable FHAMID integer; /* baska siparise satiyor ise kullaniyoruz */
declare variable FIPTAL integer; /* baska siparise satiyor ise kullaniyoruz */
declare variable FLOT varchar(150);
declare variable FDEPOID varchar(10);
declare variable FSARFORMEFISID integer;
declare variable FSARFSTOKHAREKETID integer;
declare variable FBIRIMFIYAT double precision;
declare variable FDOVIZID DOVIZSIZE;
declare variable SHTSARF integer;
declare variable Adet integer;
begin
  select shtSarf from OrmeStokHareketTipiGetir into :shtSarf;

  Select DokumaKaliteKontrol.DokumaKaliteKontrolId, DokumaKaliteKontrol.OrmeIrsaliyeId, DokumaKaliteKontrol.DokumaPlanlamaId, DokumaPlanlama.BoyaSiparisKayitId, DokumaPlanlama.OrmeSiparisDetayId,
         OrmeSiparisDetay.HamId, DokumaKaliteKontrol.RafId, DokumaKaliteKontrol.Miktar, DokumaKaliteKontrol.Kilo, DokumaKaliteKontrol.StokHareketId,
         DokumaKaliteKontrol.OrmeRafDetayId, OrmeSiparisDetay.DovizId, DokumaKaliteKontrol.SatisStokHareketId, DokumaKaliteKontrol.Sakat,
         OrmeSiparisDetay.Iptal, StokHareket.Lot, DokumaKaliteKontrol.UretimeCikisStokHareketId, DokumaKaliteKontrol.OrmeUCikisIrsaliyeId,
         DokumaKaliteKontrol.SarfOrmeFisId, DokumaKaliteKontrol.SarfStokHareketId, StokHareket.BirimFiyat, StokHareket.DovizId, OrmeRafDetay.OrmeRafId, DokumaKaliteKontrol.Adet from DokumaKaliteKontrol
  left join StokHareket on StokHareket.StokHareketID = DokumaKaliteKontrol.StokHareketId
  left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaKaliteKontrol.DokumaPlanlamaId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
  left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
  left join OrmeRafDetay on OrmeRafDetay.StokHAreketId = DokumaKaliteKontrol.StokHareketId
  where DokumaKaliteKontrol.DokumaKaliteKontrolId = :AKaliteKontrolId
  into :KKId, :FIrsaliyeId, :DokumaPlanlamaId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :HamId, :RafId,
       :Miktar, :Kilo, :StokHareketId, :OrmeRafDetayId, :SiparisDovizId, :SatisStokHareketId, :Sakat,
       :FIptal, :FLot, :UretimeCikisStokHareketId, :FUIrsaliyeId, :FSarfOrmeFisId, :FSarfStokHareketId, :FBirimFiyat, :FDovizId, :OrmeRafId, :Adet;

  if (KKId is null) then exception KaliteKontrolIdKaydiBulunamadi;

  if (AOrmeFisId is Not null and FIrsaliyeId is not null) then exception BuUretiminCikisiZatenYapilmis  '(Satis)';
  if (AOrmeFisId is Not null and FUIrsaliyeId is not null) then exception BuUretiminUCikisiZatenYapilmis  '(Uretime Cikis)';
  if (AOrmeFisId is Not null and FSarfOrmeFisId is Not null) then exception BuUretiminUCikisiZatenYapilmis  '(Sarf)';

  if (OrmeRafDetayId is not null) then
  begin
    if (Sakat = 0) then Select UrunRafId from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, Null, null, :StokHareketId, Null) where OrmeRafDetayId = :OrmeRafDetayId into :OrmeRafId;
    else /*Select UrunRafId from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, Null, Null, null, Null) where Sakat = 1 and RafID = :RafId into :OrmeRafId;*/
        OrmeRafId = OrmeRafId; /*zaten raf yukarda belli*/
  end
  else
  begin
    if (Sakat = 0) then Select Max(UrunRafId) from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, Null, Null, :StokHareketId, Null) into :OrmeRafId;
    else Select max(UrunRafId) from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, Null, Null, null, Null) where Sakat = 1 into :OrmeRafId;
  end

  if (AOrmeFisId is not null)  then
  begin
    Select StokHareketId from StokHareket
      where StokHareket.OrmeFisId = :AOrmeFisId and StokHareket.OrmeRafId = :OrmeRafId and StokHareket.OrmeIrsaliyeId is null and
           (StrKarsilastir(StokHareket.Lot, :FLot) = 1)
      into :KayitStokHareketId;


    if (KayitStokHareketId is not Null) then
    begin
      Update StokHareket Set KapAdedi = KapAdedi + 1, Miktar = Miktar + :Miktar, BrutMiktar = BrutMiktar + :Miktar, Kilo = Kilo + :Kilo, Metre = Metre + :Miktar, NeredenUpdate = 0, Adet = Adet + :Adet where StokHareket.StokHareketId = :KayitStokHareketId;
    end
    else
    begin
      KayitStokHareketId = gen_id(StokHareket_Id_Gen, 1);

      insert into StokHareket(StokHareketId, HareketTipi, OrmeFisId, NeredenUpdate, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, OrmeRafId, KapAdedi, Miktar, Kilo, Metre, BrutMiktar, Adet, BirimFiyat, Lot, BoyaTalimatKayitId, BoyaSiparisDetayId, DokumaHareketi, DovizID)
                    values(:KayitStokHareketId, :shtSarf, :AOrmeFisId, 0, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :HamId, :OrmeRafId, 1, :Miktar, :Kilo, :Miktar, :Miktar, :Adet, :FBirimFiyat, :FLot, null, :BoyaSiparisDetayId, 1, :FDovizId);

    end
    Update DokumaKaliteKontrol Set SarfOrmeFisID = :AOrmeFisId, SarfStokHareketId = :KayitStokHareketId, LinkIslemi = 1 where DokumaKaliteKontrol.DokumaKaliteKontrolId = :AKaliteKontrolId;
  end
  else
  begin
    if (FSarfStokHareketId is Null) then exception BuUretimUCikisiZatenYapilmamis;

    Select count(*) from DokumaKaliteKontrol where DokumaKaliteKontrol.SarfStokHareketId = :FSarfStokHareketId into :KayitSayisi;
    if (KayitSayisi = 1) then delete from StokHareket where StokHareket.StokHareketId = :FSarfStokHareketId;
    else
    begin
      Update StokHareket Set KapAdedi = KapAdedi - 1, Miktar = Miktar - :Miktar, Kilo = Kilo - :Kilo, Metre = Metre - :Miktar, BrutMiktar = BrutMiktar - :Miktar, Adet = Adet - :Adet, NeredenUpdate = 0 where StokHareket.StokHareketId = :FSarfStokHareketId;
      Update DokumaKaliteKontrol Set SarfOrmeFisId = Null, SarfStokHareketId = Null, LinkIslemi = 1 where DokumaKaliteKontrol.DokumaKaliteKontrolId = :AKaliteKontrolId;
    end
  end
end
