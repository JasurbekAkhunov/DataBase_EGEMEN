      into :UrunId, :UrunTipiId, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :Miktar, :UretimeCikisFiyati, :Lot, :FBaskiAsamasimi do
      begin
        Okey = 1;

        if (okey = 1) then
        begin
          if (UretimeCikisFiyati is null) then UretimeCikisFiyati = 0;

          UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;

          if (UrunTipiId = 0) then BoyaMaliyeti = BoyaMaliyeti + Miktar * UretimeCikisFiyati;
          else KimyasalMaliyeti = KimyasalMaliyeti + Miktar * UretimeCikisFiyati;

          BoyaKimyasalMaliyeti = BoyaKimyasalMaliyeti + Miktar * UretimeCikisFiyati;
          if (FBaskiAsamasimi = 1) then
            BaskiBoyaKimyasalMaliyeti = BaskiBoyaKimyasalMaliyeti + Miktar * UretimeCikisFiyati;
        end
      end

      if (PartiYil is null) then PartiIdStr = '';
      else
      begin
        PartiIdStr = '(';
        PartiIdStr = PartiIdStr || Format_Float('#0', PartiYil);
        PartiIdStr = PartiIdStr || ') ';
        PartiIdStr = PartiIdStr || Format_Float('#0', PartiId);
        PartiIdStr = PartiIdStr || ' ';
      end

      if (Fason = 1) then
      begin
        FasonStr = 'Evet';
        HamStokMaliyeti = 0;
      end
      else
      begin
        FasonStr = '';
        HamStokMaliyeti = 0;
        For Select HamStokIrsaliye.DovizId, Sum(PartiStokDetay.Kilo * HamStokMaster.NetBirimFiyat) from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
          left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
          where PartiStokDetay.PartiKayitId = :PartiKayitId  and PartiStokDetay.PartiSinifId = :PartiSinifId
          Group By HamStokIrsaliye.DovizId
          into :FIrsaliyeDovizId, :FMaliyet do
        begin
          if (FIrsaliyeDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FIrsaliyeDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
          else DovizKuru = 1;

          if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;

          HamStokMaliyeti = HamStokMaliyeti + FMaliyet  * DovizKuru;
        end
      end

      AsamaUMUzerindenAsamaMaliyeti  = 0;
      MakinaUMUzerindenAsamaMaliyeti = 0;

      if (PartiAsamaMaliyetHesapTipi = 0) then
      begin
        For Select PartiAsamalari.UretimMaliyeti, PartiAsamalari.UretimMaliyetiDovizId, Max(PartiAsamaStokDetay.UretimMaliyeti), Max(PartiAsamaStokDetay.UretimMaliyetiDovizId),
                   Min(PartiAsamaStokDetay.GirisTarihi), Max(PartiAsamaStokDetay.CikisTarihi), Max(PartiAsamaStokDetay.AraVermeSuresi) from PartiAsamalari
          left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
          left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
          where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamaStok.AsamayaDahil = 1 and
                PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is not null and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
          group By PartiAsamaStokDetay.MakinaSarjId, PartiAsamalari.UretimMaliyeti, PartiAsamalari.UretimMaliyetiDovizId
          into :FAsamaUretimMaliyeti, :FAsamaUretimMaliyetiDovizId, :FMakinaUretimMaliyeti, :FMakinaUretimMaliyetiDovizId, :FGirisTarihi, :FCikisTarihi, :FAraVermeSuresi do
        begin
          if (FAsamaUretimMaliyetiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FAsamaUretimMaliyetiDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
          else DovizKuru = 1;

          if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;
          AsamaUMUzerindenAsamaMaliyeti = AsamaUMUzerindenAsamaMaliyeti + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FAsamaUretimMaliyeti * DovizKuru;

          if (FMakinaUretimMaliyetiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FMakinaUretimMaliyetiDovizId, :ARaporDovizId, :AcilisTarihi) into :DovizKuru;
          else DovizKuru = 1;

          if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;
          MakinaUMUzerindenAsamaMaliyeti = MakinaUMUzerindenAsamaMaliyeti + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FMakinaUretimMaliyeti * DovizKuru;
        end
      end
      else
      begin
        For Select PartiAsamalari.UretimMaliyetiKg, PartiAsamalari.UretimMaliyetiKgDovizId,
                   Sum(PartiAsamaStokDetay.Kilo), Max(PartiAsamaStokDetay.UretimMaliyeti), Max(PartiAsamaStokDetay.UretimMaliyetiDovizId),
                   Min(PartiAsamaStokDetay.GirisTarihi), Max(PartiAsamaStokDetay.CikisTarihi), Max(PartiAsamaStokDetay.AraVermeSuresi) from PartiAsamalari
          left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
          left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
          where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamaStok.AsamayaDahil = 1 and
                PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is not null and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
          group By PartiAsamaStokDetay.MakinaSarjId, PartiAsamalari.UretimMaliyetiKg, PartiAsamalari.UretimMaliyetiKgDovizId
          into :FAsamaUretimMaliyeti, :FAsamaUretimMaliyetiDovizId, :FKilo, :FMakinaUretimMaliyeti, :FMakinaUretimMaliyetiDovizId, :FGirisTarihi, :FCikisTarihi, :FAraVermeSuresi do
        begin
          if (FAsamaUretimMaliyetiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FAsamaUretimMaliyetiDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
          else DovizKuru = 1;

          if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;

          AsamaUMUzerindenAsamaMaliyeti = AsamaUMUzerindenAsamaMaliyeti + FKilo * FAsamaUretimMaliyeti * DovizKuru;

          if (FMakinaUretimMaliyetiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FMakinaUretimMaliyetiDovizId, :ARaporDovizId, :AcilisTarihi) into :DovizKuru;
          else DovizKuru = 1;

          if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;
          MakinaUMUzerindenAsamaMaliyeti = MakinaUMUzerindenAsamaMaliyeti + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FMakinaUretimMaliyeti * DovizKuru;
        end
      end

      AsamaUMUzerindenPartiMaliyeti = AsamaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti;
      MakinaUMUzerindenPartiMaliyeti = MakinaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti;

/* eger recetesi arsivlenmis ise teorik kilolardan boya kimyasal maliyeti hesaplanacak */
      for select PartiReceteDetay.UrunId, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, PartiReceteDetay.HesaplananMiktar, PartiReceteDetay.BirimFiyat, PartiReceteDetay.DovizId from PartiAsamalari
        left join PartiReceteDetay on PartiReceteDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join Urun on Urun.UrunId = PartiReceteDetay.UrunId
        left join UrunBirim UBSD on UBSD.UrunBirimId = PartiReceteDetay.UrunBirimId
        left join UrunFiyat on UrunFiyat.UrunId = PartiReceteDetay.UrunId
        left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiReceteDetay.UrunId is not null --and PartiAsamalar.PartiSinifId = :PartiSinifId Mustafa İle Bak
        order by UrunFiyat.UretimeCikisFiyatiDovizId /* Aynı doviz cinsleri icin dovizkuru getire gitmesin diye sirali okuyoruz..*/
      into :UrunId, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :Miktar, :UretimeCikisFiyati, :DovizID do
      begin
        Okey = 1;

        if (okey = 1) then
        begin
          if (DovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, Current_date) into :DovizKuru;
          else DovizKuru = 1;

          if (UretimeCikisFiyati is null) then UretimeCikisFiyati = 0;

          UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;
          TeorikBoyaKimyasalMaliyeti = TeorikBoyaKimyasalMaliyeti + Miktar * UretimeCikisFiyati;
        end
      end

      OrtalamaBirimFiyat = Null;

      Select Min(BoyaFatura.DovizId), Sum(BoyaFaturaDetay.BirimFiyat * BoyaFaturaDetay.Miktar) / Sum(BoyaFaturaDetay.Miktar) from BoyaIrsaliyeDetay
        left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
        left join BoyaFatura on BoyaFatura.BoyaFaturaId = BoyaFaturaDetay.BoyaFaturaId
        where BoyaIrsaliyeDetay.PartiKayitId = :PartiKayitId and BoyaIrsaliyeDetay.BoyaFaturaId is not null and BoyaFaturaDetay.Miktar <> 0  and BoyaIrsaliyeDetay.PartiSinifId = :PartiSinifId
        into :FFaturaDovizId, :OrtalamaBirimFiyat;

      if (FFaturaDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FFaturaDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;

      OrtalamaBirimFiyat = OrtalamaBirimFiyat * DovizKuru;
      OrtalamaBirimFiyat = Yuvarla(OrtalamaBirimFiyat, 1000);

      Select DogalgazSarfiyati, ElektirikSarfiyati, OkunanSuSarfiyati from PartiEnerjiSarfiyatiHesapla(:PartiKayitID) into :DogalgazSarfiyati, :ElektirikSarfiyati, :OkunanSuSarfiyati;

      /*Yukaridaki hesaplamalarda bulunan maliyetlerin tamamini, kumulatiften yeniden aliyoruz
        yukaridakileri silmeme sebebim tedbir, maliyetin duzgun calistigi teyid edilince
        yukaridaki hesaplamalar silinebilir*/

      select first 1 HamStokMaliyeti, AsamaUMUzerindenAsamaMaliyeti, MakinaUMUzerindenAsamaMaliyeti, BoyaKimyasalMaliyeti,
           TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti, BaskiBoyaKimyasalMaliyeti from PartiSinifMaliyetKumulatif
           where PartiSinifMaliyetKumulatif.PartiSinifId = :PartiSinifId
           into :HamStokMaliyeti, :AsamaUMUzerindenAsamaMaliyeti, :MakinaUMUzerindenAsamaMaliyeti, :BoyaKimyasalMaliyeti,
           :TeorikBoyaKimyasalMaliyeti, :UnitelerASevkMaliyeti, :BaskiBoyaKimyasalMaliyeti;

      AsamaUMUzerindenPartiMaliyeti = AsamaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti + UnitelerASevkMaliyeti;
      MakinaUMUzerindenPartiMaliyeti = MakinaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti + UnitelerASevkMaliyeti;

      if (SistemDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:SistemDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
      else DovizKuru = 1;

      AsamaUMUzerindenPartiMaliyeti = AsamaUMUzerindenPartiMaliyeti * DovizKuru;
      AsamaUMUzerindenAsamaMaliyeti = AsamaUMUzerindenAsamaMaliyeti * DovizKuru;
      HamStokMaliyeti = HamStokMaliyeti * DovizKuru;
      BoyaKimyasalMaliyeti = BoyaKimyasalMaliyeti * DovizKuru;
      UnitelerASevkMaliyeti = UnitelerASevkMaliyeti * DovizKuru;
      MakinaUMUzerindenAsamaMaliyeti = MakinaUMUzerindenAsamaMaliyeti * DovizKuru;
      MakinaUMUzerindenPartiMaliyeti = MakinaUMUzerindenPartiMaliyeti * DovizKuru;
      TeorikBoyaKimyasalMaliyeti = TeorikBoyaKimyasalMaliyeti * DovizKuru;
      BaskiBoyaKimyasalMaliyeti = BaskiBoyaKimyasalMaliyeti * DovizKuru;
      /*Maliyet alimi bitti*/


      select sum(PartiStokTarti.Kilo + PartiStokTarti.Dara), Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null, 1, 0)), Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null AND  PartiStokTarti.HamStokSarfFisiDetayId is null, PartiStokTarti.Kilo + PartiStokTarti.Dara, 0)),
                Sum(iif(PartiStokTarti.KaliteNo=1, PartiStokTarti.Kilo, 0)),  Sum(iif(PartiStokTarti.KaliteNo=1, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0)),
           Sum(iif(PartiStokTarti.KaliteNo=2, PartiStokTarti.Kilo, 0)),  Sum(iif(PartiStokTarti.KaliteNo=2, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0)),
           Sum(iif(PartiStokTarti.KaliteNo=3, PartiStokTarti.Kilo, 0)),  Sum(iif(PartiStokTarti.KaliteNo=3, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0)),
           Sum(iif(PartiStokTarti.KaliteNo=4, PartiStokTarti.Kilo, 0)),  Sum(iif(PartiStokTarti.KaliteNo=4, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0)),
           Sum(iif(PartiStokTarti.KaliteNo=5, PartiStokTarti.Kilo, 0)),  Sum(iif(PartiStokTarti.KaliteNo=5, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=1, PartiStokTarti.Kilo, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=1, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=2, PartiStokTarti.Kilo, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=2, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=3, PartiStokTarti.Kilo, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=3, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=4, PartiStokTarti.Kilo, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=4, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=5, PartiStokTarti.Kilo, 0)),
           Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and  PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.KaliteNo=5, PartiStokTarti.Kilo+ PartiStokTarti.Dara, 0))
      from PartiStokTarti
      where PartiStokTarti.PartiSinifId = :PartiSinifId into :TartiBrutKilo, :KalanTopAdedi, :KalanBrutKilo,
      :K1TartiNetKilo, :K1TartiBrutKilo, :K2TartiNetKilo, :K2TartiBrutKilo, :K3TartiNetKilo, :K3TartiBrutKilo, :K4TartiNetKilo, :K4TartiBrutKilo, :K5TartiNetKilo, :K5TartiBrutKilo,
      :K1TartiKalanNetKilo, :K1TartiKalanBrutKilo, :K2TartiKalanNetKilo, :K2TartiKalanBrutKilo, :K3TartiKalanNetKilo, :K3TartiKalanBrutKilo, :K4TartiKalanNetKilo, :K4TartiKalanBrutKilo,
      :K5TartiKalanNetKilo, :K5TartiKalanBrutKilo;
      suspend;
    end
