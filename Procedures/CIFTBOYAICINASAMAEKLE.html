CREATE PROCEDURE CIFTBOYAICINASAMAEKLE(
    APARTIASAMASTOKDETAYID INTEGER)
AS
declare variable PartiAsamalariId   integer;
declare variable PartiKayitId   integer;
declare variable Sira           integer;
declare variable AsamaId        integer;
declare variable YapilanPartiAsamalariId integer;
declare variable PartiSinifId   integer;
declare variable MakinaId       varchar(10);
declare variable PartiRptTakibi integer;
begin
  Select PartiAsamaStokDetay.PartiKayitId, PartiAsamalari.Sira, PartiAsamaStok.AsamaId, PartiAsamalari.PartiAsamalariId, PartiAsamaStokDetay.MakinaId from PartiAsamaStokDetay
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariID = PartiAsamaStokDetay.PartiAsamalariID and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStokDetay.PartiAsamalariID
    where PartiAsamaStokDetay.PartiAsamaStokDetayID = :APartiAsamaStokDetayId
    into :PartiKayitId, :Sira, :AsamaId, :YapilanPartiAsamalariId, :MakinaId;

  if (PartiKayitID is null) then exit;

  Sira = Sira + 1;

  PartiAsamalariID = Gen_Id(PartiAsamalari_Id_Gen, 1);
  insert into PartiAsamalari(PartiAsamalariID, PartiKayitId, AsamaId, OtomasyonaDahilmi, Sira, ProsesNotu, InsertKullaniciId, KullaniciId)
                     values(:PartiAsamalariID, :PartiKayitId, :AsamaId, 1, :Sira, 'Kazanda Üretim Bitmeden Çıkış Verildiği Sistem Tarafından Eklendi', 1, 1);

  Update PartiAsamaStok Set AsamayaDahil = 0 where PartiAsamaStok.PartiAsamalariID= :PartiAsamalariID;

  For Select PartiAsamaStok.PartiSinifId from PartiAsamaStok
    where PartiAsamaStok.PartiAsamalariID = :YapilanPartiAsamalariID and PartiAsamaStok.AsamayaDahil = 1
    into :PartiSinifId do
  begin
    Update PartiAsamaStok Set AsamayaDahil = 1 where PartiAsamaStok.PartiAsamalariID= :PartiAsamalariID and PartiAsamaStok.PartiSinifId = :PartiSinifId;
  end


  if (MakinaId is not null) then
  begin
    Execute Procedure AsamaGirisindenPlanlamaYap(:PartiAsamalariId, :MakinaId, 1);
  end

  Select PartiRptTakibi from Parti
    left join IslemGrubu on IslemGrubu.IslemGrubuID = Parti.IslemGrubuId
    where Parti.PartiKayitId = :PartiKayitId
    into :PartiRptTakibi;

  if (PartiRptTakibi = 1) then
    insert into PartiRpt(PartiKayitId, Notu) Values(:PartiKayitId, 'Kazanda Üretim Bitmeden Çıkış Verildiği Sistem Tarafından Eklendi');
end
