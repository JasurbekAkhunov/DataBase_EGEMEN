CREATE PROCEDURE PARTIALTBEZCIKISLARIGETIR(
    APARTISINIFID INTEGER)
RETURNS (
    HAMSTOKSARFFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    HAMSTOKSARFFISIDETAYID INTEGER,
    PARTISINIFID INTEGER,
    RAFDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    METRE BOYAMIKTARSIZE,
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    LOT VARCHAR(150),
    RAFID RAFKODUSIZE,
    PARTISTOKTARTIID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER)
AS
begin
  for select HamStokSarfFisi.HamStokSarfFisiId, HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo, HamStokSarfFisi.FisTarihi, HamStokSarfFisiDetay.HamStokSarfFisiDetayId,
         HamStokSarfFisiDetay.PartiSinifId, HamStokSarfFisiDetay.RafDetayId, Ham.HamId, Ham.Adi, HamStokSarfFisiDetay.Metre, HamStokDetay.HamStokMasterId,
         HamStokDetay.HamStokDetayId, HamStokMaster.Lot, RafDetay.RafId, PartiStokTarti.PartiStokTartiId, Parti.PartiYil, Parti.PartiId
  from PartiAsamaStokDetay
  join HamStokSarfFisiDetay on HamStokSarfFisiDetay.PArtiAsamaStokDetayId = PartiAsamaStokDetay.PartiAsamaStokDetayId
  join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiId = HamStokSarfFisiDetay.HamStokSarfFisiId
  join Ham on Ham.HamId = HamStokSarfFisiDetay.HamId
  left join RafDetay on RafDetay.RafDetayId = HamStokSarfFisiDetay.RafDetayId
  left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join PartiStokTarti on PartiStokTarti.HamStokSarfFisiDetayId = HamStokSarfFisiDetay.HamStokSarfFisiDetayId
  left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
  left join Parti on Parti.PartiKayitId = PartiStok.PArtiKayitId
  where PartiAsamaStokDetay.PartiSinifId = :APartiSinifId
  into :HamStokSarfFisiId, :FisYil, :FisNo, :FisTarihi, :HamStokSarfFisiDetayId, :PartiSinifId, :RafDetayId, :HamId, :HamAdi, :Metre, :HamStokMasterId,
  :HamStokDetayId, :Lot, :RafId, :PartiStokTartiId, :PartiYil, :PartiId do
    suspend;
end
