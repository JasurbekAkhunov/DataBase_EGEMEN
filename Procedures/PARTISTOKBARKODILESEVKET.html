CREATE PROCEDURE PARTISTOKBARKODILESEVKET(
    ABOYAIRSALIYEID INTEGER,
    AFIRMAID VARCHAR(20),
    ABARKODID VARCHAR(20),
    AKULLANICIID INTEGER)
AS
declare variable PartiKayitId           integer;
declare variable PartiSinifId           integer;
declare variable DepoId                 varchar(10);
declare variable PartiStokTartiId       integer;
declare variable BoyaIrsaliyeDetayId    integer;
declare variable HamStokSarfFisiDetayId integer;
declare variable PaketId                integer;
declare variable IK                     varchar(2);
declare variable TF                     integer;
declare variable FirmaId                varchar(20);
declare variable ParcaliSevk            integer;
declare variable Durumu                 integer;
declare variable TamamiTartildi         integer;
declare variable Sevk                   integer;
declare variable OrmeFirmaKodumuz       varchar(20);
declare variable FirmaKodumuz           varchar(20);
declare variable EgemenFirmaId          varchar(20);
declare variable BoyaRezervFisiDetayId  integer;
declare variable FBarkodID          varchar(30);
declare variable IadeBoyaIrsaliyeId Integer;
declare variable KS Integer;
declare variable BaskaFirmayaSatilabilir Integer;
declare variable Fason EgeBoolean;
declare variable TopNo VarChar(25);
begin
  if (Char_Length(ABarkodId) < 3) then exception BarkodHatali1;

  ABarkodId = Replace(ABarkodId, '*', '-');

  FBarkodID = ABarkodId;
  IK = EgemenCopyStr(ABarkodId, 1, 2);

  ABarkodId = EgemenCopyStr(ABarkodId, 3, Char_Length(ABarkodId) - 2);
  TF = EgemenStrToIntOlurmu(ABarkodId);

  if (TF = 0) then exception BarkodHatali1;

  Select FirmaKodumuz, OrmeFirmaKodumuz, EgemenFirmaId from Sistem where sistemid = 1 into :FirmaKodumuz, :OrmeFirmaKodumuz, :EgemenFirmaId;

  select count(*), Max(PartiStokTartiId) from PartiStokTarti where BarkodId = :FBarkodId into :KS, :PaketId;
  if (KS > 0) then /*iade kaydidir*/
    IK = 'R-';
  else
    PaketId = EgemenStrToInt(ABarkodId);

  TopNo = '';
  if (IK = 'K-') then
  begin
    Select PsKaliteKontrol.PartiKayitId, PsKaliteKontrol.PartiSinifId, PsKaliteKontrol.PartiStokTartiId, PartiStokTarti.DepoId,
           PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, Depo.Sevk, PartiStokTarti.BoyaRezervFisiDetayId,
           PartiStokTarti.IadeBoyaIrsaliyeId, 'Top No:' || Cast(PsKaliteKontrol.TopNo as VarChar(10)) from PsKaliteKontrol
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = PsKaliteKontrol.PartiStokTartiId
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where PsKaliteKontrol.PsKaliteKontrolId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :DepoId, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :Sevk, :BoyaRezervFisiDetayId, :IadeBoyaIrsaliyeId, :TopNo;
  end
  else
  if (IK = 'R-') then /*iade kaydi*/
  begin
    Select PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId, PartiStokTarti.DepoId,
           PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, Depo.Sevk, PartiStokTarti.BoyaRezervFisiDetayId, PartiStokTarti.IadeBoyaIrsaliyeId from PartiStokTarti
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where PartiStokTarti.PartiStokTartiId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :DepoId, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :Sevk, :BoyaRezervFisiDetayId, :IadeBoyaIrsaliyeId;
  end
  else
  begin
    Select IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, IplikPaketleme.PartiStokTartiId, PartiStokTarti.DepoId,
           PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, Depo.Sevk, PartiStokTarti.BoyaRezervFisiDetayId, PartiStokTarti.IadeBoyaIrsaliyeId from IplikPaketleme
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiId
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where IplikPaketleme.IplikPaketlemeId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :DepoId, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :Sevk, :BoyaRezervFisiDetayId, :IadeBoyaIrsaliyeId;
  end


  if (PartiSinifId is null) then exception BarkodHatali2;
  if (PartiStokTartiId is null) then exception TartiKaydiYokSevkEdilemez;
  if (BoyaIrsaliyeDetayId is not null) then exception BuPaketinIrsaliyesiKesilmis;
  /* Emre kapatti, yarim cuval iade girisindeki mantik bunu kapsadi
  if (BoyaIrsaliyeDetayId is not null) then Irsaliye ID si Dolu olsada iade girişi olabilir
  begin
    Select PartiStokTartiId, BoyaRezervFisiDetayId, PartiStokTarti.DepoId, Depo.Sevk, PartiStokTarti.IadeBoyaIrsaliyeId from PartiStokTarti
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where PartiStokTarti.BarkodId = :FBarkodID and
            PartiStokTarti.BoyaIrsaliyeDetayId is Null and
            PartiStokTarti.HamStokSarfFisiDetayId is Null and
            PartiStokTarti.IadeBoyaIrsaliyeDetayId is not null
    into :PartiStokTartiId, :BoyaRezervFisiDetayId, :DepoId, :Sevk, :IadeBoyaIrsaliyeId;

    if (PartiStokTartiID is Null) then exception BuPaketinIrsaliyesiKesilmis;
  end*/

  if (HamStokSarfFisiDetayId is not null) then exception BuPaketSarFisiIleCikmis;

  if (Sevk = 0) then exception SevkiyatDeposundaDegil;

  if (BoyaRezervFisiDetayId is not null) then
  begin
    select BoyaRezervFisi.FirmaId from BoyaRezervFisiDetay
    Join BoyaRezervFisi on BoyaRezervFisi.BoyaRezervFisiId = BoyaRezervFisiDetay.BoyaRezervFisiId
    where BoyaRezervFisiDetay.BoyaRezervFisiDetayId = :BoyaRezervFisiDetayId into :FirmaId;

    if (FirmaId <> AFirmaId) then exception TartiKaydiBaskaFirmayaRezerve;
  end


  Select Parti.FirmaId, Parti.ParcaliSevk, PartiStok.Durumu, PartiStok.TamamiTartildi,
         BoyaSiparis.BaskaFirmayaSatilabilir, BoyaSiparis.Fason from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitid = Parti.BoyaSiparisKayitid
    where PartiStok.PartiSinifId = :PartiSinifId
    into :FirmaId, :ParcaliSevk, :Durumu, :TamamiTartildi, :BaskaFirmayaSatilabilir, :Fason;

  if (EgemenFirmaId <> 30 and IadeBoyaIrsaliyeId is Null ) then  /*İade alınmış bir Ürünü Kendi stoğumuza aldığımız için firma bilgisini önemsemiyoruz*/
  begin
    if (AFirmaId <> OrmeFirmaKodumuz and FirmaID <> FirmaKodumuz) then
    if (FirmaId <> AFirmaId and ((BaskaFirmayaSatilabilir = 0 and Fason = 0) or Fason = 1) ) then exception FirmaBilgisiHatali;
  end

  if (Durumu <> 7) then exception PartiStokSevkeHazirDegil;

  if (ParcaliSevk = 0 and TamamiTartildi = 0) then exception ParcaliSevkEdemezsiniz;

  execute Procedure PartiStokKapSevkEt(:ABoyaIrsaliyeId, :PartiKayitId, :PartiSinifId, :DepoId, :PartiStokTartiId, :AKullaniciId, null);
end
