CREATE PROCEDURE PARTIVERIRAPORU_2(
    APARTIYIL INTEGER,
    APARTIID INTEGER)
RETURNS (
    PARTIASAMASTOKDETAYID INTEGER,
    MAKINAID VARCHAR(10),
    MAKINABILGIID INTEGER,
    TARIH TIMESTAMP,
    NEM DOUBLE PRECISION,
    ISI DOUBLE PRECISION,
    HIZ DOUBLE PRECISION,
    BACANEM DOUBLE PRECISION,
    ELEKTIRIK INTEGER,
    DOGALGAZ INTEGER,
    RAMUSTBESLEME DOUBLE PRECISION,
    RAMALTBESLEME DOUBLE PRECISION,
    RAMCALISMAHIZI DOUBLE PRECISION,
    RAMGIRISEN DOUBLE PRECISION,
    RAMBIRBACAFANCALISMAYUZDESI DOUBLE PRECISION,
    RAMIKIBACAFANCALISMAYUZDESI DOUBLE PRECISION,
    RAMSOGUTMAFANIUST DOUBLE PRECISION,
    RAMSOGUTMAFANIALT DOUBLE PRECISION,
    RAMSOGUTMAFANIUSTMAN DOUBLE PRECISION,
    RAMSOGUTMAFANIALTOTO DOUBLE PRECISION,
    RAMBACANEMIOTOMATIK EGEBOOLEAN,
    RAMBACANEMI DOUBLE PRECISION,
    RAMCIKISEN DOUBLE PRECISION,
    RAMCIKISNEMDEGERI DOUBLE PRECISION,
    RAMBESLEMEHIZI DOUBLE PRECISION,
    RAMKALIPGENISLIGI DOUBLE PRECISION,
    RAMKUMASCEKMEZLIKDEGERI DOUBLE PRECISION,
    RAMKUMASMAYDEGERI DOUBLE PRECISION,
    RAMKUMASNEMI DOUBLE PRECISION,
    RAMMAKINEHIZI DOUBLE PRECISION,
    RAMMAKINESILINDIRSICAKLIGI DOUBLE PRECISION)
AS
declare variable GirisTarihi  TimeStamp;
declare variable CikisTarihi  TimeStamp;
begin
  for Select PartiAsamaStokDetayId, MakinaId, GirisTarihi, CikisTarihi from PartiVeriRaporu_1(:APartiYil, :APartiId) order by GirisTarihi into :PartiAsamaStokDetayId, :MakinaId, :GirisTarihi, :CikisTarihi do
  begin
    for Select MakinaBilgi.MakinaBilgiId, MakinaBilgi.tarih, MakinaBilgi.nem, MakinaBilgi.Isi, MakinaBilgi.Hiz, MakinaBilgi.BacaNem,
               MakinaBilgi.Elektirik, MakinaBilgi.Dogalgaz, MakinaBilgi.RAMUstBesleme, MakinaBilgi.RAMAltBesleme, MakinaBilgi.RAMCalismaHizi,
               MakinaBilgi.RAMGirisEn, MakinaBilgi.RAMBirBacaFanCalismaYuzdesi, MakinaBilgi.RAMIkiBacaFanCalismaYuzdesi, MakinaBilgi.RAMSogutmaFaniUst,
               MakinaBilgi.RAMSogutmaFaniAlt, MakinaBilgi.RAMSogutmaFaniUstMan, MakinaBilgi.RAMSogutmaFaniAltOto, MakinaBilgi.RAMBacaNemiOtomatik, MakinaBilgi.RAMBacaNemi,
               MakinaBilgi.RAMCikisEn, RamCikisNemDegeri, RamBeslemeHizi, RamKalipGenisligi, RamKumasCekmezlikDegeri, RamKumasMayDegeri, RamKumasNemi, RamMakineHizi, RamMakineSilindirSicakligi from MakinaBilgi
      where MakinaBilgi.Tarih >= :GirisTarihi and MakinaBilgi.Tarih <= :CikisTarihi and MakinaBilgi.MakinaId = :MakinaId
      into :MakinaBilgiId, :Tarih, :Nem, :Isi, :Hiz, :BacaNem, :Elektirik, :Dogalgaz, :RAMUstBesleme, :RAMAltBesleme, :RAMCalismaHizi,
           :RAMGirisEn, :RAMBirBacaFanCalismaYuzdesi, :RAMIkiBacaFanCalismaYuzdesi, :RAMSogutmaFaniUst, :RAMSogutmaFaniAlt, :RAMSogutmaFaniUstMan, :RAMSogutmaFaniAltOto, :RAMBacaNemiOtomatik,
           :RAMBacaNemi, :RAMCikisEn, :RamCikisNemDegeri, :RamBeslemeHizi, :RamKalipGenisligi, :RamKumasCekmezlikDegeri, :RamKumasMayDegeri, :RamKumasNemi, :RamMakineHizi, :RamMakineSilindirSicakligi do
    begin

      suspend;
    end
  end
end
