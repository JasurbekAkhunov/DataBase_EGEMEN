CREATE PROCEDURE HSFICINHAMSTOKIRSALIYELISTESI(
    AHAMSTOKHAREKETTIPIID INTEGER,
    AFIRMAID VARCHAR(20),
    AIRSALIYENO IRSALIYENOSIZE,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    HAMSTOKIRSALIYEID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    HAMSTOKMASTERID INTEGER,
    HAMADI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3))
AS
begin
  for Select HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokMaster.HamStokMasterId,
         Ham.Adi, HamStokMaster.Kilo, HamStokMaster.Metre from HamStokMaster
  left join Ham on Ham.HamID = HamStokMaster.HamId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = HamStokMaster.HamStokIrsaliyeId
  where HamStokMaster.FirmaId = :AFirmaId and HamStokMaster.HamStokFaturaId is null and
        HamStokMaster.HamStokHareketTipiId = :AHamStokHareketTipiId and HamStokMaster.Fason = 0 and HamStokIrsaliye.EvrakTipi = 1 and
        ( (HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo and :AIrsaliyeNo is not null) or
         (:AIrsaliyeNo is null and HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi ) )
  order by HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokMaster.HamStokMasterId
  into :HamStokIrsaliyeId, :IrsaliyeNo, :IrsaliyeTarihi, :HamStokMasterId, :HamAdi, :Kilo, :Metre do
  begin
    suspend;
  end
end
