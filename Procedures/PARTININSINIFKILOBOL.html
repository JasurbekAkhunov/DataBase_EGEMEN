CREATE PROCEDURE PARTININSINIFKILOBOL(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    AKILO NUMERIC(12,3),
    AKULLANICIID INTEGER,
    APSKALITEKONTROLASAMAID INTEGER,
    AMETRE NUMERIC(12,3),
    ARAFDETAYIDLER VARCHAR(100),
    ARAFMIKTARLARI VARCHAR(100),
    AOTONOMTOPPARCALANIYOR SMALLINT)
AS
declare variable FEskiSinifKilo    Numeric(12,3);
declare variable FYeniPartiSinifId integer;
declare variable FRafDetayId            integer;
declare variable FHamId                 integer;
declare variable FKapAdedi              integer;
declare variable FKilo                  Numeric(12,3);
declare variable FAdet                  integer;
declare variable FPartilemeTarihi       TimeStamp;
declare variable KayitSayisi            integer;
declare variable FMetre                 Numeric(12,3);
declare variable FMetreOran             Double Precision;
declare variable FBobinSayisi           integer;
declare variable FBobinSayisiOran       Double Precision;
declare variable FBoyaSiparisRezervId   integer;

declare variable XPartiAsamalariId   integer;
declare variable XAsamayaDahil       integer;
declare variable XMakinaPlanlamaGrupId varchar(10);
declare variable XGrupProsesNo         varchar(10);
declare variable XAsamaId              integer;
declare variable XMakinaId             varchar(10);
declare variable XGirisPersonelId      integer;
declare variable XGirisTarihi          TimeStamp;
declare variable XGirisOtomatik        integer;
declare variable XCikisPersonelId      integer;
declare variable XCikisTarihi          TimeStamp;
declare variable XCikisOtomatik        integer;
declare variable XHareketKullaniciId   integer;
declare variable XMakinaSarjId         integer;
declare variable XAraVermeSuresi       Double Precision;

declare variable OrjKilo               Numeric(12,3);

declare variable XPartiAsamaStokDetayId integer;
declare variable XKilo                  Numeric(12,3);
declare variable YKilo                  Numeric(12,3);
declare variable IlkKayit               integer;
declare variable FPartiAsamaStokDetayId integer;
declare variable PartiStokIptalKayitId integer;

declare variable XCikisKilo             Numeric(12,3);
declare variable XCikisMetre            Numeric(12,3);
declare variable FCikisKilo             Numeric(12,3);
declare variable FCikisMetre            Numeric(12,3);
declare variable CikisKiloYuvarlamaCarpani integer;
declare variable CikisMetreYuvarlamaCarpani integer;
declare variable KaliteKontrolAsamaId integer;
declare variable FPartiYil            integer;
declare variable FPartiID             integer;
declare variable BolunenPartiSiraNo   integer;
declare variable BolunenKilo          Double Precision;
declare variable FHamAdi              varchar(255);
declare variable YeniPartiSiraNo      integer;
declare variable YeniKilo             Double Precision;
declare variable Aciklama             varchar(1500);
declare variable IslemKullaniciAdi    varchar(40);
declare variable StrPos SmallInt;
declare variable RafStr varchar(50);
declare variable Temp   varchar(50);
declare variable OtonomTopParcalayabilir EgeBoolean;
begin
  Select Kilo, PartiStokIptalKayitId from PartiStok where PartiStok.PartiSinifId = :APartiSinifId
         into :FEskiSinifKilo, :PartiStokIptalKayitId;

  if (PartiStokIptalKayitId is not null) then exception IptalEdilmisDegistirilemez;

  if (AKilo is null) then exception BolunecekKiloNullOlamaz;
  if (AKilo = 0) then exception BolunecekKiloSifirOlamaz;

  OrjKilo = AKilo;

  Select OtonomTopParcalayabilir from Kullanici where Kullanici.KullaniciID = :AKullaniciId into :OtonomTopParcalayabilir;

  if (OtonomTopParcalayabilir = 0 or AOtonomTopParcalaniyor = 0) then
  begin
    Select count(*) from BoyaIrsaliyeDetay where BoyaIrsaliyeDetay.PartiSinifId = :APartiSinifId into :KayitSayisi;
    if (KayitSayisi = 0) then Select count(*) from PartiStokTarti left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId where PartiStokTarti.PartiSinifId = :APartiSinifId and PartiStok.TamamiTartildi = 1 into :KayitSayisi;

    if (KayitSayisi > 0) then exception SevkEdilmisTartilmisBolunemez;
  end

  if (FEskiSinifKilo < AKilo) then exception BolunecekKiloSinifKilodanBuyuk;

  Select KaliteKontrolAsamaId from Sistem Where SistemId = 1 into :KaliteKontrolAsamaId;

  Select Count(*) from PartiAsamaStok where PartiAsamaStok.PartiSinifId = :APartiSinifId and GirisTarihi is not null and CikisTarihi is null and (PartiAsamaStok.AsamaId <> :KaliteKontrolAsamaId or :KaliteKontrolAsamaId is Null) into :KayitSayisi;

  if (KayitSayisi > 0) then exception ImalatDevamEdiyorBolunemez;

  CikisKiloYuvarlamaCarpani = 100;
  CikisMetreYuvarlamaCarpani = 100;

  FYeniPartiSinifId = gen_id(PartiSinif_Id_gen, 1);

  insert into PartiStok(PartiKayitId, PartiSinifId, KullaniciId, IstenenRaporBoyu, IstenenOzelTuse,
             EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen, CekmezlikBoyIstenen, YikamaHasligiIstenen, PaketlemeSekliId,
             Durumu, BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, BoyaSiparisBaskiId,
             BirimFiyat, DovizId, MamulKodu, FirmaSevkAdresiId, ProsesId, MaximumFireOrani, MinimumFireOrani, LinkIslemi, HacimEn, Gramaj, TasindigiPartiKayitId, BolunduguPartiSinifId,
             FasonBirimFiyati, FasonBirimFiyatDovizID)
     Select :APartiKayitId, :FYeniPartiSinifId, :AKullaniciId, IstenenRaporBoyu, IstenenOzelTuse,
               EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen, CekmezlikBoyIstenen, YikamaHasligiIstenen, PaketlemeSekliId,
               Durumu, BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, BoyaSiparisBaskiId,
               BirimFiyat, DovizId, MamulKodu, FirmaSevkAdresiId, ProsesId, MaximumFireOrani, MinimumFireOrani, 1, HacimEn, Gramaj, TasindigiPartiKayitId, :APartiSinifId,
               FasonBirimFiyati, FasonBirimFiyatDovizID
               from PartiStok where PartiStok.PartiSinifId = :APartiSinifId;

  if (ARafDetayIdler is Null) then
  begin
    For Select RafDetayId, HamId, KapAdedi, Kilo, Metre, BobinSayisi, Adet, PartilemeTarihi, BoyaSiparisRezervId from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId
       into :FRafDetayId, :FHamId, :FKapAdedi, :FKilo, :FMetre, :FBobinSayisi, :FAdet, :FPartilemeTarihi, :FBoyaSiparisRezervId do
    begin
      if (AKilo > 0) then
      begin
        if (AKilo < FKilo) then
        begin
          if (AMetre is Null or AMetre = 0) then
          begin
            FMetreOran = FMetre / FKilo;
            FMetre = AKilo * FMetreOran;
          end
          else FMetre = AMetre;

          FMetre = Yuvarla(FMetre, 1);
          FBobinSayisiOran = FBobinSayisi / FKilo;
          FBobinSayisi = AKilo * FBobinSayisiOran;
          FBobinSayisi = Yuvarla(FBobinSayisi, 1);

          update PartiStokDetay set Kilo = Kilo - :AKilo, Metre = Metre - :FMetre,  BobinSayisi = BobinSayisi - :FBobinSayisi, KullaniciId = :AKullaniciId, PartiStokBolunuyor = 1, LinkIslemi = 1 where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;

          insert into PartiStokDetay(PartiKayitId, PartiSinifId, RafDetayId, HamId, Kilo, Metre, BobinSayisi, KullaniciId, PartilemeTarihi, BoyaSiparisRezervId, LinkIslemi)
                      values(:APartiKayitId, :FYeniPartiSinifId, :FRafDetayId, :FHamId, :AKilo, :FMetre, :FBobinSayisi, :AKullaniciId, :FPartilemeTarihi, :FBoyaSiparisRezervId, 1);

          update PartiStokDetay set KullaniciId = :AKullaniciId, PartiStokBolunuyor = 0, LinkIslemi = 1 where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;
          AKilo = 0;
        end
        else
        if (AKilo >= FKilo) then
        begin
          update PartiStokDetay set LinkIslemi = 2 where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;
          delete from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;

          insert into PartiStokDetay(PartiKayitId, PartiSinifId, RafDetayId, HamId, Kilo, Metre, BobinSayisi, KullaniciId, PartilemeTarihi, BoyaSiparisRezervId,  LinkIslemi)
                      values(:APartiKayitId, :FYeniPartiSinifId, :FRafDetayId, :FHamId, :FKilo, :FMetre, :FBobinSayisi, :AKullaniciId, :FPartilemeTarihi, :FBoyaSiparisRezervId, 1);
          AKilo = AKilo - FKilo;
        end
      end
    end
  end
  else
  begin
    For Select RafDetayId, HamId, KapAdedi, Kilo, Metre, BobinSayisi, Adet, PartilemeTarihi, BoyaSiparisRezervId from PartiStokDetay
        where PartiStokDetay.PartiSinifId = :APartiSinifId and :ARafDetayIdler Containing ','||RafDetayId||','
        into :FRafDetayId, :FHamId, :FKapAdedi, :FKilo, :FMetre, :FBobinSayisi, :FAdet, :FPartilemeTarihi, :FBoyaSiparisRezervId do
    begin
      RafStr = '@' || Cast(FRafDetayId as VarChar(15)) || '=';
      StrPos = Position(RafStr, ARafMiktarlari);
      Temp = Substring(ARafMiktarlari from StrPos + Char_length(RafStr) for 100);
      StrPos = Position('@', Temp);
      Temp = Substring(Temp from 1 for StrPos - 1);
      YeniKilo = Cast(Temp as Double Precision);

      if (YeniKilo > FKilo) then exception BolunecekKiloSinifKilodanBuyuk; /*Eğer Detaydan alınmak istenen kilo detay kilosunda büyükse kızalım*/
      else if (YeniKilo = FKilo) then /*Eğer Detayın tamamı yeni sınıfa aktarılacak ise eskisini silip yeni sinif altına oluşturalım*/
      begin
        update PartiStokDetay set LinkIslemi = 2 where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;
        delete from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;

        insert into PartiStokDetay(PartiKayitId, PartiSinifId, RafDetayId, HamId, Kilo, Metre, BobinSayisi, KullaniciId, PartilemeTarihi, BoyaSiparisRezervId,  LinkIslemi)
                    values(:APartiKayitId, :FYeniPartiSinifId, :FRafDetayId, :FHamId, :FKilo, :FMetre, :FBobinSayisi, :AKullaniciId, :FPartilemeTarihi, :FBoyaSiparisRezervId, 1);
        AKilo = AKilo - YeniKilo;
      end
      else /*Eğer Detayın bir kısmı yeni sınıfa aktarılacak ise detayı bölüp işlem yapalım*/
      begin
        if (AMetre is Null or AMetre = 0) then
        begin
          FMetreOran = FMetre / FKilo;
          FMetre = YeniKilo * FMetreOran;
        end
        else FMetre = AMetre;

        FMetre = Yuvarla(FMetre, 1);
        FBobinSayisiOran = FBobinSayisi / FKilo;
        FBobinSayisi = YeniKilo * FBobinSayisiOran;
        FBobinSayisi = Yuvarla(FBobinSayisi, 1);

        update PartiStokDetay set Kilo = Kilo - :YeniKilo, Metre = Metre - :FMetre,  BobinSayisi = BobinSayisi - :FBobinSayisi, KullaniciId = :AKullaniciId, PartiStokBolunuyor = 1, LinkIslemi = 1 where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;

        insert into PartiStokDetay(PartiKayitId, PartiSinifId, RafDetayId, HamId, Kilo, Metre, BobinSayisi, KullaniciId, PartilemeTarihi, BoyaSiparisRezervId, LinkIslemi)
                    values(:APartiKayitId, :FYeniPartiSinifId, :FRafDetayId, :FHamId, :YeniKilo, :FMetre, :FBobinSayisi, :AKullaniciId, :FPartilemeTarihi, :FBoyaSiparisRezervId, 1);

        update PartiStokDetay set KullaniciId = :AKullaniciId, PartiStokBolunuyor = 0, LinkIslemi = 1 where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;
        AKilo = AKilo - YeniKilo;
      end
    end
  end
  YeniKilo = Null;
  insert into PartiTamirSinif(PartiTamirId, PartiSinifId, Bitti)
          Select PartiTamirId, :FYeniPartiSinifId, Bitti from PartiTamirSinif where PartiTamirSinif.PartiSinifId = :APartiSinifId;

  insert into PartiAsamaStok(PartiKayitId, PartiSinifId, PartiAsamalariId, AsamayaDahil, MakinaPlanlamaGrupId, GrupProsesNo,  AsamaId)
                  Select :APartiKayitId, :FYeniPartiSinifId, PartiAsamaStok.PartiAsamalariId, PartiAsamaStok.AsamayaDahil,
                         PartiAsamaStok.MakinaPlanlamaGrupId, PartiAsamaStok.GrupProsesNo, PartiAsamaStok.AsamaId from PartiAsamaStok
                    where PartiAsamaStok.PartiSinifId = :APartiSinifId;

  for Select PartiAsamaStok.PartiAsamalariId, PartiAsamaStok.AsamayaDahil, PartiAsamaStok.MakinaPlanlamaGrupId, PartiAsamaStok.GrupProsesNo, PartiAsamaStok.AsamaId from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :APartiSinifId
      order by PartiAsamalari.Sira
      into :XPartiAsamalariId, :XAsamayaDahil, :XMakinaPlanlamaGrupId, :XGrupProsesNo, :XAsamaId do
  begin
    YKilo = 0;
    Select PartiAsamaStok.GirisKilo from PartiAsamaStok where PartiAsamaStok.PartiAsamalariId = :XPartiAsamalariId and PartiAsamaStok.PartiSinifID = :FYeniPartiSinifId into :YKilo;

/*    YKilo = OrjKilo; */
    IlkKayit = 1;

    For Select PartiAsamaStokDetayId, MakinaSarjId, Kilo, CikisKilo, CikisMetre, MakinaId, GirisPersonelId, GirisTarihi, GirisOtomatik, CikisPersonelId, CikisTarihi, CikisOtomatik, HareketKullaniciId, AraVermeSuresi from PartiAsamaStokDetay
      where PartiAsamaStokDetay.PartiSinifId = :APartiSinifId and PartiAsamaStokDetay.PartiAsamalariId = :XPartiAsamalariId
      order by PartiAsamaStokDetay.PartiAsamaStokDetayId Desc
      into :XPartiAsamaStokDetayId, :XMakinaSarjId, :XKilo, :XCikisKilo, :XCikisMetre, :XMakinaId, :XGirisPersonelId, :XGirisTarihi, :XGirisOtomatik, :XCikisPersonelId, :XCikisTarihi, :XCikisOtomatik, :XHareketKullaniciId, :XAraVermeSuresi do
    begin
      FCikisKilo   = 0;
      FCikisMetre  = 0;

      if (YKilo > 0) then
      begin
        if (XKilo < 0) then
        begin
          FKilo = Ykilo + XKilo;

          FCikisKilo = XCikisKilo * FKilo / XKilo;
          FCikisMetre = XCikisMetre * FKilo / XKilo;

          FCikisKilo = Yuvarla(FCikisKilo, CikisKiloYuvarlamaCarpani);
          FCikisMetre = Yuvarla(FCikisMetre, CikisMetreYuvarlamaCarpani);

          if (IlkKayit = 1) then
          begin
            Update PartiAsamaStokDetay Set MakinaSarjId = :XMakinaSarjId, Kilo = :FKilo, Miktar = :FKilo, CikisKilo = :FCikisKilo, CikisMetre = :FCikisMetre, MakinaId = :XMakinaId, GirisPersonelId = :XGirisPersonelId, GirisTarihi = :XGirisTarihi, GirisOtomatik = :XGirisOtomatik,
                                                                  CikisPersonelId = :XCikisPersonelId, CikisTarihi = :XCikisTarihi, CikisOtomatik = :XCikisOtomatik,
                                                                  AraVermeSuresi = :XAraVermeSuresi
               where PartiAsamaStokDetay.PartiSinifId = :FYeniPartiSinifId and PartiAsamaStokDetay.PartiAsamalariId = :XPartiAsamalariId;

            IlkKayit = 0;
          end
          else
          begin
            insert into PartiAsamaStokDetay(PartiKayitId, PartiSinifId, MakinaSarjId, PartiAsamalariId, Kilo, Miktar, CikisKilo, CikisMetre, MakinaId, GirisPersonelId, GirisTarihi,
                                       GirisOtomatik, CikisPersonelId, CikisTarihi, CikisOtomatik, AraVermeSuresi, HareketKullaniciId)
               Values(:APartiKayitId, :FYeniPartiSinifId, :XMakinaSarjId, :XPartiAsamalariId, :FKilo, :FKilo, :FCikisKilo, :FCikisMetre, :XMakinaId, :XGirisPersonelId, :XGirisTarihi,
                       :XGirisOtomatik, :XCikisPersonelId, :XCikisTarihi, :XCikisOtomatik, :XAraVermeSuresi, :XHareketKullaniciId);
          end

          Delete from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiAsamaStokDetayID = :XPartiAsamaStokDetayId;

          Select Max(PartiAsamaStokDetayId) from PartiAsamaStokDetay
            where PartiAsamaStokDetay.PartiSinifId = :APartiSinifId and PartiAsamaStokDetay.PartiAsamalariId = :XPartiAsamalariId
           into :FPartiAsamaStokDetayId;

          Update PartiAsamaStokDetay Set Kilo = Kilo + :XKilo, Miktar = Miktar + :XKilo where PartiAsamaStokDetay.PartiAsamaStokDetayId = :FPartiAsamaStokDetayId;

          YKilo = YKilo - FKilo;
        end
        else if (XKilo = 0) then
        begin
          FKilo = YKilo;

          FCikisKilo = XCikisKilo;
          FCikisMetre = XCikisMetre;

          if (IlkKayit = 1) then
          begin
            Update PartiAsamaStokDetay Set MakinaSarjId = :XMakinaSarjId, CikisKilo = :FCikisKilo, CikisMetre = :FCikisMetre, MakinaId = :XMakinaId, GirisPersonelId = :XGirisPersonelId, GirisTarihi = :XGirisTarihi, GirisOtomatik = :XGirisOtomatik,
                                           CikisPersonelId = :XCikisPersonelId, CikisTarihi = :XCikisTarihi, CikisOtomatik = :XCikisOtomatik,
                                           AraVermeSuresi = :XAraVermeSuresi
               where PartiAsamaStokDetay.PartiSinifId = :FYeniPartiSinifId and PartiAsamaStokDetay.PartiAsamalariId = :XPartiAsamalariId;
            IlkKayit = 0;
          end
          else
          begin
            insert into PartiAsamaStokDetay(PartiKayitId, PartiSinifId, MakinaSarjId, PartiAsamalariId, Kilo, Miktar, CikisKilo, CikisMetre, MakinaId, GirisPersonelId, GirisTarihi,
                                       GirisOtomatik, CikisPersonelId, CikisTarihi, CikisOtomatik, HareketKullaniciId, AraVermeSuresi)
               Values(:APartiKayitId, :FYeniPartiSinifId, :XMakinaSarjId, :XPartiAsamalariId, :FKilo, :FKilo, :FCikisKilo, :FCikisMetre, :XMakinaId, :XGirisPersonelId, :XGirisTarihi,
                       :XGirisOtomatik, :XCikisPersonelId, :XCikisTarihi, :XCikisOtomatik, :XHareketKullaniciId, :XAraVermeSuresi);
          end
          Delete from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiAsamaStokDetayID = :XPartiAsamaStokDetayId;
          YKilo = 0;
        end
        else
        begin
          FKilo = YKilo;

          FCikisKilo = XCikisKilo * FKilo / XKilo;
          FCikisMetre = XCikisMetre * FKilo / XKilo;

          FCikisKilo = Yuvarla(FCikisKilo, CikisKiloYuvarlamaCarpani);
          FCikisMetre = Yuvarla(FCikisMetre, CikisMetreYuvarlamaCarpani);

          if (IlkKayit = 1) then
          begin
            Update PartiAsamaStokDetay Set MakinaSarjId = :XMakinaSarjId, CikisKilo = :FCikisKilo, CikisMetre = :FCikisMetre, MakinaId = :XMakinaId, GirisPersonelId = :XGirisPersonelId, GirisTarihi = :XGirisTarihi, GirisOtomatik = :XGirisOtomatik,
                                           CikisPersonelId = :XCikisPersonelId, CikisTarihi = :XCikisTarihi, CikisOtomatik = :XCikisOtomatik,
                                           AraVermeSuresi = :XAraVermeSuresi
               where PartiAsamaStokDetay.PartiSinifId = :FYeniPartiSinifId and PartiAsamaStokDetay.PartiAsamalariId = :XPartiAsamalariId;

            IlkKayit = 0;
          end
          else
          begin
            insert into PartiAsamaStokDetay(PartiKayitId, PartiSinifId, MakinaSarjId, PartiAsamalariId, Kilo, Miktar, CikisKilo, CikisMetre, MakinaId, GirisPersonelId, GirisTarihi,
                                       GirisOtomatik, CikisPersonelId, CikisTarihi, CikisOtomatik, HareketKullaniciId, AraVermeSuresi)
               Values(:APartiKayitId, :FYeniPartiSinifId, :XMakinaSarjId, :XPartiAsamalariId, :FKilo, :FKilo, :FCikisKilo, :FCikisMetre, :XMakinaId, :XGirisPersonelId, :XGirisTarihi,
                       :XGirisOtomatik, :XCikisPersonelId, :XCikisTarihi, :XCikisOtomatik, :XHareketKullaniciId, :XAraVermeSuresi);
          end
          YKilo = 0;
        end
      end
    end
  end

  insert into PartiAsamaAraKontrol(PartiKayitId, PartiAsamalariId, PartiSinifId, AraKontrolTipiId, Sonuc, OnayDurumu, OnayTarihi, OnaylayanPersonelId, PersonelId, DegisimTarihi)
     Select :APartiKayitId, PartiAsamalariId, :FYeniPartiSinifId, AraKontrolTipiId, Sonuc, OnayDurumu, OnayTarihi, OnaylayanPersonelId, PersonelId, DegisimTarihi from PartiAsamaAraKontrol
     where PartiAsamaAraKontrol.PartiSinifId = :APartiSinifId;

  delete from PartiAsamaIcKontrol where PartiAsamaIcKontrol.PartiSinifId = :FYeniPartiSinifId;
