CREATE PROCEDURE SIPARISOZELTESTGUNCELLE(
    ABOYASIPARISDETAYID INTEGER)
AS
declare variable FAdi Adi80;
declare variable FOzelTestler varchar(255);
begin
  FOzelTestler = '';
  For Select OzelTest.Adi from SiparisOzelTestler
    left join OzelTest on OzelTest.OzelTestId = SiparisOzelTestler.OzelTestId
    where SiparisOzelTestler.BoyaSiparisDetayId = :ABoyaSiparisDetayId
    order By OzelTest.Adi
    into :FAdi do
  begin
    if (Char_Length(FOzelTestler) + Char_Length(FAdi) < 250) then
    begin
      if (FOzelTestler <> '') then FOzelTestler = FOzelTestler || cast(', ' as varchar(2) character set UTF8);
      FOzelTestler = FOzelTestler || FAdi;
    end
  end

  Update BoyaSiparisDetay Set OzelTestler = :FOzelTestler where BoyaSiparisDetay.BoyaSiparisDetayID = :ABoyaSiparisDetayID;
end
