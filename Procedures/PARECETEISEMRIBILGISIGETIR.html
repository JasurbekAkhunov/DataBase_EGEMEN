CREATE PROCEDURE PARECETEISEMRIBILGISIGETIR(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER)
RETURNS (
    RECETEISEMRIIDSTR VARCHAR(30),
    BIRLESTIRILENPARTILER VARCHAR(100))
AS
Declare variable FReceteIsEmriYil integer;
Declare variable FReceteIsEmriId  integer;
Declare variable FPartiYil integer;
Declare variable FPartiId  integer;
declare variable FPartiIdStr varchar(30);

begin
  ReceteIsEmriIdStr = Null;
  BirlestirilenPartiler = '';

  Select ReceteIsEmriYil, ReceteIsEmriId from ReceteIsEmriDetay where ReceteIsEmriDetay.PartiAsamalariId = :APartiAsamalariId into :FReceteIsEmriYil, :FReceteIsEmriId;

  if (FReceteIsEmriId is not null) then
  begin
    ReceteIsEmriIdStr = '(' || EgemenIntegerToStr(FReceteIsEmriYil) || ') ' || EgemenIntegerToStr(FReceteIsEmriId) || ' ';

    for select Parti.PartiYil, Parti.PartiId from ReceteIsEmriDetay
      left join Parti on Parti.PartiKayitId = ReceteIsEmriDetay.PartiKayitId
      where ReceteIsEmriDetay.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :FReceteIsEmriId and ReceteIsEmriDetay.PartiKayitId <> :APartiKayitId
      group by Parti.PartiYil, Parti.PartiId
      into :FPartiYil, :FPartiId do
    begin
      FPartiIdStr = '(' || EgemenIntegerToStr(FPartiYil)  || ') ' || EgemenIntegerToStr(FPartiId) || ' ';

      if (Char_Length(BirlestirilenPartiler) + Char_Length(FPartiIdStr) < 90) then
      begin
        if (BirlestirilenPartiler <> '' ) then BirlestirilenPartiler = BirlestirilenPartiler || cast(', ' as varchar(2) character set UTF8);
        BirlestirilenPartiler = BirlestirilenPartiler || FPartiIdStr;
      end
    end
  end

  suspend;
end
