CREATE PROCEDURE REZERVSERTIFIKAICINHAMSTOKLIST(
    AHAMID INTEGER,
    ASERTIFIKANO VARCHAR(30),
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    HAMSTOKHAREKETTIPIID INTEGER,
    HAMSTOKHAREKETTIPIIDSTR VARCHAR(10),
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30),
    TALEPKULLANICIID INTEGER,
    TALEPKULLANICIADI VARCHAR(40),
    SINIF VARCHAR(1),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYENO IRSALIYENOSIZE,
    TARIH TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30),
    URETICIFIRMAADI VARCHAR(100),
    SERTIFIKANO VARCHAR(30),
    SERTIFIKAGIRISMIKTAR BOYAMIKTARSIZE,
    SERTIFIKACIKISMIKTAR BOYAMIKTARSIZE,
    SERTIFIKAKALANMIKTAR BOYAMIKTARSIZE,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    PUS INTEGER,
    FEINE INTEGER,
    EN INTEGER,
    GRM2 INTEGER,
    ORMECIMAKINANO VARCHAR(10),
    RENK VARCHAR(20),
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    HAMKONTROLKULLANIMYERI INTEGER,
    HAMKONTROLKULLANIMYERISTR VARCHAR(20),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    IPLIKUZUNLUGU1 VARCHAR(40),
    IPLIKUZUNLUGU2 VARCHAR(40),
    IPLIKUZUNLUGU3 VARCHAR(40),
    DENYE INTEGER,
    LYCRAURETICIADI VARCHAR(20),
    BRUTNET INTEGER,
    BRUTNETSTR VARCHAR(5),
    BOBINGRAMAJI DOUBLE PRECISION,
    MUSTERIPARTINO VARCHAR(30),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    REFERANSFIRMAID VARCHAR(20),
    REFERANSFIRMAADI VARCHAR(100),
    TAMIRMI INTEGER,
    TAMIR VARCHAR(5),
    TAMIRTIPIADI VARCHAR(20),
    ESKIPARTINO INTEGER,
    KAPADEDI INTEGER,
    MIKTAR NUMERIC(12,3),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    KALANKAPADEDI INTEGER,
    PARTILENEBILIRRAFKILO NUMERIC(12,3),
    RAFIDSTR VARCHAR(50),
    HAMKONTROLNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    YARIMAMUL INTEGER,
    YARIMAMULSTR VARCHAR(20),
    ORMEURETIMPARTINO VARCHAR(30),
    URETICIID INTEGER,
    URETICIADI ADI40,
    ACIKLAMA MEMO,
    YARIMAMULPARTINOSTR VARCHAR(20),
    OLCUID INTEGER,
    OLCUADI ADI40,
    SERTIFIKAGIRISKILO BOYAMIKTARSIZE,
    SERTIFIKACIKISKILO BOYAMIKTARSIZE,
    SERTIFIKAKALANKILO BOYAMIKTARSIZE,
    SERTIFIKAGIRISMETRE BOYAMIKTARSIZE,
    SERTIFIKACIKISMETRE BOYAMIKTARSIZE,
    SERTIFIKAKALANMETRE BOYAMIKTARSIZE)
AS
declare variable IslemTipiId integer;
declare variable itNormal integer;
declare variable itRpt integer;
declare variable itKontrol integer;
declare variable itTamir integer;
declare variable itNumune integer;
declare variable KullanilanKapAdedi integer;
declare variable IptalPartiKilo Numeric(12,3);
declare variable IptalPartiMetre Numeric(12,3);
declare variable IptalPartiMiktar Numeric(12,3);
declare variable IptalPartiAdet integer;
declare variable IptalKapAdedi integer;
declare variable RafKapAdedi integer;
declare variable KapAdediSatis integer;
declare variable YariMamulPartiYil integer;
declare variable YariMamulPartiId integer;
BEGIN
  SELECT itNormal, itRpt, itKontrol, itTamir, itNumune FROM IslemTipiGetir INTO :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  FOR SELECT
         HamStokDetay.HamStokDetayId, HamStokDetay.HamStokMasterId, HamStokIrsaliye.Sinif, HamStokIrsaliye.FirmaId, IrsaliyeFirma.Adi, HamStokIrsaliye.IrsaliyeNo,
         HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, Ormeci.Adi, UreticiFirma.Adi, HamStokDetay.HamId, Ham.Adi, Ham.HamTipiId, HamTipi.Adi,
         HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.En, HamStokMaster.Grm2, HamStokMaster.OrmeciMakinaNo, HamStokMaster.Lot, HamStokMaster.MelanjKodu,
         HamStokMaster.BobinGramaji, HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo, HamStokMaster.ReferansFirmaId,
         ReferansFirma.Adi, HamStokMaster.TupAcikEnMayli, HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3,HamStokMaster.Denye, LycraUretici.Adi,
         HamStokMaster.Renk, HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartiNo, HamStokDetay.KapAdedi,
         HamStokDetay.Miktar, HamStokDetay.Kilo, HamStokDetay.Metre, HamStokDetay.Adet,
         SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId, SatinAlmaTalep.TalepKullaniciId, TalepKullanici.Adi, HamStokIrsaliye.HamStokHareketTipiId, HamStokMaster.HamKontrolKullanimYeri,
         HamStokMaster.Brut, HamStokMaster.HamKontrolNotu, Raf.YariMamul, IIF(OrmePlanlama.UretimPartiNo IS NOT NULL, OrmePlanlama.UretimPartiNo, DokumaPlanlama.UretimPartiNo), Uretici.UreticiId, Uretici.Adi, HamStokMaster.Aciklama,
         Parti.PartiYil, Parti.PartiId, Olcu.OlcuId, Olcu.OlcuAdi, HamStokMaster.SertifikaNo, HamStokDetay.SertifikaGirisMiktar, HamStokDetay.SertifikaCikisMiktar, HamStokDetay.SertifikaGirisMiktar - HamStokDetay.SertifikaCikisMiktar,
         HamStokDetay.SertifikaGirisKilo, HamStokDetay.SertifikaCikisKilo, HamStokDetay.SertifikaGirisKilo - HamStokDetay.SertifikaCikisKilo, HamStokDetay.SertifikaGirisMetre, HamStokDetay.SertifikaCikisMetre, HamStokDetay.SertifikaGirisMetre - HamStokDetay.SertifikaCikisMetre  FROM HamStokMaster
  JOIN HamStokDetay ON HamStokDetay.HamStokMasterId = HamStokMaster.HamStokMasterId
  JOIN HamstokIrsaliye ON hamstokIrsaliye.HamStokIrsaliyeid = HamstokDetay.HamStokIrsaliyeId
  JOIN RafDetay ON RafDetay.HamStokDetayid = HamStokDetay.HamStokDetayid
  LEFT JOIN VerilenSiparisDetay ON VerilenSiparisDetay.VerilenSiparisDetayId = HamstokMaster.VerilenSiparisDetayId
  left join LycraUretici on LycraUretici.LycraUreticiId = HamStokMaster.LycraUreticiId
  LEFT JOIN SatinAlmaTalepDetay ON SatinAlmaTalepDetay.SatinAlmaTalepDetayId = VerilenSiparisDetay.SatinAlmaTalepDetayId
  LEFT JOIN SatinAlmaTalep ON SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
  LEFT JOIN Kullanici TalepKullanici ON TalepKullanici.Kullaniciid = SatinAlmaTalep.TalepKullaniciId
  LEFT JOIN Firma IrsaliyeFirma ON IrsaliyeFirma.FirmaId = HamStokIrsaliye.FirmaId
  LEFT JOIN Firma ReferansFirma ON ReferansFirma.FirmaId = HamStokMaster.ReferansFirmaId
  LEFT JOIN Firma UreticiFirma ON UreticiFirma.FirmaId = HamStokMaster.UreticiFirmaId
  left join Ormeci on Ormeci.OrmeciId = HamstokIrsaliye.OrmeciId
  LEFT JOIN DisTamir ON DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
  LEFT JOIN TamirTipi ON TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  LEFT JOIN ham ON ham.hamid = HamStokMAster.Hamid
  LEFT JOIN HamTipi ON HamTipi.HamTipiid = Ham.HamTipiId
  LEFT JOIN Raf ON Raf.RafId = RafDetay.RafId
  LEFT JOIN OrmePlanlama ON OrmePlanlama.OrmePlanlamaid = HamStokMaster.OrmePlanlamaid
  LEFT JOIN DokumaPlanlama ON DokumaPlanlama.DokumaPlanlamaid = HamStokMaster.DokumaPlanlamaid
  LEFT JOIN Uretici ON Uretici.UreticiId = HamStokMaster.UreticiId
  LEFT JOIN PartiStok ON PartiStok.PartiSinifId = HamStokMaster.PartiSinifId
  LEFT JOIN Parti ON Parti.PartiKayitID = PartiStok.PartiKayitID
  LEFT JOIN Olcu ON Olcu.OlcuId = HamStokMaster.OlcuId
  WHERE
       (coalesce(HamStokMaster.SertifikaNo,'')<>'') and
       (HamStokMaster.HamId = :AHamId OR :AHamId IS NULL) AND
       (HamStokMaster.SertifikaNo = :ASertifikaNo OR :ASertifikaNo IS NULL) AND
       (HamStokMaster.HamStokMasterID = :AHamStokMasterId OR :AHamStokMasterID IS NULL) AND
        HamStokMaster.OnayDurumu = 1 AND HamStokDetay.HamStokKabulTipiId <> 5 and
       ((HamStokDetay.SertifikaGirisMiktar > HamStokDetay.SertifikaCikisMiktar) or
       (HamStokDetay.SertifikaGirisKilo > HamStokDetay.SertifikaCikisKilo) or
       (HamStokDetay.SertifikaGirisMetre > HamStokDetay.SertifikaCikisMetre))
    INTO :HamStokDetayId, :HamStokMasterId, :Sinif, :FirmaId, :FirmaAdi, :IrsaliyeNo,
         :Tarih, :EvrakTipi, :OrmeciAdi, :UreticiFirmaAdi, :HamId, :HamAdi, :HamTipiId, :HamTipiAdi,
         :Pus, :Feine, :En, :Grm2, :OrmeciMakinaNo, :Lot, :MelanjKodu,
         :BobinGramaji, :MusteriPartiNo, :MusteriSiparisNo, :ReferansSiparisNo, :ReferansFirmaId,
         :ReferansFirmaAdi, :TupAcikEnMayli, :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3, :Denye, :LycraUreticiAdi,
         :Renk, :Tamirmi, :TamirTipiAdi, :EskiPartiNo, :KapAdedi,
         :Miktar, :Kilo, :Metre, :Adet,
         :SatinAlmaTalepYil, :SatinAlmaTalepId, :TalepKullaniciId, :TalepKullaniciAdi, :HamStokHareketTipiId, :HamKontrolKullanimYeri,
         :BrutNet, :HamKontrolNotu, :YariMamul, :OrmeUretimPartiNo, :UreticiId, :UreticiAdi, :Aciklama,
         :YariMamulPartiYil, :YariMamulPartiId, :OlcuId, :OlcuAdi, :SertifikaNo, :SertifikaGirisMiktar, :SertifikaCikisMiktar, :SertifikaKalanMiktar,
         :SertifikaGirisKilo, :SertifikaCikisKilo, :SertifikaKalanKilo, :SertifikaGirisMetre, :SertifikaCikisMetre, :SertifikaKalanMetre DO
    BEGIN

    YariMamulPartiNoStr = '';
    IF (YariMamulPartiYil IS NOT NULL) THEN
      YariMamulPartiNoStr = '('||YariMamulPartiYil||') '||YariMamulPartiId;

    IF (YariMamul = 1) THEN YariMamulStr = 'Yarı Mamul';
    ELSE YariMamulStr = 'Ham';

    SELECT SUM(BoyaIrsaliyeDetay.KapAdedi) FROM RafDetay
      LEFT JOIN BoyaIrsaliyeDetay ON BoyaIrsaliyeDetay.RafDetayID =  RafDetay.RafDetayID
      WHERE RafDetay.HamStokDetayID = :HamStokDetayID AND BoyaIrsaliyeDetay.HareketTipiID = 0 INTO :KapAdediSatis;

    IF (RafKapAdedi IS NULL) THEN RafKapAdedi = 0;
    IF (KapAdediSatis IS NULL) THEN KapAdediSatis = 0;


    SELECT SUM(BoyaSiparisRezerv.KullanilanKapAdedi + :RafKapAdedi + :KapAdediSatis) FROM BoyaSiparisRezerv
      WHERE BoyaSiparisRezerv.HamStokDetayID = :HamStokDetayID INTO :KullanilanKapAdedi;

      IF (RafKapAdedi IS NULL) THEN RafKapAdedi = 0;
      IF (KullanilanKapAdedi IS NULL) THEN KullanilanKapAdedi = 0;
      IF (KapAdediSatis IS NULL) THEN KapAdediSatis = 0;

      KalanKapAdedi = RafKapAdedi - KullanilanKapAdedi -  KapAdediSatis;

    IF (Tamirmi = 1) THEN Tamir = 'Tamir';
    ELSE Tamir = '';

    IF (EvrakTipi = 0) THEN EvrakTipiStr = 'Fiş';
    ELSE IF (EvrakTipi = 1) THEN EvrakTipiStr = 'İrsaliye';
    ELSE EvrakTipiStr = 'Hatalı';

    IF (HamKontrolKullanimYeri = 0) THEN HamKontrolKullanimYeriStr = 'Beyazlık';
    ELSE IF (HamKontrolKullanimYeri = 1) THEN HamKontrolKullanimYeriStr = 'Siyah';
    ELSE IF (HamKontrolKullanimYeri = 2) THEN HamKontrolKullanimYeriStr = 'Baskı Altı';
    ELSE HamKontrolKullanimYeriStr = '';

    IF (HamStokHareketTipiId = 0) THEN HamStokHareketTipiIdStr = 'Giriş';
    ELSE IF (HamStokHareketTipiId = 1) THEN HamStokHareketTipiIdStr = 'Satış İade';
    ELSE HamStokHareketTipiIdStr = '';


    IF (BrutNet = 0) THEN BrutNetStr = 'Net';
    ELSE IF (BrutNet = 1) THEN BrutNetStr = 'Brüt';
    ELSE BrutNetStr = '';

    IptalPartiKilo = 0;
    IptalPartiMetre = 0;
    IptalPartiMiktar = 0;
    PartilenebilirRafKilo = 0;

    SELECT SUM(Kilo), SUM(Metre), SUM(Miktar), SUM(Adet), SUM(KapAdedi) FROM RafDetay
      WHERE RafDetay.HamStokDetayID = :HamStokDetayId AND RafDetay.PartiStokIptalKayitId IS NOT NULL
      INTO :IptalPartiKilo, :IptalPartiMetre, :IptalPartiMiktar, :IptalPartiAdet, :IptalKapAdedi;

    IF (IptalPartiKilo IS NULL) THEN IptalPartiKilo = 0;
    IF (IptalPartiMetre IS NULL) THEN IptalPartiMetre = 0;
    IF (IptalPartiMiktar IS NULL) THEN IptalPartiMiktar = 0;
    IF (IptalPartiAdet IS NULL) THEN IptalPartiAdet = 0;
    IF (IptalKapAdedi IS NULL) THEN IptalKapAdedi = 0;


    SatinAlmaTalepIdStr = '';
    IF (SatinAlmaTalepYil IS NOT NULL) THEN SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId)  || ' ';

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    RafIDStr = '';
    PartilenebilirRafKilo = NULL;

    SUSPEND;

  end
end
