CREATE PROCEDURE BOYASIPARISLISTESIORMEDETAYI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASIPARISSEKLI INTEGER,
    AISLEMTIPIID INTEGER,
    ASIPARISTIPIID INTEGER,
    AKAPALI INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AORMESIPARISTIPIID INTEGER,
    ADOKUMASIPARISTIPIID INTEGER,
    AREFERANSSIPARISNO VARCHAR(30))
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    SINIF VARCHAR(1),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    HAMMIKTAR DOUBLE PRECISION,
    TESLIMTARIHI TIMESTAMP,
    REVIZETESLIMTARIHI TIMESTAMP,
    PUS INTEGER,
    FEINE INTEGER,
    KAPALI INTEGER,
    KAPALISTR VARCHAR(4),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    TOPLAMDOKUNAN DOUBLE PRECISION,
    TOPLAMSAKATDOKUNAN DOUBLE PRECISION,
    PLANLANANMIKTAR DOUBLE PRECISION)
AS
begin
  For Select OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.HamId, OrmeSiparisDetay.Sinif,
     Ham.Adi, Ham.Adi2, Ham.HamTipiId, HamTipi.Adi, OrmeSiparisDetay.RenkId, Renk.Adi, OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.Teslimtarihi,
     OrmeSiparisDetay.RevizeTeslimTarihi, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmeSiparisDetay.ArtikelNo,
     OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.Kapali,
     OrmeSiparisDetay.ToplamDokunan, OrmeSiparisDetay.ToplamSakatDokunan, OrmeSiparisDetay.PlanlananMiktar from BoyaSiparisListesi(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :ASiparisSekli, :AIslemTipiId, :ASiparisTipiId, :AKapali, 1, :AMusteriSiparisNo, Null, :AOrmeSiparisTipiId, :ADokumaSiparisTipiId, :AReferansSiparisNo)
  Left Join OrmeSiparisDetay ON OrmeSiparisDetay.BoyaSiparisKayitId = BoyaSiparisListesi.BoyaSiparisKayitId
  Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
  Left Join Renk ON Renk.RenkId = OrmeSiparisDetay.RenkId
  Left Join HamTipi ON HamTipi.HamTipiId = Ham.HamTipiId
  where BoyaSiparisListesi.OrmeSiparisi = 1 and OrmeSiparisDetay.BoyaSiparisKayitId is not null
  order by OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.SiparisSiraNo
  into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :HamId, :Sinif, :HamAdi, :HamAdi2, :HamTipiId, :HamTipiAdi, :RenkId, :RenkAdi,
       :HamMiktar, :Teslimtarihi, :RevizeTeslimTarihi, :Pus, :Feine, :ArtikelNo, :MusteriSiparisNo, :ReferansSiparisNo, :Kapali,
       :ToplamDokunan, :ToplamSakatDokunan, :PlanlananMiktar do
  begin
    KapaliStr = '';
    if (Kapali = 1) then KapaliStr = 'Evet';

    suspend;
  end
end
