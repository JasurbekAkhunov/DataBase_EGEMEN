CREATE PROCEDURE KARISIMICINHARMANLISTESI
RETURNS (
    HARMANKAYITID INTEGER,
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    ADI VARCHAR(50),
    IPLIKISLETMEADI VARCHAR(30),
    TARIHI TIMESTAMP,
    HARMANMIKTARI DOUBLE PRECISION,
    HARMANCINSIID INTEGER,
    HARMANCINSIADI VARCHAR(50),
    HARMANKARISIMID INTEGER,
    HARMANKARISIMADI VARCHAR(20),
    ANALOT VARCHAR(50),
    KAPALI INTEGER,
    KARISIKHARMAN INTEGER)
AS
begin
  For Select Harman.HarmanKayitId, Harman.HarmanYil, Harman.HarmanId, Harman.Adi, Harman.Tarihi, Harman.HarmanCinsiId, HarmanCinsi.Adi,
             Harman.HarmanMiktari, IplikIsletme.Adi, Harman.Kapali,
             Harman.AnaLot, Harman.KarisikHarman, Harman.HarmanKarisimId, HarmanKarisim.Adi from Harman
  Left Join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Harman.HarmanCinsiId
  Left Join HarmanKarisim on HarmanKarisim.HarmanKarisimID = Harman.HarmanKarisimID
  Left Join IplikIsletme on IplikIsletme.IplikIsletmeId = Harman.IplikIsletmeId
  Left Join Kullanici IK on IK.KullaniciId = Harman.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = Harman.KullaniciId
  where Harman.Kapali = 0
  into :HarmanKayitId, :HarmanYil, :HarmanId, :Adi, :Tarihi, :HarmanCinsiId, :HarmanCinsiAdi, :HarmanMiktari,
       :IplikIsletmeAdi,  :Kapali,
       :AnaLot, :KarisikHarman, :HarmanKarisimId, :HarmanKarisimAdi do
  begin
    HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') ' || EgemenIntegerToStr(HarmanId) || ' ';
    suspend;
  end
end
