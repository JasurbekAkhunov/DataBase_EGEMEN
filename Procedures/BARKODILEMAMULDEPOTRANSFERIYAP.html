CREATE PROCEDURE BARKODILEMAMULDEPOTRANSFERIYAP(
    AMAMULDEPOTRANSFERFISIID INTEGER,
    ABARKODID VARCHAR(20),
    ADEPOID VARCHAR(10),
    ARAFID VARCHAR(10),
    AKULLANICIID INTEGER)
AS
declare variable PartiKayitId   integer;
declare variable PartiSinifId   integer;
declare variable PartiStokTartiId       integer;
declare variable BoyaIrsaliyeDetayId    integer;
declare variable HamStokSarfFisiDetayId integer;
declare variable PaketId   integer;
declare variable IK        varchar(2);
declare variable TF        integer;
declare variable DepoId    varchar(10);
declare variable FBarkodID          varchar(30);
declare variable KS Integer;
begin
  if (Char_Length(ABarkodId) < 3) then exception BarkodHatali1;

  IK = EgemenCopyStr(ABarkodId, 1, 2);
  FBarkodID = ABarkodId;

  ABarkodId = EgemenCopyStr(ABarkodId, 3, Char_Length(ABarkodId) - 2);
  TF = EgemenStrToIntOlurmu(ABarkodId);

  if (TF = 0) then exception BarkodHatali1;

  select count(*), Max(PartiStokTartiId) from PartiStokTarti where BarkodId = :FBarkodID into :KS, :PaketId;
  if (KS > 0) then /*iade kaydidir*/
    IK = 'R-';
  else
    PaketId = EgemenStrToInt(ABarkodId);

  if (IK = 'I-' or IK = 'I ') then
  begin
    Select IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, IplikPaketleme.PartiStokTartiId, PartiStokTarti.BoyaIrsaliyeDetayId,
           PartiStokTarti.HamStokSarfFisiDetayId, PartiStokTarti.DepoId from IplikPaketleme
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiID
      where IplikPaketleme.IplikPaketlemeId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, DepoId;
  end
  else
  if (IK = 'R-') then /*iade kaydi*/
  begin
    Select PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId, PartiStokTarti.BoyaIrsaliyeDetayId,
           PartiStokTarti.HamStokSarfFisiDetayId, PartiStokTarti.DepoId from PartiStokTarti
      where PartiStokTarti.PartiStokTartiId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, DepoId;
  end
  else if (IK = 'K-') then
  begin
    Select PsKaliteKontrol.PartiKayitId, PsKaliteKontrol.PartiSinifId, PsKaliteKontrol.PartiStokTartiId, PartiStokTarti.BoyaIrsaliyeDetayId,
           PartiStokTarti.HamStokSarfFisiDetayId, PartiStokTarti.DepoId from PsKaliteKontrol
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = PsKaliteKontrol.PartiStokTartiID
      where PsKaliteKontrol.PsKaliteKontrolId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :DepoId;
  end
  else exception BarkodHatali1;

  if (PartiSinifId is null) then exception BarkodHatali2;
  if (PartiStokTartiId is null) then exception TartiKaydiYokSevkEdilemez;
  if (BoyaIrsaliyeDetayId is not null) then exception BuPaketinIrsaliyesiKesilmis;
  if (HamStokSarfFisiDetayId is not null) then exception BuPaketSarFisiIleCikmis;

  if (DepoId <> :ADepoId) then exception OkutulanPaketinDeposuUygunDegil + ',Deposu :' || Depoid;

  Insert into MamulDepoTransferFisiDetay(MamulDepoTransferFisiId, PartiSinifId, PartiStokTartiId, RafId, InsertKullaniciId, KullaniciId)
              values(:AMamulDepoTransferFisiId, :PartiSinifId, :PartiStokTartiId, :ARafId, :AKullaniciId, :AKullaniciId);
end
