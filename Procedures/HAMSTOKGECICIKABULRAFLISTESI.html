CREATE PROCEDURE HAMSTOKGECICIKABULRAFLISTESI(
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    GECICIKABULRAFID INTEGER,
    HAMTIPIID INTEGER,
    RAFID VARCHAR(10),
    MIKTAR NUMERIC(12,3),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3))
AS
declare variable YuzeyKontroleGiden      Numeric(12,3);
declare variable YuzeyKontroldenGelen    Numeric(12,3);
declare variable KararKilo               Numeric(12,3);
declare variable KararMetre              Numeric(12,3);
declare variable DokumaKumasiKilodanTakipEt integer;
begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  Select Ham.HamTipiId from HamStokMaster
    left join Ham on Ham.HamId = HamStokMaster.HamId
    where HamStokMaster.HamStokMasterID = :AHamStokMasterId into :HamTipiId;

  For Select GeciciKabulRaf.GeciciKabulRafId, GeciciKabulRaf.RafId, GeciciKabulRaf.Miktar, GeciciKabulRaf.KapAdedi,
         GeciciKabulRaf.Kilo, GeciciKabulRaf.Metre, GeciciKabulRaf.Adet,
         GeciciKabulRaf.YuzeyKontroleGiden, GeciciKabulRaf.YuzeyKontroldenGelen, GeciciKabulRaf.KararKilo,
         GeciciKabulRaf.KararMetre from GeciciKabulRaf
    where GeciciKabulRaf.HamStokMasterId = :AHamStokMasterId
    into :GeciciKabulRafId, :RafId, :Miktar, :KapAdedi, :Kilo, :Metre, :Adet, :YuzeyKontroleGiden, :YuzeyKontroldenGelen, :KararKilo,
         :KararMetre do
  begin
    KalanKilo = Kilo - KararKilo;
    KalanMetre = Metre - KararMetre;

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
    begin
      KalanKilo = KalanKilo - YuzeyKontroleGiden + YuzeyKontroldenGelen;
      KalanMiktar = KalanKilo;
    end
    else
    begin
      KalanMetre = KalanMetre - YuzeyKontroleGiden + YuzeyKontroldenGelen;
      KalanMiktar = KalanMetre;
    end
    if (KalanMiktar > 0) then suspend;
  end
end
