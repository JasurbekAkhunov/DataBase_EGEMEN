CREATE PROCEDURE NETSISORMETALIMATBILGISI(
    AORMEPLANLAMAID INTEGER)
RETURNS (
    URETIMPARTINO VARCHAR(30),
    TAHMINIFIREORANI DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5))
AS
begin
  for Select OrmePlanlama.UretimPartiNo, OrmeSiparisDetay.TahminiFireOrani, OrmePlanlama.GerceklesenMiktar, OrmePlanlama.BirimFiyat, OrmePlanlama.DovizId from OrmePlanlama
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = OrmePlanlama.OrmeSiparisDetayId
    where OrmePlanlama.OrmePlanlamaID = :AOrmePlanlamaID
    into :UretimPartiNo, :TahminiFireOrani, :GerceklesenMiktar, :BirimFiyat, :DovizId do suspend;
end
