CREATE PROCEDURE RAFLANMAMISHAMSTOKRAPORU
RETURNS (
    HAMSTOKIRSALIYEID INTEGER,
    HAMSTOKHAREKETTIPIID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    HAMSTOKMASTERID INTEGER,
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    RAFLANANKILO NUMERIC(12,3),
    RAFLANANMETRE NUMERIC(12,3),
    ONAYDURUMU INTEGER,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8))
AS
declare variable Miktar         Numeric(12,3);
declare variable RaflananMiktar Numeric(12,3);
begin
  for select HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.HamStokHareketTipiId, HamStokIrsaliye.EvrakTipi,  HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokMaster.HamStokMasterId, Ham.HamTipiId, Ham.Adi,
             HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Adet,
             HamStokMaster.OnayDurumu, HamStokMaster.Fason, HamStokMaster.RaflananKilo, HamStokMaster.RaflananMetre from HamStokMaster
  left join Ham on Ham.HamId = HamStokMaster.HamId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeid = HamStokMaster.HamStokIrsaliyeId
  left join Firma on Firma.Firmaid = HamStokIrsaliye.Firmaid
  where HamStokMaster.OnaylayanKullaniciId is null and HamStokMaster.Miktar > HamStokMaster.RaflananMiktar
  into :HamStokIrsaliyeId, :HamStokHareketTipiId, :EvrakTipi, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :HamStokMasterId, :HamTipiId, :HamAdi,
       :KapAdedi, :Kilo, :Metre, :Adet, :Fason, :RaflananKilo, :RaflananMetre, :OnayDurumu do
  begin
    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';
    suspend;
  end

  for select HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.HamStokHareketTipiId, HamStokIrsaliye.EvrakTipi, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokMaster.HamStokMasterId, Ham.HamTipiId, Ham.Adi,
             HamStokMaster.KapAdedi, HamStokMaster.Miktar, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Adet,
             HamStokMaster.OnayDurumu, HamStokMaster.Fason, Sum(HamStokDetay.RaflananKilo), Sum(HamStokDetay.RaflananMetre), Sum(HamStokDetay.RaflananMiktar)  from HamStokMaster
  left join Ham on Ham.HamId = HamStokMaster.HamId
  left join HamStokDetay on HamStokDetay.HamStokMasterId = HamStokMaster.HamStokMasterId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeid = HamStokMaster.HamStokIrsaliyeId
  left join Firma on Firma.Firmaid = HamStokIrsaliye.Firmaid
  where HamStokMaster.OnaylayanKullaniciId is not null
  group by HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.HamStokHareketTipiId, HamStokIrsaliye.EvrakTipi, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo,
           HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokMaster.HamStokMasterId, Ham.HamTipiId, Ham.Adi,
           HamStokMaster.KapAdedi, HamStokMaster.Miktar, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Adet,
           HamStokMaster.OnayDurumu, HamStokMaster.Fason
  order by HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.FirmaId, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokMaster.HamStokMasterId
  into :HamStokIrsaliyeId, :HamStokHareketTipiId,  :EvrakTipi, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :HamStokMasterId, :HamTipiId, :HamAdi,
       :KapAdedi, :Miktar, :Kilo, :Metre, :Adet, :Fason, :OnayDurumu, :RaflananKilo, :RaflananMetre, :RaflananMiktar do
  begin
    if (Miktar > RaflananMiktar) then
    begin
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
      else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
      else EvrakTipiStr = 'Hatalı';
      suspend;
    end
  end
end
