CREATE PROCEDURE URUNSARFIYATFIYATLARIDUZENLE(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AURUNID VARCHAR(20))
RETURNS (
    STOKDETAYID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    FIFOORTALAMAMALIYET DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    URETIMECIKISTARIHI TIMESTAMP,
    EVRAKTARIHI TIMESTAMP,
    GUNCELLENENKAYITSAYISI INTEGER,
    SONUCIKISFIYATI DOUBLE PRECISION)
AS
declare variable StokTabanDovizId      varchar(5);
declare variable DefaultBirimCarpan     double precision;
declare variable StokFIFOMaliyetBaslangicTarihi timestamp;
declare variable HareketTipi integer;

begin
  Select StokTabanDovizId from Sistem where SistemId = 1 into :StokTabanDovizId;
  if (StokFifoMaliyetBaslangicTarihi is null) then Select Sistem.StokFifoMaliyetBaslangicTarihi from Sistem where Sistem.SistemId = 1 into :StokFifoMaliyetBaslangicTarihi ;

  For Select Urun.UrunId, Urun.Adi, UBD.Carpan from Urun
    Left Join UrunBirim UBD on UBD.UrunId = Urun.Urunid and UBD.DefaultBirim = 1
    Where (:AUrunId is Null or Urun.UrunId = :AUrunId)
    into :UrunId, :UrunAdi, :DefaultBirimCarpan do
  begin

    GuncellenenKayitSayisi = 0;
    FifoOrtalamaMaliyet = 0;

    For Select StokDetayId, HareketTipi, StokTabanDovizOrtalamaMaliyet, UdbMiktar, UretimeCikisTarihi, Tarih
        From UrunFifoBirimMaliyetHesapla(:UrunId, :StokFifoMaliyetBaslangicTarihi, :ABaslangicTarihi, :ABitisTarihi, null, :DefaultBirimCarpan, :StokTabanDovizId)
        Into :StokDetayId, :HareketTipi, :FifoOrtalamaMaliyet, :Miktar, :UretimeCikisTarihi, :EvrakTarihi  do
    begin
        GuncellenenKayitSayisi = GuncellenenKayitSayisi + 1 ;
        Suspend;
    end
  end
end
