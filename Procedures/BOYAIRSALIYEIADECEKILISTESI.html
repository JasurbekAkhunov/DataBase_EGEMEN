CREATE PROCEDURE BOYAIRSALIYEIADECEKILISTESI(
    ABOYAIRSALIYEID INTEGER)
RETURNS (
    BOYAIRSALIYEDETAYID INTEGER,
    IPLIKPAKETLEMEID INTEGER,
    PSKALITEKONTROLID INTEGER,
    BARKODID VARCHAR(20),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    KOLINO INTEGER,
    KAPNO INTEGER,
    KALITENO INTEGER,
    EN VARCHAR(10),
    GRAMAJ VARCHAR(10),
    DARA NUMERIC(12,3),
    AMBALAJAGIRLIGI NUMERIC(12,3),
    BOBINSAYISI INTEGER,
    BOBINDARASI NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    HACIMEN INTEGER,
    HACIMBOY INTEGER,
    HACIMYUKSEKLIK INTEGER,
    HACIM DOUBLE PRECISION,
    PALETNO VARCHAR(5),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIIDSTR VARCHAR(20),
    ACIKLAMA VARCHAR(50),
    PARTIRENKLOTNO INTEGER,
    TARTIBRUTKILO BOYAMIKTARSIZE)
AS
declare variable PartiStokTartiDara BoyaMiktarSize;
declare variable PartiStokTartiKilo BoyaMiktarSize;
begin
  For select IplikPaketleme.IplikPaketlemeId, IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PartiStokTarti.IadeBoyaIrsaliyeDetayId, PartiStokTarti.KaliteNo, IplikPaketleme.KoliNo,
             PartiStokTarti.Dara, PartiStokTarti.BobinSayisi, IplikPaketleme.BobinDarasi,
             IplikPaketleme.AmbalajAgirligi, PartiStokTarti.BrutKilo,
             PartiStokTarti.Kilo, PartiStokTarti.PaletNo,
             IplikPaketleme.AmbalajId, Ambalaj.Adi, IplikPaketleme.KonikCinsiId, KonikCinsi.Adi,
             LabRecete.RenkDerinligiId, RenkDerinligi.Adi, PartiStokTarti.BarkodID, PartiStokTarti.Dara, PartiStokTarti.Kilo from PartiStokTarti
  left join IplikPaketleme on IplikPaketleme.IplikPaketlemeID = EgemenCopyStr(PartiStokTarti.BarkodID, 3, Char_Length(PartiStokTarti.BarkodID) - 2)
  Left Join Ambalaj on Ambalaj.AmbalajId = IplikPaketleme.AmbalajId
  Left Join KonikCinsi on KonikCinsi.KonikCinsiId = IplikPaketleme.KonikCinsiId
  left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
  where (PartiStokTarti.IadeBoyaIrsaliyeId = :ABoyaIrsaliyeId) and (PartiStokTarti.IadeBoyaIrsaliyeId is not Null) and
        EgemenCopyStr(PartiStokTarti.BarkodID, 1, 1) <> 'R' and EgemenCopyStr(PartiStokTarti.BarkodID, 1, 1) <> 'K'
  into :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :KaliteNo, :KoliNo,
       :Dara, :BobinSayisi, :BobinDarasi, :AmbalajAgirligi,
       :BrutKilo, :NetKilo, :PaletNo, :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :RenkDerinligiId, :RenkDerinligiIdStr, :BarkodID, :PartiStokTartiDara, :PartiStokTartiKilo do
  begin
    Hacim = 0;
    TartiBrutKilo = PartiStokTartiKilo + PartiStokTartiDara;
    suspend;
  end

  For select null, PsKaliteKontrol.PartiKayitId, PsKaliteKontrol.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PartiStokTarti.IadeBoyaIrsaliyeDetayId, PartiStokTarti.KaliteNo, null,
             PsKaliteKontrol.Dara, PartiStokTarti.BobinSayisi, null,
             null, PartiStokTarti.BrutKilo, PartiStokTarti.Kilo, PartiStokTarti.PaletNo,
             null, null, null, null, LabRecete.RenkDerinligiId, RenkDerinligi.Adi, PartiStokTarti.BarkodID, PartiStokTarti.Dara, PartiStokTarti.Kilo, PsKaliteKontrol.Metre,
             PsKaliteKontrol.PsKaliteKontrolId, PsKaliteKontrol.Topno from PartiStokTarti
  left join PsKaliteKontrol on PsKaliteKontrol.PsKaliteKontrolID = EgemenCopyStr(PartiStokTarti.BarkodID, 3, Char_Length(PartiStokTarti.BarkodID) - 2)
  left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
  where (PartiStokTarti.IadeBoyaIrsaliyeId = :ABoyaIrsaliyeId) and (PartiStokTarti.IadeBoyaIrsaliyeId is not Null) and
        EgemenCopyStr(PartiStokTarti.BarkodID, 1, 1) = 'K'
  into :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :KaliteNo, :KoliNo,
       :Dara, :BobinSayisi, :BobinDarasi, :AmbalajAgirligi,
       :BrutKilo, :NetKilo, :PaletNo, :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :RenkDerinligiId, :RenkDerinligiIdStr, :BarkodID, :PartiStokTartiDara, :PartiStokTartiKilo, :Metre,
       :PsKaliteKontrolId, :Kapno do
  begin
    Hacim = 0;
    TartiBrutKilo = PartiStokTartiKilo + PartiStokTartiDara;
    suspend;
  end
  Kapno = null;
  For select Null, PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PartiStokTarti.IadeBoyaIrsaliyeDetayId, PartiStokTarti.KaliteNo, Null,
             PartiStokTarti.Dara, PartiStokTarti.BobinSayisi, Null,
             Null, PartiStokTarti.BrutKilo,
             PartiStokTarti.Kilo, PartiStokTarti.PaletNo,
             Null, Null, Null, Null,
             LabRecete.RenkDerinligiId, RenkDerinligi.Adi, PartiStokTarti.BarkodID, PartiStokTarti.Metre, null from PartiStokTarti
  left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
  where (PartiStokTarti.IadeBoyaIrsaliyeId = :ABoyaIrsaliyeId) and (PartiStokTarti.IadeBoyaIrsaliyeId is not Null) and
        EgemenCopyStr(PartiStokTarti.BarkodID, 1, 1) = 'R'
  into :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :KaliteNo, :KoliNo,
       :Dara, :BobinSayisi, :BobinDarasi, :AmbalajAgirligi,
       :BrutKilo, :NetKilo, :PaletNo, :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :RenkDerinligiId, :RenkDerinligiIdStr, :BarkodID, Metre,
       :PsKaliteKontrolId do
  begin
    Hacim = 0;
    TartiBrutKilo = NetKilo + Coalesce(Dara,0);
    suspend;
  end
end
