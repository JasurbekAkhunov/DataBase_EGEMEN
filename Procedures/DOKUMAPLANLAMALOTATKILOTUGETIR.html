CREATE PROCEDURE DOKUMAPLANLAMALOTATKILOTUGETIR(
    ADOKUMAPLANLAMAID INTEGER)
RETURNS (
    LOT VARCHAR(4000),
    COZGULOTU VARCHAR(4000),
    ATKILOTU VARCHAR(4000))
AS
declare variable FAtkiCozgu SmallInt;
declare variable FLot      varchar(30);
begin
  for select DokumaPlanlamaRecete.AtkiCozgu, DokumaPlanlamaRecete.Lot
    from DokumaPlanlamaRecete
    where DokumaPlanlamaRecete.DokumaPlanlamaId = :ADokumaPlanlamaId
    order by DokumaPlanlamaRecete.AtkiCozgu, DokumaPlanlamaRecete.SiraNo
    into :FAtkiCozgu, :FLot do
  begin
    if (FLot is not null and FLot <> '')  then
    begin
      if (FAtkiCozgu = 2) then
      begin
        if (CozguLotu is null) then CozguLotu = '' ;
        if (Char_Length(CozguLotu) > 0 ) then CozguLotu = CozguLotu || '+' || FLot;
        else
        CozguLotu = FLot ;
      end
      else if (FAtkiCozgu = 1) then
      begin
        if (AtkiLotu is null) then AtkiLotu = '' ;
        if (Char_Length(AtkiLotu) > 0 ) then AtkiLotu = AtkiLotu || '+' || Flot;
        else
        AtkiLotu = FLot ;
      end
    end
  end

  Lot = 'Ã‡: '||Coalesce(Replace(CozguLotu,'+',', '),'')||' + A: '||Coalesce(Replace(AtkiLotu,'+',', '),'');

  suspend;
end
