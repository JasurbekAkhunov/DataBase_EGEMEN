CREATE PROCEDURE ENERJISARFIYATRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    MAKINAID VARCHAR(10),
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    BOYAHANEBOLUMID INTEGER,
    URETIMTOPLAMI NUMERIC(12,3),
    ICTAMIRTOPLAMI NUMERIC(12,3),
    DISTAMIRTOPLAMI NUMERIC(12,3),
    NETURETIMTOPLAMI NUMERIC(12,3),
    IMALATSURESI DOUBLE PRECISION,
    IMALATSURESISAAT DOUBLE PRECISION,
    KILOWATT DOUBLE PRECISION,
    DOGALGAZM3 DOUBLE PRECISION,
    TOPLAMDOGALGAZM3 DOUBLE PRECISION,
    TOPLAMDIREKTDOGALGAZ DOUBLE PRECISION,
    KILODIREKTDOGALGAZ DOUBLE PRECISION,
    TOPLAMDIREKTELEKTIRIK DOUBLE PRECISION,
    KILODIREKTELEKTIRIK DOUBLE PRECISION)
AS
declare variable EBirimFiyat            Double Precision;
declare variable GBirimFiyat            Double Precision;
declare variable Yil                    integer;
declare variable Ay                     integer;
declare variable FDate                  Date;
begin
  Yil = f_Year(ABaslangicTarihi);
  Ay  = f_Month(ABaslangicTarihi);

  Select ElektrikBirimFiyati, DogalGazBirimFiyati from AylikMasraf where AylikMasraf.Yil = :Yil and AylikMasraf.Ay = :Ay into :EBirimFiyat, :GBirimFiyat;

  For Select Makina.MakinaId, Makina.KiloWatt, Makina.DogalGazM3, Makina.MakinaPlanlamaGrupId, MakinaPlanlamaGrup.BoyahaneBolumId from Makina
     left join MakinaPlanlamaGrup on MakinaPlanlamaGrup.MakinaPlanlamaGrupID = Makina.MakinaPlanlamaGrupId
     where makina.UretimRaporlarinaDahil = 1
     order by MakinaPlanlamaGrup.BoyahaneBolumId, Makina.MakinaPlanlamaGrupId, Makina.MakinaId
     into :MakinaId, :KiloWatt, :DogalGazM3, :MakinaPlanlamaGrupId, :BoyahaneBolumId do
  begin
    Select Sum(UretimToplami), Sum(IcTamirToplami), Sum(DisTamirToplami), Sum(ImalatSuresi) from EnerjiSarfiyatRaporuHazirla(:ABaslangictarihi, :ABitisTarihi, :MakinaId)
       into :UretimToplami, :IcTamirToplami, :DisTamirToplami, :ImalatSuresi;

    if (UretimToplami > 0) then
    begin
      NetUretimToplami = UretimToplami - IcTamirToplami - DisTamirToplami;

      ImalatSuresiSaat = ImalatSuresi * 24;
      ImalatSuresiSaat = Yuvarla(ImalatSuresiSaat, 100);

      ToplamDirektElektirik = ImalatSuresiSaat * KiloWatt * EBirimFiyat;
      KiloDirektElektirik = ToplamDirektElektirik / UretimToplami;

      ToplamDogalgazM3 = ImalatSuresiSaat * DogalGazM3;
      ToplamDirektDogalgaz = ToplamDogalgazM3 * GBirimFiyat;
      KiloDirektDogalGaz = ToplamDirektDogalgaz / UretimToplami;
      suspend;
    end
  end
end
