CREATE PROCEDURE URETIMPANOHEDEFLERIGETIR(
    APANONO INTEGER,
    ATARIH TIMESTAMP Default Null)
RETURNS (
    PANONO INTEGER,
    GUNLUK DOUBLE PRECISION,
    HAFTALIK DOUBLE PRECISION,
    AYLIK DOUBLE PRECISION,
    URETIMGUNLUK DOUBLE PRECISION,
    URETIMHAFTALIK DOUBLE PRECISION,
    URETIMAYLIK DOUBLE PRECISION,
    SEVKGUNLUK DOUBLE PRECISION,
    SEVKHAFTALIK DOUBLE PRECISION,
    SEVKAYLIK DOUBLE PRECISION,
    SEVKGUNLUKFASON DOUBLE PRECISION,
    SEVKHAFTALIKFASON DOUBLE PRECISION,
    SEVKAYLIKFASON DOUBLE PRECISION,
    SEVKGUNLUKSATIS DOUBLE PRECISION,
    SEVKHAFTALIKSATIS DOUBLE PRECISION,
    SEVKAYLIKSATIS DOUBLE PRECISION,
    PARTILEMEGUNLUK DOUBLE PRECISION,
    PARTILEMEHAFTALIK DOUBLE PRECISION,
    PARTILEMEAYLIK DOUBLE PRECISION,
    PARTILEMEGUNLUKFASON DOUBLE PRECISION,
    PARTILEMEHAFTALIKFASON DOUBLE PRECISION,
    PARTILEMEAYLIKFASON DOUBLE PRECISION,
    PARTILEMEGUNLUKSATIS DOUBLE PRECISION,
    PARTILEMEHAFTALIKSATIS DOUBLE PRECISION,
    PARTILEMEAYLIKSATIS DOUBLE PRECISION,
    SIPARISGUNLUK DOUBLE PRECISION,
    SIPARISHAFTALIK DOUBLE PRECISION,
    SIPARISAYLIK DOUBLE PRECISION,
    SIPARISGUNLUKFASON DOUBLE PRECISION,
    SIPARISHAFTALIKFASON DOUBLE PRECISION,
    SIPARISAYLIKFASON DOUBLE PRECISION,
    SIPARISGUNLUKSATIS DOUBLE PRECISION,
    SIPARISHAFTALIKSATIS DOUBLE PRECISION,
    SIPARISAYLIKSATIS DOUBLE PRECISION)
AS
declare variable KS       integer;
declare variable GunStart timestamp;
declare variable GunEnd   timestamp;
declare variable HaftaStart timestamp;
declare variable HaftaEnd timestamp;
declare variable AyStart  timestamp;
declare variable AyEnd    timestamp;
declare variable VardiyaBaslangicSaati Double Precision;
declare variable HaftaninGunu integer;
declare variable AyinGunu integer;
declare variable Bugun    Double Precision;
declare variable BuSaat   Double Precision;
begin
  if (ATarih is null) then
    ATarih = current_timestamp;

  Select TimeStampToDouble(TimeStampToTime(VardiyaBaslangicSaati)) from sistem into :VardiyaBaslangicSaati;
  HaftaninGunu = F_DAYOFWEEK(ATarih);
  HaftaninGunu = HaftaninGunu - 1;
  AyinGunu = F_DAYOFMONTH(ATarih);
  Bugun = TimeStampToDouble(cast(ATarih as date));
  BuSaat = TimeStampToDouble(ATarih);

  if ((BuSaat) > (Bugun + VardiyaBaslangicSaati)) then
  begin
    GunStart = DoubleToTimeStamp(Bugun + VardiyaBaslangicSaati);
    GunEnd = DoubleToTimeStamp((Bugun + 1) + VardiyaBaslangicSaati);

    if (HaftaninGunu = 1) then
      HaftaStart = DoubleToTimeStamp(Bugun + VardiyaBaslangicSaati); /*bugun pazartesi ve var. baslangici gecmis*/
    else
      HaftaStart = DoubleToTimeStamp((Bugun - (HaftaninGunu - 1)) + VardiyaBaslangicSaati); /*bugun pazartesi sonrasi ve var. baslangici gecmis*/

    if (AyinGunu = 1) then
      AyStart = DoubleToTimeStamp(Bugun + VardiyaBaslangicSaati); /*bugun ayin biri ve var. baslangici gecmis*/
    else
      AyStart = DoubleToTimeStamp((Bugun - (AyinGunu - 1)) + VardiyaBaslangicSaati); /*bugun ayin biri sonrasi ve var. baslangici gecmis*/
  end
  else
  begin
    GunStart = DoubleToTimeStamp((Bugun - 1) + VardiyaBaslangicSaati);
    GunEnd = DoubleToTimeStamp(Bugun + VardiyaBaslangicSaati);

    if (HaftaninGunu = 1) then
      HaftaStart = DoubleToTimeStamp((Bugun - 7) + VardiyaBaslangicSaati); /*bugun pazartesi ve var. baslangici gecmemis*/
    else
      HaftaStart = DoubleToTimeStamp((Bugun - (HaftaninGunu - 1)) + VardiyaBaslangicSaati); /*bugun pazartesi sonrasi ve var. baslangici gecmemis*/

    if (AyinGunu = 1) then
      AyStart = DoubleToTimeStamp(TimeStampToDouble(F_ADDMONTH(DoubleToTimeStamp(Bugun), -1) + VardiyaBaslangicSaati)); /*bugun ayin biri ve var. baslangici gecmemis*/
    else
      AyStart = DoubleToTimeStamp((Bugun - (AyinGunu - 1)) + VardiyaBaslangicSaati); /*bugun ayin biri sonrasi ve var. baslangici gecmemis*/
  end
  HaftaEnd = (HaftaStart + 7);
  AyEnd = F_ADDMONTH(AyStart, 1);

  /*sevk miktarlari*/
  Select Sum(BoyaIrsaliyeDetay.Miktar), Sum(iif(Parti.Fason = 1, BoyaIrsaliyeDetay.Miktar, 0)), Sum(iif(Parti.Fason = 1, 0, BoyaIrsaliyeDetay.Miktar)) from BoyaIrsaliyeDetay
  left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
  where FasonHamIade = 0 and HareketTipiID = 0 and IrsaliyeTarihi = cast(:ATarih as date) into :SevkGunluk, :SevkGunlukFason, :SevkGunlukSatis;

  Select Sum(BoyaIrsaliyeDetay.Miktar), Sum(iif(Parti.Fason = 1, BoyaIrsaliyeDetay.Miktar, 0)), Sum(iif(Parti.Fason = 1, 0, BoyaIrsaliyeDetay.Miktar)) from BoyaIrsaliyeDetay
  left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
  where FasonHamIade = 0 and HareketTipiID = 0 and  IrsaliyeTarihi >= :HaftaStart - 1 and IrsaliyeTarihi <= :HaftaEnd into :SevkHaftalik, :SevkHaftalikFason, :SevkHaftalikSatis;

  Select Sum(BoyaIrsaliyeDetay.Miktar), Sum(iif(Parti.Fason = 1, BoyaIrsaliyeDetay.Miktar, 0)), Sum(iif(Parti.Fason = 1, 0, BoyaIrsaliyeDetay.Miktar)) from BoyaIrsaliyeDetay
  left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
  where FasonHamIade = 0 and HareketTipiID = 0 and  IrsaliyeTarihi >= :AyStart  - 1  and IrsaliyeTarihi <= :AyEnd into :SevkAylik, :SevkAylikFason, :SevkAylikSatis;

  /*siparis miktarlari*/
  select Sum(BoyaSiparisDetay.SiparisKilo), Sum(iif(BoyaSiparis.Fason = 1, BoyaSiparisDetay.SiparisKilo, 0)), Sum(iif(BoyaSiparis.Fason = 1, 0, BoyaSiparisDetay.SiparisKilo)) from BoyaSiparis
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId where BoyaSiparis.SiparisTarihi = cast(:ATarih as date)
  into :SiparisGunluk, :SiparisGunlukFason, :SiparisGunlukSatis;

  select Sum(BoyaSiparisDetay.SiparisKilo), Sum(iif(BoyaSiparis.Fason = 1, BoyaSiparisDetay.SiparisKilo, 0)), Sum(iif(BoyaSiparis.Fason = 1, 0, BoyaSiparisDetay.SiparisKilo)) from BoyaSiparis
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId where BoyaSiparis.SiparisTarihi >= :HaftaStart - 1 and BoyaSiparis.SiparisTarihi <= :HaftaEnd
  into :SiparisHaftalik, :SiparisHaftalikFason, :SiparisHaftalikSatis;

  select Sum(BoyaSiparisDetay.SiparisKilo), Sum(iif(BoyaSiparis.Fason = 1, BoyaSiparisDetay.SiparisKilo, 0)), Sum(iif(BoyaSiparis.Fason = 1, 0, BoyaSiparisDetay.SiparisKilo)) from BoyaSiparis
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId where BoyaSiparis.SiparisTarihi >= :AyStart  - 1  and BoyaSiparis.SiparisTarihi <= :AyEnd
  into :SiparisAylik, :SiparisAylikFason, :SiparisAylikSatis;

  /*parti miktarlari*/
  select Sum(PartiStok.Miktar), Sum(iif(Parti.Fason = 1, PartiStok.Miktar, 0)), Sum(iif(Parti.Fason = 1, 0, PartiStok.Miktar)) from Parti
  left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId where Parti.AcilisTarihi = cast(:ATarih as date)
  into :PartilemeGunluk, :PartilemeGunlukFason, :PartilemeGunlukSatis;

  select Sum(PartiStok.Miktar), Sum(iif(Parti.Fason = 1, PartiStok.Miktar, 0)), Sum(iif(Parti.Fason = 1, 0, PartiStok.Miktar)) from Parti
  left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId where Parti.AcilisTarihi >= :HaftaStart - 1 and Parti.AcilisTarihi <= :HaftaEnd
  into :PartilemeHaftalik, :PartilemeHaftalikFason, :PartilemeHaftalikSatis;

  select Sum(PartiStok.Miktar), Sum(iif(Parti.Fason = 1, PartiStok.Miktar, 0)), Sum(iif(Parti.Fason = 1, 0, PartiStok.Miktar)) from Parti
  left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId where Parti.AcilisTarihi >= :AyStart  - 1  and Parti.AcilisTarihi <= :AyEnd
  into :PartilemeAylik, :PartilemeAylikFason, :PartilemeAylikSatis;


  for Select Makina.UretimPanoNo from Makina
    where (Makina.UretimPanoNo = :APanoNo or :APanoNo is Null)
    group by Makina.UretimPanoNo
    into :PanoNo do
  begin
    if (PanoNo is not Null) then
    begin
      Gunluk = 0.00;
      Haftalik = 0.00;
      Aylik = 0.00;
      UretimGunluk = 0.00;
      UretimHaftalik = 0.00;
      UretimAylik = 0.00;
      Select count(*) from UretimPanoHedefleri
        where UretimPanoHedefleri.PanoNo = :PanoNo
        into :KS;

      if (KS = 0) then insert into UretimPanoHedefleri(PanoNo, Gunluk, Haftalik, Aylik) values(:PanoNo, 0, 0, 0);

      Select first 1 UretimPanoHedefleri.Gunluk, UretimPanoHedefleri.Haftalik, UretimPanoHedefleri.Aylik from UretimPanoHedefleri
        where UretimPanoHedefleri.PanoNo = :PanoNo
        into :Gunluk, :Haftalik, :Aylik;

      /*uretim miktarlari*/
      select Sum(PartiAsamaStokDetay.Kilo) from makina
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.MakinaId = Makina.MakinaId and
        PartiAsamaStokDetay.CikisTarihi >= :GunStart and PartiAsamaStokDetay.CikisTarihi <= :GunEnd
        where (Makina.UretimPanoNo = :PanoNo) and Makina.UretimRaporlarinaDahil = 1 and
              PartiAsamaStokDetay.CikisTarihi is not null
        into :UretimGunluk;

      select Sum(PartiAsamaStokDetay.Kilo) from makina
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.MakinaId = Makina.MakinaId and
        PartiAsamaStokDetay.CikisTarihi >= :HaftaStart and PartiAsamaStokDetay.CikisTarihi <= :HaftaEnd
        where (Makina.UretimPanoNo = :PanoNo) and Makina.UretimRaporlarinaDahil = 1 and
              PartiAsamaStokDetay.CikisTarihi is not null
        into :UretimHaftalik;

      select Sum(PartiAsamaStokDetay.Kilo) from makina
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.MakinaId = Makina.MakinaId and
        PartiAsamaStokDetay.CikisTarihi >= :AyStart and PartiAsamaStokDetay.CikisTarihi <= :AyEnd
        where (Makina.UretimPanoNo = :PanoNo) and Makina.UretimRaporlarinaDahil = 1 and
              PartiAsamaStokDetay.CikisTarihi is not null
        into :UretimAylik;


      /*partileme miktarlari eski mantık. yenisi: açılan parti miktarı olmali
      select Sum(PartiAsamaStokDetay.Kilo), Sum(iif(Parti.Fason = 1, PartiAsamaStokDetay.Kilo, 0)), Sum(iif(Parti.Fason = 1, 0, PartiAsamaStokDetay.Kilo)) from makina
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.MakinaId = Makina.MakinaId and  PartiAsamaStokDetay.CikisTarihi >= :GunStart and PartiAsamaStokDetay.CikisTarihi <= :GunEnd
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
        left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
        where (Makina.UretimPanoNo = :PanoNo) and Makina.UretimRaporlarinaDahil = 1 and
              PartiAsamaStokDetay.CikisTarihi is not null and
              Asama.PartilemeMi = 1
        into :PartilemeGunluk, :PartilemeGunlukFason, :PartilemeGunlukSatis;

      select Sum(PartiAsamaStokDetay.Kilo), Sum(iif(Parti.Fason = 1, PartiAsamaStokDetay.Kilo, 0)), Sum(iif(Parti.Fason = 1, 0, PartiAsamaStokDetay.Kilo)) from makina
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.MakinaId = Makina.MakinaId and PartiAsamaStokDetay.CikisTarihi >= :HaftaStart and PartiAsamaStokDetay.CikisTarihi <= :HaftaEnd
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
        left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
        where (Makina.UretimPanoNo = :PanoNo) and Makina.UretimRaporlarinaDahil = 1 and
              PartiAsamaStokDetay.CikisTarihi is not null and
              Asama.PartilemeMi = 1
        into :PartilemeHaftalik, :PartilemeHaftalikFason, :PartilemeHaftalikSatis;

      select Sum(PartiAsamaStokDetay.Kilo), Sum(iif(Parti.Fason = 1, PartiAsamaStokDetay.Kilo, 0)), Sum(iif(Parti.Fason = 1, 0, PartiAsamaStokDetay.Kilo)) from makina
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.MakinaId = Makina.MakinaId and PartiAsamaStokDetay.CikisTarihi >= :AyStart and PartiAsamaStokDetay.CikisTarihi <= :AyEnd
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
        left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
        where (Makina.UretimPanoNo = :PanoNo) and Makina.UretimRaporlarinaDahil = 1 and
              PartiAsamaStokDetay.CikisTarihi is not null and
              Asama.PartilemeMi = 1
        into :PartilemeAylik, :PartilemeAylikFason, :PartilemeAylikSatis; */

      suspend;
    end
  end
