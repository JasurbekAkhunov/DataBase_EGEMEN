CREATE PROCEDURE STOKHAREKETBILGISIGETIR(
    ASTOKHAREKETID INTEGER)
RETURNS (
    STOKHAREKETID INTEGER,
    HAREKETTIPI INTEGER,
    URETIMISLEMTIPIID INTEGER,
    HAREKETTIPISTR VARCHAR(120),
    URETIMISLEMTIPIADI VARCHAR(30),
    HAMIADE INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID VARCHAR(30),
    RENKADI VARCHAR(50),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    PLANLANDIGIFIRMALAR VARCHAR(1024),
    SIPARISSIRANO INTEGER,
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISRECETEKODU VARCHAR(20),
    SIPARISHAMMIKTAR DOUBLE PRECISION,
    ARTIKELNO VARCHAR(20),
    URETIMPARTINO VARCHAR(30),
    KAPADEDI INTEGER,
    BRUTMIKTAR DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    METRE DOUBLE PRECISION,
    ADET DOUBLE PRECISION,
    GIRISKKBITISTARIHI TIMESTAMP,
    KKBITIRENKULLANICIID INTEGER,
    KKBITIRENKULLANICIADI VARCHAR(40),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    TESLIMTARIHI TIMESTAMP,
    ORMEPLANLAMAID INTEGER,
    VERILENSIPARISDETAYID INTEGER,
    SIPIGNESAYISI INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    PUS INTEGER,
    FEINE INTEGER,
    HARMANCINSIADI VARCHAR(50),
    HAMRENKADI VARCHAR(30),
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    MUSTERIPARTINO VARCHAR(30),
    DIGERRENKADI VARCHAR(100),
    KALITEADI VARCHAR(255),
    BOYATALIMATNO VARCHAR(30))
AS
declare variable Tamir integer;
declare variable shtAlis integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretim integer;
declare variable FAlinanSiparisNoDigitSayisi    integer;
declare variable FRevizeTeslimTarihi            TimeStamp;
declare variable FFirmaKisaAdi varchar(10);
begin

  select shtAlis, ShtSatisIade, ShtFasonAlis, shtFasonUretim from OrmeStokHareketTipiGetir into :shtAlis, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretim;

  Select StokHareket.StokHareketID, StokHareket.HareketTipi, StokHareket.HamIade, StokHareket.UretimIslemTipiId, StokHareket.FirmaId, SHFirma.Adi,
             OrmeFis.FisNo, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, StokHareket.HamId, Ham.Adi,
             Stokhareket.RenkId, Renk.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Stokhareket.OrmeSiparisDetayId,
             Stokhareket.BoyaSiparisDetayId,  OrmeSiparisDetay.SiparisSiraNo,  BoyaSiparis.FirmaId, SiparisFirma.Adi, Stokhareket.UretimPartiNo,
             Stokhareket.KapAdedi, Stokhareket.BrutMiktar, Stokhareket.Miktar, Stokhareket.Metre, Stokhareket.Adet, StokHareket.Tamir,
             UretimIslemTipi.Adi, Stokhareket.GirisKKBitisTarihi, Stokhareket.GirisKKBitirenKullaniciId, Kullanici.Adi,
             OrmeSiparisDetay.Teslimtarihi, OrmeSiparisDetay.RevizeTeslimtarihi,
             coalesce(BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.EnIstenen),
             coalesce(BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.GramajIstenen), BoyaSiparisDetay.LabReceteKodu,
             StokHareket.OrmePlanlamaId, OrmeSiparisDetay.ArtikelNo,
             StokHareket.VerilenSiparisDetayId, OrmeSiparisDetay.HamMiktar,
             OrmeSiparisDetay.IgneSayisi, BoyaSiparis.SiparisGrupID, SiparisGrup.Adi, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
             HarmanCinsi.Adi, HamRenk.Adi ,DokumaMakina.DokumaMakinaId,DokumaMakina.Adi,StokHareket.MusteriPartiNo,Renk.DigerAdi,Kalite.KaliteAdi,BoyaTalimat.BoyaTalimatNo
              from StokHareket
  left join Ham on Ham.HamId = StokHareket.HamId
  left join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Ham.HarmanCinsiId
  left join HamRenk on HamRenk.HamRenkId = Ham.HamRenkId
  left join Renk on Renk.RenkId = StokHareket.RenkId
  left join Firma SHFirma on SHFirma.FirmaId = StokHareket.FirmaId
  left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
  left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = StokHareket.BoyaSiparisDetayId
  left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join UretimIslemTipi on UretimIslemTipi.UretimIslemTipiId = StokHareket.UretimIslemTipiId
  left join Kullanici on Kullanici.KullaniciID = StokHareket.GirisKKBitirenKullaniciId
  Left Join VerilenSiparisDetay ON VerilenSiparisDetay.VerilenSiparisDetayId = StokHareket.VerilenSiparisDetayId
  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = StokHareket.OrmePlanlamaId
  left Join DokumaMakina on DokumaMakina.DokumaMakinaId = Stokhareket.DokumaMakinaId
  left join BoyaTalimat on BoyaTalimat.BoyaTalimatKayitId = StokHareket.BoyaTalimatKayitId
  where StokHareket.StokHareketId = :AStokHareketId
  into :StokHareketID, :HareketTipi, :HamIade, :UretimIslemTipiId, :FirmaId, :FirmaAdi, :FisNo, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi,
       :HamId, :HamAdi, :RenkId, :RenkAdi, :BoyaSiparisYil, :BoyaSiparisId,
       :OrmeSiparisDetayId, :BoyaSiparisDetayId, :SiparisSiraNo,  :SiparisFirmaId, :SiparisFirmaAdi, :UretimPartiNo, :KapAdedi, :BrutMiktar, :Miktar, :Metre, :Adet,
       :Tamir, :UretimIslemTipiAdi, :GirisKKBitisTarihi, :KKBitirenKullaniciId, :KKBitirenKullaniciAdi,
       :Teslimtarihi, :FRevizeTeslimtarihi,
       :EnIstenen, :GramajIstenen, :SiparisReceteKodu, :OrmePlanlamaId, :ArtikelNo,
       :VerilenSiparisDetayId, :SiparisHamMiktar, :SipIgneSayisi,
       :SiparisGrupID, :SiparisGrupAdi, :Pus, :Feine, :HarmanCinsiAdi, :HamRenkAdi, :DokumaMakinaId, :DokumaMakinaAdi, :MusteriPartiNo, :DigerRenkAdi, :KaliteAdi, :BoyaTalimatNo;
  if (StokHareketID is not null) then
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisYil is not null) then BoyaSiparisIdStr = '(' || Format_Float('#0', BoyaSiparisYil) || ') ' || egemenintegertostr(BoyaSiparisId) || ' ';

    if (HareketTipi = shtAlis) then HareketTipiStr = 'Alış';
    else if (HareketTipi = shtSatisIade) then HareketTipiStr = 'Satış İade';
    else if (HareketTipi = shtFasonAlis) then HareketTipiStr = 'Fason İşlenmek Üzere Giriş';
    else if (HareketTipi = shtFasonUretim and HamIade = 0) then HareketTipiStr = 'Fason Üretim';
    else if (HareketTipi = shtFasonUretim and HamIade = 1) then HareketTipiStr = '(İade) Fason Üretim';
    else HareketTipiStr = 'Bilinmiyor';

    if (UretimIslemTipiAdi is not null) then HareketTipiStr = HareketTipiStr || '-' || UretimIslemTipiAdi;

    if (Tamir = 1) then HareketTipiStr = HareketTipiStr || ' ( Tamir )';

    if (FRevizeTeslimtarihi is not null) then Teslimtarihi = FRevizeTeslimtarihi;

    PlanlandigiFirmalar = '';

    if (HareketTipi = shtFasonUretim and OrmePlanlamaId is null) then
    begin
      for Select Firma.KisaAdi from OrmePlanlama
      left join Firma on Firma.FirmaID = OrmePlanlama.FirmaId
      where OrmePlanlama.OrmeSiparisDetayID = :OrmeSiparisDetayID and OrmePlanlama.FirmaId is not null
      group by Firma.KisaAdi
      into :FFirmaKisaAdi do
      begin
        if (Char_Length(PlanlandigiFirmalar) + Char_Length(FFirmaKisaAdi) < 1000) then
        begin
          if (PlanlandigiFirmalar <> '') then PlanlandigiFirmalar = PlanlandigiFirmalar || cast(', ' as varchar(2) character set UTF8);
          PlanlandigiFirmalar = PlanlandigiFirmalar || FFirmaKisaAdi;
        end
      end
    end

    suspend;
  end
end
