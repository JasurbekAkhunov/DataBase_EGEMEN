CREATE PROCEDURE MAMULDEPOTRANSFERRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ADEPOID VARCHAR(10),
    ATRANSFERDEPOID VARCHAR(10))
RETURNS (
    MAMULDEPOTRANSFERFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    SEVKTARIHI TIMESTAMP,
    DEPOID VARCHAR(10),
    DEPOADI VARCHAR(30),
    TRANSFERDEPOID VARCHAR(10),
    TRANSFERDEPOADI VARCHAR(30),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    REFERANSKODU VARCHAR(30),
    KALITENO INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    MUSTERIPARTINOSTR VARCHAR(70),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ENSTR VARCHAR(50),
    GRM2STR VARCHAR(50),
    HAMSTOKMASTERIDSTR VARCHAR(50),
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    SIPARISRENKADI VARCHAR(40),
    TARTIKAPADEDI INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    SIPILAVEISLEMLER VARCHAR(255),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    SIPARISKILO NUMERIC(12,3),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    SIPARISBIRIMFIYAT DOUBLE PRECISION,
    SIPARISDOVIZID VARCHAR(5),
    RAFADI ADI30,
    SONLOKASYONADI ADI50)
AS
begin
  For Select MamulDepoTransferFisi.MamulDepoTransferFisiId, MamulDepoTransferFisi.FisYil, MamulDepoTransferFisi.FisNo,
             MamulDepoTransferFisi.FisTarihi, MamulDepoTransferFisi.IrsaliyeTarihi, MamulDepoTransferFisi.IrsaliyeSeriNo,
             MamulDepoTransferFisi.IrsaliyeNo, MamulDepoTransferFisi.SevkTarihi,
             MamulDepoTransferFisi.DepoId, DP.Adi, MamulDepoTransferFisi.TransferDepoId, TD.Adi,
             Parti.PartiYil, Parti.PartiId, Parti.IslemTipiId, IslemTipi.Adi, Parti.ReceteId, Recete.Adi,
             Parti.PartiGrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi,
             MamulDepoTransferFisiDetay.PartiSinifId, PartiStok.HamAdiToplami, Ham.ReferansKodu, PartiStokTarti.KaliteNo,
             Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.Metre), Sum(PartiStokTarti.KapAdedi), BoyaSiparisDetay.RenkAdi,
             SiparisProses.Adi, BoyaSiparisDetay.IlaveIslemler, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Parti.Fason,
             Ham.Adi2, Ham.Adi3, Ham.Adi4, BoyaSiparisDetay.SiparisKilo, PartiStok.BirimFiyat, PartiStok.DovizId, BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, Raf.Adi, RafLokasyon.Adi from MamulDepoTransferFisi
    left join Depo DP on DP.DepoId = MamulDepoTransferFisi.DepoId
    left join Depo TD on TD.DepoId = MamulDepoTransferFisi.TransferDepoId
    left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = MamulDepoTransferFisi.MamulDepoTransferFisiId
    left join PartiStok on PartiStok.PartiSinifId = MamulDepoTransferFisiDetay.PartiSinifId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = MamulDepoTransferFisiDetay.PartiStokTartiId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join Raf on MamulDepoTransferFisiDetay.RafId = Raf.RafId
    left join RafLokasyon on Raf.Sonraflokasyonid = RafLokasyon.RafLokasyonId
    where MamulDepoTransferFisi.FisTarihi >= :ABaslangicTarihi and MamulDepoTransferFisi.FisTarihi <= :ABitisTarihi and
         (MamulDepoTransferFisi.DepoId = :ADepoId or :ADepoId is null) and
         (MamulDepoTransferFisi.TransferDepoId = :ATransferDepoId or :ATransferDepoId is null) and
         MamulDepoTransferFisi.SistemKaydi = 0 and MamulDepoTransferFisiDetay.MamulDepoTransferFisiId is not null
    group by MamulDepoTransferFisi.MamulDepoTransferFisiId, MamulDepoTransferFisi.FisYil, MamulDepoTransferFisi.FisNo,
             MamulDepoTransferFisi.FisTarihi, MamulDepoTransferFisi.IrsaliyeTarihi, MamulDepoTransferFisi.IrsaliyeSeriNo,
             MamulDepoTransferFisi.IrsaliyeNo, MamulDepoTransferFisi.SevkTarihi,
             MamulDepoTransferFisi.DepoId, DP.Adi, MamulDepoTransferFisi.TransferDepoId, TD.Adi,
             Parti.PartiYil, Parti.PartiId, Parti.IslemTipiId, IslemTipi.Adi, Parti.ReceteId, Recete.Adi, Parti.PartiGrupId, PartiGrup.Adi,
             Parti.FirmaId, Firma.Adi, MamulDepoTransferFisiDetay.PartiSinifId, PartiStok.HamAdiToplami,
             Ham.ReferansKodu, PartiStokTarti.KaliteNo, BoyaSiparisDetay.RenkAdi,
             SiparisProses.Adi, BoyaSiparisDetay.IlaveIslemler,  BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Parti.Fason,
             Ham.Adi2, Ham.Adi3, Ham.Adi4, BoyaSiparisDetay.SiparisKilo, PartiStok.BirimFiyat, PartiStok.DovizId, BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, Raf.Adi, RafLokasyon.Adi
    order by MamulDepoTransferFisi.FisTarihi, MamulDepoTransferFisi.MamulDepoTransferFisiId
    into :MamulDepoTransferFisiId, :FisYil, :FisNo, :FisTarihi, :IrsaliyeTarihi, :IrsaliyeSeriNo,
         :IrsaliyeNo, :SevkTarihi, :DepoId, :DepoAdi, :TransferDepoId, :TransferDepoAdi,
         :PartiYil, :PartiId, :IslemTipiId, :IslemTipiAdi, :ReceteId, :ReceteAdi, :PartiGrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi,
         :PartiSinifId, :HamAdiToplami, :ReferansKodu, :KaliteNo, :Kilo, :Metre, :TartiKapAdedi,  :SiparisRenkAdi,
         :SiparisProsesAdi, SipIlaveIslemler, :BoyaSiparisYil, :BoyaSiparisId, :Fason, :HamAdi2, :HamAdi3, :HamAdi4, :SiparisKilo,
         :BirimFiyat, :DovizId, :SiparisBirimFiyat, :SiparisDovizId, :RafAdi, :SonLokasyonAdi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    else BoyaSiparisIdStr = '';

    if (Fason = 1) then  FasonStr = 'Fason'; else FasonStr = 'Satış';

    Select MusteriPartiNoStr, EnStr, Grm2Str, HamStokMasterIdStr from YariMamulPartiHamStok(:PartiSinifId) into :MusteriPartiNoStr, :EnStr, :Grm2Str, :HamStokMasterIdStr;

    suspend;
  end
end
