CREATE PROCEDURE HAMGIRISICINVSIPARISDETAYLARI(
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    AVERILENSIPARISYIL INTEGER,
    AVERILENSIPARISID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIYATGOREBILIR INTEGER)
RETURNS (
    VERILENSIPARISKAYITID INTEGER,
    VERILENSIPARISDETAYID INTEGER,
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    ONAYDURUMUSTR VARCHAR(15),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    RENKDIGERADI VARCHAR(50),
    MIKTAR DOUBLE PRECISION,
    GELENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    IADEEDILENMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    PUS INTEGER,
    FEINE INTEGER,
    MELANJKODU VARCHAR(20),
    REFERANSSIPARISNO VARCHAR(30),
    REFERANSFIRMAID VARCHAR(20),
    REFERANSFIRMAADI VARCHAR(100),
    TUPACIKENMAYLISTR VARCHAR(10),
    KENARKONSTRUKSIYONTIPIID INTEGER,
    KENARKONSTRUKSIYONTIPIADI VARCHAR(20),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TESLIMTARIHI TIMESTAMP)
AS
declare variable FHamAdi        varchar(255);
declare variable FVerilenSiparisOnayIstensinmi  integer;
declare variable FOnayDurumu integer;
declare variable Okey integer;
declare variable HamTipiId integer;
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstLycra        integer;
declare variable hstTelef        integer;
declare variable VerilenSiparisTipiId  integer;
declare variable TupAcikEnMayli integer;
declare variable VerilenSipteDetayOnayiKullan integer;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstLycra, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstLycra, :hstTelef;

  Select VerilenSiparisOnayIstensinmi, VerilenSipteDetayOnayiKullan from sistem where SistemId = 1 into :FVerilenSiparisOnayIstensinmi, :VerilenSipteDetayOnayiKullan;

  if (AHamId is not null) then
    Select HamTipiId from Ham where Ham.HamID = :AHamId into :HamTipiId;
  else
    HamTipiId = null;

  if (HamTipiId = hstOrmeKumas or HamTipiId = hstRibana or HamTipiId = hstAksesuar) then VerilenSiparisTipiID = 0;
  else if (HamTipiId = hstDokumaKumas) then VerilenSiparisTipiId = 1;
  else if (HamTipiId = hstIplik or HamTipiId = hstLycra) then VerilenSiparisTipiId = 2;
  else VerilenSiparisTipiID = null;

  if (AVerilenSiparisID is null) then
  begin
    for Select iif(:VerilenSipteDetayOnayiKullan = 0, VerilenSiparis.OnayDurumu, VerilenSiparisDetay.OnayDurumu), VerilenSiparis.VerilenSiparisKayitId, VerilenSiparisDetay.VerilenSiparisDetayId,
               VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, VerilenSiparisDetay.HamId, Ham.Adi, VerilenSiparisDetay.Adi,
               VerilenSiparisDetay.RenkId, Renk.Adi, Renk.DigerAdi, VerilenSiparisDetay.Miktar, VerilenSiparisDetay.BirimFiyat,
               VerilenSiparisDetay.DovizId, VerilenSiparisDetay.GelenMiktar, VerilenSiparisDetay.Notu,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, VerilenSiparisDetay.BoyaSiparisDetayId,
               VerilenSiparisDetay.Pus, VerilenSiparisDetay.Feine, VerilenSiparisDetay.TupAcikEnMayli, VerilenSiparisDetay.KenarKonstruksiyonTipiId,
               KenarKonstruksiyonTipi.Adi, VerilenSiparisDetay.ReferansFirmaId, RefFirma.Adi, VerilenSiparisDetay.ReferansSiparisNo, VerilenSiparisDetay.MelanjKodu,
               VerilenSiparisDetay.IadeEdilenMiktar, VerilenSiparisDetay.TeslimTarihi
               from VerilenSiparis
      left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisKayitId = VerilenSiparis.VerilenSiparisKayitId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = VerilenSiparisDetay.BoyaSiparisKayitId
      LEFT JOIN Ham ON Ham.HamId = VerilenSiparisDetay.HamId
      left join KenarKonstruksiyonTipi on KenarKonstruksiyonTipi.KenarKonstruksiyonTipiId = VerilenSiparisDetay.KenarKonstruksiyonTipiId
      LEFT JOIN Renk ON Renk.RenkId = VerilenSiparisDetay.RenkId
      Left Join Kullanici IK on IK.KullaniciId = VerilenSiparisDetay.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = VerilenSiparisDetay.KullaniciId
      left join Firma RefFirma on RefFirma.FirmaId = VerilenSiparisDetay.ReferansFirmaId
      WHERE VerilenSiparis.FirmaId = :AFirmaId and
           (VerilenSiparis.VerilenSiparisTipiID = :VerilenSiparisTipiId  or :VerilenSiparisTipiID is null) and
           (VerilenSiparis.SiparisTarihi >= :ABaslangicTarihi or :AVerilenSiparisId is not null) and
           (VerilenSiparis.SiparisTarihi <= :ABitisTarihi or :AVerilenSiparisId is not null) and
            VerilenSiparisDetay.VerilenSiparisDetayId is not null and
           (VerilenSiparisDetay.HamId = :AHamId or :AHamId is null) and VerilenSiparisDetay.Kapali = 0 and
            (VerilenSiparisDetay.Miktar + VerilenSiparisDetay.IadeEdilenMiktar) > VerilenSiparisDetay.GelenMiktar
     into :FOnayDurumu, :VerilenSiparisKayitId, :VerilenSiparisDetayId, :VerilenSiparisYil, :VerilenSiparisId,
          :HamId, :HamAdi, :FHamAdi, :RenkId, :RenkAdi, :RenkDigerAdi, :Miktar, :BirimFiyat, :DovizId, :GelenMiktar, :Notu,
          :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
          :Pus, :Feine, :TupAcikEnMayli, :KenarKonstruksiyonTipiId, :KenarKonstruksiyonTipiAdi,
          :ReferansFirmaId, :ReferansFirmaAdi, :ReferansSiparisNo, :MelanjKodu, :IadeEdilenMiktar, :TeslimTarihi do
    begin
      Okey = 1;
      if ((FVerilenSiparisOnayIstensinmi = 1 and FOnayDurumu <> 1) or FOnayDurumu = 2) then Okey = 0;

      if (Okey = 1) then
      begin
        if (HamId is null) then HamAdi = FHamAdi;

        if (AFiyatGorebilir = 0) then
        begin
          BirimFiyat = NUll;
          DovizId = Null;
        end

        Tutar = Miktar * BirimFiyat;

        Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

        if (FOnayDurumu = 1) then OnayDurumuStr = 'Onaylandı';
        else if (FOnayDurumu = 2) then OnayDurumuStr = 'Reddedildi';
        else OnayDurumuStr = '';
        KalanMiktar = Miktar - GelenMiktar;

        VerilenSiparisIdStr = '';
        if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

        BoyaSiparisIdStr = '';
        if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
        suspend;
      end
    end
  end
  else
  begin
    for Select iif(:VerilenSipteDetayOnayiKullan = 0, VerilenSiparis.OnayDurumu, VerilenSiparisDetay.OnayDurumu), VerilenSiparis.VerilenSiparisKayitId, VerilenSiparisDetay.VerilenSiparisDetayId,
               VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, VerilenSiparisDetay.HamId, Ham.Adi, VerilenSiparisDetay.Adi,
               VerilenSiparisDetay.RenkId, Renk.Adi, Renk.DigerAdi, VerilenSiparisDetay.Miktar, VerilenSiparisDetay.BirimFiyat,
               VerilenSiparisDetay.DovizId, VerilenSiparisDetay.GelenMiktar, VerilenSiparisDetay.Notu,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, VerilenSiparisDetay.BoyaSiparisDetayId,
               VerilenSiparisDetay.Pus, VerilenSiparisDetay.Feine, VerilenSiparisDetay.TupAcikEnMayli, VerilenSiparisDetay.KenarKonstruksiyonTipiId,
               KenarKonstruksiyonTipi.Adi, VerilenSiparisDetay.ReferansFirmaId, RefFirma.Adi, VerilenSiparisDetay.ReferansSiparisNo, VerilenSiparisDetay.MelanjKodu,
               VerilenSiparisDetay.IadeEdilenMiktar, VerilenSiparisDetay.TeslimTarihi
               from VerilenSiparis
      left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisKayitId = VerilenSiparis.VerilenSiparisKayitId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = VerilenSiparisDetay.BoyaSiparisKayitId
      LEFT JOIN Ham ON Ham.HamId = VerilenSiparisDetay.HamId
      left join KenarKonstruksiyonTipi on KenarKonstruksiyonTipi.KenarKonstruksiyonTipiId = VerilenSiparisDetay.KenarKonstruksiyonTipiId
      LEFT JOIN Renk ON Renk.RenkId = VerilenSiparisDetay.RenkId
      Left Join Kullanici IK on IK.KullaniciId = VerilenSiparisDetay.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = VerilenSiparisDetay.KullaniciId
      left join Firma RefFirma on RefFirma.FirmaId = VerilenSiparisDetay.ReferansFirmaId
      WHERE VerilenSiparis.FirmaId = :AFirmaId and
           (VerilenSiparis.VerilenSiparisTipiID = :VerilenSiparisTipiId  or :VerilenSiparisTipiID is null) and
           (VerilenSiparis.SiparisTarihi >= :ABaslangicTarihi or :AVerilenSiparisId is not null) and
           (VerilenSiparis.SiparisTarihi <= :ABitisTarihi or :AVerilenSiparisId is not null) and
           (VerilenSiparis.VerilenSiparisYil = :AVerilenSiparisYil or :AVerilenSiparisYil is null) and
           (VerilenSiparis.VerilenSiparisId = :AVerilenSiparisId or :AVerilenSiparisId is null) and
            VerilenSiparisDetay.VerilenSiparisDetayId is not null and
           (VerilenSiparisDetay.HamId = :AHamId or :AHamId is null) and VerilenSiparisDetay.Kapali = 0
     into :FOnayDurumu, :VerilenSiparisKayitId, :VerilenSiparisDetayId, :VerilenSiparisYil, :VerilenSiparisId,
          :HamId, :HamAdi, :FHamAdi, :RenkId, :RenkAdi, :RenkDigerAdi, :Miktar, :BirimFiyat, :DovizId, :GelenMiktar, :Notu,
          :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
          :Pus, :Feine, :TupAcikEnMayli, :KenarKonstruksiyonTipiId, :KenarKonstruksiyonTipiAdi,
          :ReferansFirmaId, :ReferansFirmaAdi, :ReferansSiparisNo, :MelanjKodu, :IadeEdilenMiktar, :TeslimTarihi do
    begin
      Okey = 1;
      if ((FVerilenSiparisOnayIstensinmi = 1 and FOnayDurumu <> 1) or FOnayDurumu = 2) then Okey = 0;

      if (Okey = 1) then
      begin
        if (HamId is null) then HamAdi = FHamAdi;
        if (AFiyatGorebilir = 0) then
        begin
          BirimFiyat = NUll;
          DovizId = Null;
        end

        Tutar = Miktar * BirimFiyat;

        Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

        if (FOnayDurumu = 1) then OnayDurumuStr = 'Onaylandı';
        else if (FOnayDurumu = 2) then OnayDurumuStr = 'Reddedildi';
        else OnayDurumuStr = '';
        KalanMiktar = Miktar - GelenMiktar;

        VerilenSiparisIdStr = '';
        if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

        BoyaSiparisIdStr = '';
        if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
        suspend;
      end
    end
  end
end
