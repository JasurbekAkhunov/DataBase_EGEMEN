CREATE PROCEDURE HAMSTOKDETAYKALANIGETIR(
    AHAMSTOKDETAYID INTEGER)
RETURNS (
    KALANMIKTAR NUMERIC(12,3),
    PARTILENEBILIRMIKTAR NUMERIC(12,3))
AS
declare variable RezervEdilenMiktar           Numeric(12,3);
declare variable KullanilanMiktar             Numeric(12,3);
declare variable Miktar                       Numeric(12,3);
declare variable IptalPartiMiktar             Numeric(12,3);
declare variable PartilenebilirRafKilo        Numeric(12,3);
declare variable PartilenebilirRafMetre       Numeric(12,3);
declare variable DokumaKumasiKilodanTakipEt   Integer;
declare variable HamTipiId                    Integer;
declare variable RezervEdilenMetre            Numeric(12,3);
declare variable RezervEdilenKilo             Numeric(12,3);
begin
  Select DokumaKumasiKilodanTakipEt from Sistem Where SistemId = 1 Into :DokumaKumasiKilodanTakipEt;

  Select Ham.HamTipiId From HamStokDetay
  Left Join Ham on HamStokDetay.HamId = Ham.HamId
  Where HamStokDetay.HamStokDetayId = :AHamStokDetayId
  Into :HamTipiId;

  select Sum(RafDetay.IadeEdilecekMiktar + RafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.SatisMiktar) from RafDetay
  where RafDetay.HamStokDetayId = :AHamStokDetayId
  into :KullanilanMiktar;

  IptalPartiMiktar = 0;

  Select HamStokDetay.Miktar, HamStokDetay.RezervEdilenMiktar, HamStokDetay.RezervEdilenMetre, HamStokDetay.RezervEdilenKilo, Sum(RafDetay.Miktar),
         Sum(Iif(Raf.PartilemeYapilabilir = 1, (RafDetay2.Kilo + RafDetay2.AktarmadanIadeEdienKilo - RafDetay2.AktarmayaCikanKilo - RafDetay2.PartilenenKilo - RafDetay2.RafTransferKilo - RafDetay2.IadeEdilecekKilo - RafDetay2.IadeEdilenKilo - RafDetay2.SatisKilo), 0)),
         Sum(Iif(Raf.PartilemeYapilabilir = 1, (RafDetay2.Metre - RafDetay2.PartilenenMetre - RafDetay2.RafTransferMetre - RafDetay2.IadeEdilecekMetre - RafDetay2.IadeEdilenMetre - RafDetay2.SatisMetre), 0)) from HamStokDetay
  left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
  left join RafDetay RafDetay2 on RafDetay2.HamStokDetayID = HamStokDetay.HamStokDetayId
  left join Raf on Raf.RafID = RafDetay2.RafId
  where HamStokDetay.HamStokDetayId = :AHamStokDetayId
  group by HamStokDetay.Miktar, HamStokDetay.RezervEdilenMiktar, HamStokDetay.RezervEdilenMetre, HamStokDetay.RezervEdilenKilo
  into :Miktar, :RezervEdilenMiktar, :RezervEdilenMetre, :RezervEdilenKilo, :IptalPartiMiktar, :PartilenebilirRafKilo, :PartilenebilirRafMetre;

  if (IptalPartiMiktar is null) then IptalPartiMiktar = 0;

  Miktar = Miktar + IptalPartiMiktar;

  if (KullanilanMiktar is null) then KullanilanMiktar = 0;

  KalanMiktar = Miktar - KullanilanMiktar - RezervEdilenMiktar;

  if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
    PartilenebilirMiktar = PartilenebilirRafKilo - RezervEdilenKilo;
  else
    PartilenebilirMiktar = PartilenebilirRafMetre - RezervEdilenMetre;

  suspend;
end
