CREATE PROCEDURE DOKUMATOPKES(
    ADOKUMAPLANLAMAID INTEGER,
    ALEVENTBITTI SMALLINT)
RETURNS (
    DOKUMAKALITEKONTROLID INTEGER,
    METRE NUMERIC(12,3),
    KILO NUMERIC(12,3),
    DOKUMADEFAULTRAFID VARCHAR(10),
    LOT VARCHAR(150),
    ATKILOTU VARCHAR(30),
    COZGULOTU VARCHAR(30),
    KUMASCOZGUTELSAYISI INTEGER,
    RECETEATKIRAPORTELTOPLAMI INTEGER,
    RECETECOZGURAPORTELTOPLAMI INTEGER,
    ATKIIPLIKCM DOUBLE PRECISION,
    COZGUIPLIKCM DOUBLE PRECISION,
    BOYDANCEKME DOUBLE PRECISION,
    TARAKENI DOUBLE PRECISION,
    HAMENCM DOUBLE PRECISION,
    TOPLAMATKIIPAGIRLIGI DOUBLE PRECISION,
    TOPLAMCOZGUIPAGIRLIGI DOUBLE PRECISION,
    GRAMMETREKARE DOUBLE PRECISION,
    TOPNO INTEGER)
AS
declare variable fatkicozgu smallint;
declare variable fhamid integer;
declare variable frecetehamid integer;
declare variable fne integer;
declare variable fdenye double precision;
declare variable fnm double precision;
declare variable ftelsayisi integer;
declare variable fbirmetreningrami double precision;
declare variable foran double precision;
declare variable TezgahtaAtilacakAtki Double Precision;
declare variable OrmeRafId integer;
declare variable UretimeCikisTarihi TimeStamp;
declare variable SonTopKesmeTarihi TimeStamp;
declare variable shtUretimeCikis Integer;
declare variable DokumaMakinaId MakinaKoduSize;
declare variable ToplamTur Integer;
declare variable DepoId DepoKoduSize;
declare variable OrmeVardiyaId integer;
begin
  /* son kesilen toptan beri gecen turu bulmamız gerek
  fatkisayisi       = 0 ;*/
  if (adokumaplanlamaid is null) then exit ;
  if (aleventbitti is null) then aleventbitti = 0 ;

  select shtUretimeCikis from OrmeStokHareketTipiGetir into :shtUretimeCikis;

  select ormesiparisdetay.hamid, ham.telsayisi, ham.atkiiplikcm, ham.cozguiplikcm, ham.boydancekme, ham.tarakeni, ham.hamencm, Coalesce(DokumaPlanlama.TezgahtaAtilacakAtki,0),
  DokumaPlanlama.DokumaMakinaId
  from dokumaplanlama
  left join ormesiparisdetay on ormesiparisdetay.ormesiparisdetayid = dokumaplanlama.ormesiparisdetayid
  left join ham on ham.hamid = ormesiparisdetay.hamid
  where dokumaplanlama.dokumaplanlamaid = :adokumaplanlamaid
  into :fhamid, :kumascozgutelsayisi, :atkiiplikcm, :cozguiplikcm, :boydancekme, :tarakeni, :hamencm, :TezgahtaAtilacakAtki, :DokumaMakinaId;

  select First 1 StokHareket.UretimeCikisTarihi, StokHareket.OrmeRafId from StokHareket
  left join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
  where DokumaHareketi = 1 and
  StokHareket.HamIade = 0 and
  HareketTipi = :shtUretimeCikis and
  StokHareket.DokumaMakinaId = :DokumaMakinaId and
  (OrmeRaf.CozguIsEmriKayitId is not null or OrmeRaf.HasilIsEmriKayitId is not null or OrmeRaf.IndigoIsEmriKayitId is not null)
  order by StokHareket.UretimeCikisTarihi desc nulls last into :UretimeCikisTarihi, :OrmeRafId;

  select Max(DokumaKaliteKontrol.InsertTarihi) from DokumaKaliteKontrol
  left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaKaliteKontrol.DokumaPlanlamaId
  where DokumaKaliteKontrol.OrmeRafId = :OrmeRafId and
  DokumaPlanlama.DokumaMakinaId = :DokumaMakinaId and
  DokumaKaliteKontrol.DokumaKaliteKontrolIdRef is null
  into :SonTopKesmeTarihi;

  if (SonTopKesmeTarihi is null) then
    SonTopKesmeTarihi = UretimeCikisTarihi;

  select Coalesce(Sum(OrmeMakinaDurumu.Tur), 0) from OrmeMakinaDurumu where OrmeMakinaDurumu.DokumaMakinaId = :DokumaMakinaId and
  OrmeMakinaDurumu.Tarih >= :SonTopKesmeTarihi
  into :ToplamTur;

  select OrmeSonVardiyaIdGetir.OrmeVardiyaId from OrmeSonVardiyaIdGetir(Null, :DokumaMakinaId) into :OrmeVardiyaId;

  select DokumaPlanlamaLotAtkiLotuGetir.lot, DokumaPlanlamaLotAtkiLotuGetir.atkilotu, DokumaPlanlamaLotAtkiLotuGetir.CozguLotu
  from DokumaPlanlamaLotAtkiLotuGetir(:adokumaplanlamaid)
  into :lot, :atkilotu, :CozguLotu;

  select sistem.dokumadefaultrafid from sistem where sistem.sistemid= 1 into :dokumadefaultrafid ;
  select DepoRaf.DepoId from DepoRaf where DepoRaf.RafId = :dokumadefaultrafid into :DepoId;

  ToplamAtkiIpAgirligi = 0 ;
  ToplamCozguIpAgirligi = 0 ;

  select sum(dokumaplanlamarecete.telsayisi) from dokumaplanlamarecete where dokumaplanlamarecete.dokumaplanlamaid = :adokumaplanlamaid and dokumaplanlamarecete.atkicozgu = 1 into :receteatkiraporteltoplami ;
  select sum(dokumaplanlamarecete.telsayisi) from dokumaplanlamarecete where dokumaplanlamarecete.dokumaplanlamaid = :adokumaplanlamaid and dokumaplanlamarecete.atkicozgu = 2 into :recetecozguraporteltoplami ;

  if (receteatkiraporteltoplami is null) then receteatkiraporteltoplami = 0 ;
  if (recetecozguraporteltoplami is null) then recetecozguraporteltoplami = 0 ;

  for select dokumaplanlamarecete.hamid, dokumaplanlamarecete.atkicozgu, dokumaplanlamarecete.telsayisi, kalinlik.kalinlikne, kalinlik.denye, kalinlik.nm
    from dokumaplanlamarecete
    left join ham on ham.hamid = dokumaplanlamarecete.hamid
    left join kalinlik on kalinlik.kalinlikid = ham.kalinlikid
    where dokumaplanlamarecete.dokumaplanlamaid = :adokumaplanlamaid
    order by dokumaplanlamarecete.atkicozgu, dokumaplanlamarecete.sirano
    into :frecetehamid, :fatkicozgu, :ftelsayisi, :fne, :fdenye, :fnm do
  begin
    /*
        Ne : Bir libre ağırlıktaki bir iplik örneğinin hank olarak uzunluğudur.
          Burada; 1 libre=453,6 gram, 1 hank ise pamuk ve ipek iplikler için 768 metredir.
          Ne'yi Nm'ye Çevirmek için Ne x 1,693 = Nm formülü kullanılır.
        Nm : 1 gram iplik kaç metre geldiğini gösterir. 30 Nm demek 1 gr 30 mt geliyor demek.
        Denye : 9000 mt iplik kaç gram geldiğini gösterir. 90 denye demek 9000 mt iplik 90 gr geliyor demek.
    */
    if (fnm is null) then
      select Nm from DokumaKalinlikDegerHesapla(:fdenye, :fne) into :fnm;

    fbirmetreningrami = 0 ;
    if (fatkicozgu = 1 ) then
    begin
      /*foran = (ftelsayisi * 1.0) / (receteatkiraporteltoplami *1.0)  ;*/
      fbirmetreningrami = ( ftelsayisi * 100 * (tarakeni /100) ) / fnm;
      toplamatkiipagirligi = toplamatkiipagirligi + fbirmetreningrami ;
    end
    else if (fatkicozgu = 2 ) then
    begin
      foran = (ftelsayisi * 1.0) / (recetecozguraporteltoplami *1.0)  ;
      fbirmetreningrami = ( kumascozgutelsayisi * foran * 100 ) / ( fnm * (100 - boydancekme) ) ;
      toplamcozguipagirligi = toplamcozguipagirligi + fbirmetreningrami ;
    end
  end

  grammetrekare = (toplamatkiipagirligi + toplamcozguipagirligi) / (hamencm / 100) ;
  metre = 0 ;
  kilo = 0 ;

  if (TezgahtaAtilacakAtki <> 0) then metre = (ToplamTur / TezgahtaAtilacakAtki) / 100 ;
  if (metre > 0 ) then kilo = metre * (grammetrekare / 1000) ;

  if (metre is not null and metre > 0 ) then
  begin
    dokumakalitekontrolid = gen_id(dokumakalitekontrol_id_gen, 1) ;

    select DokumaKKTopNoGetir.TopNo from DokumaKKTopNoGetir(:adokumaplanlamaid) into :TopNo;

    insert into dokumakalitekontrol(dokumaplanlamaid, dokumakalitekontrolid, miktar, kilo, rafid, lot, atkilotu, leventbitti, otomasyondangeldi, HesaplananMetre, DepoId, DokumaMakinaId, OrmeRafId, CozguLotu, OrmeVardiyaId, TopNo)
    values(:adokumaplanlamaid, :dokumakalitekontrolid, :Metre, Coalesce(:kilo, 0), :dokumadefaultrafid, :lot, :atkilotu, :aleventbitti, 1, :Metre, :DepoId, Null, :OrmeRafId, :CozguLotu, :OrmeVardiyaId, :TopNo);
  end

  Suspend;
end
