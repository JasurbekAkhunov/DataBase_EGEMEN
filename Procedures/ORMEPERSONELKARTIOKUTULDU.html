CREATE PROCEDURE ORMEPERSONELKARTIOKUTULDU(
    AMAKINAID VARCHAR(10),
    APERSONELID INTEGER)
RETURNS (
    HATANO INTEGER,
    HATAMESAJ VARCHAR(100))
AS
declare variable VardiyaId          integer;
declare variable VardiyaNo          integer;
declare variable KS          integer;
begin
  HataNo = 0;
  select VardiyaId, VardiyaNo from PKartiIcinSonVardiyaIdGetir(:AMakinaId, null, 1) into :VardiyaId, :VardiyaNo;

  if (VardiyaId is Null) then
  begin
    HataNo = 1;
    HataMesaj = 'Vardiya bulunamadi...';
  end
  else if (VardiyaId = 0) then
  begin
    HataNo = 1;
    HataMesaj = 'Vardiya bulunamadi...';
  end

  if (HataNo = 0) then
  begin
    Select Count(*) from OrmeVardiyaPersonel where ORMEVARDIYAID=:VardiyaId and ORMEMAKINAID=:AMAKINAID and PersonelId = :APERSONELID into :KS;
    if (KS =0) then
    begin
      HataMesaj = 'Vardiya: ' || VardiyaId || ' Personel: ' || APERSONELID || ' Makina: ' || AMakinaId;
      update ORMEVARDIYAPERSONEL set PERSONELID=:APERSONELID where ORMEVARDIYAID=:VardiyaId and ORMEMAKINAID=:AMAKINAID;
    end
    else
    begin
      HataMesaj = 'Vardiya: ' || VardiyaId || ' Personel Çıktı!! Makina: ' || AMakinaId;
      HataNo = -1;
      update ORMEVARDIYAPERSONEL set PERSONELID = Null where ORMEVARDIYAID =:VardiyaId and ORMEMAKINAID=:AMAKINAID;
    end
  end
  suspend;
end
