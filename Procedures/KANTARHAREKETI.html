CREATE PROCEDURE KANTARHAREKETI(
    AKANTARKAYITID INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    KILO NUMERIC(15,3),
    BRUTKILO NUMERIC(15,3),
    KAPADEDI INTEGER)
AS
declare variable GirisCikis integer;
begin
  Select GirisCikis from Kantar where Kantar.KantarKayitID = :AKantarKayitId into :GirisCikis;
  if (GirisCikis = 0) then
  begin
    For Select HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi,
               HamStokIrsaliye.FiiliSevkTarihi, Sum(HamStokMaster.Kilo), Sum(HamStokMaster.KapAdedi) from HamStokIrsaliye
    left join Firma On Firma.FirmaId = HamStokIrsaliye.FirmaId
    left join HamStokMaster On HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
    where HamStokIrsaliye.KantarKayitId = :AKantarKayitId and HamStokMaster.HamStokMasterID is not null
    group By HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.FiiliSevkTarihi
    into :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi, :Kilo, :KapAdedi do
    begin
      BrutKilo = Kilo;
      suspend;
    end

    For Select OrmeIrsaliye.FirmaId, Firma.Adi, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
               OrmeIrsaliye.FiiliSevkTarihi, Sum(StokHareket.Miktar), Sum(StokHareket.BrutMiktar), Sum(StokHareket.KapAdedi) from OrmeIrsaliye
    left join Firma On Firma.FirmaId = OrmeIrsaliye.FirmaId
    left join StokHareket On StokHareket.OrmeIrsaliyeId = OrmeIrsaliye.OrmeIrsaliyeId
    where OrmeIrsaliye.KantarKayitId = :AKantarKayitId and StokHareket.OrmeIrsaliyeId is not null
    group By OrmeIrsaliye.FirmaId, Firma.Adi, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.FiiliSevkTarihi
    into :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi, :Kilo, :BrutKilo, :KapAdedi do  suspend;

    FirmaId = Null;
    FirmaAdi = Null;

    for Select UniteIrsaliye.AliciUniteId, AU.Adi, UniteIrsaliye.IrsaliyeTarihi, UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo,
               UniteIrsaliye.SevkTarihi, Sum(UniteIrsaliyeDetay.Kilo), Sum(UniteIrsaliyeDetay.NetKilo), Sum(UniteIrsaliyeDetay.KapAdedi) from UniteIrsaliye
      left join UniteIrsaliyeDetay on UniteIrsaliyeDetay.UniteIrsaliyeId = UniteIrsaliye.UniteIrsaliyeId
      left join Unite AU on AU.UniteId = UniteIrsaliye.AliciUniteId
      where UniteIrsaliye.KantarKayitID = :AKantarKayitId and UniteIrsaliyeDetay.UniteIrsaliyeId is not null
      group by UniteIrsaliye.AliciUniteId, AU.Adi, UniteIrsaliye.IrsaliyeTarihi, UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo, UniteIrsaliye.SevkTarihi
      into :UniteId, :UniteAdi, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :FiiliSevkTarihi, :BrutKilo, :Kilo, :KapAdedi do suspend;


    For Select Irsaliye.FirmaId, Firma.Adi, Irsaliye.SeriNo, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi,
               Irsaliye.FiiliSevkTarihi, Sum(StokDetay.Miktar), Sum(StokDetay.Miktar), Sum(StokDetay.KapAdedi) from Irsaliye
    left join Firma On Firma.FirmaId = Irsaliye.FirmaId
    left join StokDetay On StokDetay.IrsaliyeId = Irsaliye.IrsaliyeId
    where Irsaliye.KantarKayitId = :AKantarKayitId and StokDetay.IrsaliyeId is not null
    group By Irsaliye.FirmaId, Firma.Adi, Irsaliye.SeriNo, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi, Irsaliye.FiiliSevkTarihi
    into :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi, :Kilo, :BrutKilo, :KapAdedi do  suspend;

  end
  else
  begin
    for Select UniteIrsaliye.AliciUniteId, AU.Adi, UniteIrsaliye.IrsaliyeTarihi, UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo,
               UniteIrsaliye.SevkTarihi, Sum(UniteIrsaliyeDetay.Kilo), Sum(UniteIrsaliyeDetay.NetKilo), Sum(UniteIrsaliyeDetay.KapAdedi) from UniteIrsaliye
      left join UniteIrsaliyeDetay on UniteIrsaliyeDetay.UniteIrsaliyeId = UniteIrsaliye.UniteIrsaliyeId
      left join Unite AU on AU.UniteId = UniteIrsaliye.AliciUniteId
      where UniteIrsaliye.KantarKayitID = :AKantarKayitId and UniteIrsaliyeDetay.UniteIrsaliyeId is not null
      group by UniteIrsaliye.AliciUniteId, AU.Adi, UniteIrsaliye.IrsaliyeTarihi, UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo, UniteIrsaliye.SevkTarihi
      into :UniteId, :UniteAdi, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :FiiliSevkTarihi, :BrutKilo, :Kilo, :KapAdedi do suspend;

      UniteId = Null;
      UniteAdi = Null;

    For Select OrmeIrsaliye.FirmaId, Firma.Adi, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
               OrmeIrsaliye.FiiliSevkTarihi, Sum(StokHareket.Miktar), Sum(StokHareket.BrutMiktar), Sum(StokHareket.KapAdedi) from OrmeIrsaliye
    left join Firma On Firma.FirmaId = OrmeIrsaliye.FirmaId
    left join StokHareket On StokHareket.OrmeIrsaliyeId = OrmeIrsaliye.OrmeIrsaliyeId
    where OrmeIrsaliye.KantarKayitId = :AKantarKayitId and StokHareket.OrmeIrsaliyeId is not null
    group By OrmeIrsaliye.FirmaId, Firma.Adi, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.FiiliSevkTarihi
    into :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi, :Kilo, :BrutKilo, :KapAdedi do  suspend;

    For Select BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo,
            BoyaIrsaliye.SevkTarihi,  Sum(BoyaIrsaliyeDetay.Kilo), Sum(BoyaIrsaliyeDetay.BrutKilo), Sum(BoyaIrsaliyeDetay.KapAdedi) from BoyaIrsaliye
    left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
    where BoyaIrsaliye.KantarKayitID = :AKantarKayitId and BoyaIrsaliyeDetay.BoyaIrsaliyeId is not null
    group by BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo,
            BoyaIrsaliye.SevkTarihi
    into :FirmaId, :FirmaAdi, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :FiiliSevkTarihi, :Kilo, :BrutKilo, :KapAdedi do suspend;

    For Select Irsaliye.FirmaId, Firma.Adi, Irsaliye.SeriNo, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi,
               Irsaliye.FiiliSevkTarihi, Sum(StokDetay.Miktar), Sum(StokDetay.Miktar), Sum(StokDetay.KapAdedi) from Irsaliye
    left join Firma On Firma.FirmaId = Irsaliye.FirmaId
    left join StokDetay On StokDetay.IrsaliyeId = Irsaliye.IrsaliyeId
    where Irsaliye.KantarKayitId = :AKantarKayitId and StokDetay.IrsaliyeId is not null
    group By Irsaliye.FirmaId, Firma.Adi, Irsaliye.SeriNo, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi, Irsaliye.FiiliSevkTarihi
    into :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi, :Kilo, :BrutKilo, :KapAdedi do  suspend;

  end
end
