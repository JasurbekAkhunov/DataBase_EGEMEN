CREATE PROCEDURE ORMEKALITEKONTROLRAPORU_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AORMEMAKINAID VARCHAR(10),
    AORMEPLANLAMAID INTEGER,
    ASAKAT INTEGER,
    AOTOMATIKCIKISYAPILAN EGEBOOLEAN,
    AVARDIYATARIHINEGORE INTEGER)
RETURNS (
    ORMEKALITEKONTROLID INTEGER,
    ORMEPLANLAMARECETEID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    LOT VARCHAR(250),
    RENKID INTEGER,
    RENKADI ADI50,
    ORMEFISID INTEGER,
    FISNO INTEGER,
    HESAPLANANMIKTAR BOYAMIKTARSIZE,
    CIKILANMIKTAR BOYAMIKTARSIZE,
    TAMAMICIKILDI EGEBOOLEAN,
    TAMAMICIKILDISTR VARCHAR(5),
    ACIKLAMA VARCHAR(500),
    KUMASHAMID INTEGER,
    TARIH TIMESTAMP)
AS
begin
  /*if (AOtomatikCikisYapilan is null or AOtomatikCikisYapilan = 1) then*/
  begin
    For Select OrmeKaliteKontrol.OrmeKaliteKontrolId, DokumaUretimeOtomatikCikis.OrmePlanlamaReceteId, DokumaUretimeOtomatikCikis.OrmeFisId, DokumaUretimeOtomatikCikis.HesaplananMiktar,
    DokumaUretimeOtomatikCikis.CikilanMiktar, DokumaUretimeOtomatikCikis.TamamiCikildi, DokumaUretimeOtomatikCikis.Aciklama, OrmePlanlamaRecete.HamId, Ham.Adi,
    OrmePlanlamaRecete.Lot, OrmePlanlamaRecete.RenkId, Renk.Adi, OrmeFis.FisNo, HamKumas.HamId, OrmeKaliteKontrol.Tarih from OrmeKaliteKontrol
      left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeKaliteKontrol.OrmePlanlamaId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
      left join DokumaUretimeOtomatikCikis on DokumaUretimeOtomatikCikis.OrmeKaliteKontrolId = OrmeKaliteKontrol.OrmeKaliteKontrolId
      left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaReceteId = DokumaUretimeOtomatikCikis.OrmePlanlamaReceteId
      left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
      left join Ham HamKumas on HamKumas.HamId = OrmeSiparisDetay.HamId
      left join Renk on Renk.RenkId = OrmePlanlamaRecete.RenkId
      left join OrmeFis on OrmeFis.OrmeFisId = DokumaUretimeOtomatikCikis.OrmeFisId
      where ((:AVardiyaTarihineGore = 0 and OrmeKaliteKontrol.Tarih >= :ABaslangicTarihi and OrmeKaliteKontrol.Tarih <= :ABitistarihi) or
             (:AVardiyaTarihineGore = 1 and OrmeKaliteKontrol.VardiyaTarihi >= :ABaslangicTarihi and OrmeKaliteKontrol.VardiyaTarihi <= :ABitistarihi)) and
            (OrmeKaliteKontrol.OrmePlanlamaId = :AOrmePlanlamaId or :AOrmePlanlamaId is null) and
            (OrmeKaliteKontrol.Sakat = :ASakat or :ASakat is null) and
            (OrmePlanlama.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
            DokumaUretimeOtomatikCikis.OrmeKaliteKontrolId is not null
      into :OrmeKaliteKontrolId, :OrmePlanlamaReceteId, :OrmeFisId, :HesaplananMiktar, :CikilanMiktar, :TamamiCikildi, :Aciklama, :HamId, :HamAdi, :Lot, :RenkId, :RenkAdi, :FisNo, :KumasHamId,
      :Tarih do
    begin
      TamamiCikildiStr = '';
      if (TamamiCikildi = 1) then
        TamamiCikildiStr = 'Evet';

      suspend;
    end
  end
end
