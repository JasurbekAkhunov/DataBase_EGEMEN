CREATE PROCEDURE ORMEIPLIKFIRERAPORU_1(
    ABASLANGICGIRISTARIHI TIMESTAMP,
    ABITISGIRISTARIHI TIMESTAMP,
    AHAMID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ARAFID VARCHAR(10),
    ADEPOID VARCHAR(10),
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    ASTOKGIRISNO INTEGER,
    ALOT VARCHAR(150))
RETURNS (
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    RAFID VARCHAR(10),
    DEPOID VARCHAR(10),
    URETIMPARTINO VARCHAR(30),
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    URETICIADI VARCHAR(40),
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    KAPADEDI NUMERIC(12,3),
    MIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    BRUTKALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI INTEGER)
AS
declare variable BoyaSiparisYil integer;
declare variable BoyaSiparisId  integer;

declare variable hstIplik       integer;
declare variable Okey           integer;
declare variable Ks             integer;
declare variable HamGrup1       varchar(4);
declare variable HamGrup2       varchar(4);
declare variable HamGrup3       varchar(4);
declare variable HamGrup4       varchar(4);
begin
  Select hstIplik from HamTipiGetir into :hstIplik;

  For Select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.HamId, Ham.Adi, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.RafId, DepoRaf.DepoId,
             OrmeRaf.UretimPartiNo, OrmeRaf.GirisKapAdedi, OrmeRaf.NetGirisMiktari, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
             OrmeRaf.BrutGirisMiktari, OrmeRaf.BrutGirisMiktari - OrmeRaf.BrutCikisMiktari,
             OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi, OrmeRaf.Lot, OrmeRaf.MelanjKodu, OrmeRaf.FlamKodu, Uretici.Adi,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, OrmeRaf.BoyaSiparisDetayId, OrmeRaf.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo,
             BoyaSiparis.FirmaId, SiparisFirma.Adi, Ham.HamGrup1, Ham.HamGrup2, Ham.HamGrup3, Ham.HamGrup4,
             Ham.Adi2, Ham.Adi3, Ham.Adi4 from OrmeRaf
    left join Ham on Ham.HamId = OrmeRaf.HamId
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join uretici on Uretici.UreticiId = OrmeRaf.UreticiID
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
    left join Firma SiparisFirma on SiparisFirma.Firmaid = BoyaSiparis.FirmaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = OrmeRaf.BoyaSiparisDetayId
    left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
    where (OrmeRaf.HamId = :AHamId or :AHamId is null) and OrmeRaf.FasonStok = 0 and Ham.HamTipiId = :hstIplik and
         (OrmeRaf.RafId = :ARafId or :ARafId is null) and
         (DepoRaf.DepoId = :ADepoId or :ADepoId is null) and
         (OrmeRaf.StokGirisNo = :AStokGirisNo or :AStokGirisNo is null) and
         (OrmeRaf.Lot = :ALot or :ALot is null)  and
         (Ham.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is null) and
         (Ham.KalinlikId = :AKalinlikId or :AKalinlikId is null) and
         ((:AHamGrup1Aralikli = 1 and HamGrup1 >= :ABaslangicGrup1Id  and (HamGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AHamGrup1Aralikli = 0) or (:AHamGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
         ((:AHamGrup2Aralikli = 1 and HamGrup2 >= :ABaslangicGrup2Id and (HamGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AHamGrup2Aralikli = 0) or (:AHamGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
         ((:AHamGrup3Aralikli = 1 and HamGrup3 >= :ABaslangicGrup3Id and (HamGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AHamGrup3Aralikli = 0) or (:AHamGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
         ((:AHamGrup4Aralikli = 1 and HamGrup4 >= :ABaslangicGrup4Id and (HamGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AHamGrup4Aralikli = 0) or (:AHamGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = ''))
  into :OrmeRafId, :StokGirisNo, :HamId, :HamAdi, :RenkId, :RenkAdi, :RafId, :DepoId, :UretimPartiNo,
       :KapAdedi, :Miktar, :KalanMiktar, :BrutMiktar, :BrutKalanMiktar, :KalanKapAdedi, :Lot, :MelanjKodu, :FlamKodu, :UreticiAdi,
       :BoyaSiparisYil, :BoyaSiparisID, :BoyaSiparisDetayId, :OrmeSiparisDetayId, :SiparisSiraNo, :SiparisFirmaId, :SiparisFirmaAdi,
       :HamGrup1, :HamGrup2, :HamGrup3, :HamGrup4,
       :HamAdi2, :HamAdi3, :HamAdi4 do
  begin
    okey = 1;

    if (AHamGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, HamGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, HamGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, HamGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, HamGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      KS = 0;
      Select Count(*) from OrmeRafDetay
        left join StokHareket on StokHareket.StokHareketID = OrmeRafDetay.StokHareketID
        where OrmeRafDetay.OrmeRafID = :OrmeRafID and
              StokHareket.Tarih >= :ABaslangicGirisTarihi and StokHareket.Tarih <= :ABitisGirisTarihi
        into :KS;

      if (KS = 0) then Okey = 0;
    end

    if (Okey = 1) then
    begin
      BoyaSiparisIdStr = '';

      if (BoyaSiparisYil is not null) then
        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';




      Suspend;
    end
  end
end
