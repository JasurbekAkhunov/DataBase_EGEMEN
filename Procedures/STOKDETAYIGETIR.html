               StokDetay.Notu, VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, VerilenSiparisDetay.SatinAlmaKullanimYeriId, SatinAlmaKullanimYeri.Adi,
               StokDetay.VerilenSiparisDetayId, VerilenSiparisDetay.Kapali,
               VerilenSiparisDetay.Miktar, VerilenSiparisDetay.GelenMiktar - VerilenSiparisDetay.IadeEdilenMiktar,
               StokDetay.AmbalajSekliId, AmbalajSekli.Adi, VerilenSiparisDetay.BirimFiyat, VerilenSiparisDetay.DovizID, StokDetay.DesenId, Desen.Adi,
               StokDetay.HGIsEmriKayitID, StokDetay.HGIsEmriUrunID, StokDetay.HGIsEmriPastalID, Model.ModelID, StokDetay.HGSiparisKayitId, HGSiparis.HGSiparisYil,
               HGSiparis.HGSiparisId, StokDetay.HGSiparisDetayId, HGIsEmri.UretimPartiNo, HGIsEmri.KesimTalimati, StokDetay.Iade, HgIsEmri.OrmeMakinaId, StokDetay.EntegrasyonKodu, HGSiparisDetay.MusteriSiparisNo, StokDetay.PastalNo from StokDetay
    left join Urun on Urun.UrunId = StokDetay.UrunId
    left join Kullanici on Kullanici.KullaniciId  = StokDetay.KullaniciId
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = StokDetay.VerilenSiparisDetayId
    left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
    left join SatinAlmaKullanimYeri on SatinAlmaKullanimYeri.SatinAlmaKullanimYeriId = VerilenSiparisDetay.SatinAlmaKullanimYeriId
    left join Stok on Stok.StokId = StokDetay.StokId
    left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
    left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
    left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = StokDetay.AmbalajSekliId
    left join Desen on Desen.DesenId = StokDetay.DesenId
    left join HGIsEmri on HGIsEmri.HGIsEmriKayitId = StokDetay.HGIsEmriKayitId
    left join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayId = StokDetay.HGSiparisDetayId
    left join Model on Model.ModelKayitID = HGSiparisDetay.ModelKayitID
    left join HGSiparis on HGSiparis.HGSiparisKayitId = StokDetay.HGSiparisKayitId
    where StokDetay.FaturaId = :AFaturaId
      order by StokDetayId
    into :StokDetayId, :FisNo, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi, :FaturaNo, :FaturaTarihi, :DepoId, :TransferDepoId, :UrunId, :UrunAdi, :UrunBirimId, :BirimId, :Carpan, :UrunTipiId, :Kdv, :KapAdedi, :Miktar, :BirimFiyat, :KdvHaricFiyat,
         :BirimIskontoTutar, :BirimIskontoOrani1, :BirimIskontoOrani2, :BirimIskontoOrani3, :BirimIskontoOrani4, :BirimIskontoToplami,
         :NetBirimFiyat, :Vadesi, :Lot, :ReferansStokDetayId,  :KullaniciAdi, :DegisimTarihi, :StokId, :IrsaliyeId, :FaturaId, :FAltIskontoCarpani, :FFKdvYuvarlamaCarpani, :FFNetFiyatYuvarlamaCarpani,
         :Notu, :VerilenSiparisYil, :VerilenSiparisId, :SatinAlmaKullanimYeriId, :SatinAlmaKullanimYeriAdi,
         :VerilenSiparisDetayId, :VerilenSiparisDetayKapali, :VerilenSiparisMiktari, :VerilenSiparisGelenMiktar, :AmbalajSekliId, :AmbalajSekliAdi, :VerilenSipBirimFiyat, :VerilenSipDovizID,
         :DesenId, :DesenAdi, :HGIsEmriKayitID, :HGIsEmriUrunID, :HGIsEmriPastalID, :ModelID, :HGSiparisKayitId, :HGSiparisYil, :HGSiparisId, :HGSiparisDetayId, :UretimPartiNo, :KesimTalimati, :Iade, :OrmeMakinaId,
         :EntegrasyonKodu, :MusteriSiparisNo, :PastalNo do
    begin
      BirimIskontoToplami = BirimIskontoToplami * Miktar;
      BrutTutar = KdvHaricFiyat * Miktar;
      BrutTutar = Yuvarla(BrutTutar, FKdvYuvarlamaCarpani);

      if (FAltIskontoCarpani is null) then FAltIskontoCarpani = 0;
      if (FFKdvYuvarlamaCarpani is not null) then FKdvYuvarlamaCarpani = FFKdvYuvarlamaCarpani; /* Faturanin carpanini kullansin */
      if (FFNetFiyatYuvarlamaCarpani is not null) then FNetFiyatYuvarlamaCarpani = FFNetFiyatYuvarlamaCarpani; /* Faturanin carpanini kullansin */

      T1 = NetBirimFiyat * Miktar;
      T1 = T1 - T1 * FAltIskontoCarpani;
      T1 = Yuvarla(T1, FKdvYuvarlamaCarpani);
      Tutar = T1;

      KdvTutari = T1 * Kdv / 100;
      KdvTutari = Yuvarla(:KdvTutari, :FKdvYuvarlamaCarpani);

      VerilenSiparisIdStr = '';
      if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || EgemenIntegerTostr(VerilenSiparisYil) || ') '  || EgemenIntegerTostr(VerilenSiparisId)  || ' ';

      HGSiparisIdStr = '';
      if (HGSiparisId is not null) then HGSiparisIdStr = '(' || EgemenIntegerTostr(HGSiparisYil) || ') '  || EgemenIntegerTostr(HGSiparisId)  || ' ';
      UretimIslemleri = '';
      Select UretimIslemleriID from HGIsEmriUretimIslemiGetir(:HGIsEmriKayitID, :KesimTalimati) into :UretimIslemleri;

      suspend;
    end
  end
