CREATE PROCEDURE DOKUMAURETIMHATAKAYDET(
    ADOKUMAURETIMHATAID INTEGER,
    AORMEFISID INTEGER,
    ASTOKHAREKETID INTEGER,
    ADOKUMAHATATIPIID INTEGER,
    AHATASAYISI INTEGER,
    AKULLANICIID INTEGER)
AS
begin
  if (ADokumaUretimHataID > 0) then
    Update DokumaUretimHata set HataSayisi = :AHataSayisi, KullaniciID = :AKullaniciID Where DokumaUretimHataID = :ADokumaUretimHataID;
  else
    Insert Into DokumaUretimHata (OrmeFisID, StokHareketID, DokumaHataTipiId, HataSayisi, InsertKullaniciId, KullaniciId) Values
                                 (:AOrmeFisID, :AStokHareketID, :ADokumaHataTipiId, :AHataSayisi, :AKullaniciID, :AKullaniciID);
end
