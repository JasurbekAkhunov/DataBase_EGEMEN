begin
  /*uretilen hamin DokumaTalimatRecetesiGetir cozgulerinin recetesi*/
  for select DokumaUretimRaporu3.StokHareketId,
             DokumaTalimatRecetesiGetir2.DokumaPlanlamaReceteId, DokumaTalimatRecetesiGetir2.hamid, DokumaTalimatRecetesiGetir2.hamid,
             DokumaTalimatRecetesiGetir2.hamadi, DokumaTalimatRecetesiGetir2.kullanimorani, DokumaTalimatRecetesiGetir2.TahminiFireOrani,

             DokumaTalimatRecetesiGetir2.atkicozgu, DokumaTalimatRecetesiGetir2.atkicozgustr, DokumaTalimatRecetesiGetir2.telsayisi,
             DokumaTalimatRecetesiGetir2.sirano, DokumaTalimatRecetesiGetir2.ipliksikligicm, DokumaTalimatRecetesiGetir2.agirlikkilo,
             DokumaTalimatRecetesiGetir2.uzunlukmetre, DokumaTalimatRecetesiGetir2.hatastr,
             OrmeHamStokSonAlisFiyati_v2.SonAlisFiyati, OrmeHamStokSonAlisFiyati_v2.SonAlisDovizId, DokumaTalimatRecetesiGetir2.AtkiIplikCm
      from DokumaUretimRaporu3(:ABaslangicTarihi, :ABitisTarihi, :AUretimIslemTipiId, :ABaslangicMakinaId, :ABitisMakinaId,
            :AMakinaKoduAralikli, :AMakinaGrupId, :AHamId)
      left join DokumaTalimatRecetesiGetir2(DokumaUretimRaporu3.ReceteHamId) on 1=1
      left join OrmeHamStokSonAlisFiyati_v2(DokumaTalimatRecetesiGetir2.hamid, DokumaUretimRaporu3.Tarih) on 1=1
      where DokumaUretimRaporu3.AtkiCozgu = 2 and DokumaTalimatRecetesiGetir2.HamId is not null
      into :StokHareketId, :DokumaPlanlamaReceteId, :hamid, :recetehamid, :recetehamadi, :kullanimorani, :fireorani, :cozgufireorani, :renkid,
           :renkadi, :atkicozgu, :atkicozgustr, :telsayisi, :sirano, :ipliksikligicm, :agirlikkilo, :uzunlukmetre, :hatastr, :SonAlisFiyati, :SonAlisDovizId, :AtkiIplikCm do
