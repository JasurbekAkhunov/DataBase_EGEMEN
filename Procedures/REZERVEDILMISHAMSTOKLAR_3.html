CREATE PROCEDURE REZERVEDILMISHAMSTOKLAR_3(
    ASIPARISTIPIID INTEGER,
    AONGORUONAY INTEGER,
    AIMALATTIPIID INTEGER,
    AISLEMTIPIID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEATISID INTEGER,
    LABRECETEATISNO VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    RENKADI VARCHAR(40),
    FIKSETIPIADI VARCHAR(20),
    SIPARISKILO NUMERIC(12,3),
    SIPARISBRUTKILO NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    IPTALKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    IPTALMETRE NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    HAMTIPIID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    PUS INTEGER,
    FEINE INTEGER,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    PROSESID INTEGER,
    PROSESADI VARCHAR(255),
    ISTENENRAPORBOYU VARCHAR(100),
    ISTENENOZELTUSE VARCHAR(25),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    DONMEISTENEN VARCHAR(10),
    CEKMEZLIKENISTENEN DOUBLE PRECISION,
    CEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    YIKAMAHASLIGIISTENEN VARCHAR(10),
    LOT VARCHAR(30),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    DENYE INTEGER,
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIIDSTR VARCHAR(20),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    PAKETLEMESEKLIID INTEGER,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    TERMINNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PLANLAMANOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ISLEMGRUBUID INTEGER,
    SIPARISTIPIID INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSKODU HAMKODUSIZE)
AS
declare variable FikseTipiId integer;
declare variable KS integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable PartilenenMiktardanIptaliDus EgeBoolean;

begin
  Select DokumaKumasiKilodanTakipEt, PartilenenMiktardanIptaliDus from Sistem into :DokumaKumasiKilodanTakipEt, :PartilenenMiktardanIptaliDus;

  for select BoyaSiparisDetay.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.LabReceteId,
             LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.DesenId, Desen.Adi, LabRecete.FikseTipiId, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.LabReceteAtisId, LabReceteAtis.AtisNoStr, BoyaSiparisDetay.ProsesId, Proses.Adi,
             BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisBrutKilo, BoyaSiparisDetay.PartilenenKilo, BoyaSiparisDetay.IptalKilo,
             BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.PartilenenMetre, BoyaSiparisDetay.IptalMetre, BoyaSiparisDetay.IstenenRaporBoyu, BoyaSiparisDetay.IstenenOzelTuse, BoyaSiparisDetay.PaketlemeSekliId,
             BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, BoyaSiparisDetay.VadeGun, BoyaSiparisDetay.HamId, Ham.Adi, Ham.HamTipiId,
             BoyaSiparisDetay.Pus, BoyaSiparisDetay.Feine, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen,
             BoyaSiparisDetay.DonmeIstenen, BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen, BoyaSiparisDetay.YikamaHasligiIstenen, BoyaSiparisDetay.TerminTarihi,
             BoyaSiparisDetay.RevizeTerminTarihi, BoyaSiparisDetay.TerminNotu, BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu, BoyaSiparisDetay.IslemGrubuId,
             BoyaSiparisDetay.RenkDerinligiId, RenkDerinligi.Adi, BoyaSiparisDetay.Lot, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo,
             BoyaSiparisDetay.MusteriKaliteAdi, BoyaSiparisDetay.Denye, BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi,
             BoyaSiparisDetay.OnayDurumu, BoyaSiparisDetay.OnayTarihi, BoyaSiparisDetay.OnaylayanKullaniciId, Ok.Adi, BoyaSiparisDetay.SiparisTipiId,
             IK.Adi, BoyaSiparisDetay.InsertTarihi, Sk.Adi, BoyaSiparisDetay.DegisimTarihi, Ham.ReferansKodu
             from RezervEdilmisHamStoklar_2(:ASiparisTipiId, :AOngoruOnay, :AImalatTipiId, :AIslemTipiId)
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = RezervEdilmisHamStoklar_2.BoyaSiparisKayitId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join Proses on Proses.ProsesId = BoyaSiparisDetay.ProsesId
  left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
  left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
  left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
  Left Join Kullanici OK on OK.KullaniciId = BoyaSiparisDetay.OnaylayanKullaniciId
  Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisDetay.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisDetay.KullaniciId
  where (BoyaSiparisDetay.SiparisTipiId = :ASiparisTipiId or :ASiparisTipiId is null)
  into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :MusteriSiparisNo, :LabReceteId, :LabReceteKodu, :LabReceteAdi, :DesenId, :DesenAdi, :FikseTipiId, :RenkAdi, :LabReceteAtisId, :LabReceteAtisNo,
       :ProsesId, :ProsesAdi, :SiparisProsesId, :SiparisProsesAdi, :SiparisKilo, :SiparisBrutKilo, :PartilenenKilo, :IptalKilo, :SiparisMetre, :PartilenenMetre, :IptalMetre, :IstenenRaporBoyu, :IstenenOzelTuse, :PaketlemeSekliId, :BirimFiyat,
       :DovizId, :VadeGun, :HamId, :HamAdi, :HamTipiId, :Pus, :Feine, :EnIstenen, :GramajIstenen, :DonmeIstenen, :CekmezlikEnIstenen,
       :CekmezlikBoyIstenen, :YikamaHasligiIstenen,  :TerminTarihi, :RevizeTerminTarihi, :TerminNotu, :Notu, :PlanlamaNotu, :IslemGrubuId,
       :RenkDerinligiId, :RenkDerinligiIdStr, :Lot, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
       :Denye, :IslemTipiId, :IslemTipiAdi, :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :SiparisTipiId,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :ReferansKodu do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    FikseTipiAdi = '';

    if (FikseTipiId = 0) then FikseTipiAdi = 'Ã–n Fikse';
    else if (FikseTipiId = 1) then FikseTipiAdi = 'Kasar Sonu Fikse';

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then KalanMiktar = SiparisKilo - PartilenenKilo + IptalKilo;
    else KalanMiktar = SiparisMetre - PartilenenMetre + IptalMetre;

    KS = 0;
    Select Count(*) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayID = :BoyaSiparisDetayID and BoyaSiparisRezerv.RezervVar = 1 into :KS;
    if (KS > 0) then suspend;

    if (PartilenenMiktardanIptaliDus = 1) then
    begin
      PartilenenKilo = PartilenenKilo - IptalKilo;
      PartilenenMetre = PartilenenMetre - IptalMetre;
    end
  end
end
