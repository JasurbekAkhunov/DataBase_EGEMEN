CREATE PROCEDURE LABRECETEONCALISMAURUNGETIR(
    ALABRECETEID INTEGER)
RETURNS (
    LABRECETEONCALISMAURUNID INTEGER,
    LABRECETECALISMAID INTEGER,
    LABRECETEONCALISMAATISID INTEGER,
    LABRECETEID INTEGER,
    SIRA INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    KIMYASALTIPIID INTEGER,
    KIMYASALTIPIADI VARCHAR(20),
    PIPETLEMETRANSFERKODU VARCHAR(55),
    MIKTARYUZDE DOUBLE PRECISION,
    HESAPBIRIMID VARCHAR(5),
    BIRIMID VARCHAR(2),
    PIPETLEMEBIRIMKODU VARCHAR(10),
    URUNTIPIID INTEGER,
    BOYATIPIID INTEGER,
    BIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select LabReceteOnCalismaUrun.LabReceteOnCalismaUrunID, LabReceteOnCalismaUrun.LabReceteOnCalismaAtisId, LabReceteOnCalismaUrun.LabReceteId,
   LabReceteOnCalismaUrun.UrunId, Urun.Adi, Urun.PipetlemeTransferKodu, Urun.KimyasalTipiId, KimyasalTipi.Adi, LabReceteOnCalismaUrun.MiktarYuzde, LabReceteOnCalismaUrun.BirimFiyat, LabReceteGrup.Caption, UrunBirim.BirimId,
   ((MiktarYuzde * LabReceteGrup.Carpan) * BirimFiyat  / FiyatCarpan),
   IK.Adi, LabReceteOnCalismaAtis.InsertTarihi, SK.Adi, LabReceteOnCalismaAtis.DegisimTarihi, Birim.PipetlemeBirimKodu, Urun.UrunTipiId, LabReceteOnCalismaUrun.Sira, LabReceteOnCalismaAtis.LabReceteCalismaID, Urun.BoyaTipiID  from LabReceteOnCalismaAtis
  left join LabRecete on LabRecete.LabReceteId = LabReceteOnCalismaAtis.LabReceteId
  left join LabReceteGrup on LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
  left join LabReceteOnCalismaUrun on LabReceteOnCalismaUrun.LabReceteOnCalismaAtisId = LabReceteOnCalismaAtis.LabReceteOnCalismaAtisId
  left join Urun on Urun.UrunId = LabReceteOnCalismaUrun.UrunId
  left join UrunBirim on UrunBirim.UrunBirimId = LabReceteOnCalismaUrun.UrunBirimId
  left join Birim on Birim.BirimId = UrunBirim.BirimId
  left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = urun.KimyasalTipiid
  Left Join Kullanici IK on IK.KullaniciId = LabReceteOnCalismaUrun.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = LabReceteOnCalismaUrun.KullaniciId
  where LabReceteOnCalismaAtis.LabReceteId = :ALabReceteId and LabReceteOnCalismaUrun.LabReceteOnCalismaUrunId is not null
  Order By LabReceteOnCalismaUrun.Sira
  into :LabReceteOnCalismaUrunId, :LabReceteOnCalismaAtisId, :LabReceteId, :UrunId, :UrunAdi, :PipetlemeTransferKodu,
       :KimyasalTipiId, :KimyasalTipiAdi, :MiktarYuzde, :BirimFiyat, :HesapBirimId, :BirimId,
       :Tutar, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :PipetlemeBirimKodu, :UrunTipiId, :Sira, :LabReceteCalismaID, :BoyaTipiID do
  begin
    suspend;
  end
end
