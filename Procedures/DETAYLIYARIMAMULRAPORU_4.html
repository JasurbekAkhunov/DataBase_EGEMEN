CREATE PROCEDURE DETAYLIYARIMAMULRAPORU_4(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIH TIMESTAMP,
    ABITISTARIH TIMESTAMP,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    AFASON INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(50),
    REZERVKALANKILO NUMERIC(12,3),
    REZERVKALANMETRE NUMERIC(12,3),
    RAFDETAYID INTEGER)
AS
declare variable FBoyaSiparisYil integer;
declare variable FBoyaSiparisId  integer;
begin
/*yari mamulden hangi siparislere rezerv edilmis oldugunu g√∂steriyoruz */
  BoyaSiparisIdStr  = null;

  for select DetayliYariMamulRaporu_2.RafDetayId, DetayliYariMamulRaporu_2.PartiSinifId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId,Firma.Adi,
     (BoyaSiparisRezerv.Kilo - BoyaSiparisRezerv.KullanilanKilo), (BoyaSiparisRezerv.Metre - BoyaSiparisRezerv.KullanilanMetre) from DetayliYariMamulRaporu_2(:ARaporTipi, :ABaslangicTarih, :ABitisTarih, :ABaslangicPartiYil, :ABaslangicPartiId, :ABitisPartiYil, :ABitisPartiId, :AHamTipiId, :AHamId, :AFason)
    left join RafDetay on RafDetay.RafDetayId = DetayliYariMamulRaporu_2.RafDetayId
    left join BoyaSiparisRezerv on BoyaSiparisRezerv.HamStokDetayId = RafDetay.HamStokDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisRezerv.BoyaSiparisKayitId
    left join Firma on Firma.FirmaID = BoyaSiparis.FirmaID
    where BoyaSiparisRezerv.HamStokDetayId is not null and BoyaSiparisRezerv.RezervVar = 1
    into :RafDetayId, :PartiSinifId, :FBoyaSiparisYil, :FBoyaSiparisId, :FirmaId, :FirmaAdi, :RezervKalanKilo, :RezervKalanMetre do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') '  || EgemenIntegerToStr(FBoyaSiparisId);
    suspend;
  end
end
