CREATE PROCEDURE BOYATALIMATRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASIPARISFIRMAID VARCHAR(20),
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ATALIMATFIRMAID VARCHAR(20),
    ATALIMATTIPIID INTEGER,
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISIALANKULLANICIID INTEGER)
RETURNS (
    BOYATALIMATKAYITID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYATALIMATNO VARCHAR(30),
    TALIMATTIPIID INTEGER,
    TALIMATTIPIIDSTR VARCHAR(10),
    TARIH TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISSIRANO INTEGER,
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    HAMRENKADI VARCHAR(50),
    RECETEKODU VARCHAR(20),
    LABRECETEKODU VARCHAR(20),
    SIPRENKADI VARCHAR(40),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    TALIMATPROSESID INTEGER,
    TALIMATPROSESADI VARCHAR(255),
    SIPARISKILO DOUBLE PRECISION,
    TALIMATMIKTARI DOUBLE PRECISION,
    GIDENMIKTAR DOUBLE PRECISION,
    IADEALINANMIKTAR DOUBLE PRECISION,
    BRUTGELENMIKTAR DOUBLE PRECISION,
    NETGELENMIKTAR DOUBLE PRECISION,
    BOYAHANEDEKALANMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KAPALI INTEGER,
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    PUS INTEGER,
    FEINE INTEGER,
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    SIPOZELTESTLER VARCHAR(255),
    SIPOZELAPRELER VARCHAR(255),
    SIPILAVEISLEMLER VARCHAR(255),
    TERMINTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable BaskiKilo Numeric(12,3);
begin
  for Select BoyaTalimat.BoyaTalimatKayitId, BoyaTalimat.BoyaSiparisKayitId, BoyaTalimat.BoyaSiparisDetayId, BoyaSiparisDetay.OrmeSiparisDetayId, BoyaTalimat.BoyaTalimatNo,
         BoyaTalimat.TalimatTipiId, BoyaTalimat.Tarih, BoyaTalimat.FirmaId, BTFirma.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.FirmaId, ASF.Adi,
         BoyaSiparisDetay.HamId, Ham.Adi, TalimatRenk.Adi, HamRenk.Adi, BoyaTalimat.ReceteKodu, BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.RenkAdi,
         BoyaSiparisDetay.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
         BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.ArtikelNo, TalimatProses.SiparisProsesId, TalimatProses.Adi,
         BoyaSiparisDetay.SiparisKilo, BoyaSiparisBaski.Kilo,
         BoyaTalimat.Miktar, BoyaTalimat.GidenMiktar, BoyaTalimat.IadeAlinanMiktar, BoyaTalimat.BrutGelenMiktar, BoyaTalimat.NetGelenMiktar,
         BoyaTalimat.BirimFiyat, BoyaTalimat.DovizId, BoyaTalimat.VadeGun, BoyaTalimat.Notu, BoyaTalimat.Kapali, BoyaTalimat.TerminTarihi,
         BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.OzelApreler, BoyaSiparisDetay.OzelTestler,
         BoyaSiparisDetay.Pus, BoyaSiparisDetay.Feine, OrmeSiparisDetay.TupAcikEnMayli, BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi,
         BoyaTalimat.InsertTarihi, BoyaTalimat.DegisimTarihi, Ik.Adi, Sk.Adi from BoyaTalimat
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaTalimat.BoyaSiparisDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    left join Renk TalimatRenk on TalimatRenk.RenkId = Boyatalimat.RenkId
    left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
    left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
    left join SiparisProses TalimatProses on TalimatProses.SiparisProsesId = BoyaTalimat.SiparisProsesId
    left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisBaskiId = BoyaTalimat.BoyaSiparisBaskiId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
    left join Firma BTFirma on BTFirma.FirmaId = BoyaTalimat.FirmaId
    left join Firma ASF on ASF.FirmaId = BoyaSiparis.FirmaId
    Left Join Kullanici IK on IK.KullaniciId = Boyatalimat.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = Boyatalimat.KullaniciId
    left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
    where (BoyaTalimat.Tarih >= :ABaslangicTarihi and BoyaTalimat.Tarih <= :ABitisTarihi) and
          (BoyaTalimat.FirmaId = :ATalimatFirmaId or :ATalimatFirmaId is null) and
          (BoyaSiparis.FirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
          (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
          (BoyaTalimat.Kapali = :AKapali or :AKapali is null) and
          (BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
          (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil  and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId)) and
          (BoyaTalimat.TalimatTipiId = :ATalimatTipiId or :ATalimatTipiId is null) and
          ( ( BoyaTalimat.Miktar > (BoyaTalimat.GidenMiktar - BoyaTalimat.IadeAlinanMiktar) and :ARaporTipi = 0) or
            ( (BoyaTalimat.GidenMiktar - BoyaTalimat.IadeAlinanMiktar - BoyaTalimat.BrutGelenMiktar) <> 0 and :ARaporTipi = 1) or :ARaporTipi is null) and
          (BoyaSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null)
    order by BoyaTalimat.FirmaId, BoyaTalimat.BoyaTalimatNo
    into :BoyaTalimatKayitId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :OrmeSiparisDetayId, BoyaTalimatNo, :TalimatTipiId, :Tarih, :FirmaId, :FirmaAdi, :BoyaSiparisYil, :BoyaSiparisId,
         :SiparisSiraNo, :SiparisFirmaId, :SiparisFirmaAdi, :HamId, :HamAdi, :RenkAdi, :HamRenkAdi, :ReceteKodu, :LabReceteKodu, :SipRenkAdi,
         :DesenId, :DesenAdi, :VaryantAdi, :MusteriSiparisNo, :ReferansSiparisNo, :ArtikelNo, :TalimatProsesId, :TalimatProsesAdi,
         :SiparisKilo, :BaskiKilo, :TalimatMiktari, :GidenMiktar, :IadeAlinanMiktar, :BrutGelenMiktar, :NetGelenMiktar,
         :BirimFiyat, :DovizId, :VadeGun, :Notu, :Kapali, :TerminTarihi, :SipIlaveIslemler, :SipOzelApreler, :SipOzelTestler,
         :Pus, :Feine, :TupAcikEnMayli, :IslemTipiId, :IslemTipiAdi, :InsertTarihi, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    BoyahanedeKalanMiktar = GidenMiktar - IadeAlinanMiktar - BrutGelenMiktar;

    if (TalimatTipiId = 0) then TalimatTipiIdStr = 'Boya';
    else if (TalimatTipiId = 1) then
    begin
      TalimatTipiIdStr = 'Boya+Baskı';
      SiparisKilo = BaskiKilo;
    end
    else if (TalimatTipiId = 2) then
    begin
      TalimatTipiIdStr = 'Baskı';
      SiparisKilo = BaskiKilo;
    end

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    suspend;
  end
end
