CREATE PROCEDURE RISKANALIZIRAPORU(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    ARAPORTARIHI TIMESTAMP,
    ARAPORDOVIZID VARCHAR(5))
RETURNS (
    FIRMAID VARCHAR(20),
    ADI VARCHAR(100),
    ADI2 VARCHAR(100),
    FIRMAGRUP1 VARCHAR(4),
    FIRMAGRUP2 VARCHAR(4),
    FIRMAGRUP3 VARCHAR(4),
    ACIKHESAPRISKLIMITI DOUBLE PRECISION,
    CEKSENETRISKLIMITI DOUBLE PRECISION,
    SIPARISRISKLIMITI DOUBLE PRECISION,
    RISKDOVIZID VARCHAR(5),
    DOVIZACIKHESAPRISKLIMITI DOUBLE PRECISION,
    DOVIZCEKSENETRISKLIMITI DOUBLE PRECISION,
    DOVIZSIPARISRISKLIMITI DOUBLE PRECISION,
    VADESIGECENRISK DOUBLE PRECISION,
    VADESIGELMEYENRISK DOUBLE PRECISION,
    CARIRISKI DOUBLE PRECISION,
    FIRMADOVIZID VARCHAR(5),
    DOVIZCARIRISKI DOUBLE PRECISION,
    SIPARISRISKI DOUBLE PRECISION,
    TOPLAMRISK DOUBLE PRECISION,
    RAPORDOVIZID VARCHAR(5))
AS
declare variable okey           integer;
declare variable BakiyeBa       varchar(1);
declare variable FDovizId       varchar(5);
declare variable FMeblag        Double Precision;
declare variable FDovizKuru     Double Precision;
begin
  RaporDovizId = ARaporDovizId;
  for Select FirmaId, Adi, Adi2, FirmaGrup1, FirmaGrup2, FirmaGrup3, AcikHesapRiski, CekSenetRiski, SiparisRiski, Firma.DovizId, Firma.RiskDovizId from Firma
  where ((:AFirmaAralikli = 1 and FirmaId >= :ABaslangicFirmaId and (FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0))  and
        ((:AFirmaGrup1Aralikli = 1 and FirmaGrup1 >= :ABaslangicGrup1Id  and (FirmaGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and FirmaGrup2 >= :ABaslangicGrup2Id and (FirmaGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and FirmaGrup3 >= :ABaslangicGrup3Id and (FirmaGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = ''))
  into :FirmaId, :Adi, :Adi2, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :AcikHesapRiskLimiti, :CekSenetRiskLimiti, :SiparisRiskLimiti, :FirmaDovizId, :RiskDovizId do
  begin
    Okey = 1;
    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (okey = 1) then
    begin
      VadesiGecenRisk = 0;
      VadesiGelmeyenRisk = 0;
      CariRiski = 0;
      SiparisRiski = 0;
      DovizCariRiski = 0;
      ToplamRisk = 0;
      if (RiskDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:RiskDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
      else FDovizKuru = 1;

      DovizAcikHesapRiskLimiti = AcikHesapRiskLimiti * FDovizKuru;
      DovizCekSenetRiskLimiti = CekSenetRiskLimiti * FDovizKuru;
      DovizSiparisRiskLimiti = SiparisRiskLimiti * FDovizKuru;

      select Bakiye, BakiyeBa from FirmaBakiyeGetir(:FirmaId, null, Null, Null) into :CariRiski, :BakiyeBa;
      if (CariRiski is null) then CariRiski = 0;
      else if (BakiyeBA = 'A') then CariRiski = CariRiski * -1;

      if (FirmaDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FirmaDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
      else FDovizKuru = 1;

      DovizCariRiski = CariRiski * FDovizKuru;

      for Select CekSenetDetay.DovizId, Sum(CekSenetDetay.Meblag - CekSenetDetay.TahsilEdilenMeblag)  from CekSenet
        left join CekSenetDetay on CekSenetDetay.CsBordroKayitId = CekSenet.CsBordroKayitId
        where  CekSenet.FirmaId = :FirmaId and CekSenetDetay.VadeTarihi >= :ARaporTarihi and
              (CekSenetDetay.SonBordroTipi = 0 or CekSenetDetay.SonBordroTipi = 1 or CekSenetDetay.SonBordroTipi = 2 or CekSenetDetay.SonBordroTipi = 3 or CekSenetDetay.SonBordroTipi = 4 or CekSenetDetay.SonBordroTipi = 6) and
               CekSenetDetay.Av = 'A' and (CekSenetDetay.Meblag > CekSenetDetay.TahsilEdilenMeblag)
        group by CekSenetDetay.DovizId
        into :FDovizId, :FMeblag do
      begin
        if (FDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
        else FDovizKuru = 1;
        FMeblag = FMeblag * FDovizKuru;
        VadesiGelmeyenRisk = VadesiGelmeyenRisk + FMeblag;
      end

      for Select CekSenetDetay.DovizId, Sum(CekSenetDetay.Meblag - CekSenetDetay.TahsilEdilenMeblag)  from CekSenet
        left join CekSenetDetay on CekSenetDetay.CsBordroKayitId = CekSenet.CsBordroKayitId
        where  CekSenet.FirmaId = :FirmaId and CekSenetDetay.VadeTarihi < :ARaporTarihi and (CekSenetDetay.Meblag > CekSenetDetay.TahsilEdilenMeblag) and
              (CekSenetDetay.SonBordroTipi = 0 or CekSenetDetay.SonBordroTipi = 2 or CekSenetDetay.SonBordroTipi = 3 or CekSenetDetay.SonBordroTipi = 4 or CekSenetDetay.SonBordroTipi = 6) and
               CekSenetDetay.Av = 'A'
        group by CekSenetDetay.DovizId
        into :FDovizId, :FMeblag do
      begin
        if (FDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
        else FDovizKuru = 1;
        FMeblag = FMeblag * FDovizKuru;
        VadesiGecenRisk = VadesiGecenRisk + FMeblag;
      end

/*
      For Select Sum((AlinanSiparisDetay.HamMiktar - AlinanSiparisDetay.ToplamSatis) * AlinanSiparisDetay.SatisFiyati), AlinanSiparisDetay.DovizId from AlinanSiparis
        left join AlinanSiparisDetay on AlinanSiparisDetay.AlinanSiparisKayitId = AlinanSiparis.AlinanSiparisKayitId
        left join AlinanSiparisAltDetay on AlinanSiparisAltDetay.AlinanSiparisDetayId = AlinanSiparisDetay.AlinanSiparisDetayId
        where AlinanSiparis.FirmaId = :FirmaId and AlinanSiparis.Kapali = 0 and AlinanSiparis.Iptal = 0 and AlinanSiparisAltDetay.AlinanSiparisAltDetayId is null and
              AlinanSiparisDetay.HamMiktar > AlinanSiparisDetay.ToplamSatis
        group by AlinanSiparisDetay.DovizId
        into :FMeblag, :FDovizId do
      begin
        if (FDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
        else FDovizKuru = 1;
        FMeblag = FMeblag * FDovizKuru;
        SiparisRiski = SiparisRiski + FMeblag;
      end

      For Select Sum((AlinanSiparisAltDetay.Miktar - AlinanSiparisAltDetay.ToplamSatis) * AlinanSiparisAltDetay.SatisFiyati), AlinanSiparisAltDetay.DovizId from AlinanSiparis
        left join AlinanSiparisAltDetay on AlinanSiparisAltDetay.AlinanSiparisKayitId = AlinanSiparis.AlinanSiparisKayitId
        where AlinanSiparis.FirmaId = :FirmaId and AlinanSiparis.Kapali = 0 and AlinanSiparis.Iptal = 0 and AlinanSiparisAltDetay.AlinanSiparisAltDetayId is not null and
              AlinanSiparisAltDetay.Iptal = 0 and AlinanSiparisAltDetay.Miktar > AlinanSiparisAltDetay.ToplamSatis
        group by AlinanSiparisAltDetay.DovizId
        into :FMeblag, :FDovizId do
      begin
        if (FDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
        else FDovizKuru = 1;
        FMeblag = FMeblag * FDovizKuru;
        SiparisRiski = SiparisRiski + FMeblag;
      end
*/

      ToplamRisk = DovizCariRiski + VadesiGecenRisk + VadesiGelmeyenRisk + SiparisRiski;
      if (ToplamRisk > 0) then suspend;
    end
  end
end
