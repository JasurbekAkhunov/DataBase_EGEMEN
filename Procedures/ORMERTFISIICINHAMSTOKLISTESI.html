CREATE PROCEDURE ORMERTFISIICINHAMSTOKLISTESI(
    AHAMID INTEGER)
RETURNS (
    ORMERAFID INTEGER,
    URETICIFIRMAID VARCHAR(20),
    URETICIFIRMAADI VARCHAR(100),
    STOKGIRISNO INTEGER,
    FASONSTOK INTEGER,
    FASONSTOKSTR VARCHAR(5),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RAFID VARCHAR(10),
    RENKID INTEGER,
    HAMRENKADI VARCHAR(50),
    RENKADI VARCHAR(50),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    LOT VARCHAR(150),
    PUS INTEGER,
    FEINE INTEGER,
    MIKTAR NUMERIC(12,3),
    BRUTMIKTAR BOYAMIKTARSIZE,
    KALANBRUTMIKTAR BOYAMIKTARSIZE,
    KALANMIKTAR NUMERIC(12,3),
    KAPADEDI INTEGER,
    ADET INTEGER,
    KALANKAPADEDI INTEGER,
    KALITENO INTEGER,
    MELANJKODU VARCHAR(20),
    ORMEMAKINAID VARCHAR(10),
    DOKUMAMAKINAID MAKINAKODUSIZE,
    URETIMPARTINO VARCHAR(30),
    KALITEACIKLAMATIPIADI VARCHAR(20),
    MAMULPROSESADI VARCHAR(255),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BARKODID VARCHAR(30),
    OLCUID INTEGER,
    OLCUADI ADI40,
    KALANADET INTEGER,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    ORMEMAKINAADI VARCHAR(40),
    ORMERAFSAKAT EGEBOOLEAN,
    ORMERAFSAKATSTR VARCHAR(5))
AS
declare variable CikisMiktari  Numeric(12,3);
declare variable CikisKapAdedi integer;
declare variable FUreticiAdi   varchar(40);
declare variable BrutCikisMiktari BoyaMiktarSize;
begin
  for select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.BarkodId, OrmeRaf.UreticiFirmaId, UreticiFirma.Adi, OrmeRaf.FasonStok, OrmeRaf.HamId, Ham.Adi, OrmeRaf.RafId,
             OrmeRaf.GirisKapAdedi, OrmeRaf.NetGirisMiktari, OrmeRaf.NetCikisMiktari, OrmeRaf.CikisKapAdedi,
             OrmeRaf.RenkId, SonRenk.Adi, OrmeRaf.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, OrmeRaf.Lot, OrmeRaf.KaliteNo, OrmeRaf.MelanjKodu, coalesce(OrmeRaf.OrmeMakinaId, OrmeRaf.OrmeciMakinaNo), OrmeRaf.UretimPartiNo,
             KaliteAciklamaTipi.Adi, MamulProses.Adi, OrmeRaf.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, HamRenk.Adi, Uretici.Adi, OrmeRaf.BrutGirisMiktari, OrmeRaf.BrutCikisMiktari, OrmeRaf.DokumaMakinaId,
             Olcu.OlcuId, Olcu.OlcuAdi, OrmeRaf.GirisAdet - OrmeRaf.CikisAdet, OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4,
             OrmeRaf.GirisAdet, OrmeMakina.Adi, OrmeRaf.Sakat from OrmeRaf
  left join Ham on Ham.HamId = OrmeRaf.hamid
  left join Firma UreticiFirma on UreticiFirma.Firmaid = OrmeRaf.UreticiFirmaId
  left join Renk SonRenk on SonRenk.RenkId = OrmeRaf.Renkid
  Left Join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = OrmeRaf.KaliteAciklamaTipiId
  Left Join MamulProses on MamulProses.MamulProsesId = OrmeRaf.MamulProsesId
  Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
  left join Desen on Desen.DesenId = OrmeRaf.DesenId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = OrmeRaf.DesenVaryantId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
  left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
  left join Uretici on Uretici.UreticiId  = OrmeRaf.UreticiId
  left join Olcu on Olcu.OlcuId = OrmeRaf.OlcuId
  left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmeRaf.OrmeMakinaId
  left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeRaf.OrmePlanlamaId
  where OrmeRaf.HamId = :AHamId and OrmeRaf.StokVar = 1
  order by OrmeRaf.MelanjKodu, OrmeRaf.HamId, OrmeRaf.UretimPartiNo, OrmeRaf.Lot
  into :OrmeRafId, :StokGirisNo, :BarkodId, :UreticiFirmaId, :UreticiFirmaAdi, :FasonStok, :HamId, :HamAdi, :RafId, :KapAdedi, :Miktar, :CikisMiktari, :CikisKapAdedi,
       :RenkId, :RenkAdi, :DesenId, :DesenAdi, :VaryantAdi, :Lot, :KaliteNo, :MelanjKodu, :OrmeMakinaId, :UretimPartiNo,
       :KaliteAciklamaTipiAdi, :MamulProsesAdi, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :Pus, :Feine, :HamRenkAdi, :FUreticiAdi, :BrutMiktar, :BrutCikisMiktari, :DokumaMakinaId,
       :OlcuId, :OlcuAdi, :KalanAdet, :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :Adet, :OrmeMakinaAdi, :OrmeRafSakat do
  begin
    if (FasonStok = 1) then FasonStokStr = 'Fason';
    else FasonStokStr = '';

    if (UreticiFirmaID is null) then UreticiFirmaAdi = FUreticiAdi;

    KalanMiktar = Miktar - CikisMiktari;
    KalanKapAdedi = KapAdedi - CikisKapAdedi;
    KalanBrutMiktar = BrutMiktar - BrutCikisMiktari;

    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    else
      BoyaSiparisIdStr = '';

    if (OrmeRafSakat = 1) then OrmeRafSakatStr = 'Fason';
    else OrmeRafSakatStr = '';

    if (KalanMiktar > 0) then suspend;
  end
end
