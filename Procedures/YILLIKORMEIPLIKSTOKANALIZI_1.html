CREATE PROCEDURE YILLIKORMEIPLIKSTOKANALIZI_1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMID INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    AY INTEGER,
    AYSTR VARCHAR(20),
    AYILKGUN TIMESTAMP,
    GIRIS DOUBLE PRECISION,
    GIRISIADE DOUBLE PRECISION,
    HAMSATIS DOUBLE PRECISION,
    BOYALISATIS DOUBLE PRECISION,
    HAMORGUCIKIS DOUBLE PRECISION,
    BOYALIORGUCIKIS DOUBLE PRECISION,
    BUKUMECIKIS DOUBLE PRECISION,
    BOYAHANEYECIKIS DOUBLE PRECISION,
    DIGERCIKISLAR DOUBLE PRECISION,
    DIGERGIRISLER DOUBLE PRECISION)
AS
declare variable Tarih              TimeStamp;
declare variable HareketTipiID      integer;
declare variable UretimIslemTipiId  integer;
declare variable GirisCikis         integer;
declare variable HamIade            integer;
declare variable RenkID             integer;
declare variable Miktar             Double Precision;

declare variable shtAlis               integer;
declare variable shtSatis              integer;
declare variable shtAlisIade           integer;
declare variable shtSatisIade          integer;
declare variable shtFasonUretimeCikis  integer;
declare variable shtUretimeCikis       integer;
declare variable shtFasonUretim        integer;
declare variable shtUretim             integer;
declare variable ShtKaydirmaCikis      integer;
declare variable ShtKaydirmaGiris      integer;
declare variable ShtSarf               integer;

declare variable uitBukum integer;
declare variable uitOrme integer;
declare variable uitBoya integer;

declare variable OrmeFirmaKodumuz Varchar(20);
begin
  Select OrmeFirmaKodumuz from Sistem where sistemid = 1 into :OrmeFirmaKodumuz;

  select uitBukum, uitOrme, uitBoya from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya;

  select shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, ShtKaydirmaCikis, ShtKaydirmaGiris, ShtSarf from OrmeStokHareketTipiGetir
         into :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim, :ShtKaydirmaCikis, :ShtKaydirmaGiris, :ShtSarf;

  For Select Ham.HamId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4 from Ham
    where Ham.HamTipiID = 0 and Ham.Kapali = 0 and
         (Ham.HamID = :AHamID or :AHamID is null)
    into :HamId, :HamAdi, :HamAdi2, :HamAdi3, :HamAdi4 do
  begin
    For Select StokHareket.Tarih, StokHareket.HareketTipi, StokHareket.UretimIslemTipiId, StokHareket.GirisCikis, StokHareket.HamIade, StokHareket.RenkId, StokHareket.Miktar from StokHareket
      left join OrmeRaf on OrmeRaf.OrmeRafID = StokHareket.OrmeRafID
      where StokHareket.HamID = :HamID and StokHareket.Tarih >= :ABaslangicTarihi and StokHareket.Tarih < :ABitisTarihi and
            (StokHareket.StokFirmaID = :OrmeFirmaKodumuz or OrmeRaf.StokFirmaID = :OrmeFirmaKodumuz)
      order by Tarih
      into :Tarih, :HareketTipiID, :UretimIslemTipiId, :GirisCikis, :HamIade, :RenkId, :Miktar do
    begin
      Ay = f_Month(Tarih);
      if (Ay = 1) then AyStr = 'Ocak';
      else if (Ay = 2) then AyStr = 'Şubat';
      else if (Ay = 3) then AyStr = 'Mart';
      else if (Ay = 4) then AyStr = 'Nisan';
      else if (Ay = 5) then AyStr = 'Mayıs';
      else if (Ay = 6) then AyStr = 'Haziran';
      else if (Ay = 7) then AyStr = 'Temmuz';
      else if (Ay = 8) then AyStr = 'Ağustos';
      else if (Ay = 9) then AyStr = 'Eylül';
      else if (Ay = 10) then AyStr = 'Ekim';
      else if (Ay = 11) then AyStr = 'Kasım';
      else if (Ay = 12) then AyStr = 'Aralık';
      --AyStr = f_cmonthlong(cast(Tarih as date));
      AyIlkGun = TimeStampToAyIlkGun(Tarih);

      Giris               = 0;
      GirisIade           = 0;
      HamSatis            = 0;
      BoyaliSatis         = 0;
      HamOrguCikis        = 0;
      BoyaliOrguCikis     = 0;
      BukumeCikis         = 0;
      BoyahaneyeCikis     = 0;
      DigerCikislar       = 0;
      DigerGirisler       = 0;

      if (HareketTipiID = shtAlis or (HareketTipiID = shtUretim and HamIade = 0) or (HareketTipiID = shtFasonUretim and HamIade = 0) or HareketTipiID = ShtSatisIade) then Giris = Miktar;
      else if (HareketTipiID = shtAlisIade) then GirisIade = Miktar;
      else if (HareketTipiID = shtFasonUretimeCikis or HareketTipiID = shtUretimeCikis) then
      begin
        if (UretimIslemTipiID = uitOrme) then
        begin
          if (RenkID is null) then HamOrguCikis = Miktar;
          else BoyaliOrguCikis     = Miktar;
        end
        else if (UretimIslemTipiID = uitBukum) then BukumeCikis = Miktar;
        else if (UretimIslemTipiID = uitBoya) then BoyahaneyeCikis = Miktar;
      end
      else if (HareketTipiID = shtSatis) then
      begin
        if (RenkID is null) then HamSatis = Miktar;
        else BoyaliSatis = Miktar;
      end
      else if (HamIade = 1 and (HareketTipiID = shtFasonUretim or HareketTipiID = shtUretim) ) then
      begin
        if (UretimIslemTipiID = uitOrme) then
        begin
          if (RenkID is null) then HamOrguCikis = Miktar * -1;
          else BoyaliOrguCikis     = Miktar * -1;
        end
        else if (UretimIslemTipiID = uitBukum) then BukumeCikis = Miktar * -1;
        else if (UretimIslemTipiID = uitBoya) then BoyahaneyeCikis = Miktar * -1;
      end
      else if (GirisCikis = -1) then DigerCikislar = Miktar;
      else if (GirisCikis = 1) then DigerGirisler = Miktar;
      suspend;
    end

    Giris               = 0;
    GirisIade           = 0;
    HamSatis            = 0;
    BoyaliSatis         = 0;
    HamOrguCikis        = 0;
    BoyaliOrguCikis     = 0;
    BukumeCikis         = 0;
    BoyahaneyeCikis     = 0;
    DigerCikislar       = 0;
    DigerGirisler       = 0;

    For Select BoyaIrsaliyeDetay.IrsaliyeTarihi, BoyaIrsaliyeDetay.BrutKilo from BoyaIrsaliyeDetay
      left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
      where BoyaIrsaliyeDetay.HamID = :HamID and BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliyeDetay.IrsaliyeTarihi < :ABitisTarihi and
            BoyaIrsaliyeDetay.PartiSinifID is not null and Parti.Fason = 0
      order by BoyaIrsaliyeDetay.IrsaliyeTarihi
      into :Tarih, :Miktar do
    begin
      Ay = f_Month(Tarih);
      if (Ay = 1) then AyStr = 'Ocak';
      else if (Ay = 2) then AyStr = 'Şubat';
      else if (Ay = 3) then AyStr = 'Mart';
      else if (Ay = 4) then AyStr = 'Nisan';
      else if (Ay = 5) then AyStr = 'Mayıs';
      else if (Ay = 6) then AyStr = 'Haziran';
      else if (Ay = 7) then AyStr = 'Temmuz';
      else if (Ay = 8) then AyStr = 'Ağustos';
      else if (Ay = 9) then AyStr = 'Eylül';
      else if (Ay = 10) then AyStr = 'Ekim';
      else if (Ay = 11) then AyStr = 'Kasım';
      else if (Ay = 12) then AyStr = 'Aralık';
      --AyStr = f_cmonthlong(cast(Tarih as date));
      AyIlkGun = TimeStampToAyIlkGun(Tarih);

      BoyaliSatis = Miktar;
      suspend;
    end
  end
end
