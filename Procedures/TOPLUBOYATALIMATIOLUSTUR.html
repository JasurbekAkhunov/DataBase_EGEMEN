CREATE PROCEDURE TOPLUBOYATALIMATIOLUSTUR(
    AFIRMAID VARCHAR(20),
    ABOYASIPARISDETAYIDSTR VARCHAR(1025),
    AKULLANICIID INTEGER)
AS
declare variable VirgulPos integer;
declare variable Str varchar(20);
declare variable BoyaSiparisDetayId integer;
declare variable SistemDovizId      varchar(5);
declare variable RenkId          integer;
declare variable SiparisProsesId integer;
declare variable Miktar          Numeric(12,3);
declare variable TerminTarihi    TimeStamp;
declare variable SiparisMaximumFireOrani       Double Precision;
declare variable TalimatMiktar   Numeric(12,3);
declare variable HamTipiId integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable Kilo numeric(12,3);
declare variable Metre numeric(12,3);
begin
  Select SistemDovizId, DokumaKumasiKilodanTakipEt from Sistem where sistemId = 1 into :SistemDovizId, :DokumaKumasiKilodanTakipEt;

  VirgulPos = PosBul(ABoyaSiparisDetayIdStr, ',');

  While (VirgulPos > 1) do
  begin
    Str = EgemenCopyStr(ABoyaSiparisDetayIdStr, 1, VirgulPos -1);
    BoyaSiparisDetayId = EgemenIntegerToStr(Str);

    Select BoyaSiparisDetay.SiparisKilo + BoyaSiparisDetay.SiparisKilo * BoyaSiparisDetay.MaximumFireOrani / 100,
           BoyaSiparisDetay.SiparisMetre + BoyaSiparisDetay.SiparisMetre * BoyaSiparisDetay.MaximumFireOrani / 100,
           BoyaSiparisDetay.RenkId, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.SiparisProsesId, Sum(BoyaTalimat.Miktar), Ham.HamTipiId from BoyaSiparisDetay
      left join BoyaTalimat on BoyaTalimat.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
      left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
      where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
      group by BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.MaximumFireOrani, BoyaSiparisDetay.RenkId, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.SiparisProsesId, Ham.HamTipiId
      into :Kilo, :Metre, :RenkId, :TerminTarihi, :SiparisProsesId, :TalimatMiktar, :HamTipiId;

    if (HamTipiID <> 2 or DokumaKumasiKilodanTakipEt = 1) then
    begin
      Miktar  = Kilo;
    end
    else
    begin
       Miktar  = Metre;
    end

    Miktar = Yuvarla(Miktar, 100);

    if (TalimatMiktar is not null) then Miktar = Miktar - TalimatMiktar;

    if (Miktar > 0) then
    begin
      insert into BoyaTalimat(BoyaSiparisDetayId, TalimatTipiId, FirmaId, Tarih, RenkId, SiparisProsesId, Miktar, DovizId, TerminTarihi, InsertKullaniciId, KullaniciId)
                     Values(:BoyaSiparisDetayId, 0, :AFirmaId, Current_date, :RenkId, :SiparisProsesId, :Miktar, :SistemDovizId, :TerminTarihi, :AKullaniciId, :AKullaniciId);

    end
    if (Char_Length(ABoyaSiparisDetayIdStr) > VirgulPos) then ABoyaSiparisDetayIdStr = EgemenCopyStr(ABoyaSiparisDetayIdStr, VirgulPos + 1, Char_Length(ABoyaSiparisDetayIdStr) - VirgulPos);
    else ABoyaSiparisDetayIdStr = '';

    VirgulPos = PosBul(ABoyaSiparisDetayIdStr, ',');
  end
end
