CREATE PROCEDURE MAKINASUREKAYDET(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    APARTIASAMASTOKDETAYID INTEGER)
AS
declare variable RenkDerinligiVar   integer;
declare variable LabRenkGrupVar     integer;
declare variable MigrasyonVar       integer;
declare variable IplikCinsiVar      integer;
declare variable KumasCinsiVar      integer;
declare variable KalinlikVar        integer;
declare variable IlaveIslemlerVar   integer;
declare variable BoyaAsamasimi         integer;
declare variable MiktarSureyiEtkilermi integer;
declare variable SuAnkiSira            integer;
declare variable SonReceteAsamasiSira  integer;
declare variable GirisTarihi           TimeStamp;
declare variable CikisTarihi           TimeStamp;
declare variable AraVermeSuresi        Double Precision;
declare variable HamTipiId             integer;
declare variable MakinaSarjId          integer;

declare variable MakinaSureId          integer;
declare variable MakinaPlanlamaGrupId  varchar(10);
declare variable AsamaId               integer;

declare variable RenkDerinligiId       integer;
declare variable LabRenkgrupId         integer;
declare variable Migrasyon             integer;
declare variable IplikCinsiId          integer;
declare variable KumasCinsiId          integer;
declare variable KalinlikId            integer;
declare variable IlaveIslemler         varchar(100);
declare variable SureParametresiId  integer;
declare variable SarjKilo           Numeric(12,3);
declare variable SarjMetre          Numeric(12,3);
declare variable Sure               Double Precision;
declare variable SureDakika         Double Precision;
declare variable IlaveIslemlerId       integer;
declare variable PartiAsamalariId      integer;
begin
  RenkDerinligiVar = 0;
  LabRenkGrupVar = 0;
  MigrasyonVar = 0;
  IplikCinsiVar = 0;
  KumasCinsiVar = 0;
  KalinlikVar = 0;
  IlaveIslemlerVar = 0;

  Select PartiAsamaStokDetay.PartiAsamalariId, Makina.MakinaPlanlamaGrupId, PartiAsamaStok.AsamaId, Ham.IplikCinsiId, Ham.KalinlikId, LabReceteAtis.RenkDerinligiId, Recete.LabRenkgrupId, Recete.Migrasyon,
         Asama.BoyaAsamasimi, Asama.MiktarSureyiEtkilermi, PartiAsamalari.Sira,
         ham.HamTipiId, PartiAsamaStokDetay.MakinaSarjId from PartiAsamaStokDetay
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join Recete on PartiAsamalari.ReceteId is not null and Recete.ReceteId = PartiAsamalari.ReceteId
    left join Labrecete on Recete.LabreceteId is not null and Labrecete.LabreceteId = Recete.LabreceteId
    left join LabreceteAtis on Recete.LabreceteAtisId is not null and LabreceteAtis.LabreceteAtisId = Recete.LabreceteAtisId
    left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    where PartiAsamaStokDetay.PartiAsamaStokDetayId = :APartiAsamaStokDetayId
    into :PartiAsamalariId, :MakinaPlanlamaGrupId, :AsamaId, :IplikCinsiId, :KalinlikId, :RenkDerinligiId, :LabRenkgrupId,  :Migrasyon,
         :BoyaAsamasimi, :MiktarSureyiEtkilermi, :SuAnkiSira, :HamTipiId, :MakinaSarjId;

  if (BoyaAsamasimi = 0) then
  begin
    RenkDerinligiId = Null;
    LabRenkgrupId = Null;
    Migrasyon = Null;
    SonReceteAsamasiSira = null;

    select Max(PartiAsamalari.Sira) from PartiAsamalari
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStok.PartiSinifId = :APartiSinifId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.Sira < :SuAnkiSira and PartiAsamaStok.AsamayaDahil = 1 and
            PartiAsamalari.ReceteId is not null and Asama.BoyaAsamasimi = 1
      into :SonReceteAsamasiSira;

    if (SonReceteAsamasiSira is not null) then
    begin
      Select LabReceteAtis.RenkDerinligiId, Recete.LabRenkgrupId, Recete.Migrasyon from PartiAsamalari
        left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join LabreceteAtis on LabreceteAtis.LabreceteAtisId = Recete.LabreceteAtisId
        where PartiAsamalari.PartiKayitID = :APartiKayitID and PartiAsamalari.Sira = :SonReceteAsamasiSira
        into :RenkDerinligiId, :LabRenkgrupId, :Migrasyon;
    end
  end

  For Select SureParametresiId from AsamaSureParametresi where AsamaSureParametresi.AsamaId = :AsamaId into :SureParametresiId do
  begin
    if (SureParametresiId = 1) then RenkDerinligiVar = 1;
    else if (SureParametresiId = 2) then LabRenkGrupVar = 1;
    else if (SureParametresiId = 3) then MigrasyonVar = 1;
    else if (SureParametresiId = 4) then IplikCinsiVar = 1;
    else if (SureParametresiId = 5) then KumasCinsiVar = 1;
    else if (SureParametresiId = 6) then KalinlikVar = 1;
    else if (SureParametresiId = 7) then IlaveIslemlerVar = 1;
  end

  if (RenkDerinligiVar = 0) then RenkDerinligiId = Null;
  if (LabRenkGrupVar = 0) then LabRenkGrupId = Null;
  if (MigrasyonVar = 0) then Migrasyon = Null;
  if (IplikCinsiVar = 0) then IplikCinsiID = Null;
  if (KalinlikVar = 0) then KalinlikID = Null;
  if (KumasCinsiVar = 0) then KumasCinsiID = Null;

  IlaveIslemler = Null;

  if (IlaveIslemlerVar = 1) then
  begin
    IlaveIslemler = '';
    for Select PartiIlaveIslemler.IlaveIslemlerId from PartiIlaveIslemler
      where PartiIlaveIslemler.PartiAsamalariID = :PartiAsamalariID
      order by PartiIlaveIslemler.IlaveIslemlerId
      into :IlaveIslemlerId do
    begin
      if (IlaveIslemler <> '') then IlaveIslemler = IlaveIslemler || ',';
      IlaveIslemler = IlaveIslemler || EgemenIntegertoStr(IlaveIslemlerId);
    end
  end

  MakinaSureId = Null;

  Select Max(MakinaSureId) from MakinaSure
    where MakinaSure.MakinaPlanlamaGrupId = :MakinaPlanlamaGrupId and MakinaSure.AsamaId = :AsamaId and
         (IntKarsilastir(MakinaSure.RenkDerinligiId, :RenkDerinligiId) = 1) and
         (IntKarsilastir(MakinaSure.LabRenkgrupId, :LabRenkgrupId) = 1) and
         (IntKarsilastir(MakinaSure.Migrasyon, :Migrasyon) = 1) and
         (IntKarsilastir(MakinaSure.IplikCinsiId, :IplikCinsiId) = 1) and
         (IntKarsilastir(MakinaSure.KalinlikId, :KalinlikId) = 1) and
         (IntKarsilastir(MakinaSure.KumasCinsiId, :KumasCinsiId) = 1) and
         (StrKarsilastir(MakinaSure.IlaveIslemler, :IlaveIslemler) = 1)
    into :MakinaSureId;

  Select Min(PartiAsamaStokDetay.Giristarihi), Max(PartiAsamaStokDetay.CikisTarihi), Max(PartiAsamaStokDetay.AraVermeSuresi), Sum(PartiAsamaStokDetay.Kilo), Sum(PartiAsamaStokDetay.Metre) from PartiAsamaStokDetay
    where PartiAsamaStokDetay.MakinaSarjID = :MakinaSarjID
    into :GirisTarihi, :CikisTarihi, :AraVermeSuresi, :SarjKilo, :SarjMetre;

  Sure = CikisTarihi - GirisTarihi - AraVermeSuresi;

  if (MiktarSureyiEtkilermi = 1) then
  begin
    if (SarjKilo > 0) then Sure = Sure / SarjKilo;
    else Sure = 0;
  end

  SureDakika = Sure * 1440;

  if (SureDakika > 5 and SureDakika < 1200) then
  if (MakinaSureId is null) then
  begin
    MakinaSureId = Gen_Id(MakinaSure_Id_Gen, 1);

    insert into MakinaSure(makinaSureId, MakinaPlanlamaGrupId, AsamaId, RenkDerinligiId, LabRenkGrupId, Migrasyon, IplikCinsiId, KalinlikId, KumasCinsiId, IlaveIslemler, Sure, Adet)
         values(:MakinaSureId, :MakinaPlanlamaGrupId, :AsamaId, :RenkDerinligiId, :LabRenkGrupId, :Migrasyon, :IplikCinsiId, :KalinlikId, :KumasCinsiId, :IlaveIslemler, :Sure, 1);
  end
  else Update MakinaSure Set Sure = Sure + :Sure, Adet = Adet + 1 where MakinaSure.MakinaSureID = :MakinaSureId;
end
