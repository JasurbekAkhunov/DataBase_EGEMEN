CREATE PROCEDURE OMIPLIKTRANSFERFISILISTESI(
    ABASLANGICFISTARIHI TIMESTAMP,
    ABITISFISTARIHI TIMESTAMP,
    ABASLANGICFISYIL INTEGER,
    ABASLANGICFISNO INTEGER,
    ABITISFISYIL INTEGER,
    ABITISFISNO INTEGER)
RETURNS (
    OMIPLIKTRANSFERFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    ORMEMAKINAID VARCHAR(10),
    URETIMPARTINO VARCHAR(30),
    BOYASIPARISIDSTR VARCHAR(30),
    TRANSFERORMEMAKINAID VARCHAR(10),
    TRANSFERURETIMPARTINO VARCHAR(30),
    TRANSFERBOYASIPARISIDSTR VARCHAR(30),
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    LOT VARCHAR(30),
    MELANJKODU VARCHAR(30),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP)
AS
declare variable TransferBoyaSiparisYil integer;
declare variable TransferBoyaSiparisId  integer;
declare variable BoyaSiparisYil integer;
declare variable BoyaSiparisId  integer;
begin
  For Select OMIplikTransferFisi.OMIplikTransferFisiId, OMIplikTransferFisi.FisYil, OMIplikTransferFisi.FisNo, OMIplikTransferFisi.Fistarihi,
             OMIplikTransferFisi.OrmeMakinaId, OP.UretimPartiNo, OPBoyaSiparis.BoyaSiparisYil, OPBoyaSiparis.BoyaSiparisId,
             OMIplikTransferFisi.TransferOrmeMakinaId, TOP.UretimPartiNo, TOPBoyaSiparis.BoyaSiparisYil, TOPBoyaSiparis.BoyaSiparisId,
             Ham.Adi, Renk.Adi, OMIplikTransferFisi.Lot, OMIplikTransferFisi.MelanjKodu,
             Ik.Adi, OMIplikTransferFisi.InsertTarihi from OMIplikTransferFisi
    Left Join OrmePlanlama OP on OP.OrmePlanlamaId = OMIplikTransferFisi.OrmePlanlamaId
    Left Join BoyaSiparis OPBoyaSiparis on OPBoyaSiparis.BoyaSiparisKayitId = OP.BoyaSiparisKayitId
    Left Join OrmePlanlama TOP on TOP.OrmePlanlamaId = OMIplikTransferFisi.TransferOrmePlanlamaId
    Left Join BoyaSiparis TOPBoyaSiparis on TOPBoyaSiparis.BoyaSiparisKayitId = TOP.BoyaSiparisKayitId
    Left Join Kullanici IK on IK.KullaniciId = OMIplikTransferFisi.InsertKullaniciId
    left join Renk on Renk.RenkId = OMIplikTransferFisi.RenkId
    left join Ham on Ham.HamId = OMIplikTransferFisi.HamId
    where OMIplikTransferFisi.FisTarihi >= :ABaslangicFisTarihi and OMIplikTransferFisi.FisTarihi <= :ABitisFistarihi and
         (:ABaslangicFisYil is null or (OMIplikTransferFisi.FisYil > :ABaslangicFisYil or (OMIplikTransferFisi.FisYil = :ABaslangicFisYil and OMIplikTransferFisi.FisNo >= :ABaslangicFisNo))) and
         (:ABitisFisYil is null or (OMIplikTransferFisi.FisYil < :ABitisFisYil or (OMIplikTransferFisi.FisYil = :ABitisFisYil and OMIplikTransferFisi.FisNo <= :ABitisFisNo)))
    into :OMIplikTransferFisiId, :FisYil, :FisNo, :Fistarihi, :OrmeMakinaId, :UretimPartiNo, :BoyaSiparisYil, :BoyaSiparisId,
         :TransferOrmeMakinaId, :TransferUretimPartiNo, :TransferBoyaSiparisYil, :TransferBoyaSiparisId,
         :HamAdi, :RenkAdi, :Lot, :MelanjKodu, :InsertKullaniciAdi, :InsertTarihi do
  begin
    FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';

    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    else
      BoyaSiparisIdStr = '';

    if (TransferBoyaSiparisYil is not null) then
      TransferBoyaSiparisIdStr = '(' || EgemenIntegerToStr(TransferBoyaSiparisYil) || ') ' || EgemenIntegerToStr(TransferBoyaSiparisId) || ' ';
    else
      TransferBoyaSiparisIdStr = '';

    suspend;
  end
end
