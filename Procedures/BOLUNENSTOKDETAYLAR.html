CREATE PROCEDURE BOLUNENSTOKDETAYLAR(
    ASTOKHAREKETDETAYID INTEGER)
RETURNS (
    STOKHAREKETDETAYID INTEGER)
AS
begin
  For
    Select StokHareketDetay.StokHareketDetayid from StokHareketDetay where StokHareketDetay.BolunduguStokHareketDetayId = :AStokHareketDetayID into :StokHareketDetayid
  do
  begin
    suspend;
    for select BolunenStokDetaylar.StokHareketDetayid from BolunenStokDetaylar(:StokHareketDetayid) into  :StokHareketDetayid
    do
    begin
      suspend;
    end
  end
end
