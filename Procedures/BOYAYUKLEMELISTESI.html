CREATE PROCEDURE BOYAYUKLEMELISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AARACID VARCHAR(11),
    ASEVKSEFERNO INTEGER,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICFIRMAGRUP1ID VARCHAR(4),
    ABITISFIRMAGRUP1ID VARCHAR(4),
    ABASLANGICFIRMAGRUP2ID VARCHAR(4),
    ABITISFIRMAGRUP2ID VARCHAR(4),
    ABASLANGICFIRMAGRUP3ID VARCHAR(4),
    ABITISFIRMAGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    AUNITELERARASISEVKDOKULSUN INTEGER,
    AUNITEID INTEGER,
    AHAMTIPIID INTEGER)
RETURNS (
    ORME INTEGER,
    ARACID VARCHAR(11),
    SEVKTARIHI TIMESTAMP,
    SEVKSEFERNO INTEGER,
    SOFORADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ALICIUNITEADI VARCHAR(30),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PAKETLEMESEKLIID INTEGER,
    PAKETLEMESEKLIADI VARCHAR(40),
    HAMADITOPLAMI VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    KAPADEDI INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    AMBALAJLIBRUTKILO NUMERIC(12,3),
    KILO NUMERIC(12,3),
    SEVKKILO NUMERIC(12,3),
    TAMIRKILO NUMERIC(12,3),
    IADEKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    SEVKMETRE NUMERIC(12,3),
    TAMIRMETRE NUMERIC(12,3),
    IADEMETRE NUMERIC(12,3),
    SIPARISRENKDERINLIGIID INTEGER,
    SIPARISRENKDERINLIGIADI VARCHAR(20),
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    BRUTKILO NUMERIC(12,3),
    BRUTSEVKKILO NUMERIC(12,3),
    BRUTTAMIRKILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    BRUTSEVKMETRE NUMERIC(12,3),
    BRUTTAMIRMETRE NUMERIC(12,3),
    YAPILANISLEMLER VARCHAR(1023),
    TAMIRTIPIADI VARCHAR(20),
    TAMIR INTEGER,
    KALITEID VARCHAR(20),
    KALITEADI ADI255)
AS
declare variable HamAdi                varchar(255);
declare variable IslemTipi             integer;
declare variable FirmaGrup1            varchar(4);
declare variable FirmaGrup2            varchar(4);
declare variable FirmaGrup3            varchar(4);
declare variable Okey integer;
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef       integer;
declare variable YLFasonOrmeyeCikislariGoster integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtOrmeSatis         integer;
declare variable uitOrme integer;
declare variable FHareketTipi integer;
begin
  Select YLFasonOrmeyeCikislariGoster from sistem where sistemid = 1 into :YLFasonOrmeyeCikislariGoster;

  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;


  Orme = 0;

  For Select BoyaIrsaliye.AracId, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.SevkSeferNo, Sofor.Adi, BoyaIrsaliye.FirmaId, Firma.Adi,
         BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, Parti.PartiYil, Parti.PartiId, PartiStok.HamAdiToplami, PartiStok.PaketlemeSekliId,
         PaketlemeSekli.Adi, Ham.Adi, Parti.ReceteId, Recete.Adi, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
         BoyaIrsaliyeDetay.KapAdedi, BoyaIrsaliyeDetay.Kilo, BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.Metre, BoyaIrsaliyeDetay.BrutMetre,
         BoyaIrsaliyeDetay.YapilanIslemler, Parti.IslemTipiId, Firma.FirmaGrup1,Firma.FirmaGrup2,Firma.FirmaGrup3,
         BoyaSiparisDetay.MusteriSiparisNo, TamirTipi.Adi, BoyaSiparisDetay.RenkDerinligiId, RenkDerinligi.Adi,
         Parti.IslemGrubuId, IslemGrubu.Adi, BoyaSiparisDetay.KaliteId, Kalite.KaliteAdi from BoyaIrsaliye
  left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
  Left Join Sofor On Sofor.SoforId = BoyaIrsaliye.SoforId
  left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
  left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
  left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
  left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliId = PartiStok.PaketlemeSekliId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  left join Desen on Desen.DesenId = DesenVaryant.DesenId
  left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
  where BoyaIrsaliye.Sevktarihi >= :ABaslangicTarihi and BoyaIrsaliye.Sevktarihi < :ABitisTarihi and
        (BoyaIrsaliye.UniteId = :AUniteId or :AUniteId is null) and
        (BoyaIrsaliye.AracId = :AAracId or :AAracId is null) and
        (BoyaIrsaliye.SevkSeferNo = :ASevkSeferNo or :ASevkSeferNo is null) and
        ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf  and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null) and
        ((:AFirmaAralikli = 1 and (BoyaIrsaliye.FirmaId >= :ABaslangicFirmaId) and (BoyaIrsaliye.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
        ((:AFirmaGrup1Aralikli = 1 and (Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id or :ABaslangicFirmaGrup1Id = '') and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and (Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id or :ABaslangicFirmaGrup2Id = '') and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and (Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id or :ABaslangicFirmaGrup3Id = '') and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = ''))
  order by BoyaIrsaliye.AracId, BoyaIrsaliye.SevkSeferNo, BoyaIrsaliye.Sevktarihi
  into :AracId, :SevkTarihi, :SevkSeferNo, :SoforAdi, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :PartiYil, :PartiId,
       :HamAdiToplami, :PaketlemeSekliId, :PaketlemeSekliAdi, :HamAdi, :ReceteId, :ReceteAdi, :DesenId, :DesenAdi, :VaryantAdi,
       :KapAdedi, :Kilo, :BrutKilo, :Metre, :BrutMetre, :YapilanIslemler, :IslemTipi, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :MusteriSiparisNo, :TamirTipiAdi,
       :SiparisRenkDerinligiId, :SiparisRenkDerinligiAdi, :IslemGrubuId, :IslemGrubuAdi, :KaliteId, :KaliteAdi do
  begin
    Okey = 1;
    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      SevkKilo = 0;
      TamirKilo = 0;
      IadeKilo = 0;
      BrutSevkKilo = 0;
      BrutTamirKilo = 0;

      SevkMetre = 0;
      TamirMetre = 0;
      IadeMetre = 0;
      BrutSevkMetre = 0;
      BrutTamirMetre = 0;

      if (HamAdi is not null) then HamAdiToplami = HamAdi;

      if (PartiYil is not null) then
      begin
        PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') ' || EgemenIntegerToStr(PartiId) || ' ';

        if (IslemTipi = itTamir) then
        begin
          YapilanIslemler = '(Tamir Parti) ' || YapilanIslemler;
          TamirKilo = Kilo;
          BrutTamirKilo = BrutKilo;

          TamirMetre = Metre;
          BrutTamirMetre = BrutMetre;
        end
        else
        begin
          SevkKilo = Kilo;
          BrutSevkKilo = BrutKilo;

          SevkMetre = Metre;
          BrutSevkMetre = BrutMetre;
        end
      end
      else
      begin
        PartiIdStr = Null;
        IadeKilo = Kilo;
        IadeMetre = Metre;
        YapilanIslemler = 'Ham Iade';
      end
      Suspend;
    end
  end

  FirmaId = Null;
  FirmaAdi = Null;
  SevkKilo = Null;
  BrutKilo = Null;
  TamirKilo = Null;
  BrutSevkKilo = Null;
  BrutTamirKilo = Null;
  IadeKilo = Null;
  SevkMetre = Null;
  BrutMetre = Null;
  TamirMetre = Null;
  BrutSevkMetre = Null;
  BrutTamirMetre = Null;
  IadeMetre = Null;

  YapilanIslemler = Null;

  FirmaGrup1 = Null;
  FirmaGrup2 = Null;
  FirmaGrup3 = Null;
  Tamir = Null;

  if (AUnitelerArasiSevkDokulsun = 1) then
  begin
    for Select UniteIrsaliye.AracId, UniteIrsaliye.SevkTarihi, UniteIrsaliye.SevkSeferNo, Sofor.Adi, AU.Adi,
               UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo, Parti.PartiYil, Parti.PartiId, PartiStok.HamAdiToplami,
               Parti.ReceteId, Recete.Adi, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, UniteIrsaliyeDetay.KapAdedi,
               UniteIrsaliyeDetay.AmbalajliBrutKilo, UniteIrsaliyeDetay.Kilo, UniteIrsaliyeDetay.Metre,
               BoyaSiparisDetay.MusteriSiparisNo, TamirTipi.Adi, BoyaSiparisDetay.RenkDerinligiId, RenkDerinligi.Adi,
               Parti.IslemGrubuId, IslemGrubu.Adi, BoyaSiparisDetay.KaliteId, Kalite.KaliteAdi from UniteIrsaliye
      left join Unite AU on AU.UniteId = UniteIrsaliye.AliciUniteId
      Left Join Sofor On Sofor.SoforId = UniteIrsaliye.SoforId
      left join UniteIrsaliyeDetay on UniteIrsaliyeDetay.UniteIrsaliyeId = UniteIrsaliye.UniteIrsaliyeId
      Left Join PartiStok on PartiStok.PartiSinifId = UniteIrsaliyeDetay.PartiSinifId
      Left Join Ham on Ham.HamId = PartiStok.HamId
      left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
      Left Join Parti on Parti.PartiKayitId = UniteIrsaliyeDetay.PartiKayitId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
      left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
      left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
      left join Desen on Desen.DesenId = DesenVaryant.DesenId
      left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
      where UniteIrsaliye.IrsaliyeTarihi >= :ABaslangictarihi and UniteIrsaliye.IrsaliyeTarihi < :ABitisTarihi and
            (UniteIrsaliye.GonderenUniteId = :AUniteId or :AUniteId is null) and
            ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null) and
            (UniteIrsaliye.AracId = :AAracId or :AAracId is null) and
            (UniteIrsaliye.SevkSeferNo = :ASevkSeferNo or :ASevkSeferNo is null) and
            UniteIrsaliyeDetay.UniteIrsaliyeId is not null
      into :AracId, :SevkTarihi, :SevkSeferNo, :SoforAdi, :AliciUniteAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :PartiYil, :PartiId, :HamAdiToplami,
           :ReceteId, :ReceteAdi, :DesenId, :DesenAdi, :VaryantAdi, :KapAdedi, :AmbalajliBrutKilo, :Kilo, :Metre, :MusteriSiparisNo, :TamirTipiAdi,
           :SiparisRenkDerinligiId, :SiparisRenkDerinligiAdi, :IslemGrubuId, :IslemGrubuAdi, :KaliteId, :KaliteAdi do suspend;

    AmbalajliBrutKilo = Null;

    for Select MamulDepoTransferFisi.AracId, MamulDepoTransferFisi.SevkTarihi, MamulDepoTransferFisi.SevkSeferNo, Sofor.Adi, AU.Adi,
               MamulDepoTransferFisi.IrsaliyeSeriNo, MamulDepoTransferFisi.IrsaliyeNo, Parti.PartiYil, Parti.PartiId, PartiStok.HamAdiToplami,
               Parti.FirmaId, Firma.Adi, Parti.ReceteId, Recete.Adi, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
               PartiStokTarti.KapAdedi, PartiStokTarti.Kilo, PartiStokTarti.BrutKilo, PartiStokTarti.Metre, PartiStokTarti.BrutMetre, BoyaSiparisDetay.MusteriSiparisNo, TamirTipi.Adi,
               Firma.FirmaGrup1,Firma.FirmaGrup2,Firma.FirmaGrup3,
               BoyaSiparisDetay.RenkDerinligiId, RenkDerinligi.Adi,
               Parti.IslemGrubuId, IslemGrubu.Adi, BoyaSiparisDetay.KaliteId, Kalite.KaliteAdi from MamulDepoTransferFisi
      left join Depo GD on GD.DepoId = MamulDepoTransferFisi.DepoId
      left join Depo AD on AD.DepoId = MamulDepoTransferFisi.TransferDepoId
      left join Unite AU on AU.UniteId = AD.UniteId
      Left Join Sofor On Sofor.SoforId = MamulDepoTransferFisi.SoforId
      left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = MamulDepoTransferFisi.MamulDepoTransferFisiId
      Left Join PartiStok on PartiStok.PartiSinifId = MamulDepoTransferFisiDetay.PartiSinifId
      Left Join Ham on Ham.HamId = PartiStok.HamId
      left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
      Left Join PartiStokTarti on PartiStokTarti.PartiStokTartiId = MamulDepoTransferFisiDetay.PartiStokTartiId
      Left Join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      Left Join Firma on Firma.FirmaId = Parti.FirmaId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
      left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
      left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
      left join Desen on Desen.DesenId = DesenVaryant.DesenId
      left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
      where MamulDepoTransferFisi.IrsaliyeTarihi >= :ABaslangictarihi and MamulDepoTransferFisi.IrsaliyeTarihi < :ABitisTarihi and MamulDepoTransferFisi.IrsaliyeKaydi = 1 and
            (GD.UniteId = :AUniteId or :AUniteId is null) and
            (AD.UniteId <> :AUniteId or :AUniteId is null) and
            ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstElyaf) and :AHamTipiId = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf  and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null) and
            (MamulDepoTransferFisi.AracId = :AAracId or :AAracId is null) and
            (MamulDepoTransferFisi.SevkSeferNo = :ASevkSeferNo or :ASevkSeferNo is null) and
