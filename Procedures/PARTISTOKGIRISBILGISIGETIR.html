CREATE PROCEDURE PARTISTOKGIRISBILGISIGETIR(
    APARTISINIFID INTEGER)
RETURNS (
    HAMSTOKMASTERIDSTR VARCHAR(50),
    GIRISIRSALIYENO VARCHAR(250),
    GIRISHAREKETTIPI VARCHAR(10),
    GIRISIRSALIYETARIHI VARCHAR(250),
    SONHAMSTOKGIRISIRSTARIH TIMESTAMP,
    GIRISFIRMAID VARCHAR(250),
    GIRISFIRMAADI VARCHAR(250),
    GIRISFIRMAKISAADI VARCHAR(250),
    GIRISORMECIADI VARCHAR(100),
    GIRISORMECIMAKINANO VARCHAR(100),
    GIRISMUSTERIPARTINO VARCHAR(100),
    GIRISLOT VARCHAR(150),
    GIRISDENYE VARCHAR(100),
    GIRISPUS VARCHAR(100),
    GIRISFEINE VARCHAR(100),
    GIRISMETRE DOUBLE PRECISION,
    GIRISKILO DOUBLE PRECISION,
    GIRISMIKTAR DOUBLE PRECISION,
    GIRISKAPADEDI DOUBLE PRECISION,
    GIRISFIYATI DOUBLE PRECISION,
    GIRISORMEIRSALIYENO VARCHAR(250),
    GIRISORMEIRSALIYETARIHI VARCHAR(250),
    GIRISORMEFIRMAID VARCHAR(250),
    GIRISORMEFIRMAADI VARCHAR(250),
    GIRISRENK VARCHAR(255),
    URETIMPARTINO VARCHAR(255),
    ESKIPARTINO VARCHAR(100),
    YIKAMAONCESIEN VARCHAR(150),
    YIKAMAONCESIGRAMAJ VARCHAR(150),
    YIKAMASONRASIEN VARCHAR(150),
    YIKAMASONRASIGRAMAJ VARCHAR(150),
    GIRISMUSTERISIPARISNO VARCHAR(300),
    GIRISBATCHNO VARCHAR(100),
    HAMSTOKIRSALIYEGRUPID VARCHAR(50),
    HAMSTOKIRSALIYEGRUPADI VARCHAR(250),
    GIRISREFERANSSIPARISNO VARCHAR(150),
    GIRISIPLIKUZUNLUGU1 VARCHAR(40),
    GIRISIPLIKUZUNLUGU2 VARCHAR(40),
    GIRISIPLIKUZUNLUGU3 VARCHAR(40),
    EVRAKTIPISTR VARCHAR(10),
    ILKYIKAMAONCESIEN DOUBLE PRECISION,
    ILKYIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    ORMEMAKINAADI ADI40,
    ORMEMAKINAID MAKINAKODUSIZE)
AS
declare variable FIrsaliyeNo IrsaliyeNoSize;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable Str varchar(30);
declare variable FFirmaId   varchar(20);
declare variable FFirmaAdi  varchar(100);
declare variable FFirmaKisaAdi  varchar(10);
declare variable FKilo      Numeric(12,3);
declare variable FMetre     Numeric(12,3);
declare variable FOrmeciAdi   varchar(30);
declare variable FOrmeciMakinaNo varchar(10);
declare variable FLot   varchar(150);
declare variable FDenye varchar(30);
declare variable FBatchNo varchar(30);
declare variable FPus   integer;
declare variable FFeine integer;
declare Variable FMusteriPartiNo varchar(30);
declare variable FHamStokMasterId    integer;
declare variable FGirisRenk varchar(20);
declare Variable FUretimPartiNo varchar(30);
declare variable FEskiPartiYil          integer;
declare variable FEskiPartiNo           integer;
declare variable FYikamaOncesiEn        Double Precision;
declare variable FYikamaOncesiGramaj    Double Precision;
declare variable FYikamaSonrasiEn       Double Precision;
declare variable FYikamaSonrasiGramaj   Double Precision;
declare variable SistemDovizId varchar(5);
declare variable FMusteriSiparisNo varchar(50);
declare variable FHamStokHareketTipiId   varchar(2);
declare variable FHamStokIrsaliyeGrupID  varchar(50);
declare variable FHamStokIrsaliyeGrupAdi varchar(250);
declare variable FReferansSiparisNo varchar(30);

begin
  HamStokMasterIdStr     = '';
  GirisFirmaId = '';
  GirisFirmaAdi = '';
  GirisFirmaKisaAdi = '';
  GirisMetre    = 0;
  GirisKilo     = 0;
  GirisMiktar   = 0;
  GirisKapAdedi = 0;
  GirisFiyati   = 0;

  GirisIrsaliyeNo = '';
  GirisHareketTipi = '';
  GirisIrsaliyeTarihi = '';
  GirisOrmeciAdi      = '';
  GirisOrmeciMakinaNo = '';
  GirisLot            = '';
  GirisBatchNo        = '';
  GirisDenye          = '';
  GirisPus            = '';
  GirisFeine          = '';
  GirisMusteriPartiNo = '';
  UretimPartiNo       = '';
  EskiPartiNo         = '';
  YikamaOncesiEn      = '';
  YikamaOncesiGramaj  = '';
  YikamaSonrasiEn     = '';
  YikamaSonrasiGramaj = '';
  GirisMusteriSiparisNo = '';
  HamStokIrsaliyeGrupID = '';
  HamStokIrsaliyeGrupAdi = '';
  GirisReferansSiparisNo = '';
  SonHamStokGirisIrsTarih = Null;
  IlkYikamaOncesiEn       = null;
  IlkYikamaOncesiGramaj   = null;


  Select SistemDovizId from Sistem where sistemId = 1 into :SistemDovizId;

  For select HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.FirmaId, Firma.Adi, Firma.KisaAdi, HamStokIrsaliye.HamStokHareketTipiId, HamStokIrsaliye.IrsaliyeGrupId, StokHareketGrup.Adi, Ormeci.Adi , Case When HamStokIrsaliye.EvrakTipi=0 Then 'Fiş' Else 'İrsaliye' End from PartiStok
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join RafDetay on RafDetay.RafDetayid = PartiStokDetay.RafDetayId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
    left join StokHAreketGrup on  StokHareketGrup.StokHareketGrupID = HamStokIrsaliye.IrsaliyeGrupId
    left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
    left join Ormeci on Ormeci.OrmeciID = HamStokIrsaliye.OrmeciId
    where Partistok.PartiSinifId = :APartiSinifId
    group by HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.FirmaId, Firma.Adi, Firma.KisaAdi, HamStokIrsaliye.HamStokHareketTipiId, HamStokIrsaliye.IrsaliyeGrupId, StokHareketGrup.Adi, Ormeci.Adi,HamStokIrsaliye.EvrakTipi
    into :FIrsaliyeNo, :FIrsaliyeTarihi, :FFirmaId, :FFirmaAdi, :FFirmaKisaAdi, :FHamStokHareketTipiId, :FHamStokIrsaliyeGrupID, :FHamStokIrsaliyeGrupAdi, :FOrmeciAdi, :EvrakTipiStr do
  begin
    if (FIrsaliyeNo is not null) THen Str = Format_Float('#0', FIrsaliyeNo);
    else Str = '';

    if ((FIrsaliyeTarihi > SonHamStokGirisIrsTarih) or (SonHamStokGirisIrsTarih is Null) ) then SonHamStokGirisIrsTarih = FIrsaliyeTarihi;

    if (Str <> '') then
    if (PosBul(GirisIrsaliyeNo, Str) = 0) then
    if (Char_Length(GirisIrsaliyeTarihi) + Char_Length(TimeStampToDateStr(FIrsaliyeTarihi)) < 200) then
    begin
      if (GirisIrsaliyeNo <> '') then GirisIrsaliyeNo = GirisIrsaliyeNo ||  cast(', ' as varchar(2) character set UTF8);
      if (char_length(GirisIrsaliyeNo) <200) then GirisIrsaliyeNo = GirisIrsaliyeNo || Str;

      Str = TimeStampToDateStr(FIrsaliyeTarihi);

      if (GirisIrsaliyeTarihi <> '') then GirisIrsaliyeTarihi = GirisIrsaliyeTarihi ||  cast(', ' as varchar(2) character set UTF8);
      GirisIrsaliyeTarihi = GirisIrsaliyeTarihi || Str;
    end

    if (PosBul(GirisFirmaAdi, FFirmaAdi) = 0) then
    if (Char_Length(GirisFirmaAdi) + Char_Length(FFirmaAdi) < 240) then
    begin
      if (GirisFirmaAdi <> '') then GirisFirmaAdi = GirisFirmaAdi ||  cast(', ' as varchar(2) character set UTF8);
      GirisFirmaAdi = cast(GirisFirmaAdi as blob character set utf8) || FFirmaAdi;

      if (GirisFirmaId <> '') then GirisFirmaId = GirisFirmaId ||  cast(', ' as varchar(2) character set UTF8);
      GirisFirmaId = GirisFirmaId || FFirmaId;
    end

    if (FFirmaKisaAdi is not null) then
    if (PosBul(GirisFirmaKisaAdi, FFirmaKisaAdi) = 0) then
    if (Char_Length(GirisFirmaKisaAdi) + Char_Length(FFirmaKisaAdi) < 240) then
    begin
      if (GirisFirmaKisaAdi <> '' ) then GirisFirmaKisaAdi = GirisFirmaKisaAdi || cast(', ' as varchar(2) character set UTF8);
      GirisFirmaKisaAdi = cast(GirisFirmaKisaAdi as blob character set utf8) || FFirmaKisaAdi;
    end

    if (PosBul(GirisHareketTipi, FHamStokHareketTipiId) = 0) then
    if (Char_Length(GirisHareketTipi) + Char_Length(FHamStokHareketTipiId) < 10) then
    begin
      if (GirisHareketTipi <> '' ) then GirisHareketTipi = GirisHareketTipi || cast(', ' as varchar(2) character set UTF8);
      GirisHareketTipi = GirisHareketTipi || FHamStokHareketTipiId;
    end

     if (Char_Length(HamStokIrsaliyeGrupID) + Char_Length(FHamStokIrsaliyeGrupId) < 45) then
     begin
       if (HamStokIrsaliyeGrupId <> '') then HamStokIrsaliyeGrupId = HamStokIrsaliyeGrupId || cast(', ' as varchar(2) character set UTF8);
       HamStokIrsaliyeGrupId = HamStokIrsaliyeGrupId || FHamStokIrsaliyeGrupId;
     end

     if (Char_Length(HamStokIrsaliyeGrupAdi) + Char_Length(FHamStokIrsaliyeGrupAdi) < 210) then
     begin
       if (HamStokIrsaliyeGrupAdi <> '') then HamStokIrsaliyeGrupAdi = HamStokIrsaliyeGrupAdi || cast(', ' as varchar(2) character set UTF8);
       HamStokIrsaliyeGrupAdi = HamStokIrsaliyeGrupAdi || FHamStokIrsaliyeGrupAdi;
     end

    if (Char_Length(GirisOrmeciAdi) + Char_Length(FOrmeciAdi) < 98 and FOrmeciAdi is not Null) then
    begin
      if (GirisOrmeciAdi <> '' ) then GirisOrmeciAdi = GirisOrmeciAdi || cast(', ' as varchar(2) character set UTF8);
      GirisOrmeciAdi = GirisOrmeciAdi || FOrmeciAdi;
    end

  end

  Select  Avg(HamStokMaster.NetBirimFiyat * HamStokMaster.Miktar * (select DovizKuru from DovizKuruGetir(HamStokIrsaliye.DovizID, :SistemDovizId, Current_date) ) ),
          Sum(HamStokMaster.Miktar), Sum(HamStokMaster.Kilo), Sum(HamStokMaster.Metre), Sum(HamStokMaster.KapAdedi) from PartiStokDetay
    left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
    left join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
    where PartiStokDetay.PartiSinifId = :APartiSinifId  into :GirisFiyati, GirisMiktar, GirisKilo, GirisMetre, GirisKapAdedi;


  For Select HamStokMaster.OrmeciMakinaNo, HamStokMaster.Lot, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Renk,
    OrmePlanlama.UretimPartiNo, HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo, HamStokMaster.Denye, HamStokMaster.YikamaOncesiEn,
    HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj, HamStokDetay.HamStokMasterID, HamStokMaster.MusteriPartiNo,
    HamStokMaster.MusteriSiparisNo, HamStokMaster.BatchNo, HamStokMaster.ReferansSiparisNo, HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3,
    OrmePlanlama.OrmeMakinaId, OrmeMakina.Adi
    from PartiStokDetay
    left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = HamStokMaster.OrmePlanlamaId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
    where PartiStokDetay.PartiSinifId = :APartiSinifId
    group by HamStokMaster.OrmeciMakinaNo, HamStokMaster.Lot, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Renk,
    OrmePlanlama.UretimPartiNo, HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo, HamStokMaster.Denye,
    HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj,
    HamStokDetay.HamStokMasterID, HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo, HamStokMaster.BatchNo, HamStokMaster.ReferansSiparisNo,
    HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3, OrmePlanlama.OrmeMakinaId, OrmeMakina.Adi
    into :FOrmeciMakinaNo, :FLot, :FPus, :FFeine, :FGirisRenk, :FUretimPartiNo, :FEskiPartiYil, :FEskiPartiNo, FDenye,
    :FYikamaOncesiEn, :FYikamaOncesiGramaj, :FYikamaSonrasiEn, :FYikamaSonrasiGramaj, :FHamStokMasterID, :FMusteriPartiNo, :FMusteriSiparisNo, :FBatchNo, :FReferansSiparisNo,
    :GirisIplikUzunlugu1, :GirisIplikUzunlugu2, :GirisIplikUzunlugu3, :OrmeMakinaId, :OrmeMakinaAdi do
  begin

    Str = EgemenIntegerToStr(FHamStokMasterID);
    if (Str is not null) then
    if (PosBul(HamStokMasterIDStr, Str) = 0) then
    if (Char_Length(HamStokMasterIDStr) + Char_Length(Str) < 47) then
    begin
      if (HamStokMasterIDStr <> '') then HamStokMasterIDStr = HamStokMasterIDStr || cast(', ' as varchar(2) character set UTF8);
      HamStokMasterIDStr = HamStokMasterIDStr || Str;
    end

    if (FMusteriPartiNo is not null) then
    if (PosBul(GirisMusteriPartiNo, FMusteriPartiNo) = 0) then
    if (Char_Length(GirisMusteriPartiNo) < 90) then
    begin
      if (GirisMusteriPartiNo <> '' ) then GirisMusteriPartiNo = GirisMusteriPartiNo || cast(', ' as varchar(2) character set UTF8);
      GirisMusteriPartiNo = GirisMusteriPartiNo || FMusteriPartiNo;
    end

    if (FMusteriSiparisNo is not null) then
    if (PosBul(GirisMusteriSiparisNo, FMusteriSiparisNo) = 0) then
    if (Char_Length(GirisMusteriSiparisNo) < 150) then
    begin
      if (GirisMusteriSiparisNo <> '' ) then GirisMusteriSiparisNo = GirisMusteriSiparisNo || cast(', ' as varchar(2) character set UTF8);
      GirisMusteriSiparisNo = GirisMusteriSiparisNo || FMusteriSiparisNo;
    end

    if (FReferansSiparisNo is not null) then
    if (PosBul(GirisReferansSiparisNo, FReferansSiparisNo) = 0) then
    if (Char_Length(GirisReferansSiparisNo) < 120) then
    begin
      if (GirisReferansSiparisNo <> '' ) then GirisReferansSiparisNo = GirisReferansSiparisNo || cast(', ' as varchar(2) character set UTF8);
      GirisReferansSiparisNo = GirisReferansSiparisNo || FReferansSiparisNo;
    end

    if (FYikamaOncesiEn is not null) then
    if (PosBul(YikamaOncesiEn, FYikamaOncesiEn) = 0) then
    if (Char_Length(YikamaOncesiEn) + Char_Length(FYikamaOncesiEn) < 100) then
    begin
      if (YikamaOncesiEn <> '' ) then YikamaOncesiEn = YikamaOncesiEn || cast(', ' as varchar(2) character set UTF8);
      YikamaOncesiEn = YikamaOncesiEn || FYikamaOncesiEn;
    end

    if (FYikamaOncesiEn is not null or FYikamaOncesiGramaj is not null ) then
    if (IlkYikamaOncesiEn is null and IlkYikamaOncesiGramaj is null) then
    begin
      IlkYikamaOncesiEn = FYikamaOncesiEn;
      IlkYikamaOncesiGramaj = FYikamaOncesiGramaj;
    end

    if (FYikamaOncesiGramaj is not null) then
    if (PosBul(YikamaOncesiGramaj, FYikamaOncesiGramaj) = 0) then
    if (Char_Length(YikamaOncesiGramaj) + Char_Length(FYikamaOncesiGramaj) < 100) then
    begin
      if (YikamaOncesiGramaj <> '' ) then YikamaOncesiGramaj = YikamaOncesiGramaj || cast(', ' as varchar(2) character set UTF8);
      YikamaOncesiGramaj = YikamaOncesiGramaj || FYikamaOncesiGramaj;
    end

    if (FYikamaSonrasiEn is not null) then
    if (PosBul(YikamaSonrasiEn, FYikamaSonrasiEn) = 0) then
    if (Char_Length(YikamaSonrasiEn) + Char_Length(FYikamaSonrasiEn) < 100) then
    begin
      if (YikamaSonrasiEn <> '' ) then YikamaSonrasiEn = YikamaSonrasiEn || cast(', ' as varchar(2) character set UTF8);
      YikamaSonrasiEn = YikamaSonrasiEn || FYikamaSonrasiEn;
    end

    if (FYikamaSonrasiGramaj is not null) then
    if (PosBul(YikamaSonrasiGramaj, FYikamaSonrasiGramaj) = 0) then
    if (Char_Length(YikamaSonrasiGramaj) + Char_Length(FYikamaSonrasiGramaj) < 100) then
    begin
      if (YikamaSonrasiGramaj <> '' ) then YikamaSonrasiGramaj = YikamaSonrasiGramaj || cast(', ' as varchar(2) character set UTF8);
      YikamaSonrasiGramaj = YikamaSonrasiGramaj || FYikamaSonrasiGramaj;
    end

    if (FOrmeciMakinaNo is not null) then
    if (PosBul(GirisOrmeciMakinaNo, FOrmeciMakinaNo) = 0) then
    if (Char_Length(GirisOrmeciMakinaNo) + Char_Length(FOrmeciMakinaNo) < 98) then
    begin
      if (GirisOrmeciMakinaNo <> '' ) then GirisOrmeciMakinaNo = GirisOrmeciMakinaNo || cast(', ' as varchar(2) character set UTF8);
      GirisOrmeciMakinaNo = GirisOrmeciMakinaNo || FOrmeciMakinaNo;
    end

    if (FLot is not null) then
    if (PosBul(GirisLot, FLot) = 0) then
    if (Char_Length(GirisLot) + Char_Length(FLot) < 148) then
    begin
      if (GirisLot <> '' ) then GirisLot = GirisLot || cast(', ' as varchar(2) character set UTF8);
      GirisLot = GirisLot || FLot;--cast(GirisLot as blob character set utf8) || FLot;
    end

    if (FBatchNo is not null) then
    if (PosBul(GirisBatchNo, FBatchNo) = 0) then
    if (Char_Length(GirisBatchNo) + Char_Length(FBatchNo) < 80) then
    begin
      if (GirisBatchNo <> '' ) then GirisBatchNo = GirisBatchNo || cast(', ' as varchar(2) character set UTF8);
      GirisBatchNo = GirisBatchNo || FBatchNo;
    end

    if (FDenye is not null) then
    if (PosBul(GirisDenye, FDenye) = 0) then
    if (Char_Length(GirisDenye) + Char_Length(FDenye) < 98) then
    begin
      if (GirisDenye <> '' ) then GirisDenye = GirisDenye || cast(', ' as varchar(2) character set UTF8);
      GirisDenye = GirisDenye || FDenye;
    end

    if (FPus is not null) then
    if (PosBul(GirisPus, FPus) = 0) then
    if (Char_Length(GirisPus) + Char_Length(EgemenIntegerToStr(FPus)) < 98) then
    begin
      if (GirisPus <> '' ) then GirisPus = GirisPus || cast(', ' as varchar(2) character set UTF8);
      GirisPus = GirisPus || EgemenIntegerToStr( FPus );
    end

    if (FFeine is not null) then
    if (PosBul(GirisFeine, FFeine) = 0) then
    if (Char_Length(GirisFeine) + Char_Length(EgemenIntegerToStr(FFeine)) < 98) then
    begin
      if (GirisFeine <> '' ) then GirisFeine = GirisFeine || cast(', ' as varchar(2) character set UTF8);
      GirisFeine = GirisFeine || EgemenIntegerToStr( FFeine );
    end

    if (FGirisRenk is not null) then
    if (PosBul(GirisRenk, FGirisRenk) = 0) then
    if (Char_Length(GirisRenk) + Char_Length(FGirisRenk) < 255) then
    begin
      if (GirisRenk <> '' ) then GirisRenk = GirisRenk || cast(', ' as varchar(2) character set UTF8);
      GirisRenk = cast(GirisRenk as blob character set utf8) || FGirisRenk;
    end

    if (FUretimPartiNo is not null) then
    if (PosBul(UretimPartiNo, FUretimPartiNo) = 0) then
    if (Char_Length(UretimPartiNo) + Char_Length(FUretimPartiNo) < 250) then
    begin
      if (UretimPartiNo <> '' ) then UretimPartiNo = UretimPartiNo || cast(', ' as varchar(2) character set UTF8);
      UretimPartiNo = cast(UretimPartiNo as blob character set utf8)  || FUretimPartiNo;
    end

    if (FEskiPartiNo is not null) then
    if (PosBul(EskiPartiNo, FEskiPartiYil) = 0) then
    if (Char_Length(EskiPartiNo) < 80) then
    begin
      if (EskiPartiNo <> '' ) then EskiPartiNo = EskiPartiNo || cast(', ' as varchar(2) character set UTF8);
      EskiPartiNo = EskiPartiNo || '(' || FEskiPartiYil || ') ' || FEskiPartiNo;
    end

  end

  GirisOrmeIrsaliyeNo         = '';
  GirisOrmeIrsaliyeTarihi     = '';
  GirisOrmeFirmaId            = '';
  GirisOrmeFirmaAdi           = '';

  For select OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.FirmaId, Firma.Adi from Partistok
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join RafDetay on RafDetay.RafDetayid = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join StokHareket CikisStokHareket on CikisStokHareket.StokHareketId = HamStokMaster.StokHareketId
    left join OrmeRaf on OrmeRaf.OrmeRafId = CikisStokHareket.OrmeRafId
    left join OrmeRafDetay on OrmeRafDetay.OrmeRafId = OrmeRaf.OrmeRafId
    left join StokHareket GirisStokHareket on GirisStokHareket.StokHareketId = OrmeRafDetay.StokHareketId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = GirisStokHareket.OrmeIrsaliyeId
    left join Firma on Firma.FirmaId = OrmeIrsaliye.FirmaId
    where Partistok.PartiSinifId = :APartiSinifId and HamStokMaster.StokHareketId is not null
    group by OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.FirmaId, Firma.Adi
    into :FIrsaliyeNo, :FIrsaliyeTarihi, :FFirmaId, :FFirmaAdi do
  begin
    if (FIrsaliyeNo is not null) THen Str = Format_Float('#0', FIrsaliyeNo);
    else Str = '';

    if (Str <> '') then
    if (PosBul(GirisOrmeIrsaliyeTarihi, TimeStampToDateStr(FIrsaliyeTarihi)) = 0) then
    if (Char_Length(GirisOrmeIrsaliyeTarihi) + Char_Length(TimeStampToDateStr(FIrsaliyeTarihi)) < 200) then
    begin
      if (GirisOrmeIrsaliyeNo <> '') then GirisOrmeIrsaliyeNo = GirisOrmeIrsaliyeNo ||  cast(', ' as varchar(2) character set UTF8);
      if (Char_Length(GirisOrmeIrsaliyeNo) < 230) then GirisOrmeIrsaliyeNo = GirisOrmeIrsaliyeNo || Str;

      Str = TimeStampToDateStr(FIrsaliyeTarihi);

      if (GirisOrmeIrsaliyeTarihi <> '') then GirisOrmeIrsaliyeTarihi = GirisOrmeIrsaliyeTarihi ||  cast(', ' as varchar(2) character set UTF8);
      GirisOrmeIrsaliyeTarihi = GirisOrmeIrsaliyeTarihi || Str;
    end

    if (PosBul(GirisOrmeFirmaAdi, FFirmaAdi) = 0) then
    if (Char_Length(GirisOrmeFirmaAdi) + Char_Length(FFirmaAdi) < 240) then
    begin
      if (GirisOrmeFirmaAdi <> '') then GirisOrmeFirmaAdi = GirisOrmeFirmaAdi ||  cast(', ' as varchar(2) character set UTF8);
      GirisOrmeFirmaAdi = cast(GirisOrmeFirmaAdi as blob character set utf8) || FFirmaAdi;

      if (GirisOrmeFirmaId <> '') then GirisOrmeFirmaId = GirisOrmeFirmaId ||  cast(', ' as varchar(2) character set UTF8);
      GirisOrmeFirmaId = cast(GirisOrmeFirmaId as blob character set utf8) || FFirmaId;
    end
  end
  suspend;
end
