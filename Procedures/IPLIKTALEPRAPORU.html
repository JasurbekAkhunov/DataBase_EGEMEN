AS

begin
  if (ARaporTipi = 0) then

               IplikTalepDetay.Kapali, IplikTalep.OnayDurumu,

               BoyaSiparisDetay.IslemTipiID, OrmeSiparisDetay.SatisFiyati, OrmeSiparisDetay.VadeGun, OrmeSiparisDetay.DovizId,

      LEFT JOIN Ham ON Ham.HamId = IplikTalepDetay.HamId
      LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
      left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = IplikTalepDetay.AmbalajSekliId

      LEFT JOIN Firma ON Firma.FirmaId = BoyaSiparis.FirmaId
      LEFT JOIN OrmeSiparisDetay ON OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
      left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
      left join TeslimSekli on TeslimSekli.TeslimSekliId = BoyaSiparis.TeslimSekliId
      left join OdemeSekli on OdemeSekli.OdemeSekliId = BoyaSiparis.OdemeSekliId
      LEFT JOIN BoyaSiparisDetay ON BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
      left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
      left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId

            (IplikTalepDetay.Kapali = :AKapali or :AKapali is null) and

              (:ATeslimDurumu = 1  and IplikTalepDetay.Miktar <= IplikTalepDetay.TeslimEdilenNetMiktar) or
              (:ATeslimDurumu is null) ) and
             IplikTalepDetay.IplikTalepDetayId is not null and
            (IplikTalepDetay.HamId = :AHamId or :AHamId is null)
    order by IplikTalepDetay.IplikTalepKayitId, IplikTalepDetay.IplikTalepDetayId
    into :IplikTalepYil, :IplikTalepId, :TalepTarihi, :TalepYeriId, :IplikTalepKayitId, :IplikTalepDetayId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisTarihi,
         :HamId, :HamAdi, :Miktar, :TeslimEdilenMiktar, :TeslimEdilenNetMiktar, :TeslimEdilenBobinSayisi, :IadeGelenMiktar, :TamirTeslimEdilenMiktar, :TamirTeslimEdilenNetMiktar, :MelanjKodu, :FlamKodu,  :Lot,

         :OrmeSiparisTipiId, :OrmeSiparisTipiAdi, :TeslimTarihi, :TeslimSekliId, :TeslimSekliAdi, :OdemeSekliId, :OdemeSekliAdi, :SipRenkAdi do
    begin
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';

      BoyaSiparisIdStr = '';

      Select First 1 IplikLot.BirimMaliyet, IplikLot.IscilikMaliyeti, IplikLot.DigerMaliyet, IplikLot.HarmanKayitId, Harman.ElyafBirimMaliyet, Harman.iscilikbirimmaliyeti,
             Harman.DigerBirimMaliyet  from IplikLot
        left join Harman on Harman.HarmanKayitId = IplikLot.HarmanKayitId
      where IplikLot.IplikTalepDetayID = :IplikTalepDetayID

      KapaliStr = '';
      if (Kapali = 1) then KapaliStr = 'Evet';

      SevkEdilenToplamMiktar = TeslimEdilenMiktar + TamirTeslimEdilenMiktar;

      TamirStr = '';

      if (IslemTipiId is null or IslemTipiID <> 3) then Tamir = 0;
      else
      begin
        Tamir = 1;
        TamirStr = 'Evet';
      end

      OnayDurumuStr = '';
      if (OnayDurumu = 1) then OnayDurumuStr = 'OnaylÄ±';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';

      Select TalepYeriStr from IplikTalepTalepYeriListesi where TalepYeriID = :TalepYeriID into :TalepYeriAdi;

      UretimMiktari = Null;
      TamirUretimMiktari = Null;
      SonUretimTarihi = Null;
      SonUretimGecenSure = Null;

      Select Sum(iif(IplikLot.Tamir = 0, IplikUretim.Miktar, 0)), Sum(iif(IplikLot.Tamir = 1, IplikUretim.Miktar, 0)), Max(iif(IplikLot.Tamir = 0, IplikUretim.UretimTarihi, Null)) from IplikUretim
        left Join IplikLot on IplikLot.IplikLotID = IplikUretim.IplikLotID
        where IplikUretim.IplikTalepDetayID = :IplikTalepDetayID
        into :UretimMiktari, :TamirUretimMiktari, SonUretimTarihi;

      if (SonUretimTarihi is not null) then SonUretimGecenSure = Current_date + Current_Time - SonUretimTarihi;

      if (ATamir is null or ATamir = Tamir) then suspend;
    end
  end
  else
  begin
    For Select IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, IplikTalep.TalepTarihi, IplikTalep.TalepYeriId, IplikTalepDetay.IplikTalepKayitId, IplikTalepDetay.IplikTalepDetayId,

               IplikTalepDetay.Miktar, IplikTalepDetay.TeslimEdilenMiktar, IplikTalepDetay.TeslimEdilenNetMiktar, IplikTalepDetay.TeslimEdilenBobinSayisi,
               IplikTalepDetay.IadeGelenMiktar, IplikTalepDetay.TamirTeslimEdilenMiktar, IplikTalepDetay.TamirTeslimEdilenNetMiktar,
               IplikTalepDetay.MelanjKodu, IplikTalepDetay.FlamKodu, IplikTalepDetay.Lot, IplikTalepDetay.AmbalajSekliId, AmbalajSekli.Adi,
               IplikTalepDetay.IplikTalepKullanimYeriId, IplikTalepKullanimYeri.Adi,
               IplikTalepDetay.Miktar - IplikTalepDetay.TeslimEdilenNetMiktar, IplikTalepDetay.Notu, BoyaSiparis.FirmaId, Firma.Adi,
               IplikTalepDetay.Kapali, IplikTalep.OnayDurumu,
               OrmeSiparisDetay.Notu, BoyaSiparisDetay.Notu, LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr,
               BoyaSiparisDetay.IslemTipiID, OrmeSiparisDetay.SatisFiyati, OrmeSiparisDetay.VadeGun, OrmeSiparisDetay.DovizId,
               BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, IplikTalepDetay.ReferansUretimNo,
               IplikTalepDetay.BirimMaliyet, BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.OzelTestler, BoyaSiparis.OrmeSiparisSekliId, OrmeSiparisSekli.Adi, BoyaSiparis.OrmeSiparisTipiId, OrmeSiparisTipi.Adi,
               OrmeSiparisDetay.TeslimTarihi, TeslimSekli.TeslimSekliId, TeslimSekli.Adi, OdemeSekli.OdemeSekliId, OdemeSekli.Adi, Renk.Adi FROM IplikTalep
      LEFT JOIN IplikTalepDetay ON IplikTalepDetay.IplikTalepKayitId = IplikTalep.IplikTalepKayitId
      LEFT JOIN Ham ON Ham.HamId = IplikTalepDetay.HamId
      LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
      left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = IplikTalepDetay.AmbalajSekliId
      left join IplikTalepKullanimYeri on IplikTalepKullanimYeri.IplikTalepKullanimYeriId = IplikTalepDetay.IplikTalepKullanimYeriId
      LEFT JOIN Firma ON Firma.FirmaId = BoyaSiparis.FirmaId
      LEFT JOIN OrmeSiparisDetay ON OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
      left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
      left join TeslimSekli on TeslimSekli.TeslimSekliId = BoyaSiparis.TeslimSekliId
      left join OdemeSekli on OdemeSekli.OdemeSekliId = BoyaSiparis.OdemeSekliId
      LEFT JOIN BoyaSiparisDetay ON BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
      left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
      left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
      left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join OrmeSiparisSekli On OrmeSiparisSekli.OrmeSiparisSekliId = BoyaSiparis.OrmeSiparisSekliId
      left join OrmeSiparisTipi On OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
      where IplikTalep.TalepTarihi >= :ABaslangicTalepTarihi and IplikTalep.TalepTarihi <= :ABitisTalepTarihi and
            (IplikTalep.TalepYeriId = :ATalepYeriId or :ATalepYeriId is null) and
            (IplikTalep.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
            (IplikTalepDetay.Kapali = :AKapali or :AKapali is null) and
            (IplikTalep.OnayDurumu = :AOnayDurumu or :AOnayDurumu is null) and
            (BoyaSiparis.FirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
            ( (:ATeslimDurumu = 0  and IplikTalepDetay.Miktar > IplikTalepDetay.TeslimEdilenMiktar) or
              (:ATeslimDurumu = 1  and IplikTalepDetay.Miktar <= IplikTalepDetay.TeslimEdilenMiktar) or
              (:ATeslimDurumu is null) ) and
             IplikTalepDetay.IplikTalepDetayId is not null and
            (IplikTalepDetay.HamId = :AHamId or :AHamId is null)
    order by IplikTalepDetay.IplikTalepKayitId, IplikTalepDetay.IplikTalepDetayId
    into :IplikTalepYil, :IplikTalepId, :TalepTarihi, :TalepYeriId, :IplikTalepKayitId, :IplikTalepDetayId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisTarihi,
         :HamId, :HamAdi, :Miktar, :TeslimEdilenMiktar, :TeslimEdilenNetMiktar, :TeslimEdilenBobinSayisi, :IadeGelenMiktar, :TamirTeslimEdilenMiktar, :TamirTeslimEdilenNetMiktar, :MelanjKodu, :FlamKodu, :Lot,
         :AmbalajSekliId, :AmbalajSekliAdi, :IplikTalepKullanimYeriId, :IplikTalepKullanimYeriAdi,
         :KalanMiktar, :Notu, :SiparisFirmaId, :SiparisFirmaAdi, :Kapali, :OnayDurumu,
         :OrmeSiparisDetayNotu, :BoyaSiparisDetayNotu, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisNo, :IslemTipiID, :BirimFiyat, :VadeGun, :DovizId,
         :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :ReferansUretimNo, :BirimMaliyet, :IlaveIslemler, :OzelTestler, :OrmeSiparisSekliId, :OrmeSiparisSekliAdi, :OrmeSiparisTipiId, :OrmeSiparisTipiAdi,
         :TeslimTarihi, :TeslimSekliId, :TeslimSekliAdi, :OdemeSekliId, :OdemeSekliAdi, :SipRenkAdi  do
    begin
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisYil is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';


      Select First 1 IplikLot.BirimMaliyet, IplikLot.IscilikMaliyeti, IplikLot.DigerMaliyet, IplikLot.HarmanKayitId, Harman.ElyafBirimMaliyet, Harman.iscilikbirimmaliyeti,
             Harman.DigerBirimMaliyet  from IplikLot
        left join Harman on Harman.HarmanKayitId = IplikLot.HarmanKayitId
      where IplikLot.IplikTalepDetayID = :IplikTalepDetayID
      into :LotBirimMaliyet, :LotIscilikMaliyeti, :LotDigerMaliyet, :HarmanKayitId, :HarmanElyafBirimMaliyet, :HarmanIscilikBirimMaliyeti, :HarmanDigerBirimMaliyet;

      KapaliStr = '';
      if (Kapali = 1) then KapaliStr = 'Evet';

      SevkEdilenToplamMiktar = TeslimEdilenMiktar + TamirTeslimEdilenMiktar;

      TamirStr = '';

      if (IslemTipiId is null or IslemTipiID <> 3) then Tamir = 0;
      else
      begin
        Tamir = 1;
        TamirStr = 'Evet';
      end


      OnayDurumuStr = '';
      if (OnayDurumu = 1) then OnayDurumuStr = 'OnaylÄ±';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';

      Select TalepYeriStr from IplikTalepTalepYeriListesi where TalepYeriID = :TalepYeriID into :TalepYeriAdi;

      UretimMiktari = Null;
      TamirUretimMiktari = Null;
      SonUretimTarihi = Null;
      SonUretimGecenSure = Null;

      Select Sum(iif(IplikLot.Tamir = 0, IplikUretim.Miktar, 0)), Sum(iif(IplikLot.Tamir = 1, IplikUretim.Miktar, 0)), Max(iif(IplikLot.Tamir = 0, IplikUretim.UretimTarihi, Null)) from IplikUretim
        left Join IplikLot on IplikLot.IplikLotID = IplikUretim.IplikLotID
        where IplikUretim.IplikTalepDetayID = :IplikTalepDetayID
        into :UretimMiktari, :TamirUretimMiktari, SonUretimTarihi;

      if (SonUretimTarihi is not null) then SonUretimGecenSure = Current_date + Current_Time - SonUretimTarihi;
      if (ATamir is null or ATamir = Tamir) then suspend;
    end
  end
