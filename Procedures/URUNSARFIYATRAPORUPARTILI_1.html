    ARAPORTIPI INTEGER,
    AFIRMAID VARCHAR(20),
    ABASLANGICACILISTARIHI TIMESTAMP,
    ABITISACILISTARIHI TIMESTAMP,
    ABASLANGICRECETEYAZILISTARIHI TIMESTAMP,

    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,

    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,

    ABITISURUNID VARCHAR(20),

    ABITISURUNGRUP1ID VARCHAR(4),
    ABASLANGICURUNGRUP2ID VARCHAR(4),

    ABASLANGICURUNGRUP3ID VARCHAR(4),
    ABITISURUNGRUP3ID VARCHAR(4),
    ABASLANGICURUNGRUP4ID VARCHAR(4),
    ABITISURUNGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,

RETURNS (
    PARTIKAYITID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTITIPIID INTEGER,
    PARTITIPIIDSTR VARCHAR(10),
    ACILISTARIHI TIMESTAMP,
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    MODELNO VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    RECETEYAZILDI INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION,
    NETMIKTAR DOUBLE PRECISION,
    FASONSTR VARCHAR(4),
    TAMIR VARCHAR(4),
    FATURAEDILECEK VARCHAR(4),
    TAMIRTIPIADI VARCHAR(20),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    LOT VARCHAR(150),
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    ORTALAMABIRIMFIYAT DOUBLE PRECISION,
    TEORIKBOYAKIMYASALMALIYETI DOUBLE PRECISION,
    BOYAKIMYASALMALIYETI DOUBLE PRECISION,
    BOYAMALIYETI DOUBLE PRECISION,
    KIMYASALMALIYETI DOUBLE PRECISION,
    HAMSTOKMALIYETI DOUBLE PRECISION,
    ASAMAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    MAKINAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    ASAMAUMUZERINDENPARTIMALIYETI DOUBLE PRECISION,
    MAKINAUMUZERINDENPARTIMALIYETI DOUBLE PRECISION,
    LABRECETEATISBOYAMALIYETI DOUBLE PRECISION,
    LABRECETEATISKIMYASALMALIYETI DOUBLE PRECISION,
    DOGALGAZSARFIYATI DOUBLE PRECISION,
    ELEKTIRIKSARFIYATI DOUBLE PRECISION,
    OKUNANSUSARFIYATI DOUBLE PRECISION,
    ELEKTRIKBIRIMFIYATI DOUBLE PRECISION,
    DOGALGAZBIRIMFIYATI DOUBLE PRECISION,
    BUHARBIRIMFIYATI DOUBLE PRECISION,
    SUBIRIMFIYATI DOUBLE PRECISION,
    UNITELERASEVKMALIYETI DOUBLE PRECISION,
    IMALATTIPIID INTEGER,
    RECETEYAZILISTARIHI TIMESTAMP,
    BELGEHAMADI VARCHAR(180),
    BASKIBOYAKIMYASALMALIYETI DOUBLE PRECISION,
    BOYASIPARISKAYITID INTEGER)
AS
declare variable FDefaultBirimCarpan    double precision;
declare variable Fason                  integer;
declare variable KayitSayisi            integer;
declare variable okey                   integer;
declare variable Miktar                 Double Precision;
declare variable UretimeCikisFiyati     Double Precision;
declare variable StokTabanDovizId       DovizSize;
declare variable SistemDovizId          DovizSize;
declare variable UrunId                    varchar(20);
declare variable UrunGrup1                  varchar(4);
declare variable UrunGrup2                  varchar(4);
declare variable UrunGrup3                  varchar(4);
declare variable UrunGrup4                  varchar(4);
declare variable DovizKuru                  Double Precision;
declare variable DovizID                    varchar(5);
declare variable ARaporTarihi                Date;
declare variable FFaturaKesilecek           integer;
declare variable FIslemTipi                 integer;
declare variable FIrsaliyeDovizId           varchar(5);
declare variable FFaturaDovizId             varchar(5);
declare variable FMaliyet                   Double Precision;
declare variable FAsamaUretimMaliyeti       Double Precision;
declare variable FMakinaUretimMaliyeti       Double Precision;
declare variable FAsamaUretimMaliyetiDovizId     varchar(5);
declare variable FMakinaUretimMaliyetiDovizId     varchar(5);
declare variable FGirisTarihi            TimeStamp;
declare variable FCikisTarihi            TimeStamp;
declare variable FKilo                   Numeric(12,3);
declare variable FAraVermeSuresi         Double Precision;
declare variable PartiAsamaMaliyetHesapTipi integer;
declare variable UrunTipiId integer;
declare variable FHamAdiToplami varchar(255);
declare variable LabReceteDovizId varchar(5);
declare variable Yil integer;
declare variable Ay integer;
declare variable shtUretimeCikis integer;
declare variable shtUretim integer;
declare variable FBeleHamAdi       varchar(60);
declare variable FBaskiAsamasimi integer;

begin
  ARaporTarihi = Current_TimeStamp(0);

  Select StokTabanDovizId, PartiAsamaMaliyetHesapTipi, SistemDovizId from sistem where sistemid = 1 into :StokTabanDovizId, :PartiAsamaMaliyetHesapTipi, :SistemDovizId;
  Select shtUretimeCikis, shtUretim from StokHareketTipiGetir into :shtUretimeCikis, :shtUretim;

  if (ARaporTipi = 0) then
  begin
    for select Parti.PartiKayitId , Parti.PartiYil, Parti.PartiId, Parti.IslemTipiId, Parti.FaturaKesilecek, Parti.AcilisTarihi,
               Parti.PartiGrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi, Parti.ModelNo, Parti.ReceteId,Recete.Adi, Parti.ReceteYazildi,
               Parti.Fason, Parti.PartiTipiId, Parti.IslemGrubuId, IslemGrubu.Adi,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, LabReceteAtis.BoyaMaliyeti, LabReceteAtis.KimyasalMaliyeti, LabRecete.DovizId, Parti.ImalatTipiId, Parti.ReceteYazilisTarihi,
               BoyaSiparis.BoyaSiparisKayitID from Parti
      left join Firma on Firma.FirmaId = Parti.FirmaId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
      left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = Recete.LabReceteAtisId
      left join LabRecete on LabRecete.LabReceteId = LabReceteAtis.LabReceteId
      Where (Parti.PartiYil > :ABaslangicPartiYil or (Parti.PartiYil = :ABaslangicPartiYil and Parti.PartiId >= :ABaslangicPartiId)) and
            (Parti.PartiYil < :ABitisPartiYil or (Parti.PartiYil = :ABitisPartiYil  and Parti.PartiId <= :ABitisPartiId)) and
            (Parti.FirmaId = :AFirmaID or :AFirmaId is null) and
            (Parti.PartiTipiId = :APartiTipiId or :APartiTipiId is null) and
            (Parti.Fason = :AFason or :AFason is null) and
            (:ABaslangicBoyaSiparisYil = 0 or
             ((BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
             (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId))))
    into :PartiKayitId , :PartiYil, :PartiId,  :FIslemTipi, :FFaturaKesilecek, :AcilisTarihi, :PartiGrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi,
         :ModelNo, :ReceteId, :ReceteAdi, :ReceteYazildi, :Fason, :PartiTipiId, :IslemGrubuId, :IslemGrubuAdi,
         :BoyaSiparisYil, :BoyaSiparisId, :LabReceteAtisBoyaMaliyeti, :LabReceteAtisKimyasalMaliyeti, :LabReceteDovizId, :ImalatTipiId, :ReceteYazilisTarihi,
         :BoyaSiparisKayitID do
    begin
      TamirTipiAdi = Null;

      Yil = f_Year(AcilisTarihi);
      Ay  = f_Month(AcilisTarihi);

      Select ElektrikBirimFiyati, DogalGazBirimFiyati, BuharBirimMaliyet, SuBirimFiyati from AylikMasraf
        where AylikMasraf.Yil = :Yil and AylikMasraf.Ay = :Ay
        into :ElektrikBirimFiyati, :DogalGazBirimFiyati, :BuharBirimFiyati, :SuBirimFiyati;

      if (LabReceteDovizId is not null) then
      begin
        if (LabReceteDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:LabReceteDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
        else DovizKuru = 1;

        LabReceteAtisBoyaMaliyeti = LabReceteAtisBoyaMaliyeti * DovizKuru;
        LabReceteAtisKimyasalMaliyeti = LabReceteAtisKimyasalMaliyeti * DovizKuru;

        LabReceteAtisBoyaMaliyeti = Yuvarla(LabReceteAtisBoyaMaliyeti, 10000);
        LabReceteAtisKimyasalMaliyeti = Yuvarla(LabReceteAtisKimyasalMaliyeti, 10000);
      end

      if (FIslemTipi = 3) then
      begin
        Select Max(TamirTipi.Adi) from PartiStok
          left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
          where PartiStok.PartiKayitID = :partiKayitID
          into :TamirTipiAdi;

        Tamir = 'Evet';
      end
      else Tamir = null;

      if (FFaturaKesilecek = 1) then FaturaEdilecek = 'Evet';
      else FaturaEdilecek = null;

      if (PartiTipiId = 0) then PartiTipiIdStr = 'İplik';
      else PartiTipiIdStr = 'Kumaş';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      BoyaKimyasalMaliyeti = 0;
    BaskiBoyaKimyasalMaliyeti = 0;
      TeorikBoyaKimyasalMaliyeti = 0;
      BoyaMaliyeti = 0;
      KimyasalMaliyeti = 0;

      if (AImalatiBitmisOlanlar = 1) then
      begin
        Select count(*) from PartiStok where PartiStok.PartiKayitId = :PartiKayitId and Durumu < 7 into :KayitSayisi;
        if (KayitSayisi = 0) then
        begin
          Select count(*) from PartiStok where PartiStok.PartiKayitId = :PartiKayitId into :KayitSayisi;
          if (KayitSayisi > 0) then Okey = 1;
          else Okey = 0;
        end
        else Okey = 0;
      end
      else Okey = 1;

      if (okey = 1) then
      begin
        Select Sum(PartiStok.Kilo), Sum(PartiStok.Metre), Sum(PartiStok.NetMiktar) from PartiStok
         where PartiStok.PartiKayitId = :PartiKayitId
         into :Kilo, :Metre, :NetMiktar;

        HamAdiToplami = '';

        For Select PartiStok.HamAdiToplami from PartiStok
          where PartiStok.PartiKayitId = :PartiKayitId and PartiStok.PartiStokIptalKayitID is null
          group by PartiStok.HamAdiToplami
          into :FHamAdiToplami do
        begin
          if (Char_Length(HamAdiToplami) + Char_Length(FHamAdiToplami) < 252) then
          begin
            if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set UTF8);
            HamAdiToplami = HamAdiToplami || FHamAdiToplami;
          end
        end

        BelgeHamAdi = '';
        For Select BoyaSiparisDetay.BelgeHamAdi from PartiStok
              left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
                where PartiStok.PartiKayitId = :PArtiKayitID
          into :FBeleHamAdi do
        begin
          if ((Char_Length(BelgeHamAdi) + Char_Length(FBeleHamAdi) < 180) and (Char_Length(FBeleHamAdi) > 1)) then
            if (BelgeHamAdi = '') then
              BelgeHamAdi = FBeleHamAdi;
            else
              BelgeHamAdi = BelgeHamAdi || ', ' || FBeleHamAdi;
        end

        if (StokTabanDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:StokTabanDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
        else DovizKuru = 1;

        if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;

        for select StokDetay.UrunId, Urun.UrunTipiId, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, StokDetay.Miktar, StokDetay.BirimFiyat, StokDetay.Lot, Asama.BaskiAsamasimi from stokdetay
          left join Urun on Urun.UrunId = StokDetay.UrunId
      left join PartiAsamalari on PartiAsamalari.PArtiAsamalariId = stokdetay.PartiAsamalariId
          left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
          where StokDetay.PartiKayitId = :PartiKayitId and
                StokDetay.HareketTipi = :ShtUretimeCikis and
                ((:AUrunAralikli = 1 and Urun.UrunId >= :ABaslangicUrunId and (Urun.UrunId <= :ABitisUrunId or :ABitisUrunId = '')) or (:AUrunAralikli = 0))  and
                ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicUrunGrup1Id  and (UrunGrup1 <= :ABitisUrunGrup1Id or :ABitisUrunGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicUrunGrup1Id = '' and :ABitisUrunGrup1Id = '')) and
                ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicUrunGrup2Id and (UrunGrup2 <= :ABitisUrunGrup2Id or :ABitisUrunGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicUrunGrup2Id = '' and :ABitisUrunGrup2Id = '')) and
                ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicUrunGrup3Id and (UrunGrup3 <= :ABitisUrunGrup3Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicUrunGrup3Id = '' and :ABitisUrunGrup3Id = '')) and
                ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicUrunGrup4Id and (UrunGrup4 <= :ABitisUrunGrup4Id or :ABitisUrunGrup4Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicUrunGrup4Id = '' and :ABitisUrunGrup4Id = ''))
        into :UrunId, :UrunTipiId, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :Miktar, :UretimeCikisFiyati, :Lot, :FBaskiAsamasimi do
        begin
          Okey = 1;
          if (AUrunAralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup1Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup1Id, UrunGrup1) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup2Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup2Id, UrunGrup2) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup3Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup3Id, UrunGrup3) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup4Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup4Id, UrunGrup4) = 0) then okey = 0;

          if (okey = 1) then
          begin
            if (UretimeCikisFiyati is null) then UretimeCikisFiyati = 0;

            UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;

            if (UrunTipiId = 0) then BoyaMaliyeti = BoyaMaliyeti + Miktar * UretimeCikisFiyati;
            else KimyasalMaliyeti = KimyasalMaliyeti + Miktar * UretimeCikisFiyati;

            BoyaKimyasalMaliyeti = BoyaKimyasalMaliyeti + Miktar * UretimeCikisFiyati;
      if (FBaskiAsamasimi = 1) then
              BaskiBoyaKimyasalMaliyeti = BaskiBoyaKimyasalMaliyeti + Miktar * UretimeCikisFiyati;
          end
        end

        if (PartiYil is null) then PartiIdStr = '';
        else
        begin
          PartiIdStr = '(';
          PartiIdStr = PartiIdStr || Format_Float('#0', PartiYil);
          PartiIdStr = PartiIdStr || ') ';
          PartiIdStr = PartiIdStr || Format_Float('#0', PartiId);
          PartiIdStr = PartiIdStr || ' ';
        end

        if (Fason = 1) then
        begin
          FasonStr = 'Evet';
          HamStokMaliyeti = 0;
        end
        else
        begin
          FasonStr = '';
          HamStokMaliyeti = 0;
          For Select HamStokIrsaliye.DovizId, Sum(PartiStokDetay.Kilo * HamStokMaster.NetBirimFiyat) from PartiStokDetay
            left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId

                ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicUrunGrup3Id and (UrunGrup3 <= :ABitisUrunGrup3Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicUrunGrup3Id = '' and :ABitisUrunGrup3Id = '')) and
                ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicUrunGrup4Id and (UrunGrup4 <= :ABitisUrunGrup4Id or :ABitisUrunGrup4Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicUrunGrup4Id = '' and :ABitisUrunGrup4Id = ''))
        into :UrunId, :UrunTipiId, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :Miktar, :UretimeCikisFiyati, :Lot, :FBaskiAsamasimi do
        begin
          Okey = 1;
          if (AUrunAralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup1Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup1Id, UrunGrup1) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup2Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup2Id, UrunGrup2) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup3Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup3Id, UrunGrup3) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup4Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup4Id, UrunGrup4) = 0) then okey = 0;

          if (okey = 1) then
          begin
            if (UretimeCikisFiyati is null) then UretimeCikisFiyati = 0;

            UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;

            if (UrunTipiId = 0) then BoyaMaliyeti = BoyaMaliyeti + Miktar * UretimeCikisFiyati;
            else KimyasalMaliyeti = KimyasalMaliyeti + Miktar * UretimeCikisFiyati;

            BoyaKimyasalMaliyeti = BoyaKimyasalMaliyeti + Miktar * UretimeCikisFiyati;

      if (FBaskiAsamasimi = 1) then
              BaskiBoyaKimyasalMaliyeti = BaskiBoyaKimyasalMaliyeti + Miktar * UretimeCikisFiyati;
          end
        end

        if (PartiYil is null) then PartiIdStr = '';
        else
        begin
          PartiIdStr = '(';
          PartiIdStr = PartiIdStr || Format_Float('#0', PartiYil);
          PartiIdStr = PartiIdStr || ') ';
          PartiIdStr = PartiIdStr || Format_Float('#0', PartiId);
          PartiIdStr = PartiIdStr || ' ';
        end

        if (Fason = 1) then
        begin
          FasonStr = 'Evet';
          HamStokMaliyeti = 0;
        end
        else
        begin
          FasonStr = '';
          HamStokMaliyeti = 0;

          For Select HamStokIrsaliye.DovizId, Sum(PartiStokDetay.Kilo * HamStokMaster.NetBirimFiyat) from PartiStokDetay
            left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
            left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
            left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
            left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
            where PartiStokDetay.PartiKayitId = :PartiKayitId
            Group By HamStokIrsaliye.DovizId
            into :FIrsaliyeDovizId, :FMaliyet do
          begin
            if (FIrsaliyeDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FIrsaliyeDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
            else DovizKuru = 1;

            if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;

            HamStokMaliyeti = HamStokMaliyeti + FMaliyet  * DovizKuru;
          end
        end

        AsamaUMUzerindenAsamaMaliyeti  = 0;
        MakinaUMUzerindenAsamaMaliyeti = 0;

        if (PartiAsamaMaliyetHesapTipi = 0) then
        begin
          For Select PartiAsamalari.UretimMaliyeti, PartiAsamalari.UretimMaliyetiDovizId, Max(PartiAsamaStokDetay.UretimMaliyeti), Max(PartiAsamaStokDetay.UretimMaliyetiDovizId),
                     Min(PartiAsamaStokDetay.GirisTarihi), Max(PartiAsamaStokDetay.CikisTarihi), Max(PartiAsamaStokDetay.AraVermeSuresi) from PartiAsamalari
            left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
            left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
            left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
            where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamaStok.AsamayaDahil = 1 and
                  PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is not null
            group By PartiAsamaStokDetay.MakinaSarjId, PartiAsamalari.UretimMaliyeti, PartiAsamalari.UretimMaliyetiDovizId
            into :FAsamaUretimMaliyeti, :FAsamaUretimMaliyetiDovizId, :FMakinaUretimMaliyeti, :FMakinaUretimMaliyetiDovizId, :FGirisTarihi, :FCikisTarihi, :FAraVermeSuresi do
          begin
            if (FAsamaUretimMaliyetiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FAsamaUretimMaliyetiDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
            else DovizKuru = 1;

            if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;
            AsamaUMUzerindenAsamaMaliyeti = AsamaUMUzerindenAsamaMaliyeti + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FAsamaUretimMaliyeti * DovizKuru;

            if (FMakinaUretimMaliyetiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FMakinaUretimMaliyetiDovizId, :ARaporDovizId, :AcilisTarihi) into :DovizKuru;
            else DovizKuru = 1;

            if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;
            MakinaUMUzerindenAsamaMaliyeti = MakinaUMUzerindenAsamaMaliyeti + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FMakinaUretimMaliyeti * DovizKuru;
          end
        end
        else
        begin
          For Select PartiAsamalari.UretimMaliyetiKg, PartiAsamalari.UretimMaliyetiKgDovizId,
                     Sum(PartiAsamaStokDetay.Kilo), Max(PartiAsamaStokDetay.UretimMaliyeti), Max(PartiAsamaStokDetay.UretimMaliyetiDovizId),
                     Min(PartiAsamaStokDetay.GirisTarihi), Max(PartiAsamaStokDetay.CikisTarihi), Max(PartiAsamaStokDetay.AraVermeSuresi) from PartiAsamalari
            left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
            left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
            left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
            where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamaStok.AsamayaDahil = 1 and
                  PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is not null
            group By PartiAsamaStokDetay.MakinaSarjId, PartiAsamalari.UretimMaliyetiKg, PartiAsamalari.UretimMaliyetiKgDovizId
            into :FAsamaUretimMaliyeti, :FAsamaUretimMaliyetiDovizId, :FKilo, :FMakinaUretimMaliyeti, :FMakinaUretimMaliyetiDovizId, :FGirisTarihi, :FCikisTarihi, :FAraVermeSuresi do
          begin
            if (FAsamaUretimMaliyetiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FAsamaUretimMaliyetiDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
            else DovizKuru = 1;

            if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;

            AsamaUMUzerindenAsamaMaliyeti = AsamaUMUzerindenAsamaMaliyeti + FKilo * FAsamaUretimMaliyeti * DovizKuru;

            if (FMakinaUretimMaliyetiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FMakinaUretimMaliyetiDovizId, :ARaporDovizId, :AcilisTarihi) into :DovizKuru;
            else DovizKuru = 1;

            if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;
            MakinaUMUzerindenAsamaMaliyeti = MakinaUMUzerindenAsamaMaliyeti + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FMakinaUretimMaliyeti * DovizKuru;
          end
        end

        AsamaUMUzerindenPartiMaliyeti = AsamaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti;
        MakinaUMUzerindenPartiMaliyeti = MakinaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti;

  /* eger recetesi arsivlenmis ise teorik kilolardan boya kimyasal maliyeti hesaplanacak */
        for select PartiReceteDetay.UrunId, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, PartiReceteDetay.HesaplananMiktar, PartiReceteDetay.BirimFiyat, PartiReceteDetay.DovizId from PartiAsamalari
          left join PartiReceteDetay on PartiReceteDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join Urun on Urun.UrunId = PartiReceteDetay.UrunId
          left join UrunBirim UBSD on UBSD.UrunBirimId = PartiReceteDetay.UrunBirimId
          left join UrunFiyat on UrunFiyat.UrunId = PartiReceteDetay.UrunId
          left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
          where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiReceteDetay.UrunId is not null and
                ((:AUrunAralikli = 1 and Urun.UrunId >= :ABaslangicUrunId and (Urun.UrunId <= :ABitisUrunId or :ABitisUrunId = '')) or (:AUrunAralikli = 0))  and
                ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicUrunGrup1Id  and (UrunGrup1 <= :ABitisUrunGrup1Id or :ABitisUrunGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicUrunGrup1Id = '' and :ABitisUrunGrup1Id = '')) and
                ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicUrunGrup2Id and (UrunGrup2 <= :ABitisUrunGrup2Id or :ABitisUrunGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicUrunGrup2Id = '' and :ABitisUrunGrup2Id = '')) and

          order by UrunFiyat.UretimeCikisFiyatiDovizId /* Aynı doviz cinsleri icin dovizkuru getire gitmesin diye sirali okuyoruz..*/
        into :UrunId, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :Miktar, :UretimeCikisFiyati, :DovizID do
        begin
          Okey = 1;
          if (AUrunAralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup1Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup1Id, UrunGrup1) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup2Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup2Id, UrunGrup2) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup3Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup3Id, UrunGrup3) = 0) then okey = 0;

          if (okey = 1) then
          if (AUrunGrup4Aralikli = 0) then
          if (KutuKarsilastir(ABaslangicUrunGrup4Id, UrunGrup4) = 0) then okey = 0;

          if (okey = 1) then
          begin
            if (DovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, Current_date) into :DovizKuru;
            else DovizKuru = 1;

            if (UretimeCikisFiyati is null) then UretimeCikisFiyati = 0;

            UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;
            TeorikBoyaKimyasalMaliyeti = TeorikBoyaKimyasalMaliyeti + Miktar * UretimeCikisFiyati;
          end
        end

        OrtalamaBirimFiyat = Null;

        Select Min(BoyaFatura.DovizId), Sum(BoyaFaturaDetay.BirimFiyat * BoyaFaturaDetay.Miktar) / Sum(BoyaFaturaDetay.Miktar) from BoyaIrsaliyeDetay
          left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
          left join BoyaFatura on BoyaFatura.BoyaFaturaId = BoyaFaturaDetay.BoyaFaturaId
          where BoyaIrsaliyeDetay.PartiKayitId = :PartiKayitId and BoyaIrsaliyeDetay.BoyaFaturaId is not null and BoyaFaturaDetay.Miktar <> 0
          into :FFaturaDovizId, :OrtalamaBirimFiyat;

        if (FFaturaDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FFaturaDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
        else DovizKuru = 1;

        if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;

        OrtalamaBirimFiyat = OrtalamaBirimFiyat * DovizKuru;
        OrtalamaBirimFiyat = Yuvarla(OrtalamaBirimFiyat, 1000);

        Select DogalgazSarfiyati, ElektirikSarfiyati, OkunanSuSarfiyati from PartiEnerjiSarfiyatiHesapla(:PartiKayitID) into :DogalgazSarfiyati, :ElektirikSarfiyati, :OkunanSuSarfiyati;

          yukaridakileri silmeme sebebim tedbir, maliyetin duzgun calistigi teyid edilince
          yukaridaki hesaplamalar silinebilir*/

        select first 1 HamStokMaliyeti, AsamaUMUzerindenAsamaMaliyeti, MakinaUMUzerindenAsamaMaliyeti, BoyaKimyasalMaliyeti,
             TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti, BaskiBoyaKimyasalMaliyeti from PartiMaliyetKumulatif
             where PartiKayitID = :PartiKayitID
             into :HamStokMaliyeti, :AsamaUMUzerindenAsamaMaliyeti, :MakinaUMUzerindenAsamaMaliyeti, :BoyaKimyasalMaliyeti,
             :TeorikBoyaKimyasalMaliyeti, :UnitelerASevkMaliyeti, :BaskiBoyaKimyasalMaliyeti;

        AsamaUMUzerindenPartiMaliyeti = AsamaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti + UnitelerASevkMaliyeti;
        MakinaUMUzerindenPartiMaliyeti = MakinaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti + UnitelerASevkMaliyeti;

        if (SistemDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:SistemDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
        else DovizKuru = 1;

        AsamaUMUzerindenPartiMaliyeti = AsamaUMUzerindenPartiMaliyeti * DovizKuru;
        AsamaUMUzerindenAsamaMaliyeti = AsamaUMUzerindenAsamaMaliyeti * DovizKuru;
        HamStokMaliyeti = HamStokMaliyeti * DovizKuru;
        BoyaKimyasalMaliyeti = BoyaKimyasalMaliyeti * DovizKuru;
        UnitelerASevkMaliyeti = UnitelerASevkMaliyeti * DovizKuru;
        MakinaUMUzerindenAsamaMaliyeti = MakinaUMUzerindenAsamaMaliyeti * DovizKuru;
        MakinaUMUzerindenPartiMaliyeti = MakinaUMUzerindenPartiMaliyeti * DovizKuru;
        TeorikBoyaKimyasalMaliyeti = TeorikBoyaKimyasalMaliyeti * DovizKuru;
        BaskiBoyaKimyasalMaliyeti = BaskiBoyaKimyasalMaliyeti * DovizKuru;
        /*Maliyet alimi bitti*/

        suspend;
      end
    end
