CREATE PROCEDURE BOYASEVKLERIORMEYEKAYDET(
    ABOYAIRSALIYEID INTEGER,
    AKULLANICIID INTEGER,
    ADOKUMAHAREKETI SMALLINT)
RETURNS (
    ORMEIRSALIYEID INTEGER)
AS
declare variable FirmaKodumuz varchar(20);
declare variable SistemDovizID varchar(5);
declare variable HareketTipi integer;
declare variable UretimIslemTipiId integer;
declare variable PartiIdStr        varchar(30);
declare variable RenkId            integer;
declare variable FHamRenkId            integer;
declare variable PartiYil           integer;
declare variable PartiId            integer;
declare variable PartiSinifId       integer;
declare variable BoyaIrsaliyeDetayId integer;
declare variable BoyaSiparisDetayId integer;
declare variable OrmeDefaultRafId    varchar(10);
declare variable DokumaDefaultRafId    varchar(10);
declare variable OrmeDefaultDepoId   varchar(10);
declare variable HamIade             integer;
declare variable HamIadeRezervId          integer;
declare variable BoyaHamId             integer;
declare variable OrmeHamId            integer;
declare variable KaliteNo             integer;
declare variable FReceteId              varchar(20);
declare variable FLabReceteId           integer;
declare variable FSipLabReceteId        integer;
declare variable FSipRenkId             integer;
declare variable FLabReceteKodu         varchar(20);
declare variable FLabReceteAdi          varchar(40);
declare variable RenkAdi                varchar(61);
declare variable OrmeciMakinaNo           varchar(10);
declare variable TransferYeri        integer;
declare variable OrmeSiparisTipiId      integer;
declare variable KS                     integer;
declare variable FasonHamIade           integer;
declare variable IlkKayit              integer;
declare variable StokHareketId          integer;
declare variable BoyahaneCikisPartiNoTakibiVar integer;
declare variable PartiHamStokMasterIdKarisik   integer;
declare variable CikisPartiNo varchar(30);
declare variable OrmeStokGirisindeOtomatikRafla integer;
declare variable FNetBirimFiyat Double Precision;
declare variable FDovizId       varchar(5);
declare variable DovizKuru Double Precision;
declare variable IptalNotu              Blob;
declare variable BirimMaliyet           Double Precision;
declare variable DepoRafAlimSekli       Integer;
declare variable OrmeBarkodluGirisCikisTakibi integer;
declare variable FEgemenFirmaId integer;
declare variable FPartiSiraNo integer;
declare variable AtisNoStr VarChar(20);

begin
/* DepoRafAlimSekli = 1;  /*Ozak*/
  DepoRafAlimSekli = 0;  /*DiÄŸer*/

  Select ShtFasonUretim from OrmeStokHareketTipiGetir into :HareketTipi;
  Select uitBoya from UretimIslemTipiGetir into :UretimIslemTipiId;

  Select FirmaKodumuz, SistemDovizID, OrmeDefaultRafId, DokumaDefaultRafId, BoyahaneCikisPartiNoTakibiVar, SPartiHamStokMasterIdKarisik,
         OrmeStokGirisindeOtomatikRafla, OrmeBarkodluGirisCikisTakibi, EgemenFirmaId  from Sistem
    where sistemid = 1
    into :FirmaKodumuz, :SistemDovizID, :OrmeDefaultRafId, :DokumaDefaultRafId, :BoyahaneCikisPartiNoTakibiVar, :PartiHamStokMasterIdKarisik,
         :OrmeStokGirisindeOtomatikRafla, :OrmeBarkodluGirisCikisTakibi, :FEgemenFirmaId;

  if (ADokumaHareketi = 1) then
      OrmeDefaultRafId = DokumaDefaultRafId;

  IlkKayit = 0;
  OrmeIrsaliyeID = Null;
  Select BoyaIrsaliye.OrmeIrsaliyeID from BoyaIrsaliye where BoyaIrsaliye.BoyaIrsaliyeID = :ABoyaIrsaliyeID into :OrmeIrsaliyeID;

  if (DepoRafAlimSekli = 0) then
    Select DepoId from DepoRaf where DepoRaf.RafID = :OrmeDefaultRafId into :OrmeDefaultDepoId;
  else
  begin
    Select DefaultOrmeDepoId from Kullanici where Kullanici.KullaniciID = :AKullaniciId into :OrmeDefaultDepoId;
    Select RafId from DepoRaf where DepoRaf.DepoID = :OrmeDefaultDepoId into :OrmeDefaultRafId;
  end

  if (OrmeIrsaliyeID is null) then
  begin
    IlkKayit = 1;
    OrmeIrsaliyeId = gen_Id(OrmeIrsaliye_Id_gen,1);
    insert into OrmeIrsaliye(OrmeIrsaliyeID, HareketTipi, UretimIslemTipiId, FirmaId, SeriNo, IrsaliyeNo, IrsaliyeTarihi, FiiliSevkTarihi,
                             AracId, SoforId, SevkSeferNo, Sinif, IrsaliyeGrupId, DovizId, TransferYeri, InsertKullaniciId, KullaniciId, DokumaHareketi)
         Select :OrmeIrsaliyeID, :HareketTipi, :UretimIslemTipiId, :FirmaKodumuz, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi,
                             BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.AracId, BoyaIrsaliye.SoforId, BoyaIrsaliye.SevkSeferNo,
                             BoyaIrsaliye.Sinif, BoyaIrsaliye.IrsaliyeGrupId, :SistemDovizId, 1, :AKullaniciId, :AKullaniciId, :ADokumaHareketi from BoyaIrsaliye
         where BoyaIrsaliye.BoyaIrsaliyeID = :ABoyaIrsaliyeID;
  end

  For Select BoyaIrsaliyeDetay.HamIadeRezervId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, Renk.RenkId, OrmeSiparisDetay.RenkId, Parti.PartiYil, Parti.PartiId,
             BoyaSiparisDetay.HamId, OrmeSiparisDetay.HamId, BoyaIrsaliyeDetay.KaliteNo, Parti.ReceteId, BoyaSiparisDetay.LabReceteId, BoyaSiparisDetay.RenkId,
             BoyaSiparis.OrmeSiparisTipiId, BoyaIrsaliyeDetay.PartiSinifId, PartiStok.BoyaSiparisDetayId,
             PartiStokIptal.IptalNotu, PartiStok.PartiSiraNo, LabreceteAtis.AtisNoStr from BoyaIrsaliyeDetay
      left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
      left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join Renk on Renk.LabReceteId = Recete.LabReceteId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
      left join RafDetay on RafDetay.RafDetayId = BoyaIrsaliyeDetay.RafDetayID
      left join PartiStokIptal on PartiStokIptal.PartiStokIptalKayitId = RafDetay.PartiStokIptalKayitId
      left join LabreceteAtis on LabReceteAtis.LabreceteAtisId =BoyaSiparisDetay.LabreceteAtisId
      where BoyaIrsaliyeDetay.BoyaIrsaliyeID = :ABoyaIrsaliyeID and BoyaIrsaliyeDetay.OrmeIrsaliyeId is null
      into :HamIadeRezervId, :BoyaIrsaliyeDetayId, :RenkId, :FHamRenkId, :PartiYil, :PartiId, :BoyaHamId, :OrmeHamId, :KaliteNo, :FReceteId, :FSipLabReceteId, :FSipRenkId,
           :OrmeSiparisTipiId, :PartiSinifId, :BoyaSiparisDetayId, :IptalNotu, :FPartiSiraNo, :AtisNoStr do
  begin
    PartiIdStr = '';
    OrmeciMakinaNo = Null;
    FLabreceteId = NUll;
    StokHareketID = Null;

    if (PartiYil is not null) then
    begin
     if (FEgemenFirmaId =123 and  OrmeHamId is not null and BoyaHamID = OrmeHamId and FHamRenkID is not null and OrmeSiparisTipiId = 5) then RenkID = FSipRenkId; /* ipligi boyali */
     else if (OrmeHamId is not null and BoyaHamID = OrmeHamId and FHamRenkID is not null and OrmeSiparisTipiId = 5) then RenkID = FHamRenkId; /* ipligi boyali */
      else
      begin
        KS = 0;
        Select Count(*) from PartiAsamaStok
          left join Asama on Asama.AsamaID = PartiAsamaStok.AsamaId
          where PartiAsamaStok.PartiSinifID = :PartiSinifID and PartiAsamaStok.AsamayaDahil = 1 and Asama.BoyaAsamasimi = 1
          into :KS;

        if (KS > 0) then
        begin
          if (FSipRenkId is not null) then RenkID = FSipRenkId;

          if (RenkID is null) then
          begin
            if (FSipLabReceteId is not null) then
            begin
              FLabreceteId = FSipLabReceteId;

              Select LabRecete.LabreceteKodu, LabRecete.Adi from LabRecete
                where LabRecete.LabReceteID = :FLabReceteID
                into :FLabreceteKodu, :FLabReceteAdi;
            end
            else
            begin
              Select Recete.LabreceteId, LabRecete.LabreceteKodu, LabRecete.Adi from Recete
                left join LabRecete on LabRecete.LabReceteID = Recete.LabReceteId
                where Recete.ReceteID = :FReceteID
                into :FLabreceteId, :FLabreceteKodu, :FLabReceteAdi;
            end

            if (FLabreceteId is not null) then
            begin
              RenkId = Null;
              Select Renk.RenkId from Renk where Renk.LabReceteID = :FLabReceteID into :RenkId;

              if (RenkID is null) then
              begin
                RenkAdi = FLabReceteAdi || '-' || FLabReceteKodu|| '.' || AtisNoStr;
                insert into Renk(LabReceteId, Adi) values(:FLabReceteId, EgemenCopyStr(:RenkAdi, 1, 50));

                Select Renk.RenkId from Renk where Renk.LabReceteID = :FLabReceteID into :RenkId;
              end
            end
          end
        end
        else RenkId = Null;


      end

     if (FEgemenFirmaId <> 56) then --Ns Ã–rme
        PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId);
     else
        PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || '.' || FPartiSiraNo;

      HamIade = 0;

      StokHareketID = Gen_Id(StokHareket_Id_gen, 1);

      if (BoyahaneCikisPartiNoTakibiVar = 1 and PartiHamStokMasterIdKarisik = 0) then
      begin
/* cÄ±kÄ±s parti no takibi var ise ve karÄ±sÄ±k partilenmiyor ise merkezden hangi cikis parti no ile geldigini buluyoruz */
        Select Max(BoyahaneCikisPartiNo.CikisPartiNo) from PartiStok
          left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
          left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
          left join StokHareket on StokHareket.StokHareketId = HamStokMaster.StokHareketId
          left join BoyahaneCikisPartiNo on BoyahaneCikisPartiNo.StokHareketId = StokHareket.StokHareketId
          where PartiStok.PartiSinifId = :PartiSinifId
          into :CikisPartiNo;
      end

      Select BirimMaliyet from SiparisDetayBirimMaliyetGetir(Null, :BoyaSiparisDetayID) into :BirimMaliyet;

      insert into StokHareket(OrmeIrsaliyeId, StokHareketID, HareketTipi, BoyaSiparisKayitId, OrmeSiparisDetayId, BoyaSiparisDetayId,
                  BoyaSiparisBaskiId, HamId, RenkId, DesenId, DesenVaryantId, KapAdedi, BrutMiktar, Miktar, UretimPartiNo, HamIade,
                  MelanjKodu, KaliteNo, KaliteAciklamaTipiId, MamulProsesId, BoyaIrsaliyeDetayId, CikisPartiNo, BirimFiyat, DepoId, InsertKullaniciId,
                  KullaniciId, DokumaHareketi)
           Select :OrmeIrsaliyeId, :StokHareketID, :HareketTipi, PartiStok.BoyaSiparisKayitId, BoyaSiparisDetay.OrmeSiparisDetayId,
                  PartiStok.BoyaSiparisDetayId, PartiStok.BoyaSiparisBaskiId, BoyaIrsaliyeDetay.HamId, :RenkId,
                  DesenVaryant.DesenId, Parti.DesenVaryantId, BoyaIrsaliyeDetay.KapAdedi, BoyaIrsaliyeDetay.BrutMiktar, BoyaIrsaliyeDetay.Miktar, :PartiIdStr, :HamIade, BoyaSiparisDetay.MelanjKodu, :KaliteNo, BoyaIrsaliyeDetay.KaliteAciklamaTipiId, BoyaIrsaliyeDetay.MamulProsesId, :BoyaIrsaliyeDetayId, :CikisPartiNo,
                  :BirimMaliyet, :OrmeDefaultDepoId, :AKullaniciId, :AKullaniciId, :ADokumaHareketi from BoyaIrsaliyeDetay
        left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
        left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
        where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID;
    end
    else
    begin
      if (HamIadeRezervId is not null) then
      begin
        HamIade = 1;
        Select HamStokIrsaliye.TransferYeri, HamStokMaster.OrmeciMakinaNo, HamStokIrsaliye.Fason,
               HamStokMaster.NetBirimFiyat, HamStokIrsaliye.DovizId from BoyaIrsaliyeDetay
          left join RafDetay on RafDetay.RafDetayID = BoyaIrsaliyeDetay.RafDetayID
          left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
          left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = HamStokDetay.HamStokIrsaliyeId
          left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
          left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = HamStokMaster.BoyaSiparisDetayId
          where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID
          into :TransferYeri, :OrmeciMakinaNo, :FasonHamIade, :FNetBirimFiyat, :FDovizId;

          if (TransferYeri <> 1) then OrmeciMakinaNo = Null;

      end
      else
      begin
        HamIade = 0;

        Select HamStokMaster.NetBirimFiyat, HamStokIrsaliye.DovizId from BoyaIrsaliyeDetay
          left join RafDetay on RafDetay.RafDetayID = BoyaIrsaliyeDetay.RafDetayID
          left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
          left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = HamStokDetay.HamStokIrsaliyeId
          left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
          where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID
          into :FNetBirimFiyat, :FDovizId;
      end

      if (FDovizId <> SistemDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :SistemDovizId, Current_Date) into :DovizKuru;
      else DovizKuru = 1;

      if (FNetBirimFiyat is null) then FNetBirimFiyat = 0;
      else FNetBirimFiyat = FNetBirimFiyat * DovizKuru;

      if (HamIade = 1 and FasonHamIade = 1 and TransferYeri <> 1) then
      begin
        StokHareketID = Gen_Id(StokHareket_Id_gen, 1);

        insert into StokHareket(OrmeIrsaliyeId, StokHareketID, HareketTipi, StokFirmaId, FasonStok, BoyahaneFasonIade, HamId, KapAdedi, BrutMiktar, Miktar, UretimPartiNo, HamIade, KaliteNo, OrmeciMakinaNo, BoyaIrsaliyeDetayId, BirimFiyat, DepoId, Notu, InsertKullaniciId, KullaniciId, DokumaHareketi, Adet, OlcuId)
             Select :OrmeIrsaliyeId, :StokHareketID, :HareketTipi, HamStokDetay.StokFirmaId, 1, 1, BoyaIrsaliyeDetay.HamId, BoyaIrsaliyeDetay.KapAdedi, BoyaIrsaliyeDetay.BrutMiktar, BoyaIrsaliyeDetay.Miktar, :PartiIdStr, :HamIade,
                    :KaliteNo, :OrmeciMakinaNo, :BoyaIrsaliyeDetayId, :FNetBirimFiyat, :OrmeDefaultDepoId, :IptalNotu, :AKullaniciId, :AKullaniciId, :ADokumaHareketi, BoyaIrsaliyeDetay.Adet, StokHareket.OlcuId from BoyaIrsaliyeDetay
          left join RafDetay on RafDetay.RafDetayID = BoyaIrsaliyeDetay.RafDetayID
          left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
          left join StokHareket on StokHareket.StokHareketId = HamStokMAster.StokHareketId
          where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID;
      end
      else
      begin
        StokHareketID = Gen_Id(StokHareket_Id_gen, 1);

        insert into StokHareket(OrmeIrsaliyeId, StokHareketID, HareketTipi, BoyaSiparisKayitId, OrmeSiparisDetayId, BoyaSiparisDetayId, HamId, RenkId, KapAdedi, BrutMiktar, Miktar, UretimPartiNo, HamIade, Lot, MelanjKodu, StokGirisNo, KaliteNo, OrmeciMakinaNo, BoyaIrsaliyeDetayId, BirimFiyat, CikisPartiNo, DepoId, Notu, InsertKullaniciId, KullaniciId, DokumaHareketi, Adet, OlcuId)
             Select :OrmeIrsaliyeId, :StokHareketID, :HareketTipi, HamStokMaster.BoyaSiparisKayitId, HamStokMaster.OrmeSiparisDetayId, HamStokMaster.BoyaSiparisDetayId, BoyaIrsaliyeDetay.HamId,
                    HamStokMaster.RenkId, BoyaIrsaliyeDetay.KapAdedi, BoyaIrsaliyeDetay.BrutMiktar, BoyaIrsaliyeDetay.Miktar, :PartiIdStr, :HamIade, HamStokMaster.Lot, HamStokMaster.MelanjKodu, HamStokMaster.OrmeStokGirisNo, :KaliteNo, :OrmeciMakinaNo, :BoyaIrsaliyeDetayId,
                   :FNetBirimFiyat, BoyahaneCikisPartiNo.CikisPartiNo, :OrmeDefaultDepoId, :IptalNotu, :AKullaniciId, :AKullaniciId, :ADokumaHareketi, BoyaIrsaliyeDetay.Adet, StokHareket.OlcuId from BoyaIrsaliyeDetay
          left join RafDetay on RafDetay.RafDetayID = BoyaIrsaliyeDetay.RafDetayID
          left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
          left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = HamStokMaster.BoyaSiparisDetayId
          left join StokHareket on StokHareket.StokHareketId = HamStokMaster.StokHareketId
          left join BoyahaneCikisPartiNo on BoyahaneCikisPartiNo.StokHareketId = StokHareket.StokHareketId
          where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID;
      end
    end

    if (StokHareketID is not null) then
    begin
      if (OrmeStokGirisindeOtomatikRafla = 0) then
      begin
        Insert into OrmeRafDetay(StokHareketId, RafId, KapAdedi, Miktar, BrutMiktar, InsertKullaniciId, KullaniciId, Adet)
                    Select StokHareket.StokHareketId, :OrmeDefaultRafId, StokHareket.KapAdedi, StokHareket.Miktar, StokHareket.BrutMiktar,
                          :AkullaniciId, :AKullaniciId, StokHareket.Adet from StokHareket where StokHareket.StokHareketId = :StokHareketId;
      end

      Update BoyaIrsaliyeDetay Set OrmeIrsaliyeID = :OrmeIrsaliyeID where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayId;

      if (OrmeBarkodluGirisCikisTakibi = 1) then
      begin
        insert into StokHareketDetay(StokHareketID, HareketTipi, PSKaliteKontrolID, IplikPaketlemeID, Kilo, BrutKilo, Metre, InsertKullaniciId, KullaniciId)
              Select :StokHareketID, :HareketTipi, PSKaliteKontrol.PSKaliteKontrolID, IplikPaketleme.IplikPaketlemeID,
              iif(PSKaliteKontrol.PSKaliteKontrolID is not Null, PSKaliteKontrol.Kilo-PSKaliteKontrol.Dara, IplikPaketleme.NetKilo),
              iif(PSKaliteKontrol.PSKaliteKontrolID is not Null, (PSKaliteKontrol.Kilo), IplikPaketleme.BrutKilo),
              iif(PSKaliteKontrol.PSKaliteKontrolID is not Null, (PSKaliteKontrol.Metre), 0), :AKullaniciId, :AKullaniciId from PartiStokTarti
              Left Join PSKaliteKontrol on PSKaliteKontrol.PartiStokTartiID = PartiStokTarti.PartiStokTartiID
              Left Join IplikPaketleme on IplikPaketleme.PartiStokTartiID = PartiStokTarti.PartiStokTartiID
            where PartiStokTarti.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayId and (PSKaliteKontrol.PSKaliteKontrolID is not Null or IplikPaketleme.IplikPaketlemeID is Not Null);
      end

    end
  end

  if (IlkKayit = 1) then
    Update BoyaIrsaliye Set OrmeIrsaliyeID = :OrmeIrsaliyeId where BoyaIrsaliye.BoyaIrsaliyeID = :ABoyaIrsaliyeID;

  suspend;
end
