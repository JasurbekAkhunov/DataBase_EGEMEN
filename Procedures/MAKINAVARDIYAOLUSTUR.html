CREATE PROCEDURE MAKINAVARDIYAOLUSTUR(
    ATARIH TIMESTAMP,
    AREFERANSTARIH TIMESTAMP,
    AKULLANICIID INTEGER)
AS
begin
  Insert into MakinaVardiya(Tarih, MakinaId, VardiyaSayisi, InsertKullaniciId, KullaniciId)
       Select :ATarih, Makina.MakinaId, MakinaVardiya.VardiyaSayisi, :AKullaniciId, :AKullaniciId from Makina
       left join MakinaVardiya on MakinaVardiya.MakinaId = Makina.MakinaId and MakinaVardiya.Tarih = :AReferansTarih
       where Makina.UretimRaporlarinaDahil = 1;
end
