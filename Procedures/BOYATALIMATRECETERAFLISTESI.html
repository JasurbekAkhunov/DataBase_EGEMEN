CREATE PROCEDURE BOYATALIMATRECETERAFLISTESI(
    ABOYASIPARISDETAYID INTEGER,
    ABOYATALIMATKAYITID INTEGER)
RETURNS (
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RAFID VARCHAR(10),
    RAFKALANMIKTAR NUMERIC(12,3),
    RAFKALANKAPADEDI INTEGER,
    KULLANILABILIRMIKTAR NUMERIC(12,3),
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    KALITENO INTEGER,
    ORMEMAKINAID VARCHAR(10),
    URETICIADI VARCHAR(40),
    URETIMPARTINO VARCHAR(30),
    ESKIPARTIYIL INTEGER,
    ESKIPARTINO INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    BOYATALRECETEKALANMIKTAR BOYAMIKTARSIZE)
AS
declare variable SiparisFirmaId       varchar(20);
declare variable OrmeFirmaKodumuz     varchar(20);
declare variable Fason                integer;
declare variable IslemTipiId          integer;
declare variable itTamir              integer;
declare variable KTBoyaSiparisKayitID integer;
declare variable TupAcikEnMayli integer;
declare variable TalimatFirmaID FirmaKoduSize;
declare variable OrmeSiparisDetayId integer;
declare variable AHamId integer;
begin
  Select BoyaSiparis.FirmaId, BoyaSiparis.Fason, OrmeSiparisDetay.KTBoyaSiparisKayitID, BoyaSiparisDetay.OrmeSiparisDetayId, BoyaSiparisDetay.HamId from BoyaSiparisDetay
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
  where BoyaSiparisDetay.BoyaSiparisDetayID = :ABoyaSiparisDetayID
  into :SiparisFirmaId, :Fason, :KTBoyaSiparisKayitID, :OrmeSiparisDetayId, :AHamId;

  Select BoyaSiparisDetay.IslemTipiId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId into :IslemTipiId;
  Select BoyaTalimat.FirmaID from BoyaTalimat where BoyaTalimat.BoyaTalimatKayitID = :ABoyaTalimatKayitID into :TalimatFirmaID;

  Select itTamir from islemTipiGetir into :itTamir;

  if (Fason = 0) then SiparisFirmaId = Null;

  Select OrmeFirmaKodumuz  from sistem where sistemid = 1 into :OrmeFirmaKodumuz;

  For Select OrmeRaf.OrmeRafId, OrmeRaf.HamId, Ham.Adi, OrmeRaf.StokGirisNo, OrmeRaf.RafId, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
             OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi, OrmeRaf.KaliteNo, OrmeRaf.Lot, OrmeRaf.MelanjKodu, Uretici.Adi,
             coalesce(OrmeRaf.OrmeMakinaId, OrmeRaf.OrmeciMakinaNo), OrmeRaf.UretimPartiNo, OrmeRaf.EskiPartiYil, OrmeRaf.EskiPartiNo, OrmeSiparisDetay.TupAcikEnMayli from OrmeRaf
    left join Ham on Ham.HamId = OrmeRaf.HamId
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
    left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = OrmeRaf.OrmeSiparisDetayID
    Left Join Raf on Raf.RafId = OrmeRaf.RafID
    where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and
         (OrmeRaf.FasonStok = :Fason and (OrmeRaf.FasonStok = 0 or OrmeRaf.StokFirmaID = :SiparisFirmaID)) and
         (OrmeRaf.RenkId is null or :IslemTipiId = :itTamir) and
         (OrmeRaf.DesenId is null or :IslemTipiId = :itTamir) and
         ( OrmeRaf.OrmeSiparisDetayId = :OrmeSiparisDetayId or
           OrmeRaf.OrmeSiparisDetayId is null or
          (OrmeRaf.OrmeSiparisDetayId is not null and BoyaSiparis.FirmaID = :OrmeFirmaKodumuz) or
          (OrmeRaf.BoyaSiparisKayitID = :KTBoyaSiparisKayitID) ) and
         (Raf.FirmaID = :TalimatFirmaID or :TalimatFirmaID is Null)
    into :OrmeRafId, :HamId, :HamAdi, :StokGirisNo, :RafId, :RenkId, :RenkAdi, :RafKalanMiktar, :RafKalanKapAdedi, :KaliteNo,  :Lot, :MelanjKodu,  :UreticiAdi, :OrmeMakinaId, :UretimPartiNo,
         :EskiPartiYil, :EskiPartiNo, :TupAcikEnMayli do
  begin
    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    Select Sum(Miktar - KullanilanMiktar) from BoyaTalimatRecete where BoyaTalimatRecete.OrmeRafID = :OrmeRafId into :BoyaTalReceteKalanMiktar;
    if (BoyaTalReceteKalanMiktar is Null) then BoyaTalReceteKalanMiktar = 0;

    KullanilabilirMiktar = RafKalanMiktar - BoyaTalReceteKalanMiktar;

    if (KullanilabilirMiktar > 0) then suspend;
  end
end
