CREATE PROCEDURE ONAYBILGISIKAYDET(
    AONAYTIPI INTEGER,
    AHAREKETID INTEGER,
    AONAYDEGERI INTEGER,
    AONAYNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    AKULLANICIID INTEGER,
    AONAYLAYACAKKULLANICIID INTEGER,
    ATOPLAMRISKLIMITI DOUBLE PRECISION,
    ATOPLAMRISK DOUBLE PRECISION)
AS
declare variable KayitSayisi            integer;
declare variable FFirmaID    varchar(20);
declare variable FDovizID    varchar(5);
declare variable AcikSiparisToplami  Double Precision;
declare variable SiparisRiskKontroluYap  integer;
declare variable RiskHesaplanmasin integer;
declare variable RiskVar integer;
declare variable RiskVadesi integer;
begin
  AOnayNotu = cast(AOnayNotu as blob character set UTF8);
  Select SiparisRiskKontroluYap from Sistem where SistemID = 1 into :SiparisRiskKontroluYap;

  if ((AOnayTipi = 0 or AOnayTipi = 1 or AOnayTipi = 2) and SiparisRiskKontroluYap = 1) then/*Gülle Örme Siparis ve BoyaSipariş */
  begin

    if (AOnayTipi = 0 or AOnayTipi = 1) then
      Select BoyaSiparis.FirmaID, Firma.RiskHesaplanmasin From OrmeSiparisDetay
        Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = OrmeSiparisDetay.BoyaSiparisKayitID
        Left Join Firma on Firma.FirmaID = BoyaSiparis.FirmaID
        Where OrmeSiparisDetay.OrmeSiparisDetayID = :AHareketID into :FFirmaID, :RiskHesaplanmasin;
    else
      Select BoyaSiparis.FirmaID, Firma.RiskHesaplanmasin From BoyaSiparisDetay
        Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitID
        Left Join Firma on Firma.FirmaID = BoyaSiparis.FirmaID
        Where BoyaSiparisDetay.BoyaSiparisDetayID = :AHareketID into :FFirmaID, :RiskHesaplanmasin;


    if (AOnayDegeri = 1 and RiskHesaplanmasin = 0 ) then
    begin
      if ((AToplamRiskLimiti = 0 or AToplamRiskLimiti is Null) and RiskHesaplanmasin = 0) then Exception KayitHatasi ' Risk Limiti Tanımlanmamış! Sipariş Onaylanamaz..';

      Select AcikSipRiskToplami from SiparisRiskiniGetir(:FFirmaID) into :AcikSiparisToplami;
      if (AcikSiparisToplami is Null) then AcikSiparisToplami = 0;

      if (AcikSiparisToplami + AToplamRisk > AToplamRiskLimiti ) then Exception RiskLimitiAsilamaz;
    end

    if (AOnayTipi = 0 or AOnayTipi = 1 ) then/*Örme Siparis*/
    Update BoyaSiparisDetay SET OnaylayanKullaniciID = :AKullaniciID, OnayDurumu = :AOnayDegeri, ONAYTARIHI = Current_TimeStamp(0) WHERE OrmeSiparisDetayID = :AHareketID;
    Insert Into OnayLog (BoyaSiparisDetayID, OnayDegeri, OnayTipi, OnayNotu, OnaylayanKullaniciID) Values (:AHareketID, :AOnayDegeri, :AOnayTipi, cast('Örme Sipariş Onaylandığı için Oto Onaylandı' as blob character set UTF8), :AKullaniciID);
  end
  else if ((AOnayTipi = 0 or AOnayTipi = 1 or AOnayTipi = 2) and SiparisRiskKontroluYap = 2) then /*Fayteks*/
  begin
    if (AOnayTipi = 0 or AOnayTipi = 1) then
      Select BoyaSiparis.FirmaID, Firma.RiskVadesi, Firma.RiskHesaplanmasin From OrmeSiparisDetay
        Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = OrmeSiparisDetay.BoyaSiparisKayitID
        Left Join Firma on Firma.FirmaID = BoyaSiparis.FirmaID
        Where OrmeSiparisDetay.OrmeSiparisDetayID = :AHareketID into :FFirmaID, :RiskVadesi, :RiskHesaplanmasin;
    else
      Select BoyaSiparis.FirmaID, Firma.RiskVadesi, Firma.RiskHesaplanmasin From BoyaSiparisDetay
        Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitID
        Left Join Firma on Firma.FirmaID = BoyaSiparis.FirmaID
        Where BoyaSiparisDetay.BoyaSiparisDetayID = :AHareketID into :FFirmaID, :RiskVadesi, :RiskHesaplanmasin;

    if (AOnayDegeri = 1 and RiskVadesi > 0  and RiskHesaplanmasin = 0 ) then
    begin
      Select AcikSipRiskToplami,RiskVar from SiparisRiskiniGetir(:FFirmaID) into :AcikSiparisToplami, :RiskVar;
      if (RiskVar = 1) then Exception RiskLimitiAsilamaz 'Bakiye Aşımı';
    end
  end

  if (AOnayTipi = 0 or AOnayTipi = 1 ) then/*Örme Siparis*/
  begin
    if (AOnayTipi = 0) then
      Update OrmeSiparisDetay SET OnaylayanKullaniciID = :AKullaniciID, OnayDurumu = :AOnayDegeri, ONAYTARIHI = Current_TimeStamp(0) WHERE OrmeSiparisDetayID = :AHareketID;
    else
      Update OrmeSiparisDetay SET MaliOnaylayanKullaniciID = :AKullaniciID, MaliOnayDurumu = :AOnayDegeri, MaliONAYTARIHI = Current_TimeStamp(0) WHERE OrmeSiparisDetayID = :AHareketID;

    Insert Into OnayLog (OrmeSiparisDetayID, OnayDegeri, OnayTipi, OnayNotu, OnaylayanKullaniciID) Values (:AHareketID, :AOnayDegeri, :AOnayTipi, :AOnayNotu, :AKullaniciID);
  end
  else if (AOnayTipi = 2 ) then/*Boya Siparis*/
  begin
    Update BoyaSiparisDetay SET OnaylayanKullaniciID = :AKullaniciID, OnayDurumu = :AOnayDegeri, ONAYTARIHI = Current_TimeStamp(0) WHERE BoyaSiparisDetayID = :AHareketID;
    Insert Into OnayLog (BoyaSiparisDetayID, OnayDegeri, OnayTipi, OnayNotu, OnaylayanKullaniciID) Values (:AHareketID, :AOnayDegeri, :AOnayTipi, :AOnayNotu, :AKullaniciID);
  end
  else if (AOnayTipi = 3 ) then/*Orme Planlama*/
  begin
    Update OrmePlanlama SET OnaylayanKullaniciID = :AKullaniciID, OnayDurumu = :AOnayDegeri, OnayTarihi = Current_TimeStamp(0) WHERE OrmePlanlamaID = :AHareketID;
    Insert Into OnayLog (OrmePlanlamaID, OnayDegeri, OnayTipi, OnayNotu, OnaylayanKullaniciID) Values (:AHareketID, :AOnayDegeri, :AOnayTipi, :AOnayNotu, :AKullaniciID);
  end
  else if (AOnayTipi = 4 ) then/*Verilen Siparis*/
  begin
    Update VerilenSiparis SET OnaylayanKullaniciID = :AKullaniciID, OnayDurumu = :AOnayDegeri, OnayTarihi = Current_TimeStamp(0) WHERE VerilenSiparisKayitID = :AHareketID;
    Insert Into OnayLog (VerilenSiparisKayitID, OnayDegeri, OnayTipi, OnayNotu, OnaylayanKullaniciID) Values (:AHareketID, :AOnayDegeri, :AOnayTipi, :AOnayNotu, :AKullaniciID);
  end
  else if (AOnayTipi = 6 ) then/*Verilen Siparis Detay*/
  begin
    Update VerilenSiparisDetay SET OnaylayanKullaniciID = :AKullaniciID, OnayDurumu = :AOnayDegeri, OnayTarihi = Current_TimeStamp(0) WHERE VerilenSiparisDetayID = :AHareketID;
    Insert Into OnayLog (VerilenSiparisDetayID, OnayDegeri, OnayTipi, OnayNotu, OnaylayanKullaniciID) Values (:AHareketID, :AOnayDegeri, :AOnayTipi, :AOnayNotu, :AKullaniciID);
  end
  else if (AOnayTipi = 5 ) then/*Satin Alma Talep*/
  begin
    Update SatinAlmaTalepOnay SET OnayKullaniciId = :AKullaniciID, OnayDurumu = :AOnayDegeri, OnayTarihi = Current_TimeStamp(0) WHERE SatinAlmaTalepKayitId = :AHareketID and OnaylayacakKullaniciId = :AOnaylayacakKullaniciId;
    Insert Into OnayLog (SatinAlmaTalepKayitId, OnayDegeri, OnayTipi, OnayNotu, OnaylayanKullaniciID) Values (:AHareketID, :AOnayDegeri, :AOnayTipi, :AOnayNotu, :AKullaniciID);
  end
  else if (AOnayTipi = 7 or AOnayTipi = 8 ) then/*HG Siparis*/
  begin
    if (AOnayTipi = 7) then
      Update HGSiparisDetay SET OnaylayanKullaniciID = :AKullaniciID, OnayDurumu = :AOnayDegeri, ONAYTARIHI = Current_TimeStamp(0) WHERE HGSiparisDetayID = :AHareketID;
    else
      Update HGSiparisDetay SET MaliOnaylayanKullaniciID = :AKullaniciID, MaliOnayDurumu = :AOnayDegeri, MaliONAYTARIHI = Current_TimeStamp(0) WHERE HGSiparisDetayID = :AHareketID;

    Insert Into OnayLog (HGSiparisDetayID, OnayDegeri, OnayTipi, OnayNotu, OnaylayanKullaniciID) Values (:AHareketID, :AOnayDegeri, :AOnayTipi, :AOnayNotu, :AKullaniciID);
  end
  else if (AOnayTipi = 9 ) then/*HG İş Emri*/
  begin
    Update HGIsEmri SET OnaylayanKullaniciID = :AKullaniciID, OnayDurumu = :AOnayDegeri, OnayTarihi = Current_TimeStamp(0) WHERE HGIsEmriKayitID = :AHareketID;
    Insert Into OnayLog (HGIsEmriKayitID, OnayDegeri, OnayTipi, OnayNotu, OnaylayanKullaniciID) Values (:AHareketID, :AOnayDegeri, :AOnayTipi, :AOnayNotu, :AKullaniciID);
  end

end
