CREATE PROCEDURE ILAVESARFLARISTOKTANDUS(
    APARTIASAMALARIID INTEGER,
    APARTIASAMAILAVESARFID INTEGER)
AS
declare variable APartiKayitId       integer;
declare variable AReceteId           varchar(20);
declare variable AOrjBanyoCarpani       Double Precision;
declare variable ABanyoCarpani       Double Precision;
declare variable FPartiKilo             Double Precision;
declare variable FPartiMetre            Double Precision;
declare variable FUrunId                varchar(20);
declare variable FUrunBirimId           integer;
declare variable FUrunMutfakDepoNo      varchar(10);
declare variable FBirimFiyat            Double Precision;
declare variable FUrunDovizId           varchar(5);
declare variable FStokTabanDovizId      varchar(5);
declare variable FStokId                integer;
declare variable KayitBasladi           integer;
declare variable FHareketTipi           integer;
declare variable FFisSeriNo             varchar(2);
declare variable FFisNo                 integer;
declare Variable FMiktar                Double Precision;
declare Variable FMiktarTipi            integer;
declare variable FT1                    Double Precision;
declare variable FTarih                 TimeStamp;
declare variable FReceteMiktarOrani     Double Precision;
declare variable FUrunBirimCarpan Double Precision;
declare variable FSonAlisBirimCarpan    Double Precision;
declare variable FDovizKuru             Double Precision;
declare variable FNow date;
declare variable PartiSinif  varchar(1);
declare variable FToplamKilo            Double Precision;
declare variable FKilo                  Double Precision;
declare variable FToplamMetre           Double Precision;
declare variable FMetre                 Double Precision;
declare variable FPartiKayitId          integer;
declare variable FPartiAsamalariId      integer;
declare variable FStokGrupId            varchar(4);
declare variable FAsamaMutfakDepoNo      varchar(10);
declare variable BoyaKimyasalLotTakibiFifo EgeBoolean;
declare variable FLot varchar(150);
declare variable FKalanMiktar BOYAMIKTARSIZE;
declare variable FCIKILANMIKTAR BOYAMIKTARSIZE;
declare variable FCIKILACAKMIKTAR BOYAMIKTARSIZE;
declare variable GIRISYOK EGEBOOLEAN;
declare variable FTOPLAMLOT varchar(20);
begin
  FHareketTipi = 6;  /* Uretime Cikis */
  FNow = current_TimeStamp(0);
  FTarih =  FNow;
  FToplamKilo  = 0;
  FToplamMetre = 0;

  Select PartiKayitId, ReceteId, BanyoCarpani from PartiAsamalari where PartiAsamalari.PartiAsamalariID = :APartiAsamalariID into :APartiKayitId, :AReceteId, :AOrjBanyoCarpani;

  select PartiAsamalari.StokGrupId from PartiAsamalari
    where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId
    into :FStokGrupId;

  if (FStokGrupId is Null) then
  begin
    Select PartiGrup.StokGrupID from Parti
      left join PartiGrup on PartiGrup.PartiGrupID = Parti.PartiGrupId
      where Parti.PartiKayitID = :APartiKayitID
      into :FStokGrupId;
  end

  if (AOrjBanyoCarpani is null or AOrjBanyoCarpani = 0) then AOrjBanyoCarpani = 1;

  if (AReceteID is not null) then /* recete ise aynen receteyi stoktan dustugumuz gibi stoktan dusecegiz */
  begin
    for select PartiKayitId, PartiAsamalariId from PartiRecetesiDokumu_1(:APartiKayitId, :AReceteId, :APartiAsamalariId)
      into :FPartiKayitId, :FPartiAsamalariId do
    begin
      select sum(partiasamastok.GirisKilo), sum(partiasamastok.GirisMetre) from PartiAsamalari
        left join partiasamastok on partiasamastok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        where PartiAsamalari.PartiKayitId = :FPartiKayitId and PartiAsamalari.PartiAsamalariId = :FPartiAsamalariId and PartiAsamalari.ReceteId = :AReceteId and PartiAsamaStok.AsamayaDahil = 1
        into :FKilo, :FMetre;

      FToplamKilo = FToplamKilo + FKilo;
      FToplamMetre = FToplamMetre + FMetre;
    end

  end
  else
  begin
    select Sum(PartiAsamaStok.GirisKilo), Sum(PartiAsamaStok.GirisMetre) from PartiAsamaStok
      where PartiAsamaStok.PartiAsamalariID = :APartiAsamalariID and PartiAsamaStok.AsamayaDahil = 1
      into :FPartiKilo, :FPartiMetre;
    if (FPartiKilo is null) then FPartiKilo = 0;
    if (FPartiMetre is null) then FPartiMetre = 0;
    FToplamKilo = FPartiKilo;
    FToplamMetre = FPartiMetre;
  end

  select Parti.Sinif from Parti where Parti.PartikayitId = :APartiKayitId into :PartiSinif;

  KayitBasladi = 0;

  select STOKTABANDOVIZID, RECETEMIKTARORANI, BoyaKimyasalLotTakibiFifo from SISTEM where SISTEMID = 1 into :FStokTabanDovizId, :FReceteMiktarOrani, :BoyaKimyasalLotTakibiFifo;
  Select FisSeriNo from Kullanici where KullaniciId = 1 into :FFisSeriNo;

  if (FFisSeriNo is null) then FFisSeriNo = '';

  if (FReceteMiktarOrani is null or FReceteMiktarOrani = 0) then exception RecetemiktarOraniGirilmemis;

  select PartiAsamalari.MutfakDepoNo from PartiAsamalari
        where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId
        into :FAsamaMutfakDepoNo;

  for select PartiKayitId, PartiAsamalariId from PartiRecetesiDokumu_1(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    into :FPartiKayitId, :FPartiAsamalariId do
  begin
    Select sum(PartiAsamaStok.GirisKilo), sum(PartiAsamaStok.GirisMetre) from PartiAsamalari
      left join PartiAsamaStok on partiasamastok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      where PartiAsamalari.PartiKayitId = :FPartiKayitId and PartiAsamalari.PartiAsamalariId = :FPartiAsamalariId and PartiAsamalari.ReceteId = :AReceteId and PartiAsamaStok.AsamayaDahil = 1
      into :FPartiKilo, :FPartiMetre;

    if (FPartiKilo > 0 and FToplamKilo > 0) then
    begin
      ABanyoCarpani = AOrjBanyoCarpani / FToplamKilo * FPartiKilo;
      ABanyoCarpani = Yuvarla(ABanyoCarpani, 1);

      FPartiKilo = FPartiKilo / FReceteMiktarOrani;
    end

    if (FPartiMetre > 0 and FToplamMetre > 0) then
    begin
      ABanyoCarpani = AOrjBanyoCarpani / FToplamMetre * FPartiMetre;
      ABanyoCarpani = Yuvarla(ABanyoCarpani, 1);

      FPartiMetre = FPartiMetre / FReceteMiktarOrani;
    end

    for select PartiAsamaIlaveSarf.PartiAsamaIlaveSarfId, PartiAsamaIlaveSarfDetay.UrunId, PartiAsamaIlaveSarfDetay.UrunBirimId, Urun.MutfakDepoNo,
      UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / RUB.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId,
      PartiAsamaIlaveSarfDetay.Miktar, PartiAsamaIlaveSarfDetay.MiktarTipi, RUB.Carpan from PartiAsamaIlaveSarf
      left join PartiAsamaIlaveSarfDetay on PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfId = PartiAsamaIlaveSarf.PartiAsamaIlaveSarfId
      left join UrunBirim RUB on RUB.urunBirimId = PartiAsamaIlaveSarfDetay.UrunBirimId
      left join Urun on Urun.UrunId = PartiAsamaIlaveSarfDetay.UrunId
      left join UrunFiyat on UrunFiyat.UrunId = PartiAsamaIlaveSarfDetay.UrunId
      left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
      left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = Urun.KimyasalTipiId
      where PartiAsamaIlaveSarf.PartiAsamalariId = :APartiAsamalariId and
            (PartiAsamaIlaveSarf.PartiAsamaIlaveSarfId = :APartiAsamaIlaveSarfId or :APartiAsamaIlaveSarfId is null) and
            PartiAsamaIlaveSarf.StoktanDusuldu = 0
      into :APartiAsamaIlaveSarfId, :FUrunId, :FUrunBirimId, :FUrunMutfakDepoNo, :FBirimFiyat, :FUrunDovizId,
           :FMiktar, :FMiktarTipi, :FUrunBirimCarpan do
    begin
      /*Eğer aşamada mutfak depo var ise üründeki yerine bunu kullanacak*/
      if (FAsamaMutfakDepoNo is not Null) then FUrunMutfakDepoNo = FAsamaMutfakDepoNo;

      if (FUrunId is not null) then
      begin
        if (KayitBasladi = 0) then
        begin
          select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;

          FStokId = gen_id(Stok_Id_Gen, 1);

          insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, Sinif)
                      values(:FStokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :FTarih, :FStokTabanDovizId, 1, 1, :PartiSinif);

          KayitBasladi = 1;
        end

        if (FMiktar is null) then FMiktar = 0;

        if (FMiktarTipi = 1 and FPartiKilo > 0) then FMiktar = FMiktar * FPartiKilo;
        else if (FMiktarTipi = 1 and FPartiMetre > 0) then FMiktar = FMiktar * FPartiMetre;
        else if (FMiktarTipi = 2) then FMiktar = FMiktar * ABanyoCarpani;

        if (FPartiKilo > 0 and FToplamKilo > 0) then FMiktar = (FMiktar / FToplamKilo) * FPartiKilo * FReceteMiktarOrani;
        else if (FPartiMetre > 0 and FToplamMetre > 0) then FMiktar = (FMiktar / FToplamMetre) * FPartiMetre * FReceteMiktarOrani;
        FMiktar = Yuvarla(FMiktar, 10000);

        if (FBirimFiyat is null) then FBirimFiyat = 0;

        if (FUrunDovizId is not null and FBirimFiyat > 0) then
        if (FStokTabanDovizId <> FUrunDovizId) then
        begin
          select DovizKuru from EnYakinDovizKuruGetir(:FUrunDovizId, :FStokTabanDovizId, :FTarih) into :FDovizKuru;
          FBirimFiyat = FBirimFiyat * FDovizKuru;
        end
 /*can*/
        FLot = Null;
        GirisYok = 1;
        FCikilanMiktar = 0;
        FCikilacakMiktar = 0;
        FToplamLot = '';
        --FMiktar = FMiktar + FBanyo;

        if (BoyaKimyasalLotTakibiFifo = 1) then
        begin
          for select UrunFifoBilgisi.Lot, UrunFifoBilgisi.KalanMiktar * (:FUrunBirimCarpan / UrunBirim.Carpan) from UrunFifoBilgisi(:FUrunId, :FAsamaMutfakDepoNo)
              left join UrunBirim on UrunBirim.UrunBirimID = UrunFifoBilgisi.UrunBirimId
              order by GirisTarihi into :FLot, FKalanMiktar do
          begin
            GirisYok = 0;

            if (FKalanMiktar < FMiktar - FCikilanMiktar) then
              FCikilacakMiktar = FKalanMiktar;
            else
              FCikilacakMiktar = FMiktar - FCikilanMiktar;

            insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, StokGrupId, UrunId, UrunBirimId, Miktar, BirimFiyat, PartiKayitId, PartiAsamalariId, PartiAsamaIlaveSarfID, Otomasyon, Lot)
                    values(:FTarih, :FHareketTipi, :FStokId, :FUrunMutfakDepoNo, :FStokGrupId, :FUrunId, :FUrunBirimId, :FCikilacakMiktar, :FBirimFiyat, :APartiKayitId, :APartiAsamalariId, :APartiAsamaIlaveSarfID, 1,:FLot);


            FCikilanMiktar = FCikilanMiktar + FCikilacakMiktar;

            if (FToplamLot = '') then FToplamLot = FLot;
            else if (Char_Length(FToplamLot||'+'||Flot) < 21) then FToplamLot = FToplamLot||'+'||Flot;

            if (FCikilanMiktar = FMiktar) then break;
          end
        end

        FMiktar = FMiktar - FCikilanMiktar;
        if (FMiktar > 0 or GirisYok = 1) then
 /*can*/
        insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, StokGrupId, UrunId, UrunBirimId, Miktar, BirimFiyat, PartiKayitId, PartiAsamalariId, PartiAsamaIlaveSarfID, Otomasyon)
                    values(:FTarih, :FHareketTipi, :FStokId, :FUrunMutfakDepoNo, :FStokGrupId, :FUrunId, :FUrunBirimId, :FMiktar, :FBirimFiyat, :FPartiKayitId, :FPartiAsamalariId, :APartiAsamaIlaveSarfID, 1);
      end
    end
  end

  if (APartiAsamaIlaveSarfID is null) then
    Update PartiAsamaIlaveSarf Set StoktanDusuldu = 1 where PartiAsamaIlaveSarf.PartiAsamalariId = :APartiAsamalariId and StoktanDusuldu = 0;
  else
    Update PartiAsamaIlaveSarf Set StoktanDusuldu = 1 where PartiAsamaIlaveSarf.PartiAsamaIlaveSarfID = :APartiAsamaIlaveSarfID;
end
