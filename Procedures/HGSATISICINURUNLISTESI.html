CREATE PROCEDURE HGSATISICINURUNLISTESI(
    AHGSIPARISKAYITID INTEGER,
    ADEPOID VARCHAR(10))
RETURNS (
    HGSIPARISDETAYURUNID INTEGER,
    HGSIPARISDETAYID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BIRIMID VARCHAR(2),
    SIPMIKTAR BOYAMIKTARSIZE,
    MODELID MODELKODUSIZE,
    MODELADI ADI40,
    MUSTERIMODELNO VARCHAR(30),
    DEPOID VARCHAR(10),
    DEPOSTOKMEVCUDU DOUBLE PRECISION,
    STOKMEVCUDU DOUBLE PRECISION,
    RESULTID VARCHAR(100))
AS
declare variable StokVar EgeBoolean;
declare variable HgStokEksiyeDusemez EgeBoolean;
begin
  select HgStokEksiyeDusemez from Sistem where SistemId = 1 into :HgStokEksiyeDusemez;

  if (ADepoId = '') then ADepoId = Null;

  For Select HGSiparisDetayUrun.HGSiparisDetayUrunID, HGSiparisDetayUrun.HGSiparisDetayID, HGSiparisDetayUrun.UrunId, Urun.Adi, UrunBirim.BirimId,
             HGSiparisDetayUrun.Miktar, Model.ModelID, Model.Adi, HGSiparisDetay.MusteriModelNo from HGSiparisDetayUrun
    left join Urun on Urun.UrunID = HGSiparisDetayUrun.UrunID
    left join UrunBirim on UrunBirim.UrunId = Urun.UrunId and UrunBirim.DefaultBirim = 1
    left join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayID = HGSiparisDetayUrun.HGSiparisDetayID
    left join Model on Model.ModelKayitID = HGSiparisDetay.ModelKayitID
    where HGSiparisDetayUrun.HGSiparisDetayAsortiId is null and HGSiparisDetayUrun.HGSiparisKayitID = :AHGSiparisKayitID
   into :HGSiparisDetayUrunID, :HGSiparisDetayID, :UrunId, :UrunAdi, :BirimId, :SipMiktar, :ModelID, :ModelAdi, :MusteriModelNo do
  begin
    DepoStokMevcudu = null;
    StokMevcudu = Null;
    StokVar = 0;
    DepoId = :ADepoId;

    ResultId = '"' || EgemenIntegerToStr(HGSiparisDetayID) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(HGSiparisDetayUrunID) || '",' ;
    ResultId = ResultId || '"' || UrunID || '",' ;
    ResultId = ResultId || '"' || BirimId || '"' ;

    Select Sum(DepoStokMevcudu) from UrunDepoStokGetir(:UrunId, Null) into :StokMevcudu;

    For Select Depo.DepoId from Depo
        left join Unite on Unite.UniteId  = Depo.UniteId
        where Unite.HGUnitesimi = 1 and Depo.Kapali = 0 and HgYariMamul = 0 and (DepoId = :ADepoId or :ADepoId is null) into :DepoId do
    begin
      Select Sum(DepoStokMevcudu) from UrunDepoStokGetir(:UrunId, :DepoId) into :DepoStokMevcudu;

      if (HgStokEksiyeDusemez = 0 or (HgStokEksiyeDusemez = 1 and DepoStokMevcudu > 0)) then
      begin
        StokVar = 1;
        suspend;
      end
    end

    if (StokVar = 0) then
      suspend;
  end
end
