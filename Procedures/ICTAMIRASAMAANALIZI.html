CREATE PROCEDURE ICTAMIRASAMAANALIZI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATAMIRTIPIGRUPID VARCHAR(4),
    ATAMIRTIPIID INTEGER,
    ARAPORTIPI INTEGER,
    ABITTI INTEGER,
    APARTITIPI INTEGER,
    AFIRMAID VARCHAR(20),
    AUCRETDURUMU INTEGER,
    ATAMIRDUZELTMETIPIID VARCHAR(10),
    ATAMIRCOZUMSEKLIID INTEGER,
    AISLEMEALINACAK INTEGER,
    APARTIGRUPID VARCHAR(4),
    ASIPARISGRUPID VARCHAR(4),
    ARECETETURUID INTEGER,
    AHATA1ASAMAID INTEGER,
    AHATA2ASAMAID INTEGER,
    AHATA3ASAMAID INTEGER,
    AONLEYICIFAALIYETTIPIID INTEGER)
RETURNS (
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    TOPLAMKILO NUMERIC(12,3),
    TOPLAMURETIM NUMERIC(12,3),
    ORAN DOUBLE PRECISION,
    ASAMATAMIRRAPORSIRANO INTEGER,
    TOPLAMURETIMMETRE NUMERIC(12,3),
    TOPLAMMETRE NUMERIC(12,3),
    ORANMETRE DOUBLE PRECISION)
AS
begin
  for Select AsamaId, AsamaAdi, AsamaTamirRaporSiraNo, Sum(Kilo), Sum(Metre) from IsletmeIciTamirRaporuDetayi(:ABaslangicTarihi, :ABitisTarihi, :ATamirTipiGrupId, :ATamirTipiId, :ARaporTipi, :ABitti, :APartiTipi, :AFirmaId, :AUcretDurumu, :ATamirDuzeltmeTipiId, :ATamirCozumSekliId, :AIslemeAlinacak, :APartiGrupId, :ASiparisGrupId, :AReceteTuruId, :AHata1AsamaId, :AHata2AsamaId, :AHata3AsamaId, :AOnleyiciFaaliyetTipiId)
  group by AsamaId, AsamaAdi, AsamaTamirRaporSiraNo
  into :AsamaId, :AsamaAdi, :AsamaTamirRaporSiraNo, :ToplamKilo, :ToplamMetre do
  begin
    ToplamUretim = 0;
    Select Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from PartiAsamaStok
      left join PartiStok on PartiStok.PartiSinifID = PartiAsamaStok.PartiSinifID
      where PartiAsamaStok.CikisTarihi >= :ABaslangicTarihi and PartiAsamaStok.CikisTarihi < :ABitisTarihi and PartiAsamaStok.AsamaID = :AsamaId
    into :ToplamUretim, :ToplamUretimMetre;

    if (ToplamUretim is not null and ToplamUretim > 0) then  Oran = ToplamKilo * 100 / ToplamUretim;
    else Oran = Null;

    if (ToplamUretimMetre is not null and ToplamUretimMetre > 0) then  OranMetre = ToplamMetre * 100 / ToplamUretimMetre;
    else OranMetre = Null;
    suspend;
  end
end
