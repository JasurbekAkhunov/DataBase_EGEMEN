CREATE PROCEDURE LAWERSTOKTANDUS(
    ATARIH TIMESTAMP,
    AURUNKODU VARCHAR(10),
    AURUNBIRIM VARCHAR(2),
    ATOPLAMMIKTAR DOUBLE PRECISION,
    APARTINO VARCHAR(10),
    ALOT VARCHAR(20),
    ASTOKID INTEGER)
RETURNS (
    STOKID INTEGER)
AS
declare variable StoktanDusuldu integer;
declare variable FUrunId                varchar(20);
declare variable FUrunBirimId           integer;
declare variable FUrunMutfakDepoNo      varchar(10);
declare variable FBirimFiyat            Double Precision;
declare variable FUrunDovizId           varchar(5);
declare variable FStokTabanDovizId      varchar(5);
declare variable FHareketTipi           integer;
declare variable FFisSeriNo             varchar(2);
declare variable FFisNo                 integer;
declare variable FDovizKuru             Double Precision;
declare variable FPartiYil integer;
declare variable FPartiID integer;
declare variable PartiKayitID integer;
declare variable PartiSinif varchar(2);
declare variable FUrunBirim varchar(2);
declare variable PartiTopMiktar Double Precision;
declare variable PartiMiktar Double Precision;
declare variable FMiktar Double Precision;
begin
  select STOKTABANDOVIZID from SISTEM where SISTEMID = 1 into :FStokTabanDovizId;
  FHareketTipi = 6;  /* Uretime Cikis */

  Select FisSeriNo from Kullanici where KullaniciId = 1 into :FFisSeriNo;
  if (FFisSeriNo is null) then FFisSeriNo = '';

  FPartiYil = 2000 + Cast(SubString(:APartiNo from 2 for 2) as Integer);
  FPartiID = Cast(SubString(:APartiNo from 5 for 10) as Integer);

  if (SubString(:APartiNo from 1 for 1) = 'P') then
    Select Parti.PartikayitId, Parti.Sinif from Parti where Parti.PartiYil = :FPartiYil and Parti.PartiID = :FPartiID into :PartiKayitID, :PartiSinif;
  else if (SubString(:APartiNo from 1 for 1) = 'E') then
    Select first 1 Parti.PartikayitId, Parti.Sinif from ReceteIsEmriDetay
      Left Join Parti on Parti.PartiKayitID = ReceteIsEmriDetay.PartiKayitID
    where ReceteIsEmriDetay.ReceteIsEmriYil = :FPartiYil and ReceteIsEmriDetay.ReceteIsEmriID = :FPartiID into :PartiKayitID, :PartiSinif;

  if (PartiKayitID is Null) then exception KayitHatasi ' Parti bilgisi bulunamadÄ±';

  if (AStokId is Null or AStokId = 0) then
  begin
    select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;
    StokId = gen_id(Stok_Id_Gen, 1);

    insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, Sinif)
             values(:StokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :ATarih, :FStokTabanDovizId, 1, 1, :PartiSinif);

  end
  else StokId = AStokId;

  Select first 1 BirimID from Birim where DozajlamaBirimKodu = :AUrunBirim into :FUrunBirim;

  Select first 1 DozajlamaSistemiUrun.UrunID, RUB.UrunBirimID, Urun.MutfakDepoNo, UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / RUB.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId from DozajlamaSistemiUrun
    Left Join Urun on Urun.UrunID = DozajlamaSistemiUrun.UrunID
    left join UrunBirim RUB on RUB.BirimID = :FUrunBirim and RUB.UrunId = DozajlamaSistemiUrun.UrunID
    left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
    left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
    where DozajlamaSistemiUrun.DozajlamaSistemiID = 3 and DozajlamaSistemiUrun.DozajlamaUrunID = :AUrunKodu
  into :FUrunId, :FUrunBirimID, :FUrunMutfakDepoNo, :FBirimFiyat, :FUrunDovizId;

  if (FUrunId is Null) then Exception KayitHatasi ' ÃrÃ¼n Kodu TanÄ±mlanmamÄ±Å!';

  if (FBirimFiyat is null) then FBirimFiyat = 0;

  if (FUrunDovizId is not null and FBirimFiyat > 0) then
  if (FStokTabanDovizId <> FUrunDovizId) then
  begin
    select DovizKuru from EnYakinDovizKuruGetir(:FUrunDovizId, :FStokTabanDovizId, :ATarih) into :FDovizKuru;
    FBirimFiyat = FBirimFiyat * FDovizKuru;
  end

  if (SubString(:APartiNo from 1 for 1) = 'P') then
    insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, UrunId, UrunBirimId, Miktar, BirimFiyat, PartiKayitId, Otomasyon, UretimeCikisTarihi, Lot)
                 values(:ATarih, :FHareketTipi, :StokId, :FUrunMutfakDepoNo, :FUrunId, :FUrunBirimID, :AToplamMiktar, :FBirimFiyat, :PartiKayitID, 1, :ATarih, :ALot);
  else if (SubString(:APartiNo from 1 for 1) = 'E') then
  begin
    Select Sum(PartiStok.Miktar) from ReceteIsEmriDetay
      Left Join PartiStok on PartiStok.PartiKayitID = ReceteIsEmriDetay.PartiKayitID
    where ReceteIsEmriDetay.ReceteIsEmriYil = :FPartiYil and ReceteIsEmriDetay.ReceteIsEmriID = :FPartiID into :PartiTopMiktar;

    For Select PartiStok.PartiKayitID, Sum(PartiStok.Miktar) from ReceteIsEmriDetay
      Left Join PartiStok on PartiStok.PartiKayitID = ReceteIsEmriDetay.PartiKayitID
    where ReceteIsEmriDetay.ReceteIsEmriYil = :FPartiYil and ReceteIsEmriDetay.ReceteIsEmriID = :FPartiID
    Group By PartiStok.PartiKayitID
    into :PartiKayitID, :PartiMiktar do
    begin
      FMiktar = (PartiMiktar / PartiTopMiktar) * AToplamMiktar;
      FMiktar = Yuvarla(FMiktar, 1000);
      insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, UrunId, UrunBirimId, Miktar, BirimFiyat, PartiKayitId, Otomasyon, UretimeCikisTarihi, Lot)
                 values(:ATarih, :FHareketTipi, :StokId, :FUrunMutfakDepoNo, :FUrunId, :FUrunBirimID, :FMiktar, :FBirimFiyat, :PartiKayitID, 1, :ATarih, :ALot);
    end
  end
  suspend;
end
