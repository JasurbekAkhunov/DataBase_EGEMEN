CREATE PROCEDURE ELYAFURETIMRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ALOT VARCHAR(30),
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    AELYAFURETIMISEMRIKAYITID INTEGER)
RETURNS (
    ELYAFURETIMID INTEGER,
    TELEFID INTEGER,
    ELYAFURETIMISEMRIKAYITID INTEGER,
    ELYAFURETIMISEMRIYIL INTEGER,
    ELYAFURETIMISEMRIID INTEGER,
    ELYAFURETIMISEMRISTR VARCHAR(20),
    HAMADI VARCHAR(250),
    HAMADI1 VARCHAR(250),
    HAMADI2 VARCHAR(250),
    HAMADI3 VARCHAR(250),
    LOT VARCHAR(30),
    IPLIKSTOKID INTEGER,
    KAPNO INTEGER,
    URETIMTARIHIDATE DATE,
    URETIMTARIHI TIMESTAMP,
    VARDIYANO INTEGER,
    IPLIKIRSALIYEID INTEGER,
    IPLIKIRSALIYEDETAYID INTEGER,
    MIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    IPLIKMAKINAID VARCHAR(10),
    IPLIKDEPOID VARCHAR(10),
    HAREKETTIPISTR VARCHAR(50),
    CIKISIPLIKIRSALIYEDETAYID INTEGER,
    SEVKTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    IPLIKISLETMEADI VARCHAR(30),
    HAMTIPIADI VARCHAR(20),
    KALITENO VARCHAR(60),
    GELISSIRASI VARCHAR(60),
    ISEMRILOT VARCHAR(60))
AS
declare variable HareketTipi            integer;
declare variable shtSatis               integer;
declare variable shtSarimaCikis         integer;
declare variable shtBukumeCikis         integer;
declare variable FKaliteNo              integer;
declare variable FGelisSirasi           integer;
declare variable FIsEmriLot             varchar(30);
begin
  if (AHamTipiId = 0) then Select hstHamPamuk from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 1) then Select hstTelef from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 2) then Select hstCigit from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 3) then Select hstElyaf from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 4) then Select hstKirpinti from HamTipiGetir into :AHamTipiId;
  else AHamTipiId = null;

  Select shtSatis, ShtSarimaCikis, shtBukumeCikis from IplikStokHareketTipiGetir into :shtSatis, :ShtSarimaCikis, :shtBukumeCikis;

  For Select ElyafUretim.ElyafUretimID, ElyafUretimIsEmri.ElyafUretimIsEmriKayitID, ElyafUretimIsEmri.IsEmriYil, ElyafUretimIsEmri.IsEmriID, '(' || egemenintegertostr(ElyafUretimIsEmri.IsEmriYil) || ') - ' || egemenintegertostr(ElyafUretimIsEmri.IsEmriID),
             Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, ElyafUretim.Lot, ElyafUretim.IplikStokId,  ElyafUretim.KapNo, ElyafUretim.UretimTarihi, cast(ElyafUretim.UretimTarihi as date), ElyafUretim.IplikIrsaliyeId,
             ElyafUretim.IplikIrsaliyeDetayId, ElyafUretim.Miktar, ElyafUretim.NetMiktar, ElyafUretim.IplikMakinaId,
             ElyafUretim.IplikDepoId, IplikStok.CikisIplikIrsaliyeDetayId, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.HareketTipi,
             Ik.adi, ElyafUretim.InsertTarihi, Sk.Adi, ElyafUretim.DegisimTarihi, IplikIsletme.Adi, HamTipi.Adi from ElyafUretim
    Left Join Kullanici IK on IK.KullaniciId = ElyafUretim.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = ElyafUretim.KullaniciId
    left join IplikStok on IplikStok.IplikStokId = ElyafUretim.IplikStokId and IplikStok.Numarator = ElyafUretim.Numarator
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.CikisIplikIrsaliyeDetayId
    left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikIrsaliyeDetay.IplikIrsaliyeId
    left Join ElyafUretimIsEmri on ElyafUretimIsEmri.ElyafUretimIsEmriKayitId = ElyafUretim.ElyafUretimIsEmriKayitId
    Left Join IplikIsletme on IplikIsletme.IplikIsletmeId = ElyafUretimIsEmri.IplikIsletmeId
    left join Ham on Ham.Hamid = ElyafUretimIsEmri.HamId
    left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId

    where (Cast(ElyafUretim.UretimTarihi as Date) >= :ABaslangicTarihi and Cast(ElyafUretim.UretimTarihi as Date) <= :ABitisTarihi) and
          (ElyafUretim.Lot = :ALot or :ALot is null) and
          (ElyafUretimIsEmri.HamId = :AHamId or :AHamId is null) and
          (Ham.HamTipiID = :AHamTipiID or :AHamTipiID is null) and
          (ElyafUretim.ElyafUretimIsEmriKayitId = :AElyafUretimIsEmriKayitId or :AElyafUretimIsEmriKayitId is Null)
    order by ElyafUretim.Lot, ElyafUretim.KapNo
    into :ElyafUretimID, :ElyafUretimIsEmriKayitID, :ElyafUretimIsEmriYil, :ElyafUretimIsEmriID, :ElyafUretimIsEmriStr, :HamAdi, :HamAdi1, :HamAdi2, :HamAdi3, :Lot, :IplikStokId, :KapNo, :UretimTarihi, :UretimTarihiDate, :IplikIrsaliyeId, :IplikIrsaliyeDetayId, :Miktar,
         :NetMiktar, :IplikMakinaId, :IplikDepoId, :CikisIplikIrsaliyeDetayId, :SevkTarihi, :HareketTipi,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :IplikIsletmeAdi, :HamTipiAdi do
  begin

    select VardiyaNo from OrmeVardiyaHesapla(:UretimTarihi ) into :VardiyaNo;
    if (HareketTipi is null) then HareketTipiStr = '';
    else if (HareketTipi = shtSatis) then HareketTipiStr = 'Satış';
    else if (HareketTipi = ShtSarimaCikis) then HareketTipiStr = 'Yumuşak Sarım';
    else if (HareketTipi = ShtBukumeCikis) then HareketTipiStr = 'Büküme Çıkış';
    else HareketTipiStr = 'Hatalı';

    KaliteNo = '';
    For Select IplikIrsaliyeDetay.KaliteNo from ElyafUretimIsEmriRecete
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = ElyafUretimIsEmriRecete.IplikIrsaliyeDetayId
      where ElyafUretimIsEmriRecete.ElyafUretimIsEmriKayitId = :ElyafUretimIsEmriKayitId
      Group by IplikIrsaliyeDetay.KaliteNo
    into :FKaliteNo do
    begin
    if (Char_Length(KaliteNo) + Char_Length(FKaliteNo) < 54) then
      begin
       if (KaliteNo <> '' ) then KaliteNo = KaliteNo || cast(', ' as varchar(2) character set UTF8);
       KaliteNo = cast(KaliteNo as blob character set utf8) || FKaliteNo;
      end
    end

    GelisSirasi = '';
    For Select IplikIrsaliyeDetay.GelisSirasi from ElyafUretimIsEmriRecete
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = ElyafUretimIsEmriRecete.IplikIrsaliyeDetayId
      where ElyafUretimIsEmriRecete.ElyafUretimIsEmriKayitId = :ElyafUretimIsEmriKayitId
      Group by IplikIrsaliyeDetay.GelisSirasi
    into :FGelisSirasi do
    begin
    if (Char_Length(GelisSirasi) + Char_Length(FGelisSirasi) < 54) then
      begin
       if (GelisSirasi <> '' ) then GelisSirasi = GelisSirasi || cast(', ' as varchar(2) character set UTF8);
       GelisSirasi = cast(GelisSirasi as blob character set utf8) || FGelisSirasi;
      end
    end

    IsEmriLot = '';
    For Select IplikIrsaliyeDetay.Lot from ElyafUretimIsEmriRecete
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = ElyafUretimIsEmriRecete.IplikIrsaliyeDetayId
      where ElyafUretimIsEmriRecete.ElyafUretimIsEmriKayitId = :ElyafUretimIsEmriKayitId
      Group by IplikIrsaliyeDetay.lot
    into :FIsEmriLot do
    begin
    if (Char_Length(IsEmriLot) + Char_Length(FIsEmriLot) < 54) then
      begin
       if (IsEmriLot <> '' ) then IsEmriLot = IsEmriLot || cast(', ' as varchar(2) character set UTF8);
       IsEmriLot = cast(IsEmriLot as blob character set utf8) || FIsEmriLot;
      end
    end
    suspend;
  end

  ElyafUretimID = Null;
  IplikMakinaId = Null;

  For Select Telef.TelefId, Telef.ElyafUretimIsEmriKayitID, ElyafUretimIsEmri.IsEmriYil, ElyafUretimIsEmri.IsEmriID, '(' || EgemenIntegertoStr(ElyafUretimIsEmri.IsEmriYil) || ') - ' || egemenintegertostr(ElyafUretimIsEmri.IsEmriID),
             Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Telef.Lot, Telef.IplikStokId,  Telef.KapNo, Telef.InsertTarihi, cast(Telef.InsertTarihi as date), Telef.IplikIrsaliyeId,
             Telef.IplikIrsaliyeDetayId, Telef.Miktar, Telef.NetMiktar,
             Telef.IplikDepoId, IplikStok.CikisIplikIrsaliyeDetayId, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.HareketTipi,
             Ik.adi, Telef.InsertTarihi, Sk.Adi, Telef.DegisimTarihi, IplikIsletme.Adi, HamTipi.Adi, IplikIrsaliyeDetay.KaliteNo, IplikIrsaliyeDetay.GelisSirasi,
             IplikIrsaliyeDetay.Lot   from Telef
    Left Join Kullanici IK on IK.KullaniciId = Telef.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = Telef.KullaniciId
    left join IplikStok on IplikStok.IplikStokId = Telef.IplikStokId and IplikStok.Numarator = Telef.Numarator
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.CikisIplikIrsaliyeDetayId
    left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikIrsaliyeDetay.IplikIrsaliyeId
    left Join ElyafUretimIsEmri on ElyafUretimIsEmri.ElyafUretimIsEmriKayitId = Telef.ElyafUretimIsEmriKayitId
    Left Join IplikIsletme on IplikIsletme.IplikIsletmeId = ElyafUretimIsEmri.IplikIsletmeId
    left join Ham on Ham.Hamid = Telef.HamID  --ElyafUretimIsEmri.Hamid
    left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId

    where (Cast(Telef.InsertTarihi as Date) >= :ABaslangicTarihi and Cast(Telef.InsertTarihi as Date) <= :ABitisTarihi) and
          (Telef.Lot = :ALot or :ALot is null) and
          (Telef.HamId = :AHamId or :AHamId is null) and
          (Ham.HamTipiID = :AHamTipiID or :AHamTipiID is null) and
          (Telef.ElyafUretimIsEmriKayitId = :AElyafUretimIsEmriKayitId or :AElyafUretimIsEmriKayitId is Null)
    order by Telef.Lot, Telef.KapNo
    into :TelefId, :ElyafUretimIsEmriKayitID, :ElyafUretimIsEmriYil, :ElyafUretimIsEmriID, :ElyafUretimIsEmriStr,
         :HamAdi, :HamAdi1, :HamAdi2, :HamAdi3, :Lot, :IplikStokId, :KapNo, :UretimTarihi, :UretimTarihiDate, :IplikIrsaliyeId, :IplikIrsaliyeDetayId, :Miktar,
         :NetMiktar, :IplikDepoId, :CikisIplikIrsaliyeDetayId, :SevkTarihi, :HareketTipi,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :IplikIsletmeAdi, :HamTipiAdi,
         :KaliteNo, :GelisSirasi, :IsEmriLot  do
  begin

    select VardiyaNo from OrmeVardiyaHesapla(:UretimTarihi ) into :VardiyaNo;
    if (HareketTipi is null) then HareketTipiStr = '';
    else if (HareketTipi = shtSatis) then HareketTipiStr = 'Satış';
    else if (HareketTipi = ShtSarimaCikis) then HareketTipiStr = 'Yumuşak Sarım';
    else if (HareketTipi = ShtBukumeCikis) then HareketTipiStr = 'Büküme Çıkış';
    else HareketTipiStr = 'Hatalı';

        KaliteNo = '';
    For Select IplikIrsaliyeDetay.KaliteNo from ElyafUretimIsEmriRecete
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = ElyafUretimIsEmriRecete.IplikIrsaliyeDetayId
      where ElyafUretimIsEmriRecete.ElyafUretimIsEmriKayitId = :ElyafUretimIsEmriKayitId
      Group by IplikIrsaliyeDetay.KaliteNo
    into :FKaliteNo do
    begin
    if (Char_Length(KaliteNo) + Char_Length(FKaliteNo) < 54) then
      begin
       if (KaliteNo <> '' ) then KaliteNo = KaliteNo || cast(', ' as varchar(2) character set UTF8);
       KaliteNo = cast(KaliteNo as blob character set utf8) || FKaliteNo;
      end
    end

    GelisSirasi = '';
    For Select IplikIrsaliyeDetay.GelisSirasi from ElyafUretimIsEmriRecete
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = ElyafUretimIsEmriRecete.IplikIrsaliyeDetayId
      where ElyafUretimIsEmriRecete.ElyafUretimIsEmriKayitId = :ElyafUretimIsEmriKayitId
      Group by IplikIrsaliyeDetay.GelisSirasi
    into :FGelisSirasi do
    begin
    if (Char_Length(GelisSirasi) + Char_Length(FGelisSirasi) < 54) then
      begin
       if (GelisSirasi <> '' ) then GelisSirasi = GelisSirasi || cast(', ' as varchar(2) character set UTF8);
       GelisSirasi = cast(GelisSirasi as blob character set utf8) || FGelisSirasi;
      end
    end

    IsEmriLot = '';
    For Select IplikIrsaliyeDetay.Lot from ElyafUretimIsEmriRecete
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = ElyafUretimIsEmriRecete.IplikIrsaliyeDetayId
      where ElyafUretimIsEmriRecete.ElyafUretimIsEmriKayitId = :ElyafUretimIsEmriKayitId
      Group by IplikIrsaliyeDetay.lot
    into :FIsEmriLot do
    begin
    if (Char_Length(IsEmriLot) + Char_Length(FIsEmriLot) < 54) then
      begin
       if (IsEmriLot <> '' ) then IsEmriLot = IsEmriLot || cast(', ' as varchar(2) character set UTF8);
       IsEmriLot = cast(IsEmriLot as blob character set utf8) || FIsEmriLot;
      end
    end

    suspend;
  end
end
