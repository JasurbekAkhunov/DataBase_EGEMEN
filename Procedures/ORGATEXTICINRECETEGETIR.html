CREATE PROCEDURE ORGATEXTICINRECETEGETIR(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    DOZAJLAMAURUNID VARCHAR(15),
    URUNTIPIID INTEGER,
    DOZAJLAMAPROGRAMNO INTEGER,
    OTOMASYONISTASYONNO INTEGER,
    SABLONDOZAJLAMAPROGRAMNO INTEGER,
    OTOMASYONMIKTAR DOUBLE PRECISION,
    DOZAJLAMABIRIMKODU VARCHAR(10),
    OTOMASYONRECETEMIKTAR DOUBLE PRECISION,
    DOZAJLAMASEKLI INTEGER,
    DOZAJLAMASEKLIIPLIK INTEGER,
    ISTEKNO INTEGER,
    COUNTER INTEGER,
    SIRANO INTEGER,
    URETIMECIKISFIYATIBUGUN DOUBLE PRECISION,
    BIRIMID VARCHAR(2))
AS
declare variable Miktar                Double precision;
declare variable Banyo                 Double precision;
declare variable ReceteMiktar          Double precision;
declare variable ReceteBanyo           Double precision;

declare variable ReceteAsamaId         integer;
declare variable ReceteAltAsamaId      integer;
declare variable LastReceteAsamaId integer;
declare variable LastReceteAltAsamaId integer;
declare variable LastSiraNo integer;
declare variable FLabReceteGrupID integer;
declare variable DozajlamaIplikUrunId  varchar(15);
declare variable DozajlamaKumasUrunId  varchar(15);
declare variable EgemenFirmaID integer;
declare variable OrgatexeTekBasinaAktarilir integer;
begin
  Select EgemenFirmaID from Sistem where SistemID = 1 into :EgemenFirmaID;
  LastReceteAsamaId = -2;
  LastReceteAltAsamaId = -2;
  LastSiraNo = -1;
  IstekNo = 0;

  for select DozajlamaIcinReceteHazirla.UrunId, DozajlamaIcinReceteHazirla.UrunAdi, DozajlamaIcinReceteHazirla.DozajlamaUrunId, DozajlamaIcinReceteHazirla.DozajlamaIplikUrunId,
             DozajlamaIcinReceteHazirla.UrunTipiId, DozajlamaIcinReceteHazirla.DozajlamaProgramNo, DozajlamaIcinReceteHazirla.SablonDozajlamaProgramNo,
             DozajlamaIcinReceteHazirla.DozajlamaBirimKodu, DozajlamaIcinReceteHazirla.Miktar, DozajlamaIcinReceteHazirla.Banyo,
             DozajlamaIcinReceteHazirla.ReceteMiktar, DozajlamaIcinReceteHazirla.ReceteBanyo,
             DozajlamaIcinReceteHazirla.ReceteAsamaId, DozajlamaIcinReceteHazirla.ReceteAltAsamaId, DozajlamaIcinReceteHazirla.DozajlamaSekli,
             DozajlamaIcinReceteHazirla.SiraNo, DozajlamaIcinReceteHazirla.UretimeCikisFiyatiBugun, DozajlamaIcinReceteHazirla.DozajlamaSekliIplik,
             DozajlamaIcinReceteHazirla.OrgatexeTekBasinaAktarilir, DozajlamaIcinReceteHazirla.OtomasyonIstasyonNo, DozajlamaIcinReceteHazirla.BirimId from DozajlamaIcinReceteHazirla(:APartiKayitId, :AReceteId, :APartiAsamalariId, 3)
    where UrunID is not null or OrgatexeTekBasinaAktarilir = 1
    into :UrunId, :UrunAdi, :DozajlamaKumasUrunId, :DozajlamaIplikUrunId, :UrunTipiId, :DozajlamaProgramNo, :SablonDozajlamaProgramNo, :DozajlamaBirimKodu, :Miktar, :Banyo,
         :ReceteMiktar, :ReceteBanyo, :ReceteAsamaId, :ReceteAltAsamaId, :DozajlamaSekli, :SiraNo, :UretimeCikisFiyatiBugun, :DozajlamaSekliIplik, :OrgatexeTekBasinaAktarilir, :OtomasyonIstasyonNo, :BirimId do
  begin
    if (EgemenFirmaID = 16) then /*Tekboy*/
    begin
      if (FLabReceteGrupID = 1) then DozajlamaUrunId = DozajlamaIplikUrunId; /*IplikBoya*/
      else DozajlamaUrunId = DozajlamaKumasUrunId;
    end
    else DozajlamaUrunId = DozajlamaKumasUrunId;

    if (OrgatexeTekBasinaAktarilir = 1) then  ReceteAltAsamaId = -1 ;

    if (ReceteAsamaId <> LastReceteAsamaID or SiraNo <> LastSiraNo) then
    begin
      LastReceteAsamaId = ReceteAsamaId;
      LastReceteAltAsamaId = -2;
      if (UrunId is Not Null) then IstekNo = IstekNo + 1;
      if (ReceteAltAsamaId <> LastReceteAltAsamaId) then LastReceteAltAsamaId = ReceteAltAsamaId;
      Counter = 0;
      LastSiraNo = SiraNo;
    end

    if (ReceteAltAsamaId <> LastReceteAltAsamaId) then
    begin
      LastReceteAltAsamaId = ReceteAltAsamaId;
      if (UrunId is Not Null) then IstekNo = IstekNo + 1;
      Counter = 0;
    end

    if (UrunId is Not Null) then Counter = Counter + 1;

    if (Miktar > 0) then OtomasyonMiktar = Miktar;
    else OtomasyonMiktar = Banyo;

    if (ReceteMiktar > 0) then OtomasyonReceteMiktar = ReceteMiktar;
    else OtomasyonReceteMiktar = ReceteBanyo;


    suspend;
  end
end
