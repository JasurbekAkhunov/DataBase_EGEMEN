CREATE PROCEDURE DISTAMIRLISTESIDETAYI(
    ABASLANGICBEYANTARIHI TIMESTAMP,
    ABITISBEYANTARIHI TIMESTAMP,
    ARAPORTIPI INTEGER,
    AFIRMAID VARCHAR(20),
    AHAMTIPI INTEGER,
    AHAMID INTEGER,
    APARTIGRUPID VARCHAR(4),
    ATAMIRDUZELTMETIPIID VARCHAR(10),
    ATAMIRTIPIID INTEGER,
    ATAMIRTIPIGRUPID VARCHAR(4),
    ADISTAMIRCOZUMSEKLIID VARCHAR(10),
    AISLEMEALINACAK INTEGER,
    AHAMSTOKMASTERID INTEGER,
    ASATISTIPI INTEGER)
RETURNS (
    HAMSTOKMASTERID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    PARTIGRUPID VARCHAR(4),
    PARTISINIFID INTEGER,
    HAMID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    BOYASIPARISDETAYID INTEGER,
    DURUMU INTEGER,
    DURUMUSTR VARCHAR(20),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    BRUTSEVKKILO NUMERIC(12,3),
    NETSEVKKILO NUMERIC(12,3),
    SIPARISENISTENEN VARCHAR(10),
    SIPARISGRAMAJISTENEN VARCHAR(10),
    SIPARISDONMEISTENEN VARCHAR(10),
    SIPARISCEKMEZLIKENISTENEN DOUBLE PRECISION,
    SIPARISCEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    ESKIPARTIENVERILEN VARCHAR(10),
    ESKIPARTIGRAMAJVERILEN VARCHAR(10),
    ESKIPARTIDONMEVERILEN VARCHAR(10),
    ESKIPARTICEKMEZLIKENVERILEN DOUBLE PRECISION,
    ESKIPARTICEKMEZLIKBOYVERILEN DOUBLE PRECISION,
    ESKIPARTIRKONTROLCMCDEGERI DOUBLE PRECISION,
    YENIPARTIENVERILEN VARCHAR(10),
    YENIPARTIGRAMAJVERILEN VARCHAR(10),
    YENIPARTIDONMEVERILEN VARCHAR(10),
    YENIPARTICEKMEZLIKENVERILEN DOUBLE PRECISION,
    YENIPARTICEKMEZLIKBOYVERILEN DOUBLE PRECISION,
    YENIPARTIRKONTROLCMCDEGERI DOUBLE PRECISION,
    IRSALIYENO VARCHAR(255),
    IRSALIYETARIHI VARCHAR(255),
    FATURANO VARCHAR(255),
    FATURATARIHI VARCHAR(255))
AS
declare variable FHamStokMasterId             integer;
declare variable FPartiKayitId                integer;
declare variable FPartiSinifId                integer;
declare variable FPartiYil                    integer;
declare variable FPartiId                     integer;
declare variable FPartiGrupId                 varchar(4);
declare variable FHamID                       integer;
declare variable FHamAdiToplami               varchar(255);
declare variable FDurumuStr                   varchar(20);
--declare variable FKapAdedi                    integer; --> Kullanılmamış.
declare variable FKilo                        Numeric(12,3);
declare variable FNetKilo                     Numeric(12,3);
declare variable FMetre                       Numeric(12,3);
declare variable FNetMetre                    Numeric(12,3);
declare variable FBrutSevkKilo                Numeric(12,3);
declare variable FNetSevkKilo                 Numeric(12,3);

declare variable FIrsaliyeNo                  IrsaliyeNoSize;
declare variable FFaturaTarihi              TimeStamp;

declare variable FFaturaNo                    IrsaliyeNoSize;
declare variable FIrsaliyeTarihi              TimeStamp;

declare variable FBoyaSiparisDetayId          integer;

declare variable FReceteId                    varchar(20);
declare variable FReceteAdi                   varchar(40);

declare variable LastHamStokMasterId          integer;
declare variable LastPartiSinifId             integer;
declare variable Str                          varchar(30);
declare variable FSiparisEnIstenen            varchar(10);
declare variable FSiparisGramajIstenen        varchar(10);
declare variable FSiparisDonmeIstenen         varchar(10);
declare variable FSiparisCekmezlikEnIstenen   double precision;
declare variable FSiparisCekmezlikBoyIstenen  double precision;
Declare variable FEskiPartiSinifID            integer;
Declare variable FEskiPartiYil                integer;
Declare variable FEskiPartiNo                 integer;
Declare variable EskiPartiYil                 integer;
Declare variable EskiPartiNo                  integer;
begin
  LastPartiSinifId = -1;
  LastHamStokMasterId = -1;

  for select distinct DisTamirListesi.HamStokMasterId, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.ReceteId, Recete.Adi, Parti.PartiGrupId, PartiStok.PartiSinifId,
          PartiStok.BoyaSiparisDetayId, PartiStok.HamId, PartiStok.HamAdiToplami, PartiStok.Durumu, PartiDurum.Adi, PartiStok.Kilo, PartiStok.Metre, PartiStok.NetKilo, PartiStok.NetMetre,
          PartiStok.BrutSevkKilo, PartiStok.NetSevkKilo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi,
          BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.DonmeIstenen,
          BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen,
          DisTamirListesi.EskiPartiYil, DisTamirListesi.EskiPartiNo, BoyaFatura.FaturaNo, BoyaFatura.FaturaTarihi from DisTamirListesi(:ABaslangicBeyanTarihi, :ABitisBeyanTarihi, :ARaporTipi, :AFirmaId, :AHamTipi, :AHamId, :APartiGrupId, :ATamirDuzeltmeTipiId, :ATamirTipiId, :ATamirTipiGrupId, :ADisTamirCozumSekliID, :AIslemeAlinacak, :AHamStokMasterId, :ASatisTipi)
    left join HamStokDetay on HamStokDetay.HamStokMasterId = DisTamirListesi.HamStokMasterId
    left join RafDetay on RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId
    left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
    left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayID
    left join Parti on Parti.PartiKayitId= PartiStok.PartiKayitId
    left join Recete on Recete.ReceteId= Parti.ReceteId
    left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
    left Join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiKayitId = PartiStok.PartiKayitId and BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
    left Join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    left join BoyaFatura on BoyaFatura.boyafaturaid = BoyaIrsaliyeDetay.boyafaturaid
    where PartiStok.PartiSinifId is not null and PartiStok.partiStokIptalKayitId is null
    into :FHamStokMasterId, :FPartiKayitId, :FPartiYil, :FPartiId, :FReceteId, :FReceteAdi, :FPartiGrupId, :FPartiSinifId, :FBoyaSiparisDetayId, :FHamId, :FHamAdiToplami, :Durumu, :FDurumuStr,
         :FKilo, :FMetre, :FNetKilo, :FNetMetre, :FBrutSevkKilo, :FNetSevkKilo, :FIrsaliyeNo, :FIrsaliyeTarihi,
         :FSiparisEnIstenen, :FSiparisGramajIstenen, :FSiparisDonmeIstenen, :FSiparisCekmezlikEnIstenen, :FSiparisCekmezlikBoyIstenen,
         :FEskiPartiYil, :FEskiPartiNo, :FFaturaNo, :FFaturaTarihi do
  begin
    if (LastHamStokMasterId <> FHamStokMasterId or LastPartiSinifId <> FPartiSinifId) then
    begin
      if (LastPartiSinifId <> -1) then
      begin
        select Min(PartiStok.PartiSinifId) from Parti
          left join PartiStok on PartiStok.PartiKayitID = Parti.PartiKayitID
          where Parti.PartiYil = :EskiPartiYil and Parti.PartiId = :EskiPartiNo and partiStok.HamID = :HamId
          into :FEskiPartiSinifID;

        if (FEskiPartiSinifID is not null) then
        begin
          Select EnVerilen, GramajVerilen, DonmeVerilen, CekmezlikEnVerilen, CekmezlikBoyVerilen, RKontrolCMCDegeri from PartiStokFizikKontrol where PartiStokFizikKontrol.PartiSinifID = :FEskiPartiSinifID and PartiStokFizikKontrol.SonKayit = 1
             into :EskiPartiEnVerilen, :EskiPartiGramajVerilen, :EskiPartiDonmeVerilen, :EskiPartiCekmezlikEnVerilen, :EskiPartiCekmezlikBoyVerilen, :EskiPartiRKontrolCMCDegeri;
        end

        if (FPartiSinifId is not null) then
        begin
          Select EnVerilen, GramajVerilen, DonmeVerilen, CekmezlikEnVerilen, CekmezlikBoyVerilen, RKontrolCMCDegeri from PartiStokFizikKontrol where PartiStokFizikKontrol.PartiSinifID = :FPartiSinifId and PartiStokFizikKontrol.SonKayit = 1
             into :YeniPartiEnVerilen, :YeniPartiGramajVerilen, :YeniPartiDonmeVerilen, :YeniPartiCekmezlikEnVerilen, :YeniPartiCekmezlikBoyVerilen, :YeniPartiRKontrolCMCDegeri;
        end

        suspend;
      end
      LastHamStokMasterId         = FHamStokMasterId;
      LastPartiSinifId            = FPartiSinifId;
      HamStokMasterId             = FHamStokMasterId;
      PartiKayitId                = FPartiKayitId;
      PartiYil                    = FPartiYil;
      PartiId                     = FPartiId;
      PartiGrupId                 = FPartiGrupId;
      PartiSinifId                = FPartiSinifId;
      BoyaSiparisDetayId          = FBoyaSiparisDetayId;
      HamId                       = FHamId;
      HamAdiToplami               = FHamAdiToplami;
      DurumuStr                   = FDurumuStr;
      Kilo                        = FKilo;
      NetKilo                     = FNetKilo;
      Metre                       = FMetre;
      NetMetre                    = FNetMetre;
      BrutSevkKilo                = FBrutSevkKilo;
      NetSevkKilo                 = FNetSevkKilo;
      ReceteId                    = FReceteId;
      ReceteAdi                   = FReceteAdi;
      IrsaliyeNo                  = '';
      IrsaliyeTarihi              = '';
      FaturaNo                    = '';
      FaturaTarihi                = '';
      SiparisEnIstenen            = FSiparisEnIstenen;
      SiparisGramajIstenen        = FSiparisGramajIstenen;
      SiparisDonmeIstenen         = FSiparisDonmeIstenen;
      SiparisCekmezlikEnIstenen   = FSiparisCekmezlikEnIstenen;
      SiparisCekmezlikBoyIstenen  = FSiparisCekmezlikBoyIstenen;
      PartiIdStr                  = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      EskiPartiYil                = FEskiPartiYil;
      EskiPartiNo                 = FEskiPartiNo;
      EskiPartiEnVerilen          = NUll;
      EskiPartiGramajVerilen      = NUll;
      EskiPartiDonmeVerilen       = NUll;
      EskiPartiCekmezlikEnVerilen = NUll;
      EskiPartiCekmezlikBoyVerilen= NUll;
      EskiPartiRKontrolCMCDegeri  = NUll;
    end

    if (FIrsaliyeNo is not null) then
    begin
      Str = cast(FIrsaliyeNo as varchar(30) );

      if (Str <> '') then
      if (PosBul(IrsaliyeNo, Str) = 0) then
      if (Char_Length(IrsaliyeNo) + Char_Length(Str) < 253) then
      begin
        if (IrsaliyeNo <> '') then IrsaliyeNo = left((IrsaliyeNo ||  cast(', ' as varchar(2) character set UTF8)), 255);
        IrsaliyeNo = Left(IrsaliyeNo || Str, 255);

        Str = TimeStampToDateStr(FIrsaliyeTarihi);

        if (Str <> '') then
        if (Char_Length(IrsaliyeTarihi) + Char_Length(Str) < 253) then
        begin
          if (IrsaliyeTarihi <> '') then IrsaliyeTarihi = left((IrsaliyeTarihi ||  cast(', ' as varchar(2) character set UTF8)), 255);
          IrsaliyeTarihi = left(IrsaliyeTarihi || Str, 255);
        end
      end
    end
    if (FFaturaNo is not null) then
    begin
      Str = cast(FFaturaNo as varchar(30) );

      if (Str <> '') then
      if (PosBul(FaturaNo, Str) = 0) then
      if (Char_Length(FaturaNo) + Char_Length(Str) < 253) then
      begin
        if (FaturaNo <> '') then FaturaNo = left((FaturaNo ||  cast(', ' as varchar(2) character set UTF8)), 255);
        FaturaNo = Left(FaturaNo || Str, 255);

        Str = TimeStampToDateStr(FFaturaTarihi);

        if (Str <> '') then
        if (Char_Length(FaturaTarihi) + Char_Length(Str) < 253) then
        begin
          if (FaturaTarihi <> '') then FaturaTarihi = left((FaturaTarihi ||  cast(', ' as varchar(2) character set UTF8)), 255);
          FaturaTarihi = left(FaturaTarihi || Str, 255);
        end
      end
    end
  end
  if (LastPartiSinifId <> -1) then
  begin
    select Min(PartiStok.PartiSinifId) from Parti
      left join PartiStok on PartiStok.PartiKayitID = Parti.PartiKayitID
      where Parti.PartiYil = :EskiPartiYil and Parti.PartiId = :EskiPartiNo and partiStok.HamID = :HamId
      into :FEskiPartiSinifID;

    if (FEskiPartiSinifID is not null) then
    begin
      Select EnVerilen, GramajVerilen, DonmeVerilen, CekmezlikEnVerilen, CekmezlikBoyVerilen, RKontrolCMCDegeri from PartiStokFizikKontrol where PartiStokFizikKontrol.PartiSinifID = :FEskiPartiSinifID and PartiStokFizikKontrol.SonKayit = 1
         into :EskiPartiEnVerilen, :EskiPartiGramajVerilen, :EskiPartiDonmeVerilen, :EskiPartiCekmezlikEnVerilen, :EskiPartiCekmezlikBoyVerilen, :EskiPartiRKontrolCMCDegeri;
    end
    suspend;
  end
end
