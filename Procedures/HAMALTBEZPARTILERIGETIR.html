CREATE PROCEDURE HAMALTBEZPARTILERIGETIR(
    AHAMID INTEGER,
    AFIRMAID FIRMAKODUSIZE,
    ABOYASIPARISDETAYID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMADITOPLAMI VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    KILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    KALANNETMETRE NUMERIC(12,3),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    SIPARISBIRIMFIYAT DOUBLE PRECISION,
    SIPARISDOVIZID VARCHAR(5),
    HESAPLANANFIYAT DOUBLE PRECISION,
    HESAPLANANUCRETLITAMIRFIYATI DOUBLE PRECISION,
    HESAPLANANUCRETSIZTAMIRFIYATI DOUBLE PRECISION,
    HESAPACIKLAMA VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(10),
    ISLEMTIPIID INTEGER,
    ISLEMGRUBUID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    SIPARISTIPIID INTEGER,
    SONSEVKTARIHI TIMESTAMP,
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    YAPILANISLEMLER VARCHAR(1024),
    FIYATONAYLAYANKULLANICIID INTEGER,
    FIYATONAYLAYANKULLANICIADI VARCHAR(20),
    FIYATONAYTARIHI TIMESTAMP)
AS
declare variable AsamaKisaAdi varchar(10);
declare variable IlaveIslemlerAdi varchar(30);
declare variable FirmaKodumuz   FirmaKoduSize;
begin
  select Sistem.FirmaKodumuz from Sistem where Sistem.SistemId = 1 into :FirmaKodumuz;

  for select PartiStok.PartiKayitId, PartiStok.PartiSinifId, PartiStok.BoyaSiparisDetayId, Parti.PartiYil, Parti.PartiId, Parti.ReceteId,
             Recete.Adi, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, PartiStok.HamAdiToplami, Parti.FirmaId, Firma.Adi,
             PartiStok.Kilo, PartiStok.NetKilo, PartiStok.Metre, PartiStok.NetMetre,
             Parti.Fason, Parti.IslemTipiId, IslemTipi.Adi, Parti.IslemGrubuId, IslemGrubu.Adi,
             PartiStok.BirimFiyat, PartiStok.DovizId, BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.SiparisTipiId,
             PartiStok.HesaplananFiyat, PartiStok.HesaplananUcretliTamirFiyati, PartiStok.HesaplananUCretsizTamirFiyati,
             PartiStok.SevkTarihi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, PartiStok.NetSevkMiktar + PartiStok.NetSarfMiktar + PartiStok.RezervMiktar from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = partiStok.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    left join BoyaSiparisDetayAltBez on BoyaSiparisDetayAltBez.BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisDetayAltBez.PartiSinifId = PartiStok.PartiSinifId
    where PartiStok.HamId = :AHamId and
          (Parti.FirmaId = :AFirmaId or Parti.FirmaId = :FirmaKodumuz) and
          PartiStok.Durumu < 8 and
          PartiStok.PartiStokIptalKayitId is null and
          PartiStok.NetSevkMiktar + PartiStok.NetSarfMiktar + PartiStok.RezervMiktar < PartiStok.NetMiktar and
          BoyaSiparisDetayAltBez.BoyaSiparisDetayAltBezId is null
    order by Parti.PartiYil desc, Parti.PartiId desc, PartiStok.PartiSinifId desc
    into :PartiKayitId, :PartiSinifId, :BoyaSiparisDetayId, :PartiYil, :PartiId, :ReceteId,
         :ReceteAdi, :DesenId, :DesenAdi, :VaryantAdi, :HamAdiToplami, :FirmaId, :FirmaAdi, :Kilo, :NetKilo, :Metre, :NetMetre, :Fason, :IslemTipiId, :IslemTipiAdi,
         :IslemGrubuId, :IslemGrubuAdi, :BirimFiyat, :DovizId, :SiparisBirimFiyat, :SiparisDovizId, :BoyaSiparisYil, :BoyaSiparisId,
         :SiparisTipiId, :HesaplananFiyat, :HesaplananUcretliTamirFiyati, :HesaplananUCretsizTamirFiyati, :SonSevkTarihi,
         :SiparisProsesId, :SiparisProsesAdi, :KalanNetMetre do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (HesaplananFiyat = -1) then
    begin
      HesapAciklama = 'Fiyat HesaplanamadÄ±..';
      HesaplananFiyat               = null;
      HesaplananUcretliTamirFiyati  = null;
      HesaplananUCretsizTamirFiyati = null;
    end

    if (HesaplananFiyat = 0) then HesaplananFiyat = null;
    if (HesaplananUcretliTamirFiyati = 0) then HesaplananUcretliTamirFiyati = null;
    if (HesaplananUCretsizTamirFiyati = 0) then HesaplananUCretsizTamirFiyati = null;


    YapilanIslemler = '';
    for select Coalesce(Asama.Asamakisaadi, '') from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join Recete on Recete.Receteid = PartiAsamalari.Receteid
      left join Asama on Asama.Asamaid = PartiAsamaStok.Asamaid
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1 and PartiAsamalari.SerbestAsama = 0 and Asama.Asamakisaadi is not null and Asama.Asamakisaadi <> '' and
            Asama.ParasalDegeriVar = 1
      order by PartiAsamalari.Sira into :AsamaKisaAdi do
    begin
      if (Char_Length(YapilanIslemler) + Char_Length(AsamaKisaAdi) < 1000) then
      begin
       if (YapilanIslemler <> '' ) then YapilanIslemler = YapilanIslemler || cast(', ' as varchar(2) character set UTF8);
       YapilanIslemler = YapilanIslemler || AsamaKisaAdi;
      end
    end

    for Select IlaveIslemler.Adi from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1 and PartiAsamalari.SerbestAsama = 0 and IlaveIslemler.Adi is not null
      into :IlaveIslemlerAdi do
    begin
      if (Char_Length(YapilanIslemler) + Char_Length(IlaveIslemlerAdi) < 1000) then
      begin
       if (YapilanIslemler <> '' ) then YapilanIslemler = YapilanIslemler || cast(', ' as varchar(2) character set UTF8);
       YapilanIslemler = YapilanIslemler || IlaveIslemlerAdi;
      end
    end

    suspend;
  end
end
