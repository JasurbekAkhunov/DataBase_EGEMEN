            (DesenIstekTipi.YapacakDepartmanId = :ADepartmanId or :ADepartmanId is null) and
            (DesenIstekTipi.DesenIstekGrupId = :ADesenIstekGrupId or :ADesenIstekGrupId is null) and
            (DesenIstekleri.IstekYapanKullaniciId = :AIstekYapanKullaniciId or :AIstekYapanKullaniciId is null)
      into :DesenIstekleriId, :DesenId, :DesenAdi, :FirmaId, :FirmaAdi,
           :DesenIstekTipiId, :DesenIstekTipiAdi, :DesenIstekGrupId, :IstekTarihi, :TerminTarihi, :IstekYapanKullaniciId, :IstekYapanKullaniciAdi,
           :BaslatanKullaniciAdi, :BaslamaTarihi, :BitirenKullaniciAdi, :BitisTarihi, :DepartmanId, :DepartmanAdi, :OnayDurumu,
           :OnayTarihi, :OnaylayanKullaniciAdi, :PlanlananDesenIstasyonId, :PlanlananDesenIstasyonAdi, :PlanlananHaftaKayitId,
           :PlanlananHaftaYil, :PlanlananHaftaNo, :HaftaBaslangicTarih, :HaftaBitisTarih,
           :FOrjKagitKolaj, :FOrjDijitalGoruntu, :FOrjDijitalSepare, :FOrjGiydirme, :FOrjBitmisUrun, :FOrjDiger, :SablonSayisi,
           :LaserGravur, :MaxExposer, :TasarimNo, :DesenGravurBoyu, :DesenSablonBoyu, :DesenSablonEni, :DesenIstekleriNotu do
      begin
        Okey = 1;

        if (ADesenIstekTipiAralikli = 0) then
        if (KutuKarsilastir(ABaslangicDesenIstekTipiId, DesenIstekTipiId) = 0) then okey = 0;

        if (Okey = 1) then
        begin
          if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
          else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
          else OnayDurumuStr = '';

          if (PlanlananHaftaNo is not null) then PlanlananHaftaNoStr = '(' || EgemenIntegerToStr(PlanlananHaftaYil) || ') '  || EgemenIntegerToStr(PlanlananHaftaNo);
          else PlanlananHaftaNoStr = '';

          OrjinalBicimi = '';

          if (FOrjKagitKolaj = 1) then
          begin
            if (OrjinalBicimi <> '') then OrjinalBicimi = OrjinalBicimi || cast(', ' as varchar(2) character set UTF8);
            OrjinalBicimi = 'Kağıt Kolaj';
          end
          if (FOrjDijitalGoruntu = 1) then
          begin
            if (OrjinalBicimi <> '') then OrjinalBicimi = OrjinalBicimi ||  cast(', ' as varchar(2) character set UTF8);
            OrjinalBicimi = OrjinalBicimi || 'Dijital Görüntü';
          end
          if (FOrjDijitalSepare = 1) then
          begin
            if (OrjinalBicimi <> '') then OrjinalBicimi = OrjinalBicimi || cast(', ' as varchar(2) character set UTF8);
            OrjinalBicimi = OrjinalBicimi || 'Dijital Separe';
          end
          if (FOrjGiydirme = 1) then
          begin
            if (OrjinalBicimi <> '') then OrjinalBicimi = OrjinalBicimi || cast(', ' as varchar(2) character set UTF8);
            OrjinalBicimi = OrjinalBicimi || 'Giydirme';
          end
          if (FOrjBitmisUrun = 1) then
          begin
            if (OrjinalBicimi <> '') then OrjinalBicimi = OrjinalBicimi || cast(', ' as varchar(2) character set UTF8);
            OrjinalBicimi = OrjinalBicimi || 'Bitmiş Ürün';
          end
          if (FOrjDiger = 1) then
          begin
            if (OrjinalBicimi <> '') then OrjinalBicimi = OrjinalBicimi || cast(', ' as varchar(2) character set UTF8);
            OrjinalBicimi = OrjinalBicimi || 'Diğer';
          end

          suspend;
        end
      end
    end
  end
