CREATE PROCEDURE IPLIKSARIMPLANLAMAYAZ(
    AIPLIKSARIMPLANLAMAGRUPID VARCHAR(10),
    AIPLIKMAKINAID VARCHAR(10))
RETURNS (
    SEPETNO INTEGER,
    SARIMISEMRIKAYITID INTEGER,
    SARIMISEMRIYIL INTEGER,
    SARIMISEMRIID INTEGER,
    SARIMSEKLIID INTEGER,
    IPLIKSARIMPLANLAMAGRUPID VARCHAR(10),
    PLANLAMATARIHI TIMESTAMP,
    IPLIKMAKINAID VARCHAR(10),
    IPLIKMAKINAADI VARCHAR(40),
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(30),
    MELANJKODU VARCHAR(20),
    SIRA INTEGER,
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION)
AS
declare variable KS integer;
declare variable LastIplikMakinaId varchar(10);
begin
  if (AIplikSarimPlanlamaGrupId = '') then AIplikSarimPlanlamaGrupId = Null;

  LastIplikMakinaId = '@@@';

  if (AIplikSarimPlanlamaGrupId is null) then
  begin
    for Select IplikSarimPlanlamaGrupDetayi.IplikSarimPlanlamaGrupId, SarimIsEmri.SarimIsEmriKayitId, SarimIsEmri.SarimIsEmriYil, SarimIsEmri.SarimIsEmriId,
               SarimIsEmri.SarimSekliId, SarimIsEmri.PlanlamaTarihi,
               SarimIsEmri.IplikMakinaId, IplikMakina.Adi, Ham.Adi, IplikLot.Lot, IplikLot.MelanjKodu,
               IplikSatisIade.RenkId, Renk.Adi, SarimIsEmri.Miktar, SarimIsEmri.Sira from SarimIsEmri
      left join IplikSarimPlanlamaGrupDetayi on IplikSarimPlanlamaGrupDetayi.IplikMakinaId = SarimIsEmri.IplikMakinaId
      left join IplikLot on IplikLot.IplikLotID = SarimIsEmri.IplikLotId
      left join Ham on Ham.HamID = IplikLot.HamId
      left join TamirGiris on TamirGiris.TamirGirisKayitID = IplikLot.TamirGirisKayitId
      left join IplikSatisIade on IplikSatisIade.IplikSatisIadeID = TamirGiris.IplikSatisIadeId
      left join Renk on Renk.RenkID = IplikSatisIade.RenkId
      left join IplikMakina on IplikMakina.IplikMakinaID = SarimIsEmri.IplikMakinaId
      where (SarimIsEmri.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null)  and SarimIsEmri.BitisTarihi is null
      order by SarimIsEmri.IplikMakinaId, SarimIsEmri.Sira
      into :IplikSarimPlanlamaGrupId, :SarimIsEmriKayitId, :SarimIsEmriYil, :SarimIsEmriId, :SarimSekliId,
           :PlanlamaTarihi, :IplikMakinaId, :IplikMakinaAdi, :HamAdi, :Lot, :MelanjKodu,
           :RenkId, :RenkAdi, :PlanlananMiktar, :Sira do
    begin
      if (LastIplikMakinaId <> IplikMakinaId) then
      begin
        SepetNO = 0;
        LastIplikMakinaId = IplikMakinaId;
      end


      GerceklesenMiktar = 0;

      Select Sum(IplikUretim.Miktar) from IplikUretim
        where IplikUretim.SarimIsEmriKayitId = :SarimIsEmriKayitId
        into :GerceklesenMiktar;

      SepetNO = SepetNo + 1;
      suspend;
    end
  end
  else
  begin
    for Select IplikSarimPlanlamaGrupDetayi.IplikSarimPlanlamaGrupId, SarimIsEmri.SarimIsEmriKayitId, SarimIsEmri.SarimIsEmriYil, SarimIsEmri.SarimIsEmriId,
               SarimIsEmri.SarimSekliId, SarimIsEmri.PlanlamaTarihi,
               SarimIsEmri.IplikMakinaId, IplikMakina.Adi, Ham.Adi, IplikLot.Lot, IplikLot.MelanjKodu,
               IplikSatisIade.RenkId, Renk.Adi, SarimIsEmri.Miktar, SarimIsEmri.Sira from IplikSarimPlanlamaGrupDetayi
      left join SarimIsEmri on SarimIsEmri.IplikMakinaId = IplikSarimPlanlamaGrupDetayi.IplikMakinaId
      left join IplikLot on IplikLot.IplikLotID = SarimIsEmri.IplikLotId
      left join Ham on Ham.HamID = IplikLot.HamId
      left join TamirGiris on TamirGiris.TamirGirisKayitID = IplikLot.TamirGirisKayitId
      left join IplikSatisIade on IplikSatisIade.IplikSatisIadeID = TamirGiris.IplikSatisIadeId
      left join Renk on Renk.RenkID = IplikSatisIade.RenkId
      left join IplikMakina on IplikMakina.IplikMakinaID = SarimIsEmri.IplikMakinaId
      where IplikSarimPlanlamaGrupDetayi.IplikSarimPlanlamaGrupId = :AIplikSarimPlanlamaGrupId and
           (IplikSarimPlanlamaGrupDetayi.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null)  and SarimIsEmri.BitisTarihi is null
      order by SarimIsEmri.IplikMakinaId, SarimIsEmri.Sira
      into :IplikSarimPlanlamaGrupId, :SarimIsEmriKayitId, :SarimIsEmriYil, :SarimIsEmriId,
           :SarimSekliId, :PlanlamaTarihi, :IplikMakinaId, :IplikMakinaAdi, :HamAdi, :Lot, :MelanjKodu,
           :RenkId, :RenkAdi, :PlanlananMiktar, :Sira do
    begin
      if (LastIplikMakinaId <> IplikMakinaId) then
      begin
        SepetNO = 0;
        LastIplikMakinaId = IplikMakinaId;
      end


      GerceklesenMiktar = 0;

      Select Sum(IplikUretim.Miktar) from IplikUretim
        where IplikUretim.SarimIsEmriKayitId = :SarimIsEmriKayitId
        into :GerceklesenMiktar;


      SepetNO = SepetNo + 1;
      suspend;
    end
  end
end
