CREATE PROCEDURE MODELAKSESUARDETAYGETIR(
    AMODELKAYITID INTEGER)
RETURNS (
    MODELAKSESUARID INTEGER,
    MODELAKSESUARDETAYID INTEGER,
    BEDENAKTIF EGEBOOLEAN,
    MODELBEDENID INTEGER,
    KULLANILACAKBEDENID INTEGER,
    KULLANILACAKBEDENKODU VARCHAR(7),
    KULLANILACAKBEDENADI VARCHAR(30),
    RENKAKTIF EGEBOOLEAN,
    MODELKUMASID INTEGER,
    KULLANILACAKRENKID INTEGER,
    KULLANILACAKRENKADI VARCHAR(50),
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40))
AS
begin

  For Select ModelAksesuarDetay.ModelAksesuarID, ModelAksesuarDetay.ModelAksesuarDetayId, ModelAksesuarDetay.BedenAktif, ModelAksesuarDetay.ModelBedenId, ModelAksesuarDetay.KullanilacakBedenID, Beden.KisaKodu, Beden.Adi,
             ModelAksesuarDetay.RenkAktif, ModelAksesuarDetay.ModelKumasID, ModelAksesuarDetay.KullanilacakRenkID, Renk.Adi,
             ModelAksesuarDetay.InsertKullaniciId, ModelAksesuarDetay.InsertTarihi, ModelAksesuarDetay.KullaniciId, ModelAksesuarDetay.DegisimTarihi,
             IK.Adi, SK.Adi From ModelAksesuar
    left join ModelAksesuarDetay on ModelAksesuarDetay.ModelAksesuarID = ModelAksesuar.ModelAksesuarID
    left join Renk on Renk.RenkId = ModelAksesuarDetay.KullanilacakRenkID
    left join Beden on Beden.BedenId = ModelAksesuarDetay.KullanilacakBedenID
    left join Kullanici IK on IK.KullaniciId  = ModelAksesuarDetay.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = ModelAksesuarDetay.KullaniciId
    Where ModelAksesuar.ModelKayitID = :AModelKayitID and ModelAksesuarDetay.ModelAksesuarDetayID is not Null
    into :ModelAksesuarID, :ModelAksesuarDetayId, :BedenAktif, :ModelBedenId, :KullanilacakBedenID, :KullanilacakBedenKodu, :KullanilacakBedenAdi,
         :RenkAktif, :ModelKumasID, :KullanilacakRenkID, :KullanilacakRenkAdi,
         :InsertKullaniciId, :InsertTarihi, :KullaniciId, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi do
  begin
    Suspend;
  end
end
