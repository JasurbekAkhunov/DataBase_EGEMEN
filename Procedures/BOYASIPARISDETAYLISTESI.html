CREATE PROCEDURE BOYASIPARISDETAYLISTESI(
    ABOYASIPARISKAYITID INTEGER,
    AHAMID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    HAMADI VARCHAR(255),
    SIPARISKILO NUMERIC(12,3),
    SIPARISBRUTKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    MUSTERISIPARISNO VARCHAR(50),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISNO VARCHAR(20),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5))
AS
declare variable SatSiparisKalanBruttenHesapla integer;
declare variable SiparisMaximumFireOrani       Double Precision;
declare variable FFason                        integer;
begin
  Select Fason from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitID into :FFAson;

  Select SatSiparisKalanBruttenHesapla from sistem where sistemid = 1 into :SatSiparisKalanBruttenHesapla;

  for select BoyaSiparisDetay.BoyaSiparisDetayId, Ham.Adi, BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo,
             BoyaSiparisDetay.MusteriKaliteAdi, BoyaSiparisDetay.MusteriSiparisNo,
             LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr, BoyaSiparisDetay.OnayDurumu, BoyaSiparisDetay.MaximumFireOrani,
             BoyaSiparisDetay.SiparisBrutKilo from BoyaSiparisDetay
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
  where BoyaSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisDetay.HamID = :AHamId and BoyaSiparisDetay.BoyaSiparisDetayId is not null
  order by BoyaSiparisDetay.BoyaSiparisDetayId
  into :BoyaSiparisDetayId, :HamAdi, :SiparisKilo, :SiparisMetre, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi, :MusteriSiparisNo, :LabReceteKodu,
       :LabReceteAdi, :LabReceteAtisNo, :OnayDurumu, :SiparisMaximumFireOrani, :SiparisBrutKilo do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    suspend;
  end
end
