CREATE PROCEDURE SAATEGOREMAKINACALISMARAPORU1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAGRUBU VARCHAR(4),
    AMAKINAID VARCHAR(10),
    ASAAT1 TIMESTAMP,
    ASAAT2 TIMESTAMP,
    ASAAT3 TIMESTAMP)
RETURNS (
    TARIH TIMESTAMP,
    MAKINAID VARCHAR(10),
    MAKINAGRUBU VARCHAR(4),
    CALISMASURESI1 DOUBLE PRECISION,
    CALISMASURESI2 DOUBLE PRECISION,
    CALISMASURESI3 DOUBLE PRECISION)
AS
declare variable AraVermeSuresi Double Precision;
declare variable GirisTarihi TimeStamp;
declare variable CikisTarihi TimeStamp;
declare variable FSaatDilim1   TimeStamp;
declare variable FSaatDilim2   TimeStamp;
declare variable FSaatDilim3   TimeStamp;
declare variable Dakika5       Double Precision;
declare variable FTarih   TimeStamp;
declare variable D1 Double Precision;
declare variable D2 Double Precision;
declare variable FCalismaSuresi1 Double Precision;
declare variable FCalismaSuresi2 Double Precision;
declare variable FCalismaSuresi3 Double Precision;
begin
  D1 = 5;
  D2 = 1440;
  Dakika5 = D1/D2;


  for Select Makina.MakinaId, Makina.MakinaGrubu from Makina
    where (Makina.MakinaId = :AMakinaId or :AMakinaId is null) and
          Makina.UretimRaporlarinaDahil = 1 and (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null)
    order by Makina.MakinaId
    into :MakinaId, :MakinaGrubu do
  begin
    FTarih = f_StripTime(ABaslangicTarihi)-1;

    While (FTarih <= ABitisTarihi) do
    begin
      CalismaSuresi1        = 0;
      CalismaSuresi2        = 0;
      CalismaSuresi3        = 0;
      FTarih = FTarih + 1;
      Tarih = FTarih;
      Suspend;
    end

    Tarih = f_StripTime(ABaslangicTarihi)-20;

    for Select Min(PartiAsamaStokDetay.GirisTarihi), Max(PartiAsamaStokDetay.CikisTarihi), Min(PartiAsamaStokDetay.AraVermeSuresi) from PartiAsamaStokDetay
      where PartiAsamaStokDetay.CikisTarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.CikisTarihi < :ABitisTarihi + 1 and
            PartiAsamaStokDetay.MakinaId = :MakinaId
      group by PartiAsamaStokDetay.MakinaSarjId
      order by 1
      into :GirisTarihi, :CikisTarihi, :AraVermeSuresi do
    begin
      CalismaSuresi1        = 0;
      CalismaSuresi2        = 0;
      CalismaSuresi3        = 0;

      GirisTarihi = GirisTarihi + AraVermeSuresi;

      While (GirisTarihi < CikisTarihi) do
      begin

        FTarih = f_StripTime(GirisTarihi);

        if (FTarih <> Tarih) then
        begin
          if (Tarih >= ABaslangicTarihi -1) then suspend;
          CalismaSuresi1        = 0;
          CalismaSuresi2        = 0;
          CalismaSuresi3        = 0;
          Tarih = FTarih;
        end

        FSaatDilim1 = Tarih + TimeStampToDouble(ASaat1);
        FSaatDilim2 = Tarih + TimeStampToDouble(ASaat2);
        FSaatDilim3 = Tarih + TimeStampToDouble(ASaat3);

        if (GirisTarihi >= FSaatDilim1 and GirisTarihi < FSaatDilim2) then CalismaSuresi1 = CalismaSuresi1 + Dakika5;
        else if (GirisTarihi >= FSaatDilim2 and GirisTarihi < FSaatDilim3) then CalismaSuresi2 = CalismaSuresi2 + Dakika5;
        else CalismaSuresi3 = CalismaSuresi3 + Dakika5;

        GirisTarihi = GirisTarihi + Dakika5;
      end

      suspend;
    end
  end
end
