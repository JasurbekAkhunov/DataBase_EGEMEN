CREATE PROCEDURE PARTIAARAKONTROLBILGISIGETIR(
    APARTISINIFID INTEGER)
RETURNS (
    PARTIASAMALARIID INTEGER,
    ASAMAADI VARCHAR(40),
    ASAMAKISAADI VARCHAR(10),
    ARAKONTROLTIPIKISAADI VARCHAR(11),
    ARAKONTROLTIPIADI VARCHAR(30),
    SONUC VARCHAR(50))
AS
begin
  For Select PartiAsamaAraKontrol.PartiAsamalariId, Asama.Adi, Asama.AsamaKisaAdi,
             AraKontrolTipi.Adi, AraKontrolTipi.KisaAdi, PartiAsamaAraKontrol.Sonuc from PartiAsamaAraKontrol
  Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaAraKontrol.PartiAsamalariId
  Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaAraKontrol.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaAraKontrol.PartiSinifId
  Left Join AraKontrolTipi on AraKontrolTipi.AraKontrolTipiId = PartiAsamaAraKontrol.AraKontrolTipiId
  Left Join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
  where PartiAsamaAraKontrol.PartiSinifId = :APartiSinifId and PartiAsamaStok.GirisTarihi is not null and PartiAsamaAraKontrol.Sonuc is not null
  order by PartiAsamaAraKontrol.PartiSinifId,  PartiAsamalari.Sira, PartiAsamaAraKontrol.AraKontrolTipiId
  into :PartiAsamalariId, :AsamaAdi, :AsamaKisaAdi, :AraKontrolTipiAdi, :AraKontrolTipiKisaAdi, :Sonuc do
  begin

    suspend;
  end
end
