CREATE PROCEDURE MAKINAPARTIVERIRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAID MAKINAKODUSIZE)
RETURNS (
    PARTISINIFID INTEGER,
    MAKINABILGIID INTEGER,
    MAKINAID MAKINAKODUSIZE,
    TARIH TIMESTAMP,
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    ASAMAID INTEGER,
    ASAMAADI ADI40,
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    URETIMSURESI TIMESTAMP,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    IMALATTIPIID INTEGER,
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    DESENVARYANTID INTEGER,
    VARYANTADI VARCHAR(30),
    DESENKODUADIVARYANT VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MIKTAR NUMERIC(12,3),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISTIPIID INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30))
AS
begin
  For select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.ImalatTipiId, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId,
         Firma.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.SiparisTipiId, Parti.ReceteId, Recete.Adi,
         Parti.BoyaSiparisKayitId, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.DesenVaryantId, DesenVaryant.VaryantAdi,
         BoyaSiparis.SiparisGrupID, SiparisGrup.Adi, Firma.Adi2,
         MakinaBilgi.MakinaBilgiID, MakinaBilgi.MakinaID, MakinaBilgi.Tarih, PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.CikisTarihi,
         PartiAsamalari.AsamaID, Asama.Adi AsamaAdi, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen,
         DoubletoTimeStamp(PartiAsamaStokDetay.CikisTarihi - PartiAsamaStokDetay.GirisTarihi) UretimSuresi, PartiStok.PartiSinifID, PartiStok.HamID, Ham.Adi, PartiStok.Miktar
          from MakinaBilgi
    left Join PartiAsamaStokDetay on PartiAsamaStokDetay.MakinaID = MakinaBilgi.MakinaID and PartiAsamaStokDetay.GirisTarihi <= MakinaBilgi.Tarih and PartiAsamaStokDetay.CikisTarihi >= MakinaBilgi.Tarih
    Left Join PartiStok on PartiStok.PartiSinifID = PartiAsamaStokDetay.PartiSinifID
    Left Join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
    Left Join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayID
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStokDetay.PartiAsamalariID
    Left Join Asama on Asama.AsamaID = PartiAsamalari.AsamaID
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    Left Join Ham on Ham.HamID = PartiStok.HamID
    where PartiAsamaStokDetay.PartiAsamaStokDetayId is not Null and
          MakinaBilgi.Tarih >= :ABaslangicTarihi and MakinaBilgi.Tarih <= :ABitisTarihi and
          (MakinaBilgi.MakinaID = :AMakinaID or :AMakinaID is Null)
    into :PartiKayitId, :PartiYil, :PartiId, :ImalatTipiId, :PartigrupId, :PartiGrupAdi, :FirmaId,
         :FirmaAdi, :BoyaSiparisYil, :BoyaSiparisId, :SiparisTipiId, :ReceteId, :ReceteAdi,
         :BoyaSiparisKayitId, :DesenId, :DesenAdi, :DesenVaryantId, :VaryantAdi,
         :SiparisGrupId, :SiparisGrupAdi, :FirmaAdi2, :MakinaBilgiID, :MakinaID, :Tarih, :GirisTarihi, :CikisTarihi,
         :AsamaID, :AsamaAdi, :EnIstenen, :GramajIstenen, :UretimSuresi, :PartiSinifID, :HamID, :HamAdi, :Miktar do
  begin

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    DesenKoduAdiVaryant = '(' || DesenId || ') ' || DesenAdi || ' - ' || VaryantAdi;
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';


    suspend;
  end
end
