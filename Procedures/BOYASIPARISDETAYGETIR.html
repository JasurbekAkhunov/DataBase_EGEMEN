           sum(PartiStokTarti.Kilo + PartiStokTarti.Dara)
      from PartiStok
      left join PartiStokTarti on PartiStokTarti.PartiSinifId = PartiStok.PartiSinifId
      where PartiStok.BoyaSiparisDetayId = :BoyaSiparisDetayId
       into :SevkeHazirTartiBrutKilo, :SevkEdilenTartiBrutKilo, :TartiBrutKilo;

    HamSevkBrutKilo = null;
    HamSevkNetKilo = null;

    select sum(BoyaIrsaliyeDetay.BrutKilo), sum(BoyaIrsaliyeDetay.Kilo) from BoyaIrsaliyeDetay
      /*Sebebini anlayamadk: left join RafDetay on RafDetay.RafDetayId = BoyaIrsaliyeDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId*/
      where BoyaIrsaliyeDetay.RafDetayId is not null and BoyaIrsaliyeDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
        into :HamSevkBrutKilo, :HamSevkNetKilo;

    select sum(Miktar) from StokHareket
      where StokHareket.BoyaSiparisDetayId = :BoyaSiparisDetayId and BoyaSiparisKayitId  = :BoyaSiparisKayitId and
      (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis) and StokHareket.UretimIslemTipiId = :uitBoya
       PLAN (STOKHAREKET INDEX (FK_SHOSIPARISKAYITID, STOKHAREKET_IDX_URETIMISLEMTIPI, STOKHAREKET_IDX_HAREKETTIPI, STOKHAREKET_IDX_HAREKETTIPI))
    into :OrgudenBoyaUretimeCikis;

    select sum(BoyaRezervFisiDetay.EdilecekKilo), sum(BoyaRezervFisiDetay.EdilenKilo), sum(BoyaRezervFisiDetay.EdilecekBrutMiktar), sum(BoyaRezervFisiDetay.EdilenBrutMiktar) from BoyaRezervFisiDetay where BoyaRezervFisiDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
    into RezervEdilecekMiktar, :RezervEdilenMiktar, :RezervEdilecekBrutMiktar, :RezervEdilenBrutMiktar;

    select Sum(Miktar), Sum(NetGelenMiktar), Sum(BrutGelenMiktar), Sum(GidenMiktar) from BoyaTalimat where BoyaSiparisDetayId = :BoyaSiparisDetayId into
     :TalimatBoyaMiktar, :TalimatBoyaGelenNetMiktar, :TalimatBoyaGelenBrutMiktar, :TalimatBoyaGidenMiktar;

    select Sum(Miktar), Sum(BrutMiktar) from StokHareket where StokHareket.BoyaSiparisDetayId = :BoyaSiparisDetayId and StokHareket.HareketTipi = :shtSatis and StokHareket.HamIade = 0 and  BoyaSiparisKayitId  = :BoyaSiparisKayitId
    PLAN (STOKHAREKET INDEX (FK_SHOSIPARISKAYITID, STOKHAREKET_IDX_HAREKETTIPI))
	into :OrmeSevkEdilenNetMiktar, :OrmeSevkEdilenBrutMiktar;



    if (PartilenenMiktardanIptaliDus = 1) then
    begin
      PartilenenKilo = PartilenenKilo - IptalKilo;
      PartilenenMetre = PartilenenMetre - IptalMetre;
      PartilenenKiloBaski = PartilenenKiloBaski - IptalKiloBaski;
      PartilenenMetreBaski = PartilenenMetreBaski - IptalMetreBaski;

    end
    suspend;
  end
