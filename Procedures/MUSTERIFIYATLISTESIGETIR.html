CREATE PROCEDURE MUSTERIFIYATLISTESIGETIR(
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    AFASON INTEGER)
RETURNS (
    MUSTERIFIYATLISTESIID INTEGER,
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI VARCHAR(20),
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    BIRIMFIYAT DOUBLE PRECISION,
    VADE INTEGER,
    DOVIZID VARCHAR(5))
AS
declare variable ElyafCinsiId varchar(5);
begin
  if (Afason = 1) then
  begin
    Select IplikCinsi.ElyafCinsiId from Ham
      left join IplikCinsi on IplikCinsi.IplikCinsiID = Ham.IplikCinsiId
      where Ham.HamID = :AHamId
      into :ElyafCinsiId;

    For Select MusteriFiyatListesi.MusteriFiyatListesiId, MusteriFiyatListesi.RenkDerinligiId, RenkDerinligi.Adi, MusteriFiyatListesi.IslemGrubuId, IslemGrubu.Adi,
               MusteriFiyatListesi.BirimFiyat, MusteriFiyatListesi.DovizId, MusteriFiyatListesi.Vade from MusteriFiyatListesi
      left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = MusteriFiyatListesi.RenkDerinligiId
      left join IslemGrubu on IslemGrubu.IslemGrubuId = MusteriFiyatListesi.IslemGrubuId
      where MusteriFiyatListesi.FirmaID = :AFirmaID and MusteriFiyatListesi.Fason = :AFason and
         (MusteriFiyatListesi.ElyafCinsiId = :ElyafCinsiId)
      into :MusteriFiyatListesiId, :RenkDerinligiId, :RenkDerinligiAdi, :IslemGrubuId, :IslemGrubuAdi,
           :BirimFiyat, :DovizId, :Vade do suspend;
  end
  else
  begin
    For Select MusteriFiyatListesi.MusteriFiyatListesiId, MusteriFiyatListesi.RenkDerinligiId, RenkDerinligi.Adi, MusteriFiyatListesi.IslemGrubuId, IslemGrubu.Adi,
               MusteriFiyatListesi.BirimFiyat, MusteriFiyatListesi.DovizId, MusteriFiyatListesi.Vade from MusteriFiyatListesi
      left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = MusteriFiyatListesi.RenkDerinligiId
      left join IslemGrubu on IslemGrubu.IslemGrubuId = MusteriFiyatListesi.IslemGrubuId
      where MusteriFiyatListesi.FirmaID = :AFirmaID and MusteriFiyatListesi.Fason = :AFason and
         (MusteriFiyatListesi.HamID = :AHamId or :AHamId is null)
      into :MusteriFiyatListesiId, :RenkDerinligiId, :RenkDerinligiAdi, :IslemGrubuId, :IslemGrubuAdi,
           :BirimFiyat, :DovizId, :Vade do suspend;
  end
end
