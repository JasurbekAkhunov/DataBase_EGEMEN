CREATE PROCEDURE PARTININRECETESINISTOKTANDUS(
    APARTIKAYITID INTEGER,
    APARTIASAMALARIID INTEGER,
    ARECETEID VARCHAR(20),
    ABANYOCARPANI DOUBLE PRECISION,
    ABANYOCARPANI2 DOUBLE PRECISION,
    ABANYOCARPANI3 DOUBLE PRECISION,
    ABANYOCARPANI4 DOUBLE PRECISION,
    AKOYULTMA DOUBLE PRECISION,
    AKULLANICIID INTEGER,
    ADOZAJLAMASISTEMIID INTEGER)
AS
declare variable OrjPartiKilo           Double Precision;
declare variable OrjPartiMetre          Double Precision;
declare variable FPartiKilo             Double Precision;
declare variable FUrunId                varchar(20);
declare variable FUrunBirimId           integer;
declare variable FUrunMutfakDepoNo      varchar(10);
declare variable FBirimFiyat            Double Precision;
declare variable FUrunDovizId           varchar(5);
declare variable FStokTabanDovizId      varchar(5);
declare variable FStokId                integer;
declare variable KayitBasladi           integer;
declare variable FHareketTipi           integer;
declare variable FFisSeriNo             varchar(2);
declare variable FFisNo                 integer;
declare Variable FMiktar                Double Precision;
declare Variable FBanyo                 Double Precision;
declare variable FT1                    Double Precision;
declare variable FTarih                 TimeStamp;
declare variable FReceteMiktarOrani     Double Precision;
declare variable FUrunBirimCarpan Double Precision;
declare variable FSonAlisBirimCarpan    Double Precision;
declare variable FDovizKuru             Double Precision;
declare variable FNow date;
declare variable PartiSinif  varchar(1);
declare variable FKimyasalTipiReceteOrani Double Precision;
Declare variable FReceteAsamaId           integer;
Declare variable FBoyaAsamasimi           integer;
Declare variable FFBoyaAsamasimi          integer;
Declare variable FBanyoCarpanNo           integer;
Declare variable FFBanyoCarpanNo          integer;

declare variable FToplamKilo            Double Precision;
declare variable FKilo                  Double Precision;
declare variable FPartiKayitId          integer;
declare variable FPartiAsamalariId      integer;
declare variable LabReceteKoduTipi      integer;
declare variable ReceteArsiviOlusturulsun integer;
declare variable OtomasyonaDahilmi        integer;
declare variable KayitSayisi              integer;
declare variable SarfiyatlariTeoriktenDus integer;
declare variable ORJBanyoCarpani          Double Precision;
declare variable UstAsamaBoyaAsamasimi    integer;
declare variable BoyaSarfiyatiManuelTartidanDus integer;
declare variable DozajlamaIsEmriId       integer;
declare variable UrunTipiId              integer;
declare variable FReferansStokDetayId     integer;
declare variable ManuelTartiVar           integer;
declare variable FStokGrupId            varchar(4);
declare variable FAsamaMutfakDepoNo      varchar(10);
declare variable FGirisTarihi           TimeStamp;
declare variable FCikisTarihi           TimeStamp;
declare variable ReceteStoktanDusmeSekli integer;
declare variable RecManuelDusumdeTarihGirisTar integer;
declare variable KiloSifirOlsadaReceteyiDus integer;
declare variable FLastBanyoCarpanNo   integer;
declare variable SiraNo               integer;
declare variable FLabReceteKoduTipi   integer;
declare variable FMetre               Double Precision;
declare variable FPartiMetre          Double Precision;
declare variable FToplamMetre         Double Precision;
declare variable FLot                 varchar(150) ;
declare variable myMiktar             Double Precision;
declare variable FLastReceteAsamaId   integer;
declare variable FLabReceteGrupId     integer;
declare variable BoyaKimyasalLotTakibiFifo EgeBoolean;
declare variable FKalanMiktar           double precision;
declare variable FCikilanMiktar         double precision;
declare variable FCikilacakMiktar       double precision;
declare variable GirisYok               EgeBoolean;
declare variable FToplamLot             varchar(150);
declare variable FAsamaBoyaTipiId       integer;
declare variable LastAsamaBoyaTipiId   integer;
declare variable MiktarCarpani Double Precision;
declare variable TartimYapilacak  integer;
declare variable LrgPadPatch SmallInt;
declare variable EgemenFirmaID Integer;
begin
  select StokTabanDovizId, ReceteMiktarOrani, ReceteArsiviOlusturulsun, BoyaSarfiyatiManuelTartidanDus, ReceteStoktanDusmeSekli, RecManuelDusumdeTarihGirisTar, KiloSifirOlsadaReceteyiDus, LabReceteKoduTipi, BoyaKimyasalLotTakibiFifo, EgemenFirmaID from SISTEM where SISTEMID = 1
  into :FStokTabanDovizId, :FReceteMiktarOrani, :ReceteArsiviOlusturulsun, :BoyaSarfiyatiManuelTartidanDus, :ReceteStoktanDusmeSekli, :RecManuelDusumdeTarihGirisTar, :KiloSifirOlsadaReceteyiDus, :FLabReceteKoduTipi, :BoyaKimyasalLotTakibiFifo, :EgemenFirmaID;

  select PartiAsamalari.StokGrupId from PartiAsamalari
    where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId
    into :FStokGrupId;

  if (FStokGrupId is Null) then
  begin
    Select PartiGrup.StokGrupID from Parti
      left join PartiGrup on PartiGrup.PartiGrupID = Parti.PartiGrupId
      where Parti.PartiKayitID = :APartiKayitID
      into :FStokGrupId;
  end

  FHareketTipi = 6;  /* Uretime Cikis */
  FNow = current_TimeStamp(0);
  FTarih =  FNow;

  if (ADozajlamaSistemiID is not null) then
    Select SarfiyatlariTeoriktenDus from DozajlamaSistemi where DozajlamaSistemi.DozajlamaSistemiID = :ADozajlamaSistemiID into :SarfiyatlariTeoriktenDus;
  else
    SarfiyatlariTeoriktenDus = 1;

  if (SarfiyatlariTeoriktenDus = 1) then ADozajlamaSistemiID = -1;

  if (AReceteId is null) then exception AsamadaReceteYok;

  FToplamKilo = 0;
  FToplamMetre = 0 ;

  Select Asama.BoyaAsamasimi, Recete.LabReceteGrupId from Recete
    left join Asama on Asama.AsamaId = Recete.UstAsamaId
    where Recete.ReceteID = :AReceteID
  into :UstAsamaBoyaAsamasimi, :FLabReceteGrupId;

  for select PartiKayitId, PartiAsamalariId  from PartiRecetesiDokumu_1(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    into :FPartiKayitId, :FPartiAsamalariId do
  begin
    /*Fis tarihine asama bitisini koymasi icin FTarih'e son cikisin tarihini aldik*/
    select sum(partiasamastok.GirisKilo), sum(partiasamastok.GirisMetre) ,Max(PartiAsamaStok.GirisTarihi), Max(PartiAsamaStok.CikisTarihi)  from PartiAsamalari
      left join partiasamastok on partiasamastok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      where PartiAsamalari.PartiKayitId = :FPartiKayitId and PartiAsamalari.PartiAsamalariId = :FPartiAsamalariId and PartiAsamalari.ReceteId = :AReceteId and PartiAsamaStok.AsamayaDahil = 1
      into :FKilo, :FMetre , :FGirisTarihi, :FCikisTarihi;

    if (APartiKayitId = FPartiKayitId) then
    begin
      FPartiKilo = FKilo;
      FPartiMetre = FMetre;
    end
    FToplamKilo  = FToplamKilo + FKilo;
    FToplamMetre = FToplamMetre + FMetre;

    if ((ReceteStoktanDusmeSekli = 2 or (ReceteStoktanDusmeSekli = 0 and RecManuelDusumdeTarihGirisTar = 1)) and FGirisTarihi is not Null) then FTarih = FGirisTarihi; /*2-İmalat Başlayınca*/
    else if (ReceteStoktanDusmeSekli = 3 and FCikisTarihi is not Null) then FTarih = FCikisTarihi; /*3-İmalat Bitince*/
  end

  if ((FPartiKilo is null or FPartiKilo = 0 or FToplamKilo is null or FToplamKilo = 0) and (KiloSifirOlsadaReceteyiDus = 0)) then exit;

  OrjPartiKilo = FPartiKilo;
  OrjPartiMetre = FPartiMetre;

  if (ABanyoCarpani is null or ABanyoCarpani = 0) then
  if (UstAsamaBoyaAsamasimi = 1) then exception BanyoCarpaniSifirOlamaz; /* NY icin */
  else ABanyoCarpani = 0;

  if (FLabReceteKoduTipi = 27) then /* AriTeksde Bu Kontrol Yapılacak*/
  begin
    FLastBanyoCarpanNo = -1;
    SiraNo = 0;

    for select ReceteAsama.BanyoCarpanNo from ReceteDetay
      left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
      where ReceteDetay.receteid = :AReceteId and ReceteDetay.ReceteAsamaId is not null Order By Sira
      into :FFBanyoCarpanNo do
    begin

      if (FLastBanyoCarpanNo <> FFBanyoCarpanNo and FFBanyoCarpanNo is not Null) then
      begin
        FLastBanyoCarpanNo = FFBanyoCarpanNo;
        SiraNo = SiraNo + 1;

        if (SiraNo = 1 and UstAsamaBoyaAsamasimi = 1 and (ABanyoCarpani is null or ABanyoCarpani = 0) ) then exception BanyoCarpaniSifirOlamaz;
        else if (SiraNo = 2 and (ABanyoCarpani2 is null or ABanyoCarpani2 = 0) ) then exception BanyoCarpaniSifirOlamaz;
        else if (SiraNo = 3 and (ABanyoCarpani3 is null or ABanyoCarpani3 = 0) ) then exception BanyoCarpaniSifirOlamaz;
        else if (SiraNo = 4 and (ABanyoCarpani4 is null or ABanyoCarpani4 = 0) ) then exception BanyoCarpaniSifirOlamaz;
      end
    end
  end
  else
  begin
    for select ReceteAsama.BanyoCarpanNo from ReceteDetay
      left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
      where ReceteDetay.receteid = :AReceteId and ReceteDetay.ReceteAsamaId is not null
      group by ReceteAsama.BanyoCarpanNo
      into :FFBanyoCarpanNo do
    begin
      if (FFBanyoCarpanNo = 1 and UstAsamaBoyaAsamasimi = 1 and (ABanyoCarpani is null or ABanyoCarpani = 0) ) then exception BanyoCarpaniSifirOlamaz;
      else if (FFBanyoCarpanNo = 2 and (ABanyoCarpani2 is null or ABanyoCarpani2 = 0) ) then exception BanyoCarpaniSifirOlamaz;
      else if (FFBanyoCarpanNo = 3 and (ABanyoCarpani3 is null or ABanyoCarpani3 = 0) ) then exception BanyoCarpaniSifirOlamaz;
      else if (FFBanyoCarpanNo = 4 and (ABanyoCarpani4 is null or ABanyoCarpani4 = 0) ) then exception BanyoCarpaniSifirOlamaz;
    end
  end

  if (AKoyultma is null or AKoyultma = 0) then AKoyultma = 1;

  if (FReceteMiktarOrani is null or FReceteMiktarOrani = 0) then exception RecetemiktarOraniGirilmemis;

  if (EgemenFirmaId <> 137) then
    if (UstAsamaBoyaAsamasimi = 0) then FReceteMiktarOrani = 1;  /* ömer ekledi */

  if (FPartiKilo > 0 and FToplamKilo > 0) then
  begin
    if (ABanyoCarpani <> 0) then ABanyoCarpani = ABanyoCarpani / FToplamKilo * FPartiKilo;
    else ABanyoCarpani = FPartiKilo; /* ram, merseriz gibi işlemler icin */
    if (ABanyoCarpani2 <> 0) then ABanyoCarpani2 = ABanyoCarpani2 / FToplamKilo * FPartiKilo;
    if (ABanyoCarpani3 <> 0) then ABanyoCarpani3 = ABanyoCarpani3 / FToplamKilo * FPartiKilo;
    if (ABanyoCarpani4 <> 0) then ABanyoCarpani4 = ABanyoCarpani4 / FToplamKilo * FPartiKilo;
    FPartiKilo = FPartiKilo / FReceteMiktarOrani;
  end
  else if (FPartiMetre > 0 and FToplamMetre > 0) then
  begin
    if (ABanyoCarpani <> 0) then ABanyoCarpani = ABanyoCarpani / FToplamMetre * FPartiMetre;
    else ABanyoCarpani = FPartiMetre; /* ram, merseriz gibi işlemler icin */
    if (ABanyoCarpani2 <> 0) then ABanyoCarpani2 = ABanyoCarpani2 / FToplamMetre * FPartiMetre;
    if (ABanyoCarpani3 <> 0) then ABanyoCarpani3 = ABanyoCarpani3 / FToplamMetre * FPartiMetre;
    if (ABanyoCarpani4 <> 0) then ABanyoCarpani4 = ABanyoCarpani4 / FToplamMetre * FPartiMetre;
    FPartiMetre = FPartiMetre / FReceteMiktarOrani;
  end

  ABanyoCarpani = Yuvarla(ABanyoCarpani, 1);
  ABanyoCarpani2 = Yuvarla(ABanyoCarpani2, 1);
  ABanyoCarpani3 = Yuvarla(ABanyoCarpani3, 1);
  ABanyoCarpani4 = Yuvarla(ABanyoCarpani4, 1);

  Select Parti.Sinif from Parti where Parti.PartikayitId = :APartiKayitId into :PartiSinif;

  KayitBasladi = 0;

  Select FisSeriNo from Kullanici where KullaniciId = :AKullaniciId into :FFisSeriNo;

  if (FFisSeriNo is null) then FFisSeriNo = '';


  FFBanyoCarpanNo = -1;
  ORJBanyoCarpani = ABanyoCarpani;
  SiraNo = 0;

  if (BoyaSarfiyatiManuelTartidanDus = 1) then
  begin
    Select DozajlamaIsEmriId from DozajlamaIsEmriGetir(:APartiAsamalariId) into :DozajlamaIsEmriId;

    select count(*) from DozajlamaIsEmriTartiDetay
      where DozajlamaIsEmriTartiDetay.DozajlamaIsEmriid = :DozajlamaIsEmriId
      into :ManuelTartiVar;

    if (FLabReceteKoduTipi = 4 and ManuelTartiVar = 0) then ManuelTartiVar = 1;
  end
  else ManuelTartiVar = 0;

  TartimYapilacak = Null;
  if (EgemenFirmaID = 110) /*or EgemenFirmaID = 4*/ then /*Ayyıldızda, Gülle sadece TartimYapilacak işareti olanlar düşecek(İptal)*/
  begin
    /*Select LrgPadPatch from LabReceteGrupGetir into :LrgPadPatch;
    if (FLabReceteGrupId <> LrgPadPatch and EgemenFirmaID = 4) then TartimYapilacak = 1;
    else if (EgemenFirmaID <> 4) then */ /*Gülle İptal etti*/
    TartimYapilacak = 1;
  end

  select Max(Makina.MutfakDepoId)  /*Patlamamasi garanti olsun diye max koyuldu*/
  from PartiAsamaStokDetay
  left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
  where PartiAsamaStokDetay.PartiAsamalariId = :APartiAsamalariId into :FAsamaMutfakDepoNo;

  if (FAsamaMutfakDepoNo is null) then
    select PartiAsamalari.MutfakDepoNo from PartiAsamalari
        where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId
        into :FAsamaMutfakDepoNo;

  for select ReceteDetay.ReceteAsamaId, Urun.UrunId, ReceteDetay.UrunBirimId, Urun.MutfakDepoNo,
             UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / RUB.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId,
             ReceteDetay.Miktar , ReceteDetay.Banyo, RUB.Carpan, KimyasalTipi.ReceteOrani, ReceteAsama.BoyaAsamasimi,
             ReceteAsama.BanyoCarpanNo, Urun.UrunTipiId, Recete.LabReceteGrupId, ReceteAsama.BoyaTipiId from ReceteDetay
    left join Recete on Recete.ReceteId = ReceteDetay.ReceteId
    left join UrunBirim RUB on RUB.urunBirimId = ReceteDetay.UrunBirimId
    left join Urun on Urun.UrunId = ReceteDetay.UrunId and Urun.EnvantereDahil = 1 and (Urun.TartimYapilacak = :TartimYapilacak or :TartimYapilacak is Null)
    left join UrunFiyat on UrunFiyat.UrunId = ReceteDetay.UrunId
    left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
    left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = Urun.KimyasalTipiId
    left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
    left join DozajlamaSistemiUrun on DozajlamaSistemiUrun.UrunId = ReceteDetay.UrunId and DozajlamaSistemiUrun.DozajlamaSistemiID = :ADozajlamaSistemiID
    where ReceteDetay.Receteid = :AReceteId
    order by ReceteDetay.Sira
    into :FReceteAsamaId, :FUrunId, :FUrunBirimId, :FUrunMutfakDepoNo,  :FBirimFiyat, :FUrunDovizId,
         :FMiktar, :FBanyo, :FUrunBirimCarpan, :FKimyasalTipiReceteOrani, :FBoyaAsamasimi, :FBanyoCarpanNo, :UrunTipiId, :FLabReceteGrupId, :FAsamaBoyaTipiId do
  begin
    /*Eğer aşamada mutfak depo var ise üründeki yerine bunu kullanacak*/
    if (FAsamaMutfakDepoNo is not Null) then FUrunMutfakDepoNo = FAsamaMutfakDepoNo;

    myMiktar = 0;
    if (FReceteAsamaId is not null) then
    begin
      FFBoyaAsamasimi = FBoyaAsamasimi;     /* Boya Asamasimi diye not alıyoruz.. eger boya asamasi ise kimyasal tipi carpani ile  islem yapıyoruz */
      FLastReceteAsamaId = FReceteAsamaId;
      LastAsamaBoyaTipiId = FAsamaBoyaTipiId;
      if (FLabReceteKoduTipi = 27) then /* AriTeksde */
      begin
        if (FFBanyoCarpanNo <> FBanyoCarpanNo and FBanyoCarpanNo is not Null) then
        begin
          SiraNo = SiraNo + 1;
          FFBanyoCarpanNo = FBanyoCarpanNo;

          if (SiraNo = 2) then ORJBanyoCarpani = ABanyoCarpani2;
          else if (SiraNo = 3) then ORJBanyoCarpani = ABanyoCarpani3;
          else if (SiraNo = 4) then ORJBanyoCarpani = ABanyoCarpani4;
          else ORJBanyoCarpani = ABanyoCarpani;
        end
      end
      else
      begin
        FFBanyoCarpanNo = FBanyoCarpanNo;

        if (FFBanyoCarpanNo = 2) then ORJBanyoCarpani = ABanyoCarpani2;
        else if (FFBanyoCarpanNo = 3) then ORJBanyoCarpani = ABanyoCarpani3;
        else if (FFBanyoCarpanNo = 4) then ORJBanyoCarpani = ABanyoCarpani4;
        else ORJBanyoCarpani = ABanyoCarpani;
      end
    end

    if (FUrunId is not null and (ManuelTartiVar = 0 or UrunTipiId <> 0)) then
    if ((FBanyo is not null and FBanyo > 0) or (FMiktar is not null and FMiktar > 0)) then
    begin
      if (KayitBasladi = 0) then
      begin
        Select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;

        FStokId = gen_id(Stok_Id_Gen, 1);

        insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, Sinif, KullaniciId)
                  values(:FStokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :FTarih, :FStokTabanDovizId, 1, 1, :PartiSinif, :AKullaniciId);

        KayitBasladi = 1;
      end


      Select (:FMiktar * Miktar * :AKoyultma / :FReceteMiktarOrani) From PARTIASAMALARIRECETEBOYAMIKTAR where (ReceteAsamaId = :FLastReceteAsamaId ) and Partiasamalariid = :APartiAsamalariID into :myMiktar;

      FMiktar = FMiktar * FPartiKilo * AKoyultma;

      if (LastAsamaBoyaTipiId <> 0 and FLabReceteKoduTipi = 59 and UrunTipiID <> 0) then myMiktar = myMiktar / AKoyultma;

      if (myMiktar > 0) then
      begin
        FMiktar = MyMiktar;
        if (LastAsamaBoyaTipiId = 0) then
        begin
          Select MiktarCarpani from BoyaMiktarCarpanHesapla(:APartiKayitID, Null, Null) into :MiktarCarpani;

          FMiktar = FMiktar * MiktarCarpani;
        end
      end

      if (FLabReceteGrupId = 0) then FBanyo = FBanyo * OrjBanyoCarpani * :AKoyultma;
      else FBanyo = FBanyo * OrjBanyoCarpani;

      if (FFBoyaAsamasimi = 1) then /* boya asamasi ise kimyasal tipirecete orani devreye giriyor */
      if (FKimyasalTipiReceteOrani is not null) then FBanyo = FBanyo / FKimyasalTipiReceteOrani;

      if (FMiktar is null) then FMiktar = 0;
      if (FBanyo is null) then FBanyo = 0;

      if (FBirimFiyat is null) then FBirimFiyat = 0;

      if (FUrunDovizId is not null and FBirimFiyat > 0) then
      if (FStokTabanDovizId <> FUrunDovizId) then
      begin
        select DovizKuru from EnYakinDovizKuruGetir(:FUrunDovizId, :FStokTabanDovizId, :FTarih) into :FDovizKuru;
        FBirimFiyat = FBirimFiyat * FDovizKuru;
      end

      FLot = Null;
      GirisYok = 1;
      FCikilanMiktar = 0;
      FCikilacakMiktar = 0;
      FToplamLot = '';
      FMiktar = FMiktar + FBanyo;

      if (BoyaKimyasalLotTakibiFifo = 1) then
      begin
        for select UrunFifoBilgisi.Lot, UrunFifoBilgisi.KalanMiktar * (:FUrunBirimCarpan / UrunBirim.Carpan) from UrunFifoBilgisi(:FUrunId, :FAsamaMutfakDepoNo)
            left join UrunBirim on UrunBirim.UrunBirimID = UrunFifoBilgisi.UrunBirimId
            order by GirisTarihi into :FLot, FKalanMiktar do
        begin
          GirisYok = 0;

          if (FKalanMiktar < FMiktar - FCikilanMiktar) then
            FCikilacakMiktar = FKalanMiktar;
          else
            FCikilacakMiktar = FMiktar - FCikilanMiktar;

          insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, StokGrupId, UrunId, UrunBirimId, Miktar, BirimFiyat, PartiKayitId, PartiAsamalariId, Otomasyon, KullaniciId, UretimeCikisTarihi, Lot)
                  values(:FTarih, :FHareketTipi, :FStokId, :FUrunMutfakDepoNo, :FStokGrupId, :FUrunId, :FUrunBirimId, :FCikilacakMiktar, :FBirimFiyat, :APartiKayitId, :APartiAsamalariId, 1, :AKullaniciId, :FTarih, :FLot);


          FCikilanMiktar = FCikilanMiktar + FCikilacakMiktar;

          if (FToplamLot = '') then FToplamLot = FLot;
          else if (Char_Length(FToplamLot||'+'||Flot) < 129) then FToplamLot = FToplamLot||'+'||Flot;

          if (FCikilanMiktar = FMiktar) then break;
        end
      end

      FMiktar = FMiktar - FCikilanMiktar;
      if (FMiktar > 0 or GirisYok = 1) then
          insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, StokGrupId, UrunId, UrunBirimId, Miktar, BirimFiyat, PartiKayitId, PartiAsamalariId, Otomasyon, KullaniciId, UretimeCikisTarihi, Lot)
                  values(:FTarih, :FHareketTipi, :FStokId, :FUrunMutfakDepoNo, :FStokGrupId, :FUrunId, :FUrunBirimId, :FMiktar, :FBirimFiyat, :APartiKayitId, :APartiAsamalariId, 1, :AKullaniciId, :FTarih, '');
    end
  end
  myMiktar = 0;

  if (ManuelTartiVar > 0) then
  begin
    for select DozajlamaIsEmriTarti.UrunId, DozajlamaIsEmriTarti.UrunBirimId, Urun.MutfakDepoNo,
               UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / RUB.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId,
               DozajlamaIsEmriTartiDetay.ReferansStokDetayId, DozajlamaIsEmriTartiDetay.Lot,
               Sum(DozajlamaIsEmriTartiDetay.Miktar) from DozajlamaIsEmriTartiDetay
      left join DozajlamaIsEmriTarti on DozajlamaIsEmriTarti.DozajlamaIsEmriTartiID = DozajlamaIsEmriTartiDetay.DozajlamaIsEmriTartiId
      left join UrunBirim RUB on RUB.urunBirimId = DozajlamaIsEmriTarti.urunBirimId
      left join Urun on Urun.UrunId = DozajlamaIsEmriTarti.UrunId
      left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
      left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
      where DozajlamaIsEmriTartiDetay.DozajlamaIsEmriid = :DozajlamaIsEmriId and DozajlamaIsEmriTarti.Ilave = 0
      group by DozajlamaIsEmriTarti.UrunId, DozajlamaIsEmriTarti.UrunBirimId, Urun.MutfakDepoNo,
               UrunFiyat.UretimeCikisFiyati, UBUCF.Carpan, RUB.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId,
               DozajlamaIsEmriTartiDetay.ReferansStokDetayId, DozajlamaIsEmriTartiDetay.Lot
      into :FUrunId, :FUrunBirimId, :FUrunMutfakDepoNo, :FBirimFiyat, :FUrunDovizId, :FReferansStokDetayId, :FLot, :FMiktar do
    begin
      /*Eğer aşamada mutfak depo var ise üründeki yerine bunu kullanacak*/
      if (FAsamaMutfakDepoNo is not Null) then FUrunMutfakDepoNo = FAsamaMutfakDepoNo;

      if (KayitBasladi = 0) then
      begin
        Select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;

        FStokId = gen_id(Stok_Id_Gen, 1);

        insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, Sinif, KullaniciId)
                    values(:FStokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :FTarih, :FStokTabanDovizId, 1, 1, :PartiSinif, :AKullaniciId);

        KayitBasladi = 1;
      end

      FMiktar = OrjPartiKilo * FMiktar / FToplamKilo;

      if (FBirimFiyat is null) then FBirimFiyat = 0;

      if (FUrunDovizId is not null and FBirimFiyat > 0) then
      if (FStokTabanDovizId <> FUrunDovizId) then
      begin
        select DovizKuru from EnYakinDovizKuruGetir(:FUrunDovizId, :FStokTabanDovizId, :FTarih) into :FDovizKuru;
        FBirimFiyat = FBirimFiyat * FDovizKuru;
      end

      insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, StokGrupId, UrunId, UrunBirimId, ReferansStokDetayId, Miktar, BirimFiyat, PartiKayitId, PartiAsamalariId, Otomasyon, KullaniciId, UretimeCikisTarihi, Lot)
                  values(:FTarih, :FHareketTipi, :FStokId, :FUrunMutfakDepoNo, :FStokGrupId, :FUrunId, :FUrunBirimId, :FReferansStokDetayId, :FMiktar, :FBirimFiyat, :APartiKayitId, :APartiAsamalariId, 1, :AKullaniciId, :FTarih, :FLot);
    end
  end

  Update DozajlamaIsEmri Set StoktanDusuldu = 1
     where DozajlamaIsEmri.DozajlamaIsEmriId = :DozajlamaIsEmriId;

  Update PartiAsamalari Set StoktanDusuldu = 1
     where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.PartiAsamalariId = :APartiAsamalariId;

  if (ReceteArsiviOlusturulsun = 1) then
  begin
    Select OtomasyonaDahilmi from PartiAsamalari where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId into :OtomasyonaDahilmi;

    if (OtomasyonaDahilmi = 0) then
    begin
      Select count(*) from PartiReceteDetay where PartiReceteDetay.PartiAsamalariId = :APartiAsamalariId into :KayitSayisi;
      if (KayitSayisi = 0) then
        Execute Procedure PartininRecetesiniArsivle(:APartiKayitId, :APartiAsamalariId, :AReceteId);
    end
  end
end
