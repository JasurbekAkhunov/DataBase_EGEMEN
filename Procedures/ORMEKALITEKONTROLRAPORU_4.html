CREATE PROCEDURE ORMEKALITEKONTROLRAPORU_4(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AORMEMAKINAID VARCHAR(10),
    AORMEPLANLAMAID INTEGER,
    ASAKAT INTEGER,
    AOTOMATIKCIKISYAPILAN EGEBOOLEAN,
    AVARDIYATARIHINEGORE INTEGER)
RETURNS (
    ORMEKALITEKONTROLID INTEGER,
    ORMEPLANLAMARECETEID INTEGER,
    ORMEKALITEKONTROLHATALARIID INTEGER,
    HAMHATATIPIID INTEGER,
    ORMEKALITEKONTROLHATAMETREID INTEGER,
    METRE BOYAMIKTARSIZE,
    BITISMETRE BOYAMIKTARSIZE)
AS
begin
  For Select OrmeKaliteKontrolRaporu_3.OrmeKaliteKontrolID, OrmeKaliteKontrolRaporu_3.OrmePlanlamaReceteId, OrmeKaliteKontrolRaporu_3.OrmeKaliteKontrolHatalariId, OrmeKaliteKontrolRaporu_3.HamHataTipiId,
             OrmeKaliteKontrolHataMetre.OrmeKaliteKontrolHataMetreId, OrmeKaliteKontrolHataMetre.Metre, OrmeKaliteKontrolHataMetre.BitisMetre from OrmeKaliteKontrolRaporu_3(:ABaslangicTarihi, :ABitisTarihi, :AOrmeMakinaId, :AOrmePlanlamaId, :ASakat, :AOtomatikCikisYapilan, :AVardiyaTarihineGore)

    where OrmeKaliteKontrolHataMetre.OrmeKaliteKontrolHataMetreId is not null
    order by OrmeKaliteKontrolRaporu_3.OrmeKaliteKontrolID, OrmeKaliteKontrolRaporu_3.OrmePlanlamaReceteId
  into :OrmeKaliteKontrolID, :OrmePlanlamaReceteId, :OrmeKaliteKontrolHatalariId, :HamHataTipiId, :OrmeKaliteKontrolHataMetreId, :Metre, :BitisMetre do suspend;
end
