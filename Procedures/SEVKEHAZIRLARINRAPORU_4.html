CREATE PROCEDURE SEVKEHAZIRLARINRAPORU_4(
    AFIRMAID VARCHAR(20),
    AHAMTIPIID INTEGER,
    ATEMSILCIID INTEGER,
    AFASON INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AHAMID INTEGER,
    AKALITEID VARCHAR(20),
    ALABRECETEID INTEGER,
    AIPLIKCINSIID INTEGER,
    APARTIGRUPID VARCHAR(4),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ADISUNITEDURUMU INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ADESENID VARCHAR(20),
    AUNITEID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISINIFSEVKREZERVID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    KAPADEDI INTEGER,
    MIKTAR NUMERIC(12,3),
    BIRIMFIYAT DOUBLE PRECISION,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    SEVKEDILENKILO NUMERIC(12,3),
    SEVKEDILENNETKILO NUMERIC(12,3),
    SEVKEDILENMETRE NUMERIC(12,3))
AS
declare variable MaxSira  integer;
declare variable SonKontrolAsamaId integer;
declare variable IplikSonKontrolAsamaId integer;
declare variable FMakinaId      varchar(10);
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef       integer;
declare variable FLot                   varchar(30);
declare variable FDevirPartiYil integer;
declare variable FDevirPartiID  integer;
declare variable FPartiNoStr  varchar(30);
declare variable ABoyaSiparisKayitId integer;
declare variable hstOrmeKumas   integer;
declare variable hstRibana      integer;
declare variable hstAksesuar    integer;
declare variable hstDokumaKumas integer;
begin
  Select hstIplik, hstElyaf, hstTelef, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar;

  Select SonKontrolAsamaId, IplikSonKontrolAsamaId from Sistem Where SistemId = 1 into :SonKontrolAsamaId, :IplikSonKontrolAsamaId;

  if (ABoyaSiparisYil is not Null) then
  begin
    Select BoyaSiparis.BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId into :ABoyaSiparisKayitId;
    if (ABoyaSiparisKayitId is null) then ABoyaSiparisKayitId = -1;
  end

  for select PartiStok.PartiKayitId, PartiStok.PartiSinifId from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join FirmaSevkAdresi on FirmaSevkAdresi.FirmaSevkAdresiId = PartiStok.FirmaSevkAdresiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
    left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    left join IplikCinsi On IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join DesenVaryant on DesenVaryant.DesenVaryantID = Parti.DesenVaryantID
    where PartiStok.durumu = 7 and PartiStok.PartiStokIptalKayitId is null and (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
        (PartiStok.HamID = :AHamID or :AHamID is null) and
        (Parti.Fason = :AFason or :AFason is null) and
         ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
           ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 1) or
           (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 2) or
           (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf  and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 3) or :AHamTipiId is null) and
        (Firma.TemsilciID = :ATemsilciId or :ATemsilciId is null) and
        (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
        (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
        (BoyaSiparisDetay.KaliteID = :AKaliteId or :AKaliteId is Null) and
        (BoyaSiparisDetay.LabReceteId = :ALabReceteId or :ALabReceteId is Null) and
        (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null) and
        (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is Null) and
        (IplikCinsi.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is Null) and
        ( (PartiStok.UniteID is null and :ADisUniteDurumu = 0) or
          (PartiStok.UniteID is not null and :ADisUniteDurumu = 1) or :ADisUniteDurumu is null) and
        (parti.PartiGrupId = :APartiGrupId or :APartiGrupId is Null) and
          (BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          (DesenVaryant.DesenId = :ADesenId or :ADesenId is Null ) and
          (PartiStok.UniteId = :AUniteId or :AUniteId is Null )

    into :PartiKayitId, :PartiSinifId do
  begin
    for select PartiSinifSevkRezervId, PartiSinifId, PartiSinifSevkRezerv.FirmaId, KapAdedi, Miktar, BirimFiyat, PartiSinifSevkRezerv.Notu, PartiSinifSevkRezerv.InsertKullaniciId,
        PartiSinifSevkRezerv.InsertTarihi, PartiSinifSevkRezerv.KullaniciId, PartiSinifSevkRezerv.DegisimTarihi, Firma.Adi from PartiSinifSevkRezerv
      left join Firma on Firma.FirmaId = PartiSinifSevkRezerv.FirmaId
      where PartiSinifId = :PartiSinifId
      into :PartiSinifSevkRezervId, :PartiSinifId, :FirmaId, :KapAdedi, :Miktar, :BirimFiyat, :Notu, :InsertKullaniciId,
        :InsertTarihi, :KullaniciId, :DegisimTarihi, :FirmaAdi do
    begin
      select sum(BrutKilo), sum(Kilo), sum(Metre) from BoyaIrsaliyeDetay
        where PartiSinifId = :PartiSinifId and FirmaId = :FirmaId
        into :SevkEdilenKilo, :SevkEdilenNetKilo, :SevkEdilenMetre;
      suspend;
    end
  end
end
