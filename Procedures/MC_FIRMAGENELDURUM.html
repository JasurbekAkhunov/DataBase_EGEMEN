CREATE PROCEDURE MC_FIRMAGENELDURUM(
    AFIRMAID VARCHAR(20))
RETURNS (
    DEPOSTOKIPLIKKG NUMERIC(12,3),
    DEPOSTOKIPLIKTAMIRKG NUMERIC(12,3),
    DEPOSTOKKUMASKG NUMERIC(12,3),
    DEPOSTOKKUMASTAMIRKG NUMERIC(12,3),
    DEPOSTOKDIGERKG NUMERIC(12,3),
    DEPOSTOKDIGERTAMIRKG NUMERIC(12,3),
    ISLETMEDEKIIPLIKKG NUMERIC(12,3),
    ISLETMEDEKIIPLIKUCRETLITAMIRKG NUMERIC(12,3),
    ISLETMEDEKIIPLIKUCRETSIZTAMIRKG NUMERIC(12,3),
    ISLETMEDEKIKUMASKG NUMERIC(12,3),
    ISLETMEDEKIKUMASUCRETLITAMIRKG NUMERIC(12,3),
    ISLETMEDEKIKUMASUCRETSIZTAMIRKG NUMERIC(12,3),
    SEVKEHAZIRIPLIKKG NUMERIC(12,3),
    SEVKEHAZIRIPLIKUCRETLITAMIRKG NUMERIC(12,3),
    SEVKEHAZIRIPLIKUCRETSIZTAMIRKG NUMERIC(12,3),
    SEVKEHAZIRKUMASKG NUMERIC(12,3),
    SEVKEHAZIRKUMASUCRETLITAMIRKG NUMERIC(12,3),
    SEVKEHAZIRKUMASUCRETSIZTAMIRKG NUMERIC(12,3),
    SEVKPARTIIPLIKKG NUMERIC(12,3),
    SEVKPARTIIPLIKUCRETLITAMIRKG NUMERIC(12,3),
    SEVKPARTIIPLIKUCRETSIZTAMIRKG NUMERIC(12,3),
    SEVKPARTIKUMASKG NUMERIC(12,3),
    SEVKPARTIKUMASUCRETLITAMIRKG NUMERIC(12,3),
    SEVKPARTIKUMASUCRETSIZTAMIRKG NUMERIC(12,3),
    SEVKPARTIDIGERKG NUMERIC(12,3),
    SEVKPARTIDIGERUCRETLITAMIRKG NUMERIC(12,3),
    SEVKPARTIDIGERUCRETSIZTAMIRKG NUMERIC(12,3),
    SEVKFASONHAMIADEKG NUMERIC(12,3))
AS
declare variable FHamTipiId smallint;
declare variable FTamir smallint;
declare variable FKalanKilo numeric(12,3);
declare variable FPartiTipiId smallint;
declare variable FPartiDurumu smallint;
declare variable FIslemTipiId smallint;
declare variable FFaturaKesilecek smallint;
declare variable FFason smallint;
declare variable FBrutKilo smallint;
declare variable FNetKilo smallint;
declare variable FFasonHamIade smallint;
begin
  DepoStokIplikKg       = 0 ;
  DepoStokIplikTamirKg  = 0 ;

  DepoStokKumasKg       = 0 ;
  DepoStokKumasTamirKg  = 0 ;

  DepoStokDigerKg       = 0 ;
  DepoStokDigerTamirKg  = 0 ;

  IsletmedekiIplikKg    = 0 ;
  IsletmedekiIplikUcretliTamirKg  = 0 ;
  IsletmedekiIplikUcretsizTamirKg = 0 ;

  IsletmedekiKumaskg    = 0 ;
  IsletmedekiKumasUcretliTamirKg  = 0 ;
  IsletmedekiKumasUcretsizTamirKg = 0 ;

  SevkeHazirIplikKg               = 0 ;
  SevkeHazirIplikUcretliTamirKg   = 0 ;
  SevkeHazirIplikUcretsizTamirKg  = 0 ;
  SevkeHazirKumasKg               = 0 ;
  SevkeHazirKumasUcretliTamirKg   = 0 ;
  SevkeHazirKumasUcretsizTamirKg  = 0 ;

  SevkPartiKumasKg                = 0 ;
  SevkPartiKumasUcretliTamirKg    = 0 ;
  SevkPartiKumasUcretsizTamirKg   = 0 ;

  SevkPartiIplikKg                = 0 ;
  SevkPartiIplikUcretliTamirKg    = 0 ;
  SevkPartiIplikUcretsizTamirKg   = 0 ;

  SevkPartiDigerKg                = 0 ;
  SevkPartiDigerUcretliTamirKg    = 0 ;
  SevkPartiDigerUcretsizTamirKg   = 0 ;

  SevkFasonHamIadeKg              = 0 ;

  /* geÃ§ici kabuldeki stoklarÄ± hesapla */
  for select Ham.Hamtipiid, Hamstokmaster.Tamir, sum(Gecicikabulraf.Kilo)
    from Gecicikabulraf
    left join Ham on Ham.Hamid = Gecicikabulraf.Hamid
    left join Hamstokmaster on hamstokmaster.Hamstokmasterid = Gecicikabulraf.Hamstokmasterid
    where Gecicikabulraf.Kilo > Gecicikabulraf.Kararkilo
      and Hamstokmaster.Fason = 1
      and Hamstokmaster.Firmaid = :Afirmaid
    group by Ham.Hamtipiid, Hamstokmaster.Tamir
    into :fHamTipiId, :fTamir, :fKalanKilo do
  begin
    if (fHamTipiId = 0) then
    begin
      if (fTamir = 0) then
        depoStokIplikKg = DepoStokIplikKg + fKalanKilo ;
      else If (fTamir = 1) then
        depoStokIplikTamirKg = depoStokIplikTamirKg + fKalanKilo ;
    end
    else if ( (fHamTipiId = 1) or (fHamTipiId = 2) or (fHamTipiId = 3) or (fHamTipiId = 4) ) then
    begin
      if (fTamir = 0) then
        depoStokKumasKg = depoStokKumasKg + fKalanKilo ;
      else If ( fTamir = 1 ) then
        depoStokKumasTamirKg = depoStokKumasTamirKg + fKalanKilo ;
    end
    else
    begin
      if (fTamir = 0) then
        depoStokDigerKg = depoStokDigerKg + fKalanKilo ;
      else If (fTamir = 1) then
        depoStokDigerTamirKg = depoStokDigerTamirKg + fKalanKilo ;
    end
  end

  /* ham stokdakileri hesapla */
  for select ham.Hamtipiid, hamstokmaster.Tamir, sum(RafDetay.Kilo - RafDetay.IadeEdilenKilo - RafDetay.AktarmayaCikanKilo + RafDetay.AktarmadanIadeEdienKilo - RafDetay.PartilenenKilo -  RafDetay.RafTransferKilo -  RafDetay.SatisKilo)
    from Rafdetay
    left join Ham on Ham.Hamid = Rafdetay.Hamid
    left join Hamstokdetay on Hamstokdetay.Hamstokdetayid = Rafdetay.Hamstokdetayid
    left join Hamstokmaster on Hamstokmaster.Hamstokmasterid = Hamstokdetay.Hamstokmasterid
    where Rafdetay.Stokvar = 1
      and Hamstokmaster.Fason = 1
      and Hamstokmaster.Firmaid = :Afirmaid
    group by Ham.Hamtipiid, Hamstokmaster.Tamir
    into :fHamTipiId, :fTamir, :fKalanKilo do
  begin
    if (fHamTipiId = 0) then
    begin
      if (fTamir = 0) then
        depoStokIplikKg = depoStokIplikKg + fKalanKilo ;
      else if (fTamir = 1) then
        depoStokIplikTamirKg = depoStokIplikTamirKg + fKalanKilo ;
    end
    else if ( (fHamTipiId = 1) or (fHamTipiId = 2) or (fHamTipiId = 3) or (fHamTipiId = 4) ) then
    begin
      if (fTamir = 0) then
        depoStokKumasKg = depoStokKumasKg + fKalanKilo ;
      else If (fTamir = 1) then
        depoStokKumasTamirKg = depoStokKumasTamirKg + fKalanKilo ;
    end
    else
    begin
      if (fTamir = 0) then
        depoStokDigerKg = depoStokDigerKg + fKalanKilo ;
      else If (fTamir = 1) then
        depoStokDigerTamirKg = depoStokDigerTamirKg + fKalanKilo ;
    end
  end


  /* iÅŸletmedekiler ve sevke hazÄ±rlar */
  for select parti.Partitipiid, parti.Islemtipiid, parti.Faturakesilecek, partistok.Durumu, sum(partistok.Kilo - partistok.Brutsevkkilo)
    from Partistok
    left join Parti on Parti.Partikayitid = Partistok.Partikayitid
    where Partistok.Durumu < 8 and Parti.Firmaid = :afirmaid and Partistok.Partistokiptalkayitid is null
    group by Parti.Partitipiid, Parti.Islemtipiid, Parti.Faturakesilecek, Partistok.Durumu
    into :fPartiTipiId, :fIslemTipiId, :fFaturaKesilecek, :fPartiDurumu, :fKalanKilo do
  begin
    if (fPartiTipiId = 0 ) then
    begin
      if (fPartiDurumu < 7 ) then
      begin
        if (fIslemTipiId = 3) then
        begin
          if (fFaturaKesilecek = 0 ) then
            isletmedekiIplikUcretsizTamirKg = isletmedekiIplikUcretsizTamirKg + fKalanKilo ;
          else if (fFaturaKesilecek = 1 ) then
            isletmedekiIplikUcretliTamirKg = isletmedekiIplikUcretliTamirKg + fKalanKilo ;
        end
        else
          isletmedekiIplikKg = isletmedekiIplikKg + fKalanKilo ;
      end
      else if (fPartiDurumu = 7 ) then
      begin
        if (fIslemTipiId = 3) then
        begin
          if (fFaturaKesilecek = 0 ) then
            sevkeHazirIplikUcretsizTamirkg = sevkeHazirIplikUcretsizTamirkg + fKalanKilo ;
          else if (fFaturaKesilecek = 1 ) then
            sevkeHazirIplikUcretliTamirKg = sevkeHazirIplikUcretliTamirKg + fKalanKilo ;
        end
        else
          sevkeHazirIplikKg = sevkeHazirIplikKg + fKalanKilo ;
      end
    end
    else if (fPartiTipiId = 1 ) then
    begin
      if (fPartiDurumu < 7 ) then
      begin
        if (fIslemTipiId = 3) then
        begin
          if (fFaturaKesilecek = 0 ) then
            isletmedekiKumasUcretsizTamirKg = isletmedekiKumasUcretsizTamirKg + fKalanKilo ;
          else if (fFaturaKesilecek = 1 ) then
            isletmedekiKumasUcretliTamirKg = isletmedekiKumasUcretliTamirKg + fKalanKilo ;
        end
        else
          isletmedekiKumasKg = isletmedekiKumasKg + fKalanKilo ;
      end
      else if (fPartiDurumu = 7 ) then
      begin
        if (fIslemTipiId = 3) then
        begin
          if (fFaturaKesilecek = 0 ) then
            sevkeHazirKumasUcretsizTamirKg = sevkeHazirKumasUcretsizTamirKg + fKalanKilo ;
          else if (fFaturaKesilecek = 1 ) then
            sevkeHazirKumasUcretliTamirKg = sevkeHazirKumasUcretliTamirKg + fKalanKilo ;
        end
        else
          sevkeHazirKumasKg = sevkeHazirKumasKg + fKalanKilo ;
      end
    end
  end

  /* sevkler */
    for select Boyairsaliyedetay.Fasonhamiade, Parti.Partitipiid, Parti.Fason, Parti.Islemtipiid, Parti.Faturakesilecek, sum(Boyairsaliyedetay.Brutkilo), sum(Boyairsaliyedetay.Kilo)
      from Boyairsaliye
      left join Boyairsaliyedetay on Boyairsaliyedetay.Boyairsaliyeid = Boyairsaliye.Boyairsaliyeid
      left join Partistok on Partistok.Partisinifid = Boyairsaliyedetay.Partisinifid
      left join Parti on Parti.Partikayitid = Partistok.Partikayitid
      where Boyairsaliye.Irsaliyetarihi = Current_date and Boyairsaliye.Firmaid = :afirmaid
      group by Boyairsaliyedetay.Fasonhamiade, Parti.Partitipiid, Parti.Fason, Parti.Islemtipiid, Parti.Faturakesilecek
      into :fFasonHamIade, :fPartiTipiId, :fFason, :fIslemTipiId, :fFaturaKesilecek, :fBrutKilo, :fNetKilo do
   begin
    if (fFasonHamIade = 0 ) then
    begin
      if (fPartiTipiId = 0 ) then
      begin
        if (fIslemTipiId = 3) then
        begin
          if (fFaturaKesilecek = 0) then
            sevkPartiIplikUcretsizTamirKg = sevkPartiIplikUcretsizTamirKg + fBrutKilo ;
          else if (fFaturaKesilecek = 1) then
            sevkPartiIplikUcretliTamirKg = sevkPartiIplikUcretliTamirKg + fBrutKilo ;
        end
        else
          sevkPartiIplikKg = sevkPartiIplikKg + fBrutKilo ;
      end
      else if (fPartiTipiId = 1 ) then
      begin
        if (fIslemTipiId = 3) then
        begin
          if (fFaturaKesilecek = 0) then
            sevkPartiKumasUcretsizTamirKg = sevkPartiKumasUcretsizTamirKg + fBrutKilo ;
          else if (fFaturaKesilecek = 1) then
            sevkPartiKumasUcretliTamirKg = sevkPartiKumasUcretliTamirKg + fBrutKilo ;
        end
        else
          sevkPartiKumasKg = sevkPartiKumasKg + fBrutKilo ;
      end
      else
      begin
        if (fIslemTipiId = 3) then
        begin
          if (fFaturaKesilecek = 0) then
            sevkPartiDigerUcretsizTamirKg = sevkPartiDigerUcretsizTamirKg + fBrutKilo ;
          else if (fFaturaKesilecek = 1) then
            sevkPartiDigerUcretliTamirKg = sevkPartiDigerUcretliTamirKg + fBrutKilo ;
        end
        else
          sevkPartiDigerKg = sevkPartiDigerKg + fBrutKilo ;
      end
    end
    else if (fFasonHamIade = 1 ) then
      sevkFasonHamIadeKg = sevkFasonHamIadeKg + fBrutKilo ;
   end

  suspend ;
end
