CREATE PROCEDURE COZGUHASILPLANLAMASURUKLEBIRAK(
    ACOZGUHASIL INTEGER,
    AISEMRIKAYITID INTEGER,
    AMAKINAID INTEGER,
    AFIRMAID VARCHAR(20),
    ATASISEMRIKAYITID INTEGER,
    AALTISEMRIKAYITID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable FBoyaSiparisKayitId  integer;
declare variable FSira                  integer;
declare variable OPId           integer;
declare variable ASira          integer;
declare variable FTarih         Date;
begin
  FSira = -1;
  FTarih = current_TimeStamp(0);

  if (ATasIsEmriKayitId is Not null) then
  begin
    if (AFirmaId is null and AMakinaID is null) then
    begin
        if (ACozguHasil = 1) then
          update CozguIsEmri set CozguIsEmri.Sira = null, CozguIsEmri.MakinaId = null, CozguIsEmri.PlanFirmaId = null,
          CozguIsEmri.PlanTarih = null where CozguIsEmri.CozguIsEmriKayitId = :ATasIsEmriKayitId;
        else
          update HasilIsEmri set HasilIsEmri.Sira = null, HasilIsEmri.MakinaId = null, HasilIsEmri.PlanFirmaId = null,
          HasilIsEmri.PlanTarih = null where HasilIsEmri.HasilIsEmriKayitId = :ATasIsEmriKayitId;
        exit;
    end
  end

  if (AAltIsEmriKayitId is not null) then
  begin
    if (ACozguHasil = 1) then
        Select Sira from CozguIsEmri where CozguIsEmri.CozguIsEmriKayitId = :AAltIsEmriKayitId into :FSira;
    else
        Select Sira from HasilIsEmri where HasilIsEmri.HasilIsEmriKayitId = :AAltIsEmriKayitId into :FSira;
    if (FSira is null) then exception OrmePlanlamaYerBulunamadi;
  end

  if (ATasIsEmriKayitId is null) then
  begin
    if (ACozguHasil = 1) then
        update CozguIsEmri set CozguIsEmri.Sira = :FSira, CozguIsEmri.MakinaId = :AMakinaId, CozguIsEmri.PlanFirmaId = :AFirmaId, CozguIsEmri.PlanTarih = :FTarih where CozguIsEmri.CozguIsEmriKayitId = :AIsEmriKayitId;
    else
        update HasilIsEmri set HasilIsEmri.Sira = :FSira, HasilIsEmri.MakinaId = :AMakinaId, HasilIsEmri.PlanFirmaId = :AFirmaId, HasilIsEmri.PlanTarih = :FTarih where HasilIsEmri.HasilIsEmriKayitId = :AIsEmriKayitId;
  end
  else
  begin
    if (ATasIsEmriKayitId is not null) then
    begin
      if (AMakinaId is not null) then
      begin
        if (ACozguHasil = 1) then
        begin
            if (FSira > -1) then
            begin
              ASira = FSira;
              for select CozguIsEmri.CozguIsEmriKayitId from CozguIsEmri where CozguIsEmri.MakinaId = :AMakinaId and Sira >= :FSira order by Sira into :OPId do
              begin
                ASira = ASira + 1;
                update CozguIsEmri set Sira = :ASira where CozguIsEmri.CozguIsEmriKayitId = :OPId;
              end
            end
            else
            begin
              select max(Sira) from CozguIsEmri where CozguIsEmri.MakinaId = :AMakinaId into :ASira;
              if (ASira is null) then ASira = 0;
              else ASira = ASira + 1;
              FSira = ASira;
            end
         end
         else
         begin
            if (FSira > -1) then
            begin
              ASira = FSira;
              for select HasilIsEmri.HasilIsEmriKayitId from HasilIsEmri where HasilIsEmri.MakinaId = :AMakinaId and Sira >= :FSira order by Sira into :OPId do
              begin
                ASira = ASira + 1;
                update HasilIsEmri set Sira = :ASira where HasilIsEmri.HasilIsEmriKayitId = :OPId;
              end
            end
            else
            begin
              select max(Sira) from HasilIsEmri where HasilIsEmri.MakinaId = :AMakinaId into :ASira;
              if (ASira is null) then ASira = 0;
              else ASira = ASira + 1;
              FSira = ASira;
            end
         end
      end
      else
      begin
        if (ACozguHasil = 1) then
        begin
            if (FSira > -1) then
            begin
              ASira = FSira;
              for select CozguIsEmriKayitId from CozguIsEmri where CozguIsEmri.PlanFirmaId = :AFirmaId and Sira >= :FSira order by Sira into :OPId do
              begin
                ASira = ASira + 1;
                update CozguIsEmri set Sira = :ASira where CozguIsEmriKayitId = :OPId;
              end
            end
            else
            begin
              select max(Sira) from CozguIsEmri where CozguIsEmri.PlanFirmaId = :AFirmaId into :ASira;
              if (ASira is null) then ASira = 0;
              else ASira = ASira + 1;
              FSira = ASira;
            end
        end
        else
        begin
            if (FSira > -1) then
            begin
              ASira = FSira;
              for select HasilIsEmriKayitId from HasilIsEmri where HasilIsEmri.PlanFirmaId = :AFirmaId and Sira >= :FSira order by Sira into :OPId do
              begin
                ASira = ASira + 1;
                update HasilIsEmri set Sira = :ASira where HasilIsEmriKayitId = :OPId;
              end
            end
            else
            begin
              select max(Sira) from HasilIsEmri where HasilIsEmri.PlanFirmaId = :AFirmaId into :ASira;
              if (ASira is null) then ASira = 0;
              else ASira = ASira + 1;
              FSira = ASira;
            end
        end
      end
    end
    if (ACozguHasil = 1) then
        Update CozguIsEmri Set MakinaId = :AMakinaId, PlanFirmaId = :AFirmaId, Sira = :FSira, PlanTarih = :FTarih where CozguIsEmri.CozguIsEmriKayitId = :ATasIsEmriKayitId;
    else
        Update HasilIsEmri Set MakinaId = :AMakinaId, PlanFirmaId = :AFirmaId, Sira = :FSira, PlanTarih = :FTarih  where HasilIsEmri.HasilIsEmriKayitId = :ATasIsEmriKayitId;
  end
end
