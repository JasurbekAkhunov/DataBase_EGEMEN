CREATE PROCEDURE ILAVESARFLARISTOKTANSIL(
    APARTIASAMAILAVESARFID INTEGER)
AS
declare variable FStokId integer;
declare variable KayitSayisi integer;
begin
  Select Min(StokId) from StokDetay where StokDetay.PartiAsamaIlaveSarfID = :APartiAsamaIlaveSarfID into :FStokId;

  delete from StokDetay where StokDetay.PartiAsamaIlaveSarfID = :APartiAsamaIlaveSarfID;

  Select Count(*) from StokDetay where StokDetay.StokId = :FStokId into :KayitSayisi;

  if (KayitSayisi = 0) then delete from Stok where Stok.StokId = :FStokId;

  Update PartiAsamaIlaveSarf Set StoktanDusuldu = 0 where PartiAsamaIlaveSarf.PartiAsamaIlaveSarfID = :APartiAsamaIlaveSarfId;
end
