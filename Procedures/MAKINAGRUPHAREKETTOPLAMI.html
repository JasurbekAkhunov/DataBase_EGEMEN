CREATE PROCEDURE MAKINAGRUPHAREKETTOPLAMI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    GRUPID VARCHAR(4),
    GRUPADI VARCHAR(30),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3))
AS
begin
  for select Makina.MakinaGrubu, MakinaGrup.Adi, Sum(PartiAsamaStokDetay.Kilo), Sum(PartiAsamaStokDetay.Metre) from PartiAsamaStokDetay
  left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
  left join MakinaGrup on MakinaGrup.MakinaGrupId = Makina.MakinaGrubu
  where PartiAsamaStokDetay.CikisTarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.CikisTarihi <= :ABitisTarihi and
        Makina.UretimRaporlarinaDahil = 1
  group by Makina.MakinaGrubu, MakinaGrup.Adi into :GrupId, :GrupAdi, :Kilo, :Metre do suspend;
end
