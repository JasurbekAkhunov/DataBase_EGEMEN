CREATE PROCEDURE LABRECETECALISMAGETIR(
    ALABRECETEID INTEGER)
RETURNS (
    LABRECETECALISMAID INTEGER,
    CALISMATARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    RPTMI INTEGER,
    ILAVE INTEGER,
    SOKUMUZERIBOYA INTEGER,
    RPTSTR VARCHAR(3),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    ILAVESOKUMUZERIBOYASTR VARCHAR(20),
    KRITIKCALISMASI INTEGER,
    KRITIKCALISMASISTR VARCHAR(10),
    CALISMATIPIID INTEGER,
    LABPROGRAMNO INTEGER,
    LABPROGRAMNO2 INTEGER,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    PANTONEKODU VARCHAR(20),
    TALEPEDENKULLANICIADI VARCHAR(40),
    MUSTERIRENKKODU VARCHAR(30),
    MUSTERIRENKADI VARCHAR(40),
    ISIKKAYNAGIADI VARCHAR(40),
    STANDARTTIPIADI VARCHAR(50),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PIPETLEMEYAPANKULLANICIID INTEGER,
    PIPETLEMEYAPANKULLANICIADI VARCHAR(40),
    RECETEONISLEMTIPIADI VARCHAR(30),
    ORJINALDEGISTI INTEGER,
    ORJINALDEGISTISTR VARCHAR(5),
    ORJINALDEGISIMTARIHI TIMESTAMP,
    RECETEGRAFIKID VARCHAR(10),
    RECETEGRAFIKADI VARCHAR(40),
    RECETEGRAFIKID2 VARCHAR(10),
    RECETEGRAFIKADI2 VARCHAR(40),
    USTASAMAID INTEGER,
    USTASAMAADI ADI40,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSATISNO INTEGER,
    GRAFIKRESIM BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LABRECETETALEPYIL INTEGER,
    LABRECETETALEPID INTEGER)
AS
begin
  for select LabReceteCalisma.LabReceteCalismaId, LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi, LabReceteCalisma.BitisTarihi,
    LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, LabReceteCalisma.HamId, Ham.Adi, Ham.Adi2,
    LabReceteCalisma.LabProgramNo, LabReceteCalisma.LabProgramNo2, LabReceteTalep.PantoneKodu, TEK.Adi, LabReceteTalep.MusteriRenkKodu,
    LabReceteTalep.MusteriRenkAdi, IsikKaynagi.Adi, StandartTipi.Adi, LabReceteCalisma.Notu,
    LabReceteCalisma.PipetlemeYapanKullaniciId, PYK.Adi, ReceteOnIslemTipi.Adi,
    LabReceteCalisma.orjinalDegisti, LabReceteCalisma.orjinalDegisimTarihi,
    LabReceteCalisma.ReceteGrafikId, ReceteGrafik.Adi, LabReceteCalisma.ReceteGrafikId2, ReceteGrafik2.Adi,
    IK.Adi, LabReceteCalisma.InsertTarihi, SK.Adi, LabReceteCalisma.DegisimTarihi, LabReceteCalisma.BirimFiyat, LabReceteCalisma.DovizID,
    Asama.AsamaId, Asama.Adi, LabReceteCalisma.ReferansAtisNo, ReceteGrafik.Resim, LabReceteTalep.LabReceteTalepYil, LabReceteTalep.LabReceteTalepId  from LabReceteCalisma
  left join Ham On Ham.HamId = LabReceteCalisma.HamId
  left join LabReceteTalep On LabReceteTalep.LabReceteTalepKayitId = LabReceteCalisma.LabReceteTalepKayitId
  left join IsikKaynagi on IsikKaynagi.IsikKaynagiid = LabReceteCalisma.IsikKaynagiid
  left join StandartTipi on StandartTipi.StandartTipiid = LabReceteTalep.StandartTipiid
  left join ReceteOnIslemTipi on ReceteOnIslemTipi.ReceteOnIslemTipiid = LabReceteCalisma.ReceteOnIslemTipiid
  left join ReceteGrafik on ReceteGrafik.ReceteGrafikId = LabReceteCalisma.ReceteGrafikId
  left join ReceteGrafik ReceteGrafik2 on ReceteGrafik2.ReceteGrafikId = LabReceteCalisma.ReceteGrafikId2
  left join Asama on Asama.AsamaId = LabReceteCalisma.UstAsamaId
  Left Join Kullanici IK on IK.KullaniciId = LabReceteCalisma.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = LabReceteCalisma.KullaniciId
  Left Join Kullanici TEK on TEK.KullaniciId = LabReceteTalep.TalepEdenKullaniciId
  Left Join Kullanici PYK on PYK.KullaniciId = LabReceteCalisma.PipetlemeYapanKullaniciId
  where LabReceteCalisma.LabReceteId = :ALabReceteId
  order by LabReceteCalisma.LabReceteCalismaId
  into :LabReceteCalismaId, :CalismaTarihi, :TerminTarihi, :BitisTarihi, :Rptmi, :Ilave, :SokumUzeriBoya, :KritikCalismasi, :HamId, :HamAdi, HamAdi2,
       :LabProgramNo, :LabProgramNo2, :PantoneKodu, :TalepEdenKullaniciAdi, :MusteriRenkKodu, :MusteriRenkAdi, :IsikKaynagiAdi, :StandartTipiAdi, :Notu,
       :PipetlemeYapanKullaniciId, :PipetlemeYapanKullaniciAdi, :ReceteOnIslemTipiAdi, :orjinalDegisti, :orjinalDegisimTarihi,
       :ReceteGrafikId, :ReceteGrafikAdi, :ReceteGrafikId2, :ReceteGrafikAdi2,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :BirimFiyat, :DovizID, :UstAsamaId, :UstAsamaAdi, :ReferansAtisNo, :GrafikResim, :LabReceteTalepYil,:LabReceteTalepId do
  begin
    if (Rptmi = 1) then RptStr = 'Rpt';
    else RptStr = '';

    if (OrjinalDegisti = 1) then OrjinalDegistiStr = 'Evet';
    else OrjinalDegistiStr = '';

    if (KritikCalismasi = 1) then KritikCalismasiStr = 'Kritik';
    else KritikCalismasiStr = '';

    if (Ilave = 1) then IlaveSokumUzeriBoyaStr = 'İlave';
    else if (SokumUzeriBoya = 1) then IlaveSokumUzeriBoyaStr = 'Söküm Üzeri Boya';
    else IlaveSokumUzeriBoyaStr = '';


    CalismaTipiId = 0;

    if (Rptmi = 1) then CalismaTipiId = 1;
    else if (Ilave = 1) then CalismaTipiId = 2;
    else if (SokumUzeriBoya = 1) then CalismaTipiId = 3;
    else if (KritikCalismasi = 1) then CalismaTipiId = 4;

    suspend;
  end
end
