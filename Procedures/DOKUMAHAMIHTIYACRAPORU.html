CREATE PROCEDURE DOKUMAHAMIHTIYACRAPORU(
    ABOYASIPARISKAYITID INTEGER,
    AHAMID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    ORMESIPARISDETAYRECETEID INTEGER,
    FIRMAADI ADI100,
    SIPARISHAMID INTEGER,
    SIPARISHAMADI ADI255,
    RECETEHAMID INTEGER,
    RECETEHAMADI ADI255,
    HAMSTOKMIKTARKILO BOYAMIKTARSIZE,
    HAMSTOKMIKTARBRUTKILO BOYAMIKTARSIZE,
    COZGUSTOKMIKTARKILO BOYAMIKTARSIZE,
    COZGUSTOKMIKTARMETRE BOYAMIKTARSIZE,
    GEREKENATKIMIKTARKILO BOYAMIKTARSIZE,
    GEREKENCOZGUMIKTARKILO BOYAMIKTARSIZE,
    GEREKENCOZGUMIKTARMETRE BOYAMIKTARSIZE,
    HATASTR VARCHAR(500))
AS
declare variable AtkiCozgu              integer;
begin
  for select OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Firma.Adi, Ham.HamId, Ham.Adi
      from BoyaSiparis
      left join OrmeSiparisDetay on OrmeSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
      left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
      left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
      where (:ABoyaSiparisKayitId is not null and BoyaSiparis.BoyaSiparisKayitId = :ABoyaSiparisKayitId) or
      (:ABoyaSiparisKayitId is null and
      BoyaSiparis.DokumaSiparisi = 1 and
      OrmeSiparisDetay.Kapali = 0 and
      OrmeSiparisDetay.Iptal = 0) and
      (OrmeSiparisDetay.HamId = :AHamId or :AHamId is null)
      into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaAdi, :SiparisHamId, :SiparisHamAdi do
  begin
    for select Ham.HamId, Ham.Adi, OrmeSiparisDetayRecete.OrmeSiparisDetayReceteId, OrmeSiparisDetayRecete.AtkiCozgu
        from OrmeSiparisDetayRecete
        left join Ham on Ham.HamId = OrmeSiparisDetayRecete.HamId
        where OrmeSiparisDetayRecete.OrmeSiparisDetayId = :OrmeSiparisDetayId
        into :ReceteHamId, :ReceteHamAdi, :OrmeSiparisDetayReceteId, :AtkiCozgu do
    begin
      /*reçete hamının çözgü olmayan mevcut stok miktarı*/
      select Sum(OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari), Sum(OrmeRaf.BrutGirisMiktari - OrmeRaf.BrutCikisMiktari)
      from OrmeRaf
      left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
      left join Depo on Depo.DepoId = DepoRaf.DepoId
      where OrmeRaf.HamId = :ReceteHamId and
      OrmeRaf.StokVar = 1 and
      Depo.CozguDeposu = 0 and
      Depo.HasilDeposu = 0 and
      Depo.IndigoDeposu = 0
      into :HamStokMiktarKilo, :HamStokMiktarBrutKilo;

      /*reçete hamının çözgü deposuna girilmiş miktarı. Metreyi de alabilmek için stokhareket ten gidiliyor. ormeraf ta metre yok*/
      select Sum(StokHareket.Kilo), Sum(StokHareket.Metre)
      from StokHareket
      left join OrmeRafDetay on OrmeRafDetay.StokHareketId = StokHareket.StokHareketId
      left join OrmeRaf on OrmeRaf.OrmeRafId = OrmeRafDetay.OrmeRafId
      left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
      left join Depo on Depo.DepoId = DepoRaf.DepoId
      where StokHareket.HamId = :ReceteHamId and
      OrmeRaf.StokVar = 1 and
      StokHareket.GirisCikis = 1 and
      (Depo.CozguDeposu = 1 or Depo.HasilDeposu = 1 or Depo.IndigoDeposu = 1)
      into :CozguStokMiktarKilo, :CozguStokMiktarMetre;

      /*reçete hamının çözgü deposundan çıkılmış miktarına da bakarak kalan bulunur*/
      select :CozguStokMiktarKilo - Coalesce(Sum(StokHareket.Kilo),0), :CozguStokMiktarMetre - Coalesce(Sum(StokHareket.Metre),0)
      from StokHareket
      left join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
      left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
      left join Depo on Depo.DepoId = DepoRaf.DepoId
      where StokHareket.HamId = :ReceteHamId and
      StokHareket.GirisCikis = -1 and
      OrmeRaf.StokVar = 1 and
      (Depo.CozguDeposu = 1 or Depo.HasilDeposu = 1 or Depo.IndigoDeposu = 1)
      into :CozguStokMiktarKilo, :CozguStokMiktarMetre;

      /*Sipariş reçetesi için gerekli ham miktarlar bulunur*/
      if (AtkiCozgu = 1) then
      begin
        select DokumaIplikMiktariHesaplama.AgirlikKilo, HataStr from DokumaIplikMiktariHesaplama(:OrmeSiparisDetayReceteId, Null, Null, 1, Null, Null) into :GerekenAtkiMiktarKilo, :HataStr;
        GerekenCozguMiktarKilo = Null;
        GerekenCozguMiktarMetre = Null;
      end
      else if (AtkiCozgu = 2) then
      begin
        select DokumaIplikMiktariHesaplama.AgirlikKilo, DokumaIplikMiktariHesaplama.UzunlukMetre, HataStr from DokumaIplikMiktariHesaplama(:OrmeSiparisDetayReceteId, Null, Null, 1, Null, Null)
        into :GerekenCozguMiktarKilo, :GerekenCozguMiktarMetre, :HataStr;
        GerekenAtkiMiktarKilo = Null;
      end
      suspend;

    end
  end
end
