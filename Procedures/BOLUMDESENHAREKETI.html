CREATE PROCEDURE BOLUMDESENHAREKETI(
    ABOYAHANEBOLUMID INTEGER)
RETURNS (
    DESENID DESENKODUSIZE,
    DESENISTEKLERIID INTEGER,
    SIRA INTEGER,
    ISTEKTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    DESENISTEKTIPIID VARCHAR(5),
    DESENISTEKTIPIADI VARCHAR(50),
    ISTEKBEKLEMEDE INTEGER,
    BASLATANKULLANICIADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENKULLANICIADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIADI VARCHAR(40))
AS
begin
  For Select DesenVaryant.DesenId, DesenIstekleri.DesenIstekleriId, DesenIstekleri.Sira, DesenIstekleri.IstekTarihi,
             DesenIstekleri.TerminTarihi, DesenIstekleri.DesenIstekTipiId, DesenIstekTipi.Adi, DesenIstekleri.Beklemede,
             BasK.Adi, DesenIstekleri.BaslamaTarihi, BitK.Adi, DesenIstekleri.BitisTarihi,
             DesenIstekleri.OnayDurumu, DesenIstekleri.OnayTarihi, OK.Adi from BolumPlanlama
  left join Parti on Parti.PartiKayitId = BolumPlanlama.PartiKayitId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  left join DesenIstekleri on DesenIstekleri.DesenId = DesenVaryant.DesenId
  Left Join Kullanici BasK on BasK.KullaniciId = DesenIstekleri.BaslatanKullaniciId
  Left Join Kullanici BitK on BitK.KullaniciId = DesenIstekleri.BitirenKullaniciId
  Left Join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiId
  left join Kullanici OK on OK.KullaniciId = DesenIstekleri.OnaylayanKullaniciId
  where BolumPlanlama.BoyahaneBolumID = :ABoyahaneBolumID and BolumPlanlama.MakinaPlanlamagrupId is null and
        DesenIstekleri.DesenIstekleriId is not null
  group by DesenVaryant.DesenId, DesenIstekleri.DesenIstekleriId, DesenIstekleri.Sira, DesenIstekleri.IstekTarihi,
           DesenIstekleri.TerminTarihi, DesenIstekleri.DesenIstekTipiId, DesenIstekTipi.Adi, DesenIstekleri.Beklemede,
           BasK.Adi, DesenIstekleri.BaslamaTarihi, BitK.Adi, DesenIstekleri.BitisTarihi,
           DesenIstekleri.OnayDurumu, DesenIstekleri.OnayTarihi, OK.Adi
  into :DesenId, :DesenIstekleriId, :Sira, :IstekTarihi, :TerminTarihi, :DesenIstekTipiId, :DesenIstekTipiAdi,
       :IstekBeklemede, :BaslatanKullaniciAdi, :BaslamaTarihi, :BitirenKullaniciAdi, :BitisTarihi,
       :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciAdi do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    suspend;
  end
end
