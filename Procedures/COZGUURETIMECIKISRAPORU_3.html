CREATE PROCEDURE COZGUURETIMECIKISRAPORU_3(
    ABASLAMATARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    STOKHAREKETID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    LOT VARCHAR(150),
    REFERANSKODU HAMKODUSIZE,
    KUMASHAMID INTEGER,
    KUMASHAMADI ADI255,
    TOPLAMKKMETRE BOYAMIKTARSIZE,
    COZGUMETRE BOYAMIKTARSIZE,
    STOKMIKTARI BOYAMIKTARSIZE,
    GIRISMIKTARI BOYAMIKTARSIZE,
    URETIMPARTINO VARCHAR(30))
AS
declare variable ShtUretimeCikis integer;
declare variable shtAlis         integer;
declare variable OrmeRafId       integer;
declare variable DokumaPlanlamaId integer;
begin
  Select ShtUretimeCikis, shtAlis from OrmeStokHareketTipiGetir into :shtUretimeCikis, :shtAlis;

  /*Çözgü çıkışı ile aynı ham ve lottaki çözgü çıkışları*/
  for select CozguUretimeCikisRaporu.StokHareketId, StokHareket.HamId, Ham.Adi, StokHareket.Lot, HamKumas.ReferansKodu, HamKumas.HamId, HamKumas.Adi,
    Sum((select Sum(DokumaKaliteKontrol.Miktar) from DokumaKaliteKontrol where DokumaKaliteKontrol.OrmeRafId = StokHareket.OrmeRafId)),
    Sum(StokHareket.Miktar), DokumaPlanlama.UretimPartiNo
    from CozguUretimeCikisRaporu(:ABaslamaTarihi, :ABitisTarihi, 0)
    left join StokHareket on CozguUretimeCikisRaporu.HamId = StokHareket.HamId and StrKarsilastir(CozguUretimeCikisRaporu.Lot, StokHareket.Lot) = 1
    left join Depo on Depo.DepoId = StokHareket.DepoId
    left join Ham on Ham.HamId = StokHareket.HamId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join Ham HamKumas on HamKumas.HamId = OrmeSiparisDetay.HamId
    left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = StokHareket.DokumaPlanlamaId
    where StokHareket.DokumaHareketi = 1 and
    StokHareket.HareketTipi = :ShtUretimeCikis and
    (Depo.CozguDeposu = 1 or Depo.HasilDeposu = 1 or Depo.IndigoDeposu = 1) and
    StokHareket.HamIade = 0
    group by CozguUretimeCikisRaporu.StokHareketId, StokHareket.HamId, Ham.Adi, StokHareket.Lot, HamKumas.ReferansKodu, HamKumas.HamId, HamKumas.Adi, DokumaPlanlama.UretimPartiNo
    into :StokHareketId, :HamId, :HamAdi, :Lot, :ReferansKodu, :KumasHamId, :KumasHamAdi, :ToplamKKMetre, :CozguMetre, :UretimPartiNo do
  begin
    /*Çözgünün stok miktarı*/
    select Sum(OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari) from OrmeRaf
    left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
    left join Depo on Depo.DepoId = DepoRaf.DepoId
    where OrmeRaf.StokVar = 1 and
    OrmeRaf.HamId = :HamId and
    StrKarsilastir(OrmeRaf.Lot, :Lot) = 1 and
    (Depo.CozguDeposu = 1 or Depo.HasilDeposu = 1 or Depo.IndigoDeposu = 1) into :StokMiktari;

    /*Çözgünün alım miktarı*/
    select Sum(StokHareket.Miktar) from StokHareket
    left join Depo on Depo.DepoId = StokHareket.DepoId
    where StokHareket.DokumaHareketi = 1 and
    StokHareket.HareketTipi = :shtAlis and
    (Depo.CozguDeposu = 1 or Depo.HasilDeposu = 1 or Depo.IndigoDeposu = 1) and
    StokHareket.HamIade = 0 and
    StokHareket.HamId = :HamId and
    StrKarsilastir(StokHareket.Lot, :Lot) = 1 into :GirisMiktari;

    suspend;
  end
end
