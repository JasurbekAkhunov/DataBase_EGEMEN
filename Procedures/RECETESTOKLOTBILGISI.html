CREATE PROCEDURE RECETESTOKLOTBILGISI(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER)
RETURNS (
    RENK VARCHAR(100),
    LOT VARCHAR(150),
    ANAKUMASEN INTEGER,
    ANAKUMASGRM2 INTEGER,
    ANAKUMASYIKAMAONCESIEN DOUBLE PRECISION,
    ANAKUMASYIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    ANAKUMASTUPACIKENMAYLI INTEGER,
    ONFIKSEASAMASIVAR INTEGER,
    PARTISINIFID INTEGER,
    ILKKEZBOYANACAK INTEGER)
AS
declare variable FLot varchar(150);
declare variable FRenk varchar(20);
declare variable MaxKilo Numeric(12,3);
declare variable HamStokDetayId integer;
declare variable KS integer;
declare variable EskiLabReceteKodu  varchar(20);
declare variable IlkBoyananPartiKayitId integer;
declare variable Ilave integer;
declare variable SokumUzeriBoya integer;
begin
  Lot = '';
  for Select HamStokMaster.Lot from PartiRecetesiDokumu_3(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiRecetesiDokumu_3.PartiSinifID
    left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    group by HamStokMaster.Lot into :FLot do
  begin
    if (FLot <> '' and FLot is not null) then
    if (Char_Length(Lot) + Char_Length(FLot) < 148) then
    begin
      if (Lot <> '' ) then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
      Lot = Lot || cast(FLot as blob character set UTF8);
    end
  end

  Renk = '';
  for Select HamStokMaster.Renk from PartiRecetesiDokumu_3(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiRecetesiDokumu_3.PartiSinifID
    left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    group by HamStokMaster.Renk into :FRenk do
  begin
    if (FRenk <> '' and FRenk is not null) then
    if (Char_Length(Renk) + Char_Length(FRenk) < 95) then
    begin
      if (Renk <> '' ) then Renk = Renk || cast(', ' as varchar(2) character set UTF8);
      Renk = Renk || cast(FRenk as blob character set UTF8);
    end
  end

  Select Recete.IlkBoyananPartiKayitId, LabRecete.EskiLabReceteKodu, Recete.SokumUzeriBoya, Recete.Ilave from Recete
    left join LabRecete on LabRecete.LabReceteID = Recete.LabReceteId
    where Recete.ReceteID = :AReceteId
    into :IlkBoyananPartiKayitId, :EskiLabReceteKodu, :SokumUzeriBoya, :Ilave;

  IlkKezBoyanacak = 0;


  if (SokumUzeriBoya = 0 and Ilave = 0) then
  if (EskiLabReceteKodu is null or EskiLabReceteKodu = '') then
  begin
    if (IlkBoyananPartiKayitId is not null) then
    begin
      KS = 0;

      Select Count(*) from PartiRecetesiDokumu_1(:APartiKayitId, :AReceteId, :APartiAsamalariId)
        where PartiRecetesiDokumu_1.PartiKayitID = :IlkBoyananPartiKayitId
        into :KS;

      if (Ks > 0) then IlkKezBoyanacak = 1;
    end
    else IlkKezBoyanacak = 1;
  end


  Select max(PartiRecetesiDokumu_3.Kilo) from PartiRecetesiDokumu_3(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    into :MaxKilo;

  Select Max(PartiRecetesiDokumu_3.PartiSinifID) from PartiRecetesiDokumu_3(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiRecetesiDokumu_3.PartiSinifID
    where PartiRecetesiDokumu_3.Kilo = :MaxKilo
    into :PartiSinifId;

  Select max(RafDetay.HamStokDetayId) from PartiRecetesiDokumu_3(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiRecetesiDokumu_3.PartiSinifID
    left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
    where PartiRecetesiDokumu_3.Kilo = :MaxKilo
    into :HamStokDetayId;

  Select HamStokMaster.En, HamStokMaster.Grm2, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.TupAcikEnMayli  from HamStokDetay
    left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterID
    where HamStokDetay.HamStokDetayId = :HamStokDetayId
    into :AnaKumasEn, :AnaKumasGrm2, :AnaKumasYikamaOncesiEn, :AnaKumasYikamaOncesiGramaj, :AnaKumasTupAcikEnMayli;

  Select count(*) from PartiAsamalari
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    where PartiAsamalari.PartiKayitID = :APartiKayitID and Asama.OnFikseAsamasimi = 1 and PartiAsamaStok.AsamayaDahil = 1
    into :KS;

  if (KS = 0) then OnFikseAsamasiVar = 0;
  else OnFikseAsamasiVar = 1;

  suspend;
end
