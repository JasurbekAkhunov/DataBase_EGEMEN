CREATE PROCEDURE BAKIMTIPINDENISEMRIKOPYALA(
    AMAKINABAKIMISEMRIID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable MakinaBakimIsEmriId        integer;
begin
  select max(MakinaBakimIsEmriId) from MakinaBakimIsEmri into :MakinaBakimIsEmriId;
  if (MakinaBakimIsEmriId is Null) then MakinaBakimIsEmriId = 0;
  MakinaBakimIsEmriId = MakinaBakimIsEmriId + 1;

  insert into MakinaBakimIsEmri(MakinaBakimIsEmriId, Adi, MakinaBakimTipiId, MakinaBakimBolumuId, TahminiBaslangic, TahminiSaat,
    Notu, MakinaId, OrmeMakinaId, IplikMakinaId, PlanlananBaslangic, Tolerans, ArizaIsEmri,
    SistemOlusturdu, Kapali, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi, BakimDairesiId)
  Select :MakinaBakimIsEmriId, Adi, MakinaBakimTipiId, MakinaBakimBolumuId, TahminiBaslangic, TahminiSaat,
    Notu, MakinaId, OrmeMakinaId, IplikMakinaId, PlanlananBaslangic, Tolerans, 0,
    0, Kapali, :AKullaniciId, current_timestamp(0), :AKullaniciId, current_timestamp(0), BakimDairesiId from MakinaBakimIsEmri
    where MakinaBakimIsEmriId = :AMakinaBakimIsEmriId;
end
