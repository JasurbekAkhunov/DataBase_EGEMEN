CREATE PROCEDURE FIRMAHAMIADEREZERVLISTESI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    HAMIADEREZERVID INTEGER,
    HAMSTOKIRSALIYEID INTEGER,
    HAMSTOKDETAYID INTEGER,
    HAMSTOKMASTERID INTEGER,
    RAFDETAYID INTEGER,
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    LOT VARCHAR(150),
    RENK VARCHAR(20),
    REZERVTARIHI TIMESTAMP,
    RAFID VARCHAR(10),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    IADEEDILENKILO NUMERIC(12,3),
    IADEEDILENMETRE NUMERIC(12,3),
    REZERVVAR INTEGER,
    HAMSTOKIADESEBEBIID INTEGER,
    HAMSTOKIADESEBEBIADI VARCHAR(30),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    REFERANSSIPARISNO VARCHAR(30),
    REFERANSFIRMAID VARCHAR(20),
    REFERANSFIRMAADI VARCHAR(100),
    GIRISIRSALIYEKAPADEDI INTEGER,
    GIRISIRSALIYEKILO NUMERIC(12,3),
    GIRISIRSALIYENO IRSALIYENOSIZE,
    GIRISIRSALIYETARIHI TIMESTAMP,
    GIRISEVRAKTIPI INTEGER,
    GIRISEVRAKTIPISTR VARCHAR(8),
    KALITENO INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    IPTALPARTIIDSTR VARCHAR(30),
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    ORMECIADI ADI30,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    OLCUADI ADI40,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20))
AS
declare variable BoyaSiparisYil integer;
declare variable BoyaSiparisId  integer;
declare variable IptalPartiYil  integer;
declare variable IptalPartiId   integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable TupAcikEnMayli integer;
begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  for select HamIadeRezerv.HamIadeRezervId, HamIadeRezerv.HamStokIrsaliyeId, HamIadeRezerv.HamStokDetayId, HamStokMaster.HamStokMasterId, HamIadeRezerv.RafDetayId,
            Ham.Adi, HamStokMaster.Renk, HamStokMaster.Lot, Ham.HamTipiId, HamIadeRezerv.RezervTarihi, RafDetay.RafId, HamIadeRezerv.KapAdedi, HamIadeRezerv.Kilo, HamIadeRezerv.Metre, HamIadeRezerv.Adet,
            HamIadeRezerv.IadeEdilenKilo, HamIadeRezerv.IadeEdilenMetre,
            HamIadeRezerv.HamStokIadeSebebiId, HamStokIadeSebebi.Adi, HamIadeRezerv.Notu, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
            HamStokMaster.ReferansSiparisNo, HamStokMaster.ReferansFirmaId, ReferansFirma.Adi, RafDetay.KaliteNo,
            IptalParti.PartiYil, IptalParti.PartiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, SiparisFirma.Adi,
            HamStokMaster.KapAdedi, HamStokMaster.Kilo,
            Ik.Adi, HamIadeRezerv.InsertTarihi, Sk.Adi, HamIadeRezerv.DegisimTarihi, HamStokMaster.TupAcikEnMayli, Ormeci.Adi, Olcu.OlcuAdi,
            OrmeSiparisDetay.IplikBoyu1, OrmeSiparisDetay.IplikBoyu2, OrmeSiparisDetay.IplikBoyu3 from HamIadeRezerv
  left join Rafdetay on RafDetay.Rafdetayid = HamIadeRezerv.Rafdetayid
  left join hamstokdetay on hamstokdetay.hamstokdetayid = HamIadeRezerv.hamstokdetayid
  left join HamStokMaster on HamStokMaster.HamStokMasterid = hamstokdetay.HamStokMasterid
  left join HamStokIrsaliye on hamstokIrsaliye.HamStokIrsaliyeid = HamstokDetay.HamStokIrsaliyeid
  left join Ormeci on Ormeci.OrmeciId = HamStokIrsaliye.OrmeciId
  left join HamStokIadeSebebi on HamStokIadeSebebi.HamStokIadeSebebiid = HamIadeRezerv.HamStokIadeSebebiid
  left join Firma ReferansFirma on ReferansFirma.FirmaId = HamStokMaster.ReferansFirmaId
  left join Ham on ham.hamid = RafDetay.hamid
  left join PartiStok on PartiStok.PartiStokIptalKayitId = RafDetay.PartiStokIptalKayitId
  left join Parti IptalParti on IptalParti.PartiKayitId = PartiStok.PartiKayitId
  Left Join Kullanici IK on IK.KullaniciId = HamIadeRezerv.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HamIadeRezerv.KullaniciId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = HamStokMaster.BoyaSiparisKayitId
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join StokHareket on StokHareket.StokHareketId = HamStokMaster.StokHareketID
  left join Olcu on Olcu.OlcuId = StokHareket.OlcuId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = HamStokMaster.OrmeSiparisDetayId
  where HamIadeRezerv.FirmaId = :AFirmaId and HamIadeRezerv.RezervTarihi >= :ABaslangicTarihi and HamIadeRezerv.Rezervtarihi <= :ABitistarihi
  into :HamIadeRezervId, :HamStokIrsaliyeId, :HamStokDetayId, :HamStokMasterId, :RafDetayId, :HamAdi, :Renk, :Lot, :HamTipiId, :RezervTarihi, :RafId, :KapAdedi, :Kilo, :Metre, :Adet,
       :IadeEdilenKilo, :IadeEdilenMetre,
       :HamStokIadeSebebiId, :HamStokIadeSebebiAdi, :Notu, :GirisIrsaliyeNo, :GirisIrsaliyeTarihi, :GirisEvrakTipi,
       :ReferansSiparisNo, :ReferansFirmaId, :ReferansFirmaAdi, :KaliteNo,
       :IptalPartiYil, :IptalPartiId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :GirisIrsaliyeKapAdedi, :GirisIrsaliyeKilo,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :TupAcikEnMayli, :OrmeciAdi, :OlcuAdi,
       :IplikBoyu1, :IplikBoyu2, :IplikBoyu3 do
  begin
    if (GirisEvrakTipi = 0) then GirisEvrakTipiStr = 'Fiş';
    else if (GirisEvrakTipi = 1) then GirisEvrakTipiStr = 'İrsaliye';
    else GirisEvrakTipiStr = 'Hatalı';

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    if (hamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
    begin
      if (Kilo <> IadeEdilenKilo) then Rezervvar = 1;
      else RezervVar = 0;
    end
    else
    begin
      if (Metre <> IadeEdilenMetre) then Rezervvar = 1;
      else RezervVar = 0;
    end


    if (IptalPartiYil is not null) then IptalPartiIdStr = '(' || EgemenIntegerToStr(IptalPartiYil)  || ') ' || EgemenIntegerToStr(IptalPartiId) || ' ';
    else IptalPartiIdStr = '';

    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    else BoyaSiparisIdStr = '';


    suspend;
  end
end
