CREATE PROCEDURE RENKMALIYETLISTESI_2(
    ARECETELER VARCHAR(200),
    ARAPORDOVIZID VARCHAR(5),
    AFIYATTIPI INTEGER,
    AURUNFIYATGOREBILIR SMALLINT)
RETURNS (
    LABRECETEATISID INTEGER,
    LABRECETEURUNID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNTIPIID INTEGER,
    MIKTARYUZDE DOUBLE PRECISION,
    HESAPBIRIMID VARCHAR(5),
    BIRIMID VARCHAR(2),
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATDOVIZID VARCHAR(5),
    RAPORBIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5))
AS
declare variable LabReceteDovizId varchar(20);
declare variable AlisFiyati       double precision;
declare variable AlisFiyatiDovizId   varchar(5);
declare variable UretimeCikisFiyati     double precision;
declare variable UretimeCikisFiyatiDovizId  varchar(5);
declare variable DovizKuru  Double Precision;
declare variable FirmaId    varchar(20);
begin
  DovizId = ARaporDovizId;

  for select RenkMaliyetListesi_1.LabReceteAtisId,  RenkMaliyetListesi_1.FirmaId, LabReceteUrun.LabReceteUrunId, LabReceteUrun.UrunId, Urun.Adi, LabReceteUrun.MiktarYuzde,
             LabReceteUrun.BirimFiyat, UrunBirim.BirimId, LabRecete.DovizId, Urun.UrunTipiId,
             UrunFiyat.AlisFiyati,  UrunFiyat.AlisFiyatiDovizId,  UrunFiyat.UretimeCikisFiyati, UrunFiyat.UretimeCikisFiyatiDovizId from RenkMaliyetListesi_1(:AReceteler, :ARaporDovizId, :AFiyatTipi)
  left join LabRecete on LabRecete.LabReceteId = RenkMaliyetListesi_1.LabReceteId
  left join LabReceteUrun on LabReceteUrun.LabReceteAtisId = RenkMaliyetListesi_1.LabReceteAtisId
  left join Urun on Urun.UrunId = LabReceteUrun.UrunId
  left join UrunFiyat on UrunFiyat.UrunId = LabReceteUrun.UrunId
  left join UrunBirim on UrunBirim.UrunBirimId = LabReceteUrun.UrunBirimId
  where LabReceteUrun.LabReceteUrunId is not null
  into :LabReceteAtisId, :FirmaId, :LabReceteUrunId, :UrunId, :UrunAdi, :MiktarYuzde, :BirimFiyat, :BirimId, :LabReceteDovizId, :UrunTipiId,
       :AlisFiyati,  :AlisFiyatiDovizId,  :UretimeCikisFiyati, :UretimeCikisFiyatiDovizId do
  begin
    if (AFiyatTipi = 0) then
    begin
      BirimFiyat = UretimeCikisFiyati;
      BirimFiyatDovizId = UretimeCikisFiyatiDovizID;
    end
    else
    begin
      BirimFiyat = AlisFiyati;
      BirimFiyatDovizId = AlisFiyatiDovizID;
    end

    if (BirimFiyatDovizId <> ARaporDovizId) then
      select DovizKuru from FirmaDovizKurlariniGetir(:FirmaId, :BirimFiyatDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
    else
      DovizKuru = 1;

    RaporBirimFiyat = BirimFiyat * DovizKuru;

    if (AUrunFiyatGorebilir = 0) then
    begin
      BirimFiyat = Null;
      RaporBirimFiyat = Null;
    end

    suspend;
  end
end
