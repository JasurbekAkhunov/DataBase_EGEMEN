CREATE PROCEDURE STOKGIRISRAPORUIPLIK_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    ALOT VARCHAR(30),
    ASIFIROLANLARIDOK INTEGER,
    AALIS INTEGER,
    AURETIM INTEGER,
    AURETIMIADE INTEGER,
    ATELEF INTEGER,
    ASATISIADE INTEGER,
    ADEPOTRANSFER INTEGER,
    ABUKUMURETIMI INTEGER,
    ASARIMURETIMI INTEGER)
RETURNS (
    IPLIKIRSALIYEDETAYID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ARACID VARCHAR(11),
    IPLIKTALEPYIL INTEGER,
    IPLIKTALEPID INTEGER,
    IPLIKTALEPIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    KIRPINTITASNIFYIL INTEGER,
    KIRPINTITASNIFID INTEGER,
    KIRPINTITASNIFSTR VARCHAR(30),
    MIKTAR DOUBLE PRECISION,
    NETMIKTAR DOUBLE PRECISION,
    IPLIKDEPOID VARCHAR(10),
    HEDEFIPLIKDEPOID VARCHAR(10))
AS
declare variable shtAlisIade integer;
declare variable shtSatis integer;
declare variable shtUretimeCikis    integer;
declare variable shtUretimCikisBoya integer;
declare variable shtSarimaCikis     integer;
declare variable shtBukumeCikis     integer;
declare variable ShtDepoTransfer    integer;
declare variable shtKirpintiCikis    integer;

begin
  Select shtAlisIade, shtSatis, shtUretimeCikis, shtUretimCikisBoya, ShtSarimaCikis, shtBukumeCikis, ShtDepoTransfer, shtKirpintiCikis from IplikStokHareketTipiGetir into :shtAlisIade, :shtSatis, :shtUretimeCikis, :shtUretimCikisBoya, :ShtSarimaCikis, :shtBukumeCikis, :ShtDepoTransfer, :shtKirpintiCikis;

  For Select StokGirisRaporuIplik_1.IplikIrsaliyeDetayId, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
             IplikIrsaliye.HareketTipi, IplikIrsaliye.FirmaId, Firma.Adi, IplikIrsaliye.AracId,
             IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             Sum(IplikStok.Miktar), Sum(IplikStok.NetMiktar), IplikStok.IplikDepoID, IplikIrsaliye.IplikTransferDepoID,
             KirpintiTasnif.KirpintiTasnifYil, KirpintiTasnif.KirpintiTasnifId from StokGirisRaporuIplik_1(:ABaslangicTarihi, :ABitisTarihi, :AFirmaId, :AHamTipiId, :AHamId, :ALot, :ASifirOlanlariDok, :AAlis, :AUretim, :AUretimIade, :ATelef, :ASatisIade, :ADepoTransfer, :ABukumUretimi, :ASarimUretimi)
    left join IplikStok on IplikStok.IplikIrsaliyeDetayId = StokGirisRaporuIplik_1.IplikIrsaliyeDetayId
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = IplikStok.CikisIplikIrsaliyeDetayId
    left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
    left join Firma on Firma.FirmaId = IplikIrsaliye.FirmaId
    left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
    left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikIrsaliyeDetay.IplikTalepDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
    left join KirpintiTasnifDetay ON KirpintiTasnifDetay.KirpintiTasnifDetayId = IplikIrsaliyeDetay.KirpintiTasnifDetayId
    left join KirpintiTasnif ON KirpintiTasnif.KirpintiTasnifKayitId = KirpintiTasnifDetay.KirpintiTasnifKayitId
    where IplikIrsaliyeDetay.IplikIrsaliyeDetayID is not null
    group by StokGirisRaporuIplik_1.IplikIrsaliyeDetayId, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
             IplikIrsaliye.HareketTipi, IplikIrsaliye.FirmaId, Firma.Adi, IplikIrsaliye.AracId,
             IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             IplikStok.IplikDepoID, IplikIrsaliye.IplikTransferDepoID, KirpintiTasnif.KirpintiTasnifYil, KirpintiTasnif.KirpintiTasnifId
    into :IplikIrsaliyeDetayId, :IrsaliyeNo, :IrsaliyeTarihi, :HareketTipi, :FirmaId, :FirmaAdi, :AracId,
         :IplikTalepYil, :IplikTalepId, :BoyaSiparisYil, :BoyaSiparisId, :Miktar, :NetMiktar,:IplikDepoID, :HedefIplikDepoID,
         :KirpintiTasnifYil, :KirpintiTasnifId do
  begin
    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';
    else
      BoyaSiparisIdStr = '';

    if (IplikTalepYil is not null) then
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';
    else
      IplikTalepIdStr = '';

    KirpintiTasnifStr = '';
    if (KirpintiTasnifYil is not null) then
      KirpintiTasnifStr = '(' || EgemenIntegerToStr(KirpintiTasnifYil) || ') '  || EgemenIntegerToStr(KirpintiTasnifId)  || ' ';

    if (HareketTipi = shtSatis) then HareketTipiStr = 'Satış İrsaliyesi';
    else if (HareketTipi = shtAlisIade) then HareketTipiStr = 'Alış İade İrsaliyesi';
    else if (HareketTipi = shtUretimeCikis) then HareketTipiStr = 'Üretime Çıkış';
    else if (HareketTipi = shtUretimCikisBoya) then HareketTipiStr = 'Üretime Çıkış (Boya)';
    else if (HareketTipi = ShtSarimaCikis) then HareketTipiStr = 'Sarım''a Çıkış';
    else if (HareketTipi = ShtBukumeCikis) then HareketTipiStr = 'Büküm''e Çıkış';
    else if (HareketTipi = ShtDepoTransfer) then HareketTipiStr = 'Depo Transfer';
    else if (HareketTipi = shtKirpintiCikis and KirpintiTasnifYil is not Null) then HareketTipiStr = 'Üretime Çıkış (Tasnif)';
    else if (HareketTipi = shtKirpintiCikis) then HareketTipiStr = 'Üretime Çıkış(Kırpıntı)';
    else HareketTipiStr = 'Hatalı';

    suspend;
  end
end
