CREATE PROCEDURE PARTIGENELRAPORUDETAYLI_6(
    AIMALATTIPIID INTEGER,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    AFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    ARECETEID VARCHAR(20),
    ASABLONRECETEID VARCHAR(20),
    ADESENID VARCHAR(20),
    ADESENVARYANTID INTEGER,
    AMODELNO VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AHAMID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ASINIF VARCHAR(1),
    AISLEMGRUBU INTEGER,
    AISLEMTIPI INTEGER,
    AACILIYETDERECESI INTEGER,
    AFASON INTEGER,
    ARAPORTIPI INTEGER,
    ASEVKDURUMU INTEGER,
    ARECETEYAZILDI INTEGER,
    AYURTDISI INTEGER,
    APARTIGRUPID VARCHAR(4),
    AHAMTIPIID INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AKALINLIKID INTEGER,
    AIPLIKCINSIID INTEGER,
    AOZELLIKID INTEGER,
    AKUMASCINSIID INTEGER,
    ALYCRATIPIID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ABASLANGICKILO NUMERIC(12,3),
    ABITISKILO NUMERIC(12,3),
    ALOT VARCHAR(30),
    AKALITEID VARCHAR(20),
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AURUNFIYATGOREBILIR SMALLINT,
    AARGENO VARCHAR(30),
    ADESENGRUPID URUNGRUPKODUSIZE,
    AIPTALPARTILERIGOREBILIR SMALLINT,
    AYARIMAMULTAMAMLANDI SMALLINT,
    AMUSTERIKALITENO VARCHAR(20))
RETURNS (
    PARTIKAYITID INTEGER,
    KALITE INTEGER,
    SEVKEHAZIRKILO DOUBLE PRECISION,
    SEVKEHAZIRMETRE DOUBLE PRECISION,
    TOPLAMKILO DOUBLE PRECISION,
    TOPLAMMETRE DOUBLE PRECISION)
AS
begin
  for select    PartiGenelRaporuDetayli.PartiKayitId, PartiStokTarti.KaliteNo, Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.Metre)
              , Sum(iif(PartiStokTarti.BoyaIrsaliyeId is NULL AND PartiStokTarti.HamStokSarfFisiID is NULL, PartiStokTarti.Kilo, 0))
              , Sum(iif(PartiStokTarti.BoyaIrsaliyeId is NULL AND PartiStokTarti.HamStokSarfFisiID is NULL, PartiStokTarti.Metre, 0))
      from      PartiGenelRaporuDetayli(:AImalatTipiId, :ABaslangicPartiYil, :ABaslangicPartiId, :ABitisPartiYil, :ABitisPartiId, :AFirmaId, :ATemsilciId, :AReceteId, :ASablonReceteId, :ADesenId, :ADesenVaryantId, :AModelNo,
                :AMusteriSiparisNo, :AReferansSiparisNo, :AHamId, :ABaslangicTarihi, :ABitisTarihi, :ABoyaSiparisYil, :ABoyaSiparisId, :ASinif, :AIslemGrubu, :AIslemTipi, :AAciliyetDerecesi, :AFason, :ARaporTipi, :ASevkDurumu,
                :AReceteYazildi, :AYurtDisi, :APartiGrupId, :AHamTipiId, :ASiparisiAlanKullaniciId, :AKalinlikId, :AIplikCinsiId, :AOzellikId, :AKumasCinsiId, :ALycraTipiId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id,
                :ABitisGrup2Id, :ABaslangicGrup3Id, :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id, :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ABaslangicKilo, :ABitisKilo, :ALot, :AKaliteId,
                :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AUrunFiyatGorebilir, :AArgeNo, :ADesenGrupId, :AIptalPartileriGorebilir, :AYariMamulTamamlandi, :AMusteriKaliteNo)
      left join PartiStokTarti on PartiStokTarti.PartiKayitId = PartiGenelRaporuDetayli.PartiKayitId
      group by  PartiGenelRaporuDetayli.PartiKayitId,
                PartiStokTarti.KaliteNo
      into      :PartiKayitId, :Kalite, :ToplamKilo, :ToplamMetre, :SevkeHazirKilo, :SevkeHazirMetre
      do
  begin
    suspend;
  end
end
