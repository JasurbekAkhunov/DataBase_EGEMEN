CREATE PROCEDURE MAKINASARJRAPORUPARTIDETAY(
    AUNITEID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAGRUBU VARCHAR(4),
    AMAKINAID VARCHAR(10))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    IRSALIYETARIHI TIMESTAMP,
    EN INTEGER,
    GRAMAJ INTEGER)
AS
begin
  for select Distinct HamStokMaster.En, HamStokMaster.Grm2, HamStokIrsaliye.IrsaliyeTarihi, PartiStokDetay.PartiKayitId, PartiStokDetay.PartiSinifId from MakinaSarjRaporuPartiler(:AUniteId, :ABaslangicTarihi, :ABitisTarihi, :AMakinaGrubu, :AMakinaId)
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = MakinaSarjRaporuPartiler.PartiSinifId
    left join RafDetay on RafDetay.RafDetayid = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayid = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterid = HamStokDetay.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
    into :En, :Gramaj, :IrsaliyeTarihi, :PartiKayitId, :PartiSinifId do
  begin
    suspend;
  end
end
