CREATE PROCEDURE GUNBAZINDAAYLIKURETIMRAPORU_3(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMTIPIID INTEGER,
    APARTIGRUPID VARCHAR(4))
RETURNS (
    TARIH TIMESTAMP,
    NORMALSEVKIYAT NUMERIC(12,3),
    TAMIRSEVKIYAT NUMERIC(12,3),
    TOPLAMSEVKIYAT NUMERIC(12,3),
    TAMIRORANI DOUBLE PRECISION,
    NORMALNETSEVKIYAT NUMERIC(12,3),
    TAMIRNETSEVKIYAT NUMERIC(12,3),
    TOPLAMNETSEVKIYAT NUMERIC(12,3))
AS
declare variable FIslemTipiId integer;
declare variable FKilo        integer;
declare variable FNetKilo        integer;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable FLastIrsaliyeTarihi TimeStamp;
declare variable itTamir          integer;

declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;
  Select itTamir from IslemTipiGetir into :itTamir;

  FLastIrsaliyeTarihi = null;

  For Select BoyaIrsaliyeDetay.IrsaliyeTarihi, Parti.IslemTipiId, Sum(BoyaIrsaliyeDetay.BrutKilo), Sum(BoyaIrsaliyeDetay.Kilo) from BoyaIrsaliyeDetay
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
    where BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliyeDetay.IrsaliyeTarihi <= :ABitisTarihi and
          BoyaIrsaliyeDetay.HareketTipiID = 0 and BoyaIrsaliyeDetay.PartiSinifID is not null and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null) and
          (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null)
    group by BoyaIrsaliyeDetay.IrsaliyeTarihi, Parti.IslemTipiId
    order by BoyaIrsaliyeDetay.IrsaliyeTarihi
    into :FIrsaliyeTarihi, :FIslemTipiId, :FKilo, :FNetKILO do
  begin
    if (FLastIrsaliyeTarihi is null or FIrsaliyeTarihi <> FLastIrsaliyeTarihi) then
    begin
      if (FLastIrsaliyeTarihi is not null) then
      begin
        if (ToplamSevkiyat <> 0) then TamirOrani = TamirSevkiyat * 100.00 / ToplamSevkiyat;
        if (ToplamNetSevkiyat <> 0) then TamirOrani = TamirSevkiyat * 100.00 / ToplamNetSevkiyat;

        suspend;
      end

      NormalSevkiyat    = 0;
      NormalNetSevkiyat    = 0;
      TamirSevkiyat     = 0;
      TamirNetSevkiyat   = 0;
      ToplamNetSevkiyat = 0;
      ToplamSevkiyat    = 0;
      TamirOrani        = 0;
      Tarih = FIrsaliyeTarihi;
      FLastIrsaliyeTarihi = FIrsaliyeTarihi;

    end

    if (FIslemTipiId = itTamir) then TamirSevkiyat = TamirSevkiyat + FKilo;
    else NormalSevkiyat = NormalSevkiyat + FKilo;

    if (FIslemTipiId = itTamir) then TamirNetSevkiyat = TamirSevkiyat + FNetKilo;
    else NormalNetSevkiyat = NormalNetSevkiyat + FNetKilo;

    ToplamSevkiyat = ToplamSevkiyat + FKilo;
    ToplamNetSevkiyat = ToplamNetSevkiyat + FNetKilo;
  end

  if (FLastIrsaliyeTarihi is not null) then
  begin
    if (ToplamSevkiyat <> 0) then TamirOrani = TamirSevkiyat * 100.00 / ToplamSevkiyat;
    suspend;
  end
end
