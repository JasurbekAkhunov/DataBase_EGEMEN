CREATE PROCEDURE HAMSTOKMASTERGETIR(
    AHAMSTOKIRSALIYEID INTEGER)
RETURNS (
    HAMSTOKMASTERID INTEGER,
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    HAMID INTEGER,
    PUS INTEGER,
    FEINE INTEGER,
    ORMECIMAKINANO VARCHAR(10),
    EN INTEGER,
    GRM2 INTEGER,
    RENK VARCHAR(20),
    TUPACIKENMAYLISTR VARCHAR(10),
    LOT VARCHAR(150),
    DENYE INTEGER,
    LYCRAURETICIADI VARCHAR(20),
    IPLIKUZUNLUGU1 VARCHAR(40),
    IPLIKUZUNLUGU2 VARCHAR(40),
    IPLIKUZUNLUGU3 VARCHAR(40),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    MIKTAR2 NUMERIC(12,3),
    BRUTNET INTEGER,
    BRUTNETSTR VARCHAR(5),
    ACIKLAMA BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TAMIR INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    HARMANADI VARCHAR(50),
    KONIKRENGIADI VARCHAR(20),
    KONIKCINSIADI ADI20,
    AMBALAJADI ADI20,
    MUSTERIPARTINO VARCHAR(30),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    REFERANSFIRMAID VARCHAR(20),
    REFERANSFIRMAADI VARCHAR(100),
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    RAFLANMAYANMIKTAR NUMERIC(12,3),
    HAMSTOKIADESEBEBIID INTEGER,
    HAMSTOKIADESEBEBIADI VARCHAR(30),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(4),
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP,
    HAMSTOKFATURAID INTEGER,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    SIPARISRENKADI VARCHAR(40),
    SIPARISENISTENEN VARCHAR(10),
    SIPARISGRAMAJISTENEN VARCHAR(10),
    SIPARISDONMEISTENEN VARCHAR(10),
    SIPARISCEKMEZLIKENISTENEN DOUBLE PRECISION,
    SIPARISCEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    ORMEGIRISIRSALIYENO IRSALIYENOSIZE,
    TOPRAFLAMAYAPILDI INTEGER,
    KALITEID VARCHAR(20),
    DEVIRPARTIYIL INTEGER,
    DEVIRPARTIID INTEGER,
    PARTIGRUPID VARCHAR(4),
    ESKIPARTIYIL INTEGER,
    ESKIPARTINO INTEGER,
    URETIMPARTINO VARCHAR(30),
    ELYAFICERIGIANALIZI EGEBOOLEAN,
    BATCHNO VARCHAR(30),
    URETICIFIRMAID VARCHAR(20),
    URETICIFIRMAADI VARCHAR(100),
    BOYASIPARISFIRMAID VARCHAR(20),
    BOYASIPARISFIRMAADI VARCHAR(100),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    URETICIID INTEGER,
    URETICIADI ADI40,
    DEGISIMLOG BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    REFERANSKODU HAMKODUSIZE,
    OLCUID INTEGER,
    OLCUADI ADI40,
    SERTIFIKANO VARCHAR(30),
    ENTEGRASYONKODU VARCHAR(50),
    KALITEADI VARCHAR(255),
    KONTROLDURUMU VARCHAR(10))
AS
declare variable Miktar  numeric(12,3);
declare variable TupAcikEnMayli  integer;
begin
  for select HSM.HamStokMasterId, Ham.HamTipiId, HSM.HamID, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, HSM.Pus, HSM.Feine, HSM.OrmeciMakinaNo, HSM.En, HSM.Grm2,  HSM.Renk, HSM.Lot, HSM.TupAcikEnMayli,
             HSM.Denye, LycraUretici.Adi, HSM.IplikUzunlugu1, HSM.IplikUzunlugu2, HSM.IplikUzunlugu3,
             HSM.KapAdedi, HSM.Miktar2, HSM.Miktar, (HSM.Miktar - HSM.RaflananMiktar), HSM.Kilo, HSM.Metre, HSM.Adet, HSM.Brut, HSM.Aciklama,  HSM.Tamir, TamirTipi.Adi, HSM.EskiPartino, HSM.MusteriPartiNo,
             HSM.MusteriSiparisNo, HSM.ReferansSiparisNo, HSM.ReferansFirmaId, ReferansFirma.Adi,
             IK.Adi, HSM.InsertTarihi, Sk.Adi, HSM.DegisimTarihi,
             VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             coalesce(VerilenSiparisDetay.BoyaSiparisDetayId, HSM.BoyaSiparisDetayId),
             HSM.OnayDurumu, HSM.OnaylayanKullaniciId, OK.Adi, HSM.OnayTarihi, HSM.HamStokFaturaId,
             HSM.HamStokIadeSebebiId, HamStokIadeSebebi.Adi, BoyaSiparisDetay.RenkAdi,
             BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.DonmeIstenen,
             BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen,
             BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, HSM.OrmeGirisIrsaliyeNo,
             HamStokFatura.FaturaNo, HamStokFatura.FaturaTarihi, HSM.HarmanAdi, KonikRengi.Adi, HSM.UreticiFirmaId, UreticiFirma.Adi,
             HSM.TopRaflamaYapildi, HSM.KaliteId, HSM.DevirPartiYil, HSM.DevirPartiId, HSM.PartiGrupId,
             HSM.EskiPartiYil, OrmeRaf.UretimPartiNo,
             BoyaSiparis.FirmaId, SiparisFirma.Adi, HSM.ElyafIcerigiAnalizi, HSM.BatchNo, KonikCinsi.Adi, Ambalaj.Adi, Uretici.UreticiId, Uretici.Adi, HSM.DegisimLog, Ham.ReferansKodu,
             Olcu.OlcuId, Olcu.OlcuAdi, HSM.SertifikaNo, HSM.EntegrasyonKodu, Kalite.KaliteAdi from HamStokMaster HSM
  left join Ham on Ham.HamId = HSM.HamId
  left join DisTamir on DisTamir.HamStokMasterId = HSM.HamStokMasterId
  left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  left join LycraUretici on LycraUretici.LycraUreticiId = HSM.LycraUreticiId
  left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = HSM.VerilenSiparisDetayId
  left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = coalesce(VerilenSiparisDetay.BoyaSiparisKayitId, HSM.BoyaSiparisKayitId)
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = coalesce(VerilenSiparisDetay.BoyaSiparisDetayId, HSM.BoyaSiparisDetayId)
  left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  left join Firma ReferansFirma on ReferansFirma.FirmaId = HSM.ReferansFirmaId
  left join Firma UreticiFirma on UreticiFirma.FirmaId = HSM.UreticiFirmaId
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join HamStokIadeSebebi on HamStokIadeSebebi.HamStokIadeSebebiId = HSM.HamStokIadeSebebiId
  left join HamStokFatura on HamStokFatura.HamStokFaturaId = HSM.HamStokFaturaId
  left join KonikRengi on KonikRengi.KonikRengiId = HSM.KonikRengiId
  left join KonikCinsi on KonikCinsi.KonikCinsiId = HSM.KonikCinsiId
  left join Ambalaj on Ambalaj.AmbalajId = HSM.AmbalajId
  left join StokHareket on StokHareket.StokHareketId = HSM.StokHareketId
  left join OrmeRaf on OrmeRaf.ORmeRAfID = StokHareket.OrmeRafId
  Left Join Kullanici OK on OK.KullaniciId = HSM.OnaylayanKullaniciId
  Left Join Kullanici IK on IK.KullaniciId = HSM.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HSM.KullaniciId
  left join Uretici on Uretici.UreticiId = HSM.UreticiId
  left join Olcu on Olcu.OlcuId = HSM.OlcuId
  left join Kalite on Kalite.KaliteId  =   HSM.KaliteId
  where HSM.HamStokIrsaliyeId = :AHamStokIrsaliyeId order by HSM.HamStokMasterId
  into :HamStokMasterId, :HamTipiId, :HamID, :HamAdi, :HamAdi2, :HamAdi3, :HamAdi4, :Pus, :Feine, :OrmeciMakinaNo, :En, :Grm2, :Renk, :Lot, :TupAcikEnMayli, :Denye, :LycraUreticiAdi,
       :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3,
       :KapAdedi, :Miktar2, :Miktar, :RaflanmayanMiktar, :Kilo, :Metre, :Adet, :BrutNet, :Aciklama, :Tamir, :TamirTipiAdi, :EskiPartino, :MusteriPartiNo,
       :MusteriSiparisNo, :ReferansSiparisNo, :ReferansFirmaId, :ReferansFirmaAdi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :VerilenSiparisYil, :VerilenSiparisId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
       :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi, :HamStokFaturaId, :HamStokIadeSebebiId, :HamStokIadeSebebiAdi,
       :SiparisRenkAdi, :SiparisEnIstenen, :SiparisGramajIstenen, :SiparisDonmeIstenen,
       :SiparisCekmezlikEnIstenen, :SiparisCekmezlikBoyIstenen, :SiparisProsesId, :SiparisProsesAdi, :OrmeGirisIrsaliyeNo,
       :FaturaNo, :FaturaTarihi, :HarmanAdi, :KonikRengiAdi, :UreticiFirmaId, :UreticiFirmaAdi,
       :TopRaflamaYapildi, :KaliteId, :DevirPartiYil, :DevirPartiId, :PartiGrupId,
       :EskiPartiYil, :UretimPartiNo, :BoyaSiparisFirmaId, :BoyaSiparisFirmaAdi, :ElyafIcerigiAnalizi, :BatchNo, :KonikCinsiAdi, :AmbalajAdi, :UreticiId, :UreticiAdi, :DegisimLog, :ReferansKodu,
       :OlcuId, :OlcuAdi, :SertifikaNo, :EntegrasyonKodu, :KaliteAdi do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Okey';
    else OnayDurumuStr = '';

    if (BrutNet = 0) then BrutNetStr = 'Net';
    else if (BrutNet = 1) then BrutNetStr = 'Brüt';
    else BrutNetStr = '';

    Select TupAcikEnMayliStrGetir.tupacikenmaylistr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    VerilenSiparisIdStr = '';
    if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    KontrolDurumu = '';
    Select Count(*) || '/'|| Sum(HamStokMasterTop.KontrolEdildi) from HamStokMasterTop where HamStokMasterTop.HamStokMasterId = :HamStokMasterId
    into :KontrolDurumu;

    suspend;
  end
end
