CREATE PROCEDURE STOKIRSALIYERAPORU_1(
    AHAREKETTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AIRSALIYENO IRSALIYENOSIZE,
    AURUNTIPIID INTEGER)
RETURNS (
    IRSALIYEID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYEGRUPID VARCHAR(4),
    REFERANSEVRAKNO VARCHAR(10),
    FIILISEVKTARIHI TIMESTAMP,
    DOVIZID VARCHAR(5),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
begin
  for select DISTINCT Irsaliye.IrsaliyeId, Irsaliye.FirmaId, Firma.Adi, Irsaliye.SeriNo,
             Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi, Irsaliye.IrsaliyeGrupId, Irsaliye.ReferansEvrakNo,
             Irsaliye.FiiliSevkTarihi, Irsaliye.Notu, Irsaliye.DovizId from StokIrsaliyeRaporu_2(:AHareketTipiId, :ABaslangicTarihi, :ABitisTarihi, :AFirmaId, :AIrsaliyeNo, :AUrunTipiId)
    Left Join Irsaliye on Irsaliye.IrsaliyeId = StokIrsaliyeRaporu_2.IrsaliyeId
    left join Firma on Firma.Firmaid = Irsaliye.Firmaid
    into :IrsaliyeId, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :IrsaliyeGrupId, :ReferansEvrakNo,
         :FiiliSevkTarihi, :Notu, :DovizId do
  begin
    suspend;
  end
end
