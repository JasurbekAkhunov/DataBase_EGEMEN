CREATE PROCEDURE MANUELGIRISCIKISYAP(
    APARTISINIFID VARCHAR(8),
    AMAKINAID VARCHAR(10),
    AKULLANICIID INTEGER)
RETURNS (
    STATUS INTEGER,
    YER VARCHAR(100),
    MESAJ VARCHAR(255))
AS
declare variable ProxyId       varchar(8);
declare variable PersonelId    integer;
begin
  Select ProxyId from PartiStok where PartiStok.PartiSinifId = :APartiSinifId into :ProxyId;
  Select PersonelId from Kullanici where Kullanici.KullaniciId = :AKullaniciId into :PersonelId;
  Select Status, Yer, Mesaj from GirisCikisYap(:ProxyId, :AMakinaId, :PersonelId, 0) into :Status, :Yer, :Mesaj;
  suspend;
end
