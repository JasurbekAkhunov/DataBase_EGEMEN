CREATE PROCEDURE ICTAMIRTAMIRTIPIANALIZI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATAMIRTIPIGRUPID VARCHAR(4),
    ATAMIRTIPIID INTEGER,
    ARAPORTIPI INTEGER,
    ABITTI INTEGER,
    APARTITIPI INTEGER,
    AFIRMAID VARCHAR(20),
    AUCRETDURUMU INTEGER,
    ATAMIRDUZELTMETIPIID VARCHAR(10),
    ATAMIRCOZUMSEKLIID INTEGER,
    AISLEMEALINACAK INTEGER,
    APARTIGRUPID VARCHAR(4),
    ASIPARISGRUPID VARCHAR(4),
    ARECETETURUID INTEGER,
    AHATA1ASAMAID INTEGER,
    AHATA2ASAMAID INTEGER,
    AHATA3ASAMAID INTEGER,
    AONLEYICIFAALIYETTIPIID INTEGER)
RETURNS (
    TAMIRTIPIID INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    TOPLAMKILO NUMERIC(12,3),
    UCRETLITOPLAMKILO NUMERIC(12,3),
    UCRETSIZTOPLAMKILO NUMERIC(12,3),
    TOPLAMMETRE NUMERIC(12,3),
    UCRETLITOPLAMMETRE NUMERIC(12,3),
    UCRETSIZTOPLAMMETRE NUMERIC(12,3),
    ORAN DOUBLE PRECISION,
    ORANMETRE DOUBLE PRECISION)
AS
declare variable FKilo       Numeric(12,3);
declare variable FToplamKilo Numeric(12,3);
declare variable FToplamMetre Numeric(12,3);
declare variable FUcretli    integer;
declare variable FMetre   Numeric(12,3);
begin
  Select Sum(kilo), Sum(metre) from IsletmeIciTamirRaporu(:ABaslangicTarihi, :ABitisTarihi, :ATamirTipiGrupId, :ATamirTipiId, :ARaporTipi, :ABitti, :APartiTipi, :AFirmaId, :AUcretDurumu, :ATamirDuzeltmeTipiId, :ATamirCozumSekliId, :AIslemeAlinacak, :APartiGrupId, :ASiparisGrupId, :AReceteTuruId, :AHata1AsamaId, :AHata2AsamaId, :AHata3AsamaId, :AOnleyiciFaaliyetTipiId) into :FKilo, :FMetre;

  for Select TamirTipiId, TamirTipiAdi, Sum(Kilo), Sum(Metre) from IsletmeIciTamirRaporu(:ABaslangicTarihi, :ABitisTarihi, :ATamirTipiGrupId, :ATamirTipiId, :ARaporTipi, :ABitti, :APartiTipi, :AFirmaId, :AUcretDurumu, :ATamirDuzeltmeTipiId, :ATamirCozumSekliId, :AIslemeAlinacak, :APartiGrupId, :ASiparisGrupId, :AReceteTuruId, :AHata1AsamaId, :AHata2AsamaId, :AHata3AsamaId, :AOnleyiciFaaliyetTipiId)
  group by TamirTipiId, TamirTipiAdi
  into :TamirTipiId, :TamirTipiAdi, :ToplamKilo, :ToplamMetre do
  begin
    if (FKilo > 0) then Oran = ToplamKilo * 100 / FKilo;
    else Oran = Null;

    if (FMetre > 0) then OranMetre = ToplamMetre * 100 / FMetre;
    else OranMetre = Null;

    UcretliToplamKilo = 0;
    UcretsizToplamKilo = 0;

    for Select Ucretli, Sum(Kilo), Sum(Metre) from IsletmeIciTamirRaporu(:ABaslangicTarihi, :ABitisTarihi, :ATamirTipiGrupId, :TamirTipiId, :ARaporTipi, :ABitti, :APartiTipi, :AFirmaId, :AUcretDurumu, :ATamirDuzeltmeTipiId, :ATamirCozumSekliId, :AIslemeAlinacak, :APartiGrupId, :ASiparisGrupId, :AReceteTuruId, :AHata1AsamaId, :AHata2AsamaId, :AHata3AsamaId, :AOnleyiciFaaliyetTipiId)
      group by Ucretli
      into :FUcretli, :FToplamKilo, :FToplamMetre do
    begin
      if (FUcretli = 1) then
      begin
        UcretliToplamKilo = FToplamKilo;
        UcretliToplamMetre = FToplamMetre;
      end
      else
      begin
        UcretsizToplamKilo = FToplamKilo;
        UcretsizToplamMetre = FToplamMetre;
      end
    end

    suspend;
  end
end
