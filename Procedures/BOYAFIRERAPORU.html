          FSiparisFire = FBrutMiktar - FNetMiktar;

          into :FDepoId do
        begin
          if (Char_Length(DepoId) + Char_Length(FDepoId) < 97) then
          begin
           if (DepoId <> '') then DepoId = DepoId || ',';
           DepoId = DepoId || FDepoId;
          end
        end

        for select PartiAsamalari.ReceteId, Asama.Adi from PartiAsamaStok
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          left join Asama on Asama.Asamaid = PartiAsamalari.Asamaid
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1
          order by PartiAsamalari.Sira
          into :FReceteId, :FAsamaAdi do
        begin
          if (FReceteID <> '') then
          begin
           if (Char_Length(Asamalar) + Char_Length(FReceteId) < 1000) then
           begin
             if (Asamalar <> '' ) then Asamalar = Asamalar || cast(', ' as varchar(2) character set UTF8);
             Asamalar = Asamalar || FReceteId;
           end
          end
          else
          begin
           if (Char_Length(Asamalar) + Char_Length(FAsamaAdi) < 1000) then
           begin
             if (Asamalar <> '' ) then Asamalar = Asamalar || cast(', ' as varchar(2) character set UTF8);
             Asamalar = Asamalar || FAsamaAdi;
           end
          end
        end

        KKBrutKilo  =Null;
        KKNetKilo   =Null;
        KKNetMetre  =Null;
        KKBrutMetre =Null;
        if (PartiSinifID is not Null) then
        begin
         Select Sum(PSKaliteKontrol.Kilo), Sum(PSKaliteKontrol.Kilo - PSKaliteKontrol.Dara), Sum(PSKaliteKontrol.Metre), Sum(PSKaliteKontrol.BrutMetre) from PSKaliteKontrolAsama

          into :KKBrutKilo, :KKNetKilo, :KKNetMetre, :KKBrutMetre ;

        end

        IlaveIslemlerAdi = '';
        for Select IlaveIslemler.Adi from PartiAsamaStok

               PartiStok.TamirTipiId, TamirTipi.Adi, PaketlemeSekli.Tup,
               BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaSiparisDetay.IlaveIslemler,
               SipRenkDerinligi.Adi, ReceteRenkDerinligi.Adi, Parti.SevkiyatNotu, PartiStok.IlaveSevkNotu,
               Ham.Adi2, Ham.Adi3, Ham.Adi4, PartiStok.KKesme, PartiStok.ParcaFire, Parti.IslemGrubuID, IslemGrubu.Adi, BoyaSiparisDetay.DesenID, PartiStok.SonCikisMiktar, BoyaSiparisDetay.RenkAdi from BoyaIrsaliye
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
      left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
      left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join Ham on Ham.HamId = PartiStok.HamId
      left join FireSebepTipi on FireSebepTipi.FireSebepTipiId = PartiStok.FireSebepTipiId
      left join Firma IrsFirma on IrsFirma.FirmaId = BoyaIrsaliye.FirmaId
      left join Firma SipFirma on SipFirma.FirmaId = BoyaSiparis.FirmaId
      left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
      Left join Kalite on Kalite.KaliteId = PartiStok.KaliteId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
      left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliId = PartiStok.PaketlemeSekliId
      left join Kullanici FOK on FOK.KullaniciId = PartiStok.FireOnayiKullaniciId
      left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
      left join RenkDerinligi SipRenkDerinligi on SipRenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
      left join RenkDerinligi ReceteRenkDerinligi on ReceteRenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
      left join IslemGrubu on IslemGrubu.IslemGrubuID = Parti.IslemGrubuID
      where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicSevkTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisSevkTarihi and
            BoyaIrsaliyeDetay.PartiKayitId is not null and PartiStok.TamamiTartildi = 1 and
            (IrsFirma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
            (Parti.Fason = :AFason or :AFason is null) and
            (PartiStok.HamId = :AHamId or :AHamId is null) and
             ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :APartiTipiId = 0) or
               (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :APartiTipiId = 1) or
               ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :APartiTipiId = 2) or
               (Ham.HamTipiId = :hstDokumaKumas and :APartiTipiId = 3) or :APartiTipiId is null) and
            (PartiStok.Miktar >= :ABaslangicPartiMiktar and PartiStok.Miktar <= :ABitisPartiMiktar) and
            (PartiStok.FireSebepTipiId = :AFireSebepTipiId or :AFireSebepTipiId is null) and
            (SipFirma.AnaHesapFirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
            (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
            (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
            (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is Null) and
            (Parti.ReceteId = :AReceteId or :AReceteId is null)
      into :IrsaliyeNo, :IrsaliyeTarihi, :FirmaId, :FirmaAdi, :FirmaAdi2, :SiparisFirmaId, :SiparisFirmaAdi, :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :HamAdiToplami,
           :BrutMiktar, :NetMiktar, :BrutKilo, :NetKilo, :FireOrani, :NetSevkMiktar, :BrutSevkMiktar, :NetSevkKilo, :BrutSevkKilo,
           :IslemTipiId, :IslemTipiAdi, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId, :UygunGorulenFireOrani,
           :FFireOnayNotu, :FireSebepTipiId, :FireSebepTipiAdi, :FireOnayiIstensin, :HamNemOrani, :CikisNemOrani, :Fason, :PartiTipiId, :ReceteId, :ReceteAdi,
           :FireOnayiKullaniciId, :FireOnayiKullaniciAdi, :FireOnayTarihi,
           :MusteriSiparisNo, :ReferansSiparisNo, :KaliteId, :KaliteAdi, :TamirTipiId, :TamirTipiAdi, :PaketlemeSekliTup,
           :SiparisProsesId, :SiparisProsesAdi, :SipIlaveIslemler, SiparisRenkDerinligiAdi, :ReceteRenkDerinligiAdi, :PartiSevkiyatNotu, :IlaveSevkNotu,
           :HamAdi2, :HamAdi3, :HamAdi4, :KKesme, :ParcaFire, :IslemGrubuID, :IslemGrubuAdi, :DesenID, :SonCikisMiktar, :RenkAdi do
    begin
      FireOrani = ((BrutMiktar - NetMiktar) / BrutMiktar) * 100;
      FireOrani = Yuvarla(FireOrani, 100);

      if (FireOrani >= 100) then Fire = NetMiktar;
      else Fire = (BrutMiktar - NetMiktar); /*NetMiktar * 100 / (100 - FireOrani) - NetMiktar; FireOranı bir çok kayıtta aradaki farkı gösterir değerde değil. Sorun olabilir!!!*/

      if (AOranTipi is null) then Okey = 1;
      else if (AOranTipi = 0 and FireOrani > UygunGorulenFireOrani) then Okey = 1;
      else if (AOranTipi = 1 and FireOrani < UygunGorulenFireOrani) then Okey = 1;
      else Okey = 0;

      if (AFirmaAdi2yiGorebilir = 0) then FirmaAdi2 = Null;

      if (Okey = 1 and AAsamaId is not null) then
      begin
        KS = 0;
        Select count(*) from PartiStok
        left join PartiAsamaStok on PartiAsamaStok.PartiSinifID = PartiStok.PartiSinifID
        where PartiStok.PartiSinifID = :PartiSinifID and PartiAsamaStok.AsamaId = :AAsamaId and PartiAsamaStok.AsamayaDahil = 1 into :KS;

        if (KS = 0) then Okey = 0;
      end

      if (Okey = 1) then
      begin
        BoyaSiparisIdStr = '';
        if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
        if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
        else PartiIdStr = Null;

        Select sum(Miktar), sum(NetMiktar) from PartiStok
           where PartiStok.BoyaSiparisDetayID = :BoyaSiparisDetayID and PartiStok.TamamiTartildi = 1
           into :FBrutMiktar, :FNetMiktar;


        if (FNetMiktar <> 0 and FBrutMiktar <> 0) then
        begin
          FSiparisFire = FBrutMiktar - FNetMiktar;

          if (FSiparisFire <> 0) then
          begin
             SiparisFireOrani = (FSiparisFire / FBrutMiktar) * 100;
             SiparisFireOrani = Yuvarla(SiparisFireOrani, 100);
          end
          else FireOrani = 0;
        end else SiparisFireOrani = Null;

        if (char_length(FFireOnayNotu) > 1023) then FireOnayNotu = Left(FFireOnayNotu, 1023);
        else FireOnayNotu = Left(FFireOnayNotu, char_length(FFireOnayNotu));

        if (Fason = 0) then FasonStr = 'Satış';
        else FasonStr = 'Fason';

        if (PartiTipiId = 0) then PartiTipiIdStr = 'İplik';
        else PartiTipiIdStr = 'Kumaş';

        RafId = '';
        for select RafDetay.RafId from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          where PartiStokDetay.PartiSinifID = :PartiSinifID
          group by RafDetay.RafId
          into :FRafID do
        begin
          if (Char_Length(RafId) + Char_Length(FRafId) < 97) then
          begin
           if (RafId <> '') then RafId = RafId || ',';
           RafId = RafId || FRafId;
          end
        end
        DepoId = '';

        for select DepoRaf.DepoId from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          left join DepoRaf on DepoRaf.RafId = RafDetay.RafId
          where PartiStokDetay.PartiSinifID = :PartiSinifID
          group by DepoRaf.DepoId
          into :FDepoId do
        begin
          if (Char_Length(DepoId) + Char_Length(FDepoId) < 97) then
          begin
           if (DepoId <> '') then DepoId = DepoId || ',';
           DepoId = DepoId || FDepoId;
          end
        end

        Asamalar = '';
        for select PartiAsamalari.ReceteId, Asama.Adi from PartiAsamaStok
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          left join Asama on Asama.Asamaid = PartiAsamalari.Asamaid
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1
          order by PartiAsamalari.Sira
          into :FReceteId, :FAsamaAdi do
        begin
          if (FReceteID <> '') then
          begin
           if (Char_Length(Asamalar) + Char_Length(FReceteId) < 1000) then
           begin
             if (Asamalar <> '' ) then Asamalar = Asamalar || cast(', ' as varchar(2) character set UTF8);
             Asamalar = Asamalar || FReceteId;
           end
          end
          else
          begin
           if (Char_Length(Asamalar) + Char_Length(FAsamaAdi) < 1000) then
           begin
             if (Asamalar <> '' ) then Asamalar = Asamalar || cast(', ' as varchar(2) character set UTF8);
             Asamalar = Asamalar || FAsamaAdi;
           end
          end
        end
        KKBrutKilo  =Null;
        KKNetKilo   =Null;
        KKNetMetre  =Null;
        KKBrutMetre =Null;
        if (PartiSinifID is not Null) then
        begin
          Select Sum(PSKaliteKontrol.Kilo), Sum(PSKaliteKontrol.Kilo - PSKaliteKontrol.Dara), Sum(PSKaliteKontrol.Metre), Sum(PSKaliteKontrol.BrutMetre) from PSKaliteKontrolAsama
            Left Join PSKaliteKontrol on PSKaliteKontrol.PSKaliteKontrolAsamaID = PSKaliteKontrolAsama.PSKaliteKontrolAsamaID
            where PSKaliteKontrolAsama.PartiSinifId = :PartiSinifId and PSKaliteKontrolAsama.SonKayit = 1 group  by pskalitekontrolasama.partisinifid
          into :KKBrutKilo, :KKNetKilo, :KKNetMetre, :KKBrutMetre;

        end

        IlaveIslemlerAdi = '';

        for Select IlaveIslemler.Adi from PartiAsamaStok
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1 and IlaveIslemler.Adi is not null
          into :FIlaveIslemlerAdi do
        begin
           if (Char_Length(IlaveIslemlerAdi) + Char_Length(FAsamaAdi) < 1000) then
           begin
             if (IlaveIslemlerAdi <> '' ) then IlaveIslemlerAdi = IlaveIslemlerAdi || cast(', ' as varchar(2) character set UTF8);
             IlaveIslemlerAdi = IlaveIslemlerAdi || FIlaveIslemlerAdi;
           end
        end

        HamStokMasterIdStr = '';

        for select HamStokDetay.HamStokMasterId, HamStokMaster.Lot from PartiStokDetay
          left join BoyaSiparisRezerv on BoyaSiparisRezerv.BoyaSiparisRezervId = PartiStokDetay.BoyaSiparisRezervId
          left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
          where PartiStokDetay.PartiSinifId = :PartiSinifId
          group by HamStokDetay.HamStokMasterId, HamStokMaster.Lot
          into :FHamStokMasterId, :FLot do
        begin
          if (Char_Length(HamSTokMasterIdStr) + Char_Length(FHamStokMasterId) < 100) then
          begin
           if (HamSTokMasterIdStr <> '' ) then HamSTokMasterIdStr = HamSTokMasterIdStr || cast(', ' as varchar(2) character set UTF8);
           HamSTokMasterIdStr = HamSTokMasterIdStr || FHamStokMasterId;
          end

          if (FLot is not null) then
            if (PosBul(GirisLot, FLot) = 0) then
              if (Char_Length(GirisLot) + Char_Length(FLot) < 148) then
              begin
                if (GirisLot <> '' ) then GirisLot = GirisLot || cast(', ' as varchar(2) character set UTF8);
                GirisLot = cast(GirisLot as blob character set utf8) || FLot;
              end
        end

        suspend;
      end
    end
