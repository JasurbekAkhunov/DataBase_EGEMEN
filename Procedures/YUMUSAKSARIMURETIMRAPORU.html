CREATE PROCEDURE YUMUSAKSARIMURETIMRAPORU(
    ATARIH TIMESTAMP)
RETURNS (
    IPLIKMAKINAID VARCHAR(10),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIADI VARCHAR(80),
    LOT VARCHAR(30),
    LOTKAPALI INTEGER,
    VARDIYA1KAPADEDI INTEGER,
    VARDIYA1MIKTAR DOUBLE PRECISION,
    VARDIYA2KAPADEDI INTEGER,
    VARDIYA2MIKTAR DOUBLE PRECISION,
    VARDIYA3KAPADEDI INTEGER,
    VARDIYA3MIKTAR DOUBLE PRECISION)
AS
declare variable ABaslangicTarihi  TimeStamp;
declare variable ABitisTarihi  TimeStamp;
declare variable FIplikMakinaID varchar(10);
declare variable FHamId         integer;
declare variable FHamAdi        varchar(255);
declare variable FLot           varchar(30);
declare variable FUretimTarihi  TimeStamp;
declare variable FMiktar        Numeric(12,3);
declare variable FLastIplikMakinaID varchar(10);
declare variable FLastHamId         integer;
declare variable FLastLot           varchar(30);
declare variable FLotKapali         integer;
declare variable FKalinlikAdi       varchar(50);
declare variable FIplikCinsiAdi     varchar(80);
declare variable Vardiya           integer;
declare variable VardiyaSayisi     integer;
declare variable T1                Double Precision;
begin
  ABaslangicTarihi = ATarih;
  ABitisTarihi = ABaslangicTarihi + 1;

  Select IplikVardiyaSayisi from sistem where sistemid = 1 into :VardiyaSayisi;

  if (VardiyaSayisi = 0 or VardiyaSayisi is null) then VardiyaSayisi = 3;

  FLastIplikMakinaID = '@@';
  FLastHamId = -1;
  FLastLot = '@@';

  For Select IplikUretim.IplikMakinaId, IplikLot.HamId, Ham.Adi, IplikLot.Lot, IplikLot.Kapali,
             Kalinlik.Adi, IplikCinsi.Adi, IplikUretim.UretimTarihi, IplikUretim.Miktar from IplikUretim
  left join IplikLot On IplikLot.IplikLotID = IplikUretim.IplikLotId
  left join Ham On Ham.HamID = IplikLot.HamId
  left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
  left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
  where IplikUretim.UretimTarihi >= :ABaslangicTarihi and IplikUretim.UretimTarihi <= :ABitisTarihi and
        IplikUretim.SarimSekliID = 1
  order by IplikUretim.IplikMakinaId, Kalinlik.Adi, IplikCinsi.Adi, IplikLot.Lot
  into :FIplikMakinaId, :FHamId, :FHamAdi, :FLot, FLotKapali, :FKalinlikAdi, :FIplikCinsiAdi, :FUretimTarihi, :FMiktar do
  begin
    if (FLastIplikMakinaId <> FIplikMakinaId or FLastHamId <> FHamId or FLastLot <> FLot) then
    begin
      if (FLastHamId <> -1) then suspend;

      FLastIplikMakinaID = FIplikMakinaId;
      FLastHamId = FHamId;
      FLastLot = FLot;

      IplikMakinaId = FIplikMakinaId;
      HamId = FHamId;
      HamAdi = FHamAdi;
      Lot = FLot;
      LotKapali = FLotKapali;
      KalinlikAdi = FKalinlikAdi;
      IplikCinsiAdi = FIplikCinsiAdi;

      Vardiya1KapAdedi = 0;
      Vardiya1Miktar = 0;
      Vardiya2KapAdedi = 0;
      Vardiya2Miktar = 0;
      Vardiya3KapAdedi = 0;
      Vardiya3Miktar = 0;
    end

    T1 = FUretimTarihi - ABaslangicTarihi;
    vardiya = VardiyaHesapla(T1, VardiyaSayisi);

    if (Vardiya = 1) then
    begin
      Vardiya1KapAdedi = Vardiya1KapAdedi + 1;
      Vardiya1Miktar = Vardiya1Miktar + FMiktar;
    end
    else if (Vardiya = 2) then
    begin
      Vardiya2KapAdedi = Vardiya2KapAdedi + 1;
      Vardiya2Miktar = Vardiya2Miktar + FMiktar;
    end
    else if (Vardiya >= 3) then
    begin
      Vardiya3KapAdedi = Vardiya3KapAdedi + 1;
      Vardiya3Miktar = Vardiya3Miktar + FMiktar;
    end
  end
  if (FLastHamId <> -1) then suspend;
end
