CREATE PROCEDURE PARTIARAKONTROLRAPORUKAGIT(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AASAMAID INTEGER,
    AHAMID INTEGER,
    AFIRMAID VARCHAR(20),
    AONAYDURUMU INTEGER)
RETURNS (
    PARTIASAMALARIID INTEGER,
    PARTISINIFID INTEGER,
    ARAKONTROLTIPIID INTEGER,
    SONUC VARCHAR(50),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(4),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANPERSONELADI VARCHAR(40))
AS
begin
  For Select PartiAraKontrolRaporu.PartiAsamalariId, PartiAraKontrolRaporu.PartiSinifId from PartiAraKontrolRaporu(:ABaslangicTarihi, :ABitisTarihi, :AAsamaId, :AHamId, :AFirmaId, :AOnayDurumu)
    into :PartiAsamalariId, :PartiSinifId do
  begin
    For Select AsamaAraKontrolTipi.AraKontrolTipiId, PartiAsamaAraKontrol.Sonuc, PartiAsamaAraKontrol.OnayDurumu, PartiAsamaAraKontrol.Onaytarihi, Personel.Adi from AsamaAraKontrolTipi
      left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId
      left join PartiAsamaAraKontrol on PartiAsamaAraKontrol.PartiSinifId = :PartiSinifId and PartiAsamaAraKontrol.PartiAsamalariId = :PartiAsamalariId and PartiAsamaAraKontrol.AraKontrolTipiId = AsamaAraKontrolTipi.AraKontrolTipiId
      Left Join Personel on Personel.PersonelId = PartiAsamaAraKontrol.OnaylayanPersonelId
      where AsamaAraKontrolTipi.AsamaId = :AAsamaId and
            (PartiAsamaAraKontrol.OnayDurumu = :AOnayDurumu + 1 or :AOnayDurumu is null) and
            PartiAsamaStok.CikisTarihi >= :ABaslangicTarihi and PartiAsamaStok.CikisTarihi <= :ABitisTarihi
      into :AraKontrolTipiId, :Sonuc, :OnayDurumu, :Onaytarihi, :OnaylayanPersonelAdi do
    begin
      if (OnayDurumu = 1) then OnayDurumuStr = 'Okey';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';
      suspend;
    end
  end
end
