CREATE PROCEDURE MC_SONKONTROLRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    AFIRMAID VARCHAR(20),
    AKULLANICIID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    PARTIASAMALARIID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KALITEID VARCHAR(20),
    KALITEADI ADI255,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    FKONTROL1KONTROLDURUMU INTEGER,
    FKONTROL1KONTROLDURUMUSTR VARCHAR(4),
    FKONTROL1KONTROLTARIHI TIMESTAMP,
    FKONTROL1KONTROLKULLANICIID INTEGER,
    FKONTROL1KKULLANICIADI VARCHAR(40),
    FKONTROL1ONAYDURUMU INTEGER,
    FKONTROL1ONAYDURUMUSTR VARCHAR(4),
    FKONTROL1ONAYTARIHI TIMESTAMP,
    FKONTROL1ONAYKULLANICIID INTEGER,
    FKONTROL1ONAYKULLANICIADI VARCHAR(40),
    FKONTROL2KONTROLDURUMU INTEGER,
    FKONTROL2KONTROLDURUMUSTR VARCHAR(4),
    FKONTROL2KONTROLTARIHI TIMESTAMP,
    FKONTROL2KONTROLKULLANICIID INTEGER,
    FKONTROL2KKULLANICIADI VARCHAR(40),
    FKONTROL2ONAYDURUMU INTEGER,
    FKONTROL2ONAYDURUMUSTR VARCHAR(4),
    FKONTROL2ONAYTARIHI TIMESTAMP,
    FKONTROL2ONAYKULLANICIID INTEGER,
    FKONTROL2ONAYKULLANICIADI VARCHAR(40),
    FIZIKSELKONTROLNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    FIZIKSELKONTROLNOTU2 BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SONKONTROLONAYDURUMU INTEGER,
    SONKONTROLONAYDURUMUSTR VARCHAR(4),
    SONKONTROLONAYKULLANICIADI VARCHAR(40),
    SONKONTROLONAYTARIHI TIMESTAMP)
AS
declare variable Okey     integer;
declare variable APartiKayitId integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  if (APartiId is not null) then
  begin
    Select PartiKayitId from Parti
       where Parti.PartiYil = :APartiYil and Parti.PartiId = :APartiId into :APartiKayitId;
  end
  else APartiKayitId = Null;

  for Select PartiStokFizikKontrol.PartiKayitId, PartiStokFizikKontrol.PartiAsamalariId, Parti.PartiYil, Parti.PartiId, PartiStokFizikKontrol.PartiSinifId, PartiStok.BoyaSiparisDetayId,
              PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.Metre,
              PartiStokFizikKontrol.FKontrol1KontrolDurumu, PartiStokFizikKontrol.FKontrol1KontrolTarihi, PartiStokFizikKontrol.FKontrol1KontrolKullaniciId, FK1KK.Adi,
              PartiStokFizikKontrol.FKontrol1OnayDurumu, PartiStokFizikKontrol.FKontrol1OnayTarihi, PartiStokFizikKontrol.FKontrol1OnayKullaniciId, FK1OK.Adi,
              PartiStokFizikKontrol.FKontrol2KontrolDurumu, PartiStokFizikKontrol.FKontrol2KontrolTarihi, PartiStokFizikKontrol.FKontrol2KontrolKullaniciId, FK2KK.Adi,
              PartiStokFizikKontrol.FKontrol2OnayDurumu, PartiStokFizikKontrol.FKontrol2OnayTarihi, PartiStokFizikKontrol.FKontrol2OnayKullaniciId, FK2OK.Adi,
              PartiStokFizikKontrol.FizikselKontrolNotu, PartiStokFizikKontrol.FizikselKontrolNotu2,
              PartiStokSonKontrol.SonKontrolOnayDurumu, SKOK.Adi, PartiStokSonKontrol.SonKontrolOnayTarihi,
              Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
              PartiStok.KaliteId, Kalite.KaliteAdi, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.MusteriRenkKoduveAdi,
              BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi from PartiStokFizikKontrol
    left join Parti On Parti.PartiKayitId = PartiStokFizikKontrol.PartiKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join PartiStok on PartiStok.PartiSinifId = PartiStokFizikKontrol.PartiSinifId
    left join Kalite on Kalite.KaliteId = PartiStok.KaliteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Kullanici FK1KK  On FK1KK.KullaniciId = PartiStokFizikKontrol.FKontrol1KontrolKullaniciId
    left join Kullanici FK2KK  On FK2KK.KullaniciId = PartiStokFizikKontrol.FKontrol2KontrolKullaniciId
    left join Kullanici FK1OK  On FK1OK.KullaniciId = PartiStokFizikKontrol.FKontrol1OnayKullaniciId
    left join Kullanici FK2OK  On FK2OK.KullaniciId = PartiStokFizikKontrol.FKontrol2OnayKullaniciId
    left join PartiStokSonKontrol On PartiStokSonKontrol.PartiSinifId = PartiStokFizikKontrol.PartiSinifId and PartiStokSonKontrol.SonKayit = 1
    left join Kullanici SKOK  On SKOK.KullaniciId = PartiStokSonKontrol.SonKontrolOnayKullaniciId
    where ((PartiStokFizikKontrol.FKontrol1OnayTarihi >= :ABaslangicTarihi and PartiStokFizikKontrol.FKontrol1OnayTarihi <= :ABitisTarihi and
            PartiStokFizikKontrol.SonKayit = 1 and :APartiKayitId is null) or
          (:APartiKayitId is not null and PartiStokFizikKontrol.PartiKayitId = :APartiKayitId)) and
          Parti.FirmaID = :AFirmaId
    into :PartiKayitId, :PartiAsamalariId, :PartiYil, :PartiId, :PartiSinifId, :BoyaSiparisDetayId, :HamAdiToplami, :Kilo, :Metre,
         :FKontrol1KontrolDurumu, :FKontrol1KontrolTarihi, :FKontrol1KontrolKullaniciId, :FKontrol1KKullaniciAdi,
         :FKontrol1OnayDurumu, :FKontrol1OnayTarihi, :FKontrol1OnayKullaniciId, :FKontrol1OnayKullaniciAdi,
         :FKontrol2KontrolDurumu, :FKontrol2KontrolTarihi, :FKontrol2KontrolKullaniciId, :FKontrol2KKullaniciAdi,
         :FKontrol2OnayDurumu, :FKontrol2OnayTarihi, :FKontrol2OnayKullaniciId, :FKontrol2OnayKullaniciAdi,
         :FizikselKontrolNotu, :FizikselKontrolNotu2,
         :SonKontrolOnayDurumu, :SonKontrolOnayKullaniciAdi, :SonKontrolOnayTarihi,
         :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId,
         :KaliteId, :KaliteAdi, :MusteriSiparisNo, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    if (BoyaSiparisYil is null) then BoyaSiparisIdStr = '';
    else BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    FKontrol1KontrolDurumuStr = '';
    if (FKontrol1KontrolDurumu = 1) then FKontrol1KontrolDurumuStr = 'Onay';
    else if (FKontrol1KontrolDurumu = 2) then FKontrol1KontrolDurumuStr = 'Ret';

    FKontrol1OnayDurumuStr = '';
    if (FKontrol1OnayDurumu = 1) then FKontrol1OnayDurumuStr = 'Onay';
    else if (FKontrol1OnayDurumu = 2) then FKontrol1OnayDurumuStr = 'Ret';

    FKontrol2KontrolDurumuStr = '';
    if (FKontrol2KontrolDurumu = 2) then FKontrol2KontrolDurumuStr = 'Onay';
    else if (FKontrol2KontrolDurumu = 2) then FKontrol2KontrolDurumuStr = 'Ret';

    FKontrol2OnayDurumuStr = '';
    if (FKontrol2OnayDurumu = 2) then FKontrol2OnayDurumuStr = 'Onay';
    else if (FKontrol2OnayDurumu = 2) then FKontrol2OnayDurumuStr = 'Ret';

    SonKontrolOnayDurumuStr = '';
    if (SonKontrolOnayDurumu = 1) then SonKontrolOnayDurumuStr = 'Onay';
    else if (SonKontrolOnayDurumu = 2) then SonKontrolOnayDurumuStr = 'Ret';

    SonKontrolOnayDurumuStr = '';
    if (SonKontrolOnayDurumu = 1) then SonKontrolOnayDurumuStr = 'Onay';
    else if (SonKontrolOnayDurumu = 2) then SonKontrolOnayDurumuStr = 'Ret';
    suspend;
  end
end
