CREATE PROCEDURE PSKALITEKONTROLASAMAGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PSKALITEKONTROLASAMAID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTISINIFID INTEGER,
    SIRA INTEGER,
    KKBASLAMATARIHI TIMESTAMP,
    KKBASLATANPERSONELID INTEGER,
    BASLATANPERSONELADI VARCHAR(40),
    KKBITISTARIHI TIMESTAMP,
    KKBITIRENPERSONELID INTEGER,
    BITIRENPERSONELADI VARCHAR(40),
    KKNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SONKAYIT INTEGER,
    TARTIYAAKTARILDI INTEGER,
    TOPLAMKILO DOUBLE PRECISION,
    TOPLAMNETKILO DOUBLE PRECISION,
    TOPLAMMETRE DOUBLE PRECISION,
    REDTOPLAMKILO DOUBLE PRECISION,
    REDTOPLAMMETRE DOUBLE PRECISION,
    REDTOPLAMNETKILO DOUBLE PRECISION,
    AKTARMAKONTROLSTR VARCHAR(20))
AS
begin
  For Select PSKaliteKontrolAsama.PSKaliteKontrolAsamaId, PSKaliteKontrolAsama.PartiKayitId, PSKaliteKontrolAsama.PartiAsamalariId, PSKaliteKontrolAsama.PartiSinifId,
             PartiAsamalari.Sira, PSKaliteKontrolAsama.KKBaslamaTarihi, PSKaliteKontrolAsama.KKBaslatanPersonelId, PSKaliteKontrolAsama.KKBitisTarihi,
             PSKaliteKontrolAsama.KKBitirenPersonelId, PSKaliteKontrolAsama.KKNotu, PSKaliteKontrolAsama.SonKayit, PSKaliteKontrolAsama.TartiyaAktarildi,
             BASP.Adi, BITP.ADI,
             Sum(iif(PsKaliteKontrol.OnayDurumu <> 2, PsKaliteKontrol.Kilo, 0)), Sum(iif(PsKaliteKontrol.OnayDurumu <> 2, (PsKaliteKontrol.Kilo-PsKaliteKontrol.Dara), 0)), Sum(iif(PsKaliteKontrol.OnayDurumu <> 2, PsKaliteKontrol.Metre, 0)),
             Sum(iif(PsKaliteKontrol.OnayDurumu = 2, PsKaliteKontrol.Kilo, 0)), Sum(iif(PsKaliteKontrol.OnayDurumu = 2, (PsKaliteKontrol.Kilo-PsKaliteKontrol.Dara), 0)), Sum(iif(PsKaliteKontrol.OnayDurumu = 2, PsKaliteKontrol.Metre, 0)),
             case when coalesce(PSKaliteKontrolAsama.AKTARMAKONTROL,0)=0 then 'KALİTE KONTROL' ELSE 'AKTARMA KONTROL' END AS KONTROLADI
             FROM PSKaliteKontrolAsama
  Left Join PERSONEL BASP ON BASP.PERSONELID = PSKALITEKONTROLASAMA.KKBASLATANPERSONELID
  Left Join PERSONEL BITP ON BITP.PERSONELID = PSKALITEKONTROLASAMA.KKBITIRENPERSONELID
  Left Join PARTIASAMALARI On PARTIASAMALARI.PARTIASAMALARIID = PSKALITEKONTROLASAMA.PARTIASAMALARIID
  Left Join PSKALITEKONTROL On PSKALITEKONTROL.PSKALITEKONTROLASAMAID = PSKALITEKONTROLASAMA.PSKALITEKONTROLASAMAID
  Where PSKaliteKontrolAsama.PartiKayitId = :APartiKayitId
  Group By PSKaliteKontrolAsama.PSKaliteKontrolAsamaId, PSKaliteKontrolAsama.PartiKayitId, PSKaliteKontrolAsama.PartiAsamalariId, PSKaliteKontrolAsama.PartiSinifId,
             PartiAsamalari.Sira, PSKaliteKontrolAsama.KKBaslamaTarihi, PSKaliteKontrolAsama.KKBaslatanPersonelId, PSKaliteKontrolAsama.KKBitisTarihi,
             PSKaliteKontrolAsama.KKBitirenPersonelId, PSKaliteKontrolAsama.KKNotu, PSKaliteKontrolAsama.SonKayit, PSKaliteKontrolAsama.TartiyaAktarildi,
             BASP.Adi, BITP.ADI, case when coalesce(PSKaliteKontrolAsama.AKTARMAKONTROL,0)=0 then 'KALİTE KONTROL' ELSE 'AKTARMA KONTROL' END
  Order By PartiAsamalari.Sira
  into :PSKaliteKontrolAsamaId, :PartiKayitId, :PartiAsamalariId, :PartiSinifId, :Sira, :KKBaslamaTarihi, :KKBaslatanPersonelId,
       :KKBitisTarihi, :KKBitirenPersonelId, :KKNotu, :SonKayit, :TartiyaAktarildi, :BaslatanPersonelAdi, :BitirenPersonelAdi,
       :ToplamKilo, :ToplamNetKilo, :ToplamMetre, :RedToplamKilo, :RedToplamNetKilo, :RedToplamMetre , :AktarmaKontrolStr
  do suspend;
end
