CREATE PROCEDURE HAMDEPOVEISLETMESTOKRAPORU(
    ASTOKFIRMAID VARCHAR(20),
    ASIPARISFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATAMIR INTEGER,
    AFASON INTEGER,
    ARAPORHAMTIPI INTEGER,
    AHAMID INTEGER)
RETURNS (
    PARTIIDSTR VARCHAR(30),
    STOKFIRMAID VARCHAR(20),
    STOKFIRMAADI VARCHAR(100),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    MUSTERIRENKKODUVEADI VARCHAR(30),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    RECETEID VARCHAR(20),
    RENKADI VARCHAR(40),
    MELANJKODU VARCHAR(20),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    HAMTIPIID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    TAMIRMI VARCHAR(10),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    DURUMUSTR VARCHAR(20))
AS
declare variable IslemTipiId integer;
begin
  for Select StokFirmaId, StokFirmaAdi, SiparisFirmaId, SiparisFirmaAdi, BoyaSiparisYil, BoyaSiparisId, BoyaSiparisIdStr, BoyaSiparisDetayId, Fason, FasonStr, HamTipiId,
             HamId, HamAdi, Tamirmi, MusteriRenkKoduveAdi, LabReceteKodu, labReceteAdi, MelanjKodu, Sum(KalanKilo), Sum(KalanMetre) from HamDepoveIsletmeStokRaporu_1(:AStokFirmaId, :ASiparisFirmaId, :ABaslangicTarihi, :ABitisTarihi, :ATamir, :AFason, :ARaporHamTipi, :AHamId)
     Group by StokFirmaId, StokFirmaAdi, SiparisFirmaId, SiparisFirmaAdi, BoyaSiparisYil, BoyaSiparisId, BoyaSiparisIdStr, BoyaSiparisDetayId, Fason, FasonStr, HamTipiId,
             HamId, HamAdi, Tamirmi, MusteriRenkKoduveAdi, LabReceteKodu, labReceteAdi, MelanjKodu
     into :StokFirmaId, :StokFirmaAdi, :SiparisFirmaId, :SiparisFirmaAdi, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisIdStr, :BoyaSiparisDetayId, :Fason, :FasonStr, :HamTipiId,
           :HamId, :HamAdi, :Tamirmi, :MusteriRenkKoduveAdi, :LabReceteKodu, :labReceteAdi, :MelanjKodu, :Kilo, :Metre do suspend;

  LabReceteKodu = Null;
  LabReceteAdi = NUll;

  For Select PartiIdStr, FirmaId, FirmaAdi, BoyaSiparisYil, BoyaSiparisId, BoyaSiparisIdStr, BoyaSiparisDetayId, Fason, FasonStr, HamTipiId, HamId, HamAdiToplami, IslemTipiId, Kilo - BrutSevkKilo, Metre - BrutSevkMetre, DurumuStr,

             MusteriRenkKoduveAdi, ReceteId, RenkAdi, MelanjKodu, NetKilo, SiparisProsesId, SiparisProsesAdi from IsletmedekiPartilerinListesi(0, null, null, null, :ASiparisFirmaId, null, :ARaporHamTipi, Null, :AHamId, :ATamir, Null, Null, Null, null, null, null, null, null, null, Null, null, null, null, null, null, 1, null, null, null, Null, null, null, null, :AFason, null, 0, 0, null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, null, null, null, null, null)
   where IsletmedekiPartilerinListesi.Fason = :AFason and
         (IsletmedekiPartilerinListesi.FirmaID = :AStokFirmaID or :AStokFirmaID is null)
  into :PartiIdStr, :SiparisFirmaId, :SiparisFirmaAdi, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisIdStr, :BoyaSiparisDetayId, :Fason, :FasonStr, :HamTipiId, :HamId, :HamAdi, :IslemTipiId, :Kilo, :Metre, :DurumuStr,
       :MusteriRenkKoduveAdi, :ReceteId, :RenkAdi, :MelanjKodu, :NetKilo, :SiparisProsesId, :SiparisProsesAdi do
  begin
    if (IslemTipiId <> 3) then Tamirmi = null;
    else Tamirmi = 'Tamir';

    if (Fason = 0) then
    begin
      StokFirmaId = Null;
      StokFirmaAdi = Null;
    end
    else
    begin
      StokFirmaId = SiparisFirmaId;
      StokFirmaAdi = SiparisFirmaAdi;
    end

    suspend;
  end
end
