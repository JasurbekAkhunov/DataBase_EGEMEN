CREATE PROCEDURE URUNSARFIYATRAPORUDETAYLI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICURUNGRUP1ID VARCHAR(4),
    ABITISURUNGRUP1ID VARCHAR(4),
    ABASLANGICURUNGRUP2ID VARCHAR(4),
    ABITISURUNGRUP2ID VARCHAR(4),
    ABASLANGICURUNGRUP3ID VARCHAR(4),
    ABITISURUNGRUP3ID VARCHAR(4),
    ABASLANGICURUNGRUP4ID VARCHAR(4),
    ABITISURUNGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    AHAREKETIOLMAYANLARIDOKSUNMU INTEGER,
    ARAPORDOVIZID VARCHAR(5),
    ASINIF VARCHAR(1),
    AURUNTIPIID INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    KIMYASALTIPIID INTEGER,
    KIMYASALTIPIADI VARCHAR(20),
    BIRIMID VARCHAR(2),
    DEVREDENSTOKMIKTARI DOUBLE PRECISION,
    GIRISTOPLAMI DOUBLE PRECISION,
    OTOMATIKURETIMECIKIS DOUBLE PRECISION,
    MANUELURETIMECIKIS DOUBLE PRECISION,
    UCRETSIZTAMIRECIKIS DOUBLE PRECISION,
    URUNURETIMECIKIS DOUBLE PRECISION,
    TOPLAMURETIMECIKIS DOUBLE PRECISION,
    KALANSTOK DOUBLE PRECISION,
    URETIMECIKISFIYATI DOUBLE PRECISION,
    RAPORDOVIZID VARCHAR(5),
    OTOMATIKURETIMECIKISMALIYETI DOUBLE PRECISION,
    MANUELURETIMECIKISMALIYETI DOUBLE PRECISION,
    UCRETSIZTAMIRECIKISMALIYETI DOUBLE PRECISION,
    URUNURETIMECIKISMALIYETI DOUBLE PRECISION,
    SH_OTOMATIKURETIMECIKISMALIYETI DOUBLE PRECISION,
    SH_MANUELURETIMECIKISMALIYETI DOUBLE PRECISION,
    SH_UCRETSIZTAMIRECIKISMALIYETI DOUBLE PRECISION,
    SH_URUNURETIMECIKISMALIYETI DOUBLE PRECISION,
    URUNTIPIID INTEGER,
    URUNTIPIADI VARCHAR(20),
    URUNGRUP1ADI VARCHAR(30),
    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30),
    ORMEMAKINAIDLER VARCHAR(100))
AS
declare variable FDefaultBirimCarpan double precision;
declare variable FOtomasyon integer;
declare variable FFaturaEdilecek integer;
declare variable FUretimeCikisMiktari Double Precision;
declare variable okey integer;
declare variable UretimeCikisFiyatiDovizId     varchar(5);
declare variable LastDovizId   varchar(5);
declare variable StokTabanDovizId   varchar(5);
declare variable ARaporTarihi           Date;
declare variable DovizKuru Double Precision;
declare variable TabanDovizKuru Double Precision;
declare variable FMaliyet Double Precision;
declare variable FOrmeMakinaId varchar(10);
begin
  ARaportarihi = Current_TimeStamp(0);
  DevredenStokMiktari = 0.0;
  LastDovizId = '';
  RaporDovizID = ARaporDovizId;

  Select StokTabanDovizId from Sistem where sistemid = 1 into :StokTabanDovizId;

  if (StokTabanDovizId <> ARaporDovizId) then
    select DovizKuru from EnYakinDovizKuruGetir(:StokTabanDovizId, :ARaporDovizId, :ARaporTarihi) into :TabanDovizKuru;
  else
    TabanDovizKuru = 1;

  for Select Urun.UrunId, Urun.Adi, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, UBD.BirimId, UBD.Carpan,
             UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / UBD.Carpan,  UrunFiyat.UretimeCikisFiyatiDovizId,
             Urun.KimyasalTipiId, KimyasalTipi.Adi, Urun.UrunTipiId, UrunTipi.Adi,
             UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi from Urun
      left join UrunBirim UBD on UBD.UrunId = Urun.UrunId and UBD.DefaultBirim = 1
      left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
      left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
      left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = urun.KimyasalTipiid
      left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
      left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
      left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
      left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
      left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
  where ((:AUrunAralikli = 1 and Urun.UrunId >= :ABaslangicUrunId and (Urun.UrunId <= :ABitisUrunId or :ABitisUrunId = '')) or (:AUrunAralikli = 0))  and
        ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicUrunGrup1Id  and (UrunGrup1 <= :ABitisUrunGrup1Id or :ABitisUrunGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicUrunGrup1Id = '' and :ABitisUrunGrup1Id = '')) and
        ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicUrunGrup2Id and (UrunGrup2 <= :ABitisUrunGrup2Id or :ABitisUrunGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicUrunGrup2Id = '' and :ABitisUrunGrup2Id = '')) and
        ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicUrunGrup3Id and (UrunGrup3 <= :ABitisUrunGrup3Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicUrunGrup3Id = '' and :ABitisUrunGrup3Id = '')) and
        ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicUrunGrup4Id and (UrunGrup4 <= :ABitisUrunGrup4Id or :ABitisUrunGrup4Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicUrunGrup4Id = '' and :ABitisUrunGrup4Id = '')) and
        (Urun.UrunTipiId = :AUrunTipiId or :AUrunTipiId is null)
  into :UrunId, :UrunAdi, :UrunGrup1,:UrunGrup2,:UrunGrup3, :UrunGrup4, :BirimId, :FDefaultBirimCarpan, :UretimeCikisFiyati, :UretimeCikisFiyatiDovizId,
       :KimyasalTipiId, :KimyasalTipiAdi,
       :UrunTipiId, :UrunTipiAdi, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi do
  begin
    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup4Id, UrunGrup4) = 0) then okey = 0;

    if (okey = 1) then
    begin
      /*select sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from stokdetay
        left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
        where StokDetay.UrunId = :UrunId and StokDetay.Tarih < :ABaslangicTarihi and StokDetay.HareketTipi <> 5 and
              (StokDetay.Sinif = :ASinif or :ASinif is null)
        into :DevredenStokMiktari;


      select sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from stokdetay
        left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
        where StokDetay.UrunId = :UrunId and StokDetay.Tarih >= :ABaslangicTarihi and StokDetay.Tarih <= :ABitisTarihi and
              (StokDetay.HareketTipi = 1 or StokDetay.HareketTipi = 3) and
              (StokDetay.Sinif = :ASinif or :ASinif is null)
        into :GirisToplami;*/

      DevredenStokMiktari = Null;
      GirisToplami = Null;

      OtomatikUretimeCikis = 0;
      ManuelUretimeCikis = 0;
      UcretsizTamireCikis = 0;
      SH_OtomatikUretimeCikisMaliyeti = 0;
      SH_ManuelUretimeCikisMaliyeti   = 0;
      SH_UcretsizTamireCikisMaliyeti  = 0;
      SH_UrunUretimeCikisMaliyeti     = 0;

      for select StokDetay.Otomasyon, Parti.FaturaKesilecek, sum(StokDetay.Miktar / UrunBirim.Carpan), sum(StokDetay.Miktar * StokDetay.BirimFiyat) from stokdetay
        left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
        left join Parti on Parti.PartiKayitId = StokDetay.PartiKayitId
        where StokDetay.UrunId = :UrunId and StokDetay.UretimeCikisTarihi >= :ABaslangicTarihi and StokDetay.UretimeCikisTarihi <= :ABitisTarihi and
              StokDetay.HareketTipi = 6 and StokDetay.PartiKayitId is not null and
              (StokDetay.Sinif = :ASinif or :ASinif is null) and
              (Parti.FirmaID = :AFirmaID or :AFirmaID is null)
        group by StokDetay.Otomasyon, Parti.FaturaKesilecek
        into :FOtomasyon, :FFaturaEdilecek, :FUretimeCikisMiktari, :FMaliyet  do
      begin
         if (FFaturaEdilecek = 0) then
         begin
           UcretsizTamireCikis = UcretsizTamireCikis + FUretimeCikisMiktari;
           SH_UcretsizTamireCikisMaliyeti = SH_UcretsizTamireCikisMaliyeti + FMaliyet;
         end
         else if (FOtomasyon <> 0) then
         begin
           OtomatikUretimeCikis = OtomatikUretimeCikis + FUretimeCikisMiktari;
           SH_OtomatikUretimeCikisMaliyeti = SH_OtomatikUretimeCikisMaliyeti + FMaliyet;
         end
         else
         begin
           ManuelUretimeCikis = ManuelUretimeCikis + FUretimeCikisMiktari;
           SH_ManuelUretimeCikisMaliyeti = SH_ManuelUretimeCikisMaliyeti + FMaliyet;
         end
      end

      if (AFirmaID is null) then
      begin
        select sum(StokDetay.Miktar / UrunBirim.Carpan), sum(StokDetay.Miktar * StokDetay.BirimFiyat) from stokdetay
          left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
          where StokDetay.UrunId = :UrunId and StokDetay.UretimeCikisTarihi >= :ABaslangicTarihi and StokDetay.UretimeCikisTarihi <= :ABitisTarihi and
                StokDetay.HareketTipi = 6 and StokDetay.PartiKayitId is null and
                (StokDetay.Sinif = :ASinif or :ASinif is null)
          into :UrunUretimeCikis, :SH_UrunUretimeCikisMaliyeti;
      end

      if (OtomatikUretimeCikis is null) then OtomatikUretimeCikis = 0;
      if (ManuelUretimeCikis is null) then ManuelUretimeCikis = 0;
      if (UcretsizTamireCikis is null) then UcretsizTamireCikis = 0;
      if (UrunUretimeCikis is null) then UrunUretimeCikis = 0;

      OtomatikUretimeCikis = OtomatikUretimeCikis * FDefaultBirimCarpan;
      ManuelUretimeCikis = ManuelUretimeCikis * FDefaultBirimCarpan;
      UrunUretimeCikis = UrunUretimeCikis * FDefaultBirimCarpan;
      UcretsizTamireCikis = UcretsizTamireCikis * FDefaultBirimCarpan;

      ToplamUretimeCikis = ManuelUretimeCikis + OtomatikUretimeCikis + UcretsizTamireCikis + UrunUretimeCikis;

      OrmeMakinaIDler = '';
      for select StokDetay.OrmeMakinaID from StokDetay
        left join Parti on Parti.PartiKayitId = StokDetay.PartiKayitId
        where StokDetay.UrunId = :UrunId and StokDetay.UretimeCikisTarihi >= :ABaslangicTarihi and StokDetay.UretimeCikisTarihi <= :ABitisTarihi and
              StokDetay.HareketTipi = 6 and StokDetay.PartiKayitId is not null and
              (StokDetay.Sinif = :ASinif or :ASinif is null) and
              (Parti.FirmaID = :AFirmaID or :AFirmaID is null) and StokDetay.OrmeMakinaID is not Null
        group by StokDetay.OrmeMakinaID
        into :FOrmeMakinaID  do
      begin
        if (Char_Length(FOrmeMakinaID) < 90) then
        begin
          if (OrmeMakinaIDler <> '' ) then OrmeMakinaIDler = OrmeMakinaIDler || cast(', ' as varchar(2) character set UTF8);
          OrmeMakinaIDler = OrmeMakinaIDler || FOrmeMakinaID;
        end
      end

      if (AHareketiOlmayanlariDoksunmu = 0) then
      begin
        if (GirisToplami = 0 and ToplamUretimeCikis = 0) then Okey = 0;
      end
      if (Okey = 1) then
      begin
        if (UretimeCikisFiyatiDovizId is null) then UretimeCikisFiyati = 0;
        else if (LastDovizId <> UretimeCikisFiyatiDovizId) then
        begin
          if (UretimeCikisFiyatiDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:UretimeCikisFiyatiDovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
              else DovizKuru = 1;
          if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;
          LastDovizId = UretimeCikisFiyatiDovizId;
        end

        UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;

        OtomatikUretimeCikisMaliyeti = OtomatikUretimeCikis * UretimeCikisFiyati;
        ManuelUretimeCikisMaliyeti = ManuelUretimeCikis * UretimeCikisFiyati;
        UrunUretimeCikisMaliyeti = UrunUretimeCikis * UretimeCikisFiyati;
        UcretsizTamireCikisMaliyeti = UcretsizTamireCikis * UretimeCikisFiyati;

        /*select sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan)   from stokdetay
          left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
          where StokDetay.UrunId = :UrunId and StokDetay.UretimeCikisTarihi <= :ABitisTarihi and StokDetay.HareketTipi <> 5 and
                (StokDetay.Sinif = :ASinif or :ASinif is null)
          into :KalanStok;*/
        KalanStok = Null;

        SH_OtomatikUretimeCikisMaliyeti  =  SH_OtomatikUretimeCikisMaliyeti * TabanDovizKuru;
        SH_ManuelUretimeCikisMaliyeti    =  SH_ManuelUretimeCikisMaliyeti * TabanDovizKuru;
        SH_UcretsizTamireCikisMaliyeti   =  SH_UcretsizTamireCikisMaliyeti * TabanDovizKuru;
        SH_UrunUretimeCikisMaliyeti      =  SH_UrunUretimeCikisMaliyeti * TabanDovizKuru;

        suspend;
      end
    end
  end
end
