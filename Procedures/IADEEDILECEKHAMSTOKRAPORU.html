CREATE PROCEDURE IADEEDILECEKHAMSTOKRAPORU(
    AFIRMAID VARCHAR(20))
RETURNS (
    REZERVTARIHI TIMESTAMP,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYENO IRSALIYENOSIZE,
    TARIH TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30),
    HAMADI VARCHAR(255),
    MIKTAR NUMERIC(12,3),
    IADEEDILENMIKTAR NUMERIC(12,3),
    IADEEDILECEKMIKTAR NUMERIC(12,3),
    LOT VARCHAR(150),
    HAMSTOKIADESEBEBIID INTEGER,
    HAMSTOKIADESEBEBIADI VARCHAR(30),
    IRSALIYEKILO NUMERIC(12,3),
    IRSALIYEMETRE NUMERIC(12,3),
    RAFID VARCHAR(10),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    HAMSTOKMASTERID INTEGER)
AS
begin
  if (AFirmaId is null) then
  begin
    For Select HamIadeRezerv.RezervTarihi, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, Ormeci.Adi, Ham.Adi,
               HamStokIrsaliye.Fason, HamStokIrsaliye.EvrakTipi, HamIadeRezerv.Miktar, HamIadeRezerv.IadeEdilenMiktar,
               HamStokMaster.Lot, HamIadeRezerv.HamStokIadeSebebiId, HamStokIadeSebebi.Adi,
               HamStokDetay.Kilo, HamStokDetay.Metre, RafDetay.RafId, HamIadeRezerv.Notu, HamStokDetay.HamStokMasterId from HamIadeRezerv
        left join HamStokIadeSebebi on HamStokIadeSebebi.HamStokIadeSebebiid = HamIadeRezerv.HamStokIadeSebebiid
        left join RafDetay on RafDetay.RafDetayId = HamIadeRezerv.RafDetayId
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
        left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
        left join Ormeci on Ormeci.OrmeciId = HamStokIrsaliye.OrmeciId
        left join Ham on Ham.HamId = RafDetay.HamId
        where HamIadeRezerv.Miktar > HamIadeRezerv.IadeEdilenMiktar
        into :RezervTarihi, :FirmaId, :FirmaAdi, :IrsaliyeNo, :Tarih, :OrmeciAdi, :HamAdi, :Fason, :EvrakTipi, :Miktar, :IadeEdilenMiktar,
             :Lot, :HamStokIadeSebebiId, :HamStokIadeSebebiAdi,
             :IrsaliyeKilo, :IrsaliyeMetre, :RafId, :Notu, :HamStokMasterId do
    begin
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';
      if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
      else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
      else EvrakTipiStr = 'Hatalı';

      IadeEdilecekMiktar = Miktar - IadeEdilenMiktar;
      suspend;
    end
  end
  else
  begin
    For Select HamIadeRezerv.RezervTarihi, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, Ormeci.Adi, Ham.Adi,
               HamStokIrsaliye.Fason, HamStokIrsaliye.EvrakTipi, HamIadeRezerv.Miktar, HamIadeRezerv.IadeEdilenMiktar,
               HamStokMaster.Lot, HamIadeRezerv.HamStokIadeSebebiId, HamStokIadeSebebi.Adi,
               HamStokDetay.Kilo, HamStokDetay.Metre, RafDetay.RafId, HamIadeRezerv.Notu, HamStokDetay.HamStokMasterId from HamIadeRezerv
        left join HamStokIadeSebebi on HamStokIadeSebebi.HamStokIadeSebebiid = HamIadeRezerv.HamStokIadeSebebiid
        left join RafDetay on RafDetay.RafDetayId = HamIadeRezerv.RafDetayId
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
        left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
        left join Ormeci on Ormeci.OrmeciId = HamStokIrsaliye.OrmeciId
        left join Ham on Ham.HamId = RafDetay.HamId
        where HamIadeRezerv.Miktar > HamIadeRezerv.IadeEdilenMiktar and
              Firma.AnaHesapFirmaId = :AFirmaId
        into :RezervTarihi, :FirmaId, :FirmaAdi, :IrsaliyeNo, :Tarih, :OrmeciAdi, :HamAdi, :Fason, :EvrakTipi, :Miktar, :IadeEdilenMiktar,
             :Lot, :HamStokIadeSebebiId, :HamStokIadeSebebiAdi,
             :IrsaliyeKilo, :IrsaliyeMetre, :RafId, :Notu, :HamStokMasterId do
    begin
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
      else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
      else EvrakTipiStr = 'Hatalı';

      IadeEdilecekMiktar = Miktar - IadeEdilenMiktar;
      suspend;
    end
  end
end
