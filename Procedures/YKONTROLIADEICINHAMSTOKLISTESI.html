CREATE PROCEDURE YKONTROLIADEICINHAMSTOKLISTESI
RETURNS (
    YKONTROLCIKISFISIDETAYID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(20),
    FISTARIHI TIMESTAMP,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    GELENMIKTAR NUMERIC(12,3),
    IADEEDILENMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    GIRISEVRAKTIPI INTEGER,
    GIRISEVRAKTIPISTR VARCHAR(8))
AS
begin
  for select YKontrolCikisFisiDetay.YKontrolCikisFisiDetayId, YKontrolCikisFisi.FisTarihi, YKontrolCikisFisi.FisYil, YKontrolCikisFisi.FisNo,
             Ham.HamId, Ham.Adi, YKontrolCikisFisiDetay.Miktar, YKontrolCikisFisiDetay.IadeEdilenMiktar,
             HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo,
             HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi from YKontrolCikisFisiDetay
  left join YKontrolCikisFisi on YKontrolCikisFisi.YKontrolCikisFisiId = YKontrolCikisFisiDetay.YKontrolCikisFisiId
  Left join YuzeyKontrolTalep on YuzeyKontrolTalep.YuzeyKontrolTalepKayitId = YKontrolCikisFisiDetay.YuzeyKontrolTalepKayitId
  Left join HamStokMaster on HamStokMaster.HamStokMasterId = YuzeyKontrolTalep.HamStokMasterId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
  left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
  left join Ham on Ham.HamId = HamStokMaster.HamId
  where YKontrolCikisFisiDetay.StokVar = 1 and YKontrolCikisFisi.TeslimAlindiDurumu = 1
  into :YKontrolCikisFisiDetayId, :FisTarihi, :FisYil, :FisNo, :HamId, :HamAdi, :GelenMiktar, :IadeEdilenMiktar, :FirmaId, :FirmaAdi,
       :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :GirisEvrakTipi do
  begin
     KalanMiktar = GelenMiktar - IadeEdilenMiktar;
     FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';

     if (GirisEvrakTipi = 0) then GirisEvrakTipiStr = 'Fiş';
     else if (GirisEvrakTipi = 1) then GirisEvrakTipiStr = 'İrsaliye';
     else GirisEvrakTipiStr = 'Hatalı';

     if (KalanMiktar <> 0) then suspend;
  end
end
