CREATE PROCEDURE KANTARILKTARTIKARTIDETAYI
RETURNS (
    KANTARKAYITID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable GirisCikis integer;
begin
  For Select KantarKayitId, GirisCikis from KantarIlkTartiKarti into :KantarKayitId, :GirisCikis do
  begin
    if (GirisCikis = 0) then
    begin
      For Select HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi,
                 HamStokIrsaliye.FiiliSevkTarihi, IK.Adi, HamStokIrsaliye.InsertTarihi, Sk.Adi, HamStokIrsaliye.DegisimTarihi from HamStokIrsaliye
      left join Firma On Firma.FirmaId = HamStokIrsaliye.FirmaId
      Left Join Kullanici IK on IK.KullaniciId = HamstokIrsaliye.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = HamstokIrsaliye.KullaniciId
      where HamStokIrsaliye.KantarKayitId = :KantarKayitId
      into :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi,
      :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do  suspend;

      For Select OrmeIrsaliye.FirmaId, Firma.Adi, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
                 OrmeIrsaliye.FiiliSevkTarihi from OrmeIrsaliye
      left join Firma On Firma.FirmaId = OrmeIrsaliye.FirmaId
      where OrmeIrsaliye.KantarKayitId = :KantarKayitId
      into :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi do  suspend;
    end
    else
    begin
      For Select BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo,
              BoyaIrsaliye.SevkTarihi, IK.Adi, BoyaIrsaliye.InsertTarihi, Sk.Adi, BoyaIrsaliye.DegisimTarihi from BoyaIrsaliye
      left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
      Left Join Kullanici IK on IK.KullaniciId = BoyaIrsaliye.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = BoyaIrsaliye.KullaniciId
      where BoyaIrsaliye.KantarKayitID = :KantarKayitId
      into :FirmaId, :FirmaAdi, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :FiiliSevkTarihi,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do suspend;

      For Select OrmeIrsaliye.FirmaId, Firma.Adi, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
                 OrmeIrsaliye.FiiliSevkTarihi from OrmeIrsaliye
      left join Firma On Firma.FirmaId = OrmeIrsaliye.FirmaId
      where OrmeIrsaliye.KantarKayitId = :KantarKayitId
      into :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi do  suspend;
    end

    FirmaId = Null;
    FirmaAdi = Null;

    for Select UniteIrsaliye.AliciUniteId, Unite.Adi, UniteIrsaliye.IrsaliyeTarihi, UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo,
               UniteIrsaliye.SevkTarihi, IK.Adi, UniteIrsaliye.InsertTarihi, Sk.Adi, UniteIrsaliye.DegisimTarihi from UniteIrsaliye
      left join Unite on Unite.UniteId = UniteIrsaliye.AliciUniteId
      Left Join Kullanici IK on IK.KullaniciId = UniteIrsaliye.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = UniteIrsaliye.KullaniciId
      where UniteIrsaliye.KantarKayitID = :KantarKayitId
      into :UniteId, :UniteAdi, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :FiiliSevkTarihi,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do suspend;

    UniteId = Null;
    UniteAdi = Null;

    for Select Depo.UniteId, Unite.Adi, MamulDepoTransferFisi.IrsaliyeTarihi, MamulDepoTransferFisi.IrsaliyeSeriNo, MamulDepoTransferFisi.IrsaliyeNo,
               MamulDepoTransferFisi.SevkTarihi, IK.Adi, MamulDepoTransferFisi.InsertTarihi, Sk.Adi, MamulDepoTransferFisi.DegisimTarihi from MamulDepoTransferFisi
      left join Depo on Depo.DepoId = MamulDepoTransferFisi.TransferDepoId
      left join Unite on Unite.UniteId = Depo.UniteId
      Left Join Kullanici IK on IK.KullaniciId = MamulDepoTransferFisi.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = MamulDepoTransferFisi.KullaniciId
      where MamulDepoTransferFisi.KantarKayitID = :KantarKayitId
      into :UniteId, :UniteAdi, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :FiiliSevkTarihi,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do suspend;


    UniteId = Null;
    UniteAdi = Null;

    For Select Irsaliye.FirmaId, Firma.Adi, Irsaliye.SeriNo, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi,
               Irsaliye.FiiliSevkTarihi, IK.Adi, Irsaliye.InsertTarihi, Sk.Adi, Irsaliye.DegisimTarihi from Irsaliye
    left join Firma On Firma.FirmaId = Irsaliye.FirmaId
    Left Join Kullanici IK on IK.KullaniciId = Irsaliye.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = Irsaliye.KullaniciId
    where Irsaliye.KantarKayitId = :KantarKayitId
    into :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do suspend;
  end
end
