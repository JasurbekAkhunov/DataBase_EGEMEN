CREATE PROCEDURE LOTTANITIMIIPLIKTALEPLISTESI(
    ALOTACILANLARIGOSTERME INTEGER,
    AFIRMANINILKTALEBINIGOSTER INTEGER)
RETURNS (
    IPLIKTALEPYIL INTEGER,
    IPLIKTALEPID INTEGER,
    IPLIKTALEPIDSTR VARCHAR(30),
    IPLIKTALEPKAYITID INTEGER,
    IPLIKTALEPDETAYID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    KALINLIKID INTEGER,
    IPLIKCINSIID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    MIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    LOT VARCHAR(30),
    ACILANLOTLAR VARCHAR(100),
    URETIMLOTLARI VARCHAR(100),
    TALEPYERIID INTEGER,
    TALEPYERIIDSTR VARCHAR(8),
    AMBALAJSEKLIID INTEGER,
    AMBALAJSEKLIADI VARCHAR(10),
    IPLIKTALEPKULLANIMYERIID INTEGER,
    IPLIKTALEPKULLANIMYERIADI VARCHAR(50),
    NOTU VARCHAR(255))
AS
declare variable FLot varchar(30);
declare variable FLastFirmaID varchar(20);
declare variable FLastHamID   integer;
begin
  if (AFirmaninIlkTalebiniGoster = 0) then
  begin
    For Select IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, IplikTalepDetay.IplikTalepKayitId, IplikTalepDetay.IplikTalepDetayId,
               IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId, IplikTalepDetay.BoyaSiparisDetayId,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, IplikTalepDetay.HamId, Ham.Adi, Ham.KalinlikId, Ham.IplikCinsiId,
               IplikTalepDetay.Miktar,
               IplikTalepDetay.MelanjKodu, IplikTalepDetay.FlamKodu, IplikTalepDetay.Lot, IplikTalepDetay.AmbalajSekliId, AmbalajSekli.Adi,
               IplikTalepDetay.IplikTalepKullanimYeriId, IplikTalepKullanimYeri.Adi,
               IplikTalepDetay.Miktar - IplikTalepDetay.TeslimEdilenMiktar, IplikTalepDetay.Notu, BoyaSiparis.FirmaId, Firma.Adi,
               IplikTalep.TalepYeriId FROM IplikTalepDetay
      LEFT JOIN Ham ON Ham.HamId = IplikTalepDetay.HamId
      LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
      left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = IplikTalepDetay.AmbalajSekliId
      left join IplikTalepKullanimYeri on IplikTalepKullanimYeri.IplikTalepKullanimYeriId = IplikTalepDetay.IplikTalepKullanimYeriId
      LEFT JOIN IplikTalep ON IplikTalep.IplikTalepKayitId = IplikTalepDetay.IplikTalepKayitId
      LEFT JOIN Firma ON Firma.FirmaId = BoyaSiparis.FirmaId
      WHERE IplikTalepDetay.Kapali = 0 and IplikTalep.OnayDurumu = 1
      order by BoyaSiparis.FirmaId, IplikTalepDetay.HamId, IplikTalepDetay.IplikTalepDetayId
    into :IplikTalepYil, :IplikTalepId, :IplikTalepKayitId, :IplikTalepDetayId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
         :HamId, :HamAdi, :KalinlikId, :IplikCinsiId, :Miktar,:MelanjKodu, :FlamKodu,  :Lot,
         :AmbalajSekliId, :AmbalajSekliAdi, :IplikTalepKullanimYeriId, :IplikTalepKullanimYeriAdi,
         :KalanMiktar, :Notu, :SiparisFirmaId, :SiparisFirmaAdi, :TalepYeriId do
    begin
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId);

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      if (TalepYeriId = 1) then TalepYeriIdStr = 'Boyahane';
      else if (TalepYeriId = 2) then TalepYeriIdStr = 'Merkez';
      else if (TalepYeriId = 3) then TalepYeriIdStr = 'İplik';
      else TalepYeriIdStr = '';

      AcilanLotlar = '';

      For Select IplikLot.Lot from IplikLot
        where IplikLot.IplikTalepDetayID = :IplikTalepDetayID
        into :FLot do
      begin
        if (Char_Length(AcilanLotlar) + Char_Length(FLot) < 99) then
        begin
          if (AcilanLotlar <> '') then AcilanLotlar = AcilanLotlar || ',';
          AcilanLotlar = AcilanLotlar || Flot;
        end
      end

      UretimLotlari = '';

      For Select IplikUretim.Lot from IplikUretim
        where IplikUretim.IplikTalepDetayID = :IplikTalepDetayID
        group by IplikUretim.Lot
        into :FLot do
      begin
        if (Char_Length(UretimLotlari) + Char_Length(FLot) < 99) then
        begin
          if (UretimLotlari <> '') then UretimLotlari = UretimLotlari || ',';
          UretimLotlari = UretimLotlari || Flot;
        end
      end

      if (ALotAcilanlariGosterme = 0 or (UretimLotlari = '' and AcilanLotlar = '') ) then suspend;
    end
  end
  else
  begin
    FLastFirmaID = '@@@';
    FLastHamID = -1;

    For Select IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, IplikTalepDetay.IplikTalepKayitId, IplikTalepDetay.IplikTalepDetayId,
               IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId, IplikTalepDetay.BoyaSiparisDetayId,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, IplikTalepDetay.HamId, Ham.Adi, Ham.KalinlikId, Ham.IplikCinsiId,
               IplikTalepDetay.Miktar,
               IplikTalepDetay.MelanjKodu, IplikTalepDetay.FlamKodu, IplikTalepDetay.Lot, IplikTalepDetay.AmbalajSekliId, AmbalajSekli.Adi,
               IplikTalepDetay.IplikTalepKullanimYeriId, IplikTalepKullanimYeri.Adi,
               IplikTalepDetay.Miktar - IplikTalepDetay.TeslimEdilenMiktar, IplikTalepDetay.Notu, BoyaSiparis.FirmaId, Firma.Adi,
               IplikTalep.TalepYeriId FROM IplikTalepDetay
      LEFT JOIN Ham ON Ham.HamId = IplikTalepDetay.HamId
      LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
      left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = IplikTalepDetay.AmbalajSekliId
      left join IplikTalepKullanimYeri on IplikTalepKullanimYeri.IplikTalepKullanimYeriId = IplikTalepDetay.IplikTalepKullanimYeriId
      LEFT JOIN IplikTalep ON IplikTalep.IplikTalepKayitId = IplikTalepDetay.IplikTalepKayitId
      LEFT JOIN Firma ON Firma.FirmaId = BoyaSiparis.FirmaId
      WHERE IplikTalepDetay.Kapali = 0 and IplikTalep.OnayDurumu = 1
      order by BoyaSiparis.FirmaId, IplikTalepDetay.HamId, IplikTalepDetay.IplikTalepDetayId
    into :IplikTalepYil, :IplikTalepId, :IplikTalepKayitId, :IplikTalepDetayId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
         :HamId, :HamAdi, :KalinlikId, :IplikCinsiId, :Miktar,:MelanjKodu, :FlamKodu,  :Lot,
         :AmbalajSekliId, :AmbalajSekliAdi, :IplikTalepKullanimYeriId, :IplikTalepKullanimYeriAdi,
         :KalanMiktar, :Notu, :SiparisFirmaId, :SiparisFirmaAdi, :TalepYeriId do
    begin
      if (SiparisFirmaID is null) then SiparisFirmaId = '';

      AcilanLotlar = '';

      For Select IplikLot.Lot from IplikLot
        where IplikLot.IplikTalepDetayID = :IplikTalepDetayID
        into :FLot do
      begin
        if (Char_Length(AcilanLotlar) + Char_Length(FLot) < 99) then
        begin
          if (AcilanLotlar <> '') then AcilanLotlar = AcilanLotlar || ',';
          AcilanLotlar = AcilanLotlar || Flot;
        end
      end

      UretimLotlari = '';

      For Select IplikUretim.Lot from IplikUretim
        where IplikUretim.IplikTalepDetayID = :IplikTalepDetayID
        group by IplikUretim.Lot
        into :FLot do
      begin
        if (Char_Length(UretimLotlari) + Char_Length(FLot) < 99) then
        begin
          if (UretimLotlari <> '') then UretimLotlari = UretimLotlari || ',';
          UretimLotlari = UretimLotlari || Flot;
        end
      end

      if (FLastFirmaID <> SiparisFirmaId or FLastHamID <> HamId) then
      begin
        IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId);

        BoyaSiparisIdStr = '';
        if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

        if (TalepYeriId = 1) then TalepYeriIdStr = 'Boyahane';
        else if (TalepYeriId = 2) then TalepYeriIdStr = 'Merkez';
        else if (TalepYeriId = 3) then TalepYeriIdStr = 'İplik';
        else TalepYeriIdStr = '';

        if (ALotAcilanlariGosterme = 0 or (AcilanLotlar = '' and UretimLotlari = '') ) then
        begin
          FLastFirmaID = SiparisFirmaId;
          FLastHamID = HamId;
          suspend;
        end
      end
    end
  end
end
