CREATE PROCEDURE BUKUMISEMRIRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMID INTEGER,
    ALOT VARCHAR(30),
    ABITTI INTEGER)
RETURNS (
    BUKUMISEMRIKAYITID INTEGER,
    BUKUMISEMRIYIL INTEGER,
    BUKUMISEMRIID INTEGER,
    BUKUMISEMRIIDSTR VARCHAR(30),
    TARIHI TIMESTAMP,
    IPLIKLOTID INTEGER,
    LOT VARCHAR(30),
    BOYASIPARISIDSTR VARCHAR(20),
    SIPFIRMAID VARCHAR(20),
    SIPFIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MIKTAR NUMERIC(12,3),
    BUKUMSAYISI INTEGER,
    BOBINGRAMAJI NUMERIC(12,3),
    SARIMMETRAJI INTEGER,
    BUKUMHIZI INTEGER,
    IPLIKKODU VARCHAR(20),
    BUKUMT DOUBLE PRECISION,
    BUKUMTM DOUBLE PRECISION,
    BUKUMTIPI VARCHAR(50),
    BUKUMCAPI DOUBLE PRECISION,
    KONIKRENGI VARCHAR(50),
    FLAYERAGIRLIGI DOUBLE PRECISION,
    MAKARAPOZISYONU VARCHAR(50),
    CIKISMIKTARI NUMERIC(12,3),
    GERCEKLESENMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    NOTU VARCHAR(255),
    BITIRENKULLANICIID INTEGER,
    BITIRENKULLANICIADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    BITTI INTEGER,
    BITTISTR VARCHAR(5),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin

  if (ABitti = 1) then
  begin
    for Select BukumIsEmri.BukumIsEmriKayitId, BukumIsEmri.BukumIsEmriyil, BukumIsEmri.BukumIsEmriId,
               BukumIsEmri.Tarihi, BukumIsEmri.IplikLotId, Coalesce(IplikLot.Lot, BukumIsEmri.Lot), BukumIsEmri.HamId, Ham.Adi,
               BukumIsEmri.Miktar, BukumIsEmri.BukumSayisi, BukumIsEmri.BobinGramaji, BukumIsEmri.SarimMetraji,
               BukumIsEmri.BukumHizi, BukumIsEmri.GerceklesenMiktar, BukumIsEmri.CikisMiktari, BukumIsEmri.Notu,
               BukumIsEmri.BitirenKullaniciId, Bk.Adi, BukumIsEmri.BitisTarihi, BukumIsEmri.Bitti,
               Ik.Adi, BukumIsEmri.InsertTarihi, Sk.Adi, BukumIsEmri.DegisimTarihi, BukumIsEmri.IplikKodu, BukumIsEmri.BukumT,
               BukumIsEmri.BukumTM, BukumIsEmri.BukumTipi, BukumIsEmri.BukumCapi, BukumIsEmri.KonikRengi, BukumIsEmri.FlayerAgirligi,
               BukumIsEmri.MakaraPozisyonu, '(' || BoyaSiparis.BoyaSiparisYil || ')' ||
               BoyaSiparis.BoyaSiparisID, BoyaSiparis.FirmaID, Firma.Adi  from BukumIsEmri
      left join IplikLot on IplikLot.IplikLotID = BukumIsEmri.IplikLotId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BukumIsEmri.BoyaSiparisKayitID
      Left Join Firma on Firma.FirmaId = BoyaSiparis.FirmaID
      left join Ham on Ham.HamID = BukumIsEmri.HamID
      left join Kullanici BK on BK.KullaniciId = BukumIsEmri.BitirenKullaniciId
      Left Join Kullanici IK on IK.KullaniciId = BukumIsEmri.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = BukumIsEmri.KullaniciId
      where (BukumIsEmri.BitisTarihi >= :ABaslangicTarihi and BukumIsEmri.BitisTarihi < :ABitisTarihi + 1) and BukumIsEmri.Bitti = 1 and
            (BukumIsEmri.HamId = :AHamId or :AHamId is null) and
            (IplikLot.Lot = :ALot or :ALot is null)
      order by BukumIsEmri.Tarihi
      into :BukumIsEmriKayitId, :BukumIsEmriyil, :BukumIsEmriId, :Tarihi, :IplikLotId, :Lot, :HamId, :HamAdi,
           :Miktar, :BukumSayisi, :BobinGramaji, :SarimMetraji, :BukumHizi, :GerceklesenMiktar, :CikisMiktari, :Notu,
           :BitirenKullaniciId, :BitirenKullaniciAdi, :BitisTarihi, :Bitti,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :IplikKodu, :BukumT, :BukumTM, :BukumTipi,
           :BukumCapi, :KonikRengi, :FlayerAgirligi, :MakaraPozisyonu, :BoyaSiparisIDStr, :SipFirmaID, :SipFirmaAdi  do
    begin
      if (Bitti = 0) then BittiStr = '';
      else BittiStr = 'Bitti';

      BukumIsEmriIdStr = '(' || EgemenIntegerToStr(BukumIsEmriYil) || ') ' || EgemenIntegerToStr(BukumIsEmriId) || ' ';

      KalanMiktar = Miktar - GerceklesenMiktar;

      suspend;
    end
  end
  else
  begin
    for Select BukumIsEmri.BukumIsEmriKayitId, BukumIsEmri.BukumIsEmriyil, BukumIsEmri.BukumIsEmriId,
               BukumIsEmri.Tarihi, BukumIsEmri.IplikLotId, Coalesce(IplikLot.Lot, BukumIsEmri.Lot), BukumIsEmri.HamId, Ham.Adi,
               BukumIsEmri.Miktar, BukumIsEmri.BukumSayisi, BukumIsEmri.BobinGramaji, BukumIsEmri.SarimMetraji,
               BukumIsEmri.BukumHizi, BukumIsEmri.GerceklesenMiktar, BukumIsEmri.CikisMiktari, BukumIsEmri.Notu,
               BukumIsEmri.BitirenKullaniciId, Bk.Adi, BukumIsEmri.BitisTarihi, BukumIsEmri.Bitti,
               Ik.Adi, BukumIsEmri.InsertTarihi, Sk.Adi, BukumIsEmri.DegisimTarihi, BukumIsEmri.IplikKodu, BukumIsEmri.BukumT,
               BukumIsEmri.BukumTM, BukumIsEmri.BukumTipi, BukumIsEmri.BukumCapi, BukumIsEmri.KonikRengi, BukumIsEmri.FlayerAgirligi,
               BukumIsEmri.MakaraPozisyonu, '(' || BoyaSiparis.BoyaSiparisYil || ')' ||
               BoyaSiparis.BoyaSiparisID, BoyaSiparis.FirmaID, Firma.Adi  from BukumIsEmri
      left join IplikLot on IplikLot.IplikLotID = BukumIsEmri.IplikLotId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BukumIsEmri.BoyaSiparisKayitID
      Left Join Firma on Firma.FirmaId = BoyaSiparis.FirmaID
      left join Ham on Ham.HamID = BukumIsEmri.HamID
      left join Kullanici BK on BK.KullaniciId = BukumIsEmri.BitirenKullaniciId
      Left Join Kullanici IK on IK.KullaniciId = BukumIsEmri.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = BukumIsEmri.KullaniciId
      where (BukumIsEmri.Tarihi >= :ABaslangicTarihi and BukumIsEmri.Tarihi < :ABitisTarihi + 1) and
            (BukumIsEmri.HamId = :AHamId or :AHamId is null) and
            (IplikLot.Lot = :ALot or :ALot is null) and
            (BukumIsEmri.Bitti = :ABitti or :ABitti is null)
      order by BukumIsEmri.Tarihi
      into :BukumIsEmriKayitId, :BukumIsEmriyil, :BukumIsEmriId, :Tarihi, :IplikLotId, :Lot, :HamId, :HamAdi,
           :Miktar, :BukumSayisi, :BobinGramaji, :SarimMetraji, :BukumHizi, :GerceklesenMiktar, :CikisMiktari, :Notu,
           :BitirenKullaniciId, :BitirenKullaniciAdi, :BitisTarihi, :Bitti,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :IplikKodu, :BukumT, :BukumTM, :BukumTipi,
           :BukumCapi, :KonikRengi, :FlayerAgirligi, :MakaraPozisyonu, :BoyaSiparisIDStr, :SipFirmaID, :SipFirmaAdi  do
    begin
      if (Bitti = 0) then BittiStr = '';
      else BittiStr = 'Evet';

      BukumIsEmriIdStr = '(' || EgemenIntegerToStr(BukumIsEmriYil) || ') ' || EgemenIntegerToStr(BukumIsEmriId) || ' ';

      KalanMiktar = Miktar - GerceklesenMiktar;

      suspend;
    end
  end
end
