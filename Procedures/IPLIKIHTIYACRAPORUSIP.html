CREATE PROCEDURE IPLIKIHTIYACRAPORUSIP(
    AIPLIKLERIDOK INTEGER,
    ALYCRALARIDOK INTEGER,
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    ATARIHEGORESIPRECETETOPLAMI INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    ARAPORTIPI INTEGER)
RETURNS (
    HAMID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    FIRMAID FIRMAKODUSIZE,
    SIPARISSIRANO INTEGER,
    SIPARISHAMID INTEGER,
    HAMADI VARCHAR(255),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    KALITEID KALITEKODUSIZE,
    MELANJKODU VARCHAR(20),
    TESLIMTARIHI TIMESTAMP,
    PUS INTEGER,
    FEINE INTEGER,
    GEREKLIMIKTAR BOYAMIKTARSIZE,
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUPID VARCHAR(4),
    ACIKTALIMATRECETET BOYAMIKTARSIZE,
    ACIKSTALIMATCIKIST DOUBLE PRECISION,
    FIRMAADI ADI100)
AS
declare variable hstIplik       integer;
declare variable hstLycra       integer;
declare variable AstIplik               integer;
declare variable astBoyaliIplik         integer;
declare variable astLycra               integer;
declare variable astHamKumas            integer;
declare variable astBoyaliKumas         integer;
declare variable astIplikBoyaliKumas    integer;
declare variable KullanimOrani  Double Precision;
declare variable FireOrani      Double Precision;
declare variable assSatis          integer;
declare variable assFasonArtiSatis integer;
declare variable assFason          integer;
declare variable FOrmeSiparisDetayID    integer;
declare variable shtUretimeCikis        integer;
declare variable KaliteRecetesiniKullan integer;
begin
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;
  select shtUretimeCikis from OrmeStokHareketTipiGetir into :shtUretimeCikis;

  select KaliteRecetesiniKullan from Sistem into KaliteRecetesiniKullan;

  Select astIplik, astBoyaliIplik, AstLycra, AstHamKumas, AstBoyaliKumas, AstIplikBoyaliKumas from OrmeSiparisTipiGetir
    into :astIplik, :astBoyaliIplik, :AstLycra, :AstHamKumas, :AstBoyaliKumas, :AstIplikBoyaliKumas;

  Select assSatis, assFasonArtiSatis, assFason from OrmeSiparisSekliGetir into :assSatis, :assFasonArtiSatis, :assFason;

  for Select Ham.HamId  From Ham
    where (HamTipiID = :hstIplik or HamTipiID = :hstLycra) and
    (:AIplikleriDok = 1 or HamTipiID <> :hstIplik) and
    (:ALycralariDok = 1 or HamTipiID <> :hstLycra)
    into :HamID do
  begin
    if (KaliteRecetesiniKullan = 1) then
    begin
      for select Kaliteid, KullanimOrani, FireOrani from KaliteRecete where ReceteHamId = :HamId and FireOrani < 100
        into :KaliteId, :KullanimOrani, :FireOrani do
      begin
        GerekliMiktar = 0;
        for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, BoyaSiparis.SiparisTarihi,
                   OrmeSiparisDetay.Kaliteid, OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo,
                   OrmeSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
                   Sum((OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar - OrmeSiparisDetay.ToplamOrulen) * (:KullanimOrani / 100.00) / ((100.00 - :FireOrani) / 100.00) ),                 
                  SiparisGrup.Adi, SiparisGrup.SiparisGrupId, Firma.Adi from OrmeSiparisDetay
          left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
          left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
          left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
          left join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
          where (OrmeSiparisDetay.kaliteid = :kaliteid) and
                (OrmeSiparisDetay.Kapali = 0) and
                /*tarih filtresi gir parametreye göre*/
                ((:ATariheGoreSipReceteToplami = 0) or
                (:ATariheGoreSipReceteToplami = 1 and BoyaSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and BoyaSiparis.SiparisTarihi <= :ABitisSiparisTarihi ))   and
                ((BoyaSiparis.OrmeSiparisTipiId = :AstHamKumas) or
                (BoyaSiparis.OrmeSiparisTipiId = :AstBoyaliKumas) or
                (BoyaSiparis.OrmeSiparisTipiId = :AstIplikBoyaliKumas)) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1)
        Group By BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, BoyaSiparis.SiparisTarihi,
                 OrmeSiparisDetay.Kaliteid, OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo,
                 OrmeSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
                 SiparisGrup.Adi, SiparisGrup.SiparisGrupId, Firma.Adi
        into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :SiparisTarihi, :Kaliteid, :SiparisSiraNo, :MusteriSiparisNo, :ReferansSiparisNo,
             :SiparisHamId, :HamAdi, :MelanjKodu, :TeslimTarihi, :Pus, :Feine, :GerekliMiktar, :SiparisGrupAdi, :SiparisGrupId, :FirmaAdi do
        begin
          BoyaSiparisIdStr = '';
          if (BoyaSiparisYil is not null) then
            BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

          ---
          AcikTalimatReceteT = 0;
          select sum( (OrmePlanlama.PlanlananMiktar - OrmePlanlama.GerceklesenMiktar + OrmePlanlama.ToplamSakatOrulen) * (OrmePlanlamaRecete.KullanimOrani / 100.00) / ((100.00 - OrmePlanlamaRecete.TahminiFireOrani) / 100.00) )
            from OrmeSiparisDetay
              left join OrmePlanlama on OrmePlanlama.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
              left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = OrmePlanlama.OrmePlanlamaId
              left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
           where (OrmePlanlamaRecete.HamId = :HamId) and
            ORmeSiparisDetay.ormesiparisdetayid = :FOrmeSiparisDetayID   and
              (OrmeSiparisDetay.Kapali = 0)  and
              (OrmePlanlama.Bitti = 0)  and
              (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
              (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
              (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1) and
              (
                  :ARaporTipi is null or
                  (OrmePlanlama.OrmeMakinaId is not null and :ARaporTipi = 0) or
                  (OrmePlanlama.FirmaId is not null and :ARaporTipi = 1)
               ) and
               OrmePlanlamaRecete.TahminiFireOrani < 100
            into :AcikTalimatReceteT;
            if (AcikTalimatReceteT is Null) then AcikTalimatReceteT = 0;
          ---


          ---
          select sum(StokHareket.Miktar) from StokHareket
            left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = StokHareket.OrmePlanlamaId
            left join OrmeSiparisDetay On OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
            left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
            --Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
            where (StokHareket.OrmePlanlamaId is not Null) and
                  (StokHareket.ORmeSiparisDetayId = :FOrmeSiparisDetayID) and
                  (StokHareket.HareketTipi = :shtUretimeCikis) and
                  (StokHareket.HamId = :HamId) and
                  (OrmeSiparisDetay.Kapali = 0)  and
                  (OrmePlanlama.Bitti = 0) and
                  (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
                  (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
                  (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1) and
                  (:ARaporTipi is null or
                   (OrmePlanlama.OrmeMakinaId is not null and :ARaporTipi = 0) or
                   (OrmePlanlama.FirmaId is not null and :ARaporTipi = 1) )
            into :AcikSTalimatCikisT;
          if (AcikSTalimatCikisT is Null) then AcikSTalimatCikisT = 0;
          ---
          --if (GerekliMiktar > 0) then
          suspend;
        end
      end
    end
    else
    begin
      for select HamId, KullanimOrani, FireOrani from HamStokRecete where ReceteHamId = :HamId and FireOrani < 100
        into :SiparisHamId, :KullanimOrani, :FireOrani do
      begin
        GerekliMiktar = 0;
        for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, BoyaSiparis.SiparisTarihi,
                   OrmeSiparisDetay.Kaliteid, OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo,
                   OrmeSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
                   Sum((OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar - OrmeSiparisDetay.ToplamOrulen) * (:KullanimOrani / 100.00) / ((100.00 - :FireOrani) / 100.00) ),
                  SiparisGrup.Adi, SiparisGrup.SiparisGrupId, OrmeSiparisDetay.OrmeSiparisDetayID, Firma.Adi from OrmeSiparisDetay
          left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
          left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
          left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
          left join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
          where (OrmeSiparisDetay.HamId = :SiparisHamId) and
                (OrmeSiparisDetay.Kapali = 0) and
                /*tarih filtresi gir parametreye göre*/
                ((:ATariheGoreSipReceteToplami = 0) or
                (:ATariheGoreSipReceteToplami = 1 and BoyaSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and BoyaSiparis.SiparisTarihi <= :ABitisSiparisTarihi ))   and
                ((BoyaSiparis.OrmeSiparisTipiId = :AstHamKumas) or
                (BoyaSiparis.OrmeSiparisTipiId = :AstBoyaliKumas) or
                (BoyaSiparis.OrmeSiparisTipiId = :AstIplikBoyaliKumas)) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1)
        Group By BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, BoyaSiparis.SiparisTarihi,
                 OrmeSiparisDetay.Kaliteid, OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo,
                 OrmeSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
                  SiparisGrup.Adi, SiparisGrup.SiparisGrupId, OrmeSiparisDetay.OrmeSiparisDetayID, Firma.Adi
        into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :SiparisTarihi, :Kaliteid, :SiparisSiraNo, :MusteriSiparisNo, :ReferansSiparisNo,
             :SiparisHamId, :HamAdi, :MelanjKodu, :TeslimTarihi, :Pus, :Feine, :GerekliMiktar, :SiparisGrupAdi, :SiparisGrupId, :FOrmeSiparisDetayID, FirmaAdi do
        begin
          BoyaSiparisIdStr = '';
          if (BoyaSiparisYil is not null) then
            BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';


          ---
          AcikTalimatReceteT = 0;
          select sum( (OrmePlanlama.PlanlananMiktar - OrmePlanlama.GerceklesenMiktar + OrmePlanlama.ToplamSakatOrulen) * (OrmePlanlamaRecete.KullanimOrani / 100.00) / ((100.00 - OrmePlanlamaRecete.TahminiFireOrani) / 100.00) )
            from OrmeSiparisDetay
              left join OrmePlanlama on OrmePlanlama.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
              left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = OrmePlanlama.OrmePlanlamaId
              left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
           where (OrmePlanlamaRecete.HamId = :HamId) and
            ORmeSiparisDetay.ormesiparisdetayid = :FOrmeSiparisDetayID   and
              (OrmeSiparisDetay.Kapali = 0)  and
              (OrmePlanlama.Bitti = 0)  and
              (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
              (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
              (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1) and
              (
                  :ARaporTipi is null or
                  (OrmePlanlama.OrmeMakinaId is not null and :ARaporTipi = 0) or
                  (OrmePlanlama.FirmaId is not null and :ARaporTipi = 1)
               ) and
               OrmePlanlamaRecete.TahminiFireOrani < 100
            into :AcikTalimatReceteT;
            if (AcikTalimatReceteT is Null) then AcikTalimatReceteT = 0;
          ---


          ---
          select sum(StokHareket.Miktar) from StokHareket
            left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = StokHareket.OrmePlanlamaId
            left join OrmeSiparisDetay On OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
            left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
            --Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
            where (StokHareket.OrmePlanlamaId is not Null) and
                  (StokHareket.ORmeSiparisDetayId = :FOrmeSiparisDetayID) and
                  (StokHareket.HareketTipi = :shtUretimeCikis) and
                  (StokHareket.HamId = :HamId) and
                  (OrmeSiparisDetay.Kapali = 0)  and
                  (OrmePlanlama.Bitti = 0) and
                  (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
                  (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
                  (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1) and
                  (:ARaporTipi is null or
                   (OrmePlanlama.OrmeMakinaId is not null and :ARaporTipi = 0) or
                   (OrmePlanlama.FirmaId is not null and :ARaporTipi = 1) )
            into :AcikSTalimatCikisT;
          if (AcikSTalimatCikisT is Null) then AcikSTalimatCikisT = 0;
          ---

          --if (GerekliMiktar > 0) then
          suspend;
        end
      end
    end
  end
end
