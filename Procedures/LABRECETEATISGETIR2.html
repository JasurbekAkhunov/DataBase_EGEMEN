CREATE PROCEDURE LABRECETEATISGETIR2(
    ALABRECETEID INTEGER,
    ALABRECETEATISID INTEGER,
    AURUNFIYATGOREBILIR SMALLINT)
RETURNS (
    LABRECETECALISMAID INTEGER,
    LABRECETEATISID INTEGER,
    LABRECETEID INTEGER,
    HAMADI VARCHAR(255),
    ATISNOSTR VARCHAR(20),
    PIPETLEMEKODU VARCHAR(30),
    MIGRASYON INTEGER,
    MIGRASYONSTR VARCHAR(5),
    BOYAMALIYETI DOUBLE PRECISION,
    BOYAMALIYETITL DOUBLE PRECISION,
    KIMYASALMALIYETI DOUBLE PRECISION,
    KIMYASALMALIYETITL DOUBLE PRECISION,
    DOVIZKURU DOUBLE PRECISION,
    DOVIZKURUTARIHI TIMESTAMP,
    BOYAYUZDESI DOUBLE PRECISION,
    HESAPBIRIMID VARCHAR(5),
    ATISTARIHI TIMESTAMP,
    GIDISTARIHI TIMESTAMP,
    OKEY INTEGER,
    DURUMUSTR VARCHAR(6),
    TARIHI TIMESTAMP,
    RECETEASAMAID INTEGER,
    RECETEASAMAADI VARCHAR(40),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    BOYATIPIFARKLIURUNVAR INTEGER,
    KARTELATARIHI TIMESTAMP,
    MUSTERIFIYATONAYTARIHI TIMESTAMP,
    MUSTERIFIYATONAYSEKLI INTEGER,
    MUSTERIFIYATONAYSEKLISTR VARCHAR(10),
    MUSTERIFIYATONAYNOTU VARCHAR(50),
    RPTSTR VARCHAR(3),
    ILAVESOKUMUZERIBOYASTR VARCHAR(20),
    KRITIKCALISMASI INTEGER,
    KRITIKCALISMASISTR VARCHAR(10),
    RENKDERINLIGIID INTEGER,
    CALISMATARIHI TIMESTAMP,
    URUNSAYISI INTEGER,
    PIPETLEMEYEGONDERILDI INTEGER,
    PIPETLEMEYEGONDERILDISTR VARCHAR(5),
    COLOR INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    FIYATFARKI DOUBLE PRECISION)
AS
declare variable DovizId        varchar(5);
declare variable SistemDovizId  varchar(5);
declare variable MinBoyaTipiId  integer;
declare variable MaxBoyaTipiId  integer;
declare variable Rptmi          integer;
declare variable Ilave          integer;
declare variable SokumUzeriBoya integer;
declare variable FUrunID        varchar(20);
begin
  Select SistemDovizId from Sistem where sistemId = 1 into :SistemDovizId; /* YTL*/

  Select LabRecete.DovizId from LabRecete where LabRecete.LabReceteID = :ALabReceteID into :DovizId;

  if (DovizID <> SistemDovizId) then
  begin
    select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :SistemDovizId, Current_Date) into :DovizKuru, :DovizKuruTarihi;
  end
  else DovizKuru = 1;

  for select LabReceteAtis.LabReceteId, LabReceteAtis.LabReceteCalismaId, LabReceteAtis.LabReceteAtisId, LabReceteAtis.AtisNoStr,
   Ham.Adi, LabReceteAtis.AtisTarihi, LabReceteAtis.GidisTarihi, LabReceteAtis.Okey, LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi, LabReceteAtis.KartelaTarihi,
   LabReceteAtis.BoyaMaliyeti, LabReceteAtis.KimyasalMaliyeti, LabReceteAtis.BoyaYuzdesi, LabReceteGrup.Caption, LabReceteAtis.Notu, LabReceteAtis.Migrasyon,
   LabReceteAtis.ReceteAsamaId, ReceteAsama.Adi, LabReceteAtis.MusteriFiyatOnayTarihi, LabReceteAtis.MusteriFiyatOnaySekli,
   LabReceteAtis.MusteriFiyatOnayNotu, LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi,
   LabReceteCalisma.CalismaTarihi, LabReceteAtis.RenkDerinligiId, LabReceteAtis.PipetlemeyeGonderildi, LabReceteAtis.Color,
   IK.Adi, LabReceteAtis.InsertTarihi, SK.Adi, LabReceteAtis.DegisimTarihi, LabReceteCalisma.FiyatFarki  from LabReceteAtis
  left join LabRecete on LabRecete.LabReceteId = LabReceteAtis.LabReceteId
  left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
  left join Ham on Ham.HamId = LabReceteCalisma.HamId
  left join ReceteAsama on ReceteAsama.ReceteAsamaId = LabReceteAtis.ReceteAsamaId
  left join LabReceteGrup on LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
  Left Join Kullanici IK on IK.KullaniciId = LabReceteAtis.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = LabReceteAtis.KullaniciId
  where LabReceteAtis.LabReceteAtisId = :ALabReceteAtisId
  order by LabReceteAtis.labReceteAtisId
  into :LabReceteId, :LabReceteCalismaId, :LabReceteAtisId, :AtisNoStr, :HamAdi, :AtisTarihi, :GidisTarihi, :Okey, :DurumuStr, :Tarihi,
       :KartelaTarihi, :BoyaMaliyeti, :KimyasalMaliyeti, :BoyaYuzdesi, :HesapBirimId, :Notu, :Migrasyon, :ReceteAsamaId, :ReceteAsamaAdi,
       :MusteriFiyatOnayTarihi, :MusteriFiyatOnaySekli, :MusteriFiyatOnayNotu, :Rptmi, :Ilave, :SokumUzeriBoya, :KritikCalismasi,
       :CalismaTarihi, :RenkDerinligiId, :PipetlemeyeGonderildi, :Color, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :FiyatFarki do
  begin
    BoyaMaliyetiTL = BoyaMaliyeti * DovizKuru;
    KimyasalMaliyetiTL = KimyasalMaliyeti * DovizKuru;
    if (Migrasyon = 0) then MigrasyonStr = '';
    else if (Migrasyon = 1) then MigrasyonStr = 'Soğuk';
    else if (Migrasyon = 2) then MigrasyonStr = 'Sıcak';
    else MigrasyonStr = '??';


    if (PipetlemeyeGonderildi = 0) then PipetlemeyeGonderildiStr = '';
    else PipetlemeyeGonderildiStr = 'Evet';


    if (MusteriFiyatOnaySekli = 0) then MusteriFiyatOnaySekliStr = 'Fax';
    else if (MusteriFiyatOnaySekli = 1) then MusteriFiyatOnaySekliStr = 'İrsaliye';
    else MusteriFiyatOnaySekliStr = '';

    BoyaTipiFarkliUrunVar = 0;
    MinBoyaTipiId = 0;
    MaxBoyaTipiId = 0;

    Select Min(BoyaTipiId), Max(BoyaTipiId) from LabReceteUrun
      left join Urun on Urun.UrunId = LabReceteUrun.UrunId
      where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and Urun.UrunTipiID = 0
      into :MinBoyaTipiId, :MaxBoyaTipiId;

    if (MinBoyaTipiId <> MaxBoyaTipiId) then BoyaTipiFarkliUrunVar = 1;

    if (Rptmi = 1) then RptStr = 'Rpt';
    else RptStr = '';

    if (KritikCalismasi = 1) then KritikCalismasiStr = 'Kritik';
    else KritikCalismasiStr = '';

    if (Ilave = 1) then IlaveSokumUzeriBoyaStr = 'İlave';
    else if (SokumUzeriBoya = 1) then IlaveSokumUzeriBoyaStr = 'Söküm Üzeri Boya';
    else IlaveSokumUzeriBoyaStr = '';

    UrunSayisi = 0;

    For Select distinct LabReceteUrun.UrunID from LabReceteUrun
      where LabReceteUrun.LabReceteAtisID = :LabReceteAtisID
      into :FUrunID do
    begin
      UrunSayisi = UrunSayisi + 1;
    end

    PipetlemeKodu = EgemenIntegerToStr(LabReceteCalismaId) || '.' || AtisNoStr;

    if (AUrunFiyatGorebilir = 0) then
    begin
      BoyaMaliyeti     = Null;
      BoyaMaliyetiTL   = Null;
      KimyasalMaliyeti = Null;
      KimyasalMaliyetiTL = Null;
    end
    suspend;
  end
end
