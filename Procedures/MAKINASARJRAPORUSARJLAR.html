CREATE PROCEDURE MAKINASARJRAPORUSARJLAR(
    AUNITEID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAGRUBU VARCHAR(4),
    AMAKINAID VARCHAR(10))
RETURNS (
    ACIKLAMA VARCHAR(20),
    MAKINAID VARCHAR(10),
    MAKINASARJID INTEGER,
    IMALATAARAVERMEID INTEGER,
    SIRANO INTEGER,
    ID INTEGER,
    TOPLAMKILO NUMERIC(12,3),
    TOPLAMMETRE NUMERIC(12,3),
    ILKGIRISTARIHI TIMESTAMP,
    SONCIKISTARIHI TIMESTAMP,
    BOSGECENSURE DOUBLE PRECISION,
    BOSGECENSURESTR VARCHAR(20),
    ARAVERMESURESI DOUBLE PRECISION,
    ARAVERMESURESISTR VARCHAR(20),
    IMALATSURESI DOUBLE PRECISION,
    IMALATSURESISTR VARCHAR(20),
    ARAVERMESEBEBIIDSTR VARCHAR(55),
    ARAVERMESEBEBIADISTR VARCHAR(512),
    DOGALGAZSARFIYATI DOUBLE PRECISION,
    ELEKTIRIKSARFIYATI DOUBLE PRECISION,
    OKUNANSUSARFIYATI DOUBLE PRECISION,
    BASELEKTRIKBIRIMFIYATI DOUBLE PRECISION,
    BASDOGALGAZBIRIMFIYATI DOUBLE PRECISION,
    BASBUHARBIRIMFIYATI DOUBLE PRECISION,
    BASSUBIRIMFIYATI DOUBLE PRECISION,
    BITELEKTRIKBIRIMFIYATI DOUBLE PRECISION,
    BITDOGALGAZBIRIMFIYATI DOUBLE PRECISION,
    BITBUHARBIRIMFIYATI DOUBLE PRECISION,
    BITSUBIRIMFIYATI DOUBLE PRECISION)
AS
declare variable LastSonCikisTarihi TimeStamp;
declare variable GecenSure  Double precision;

declare variable FToplamKilo Numeric(12,3);
declare variable FToplamMetre           Numeric(12,3);
declare variable FIlkGirisTarihi        Timestamp;
declare variable FSonCikisTarihi        TimeStamp;
declare variable FAraVermeSuresi        Double Precision;

declare variable MDIlkGirisTarihi        Timestamp;
declare variable MDSonCikisTarihi        TimeStamp;
declare variable MDImalataAraVermeId     integer;
declare variable MDAciklama              varchar(50);
declare variable FMakinaId               varchar(10);
declare variable FMakinaSarjId           integer;
declare variable AraVermeSebebiId        integer;
declare variable AraVermeSebebiAdi       varchar(50);
declare variable FMsg varchar(255);
declare variable Ay integer;
declare variable Yil integer;
begin
  LastSonCikisTarihi = Null;
  SiraNo = 0;
  BosGecenSure = 0;
  FMakinaID = '';
  FMakinaSarjID = null;

  for select MakinaSarjRaporuPartiler.MakinaId, MakinaSarjRaporuPartiler.MakinaSarjId, Sum(Kilo), Sum(Metre),
             Min(GirisTarihi), Max(CikisTarihi), Max(AraVermeSuresi) from MakinaSarjRaporuPartiler(:AUniteId, :ABaslangicTarihi, :ABitisTarihi, :AMakinaGrubu, :AMakinaId)
    group by MakinaSarjRaporuPartiler.MakinaId, MakinaSarjRaporuPartiler.MakinaSarjId
  into :MakinaId, :MakinaSarjId, :FToplamKilo, :FToplamMetre, :FIlkGirisTarihi, :FSonCikistarihi, :FAraVermeSuresi do
  begin
    if (FMakinaId <> MakinaId) then LastSonCikisTarihi = Null;

    if (LastSonCikisTarihi is not null) then
    begin
      GecenSure = FIlkGirisTarihi - LastSonCikisTarihi;
      GecenSure = GecenSure * 24 * 60;

      if (GecenSure > 3) then
      begin
        ToplamKilo = 0;
        ToplamMetre = 0;
        ID = 0;

        for Select Md.Tarih, BMD.Tarih, AraVermeSebebi.Adi, ImalataAraVerme.ImalataAraVermeId from MakinaDurumu MD
          left join ImalataAraVerme on ImalataAraVerme.BasladiMakinaDurumuId = MD.MakinaDurumuId
          left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
          left join MakinaDurumu BMD on BMD.MakinaDurumuId = ImalataAraVerme.BittiMakinaDurumuId
          where MD.MakinaId = :MakinaId and MD.Tarih > :LastSonCikisTarihi and MD.Tarih < :FIlkGirisTarihi and ImalataAraVerme.ImalataAraVermeId is not null
          order by MD.MakinaId, MD.Tarih
         into :MDIlkGirisTarihi, :MDSonCikisTarihi, :MDAciklama, :MDImalataAraVermeId do
        begin
          if (MDIlkGirisTarihi > LastSonCikisTarihi) then
          begin
            GecenSure = MDIlkGirisTarihi - LastSonCikisTarihi;
            GecenSure = GecenSure * 24 * 60;

            if (GecenSure > 3) then
            begin
              Aciklama            = '!! BOŞ GEÇEN SÜRE !! ';
              ID = -1;
              ImalataAraVermeId   = Null;
              SiraNo = SiraNo + 1;

              DogalgazSarfiyati     = 0;
              ElektirikSarfiyati    = 0;
              OkunanSuSarfiyati     = 0;

              BasElektrikBirimFiyati   = 0;
              BasDogalGazBirimFiyati   = 0;
              BasBuharBirimFiyati      = 0;
              BasSuBirimFiyati         = 0;
              BitElektrikBirimFiyati   = 0;
              BitDogalGazBirimFiyati   = 0;
              BitBuharBirimFiyati      = 0;
              BitSuBirimFiyati         = 0;

              ToplamKilo           = 0;
              ToplamMetre          = 0;
              IlkGirisTarihi       = LastSonCikisTarihi;
              SonCikisTarihi       = MDIlkGirisTarihi;

              BosGecenSure = SonCikisTarihi - IlkGirisTarihi;
              BosGecenSureStr = TarihtoStr(BosGecenSure);

              AraVermeSuresi       = 0;
              AraVermeSuresiStr    = '';
              ImalatSuresi         = 0;
              ImalatSuresiStr      = '';
              suspend;
            end
          end

          ID = 0;
          IlkGirisTarihi = MDIlkGirisTarihi;
          SonCikisTarihi = MDSonCikisTarihi;
          Aciklama = MDAciklama;
          ImalataAraVermeId = MDImalataAraVermeId;
          BosGecenSure = 0;
          BosGecenSureStr = '';

          AraVermeSuresi = SonCikisTarihi - IlkGirisTarihi;

          ImalatSuresi = 0;
          ImalatSuresiStr = '';

          if (AraVermeSuresi <> 0) then AraVermeSuresiStr = TarihtoStr(AraVermeSuresi);
          else AraVermeSuresiStr = '';
          SiraNo = SiraNo + 1;
          DogalgazSarfiyati     = 0;
          ElektirikSarfiyati    = 0;
          OkunanSuSarfiyati     = 0;

          BasElektrikBirimFiyati   = 0;
          BasDogalGazBirimFiyati   = 0;
          BasBuharBirimFiyati      = 0;
          BasSuBirimFiyati         = 0;
          BitElektrikBirimFiyati   = 0;
          BitDogalGazBirimFiyati   = 0;
          BitBuharBirimFiyati      = 0;
          BitSuBirimFiyati         = 0;


          suspend;
          LastSonCikisTarihi = SonCikisTarihi;
        end

        if (SonCikisTarihi < FIlkGirisTarihi) then
        begin
          GecenSure = FIlkGirisTarihi - SonCikisTarihi;
          GecenSure = GecenSure * 24 * 60;
          if (GecenSure > 3) then
          begin
            Aciklama            = '!! BOŞ GEÇEN SÜRE !! ';
            ID = -1;
            ImalataAraVermeId   = Null;
            SiraNo = SiraNo + 1;

            ToplamKilo           = 0;
            ToplamMetre          = 0;
            IlkGirisTarihi       = SonCikisTarihi;
            SonCikisTarihi       = FIlkGirisTarihi;

            BosGecenSure = SonCikisTarihi - IlkGirisTarihi;
            BosGecenSureStr = TarihtoStr(BosGecenSure);

            DogalgazSarfiyati     = 0;
            ElektirikSarfiyati    = 0;
            OkunanSuSarfiyati     = 0;
            BasElektrikBirimFiyati   = 0;
            BasDogalGazBirimFiyati   = 0;
            BasBuharBirimFiyati      = 0;
            BasSuBirimFiyati         = 0;
            BitElektrikBirimFiyati   = 0;
            BitDogalGazBirimFiyati   = 0;
            BitBuharBirimFiyati      = 0;
            BitSuBirimFiyati         = 0;


            AraVermeSuresi       = 0;
            AraVermeSuresiStr    = '';
            ImalatSuresi         = 0;
            ImalatSuresiStr      = '';
            suspend;
          end
        end
      end
    end

    ImalataAraVermeId   = Null;
    LastSonCikisTarihi = FSonCikistarihi;

    ToplamKilo     = FToplamKilo;
    ToplamMetre    = FToplamMetre;
    IlkGirisTarihi = FIlkGirisTarihi;
    SonCikistarihi = FSonCikistarihi;
    AraVermeSuresi = FAraVermeSuresi;
    BosGecenSure    = 0;
    BosGecenSureStr = '';

    SiraNo = SiraNo + 1;
    ID = 1;

    Aciklama = 'İMALAT';
    ImalatSuresi = SonCikisTarihi - IlkGirisTarihi;
    ImalatSuresi = ImalatSuresi - AraVermeSuresi;

    ImalatSuresiStr = TarihtoStr(ImalatSuresi);

    AraVermeSebebiIdStr  = '';
    AraVermeSebebiAdiStr = '';

    if (AraVermeSuresi <> 0) then
    begin
      AraVermeSuresiStr = TarihtoStr(AraVermeSuresi);

      for Select ImalataAraVerme.AraVermeSebebiId, AraVermeSebebi.Adi from MakinaDurumu MD
        left join ImalataAraVerme on ImalataAraVerme.BasladiMakinaDurumuId = MD.MakinaDurumuId
        left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
        where MD.MakinaId = :MakinaId and MD.Tarih >= :IlkGirisTarihi and MD.Tarih <= :SonCikisTarihi and ImalataAraVerme.ImalataAraVermeId is not null
        group by ImalataAraVerme.AraVermeSebebiId,  AraVermeSebebi.Adi
        into :AraVermeSebebiId, :AraVermeSebebiAdi do
      begin
        if (Char_Length(AraVermeSebebiIdStr) + Char_Length(EgemenIntegertoStr(AraVermeSebebiId)) < 50) then
        begin
          if (AraVermeSebebiIdStr <> '' ) then AraVermeSebebiIdStr = AraVermeSebebiIdStr || cast(', ' as varchar(2) character set UTF8);
          AraVermeSebebiIdStr = AraVermeSebebiIdStr || EgemenIntegertoStr(AraVermeSebebiId);
        end

        if (Char_Length(AraVermeSebebiAdiStr) + Char_Length(AraVermeSebebiAdi) < 500) then
        begin
          if (AraVermeSebebiAdiStr <> '' ) then AraVermeSebebiAdiStr = AraVermeSebebiAdiStr || cast(', ' as varchar(2) character set UTF8);
          AraVermeSebebiAdiStr = AraVermeSebebiAdiStr || AraVermeSebebiAdi;
        end
      end
    end
    else AraVermeSuresiStr = '';

    Select Dogalgaz, Elektirik, OkunanSu from MakinaSarj where MakinaSarj.MakinaSarjID = :MakinaSarjID into :DogalgazSarfiyati, :ElektirikSarfiyati, :OkunanSuSarfiyati;

    Yil = f_Year(IlkGirisTarihi);
    Ay  = f_Month(IlkGirisTarihi);

    Select ElektrikBirimFiyati, DogalGazBirimFiyati, BuharBirimMaliyet, SuBirimFiyati from AylikMasraf
      where AylikMasraf.Yil = :Yil and AylikMasraf.Ay = :Ay
      into :BasElektrikBirimFiyati, :BasDogalGazBirimFiyati, :BasBuharBirimFiyati, :BasSuBirimFiyati;

    Yil = f_Year(SonCikisTarihi);
    Ay  = f_Month(SonCikisTarihi);

    Select ElektrikBirimFiyati, DogalGazBirimFiyati, BuharBirimMaliyet, SuBirimFiyati from AylikMasraf
      where AylikMasraf.Yil = :Yil and AylikMasraf.Ay = :Ay
      into :BitElektrikBirimFiyati, :BitDogalGazBirimFiyati, :BitBuharBirimFiyati, :BitSuBirimFiyati;


    suspend;
    AraVermeSebebiIdStr  = '';
    AraVermeSebebiAdiStr = '';
    FMakinaID = MakinaId;
    FMakinaSarjID = MakinaSarjId;
  end
end
