CREATE PROCEDURE ORMESIPARISHAREKETOZETI(
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AORMEMAKINAID VARCHAR(10),
    AFASONURETIMECIKIS INTEGER,
    AURETIMECIKIS INTEGER,
    AFASONURETIM INTEGER,
    AURETIM INTEGER)
RETURNS (
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    ORMEYECIKANIPLIK DOUBLE PRECISION,
    ORMEDENIADEIPLIK DOUBLE PRECISION,
    ORMEYECIKANNETIPLIK DOUBLE PRECISION,
    ORMEYECIKANLYCRA DOUBLE PRECISION,
    ORMEDENIADELYCRA DOUBLE PRECISION,
    ORMEYECIKANNETLYCRA DOUBLE PRECISION,
    ORULEN DOUBLE PRECISION,
    ORMEFIRE DOUBLE PRECISION,
    ORMELYCRAKULLANIMORANI DOUBLE PRECISION,
    ORMEFIREORANI DOUBLE PRECISION,
    BOYAYACIKAN DOUBLE PRECISION,
    BOYADANIADE DOUBLE PRECISION,
    BOYAYACIKANNET DOUBLE PRECISION,
    BOYANAN DOUBLE PRECISION,
    BRUTBOYANAN DOUBLE PRECISION,
    BOYAFIRE DOUBLE PRECISION,
    BOYAFIREORANI DOUBLE PRECISION,
    BUKUMECIKANIPLIK DOUBLE PRECISION,
    BUKUMDENIADEIPLIK DOUBLE PRECISION,
    BUKUMECIKANNETIPLIK DOUBLE PRECISION,
    BUKUMECIKANLYCRA DOUBLE PRECISION,
    BUKUMDENIADELYCRA DOUBLE PRECISION,
    BUKUMECIKANNETLYCRA DOUBLE PRECISION,
    BUKULEN DOUBLE PRECISION,
    BUKUMFIRE DOUBLE PRECISION,
    BUKUMLYCRAKULLANIMORANI DOUBLE PRECISION,
    BUKUMFIREORANI DOUBLE PRECISION)
AS
declare variable FHamTipiId integer;
declare variable FHareketTipi integer;
declare variable FHamIade integer;
declare variable FMiktar  Double Precision;
declare variable FBrutMiktar  Double Precision;
declare variable hstIplik       integer;
declare variable hstLycra       integer;
declare variable uitBukum integer;
declare variable uitOrme integer;
declare variable uitBoya integer;



declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;


declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable Okey integer;
declare variable FireHesapTipi integer;
declare variable BoyaSiparisFirmaId varchar(20);








declare variable FBoyaSiparisKayitId integer;
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
   into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
        :shtFasonUretim, :ShtUretim;

  select uitBukum, uitOrme, uitBoya from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya;

  Select FireHesapTipi from Sistem where SistemId = 1 into :FireHesapTipi;

  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  Select BoyaSiparisKayitId, FirmaId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId into :FBoyaSiparisKayitId, :BoyaSiparisFirmaId;

  if (ABoyaSiparisYil is not null and ABoyaSiparisId is not null and FBoyaSiparisKayitId is null) then exit;

  For Select OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.HamId, Ham.Adi,
      OrmeSiparisDetay.RenkId, Renk.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4 from OrmeSiparisDetay
    left join Ham On Ham.HamID = OrmeSiparisDetay.HamId
    left join Renk On Renk.RenkID = OrmeSiparisDetay.RenkId
    where OrmeSiparisDetay.BoyaSiparisKayitId = :FBoyaSiparisKayitId
    into :OrmeSiparisDetayId, :SiparisSiraNo, :HamId, :HamAdi, :RenkId, :RenkAdi, :HamAdi2, :HamAdi3, :HamAdi4 do
  begin
    OrmeyeCikanIplik          = 0;
    OrmedenIadeIplik          = 0;
    OrmeyeCikanNetIplik       = 0;
    OrmeyeCikanLycra          = 0;
    OrmedenIadeLycra          = 0;
    OrmeyeCikanNetLycra       = 0;
    Orulen                    = 0;
    OrmeFire                  = 0;
    OrmeLycraKullanimOrani    = 0;
    OrmeFireOrani             = 0;
    BoyayaCikan               = 0;
    BoyadanIade               = 0;
    BoyayaCikanNet            = 0;
    Boyanan                   = 0;
    BrutBoyanan               = 0;
    BoyaFire                  = 0;
    BoyaFireOrani             = 0;


    for Select Ham.HamTipiId, StokHareket.HareketTipi, StokHareket.HamIade, StokHareket.Miktar, StokHareket.BrutMiktar from StokHareket
    left join Ham on Ham.HamId = StokHareket.HamId
    where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and
          (((StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtFasonUretim) and StokHareket.FirmaId = :AFirmaId) or :AFirmaId is null) and
          (StokHareket.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
          ((StokHareket.HareketTipi = :shtFasonUretimeCikis and :AFasonUretimeCikis = 1) or
           (StokHareket.HareketTipi = :shtFasonUretim and :AFasonUretim = 1) or
           (StokHareket.HareketTipi = :shtUretimeCikis and :AUretimeCikis = 1) or
           (StokHareket.HareketTipi = :shtUretim and :AUretim = 1) ) and
            StokHareket.UretimIslemTipiId = :uitOrme
    order by StokHareket.Tarih, StokHareket.StokHareketId
    into :FHamTipiId, :FHareketTipi, :FHamIade, :FMiktar, :FBrutMiktar do
    begin
      if (FHareketTipi = shtFasonUretimeCikis or FHareketTipi = shtUretimeCikis) then
      begin
        if (FHamTipiId = hstIplik) then OrmeyeCikanIplik = OrmeyeCikanIplik + FMiktar;
        else if (FHamTipiId = hstLycra) then OrmeyeCikanLycra = OrmeyeCikanLycra + FMiktar;
      end
      else if (FHareketTipi = shtFasonUretim or FHareketTipi = shtUretim) then
      begin
        if (FHamIade = 0) then Orulen = Orulen + FMiktar;
        else
        begin
          if (FHamTipiId = hstIplik) then OrmedenIadeIplik = OrmedenIadeIplik + FMiktar;
          else if (FHamTipiId = hstLycra) then OrmedenIadeLycra = OrmedenIadeLycra + FMiktar;
        end
      end
    end
    OrmeyeCikanNetIplik = OrmeyeCikanIplik - OrmedenIadeIplik;
    OrmeyeCikanNetLycra = OrmeyeCikanLycra - OrmedenIadeLycra;

    OrmeFire = OrmeyeCikanNetIplik - (Orulen - OrmeyeCikanNetLycra);
    if (Orulen > 0) then OrmeLycraKullanimOrani = (OrmeyeCikanNetLycra / Orulen) * 100;

    OrmeFireOrani = Null;

    if (FireHesapTipi = 0) then
    begin
      if ((Orulen - OrmeyeCikanNetLycra)> 0) then OrmeFireOrani = (OrmeFire / (Orulen - OrmeyeCikanNetLycra)) * 100;
    end
    else
    begin
      if (OrmeyeCikanNetIplik  > 0) then OrmeFireOrani = (OrmeFire / OrmeyeCikanNetIplik ) * 100;
    end

    OrmeFireOrani = Yuvarla(OrmeFireOrani, 100);
    /*b端k端m*/
    BukumeCikanIplik          = 0;
    BukumdenIadeIplik          = 0;
    BukumeCikanNetIplik       = 0;
    BukumeCikanLycra          = 0;
    BukumdenIadeLycra          = 0;
    BukumeCikanNetLycra       = 0;
    Bukulen                    = 0;
    BukumFire                  = 0;
    BukumLycraKullanimOrani    = 0;
    BukumFireOrani             = 0;

    for Select Ham.HamTipiId, StokHareket.HareketTipi, StokHareket.HamIade, StokHareket.Miktar, StokHareket.BrutMiktar from StokHareket
    left join Ham on Ham.HamId = StokHareket.HamId
    where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and
          (((StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtFasonUretim) and StokHareket.FirmaId = :AFirmaId) or :AFirmaId is null) and
          (StokHareket.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
          ((StokHareket.HareketTipi = :shtFasonUretimeCikis and :AFasonUretimeCikis = 1) or
           (StokHareket.HareketTipi = :shtFasonUretim and :AFasonUretim = 1) or
           (StokHareket.HareketTipi = :shtUretimeCikis and :AUretimeCikis = 1) or
           (StokHareket.HareketTipi = :shtUretim and :AUretim = 1) ) and
            StokHareket.UretimIslemTipiId = :uitBukum
    order by StokHareket.Tarih, StokHareket.StokHareketId
    into :FHamTipiId, :FHareketTipi, :FHamIade, :FMiktar, :FBrutMiktar do
    begin
      if (FHareketTipi = shtFasonUretimeCikis or FHareketTipi = shtUretimeCikis) then
      begin
        if (FHamTipiId = hstIplik) then BukumeCikanIplik = BukumeCikanIplik + FMiktar;
        else if (FHamTipiId = hstLycra) then BukumeCikanLycra = BukumeCikanLycra + FMiktar;
      end
      else if (FHareketTipi = shtFasonUretim or FHareketTipi = shtUretim) then
      begin
        if (FHamIade = 0) then Bukulen = Bukulen + FMiktar;
        else
        begin
          if (FHamTipiId = hstIplik) then BukumdenIadeIplik = BukumdenIadeIplik + FMiktar;
          else if (FHamTipiId = hstLycra) then BukumdenIadeLycra = BukumdenIadeLycra + FMiktar;
        end
      end
    end

    BukumeCikanNetIplik = BukumeCikanIplik - BukumdenIadeIplik;
    BukumeCikanNetLycra = BukumeCikanLycra - BukumdenIadeLycra;

    BukumFire = BukumeCikanNetIplik - (Bukulen - BukumeCikanNetLycra);
    if (Bukulen > 0) then BukumLycraKullanimOrani = (BukumeCikanNetLycra / Bukulen) * 100;

    BukumFireOrani = Null;

    if (FireHesapTipi = 0) then
    begin
      if ((Bukulen - BukumeCikanNetLycra)> 0) then BukumFireOrani = (BukumFire / (Bukulen - BukumeCikanNetLycra)) * 100;
    end
    else
    begin
      if (BukumeCikanNetIplik  > 0) then BukumFireOrani = (BukumFire / BukumeCikanNetIplik ) * 100;
    end

    BukumFireOrani = Yuvarla(BukumFireOrani, 100);
    /*b端k端m*/

    suspend;
  end


  For Select BoyaSiparisDetay.OrmeSiparisDetayId, BoyaSiparisDetay.BoyaSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo,
      BoyaSiparisDetay.HamId, Ham.Adi, BoyaSiparisDetay.RenkAdi, Ham.Adi2, Ham.Adi3, Ham.Adi4 from BoyaSiparisDetay
    left Join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = BoyaSiparisDetay.OrmeSiparisDetayID
    left join Ham On Ham.HamID = BoyaSiparisDetay.HamId
    where BoyaSiparisDetay.BoyaSiparisKayitId = :FBoyaSiparisKayitId
    into :OrmeSiparisDetayId, :BoyaSiparisDetayId, :SiparisSiraNo, :HamId, :HamAdi, :RenkAdi, :HamAdi2, :HamAdi3, :HamAdi4 do
  begin
    RenkId = Null;
    OrmeyeCikanIplik          = 0;
    OrmedenIadeIplik          = 0;
    OrmeyeCikanNetIplik       = 0;
    OrmeyeCikanLycra          = 0;
    OrmedenIadeLycra          = 0;
    OrmeyeCikanNetLycra       = 0;
    Orulen                    = 0;
    OrmeFire                  = 0;
    OrmeLycraKullanimOrani    = 0;
    OrmeFireOrani             = 0;
    BoyayaCikan               = 0;
    BoyadanIade               = 0;
    BoyayaCikanNet            = 0;
    Boyanan                   = 0;
    BrutBoyanan               = 0;
    BoyaFire                  = 0;
    BoyaFireOrani             = 0;

    for Select Ham.HamTipiId, StokHareket.HareketTipi, StokHareket.HamIade, StokHareket.Miktar, StokHareket.BrutMiktar from StokHareket
    left join Ham on Ham.HamId = StokHareket.HamId
    where StokHareket.BoyaSiparisDetayId = :BoyaSiparisDetayId and
          (((StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtFasonUretim) and StokHareket.FirmaId = :AFirmaId) or :AFirmaId is null) and
          ((StokHareket.HareketTipi = :shtFasonUretimeCikis and :AFasonUretimeCikis = 1) or
           (StokHareket.HareketTipi = :shtFasonUretim and :AFasonUretim = 1) or
           (StokHareket.HareketTipi = :shtUretimeCikis and :AUretimeCikis = 1) or
           (StokHareket.HareketTipi = :shtUretim and :AUretim = 1) ) and
           StokHareket.UretimIslemTipiId = :uitBoya
    order by StokHareket.Tarih, StokHareket.StokHareketId
    into :FHamTipiId, :FHareketTipi, :FHamIade, :FMiktar, :FBrutMiktar do
    begin
      if (FHareketTipi = shtFasonUretimeCikis or FHareketTipi = shtUretimeCikis) then
      begin
        BoyayaCikan = BoyayaCikan + FMiktar;
      end
      else if (FHareketTipi = shtFasonUretim or FHareketTipi = shtUretim) then
      begin
        if (FHamIade = 0) then
        begin
          Boyanan = Boyanan + FMiktar;
          BrutBoyanan = BrutBoyanan + FBrutMiktar;
        end
        else BoyadanIade = BoyadanIade + FMiktar;
      end
    end

    FBrutMiktar = 0;
    FMiktar = 0;
    Select Sum(BoyaIrsaliyeDetay.BrutKilo), Sum(BoyaIrsaliyeDetay.Kilo) from PartiStok
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifID = PartiStok.PartiSinifID
    where PartiStok.BoyaSiparisDetayID = :BoyaSiparisDetayID and BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID is not null and BoyaIrsaliyeDetay.FirmaID = :BoyaSiparisFirmaId
    into :FBrutMiktar, :FMiktar;

    if (FBrutMiktar is null) then FBrutMiktar = 0;
    if (FMiktar is null) then FMiktar = 0;
    BrutBoyanan = BrutBoyanan + FBrutMiktar;
    Boyanan = Boyanan + FMiktar;

    BoyayaCikanNet      = BoyayaCikan - BoyadanIade;
    BoyaFire            = BrutBoyanan - Boyanan;

    if (BrutBoyanan > 0) then BoyaFireOrani = (BoyaFire / BrutBoyanan) * 100;
    else BoyaFireOrani = 0;

    BoyaFireOrani = Yuvarla(BoyaFireOrani, 100);

    suspend;
  end
end
