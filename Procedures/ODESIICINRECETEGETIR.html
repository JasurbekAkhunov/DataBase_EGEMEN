CREATE PROCEDURE ODESIICINRECETEGETIR(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    DOZAJLAMAURUNID VARCHAR(15),
    URUNTIPIID INTEGER,
    DOZAJLAMAPROGRAMNO INTEGER,
    SABLONDOZAJLAMAPROGRAMNO INTEGER,
    OTOMASYONMIKTAR DOUBLE PRECISION,
    DOZAJLAMABIRIMKODU VARCHAR(10),
    OTOMASYONRECETEMIKTAR DOUBLE PRECISION,
    DOZAJLAMASEKLI INTEGER,
    ISTEKNO INTEGER,
    COUNTER INTEGER,
    SIRANO INTEGER,
    URETIMECIKISFIYATIBUGUN DOUBLE PRECISION,
    MAKINAID VARCHAR(10),
    OTOMASYONMAKINANO VARCHAR(10),
    TANKNO VARCHAR(5))
AS
declare variable Miktar                Double precision;
declare variable Banyo                 Double precision;
declare variable ReceteMiktar          Double precision;
declare variable ReceteBanyo           Double precision;

declare variable ReceteAsamaId         integer;
declare variable ReceteAltAsamaId      integer;
declare variable LastReceteAsamaId integer;
declare variable LastReceteAltAsamaId integer;
declare variable LastSiraNo integer;
declare variable TuzKimyasalTipiID integer;
declare variable FKimyasalTipiId integer;
declare variable EgemenFirmaId   integer;
declare variable TartimYapilacak EgeBoolean;
begin
  LastReceteAsamaId = -2;
  LastReceteAltAsamaId = -2;
  LastSiraNo = -1;
  IstekNo = 0;

  select TuzKimyasalTipiID, EgemenFirmaId from Sistem into :TuzKimyasalTipiID, :EgemenFirmaId;

  for select DozajlamaIcinReceteHazirla.UrunId, DozajlamaIcinReceteHazirla.UrunAdi, DozajlamaIcinReceteHazirla.DozajlamaUrunId, DozajlamaIcinReceteHazirla.UrunTipiId,
             DozajlamaIcinReceteHazirla.DozajlamaProgramNo, DozajlamaIcinReceteHazirla.SablonDozajlamaProgramNo, DozajlamaIcinReceteHazirla.DozajlamaBirimKodu,
             DozajlamaIcinReceteHazirla.Miktar, DozajlamaIcinReceteHazirla.Banyo,
             DozajlamaIcinReceteHazirla.ReceteMiktar, DozajlamaIcinReceteHazirla.ReceteBanyo,
             DozajlamaIcinReceteHazirla.ReceteAsamaId, DozajlamaIcinReceteHazirla.ReceteAltAsamaId, DozajlamaIcinReceteHazirla.DozajlamaSekli,
             DozajlamaIcinReceteHazirla.SiraNo, DozajlamaIcinReceteHazirla.UretimeCikisFiyatiBugun, DozajlamaIcinReceteHazirla.MakinaID, DozajlamaIcinReceteHazirla.OtomasyonMakinaNo, Urun.KimyasalTipiId,
             Urun.TartimYapilacak from DozajlamaIcinReceteHazirla(:APartiKayitId, :AReceteId, :APartiAsamalariId, 4)
    Left Join Urun on Urun.UrunID = DozajlamaIcinReceteHazirla.UrunID
    where DozajlamaIcinReceteHazirla.UrunID is not null
    into :UrunId, :UrunAdi, :DozajlamaUrunId, :UrunTipiId, :DozajlamaProgramNo, :SablonDozajlamaProgramNo, :DozajlamaBirimKodu, :Miktar, :Banyo,
         :ReceteMiktar, :ReceteBanyo, :ReceteAsamaId, :ReceteAltAsamaId, :DozajlamaSekli, :SiraNo, :UretimeCikisFiyatiBugun, :MakinaID, :OtomasyonMakinaNo, :FKimyasalTipiId,
         :TartimYapilacak do
  begin
    if (ReceteAsamaId <> LastReceteAsamaID or SiraNo <> LastSiraNo) then
    begin
      LastReceteAsamaId = ReceteAsamaId;
      LastReceteAltAsamaId = -2;
      IstekNo = IstekNo + 1;
      if (ReceteAltAsamaId <> LastReceteAltAsamaId) then LastReceteAltAsamaId = ReceteAltAsamaId;
      Counter = 0;
      LastSiraNo = SiraNo;
    end

    if (ReceteAltAsamaId <> LastReceteAltAsamaId) then
    begin
      LastReceteAltAsamaId = ReceteAltAsamaId;
      IstekNo = IstekNo + 1;
      Counter = 0;
    end

    TankNo = 'T01';
    if (FKimyasalTipiId = TuzKimyasalTipiID) then TankNo = 'T02';

    Counter = Counter + 1;
    if (Miktar > 0) then OtomasyonMiktar = Miktar;
    else OtomasyonMiktar = Banyo;

    if (ReceteMiktar > 0) then OtomasyonReceteMiktar = ReceteMiktar;
    else OtomasyonReceteMiktar = ReceteBanyo;

    OtomasyonMiktar = Trunc(OtomasyonMiktar);

    if (EgemenFirmaId <> 76) then /*new jacard*/
      suspend;
    else if (UrunTipiId <> 1 or TartimYapilacak = 0)  then
      suspend;
  end
end
