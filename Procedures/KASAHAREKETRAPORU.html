CREATE PROCEDURE KASAHAREKETRAPORU(
    AKASAID VARCHAR(10),
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABORCALACAK VARCHAR(1),
    ASINIF VARCHAR(1),
    ARAPORDOVIZID VARCHAR(5),
    AISLEMDOVIZID VARCHAR(5))
RETURNS (
    KASAHAREKETID INTEGER,
    KASAID VARCHAR(10),
    TARIH TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BANKAHESAPID VARCHAR(20),
    BANKAHESAPADI VARCHAR(50),
    ACIKLAMA VARCHAR(50),
    KASAHARCAMATIPIID VARCHAR(10),
    KASAHARCAMATIPIADI VARCHAR(30),
    BORCALACAK VARCHAR(1),
    SINIF VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    DOVIZKURUSTR VARCHAR(40),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    ISLEMDOVIZMEBLAG DOUBLE PRECISION,
    ISLEMDOVIZID VARCHAR(5))
AS
declare Variable ToplamBorc double precision;
declare variable ToplamAlacak double precision;
declare variable DovizKuru Double Precision;
declare variable DovizKuru1 Double Precision;
declare variable DovizKuru2 Double Precision;
declare variable FirmaDovizId varchar(5);
declare variable BankaDovizId varchar(5);
declare variable HedefDovizId varchar(5);
declare variable KasaDovizId varchar(5);
declare variable IslemDovizKuru1 Double Precision;
declare variable IslemDovizKuru2 Double Precision;
begin
  for select KasaHareket.KasaId, KasaHareket.Tarih, KasaHareket.KasaHareketId, Firma.FirmaId, Firma.Adi, Firma.DovizId, KasaHareket.BankaHesapId, BankaHesap.Adi, BankaHesap.DovizId, KasaHareket.Aciklama,
             KasaHareket.BorcAlacak, KasaHareket.Sinif, KasaHareket.Meblag, KasaHareket.DovizKuru1, KasaHareket.DovizKuru2, KasaHareket.Notu, KasaHareket.KasaHarcamaTipiId,
             KasaHarcamaTipi.Adi, Kullanici.Adi, KasaHareket.DegisimTarihi, Kasa.DovizId, KasaHareket.IslemDovizMeblag, KasaHareket.IslemDovizID, KasaHareket.IslemDovizKuru1, KasaHareket.IslemDovizKuru2 from KasaHareket
  left join Firma on Firma.FirmaId = KasaHareket.FirmaId
  left join BankaHesap on BankaHesap.BankaHesapId = KasaHareket.BankaHesapId
  left join KasaHarcamaTipi on KasaHarcamaTipi.KasaHarcamaTipiId = KasaHareket.KasaHarcamaTipiId
  left join Kullanici on Kullanici.KullaniciId = KasaHareket.KullaniciId
  left join Kasa on Kasa.KasaId = KasaHareket.KasaId
  where (KasaHareket.KasaId = :AKasaId or :AKasaId is Null) and KasaHareket.Tarih >= :ABaslangicTarihi and
  KasaHareket.Tarih <= :ABitisTarihi and (KasaHareket.Sinif = :ASinif or :ASinif is null)
  and (KasaHareket.FirmaId = :AFirmaId or :AFirmaId is Null)
  and (KasaHareket.BorcAlacak = :ABorcAlacak or :ABorcAlacak is Null)
  and (KasaHareket.IslemDovizID = :AIslemDovizId or :AIslemDovizId is Null)
  order by KasaHareket.Tarih
  into :KasaId, :Tarih, :KasaHareketId, :FirmaId, :FirmaAdi, :FirmaDovizId, :BankaHesapId, :BankaHesapAdi, :BankaDovizId, :Aciklama,
       :BorcAlacak, :Sinif,  :Meblag, :DovizKuru1, :DovizKuru2, :Notu, :KasaHarcamaTipiId, :KasaHarcamaTipiAdi, :KullaniciAdi, :DegisimTarihi, :KasaDovizId, :IslemDovizMeblag, :IslemDovizID, :IslemDovizKuru1, :IslemDovizKuru2 do
  begin
    Borc = Null;
    Alacak = Null;

    if (BorcAlacak = 'B') then Borc = Meblag;
    else Alacak = Meblag;

    if (BankaDovizId is not null) then HedefDovizId = BankaDovizId;
    else if (FirmaDovizId is not null) then HedefDovizId = FirmaDovizId;
    else HedefDovizId = KasaDovizId;

    DovizKuru=1;
    if (KasaDovizID <> HedefDovizId) then
    begin
      DovizKuruStr = Format_Float('', IslemDovizKuru1) || ' ' || KasaDovizId || '-' || Format_Float('', IslemDovizKuru2) || ' ' || IslemDovizId;
      DovizKuru = IslemDovizKuru2 / IslemDovizKuru1;
    end
    else DovizKuruStr = '';

    Meblag=Meblag*DovizKuru;
    Borc=Borc*DovizKuru;
    Alacak=Alacak*DovizKuru;

    if (IslemDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:IslemDovizId, :ARaporDovizId, :Tarih) into :DovizKuru;
    else DovizKuru = 1;

    Meblag=Meblag*DovizKuru;
    Borc=Borc*DovizKuru;
    Alacak=Alacak*DovizKuru;

    suspend;
  end
end
