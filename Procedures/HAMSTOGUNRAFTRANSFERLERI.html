CREATE PROCEDURE HAMSTOGUNRAFTRANSFERLERI(
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    RAFTRANSFERFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    ESKIRAFID VARCHAR(10),
    YENIRAFID VARCHAR(10),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3))
AS
begin
  For Select RafTransferFisi.RafTransferFisiId, RafTransferFisi.FisYil, RafTransferFisi.FisNo, RafTransferFisi.FisTarihi,
             RafDetay.RafId, RafTransferFisiDetay.RafId, RafTransferFisiDetay.Kilo, RafTransferFisiDetay.Metre from HamStokDetay
  left join RafDetay on RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId
  left join RafTransferFisiDetay on RafTransferFisiDetay.RafDetayId = RafDetay.RafDetayId
  left join RafTransferFisi on RafTransferFisi.RafTransferFisiId = RafTransferFisiDetay.RafTransferFisiId
  where HamStokDetay.HamStokMasterID = :AHamStokMasterId and RafDetay.RafDetayId is not null and RafTransferFisiDetay.RafTransferFisiId is not null
  into :RafTransferFisiId, :FisYil, :FisNo, :FisTarihi, :EskiRafId, :YeniRafId, :Kilo, Metre do
  begin
    FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';
    suspend;
  end
end
