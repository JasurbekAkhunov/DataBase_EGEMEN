CREATE PROCEDURE IADEHAMSTOKRAPORU(
    AIADETIPI INTEGER,
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ARAPORTARIHTIPI INTEGER,
    AHAMSTOKTIPI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    SEVKTARIHI TIMESTAMP,
    ARACID VARCHAR(11),
    HAMSTOKMASTERID INTEGER,
    MIKTAR NUMERIC(12,3),
    GIRISIRSALIYENO IRSALIYENOSIZE,
    GIRISTARIHI TIMESTAMP,
    GIRISEVRAKTIPI INTEGER,
    GIRISEVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30),
    ORMECIMAKINANO VARCHAR(10),
    HAMADI VARCHAR(255),
    TAMIR INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    PUS INTEGER,
    FEINE INTEGER,
    RENK VARCHAR(20),
    LOT VARCHAR(150),
    MUSTERIPARTINO VARCHAR(30),
    ESKIPARTIYIL INTEGER,
    ESKIPARTINO INTEGER,
    HAMSTOKIADESEBEBIID INTEGER,
    HAMSTOKIADESEBEBIADI VARCHAR(30),
    RAFID VARCHAR(10))
AS
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef       integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  if (ARaporTarihTipi = 0) then
    For Select BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.SevkTarihi,
               BoyaIrsaliye.AracId, BoyaIrsaliyeDetay.Miktar, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
               HamStokDetay.HamStokMasterId, Ormeci.Adi, HamStokMaster.OrmecimakinaNo,
               Ham.Adi, TamirTipi.Adi, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Renk, HamStokMaster.Lot, HamStokMaster.MusteriPartiNo,
               RafDetay.RafId, HamIadeRezerv.HamStokIadeSebebiId, HamStokIadeSebebi.Adi,
               HamStokMaster.Tamir, HamStokMaster.EskiPartiNo, HamStokMaster.EskiPartiYil from BoyaIrsaliye
    left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
    left join RafDetay on RafDetay.RafDetayId = BoyaIrsaliyeDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
    left join Ormeci on Ormeci.OrmeciId = HamStokIrsaliye.OrmeciId
    left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join HamIadeRezerv on HamIadeRezerv.HamIadeRezervid = BoyaIrsaliyeDetay.HamIadeRezervId
    left join HamStokIadeSebebi on HamStokIadeSebebi.HamStokIadeSebebiid = HamIadeRezerv.HamStokIadeSebebiid
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
         (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and BoyaIrsaliyeDetay.RafDetayId is not null and
         ((:AIadeTipi = 0 and BoyaIrsaliyeDetay.FasonHamIade = 1) or (:AIadeTipi = 1 and BoyaIrsaliyeDetay.HareketTipiId = 1)) and
         ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamStokTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf  and Ham.HamTipiId <> :hstTelef and :AHamStokTipi = 1) or :AHamStokTipi is null)
    order by BoyaIrsaliye.FirmaId, BoyaIrsaliye.IrsaliyeTarihi
    into :FirmaId, :FirmaAdi, :IrsaliyeNo, :IrsaliyeTarihi, :SevkTarihi, :AracId, :Miktar, :GirisIrsaliyeNo, :GirisTarihi, :GirisEvrakTipi,
         :HamStokMasterId, :OrmeciAdi, :OrmecimakinaNo, :HamAdi, :TamirTipiAdi, :Pus, :Feine, :Renk, :Lot, :MusteriPartiNo, :RafId,
         :HamStokIadeSebebiId, :HamStokIadeSebebiAdi, :Tamir, :EskiPartiNo, :EskiPartiYil do
  begin
    if (GirisEvrakTipi = 0) then GirisEvrakTipiStr = 'Fiş';
    else if (GirisEvrakTipi = 1) then GirisEvrakTipiStr = 'İrsaliye';
    else GirisEvrakTipiStr = 'Hatalı';
    suspend;
  end
  else
    For Select BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.SevkTarihi,
               BoyaIrsaliye.AracId, BoyaIrsaliyeDetay.Miktar, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, Ormeci.Adi,
               HamStokMaster.OrmecimakinaNo, Ham.Adi, TamirTipi.Adi, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Renk,
               HamStokMaster.Lot, HamStokMaster.MusteriPartiNo, RafDetay.RafId, HamIadeRezerv.HamStokIadeSebebiId, HamStokIadeSebebi.Adi,
               HamStokMaster.Tamir, HamStokMaster.EskiPartiNo, HamStokMaster.EskiPartiYil from BoyaIrsaliye
    left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
    left join RafDetay on RafDetay.RafDetayId = BoyaIrsaliyeDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
    left join Ormeci on Ormeci.OrmeciId = HamStokIrsaliye.OrmeciId
    left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join HamIadeRezerv on HamIadeRezerv.HamIadeRezervid = BoyaIrsaliyeDetay.HamIadeRezervId
    left join HamStokIadeSebebi on HamStokIadeSebebi.HamStokIadeSebebiid = HamIadeRezerv.HamStokIadeSebebiid
    where BoyaIrsaliye.Sevktarihi >= :ABaslangicTarihi and BoyaIrsaliye.Sevktarihi <= :ABitisTarihi and
         (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and BoyaIrsaliyeDetay.RafDetayId is not null and
         ((:AIadeTipi = 0 and BoyaIrsaliyeDetay.FasonHamIade = 1) or (:AIadeTipi = 1 and BoyaIrsaliyeDetay.HareketTipiId = 1)) and
         ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamStokTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf  and Ham.HamTipiId <> :hstTelef and :AHamStokTipi = 1) or :AHamStokTipi is null)
    order by BoyaIrsaliye.FirmaId, BoyaIrsaliye.SevkTarihi
    into :FirmaId, :FirmaAdi, :IrsaliyeNo, :IrsaliyeTarihi, :SevkTarihi, :AracId, :Miktar, :GirisIrsaliyeNo, :GirisTarihi, :GirisEvrakTipi, :OrmeciAdi,
         :OrmecimakinaNo, :HamAdi, :TamirTipiAdi, :Pus, :Feine, :Renk, :Lot, :MusteriPartiNo, :RafId,
         :HamStokIadeSebebiId, :HamStokIadeSebebiAdi, :Tamir, :EskiPartiNo, :EskiPartiYil do
  begin
    if (GirisEvrakTipi = 0) then GirisEvrakTipiStr = 'Fiş';
    else if (GirisEvrakTipi = 1) then GirisEvrakTipiStr = 'İrsaliye';
    else GirisEvrakTipiStr = 'Hatalı';
    suspend;
  end
end
