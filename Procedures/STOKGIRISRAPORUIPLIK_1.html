CREATE PROCEDURE STOKGIRISRAPORUIPLIK_1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    ALOT VARCHAR(30),
    ASIFIROLANLARIDOK INTEGER,
    AALIS INTEGER,
    AURETIM INTEGER,
    AURETIMIADE INTEGER,
    ATELEF INTEGER,
    ASATISIADE INTEGER,
    ADEPOTRANSFER INTEGER,
    ABUKUMURETIMI INTEGER,
    ASARIMURETIMI INTEGER)
RETURNS (
    IPLIKIRSALIYEID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ARACID VARCHAR(11),
    IPLIKIRSALIYEDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(30),
    TAMIR INTEGER,
    TAMIRSTR VARCHAR(5),
    IPLIKISLETMEID INTEGER,
    IPLIKISLETMEADI VARCHAR(30),
    KAPADEDI DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    NETMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    KALANNETMIKTAR DOUBLE PRECISION,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKRENGIID INTEGER,
    KONIKRENGIADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    HARMANKAYITID INTEGER,
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    HARMANADI VARCHAR(50),
    HARMANCINSIID INTEGER,
    HARMANCINSIADI VARCHAR(50),
    IPLIKTALEPYIL INTEGER,
    IPLIKTALEPID INTEGER,
    IPLIKTALEPIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    IPLIKFATURAID INTEGER,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    ARTIKELNO VARCHAR(20),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    IPLIKDEPOID VARCHAR(10),
    KAYNAKIPLIKDEPOID VARCHAR(10),
    BUKUMISEMRIKAYITID INTEGER,
    SARIMISEMRIKAYITID INTEGER,
    ELYAFURETIMISEMRIKAYITID INTEGER,
    ELYAFURETIMISEMRIYIL INTEGER,
    ELYAFURETIMISEMRIID INTEGER,
    ELYAFURETIMISEMRIIDSTR VARCHAR(30),
    ELYAFURETIMIEIPLIKDEPOID DEPOKODUSIZE,
    ELYAFURETIMIEIPLIKDEPOADI ADI30,
    ELYAFURETIMIEHAMID INTEGER,
    ELYAFURETIMIEHAMADI ADI255,
    ELYAFURETIMIELOT VARCHAR(30),
    KIRPINTITASNIFYIL INTEGER,
    KIRPINTITASNIFID INTEGER,
    KIRPINTITASNIFSTR VARCHAR(30),
    GELISSIRASI INTEGER,
    KALITENO INTEGER,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    VERGIHESAPNO VARCHAR(20))
AS
declare variable KullanilanMiktar Double Precision;
declare variable KullanilanNetMiktar Double Precision;
declare variable shtAlis integer;
declare variable shtSatisIade integer;
declare variable shtUretim integer;
declare variable shtUretimIade integer;
declare variable shtTelefGirisi integer;
declare variable shtUretimBoya  integer;
declare variable ShtDepoTransfer  integer;
declare variable shtUretimKirpinti integer;
declare variable TamirGirisKayitID integer;
begin
  Select shtAlis, shtSatisIade, ShtUretim, shtUretimIade, shtTelefGirisi, shtUretimBoya, ShtDepoTransfer, shtUretimKirpinti from IplikStokHareketTipiGetir
         into :shtAlis, :shtSatisIade, :ShtUretim, :shtUretimIade, :shtTelefGirisi, :shtUretimBoya, :ShtDepoTransfer, :shtUretimKirpinti;

  if (AHamTipiId = 0) then Select hstIplik from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 1) then Select hstElyaf from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 2) then Select hstTelef from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 3) then Select hstHamPamuk from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 4) then Select hstCigit from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 5) then Select hstKirpinti from HamTipiGetir into :AHamTipiId;
  else AHamTipiId = null;

  for select IplikIrsaliye.IplikIrsaliyeId, IplikIrsaliye.HareketTipi, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.FirmaId,
             IFirma.Adi, IplikIrsaliye.AracId, IplikIrsaliyeDetay.IplikIrsaliyeDetayId, IplikIrsaliyeDetay.HamId, Ham.Adi,
             Ham.HamTipiId, IplikIrsaliyeDetay.RenkId, Renk.Adi, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.Lot,
             IplikIrsaliyeDetay.IplikIsletmeId, IplikIsletme.Adi,
             Coalesce(iif(IplikUretim.Miktar is not Null, 1, Null),IplikIrsaliyeDetay.KapAdedi), Coalesce(IplikUretim.Miktar,IplikIrsaliyeDetay.Miktar), Coalesce(IplikUretim.NetMiktar, IplikIrsaliyeDetay.NetMiktar),
             IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             BoyaSiparis.FirmaId, SiparisFirma.Adi,
             IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId, IplikTalepDetay.BoyaSiparisDetayId,
             BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, IplikIrsaliyeDetay.AmbalajId, Ambalaj.Adi,
             IplikIrsaliyeDetay.KonikRengiId, KonikRengi.Adi, IplikIrsaliyeDetay.KonikCinsiId, KonikCinsi.Adi,
             IplikIrsaliyeDetay.IplikFaturaId,
             coalesce(OrmeSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.MusteriSiparisNo),
             coalesce(OrmeSiparisDetay.ReferansSiparisNo, BoyaSiparisDetay.ReferansSiparisNo),
             BoyaSiparisDetay.MusteriKaliteNo, OrmeSiparisDetay.ArtikelNo, IplikIrsaliyeDetay.Notu,
             IplikIrsaliyeDetay.HarmanKayitId, harman.HarmanYil, Harman.HarmanId, Harman.Adi, Harman.HarmanCinsiId, HarmanCinsi.Adi,
             IplikIrsaliyeDetay.TamirGirisKayitID, IplikIrsaliyeDetay.IplikDepoId, IplikIrsaliye.IplikDepoID, IplikLot.BukumIsEmriKayitID, IplikUretim.SarimIsEmriKayitID, IplikIrsaliyeDetay.BirimFiyat, IplikIrsaliye.DovizID,
             IFirma.VergiHesapNo,
             IplikIrsaliyeDetay.ElyafUretimIsEmriKayitId, ElyafUretimIsEmri.IsEmriYil, ElyafUretimIsEmri.IsEmriId, ElyafUretimIsEmri.IplikDepoId, ElyafUretimIEIplikDepo.Adi,
             ElyafUretimIsEmri.HamId, ElyafUretimIEHam.Adi, ElyafUretimIsEmri.Lot, IplikIrsaliyeDetay.GelisSirasi, IplikIrsaliyeDetay.KaliteNo, KirpintiTasnif.KirpintiTasnifYil, KirpintiTasnif.KirpintiTasnifId from IplikIrsaliye
  left join Firma IFirma on IFirma.FirmaId = IplikIrsaliye.FirmaId
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeId = IplikIrsaliye.IplikIrsaliyeId
  left join IplikIsletme on IplikIsletme.IplikIsletmeId = IplikIrsaliyeDetay.IplikIsletmeId
  left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
  left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikIrsaliyeDetay.IplikTalepDetayId
  left join Ambalaj on Ambalaj.AmbalajId = IplikIrsaliyeDetay.AmbalajId
  left join KonikRengi on KonikRengi.KonikRengiId = IplikIrsaliyeDetay.KonikRengiId
  left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikIrsaliyeDetay.KonikCinsiId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
  left join IplikFatura on IplikFatura.IplikFaturaId = IplikIrsaliyeDetay.IplikFaturaId
  Left join Renk on Renk.RenkID = IplikIrsaliyeDetay.RenkId
  left join Harman on Harman.HarmanKayitId = IplikIrsaliyeDetay.HarmanKayitId
  left join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Harman.HarmanCinsiId
  left join IplikLot on IplikLot.IplikLotId = IplikIrsaliyeDetay.IplikLotId
  left join IplikUretim on IplikUretim.IplikIrsaliyeDetayId = IplikIrsaliyeDetay.IplikIrsaliyeDetayId
  left join ElyafUretimIsEmri ON ElyafUretimIsEmri.ElyafUretimIsEmriKayitId = IplikIrsaliyeDetay.ElyafUretimIsEmriKayitId
  left join KirpintiTasnifDetay ON KirpintiTasnifDetay.KirpintiTasnifDetayId = IplikIrsaliyeDetay.KirpintiTasnifDetayId
  left join KirpintiTasnif ON KirpintiTasnif.KirpintiTasnifKayitId = KirpintiTasnifDetay.KirpintiTasnifKayitId
  left join Ham ElyafUretimIEHam on ElyafUretimIEHam.HamId = ElyafUretimIsEmri.HamId
  left join IplikDepo ElyafUretimIEIplikDepo on ElyafUretimIEIplikDepo.IplikDepoId = ElyafUretimIsEmri.IplikDepoId
  where IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and IplikIrsaliyeDetay.IplikIrsaliyeId is not null and
        IplikIrsaliyeDetay.GirisCikis = 1 and
        (IplikIrsaliye.FirmaID = :AFirmaID or :AFirmaID is null) and
        (IplikIrsaliyeDetay.HamID = :AHamID or :AHamID is null) and
        (Ham.HamTipiID = :AHamTipiID or :AHamTipiID is null) and
        (IplikIrsaliyeDetay.Lot = :ALot or :ALot is null) and
        (IplikIrsaliye.HareketTipi <> :shtAlis or :AAlis = 1) and
        (
         ((IplikIrsaliye.HareketTipi <> :shtUretim and IplikIrsaliye.HareketTipi <> :shtUretimBoya) or IplikLot.BukumIsEmriKayitID is not Null or IplikUretim.SarimIsEmriKayitID is Not Null)
         or
         (:AUretim = 1 and IplikLot.BukumIsEmriKayitID is Null and IplikUretim.SarimIsEmriKayitID is Null)
         ) and
        (IplikIrsaliye.HareketTipi <> :shtUretimIade or :AUretimIade = 1) and
        (IplikIrsaliye.HareketTipi <> :shtTelefGirisi or :ATelef = 1) and
        (IplikIrsaliye.HareketTipi <> :shtSatisIade or :ASatisIade = 1) and
        (IplikIrsaliye.HareketTipi <> :shtDepoTransfer or :ADepoTransfer = 1) and
        (IplikLot.BukumIsEmriKayitID is Null or :ABukumUretimi = 1) and
        (IplikUretim.SarimIsEmriKayitID is Null or :ASarimUretimi = 1)
  into :IplikIrsaliyeId, :HareketTipi, :IrsaliyeNo, :IrsaliyeTarihi, :FirmaId, :FirmaAdi, :AracId, :IplikIrsaliyeDetayId, :HamId, :HamAdi, :HamTipiId, :RenkId, :RenkAdi, :MelanjKodu, :Lot,
       :IplikIsletmeId, :IplikIsletmeAdi, :KapAdedi, :Miktar, :NetMiktar, :IplikTalepYil, :IplikTalepId, :BoyaSiparisYil, :BoyaSiparisId,  :SiparisFirmaId, :SiparisFirmaAdi,
       :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :AmbalajId, :AmbalajAdi,
       :KonikRengiId, :KonikRengiAdi, :KonikCinsiId, :KonikCinsiAdi,
       :IplikFaturaId, :MusteriSiparisNo, :ReferansSiparisNo, :MusteriKaliteNo, :ArtikelNo, :Notu,
       :HarmanKayitId, :HarmanYil, :HarmanId, :HarmanAdi, :HarmanCinsiId, :HarmanCinsiAdi, :TamirGirisKayitID, :IplikDepoID, :KaynakIplikDepoID, :BukumIsEmriKayitID, :SarimIsEmriKayitID, :BirimFiyat, :DovizID,
       :VergiHesapNo,
       :ElyafUretimIsEmriKayitId, :ElyafUretimIsEmriYil, :ElyafUretimIsEmriId, :ElyafUretimIEIplikDepoId, :ElyafUretimIEIplikDepoAdi,
       :ElyafUretimIEHamId, :ElyafUretimIEHamAdi, :ElyafUretimIELot, :GelisSirasi, :KaliteNo, :KirpintiTasnifYil, :KirpintiTasnifId do
  begin
    KullanilanMiktar = Null;
    KullanilanNetMiktar = Null;

    if (HareketTipi = shtAlis) then HareketTipiStr = 'Alış İrsaliyesi';
    else if (HareketTipi = shtSatisIade) then HareketTipiStr = 'Satış İade';
    else if (HareketTipi = ShtUretim and ElyafUretimIsEmriKayitId is not null) then HareketTipiStr = 'Elyaf Üretimi';
    else if (HareketTipi = ShtUretim and BukumIsEmriKayitID is not null) then HareketTipiStr = 'Büküm Üretimi';
    else if (HareketTipi = ShtUretim and SarimIsEmriKayitID is not null) then HareketTipiStr = 'Sarım Üretimi';
    else if (HareketTipi = ShtUretim) then HareketTipiStr = 'Iplik Üretim';
    else if (HareketTipi = shtUretimIade) then HareketTipiStr = 'Üretim İade';
    else if (HareketTipi = shtTelefGirisi) then HareketTipiStr = 'Telef';
    else if (HareketTipi = ShtUretimBoya) then HareketTipiStr = 'Elyaf Boya';
    else if (HareketTipi = ShtDepoTransfer) then HareketTipiStr = 'Depo Transfer';
    else if (HareketTipi = shtUretimKirpinti and KirpintiTasnifYil is not Null) then HareketTipiStr = 'Kırpıntı Üretim (Tasnif)';
    else if (HareketTipi = shtUretimKirpinti) then HareketTipiStr = 'Kırpıntı Üretim';
    else HareketTipiStr = 'Hatalı';

    Select Sum(IplikStok.Miktar) from IplikStok
      where IplikStok.IplikIrsaliyeDetayId = :IplikIrsaliyeDetayId and IplikStok.CikisIplikIrsaliyeDetayId is not null
      into :KullanilanMiktar;

    Select Sum(IplikStok.NetMiktar) from IplikStok
      where IplikStok.IplikIrsaliyeDetayId = :IplikIrsaliyeDetayId and IplikStok.CikisIplikIrsaliyeDetayId is not null
      into :KullanilanNetMiktar;

    if (KullanilanNetMiktar is null) then KullanilanNetMiktar = 0;
    KalanNetMiktar = NetMiktar - KullanilanNetMiktar;

    if (KullanilanMiktar is null) then KullanilanMiktar = 0;
    KalanMiktar = Miktar - KullanilanMiktar;

    KirpintiTasnifStr = '';
    if (KirpintiTasnifYil is not null) then
      KirpintiTasnifStr = '(' || EgemenIntegerToStr(KirpintiTasnifYil) || ') '  || EgemenIntegerToStr(KirpintiTasnifId)  || ' ';

    if (ASifirOlanlariDok = 1 or KalanMiktar > 0) then
    begin
      if (BoyaSiparisYil is not null) then
        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';
      else
        BoyaSiparisIdStr = '';

      if (ElyafUretimIsEmriYil is not null) then
        ElyafUretimIsEmriIdStr = '(' || EgemenIntegerToStr(ElyafUretimIsEmriYil) || ') '  || EgemenIntegerToStr(ElyafUretimIsEmriId)  || ' ';
      else
        ElyafUretimIsEmriIdStr = '';

      if (IplikTalepYil is not null) then
        IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';
