CREATE PROCEDURE ONAYLANMAMISHAMSTOKIRSALIYE(
    ARAPORTIPI INTEGER,
    ADEPARTMANID INTEGER,
    AIRSALIYENO IRSALIYENOSIZE,
    AORMEGIRISIRSALIYENO IRSALIYENOSIZE,
    AHAMSTOKMASTERID INTEGER,
    AFIRMAID VARCHAR(20),
    AREFERANSSIPARISNO VARCHAR(30),
    AREFERANSFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AHAMID INTEGER,
    AFASON INTEGER)
RETURNS (
    HAMSTOKIRSALIYEID INTEGER,
    HAMSTOKHAREKETTIPIID INTEGER,
    HAMSTOKHAREKETTIPIIDSTR VARCHAR(10),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SINIF VARCHAR(1),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    EVRAKNO VARCHAR(10),
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30))
AS
begin
  For Select HamStokIrsaliyeId, HamStokHareketTipiId, Fason, FasonStr, FirmaId, FirmaAdi, Sinif, IrsaliyeSeriNo, IrsaliyeNo, EvrakNo, IrsaliyeTarihi, OrmeciAdi, EvrakTipi, EvrakTipiStr from OnaylanmamisHamStokListesi(:ARaporTipi, :ADepartmanId, :AIrsaliyeNo, :AOrmeGirisIrsaliyeNo, :AHamStokMasterId,  :AFirmaId, :AReferansSiparisNo, :AReferansFirmaId, :ATemsilciId, :AHamId, :AFason)
   Group By HamStokIrsaliyeId, HamStokHareketTipiId, Fason, FasonStr, FirmaId, FirmaAdi, Sinif, IrsaliyeSeriNo, IrsaliyeNo, EvrakNo, IrsaliyeTarihi, OrmeciAdi, EvrakTipi, EvrakTipiStr
   into :HamStokIrsaliyeId, :HamStokHareketTipiId, :Fason, :FasonStr, :FirmaId, :FirmaAdi, :Sinif, :IrsaliyeSeriNo, :IrsaliyeNo, :EvrakNo, :IrsaliyeTarihi, :OrmeciAdi, :EvrakTipi, :EvrakTipiStr do
  begin
    if (HamStokHareketTipiId = 0) then HamStokHareketTipiIdStr = 'Giriş';
    else if (HamStokHareketTipiId = 1) then HamStokHareketTipiIdStr = 'Satış İade';
    else HamStokHareketTipiIdStr = '';
    suspend;
  end
end
