CREATE PROCEDURE PARTIRECETESISTOKKONTROL(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    MIKTAR DOUBLE PRECISION,
    BANYO DOUBLE PRECISION,
    BOYAKIMYASALMIKTARI DOUBLE PRECISION,
    STOKMIKTARI DOUBLE PRECISION,
    SONUC INTEGER)
AS
  declare variable ReceteYazimdaMiktarKontrol        integer;
  declare variable ReceteYazimdaMiktarZorunlu        integer;
begin
  select ReceteYazimdaMiktarKontrol, ReceteYazimdaMiktarZorunlu from Sistem into :ReceteYazimdaMiktarKontrol, :ReceteYazimdaMiktarZorunlu;

  if (ReceteYazimdaMiktarKontrol > 0) then
  begin
    Sonuc = 0;
    for select UrunId, UrunAdi, Miktar, Banyo from PartiRecetesiDokumu_2(:APartiKayitId, :AReceteId, :APartiAsamalariId, 1)
      into :UrunId, :UrunAdi, :Miktar, :Banyo do
    begin
      if ((Miktar is not Null) and (Miktar > 0)) then BoyaKimyasalMiktari = Miktar;
      else if ((Banyo is not Null) and (Banyo > 0)) then BoyaKimyasalMiktari = Banyo;

      if ((BoyaKimyasalMiktari is not Null) and (BoyaKimyasalMiktari > 0)) then
      begin
        Select ToplamStokMevcudu * ReceteBirim.Carpan from UrunStokBilgisiDetayliGetir(:UrunId, Null)
          left join UrunBirim ReceteBirim on ReceteBirim.UrunId = :UrunId and ReceteBirim.ReceteDefaultBirim = 1
        into :StokMiktari;

        Sonuc = 0;
        if (StokMiktari is not Null) then
        begin
          if (StokMiktari < BoyaKimyasalMiktari) then
          begin
            if (ReceteYazimdaMiktarZorunlu = 1) then Sonuc = -2;
            else Sonuc = -1;
          end
          Suspend;
        end
      end
    end
  end
  else
  begin
    Sonuc = 0;
    Suspend;
  end
end
