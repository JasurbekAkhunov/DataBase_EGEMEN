CREATE PROCEDURE MODELLISTESIDETAYI(
    AMODELID MODELKODUSIZE,
    ASEZONID INTEGER,
    AMODELGRUPID MODELGRUPKODUSIZE,
    AMODELGRUPID2 MODELGRUPKODUSIZE,
    AMUSTERIMODELNO VARCHAR(30),
    AONAYDURUMU INTEGER)
RETURNS (
    MODELKUMASID INTEGER,
    MODELKAYITID INTEGER,
    MODELID MODELKODUSIZE,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MIKTAR BOYAMIKTARSIZE,
    FIREORANI DOUBLE PRECISION,
    RENKID INTEGER,
    RENKADI ADI50,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATDOVIZID DOVIZSIZE,
    LABRECETEID INTEGER,
    LABRECETEADI VARCHAR(40),
    LABRECETEKODU RECETEKODUSIZE,
    LABRECETEATISID INTEGER,
    LABRECETEATISNO VARCHAR(20),
    PANTONEKODU VARCHAR(30),
    ANAKUMAS EGEBOOLEAN,
    ANAKUMASMODELKUMASID INTEGER,
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(20),
    BOYALIGRAMAJ INTEGER,
    BOYALIEN INTEGER,
    NOTU MEMO,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40),
    TEDARIKCIRENKKODU VARCHAR(30))
AS
begin
  For Select ModelKumas.ModelKumasID, ModelKumas.ModelKayitId, Model.ModelId, ModelKumas.HamId, Ham.Adi, ModelKumas.FireOrani, ModelKumas.RenkId, ModelKumas.BirimFiyat,
             ModelKumas.BirimFiyatDovizId, ModelKumas.InsertKullaniciId, ModelKumas.InsertTarihi, ModelKumas.KullaniciId, ModelKumas.DegisimTarihi,
             IK.Adi, SK.Adi, ModelKumas.Notu, ModelKumas.Miktar, ModelKumas.LabReceteId, LabRecete.Adi, ModelKumas.LabReceteKodu, ModelKumas.LabReceteAtisID,
             LabReceteAtis.AtisNoStr, ModelKumas.RenkAdi, ModelKumas.PantoneKodu, ModelKumas.AnaKumas, ModelKumas.AnaKumasModelKumasID, ModelKumas.TupAcikEnMayli,
             ModelKumas.BoyaliGramaj, ModelKumas.BoyaliEn, ModelKumas.TedarikciRenkKodu From Model
    left join ModelKumas on ModelKumas.ModelKayitId = Model.ModelKayitID
    left join Ham on Ham.HamId = ModelKumas.HamId
    left join LabRecete on LabRecete.LabReceteId = ModelKumas.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = ModelKumas.LabReceteAtisId
    left join Kullanici IK on IK.KullaniciId  = ModelKumas.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = ModelKumas.KullaniciId
    where (Model.ModelId like :AModelId || '%' or :AModelId is Null) and
          (Model.SezonId = :ASezonId or :ASezonId is null) and
          (Model.MusteriModelNo = :AMusteriModelNo or :AMusteriModelNo is null) and
          (Model.ModelGrupId = :AModelGrupID or :AModelGrupID is null) and
          (Model.ModelGrup2Id = :AModelGrupId2 or :AModelGrupId2 is null) and
          (Model.OnayDurumu = :AOnayDurumu or :AOnayDurumu is null) and
          ModelKumas.ModelKayitId is not Null and ModelKumas.AnaKumas = 1
    into :ModelKumasID, :ModelKayitId, :ModelId, :HamId, :HamAdi, :FireOrani, :RenkId, :BirimFiyat, :BirimFiyatDovizId,
         :InsertKullaniciId, :InsertTarihi, :KullaniciId, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi, :Notu, :Miktar,
         :LabReceteId, :LabReceteAdi, :LabReceteKodu, :LabReceteAtisID, :LabReceteAtisNo, :RenkAdi, :PantoneKodu, :AnaKumas,
         :AnaKumasModelKumasID, :TupAcikEnMayli, :BoyaliGramaj, :BoyaliEn, :TedarikciRenkKodu do
  begin
    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikEnMayli) into :TupAcikEnMayliStr;
    Suspend;
  end
end
