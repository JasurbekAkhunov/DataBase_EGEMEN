CREATE PROCEDURE TEKLIFTALEPLISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ATEKLIFTALEPDETAYID INTEGER)
RETURNS (
    TEKLIFTALEPKAYITID INTEGER,
    TEKLIFTALEPYIL INTEGER,
    TEKLIFTALEPID INTEGER,
    TEKLIFTALEPNOSTR VARCHAR(30),
    TALEPTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable KS integer;
begin
  For Select TeklifTalep.TeklifTalepKayitId, TeklifTalep.TeklifTalepYil, TeklifTalep.TeklifTalepId,
             TeklifTalep.TalepTarihi, TeklifTalep.Notu,
             IK.Adi, TeklifTalep.InsertTarihi, Sk.Adi, TeklifTalep.DegisimTarihi from TeklifTalep
    Left Join Kullanici IK on IK.KullaniciId = TeklifTalep.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = TeklifTalep.KullaniciId
    where TeklifTalep.TalepTarihi >= :ABaslangicTarihi and TeklifTalep.TalepTarihi <= :ABitisTarihi
    into :TeklifTalepKayitId, :TeklifTalepYil, :TeklifTalepId, :TalepTarihi, :Notu,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
   begin
     TeklifTalepNoStr = '(' || EgemenIntegerToStr(TeklifTalepYil) || ') ' || EgemenIntegerToStr(TeklifTalepId) || ' ';

     KS = 1;

     if (AFirmaID is not null) then
       Select Count(*) from TeklifTalepDetay where TeklifTalepDetay.TeklifTalepKayitID = :TeklifTalepKayitID and TeklifTalepDetay.FirmaID = :AFirmaID into :KS;

     if (AFirmaId is null or KS > 0) then
     begin
       KS = 1;

       if (ATeklifTalepDetayId is not null) then
          Select Count(*) from TeklifTalepDetay where TeklifTalepDetay.TeklifTalepKayitID = :TeklifTalepKayitID and TeklifTalepDetay.TeklifTalepDetayId = :ATeklifTalepDetayId into :KS;

       if (ATeklifTalepDetayId is null or KS > 0) then suspend;
     end
   end
end
