CREATE PROCEDURE KAPANACAKORMESIPARISLISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AORMESIPARISSEKLIID INTEGER,
    AORAN DOUBLE PRECISION,
    ASIPARISIALANKULLANICIID INTEGER,
    ADOKUMASIPARISI INTEGER)
RETURNS (
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    HAMTIPIID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    HAMMIKTAR DOUBLE PRECISION,
    TOPLAMBOYALISIPARISMIKTARI DOUBLE PRECISION,
    SATINALINACAKMIKTAR DOUBLE PRECISION,
    STOKTANKULLANILACAKMIKTAR DOUBLE PRECISION,
    TESLIMTARIHI TIMESTAMP,
    REVIZETESLIMTARIHI TIMESTAMP,
    ILKSEVKTARIHI TIMESTAMP,
    TOPLAMORULEN NUMERIC(12,3),
    TOPLAMSATIS NUMERIC(12,3),
    TOPLAMBOYALISATIS NUMERIC(12,3),
    TOPLAMSATISIADE NUMERIC(12,3),
    TOPLAMFATURAMIKTAR NUMERIC(12,3),
    TOPLAMBOYALIFATURAMIKTAR NUMERIC(12,3),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    MELANJKODU VARCHAR(20),
    PUS INTEGER,
    FEINE INTEGER,
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    TAHMINIFIREORANI DOUBLE PRECISION,
    ORMEFIREORANI DOUBLE PRECISION,
    ORMEFIREMIKTARI DOUBLE PRECISION,
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    SIPARISSIRANO INTEGER,
    ORMESIPARISSEKLIID INTEGER,
    ORMESIPARISSEKLIADI VARCHAR(15),
    ORMESIPARISTIPIID INTEGER,
    ORMESIPARISTIPIADI VARCHAR(30),
    SIPARISIVEREN VARCHAR(50),
    KALITEID VARCHAR(20),
    KALITEADI ADI255,
    SEVKEDILMEMISPARTIVAR INTEGER,
    SEVKEDILMEMISSTOKVAR INTEGER,
    ACIKTALIMATVAR INTEGER,
    KAPATILABILIR INTEGER,
    TOPLAMPARTILENENKILO NUMERIC(12,3),
    TOPLAMPARTILENENKILOBASKI NUMERIC(12,3))
AS
declare variable FireHesapTipi          integer;
declare variable FHamTipiId             integer;
declare variable FMiktar                Double Precision;
declare variable OrmeyeCikanNetIplik    Double Precision;
declare variable OrmeyeCikanNetLycra    Double Precision;
declare variable hstElyaf               integer;
declare variable hstIplik               integer;
declare variable hstOrmeKumas           integer;
declare variable hstLycra               integer;
declare variable hstTelef        integer;
declare variable KS   integer;
declare variable FToplamBoyaliFaturaMiktar Numeric(12,3);
declare variable FToplamBoyaliSatis        Numeric(12,3);
declare variable OrmeFirmaKodumuz  varchar(20);
declare variable shtSatis integer;
declare variable FarkMiktar  Double Precision;
declare variable FarkOran    Double Precision;
declare variable AOrmeSiparisi integer;
declare variable OrmeKapSipRaporTipi integer;

begin

  AOran = 100 - AOran;
  AOrmeSiparisi = 1;
  if (ADokumaSiparisi = 1) then  AOrmeSiparisi = 0;

  Select FireHesapTipi, OrmeFirmaKodumuz, OrmeKapSipRaporTipi from Sistem where SistemId = 1 into :FireHesapTipi, :OrmeFirmaKodumuz, :OrmeKapSipRaporTipi;
  Select hstIplik, hstOrmeKumas, hstLycra, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas,:hstLycra, :hstElyaf, :hstTelef;
  Select shtSatis from OrmeStokHareketTipiGetir into :shtSatis;

  for Select BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, BoyaSiparis.SiparisTarihi, FirmaYetkili.Adi, OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayId,
     OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.OrmeSiparisSekliId, OrmeSiparisSekli.Adi, BoyaSiparis.OrmeSiparisTipiId, OrmeSiparisTipi.Adi,
     OrmeSiparisDetay.HamId, Ham.Adi, Ham.HamtipiId, OrmeSiparisDetay.RenkId, Renk.Adi, OrmeSiparisDetay.hamMiktar, OrmeSiparisDetay.SatinAlinacakMiktar, OrmeSiparisDetay.StoktanKullanilacakMiktar,
     OrmeSiparisDetay.Teslimtarihi, OrmeSiparisDetay.RevizeTeslimtarihi,
     OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MelanjKodu,
     OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmeSiparisDetay.TupAcikEnMayli,
     case when :ADokumaSiparisi = 1 then OrmeSiparisDetay.ToplamDokunan else OrmeSiparisDetay.ToplamOrulen end, OrmeSiparisDetay.ToplamSatis, OrmeSiparisDetay.ToplamBoyaliSatis, OrmeSiparisDetay.ToplamSatisIade, OrmeSiparisDetay.TahminiFireOrani,
     BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi, OrmeSiparisDetay.KaliteId, Kalite.KaliteAdi,
     BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi from BoyaSiparis
  left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
  Left Join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
  left join OrmeSiparisSekli On OrmeSiparisSekli.OrmeSiparisSekliId = BoyaSiparis.OrmeSiparisSekliId
  left join OrmeSiparisTipi On OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
  Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  left join OrmeSiparisDetay on OrmeSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
  Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
  Left Join Kalite ON Kalite.KaliteId = OrmeSiparisDetay.KaliteId
  Left Join Renk On Renk.RenkId = OrmeSiparisDetay.RenkId
  where (BoyaSiparis.SiparisTarihi >= :ABaslangicTarihi and BoyaSiparis.SiparisTarihi <= :ABitisTarihi) and
        (BoyaSiparis.FirmaId = :AFirmaId or :AFirmaId is null) and
         BoyaSiparis.KumasTemin = 0 and
         BoyaSiparis.OrmeSiparisi = :AOrmeSiparisi  and
         BoyaSiparis.DokumaSiparisi = :ADokumaSiparisi and
         OrmeSiparisDetay.Kapali = 0 and
        (BoyaSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
        (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
        (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
        (BoyaSiparis.OrmeSiparisSekliId = :AOrmeSiparisSekliId or :AOrmeSiparisSekliId is null)
  into :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :SiparisTarihi, :SiparisiVeren, :BoyaSiparisKayitId, :OrmeSiparisDetayId,
       :SiparisSiraNo, :OrmeSiparisSekliId, :OrmeSiparisSekliAdi, :OrmeSiparisTipiId, :OrmeSiparisTipiAdi,
       :HamId, :HamAdi, :HamtipiId, :RenkId, :RenkAdi, :HamMiktar, :SatinAlinacakMiktar, :StoktanKullanilacakMiktar,
       :Teslimtarihi, :RevizeTeslimtarihi,
       :MusteriSiparisNo, :ReferansSiparisNo, :ArtikelNo, :MelanjKodu, :Pus, :Feine, :TupAcikEnMayli,
       :ToplamOrulen, :ToplamSatis, :ToplamBoyaliSatis, :ToplamSatisIade, :TahminiFireOrani,
       :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi, :KaliteId, :KaliteAdi,
       :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    KapatilaBilir = 0;
    OrmeyeCikanNetIplik = 0;
    OrmeyeCikanNetLycra = 0;
    OrmeFireOrani = Null;
    ToplamBoyaliSiparisMiktari = Null;

    Select Sum(BoyaSiparisDetay.SiparisKilo), Sum(BoyaSiparisDetay.PartilenenKilo), Sum(BoyaSiparisDetay.PartilenenKiloBaski) from OrmeSiparisDetay
        left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
        where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and BoyaSiparisDetay.HamID = OrmeSiparisDetay.HamId
        into :ToplamBoyaliSiparisMiktari, :ToplamPartilenenKilo, :ToplamPartilenenKiloBaski;

    if (ToplamBoyaliSiparisMiktari is null) then ToplamBoyaliSiparisMiktari = 0;
  if (ToplamPartilenenKilo is null) then ToplamPartilenenKilo = 0;
    if (ToplamPartilenenKiloBaski is null) then ToplamPartilenenKiloBaski = 0;

    if (ToplamOrulen > 0) then
    begin
    if (AOrmeSiparisi = 1) then
      begin
        for Select Ham.HamTipiId, sum(CikisMiktari - IadeMiktari) from OrmePlanlama
          left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaID = OrmePlanlama.OrmePlanlamaID
          left join Ham on Ham.HamID = OrmePlanlamaRecete.HamID
          where OrmePlanlama.OrmeSiparisDetayId = :OrmeSiparisDetayId and OrmePlanlamaRecete.OrmePlanlamaID is not null
          group by Ham.HamTipiID
          into :FHamTipiId, :FMiktar do
        begin
          if (FHamTipiId = hstLycra) then OrmeyeCikanNetLycra = OrmeyeCikanNetLycra + FMiktar;
          else OrmeyeCikanNetIplik = OrmeyeCikanNetIplik + FMiktar;
        end
      end

      OrmeFireMiktari = OrmeyeCikanNetLycra + OrmeyeCikanNetIplik - ToplamOrulen;

      if (FireHesapTipi = 0) then
        if ((ToplamOrulen - OrmeyeCikanNetLycra)> 0) then OrmeFireOrani = (OrmeFireMiktari / (ToplamOrulen - OrmeyeCikanNetLycra)) * 100;
      else
        if (OrmeyeCikanNetIplik  > 0) then OrmeFireOrani = (OrmeFireMiktari / OrmeyeCikanNetIplik ) * 100;

      OrmeFireOrani = Yuvarla(OrmeFireOrani, 100);
    end

    IlkSevkTarihi = Null;
    if (ToplamSatis > 0) then
    begin
      Select Min(StokHareket.Tarih) from OrmeSiparisDetay
        left join StokHareket on StokHareket.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
        where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and StokHareket.HareketTipi = 2
      into :IlkSevkTarihi;
    end

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;


    Select Sum(BoyaIrsaliyeDetay.Kilo) from OrmeSiparisDetay
    left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayID = OrmeSiparisDetay.OrmeSiparisDetayId
    left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and BoyaSiparisDetay.BoyaSiparisDetayID is not null and
          PartiStok.PartiSinifID is not null and BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID is not null and
          BoyaIrsaliye.FirmaID <> :OrmeFirmaKodumuz
    into :FToplamBoyaliSatis;

    if (ToplamBoyaliSatis is null) then ToplamBoyaliSatis = 0;
    if (FToplamBoyaliSatis is null) then FToplamBoyaliSatis = 0;
    ToplamBoyaliSatis = ToplamBoyaliSatis + FToplamBoyaliSatis;

    Select Sum(StokHareket.Miktar) from OrmeSiparisDetay
      left join StokHareket on StokHareket.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
      where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and StokHareket.StokHareketID is not null and
            StokHareket.BoyaSiparisDetayID is null and StokHareket.HareketTipi = :shtSatis and StokHareket.HamIade = 0
      into :ToplamFaturaMiktar;


    Select Sum(StokHareket.Miktar) from OrmeSiparisDetay
      left join StokHareket on StokHareket.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
      where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and StokHareket.StokHareketID is not null and
            StokHareket.BoyaSiparisDetayID is not null and StokHareket.HareketTipi = :shtSatis and StokHareket.HamIade = 0
      into :ToplamBoyaliFaturaMiktar;

    Select Sum(BoyaFaturaDetay.Miktar) from OrmeSiparisDetay
      left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
      left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
      left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
      where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and BoyaSiparisDetay.BoyaSiparisDetayID is not null and
            PartiStok.PartiSinifID is not null and BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID is not null and BoyaFaturaDetay.BoyaFaturaId is not null
      into :FToplamBoyaliFaturaMiktar;

    if (ToplamBoyaliFaturaMiktar is null) then ToplamBoyaliFaturaMiktar = 0;
    if (FToplamBoyaliFaturaMiktar is null) then FToplamBoyaliFaturaMiktar = 0;

    ToplamBoyaliFaturaMiktar = ToplamBoyaliFaturaMiktar + FToplamBoyaliFaturaMiktar;
    if (ToplamBoyaliFaturaMiktar = 0) then ToplamBoyaliFaturaMiktar = null;


    Select count(*) from OrmeRaf
     where OrmeRaf.BoyaSiparisKayitID = :BoyaSiparisKayitID and OrmeRaf.OrmeSiparisDetayID = :OrmeSiparisDetayID and
           OrmeRaf.HamID = :HamId and OrmeRaf.StokVar = 1
     into :KS;

    if (KS > 0) then SevkEdilmemisStokVar = 1;
    else SevkEdilmemisStokVar = 0;


    Select count(*) from OrmePlanlama
      where OrmePlanlama.OrmeSiparisDetayId = :OrmeSiparisDetayId and OrmePlanlama.bitti = 0
      into :Ks;


   if (OrmeKapSipRaporTipi = 1) then
   begin
     if (AOran < 100) then
       FarkOran = AOran + 1;
     else
       FarkOran = 0;

     if (KS>0) then AcikTalimatVar = 1; else AcikTalimatVar = 0;

     if  ((ToplamOrulen  + StoktanKullanilacakMiktar) >= HamMiktar) then KapatilaBilir = 1;
     if (AcikTalimatVar = 1) then KapatilaBilir = 0;
     FarkMiktar = HamMiktar - (ToplamOrulen + StoktanKullanilacakMiktar);
     FarkOran = FarkMiktar / HamMiktar * 100;

     if (FarkOran >= AOran) then KapatilaBilir = 0;
     suspend;
   end
   else
   begin
     if (AOran < 100) then
       FarkOran = AOran + 1;
     else
       FarkOran = 0;

      if (ToplamBoyaliSiparisMiktari > 0) then
      begin
        Select Count(*) from OrmeSiparisDetay
         left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
         left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
         where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and BoyaSiparisDetay.OrmeSiparisDetayId is not null and
               PartiStok.PartiSinifId is not null and PartiStok.Durumu <> 8 and PartiStok.PartiStokIptalKayitId is null into :KS;

        if (KS > 0) then SevkEdilmemisPartiVar = 1;
        else SevkEdilmemisPartiVar = 0;

        FarkMiktar = ToplamBoyaliSiparisMiktari - ToplamBoyaliSatis;

        FarkOran = FarkMiktar / ToplamBoyaliSiparisMiktari * 100;
      end
      else if (ToplamSatis > 0) then
      begin
        SevkEdilmemisPartiVar = 0;
        FarkMiktar = HamMiktar - ToplamSatis;

        FarkOran = FarkMiktar / ToplamSatis * 100;
      end

      if (FarkOran <= AOran) then suspend;
    end
  end
end
