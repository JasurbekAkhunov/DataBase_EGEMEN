CREATE PROCEDURE DOKUMASIPARISRAPORU_5(
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    AIPLIKSIPARISI INTEGER,
    ABOYALIIPLIKSIPARISI INTEGER,
    ADOKUMASIPARISI INTEGER,
    ABOYALIDOKUMASIPARISI INTEGER,
    AIPLIKBOYALIDOKUMASIPARISI INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AKUMASCINSIID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    AYURTDISI INTEGER,
    AKUMASTEMINSIPARISI INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ASINIF VARCHAR(1),
    AKALITEID VARCHAR(20))
RETURNS (
    DOKUMAPLANLAMAID INTEGER,
    TARIH TIMESTAMP,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    SIRA INTEGER,
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    SAKATURETIMTOPLAMI DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    IPLIKFIREORANI DOUBLE PRECISION,
    LYCRAKULLANIMORANI DOUBLE PRECISION,
    MIKTARGUN DOUBLE PRECISION,
    BASLAMATARIHI TIMESTAMP,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    URETIMPARTINO VARCHAR(30),
    DOKUMAYACIKANIPLIK DOUBLE PRECISION,
    DOKUMADANIADEIPLIK DOUBLE PRECISION,
    DOKUMAYACIKANLYCRA DOUBLE PRECISION,
    DOKUMADANIADELYCRA DOUBLE PRECISION,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LOT VARCHAR(150),
    BITTI INTEGER,
    BITTISTR VARCHAR(5),
    BEKLEMEDE INTEGER,
    BEKLEMEDESTR VARCHAR(9),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    HAMID INTEGER,
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable Iplik        Double Precision;
declare variable Lycra        Double Precision;
declare variable FHamTipiId           integer;
declare variable FCikisMiktari          Double Precision;
declare variable FIadeMiktari           Double Precision;
declare variable YuvarlamaCarpani       Double Precision;
declare variable uitBoya integer;
begin
  select uitBoya from UretimIslemTipiGetir into :uitBoya;

  For Select DokumaSiparisRaporu_1.BoyaSiparisKayitId, DokumaSiparisRaporu_1.OrmeSiparisDetayId,
             DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlama.Tarih, DokumaPlanlama.FirmaId, DokumaPlanlama.DokumaMakinaId, DokumaPlanlama.Sira,
             DokumaPlanlama.PlanlananMiktar, DokumaPlanlama.GerceklesenMiktar, DokumaPlanlama.ToplamSakatDokunan,
             DokumaPlanlama.MiktarGun, DokumaPlanlama.BaslamaTarihi, DokumaPlanlama.BirimFiyat, DokumaPlanlama.DovizId, DokumaPlanlama.VadeGun,
             DokumaPlanlama.UretimPartiNo, DokumaPlanlama.Lot, DokumaPlanlama.Notu, DokumaPlanlama.Bitti, Firma.Adi, DokumaMakina.Adi, DokumaPlanlama.Beklemede,
             Ik.Adi, DokumaPlanlama.InsertTarihi, Sk.Adi, DokumaPlanlama.DegisimTarihi,
             OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.HamID
             from DokumaSiparisRaporu_1(:ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ABaslangicSiparisTarihi, :ABitisSiparisTarihi, :AFirmaId,
             :AFASONSIPARISLER, :AFASONSATISSIPARISLER, :ASATISSIPARISLER, :AIplikSiparisi, :ABoyaliIplikSiparisi, :ADokumaSiparisi, :ABoyaliDokumaSiparisi,
             :AIplikBoyaliDokumaSiparisi, :AMusteriSiparisNo, :AReferansSiparisNo, :ARaporTipi, :AKapali, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisiAlanKullaniciId,
             :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId, :AYurtDisi, :AKumasTeminSiparisi, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASinif, :AKaliteId)
    left join DokumaPlanlama on DokumaPlanlama.OrmeSiparisDetayId = DokumaSiparisRaporu_1.OrmeSiparisDetayId
    left join Firma on Firma.FirmaId = DokumaPlanlama.FirmaId
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = DokumaPlanlama.DokumaMakinaId
    left join OrmesiparisDetay on OrmesiparisDetay.OrmesiparisDetayId = DokumaPlanlama.OrmesiparisDetayId
    Left Join Kullanici IK on IK.KullaniciId = DokumaPlanlama.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = DokumaPlanlama.KullaniciId
    where DokumaPlanlama.DokumaPlanlamaID is not null
    into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :DokumaPlanlamaId, :Tarih, :FirmaId, :DokumaMakinaId, :Sira, :PlanlananMiktar, :GerceklesenMiktar,
         :SakatUretimToplami, :MiktarGun, :BaslamaTarihi, :BirimFiyat, :DovizId, :VadeGun,
         :UretimPartiNo, :Lot, :Notu, :Bitti, :FirmaAdi, :DokumaMakinaAdi, :Beklemede, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi,
         :DegisimTarihi, :SiparisSiraNo, :HamID do
  begin
    if (Bitti = 1) then BittiStr = 'Bitti';
    else BittiStr = '';

    BeklemedeStr = '';
    if (Beklemede = 1) then BeklemedeStr = 'Beklemede';

    KalanMiktar = PlanlananMiktar - GerceklesenMiktar + SakatUretimToplami;

    Iplik = 0;
    Lycra = 0;
    IplikFireOrani = 0;
    LycraKullanimOrani = 0;

    For Select Ham.HamTipiId, Sum(CikisMiktari), sum(IadeMiktari) from DokumaPlanlamaRecete
       left join Ham on Ham.HamId = DokumaPlanlamaRecete.HamId
       where DokumaPlanlamaRecete.DokumaPlanlamaId = :DokumaPlanlamaId
      group by Ham.HamTipiId
      into :FHamTipiId, :FCikisMiktari, :FIadeMiktari do
    begin
      if (FHamTipiId = 0) then
      begin
        DokumayaCikanIplik = FCikisMiktari;
        DokumadanIadeIplik = FIadeMiktari;

        Iplik = FCikisMiktari - FIadeMiktari;
      end
      else if (FHamTipiId = 6) then
      begin
        DokumayaCikanLycra = FCikisMiktari;
        DokumadanIadeLycra = FIadeMiktari;

        Lycra = FCikisMiktari - FIadeMiktari;
      end
    end

    if (GerceklesenMiktar > 0) then
    begin

      IplikFireOrani = ((Iplik + Lycra - GerceklesenMiktar) / GerceklesenMiktar) * 100;
      LycraKullanimOrani = Lycra / GerceklesenMiktar * 100;

      YuvarlamaCarpani = 100;
      IplikFireOrani = Yuvarla(IplikFireOrani, YuvarlamaCarpani);
      LycraKullanimOrani = Yuvarla(LycraKullanimOrani, YuvarlamaCarpani);

    end

    suspend;
  end
end
