CREATE PROCEDURE ASAMAGIRISINDENPLANLAMAYAP(
    APARTIASAMALARIID INTEGER,
    AMAKINAID VARCHAR(10),
    AKULLANICIID INTEGER)
AS
declare variable BolumPlanlamaId        integer;
declare variable GrupPlanlamaId         integer;
declare variable UretimPlanlamaDetayId  integer;
declare variable PlanlamaMakinaId varchar(10);
declare variable XUretimPlanlamaId integer;

declare variable FReceteIsEmriYil integer;
declare variable FReceteIsEmriId  integer;
declare variable FPartiAsamalariId integer;
declare variable YeniSepetAc integer;
begin
  XUretimPlanlamaId = null;

  Select ReceteIsEmriYil, ReceteIsEmriId from ReceteIsEmriDetay where ReceteIsEmriDetay.PartiAsamalariId = :APartiAsamalariId into :FReceteIsEmriYil, :FReceteIsEmriId;

  for Select BolumPlanlama.BolumPlanlamaId, GrupPlanlama.GrupPlanlamaId, UretimPlanlamaDetay.UretimPlanlamaDetayId, UretimPlanlama.MakinaId from BolumPlanlama
    left join GrupPlanlama on GrupPlanlama.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
    left join UretimPlanlamaDetay on UretimPlanlamaDetay.GrupPlanlamaID = GrupPlanlama.GrupPlanlamaId
    left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
    where BolumPlanlama.PartiAsamalariID = :APartiAsamalariID
    into :BolumPlanlamaId, :GrupPlanlamaId, :UretimPlanlamaDetayId, :PlanlamaMakinaId do
  begin
    if (BolumPlanlamaId is null) then exit;

    if (XUretimPlanlamaId is null) then
    begin
      Select Max(UretimPlanlamaDetay.UretimPlanlamaId) from BolumPlanlama
        left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
        left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
        where (BolumPlanlama.BolumPlanlamaID = :BolumPlanlamaID or BolumPlanlama.PartiAsamalariID = :APartiAsamalariID) and UretimPlanlama.MakinaID = :AMakinaId
        into :XUretimPlanlamaId;
    end

    if (XUretimPlanlamaId is null) then
    begin
      Select Max(UretimPlanlamaDetay.UretimPlanlamaId) from ReceteIsEmriDetay
        left join BolumPlanlama on BolumPlanlama.PartiAsamalariId = ReceteIsEmriDetay.PartiAsamalariId
        left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
        left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
        where ReceteIsEmriDetay.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :FReceteIsEmriId and
          ReceteIsEmriDetay.PartiAsamalariId <> :APartiAsamalariID and UretimPlanlama.MakinaID = :AMakinaId
        into :XUretimPlanlamaId;
    end

    if (XUretimPlanlamaID is null) then YeniSepetAc = 1;
    else YeniSepetAc = 0;

    if (UretimPlanlamaDetayId is not null) then
    begin
      if (AMakinaId <> PlanlamaMakinaId) then
      begin
        execute Procedure MakinadanMakinayaSurukle(:UretimPlanlamaDetayId, :AMakinaId, :YeniSepetAc, :XUretimPlanlamaId, :AKullaniciId);
      end
    end
    else if (GrupPlanlamaId is not null) then
      Select UretimPlanlamaId from GruptanMakinayaSurukle(:AMakinaId, :YeniSepetAc, :XUretimPlanlamaId, :GrupPlanlamaId, :AKullaniciId) into :XUretimPlanlamaId;
    else
      Select UretimPlanlamaId from BolumdenMakinayaSurukle(:AMakinaId, :YeniSepetAc, :XUretimPlanlamaId, :BolumPlanlamaId, :AKullaniciId) into :XUretimPlanlamaId;
  end

  if (XUretimPlanlamaId is null) then
  begin
    Select Max(UretimPlanlamaDetay.UretimPlanlamaId) from BolumPlanlama
      left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
      left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
      where BolumPlanlama.BolumPlanlamaID = :BolumPlanlamaID and UretimPlanlama.MakinaID = :AMakinaId
      into :XUretimPlanlamaId;
  end

  if (XUretimPlanlamaID is null) then YeniSepetAc = 1;
  else YeniSepetAc = 0;

  for select ReceteIsEmriDetay.PartiAsamalariId from ReceteIsEmriDetay
      where ReceteIsEmriDetay.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :FReceteIsEmriId and
            ReceteIsEmriDetay.PartiAsamalariId <> :APartiAsamalariId
      into :FPartiAsamalariId do
  begin
    for Select BolumPlanlama.BolumPlanlamaId, GrupPlanlama.GrupPlanlamaId, UretimPlanlamaDetay.UretimPlanlamaDetayId, UretimPlanlama.MakinaId from BolumPlanlama
      left join GrupPlanlama on GrupPlanlama.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
      left join UretimPlanlamaDetay on UretimPlanlamaDetay.GrupPlanlamaID = GrupPlanlama.GrupPlanlamaId
      left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
      where BolumPlanlama.PartiAsamalariID = :FPartiAsamalariID
      into :BolumPlanlamaId, :GrupPlanlamaId, :UretimPlanlamaDetayId, :PlanlamaMakinaId do
    begin
      if (BolumPlanlamaId is null) then exit;

      YeniSepetAc = 0;

      if (UretimPlanlamaDetayId is not null) then
      begin
        if (AMakinaId <> PlanlamaMakinaId) then
          execute Procedure MakinadanMakinayaSurukle(:UretimPlanlamaDetayId, :AMakinaId, :YeniSepetAc, :XUretimPlanlamaId, :AKullaniciId);
      end
      else if (GrupPlanlamaId is not null) then
        Select UretimPlanlamaId from GruptanMakinayaSurukle(:AMakinaId, :YeniSepetAc, :XUretimPlanlamaId, :GrupPlanlamaId, :AKullaniciId) into :XUretimPlanlamaId;
      else
        Select UretimPlanlamaId from BolumdenMakinayaSurukle(:AMakinaId, :YeniSepetAc, :XUretimPlanlamaId, :BolumPlanlamaId, :AKullaniciId) into :XUretimPlanlamaId;
    end

  end
end
