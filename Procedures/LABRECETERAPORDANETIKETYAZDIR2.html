  left join KimyasalTipi On KimyasalTipi.KimyasalTipiId = Urun.KimyasalTipiId
  where LabReceteUrun.LabReceteUrunID is not null
  into :LabReceteAtisId, :UrunId, :UrunAdi, :MiktarYuzde, :BirimId, :UrunTipiId, :BoyaTipiId, :PipetlemeTransferKodu, :KimyasalTipiId, :KimyasalTipiAdi do

  end
end
