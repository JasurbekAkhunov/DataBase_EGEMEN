CREATE PROCEDURE NUMUNEBASKIPERFORMANSRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DESENSAYISI INTEGER,
    ATISSAYISI INTEGER,
    OKEYLIATISSAYISI INTEGER)
AS
declare variable itDesenTasarim       integer;
declare variable FFirmaId              varchar(20);
declare variable FLastFirmaId          varchar(20);
declare variable FFirmaAdi             varchar(100);
declare variable FDesenId            varchar(20);
declare variable FLastDesenId        varchar(20);
declare variable FTasarimVaryantNo   integer;
begin
  Select itDesenTasarim from IslemTipiGetir into :itDesenTasarim;

  FLastFirmaId = '';
  FLastDesenId = -1;

  for select Parti.FirmaId, Firma.Adi, BoyaSiparisDetay.DesenId, Parti.DesenVaryantId from PartiAsamaStokDetay
    left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join BoyahaneBolum on BoyahaneBolum.BoyahaneBolumId = Asama.BoyahaneBolumId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Kullanici on Kullanici.KullaniciID = BoyaSiparis.SiparisiAlanKullaniciId
    where PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi <= :ABitisTarihi and
          Parti.IslemTipiID = :itDesenTasarim and Asama.BoyahaneBolumId is not null and BoyahaneBolum.NumuneBaski = 1
    order by Parti.FirmaId, Firma.Adi, BoyaSiparisDetay.DesenId
    into :FFirmaId, :FFirmaAdi, :FDesenId, :FTasarimVaryantNo do
  begin
    if (FFirmaID <> FLastFirmaID) then
    begin
      if (FLastFirmaId <> '') then suspend;

      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      FLastFirmaId = FFirmaId;
      FLastDesenId = -1;
      DesenSayisi = 0;
      AtisSayisi = 0;
      OkeyliAtisSayisi = 0;
      FLastDesenId = -1;
    end

    if (FLastDesenId <> FDesenId) then DesenSayisi = DesenSayisi + 1;
    FLastDesenId = FDesenId;

    AtisSayisi = AtisSayisi + 1;
    if (FTasarimVaryantNo is not null) then OkeyliAtisSayisi = OkeyliAtisSayisi + 1;
  end

  if (FLastFirmaId <> '') then suspend;
end
