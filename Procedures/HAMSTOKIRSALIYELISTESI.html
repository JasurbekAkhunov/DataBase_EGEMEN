CREATE PROCEDURE HAMSTOKIRSALIYELISTESI(
    AHAMSTOKHAREKETTIPIID INTEGER,
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AORMECIID INTEGER,
    AFASON INTEGER,
    ALOT VARCHAR(150),
    AHAMSTOKMASTERID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AREFERANSFIRMAID VARCHAR(20),
    AEVRAKTIPI INTEGER)
RETURNS (
    HAMSTOKIRSALIYEID INTEGER,
    KANTARKAYITID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SINIF VARCHAR(1),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYEGRUPID VARCHAR(4),
    EVRAKNO VARCHAR(10),
    ORMECIADI VARCHAR(30),
    ARACID VARCHAR(11),
    SOFORADI VARCHAR(40),
    DOVIZID VARCHAR(5),
    DOVIZKURUSTR VARCHAR(30),
    KDVDAHIL INTEGER,
    KDVDAHILSTR VARCHAR(4),
    FIILISEVKTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable TabanDovizKuru1   Double Precision;
declare variable TabanDovizKuru2   Double Precision;
declare variable FStokTabanDovizId varchar(5);
begin
  select STOKTABANDOVIZID from SISTEM where SISTEMID = 1 into :FStokTabanDovizId;

  if (AHamStokMasterId is null) then
  begin
    For Select HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.Sinif, HamStokIrsaliye.IrsaliyeSeriNo,
               HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakNo, HamStokIrsaliye.IrsaliyeGrupId, Ormeci.Adi,
               HamStokIrsaliye.AracId, Sofor.Adi, HamStokIrsaliye.DovizId, HamStokIrsaliye.TabanDovizKuru1, HamStokIrsaliye.TabanDovizKuru2, HamStokIrsaliye.KdvDahil,
               HamStokIrsaliye.FiiliSevkTarihi, HamStokIrsaliye.KantarKayitId,
               IK.Adi, HamStokIrsaliye.InsertTarihi, Sk.Adi, HamStokIrsaliye.DegisimTarihi from HamStokIrsaliye
      left join Firma on Firma.Firmaid = hamstokIrsaliye.Firmaid
      left join Ormeci on Ormeci.Ormeciid = HamstokIrsaliye.Ormeciid
      left join Sofor on Sofor.Soforid = HamstokIrsaliye.Soforid
      Left Join Kullanici IK on IK.KullaniciId = HamstokIrsaliye.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = HamstokIrsaliye.KullaniciId
      left join HamStokMaster on HamStokMaster.HamStokIrsaliyeid = HamstokIrsaliye.HamStokIrsaliyeid
      where HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
            HamStokIrsaliye.EvrakTipi = :AEvrakTipi and
            HamStokIrsaliye.HamStokHareketTipiId = :AHamStokHareketTipiId and HamStokIrsaliye.Fason = :AFason and
            (HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
            (HamStokIrsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (HamStokIrsaliye.OrmeciId = :AOrmeciId or :AOrmeciId is null) and
            (HamStokMaster.Lot = :ALot or :ALot is null) and
            (HamStokMaster.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
            (HamStokMaster.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
            (HamStokMaster.ReferansFirmaId = :AReferansFirmaId or :AReferansFirmaId is null)
      group by HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.Sinif, HamStokIrsaliye.IrsaliyeSeriNo,
               HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakNo, HamStokIrsaliye.IrsaliyeGrupId, Ormeci.Adi,
               HamStokIrsaliye.AracId, Sofor.Adi, HamStokIrsaliye.DovizId, HamStokIrsaliye.TabanDovizKuru1, HamStokIrsaliye.TabanDovizKuru2, HamStokIrsaliye.KdvDahil,
               HamStokIrsaliye.FiiliSevkTarihi, HamStokIrsaliye.KantarKayitId,
               IK.Adi, HamStokIrsaliye.InsertTarihi, Sk.Adi, HamStokIrsaliye.DegisimTarihi
      into :HamStokIrsaliyeId, :FirmaId, :FirmaAdi, :Sinif, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakNo, :IrsaliyeGrupId, :OrmeciAdi,
           :AracId, :SoforAdi, :DovizId, :TabanDovizKuru1, :TabanDovizKuru2, :KdvDahil, :FiiliSevkTarihi, :KantarKayitId,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
    begin
      if (KdvDahil = 1) then KdvDahilStr = 'Evet';
      else KdvDahilStr = '';

      if (FStokTabanDovizId <> DovizId) then DovizKuruStr = Format_Float('', TabanDovizKuru1) || ' ' || DovizId || '-' || Format_Float('', TabanDovizKuru2) || ' ' || FStokTabanDovizId;
      else DovizKuruStr = '';
      suspend;
    end
  end
  else
  begin
    For Select HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.Sinif, HamStokIrsaliye.IrsaliyeSeriNo,
               HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakNo, HamStokIrsaliye.IrsaliyeGrupId, Ormeci.Adi,
               HamStokIrsaliye.AracId, Sofor.Adi, HamStokIrsaliye.DovizId, HamStokIrsaliye.TabanDovizKuru1, HamStokIrsaliye.TabanDovizKuru2, HamStokIrsaliye.KdvDahil,
               HamStokIrsaliye.FiiliSevkTarihi, HamStokIrsaliye.KantarKayitId,
               IK.Adi, HamStokIrsaliye.InsertTarihi, Sk.Adi, HamStokIrsaliye.DegisimTarihi from HamStokMaster
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeid = HamStokMaster.HamStokIrsaliyeId
      left join Firma on Firma.Firmaid = hamstokIrsaliye.Firmaid
      left join Ormeci on Ormeci.Ormeciid = HamstokIrsaliye.Ormeciid
      left join Sofor on Sofor.Soforid = HamstokIrsaliye.Soforid
      Left Join Kullanici IK on IK.KullaniciId = HamstokIrsaliye.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = HamstokIrsaliye.KullaniciId
      where  HamStokMaster.HamStokMasterID = :AHamStokMasterID and HamStokIrsaliye.HamStokHareketTipiId = :AHamStokHareketTipiId and HamStokIrsaliye.Fason = :AFason and
            HamStokIrsaliye.EvrakTipi = :AEvrakTipi
      group by HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.Sinif, HamStokIrsaliye.IrsaliyeSeriNo,
               HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakNo, HamStokIrsaliye.IrsaliyeGrupId, Ormeci.Adi,
               HamStokIrsaliye.AracId, Sofor.Adi, HamStokIrsaliye.DovizId, HamStokIrsaliye.TabanDovizKuru1, HamStokIrsaliye.TabanDovizKuru2, HamStokIrsaliye.KdvDahil,
               HamStokIrsaliye.FiiliSevkTarihi, HamStokIrsaliye.KantarKayitId,
               IK.Adi, HamStokIrsaliye.InsertTarihi, Sk.Adi, HamStokIrsaliye.DegisimTarihi
      into :HamStokIrsaliyeId, :FirmaId, :FirmaAdi, :Sinif, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakNo, :IrsaliyeGrupId, :OrmeciAdi,
           :AracId, :SoforAdi, :DovizId, :TabanDovizKuru1, :TabanDovizKuru2, :KdvDahil, :FiiliSevkTarihi, :KantarKayitId,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
    begin
      if (KdvDahil = 1) then KdvDahilStr = 'Evet';
      else KdvDahilStr = '';

      if (FStokTabanDovizId <> DovizId) then DovizKuruStr = Format_Float('', TabanDovizKuru1) || ' ' || DovizId || '-' || Format_Float('', TabanDovizKuru2) || ' ' || FStokTabanDovizId;
      else DovizKuruStr = '';
      suspend;
    end
  end
end
