CREATE PROCEDURE URETIMPARTINOBILGISIGETIR(
    AORMEPLANLAMAID INTEGER)
RETURNS (
    ORMEPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    PLANLANANMIKTAR NUMERIC(12,3),
    GERCEKLESENMIKTAR NUMERIC(12,3),
    TOPLAMSAKATORULEN NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    TARIH TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAIDSTR VARCHAR(135),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    ORMEMAKINAIDSTR VARCHAR(55),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    HAMADI VARCHAR(255),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100))
AS
begin

  Select OrmePlanlama.OrmePlanlamaId, OrmePlanlama.UretimPartiNo, OrmePlanlama.PlanlananMiktar, OrmePlanlama.Tarih, OrmePlanlama.OrmeMakinaId, OrmeMakina.Adi, OrmePlanlama.FirmaId, PFirma.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Ham.Adi,
         OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.ArtikelNo, SFirma.Adi,
         OrmePlanlama.GerceklesenMiktar, OrmePlanlama.ToplamSakatOrulen from OrmePlanlama
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
    left join Firma PFirma on PFirma.FirmaId = OrmePlanlama.FirmaId
    left join Firma SFirma on SFirma.FirmaId = BoyaSiparis.FirmaId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    where OrmePlanlama.OrmePlanlamaId = :AOrmePlanlamaId
    into :OrmePlanlamaId, :UretimPartiNo, :PlanlananMiktar, :Tarih, :OrmeMakinaId, :OrmeMakinaAdi, :FirmaId, :FirmaAdi, :BoyaSiparisYil, :BoyaSiparisId, :HamAdi,
         :MusteriSiparisNo, :ReferansSiparisNo, :ArtikelNo, :SiparisFirmaAdi, :GerceklesenMiktar, :ToplamSakatOrulen;

    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    if (FirmaId is not null) then FirmaIdStr = '(' || FirmaId || ') ' || FirmaAdi;
    if (OrmeMakinaId is not null) then OrmeMakinaIdStr = '(' || OrmeMakinaId || ') ' || OrmeMakinaAdi;

    KalanMiktar = PlanlananMiktar - GerceklesenMiktar + ToplamSakatOrulen;
    suspend;
end
