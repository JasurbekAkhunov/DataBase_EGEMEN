CREATE PROCEDURE BOYADEPOSAYIMBARKODLUDETAYEKLE(
    ABARKOD VARCHAR(40),
    ADEPOSAYIMID INTEGER,
    AKILO DOUBLE PRECISION,
    ABRUTKILO DOUBLE PRECISION,
    AMETRE DOUBLE PRECISION,
    AKAPADEDI INTEGER,
    AKULLANICIID INTEGER,
    ARAFID RAFKODUSIZE)
AS
declare variable Renkid integer;
declare variable DESENVARYANTID integer;
declare variable KaliteNo integer;
declare variable DesenID DESENKODUSIZE;
declare variable Notu    Memo;
declare variable FlamKodu   varchar(20);
declare variable MelanjKodu varchar(20);
declare variable Lot        varchar(30);
declare variable HamID      integer;
declare variable shtAlis        SmallInt;
declare variable shtFasonUretim SmallInt;
declare variable shtUretim      SmallInt;

declare variable FBoyaSiparisKayitID integer;
declare variable FBoyaSiparisDetayID integer;
declare variable FBoyaSiparisBaskiId integer;
declare variable FOrmeSiparisDetayID integer;
declare variable FUreticiId integer;
declare variable FDepoId     varchar(10);
declare variable IplikStokNumarator integer;
declare variable Sakat integer;
declare variable StokHareketID integer;
declare variable OrmeRafId integer;
declare variable IK     varchar(2);
declare variable TF     integer;
declare variable PaketId integer;
declare variable Kilo  Double Precision;
declare variable BrutKilo Double Precision;
declare variable FBoyaIrsaliyeDetayId Integer;
declare variable FHamStokSarfFisiDetayId integer;
declare variable Metre  Double Precision;
declare variable KapAdedi integer;
declare variable PartiKayitId integer;
declare variable PartiSinifId integer;
declare variable KS Integer;
declare variable FPartiStokTartiId integer;
begin
  if (AKilo is Null) then AKilo = 0;

  ABarkod = Replace(ABarkod, '*', '-');

  IK = EgemenCopyStr(ABarkod, 1, 2);

  TF = EgemenStrToIntOlurmu(EgemenCopyStr(ABarkod, 3, Char_Length(ABarkod) - 2));
  if (TF = 0) then exception BarkodHatali1;
  PaketId = EgemenStrToInt(EgemenCopyStr(ABarkod, 3, Char_Length(ABarkod) - 2));

  select count(*), Max(PartiStokTartiId) from PartiStokTarti where BarkodId = :ABarkod into :KS, :PaketId;
  if (KS > 0) then /*iade kaydidir*/
    IK = 'R-';
  else
    PaketId = EgemenStrToInt(EgemenCopyStr(ABarkod, 3, Char_Length(ABarkod) - 2));

  if (IK = 'K-') then
  begin
    Select PartiStokTarti.PartiStokTartiId, PartiStokTarti.Kilo, PartiStokTarti.BrutKilo, PartiStokTarti.Metre, PartiStokTarti.KapAdedi, PsKaliteKontrol.PartiKayitId, PsKaliteKontrol.PartiSinifId,
           PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, PartiStok.HamID, PartiStokTarti.DepoID from PsKaliteKontrol
    left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = PsKaliteKontrol.PartiStokTartiId
    Left Join StokHareket on StokHareket.BoyaIrsaliyeDetayID = PartiStokTarti.BoyaIrsaliyeDetayID
    left join PartiStok on PartiStok.PartiSinifId = PsKaliteKontrol.PartiSinifId
    where PsKaliteKontrol.PsKaliteKontrolId = :PaketId
    into :FPartiStokTartiId, :Kilo, :BrutKilo, :Metre, :KapAdedi, :PartiKayitId, :PartiSinifId, :FBoyaIrsaliyeDetayId, :FHamStokSarfFisiDetayId, :HamId, :FDepoId;
  end
  else
  if (IK = 'R-') then
  begin
    Select PartiStokTarti.PartiStokTartiId, PartiStokTarti.Kilo, PartiStokTarti.BrutKilo, PartiStokTarti.Metre, PartiStokTarti.KapAdedi, PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId,
           PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, PartiStok.HamID, PartiStokTarti.DepoID from PartiStokTarti
    left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
    where PartiStokTarti.PartiStokTartiId = :PaketId
    into :FPartiStokTartiId, :Kilo, :BrutKilo, :Metre, :KapAdedi, :PartiKayitId, :PartiSinifId, :FBoyaIrsaliyeDetayId, :FHamStokSarfFisiDetayId, :HamId, :FDepoId;
  end
  else
  if (IK = 'I-') then
  begin
    Select PartiStokTarti.PartiStokTartiId, PartiStokTarti.Kilo, PartiStokTarti.BrutKilo, PartiStokTarti.Metre, PartiStokTarti.KapAdedi, IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId,
           PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, PartiStok.HamID, PartiStokTarti.DepoID from IplikPaketleme
    left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiId
    Left Join StokHareket on StokHareket.BoyaIrsaliyeDetayID = PartiStokTarti.BoyaIrsaliyeDetayID
    left join PartiStok on PartiStok.PartiSinifId = IplikPaketleme.PartiSinifId
    where IplikPaketleme.IplikPaketlemeId = :PaketId
    into :FPartiStokTartiId, :Kilo, :BrutKilo, :Metre, :KapAdedi, :PartiKayitId, :PartiSinifId, :FBoyaIrsaliyeDetayId, :FHamStokSarfFisiDetayId, :HamId, :FDepoId;
  end

  if (PartiKayitId is Null) then Exception BarkodHatali2;

  if (AKilo = 0) then
  begin
    AKilo       = Kilo;
    ABrutKilo   = Kilo;
    AMetre      = Metre;
    AKapAdedi   = KapAdedi;

    if (AKilo is Null) then AKilo = 0;
  end

  insert into DepoSayimDetay(DepoSayimID, Barkod, PartiKayitId, PartiSinifId, HamID,  Kilo, BrutKilo, Metre, KapAdedi, KullaniciID, InsertKullaniciID, PartiStokTartiID, RafId)
                   Values (:ADepoSayimID, :ABarkod, :PartiKayitId, :PartiSinifId, :HamID, :AKilo, :ABrutKilo, :AMetre, :AKapAdedi, :AKullaniciID, :AKullaniciID, :FPartiStokTartiId, :ARafId);
end
