CREATE PROCEDURE BOYASIPARISRAPORU_10(
    AIMALATTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AFIRMAYETKILIID INTEGER,
    AISLEMTIPIID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AMUSTERIKALITENO VARCHAR(20),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AHAMID INTEGER,
    AKULLANICIID INTEGER,
    AKALITEID VARCHAR(20),
    ATALIMATKODU VARCHAR(20),
    AMAKINAID VARCHAR(10),
    ATEKRARURETIM INTEGER,
    ADESENID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AMARKALI INTEGER,
    AKOMISYONLU INTEGER,
    ALAMINASYON INTEGER,
    ARECETETURUID INTEGER,
    ABATCHNO VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ABASKITIPIID INTEGER,
    AHAMGRUP1ID VARCHAR(4),
    AHAMGRUP2ID VARCHAR(4),
    AHAMGRUP3ID VARCHAR(4),
    AHAMGRUP4ID VARCHAR(4),
    ASIPARISRECETEDURUMU INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    DESENVARYANTID INTEGER,
    VARYANTADI VARCHAR(40),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    DESENVINSERTKULLANICIADI VARCHAR(40),
    DESENVINSERTTARIHI TIMESTAMP,
    DESENVKULLANICIADI VARCHAR(40),
    DESENVDEGISIMTARIHI TIMESTAMP,
    NUMUNEGIDISTARIHI TIMESTAMP,
    OKEYGELISTARIHI TIMESTAMP,
    ACIKLAMA VARCHAR(255))
AS
begin
  For Select BoyaSiparisRaporu_2.BoyaSiparisKayitID, BoyaSiparisRaporu_2.BoyaSiparisDetayID, BoyaSiparisBaski.BoyaSiparisBaskiId,
             BoyaSiparisDetay.DesenId, Desen.Adi, BoyaSiparisBaski.DesenVaryantId, DesenVaryant.VaryantAdi,
             BoyaSiparisBaski.Kilo, BoyaSiparisBaski.Metre, IK.Adi, BoyaSiparisBaski.InsertTarihi, Sk.Adi,
             BoyaSiparisBaski.DegisimTarihi, DIK.Adi, DesenVaryant.InsertTarihi, DSK.Adi, DesenVaryant.DegisimTarihi,
             BoyaSiparisBaski.NumuneGidisTarihi, BoyaSiparisBaski.OkeyGelisTarihi, DesenVaryant.Aciklama
             from  BoyaSiparisRaporu_2(:AImalatTipiId,:ABaslangicTarihi,:ABitisTarihi,
                   :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AFirmaYetkiliId, :AIslemTipiId, :AIslemGrubuId, :ASatisTipi,
                   :ABoyaSiparisTipiId, :ALabReceteId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMusteriKaliteNo, :AOnayDurumu, :APartilenen,
                   :AKapaliDurumu, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASiparisiAlanKullaniciId,
                   :AYurtDisi, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AHamId, :AKullaniciId, :AKaliteId, :ATalimatKodu, :AMakinaId, :ATekrarUretim, :ADesenID, :ATemsilciId, :AMarkali, :AKomisyonlu, :ALaminasyon, :AReceteTuruId, :ABatchNo, 1, :ASertifikaNo, :ABaskiTipiId, :AHamGrup1Id, :AHamGrup2Id, :AHamGrup3Id, :AHamGrup4Id, :ASiparisReceteDurumu)
    left join BoyaSiparisBaski ON BoyaSiparisBaski.BoyaSiparisDetayId = BoyaSiparisRaporu_2.BoyaSiparisDetayId
    left join BoyaSiparisDetay ON BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisBaski.BoyaSiparisDetayId
    left join Desen ON Desen.DesenId = BoyaSiparisDetay.DesenId
    left join DesenVaryant ON DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
    Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisBaski.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisBaski.KullaniciId
    Left Join Kullanici DIK on DIK.KullaniciId = DesenVaryant.InsertKullaniciId
    Left Join Kullanici DSK on DSK.KullaniciId = DesenVaryant.KullaniciId
    where BoyaSiparisBaski.BoyaSiparisBaskiId is not null
    into :BoyaSiparisKayitID, :BoyaSiparisDetayID, :BoyaSiparisBaskiId, :DesenId, :DesenAdi, :DesenVaryantId, :VaryantAdi, :Kilo, :Metre, :InsertKullaniciAdi, :InsertTarihi,
         :KullaniciAdi, :DegisimTarihi, :DesenVInsertKullaniciAdi, :DesenVInsertTarihi, :DesenVKullaniciAdi, :DesenVDegisimTarihi, :NumuneGidisTarihi, :OkeyGelisTarihi, :Aciklama do
  begin
    suspend;
  end
end
