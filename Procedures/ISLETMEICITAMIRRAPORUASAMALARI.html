CREATE PROCEDURE ISLETMEICITAMIRRAPORUASAMALARI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATAMIRTIPIGRUPID VARCHAR(4),
    ATAMIRTIPIID INTEGER,
    ARAPORTIPI INTEGER,
    ABITTI INTEGER,
    APARTITIPI INTEGER,
    AFIRMAID VARCHAR(20),
    AUCRETDURUMU INTEGER,
    ATAMIRDUZELTMETIPIID VARCHAR(10),
    ATAMIRCOZUMSEKLIID INTEGER,
    AISLEMEALINACAK INTEGER,
    APARTIGRUPID VARCHAR(4),
    ASIPARISGRUPID VARCHAR(4),
    ARECETETURUID INTEGER,
    AHATA1ASAMAID INTEGER,
    AHATA2ASAMAID INTEGER,
    AHATA3ASAMAID INTEGER,
    AONLEYICIFAALIYETTIPIID INTEGER)
RETURNS (
    PARTITAMIRID INTEGER,
    PARTIKAYITID INTEGER,
    SIRA INTEGER,
    SERBESTTAMIR INTEGER,
    PARTIASAMALARIID INTEGER,
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    MAKINAID VARCHAR(10),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    GIRISPERSONELID INTEGER,
    GIRISPERSONELADI VARCHAR(40),
    CIKISPERSONELID INTEGER,
    CIKISPERSONELADI VARCHAR(40))
AS
declare variable PartiTamirSinifId integer;
begin
  For Select PartiTamirId, PartiKayitId from IsletmeIciTamirRaporu(:ABaslangicTarihi, :ABitisTarihi, :ATamirTipiGrupId, :ATamirTipiId, :ARaporTipi, :ABitti, :APartiTipi, :AFirmaId, :AUcretDurumu, :ATamirDuzeltmeTipiId, :ATamirCozumSekliId, :AIslemeAlinacak, :APartiGrupId, :ASiparisGrupId, :AReceteTuruId, :AHata1AsamaId, :AHata2AsamaId, :AHata3AsamaId, :AOnleyiciFaaliyetTipiId)
  into :PartiTamirId, :PartiKayitId do
  begin
    For Select PartiAsamalari.Sira, PartiAsamalari.PartiAsamalariId, PartiAsamalari.SerbestAsama, PartiAsamalari.AsamaId, Asama.Adi,
               PartiAsamalari.ReceteId, Recete.Adi, Min(PartiAsamaStokDetay.MakinaId), Min(PartiAsamaStok.GirisTarihi), Max(PartiAsamaStok.CikisTarihi),
               Min(PartiAsamaStokDetay.GirispersonelId), Gp.Adi, Min(PartiAsamaStokDetay.CikispersonelId), CP.Adi from PartiAsamalari
      left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
      left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId or Asama.AsamaId = recete.UstAsamaId
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and AsamayaDahil = 1
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
      left join Personel GP on GP.personelId = PartiAsamaStokDetay.GirispersonelId
      left join Personel CP on CP.personelId = PartiAsamaStokDetay.CikispersonelId
      where PartiAsamalari.PartiTamirId = :PartiTamirId
      group By PartiAsamalari.Sira, PartiAsamalari.PartiAsamalariId, PartiAsamalari.SerbestAsama, PartiAsamalari.AsamaId, Asama.Adi, PartiAsamalari.ReceteId, Recete.Adi, Gp.Adi, CP.adi
      into :Sira, :PartiAsamalariId, :SerbestTamir, :AsamaId, :AsamaAdi, :ReceteId, :ReceteAdi, :MakinaId, :GirisTarihi, :CikisTarihi, :GirisPersonelId, :GirisPersonelAdi, :CikisPersonelId, :CikisPersonelAdi
      do suspend;

    For Select PartiTamirSinifId from PartiTamirSinif where PartiTamirSinif.PartiTamirId = :PartiTamirId into :PartiTamirSinifId do
    begin
      for Select PartiAsamalari.Sira, PartiAsamalari.PartiAsamalariId, PartiAsamalari.SerbestAsama, PartiAsamalari.AsamaId, Asama.Adi,
              PartiAsamalari.ReceteId, Recete.Adi, Min(PartiAsamaStokDetay.MakinaId), Min(PartiAsamaStokDetay.GirisTarihi), Min(PartiAsamaStokDetay.CikisTarihi),
               Min(PartiAsamaStokDetay.GirispersonelId), Gp.Adi, Min(PartiAsamaStokDetay.CikispersonelId), CP.adi from PartiAsamaStokDetay
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
      left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
      left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId or Asama.AsamaId = recete.UstAsamaId
      left join Personel GP on GP.personelId = PartiAsamaStokDetay.GirispersonelId
      left join Personel CP on CP.personelId = PartiAsamaStokDetay.CikispersonelId
      where PartiAsamaStokDetay.PartiTamirSinifId = :PartiTamirSinifId and PartiAsamaStok.AsamayaDahil = 1
      group by PartiAsamalari.Sira, PartiAsamalari.PartiAsamalariId, PartiAsamalari.SerbestAsama, PartiAsamalari.AsamaId, Asama.Adi,
              PartiAsamalari.ReceteId, Recete.Adi, Gp.Adi, CP.adi
      into :Sira, :PartiAsamalariId, :SerbestTamir, :AsamaId, :AsamaAdi, :ReceteId, :ReceteAdi, :MakinaId, :GirisTarihi, :CikisTarihi, :GirisPersonelId, :GirisPersonelAdi, :CikisPersonelId, :CikisPersonelAdi
      do suspend;
    end
  end
end
