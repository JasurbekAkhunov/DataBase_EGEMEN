CREATE PROCEDURE RAFTRANSFERFISILISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    RAFTRANSFERFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  if (AHamStokMasterID is Null) then
  begin
    For Select RafTransferFisi.RafTransferFisiId, RafTransferFisi.FisYil, RafTransferFisi.FisNo, RafTransferFisi.FisTarihi,
               RafTransferFisi.Notu, IK.Adi, RafTransferFisi.InsertTarihi, Sk.Adi, RafTransferFisi.DegisimTarihi from RafTransferFisi
      Left Join Kullanici IK on IK.KullaniciId = RafTransferFisi.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = RafTransferFisi.KullaniciId
      where RafTransferFisi.FisTarihi >= :ABaslangicTarihi and RafTransferFisi.FisTarihi <= :ABitisTarihi
      into :RafTransferFisiId, :FisYil, :FisNo, :FisTarihi, :Notu,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
    begin
      FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';
      suspend;
    end
  end
  else
  begin
    For Select distinct RafTransferFisi.RafTransferFisiId, RafTransferFisi.FisYil, RafTransferFisi.FisNo, RafTransferFisi.FisTarihi,
               RafTransferFisi.Notu, IK.Adi, RafTransferFisi.InsertTarihi, Sk.Adi, RafTransferFisi.DegisimTarihi from RafTransferFisi
      left join RafTransferFisiDetay on RafTransferFisiDetay.RafTransferFisiID = RafTransferFisi.RafTransferFisiId
      left join RafDetay on RafDetay.RafDetayID = RafTransferFisiDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
      Left Join Kullanici IK on IK.KullaniciId = RafTransferFisi.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = RafTransferFisi.KullaniciId
      where HamStokDetay.HamStokMasterID = :AHamStokMasterID
      into :RafTransferFisiId, :FisYil, :FisNo, :FisTarihi, :Notu,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
    begin
      FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';
      suspend;
    end
  end
end
