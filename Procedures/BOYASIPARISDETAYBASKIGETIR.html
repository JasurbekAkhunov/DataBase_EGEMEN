CREATE PROCEDURE BOYASIPARISDETAYBASKIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    PROSESID INTEGER,
    PROSESADI VARCHAR(255),
    LOT VARCHAR(30),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIIDSTR VARCHAR(20),
    MUSTERISIPARISNO VARCHAR(50),
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISID INTEGER,
    LABRECETEATISNO VARCHAR(20),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    PAKETLEMESEKLIID INTEGER,
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    TERMINNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PLANLAMANOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    NUMUNEGIDISTARIHI TIMESTAMP,
    OKEYGELISTARIHI TIMESTAMP)
AS
begin
  for select BoyaSiparisDetay.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisBaski.BoyaSiparisBaskiId,
             BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.LabReceteId,
             LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.LabReceteAtisId,
             LabReceteAtis.AtisNoStr,
             BoyaSiparisDetay.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, BoyaSiparisBaski.Kilo, BoyaSiparisBaski.Metre,
             BoyaSiparisDetay.ProsesId, Proses.Adi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi,
             BoyaSiparisDetay.PaketlemeSekliId, BoyaSiparisDetay.HamId, Ham.Adi,
             BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi, BoyaSiparisDetay.TerminNotu, BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu,
             BoyaSiparisDetay.RenkDerinligiId, RenkDerinligi.Adi, BoyaSiparisDetay.Lot, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo,
             BoyaSiparisDetay.MusteriKaliteAdi, BoyaSiparisDetay.OnayDurumu, BoyaSiparisDetay.OnayTarihi, BoyaSiparisDetay.OnaylayanKullaniciId, Ok.Adi,
             Ham.Adi2, Ham.Adi3, Ham.Adi4, BoyaSiparisBaski.NumuneGidisTarihi, BoyaSiparisBaski.OkeyGelisTarihi from BoyaSiparisDetay
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join Proses on Proses.ProsesId = BoyaSiparisDetay.ProsesId
  left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
  left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
  Left Join Kullanici OK on OK.KullaniciId = BoyaSiparisDetay.OnaylayanKullaniciId
  left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
  where BoyaSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisBaski.BoyaSiparisBaskiId is not null
  order by BoyaSiparisDetay.BoyaSiparisDetayId
  into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :MusteriSiparisNo, :LabReceteId, :LabReceteKodu,
       :LabReceteAdi, :LabReceteAtisId, :LabReceteAtisNo, :DesenId, :DesenAdi, :VaryantAdi, :SiparisKilo, :SiparisMetre,
       :ProsesId, :ProsesAdi, :SiparisProsesId, :SiparisProsesAdi, :PaketlemeSekliId,
       :HamId, :HamAdi, :TerminTarihi, :RevizeTerminTarihi, :TerminNotu, :Notu, :PlanlamaNotu,
       :RenkDerinligiId, :RenkDerinligiIdStr, :Lot, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
       :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi,
       :HamAdi2, :HamAdi3, :HamAdi4, :NumuneGidisTarihi, :OkeyGelisTarihi do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    Select Sum(PartiStok.Kilo), Sum(PartiStok.Metre)  from PartiStok where PartiStok.BoyaSiparisBaskiId = :BoyaSiparisBaskiId and PartiStok.PartiStokIptalKayitId is null
      into :PartilenenKilo, :PartilenenMetre;

    if (PartilenenKilo is null) then PartilenenKilo = 0;
    if (PartilenenMetre is null) then PartilenenMetre = 0;

    KalanKilo = SiparisKilo - PartilenenKilo;
    KalanMetre = SiparisMetre - PartilenenMetre;
    suspend;
  end
end
