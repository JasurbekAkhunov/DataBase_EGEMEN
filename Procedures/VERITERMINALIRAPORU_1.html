CREATE PROCEDURE VERITERMINALIRAPORU_1(
    AMAKINAID VARCHAR(10),
    AMAKINAGRUBU VARCHAR(4),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKALINLIKID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKUMASCINSIID INTEGER,
    ALYCRATIPIID INTEGER,
    ARECETEID VARCHAR(20))
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTIASAMASTOKDETAYID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    SINIFKILO NUMERIC(12,3),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    CIKISKILO NUMERIC(12,3),
    CIKISMETRE NUMERIC(12,3),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    MAKINAID VARCHAR(20),
    MAKINASARJID INTEGER,
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30))
AS
declare variable PartiKayitID integer;
begin
  For Select Parti.PartiYil, Parti.PartiId, Parti.PartiGrupId, PartiGrup.Adi, Parti.AcilisTarihi, Parti.FirmaID, Firma.Adi,
             Parti.ReceteID, Recete.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             PartiAsamaStokDetay.PartiAsamaStokDetayID, PartiAsamaStokDetay.PartiSinifId, PartiAsamaStokDetay.PartiAsamalariId, partiStok.HamAdiToplami, PartiStok.Kilo,
             PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.Metre, PartiAsamaStokDetay.CikisKilo, PartiAsamaStokDetay.CikisMetre,
             PartiAsamaStok.AsamaId, Asama.Adi, PartiAsamaStokDetay.makinaId, PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.CikisTarihi,
             PartiAsamaStokDetay.MakinaSarjId from PartiAsamaStokDetay
    left join Makina on Makina.MakinaID = PartiAsamaStokDetay.MakinaId
    left join PartiStok on PartiStok.PartiSinifID = PartiAsamaStokDetay.PartiSinifId
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
    left join Ham on Ham.HamID = PartiStok.HamID
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Asama on Asama.AsamaID = PartiAsamaStok.AsamaId
    left join Firma on Firma.FirmaID = Parti.FirmaId
    left join PartiGrup on PartiGrup.PartiGrupID = Parti.PartiGrupId
    left join Recete on Recete.ReceteID = Parti.ReceteId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = Parti.BoyaSiparisKayitId
    where PartiAsamaStokDetay.GirisTarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.GirisTarihi <= :ABitisTarihi and
          (PartiAsamaStokDetay.MakinaId = :AMakinaID or :AMakinaID is null) and
          (Makina.MakinaGrubu = :AMakinaGrubu or (:AMakinaGrubu is null and Makina.TerminalNo > 0)) and
          (Parti.ReceteID = :AReceteID or :AReceteID is null) and
          (Ham.KalinlikID = :AKalinlikID or :AKalinlikID is null) and
          (Ham.IplikCinsiID = :AIplikCinsiID or :AIplikCinsiID is null) and
          (Ham.KumasCinsiID = :AKumasCinsiID or :AKumasCinsiID is null) and
          (Ham.LycraTipiID = :ALycraTipiID or :ALycraTipiID is null)
    into :PartiYil, :PartiId, :PartiGrupId, :PartiGrupAdi, :AcilisTarihi, :FirmaID, :FirmaAdi, :ReceteID, :ReceteAdi, :BoyaSiparisYil, :BoyaSiparisId,
         :PartiAsamaStokDetayID, :PartiSinifId, :PartiAsamalariId, :HamAdiToplami, :SinifKilo, :Kilo, :Metre, :CikisKilo, :CikisMetre,
         :AsamaId, :AsamaAdi, :MakinaId, :GirisTarihi, :CikisTarihi, :MakinaSarjId do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    suspend;
  end
end
