    BOYATL DOUBLE PRECISION,
    TUZAYTL DOUBLE PRECISION,
    SODAAYTL DOUBLE PRECISION,
    SULFATAYTL DOUBLE PRECISION,
    SIVIKOSTIKAYTL DOUBLE PRECISION,
    HPEROKSITAYTL DOUBLE PRECISION,
    KIMYASALAYTL DOUBLE PRECISION,
    BOYAAYTL DOUBLE PRECISION,
    KAZAN INTEGER,

AS
declare variable FIslemTipiId integer;
declare variable FPartiTamirId integer;
declare variable FIlaveProses integer;
declare variable FHamTipiId    integer;
declare variable FCikisTarihi TimeStamp;
declare variable FKilo Double Precision;
declare variable FMetre Double Precision;

declare variable FIslemGrubuId    integer;
declare variable FIslemGrubuAdi   varchar(20);
declare variable FReceteTuruId    integer;
declare variable FReceteTuruAdi   varchar(15);
declare variable FMakinaId        varchar(10);
declare variable FLastMakinaId    varchar(10);
declare variable FPartininAsilRecetesimi integer;
declare variable FLastPartininAsilRecetesimi integer;
declare variable FGunlukKapasite  Double Precision;
declare variable FAsamaId         integer;
declare variable FAsamaAdi        varchar(40);
declare variable FMakinaRaporGrupId  integer;
declare variable FMakinaGrubu      varchar(4);

declare variable FLastIslemGrubuId    integer;
declare variable FLastReceteTuruId    integer;
declare variable FLastAsamaId         integer;
declare variable FLastMakinaRaporGrupId  integer;
declare variable FAsamaGURaporGrupNo      integer;
declare variable FIslemGrubuGURaporGrupNo integer;
declare variable FReceteTuruGURaporGrupNo integer;
declare variable FPaketlemeSekliTup integer;
declare variable FFaturaKesilecek integer;
declare variable FUcretliTamir integer;
declare variable FFason integer;
declare variable FDogalgazSarfiyati  Double Precision;
declare variable FElektirikSarfiyati  Double Precision;
declare variable FOkunanSuSarfiyati  Double Precision;
declare variable FPartiKayitId    integer;
declare variable FPartiAsamalariId    integer;

declare variable TuzKimyasalTipiID    integer;
declare variable SodaKimyasalTipiID    integer;
declare variable SulfatKimyasalTipiID    integer;
declare variable SiviKostikKimyasalTipiID    integer;
declare variable HidrojenPerOksitKimyasalTipiID    integer;

declare variable FUrunTipiID    integer;
declare variable FKimyasalTipiID    integer;
declare variable SarfMiktar  Double Precision;
declare variable SarfMaliyet  Double Precision;

declare variable FTuz  Double Precision;
declare variable FSoda  Double Precision;
declare variable FSulfat  Double Precision;
declare variable FSiviKostik  Double Precision;
declare variable FHPeroksit  Double Precision;
declare variable FKimyasal  Double Precision;
declare variable FBoya  Double Precision;

declare variable FTuzTL  Double Precision;
declare variable FSodaTL  Double Precision;
declare variable FSulfatTL  Double Precision;
declare variable FSiviKostikTL  Double Precision;
declare variable FHPeroksitTL  Double Precision;
declare variable FKimyasalTL  Double Precision;
declare variable FBoyaTL  Double Precision;
declare variable FKazan    integer;
declare variable VardiyaSayisi      integer;
declare variable T1                Double Precision;
declare variable FLastVardiyaNo    integer;
declare variable FVardiyaNo       integer;
begin
  select TuzKimyasalTipiID, SodaKimyasalTipiID, SulfatKimyasalTipiID, SiviKostikKimyasalTipiID, HidrojenPerOksitKimyasalTipiID from Sistem
    into :TuzKimyasalTipiID, :SodaKimyasalTipiID, :SulfatKimyasalTipiID, :SiviKostikKimyasalTipiID, :HidrojenPerOksitKimyasalTipiID;

  Select Sum(coalesce(BoyahaneCalismaSaati, 0)), Sum(coalesce(KuruBolumCalismaSaati, 0)) from GunlukCalismaSaati
    where GunlukCalismaSaati.Tarih >= :AAylikBaslangic - 1 and GunlukCalismaSaati.Tarih <= :AAylikBitis
    into :CalismaGunSayisi, :KuruBolumCalismaGunSayisi;


  if (CalismaGunSayisi is not null) then CalismaGunSayisi = CalismaGunSayisi / 24;
  else CalismaGunSayisi = AAylikBitis - AAylikBaslangic;
  CalismaGunSayisi = Yuvarla(CalismaGunSayisi, 10);


  if (KuruBolumCalismaGunSayisi is not null) then KuruBolumCalismaGunSayisi = KuruBolumCalismaGunSayisi / 24;
  else KuruBolumCalismaGunSayisi = AAylikBaslangic - AAylikBaslangic;
  KuruBolumCalismaGunSayisi = Yuvarla(KuruBolumCalismaGunSayisi, 10);

  Select VardiyaSayisi from sistem where sistemid = 1 into :VardiyaSayisi;

  if (VardiyaSayisi = 0 or VardiyaSayisi is null) then VardiyaSayisi = 3;

  FLastVardiyaNo = -1;
  FLastIslemGrubuId = -1;
  FLastReceteTuruId = -1;
  FLastAsamaId      = -1;
  FLastMakinaId     = '';
  FLastMakinaRaporGrupId = -1;
  FLastPartininAsilRecetesimi = -1;

  for select PartiAsamalari.PartiAsamalariId, Parti.PartiKayitId, Parti.IslemGrubuId, IslemGrubu.Adi, IslemGrubu.GURaporGrupNo, Recete.ReceteTuruId, ReceteTuru.Adi, ReceteTuru.GURaporGrupNo,
             PartiAsamaStok.AsamaId, Asama.Adi, Asama.GURaporGrupNo, Parti.IslemTipiId, PartiStok.HamTipiId, PartiAsamalari.PartiTamirId,
             PartiAsamaStokDetay.MakinaId, Makina.MakinaGrubu, Makina.RaporGrupId, Makina.GunlukKapasite, PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.CikisTarihi,
             PaketlemeSekli.Tup, PartiAsamalari.IlaveProses, Parti.FaturaKesilecek, TamirCozumSekli.Ucretli, PartiAsamaStokDetay.Metre, Parti.Fason, Makina.Kazan, PartiAsamalari.PartininAsilRecetesimi from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    left join Unite on Unite.UniteId = Makina.UniteId
    left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join ReceteTuru on ReceteTuru.ReceteTuruId = Recete.ReceteTuruId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliId = BoyaSiparisDetay.PaketlemeSekliId
    left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
    where PartiAsamaStokDetay.Cikistarihi >= :AAylikBaslangic and PartiAsamaStokDetay.Cikistarihi < :AAylikBitis and
          Makina.UretimRaporlarinaDahil = 1 and Asama.GunlukUretimRaporunaDahil = 1 and
         (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
         (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null) and
         (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is Null) and
          Parti.PartiTipiId = :APartiTipiId
    order by Parti.IslemGrubuId, Recete.ReceteTuruId, PartiAsamaStok.AsamaId /* orderby değiştirme rapor bozulur */
    into :FPartiAsamalariId, :FPartiKayitId, :FIslemGrubuId, :FIslemGrubuAdi, :FIslemGrubuGURaporGrupNo, :FReceteTuruId, :FReceteTuruAdi, :FReceteTuruGURaporGrupNo,
         :FAsamaId, :FAsamaAdi, :FAsamaGURaporGrupNo, :FIslemTipiId, :FHamTipiId, :FPartiTamirId, :FMakinaId, :FMakinaGrubu, :FMakinaRaporGrupId,
         :FGunlukKapasite, :FKilo, :FCikisTarihi, :FPaketlemeSekliTup, :FIlaveProses, :FFaturaKesilecek, :FUcretliTamir, :FMetre, :FFason, :FKazan, :FPartininAsilRecetesimi do
  begin
    select DogalgazSarfiyati, ElektirikSarfiyati, OkunanSuSarfiyati from PartiEnerjiSarfiyatiHesapla(:FPartiKayitId)
         into :FDogalgazSarfiyati, :FElektirikSarfiyati, :FOkunanSuSarfiyati;

    T1 = TimeStampToDouble(FCikisTarihi) - TimeStampToDouble(f_stripTime(FCikisTarihi));
    FVardiyaNo = VardiyaHesapla(T1, VardiyaSayisi);

    FTuz = 0;
    FSoda = 0;
    FSulfat = 0;
    FSiviKostik = 0;
    FHPeroksit = 0;
    FKimyasal = 0;
    FBoya = 0;
    FTuzTL = 0;
    FSodaTL = 0;
    FSulfatTL = 0;
    FSiviKostikTL = 0;
    FHPeroksitTL = 0;
    FKimyasalTL = 0;
    FBoyaTL = 0;

    for select Urun.UrunTipiId, Urun.KimyasalTipiId, sum(StokDetay.Miktar / UrunBirim.Carpan), sum(StokDetay.Miktar  * StokDetay.NetBirimFiyat) from stokdetay
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = StokDetay.PartiAsamalariId
      left join Urun on Urun.UrunId = StokDetay.UrunId
      where StokDetay.PartiAsamalariId = :FPartiAsamalariId
      group by Urun.UrunTipiId, Urun.KimyasalTipiId
      into :FUrunTipiID, :FKimyasalTipiID, :SarfMiktar, :SarfMaliyet do
    begin
      if (FUrunTipiId = 0) then
      begin
        FBoya = SarfMiktar;
        FBoyaTL = SarfMaliyet;
      end
      else if (FKimyasalTipiId = TuzKimyasalTipiID) then
      begin
        FTuz = SarfMiktar;
        FTuzTL = SarfMaliyet;
      end
      else if (FKimyasalTipiId = SodaKimyasalTipiID) then
      begin
        FSoda = SarfMiktar;
        FSodaTL = SarfMaliyet;
      end
      else if (FKimyasalTipiId = SulfatKimyasalTipiID) then
      begin
        FSulfat = SarfMiktar;
        FSulfatTL = SarfMaliyet;
      end
      else if (FKimyasalTipiId = SiviKostikKimyasalTipiID) then
      begin
        FSiviKostik = SarfMiktar;
        FSiviKostikTL = SarfMaliyet;
      end
      else if (FKimyasalTipiId = HidrojenPerOksitKimyasalTipiID) then
      begin
        FHPeroksit = SarfMiktar;
        FHPeroksitTL = SarfMaliyet;
      end
      else
      begin
        FKimyasal = SarfMiktar;
        FKimyasalTL = SarfMaliyet;
      end
    end
