CREATE PROCEDURE BOYAFATURAPROSESRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASON INTEGER,
    AILAVEISLEMLERIDIKKATEAL INTEGER)
RETURNS (
    HIZMETACIKLAMASI VARCHAR(550),
    AY1MIKTAR NUMERIC(12,3),
    AY1TUTAR DOUBLE PRECISION,
    AY2MIKTAR NUMERIC(12,3),
    AY2TUTAR DOUBLE PRECISION,
    AY3MIKTAR NUMERIC(12,3),
    AY3TUTAR DOUBLE PRECISION,
    AY4MIKTAR NUMERIC(12,3),
    AY4TUTAR DOUBLE PRECISION,
    AY5MIKTAR NUMERIC(12,3),
    AY5TUTAR DOUBLE PRECISION,
    AY6MIKTAR NUMERIC(12,3),
    AY6TUTAR DOUBLE PRECISION,
    AY7MIKTAR NUMERIC(12,3),
    AY7TUTAR DOUBLE PRECISION,
    AY8MIKTAR NUMERIC(12,3),
    AY8TUTAR DOUBLE PRECISION,
    AY9MIKTAR NUMERIC(12,3),
    AY9TUTAR DOUBLE PRECISION,
    AY10MIKTAR NUMERIC(12,3),
    AY10TUTAR DOUBLE PRECISION,
    AY11MIKTAR NUMERIC(12,3),
    AY11TUTAR DOUBLE PRECISION,
    AY12MIKTAR NUMERIC(12,3),
    AY12TUTAR DOUBLE PRECISION,
    TOPLAMMIKTAR NUMERIC(12,3),
    TOPLAMTUTAR DOUBLE PRECISION,
    ORTALAMABIRIMFIYAT DOUBLE PRECISION)
AS
declare variable FSiparisProsesAdi       varchar(255);
declare variable FIlaveIslemlerAdiStr    varchar(255);
declare variable FAy                     integer;
declare variable FMiktar                 Numeric(12,3);
declare variable FTutar                  Double Precision;
declare variable FLastSiparisProsesAdi       varchar(255);
declare variable FLastIlaveIslemlerAdiStr    varchar(255);
begin
  FLastSiparisProsesAdi = '@';
  FLastIlaveIslemlerAdiStr = '@';

  For Select Ay, SiparisProsesAdi, IlaveIslemlerAdiStr, Sum(Miktar), Sum(Tutar) from BoyaFaturaProsesRaporuHazirla(:ABaslangicTarihi, :ABitisTarihi, :AFirmaId, :AFason, :AIlaveIslemleriDikkateAl)
  group by Ay, SiparisProsesAdi, IlaveIslemlerAdiStr
  order by SiparisProsesAdi, IlaveIslemlerAdiStr, Ay
/* order by degismeyecek yanlıs calısır */
  into :FAy, :FSiparisProsesAdi, :FIlaveIslemlerAdiStr, :FMiktar, :FTutar do
  begin
    if (FLastSiparisProsesAdi <> FSiparisProsesAdi or FLastIlaveIslemlerAdiStr <> FIlaveIslemlerAdiStr) then
    begin
      if (FLastSiparisProsesAdi <> '@') then
      begin
        if (ToplamMiktar <> 0) then OrtalamaBirimFiyat = ToplamTutar / ToplamMiktar;
        else OrtalamaBirimFiyat = 0;
        OrtalamaBirimFiyat = Yuvarla(OrtalamaBirimFiyat, 100);
        suspend;
      end

      FLastSiparisProsesAdi = FSiparisProsesAdi;
      FLastIlaveIslemlerAdiStr = FIlaveIslemlerAdiStr;

      Ay1Miktar  = 0;
      Ay1Tutar   = 0;
      Ay2Miktar  = 0;
      Ay2Tutar   = 0;
      Ay3Miktar  = 0;
      Ay3Tutar   = 0;
      Ay4Miktar  = 0;
      Ay4Tutar   = 0;
      Ay5Miktar  = 0;
      Ay5Tutar   = 0;
      Ay6Miktar  = 0;
      Ay6Tutar   = 0;
      Ay7Miktar  = 0;
      Ay7Tutar   = 0;
      Ay8Miktar  = 0;
      Ay8Tutar   = 0;
      Ay9Miktar  = 0;
      Ay9Tutar   = 0;
      Ay10Miktar = 0;
      Ay10Tutar  = 0;
      Ay11Miktar = 0;
      Ay11Tutar  = 0;
      Ay12Miktar = 0;
      Ay12Tutar  = 0;
      ToplamMiktar = 0;
      ToplamTutar  = 0;
      HizmetAciklamasi = FSiparisProsesAdi;
      if (FIlaveIslemlerAdiStr <> '') then HizmetAciklamasi = HizmetAciklamasi || ' + (' || FIlaveIslemlerAdiStr || ') ';
    end

    if (FAY = 1) then
    begin
      Ay1Miktar  = Ay1Miktar + FMiktar;
      Ay1Tutar   = Ay1Tutar + FTutar;
    end
    else if (FAY = 2) then
    begin
      Ay2Miktar  = Ay2Miktar + FMiktar;
      Ay2Tutar   = Ay2Tutar + FTutar;
    end
    else if (FAY = 3) then
    begin
      Ay3Miktar  = Ay3Miktar + FMiktar;
      Ay3Tutar   = Ay3Tutar + FTutar;
    end
    else if (FAY = 4) then
    begin
      Ay4Miktar  = Ay4Miktar + FMiktar;
      Ay4Tutar   = Ay4Tutar + FTutar;
    end
    else if (FAY = 5) then
    begin
      Ay5Miktar  = Ay5Miktar + FMiktar;
      Ay5Tutar   = Ay5Tutar + FTutar;
    end
    else if (FAY = 6) then
    begin
      Ay6Miktar  = Ay6Miktar + FMiktar;
      Ay6Tutar   = Ay6Tutar + FTutar;
    end
    else if (FAY = 7) then
    begin
      Ay7Miktar  = Ay7Miktar + FMiktar;
      Ay7Tutar   = Ay7Tutar + FTutar;
    end
    else if (FAY = 8) then
    begin
      Ay8Miktar  = Ay8Miktar + FMiktar;
      Ay8Tutar   = Ay8Tutar + FTutar;
    end
    else if (FAY = 9) then
    begin
      Ay9Miktar  = Ay9Miktar + FMiktar;
      Ay9Tutar   = Ay9Tutar + FTutar;
    end
    else if (FAY = 10) then
    begin
      Ay10Miktar  = Ay10Miktar + FMiktar;
      Ay10Tutar   = Ay10Tutar + FTutar;
    end
    else if (FAY = 11) then
    begin
      Ay11Miktar  = Ay11Miktar + FMiktar;
      Ay11Tutar   = Ay11Tutar + FTutar;
    end
    else if (FAY = 12) then
    begin
      Ay12Miktar  = Ay12Miktar + FMiktar;
      Ay12Tutar   = Ay12Tutar + FTutar;
    end

    ToplamMiktar = ToplamMiktar + FMiktar;
    ToplamTutar  = ToplamTutar + FTutar;
  end

  if (FLastSiparisProsesAdi <> '@') then
  begin
    if (ToplamMiktar <> 0) then OrtalamaBirimFiyat = ToplamTutar / ToplamMiktar;
    else OrtalamaBirimFiyat = 0;
    OrtalamaBirimFiyat = Yuvarla(OrtalamaBirimFiyat, 100);
    suspend;
  end
end
