CREATE PROCEDURE DESENISTASYONPLANLAMAGETIR(
    ADEPARTMANID INTEGER,
    ADESENISTASYONID INTEGER)
RETURNS (
    DESENISTASYONPLANLAMAID INTEGER,
    DESENPLANLAMAID INTEGER,
    DESENISTASYONID INTEGER,
    SEPETNO INTEGER,
    PLANLANANHAFTA VARCHAR(10),
    DESENISTEKLERIID INTEGER,
    DESENISTEKTIPIID VARCHAR(5),
    DESENISTEKTIPIADI VARCHAR(50),
    DESENISTEKTIPIKISAADI VARCHAR(10),
    DESENISTEKNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PLANLAMAACIKLAMA VARCHAR(100),
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(80),
    ISTEKTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    BASLAMATARIHI TIMESTAMP,
    BASLATANKULLANICIID INTEGER,
    BASLATANKULLANICIADI VARCHAR(40),
    ISTEKYAPANKULLANICIID INTEGER,
    ISTEKYAPANKULLANICIADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAKISAADI VARCHAR(10),
    DESENGRUPID VARCHAR(4),
    SABLONBOYUID VARCHAR(10),
    BASKICINSIADI VARCHAR(20),
    BASKITEKNIGIADI VARCHAR(20),
    DESENBEKLEMEDE INTEGER,
    DESENBEKLEBASLATANKULLANICIID INTEGER,
    DESENBEKLEBASLATANKULLANICIADI VARCHAR(40),
    DESENBEKLEMENOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ISTEKBEKLEMEDE INTEGER,
    ONCELIKNO INTEGER,
    HAFTAYIL INTEGER,
    HAFTANO INTEGER,
    NUMUNEBASKIPARTIYIL INTEGER,
    NUMUNEBASKIPARTIID INTEGER,
    TASARIMNO VARCHAR(20))
AS
declare variable MaxPartiKayitId integer;
begin
  For Select DesenIstasyonPlanlama.DesenIstasyonPlanlamaId, DesenIstasyonPlanlama.DesenIstasyonId, DesenIstasyonPlanlama.DesenPlanlamaID, DesenIstasyonPlanlama.SepetNo,
             DesenPlanlama.DesenIstekleriId, DesenIstekleri.DesenIstekTipiId, DesenIstekTipi.Adi, DesenIstekTipi.KisaAdi,
             Desen.DesenId, Desen.Adi, DesenIstekleri.IstekTarihi, DesenIstekleri.TerminTarihi, DesenIstekleri.IstekYapanKullaniciId,
             IYK.Adi, DesenIstekleri.UniteId, Unite.Adi, Desen.FirmaId, Firma.Adi, Firma.KisaAdi, Desen.DesenGrupId, SablonB.Olcu, BaskiCinsi.Adi, BaskiTeknigi.Adi,
             Desen.Beklemede, DesenBekleme.BaslatanKullaniciId, DBBK.Adi, DesenBekleme.Notu,
             DesenIstekleri.Beklemede, Hafta.Yil, Hafta.HaftaNo, DesenIstekleri.Notu, DesenPlanlama.PlanlamaAciklama,

             DesenIstekleri.BaslamaTarihi, DesenIstekleri.BaslatanKullaniciId, DIBK.Adi, DesenPlanlama.OncelikNo, Desen.TasarimNo from DesenIstasyon
  left join DesenIstasyonPlanlama on DesenIstasyonPlanlama.DesenIstasyonID = DesenIstasyon.DesenIstasyonId
  left join DesenPlanlama on DesenPlanlama.DesenPlanlamaID = DesenIstasyonPlanlama.DesenPlanlamaId
  left join Hafta on Hafta.HaftaKayitId = DesenPlanlama.HaftaKayitId
  left join DesenIstekleri on DesenIstekleri.DesenIstekleriId = DesenPlanlama.DesenIstekleriId
  left join Desen on Desen.DesenId = DesenPlanlama.DesenId
  left join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiId
  left join Firma on Firma.FirmaId = Desen.FirmaId
  left join DesenOlcu SablonB on SablonB.DesenOlcuID = Desen.SablonBoyuDesenOlcuId
  left join BaskiCinsi on BaskiCinsi.BaskiCinsiId = Desen.BaskiCinsiId
  left join BaskiTeknigi on BaskiTeknigi.BaskiTeknigiId = Desen.BaskiTeknigiId
  left join Kullanici IYK on IYK.KullaniciId = DesenIstekleri.IstekYapanKullaniciId
  left join DesenBekleme on DesenBekleme.DesenId = Desen.DesenId and DesenBekleme.Bitti = 0
  left join Kullanici DBBK on DBBK.KullaniciId = DesenBekleme.BaslatanKullaniciId
  left join Kullanici DIBK on DIBK.KullaniciId = DesenIstekleri.BaslatanKullaniciId
  left join Unite on Unite.UniteId = DesenIstekleri.UniteId
  where DesenIstasyon.DepartmanId = :ADepartmanId and (DesenIstasyon.DesenIstasyonID = :ADesenIstasyonID or :ADesenIstasyonID is null) and
        DesenIstasyonPlanlama.DesenIstasyonPlanlamaID is not null
  order by DesenIstasyonPlanlama.SepetNo
  into :DesenIstasyonPlanlamaId, :DesenIstasyonId, :DesenPlanlamaID, :SepetNo, :DesenIstekleriId, :DesenIstekTipiId, :DesenIstekTipiAdi, :DesenIstekTipiKisaAdi,
       :DesenId, :DesenAdi, :IstekTarihi, :TerminTarihi, :IstekYapanKullaniciId, :IstekYapanKullaniciAdi,
       :UniteId, :UniteAdi, :FirmaId, :FirmaAdi, :FirmaKisaAdi, :DesenGrupId,
       :SablonBoyuId, :BaskiCinsiAdi, :BaskiTeknigiAdi, :DesenBeklemede, :DesenBekleBaslatanKullaniciId, :DesenBekleBaslatanKullaniciAdi,
       :DesenBeklemeNotu, :IstekBeklemede, :HaftaYil, :HaftaNo, :DesenIstekNotu, :PlanlamaAciklama,

       :Baslamatarihi, :BaslatanKullaniciId, :BaslatanKullaniciAdi, :OncelikNo, :TasarimNo do
  begin
    PlanlananHafta = '';

    if (HaftaNo is not null) then
    begin
      PlanlananHafta = '(' || EgemenIntegerToStr(HaftaYil) || ') ' || EgemenIntegerToStr(HaftaNo);
    end

    MaxPartiKayitId = Null;
    NumuneBaskiPartiYil = Null;
    NumuneBaskiPartiId = Null;

    Select max(parti.PartiKayitId) from DesenVaryant
      left join Parti on Parti.DesenVaryantID = DesenVaryant.DesenVaryantID
      where DesenVaryant.DesenId = :DesenId and DesenVaryant.DesenVaryantID is not null and
            Parti.IslemTipiId = 5
      into :MaxPartiKayitId;

    if (MaxPartiKayitId is not null) then
    begin
      Select Parti.PartiYil, Parti.PartiId from Parti
        where Parti.PartiKayitId = :MaxPartiKayitId
        into :NumuneBaskiPartiYil, :NumuneBaskiPartiId;
    end


    suspend;
  end
end
