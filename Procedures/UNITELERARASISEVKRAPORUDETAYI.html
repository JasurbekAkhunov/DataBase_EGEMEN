CREATE PROCEDURE UNITELERARASISEVKRAPORUDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AGONDERENUNITEID INTEGER,
    AALICIUNITEID INTEGER,
    AARACID VARCHAR(11),
    ASINIF VARCHAR(1))
RETURNS (
    UNITEIRSALIYEID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    KAPADEDI INTEGER,
    AMBALAJLIBRUTKILO NUMERIC(12,3),
    KILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    MIKTARBIRIM SMALLINT,
    MIKTARBIRIMSTR VARCHAR(7),
    YAPILACAKISLEMLER VARCHAR(255),
    ILAVEISLEMLER VARCHAR(255),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    RENKADI VARCHAR(40),
    HAMRENKADI VARCHAR(50),
    BIRIMFIYAT DOUBLE PRECISION,
    ACIKLAMA VARCHAR(60),
    MUSTERISIPARISNO VARCHAR(50),
    GIDENKILO DOUBLE PRECISION,
    GIDENNETKILO DOUBLE PRECISION,
    GIDENMETRE DOUBLE PRECISION,
    GELENKILO DOUBLE PRECISION,
    GELENNETKILO DOUBLE PRECISION,
    GELENMETRE DOUBLE PRECISION)
AS
Declare Variable FGonderenUniteID  integer;
Declare Variable FAliciUniteID  integer;
begin
  for Select UniteIrsaliye.UniteIrsaliyeId, UniteIrsaliyeDetay.PartiKayitId, Parti.PartiYil, Parti.PartiId,UniteIrsaliyeDetay.PartiSinifId,
             PartiStok.HamAdiToplami, UniteIrsaliyeDetay.ReceteId, Recete.Adi, UniteIrsaliyeDetay.KapAdedi, UniteIrsaliyeDetay.AmbalajliBrutKilo, UniteIrsaliyeDetay.Kilo, UniteIrsaliyeDetay.NetKilo,
             UniteIrsaliyeDetay.YapilacakIslemler, UniteIrsaliyeDetay.IlaveIslemler, Parti.FirmaId, Firma.Adi,
             BoyaSiparisDetay.RenkAdi, Renk.Adi, UniteIrsaliyeDetay.BirimFiyat, UniteIrsaliyeDetay.Aciklama, BoyaSiparisDetay.BoyaSiparisKayitID,
             UniteIrsaliyeDetay.Metre, UniteIrsaliyeDetay.MiktarBirim, BoyaSiparisDetay.MusteriSiparisNo,
             UniteIrsaliye.GonderenUniteID, UniteIrsaliye.AliciUniteID from UniteIrsaliye
    left join UniteIrsaliyeDetay on UniteIrsaliyeDetay.UniteIrsaliyeId = UniteIrsaliye.UniteIrsaliyeId
    Left Join Parti on Parti.PartiKayitId = UniteIrsaliyeDetay.PartiKayitId
    Left Join Firma on Firma.FirmaId = Parti.FirmaId
    Left Join Recete on Recete.ReceteId = UniteIrsaliyeDetay.ReceteId
    Left Join PartiStok on PartiStok.PartiSinifId = UniteIrsaliyeDetay.PartiSinifId
    Left Join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    Left Join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
    Left Join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
    where UniteIrsaliye.IrsaliyeTarihi >= :ABaslangictarihi and UniteIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
          (UniteIrsaliye.GonderenUniteId = :AGonderenUniteId or :AGonderenUniteId is null) and
          (UniteIrsaliye.AliciUniteId = :AAliciUniteId or :AAliciUniteId is null) and
          (UniteIrsaliye.AracId = :AAracId or :AAracId is null) and
          (UniteIrsaliye.Sinif = :ASinif or :ASinif is null)
    into :UniteIrsaliyeId, :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :HamAdiToplami, :ReceteId, :ReceteAdi,
         :KapAdedi, :AmbalajliBrutKilo, :Kilo, :NetKilo, :YapilacakIslemler, :IlaveIslemler, :FirmaId, :FirmaAdi, :RenkAdi, :HamRenkAdi,
         :BirimFiyat, :Aciklama, :BoyaSiparisKayitID, :Metre, :MiktarBirim, :MusteriSiparisNo, :FGonderenUniteID, :FAliciUniteID do
  begin
    MiktarBirimStr = '';
    if (MiktarBirim = 0) then MiktarBirimStr = 'Kilo';
    else MiktarBirimStr = 'Metre';

    if (PartiYil is null) then PartiIdStr = '';
    else PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    Select GidenKilo, GidenNetKilo, GidenMetre, GelenKilo, GelenNetKilo, GelenMetre from UniteGidenGelenMiktarGetir(:FGonderenUniteID, :FAliciUniteID, :PartiSinifID)
      into :GidenKilo, :GidenNetKilo, :GidenMetre, :GelenKilo, :GelenNetKilo, :GelenMetre;

    suspend;
  end
end
