CREATE PROCEDURE ELYAFURETIMISEMRIRECETEGETIR(
    AELYAFURETIMISEMRIKAYITID INTEGER)
RETURNS (
    ELYAFURETIMISEMRIRECETEID INTEGER,
    ELYAFURETIMISEMRIKAYITID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MIKTAR DOUBLE PRECISION,
    LOT VARCHAR(30),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    KALITENO INTEGER,
    GELISSIRASI INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For select ElyafUretimIsEmriRecete.ElyafUretimIsEmriKayitId, ElyafUretimIsEmriRecete.ElyafUretimIsEmriReceteId, IplikIrsaliyeDetay.HamId,
             Ham.Adi, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.Lot, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
             IplikIrsaliyeDetay.HareketTipi, Ik.Adi, ElyafUretimIsEmriRecete.InsertTarihi, Sk.Adi, ElyafUretimIsEmriRecete.DegisimTarihi,
             ElyafUretimIsEmriRecete.Notu, IplikIrsaliyeDetay.KaliteNo, IplikIrsaliyeDetay.GelisSirasi from ElyafUretimIsEmriRecete
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = ElyafUretimIsEmriRecete.IplikIrsaliyeDetayId
  left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
  left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
  Left Join Kullanici IK on IK.KullaniciId = ElyafUretimIsEmriRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = ElyafUretimIsEmriRecete.KullaniciId
  where ElyafUretimIsEmriRecete.ElyafUretimIsEmriKayitId = :AElyafUretimIsEmriKayitId
  into :ElyafUretimIsEmriKayitId, :ElyafUretimIsEmriReceteId, :HamId,
       :HamAdi, :Miktar, :Lot, :IrsaliyeNo, :IrsaliyeTarihi, :HareketTipi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :Notu, :KaliteNo, :GelisSirasi do
  begin
    if (HareketTipi = 0) then HareketTipiStr = 'Alış';
    else if (HareketTipi = 5) then HareketTipiStr = 'Üretim İade';
    else if (HareketTipi = 6) then HareketTipiStr = 'Telef Giriş Fişi';
    else if (HareketTipi = 8) then HareketTipiStr = 'Üretim (boya)';
    else if (HareketTipi = 12) then HareketTipiStr = 'Depo Transfer';
    else HareketTipiStr = '????';

    suspend;
  end
end
