CREATE PROCEDURE BOYASIPARISKOPYALA(
    ABOYASIPARISKAYITID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER)
AS
declare variable BoyaSiparisKayitID  integer;
declare variable BoyaSiparisDetayId integer;
declare variable ABoyaSiparisDetayId integer;
declare variable SiparisKopyalarkenFiyatiTasi integer;
declare variable PartilenmisBSipKopyalanabilir integer;
declare variable PartiVar integer;
begin
  Select SiparisKopyalarkenFiyatiTasi, PartilenmisBSipKopyalanabilir from sistem where sistemid = 1 into :SiparisKopyalarkenFiyatiTasi, :PartilenmisBSipKopyalanabilir;

  if (PartilenmisBSipKopyalanabilir = 0) then
  begin
    Select Count(*) from Parti where Parti.BoyaSiparisKayitId = :ABoyaSiparisKayitId into :PartiVar;

    if (PartiVar > 0) then exception KayitHatasi ' Partilenmiş Sipariş Kopyalanamaz.';
  end

  BoyaSiparisYil = f_Year(Current_date);

  BoyaSiparisId = Null;

  Select Id from BoyaSiparisNumaratoruAl(:BoyaSiparisYil) into :BoyaSiparisId;



  insert into BoyaSiparis(BoyaSiparisYil, BoyaSiparisId, FirmaId, SiparisTarihi, SiparisiAlanKullaniciId, FirmaYetkiliId, IslemTipiId,
                          SiparisGrupId, SiparisGrup2Id, Fason, OdemeSekliId, TeslimSekliId, YurtDisi, Notu, SiparisTipiId, PSKKEtiketTipiId, InsertKullaniciId, KullaniciId)
             Select :BoyaSiparisYil, :BoyaSiparisId, FirmaId, Current_date, :AKullaniciId, FirmaYetkiliId, IslemTipiId,
                     SiparisGrupId, SiparisGrup2Id, Fason, OdemeSekliId, TeslimSekliId, YurtDisi, Notu, SiparisTipiId, PSKKEtiketTipiId, :AKullaniciId, :AKullaniciId from BoyaSiparis
               where BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitID;

  Select BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :BoyaSiparisYil and BoyaSiparis.BoyaSiparisID = :BoyaSiparisID into :BoyaSiparisKayitId;

  For Select BoyaSiparisDetayId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisKayitID = :ABoyaSiparisKayitID into :ABoyaSiparisDetayId do
  begin
    BoyaSiparisDetayId = gen_id(BoyaSiparisDetay_Id_Gen, 1);

    if (SiparisKopyalarkenFiyatiTasi = 0) then
    begin
      Insert into BoyaSiparisDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisRenkNo, FirmaSevkAdresiId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
                                   RenkId, RenkAdi, ProsesId, SiparisProsesId, SiparisKilo, SiparisMetre, PaketlemeSekliId, RenkDerinligiId, BirimFiyat, HamStokBirimId,
                                   FaturaEdilecekMiktarTipi, DovizId, UcretliTamir, FaturaKesilecek, VadeGun, OdemeSekliId, HamId, Pus, Feine, Lot, MusteriRenkKoduveAdi,
                                   DesenId, DesenKenarYazisi, DesenIlkBaski, MelanjKodu, FlamKodu,
                                   KaliteId, MusteriKaliteNo, MusteriKaliteAdi, Denye, IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen,
                                   CekmezlikBoyIstenen, YikamaHasligiIstenen, SansIstenen, ToleransEn, ToleransGramaj, ToleransEndenCekmezlik, ToleransBoydanCekmezlik,
                                   ToleransDonme, ToleransSans, TerminTarihi, RevizeTerminTarihi, TerminNotu, Notu, PlanlamaNotu,
                                   YikamaDereceId, KurutmaSekliId, YikamaHasligiTestiId, OngoruYapilacak, FirmaAdimizEtiketteCiksinmi, MaximumFireOrani, MinimumFireOrani, IslemGrubuId,
                                   FasonBirimFiyati, FasonBirimFiyatDovizId, BaskiBirimFiyati, BaskiBirimFiyatDovizId, InsertKullaniciId, KullaniciId, KopyaSonuUpdateYok, SiparisteISBrutNetDaradan, PSKKEtiketTipiId)
        Select :BoyaSiparisKayitId, :BoyaSiparisDetayId, BoyaSiparisRenkNo, FirmaSevkAdresiId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
               RenkId, RenkAdi, ProsesId, SiparisProsesId, SiparisKilo, SiparisMetre, PaketlemeSekliId, RenkDerinligiId, 0, HamStokBirimId,
               FaturaEdilecekMiktarTipi, DovizId, UcretliTamir, FaturaKesilecek, VadeGun, OdemeSekliId, HamId, Pus, Feine, Lot, MusteriRenkKoduveAdi,
               DesenId, DesenKenarYazisi, DesenIlkBaski, MelanjKodu, FlamKodu,
               KaliteId, MusteriKaliteNo, MusteriKaliteAdi, Denye, IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen,
               CekmezlikBoyIstenen, YikamaHasligiIstenen, SansIstenen, ToleransEn, ToleransGramaj, ToleransEndenCekmezlik, ToleransBoydanCekmezlik,
               ToleransDonme, ToleransSans, TerminTarihi, RevizeTerminTarihi, TerminNotu, Notu, PlanlamaNotu,
               YikamaDereceId, KurutmaSekliId, YikamaHasligiTestiId, OngoruYapilacak, FirmaAdimizEtiketteCiksinmi, MaximumFireOrani, MinimumFireOrani, IslemGrubuId, 0, FasonBirimFiyatDovizId, 0, BaskiBirimFiyatDovizId, :AKullaniciId, :AKullaniciId, 1, SiparisteISBrutNetDaradan, PSKKEtiketTipiId from BoyaSiparisDetay
        where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId;
    end
    else
    begin
      Insert into BoyaSiparisDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisRenkNo, FirmaSevkAdresiId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
                                   RenkId, RenkAdi, ProsesId, SiparisProsesId, SiparisKilo, SiparisMetre, PaketlemeSekliId, RenkDerinligiId, BirimFiyat, HamStokBirimId,
                                   FaturaEdilecekMiktarTipi, DovizId, UcretliTamir, FaturaKesilecek, VadeGun, OdemeSekliId, HamId, Pus, Feine, Lot, MusteriRenkKoduveAdi,
                                   DesenId, DesenKenarYazisi, DesenIlkBaski, MelanjKodu, FlamKodu,
                                   KaliteId, MusteriKaliteNo, MusteriKaliteAdi, Denye, IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen,
                                   CekmezlikBoyIstenen, YikamaHasligiIstenen, SansIstenen, ToleransEn, ToleransGramaj, ToleransEndenCekmezlik, ToleransBoydanCekmezlik,
                                   ToleransDonme, ToleransSans, TerminTarihi, RevizeTerminTarihi, TerminNotu, Notu, PlanlamaNotu,
                                   YikamaDereceId, KurutmaSekliId, YikamaHasligiTestiId, OngoruYapilacak, FirmaAdimizEtiketteCiksinmi, MaximumFireOrani, MinimumFireOrani, IslemGrubuId,
                                   FasonBirimFiyati, FasonBirimFiyatDovizId, BaskiBirimFiyati, BaskiBirimFiyatDovizId, InsertKullaniciId, KullaniciId, KopyaSonuUpdateYok, SiparisteISBrutNetDaradan, PSKKEtiketTipiId)
        Select :BoyaSiparisKayitId, :BoyaSiparisDetayId, BoyaSiparisRenkNo, FirmaSevkAdresiId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
               RenkId, RenkAdi, ProsesId, SiparisProsesId, SiparisKilo, SiparisMetre, PaketlemeSekliId, RenkDerinligiId, BirimFiyat, HamStokBirimId,
               FaturaEdilecekMiktarTipi, DovizId, UcretliTamir, FaturaKesilecek, VadeGun, OdemeSekliId, HamId, Pus, Feine, Lot, MusteriRenkKoduveAdi,
               DesenId, DesenKenarYazisi, DesenIlkBaski, MelanjKodu, FlamKodu,
               KaliteId, MusteriKaliteNo, MusteriKaliteAdi, Denye, IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen,
               CekmezlikBoyIstenen, YikamaHasligiIstenen, SansIstenen, ToleransEn, ToleransGramaj, ToleransEndenCekmezlik, ToleransBoydanCekmezlik,
               ToleransDonme, ToleransSans, TerminTarihi, RevizeTerminTarihi, TerminNotu, Notu, PlanlamaNotu,
               YikamaDereceId, KurutmaSekliId, YikamaHasligiTestiId, OngoruYapilacak, FirmaAdimizEtiketteCiksinmi, MaximumFireOrani, MinimumFireOrani, IslemGrubuId,
               FasonBirimFiyati, FasonBirimFiyatDovizId, BaskiBirimFiyati, BaskiBirimFiyatDovizId, :AKullaniciId, :AKullaniciId, 1, SiparisteISBrutNetDaradan, PSKKEtiketTipiId from BoyaSiparisDetay
        where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId;
    end

    Insert into SiparisIlaveIslemler(BoyaSiparisDetayId, BoyaSiparisKayitId, BirimFiyat, DovizId, IlaveIslemlerId, InsertKullaniciId, KullaniciId)
      Select :BoyaSiparisDetayId, :BoyaSiparisKayitId, 0, DovizId, IlaveIslemlerId, :AKullaniciId, :AKullaniciId from SiparisIlaveIslemler
      where SiparisIlaveIslemler.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

    Insert into SiparisOzelTestler(BoyaSiparisDetayId, BoyaSiparisKayitId, OzelTestId, OzelTestStandardiId, Istenen, InsertKullaniciId, KullaniciId)
      Select :BoyaSiparisDetayId, :BoyaSiparisKayitId, OzelTestId, OzelTestStandardiId, Istenen, :AKullaniciId, :AKullaniciId from SiparisOzelTestler
      where SiparisOzelTestler.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

    Insert into SiparisOzelApreler(BoyaSiparisDetayId, BoyaSiparisKayitId, OzelApreId, InsertKullaniciId, KullaniciId)
      Select :BoyaSiparisDetayId, :BoyaSiparisKayitId, OzelApreId, :AKullaniciId, :AKullaniciId from SiparisOzelApreler
      where SiparisOzelApreler.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

    update BoyaSiparisDetay set KopyaSonuUpdateYok = 0 where BoyaSiparisDetayId = :BoyaSiparisDetayId;
  end
  suspend;
end
