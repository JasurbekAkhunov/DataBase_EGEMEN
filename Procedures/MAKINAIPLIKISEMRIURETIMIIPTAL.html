CREATE PROCEDURE MAKINAIPLIKISEMRIURETIMIIPTAL(
    AIPLIKMAKINAID VARCHAR(10),
    AIPLIKURETIMISEMRIKAYITID INTEGER)
RETURNS (
    STATUS INTEGER,
    MESAJ VARCHAR(100),
    IPLIKURETIMISEMRIKAYITID INTEGER,
    IPLIKURETIMISEMRIYIL INTEGER,
    IPLIKURETIMISEMRIID INTEGER,
    LOT VARCHAR(30),
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    BUKUMTIPIADI VARCHAR(10),
    HARMANADI VARCHAR(50),
    HARMANCINSIADI VARCHAR(50),
    KALINLIKNE INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    IPLIKCINSIURETIMPARTINOLABEL VARCHAR(10),
    COPSTIPIADI VARCHAR(20),
    BUKUMSAYISI INTEGER,
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    BASLAMATARIHI TIMESTAMP,
    TAHMINIBITISTARIHI TIMESTAMP)
AS
declare variable FIplikUretimIsEmriKayitId integer;
declare variable EskiIplikUretimIsEmriKayitId integer;
declare variable FIplikMakinaTipiId         integer;
declare variable FIplikMakinaId             Varchar(10);
declare variable FBitisTarihi               TimeStamp;
declare variable FIsEmriIplikMakinaTipiId   integer;
declare variable KS  integer;
begin
  Select count(*) from IplikUretimIsEmri
    where IplikUretimIsEmri.BitisTarihi is null and IplikUretimIsEmri.BaslamaTarihi is not null and IplikUretimIsEmri.IplikMakinaID = :AIplikMakinaID
    into :KS;

  if (KS = 0) then
  begin
    Status = -5;
    Mesaj = 'Makina da Başlamış iş Yok...';
  end

  Status = 0;

  if (AIplikUretimIsEmriKayitId is not null) then
  begin
    Select IplikUretimIsEmri.IplikUretimIsEmriKayitId, IplikUretimIsEmri.IplikUretimIsEmriYil, IplikUretimIsEmri.IplikUretimIsEmriId,
           IplikUretimIsEmri.IplikMakinaTipiId, IplikLot.Lot, IplikLot.KalinlikNe, IplikCinsi.Adi, IplikUretimIsEmri.PlanlananMiktar,
           IplikUretimIsEmri.GerceklesenMiktar, Harman.HarmanYil, Harman.HarmanId, Harman.Adi, HarmanCinsi.Adi,
           IplikUretimIsEmri.IplikMakinaId, IplikUretimIsEmri.BaslamaTarihi, BukumTipi.Adi, IplikUretimIsEmri.BitisTarihi,
           IplikLot.BukumSayisi, IplikCinsi.UretimPartiNoLabel, CopsTipi.Adi from IplikUretimIsEmri
      left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotID
      left join IplikCinsi on IplikCinsi.IplikCinsiID = IplikLot.IplikCinsiID
      left join BukumTipi on BukumTipi.BukumTipiID = IplikLot.BukumTipiID
      left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitID
      left join HarmanCinsi on HarmanCinsi.HarmanCinsiID = Harman.HarmanCinsiID
      left join CopsTipi on CopsTipi.CopsTipiID = IplikUretimIsEmri.CopsTipiID
      where IplikUretimIsEmri.IplikUretimIsEmriKayitId = :AIplikUretimIsEmriKayitId
      order by IplikUretimIsEmri.Sira
      into :IplikUretimIsEmriKayitId, :IplikUretimIsEmriYil, :IplikUretimIsEmriId, :FIsEmriIplikMakinaTipiId, :Lot, :KalinlikNe, :IplikCinsiAdi,
           :PlanlananMiktar, :GerceklesenMiktar, :HarmanYil, :HarmanId, :HarmanAdi, :HarmanCinsiAdi, :FIplikMakinaId, :BaslamaTarihi, :BukumTipiAdi,
           :FBitisTarihi, :BukumSayisi, :IplikCinsiUretimPartiNoLabel, :CopsTipiAdi;

    if (FBitisTarihi is not null) then
    begin
      Status = -1;
      Mesaj = 'İş Emri''nin Üretimi Bitmiş.';
    end

    if (BaslamaTarihi is not null and AIplikMakinaId = FIplikMakinaId) then
    begin
      Status = -2;
      Mesaj = 'İş Emri''nin Üretimine Zaten Başlanmış.';
    end

    if (AIplikMakinaId <> FIplikMakinaId) then
    begin
      Status = -3;
      Mesaj = 'Bu İş Emri başka Makinaya Planlanmış.' || ' (Makina Kodu = ' || FIplikMakinaId || ')';
    end

    Select IplikMakina.IplikMakinaTipiID from IplikMakina where IplikMakina.IplikMakinaID = :AIplikMakinaID into :FIplikMakinaTipiID;

    if (FIplikMakinaTipiID <> FIsEmriIplikMakinaTipiId) then
    begin
      Status = -4;
      Mesaj = 'İş Emri ile Makina''nın Tipi Uygun Değil.';
    end
  end


  if (Status = 0) then
  begin
    Select First 1 IplikUretimIsEmri.IplikUretimIsEmriKayitId, IplikUretimIsEmri.BaslamaTarihi from IplikUretimIsEmri
      where IplikUretimIsEmri.BitisTarihi is null and IplikUretimIsEmri.BaslamaTarihi is not null and IplikUretimIsEmri.IplikMakinaID = :AIplikMakinaID
      order by IplikUretimIsEmri.Sira
      into :EskiIplikUretimIsEmriKayitId, :BaslamaTarihi;

    if (AIplikUretimIsEmriKayitId is not null) then
    begin
      Update IplikUretimIsEmri Set BaslamaTarihi = Null where IplikUretimIsEmri.IplikUretimIsEmriKayitId = :EskiIplikUretimIsEmriKayitId;

  /* sepetteki ilk sırada iş emrini buluyoruz ona göre sürükle bırak yapıcaz..*/
      Select First 1 IplikUretimIsEmri.IplikUretimIsEmriKayitId from IplikUretimIsEmri
        where IplikUretimIsEmri.BitisTarihi is null and IplikUretimIsEmri.IplikMakinaId = :AIplikMakinaId
        order by IplikUretimIsEmri.Sira
        into :FIplikUretimIsEmriKayitId;

      if (FIplikUretimIsEmriKayitId is null or FIplikUretimIsEmriKayitId <> AIplikUretimIsEmriKayitId) then
      begin
        Execute procedure IplikPlanlamaSurukleBirak(Null, :AIplikMakinaId, :AIplikUretimIsEmriKayitId, :FIplikUretimIsEmriKayitId, Null, 1);
      end

      Update IplikMakinaSayac Set IplikUretimIsEmriKayitId = :AIplikUretimIsEmriKayitId
        where IplikMakinaSayac.IplikUretimIsEmriKayitId = :EskiIplikUretimIsEmriKayitId and IplikMakinaSayac.IplikMakinaID = :AIplikMakinaID;

      Update IplikUretimIsEmri Set BaslamaTarihi = :BaslamaTarihi where IplikUretimIsEmri.IplikUretimIsEmriKayitID = :AIplikUretimIsEmriKayitID;
      Status = 1;
    end
    else
    begin
      Select Count(*) from IplikMakinaSayac
        where IplikMakinaSayac.IplikUretimIsEmriKayitId = :EskiIplikUretimIsEmriKayitId and IplikMakinaSayac.IplikMakinaID = :AIplikMakinaID
        into :KS;

      if (KS > 0) then
      begin
        Status = -6;
      end
      else
      begin
        Update IplikUretimIsEmri Set BaslamaTarihi = Null where IplikUretimIsEmri.IplikUretimIsEmriKayitId = :EskiIplikUretimIsEmriKayitId;
        Status = 1;
      end
    end
  end

  suspend;
end
