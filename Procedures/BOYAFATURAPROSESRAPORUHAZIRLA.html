CREATE PROCEDURE BOYAFATURAPROSESRAPORUHAZIRLA(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASON INTEGER,
    AILAVEISLEMLERIDIKKATEAL INTEGER)
RETURNS (
    AY INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BOYAIRSALIYEDETAYID INTEGER,
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    ILAVEISLEMLERIDSTR VARCHAR(255),
    ILAVEISLEMLERADISTR VARCHAR(255),
    MIKTAR NUMERIC(12,3),
    TUTAR DOUBLE PRECISION)
AS
declare variable FaturaTarihi       TimeStamp;
declare variable BoyaSiparisDetayID integer;
declare variable IlaveIslemlerId    integer;
declare variable IlaveIslemlerAdi   varchar(30);
begin
  For Select BoyaFatura.FaturaTarihi, BoyaFatura.FirmaId, Firma.Adi, BoyaFaturaDetay.BoyaIrsaliyeDetayId, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi,
             PartiStok.BoyaSiparisDetayID,
             Max(BoyaFaturaDetay.Miktar), Sum(BoyaFaturaDetay.Miktar * BoyaFaturaDetay.BirimFiyat) from BoyaFatura
  left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaFaturaId = BoyaFatura.BoyaFaturaId
  left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
  left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
  left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  left join Firma on Firma.FirmaId = BoyaFatura.FirmaId
  where Boyafatura.FaturaTarihi >= :ABaslangicTarihi and Boyafatura.FaturaTarihi <= :ABitisTarihi and
        (Boyafatura.FirmaId = :AFirmaId or :AFirmaId is null) and
        (Parti.Fason = :AFason or (:AFason = 0 and BoyaIrsaliyeDetay.PartiKayitId is null) or :AFason is null) and
        BoyaFaturaDetay.BoyaFaturaDetayID is not null
  group by BoyaFatura.FaturaTarihi, BoyaFatura.FirmaId, Firma.Adi, BoyaFaturaDetay.BoyaIrsaliyeDetayId, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi,
           PartiStok.BoyaSiparisDetayID
  into :FaturaTarihi, :FirmaId, :FirmaAdi, :BoyaIrsaliyeDetayId, :SiparisProsesId, :SiparisProsesAdi, :BoyaSiparisDetayID, :Miktar, :Tutar do
  begin
    if (SiparisProsesAdi is null) then SiparisProsesAdi = '';

    IlaveIslemlerIdStr  = '';
    IlaveIslemlerAdiStr = '';
    Ay = f_Month(FaturaTarihi);

    Tutar = Yuvarla(Tutar, 100);

    if (AIlaveIslemleriDikkateAl = 1 and BoyaSiparisDetayId is not null) then
    begin
      For Select SiparisIlaveIslemler.IlaveIslemlerId, IlaveIslemler.Adi from SiparisIlaveIslemler
      left join IlaveIslemler ON IlaveIslemler.IlaveIslemlerId = SiparisIlaveIslemler.IlaveIslemlerId
      where SiparisIlaveIslemler.BoyaSiparisDetayId = :BoyaSiparisDetayId
      group by SiparisIlaveIslemler.IlaveIslemlerId, IlaveIslemler.Adi
      order by SiparisIlaveIslemler.IlaveIslemlerId
      into :IlaveIslemlerId, :IlaveIslemlerAdi do
      begin
        if (Char_Length(IlaveIslemlerIdStr) + Char_Length(EgemenIntegertoStr(IlaveIslemlerId)) < 250) then
        begin
          if (IlaveIslemlerIdStr <> '' ) then IlaveIslemlerIdStr = IlaveIslemlerIdStr || '+';
          IlaveIslemlerIdStr = IlaveIslemlerIdStr || IlaveIslemlerId;
        end

        if (Char_Length(IlaveIslemlerAdiStr) + Char_Length(IlaveIslemlerAdi) < 250) then
        begin
          if (IlaveIslemlerAdiStr <> '' ) then IlaveIslemlerAdiStr = IlaveIslemlerAdiStr || '+';
          IlaveIslemlerAdiStr = IlaveIslemlerAdiStr || IlaveIslemlerAdi;
        end
      end
    end
    suspend;
  end
end
