CREATE PROCEDURE MC_BOYASIPARISRAPORU_3(
    ABOYASIPARISDETAYID INTEGER,
    AFIRMAID VARCHAR(20),
    AKULLANICIID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISNO VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    ISLEMTIPIADI VARCHAR(20),
    ACILIYETDERECESIADI VARCHAR(20),
    TERMINTARIHI TIMESTAMP)
AS
declare variable Okey integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  for select DISTINCT PartiStok.BoyaSiparisDetayId, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.ReceteId,
             Recete.Adi, IslemGrubu.Adi, IslemTipi.Adi, AciliyetDerecesi.Adi, Parti.TerminTarihi,
             LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
    where PartiStok.BoyaSiparisDetayID = :ABoyaSiparisDetayId and Parti.FirmaId = :AFirmaId and PartiStok.PartiStokIptalKayitID is null
    into :BoyaSiparisDetayId, :PartiKayitId, :PartiYil, :PartiId, :AcilisTarihi, :ReceteId, :ReceteAdi, :IslemGrubuAdi, :IslemTipiAdi,
         :AciliyetDerecesiAdi, :TerminTarihi, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisNo do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    suspend;
  end
end
