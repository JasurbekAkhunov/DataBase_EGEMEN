    DEGISIMTARIHI TIMESTAMP,

    URETIMECIKISDOKUMAMAKINAID VARCHAR(100),
    URETIMECIKISURETIMPARTINO VARCHAR(300),

          UretimeCikisDokumaMakinaId = UretimeCikisDokumaMakinaId || ' ' || FDokumaMakinaId;

          UretimeCikisUretimPartiNo = FUretimPartiNo;

          UretimeCikisUretimPartiNo  = UretimeCikisUretimPartiNo || ' ' || FUretimPartiNo;

    left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId

         :Kilo, :Metre, :BrutKilo, :BrutMetre, :FatEdilecekHamStokBirimID, :FatEdilecekHamStokBirimStr, :HamID, :HamAdi, :UretimPartiNo, :DokumaPlanlamaId, :FOrmeRafId,

        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      suspend;
    end

  begin
    For Select HasilIsEmri.HasilIsEmriKayitID, StokHareket.StokHareketId, StokHareket.HareketTipi, StokHareket.FirmaId, SHFirma.Adi,
               OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, StokHareket.DokumaMakinaId,
               StokHareket.Miktar, StokHareket.BrutMiktar, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, StokHareket.BoyaSiparisDetayId,
               StokHareket.OrmeSiparisDetayId, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, StokHareket.HamIade,
               Ik.Adi, StokHareket.InsertTarihi, Sk.Adi, StokHareket.DegisimTarihi, StokHareket.MusteriPartiNo, StokHareket.Lot, StokHareket.AtkiLotu,
               StokHareket.RenkID, Renk.Adi, StokHareket.DepoID, StokHareket.LeventNo, StokHareket.TelAdedi, StokHareket.BitenBobin, StokHareket.SarimHataliBobinSayisi, StokHareket.UretimTarihi,
               StokHareket.Viscosity, StokHareket.Refacto, StokHareket.DepoPartiNo,
               StokHareket.Kilo, StokHareket.FatEdilecekHamStokBirimID, HamStokBirim.BirimID ,StokHareket.HamID, Ham.Adi, UretimIslemTipi.UretimIslemTipiId, UretimIslemTipi.Adi,
               StokHareket.IcCevreMM, StokHareket.DisCevreMM, StokHareket.TurSayisi from HasilIsEmri
    left join StokHareket on StokHareket.HasilIsEmriKayitId = HasilIsEmri.HasilIsEmriKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
    left join Firma SHFirma on SHFirma.Firmaid = StokHareket.Firmaid
    left join Kullanici IK on IK.KullaniciId = StokHareket.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId = StokHareket.KullaniciId
    left join Renk on Renk.RenkId = StokHareket.RenkId
    left join Ham on Ham.HamId = StokHareket.HamId
    Left join HamStokBirim on HamStokBirim.HamStokBirimId = StokHareket.FatEdilecekHamStokBirimID
    left join UretimIslemTipi on UretimIslemTipi.UretimIslemTipiId = StokHareket.UretimIslemTipiId
    where StokHareket.HasilIsEmriKayitId is not null and
         (HasilIsEmri.HasilIsEmriKayitId = :AIsEmriKayitID or :AIsEmriKayitID is Null) and
         ((HasilIsEmri.Tarih >= :ABaslangicTarihi and HasilIsEmri.Tarih <= :ABitisTarihi) or :AIsEmriKayitID is Not Null or
         (:ABaslangicTarihi is null and :ABitisTarihi is null and :AIsEmriKayitID is null)) and
         (HasilIsEmri.Kapali = :AKapaliDurumu or :AKapaliDurumu is Null) and
         (HasilIsEmri.FirmaID = :AFirmaID or :AFirmaID is Null)
    Order By StokHareket.Tarih
    into :HasilIsEmriKayitID, :StokHareketId, :HareketTipi, :FirmaId, :FirmaAdi, :Fisno, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi,
         :DokumaMakinaId, :Miktar, :BrutMiktar,  :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
         :OrmeSiparisDetayId, :MusteriSiparisNo, :ReferansSiparisNo, :HamIade,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :MusteriPartiNo, :Lot, :AtkiLotu, :RenkID, :RenkAdi,
         :DepoID, :LeventNo, :TelAdedi, :BitenBobin, :SarimHataliBobinSayisi, :UretimTarihi, :Viscosity, :Refacto, :DepoPartiNo,
         :Kilo, :FatEdilecekHamStokBirimID, :FatEdilecekHamStokBirimStr, :HamID, :HamAdi, :UretimIslemTipiId, :UretimIslemTipiAdi,
         :IcCevreMM, :DisCevreMM, :TurSayisi do
    begin
      Select HareketTipiStr from OrmeStokHareketTipiStrGetir(:HareketTipi, :HamIade, 0, Null) into :HareketTipiStr;

      BoyaSiparisIdStr = '';
      if (BoyaSiparisYil is not null) then
        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      suspend;
    end
  end
