CREATE PROCEDURE TAMIRINSARIMACIKISINIKAYDET(
    ATAMIRGIRISKAYITID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable HamId integer;
declare variable IplikDepoId    varchar(10);
declare variable IplikLotId      integer;
declare variable IplikIrsaliyeId integer;
declare variable IplikIrsaliyeDetayId integer;
declare variable Lot                  varchar(30);
declare variable MelanjKodu           varchar(30);
declare variable FlamKodu             varchar(20);
declare variable RenkId               integer;
declare variable KapAdedi             integer;
declare variable Miktar               numeric(12,3);
declare variable AmbalajId            integer;
declare variable KS integer;
declare variable shtSarimaCikis       integer;
declare variable SeriNo               varchar(2);
declare variable IrsaliyeNo           IrsaliyeNoSize;
declare variable SistemDovizId        varchar(5);
begin
  Select SistemDovizId from Sistem where sistemId = 1 into :SistemDovizId;

  select shtSarimaCikis from IplikStokHareketTipiGetir into :shtSarimaCikis;

  Select Max(IplikLot.IplikLotId) from IplikLot where IplikLot.TamirGirisKayitId = :ATamirGirisKayitId into :IplikLotId;

  Select TamirGiris.HamId, IplikSatisIade.IplikDepoId, IplikSatisIade.EskiLot, IplikSatisIade.MelanjKodu, IplikSatisIade.FlamKodu,
         IplikSatisIade.RenkId, IplikSatisIade.AmbalajId, IplikSatisIade.KapAdedi, TamirGiris.Miktar - TamirGiris.SarimaCikanMiktar from TamirGiris
    left join IplikSatisIade on IplikSatisIade.IplikSatisIadeID = TamirGiris.IplikSatisIadeId
    where TamirGiris.TamirGirisKayitId = :ATamirGirisKayitId
    into :HamId, :IplikDepoId, :Lot, :MelanjKodu, :FlamKodu, :RenkId, :AmbalajId, :KapAdedi, :Miktar;

  if (HamId is null) then exception KayitHatasi ' Stok Bulunamadı, Status-1';

  if (Miktar = 0) then exception KayitHatasi ' Stok Daha Önce Cıkılmıs, Status-2';

  Select Max(IplikIrsaliyeId) from IplikIrsaliye
    where IplikIrsaliye.IrsaliyeTarihi = Current_date and IplikIrsaliye.HareketTipi = :shtSarimaCikis
    into :IplikIrsaliyeId;

  if (IplikIrsaliyeId is null) then
  begin
    Select IplikUretimeCikisSeriNo from Kullanici where KullaniciId = :AKullaniciId into :SeriNo;

    if (SeriNo is null) then SeriNo = '';

    Select IrsaliyeNo from IrsaliyeNumaratorAl(:SeriNo) into :IrsaliyeNo;

    IplikIrsaliyeId = gen_id(IplikIrsaliye_Id_Gen, 1);

    insert into IplikIrsaliye(IplikIrsaliyeId, HareketTipi, SeriNo, IrsaliyeNo, IrsaliyeTarihi, DovizId, InsertKullaniciId, KullaniciId)
             values(:IplikIrsaliyeId, :shtSarimaCikis, :SeriNo, :IrsaliyeNo, Current_date, :SistemDovizId, :AKullaniciId, :AKullaniciId);
  end

  IplikIrsaliyeDetayId = gen_Id(IplikIrsaliyeDetay_Id_gen, 1);
  insert into IplikIrsaliyeDetay(IplikIrsaliyeId, IplikIrsaliyeDetayId, HareketTipi, IplikDepoId, HamId, AmbalajId, RenkId, Lot, MelanjKodu, FlamKodu, IplikLotId, KapAdedi, Miktar, NetMiktar, InsertKullaniciId, KullaniciId)
              values(:IplikIrsaliyeId, :IplikIrsaliyeDetayId, :shtSarimaCikis, :IplikDepoId, :HamId, :AmbalajId, :RenkId, :Lot, :MelanjKodu, :FlamKodu, :IplikLotId, :KapAdedi, :Miktar, :Miktar, :AKullaniciID, :AKullaniciID);

end
