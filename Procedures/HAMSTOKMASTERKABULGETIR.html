CREATE PROCEDURE HAMSTOKMASTERKABULGETIR(
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    HAMSTOKMASTERKABULID INTEGER,
    DEPARTMANID INTEGER,
    DEPARTMANADI VARCHAR(30),
    KABUL INTEGER,
    KABULMIKTAR NUMERIC(12,3),
    RET INTEGER,
    RETMIKTAR NUMERIC(12,3),
    SARTLIKABUL INTEGER,
    SARTLIKABULMIKTAR NUMERIC(12,3),
    ZORUNLUKABUL INTEGER,
    ZORUNLUKABULMIKTAR NUMERIC(12,3),
    KONTROLDURUMU INTEGER,
    KONTROLKULLANICIID INTEGER,
    KONTROLKULLANICIADI VARCHAR(40),
    KONTROLTARIHI TIMESTAMP,
    KONTROLNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KABULSTR VARCHAR(5),
    RETSTR VARCHAR(3),
    SARTLIKABULSTR VARCHAR(13),
    ZORUNLUKABULSTR VARCHAR(18))
AS
begin
  for select HamStokMasterKabul.HamStokMasterKabulId, HamStokMasterKabul.DepartmanId, Departman.Adi,
             HamStokMasterKabul.Kabul, HamStokMasterKabul.KabulMiktar, HamStokMasterKabul.Ret, HamStokMasterKabul.RetMiktar,
             HamStokMasterKabul.SartliKabul, HamStokMasterKabul.SartliKabulMiktar, HamStokMasterKabul.ZorunluKabul, HamStokMasterKabul.ZorunluKabulMiktar,
             HamStokMasterKabul.KontrolDurumu, KK.Adi, HamStokMasterKabul.KontrolTarihi, HamStokMasterKabul.KontrolNotu from HamStokMasterKabul
  left join Departman on Departman.DepartmanId = HamStokMasterKabul.DepartmanId
  Left Join Kullanici KK on KK.KullaniciId = HamStokMasterKabul.KontrolKullaniciId
  where HamStokMasterKabul.HamStokMasterId = :AHamStokMasterId
  order by HamStokMasterKabul.SonKarar
  into :HamStokMasterKabulId, :DepartmanId, :DepartmanAdi, :Kabul, :KabulMiktar, :Ret, :RetMiktar, :SartliKabul, :SartliKabulMiktar,
       :ZorunluKabul, :ZorunluKabulMiktar, :KontrolDurumu, KontrolKullaniciAdi, :KontrolTarihi, :KontrolNotu do
  begin
    if (Kabul = 1) then KabulStr = 'Kabul';
    else KabulStr = '';

    if (Ret = 1) then RetStr = 'Ret';
    else RetStr = '';

    if (SartliKabul = 1) then SartliKabulStr = 'Şartlı Kabul';
    else SartliKabulStr = '';

    if (ZorunluKabul = 1) then ZorunluKabulStr = 'Zorunlu Kabul';
    else ZorunluKabulStr = '';

    if (KabulMiktar = 0) then KabulMiktar = Null;
    if (RetMiktar = 0) then RetMiktar = Null;
    if (SartliKabulMiktar = 0) then SartliKabulMiktar = Null;
    if (ZorunluKabulMiktar = 0) then ZorunluKabulMiktar = Null;

    suspend;
  end
end
