CREATE PROCEDURE ORJINALFIELDFORMATLARIYUKLE
AS
declare variable ModulFormatId         integer;
declare variable Basamak               integer;
declare variable Ondalik               integer;
declare variable Yuzde                 integer;

declare variable TableAdi           varchar(40);
declare variable FieldAdi           varchar(40);
begin
  delete from FieldFormat;
  delete from ModulFormat;

  For Select ModulFormatId, Basamak, Ondalik, Yuzde from ModulFormatOrj into :ModulFormatId, :Basamak, :Ondalik, :Yuzde do
  begin
   insert into MODULFORMAT(ModulFormatId, Basamak, Ondalik, Yuzde) values(:ModulFormatId, :Basamak, :Ondalik, :Yuzde);
  end

  For Select TableAdi, FieldAdi, ModulFormatId from FieldFormatOrj into :TableAdi, :FieldAdi, :ModulFormatId do
  begin
    INSERT INTO FIELDFORMAT(TABLEADI, FIELDADI, MODULFORMATID) VALUES(:TABLEADI, :FIELDADI, :MODULFORMATID);
  end
end
