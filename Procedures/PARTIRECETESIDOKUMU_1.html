CREATE PROCEDURE PARTIRECETESIDOKUMU_1(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER)
RETURNS (
    ISEMRIYIL INTEGER,
    ISEMRIID INTEGER,
    DOZAJLAMAISEMRIID INTEGER,
    PARTIKAYITID INTEGER,
    ARGENO VARCHAR(30),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAKISAADI VARCHAR(10),
    PARTIASAMALARIID INTEGER,
    RECETETARIHI TIMESTAMP,
    GIRISKILO NUMERIC(12,3),
    GIRISMETRE NUMERIC(12,3),
    GIRISMIKTAR NUMERIC(12,3),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    TERMINTARIHI TIMESTAMP,
    PARTITAMIRID INTEGER,
    BANYOCARPANI DOUBLE PRECISION,
    BANYOCARPANI2 DOUBLE PRECISION,
    BANYOCARPANI3 DOUBLE PRECISION,
    BANYOCARPANI4 DOUBLE PRECISION,
    FULARBASINCI DOUBLE PRECISION,
    KOYULTMA DOUBLE PRECISION,
    DURULAMASUYU INTEGER,
    BOYASUYU INTEGER,
    BOYAMASEVIYESI INTEGER,
    TUR DOUBLE PRECISION,
    POMPADEGERI INTEGER,
    DUZE INTEGER,
    GRUPPROSESNO VARCHAR(10),
    GRUPPROSESADI VARCHAR(80),
    PLANLANANMAKINAID VARCHAR(10),
    PLANLANANOTOMASYONMAKINANO VARCHAR(10),
    PARTINOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PROSESNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    DOZAJLAMAYAGONDERILDI INTEGER,
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    STOKTANDUSULDU INTEGER,
    STOKTANDUSMETARIHI TIMESTAMP,
    KUMASTAKISU DOUBLE PRECISION,
    ILAVEMIKTAR DOUBLE PRECISION,
    GOZUZUNLUGU DOUBLE PRECISION,
    ISLAKLIKORANI DOUBLE PRECISION,
    ICTAMBURHIZI DOUBLE PRECISION,
    YIKAMAKATSAYISI DOUBLE PRECISION,
    DEVIRLICALISMAHIZI DOUBLE PRECISION,
    MAKINAICIYIKAMA DOUBLE PRECISION,
    KULEHIZI DOUBLE PRECISION,
    HAMSTOKFLOTTECARPANI DOUBLE PRECISION,
    IDSURE DOUBLE PRECISION,
    DISURE DOUBLE PRECISION,
    POMPAHIZID DOUBLE PRECISION,
    POMPAHIZDI DOUBLE PRECISION,
    FARKBASINCID DOUBLE PRECISION,
    FARKBASINCDI DOUBLE PRECISION,
    AKISID DOUBLE PRECISION,
    AKISDI DOUBLE PRECISION,
    TUZCARPANI DOUBLE PRECISION,
    FULARHAVABASINCI DOUBLE PRECISION,
    BOYAMETODU DOUBLE PRECISION,
    FULARLITRE DOUBLE PRECISION,
    TUPSAYISI INTEGER,
    FANGUCU INTEGER,
    TORK INTEGER,
    SIPARISGRUP3ID VARCHAR(4),
    SIPARISGRUP3ADI VARCHAR(30),
    SIPARISGRUP4ID VARCHAR(4),
    SIPARISGRUP4ADI VARCHAR(30),
    FASON SMALLINT,
    RECETEYAZMASAYISI INTEGER,
    ISLEMGRUBUADI VARCHAR(20))
AS
declare variable FReceteIsEmriYil integer;
declare variable FReceteIsEmriId integer;
declare variable FSira integer;
declare variable RYDozajlamaIsEmriNoOlustur integer;
declare variable EgemenFirmaID integer;
begin
  Select RYDozajlamaIsEmriNoOlustur, EgemenFirmaID from Sistem where sistemid = 1 into :RYDozajlamaIsEmriNoOlustur, :EgemenFirmaID;

  PartiTamirId = Null;
  select ReceteIsEmriDetay.ReceteIsEmriYil, ReceteIsEmriDetay.ReceteIsEmriId, ReceteIsEmri.BanyoCarpani,
         ReceteIsEmri.BanyoCarpani2, ReceteIsEmri.BanyoCarpani3,  ReceteIsEmri.BanyoCarpani4, ReceteIsEmri.HamStokFlotteCarpani, ReceteIsEmri.FularBasinci,
         ReceteIsEmri.Koyultma, ReceteIsEmri.DurulamaSuyu, ReceteIsEmri.BoyaSuyu, ReceteIsEmri.Tur, ReceteIsEmri.PompaDegeri, ReceteIsEmri.Duze, ReceteIsEmri.BoyamaSeviyesi,
         ReceteIsEmri.FanGucu, ReceteIsEmri.TupSayisi, ReceteIsEmri.TuzCarpani, ReceteIsEmri.KuleHizi, ReceteIsEmri.IslaklikOrani  from ReceteIsEmriDetay
    left join ReceteIsEmri on ReceteIsEmri.ReceteIsEmriYil = ReceteIsEmriDetay.ReceteIsEmriYil and ReceteIsEmri.ReceteIsEmriId = ReceteIsEmriDetay.ReceteIsEmriId
    where ReceteIsEmriDetay.PartiKayitId = :APartiKayitId and ReceteIsEmriDetay.PartiAsamalariId = :APartiAsamalariId and ReceteIsEmri.ReceteId = :AReceteId
    into :FReceteIsEmriYil, :FReceteIsEmriId, :BanyoCarpani, :BanyoCarpani2, :BanyoCarpani3, :BanyoCarpani4, :HamStokFlotteCarpani,
         :FularBasinci, :Koyultma, :DurulamaSuyu, :BoyaSuyu, :Tur, :PompaDegeri, Duze, :BoyamaSeviyesi,
         :FanGucu, :TupSayisi, :TuzCarpani, :KuleHizi, :IslaklikOrani;

  if (RYDozajlamaIsEmriNoOlustur = 1) then
    Select DozajlamaIsEmriId, Gonderildi from DozajlamaIsEmriGetir(:APartiAsamalariId) into :DozajlamaIsEmriId, :DozajlamayaGonderildi;

  if (FReceteIsEmriId is null) then
  begin
    if (RYDozajlamaIsEmriNoOlustur = 0) then
      Select DozajlamaIsEmriId, Gonderildi from DozajlamaIsEmri where DozajlamaIsEmri.PartiAsamalariId = :APartiAsamalariId into :DozajlamaIsEmriId,:DozajlamayaGonderildi;

    select BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, FularBasinci, Koyultma, DurulamaSuyu, BoyaSuyu, Tur, PompaDegeri, Duze, Sira, PartiTamirId, PartiAsamalari.GrupProsesNo, MakinaPlanlamaGrupProses.Adi, BoyamaSeviyesi, ProsesNotu, StoktanDusuldu,
           KumastakiSu, IlaveMiktar, GozUzunlugu, IslaklikOrani, IcTamburHizi, YikamaKatSayisi, DevirliCalismaHizi, MakinaIciYikama, KuleHizi, HamStokFlotteCarpani, TuzCarpani,
           PartiAsamalari.FularBasinci, PartiAsamalari.BoyaMetodu, PartiAsamalari.FularLitre, PartiAsamalari.FanGucu, PartiAsamalari.TupSayisi from PartiAsamalari
          left join MakinaPlanlamaGrupProses on MakinaPlanlamaGrupProses.MakinaPlanlamaGrupId = PartiAsamalari.MakinaPlanlamaGrupId and MakinaPlanlamaGrupProses.GrupProsesNo = PartiAsamalari.GrupProsesNo
    where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamalari.PartiAsamalariId = :APartiAsamalariId
      into :BanyoCarpani, :BanyoCarpani2, :BanyoCarpani3, :BanyoCarpani4, :FularBasinci, :Koyultma, :DurulamaSuyu, :BoyaSuyu, :Tur, :PompaDegeri, :Duze,  :FSira, :PartiTamirId, :GrupProsesNo, :GrupProsesAdi, :BoyamaSeviyesi, :ProsesNotu, :StoktanDusuldu,
           :KumastakiSu, :IlaveMiktar, :GozUzunlugu, :IslaklikOrani, :IcTamburHizi, :YikamaKatSayisi, :DevirliCalismaHizi, :MakinaIciYikama, :KuleHizi, :HamStokFlotteCarpani, :TuzCarpani,
           :FularBasinci, :BoyaMetodu, :FularLitre, :FanGucu, :TupSayisi;

    if (EgemenFirmaid<>132) then
    begin
      if (FSira is null) then exception ReceteAsamalardaBulunamadi;
    end

    PartiAsamalariId = APartiAsamalariId;

    Select Max(UretimPlanlama.MakinaId) from BolumPlanlama
      left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
      left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
      where BolumPlanlama.PartiAsamalariId = :APartiAsamalariId
      into :PlanlananMakinaId;

    if (PlanlananMakinaId is null) then
    begin
      Select Max(PartiAsamaStokDetay.MakinaId) from PartiAsamalari
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
        where PartiAsamalari.PartiAsamalariID = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamaStokDetay.GirisTarihi is not null
        into :PlanlananMakinaId;
    end

    Select IDSure, DISure, PompaHizID, PompaHizDI, FarkBasincID, FarkBasincDI, AkisID, AkisDI, Tork, OtomasyonMakinaNo from Makina Where MakinaID = :PlanlananMakinaId
      into :IDSure, :DISure, :PompaHizID, :PompaHizDI, :FarkBasincID, :FarkBasincDI, :AkisID, :AkisDI, :Tork, :PlanlananOtomasyonMakinaNo;

    select Parti.PartiKayitId, Parti.ArgeNo, Parti.PartiYil, Parti.PartiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
           Parti.FirmaId, Firma.Adi, Parti.Notu, Parti.TerminTarihi, Parti.ReceteTarihi,
           BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, Parti.IslemTipiId, IslemTipi.Adi, Firma.KisaAdi,
           BoyaSiparis.SiparisGrup3Id, SiparisGrup3.Adi, BoyaSiparis.SiparisGrup4Id, SiparisGrup4.Adi, BoyaSiparis.Fason, Parti.ReceteYazmaSayisi,
		   IslemGrubu.Adi from Parti

      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      left join Firma on Firma.FirmaId = Parti.FirmaId
      left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join SiparisGrup SiparisGrup3 on SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
      left join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
	  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
      where Parti.PartiKayitId = :APartiKayitId
      into :PartiKayitId, :ArgeNo, :PartiYil, :PartiId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :PartiNotu, :TerminTarihi, :ReceteTarihi,
           :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :IslemTipiId, :IslemTipiAdi, :FirmaKisaAdi,
           :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi, :Fason, :ReceteYazmaSayisi, :IslemGrubuAdi;

    Select Sum(PartiAsamaStok.GirisKilo), Sum(PartiAsamaStok.GirisMetre), Sum(PartiAsamaStok.GirisMiktar) from PartiAsamaStok
      where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
      into :GirisKilo, :GirisMetre, :GirisMiktar;

    if (EgemenFirmaID = 99) then /* megam */
    begin
       select Recete.GrupProsesNo from Recete where ReceteId = :AReceteId into :GrupProsesNo;
    end

    StoktanDusmeTarihi = Null;

    if (StoktanDusuldu = 1) then
    begin
      Select Max(StokDetay.Tarih) from StokDetay
        where StokDetay.PartiAsamalariID = :PartiAsamalariId and StokDetay.PartiAsamaIlaveSarfID is null
        into :StoktanDusmeTarihi;
    end

    suspend;
    exit;
  end
  else
  begin
    IsEmriYil = :FReceteIsEmriYil;
    IsEmriId = :FReceteIsEmriId;

    Select Max(UretimPlanlama.MakinaId) from BolumPlanlama
      left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
      left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
      where BolumPlanlama.PartiAsamalariId = :APartiAsamalariId
      into :PlanlananMakinaId;

    if (PlanlananMakinaId is null) then
    begin
      Select Max(PartiAsamaStokDetay.MakinaId) from PartiAsamalari
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
        where PartiAsamalari.PartiAsamalariID = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamaStokDetay.GirisTarihi is not null
        into :PlanlananMakinaId;
    end

    Select IDSure, DISure, PompaHizID, PompaHizDI, FarkBasincID, FarkBasincDI, AkisID, AkisDI, Tork, OtomasyonMakinaNo from Makina Where MakinaID = :PlanlananMakinaId
      into :IDSure, :DISure, :PompaHizID, :PompaHizDI, :FarkBasincID, :FarkBasincDI, :AkisID, :AkisDI, :Tork, :PlanlananOtomasyonMakinaNo;

    if (RYDozajlamaIsEmriNoOlustur = 0) then
      Select DozajlamaIsEmriId, Gonderildi from DozajlamaIsEmri where DozajlamaIsEmri.ReceteIsEmriYil = :FReceteIsEmriYil and DozajlamaIsEmri.ReceteIsEmriId = :FReceteIsEmriId into :DozajlamaIsEmriId,:DozajlamayaGonderildi;

    for select Parti.PartiKayitId, Parti.ArgeNo, Parti.PartiYil, Parti.PartiId, ReceteIsEmriDetay.PartiAsamalariId, PartiAsamalari.PartiTamirId,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Parti.FirmaId, Firma.Adi, Parti.Notu, PartiAsamalari.GrupProsesNo, MakinaPlanlamaGrupProses.Adi, Parti.TerminTarihi,
               Parti.ReceteTarihi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, PartiAsamalari.ProsesNotu,
               Parti.IslemTipiId, IslemTipi.Adi, PartiAsamalari.StoktanDusuldu, Firma.KisaAdi, BoyaSiparis.SiparisGrup3Id, SiparisGrup3.Adi, BoyaSiparis.SiparisGrup4Id, SiparisGrup4.Adi, BoyaSiparis.Fason from ReceteIsEmriDetay
      left join Parti on Parti.PartiKayitId = ReceteIsEmriDetay.PartiKayitId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      left join Firma on Firma.FirmaId = Parti.FirmaId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = ReceteIsEmriDetay.PartiAsamalariId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join SiparisGrup SiparisGrup3 on SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
      left join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
      left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
      left join MakinaPlanlamaGrupProses on MakinaPlanlamaGrupProses.MakinaPlanlamaGrupId = PartiAsamalari.MakinaPlanlamaGrupId and MakinaPlanlamaGrupProses.GrupProsesNo = PartiAsamalari.GrupProsesNo
      where ReceteIsEmriDetay.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :FReceteIsEmriId
      into :PartiKayitId, :ArgeNo, :PartiYil, :PartiId, :PartiAsamalariId, :PartiTamirId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi,
           :PartiNotu, :GrupProsesNo, :GrupProsesAdi, :TerminTarihi, :ReceteTarihi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :ProsesNotu,
           :IslemTipiId, :IslemTipiAdi, :StoktanDusuldu, :FirmaKisaAdi, :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi, :Fason do
    begin
      Select Sum(PartiAsamaStok.GirisKilo), Sum(PartiAsamaStok.GirisMetre), Sum(PartiAsamaStok.GirisMiktar) from PartiAsamaStok
        where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
        into :GirisKilo, :GirisMetre, :GirisMiktar;

      StoktanDusmeTarihi = Null;

      if (StoktanDusuldu = 1) then
      begin
        Select Max(StokDetay.Tarih) from StokDetay
          where StokDetay.PartiAsamalariID = :PartiAsamalariId and StokDetay.PartiAsamaIlaveSarfID is null
          into :StoktanDusmeTarihi;
      end

      if (EgemenFirmaID = 99) then /* megam */
      begin
         select Recete.GrupProsesNo from Recete where ReceteId = :AReceteId into :GrupProsesNo;
      end

      suspend;
    end
  end
end
