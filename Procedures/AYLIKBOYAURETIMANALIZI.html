CREATE PROCEDURE AYLIKBOYAURETIMANALIZI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMTIPIID INTEGER)
RETURNS (
    URETIMTOPLAMI NUMERIC(12,3),
    ESDEGERURETIMTOPLAMI NUMERIC(12,3),
    ILAVEPROSESTOPLAMI NUMERIC(12,3),
    DISTAMIRTOPLAMI NUMERIC(12,3),
    ICTAMIRTOPLAMI NUMERIC(12,3),
    ISLEMSIZIADETOPLAMI NUMERIC(12,3),
    UCRETLIICTAMIRTOPLAMI NUMERIC(12,3),
    UCRETLIDISTAMIRTOPLAMI NUMERIC(12,3))
AS
declare variable IslemGrubuId   integer;
declare variable IslemTipiId    integer;
declare variable IlaveProses    integer;
declare variable KatSayi        Double Precision;
declare variable IcTamir        integer;
declare variable Kilo           Numeric(12,3);
declare variable Ucretli        integer;
declare variable FaturaKesilecek integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  UretimToplami           = 0;
  EsdegerUretimToplami    = 0;
  IlaveProsesToplami      = 0;
  DisTamirToplami         = 0;
  IcTamirToplami          = 0;
  IslemsizIadeToplami     = 0;
  UcretliIcTamirToplami   = 0;
  UcretliDisTamirToplami  = 0;

  for select Parti.IslemTipiId, Parti.IslemGrubuId, IslemGrubu.Katsayi, (PartiAsamalari.PartiTamirId - PartiAsamalari.PartiTamirId + 1),
             PartiAsamalari.IlaveProses, Sum(PartiAsamaStokDetay.Kilo), TamirCozumSekli.Ucretli, PartiStok.FaturaKesilecek from PartiAsamaStokDetay
    left join PartiStok On PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Ham On Ham.HamId = PartiStok.HamId
    left join PartiAsamalari On PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join PartiTamir On PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    left join TamirCozumSekli On TamirCozumSekli.TamirCozumSekliId = PartiTamir.TamirCozumSekliId
    where PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.CikisTarihi <= :ABitisTarihi and Makina.Kazan = 1 and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null)
    group by Parti.IslemTipiId, Parti.IslemGrubuId, IslemGrubu.Katsayi, PartiAsamalari.PartiTamirId, PartiAsamalari.IlaveProses,
              TamirCozumSekli.Ucretli, PartiStok.FaturaKesilecek
    into :IslemTipiId, :IslemGrubuId, :KatSayi, :IcTamir, :IlaveProses, :Kilo, :Ucretli, :FaturaKesilecek do
  begin
    UretimToplami = UretimToplami + Kilo;
    EsdegerUretimToplami = EsdegerUretimToplami + Kilo * KatSayi;

    if (IlaveProses = 1) then IlaveProsesToplami = IlaveProsesToplami + Kilo;
    else if (IcTamir = 1) then
    begin
      if (Ucretli = 1) then UcretliIcTamirToplami = UcretliIcTamirToplami + Kilo;
      IcTamirToplami = IcTamirToplami + Kilo;
    end
    else if (IslemTipiId = 3) then
    begin
      if (FaturaKesilecek = 1) then UcretliDisTamirToplami = UcretliDisTamirToplami + Kilo;
      DisTamirToplami = DisTamirToplami + Kilo;
    end
  end
  suspend;
end
