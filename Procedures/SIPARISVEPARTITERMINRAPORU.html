CREATE PROCEDURE SIPARISVEPARTITERMINRAPORU(
    ABASLANGICTERMINTARIHI TIMESTAMP,
    ABITISTERMINTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASON INTEGER,
    ASIPARISONAYDURUMU INTEGER,
    AASAMAID INTEGER,
    AASAMABASLAMAMIS INTEGER,
    ABOYAHANEBOLUMID INTEGER,
    AHAMID INTEGER,
    APARTITIPIID INTEGER,
    AIMALATTIPIID INTEGER,
    ASIPARISKAPALIDURUMU INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ACILISTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    HAMADI VARCHAR(255),
    REFERANSKODU VARCHAR(30),
    PROSESID INTEGER,
    PROSESADI VARCHAR(255),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    RECETEKODU VARCHAR(20),
    RECETEADI VARCHAR(50),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    IPTALKILO NUMERIC(12,3),
    IPTALMETRE NUMERIC(12,3),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    SONYAPILANASAMAADI VARCHAR(40),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    SONRAKIASAMAADI VARCHAR(40),
    SONRAKIASAMAID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    TERMINNOTU VARCHAR(1024),
    SIPARISNOTU VARCHAR(1024),
    PLANLAMANOTU VARCHAR(1024),
    PARTIGECIKMETIPIADI VARCHAR(30),
    PARTIGECIKMENOTU VARCHAR(1024),
    BEKLEMEDE INTEGER,
    ISLEMTIPIID INTEGER,
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISTERMINTARIHI TIMESTAMP,
    SIPARISREVIZETERMINTARIHI TIMESTAMP,
    SIPARISTARIHI TIMESTAMP,
    SIPARISINSERTTARIHI TIMESTAMP,
    HAMSTOKTEMINTARIHI TIMESTAMP,
    IMALATTIPISTR VARCHAR(5),
    IMALATTIPIID INTEGER,
    KAPALI INTEGER,
    KAPALISTR VARCHAR(4),
    DESENID DESENKODUSIZE,
    DESENTASARIMNO VARCHAR(20),
    DESENVARYANTID INTEGER,
    DESENVARYANTADI ADI30,
    DESENSABLONENI INTEGER,
    DESENGRAVURBOYU INTEGER,
    PARTILENENKILOBASKI NUMERIC(12,3),
    PARTILENENMETREBASKI NUMERIC(12,3),
    IPTALKILOBASKI NUMERIC(12,3),
    IPTALMETREBASKI NUMERIC(12,3),
    DESENADI ADI50,
    VARYANTADI ADI30,
    BOYASIPARISBASKIID INTEGER,
    VARYANTACIKLAMA VARCHAR(255),
    PUSSTR VARCHAR(50),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    KONFEKSIYON INTEGER,
    KONFEKSIYONSTR VARCHAR(5),
    HAMSTOKMASTERIDSTR VARCHAR(100),
    LOT VARCHAR(150),
    YARIMAMULPARTINOSTR VARCHAR(200))
AS
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
declare variable AtisNoStr       varchar(20);
declare variable SiparisKilo     numeric(12,3);
declare variable SiparisMetre     numeric(12,3);
declare variable SonYapilanSira      integer;
declare variable SonrakiSira      integer;
declare variable FTerminNotu   blob;
declare variable FSiparisNotu blob;
declare variable FPlanlamaNotu blob;
declare variable FPartiGecikmeNotu blob;
declare variable BlobSize    integer;
declare variable Okey integer;
declare variable KayitSayisi integer;
declare variable FPartiYil integer;
declare variable FPartiId  integer;
declare variable HamTipiId integer;
declare variable PartilenenMiktardanIptaliDus EgeBoolean;
declare variable LabReceteId integer;
declare variable BaskiSiparisKilo  BoyaMiktarSize;
declare variable BaskiSiparisMetre BoyaMiktarSize;
declare variable BaskiPartilenenKilo  BoyaMiktarSize;
declare variable BaskiPartilenenMetre BoyaMiktarSize;
declare variable BaskiDetayIptalKilo  BoyaMiktarSize;
declare variable BaskiDetayIptalMetre BoyaMiktarSize;
declare variable FLot  varchar(150);
declare variable FHamStokMasterID  Integer;
declare variable Str  varchar(20);
declare variable YariMamulPartiYil integer;
declare variable YariMamulPartiId  integer;
begin
  select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;
  select PartilenenMiktardanIptaliDus from Sistem where Sistem.SistemId = 1 into :PartilenenMiktardanIptaliDus;

  PartilenenKilo    = 0;
  PartilenenMetre   = 0;
  IptalKilo    = 0;
  IptalMetre   = 0;

  for select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.BoyaSiparisKayitId, PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.OrmeSiparisDetayId,
       Parti.AcilisTarihi, Parti.Termintarihi, PartiStok.HamAdiToplami, Ham.ReferansKodu, PartiStok.ProsesId, Proses.Adi, Parti.ReceteId, Recete.Adi, Parti.FirmaId, Firma.Adi, BoyaSiparis.BoyaSiparisYil,
       BoyaSiparis.BoyaSiparisId, SiparisProses.SiparisProsesID, SiparisProses.Adi, PartiStok.Kilo, PartiStok.Metre, Parti.Fason, Asama.Adi, PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi,
       PartiAsamalari.Sira, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.MusteriRenkKoduVeAdi, BoyaSiparisDetay.TerminNotu, BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu, BoyaSiparisDetay.TerminTarihi,
       BoyaSiparisDetay.RevizeTerminTarihi, GecikmeTipi.Adi, PartiGecikme.Notu, Parti.IslemTipiId, BoyaSiparis.SiparisiAlanKullaniciId, PartiStok.HamTipiId, BoyaSiparis.SiparisTarihi, BoyaSiparisDetay.InsertTarihi,
       BoyaSiparisDetay.HamStokTeminTarihi, Parti.ImalatTipiId,BoyaSiparisDetay.Kapali, Desen.DesenId, Desen.TasarimNo, DesenVaryant.DesenVaryantId, DesenGravurBoyu.Olcu, DesenSablonEni.Olcu, DesenVaryant.VaryantAdi,
       BoyaSiparisDetay.PartilenenKiloBaski, BoyaSiparisDetay.PartilenenMetreBaski, BoyaSiparisDetay.IptalKiloBaski, BoyaSiparisDetay.IptalMetreBaski, Desen.Adi, Desenvaryant.Aciklama, BoyaSiparisDetay.Pus,
       BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, BoyaSiparisBaski.Konfeksiyon from PartiStok
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
    left join Proses on Proses.ProsesId = PartiStok.ProsesId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisBaskiId = PartiStok.BoyaSiparisBaskiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiStok.PartiAsamalariId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiStok.PartiAsamalariId and PartiAsamaStok.PartiSinifID = PartiStok.PartiSinifId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiGecikme on PartiGecikme.PartiKayitId = Parti.PartiKayitId and PartiGecikme.Bitti = 0
    left join GecikmeTipi on GecikmeTipi.GecikmeTipiId = PartiGecikme.GecikmeTipiId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
    left join DesenOlcu DesenGravurBoyu on DesenGravurBoyu.DesenolcuId = Desen.GravurBoyuDesenOlcuId
    left join DesenOlcu DesenSablonEni on DesenSablonEni.DesenolcuId = Desen.SablonEniDesenOlcuId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    where PartiStok.Durumu < 7 and PartiStok.PartiStokIptalKayitId is null and
         (BoyaSiparisDetay.Kapali = :ASiparisKapaliDurumu or :ASiparisKapaliDurumu is null) and
         (Parti.TerminTarihi >= :ABaslangicTerminTarihi and Parti.TerminTarihi <= :ABitisTerminTarihi) and
         (PartiStok.HamId = :AHamId or :AHamId is null) and
         (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null) and
         (Parti.Fason = :AFason or :AFason is null) and
         (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
         ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :APartiTipiId = 0) or
           (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :APartiTipiId = 1) or
           ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :APartiTipiId = 2) or
           (Ham.HamTipiId = :hstDokumaKumas and :APartiTipiId = 3) or :APartiTipiId is null)
    into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :OrmeSiparisDetayId, :AcilisTarihi, :Termintarihi,
         :HamAdi, :ReferansKodu,  :ProsesId, :ProsesAdi, :ReceteKodu, :ReceteAdi, :FirmaId, :FirmaAdi, :BoyaSiparisYil, :BoyaSiparisId,
         :SiparisProsesID, :SiparisProsesAdi, :Kilo, :Metre, :Fason, :SonYapilanAsamaAdi, :GirisTarihi, :CikisTarihi, :SonYapilanSira,
         :MusteriSiparisNo, MusteriRenkKoduVeAdi, :FTerminNotu, :FSiparisNotu, :FPlanlamaNotu, :SiparisTerminTarihi, :SiparisRevizeTerminTarihi,
         :PartiGecikmeTipiAdi, :FPartiGecikmeNotu, :IslemTipiId, :SiparisiAlanKullaniciId, :HamTipiId,
         :SiparisTarihi, :SiparisInsertTarihi, :HamStokTeminTarihi, :ImalatTipiId, :Kapali, :DesenId, :DesenTasarimNo, :DesenVaryantId,
         :DesenGravurBoyu, :DesenSablonEni, :DesenVaryantAdi, :PartilenenKiloBaski, :PartilenenMetreBaski, :IptalKiloBaski, :IptalMetreBaski, :DesenAdi,
         :VaryantAciklama, :PusStr, :EnIstenen, :GramajIstenen, :BirimFiyat,  :DovizId, :Konfeksiyon do
  begin
    Okey = 1;
    if (AAsamaId is not null) then
    begin   /* 0 baslamamıs, 1 baslamıs, null hepsi*/
     if (AAsamaBaslamamis = 0) then
      select count(*) from PartiAsamaStok
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamaID = :AAsamaId and PartiAsamaStok.AsamayaDahil = 1 and
              PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.CikisTarihi is null into :Okey;
     else if (AAsamaBaslamamis = 1) then
      select count(*) from PartiAsamaStok
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamaID = :AAsamaId and PartiAsamaStok.AsamayaDahil = 1 and
              PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is null into :Okey;
     else
        select count(*) from PartiAsamaStok
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamaID = :AAsamaId and PartiAsamaStok.AsamayaDahil = 1 and
              PartiAsamaStok.CikisTarihi is null into :Okey;
    end
    else if (ABoyahaneBolumId is not null) then
    begin
     if (AAsamaBaslamamis = 0) then
       select count(*) from BolumPlanlama
         left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
         where BolumPlanlama.PartiKayitId = :PartiKayitId and BolumPlanlama.PartiSinifId = :PartiSinifId and BolumPlanlama.BoyahaneBolumId = :ABoyahaneBolumId and
               PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.CikisTarihi is null into :Okey;
     else if (AAsamaBaslamamis = 1) then
       select count(*) from BolumPlanlama
         left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
         where BolumPlanlama.PartiKayitId = :PartiKayitId and BolumPlanlama.PartiSinifId = :PartiSinifId and BolumPlanlama.BoyahaneBolumId = :ABoyahaneBolumId and
               PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is null into :Okey;
     else
       select count(*) from BolumPlanlama
         left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
         where BolumPlanlama.PartiKayitId = :PartiKayitId and BolumPlanlama.PartiSinifId = :PartiSinifId and BolumPlanlama.BoyahaneBolumId = :ABoyahaneBolumId and
               PartiAsamaStok.CikisTarihi is null into :Okey;
    end

    Lot = '';
    HamStokMasterIDStr = '';
    YariMamulPartiNoStr = '';
    For Select HamStokMaster.HamStokMasterID, HamStokMaster.Lot, Parti.PartiYil, Parti.PartiID from PartiStok
      left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiStok.PartiSinifID
      left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left Join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join PartiStok YariMamulPartiStok on YariMamulPartiStok.PartiSinifID = HamStokMaster.PartiSinifId
      left join Parti on Parti.PartiKayitID = YariMamulPartiStok.PartiKayitId
      where PartiStok.PartiSinifId = :PartiSinifId
      Group By HamStokMaster.HamStokMasterID, HamStokMaster.Lot, Parti.PartiYil, Parti.PartiID
      into :FHamStokMasterID, :FLot, :YariMamulPartiYil, :YariMamulPartiId do
    begin
      Str = EgemenIntegerToStr(FHamStokMasterID);
      if (Str is not null) then
      if (PosBul(HamStokMasterIDStr, Str) = 0) then
      if (Char_Length(HamStokMasterIDStr) + Char_Length(Str) < 97) then
      begin
        if (HamStokMasterIDStr <> '') then HamStokMasterIDStr = HamStokMasterIDStr || cast(', ' as varchar(2) character set UTF8);
            HamStokMasterIDStr = HamStokMasterIDStr || Str;
      end

      if (Char_Length(Lot) + Char_Length(FLot) < 150) then
      begin
        if (Lot <> '' ) then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
         Lot = Lot || FLot;
      end

      Str = '('||YariMamulPartiYil||') '||YariMamulPartiId;
      if (YariMamulPartiYil is not null) then
        if (PosBul(YariMamulPartiNoStr, Str) = 0) then
          if (Char_Length(YariMamulPartiNoStr) + Char_Length(Str) < 198) then
          begin
            if (YariMamulPartiNoStr <> '') then YariMamulPartiNoStr = YariMamulPartiNoStr ||  cast(', ' as varchar(2) character set UTF8);
            YariMamulPartiNoStr = YariMamulPartiNoStr || Str;
          end
    end

    if (Okey > 0) then
    begin
      PartiIdStr = '';
      BoyaSiparisIdStr = '';

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      BlobSize = char_length(FTerminNotu);
      if (BlobSize > 1023) then TerminNotu = left(FTerminNotu, 1023);
      else if (BlobSize > 0) then TerminNotu = left(FTerminNotu, BlobSize);
      else TerminNotu = '';

      BlobSize = char_length(FSiparisNotu);
      if (BlobSize > 1023) then SiparisNotu = left(FSiparisNotu, 1023);
      else if (BlobSize > 0) then SiparisNotu = left(FSiparisNotu, BlobSize);
      else SiparisNotu = '';

      BlobSize = char_length(FPlanlamaNotu);
      if (BlobSize > 1023) then PlanlamaNotu = left(FPlanlamaNotu, 1023);
      else if (BlobSize > 0) then PlanlamaNotu = left(FPlanlamaNotu, BlobSize);
      else PlanlamaNotu = '';

      BlobSize = char_length(FPartiGecikmeNotu);
      if (BlobSize > 1023) then PartiGecikmeNotu = left(FPartiGecikmeNotu, 1023);
      else if (BlobSize > 0) then PartiGecikmeNotu = left(FPartiGecikmeNotu, BlobSize);
      else PartiGecikmeNotu = '';

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (Fason = 1) then FasonStr = 'Fason';

      select min(PartiAsamalari.Sira) from PartiAsamaStok

        into :SonrakiSira;

      if (SonrakiSira is not null) then
      begin
        select PartiAsamaStok.AsamaId, Asama.Adi from PartiAsamaStok
          left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :SonrakiSira
          into :SonrakiAsamaId, :SonrakiAsamaAdi;
      end
      else
      begin
        SonrakiAsamaAdi    = null;
        SonrakiAsamaId     = null;
      end

      select count(*) from PartiStokBekle where PartiStokBekle.PartiSinifId = :PartiSinifId and PartiStokBekle.Bitti = 0 into :KayitSayisi;
      if (KayitSayisi > 0) then Beklemede = 1;
      else Beklemede = 0;
      suspend;
    end
  end

           (Ham.HamTipiId = :hstDokumaKumas and :APartiTipiId = 3) or :APartiTipiId is null) and
         ( (BoyaSiparisDetay.LabReceteId is not null and :AImalatTipiId = 0) or (BoyaSiparisDetay.DesenId is not null and :AImalatTipiId = 1) or :AImalatTipiId is null )

       :SiparisKilo, :SiparisMetre, :PartilenenKilo, :IptalKilo, :PartilenenMetre, :IptalMetre, :HamAdi, :ReferansKodu, :SiparisProsesId, :SiparisProsesAdi,

    Okey = 1;

    if (AAsamaId is not null) then
    begin
      select count(*) from SiparisProsesAsamalari
      where SiparisProsesAsamalari.SiparisProsesId = :SiparisProsesId and SiparisProsesAsamalari.AsamaId = :AAsamaId
      into :Okey;
    end
    else if (ABoyahaneBolumId is not null) then
    begin

      left join Asama on Asama.AsamaId = SiparisProsesAsamalari.AsamaId
      where SiparisProsesAsamalari.SiparisProsesId = :SiparisProsesId and Asama.BoyahaneBolumId = :ABoyahaneBolumId
      into :Okey;

    For Select HamStokMaster.HamStokMasterID, HamStokMaster.Lot from PartiStok
      left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiStok.PartiSinifID
      left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId

      left Join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      where PartiStok.PartiSinifId = :PartiSinifId
      Group By HamStokMaster.HamStokMasterID, HamStokMaster.Lot
      into :FHamStokMasterID, :FLot do
    begin
      Str = EgemenIntegerToStr(FHamStokMasterID);
      if (Str is not null) then
      if (PosBul(HamStokMasterIDStr, Str) = 0) then
      if (Char_Length(HamStokMasterIDStr) + Char_Length(Str) < 97) then
      begin
        if (HamStokMasterIDStr <> '') then HamStokMasterIDStr = HamStokMasterIDStr || cast(', ' as varchar(2) character set UTF8);

      end

      if (Char_Length(Lot) + Char_Length(FLot) < 150) then

        if (Lot <> '' ) then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
        Lot = Lot || FLot;
      end
    end

      else if (BlobSize > 0) then PlanlamaNotu = left(FPlanlamaNotu, BlobSize);

        Metre = SiparisMetre - PartilenenMetre + IptalMetre;
      end
      else
      begin

        begin

          PartilenenMetre = BaskiPartilenenMetre;
          IptalKilo = BaskiDetayIptalKilo;
          IptalMetre = BaskiDetayIptalMetre;
        end
        else
        begin
          Kilo = SiparisKilo - PartilenenKiloBaski + IptalKiloBaski;
          Metre = SiparisMetre - PartilenenMetreBaski + IptalMetreBaski;
        end
      end

      if (Kilo < 0) then Kilo = 0;
      if (Metre < 0) then Metre = 0;

      ImalatTipiStr = '';

      if (LabReceteId is not null) then
      begin
        ImalatTipiStr = 'Boya';
        ImalatTipiId = 0;
      end
      if (DesenID is not null) then
      begin
        ImalatTipiStr = 'Baskı';
        ImalatTipiId = 1;
      end

      if (Kapali = 1) then KapaliStr = 'Evet';

        PartilenenMetreBaski = PartilenenMetreBaski - IptalMetreBaski;
      end
      if (Kilo > 0 or Metre > 0) then suspend;
    end
