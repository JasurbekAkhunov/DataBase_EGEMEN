CREATE PROCEDURE IPLIKSEVKLERIORMEYEKAYDET(
    AIPLIKIRSALIYEID INTEGER,
    AKULLANICIID INTEGER,
    ADOKUMAHAREKETI INTEGER)
RETURNS (
    ORMEIRSALIYEID INTEGER)
AS
declare variable IplikFirmaKodumuz      varchar(20);
declare variable SistemDovizID          varchar(5);
declare variable HareketTipi            integer;
declare variable IplikTalepIdStr        varchar(30);
declare variable RenkId                 integer;
declare variable IplikTalepYil          integer;
declare variable IplikTalepId           integer;
declare variable IplikIrsaliyeDetayId   integer;
declare variable OrmeDefaultRafId       varchar(10);
declare variable DokumaDefaultRafId     varchar(10);
declare variable OrmeDefaultDepoId      varchar(10);

declare variable TransferYeri           integer;
declare variable KS                     integer;
declare variable IlkKayit               integer;
declare variable StokHareketId          integer;
declare variable OrmeStokGirisindeOtomatikRafla integer;
declare variable OrmeBarkodluGirisCikisTakibi SmallInt;
declare variable TalBirimMaliyet  Double Precision;
declare variable SipSatisFayati   Double Precision;
declare variable SipDovizID       Varchar(5);
declare variable OrmeSiparisDetayId Integer;
declare variable BirimFiyat Double Precision;
declare variable EgemenFirmaID integer;
declare variable DovizKuru Double Precision;
begin
  Select shtAlis from OrmeStokHareketTipiGetir into :HareketTipi;

  Select IplikFirmaKodumuz, SistemDovizID, OrmeDefaultRafId, DokumaDefaultRafId, OrmeStokGirisindeOtomatikRafla, OrmeBarkodluGirisCikisTakibi, EgemenFirmaID  from Sistem
    where sistemid = 1
    into :IplikFirmaKodumuz, :SistemDovizID, :OrmeDefaultRafId, :DokumaDefaultRafId, :OrmeStokGirisindeOtomatikRafla ,:OrmeBarkodluGirisCikisTakibi, :EgemenFirmaID;

    if (ADokumaHareketi = 1) then
        OrmeDefaultRafId = DokumaDefaultRafId;

  Select DepoRaf.DepoId from DepoRaf where DepoRaf.RafID = :OrmeDefaultRafId into :OrmeDefaultDepoId;
  IlkKayit = 0;
  OrmeIrsaliyeID = Null;
  Select IplikIrsaliye.OrmeIrsaliyeID from IplikIrsaliye where IplikIrsaliye.IplikIrsaliyeID = :AIplikIrsaliyeID into :OrmeIrsaliyeID;

  if (OrmeIrsaliyeID is null) then
  begin
    IlkKayit = 1;
    OrmeIrsaliyeId = gen_Id(OrmeIrsaliye_Id_gen,1);
    insert into OrmeIrsaliye(OrmeIrsaliyeID, HareketTipi, FirmaId, SeriNo, IrsaliyeNo, IrsaliyeTarihi, FiiliSevkTarihi,
                             AracId, SoforId, Sinif, IrsaliyeGrupId, DovizId, TransferYeri, InsertKullaniciId, KullaniciId, DokumaHareketi)
         Select :OrmeIrsaliyeID, :HareketTipi, :IplikFirmaKodumuz, IplikIrsaliye.SeriNo, IplikIrsaliye.IrsaliyeNo,
                IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.FiiliSevkTarihi, IplikIrsaliye.AracId,
                IplikIrsaliye.SoforId, IplikIrsaliye.Sinif, IplikIrsaliye.IrsaliyeGrupId, :SistemDovizId, 2,
                :AKullaniciId, :AKullaniciId, :ADokumaHareketi from IplikIrsaliye
         where IplikIrsaliye.IplikIrsaliyeID = :AIplikIrsaliyeID;
  end

  For Select IplikIrsaliyeDetay.IplikIrsaliyeDetayId, IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, IplikTalepDetay.BirimMaliyet, OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.SatisFiyati, OrmeSiparisDetay.DovizID from IplikIrsaliyeDetay
      left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
      left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayID = IplikIrsaliyeDetay.IplikTalepDetayID
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
      where IplikIrsaliyeDetay.IplikIrsaliyeID = :AIplikIrsaliyeID and IplikIrsaliyeDetay.OrmeIrsaliyeId is null
      into :IplikIrsaliyeDetayId, :IplikTalepYil, :IplikTalepId, :TalBirimMaliyet, :OrmeSiparisDetayId, :SipSatisFayati, :SipDovizID do
  begin
    BirimFiyat = 0;

    if (EgemenFirmaID = 56) then /*Rem Ã–rme BirimFiyat*/
    begin
      if (OrmeSiparisDetayId is not Null) then
      begin
        if (SipDovizID <> SistemDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:SipDovizID, :SistemDovizId, Current_Date) into :DovizKuru;
        else DovizKuru = 1;

        BirimFiyat = SipSatisFayati * DovizKuru;
      end
      else BirimFiyat = TalBirimMaliyet;
    end

    IplikTalepIdStr = '';

    if (IplikTalepID is not null) then
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil)  || ') ' || EgemenIntegerToStr(IplikTalepId);

    StokHareketID = Gen_Id(StokHareket_Id_gen, 1);

    insert into StokHareket(OrmeIrsaliyeId, StokHareketID, HareketTipi, HamId, RenkId, KapAdedi, BrutMiktar,
                            Miktar, UretimPartiNo, HamIade, Lot, FlamKodu, MelanjKodu, IplikIrsaliyeDetayId, KonikRengiId,
                            IplikTalepDetayId, BoyaSiparisKayitId, OrmeSiparisDetayId, BoyaSiparisDetayId, DepoId, BirimFiyat,
                            InsertKullaniciId, KullaniciId, DokumaHareketi)
           Select :OrmeIrsaliyeId, :StokHareketID, :HareketTipi, IplikIrsaliyeDetay.HamId, coalesce(IplikIrsaliyeDetay.RenkId, HarmanRenk.RenkId),
                  IplikIrsaliyeDetay.KapAdedi, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.NetMiktar, :IplikTalepIdStr, 0,
                  IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.FlamKodu, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.IplikIrsaliyeDetayId,
                  IplikIrsaliyeDetay.KonikRengiId, IplikIrsaliyeDetay.IplikTalepDetayId,
                  IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId, IplikTalepDetay.BoyaSiparisDetayId,
                  :OrmeDefaultDepoId, :BirimFiyat, :AKullaniciId, :AKullaniciId, :ADokumaHareketi from IplikIrsaliyeDetay
        left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayID = IplikIrsaliyeDetay.IplikTalepDetayID
        left join Harman on Harman.HarmanKayitId = IplikIrsaliyeDetay.HarmanKayitId
        left join Renk HarmanRenk on HarmanRenk.RenkId = Harman.RenkId
        where IplikIrsaliyeDetay.IplikIrsaliyeDetayID = :IplikIrsaliyeDetayID;

    if (OrmeStokGirisindeOtomatikRafla = 0) then
    begin
      Insert into OrmeRafDetay(StokHareketId, RafId, KapAdedi, Miktar, BrutMiktar, InsertKullaniciId, KullaniciId)
                  Select StokHareket.StokHareketId, :OrmeDefaultRafId, StokHareket.KapAdedi, StokHareket.Miktar, StokHareket.BrutMiktar,
                        :AkullaniciId, :AKullaniciId from StokHareket where StokHareket.StokHareketId = :StokHareketId;
    end

    Update IplikIrsaliyeDetay Set OrmeIrsaliyeID = :OrmeIrsaliyeID, NeredenUpdate = 3 where IplikIrsaliyeDetay.IplikIrsaliyeDetayID = :IplikIrsaliyeDetayId;

    if (IlkKayit = 1) then
      Update IplikIrsaliye Set OrmeIrsaliyeID = :OrmeIrsaliyeId where IplikIrsaliye.IplikIrsaliyeID = :AIplikIrsaliyeID;


    if (OrmeBarkodluGirisCikisTakibi = 1) then
    begin
      insert into StokHareketDetay(StokHareketID, HareketTipi, IplikStokID, IplikStokNumarator, Kilo, BrutKilo, InsertKullaniciId, KullaniciId)
             Select :StokHareketID, :HareketTipi, IplikStok.IplikStokID, IplikStok.Numarator, IplikStok.NetMiktar, IplikStok.Miktar, :AKullaniciId, :AKullaniciId from IplikStok
          where IplikStok.CikisIplikIrsaliyeDetayId = :IplikIrsaliyeDetayID;
    end
  end
  suspend;
end
