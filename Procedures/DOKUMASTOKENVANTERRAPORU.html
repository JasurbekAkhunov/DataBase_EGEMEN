     into :HamId, :HamTipiId, :HamTipiIdStr, :HamKodu, :HamAdi, :HamAdi2, :HamAdi3, :HamAdi4, :KumasCinsiId, :KumasCinsiAdi,

          :Kapali, :AlisFiyati, :FDovizId, :HamSatisTransferKodu, :FasonTransferKodu, :FasonOrmeTransferKodu, :FasonArtiSatOrmeTransferKodu, :AlisTransferKodu, :RenkliAlisTransferKodu, :BoyaliTransferKodu, :SekerKasarTransferKodu,
       :OptikBeyazTransferKodu, :YikamaTransferKodu, :SABoyaliTransferKodu, :SASekerKasarTransferKodu, :SAOptikBeyazTransferKodu, :SAYikamaTransferKodu, :AlisBoyaliTransferKodu, :AlisSekerKasarTransferKodu,
       :AlisOptikBeyazTransferKodu, :AlisYikamaTransferKodu, :AlisBaskiliTransferKodu, :HamGrup1Adi, :HamGrup2Adi, :HamGrup3Adi, :HamGrup4Adi do
  begin
    KapaliStr = '';
    if (Kapali = 1) then KapaliStr = 'Evet';

    Okey = 1;
    if (AHamGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, HamGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, HamGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, HamGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, HamGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      SonAlisFiyati = null ;
      SonAlisFiyatiDovizId = null ;
      OrtalamaAlisFiyati = null ;

      select SonAlisFiyati, DovizId, OrtalamaAlisFiyati from OrmeHamStokSonAlisFiyati(:HamId, Null, Null, Null, 1, Null, Null) into :SonAlisFiyati, :SonAlisFiyatiDovizId, :OrtalamaAlisFiyati;

      For select CASE WHEN Char_Length(RenkId) > 0 THEN 1 ELSE 0 END AS RenkDurumu from stokhareket
        where StokHareket.HamId = :HamId group by RenkDurumu into :RenkDurumuId do
      begin

        StokMiktari = 0;
        StokMiktariTL = 0;

        select Devir, AlisToplami, AlisIadeToplami, FasonGirisToplami, FasonGirisIadeToplami, SatisToplami,
                BrutSatisToplami, SatisIadeToplami, UretimeCikisToplami, UretimdenIadeToplami, NormalUretimToplami, SakatUretimToplami,
                UretimToplami, FasonUretimeCikisToplami, FasonUretimdenIadeToplami, FasonUretimToplami, IcUretimeCikisToplami,
                IcUretimdenIadeToplami, IcUretimToplami, FasonUretimeCikisToplamiBoya, FasonUretimdenIadeToplamiBoya, FasonUretimToplamiBoya,
                KendiBoyamizaCikisToplamiBoya, KendiBoyamizdanIadeToplamiBoya, KendiBoyamizUretimToplamiBoya, IcUretimeCikisToplamiBoya,
                IcUretimdenIadeToplamiBoya, IcUretimToplamiBoya, FasonUretimeCikisToplamiBukum, FasonUretimdenIadeToplamiBukum, FasonUretimToplamiBukum,
                IcUretimeCikisToplamiBukum, IcUretimdenIadeToplamiBukum, IcUretimToplamiBukum, StokKaydirmaGiris, StokKaydirmaCikis,
                SarfToplami, FStokKapAdedi, DevirTL, AlisToplamiTL, AlisIadeToplamiTL, FasonGirisToplamiTL, FasonGirisIadeToplamiTL,
                SatisToplamiTL, BrutSatisToplamiTL, SatisIadeToplamiTL, UretimeCikisToplamiTL, UretimdenIadeToplamiTL, NormalUretimToplamiTL,
                SakatUretimToplamiTL, UretimToplamiTL, FasonUretimeCikisToplamiTL, FasonUretimdenIadeToplamiTL, FasonUretimToplamiTL,
                IcUretimeCikisToplamiTL, IcUretimdenIadeToplamiTL, IcUretimToplamiTL, FasonUretimeCikisTBoyaTL, FasonUretimdenIadeTBoyaTL,
                FasonUretimTBoyaTL, KendiBoyamizaCikisTBoyaTL, KendiBoyamizdanIadeTBoyaTL, KendiBoyamizUretimTBoyaTL,
                IcUretimeCikisTBoyaTL, IcUretimdenIadeTBoyaTL, IcUretimTBoyaTL, FasonUretimeCikisTBukumTL, FasonUretimdenIadeTBukumTL,
                FasonUretimTBukumTL, IcUretimeCikisTBukumTL, IcUretimdenIadeTBukumTL, IcUretimTBukumTL, StokKaydirmaGirisTL,
                StokKaydirmaCikisTL, SarfToplamiTL from DokumaStokEnvanterRaporu_Ic
                (:ARaporBaslangicTarihi, :ARaporBitisTarihi, :HamId, :RenkDurumuId)
        into :Devir, :AlisToplami, :AlisIadeToplami, :FasonGirisToplami, :FasonGirisIadeToplami, :SatisToplami,
                :BrutSatisToplami, :SatisIadeToplami, :UretimeCikisToplami, :UretimdenIadeToplami, :NormalUretimToplami, :SakatUretimToplami,
                :UretimToplami, :FasonUretimeCikisToplami, :FasonUretimdenIadeToplami, :FasonUretimToplami, :IcUretimeCikisToplami,
                :IcUretimdenIadeToplami, :IcUretimToplami, :FasonUretimeCikisToplamiBoya, :FasonUretimdenIadeToplamiBoya, :FasonUretimToplamiBoya,
                :KendiBoyamizaCikisToplamiBoya, :KendiBoyamizdanIadeToplamiBoya, :KendiBoyamizUretimToplamiBoya, :IcUretimeCikisToplamiBoya,
                :IcUretimdenIadeToplamiBoya, :IcUretimToplamiBoya, :FasonUretimeCikisToplamiBukum, :FasonUretimdenIadeToplamiBukum, :FasonUretimToplamiBukum,
                :IcUretimeCikisToplamiBukum, :IcUretimdenIadeToplamiBukum, :IcUretimToplamiBukum, :StokKaydirmaGiris, :StokKaydirmaCikis,
                :SarfToplami, :FStokKapAdedi, :DevirTL, :AlisToplamiTL, :AlisIadeToplamiTL, :FasonGirisToplamiTL, :FasonGirisIadeToplamiTL,
                :SatisToplamiTL, :BrutSatisToplamiTL, :SatisIadeToplamiTL, :UretimeCikisToplamiTL, :UretimdenIadeToplamiTL, :NormalUretimToplamiTL,
                :SakatUretimToplamiTL, :UretimToplamiTL, :FasonUretimeCikisToplamiTL, :FasonUretimdenIadeToplamiTL, :FasonUretimToplamiTL,
                :IcUretimeCikisToplamiTL, :IcUretimdenIadeToplamiTL, :IcUretimToplamiTL, :FasonUretimeCikisTBoyaTL, :FasonUretimdenIadeTBoyaTL,
                :FasonUretimTBoyaTL, :KendiBoyamizaCikisTBoyaTL, :KendiBoyamizdanIadeTBoyaTL, :KendiBoyamizUretimTBoyaTL,
                :IcUretimeCikisTBoyaTL, :IcUretimdenIadeTBoyaTL, :IcUretimTBoyaTL, :FasonUretimeCikisTBukumTL, :FasonUretimdenIadeTBukumTL,
                :FasonUretimTBukumTL, :IcUretimeCikisTBukumTL, :IcUretimdenIadeTBukumTL, :IcUretimTBukumTL, :StokKaydirmaGirisTL,
                :StokKaydirmaCikisTL, :SarfToplamiTL;

        SakatUretimToplami = Null;
        select Sum(StokHareket.Miktar),
                 sum(StokHareket.Miktar * (StokHareket.BirimMaliyet) / StokHareket.TabanDovizKuru1 * StokHareket.TabanDovizKuru2) from StokHareket
          where StokHareket.HamId = :HamId and (StokHareket.Tarih <= :ARaporBitisTarihi and StokHareket.Tarih >= :ARaporBaslangicTarihi) and
                StokHareket.GirisCikis <> 0 and StokHareket.Sakat = 1 and ((RenkId is Null and :RenkDurumuId = 0) or (RenkId is not Null and :RenkDurumuId >= 1))
          into :SakatUretimToplami, :FMaliyet;
        if (SakatUretimToplami is null) then SakatUretimToplami = 0;
        if (FMaliyet is null) then SakatUretimToplamiTL = 0;
        else SakatUretimToplamiTL = FMaliyet;

        NormalUretimToplami = UretimToplami - SakatUretimToplami;
        NormalUretimToplamiTL = UretimToplamiTL - SakatUretimToplamiTL;

        StokMiktari = Devir + AlisToplami - SatisToplami + FasonGirisToplami  -
                      AlisIadeToplami + SatisIadeToplami - FasonGirisIadeToplami - UretimeCikisToplami +
                      UretimdenIadeToplami + UretimToplami  -
                      StokKaydirmaCikis + StokKaydirmaGiris - SarfToplami;

        StokMiktariTL = DevirTL + AlisToplamiTL - SatisToplamiTL + FasonGirisToplamiTL  -
                      AlisIadeToplamiTL + SatisIadeToplamiTL - FasonGirisIadeToplamiTL - UretimeCikisToplamiTL +
                      UretimdenIadeToplamiTL + UretimToplamiTL  -
                      StokKaydirmaCikisTL + StokKaydirmaGirisTL - SarfToplamiTL;

        KapAdedi = FStokKapAdedi;

        StokMiktari = Yuvarla(StokMiktari, 1000);

        if (ASifirOlanlarDokulsunmu = 1 or StokMiktari <> 0) then
        begin
          if (FDovizId is not null and FDovizId <> DovizId) then
          begin
            select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :DovizId, Current_Date) into :FDovizKuru;
            AlisFiyati = AlisFiyati * FDovizKuru;
          end

          if (RenkDurumuId = 0) then RenkDurumu = 'Ham';
          else RenkDurumu = 'Renkli';

          suspend;
        end
      end
    end
  end
end
