CREATE PROCEDURE DOKUMAPLANLAMAARAVERMEYAZ(
    ADOKUMAPLANLAMAGRUPID VARCHAR(10),
    ADOKUMAMAKINAID VARCHAR(10),
    AFIRMAID VARCHAR(20))
RETURNS (
    DOKUMAPLANLAMAID INTEGER,
    ARAVERMESEBEBIID INTEGER,
    ARAVERMESEBEBIADI ADI50,
    ARAVERDIRENPERSONELID INTEGER,
    ARAVERDIRENPERSONELADI ADI40,
    ARAYIBITIRENPERSONELID INTEGER,
    ARAYIBITIRENPERSONELADI ADI40,
    ARAVERMETARIHI TIMESTAMP,
    ARABITISTARIHI TIMESTAMP)
AS
begin
  for  select DokumaPlanlamaYaz.DokumaPlanlamaId, AraVermeSebebi.AraVermeSebebiId, AraVermeSebebi.Adi AraVermeSebebiAdi, AVP.PersonelId AraVerdirenPersonelId, AVP.Adi AraVerdirenPersonelAdi,
    AVBP.PersonelId ArayiBitirenPersonelId, AVBP.Adi ArayiBitirenPersonelAdi,
    OrmeMakinaDurumu.Tarih AraVermeTarihi, OMD.Tarih AraBitisTarihi
    from DokumaPlanlamaYaz(:ADokumaPlanlamaGrupId, :ADokumaMakinaId, :AFirmaId)
    left join OrmeMakinaDurumu on OrmeMakinaDurumu.DokumaPlanlamaId = DokumaPlanlamaYaz.DokumaPlanlamaId
    left join ImalataAraVerme on ImalataAraVerme.BasladiOrmeMakinaDurumuId = OrmeMakinaDurumu.OrmeMakinaDurumuId
    left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
    left join Personel AVP on AVP.PersonelId = ImalataAraVerme.AraVerdirenPersonelId
    left join Personel AVBP on AVBP.PersonelId = ImalataAraVerme.ArayiBitirenPersonelId
    left join OrmeMakinaDurumu OMD on OMD.OrmeMakinaDurumuId = ImalataAraVerme.BittiOrmeMakinaDurumuId
    where OrmeMakinaDurumu.ImalataAraVerildi = 1 into :DokumaPlanlamaId, :AraVermeSebebiId, :AraVermeSebebiAdi, :AraVerdirenPersonelId, :AraVerdirenPersonelAdi,
    :ArayiBitirenPersonelId, :ArayiBitirenPersonelAdi, :AraVermeTarihi, :AraBitisTarihi do
      suspend;
end
