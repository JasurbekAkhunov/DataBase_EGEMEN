CREATE PROCEDURE ORMESIPARISOLCUDETAYIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    ORMESIPARISDETAYID INTEGER,
    ORMESIPARISOLCUID INTEGER,
    OLCUID INTEGER,
    OLCUADI VARCHAR(40),
    ADET DOUBLE PRECISION,
    KILO DOUBLE PRECISION,
    TOPLAMORULEN BOYAMIKTARSIZE,
    TOPLAMSAKATORULEN BOYAMIKTARSIZE,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    KAPADEDI INTEGER,
    TOPLAMORULENADET NUMERIC(12,3),
    TOPLAMORULENKILO NUMERIC(12,3),
    KALANADET NUMERIC(12,3),
    KALANKILO NUMERIC(12,3))
AS
  declare uitOrme integer;
  declare shtUretim integer;
  declare shtFasonUretim integer;
begin
  select first 1 shtUretim, shtFasonUretim from OrmeStokHareketTipiGetir into :shtUretim, :shtFasonUretim;
  select first 1 uitOrme from UretimIslemTipiGetir into :uitOrme;
  for select OrmeSiparisOlcu.OrmeSiparisDetayId, OrmeSiparisOlcu.OlcuId, Olcu.OlcuAdi, OrmeSiparisOlcu.Adet, OrmeSiparisOlcu.Kilo, OrmeSiparisOlcu.ormesiparisolcuid,
             Ik.Adi, OrmeSiparisOlcu.InsertTarihi, Sk.Adi, OrmeSiparisOlcu.DegisimTarihi, OrmeSiparisOlcu.ToplamOrulen, OrmeSiparisOlcu.ToplamSakatOrulen, OrmeSiparisOlcu.KapAdedi from OrmeSiparisOlcu
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeSiparisOlcu.OrmeSiparisDetayId
      left join Olcu on Olcu.OlcuId = OrmeSiparisOlcu.OlcuId
      Left Join Kullanici IK on IK.KullaniciId = OrmeSiparisOlcu.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = OrmeSiparisOlcu.KullaniciId
      where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId
      into :OrmeSiparisDetayId, :OlcuId, :OlcuAdi, :Adet, :Kilo, :OrmeSiparisOlcuId, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
      :ToplamOrulen, :ToplamSakatOrulen, :KapAdedi do
      begin
        ToplamOrulenAdet  = 0;
        ToplamOrulenKilo  = 0;

        select coalesce(sum(StokHareket.Adet), 0), coalesce(sum(StokHareket.Kilo), 0) from StokHareket
        where (StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId) and (StokHareket.Olcuid = :Olcuid) and (StokHareket.UretimIslemTipiId = :uitOrme) and
        (StokHareket.HareketTipi = :shtUretim or StokHareket.HareketTipi = :shtFasonUretim) and (StokHareket.HamIade = 0)
        into :ToplamOrulenAdet, :ToplamOrulenKilo;
        KalanKilo = :Kilo - :ToplamOrulenKilo;
        KalanAdet = :Adet - :ToplamOrulenAdet;

        suspend;
      end
end
