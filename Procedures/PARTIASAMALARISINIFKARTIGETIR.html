CREATE PROCEDURE PARTIASAMALARISINIFKARTIGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTISINIFID INTEGER,
    PARTISIRANO INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    HAMTIPIID INTEGER,
    ASAMAYADAHIL INTEGER,
    ASAMAYADAHILSTR VARCHAR(5),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    PROXYID VARCHAR(8),
    PAKETLEMENO VARCHAR(30),
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    GRUPPROSESNO VARCHAR(10),
    GRUPPROSESADI VARCHAR(80),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    PARTISTOKBEKLEID INTEGER,
    PARTISTOKBEKLEBITTI INTEGER,
    DURUMU INTEGER,
    BELGEHAMADI VARCHAR(60),
    LABRECETEKODU VARCHAR(20),
    LABRECETEATISID INTEGER,
    LABRECETEATISNO INTEGER,
    HESAPLANANMETRE NUMERIC(12,3))
AS
declare variable GirisOtomatik integer;
declare variable CikisOtomatik integer;
declare variable PaketlemeYil  varchar(4);
declare variable PaketlemeId   varchar(10);
declare variable KumasPaketlemeAsamaId integer;
declare variable AsamaId integer;
declare variable FEnIstenen varchar(10);
declare variable FGramajIstenen varchar(10);
declare variable FMetre Double Precision;
begin
   select KumasPaketlemeAsamaId from Sistem where SistemId = 1 into :KumasPaketlemeAsamaId;

   For Select PartiStok.BoyaSiparisDetayId, PartiStok.PartiKayitId, PartiAsamaStok.PartiAsamalariId, PartiAsamaStok.PartiSinifId, PartiStok.PartiSiraNo,
              PartiAsamaStok.AsamayaDahil, PartiStok.hamTipiId, PartiAsamaStok.Giristarihi,  PartiAsamaStok.Cikistarihi,
              PartiStok.HamAdiToplami, PartiAsamaStok.GirisKilo, PartiAsamaStok.GirisMetre, PartiStok.KapAdedi, PartiStok.Adet,
              PartiStok.ProxyId, IK.Adi, PartiAsamaStok.InsertTarihi, SK.Adi, PartiAsamaStok.DegisimTarihi,
              PartiAsamaStok.MakinaPlanlamaGrupId, PartiAsamaStok.GrupProsesNo, MakinaPlanlamaGrupProses.Adi,
              PartiStokBekle.PartiStokBekleId, PartiStokBekle.Bitti, PartiStok.Durumu,
              PartiAsamaStok.PaketlemeYil, PartiAsamaStok.PaketlemeId, PartiAsamaStok.AsamaId, BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.LabReceteAtisId, LabReceteAtis.AtisNo,
              BoyaSiparisDetay.BelgeHamAdi, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen from PartiAsamaStok
   left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
   left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
   left join MakinaPlanlamaGrupProses on MakinaPlanlamaGrupProses.MakinaPlanlamaGrupId = PartiAsamaStok.MakinaPlanlamaGrupId and MakinaPlanlamaGrupProses.GrupProsesNo = PartiAsamaStok.GrupProsesNo
   left join PartiStokBekle on PartiStokBekle.PartiSinifId = PartiAsamaStok.PartiSinifId and PartiStokBekle.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
   left join Kullanici IK on IK.KullaniciId = PartiAsamaStok.InsertKullaniciId
   left join Kullanici SK on SK.KullaniciId = PartiAsamaStok.KullaniciId
   left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
   left join LabReceteAtis on LabReceteAtis.LabReceteAtisID = BoyaSiparisDetay.LabReceteAtisID
   where PartiAsamaStok.PartiKayitId = :APartiKayitId order by PartiStok.PartiSinifId
  into :BoyaSiparisDetayId, :PartiKayitId, :PartiAsamalariId, :PartiSinifId, :PartiSiraNo, :AsamayaDahil, :HamTipiId,
       :Giristarihi, :Cikistarihi, :HamAdiToplami, :Kilo, :Metre, :KapAdedi,
       :Adet, :ProxyId, :InsertKullaniciAdi, :InsertTarihi , :KullaniciAdi, :DegisimTarihi,
       :MakinaPlanlamaGrupId, :GrupProsesNo, :GrupProsesAdi, :PartiStokBekleId, :PartiStokBekleBitti, :Durumu,
       :PaketlemeYil, :PaketlemeId, :AsamaId, :LabReceteKodu, :LabReceteAtisId, :LabReceteAtisNo, :BelgeHamAdi, :FEnIstenen, :FGramajIstenen do
   begin
     AsamayaDahilStr = Null;
     if (AsamayaDahil = 1) then AsamayaDahilStr = 'Dahil';

     PaketlemeNo = '';

     if (AsamaId = KumasPaketlemeAsamaId) then
     begin
       if (PaketlemeYil is not null) then
       begin
         PaketlemeNo = EgemenCopyStr(PaketlemeYil, 3, 2) || '-' || PaketlemeId;
       end
     end

     HesaplananMetre = 0;
     if (EgemenStrToIntOlurmu(FEnIstenen) = 1 and EgemenStrToIntOlurmu(FGramajIstenen) = 1 and (EgemenStrToInt(FEnIstenen) > 0 and EgemenStrToInt(FGramajIstenen) > 0)) then
     begin
       FMetre = EgemenStrToInt(FEnIstenen) * EgemenStrToInt(FGramajIstenen);
       HesaplananMetre = (1000/(EgemenStrToInt(FEnIstenen) * EgemenStrToInt(FGramajIstenen))) * 100 * Kilo;
     end

     suspend;
   end
end
