CREATE PROCEDURE CSODEMEPLANIAYTOPLAMLI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APROTESTO INTEGER,
    ABANKACIKIS INTEGER,
    ADOVIZID VARCHAR(5),
    ARAPORDOVIZID VARCHAR(5))
RETURNS (
    AV VARCHAR(1),
    CS VARCHAR(1),
    VADETARIHI TIMESTAMP,
    ODEME DOUBLE PRECISION,
    TAHSILAT DOUBLE PRECISION,
    BAKIYE DOUBLE PRECISION,
    BAKIYEBA VARCHAR(1))
AS
declare variable FBakiye        Double Precision;
declare variable FOdeme         Double precision;
declare variable FTahsilat      Double precision;
begin
  FBakiye = 0;
  Bakiye = 0;
  for select vadetarihiAy, av,cs, Sum(Tahsilat), Sum(odeme) from CSOdemePlani(:ABaslangicTarihi, :ABitisTarihi, :AProtesto, :ABankaCikis, :ADovizId, :ARaporDovizId)
  group by vadetarihiAy, av,cs
  into :vadetarihi, :Av, :Cs, :FTahsilat, :FOdeme do
  begin
    Tahsilat = FTahsilat;
    Odeme = FOdeme;
    if (FTahsilat is null) then FTahsilat = 0;
    if (FOdeme is null) then FOdeme = 0;
    FBakiye = FBakiye + FTahsilat - FOdeme;

    if (FBakiye < 0) then
    begin
      Bakiye = FBakiye * -1;
      BakiyeBa = 'O';
    end
    else
    begin
      Bakiye = FBakiye;
      BakiyeBa = 'T';
    end
    suspend;
  end
end
