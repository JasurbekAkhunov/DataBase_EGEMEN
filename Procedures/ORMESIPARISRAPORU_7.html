  IplikHareketTipi = Null;
  MakinalarArasiTransfer = Null;
  OMIplikTransferFisiFisYil = Null;
  OMIplikTransferFisiFisNo  = Null;
  UcretliTamir = Null;
  BoyaSipBirimFiyat = Null;
  Adet = null;


  For Select IplikIrsaliye.HareketTipi, IplikIrsaliyeDetay.IplikIrsaliyeDetayId, IplikIrsaliyeDetay.IplikIrsaliyeId, BoyaSiparisDetay.HamId, Ham.Adi,
             IplikIrsaliye.IrsaliyeTarihi, BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi, IplikIrsaliye.FirmaId, Firma.Adi,
             IplikIrsaliyeDetay.KapAdedi, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.NetMiktar, IplikTalepDetay.BoyaSiparisDetayId,
             BoyaSiparisDetay.RenkAdi, IplikIrsaliyeDetay.Lot,
             IplikIrsaliye.Sinif, IplikIrsaliye.SeriNo, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
             OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.FlamKodu, IK.Adi, IplikIrsaliyeDetay.InsertTarihi, SK.Adi,
             IplikIrsaliyeDetay.DegisimTarihi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Ham.HamTipiId, HamTipi.Adi, BoyaSiparisDetay.UcretliTamir, BoyaSiparisDetay.BirimFiyat, OrmeSiparisDetay.OrmeSiparisDetayID,
              OS_1.BoyaSiparisKayitId from
            (select distinct OrmeSiparisRaporu_1.BoyaSiparisKayitId from OrmeSiparisRaporu_1(:ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ABaslangicSiparisTarihi, :ABitisSiparisTarihi, :AFirmaId,
            :AFASONSIPARISLER, :AFASONSATISSIPARISLER, :ASATISSIPARISLER, :AIplikSiparisi, :ABoyaliIplikSiparisi, :ALycraSiparisi, :AHamKumasSiparisi, :ABoyaliKumasSiparisi,
            :AIplikBoyaliKumasSiparisi, :AMusteriSiparisNo, :AReferansSiparisNo, :ARaporTipi, :AKapali, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :ASiparisiAlanKullaniciId,
            :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId, :AYurtDisi, :AKumasTeminSiparisi, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASinif, :AKaliteId, :ASipOnayDurumu, :AMaliOnayDurumu, :AKapatmaTarihiBaslangic, :AKapatmaTarihiBitis, :AImalatTipiId)) OS_1
  left join IplikTalepDetay on IplikTalepDetay.BoyaSiparisKayitId = OS_1.BoyaSiparisKayitId
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayID = IplikTalepDetay.IplikTalepDetayID
  left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikIrsaliyeDetay.IplikIrsaliyeId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = IplikTalepDetay.BoyaSiparisDetayId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = IplikTalepDetay.OrmeSiparisDetayId
  left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
  left join Firma on Firma.FirmaId = IplikIrsaliye.FirmaId
  left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
  left join Kullanici IK on IK.KullaniciId  = IplikIrsaliyeDetay.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = IplikIrsaliyeDetay.KullaniciId
  Left Join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  where IplikTalepDetay.IplikTalepDetayId is not null and
        IplikIrsaliyeDetay.IplikIrsaliyeDetayId is not null and
        (IplikIrsaliyeDetay.HareketTipi = :ShtSatisIplik or IplikIrsaliyeDetay.HareketTipi = :ShtSatisIadeIplik)
  into :IplikHareketTipi, :IplikIrsaliyeDetayId, :IplikIrsaliyeId, :HamId, :HamAdi, :Tarih, :BoyaIslemTipiId, :BoyaIslemTipiAdi, :FirmaId, :FirmaAdi,
       :KapAdedi, :BrutMiktar, :Miktar, :BoyaSiparisDetayId, :BoyaSiparisDetayRenkAdi, :Lot,  :FIrsaliyeSinif, :FIrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi,
       :SiparisSiraNo, :MelanjKodu, :FlamKodu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :HamAdi2, :HamAdi3, :HamAdi4, :HamTipiId, :HamTipiIdStr, :UcretliTamir, :BoyaSipBirimFiyat,
       :OrmeSiparisDetayID, :BoyaSiparisKayitId do
  begin
    if (IplikHareketTipi = shtSatisIplik) then HareketTipiStr = 'plik Fab. Sat';
    else
    begin
      if (IplikHareketTipi = shtSatisIadeIplik) then HareketTipiStr = 'plik Fab. Sat ade';
      else HareketTipiStr = 'Hatal...';
      IplikIrsaliyeID = Null;
    end

    IrsaliyeSinifVeSeri = FIrsaliyeSinif || '-';
    IrsaliyeSinifVeSeri = IrsaliyeSinifVeSeri || FIrsaliyeSeriNo;

    suspend;
  end
end
