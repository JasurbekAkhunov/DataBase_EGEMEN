CREATE PROCEDURE TRANSFERICINBOYAFATURALISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AEVRAKTIPI INTEGER,
    ATRANSFERDURUMU INTEGER,
    ASINIF VARCHAR(1))
RETURNS (
    ORMEFATURAID INTEGER,
    ORMESTOKHAREKETTIPIID INTEGER,
    IPLIKSTOKHAREKETTIPIID INTEGER,
    BOYAFATURAID INTEGER,
    IPLIKFATURAID INTEGER,
    HAMSTOKFATURAID INTEGER,
    HAMSTOKHAREKETTIPIID INTEGER,
    HAREKETTIPIID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    TRANSFEREDILDI INTEGER,
    TRANSFEREDILDISTR VARCHAR(7),
    TRANSFERTARIHI TIMESTAMP,
    TRANSFERKULLANICIID INTEGER,
    TRANSFERKULLANICIADI VARCHAR(40),
    FASON INTEGER,
    STATUS INTEGER,
    IHRACATFATURASI INTEGER,
    TEVKIFAT INTEGER,
    IPTALKAYDI INTEGER,
    EFATURA INTEGER,
    EFATURASTR VARCHAR(5))
AS
declare variable StokMuhasebeYazilimTipi  integer;
declare variable EgemenFirmaID  integer;

begin
  Select StokMuhasebeYazilimTipi, EgemenFirmaID from Sistem where SistemID = 1 into :StokMuhasebeYazilimTipi, :EgemenFirmaID;

  IptalKaydi = 0;
  EFatura   = 0;
  EFaturaStr = '';

  if (AEvrakTipi = 0) then
  begin
    HareketTipiId = null;
    for Select HamStokFatura.HamStokFaturaId, HamStokFatura.HamStokHareketTipiId, HamStokFatura.FirmaId, Firma.Adi, HamStokFatura.FaturaSeriNo, HamStokFatura.FaturaNo,
               HamStokFatura.FaturaTarihi, HamStokFatura.TransferEdildi, HamStokFatura.TransferTarihi, HamStokFatura.TransferKullaniciId,
               TK.Adi from HamStokFatura
      left join Firma On Firma.FirmaId = HamStokFatura.FirmaId
      left join Kullanici TK On TK.KullaniciId = HamStokFatura.TransferKullaniciId
      where HamStokFatura.FaturaTarihi >= :ABaslangicTarihi and HamStokFatura.FaturaTarihi <= :ABitisTarihi and
           (HamStokFatura.TransferEdildi = :ATransferDurumu or :ATransferDurumu is null) and
            HamStokFatura.HamStokHareketTipiId = 0
      order by HamStokFatura.FaturaTarihi, HamStokFatura.FaturaSeriNo, HamStokFatura.FaturaNo
    into :HamStokFaturaId, :HamStokHareketTipiId, :FirmaId, :FirmaAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :TransferEdildi, :TransferTarihi, :TransferKullaniciId,
         :TransferKullaniciAdi do
    begin
      if (TransferEdildi = 1) then TransferEdildiStr = 'Yapıldı';
      else TransferEdildiStr = '';
      Fason = Null;
      suspend;
    end

    if (StokMuhasebeYazilimTipi = 5 or StokMuhasebeYazilimTipi = 9) then
    begin
      HamStokFaturaId = NUll;
      HamStokHareketTipiId = Null;
      EFatura   = 0;
      EFaturaStr = '';

      for Select OrmeFatura.OrmeFaturaId, OrmeFatura.FirmaId, Firma.Adi, OrmeFatura.SeriNo, OrmeFatura.FaturaNo,
                 OrmeFatura.FaturaTarihi, OrmeFatura.TransferEdildi, OrmeFatura.TransferTarihi, OrmeFatura.TransferKullaniciId,
                 TK.Adi, OrmeFatura.HareketTipi, OrmeFatura.Tevkifat, OrmeFatura.EFatura from OrmeFatura
        left join Firma On Firma.FirmaId = OrmeFatura.FirmaId
        left join Kullanici TK On TK.KullaniciId = OrmeFatura.TransferKullaniciId
        where OrmeFatura.FaturaTarihi >= :ABaslangicTarihi and OrmeFatura.FaturaTarihi <= :ABitisTarihi and (OrmeFatura.HareketTipi = 1 or OrmeFatura.HareketTipi = 8 or OrmeFatura.HareketTipi = 4) and
              OrmeFatura.Sinif = :ASinif and
             (OrmeFatura.TransferEdildi = :ATransferDurumu or :ATransferDurumu is null) and OrmeFatura.Entegrasyon = 1
      order by OrmeFatura.SeriNo, OrmeFatura.FaturaNo
      into :OrmeFaturaId, :FirmaId, :FirmaAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :TransferEdildi, :TransferTarihi, :TransferKullaniciId,
           :TransferKullaniciAdi, :OrmeStokHareketTipiID, :Tevkifat, :EFatura do
      begin
        if (TransferEdildi = 1) then TransferEdildiStr = 'Yapıldı';
        else TransferEdildiStr = '';
        Fason = 0;

        EFaturaStr = '';
        if (EFatura = 1) then EFaturaStr = 'Evet';

        Select Min(BoyaSiparis.Fason) from StokHareket
          left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = StokHareket.BoyaSiparisKayitID
          where StokHareket.OrmeFaturaId = :OrmeFaturaId
          into :Fason;

        suspend;
      end
    end
    if (StokMuhasebeYazilimTipi = 4 and EgemenFirmaId = 46) then
    begin
      HamStokFaturaId       = NUll;
      HamStokHareketTipiId  = Null;
      OrmeFaturaId          = Null;
      OrmeStokHareketTipiID = Null;
      EFatura   = 0;
      EFaturaStr = '';

      for Select IplikFatura.IplikFaturaId, IplikFatura.FirmaId, Firma.Adi, IplikFatura.SeriNo, IplikFatura.FaturaNo,
               IplikFatura.FaturaTarihi, IplikFatura.TransferEdildi, IplikFatura.TransferTarihi, IplikFatura.TransferKullaniciId,
               TK.Adi, IplikFatura.HareketTipi, IplikFatura.EFatura from IplikFatura
        left join Firma On Firma.FirmaId = IplikFatura.FirmaId
        left join Kullanici TK On TK.KullaniciId = IplikFatura.TransferKullaniciId
        where IplikFatura.FaturaTarihi >= :ABaslangicTarihi and IplikFatura.FaturaTarihi <= :ABitisTarihi and IplikFatura.HareketTipi = 0 and
              IplikFatura.Sinif = :ASinif and
              ((:ATransferDurumu = 0 and IplikFatura.TransferEdildi = 0) or (:ATransferDurumu = 1 and IplikFatura.TransferEdildi <> 0) or :ATransferDurumu is null)
              and IplikFatura.Entegrasyon = 1
      order by IplikFatura.SeriNo, IplikFatura.FaturaNo
      into :IplikFaturaId, :FirmaId, :FirmaAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :TransferEdildi, :TransferTarihi, :TransferKullaniciId,
           :TransferKullaniciAdi, :IplikStokHareketTipiID, :EFatura do
      begin
        if (TransferEdildi > 0) then TransferEdildiStr = 'Yapıldı';
        else TransferEdildiStr = '';

        EFaturaStr = '';
        if (EFatura = 1) then EFaturaStr = 'Evet';

        Fason = 0;

        suspend;
      end
    end
  end
  else if (AEvrakTipi = 1) then
  begin
    HareketTipiId = 0;
    HamStokHareketTipiId = NUll;
    EFatura   = 0;
    EFaturaStr = '';

    for Select BoyaFatura.BoyaFaturaId, BoyaFatura.FirmaId, Firma.Adi, BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo,
               BoyaFatura.FaturaTarihi, BoyaFatura.TransferEdildi, BoyaFatura.TransferTarihi, BoyaFatura.TransferKullaniciId,
               TK.Adi, BoyaFatura.Status, BoyaFatura.IhracatFaturasi, BoyaFatura.Tevkifat, BoyaFatura.EFatura from BoyaFatura
      left join Firma On Firma.FirmaId = BoyaFatura.FirmaId
      left join Kullanici TK On TK.KullaniciId = BoyaFatura.TransferKullaniciId
      left join BoyaFaturaDetay On BoyaFaturaDetay.BoyaFaturaId = BoyaFatura.BoyaFaturaId
      where BoyaFatura.FaturaTarihi >= :ABaslangicTarihi and BoyaFatura.FaturaTarihi <= :ABitisTarihi and BoyaFatura.HareketTipiId = 0 and
            BoyaFatura.Sinif = :ASinif and
           ((:ATransferDurumu = 0 and BoyaFatura.TransferEdildi = 0) or (:ATransferDurumu = 1 and BoyaFatura.TransferEdildi <> 0) or :ATransferDurumu is null)
           and BoyaFaturaDetay.BoyaFaturaDetayId is not null and
            BoyaFatura.Entegrasyon = 1
      group by BoyaFatura.BoyaFaturaId, BoyaFatura.FirmaId, Firma.Adi, BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo,
               BoyaFatura.FaturaTarihi, BoyaFatura.TransferEdildi, BoyaFatura.TransferTarihi, BoyaFatura.TransferKullaniciId,
               TK.Adi, BoyaFatura.Status, BoyaFatura.IhracatFaturasi, BoyaFatura.Tevkifat, BoyaFatura.EFatura
      order by BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo
    into :BoyaFaturaId, :FirmaId, :FirmaAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :TransferEdildi, :TransferTarihi, :TransferKullaniciId,
         :TransferKullaniciAdi, :Status, :IhracatFaturasi, :Tevkifat, :EFatura do
    begin
      if (TransferEdildi > 0) then TransferEdildiStr = 'Yapıldı';
      else TransferEdildiStr = '';

      EFaturaStr = '';
      if (EFatura = 1) then EFaturaStr = 'Evet';

      Select Min(Parti.Fason) from BoyaFaturaDetay
        left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = BoyaFaturaDetay.BoyaIrsaliyeDetayId
        left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
        where BoyaFaturaDetay.BoyaFaturaID = :BoyaFaturaID
        into :Fason;

      suspend;
    end

    HareketTipiId = Null;
    HamStokHareketTipiId = NUll;
    BoyaFaturaId   = Null;
    HamStokFaturaId = Null;
    IplikFaturaId = NUll;
    IhracatFaturasi = Null;
    Tevkifat  = Null;
    EFatura   = 0;
    EFaturaStr = '';

/********************* ÖRME FATURA *****************************/

    for Select OrmeFatura.OrmeFaturaId, OrmeFatura.FirmaId, Firma.Adi, OrmeFatura.SeriNo, OrmeFatura.FaturaNo,
               OrmeFatura.FaturaTarihi, OrmeFatura.TransferEdildi, OrmeFatura.TransferTarihi, OrmeFatura.TransferKullaniciId,
               TK.Adi, OrmeFatura.HareketTipi, OrmeFatura.Tevkifat, OrmeFatura.EFatura from OrmeFatura
      left join Firma On Firma.FirmaId = OrmeFatura.FirmaId
      left join Kullanici TK On TK.KullaniciId = OrmeFatura.TransferKullaniciId
      where OrmeFatura.FaturaTarihi >= :ABaslangicTarihi and OrmeFatura.FaturaTarihi <= :ABitisTarihi and (OrmeFatura.HareketTipi = 2 or OrmeFatura.HareketTipi = 3) and
            OrmeFatura.Sinif = :ASinif and
            (OrmeFatura.TransferEdildi = :ATransferDurumu or :ATransferDurumu is null) and OrmeFatura.Entegrasyon = 1
    order by OrmeFatura.SeriNo, OrmeFatura.FaturaNo
    into :OrmeFaturaId, :FirmaId, :FirmaAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :TransferEdildi, :TransferTarihi, :TransferKullaniciId,
         :TransferKullaniciAdi, :OrmeStokHareketTipiID, :Tevkifat, :EFatura do
    begin
      if (TransferEdildi = 1) then TransferEdildiStr = 'Yapıldı';
      else TransferEdildiStr = '';

      EFaturaStr = '';
      if (EFatura = 1) then EFaturaStr = 'Evet';

      Fason = 0;

      Select Min(BoyaSiparis.Fason) from StokHareket
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = StokHareket.BoyaSiparisKayitID
        where StokHareket.OrmeFaturaId = :OrmeFaturaId
        into :Fason;

      suspend;
    end


    HareketTipiId = Null;
    HamStokHareketTipiId = NUll;
    BoyaFaturaId   = Null;
    HamStokFaturaId = Null;
    IplikFaturaId = NUll;
    OrmeFaturaId = Null;
    IhracatFaturasi = Null;
    Tevkifat  = Null;
    EFatura   = 0;
    EFaturaStr = '';

/********************* İPLİK FATURA *****************************/

    for Select IplikFatura.IplikFaturaId, IplikFatura.FirmaId, Firma.Adi, IplikFatura.SeriNo, IplikFatura.FaturaNo,
               IplikFatura.FaturaTarihi, IplikFatura.TransferEdildi, IplikFatura.TransferTarihi, IplikFatura.TransferKullaniciId,
               TK.Adi, IplikFatura.HareketTipi, IplikFatura.EFatura from IplikFatura
      left join Firma On Firma.FirmaId = IplikFatura.FirmaId
      left join Kullanici TK On TK.KullaniciId = IplikFatura.TransferKullaniciId
      where IplikFatura.FaturaTarihi >= :ABaslangicTarihi and IplikFatura.FaturaTarihi <= :ABitisTarihi and IplikFatura.HareketTipi = 1 and
            IplikFatura.Sinif = :ASinif and
            ((:ATransferDurumu = 0 and IplikFatura.TransferEdildi = 0) or (:ATransferDurumu = 1 and IplikFatura.TransferEdildi <> 0) or :ATransferDurumu is null)
            and IplikFatura.Entegrasyon = 1
    order by IplikFatura.SeriNo, IplikFatura.FaturaNo
    into :IplikFaturaId, :FirmaId, :FirmaAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :TransferEdildi, :TransferTarihi, :TransferKullaniciId,
         :TransferKullaniciAdi, :IplikStokHareketTipiID, :EFatura do
    begin
      if (TransferEdildi > 0) then TransferEdildiStr = 'Yapıldı';
      else TransferEdildiStr = '';

      EFaturaStr = '';
      if (EFatura = 1) then EFaturaStr = 'Evet';

      Fason = 0;

      suspend;
    end
  end
  else if (AEvrakTipi = 2) then
  begin
    HareketTipiId = null;
    for Select HamStokFatura.HamStokFaturaId, HamStokFatura.HamStokHareketTipiId, HamStokFatura.FirmaId, Firma.Adi, HamStokFatura.FaturaSeriNo, HamStokFatura.FaturaNo,
               HamStokFatura.FaturaTarihi, HamStokFatura.TransferEdildi, HamStokFatura.TransferTarihi, HamStokFatura.TransferKullaniciId,
               TK.Adi from HamStokFatura
      left join Firma On Firma.FirmaId = HamStokFatura.FirmaId
      left join Kullanici TK On TK.KullaniciId = HamStokFatura.TransferKullaniciId
      where HamStokFatura.FaturaTarihi >= :ABaslangicTarihi and HamStokFatura.FaturaTarihi <= :ABitisTarihi and
           (HamStokFatura.TransferEdildi = :ATransferDurumu or :ATransferDurumu is null) and
            HamStokFatura.HamStokHareketTipiId = 1
      order by HamStokFatura.FaturaTarihi, HamStokFatura.FaturaSeriNo, HamStokFatura.FaturaNo
    into :HamStokFaturaId, :HamStokHareketTipiId, :FirmaId, :FirmaAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :TransferEdildi, :TransferTarihi, :TransferKullaniciId,
         :TransferKullaniciAdi do
    begin
      if (TransferEdildi = 1) then TransferEdildiStr = 'Yapıldı';
      else TransferEdildiStr = '';
      Fason = Null;
      suspend;
    end
  end


  if (AEvrakTipi = 1 and ATransferDurumu = 0) then
  begin
    if (StokMuhasebeYazilimTipi = 5) then
    begin
      OrmeFaturaId            = null;
      IplikFaturaId            = null;
      OrmeStokHareketTipiID   = null;
      BoyaFaturaId            = null;
      HamStokFaturaId         = null;
      HamStokHareketTipiId    = null;
      HareketTipiId           = null;
      FirmaId                 = null;
      FirmaAdi                = null;
      FaturaSeriNo            = null;
      FaturaNo                = null;
      FaturaTarihi            = null;
      TransferEdildi          = null;
      TransferEdildiStr       = null;
      TransferTarihi          = null;
      TransferKullaniciId     = null;
      TransferKullaniciAdi    = null;
      Fason                   = null;
      Status                  = null;
      IhracatFaturasi         = Null;
      Tevkifat                = Null;
      EFatura   = 0;
      EFaturaStr = '';


      for Select BoyaFaturaIptal.BoyaFaturaId, BoyaFaturaIptal.FirmaId, Firma.Adi, BoyaFaturaIptal.FaturaSeriNo, BoyaFaturaIptal.FaturaNo,
                 BoyaFaturaIptal.FaturaTarihi from BoyaFaturaIptal
        left join Firma On Firma.FirmaId = BoyaFaturaIptal.FirmaId
        where BoyaFaturaIptal.IptalTarihi >= :ABaslangicTarihi and BoyaFaturaIptal.IptalTarihi <= :ABitisTarihi + 1 and BoyaFaturaIptal.HareketTipiId = 0 and
              BoyaFaturaIptal.TransferEdilecek = 1
        order by BoyaFaturaIptal.FaturaSeriNo, BoyaFaturaIptal.FaturaNo
      into :BoyaFaturaId, :FirmaId, :FirmaAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi do
      begin
        IptalKaydi = 1;
        suspend;
      end
    end
  end
end
