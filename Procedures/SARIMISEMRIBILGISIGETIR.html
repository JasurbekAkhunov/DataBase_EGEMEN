CREATE PROCEDURE SARIMISEMRIBILGISIGETIR(
    ASARIMISEMRIKAYITID INTEGER)
RETURNS (
    SARIMISEMRIKAYITID INTEGER,
    SARIMISEMRIYIL INTEGER,
    SARIMISEMRIID INTEGER,
    ISEMRIMIKTARI NUMERIC(12,3),
    IPLIKLOTID INTEGER,
    LOT VARCHAR(30),
    ELYAFTIPIID INTEGER,
    TARIHI TIMESTAMP,
    HARMANKAYITID INTEGER,
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    ETIKETTIPIID INTEGER,
    ETIKETTIPIADI VARCHAR(30),
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    NOTU VARCHAR(250),
    TAMIR INTEGER,
    IPLIKTAMIRTIPIID INTEGER,
    IPLIKTAMIRTIPIADI VARCHAR(20),
    TAMIRGIRISMIKTAR NUMERIC(12,3),
    KAPALI INTEGER,
    IPLIKTALEPDETAYID INTEGER,
    BITISTARIHI TIMESTAMP,
    BITIRENKULLANICIID INTEGER,
    BITIRENKULLANICIADI VARCHAR(40),
    PAKETLEMEETIKETNO INTEGER)
AS
begin
  Select SarimIsEmri.SarimIsEmriKayitId, SarimIsEmri.SarimIsEmriYil, SarimIsEmri.SarimIsEmriId, SarimIsEmri.Miktar, IplikLot.IplikLotId, IplikLot.Lot, IplikLot.ElyafTipiId, IplikLot.Tarihi, IplikLot.HarmanKayitId,
         Harman.HarmanYil, Harman.HarmanId, IplikLot.HamId, Ham.Adi, IplikLot.MelanjKodu, IplikLot.FlamKodu,
         IplikLot.KonikCinsiId, KonikCinsi.Adi, IplikLot.AmbalajId, Ambalaj.Adi,
         IplikLot.EtiketTipiId, EtiketTipi.Adi, EtiketTipi.PaketlemeEtiketNo, IplikLot.Notu, IplikLot.Tamir, IplikLot.Kapali,
         IplikSatisIade.IplikTamirTipiId, IplikTamirTipi.Adi, TamirGiris.Miktar,
         IplikSatisIade.RenkId, Renk.Adi, IplikLot.IplikTalepDetayId,
         SarimIsEmri.Bitistarihi, SarimIsEmri.BitirenKullaniciId, BK.Adi from SarimIsEmri
    left join IplikLot on IplikLot.IplikLotID = SarimIsEmri.IplikLotId
    left join Ham on Ham.HamID = IplikLot.HamId
    left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitId
    left join EtiketTipi on EtiketTipi.EtiketTipiId = IplikLot.EtiketTipiId
    left join Ambalaj on Ambalaj.AmbalajId = IplikLot.AmbalajId
    left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikLot.KonikCinsiId
    left join TamirGiris on TamirGiris.TamirGirisKayitID = IplikLot.TamirGirisKayitId
    left join IplikSatisIade on IplikSatisIade.IplikSatisIadeID = TamirGiris.IplikSatisIadeId
    left join IplikTamirTipi on IplikTamirTipi.IplikTamirTipiID = IplikSatisIade.IplikTamirTipiId
    left join Renk on Renk.RenkID = IplikSatisIade.RenkId
    left join Kullanici BK on BK.KullaniciID = SarimIsEmri.BitirenKullaniciId
    where SarimIsEmri.SarimIsEmriKayitID = :ASarimIsEmriKayitID
    into :SarimIsEmriKayitId, :SarimIsEmriYil, :SarimIsEmriId, :IsEmriMiktari, :IplikLotId, :Lot, :ElyafTipiId,
         :Tarihi, :HarmanKayitId, :HarmanYil, :HarmanId, :HamId, :HamAdi, :MelanjKodu, :FlamKodu,
         :KonikCinsiId, :KonikCinsiAdi, :AmbalajId, :AmbalajAdi,
         :EtiketTipiId, :EtiketTipiAdi, :PaketlemeEtiketNo, :Notu, :Tamir, :Kapali,
         :IplikTamirTipiId, :IplikTamirTipiAdi, :TamirGirisMiktar, :RenkId, :RenkAdi, :IplikTalepDetayId,
         :Bitistarihi, :BitirenKullaniciId, BitirenKullaniciAdi;

  if (IplikLotId is not null) then
  begin
    if (HarmanYil is not null) then
      HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') ' || EgemenIntegerToStr(HarmanId) || ' ';
    suspend;
  end
end
