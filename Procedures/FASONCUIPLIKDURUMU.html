CREATE PROCEDURE FASONCUIPLIKDURUMU(
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    AFIRMAARALIKLI INTEGER,
    AKALINLIKID INTEGER,
    AIPLIKCINSIID INTEGER,
    ATALIMATBITTI INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ORMESIPARISDETAYID INTEGER,
    ORMEPLANLAMAID INTEGER,
    URUNADI VARCHAR(255),
    URUNADI2 VARCHAR(255),
    URUNADI3 VARCHAR(255),
    URUNADI4 VARCHAR(255),
    URETIMPARTINO VARCHAR(30),
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIIDSTR VARCHAR(20),
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    KULLANIMORANI INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PUS INTEGER,
    FEINE INTEGER,
    HAMIADE INTEGER,
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    KULLANIMMIKTARI DOUBLE PRECISION,
    TAHMINIFIREMIKTARI DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    TAHMINIFIREORANI DOUBLE PRECISION,
    AKTARILANMIKTAR DOUBLE PRECISION,
    HAMMIKTAR DOUBLE PRECISION,
    SONALISFIYATI DOUBLE PRECISION,
    SONSATISFIYATI DOUBLE PRECISION,
    SONURETIMECIKISFIYATI DOUBLE PRECISION,
    ORTALAMAFASONUCIKISFIYATI DOUBLE PRECISION,
    TALIMATTAHMINIFIREMIKTARI DOUBLE PRECISION,
    TAHMINIRECETEFIREMIKTARI DOUBLE PRECISION,
    RECETEFIREORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    TALIMATRECETEFIREORANI DOUBLE PRECISION)
AS
declare Variable Gonderildi           integer;
declare variable FBoyaSiparisKayitId  integer;
declare variable FBoyaSiparisYil      Integer;
declare variable FBoyaSiparisId       Integer;
declare variable FBoyaSiparisIdStr    varchar(30);
declare variable FOrmeSiparisDetayId  integer;
declare variable FUrunAdi              varchar(255);
declare variable FOrmePlanlamaId        integer;
declare variable FUretimPartiNo         varchar(30);
declare variable FHamId                 integer;
declare variable FHamTipiId            integer;
declare variable FHamTipiIdStr         varchar(20);
declare variable FHamAdi               varchar(255);
declare variable FRenkId               integer;
declare variable FRenkAdi              varchar(50);
declare variable FKullanimOrani         integer;
declare variable FPus                   integer;
declare variable FFeine                 integer;
declare variable FHamIade                integer;
declare variable FFirmaID               varchar(20);
declare variable FFirmaAdi              varchar(100);
declare variable FLot                   varchar(150);
declare variable FGerceklesenMiktar     Double precision;
declare variable FCikisMiktari          Double precision;
declare variable FIadeMiktari           Double precision;
declare variable FKullanimMiktari       Double precision;
declare variable FTahminiFireMiktari    Double precision;
declare variable FKalanMiktar           Double precision;
declare variable FTahminiFireOrani      Double precision;
declare Variable AktarilacakMiktar      Double Precision;
declare Variable FHamMiktar             Double Precision;
declare Variable FSonAlisFiyati         Double precision;
declare Variable FSonSatisFiyati        Double precision;
declare Variable FSonUretimeCikisFiyati       Double precision;
declare Variable FOrtalamaFasonUCikisFiyati   Double precision;
declare variable FTalimatTahminiFireMiktari   double precision;
declare variable FTahminiReceteFireMiktari    double precision;
declare variable FReceteFireOrani             double precision;
declare variable FKullanilacakMiktar          double precision;
declare variable FTalimatReceteFireOrani      double precision;
begin
  Gonderildi = 1;
  AktarilacakMiktar = 0;
  AktarilanMiktar   = 0;

  For select BoyaSiparisKayitId, BoyaSiparisYil, BoyaSiparisId, BoyaSiparisIdStr, OrmeSiparisDetayId, UrunAdi, OrmePlanlamaId, UretimPartiNo,
             HamId, HamTipiId, HamTipiIdStr, HamAdi, RenkId, RenkAdi, Lot, UreticiId, UreticiAdi, KullanimOrani, FirmaId, FirmaAdi, CikisMiktari, IadeMiktari, GerceklesenMiktar,
             KullanimMiktari, TahminiFireMiktari, KalanMiktar, TahminiFireOrani, HamAdi2, HamAdi3, HamAdi4, UrunAdi2, UrunAdi3, UrunAdi4, Pus, Feine, HamMiktar, SonAlisFiyati,                                                                                                                                                                                                                    
             SonSatisFiyati, SonUretimeCikisFiyati, OrtalamaFasonUCikisFiyati, Hamiade,
             TalimatTahminiFireMiktari , TahminiReceteFireMiktari , ReceteFireOrani , KullanilacakMiktar , TalimatReceteFireOrani
               from FasoncuIplikDurumuHazirla(:ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ABaslangicFirmaId, :ABitisFirmaId, :AFirmaAralikli, :ATalimatBitti)
   where (FasoncuIplikDurumuHazirla.KalinlikId = :AKalinlikId or :AKalinlikId is null) and
         (FasoncuIplikDurumuHazirla.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is null)
   order by OrmeSiparisDetayId, FirmaId, HamId, Ormeplanlamaid
   into :FBoyaSiparisKayitId, :FBoyaSiparisYil, :FBoyaSiparisId, :FBoyaSiparisIdStr, :FOrmeSiparisDetayId, :FUrunAdi, :FOrmePlanlamaId, :FUretimPartiNo,
        :FHamId, :FHamTipiId, :FHamTipiIdStr, :FHamAdi, :FRenkId, :FRenkAdi, :FLot, :UreticiId, :UreticiAdi,
        :FKullanimOrani, :FFirmaId, :FFirmaAdi, :FCikisMiktari, :FIadeMiktari, :FGerceklesenMiktar,
        :FKullanimMiktari, :FTahminiFireMiktari, :FKalanMiktar, :FTahminiFireOrani, :HamAdi2, :HamAdi3, :HamAdi4, :UrunAdi2, :UrunAdi3, :UrunAdi4,
        :FPus, :FFeine, :FHamMiktar, :FSonAlisFiyati, :FSonSatisFiyati, :FSonUretimeCikisFiyati, :FOrtalamaFasonUCikisFiyati, :FHamIade,
        :FTalimatTahminiFireMiktari , :FTahminiReceteFireMiktari , :FReceteFireOrani , :FKullanilacakMiktar , :FTalimatReceteFireOrani  do
  begin
    if (Gonderildi = 0) then
    begin
      if (KalanMiktar > 0) then
      if (FHamId = HamId and FOrmeSiparisDetayId = OrmeSiparisDetayId and FFirmaId = FirmaId and StrKarsilastir(Lot, FLot) = 1 ) then
      begin
        AktarilacakMiktar = KalanMiktar;
        KalanMiktar = 0;
      end
      suspend;
      Gonderildi = 1;
    end

    BoyaSiparisKayitId  = FBoyaSiparisKayitId;
    BoyaSiparisYil      = FBoyaSiparisYil;
    BoyaSiparisId       = FBoyaSiparisId;
    BoyaSiparisIdStr    = FBoyaSiparisIdStr;
    OrmeSiparisDetayId  = FOrmeSiparisDetayId;
    OrmePlanlamaId      = FOrmePlanlamaId;
    UrunAdi             = FUrunAdi;
    UretimPartiNo       = FUretimPartiNo;
    HamId               = FHamId;
    HamTipiId           = FHamTipiId;
    HamTipiIdStr        = FHamTipiIdStr;
    HamAdi              = FHamAdi;
    RenkId              = FRenkId;
    RenkAdi             = FRenkAdi;
    KullanimOrani       = FKullanimOrani;
    FirmaId             = FFirmaId;
    Pus                   = FPus;
    Feine                 = FFeine;
    HamIade               = FHamIade;
    FirmaAdi              = FFirmaAdi;
    CikisMiktari          = FCikisMiktari;
    IadeMiktari           = FIadeMiktari;
    GerceklesenMiktar     = FGerceklesenMiktar;
    KullanimMiktari       = FKullanimMiktari;
    TahminiFireMiktari    = FTahminiFireMiktari;
    AktarilanMiktar       = AktarilacakMiktar;
    KalanMiktar           = FKalanMiktar + AktarilanMiktar;
    TahminiFireOrani      = FTahminiFireOrani;
    AktarilacakMiktar = 0;
    HamMiktar             = FHamMiktar;
    SonAlisFiyati            = FSonAlisFiyati;
    SonSatisFiyati           = FSonSatisFiyati;
    SonUretimeCikisFiyati    = FSonUretimeCikisFiyati;
    OrtalamaFasonUCikisFiyati = FOrtalamaFasonUCikisFiyati;
    Lot = FLot;
    Gonderildi = 0;

    TalimatTahminiFireMiktari  = FTalimatTahminiFireMiktari;
    TahminiReceteFireMiktari   = FTahminiReceteFireMiktari;
    ReceteFireOrani            = FReceteFireOrani;
    KullanilacakMiktar         = FKullanilacakMiktar;
    TalimatReceteFireOrani     = FTalimatReceteFireOrani;

  end
  if (Gonderildi = 0) then suspend;
end
