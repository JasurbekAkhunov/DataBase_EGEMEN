              BoyaSiparisDetayID is not null and HareketTipi = :shtSatisIade

            BoyaliSatisEksigi = (ToplamBoyaliSiparisMiktari + BoyaliSatisIade) - ToplamBoyaliSatis;

            BukumGerceklesenMiktar = null;
            Select Sum(BukumTalimat.GerceklesenMiktar) From BukumTalimat where BukumTalimat.OrmeSiparisDetayId = :OrmeSiparisDetayID into :BukumGerceklesenMiktar;

            PlanlanmayanMiktar = (:HamMiktar - :PlanlananMiktar);
            suspend;
          end
