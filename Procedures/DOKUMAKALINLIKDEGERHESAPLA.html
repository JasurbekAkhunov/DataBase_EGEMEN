CREATE PROCEDURE DOKUMAKALINLIKDEGERHESAPLA(
    ADENYE DOUBLE PRECISION,
    AKALINLIKNE DOUBLE PRECISION)
RETURNS (
    NM DOUBLE PRECISION)
AS
begin
  Nm = null;
  if (AKalinlikNe is not null) then
    Nm =  AKalinlikNe * 1.693 ;

end
