CREATE PROCEDURE BOYATALIMATRECETEGETIR(
    ABOYASIPARISKAYITID INTEGER,
    ABOYATALIMATKAYITID INTEGER)
RETURNS (
    BOYATALIMATRECETEID INTEGER,
    BOYATALIMATKAYITID INTEGER,
    ORMERAFID INTEGER,
    RAFID RAFKODUSIZE,
    MIKTAR BOYAMIKTARSIZE,
    KULLANILANMIKTAR BOYAMIKTARSIZE,
    KALANMIKTAR BOYAMIKTARSIZE,
    RAFKALANMIKTAR BOYAMIKTARSIZE,
    RAFKULLANILABILIRMIKTAR BOYAMIKTARSIZE,
    BASKATALIMATRECETEMIKTAR BOYAMIKTARSIZE,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
Begin
  For Select BoyaTalimatRecete.BoyaTalimatReceteId, BoyaTalimatRecete.BoyaTalimatKayitID, BoyaTalimatRecete.OrmeRafID, Raf.RafID,
             BoyaTalimatRecete.Miktar, BoyaTalimatRecete.KullanilanMiktar, (BoyaTalimatRecete.Miktar - BoyaTalimatRecete.KullanilanMiktar),
             IK.Adi, BoyaTalimatRecete.InsertTarihi, Sk.Adi, BoyaTalimatRecete.DegisimTarihi, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari From BoyaTalimat
  Left Join BoyaTalimatRecete on BoyaTalimatRecete.BoyaTalimatKayitID = BoyaTalimat.BoyaTalimatKayitID
  Left Join OrmeRaf on OrmeRaf.OrmeRafId = BoyaTalimatRecete.OrmeRafId
  Left Join Raf on Raf.RafId = OrmeRaf.RafId
  Left Join Kullanici IK on IK.KullaniciId = BoyaTalimatRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = BoyaTalimatRecete.KullaniciId
  where BoyaTalimat.BoyaSiparisKayitID = :ABoyaSiparisKayitID and
        (BoyaTalimat.BoyaTalimatKayitID = :ABoyaTalimatKayitID or :ABoyaTalimatKayitID is Null)
  Into :BoyaTalimatReceteId, :BoyaTalimatKayitID, :OrmeRafID, :RafID, :Miktar, :Kullanilanmiktar, :KalanMiktar,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :RafKalanMiktar do
  begin

    Select Sum(Miktar - KullanilanMiktar) from BoyaTalimatRecete where BoyaTalimatRecete.OrmeRafID = :OrmeRafId and BoyaTalimatRecete.BoyaTalimatReceteID <> :BoyaTalimatReceteID
      into :BaskaTalimatReceteMiktar;
    if (BaskaTalimatReceteMiktar is Null) then BaskaTalimatReceteMiktar = 0;

    RafKullanilabilirMiktar = RafKalanMiktar - BaskaTalimatReceteMiktar;

    Suspend;
  end
end
