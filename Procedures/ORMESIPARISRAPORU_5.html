CREATE PROCEDURE ORMESIPARISRAPORU_5(
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    AIPLIKSIPARISI INTEGER,
    ABOYALIIPLIKSIPARISI INTEGER,
    ALYCRASIPARISI INTEGER,
    AHAMKUMASSIPARISI INTEGER,
    ABOYALIKUMASSIPARISI INTEGER,
    AIPLIKBOYALIKUMASSIPARISI INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AKUMASCINSIID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    AYURTDISI INTEGER,
    AKUMASTEMINSIPARISI INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ASINIF VARCHAR(1),
    AKALITEID VARCHAR(20),
    ASIPONAYDURUMU SMALLINT,
    AMALIONAYDURUMU SMALLINT,
    AKAPATMATARIHIBASLANGIC TIMESTAMP,
    AKAPATMATARIHIBITIS TIMESTAMP,
    AIMALATTIPIID INTEGER)
RETURNS (
    ORMEPLANLAMAID INTEGER,
    TARIH TIMESTAMP,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    SIRA INTEGER,
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    SAKATURETIMTOPLAMI DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    IPLIKFIREORANI DOUBLE PRECISION,
    LYCRAKULLANIMORANI DOUBLE PRECISION,
    MIKTARGUN DOUBLE PRECISION,
    BASLAMATARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    URETIMPARTINO VARCHAR(30),
    MAKINAUSTUGRAMAJ INTEGER,
    DINLENMISGRAMAJ INTEGER,
    IGNESAYISI INTEGER,
    MAYSAYISI VARCHAR(10),
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    SIPTUPACIKENMAYLI INTEGER,
    SIPTUPACIKENMAYLISTR VARCHAR(10),
    ORMEYECIKANIPLIK DOUBLE PRECISION,
    ORMEDENIADEIPLIK DOUBLE PRECISION,
    ORMEYECIKANLYCRA DOUBLE PRECISION,
    ORMEDENIADELYCRA DOUBLE PRECISION,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LOT VARCHAR(50),
    BITTI INTEGER,
    BITTISTR VARCHAR(5),
    BEKLEMEDE INTEGER,
    BEKLEMEDESTR VARCHAR(9),
    MAKINASAYISI INTEGER,
    TERMINTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    HAMID INTEGER,
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable Iplik        Double Precision;
declare variable Lycra        Double Precision;
declare variable FHamTipiId        integer;
declare variable FCikisMiktari     Double Precision;
declare variable FIadeMiktari      Double Precision;
declare variable YuvarlamaCarpani  Double Precision;
declare variable uitBoya           integer;
declare variable FKalanMiktar      Double Precision;
declare variable Sure              Double Precision;
declare variable FSure             Double Precision;
declare variable FPlanlananMiktar    BoyaMiktarSize;
declare variable FGerceklesenMiktar  BoyaMiktarSize;
declare variable FSakatUretimToplami BoyaMiktarSize;
declare variable FMiktarGun          BoyaMiktarSize;
declare variable BitirmeTarihi       TimeStamp;
begin
  select uitBoya from UretimIslemTipiGetir into :uitBoya;

  For Select OrmeSiparisRaporu_1.BoyaSiparisKayitId, OrmeSiparisRaporu_1.OrmeSiparisDetayId,
             OrmePlanlama.OrmePlanlamaId, OrmePlanlama.Tarih, OrmePlanlama.FirmaId, OrmePlanlama.OrmeMakinaId, OrmePlanlama.Sira,
              OrmePlanlama.PlanlananMiktar, OrmePlanlama.GerceklesenMiktar, OrmePlanlama.ToplamSakatOrulen,
              OrmePlanlama.MiktarGun, OrmePlanlama.BaslamaTarihi, OrmePlanlama.BirimFiyat, OrmePlanlama.DovizId, OrmePlanlama.VadeGun,
              OrmePlanlama.UretimPartiNo, OrmePlanlama.MakinaUstuGramaj, OrmePlanlama.DinlenmisGramaj, OrmePlanlama.IgneSayisi,
              OrmePlanlama.MaySayisi, OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4,
              OrmePlanlama.TupAcikEnMayli, OrmePlanlama.Lot, OrmePlanlama.Notu, OrmePlanlama.Bitti, Firma.Adi, OrmeMakina.Adi, OrmePlanlama.Beklemede,
              Ik.Adi, OrmePlanlama.InsertTarihi, Sk.Adi, OrmePlanlama.DegisimTarihi,
              OrmeSiparisRaporu_1.TupAcikEnMayli, OrmeSiparisRaporu_1.TupAcikEnMayliStr, OrmeSiparisRaporu_1.SiparisSiraNo, OrmeSiparisRaporu_1.HamID, OrmePlanlama.MakinaSayisi, OrmePlanlama.TerminTarihi,
              OrmePlanlama.BitirmeTarihi from OrmeSiparisRaporu_1(:ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ABaslangicSiparisTarihi, :ABitisSiparisTarihi, :AFirmaId,
              :AFASONSIPARISLER, :AFASONSATISSIPARISLER, :ASATISSIPARISLER, :AIplikSiparisi, :ABoyaliIplikSiparisi, :ALycraSiparisi, :AHamKumasSiparisi, :ABoyaliKumasSiparisi,
              :AIplikBoyaliKumasSiparisi, :AMusteriSiparisNo, :AReferansSiparisNo, :ARaporTipi, :AKapali, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :ASiparisiAlanKullaniciId,
              :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId, :AYurtDisi, :AKumasTeminSiparisi, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASinif, :AKaliteId, :ASipOnayDurumu, :AMaliOnayDurumu, :AKapatmaTarihiBaslangic, :AKapatmaTarihiBitis, :AImalatTipiID)
    left join OrmePlanlama on OrmePlanlama.OrmeSiparisDetayId = OrmeSiparisRaporu_1.OrmeSiparisDetayId
    left join Firma on Firma.FirmaId = OrmePlanlama.FirmaId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
    Left Join Kullanici IK on IK.KullaniciId = OrmePlanlama.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = OrmePlanlama.KullaniciId
    where OrmePlanlama.OrmePlanlamaID is not null
    into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :OrmePlanlamaId, :Tarih, :FirmaId, :OrmeMakinaId, :Sira, :PlanlananMiktar, :GerceklesenMiktar,
         :SakatUretimToplami, :MiktarGun, :BaslamaTarihi, :BirimFiyat, :DovizId, :VadeGun,
         :UretimPartiNo, :MakinaUstuGramaj, :DinlenmisGramaj, :IgneSayisi, :MaySayisi, :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :TupAcikEnMayli,
         :Lot, :Notu, :Bitti, :FirmaAdi, :OrmeMakinaAdi, :Beklemede, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi,
         :DegisimTarihi, :SipTupAcikEnMayli, :SipTupAcikEnMayliStr, :SiparisSiraNo, :HamID, :MakinaSayisi, :TerminTarihi, :BitirmeTarihi do
  begin
    if (Bitti = 0) then
    begin
      Sure = 0;
      for select OrmePlanlama.PlanlananMiktar, OrmePlanlama.GerceklesenMiktar, OrmePlanlama.ToplamSakatOrulen, OrmePlanlama.MiktarGun
        from OrmePlanlama where OrmePlanlama.BitirenKullaniciId is Null and (OrmePlanlama.OrmeMakinaId = :OrmeMakinaId or OrmePlanlama.FirmaId = :FirmaId) and
        OrmePlanlama.Sira <= :Sira
        order by Sira into :FPlanlananMiktar, :FGerceklesenMiktar, :FSakatUretimToplami, :FMiktarGun do
      begin
        FKalanMiktar = FPlanlananMiktar - FGerceklesenMiktar + FSakatUretimToplami;
        if (FMiktarGun > 0 and FKalanMiktar > 0) then FSure = FKalanMiktar / FMiktarGun;
        else FSure = 0;

        Sure = Sure + FSure;
      end

      if (Sure = 0) then
        BitisTarihi = Current_Timestamp(0);
      else
        BitisTarihi = DoubleToTimeStamp(TimeStampToDouble(Current_Timestamp(0)) + Sure);
    end
    else
        BitisTarihi = BitirmeTarihi;

    if (Bitti = 1) then BittiStr = 'Bitti';
    else BittiStr = '';

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    BeklemedeStr = '';
    if (Beklemede = 1) then BeklemedeStr = 'Beklemede';

    KalanMiktar = PlanlananMiktar - GerceklesenMiktar + SakatUretimToplami;

    Iplik = 0;
    Lycra = 0;
    IplikFireOrani = 0;
    LycraKullanimOrani = 0;

    For Select Ham.HamTipiId, Sum(CikisMiktari), sum(IadeMiktari) from OrmePlanlamaRecete
       left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
       where OrmePlanlamaRecete.OrmePlanlamaId = :OrmePlanlamaId
      group by Ham.HamTipiId
      into :FHamTipiId, :FCikisMiktari, :FIadeMiktari do
    begin
      if (FHamTipiId = 0) then
      begin
        OrmeyeCikanIplik = FCikisMiktari;
        OrmedenIadeIplik = FIadeMiktari;

        Iplik = FCikisMiktari - FIadeMiktari;
      end
      else if (FHamTipiId = 6) then
      begin
        OrmeyeCikanLycra = FCikisMiktari;
        OrmedenIadeLycra = FIadeMiktari;

        Lycra = FCikisMiktari - FIadeMiktari;
      end
    end

    if (GerceklesenMiktar > 0) then
    begin

      IplikFireOrani = ((Iplik + Lycra - GerceklesenMiktar) / GerceklesenMiktar) * 100;
      LycraKullanimOrani = Lycra / GerceklesenMiktar * 100;

      YuvarlamaCarpani = 100;
      IplikFireOrani = Yuvarla(IplikFireOrani, YuvarlamaCarpani);
      LycraKullanimOrani = Yuvarla(LycraKullanimOrani, YuvarlamaCarpani);

    end

    suspend;
  end
end
