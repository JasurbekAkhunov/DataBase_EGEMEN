CREATE PROCEDURE HAMSTOKSARFFISIDETAYGETIR(
    AHAMSTOKSARFFISIID INTEGER)
RETURNS (
    HAMSTOKSARFFISIDETAYID INTEGER,
    HAMSTOKMASTERID INTEGER,
    RAFDETAYID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISNOSTR VARCHAR(30),
    HAMADI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    KAPADEDI INTEGER,
    RAFID VARCHAR(10),
    LOT VARCHAR(150),
    GIRISIRSALIYENO IRSALIYENOSIZE,
    GIRISIRSALIYETARIHI TIMESTAMP,
    GIRISEVRAKTIPI INTEGER,
    GIRISEVRAKTIPISTR VARCHAR(10),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSKODU HAMKODUSIZE,
    YARIMAMULPARTIIDSTR VARCHAR(30),
    DESENVARYANTADI VARCHAR(30),
    KALITEADI ADI255,
    RECETEID RECETEKODUSIZE,
    DESENID DESENKODUSIZE,
    SERTIFIKAYISTOKTANDUS INTEGER)
AS
declare variable PartiFirmaId  varchar(20);
declare variable PartiFirmaAdi varchar(100);
declare variable PartiFason    integer;
declare variable FPartiYil      integer;
declare variable FPartiId       integer;
begin
  For Select HamStokSarfFisiDetay.HamStokSarfFisiDetayId, HamStokDetay.HamStokMasterId, HamStokSarfFisiDetay.RafDetayId,
             Parti.PartiYil, Parti.PartiId, HamStokSarfFisiDetay.PartiSinifId, Ham.Adi, HamStokSarfFisiDetay.Kilo,
             HamStokSarfFisiDetay.Metre, HamStokSarfFisiDetay.BrutKilo, HamStokSarfFisiDetay.BrutMetre, RafDetay.RafId, HamStokMaster.Lot,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, HamStokIrsaliye.FirmaId, HFirma.Adi,
             Parti.FirmaId, PFirma.Adi, HamStokIrsaliye.Fason, Parti.Fason, Ik.Adi, HamStokSarfFisiDetay.InsertTarihi, Sk.Adi,
             HamStokSarfFisiDetay.DegisimTarihi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, HamStokSarfFisiDetay.KapAdedi,
             Ham.ReferansKodu, P.PartiYil, P.PartiId, DesenVaryant.VaryantAdi, Kalite.KaliteAdi, Parti.ReceteId, BoyaSiparisDetay.DesenId, HamStokSarfFisiDetay.SertifikayiStoktanDus
             from HamStokSarfFisiDetay
  left join PartiStok on PartiStok.PartiSinifId = HamStokSarfFisiDetay.PartiSinifId
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  left join RafDetay on RafDetay.RafDetayID = HamStokSarfFisiDetay.RafDetayId
  left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join PartiStok PS on PS.PartiSinifID = HamStokMaster.PartiSinifId
  left join Parti P on P.PartiKayitId = PS.PartiKayitId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = HamStokDetay.HamStokIrsaliyeId
  left join Firma PFirma on PFirma.FirmaId = Parti.FirmaId
  left join Firma HFirma on HFirma.FirmaId = HamStokIrsaliye.FirmaId
  left join Ham on (Ham.HamID = HamStokDetay.HamId or Ham.HamID = PartiStok.HamId)
  Left Join Kullanici IK on IK.KullaniciId = HamStokSarfFisiDetay.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HamStokSarfFisiDetay.KullaniciId
  Left Join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  Left Join Kalite on Kalite.KaliteId = PartiStok.KaliteId
  Left Join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  where HamStokSarfFisiDetay.HamStokSarfFisiId = :AHamStokSarfFisiId
  into :HamStokSarfFisiDetayId, :HamStokMasterId, :RafDetayId, :PartiYil, :PartiId, :PartiSinifId, :HamAdi, :Kilo,
       :Metre, :BrutKilo, :BrutMetre, :RafId, :Lot,
       :GirisIrsaliyeNo, :GirisIrsaliyeTarihi, :GirisEvrakTipi, :FirmaId, :FirmaAdi,
       :PartiFirmaId, :PartiFirmaAdi, :Fason, :PartiFason, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi,
       :DegisimTarihi, :BoyaSiparisYil, :BoyaSiparisID, :KapAdedi,
       :ReferansKodu, :FPartiYil, :FPartiId, :DesenVaryantAdi, :KaliteAdi, :ReceteId, :DesenId, :SertifikayiStoktanDus do
  begin
    if (BoyaSiparisYil is not null) then BoyaSiparisNoStr = '(' || BoyaSiparisYil || ') ' || BoyaSiparisId || ' ';
    else BoyaSiparisNoStr = '';

    PartiIdStr = '';
    if (PartiId is not null) then
    begin
      PartiIdStr = '(' || PartiYil || ') '  || PartiId;
      FirmaID = PartiFirmaId;
      FirmaAdi = PartiFirmaAdi;
      Fason = PartiFason;
    end

    YariMamulPartiIdStr = '';
    if (FPartiId is not null) then
    begin
      YariMamulPartiIdStr = '(' || FPartiYil || ') '  || FPartiId;
    end

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (GirisEvrakTipi is null) then GirisEvrakTipiStr = '';
    else if (GirisEvrakTipi = 0) then GirisEvrakTipiStr = 'Fiş';
    else if (GirisEvrakTipi = 1) then GirisEvrakTipiStr = 'İrsaliye';
    else GirisEvrakTipiStr = 'HATALI!!';

    suspend;
  end
end
