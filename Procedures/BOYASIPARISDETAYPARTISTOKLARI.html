CREATE PROCEDURE BOYASIPARISDETAYPARTISTOKLARI(
    ABOYASIPARISDETAYID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    HAMADITOPLAMI VARCHAR(255),
    DURUMUSTR VARCHAR(20),
    BOBINSAYISI INTEGER,
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    PROSESID INTEGER,
    PROSESADI VARCHAR(255),
    MEVCUTBIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    PARTIPAKETTENFIYATLANDIRILDI INTEGER,
    PARTIPAKETTENFIYATLANDIRILDISTR VARCHAR(4),
    FIYATONAYLAYANKULLANICIID INTEGER,
    FIYATONAYLAYANKULLANICIADI VARCHAR(40),
    FIYATONAYTARIHI TIMESTAMP,
    FASONBIRIMFIYATI DOUBLE PRECISION,
    FASONBIRIMFIYATDOVIZID VARCHAR(5))
AS
begin
  for select PartiStok.PartiSinifId, PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.HamAdiToplami, PartiDurum.Adi,
        PartiStok.BobinSayisi, PartiStok.KapAdedi, PartiStok.Kilo, PartiStok.Metre, PartiStok.Adet, PartiStok.ProsesId, Proses.Adi,
        PartiStok.BirimFiyat, PartiStok.DovizId, PartiStok.PartiPakettenFiyatlandirildi, PartiStok.FiyatOnaylayanKullaniciId, FOK.Adi,
        PartiStok.FiyatOnayTarihi, PartiStok.FasonBirimFiyati, PartiStok.FasonBirimFiyatDovizID from PartiStok
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
  left join Proses on Proses.ProsesId = PartiStok.ProsesId
  left join Kullanici FOK on FOK.KullaniciId = PartiStok.FiyatOnaylayanKullaniciId
  where PartiStok.BoyaSiparisDetayId = :ABoyaSiparisDetayId and PartiStok.PartiStokIptalKayitID is null
  order by Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId
  into :PartiSinifId, :PartiKayitId, :PartiYil, :PartiId, :HamAdiToplami, :DurumuStr, :BobinSayisi, :KapAdedi, :Kilo, :Metre,
       :Adet, :ProsesId, :ProsesAdi, :MevcutBirimFiyat, :DovizId,
       :PartiPakettenFiyatlandirildi, :FiyatOnaylayanKullaniciId, :FiyatOnaylayanKullaniciAdi,
       :FiyatOnayTarihi, :FasonBirimFiyati, :FasonBirimFiyatDovizID do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (PartiPakettenFiyatlandirildi = 1) then PartiPakettenFiyatlandirildiStr = 'Evet';
    suspend;
  end
end
