CREATE PROCEDURE PARTIASAMAFIYATLARIDUZENLE(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIASAMALARIID INTEGER,
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    RECETEID VARCHAR(20),
    INSERTTARIHI TIMESTAMP,
    ESKIURETIMMALIYETI DOUBLE PRECISION,
    YENIURETIMMALIYETI DOUBLE PRECISION,
    ESKIISCILIKURETIMMALIYETI DOUBLE PRECISION,
    YENIISCILIKURETIMMALIYETI DOUBLE PRECISION,
    ESKIURETIMMALIYETIKG DOUBLE PRECISION,
    YENIURETIMMALIYETIKG DOUBLE PRECISION,
    ESKIISCILIKURETIMMALIYETIKG DOUBLE PRECISION,
    YENIISCILIKURETIMMALIYETIKG DOUBLE PRECISION,
    ESKIURETIMMALIYETIDOVIZID VARCHAR(5),
    ESKIURETIMMALIYETIKGDOVIZID VARCHAR(5),
    YENIURETIMMALIYETIDOVIZID VARCHAR(5),
    YENIURETIMMALIYETIKGDOVIZID VARCHAR(5),
    ESKIISCILIKURETIMMDOVIZID VARCHAR(5),
    ESKIISCILIKURETIMMKGDOVIZID VARCHAR(5),
    YENIISCILIKURETIMMDOVIZID VARCHAR(5),
    YENIISCILIKURETIMMKGDOVIZID VARCHAR(5))
AS
declare variable DefaultBirimCarpan     double precision;
declare variable KalanMaliyet     double precision;
declare variable KalanMiktar     double precision;
declare variable DovizKuru       double precision;
begin

  for select PartiAsamalari.PartiAsamalariId, PartiAsamalari.AsamaId, Asama.Adi, PartiAsamalari.ReceteId, PartiAsamalari.UretimMaliyeti, PartiAsamalari.UretimMaliyetiDovizId,
        PartiAsamalari.IscilikUretimMaliyeti, PartiAsamalari.IscilikUretimMaliyetiDovizId, PartiAsamalari.IscilikUretimMaliyetiKg, PartiAsamalari.IscilikUretimMaliyetiKgDovizId,
        PartiAsamalari.UretimMaliyetiKg, PartiAsamalari.UretimMaliyetiKgDovizId, PartiAsamalari.InsertTarihi, Parti.PartiYil, Parti.PartiId, Parti.PartiKayitId from PartiAsamalari
    left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
    left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
    left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
    where PartiAsamalari.InsertTarihi >= :ABaslangicTarihi and PartiAsamalari.InsertTarihi <= :ABitisTarihi
    into :PartiAsamalariId, :AsamaId, :AsamaAdi, :ReceteId, :EskiUretimMaliyeti, :EskiUretimMaliyetiDovizId,
        :EskiIscilikUretimMaliyeti, :EskiIscilikUretimMDovizId, :EskiIscilikUretimMaliyetiKg, :EskiIscilikUretimMKgDovizId,
        :EskiUretimMaliyetiKg, :EskiUretimMaliyetiKgDovizId, :InsertTarihi, :PartiYil, :PartiId, :PartiKayitId do
  begin

    if (:AsamaId is not null) then
    begin
      Select Asama.UretimMaliyeti, Asama.UretimMaliyetiDovizId, Asama.UretimMaliyetiKg, Asama.UretimMaliyetiKgDovizId,
             Asama.IscilikUretimMaliyeti, Asama.IscilikUretimMaliyetiDovizId, Asama.IscilikUretimMaliyetiKg, Asama.IscilikUretimMaliyetiKgDovizId from Asama
        where Asama.AsamaId = :AsamaId
        into :YeniUretimMaliyeti, :YeniUretimMaliyetiDovizId, :YeniUretimMaliyetiKg, :YeniUretimMaliyetiKgDovizId,
             :YeniIscilikUretimMaliyeti, :YeniIscilikUretimMDovizId, :YeniIscilikUretimMaliyetiKg, :YeniIscilikUretimMKgDovizId;
    end
    else
    begin
      Select Asama.UretimMaliyeti, Asama.UretimMaliyetiDovizId, Asama.UretimMaliyetiKg, Asama.UretimMaliyetiKgDovizId,
             Asama.IscilikUretimMaliyeti, Asama.IscilikUretimMaliyetiDovizId, Asama.IscilikUretimMaliyetiKg, Asama.IscilikUretimMaliyetiKgDovizId from Recete
        left join Asama on Asama.AsamaId = Recete.UstAsamaId
        where Recete.ReceteId = :ReceteId
        into :YeniUretimMaliyeti, :YeniUretimMaliyetiDovizId, :YeniUretimMaliyetiKg, YeniUretimMaliyetiKgDovizId,
             :YeniIscilikUretimMaliyeti, :YeniIscilikUretimMDovizId, :YeniIscilikUretimMaliyetiKg, :YeniIscilikUretimMKgDovizId;
    end

    if (EskiUretimMaliyetiDovizId <> YeniUretimMaliyetiDovizId) then
    begin
      select DovizKuru from EnYakinDovizKuruGetir(:EskiUretimMaliyetiDovizId, :YeniUretimMaliyetiDovizId, :InsertTarihi) into :DovizKuru;
      if (DovizKuru is Null) then DovizKuru = 0;
      YeniUretimMaliyeti = YeniUretimMaliyeti * DovizKuru;
    end

    if (EskiUretimMaliyetiKgDovizId <> YeniUretimMaliyetiKgDovizId) then
    begin
      select DovizKuru from EnYakinDovizKuruGetir(:EskiUretimMaliyetiKgDovizId, :YeniUretimMaliyetiKgDovizId, :InsertTarihi) into :DovizKuru;
      if (DovizKuru is Null) then DovizKuru = 0;
      YeniUretimMaliyetiKg = YeniUretimMaliyetiKg * DovizKuru;
    end

    if (EskiIscilikUretimMDovizId <> YeniIscilikUretimMDovizId) then
    begin
      select DovizKuru from EnYakinDovizKuruGetir(:EskiIscilikUretimMDovizId, :YeniIscilikUretimMDovizId, :InsertTarihi) into :DovizKuru;
      if (DovizKuru is Null) then DovizKuru = 0;
      YeniIscilikUretimMaliyeti = YeniIscilikUretimMaliyeti * DovizKuru;
    end

    if (EskiIscilikUretimMKgDovizId <> YeniIscilikUretimMKgDovizId) then
    begin
      select DovizKuru from EnYakinDovizKuruGetir(:EskiIscilikUretimMKgDovizId, :YeniIscilikUretimMKgDovizId, :InsertTarihi) into :DovizKuru;
      if (DovizKuru is Null) then DovizKuru = 0;
      YeniIscilikUretimMaliyetiKg = YeniIscilikUretimMaliyetiKg * DovizKuru;
    end

    update PartiAsamalari set UretimMaliyeti = :YeniUretimMaliyeti, UretimMaliyetiDovizId = :YeniUretimMaliyetiDovizId,
            UretimMaliyetiKg = :YeniUretimMaliyetiKg, UretimMaliyetiKgDovizId = :YeniUretimMaliyetiKgDovizId,
            IscilikUretimMaliyeti = :YeniIscilikUretimMaliyeti, IscilikUretimMaliyetiDovizId = :YeniIscilikUretimMDovizId,
            IscilikUretimMaliyetiKg = :YeniIscilikUretimMaliyetiKg, IscilikUretimMaliyetiKgDovizId = :YeniIscilikUretimMKgDovizId
      where PartiAsamalariId = :PartiAsamalariId;

    Suspend;
  end

end
