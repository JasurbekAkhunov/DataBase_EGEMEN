CREATE PROCEDURE ELYAFKALITEDEGERGIRISRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AIRSALIYENO IRSALIYENOSIZE,
    AHAMID INTEGER,
    AIPLIKDEPOID VARCHAR(10),
    ALOT VARCHAR(30),
    AHAMTIPIID INTEGER,
    AKONTROLDURUMU INTEGER)
RETURNS (
    IPLIKDEPOID VARCHAR(10),
    IPLIKIRSALIYEID INTEGER,
    IPLIKIRSALIYEDETAYID INTEGER,
    IPLIKSTOKID INTEGER,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMTIPIID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    LOT VARCHAR(30),
    IPLIKLOTID INTEGER,
    ELYAFDEGERLENDIRME INTEGER,
    MIKTAR DOUBLE PRECISION,
    NETMIKTAR DOUBLE PRECISION,
    KALITENO INTEGER,
    KALITEACIKLAMATIPIID INTEGER,
    KALITEACIKLAMATIPIADI VARCHAR(20),
    HARMANKAYITID INTEGER,
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    HARMANADI VARCHAR(50),
    HARMANCINSIID INTEGER,
    HARMANCINSIADI VARCHAR(50),
    IPLIKTALEPYIL INTEGER,
    IPLIKTALEPID INTEGER,
    IPLIKTALEPIDSTR VARCHAR(30),
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKRENGIID INTEGER,
    KONIKRENGIADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    ARTIKELNO VARCHAR(20),
    SARIMSEKLIID INTEGER,
    SARIMSEKLIIDSTR VARCHAR(20),
    TAMIRGIRISKAYITID INTEGER,
    TAMIRGIRISYIL INTEGER,
    TAMIRGIRISID INTEGER,
    TAMIRGIRISIDSTR VARCHAR(30),
    TALEPBIRIMMALIYET DOUBLE PRECISION,
    INCELIK DOUBLE PRECISION,
    PARLAKLIK DOUBLE PRECISION,
    SARILIK DOUBLE PRECISION,
    UZUNLUK DOUBLE PRECISION,
    KISAELYAFINDEKSI DOUBLE PRECISION,
    MUKAVEMET DOUBLE PRECISION,
    NEMLILIK DOUBLE PRECISION,
    KONTROLDURUMU INTEGER,
    KONTROLDURUMUSTR VARCHAR(5),
    KONTROLKULLANICIID INTEGER,
    KONTROLKULLANICIADI VARCHAR(40),
    KONTROLTARIHI TIMESTAMP)
AS
declare variable shtAlis integer;
declare variable shtSatisIade integer;
declare variable shtUretim integer;
declare variable shtUretimIade integer;
declare variable shtTelefGirisi integer;
declare variable shtUretimBoya integer;
declare variable shtDepoTransfer integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
declare variable hstHamPamuk     integer;
begin
  Select shtAlis, shtSatisIade, ShtUretim, shtUretimIade, shtTelefGirisi, shtUretimBoya, shtDepoTransfer from IplikStokHareketTipiGetir
         into :shtAlis, :shtSatisIade, :ShtUretim, :shtUretimIade, :shtTelefGirisi, :shtUretimBoya, :shtDepoTransfer;

  Select hstElyaf, hstTelef, hstHamPamuk from HamTipiGetir into :hstElyaf, :hstTelef, :hstHamPamuk;

  if (AHamTipiId = 0) then AHamTipiId = hstElyaf;
  else if (AHamTipiId = 1) then AHamTipiId = hstTelef;
  else AHamTipiId = null;


  for Select IplikIrsaliyeDetay.IplikDepoId, IplikStok.IplikIrsaliyeId, IplikStok.IplikIrsaliyeDetayId, IplikStok.IplikStokId, IplikIrsaliyeDetay.HareketTipi,
         IplikIrsaliye.FirmaId, IrsaliyeFirma.Adi,
         IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, Ham.HamTipiId, IplikIrsaliyeDetay.HamId, Ham.Adi,
         IplikIrsaliyeDetay.RenkId, Renk.Adi, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu,
         IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.IplikLotID, IplikStok.Miktar, IplikStok.NetMiktar, IplikStok.KaliteNo,
         IplikStok.KaliteAciklamaTipiId, KaliteAciklamaTipi.Adi,
         Harman.HarmanKayitId, Harman.HarmanYil, Harman.HarmanId, Harman.Adi, Harman.HarmanCinsiId, HarmanCinsi.Adi,
         IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, IplikIrsaliyeDetay.AmbalajId, Ambalaj.Adi, IplikIrsaliyeDetay.KonikRengiId,
         KonikRengi.Adi, IplikIrsaliyeDetay.KonikCinsiId, KonikCinsi.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
         BoyaSiparis.FirmaId, SiparisFirma.Adi, IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId, IplikTalepDetay.BoyaSiparisDetayId,
         Coalesce(BoyaSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.MusteriSiparisNo),
         Coalesce(BoyaSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.ReferansSiparisNo),
         BoyaSiparisDetay.MusteriKaliteNo, OrmeSiparisDetay.ArtikelNo,
         IplikIrsaliyeDetay.TamirGirisKayitId, TamirGiris.TamirGirisYil, TamirGiris.TamirGirisId, IplikStok.SarimSekliId,
         IplikLot.ElyafDegerlendirme, IplikTalepDetay.BirimMaliyet, IplikStok.Incelik, IplikStok.Parlaklik, IplikStok.Sarilik, IplikStok.Uzunluk,
         IplikStok.KisaElyafIndeksi, IplikStok.Mukavemet, IplikStok.KontrolDurumu, IplikStok.KontrolKullaniciID, KKullanici.Adi,
         IplikStok.KontrolTarihi, IplikStok.Nemlilik  from IplikStok
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.IplikIrsaliyeDetayId
  left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
  left join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = IplikStok.KaliteAciklamaTipiId
  left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
  left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikIrsaliyeDetay.IplikTalepDetayId
  left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
  left join Renk on Renk.RenkId = IplikIrsaliyeDetay.RenkId
  left join KonikRengi on KonikRengi.KonikRengiId = IplikIrsaliyeDetay.KonikRengiId
  left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikIrsaliyeDetay.KonikCinsiId
  left join Ambalaj on Ambalaj.AmbalajId = IplikIrsaliyeDetay.AmbalajId
  left join Harman on Harman.HarmanKayitId = IplikIrsaliyeDetay.HarmanKayitId
  left join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Harman.HarmanCinsiId
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join Firma IrsaliyeFirma on IrsaliyeFirma.FirmaId = IplikIrsaliye.FirmaId
  left join TamirGiris on TamirGiris.TamirGirisKayitId = IplikIrsaliyeDetay.TamirGirisKayitId
  left join IplikLot on IplikLot.Lot = IplikIrsaliyeDetay.Lot and IplikLot.HamID = IplikIrsaliyeDetay.HamId
  left join Kullanici KKullanici on KKullanici.KullaniciId = IplikStok.KontrolKullaniciID
  where IplikStok.CikisIplikIrsaliyeDetayID is null and
       (IplikStok.IplikDepoId = :AIplikDepoId or :AIplikDepoId is null) and
       (IplikStok.HamID = :AHamId or :AHamId is null) and
       (Ham.HamTipiID = :AHamTipiID or (:AHamTipiID is null and (Ham.HamTipiID = :hstElyaf or Ham.HamTipiID = :hstTelef or Ham.HamTipiID = :hstHamPamuk))) and
       (IplikIrsaliyeDetay.Lot = :ALot or :ALot is null) and
       (IplikIrsaliyeDetay.Tarih >= :ABaslangicTarihi and IplikIrsaliyeDetay.Tarih <= :ABitisTarihi) and
       (IplikIrsaliyeDetay.FirmaID = :AFirmaID or :AFirmaID is null) and
       (IplikIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
       (IplikStok.KontrolDurumu = :AKontrolDurumu or :AKontrolDurumu is null)
  into :IplikDepoId, :IplikIrsaliyeId, :IplikIrsaliyeDetayId, :IplikStokId, :HareketTipi, :FirmaId, :FirmaAdi, :IrsaliyeNo, :IrsaliyeTarihi, :HamTipiId, :HamId, :HamAdi,
       :RenkId, :RenkAdi, :MelanjKodu, :FlamKodu, :Lot, :IplikLotID,  :Miktar, :NetMiktar, :KaliteNo, :KaliteAciklamaTipiId, :KaliteAciklamaTipiAdi, :HarmanKayitId, :HarmanYil, :HarmanId, :HarmanAdi, :HarmanCinsiId, :HarmanCinsiAdi,
       :IplikTalepYil, :IplikTalepId, :AmbalajId, :AmbalajAdi, :KonikRengiId, :KonikRengiAdi, :KonikCinsiId, :KonikCinsiAdi,
       :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :BoyaSiparisKayitId, :OrmeSiparisDetayId,
       :BoyaSiparisDetayId, :MusteriSiparisNo, :ReferansSiparisNo, :MusteriKaliteNo, :ArtikelNo,
       :TamirGirisKayitId, :TamirGirisYil, :TamirGirisId, :SarimSekliId, :ElyafDegerlendirme, :TalepBirimMaliyet, :Incelik, :Parlaklik, :Sarilik, :Uzunluk, :KisaElyafIndeksi,
       :Mukavemet, :KontrolDurumu, :KontrolKullaniciID, :KontrolKullaniciAdi, :KontrolTarihi, :Nemlilik do
  begin
    if (KontrolDurumu = 1) then KontrolDurumuStr = 'Evet';
    else KontrolDurumuStr = '';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    IplikTalepIdStr = '';
    if (IplikTalepId is not null) then IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId);

    HarmanIdStr = '';
    if (HarmanId is not null) then HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') '  || EgemenIntegerToStr(HarmanId);

    if (HareketTipi = shtAlis) then HareketTipiStr = 'Alış';
    else if (HareketTipi = ShtSatisIade) then HareketTipiStr = 'Satış İade';
    else if (HareketTipi = ShtUretim) then HareketTipiStr = 'Üretim';
    else if (HareketTipi = shtUretimIade) then HareketTipiStr = 'Üretim İade';
    else if (HareketTipi = shtTelefGirisi) then HareketTipiStr = 'Telef Girişi';
    else if (HareketTipi = shtUretimBoya) then HareketTipiStr = 'Üretim (Boya)';
    else if (HareketTipi = shtDepoTransfer) then HareketTipiStr = 'Depo Transfer';
    else HareketTipiStr = 'Hatalı';

    TamirGirisIdStr = '';
    if (TamirGirisId is not null) then TamirGirisIdStr = '(' || EgemenIntegerToStr(TamirGirisYil) || ') '  || EgemenIntegerToStr(TamirGirisId);


    SarimSekliIdStr = Null;

    if (HamTipiId = 0) then
    if (SarimSekliId = 0) then SarimSekliIdStr = 'Sert Sarım';
    else SarimSekliIdStr = 'Yumuşak Sarım';

    suspend;
  end
end
