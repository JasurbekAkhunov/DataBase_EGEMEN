CREATE PROCEDURE SABLONSILMETALIMATIRAPORU_3(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    SABLONSILMETALIMATIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(80),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SABLONBOYUID VARCHAR(10),
    SABLONBASLIKSIZ INTEGER,
    SABLONMUSTERIDE INTEGER,
    SABLONMUSTERIYEAIT INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SABLONSAYISI INTEGER,
    SABLONADI VARCHAR(1),
    MESHID INTEGER,
    ACIKLAMA VARCHAR(100),
    ILAVESABLONNO INTEGER,
    ORTAKSABLON INTEGER,
    ORTAKSABLONDESENID VARCHAR(20),
    SABLONUKULLANANDESENLER VARCHAR(200))
AS
declare variable OrtakSabloniseDesenId varchar(20);
declare variable FDesenId varchar(20);
declare variable SablonSiraNo integer;
begin
  For Select SablonSilmeTalimati.SablonSilmeTalimatiId, SablonSilmeTalimati.FisYil, SablonSilmeTalimati.FisNo, SablonSilmeTalimati.FisTarihi,
             Desen.DesenId, Desen.Adi, Desen.FirmaId, Firma.Adi,
             SablonB.Olcu, Desen.SablonBasliksiz, Desen.SablonMusteride, Desen.SablonMusteriyeAit,
             Desen.Notu, Desen.SablonSayisi, DesenSablon.SablonAdi, DesenSablon.SablonSiraNo, DesenSablon.MeshId, DesenSablon.Aciklama,
             DesenSablon.IlaveSablonNo, DesenSablon.OrtakSablon, DesenSablon.OrtakSabloniseDesenId from SablonSilmeTalimati
  left join SablonSilmeTalimatDetay on SablonSilmeTalimatDetay.SablonSilmeTalimatiId = SablonSilmeTalimati.SablonSilmeTalimatiId
  left join Desen on Desen.DesenId = SablonSilmeTalimatDetay.DesenId
  left join Firma on Firma.FirmaId = Desen.FirmaId
  left join DesenOlcu SablonB on SablonB.DesenOlcuID = Desen.SablonBoyuDesenOlcuId
  left join DesenSablon on DesenSablon.DesenId = Desen.DesenId
  where SablonSilmeTalimati.FisTarihi >= :ABaslangicTarihi and SablonSilmeTalimati.FisTarihi <= :ABitisTarihi and
        SablonSilmeTalimatDetay.DesenId is not null
  into :SablonSilmeTalimatiId, :FisYil, :FisNo, :FisTarihi, :DesenId, :DesenAdi, :FirmaId, :FirmaAdi,
       :SablonBoyuId, :SablonBasliksiz, :SablonMusteride, :SablonMusteriyeAit, :Notu, :SablonSayisi,
       :SablonAdi, :SablonSiraNo,  :MeshId, :Aciklama, :IlaveSablonNo, :OrtakSablon, :OrtakSabloniseDesenId do
  begin
    FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';

    if (OrtakSabloniseDesenId is not null and OrtakSablon = 1) then
      Select DesenId from Desen where Desen.DesenId = :OrtakSabloniseDesenId into :OrtakSablonDesenId;
    else
      OrtakSablonDesenId = Null;

    SablonuKullananDesenler = '';

    for Select Desen.DesenId from DesenSablon
      left join Desen on Desen.DesenId = DesenSablon.DesenId
      Where DesenSablon.OrtakSabloniseDesenId = :DesenId and DesenSablon.OrtakSabloniseSablonSiraNo = :SablonSiraNo
      group by Desen.DesenId
      into :FDesenId do
    begin
      if (Char_Length(SablonuKullananDesenler) + Char_Length(FDesenId) < 195) then
      begin
        if (SablonuKullananDesenler <> '' ) then SablonuKullananDesenler = SablonuKullananDesenler || cast(', ' as varchar(2) character set UTF8);
        SablonuKullananDesenler = SablonuKullananDesenler || FDesenId;
      end
    end

    suspend;
  end
end
