CREATE PROCEDURE KIRPINTITASNIFYAP(
    AKIRPINTITASNIFKAYITID INTEGER,
    AKIRPINTITASNIFDETAYID INTEGER,
    AHAMID INTEGER,
    ARENKID INTEGER,
    AIPLIKDEPOID DEPOKODUSIZE,
    AKILO DOUBLE PRECISION,
    AKULLANICIID INTEGER)
RETURNS (
    CIKISIPLIKIRSALIYEID INTEGER,
    CIKISIPLIKIRSALIYEDETAYID INTEGER,
    GIRISIPLIKIRSALIYEID INTEGER,
    GIRISIPLIKIRSALIYEDETAYID INTEGER)
AS
declare variable SistemDovizId varchar(5);
declare variable FKaynakHamId      integer;
declare variable FHamId      integer;
declare variable LastHamId      integer;
declare variable shtKirpintiCikis      integer;
declare variable shtUretimKirpinti      integer;
declare variable FTasnifTarih      timestamp;

declare variable FIplikStokId           integer;
declare variable FNumarator             integer;
declare variable SeriNo          varchar(2);
declare variable IrsaliyeNo      integer;
declare variable FKapNo          integer;
declare variable FKaynakIplikDepoId  DepoKoduSize;
declare variable DepoKalanKilo  Double precision;
declare variable DepoKilo       Double precision;
declare variable FKilo          Double precision;

begin
  select IplikDepoId, HamId, Tarih from KirpintiTasnif where KirpintiTasnif.KirpintiTasnifKayitId = :AKirpintiTasnifKayitId into :FKaynakIplikDepoId, :FKaynakHamId, :FTasnifTarih;

  select sum(IplikStok.Miktar) from IplikStok where IplikStok.IplikDepoId = :FKaynakIplikDepoId and IplikStok.CikisIplikIrsaliyeDetayId is null and (HamID = :FKaynakHamId or :FKaynakHamId is null) into :DepoKalanKilo;
  if (AKilo > DepoKalanKilo) then exception KayitHatasi 'Depoda yeterli miktar yok! Depo mevcudu: '|| Format_float('',DepoKalanKilo) || ' Tasnif edilen: ' || Format_float('', AKilo) ;


  select shtKirpintiCikis, shtUretimKirpinti from IplikStokHareketTipiGetir into :shtKirpintiCikis, :shtUretimKirpinti;

  select SistemDovizId from Sistem where SISTEMID = 1 into :SistemDovizId;


  select max(IplikIrsaliyeId) from IplikIrsaliye
    where IplikIrsaliye.IrsaliyeTarihi = :FTasnifTarih and IplikIrsaliye.HareketTipi = :shtKirpintiCikis
    into :CikisIplikIrsaliyeId;

  if (CikisIplikIrsaliyeId is null) then
  begin
    select IplikUretimeCikisSeriNo from Kullanici where KullaniciId = :AKullaniciId into :SeriNo;

    if (SeriNo is null) then SeriNo = '';

    select IrsaliyeNo from IrsaliyeNumaratorAl(:SeriNo) into :IrsaliyeNo;

    CikisIplikIrsaliyeId = gen_id(IplikIrsaliye_Id_Gen, 1);

    insert into IplikIrsaliye(IplikIrsaliyeId, HareketTipi, SeriNo, IrsaliyeNo, IrsaliyeTarihi, DovizId, InsertKullaniciId, KullaniciId)
             values(:CikisIplikIrsaliyeId, :shtKirpintiCikis, :SeriNo, :IrsaliyeNo, :FTasnifTarih, :SistemDovizId, :AKullaniciId, :AKullaniciId);
  end

  CikisIplikIrsaliyeDetayId = Null;
  FKilo = AKilo;
  for select IplikStok.IplikStokId, IplikStok.Numarator, IplikStok.Miktar, IplikStok.HamId from IplikStok where IplikStok.IplikDepoId = :FKaynakIplikDepoId and IplikStok.CikisIplikIrsaliyeDetayId is null and (HamID = :FKaynakHamId or :FKaynakHamId is null)
      into :FIplikStokId, :FNumarator, :DepoKilo, :FHamId do
  begin
    if (CikisIplikIrsaliyeDetayId is Null or FHamId <> LastHamId) then
    begin
      if (CikisIplikIrsaliyeDetayId is not Null) then
      begin
        Update IplikIrsaliyeDetay Set Miktar = :AKilo - :FKilo, NetMiktar = :AKilo - :FKilo, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :CikisIplikIrsaliyeDetayId;
      end

      CikisIplikIrsaliyeDetayId = gen_id(IplikIrsaliyeDetay_Id_Gen, 1);

      insert into IplikIrsaliyeDetay(IplikIrsaliyeId, IplikIrsaliyeDetayId, HareketTipi, IplikDepoId, HamId, KapAdedi, Miktar, NetMiktar, KirpintiTasnifDetayId, InsertKullaniciId, KullaniciId)
                         values (:CikisIplikIrsaliyeId, :CikisIplikIrsaliyeDetayId, :shtKirpintiCikis, :FKaynakIplikDepoId, :FHamId, 1, :FKilo, :FKilo, :AKirpintiTasnifDetayId, :AKullaniciId, :AKullaniciId);
      LastHamId = FHamId;
    end

    if (DepoKilo > FKilo) then
    begin
      FNumarator = 1;
      Select YENIIPLIKSTOKID from IPLIKSTOKBOL(:FIplikStokId, :FKilo, 0) into :FIplikStokId;
      FKilo = 0;
    end
    else
      FKilo = FKilo - DepoKilo;

    Update IplikStok set CikisIplikIrsaliyeDetayId = :CikisIplikIrsaliyeDetayId where IplikStok.IplikStokId = :FIplikStokId and IplikStok.Numarator = :FNumarator;
    if (FKilo <= 0 ) then break;

  end


  select max(IplikIrsaliyeId) from IplikIrsaliye
    where IplikIrsaliye.IrsaliyeTarihi = :FTasnifTarih and IplikIrsaliye.HareketTipi = :shtUretimKirpinti
    into :GirisIplikIrsaliyeId;

  if (GirisIplikIrsaliyeId is null) then
  begin
    select IplikUretimSeriNo from Kullanici where KullaniciId = :AKullaniciId into :SeriNo;

    if (SeriNo is null) then SeriNo = '';

    select IrsaliyeNo from IrsaliyeNumaratorAl(:SeriNo) into :IrsaliyeNo;

    GirisIplikIrsaliyeId = gen_id(IplikIrsaliye_Id_Gen, 1);

    insert into IplikIrsaliye(IplikIrsaliyeId, HareketTipi, SeriNo, IrsaliyeNo, IrsaliyeTarihi, DovizId, InsertKullaniciId, KullaniciId)
             values(:GirisIplikIrsaliyeId, :shtUretimKirpinti, :SeriNo, :IrsaliyeNo, :FTasnifTarih, :SistemDovizId, :AKullaniciId, :AKullaniciId);
  end

  select max(IplikIrsaliyeDetayId) from IplikIrsaliyeDetay
    where IplikIrsaliyeDetay.HareketTipi = :shtUretimKirpinti and
          IplikIrsaliyeDetay.IplikIrsaliyeId = :GirisIplikIrsaliyeId and
          IplikIrsaliyeDetay.IplikDepoId = :AIplikDepoId and
          IplikIrsaliyeDetay.KirpintiTasnifDetayId = :AKirpintiTasnifDetayId and
          IplikIrsaliyeDetay.RenkId = :ARenkId and
         (IplikIrsaliyeDetay.HamID = :AHamId)
    into :GirisIplikIrsaliyeDetayId;

  if (GirisIplikIrsaliyeDetayId is not Null) then Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi + 1, Miktar = Miktar + :AKilo, NetMiktar = NetMiktar + :AKilo, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :GirisIplikIrsaliyeDetayId;
  else
  begin
    GirisIplikIrsaliyeDetayId = gen_id(IplikIrsaliyeDetay_Id_Gen, 1);
    insert into IplikIrsaliyeDetay(IplikIrsaliyeId, IplikIrsaliyeDetayId, HareketTipi, IplikDepoId, HamId, RenkId, KirpintiTasnifDetayId, KapAdedi, Miktar, NetMiktar, InsertKullaniciId, KullaniciId)
                     Values (:GirisIplikIrsaliyeId, :GirisIplikIrsaliyeDetayId, :shtUretimKirpinti, :AIplikDepoId, :AHamId, :ARenkId, :AKirpintiTasnifDetayId, 1, :AKilo, :AKilo, :AKullaniciId, :AKullaniciId );
  end

  suspend;
end
