CREATE PROCEDURE MAKINASARJSAYACARTTIR(
    AMAKINAID VARCHAR(10),
    ADOGALGAZ INTEGER,
    AELEKTIRIK INTEGER,
    AOKUNANSU INTEGER)
AS
declare variable MakinaSarjId    integer;
declare variable AsilMakinaID    MakinaKoduSize;
declare variable EgemenFirmaID   Integer;
begin
  if (ADogalgaz > 0 or AElektirik > 0 or AOkunanSu > 0) then
  begin
    Select AsilMakinaID from Makina where Makina.MakinaId = :AMakinaId into :AsilMakinaID;
    if (AsilMakinaID is Not Null) then AMakinaID = :AsilMakinaID;

    Select Max(MakinaSarjID) from MakinaSarj
      where MakinaSarj.BitisTarihi is null and MakinaSarj.MakinaID = :AMakinaID
      into :MakinaSarjId;

    if (MakinaSarjId is null) then
    begin
      MakinaSarjId = Gen_ID(MakinaSarj_Id_Gen, 1);

      insert into MakinaSarj(MakinaSarjID, MakinaId, BaslamaTarihi, BitisTarihi, PartiVar)
                      values(:MakinaSarjID, :AMakinaId, Current_TimeStamp(0), Null, 0);
    end

    if (ADogalgaz > 0) then
      Update MakinaSarj Set Dogalgaz = Dogalgaz + 1 where MakinaSarj.MakinaSarjID = :MakinaSarjID;
    else if (AElektirik > 0) then
      Update MakinaSarj Set Elektirik = Elektirik + 1 where MakinaSarj.MakinaSarjID = :MakinaSarjID;
    else
    begin
      select EgemenFirmaId from Sistem where SistemId = 1 into :EgemenFirmaId;

      if (EgemenFirmaId = 0) then/*0 Nuryıldız*/
        Update MakinaSarj Set Metre = Metre + :AOkunanSu where MakinaSarj.MakinaSarjID = :MakinaSarjID;
      else
        Update MakinaSarj Set OkunanSu = OkunanSu + 1 where MakinaSarj.MakinaSarjID = :MakinaSarjID;
    end
  end
end
