CREATE PROCEDURE FIRMANINHAMSTOKLARI(
    AFIRMAID VARCHAR(20))
RETURNS (
    PARTILENMISHAMSTOKTOPLAMI NUMERIC(12,3),
    PARTILENMISTAMIRTOPLAMI NUMERIC(12,3),
    PARTILENMEMISHAMSTOKTOPLAMI NUMERIC(12,3),
    IADEEDILECEKHAMSTOKTOPLAMI NUMERIC(12,3))
AS
declare variable T1 Numeric(12,3);
declare variable T2 Numeric(12,3);
declare variable T3 Numeric(12,3);
declare variable T4 Numeric(12,3);
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
begin
  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  select sum(PartiStok.Kilo) from Parti
  left join Partistok on Partistok.PartiKayitId = Parti.PartiKayitId
  where Parti.FirmaId = :AFirmaId and Parti.Fason = 1 and Partistok.durumu < 7 and Parti.IslemTipiId = :itTamir into :PartilenmisTamirToplami;

  select sum(PartiStok.Kilo) from Parti
  left join Partistok on Partistok.PartiKayitId = Parti.PartiKayitId
  where Parti.FirmaId = :AFirmaId and Parti.Fason = 1 and Partistok.durumu < 7 and Parti.IslemTipiId <> :itTamir into :PartilenmisHamStokToplami;


/* Gelen Kilo */
  Select sum(HamStokDetay.kilo) from HamStokDetay
  where HamStokDetay.StokFirmaId = :AFirmaId and HamStokDetay.Fason = 1
  into :T1;

/* Iade Edilecek Kilo */
  select sum(RafDetay.IadeEdilecekKilo),sum(RafDetay.IadeEdilenKilo) from HamStokDetay
  left join Rafdetay on Rafdetay.HamStokDetayId = HamStokDetay.HamStokDetayId
  where HamStokDetay.StokFirmaId = :AFirmaId and HamStokDetay.Fason = 1  into :T2, :T3;

/* Partilenen */
  select sum(PartiStok.Kilo) from Parti
  left join Partistok on Partistok.PartiKayitId = Parti.PartiKayitId
  where Parti.FirmaId = :AFirmaId and Parti.Fason = 1 into :T4;

  if (T1 is null) then T1 = 0; /* Gelen Kilo */
  if (T2 is null) then T2 = 0; /* Iade Edilecek Kilo */
  if (T3 is null) then T3 = 0; /* Iade Edilen Kilo */
  if (T4 is null) then T4 = 0; /* Partilenen */

  PartilenmemisHamStokToplami = T1 - T3 - T4;
  IadeEdilecekHamStokToplami = T2;
  suspend;
end
