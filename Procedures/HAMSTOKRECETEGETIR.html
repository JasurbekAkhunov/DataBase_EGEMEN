CREATE PROCEDURE HAMSTOKRECETEGETIR(
    AHAMID INTEGER)
RETURNS (
    HAMSTOKRECETEID INTEGER,
    HAMID INTEGER,
    RECETEHAMID INTEGER,
    RECETEHAMADI ADI255,
    KULLANIMORANI DOUBLE PRECISION,
    FIREORANI DOUBLE PRECISION,
    COZGUFIREORANI DOUBLE PRECISION,
    RENKID INTEGER,
    RENKADI ADI50,
    ATKICOZGU SMALLINT,
    ATKICOZGUSTR VARCHAR(5),
    TELSAYISI INTEGER,
    SIRANO SMALLINT,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATDOVIZID DOVIZSIZE,
    INSERTKULLANICIID INTEGER,
    INSERTKULLANICIADI ADI40,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    KULLANICIADI ADI40,
    DEGISIMTARIHI TIMESTAMP,
    MALIYETFIREORANI DOUBLE PRECISION,
    IPLIKSIKLIGICM DOUBLE PRECISION,
    ATKIIPLIKCM DOUBLE PRECISION,
    AGIRLIKKILO BOYAMIKTARSIZE,
    UZUNLUKMETRE BOYAMIKTARSIZE,
    HATASTR VARCHAR(500))
AS
begin
  for select HamStokRecete.HamStokReceteId, HamStokRecete.HamId, HamStokRecete.ReceteHamId, Ham.Adi, HamStokRecete.KullanimOrani, HamStokRecete.FireOrani, HamStokRecete.CozguFireOrani, HamStokRecete.RenkId,
         Renk.Adi, HamStokRecete.AtkiCozgu, Decode(HamStokRecete.AtkiCozgu, 1 , 'Atkı', 2, 'Çözgü', ''), HamStokRecete.TelSayisi, HamStokRecete.SiraNo, HamStokRecete.BirimFiyat, HamStokRecete.BirimFiyatDovizId,
         HamStokRecete.InsertKullaniciId, InsKul.Adi, HamStokRecete.InsertTarihi, HamStokRecete.KullaniciId, Kullanici.Adi, HamStokRecete.DegisimTarihi, HamStokRecete.MaliyetFireOrani, HamStokRecete.IplikSikligiCm,
         iif(Ham.BolgeselSiklik = 1, HamStokRecete.IplikSikligiCm, Ham.AtkiIplikCm) from hamstokrecete
      left join Ham on Ham.HamId = HamStokRecete.ReceteHamId
      left join Kullanici on Kullanici.KullaniciId = HamStokRecete.KullaniciId
      left join Kullanici InsKul on InsKul.KullaniciId = HamStokRecete.InsertKullaniciId
      left join Renk on Renk.RenkId = HamStokRecete.RenkId
      where HamStokRecete.HamId = :AHamId
      order by AtkiCozgu, SiraNo
      into :HamStokReceteId, :HamId, :ReceteHamId, :ReceteHamAdi, :KullanimOrani, :FireOrani, :CozguFireOrani, :RenkId, :RenkAdi, :AtkiCozgu, :AtkiCozguStr, :TelSayisi, :SiraNo, :BirimFiyat,
      :BirimFiyatDovizId, :InsertKullaniciId, :InsertKullaniciAdi, :InsertTarihi, :KullaniciId, :KullaniciAdi, :DegisimTarihi, :MaliyetFireOrani, :IplikSikligiCm, :AtkiIplikCm do
  begin
    select DokumaIplikMiktariHesaplama.AgirlikKilo, DokumaIplikMiktariHesaplama.Uzunlukmetre, DokumaIplikMiktariHesaplama.HataStr
    from DokumaIplikMiktariHesaplama(null, null, null, 0, 1, :HamStokReceteId) into :AgirlikKilo, :UzunlukMetre, :HataStr;

    suspend;
  end
end
