CREATE PROCEDURE SIPARISVEHAMSTOKMIKTARANALIZI(
    MEVCUTSTOKLARIDAGOSTER INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    SIPARISBRUTKILO NUMERIC(12,3),
    MEVCUTMIKTAR NUMERIC(12,3),
    BEKLENENMIKTAR NUMERIC(12,3),
    SIPARISIVERILMEMISTALEP NUMERIC(12,3),
    ISLETMEMIKTARDEPO NUMERIC(12,3),
    ISLETMEMIKTARBOYANACAK NUMERIC(12,3),
    ISLETMEMIKTARKAZAN NUMERIC(12,3),
    ISLETMEMIKTARKAZANSONRASI NUMERIC(12,3))
AS
declare variable FirmaKodumuz varchar(20);
declare variable SatSiparisKalanBruttenHesapla integer;
declare variable BSatisSiparisleriNetGirilecek  integer;
declare variable SipBrutMiktarHesaplamaSekli   integer;
declare variable FSiparisBrutKilo Numeric(12,3);
declare variable VerilenSipteDetayOnayiKullan integer;
declare variable FEgemenFirmaId integer;
begin
    Select FirmaKodumuz, SatSiparisKalanBruttenHesapla, BSatisSiparisleriNetGirilecek, SipBrutMiktarHesaplamaSekli, VerilenSipteDetayOnayiKullan, EgemenFirmaId from Sistem where sistemid = 1 into :FirmaKodumuz, :SatSiparisKalanBruttenHesapla, :BSatisSiparisleriNetGirilecek, :SipBrutMiktarHesaplamaSekli, :VerilenSipteDetayOnayiKullan, :FEgemenFirmaId;

 if (MevcutStoklaridaGoster = 1) then
 begin
   For  Select Ham.HamId, Ham.Adi, Ham.HamTipiId,  Sum(GirisMiktar - (PartilenenMiktar + IadeEdilenMiktar + SatisMiktar))  from Hamstokkumilatif
      Left Join Ham on Ham.HamId = Hamstokkumilatif.HamId
      Where FirmaID = :FirmaKodumuz
      Group by Ham.HamId, Ham.Adi, Ham.Hamtipiid
      Having Sum(GirisMiktar - (PartilenenMiktar + IadeEdilenMiktar + SatisMiktar)) > 0
     into :HamId, :HamAdi, HamTipiId, :MevcutMiktar do
    begin
      SiparisKilo = null;
      SiparisMetre = null;
      SiparisBrutKilo = null;

      for select BoyaSiparisDetay.HamId, Ham.Adi, Ham.HamTipiId,
                 Sum(BoyaSiparisDetay.SiparisKilo - (BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo)),
                 Sum(BoyaSiparisDetay.SiparisMetre - (BoyaSiparisDetay.PartilenenMetre - BoyaSiparisDetay.IptalMetre)),
                 Sum(BoyaSiparisDetay.SiparisKilo / ((100 - BoyaSiparisDetay.MaximumFireOrani) / 100) - (BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo)),
                 Sum(BoyaSiparisDetay.SiparisKilo + BoyaSiparisDetay.SiparisKilo * BoyaSiparisDetay.MaximumFireOrani / 100 - (BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo)) from BoyaSiparis
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
      left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
      where BoyaSiparisDetay.Kapali = 0 and BoyaSiparis.Fason = 0 and  BoyaSiparisDetay.HamId = :HamId and
            ( (Ham.HamTipiID <> 2 and BoyaSiparisDetay.SiparisKilo > (BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo)) or
              (Ham.HamTipiID = 2 and BoyaSiparisDetay.SiparisMetre > (BoyaSiparisDetay.PartilenenMetre - BoyaSiparisDetay.IptalMetre)))
      group by BoyaSiparisDetay.HamId, Ham.Adi, Ham.HamTipiId
      into :HamId, :HamAdi, HamTipiId, :SiparisKilo, :SiparisMetre, :SiparisBrutKilo, :FSiparisBrutKilo do
      begin
        MevcutMiktar = 0;
        BeklenenMiktar = 0;
        SiparisiVerilmemisTalep = 0;
        if (BSatisSiparisleriNetGirilecek = 0 or SatSiparisKalanBruttenHesapla = 0) then SiparisBrutKilo = SiparisKilo;
        else if (SipBrutMiktarHesaplamaSekli <> 0) then SiparisBrutKilo = FSiparisBrutKilo;
      end

      --select sum(GirisMiktar)- sum(PartilenenMiktar)- sum(iadeedilenMiktar) - sum(satisMiktar) from HamStokKumilatif
      --  where HamStokKumilatif.Firmaid = :FirmaKodumuz and HamStokKumilatif.HamId = :HamId
      --  into :MevcutMiktar;

      Select Sum(VerilenSiparisDetay.Miktar - VerilenSiparisDetay.GelenMiktar) from VerilenSiparisDetay
        left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
        where VerilenSiparisDetay.HamId = :HamId  and ((VerilenSiparis.OnayDurumu = 1 and :VerilenSipteDetayOnayiKullan = 0) or (VerilenSiparisDetay.OnayDurumu = 1 and :VerilenSipteDetayOnayiKullan = 1) ) and
               VerilenSiparisDetay.Kapali = 0 and VerilenSiparisDetay.Miktar > VerilenSiparisDetay.GelenMiktar
        into :BeklenenMiktar;

      Select Sum(SatinAlmaTalepDetay.Miktar - SatinAlmaTalepDetay.SiparisiVerilenMiktar) from SatinAlmaTalepDetay
        left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
        where SatinAlmaTalepDetay.HamId = :HamId and SatinAlmaTalep.OnayDurumu <> 2 and
               SatinAlmaTalepDetay.Kapali = 0 and SatinAlmaTalep.IstekYeri = 1 and SatinAlmaTalep.SadeceFiyatArastirmasi = 0 and
               SatinAlmaTalepDetay.Miktar > SatinAlmaTalepDetay.SiparisiVerilenMiktar
        into :SiparisiVerilmemisTalep;

      BeklenenMiktar = Yuvarla(BeklenenMiktar, 100);
      MevcutMiktar = Yuvarla(MevcutMiktar, 100);
      SiparisiVerilmemisTalep = Yuvarla(SiparisiVerilmemisTalep, 100);

      if (MevcutMiktar is null) then MevcutMiktar = 0;
      if (BeklenenMiktar is null) then BeklenenMiktar = 0;
      if (SiparisiVerilmemisTalep is null) then SiparisiVerilmemisTalep = 0;

      IsletmeMiktarDepo =0;
      IsletmeMiktarBoyanacak =0;
      IsletmeMiktarKazan        = 0;
      IsletmeMiktarKazanSonrasi = 0;
      if (FEgemenFirmaId = 142 or FEgemenFirmaId = 124)  then   /*Abm Cotton , Miray*/
      begin
        Select sum(Kilo) from IsletmedekiPartilerinListesi(0, null, null, null, null, null, null, Null, :HamId, Null, Null, Null, Null, null, null, null, null, null, null, Null, null, null, null, null, null, 0, null, null, null, Null, null, null, null, 0, null, 0, 0, null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, null, null, null, null, null)
          where IsletmedekiPartilerinListesi.Durumu =0
          into :IsletmeMiktarDepo;

        Select sum(Kilo) from IsletmedekiPartilerinListesi(0, null, null, null, null, null, null, Null, :HamId, Null, Null, Null, Null, null, null, null, null, null, null, Null, null, null, null, null, null, 0, null, null, null, Null, null, null, null, 0, null, 0, 0, null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, null, null, null, null, null)
          where IsletmedekiPartilerinListesi.Durumu <=4
          into :IsletmeMiktarBoyanacak;

      end
      else
      begin
        Select sum(PartiStok.Kilo) from Parti
          left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
          left join PartiStokIptal on PartiStokIptal.PartiSinifId = PartiStok.PartiSinifId
          where Parti.Fason = 0 and PartiStok.HamId = :HamId and PartiStok.Durumu = 0  and PartiStokIptal.PartiSinifId is null
            into :IsletmeMiktarDepo;

        Select sum(PartiStok.Kilo) from Parti
          left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
          left join PartiStokIptal on PartiStokIptal.PartiSinifId = PartiStok.PartiSinifId
          where Parti.Fason = 0 and PartiStok.HamId = :HamId and PartiStok.Durumu < 4 and PartiStokIptal.PartiSinifId is null
            into :IsletmeMiktarBoyanacak;
      end

      Select sum(PartiStok.Kilo) from Parti
        left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
        left join PartiStokIptal on PartiStokIptal.PartiSinifId = PartiStok.PartiSinifId
        where Parti.Fason = 0 and PartiStok.HamId = :HamId and PartiStok.Durumu = 4 and PartiStokIptal.PartiSinifId is null
          into :IsletmeMiktarKazan;

      Select sum(PartiStok.Kilo) from Parti
        left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
        left join PartiStokIptal on PartiStokIptal.PartiSinifId = PartiStok.PartiSinifId
        where Parti.Fason = 0 and PartiStok.HamId = :HamId and PartiStok.Durumu > 4 and PartiStok.Durumu < 7 and PartiStokIptal.PartiSinifId is null
          into :IsletmeMiktarKazanSonrasi;
      suspend;
    end
  end
  else
  begin
    for select BoyaSiparisDetay.HamId, Ham.Adi, Ham.HamTipiId,
               Sum(BoyaSiparisDetay.SiparisKilo - (BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo)),
               Sum(BoyaSiparisDetay.SiparisMetre - (BoyaSiparisDetay.PartilenenMetre - BoyaSiparisDetay.IptalMetre)),
               Sum(BoyaSiparisDetay.SiparisKilo / ((100 - BoyaSiparisDetay.MaximumFireOrani) / 100) - (BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo)),
               Sum(BoyaSiparisDetay.SiparisKilo + BoyaSiparisDetay.SiparisKilo * BoyaSiparisDetay.MaximumFireOrani / 100 - (BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo)) from BoyaSiparis
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    where BoyaSiparisDetay.Kapali = 0 and BoyaSiparis.Fason = 0 and
          ( (Ham.HamTipiID <> 2 and BoyaSiparisDetay.SiparisKilo > (BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo)) or
            (Ham.HamTipiID = 2 and BoyaSiparisDetay.SiparisMetre > (BoyaSiparisDetay.PartilenenMetre - BoyaSiparisDetay.IptalMetre)))
    group by BoyaSiparisDetay.HamId, Ham.Adi, Ham.HamTipiId
    into :HamId, :HamAdi, HamTipiId, :SiparisKilo, :SiparisMetre, :SiparisBrutKilo, :FSiparisBrutKilo do
    begin
      MevcutMiktar = 0;
      BeklenenMiktar = 0;
      SiparisiVerilmemisTalep = 0;
      if (BSatisSiparisleriNetGirilecek = 0 or SatSiparisKalanBruttenHesapla = 0) then SiparisBrutKilo = SiparisKilo;
      else if (SipBrutMiktarHesaplamaSekli <> 0) then SiparisBrutKilo = FSiparisBrutKilo;

      select sum(GirisMiktar)- sum(PartilenenMiktar)- sum(iadeedilenMiktar) - sum(satisMiktar) from HamStokKumilatif
      where HamStokKumilatif.Firmaid = :FirmaKodumuz and HamStokKumilatif.HamId = :HamId
      into :MevcutMiktar;

      Select Sum(VerilenSiparisDetay.Miktar - VerilenSiparisDetay.GelenMiktar) from VerilenSiparisDetay
       left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
       where VerilenSiparisDetay.HamId = :HamId  and ((VerilenSiparis.OnayDurumu = 1 and :VerilenSipteDetayOnayiKullan = 0) or (VerilenSiparisDetay.OnayDurumu = 1 and :VerilenSipteDetayOnayiKullan = 1) ) and
             VerilenSiparisDetay.Kapali = 0 and VerilenSiparisDetay.Miktar > VerilenSiparisDetay.GelenMiktar
       into :BeklenenMiktar;

      Select Sum(SatinAlmaTalepDetay.Miktar - SatinAlmaTalepDetay.SiparisiVerilenMiktar) from SatinAlmaTalepDetay
       left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
       where SatinAlmaTalepDetay.HamId = :HamId and SatinAlmaTalep.OnayDurumu <> 2 and
             SatinAlmaTalepDetay.Kapali = 0 and SatinAlmaTalep.IstekYeri = 1 and SatinAlmaTalep.SadeceFiyatArastirmasi = 0 and
             SatinAlmaTalepDetay.Miktar > SatinAlmaTalepDetay.SiparisiVerilenMiktar
       into :SiparisiVerilmemisTalep;

      BeklenenMiktar = Yuvarla(BeklenenMiktar, 100);
      MevcutMiktar = Yuvarla(MevcutMiktar, 100);
      SiparisiVerilmemisTalep = Yuvarla(SiparisiVerilmemisTalep, 100);

      if (MevcutMiktar is null) then MevcutMiktar = 0;
      if (BeklenenMiktar is null) then BeklenenMiktar = 0;
      if (SiparisiVerilmemisTalep is null) then SiparisiVerilmemisTalep = 0;

      IsletmeMiktarDepo =0;
      IsletmeMiktarBoyanacak =0;
      IsletmeMiktarKazan        = 0;
      IsletmeMiktarKazanSonrasi = 0;

      if (FEgemenFirmaId = 142 or FEgemenFirmaId = 124)  then   /*Abm Cotton , Miray*/
      begin
        Select sum(Kilo) from IsletmedekiPartilerinListesi(0, null, null, null, null, null, null, Null, :HamId, Null, Null, Null, Null, null, null, null, null, null, null, Null, null, null, null, null, null, 0, null, null, null, Null, null, null, null, 0, null, 0, 0, null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, null, null, null, null, null)
        where IsletmedekiPartilerinListesi.Durumu =0
          into :IsletmeMiktarDepo;

        Select sum(Kilo) from IsletmedekiPartilerinListesi(0, null, null, null, null, null, null, Null, :HamId, Null, Null, Null, Null, null, null, null, null, null, null, Null, null, null, null, null, null, 0, null, null, null, Null, null, null, null, 0, null, 0, 0, null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, Null, null, null, null, null, null)
        where IsletmedekiPartilerinListesi.Durumu <=4
          into :IsletmeMiktarBoyanacak;

      end
    else
    begin
      Select sum(PartiStok.Kilo) from Parti
        left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
        left join PartiStokIptal on PartiStokIptal.PartiSinifId = PartiStok.PartiSinifId
        where Parti.Fason = 0 and PartiStok.HamId = :HamId and PartiStok.Durumu = 0  and PartiStokIptal.PartiSinifId is null
          into :IsletmeMiktarDepo;

        Select sum(PartiStok.Kilo) from Parti
        left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
        left join PartiStokIptal on PartiStokIptal.PartiSinifId = PartiStok.PartiSinifId
        where Parti.Fason = 0 and PartiStok.HamId = :HamId and PartiStok.Durumu < 4 and PartiStokIptal.PartiSinifId is null
          into :IsletmeMiktarBoyanacak;
    end

    Select sum(PartiStok.Kilo) from Parti
        left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
        left join PartiStokIptal on PartiStokIptal.PartiSinifId = PartiStok.PartiSinifId
        where Parti.Fason = 0 and PartiStok.HamId = :HamId and PartiStok.Durumu = 4 and PartiStokIptal.PartiSinifId is null
        into :IsletmeMiktarKazan;

      Select sum(PartiStok.Kilo) from Parti
        left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
        left join PartiStokIptal on PartiStokIptal.PartiSinifId = PartiStok.PartiSinifId
        where Parti.Fason = 0 and PartiStok.HamId = :HamId and PartiStok.Durumu > 4 and PartiStok.Durumu < 7 and PartiStokIptal.PartiSinifId is null
        into :IsletmeMiktarKazanSonrasi;

      suspend;
    end
  end
