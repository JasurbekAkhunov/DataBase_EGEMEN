CREATE PROCEDURE ELIARTOEGEMENGIRISCIKISYAP(
    ADOZAJLAMAISEMRIID INTEGER,
    AOTOMASYONMAKINANO VARCHAR(10),
    AGIRISTARIHI TIMESTAMP,
    AGIRISPERSONELID INTEGER,
    ACIKISTARIHI TIMESTAMP,
    ACIKISPERSONELID INTEGER)
RETURNS (
    STATUS INTEGER,
    MESAJ VARCHAR(100))
AS
declare variable PartiKayitID integer;
declare variable PartiSinifID integer;
declare variable FPartiAsamalariID integer;
declare variable MakinaID MakinaKoduSize;
declare variable ProxyID varchar(8);
declare variable ReceteIsEmriYil integer;
declare variable ReceteIsEmriID integer;
declare variable PartiAsamaStokDetayId integer;
begin
  Select DozajlamaIsEmri.PartiAsamalariID, DozajlamaIsEmri.ReceteIsEmriYil, DozajlamaIsEmri.ReceteIsEmriID from DozajlamaIsEmri where DozajlamaIsEmri.DozajlamaIsEmriID = :ADozajlamaIsEmriID
    into :FPartiAsamalariID, :ReceteIsEmriYil, :ReceteIsEmriID;

  if (FPartiAsamalariID is null and ReceteIsEmriID is Null) then
  begin
    Status = -1;
    Mesaj = 'Dozajlama İş Emri Kaydına Ulaşılamadı. DozajlamaIsEmriNo : ' || EgemenIntegertoStr(:ADozajlamaIsEmriID);
    suspend;
    exit;
  end

  Select MakinaID from Makina where Makina.OtomasyonMakinaNo = :AOtomasyonMakinaNo into :MakinaID;

  if (MakinaID is Null) then
  begin
    Status = -1;
    Mesaj = 'Otomasyon Makina No ile eşleşen makina Bulunamadı. OtomasyonMakinaNo : ' || :AOtomasyonMakinaNo;
    suspend;
    exit;
  end

  if (AGirisTarihi is not null and (AGirisPersonelID is Null or AGirisPersonelID < 0)) then
  begin
    Status = -1;
    Mesaj = 'Giriş Personel Kodu Yok.';
    suspend;
    exit;
  end
  if (ACikisTarihi is not null and (ACikisPersonelID is Null or ACikisPersonelID < 0)) then
  begin
    Status = -1;
    Mesaj = 'Çıkış Personel Kodu Yok.';
    suspend;
    exit;
  end

  if (FPartiAsamalariID is not Null) then
  begin
    for select PartiStok.ProxyID from PartiAsamalari
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join PartiStok on PartiStok.PartiSinifID = PartiAsamaStok.PartiSinifID
        where PartiAsamalari.PartiAsamalariID = :FPartiAsamalariID and PartiAsamalari.OtomasyonaDahilmi = 1 and
              PartiAsamaStok.AsamayaDahil = 1
        into :ProxyID do
    begin
      Select Status, Mesaj, PartiAsamaStokDetayId from GirisCikisYap(:ProxyId, :MakinaId, Coalesce(:AGirisPersonelID, :ACikisPersonelID), 1) into :Status, :Mesaj, :PartiAsamaStokDetayId;
      if ((Status = 1 or Status = 2 or Status = 5 or Status = 15) and (PartiAsamaStokDetayId is not Null)) then
      begin
        if (AGirisTarihi is not null and ACikisTarihi is not null) then
          Update PartiAsamaStokDetay Set GirisTarihi = :AGirisTarihi, GirisPersonelId = :AGirisPersonelID, CikisTarihi = :ACikisTarihi, CikisPersonelId = :ACikisPersonelID where PartiAsamaStokDetay.PartiAsamaStokDetayID = :PartiAsamaStokDetayId;
        else if (AGirisTarihi is not null) then
          Update PartiAsamaStokDetay Set GirisTarihi = :AGirisTarihi, GirisPersonelId = :AGirisPersonelID where PartiAsamaStokDetay.PartiAsamaStokDetayID = :PartiAsamaStokDetayId;
        else if (ACikisTarihi is not null) then
          Update PartiAsamaStokDetay Set CikisTarihi = :ACikisTarihi, CikisPersonelId = :ACikisPersonelID where PartiAsamaStokDetay.PartiAsamaStokDetayID = :PartiAsamaStokDetayId;
        Status = 1;
        Mesaj = 'Giris/Çıkış İşlemi Yapıldı...';
      end

      Suspend;
    end
  end
  else if (ReceteIsEmriID is not Null) then
  begin
    for select PartiStok.ProxyID from ReceteIsEmriDetay
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = ReceteIsEmriDetay.PartiAsamalariId
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join PartiStok on PartiStok.PartiSinifID = PartiAsamaStok.PartiSinifID
        where ReceteIsEmriDetay.ReceteIsEmriYil = :ReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriID = :ReceteIsEmriID and PartiAsamalari.OtomasyonaDahilmi = 1 and
              PartiAsamaStok.AsamayaDahil = 1
        into :ProxyID do
    begin
      Select Status, Mesaj, PartiAsamaStokDetayId from GirisCikisYap(:ProxyId, :MakinaId, Coalesce(:AGirisPersonelID, :ACikisPersonelID), 1) into :Status, :Mesaj, :PartiAsamaStokDetayId;

      if ((Status = 1 or Status = 2 or Status = 5 or Status = 15) and (PartiAsamaStokDetayId is not Null)) then
      begin
        if (AGirisTarihi is not null and ACikisTarihi is not null) then
          Update PartiAsamaStokDetay Set GirisTarihi = :AGirisTarihi, GirisPersonelId = :AGirisPersonelID, CikisTarihi = :ACikisTarihi, CikisPersonelId = :ACikisPersonelID where PartiAsamaStokDetay.PartiAsamaStokDetayID = :PartiAsamaStokDetayId;
        else if (AGirisTarihi is not null) then
          Update PartiAsamaStokDetay Set GirisTarihi = :AGirisTarihi, GirisPersonelId = :AGirisPersonelID where PartiAsamaStokDetay.PartiAsamaStokDetayID = :PartiAsamaStokDetayId;
        else if (ACikisTarihi is not null) then
          Update PartiAsamaStokDetay Set CikisTarihi = :ACikisTarihi, CikisPersonelId = :ACikisPersonelID where PartiAsamaStokDetay.PartiAsamaStokDetayID = :PartiAsamaStokDetayId;

        Status = 1;
        Mesaj = 'Giris/Çıkış İşlemi Yapıldı...';
      end

      Suspend;
    end
  end
end
