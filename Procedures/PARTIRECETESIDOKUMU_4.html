CREATE PROCEDURE PARTIRECETESIDOKUMU_4(
    APARTIASAMALARIID INTEGER)
RETURNS (
    URUNID URUNKODUSIZE,
    URUNADI ADI80,
    LOT VARCHAR(150),
    MIKTAR BOYAMIKTARSIZE,
    BIRIMID BIRIMKODUSIZE)
AS
begin
  for select StokDetay.UrunId, Urun.Adi, StokDetay.Lot, StokDetay.Miktar, UrunBirim.BirimId from StokDetay
    left join Urun on Urun.UrunId = StokDetay.UrunId
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    where StokDetay.PartiAsamalariId = :APartiAsamalariId
    into :UrunId, :UrunAdi, :Lot, :Miktar, :BirimId do
  begin
    suspend;
  end
end
