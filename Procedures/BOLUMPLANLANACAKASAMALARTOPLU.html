CREATE PROCEDURE BOLUMPLANLANACAKASAMALARTOPLU(
    ABOYAHANEBOLUMID INTEGER)
RETURNS (
    BOLUMPLANLAMAID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(20),
    PARTIASAMALARIID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    PARTIKILO NUMERIC(12,3),
    PARTIMETRE NUMERIC(12,3),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    RECETEISEMRIIDSTR VARCHAR(30),
    BIRLESTIRILENPARTILER VARCHAR(100),
    BOYAASAMASIMI INTEGER,
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    SONRAKIASAMAID INTEGER,
    SONRAKIASAMAADI VARCHAR(40))
AS
begin
  For Select BolumPlanlama.BolumPlanlamaId, BolumPlanlama.PartiKayitId, BolumPlanlama.PartiSinifId, BolumPlanlama.PartiAsamalariId,
            Parti.PartiYil, Parti.PartiId, Parti.FirmaId, Firma.Adi, PartiAsamalari.ReceteId, Recete.Adi, BPPartiStok.SonCikisKilo, BPPartiStok.SonCikisMetre,
            BPPartiStok.HamAdiToplami, Asama.BoyaAsamasimi, Asama.AsamaID, Asama.Adi from BolumPlanlama
  Left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = BolumPlanlama.PartiAsamalariId
  Left join PartiStok BPPartiStok on BPPartiStok.PartiSinifId = BolumPlanlama.PartiSinifId
  Left join Parti on Parti.PartiKayitId = BolumPlanlama.PartiKayitId
  Left join Firma on Firma.FirmaId = Parti.FirmaId
  left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
  left join Asama on (Asama.AsamaId = PartiAsamalari.AsamaId or Asama.AsamaId = Recete.UstAsamaId)
  where BolumPlanlama.BoyahaneBolumID = :ABoyahaneBolumID and BolumPlanlama.MakinaPlanlamagrupId is null
  Group By BolumPlanlama.BolumPlanlamaId, BolumPlanlama.PartiKayitId, BolumPlanlama.PartiSinifId, BolumPlanlama.PartiAsamalariId,
            Parti.PartiYil, Parti.PartiId, Parti.FirmaId, Firma.Adi, PartiAsamalari.ReceteId, Recete.Adi, BPPartiStok.SonCikisKilo, BPPartiStok.SonCikisMetre,
            BPPartiStok.HamAdiToplami, Asama.BoyaAsamasimi, Asama.AsamaID, Asama.Adi
  order by BolumPlanlama.BolumPlanlamaId
  into :BolumPlanlamaId, :PartiKayitId, :PartiSinifId, :PartiAsamalariId, :PartiYil, :PartiId, :FirmaId, :FirmaAdi, :ReceteId, :ReceteAdi, :Kilo, :Metre, :HamAdiToplami, :BoyaAsamasimi, :AsamaID, :AsamaAdi
  do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    Select Sum(PartiStok.SonCikisKilo), Sum(PartiStok.SonCikisMetre) from PartiStok where PartiStok.PartiKayitID = :PartiKayitID and PartiStok.PartiStokIptalKayitId is null
      into :PartiKilo, :PartiMetre;

    ReceteIsEmriIdStr = Null;
    BirlestirilenPartiler = '';

    if (ReceteId is not null) then
      Select ReceteIsEmriIdStr, BirlestirilenPartiler from PAReceteIsEmriBilgisiGetir(:PartikayitId, :ReceteId, :PartiAsamalariId) into :ReceteIsEmriIdStr, :BirlestirilenPartiler;

    SonrakiAsamaID = NULL;
    SonrakiAsamaAdi = NULL;

    select first 1 PartiAsamaStok.AsamaID, Asama.Adi from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariID
    left join Asama on  Asama.AsamaID = PartiAsamaStok.AsamaID
    where PartiAsamaStok.PartiSinifID = :PartiSinifId and PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.AsamayaDahil = 1
    order by PartiAsamalari.Sira ASC
    INTO :SonrakiAsamaID, :SonrakiAsamaAdi;

    suspend;
  end
end
