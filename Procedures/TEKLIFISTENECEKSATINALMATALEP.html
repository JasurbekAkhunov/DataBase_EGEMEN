CREATE PROCEDURE TEKLIFISTENECEKSATINALMATALEP(
    AVERILENSIPARISTIPIID INTEGER,
    ASATINALMABIRIMID INTEGER,
    AURUNKATEGORIID VARCHAR(4))
RETURNS (
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30),
    TALEPEDENKULLANICIADI VARCHAR(40),
    TALEPTARIHI TIMESTAMP,
    SATINALMATALEPDETAYID INTEGER,
    URUNKATEGORIID VARCHAR(4),
    URUNID VARCHAR(20),
    HAMID INTEGER,
    URUNADI VARCHAR(80),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    RENKDIGERADI VARCHAR(50),
    MIKTAR DOUBLE PRECISION,
    SIPARISIVERILENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMID VARCHAR(2),
    DOVIZID VARCHAR(5),
    TEDARIKCIFIRMA BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    VERILENSIPARISTIPIID INTEGER,
    VERILENSIPARISTIPIIDSTR VARCHAR(15),
    DEPARTMANID INTEGER,
    DEPARTMANADI VARCHAR(30),
    SATINALMATALEPYERIID INTEGER,
    SATINALMATALEPYERIADI VARCHAR(30),
    SATINALMAKULLANIMYERIID INTEGER,
    SATINALMAKULLANIMYERIADI VARCHAR(50),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    TALEPNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable FUrunAdi        varchar(80);
declare variable FHamAdi         varchar(255);
declare variable KS              integer;
begin
  For Select SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId, SatinAlmaTalep.Tarih, TK.Adi, SatinAlmaTalep.DepartmanId, Departman.Adi,
             SatinAlmaTalep.SatinAlmaTalepYeriId, SatinAlmaTalepYeri.Adi, SatinAlmaTalepDetay.SatinAlmaTalepDetayId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, SatinAlmaTalepDetay.BoyaSiparisDetayId,
             SatinAlmaTalepDetay.UrunId, SatinAlmaTalepDetay.HamId, Urun.Adi, Ham.Adi, SatinAlmaTalepDetay.Adi,
             SatinAlmaTalepDetay.RenkId, Renk.Adi, Renk.DigerAdi,
             SatinAlmaTalepDetay.Miktar, SatinAlmaTalepDetay.SiparisiVerilenMiktar, SatinAlmaTalepDetay.BirimFiyat, SatinAlmaTalepDetay.DovizId,
             UrunBirim.BirimId, SatinAlmaTalepDetay.TedarikciFirma, SatinAlmaTalepDetay.SatinAlmaKullanimYeriId,
             SatinAlmaKullanimYeri.Adi, SatinAlmaTalepDetay.TalepNotu, SatinAlmaTalepDetay.Notu,
             SatinAlmaTalep.VerilenSiparisTipiId, VerilenSiparisTipi.Adi,
             coalesce(Urun.UrunKategoriId, HamTipi.UrunKategoriId) FROM SatinAlmaTalepDetay
    LEFT JOIN SatinAlmaTalep ON SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
    LEFT JOIN Urun ON Urun.UrunId = SatinAlmaTalepDetay.UrunId
    LEFT JOIN Ham ON Ham.HamId = SatinAlmaTalepDetay.HamId
    LEFT JOIN HamTipi ON HamTipi.HamTipiId = Ham.HamTipiId
    left join UrunBirim on UrunBirim.UrunBirimId = SatinAlmaTalepDetay.UrunBirimId and UrunBirim.UrunId = SatinAlmaTalepDetay.UrunId
    LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = SatinAlmaTalepDetay.BoyaSiparisKayitId
    left join SatinAlmaKullanimYeri on SatinAlmaKullanimYeri.SatinAlmaKullanimYeriId = SatinAlmaTalepDetay.SatinAlmaKullanimYeriId
    LEFT JOIN Renk ON Renk.RenkId = SatinAlmaTalepDetay.RenkId
    left join Departman on Departman.DepartmanId = SatinAlmaTalep.DepartmanId
    left join SatinAlmaTalepYeri on SatinAlmaTalepYeri.SatinAlmaTalepYeriId = SatinAlmaTalep.SatinAlmaTalepYeriId
    left join VerilenSiparisTipi on VerilenSiparisTipi.VerilenSiparisTipiId = SatinAlmaTalep.VerilenSiparisTipiId
    Left Join Kullanici TK on TK.KullaniciId = SatinAlmaTalep.TalepKullaniciId
    WHERE SatinAlmaTalepDetay.Miktar > SatinAlmaTalepDetay.SiparisiVerilenMiktar and SatinAlmaTalepDetay.Kapali = 0 and
          (SatinAlmaTalep.SatinAlmaBirimID = :ASatinAlmaBirimID or :ASatinAlmaBirimID is null) and
          SatinAlmaTalep.OnayDurumu = 1 and
          (SatinAlmaTalep.VerilenSiparisTipiID = :AVerilenSiparisTipiID) and
          (Urun.UrunKategoriID = :AUrunKategoriID or HamTipi.UrunKategoriID = :AUrunKategoriID or :AUrunKategoriID is null)
  into :SatinAlmaTalepYil, :SatinAlmaTalepId, :TalepTarihi, :TalepEdenKullaniciAdi, :DepartmanId, :DepartmanAdi, :SatinAlmaTalepYeriId,
       :SatinAlmaTalepYeriAdi, :SatinAlmaTalepDetayId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId, :UrunId, :HamId, :UrunAdi, :FHamAdi,
       :FUrunAdi, :RenkId, :RenkAdi, :RenkDigerAdi, :Miktar, :SiparisiVerilenMiktar, :BirimFiyat, :DovizId, :BirimId, :TedarikciFirma, :SatinAlmaKullanimYeriId, :SatinAlmaKullanimYeriAdi,
       :TalepNotu, :Notu, :VerilenSiparisTipiId, :VerilenSiparisTipiIdStr, :UrunKategoriId do
  begin
    if (UrunId is null and HamId is null) then UrunAdi = FUrunAdi;
    else if (HamId is not null) then UrunAdi = FHamAdi;

    SatinAlmaTalepIdStr = '';
    if (SatinAlmaTalepId is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId);

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    KalanMiktar = Miktar - SiparisiVerilenMiktar;

    KS = 0;
    Select Count(*) from TeklifTalepDetay where TeklifTalepDetay.SatinAlmaTalepDetayId = :SatinAlmaTalepDetayId into :KS;

    if (KS = 0) then suspend;
  end
end
