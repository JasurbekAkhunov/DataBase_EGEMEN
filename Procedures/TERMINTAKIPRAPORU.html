CREATE PROCEDURE TERMINTAKIPRAPORU(
    ARAPORTIPI SMALLINT,
    ABASLANGICTERMINTARIHI TIMESTAMP,
    ABITISTERMINTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASON INTEGER,
    ASIPARISONAYDURUMU INTEGER,
    AASAMAID INTEGER,
    AASAMABASLAMAMIS INTEGER,
    ABOYAHANEBOLUMID INTEGER,
    AHAMID INTEGER,
    APARTITIPIID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ACILISTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    HAMADI VARCHAR(255),
    REFERANSKODU VARCHAR(30),
    PROSESID INTEGER,
    PROSESADI VARCHAR(255),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    RECETEKODU VARCHAR(20),
    RECETEADI VARCHAR(50),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    IPTALKILO NUMERIC(12,3),
    IPTALMETRE NUMERIC(12,3),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    SONYAPILANASAMAADI VARCHAR(40),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    SONRAKIASAMAADI VARCHAR(40),
    SONRAKIASAMAID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    TERMINNOTU VARCHAR(1024),
    SIPARISNOTU VARCHAR(1024),
    PLANLAMANOTU VARCHAR(1024),
    PARTIGECIKMETIPIADI VARCHAR(30),
    PARTIGECIKMENOTU VARCHAR(1024),
    BEKLEMEDE INTEGER,
    ISLEMTIPIID INTEGER,
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISTERMINTARIHI TIMESTAMP,
    SIPARISREVIZETERMINTARIHI TIMESTAMP,
    SIPARISTARIHI TIMESTAMP,
    SIPARISINSERTTARIHI TIMESTAMP,
    HAMSTOKTEMINTARIHI TIMESTAMP,
    SEVKEHAZIR VARCHAR(10),
    BEKLEMESURESI DOUBLE PRECISION,
    BEKLEMESURESISTR VARCHAR(30),
    BOLUMDEFAULTGUN DOUBLE PRECISION)
AS
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;

declare variable AtisNoStr       varchar(20);
declare variable SiparisKilo     Numeric(12,3);
declare variable SiparisMetre     Numeric(12,3);

declare variable SonYapilanSira      integer;
declare variable SonrakiSira      integer;

declare variable FTerminNotu   Blob;
declare variable FSiparisNotu Blob;
declare variable FPlanlamaNotu Blob;
declare variable FPartiGecikmeNotu Blob;
declare variable BlobSize    integer;
declare variable Okey integer;
declare variable KayitSayisi integer;

declare variable FPartiYil integer;
declare variable FPartiId  integer;
declare variable HamTipiId            integer;
declare variable FDurumu  integer;
declare variable PartilenenMiktardanIptaliDus EgeBoolean;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;
  select PartilenenMiktardanIptaliDus from Sistem where Sistem.SistemId = 1 into :PartilenenMiktardanIptaliDus;

  PartilenenKilo    = 0;
  PartilenenMetre   = 0;
  IptalKilo = 0;
  IptalMetre = 0;

  if (ARaporTipi = 2) then
  begin
    For Select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.BoyaSiparisKayitId, PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.OrmeSiparisDetayId,
               Parti.AcilisTarihi, Parti.Termintarihi, PartiStok.HamAdiToplami, Ham.ReferansKodu, PartiStok.ProsesId, Proses.Adi, Parti.ReceteId, Recete.Adi, Parti.FirmaId, Firma.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               SiparisProses.SiparisProsesID, SiparisProses.Adi, PartiStok.Kilo, PartiStok.Metre,
               Parti.Fason, Asama.Adi, PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi, PartiAsamalari.Sira, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.MusteriRenkKoduVeAdi,
               BoyaSiparisDetay.TerminNotu, BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu, BoyaSiparisDetay.TerminTarihi,
               BoyaSiparisDetay.RevizeTerminTarihi, GecikmeTipi.Adi, PartiGecikme.Notu, Parti.IslemTipiId,
               BoyaSiparis.SiparisiAlanKullaniciId, PartiStok.HamTipiId, BoyaSiparis.SiparisTarihi, BoyaSiparisDetay.InsertTarihi,
               BoyaSiparisDetay.HamStokTeminTarihi, PArtiStok.Durumu, BoyahaneBolum.DefaultGun from PartiStok
      left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
      left join Proses on Proses.ProsesId = PartiStok.ProsesId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join Firma on Firma.FirmaId = Parti.FirmaId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiStok.PartiAsamalariId
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiStok.PartiAsamalariId and PartiAsamaStok.PartiSinifID = PartiStok.PartiSinifId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      left join BoyahaneBolum on BoyahaneBolum.BoyahaneBolumID = Asama.BoyahaneBolumID
      left join PartiGecikme on PartiGecikme.PartiKayitId = Parti.PartiKayitId and PartiGecikme.Bitti = 0
      left join GecikmeTipi on GecikmeTipi.GecikmeTipiId = PartiGecikme.GecikmeTipiId
      left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
      where PartiStok.Durumu <= 7 and PartiStok.PartiStokIptalKayitId is null and
           (Parti.TerminTarihi >= :ABaslangicTerminTarihi and Parti.TerminTarihi <= :ABitisTerminTarihi) and
           (PartiStok.HamId = :AHamId or :AHamId is null) and
           (Parti.Fason = :AFason or :AFason is null) and
           (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
           ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :APartiTipiId = 0) or
             (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :APartiTipiId = 1) or
             ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :APartiTipiId = 2) or
             (Ham.HamTipiId = :hstDokumaKumas and :APartiTipiId = 3) or :APartiTipiId is null)
      into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :OrmeSiparisDetayId, :AcilisTarihi, :Termintarihi,
           :HamAdi, :ReferansKodu,  :ProsesId, :ProsesAdi, :ReceteKodu, :ReceteAdi, :FirmaId, :FirmaAdi, :BoyaSiparisYil, :BoyaSiparisId,
           :SiparisProsesID, :SiparisProsesAdi, :Kilo, :Metre, :Fason, :SonYapilanAsamaAdi, :GirisTarihi, :CikisTarihi, :SonYapilanSira,
           :MusteriSiparisNo, MusteriRenkKoduVeAdi, :FTerminNotu, :FSiparisNotu, :FPlanlamaNotu, :SiparisTerminTarihi, :SiparisRevizeTerminTarihi,
           :PartiGecikmeTipiAdi, :FPartiGecikmeNotu, :IslemTipiId, :SiparisiAlanKullaniciId, :HamTipiId,
           :SiparisTarihi, :SiparisInsertTarihi, :HamStokTeminTarihi, :FDurumu, :BolumDefaultGun do
    begin
      Okey = 1;
      if (FDurumu = 7) then SevkeHazir = 'Evet';
      else SevkeHazir = 'Hayir';

      if (GirisTarihi is not null and CikisTarihi is Null) then
      begin
        BeklemeSuresi = Current_TimeStamp(0) - GirisTarihi;
        BeklemeSuresiStr = TarihToStr(BeklemeSuresi);
      end
      else
      begin
        BeklemeSuresi = 0;
        BeklemeSuresiStr = '';
      end

      if (AAsamaId is not null) then
      begin
  /* 0 baslamamıs, 1 baslamıs, null hepsi*/
       if (AAsamaBaslamamis = 0) then
        Select Count(*) from PartiAsamaStok
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamaID = :AAsamaId and PartiAsamaStok.AsamayaDahil = 1 and
                PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.CikisTarihi is null into :Okey;
       else if (AAsamaBaslamamis = 1) then
        Select Count(*) from PartiAsamaStok
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamaID = :AAsamaId and PartiAsamaStok.AsamayaDahil = 1 and
                PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is null into :Okey;
       else
          Select Count(*) from PartiAsamaStok
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamaID = :AAsamaId and PartiAsamaStok.AsamayaDahil = 1 and
                PartiAsamaStok.CikisTarihi is null into :Okey;
      end
      else if (ABoyahaneBolumId is not null) then
      begin
       if (AAsamaBaslamamis = 0) then
         Select Count(*) from BolumPlanlama
           left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
           where BolumPlanlama.PartiKayitId = :PartiKayitId and BolumPlanlama.PartiSinifId = :PartiSinifId and BolumPlanlama.BoyahaneBolumId = :ABoyahaneBolumId and
                 PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.CikisTarihi is null into :Okey;
       else if (AAsamaBaslamamis = 1) then
         Select Count(*) from BolumPlanlama
           left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
           where BolumPlanlama.PartiKayitId = :PartiKayitId and BolumPlanlama.PartiSinifId = :PartiSinifId and BolumPlanlama.BoyahaneBolumId = :ABoyahaneBolumId and
                 PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is null into :Okey;
       else
         Select Count(*) from BolumPlanlama
           left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
           where BolumPlanlama.PartiKayitId = :PartiKayitId and BolumPlanlama.PartiSinifId = :PartiSinifId and BolumPlanlama.BoyahaneBolumId = :ABoyahaneBolumId and
                 PartiAsamaStok.CikisTarihi is null into :Okey;
      end


      if (Okey > 0) then
      begin
        PartiIdStr = '';
        BoyaSiparisIdStr = '';

        if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

        BlobSize = char_length(FTerminNotu);
        if (BlobSize > 1023) then TerminNotu = Left(FTerminNotu, 1023);
        else if (BlobSize > 0) then TerminNotu = Left(FTerminNotu, BlobSize);
        else TerminNotu = '';

        BlobSize = char_length(FSiparisNotu);
        if (BlobSize > 1023) then SiparisNotu = Left(FSiparisNotu, 1023);
        else if (BlobSize > 0) then SiparisNotu = Left(FSiparisNotu, BlobSize);
        else SiparisNotu = '';

        BlobSize = char_length(FPlanlamaNotu);
        if (BlobSize > 1023) then PlanlamaNotu = Left(FPlanlamaNotu, 1023);
        else if (BlobSize > 0) then PlanlamaNotu = Left(FPlanlamaNotu, BlobSize);
        else PlanlamaNotu = '';

        BlobSize = char_length(FPartiGecikmeNotu);
        if (BlobSize > 1023) then PartiGecikmeNotu = Left(FPartiGecikmeNotu, 1023);
        else if (BlobSize > 0) then PartiGecikmeNotu = Left(FPartiGecikmeNotu, BlobSize);
        else PartiGecikmeNotu = '';

        PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

        if (Fason = 1) then FasonStr = 'Fason';
        else FasonStr = '';

        if (SonYapilanSira is null) then SonYapilanSira = -1;

        select Min(PartiAsamalari.Sira) from PartiAsamaStok
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.Sira > :SonYapilanSira
          into :SonrakiSira;

        if (SonrakiSira is not null) then
        begin
          select PartiAsamaStok.AsamaId, Asama.Adi from PartiAsamaStok
            left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
            left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
            where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :SonrakiSira
            into :SonrakiAsamaId, :SonrakiAsamaAdi;
        end
        else
        begin
          SonrakiAsamaAdi    = Null;
          SonrakiAsamaId     = Null;
        end

        Select Count(*) from PartiStokBekle where PartiStokBekle.PartiSinifId = :PartiSinifId and PartiStokBekle.Bitti = 0 into :KayitSayisi;
        if (KayitSayisi > 0) then Beklemede = 1;
        else Beklemede = 0;

        suspend;
      end
    end
  end

  SevkeHazir = '';
  PartiKayitId       = Null;
  PartiYil           = Null;
  PartiId            = Null;
  PartiIdStr         = Null;
  PartiSinifId       = Null;
  BoyaSiparisDetayId = Null;
  BoyaSiparisKayitId = Null;
  TerminTarihi       = Null;
  HamAdi             = Null;
  ProsesId           = Null;
  ProsesAdi          = Null;
  FirmaId            = Null;
  FirmaAdi           = Null;
  BoyaSiparisYil     = Null;
  BoyaSiparisId      = Null;
  BoyaSiparisIdStr   = Null;
  ReceteKodu         = Null;
  ReceteAdi          = Null;
  SiparisProsesId    = Null;
  SiparisProsesAdi   = Null;
  Kilo               = Null;
  Metre              = Null;
  SonYapilanAsamaAdi = Null;
  GirisTarihi        = Null;
  CikisTarihi        = Null;
  SiparisiAlanKullaniciId = Null;
  SonrakiAsamaAdi    = Null;
  SonrakiAsamaId     = Null;
  PartiGecikmeTipiAdi = Null;
  PartiGecikmeNotu   = Null;
  Beklemede = Null;
  AcilisTarihi        = Null;
  HamTipiID = Null;
  SevkeHazir       = Null;
  BeklemeSuresi    = Null;
  BeklemeSuresiStr = Null;
  BolumDefaultGun  = Null;

  if (ARaporTipi = 1) then /*Boya Sipariş*/
  begin
    for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.OrmeSiparisDetayId,
               BoyaSiparis.FirmaId, Firma.Adi, LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr,
               BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisMetre,
               BoyaSiparisDetay.PartilenenKilo, BoyaSiparisDetay.IptalKilo, BoyaSiparisDetay.PartilenenMetre, BoyaSiparisDetay.IptalMetre,
               Ham.Adi, Ham.ReferansKodu, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi,
               BoyaSiparis.Fason, BoyaSiparisDetay.SonTerminTarihi,
               BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.MusteriRenkKoduVeAdi, BoyaSiparisDetay.TerminNotu,
               BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi,
               BoyaSiparisDetay.IslemTipiId, BoyaSiparis.SiparisiAlanKullaniciId, Ham.HamTipiId,
               BoyaSiparis.SiparisTarihi, BoyaSiparisDetay.InsertTarihi, BoyaSiparisDetay.HamStokTeminTarihi from BoyaSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
    where BoyaSiparisDetay.PartilenenKilo < BoyaSiparisDetay.SiparisKilo and BoyaSiparisDetay.OrmeSiparisDetayId is Null and
          (BoyaSiparisDetay.SonTerminTarihi >= :ABaslangicTerminTarihi and BoyaSiparisDetay.SonTerminTarihi <= :ABitisTerminTarihi) and
           BoyaSiparisDetay.KapatmaTarihi is null and
          (BoyaSiparisDetay.HamId = :AHamId or :AHamId is null) and
          (BoyaSiparis.Fason = :AFason or :AFason is null) and
          (BoyaSiparis.FirmaId = :AFirmaId or :AFirmaId is null) and
          (BoyaSiparisDetay.OnayDurumu = :ASiparisOnayDurumu or :ASiparisOnayDurumu is null) and
           ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :APartiTipiId = 0) or
             (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :APartiTipiId = 1) or
             ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :APartiTipiId = 2) or
             (Ham.HamTipiId = :hstDokumaKumas and :APartiTipiId = 3) or :APartiTipiId is null)
