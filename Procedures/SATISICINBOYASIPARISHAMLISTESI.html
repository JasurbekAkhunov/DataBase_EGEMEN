CREATE PROCEDURE SATISICINBOYASIPARISHAMLISTESI(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    RESULTID VARCHAR(100),
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER,
    SIPARISSIRANO INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    BOYASIPARISDETAYRENKADI VARCHAR(40),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    SIPARISKILO NUMERIC(12,3),
    MUSTERISIPARISNO VARCHAR(50),
    ARTIKELNO VARCHAR(20),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    TOPLAMSATIS NUMERIC(12,3),
    FATURAKESILECEK INTEGER,
    REZERVKILO NUMERIC(12,3),
    REFERANSSIPARISNO VARCHAR(30))
AS
declare variable FHamId                 integer;
declare variable FRenkId                integer;
declare variable DesenVaryantId         integer;
declare variable HamRenkadi             varchar(50);
declare variable FSevkKilo              Numeric(12,3);
declare variable FFirmaKodumuz          varchar(20);
declare variable DokumaSiparisi         integer;
declare variable shtSatis               integer;
declare variable FHamStokBirimID        integer;
begin
  select shtSatis from StokHareketTipiGetir into :shtSatis;

  BoyaSiparisBaskiId  = Null;
  DesenId             = Null;
  DesenAdi            = Null;
  VaryantAdi          = Null;

  Select BoyaSiparis.DokumaSiparisi from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitId = :ABoyaSiparisKayitId into :DokumaSiparisi;

  Select iif(:DokumaSiparisi = 1, Sistem.DokumaFirmaKodumuz, Sistem.OrmeFirmaKodumuz) from Sistem where sistemid = 1 into :FFirmaKodumuz;

  for Select OrmeSiparisDetay.OrmeSiparisDetayID, OrmeSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.RenkId, Renk.Adi,
             OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ToplamSatis, OrmeSiparisDetay.HamStokBirimID,
             OrmeSiparisDetay.ReferansSiparisno    from OrmeSiparisDetay
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = OrmeSiparisDetay.BoyaSiparisKayitID
  left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
  left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
  where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  order by OrmeSiparisDetay.SiparisSiraNo
  into :OrmeSiparisDetayID, :HamId, :HamAdi, :SiparisSiraNo, :RenkId, :RenkAdi, :SiparisKilo, :ArtikelNo, :MusteriSiparisNo, :ToplamSatis, :FHamStokBirimID, :ReferansSiparisNo do
  begin
    ResultId = '"' || EgemenIntegerToStr(OrmeSiparisDetayID) || '",' ;

    ResultId = ResultId ||'"",' ;
    ResultId = ResultId ||'"",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",' ;

    if (RenkId is not null) then
      ResultId = ResultId || '"' || EgemenIntegerToStr(RenkId) || '",' ;
    else
      ResultId = ResultId || '"",' ;

    ResultId = ResultId || '"", ' ;
    ResultId = ResultId || '"", ' ;
    ResultId = ResultId || '"", ' ;

    FaturaKesilecek = 1;
    ResultId = ResultId || '"' || EgemenIntegerToStr(FaturaKesilecek) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(FHamStokBirimID) || '"' ;


    suspend;
  end

  for Select OrmeSiparisDetay.OrmeSiparisDetayID, BoyaSiparisDetay.BoyaSiparisDetayId, OrmeSiparisDetay.HamId, BoyaSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.RenkId,
             OrmeSiparisDetay.RenkId, BoyaliRenk.Adi, HamRenk.Adi,
             BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.MusteriRenkKoduveAdi,
             BoyaSiparisDetay.SiparisKilo, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo,
             BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi, BoyaSiparisDetay.FaturaKesilecek, OrmeSiparisDetay.HamStokBirimID, OrmeSiparisDetay.ReferansSiparisno from BoyaSiparisDetay
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join Renk BoyaliRenk on BoyaliRenk.RenkId = BoyaSiparisDetay.RenkId
  left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
  left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
  where BoyaSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisDetay.BoyaSiparisDetayID is not null and BoyaSiparisDetay.Kapali = 0 --<> 2
  order by OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.BoyaSiparisDetayId
  into :OrmeSiparisDetayID, :BoyaSiparisDetayId, :FHamId, :HamId, :HamAdi, :SiparisSiraNo, :RenkId, :FRenkId, :RenkAdi, :HamRenkAdi,
       :LabReceteKodu, :LabReceteAdi, :BoyaSiparisDetayRenkAdi, :MusteriRenkKoduveAdi, :SiparisKilo, :ArtikelNo, :MusteriSiparisNo,
       :IslemTipiId, :IslemTipiAdi, :FaturaKesilecek, :FHamStokBirimID, :ReferansSiparisNo do
  begin
    if (HamID = FHamId and FRenkID is not null and RenkID is null) then
    begin
      RenkID = FRenkId;
      RenkAdi = HamRenkAdi;
    end

    if (OrmeSiparisDetayId is not null) then
      ResultId = '"' || EgemenIntegerToStr(OrmeSiparisDetayID) || '",' ;
    else
      ResultId = '"",' ;

    ResultId = ResultId ||'"' || EgemenIntegerToStr(BoyaSiparisDetayId) || '",' ;
    ResultId = ResultId ||'"",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",' ;

    if (RenkId is not null) then
      ResultId = ResultId || '"' || EgemenIntegerToStr(RenkId) || '",' ;
    else
      ResultId = ResultId || '"",' ;

    ResultId = ResultId || '"", ' ;
    ResultId = ResultId || '"", ' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(ISLEMTIPIID) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(FaturaKesilecek) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(FHamStokBirimID) || '"' ;

    ToplamSatis = Null;
    FSevkKilo = null;

    Select Sum(BoyaIrsaliyeDetay.Kilo) from BoyaSiparisDetay
    left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and PartiStok.BoyaSiparisBaskiId is null and BoyaIrsaliye.FirmaID <> :FFirmaKodumuz
    into :ToplamSatis;

    if (ToplamSatis is null) then ToplamSatis = 0;

    Select Sum(StokHareket.Miktar) from BoyaSiparisDetay
    left join StokHareket on StokHareket.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
    where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and StokHareket.BoyaSiparisBaskiId is null and
          StokHareket.HareketTipi = :shtSatis and StokHareket.HamIade = 0
    into :FSevkKilo;

    if (FSevkKilo is not null) then ToplamSatis = ToplamSatis + FSevkKilo;

    Select Sum(Kilo) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayId = :BoyaSiparisDetayId into :RezervKilo;

    suspend;
  end

  for Select OrmeSiparisDetay.OrmeSiparisDetayID, BoyaSiparisBaski.BoyaSiparisDetayId, BoyaSiparisBaski.BoyaSiparisBaskiId,
             OrmeSiparisDetay.HamId, BoyaSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.RenkId,
             OrmeSiparisDetay.RenkId, BoyaliRenk.Adi, HamRenk.Adi, BoyaSiparisDetay.DesenId, Desen.Adi, BoyaSiparisBaski.DesenVaryantId, DesenVaryant.VaryantAdi,
             BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.MusteriRenkKoduveAdi,
             BoyaSiparisDetay.SiparisKilo, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo,
             BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi, BoyaSiparisDetay.FaturaKesilecek, OrmeSiparisDetay.HamStokBirimID, OrmeSiparisDetay.ReferansSiparisno from BoyaSiparisBaski
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisBaski.BoyaSiparisDetayId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join Renk BoyaliRenk on BoyaliRenk.RenkId= BoyaSiparisDetay.RenkId
  left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
  left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
  left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
  where BoyaSiparisBaski.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisDetay.Kapali = 0 --<> 2
  order by BoyaSiparisBaski.BoyaSiparisKayitId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisBaski.BoyaSiparisDetayId, BoyaSiparisBaski.BoyaSiparisBaskiId
  into :OrmeSiparisDetayID, :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :FHamId, :HamId, :HamAdi, :SiparisSiraNo, :RenkId, :FRenkId, :RenkAdi, :HamRenkAdi,
       :DesenId, :DesenAdi, :DesenVaryantId, :VaryantAdi,
       :LabReceteKodu, :LabReceteAdi, :BoyaSiparisDetayRenkAdi, :MusteriRenkKoduveAdi, :SiparisKilo, :ArtikelNo, :MusteriSiparisNo,
       :IslemTipiId, :IslemTipiAdi, :FaturaKesilecek, :FHamStokBirimID, :ReferansSiparisNo do
  begin
    if (HamID = FHamId and FRenkID is not null) then
    begin
      RenkID = FRenkId;
      RenkAdi = HamRenkAdi;
    end
    if (OrmeSiparisDetayId is not null) then
      ResultId = '"' || EgemenIntegerToStr(OrmeSiparisDetayID) || '",' ;
    else
      ResultId = '"",' ;

    ResultId = ResultId ||'"' || EgemenIntegerToStr(BoyaSiparisDetayId) || '",' ;
    ResultId = ResultId ||'"' || EgemenIntegerToStr(BoyaSiparisBaskiId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",' ;

    if (RenkId is not null) then
      ResultId = ResultId || '"' || EgemenIntegerToStr(RenkId) || '",' ;
    else
      ResultId = ResultId || '"",' ;

    ResultId = ResultId || '"' || DesenId || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(DesenVaryantId) || '",' ;

    ResultId = ResultId || '"' || EgemenIntegerToStr(ISLEMTIPIID) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(FaturaKesilecek) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(FHamStokBirimID) || '"' ;

    ToplamSatis = Null;
    FSevkKilo = null;

    Select Sum(BoyaIrsaliyeDetay.Kilo) from BoyaSiparisBaski
    left join PartiStok on PartiStok.BoyaSiparisBaskiId = BoyaSiparisBaski.BoyaSiparisBaskiId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    where BoyaSiparisBaski.BoyaSiparisBaskiId = :BoyaSiparisBaskiId and BoyaIrsaliye.FirmaID <> :FFirmaKodumuz
    into :ToplamSatis;

    if (ToplamSatis is null) then ToplamSatis = 0;

    Select Sum(StokHareket.Miktar) from BoyaSiparisBaski
    left join StokHareket on StokHareket.BoyaSiparisBaskiID = BoyaSiparisBaski.BoyaSiparisBaskiId
    where BoyaSiparisBaski.BoyaSiparisBaskiId = :BoyaSiparisBaskiId and
          StokHareket.HareketTipi = :shtSatis and StokHareket.HamIade = 0
    into :FSevkKilo;

    if (FSevkKilo is not null) then ToplamSatis = ToplamSatis + FSevkKilo;

    Select Sum(Kilo) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayId = :BoyaSiparisDetayId into :RezervKilo;

    suspend;
  end
end
