CREATE PROCEDURE SATINALMATALEPKAYDET(
    AHGSIPARISKAYITID INTEGER,
    AHGSIPARISDETAYID INTEGER,
    ADEPARTMANID INTEGER,
    ASATINALMABIRIMID INTEGER,
    ASATINALMATALEPYERIID INTEGER,
    ATERMINTARIHI TIMESTAMP,
    AENKISASUREDE SMALLINT,
    ASATINALMAKULLANIMYERIID INTEGER,
    ASISTEMDOVIZID DOVIZSIZE,
    KUMASTALIMATIVER SMALLINT,
    AKSESUARTALIMATIVER SMALLINT,
    AKULLANICIID INTEGER)
RETURNS (
    SATINALMATALEPKAYITID INTEGER,
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER)
AS
declare variable DefaultSatinAlmaBirimID Integer;
declare variable FVerilenSiparisTipi SmallInt;
declare variable KS Integer;
begin
  if (KumasTalimatiVer = 1) then
  begin
    Select Count(*) from HGSiparisDetayKumas where HGSiparisDetayKumas.HGSiparisKayitID = :AHGSiparisKayitID and (HGSiparisDetayKumas.HGSiparisDetayID = :AHGSiparisDetayID or :AHGSiparisDetayID is Null) into :KS;
    if (KS = 0) then KumasTalimatiVer = 0;
  end
  if (AksesuarTalimatiVer = 1) then
  begin
    Select Count(*) from HGSiparisDetayAksesuar where HGSiparisDetayAksesuar.HGSiparisKayitID = :AHGSiparisKayitID and (HGSiparisDetayAksesuar.HGSiparisDetayID = :AHGSiparisDetayID or :AHGSiparisDetayID is Null) into :KS;
    if (KS = 0) then AksesuarTalimatiVer = 0;
  end

  if (KumasTalimatiVer <= 0 and AksesuarTalimatiVer <= 0) then Exception KayitHatasi ' Talimat oluşturmak için uygun Kumaş yada Aksesuar yok!';

  if (KumasTalimatiVer = 1) then
  begin
    KumasTalimatiVer = 0;
    for Select Distinct KumasSatinAlmaTalepKayitID, SatinAlmaTalepYil, SatinAlmaTalepID From HGSiparisDetay
        Left Join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitID = HGSiparisDetay.KumasSatinAlmaTalepKayitID
        where HGSiparisDetay.HGSiparisKayitID = :AHGSiparisKayitID and
        (HGSiparisDetay.HGSiparisDetayID = :AHGSiparisDetayID or :AHGSiparisDetayID is Null)
    into :SatinAlmaTalepKayitID, :SatinAlmaTalepYil, :SatinAlmaTalepID do
    begin
      if (SatinAlmaTalepKayitID is Null and KumasTalimatiVer = 0) then KumasTalimatiVer = 1;
      else if (SatinAlmaTalepKayitID is not Null) then Suspend;
    end
  end
  if (AksesuarTalimatiVer = 1) then
  begin
    AksesuarTalimatiVer = 0;
    for Select Distinct AksesuarSatinAlmaTalepKayitID, SatinAlmaTalepYil, SatinAlmaTalepID From HGSiparisDetay
        Left Join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitID = HGSiparisDetay.AksesuarSatinAlmaTalepKayitID
        where HGSiparisDetay.HGSiparisKayitID = :AHGSiparisKayitID and
        (HGSiparisDetay.HGSiparisDetayID = :AHGSiparisDetayID or :AHGSiparisDetayID is Null)
    into :SatinAlmaTalepKayitID, :SatinAlmaTalepYil, :SatinAlmaTalepID do
    begin
      if (SatinAlmaTalepKayitID is Null and AksesuarTalimatiVer = 0) then AksesuarTalimatiVer = 1;
      else if (SatinAlmaTalepKayitID is not Null) then Suspend;
    end
  end

  if (KumasTalimatiVer > 0) then
  begin
    FVerilenSiparisTipi = 0; /*Örme Kumas*/
    SatinAlmaTalepKayitId = gen_id(SatinAlmaTalepKayit_Id_Gen, 1);
    SatinAlmaTalepYil = F_Year(Current_Date);
    select ID from SATINALMATALEPNUMARATORUAL(:SatinAlmaTalepYil) into :SatinAlmaTalepId;

    Insert Into SatinAlmaTalep(SatinAlmaTalepKayitId, SatinAlmaTalepYil, SatinAlmaTalepId, DepartmanId, SatinAlmaTalepYeriId,
                                SatinAlmaBirimId, IstekYeri, VerilenSiparisTipiId, InsertKullaniciId, KullaniciId)
                         values (:SatinAlmaTalepKayitId, :SatinAlmaTalepYil, :SatinAlmaTalepId, :ADepartmanID, :ASatinAlmaTalepYeriId,
                                :ASatinAlmaBirimID, 1, :FVerilenSiparisTipi, :AKullaniciID, :AKullaniciId);

    Insert Into SatinAlmaTalepDetay(SatinAlmaTalepKayitId, HGSiparisKayitId, HGSiparisDetayId, HamId, RenkId, Miktar, BirimFiyat, DovizId,
                                    TupAcikEnMayli, MamulEn, MamulGramaj, TerminTarihi, EnKisaSurede, SatinAlmaKullanimYeriId, InsertKullaniciId, KullaniciId)
                             Select :SatinAlmaTalepKayitId, HGSiparisKayitId, HGSiparisDetayId, HGSiparisDetayKumas.HamID, HGSiparisDetayKumas.RenkID,
                                    Round(
                                      ( ((HGSiparisDetayKumas.Miktar / 1000) * iif(HGSiparisDetayKumas.FireOrani > 0, HGSiparisDetayKumas.FireOrani / 100 + 1, 1)) *
                                         iif(HGSiparisDetayKumas.FazlaKesimOrani > 0, HGSiparisDetayKumas.FazlaKesimOrani / 100 + 1, 1)
                                      ) *
                                      (Select Sum(Miktar) from HGSiparisDetayUrun where HGSiparisDetayUrun.HGSiparisDetayId = HGSiparisDetayKumas.HGSiparisDetayId and (HGSiparisDetayUrun.ModelKumasID = HGSiparisDetayKumas.ModelKumasID or HGSiparisDetayUrun.ModelKumasID = HGSiparisDetayKumas.AnaKumasModelKumasID ))
                                    ), HGSiparisDetayKumas.BirimFiyat, HGSiparisDetayKumas.BirimFiyatDovizID,
                                    iif(HGSiparisDetayKumas.TupAcikEnMayli is Null, 0, HGSiparisDetayKumas.TupAcikEnMayli), HGSiparisDetayKumas.BoyaliEn, HGSiparisDetayKumas.BoyaliGramaj, :ATerminTarihi, :AEnKisaSurede, :ASatinAlmaKullanimYeriId, :AKullaniciId, :AKullaniciId From HGSiparisDetayKumas
                               where HGSiparisDetayKumas.HGSiparisKayitID = :AHGSiparisKayitID and (HGSiparisDetayKumas.HGSiparisDetayID = :AHGSiparisDetayID or :AHGSiparisDetayID is Null);

    Update HGSiparisDetay set KumasSatinAlmaTalepKayitID = :SatinAlmaTalepKayitId where HGSiparisDetay.HGSiparisKayitID = :AHGSiparisKayitID and (HGSiparisDetay.HGSiparisDetayID = :AHGSiparisDetayID or :AHGSiparisDetayID is Null);
    Suspend;
  end
  if (AksesuarTalimatiVer > 0) then
  begin
    FVerilenSiparisTipi = 3; /*Diğer*/
    SatinAlmaTalepKayitId = gen_id(SatinAlmaTalepKayit_Id_Gen, 1);
    SatinAlmaTalepYil = F_Year(Current_Date);
    select ID from SATINALMATALEPNUMARATORUAL(:SatinAlmaTalepYil) into :SatinAlmaTalepId;

    Insert Into SatinAlmaTalep(SatinAlmaTalepKayitId, SatinAlmaTalepYil, SatinAlmaTalepId, DepartmanId, SatinAlmaTalepYeriId,
                                SatinAlmaBirimId, IstekYeri, VerilenSiparisTipiId, InsertKullaniciId, KullaniciId)
                         values (:SatinAlmaTalepKayitId, :SatinAlmaTalepYil, :SatinAlmaTalepId, :ADepartmanID, :ASatinAlmaTalepYeriId,
                                :ASatinAlmaBirimID, 1, :FVerilenSiparisTipi, :AKullaniciID, :AKullaniciId);

    Insert Into SatinAlmaTalepDetay(SatinAlmaTalepKayitId, HGSiparisKayitId, HGSiparisDetayId, UrunId, Miktar, UrunBirimId, BirimFiyat, DovizId,
                                    TerminTarihi, EnKisaSurede, SatinAlmaKullanimYeriId, InsertKullaniciId, KullaniciId, ResimDosyaAdi)
                            Select :SatinAlmaTalepKayitId, HGSiparisDetayAksesuar.HGSiparisKayitId, HGSiparisDetayAksesuar.HGSiparisDetayId, HGSiparisDetayAksesuar.UrunId,
                                   Round(Sum(HGSiparisDetayUrun.Miktar  * iif(FireOrani > 0, FireOrani / 100 + 1,1) *
                                        (HGSiparisDetayAksesuar.Miktar / Case When HGSiparisDetayAksesuar.BirimId Is Not Null then Iif(AksUrunBirim.Carpan = 0, 1.0, AksUrunBirim.Carpan) Else Iif(UrunBirim.Carpan = 0, 1.0, UrunBirim.Carpan) End)
                                      )),
                                   UrunBirim.UrunBirimID, HGSiparisDetayAksesuar.BirimFiyat, HGSiparisDetayAksesuar.BirimFiyatDovizId,
                                   :ATerminTarihi, :AEnKisaSurede, :ASatinAlmaKullanimYeriId, :AKullaniciId, :AKullaniciId, Urun.ResimDosyasi From HGSiparisDetayUrun
                              Left Join HGSiparisDetayAksesuar on HGSiparisDetayAksesuar.HGSiparisDetayID = HGSiparisDetayUrun.HGSiparisDetayID and
                                        (HGSiparisDetayAksesuar.ModelKumasID = HGSiparisDetayUrun.ModelKumasID or HGSiparisDetayAksesuar.ModelKumasID is Null) and
                                        (HGSiparisDetayAksesuar.ModelBedenID = HGSiparisDetayUrun.ModelBedenID or HGSiparisDetayAksesuar.ModelBedenID is Null)
                            left join Urun on Urun.UrunId = HGSiparisDetayAksesuar.UrunId
                            left join UrunBirim on UrunBirim.UrunId = HGSiparisDetayAksesuar.UrunId and UrunBirim.DefaultBirim = 1
                            left join UrunBirim AksUrunBirim on AksUrunBirim.UrunId = HGSiparisDetayAksesuar.UrunId and AksUrunBirim.BirimId = HGSiparisDetayAksesuar.BirimId
                            where HGSiparisDetayUrun.HGSiparisKayitID = :AHGSiparisKayitID and (HGSiparisDetayUrun.HGSiparisDetayID = :AHGSiparisDetayID or :AHGSiparisDetayID is Null)
                            Group By HGSiparisDetayAksesuar.HGSiparisKayitId, HGSiparisDetayAksesuar.HGSiparisDetayId, HGSiparisDetayAksesuar.UrunId,
                                     HGSiparisDetayAksesuar.BirimId, UrunBirim.UrunBirimID, HGSiparisDetayAksesuar.BirimFiyat, HGSiparisDetayAksesuar.BirimFiyatDovizId, Urun.ResimDosyasi;


    Update HGSiparisDetay set AksesuarSatinAlmaTalepKayitID = :SatinAlmaTalepKayitId where HGSiparisDetay.HGSiparisKayitID = :AHGSiparisKayitID and (HGSiparisDetay.HGSiparisDetayID = :AHGSiparisDetayID or :AHGSiparisDetayID is Null);

    Suspend;
  end
end
