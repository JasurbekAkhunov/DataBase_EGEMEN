CREATE PROCEDURE URUNSARFIYATRAPORUPARTILI_2(
    ARAPORTIPI INTEGER,
    AFIRMAID VARCHAR(20),
    ABASLANGICACILISTARIHI TIMESTAMP,
    ABITISACILISTARIHI TIMESTAMP,
    ABASLANGICRECETEYAZILISTARIHI TIMESTAMP,
    ABITISRECETEYAZILISTARIHI TIMESTAMP,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICURUNGRUP1ID VARCHAR(4),
    ABITISURUNGRUP1ID VARCHAR(4),
    ABASLANGICURUNGRUP2ID VARCHAR(4),
    ABITISURUNGRUP2ID VARCHAR(4),
    ABASLANGICURUNGRUP3ID VARCHAR(4),
    ABITISURUNGRUP3ID VARCHAR(4),
    ABASLANGICURUNGRUP4ID VARCHAR(4),
    ABITISURUNGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    ARAPORDOVIZID VARCHAR(5),
    AIMALATIBITMISOLANLAR INTEGER,
    APARTITIPIID INTEGER,
    AFASON INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIASAMALARIID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    BIRIMID VARCHAR(2),
    URUNTIPIID INTEGER,
    URUNTIPIADI VARCHAR(20),
    KIMYASALTIPIID INTEGER,
    KIMYASALTIPIADI VARCHAR(20),
    PLANLANANMIKTAR DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    RAPORDOVIZID VARCHAR(5),
    DOVIZBIRIMFIYAT DOUBLE PRECISION,
    DOVIZTUTAR DOUBLE PRECISION,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    FISNO INTEGER,
    STOKID INTEGER,
    STOKGRUPID VARCHAR(4),
    STOKGRUPADI VARCHAR(30),
    BIRIMFIYAT DOUBLE PRECISION,
    URETIMECIKISTARIHI TIMESTAMP,
    URETIMECIKISFIYATI DOUBLE PRECISION,
    URETIMECIKISFIYATIDOVIZ DOUBLE PRECISION,
    URETIMECIKISFIYATIDOVIZID VARCHAR(5),
    FISURETIMECIKISFIYATI DOUBLE PRECISION,
    OTOMASYON INTEGER,
    STOKDUSUMTIPI VARCHAR(30),
    PARTIASAMAILAVESARFID INTEGER,
    URUNGRUP1ADI VARCHAR(30),
    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30),
    LOT VARCHAR(150),
    MUHASEBEHESAPKODU ADI30)
AS
declare variable FDefaultBirimCarpan    double precision;
declare variable KayitSayisi            integer;
declare variable okey                   integer;
declare variable DovizKuru              Double Precision;
declare variable TabanDovizKuru              Double Precision;
declare variable StokTabanDovizId       varchar(5);
declare variable LastDovizId       varchar(5);
declare variable ARaporTarihi           Date;
begin
  ARaporTarihi = Current_TimeStamp(0);
  LastDovizId = '';
  RaporDovizId = ARaporDOvizId;

  Select StokTabanDovizId from sistem where sistemid = 1 into :StokTabanDovizId;

  if (StokTabanDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:StokTabanDovizId, :ARaporDovizId, Current_date) into :TabanDovizKuru;
  else TabanDovizKuru = 1;

  if (ARaporTipi = 0) then
  begin
    for select Parti.PartiKayitId  from Parti
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      Where (Parti.PartiYil > :ABaslangicPartiYil or (Parti.PartiYil = :ABaslangicPartiYil and Parti.PartiId >= :ABaslangicPartiId)) and
            (Parti.PartiYil < :ABitisPartiYil or (Parti.PartiYil = :ABitisPartiYil  and Parti.PartiId <= :ABitisPartiId)) and
            (Parti.FirmaId = :AFirmaID or :AFirmaId is null) and
            (Parti.PartiTipiId = :APartiTipiId or :APartiTipiId is null) and
            (Parti.Fason = :AFason or :AFason is null) and
            (:ABaslangicBoyaSiparisYil = 0 or
             ((BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
             (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId))))
    into :PartiKayitId do
    begin
      if (AImalatiBitmisOlanlar = 1) then
      begin
        Select count(*) from PartiStok where PartiStok.PartiKayitId = :PartiKayitId and Durumu < 7 into :KayitSayisi;
        if (KayitSayisi = 0) then Okey = 1;
        else Okey = 0;
      end
      else Okey = 1;

      if (okey = 1) then
      for select StokDetay.UrunId, Urun.Adi, Urun.UrunTipiId, UrunTipi.Adi, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3, Urun.UrunGrup4,
                   PartiAsamalari.ReceteId, Recete.Adi, UBSD.BirimId, StokDetay.UretimeCikisTarihi, Stok.FisNo, StokDetay.StokId,
                   StokDetay.Miktar, StokDetay.PlanlananMiktar, StokDetay.BirimFiyat,
                   UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / UBSD.Carpan, UrunFiyat.UretimeCikisFiyati, UrunFiyat.UretimeCikisFiyatiDovizId, StokDetay.PartiAsamalariId,
                   StokDetay.Otomasyon, Urun.KimyasalTipiId, KimyasalTipi.Adi, StokDetay.PartiAsamaIlaveSarfID,
                   stokdetay.StokGrupId, StokHareketGrup.Adi,
                   UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi, StokDetay.BirimFiyat, StokDetay.Lot, Urun.MuhasebeHesapKodu from stokdetay
        left join Urun on Urun.UrunId = StokDetay.UrunId
        left join UrunBirim UBSD on UBSD.UrunBirimId = StokDetay.UrunBirimId
        left join UrunFiyat on UrunFiyat.UrunId = StokDetay.UrunId
        left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
        left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
        left join Parti on Parti.PartiKayitId = StokDetay.PartiKayitId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = StokDetay.PartiAsamalariId
        left join recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join Stok on Stok.StokId = StokDetay.StokId
        left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = urun.KimyasalTipiid
        left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = stokdetay.StokGrupId
        left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
        left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
        left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
        left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
       where StokDetay.PartiKayitId = :PartiKayitId and StokDetay.HareketTipi = 6 and
              ((:AUrunAralikli = 1 and StokDetay.UrunId >= :ABaslangicUrunId and (StokDetay.UrunId <= :ABitisUrunId or :ABitisUrunId = '')) or (:AUrunAralikli = 0))  and
              ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicUrunGrup1Id  and (UrunGrup1 <= :ABitisUrunGrup1Id or :ABitisUrunGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicUrunGrup1Id = '' and :ABitisUrunGrup1Id = '')) and
              ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicUrunGrup2Id and (UrunGrup2 <= :ABitisUrunGrup2Id or :ABitisUrunGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicUrunGrup2Id = '' and :ABitisUrunGrup2Id = '')) and
              ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicUrunGrup3Id and (UrunGrup3 <= :ABitisUrunGrup3Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicUrunGrup3Id = '' and :ABitisUrunGrup3Id = '')) and
              ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicUrunGrup4Id and (UrunGrup4 <= :ABitisUrunGrup4Id or :ABitisUrunGrup4Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicUrunGrup4Id = '' and :ABitisUrunGrup4Id = ''))
        order by UrunFiyat.UretimeCikisFiyatiDovizId /* AynÄ± doviz cinsleri icin dovizkuru getire gitmesin diye sirali okuyoruz..*/
        into :UrunId, :UrunAdi, :UrunTipiId, :UrunTipiAdi, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :ReceteId, :ReceteAdi, :BirimId,  :UretimeCikisTarihi, :FisNo, :StokId,
           :Miktar, :PlanlananMiktar, :BirimFiyat, :UretimeCikisFiyati, :UretimeCikisFiyatiDoviz, :UretimeCikisFiyatiDovizId, :PartiAsamalariId, :Otomasyon,
           :KimyasalTipiId, :KimyasalTipiAdi, :PartiAsamaIlaveSarfID, :StokGrupId, :StokGrupAdi, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi,
           :FisUretimeCikisFiyati, :Lot, :MuhasebeHesapKodu do
      begin
        Okey = 1;
        if (AUrunAralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

        if (okey = 1) then
        if (AUrunGrup1Aralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunGrup1Id, UrunGrup1) = 0) then okey = 0;

        if (okey = 1) then
        if (AUrunGrup2Aralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunGrup2Id, UrunGrup2) = 0) then okey = 0;

        if (okey = 1) then
        if (AUrunGrup3Aralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunGrup3Id, UrunGrup3) = 0) then okey = 0;

        if (okey = 1) then
        if (AUrunGrup4Aralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunGrup4Id, UrunGrup4) = 0) then okey = 0;

        if (okey = 1) then
        begin
          if (UretimeCikisFiyatiDovizId is not null) then
          begin
            if (LastDovizId <> UretimeCikisFiyatiDovizId) then
            begin
              if (UretimeCikisFiyatiDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:UretimeCikisFiyatiDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
              else DovizKuru = 1;

              if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;
            end

            UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;

            LastDovizId = UretimeCikisFiyatiDovizId;
            if (UretimeCikisFiyati is null) then UretimeCikisFiyati = 0;

          end

          DovizBirimFiyat = BirimFiyat * TabanDovizKuru;
          DovizTutar = DovizBirimFiyat * Miktar;
          if (PartiAsamalariId is null) then PartiAsamalariId = -1;

          if (Otomasyon = 0) then StokDusumTipi = 'Manuel';
          else if (Otomasyon = 1) then StokDusumTipi = 'Teorik';
          else if (Otomasyon = 2) then StokDusumTipi = 'Dozajlama Otomatik';
          else if (Otomasyon = 3) then StokDusumTipi = 'Dozajlama Teorik';
          else if (Otomasyon = 4) then StokDusumTipi = 'Dozajlama Manuel';
          suspend;
        end
      end
    end
  end
  else
  begin
    for select Parti.PartiKayitId  from Parti
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      where (((:ARaporTipi = 1) and (Parti.AcilisTarihi >= :ABaslangicAcilisTarihi and Parti.AcilisTarihi <= :ABitisAcilisTarihi )) or
             ((:ARaporTipi = 2) and (Parti.ReceteYazilisTarihi >= :ABaslangicReceteYazilisTarihi and Parti.ReceteYazilisTarihi <= :ABitisReceteYazilisTarihi)))  and
            (Parti.FirmaId = :AFirmaID or :AFirmaId is null) and
            (Parti.PartiTipiId = :APartiTipiId or :APartiTipiId is null) and
            (Parti.Fason = :AFason or :AFason is null) and
            (:ABaslangicBoyaSiparisYil = 0 or
             ((BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
             (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId))))
    into :PartiKayitId do
    begin
      if (AImalatiBitmisOlanlar = 1) then
      begin
        Select count(*) from PartiStok where PartiStok.PartiKayitId = :PartiKayitId and Durumu < 7 into :KayitSayisi;
        if (KayitSayisi = 0) then Okey = 1;
        else Okey = 0;
      end
      else Okey = 1;

      if (okey = 1) then
      for select StokDetay.UrunId, Urun.Adi, Urun.UrunTipiId, UrunTipi.Adi, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3, Urun.UrunGrup4,
                   PartiAsamalari.ReceteId, Recete.Adi, UBSD.BirimId, StokDetay.UretimeCikisTarihi, Stok.FisNo, StokDetay.StokId,
                   StokDetay.Miktar, StokDetay.PlanlananMiktar, StokDetay.BirimFiyat,
                   UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / UBSD.Carpan, UrunFiyat.UretimeCikisFiyati, UrunFiyat.UretimeCikisFiyatiDovizId, StokDetay.PartiAsamalariId,
                   StokDetay.Otomasyon, Urun.KimyasalTipiId, KimyasalTipi.Adi, StokDetay.PartiAsamaIlaveSarfID,
                   stokdetay.StokGrupId, StokHareketGrup.Adi,
                   UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi, StokDetay.BirimFiyat, StokDetay.Lot, Urun.MuhasebeHesapKodu from stokdetay
        left join Urun on Urun.UrunId = StokDetay.UrunId
        left join UrunBirim UBSD on UBSD.UrunBirimId = StokDetay.UrunBirimId
        left join UrunFiyat on UrunFiyat.UrunId = StokDetay.UrunId
        left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
        left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
        left join Parti on Parti.PartiKayitId = StokDetay.PartiKayitId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = StokDetay.PartiAsamalariId
        left join recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join Stok on Stok.StokId = StokDetay.StokId
        left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = stokdetay.StokGrupId
        left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = urun.KimyasalTipiid
        left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
        left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
        left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
        left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
        where StokDetay.PartiKayitId = :PartiKayitId and StokDetay.HareketTipi = 6 and
              ((:AUrunAralikli = 1 and StokDetay.UrunId >= :ABaslangicUrunId and (StokDetay.UrunId <= :ABitisUrunId or :ABitisUrunId = '')) or (:AUrunAralikli = 0))  and
              ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicUrunGrup1Id  and (UrunGrup1 <= :ABitisUrunGrup1Id or :ABitisUrunGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicUrunGrup1Id = '' and :ABitisUrunGrup1Id = '')) and
              ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicUrunGrup2Id and (UrunGrup2 <= :ABitisUrunGrup2Id or :ABitisUrunGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicUrunGrup2Id = '' and :ABitisUrunGrup2Id = '')) and
              ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicUrunGrup3Id and (UrunGrup3 <= :ABitisUrunGrup3Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicUrunGrup3Id = '' and :ABitisUrunGrup3Id = '')) and
              ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicUrunGrup4Id and (UrunGrup4 <= :ABitisUrunGrup4Id or :ABitisUrunGrup4Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicUrunGrup4Id = '' and :ABitisUrunGrup4Id = ''))
        order by UrunFiyat.UretimeCikisFiyatiDovizId /* AynÄ± doviz cinsleri icin dovizkuru getire gitmesin diye sirali okuyoruz..*/
        into :UrunId, :UrunAdi, :UrunTipiId, :UrunTipiAdi, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :ReceteId, :ReceteAdi, :BirimId,  :UretimeCikisTarihi, :FisNo, :StokId,
           :Miktar, :PlanlananMiktar,  :BirimFiyat, :UretimeCikisFiyati, :UretimeCikisFiyatiDoviz, :UretimeCikisFiyatiDovizId, :PartiAsamalariId, :Otomasyon,
           :KimyasalTipiId, :KimyasalTipiAdi, :PartiAsamaIlaveSarfID, :StokGrupId, :StokGrupAdi, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi,
           :FisUretimeCikisFiyati, :Lot, :MuhasebeHesapKodu do
      begin
        Okey = 1;
        if (AUrunAralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

        if (okey = 1) then
        if (AUrunGrup1Aralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunGrup1Id, UrunGrup1) = 0) then okey = 0;

        if (okey = 1) then
        if (AUrunGrup2Aralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunGrup2Id, UrunGrup2) = 0) then okey = 0;

        if (okey = 1) then
        if (AUrunGrup3Aralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunGrup3Id, UrunGrup3) = 0) then okey = 0;

        if (okey = 1) then
        if (AUrunGrup4Aralikli = 0) then
        if (KutuKarsilastir(ABaslangicUrunGrup4Id, UrunGrup4) = 0) then okey = 0;

        if (okey = 1) then
        begin
          if (UretimeCikisFiyatiDovizId is not null) then
          begin
            if (LastDovizId <> UretimeCikisFiyatiDovizId) then
            begin
              if (UretimeCikisFiyatiDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:UretimeCikisFiyatiDovizId, :ARaporDovizId, Current_date) into :DovizKuru;
              else DovizKuru = 1;

              if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;
            end

            UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;

            LastDovizId = UretimeCikisFiyatiDovizId;
            if (UretimeCikisFiyati is null) then UretimeCikisFiyati = 0;

          end

          DovizBirimFiyat = BirimFiyat * TabanDovizKuru;
          DovizTutar = DovizBirimFiyat * Miktar;
          if (PartiAsamalariId is null) then PartiAsamalariId = -1;

          if (Otomasyon = 0) then StokDusumTipi = 'Manuel';
          else if (Otomasyon = 1) then StokDusumTipi = 'Teorik';
          else if (Otomasyon = 2) then StokDusumTipi = 'Dozajlama Otomatik';
          else if (Otomasyon = 3) then StokDusumTipi = 'Dozajlama Teorik';
          else if (Otomasyon = 4) then StokDusumTipi = 'Dozajlama Manuel';
          suspend;
        end
      end
    end
  end
