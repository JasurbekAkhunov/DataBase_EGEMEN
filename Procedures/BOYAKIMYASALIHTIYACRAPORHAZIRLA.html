CREATE PROCEDURE BOYAKIMYASALIHTIYACRAPORHAZIRLA(
    AIMALATTIPIID INTEGER,
    AHAMTIPIID INTEGER)
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(50),
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BIRIMID VARCHAR(2),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    URUNGRUP1ADI VARCHAR(30),
    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30),
    MIKTAR DOUBLE PRECISION,
    FIRMAADI VARCHAR(100),
    MAKINAPLANLAMAGRUPID VARCHAR(10))
AS
declare variable APartiKayitId       integer;
declare variable APartiAsamalariId   integer;
declare variable AReceteId           varchar(20);
declare variable ABanyoCarpani       Double Precision;
declare variable ABanyoCarpani2      Double Precision;
declare variable ABanyoCarpani3      Double Precision;
declare variable ABanyoCarpani4      Double Precision;
declare variable AKoyultma           Double Precision;
declare variable FLabReceteGrupId    integer;

declare variable OrjPartiKilo             Double Precision;
declare variable FPartiKilo             Double Precision;
declare Variable FMiktar                Double Precision;
declare Variable FBanyo                 Double Precision;
declare variable FReceteMiktarOrani     Double Precision;
declare variable FKimyasalTipiReceteOrani Double Precision;
Declare variable FReceteAsamaId           integer;
Declare variable FBoyaAsamasimi           integer;
Declare variable FFBoyaAsamasimi          integer;
Declare variable FBanyoCarpanNo           integer;
Declare variable FFBanyoCarpanNo          integer;

declare variable FToplamKilo            Double Precision;
declare variable FKilo                  Double Precision;
declare variable FPartiKayitId          integer;
declare variable FPartiAsamalariId      integer;
declare variable ReceteArsiviOlusturulsun integer;
declare variable OtomasyonaDahilmi        integer;
declare variable ORJBanyoCarpani          Double Precision;
declare variable UstAsamaBoyaAsamasimi    integer;
declare variable KiloSifirOlsadaReceteyiDus    integer;

declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
begin
  select RECETEMIKTARORANI, KiloSifirOlsadaReceteyiDus from SISTEM where SISTEMID = 1 into :FReceteMiktarOrani, :KiloSifirOlsadaReceteyiDus;
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;

  For Select Parti.PartiYil, Parti.PartiId, Parti.ReceteId, Recete.Adi,
             PartiAsamaStok.PartiKayitId, PartiAsamaStok.PartiAsamalariId, PartiAsamalari.ReceteId,
             PartiAsamalari.BanyoCarpani, PartiAsamalari.BanyoCarpani2, PartiAsamalari.BanyoCarpani3, PartiAsamalari.BanyoCarpani4,
             PartiAsamalari.Koyultma, Recete.LabReceteGrupId, Firma.Adi, PartiAsamaStok.MakinaPlanlamaGrupId From PartiAsamaStok
  left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
  left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
  left join Ham on Ham.HamId = PartiStok.HamId
  left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
  left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
  left join Firma on Firma.FirmaId = Parti.FirmaId
  where PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.ReceteID is not null and PartiAsamalari.otomasyonadahilmi=1 and
        (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null) and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
            (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or
            ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 2) or
            (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 3) or :AHamTipiId is null)
  Group by Parti.PartiYil, Parti.PartiId, Parti.ReceteId, Recete.Adi,PartiAsamaStok.PartiKayitId, PartiAsamaStok.PartiAsamalariId, PartiAsamalari.ReceteId,
             PartiAsamalari.BanyoCarpani, PartiAsamalari.BanyoCarpani2, PartiAsamalari.BanyoCarpani3, PartiAsamalari.BanyoCarpani4,
             PartiAsamalari.Koyultma, Recete.LabReceteGrupId, Firma.Adi,PartiAsamaStok.MakinaPlanlamaGrupId
  into :PartiYil, :PartiId, :ReceteId, :ReceteAdi, :APartiKayitId, :APartiAsamalariId, :AReceteId,
  :ABanyoCarpani, :ABanyoCarpani2, :ABanyoCarpani3, :ABanyoCarpani4, :AKoyultma, :FLabReceteGrupId, :FirmaAdi, :MakinaPlanlamaGrupId do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    FToplamKilo = 0;
    FPartiKilo = 0;

    Select Asama.BoyaAsamasimi from Recete
      left join Asama on Asama.AsamaId = Recete.UstAsamaId
      where Recete.ReceteID = :AReceteID
      into :UstAsamaBoyaAsamasimi;

    if (ABanyoCarpani is null or ABanyoCarpani = 0) then
    if (UstAsamaBoyaAsamasimi = 1) then ABanyoCarpani = -1;
    else ABanyoCarpani = 0;

    if (ABanyoCarpani2 is null) then ABanyoCarpani2 = 0;
    if (ABanyoCarpani3 is null) then ABanyoCarpani3 = 0;
    if (ABanyoCarpani4 is null) then ABanyoCarpani4 = 0;

    if (ABanyoCarpani >= 0) then
    begin
      for select ReceteAsama.BanyoCarpanNo from ReceteDetay
        left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
        where ReceteDetay.receteid = :AReceteId and ReceteDetay.ReceteAsamaId is not null
        group by ReceteAsama.BanyoCarpanNo
        into :FFBanyoCarpanNo do
      begin
        if (FFBanyoCarpanNo = 1 and UstAsamaBoyaAsamasimi = 1 and (ABanyoCarpani is null or ABanyoCarpani = 0) ) then ABanyoCarpani = -1;
        else if (FFBanyoCarpanNo = 2 and (ABanyoCarpani2 is null or ABanyoCarpani2 = 0) ) then ABanyoCarpani = -1;
        else if (FFBanyoCarpanNo = 3 and (ABanyoCarpani3 is null or ABanyoCarpani3 = 0) ) then ABanyoCarpani = -1;
        else if (FFBanyoCarpanNo = 4 and (ABanyoCarpani4 is null or ABanyoCarpani4 = 0) ) then ABanyoCarpani = -1;
      end
    end

    if (ABanyoCarpani > -1 and ABanyoCarpani2 > -1 and ABanyoCarpani3 > -1 and ABanyoCarpani4 > -1) then
    begin
      for select PartiKayitId, PartiAsamalariId from PartiRecetesiDokumu_1(:APartiKayitId, :AReceteId, :APartiAsamalariId)
        into :FPartiKayitId, :FPartiAsamalariId do
      begin
        select Sum(iif(:KiloSifirOlsadaReceteyiDus = 0, PartiAsamaStok.GirisKilo, PartiAsamaStok.GirisMetre))  from PartiAsamalari
          left join partiasamastok on partiasamastok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
          where PartiAsamalari.PartiKayitId = :FPartiKayitId and PartiAsamalari.PartiAsamalariId = :FPartiAsamalariId and PartiAsamalari.ReceteId = :AReceteId and PartiAsamaStok.AsamayaDahil = 1
          into :FKilo;

        if (APartiKayitId = FPartiKayitId) then FPartiKilo = FKilo;
        FToplamKilo = FToplamKilo + FKilo;
      end

      if (FPartiKilo is null) then FPartiKilo = 0;
      if (FToplamKilo is null) then FToplamKilo = 0;

      if (FPartiKilo > 0 and FToplamKilo > 0) then
      begin
        OrjPartiKilo = FPartiKilo;

        if (AKoyultma is null or AKoyultma = 0) then AKoyultma = 1;

        if (ABanyoCarpani <> 0) then ABanyoCarpani = ABanyoCarpani / FToplamKilo * FPartiKilo;
        else ABanyoCarpani = FPartiKilo;

        ABanyoCarpani = Yuvarla(ABanyoCarpani, 1);

        if (ABanyoCarpani2 <> 0) then ABanyoCarpani2 = ABanyoCarpani2 / FToplamKilo * FPartiKilo;
        ABanyoCarpani2 = Yuvarla(ABanyoCarpani2, 1);

        if (ABanyoCarpani3 <> 0) then ABanyoCarpani3 = ABanyoCarpani3 / FToplamKilo * FPartiKilo;
        ABanyoCarpani3 = Yuvarla(ABanyoCarpani3, 1);

        if (ABanyoCarpani4 <> 0) then ABanyoCarpani4 = ABanyoCarpani4 / FToplamKilo * FPartiKilo;
        ABanyoCarpani4 = Yuvarla(ABanyoCarpani4, 1);

        FPartiKilo = FPartiKilo / FReceteMiktarOrani;

        FFBanyoCarpanNo = 1;
        ORJBanyoCarpani = ABanyoCarpani;

        for select ReceteDetay.ReceteAsamaId, ReceteDetay.UrunId, Urun.Adi, DUB.BirimId,
                   Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi,
                   ReceteDetay.Miktar/ RUB.Carpan, ReceteDetay.Banyo/ RUB.Carpan, KimyasalTipi.ReceteOrani, ReceteAsama.BoyaAsamasimi,
                   ReceteAsama.BanyoCarpanNo from recetedetay
          left join UrunBirim RUB on RUB.urunBirimId = ReceteDetay.urunBirimId
          left join Urun on Urun.UrunId = ReceteDetay.UrunId
          left join UrunBirim DUB on DUB.UrunId = ReceteDetay.UrunId and DUB.DefaultBirim = 1
          left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = Urun.KimyasalTipiId
          left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
          left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
          left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
          left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
          left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
          where ReceteDetay.Receteid = :AReceteId
          order by ReceteDetay.Sira
          into :FReceteAsamaId, :UrunId, :UrunAdi, :BirimId, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi,
               :FMiktar, :FBanyo, :FKimyasalTipiReceteOrani, :FBoyaAsamasimi, :FBanyoCarpanNo do
        begin
          if (FReceteAsamaId is not null) then
          begin
            FFBoyaAsamasimi = FBoyaAsamasimi;
            FFBanyoCarpanNo = FBanyoCarpanNo;

            if (FFBanyoCarpanNo = 2) then ORJBanyoCarpani = ABanyoCarpani2;
            else if (FFBanyoCarpanNo = 3) then ORJBanyoCarpani = ABanyoCarpani3;
            else if (FFBanyoCarpanNo = 4) then ORJBanyoCarpani = ABanyoCarpani4;
            else ORJBanyoCarpani = ABanyoCarpani;
          end

          if (UrunId is not null) then
          if ((FBanyo is not null and FBanyo > 0) or (FMiktar is not null and FMiktar > 0)) then
          begin
            FMiktar = FMiktar * FPartiKilo * AKoyultma;

            if (FLabReceteGrupId = 0) then FBanyo = FBanyo * OrjBanyoCarpani * AKoyultma;
            else FBanyo = FBanyo * OrjBanyoCarpani;

            if (FFBoyaAsamasimi = 1) then
            if (FKimyasalTipiReceteOrani is not null) then FBanyo = FBanyo / FKimyasalTipiReceteOrani;

            if (FMiktar is null) then FMiktar = 0;
            if (FBanyo is null) then FBanyo = 0;

            Miktar = FMiktar + FBanyo;
            Suspend;
          end
        end
      end
    end

  end
end
