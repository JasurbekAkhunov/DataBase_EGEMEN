CREATE PROCEDURE BOYASIPARISREZERVSERTIFIKAGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISREZERVSERTIFIKAID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISSARJID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    HAMSTOKIRSALIYEID INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    KAPADEDI INTEGER,
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    PARTILENENADET INTEGER,
    PARTILENENKAPADEDI INTEGER,
    PARTIIDSTR VARCHAR(255),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    KALANADET INTEGER,
    KALANKAPADEDI INTEGER,
    LOT VARCHAR(150),
    PUS INTEGER,
    FEINE INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    ESKIPARTIYIL INTEGER,
    ESKIPARTINO INTEGER,
    ESKIPARTINOSTR VARCHAR(30),
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30),
    ORMECIMAKINANO VARCHAR(10),
    MUSTERIPARTINO VARCHAR(30),
    RAFIDSTR VARCHAR(50),
    TEDARIKCIFIRMAADI VARCHAR(100),
    URETICIFIRMAADI VARCHAR(100),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    ORMESIPARISYIL INTEGER,
    ORMESIPARISID INTEGER,
    ORMESIPARISIDSTR VARCHAR(30),
    ORMEURETIMPARTINO VARCHAR(30),
    YARIMAMULPARTINOSTR VARCHAR(20),
    OLCUID INTEGER,
    OLCUADI ADI40,
    SERTIFIKANO VARCHAR(30),
    MIKTAR BOYAMIKTARSIZE,
    BOYASIPARISREZERVID INTEGER)
AS
declare variable LastBoyaSiparisDetayId integer;
declare variable LastBoyaSiparisSarjId  integer;
declare variable LastHamStokDetayId     integer;
declare variable KullanilanKilo         Numeric(12,3);
declare variable KullanilanMetre        Numeric(12,3);
declare variable KullanilanAdet         integer;
declare variable KullanilanKapAdedi     integer;
declare variable FKullanilanKilo        Numeric(12,3);
declare variable FKullanilanMetre       Numeric(12,3);
declare variable FKullanilanAdet        integer;
declare variable FKullanilanKapAdedi    integer;
declare variable FRafId                 varchar(10);
Declare variable FPartiYil integer;
Declare variable FPartiId  integer;
Declare variable FPartiSiraNo integer;
declare variable FPartiIdStr varchar(30);
Declare variable YariMamulPartiYil integer;
Declare variable YariMamulPartiId  integer;
begin
  LastBoyaSiparisDetayId = -1;
  LastHamStokDetayId = -1;

  For select BoyaSiparisRezerv.BoyaSiparisRezervId, BoyaSiparisRezervSertifika.BoyaSiparisRezervSertifikaId, BoyaSiparisRezervSertifika.BoyaSiparisKayitId, BoyaSiparisRezervSertifika.BoyaSiparisDetayId,
             BoyaSiparisRezervSertifika.HamStokDetayId, HamStokDetay.HamStokMasterId, HamStokDetay.HamStokIrsaliyeId,
             HamStokMaster.Lot, HamStokMaster.Pus, HamStokMaster.Feine,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, Ormeci.Adi, HamStokMaster.OrmeciMakinaNo, HamStokMaster.MusteriPartiNo,
             Ik.Adi, BoyaSiparisRezervSertifika.Inserttarihi, Sk.Adi, BoyaSiparisRezervSertifika.DegisimTarihi,
             BoyaSiparisRezerv.KullanilanKilo, BoyaSiparisRezerv.KullanilanMetre, BoyaSiparisRezerv.KullanilanAdet, BoyaSiparisRezerv.KullanilanKapAdedi,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, IrsaliyeFirma.Adi, UreticiFirma.Adi, HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo, HamStokMaster.TupAcikEnMayli,
             Coalesce(OrmePlanlama.UretimPartiNo, DokumaPlanlama.UretimPartiNo), BS.BoyaSiparisId, BS.BoyaSiparisYil, Parti.PartiYil, Parti.PartiId, Olcu.OlcuId, Olcu.OlcuAdi, HamStokMaster.SertifikaNo, BoyaSiparisRezervSertifika.Miktar,
             BoyaSiparisRezervSertifika.Kilo, BoyaSiparisRezervSertifika.Metre
             from BoyaSiparisRezervSertifika
   left join BoyaSiparisRezerv on BoyaSiparisRezerv.BoyaSiparisRezervID = BoyaSiparisRezervSertifika.BoyaSiparisRezervID
   left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisRezervSertifika.HamStokDetayId
   left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
   left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = HamStokMaster.OrmePlanlamaId
   left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = HamStokMaster.DokumaPlanlamaId
   left join BoyaSiparis BS on BS.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
   left join HamstokIrsaliye on hamstokIrsaliye.HamStokIrsaliyeid = HamstokDetay.HamStokIrsaliyeid
   left join Ormeci on Ormeci.OrmeciId = HamstokIrsaliye.OrmeciId
   Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisRezervSertifika.BoyaSiparisKayitId
   left join Firma IrsaliyeFirma on IrsaliyeFirma.FirmaId = HamStokIrsaliye.FirmaId
   left join Firma UreticiFirma on UreticiFirma.FirmaId = HamStokMaster.UreticiFirmaId
   Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisRezervSertifika.InsertKullaniciId
   Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisRezervSertifika.KullaniciId
   left join PartiStok on PartiStok.PartiSinifId = HamStokMaster.PartiSinifId
   left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitID
   left join Olcu on Olcu.OlcuId = HamStokMaster.OlcuId
   where BoyaSiparisRezervSertifika.BoyaSiparisKayitId = :ABoyaSiparisKayitId
   order by BoyaSiparisRezervSertifika.BoyaSiparisDetayId, BoyaSiparisRezervSertifika.HamStokDetayId, BoyaSiparisRezervSertifika.BoyaSiparisRezervSertifikaId
/*DİKKAT ÖNEMLİ  Order by sakın degistirme else procedure yanlis calisir */
   into :BoyaSiparisRezervId, :BoyaSiparisRezervSertifikaId, :BoyaSiparisKayitId, :BoyaSiparisDetayId,  :HamStokDetayId, :HamStokMasterId, :HamStokIrsaliyeId,
        :Lot, :Pus, :Feine, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi, :OrmeciAdi, :OrmeciMakinaNo, :MusteriPartiNo,
        :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
        :KullanilanKilo, :KullanilanMetre, :KullanilanAdet, :KullanilanKapAdedi, :BoyaSiparisYil, :BoyaSiparisId, :TedarikciFirmaAdi, :UreticiFirmaAdi, :EskiPartiYil, :EskiPartiNo, :TupAcikEnMayli,
        :OrmeUretimPartiNo, :OrmeSiparisId, :OrmeSiparisYil, :YariMamulPartiYil, :YariMamulPartiId, :OlcuId, :OlcuAdi, :SertifikaNo, :Miktar, :Kilo, :Metre do
   begin
     if (EskiPartiNo is not null) then EskiPartiNoStr = '(' || EgemenIntegerToStr(EskiPartiYil)  || ') ' || EgemenIntegerToStr(EskiPartiNo) || ' ';
     else EskiPartiNoStr = '';

     YariMamulPartiNoStr = '';
     if (YariMamulPartiYil is not null) then
       YariMamulPartiNoStr = '('||YariMamulPartiYil||') '||YariMamulPartiId;

     OrmeSiparisIdStr = '';
     if (OrmeSiparisId is not null) then OrmeSiparisIdStr = '(' || EgemenIntegerToStr(OrmeSiparisYil) || ') '  || EgemenIntegerToStr(OrmeSiparisId);

     if (BoyaSiparisDetayId <> LastBoyaSiparisDetayId or HamStokDetayId <> LastHamStokDetayId) then
     begin
       FKullanilanKilo = KullanilanKilo;
       FKullanilanMetre = KullanilanMetre;
       FKullanilanAdet = KullanilanAdet;
       FKullanilanKapAdedi = KullanilanKapAdedi;
       LastBoyaSiparisDetayId = BoyaSiparisDetayId;

       LastHamStokDetayId = HamStokDetayId;
     end
     PartilenenKilo = 0;
     PartilenenMetre = 0;
     PartilenenAdet = 0;
     PartilenenKapAdedi = 0;

     if (FKullanilanKilo > Kilo) then
     begin
       PartilenenKilo = Kilo;
       FKullanilanKilo = FKullanilanKilo - Kilo;
     end
     else
     begin
       PartilenenKilo = FKullanilanKilo;
       FKullanilanKilo = 0;
     end
     if (FKullanilanMetre > Metre) then
     begin
       PartilenenMetre = Metre;
       FKullanilanMetre = FKullanilanMetre - Metre;
     end
     else
     begin
       PartilenenMetre = FKullanilanMetre;
       FKullanilanMetre = 0;
     end
     if (FKullanilanAdet > Adet) then
     begin
       PartilenenAdet = Adet;
       FKullanilanAdet = FKullanilanAdet - Adet;
     end
     else
     begin
       PartilenenAdet = FKullanilanAdet;
       FKullanilanAdet = 0;
     end
     if (FKullanilanKapAdedi > KapAdedi) then
     begin
       PartilenenKapAdedi = KapAdedi;
       FKullanilanKapAdedi = FKullanilanKapAdedi - KapAdedi;
     end
     else
     begin
       PartilenenKapAdedi = FKullanilanKapAdedi;
       FKullanilanKapAdedi = 0;
     end

     KalanKilo = Kilo - PartilenenKilo;
     KalanMetre = Metre - PartilenenMetre;
     KalanAdet = Adet - PartilenenAdet;
     KalanKapAdedi = KapAdedi - PartilenenKapAdedi;

     Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

     if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
     else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
     else EvrakTipiStr = 'Hatalı';

     RafIDStr = '';

     For select RafDetay.RafId from RafDetay
      where RafDetay.HamStokDetayId = :HamStokDetayId and (RafDetay.Miktar + AktarmadanIadeEdienKilo - RafDetay.IadeEdilecekMiktar - RafDetay.IadeEdilenMiktar - RafDetay.AktarmayaCikanKilo - RafDetay.PartilenenMiktar - RafDetay.RafTransferMiktar - RafDetay.SatisMiktar) > 0
      group by RafDetay.RafId into :FRafID do
     begin
       if (Char_Length(FRafId) + Char_Length(RafIDStr) < 48) then
       begin
         if (RafIDStr <> '' ) then RafIDStr = RafIDStr || cast(', ' as varchar(2) character set UTF8);
         RafIDStr = RafIDStr || FRafId;
       end

     end


     PartiIdStr = '';

     for select Parti.PartiYil, Parti.PartiId, PartiStok.PartiSiraNo from PartiStokDetay
       left join PartiStok on PartiStok.PartiSinifID = PartiStokDetay.PartiSinifID
       left join Parti on Parti.PartiKayitId = PartiStokDetay.PartiKayitId
       where PartiStokDetay.BoyaSiparisRezervId = :BoyaSiparisRezervId
       group by Parti.PartiYil, Parti.PartiId, PartiStok.PartiSiraNo
       into :FPartiYil, :FPartiId, :FPartiSiraNo do
     begin
       FPartiIdStr = '(' || EgemenIntegerToStr(FPartiYil)  || ') ' || EgemenIntegerToStr(FPartiId) || '.' || EgemenIntegerToStr(FPartiSiraNo) || ' ';

       if (Char_Length(PartiIdStr) + Char_Length(FPartiIdStr) < 250) then
       begin
         if (PartiIdStr <> '' ) then PartiIdStr = PartiIdStr || cast(', ' as varchar(2) character set UTF8);
         PartiIdStr = PartiIdStr || FPartiIdStr;
       end
     end

     BoyaSiparisIdStr = '';
     if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

     suspend;
   end
end
