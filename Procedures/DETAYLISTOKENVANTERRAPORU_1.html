           ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicGrup2Id and (UrunGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
           ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicGrup3Id and (UrunGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
           ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicGrup4Id and (UrunGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = '')) and
           ((:AUrunKategoriAralikli = 1 and Urun.UrunKategoriId >= :ABaslangicUrunKategoriId and (Urun.UrunKategoriId <= :ABitisUrunKategoriId or :ABitisUrunKategoriId = '')) or (:AUrunKategoriAralikli = 0) or (:AUrunKategoriAralikli = 1 and :ABaslangicUrunKategoriId = '' and :ABitisUrunKategoriId = '')) and
           ((:ARafAralikli = 1 and Urun.Raf >= :ABaslangicRaf and (Urun.Raf <= :ABitisRaf or :ABitisRaf = '')) or (:ARafAralikli = 0) or (:ARafAralikli = 1 and :ABaslangicRaf = '' and :ABitisRaf = '')) and
           (Urun.Kapali = :AKapaliDurumu or :AKapaliDurumu is null) and
           (:AUrunParcaDurumu is null or (Iif(:AUrunParcaDurumu = 0, Iif(Urun.ModelParcaId Is Null, 1, 0), Iif(:AUrunParcaDurumu = 1, Iif(Urun.ModelParcaId Is Not Null, 1, 0), 1)) = 1)) and
           Urun.EnvantereDahil = 1 and
           ((:ABoyaUrunTipi = 1 or Urun.UrunTipiId <> 0) and
            (:AModelId is null or (:AModelId is not null and Model.ModelId = :AModelId)) and
            (:AKimyasalUrunTipi = 1 or Urun.UrunTipiId <> 1) and
            (:ADigerUrunTipi = 1 or Urun.UrunTipiId <> 2) and
            (:AYedekParcaUrunTipi = 1 or Urun.UrunTipiId <> 3) and
            (:ASarfMalzemesiUrunTipi = 1 or Urun.UrunTipiId <> 4) and
            (:AHizmetUrunTipi = 1 or Urun.UrunTipiId <> 5) and
            (:ASablonUrunTipi = 1 or Urun.UrunTipiId <> 6) and
            (:AAksesuarUrunTipi = 1 or Urun.UrunTipiId <> 7) and
            (:ATicariMalUrunTipi = 1 or Urun.UrunTipiId <> 8)
           )
     into :UrunId, :UrunAdi, :Raf, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi,
          :UrunKategoriId, :KdvOrani, :Vadesi, :PesinIskontosu, :VadeliIskontosu, :TransferKodu, :AlisFiyati, :ALisFiyatiDovizId, :AlisFiyatiBirimId,
          :SatisFiyati, :SatisFiyatiDovizId, :SatisFiyatiBirimId, :UretimeCikisFiyati, :UretimeCikisFiyatiDovizId, :UretimeCikisFiyatiBirimId,
          :BirimId, :DefaultBirimCarpan, :UrunBirimId, :KimyasalTipiId, :KimyasalTipiAdi, :UstSeviye, :AltSeviye, :TeminSuresi, :MutfakDepoNo,
          :UrunTipiId, :UrunTipiAdi, :StokUreticiId, :StokUreticiAdi, :UrunAdi2, :FUrunNotu, :UrunMarkaID, :UrunMarkaAdi, :UreticiUrunKodu, :ParcaNo,
          :KatalogNo, :KatalogIndexNo, :KatalogDosyaAdi, :ResimDosyasi, :MakinaID, :MakinaAdi, :OrmeMakinaID, :OrmeMakinaAdi, :IplikMakinaID, :IplikMakinaAdi,
          :AcentaDosyaNo, :MalzemeOzelligi, :EtiketAciklama, :UniteId, :UniteAdi, :ModelId, :ModelAdi, :CasNo, :EkolojiNo, :Mrsl, :TheList,
          :Zdhc, :Gots, :Grs, :Msds, :Tds, :Koi, :Ctw, :TehlikeTanitimi,:Konsantrasyon, :AmbalajSekliAdi, :EcoPassport, :CMR, :Bluesign,
          :RenkAdi, :BedenAdi, :UrunKategoriAdi, :BoyaGrupId, :BoyaGrupAdi do
  begin
    select Notu from  BlobtanStringVer(:FUrunNotu) into :UrunNotu;


    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, UrunGrup4) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunKategoriAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunKategoriId, UrunKategoriId) = 0) then okey = 0;

    if (okey = 1) then
    if (ARafAralikli = 0) then
    if (KutuKarsilastir(ABaslangicRaf, Raf) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      DevirMiktari        = 0;
      AlisMiktari         = 0;
      AlisMaliyeti        = 0;
      DengelemeMiktari    = 0;
      DengelemeMaliyeti   = 0;
      UretimMiktari       = 0;
      UretimMaliyeti      = 0;
      AlisIadesiMiktari   = 0;
      AlisIadesiMaliyeti  = 0;
      SatisMiktari        = 0;
      SatisMaliyeti       = 0;
      SatisIadesiMiktari  = 0;
      SatisIadesiMaliyeti = 0;
      UretimeCikisMiktari = 0;
      UretimeCikisMaliyeti= 0;
      PartiliUCMiktari    = 0;
      PartiliUCMaliyeti   = 0;
      PartisizUCMiktari   = 0;
      PartisizUCMaliyeti  = 0;
      StokMiktari         = 0;
      OrtalamaMaliyet     = 0;
      FasonUretimeCikisMiktari  = 0;
      FasonUretimeCikisMaliyeti = 0;
      FasonUretimMiktari        = 0;
      FasonUretimMaliyeti       = 0;
      FasonUretimIadeMiktari    = 0;
      FasonUretimIadeMaliyeti   = 0;
      DepoTransferGelen         = 0;
      DepoTransferGelenMaliyeti = 0;
      DepoTransferGiden         = 0;
      DepoTransferGidenMaliyeti = 0;

      For Select StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId, sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from stokdetay
        left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
        where StokDetay.UrunId = :UrunId and stokdetay.StokDetayID is not null and StokDetay.Tarih < :ABaslangicTarihi and
              (StokDetay.Sinif = :ASinif or :ASinif is null) and
         ((:ADepoId is null and StokDetay.HareketTipi <> 5) or (:ADepoId is not null and stokdetay.DepoId = :ADepoId or StokDetay.TransferDepoId = :ADepoId))
        group by StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId
      into :HareketTipi, :FDepoId, :FTransferDepoId, :Miktar do
      begin
        if (HareketTipi = 5 and ADepoId = FTransferDepoId) then DevirMiktari = DevirMiktari - Miktar;
        else DevirMiktari = DevirMiktari + Miktar;
      end

      StokMiktari = DevirMiktari;

      For select StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId, sum(StokDetay.Miktar/ UrunBirim.Carpan),
           Sum(
            iif(StokDetay.FaturaId is not null, (StokDetay.Miktar * (StokDetay.NetBirimFiyat + StokDetay.NetBirimFiyat * Fatura.AltIskontoCarpani ) / Fatura.TabanDovizKuru1 * Fatura.TabanDovizKuru2),
              iif(StokDetay.IrsaliyeId is not null, (StokDetay.Miktar * StokDetay.NetBirimFiyat / Irsaliye.TabanDovizKuru1 * Irsaliye.TabanDovizKuru2),
                (StokDetay.Miktar * StokDetay.NetBirimFiyat  / Stok.TabanDovizKuru1 * Stok.TabanDovizKuru2)
              )
            )
           ),
                 Stok.DengelemeEvraki, iif(StokDetay.StokId is not null, StokDetay.PartiKayitId, Null), StokDetay.Iade from stokdetay
          left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
          left join fatura On Fatura.FaturaId = StokDetay.faturaId
          left join Irsaliye On Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
          left join Stok On Stok.StokId = StokDetay.StokId
          where stokdetay.Urunid = :UrunId and stokdetay.StokDetayID is not null and StokDetay.Tarih >= :ABaslangicTarihi and StokDetay.Tarih <= :ABitisTarihi and
              (StokDetay.Sinif = :ASinif or :ASinif is null) and
         (:ADepoId is null or (:ADepoId is not null and ( stokdetay.DepoId = :ADepoId or  StokDetay.TransferDepoId = :ADepoId))) and
         (:ADepoId is not null  or (:ADepoId is null and StokDetay.HareketTipi <> 5))
        /*and (StokDetay.faturaid is not null) or (StokDetay.FaturaId is null and StokDetay.IrsaliyeId is not null) or (StokDetay.FaturaId is null and StokDetay.IrsaliyeId is null)*/
          group by StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId, Stok.DengelemeEvraki, iif(StokDetay.StokId is not null, StokDetay.PartiKayitId, Null), StokDetay.Iade
          into :HareketTipi, :FDepoId, :FTransferDepoId, :Miktar, :Maliyet, :FDengeleme, :PartiKayitId, :Iade do
      begin
        if (HareketTipi = shtAlis or (HareketTipi = shtDepoTransfer and ADepoId = FTransferDepoId)) then
        begin
          if (FDengeleme is Null or FDengeleme = 0) then
          begin
            if (HareketTipi = shtDepoTransfer and ADepoId = FTransferDepoId) then
            begin
              DepoTransferGelen = DepoTransferGelen + Miktar;
              DepoTransferGelenMaliyeti = DepoTransferGelenMaliyeti + Maliyet;
              StokMiktari  = StokMiktari + Miktar;
            end
            else
            begin
              AlisMiktari  = AlisMiktari + Miktar;
              AlisMaliyeti = AlisMaliyeti + Maliyet;
              StokMiktari  = StokMiktari + Miktar;
            end
          end
          else
          begin
            DengelemeMiktari = DengelemeMiktari + Miktar;
            DengelemeMaliyeti = DengelemeMaliyeti + Maliyet;
            StokMiktari = StokMiktari + Miktar;
          end
        end
        else
        if (HareketTipi = shtSatis or (HareketTipi = shtDepoTransfer and ADepoId = FDepoId)) then
        begin
          if (HareketTipi = shtDepoTransfer and ADepoId = FDepoId) then
          begin
            DepoTransferGiden = DepoTransferGiden + Miktar;
            DepoTransferGidenMaliyeti = DepoTransferGidenMaliyeti + Maliyet;
            StokMiktari  = StokMiktari - Miktar;
          end
          else
          begin
            SatisMiktari = SatisMiktari + Miktar;
            SatisMaliyeti = SatisMaliyeti + Maliyet;
            StokMiktari = StokMiktari - Miktar;
          end
        end
        else
        if (HareketTipi = shtAlisIade) then
        begin
          AlisIadesiMiktari = AlisIadesiMiktari + Miktar;
          AlisIadesiMaliyeti = AlisIadesiMaliyeti + Maliyet;
          StokMiktari = StokMiktari - Miktar;
        end
        else
        if (HareketTipi = shtSatisIade) then
        begin
          SatisIadesiMiktari = SatisIadesiMiktari + Miktar;
          SatisIadesiMaliyeti = SatisIadesiMaliyeti + Maliyet;
          StokMiktari = StokMiktari + Miktar;
        end
        else
        if (HareketTipi = shtUretimeCikis) then
        begin
          UretimeCikisMiktari = UretimeCikisMiktari + Miktar;
          UretimeCikisMaliyeti = UretimeCikisMaliyeti + Maliyet;
          StokMiktari = StokMiktari - Miktar;
          if (PartiKayitId is not Null) then
          begin
            PartiliUCMiktari = PartiliUCMiktari + Miktar;
            PartiliUCMaliyeti = PartiliUCMaliyeti + Maliyet;
          end
          else
          begin
            PartisizUCMiktari = PartisizUCMiktari + Miktar;
            PartisizUCMaliyeti = PartisizUCMaliyeti + Maliyet;
          end
        end
        else
        if (HareketTipi = shtUretim) then
        begin
          UretimMiktari = UretimMiktari + Miktar;
          UretimMaliyeti = UretimMaliyeti + Maliyet;
          StokMiktari = StokMiktari + Miktar;
        end
        else
        if (HareketTipi = shtFasonUretimeCikis) then
        begin
          FasonUretimeCikisMiktari = FasonUretimeCikisMiktari + Miktar;
          FasonUretimeCikisMaliyeti = FasonUretimeCikisMaliyeti + Maliyet;
          StokMiktari = StokMiktari - Miktar;
        end
        else
        if (HareketTipi = shtFasonUretim and Iade = 0) then
        begin
          FasonUretimMiktari = FasonUretimMiktari + Miktar;
          FasonUretimMaliyeti = FasonUretimMaliyeti + Maliyet;
          StokMiktari = StokMiktari + Miktar;
        end
        else
        if (HareketTipi = shtFasonUretim and Iade = 1) then
        begin
          FasonUretimIadeMiktari = FasonUretimIadeMiktari + Miktar;
          FasonUretimIadeMaliyeti = FasonUretimIadeMaliyeti + Maliyet;
          StokMiktari = StokMiktari + Miktar;
        end
      end

      DevirMiktari         = DevirMiktari  * DefaultBirimCarpan;
      AlisMiktari          = AlisMiktari * DefaultBirimCarpan;
      DengelemeMiktari     = DengelemeMiktari * DefaultBirimCarpan;
      UretimMiktari        = UretimMiktari * DefaultBirimCarpan;
      AlisIadesiMiktari    = AlisIadesiMiktari * DefaultBirimCarpan;
      SatisMiktari         = SatisMiktari * DefaultBirimCarpan;
      SatisIadesiMiktari   = SatisIadesiMiktari * DefaultBirimCarpan;
      UretimeCikisMiktari  = UretimeCikisMiktari * DefaultBirimCarpan;
      PartiliUCMiktari     = PartiliUCMiktari * DefaultBirimCarpan;
      PartisizUCMiktari    = PartisizUCMiktari * DefaultBirimCarpan;
      FasonUretimeCikisMiktari  = FasonUretimeCikisMiktari * DefaultBirimCarpan;
      FasonUretimMiktari        = FasonUretimMiktari * DefaultBirimCarpan;
      FasonUretimIadeMiktari    = FasonUretimIadeMiktari * DefaultBirimCarpan;
      StokMiktari               = StokMiktari * DefaultBirimCarpan;
      DepoTransferGelen         = DepoTransferGelen * DefaultBirimCarpan;
      DepoTransferGiden         = DepoTransferGiden * DefaultBirimCarpan;

      if (AlisMiktari + UretimMiktari + SatisIadesiMiktari > 0) then
        OrtalamaMaliyet = (AlisMaliyeti + UretimMaliyeti + SatisIadesiMaliyeti + DepoTransferGelenMaliyeti) / (AlisMiktari + UretimMiktari + SatisIadesiMiktari + DepoTransferGelen);

      select SonAlisTarihi, SonAlisFiyati, SonAlisDovizId, SonAlisBirimId, SonAlisBirimCarpan, SonAlisLot, SonAlisMiktari from EnvanterIcinSonAlisBilgisiGetir(:UrunId, :Lot)
             into :SonAlisTarihi, :SonAlisFiyati, :SonAlisDovizId, :SonAlisBirimId, :SonAlisBirimCarpan, :SonAlisLot, :SonAlisMiktari;

      if (SonAlisDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:SonAlisDovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuruAlis;
      else DovizKuruAlis = 1;

      SonAlisFiyatiStokMaliyeti = StokMiktari * (SonAlisFiyati * SonAlisBirimCarpan / DefaultBirimCarpan) * DovizKuruAlis;
      OrtalamaMaliyetStokMaliyeti = StokMiktari * OrtalamaMaliyet;

      SonAlisFiyatindanDevirMaliyeti = DevirMiktari * (SonAlisFiyati * SonAlisBirimCarpan / DefaultBirimCarpan) * DovizKuruAlis;

      RaporSonAlisFiyati = (SonAlisFiyati * SonAlisBirimCarpan / DefaultBirimCarpan) * DovizKuruAlis;

      if (UretimeCikisFiyatiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:UretimeCikisFiyatiDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
      else FDovizKuru = 1;
      RaporUretimeCikisFiyati = UretimeCikisFiyati * FDovizKuru;

      if (AlisFiyatiDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:UretimeCikisFiyatiDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
      else FDovizKuru = 1;
      RaporAlisFiyati = AlisFiyati * FDovizKuru;

      AlisMaliyeti                = AlisMaliyeti * DovizKuru;
      DengelemeMaliyeti           = DengelemeMaliyeti * DovizKuru;
      AlisIadesiMaliyeti          = AlisIadesiMaliyeti * DovizKuru;
      SatisMaliyeti               = SatisMaliyeti * DovizKuru;
      SatisIadesiMaliyeti         = SatisIadesiMaliyeti * DovizKuru;
      UretimeCikisMaliyeti        = UretimeCikisMaliyeti * DovizKuru;
      PartiliUCMaliyeti           = PartiliUCMaliyeti * DovizKuru;
      PartisizUCMaliyeti          = PartisizUCMaliyeti * DovizKuru;
      OrtalamaMaliyet             = OrtalamaMaliyet * DovizKuru;
      OrtalamaMaliyetStokMaliyeti = OrtalamaMaliyetStokMaliyeti * DovizKuru;
      FasonUretimeCikisMaliyeti   = FasonUretimeCikisMaliyeti * DovizKuru;
      FasonUretimMaliyeti         = FasonUretimMaliyeti * DovizKuru;
      FasonUretimIadeMaliyeti     = FasonUretimIadeMaliyeti * DovizKuru;
      DepoTransferGelenMaliyeti   = DepoTransferGelenMaliyeti * DovizKuru;
      DepoTransferGidenMaliyeti   = DepoTransferGidenMaliyeti * DovizKuru;

      Select Miktar from FormataGoreKiloYuvarla(:StokMiktari, 'STOKRAPOR', 'STOKMIKTARI') into :StokMiktari;

      SonAlisOrtalamaFiyati = NUll;
      Select Sum(KalanMaliyet) / sum(KalanMiktar) from UrunFifoMaliyetiGetir(:UrunId, :DefaultBirimCarpan, :ABitisTarihi, :FStokTabanDovizId) into :SonAlisOrtalamaFiyati;

      Select FifoOrtalamaMaliyet from UrunFifoMaliyetiBul(:UrunId, :DefaultBirimCarpan, :ABaslangicTarihi - (1.00 / 24.00 / 60.00 / 60.00), :FStokTabanDovizId, :DevirMiktari, :DefaultBirimCarpan) into :DevirFifoBirimMaliyeti;

      Select FifoOrtalamaMaliyet from UrunFifoMaliyetiBul(:UrunId, :DefaultBirimCarpan, :ABitisTarihi, :FStokTabanDovizId, :StokMiktari, :DefaultBirimCarpan) into :StokFifoBirimMaliyeti;

      DevirFifoBirimMaliyeti = DevirFifoBirimMaliyeti * DovizKuru;
      StokFifoBirimMaliyeti  = StokFifoBirimMaliyeti * DovizKuru;

      if (ASifirOlanlarDokulsunmu = 1 or StokMiktari <> 0) then  suspend;
    end
  end
