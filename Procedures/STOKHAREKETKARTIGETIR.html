CREATE PROCEDURE STOKHAREKETKARTIGETIR(
    AURUNID VARCHAR(20),
    ADEPOID VARCHAR(10),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AURUNBIRIMID INTEGER)
RETURNS (
    FATURAID INTEGER,
    IRSALIYEID INTEGER,
    STOKID INTEGER,
    STOKDETAYID INTEGER,
    HAREKETTARIHI TIMESTAMP,
    HAREKETTIPI INTEGER,
    ACIKLAMA VARCHAR(100),
    DEPOID VARCHAR(10),
    TRANSFERDEPOID VARCHAR(10),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTIIDSTR VARCHAR(30),
    HBIRIMID VARCHAR(2),
    HMIKTAR DOUBLE PRECISION,
    KDV DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    STOKMIKTARI DOUBLE PRECISION,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    FISGRUPID VARCHAR(4),
    FISGRUPADI VARCHAR(30),
    IRSALIYEGRUPID VARCHAR(4),
    IRSALIYEGRUPADI VARCHAR(30),
    FATURAGRUPID VARCHAR(4),
    FATURAGRUPADI VARCHAR(30),
    MASRAFMERKEZIID INTEGER,
    MASRAFMERKEZIADI VARCHAR(30),
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    DEPARTMANID INTEGER,
    DEPARTMANADI VARCHAR(30),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    YATIRIM INTEGER,
    YATIRIMSTR VARCHAR(10),
    BIRIMFIYATTL DOUBLE PRECISION,
    DOVIZKURU DOUBLE PRECISION,
    DOVIZKURUMESAJ VARCHAR(255),
    DEFAULTBIRIM VARCHAR(2),
    DEFAULTBIRIMCARPAN DOUBLE PRECISION,
    DEFAULTBIRIMMIKTAR DOUBLE PRECISION,
    DEFAULTBIRIMFIYAT DOUBLE PRECISION,
    DEFAULTBIRIMFIYATTL DOUBLE PRECISION,
    URUNALISFIYATI DOUBLE PRECISION,
    URUNALISFIYATIDOVIZID VARCHAR(5),
    HGSIPARISKAYITID INTEGER,
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER,
    HGSIPARISIDSTR VARCHAR(30),
    HGSIPARISDETAYID INTEGER,
    HGISEMRIKAYITID INTEGER,
    HGISEMRIPASTALID INTEGER,
    HGISEMRIURUNID INTEGER,
    URETIMPARTINO VARCHAR(30),
    MODELID MODELKODUSIZE,
    MODELADI ADI40,
    KESIMTALIMATI INTEGER,
    URETIMISLEMLERIID VARCHAR(100),
    URETIMISLEMLERIADI VARCHAR(400),
    ISLEMKAYNAGI SMALLINT,
    IADE EGEBOOLEAN,
    LOT VARCHAR(150))
AS
declare variable FRaporBirimCarpan Double Precision;
declare variable FGirisCikis integer;
declare variable str varchar(30) character set dos857;
declare Variable FBirimCarpan Double Precision;
declare variable FAltIskontoCarpani Double Precision;
declare variable FPartiYil Integer;
declare variable FPartiId Integer;
declare variable SDovizId varchar(5);
declare variable IDovizId varchar(5);
declare variable FDovizId varchar(5);
declare variable FDepoId varchar(10);
declare variable FTransferDepoId varchar(10);
declare variable StokTabanDovizId varchar(5);
declare variable DovizKuruTarihi TimeStamp;

begin
/*
select * from stokhareketkartigetir('B010','BOYA',null,null,132)
*/
  StokMiktari = 0.0;
  if (ABitisTarihi is null) then ABitisTarihi = Current_TimeStamp(0);

  ABitisTarihi = f_stripTime(ABitisTarihi) || ' 23:59:59';

  select UrunBirim.Carpan from UrunBirim where UrunBirim.UrunId = :AUrunId and UrunBirim.UrunBirimId = :AUrunBirimId into :FRaporBirimCarpan;

  select UrunBirim.Carpan, UrunBirim.BirimId from UrunBirim where UrunBirim.UrunId = :AUrunId and UrunBirim.DefaultBirim = 1 into :DefaultBirimCarpan, :DefaultBirim;

  select StokTabanDovizId from Sistem into :StokTabanDovizId;

  if (FRaporBirimCarpan is null or FRaporBirimCarpan = 0) then FRaporBirimCarpan = 1.0;
  if (DefaultBirimCarpan is null or DefaultBirimCarpan = 0) then DefaultBirimCarpan = 1.0;
  StokMiktari = 0;
  if (ABaslangictarihi is Not Null) then
  begin
    For Select StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId, sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan)   from stokdetay
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    where stokdetay.Urunid = :AUrunId and StokDetay.Tarih < :ABaslangicTarihi and
     ((:ADepoId is null and StokDetay.HareketTipi <> 5) or (:ADepoId is not null and stokdetay.DepoId = :ADepoId or StokDetay.TransferDepoId = :ADepoId))
    group by StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId
    into :HareketTipi, :FDepoId, :FTransferDepoId, :Miktar do
    begin
      if (StokMiktari is null) then StokMiktari = 0.0;
      else StokMiktari = StokMiktari * FRaporBirimCarpan;

      if (HareketTipi = 1) then StokMiktari = StokMiktari + Miktar;
      else if (HareketTipi = 2) then StokMiktari = StokMiktari + Miktar;
      else if (HareketTipi = 3) then StokMiktari = StokMiktari + Miktar;
      else if (HareketTipi = 5 and ADepoId = FTransferDepoId) then StokMiktari = StokMiktari - Miktar;
      else if (HareketTipi = 5 and ADepoId = FDepoId) then StokMiktari = StokMiktari + Miktar;
      else if (HareketTipi = 4) then StokMiktari = StokMiktari + Miktar;
      else if (HareketTipi = 6) then StokMiktari = StokMiktari + Miktar;
      else if (HareketTipi = 7) then StokMiktari = StokMiktari + Miktar;
      else if (HareketTipi = 8) then StokMiktari = StokMiktari + Miktar;
      else if (HareketTipi = 9) then StokMiktari = StokMiktari + Miktar;
    end
  end

  if (ABaslangicTarihi is not Null) then
  begin
    For select StokDetay.FaturaId, StokDetay.IrsaliyeId, StokDetay.StokId, StokDetay.StokDetayId, StokDetay.HareketTipi, StokDetay.GirisCikis, StokDetay.Tarih, StokDetay.DepoId, StokDetay.TransferDepoId, Firma.FirmaId, Firma.Adi,
               StokDetay.Kdv, StokDetay.Miktar, UrunBirim.Carpan, UrunBirim.BirimId, NetBirimFiyat, AltIskontoCarpani, Fatura.FaturaNo,
               Fatura.FaturaTarihi, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi, Stok.FisNo, Stok.FisTarihi, Parti.PartiYil, Parti.PartiId, Stok.DovizId, Irsaliye.DovizId, fatura.DovizId,
               stokdetay.StokGrupId, SGrup.Adi, Irsaliye.IrsaliyeGrupId, IGrup.Adi, Fatura.FaturaGrupId, FGrup.Adi, StokDetay.Notu, Stokdetay.Yatirim,
               StokDetay.UniteID, Unite.Adi, coalesce(Stok.MasrafMerkeziId, StokDetay.MasrafMerkeziId), coalesce(MasrafMerkezi.Adi, DetayMasrafMerkezi.Adi), coalesce(Stok.DepartmanID, StokDetay.DepartmanID), coalesce(Departman.Adi, DetayDepartman.Adi),
               UrunFiyat.AlisFiyati, UrunFiyat.AlisFiyatiDovizId,
               StokDetay.HGIsEmriKayitID, StokDetay.HGIsEmriUrunID, StokDetay.HGIsEmriPastalID, Model.ModelID, Model.Adi, StokDetay.HGSiparisKayitId, HGSiparis.HGSiparisYil,
               HGSiparis.HGSiparisId, StokDetay.HGSiparisDetayId, HGIsEmri.UretimPartiNo, HGIsEmri.KesimTalimati, coalesce(Irsaliye.IslemKaynagi, Coalesce(Fatura.IslemKaynagi, Stok.IslemKaynagi)), StokDetay.Iade, StokDetay.Lot from Stokdetay
      left join Urun on Urun.UrunId = StokDetay.UrunId
      left join UrunFiyat ON UrunFiyat.UrunId = StokDetay.UrunId
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
      left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
      left join Stok on Stok.StokId = StokDetay.StokId
      left join Firma Firma on Firma.FirmaId = StokDetay.FirmaId
      left join Parti on Parti.PartiKayitId = StokDetay.PartiKayitId
      left join StokHareketGrup SGrup on SGrup.StokHareketGrupId = stokdetay.StokGrupId
      left join StokHareketGrup IGrup on IGrup.StokHareketGrupId = Irsaliye.IrsaliyeGrupId
      left join StokHareketGrup FGrup on FGrup.StokHareketGrupId = Fatura.FaturaGrupId
      left join MasrafMerkezi on MasrafMerkezi.MasrafMerkeziId = Stok.MasrafMerkeziId
      left Join Departman on Departman.DepartmanID = Stok.DepartmanID
      left join MasrafMerkezi DetayMasrafMerkezi on DetayMasrafMerkezi.MasrafMerkeziId = StokDetay.MasrafMerkeziId
      left Join Departman DetayDepartman on DetayDepartman.DepartmanID = StokDetay.DepartmanID
      left Join Unite on Unite.UniteID = StokDetay.UniteID
      Left Join HGSiparis on HGSiparis.HGSiparisKayitID = StokDetay.HGSiparisKayitID
      Left Join HGIsEmri on HGIsEmri.HGIsEmriKayitID = StokDetay.HGIsEmriKayitID
      left join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayId = StokDetay.HGSiparisDetayId
      left join Model on Model.ModelKayitID = Urun.ModelKayitID
      where stokdetay.Urunid = :AUrunId and StokDetay.Tarih >= :ABaslangicTarihi  and StokDetay.Tarih <= :ABitisTarihi and
           ((:ADepoId is null ) or
           (( stokdetay.DepoId = :ADepoId or  StokDetay.TransferDepoId = :ADepoId) and :ADepoId is not null ))
      order By StokDetay.UrunId, StokDetay.Tarih, StokDetay.StokDetayId
      into :FaturaId, :IrsaliyeId, :StokId, :StokDetayId, :HareketTipi, :FGirisCikis, :HareketTarihi, :DepoId, :TransferDepoId,
           :FirmaId,  :FirmaAdi, :Kdv, :HMiktar, :FBirimCarpan, :HBirimId, :BirimFiyat, :FAltIskontoCarpani, :FaturaNo,
           :FaturaTarihi, :IrsaliyeNo, :IrsaliyeTarihi, :FisNo, :FisTarihi, :FPartiYil, :FPartiId, :SDovizId, :IDovizId, :FDovizId,
           :FisGrupId, :FisGrupAdi, :IrsaliyeGrupId, :IrsaliyeGrupAdi, :FaturaGrupId, :FaturaGrupAdi, :Notu, :Yatirim,
           :UniteID, :UniteAdi, :MasrafMerkeziId, :MasrafMerkeziAdi, :DepartmanID, :DepartmanAdi, :UrunAlisFiyati, :UrunAlisFiyatiDovizId,
           :HGIsEmriKayitID, :HGIsEmriUrunID, :HGIsEmriPastalID, :ModelID, :ModelAdi, :HGSiparisKayitId, :HGSiparisYil, :HGSiparisId,
           :HGSiparisDetayId, :UretimPartiNo, :KesimTalimati, :IslemKaynagi, :Iade, :Lot do
    begin
      HGSiparisIdStr = '';
      if (HGSiparisId is not null) then HGSiparisIdStr = '(' || EgemenIntegerTostr(HGSiparisYil) || ') '  || EgemenIntegerTostr(HGSiparisId)  || ' ';
      UretimIslemleriID = '';
      UretimIslemleriAdi = '';
      Select UretimIslemleriID, UretimIslemleriAdi from HGIsEmriUretimIslemiGetir(:HGIsEmriKayitID, :KesimTalimati) into :UretimIslemleriId, :UretimIslemleriAdi;

      Aciklama = '';
      if (FAltIskontoCarpani is Null) then FAltIskontoCarpani = 0;

      BirimFiyat = BirimFiyat - BirimFiyat * FAltIskontoCarpani;

      Miktar = HMiktar;

      if (FBirimCarpan <> FRaporBirimCarpan) then
         Miktar = Miktar / FBirimCarpan * FRaporBirimCarpan;

      if (HareketTipi = 5) then
      begin
        if (ADepoId = DepoId) then FGirisCikis = -1;
        else if (ADepoId = TransferDepoId) then FGirisCikis = 1;
        else FGirisCikis = 0;
      end

      StokMiktari = StokMiktari + Miktar * FGirisCikis;

      if (FPartiYil is null) then PartiIdStr = '';
      else
      begin
        PartiIdStr = '(';
        PartiIdStr = PartiIdStr || Format_Float('#0', FPartiYil);
        PartiIdStr = PartiIdStr || ') ';
        PartiIdStr = PartiIdStr || Format_Float('#0', FPartiId);
        PartiIdStr = PartiIdStr || ' ';
      end

      if (FDovizId is not null) then DovizId = FDovizId;
      else if (IDovizId is not null) then DovizId = IDovizId;
      else if (SDovizId is not null) then DovizId = SDovizId;

      if (HareketTipi = 0) then  Aciklama = Format_Float('#0', FisNo) || '.Nolu Devir Fisi';
      else
      if (HareketTipi = 1) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Alis Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Alis Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Stok Giris Fisi';
      end
      else
      if (HareketTipi = 2) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Satis Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Satis Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Stok Cikis Fisi';
      end
      else
      if (HareketTipi = 3) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Alis Iade Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Alis Iade Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Stok Giris Iade Fisi';
      end
      else
      if (HareketTipi = 4) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Satis Iade Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Satis Iade Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Stok Cikis Iade Fisi';
      end
      else
      if (HareketTipi = 5) then
      begin
        if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Depo Transfer Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Depo Transfer Fisi';
        if (ADepoId = DepoId) then Aciklama = Aciklama || ' ile Cikis';
        if (ADepoId = TransferDepoId) then Aciklama = Aciklama || ' ile Giriş';
      end
      else
      if (HareketTipi = 6) then Aciklama = Format_Float('#0', FisNo) || '.Nolu Uretime Cikis Fisi';
      else
      if (HareketTipi = 7) then Aciklama = Format_Float('#0', FisNo) || '.Nolu Uretim Fisi';
      else
      if (HareketTipi = 8) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Fason Üretime Çıkış Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Fason Üretime Çıkış Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Fason Üretime Çıkış Fisi';
      end
      else
      if (HareketTipi = 9) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Fason Üretim Faturasi';
        else if (IrsaliyeNo is not null and Iade = 0) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Fason Üretim Irsaliyesi';
        else if (IrsaliyeNo is not null and Iade = 1) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Iade Fason Üretim Irs.';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Fason Üretime Fisi';
      end
      else Aciklama = 'Hareket Tipi Hatali';

      if (Yatirim = 1) then YatirimStr = 'Yatırım';
      else YatirimStr = '';

      if (DovizId <> StokTabanDovizId) then
      begin
        select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :StokTabanDovizId, :HareketTarihi) into :DovizKuru, :DovizKuruTarihi;
        if (HareketTarihi <> DovizKuruTarihi) then DovizKuruMesaj = 'Fatura Tarihine Ait Kur Girisi Bulunamadigi icin ' || TimeStampToDateTimeStr(DovizKuruTarihi) || ' Tarihli Kur ile Hesaplandi';
      end
      else DovizKuru = 1;

      BirimFiyatTL = BirimFiyat * DovizKuru;

      DefaultBirimMiktar = HMiktar;
      DefaultBirimFiyat = BirimFiyat;

      if (FBirimCarpan <> DefaultBirimCarpan) then
      begin
         DefaultBirimMiktar = DefaultBirimMiktar / FBirimCarpan * DefaultBirimCarpan;
         DefaultBirimFiyat = DefaultBirimFiyat * FBirimCarpan * DefaultBirimCarpan;
      end
      DefaultBirimFiyatTL = DefaultBirimFiyat * DovizKuru;
      suspend;
    end
  end
  else
  begin
    For select StokDetay.FaturaId, StokDetay.IrsaliyeId, StokDetay.StokId, StokDetay.StokDetayId, StokDetay.HareketTipi, StokDetay.GirisCikis, StokDetay.Tarih, StokDetay.DepoId, StokDetay.TransferDepoId, Firma.FirmaId, Firma.Adi,
               StokDetay.Kdv, StokDetay.Miktar, UrunBirim.Carpan, UrunBirim.BirimId, NetBirimFiyat, AltIskontoCarpani, Fatura.FaturaNo,
               Fatura.FaturaTarihi, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi, Stok.FisNo, Stok.FisTarihi, Parti.PartiYil, Parti.PartiId, Stok.DovizId, Irsaliye.DovizId, fatura.DovizId,
               stokdetay.StokGrupId, SGrup.Adi, Irsaliye.IrsaliyeGrupId, IGrup.Adi, Fatura.FaturaGrupId, FGrup.Adi, StokDetay.Notu, stokdetay.Yatirim,
               StokDetay.UniteID, Unite.Adi, coalesce(Stok.MasrafMerkeziId, StokDetay.MasrafMerkeziId), coalesce(MasrafMerkezi.Adi, DetayMasrafMerkezi.Adi), coalesce(Stok.DepartmanID, StokDetay.DepartmanID), coalesce(Departman.Adi, DetayDepartman.Adi),
               UrunFiyat.AlisFiyati, UrunFiyat.AlisFiyatiDovizId,
               StokDetay.HGIsEmriKayitID, StokDetay.HGIsEmriUrunID, StokDetay.HGIsEmriPastalID, Model.ModelID, Model.Adi, StokDetay.HGSiparisKayitId, HGSiparis.HGSiparisYil,
               HGSiparis.HGSiparisId, StokDetay.HGSiparisDetayId, HGIsEmri.UretimPartiNo, HGIsEmri.KesimTalimati, coalesce(Irsaliye.IslemKaynagi, Coalesce(Fatura.IslemKaynagi, Stok.IslemKaynagi)), StokDetay.Iade, StokDetay.Lot from stokdetay
      left join Urun on Urun.UrunId = StokDetay.UrunId
      left join UrunFiyat ON UrunFiyat.UrunId = StokDetay.UrunId
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
      left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
      left join Stok on Stok.StokId = StokDetay.StokId
      left join Firma Firma on Firma.FirmaId = StokDetay.FirmaId
      left join Parti on Parti.PartiKayitId = StokDetay.PartiKayitId
      left join StokHareketGrup SGrup on SGrup.StokHareketGrupId = stokdetay.StokGrupId
      left join StokHareketGrup IGrup on IGrup.StokHareketGrupId = Irsaliye.IrsaliyeGrupId
      left join StokHareketGrup FGrup on FGrup.StokHareketGrupId = Fatura.FaturaGrupId
      left join MasrafMerkezi on MasrafMerkezi.MasrafMerkeziId = Stok.MasrafMerkeziId
      left Join Departman on Departman.DepartmanID = Stok.DepartmanID
      left join MasrafMerkezi DetayMasrafMerkezi on DetayMasrafMerkezi.MasrafMerkeziId = StokDetay.MasrafMerkeziId
      left Join Departman DetayDepartman on DetayDepartman.DepartmanID = StokDetay.DepartmanID
      left Join Unite on Unite.UniteID = StokDetay.UniteID
      Left Join HGSiparis on HGSiparis.HGSiparisKayitID = StokDetay.HGSiparisKayitID
      Left Join HGIsEmri on HGIsEmri.HGIsEmriKayitID = StokDetay.HGIsEmriKayitID
      left join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayId = StokDetay.HGSiparisDetayId
      left join Model on Model.ModelKayitID = Urun.ModelKayitID
      where stokdetay.Urunid = :AUrunId and StokDetay.Tarih <= :ABitisTarihi and
           ((:ADepoId is null ) or
           (( stokdetay.DepoId = :ADepoId or  StokDetay.TransferDepoId = :ADepoId) and :ADepoId is not null ))
      order By StokDetay.UrunId, StokDetay.Tarih, StokDetay.StokDetayId
      into :FaturaId, :IrsaliyeId, :StokId, :StokDetayId, :HareketTipi, :FGirisCikis, :HareketTarihi, :DepoId, :TransferDepoId,
           :FirmaId,  :FirmaAdi, :Kdv, :HMiktar, :FBirimCarpan, :HBirimId, :BirimFiyat, :FAltIskontoCarpani, :FaturaNo,
           :FaturaTarihi, :IrsaliyeNo, :IrsaliyeTarihi, :FisNo, :FisTarihi, :FPartiYil, :FPartiId, :SDovizId, :IDovizId, :FDovizId,
           :FisGrupId, :FisGrupAdi, :IrsaliyeGrupId, :IrsaliyeGrupAdi, :FaturaGrupId, :FaturaGrupAdi, :Notu, :Yatirim,
           :UniteID, :UniteAdi, :MasrafMerkeziId, :MasrafMerkeziAdi, :DepartmanID, :DepartmanAdi, :UrunAlisFiyati, :UrunAlisFiyatiDovizId,
           :HGIsEmriKayitID, :HGIsEmriUrunID, :HGIsEmriPastalID, :ModelID, :ModelAdi, :HGSiparisKayitId, :HGSiparisYil, :HGSiparisId,
           :HGSiparisDetayId, :UretimPartiNo, :KesimTalimati, :IslemKaynagi, :Iade, :Lot do
    begin
      HGSiparisIdStr = '';
      if (HGSiparisId is not null) then HGSiparisIdStr = '(' || EgemenIntegerTostr(HGSiparisYil) || ') '  || EgemenIntegerTostr(HGSiparisId)  || ' ';
      UretimIslemleriID = '';
      UretimIslemleriAdi = '';
      Select UretimIslemleriID, UretimIslemleriAdi from HGIsEmriUretimIslemiGetir(:HGIsEmriKayitID, :KesimTalimati) into :UretimIslemleriId, :UretimIslemleriAdi;

      Aciklama = '';
      if (FAltIskontoCarpani is Null) then FAltIskontoCarpani = 0;

      BirimFiyat = BirimFiyat - BirimFiyat * FAltIskontoCarpani;

      Miktar = HMiktar;

      if (FBirimCarpan <> FRaporBirimCarpan) then
         Miktar = Miktar / FBirimCarpan * FRaporBirimCarpan;

      if (HareketTipi = 5) then
      begin
        if (ADepoId = DepoId) then FGirisCikis = -1;
        else if (ADepoId = TransferDepoId) then FGirisCikis = 1;
        else FGirisCikis = 0;
      end

      StokMiktari = StokMiktari + Miktar * FGirisCikis;

      if (FPartiYil is null) then PartiIdStr = '';
      else
      begin
        PartiIdStr = '(';
        PartiIdStr = PartiIdStr || Format_Float('#0', FPartiYil);
        PartiIdStr = PartiIdStr || ') ';
        PartiIdStr = PartiIdStr || Format_Float('#0', FPartiId);
        PartiIdStr = PartiIdStr || ' ';
      end

      if (FDovizId is not null) then DovizId = FDovizId;
      else if (IDovizId is not null) then DovizId = IDovizId;
      else if (SDovizId is not null) then DovizId = SDovizId;

      if (HareketTipi = 0) then  Aciklama = Format_Float('#0', FisNo) || '.Nolu Devir Fisi';
      else
      if (HareketTipi = 1) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Alis Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Alis Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Stok Giris Fisi';
      end
      else
      if (HareketTipi = 2) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Satis Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Satis Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Stok Cikis Fisi';
      end
      else
      if (HareketTipi = 3) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Alis Iade Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Alis Iade Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Stok Giris Iade Fisi';
      end
      else
      if (HareketTipi = 4) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Satis Iade Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Satis Iade Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Stok Cikis Iade Fisi';
      end
      else
      if (HareketTipi = 5) then
      begin
        if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Depo Transfer Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Depo Transfer Fisi';
        if (ADepoId = DepoId) then Aciklama = Aciklama || ' ile Cikis';
        if (ADepoId = TransferDepoId) then Aciklama = Aciklama || ' ile Giris';
      end
      else
      if (HareketTipi = 6) then Aciklama = Format_Float('#0', FisNo) || '.Nolu Uretime Cikis Fisi';
      else
      if (HareketTipi = 7) then Aciklama = Format_Float('#0', FisNo) || '.Nolu Uretim Fisi';
      else
      if (HareketTipi = 8) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Fason Üretime Çıkış Faturasi';
        else if (IrsaliyeNo is not null) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Fason Üretime Çıkış Irsaliyesi';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Fason Üretime Çıkış Fisi';
      end
      else
      if (HareketTipi = 9) then
      begin
        if (FaturaNo is not null) then Aciklama = Format_Float('#0', FaturaNo) || '.Nolu Fason Üretim Faturasi';
        else if (IrsaliyeNo is not null and Iade = 0) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Fason Üretim Irsaliyesi';
        else if (IrsaliyeNo is not null and Iade = 1) then Aciklama = Format_Float('#0', IrsaliyeNo) || '.Nolu Iade Fason Üretim Irs.';
        else Aciklama = Format_Float('#0', FisNo) || '.Nolu Fason Üretim Fisi';
      end
      else Aciklama = 'Hareket Tipi Hatali';

      if (Yatirim = 1) then YatirimStr = 'Yatırım';
      else YatirimStr = '';

      if (DovizId <> StokTabanDovizId) then
      begin
        select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :StokTabanDovizId, :HareketTarihi) into :DovizKuru, :DovizKuruTarihi;
        if (HareketTarihi <> DovizKuruTarihi) then DovizKuruMesaj = 'Fatura Tarihine Ait Kur Girisi Bulunamadigi icin ' || TimeStampToDateTimeStr(DovizKuruTarihi) || ' Tarihli Kur ile Hesaplandi';
      end
      else DovizKuru = 1;

      BirimFiyatTL = BirimFiyat * DovizKuru;

      DefaultBirimMiktar = HMiktar;
      DefaultBirimFiyat = BirimFiyat;

      if (FBirimCarpan <> DefaultBirimCarpan) then
      begin
         DefaultBirimMiktar = DefaultBirimMiktar / FBirimCarpan * DefaultBirimCarpan;
         DefaultBirimFiyat = DefaultBirimFiyat * FBirimCarpan * DefaultBirimCarpan;
      end
      DefaultBirimFiyatTL = DefaultBirimFiyat * DovizKuru;

      suspend;
    end
  end
end
