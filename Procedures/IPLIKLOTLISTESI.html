CREATE PROCEDURE IPLIKLOTLISTESI(
    ABASLANGICISEMRITARIHI TIMESTAMP,
    ABITISISEMRITARIHI TIMESTAMP,
    AIPLIKISLETMEID INTEGER,
    AKAPALI INTEGER,
    AHAMID INTEGER,
    ABASKAPATMATARIHI TIMESTAMP,
    ABITKAPATMATARIHI TIMESTAMP)
RETURNS (
    IPLIKLOTID INTEGER,
    TARIHI TIMESTAMP,
    LOT VARCHAR(30),
    ESKILOT VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    HARMANKAYITID INTEGER,
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    HARMANADI VARCHAR(50),
    HARMANCINSIID INTEGER,
    HARMANCINSIADI VARCHAR(50),
    IPLIKISLETMEID INTEGER,
    IPLIKISLETMEADI VARCHAR(30),
    BOBINGRAMAJI NUMERIC(12,3),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    KONIKRENGIID INTEGER,
    KONIKRENGIADI VARCHAR(20),
    ETIKETTIPIID INTEGER,
    ETIKETTIPIADI VARCHAR(30),
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    IPLIKTALEPYIL INTEGER,
    IPLIKTALEPID INTEGER,
    IPLIKTALEPIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    URETIMMIKTARI NUMERIC(12,3),
    SARIMURETIMMIKTARI NUMERIC(12,3),
    TAMIR INTEGER,
    TAMIRSTR VARCHAR(6),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    TAMIRGIRISMIKTAR NUMERIC(12,3),
    KAPALI INTEGER,
    KAPALISTR VARCHAR(6),
    SARIMSEKLIID INTEGER,
    SARIMSEKLIIDSTR VARCHAR(20),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    KAPATANKULLANICIADI VARCHAR(40),
    KAPATMATARIHI TIMESTAMP,
    IPLIKTALEPDETAYID INTEGER,
    IPLIKTALEPMIKTAR DOUBLE PRECISION,
    URETIMTARIHI TIMESTAMP,
    IPLIKLOTGRUBUID INTEGER,
    IPLIKLOTGRUBUADI ADI20,
    HARMANRENKADI ADI50,
    HARMANRENKDIGERADI ADI50,
    HARMANRENKID INTEGER)
AS
begin
  For Select IplikLot.IplikLotId, IplikLot.Tarihi, IplikLot.Lot, IplikLot.EskiLot, IplikLot.hamId, Ham.Adi, IplikLot.MelanjKodu, IplikLot.FlamKodu,
             IplikLot.HarmanKayitId, Harman.HarmanYil, Harman.HarmanId, Harman.Adi, Harman.HarmanCinsiId,
             HarmanCinsi.Adi, Harman.IplikIsletmeId, IplikIsletme.Adi, IplikLot.BobinGramaji,
             IplikLot.KonikCinsiId, KonikCinsi.Adi, IplikLot.KonikRengiId, KonikRengi.Adi,
             IplikLot.AmbalajId, Ambalaj.Adi, IplikLot.EtiketTipiId, EtiketTipi.Adi,
             IplikLot.UretimMiktari, IplikLot.SarimUretimMiktari, IplikLot.Kapali, IplikLot.Tamir,
             IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, SiparisFirma.Adi,
             IplikSatisIade.RenkId, Renk.Adi, TamirGiris.Miktar, IplikLot.SarimSekliId,
             Ik.Adi, IplikLot.InsertTarihi, Sk.Adi, IplikLot.DegisimTarihi, KapatanKullanici.Adi, IplikLot.KapatmaTarihi, IplikTalepDetay.Miktar, IplikLot.IplikTalepDetayId,
             IplikLot.UretimTarihi, IplikLotGrubu.IplikLotGrubuId, IplikLotGrubu.Adi, HarmanRenk.Adi , HarmanRenk.DigerAdi, HarmanRenk.RenkId from IplikLot
    left join Ham on Ham.HamId = IplikLot.HamId
    left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitId
    left join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Harman.HarmanCinsiId
    left join IplikIsletme on IplikIsletme.IplikIsletmeId = Harman.IplikIsletmeId
    left join Ambalaj on Ambalaj.AmbalajId = IplikLot.AmbalajId
    left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikLot.KonikCinsiId
    left join KonikRengi on KonikRengi.KonikRengiId = IplikLot.KonikRengiId
    left join EtiketTipi on EtiketTipi.EtiketTipiId = IplikLot.EtiketTipiId
    Left Join Kullanici IK on IK.KullaniciId = IplikLot.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = IplikLot.KullaniciId
    Left Join Kullanici KapatanKullanici on KapatanKullanici.KullaniciId = IplikLot.KapatanKullaniciId
    left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikLot.IplikTalepKayitId
    left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikLot.IplikTalepDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
    left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
    left join TamirGiris on TamirGiris.TamirGirisKayitID = IplikLot.TamirGirisKayitId
    left join IplikSatisIade on IplikSatisIade.IplikSatisIadeID = TamirGiris.IplikSatisIadeId
    left join Renk on Renk.RenkID = IplikSatisIade.RenkId
    left join IplikLotGrubu on IplikLotGrubu.IplikLotGrubuId = IplikLot.IplikLotGrubuId
    left join Renk HarmanRenk on HarmanRenk.RenkId = Harman.RenkId
  where (IplikLot.Tarihi >= :ABaslangicIsEmriTarihi and IplikLot.Tarihi <= :ABitisIsEmriTarihi) and
        (IplikLot.Kapali = :AKapali or :AKapali is null) and
        (IplikLot.HamID = :AHamID or :AHamID is null) and
        (Harman.IplikIsletmeId = :AIplikIsletmeId or :AIplikIsletmeId is null) and
        (
          (:AKapali = 1 and IplikLot.KapatmaTarihi >= :ABasKapatmaTarihi and IplikLot.KapatmaTarihi <= (:ABitKapatmaTarihi + 1)) or
          (:AKapali is Null and (IplikLot.KapatmaTarihi >= :ABasKapatmaTarihi and IplikLot.KapatmaTarihi <= (:ABitKapatmaTarihi + 1) or (IplikLot.Kapali = 0))) or
          (:AKapali = 0)
        )
  into :IplikLotId, :Tarihi, :Lot, :EskiLot, :hamId, :HamAdi, :MelanjKodu, :FlamKodu, :HarmanKayitId, :HarmanYil, :HarmanId, :HarmanAdi, :HarmanCinsiId,
       :HarmanCinsiAdi, :IplikIsletmeId, :IplikIsletmeAdi, :BobinGramaji, :KonikCinsiId, :KonikCinsiAdi, :KonikRengiId, :KonikRengiAdi, :AmbalajId, :AmbalajAdi,
       :EtiketTipiId, :EtiketTipiAdi, :UretimMiktari, :SarimUretimMiktari,  :Kapali, :Tamir, :IplikTalepYil, :IplikTalepId, :BoyaSiparisYil, :BoyaSiparisId,
       :SiparisFirmaId, :SiparisFirmaAdi, :RenkId, :RenkAdi, :TamirGirisMiktar, :SarimSekliId,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :KapatanKullaniciAdi, :KapatmaTarihi, :IplikTalepMiktar, :IplikTalepDetayId, :UretimTarihi,
       :IplikLotGrubuId, :IplikLotGrubuAdi, :HarmanRenkAdi, :HarmanRenkDigerAdi, :HarmanRenkId do
  begin
    HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') ' || EgemenIntegerToStr(HarmanId) || ' ';

    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';
    else
      BoyaSiparisIdStr = '';

    if (IplikTalepYil is not null) then
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';
    else
      IplikTalepIdStr = '';

    if (Kapali = 0) then KapaliStr = ''; else KapaliStr = 'Kapalı';
    if (Tamir = 0) then TamirStr = ''; else TamirStr = 'Tamir';

    if (SarimSekliId = 0) then SarimSekliIdStr = 'Sert Sarım';
    else SarimSekliIdStr ='Yumuşak Sarım';

    suspend;
  end
end
