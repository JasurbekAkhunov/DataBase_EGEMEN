CREATE PROCEDURE BSDPARTISTOKTARTISEVKGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    MIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    BRUTSEVKMIKTAR NUMERIC(12,3),
    NETSEVKMIKTAR NUMERIC(12,3),
    FIREORANI DOUBLE PRECISION)
AS
declare variable FBrutSevkMiktar   Numeric(12,3);
declare variable FNetSevkMiktar    Numeric(12,3);
begin
  For Select PartiStok.BoyaSiparisDetayID from PartiStok
    where Partistok.PartiKayitID = :APartiKayitId
    group by PartiStok.BoyaSiparisDetayID
    into :BoyaSiparisDetayID do
  begin
    FBrutSevkMiktar = 0;
    FNetSevkMiktar  = 0;

    For Select Parti.PartiYil, Parti.PartiID, PartiStok.Miktar, PartiStok.NetMiktar, PartiStok.BrutSevkMiktar, PartiStok.NetSevkMiktar from PartiStok
      left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
      where PartiStok.BoyaSiparisDetayID = :BoyaSiparisDetayID and PartiStok.PartiStokIptalKayitID is null and
            PartiStok.PartiKayitID <> :APartiKayitID
      order by PartiStok.SevkTarihi
      into :PartiYil, :PartiID, :Miktar, :NetMiktar, :BrutSevkMiktar, :NetSevkMiktar do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';


      if (BrutSevkMiktar <> 0) then
        FireOrani = (BrutSevkMiktar - NetSevkMiktar) * 100 / BrutSevkMiktar;
      else
        FireOrani = Null;

      FBrutSevkMiktar = FBrutSevkMiktar + BrutSevkMiktar;
      FNetSevkMiktar  = FNetSevkMiktar + NetSevkMiktar;
      suspend;

      if (FBrutSevkMiktar > 0) then
      begin
        PartiYil = Null;
        PartiID = Null;
        PartiIdStr = '    Ara Toplam';
        Miktar      = Null;
        NetMiktar   = Null;
        BrutSevkMiktar = FBrutSevkMiktar;
        NetSevkMiktar  = FNetSevkMiktar;
        FireOrani = (FBrutSevkMiktar - FNetSevkMiktar) * 100 / FBrutSevkMiktar;
        suspend;
      end
    end
  end
end
