CREATE PROCEDURE MAMULDEPOSAYIMRAPORU_2(
    ATARIH TIMESTAMP,
    ADEPOID DEPOKODUSIZE,
    ARAFID RAFKODUSIZE,
    ATIP INTEGER)
RETURNS (
    PARTISTOKTARTIID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    TARIH TIMESTAMP,
    KAPADEDI INTEGER,
    DARA BOYAMIKTARSIZE,
    KILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    ADET INTEGER,
    BOBINSAYISI INTEGER,
    MAMULDEPOTRANSFERFISIID INTEGER,
    KALITENO INTEGER,
    KALITEKONTROL INTEGER,
    DEPOID VARCHAR(10),
    RAFID RAFKODUSIZE,
    BRUTKILO BOYAMIKTARSIZE,
    TARTIBRUTKILO BOYAMIKTARSIZE,
    BRUTMETRE BOYAMIKTARSIZE,
    PALETNO VARCHAR(5),
    KALITEACIKLAMATIPIID INTEGER,
    MAMULPROSESID INTEGER,
    BOYAREZERVFISIDETAYID INTEGER,
    IADEBOYAIRSALIYEDETAYID INTEGER,
    IADEBOYAIRSALIYEID INTEGER,
    BARKODID VARCHAR(30),
    PARTIIDSTR VARCHAR(30),
    RENKADI ADI50,
    FIRMAID FIRMAKODUSIZE,
    FIRMAADI ADI100,
    ACIKLAMA VARCHAR(1000),
    SAYIMTAMAM EGEBOOLEAN,
    SAYIMRAFID RAFKODUSIZE)
AS
begin
  for Select PartiStokTarti.PartiStokTartiId, PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.Tarih, PartiStokTarti.KapAdedi, PartiStokTarti.Dara, PartiStokTarti.Kilo,
             PartiStokTarti.Metre, PartiStokTarti.Adet, PartiStokTarti.BobinSayisi, PartiStokTarti.MamulDepoTransferFisiId, PartiStokTarti.KaliteNo, PartiStokTarti.KaliteKontrol, PartiStokTarti.DepoId,
             PartiStokTarti.BrutKilo, PartiStokTarti.BrutMetre, PartiStokTarti.PaletNo, PartiStokTarti.KaliteAciklamaTipiId, PartiStokTarti.MamulProsesId, PartiStokTarti.BoyaRezervFisiDetayId,
             PartiStokTarti.IadeBoyaIrsaliyeDetayId, PartiStokTarti.IadeBoyaIrsaliyeId, Coalesce(PartiStokTarti.BarkodId, 'I-'||IplikPaketleme.IplikPaketlemeId), '('||Parti.PartiYil||') '||Parti.PartiId, Renk.Adi, Firma.FirmaId,
             Firma.Adi, PartiStokTarti.Dara + PartiStokTarti.Kilo, MamulDepoTransferFisiDetay.RafId, DepoSayimDetay.RafId
    from DepoSayim
    join DepoSayimDetay on DepoSayim.DepoSayimId = DepoSayimDetay.DepoSayimId
    join PartiStokTarti on PartiStokTarti.PartiStokTartiId = DepoSayimDetay.PartiStokTartiId
    left join MamulDepoTransferFisiDetay on PartiStokTarti.PartiStokTartiId = MamulDepoTransferFisiDetay.PartiStokTartiId and PartiStokTarti.MamulDepoTransferFisiId = MamulDepoTransferFisiDetay.MamulDepoTransferFisiId
    left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Renk on BoyaSiparisDetay.RenkId = Renk.RenkID
    left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    left join DepoRaf on DepoRaf.RafId = DepoSayimDetay.RafId
    where DepoSayim.FisTarihi = :ATarih and
         (DepoSayimDetay.RafID = :ARafId or :ARafId is null) and
         (DepoRaf.DepoId = :ADepoId or :ADepoId is null)
  into :PartiStokTartiId, :PartiKayitId, :PartiSinifId, :Tarih, :KapAdedi, :Dara, :Kilo,
             :Metre, :Adet, :BobinSayisi, :MamulDepoTransferFisiId, :KaliteNo, :KaliteKontrol, :DepoId,
             :BrutKilo, :BrutMetre, :PaletNo, :KaliteAciklamaTipiId, :MamulProsesId, :BoyaRezervFisiDetayId,
             :IadeBoyaIrsaliyeDetayId, :IadeBoyaIrsaliyeId, :BarkodId, :PartiIdStr, :RenkAdi, :FirmaId, :FirmaAdi, :TartiBrutKilo, :RafId, :SayimRafId do
  begin
    SayimTamam = 0;
    Aciklama = '';
    if (SayimRafId <> RafId) then
      Aciklama = 'Farklı rafta ('||RafId||')';
    else
      SayimTamam = 1;
    suspend;
  end
end
