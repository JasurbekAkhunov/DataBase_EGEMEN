CREATE PROCEDURE DOKUMASIPARISRAPORU_3(
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    AIPLIKSIPARISI INTEGER,
    ABOYALIIPLIKSIPARISI INTEGER,
    ADOKUMASIPARISI INTEGER,
    ABOYALIDOKUMASIPARISI INTEGER,
    AIPLIKBOYALIDOKUMASIPARISI INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AKUMASCINSIID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    AYURTDISI INTEGER,
    AKUMASTEMINSIPARISI INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ASINIF VARCHAR(1),
    AKALITEID VARCHAR(20))
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER,
    KILO NUMERIC(12,3),
    SIPARISBRUTKILO NUMERIC(12,3),
    VARYANTADI VARCHAR(30),
    BRUTSEVKKILO NUMERIC(12,3),
    NETSEVKKILO NUMERIC(12,3),
    BOYAHANEBRUTSEVKKILO NUMERIC(12,3),
    BOYAHANENETSEVKKILO NUMERIC(12,3),
    BOYAHANEDENGELENBRUTTOPLAM NUMERIC(12,3),
    BOYAHANEDENGELENNETTOPLAM NUMERIC(12,3),
    BOYAHANEDENGELENBRUTTOPLAMK1 NUMERIC(12,3),
    BOYAHANEDENGELENNETTOPLAMK1 NUMERIC(12,3),
    TOPLAMSATISIADE NUMERIC(12,3),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable FirmaId          varchar(20);
declare variable shtSatis integer;
declare variable shtSatisIade integer;
declare variable FBrutMiktar  Numeric(15,3);
declare variable FNetMiktar   Numeric(15,3);
declare variable FBrutKilo  Numeric(15,3);
declare variable FNetKilo   Numeric(15,3);
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable FKaliteNo integer;
declare variable FBrutToplam Numeric(12,3);
declare variable FNetToplam Numeric(12,3);
declare variable DokumaFirmaKodumuz varchar(20);
declare variable PartilenenKilo Numeric(12,3);
begin
  Select DokumaFirmaKodumuz from Sistem where SistemID = 1 into :DokumaFirmaKodumuz;

  Select shtFasonUretim, ShtUretim, ShtSatis, ShtSatisIade from OrmeStokHareketTipiGetir into :shtFasonUretim, :ShtUretim, :ShtSatis, :ShtSatisIade;

  For Select DokumaSiparisRaporu_2.BoyaSiparisKayitId, DokumaSiparisRaporu_2.OrmeSiparisDetayId, BoyaSiparisBaski.BoyaSiparisDetayId, BoyaSiparisBaski.BoyaSiparisBaskiId,
             BoyaSiparisBaski.Kilo, Desenvaryant.VaryantAdi, DokumaSiparisRaporu_2.FirmaId, BoyaSiparisBaski.PartilenenKilo,
             Ik.Adi, BoyaSiparisBaski.InsertTarihi, Sk.Adi, BoyaSiparisBaski.DegisimTarihi, DokumaSiparisRaporu_2.SiparisBrutKilo from DokumaSiparisRaporu_2(:ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ABaslangicSiparisTarihi, :ABitisSiparisTarihi, :AFirmaId,
              :AFASONSIPARISLER, :AFASONSATISSIPARISLER, :ASATISSIPARISLER, :AIplikSiparisi, :ABoyaliIplikSiparisi, :ADokumaSiparisi, :ABoyaliDokumaSiparisi,
              :AIplikBoyaliDokumaSiparisi, :AMusteriSiparisNo, :AReferansSiparisNo, :ARaporTipi, :AKapali, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisiAlanKullaniciId,
              :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId, :AYurtDisi, :AKumasTeminSiparisi, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASinif, :AKaliteId)
  left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisDetayId = DokumaSiparisRaporu_2.BoyaSiparisDetayId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisBaski.BoyaSiparisDetayId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
  left join Kullanici IK on IK.KullaniciId = BoyaSiparisBaski.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId = BoyaSiparisBaski.KullaniciId
  where BoyaSiparisBaski.BoyaSiparisBaskiid is not null
  into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :Kilo, :VaryantAdi,
       :FirmaId, :PartilenenKilo, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :SiparisBrutKilo do
  begin

    BrutSevkKilo = 0;
    NetSevkKilo = 0;
    BoyahaneBrutSevkKilo = 0;
    BoyahaneNetSevkKilo = 0;

    BoyahanedenGelenBrutToplam = 0;
    BoyahanedenGelenNetToplam = 0;
    BoyahanedenGelenBrutToplamK1 = 0;
    BoyahanedenGelenNetToplamK1 = 0;
    ToplamSatisIade = 0;

    if (PartilenenKilo > 0) then
    begin
      Select Sum(BoyaIrsaliyeDetay.BrutKilo), Sum(BoyaIrsaliyeDetay.Kilo) from BoyaSiparisBaski
      left join PartiStok on PartiStok.BoyaSiparisBaskiId = BoyaSiparisBaski.BoyaSiparisBaskiId
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      where BoyaSiparisBaski.BoyaSiparisBaskiId = :BoyaSiparisBaskiId and BoyaIrsaliye.FirmaID <> :DokumaFirmaKodumuz
      into :BoyahaneBrutSevkKilo, :BoyahaneNetSevkKilo;

      if (BoyahaneBrutSevkKilo is null) then BoyahaneBrutSevkKilo = 0;
      if (BoyahaneNetSevkKilo is null) then BoyahaneNetSevkKilo = 0;

    end

    Select Sum(StokHareket.BrutMiktar), Sum(StokHareket.Miktar) from BoyaSiparisBaski
    left join StokHareket on StokHareket.BoyaSiparisBaskiID = BoyaSiparisBaski.BoyaSiparisBaskiId
    where BoyaSiparisBaski.BoyaSiparisBaskiId = :BoyaSiparisBaskiId and StokHareket.HareketTipi = :shtSatis and StokHareket.HamIade = 0
    into :BrutSevkKilo, :NetSevkKilo;

    if (BrutSevkKilo is null) then BrutSevkKilo = 0;
    if (NetSevkKilo is null) then NetSevkKilo = 0;

    BrutSevkKilo = BoyahaneBrutSevkKilo + BrutSevkKilo;
    NetSevkKilo = BoyahaneNetSevkKilo + NetSevkKilo;

    Select Sum(StokHareket.Miktar) from BoyaSiparisBaski
    left join StokHareket on StokHareket.BoyaSiparisBaskiID = BoyaSiparisBaski.BoyaSiparisBaskiId
    where BoyaSiparisBaski.BoyaSiparisBaskiId = :BoyaSiparisBaskiId and StokHareket.HareketTipi = :shtSatisIade
    into :ToplamSatisIade;

    for Select StokHareket.KaliteNo, Sum(StokHareket.BrutMiktar), Sum(StokHareket.Miktar) from BoyaSiparisBaski
    left join StokHareket on StokHareket.BoyaSiparisBaskiID = BoyaSiparisBaski.BoyaSiparisBaskiId
    where BoyaSiparisBaski.BoyaSiparisBaskiId = :BoyaSiparisBaskiId and
          (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :ShtUretim) and StokHareket.HamIade = 0
    group by StokHareket.KaliteNo
    into :FKaliteNo, :FBrutToplam, :FNetToplam do
    begin
      BoyahanedenGelenBrutToplam = BoyahanedenGelenBrutToplam + FBrutToplam;
      BoyahanedenGelenNetToplam  = BoyahanedenGelenNetToplam + FNetToplam;

      if (FKaliteNo = 1) then
      begin
        BoyahanedenGelenBrutToplamK1 = BoyahanedenGelenBrutToplamK1  + FBrutToplam;
        BoyahanedenGelenNetToplamK1 = BoyahanedenGelenNetToplamK1 + FNetToplam;
      end
    end

    suspend;
  end
end
