CREATE PROCEDURE ORMEHAMKONTROLKTBOYASIPDLISTESI(
    AORMEPLANLAMAID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    HAMADI ADI255,
    RENKADI ADI50)
AS
begin
  for select BoyaSiparisDetay.BoyaSiparisDetayId, Ham.Adi, Renk.Adi
    from OrmePlanlama
    join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    join OrmeSiparisDetay OrmeSiparisDetay2 on OrmeSiparisDetay.BoyaSiparisKayitId = OrmeSiparisDetay2.KtBoyaSiparisKayitId
    join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay2.OrmeSiparisDetayId and BoyaSiparisDetay.HamId = OrmeSiparisDetay.HamId
    join Ham on Ham.HamId = BoyaSiparisDetay.HamID
    left join Renk on Renk.RenkId = BoyaSiparisDetay.RenkID
    where OrmePlanlama.OrmePlanlamaId = :AOrmePlanlamaId into :BoyaSiparisDetayId, :HamAdi, :RenkAdi do suspend;
end
