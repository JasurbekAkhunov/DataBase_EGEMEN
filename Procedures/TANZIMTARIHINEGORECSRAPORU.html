CREATE PROCEDURE TANZIMTARIHINEGORECSRAPORU(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAV VARCHAR(1),
    ACS VARCHAR(1),
    ATAHSIL INTEGER,
    APROTESTO INTEGER,
    AAVUKAT INTEGER,
    AIADEEDILENLER INTEGER,
    ABANKACIKIS INTEGER,
    ACIRO INTEGER,
    ABANKAHESAPID VARCHAR(20),
    ABASLANGICCSGRUP1ID VARCHAR(4),
    ABITISCSGRUP1ID VARCHAR(4),
    ABASLANGICCSGRUP2ID VARCHAR(4),
    ABITISCSGRUP2ID VARCHAR(4),
    ABASLANGICCSGRUP3ID VARCHAR(4),
    ABITISCSGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    ACSGRUP1ARALIKLI INTEGER,
    ACSGRUP2ARALIKLI INTEGER,
    ACSGRUP3ARALIKLI INTEGER,
    ACEKSENETNO VARCHAR(30),
    AILKBORCLU VARCHAR(80),
    ADOVIZID VARCHAR(5),
    ARAPORDOVIZID VARCHAR(5),
    ASINIF VARCHAR(1))
RETURNS (
    AV VARCHAR(1),
    CS VARCHAR(1),
    BORDROID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    BORDROTARIHI TIMESTAMP,
    CEKSENETID INTEGER,
    GENELKAYITID INTEGER,
    ACIKLAMA VARCHAR(40),
    SINIF VARCHAR(1),
    VADETARIHI TIMESTAMP,
    MEBLAG DOUBLE PRECISION,
    TAHSILEDILENMEBLAG DOUBLE PRECISION,
    KALANMEBLAG DOUBLE PRECISION,
    DOVIZMEBLAG DOUBLE PRECISION,
    DOVIZTAHSILEDILENMEBLAG DOUBLE PRECISION,
    DOVIZKALANMEBLAG DOUBLE PRECISION,
    TAHSILTARIHI TIMESTAMP,
    DOVIZID VARCHAR(5),
    ISLEMDOVIZID VARCHAR(5),
    DOVIZKURUSTR VARCHAR(40),
    CEKSENETNO VARCHAR(30),
    HESAPNO VARCHAR(40),
    ILKBORCLU VARCHAR(80),
    KESIDEYERI VARCHAR(20),
    AITOLDUGUBANKAID INTEGER,
    AITOLDUGUBANKAADI VARCHAR(40),
    BANKAHESAPID VARCHAR(20),
    CEKSENETGRUP1 VARCHAR(4),
    CEKSENETGRUP2 VARCHAR(4),
    CEKSENETGRUP3 VARCHAR(4),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KARSILIKSIZ INTEGER,
    KARSILIKSIZTARIHI TIMESTAMP,
    CIROBORDROID INTEGER,
    CIROTARIHI TIMESTAMP,
    CIROFIRMAID VARCHAR(20),
    CIROFIRMAADI VARCHAR(100),
    CIROFIRMAADI2 VARCHAR(100),
    CIRODOVIZKURUSTR VARCHAR(40),
    BANKAYACIKISBORDROID INTEGER,
    BANKAYACIKISTARIHI TIMESTAMP,
    CIKISBANKAHESAPID VARCHAR(20),
    AVUKATID VARCHAR(10),
    AVUKATADI VARCHAR(50),
    AVUKATCIKISBORDROID INTEGER,
    AVUKATCIKISTARIHI TIMESTAMP,
    IADETARIHI TIMESTAMP,
    SONBORDROTIPI INTEGER,
    TAHSILEDILMIS INTEGER,
    VADETARIHIAYTOPLAM TIMESTAMP)
AS
declare variable DovizKuru1 Double Precision;
declare variable DovizKuru2 Double Precision;
declare variable CiroDovizKuru1 Double Precision;
declare variable CiroDovizKuru2 Double Precision;
declare variable FirmaDovizId varchar(5);
declare variable CiroFirmaDovizId varchar(5);
declare variable SonHareketId integer;
declare variable Okey integer;
declare variable FRaportarihi  Date;
declare variable DovizKuru Double Precision;
declare variable FDovizID varchar(5);
declare variable FaturaCariyeIslemDovizinden integer;
begin
  Select FaturaCariyeIslemDovizinden from Sistem into :FaturaCariyeIslemDovizinden;

  if (AIlkBorclu is not null and AIlkBorclu <> '') then AIlkBorclu = '%' || AIlkBorclu || '%';
  else AIlkBorclu = Null;

  FRaporTarihi = current_TimeStamp(0);
  for Select CekSenetDetay.Av, CekSenetDetay.Cs, CekSenet.BordroId, CekSenet.FirmaId, Firma1.Adi, Firma1.Adi2, Firma1.DovizId, CekSenet.tarih,
             CekSenetDetay.CekSenetId, CekSenetDetay.genelKayitId, CekSenetDetay.Aciklama, CekSenetDetay.Sinif, CekSenetDetay.VadeTarihi,
             CekSenetDetay.Meblag, CekSenetDetay.TahsilEdilenMeblag, CekSenetDetay.DovizId, CekSenetDetay.DovizKuru1, CekSenetDetay.DovizKuru2,
             CekSenetDetay.CekSenetNo, CekSenetDetay.HesapNo, CekSenetDetay.IlkBorclu, CekSenetDetay.KesideYeri,
             CekSenetDetay.AitOlduguBankaId, Banka.Adi, CekSenetDetay.BankaHesapId, CekSenetDetay.CekSenetGrup1,
             CekSenetDetay.CekSenetGrup2, CekSenetDetay.CekSenetGrup3, CekSenetDetay.Notu, CekSenetDetay.Karsiliksiz,
             CekSenetDetay.KarsiliksizTarihi, CekSenetDetay.SonBordroTipi, CekSenetDetay.SonHareketId, CekSenetDetay.IslemDovizId  from CekSenet
    left join CekSenetDetay on CekSenetDetay.CsBordroKayitId = CekSenet.CsBordroKayitId
    left join Firma Firma1 on Firma1.FirmaId = CekSenet.FirmaId
    left join BankaHesap on BankaHesap.BankaHesapId = CekSenetDetay.BankaHesapId
    left join banka on (banka.bankaId = CekSenetDetay.AitOlduguBankaId or banka.bankaId = BankaHesap.BankaId)
    where (CekSenet.Tarih >= :ABaslangicTarihi and CekSenet.Tarih <= :ABitistarihi) and
          CekSenet.AV = :AAv and CekSenet.CS = :ACs and
          (CekSenetDetay.DovizId = :ADovizId or :ADovizId is null) and
          (CekSenetDetay.Karsiliksiz = :AProtesto or :AProtesto is null) and
          (CekSenetDetay.BankaHesapId = :ABankaHesapId or (:ABankaHesapId is null or :AAv = 'A')) and
          (CekSenetDetay.Sinif = :ASinif or :ASinif is null) and
          (CekSenetDetay.CekSenetNo = :ACekSenetNo or :ACekSenetNo is null) and
          (CekSenetDetay.IlkBorclu Like :AIlkBorclu or :AIlkBorclu is null) and
          ((:AFirmaAralikli = 1 and CekSenet.FirmaId >= :ABaslangicFirmaId and (CekSenet.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
          ((:ACSGrup1Aralikli = 1 and CekSenetDetay.CekSenetGrup1 >= :ABaslangicCSGrup1Id and (CekSenetDetay.CekSenetGrup1 <= :ABitisCsGrup1Id or :ABitisCsGrup1Id = '')) or (:ACSGrup1Aralikli = 0)) and
          ((:ACSGrup2Aralikli = 1 and CekSenetDetay.CekSenetGrup2 >= :ABaslangicCSGrup2Id and (CekSenetDetay.CekSenetGrup2 <= :ABitisCsGrup2Id or :ABitisCsGrup2Id = '')) or (:ACSGrup2Aralikli = 0)) and
          ((:ACSGrup3Aralikli = 1 and CekSenetDetay.CekSenetGrup3 >= :ABaslangicCSGrup3Id and (CekSenetDetay.CekSenetGrup3 <= :ABitisCsGrup3Id or :ABitisCsGrup3Id = '')) or (:ACSGrup3Aralikli = 0)) and
          CekSenetDetay.GenelKayitId is not null
          order by CekSenet.Tarih, CeksenetDetay.VadeTarihi
    into :Av, :Cs, :BordroId, :FirmaId, :FirmaAdi, :FirmaAdi2, :FirmaDovizId, :BordroTarihi, :CekSenetId, :GenelKayitId, :Aciklama, :Sinif, :VadeTarihi,
         :Meblag, :TahsilEdilenMeblag, :DovizId, :DovizKuru1, :DovizKuru2, :CekSenetNo, :HesapNo, :IlkBorclu, :KesideYeri,
         :AitOlduguBankaId, :AitOlduguBankaAdi, :BankaHesapId, :CekSenetGrup1, :CekSenetGrup2, :CekSenetGrup3,
         :Notu, :Karsiliksiz, :KarsiliksizTarihi, :SonBordroTipi, :SonHareketId, :IslemDovizId do
  begin
    Okey = 1;
    if (FaturaCariyeIslemDovizinden = 1) then FDovizId = IslemDovizId;
    else FDovizId = DovizId;

    if (FirmaDovizId <> FDovizID) then
    begin
      DovizKuruStr = Format_Float('', DovizKuru1) || ' ' || DovizId || '-' || Format_Float('', DovizKuru2) || ' ' || FirmaDovizId;
    end
    else DovizKuruStr = '';

    KalanMeblag = Meblag - TahsilEdilenMeblag;

    if (TahsilEdilenMeblag = 0) then TahsilEdilmis = 0;
    else if (TahsilEdilenMeblag < Meblag) then TahsilEdilmis = 1;
    else TahsilEdilmis = 2;

    Okey = 1;
    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (ACSGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicCsGrup1Id, CekSenetGrup1) = 0) then okey = 0;

    if (ACSGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicCsGrup2Id, CekSenetGrup2) = 0) then okey = 0;

    if (ACSGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicCsGrup3Id, CekSenetGrup3) = 0) then okey = 0;


    if (ATahsil is not null) then
    if (ATahsil = 0) then /* Tahsil Edilememişleri istemiş ise */
    begin
      if (KalanMeblag = 0) then okey = 0;
    end
    else /* ATahsil = 1 ise yani tahsil edilmişleri istemiş ise */
    begin
      if (KalanMeblag > 0) then okey = 0;
    end

    if (Okey = 1) then
    if (AProtesto is not null) then
    if (AProtesto = 0 And Karsiliksiz = 1) then okey = 0;
    else if (AProtesto = 1 And Karsiliksiz = 0) then okey = 0;

    if (Okey = 1) then
    if (AAvukat is not null) then
    if (AAvukat = 0 and SonBordroTipi = 5) then Okey = 0;
    else if (AAvukat = 1 and SonBordroTipi <> 5) then Okey = 0;

    if (Okey = 1) then
    if (AIadeEdilenler is not null) then
    if (AIadeEdilenler = 0 and (SonBordroTipi = 7 or SonBordroTipi = 8)) then Okey = 0;
    else if (AIadeEdilenler = 1 and (SonBordroTipi <> 7 and SonBordroTipi <> 8)) then Okey = 0;

    if (Okey = 1) then
    if (ABankaCikis is not null) then
    if (ABankaCikis = 0 and SonBordroTipi = 3) then Okey = 0;
    else if (ABankaCikis = 1 and SonBordroTipi <> 3) then Okey = 0;

    if (Okey = 1) then
    if (ACiro is not null) then
    if (ACiro = 0 and SonBordroTipi = 1) then Okey = 0;
    else if (ACiro = 1 and SonBordroTipi <> 1) then Okey = 0;

    CiroBordroId = Null;
    CiroTarihi   = Null;
    CiroFirmaId  = Null;
    CiroFirmaAdi = Null;
    CiroFirmaAdi2 = Null;
    CiroDovizKuruStr = Null;

    BankayaCikisTarihi = Null;
    BankayaCikisBordroId = NUll;
    CikisBankaHesapId = Null;

    AvukatId             = Null;
    AvukatAdi            = Null;
    AvukatCikisBordroId  = Null;
    AvukatCikisTarihi    = Null;

    TahsilTarihi = Null;
    IadeTarihi = Null;

    if (KalanMeblag = 0) then
      select Max(tarih) from CekSenetTahsilat where GenelKayitId = :GenelKayitId into :TahsilTarihi;

    if (Okey = 1) then
    begin
      if (SonBordroTipi = 1) then
      begin
        select CekSenetHareket.tarih, CekSenetHareket.BordroId,CekSenetHareket.FirmaId, Firma.Adi, Firma.Adi2, Firma.DovizId, CekSenetHareketDetay.DovizKuru1, CekSenetHareketDetay.DovizKuru2 from ceksenethareketdetay
          left join CekSenethareket on CekSenetHareket.CshBordroKayitId = CekSenetHareketDetay.CshBordroKayitId
          left Join Firma On Firma.FirmaId = CekSenetHareket.FirmaId
          where ceksenethareketdetay.cshareketdetayid = :SonHareketId
          into :CiroTarihi, :CiroBordroId, :CiroFirmaId, :CiroFirmaAdi, :CiroFirmaAdi2, :FirmaDovizId, :CiroDovizKuru1, :CiroDovizKuru2;

          if (FirmaDovizId <> DovizId) then
          begin
            CiroDovizKuruStr = Format_Float('', CiroDovizKuru1) || ' ' || DovizId || '-' || Format_Float('', CiroDovizKuru2) || ' ' || FirmaDovizId;
          end
          else CiroDovizKuruStr = '';

      end
      else
      if (SonBordroTipi = 3) then
      begin
        select CekSenetHareket.tarih, CekSenetHareket.BordroId,CekSenetHareket.BankaHesapId from ceksenethareketdetay
          left join CekSenethareket on CekSenetHareket.CshBordroKayitId = CekSenetHareketDetay.CshBordroKayitId
          where ceksenethareketdetay.cshareketdetayid = :SonHareketId
          into :BankayaCikisTarihi, :BankayaCikisBordroId, :CikisBankaHesapId;
          if (ABankaHesapId is not null and CikisBankaHesapId <> ABankaHesapId) then Okey = 0;
      end
      else
      if (SonBordroTipi = 5) then
      begin
        select CekSenetHareket.tarih, CekSenetHareket.BordroId,CekSenetHareket.AvukatId, Avukat.Adi from ceksenethareketdetay
          left join CekSenethareket on CekSenetHareket.CshBordroKayitId = CekSenetHareketDetay.CshBordroKayitId
          left Join Avukat on Avukat.AvukatId = CekSenetHareket.AvukatId
          where ceksenethareketdetay.cshareketdetayid = :SonHareketId
          into :AvukatCikisTarihi, :AvukatCikisBordroId, :AvukatId, :AvukatAdi;
      end
      else
      if (SonBordroTipi = 7 or SonBordroTipi = 8) then
      begin
        select CekSenetHareket.tarih from ceksenethareketdetay
          left join CekSenethareket on CekSenetHareket.CshBordroKayitId = CekSenetHareketDetay.CshBordroKayitId
          where ceksenethareketdetay.cshareketdetayid = :SonHareketId
          into :IadeTarihi;
      end

      if (Okey = 1) then
      begin
        if (DovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :FRaporTarihi) into :DovizKuru;
        else DovizKuru = 1;


        DovizMeblag = Meblag * DovizKuru;
        DovizTahsilEdilenMeblag = TahsilEdilenMeblag * DovizKuru;
        DovizKalanMeblag = KalanMeblag * DovizKuru;

        DovizMeblag = Yuvarla(DovizMeblag, 1000);
        DovizTahsilEdilenMeblag = Yuvarla(DovizTahsilEdilenMeblag, 1000);
        DovizKalanMeblag = Yuvarla(DovizKalanMeblag, 1000);
        VadeTarihiAyToplam = TimeStampToAyIlkGun(VadeTarihi);
        suspend;
      end
    end
  end
end
