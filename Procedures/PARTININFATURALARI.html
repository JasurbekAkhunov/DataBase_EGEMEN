CREATE PROCEDURE PARTININFATURALARI(
    APARTIKAYITID INTEGER)
RETURNS (
    BOYAFATURAID INTEGER,
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    ENTEGRASYONSTR VARCHAR(4),
    ODEMETARIHI TIMESTAMP,
    TEVKIFATSTR VARCHAR(3),
    FATURAGRUPID VARCHAR(4),
    DOVIZID VARCHAR(5),
    DOVIZKURUSTR VARCHAR(40),
    SINIF VARCHAR(1))
AS
declare variable Entegrasyon           integer;
declare variable Tevkifat              integer;
declare variable DovizKuru1            Double Precision;
declare variable DovizKuru2            Double Precision;
declare variable FirmaDovizId          varchar(5);
begin
  For Select BoyaFatura.BoyaFaturaId, Firma.DovizId, BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo, BoyaFatura.FaturaTarihi, BoyaFatura.Entegrasyon,
             BoyaFatura.Odemetarihi, BoyaFatura.Tevkifat, BoyaFatura.FaturaGrupId, BoyaFatura.DovizId, BoyaFatura.DovizKuru1,
             BoyaFatura.DovizKuru2, BoyaFatura.Sinif from BoyaIrsaliyeDetay
      left join BoyaFatura on BoyaFatura.BoyaFaturaId = BoyaIrsaliyeDetay.BoyaFaturaId
      left join Firma on Firma.FirmaId = BoyaFatura.FirmaId
      where BoyaIrsaliyeDetay.PartiKayitId = :APartiKayitId and BoyaIrsaliyeDetay.BoyaFaturaId is not null
      group By BoyaFatura.BoyaFaturaId, Firma.DovizId, BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo, BoyaFatura.FaturaTarihi, BoyaFatura.Entegrasyon,
             BoyaFatura.Odemetarihi, BoyaFatura.Tevkifat, BoyaFatura.FaturaGrupId, BoyaFatura.DovizId, BoyaFatura.DovizKuru1,
             BoyaFatura.DovizKuru2, BoyaFatura.Sinif
      into :BoyaFaturaId, :FirmaDovizId, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :Entegrasyon, :Odemetarihi, :Tevkifat, :FaturaGrupId, :DovizId,
           :DovizKuru1, :DovizKuru2, :Sinif  do
  begin
    if (Entegrasyon = 1) then EntegrasyonStr = 'Evet';
    else EntegrasyonStr = '';

    if (Tevkifat = 1) then TevkifatStr = 'Var';
    else TevkifatStr = '';

    if (FirmaDovizId <> DovizId) then
    begin
      DovizKuruStr = Format_Float('', DovizKuru1) || ' ' || DovizId || '-' || Format_Float('', DovizKuru2) || ' ' || FirmaDovizId;
    end
    else DovizKuruStr = '';

    suspend;
  end
end
