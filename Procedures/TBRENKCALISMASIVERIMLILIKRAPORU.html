CREATE PROCEDURE TBRENKCALISMASIVERIMLILIKRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TALEPSAYISI DOUBLE PRECISION,
    CALISILANTALEPSAYISI INTEGER,
    TOPLAMATISSAYISI DOUBLE PRECISION,
    OKEYATISSAYISI INTEGER,
    OKEYTALEPSAYISI DOUBLE PRECISION,
    KRITIKATISSAYISI INTEGER,
    KRITIKTALEPSAYISI INTEGER,
    OKEYTALEPORANI DOUBLE PRECISION)
AS
declare variable LastFirmaId      varchar(20);
declare variable FLastLabReceteTalepkayitId integer;
declare variable FLabReceteTalepkayitId integer;
declare variable FLabReceteCalismaId integer;
declare variable FLabReceteAtisId    integer;
declare variable FFirmaId            varchar(20);
declare variable FFirmaAdi           varchar(100);
declare variable FOkey     integer;
declare variable FKritik   integer;
declare variable TalebinOkeyiVar  integer;
begin
  LastFirmaID = '';

  FLastLabReceteTalepkayitId = -1;

  For Select LabReceteTalep.FirmaId, Firma.Adi, LabReceteTalep.LabReceteTalepkayitId, LabReceteTalep.LabReceteCalismaId, LabReceteAtis.LabReceteAtisId, LabReceteAtis.Okey, LabReceteAtis.Kritik from LabReceteTalep
    left join LabReceteCalisma on LabReceteCalisma.LabReceteTalepKayitID = LabReceteTalep.LabReceteTalepKayitId
    left join LabReceteAtis on LabReceteAtis.LabReceteCalismaID = LabReceteCalisma.LabReceteCalismaId
    left join LabRecete on LabRecete.LabReceteId = LabReceteCalisma.LabReceteId
    left join Firma on Firma.FirmaID = LabReceteTalep.FirmaId
    left join Kullanici on Kullanici.KullaniciID = LabReceteTalep.TalepEdenKullaniciId
    where LabReceteTalep.TalepTarihi >= :ABaslangicTarihi and LabReceteTalep.TalepTarihi <= :ABitisTarihi
    order by LabReceteTalep.FirmaId, Firma.Adi, LabReceteTalep.TalepEdenKullaniciId, LabReceteTalep.LabReceteTalepkayitId
    into :FFirmaId, :FFirmaAdi, :FLabReceteTalepkayitId, :FLabReceteCalismaId, :FLabReceteAtisId, :FOkey, :FKritik do
  begin
    if (LastFirmaId <> FFirmaId) then
    begin
      if (LastFirmaId <> '') then
      begin
        OkeyTalepOrani = OkeyTalepSayisi / TalepSayisi * 100;
        suspend;
      end

      FLastLabReceteTalepkayitId = -1;
      LastFirmaId = FFirmaId;
      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;


      TalepSayisi           = 0;
      CalisilanTalepSayisi  = 0;
      ToplamAtisSayisi      = 0;
      OkeyAtisSayisi        = 0;
      OkeyTalepSayisi       = 0;
      KritikTalepSayisi     = 0;
      KritikAtisSayisi      = 0;
      OkeyTalepOrani        = 0;
    end

    if (FLastLabReceteTalepkayitId <> FLabReceteTalepkayitId) then
    begin
      TalepSayisi = TalepSayisi + 1;
      if (FLabReceteCalismaId is not null) then CalisilanTalepSayisi = CalisilanTalepSayisi + 1;
      TalebinOkeyiVar = 0;
      FLastLabReceteTalepkayitId = FLabReceteTalepkayitId;
    end

    if (FLabReceteAtisId is not null) then
    begin
      ToplamAtisSayisi = ToplamAtisSayisi + 1;
      if (FOkey = 1) then
      begin
        OkeyAtisSayisi = OkeyAtisSayisi + 1;

        if (TalebinOkeyiVar = 0) then
        begin
          OkeyTalepSayisi = OkeyTalepSayisi + 1;
          TalebinOkeyiVar = 1;
        end
      end
      if (FKritik = 1) then
      begin
        KritikAtisSayisi = KritikAtisSayisi + 1;
        KritikTalepSayisi = KritikTalepSayisi + 1;
      end
    end
  end

  if (LastFirmaId <> '') then
  begin
    OkeyTalepOrani = OkeyTalepSayisi / TalepSayisi * 100;
    suspend;
  end
end
