CREATE PROCEDURE TRANSFERICINORMEFATURAGETIR(
    AORMEFATURAID INTEGER)
RETURNS (
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    ODEMETARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DOVIZID VARCHAR(5),
    TRANSFERKODU VARCHAR(55),
    ATRANSFERKODU VARCHAR(55),
    MUHASEBEKODU VARCHAR(20),
    BRUTTOPLAM DOUBLE PRECISION,
    ISKONTOTOPLAMI DOUBLE PRECISION,
    PARASALISKONTO DOUBLE PRECISION,
    ISKONTOORANI DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    ARATOPLAM DOUBLE PRECISION,
    GENELTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZKURU1 DOUBLE PRECISION,
    ISLEMDOVIZKURU2 DOUBLE PRECISION,
    ISLEMDOVIZTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZTAHSILEDILECEKKDV DOUBLE PRECISION,
    FATURAGRUPTRANSFERKODU VARCHAR(10),
    TEVKIFAT INTEGER,
    TEVKIFATORANI DOUBLE PRECISION,
    SINIF VARCHAR(1),
    IRSALIYENO VARCHAR(255),
    EFATURA INTEGER,
    FATURAISKONTOSUTUTARI DOUBLE PRECISION,
    FIRMADOVIZKURU1 DOUBLE PRECISION,
    FIRMADOVIZKURU2 DOUBLE PRECISION,
    NOTU VARCHAR(1024),
    INSERTTARIHI TIMESTAMP)
AS
declare variable FIrsaliyeNo varchar(20);
declare variable FMusteriSiparisNo varchar(50);
declare variable FNotu Blob ;
declare variable BlobSize integer;
declare variable IlkSiparis SmallInt;
begin
  For Select OrmeFatura.FaturaTarihi, OrmeFatura.OdemeTarihi, OrmeFatura.SeriNo, OrmeFatura.FaturaNo, OrmeFatura.FirmaId,
         Coalesce(DovizliFirmaTransfer.TransferKodu,Firma.TransferKodu), Firma.ATransferKodu, Firma.MuhasebeKodu, OrmeFatura.DovizId,
         OrmeFatura.BrutToplam, OrmeFatura.ParasalIskonto, OrmeFatura.FaturaIskontosuOrani, OrmeFatura.IskontoToplami,
         OrmeFatura.TahsilEdilecekKdv, OrmeFatura.IndirilecekKdv, OrmeFatura.IslemDovizID, OrmeFatura.IslemDovizKuru1, OrmeFatura.IslemDovizKuru2,
         IslemDovizToplam, IslemDovizTahsilEdilecekKdv, StokHareketGrup.TransferKodu, OrmeFatura.Tevkifat, OrmeFatura.Sinif,
         (OrmeFatura.BrutToplam + OrmeFatura.TahsilEdilecekKDV - OrmeFatura.IskontoToplami) GenelToplam,
         (OrmeFatura.BrutToplam + OrmeFatura.TahsilEdilecekKDV - OrmeFatura.IskontoToplami + OrmeFatura.IndirilecekKdv) AraToplam, (cast(OrmeFatura.TevkifatOrani1 as Double Precision) / OrmeFatura.TevkifatOrani2) TevkifatOrani,
         OrmeFatura.EFatura, OrmeFatura.FaturaIskontosuTutari, OrmeFatura.FirmaDovizKuru1, OrmeFatura.FirmaDovizKuru2, OrmeFatura.Notu, OrmeFatura.InsertTarihi, Firma.Adi from OrmeFatura
  left join Firma On Firma.FirmaID = OrmeFatura.FirmaId
  left join DovizliFirmaTransfer On DovizliFirmaTransfer.FirmaID = OrmeFatura.FirmaId and DovizliFirmaTransfer.DovizID = OrmeFatura.IslemDovizID
  left join StokHareketGrup On StokHareketGrup.StokHareketGrupID = OrmeFatura.FaturaGrupId
  where OrmeFatura.OrmeFaturaID = :AOrmeFaturaId
  into :FaturaTarihi, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FirmaId, TransferKodu, :ATransferKodu, :MuhasebeKodu, :DovizId,
       :BrutToplam,
       :ParasalIskonto, :IskontoOrani, :IskontoToplami, :TahsilEdilecekKdv, :IndirilecekKdv, :IslemDovizID,
       :IslemDovizKuru1, :IslemDovizKuru2, :IslemDovizToplam, :IslemDovizTahsilEdilecekKdv, :FaturaGrupTransferKodu,
       :Tevkifat, :Sinif, :GenelToplam, :AraToplam, :TevkifatOrani, :EFatura, :FaturaIskontosuTutari, :FirmaDovizKuru1, :FirmaDovizKuru2, :FNotu, :InsertTarihi, :FirmaAdi do
  begin
    IrsaliyeNo = '';
    For Select OrmeIrsaliye.SeriNo || OrmeIrsaliye.IrsaliyeNo from StokHareket
      left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
      where StokHareket.OrmeFaturaID = :AOrmeFaturaId
      Group By OrmeIrsaliye.SeriNo,OrmeIrsaliye.IrsaliyeNo
    into :FIrsaliyeNo do
    begin
      if (Char_Length(IrsaliyeNo) + Char_Length(FIrsaliyeNo) < 240) then
      begin
        if (IrsaliyeNo <> '' ) then IrsaliyeNo = IrsaliyeNo || cast(', ' as varchar(2) character set UTF8);
        IrsaliyeNo = IrsaliyeNo || FIrsaliyeNo;
      end
    end

    BlobSize = char_length(FNotu);

    if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
    else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
    else Notu = '';

    IlkSiparis = 1;

    For Select OrmeSiparisDetay.MusteriSiparisNo from StokHareket
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
      where StokHareket.OrmeFaturaID = :AOrmeFaturaID
      Group By OrmeSiparisDetay.MusteriSiparisNo
    into :FMusteriSiparisNo do
    begin
      if (Char_Length(Notu) + Char_Length(FMusteriSiparisNo) < 970) then
      begin
        if (IlkSiparis = 1) then
        begin
          Notu = Notu || ' Sipler: ';
          IlkSiparis = 0;
        end
        else if (Notu <> '' ) then Notu = Notu || cast(', ' as varchar(2) character set UTF8);
        Notu = Notu || FMusteriSiparisNo;
      end
    end


    suspend;
  end
end
