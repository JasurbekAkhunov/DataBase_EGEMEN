CREATE PROCEDURE PARTISTOKTARTIBOL(
    APARTISTOKTARTIID INTEGER,
    AMIKTAR NUMERIC(12,3),
    AKULLANICIID INTEGER,
    AMETREKULLAN INTEGER)
RETURNS (
    ESKIPSKALITEKONTROLID INTEGER,
    PSKALITEKONTROLID INTEGER,
    PSKALITEKONTROLASAMAID INTEGER,
    YENIPARTISTOKTARTIID INTEGER,
    YENIBARKODID VARCHAR(30))
AS
declare variable PartiKayitId           integer;
declare variable PartiSinifId           integer;
declare variable HamStokSarfFisiDetayId integer;
declare variable BoyaIrsaliyeDetayId    integer;
declare variable HamTipiId              integer;
declare variable Kilo                   Numeric(12,3);
declare variable Metre                  Numeric(12,3);
declare variable AKilo                  Numeric(12,3);
declare variable AMetre                 Numeric(12,3);
declare variable PartiStokTartiId       integer;
declare variable KaliteKontrol          integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable BarkodId               varchar(30);
declare variable FDepoId                 DepoKoduSize;
begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  Select PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, Ham.HamTipiId,
         PartiStokTarti.Kilo, PartiStokTarti.Metre, PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.PSKaliteKontrolAsamaId, PartiStokTarti.BarkodId, PartiStokTarti.DepoId from PartiStokTarti
  left join PartiStok on PartiStok.PartiSinifID = PartiStokTarti.PartiSinifID
  left join PSKaliteKontrol on PSKaliteKontrol.PartiStokTartiID = PartiStokTarti.PartiStokTartiId
  left join Ham on Ham.HamID = PartiStok.HamId
  where PartistokTarti.PartistokTartiId = :APartistokTartiId
  into :PartiKayitId, :PartiSinifId, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :HamTipiId, :Kilo, :Metre, :EskiPSKaliteKontrolId, :PSKaliteKontrolAsamaId, :BarkodId, :FDepoId;

  if (BoyaIrsaliyeDetayId is not null or HamStokSarfFisiDetayId is not null) then exception KayitHatasi ' Sevk Edilmiş Top Bölünemez..';

  if ((HamTipiID <> 2 or DokumaKumasiKilodanTakipEt = 1) and (AMetreKullan = 0)) then
  begin
    if (AMiktar >= Kilo) then exception KayitHatasi ' Girilen Miktar Hatali';

    AKilo = AMiktar;

    if (Kilo <> 0) then AMetre = AKilo * Metre / Kilo;
    else AMetre = 0;

    AMetre = Yuvarla(AMetre, 1);
  end
  else
  begin
    if (AMiktar >= Metre) then exception KayitHatasi ' Girilen Miktar Hatali';
    AMetre = AMiktar;

    if (Metre <> 0) then AKilo = AMetre * Kilo / Metre;
    else AKilo = 0;
    AKilo = Yuvarla(AKilo, 100);
  end

  if (EskiPSKaliteKontrolId is not null and BarkodId is null) then
  begin
    Update PSKaliteKontrol Set Kilo = Kilo - :AKilo, Metre = Metre - :AMetre, LinkIslemi = 1, KullaniciId = :AKullaniciId where PSKaliteKontrol.PSKaliteKontrolId = :EskiPSKaliteKontrolId;
    KaliteKontrol = 1;
    YeniBarkodId = null;
  end
  else KaliteKontrol = 0;

  Update PartiStokTarti Set Kilo = Kilo - :AKilo, Metre = Metre - :AMetre, LinkIslemi = 2 where PartiStokTarti.PartiStokTartiID = :APartiStokTartiID;


  if (EskiPSKaliteKontrolId is not null and BarkodId is null) then
  begin
    PsKaliteKontrolID = Gen_Id(PsKaliteKontrol_ID_gen, 1);

    insert into PSKaliteKontrol(PSKaliteKontrolAsamaId, PsKaliteKontrolID, Tarih, PartiKayitId, PartiSinifId, MakinaId, PersonelId,
                                KaliteNo, Kilo, Metre, En, Gramaj, Donme, Egrilik, BaslangicEn, OrtaEn, BitisEn, OrtalamaEn, Aciklama, OnayDurumu, OnayTarihi, OnayKullaniciId,
                                InsertKullaniciId, KullaniciId, LinkIslemi, OlcuId, InsertTarihi)
              Select PSKaliteKontrolAsamaId, :PsKaliteKontrolID, Tarih, PartiKayitId, PartiSinifId, MakinaId, PersonelId,
                                KaliteNo, :AKilo, :AMetre, En, Gramaj, Donme, Egrilik, BaslangicEn, OrtaEn, BitisEn, OrtalamaEn, Aciklama, OnayDurumu, OnayTarihi, OnayKullaniciId,
                                InsertKullaniciId, KullaniciId, 1, OlcuId, InsertTarihi from PsKaliteKontrol
              where PSKaliteKontrol.PartiStokTartiID = :APartiStokTartiID;

    PartiStokTartiId = gen_id(PartiStokTarti_Id_Gen, 1);

    Insert Into PartiStokTarti(PartiStokTartiId, PartiKayitId, PartiSinifId, Kilo, Metre, KapAdedi, KaliteNo, KaliteKontrol, InsertKullaniciId, KullaniciId, LinkIslemi, BarkodId, BolunduguPartiStokTartiId, OlcuId, DepoId)
              Select :PartiStokTartiId, PartiKayitId, PartiSinifId, :AKilo, :AMetre, 1, KaliteNo, :KaliteKontrol, :AKullaniciId, :AKullaniciId, 2, :YeniBarkodId, :APartiStokTartiID, OlcuId, :FDepoId
              from PsKaliteKontrol where PsKaliteKontrol.PsKaliteKontrolID = :PsKaliteKontrolID;

    Update PsKaliteKontrol set PartiStokTartiId = :PartiStokTartiId, LinkIslemi = 1, KullaniciId = :AKullaniciId where PsKaliteKontrol.PsKaliteKontrolId = :PsKaliteKontrolId;
  end
  else
  begin
    PartiStokTartiId = gen_id(PartiStokTarti_Id_Gen, 1);

    YeniBarkodId = 'R-'||PartiStokTartiId;

    Insert Into PartiStokTarti(PartiStokTartiId, PartiKayitId, PartiSinifId, Kilo, Metre, KapAdedi, KaliteNo, KaliteKontrol, InsertKullaniciId, KullaniciId, LinkIslemi, BarkodId, BolunduguPartiStokTartiId, OlcuId, DepoId)
            Select :PartiStokTartiId, PartiKayitId, PartiSinifId, :AKilo, :AMetre, 1, KaliteNo, :KaliteKontrol, :AKullaniciId, :AKullaniciId, 2, :YeniBarkodId, :APartiStokTartiID, OlcuId, :FDepoId
            from PartiStokTarti where PartiStokTarti.PartiStokTartiID = :APartiStokTartiId;
  end

  insert into MamulDepoTransferFisiDetay(MamulDepoTransferFisiId, PartiSinifId, PartiStokTartiId, RafId, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
         Select MamulDepoTransferFisiDetay.MamulDepoTransferFisiId, :PartiSinifId, :PartiStokTartiId, MamulDepoTransferFisiDetay.RafId,
                MamulDepoTransferFisiDetay.InsertKullaniciId, MamulDepoTransferFisiDetay.InsertTarihi, MamulDepoTransferFisiDetay.KullaniciId, MamulDepoTransferFisiDetay.DegisimTarihi from MamulDepoTransferFisiDetay
         left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiID = MamulDepoTransferFisiDetay.MamulDepoTransferFisiId
         where MamulDepoTransferFisiDetay.PartiStokTartiID = :APartiStokTartiID and MamulDepoTransferFisi.SistemKaydi = 0 Order By MamulDepoTransferFisi.InsertTarihi;

  if (EskiPSKaliteKontrolId is not null and BarkodId is null) then
  begin
    update PSKaliteKontrolHataMetre Set PsKaliteKontrolID = :PsKaliteKontrolID, BaslangicMetre = BaslangicMetre - (:Metre - :AMetre), BitisMetre = BitisMetre - (:Metre - :AMetre), LinkIslemi = 1
      where PSKaliteKontrolHataMetre.PsKaliteKontrolID = :EskiPsKaliteKontrolID and PSKaliteKontrolHataMetre.BaslangicMetre > (:Metre - :AMetre);
  end

  YeniPartiStokTartiId = PartiStokTartiId;
  suspend;
end
