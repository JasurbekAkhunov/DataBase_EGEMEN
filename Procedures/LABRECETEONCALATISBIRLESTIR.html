CREATE PROCEDURE LABRECETEONCALATISBIRLESTIR(
    ABIRINCIONCALISMAATISID INTEGER,
    AIKINCIONCALISMAATISID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable LabReceteAtisId integer;
begin
  LabReceteAtisId = gen_id(LabReceteOnCalismaAtis_Id_Gen, 1);

  Insert into LabReceteOnCalismaAtis(LabReceteCalismaId, LabReceteId, LabReceteOnCalismaAtisId, Migrasyon, Color, Notu, NotuStr, RenkDerinligiId, ReceteAsamaId, GidisTarihi, InsertKullaniciId, KullaniciId, YeniLabReceteOnCalismaAtisID, Birlestirildi)
      Select LabReceteCalismaId, LabReceteId, :LabReceteAtisId, Migrasyon, Color, Notu, NotuStr, RenkDerinligiId, ReceteAsamaId, Current_date, :AKullaniciId, :AKullaniciId, LabReceteOnCalismaAtisID, 1 from LabReceteOnCalismaAtis
        where LabReceteOnCalismaAtis.LabReceteOnCalismaAtisID = :ABirinciOnCalismaAtisId;

  Insert into LabReceteOnCalismaUrun(LabReceteOnCalismaAtisId, LabReceteId, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, InsertKullaniciId, KullaniciId, LabReceteOnCalismaUrunID)
      Select :LabReceteAtisId, LabReceteId, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, :AKullaniciId, :AKullaniciId, LabReceteOnCalismaUrunID from LabReceteOnCalismaUrun
        where LabReceteOnCalismaUrun.LabReceteOnCalismaAtisID = :ABirinciOnCalismaAtisId
        order by LabReceteOnCalismaUrun.LabReceteOnCalismaUrunId;

  Insert into LabReceteOnCalismaUrun(LabReceteOnCalismaAtisId, LabReceteId, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, InsertKullaniciId, KullaniciId, LabReceteOnCalismaUrunID)
      Select :LabReceteAtisId, LabReceteId, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, :AKullaniciId, :AKullaniciId, LabReceteOnCalismaUrunID from LabReceteOnCalismaUrun
        where LabReceteOnCalismaUrun.LabReceteOnCalismaAtisID = :AIkinciOnCalismaAtisId
        order by LabReceteOnCalismaUrun.LabReceteOnCalismaUrunId;

  Update LabReceteOnCalismaAtis set YeniLabReceteOnCalismaAtisID = :LabReceteAtisId where (LabReceteOnCalismaAtis.LabReceteOnCalismaAtisID = :ABirinciOnCalismaAtisId or LabReceteOnCalismaAtis.LabReceteOnCalismaAtisID = :AIkinciOnCalismaAtisId );
  Execute Procedure LabReceteOnCalAtisGonder(:LabReceteAtisId, :AKullaniciId);
end
