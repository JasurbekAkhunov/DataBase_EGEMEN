CREATE PROCEDURE SATINALMATALEPTEKLIFGETIR(
    ASATINALMATALEPKAYITID INTEGER)
RETURNS (
    TEKLIFTALEPDETAYID INTEGER,
    TEKLIFTALEPYIL INTEGER,
    TEKLIFTALEPID INTEGER,
    TEKLIFTALEPIDSTR VARCHAR(30),
    TALEPTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SATINALMATALEPDETAYID INTEGER,
    MIKTAR DOUBLE PRECISION,
    TEKLIFGELDI INTEGER,
    TEKLIFGELDISTR VARCHAR(5),
    TEKLIFGELISTARIHI TIMESTAMP,
    TEKLIFDOSYAADI VARCHAR(255),
    BIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    BIRIMFIYATTL DOUBLE PRECISION,
    TUTARTL DOUBLE PRECISION,
    SISTEMDOVIZID VARCHAR(5),
    TERMINTARIHI TIMESTAMP,
    YUKLEMETARIHI TIMESTAMP,
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(40),
    TESLIMSEKLIID INTEGER,
    TESLIMSEKLIADI VARCHAR(40),
    VADEGUN INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(4),
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable DovizKuru Double Precision;
begin
  Select SistemDovizId from Sistem where SistemID = 1 into :SistemDovizId;

  For Select TeklifTalep.TeklifTalepYil, TeklifTalep.TeklifTalepId, TeklifTalepDetay.TeklifTalepDetayId, TeklifTalep.TalepTarihi, TeklifTalepDetay.FirmaId, Firma.Adi, TeklifTalepDetay.SatinAlmaTalepDetayId,
             TeklifTalepDetay.Miktar, TeklifTalepDetay.TeklifGeldi, TeklifTalepDetay.TeklifGelisTarihi, TeklifTalepDetay.TeklifDosyaAdi, TeklifTalepDetay.BirimFiyat,
             TeklifTalepDetay.DovizId, TeklifTalepDetay.TerminTarihi, TeklifTalepDetay.YuklemeTarihi, TeklifTalepDetay.TeslimSekliId, TeslimSekli.Adi,
             TeklifTalepDetay.OdemeSekliId, OdemeSekli.Adi,
             TeklifTalepDetay.VadeGun, TeklifTalepDetay.Notu,
             IK.Adi, TeklifTalepDetay.InsertTarihi, SK.Adi, TeklifTalepDetay.DegisimTarihi,
             TeklifTalepDetay.OnayDurumu, TeklifTalepDetay.OnaylayanKullaniciId, OK.Adi, TeklifTalepDetay.OnayTarihi,
             VerilenSiparis.VerilenSiparisyil, VerilenSiparis.VerilenSiparisId from TeklifTalepDetay
    Left Join Kullanici IK on IK.KullaniciId = TeklifTalepDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = TeklifTalepDetay.KullaniciId
    Left Join Kullanici OK on OK.KullaniciId = TeklifTalepDetay.OnaylayanKullaniciId
    left join OdemeSekli On OdemeSekli.OdemeSekliId = TeklifTalepDetay.OdemeSekliId
    left join TeslimSekli On TeslimSekli.TeslimSekliId = TeklifTalepDetay.TeslimSekliId
    left join Firma On Firma.FirmaId = TeklifTalepDetay.FirmaId
    Left Join TeklifTalep on TeklifTalep.TeklifTalepKayitId = TeklifTalepDetay.TeklifTalepKayitId
    left join VerilenSiparisDetay On VerilenSiparisDetay.VerilenSiparisDetayId = TeklifTalepDetay.VerilenSiparisDetayId
    left join VerilenSiparis On VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
    where TeklifTalepDetay.SatinAlmaTalepKayitId = :ASatinAlmaTalepKayitId
    into :TeklifTalepYil, :TeklifTalepId, :TeklifTalepDetayId, :TalepTarihi, :FirmaId, :FirmaAdi, :SatinAlmaTalepDetayId, :Miktar, :TeklifGeldi, :TeklifGelisTarihi,
         :TeklifDosyaAdi, :BirimFiyat, :DovizId, :TerminTarihi,
         :YuklemeTarihi, :TeslimSekliId, :TeslimSekliAdi, :OdemeSekliId, :OdemeSekliAdi,
         :VadeGun, :Notu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
         :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi,
         :VerilenSiparisyil, :VerilenSiparisId do
  begin
    if (TeklifGeldi = 1) then TeklifGeldiStr = 'Evet';
    else TeklifGeldiStr = '';

    if (OnayDurumu = 1) then OnayDurumuStr = 'Okey';
    else OnayDurumuStr = '';

    BirimFiyatTL = Null;
    TutarTL = NUll;

    Tutar = BirimFiyat * Miktar;


    TeklifTalepIdStr = '(' || EgemenIntegerToStr(TeklifTalepYil) || ') '  || EgemenIntegerToStr(TeklifTalepId)  || ' ';


    VerilenSiparisIdStr = '';
    if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || EgemenIntegerToStr(VerilenSiparisYil) || ') '  || EgemenIntegerToStr(VerilenSiparisId)  || ' ';

    if (DovizId <> SistemDovizId) then  select DovizKuru from DovizKuruGetir(:DovizId, :SistemDovizId, Current_date) into :DovizKuru;
     else DovizKuru = 1;

    BirimFiyatTL = BirimFiyat * DovizKuru;
    TutarTL = BirimFiyatTL * Miktar;

    suspend;
  end
end
