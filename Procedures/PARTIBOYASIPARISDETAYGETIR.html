CREATE PROCEDURE PARTIBOYASIPARISDETAYGETIR(
    APARTIKAYITID INTEGER,
    ASIPARISFASONBRFIYATGOREBILIR INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    LABRECETEKODU VARCHAR(20),
    BOYASIPARISRENKNO INTEGER,
    RENKADI VARCHAR(40),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    PLANLAMANOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SIPARISNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    MUSTERISIPARISNO VARCHAR(50),
    FIRMASEVKADRESIADI VARCHAR(50),
    FASONBIRIMFIYATI DOUBLE PRECISION,
    FASONBIRIMFIYATDOVIZID VARCHAR(5),
    BOYANANKILO DOUBLE PRECISION,
    PARTILENENKILO DOUBLE PRECISION,
    IPTALKILO DOUBLE PRECISION,
    BOBINSAYISI INTEGER,
    REFERANSACIKLAMA VARCHAR(50),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    DONMEISTENEN VARCHAR(10),
    CEKMEZLIKENISTENEN DOUBLE PRECISION,
    CEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI ADI20,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    REFERANSKODU HAMKODUSIZE,
    HAMSTOKMASTERIDSTR VARCHAR(100),
    LOT VARCHAR(1000),
    YARIMAMULPARTINOSTR VARCHAR(200),
    MUSTERIKALITENO VARCHAR(20),
    SIPARISKILO NUMERIC(12,3))
AS
declare variable PartilenenMiktardanIptaliDus Boolean;
declare variable FLot                   varchar(1500);
declare variable FHamStokMasterID       integer;
declare variable Str                    varchar(20);
declare variable YariMamulPartiYil integer;
declare variable YariMamulPartiId  integer;
begin
  select PartilenenMiktardanIptaliDus from Sistem where Sistem.SistemId = 1 into :PartilenenMiktardanIptaliDus;

  for select PartiStok.BoyaSiparisDetayId, PartiStok.HamAdiToplami, PartiStok.Kilo, BoyaSiparisDetay.MusteriRenkKoduveAdi,
      BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.BoyaSiparisRenkNo, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.PlanlamaNotu, BoyaSiparisDetay.Notu,
      BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi, BoyaSiparisDetay.MusteriSiparisNo,
      FirmaSevkAdresi.AdresAdi, BoyaSiparisDetay.FasonBirimFiyati, BoyaSiparisDetay.FasonBirimFiyatDovizId,
      BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparisDetay.PartilenenKilo, BoyaSiparisDetay.IptalKilo,
      BoyaSiparisDetay.BobinSayisi, PartiStokFizikKontrol.ReferansAciklama, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen,
      BoyaSiparisDetay.DonmeIstenen, BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen,
      RenkDerinligi.RenkDerinligiId, RenkDerinligi.Adi, BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, Ham.ReferansKodu, BoyaSiparisDetay.MusteriKaliteNo,
      BoyaSiparisDetay.SiparisKilo from PartiStok
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join FirmaSevkAdresi on FirmaSevkAdresi.FirmaSevkAdresiId = PartiStok.FirmaSevkAdresiId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
  left join Ham on Ham.HamId = PartiStok.HamId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
  where PartiStok.PartiKayitId = :APARTIKAYITID
  into :BoyaSiparisDetayId, :HamAdiToplami, :Kilo, :MusteriRenkKoduveAdi,
       :LabReceteKodu, :BoyaSiparisRenkNo, :RenkAdi, :PlanlamaNotu, :SiparisNotu,
       :TerminTarihi, :RevizeTerminTarihi, :MusteriSiparisNo,
       :FIRMASEVKADRESIADI, :FasonBirimFiyati, :FasonBirimFiyatDovizId,
       :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :PartilenenKilo, :IptalKilo,
       :BobinSayisi, :ReferansAciklama, :EnIstenen, :GramajIstenen,
       :DonmeIstenen, :CekmezlikEnIstenen, :CekmezlikBoyIstenen,
       :RenkDerinligiId, :RenkDerinligiAdi, :BirimFiyat, :DovizId, :ReferansKodu, :MusteriKaliteNo,
       :SiparisKilo do
  begin
    if (ASiparisFasonBrFiyatGorebilir = 0) then FasonBirimFiyati = null;

    select sum(PartiStok.Kilo) from PartiStok where PartiStok.Durumu > 4 and PartiStok.BoyaSiparisDetayID = :BoyaSiparisDetayId into :BoyananKilo;

    if (PartilenenMiktardanIptaliDus = 1) then
      PartilenenKilo = PartilenenKilo - IptalKilo;

    Lot = '';
    HamStokMasterIDStr = '';
    YariMamulPartiNoStr = '';
    for select HamStokMaster.HamStokMasterID, HamStokMaster.Lot, Parti.PartiYil, Parti.PartiID from PartiStok
      left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiStok.PartiSinifID
      left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join PartiStok YariMamulPartiStok on YariMamulPartiStok.PartiSinifId = HamStokMaster.PartiSinifId
      left join Parti on Parti.PartiKayitID = YariMamulPartiStok.PartiKayitId
      where PartiStok.PartiKayitId = :APartiKayitId
      group by HamStokMaster.HamStokMasterID, HamStokMaster.Lot, Parti.PartiYil, Parti.PartiID
      into :FHamStokMasterID, :FLot, :YariMamulPartiYil, :YariMamulPartiId do
    begin

      Str = EgemenIntegerToStr(FHamStokMasterID);
      if (Str is not null) then
      if (PosBul(HamStokMasterIDStr, Str) = 0) then
      if (char_length(HamStokMasterIDStr) + char_length(Str) < 97) then
      begin
        if (HamStokMasterIDStr <> '') then HamStokMasterIDStr = HamStokMasterIDStr || cast(', ' as varchar(2) character set UTF8);
        HamStokMasterIDStr = HamStokMasterIDStr || Str;
      end

     if (char_length(Lot) + char_length(FLot) < 998) then
      begin
       if (Lot <> '' ) then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
       Lot = Lot || FLot;
      end
    end

      Str = '('||YariMamulPartiYil||') '||YariMamulPartiId;
      if (YariMamulPartiYil is not null) then
      if (PosBul(YariMamulPartiNoStr, Str) = 0) then
      if (char_length(YariMamulPartiNoStr) + char_length(Str) < 198) then
      begin
        if (YariMamulPartiNoStr <> '') then YariMamulPartiNoStr = YariMamulPartiNoStr || cast(', ' as varchar(2) character set UTF8);
        YariMamulPartiNoStr = YariMamulPartiNoStr || Str;
      end
    suspend;
  end
end
