CREATE PROCEDURE SARFFISIICINSTOKLISTESI(
    AHAMID INTEGER,
    ADEPOID VARCHAR(10))
RETURNS (
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    RAFID VARCHAR(10),
    KALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI NUMERIC(12,3),
    KALITENO INTEGER,
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    URETICIADI VARCHAR(40),
    URETIMPARTINO VARCHAR(30),
    STOKFIRMAID VARCHAR(20),
    STOKFIRMAADI VARCHAR(50),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    BOYASIPARISIDSTR VARCHAR(30),
    KALITEACIKLAMATIPIID INTEGER,
    KALITEACIKLAMATIPIADI VARCHAR(20),
    OLCUADI VARCHAR(40),
    ORMEMAKINAADI VARCHAR(40))
AS
declare variable BoyaSiparisYil integer;
declare variable BoyaSiparisId  integer;
declare variable KS             integer;
begin
  For Select OrmeRaf.OrmeRafId, OrmeRaf.RafId, OrmeRaf.StokGirisNo, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari, OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi,
             OrmeRaf.Lot, OrmeRaf.MelanjKodu, OrmeRaf.KaliteNo, Uretici.Adi, UretimPartiNo,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID,
             OrmeRaf.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,  OrmeRaf.KaliteAciklamaTipiId, KaliteAciklamaTipi.Adi, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.MelanjKodu,
             Olcu.OlcuAdi, OrmeMakina.Adi from OrmeRaf
    left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
    left join Firma StokFirma on StokFirma.FirmaId = OrmeRaf.StokFirmaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
    left join Desen on Desen.DesenId = OrmeRaf.DesenId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = OrmeRaf.DesenVaryantId
    left join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = OrmeRaf.KaliteAciklamaTipiId
    left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
    left Join Renk ON Renk.RenkId = OrmeRaf.RenkId
    left join Olcu on Olcu.OlcuId = OrmeRaf.OlcuId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmeRaf.OrmeMakinaId
    where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and
         (DepoRaf.DepoId = :ADepoID or :ADepoID is null)
    into :OrmeRafId, :RafId, :StokGirisNo, :KalanMiktar, :KalanKapAdedi, :Lot, :MelanjKodu,  :KaliteNo, :UreticiAdi, :UretimPartiNo,
         :BoyaSiparisYil, :BoyaSiparisID, :DesenId, :DesenAdi, :VaryantAdi, :KaliteAciklamaTipiId, :KaliteAciklamaTipiAdi, :RenkId, :RenkAdi, :MelanjKodu,
         :OlcuAdi, :OrmeMakinaAdi do
  begin
    /*top top kayit varsa raf sectirmiyoruz, top top isleme izin verecegiz*/
    select count(*) from OrmeRafDetay
    left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketId
    left join GirisKaliteKontrol on GirisKaliteKontrol.StokHareketId = StokHareket.StokHareketId
    left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketId = StokHareket.StokHareketId
    where OrmeRafDetay.OrmeRafId = :OrmeRafId and
    StokHareket.DokumaHareketi = 1 and
    (GirisKaliteKontrol.GirisKaliteKontrolId is not null or DokumaKaliteKontrol.DokumaKaliteKontrolId is not null) into :KS;

    if (KS = 0) then
      suspend;
  end
end
