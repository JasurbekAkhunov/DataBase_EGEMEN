CREATE PROCEDURE IPLIKSARIMISEMIRLERIRAPORU(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMID INTEGER,
    ALOT VARCHAR(30),
    AIPLIKMAKINAID VARCHAR(10))
RETURNS (
    SARIMISEMRIKAYITID INTEGER,
    SARIMISEMRIYIL INTEGER,
    SARIMISEMRIID INTEGER,
    SARIMISEMRIIDSTR VARCHAR(30),
    PLANLAMATARIHI TIMESTAMP,
    IPLIKMAKINAID VARCHAR(10),
    SARIMSEKLIID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    LOT VARCHAR(30),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    MIKTAR NUMERIC(12,3),
    BITIRENKULLANICIID INTEGER,
    BITIRENKULLANICIADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select SarimIsEmri.SarimIsEmriKayitId, SarimIsEmri.SarimIsEmriYil, SarimIsEmri.SarimIsEmriId, SarimIsEmri.IplikMakinaId,
             SarimIsEmri.PlanlamaTarihi, IplikLot.HamId, Ham.Adi, Renk.Adi, IplikLot.Lot, IplikLot.MelanjKodu, IplikLot.FlamKodu, SarimIsemri.Miktar,
             SarimIsEmri.SarimSekliId, SarimIsEmri.BitirenKullaniciId, BitP.Adi, SarimIsEmri.BitisTarihi, Ik.Adi, SarimIsEmri.InsertTarihi, Sk.Adi, SarimIsEmri.DegisimTarihi from SarimIsEmri
  left join IplikLot on IplikLot.IplikLotId = SarimIsEmri.IplikLotId
  left join Ham on Ham.Hamid = IplikLot.HamId
  left join TamirGiris on TamirGiris.TamirGirisKayitID = IplikLot.TamirGirisKayitId
  left join IplikSatisIade on IplikSatisIade.IplikSatisIadeID = TamirGiris.IplikSatisIadeId
  left join Renk on Renk.RenkID = IplikSatisIade.RenkId
  left join Kullanici BitP on BitP.KullaniciId = SarimIsEmri.BitirenKullaniciId
  Left Join Kullanici IK on IK.KullaniciId = SarimIsEmri.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = SarimIsEmri.KullaniciId
  where SarimIsEmri.PlanlamaTarihi >= :ABaslangicTarihi and SarimIsEmri.PlanlamaTarihi < :ABitisTarihi + 1 and
        (IplikLot.HamID = :AHamId or :AHamId is null) and
        (IplikLot.Lot = :ALot or :ALot is null) and
        (SarimIsEmri.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null) and
        ( :ARaporTipi is null or (SarimIsEmri.BitisTarihi is null and :ARaporTipi = 0) or (SarimIsEmri.BitisTarihi is not null and :ARaporTipi = 1) )
  into :SarimIsEmriKayitId, :SarimIsEmriYil, :SarimIsEmriId, :IplikMakinaId, :PlanlamaTarihi, :HamId, :HamAdi, :RenkAdi, :Lot,
       :MelanjKodu, :FlamKodu,  :Miktar, :SarimSekliId,  :BitirenKullaniciId, :BitirenKullaniciAdi, :BitisTarihi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    SarimIsEmriIdStr = '(' || EgemenIntegerToStr(SarimIsEmriYil) || ') ' || EgemenIntegerToStr(SarimIsEmriId) || ' ';
    suspend;
  end
end
