CREATE PROCEDURE UNITELERARASISEVKPARTISECIMI(
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    AGONDERENUNITEID INTEGER,
    AALICIUNITEID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    HAMTIPIID INTEGER,
    HAMTIPISTR VARCHAR(20),
    NETKILO BOYAMIKTARSIZE,
    KILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    DURUMUSTR VARCHAR(20),
    GIDENKILO DOUBLE PRECISION,
    GIDENNETKILO DOUBLE PRECISION,
    GIDENMETRE DOUBLE PRECISION,
    GELENKILO DOUBLE PRECISION,
    GELENNETKILO DOUBLE PRECISION,
    GELENMETRE DOUBLE PRECISION)
AS
begin
  For Select PartiStok.PartiKayitID, PartiStok.PartiSinifID, PartiStok.HamAdiToplami, PartiStok.HamTipiID, HamTipi.Adi, PartiStok.NetKilo,
             PartiStok.Kilo, PartiStok.Metre, Parti.ReceteID, Recete.Adi, PartiDurum.Adi From Parti
        Left Join Recete ON Recete.ReceteID = Parti.ReceteID
        Left Join PartiStok ON  PartiStok.PartiKayitID = Parti.PartiKayitID
        Left Join HamTipi ON HamTipi.HamTipiID = PartiStok.HamTipiID
        Left Join PartiDurum ON PartiDurum.PartiDurumID = PartiStok.Durumu
      where Parti.PartiYil = :APartiYil AND Parti.PartiID = :APartiID
      into :PartiKayitID, :PartiSinifID, :HamAdiToplami, :HamTipiID, :HamTipiStr, :NetKilo, :Kilo, :Metre, :ReceteID, :ReceteAdi, :DurumuStr
  do
  begin
    Select GidenKilo, GidenNetKilo, GidenMetre, GelenKilo, GelenNetKilo, GelenMetre from UniteGidenGelenMiktarGetir(:AGonderenUniteID, :AAliciUniteID, :PartiSinifID)
      into :GidenKilo, :GidenNetKilo, :GidenMetre, :GelenKilo, :GelenNetKilo, :GelenMetre;

    suspend;
  end
end
