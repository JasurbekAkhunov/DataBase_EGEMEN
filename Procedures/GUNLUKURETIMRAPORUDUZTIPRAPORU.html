CREATE PROCEDURE GUNLUKURETIMRAPORUDUZTIPRAPORU(
    AMAKINAGRUBU VARCHAR(4),
    AMAKINARAPORGRUPID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP)
RETURNS (
    MAKINARAPORGRUPID INTEGER,
    TAMIRDUZELTMETIPIID VARCHAR(10),
    IPLIKKILO DOUBLE PRECISION,
    IPLIKAYKILO DOUBLE PRECISION,
    ORMEKUMASKILO DOUBLE PRECISION,
    ORMEKUMASAYKILO DOUBLE PRECISION,
    DOKUMAKUMASKILO DOUBLE PRECISION,
    DOKUMAKUMASAYKILO DOUBLE PRECISION)
AS
declare variable FCikisTarihi TimeStamp;
declare variable FHamTipiId    integer;
declare variable FKilo Double Precision;
declare variable FMakinaRaporGrupId  integer;
declare variable FTamirDuzeltmeTipiId   varchar(10);

declare variable FLastMakinaRaporGrupId  integer;
declare variable FLastTamirDuzeltmeTipiId   varchar(10);
begin
  FLastMakinaRaporGrupId = -2;
  FLastTamirDuzeltmeTipiId = '@';

  for select Makina.RaporGrupId, PartiTamir.TamirDuzeltmeTipiId, PartiStok.HamTipiId, PartiAsamaStokDetay.CikisTarihi, PartiAsamaStokDetay.Kilo from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    where PartiAsamaStokDetay.Cikistarihi >= :AAylikBaslangic and PartiAsamaStokDetay.Cikistarihi < :AAylikBitis and
          Makina.UretimRaporlarinaDahil = 1 and Asama.GunlukUretimRaporunaDahil = 1 and
         (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
         (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null) and
         (Parti.IslemTipiId <> 3 and PartiAsamalari.PartiTamirID is not null)
    order by Makina.RaporGrupId, PartiTamir.TamirDuzeltmeTipiId, PartiStok.HamTipiId
    into :FMakinaRaporGrupId, :FTamirDuzeltmeTipiId, :FHamTipiId, :FCikisTarihi, :FKilo do
  begin
    if (FLastMakinaRaporGrupId <> FMakinaRaporGrupId or StrKarsilastir(FlastTamirDuzeltmeTipiId, FTamirDuzeltmeTipiId) = 0) then
    begin
      if (FLastMakinaRaporGrupId <> -2) then suspend;

      if (FMakinaRaporGrupId is null) then FMakinaRaporGrupId = -1;
      if (FTamirDuzeltmeTipiId is null) then FTamirDuzeltmeTipiId = '-1';

      MakinaRaporGrupId     = FMakinaRaporGrupId;
      TamirDuzeltmeTipiId   = FTamirDuzeltmeTipiId;
      IplikKilo             = 0;
      IplikAyKilo           = 0;
      OrmeKumasKilo         = 0;
      OrmeKumasAyKilo       = 0;
      DokumaKumasKilo       = 0;
      DokumaKumasAyKilo     = 0;

      FLastMakinaRaporGrupId     = FMakinaRaporGrupId;
      FLastTamirDuzeltmeTipiId   = FTamirDuzeltmeTipiId;
    end

    if (FCikisTarihi >= ABaslangicTarihi and FCikisTarihi < ABitisTarihi) then
    begin
      if (FHamTipiID <> 2 and FHamtipiID <> 0 and FHamtipiID <> 5) then
        OrmeKumasKilo = OrmeKumasKilo + FKilo;
      else if (FHamTipiID = 2) then
        DokumaKumasKilo = DokumaKumasKilo + FKilo;
      else
        IplikKilo = IplikKilo + FKilo;
    end

    if (FHamTipiID <> 2 and FHamtipiID <> 0 and FHamtipiID <> 5) then
      OrmeKumasAyKilo = OrmeKumasAyKilo + FKilo;
    else if (FHamTipiID = 2) then
      DokumaKumasAyKilo = DokumaKumasAyKilo + FKilo;
    else
      IplikAyKilo = IplikAyKilo + FKilo;
  end

  if (FLastMakinaRaporGrupId <> -2) then suspend;

  FLastMakinaRaporGrupId = -2;
  FLastTamirDuzeltmeTipiId = '@';

  for select max(DisTamir.TamirDuzeltmeTipiId), Makina.RaporGrupId, PartiStok.HamTipiId, PartiAsamaStokDetay.CikisTarihi, PartiAsamaStokDetay.Kilo from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
    where PartiAsamaStokDetay.Cikistarihi >= :AAylikBaslangic and PartiAsamaStokDetay.Cikistarihi < :AAylikBitis and
          Makina.UretimRaporlarinaDahil = 1 and Asama.GunlukUretimRaporunaDahil = 1 and
         (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
         (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null) and
         (Parti.IslemTipiId = 3)
    group by PartiAsamaStokDetay.PartiAsamaStokDetayId, Makina.RaporGrupId, PartiStok.HamTipiId, PartiAsamaStokDetay.CikisTarihi, PartiAsamaStokDetay.Kilo
    order by Makina.RaporGrupId, 1
    into :FTamirDuzeltmeTipiId, :FMakinaRaporGrupId, :FHamTipiId, :FCikisTarihi, :FKilo do
  begin
    if (FLastMakinaRaporGrupId <> FMakinaRaporGrupId or StrKarsilastir(FlastTamirDuzeltmeTipiId, FTamirDuzeltmeTipiId) = 0) then
    begin
      if (FLastMakinaRaporGrupId <> -2) then suspend;

      if (FMakinaRaporGrupId is null) then FMakinaRaporGrupId = -1;
      if (FTamirDuzeltmeTipiId is null) then FTamirDuzeltmeTipiId = '-1';

      MakinaRaporGrupId     = FMakinaRaporGrupId;
      TamirDuzeltmeTipiId   = FTamirDuzeltmeTipiId;
      IplikKilo             = 0;
      IplikAyKilo           = 0;
      OrmeKumasKilo         = 0;
      OrmeKumasAyKilo       = 0;
      DokumaKumasKilo       = 0;
      DokumaKumasAyKilo     = 0;

      FLastMakinaRaporGrupId     = FMakinaRaporGrupId;
      FLastTamirDuzeltmeTipiId   = FTamirDuzeltmeTipiId;
    end

    if (FCikisTarihi >= ABaslangicTarihi and FCikisTarihi < ABitisTarihi) then
    begin
      if (FHamTipiID <> 2 and FHamtipiID <> 0 and FHamtipiID <> 5) then
        OrmeKumasKilo = OrmeKumasKilo + FKilo;
      else if (FHamTipiID = 2) then
        DokumaKumasKilo = DokumaKumasKilo + FKilo;
      else
        IplikKilo = IplikKilo + FKilo;
    end

    if (FHamTipiID <> 2 and FHamtipiID <> 0 and FHamtipiID <> 5) then
      OrmeKumasAyKilo = OrmeKumasAyKilo + FKilo;
    else if (FHamTipiID = 2) then
      DokumaKumasAyKilo = DokumaKumasAyKilo + FKilo;
    else
      IplikAyKilo = IplikAyKilo + Fkilo;
  end

  if (FLastMakinaRaporGrupId <> -2) then suspend;

  MakinaRaporGrupId    = 1;
  TamirDuzeltmeTipiId  = 'Nüanslama';

  OrmeKumasKilo         = 0;
  OrmeKumasAyKilo       = 0;
  DokumaKumasKilo       = 0;
  DokumaKumasAyKilo     = 0;

  for select PartiStok.HamTipiId, PartiAsamaStokDetay.CikisTarihi, PartiAsamaStokDetay.Kilo from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join PartiAsamaIlaveSarf on PartiAsamaIlaveSarf.PartiAsamalariID = PartiAsamalari.PartiAsamalariId
    where PartiAsamaStokDetay.Cikistarihi >= :AAylikBaslangic and PartiAsamaStokDetay.Cikistarihi < :AAylikBitis and
          PartiAsamaIlaveSarf.PartiAsamaIlaveSarfId is not null
    into :FHamTipiId, :FCikisTarihi, :FKilo do
  begin
    if (FCikisTarihi >= ABaslangicTarihi and FCikisTarihi < ABitisTarihi) then
    begin
      if (FHamTipiID <> 2 and FHamtipiID <> 0 and FHamtipiID <> 5) then
        OrmeKumasKilo = OrmeKumasKilo + FKilo;
      else if (FHamTipiID = 2) then
        DokumaKumasKilo = DokumaKumasKilo + FKilo;
      else
        IplikKilo = IplikKilo + FKilo;
    end

    if (FHamTipiID <> 2 and FHamtipiID <> 0 and FHamtipiID <> 5) then
      OrmeKumasAyKilo = OrmeKumasAyKilo + FKilo;
    else if (FHamTipiID = 2) then
      DokumaKumasAyKilo = DokumaKumasAyKilo + FKilo;
    else
      IplikAyKilo = IplikAyKilo + FKilo;
  end
  suspend;
end
