CREATE PROCEDURE DESENVARYANTSABLONGETIR(
    ADESENID VARCHAR(20),
    ADESENVARYANTID INTEGER)
RETURNS (
    DESENVARYANTID INTEGER,
    DESENVARYANTSABLONID INTEGER,
    SABLONNO INTEGER,
    ACILIM VARCHAR(10),
    SABLONRENKID INTEGER,
    SABLONRENKADI VARCHAR(30),
    DEPOID VARCHAR(10),
    MESHID INTEGER,
    DOLULUKORANI DOUBLE PRECISION,
    EMULSIYONURUNID VARCHAR(20),
    EMULSIYONURUNADI VARCHAR(80))
AS
declare variable Acilim1               Double Precision;
declare variable Acilim2               Double Precision;
begin

  for select DesenVaryantSablon.DesenvaryantId, DesenVaryantSablon.DesenVaryantSablonId, DesenVaryantSablon.SablonNo, DesenVaryantSablon.Acilim1, DesenVaryantSablon.Acilim2,
             DesenVaryantSablon.DepoId, DesenVaryantSablon.SablonRenkId, SablonRenk.Adi, DesenVaryantSablon.MeshId, DesenVaryantSablon.DolulukOrani,
             DesenVaryantSablon.EmulsiyonUrunId, Urun.Adi from DesenVaryant
  left join DesenVaryantSablon on DesenVaryantSablon.DesenVaryantID = DesenVaryant.DesenVaryantId
  left join SablonRenk on SablonRenk.SablonRenkID = DesenVaryantSablon.SablonRenkId
  left join Urun on Urun.UrunId = DesenVaryantSablon.EmulsiyonUrunId
  where DesenVaryant.DesenID = :ADesenID and (DesenVaryant.DesenVaryantId = :ADesenVaryantId or :ADesenVaryantId is null) and
        DesenVaryantSablon.DesenVaryantID is not null
  order by DesenVaryantSablon.DesenVaryantID, DesenVaryantSablon.SablonNo
  into :DesenvaryantId, :DesenVaryantSablonId, :SablonNo, :Acilim1, :Acilim2, :DepoId, :SablonRenkId, :SablonRenkAdi, :MeshId, :DolulukOrani,
       :EmulsiyonUrunId, :EmulsiyonUrunAdi do
  begin
    Acilim = Format_Float('',Acilim1) || ' / ' || Format_Float('',Acilim2);
    suspend;
  end
end
