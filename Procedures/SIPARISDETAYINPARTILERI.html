CREATE PROCEDURE SIPARISDETAYINPARTILERI(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ISLEMGRUBUADI VARCHAR(20),
    ISLEMTIPIADI VARCHAR(20),
    ACILIYETDERECESIADI VARCHAR(20),
    PLANLANANIMALATTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
begin
  For Select DISTINCT BoyaSiparisDetay.BoyaSiparisDetayId, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.ReceteId,
             Recete.Adi, IslemGrubu.Adi, IslemTipi.Adi, AciliyetDerecesi.Adi, Parti.PlanlananImalatTarihi, Parti.TerminTarihi,
             Parti.Notu from BoyaSiparisDetay
    left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    where BoyaSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and PartiStok.PartiKayitId is not null
    into :BoyaSiparisDetayId, :PartiKayitId, :PartiYil, :PartiId, :AcilisTarihi, :ReceteId, :ReceteAdi, :IslemGrubuAdi, :IslemTipiAdi,
         :AciliyetDerecesiAdi, :PlanlananImalatTarihi, :TerminTarihi, :Notu do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    suspend;
  end
end
