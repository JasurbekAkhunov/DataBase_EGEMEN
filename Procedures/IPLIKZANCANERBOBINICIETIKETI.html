CREATE PROCEDURE IPLIKZANCANERBOBINICIETIKETI(
    AIPLIKLOTID INTEGER)
RETURNS (
    DEGER VARCHAR(100),
    SIRASI INTEGER)
AS
declare variable BELotSirasi                     integer;
declare variable BESiparisNoSirasi               integer;
declare variable BEFirmaAdiSirasi                integer;
declare variable BEMusteriSiparisNoSirasi        integer;
declare variable BEHamAdiSirasi                  integer;
declare variable BEHamAdi2Sirasi                 integer;
declare variable BEHamAdi3Sirasi                 integer;
declare variable BEMelanjSirasi                  integer;
declare variable BEFlamSirasi                    integer;
declare variable BEIplikCinsiSirasi              integer;
declare variable BEIplikCinsiDigerAdiSirasi      integer;
declare variable BEKalinlikSirasi                integer;
declare variable BEOzellikSirasi                 integer;
declare variable BEOzellikDigerAdiSirasi         integer;
declare variable BEElyafSinifSirasi              integer;
declare variable BEElyafTipiSirasi               integer;
declare variable BEBobinGramajiSirasi            integer;
declare variable BETarihSirasi                   integer;
declare variable BESaatSirasi                    integer;
declare variable BEKonikRengiSirasi              integer;
declare variable BEKonikCinsiSirasi              integer;
declare variable BELotNotuSirasi                 integer;
declare variable BELotNotu2Sirasi                 integer;

declare variable Lot  varchar(30);
declare variable BoyaSiparisYil  integer;
declare variable BoyaSiparisId   integer;
declare variable FirmaAdi        varchar(100);
declare variable MusteriSiparisNo varchar(50);
declare variable HamAdi           varchar(255);
declare variable HamAdi2          varchar(255);
declare variable HamAdi3          varchar(255);
declare variable MelanjKodu       varchar(20);
declare variable FlamKodu         varchar(20);
declare variable LotNotu          varchar(250);
declare variable LotNotu2         varchar(250);
declare variable IplikCinsiAdi      varchar(80);
declare variable IplikCinsiDigerAdi varchar(30);
declare variable KalinlikAdi      varchar(50);
declare variable OzellikAdi       varchar(40);
declare variable OzellikDigerAdi  varchar(40);
declare variable ElyafSinifAdi    varchar(20);
declare variable ElyafTipiAdi     varchar(30);
declare variable BobinGramaji     varchar(20);
declare variable KonikRengiAdi    varchar(20);
declare variable KonikCinsiAdi    varchar(20);
declare variable EtiketTipiId     integer;
declare variable EgemenFirmaId    integer;
begin
  Select EgemenFirmaId from sistem where sistemID = 1 into :EgemenFirmaId;

  select IplikLot.Lot, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Firma.Adi, coalesce(BoyaSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.MusteriSiparisNo),
         Ham.Adi, Ham.Adi2, Ham.Adi3, IplikLot.MelanjKodu, IplikLot.FlamKodu, IplikCinsi.Adi, Kalinlik.Adi, Ozellik.Adi, ElyafSinif.Adi,
         ElyafTipi.Adi, IplikLot.BobinGramaji, IplikLot.EtiketTipiId, KonikRengi.Adi, KonikCinsi.Adi, IplikLot.Notu, IplikLot.Notu2, IplikCinsi.DigerAdi, Ozellik.DigerAdi from IplikLot
    left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikLot.IplikTalepDetayID
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join Ham on Ham.HamId = IplikLot.HamId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
    left join Ozellik on Ozellik.OzellikId = Ham.OzellikId
    left join ElyafSinif on ElyafSinif.ElyafSinifId = Ham.ElyafSinifId
    left join ElyafTipi on ElyafTipi.ElyafTipiId = IplikLot.ElyafTipiId
    left join KonikRengi on KonikRengi.KonikRengiId = IplikLot.KonikRengiId
    left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikLot.KonikCinsiId
    where IplikLot.IplikLotId = :AIplikLotId
    into :Lot, :BoyaSiparisYil, :BoyaSiparisId, :FirmaAdi, :MusteriSiparisNo, :HamAdi, :HamAdi2, :HamAdi3, :MelanjKodu, :FlamKodu,
         :IplikCinsiAdi, :KalinlikAdi, :OzellikAdi, :ElyafSinifAdi, :ElyafTipiAdi, :BobinGramaji, :EtiketTipiId,
         :KonikRengiAdi, :KonikCinsiAdi, :LotNotu, :LotNotu2, :IplikCinsiDigerAdi, :OzellikDigerAdi;

  if (EtiketTipiID is null) then exit;

  Select BELotSirasi, BESiparisNoSirasi, BEFirmaAdiSirasi, BEMusteriSiparisNoSirasi, BEHamAdiSirasi, BEHamAdi2Sirasi, BEHamAdi3Sirasi, BEMelanjSirasi,
         BEFlamSirasi, BEIplikCinsiSirasi, BEKalinlikSirasi, BEOzellikSirasi, BEElyafSinifSirasi, BEElyafTipiSirasi, BEBobinGramajiSirasi, BETarihSirasi,
         BESaatSirasi, BEKonikRengiSirasi, BEKonikCinsiSirasi, BELotNotuSirasi, BELotNotu2Sirasi, BEIplikCinsiDigerAdiSirasi, BEOzellikDigerAdiSirasi from EtiketTipi
    where EtiketTipi.EtiketTipiID = :EtiketTipiId
    into :BELotSirasi, :BESiparisNoSirasi, :BEFirmaAdiSirasi, :BEMusteriSiparisNoSirasi, :BEHamAdiSirasi, :BEHamAdi2Sirasi, :BEHamAdi3Sirasi, :BEMelanjSirasi,
         :BEFlamSirasi, :BEIplikCinsiSirasi, :BEKalinlikSirasi, :BEOzellikSirasi, :BEElyafSinifSirasi, :BEElyafTipiSirasi, :BEBobinGramajiSirasi, :BETarihSirasi,
         :BESaatSirasi, :BEKonikRengiSirasi, :BEKonikCinsiSirasi, :BELotNotuSirasi, :BELotNotu2Sirasi, :BEIplikCinsiDigerAdiSirasi, :BEOzellikDigerAdiSirasi;


  if (BELotSirasi > 0) then
  begin
    Sirasi = BELotSirasi;
    if (EgemenFirmaId = 46) then Deger = SubString(Lot from 5 for Char_Length(Lot) - 4); /*Başyazıcı*/
    else Deger = Lot;
    suspend;
  end

  if (BESiparisNoSirasi > 0) then
  begin
    Sirasi = BESiparisNoSirasi;

    if (BoyaSiparisYil is not null) then
      Deger = EgemenIntegerToStr(BoyaSiparisYil)  || '-' || EgemenIntegerToStr(BoyaSiparisId);
    else
      Deger = '';
    suspend;
  end

  if (BEFirmaAdiSirasi > 0) then
  begin
    if (FirmaAdi is null) then FirmaAdi = '';

    Sirasi = BEFirmaAdiSirasi;
    Deger = SubString(FirmaAdi from 1 for 30);
    suspend;
  end


  if (BEMusteriSiparisNoSirasi > 0) then
  begin
    if (MusteriSiparisNo is null) then MusteriSiparisNo = '';

    Sirasi = BEMusteriSiparisNoSirasi;
    Deger = MusteriSiparisNo;
    suspend;
  end

  if (BEHamAdiSirasi > 0) then
  begin
    Sirasi = BEHamAdiSirasi;
    if (EgemenFirmaId = 4) then Deger = SubString(HamAdi from 1 for 40); /*Gülle*/
    else Deger = SubString(HamAdi from 1 for 24);
    suspend;
  end

  if (BEHamAdi2Sirasi > 0) then
  begin
    if (HamAdi2 is null) then HamAdi2 = '';

    Sirasi = BEHamAdi2Sirasi;
    Deger = SubString(HamAdi2 from 1 for 24);
    suspend;
  end

  if (BEHamAdi3Sirasi > 0) then
  begin
    if (HamAdi3 is null) then HamAdi3 = '';

    Sirasi = BEHamAdi3Sirasi;
    Deger = SubString(HamAdi3 from 1 for 24);
    suspend;
  end

  if (BEMelanjSirasi > 0) then
  begin
    if (MelanjKodu is null) then MelanjKodu = '';

    Sirasi = BEMelanjSirasi;
    Deger = MelanjKodu;
    suspend;
  end

  if (BEFlamSirasi > 0) then
  begin
    if (FlamKodu is null) then FlamKodu = '';

    Sirasi = BEFlamSirasi;
    Deger = FlamKodu;
    suspend;
  end


  if (BEIplikCinsiSirasi > 0) then
  begin
    if (IplikCinsiAdi is null) then IplikCinsiAdi = '';

    Sirasi = BEIplikCinsiSirasi;
    Deger = IplikCinsiAdi;
    suspend;
  end

  if (BEKalinlikSirasi > 0) then
  begin
    if (KalinlikAdi is null) then KalinlikAdi = '';

    Sirasi = BEKalinlikSirasi;
    Deger = KalinlikAdi;
    suspend;
  end

  if (BEOzellikSirasi > 0) then
  begin
    if (OzellikAdi is null) then OzellikAdi = '';

    Sirasi = BEOzellikSirasi;
    Deger = OzellikAdi;
    suspend;
  end

  if (BEOzellikDigerAdiSirasi > 0) then
  begin
    if (OzellikDigerAdi is null) then OzellikDigerAdi = '';
    Sirasi = BEOzellikDigerAdiSirasi;
    Deger = OzellikDigerAdi;
    suspend;
  end

  if (BEIplikCinsiDigerAdiSirasi > 0) then
  begin
    if (IplikCinsiDigerAdi is null) then IplikCinsiDigerAdi = '';
    Sirasi = BEIplikCinsiDigerAdiSirasi;
    Deger = IplikCinsiDigerAdi;
    suspend;
  end


  if (BEElyafSinifSirasi > 0) then
  begin
    if (ElyafSinifAdi is null) then ElyafSinifAdi = '';

    Sirasi = BEElyafSinifSirasi;
    Deger = ElyafSinifAdi;
    suspend;
  end

  if (BEElyafTipiSirasi > 0) then
  begin
    if (ElyafTipiAdi is null) then ElyafTipiAdi = '';

    Sirasi = BEElyafTipiSirasi;
    Deger = ElyafTipiAdi;
    suspend;
  end

  if (BEKonikRengiSirasi > 0) then
  begin
    if (KonikRengiAdi is null) then KonikRengiAdi = '';

    Sirasi = BEKonikRengiSirasi;
    Deger = KonikRengiAdi;
    suspend;
  end

  if (BEKonikCinsiSirasi > 0) then
  begin
    if (KonikCinsiAdi is null) then KonikCinsiAdi = '';

    Sirasi = BEKonikCinsiSirasi;
    Deger = KonikCinsiAdi;
    suspend;
  end

  if (BELotNotuSirasi > 0) then
  begin
    if (LotNotu is null) then LotNotu = '';

    Sirasi = BELotNotuSirasi;
    Deger = SubString(LotNotu from 1 for 24);
    suspend;
  end

  if (BELotNotu2Sirasi > 0) then
  begin
    if (LotNotu2 is null) then LotNotu2 = '';

    Sirasi = BELotNotu2Sirasi;
    Deger = SubString(LotNotu2 from 1 for 24);
    suspend;
  end


  if (BEBobinGramajiSirasi > 0) then
  begin
    if (BobinGramaji is null) then BobinGramaji = '';

    Sirasi = BEBobinGramajiSirasi;
    Deger = ElyafTipiAdi;
    suspend;
  end

  if (BETarihSirasi > 0) then
  begin
    Sirasi = BETarihSirasi;
    Deger = TimeStampToDateStr(Current_Date);
    suspend;
  end

  if (BESaatSirasi > 0) then
  begin
    Sirasi = BESaatSirasi;
    Deger = current_TimeStamp(0);
    Deger = SubString(Deger from 1 for 16);
    suspend;
  end
end
