CREATE PROCEDURE BOYATEKLIFKOPYALA(
    ABOYATEKLIFKAYITID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    BOYATEKLIFYIL INTEGER,
    BOYATEKLIFID INTEGER)
AS
declare variable YeniBoyaTeklifKayitId integer;
declare variable YeniBoyaTeklifDetayId integer;
declare variable FBoyaTeklifDetayId    integer;

declare variable hamid integer;
declare variable kullanimorani double precision;
declare variable AlisFiyati double precision;
declare variable fireorani double precision;
declare variable fasonboyabirimfiyat double precision;
declare variable renkid integer;
declare variable AlisfiyatiDovizId DovizSize;
declare variable TeklifDovizId DovizSize;
declare variable TeklifTarihi TimeStamp;
declare variable FDovizKuru Double Precision;
begin
  BoyaTeklifYil = f_Year(Current_date);
  BoyaTeklifId = Null;
  Select Id from BoyaTeklifNumaratoruAl(:BoyaTeklifYil) into :BoyaTeklifId;
  YeniBoyaTeklifKayitId = gen_id(BoyaTeklifKayit_Id_Gen, 1);
  /*BoyaTeklif Tablosu*/
  insert into BoyaTeklif (BoyaTeklifKayitId, BoyaTeklifYil, BoyaTeklifId, FirmaId, TeklifTarihi, Fason, DovizId,
                          Musterireferansi_1, Musterireferansi_2, Firmayetkiliid, kontroldurumu, kontrolkullaniciid,
                          kontroltarihi, onaydurumu, onaylayankullaniciid, onaytarihi, notu, teklifsuresi,
                          duzeltmefaktoru, uretimgideri, yonetimgideri, amortisman, vadefarkiorani, IslemKaynagi, linkislemi,
                          insertkullaniciid, OdemeSekliID)
  select :YeniBoyaTeklifKayitId, :BoyaTeklifYil, :BoyaTeklifId, firmaid, current_date, fason, dovizid, musterireferansi_1,
         musterireferansi_2, firmayetkiliid, kontroldurumu, kontrolkullaniciid, kontroltarihi, onaydurumu,
         onaylayankullaniciid, onaytarihi, notu, teklifsuresi, duzeltmefaktoru, uretimgideri, yonetimgideri, amortisman,
         vadefarkiorani, IslemKaynagi, 2, :AKullaniciId, OdemeSekliID
  from BoyaTeklif where BoyaTeklif.BoyaTeklifKayitId = :ABoyaTeklifKayitId;

  for select BoyaTeklifDetayId from BoyaTeklifDetay where BoyaTeklifKayitId = :ABoyaTeklifKayitId into :FBoyaTeklifDetayId do
  begin
    /*BoyaTeklifDetay tablosu*/
    YeniBoyaTeklifDetayId = gen_id(BoyaTeklifDetay_Id_Gen, 1);
    insert into BoyaTeklifDetay (boyateklifkayitid, boyateklifdetayid, fason, hamid, kaliteid, labreceteatisid,
                                 globalteklif, renkderinligiid, hesaplananfiyat, teklifedilenfiyat, recetemaliyeti, kilo,
                                 metre, adet, siparisprosesid, vadegun, uretimkarorani, satiskarorani, tekliffireorani,
                                 hamstokbedeli, teklifsuresi, ormebirimmaliyet, iplikfireorani, komisyon, navlun,
                                 fasonbirimfiyat, fasonfireorani, enistenen, gramajistenen, notu, linkislemi,
                                 insertkullaniciid, inserttarihi, ModelKayitId, urunid)
    select :YeniBoyaTeklifKayitId, :YeniBoyaTeklifDetayId, fason, hamid, kaliteid, labreceteatisid, globalteklif, renkderinligiid,
           hesaplananfiyat, teklifedilenfiyat, recetemaliyeti, kilo, metre, adet, siparisprosesid, vadegun, uretimkarorani,
           satiskarorani, tekliffireorani, hamstokbedeli, teklifsuresi, ormebirimmaliyet, iplikfireorani, komisyon, navlun,
           fasonbirimfiyat, fasonfireorani, enistenen, gramajistenen, notu, 2, :AKullaniciId, current_TimeStamp(0), ModelKayitId, urunid
    from BoyaTeklifDetay where BoyaTeklifDetay.BoyaTeklifDetayId = :FBoyaTeklifDetayId;

    /*BoyaTeklifAsama tablosu*/
    insert into BoyaTeklifAsama (boyateklifkayitid, boyateklifdetayid, asamaid, birimfiyat, insertkullaniciid, inserttarihi, linkislemi)
    select :YeniBoyaTeklifKayitId, :YeniBoyaTeklifDetayId, asamaid, birimfiyat, :AKullaniciId, current_TimeStamp(0), 2
    from BoyaTeklifAsama where BoyaTeklifAsama.BoyaTeklifDetayId = :FBoyaTeklifDetayId;

    /*BoyaTeklifHammadde tablosu*/
    select BoyaTeklif.DovizId, BoyaTeklif.TeklifTarihi from BoyaTeklif where BoyaTeklif.BoyaTeklifKayitId = :YeniBoyaTeklifKayitId into :TeklifDovizId, :TeklifTarihi;

    for select BoyaTeklifHamMadde.hamid, BoyaTeklifHamMadde.kullanimorani, Ham.AlisFiyati, Ham.AlisfiyatiDovizId, BoyaTeklifHamMadde.fireorani, BoyaTeklifHamMadde.fasonboyabirimfiyat, BoyaTeklifHamMadde.renkid
        from BoyaTeklifHamMadde
        left join Ham on Ham.HamId = BoyaTeklifHamMadde.HamId
        where BoyaTeklifHamMadde.BoyaTeklifDetayId = :FBoyaTeklifDetayId into :hamid, :kullanimorani, :AlisFiyati, :AlisfiyatiDovizId, :fireorani, :fasonboyabirimfiyat, :renkid do
    begin
      if (TeklifDovizId <> AlisfiyatiDovizId) then
      begin
        select DovizKuru from DovizKuruGetir(:AlisfiyatiDovizId, :TeklifDovizId, :TeklifTarihi) into :FDovizKuru;
        if (FDovizKuru = 0) then exception DovizKuruGirisiYapilmamis;
        AlisFiyati = AlisFiyati * FDovizKuru;

        Select Fiyat from FormataGoreFiyatYuvarla(:AlisFiyati, 'BOYA', 'BIRIMFIYAT') into :AlisFiyati;
      end

      insert into BoyaTeklifHamMadde (boyateklifkayitid, boyateklifdetayid, hamid, kullanimorani, birimfiyat, insertkullaniciid, inserttarihi, fireorani, fasonboyabirimfiyat, renkid, linkislemi)
                              values (:YeniBoyaTeklifKayitId, :YeniBoyaTeklifDetayId, :hamid, :kullanimorani, :AlisFiyati, :AKullaniciId, current_timestamp(0), :fireorani, :fasonboyabirimfiyat, :renkid, 2);
    end

    /*BoyaTeklifRenk tablosu*/
    insert into BoyaTeklifRenk (boyateklifkayitid, boyateklifdetayid, labreceteatisid, renkderinligiid,
                                miktar, birimfiyat, insertkullaniciid, inserttarihi, linkislemi)
    select :YeniBoyaTeklifKayitId, :YeniBoyaTeklifDetayId, labreceteatisid, renkderinligiid, miktar, birimfiyat,
           :AKullaniciId, current_TimeStamp(0), 2
    from BoyaTeklifRenk where BoyaTeklifRenk.BoyaTeklifDetayId = :FBoyaTeklifDetayId;

    /*BoyaTeklifOzelTest tablosu*/
    insert into BoyaTeklifOzelTest (boyateklifkayitid, boyateklifdetayid, ozeltestid, birimfiyat, insertkullaniciid, inserttarihi, linkislemi)
    select :YeniBoyaTeklifKayitId, :YeniBoyaTeklifDetayId, ozeltestid, birimfiyat, :AKullaniciId, current_TimeStamp(0), 2
    from BoyaTeklifOzelTest where BoyaTeklifOzelTest.BoyaTeklifDetayId = :FBoyaTeklifDetayId;

    /*BoyaTeklifIlaveIslem tablosu*/
    insert into BoyaTeklifIlaveIslem (boyateklifkayitid, boyateklifdetayid, ilaveislemlerid, birimfiyat, insertkullaniciid, inserttarihi, linkislemi)
    select :YeniBoyaTeklifKayitId, :YeniBoyaTeklifDetayId, ilaveislemlerid, birimfiyat, :AKullaniciID, current_TimeStamp(0), 2
    from BoyaTeklifIlaveIslem where BoyaTeklifIlaveIslem.BoyaTeklifDetayId = :FBoyaTeklifDetayId;
  end

  suspend;
end
