CREATE PROCEDURE IPLIKFATURALISTESIDETAYI(
    AHAREKETTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFATURANO FATURANOSIZE,
    ASINIF VARCHAR(1))
RETURNS (
    IPLIKIRSALIYEDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    IPLIKSTOKGIRISNO INTEGER,
    IPLIKDEPOID VARCHAR(10),
    HAMTIPIID INTEGER,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(30),
    KDV DOUBLE PRECISION,
    TEVKIFAT INTEGER,
    TEVKIFATSTR VARCHAR(3),
    KAPADEDI DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    NETMIKTAR DOUBLE PRECISION,
    FATURAMIKTAR DOUBLE PRECISION,
    BRUT INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKRENGIID INTEGER,
    KONIKRENGIADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    HARMANKAYITID INTEGER,
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    HARMANADI VARCHAR(50),
    HARMANCINSIID INTEGER,
    HARMANCINSIADI VARCHAR(50),
    IPLIKTALEPYIL INTEGER,
    IPLIKTALEPID INTEGER,
    IPLIKTALEPIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    BIRIMFIYAT DOUBLE PRECISION,
    BRUTTUTAR DOUBLE PRECISION,
    BIRIMISKONTOPARASAL DOUBLE PRECISION,
    BIRIMISKONTOORANI1 DOUBLE PRECISION,
    BIRIMISKONTOORANI2 DOUBLE PRECISION,
    BIRIMISKONTOTOPLAMI DOUBLE PRECISION,
    KDVHARICFIYAT DOUBLE PRECISION,
    NETBIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    IPLIKIRSALIYEID INTEGER,
    IPLIKFATURAID INTEGER,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    ARTIKELNO VARCHAR(20),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select IplikIrsaliyeDetay.IplikIrsaliyeDetayId, IplikIrsaliyeDetay.HamId, Ham.Adi, Ham.Adi2, Ham.HamTipiId,
             IplikIrsaliyeDetay.IplikStokGirisNo, IplikIrsaliyeDetay.IplikDepoId, IplikIrsaliyeDetay.RenkId, Renk.Adi, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.Lot,
             IplikIrsaliyeDetay.Kdv, IplikIrsaliyeDetay.Tevkifat, IplikIrsaliyeDetay.KapAdedi, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.NetMiktar, IplikIrsaliyeDetay.FaturaMiktar, IplikIrsaliyeDetay.Brut,
             IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, SiparisFirma.Adi,
             IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId, IplikTalepDetay.BoyaSiparisDetayId,
             BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, IplikIrsaliyeDetay.AmbalajId, Ambalaj.Adi,
             IplikIrsaliyeDetay.KonikRengiId, KonikRengi.Adi, IplikIrsaliyeDetay.KonikCinsiId, KonikCinsi.Adi,
             IplikIrsaliyeDetay.BirimFiyat, IplikIrsaliyeDetay.BirimIskontoParasal, IplikIrsaliyeDetay.BirimIskontoOrani1,
             IplikIrsaliyeDetay.BirimIskontoOrani2, IplikIrsaliyeDetay.BirimIskontoToplami,
             IplikIrsaliyeDetay.KdvHaricFiyat, IplikIrsaliyeDetay.NetBirimFiyat,
             IplikIrsaliyeDetay.IplikIrsaliyeId, IplikIrsaliyeDetay.IplikFaturaId,
             coalesce(BoyaSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.MusteriSiparisNo),
             coalesce(BoyaSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.ReferansSiparisNo),
             BoyaSiparisDetay.MusteriKaliteNo,
             OrmeSiparisDetay.ArtikelNo, IplikIrsaliyeDetay.Notu,
             IplikIrsaliyeDetay.HarmanKayitId, harman.HarmanYil, Harman.HarmanId, Harman.Adi, Harman.HarmanCinsiId, HarmanCinsi.Adi,
             IK.Adi, IplikIrsaliyeDetay.InsertTarihi, SK.Adi, IplikIrsaliyeDetay.DegisimTarihi from IplikFatura
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikFaturaID = IplikFatura.IplikFaturaId
    left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
    left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
    left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
    left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikIrsaliyeDetay.IplikTalepDetayId
    left join Ambalaj on Ambalaj.AmbalajId = IplikIrsaliyeDetay.AmbalajId
    left join KonikRengi on KonikRengi.KonikRengiId = IplikIrsaliyeDetay.KonikRengiId
    left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikIrsaliyeDetay.KonikCinsiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
    left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
    Left join Renk on Renk.RenkID = IplikIrsaliyeDetay.RenkId
    left join Harman on Harman.HarmanKayitId = IplikIrsaliyeDetay.HarmanKayitId
    left join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Harman.HarmanCinsiId
    left join Kullanici IK on IK.KullaniciId  = IplikIrsaliyeDetay.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = IplikIrsaliyeDetay.KullaniciId
    where IplikFatura.HareketTipi = :AHareketTipi And IplikIrsaliyeDetay.IplikFaturaId is not null and
          ( (IplikFatura.FaturaTarihi >= :ABaslangicTarihi and IplikFatura.FaturaTarihi <= :ABitisTarihi) or (IplikFatura.FaturaNo = :AFaturaNo and :AFaturaNo is not null)) and
          (IplikFatura.FirmaId = :AFirmaId or :AFirmaId is null) and
          (IplikFatura.Sinif = :ASinif or :ASinif is null)
  into :IplikIrsaliyeDetayId, :HamId, :HamAdi, :HamAdi2, :HamTipiId, :IplikStokGirisNo, :IplikDepoId, :RenkId, :RenkAdi, :MelanjKodu, :Lot, :Kdv,
       :Tevkifat, :KapAdedi, :Miktar, :NetMiktar, :FaturaMiktar, :Brut, :IplikTalepYil, :IplikTalepId, :BoyaSiparisYil, :BoyaSiparisId,  :SiparisFirmaId, :SiparisFirmaAdi,
       :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :AmbalajId, :AmbalajAdi,
       :KonikRengiId, :KonikRengiAdi, :KonikCinsiId, :KonikCinsiAdi,
       :BirimFiyat, :BirimIskontoParasal, :BirimIskontoOrani1, :BirimIskontoOrani2, :BirimIskontoToplami, :KdvHaricFiyat,
       :NetBirimFiyat, :IplikIrsaliyeId, :IplikFaturaId, :MusteriSiparisNo, :ReferansSiparisNo,
       :MusteriKaliteNo, :ArtikelNo, :Notu,
       :HarmanKayitId, :HarmanYil, :HarmanId, :HarmanAdi, :HarmanCinsiId, :HarmanCinsiAdi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    if (Tevkifat = 1) then TevkifatStr = 'Var';
    else TevkifatStr = '';

    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';
    else
      BoyaSiparisIdStr = '';

    if (IplikTalepYil is not null) then
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';
    else
      IplikTalepIdStr = '';

    if (HarmanYil is not null) then
      HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') '  || EgemenIntegerToStr(HarmanId)  || ' ';
    else
      HarmanIdStr = '';
    suspend;
  end
end
