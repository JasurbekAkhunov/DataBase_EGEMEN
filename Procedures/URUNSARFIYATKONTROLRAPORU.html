CREATE PROCEDURE URUNSARFIYATKONTROLRAPORU(
    ABASLANGICTARIH TIMESTAMP,
    ABITISTARIH TIMESTAMP,
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICURUNGRUP1ID VARCHAR(4),
    ABITISURUNGRUP1ID VARCHAR(4),
    ABASLANGICURUNGRUP2ID VARCHAR(4),
    ABITISURUNGRUP2ID VARCHAR(4),
    ABASLANGICURUNGRUP3ID VARCHAR(4),
    ABITISURUNGRUP3ID VARCHAR(4),
    ABASLANGICURUNGRUP4ID VARCHAR(4),
    ABITISURUNGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    APARTITIPIID INTEGER,
    AFIRMAID VARCHAR(20),
    AOTOMASYON INTEGER,
    AASAMAID INTEGER,
    ADEPOID VARCHAR(10),
    AIMALATTIPIID INTEGER,
    APARTIGRUPID VARCHAR(4))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTIKILO NUMERIC(12,3),
    DOZAJLAMAPARTINO VARCHAR(10),
    DOZAJLAMASARJNO INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTIASAMAILAVESARFID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    BIRIMID VARCHAR(2),
    PLANLANANMIKTAR DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    ICTAMIR INTEGER,
    ICTAMIRSTR VARCHAR(5),
    FISNO INTEGER,
    STOKID INTEGER,
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    STOKGRUPID VARCHAR(4),
    STOKGRUPADI VARCHAR(30),
    OTOMASYON INTEGER,
    STOKDUSUMTIPI VARCHAR(30),
    BIRIMFIYAT DOUBLE PRECISION,
    FISTARIHI TIMESTAMP,
    URETIMECIKISTARIHI TIMESTAMP,
    URUNTIPIID INTEGER,
    ISLEMTIPIID INTEGER,
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    IMALATTIPIID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    LABRECETERENKDERINLIGIADI VARCHAR(20),
    HAMADITOPLAMI VARCHAR(1024),
    MASRAFMERKEZIID INTEGER,
    MASRAFMERKEZIADI VARCHAR(30),
    DEPARTMANID INTEGER,
    DEPARTMANADI VARCHAR(30),
    URUNTIPIADI VARCHAR(20),
    URUNGRUP1ADI VARCHAR(30),
    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30),
    ARAVERMESURESI DOUBLE PRECISION,
    ARAVERMESURESISTR VARCHAR(20),
    IMALATSURESI DOUBLE PRECISION,
    IMALATSURESISTR VARCHAR(20),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    MAKINAID VARCHAR(100),
    ASAMAKILO NUMERIC(12,3),
    MIKTARTIPI INTEGER,
    MIKTARTIPISTR VARCHAR(5),
    KIMYASALTIPIID INTEGER,
    KIMYASALTIPIADI VARCHAR(20),
    PARTISINIFID INTEGER,
    DEFAULTBIRIM VARCHAR(2),
    DEFAULTBIRIMCARPAN DOUBLE PRECISION,
    DEFAULTBIRIMMIKTAR DOUBLE PRECISION,
    DEFAULTBIRIMFIYAT DOUBLE PRECISION,
    DEFAULTBIRIMFIYATTL DOUBLE PRECISION,
    DOVIZKURUTARIHI TIMESTAMP,
    DOVIZKURUMESAJ VARCHAR(30),
    DOVIZID VARCHAR(5),
    TARIH TIMESTAMP,
    REFERANSSTOKDETAYID INTEGER,
    REFERANSIRSFIRMAID VARCHAR(20),
    REFERANSIRSTARIHI TIMESTAMP,
    REFERANSIRSNO VARCHAR(40),
    PARTISTOKDURUMU INTEGER,
    PARTISTOKDURUMUSTR ADI20)
AS
declare variable Okey               integer;
declare variable FHamAdiToplami     varchar(255);
declare variable FPartiTamirId      integer;
declare variable ImalatSuresiDakika Double Precision;
declare variable DovizKuru          Double Precision;
declare variable StokTabanDovizId   varchar(5);
declare Variable FBirimCarpan       Double Precision;
begin
  Select StokTabanDovizId from Sistem where SistemId = 1 into :StokTabanDovizId;

  for select StokDetay.StokId, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiGrupId, PartiGrup.Adi, StokDetay.UrunId, Urun.Adi, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3, Urun.UrunGrup4,
             PartiAsamalari.AsamaId, Asama.Adi, PartiAsamalari.ReceteId, Recete.Adi, UBSD.BirimId, Stok.FisTarihi, StokDetay.UretimeCikisTarihi, Stok.FisNo,
             StokDetay.PlanlananMiktar, StokDetay.Miktar, StokDetay.PartiAsamalariId, StokDetay.PartiAsamaIlaveSarfID, StokDetay.Otomasyon, StokDetay.EasPartiNo, StokDetay.EasSarjNo,
             Parti.FirmaId, Firma.Adi, Parti.IslemTipiId, Parti.IslemGrubuId, IslemGrubu.Adi, Parti.ImalatTipiId, Urun.UrunTipiId, StokDetay.BirimFiyat * StokDetay.TabanDovizKuru2 / StokDetay.TabanDovizKuru1,
             stokdetay.StokGrupId, StokHareketGrup.Adi,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, RenkDerinligi.Adi, Stok.MasrafMerkeziId, MasrafMerkezi.Adi,
             UrunTipi.Adi, UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi,
             PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.Cikistarihi, PartiAsamaStokDetay.AraVermeSuresi, PartiAsamaStokDetay.MakinaId,
             PartiAsamaStokDetay.Kilo, Urun.KimyasalTipiId, KimyasalTipi.Adi, PartiAsamaStokDetay.PartiSinifId, Stok.DovizId, UBSD.Carpan, stokdetay.Tarih, Stok.DepartmanID, Departman.Adi, StokDetay.ReferansStokDetayID from stokdetay
    left join Urun on Urun.UrunId = StokDetay.UrunId
    left join UrunBirim UBSD on UBSD.UrunBirimId = StokDetay.UrunBirimId
    left join Parti on Parti.PartiKayitId = StokDetay.PartiKayitId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = StokDetay.PartiAsamalariId
    left join recete on Recete.ReceteId = PartiAsamalari.ReceteId
    left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = StokDetay.PartiAsamalariId
    left join Stok on Stok.StokId = StokDetay.StokId
    left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = stokdetay.StokGrupId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = Parti.BoyaSiparisKayitId
    left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
    left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
    left join MasrafMerkezi on MasrafMerkezi.MasrafMerkeziId = Stok.MasrafMerkeziId
    left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
    left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
    left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
    left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
    left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
    left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = urun.KimyasalTipiId
    left Join Departman on Departman.DepartmanID = Stok.DepartmanID
    where StokDetay.UretimeCikisTarihi >= :ABaslangicTarih and StokDetay.UretimeCikisTarihi <= :ABitisTarih and StokDetay.HareketTipi = 6 and
          (StokDetay.Otomasyon = :AOtomasyon or :AOtomasyon is null) and
          ((:AUrunAralikli = 1 and StokDetay.UrunId >= :ABaslangicUrunId and (StokDetay.UrunId <= :ABitisUrunId or :ABitisUrunId = '')) or (:AUrunAralikli = 0))  and
          ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicUrunGrup1Id  and (UrunGrup1 <= :ABitisUrunGrup1Id or :ABitisUrunGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicUrunGrup1Id = '' and :ABitisUrunGrup1Id = '')) and
          ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicUrunGrup2Id and (UrunGrup2 <= :ABitisUrunGrup2Id or :ABitisUrunGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicUrunGrup2Id = '' and :ABitisUrunGrup2Id = '')) and
          ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicUrunGrup3Id and (UrunGrup3 <= :ABitisUrunGrup3Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicUrunGrup3Id = '' and :ABitisUrunGrup3Id = '')) and
          ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicUrunGrup4Id and (UrunGrup4 <= :ABitisUrunGrup4Id or :ABitisUrunGrup4Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicUrunGrup4Id = '' and :ABitisUrunGrup4Id = '')) and
          (Parti.PartiTipiId = :APartiTipiId or :APartiTipiId is null) and
          (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
          (PartiAsamalari.AsamaId = :AAsamaId or Recete.UstAsamaId = :AAsamaId or :AAsamaId is null) and
          (StokDetay.DepoID = :ADepoID or :ADepoID is null) and
          (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null)and
          (PartiGrup.PartiGrupId = :APartiGrupId or :APartiGrupId is null)
    order by StokDetay.UretimeCikisTarihi, Stok.FisNo, StokDetay.PartiAsamalariId
    into :StokId, :PartiKayitId, :PartiYil, :PartiId, :PartiGrupId, :PartiGrupAdi, :UrunId, :UrunAdi, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4,
         :AsamaId, :AsamaAdi, :ReceteId, :ReceteAdi, :BirimId, :FisTarihi, :UretimeCikisTarihi, :FisNo,
         :PlanlananMiktar, :Miktar, :PartiAsamalariId, :PartiAsamaIlaveSarfID, :Otomasyon, :DozajlamaPartiNo, :DozajlamaSarjNo,
         :FirmaId, :FirmaAdi, :IslemTipiId, :IslemGrubuId, :IslemGrubuAdi, :ImalatTipiId, :UrunTipiId, :BirimFiyat,
         :StokGrupId, :StokGrupAdi, :BoyaSiparisYil, :BoyaSiparisId, :LabReceteRenkDerinligiAdi, :MasrafMerkeziId, :MasrafMerkeziAdi,
         :UrunTipiAdi, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi,
         :GirisTarihi, :Cikistarihi, :AraVermeSuresi, :MakinaId,
         :AsamaKilo, :KimyasalTipiId, :KimyasalTipiAdi, :PartiSinifId, :DovizId, :FBirimCarpan, :Tarih, :DepartmanID, :DepartmanAdi, :ReferansStokDetayID do
  begin
    select UrunBirim.Carpan, UrunBirim.BirimId from UrunBirim where UrunBirim.UrunId = :UrunId and UrunBirim.DefaultBirim = 1 into :DefaultBirimCarpan, :DefaultBirim;
    if (DefaultBirimCarpan is null or DefaultBirimCarpan = 0) then DefaultBirimCarpan = 1.0;
    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunGrup4Id, UrunGrup4) = 0) then okey = 0;

    ReferansIrsFirmaID = Null;
    ReferansIrsTarihi = Null;
    ReferansIrsNo = Null;

    if (ReferansStokDetayID is not Null) then
    begin
      Select First 1 Irsaliye.FirmaID, Irsaliye.IrsaliyeTarihi, Irsaliye.SeriNo || Irsaliye.IrsaliyeNo from StokDetay
        left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeID
        where StokDetay.StokDetayID = :ReferansStokDetayID
      into :ReferansIrsFirmaID, :ReferansIrsTarihi, :ReferansIrsNo;
    end

    IcTamir = 0;

    AraVermeSuresiStr = TarihtoStr(AraVermeSuresi);
    ImalatSuresi = CikisTarihi - GirisTarihi;
    ImalatSuresi = ImalatSuresi - AraVermeSuresi;
    ImalatSuresiDakika = ImalatSuresi * 1440;
    ImalatSuresiStr = TarihtoStr(ImalatSuresi);

    MiktarTipi = 0;
    MiktarTipiStr = Null;

    Select First 1 PartiStok.Durumu, PartiDurum.Adi FROM PartiStok LEFT JOIN PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
    WHERE PartiStok.PartiKayitID = :PartiKayitID AND PartiStok.PartiSinifID = :PartiSinifID
    into :PartiStokDurumu, :PartiStokDurumuStr;

    if (okey = 1) then
    begin
      if (Otomasyon = 0) then StokDusumTipi = 'Manuel';
      else if (Otomasyon = 1) then StokDusumTipi = 'Teorik';
      else if (Otomasyon = 2) then StokDusumTipi = 'Dozajlama Otomatik';
      else if (Otomasyon = 3) then StokDusumTipi = 'Dozajlama Teorik';
      else if (Otomasyon = 4) then StokDusumTipi = 'Dozajlama Manuel';

      if (PartiAsamaIlaveSarfID is not null) then StokDusumTipi = StokDusumTipi || ' (İlave)';

      if (PartiId is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      else PartiIdStr = '';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      PartiKilo = Null;
      HamAdiToplami = '';

      if (PartiKayitID is not null) then
      begin
        if (PartiAsamalariId is not null) then
        begin
          Select Max(PartiAsamalari.PartiTamirId), Sum(PartiStok.Kilo) from PartiAsamalari
            left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
            left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
            where PartiAsamalari.PartiAsamalariID = :PartiAsamalariID and PartiAsamaStok.AsamayaDahil = 1
            into :FPartiTamirId, :PartiKilo;

          if (FPartiTamirId is not null) then IcTamir = 1;

          for Select PartiStok.HamAdiToplami from PartiAsamalari
            left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
            left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
            where PartiAsamalari.PartiAsamalariID = :PartiAsamalariID and PartiAsamaStok.AsamayaDahil = 1
            group by PartiStok.HamAdiToplami
            into :FHamAdiToplami do
          begin
            if (Char_Length(HamAdiToplami) + Char_Length(FHamAdiToplami) < 1020) then
            begin
              if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set UTF8);
              HamAdiToplami = HamAdiToplami || FHamAdiToplami;
            end
          end
        end
        else
        begin
          Select Sum(PartiStok.Kilo) from Parti
            left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
            where Parti.PartiKayitId = :PartiKayitId
            into :PartiKilo;

          For Select PartiStok.HamAdiToplami from PartiStok
            where PartiStok.PartiKayitId = :PartiKayitId
            group By PartiStok.HamAdiToplami
            into :FHamAdiToplami do
          begin
            if (Char_Length(HamAdiToplami) + Char_Length(FHamAdiToplami) < 1020) then
            begin
              if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set UTF8);
              HamAdiToplami = HamAdiToplami || FHamAdiToplami;
            end
          end
        end
      end
      IcTamirStr = '';

      if (IcTamir = 1) then IcTamirStr = 'Evet';

      if (PartiAsamaIlaveSarfID is not Null) then
      begin
        Select first 1 PartiAsamaIlaveSarfDetay.MiktarTipi from PartiAsamaIlaveSarf
          left join PartiAsamaIlaveSarfDetay on PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfID = PartiAsamaIlaveSarf.PartiAsamaIlaveSarfId
          where PartiAsamaIlaveSarf.PartiAsamaIlaveSarfID = :PartiAsamaIlaveSarfID and PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfDetayId is not null and
          PartiAsamaIlaveSarfDetay.UrunId = :UrunId
          into :MiktarTipi;

        if (MiktarTipi = 0) then MiktarTipiStr = 'Gr';
        if (MiktarTipi = 1) then MiktarTipiStr = 'Gr/Kg';
        else if (MiktarTipi = 2) then MiktarTipiStr = 'Gr/Lt';
      end

      if (AsamaId is Null and ReceteId is not Null) then
      begin
        select first 1 Recete.UstAsamaId, Asama.Adi from Recete
          left join Asama on Asama.AsamaId = Recete.UstAsamaId
          where Recete.ReceteId = :ReceteId
          into :AsamaId, :AsamaAdi;
      end

      if (DovizId <> StokTabanDovizId) then
      begin
        select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :StokTabanDovizId, :Tarih) into :DovizKuru, :DovizKuruTarihi;
        if (Tarih <> DovizKuruTarihi) then DovizKuruMesaj = 'Fatura Tarihine Ait Kur Girisi Bulunamadigi icin ' || TimeStampToDateTimeStr(DovizKuruTarihi) || ' Tarihli Kur ile Hesaplandi';
      end
      else DovizKuru = 1;

      DefaultBirimMiktar = Miktar;
      DefaultBirimFiyat = BirimFiyat;

      if (FBirimCarpan <> DefaultBirimCarpan) then
      begin
         DefaultBirimMiktar = DefaultBirimMiktar / FBirimCarpan * DefaultBirimCarpan;
         DefaultBirimFiyat = DefaultBirimFiyat / FBirimCarpan * DefaultBirimCarpan;
      end
      DefaultBirimFiyatTL = DefaultBirimFiyat * DovizKuru;

      suspend;
    end
  end
