CREATE PROCEDURE SATINALMATALEPDETAYSARF(
    ASATINALMATALEPKAYITID INTEGER)
RETURNS (
    SATINALMATALEPDETAYID INTEGER,
    URUNID VARCHAR(20),
    YILTOPLAMTUKETILEN DOUBLE PRECISION,
    AYLIKORTALAMATUKETILEN DOUBLE PRECISION,
    GECENYILAYLIKORTALAMATUKETILEN DOUBLE PRECISION,
    GECENYILTOPLAMTUKETILEN DOUBLE PRECISION,
    GECENAYTUKETILEN DOUBLE PRECISION,
    BUAYTUKETILEN DOUBLE PRECISION,
    STOKMIKTARI DOUBLE PRECISION,
    DEPOSTOKMIKTARI DOUBLE PRECISION,
    SATINALMABIRIMDEPOID VARCHAR(10),
    GECENYILILKGUN TIMESTAMP,
    GECENYILSONGUN TIMESTAMP)
AS
declare variable VerilenSiparisTipiId   integer;
declare variable TalepTarihi            TimeStamp;
declare variable IlkGun                 TimeStamp;
declare variable SonGun                 TimeStamp;
declare variable AySayisi               integer;
declare variable FHareketTipi            integer;
declare variable FDepoId                varchar(10);
declare variable FTransferDepoId        varchar(10);
declare variable FMiktar                 Double Precision;
begin
  Select SatinAlmaTalep.VerilenSiparisTipiId, SatinAlmaTalep.Tarih, SatinAlmaBirim.DepoID from SatinAlmaTalep
    left join SatinAlmaBirim on SatinAlmaBirim.SatinAlmaBirimID = SatinAlmaTalep.SatinAlmaBirimID
    where SatinAlmaTalep.SatinAlmaTalepkayitId = :ASatinAlmaTalepkayitId
    into :VerilenSiparisTipiId, :TalepTarihi, :SatinAlmaBirimDepoID;

  AySayisi = f_Month(TalepTarihi) - 1;

  if (VerilenSiparisTipiId <> 3) then exit;

  For Select SatinAlmaTalepDetayGetir.SatinAlmaTalepDetayId, SatinAlmaTalepDetayGetir.UrunId FROM SatinAlmaTalepDetayGetir(:ASatinAlmaTalepKayitId) where SatinAlmaTalepDetayGetir.UrunID is not null
  into :SatinAlmaTalepDetayId, :UrunId do
  begin
    YilToplamTuketilen = 0;
    AylikOrtalamaTuketilen = 0;
    GecenAyTuketilen       = 0;
    GecenYilAylikOrtalamaTuketilen = 0;
    GecenYilToplamTuketilen = 0;
    BuAyTuketilen = 0;
    StokMiktari            = 0;


    IlkGun = TalepTarihi - f_DayOfYear(TalepTarihi) + 1;
    SonGun = TalepTarihi - f_DayOfMonth(TalepTarihi) + 1;

    Select sum(StokDetay.Miktar / UrunBirim.Carpan)   from stokdetay
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      where StokDetay.UrunId = :UrunId and StokDetay.Tarih >= :IlkGun and StokDetay.Tarih < :SonGun and
            StokDetay.HareketTipi = 6
      into :YilToplamTuketilen;

    AylikOrtalamaTuketilen = YilToplamTuketilen / AySayisi;

    SonGun = TalepTarihi - f_DayOfMonth(TalepTarihi);
    IlkGun = SonGun - f_DayOfMonth(SonGun)+1;

    SonGun = SonGun + 1;

    Select sum(StokDetay.Miktar / UrunBirim.Carpan)   from stokdetay
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      where StokDetay.UrunId = :UrunId and StokDetay.Tarih >= :IlkGun and StokDetay.Tarih < :SonGun and
            StokDetay.HareketTipi = 6
      into :GecenAyTuketilen;

    GecenYilSonGun = TalepTarihi - f_DayOfYear(TalepTarihi);
    GecenYilIlkGun = GecenYilSonGun - f_DayOfYear(GecenYilSonGun) + 1;

    Select sum(StokDetay.Miktar / UrunBirim.Carpan)   from stokdetay
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      where StokDetay.UrunId = :UrunId and StokDetay.Tarih >= :GecenYilIlkGun and StokDetay.Tarih < :GecenYilSonGun and
            StokDetay.HareketTipi = 6
      into :GecenYilToplamTuketilen;

    GecenYilAylikOrtalamaTuketilen = GecenYilToplamTuketilen / 12;

    IlkGun = TalepTarihi - f_DayOfMonth(TalepTarihi) + 1;
    SonGun = TalepTarihi;

    Select sum(StokDetay.Miktar / UrunBirim.Carpan)   from stokdetay
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      where StokDetay.UrunId = :UrunId and StokDetay.Tarih >= :IlkGun and StokDetay.Tarih <= :SonGun and
            StokDetay.HareketTipi = 6
      into :BuAyTuketilen;


    Select sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from stokdetay
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      where Stokdetay.Urunid = :UrunId and StokDetay.Tarih <= :TalepTarihi and StokDetay.HareketTipi <> 5
    into :StokMiktari;


    DepoStokMiktari = 0;

    if (SatinAlmaBirimDepoID is not null) then
    begin
      For Select StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId, sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from Urun
        left join stokdetay on stokdetay.UrunID = Urun.UrunId
        left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
        where Urun.Urunid = :UrunId and stokdetay.StokDetayID is not null and StokDetay.Tarih <= :TalepTarihi and
              (stokdetay.DepoId = :SatinAlmaBirimDepoID or StokDetay.TransferDepoId = :SatinAlmaBirimDepoID)
        group by StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId
      into :FHareketTipi, :FDepoId, :FTransferDepoId, :FMiktar do
      begin
        if (FHareketTipi = 1) then DepoStokMiktari = DepoStokMiktari + FMiktar;
        else if (FHareketTipi = 2) then DepoStokMiktari = DepoStokMiktari + FMiktar;
        else if (FHareketTipi = 3) then DepoStokMiktari = DepoStokMiktari + FMiktar;
        else if (FHareketTipi = 5 and SatinAlmaBirimDepoID = FTransferDepoId) then DepoStokMiktari = DepoStokMiktari - FMiktar;
        else if (FHareketTipi = 5 and SatinAlmaBirimDepoID = FDepoId) then DepoStokMiktari = DepoStokMiktari + FMiktar;
        else if (FHareketTipi = 4) then DepoStokMiktari = DepoStokMiktari + FMiktar;
        else if (FHareketTipi = 6) then DepoStokMiktari = DepoStokMiktari + FMiktar;
        else if (FHareketTipi = 7) then DepoStokMiktari = DepoStokMiktari + FMiktar;
      end
    end

    suspend;
  end
end
