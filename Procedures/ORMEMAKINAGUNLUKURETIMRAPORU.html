CREATE PROCEDURE ORMEMAKINAGUNLUKURETIMRAPORU(
    ATARIH TIMESTAMP,
    AVARDIYA INTEGER,
    AVARDIYAAMIRIPERSONELID INTEGER,
    APERSONELID INTEGER,
    AORMEMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4),
    AUNITEID INTEGER,
    AURETIMIOLMAYANLARDOKULSUNMU INTEGER)
RETURNS (
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    ORMEPLANLAMAGRUPID VARCHAR(10),
    BOYASIPARISKAYITID INTEGER,
    ORMEFISID INTEGER,
    HAMID INTEGER,
    STOKHAREKETID INTEGER,
    HAREKETTIPI INTEGER,
    URETIMISLEMTIPIID INTEGER,
    SINIF VARCHAR(1),
    TARIH TIMESTAMP,
    VARDIYA INTEGER,
    FISSERINO VARCHAR(2),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    SIPARISHAMRENKADI VARCHAR(50),
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    SIPARISSIRANO INTEGER,
    HAMADI VARCHAR(255),
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION,
    URETIMPARTINO VARCHAR(30),
    VARDIYAAMIRIPERSONELID INTEGER,
    VARDIYAAMIRIPERSONELADI VARCHAR(40),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    AYARTOPU INTEGER,
    AYARTOPUSTR VARCHAR(10),
    VADEGUN INTEGER,
    SIPARISPUS INTEGER,
    SIPARISFEINE INTEGER,
    MAKINAPUS INTEGER,
    MAKINAFEINE INTEGER,
    MAKINAMARKASI VARCHAR(30),
    MAKINATIPIADI VARCHAR(20),
    SIPARISMIKTARI DOUBLE PRECISION,
    SIPARISKALANMIKTAR DOUBLE PRECISION,
    TOPLAMORULEN DOUBLE PRECISION,
    TOPLAMSAKATORULEN DOUBLE PRECISION,
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    SIPARISTARIHI TIMESTAMP,
    SIPTUPACIKENMAYLI INTEGER,
    SIPTUPACIKENMAYLISTR VARCHAR(10),
    ORMESIPARISNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    MIKTARGUN DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    BINTURDAORULECEKKG DOUBLE PRECISION,
    PLANLANANTURSAYISI INTEGER,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    AYLIKKAPADEDI INTEGER,
    AYLIKMIKTAR DOUBLE PRECISION,
    AYLIKSAKATMIKTAR DOUBLE PRECISION,
    AYLIKSAKATKAPADEDI INTEGER)
AS
declare variable uitBukum integer;
declare variable uitOrme integer;
declare variable uitBoya integer;

declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable RaporVar integer;
declare variable FisSinif varchar(1);
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim;

  select uitBukum, uitOrme, uitBoya from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya;

  For Select OrmeMakina.OrmeMakinaId, OrmeMakina.Adi, OrmeMakina.Pus, OrmeMakina.Feine, OrmeMakina.Markasi, MakinaTipi.Adi from OrmeMakina
    left join MakinaTipi on MakinaTipi.MakinaTipiId = OrmeMakina.MakinaTipiId
    where (OrmeMakina.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
          (OrmeMakina.UniteId = :AUniteId or :AUniteId is null) and
          (OrmeMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null)
    into :OrmeMakinaId, :OrmeMakinaAdi, :MakinaPus, :MakinaFeine, :MakinaMarkasi, :MakinaTipiAdi do
  begin
    RaporVar = 0;
    BoyaSiparisKayitId  = Null;
    OrmeFisId             = Null;
    HamId                 = Null;
    StokHareketId         = Null;
    HareketTipi           = Null;
    Sinif                 = Null;
    Tarih                 = ATarih;
    Vardiya               = Null;
    FisSeriNo             = Null;
    FisNo                 = Null;
    FisTarihi             = Null;
    BoyaSiparisYil        = Null;
    BoyaSiparisId         = Null;
    BoyaSiparisIdStr      = Null;
    SiparisFirmaId        = Null;
    SiparisFirmaAdi       = Null;
    HamAdi                = Null;
    KapAdedi              = Null;
    Miktar                = Null;
    UretimPartiNo         = Null;
    VardiyaAmiriPersonelId      = Null;
    VardiyaAmiriPersonelAdi     = Null;
    PersonelId                  = Null;
    PersonelAdi                 = Null;
    VadeGun               = Null;
    InsertKullaniciAdi    = Null;
    InsertTarihi          = Null;
    KullaniciAdi          = Null;
    DegisimTarihi         = Null;
    UretimIslemTipiId     = Null;
    SiparisSiraNo         = Null;
    SiparisGrupId         = Null;
    SiparisGrupAdi        = Null;
    SiparisGrup2Id        = Null;
    SiparisGrup2Adi       = Null;
    MusteriSiparisNo      = Null;
    ReferansSiparisNo     = Null;
    SiparisHamRenkAdi     = Null;
    SiparisiAlanKullaniciId = Null;
    SiparisiAlanKullaniciAdi = Null;
    SiparisMiktari        = Null;
    SiparisKalanMiktar    = Null;
    ToplamOrulen          = Null;
    ToplamSakatOrulen     = Null;
    PlanlananMiktar       = Null;
    GerceklesenMiktar     = Null;
    SiparisTarihi         = Null;
    SipTupAcikEnMayli     = Null;
    SipTupAcikEnMayliStr  = Null;
    MiktarGun             = Null;
    OrmeSiparisNotu       = NUll;
    Sakat = Null;
    SakatStr = '';
    AyarTopu = Null;
    AyarTopuStr = '';
    OrmePlanlamaGrupId = Null;
    BinTurdaOrulecekKg = null;
    PlanlananTurSayisi = Null;
    AylikKapAdedi         = Null;
    AylikMiktar           = Null;
    AylikSakatMiktar      = Null;
    AylikSakatKapAdedi    = Null;

    Select Min(OrmePlanlamaGrupId) from OrmePlanlamaGrupDetayi where OrmePlanlamaGrupDetayi.OrmeMakinaID = :OrmeMakinaID into :OrmePlanlamaGrupId;

    For Select StokHareket.StokHareketId, StokHareket.OrmeFisId, StokHareket.HareketTipi, StokHareket.Tarih, OrmeFis.SeriNo, OrmeFis.FisNo, OrmeFis.FisTarihi,
               OrmeFis.Vardiya, OrmeFis.VardiyaAmiriPersonelId, VAPersonel.Adi,
               StokHareket.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               StokHareket.HamId, Ham.Adi, StokHareket.KapAdedi, StokHareket.Miktar,
               StokHareket.UretimPartiNo, StokHareket.PersonelId, P.Adi, StokHareket.VadeGun, StokHareket.InsertTarihi,
               StokHareket.DegisimTarihi, IK.Adi, SK.Adi, OrmeFis.Sinif, BoyaSiparis.FirmaId, SiparisFirma.Adi,
               StokHareket.UretimIslemTipiId, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, OrmeSiparisDetay.SiparisSiraNo,
               OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmeSiparisDetay.HamMiktar,
               OrmeSiparisDetay.ToplamOrulen, OrmeSiparisDetay.ToplamSakatOrulen, OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.ToplamOrulen + OrmeSiparisDetay.ToplamSakatOrulen,
               OrmePlanlama.PlanlananMiktar, OrmePlanlama.GerceklesenMiktar, BoyaSiparis.SiparisTarihi,
               OrmeSiparisDetay.TupAcikEnMayli, OrmePlanlama.MiktarGun, OrmeSiparisDetay.Notu, StokHareket.Sakat, StokHareket.AyarTopu,
               BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, OrmePlanlama.BinTurdaOrulecekKg, OrmePlanlama.PlanlananTurSayisi,
               OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4
               from StokHareket
      left join Ham on Ham.HamId = StokHareket.HamId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
      left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
      left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
      left join OrmeMakina on OrmeMakina.OrmeMakinaId = StokHareket.OrmeMakinaId
      left join Personel VAPersonel on VAPersonel.PersonelId  = OrmeFis.VardiyaAmiriPersonelId
      left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId  = StokHareket.OrmePlanlamaId
      left join Personel P on P.PersonelId  = StokHareket.PersonelId
      left join Kullanici IK on IK.KullaniciId  = StokHareket.InsertKullaniciId
      left join Kullanici SK on SK.KullaniciId  = StokHareket.KullaniciId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
      Left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
      where StokHareket.Tarih = :ATarih and
           (StokHareket.HareketTipi = :ShtUretim and StokHareket.HamIade = 0) and StokHareket.OrmeMakinaId = :OrmeMakinaId and
           (OrmeFis.Vardiya = :AVardiya or :Avardiya is null) and
           (OrmeFis.VardiyaAmiriPersonelId = :AVardiyaAmiriPersonelId or :AVardiyaAmiriPersonelId is null) and
           (StokHareket.PersonelId = :APersonelId or :APersonelId is null)
    into :StokHareketId, :OrmeFisId, :HareketTipi, :Tarih, :FisSeriNo, :FisNo, :FisTarihi, :Vardiya,
         :VardiyaAmiriPersonelId, :VardiyaAmiriPersonelAdi, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :HamId,
         :HamAdi, :KapAdedi, :Miktar, :UretimPartiNo, :PersonelId, :PersonelAdi, :VadeGun, :InsertTarihi, :DegisimTarihi,
         :InsertKullaniciAdi, :KullaniciAdi, :FisSinif, :SiparisFirmaId, :SiparisFirmaAdi, :UretimIslemTipiId,
         :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisSiraNo, :SiparisPus, :SiparisFeine, :SiparisMiktari, :ToplamOrulen, :ToplamSakatOrulen, :SiparisKalanMiktar,
         :PlanlananMiktar, :GerceklesenMiktar, :SiparisTarihi, :SipTupAcikEnMayli, :MiktarGun, :OrmeSiparisNotu, :Sakat, :AyarTopu,
         :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi, :MusteriSiparisNo, :ReferansSiparisNo, :BinTurdaOrulecekKg, :PlanlananTurSayisi,
         :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4 do
    begin
      if (FisSinif is not null) then Sinif = FisSinif;

      BoyaSiparisIdStr = '';
      if (BoyaSiparisYil is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      RaporVar = 1;

      Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:SipTupAcikenMayli) into :SipTupAcikEnMayliStr;

      if (Sakat = 1) then SakatStr = 'Sakat';
      else SakatStr = '';

      if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
      else AyarTopuStr = '';
            /*ay toplamları ıcın eklendı omer */
     Select Sum(StokHareket.KapAdedi), Sum(StokHareket.Miktar) from StokHareket
     left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
          where StokHareket.Tarih >= TimeStampToAyIlkGun(:ATarih) and  StokHareket.Tarih <= :ATarih and
           (StokHareket.HareketTipi = :shtUretim and StokHareket.HamIade = 0) and StokHareket.OrmeMakinaId = :OrmeMakinaId and
           (OrmeFis.Vardiya = :AVardiya or :Avardiya is null) and
           (OrmeFis.VardiyaAmiriPersonelId = :AVardiyaAmiriPersonelId or :AVardiyaAmiriPersonelId is null) and
           (StokHareket.PersonelId = :APersonelId or :APersonelId is null)  and StokHareket.Sakat = 0 into
          :AylikKapAdedi, :AylikMiktar ;

     Select Sum(StokHareket.KapAdedi), Sum(StokHareket.Miktar) from StokHareket
     left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
          where StokHareket.Tarih >= TimeStampToAyIlkGun(:ATarih) and  StokHareket.Tarih <= :ATarih and
           (StokHareket.HareketTipi = :shtUretim and StokHareket.HamIade = 0) and StokHareket.OrmeMakinaId = :OrmeMakinaId and
           (OrmeFis.Vardiya = :AVardiya or :Avardiya is null) and
           (OrmeFis.VardiyaAmiriPersonelId = :AVardiyaAmiriPersonelId or :AVardiyaAmiriPersonelId is null) and
           (StokHareket.PersonelId = :APersonelId or :APersonelId is null)  and StokHareket.Sakat = 1 into
          :AylikSakatKapAdedi, :AylikSakatMiktar ;
      suspend;
    end

      /*ay toplamları ıcın eklendı omer */
  if (RaporVar = 0) then
     begin
     Select Sum(StokHareket.KapAdedi), Sum(StokHareket.Miktar) from StokHareket
     left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
          where StokHareket.Tarih >= TimeStampToAyIlkGun(:ATarih) and  StokHareket.Tarih <= :ATarih and
           (StokHareket.HareketTipi = :shtUretim and StokHareket.HamIade = 0) and StokHareket.OrmeMakinaId = :OrmeMakinaId and
           (OrmeFis.Vardiya = :AVardiya or :Avardiya is null) and
           (OrmeFis.VardiyaAmiriPersonelId = :AVardiyaAmiriPersonelId or :AVardiyaAmiriPersonelId is null) and
           (StokHareket.PersonelId = :APersonelId or :APersonelId is null)  and StokHareket.Sakat = 0 into
          :AylikKapAdedi, :AylikMiktar ;

     Select Sum(StokHareket.KapAdedi), Sum(StokHareket.Miktar) from StokHareket
     left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
          where StokHareket.Tarih >= TimeStampToAyIlkGun(:ATarih) and  StokHareket.Tarih <= :ATarih and
           (StokHareket.HareketTipi = :shtUretim and StokHareket.HamIade = 0) and StokHareket.OrmeMakinaId = :OrmeMakinaId and
           (OrmeFis.Vardiya = :AVardiya or :Avardiya is null) and
           (OrmeFis.VardiyaAmiriPersonelId = :AVardiyaAmiriPersonelId or :AVardiyaAmiriPersonelId is null) and
           (StokHareket.PersonelId = :APersonelId or :APersonelId is null)  and StokHareket.Sakat = 1 into
          :AylikSakatKapAdedi, :AylikSakatMiktar ;
  end
    --if (AylikMiktar > 0 and AylikSakatMiktar > 0)  then  RaporVar = 0;

    if (RaporVar = 0 and AUretimiOlmayanlarDokulsunmu = 1) then suspend;
  end
end
