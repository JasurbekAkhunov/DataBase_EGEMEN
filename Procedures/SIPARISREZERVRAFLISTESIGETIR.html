CREATE PROCEDURE SIPARISREZERVRAFLISTESIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISREZERVID INTEGER,
    RAFDETAYID INTEGER,
    HAMTIPIID INTEGER,
    PARTISTOKIPTALKAYITID INTEGER,
    BOBINGRAMAJI DOUBLE PRECISION,
    RAFID VARCHAR(10),
    LOT VARCHAR(150),
    GRM2 INTEGER,
    KALANKILO NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    AKTARMA INTEGER,
    AKTARMASTR VARCHAR(15),
    ACIKLAMA VARCHAR(30),
    KALITENO INTEGER,
    YARIMAMUL INTEGER,
    YARIMAMULSTR VARCHAR(10),
    KALANKAPADEDI INTEGER,
    KALANADET INTEGER)
AS
begin
  For select BoyaSiparisRezerv.BoyaSiparisRezervId, Ham.HamTipiId, RafDetay.RafDetayId, RafDetay.RafId,
             HamStokMaster.Lot, HamStokMaster.Grm2, RafDetay.PartiStokIptalKayitId, RafDetay.KaliteNo,
             (RafDetay.Miktar + RafDetay.AktarmadanIadeEdienKilo - RafDetay.IadeEdilecekMiktar - RafDetay.IadeEdilenMiktar - RafDetay.AktarmayaCikanKilo - RafDetay.PartilenenMiktar - RafDetay.RafTransferMiktar - RafDetay.SatisMiktar),
             (RafDetay.Kilo + RafDetay.AktarmadanIadeEdienKilo - RafDetay.IadeEdilecekKilo - RafDetay.IadeEdilenKilo - RafDetay.AktarmayaCikanKilo - RafDetay.PartilenenKilo - RafDetay.RafTransferKilo - RafDetay.SatisKilo),
              RafDetay.Aktarma, HamStokMaster.BobinGramaji, Raf.YariMamul,
              (RafDetay.KapAdedi - RafDetay.IadeEdilecekKapAdedi - RafDetay.IadeEdilenKapAdedi - RafDetay.RafTransferKapAdedi - RafDetay.PartilenenKapAdedi),
              (RafDetay.Adet - RafDetay.IadeEdilecekAdet - RafDetay.IadeEdilenAdet - RafDetay.PartilenenAdet) from BoyaSiparisRezerv
   left Join RafDetay on RafDetay.HamStokDetayID = BoyaSiparisRezerv.HamStokDetayId
   left Join Raf on Raf.RafID = RafDetay.RafId
   left Join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
   left Join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
   left Join Ham on Ham.HamID = RafDetay.HamId
   left join BoyaSiparisDetayOngoru on BoyaSiparisDetayOngoru.BoyaSiparisDetayId = BoyaSiparisRezerv.BoyaSiparisDetayId and BoyaSiparisDetayOngoru.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
   where BoyaSiparisRezerv.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisDetayOngoru.OnayDurumu = 1 and BoyaSiparisRezerv.RezervVar = 1 and
        (RafDetay.Miktar + AktarmadanIadeEdienKilo - RafDetay.IadeEdilecekMiktar - RafDetay.IadeEdilenMiktar - RafDetay.AktarmayaCikanKilo - RafDetay.PartilenenMiktar - RafDetay.RafTransferMiktar - RafDetay.SatisMiktar) > 0 and
         Raf.PartilemeYapilabilir = 1
   order by BoyaSiparisRezerv.BoyaSiparisDetayId, BoyaSiparisRezerv.HamStokDetayId
   into :BoyaSiparisRezervId, :HamTipiId, :RafDetayId, :RafId, :Lot, :Grm2, :PartiStokIptalKayitId, :KaliteNo,
        :KalanMiktar, :KalanKilo, :Aktarma, :BobinGramaji, :YariMamul, :KalanKapAdedi, :KalanAdet do
  begin
    if (Aktarma = 1) then AktarmaStr = 'Ham Aktarma';
    else AktarmaStr = '';

    if (PartiStokIptalKayitId is not null) then Aciklama = 'Parti İptal';
    else Aciklama = '';

    YariMamulStr = Null;
    if (YariMamul = 1) then YariMamulStr = 'Yarı Mamul';

    suspend;
  end
end
