CREATE PROCEDURE DOZAJLAMAISEMRITARTIGETIR(
    ADOZAJLAMAISEMRIID INTEGER)
RETURNS (
    DOZAJLAMAISEMRIID INTEGER,
    DOZAJLAMAISEMRITARTIID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNRAF VARCHAR(15),
    URUNTIPIID TYPE OF COLUMN URUN.URUNTIPIID,
    AMBALAJSEKLIADI TYPE OF COLUMN AMBALAJSEKLI.ADI,
    AMBALAJAGIRLIGI DOUBLE PRECISION,
    BIRIMID VARCHAR(2),
    RECETEMIKTAR DOUBLE PRECISION,
    PLANLANANMIKTAR DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    BARKOD VARCHAR(255),
    ILAVE EGEBOOLEAN,
    ILAVESTR VARCHAR(5),
    BIRIMMIKTAR DOUBLE PRECISION,
    MIKTARTIPISTR VARCHAR(10),
    SABLONNO INTEGER,
    STOKTANDUSULDU SMALLINT,
    STOKTANDUSULDUSTR VARCHAR(5),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    BOYATIPIID INTEGER,
    URUNADI2 VARCHAR(80),
    RENKADI VARCHAR(30),
    BIDONNO INTEGER,
    BDRCTP VARCHAR(3),
    BDGINO INTEGER,
    BDPANO INTEGER)
AS
declare variable MiktarTipi          Integer;
declare variable FAmbalajSekliId     Integer;
declare variable FAmbalajId          Integer;
declare variable LastUrunTipiId      Integer;
begin
  BDGINO = 0;
  BDPANO = 0;
  LastUrunTipiId = -1;

  For Select DozajlamaIsEmriTarti.DozajlamaIsEmriID, DozajlamaIsEmriTarti.DozajlamaIsEmriTartiId, DozajlamaIsEmriTarti.UrunId, Urun.Adi, UrunBirim.BirimId,
             DozajlamaIsEmriTarti.ReceteMiktar, DozajlamaIsEmriTarti.PlanlananMiktar, DozajlamaIsEmriTarti.Miktar,
             Ik.Adi, DozajlamaIsEmriTarti.InsertTarihi, Sk.Adi, DozajlamaIsEmriTarti.DegisimTarihi, Urun.Barkod, Urun.Raf, DozajlamaIsEmriTarti.Ilave,
             DozajlamaIsEmriTarti.BirimMiktar, DozajlamaIsEmriTarti.MiktarTipi, DozajlamaIsEmriTarti.SablonNo, PartiSablonIlave.StoktanDusuldu, Urun.BoyaTipiId, Urun.Adi2,
             SablonRenk.Adi, DozajlamaIsEmriTarti.BidonNo, Urun.AmbalajSekliId, AmbalajSekli.Adi, Urun.UrunTipiId FROM DozajlamaIsEmriTarti
    Left Join Urun ON Urun.UrunID = DozajlamaIsEmriTarti.UrunID
    Left Join UrunBirim ON UrunBirim.UrunBirimID = DozajlamaIsEmriTarti.UrunBirimID
    Left Join AmbalajSekli ON AmbalajSekli.AmbalajSekliId = Urun.AmbalajSekliId
    Left Join Kullanici IK on IK.KullaniciId = DozajlamaIsEmriTarti.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = DozajlamaIsEmriTarti.KullaniciId
    Left Join PartiSablonIlave on PartiSablonIlave.PartiSablonIlaveId = DozajlamaIsEmriTarti.PartiSablonIlaveId
    Left Join DozajlamaIsEmri on DozajlamaIsEmri.DozajlamaIsEmriId = DozajlamaIsEmriTarti.DozajlamaIsEmriId
    Left Join Parti on Parti.PartiKayitId = DozajlamaIsEmri.PartiKayitId
    Left Join PartiSablonIsEmri on PartiSablonIsEmri.PartiSablonIsEmriId = DozajlamaIsEmri.PartiSablonIsEmriId
    Left Join DesenVaryantSablon on (DesenVaryantSablon.DesenVaryantID = PartiSablonIsEmri.DesenVaryantID or DesenVaryantSablon.DesenVaryantID = Parti.DesenVaryantID)  and DesenVaryantSablon.SablonNo = DozajlamaIsEmriTarti.SablonNo
    Left Join SablonRenk on SablonRenk.SablonRenkID = DesenVaryantSablon.SablonRenkId
    where DozajlamaIsEmriTarti.DozajlamaIsEmriID = :ADozajlamaIsEmriID
    ORDER BY DozajlamaIsEmriTarti.SablonNo, DozajlamaIsEmriTarti.DozajlamaIsEmriTartiID
    into :DozajlamaIsEmriID, :DozajlamaIsEmriTartiId, :UrunId, :UrunAdi, :BirimId, :ReceteMiktar,  :PlanlananMiktar, :Miktar,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :Barkod, :UrunRaf, :Ilave, :BirimMiktar, :MiktarTipi, :SablonNo, :StoktanDusuldu, :BoyaTipiId, :UrunAdi2,
         :RenkAdi, :BidonNo, :FAmbalajSekliId, :AmbalajSekliAdi, :UrunTipiId do
  begin

    AmbalajAgirligi = 0;
    BDPANO = 0;
    FAmbalajId      = Null;
    Select First 1 AmbalajId from Ambalaj Where Ambalaj.AmbalajSekliId = :FAmbalajSekliId into :FAmbalajId;
    select AmbalajAgirligi from AmbalajAgirligiHesapla(:FAmbalajId, 1) into :AmbalajAgirligi;

    IlaveStr = '';
    if (Ilave = 1) then
      IlaveStr = 'Evet';

    StoktanDusulduStr = '';
    if (StoktanDusuldu = 1) then
      StoktanDusulduStr = 'Evet';

    if (MiktarTipi = 0) then MiktarTipiStr = 'Gr';
    else if (MiktarTipi = 1) then MiktarTipiStr = 'Gr/Kg';
    else if (MiktarTipi = 2) then MiktarTipiStr = 'Gr/Lt';

    if (UrunTipiId = 0) then BDRCTP = 'BOY'; else BDRCTP = 'KIM';

    if (LastUrunTipiId <> UrunTipiId) then
    begin
        BDPANO = 0;
        BDGINO = BDGINO + 1;
    end

    BDPANO = BDPANO + 1;

    suspend;
  end
end
