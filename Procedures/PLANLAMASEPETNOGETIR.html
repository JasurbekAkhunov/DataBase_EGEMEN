CREATE PROCEDURE PLANLAMASEPETNOGETIR(
    APARTIKAYITID INTEGER,
    APARTIASAMALARIID INTEGER,
    APARTISINIFID INTEGER)
RETURNS (
    PLANLAMASEPETNO INTEGER,
    PLANLANDIGIMAKINAID VARCHAR(10),
    PLANLANDIGIMAKINARAPORGRUPID INTEGER)
AS
declare variable FSepetNo         integer;
begin
  PlanlamaSepetNo = Null;
  PlanlandigiMakinaId = Null;
  PlanlandigiMakinaRaporGrupId = Null;
  Select UretimPlanlama.MakinaId, UretimPlanlama.SepetNo from BolumPlanlama
    left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
    left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
    where BolumPlanlama.PartiAsamalariId = :APartiAsamalariId and BolumPlanlama.PartiSinifId = :APartiSinifId
    into :PlanlandigiMakinaId, :FSepetNo;

  if (FSepetNo is not null) then
  begin
    Select count(*) from UretimPlanlama
      where UretimPlanlama.MakinaID = :PlanlandigiMakinaId and UretimPlanlama.SepetNo < :FSepetNo
      into :PlanlamaSepetNo;

    if (PlanlamaSepetNo is null) then PlanlamaSepetNo = 0;
    PlanlamaSepetNo = PlanlamaSepetNo + 1;
  end
  else
  begin
    Select Max(PartiAsamaStokDetay.MakinaId) from PartiAsamalari
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariID = PartiAsamalari.PartiAsamalariId
      where PartiAsamalari.PartiKayitID = :APartiKayitId and PartiAsamalari.PartiAsamalariID = :APartiAsamalariId
      into :PlanlandigiMakinaId;
  end

  if (PlanlandigiMakinaId is not null) then
     Select RaporGrupID from Makina where Makina.MakinaID = :PlanlandigiMakinaId into :PlanlandigiMakinaRaporGrupId;
  suspend;
end
