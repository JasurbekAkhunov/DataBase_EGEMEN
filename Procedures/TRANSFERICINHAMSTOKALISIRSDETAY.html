CREATE PROCEDURE TRANSFERICINHAMSTOKALISIRSDETAY(
    AHAMSTOKIRSALIYEID INTEGER)
RETURNS (
    HAMSTOKMASTERID INTEGER,
    HAMTIPIID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATTL DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    TRANSFERKODU VARCHAR(55),
    MUHHESAPKODU VARCHAR(55),
    VADEGUN INTEGER,
    DOVIZKURU DOUBLE PRECISION,
    KDVORANI DOUBLE PRECISION)
AS
declare variable IrsaliyeTarihi     TimeStamp;
declare variable EgemenFirmaID integer;
declare variable SistemDovizId  varchar(5);
declare variable FYariMamulTransferKodu varchar(55);
declare variable FGirisTipi integer;
declare variable FFason integer;
declare variable AlisTransferKodu varchar(55);
declare variable FasonTransferKodu varchar(55);
declare variable MuhAlisHesapKodu varchar(55);
begin
  Select SistemDovizId, EgemenFirmaID from Sistem where sistemId = 1 into :SistemDovizId, :EgemenFirmaID; /* YTL*/
  Select IrsaliyeTarihi, DovizId, GirisTipi from HamStokIrsaliye where HamStokIrsaliye.HamStokIrsaliyeID = :AHamStokIrsaliyeID into :IrsaliyeTarihi, :DovizId, :FGirisTipi;

  For select HamStokMaster.HamStokMasterId, Ham.HamTipiId, HamStokMaster.HamId, HamStokMaster.Miktar, HamStokMaster.NetBirimFiyat,
             Ham.HamSatisTransferKodu, Ham.YariMamulTransferKodu, Ham.Adi, VerilenSiparisDetay.VadeGun, Ham.KdvOrani,
             HamStokMaster.Fason, Ham.AlisTransferKodu, Ham.FasonTransferKodu, Ham.MuhAlisHesapKodu from HamStokMaster
    left join Ham on Ham.HamId = HamStokMaster.HamId
    left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = HamStokMaster.VerilenSiparisDetayId
    where HamStokMaster.HamStokIrsaliyeId = :AHamStokIrsaliyeId and HamStokMaster.HamStokMasterId is not null
    order by HamStokMaster.HamStokMasterId
    into :HamStokMasterId, :HamTipiId, :HamId, :Miktar, :BirimFiyat, :TransferKodu, :FYariMamulTransferKodu, :HamAdi, :VadeGun,
         :KdvOrani, :FFason, :AlisTransferKodu, :FasonTransferKodu, :MuhAlisHesapKodu do
  begin
    if (DovizID <> SistemDovizId) then Select DovizKuru from DovizKuruGetir(:DovizId, :SistemDovizId, :IrsaliyeTarihi) into :DovizKuru;
    else DovizKuru = 1;

    if (BirimFiyat is null) then BirimFiyat = 0;

    BirimFiyatTL = BirimFiyat * DovizKuru;

    if (FGirisTipi = 1) then TransferKodu = FYariMamulTransferKodu;

    if (EgemenFirmaID = 19) then /*osborn*/
    begin
      MuhHesapKodu = MuhAlisHesapKodu;
      if (FFason = 1) then
        TransferKodu = FasonTransferKodu;
      else
        TransferKodu = AlisTransferKodu;
    end

    suspend;
  end
end
