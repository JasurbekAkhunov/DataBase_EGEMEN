CREATE PROCEDURE AYLIKBOYAURETIMANALIZIMAKINA(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMTIPIID INTEGER)
RETURNS (
    MAKINAID VARCHAR(10),
    URETIM DOUBLE PRECISION,
    DURUSVEKAYIPZAMAN DOUBLE PRECISION,
    DURUSVEKAYIPZAMANSTR VARCHAR(40),
    KAYIPKG DOUBLE PRECISION)
AS
declare variable FDurusZaman  Double precision;
declare variable HedeflenenUretim Double precision;
declare variable vardiyaSaati Double precision;
declare variable VardiyaSayisi integer;
declare variable HedefTarih    TimeStamp;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  select VardiyaSayisi from sistem into :VardiyaSayisi;

  if (VardiyaSayisi > 0) then
    vardiyaSaati = 24 / VardiyaSayisi;
  else
    vardiyaSaati = 8;

  Select Max(Tarih) from MakinaGunlukUretimHedefi where MakinaGunlukUretimHedefi.Tarih <= :ABitisTarihi into :HedefTarih;

  For Select Makina.MakinaId, MakinaGunlukUretimHedefi.HedeflenenUretim, Sum(PartiAsamaStokDetay.Kilo) from Makina
    left join MakinaGunlukUretimHedefi on MakinaGunlukUretimHedefi.MakinaId = Makina.MakinaId and MakinaGunlukUretimHedefi.Tarih = :HedefTarih
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.CikisTarihi <= :ABitisTarihi and PartiAsamaStokDetay.MakinaId = Makina.MakinaId
    where Makina.Kazan = 1 and ( (:AHamTipiId = 0 and Makina.KumasBoyami = 0) or (:AHamTipiId = 1 and Makina.KumasBoyami = 1) or :AHamTipiId is null)
    group by Makina.MakinaId, MakinaGunlukUretimHedefi.HedeflenenUretim
    into :MakinaId, :HedeflenenUretim, :Uretim do
  begin
    DurusveKayipZaman = 0;
    FDurusZaman = 0;

    Select Sum(Bmk.tarih - MK.Tarih)  from MakinaDurumu MK
      left join Imalataaraverme on Imalataaraverme.BasladiMakinaDurumuId = MK.MakinaDurumuId
      left join MakinaDurumu BMK on BMK.MakinaDurumuId = ImalataAraVerme.bittiMakinaDurumuId
      where MK.tarih >= :ABaslangicTarihi and MK.tarih <= :ABitisTarihi and MK.ImalataAraVerildi = 1 and
           (Mk.MakinaId = :MakinaId) and
            Imalataaraverme.BasladiMakinaDurumuId is not null and BMK.tarih is not null
      into :DurusveKayipZaman;

    if (DurusveKayipZaman is null) then DurusveKayipZaman = 0;
    if (FDurusZaman is null) then FDurusZaman = 0;

    DurusveKayipZaman = DurusveKayipZaman + FDurusZaman;

    if (DurusveKayipZaman = 0) then DurusveKayipZamanStr = '';
    else DurusveKayipZamanStr = TarihToStr(DurusveKayipZaman);

/* ???? neden böyle hesapladıgımı anlamadım!!!   KayipKg = DurusveKayipZaman * 24 * HedeflenenUretim / VardiyaSaati; */
    KayipKg = DurusveKayipZaman * HedeflenenUretim;
    suspend;
  end
end
