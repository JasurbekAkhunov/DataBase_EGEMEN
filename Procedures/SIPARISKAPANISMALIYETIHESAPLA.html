CREATE PROCEDURE SIPARISKAPANISMALIYETIHESAPLA(
    AORMESIPARISDETAYID INTEGER)
RETURNS (
    KPORMEYECIKANIPLIKMIKTARI NUMERIC(12,3),
    KPORMEYECIKANIPLIKBIRIMMALIYET DOUBLE PRECISION,
    KPORMEYECIKANLYCRAMIKTARI NUMERIC(12,3),
    KPORMEYECIKANLYCRABIRIMMALIYET DOUBLE PRECISION,
    KPORMEMIKTARI NUMERIC(12,3),
    KPORMEBIRIMMALIYET DOUBLE PRECISION,
    KPIPLIKBOYAYACIKANBRUTMIKTAR NUMERIC(12,3),
    KPIPLIKBOYAYACIKANBIRIMMALIYET DOUBLE PRECISION,
    KPIPLIKBOYABRUTMIKTAR NUMERIC(12,3),
    KPIPLIKBOYANETMIKTAR NUMERIC(12,3),
    KPIPLIKBOYABIRIMMALIYET DOUBLE PRECISION,
    KPKUMASBOYAYACIKANBRUTMIKTAR NUMERIC(12,3),
    KPKUMASBOYAYACIKANBIRIMMALIYET DOUBLE PRECISION,
    KPKUMASBOYABRUTMIKTAR NUMERIC(12,3),
    KPKUMASBOYANETMIKTAR NUMERIC(12,3),
    KPKUMASBOYABIRIMMALIYET DOUBLE PRECISION,
    KPTOPLAMBIRIMMALIYET DOUBLE PRECISION,
    KPSATISMIKTARI NUMERIC(12,3),
    KPBRUTSATISMIKTARI NUMERIC(12,3),
    KPSATISBIRIMFIYATI DOUBLE PRECISION,
    KPDOVIZID VARCHAR(5))
AS
declare variable uitBukum integer;
declare variable uitOrme integer;
declare variable uitBoya integer;

declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;

declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstLycra        integer;
declare variable hstTelef        integer;

declare variable FHareketTipi         integer;
declare variable FUretimIslemTipiId   integer;
declare variable FHamTipiId           integer;
declare variable FMiktar              Numeric(12,3);
declare variable FBrutMiktar          Numeric(12,3);
declare variable FBirimFiyat          Double Precision;
declare variable FBirimMaliyet        Double Precision;
declare variable FHamBirimMaliyet     Double Precision;
declare variable FHamIade             integer;
declare variable FOrmeRafId           integer;

declare variable FHamId                        integer;
declare variable DovizKuru                     Double Precision;

declare variable FTutar                        Double Precision;
declare variable FPartiStokDovizId             varchar(5);
declare variable BoyaSiparisDetayId            integer;
declare variable FBoyaSiparisFirmaId           varchar(20);
declare variable PartiSinifId                 integer;
begin
  Select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim;

  Select uitBukum, uitOrme, uitBoya from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya;

  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef, hstLycra from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef, :hstLycra;

  KPOrmeyeCikanIplikMiktari        = 0;
  KPOrmeyeCikanIplikBirimMaliyet   = 0;
  KPOrmeyeCikanLycraMiktari        = 0;
  KPOrmeyeCikanLycraBirimMaliyet   = 0;
  KPOrmeMiktari                    = 0;
  KPOrmeBirimMaliyet               = 0;

  KPIplikBoyayaCikanBrutMiktar     = 0;
  KPIplikBoyayaCikanBirimMaliyet   = 0;
  KPIplikBoyaBrutMiktar            = 0;
  KPIplikBoyaNetMiktar             = 0;
  KPIplikBoyaBirimMaliyet          = 0;

  KPKumasBoyayaCikanBrutMiktar     = 0;
  KPKumasBoyayaCikanBirimMaliyet   = 0;
  KPKumasBoyaBrutMiktar            = 0;
  KPKumasBoyaNetMiktar             = 0;
  KPKumasBoyaBirimMaliyet          = 0;

  KPToplamBirimMaliyet             = 0;
  KPSatisMiktari                   = 0;
  KPBrutSatisMiktari               = 0;
  KPSatisBirimFiyati               = 0;

  Select SistemDovizId from Sistem where SistemID = 1 into :KPDovizId;

  Select OrmeSiparisDetay.HamId, Boyasiparis.FirmaId from OrmeSiparisDetay
    left join Ham on Ham.HamID = OrmeSiparisDetay.HamId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = OrmeSiparisDetay.BoyaSiparisKayitId
    where OrmeSiparisDetay.OrmeSiparisDetayID = :AOrmeSiparisDetayId
    into :FHamId, :FBoyaSiparisFirmaId;

  for Select StokHareket.HareketTipi, StokHareket.UretimIslemTipiId, Ham.HamTipiId, StokHareket.Miktar, StokHareket.BrutMiktar, StokHareket.BirimFiyat, StokHareket.BirimMaliyet,
             StokHareket.HamIade, StokHareket.OrmeRafId from StokHareket
      left join Ham on Ham.HamID = StokHareket.HamId
      where StokHareket.OrmeSiparisDetayID = :AOrmeSiparisDetayID
      into :FHareketTipi, :FUretimIslemTipiId, :FHamTipiId, :FMiktar, :FBrutMiktar, :FBirimFiyat, :FBirimMaliyet, :FHamIade, :FOrmeRafId do
  begin
    if (FHareketTipi = shtSatis and FHamIade = 0) then
    begin
      KPSatisMiktari = KPSatisMiktari + FMiktar;
      KPBrutSatisMiktari = KPBrutSatisMiktari + FBrutMiktar;
      KPSatisBirimFiyati = KPSatisBirimFiyati + FMiktar * FBirimFiyat;
      KPToplamBirimMaliyet = KPToplamBirimMaliyet + FMiktar * FBirimMaliyet;
    end
    else if (FHareketTipi = shtSatisIade) then
    begin
      KPSatisMiktari = KPSatisMiktari - FMiktar;
      KPBrutSatisMiktari = KPBrutSatisMiktari - FBrutMiktar;

      KPSatisBirimFiyati = KPSatisBirimFiyati - FMiktar * FBirimFiyat;
      KPToplamBirimMaliyet = KPToplamBirimMaliyet - FMiktar * FBirimMaliyet;
    end
    else if (FHareketTipi = shtFasonUretimeCikis or FHareketTipi = shtUretimeCikis) then
    begin
      if (FUretimIslemTipiId = uitOrme) then
      begin
        if (FHamTipiId = hstIplik) then
        begin
          KPOrmeyeCikanIplikMiktari = KPOrmeyeCikanIplikMiktari + FMiktar;
          KPOrmeyeCikanIplikBirimMaliyet = KPOrmeyeCikanIplikBirimMaliyet + FMiktar * FBirimMaliyet;
        end
        else if (FHamTipiId = hstLycra) then
        begin
          KPOrmeyeCikanLycraMiktari = KPOrmeyeCikanLycraMiktari + FMiktar;
          KPOrmeyeCikanLycraBirimMaliyet = KPOrmeyeCikanLycraBirimMaliyet + FMiktar * FBirimMaliyet;
        end
      end
      else if (FUretimIslemTipiId = uitBoya) then
      begin
        if (FHamTipiId = hstIplik) then
        begin
          KPIplikBoyayaCikanBrutMiktar = KPIplikBoyayaCikanBrutMiktar + FMiktar;
          KPIplikBoyayaCikanBirimMaliyet = KPIplikBoyayaCikanBirimMaliyet + FMiktar * FBirimMaliyet;
        end
        else
        begin
          KPKumasBoyayaCikanBrutMiktar = KPKumasBoyayaCikanBrutMiktar  + FMiktar;
          KPKumasBoyayaCikanBirimMaliyet = KPKumasBoyayaCikanBirimMaliyet + FMiktar * FBirimMaliyet;
        end
      end
    end
    else if ( (FHareketTipi = shtFasonUretim or FHareketTipi = shtUretim) and FHamIade = 1) then
    begin
      if (FUretimIslemTipiId = uitOrme) then
      begin
        if (FHamTipiId = hstIplik) then
        begin
          KPOrmeyeCikanIplikMiktari = KPOrmeyeCikanIplikMiktari - FMiktar;
          KPOrmeyeCikanIplikBirimMaliyet = KPOrmeyeCikanIplikBirimMaliyet - FMiktar * FBirimMaliyet;
        end
        else if (FHamTipiId = hstLycra) then
        begin
          KPOrmeyeCikanLycraMiktari = KPOrmeyeCikanLycraMiktari - FMiktar;
          KPOrmeyeCikanLycraBirimMaliyet = KPOrmeyeCikanLycraBirimMaliyet - FMiktar * FBirimMaliyet;
        end
      end
      else if (FUretimIslemTipiId = uitBoya) then
      begin
        if (FHamTipiId = hstIplik) then
        begin
          KPIplikBoyayaCikanBrutMiktar = KPIplikBoyayaCikanBrutMiktar - FMiktar;
          KPIplikBoyayaCikanBirimMaliyet = KPIplikBoyayaCikanBirimMaliyet - FMiktar * FBirimMaliyet;
        end
        else
        begin
          KPKumasBoyayaCikanBrutMiktar = KPKumasBoyayaCikanBrutMiktar - FMiktar;
          KPKumasBoyayaCikanBirimMaliyet = KPKumasBoyayaCikanBirimMaliyet - FMiktar * FBirimMaliyet;
        end
      end
    end
    else if ( (FHareketTipi = shtFasonUretim or FHareketTipi = shtUretim) and FHamIade = 0) then
    begin
      if (FUretimIslemTipiId = uitOrme) then
      begin
        KPOrmeMiktari = KPOrmeMiktari + FMiktar;
        KPOrmeBirimMaliyet = KPOrmeBirimMaliyet + FMiktar * FBirimFiyat;
      end
      else if (FUretimIslemTipiId = uitBoya) then
      begin
        if (FHamTipiId = hstIplik) then
        begin
          KPIplikBoyaBrutMiktar = KPIplikBoyaBrutMiktar + FBrutMiktar;
          KPIplikBoyaNetMiktar = KPIplikBoyaNetMiktar + FMiktar;
          KPIplikBoyaBirimMaliyet = KPIplikBoyaBirimMaliyet + FMiktar * FBirimFiyat;
        end
        else
        begin
          KPKumasBoyaBrutMiktar = KPKumasBoyaBrutMiktar + FBrutMiktar;
          KPKumasBoyaNetMiktar = KPKumasBoyaNetMiktar + FMiktar;
          KPKumasBoyaBirimMaliyet = KPKumasBoyaBirimMaliyet + FMiktar * FBirimFiyat;
        end
      end
    end
  end

  FMiktar = 0;
  For Select BoyaIrsaliyeDetay.Miktar, BoyaIrsaliyeDetay.BrutMiktar, Ham.HamTipiId, BoyaIrsaliyeDetay.PartiSinifId,
             PartiStok.BirimFiyat, PartiStok.DovizId, PartiStok.BoyaSiparisDetayId from OrmeSiparisDetay
  left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayID = OrmeSiparisDetay.OrmeSiparisDetayId
  left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
  left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifID = PartiStok.PartiSinifID
  left join Ham on Ham.HamID = BoyaIrsaliyeDetay.HamId
  left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
  left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
  where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId and
        BoyaSiparisDetay.BoyaSiparisDetayId is not null and
        BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID is not null and
        BoyaIrsaliye.FirmaID = :FBoyaSiparisFirmaId
  into :FMiktar, :FBrutMiktar, :FHamTipiId, :PartiSinifId, :FBirimFiyat, :FPartiStokDovizId, :BoyaSiparisDetayId do
  begin
    KPSatisMiktari = KPSatisMiktari + FMiktar;
    KPBrutSatisMiktari = KPBrutSatisMiktari + FBrutMiktar;

    if (FPartiStokDovizId <> KpDovizId) then
      select DovizKuru from EnYakinDovizKuruGetir(:FPartiStokDovizId, :KpDovizId, Current_Date) into :DovizKuru;
    else
      DovizKuru = 1;

    FBirimFiyat = FBirimFiyat * DovizKuru;

    KPSatisBirimFiyati = KPSatisBirimFiyati + FMiktar * FBirimFiyat;

    Select HamStokBirimMaliyeti, BirimMaliyet from SiparisDetayBirimMaliyetGetir(:PartiSinifId, :BoyaSiparisDetayID) into :FHamBirimMaliyet, :FBirimMaliyet;

    KPToplamBirimMaliyet = KPToplamBirimMaliyet + FMiktar * FHamBirimMaliyet + FMiktar * FBirimMaliyet;

    if (FHamTipiId = hstIplik) then
    begin
      KPIplikBoyaBrutMiktar = KPIplikBoyaBrutMiktar + FBrutMiktar;
      KPIplikBoyaNetMiktar = KPIplikBoyaNetMiktar + FMiktar;
      KPIplikBoyaBirimMaliyet = KPIplikBoyaBirimMaliyet + FMiktar * FBirimMaliyet;
    end
    else
    begin
      KPKumasBoyaBrutMiktar = KPKumasBoyaBrutMiktar + FBrutMiktar;
      KPKumasBoyaNetMiktar = KPKumasBoyaNetMiktar + FMiktar;
      KPKumasBoyaBirimMaliyet = KPKumasBoyaBirimMaliyet + FMiktar * FBirimMaliyet;
    end
  end

  if (KPOrmeyeCikanIplikMiktari <> 0) then KPOrmeyeCikanIplikBirimMaliyet = KPOrmeyeCikanIplikBirimMaliyet / KPOrmeyeCikanIplikMiktari;
  if (KPOrmeyeCikanLycraMiktari <> 0) then KPOrmeyeCikanLycraBirimMaliyet = KPOrmeyeCikanLycraBirimMaliyet / KPOrmeyeCikanLycraMiktari;
  if (KPOrmeMiktari <> 0) then KPOrmeBirimMaliyet = KPOrmeBirimMaliyet / KPOrmeMiktari;
  if (KPIplikBoyayaCikanBrutMiktar <> 0) then KPIplikBoyayaCikanBirimMaliyet = KPIplikBoyayaCikanBirimMaliyet / KPIplikBoyayaCikanBrutMiktar;
  if (KPKumasBoyayaCikanBrutMiktar <> 0) then KPKumasBoyayaCikanBirimMaliyet = KPKumasBoyayaCikanBirimMaliyet / KPKumasBoyayaCikanBrutMiktar;

  if (KPIplikBoyaNetMiktar <> 0) then KPIplikBoyaBirimMaliyet = KPIplikBoyaBirimMaliyet / KPIplikBoyaNetMiktar;
  if (KPKumasBoyaNetMiktar <> 0) then KPKumasBoyaBirimMaliyet = KPKumasBoyaBirimMaliyet / KPKumasBoyaNetMiktar;

  if (KPSatisMiktari <> 0) then
  begin
    KPSatisBirimFiyati = KPSatisBirimFiyati / KPSatisMiktari;
    KPToplamBirimMaliyet = KPToplamBirimMaliyet / KPSatisMiktari;
  end

  KPOrmeyeCikanIplikBirimMaliyet = Yuvarla(KPOrmeyeCikanIplikBirimMaliyet, 1000);
  KPOrmeyeCikanLycraBirimMaliyet = Yuvarla(KPOrmeyeCikanLycraBirimMaliyet, 1000);
  KPOrmeBirimMaliyet = Yuvarla(KPOrmeBirimMaliyet, 1000);
  KPIplikBoyayaCikanBirimMaliyet = Yuvarla(KPIplikBoyayaCikanBirimMaliyet, 1000);
  KPIplikBoyaBirimMaliyet = Yuvarla(KPIplikBoyaBirimMaliyet, 1000);
  KPKumasBoyayaCikanBirimMaliyet = Yuvarla(KPKumasBoyayaCikanBirimMaliyet, 1000);
  KPKumasBoyaBirimMaliyet = Yuvarla(KPKumasBoyaBirimMaliyet, 1000);
  KPToplamBirimMaliyet = Yuvarla(KPToplamBirimMaliyet, 1000);

  KPSatisBirimFiyati = Yuvarla(KPSatisBirimFiyati, 1000);

  suspend;
end
