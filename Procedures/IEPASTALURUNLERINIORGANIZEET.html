CREATE PROCEDURE IEPASTALURUNLERINIORGANIZEET(
    AHGISEMRIPASTALID MODELKODUSIZE)
AS
declare variable FModelKayitID ModelKoduSize;
declare variable FHGSiparisDEtayID Integer;
declare variable HGIsEmriPastalIDUrun Integer;
declare variable FPlanlananMiktar   Integer;
declare variable UrunID        Urunkodusize;
begin
  Select HGSiparisDetay.HGSiparisDetayID, HGSiparisDetay.ModelKayitID from HGISEmriPastal
    Left Join HGIsEmri on HGISEmri.HGISEmriKayitID = HGISEmriPastal.HGISEmriKayitID
    Left Join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayID = HGISEmri.HGSiparisDetayID
  where HGISEmriPastal.HGISEmriPastalID = :AHGIsEmriPastalID into :FHGSiparisDEtayID, :FModelKayitID;

  for Select HgIsEmriPastalUrun.HGIsEmriPastalID, Urun.UrunID, (HGIsEmriPastalBeden.Miktar * HGIsEmriPastalKumas.KatSayisi) from Urun
    Left Join ModelBeden on ModelBeden.ModelBedenID = Urun.ModelBedenID
    Left Join HGSiparisDetayKumas on HGSiparisDetayKumas.HGSiparisDetayID = :FHGSiparisDEtayID and HGSiparisDetayKumas.ModelKumasID = Urun.ModelKumasID
    Left Join HGIsEmriPastalBeden on HGIsEmriPastalBeden.HGIsEmriPastalID = :AHGIsEmriPastalID and HGIsEmriPastalBeden.BedenID = ModelBeden.BedenID
    Left Join HGIsEmriPastalKumas on HGIsEmriPastalKumas.HGIsEmriPastalID = :AHGIsEmriPastalID and HGIsEmriPastalKumas.HGSiparisDetayKumasID = HGSiparisDetayKumas.HGSiparisDetayKumasID
    Left Join HGIsEmriPastalUrun on HGIsEmriPastalUrun.HGIsEmriPastalID = :AHGIsEmriPastalID and HGIsEmriPastalUrun.UrunID = Urun.UrunID
    Where Urun.ModelKayitID = :FModelKayitID and
          HGIsEmriPastalBeden.HGIsEmriPastalBedenID is not Null and
          HGIsEmriPastalKumas.HGIsEmriPastalKumasID is Not Null and
          (HGIsEmriPastalUrun.HGIsEmriPastalID is Null or HGIsEmriPastalUrun.PlanlananMiktar <> (HGIsEmriPastalBeden.Miktar * HGIsEmriPastalKumas.KatSayisi) )
  into :HGIsEmriPastalIDUrun, :UrunID, :FPlanlananMiktar do
  begin
    if (HGIsEmriPastalIDUrun is not Null) then
      Update HGIsEmriPastalUrun set PlanlananMiktar = :FPlanlananMiktar where HGIsEmriPastalUrun.HGIsEmriPastalID = :AHGIsEmriPastalID and HGIsEmriPastalUrun.UrunID = :UrunID;
    else
      Insert Into HGIsEmriPastalUrun(HGIsEmriPastalID, UrunID, PlanlananMiktar) Values (:AHGIsEmriPastalID, :UrunID, :FPlanlananMiktar);
  end
end
