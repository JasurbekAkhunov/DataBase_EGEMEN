CREATE PROCEDURE LABRECETESINDENRECETEOLUSTUR(
    ALABRECETEATISID INTEGER)
AS
declare variable LabReceteKodu varchar(20);
declare variable AtisNoStr     varchar(20);
declare variable ReceteId      varchar(20);
declare variable Str           varchar(50);
declare variable Adi           varchar(40);
declare variable FirmaId       varchar(20);
declare variable UstAsamaId    integer;
declare variable ReceteTuruId  integer;
declare variable LabReceteGrupId          integer;
declare variable CalismaTarihi TimeStamp;
declare variable RenkOkey      integer;
declare variable KayitSayisi   integer;
declare variable LabReceteId          integer;
declare variable Migrasyon  integer;
declare variable UrunId         varchar(20);
declare variable MiktarYuzde    Double Precision;
declare variable UrunBirimId    integer;
declare variable Ilave              integer;
declare variable SokumUzeriBoya     integer;
declare variable LabRenkGrupId          integer;
declare variable RenkDerinligiId        integer;

begin
  Select Count(*) from Recete where Recete.LabReceteAtisId = :ALabReceteAtisId into :KayitSayisi;
  if (KayitSayisi = 1) then exit;

  Select LabRecete.LabReceteId, LabRecete.LabReceteKodu, LabReceteAtis.AtisNoStr, LabRecete.Adi, LabRecete.FirmaId, Coalesce(LabReceteCalisma.UstAsamaId, LabRecete.UstAsamaId),
         LabRecete.LabReceteGrupId, LabRecete.LabRenkGrupId, LabRecete.RenkDerinligiId,
         LabReceteCalisma.CalismaTarihi, LabReceteAtis.Okey, LabRecete.ReceteTuruId,
         LabReceteAtis.Migrasyon, LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya from LabReceteAtis
  left join LabRecete on LabRecete.LabReceteId = LabReceteAtis.LabReceteId
  left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
  where LabReceteAtis.LabReceteAtisId = :ALabReceteAtisId
  into :LabReceteId, :LabReceteKodu, :AtisNoStr, :Adi, :FirmaId, :UstAsamaId, :LabReceteGrupId, :LabRenkGrupId, :RenkDerinligiId, :CalismaTarihi,
       :RenkOkey, :ReceteTuruId, :Migrasyon, :Ilave, :SokumUzeriBoya;

  Str = LabReceteKodu || '.' || AtisNoStr;

  if ( Char_Length(Str) > 20) then exception ReceteKoduAlaniYetersiz;

  ReceteId = Str;

  Select Count(*) from Recete where Recete.ReceteId = :ReceteId into :KayitSayisi;

  if (KayitSayisi = 1) then
    Update Recete Set LabReceteId = :LabReceteId, LabReceteAtisId = :ALabReceteAtisId where Recete.ReceteId = :ReceteId;
  else
    insert into Recete(ReceteId, ReceteTuruId, Adi, FirmaId, UstAsamaId, CalismaTarihi, RenkOkey, LabReceteId, LabReceteAtisId, LabReceteGrupId,
                       LabRenkGrupId, RenkDerinligiId, Migrasyon, Ilave, SokumUzeriBoya)
                values(:ReceteId, :ReceteTuruId, :Adi, :FirmaId, :UstAsamaId, :CalismaTarihi, :RenkOkey, :LabReceteId, :ALabReceteAtisId, :LabReceteGrupId,
                       :LabRenkGrupId, :RenkDerinligiId, :Migrasyon, :Ilave, :SokumUzeriBoya);
end
