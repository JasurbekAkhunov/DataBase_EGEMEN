CREATE PROCEDURE MAKINAPLANLAMAGRUPLISTESIYAZDIR(
    ABOYAHANEBOLUMID INTEGER)
RETURNS (
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    ADI VARCHAR(30),
    MAKINAPLANLAMAGRUPRAPORSIRANO INTEGER,
    YIKAMASARJGUN INTEGER,
    BEYAZSARJGUN INTEGER,
    BOYAMASARJGUN INTEGER,
    MAKINASAYISI INTEGER)
AS
begin
  For Select MakinaPlanlamaGrup.MakinaPlanlamaGrupId, MakinaPlanlamaGrup.Adi, MakinaPlanlamaGrup.RaporSiraNo, YikamaSarjGun, BeyazSarjGun, BoyamaSarjGun, Count(*) from MakinaPlanlamaGrup
    left join Makina On Makina.MakinaPlanlamaGrupID = MakinaPlanlamaGrup.MakinaPlanlamaGrupId
    where MakinaPlanlamaGrup.BoyahaneBolumID = :ABoyahaneBolumID
    group by MakinaPlanlamaGrup.MakinaPlanlamaGrupId, MakinaPlanlamaGrup.Adi, MakinaPlanlamaGrup.RaporSiraNo, YikamaSarjGun, BeyazSarjGun, BoyamaSarjGun, MakinaPlanlamaGrup.RaporSiraNo
    order by MakinaPlanlamaGrup.RaporSiraNo
    into :MakinaPlanlamaGrupId, :Adi, :MakinaPlanlamaGrupRaporSiraNo, :YikamaSarjGun, :BeyazSarjGun, :BoyamaSarjGun,  :MakinaSayisi do suspend;

  MakinaPlanlamaGrupId    = '';
  Adi                     = 'Planlanmamışlar';
  MakinaPlanlamaGrupRaporSiraNo = 9999;
  YikamaSarjGun           = 0;
  BeyazSarjGun            = 0;
  BoyamaSarjGun           = 0;
  MakinaSayisi            = 0;
  suspend;
end
