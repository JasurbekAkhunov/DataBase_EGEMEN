CREATE PROCEDURE CEKSENETDETAYIGETIR(
    AV VARCHAR(1),
    CS VARCHAR(1),
    ABORDROID INTEGER)
RETURNS (
    BORDROID INTEGER,
    CEKSENETID INTEGER,
    GENELKAYITID INTEGER,
    ACIKLAMA VARCHAR(40),
    VADETARIHI TIMESTAMP,
    SINIF VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZMEBLAG DOUBLE PRECISION,
    DOVIZKURUSTR VARCHAR(40),
    DOVIZMEBLAG DOUBLE PRECISION,
    CEKSENETNO VARCHAR(30),
    HESAPNO VARCHAR(40),
    ILKBORCLU VARCHAR(80),
    KESIDEYERI VARCHAR(20),
    AITOLDUGUBANKAADI VARCHAR(40),
    BANKAHESAPID VARCHAR(20),
    BANKAHESAPADI VARCHAR(40),
    CEKSENETGRUP1 VARCHAR(5),
    CEKSENETGRUP2 VARCHAR(5),
    CEKSENETGRUP3 VARCHAR(5),
    KARSILIKSIZ INTEGER,
    SONBORDROTIPI INTEGER,
    TAHSILEDILMIS INTEGER,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable TahsilEdilenMeblag double precision;
declare variable AGenelKayitId integer;
declare variable DovizKuru1 double precision;
declare variable DovizKuru2 double precision;
declare variable FFirmaId varchar(20);
declare variable FirmaDovizId varchar(5);
declare variable FaturaCariyeIslemDovizinden integer;
begin
  Select FaturaCariyeIslemDovizinden from Sistem where SistemID = 1 into :FaturaCariyeIslemDovizinden;

  select FirmaId  from CekSenet Where CekSenet.AV = :Av and CekSenet.CS = :Cs and CekSenet.BordroId = :ABordroId into :FFirmaId;

  Select DovizId from Firma where Firma.FirmaId = :FFirmaId into :FirmaDovizId;

  for select CekSenetDetay.BordroId, CekSenetDetay.GenelKayitId, CekSenetDetay.CekSenetId, CekSenetDetay.Aciklama,  CekSenetDetay.Vadetarihi,
             CekSenetDetay.Meblag, CekSenetDetay.TahsilEdilenMeblag, CekSenetDetay.Sinif, CekSenetDetay.DovizId, CekSenetDetay.DovizKuru1, CekSenetDetay.DovizKuru2,
             Banka.Adi, CekSenetDetay.BankaHesapId, BankaHesap.Adi, CekSenetDetay.CekSenetno, CekSenetDetay.HesapNo,
             CekSenetDetay.IlkBorclu, CekSenetDetay.KesideYeri, CekSenetDetay.CekSenetGrup1, CekSenetDetay.CekSenetGrup2, CekSenetDetay.CekSenetGrup3, CekSenetDetay.Karsiliksiz,
             CekSenetDetay.SonBordroTipi, Kullanici.Adi, CekSenetDetay.DegisimTarihi, CekSenetDetay.IslemDovizId, CekSenetDetay.IslemDovizMeblag, CekSenetDetay.Notu from CekSenetDetay
  left join BankaHesap on BankaHesap.BankaHesapID = CekSenetDetay.BankaHesapID
  left join Banka on (Banka.bankaId = CekSenetDetay.AitOlduguBankaId or Banka.BankaId = BankaHesap.BankaId)
  left join Kullanici on Kullanici.KullaniciId = CekSenetDetay.KullaniciId
  where CekSenetDetay.AV = :Av and CekSenetDetay.CS = :Cs and CekSenetDetay.BordroId = :ABordroId
  order by CekSenetDetay.VadeTarihi, CekSenetDetay.CekSenetId
  into :BordroId, :GenelKayitId, :CekSenetId, :Aciklama, :Vadetarihi, :Meblag, :TahsilEdilenMeblag, :Sinif, :DovizId, :DovizKuru1, :DovizKuru2, :AitOlduguBankaAdi, :BankaHesapId,
       :BankaHesapAdi, :CekSenetno, :HesapNo,
       :IlkBorclu, :KesideYeri, :CekSenetGrup1, :CekSenetGrup2, :CekSenetGrup3, :Karsiliksiz, :SonBordroTipi, :KullaniciAdi, :DegisimTarihi, :IslemDovizId, :IslemDovizMeblag, :Notu do
  begin
    DovizMeblag = Null;

    if (FaturaCariyeIslemDovizinden = 0) then
    begin
      DovizMeblag = Meblag / DovizKuru1 * DovizKuru2;

      if (FirmaDovizId <> DovizId) then
      begin
        DovizKuruStr = Format_Float('', DovizKuru1) || ' ' || DovizId || '-' || Format_Float('', DovizKuru2) || ' ' || FirmaDovizId;
      end
      else DovizKuruStr = '';

    end
    else
    begin
      DovizMeblag = IslemDovizMeblag / DovizKuru1 * DovizKuru2;

      if (FirmaDovizId <> IslemDovizId) then
      begin
        DovizKuruStr = Format_Float('', DovizKuru1) || ' ' || IslemDovizId || '-' || Format_Float('', DovizKuru2) || ' ' || FirmaDovizId;
      end
      else DovizKuruStr = '';

    end

    if (TahsilEdilenMeblag = 0) then TahsilEdilmis = 0;
    else if (TahsilEdilenMeblag < Meblag) then TahsilEdilmis = 1;
    else TahsilEdilmis = 2;

    suspend;
  end
end
