CREATE PROCEDURE BOYAFATURASIKESILMEMISIRSALIYE(
    AFIRMAID VARCHAR(20),
    AIRSALIYENO IRSALIYENOSIZE,
    AHAREKETTIPIID INTEGER)
RETURNS (
    BOYAIRSALIYEID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    HAMADI VARCHAR(255),
    BOYASIPARISDETAYID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    KILO NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    FASON INTEGER,
    FASONSTR VARCHAR(10),
    ISLEMTIPIID INTEGER,
    ISLEMGRUBUID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    TARTIBRUT NUMERIC(12,3),
    SIPENISTENEN VARCHAR(10),
    SIPGRAMAJISTENEN VARCHAR(10),
    ODEMESEKLIADI VARCHAR(40))
AS
begin
  for select BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliyeDetay.PartiKayitId,
             BoyaIrsaliyeDetay.PartiSinifId, PartiStok.BoyaSiparisDetayId, Parti.PartiYil, Parti.PartiId, Parti.ReceteId,
             Recete.Adi, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
             BoyaIrsaliyeDetay.Kilo, BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.Metre, BoyaIrsaliyeDetay.BrutMetre,
             Parti.Fason, Parti.IslemTipiId, IslemTipi.Adi, Parti.IslemGrubuId, IslemGrubu.Adi,
             PartiStok.BirimFiyat, PartiStok.DovizId, Ham.Adi, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, OdemeSekli.Adi from BoyaIrsaliyeDetay
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join OdemeSekli on OdemeSekli.OdemeSekliID = BoyaSiparis.OdemeSekliID
    where BoyaIrsaliyeDetay.BoyaFaturaId is null and BoyaIrsaliyeDetay.FirmaId = :AFirmaId and BoyaIrsaliyeDetay.HareketTipiId = :AHareketTipiId and BoyaIrsaliyeDetay.FaturaKesilecek = 1 and
          BoyaIrsaliye.FaturaKesilecek = 1 and
         (BoyaIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
         (PartiStok.PartiSinifId is null or PartiStok.FiyatOnayiIstensin = 0 or PartiStok.FiyatOnaylayanKullaniciid is not null)
    order by Parti.FirmaId, Parti.PartiKayitId
    into :BoyaIrsaliyeId, :BoyaIrsaliyeDetayId, :IrsaliyeNo, :IrsaliyeTarihi, :PartiKayitId, :PartiSinifId, :BoyaSiparisDetayId, :PartiYil, :PartiId, :ReceteId,
         :ReceteAdi, :DesenId, :DesenAdi, :VaryantAdi, :Kilo, :BrutKilo, :Metre, :BrutMetre, :Fason, :IslemTipiId, :IslemTipiAdi,
         :IslemGrubuId, :IslemGrubuAdi, :BirimFiyat, :DovizId, :HamAdi, :SipEnIstenen, :SipGramajIstenen, :OdemeSekliAdi do
  begin
    if (PartiSinifId is not null) then
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';
    end
    else
    begin
      if (AHareketTipiID = 0) then PartiIdStr = 'Ham Stok Satış';
      else if (AHareketTipiID = 1) then PartiIdStr = 'Alış İade';
      else exception KayitHatasi;

      FasonStr = '';
    end

    Select Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara) from PartiStokTarti
      where PartiStokTarti.BoyaIrsaliyeDetayId = :BoyaIrsaliyeDetayId
      into :TartiBrut;

    suspend;
  end
end
