CREATE PROCEDURE FIRMACARIBAKIYEVADESIHESAPLA(
    AFIRMAID VARCHAR(20),
    ABAKIYE DOUBLE PRECISION,
    ABAKIYEBA VARCHAR(1),
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1),
    ADOVIZID VARCHAR(5))
RETURNS (
    BAKIYEVADETARIHI TIMESTAMP,
    BAKIYESISTEMDOVIZ DOUBLE PRECISION,
    SISTEMDOVIZID VARCHAR(5))
AS
declare variable Bakiye  Double Precision;
declare variable AraTutar Double Precision;
declare variable Meblag   Double Precision;
declare variable Valortarihi timestamp;
declare variable TGun Double Precision;
declare variable ToplamMeblag Double Precision;
declare variable CariTipiId integer;
declare variable FirmaDovizId   varchar(5);
declare variable FDovizKuru Double Precision;
declare variable FMSg varchar(255);

begin
  if (ABakiye is not null) then ABakiye = Yuvarla(ABakiye, 100);

  if (ABakiye is null or ABakiye <= 0.01) then
  begin
    suspend;
    exit;
  end

  Select Firma.DovizId from Firma where Firma.FirmaId = :AFirmaId into :FirmaDovizId;
  Select SistemDovizId from Sistem where SistemId = 1 into :SistemDovizId;

  if (ADovizID is not Null) then FirmaDovizId = :ADovizID;

  if (FirmaDovizId <> SistemDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FirmaDovizId, :SistemDovizId, Current_Date) into :FDovizKuru;
  else FDovizKuru = 1;

  BakiyeSistemDoviz = ABakiye * FDovizKuru;

  AraTutar = ABakiye;
  Bakiye = ABakiye;
  ToplamMeblag = 0;

  for select iif(:ADovizID is Null, (Meblag / DovizKuru1 * DovizKuru2), (Meblag)), ValorTarihi,CariTipiId  from Cari
          where Cari.FirmaId = :AFirmaId and Cari.Tarih <= :ABitistarihi and Cari.BorcAlacak = :ABakiyeBa and (Cari.Sinif = :Asinif or :ASinif is null) and
                (DovizID = :ADovizID or :ADovizID is Null)
     order by Tarih Desc, CariId Desc
    into :Meblag, :ValorTarihi, :CariTipiId do
  begin
    if (CariTipiId <> 4 and CariTipiId <> 5 and CariTipiId <> 6) then
    begin
      Meblag = Yuvarla(Meblag, 100);

      TGun = TimeStampToDouble(ValorTarihi);
      if(Meblag > bakiye) then
      begin
        ToplamMeblag = ToplamMeblag  + TGun * Bakiye;
        bakiye = 0;
        AraTutar = Bakiye;
      end
      else
      begin
        Bakiye = Bakiye - Meblag;
        ToplamMeblag = ToplamMeblag  + TGun * Meblag;
        AraTutar = Bakiye;
      end

      Bakiye = Yuvarla(Bakiye, 100);

      if(Bakiye <= 0.01) then
      begin
        TGun = ToplamMeblag / ABakiye;
        TGun = Yuvarla(TGun, 1.0);
        BakiyeVadeTarihi = DoubleToTimeStamp(TGun);
        suspend;
        exit;
      end
    end
  end
end
