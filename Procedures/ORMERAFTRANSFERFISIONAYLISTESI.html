CREATE PROCEDURE ORMERAFTRANSFERFISIONAYLISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AONAYDURUMU INTEGER)
RETURNS (
    ORMERAFTRANSFERFISIID INTEGER,
    ORMERAFTRANSFERFISIDETAYID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    REFERANSNO VARCHAR(30),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ORMERAFID INTEGER,
    YENIORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    ESKIRAFID VARCHAR(10),
    YENIRAFID VARCHAR(10),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3),
    KALITENO INTEGER,
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    HAMRENKADI VARCHAR(50),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    FASONSTOK INTEGER,
    URETIMPARTINO VARCHAR(30),
    ORMECIMAKINANO VARCHAR(10),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(12),
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP,
    YENIDEPOID VARCHAR(10),
    BARKODID VARCHAR(30))
AS
begin
  For Select OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId, OrmeRafTransferFisiDetay.OrmeRafId, OrmeRafTransferFisiDetay.YeniOrmeRafId,
             Ham.Adi, RafRenk.Adi, OrmeRaf.RafId, OrmeRafTransferFisiDetay.RafId, OrmeRaf.Lot, OrmeRaf.MelanjKodu, HamRenk.Adi,
             OrmeRafTransferFisiDetay.KapAdedi, OrmeRafTransferFisiDetay.Kilo, OrmeRafTransferFisiDetay.BrutKilo, OrmeRafTransferFisiDetay.KaliteNo,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, SiparisFirma.Adi,
             OrmeSiparisDetay.SiparisSiraNo, OrmeRaf.OrmeSiparisDetayId, OrmeRaf.BoyaSiparisDetayId, OrmeRaf.StokGirisNo, OrmeRaf.BarkodId,
             Ik.Adi, OrmeRafTransferFisiDetay.InsertTarihi, Sk.Adi, OrmeRafTransferFisiDetay.DegisimTarihi, OrmeRaf.FasonStok, OrmeRaf.UretimPartiNo, OrmeRaf.OrmeciMakinaNo,
             OrmeRafTransferFisi.OrmeRafTransferFisiId, OrmeRafTransferFisi.FisYil, OrmeRafTransferFisi.FisNo, OrmeRafTransferFisi.FisTarihi,
             OrmeRafTransferFisi.Notu, OrmeRafTransferFisi.ReferansNo, OrmeRafTransferFisiDetay.OnayDurumu, OrmeRafTransferFisiDetay.OnaylayanKullaniciId, OK.Adi,
             OrmeRafTransferFisiDetay.OnayTarihi, DepoRaf.DepoID from OrmeRafTransferFisiDetay
  Left Join OrmeRafTransferFisi on OrmeRafTransferFisi.OrmeRafTransferFisiID = OrmeRafTransferFisiDetay.OrmeRafTransferFisiID
  left join OrmeRaf on OrmeRaf.OrmeRafID = OrmeRafTransferFisiDetay.OrmeRafId
  Left Join Kullanici IK on IK.KullaniciId = OrmeRafTransferFisiDetay.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = OrmeRafTransferFisiDetay.KullaniciId
  left join Ham on Ham.HamId = OrmeRaf.HamId
  left join Renk RafRenk on RafRenk.RenkId = OrmeRaf.RenkId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
  left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
  left join DepoRaf on DepoRaf.RafID = OrmeRafTransferFisiDetay.RafId
  Left Join Kullanici OK on OK.KullaniciId = OrmeRafTransferFisiDetay.OnaylayanKullaniciId
  where (OrmeRafTransferFisiDetay.OnayDurumu = :AOnayDurumu or :AOnayDurumu is Null) and
         OrmeRafTransferFisi.FisTarihi >= :ABaslangicTarihi and OrmeRafTransferFisi.FisTarihi <= :ABitisTarihi
  order by OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId
  into :OrmeRafTransferFisiDetayId, :OrmeRafId, :YeniOrmeRafId, :HamAdi,  :RenkAdi, :EskiRafId, :YeniRafId, :Lot, :MelanjKodu, :HamRenkAdi,
       :KapAdedi, :Kilo, :BrutKilo, :KaliteNo, :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi,
       :SiparisSiraNo, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :StokGirisNo, :BarkodId,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :FasonStok, :UretimPartiNo, :OrmeciMakinaNo,
       :OrmeRafTransferFisiId, :FisYil, :FisNo, :FisTarihi, :Notu, :ReferansNo, :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi, :YeniDepoId do
  begin
    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';
    else
      BoyaSiparisIdStr = '';

    if (OnayDurumu = 0) then OnayDurumuStr = 'Beklemede';
    else if (OnayDurumu = 1) then OnayDurumuStr = 'Onaylanmış';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Reddedilmiş';
    else OnayDurumuStr = '';

    FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';

    suspend;
  end
end
