CREATE PROCEDURE PSSABLONRECETESIKOPYALA(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    AYENIPARTISINIFID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable EskiPartiSiraNo integer;
declare variable SablonNo        integer;
declare variable Acilim1         integer;
declare variable Acilim2         integer;
declare variable UrunId          varchar(20);
declare variable UrunBirimId     integer;
declare variable Miktar          Double precision;
declare variable PartiSablonId   integer;
declare variable Lot             varchar(150);
begin
  For Select SablonNo, Acilim1, Acilim2 from PartiSablon where PartiSablon.PartiSinifId = :APartiSinifId into :SablonNo, :Acilim1, :Acilim2 do
  begin
    Update PartiSablon Set Acilim1 = :Acilim1, Acilim2 = :Acilim2 where PartiSablon.PartiSinifId = :AYeniPartiSinifId and PartiSablon.SablonNo = :SablonNo;
  end


  For Select PartiSablon.SablonNo, PartiSablonRecete.UrunId, PartiSablonRecete.UrunBirimId, PartiSablonRecete.Miktar, PartiSablonRecete.Lot From PartiSablon
    left join PartiSablonRecete on PartiSablonRecete.PartiSablonId = PartiSablon.PartiSablonId
    where PartiSablon.PartiSinifId = :APartiSinifId and PartiSablonRecete.UrunId is not null

    into :SablonNo, :UrunId, :UrunBirimId, :Miktar, :Lot do /* her sablon icin receteleri okuyoyoruz */
  begin
    For Select PartiSablonId from PartiSablon where PartiSablon.PartiSinifID = :AYeniPartiSinifID and PartiSablon.SablonNo = :SablonNo into :PartiSablonId do
    begin

      insert into PartiSablonRecete(PartiSablonId, PartiKayitId, UrunId, UrunBirimId, Miktar, InsertKullaniciId, KullaniciId, Lot)
                  values(:PartiSablonId, :APartiKayitId, :UrunId, :UrunBirimId, :Miktar, :AKullaniciId, :AKullaniciId, :Lot);
    end
  end
end
