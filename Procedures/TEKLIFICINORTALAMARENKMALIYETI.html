CREATE PROCEDURE TEKLIFICINORTALAMARENKMALIYETI(
    ABOYATEKLIFDETAYID INTEGER)
RETURNS (
    RECETEMALIYETI DOUBLE PRECISION)
AS
declare variable ToplamMaliyet   Double Precision;
declare variable ToplamMiktar    Numeric(12,3);
begin
   Select Sum(Miktar * BirimFiyat), Sum(Miktar) from BoyaTeklifRenk where BoyaTeklifRenk.BoyaTeklifDetayId = :ABoyaTeklifDetayId
     into :ToplamMaliyet, :ToplamMiktar;

  ReceteMaliyeti = 0;
  if (ToplamMiktar > 0) then ReceteMaliyeti = ToplamMaliyet / ToplamMiktar;
  suspend;
end
