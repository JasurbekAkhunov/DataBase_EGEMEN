CREATE PROCEDURE PARTISTOKBSDETAYIDDEGISTIR(
    APARTISINIFID INTEGER,
    ABOYASIPARISDETAYID INTEGER,
    ABOYASIPARISSARJID INTEGER,
    ABOYASIPARISBASKIID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable KayitSayisi            integer;
declare variable PartiKayitId           integer;
declare variable PartiBoyaSiparisKayitId  integer;
declare variable BoyaSiparisKayitId     integer;
declare variable PartiSinifId           integer;
declare variable HamId                  integer;
declare variable EskiPartiStokDurumu    integer;

declare variable EHamStokDetayId        integer;
declare variable EBoyaSiparisRezervId   integer;
declare variable ERafDetayId            integer;

declare variable EKapAdedi              integer;
declare variable EKilo                  Numeric(12,3);
declare variable EMetre                 Numeric(12,3);
declare variable EAdet                  integer;
declare variable EBobinSayisi           integer;

declare variable YKilo                  Numeric(12,3);
declare variable YMetre                 Numeric(12,3);
declare variable YAdet                  integer;

declare variable RBoyaSiparisRezervDetayId integer;
declare variable RKilo                  Numeric(12,3);
declare variable RMetre                 Numeric(12,3);
declare variable RAdet                  integer;
declare variable RKapAdedi              integer;

declare variable KapAdedi              integer;
declare variable Kilo                  Numeric(12,3);
declare variable Metre                 Numeric(12,3);
declare variable Adet                  integer;
declare variable BobinSayisi           integer;
declare variable BoyaSiparisRezervId   integer;
declare variable FHamTipiId                      integer;
declare variable ProxyId               varchar(8);
declare variable PartiIslemTipiId      integer;
declare variable SiparisIslemTipiId    integer;
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable YeniDesenVaryantId integer;
declare variable FKaliteKontrolAsamaId integer;
declare variable PartiSiraNo          integer;
begin
  Select DokumaKumasiKilodanTakipEt, KaliteKontrolAsamaId from Sistem into :DokumaKumasiKilodanTakipEt, :FKaliteKontrolAsamaId;
  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  Select Parti.BoyaSiparisKayitId, PartiStok.PartiKayitId, PartiStok.HamTipiId, PartiStok.Durumu, PartiStok.ProxyId, Parti.IslemTipiId, PartiStok.PartiSiraNo from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    where PartiStok.PartiSinifId = :APartiSinifId
    into :PartiBoyaSiparisKayitId, :PartiKayitId, :FHamTipiId, :EskiPartiStokDurumu, :ProxyId, :PartiIslemTipiId, :PartiSiraNo;

  Select count(*) from PartiAsamaStok
        where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and (PartiAsamaStok.AsamaID <> :FKaliteKontrolAsamaId or :FKaliteKontrolAsamaId is Null)and
              PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is null into :KayitSayisi;

  if (KayitSayisi > 0) then exception BitmemisAsamasiVarTasinamaz;

  if (ABoyaSiparisSarjId is not null) then
  begin
    Select BoyaSiparisSarj.BoyaSiparisKayitId, BoyaSiparisSarj.BoyaSiparisDetayId, BoyaSiparisDetay.HamId, BoyaSiparisDetay.IslemTipiId from BoyaSiparisSarj
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisSarj.BoyaSiparisDetayId
       where BoyaSiparisSarj.BoyaSiparisSarjId = :ABoyaSiparisSarjId
       into :BoyaSiparisKayitId, :ABoyaSiparisDetayId, :HamId, :SiparisIslemTipiId;
  end
  else if (ABoyaSiparisBaskiId is not null) then
  begin
    Select BoyaSiparisBaski.BoyaSiparisKayitId, BoyaSiparisBaski.BoyaSiparisDetayId, BoyaSiparisDetay.HamId,
           BoyaSiparisDetay.IslemTipiId, BoyaSiparisBaski.DesenVaryantId from BoyaSiparisBaski
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisBaski.BoyaSiparisDetayId
       where BoyaSiparisBaski.BoyaSiparisBaskiId = :ABoyaSiparisBaskiId
       into :BoyaSiparisKayitId, :ABoyaSiparisDetayId, :HamId, :SiparisIslemTipiId, :YeniDesenVaryantId;
  end
  else
  begin
    Select BoyaSiparisDetay.BoyaSiparisKayitId, BoyaSiparisDetay.HamId, BoyaSiparisDetay.IslemTipiId from BoyaSiparisDetay
       where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId
       into :BoyaSiparisKayitId, :HamId, :SiparisIslemTipiId;
  end

  if (PartiBoyaSiparisKayitId <> BoyaSiparisKayitId) then exception BoyaSiparisDetayNoHatasi;

  if (PartiIslemTipiId <> SiparisIslemTipiId) then
  begin
    if (PartiIslemTipiId = itTamir or SiparisIslemTipiId = itTamir) then exception IslemTipiUymuyorAktarilamaz;
  end

  Select Count(*) from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.HamId <> :HamId into :KayitSayisi;

  if (KayitSayisi > 0) then exception HamStokSipariseUygunDegil;
  PartiSinifId = gen_id(PartiSinif_Id_gen, 1);

  Update PartiStok Set ProxyId = Null where PartiStok.PartiSinifId = :APartiSinifId;

  if (ABoyaSiparisBaskiId is not null) then
    Update Parti Set DesenVaryantId = :YeniDesenVaryantId where Parti.PartiKayitId = :PartiKayitId;

/* Yeni Sinif aciyoruz */
  Insert Into PartiStok(PartiKayitId, PartiSinifId, BoyaSiparisDetayId, BoyaSiparisSarjId, BoyaSiparisBaskiId, MamulKodu, ProxyId,
             IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen, CekmezlikBoyIstenen,
             YikamaHasligiIstenen, Durumu, MaximumFireOrani, MinimumFireOrani, InsertKullaniciId, KullaniciId, SonCikisMetre, LinkIslemi,
             HacimEn, HacimBoy, HacimYukseklik, KKPartiRenkLotNo, KKNotu, Gramaj)
        Select :PartiKayitId, :PartiSinifId, :ABoyaSiparisDetayId, :ABoyaSiparisSarjId, :ABoyaSiparisBaskiId, MamulKodu, :ProxyId,
             IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen, CekmezlikBoyIstenen,
             YikamaHasligiIstenen,:EskiPartiStokDurumu, MaximumFireOrani, MinimumFireOrani, :AKullaniciId, :AKullaniciId, 0, 1,
             HacimEn, HacimBoy, HacimYukseklik, KKPartiRenkLotNo, KKNotu, Gramaj
             from PartiStok where PartiStok.PartiSinifId = :APartiSinifId;

  Update PartiAsamaStokDetay Set PartiSinifId = :PartiSinifId where PartiAsamaStokDetay.PartiSinifId = :APartiSinifId;

  Update PartiAsamaStok Set PartiSinifId = :PartiSinifId where PartiAsamaStok.PartiSinifId = :APartiSinifId;

  Update PartiTamirSinif Set PartiSinifId = :PartiSinifId where PartiTamirSinif.PartiSinifId = :APartiSinifId;

  Update BolumPlanlama Set PartiSinifId = :PartiSinifId where BolumPlanlama.PartiSinifId = :APartiSinifId;

  Update PartiAsamaIcKontrol Set PartiSinifId = :PartiSinifId where PartiAsamaIcKontrol.PartiSinifId = :APartiSinifId;

  Update PartiAsamaDeger Set PartiSinifId = :PartiSinifId where PartiAsamaDeger.PartiSinifId = :APartiSinifId;

  Update PartiAsamaAraKontrol Set PartiSinifId = :PartiSinifId where PartiAsamaAraKontrol.PartiSinifId = :APartiSinifId;

  Update PartiStokSonKontrol Set PartiSinifId = :PartiSinifId, LinkIslemi = 1 where PartiStokSonKontrol.PartiSinifId = :APartiSinifId;

  Update PartiStokFizikKontrol Set PartiSinifId = :PartiSinifId, LinkIslemi = 1 where PartiStokFizikKontrol.PartiSinifId = :APartiSinifId;

  Update MamulDepoTransferFisiDetay Set PartiSinifId = :PartiSinifId, LinkIslemi = 1 where MamulDepoTransferFisiDetay.PartiSinifId = :APartiSinifId;

  Update PartiStokTarti set PartiKayitId = :PartiKayitId, PartiSinifId = :PartiSinifId where PartiStokTarti.PartiSinifId = :APartiSinifId;

  Update IplikPaketleme Set PartiSinifId = :PartiSinifId where IplikPaketleme.PartiSinifId = :APartiSinifId;

  Update PsKaliteKontrol Set PartiSinifId = :PartiSinifId, LinkIslemi = 1, KullaniciId = :AKullaniciId where PsKaliteKontrol.PartiSinifId = :APartiSinifId;

  Update PartiKumasKBilgi Set PartiSinifId = :PartiSinifId where PartiKumasKBilgi.PartiSinifId = :APartiSinifId;

  Update PSKaliteKontrolAsama Set PartiSinifId = :PartiSinifId, LinkIslemi = 1 where PSKaliteKontrolAsama.PartiSinifId = :APartiSinifId;

  Update UniteIrsaliyeDetay set PartiSinifId = :PartiSinifId where UniteIrsaliyeDetay.PartiSinifId = :APartiSinifId;

  Update PartiStokPartileme set PartiSinifId = :PartiSinifId where PartiStokPartileme.PartiSinifId = :APartiSinifId;
  /* PartiStokDetay ve rezerv aktarmasi yapiyoruz */
  For Select PartiStokDetay.RafDetayId, RafDetay.HamStokDetayId, PartiStokDetay.BoyaSiparisRezervId, PartiStokDetay.KapAdedi, PartiStokDetay.Kilo,
             PartiStokDetay.Metre, PartiStokDetay.Adet, PartiStokDetay.BobinSayisi from PartiStokDetay
     left join RafDetay on rafDetay.rafDetayId = PartiStokDetay.RafDetayId
     where PartiStokDetay.PartiSinifId = :APartiSinifId
     into :ERafDetayId, :EHamStokDetayId, :EBoyaSiparisRezervId, :EKapAdedi, :EKilo, :EMetre, :EAdet, :EBobinSayisi do
  begin
    update PartiStokDetay set LinkIslemi = 2 where PartiStokDetay.PartiSinifId = :APartiSinifId and
                                                   PartiStokDetay.RafDetayId = :ERafDetayId;

    Delete from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId and
                                     PartiStokDetay.RafDetayId = :ERafDetayId;

    KapAdedi = EKapAdedi;
    Kilo = EKilo;
    Metre = EMetre;
    Adet = EAdet;
    BobinSayisi = EBobinSayisi;

    For Select BoyaSiparisRezervDetay.BoyaSiparisRezervDetayId, BoyaSiparisRezervDetay.Kilo, BoyaSiparisRezervDetay.Metre, BoyaSiparisRezervDetay.Adet, BoyaSiparisRezervDetay.KapAdedi from BoyaSiparisRezerv
        left join BoyaSiparisRezervDetay on BoyaSiparisRezervDetay.BoyaSiparisDetayId = BoyaSiparisRezerv.BoyaSiparisDetayId and IntKarsilastir(BoyaSiparisRezervDetay.BoyaSiparisSarjId, BoyaSiparisRezerv.BoyaSiparisSarjId) = 1 and BoyaSiparisRezervDetay.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
        where BoyaSiparisRezerv.BoyaSiparisRezervId = :EBoyaSiparisRezervId and BoyaSiparisRezervDetay.BoyaSiparisRezervDetayId is not null
        into :RBoyaSiparisRezervDetayId, :RKilo, :RMetre, :RAdet, :RKapAdedi do
    begin
      if (FHamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
      begin
        if (Ekilo > 0) then
        if (EKilo < RKilo) then
        begin
          Update BoyaSiparisRezervDetay Set Kilo = Kilo - :EKilo, Metre = Metre - :EMetre, Adet = Adet - :EAdet, KapAdedi = KapAdedi - :EKapAdedi where BoyaSiparisRezervDetayId = :RBoyaSiparisRezervDetayId;

          Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, HamStokDetayId, Kilo, Metre, Adet, KapAdedi, InsertKullaniciId, KullaniciId)
                                 values(:BoyaSiparisKayitId, :ABoyaSiparisDetayId, :ABoyaSiparisSarjId, :EHamStokDetayId, :EKilo, :EMetre, :EAdet, :EKapAdedi, :AKullaniciId, :AKullaniciId);
          EKilo = 0;
          EMetre = 0;
          EAdet = 0;
          EKapAdedi = 0;
        end
        else
        if (EKilo >= RKilo) then
        begin
          Delete from BoyaSiparisRezervDetay where BoyaSiparisRezervDetayId = :RBoyaSiparisRezervDetayId;

          Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, HamStokDetayId, Kilo, Metre, Adet, KapAdedi, InsertKullaniciId, KullaniciId)
                                 values(:BoyaSiparisKayitId, :ABoyaSiparisDetayId, :ABoyaSiparisSarjId, :EHamStokDetayId, :RKilo, :RMetre, :RAdet, :RKapAdedi, :AKullaniciId, :AKullaniciId);

          EKilo = EKilo - RKilo;
          EMetre = EMetre - RMetre;
          EAdet = EAdet - RAdet;
          EKapAdedi = EKapAdedi - RKapAdedi;
        end
      end
      else
      begin
        if (EMetre > 0) then
        if (EMetre < RMetre) then
        begin
          Update BoyaSiparisRezervDetay Set Metre = Metre - :EMetre, Kilo = Kilo - :EKilo, Adet = Adet - :EAdet, KapAdedi = KapAdedi - :EKapAdedi where BoyaSiparisRezervDetayId = :RBoyaSiparisRezervDetayId;

          Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, HamStokDetayId, Metre, Kilo, Adet, KapAdedi, InsertKullaniciId, KullaniciId)
                                 values(:BoyaSiparisKayitId, :ABoyaSiparisDetayId, :ABoyaSiparisSarjId, :EHamStokDetayId, :EMetre, :EKilo, :EAdet, :EKapAdedi, :AKullaniciId, :AKullaniciId);
          EKilo = 0;
          EMetre = 0;
          EAdet = 0;
          EKapAdedi = 0;
        end
        else
        if (EMetre >= RMetre) then
        begin
          Delete from BoyaSiparisRezervDetay where BoyaSiparisRezervDetayId = :RBoyaSiparisRezervDetayId;

          Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, HamStokDetayId, Metre, Kilo, Adet, KapAdedi, InsertKullaniciId, KullaniciId)
                                 values(:BoyaSiparisKayitId, :ABoyaSiparisDetayId, :ABoyaSiparisSarjId, :EHamStokDetayId, :RMetre, :RKilo, :RAdet, :RKapAdedi, :AKullaniciId, :AKullaniciId);

          EKilo = EKilo - RKilo;
          EMetre = EMetre - RMetre;
          EAdet = EAdet - RAdet;
          EKapAdedi = EKapAdedi - RKapAdedi;

        end
      end
    end

    Select BoyaSiparisRezervId from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayId = :ABoyaSiparisDetayId and
                                     BoyaSiparisRezerv.BoyaSiparisKayitId = :BoyaSiparisKayitId and
                                     IntKarsilastir(BoyaSiparisRezerv.BoyaSiparisSarjId, :ABoyaSiparisSarjId) = 1 and
                                     BoyaSiparisRezerv.HamStokDetayId = :EHamStokDetayId into :BoyaSiparisRezervId;

    insert into PartiStokDetay(PartiKayitId, PartiSinifId, BoyaSiparisRezervId, RafDetayId, KapAdedi, Kilo, Metre, Adet, BobinSayisi, InsertKullaniciId, KullaniciId, LinkIslemi)
                     values(:PartiKayitId, :PartiSinifId, :BoyaSiparisRezervId, :ERafDetayId, :KapAdedi, :Kilo, :Metre, :Adet, :BobinSayisi, :AKullaniciId, :AKullaniciId, 1);
  end
  Update PartiStok Set PartiSiraNo = :PartiSiraNo where PartiStok.PartiSinifId = :PartiSinifId;


  Execute Procedure PartiSinifDurumuHesapla(:PartiKayitId, :PartiSinifId);
end
