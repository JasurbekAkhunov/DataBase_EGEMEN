CREATE PROCEDURE FIRMANINPARTIRAPORU_1(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AISLEMTIPIID INTEGER,
    ASATISTIPI INTEGER,
    APARTITIPIID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    ADURUMU INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    ISLEMTIPIADI VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    MUSTERISIPARISNO VARCHAR(50),
    BOYASIPARISDETAYID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    DURUMU INTEGER,
    DURUMUSTR VARCHAR(20),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    GIRISIRSALIYENO VARCHAR(100),
    GIRISIRSALIYETARIHI VARCHAR(100),
    BRUTSEVKKILO NUMERIC(12,3),
    BRUTSEVKMETRE NUMERIC(12,3),
    NETSEVKKILO NUMERIC(12,3),
    NETSEVKMETRE NUMERIC(12,3))
AS
declare variable FIrsaliyeNo IrsaliyeNoSize;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable Str varchar(30);
begin
  For Select Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.SiparisTarihi,
             Parti.Fason, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.ReceteId, Recete.Adi, PartiStok.BoyaSiparisDetayId,
             PartiStok.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.Metre, PartiStok.Durumu, PartiDurum.Adi, IslemTipi.Adi,
             BoyaSiparisDetay.MusteriSiparisNo, LabRecete.LabReceteKodu, LabRecete.Adi,
             PartiStok.Kilo, PartiStok.Metre, PartiStok.BrutSevkKilo, PartiStok.NetSevkKilo, PartiStok.BrutSevkMetre, PartiStok.NetSevkMetre,
             DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi from Firma
    left join Parti on Parti.FirmaID = Firma.FirmaId
    left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    where Firma.AnaHesapFirmaId = :AFirmaId and
         (Parti.AcilisTarihi >= :ABaslangicTarihi and Parti.AcilisTarihi <= :ABitisTarihi) and
         (Parti.IslemTipiId = :AIslemTipiId or :AIslemTipiId is null) and
         (Parti.Fason = :ASatisTipi or :ASatisTipi is null) and
         (Parti.PartiTipiId = :APartiTipiId or :APartiTipiId is null) and
         (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
         (PartiStok.PartiStokIptalKayitId is null) and
         ( (PartiStok.Durumu < 8  and :ADurumu = 0) or (PartiStok.Durumu = 8 and :ADurumu = 1) or :ADurumu is null)
    into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisTarihi, :Fason, :PartiKayitId,
         :PartiYil, :PartiId, :AcilisTarihi, :ReceteId, :ReceteAdi, :BoyaSiparisDetayId,
         :PartiSinifId, :HamAdiToplami, :Kilo, :Metre, :Durumu, :DurumuStr, :IslemTipiAdi, :MusteriSiparisNo, :LabReceteKodu, :LabReceteAdi,
         :Kilo, :Metre, :BrutSevkKilo, :NetSevkKilo, :BrutSevkMetre, :NetSevkMetre,
         :DesenId, :DesenAdi, :VaryantAdi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    For select HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi from Partistok
      left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
      left join RafDetay on RafDetay.RafDetayid = PartiStokDetay.RafDetayId
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
      where Partistok.PartiSinifId = :PartiSinifId
      group by HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi
      into :FIrsaliyeNo, :FIrsaliyeTarihi do
    begin
      if (FIrsaliyeNo is not null) THen Str = Format_Float('#0', FIrsaliyeNo);
      else Str = '';

      GirisIrsaliyeNo = '';
      GirisIrsaliyeTarihi = '';

      if (Str <> '') then
      if (PosBul(GirisIrsaliyeNo, Str) = 0) then
      if (Char_Length(GirisIrsaliyeNo) + Char_Length(Str) < 250) then
      begin
        if (GirisIrsaliyeNo <> '') then GirisIrsaliyeNo = GirisIrsaliyeNo ||  cast(', ' as varchar(2) character set UTF8);
        GirisIrsaliyeNo = GirisIrsaliyeNo || Str;

        Str = TimeStampToDateStr(FIrsaliyeTarihi);

        if (GirisIrsaliyeTarihi <> '') then GirisIrsaliyeTarihi = GirisIrsaliyeTarihi ||  cast(', ' as varchar(2) character set UTF8);
        GirisIrsaliyeTarihi = GirisIrsaliyeTarihi || Str;
      end
      suspend;
    end
  end
end
