      else if (BlobSize > 0) then Notu = Left(cast(FNotu as blob character set UTF8), BlobSize);
      else Notu = '';

      if (IrsaliyeNo is not null) then
      begin
        if (EvrakTipi = 1) then IrsaliyeTarihveNo = TimeStampToDateStr(IrsaliyeTarihi) || ' Tarih ' || Format_Float('',IrsaliyeNo) || ' Nolu Irsaliye ';

               LabReceteTalep.HamStokMasterId, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, LabReceteTalep.Notu,
               LabReceteTalep.LabReceteCalismaId, LabRecete.LabReceteKodu, CK.Adi,
               LabReceteTalep.TalepYeriId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, LabReceteTalep.BoyaSiparisDetayId,
               BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.RenkAdi,
               LabReceteTalep.PartiYil, LabReceteTalep.PartiId, LabReceteTalep.CalismaOncelikNo,
               LabReceteTalep.Rptmi, LabReceteTalep.Ilave, LabReceteTalep.SokumUzeriBoya, LabReceteTalep.KritikCalismasi,
               LabReceteTalep.GidisTarihi, LabReceteTalep.RenkGelisTarihi, LabReceteTalep.EvrakNo, LabReceteTalep.ReceteTalepGrup1,
               LabReceteTalep.ReceteTalepGrup2, RTG1.Adi, RTG2.Adi, LabReceteCalisma.CalismaTarihi, SK.Adi, LabReceteTalep.DegisimTarihi, LabReceteTalep.MusteriTalepNo,
               LabReceteTalep.MusteriModelNo, Ham.ReferansKodu, LabReceteAtis.AtisNoStr, LabReceteTalep.AtisOkeyTarihi, LabReceteCalisma.BitisTarihi, LabReceteTalep.Data from LabReceteTalep
    left join Firma on Firma.Firmaid = LabReceteTalep.Firmaid
    left join IsikKaynagi on IsikKaynagi.IsikKaynagiid = LabReceteTalep.IsikKaynagiid
    left join StandartTipi on StandartTipi.StandartTipiid = LabReceteTalep.StandartTipiid
    left join Ham on Ham.HamId = LabReceteTalep.HamId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = LabReceteTalep.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
    left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteTalep.LabReceteCalismaId
    left join LabRecete on LabRecete.LabReceteId = LabReceteCalisma.LabReceteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = LabReceteTalep.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join ReceteTalepGrup RTG1 on RTG1.ReceteTalepGrupId = LabReceteTalep.ReceteTalepGrup1
    left join ReceteTalepGrup RTG2 on RTG2.ReceteTalepGrupId = LabReceteTalep.ReceteTalepGrup2
    Left Join Kullanici CK on CK.KullaniciId = LabReceteCalisma.InsertKullaniciId
    Left Join Kullanici TEK on TEK.KullaniciId = LabReceteTalep.TalepEdenKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = LabReceteTalep.KullaniciId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = LabReceteTalep.OkeyliLabReceteAtisId
    where LabReceteTalep.TalepTarihi >= :ABaslangicTarihi and LabReceteTalep.TalepTarihi <= :ABitisTarihi and
         (LabreceteTalep.EvrakNo = :AEvrakNo or :AEvrakNo is null) and
         (LabreceteTalep.ReceteTalepGrup1 = :AReceteTalepGrup1 or :AReceteTalepGrup1 is null) and
         (LabreceteTalep.ReceteTalepGrup2 = :AReceteTalepGrup2 or :AReceteTalepGrup2 is null) and
         (LabreceteTalep.KritikCalismasi = :AKritikCalismasi or :AKritikCalismasi is null) and
         (LabreceteTalep.Rptmi = :ARpt or :ARpt is null) and
          (LabReceteTalep.FirmaId = :AFirmaId or :AFirmaId is null)
    order by LabReceteTalep.LabReceteTalepYil, LabReceteTalep.LabReceteTalepId

         :SipLabReceteKodu, :SipRenkAdi, :PartiYil, :PartiId, :CalismaOncelikNo, :Rptmi, :Ilave, :SokumUzeriBoya, :KritikCalismasi,
         :GidisTarihi, :RenkGelisTarihi, :EvrakNo, :ReceteTalepGrup1, :ReceteTalepGrup2, :ReceteTalepGrup1Adi, :ReceteTalepGrup2Adi,
         :CalismaTarihi, :KullaniciAdi, :DegisimTarihi, :MusteriTalepNo, :MusteriModelNo, :ReferansKodu, :AtisNoStr, :AtisOkeyTarihi, :CalismaBitisTarihi, :Data do
    begin
      LabReceteTalepIdStr = '(' || EgemenIntegerToStr(LabReceteTalepYil)  || ') ' || EgemenIntegerToStr(LabReceteTalepId) || ' ';

      if (Rptmi = 1) then TipStr = 'Rpt';
      else if (KritikCalismasi = 1) then TipStr = 'Kritik';
      else if (Ilave = 1) then TipStr = 'İlave';
      else if (SokumUzeriBoya = 1) then TipStr = 'Söküm Üzeri Boya';
      else TipStr = '';

      BlobSize = char_length(FNotu);

      if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
      else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
      else Notu = '';

      if (IrsaliyeNo is not null) then
      begin
        if (EvrakTipi = 1) then IrsaliyeTarihveNo = TimeStampToDateStr(IrsaliyeTarihi) || ' Tarih ' || Format_Float('',IrsaliyeNo) || ' Nolu Irsaliye ';
        else IrsaliyeTarihveNo = TimeStampToDateStr(IrsaliyeTarihi) || ' Tarih ' || Format_Float('',IrsaliyeNo) || ' Nolu Fiş ';
      end
      else IrsaliyeTarihveNo = '';

      if (TalepYeriId = 0) then TalepYeriIdStr = 'Teklif';
      else if (TalepYeriId = 1) then TalepYeriIdStr = 'Sipariş';
      else if (TalepYeriId = 2) then TalepYeriIdStr = 'Üretim';
      else TalepYeriIdStr = '';

      if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      else PartiIdStr = '';

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      else BoyaSiparisIdStr = '';

      suspend;
    end
  end
