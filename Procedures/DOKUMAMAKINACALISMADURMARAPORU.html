CREATE PROCEDURE DOKUMAMAKINACALISMADURMARAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APERSONELIDLER VARCHAR(255),
    AMAKINAGRUPID MAKINAGRUPKODUSIZE,
    ADOKUMAMAKINAIDLER VARCHAR(255),
    AHAMID INTEGER)
RETURNS (
    BASLAMATARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    DOKUMAMAKINAID MAKINAKODUSIZE,
    CALISIYORSTR VARCHAR(20),
    DURUSSEBEBI ADI50,
    URETIMPARTINO VARCHAR(30),
    HAMADI ADI255,
    SURE DOUBLE PRECISION,
    SURESTR VARCHAR(30),
    SERVISCIADI ADI40,
    SERVISCIGELDITARIHI TIMESTAMP,
    SERVISCIGITTITARIHI TIMESTAMP,
    VARDIYATARIHI TIMESTAMP,
    VARDIYANO INTEGER,
    TOPLAMTUR INTEGER,
    ATMASIGEREKENTOPLAMTUR INTEGER)
AS
declare variable LeventSonu integer;
declare variable AtkiKoptu  integer;
declare variable CozguKoptu integer;
declare variable StartStop  integer;
declare variable ImalataAraVerildi integer;
declare variable AraVermeSebebiId integer;
declare variable Tarih TimeStamp;

declare variable SonTarih timestamp;
declare variable Durum integer;
declare variable SonDurum integer;

declare variable TmpDokumaMakinaId  MakinaKoduSize;
declare variable TmpUretimPartiNo   varchar(30);
declare variable TmpHamAdi          Adi255;
declare variable TmpCalisiyorStr    varchar(20);
declare variable TmpDurusSebebi     Adi50;
declare variable TmpVardiyaTarihi   TimeStamp;
declare variable TmpVardiyaNo       integer;

declare variable SonDokumaMakinaId  MakinaKoduSize;
declare variable SonUretimPartiNo   varchar(30);
declare variable SonHamAdi          Adi255;
declare variable SonCalisiyorStr    varchar(20);
declare variable SonDurusSebebi     Adi50;
declare variable SonVardiyaTarihi   TimeStamp;
declare variable SonVardiyaNo       integer;

declare variable ServisciGeldi      integer;
declare variable OrmeMakinaDurumuId integer;
declare variable Tur                integer;
declare variable AtmasiGerekenTur   integer;

declare variable SonTur1            integer;
declare variable SonTur2            integer;
declare variable SonTur3            integer;
declare variable SonTur4            integer;
declare variable SonTur5            integer;
declare variable SonTur6            integer;
declare variable SonTur7            integer;
declare variable SonTur8            integer;
declare variable SonTur9            integer;
declare variable SonTur10           integer;
declare variable SonTurSayisi       integer;

begin
  if (ABaslangicTarihi > current_timestamp) then exit;

  Sure = 0;
  SonDurum = null;
  SonDokumaMakinaId = null;
  ToplamTur = 0;
  SonTurSayisi = 0;
  AtmasiGerekenToplamTur = 0;

  for select OrmeMakinaDurumu.Tarih, DokumaMakina.DokumaMakinaId, LeventSonu, AtkiKoptu, CozguKoptu, StartStop, OrmeMakinaDurumu.ImalataAraVerildi, DokumaPlanlama.AraVermeSebebiId,
  DokumaPlanlama.UretimPartiNo, Ham.Adi, OrmeMakinaDurumu.ServisciGeldi, OrmeMakinaDurumu.OrmeMakinaDurumuId, OrmeMakinaDurumu.Tur
      from OrmeMakinaDurumu
      left join Personel on Personel.PersonelId = OrmeMakinaDurumu.PersonelId
      left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = OrmeMakinaDurumu.DokumaPlanlamaId
      left join DokumaMakina on DokumaMakina.DokumaMakinaId = OrmeMakinaDurumu.DokumaMakinaId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
      left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
      where OrmeMakinaDurumu.Tarih >= :ABaslangicTarihi and OrmeMakinaDurumu.Tarih <= :ABitisTarihi and
      (:APersonelIdler is null or position('-'||OrmeMakinaDurumu.PersonelId||'-', :APersonelIdler) > 0) and
      (:ADokumaMakinaIdler is null or position('-'||OrmeMakinaDurumu.DokumaMakinaId||'-', :ADokumaMakinaIdler) > 0) and
      (:AMakinaGrupId is null or DokumaMakina.MakinaGrupId = :AMakinaGrupId) and
      (:AHamId is null or  OrmeSiparisDetay.HamId = :AHamId)
      order by OrmeMakinaDurumu.DokumaMakinaId, OrmeMakinaDurumu.Tarih
      into :Tarih, :DokumaMakinaId, :LeventSonu, :AtkiKoptu, :CozguKoptu, :StartStop, :ImalataAraVerildi, :AraVermeSebebiId, :UretimPartiNo, :HamAdi, :ServisciGeldi, :OrmeMakinaDurumuId, :Tur do
  begin
    if (BaslamaTarihi is null) then BaslamaTarihi = Tarih;
    select VardiyaTarihi, VardiyaNo from DokumaVardiyaHesapla(:Tarih) into :VardiyaTarihi, :VardiyaNo;

    if (ServisciGeldi = 1) then
    begin
      select Personel.Adi, OrmeMakinaDurumu.Tarih from OrmeArizaBakim
      left join Personel on Personel.PersonelId = OrmeArizaBakim.PersonelId
      left join OrmeMakinaDurumu on OrmeMakinaDurumu.OrmeMakinaDurumuId = OrmeArizaBakim.GittiMakinaDurumuId
      where OrmeArizaBakim.GeldiMakinaDurumuId = :OrmeMakinaDurumuId into :ServisciAdi, :ServisciGittiTarihi;

      ServisciGeldiTarihi = Tarih;
    end

    if (LeventSonu = 0 and AtkiKoptu = 0 and CozguKoptu = 0 and StartStop = 0 and ImalataAraVerildi = 0) then
    begin
      CalisiyorStr = 'Çalıştı';
      DurusSebebi = '';
      Durum = 1;
    end
    else if (LeventSonu = 1) then
    begin
      CalisiyorStr = 'Durdu';
      DurusSebebi = 'Levent Sonu';
      Durum = 2;
    end
    else if (AtkiKoptu = 1) then
    begin
      CalisiyorStr = 'Durdu';
      DurusSebebi = 'Atkı Koptu';
      Durum = 3;
    end
    else if (CozguKoptu = 1) then
    begin
      CalisiyorStr = 'Durdu';
      DurusSebebi = 'Çözgü Koptu';
      Durum = 4;
    end
    else if (StartStop = 1) then
    begin
      CalisiyorStr = 'Durdu';
      DurusSebebi = 'Start/Stop';
      Durum = 5;
    end
    else if (ImalataAraVerildi = 1) then
    begin
      CalisiyorStr = 'Ara Verildi';
      Durum = 6;
      select AraVermeSebebi.Adi from AraVermeSebebi where AraVermeSebebiId = :AraVermeSebebiId into :DurusSebebi;
    end

    if ((Durum = 1 and SonTurSayisi = 10 and Tur > 0) or (Durum = 1 and SonTurSayisi < 10) ) then
    begin
      /*Makinanin atmasi gereken turu hesapla baslangici*/
      SonTurSayisi = SonTurSayisi + 1;
      if (SonTurSayisi > 10) then SonTurSayisi = 10;

      SonTur10 = Coalesce(SonTur9, 0);
      SonTur9 = Coalesce(SonTur8, 0);
      SonTur8 = Coalesce(SonTur7, 0);
      SonTur7 = Coalesce(SonTur6, 0);
      SonTur6 = Coalesce(SonTur5, 0);
      SonTur5 = Coalesce(SonTur4, 0);
      SonTur4 = Coalesce(SonTur3, 0);
      SonTur3 = Coalesce(SonTur2, 0);
      SonTur2 = Coalesce(SonTur1, 0);
      SonTur1 = Tur;

      AtmasiGerekenTur = Round((SonTur1 + SonTur2 + SonTur3 + SonTur4 + SonTur5 + SonTur6 + SonTur7 + SonTur8 + SonTur9 + SonTur10) / SonTurSayisi);
      /*Makinanin atmasi gereken turu hesapla sonu*/
    end

    if (SonDurum is not null and SonDurum <> Durum or SonDokumaMakinaId is not null and SonDokumaMakinaId <> DokumaMakinaId or SonVardiyaNo is not null and SonVardiyaNo <> VardiyaNo)  then
    begin
      if (SonDokumaMakinaId is not null and SonDokumaMakinaId <> DokumaMakinaId) then
      begin
        SonTurSayisi = 0;
        SonTur1 = 0;
        SonTur2 = 0;
        SonTur3 = 0;
        SonTur4 = 0;
        SonTur5 = 0;
        SonTur6 = 0;
        SonTur7 = 0;
        SonTur8 = 0;
        SonTur9 = 0;
        SonTur10 = 0;
      end
      TmpDokumaMakinaId = DokumaMakinaId;
      TmpUretimPartiNo = UretimPartiNo;
      TmpHamAdi = HamAdi;
      TmpCalisiyorStr = CalisiyorStr;
      TmpDurusSebebi = DurusSebebi;
      TmpVardiyaTarihi = VardiyaTarihi;
      TmpVardiyaNo = VardiyaNo;

      DokumaMakinaId = SonDokumaMakinaId;
      UretimPartiNo = SonUretimPartiNo;
      HamAdi = SonHamAdi;
      CalisiyorStr = SonCalisiyorStr;
      DurusSebebi = SonDurusSebebi;
      VardiyaTarihi = SonVardiyaTarihi;
      VardiyaNo = SonVardiyaNo;

      BitisTarihi = SonTarih;
      if (Sure = 0) then
        Sure = 0.0006944444467;
      SureStr = TarihToStr(Sure);

      if (SonDurum = 1 and Durum <> 1) then
      begin
        ToplamTur = ToplamTur + Tur;
        if ((SonTarih is not null) and (TimeStampToDouble(Tarih) - TimeStampToDouble(SonTarih) > 0) and (TimeStampToDouble(Tarih) - TimeStampToDouble( SonTarih) < 0.0006944444467)) then
          AtmasiGerekenToplamTur = AtmasiGerekenToplamTur + ((TimeStampToDouble(Tarih) - TimeStampToDouble( SonTarih)) * 24 * 60) * AtmasiGerekenTur;
        else
          AtmasiGerekenToplamTur = AtmasiGerekenToplamTur + AtmasiGerekenTur;
      end

      suspend;

      Sure = 0;
      SureStr = Null;
      BitisTarihi = Null;
      DokumaMakinaId = TmpDokumaMakinaId;
      UretimPartiNo = TmpUretimPartiNo;
      HamAdi = TmpHamAdi;
      CalisiyorStr = TmpCalisiyorStr;
      DurusSebebi = TmpDurusSebebi;
      ServisciGeldiTarihi = null;
      ServisciGittiTarihi = null;
      ServisciAdi = null;
      if (SonDokumaMakinaId is not null and SonDokumaMakinaId <> DokumaMakinaId) then
        BaslamaTarihi = Tarih;
      else
        BaslamaTarihi = SonTarih;

      SonTarih = null;
      VardiyaTarihi = TmpVardiyaTarihi;
      VardiyaNo = TmpVardiyaNo;
      ToplamTur = 0;
      AtmasiGerekenToplamTur = 0;
    end

    if (SonTarih is not null) then
      Sure = Sure + (Tarih - SonTarih);
    else
      Sure = Sure + 0.0006944444467;

    if (SonDurum = 1 and Durum <> 1) then
    begin
      ToplamTur = ToplamTur;
      AtmasiGerekenToplamTur = AtmasiGerekenToplamTur;
    end
    else
    begin
      ToplamTur = ToplamTur + Tur;
      if ((SonTarih is not null) and (TimeStampToDouble(Tarih) - TimeStampToDouble(SonTarih) > 0) and (TimeStampToDouble(Tarih) - TimeStampToDouble( SonTarih) < 0.0006944444467)) then
        AtmasiGerekenToplamTur = AtmasiGerekenToplamTur + ((TimeStampToDouble(Tarih) - TimeStampToDouble( SonTarih)) * 24 * 60) * AtmasiGerekenTur;
      else
        AtmasiGerekenToplamTur = AtmasiGerekenToplamTur + AtmasiGerekenTur;
    end

    SonTarih = Tarih;
    SonDurum = Durum;
    SonDokumaMakinaId = DokumaMakinaId;
    SonUretimPartiNo = UretimPartiNo;
    SonHamAdi = HamAdi;
    SonCalisiyorStr = CalisiyorStr;
    SonDurusSebebi = DurusSebebi;
    SonVardiyaTarihi = VardiyaTarihi;
    SonVardiyaNo = VardiyaNo;
  end

  BitisTarihi = SonTarih;
  if (Sure = 0) then
    Sure = 0.0006944444467;
  SureStr = TarihToStr(Sure);
  select VardiyaTarihi, VardiyaNo from DokumaVardiyaHesapla(:BaslamaTarihi) into :VardiyaTarihi, :VardiyaNo;

  suspend;
end
