CREATE PROCEDURE LABRECETEATISRENKLISTESI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTALEPTARIHI TIMESTAMP,
    ABITISTALEPTARIHI TIMESTAMP,
    ALABRECETEGRUPID INTEGER,
    AGRUPIPLIKBOYAMI INTEGER,
    ALABRENKGRUPID INTEGER,
    AFASON INTEGER,
    AESKILABRECETEKODU VARCHAR(20),
    APANTONEKODU VARCHAR(20),
    AHAMID INTEGER,
    AKALITEID VARCHAR(20),
    AURUNFIYATGOREBILIR SMALLINT)
RETURNS (
    LABRECETEATISRENKID INTEGER,
    LABRECETEATISID INTEGER,
    LABRECETEID INTEGER,
    SIRA INTEGER,
    SIRANOSTR VARCHAR(25),
    ATISNOSTR VARCHAR(20),
    RENKKODU VARCHAR(50),
    RENKADI VARCHAR(120),
    COLOR INTEGER,
    INSERTKULLANICIID INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select LabReceteAtisRenk.LabReceteAtisRenkid, LabReceteAtisRenk.labreceteatisid, LabReceteAtisRenk.labreceteid,
             LabReceteAtisRenk.sira, LabReceteAtisRenk.renkkodu, LabReceteAtisRenk.renkadi, LabReceteAtisRenk.color,
             LabReceteAtisRenk.insertkullaniciid, LabReceteAtisRenk.inserttarihi, LabReceteAtisRenk.kullaniciid,
             LabReceteAtisRenk.degisimtarihi, IK.Adi, SK.Adi, LabReceteAtisListesi.AtisNoStr from
             LabReceteAtisListesi(:AFirmaId, :ABaslangicTalepTarihi, :ABitisTalepTarihi, :ALabReceteGrupid, :AGrupIplikBoyami, :ALabRenkGrupid, :AFason, :AEskiLabReceteKodu, :APantoneKodu, :AHamId, :AKaliteId, :AUrunFiyatGorebilir, null, null, null)
    Left Join LabReceteAtisRenk on LabReceteAtisRenk.LabReceteAtisId = LabReceteAtisListesi.LabReceteAtisId
    Left Join Kullanici IK on (IK.KullaniciID = LabreceteAtisRenk.InsertKullaniciID)
    Left Join Kullanici SK on (SK.KullaniciID = LabreceteAtisRenk.KullaniciID)
    Order By LabReceteAtisRenk.Sira
    into :LabReceteAtisRenkid, :LabReceteAtisid, :LabReceteid, :Sira, :Renkkodu, :Renkadi, :Color, :InsertKullaniciID,
         :InsertTarihi, :KullaniciID, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi, :AtisNoStr
  do
  begin
    SiraNoStr = '';
    SiraNoStr = AtisNoStr || '.' || Sira;
    suspend;
  end
end
