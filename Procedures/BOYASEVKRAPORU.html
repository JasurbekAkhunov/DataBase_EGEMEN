CREATE PROCEDURE BOYASEVKRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    RAPORHAMTIPIID INTEGER,
    KILO NUMERIC(12,3),
    UCRETLITAMIRKILO NUMERIC(12,3),
    UCRETSIZTAMIRKILO NUMERIC(12,3),
    HAMSATISKILO NUMERIC(12,3),
    HAMIADEKILO NUMERIC(12,3),
    TOPLAMKILO NUMERIC(12,3))
AS
declare variable FFirmaId       varchar(20);
declare variable FFirmaAdi      varchar(100);
declare variable FHamTipiId     integer;
declare variable FHamTipiAdi    varchar(20);
declare variable PartiKayitId   integer;
declare variable FasonHamIade   integer;
declare variable IslemTipiId    integer;
declare variable FaturaKesilecek integer;
declare variable FKilo           Numeric(12,3);
declare variable SonFirmaId     varchar(20);
declare variable SonHamTipiId   integer;
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;

begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;
  SonFirmaId = '';
  SonHamTipiId = -1;

  for select BoyaIrsaliye.Firmaid, Firma.Adi, Ham.HamTipiId, HamTipi.Adi, Parti.IslemTipiId, Parti.FaturaKesilecek, BoyaIrsaliyeDetay.PartiKayitId, BoyaIrsaliyeDetay.FasonHamIade, BoyaIrsaliyeDetay.Kilo from BoyaIrsaliye
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
    left join Firma on Firma.Firmaid = BoyaIrsaliye.Firmaid
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
    left join Parti on Parti.PartiKayitid = BoyaIrsaliyeDetay.PartiKayitId
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
          BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID is not null and BoyaIrsaliye.HareketTipiID = 0
    order by BoyaIrsaliye.Firmaid, Ham.HamTipiId
    into :FFirmaId, :FFirmaAdi, :FHamTipiId, :FHamTipiAdi, :IslemTipiId, :FaturaKesilecek, :PartiKayitId, :FasonHamIade, :FKilo do
  begin
    if (SonFirmaId <> FFirmaId or SonHamTipiId <> :FHamTipiId) then
    begin
      if (SonFirmaId <> '') then
      begin
        ToplamKilo = Kilo + UCretliTamirKilo + UcretsizTamirKilo + HamIadeKilo + HamSatisKilo;

        if (HamTipiID = hstIplik or HamTipiID = hstElyaf or HamTipiID = hstTelef) then RaporHamTipiId = 0;
        else if (HamTipiID = hstOrmeKumas or HamTipiID = hstRibana or HamTipiID = hstAksesuar) then RaporHamTipiId = 1;
        else if (HamTipiID = hstDokumaKumas) then RaporHamTipiId = 2;
        else RaporHamTipiId = -1;

        suspend;
      end
      SonFirmaId = FFirmaId;
      SonHamTipiId = FHamTipiId;
      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      HamTipiId = FHamTipiId;
      HamTipiAdi = FHamTipiAdi;
      Kilo = 0;
      UcretliTamirKilo      = 0;
      UcretsizTamirKilo     = 0;
      HamSatisKilo = 0;
      HamIadeKilo = 0;
      ToplamKilo = 0;
    end

    if (PartiKayitID is not null) then
    begin
      if (IslemTipiId <> 3) then
      begin
        Kilo = Kilo + FKilo;
      end
      else
      begin
        if (FaturaKesilecek = 0) then
          UcretsizTamirKilo = UcretsizTamirKilo + FKilo;
        else
          UcretliTamirKilo = UcretliTamirKilo + FKilo;
      end
    end
    else if (FasonHamIade = 1) then
      HamIadeKilo = HamIadeKilo + FKilo;
    else
      HamSatisKilo = HamSatisKilo + FKilo;
  end

  if (SonFirmaId <> '') then
  begin
    ToplamKilo = Kilo + UCretliTamirKilo + UcretsizTamirKilo + HamIadeKilo + HamSatisKilo;


    if (HamTipiID = hstIplik or HamTipiID = hstElyaf or HamTipiID = hstTelef) then RaporHamTipiId = 0;
    else if (HamTipiID = hstOrmeKumas or HamTipiID = hstRibana or HamTipiID = hstAksesuar) then RaporHamTipiId = 1;
    else if (HamTipiID = hstDokumaKumas) then RaporHamTipiId = 2;
    else RaporHamTipiId = -1;
    suspend;
  end
end
