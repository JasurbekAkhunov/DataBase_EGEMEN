CREATE PROCEDURE BOYACIRORAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASERINO SERINOSIZE5,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICFIRMAGRUP1ID VARCHAR(4),
    ABITISFIRMAGRUP1ID VARCHAR(4),
    ABASLANGICFIRMAGRUP2ID VARCHAR(4),
    ABITISFIRMAGRUP2ID VARCHAR(4),
    ABASLANGICFIRMAGRUP3ID VARCHAR(4),
    ABITISFIRMAGRUP3ID VARCHAR(4),
    ABASLANGICFATURAGRUPID VARCHAR(4),
    ABITISFATURAGRUPID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    AFATURAGRUPARALIKLI INTEGER,
    ATEVKIFAT INTEGER,
    ARAPORDOVIZID VARCHAR(5),
    AFIRMAADI2YIGOREBILIR INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    FIRMAGRUP1 VARCHAR(4),
    FIRMAGRUP2 VARCHAR(4),
    FIRMAGRUP3 VARCHAR(4),
    BRUTTOPLAM DOUBLE PRECISION,
    ISKONTOTOPLAMI DOUBLE PRECISION,
    ISKONTO1TUTARI DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    FATURATUTARI DOUBLE PRECISION)
AS
declare variable Okey integer;
declare variable Tarih  TimeStamp;
declare variable DovizKuru Double Precision;
declare variable FaturaDovizId  varchar(5);
declare variable FFirmaId       varchar(20);
declare variable FFirmaAdi      varchar(100);
declare variable FFirmaAdi2     varchar(100);
declare variable FFirmaGrup1            varchar(4);
declare variable FFirmaGrup2            varchar(4);
declare variable FFirmaGrup3            varchar(4);
declare variable FFaturaGrupId          varchar(4);
declare variable FLastFirmaId   varchar(20);
declare variable FBrutToplam            Double Precision;
declare variable FIskontoToplami        Double PRecision;
declare variable FIskonto1Tutari        Double PRecision;
declare variable FIndirilecekKdv        double Precision;
declare variable FTahsilEdilecekKdv     double Precision;
declare variable FFaturaTutari          double Precision;
begin
  FLastFirmaId = '';
  For Select BoyaFatura.FaturaTarihi, BoyaFatura.FaturaGrupId, BoyaFatura.FirmaId, Firma.Adi, Firma.Adi2, Firma.FirmaGrup1, Firma.FirmaGrup2, Firma.FirmaGrup3, BoyaFatura.DovizId,
      BoyaFatura.BrutToplam, BoyaFatura.ParasalIskonto, BoyaFatura.IskontoToplami, BoyaFatura.IndirilecekKdv,
      BoyaFatura.TahsilEdilecekKdv from BoyaFatura
    left join Firma on Firma.FirmaId = BoyaFatura.FirmaId
  where BoyaFatura.FaturaTarihi >= :ABaslangicTarihi and BoyaFatura.FaturaTarihi <= :ABitisTarihi and BoyaFatura.HareketTipiId = 0 and
       (BoyaFatura.FaturaSeriNo = :ASeriNo or :ASeriNo is null) and
       ((:AFirmaAralikli = 1 and (BoyaFatura.FirmaId >= :ABaslangicFirmaId or :ABaslangicFirmaId = '') and (BoyaFatura.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
       ((:AFaturaGrupAralikli = 1 and (BoyaFatura.FaturaGrupId >= :ABaslangicFaturaGrupId or :ABaslangicFaturaGrupId = '') and (BoyaFatura.FaturaGrupId <= :ABitisFaturaGrupId or :ABitisFaturaGrupId = '')) or (:AFaturaGrupAralikli = 0)) and
       ((:AFirmaGrup1Aralikli = 1 and (Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id or :ABaslangicFirmaGrup1Id = '') and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
       ((:AFirmaGrup2Aralikli = 1 and (Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id or :ABaslangicFirmaGrup2Id = '') and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
       ((:AFirmaGrup3Aralikli = 1 and (Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id or :ABaslangicFirmaGrup3Id = '') and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = '')) and
       (BoyaFatura.Entegrasyon = 1) and (BoyaFatura.Tevkifat = :ATevkifat or :ATevkifat is null)
       order by BoyaFatura.FirmaId
  into :Tarih, :FFaturaGrupId, :FFirmaId, :FFirmaAdi,:FFirmaAdi2, :FFirmaGrup1, :FFirmaGrup2, :FFirmaGrup3,
       :FaturaDovizId, :FBrutToplam, :FIskonto1Tutari, :FIskontoToplami, :FIndirilecekKdv, :FTahsilEdilecekKdv do
  begin
    if (AFirmaAdi2yiGorebilir = 0) then FFirmaAdi2 = NUll;

    if (FFirmaId <> FLastFirmaId) then
    begin
      if (FLastFirmaId <> '') then
      if (Okey = 1) then suspend;

      FLastFirmaId = FFirmaId;
      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      FirmaAdi2 = FFirmaAdi2;
      FirmaGrup1 = FFirmaGrup1;
      FirmaGrup2 = FFirmaGrup2;
      FirmaGrup3 = FFirmaGrup3;
      BrutToplam            = 0;
      IskontoToplami        = 0;
      Iskonto1Tutari        = 0;
      IndirilecekKdv        = 0;
      TahsilEdilecekKdv     = 0;
      FaturaTutari          = 0;

      Okey = 1;
      if (AFirmaAralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaId, FFirmaId) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup1Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup1Id, FFirmaGrup1) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup2Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup2Id, FFirmaGrup2) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup3Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup3Id, FFirmaGrup3) = 0) then okey = 0;

      if (okey = 1) then
      if (AFaturaGrupAralikli = 0) then
      if (KutuKarsilastir(ABaslangicFaturaGrupId, FFaturaGrupId) = 0) then okey = 0;
    end

    if (okey = 1) then
    begin
      if (FaturaDovizId <> ARaporDovizId) then
      begin
        select DovizKuru from EnYakinDovizKuruGetir(:FaturaDovizId, :ARaporDovizId, :Tarih) into :DovizKuru;
      end
      else DovizKuru = 1;

      FBrutToplam = FBrutToplam * DovizKuru;
      FIskontoToplami = FIskontoToplami * DovizKuru;
      FIndirilecekKdv = FIndirilecekKdv * DovizKuru;
      FTahsilEdilecekKdv = FTahsilEdilecekKdv * DovizKuru;

      FFaturaTutari = FBrutToplam - FIskontoToplami + FTahsilEdilecekKdv;

      BrutToplam            = BrutToplam + FBrutToplam;
      IskontoToplami        = IskontoToplami + FIskontoToplami;
      Iskonto1Tutari        = Iskonto1Tutari + FIskonto1Tutari;
      IndirilecekKdv        = IndirilecekKdv + FIndirilecekKdv;
      TahsilEdilecekKdv     = TahsilEdilecekKdv + FTahsilEdilecekKdv;
      FaturaTutari          = FaturaTutari + FFaturaTutari;
    end
  end
  if (FLastFirmaId <> '') then
  if (Okey = 1) then suspend;
/*

*/
end
