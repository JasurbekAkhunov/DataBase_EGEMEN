CREATE PROCEDURE PARTIILAVEISLEMLERGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIASAMALARIID INTEGER,
    ILAVEISLEMLERID INTEGER,
    ILAVEISLEMLERADI VARCHAR(30))
AS
begin
  For Select PartiIlaveIslemler.PartiAsamalariId, PartiIlaveIslemler.IlaveIslemlerId, IlaveIslemler.Adi from PartiIlaveIslemler
    left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
    where PartiKayitId = :APartiKayitId
    into :PartiAsamalariId, :IlaveIslemlerId, :IlaveIslemlerAdi do suspend;
end
