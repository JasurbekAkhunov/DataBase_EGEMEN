CREATE PROCEDURE MAKINAIPLIKDURUMUHAZIRLA(
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICMAKINAID VARCHAR(10),
    ABITISMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4),
    AMAKINAARALIKLI INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ORMESIPARISDETAYID INTEGER,
    ORMEPLANLAMAID INTEGER,
    URUNADI VARCHAR(255),
    URETIMPARTINO VARCHAR(30),
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIIDSTR VARCHAR(20),
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    KULLANIMORANI INTEGER,
    ORMEMAKINAID VARCHAR(10),
    PLANLANANMIKTAR DOUBLE PRECISION,
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    KULLANIMMIKTARI DOUBLE PRECISION,
    TAHMINIFIREMIKTARI DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    TAHMINIFIREORANI DOUBLE PRECISION)
AS
declare variable TalimatTahminiFireMiktari    Double precision;
declare Variable Okey                         integer;
declare variable hstIplik       integer;
declare variable hstLycra       integer;
begin
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  For Select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.OrmeSiparisDetayId, OrmePlanlama.OrmePlanlamaId,
             OrmePlanlama.UretimPartiNo, OrmePlanlama.OrmeMakinaId, ReceteHam.HamTipiId, HamTipi.Adi, ReceteHam.Adi, OrmePlanlama.PlanlananMiktar, OrmePlanlama.GerceklesenMiktar, OrmePlanlamaRecete.HamId,
             OrmePlanlamaRecete.RenkId, Renk.Adi, OrmePlanlamaRecete.Lot, OrmePlanlamaRecete.UreticiId, Uretici.Adi,
             KullanimOrani, CikisMiktari, IadeMiktari,
             OrmePlanlamaRecete.TahminiFireOrani, SipHam.Adi From BoyaSiparis
     left join OrmeSiparisDetay on OrmeSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
     left join Ham SipHam on SipHam.HamId = OrmeSiparisDetay.HamId
     left join OrmePlanlama on OrmePlanlama.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId and OrmePlanlama.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
     left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
     left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = OrmePlanlama.OrmePlanlamaId
     left join Ham ReceteHam on ReceteHam.HamId = OrmePlanlamaRecete.HamId
     left join Renk on Renk.RenkId = OrmePlanlamaRecete.RenkId
     left join Uretici on Uretici.UreticiId = OrmePlanlamaRecete.UreticiId
     left join HamTipi on HamTipi.HamTipiId = ReceteHam.HamTipiId
  where (BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
        (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil  and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId)) and
        OrmeSiparisDetay.OrmeSiparisDetayId is not null and OrmePlanlama.OrmePlanlamaId is not null and OrmeSiparisDetay.Kapali = 0 and
        OrmePlanlama.Bitti = 0 and
        OrmePlanlama.OrmeMakinaId is not null and OrmePlanlamaRecete.OrmePlanlamaReceteId is not null and
        (OrmeMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
        ((:AMakinaAralikli = 1 and OrmePlanlama.OrmeMakinaId >= :ABaslangicMakinaId and (OrmePlanlama.OrmeMakinaId <= :ABitisMakinaId or :ABitisMakinaId = '')) or (:AMakinaAralikli = 0))
  order by OrmePlanlama.OrmeMakinaId, OrmePlanlama.Tarih, OrmePlanlama.OrmePlanlamaId
  into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :OrmeSiparisDetayId, :OrmePlanlamaId, :UretimPartiNo, :OrmeMakinaId,
       :HamTipiId, :HamTipiIdStr, :HamAdi, :PlanlananMiktar, :GerceklesenMiktar, :HamId, :RenkId, :RenkAdi, :Lot, :UreticiId, :UreticiAdi,
       :KullanimOrani, :CikisMiktari, :IadeMiktari, :TahminiFireOrani, :UrunAdi do
  begin
    Okey = 1;
    if (AMakinaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicMakinaId, OrmeMakinaId) = 0) then okey = 0;

    if (okey = 1) then
    Begin
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      TalimatTahminiFireMiktari = 0;
      TahminiFireMiktari = 0;

      if (hamTipiId = hstIplik) then
      begin
        TalimatTahminiFireMiktari = GerceklesenMiktar * TahminiFireOrani / 100;
        TahminiFireMiktari = TalimatTahminiFireMiktari * KullanimOrani / 100;
        KullanimMiktari = GerceklesenMiktar * KullanimOrani / 100 + TahminiFireMiktari;
        KalanMiktar = CikisMiktari - IadeMiktari - KullanimMiktari;
      end
      else if (HamTipiId = hstLycra) then
      begin
        KullanimMiktari = GerceklesenMiktar * KullanimOrani / 100;
        KalanMiktar = CikisMiktari - IadeMiktari - KullanimMiktari;
      end
      suspend;
    end
  end
end
