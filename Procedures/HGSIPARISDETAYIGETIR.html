CREATE PROCEDURE HGSIPARISDETAYIGETIR(
    AHGSIPARISKAYITID INTEGER)
RETURNS (
    HGSIPARISKAYITID INTEGER,
    HGSIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    SINIF VARCHAR(1),
    MODELID MODELKODUSIZE,
    MODELADI ADI40,
    SATISFIYATI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    TESLIMTARIHI TIMESTAMP,
    VADEGUN INTEGER,
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(40),
    KAPALI INTEGER,
    KAPALISTR VARCHAR(4),
    KAPATMATARIHI TIMESTAMP,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIMODELNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    RESIMDOSYAADI1 VARCHAR(255),
    RESIMDOSYAYAZDIR1 INTEGER,
    RESIMDOSYAADI2 VARCHAR(255),
    RESIMDOSYAYAZDIR2 INTEGER,
    MODELRESIMDOSYAADI1 VARCHAR(255),
    MODELRESIMDOSYAADI2 VARCHAR(255),
    DEGISIMLOG BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SISTEMDOVIZID VARCHAR(5),
    KUMASSATINALMATALEPKAYITID INTEGER,
    KUMASSATINALMATALEPYIL INTEGER,
    KUMASSATINALMATALEPID INTEGER,
    AKSESUARSATINALMATALEPKAYITID INTEGER,
    AKSESUARSATINALMATALEPYIL INTEGER,
    AKSESUARSATINALMATALEPID INTEGER,
    INSERTKULLANICIID INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYLAYANKULLANICIIMZA BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable DovizKuru                     Double Precision;
declare variable SiparisTarihi                  TimeStamp;
declare variable OrmeFirmaKodumuz Varchar(20);
begin
  Select SistemDovizID, OrmeFirmaKodumuz from Sistem where SistemID = 1 into :SistemDovizID, :OrmeFirmaKodumuz;
  Select SiparisTarihi from HGSiparis where HGSiparis.HGSiparisKayitID = :AHGSiparisKayitID into :SiparisTarihi;

  For Select HGSiparisDetay.HGSiparisKayitId, HGSiparisDetay.HGSiparisDetayId, HGSiparisDetay.SiparisSiraNo, HGSiparisDetay.SatisFiyati, HGSiparisDetay.DovizId,
     HGSiparisDetay.Teslimtarihi, HGSiparisDetay.VadeGun, HGSiparisDetay.OdemeSekliId, OdemeSekli.Adi, HGSiparisDetay.MusteriSiparisNo, HGSiparisDetay.MusteriModelNo,
     HGSiparisDetay.ReferansSiparisNo, HGSiparisDetay.Kapali, HGSiparisDetay.KapatmaTarihi, HGSiparisDetay.Notu,
     Model.ModelId, Model.Adi, HGSiparisDetay.ResimDosyaAdi1, HGSiparisDetay.ResimDosyayazdir1, HGSiparisDetay.ResimDosyaAdi2, HGSiparisDetay.ResimDosyayazdir2,
     HGSiparisDetay.InsertKullaniciID, IK.Adi, HGSiparisDetay.InsertTarihi, SK.Adi, HGSiparisDetay.DegisimTarihi, HGSiparisDetay.DegisimLog,
     HGSiparisDetay.OnayDurumu, HGSiparisDetay.OnayTarihi, HGSiparisDetay.OnaylayanKullaniciId, Ok.Adi, KullaniciImza.Imza,
     HGSiparisDetay.KumasSatinAlmaTalepKayitID, KumasSatinAlmaTalep.SatinAlmaTalepYil, KumasSatinAlmaTalep.SatinAlmaTalepID,
     HGSiparisDetay.AksesuarSatinAlmaTalepKayitID, AksSatinAlmaTalep.SatinAlmaTalepYil, AksSatinAlmaTalep.SatinAlmaTalepID,
     Model.ResimDosyaAdi1, Model.ResimDosyaAdi2 from HGSiparisDetay
  left join Model on Model.ModelKayitID  = HGSiparisDetay.ModelKayitID
  Left Join SatinAlmaTalep KumasSatinAlmaTalep on KumasSatinAlmaTalep.SatinAlmaTalepKayitID = HGSiparisDetay.KumasSatinAlmaTalepKayitID
  Left Join SatinAlmaTalep AksSatinAlmaTalep on AksSatinAlmaTalep.SatinAlmaTalepKayitID = HGSiparisDetay.AksesuarSatinAlmaTalepKayitID
  Left Join Kullanici IK  ON IK.KullaniciId = HGSiparisDetay.InsertKullaniciID
  Left Join Kullanici SK  ON SK.KullaniciId = HGSiparisDetay.KullaniciID
  Left Join Kullanici OK  ON OK.KullaniciId = HGSiparisDetay.OnaylayanKullaniciId
  left join OdemeSekli on OdemeSekli.OdemeSekliId  = HGSiparisDetay.OdemeSekliId
  Left Join KullaniciImza On KullaniciImza.KullaniciID = HGSiparisDetay.OnaylayanKullaniciId
  where HGSiparisDetay.HGSiparisKayitId = :AHGSiparisKayitId
  order by HGSiparisDetay.HGSiparisKayitId, HGSiparisDetay.SiparisSiraNo
  into :HGSiparisKayitId, :HGSiparisDetayId, :SiparisSiraNo, :SatisFiyati, :DovizId, :Teslimtarihi, :VadeGun, :OdemeSekliId, :OdemeSekliAdi, :MusteriSiparisNo, :MusteriModelNo,
       :ReferansSiparisNo, :Kapali, :KapatmaTarihi, :Notu, :ModelId, :ModelAdi, :ResimDosyaAdi1, :ResimDosyayazdir1, :ResimDosyaAdi2, :ResimDosyayazdir2,
       :InsertKullaniciID, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :DegisimLog, :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi,
       :OnaylayanKullaniciImza, :KumasSatinAlmaTalepKayitID, :KumasSatinAlmaTalepYil, :KumasSatinAlmaTalepID, :AksesuarSatinAlmaTalepKayitID, :AksesuarSatinAlmaTalepYil, :AksesuarSatinAlmaTalepID,
       :ModelResimDosyaAdi1, :ModelResimDosyaAdi2 do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    KapaliStr = '';
    if (Kapali = 1) then KapaliStr = 'Evet';

    suspend;
  end
end
