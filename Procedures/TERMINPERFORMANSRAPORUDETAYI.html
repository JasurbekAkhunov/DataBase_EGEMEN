CREATE PROCEDURE TERMINPERFORMANSRAPORUDETAYI(
    ABASLANGICSEVKTARIHI TIMESTAMP,
    ABITISSEVKTARIHI TIMESTAMP,
    APARTITIPI INTEGER,
    AFASON INTEGER,
    AISLEMTIPIID INTEGER,
    AFIRMAID VARCHAR(20),
    AGECIKMETIPIID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    PARTIIDSTR VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYENO IRSALIYENOSIZE,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTIONAYTARIHI TIMESTAMP,
    HAMSTOKGIRISTARIHI TIMESTAMP,
    PARTITIPIID INTEGER,
    PARTITIPIIDSTR VARCHAR(5),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID FIRMAKODUSIZE,
    SIPARISFIRMAADI ADI100,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    HAMADITOPLAMI VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    BRUTKILO NUMERIC(12,3),
    GECIKMENOTU VARCHAR(50),
    GECIKMETIPIID INTEGER,
    GECIKMETIPIADI VARCHAR(30),
    MUSTERIKAYNAKLI INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    SONUC INTEGER,
    SONUCSTR VARCHAR(20),
    KKBITISTARIHI TIMESTAMP,
    NETSEVKMETRE BOYAMIKTARSIZE,
    BRUTSEVKMETRE BOYAMIKTARSIZE)
AS
declare variable TerminToleransGun integer;
begin
  Select TerminToleransGun from sistem where sistemid = 1 into :TerminToleransGun;

  For Select BoyaIrsaliyeDetay.partiKayitId, BoyaIrsaliyeDetay.PartiSinifId, Parti.PartiYil, Parti.PartiId, Parti.TerminTarihi,
             Parti.AcilisTarihi, Parti.OnayTarihi,
             BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.FirmaId, Firma.Adi, Parti.PartiTipiId,
             Parti.IslemTipiId, IslemTipi.Adi, Parti.Fason, PartiStok.HamAdiToplami, Parti.ReceteId, Recete.Adi,
             BoyaIrsaliyeDetay.BrutKilo, PartiStok.GecikmeNotu, PartiStok.GecikmeTipiId, GecikmeTipi.Adi, GecikmeTipi.MusteriKaynakli,
             BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaID, SipFirma.Adi, PartiStok.KKBitisTarihi, PartiStok.NetSevkMetre, PartiStok.BrutSevkMetre from BoyaIrsaliye
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
    left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
    left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
    left join GecikmeTipi on GecikmeTipi.GecikmeTipiId = PartiStok.GecikmeTipiId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Firma SipFirma on SipFirma.FirmaId = BoyaSiparis.FirmaId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicSevkTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisSevkTarihi and
          BoyaIrsaliyeDetay.PartiKayitId is not null and
         (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
         (Parti.IslemTipiID = :AIslemTipiID or :AIslemTipiID is null) and
         (Parti.Fason = :AFason or :AFason is null) and
         (Parti.PartiTipiId = :APartiTipi or :APartiTipi is null) and
         (PartiStok.GecikmeTipiId = :AGecikmeTipiId or :AGecikmeTipiId is null)
    into :PartiKayitId, :PartiSinifId, :PartiYil, :PartiId, :TerminTarihi, :AcilisTarihi, :PartiOnayTarihi, :IrsaliyeTarihi, :IrsaliyeNo, :FirmaId, :FirmaAdi,
         :PartiTipiId, :IslemTipiId, :IslemTipiAdi, :Fason, :HamAdiToplami, :ReceteId, :ReceteAdi, :BrutKilo, :GecikmeNotu,
         :GecikmeTipiId, :GecikmeTipiAdi, :MusteriKaynakli, :MusteriSiparisNo, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
         :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaID, :SiparisFirmaAdi, :KKBitisTarihi, :NetSevkMetre, :BrutSevkMetre do
  begin

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId)  || ' ';
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    if (PartiTipiId = 0) then PartiTipiIdStr = 'İplik';  else PartiTipiIdStr = 'Diger';

    if (Fason = 1) then FasonStr = 'Fason';  else FasonStr = '';

    if (IrsaliyeTarihi < TerminTarihi) then
    begin
      Sonuc = 1;
      SonucStr = 'Erken';
      MusteriKaynakli = Null;
    end
    else if (IrsaliyeTarihi > (TerminTarihi + TerminToleransGun)) then
    begin
      Sonuc = -1;
      SonucStr = 'GEÇ';
    end
    else if (IrsaliyeTarihi = TerminTarihi) then
    begin
      Sonuc = 0;
      SonucStr = 'Zamanında';
      MusteriKaynakli = Null;
    end
    else if (IrsaliyeTarihi = (TerminTarihi + 1) ) then
    begin
      Sonuc = 0;
      SonucStr = 'Zamanında (T)';
      MusteriKaynakli = Null;
    end

    if (GecikmeTipiId is null) then GecikmeTipiId = -1;

    HamStokGirisTarihi = Null;
    Select Max(HamStokIrsaliye.IrsaliyeTarihi) from PartiStokDetay
      left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left Join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
      where PartiStokDetay.PartiSinifId = :PartiSinifId
      into :HamStokGirisTarihi;

    if (SiparisGrupId is null) then SiparisGrupId = 'BOŞ';
    if (SiparisGrup2Id is null) then SiparisGrup2Id = 'BOŞ';
    suspend;
  end
end
