CREATE PROCEDURE FASONCUIPLIKDURUMUHAZIRLA(
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    AFIRMAARALIKLI INTEGER,
    ATALIMATBITTI INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ORMESIPARISDETAYID INTEGER,
    ORMEPLANLAMAID INTEGER,
    URUNADI VARCHAR(255),
    URUNADI2 VARCHAR(255),
    URUNADI3 VARCHAR(255),
    URUNADI4 VARCHAR(255),
    URETIMPARTINO VARCHAR(30),
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIIDSTR VARCHAR(20),
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    KALINLIKID INTEGER,
    IPLIKCINSIID INTEGER,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    KULLANIMORANI INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PUS INTEGER,
    FEINE INTEGER,
    HAMIADE INTEGER,
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    KULLANIMMIKTARI DOUBLE PRECISION,
    TAHMINIFIREMIKTARI DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    TAHMINIFIREORANI DOUBLE PRECISION,
    HAMMIKTAR DOUBLE PRECISION,
    SONALISFIYATI DOUBLE PRECISION,
    SONSATISFIYATI DOUBLE PRECISION,
    SONURETIMECIKISFIYATI DOUBLE PRECISION,
    ORTALAMAFASONUCIKISFIYATI DOUBLE PRECISION,
    TALIMATTAHMINIFIREMIKTARI DOUBLE PRECISION,
    TAHMINIRECETEFIREMIKTARI DOUBLE PRECISION,
    RECETEFIREORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    TALIMATRECETEFIREORANI DOUBLE PRECISION)
AS
declare Variable Okey                         integer;
declare variable FMiktar        Double precision;
declare variable hstIplik       integer;
declare variable hstLycra       integer;
declare variable shtAlis        integer;
declare variable shtSatis       integer;
declare variable ShtFasonAlis   integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis      integer;
declare variable shtFasonUretim       integer;
declare variable shtUretim            integer;
declare variable FHareketTipi         integer;
declare variable FPlanlananMiktar     double precision;
declare variable FGerceklesenMiktar   double precision;
declare variable OrmeTalimattaReceteFiresiYap integer;
declare variable OtomatikUCikislardaReceteFire integer;
declare variable UrunHamId integer;
begin
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  select shtAlis, shtSatis, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir into :shtAlis, :shtSatis, :ShtFasonAlis, :shtFasonUretimeCikis, :shtUretimeCikis, :shtFasonUretim, shtUretim;


  For Select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.OrmeSiparisDetayId, OrmePlanlama.OrmePlanlamaId,
             OrmePlanlama.UretimPartiNo, OrmePlanlama.FirmaId, Firma.Adi, ReceteHam.HamTipiId, HamTipi.Adi, ReceteHam.Adi, ReceteHam.Adi2, ReceteHam.Adi3, ReceteHam.Adi4,
             ReceteHam.KalinlikId, ReceteHam.IplikCinsiId,
             OrmePlanlama.GerceklesenMiktar, StokHareket.HamId,
             StokHareket.RenkId, Renk.Adi, StokHareket.Lot, StokHareket.UreticiId, Uretici.Adi,
             OrmePlanlamaRecete.KullanimOrani, Sum(Miktar),
             OrmeSiparisDetay.TahminiFireOrani, SipHam.Adi, SipHam.Adi2, SipHam.Adi3, SipHam.Adi4, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmeSiparisDetay.HamMiktar, 0, null,
             OrmePlanlama.PlanlananMiktar,
             OrmePlanlama.PlanlananMiktar - sum(case when ((StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and (StokHareket.HamIade = 0))  then  Miktar else 0 end),
             sum(case when ((StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and (StokHareket.HamIade = 0))  then Miktar else 0 end),
             sum(case when ((StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and (StokHareket.HamIade = 1))  then Miktar else 0 end),
             sum(case when ((StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis) and (StokHareket.HamIade = 0))  then Miktar else 0 end),
             OrmeSiparisDetay.HamId,OrmePlanlama.PlanlananMiktar * OrmePlanlamaRecete.KullanimOrani / 100, OrmePlanlamaRecete.TahminiFireOrani,
             OrmePlanlama.GerceklesenMiktar * OrmePlanlamaRecete.KullanimOrani / 100
             From BoyaSiparis
     left join OrmeSiparisDetay on OrmeSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
     left join Ham SipHam on SipHam.HamId = OrmeSiparisDetay.HamId
     left join OrmePlanlama on OrmePlanlama.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId and OrmePlanlama.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
     left join Firma on Firma.FirmaId = OrmePlanlama.FirmaId
     left join StokHareket on StokHareket.OrmePlanlamaId = OrmePlanlama.OrmePlanlamaId
     left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = StokHareket.OrmePlanlamaId and IntKarsilastir(OrmePlanlamaRecete.RenkID,StokHareket.RenkID) = 1 and StrKarsilastir(OrmePlanlamaRecete.Lot,StokHareket.Lot) = 1 and IntKarsilastir(OrmePlanlamaRecete.UreticiID,StokHareket.UreticiID) = 1 and OrmePlanlamaRecete.HamId = StokHareket.HamId
     left join Ham ReceteHam on ReceteHam.HamId = StokHareket.HamId
     left join Uretici on Uretici.UreticiId = StokHareket.UreticiId
     left join Renk on Renk.RenkId = StokHareket.RenkId
     left join HamTipi on HamTipi.HamTipiId = ReceteHam.HamTipiId
     where (BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
           (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil  and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId)) and
           OrmeSiparisDetay.OrmeSiparisDetayId is not null and OrmePlanlama.OrmePlanlamaId is not null and OrmeSiparisDetay.Kapali = 0 and
           (OrmePlanlama.Bitti = :ATalimatBitti or :ATalimatBitti is null) and
           OrmePlanlama.FirmaId is not null and StokHareket.OrmePlanlamaId is not null and
           ((:AFirmaAralikli = 1 and OrmePlanlama.FirmaId >= :ABaslangicFirmaId and (OrmePlanlama.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0))
  Group by BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.OrmeSiparisDetayId, OrmePlanlama.OrmePlanlamaId,
             OrmePlanlama.UretimPartiNo, OrmePlanlama.FirmaId, Firma.Adi, ReceteHam.HamTipiId, HamTipi.Adi, ReceteHam.Adi, ReceteHam.Adi2, ReceteHam.Adi3, ReceteHam.Adi4,
             ReceteHam.KalinlikId, ReceteHam.IplikCinsiId,
             OrmePlanlama.GerceklesenMiktar, StokHareket.HamId,
             StokHareket.RenkId, Renk.Adi, StokHareket.Lot, StokHareket.UreticiId, Uretici.Adi,
             OrmePlanlamaRecete.KullanimOrani,
             OrmeSiparisDetay.TahminiFireOrani, OrmePlanlama.Tarih,SipHam.Adi, SipHam.Adi2, SipHam.Adi3, SipHam.Adi4, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmeSiparisDetay.HamMiktar,
             OrmePlanlama.PlanlananMiktar,OrmeSiparisDetay.HamId,OrmePlanlama.PlanlananMiktar * OrmePlanlamaRecete.KullanimOrani / 100, OrmePlanlamaRecete.TahminiFireOrani
  order by OrmePlanlama.FirmaId, OrmePlanlama.Tarih, OrmePlanlama.OrmePlanlamaId
  into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :OrmeSiparisDetayId, :OrmePlanlamaId, :UretimPartiNo, :FirmaId, :FirmaAdi,
       :HamTipiId, :HamTipiIdStr, :HamAdi, :HamAdi2, :HamAdi3, :HamAdi4, :KalinlikId, :IplikCinsiId, :FGerceklesenMiktar, :HamId, :RenkId, :RenkAdi, :Lot, :UreticiId, :UreticiAdi,
       :KullanimOrani, :FMiktar, :TahminiFireOrani, :UrunAdi, :UrunAdi2, :UrunAdi3, :UrunAdi4, :Pus, :Feine, :HamMiktar, :Hamiade, :FHareketTipi, :FPlanlananMiktar,
       :KalanMiktar, :GerceklesenMiktar, :IadeMiktari, :CikisMiktari, :UrunHamId, :KullanilacakMiktar, :TalimatReceteFireOrani, :KullanimMiktari do
  begin
    Okey = 1;
    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    Begin
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

TalimatTahminiFireMiktari = 0;
    TahminiFireMiktari        = 0;
    TahminiReceteFireMiktari  = 0;

    if ((hamTipiId = hstIplik) or (hamTipiId = hstLycra)) then
    begin
      select first 1 OrmeTalimattaReceteFiresiYap, OtomatikUCikislardaReceteFire from Sistem into :OrmeTalimattaReceteFiresiYap, :OtomatikUCikislardaReceteFire;
      select first 1 FireOrani from HamStokRecete where HamId = :UrunHamId and ReceteHamId = :HamId into :ReceteFireOrani;

      if (OtomatikUCikislardaReceteFire = 1) then
      begin
        KullanilacakMiktar = KullanilacakMiktar / ((100 - TalimatReceteFireOrani) / 100);
        if (hamTipiId = hstIplik) then
        begin
          TalimatTahminiFireMiktari = (((FGerceklesenMiktar * KullanimOrani) / 100) * (TalimatReceteFireOrani / 100));
          KalanMiktar = CikisMiktari - IadeMiktari - KullanimMiktari - TalimatTahminiFireMiktari ;
        end
        else if (HamTipiId = hstLycra) then
          KalanMiktar = CikisMiktari - IadeMiktari - KullanimMiktari ;
      end
      else if (OrmeTalimattaReceteFiresiYap <> 1) then
      begin
        KullanilacakMiktar = KullanilacakMiktar / ((100 - TahminiFireOrani) / 100);

        if (hamTipiId = hstIplik) then
        begin
          TahminiFireMiktari = (((FGerceklesenMiktar * KullanimOrani) / 100) * (TahminiFireOrani / 100));
          KalanMiktar = CikisMiktari - IadeMiktari - KullanimMiktari - TahminiFireMiktari;
        end
        else if (HamTipiId = hstLycra) then
          KalanMiktar = CikisMiktari - IadeMiktari - KullanimMiktari ;
      end
      else
      begin
        KullanilacakMiktar = KullanilacakMiktar / ((100 - ReceteFireOrani) / 100);

        if (hamTipiId = hstIplik) then
        begin
          TahminiReceteFireMiktari =  (((FGerceklesenMiktar * KullanimOrani) / 100) * (ReceteFireOrani / 100));
          KalanMiktar = CikisMiktari - IadeMiktari - KullanimMiktari - TahminiReceteFireMiktari ;
        end
        else if (HamTipiId = hstLycra) then
          KalanMiktar = CikisMiktari - IadeMiktari - KullanimMiktari ;
      end
    end

      Select first 1 BirimFiyat from StokHareket where HamID = :HamId and (HareketTipi = :shtAlis or HareketTipi = :shtFasonAlis) and BirimFiyat > 0 Order by Tarih Desc into :SonAlisFiyati;
      Select first 1 BirimFiyat from StokHareket where  HamID = :HamId and (HareketTipi = :shtSatis ) and BirimFiyat > 0 Order by Tarih Desc into :SonSatisFiyati;
      Select first 1 BirimFiyat from StokHareket where  OrmePlanlamaID = :OrmePlanlamaId and HamID = :HamId and (HareketTipi = :shtUretimeCikis) and BirimFiyat > 0 Order by Tarih Desc into :SonUretimeCikisFiyati;
      Select Avg(BirimFiyat) from StokHareket where  OrmePlanlamaID = :OrmePlanlamaId and HamID = :HamId and (HareketTipi = :shtFasonUretimeCikis) and BirimFiyat > 0 into :OrtalamaFasonUCikisFiyati;

      suspend;
    end
  end
end
