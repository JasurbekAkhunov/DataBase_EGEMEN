CREATE PROCEDURE VERILENCEKSENETBILGISIGETIR(
    CS VARCHAR(1),
    ACEKSENETID INTEGER,
    AGENELKAYITID INTEGER)
RETURNS (
    BORDROID INTEGER,
    BORDROTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    CEKSENETID INTEGER,
    GENELKAYITID INTEGER,
    ACIKLAMA VARCHAR(40),
    VADETARIHI TIMESTAMP,
    MEBLAG DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    DOVIZKURU1 DOUBLE PRECISION,
    DOVIZKURU2 DOUBLE PRECISION,
    ISLEMDOVIZMEBLAG DOUBLE PRECISION,
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZKURU1 DOUBLE PRECISION,
    ISLEMDOVIZKURU2 DOUBLE PRECISION,
    DOVIZMEBLAG DOUBLE PRECISION,
    BANKAHESAPID VARCHAR(20),
    CEKSENETNO VARCHAR(30),
    KESIDEYERI VARCHAR(20),
    CEKSENETGRUP1 VARCHAR(5),
    CEKSENETGRUP2 VARCHAR(5),
    CEKSENETGRUP3 VARCHAR(5),
    KARSILIKSIZ INTEGER,
    SONBORDROTIPI INTEGER,
    TAHSILEDILMIS INTEGER)
AS
declare variable TahsilEdilenMeblag Double Precision;
declare variable FaturaCariyeIslemDovizinden integer;
begin
  Select FaturaCariyeIslemDovizinden from Sistem where SistemID = 1 into :FaturaCariyeIslemDovizinden;

  if (ACekSenetId = -1) then
  begin
    select CekSenet.BordroId, CekSenet.Tarih, CekSenet.FirmaId, CekSenetDetay.GenelKayitId, CekSenetId, CekSenetDetay.Aciklama,
           Vadetarihi, Meblag, TahsilEdilenMeblag, CekSenetDetay.DovizId, DovizKuru1, DovizKuru2, BankaHesapId, CekSenetno,
               KesideYeri, CekSenetGrup1, CekSenetGrup2, CekSenetGrup3, Karsiliksiz, SonBordroTipi, CekSenetDetay.IslemDovizID, CekSenetDetay.IslemDovizKuru1, CekSenetDetay.IslemDovizKuru2, CekSenetDetay.IslemDovizMeblag from CekSenetDetay
    left join CekSenet on CekSenet.CsBordroKayitId = CekSenetDetay.CsBordroKayitId
    where CekSenetDetay.GenelKayitId = :AGenelKayitId and CekSenetDetay.AV = 'V' and CekSenetDetay.CS = :Cs
    into :BordroId,:BordroTarihi, :FirmaId, :GenelKayitId, :CekSenetId, :Aciklama, :Vadetarihi, :Meblag, :TahsilEdilenMeblag,
         :DovizId, :DovizKuru1, :DovizKuru2, :BankaHesapId, :CekSenetno, :KesideYeri, :CekSenetGrup1, :CekSenetGrup2,
         :CekSenetGrup3, :Karsiliksiz, :SonBordroTipi, :IslemDovizID, :IslemDovizKuru1, :IslemDovizKuru2, :IslemDovizMeblag;
  end
  else
  begin
    select CekSenet.BordroId, CekSenet.Tarih, CekSenet.FirmaId, CekSenetDetay.GenelKayitId, CekSenetId, CekSenetDetay.Aciklama,
           Vadetarihi, Meblag, TahsilEdilenMeblag, CekSenetDetay.DovizId, DovizKuru1, DovizKuru2, BankaHesapId, CekSenetno,
               KesideYeri, CekSenetGrup1, CekSenetGrup2, CekSenetGrup3, Karsiliksiz, SonBordroTipi, CekSenetDetay.IslemDovizID, CekSenetDetay.IslemDovizKuru1, CekSenetDetay.IslemDovizKuru2, CekSenetDetay.IslemDovizMeblag from CekSenetDetay
    left join CekSenet on CekSenet.CsBordroKayitId = CekSenetDetay.CsBordroKayitId
    where CekSenetDetay.CekSenetId = :ACekSenetId and CekSenetDetay.AV = 'V' and CekSenetDetay.CS = :Cs
    into :BordroId,:BordroTarihi, :FirmaId, :GenelKayitId, :CekSenetId, :Aciklama, :Vadetarihi, :Meblag, :TahsilEdilenMeblag,
         :DovizId, :DovizKuru1, :DovizKuru2, :BankaHesapId, :CekSenetno, :KesideYeri, :CekSenetGrup1, :CekSenetGrup2,
         :CekSenetGrup3, :Karsiliksiz, :SonBordroTipi, :IslemDovizID, :IslemDovizKuru1, :IslemDovizKuru2, :IslemDovizMeblag;
  end

  if (FaturaCariyeIslemDovizinden = 0) then
    DovizMeblag = Meblag / DovizKuru1 * DovizKuru2;
  else
    DovizMeblag = IslemDovizMeblag / DovizKuru1 * DovizKuru2;

  if (TahsilEdilenMeblag = 0) then TahsilEdilmis = 0;
  else if (TahsilEdilenMeblag < Meblag) then TahsilEdilmis = 1;
  else TahsilEdilmis = 2;

  if (CekSenetId is not null) then suspend;
end
