CREATE PROCEDURE KARTELAFISILISTESIDETAYI(
    AGIRIS INTEGER,
    ABASLANGICFISTARIHI TIMESTAMP,
    ABITISFISTARIHI TIMESTAMP,
    ABASLANGICFISYIL INTEGER,
    ABASLANGICFISNO INTEGER,
    ABITISFISYIL INTEGER,
    ABITISFISNO INTEGER)
RETURNS (
    KARTELAFISIID INTEGER,
    KARTELAFISIDETAYID INTEGER,
    KALITEID VARCHAR(20),
    KALITEADI ADI255,
    HAMADI VARCHAR(255),
    ACIKLAMA VARCHAR(255),
    MIKTAR INTEGER,
    METRE NUMERIC(12,3),
    RECETEKODU VARCHAR(20),
    RECETEADI VARCHAR(40),
    DESENKODU VARCHAR(20),
    DESENADI VARCHAR(50),
    PUS INTEGER,
    FEINE INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(20),
    IMALATTIPISTR VARCHAR(5))
AS
declare variable FImalatTipiId integer;

begin
  For select KartelaFisiListesi.KartelaFisiId, KartelaFisiDetay.KartelaFisiDetayId, KartelaFisiDetay.KaliteID, Kalite.KaliteAdi,
             Kalite.HamAdi, KartelaFisiDetay.Miktar, KartelaFisiDetay.Metre, Kalite.ReceteKodu, Kalite.ReceteAdi, Kalite.DesenKodu, Kalite.Desenadi,
             Kalite.Pus, Kalite.Feine, KartelaFisiDetay.Aciklama, Kalite.PartiYil, Kalite.PartiID, Parti.ImalatTipiId from KartelaFisiListesi(:AGiris, :ABaslangicFisTarihi, :ABitisFisTarihi, :ABaslangicFisYil, :ABaslangicFisNo, :ABitisFisYil, :ABitisFisNo)
  left join KartelaFisiDetay on KartelaFisiDetay.KartelaFisiId = KartelaFisiListesi.KartelaFisiId
  left join Kalite on Kalite.KaliteId = KartelaFisiDetay.KaliteId
  left join Parti on Parti.PartiYil = Kalite.PartiYil and Parti.PartiId = Kalite.PartiID
  where KartelaFisiDetay.KartelaFisiDetayId is not null
  order by KartelaFisiDetay.KartelaFisiId, KartelaFisiDetay.KartelaFisiDetayId
  into :KartelaFisiId, :KartelaFisiDetayId, :KaliteID, :KaliteAdi, :HamAdi, :Miktar, :Metre, :ReceteKodu, :ReceteAdi, :DesenKodu, :Desenadi,
       :Pus, :Feine, :Aciklama, :PartiYil, :PartiID, :FImalatTipiId do
  begin
    if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    else PartiIdStr = '';

    if (FImalatTipiId = 0) then ImalatTipiStr = 'Boya';
    if (FImalatTipiId = 1) then ImalatTipiStr = 'BaskÄ±';

    suspend;
  end
end
