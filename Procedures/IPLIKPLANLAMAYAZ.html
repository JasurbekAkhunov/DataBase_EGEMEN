CREATE PROCEDURE IPLIKPLANLAMAYAZ(
    AIPLIKMAKINATIPIID INTEGER,
    AIPLIKPLANLAMAGRUPID VARCHAR(10),
    AIPLIKMAKINAID VARCHAR(10))
RETURNS (
    SEPETNO INTEGER,
    IPLIKPLANLAMAGRUPID VARCHAR(10),
    IPLIKMAKINAID VARCHAR(10),
    IPLIKMAKINAADI VARCHAR(40),
    IPLIKMAKINATIPIID INTEGER,
    IPLIKURETIMISEMRIKAYITID INTEGER,
    IPLIKURETIMISEMRIYIL INTEGER,
    IPLIKURETIMISEMRIID INTEGER,
    PLANLAMATARIHI TIMESTAMP,
    IPLIKLOTID INTEGER,
    LOT VARCHAR(30),
    BUKUMTIPIID INTEGER,
    BUKUMTIPIADI VARCHAR(10),
    KALINLIKNE INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    COPSTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    PLANLANANMIKTAR NUMERIC(12,3),
    GERCEKLESENMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    SIRA INTEGER,
    BASLAMATARIHI TIMESTAMP,
    PLANLANDIGIMAKINALAR VARCHAR(255),
    BUKUMSAYISI INTEGER,
    TOPLAMIGSAYISI INTEGER,
    EKSIKIGSAYISI INTEGER,
    IGHIZI INTEGER)
AS
declare variable LASTIPLIKMAKINAID varchar(10);
declare variable FIPLIKMAKINAID varchar(10);
declare variable FBITTI integer;
begin
  if (AIplikPlanlamaGrupId = '') then AIplikPlanlamaGrupId = Null;

  LastIplikMakinaID = '@@@';

  if (AIplikPlanlamaGrupId is null) then
  begin
    for Select IplikPlanlamaGrupDetayi.IplikPlanlamaGrupId, IplikUretimIsEmri.IplikUretimIsEmriKayitId, IplikUretimIsEmri.IplikUretimIsEmriYil, IplikUretimIsEmri.IplikUretimIsEmriId,
             IplikUretimIsEmri.PlanlamaTarihi, IplikUretimIsEmri.IplikMakinaId, IplikMakina.Adi, IplikUretimIsEmri.IplikLotId,
             IplikLot.Lot, IplikLot.BukumTipiId, BukumTipi.Adi, IplikLot.KalinlikNe, IplikCinsi.Adi,
             Ham.Adi, IplikUretimIsEmri.PlanlananMiktar, IplikUretimIsEmri.GerceklesenMiktar,
             IplikUretimIsEmri.BaslamaTarihi, IplikUretimIsEmri.Sira, CopsTipi.Adi, IplikUretimIsEmri.IplikMakinaTipiId, IplikLot.BukumSayisi, IplikMakina.ToplamIgSayisi, IplikUretimIsemri.IgHizi from IplikUretimIsEmri
      left join IplikPlanlamaGrupDetayi on IplikPlanlamaGrupDetayi.IplikMakinaId = IplikUretimIsEmri.IplikMakinaId
      left join IplikMakina on IplikMakina.IplikMakinaID = IplikUretimIsEmri.IplikMakinaId
      left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotId
      left join Ham on Ham.HamID = IplikLot.HamId
      left join IplikCinsi on IplikCinsi.IplikCinsiID = IplikLot.IplikCinsiId
      left join BukumTipi on BukumTipi.BukumTipiID = IplikLot.BukumTipiId
      left join CopsTipi on CopsTipi.CopsTipiID = IplikUretimIsEmri.CopsTipiId
      where (IplikUretimIsEmri.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null) and
            (IplikUretimIsEmri.IplikMakinaTipiID = :AIplikMakinaTipiID) and
            IplikUretimIsEmri.Bitti = 0
    order by IplikUretimIsEmri.IplikMakinaId, IplikUretimIsEmri.Sira
    into :IplikPlanlamaGrupId, :IplikUretimIsEmriKayitId, :IplikUretimIsEmriYil, :IplikUretimIsEmriId, :PlanlamaTarihi, :IplikMakinaId, :IplikMakinaAdi,
         :IplikLotId, :Lot, :BukumTipiId, :BukumTipiAdi, :KalinlikNe, :IplikCinsiAdi, :HamAdi,  :PlanlananMiktar, :GerceklesenMiktar,
         :BaslamaTarihi, :Sira, :CopsTipiAdi, :IplikMakinaTipiId, :BukumSayisi, :ToplamIgSayisi, :IgHizi do
    begin
      if (LastIplikMakinaId <> IplikMakinaId) then
      begin
        SepetNO = 0;
        LastIplikMakinaId = IplikMakinaId;
      end

      Select First 1 IplikMakinaSayac.EksikIgsayisi from IplikMakinaSayac where IplikMakinaSayac.IplikMakinaID = :IplikMakinaID and VardiyaTarihi = Current_Date Order By  IplikMakinaSayacID Desc into :EksikIgsayisi;

      SepetNO = SepetNo + 1;
      KalanMiktar = PlanlananMiktar - GerceklesenMiktar;

      PlanlandigiMakinalar = '';

      for Select IplikUretimIsEmri.IplikMakinaId, Min(Bitti) from IplikUretimIsEmri
      where IplikUretimIsEmri.IplikLotID = :IplikLotID and IplikUretimIsEmri.IplikMakinaTipiID = :IplikMakinaTipiID
      group by IplikUretimIsEmri.IplikMakinaId
      into :FIplikMakinaId, :FBitti do
      begin
        if ( Char_Length(PlanlandigiMakinalar) + Char_Length(FIplikMakinaId) < 230) then
        begin
          if (PlanlandigiMakinalar <> '') then PlanlandigiMakinalar = PlanlandigiMakinalar || cast(', ' as varchar(2) character set UTF8);
          PlanlandigiMakinalar = PlanlandigiMakinalar || FIplikMakinaId;

          if (FBitti = 0) then PlanlandigiMakinalar = PlanlandigiMakinalar || '(-)';
          else PlanlandigiMakinalar = PlanlandigiMakinalar || '(+)';
        end
      end

      suspend;
    end
  end
  else
  begin
    for Select IplikPlanlamaGrupDetayi.IplikPlanlamaGrupId, IplikUretimIsEmri.IplikUretimIsEmriKayitId, IplikUretimIsEmri.IplikUretimIsEmriYil, IplikUretimIsEmri.IplikUretimIsEmriId,
             IplikUretimIsEmri.PlanlamaTarihi, IplikUretimIsEmri.IplikMakinaId, IplikMakina.Adi, IplikUretimIsEmri.IplikLotId,
             IplikLot.Lot, IplikLot.BukumTipiId, BukumTipi.Adi, IplikLot.KalinlikNe, IplikCinsi.Adi,
             Ham.Adi, IplikUretimIsEmri.PlanlananMiktar, IplikUretimIsEmri.GerceklesenMiktar,
             IplikUretimIsEmri.BaslamaTarihi, IplikUretimIsEmri.Sira, CopsTipi.Adi, IplikUretimIsEmri.IplikMakinaTipiId, IplikLot.BukumSayisi, IplikMakina.ToplamIgSayisi, IplikUretimIsemri.IgHizi from IplikPlanlamaGrupDetayi
      left join IplikUretimIsEmri on IplikUretimIsEmri.IplikMakinaID = IplikUretimIsEmri.IplikMakinaId
      left join IplikMakina on IplikMakina.IplikMakinaID = IplikUretimIsEmri.IplikMakinaId
      left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotId
      left join Ham on Ham.HamID = IplikLot.HamId
      left join IplikCinsi on IplikCinsi.IplikCinsiID = IplikLot.IplikCinsiId
      left join BukumTipi on BukumTipi.BukumTipiID = IplikLot.BukumTipiId
      left join CopsTipi on CopsTipi.CopsTipiID = IplikUretimIsEmri.CopsTipiId
      where IplikPlanlamaGrupDetayi.IplikPlanlamaGrupId = :AIplikPlanlamaGrupId and
            (IplikUretimIsEmri.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null) and
            (IplikUretimIsEmri.IplikMakinaTipiID = :AIplikMakinaTipiID) and
            IplikUretimIsEmri.Bitti = 0
      order by IplikPlanlamaGrupDetayi.Sira, IplikUretimIsEmri.IplikMakinaId, IplikUretimIsEmri.Sira
      into :IplikPlanlamaGrupId, :IplikUretimIsEmriKayitId, :IplikUretimIsEmriYil, :IplikUretimIsEmriId, :PlanlamaTarihi, :IplikMakinaId, :IplikMakinaAdi,
           :IplikLotId, :Lot, :BukumTipiId, :BukumTipiAdi, :KalinlikNe, :IplikCinsiAdi, :HamAdi,  :PlanlananMiktar, :GerceklesenMiktar,
           :BaslamaTarihi, :Sira, :CopsTipiAdi, :IplikMakinaTipiId, :BukumSayisi, :ToplamIgSayisi, :IgHizi do
    begin
      if (LastIplikMakinaId <> IplikMakinaId) then
      begin
        SepetNO = 0;
        LastIplikMakinaId = IplikMakinaId;
      end

      Select First 1 IplikMakinaSayac.EksikIgsayisi from IplikMakinaSayac where IplikMakinaSayac.IplikMakinaID = :IplikMakinaID and VardiyaTarihi = Current_Date Order By  IplikMakinaSayacID Desc into :EksikIgsayisi;

      SepetNO = SepetNo + 1;
      KalanMiktar = PlanlananMiktar - GerceklesenMiktar;

      PlanlandigiMakinalar = '';

      for Select IplikUretimIsEmri.IplikMakinaId, Min(Bitti) from IplikUretimIsEmri
      where IplikUretimIsEmri.IplikLotID = :IplikLotID and IplikUretimIsEmri.IplikMakinaTipiID = :IplikMakinaTipiID
      group by IplikUretimIsEmri.IplikMakinaId
      into :FIplikMakinaId, :FBitti do
      begin
        if ( Char_Length(PlanlandigiMakinalar) + Char_Length(FIplikMakinaId) < 230) then
        begin
          if (PlanlandigiMakinalar <> '') then PlanlandigiMakinalar = PlanlandigiMakinalar || cast(', ' as varchar(2) character set UTF8);
          PlanlandigiMakinalar = PlanlandigiMakinalar || FIplikMakinaId;

          if (FBitti = 0) then PlanlandigiMakinalar = PlanlandigiMakinalar || '(-)';
          else PlanlandigiMakinalar = PlanlandigiMakinalar || '(+)';
        end
      end
      suspend;
    end
  end
end
