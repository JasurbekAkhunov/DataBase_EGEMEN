CREATE PROCEDURE STOKHAREKETDETAYETIKET(
    ASTOKHAREKETDETAYID INTEGER)
RETURNS (
    STOKHAREKETDETAYID INTEGER,
    STOKHAREKETID INTEGER,
    HAREKETTIPI INTEGER,
    BARKOD VARCHAR(120),
    IPLIKSTOKID INTEGER,
    IPLIKSTOKNUMARATOR INTEGER,
    KILO NUMERIC(18,3),
    BRUTKILO NUMERIC(18,3),
    METRE NUMERIC(18,3),
    KAPADEDI INTEGER,
    AMBALAJNO VARCHAR(40),
    NOTU MEMO,
    PSKALITEKONTROLID INTEGER,
    IPLIKPAKETLEMEID INTEGER,
    ORMERAFID INTEGER,
    ORMERAFTRANSFERFISIDETAYID INTEGER,
    GIRISKALITEKONTROLID INTEGER,
    LINKISLEMI INTEGER,
    KONIKRENGIID INTEGER,
    BOBINGRAMAJI DOUBLE PRECISION,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    BOLUNDUGUSTOKHAREKETDETAYID INTEGER,
    KONIKRENGIADI VARCHAR(80),
    KALANKILO NUMERIC(18,3),
    KALANBRUTKILO NUMERIC(18,3),
    KALANMETRE NUMERIC(18,3),
    KALANKAPADEDI INTEGER)
AS
Begin
  For
    Select StokHareketDetay.StokHareketDetayId, StokHareketDetay.StokHareketId, StokHareketDetay.HareketTipi, StokHareketDetay.Barkod, StokHareketDetay.IplikStokId,
      StokHareketDetay.IplikStokNumarator, StokHareketDetay.Kilo, StokHareketDetay.BrutKilo, StokHareketDetay.Metre, StokHareketDetay.KapAdedi, StokHareketDetay.AmbalajNo,
      StokHareketDetay.Notu, StokHareketDetay.PSKaliteKontrolId, StokHareketDetay.IplikPaketlemeId, StokHareketDetay.OrmeRafId, StokHareketDetay.OrmeRafTransferFisiDetayId,
      StokHareketDetay.GirisKaliteKontrolId, StokHareketDetay.LinkIslemi, StokHareketDetay.KonikRengiId, StokHareketDetay.BobinGramaji, StokHareketDetay.InsertKullaniciId,
      StokHareketDetay.InsertTarihi, StokHareketDetay.KullaniciId, StokHareketDetay.DegisimTarihi, StokHareketDetay.BolunduguStokHareketDetayId, KonikRengi.Adi,
      StokHareketDetayMIKTAR.girisKilo-StokHareketDetayMIKTAR.cikisKilo KalanKilo, StokHareketDetayMIKTAR.girisBrutKilo-StokHareketDetayMIKTAR.cikisBrutKilo KalanBrutKilo,
      StokHareketDetayMIKTAR.GirisMetre-StokHareketDetayMIKTAR.cikisMetre  KalanMetre, StokHareketDetayMIKTAR.GirisKapAdedi-StokHareketDetayMIKTAR.cikisKapAdedi  KalanKapAdedi From StokHareketDetay
    left join StokHareketDetayMIKTAR on StokHareketDetayMIKTAR.Barkod = StokHareketDetay.Barkod
      left join KonikRengi on KonikRengi.KonikRengiId = StokHareketDetay.KonikRengiId
    Where StokHareketDetayId  = :AStokHareketDetayId
    Into :StokHareketDetayId, :StokHareketId, :HareketTipi, :Barkod, :IplikStokId, :IplikStokNumarator, :Kilo, :BrutKilo, :Metre, :KapAdedi, :AmbalajNo, :Notu, :PSKaliteKontrolId,
         :IplikPaketlemeId, :OrmeRafId, :OrmeRafTransferFisiDetayId, :GirisKaliteKontrolId, :LinkIslemi, :KonikRengiId, :BobinGramaji, :InsertKullaniciId, :InsertTarihi, :KullaniciId,
         :DegisimTarihi, :BolunduguStokHareketDetayId, :KonikRengiAdi, :KalanKilo, :KalanBrutKilo, :KalanMetre, :KalanKapAdedi
  do
  Begin
    Suspend;
  end
end
