CREATE PROCEDURE IPLIKTALEPDETAYGETIR(
    AIPLIKTALEPKAYITID INTEGER)
RETURNS (
    IPLIKTALEPDETAYID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FASON SMALLINT,
    FASONSTR VARCHAR(5),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMKDVORANI DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    BIRIMMALIYET DOUBLE PRECISION,
    TESLIMEDILENMIKTAR DOUBLE PRECISION,
    TESLIMEDILENNETMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    LOT VARCHAR(30),
    ILAVEISLEMLER VARCHAR(255),
    OZELTESTLER VARCHAR(255),
    REFERANSURETIMNO VARCHAR(30),
    KAPALI INTEGER,
    KAPALISTR VARCHAR(6),
    AMBALAJSEKLIID INTEGER,
    AMBALAJSEKLIADI VARCHAR(10),
    IPLIKTALEPKULLANIMYERIID INTEGER,
    IPLIKTALEPKULLANIMYERIADI VARCHAR(50),
    TESLIMTARIHI TIMESTAMP,
    NOTU VARCHAR(255),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    ORMESIPARISDETAYNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    BOYASIPARISDETAYNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISNO VARCHAR(20),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select IplikTalepDetay.IplikTalepDetayId, IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId, IplikTalepDetay.BoyaSiparisDetayId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, IplikTalepDetay.HamId, Ham.Adi,
             IplikTalepDetay.Miktar,
             IplikTalepDetay.MelanjKodu, IplikTalepDetay.FlamKodu, IplikTalepDetay.Lot, IplikTalepDetay.AmbalajSekliId, AmbalajSekli.Adi,
             IplikTalepDetay.IplikTalepKullanimYeriId, IplikTalepKullanimYeri.Adi,
             IplikTalepDetay.TeslimEdilenMiktar, IplikTalepDetay.TeslimTarihi, IplikTalepDetay.Kapali, BoyaSiparis.FirmaId, Firma.Adi,
             IplikTalepDetay.Notu, OrmeSiparisDetay.Notu, BoyaSiparisDetay.Notu,
             LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr,
             Ik.Adi, IplikTalepDetay.InsertTarihi, SK.Adi, IplikTalepDetay.DegisimTarihi, IplikTalepDetay.ReferansUretimNo, IplikTalepDetay.BirimMaliyet, IplikTalepDetay.TeslimEdilenNetMiktar,
             BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.OzelTestler, Ham.KdvOrani, BoyaSiparis.Fason FROM IplikTalepDetay
    LEFT JOIN Ham ON Ham.HamId = IplikTalepDetay.HamId
    LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
    left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = IplikTalepDetay.AmbalajSekliId
    LEFT JOIN Firma ON Firma.FirmaId = BoyaSiparis.FirmaId
    left join IplikTalepKullanimYeri on IplikTalepKullanimYeri.IplikTalepKullanimYeriId = IplikTalepDetay.IplikTalepKullanimYeriId
    Left Join Kullanici IK on IK.KullaniciId = IplikTalepDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = IplikTalepDetay.KullaniciId
    LEFT JOIN OrmeSiparisDetay ON OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    LEFT JOIN BoyaSiparisDetay ON BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
    WHERE IplikTalepDetay.IplikTalepKayitId = :AIplikTalepKayitId
    order by IplikTalepDetay.IplikTalepDetayId
  into :IplikTalepDetayId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
       :HamId, :HamAdi, :Miktar,:MelanjKodu, :FlamKodu, :Lot,
       :AmbalajSekliId, :AmbalajSekliAdi, :IplikTalepKullanimYeriId, :IplikTalepKullanimYeriAdi,
       :TeslimEdilenMiktar, :Teslimtarihi, :Kapali, :SiparisFirmaId, :SiparisFirmaAdi,
       :Notu, :OrmeSiparisDetayNotu, :BoyaSiparisDetayNotu, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisNo,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :ReferansUretimNo, :BirimMaliyet,
       :TeslimEdilenNetMiktar, :IlaveIslemler, :OzelTestler, :HamKdvOrani, :Fason do
  begin
    if (Kapali = 1) then KapaliStr = 'Kapalı';
    else KapaliStr = '';


    if (Fason = 1) then FasonStr = 'Fason';
    else if (Fason = 0) then FasonStr = 'Satış';
    else FasonStr = '';

    KalanMiktar = Miktar - TeslimEdilenMiktar;

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    TeslimEdilenMiktar = Yuvarla(TeslimEdilenMiktar, 1000);
    suspend;
  end
end
