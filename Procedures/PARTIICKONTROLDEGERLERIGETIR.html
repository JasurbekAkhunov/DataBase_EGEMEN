CREATE PROCEDURE PARTIICKONTROLDEGERLERIGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIASAMALARIID INTEGER,
    SIRA INTEGER,
    PARTISINIFID INTEGER,
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    ICKONTROLTIPIID INTEGER,
    ICKONTROLTIPIADI VARCHAR(30),
    ONERILENDEGER VARCHAR(15),
    ISTENEN VARCHAR(50),
    SONUC VARCHAR(15),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select PartiAsamaIcKontrol.PartiAsamalariId, PartiAsamaIcKontrol.PartiSinifId, PartiAsamalari.Sira, PartiAsamaStok.AsamaId, Asama.Adi, PartiAsamaIcKontrol.IcKontrolTipiId,
             IcKontrolTipi.Adi, IcKontrolTipi.OnerilenDeger, PartiAsamaIcKontrol.Istenen, PartiAsamaIcKontrol.Sonuc, PartiAsamaIcKontrol.PersonelId, Personel.Adi,
             Sk.Adi, PartiAsamaIcKontrol.DegisimTarihi from PartiAsamaIcKontrol
  Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaIcKontrol.PartiAsamalariId
  Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaIcKontrol.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaIcKontrol.PartiSinifId
  Left Join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
  Left Join IcKontrolTipi on IcKontrolTipi.IcKontrolTipiId = PartiAsamaIcKontrol.IcKontrolTipiId
  Left Join Personel on Personel.PersonelId = PartiAsamaIcKontrol.PersonelId
  Left Join Kullanici SK on SK.KullaniciId = PartiAsamaIcKontrol.KullaniciId
  where PartiAsamaIcKontrol.PartiKayitId = :APartiKayitId
  order by PartiAsamaIcKontrol.PartiKayitId, PartiAsamalari.Sira
  into :PartiAsamalariId, :PartiSinifId, :Sira, :AsamaId, :AsamaAdi, :IcKontrolTipiId, :IcKontrolTipiAdi, :OnerilenDeger, :Istenen, :Sonuc,
       :PersonelId, :PersonelAdi, :KullaniciAdi, :DegisimTarihi do
  begin
    suspend;
  end
end
