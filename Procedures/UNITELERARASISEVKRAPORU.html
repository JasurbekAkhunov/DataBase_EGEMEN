CREATE PROCEDURE UNITELERARASISEVKRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AGONDERENUNITEID INTEGER,
    AALICIUNITEID INTEGER,
    AARACID VARCHAR(11),
    ASINIF VARCHAR(1))
RETURNS (
    UNITEIRSALIYEID INTEGER,
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    SEVKTARIHI TIMESTAMP,
    GONDERENUNITEID INTEGER,
    GONDERENUNITEADI VARCHAR(30),
    ALICIUNITEID INTEGER,
    ALICIUNITEADI VARCHAR(30),
    ARACID VARCHAR(11),
    SEVKSEFERNO INTEGER,
    SINIF VARCHAR(1),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
begin
  for Select UniteIrsaliye.UniteIrsaliyeId, UniteIrsaliye.IrsaliyeTarihi, UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo,
         UniteIrsaliye.SevkTarihi, UniteIrsaliye.GonderenUniteId, GU.Adi, UniteIrsaliye.AliciUniteId, AU.Adi, UniteIrsaliye.AracId,
         UniteIrsaliye.SevkSeferNo, UniteIrsaliye.Sinif, UniteIrsaliye.Notu from UniteIrsaliye
    left join Unite GU on GU.UniteId = UniteIrsaliye.GonderenUniteId
    left join Unite AU on AU.UniteId = UniteIrsaliye.AliciUniteId
    where UniteIrsaliye.IrsaliyeTarihi >= :ABaslangictarihi and UniteIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
          (UniteIrsaliye.GonderenUniteId = :AGonderenUniteId or :AGonderenUniteId is null) and
          (UniteIrsaliye.AliciUniteId = :AAliciUniteId or :AAliciUniteId is null) and
          (UniteIrsaliye.AracId = :AAracId or :AAracId is null) and
          (UniteIrsaliye.Sinif = :ASinif or :ASinif is null)
    into :UniteIrsaliyeId, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :SevkTarihi, :GonderenUniteId, :GonderenUniteAdi,
         :AliciUniteId, :AliciUniteAdi, :AracId, :SevkSeferNo, :Sinif, :Notu do suspend;
end
