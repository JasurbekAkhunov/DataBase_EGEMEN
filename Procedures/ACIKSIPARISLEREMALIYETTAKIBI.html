CREATE PROCEDURE ACIKSIPARISLEREMALIYETTAKIBI(
    AORMESIPARISDETAYID INTEGER)
RETURNS (
    ORMESIPARISDETAYID INTEGER,
    STOKHAREKETID INTEGER,
    BIRIMMALIYET DOUBLE PRECISION)
AS
declare variable SiparisMaliyetTakibiVar integer;
begin
  Select SiparisMaliyetTakibiVar from Sistem where SistemID = 1 into :SiparisMaliyetTakibiVar;

  if (SiparisMaliyetTakibiVar = 1) then
  begin
    for select OrmeSiparisDetay.OrmeSiparisDetayId from OrmeSiparisDetay
      where OrmeSiparisDetay.Kapali = 0 and (OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId or :AOrmeSiparisDetayId is Null) and
        (OrmeSiparisDetay.MaliyetHesaplaniyor = 0 or OrmeSiparisDetay.MaliyetHesaplaniyor is Null)
      into :OrmeSiparisDetayId do
    begin
      update OrmeSiparisDetay set MaliyetHesaplaniyor = 1 where OrmeSiparisDetayId = :OrmeSiparisDetayId;
      For Select StokHareket.StokHareketId from StokHareket
        where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId
        order by StokHareket.Tarih, StokHareket.UretimIslemTipiId, StokHareket.HareketTipi, StokHareket.StokHareketId asc
        into :StokHareketId do
      begin
        Update StokHareket Set BirimMaliyet = 0, NeredenUpdate = 4, LinkIslemi = 1 where StokHareket.StokHareketId = :StokHareketId;
        Select BirimMaliyet from Stokhareket where StokHareket.StokHareketId = :StokHareketId into :BirimMaliyet;
        Update HamStokMaster Set NeredenUpdate = 3, BirimFiyat = :BirimMaliyet where HamStokMaster.StokhareketID = :StokhareketID;
        suspend;
      end
      update OrmeSiparisDetay set MaliyetHesaplaniyor = 0 where OrmeSiparisDetayId = :OrmeSiparisDetayId;
    end
  end
end
