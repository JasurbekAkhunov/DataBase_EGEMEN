CREATE PROCEDURE TERMINGECIKMESEBEPANALIZI(
    ABASLANGICSEVKTARIHI TIMESTAMP,
    ABITISSEVKTARIHI TIMESTAMP,
    APARTITIPI INTEGER,
    AFASON INTEGER,
    AISLEMTIPIID INTEGER,
    AFIRMAID VARCHAR(20))
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    GECIKMETIPIADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    PARTITERMINTARIHI TIMESTAMP,
    SIPARISTERMINTARIHI TIMESTAMP,
    SIPARISREVIZETERMINTARIHI TIMESTAMP,
    SIPARISTARIHI TIMESTAMP,
    SIPARISINSERTTARIHI TIMESTAMP,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(50),
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    PARTIACILISTARIHI TIMESTAMP,
    DURUMU INTEGER,
    DURUMUSTR VARCHAR(20),
    SONSEVKTARIHI TIMESTAMP,
    IMALATINBASLADIGITARIH TIMESTAMP,
    HAMSTOKILKGELISTARIHI TIMESTAMP,
    HAMSTOKSONGELISTARIHI TIMESTAMP,
    LABRECETECALISMATARIHI TIMESTAMP,
    LABRECETEATISTARIHI TIMESTAMP,
    LABRECETEATISOKEYTARIHI TIMESTAMP,
    SONHAREKETTARIHI TIMESTAMP,
    SEVKIRSALIYENOVETARIHI VARCHAR(100))
AS
declare variable FPartiAsamalariID integer;
declare variable MaxGiristarihi TimeStamp;
declare variable MaxCikisTarihi TimeStamp;
declare variable IrsaliyeNo     IrsaliyeNoSize;
declare variable IrsaliyeTarihi TimeStamp;
begin
  For Select Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, GecikmeTipi.Adi, Parti.FirmaId, Firma.Adi, BoyaSiparis.BoyaSiparisYil,
             BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId, Parti.Termintarihi, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi,
             Parti.ReceteId, Recete.Adi, PartiStok.HamAdiToplami,
             PartiStok.Kilo, PartiStok.Metre, Parti.AcilisTarihi, PartiStok.Durumu, PartiDurum.Adi, PartiStok.SevkTarihi,
             PartiStok.PartiAsamalariID, LabReceteCalisma.CalismaTarihi, LabReceteAtis.InsertTarihi, LabReceteAtis.Tarihi,
             BoyaSiparis.SiparisTarihi, BoyaSiparisDetay.InsertTarihi,
             Min(HamStokIrsaliye.IrsaliyeTarihi), Max(HamStokIrsaliye.IrsaliyeTarihi) from PartiStok
  Left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  left join GecikmeTipi on GecikmeTipi.GecikmeTipiId = PartiStok.GecikmeTipiId
  Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  Left join Firma on Firma.FirmaId = Parti.FirmaId
  Left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = Recete.LabReceteAtisId
  left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
  Left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
  Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
  Left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
  where PartiStok.SevkTarihi >= :ABaslangicSevktarihi and PartiStok.SevkTarihi <= :ABitisSevktarihi and
        (Parti.TerminTarihi < PartiStok.SevkTarihi) and
        (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
        (Parti.IslemTipiID = :AIslemTipiID or :AIslemTipiID is null) and
        (Parti.Fason = :AFason or :AFason is null) and
        (Parti.PartiTipiId = :APartiTipi or :APartiTipi is null)
  group by Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, GecikmeTipi.Adi, Parti.FirmaId, Firma.Adi, BoyaSiparis.BoyaSiparisYil,
           BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId, Parti.Termintarihi, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi,
           Parti.ReceteId, Recete.Adi, PartiStok.HamAdiToplami,
           PartiStok.Kilo, PartiStok.Metre, Parti.AcilisTarihi, PartiStok.Durumu, PartiDurum.Adi, PartiStok.SevkTarihi,
           PartiStok.PartiAsamalariID, LabReceteCalisma.CalismaTarihi, LabReceteAtis.InsertTarihi, LabReceteAtis.Tarihi,
           BoyaSiparis.SiparisTarihi, BoyaSiparisDetay.InsertTarihi
  into :PartiYil, :PartiId, :PartiSinifId, :GecikmeTipiAdi, :FirmaId, :FirmaAdi, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId, :PartiTerminTarihi,
       :SiparisTerminTarihi, :SiparisRevizeTerminTarihi, :ReceteId, :ReceteAdi,
       :HamAdiToplami, :Kilo, :Metre, :PartiAcilisTarihi, :Durumu, :DurumuStr, :SonSevkTarihi, :FPartiAsamalariID,
       :LabReceteCalismaTarihi, :LabReceteAtisTarihi, :LabReceteAtisOkeyTarihi,
       :SiparisTarihi, :SiparisInsertTarihi, :HamStokIlkGelisTarihi, :HamStokSonGelisTarihi do
  begin
    ImalatinBasladigiTarih = Null;
    SonHareketTarihi = Null;

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    Select Min(PartiAsamaStok.GirisTarihi) from PartiAsamaStok where PartiAsamaStok.PartiSinifID = :PartiSinifID and PartiAsamaStok.GirisTarihi is not null
       into :ImalatinBasladigiTarih;

    if (FPartiAsamalariID is not null) then
    begin
      Select Max(PartiAsamaStokDetay.Giristarihi), Max(PartiAsamaStokDetay.CikisTarihi) from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
        into :MaxGirisTarihi, :MaxCikisTarihi;

      if (MaxCikisTarihi is not null) then SonHareketTarihi = MaxCikisTarihi;
      else SonHareketTarihi = MaxGirisTarihi;
    end

    SevkIrsaliyeNoveTarihi = '';

    for Select BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeNo from BoyaIrsaliyeDetay
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      where BoyaIrsaliyeDetay.PartiSinifID = :PartiSinifID
      group by BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeNo
      into :IrsaliyeTarihi, :IrsaliyeNo do
    begin
      if (Char_Length(SevkIrsaliyeNoveTarihi) < 70) then
      begin
        if (SevkIrsaliyeNoveTarihi <> '') then SevkIrsaliyeNoveTarihi = SevkIrsaliyeNoveTarihi || cast(', ' as varchar(2) character set UTF8);
        SevkIrsaliyeNoveTarihi = TimeStampToDateStr(IrsaliyeTarihi) || ' - ' || EgemenIntegerToStr(IrsaliyeNo);
      end
    end

    suspend;
  end
end
