CREATE PROCEDURE BOYASIPARISRENKGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISRENKNO INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISRENKNOSTR VARCHAR(50),
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEATISID INTEGER,
    LABRECETEATISNO VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    RENKADI VARCHAR(40),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select BoyaSiparisRenk.BoyaSiparisKayitId, BoyaSiparisRenk.BoyaSiparisRenkNo, BoyaSiparisRenk.LabReceteId,
             LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisRenk.LabReceteAtisId,
             LabReceteAtis.AtisNoStr, BoyaSiparisRenk.RenkAdi,
             BoyaSiparisRenk.MusteriRenkKoduveAdi,
             IK.Adi, BoyaSiparisRenk.InsertTarihi, Sk.Adi, BoyaSiparisRenk.DegisimTarihi from BoyaSiparisRenk
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisRenk.LabReceteId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisRenk.LabReceteAtisId
  left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
  Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisRenk.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisRenk.KullaniciId
  where BoyaSiparisRenk.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  order by BoyaSiparisRenk.BoyaSiparisRenkNo
  into :BoyaSiparisKayitId, :BoyaSiparisRenkNo, :LabReceteId, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisId,
       :LabReceteAtisNo, :RenkAdi,
       :MusteriRenkKoduveAdi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    BoyaSiparisRenkNoStr = '(' || EgemenIntToLzStr(BoyaSiparisRenkNo, 2) || ') - ' || RenkAdi;

    suspend;
  end
end
