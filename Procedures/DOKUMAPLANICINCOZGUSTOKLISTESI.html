CREATE PROCEDURE DOKUMAPLANICINCOZGUSTOKLISTESI
RETURNS (
    RAFID RAFKODUSIZE,
    URETICIADI ADI40,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    RENKID INTEGER,
    RENKADI ADI50,
    FASONSTOK EGEBOOLEAN,
    DEPOID DEPOKODUSIZE,
    DEPOADI ADI30,
    STOKFIRMAID FIRMAKODUSIZE,
    STOKFIRMAADI ADI100,
    STOKFIRMAADI2 ADI100,
    GIRISKAPADEDI INTEGER,
    NETGIRISMIKTARI BOYAMIKTARSIZE,
    KALANMIKTAR BOYAMIKTARSIZE,
    BRUTGIRISMIKTARI BOYAMIKTARSIZE,
    BRUTKALANMIKTAR BOYAMIKTARSIZE,
    MIKTARBIRIMSTR VARCHAR(5),
    LEVENTNO INTEGER,
    TELADEDI INTEGER,
    ORMERAFID INTEGER,
    ISEMRIID VARCHAR(15),
    PLANLANABILIRMIKTAR BOYAMIKTARSIZE,
    ATKILOTU VARCHAR(30),
    LOT VARCHAR(150),
    LEVENTTIPIID INTEGER,
    LEVENTTIPIADI ADI40,
    BARKODID VARCHAR(30),
    HASILISEMRIID VARCHAR(15),
    INDIGOISEMRIID VARCHAR(15),
    RECETEADI VARCHAR(40),
    LEVENTSIRANO INTEGER,
    SETTEKILEVENTADEDI INTEGER,
    KALITE1MIKTAR DOUBLE PRECISION,
    KALITE2MIKTAR DOUBLE PRECISION,
    KALITE3MIKTAR DOUBLE PRECISION,
    DEPOPARTINO VARCHAR(30))
AS
declare variable FPlanlananMiktar          BoyaMiktarSize;
declare variable FPlanlamaIcinCikisMiktari BoyaMiktarSize;
declare variable FNetCikisMiktari          BoyaMiktarSize;
declare variable FOrmeRafId                Integer;
declare variable shtFasonUretimeCikis      Smallint;
declare variable shtUretimeCikis           Smallint;
declare variable FBarkodId Varchar(30);
declare variable DokumaBarkodluGirisCikisTakibi integer;
begin
  select shtFasonUretimeCikis, shtUretimeCikis from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis, :shtUretimeCikis;
  select sistem.DokumaBarkodluGirisCikisTakibi from sistem  into :DokumaBarkodluGirisCikisTakibi;

  for select OrmeRaf.RafId, Uretici.Adi UreticiAdi, OrmeRaf.StokGirisNo, OrmeRaf.HamId, Ham.Adi HamAdi, OrmeRaf.RenkId, Renk.Adi RenkAdi, OrmeRaf.FasonStok,
       DepoRaf.DepoId, Depo.Adi DepoAdi, OrmeRaf.StokFirmaId, StokFirma.Adi StokFirmaAdi, StokFirma.Adi2 StokFirmaAdi2,
       OrmeRaf.GirisKapAdedi, OrmeRaf.NetGirisMiktari, OrmeRaf.NetCikisMiktari, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari KalanMiktar,
       OrmeRaf.BrutGirisMiktari, OrmeRaf.BrutGirisMiktari - OrmeRaf.BrutCikisMiktari BrutKalanMiktar,
       iif(OrmeRaf.MiktarBirim = 0, 'Kilo', 'Metre') MiktarBirimStr, OrmeRaf.LeventNo, OrmeRaf.TelAdedi, OrmeRaf.AtkiLotu, OrmeRaf.Lot, OrmeRaf.OrmeRafId,
       CozguIsEmri.IsEmriId, OrmeRaf.OrmeRafId, LeventTipi.LeventTipiId, LeventTipi.Adi, OrmeRaf.BarkodId, HasilIsEmri.IsEmriId, IndigoIsEmri.IsEmriId, Recete.Adi, OrmeRaf.LeventSiraNo,
       CozguIsEmri.SettekiLeventAdedi, OrmeRaf.DepoPartiNo
    from OrmeRaf
    join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
    join Depo on Depo.DepoId = DepoRaf.DepoId
    left join Ham on Ham.HamId = OrmeRaf.HamId
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join Firma StokFirma on StokFirma.Firmaid = OrmeRaf.StokFirmaid
    left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiID
    left join CozguIsEmri on CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId
    left join HasilIsEmri on HasilIsEmri.HasilIsEmriKayitId = OrmeRaf.HasilIsEmriKayitId
    left join IndigoIsEmri on IndigoIsEmri.IndigoIsEmriKayitId = OrmeRaf.IndigoIsEmriKayitId
    left join Recete on Recete.ReceteId = IndigoIsEmri.ReceteId
    left join LeventTipi on LeventTipi.LeventTipiId = OrmeRaf.LeventTipiId
    where OrmeRaf.StokVar = 1 and (Depo.CozguDeposu = 1 or Depo.HasilDeposu = 1 or Depo.IndigoDeposu = 1)
    into :RafId, :UreticiAdi, :StokGirisNo, :HamId, :HamAdi, :RenkId, :RenkAdi,
      :FasonStok, :DepoId, :DepoAdi, :StokFirmaId, :StokFirmaAdi, :StokFirmaAdi2, :GirisKapAdedi,
      :NetGirisMiktari, :FNetCikisMiktari, :KalanMiktar, :BrutGirisMiktari, :BrutKalanMiktar, :MiktarBirimStr, :LeventNo, :TelAdedi, :AtkiLotu, :Lot, :OrmeRafId,
      :IsEmriId, :FOrmeRafId, :LeventTipiId, :LeventTipiAdi, :BarkodId, :HasilIsEmriId, :IndigoIsEmriId, :ReceteAdi , :LeventSiraNo, :SettekiLeventAdedi, :DepoPartiNo do
    begin

      select Coalesce(Sum(Miktar),0) from (
      Select Max(StokHareket.Miktar) Miktar from StokHareket
      join DokumaStokPlanlama on DokumaStokPlanlama.DokumaPlanlamaId = StokHareket.DokumaPlanlamaId
      where (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis)
      and StokHareket.OrmeRafId =:FOrmeRafId and DokumaStokPlanlama.OrmeRafId = :FOrmeRafId
      group by StokHareket.StokHareketId ) into :FPlanlamaIcinCikisMiktari;

      select Coalesce(sum(DokumaStokPlanlama.Miktar),0) from DokumaStokPlanlama where DokumaStokPlanlama.OrmeRafId = :FOrmeRafId into :FPlanlananMiktar;

      Select Sum(Kalite1Miktar), Sum(Kalite2Miktar), Sum(Kalite3Miktar) from OrmeRafDetay
        left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketId
          where OrmeRafDetay.OrmeRafId = :OrmeRafId into :Kalite1Miktar, :Kalite2Miktar, :Kalite3Miktar;

      FBarkodId = Null;
      Select Substring(List(StokHareketDetay.Barkod, ',') From 1 for 30) from StokHareket
        Left Join StokHareketDetay on StokHareketDetay.Stokhareketid = StokHareket.Stokhareketid
        where StokGirisNo = :StokGirisNo into :FBarkodId;

      if ((FBarkodId is not Null) and (DokumaBarkodluGirisCikisTakibi =1)) then BarkodId = FBarkodId;
      PlanlanabilirMiktar = KalanMiktar + FPlanlamaIcinCikisMiktari - FPlanlananMiktar;

      if (PlanlanabilirMiktar > KalanMiktar) then
        PlanlanabilirMiktar = KalanMiktar;

      if (PlanlanabilirMiktar > 0) then
      begin
        suspend;
      end
    end
end
