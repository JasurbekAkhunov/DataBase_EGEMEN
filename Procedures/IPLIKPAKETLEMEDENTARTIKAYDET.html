CREATE PROCEDURE IPLIKPAKETLEMEDENTARTIKAYDET(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    AIPLIKPAKETLEMEID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable ParcaliSevk    integer;
declare variable TamamiTartildi integer;
declare variable KayitSayisi    integer;
declare variable NetKilo        Numeric(12,3);
declare variable Dara           Numeric(12,3);
declare variable KapAdedi       integer;
declare variable BobinSayisi    integer;
declare variable KaliteNo       integer;
declare variable IplikPaketlemeId       integer;
declare variable PartiStokTartiId       integer;
declare variable PaketlemeTarihi        TimeStamp;
begin
  Select TamamiTartildi from PartiStok where PartiStok.PartiSinifId = :APartiSinifId into :TamamiTartildi;

  if (TamamiTartildi = 1) then exception TamamiTartilmisKiloGiremezsiniz;

  Select ParcaliSevk from Parti where Parti.PartiKayitId = :APartiKayitId into :ParcaliSevk;
  KapAdedi = 1;

  For Select IplikPaketlemeId, KaliteNo, AmbalajAgirligi + PaletAgirligi, BobinSayisi, NetKilo, PaketlemeTarihi from IplikPaketleme
    where IplikPaketleme.PartiSinifId = :APartiSinifId and
          (IplikPaketleme.IplikPaketlemeId = :AIplikPaketlemeId or :AIplikPaketlemeId is null) and
           IplikPaketleme.PartiStokTartiId is null
    into :IplikPaketlemeId, :KaliteNo, :Dara, :BobinSayisi, :NetKilo, :PaketlemeTarihi do
  begin
    PartiStokTartiId = gen_id(PartiStokTarti_Id_Gen, 1);

    Insert Into PartiStokTarti(PartiStokTartiId, PartiKayitId, PartiSinifId, Tarih, Kilo, Dara, KapAdedi, BobinSayisi, KaliteKontrol, KaliteNo, InsertKullaniciId, KullaniciId)
                        values(:PartiStokTartiId, :APartiKayitId, :APartiSinifId, :PaketlemeTarihi, :NetKilo, :Dara, :KapAdedi, :BobinSayisi, 1, :KaliteNo, :AKullaniciId, :AKullaniciId);

    Update IplikPaketleme Set PartiStokTartiId = :PartiStokTartiId where IplikPaketleme.IplikPaketlemeId = :IplikPaketlemeId;
  end

  if (ParcaliSevk = 0) then
  begin
    Update PartiStok Set TamamiTartildi = 1 where PartiStok.PartiSinifId = :APartiSinifId;

    execute Procedure PartiStokTartiBrutHesapla(:APartiSinifId, 0);
  end
end
