CREATE PROCEDURE DESENRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ATASARIMFIRMAID VARCHAR(20),
    ASABLONSAYISI INTEGER,
    ADESENKATALOGID INTEGER,
    ADESENGRUPID VARCHAR(4),
    ADESENICERIKID INTEGER,
    ACALISMANO VARCHAR(20),
    ATASARIMNO VARCHAR(20))
RETURNS (
    DESENKAYITID INTEGER,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SABLONSAYISI INTEGER,
    CALISMANO VARCHAR(20),
    TASARIMNO VARCHAR(20),
    GELISTARIHI TIMESTAMP,
    DESENDURUMU INTEGER,
    DESENDURUMUSTR VARCHAR(25),
    DESENKATALOGID INTEGER,
    DESENKATALOGADI VARCHAR(50),
    DESENGRUPID VARCHAR(4),
    DESENGRUPADI VARCHAR(30),
    ESKIDESENNO VARCHAR(20),
    DESENICERIKID INTEGER,
    DESENICERIKADI VARCHAR(30),
    REFERANSDESENNO VARCHAR(20),
    REFERANSDESENADI VARCHAR(50),
    MUSTERIDESENNO1 VARCHAR(20),
    MUSTERIDESENADI1 VARCHAR(50),
    MUSTERIDESENNO2 VARCHAR(20),
    MUSTERIDESENADI2 VARCHAR(50),
    TASARIMFIRMAID VARCHAR(20),
    TASARIMFIRMAADI VARCHAR(100),
    DESINATORADI VARCHAR(50),
    SABLONBOYUID VARCHAR(10),
    DESENRAPORTBOYUID VARCHAR(10),
    DOLULUKORANI DOUBLE PRECISION,
    DESENOKEYTARIHI TIMESTAMP,
    PRINTVERILISTARIHI TIMESTAMP,
    SABLONCEKIMTARIHI TIMESTAMP,
    SABLONGONDERMETARIHI TIMESTAMP,
    SABLONDURUMUID INTEGER,
    SABLONDURUMUADI VARCHAR(30),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    KULLANILANPARTISAYISI DOUBLE PRECISION,
    KULLANILANTOPLAMMIKTAR DOUBLE PRECISION,
    BRUTSEVKMIKTAR DOUBLE PRECISION,
    NETSEVKMIKTAR DOUBLE PRECISION,
    BRUTSEVKKILO DOUBLE PRECISION,
    NETSEVKKILO DOUBLE PRECISION,
    BRUTSEVKMETRE DOUBLE PRECISION,
    NETSEVKMETRE DOUBLE PRECISION,
    BASKIYAGIRENTOPMIKTAR DOUBLE PRECISION,
    SABLONENIOLCU DESENOLCUSIZE,
    GRAVURBOYUOLCU DESENOLCUSIZE,
    OZELYERLESIMDOSYAADI VARCHAR(255),
    ARSIVNOTU VARCHAR(255),
    RAPORTLAMA VARCHAR(5),
    DESENCALISMASEKLIADI VARCHAR(30),
    DESENCASLIMASEKLIADI VARCHAR(30),
    SABLONTAMCEKIMSAYISI INTEGER,
    SABLONTAMIRCEKIMSAYISI INTEGER,
    SABLONKIRIKCEKIMSAYISI INTEGER,
    SABLONRENKAYRIMICEKIMSAYISI INTEGER,
    DESENSONSEVKTARIHI TIMESTAMP,
    KAPALI INTEGER)
AS
Declare Variable FMiktar Double Precision;
Declare Variable PartiKayitID Integer;
begin
  for select Desen.DesenKayitId, Desen.DesenId, Desen.Adi, Desen.FirmaId, DesenFirma.Adi, Desen.SablonSayisi, Desen.CalismaNo, Desen.TasarimNo, Desen.GelisTarihi, Desen.DesenKatalogId, DesenKatalog.Adi,
             Desen.DesenGrupId, DesenGrup.Adi, Desen.EskiDesenNo, Desen.DesenIcerikId, DesenIcerik.Adi, Desen.ReferansDesenNo, Desen.ReferansDesenAdi, Desen.MusteriDesenNo1,
             Desen.MusteriDesenAdi1, Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2, Desen.TasarimFirmaId, TasarimFirma.Adi, Desen.DesinatorAdi,  SablonB.Olcu,
             RaportB.Olcu, Desen.DolulukOrani, Desen.Notu,
             Desen.DesenOkeyTarihi, Desen.PrintVerilisTarihi, Desen.SablonCekimTarihi, Desen.SablonGondermeTarihi,
             Desen.SablonDurumuId, SablonDurumu.Adi,
             IK.Adi, Desen.InsertTarihi, SK.Adi, Desen.DegisimTarihi, Desen.DesenDurumu, SablonE.Olcu, GravurB.Olcu, Desen.OzelYerlesimDosyaAdi,
       Desen.ArsivNotu, Desen.Raportlama, DesenCalismaSekli.Adi, Desen.Kapali from Desen
    left join Firma DesenFirma on DesenFirma.FirmaId = Desen.FirmaId
    left join Firma TasarimFirma on TasarimFirma.FirmaId = Desen.TasarimFirmaId
    left join DesenCalismaSekli on DesenCalismaSekli.DesenCalismaSekliID = Desen.DesenCalismaSekliID
    left join DesenKatalog on DesenKatalog.DesenKatalogId = Desen.DesenKatalogId
    left join DesenGrup on DesenGrup.DesenGrupId = Desen.DesenGrupId
    left join DesenIcerik on DesenIcerik.DesenIcerikId = Desen.DesenIcerikId
    left join DesenOlcu SablonB on SablonB.DesenOlcuID = Desen.SablonBoyuDesenOlcuId
    left join DesenOlcu RaportB on RaportB.DesenOlcuID = Desen.DesenRaportBoyuDesenOlcuId
    left join DesenOlcu SablonE on SablonE.DesenOlcuID = Desen.SablonEniDesenOlcuId
    left join DesenOlcu GravurB on GravurB.DesenOlcuID = Desen.GravurBoyuDesenOlcuId
    left join SablonDurumu on SablonDurumu.SablonDurumuId = Desen.SablonDurumuId
    Left Join Kullanici IK on IK.KullaniciId = Desen.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = Desen.KullaniciId
    where Desen.GelisTarihi >= :ABaslangicTarihi and Desen.GelisTarihi <= :ABitisTarihi and
          (Desen.FirmaID = :AFirmaID or :AFirmaId is null) and
          (Desen.TasarimFirmaID = :ATasarimFirmaID or :ATasarimFirmaId is null) and
          (Desen.SablonSayisi = :ASablonSayisi or :ASablonSayisi is null) and
          (Desen.DesenKatalogId = :ADesenKatalogId or :ADesenKatalogId is null) and
          (Desen.DesenGrupId = :ADesenGrupId or :ADesenGrupId is null) and
          (Desen.DesenIcerikId = :ADesenIcerikId or :ADesenIcerikId is null) and
          (Desen.CalismaNo = :ACalismaNo or :ACalismaNo is null) and
          (Desen.TasarimNo = :ATasarimNo or :ATasarimNo is null)
  into :DesenKayitId, :DesenId, :DesenAdi, :FirmaId, :FirmaAdi, :SablonSayisi, :CalismaNo, :TasarimNo, :GelisTarihi, :DesenKatalogId, :DesenKatalogAdi,
       :DesenGrupId, :DesenGrupAdi, :EskiDesenNo, :DesenIcerikId, :DesenIcerikAdi, :ReferansDesenNo, :ReferansDesenAdi, :MusteriDesenNo1,
       :MusteriDesenAdi1, :MusteriDesenNo2, :MusteriDesenAdi2, :TasarimFirmaId, :TasarimFirmaAdi, :DesinatorAdi, :SablonBoyuId,
       :DesenRaportBoyuId, :DolulukOrani, :Notu, :DesenOkeyTarihi, :PrintVerilisTarihi, :SablonCekimTarihi, :SablonGondermeTarihi,
       :SablonDurumuId, :SablonDurumuAdi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :DesenDurumu, :SablonEniOlcu, :GravurBoyuOlcu, :OzelYerlesimDosyaAdi, :ArsivNotu,
       :Raportlama, :DesenCalismaSekliAdi, :Kapali do
  begin
    if (DesenDurumu = 0) then DesenDurumuStr = 'Exclusive';
    if (DesenDurumu = 1) then DesenDurumuStr = 'Açık';
    if (DesenDurumu = 2) then DesenDurumuStr = 'Coğrafi';
    if (DesenDurumu = 3) then DesenDurumuStr = 'Bölgeye Sınırlı';
    if (DesenDurumu = 4) then DesenDurumuStr = 'Opsiyonlu';
    else DesenDurumuStr = '';



    Select Count(Parti.PartiKayitID), Sum(PartiStok.Miktar), Sum(PartiStok.BrutSevkMiktar) , Sum(PartiStok.NetSevkMiktar),
    Sum(PartiStok.BrutSevkKilo), Sum(PartiStok.NetSevkKilo) , Sum(PartiStok.BrutSevkMetre),Sum(PartiStok.NetSevkMetre) from DesenVaryant
      Left Join Parti on Parti.DesenVaryantID = DesenVaryant.DesenVaryantID
      Left Join PartiStok on PartiStok.PartiKayitID = Parti.PartiKayitID
      Where DesenVaryant.DesenID = :DesenId and Parti.PartiKayitID is not Null
      into :KullanilanPartiSayisi, :KullanilanToplamMiktar, :BrutSevkMiktar, :NetSevkMiktar,:BrutSevkKilo, :NetSevkKilo,:BrutSevkMetre, :NetSevkMetre;

    BaskiyaGirenTopMiktar = 0;

    For Select Parti.PartiKayitID from DesenVaryant
      Left Join Parti on Parti.DesenVaryantID = DesenVaryant.DesenVaryantID
      Where DesenVaryant.DesenID = :DesenId and Parti.PartiKayitID is not Null into :PartiKayitID do
    begin
      For Select First 1 GirisMiktar from PartiStok
        Left Join PartiAsamaStok on PartiAsamaStok.PartiSinifID = PartiStok.PartiSinifID
        Left Join Asama  on Asama.AsamaID = PartiAsamaStok.AsamaID
        Where PartiStok.PartiKayitID = :PartiKayitID and Asama.BaskiAsamasimi = 1 and PartiAsamaStok.GirisTarihi is not null
        into :FMiktar do
      begin
        BaskiyaGirenTopMiktar = BaskiyaGirenTopMiktar + FMiktar;
      end
    end

    Select sum (desenistektipi.sabloncekimmi) , sum (desenistektipi.tamir),sum (desenistektipi.kirik),sum (desenistektipi.renkayrimistegimi) from desenistekleri
      left join desenistektipi on desenistektipi.desenistektipiid = desenistekleri.desenistektipiid
      where  DesenIstekleri.DesenID = :DesenId and desenistekleri.stoktandusuldu=1
      group by desenistekleri.desenid
      into :SablonTamCekimSayisi, :SablonTamirCekimSayisi, :SablonKirikCekimSayisi, :SablonRenkAyrimiCekimSayisi;


    Select  max(PartiStok.SevkTarihi)  from DesenVaryant
      left Join Parti on Parti.DesenVaryantID = DesenVaryant.DesenVaryantID
      left Join PartiStok on PartiStok.PartiKayitID = Parti.PartiKayitID
      where DesenVaryant.DesenID = :DesenId and Parti.PartiKayitID is not Null into :DesenSonSevkTarihi;

    suspend;
  end
end
