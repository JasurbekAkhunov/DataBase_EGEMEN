CREATE PROCEDURE VALORLUCARIKARTFIRMADOKUMU(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1))
RETURNS (
    FIRMAID VARCHAR(20),
    ADI VARCHAR(100),
    ADI2 VARCHAR(100),
    FIRMAGRUP1 VARCHAR(4),
    FIRMAGRUP2 VARCHAR(4),
    FIRMAGRUP3 VARCHAR(4),
    ADRESI1 VARCHAR(40),
    ADRESI2 VARCHAR(40),
    ADRESI3 VARCHAR(40),
    SEMT VARCHAR(20),
    SEHIR VARCHAR(20),
    VERGIDAIRESI VARCHAR(20),
    VERGIHESAPNO VARCHAR(20),
    DOVIZID VARCHAR(5),
    ACIKHESAPRISKI DOUBLE PRECISION,
    CEKSENETRISKI DOUBLE PRECISION,
    SIPARISRISKI DOUBLE PRECISION,
    TAHSILATHAFTA1 INTEGER,
    TAHSILATHAFTA2 INTEGER,
    TAHSILATHAFTA3 INTEGER,
    TAHSILATHAFTA4 INTEGER,
    TAHSILATGUN1 INTEGER,
    TAHSILATGUN2 INTEGER,
    TAHSILATGUN3 INTEGER,
    TAHSILATGUN4 INTEGER,
    TAHSILATGUN5 INTEGER,
    TAHSILATGUN6 INTEGER)
AS
declare variable okey integer;
declare variable KayitSayisi integer;
begin
  for Select Firma.FirmaId, Adi, Adi2, FirmaGrup1, FirmaGrup2, FirmaGrup3, Adresi1, Adresi2, Adresi3, Semt, Sehir, VergiDairesi, VergiHesapNo, DovizId, AcikHesapRiski, CekSenetRiski, SiparisRiski,
  TahsilatHafta1, TahsilatHafta2, TahsilatHafta3, TahsilatHafta4, TahsilatGun1, TahsilatGun2, TahsilatGun3, TahsilatGun4, TahsilatGun5, TahsilatGun6 from Firma
  where ((:AFirmaAralikli = 1 and FirmaId >= :ABaslangicFirmaId and (FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0))  and
        ((:AFirmaGrup1Aralikli = 1 and FirmaGrup1 >= :ABaslangicGrup1Id  and (FirmaGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and FirmaGrup2 >= :ABaslangicGrup2Id and (FirmaGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and FirmaGrup3 >= :ABaslangicGrup3Id and (FirmaGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = ''))
  into :FirmaId, :Adi, :Adi2, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :Adresi1, :Adresi2, :Adresi3, :Semt, :Sehir, :VergiDairesi, :VergiHesapNo, :DovizId, :AcikHesapRiski, :CekSenetRiski, :SiparisRiski,
  :TahsilatHafta1, :TahsilatHafta2, :TahsilatHafta3, :TahsilatHafta4, :TahsilatGun1, :TahsilatGun2, :TahsilatGun3, :TahsilatGun4, :TahsilatGun5, :TahsilatGun6 do
  begin
    Okey = 1;
    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (okey = 1) then
    begin
      select count(*) from cari where Cari.FirmaId = :FirmaId and Cari.Tarih >= :ABaslangicTarihi and Cari.tarih <= :ABitisTarihi and (cari.Sinif = :ASinif or :ASinif is null) into :KayitSayisi;
      if (KayitSayisi = 0) then okey = 0;
    end
    if (okey = 1) then suspend;
  end

end
