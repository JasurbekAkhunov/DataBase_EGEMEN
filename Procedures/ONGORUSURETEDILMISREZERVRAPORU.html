CREATE PROCEDURE ONGORUSURETEDILMISREZERVRAPORU(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    HAMSTOKDETAYID INTEGER,
    HAMSTOKIRSALIYEID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30),
    ORMECIMAKINANO VARCHAR(10),
    HAMADI VARCHAR(255),
    PUS INTEGER,
    FEINE INTEGER,
    KUMASCINSIADI VARCHAR(40),
    REZERVKILO NUMERIC(12,3),
    REZERVMETRE NUMERIC(12,3),
    REZERVADET INTEGER,
    YIKAMAONCESIEN DOUBLE PRECISION,
    YIKAMASONRASIEN DOUBLE PRECISION,
    ENISTENEN VARCHAR(10),
    ONGORULENEN INTEGER,
    YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    YIKAMASONRASIGRAMAJ DOUBLE PRECISION,
    GRAMAJISTENEN VARCHAR(10),
    ONGORULENGRAMAJ INTEGER,
    ENDENCEKMEZLIK DOUBLE PRECISION,
    BOYDANCEKMEZLIK DOUBLE PRECISION,
    CEKMEZLIKENISTENEN DOUBLE PRECISION,
    CEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    ONGORULENENDENCEKMEZLIK DOUBLE PRECISION,
    ONGORULENBOYDANCEKMEZLIK DOUBLE PRECISION,
    YIKAMAHASLIGIISTENEN VARCHAR(10),
    HAMDONME DOUBLE PRECISION,
    DONMEISTENEN VARCHAR(10),
    ONGORULENDONME VARCHAR(10),
    YIKAMAONCESIRAPORBOYU VARCHAR(100),
    YIKAMASONRASIRAPORBOYU VARCHAR(100),
    ISTENENRAPORBOYU VARCHAR(100),
    ONGORULENRAPORBOYU VARCHAR(1000),
    YIKAMAONCESICPI VARCHAR(10),
    YIKAMASONRASICPI VARCHAR(10),
    YIKAMAONCESIWPI VARCHAR(10),
    YIKAMASONRASIWPI VARCHAR(10),
    HAMKONTROLIPLIKUZUNLUGU1 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU2 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU3 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU4 VARCHAR(40),
    ONGORUNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ONAYNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    NEMORANI DOUBLE PRECISION,
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    SIPARISNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PLANLAMANOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KONTROLEDENKULLANICIADI VARCHAR(40),
    KONTROLTARIHI TIMESTAMP,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP)
AS
begin
  if (ARaporTipi is null) then
  begin
    For Select BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetayOngoru.BoyaSiparisKayitId, BoyaSiparisDetayOngoru.BoyaSiparisDetayId, BoyaSiparisDetayOngoru.HamStokDetayId,
               HamStokDetay.HamStokIrsaliyeId, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
               Ormeci.Adi, HamStokMaster.OrmeciMakinaNo, Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, KumasCinsi.Adi,
               HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaSonrasiEn, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetayOngoru.OngorulenEn,
               HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiGramaj, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetayOngoru.OngorulenGramaj,
               HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen, BoyaSiparisDetay.YikamaHasligiIstenen,
               BoyaSiparisDetayOngoru.OngorulenEndenCekmezlik, BoyaSiparisDetayOngoru.OngorulenBoydanCekmezlik, HamStokMaster.HamDonme, BoyaSiparisDetay.DonmeIstenen,
               BoyaSiparisDetayOngoru.OngorulenDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu, BoyaSiparisDetay.IstenenRaporBoyu,
               HamStokMaster.YikamaOncesiCPI, HamStokMaster.YikamaSonrasiCPI, HamStokMaster.YikamaOncesiWPI, HamStokMaster.YikamaSonrasiWPI,
               HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2, HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4,
               BoyaSiparisDetayOngoru.OngorulenRaporBoyu, BoyaSiparisDetayOngoru.OngoruNotu, BoyaSiparisDetayOngoru.OnayNotu,
               HamStokMaster.NemOrani, BoyaSiparisDetayOngoru.KontrolTarihi, OKK.Adi,
               BoyaSiparisDetayOngoru.OnayDurumu, BoyaSiparisDetayOngoru.OnaylayanKullaniciId, OOK.Adi, BoyaSiparisDetayOngoru.OnayTarihi,
               BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu from BoyaSiparisDetayOngoru
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisDetayOngoru.BoyaSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
      left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisDetayOngoru.HamStokDetayId
      Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
      left join Firma on Firma.Firmaid = hamstokIrsaliye.Firmaid
      left join Ormeci on Ormeci.Ormeciid = HamstokIrsaliye.Ormeciid
      left join Ham on Ham.Hamid = HamStokDetay.Hamid
      left join KumasCinsi on KumasCinsi.KumasCinsiid = Ham.KumasCinsiId
      Left Join Kullanici OKK on OKK.KullaniciId = BoyaSiparisDetayOngoru.KontrolKullaniciId
      Left Join Kullanici OOK on OOK.KullaniciId = BoyaSiparisDetayOngoru.OnaylayanKullaniciId
      where BoyaSiparisDetayOngoru.KontrolTarihi >= :ABaslangicTarihi and BoyaSiparisDetayOngoru.KontrolTarihi <= :ABitisTarihi and
            BoyaSiparisDetayOngoru.KontrolDurumu = 2
      into :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :HamStokDetayId, :HamStokIrsaliyeId, :FirmaId, :FirmaAdi, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi,
           :OrmeciAdi, :OrmeciMakinaNo, :HamAdi, :Pus, :Feine, :KumasCinsiAdi, :YikamaOncesiEn, :YikamaSonrasiEn, :EnIstenen, :OngorulenEn,
           :YikamaOncesiGramaj, :YikamaSonrasiGramaj, :GramajIstenen, :OngorulenGramaj, :EndenCekmezlik, :BoydanCekmezlik, :CekmezlikEnIstenen,
           :CekmezlikBoyIstenen,  :YikamaHasligiIstenen, :OngorulenEndenCekmezlik, :OngorulenBoydanCekmezlik, :HamDonme, :DonmeIstenen, :OngorulenDonme,
           :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu, :IstenenRaporBoyu,
           :YikamaOncesiCPI, :YikamaSonrasiCPI, :YikamaOncesiWPI, :YikamaSonrasiWPI, :HamKontrolIplikUzunlugu1, :HamKontrolIplikUzunlugu2,
           :HamKontrolIplikUzunlugu3, :HamKontrolIplikUzunlugu4,
           :OngorulenRaporBoyu, :OngoruNotu, :OnayNotu, :NemOrani, :KontrolTarihi, :KontrolEdenKullaniciAdi,
           :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi,
           :SiparisProsesId, :SiparisProsesAdi, :SiparisNotu, :PlanlamaNotu do
    begin
      if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';

      Select Sum(Kilo), Sum(Metre), Sum(Adet) from BoyaSiparisRezerv
      where BoyaSiparisRezerv.BoyaSiparisDetayId = :BoyaSiparisDetayId and BoyaSiparisRezerv.HamStokDetayId = :HamStokDetayId
      into :RezervKilo, :RezervMetre, :RezervAdet;

      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
      else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
      else EvrakTipiStr = 'Hatalı';

      suspend;
    end
  end
  else
  begin
    For Select BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetayOngoru.BoyaSiparisKayitId, BoyaSiparisDetayOngoru.BoyaSiparisDetayId, BoyaSiparisDetayOngoru.HamStokDetayId,
               HamStokDetay.HamStokIrsaliyeId, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
               Ormeci.Adi, HamStokMaster.OrmeciMakinaNo, Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, KumasCinsi.Adi,
               HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaSonrasiEn, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetayOngoru.OngorulenEn,
               HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiGramaj, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetayOngoru.OngorulenGramaj,
               HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen, BoyaSiparisDetay.YikamaHasligiIstenen,
               BoyaSiparisDetayOngoru.OngorulenEndenCekmezlik, BoyaSiparisDetayOngoru.OngorulenBoydanCekmezlik, HamStokMaster.HamDonme, BoyaSiparisDetay.DonmeIstenen,
               BoyaSiparisDetayOngoru.OngorulenDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu, BoyaSiparisDetay.IstenenRaporBoyu,
               HamStokMaster.YikamaOncesiCPI, HamStokMaster.YikamaSonrasiCPI, HamStokMaster.YikamaOncesiWPI, HamStokMaster.YikamaSonrasiWPI,
               HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2, HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4,
               BoyaSiparisDetayOngoru.OngorulenRaporBoyu, BoyaSiparisDetayOngoru.OngoruNotu, BoyaSiparisDetayOngoru.OnayNotu,
               HamStokMaster.NemOrani, BoyaSiparisDetayOngoru.KontrolTarihi, OKK.Adi,
               BoyaSiparisDetayOngoru.OnayDurumu, BoyaSiparisDetayOngoru.OnaylayanKullaniciId, OOK.Adi, BoyaSiparisDetayOngoru.OnayTarihi,
               BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu from BoyaSiparisDetayOngoru
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisDetayOngoru.BoyaSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
      left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisDetayOngoru.HamStokDetayId
      Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
      left join Firma on Firma.Firmaid = hamstokIrsaliye.Firmaid
      left join Ormeci on Ormeci.Ormeciid = HamstokIrsaliye.Ormeciid
      left join Ham on Ham.Hamid = HamStokDetay.Hamid
      left join KumasCinsi on KumasCinsi.KumasCinsiid = Ham.KumasCinsiId
      Left Join Kullanici OKK on OKK.KullaniciId = BoyaSiparisDetayOngoru.KontrolKullaniciId
      Left Join Kullanici OOK on OOK.KullaniciId = BoyaSiparisDetayOngoru.OnaylayanKullaniciId
      where BoyaSiparisDetayOngoru.KontrolTarihi >= :ABaslangicTarihi and BoyaSiparisDetayOngoru.KontrolTarihi <= :ABitisTarihi and
            BoyaSiparisDetayOngoru.KontrolDurumu = 2 and BoyaSiparisDetayOngoru.OnayDurumu = :ARaporTipi
      into :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :HamStokDetayId, :HamStokIrsaliyeId, :FirmaId, :FirmaAdi, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi,
           :OrmeciAdi, :OrmeciMakinaNo, :HamAdi, :Pus, :Feine, :KumasCinsiAdi, :YikamaOncesiEn, :YikamaSonrasiEn, :EnIstenen, :OngorulenEn,
           :YikamaOncesiGramaj, :YikamaSonrasiGramaj, :GramajIstenen, :OngorulenGramaj, :EndenCekmezlik, :BoydanCekmezlik, :CekmezlikEnIstenen,
           :CekmezlikBoyIstenen,  :YikamaHasligiIstenen, :OngorulenEndenCekmezlik, :OngorulenBoydanCekmezlik, :HamDonme, :DonmeIstenen, :OngorulenDonme,
           :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu, :IstenenRaporBoyu,
           :YikamaOncesiCPI, :YikamaSonrasiCPI, :YikamaOncesiWPI, :YikamaSonrasiWPI, :HamKontrolIplikUzunlugu1, :HamKontrolIplikUzunlugu2,
           :HamKontrolIplikUzunlugu3, :HamKontrolIplikUzunlugu4,
           :OngorulenRaporBoyu, :OngoruNotu, :OnayNotu, :NemOrani, :KontrolTarihi, :KontrolEdenKullaniciAdi,
           :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi,
           :SiparisProsesId, :SiparisProsesAdi, :SiparisNotu, :PlanlamaNotu do
    begin

      if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';

      Select Sum(Kilo), Sum(Metre), Sum(Adet) from BoyaSiparisRezerv
      where BoyaSiparisRezerv.BoyaSiparisDetayId = :BoyaSiparisDetayId and BoyaSiparisRezerv.HamStokDetayId = :HamStokDetayId
      into :RezervKilo, :RezervMetre, :RezervAdet;

      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
      else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
      else EvrakTipiStr = 'Hatalı';
      suspend;
    end
  end
end
