CREATE PROCEDURE DEPOSAYIMDETAYIGETIR(
    ADEPOSAYIMID INTEGER)
RETURNS (
    DEPOSAYIMDETAYID INTEGER,
    DEPOSAYIMID INTEGER,
    BARKOD VARCHAR(30),
    KILO BOYAMIKTARSIZE,
    BRUTKILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    KAPADEDI INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(30),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    KALITENO INTEGER,
    DESENVARYANTID INTEGER,
    VARYANTADI VARCHAR(30),
    DESENID DESENKODUSIZE,
    DESENADI VARCHAR(50),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    RAFID RAFKODUSIZE)
AS
begin
  for select DepoSayimDetay.DepoSayimDetayID, DepoSayimDetay.DepoSayimID, DepoSayimDetay.Kilo, DepoSayimDetay.BrutKilo, DepoSayimDetay.Metre, DepoSayimDetay.KapAdedi,
             DepoSayimDetay.Barkod, DepoSayimDetay.HamID, Ham.HamTipiId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4,
             DepoSayimDetay.RenkId, Renk.Adi, DepoSayimDetay.Lot, DepoSayimDetay.MelanjKodu,
             DepoSayimDetay.FlamKodu, DepoSayimDetay.Notu,DepoSayimDetay.KaliteNo,
             DepoSayimDetay.DesenId, SHDesen.Adi, SHDesenVaryant.VaryantAdi, DepoSayimDetay.PartiKayitId, DepoSayimDetay.PartiSinifID, Parti.PartiYil, Parti.PartiID,
             DepoSayimDetay.RafId from DepoSayimDetay
  left join Parti on Parti.PartiKayitId = DepoSayimDetay.PartiKayitId
  left join Ham on Ham.HamId = DepoSayimDetay.HamId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  Left join Renk on Renk.RenkID = DepoSayimDetay.RenkId
  left join Desen ShDesen on ShDesen.DesenId = DepoSayimDetay.DesenId
  left join DesenVaryant SHDesenVaryant on SHDesenVaryant.DesenVaryantId = DepoSayimDetay.DesenVaryantId
  where DepoSayimDetay.DepoSayimID = :ADepoSayimID
  into :DepoSayimDetayId, :DepoSayimId, :Kilo, :BrutKilo, :Metre, :KapAdedi, :Barkod, :HamId, :HamTipiId, :HamAdi, :HamAdi2, :HamAdi3, :HamAdi4, :RenkId, :RenkAdi,
       :Lot, :MelanjKodu, :FlamKodu, :Notu, :KaliteNo, :DesenId, :DesenAdi, :VaryantAdi, :PartiKayitId, :PartiSinifID, :PartiYil, :PartiID, :RafId do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    suspend;
  end
end
