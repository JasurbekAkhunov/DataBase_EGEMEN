CREATE PROCEDURE BOYATALIMATIOLMAYANSEVKRAPORU(
    ABASLANGICSEVKTARIHI TIMESTAMP,
    ABITISSEVKTARIHI TIMESTAMP,
    ASEVKFIRMAID VARCHAR(20),
    KENDIBOYAHANEMIZIGOSTERME SMALLINT)
RETURNS (
    STOKHAREKETID INTEGER,
    IRSALIYENO VARCHAR(30),
    SEVKFIRMAID VARCHAR(20),
    SEVKFIRMAADI VARCHAR(50),
    IRSALIYETARIHI TIMESTAMP,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    DEPOID VARCHAR(10),
    LOT VARCHAR(150),
    MIKTAR DOUBLE PRECISION,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    TALIMATVERILENMIKTAR DOUBLE PRECISION,
    TALIMATVERILMEYENMIKTAR DOUBLE PRECISION,
    IADEMIKTAR DOUBLE PRECISION,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER)
AS
declare variable uitBoya integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretimeCikis integer;
declare variable FAlinanSiparisDetayID integer;
declare variable OrmeFirmaKodumuz varchar(20);
declare variable FirmaKodumuz varchar(20);
begin
  Select OrmeFirmaKodumuz, FirmaKodumuz from sistem where sistemid = 1 into :OrmeFirmaKodumuz, :FirmaKodumuz;
  select shtFasonUretimeCikis,  shtFasonUretim from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis,  :shtFasonUretim;
  select uitBoya from UretimIslemTipiGetir into :uitBoya;

  For Select StokHareket.StokHareketID, (OrmeIrsaliye.SeriNo ||  OrmeIrsaliye.IrsaliyeNo) IrsaliyeNo, OrmeIrsaliye.FirmaID, Firma.Adi, OrmeIrsaliye.IrsaliyeTarihi, StokHareket.RenkID, Renk.Adi,
         StokHareket.DepoID, StokHareket.Lot, StokHareket.Miktar, StokHareket.HamID, Ham.Adi, StokHareket.BoyaSiparisDetayID, StokHareket.BoyaSiparisBaskiID, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId  from StokHareket
     Left Join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeID = StokHareket.OrmeIrsaliyeID
     Left Join Ham on Ham.HamID = StokHareket.HamID
     Left Join Firma on Firma.FirmaID = OrmeIrsaliye.FirmaID
     Left Join Renk on Renk.RenkID = StokHareket.RenkID
     left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
  where StokHareket.Tarih >= :ABaslangicSevkTarihi and StokHareket.Tarih <= :ABitisSevkTarihi and
        (StokHareket.FirmaId = :ASevkFirmaId or :ASevkFirmaId is null) and
        ((OrmeIrsaliye.FirmaID <> :FirmaKodumuz and OrmeIrsaliye.FirmaID <> :OrmeFirmaKodumuz and :KendiBoyahanemiziGosterme = 1) or :KendiBoyahanemiziGosterme = 0) and
        (StokHareket.HareketTipi = :shtFasonUretimeCikis) and
        (StokHareket.UretimIslemTipiID = :uitBoya) and StokHareket.BoyaTalimatKayitID is Null and
        (Miktar > (Select iif(Sum(GidenMiktar) is null, 0, Sum(GidenMiktar)) from BoyaTalimat where BoyaTalimat.UretimeCikisStokHareketID = StokHareket.StokHareketID))
        order by StokHareket.Tarih, StokHareket.StokHareketId
  into :StokHareketId, :IrsaliyeNo, :SevkFirmaId, :SevkFirmaAdi, :IrsaliyeTarihi, :RenkId, :RenkAdi,
       :DepoID, :Lot, :Miktar, :HamID, :HamAdi, :BoyaSiparisDetayID, :BoyaSiparisBaskiID, :BoyaSiparisYil, :BoyaSiparisId do
  begin
      Select Sum(GidenMiktar) from BoyaTalimat where BoyaTalimat.UretimeCikisStokHareketID = :StokHareketId into :TalimatVerilenMiktar;
      /*Select Sum(Miktar) from StokHareket where  StokHareket.HareketTipi = :shtFasonUretim and StokHareket.UretimIslemTipiID = :uitBoya and HamIade = 1 and BoyaSiparisDetayID = :BoyaSiparisDetayID into :IadeMiktar;*/
      Select Sum(Miktar) from StokHareket where  StokHareket.HareketTipi = :shtFasonUretim and StokHareket.UretimIslemTipiID = :uitBoya and HamIade = 1 and StokHareket.ReferansCikisStokHareketID = :StokHareketID into :IadeMiktar;

      if (TalimatVerilenMiktar is Null) then TalimatVerilenMiktar = 0;
      if (IadeMiktar is Null) then IadeMiktar = 0;
      TalimatVerilmeyenMiktar = Miktar - TalimatVerilenMiktar - IadeMiktar;

      BoyaSiparisIdStr = '';
      if (BoyaSiparisYil is not null) then BoyaSiparisIdStr = '(' || Format_Float('#0', BoyaSiparisYil) || ') ' || Format_Float('#0', BoyaSiparisId);

      suspend;
  end
end
