CREATE PROCEDURE ORMESIPARISDETAYKOPYALA(
    AORMESIPARISDETAYID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable ABoyaSiparisDetayId  integer;
declare variable BoyaSiparisDetayId   integer;
declare variable OrmeSiparisDetayId   integer;
declare variable SiparisKopyalarkenFiyatiTasi integer;
declare variable BirimFiyat Double Precision;
declare variable FFasonBirimFiyati Double Precision;
declare variable FBaskiBirimFiyati Double Precision;
begin
  Select SiparisKopyalarkenFiyatiTasi from sistem where sistemid = 1 into :SiparisKopyalarkenFiyatiTasi;

  OrmeSiparisDetayId = gen_id(OrmeSiparisDetay_Id_Gen, 1);

  if (SiparisKopyalarkenFiyatiTasi = 0) then
  begin
    Insert Into OrmeSiparisDetay(BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, MelanjKodu, FlamKodu, HamMiktar, Metre, SatinAlinacakMiktar, StoktanKullanilacakMiktar, SatisFiyati,
                                   DovizId, OrmeBirimMaliyet, OrmeBirimMaliyetDovizId, TeslimTarihi, RevizeTeslimTarihi,
                                   VadeGun, OdemeSekliId, MakinaUstuGramaj, DinlenmisGramaj, IgneSayisi, MaySayisi, Pus, Feine, TupAcikEnMayli, ArtikelNo,
                                   Beyazlik, TahminiFireOrani, IplikBoyu1, IplikBoyu2, IplikBoyu3, IplikBoyu4, RaporBoyu, SevkYeri, MusteriTalimati, Notu, KaliteId, KopyaId, HamStokBirimId,
                                   InsertKullaniciId, KullaniciId, PaketlemeSekliId)
         Select BoyaSiparisKayitId, :OrmeSiparisDetayId, HamId, RenkId, MelanjKodu, FlamKodu, HamMiktar, Metre, SatinAlinacakMiktar, StoktanKullanilacakMiktar, 0,
                 DovizId, 0, OrmeBirimMaliyetDovizId, TeslimTarihi, RevizeTeslimTarihi,
                 VadeGun, OdemeSekliId, MakinaUstuGramaj, DinlenmisGramaj, IgneSayisi, MaySayisi, Pus, Feine, TupAcikEnMayli, ArtikelNo,
                 Beyazlik, TahminiFireOrani, IplikBoyu1, IplikBoyu2, IplikBoyu3, IplikBoyu4, RaporBoyu, SevkYeri, MusteriTalimati, Notu, KaliteId, OrmeSiparisDetayId, HamStokBirimId, :AKullaniciId, :AKullaniciId,
                 PaketlemeSekliId from OrmeSiparisDetay
             where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId;
  end
  else
  begin
    Insert Into OrmeSiparisDetay(BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, MelanjKodu, FlamKodu, HamMiktar, Metre, SatinAlinacakMiktar, StoktanKullanilacakMiktar, SatisFiyati,
                                   DovizId, OrmeBirimMaliyet, OrmeBirimMaliyetDovizId, TeslimTarihi, RevizeTeslimTarihi,
                                   VadeGun, OdemeSekliId, MakinaUstuGramaj, DinlenmisGramaj, IgneSayisi, MaySayisi, Pus, Feine, TupAcikEnMayli, ArtikelNo,
                                   Beyazlik, TahminiFireOrani, IplikBoyu1, IplikBoyu2, IplikBoyu3, IplikBoyu4, RaporBoyu, SevkYeri, MusteriTalimati, Notu, KaliteId, KopyaId, HamStokBirimId,
                                   InsertKullaniciId, KullaniciId, PaketlemeSekliId)
         Select BoyaSiparisKayitId, :OrmeSiparisDetayId, HamId, RenkId, MelanjKodu, FlamKodu, HamMiktar, Metre, SatinAlinacakMiktar, StoktanKullanilacakMiktar, SatisFiyati,
                 DovizId, OrmeBirimMaliyet, OrmeBirimMaliyetDovizId, TeslimTarihi, RevizeTeslimTarihi,
                 VadeGun, OdemeSekliId, MakinaUstuGramaj, DinlenmisGramaj, IgneSayisi, MaySayisi, Pus, Feine, TupAcikEnMayli, ArtikelNo,
                 Beyazlik, TahminiFireOrani, IplikBoyu1, IplikBoyu2, IplikBoyu3, IplikBoyu4, RaporBoyu, SevkYeri, MusteriTalimati, Notu, KaliteId, OrmeSiparisDetayId, HamStokBirimId, :AKullaniciId, :AKullaniciId,
                 PaketlemeSekliId from OrmeSiparisDetay
             where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId;
  end

  For Select BoyaSiparisDetayId, BirimFiyat, FasonBirimFiyati, BaskiBirimFiyati from BoyaSiparisDetay
     where BoyaSiparisDetay.OrmeSiparisDetayID = :AOrmeSiparisDetayID
     order by BoyaSiparisDetay.BoyaSiparisDetayID
     into :ABoyaSiparisDetayId, :BirimFiyat, :FFasonBirimFiyati, :FBaskiBirimFiyati do
  begin
    if (SiparisKopyalarkenFiyatiTasi = 0) then
    begin
      BirimFiyat = 0;
      FFasonBirimFiyati = 0;
      FBaskiBirimFiyati = 0;
    end

    BoyaSiparisDetayId = gen_id(BoyaSiparisDetay_Id_Gen, 1);

    Insert into BoyaSiparisDetay(BoyaSiparisDetayId, OrmeSiparisDetayId, BoyaSiparisRenkNo, BoyaSiparisKayitId, FirmaSevkAdresiId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
                                 MelanjKodu, FlamKodu, RenkId, RenkAdi, ProsesId, SiparisProsesId, SiparisKilo, SiparisMetre, PaketlemeSekliId, RenkDerinligiId, BirimFiyat, HamStokBirimId,
                                 FaturaEdilecekMiktarTipi, DovizId, UcretliTamir, FaturaKesilecek, VadeGun, OdemeSekliId, HamId, Pus, Feine, Lot, MusteriRenkKoduveAdi,
                                 DesenId, DesenKenarYazisi, DesenIlkBaski,
                                 KaliteId, MusteriKaliteNo, MusteriKaliteAdi, Denye, IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen,
                                 CekmezlikBoyIstenen, YikamaHasligiIstenen, SansIstenen, ToleransEn, ToleransGramaj, ToleransEndenCekmezlik, ToleransBoydanCekmezlik,
                                 ToleransDonme, ToleransSans, TerminTarihi, RevizeTerminTarihi, TerminNotu, Notu, PlanlamaNotu,
                                 YikamaDereceId, KurutmaSekliId, YikamaHasligiTestiId, OngoruYapilacak, FirmaAdimizEtiketteCiksinmi, IslemGrubuId, MaximumFireOrani, MinimumFireOrani, FasonBirimFiyati, FasonBirimFiyatDovizId, BaskiBirimFiyati, BaskiBirimFiyatDovizId, InsertKullaniciId, KullaniciId, KopyaSonuUpdateYok, TupAcikEnMayli, PSKKEtiketTipiId)
      Select :BoyaSiparisDetayId, :OrmeSiparisDetayId, BoyaSiparisRenkNo, BoyaSiparisKayitId, FirmaSevkAdresiId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
             MelanjKodu, FlamKodu, RenkId, RenkAdi, ProsesId, SiparisProsesId, SiparisKilo, SiparisMetre, PaketlemeSekliId, RenkDerinligiId, :BirimFiyat, HamStokBirimId,
             FaturaEdilecekMiktarTipi, DovizId, UcretliTamir, FaturaKesilecek, VadeGun, OdemeSekliId, HamId, Pus, Feine, Lot, MusteriRenkKoduveAdi,
             DesenId, DesenKenarYazisi, DesenIlkBaski,
             KaliteId, MusteriKaliteNo, MusteriKaliteAdi, Denye, IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen,
             CekmezlikBoyIstenen, YikamaHasligiIstenen, SansIstenen, ToleransEn, ToleransGramaj, ToleransEndenCekmezlik, ToleransBoydanCekmezlik,
             ToleransDonme, ToleransSans, TerminTarihi, RevizeTerminTarihi, TerminNotu, Notu, PlanlamaNotu,
             YikamaDereceId, KurutmaSekliId, YikamaHasligiTestiId, OngoruYapilacak, FirmaAdimizEtiketteCiksinmi, IslemGrubuId, MaximumFireOrani, MinimumFireOrani, :FFasonBirimFiyati, FasonBirimFiyatDovizId, :FBaskiBirimFiyati, BaskiBirimFiyatDovizId, :AKullaniciId, :AKullaniciId, 1, TupAcikEnMayli, PSKKEtiketTipiId from BoyaSiparisDetay
      where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

    Insert into BoyaSiparisBaski(BoyaSiparisDetayId, BoyaSiparisKayitId, DesenVaryantId, VaryantIlkBaski, OzelYerlesim, OzelYerlesimDosyaAdi,
                                 YanRaportVarmi, KenarBeyazligi, Konfeksiyon, Tasma, Kilo, Metre, InsertKullaniciId, KullaniciId)
      Select :BoyaSiparisDetayId, BoyaSiparisKayitId, DesenVaryantId, VaryantIlkBaski, OzelYerlesim, OzelYerlesimDosyaAdi,
              YanRaportVarmi, KenarBeyazligi, Konfeksiyon, Tasma, Kilo, Metre, :AKullaniciId, :AKullaniciId From BoyaSiparisBaski
         where BoyaSiparisBaski.BoyaSiparisDetayID = :ABoyaSiparisDetayId;


    Insert into SiparisIlaveIslemler(BoyaSiparisDetayId, BoyaSiparisKayitId, BirimFiyat, DovizId, IlaveIslemlerId, InsertKullaniciId, KullaniciId)
      Select :BoyaSiparisDetayId, BoyaSiparisKayitId, BirimFiyat, DovizId, IlaveIslemlerId, :AKullaniciId, :AKullaniciId from SiparisIlaveIslemler
      where SiparisIlaveIslemler.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

    Insert into SiparisOzelTestler(BoyaSiparisDetayId, BoyaSiparisKayitId, OzelTestId, InsertKullaniciId, KullaniciId)
      Select :BoyaSiparisDetayId, BoyaSiparisKayitId, OzelTestId, :AKullaniciId, :AKullaniciId from SiparisOzelTestler
      where SiparisOzelTestler.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

    Insert into SiparisOzelApreler(BoyaSiparisDetayId, BoyaSiparisKayitId, OzelApreId, InsertKullaniciId, KullaniciId)
      Select :BoyaSiparisDetayId, BoyaSiparisKayitId, OzelApreId, :AKullaniciId, :AKullaniciId from SiparisOzelApreler
      where SiparisOzelApreler.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

    update BoyaSiparisDetay set KopyaSonuUpdateYok = 0 where BoyaSiparisDetayId = :BoyaSiparisDetayId;
  end
end
