CREATE PROCEDURE NAKITAKISRAPORUCEK(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ARAPORTARIHI TIMESTAMP,
    ARAPORDOVIZID VARCHAR(5))
RETURNS (
    ALINANCEKLERTOPLAMI DOUBLE PRECISION,
    VERILENCEKLERTOPLAMI DOUBLE PRECISION)
AS
declare Variable FAv        varchar(1);
declare Variable FDovizId   varchar(5);
declare Variable FMeblag    Double Precision;
declare Variable FDovizKuru Double Precision;
begin
  AlinanCeklerToplami  = 0;
  VerilenCeklerToplami = 0;
  for Select CekSenetDetay.Av, CekSenetDetay.DovizId, Sum(CekSenetDetay.Meblag - CekSenetDetay.TahsilEdilenMeblag)  from CekSenetDetay
    left join CekSenet on CekSenet.CsBordroKayitId = CekSenetDetay.CsBordroKayitId
    left join Firma Firma1 on Firma1.FirmaId = CekSenet.FirmaId
    left join BankaHesap on BankaHesap.BankaHesapId = CekSenetDetay.BankaHesapId
    left join banka on (banka.bankaId = CekSenetDetay.AitOlduguBankaId or banka.bankaId = BankaHesap.BankaId)
    where (CekSenetDetay.VadeTarihi >= :ABaslangicTarihi and (CekSenetDetay.VadeTarihi <= :ABitistarihi or :ABitisTarihi is null) and CekSenetDetay.Cs = 'C') and
          (CekSenetDetay.Karsiliksiz = 0) and
          (CekSenetDetay.SonBordroTipi = 0 or CekSenetDetay.SonBordroTipi = 2 or CekSenetDetay.SonBordroTipi = 3 or CekSenetDetay.SonBordroTipi = 4 or CekSenetDetay.SonBordroTipi = 6) and
          (CekSenetDetay.Meblag > CekSenetDetay.TahsilEdilenMeblag)
    group by CekSenetDetay.Av, CekSenetDetay.DovizId
    into :FAv, :FDovizId, :FMeblag do
  begin
    if (FDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
    else FDovizKuru = 1;
    FMeblag = FMeblag * FDovizKuru;
    if (FAv = 'A') then AlinanCeklerToplami = AlinanCeklerToplami + FMeblag;
    else VerilenCeklerToplami = VerilenCeklerToplami + FMeblag;
  end
  suspend;
end
