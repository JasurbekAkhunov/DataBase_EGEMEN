CREATE PROCEDURE FIRMABAZINDABOYAHANEANALIZI_3(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    KSEVKIYATNORMALKILOTOPLAM DOUBLE PRECISION,
    KSEVKIYATTAMIRKILOTOPLAM DOUBLE PRECISION,
    KSEVKIYATNUMUNEKILOTOPLAM DOUBLE PRECISION,
    KSEVKIYATKONTROLKILOTOPLAM DOUBLE PRECISION,
    KSEVKIYATRPTKILOTOPLAM DOUBLE PRECISION,
    KSEVKIYATNORMALMETRETOPLAM DOUBLE PRECISION,
    KSEVKIYATTAMIRMETRETOPLAM DOUBLE PRECISION,
    KSEVKIYATNUMUNEMETRETOPLAM DOUBLE PRECISION,
    KSEVKIYATKONTROLMETRETOPLAM DOUBLE PRECISION,
    KSEVKIYATRPTMETRETOPLAM DOUBLE PRECISION,
    ISEVKIYATNORMALKILOTOPLAM DOUBLE PRECISION,
    ISEVKIYATTAMIRKILOTOPLAM DOUBLE PRECISION,
    ISEVKIYATNUMUNEKILOTOPLAM DOUBLE PRECISION,
    ISEVKIYATKONTROLKILOTOPLAM DOUBLE PRECISION,
    ISEVKIYATRPTKILOTOPLAM DOUBLE PRECISION,
    SEVKIYATHAMIADEKILOTOPLAM DOUBLE PRECISION,
    SEVKIYATISLEMSIZIADEKILOTOPLAM DOUBLE PRECISION)
AS
declare variable FlastFirmaId  varchar(20);
declare variable FFirmaId     varchar(20);
declare variable FFirmaAdi    varchar(100);
declare variable FIslemTipiId integer;
declare variable FPartiTipiId integer;
declare variable FIrsaliyeTarihi  TimeStamp;
declare variable FFasonHamIade    integer;
declare variable FTamir           integer;
declare variable FBrutKilo        Numeric(12,3);
declare variable FNetKilo         Numeric(12,3);
declare variable FBrutMetre        Numeric(12,3);
begin
  FlastFirmaId = '';

  For Select BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliyeDetay.IrsaliyeTarihi, Parti.IslemTipiId, Parti.PartiTipiId, BoyaIrsaliyeDetay.FasonHamIade, HamStokMaster.Tamir, Sum(BoyaIrsaliyeDetay.BrutKilo), Sum(BoyaIrsaliyeDetay.Kilo), Sum(BoyaIrsaliyeDetay.BrutMetre) from BoyaIrsaliye
    left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
    left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
    left join RafDetay on RafDetay.RafDetayId = BoyaIrsaliyeDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
          (Firma.AnaHesapFirmaID = :AFirmaId or :AFirmaID is null) and BoyaIrsaliye.HareketTipiID = 0
    group by BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliyeDetay.IrsaliyeTarihi, Parti.IslemTipiId, Parti.PartiTipiId, BoyaIrsaliyeDetay.FasonHamIade, HamStokMaster.Tamir
    into :FFirmaId, :FFirmaAdi, :FIrsaliyeTarihi, :FIslemTipiId, :FPartiTipiId, :FFasonHamIade, :FTamir, :FBrutKilo, :FNetKilo, :FBrutMetre do
  begin
    if (FLastFirmaID <> FFirmaID) then
    begin
      if (FLastFirmaID <> '') then suspend;
      FirmaID = FFirmaId;
      FLastFirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;

      KSevkiyatNormalKiloToplam    = 0;
      KSevkiyatRptKiloToplam       = 0;
      KSevkiyatKontrolKiloToplam   = 0;
      KSevkiyatTamirKiloToplam     = 0;
      KSevkiyatNumuneKiloToplam    = 0;

      KSevkiyatNormalMetreToplam    = 0;
      KSevkiyatRptMetreToplam       = 0;
      KSevkiyatKontrolMetreToplam   = 0;
      KSevkiyatTamirMetreToplam     = 0;
      KSevkiyatNumuneMetreToplam    = 0;

      ISevkiyatNormalKiloToplam    = 0;
      ISevkiyatRptKiloToplam       = 0;
      ISevkiyatKontrolKiloToplam   = 0;
      ISevkiyatTamirKiloToplam     = 0;
      ISevkiyatNumuneKiloToplam    = 0;

      SevkiyatHamIadeKiloToplam        = 0;
      SevkiyatIslemsizIadeKiloToplam   = 0;
    end

    if (FIslemTipiID is not null) then
    begin
      if (FPartiTipiId = 0) then
      begin
        if (FIslemTipiID = 0) then ISevkiyatNormalKiloToplam = ISevkiyatNormalKiloToplam + FBrutKilo;
        else if (FIslemTipiID = 1) then ISevkiyatRptKiloToplam = ISevkiyatRptKiloToplam + FBrutKilo;
        else if (FIslemTipiID = 2) then ISevkiyatKontrolKiloToplam = ISevkiyatKontrolKiloToplam + FBrutKilo;
        else if (FIslemTipiID = 3) then ISevkiyatTamirKiloToplam = ISevkiyatTamirKiloToplam + FBrutKilo;
        else if (FIslemTipiID = 4) then ISevkiyatNumuneKiloToplam = ISevkiyatNumuneKiloToplam + FBrutKilo;
      end
      else
      begin
        if (FIslemTipiID = 0) then KSevkiyatNormalKiloToplam = KSevkiyatNormalKiloToplam + FBrutKilo;
        else if (FIslemTipiID = 1) then KSevkiyatRptKiloToplam = KSevkiyatRptKiloToplam + FBrutKilo;
        else if (FIslemTipiID = 2) then KSevkiyatKontrolKiloToplam = KSevkiyatKontrolKiloToplam + FBrutKilo;
        else if (FIslemTipiID = 3) then KSevkiyatTamirKiloToplam = KSevkiyatTamirKiloToplam + FBrutKilo;
        else if (FIslemTipiID = 4) then KSevkiyatNumuneKiloToplam = KSevkiyatNumuneKiloToplam + FBrutKilo;

        if (FIslemTipiID = 0) then KSevkiyatNormalMetreToplam = KSevkiyatNormalMetreToplam + FBrutMetre;
        else if (FIslemTipiID = 1) then KSevkiyatRptMetreToplam = KSevkiyatRptMetreToplam + FBrutMetre;
        else if (FIslemTipiID = 2) then KSevkiyatKontrolMetreToplam = KSevkiyatKontrolMetreToplam + FBrutMetre;
        else if (FIslemTipiID = 3) then KSevkiyatTamirMetreToplam = KSevkiyatTamirMetreToplam + FBrutMetre;
        else if (FIslemTipiID = 4) then KSevkiyatNumuneMetreToplam = KSevkiyatNumuneMetreToplam + FBrutMetre;
      end
    end
    else
    begin
      if (FFasonHamIade = 1) then
      begin
        if (FTamir = 1) then SevkiyatIslemsizIadeKiloToplam = SevkiyatIslemsizIadeKiloToplam + FNetKilo;
        else SevkiyatHamIadeKiloToplam = SevkiyatHamIadeKiloToplam + FNetKilo;
      end
    end
  end

  if (FLastFirmaID <> '') then suspend;
end
