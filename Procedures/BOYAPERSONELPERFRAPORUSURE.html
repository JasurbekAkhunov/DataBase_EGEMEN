CREATE PROCEDURE BOYAPERSONELPERFRAPORUSURE(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATEORIKIMALATSURESI DOUBLE PRECISION,
    ATEORIKIMALATSURESI2 DOUBLE PRECISION,
    AMAKINAID VARCHAR(10))
RETURNS (
    ISBASLAMATARIHI TIMESTAMP,
    ISBITISTARIHI TIMESTAMP,
    ISGECENSURE DOUBLE PRECISION,
    ISGECENSUREDK DOUBLE PRECISION,
    ISGECENSURESTR VARCHAR(20),
    ISSTANDARTSURE DOUBLE PRECISION,
    ISSTANDARTSUREDK DOUBLE PRECISION,
    ISSTANDARTSURESTR VARCHAR(20),
    ISSTANDARTSURE2 DOUBLE PRECISION,
    ISSTANDARTSURE2DK DOUBLE PRECISION,
    ISSTANDARTSURE2STR VARCHAR(20),
    ISCALISMASURE DOUBLE PRECISION,
    ISCALISMASUREDK DOUBLE PRECISION,
    ISCALISMASURESTR VARCHAR(20),
    ISARAVERMESURE DOUBLE PRECISION,
    ISARAVERMESUREDK DOUBLE PRECISION,
    ISARAVERMESURESTR VARCHAR(20))
AS
declare variable FSonTarih TimeStamp;
declare variable FYeniTarih TimeStamp;
declare variable FSonMakinaDurumu Integer;
declare variable FYeniMakinaDurumu Integer;
begin
  IsBaslamaTarihi = ABaslangicTarihi;
  IsBitisTarihi = ABitisTarihi;

  IsGecenSure = TimeStampToDouble(IsBitisTarihi) - TimeStampToDouble(IsBaslamaTarihi);
  IsGecenSureDk = IsGecenSure * 1440;
  IsGecenSureStr = TarihtoStr(IsGecenSure);

  IsStandartSure = ATeorikImalatSuresi / 1440;
  IsStandartSureDk = IsStandartSure * 1440;
  IsStandartSureStr = TarihtoStr(IsStandartSure);

  IsStandartSure2    = ATeorikImalatSuresi2 / 1440;
  IsStandartSure2Dk  = IsStandartSure2 * 1440;
  IsStandartSure2Str = TarihtoStr(IsStandartSure2);

  IsCalismaSure = 0;
  FSonMakinaDurumu = -1;
  FSonTarih = IsBaslamaTarihi;
  for select Tarih, Calisiyor from MakinaDurumu
    where MakinaId = :AMakinaId and
    Tarih >= :IsBaslamaTarihi and Tarih <= :IsBitisTarihi
    order by Tarih
    into :FYeniTarih, :FYeniMakinaDurumu do
  begin
    if (FSonMakinaDurumu = 1) then
      IsCalismaSure = TimeStampToDouble(FYeniTarih) - TimeStampToDouble(FSonTarih);
    FSonMakinaDurumu = FYeniMakinaDurumu;
    FSonTarih = FYeniTarih;
  end
  if (FSonMakinaDurumu = 1) then
    IsAraVermeSure = TimeStampToDouble(IsBitisTarihi) - TimeStampToDouble(FSonTarih);
  IsCalismaSureDk = IsCalismaSure * 1440;
  IsCalismaSureStr = TarihtoStr(IsCalismaSure);

  IsAraVermeSure = 0;
  FSonMakinaDurumu = -1;
  FSonTarih = IsBaslamaTarihi;
  for select Tarih, ImalataAraVerildi from MakinaDurumu
    where MakinaId = :AMakinaId and
    Tarih >= :IsBaslamaTarihi and Tarih <= :IsBitisTarihi and
    ImalataAraVerildi > 0
    order by Tarih
    into :FYeniTarih, :FYeniMakinaDurumu do
  begin
    if (FSonMakinaDurumu = 1 and FYeniMakinaDurumu = 2) then
      IsAraVermeSure = TimeStampToDouble(FYeniTarih) - TimeStampToDouble(FSonTarih);
    FSonMakinaDurumu = FYeniMakinaDurumu;
    FSonTarih = FYeniTarih;
  end
  if (FSonMakinaDurumu = 1) then
    IsAraVermeSure = TimeStampToDouble(IsBitisTarihi) - TimeStampToDouble(FSonTarih);
  IsAraVermeSureDk = IsAraVermeSure * 1440;
  IsAraVermeSureStr = TarihtoStr(IsAraVermeSure);

  Suspend;
end
