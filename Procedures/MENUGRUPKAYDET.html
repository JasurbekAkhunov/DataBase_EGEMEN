CREATE PROCEDURE MENUGRUPKAYDET(
    AMENUGRUPID INTEGER,
    ATAG INTEGER)
AS
declare variable FSiraId        varchar(30);
declare variable FTag           integer;
declare variable KayitSayisi    integer;
begin
  if (ATag > -1) then exit;
  Select SiraId from Menu where Tag = :ATag into :FSiraId;
  FSiraId = FSiraId || '%';

  For Select Tag from Menu where Menu.SiraId like :FSiraId into :FTag do
  begin
    Select Count(*) from MenuGrupElemanlari where MenuGrupElemanlari.MenuGrupId = :AMenuGrupId and MenuGrupElemanlari.Tag = :FTag into :KayitSayisi;
    if (KayitSayisi = 0) then insert into MenuGrupElemanlari(MenuGrupId, Tag) values(:AMenuGrupId, :FTag);
  end
end
