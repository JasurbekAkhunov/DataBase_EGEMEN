CREATE PROCEDURE BEKLEYENMALLARDURUMU
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TEMSILCIID INTEGER,
    TEMSILCIADI VARCHAR(40),
    RPT NUMERIC(12,3),
    KONTROLBOYA NUMERIC(12,3),
    RENKOKEYIBEKLEYEN NUMERIC(12,3),
    TALIMATIOLMAYAN NUMERIC(12,3),
    TALIMATOKEYIBEKLEYEN NUMERIC(12,3),
    TOPLAMKILO NUMERIC(12,3))
AS
declare variable FFirmaId       varchar(20);
declare variable FFirmaAdi      varchar(100);
declare variable FTemsilciId    integer;
declare variable FTemsilciAdi   varchar(40);
declare variable Kilo           Numeric(12,3);
declare variable ReceteOkey     integer;
declare variable Talimat        integer;
declare variable TalimatOkey    integer;
declare variable IslemTipi      integer;
declare variable SonFirmaId     varchar(20);
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
begin
  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  SonFirmaId = '';
  for select Parti.FirmaId, Firma.Adi, Firma.TemsilciId, Kullanici.Adi, Parti.ReceteOkey, Parti.Talimat, Parti.TalimatOkey, Parti.IslemTipiId, sum(PartiStok.Kilo) from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    Left join Kullanici on Kullanici.KullaniciId = Firma.TemsilciId
    where PartiStok.Durumu <= 3
    group by Parti.FirmaId, Firma.Adi, Firma.TemsilciId, Kullanici.Adi, Parti.IslemTipiId, Parti.ReceteOkey, Parti.Talimat, Parti.TalimatOkey
    into :FFirmaId, :FFirmaAdi, :FTemsilciId, :FTemsilciAdi, :ReceteOkey, :Talimat, :TalimatOkey, :IslemTipi, Kilo do
  begin
    if (SonFirmaId <> FFirmaId) then
    begin
      if (SonFirmaId <> '') then
      begin
        ToplamKilo = Rpt + KontrolBoya + RenkOkeyiBekleyen + TalimatiOlmayan + TalimatOkeyiBekleyen;
        suspend;
      end
      SonFirmaId = FFirmaId;

      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      TemsilciId = FTemsilciId;
      TemsilciAdi = FTemsilciAdi;

      Rpt = 0;
      KontrolBoya = 0;
      RenkOkeyiBekleyen = 0;
      TalimatiOlmayan = 0;
      TalimatOkeyiBekleyen = 0;
    end
    if (IslemTipi = itRpt) then Rpt = Rpt + Kilo;     /* Rpt */
    else if (Talimat = 0) then TalimatiOlmayan = TalimatiOlmayan + Kilo;
    else if (TalimatOkey = 0) then TalimatOkeyiBekleyen = TalimatOkeyiBekleyen + Kilo;
    else if (ReceteOkey = 0) then RenkOkeyiBekleyen = RenkOkeyiBekleyen + Kilo;
    else if (IslemTipi = itKontrol) then KontrolBoya = KontrolBoya + Kilo;     /* Kontrol  */
  end
  ToplamKilo = Rpt + KontrolBoya + RenkOkeyiBekleyen + TalimatiOlmayan + TalimatOkeyiBekleyen;
  if (SonFirmaId <> '') then suspend;
end
