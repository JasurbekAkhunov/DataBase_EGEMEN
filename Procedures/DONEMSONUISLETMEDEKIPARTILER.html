    UNITELERASEVKMALIYETI DOUBLE PRECISION)
AS
declare variable FFirmaid   varchar(20);
declare variable FFirmaAdi  varchar(100);
declare variable FHamId     integer;
declare variable FHamAdi    varchar(255);
declare variable FLastFirmaid   varchar(20);
declare variable FLastHamId     integer;
declare variable FProsesId     integer;
declare variable FProsesAdi    varchar(255);
declare variable FLastProsesId     integer;
declare variable FAsamaAdi     Varchar(50);
declare variable FRAsamaAdi    Varchar(50);
declare variable FAsamaSon     Varchar(50);
declare variable FRAsamaSon    Varchar(50);
declare variable FAsamaGiris    TimeStamp;
declare variable FAsamaCikis    TimeStamp;
begin
  FLastFirmaid = '';
  FLastHamId   = -1;
  BrutKilo = 0;
  FLastProsesId = -1;

  /*GECICI*/
  ATarih = ATarih - 1;

  For Select Parti.PartiYil, Parti.PartiId, Parti.FirmaId, Firma.Adi, PartiStok.HamId, Ham.Adi, Parti.IslemTipiId, IslemTipi.Adi,
             PartiStok.PartiSinifId, Parti.ReceteId, Recete.Adi, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi,
             BoyaSiparisDetay.IlaveIslemler, PartiStok.ProsesId, Proses.Adi, PartiStok.BirimFiyat, PartiStok.DovizId,
             BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, PartiStok.Kilo, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,

             BoyaSiparisDetay.DesenId, Desen.Adi, Parti.DesenVaryantId, DesenVaryant.VaryantAdi, PartiStok.HamTipiId, HamTipi.Adi,
             Ham.HamSatisTransferKodu, Ham.FasonTransferKodu, Ham.FasonOrmeTransferKodu, Ham.FasonArtiSatOrmeTransferKodu, Ham.AlisTransferKodu, Ham.RenkliAlisTransferKodu, Ham.BoyaliTransferKodu,
             Ham.SekerKasarTransferKodu, Ham.OptikBeyazTransferKodu, Ham.YikamaTransferKodu, Ham.SABoyaliTransferKodu, Ham.SASekerKasarTransferKodu, Ham.SAOptikBeyazTransferKodu, Ham.SAYikamaTransferKodu,
             Ham.AlisBoyaliTransferKodu, Ham.AlisSekerKasarTransferKodu, Ham.AlisOptikBeyazTransferKodu, Ham.AlisYikamaTransferKodu, Ham.AlisBaskiliTransferKodu,
             IslemGrubu.FasonIplikTransferKodu, IslemGrubu.FasonIplikMtTransferKodu, IslemGrubu.FasonKumasTransferKodu, IslemGrubu.FasonKumasMtTransferKodu, IslemGrubu.SatisIplikTransferKodu, IslemGrubu.SatisIplikMtTransferKodu,
             IslemGrubu.SatisKumasTransferKodu, IslemGrubu.SatisKumasMtTransferKodu, IslemGrubu.BaskiFasonKumasTransferKodu, IslemGrubu.BaskiFasonKumasMtTransferKodu, IslemGrubu.BaskiSatisKumasTransferKodu,
             IslemGrubu.BaskiSatisKumasMtTransferKodu, BoyaSiparisDetay.IslemGrubuID, SipIslemGrubu.Adi, Parti.IslemGrubuID, IslemGrubu.Adi, OrmeSiparisDetay.OrmeBirimMaliyet, OrmeSiparisDetay.OrmeBirimMaliyetDovizId,
             BoyaSiparisDetay.FasonBirimFiyati, BoyaSiparisDetay.FasonBirimFiyatDovizId, BoyaSiparisDetay.MusteriSiparisNo from Parti
  left join Firma on Firma.FirmaId = Parti.FirmaId
  left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
  left join Ham on Ham.HamId = PartiStok.HamId
  left join Proses on Proses.ProsesId = PartiStok.ProsesId
  left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
  left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  left join IslemGrubu SipIslemGrubu on SipIslemGrubu.IslemGrubuId = BoyaSiparisDetay.IslemGrubuId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
  left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  left join PartiStokIptal on PartiStokIptal.PartiSinifId = PartiStok.PartiSinifId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
  left join HamTipi on HamTipi.HamTipiid = PartiStok.HamTipiId

  (PartiStok.PartiStokIptalKayitId is null or PartiStokIptal.IptalTarihi >= :ATarih)
  /*and PartiStok.PartiStokIptalKayitId is null and
  (PartiStok.SevkTarihi > :ATarih or PartiStok.SevkTarihi is Null)*/
  order by Parti.FirmaId, Firma.Adi, PartiStok.HamId, PartiStok.ProsesId
  into :PartiYil, :PartiId, :FirmaId, :FirmaAdi, :HamId, :HamAdi, :IslemTipiId, :IslemTipiAdi, :PartiSinifId, :ReceteId, :ReceteAdi, :SiparisRenkAdi,

       :SiparisEnIstenen, :SiparisGramajIstenen, :DesenId, :DesenAdi, :DesenVaryantId, :VaryantAdi, :HamTipiId, :HamTipiAdi,
       :HamSatisTransferKodu, :FasonTransferKodu, :FasonOrmeTransferKodu, :FasonArtiSatOrmeTransferKodu, :AlisTransferKodu, :RenkliAlisTransferKodu, :BoyaliTransferKodu, :SekerKasarTransferKodu,
       :OptikBeyazTransferKodu, :YikamaTransferKodu, :SABoyaliTransferKodu, :SASekerKasarTransferKodu, :SAOptikBeyazTransferKodu, :SAYikamaTransferKodu, :AlisBoyaliTransferKodu, :AlisSekerKasarTransferKodu,
       :AlisOptikBeyazTransferKodu, :AlisYikamaTransferKodu, :AlisBaskiliTransferKodu, :FasonIplikTransferKodu, :FasonIplikMtTransferKodu, :FasonKumasTransferKodu, :FasonKumasMtTransferKodu, :SatisIplikTransferKodu,
       :SatisIplikMtTransferKodu, :SatisKumasTransferKodu, :SatisKumasMtTransferKodu, :BaskiFasonKumasTransferKodu, :BaskiFasonKumasMtTransferKodu, :BaskiSatisKumasTransferKodu, :BaskiSatisKumasMtTransferKodu,
       SipIslemGrubuID, SipIslemGrubuAdi, PartiIslemGrubuID, PartiIslemGrubuAdi, :OrmeBirimMaliyet, :OrmeBirimMaliyetDovizId, :FasonBirimFiyati, :FasonBirimFiyatDovizId, :MusteriSiparisNo do
  begin
    SevkBrutKilo = 0;
    if (PartiBrutKilo is null) then PartiBrutKilo = 0;

    if (YurtDisi = 1) then YurtDisiStr = 'Yurt Dışı';
    else YurtDisiStr = '';

    BoyaSiparisNoStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    if (PartiSinifId is not null) then
    begin
      Select Sum(BoyaIrsaliyeDetay.BrutKilo) from BoyaIrsaliyeDetay
        where BoyaIrsaliyeDetay.PartiSinifId = :PartiSinifId and BoyaIrsaliyeDetay.IrsaliyeTarihi < :ATarih
        into :SevkBrutKilo;

      Select Sum(HamStokSarfFisiDetay.BrutKilo) from HamStokSarfFisiDetay
        left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiId = HamStokSarfFisiDetay.HamStokSarfFisiId
        where HamStokSarfFisiDetay.PartiSinifId = :PartiSinifId and HamStokSarfFisi.FisTarihi < :ATarih
        into :SarfBrutKilo;

      Select Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.Dara), Sum(PartiStokTarti.Metre), Sum(PartiStokTarti.Adet) from PartiStokTarti
        where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.InsertTarihi < :ATarih
        into :TartiKilo, :TartiDara, :TartiMetre, :TartiAdet;

      Select GirisLot, GirisDenye, GirisPus, GirisFeine, GirisOrmeciMakinaNo, EskiPartiNo,
        YikamaOncesiEn, YikamaOncesiGramaj, YikamaSonrasiEn, YikamaSonrasiGramaj, GirisFirmaId from PartiStokGirisBilgisiGetir(:PartiSinifID)
        into :GirisLot, :GirisDenye, :GirisPus, :GirisFeine, :GirisOrmeciMakinaNo, :EskiPartiNo,
        :GirisYikamaOncesiEn, :GirisYikamaOncesiGramaj, :GirisYikamaSonrasiEn, :GirisYikamaSonrasiGramaj, :GirisFirmaId;


      Select first 1 HamStokMaliyeti, AsamaUMUzerindenAsamaMaliyeti, AsamaUMUzerindenIscilikAsamaM, MakinaUMUzerindenAsamaMaliyeti, BoyaKimyasalMaliyeti,
         TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti from PartiSinifMaliyetKumulatif
         where PartiSinifID = :PartiSinifID
         into :HamStokMaliyeti, :AsamaUMUzerindenAsamaMaliyeti, :AsamaUMUzerindenIscilikAsamaM, :MakinaUMUzerindenAsamaMaliyeti, :BoyaKimyasalMaliyeti,
         :TeorikBoyaKimyasalMaliyeti, :UnitelerASevkMaliyeti;

    end

    if (SevkBrutKilo is null) then SevkBrutKilo = 0;
    if (SarfBrutKilo is null) then SarfBrutKilo = 0;

    BrutKilo = PartiBrutKilo - (SevkBrutKilo + SarfBrutKilo);

    if (AFasonBoyaSiparisFiyatGorebilir = 0) then
    begin
      BirimFiyat = null;
      DovizId    = null;
    end

    OTarihtekiDurumu = '';
    FAsamaAdi = Null;
    FRAsamaAdi = Null;
    FAsamaSon = Null;
    FRAsamaSon = Null;
    FAsamaGiris = Null;
    FAsamaCikis = Null;
    if (BrutKilo <= 0) then
    begin
      OTarihtekiDurumu = 'Tamamı sevk edilmiş...';
    end
    else if (SevkBrutKilo > 0) then
    begin
      OTarihtekiDurumu = 'Bir kısmı sevk edilmiş...';
    end
    else
    begin
      select first(1) Asama.Adi, RAsama.Adi, Asama.SonAsamami, RAsama.SonAsamami,
        PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
        left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join Asama RAsama on RAsama.AsamaId = Recete.UstAsamaId
      where ((PartiAsamaStok.CikisTarihi is not Null and PartiAsamaStok.CikisTarihi < :ATarih) or
      (PartiAsamaStok.CikisTarihi is Null and PartiAsamaStok.GirisTarihi is not Null and PartiAsamaStok.GirisTarihi < :ATarih)) and
      (PartiAsamaStok.PartiSinifId = :PartiSinifId) and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamaStok.GirisTarihi is Not Null
      order by PartiAsamaStok.GirisTarihi desc, PartiAsamaStok.CikisTarihi desc
      into :FAsamaAdi, :FRAsamaAdi, :FAsamaSon, :FRAsamaSon, :FAsamaGiris, :FAsamaCikis;

      if ((FAsamaAdi is Null) and (FRAsamaAdi is not Null)) then
      begin
        FAsamaAdi = FRAsamaAdi;
        FAsamaSon = FRAsamaSon;
      end

      if (FAsamaSon is Null) then FAsamaSon = 0;

      if (FAsamaAdi is Null) then
        OTarihtekiDurumu = 'Herhangi bir aşama başlamamış...';
      else if ((FAsamaSon = 1) and (FAsamaCikis is not Null)) then
        OTarihtekiDurumu = 'Partinin son aşaması yapılmış...';
      else if (FAsamaCikis is Null) then
        OTarihtekiDurumu = 'Parti ' || FAsamaAdi || ' aşamasında...';
      else if (FAsamaCikis is not Null) then
        OTarihtekiDurumu = 'Parti ' || FAsamaAdi || ' aşamasından çıkmış...';

    end


    if (BrutKilo > 0) then
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      Select IlaveIslemlerAdi from PartiStokYapilacaklar(:PartiSinifId) into :IlaveIslemlerAdi;
      suspend;
    end
  end
