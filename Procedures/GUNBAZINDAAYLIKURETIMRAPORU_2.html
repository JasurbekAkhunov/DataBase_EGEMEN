CREATE PROCEDURE GUNBAZINDAAYLIKURETIMRAPORU_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMTIPIID INTEGER,
    APARTIGRUPID VARCHAR(4))
RETURNS (
    TARIH TIMESTAMP,
    BEYAZURETIMI NUMERIC(12,3),
    RENKLIURETIMI NUMERIC(12,3),
    YIKAMAURETIMI NUMERIC(12,3),
    SEKERKASARURETIMI NUMERIC(12,3),
    CIFTBOYAMAURETIMI NUMERIC(12,3),
    SIYAHURETIMI NUMERIC(12,3),
    NAYLONBOYAMAURETIMI NUMERIC(12,3),
    POLYESTERBOYAMAURETIMI NUMERIC(12,3),
    DIGERURETIM NUMERIC(12,3),
    NORMALURETIMTOPLAMI NUMERIC(12,3),
    ILAVEPROSESURETIMTOPLAMI NUMERIC(12,3),
    ICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    UCRETLIICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    UCRETSIZICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    DISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    UCRETLIDISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    UCRETSIZDISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    TOPLAMURETIM NUMERIC(12,3),
    TAMIRORANI DOUBLE PRECISION)
AS
declare variable FPartiTamirId    integer;
declare variable FIlaveProses     integer;
declare variable FKilo            Numeric(12,3);
declare variable FCikisTarihi     TimeStamp;
declare variable FLastCikisTarihi TimeStamp;
declare variable VardiyaBaslangicSaati  Double Precision;
declare variable FSaat                  Double Precision;
Declare variable FDate                  date;
declare variable itTamir          integer;
declare variable FIslemTipiId     integer;
declare variable FIslemGrubuId    integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable FFaturaKesilecek integer;
declare variable FUcretliTamir integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf,:hstTelef;
  Select itTamir from IslemTipiGetir into :itTamir;

  Select TimeStampToDouble(TimeStampToTime(VardiyaBaslangicSaati)) from sistem into :VardiyaBaslangicSaati;

  FLastCikisTarihi = Null;

  for select Parti.IslemGrubuId, parti.IslemTipiId, PartiAsamalari.PartiTamirId, PartiAsamaStokDetay.kilo,
             PartiAsamaStokDetay.Cikistarihi, PartiAsamalari.IlaveProses, Parti.FaturaKesilecek, TamirCozumSekli.Ucretli from PartiAsamaStokDetay
  left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
  left join Ham on Ham.HamId = PartiStok.HamId
  left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
  left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
  left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
  left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
  left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
  where PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi <= :ABitisTarihi and
        ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null) and
        Makina.Kazan = 1 and
       (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null)
  order by PartiAsamaStokDetay.CikisTarihi
  into :FIslemGrubuId, :FIslemTipiId, :FPartiTamirId, :FKilo, :FCikistarihi, :FIlaveProses, :FFaturaKesilecek, :FUcretliTamir do
  begin
    FDate = FCikisTarihi;
    FSaat = TimeStampToDouble(FCikisTarihi) - TimeStampToDouble(FDate);
    FCikisTarihi = FDate;
    if (FSaat < VardiyaBaslangicSaati) then FCikisTarihi = FCikisTarihi - 1;

    if (FLastCikisTarihi is null or FCikisTarihi <> FLastCikisTarihi) then
    begin
      if (FLastCikisTarihi is not null) then
      begin
        NormalUretimToplami = BeyazUretimi + RenkliUretimi + YikamaUretimi + SekerKasarUretimi + CiftBoyamaUretimi + SiyahUretimi + NaylonBoyamaUretimi + PolyesterBoyamaUretimi + DigerUretim;
        ToplamUretim = NormalUretimToplami + IcTamirUretimToplami + DisTamirUretimToplami;

        if (ToplamUretim <> 0) then TamirOrani = (IcTamirUretimToplami + DisTamirUretimToplami) * 100 / ToplamUretim;
        suspend;
      end

      BeyazUretimi       = 0;
      RenkliUretimi      = 0;
      YikamaUretimi      = 0;
      SekerKasarUretimi  = 0;
      CiftBoyamaUretimi  = 0;
      SiyahUretimi       = 0;
      NaylonBoyamaUretimi = 0;
      PolyesterBoyamaUretimi = 0;
      DigerUretim        = 0;
      NormalUretimToplami = 0;
      IlaveProsesUretimToplami = 0;

      IcTamirUretimToplami  = 0;
      UcretliIcTamirUretimToplami = 0;
      UcretsizIcTamirUretimToplami = 0;
      DisTamirUretimToplami = 0;
      UcretliDisTamirUretimToplami = 0;
      UcretsizDisTamirUretimToplami = 0;
      ToplamUretim = 0;
      TamirOrani          = 0;
      Tarih = FCikisTarihi;
      FLastCikisTarihi = FCikisTarihi;
    end

    if (FIlaveProses = 1) then IlaveProsesUretimToplami = IlaveProsesUretimToplami + FKilo;
    else if (FPartiTamirID is not null) then
    begin
      IcTamirUretimToplami = IcTamirUretimToplami + FKilo;
      if (FUcretliTamir = 1) then
        UcretliIcTamirUretimToplami = UcretliIcTamirUretimToplami + FKilo;
      else
        UcretsizIcTamirUretimToplami = UcretsizIcTamirUretimToplami + FKilo;
    end
    else if (FIslemTipiID = itTamir) then
    begin
      DisTamirUretimToplami = DisTamirUretimToplami + FKilo;

      if (FFaturaKesilecek = 1) then
        UcretliDisTamirUretimToplami = UcretliDisTamirUretimToplami + FKilo;
      else
        UcretsizDisTamirUretimToplami = UcretsizDisTamirUretimToplami + FKilo;
    end
    else if (FIslemGrubuId = 0) then BeyazUretimi = BeyazUretimi + FKilo;
    else if (FIslemGrubuId = 1) then RenkliUretimi = RenkliUretimi + FKilo;
    else if (FIslemGrubuId = 2) then YikamaUretimi = YikamaUretimi + FKilo;
    else if (FIslemGrubuId = 3) then SekerKasarUretimi = SekerKasarUretimi + FKilo;
    else if (FIslemGrubuId = 5) then CiftBoyamaUretimi = CiftBoyamaUretimi + FKilo;
    else if (FIslemGrubuId = 6) then SiyahUretimi = SiyahUretimi + FKilo;
    else if (FIslemGrubuId = 8) then NaylonBoyamaUretimi = NaylonBoyamaUretimi + FKilo;
    else if (FIslemGrubuId = 20) then PolyesterBoyamaUretimi = PolyesterBoyamaUretimi + FKilo;
    else DigerUretim = DigerUretim + FKilo;
  end

  if (FLastCikisTarihi is not null) then
  begin
    NormalUretimToplami = BeyazUretimi + RenkliUretimi + YikamaUretimi + SekerKasarUretimi + CiftBoyamaUretimi + SiyahUretimi + NaylonBoyamaUretimi + PolyesterBoyamaUretimi + DigerUretim;
    ToplamUretim = NormalUretimToplami + IcTamirUretimToplami + DisTamirUretimToplami + IlaveProsesUretimToplami;

    if (ToplamUretim <> 0) then TamirOrani = (IcTamirUretimToplami + DisTamirUretimToplami) * 100 / ToplamUretim;
    suspend;
  end
end
