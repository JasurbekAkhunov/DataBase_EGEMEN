CREATE PROCEDURE SIPARISDETAYBILGILERINIKOPYALA(
    ABOYASIPARISDETAYID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable BoyaSiparisKayitId       integer;
declare variable FEnIstenen               varchar(10);
declare variable FGramajIstenen           varchar(10);
declare variable FPaketlemeSekliId        Integer;
declare variable FMaximumFireOrani        Double Precision;
declare variable FMinimumFireOrani        Double Precision;
declare variable FTartiIcinFireOrani      Double Precision;
declare variable FBirimFiyat              Double Precision;
declare variable FIkinciKaliteBirimFiyat  Double Precision;
declare variable FFasonBirimFiyati        Double Precision;
declare variable FBaskiBirimFiyati        Double Precision;
declare variable FDovizId                 DovizSize;
declare variable FFasonBirimFiyatDovizId  DovizSize;
declare variable FBaskiBirimFiyatDovizId  DovizSize;
declare variable FProsesId                Integer;
declare variable FSiparisProsesId         Integer;
declare variable OrmeSiparisDetayId       Integer;
declare variable FIslemGrubuId            Integer;
declare variable FTerminTarihi            TimeStamp;
begin
  select BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.PaketlemeSekliId, BoyaSiparisDetay.MaximumFireOrani,
  BoyaSiparisDetay.MinimumFireOrani, BoyaSiparisDetay.TartiIcinFireOrani, BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.IkinciKaliteBirimFiyat,
  BoyaSiparisDetay.FasonBirimFiyati, BoyaSiparisDetay.BaskiBirimFiyati, BoyaSiparisDetay.DovizId, BoyaSiparisDetay.FasonBirimFiyatDovizId,
  BoyaSiparisDetay.BaskiBirimFiyatDovizId, BoyaSiparisDetay.ProsesId, BoyaSiparisDetay.BoyaSiparisKayitId, BoyaSiparisDetay.OrmeSiparisDetayId,
  BoyaSiparisDetay.IslemGrubuId, BoyaSiparisDetay.SiparisProsesId, BoyaSiparisDetay.TerminTarihi
  from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId
  into :FEnIstenen, :FGramajIstenen, :FPaketlemeSekliId, :FMaximumFireOrani, :FMinimumFireOrani, :FTartiIcinFireOrani, :FBirimFiyat, :FIkinciKaliteBirimFiyat,
  :FFasonBirimFiyati, :FBaskiBirimFiyati, :FDovizId, :FFasonBirimFiyatDovizId, :FBaskiBirimFiyatDovizId, :FProsesId, :BoyaSiparisKayitId, :OrmeSiparisDetayId,
  :FIslemGrubuId, :FSiparisProsesId, :FTerminTarihi;

  if (OrmeSiparisDetayId is null) then
  begin
    update BoyaSiparisDetay set
    BoyaSiparisDetay.EnIstenen = :FEnIstenen, BoyaSiparisDetay.GramajIstenen = :FGramajIstenen, BoyaSiparisDetay.PaketlemeSekliId = :FPaketlemeSekliId,
    BoyaSiparisDetay.MaximumFireOrani = :FMaximumFireOrani, BoyaSiparisDetay.MinimumFireOrani = :FMinimumFireOrani, BoyaSiparisDetay.TartiIcinFireOrani = :FTartiIcinFireOrani,
    BoyaSiparisDetay.BirimFiyat = :FBirimFiyat, BoyaSiparisDetay.IkinciKaliteBirimFiyat = :FIkinciKaliteBirimFiyat, BoyaSiparisDetay.FasonBirimFiyati = :FFasonBirimFiyati,
    BoyaSiparisDetay.BaskiBirimFiyati = :FBaskiBirimFiyati, BoyaSiparisDetay.DovizId = :FDovizId, BoyaSiparisDetay.FasonBirimFiyatDovizId = :FFasonBirimFiyatDovizId,
    BoyaSiparisDetay.BaskiBirimFiyatDovizId = :FBaskiBirimFiyatDovizId, BoyaSiparisDetay.ProsesId = :FProsesId, BoyaSiparisDetay.KullaniciId = :AKullaniciId,
    BoyaSiparisDetay.IslemGrubuId = :FIslemGrubuId, BoyaSiparisDetay.SiparisProsesId = :FSiparisProsesId, BoyaSiparisDetay.TerminTarihi = :FTerminTarihi
    where BoyaSiparisDetay.BoyaSiparisKayitId = :BoyaSiparisKayitId and BoyaSiparisDetay.BoyaSiparisDetayId <> :ABoyaSiparisDetayId;
  end
  else
  begin
    update BoyaSiparisDetay set
    BoyaSiparisDetay.EnIstenen = :FEnIstenen, BoyaSiparisDetay.GramajIstenen = :FGramajIstenen, BoyaSiparisDetay.PaketlemeSekliId = :FPaketlemeSekliId,
    BoyaSiparisDetay.MaximumFireOrani = :FMaximumFireOrani, BoyaSiparisDetay.MinimumFireOrani = :FMinimumFireOrani, BoyaSiparisDetay.TartiIcinFireOrani = :FTartiIcinFireOrani,
    BoyaSiparisDetay.BirimFiyat = :FBirimFiyat, BoyaSiparisDetay.IkinciKaliteBirimFiyat = :FIkinciKaliteBirimFiyat, BoyaSiparisDetay.FasonBirimFiyati = :FFasonBirimFiyati,
    BoyaSiparisDetay.BaskiBirimFiyati = :FBaskiBirimFiyati, BoyaSiparisDetay.DovizId = :FDovizId, BoyaSiparisDetay.FasonBirimFiyatDovizId = :FFasonBirimFiyatDovizId,
    BoyaSiparisDetay.BaskiBirimFiyatDovizId = :FBaskiBirimFiyatDovizId, BoyaSiparisDetay.ProsesId = :FProsesId, BoyaSiparisDetay.KullaniciId = :AKullaniciId,
    BoyaSiparisDetay.IslemGrubuId = :FIslemGrubuId, BoyaSiparisDetay.SiparisProsesId = :FSiparisProsesId, BoyaSiparisDetay.TerminTarihi = :FTerminTarihi
    where BoyaSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and BoyaSiparisDetay.BoyaSiparisDetayId <> :ABoyaSiparisDetayId;
  end
end
