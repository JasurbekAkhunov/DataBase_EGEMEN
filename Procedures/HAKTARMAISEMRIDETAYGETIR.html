CREATE PROCEDURE HAKTARMAISEMRIDETAYGETIR(
    AHAKTARMAISEMRIKAYITID INTEGER)
RETURNS (
    HAKTARMAISEMRIDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    LOT VARCHAR(150),
    KILO NUMERIC(12,3),
    BOBINGRAMAJI NUMERIC(12,3),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(15),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select HAktarmaIsEmriDetay.HAktarmaIsEmriDetayId, HamStokDetay.HamId, Ham.Adi, HAktarmaIsEmriDetay.BobinGramaji,
             HamStokMaster.Lot, HAktarmaIsEmriDetay.Kilo, HamStokIrsaliye.FirmaId, Firma.Adi,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             Ik.Adi, HAktarmaIsEmriDetay.InsertTarihi, Sk.Adi, HAktarmaIsEmriDetay.DegisimTarihi from HAktarmaIsEmriDetay
  left join HAktarmaFisiDetay on HAktarmaFisiDetay.HAktarmaFisiDetayid = HAktarmaIsEmriDetay.HAktarmaFisiDetayId
  left join HamStokDetay on HamStokDetay.HamStokDetayid = HAktarmaFisiDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterid = HamStokDetay.HamStokMasterId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
  left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
  left join Ham on Ham.HamId = HamStokDetay.HamId
  Left Join Kullanici IK on IK.KullaniciId = HAktarmaIsEmriDetay.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HAktarmaIsEmriDetay.KullaniciId
  where HAktarmaIsEmriDetay.HAktarmaIsEmriKayitId = :AHAktarmaIsEmriKayitId order by HAktarmaIsEmriDetay.HAktarmaIsEmriDetayId
  into :HAktarmaIsEmriDetayId, :HamId, :HamAdi, :BobinGramaji, :Lot, :Kilo, :FirmaId, :FirmaAdi, :IrsaliyeNo, :IrsaliyeSeriNo, :IrsaliyeTarihi, :EvrakTipi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';
    suspend;
  end
end
