CREATE PROCEDURE HGSIPARISTALIMATLISTESI(
    AHGSIPARISKAYITID INTEGER,
    AKESIMTALIMATI EGEBOOLEAN)
RETURNS (
    HGISEMRIKAYITID INTEGER,
    HGSIPARISKAYITID INTEGER,
    HGSIPARISDETAYID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ORMEMAKINAID MAKINAKODUSIZE,
    ORMEMAKINAADI ADI40,
    URETIMPARTINO VARCHAR(30),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    TERMINTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    URETIMISLEMLERI VARCHAR(100),
    PARCAISCILIK EGEBOOLEAN,
    PARCAISCILIKSTR VARCHAR(10),
    KESIMTALIMATI EGEBOOLEAN,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable Iplik        Double Precision;
declare variable Lycra        Double Precision;
declare variable FHamTipiId           integer;
declare variable FCikisMiktari          Double Precision;
declare variable YuvarlamaCarpani       Double Precision;
declare variable uitBoya integer;
declare variable FModelUretimIslemTipiID ModelGrupKoduSize;
begin
  select uitBoya from UretimIslemTipiGetir into :uitBoya;

  For select HGIsEmri.HGIsEmriKayitID, HGIsEmri.HGSiparisKayitId, HGIsEmri.HGSiparisDetayId,
    HGIsEmri.FirmaId, HGIsEmri.OrmeMakinaID, OrmeMakina.Adi, HGIsEmri.BirimFiyat, HGIsEmri.DovizId, HGIsEmri.VadeGun,
    HGIsEmri.Notu, Firma.Adi, Ik.Adi, HGIsEmri.InsertTarihi, Sk.Adi, HGIsEmri.DegisimTarihi,
    HGIsEmri.TerminTarihi, HGIsEmri.URetimPArtiNo, HGIsEmri.KesimTalimati, HGIsEmri.ParcaIscilik From HGIsEmri
    left join Firma on Firma.FirmaId = HGIsEmri.FirmaId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = HGIsEmri.OrmeMakinaId
    left join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayID = HGIsEmri.HGSiparisDetayId
    Left Join Kullanici IK on IK.KullaniciId = HGIsEmri.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = HGIsEmri.KullaniciId
    where HGIsEmri.HGSiparisKayitID = :AHGSiparisKayitID and (HGIsEmri.KesimTalimati = :AKesimTalimati or :AKesimTalimati is Null)
    into :HGIsEmriKayitID, :HGSiparisKayitId, :HGSiparisDetayId, :FirmaId, :OrmeMakinaID, :OrmeMakinaAdi, :BirimFiyat, :DovizId, :VadeGun,
         :Notu, :FirmaAdi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :TerminTarihi, :UretimPArtiNo,
         :KesimTalimati, :ParcaIscilik do
  begin
    if (ParcaIscilik = 1) then ParcaIscilikStr = 'Evet';
    else ParcaIscilikStr = 'Hayır';

    UretimIslemleri = '';
    Select UretimIslemleriID from HGIsEmriUretimIslemiGetir(:HGIsEmriKayitID, :KesimTalimati) into :UretimIslemleri;

    suspend;
  end
end
