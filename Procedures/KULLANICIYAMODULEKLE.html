CREATE PROCEDURE KULLANICIYAMODULEKLE(
    AKULLANICIID INTEGER)
AS
declare variable ProgramModulId integer;
begin
  for select ProgramModul.ProgramModulId from ProgramModul
  left join KullaniciModul on KullaniciModul.KullaniciId = :AKullaniciId and KullaniciModul.ProgramModulId = ProgramModul.ProgramModulId
  where KullaniciModul.ProgramModulId is null into :ProgramModulId
  do insert into KullaniciModul(KullaniciId, ProgramModulId) values(:AKullaniciId, :ProgramModulId);
end
