CREATE PROCEDURE BOYASIPARISPARTILERI(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    CIKISNEMORANI DOUBLE PRECISION,
    TAMIREGIRDIMI INTEGER,
    TAMIREGIRDIMISTR VARCHAR(15),
    TAMIRASAMALARI VARCHAR(255),
    TAMIRSEBEBI VARCHAR(255),
    DISTAMIRTIPIADI VARCHAR(20),
    TAMIRTIPI VARCHAR(255),
    ORMECILER VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(50),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    NETSEVKKILO NUMERIC(12,3),
    BRUTSEVKKILO NUMERIC(12,3),
    RECETEISEMRIIDSTR VARCHAR(30),
    BIRLESTIRILENPARTILER VARCHAR(100),
    SEVKIRSALIYENO VARCHAR(100),
    SEVKIRSALIYETARIHI VARCHAR(100))
AS
declare variable AsilRecetePartiAsamalariId integer;
declare variable FIrsaliyeNo     IrsaliyeNoSize;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable FTamirSebepAdi varchar(20);
declare variable FAsamaAdi varchar(40);
declare variable FPartiTamirId integer;
declare variable FOrmeciAdi varchar(30);
declare variable FTamirTipiAdi varchar(20);
begin
  For Select PartiStok.BoyaSiparisDetayID, PartiStok.PartiKayitID, Parti.PartiYil, Parti.PartiID, PartiStok.PartiSinifId,
             PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.Metre, PartiStok.NetSevkKilo, PartiStok.BrutSevkKilo,
             Parti.ReceteId, Recete.Adi, PartiStokFizikKontrol.CikisNemOrani, TamirTipi.Adi from PartiStok
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId
    left join Recete on Recete.ReceteID = Parti.ReceteID
    left join TamirTipi on TamirTipi.TamirTipiID = PartiStok.TamirTipiID
    where PartiStok.BoyaSiparisKayitID = :ABoyaSiparisKayitID and PartiStok.PartiStokIptalKayitID is null
    into :BoyaSiparisDetayID, :PartiKayitID, :PartiYil, :PartiID, :PartiSinifId,
         :HamAdiToplami, :Kilo, :Metre, :NetSevkKilo, :BrutSevkKilo, :ReceteID, :ReceteAdi, :CikisNemOrani, :DisTamirTipiAdi do
  begin
    CikisNemOrani = Null;
    TamirAsamalari = '';
    TamirSebebi = '';
    Ormeciler = '';
    TamirTipi = '';
    SevkIrsaliyeNo = '';
    SevkIrsaliyeTarihi = '';

    ReceteIsEmriIdStr = '';
    BirlestirilenPartiler = '';
    if (ReceteId is not null) then
    begin
      AsilRecetePartiAsamalariId = Null;
      Select PartiAsamalariId from PartiAsamalari where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.PartininAsilRecetesimi = 1 into :AsilRecetePartiAsamalariId;

      if (AsilRecetePartiAsamalariId is not null) then
        Select ReceteIsEmriIdStr, BirlestirilenPartiler from PAReceteIsEmriBilgisiGetir(:PartikayitId, :ReceteId, :AsilRecetePartiAsamalariId) into :ReceteIsEmriIdStr, :BirlestirilenPartiler;
    end

    TamireGirdimi = 0;
    TamireGirdimiStr = 'Hayır';

    for select PartiTamir.PartiTamirId, TamirSebebi.Adi, TamirTipi.Adi from PartiTamirSinif
      left join PartiTamir on PartiTamir.PartiTamirID = PartiTamirSinif.PartiTamirID
      left join TamirSebebi on TamirSebebi.TamirSebebiId = PartiTamir.TamirSebebiId
      left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
      where PartiTamirSinif.PartiSinifId = :PartiSinifId
      into :FPartiTamirId, :FTamirSebepAdi, FTamirTipiAdi do
    begin
      TamireGirdimi = 1;
      TamireGirdimiStr = 'Tamire girmis';
      if (Char_Length(TamirSebebi) < 230) then
      begin
        if (TamirSebebi <> '') then TamirSebebi = TamirSebebi || cast(', ' as varchar(2) character set UTF8);
        TamirSebebi = TamirSebebi || FTamirSebepAdi;
      end
      if (Char_Length(TamirTipi) < 230) then
      begin
        if (TamirTipi <> '') then TamirTipi = TamirTipi || cast(', ' as varchar(2) character set UTF8);
        TamirTipi = TamirTipi || FTamirTipiAdi;
      end


      for select Asama.Adi from PartiAsamalari
        left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join Asama on Asama.AsamaId = coalesce(PartiAsamalari.AsamaId, Recete.UstAsamaID)
        where PartiAsamalari.PartiTamirId = :FPartiTamirId
        into :FAsamaAdi do
      begin
        if (Char_Length(TamirAsamalari) < 210) then
        begin
          if (TamirAsamalari <> '') then TamirAsamalari = TamirAsamalari || cast(', ' as varchar(2) character set UTF8);
          TamirAsamalari = TamirAsamalari || FAsamaAdi;
        end
      end
    end

    for select Ormeci.Adi from PartiStokDetay
      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
      left join Ormeci on Ormeci.OrmeciId = HamStokIrsaliye.OrmeciId
      where PartiStokDetay.PartiSinifId = :PartiSinifId
      into :FOrmeciAdi do
    begin
      if (Char_Length(Ormeciler) < 220) then
      begin
        if (Ormeciler <> '') then Ormeciler = Ormeciler || cast(', ' as varchar(2) character set UTF8);
        Ormeciler = Ormeciler || FOrmeciAdi;
      end
    end

    for Select BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeNo from BoyaIrsaliyeDetay
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      where BoyaIrsaliyeDetay.PartiSinifID = :PartiSinifID
      group by BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeNo
      into :FIrsaliyeTarihi, :FIrsaliyeNo do
    begin
      if (Char_Length(SevkIrsaliyeNo) < 70) then
      begin
        if (SevkIrsaliyeNo <> '') then SevkIrsaliyeNo = SevkIrsaliyeNo || cast(', ' as varchar(2) character set UTF8);
        SevkIrsaliyeNo = SevkIrsaliyeNo || Format_Float('',FIrsaliyeNo); --EgemenIntegerToStr(FIrsaliyeNo);
      end
      if (Char_Length(SevkIrsaliyeTarihi) < 70) then
      begin
        if (SevkIrsaliyeTarihi <> '') then SevkIrsaliyeTarihi = SevkIrsaliyeTarihi || cast(', ' as varchar(2) character set UTF8);
        SevkIrsaliyeTarihi = SevkIrsaliyeTarihi || TimeStampToDateStr(FIrsaliyeTarihi);
      end
    end

    suspend;
  end
end
