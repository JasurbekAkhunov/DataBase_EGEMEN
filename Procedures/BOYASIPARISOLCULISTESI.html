CREATE PROCEDURE BOYASIPARISOLCULISTESI(
    ABOYASIPARISDETAYID INTEGER)
RETURNS (
    RESULTID VARCHAR(100),
    OLCUADI VARCHAR(40),
    BOYASIPARISOLCUID INTEGER,
    OLCUID INTEGER)
AS
begin
  for select BoyaSiparisOlcu.BoyaSiparisOlcuId, Olcu.OlcuId, Olcu.OlcuAdi from BoyaSiparisOlcu
      left join Olcu on Olcu.OlcuId = BoyaSiparisOlcu.OlcuId
      where BoyaSiparisOlcu.BoyaSiparisDetayId = :ABoyaSiparisDetayId into :BoyaSiparisOlcuId, :OlcuId, :OlcuAdi do
  begin
    ResultId = OlcuId||','||BoyaSiparisOlcuId||','||'"'||Replace(OlcuAdi, ',', '.')||'"';
    suspend;
  end
end
