    APARTIID INTEGER)

    HAMID INTEGER,

    METRE DOUBLE PRECISION,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER)

    into :PartiSinifId, :HamId, :HamAdi, :Kilo, :Metre, :BoyaSiparisYil, :BoyaSiparisID do
  begin
    suspend;
  end
end
