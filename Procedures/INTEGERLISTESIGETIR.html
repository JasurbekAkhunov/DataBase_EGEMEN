Declare Variable VirgulPos      Integer;
Declare Variable IntegerDeger   Varchar(10);
Begin
  IntegerListesi = AIntegerListesi || ' ';
  VirgulPos = Position(',', IntegerListesi);

  While (VirgulPos > 0) Do
  Begin
    IntegerDeger = Trim(SubString(IntegerListesi from 1 for VirgulPos - 1));

    if (Char_Length(IntegerDeger) > 0) Then

      IntegerListesi = SubString(IntegerListesi From VirgulPos + 1);
    else
      IntegerListesi = '';

    VirgulPos = Position(',', IntegerListesi);
  end

  IntegerListesi = Trim(IntegerListesi);

  if (Char_Length(IntegerListesi) > 0) Then
  begin

end
