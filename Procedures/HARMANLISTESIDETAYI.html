CREATE PROCEDURE HARMANLISTESIDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHARMANCINSIID INTEGER,
    AIPLIKISLETMEID INTEGER)
RETURNS (
    HARMANRECETEID INTEGER,
    HARMANKAYITID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MIKTAR DOUBLE PRECISION,
    KULLANIMORANI DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable HarmanMiktari Numeric(12,3);
begin
  For select HarmanRecete.HarmanKayitId, HarmanRecete.HarmanReceteId, HarmanRecete.HamId,
             Ham.Adi, HarmanRecete.Miktar, HarmanListesi.HarmanMiktari, Ik.Adi, HarmanRecete.InsertTarihi, Sk.Adi, HarmanRecete.DegisimTarihi from HarmanListesi(:ABaslangicTarihi, :ABitisTarihi, :AHarmanCinsiId, :AIplikIsletmeID)
  left join HarmanRecete on HarmanRecete.HarmanKayitId = HarmanListesi.HarmanKayitId
  left join Ham on Ham.HamId = HarmanRecete.HamId
  Left Join Kullanici IK on IK.KullaniciId = HarmanRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HarmanRecete.KullaniciId
  where HarmanRecete.HarmanKayitId is not null
  into :HarmanKayitId, :HarmanReceteId, :HamId, :HamAdi, :Miktar, :HarmanMiktari,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    if (HarmanMiktari > 0) then KullanimOrani = (Miktar / HarmanMiktari) * 100;
    suspend;
  end
end
