CREATE PROCEDURE DESENPLANLAMAHAFTA(
    ADEPARTMANID INTEGER)
RETURNS (
    DESENPLANLAMAID INTEGER,
    ONCELIKNO INTEGER,
    DESENISTEKLERIID INTEGER,
    DESENISTEKTIPIID VARCHAR(5),
    DESENISTEKTIPIADI VARCHAR(50),
    DESENISTEKTIPIKISAADI VARCHAR(10),
    DESENISTEKNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PLANLAMAACIKLAMA VARCHAR(100),
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(80),
    ISTEKTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    ISTEKYAPANKULLANICIID INTEGER,
    ISTEKYAPANKULLANICIADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DESENGRUPID VARCHAR(4),
    SABLONBOYUID VARCHAR(10),
    BASKICINSIADI VARCHAR(20),
    BASKITEKNIGIADI VARCHAR(20),
    DESENBEKLEMEDE INTEGER,
    DESENBEKLEBASLATANKULLANICIID INTEGER,
    DESENBEKLEBASLATANKULLANICIADI VARCHAR(40),
    DESENBEKLEMENOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ISTEKBEKLEMEDE INTEGER,
    RENKAYRIMINIYAPANKULLANICIADI VARCHAR(40),
    HAFTAKAYITID INTEGER,
    HAFTAYIL INTEGER,
    HAFTANO INTEGER,
    HAFTABASLANGICTARIH TIMESTAMP,
    HAFTABITISTARIH TIMESTAMP,
    HAFTAYAPLANLAYANKULLANICIID INTEGER,
    HAFTAYAPLANLAYANKULLANICIADI VARCHAR(40),
    LASERGRAVUR INTEGER,
    MAXEXPOSER INTEGER)
AS
begin
  For Select DesenPlanlama.DesenPlanlamaID, DesenPlanlama.OncelikNo,
             DesenPlanlama.DesenIstekleriId, DesenIstekleri.DesenIstekTipiId, DesenIstekTipi.Adi, DesenIstekTipi.KisaAdi,
             Desen.DesenId, Desen.Adi, DesenIstekleri.IstekTarihi, DesenIstekleri.TerminTarihi, DesenIstekleri.IstekYapanKullaniciId,
             IYK.Adi, DesenIstekleri.UniteId, Unite.Adi, Desen.FirmaId, Firma.Adi, Desen.DesenGrupId, SablonB.Olcu, BaskiCinsi.Adi, BaskiTeknigi.Adi,
             Desen.Beklemede, DesenBekleme.BaslatanKullaniciId, DBBK.Adi, DesenBekleme.Notu,
             DesenIstekleri.Beklemede, DesenPlanlama.HaftaKayitId, Hafta.Yil, Hafta.HaftaNo, Hafta.BaslangicTarih, Hafta.BitisTarih,
             DesenPlanlama.HaftayaPlanlayanKullaniciId, HPK.Adi, DesenIstekleri.Notu, DesenPlanlama.PlanlamaAciklama,
             Desen.LaserGravur, Desen.MaxExposer from DesenPlanlama
  left join DesenIstekleri on DesenIstekleri.DesenIstekleriId = DesenPlanlama.DesenIstekleriId
  left join Desen on Desen.DesenId = DesenPlanlama.DesenId
  left join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiId
  left join Firma on Firma.FirmaId = Desen.FirmaId
  left join BaskiCinsi on BaskiCinsi.BaskiCinsiId = Desen.BaskiCinsiId
  left join BaskiTeknigi on BaskiTeknigi.BaskiTeknigiId = Desen.BaskiTeknigiId
  left join DesenOlcu SablonB on SablonB.DesenOlcuID = Desen.SablonBoyuDesenOlcuId
  left join Kullanici IYK on IYK.KullaniciId = DesenIstekleri.IstekYapanKullaniciId
  left join DesenBekleme on DesenBekleme.DesenId = Desen.DesenId and DesenBekleme.Bitti = 0
  left join Kullanici DBBK on DBBK.KullaniciId = DesenBekleme.BaslatanKullaniciId
  left join Unite on Unite.UniteId = DesenIstekleri.UniteId
  left join Hafta on Hafta.HaftaKayitId = DesenPlanlama.HaftaKayitId
  left join Kullanici HPK on HPK.KullaniciId = DesenPlanlama.HaftayaPlanlayanKullaniciId
  where DesenPlanlama.DepartmanId = :ADepartmanId and DesenPlanlama.HaftaKayitId is not null and DesenPlanlama.DesenIstasyonID is null
  order by DesenPlanlama.OncelikNo, DesenIstekleri.TerminTarihi
  into :DesenPlanlamaID, :OncelikNo, :DesenIstekleriId, :DesenIstekTipiId, :DesenIstekTipiAdi, :DesenIstekTipiKisaAdi,
       :DesenId, :DesenAdi, :IstekTarihi, :TerminTarihi, :IstekYapanKullaniciId, :IstekYapanKullaniciAdi,
       :UniteId, :UniteAdi, :FirmaId, :FirmaAdi, :DesenGrupId,
       :SablonBoyuId, :BaskiCinsiAdi, :BaskiTeknigiAdi, :DesenBeklemede, :DesenBekleBaslatanKullaniciId, :DesenBekleBaslatanKullaniciAdi,
       :DesenBeklemeNotu, :IstekBeklemede, :HaftaKayitId, :HaftaYil, :HaftaNo, :HaftaBaslangicTarih, :HaftaBitisTarih,
       :HaftayaPlanlayanKullaniciId, :HaftayaPlanlayanKullaniciAdi, :DesenIstekNotu, :PlanlamaAciklama, :LaserGravur, :MaxExposer  do
  begin
    RenkAyriminiYapanKullaniciAdi = '';

    Select Max(Kullanici.Adi) from DesenIstekleri
      left join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiID = DesenIstekleri.DesenIstekTipiId
      left join Kullanici on Kullanici.KullaniciID = DesenIstekleri.BitirenKullaniciId
      where DesenIstekleri.DesenId = :DesenId and DesenIstekTipi.RenkAyrimIstegimi = 1
      into :RenkAyriminiYapanKullaniciAdi;

    suspend;
  end
end
