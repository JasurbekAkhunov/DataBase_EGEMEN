CREATE PROCEDURE PLANLAMAGRUPANALIZISIPGRUP(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    MAKINAPLANLAMAGRUPADI VARCHAR(30),
    MAKINAPLANLAMAGRUPRAPORSIRANO INTEGER,
    MAXKAPASITE DOUBLE PRECISION,
    BEYAZSAYISI INTEGER,
    BEYAZ NUMERIC(12,3),
    RENKLISAYISI INTEGER,
    RENKLI NUMERIC(12,3),
    YIKAMASAYISI INTEGER,
    YIKAMA NUMERIC(12,3),
    SEKERKASARSAYISI INTEGER,
    SEKERKASAR NUMERIC(12,3),
    KURUBOLUMSAYISI INTEGER,
    KURUBOLUM NUMERIC(12,3),
    CIFTBOYAMASAYISI INTEGER,
    CIFTBOYAMA NUMERIC(12,3),
    SIYAHSAYISI INTEGER,
    SIYAH NUMERIC(12,3),
    NAYLONBOYAMASAYISI INTEGER,

    ANTIKBOYAMASAYISI INTEGER,
    ANTIKBOYAMA NUMERIC(12,3),
    BATIKBOYAMASAYISI INTEGER,

    KUKURTBOYAMASAYISI INTEGER,

declare variable FLastSiparisGrupId varchar(4);
declare variable FLastSiparisGrup2Id varchar(4);
declare variable FlastFirmaId       varchar(20);
declare variable FSiparisGrupId     varchar(4);
declare variable FSiparisGrupAdi    varchar(30);
declare variable FSiparisGrup2Id     varchar(4);
declare variable FSiparisGrup2Adi    varchar(30);
declare variable FFirmaId           varchar(20);
declare variable FFirmaAdi          varchar(100);
declare variable FIslemGrubuId      integer;
declare variable FKilo              Numeric(12,3);
declare variable FPartiAsamalariId  integer;
declare variable FLastMakinaPlanlamaGrupId  varchar(10);
declare variable FMakinaPlanlamaGrupRaporSiraNo integer;
declare variable FMakinaPlanlamaGrupId  varchar(10);
declare variable FMakinaPlanlamaGrupAdi varchar(30);
declare variable FMaxKapasite double precision;
begin
/* Sipariş Grubuna göre Makina Planlama Grubu analiz */
 FLastSiparisGrupId  = '-1';
 FLastSiparisGrup2Id = '-1';
 FlastFirmaId = '-1';
 FLastMakinaPlanlamaGrupId = '-1';

  if (ARaporTipi = 0) then
  begin
    for select Parti.FirmaId, Firma.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, Parti.IslemGrubuId, Sum(PartiStok.Kilo), PartiAsamalari.PartiAsamalariId,

        NaylonBoyamaSayisi    = 0;
        NaylonBoyama          = 0;
        AntikBoyamaSayisi     = 0;

        BatikBoyamaSayisi     = 0;

      end

      begin
        BeyazSayisi = BeyazSayisi + 1;
        Beyaz       = Beyaz + FKilo;
      end
      else if (FIslemGrubuId = 1) then /* Renkli */
      begin
        RenkliSayisi = RenkliSayisi + 1;
        Renkli       = Renkli + FKilo;
      end
      else if (FIslemGrubuId = 2) then /* Yikama */

      end

      end

    left join BolumPlanlama on BolumPlanlama.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and BolumPlanlama.PartiSinifId = PartiStok.PartiSinifId
    left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaId = BolumPlanlama.BolumPlanlamaId
    left join UretimPlanlama on UretimPlanlama.UretimPlanlamaId = UretimPlanlamaDetay.UretimPlanlamaId
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join Makina Mk1 on Mk1.MakinaID = PartiAsamaStokDetay.MakinaId
    left join Makina Mk2 on Mk2.MakinaID = UretimPlanlama.MakinaId
    where (PartiAsamaStok.GirisTarihi >= :ABaslangicTarihi and PartiAsamaStok.GirisTarihi < :ABitisTarihi +1) and
          (Firma.AnaHesapFirmaID  = :AFirmaID or :AFirmaID is null) and

        SiparisGrupAdi        = FSiparisGrupAdi;
        SiparisGrup2Id         = FSiparisGrup2Id;

        FirmaId         = FFirmaId;
        FirmaAdi        = FFirmaAdi;
        MakinaPlanlamaGrupId  = FMakinaPlanlamaGrupId;
        MakinaPlanlamaGrupAdi = FMakinaPlanlamaGrupAdi;
        MakinaPlanlamaGrupRaporSiraNo = FMakinaPlanlamaGrupRaporSiraNo;
        BeyazSayisi           = 0;
        Beyaz                 = 0;
        RenkliSayisi          = 0;
        Renkli                = 0;
        YikamaSayisi          = 0;
        Yikama                = 0;
        SekerKasarSayisi      = 0;

        AntikBoyamaSayisi     = 0;

        KukurtBoyamaSayisi    = 0;

        DigerSayisi           = 0;
        Diger                 = 0;
        ToplamPartiSayisi     = 0;
        ToplamKilo            = 0;

        FLastSiparisGrupId = FSiparisGrupId;

      else if (FIslemGrubuId = 5) then /* CiftBoyama */
      begin
        CiftBoyamaSayisi = CiftBoyamaSayisi + 1;
        CiftBoyama       = CiftBoyama + FKilo;
      end
      else if (FIslemGrubuId = 6) then /* Siyah */

        Diger       = Diger + FKilo;
      end
      ToplamPartiSayisi     = ToplamPartiSayisi + 1;
      ToplamKilo            = ToplamKilo + FKilo;

    left join UretimPlanlama on UretimPlanlama.UretimPlanlamaId = UretimPlanlamaDetay.UretimPlanlamaId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiStok.PartiSinifId
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join Makina Mk1 on Mk1.MakinaID = PartiAsamaStokDetay.MakinaId
    left join Makina Mk2 on Mk2.MakinaID = UretimPlanlama.MakinaId
    where (PartiStok.SevkTarihi >= :ABaslangicTarihi and PartiStok.SevkTarihi <= :ABitisTarihi ) and
          (Firma.AnaHesapFirmaID = :AFirmaID or :AFirmaID is null) and
          Parti.IslemTipiId <> 3 and PartiAsamalari.PartininAsilRecetesimi = 1 and PartiAsamaStok.AsamayaDahil = 1 and
          PartiStok.PartiStokIptalKayitId is null
    group by Parti.PartiKayitId, Parti.FirmaId, Firma.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, Parti.IslemGrubuId, PartiAsamalari.PartiAsamalariId
    order by Parti.FirmaId, BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id, 8
    into :FFirmaId, :FFirmaAdi, :FSiparisGrupId, :FSiparisGrupAdi, :FSiparisGrup2Id, :FSiparisGrup2Adi, :FIslemGrubuId, :FKilo, :FPartiAsamalariId, :FMakinaPlanlamaGrupId do
    begin
      if (FSiparisGrupId is null) then FSiparisGrupId = '';
      if (FSiparisGrup2Id is null) then FSiparisGrup2Id = '';

      FMakinaPlanlamaGrupAdi= Null;
      FMaxKapasite = Null;
      FMakinaPlanlamaGrupRaporSiraNo = 0;

      if (FMakinaPlanlamaGrupId is not null) then
      begin
        Select MakinaPlanlamaGrup.Adi, MakinaPlanlamaGrup.RaporSiraNo, Max(Makina.MaxKapasite) from MakinaPlanlamaGrup
          left join Makina on Makina.MakinaPlanlamaGrupID = MakinaPlanlamaGrup.MakinaPlanlamaGrupId
          where MakinaPlanlamaGrup.MakinaPlanlamaGrupID = :FMakinaPlanlamaGrupId
          group by MakinaPlanlamaGrup.Adi, MakinaPlanlamaGrup.RaporSiraNo
          into :FMakinaPlanlamaGrupAdi, :FMakinaPlanlamaGrupRaporSiraNo, :FMaxKapasite;
      end

      if (FMakinaPlanlamaGrupId is null) then FMakinaPlanlamaGrupId = '';

      if (FLastFirmaID <> FFirmaId or FLastSiparisGrupId <> FSiparisGrupId or FLastSiparisGrup2Id <> FSiparisGrup2Id or FLastMakinaPlanlamaGrupId <> FMakinaPlanlamaGrupId) then
      begin
        if (FLastSiparisGrupId <> '-1') then suspend;
        if (FLastSiparisGrup2Id <> '-1') then suspend;
        SiparisGrupId         = FSiparisGrupId;
        SiparisGrupAdi        = FSiparisGrupAdi;
        SiparisGrup2Id         = FSiparisGrup2Id;
        SiparisGrup2Adi        = FSiparisGrup2Adi;
        FirmaId         = FFirmaId;
        FirmaAdi        = FFirmaAdi;
        MakinaPlanlamaGrupId  = FMakinaPlanlamaGrupId;
        MakinaPlanlamaGrupAdi = FMakinaPlanlamaGrupAdi;
        MakinaPlanlamaGrupRaporSiraNo = FMakinaPlanlamaGrupRaporSiraNo;

        BeyazSayisi           = 0;
        Beyaz                 = 0;
        RenkliSayisi          = 0;
        Renkli                = 0;
        YikamaSayisi          = 0;
        Yikama                = 0;
        SekerKasarSayisi      = 0;
        SekerKasar            = 0;
        KuruBolumSayisi       = 0;
        KuruBolum             = 0;
        CiftBoyamaSayisi      = 0;
        CiftBoyama            = 0;
        SiyahSayisi           = 0;
        Siyah                 = 0;
        NaylonBoyamaSayisi    = 0;
        NaylonBoyama          = 0;
        AntikBoyamaSayisi     = 0;
        AntikBoyama           = 0;
        BatikBoyamaSayisi     = 0;
        BatikBoyama           = 0;
        KukurtBoyamaSayisi    = 0;
        KukurtBoyama          = 0;
        DigerSayisi           = 0;
        Diger                 = 0;
        ToplamPartiSayisi     = 0;
        ToplamKilo            = 0;
        MaxKapasite = FMaxKapasite;

        FLastSiparisGrupId = FSiparisGrupId;
        FLastSiparisGrup2Id = FSiparisGrup2Id;
        FLastFirmaId = FFirmaId;
        FLastMakinaPlanlamaGrupId = FMakinaPlanlamaGrupId;
      end

      if (FIslemGrubuId = 0) then /* beyaz */
      begin
        BeyazSayisi = BeyazSayisi + 1;
        Beyaz       = Beyaz + FKilo;
      end
      else if (FIslemGrubuId = 1) then /* Renkli */
      begin
        RenkliSayisi = RenkliSayisi + 1;
        Renkli       = Renkli + FKilo;
      end
      else if (FIslemGrubuId = 2) then /* Yikama */
      begin
        YikamaSayisi = YikamaSayisi + 1;
        Yikama       = Yikama + FKilo;
      end
      else if (FIslemGrubuId = 3) then /* SekerKasar */
      begin
        SekerKasarSayisi = SekerKasarSayisi + 1;
        SekerKasar       = SekerKasar + FKilo;
      end
      else if (FIslemGrubuId = 4) then /* KuruBolum */
      begin
        KuruBolumSayisi = KuruBolumSayisi + 1;
        KuruBolum       = KuruBolum + FKilo;
      end
      else if (FIslemGrubuId = 5) then /* CiftBoyama */
      begin
        CiftBoyamaSayisi = CiftBoyamaSayisi + 1;
        CiftBoyama       = CiftBoyama + FKilo;
      end
      else if (FIslemGrubuId = 6) then /* Siyah */
      begin
        SiyahSayisi  = SiyahSayisi + 1;
        Siyah        = Siyah + FKilo;
      end
      else if (FIslemGrubuId = 9) then /* AntikBoyama */
      begin
        AntikBoyamaSayisi = AntikBoyamaSayisi + 1;
        AntikBoyama       = AntikBoyama + FKilo;
      end
      else if (FIslemGrubuId = 10) then /* BatikBoyama */
      begin
        BatikBoyamaSayisi = BatikBoyamaSayisi + 1;
        BatikBoyama       = BatikBoyama + FKilo;
      end
      else if (FIslemGrubuId = 11) then /* KukurtBoyama */
      begin
        KukurtBoyamaSayisi = KukurtBoyamaSayisi + 1;
        KukurtBoyama       = KukurtBoyama + FKilo;
      end
      else
      begin
        DigerSayisi = DigerSayisi + 1;
        Diger       = Diger + FKilo;
      end
      ToplamPartiSayisi     = ToplamPartiSayisi + 1;
      ToplamKilo            = ToplamKilo + FKilo;
    end
  end


 if (FLastSiparisGrupId <> '-1') then suspend;
 if (FLastSiparisGrup2Id <> '-1') then suspend;
end
