CREATE PROCEDURE BOYAFATURARAPORUDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASERINO SERINOSIZE5,
    ABASLANGICFATURANO FATURANOSIZE,
    ABITISFATURANO FATURANOSIZE,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICFIRMAGRUP1ID VARCHAR(4),
    ABITISFIRMAGRUP1ID VARCHAR(4),
    ABASLANGICFIRMAGRUP2ID VARCHAR(4),
    ABITISFIRMAGRUP2ID VARCHAR(4),
    ABASLANGICFIRMAGRUP3ID VARCHAR(4),
    ABITISFIRMAGRUP3ID VARCHAR(4),
    ABASLANGICFATURAGRUPID VARCHAR(4),
    ABITISFATURAGRUPID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    AFATURAGRUPARALIKLI INTEGER,
    AENTEGRASYON INTEGER,
    ARAPORDOVIZID VARCHAR(5),
    AIHRACATFATURASI INTEGER,
    ASATISTIPI INTEGER)
RETURNS (
    BOYAFATURADETAYID INTEGER,
    BOYAFATURAID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    KALITENO INTEGER,
    FASON INTEGER,
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    HAMADITOPLAMI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    SIPARISPROSESADI2 VARCHAR(255),
    ACIKLAMA VARCHAR(255),
    PAKETACIKLAMA1 VARCHAR(255),
    PAKETACIKLAMA2 VARCHAR(255),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    SIPARISRENKADI VARCHAR(40),
    MIKTAR DOUBLE PRECISION,
    SIPARISBIRIMFIYAT DOUBLE PRECISION,
    SIPARISFASONBIRIMFIYAT DOUBLE PRECISION,
    SIPARISDOVIZID VARCHAR(5),
    SIPARISFASONDOVIZID VARCHAR(5),
    BIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    DOVIZKURU1 DOUBLE PRECISION,
    DOVIZKURU2 DOUBLE PRECISION,
    DOVIZKURUSTR VARCHAR(50),
    VADEGUN INTEGER,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    BOYAIRSALIYEID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    GIRISIRSALIYENOSTR VARCHAR(50),
    GIRISIRSALIYETARIHISTR VARCHAR(50),
    HAMSTOKBIRIMID INTEGER,
    BIRIMID VARCHAR(2),
    PARTIPAKETID INTEGER,
    KDVORANI DOUBLE PRECISION,
    SIPILAVEISLEMLER VARCHAR(255),
    SIPOZELTESTLER VARCHAR(255),
    SIPOZELAPRELER VARCHAR(255),
    NETMIKTAR DOUBLE PRECISION,
    KAPADEDI INTEGER,
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIIDSTR VARCHAR(20),
    PARTIMIKTAR BOYAMIKTARSIZE,
    PARTIGRUPID TYPE OF COLUMN PARTIGRUP.PARTIGRUPID)
AS
declare variable FNetFiyatYuvarlamaCarpani Double Precision;
declare variable FFaturaDovizId            varchar(5);

declare variable FHareketTipiId            integer;
declare variable FHamTipiId                integer;
begin
  For Select BoyaFaturaRaporu.BoyaFaturaId, BoyaFaturaRaporu.DovizId, BoyaFaturaDetay.hareketTipiId, BoyaFaturaDetay.BoyaFaturaDetayId, BoyaIrsaliyeDetay.PartiKayitId, BoyaIrsaliyeDetay.PartiSinifId,
             Ham.Adi, Parti.PartiYil, Parti.PartiId, Parti.Fason, Parti.ReceteId, Recete.Adi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, SiparisProses.Adi2,
             BoyaFaturaDetay.Aciklama, PartiPaket.Aciklama1, PartiPaket.Aciklama2, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.MusteriRenkKoduVeAdi,
             BoyaIrsaliyeDetay.KaliteNo, BoyaFaturaDetay.Miktar, BoyaFaturaDetay.BirimFiyat, BoyaFaturaDetay.DovizKuru1, BoyaFaturaDetay.DovizKuru2, BoyaFaturaDetay.VadeGun,
             BoyaSiparisDetay.BirimFiyat, BoyaFaturaDetay.KdvOrani,
             PartiStok.DovizId, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             PartiStok.BoyaSiparisDetayId, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, BoyaSiparisDetay.HamStokBirimId, HamStokBirim.BirimId,
             Ham.HamTipiId, Parti.IslemTipiId, IslemTipi.Adi, Parti.IslemGrubuId, IslemGrubu.Adi, BoyaFaturaDetay.PartiPaketId,
             BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.OzelTestler, BoyaSiparisDetay.OzelApreler,
             BoyaFaturaRaporu.NetFiyatYuvarlamaCarpani, Ham.Adi2, Ham.Adi3, Ham.Adi4, PartiStok.NetMiktar,
             PartiStok.KapAdedi, BoyaSiparisDetay.RenkDerinligiId, RenkDerinligi.Adi, BoyaSiparisDetay.FasonBirimFiyati, BoyaSiparisDetay.FasonBirimFiyatDovizId,
             PartiStok.Miktar, Parti.PartiGrupId from BoyaFaturaRaporu(:ABaslangicTarihi, :ABitisTarihi, :ASeriNo, :ABaslangicFaturaNo, :ABitisFaturaNo, :ABaslangicFirmaId, :ABitisFirmaId, :ABaslangicFirmaGrup1Id, :ABitisFirmaGrup1Id, :ABaslangicFirmaGrup2Id,
                              :ABitisFirmaGrup2Id, :ABaslangicFirmaGrup3Id, :ABitisFirmaGrup3Id, :ABaslangicFaturaGrupId, :ABitisFaturaGrupId, :AFirmaAralikli, :AFirmaGrup1Aralikli,
                              :AFirmaGrup2Aralikli, :AFirmaGrup3Aralikli, :AFaturaGrupAralikli, :AEntegrasyon, :ARaporDovizId, 0, :AIhracatFaturasi, :ASatisTipi, Null)
   Left Join BoyaFaturaDetay on BoyaFaturaDetay.BoyaFaturaId =  BoyaFaturaRaporu.BoyaFaturaId
   Left Join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
   Left Join PartiPaket on PartiPaket.PartiPaketId = BoyaFaturaDetay.PartiPaketId
   left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
   left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
   Left Join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
   left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
   left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
   Left Join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
   Left Join Recete on Recete.ReceteId = Parti.ReceteId
   left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
   Left Join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
   left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
   Left Join RafDetay on RafDetay.RafDetayId = BoyaIrsaliyeDetay.RafDetayId
   Left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
   Left Join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
   left join HamStokBirim on HamStokBirim.HamStokBirimId = BoyaFaturaDetay.Miktarbirim
   left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
   where BoyaFaturaDetay.BoyaFaturaId is not null
   into :BoyaFaturaId, :FFaturaDovizId, :FHareketTipiId, :BoyaFaturaDetayId, :PartiKayitId, :PartiSinifId, :HamAdiToplami, :PartiYil, :PartiId, :Fason,
        :ReceteId, :ReceteAdi, :SiparisProsesId, :SiparisProsesAdi, :SiparisProsesAdi2, :Aciklama, :PaketAciklama1, :PaketAciklama2,
        :SiparisRenkAdi, :MusteriRenkKoduVeAdi, :KaliteNo, :Miktar, :BirimFiyat, :DovizKuru1, :DovizKuru2, :VadeGun,
        :SiparisBirimFiyat, :KdvOrani, :SiparisDovizId,
        :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :BoyaIrsaliyeId, :BoyaIrsaliyeDetayId, :BoyaSiparisYil, :BoyaSiparisId,
        :BoyaSiparisDetayId, :GirisIrsaliyeNoStr, :GirisIrsaliyeTarihiStr, :HamStokBirimId, :BirimId, :FHamTipiId,
        :IslemTipiId, :IslemTipiAdi, :IslemGrubuId, :IslemGrubuAdi, :PartiPaketId, :SipIlaveIslemler, :SipOzelTestler, :SipOzelApreler,
        :FNetFiyatYuvarlamaCarpani, :HamAdi2, :HamAdi3, :HamAdi4, :NetMiktar, :KapAdedi, :RenkDerinligiId, :RenkDerinligiIdStr, :SiparisFasonBirimFiyat, :SiparisFasonDovizId,
        :PartiMiktar, :PartiGrupId do
   begin
     Tutar = Miktar * BirimFiyat;
     Tutar = Yuvarla(Tutar, FNetFiyatYuvarlamaCarpani);

     if (PartiSinifId is null) then
     begin
        Fason = 0;
        if (FHareketTipiId = 0) then PartiIdStr = 'Ham Stok Satış';
        else if (FHareketTipiId = 1) then PartiIdStr = 'Ham Stok Alış İade';
        else exception Kayithatasi;
        BoyaSiparisIdStr = '';

        if (FHamTipiId <> 2) then
        begin
          HamStokBirimId = 0;
          BirimId = 'Kg';
        end
        else
        begin
          HamStokBirimId = 1;
          BirimId = 'Mt';
        end
     end
     else
     begin
       PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
       BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
     end

     if (SiparisDovizId is not null and FFaturaDovizId <> SiparisDovizId) then DovizKuruStr = Format_Float('', DovizKuru1) || ' ' || SiparisDovizId || '-' || Format_Float('', DovizKuru2) || ' ' || FFaturaDovizId;
     else DovizKuruStr = '';
     suspend;
   end
end
