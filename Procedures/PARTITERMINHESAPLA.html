CREATE PROCEDURE PARTITERMINHESAPLA(
    APARTIKAYITID INTEGER,
    AMAKINAPLANLAMAGRUPID VARCHAR(10))
RETURNS (
    TERMINTARIHI TIMESTAMP,
    SARJSAYISI DOUBLE PRECISION,
    SARJGUN INTEGER,
    KAZANONCESITERMINGUN INTEGER,
    KURUBOLUMTERMINGUN INTEGER,
    BOYAHANETERMINGUN INTEGER,
    MAKINASAYISI INTEGER,
    PAZARGUNLERISAYISI INTEGER)
AS
declare variable TerminGun              integer;
declare variable KS                     integer;
declare variable AsilReceteSiraNo       integer;
declare variable GunlukKapasite         Double Precision;
declare variable DokumaGunlukKapasite   Double Precision;
declare variable ToplamKilo             Double Precision;
declare variable AsamaId                integer;
declare variable HamTipiId              integer;
declare variable KatSayi                Double Precision;
declare variable OtoTerminHesapSekli    EgeBoolean;
declare variable Sayac integer;
begin
  Select Sistem.Ototerminhesapsekli From Sistem Where Sistem.SistemId = 1 Into :OtoTerminHesapSekli ;

  TerminTarihi = null;
  SarjSayisi = 0;

  Select Sira From PartiAsamalari
    where PartiAsamalari.PartiKayitID = :APartiKayitID and PartiAsamalari.PartininAsilRecetesimi = 1 into :AsilReceteSiraNo;

  if (AsilReceteSiraNo is null) then AsilReceteSiraNo = -1;

  KazanOncesiTerminGun = 0;

  For Select PartiAsamaStok.AsamaId, Asama.GunlukKapasite, Asama.DokumaGunlukKapasite, Max(PartiStok.HamTipiId) from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    where PartiAsamaStok.PartiKayitID = :APartiKayitId and PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.AsamayaDahil = 1 and
          PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamalari.Sira < :AsilReceteSiraNo and
          Asama.CikisOtomatikmi = 0 and Asama.AraKontrolAsamasimi = 0 and Asama.KaliteKontrolAsamasimi = 0 and asama.BoyahaneBolumId is not null
    group by PartiAsamaStok.AsamaId, Asama.GunlukKapasite, Asama.DokumaGunlukKapasite
    into :AsamaId, :GunlukKapasite, :DokumaGunlukKapasite, :HamTipiId do
  begin
    if (GunlukKapasite is null) then GunlukKapasite = 0;
    if (DokumaGunlukKapasite is null) then DokumaGunlukKapasite = 0;
    ToplamKilo = Null;
    TerminGun = 0;

    if ( (HamTipiID <> 2 and GunlukKapasite <> 0) or (HamTipiID = 2 and DokumaGunlukKapasite <> 0) ) then
    begin
      if (HamTipiID <> 2) then
      begin
        Select Sum(PartiStok.Kilo) from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariId
        left join PartiStok on PartiStok.PartiSinifID = PartiAsamaStok.PartiSinifID
        where PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.AsamaID = :AsamaId and PartiAsamaStok.AsamayaDahil = 1 and
              PartiAsamalari.OtomasyonaDahilmi = 1 and PartiStok.HamTipiID <> 2
        into :ToplamKilo;
      end
      else
      begin
        GunlukKapasite = DokumaGunlukKapasite;
        Select Sum(PartiStok.Kilo) from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariId
        left join PartiStok on PartiStok.PartiSinifID = PartiAsamaStok.PartiSinifID
        where PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.AsamaID = :AsamaId and PartiAsamaStok.AsamayaDahil = 1 and
              PartiAsamalari.OtomasyonaDahilmi = 1 and PartiStok.HamTipiID = 2
        into :ToplamKilo;
      end

      if (ToplamKilo is not null and ToplamKilo <> 0) then
        TerminGun = ToplamKilo / GunlukKapasite;

      if (TerminGun > KazanOncesiTerminGun) then KazanOncesiTerminGun = TerminGun;
    end
  end

  BoyahaneTerminGun = 0;
  if (AMakinaPlanlamaGrupId is not null and AMakinaPlanlamaGrupId <> '') then
  begin
    for Select BolumPlanlama.PartiKayitId, IslemGrubu.katSayi from BolumPlanlama
      left join Parti on Parti.PartiKayitID = BolumPlanlama.PartiKayitId
      left join IslemGrubu on IslemGrubu.IslemGrubuID = Parti.IslemGrubuId
      where BolumPlanlama.MakinaPlanlamaGrupId = :AMakinaPlanlamaGrupId and BolumPlanlama.PartiKayitId <> :APartiKayitId
      group by BolumPlanlama.PartiKayitId, IslemGrubu.katSayi
      into :KS, :katSayi do
    begin
      SarjSayisi = SarjSayisi + 1.0 * KatSayi;
    end

    Select BoyamaSarjGun, Count(*) from MakinaPlanlamaGrup
      left join Makina On Makina.MakinaPlanlamaGrupID = MakinaPlanlamaGrup.MakinaPlanlamaGrupId
      where MakinaPlanlamaGrup.MakinaPlanlamaGrupID= :AMakinaPlanlamaGrupId
      group by BoyamaSarjGun
      into :SarjGun, :MakinaSayisi;

    SarjGun = SarjGun * MakinaSayisi;
    if (SarjGun is null) then SarjGun = 0;

    if (SarjGun > 0) then
      BoyahaneTerminGun = SarjSayisi / SarjGun  + 0.5;
    else
      BoyahaneTerminGun = 0;
  end

  KuruBolumTerminGun = 0;


  For Select PartiAsamaStok.AsamaId, Asama.GunlukKapasite, Asama.DokumaGunlukKapasite, Max(PartiStok.HamTipiId) from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    where PartiAsamaStok.PartiKayitID = :APartiKayitId and PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.AsamayaDahil = 1 and
          PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamalari.Sira > :AsilReceteSiraNo and
          Asama.CikisOtomatikmi = 0 and Asama.AraKontrolAsamasimi = 0 and Asama.KaliteKontrolAsamasimi = 0 and asama.BoyahaneBolumId is not null
    group by PartiAsamaStok.AsamaId, Asama.GunlukKapasite, Asama.DokumaGunlukKapasite
    into :AsamaId, :GunlukKapasite, :DokumaGunlukKapasite, :HamTipiId do
  begin
    if (GunlukKapasite is null) then GunlukKapasite = 0;
    if (DokumaGunlukKapasite is null) then DokumaGunlukKapasite = 0;
    ToplamKilo = Null;
    TerminGun = 0;

    if ( (HamTipiID <> 2 and GunlukKapasite <> 0) or (HamTipiID = 2 and DokumaGunlukKapasite <> 0) ) then
    begin
      if (HamTipiID <> 2) then
      begin
        Select Sum(PartiStok.Kilo) from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariId
        left join PartiStok on PartiStok.PartiSinifID = PartiAsamaStok.PartiSinifID
        where PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.AsamaID = :AsamaId and PartiAsamaStok.AsamayaDahil = 1 and
              PartiAsamalari.OtomasyonaDahilmi = 1 and PartiStok.HamTipiID <> 2
        into :ToplamKilo;
      end
      else
      begin
        GunlukKapasite = DokumaGunlukKapasite;
        Select Sum(PartiStok.Kilo) from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariId
        left join PartiStok on PartiStok.PartiSinifID = PartiAsamaStok.PartiSinifID
        where PartiAsamaStok.GirisTarihi is null and PartiAsamaStok.AsamaID = :AsamaId and PartiAsamaStok.AsamayaDahil = 1 and
              PartiAsamalari.OtomasyonaDahilmi = 1 and PartiStok.HamTipiID = 2
        into :ToplamKilo;
      end

      if (ToplamKilo is not null and ToplamKilo <> 0) then
        TerminGun = ToplamKilo / GunlukKapasite;

      if (TerminGun > KuruBolumTerminGun) then KuruBolumTerminGun = TerminGun;
    end
  end

  if (KazanOncesiTerminGun > BoyahaneTerminGun or KazanOncesiTerminGun > KuruBolumTerminGun) then
    TerminTarihi = Current_date + KazanOncesiTerminGun + 2;
  else if (KuruBolumTerminGun > BoyahaneTerminGun) then
    TerminTarihi = Current_date + KuruBolumTerminGun + 1;
  else
    TerminTarihi = Current_date + BoyahaneTerminGun + 1;

  /* Palben*/
  if (OtoTerminHesapSekli = 1 ) then
  begin
    /*
      Kazan öncesi için 1 gün ve kazan sonrasy için 2 gün eklenmesi isteniyor.
      ?ardon yada tyra? yada hem ?ardon-tyra? olursa da 1 gün ekleyece?iz.
      Bulunan termin 5 günden az ise 5 güne tamamlaca?yz.
      Pazar günleri çaly?ma olmady?y için pazar gün sayysy kadar ekleme yapaca?yz.
    */
    PazarGunleriSayisi = 0 ;
    TerminGun = 0 ;
    TerminGun = BoyahaneTerminGun + 3 ;

    select count(partiasamastok.Asamaid)
    from partiasamastok
    where partiasamastok.Partikayitid = :Apartikayitid
      and ( ( partiasamastok.Asamaid = 9 ) or (partiasamastok.Asamaid = 18) or (partiasamastok.Asamaid = 19) )
      and partiasamastok.Giristarihi is null and partiasamastok.Cikistarihi is null
      and partiasamastok.Asamayadahil = 1
    into :Ks ;

    if (Ks > 0 ) then TerminGun = TerminGun + 1 ;

    if ( TerminGun < 5 ) then TerminGun =  5 ;

    TerminTarihi = Current_date + TerminGun ;

    Sayac = 0 ;
    while(  Sayac  <= TerminGun ) do
    begin
      if ( F_Dayofweek(Current_Date + sayac) = 1 ) Then PazarGunleriSayisi = PazarGunleriSayisi + 1 ;
      Sayac = Sayac + 1 ;
    end

    TerminTarihi = TerminTarihi + PazarGunleriSayisi ;
    /* bulunan termin pazara denk geliyor ise bir gün daha arttyr*/
    if ( F_Dayofweek(TerminTarihi) = 1 ) then TerminTarihi = TerminTarihi + 1 ;
   end

  suspend;
end
