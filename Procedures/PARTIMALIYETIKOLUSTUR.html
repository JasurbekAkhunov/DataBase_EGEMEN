CREATE PROCEDURE PARTIMALIYETIKOLUSTUR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    ACILISTARIHI TIMESTAMP,
    KILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION,
    ORTALAMABIRIMFIYAT DOUBLE PRECISION,
    TEORIKBOYAKIMYASALMALIYETI DOUBLE PRECISION,
    BOYAKIMYASALMALIYETI DOUBLE PRECISION,
    BOYAMALIYETI DOUBLE PRECISION,
    KIMYASALMALIYETI DOUBLE PRECISION,
    HAMSTOKMALIYETI DOUBLE PRECISION,
    ASAMAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    ASAMAUMUZERINDENISCILIKASAMAM DOUBLE PRECISION,
    MAKINAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    DOGALGAZSARFIYATI DOUBLE PRECISION,
    ELEKTIRIKSARFIYATI DOUBLE PRECISION,
    OKUNANSUSARFIYATI DOUBLE PRECISION,
    ELEKTRIKBIRIMFIYATI DOUBLE PRECISION,
    DOGALGAZBIRIMFIYATI DOUBLE PRECISION,
    BUHARBIRIMFIYATI DOUBLE PRECISION,
    SUBIRIMFIYATI DOUBLE PRECISION,
    UNITELERASEVKMALIYETI DOUBLE PRECISION,
    BASKIBOYAKIMYASALMALIYETI DOUBLE PRECISION)
AS
declare variable FDefaultBirimCarpan    double precision;
declare variable Fason                  integer;
declare variable KayitSayisi            integer;
declare variable Miktar                 Double Precision;
declare variable UretimeCikisFiyati     Double Precision;
declare variable StokTabanDovizId       varchar(5);
declare variable DovizKuru                  Double Precision;
declare variable DovizID                    varchar(5);
declare variable ARaporTarihi                Date;
declare variable FIrsaliyeDovizId           varchar(5);
declare variable FFaturaDovizId             varchar(5);
declare variable FMaliyet                   Double Precision;
declare variable FAsamaUretimMaliyeti       Double Precision;
declare variable FAsamaIscilikUretimMaliyeti    Double Precision;
declare variable FMakinaUretimMaliyeti          Double Precision;
declare variable FAsamaUretimMaliyetiDovizId    varchar(5);
declare variable FAsamaIscilikUretimMDovizId    varchar(5);
declare variable FMakinaUretimMaliyetiDovizId    varchar(5);
declare variable FGirisTarihi            TimeStamp;
declare variable FCikisTarihi            TimeStamp;
declare variable FKilo                   Numeric(12,3);
declare variable PartiAsamaMaliyetHesapTipi integer;
declare variable Yil integer;
declare variable Ay integer;
declare variable FAraVermeSuresi         Double Precision;
declare variable UrunTipiId     integer;
declare variable MaliyetKaydiVar integer;
declare variable UntIrsKilo       Double Precision;
declare variable UntIrsFiyat       Double Precision;
declare variable UntIrsDovizId  varchar(5);
declare variable FPartiSinifId  integer;
declare variable EgemenFirmaId  integer;
declare variable FPartiAsamalariId integer;
declare variable FBaskiAsamasimi integer;
declare variable FBoyakimyasalmaliyeti        Double Precision;
declare variable FBaskiBoyaKimyasalMaliyeti   Double Precision;
declare variable FUnitelerASevkMaliyeti       Double Precision;
begin
  ARaporTarihi = Current_TimeStamp(0);
  PartiKayitId = APartiKayitId;

  Select StokTabanDovizId, PartiAsamaMaliyetHesapTipi, EgemenFirmaId from sistem where sistemid = 1 into :StokTabanDovizId, :PartiAsamaMaliyetHesapTipi, :EgemenFirmaId;

  if (EgemenFirmaId = 4) then Exit;

  select Parti.AcilisTarihi from Parti
      left join Firma on Firma.FirmaId = Parti.FirmaId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
      left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = Recete.LabReceteAtisId
      left join LabRecete on LabRecete.LabReceteId = LabReceteAtis.LabReceteId
      Where Parti.PartiKayitId = :APartiKayitId
    into  :AcilisTarihi;

  Yil = f_Year(AcilisTarihi);
  Ay  = f_Month(AcilisTarihi);

  Select ElektrikBirimFiyati, DogalGazBirimFiyati, BuharBirimMaliyet, SuBirimFiyati from AylikMasraf
    where AylikMasraf.Yil = :Yil and AylikMasraf.Ay = :Ay
    into :ElektrikBirimFiyati, :DogalGazBirimFiyati, :BuharBirimFiyati, :SuBirimFiyati;

  BoyaKimyasalMaliyeti = 0;
  TeorikBoyaKimyasalMaliyeti = 0;
  BoyaMaliyeti = 0;
  KimyasalMaliyeti = 0;
  BaskiBoyaKimyasalMaliyeti = 0;

  Select Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from PartiStok
   where PartiStok.PartiKayitId = :APartiKayitId
   into :Kilo, :Metre;

  FIrsaliyeDovizId = Null;

  if (Fason = 1) then
  begin
    HamStokMaliyeti = 0;
  end
  else
  begin
    HamStokMaliyeti = 0;
    For Select HamStokIrsaliye.DovizId, Sum(PartiStokDetay.Kilo * HamStokMaster.BirimMaliyet) from PartiStokDetay
      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
      where PartiStokDetay.PartiKayitId = :APartiKayitId
      Group By HamStokIrsaliye.DovizId
      into :FIrsaliyeDovizId, :FMaliyet do
    begin
      /*Emre KORKMAZ-28.03.2019: sistem dovize donusturulmus oldugu icin cevirim yapmadim*/
      /*if (FIrsaliyeDovizId <> SistemDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FIrsaliyeDovizId, :SistemDovizId, :AcilisTarihi) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then DovizKuru = 0;

      HamStokMaliyeti = HamStokMaliyeti + FMaliyet  * DovizKuru;*/
      HamStokMaliyeti = HamStokMaliyeti + FMaliyet;
    end
  end

  AsamaUMUzerindenAsamaMaliyeti  = 0;
  MakinaUMUzerindenAsamaMaliyeti = 0;

  if (PartiAsamaMaliyetHesapTipi = 0) then
  begin
    For Select PartiAsamalari.UretimMaliyeti, PartiAsamalari.UretimMaliyetiDovizId, PartiAsamalari.IscilikUretimMaliyeti, PartiAsamalari.IscilikUretimMaliyetiDovizId, Max(PartiAsamaStokDetay.UretimMaliyeti), Max(PartiAsamaStokDetay.UretimMaliyetiDovizId),
               Min(PartiAsamaStokDetay.GirisTarihi), Max(PartiAsamaStokDetay.CikisTarihi), Max(PartiAsamaStokDetay.AraVermeSuresi) from PartiAsamalari
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamaStok.AsamayaDahil = 1 and
            PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is not null
      group By PartiAsamaStokDetay.MakinaSarjId, PartiAsamalari.UretimMaliyeti, PartiAsamalari.UretimMaliyetiDovizId, PartiAsamalari.IscilikUretimMaliyeti, PartiAsamalari.IscilikUretimMaliyetiDovizId
      into :FAsamaUretimMaliyeti, :FAsamaUretimMaliyetiDovizId, :FAsamaIscilikUretimMaliyeti, :FAsamaIscilikUretimMDovizId, :FMakinaUretimMaliyeti, :FMakinaUretimMaliyetiDovizId, :FGirisTarihi, :FCikisTarihi, :FAraVermeSuresi do
    begin
      if (FAsamaUretimMaliyetiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FAsamaUretimMaliyetiDovizId, :StokTabanDovizId, :FCikisTarihi) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then DovizKuru = 0;
      AsamaUMUzerindenAsamaMaliyeti = AsamaUMUzerindenAsamaMaliyeti + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FAsamaUretimMaliyeti * DovizKuru;

      if (FAsamaIscilikUretimMDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FAsamaIscilikUretimMDovizId, :StokTabanDovizId, :FCikisTarihi) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then DovizKuru = 0;
      AsamaUMUzerindenIscilikAsamaM = AsamaUMUzerindenIscilikAsamaM + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FAsamaIscilikUretimMaliyeti * DovizKuru;

      if (FMakinaUretimMaliyetiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FMakinaUretimMaliyetiDovizId, :StokTabanDovizId, :AcilisTarihi) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then DovizKuru = 0;
      MakinaUMUzerindenAsamaMaliyeti = MakinaUMUzerindenAsamaMaliyeti + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FMakinaUretimMaliyeti * DovizKuru;
    end
  end
  else if (PartiAsamaMaliyetHesapTipi = 3) then
  begin
    Select Sum(GenelUretimGiderPayiTutari + ElektrikPayiTutari + DogalgazPayiTutari + BuharPayiTutari + SuPayiTutari), Sum(IscilikPayiTutari) from PartiAsamalari
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamaStok.AsamayaDahil = 1 and
            PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is not null
      into :MakinaUMUzerindenAsamaMaliyeti, :AsamaUMUzerindenIscilikAsamaM;

  end
  else
  begin
    For Select PartiAsamalari.UretimMaliyetiKg, PartiAsamalari.UretimMaliyetiKgDovizId, PartiAsamalari.IscilikUretimMaliyetiKg, PartiAsamalari.IscilikUretimMaliyetiKgDovizId,
               Sum(PartiAsamaStokDetay.Kilo), Max(PartiAsamaStokDetay.UretimMaliyeti), Max(PartiAsamaStokDetay.UretimMaliyetiDovizId),
               Min(PartiAsamaStokDetay.GirisTarihi), Max(PartiAsamaStokDetay.CikisTarihi), Max(PartiAsamaStokDetay.AraVermeSuresi) from PartiAsamalari
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamaStok.AsamayaDahil = 1 and
            PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is not null
      group By PartiAsamaStokDetay.MakinaSarjId, PartiAsamalari.UretimMaliyetiKg, PartiAsamalari.UretimMaliyetiKgDovizId, PartiAsamalari.IscilikUretimMaliyetiKg, PartiAsamalari.IscilikUretimMaliyetiKgDovizId
      into :FAsamaUretimMaliyeti, :FAsamaUretimMaliyetiDovizId, :FAsamaIscilikUretimMaliyeti, :FAsamaIscilikUretimMDovizId, :FKilo, :FMakinaUretimMaliyeti, :FMakinaUretimMaliyetiDovizId, :FGirisTarihi, :FCikisTarihi, :FAraVermeSuresi do
    begin
      if (FAsamaUretimMaliyetiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FAsamaUretimMaliyetiDovizId, :StokTabanDovizId, :FCikisTarihi) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then DovizKuru = 0;
      AsamaUMUzerindenAsamaMaliyeti = AsamaUMUzerindenAsamaMaliyeti + FKilo * FAsamaUretimMaliyeti * DovizKuru;

      if (FAsamaIscilikUretimMDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FAsamaIscilikUretimMDovizId, :StokTabanDovizId, :FCikisTarihi) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then DovizKuru = 0;
      AsamaUMUzerindenIscilikAsamaM = AsamaUMUzerindenIscilikAsamaM + FKilo * FAsamaIscilikUretimMaliyeti * DovizKuru;

      if (FMakinaUretimMaliyetiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FMakinaUretimMaliyetiDovizId, :StokTabanDovizId, :AcilisTarihi) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then DovizKuru = 0;
      MakinaUMUzerindenAsamaMaliyeti = MakinaUMUzerindenAsamaMaliyeti + ((FCikisTarihi - FGirisTarihi - FAraVermeSuresi) / 0.000694444) * FMakinaUretimMaliyeti * DovizKuru;
    end
  end

  /*AsamaUMUzerindenPartiMaliyeti = AsamaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti;
  MakinaUMUzerindenPartiMaliyeti = MakinaUMUzerindenAsamaMaliyeti + HamStokMaliyeti + BoyaKimyasalMaliyeti;*/

/* eger recetesi arsivlenmis ise teorik kilolardan boya kimyasal maliyeti hesaplanacak */
  for select PartiReceteDetay.HesaplananMiktar, PartiReceteDetay.BirimFiyat, PartiReceteDetay.DovizId from PartiAsamalari
    left join PartiReceteDetay on PartiReceteDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
    left join Urun on Urun.UrunId = PartiReceteDetay.UrunId
    left join UrunBirim UBSD on UBSD.UrunBirimId = PartiReceteDetay.UrunBirimId
    left join UrunFiyat on UrunFiyat.UrunId = PartiReceteDetay.UrunId
    left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
    where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiReceteDetay.UrunId is not null
    order by UrunFiyat.UretimeCikisFiyatiDovizId /* AynÄ± doviz cinsleri icin dovizkuru getire gitmesin diye sirali okuyoruz..*/
  into :Miktar, :UretimeCikisFiyati, :DovizID do
  begin
    if (DovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :StokTabanDovizId, :AcilisTarihi) into :DovizKuru;
    else DovizKuru = 1;

    if (UretimeCikisFiyati is null) then UretimeCikisFiyati = 0;

    UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;
    TeorikBoyaKimyasalMaliyeti = TeorikBoyaKimyasalMaliyeti + Miktar * UretimeCikisFiyati;
  end

  OrtalamaBirimFiyat = Null;

  Select Min(BoyaFatura.DovizId), Sum(BoyaFaturaDetay.BirimFiyat * BoyaFaturaDetay.Miktar) / Sum(BoyaFaturaDetay.Miktar) from BoyaIrsaliyeDetay
    left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
    left join BoyaFatura on BoyaFatura.BoyaFaturaId = BoyaFaturaDetay.BoyaFaturaId
    where BoyaIrsaliyeDetay.PartiKayitId = :APartiKayitId and BoyaIrsaliyeDetay.BoyaFaturaId is not null and BoyaFaturaDetay.Miktar <> 0
    into :FFaturaDovizId, :OrtalamaBirimFiyat;

  if (FFaturaDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FFaturaDovizId, :StokTabanDovizId, :AcilisTarihi) into :DovizKuru;
  else DovizKuru = 1;

  if (DovizKuru is null) then DovizKuru = 0;

  OrtalamaBirimFiyat = OrtalamaBirimFiyat * DovizKuru;
  OrtalamaBirimFiyat = Yuvarla(OrtalamaBirimFiyat, 1000);

  Select DogalgazSarfiyati, ElektirikSarfiyati, OkunanSuSarfiyati from PartiEnerjiSarfiyatiHesapla(:APartiKayitId) into :DogalgazSarfiyati, :ElektirikSarfiyati, :OkunanSuSarfiyati;

  UnitelerASevkMaliyeti = 0;
  select count(PartiKayitId) from PartiMaliyetKumulatif where PartiKayitId = :APartiKayitId into :MaliyetKaydiVar;

  if (MaliyetKaydiVar > 0) then
  begin
    update PartiMaliyetKumulatif set HamStokMaliyeti = :HamStokMaliyeti,
      AsamaUMUzerindenAsamaMaliyeti = :AsamaUMUzerindenAsamaMaliyeti, AsamaUMUzerindenIscilikAsamaM = :AsamaUMUzerindenIscilikAsamaM, MakinaUMUzerindenAsamaMaliyeti = :MakinaUMUzerindenAsamaMaliyeti,
      BoyaKimyasalMaliyeti = :BoyaKimyasalMaliyeti, TeorikBoyaKimyasalMaliyeti = :TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti = :UnitelerASevkMaliyeti,
      BaskiBoyaKimyasalMaliyeti = :BaskiBoyaKimyasalMaliyeti, DegisimTarihi = current_timestamp(0) where PartiKayitId = :APartiKayitId;
  end
  else
  begin
    insert into PartiMaliyetKumulatif(PartiKayitId, HamStokMaliyeti, AsamaUMUzerindenAsamaMaliyeti, AsamaUMUzerindenIscilikAsamaM, MakinaUMUzerindenAsamaMaliyeti,
          BoyaKimyasalMaliyeti, TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti, InsertTarihi, DegisimTarihi, BaskiBoyaKimyasalMaliyeti)
      values(:APartiKayitId, :HamStokMaliyeti, :AsamaUMUzerindenAsamaMaliyeti, :AsamaUMUzerindenIscilikAsamaM, :MakinaUMUzerindenAsamaMaliyeti,
          :BoyaKimyasalMaliyeti, :TeorikBoyaKimyasalMaliyeti, :UnitelerASevkMaliyeti, current_timestamp(0), current_timestamp(0), :BaskiBoyaKimyasalMaliyeti);
  end

  for Select PartiStok.PartiSinifId from PartiStok
    where PartiStok.PartiKayitId = :APartiKayitId
    into :FPartiSinifId do
  begin
    if (FPartiSinifId is not Null) then
    begin
      select Coalesce(BoyaKimyasalMaliyeti, 0), Coalesce(BaskiBoyaKimyasalMaliyeti, 0), Coalesce(UnitelerASevkMaliyeti, 0) from PartiSinifMaliyetiKOlustur(:FPartiSinifId) into :FBoyaKimyasalMaliyeti, :FBaskiBoyaKimyasalMaliyeti, :FUnitelerASevkMaliyeti;
      BoyaKimyasalMaliyeti = BoyaKimyasalMaliyeti + FBoyaKimyasalMaliyeti;
      BaskiBoyaKimyasalMaliyeti = BaskiBoyaKimyasalMaliyeti + FBaskiBoyaKimyasalMaliyeti;
      UnitelerASevkMaliyeti = UnitelerASevkMaliyeti + FUnitelerASevkMaliyeti;
    end
  end
  update PartiMaliyetKumulatif set BoyaKimyasalMaliyeti = :BoyaKimyasalMaliyeti, BaskiBoyaKimyasalMaliyeti = :BaskiBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti = :UnitelerASevkMaliyeti , DegisimTarihi = current_timestamp(0) where PartiKayitId = :APartiKayitId;

  suspend;
end
