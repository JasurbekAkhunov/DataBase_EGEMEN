CREATE PROCEDURE PARTININSINIFINIBIRLESTIR(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    ABIRLESTIRILECEKPARTISINIFID INTEGER)
AS
declare variable KayitSayisi integer;
declare variable HamTipi1  integer;
declare variable HamTipi2  integer;
declare variable Durumu1  integer;
declare variable Durumu2  integer;
declare variable BoyaSiparisDetayId1  integer;
declare variable BoyaSiparisDetayId2  integer;
declare variable BoyaSiparisSarjId1  integer;
declare variable BoyaSiparisSarjId2  integer;
declare variable HamStokMasterId1 integer;
declare variable HamStokMasterId2 integer;
declare variable Fason  integer;

declare variable FRafDetayId            integer;
declare variable FHamId                 integer;
declare variable FKapAdedi              integer;
declare variable FKilo                  Numeric(12,3);
declare variable FMetre                 Numeric(12,3);
declare variable FAdet                  integer;
declare variable FBobinSayisi           integer;
declare variable FPartilemeTarihi       TimeStamp;
declare variable FPartiAsamalariId      integer;
declare variable FBoyaSiparisRezervId   integer;

declare variable FPartiAsamaStokDetayId integer;
declare variable Fmsg varchar(255);
declare variable PartiHamStokMasterIdKarisik integer;
declare variable FKaliteKontrolAsamaId integer;
begin
  Select Parti.Fason from Parti where Parti.PartiKayitId = :APartiKayitId into :Fason;

  if (Fason = 0) then
    Select SPartiHamStokMasterIdKarisik, KaliteKontrolAsamaId from Sistem where SistemID = 1 into :PartiHamStokMasterIdKarisik, :FKaliteKontrolAsamaId;
  else
    Select FPartiHamStokMasterIdKarisik, KaliteKontrolAsamaId from Sistem where SistemID = 1 into :PartiHamStokMasterIdKarisik, :FKaliteKontrolAsamaId;

  Select PartiStok.HamTipiId, PartiStok.Durumu, PartiStok.BoyaSiparisDetayId, PartiStok.BoyaSiparisSarjId, Max(HamStokDetay.HamStokMasterId) from PartiStok
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    where PartiStok.PartiSinifId = :APartiSinifId
    group by PartiStok.HamTipiId, PartiStok.Durumu, PartiStok.BoyaSiparisDetayId, PartiStok.BoyaSiparisSarjId
    into :HamTipi1, :Durumu1, :BoyaSiparisDetayId1, :BoyaSiparisSarjId1, :HamStokMasterId1;

  Select PartiStok.HamTipiId, PartiStok.Durumu, PartiStok.BoyaSiparisDetayId, PartiStok.BoyaSiparisSarjId, Max(HamStokDetay.HamStokMasterId) from PartiStok
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    where PartiStok.PartiSinifId = :ABirlestirilecekPartiSinifId
    group by PartiStok.HamTipiId, PartiStok.Durumu, PartiStok.BoyaSiparisDetayId, PartiStok.BoyaSiparisSarjId
    into :HamTipi2, :Durumu2, :BoyaSiparisDetayId2, :BoyaSiparisSarjId2, :HamStokMasterId2;

  if (HamTipi1 <> HamTipi2) then exception TiplerUymuyorBirlesemez;
  if (Durumu1 <> Durumu2) then exception StokDurumlariUymuyorBirlesemez;

  if (BoyaSiparisDetayId1 <> BoyaSiparisDetayId2) then exception SiparisleriUymuyorBirlesemez;

  if (BoyaSiparisSarjId1 <> BoyaSiparisSarjId2) then exception SiparisleriUymuyorBirlesemez;

  if (PartiHamStokMasterIdKarisik = 0 and HamStokMasterId1 <> HamStokMasterId2) then exception HamStokGirisNoKarisikOlamaz;

  For select count(*) from PartiAsamaStok
    where PartiAsamaStok.PartiKayitId = :APartiKayitId and (PartiAsamaStok.PartiSinifId = :APartiSinifId or PartiAsamaStok.PartiSinifId = :ABirlestirilecekPartiSinifId)
    group by PartiAsamalariId, asamayadahil
    having count(*) < 2 into :KayitSayisi do
  begin
    exception DahilDurumlariUymuyorBirlesemez;
  end

  Select Count(*) from PartiAsamaStokDetay
  Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariID = PartiAsamaStokDetay.PartiAsamalariID
  where PartiAsamaStokDetay.PartiSinifId = :APartiSinifId and PartiAsamaStokDetay.GirisTarihi is not null and PartiAsamaStokDetay.CikisTarihi is null and (PartiAsamaStok.AsamaID <> :FKaliteKontrolAsamaId or :FKaliteKontrolAsamaId is Null) into :KayitSayisi;

  if (KayitSayisi > 0) then exception ImalatDevamEdiyorBirlesemez;

  Select Count(*) from PartiAsamaStokDetay
  Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariID = PartiAsamaStokDetay.PartiAsamalariID
  where PartiAsamaStokDetay.PartiSinifId = :ABirlestirilecekPartiSinifId and PartiAsamaStokDetay.GirisTarihi is not null and PartiAsamaStokDetay.CikisTarihi is null and (PartiAsamaStok.AsamaID <> :FKaliteKontrolAsamaId or :FKaliteKontrolAsamaId is Null) into :KayitSayisi;

  if (KayitSayisi > 0) then exception ImalatDevamEdiyorBirlesemez;


  For select count(*) from PartiTamirSinif
    where PartiTamirSinif.PartiSinifId = :APartiSinifId or PartiTamirSinif.PartiSinifId = :ABirlestirilecekPartiSinifId
    group by Partitamirid
    having count(*) < 2
    into :KayitSayisi do exception FarkliTamirVarBirlesemez;


  Select count(*) from BoyaIrsaliyeDetay
   where BoyaIrsaliyeDetay.PartiSinifId = :APartiSinifId or BoyaIrsaliyeDetay.PartiSinifId = :ABirlestirilecekPartiSinifId into :KayitSayisi;

  if (KayitSayisi = 0) then Select count(*) from PartiStokTarti where PartiStokTarti.PartiSinifId = :APartiSinifId and PartiStokTarti.PartiSinifId = :ABirlestirilecekPartiSinifId into :KayitSayisi;

  if (KayitSayisi > 0) then exception SevkEdilmisTartilmisBirlesemez;

/*  Delete From PartiTamirSinif where PartiTamirSinif.PartiSinifId = :ABirlestirilecekPartiSinifId; */
  Delete From PartiAsamaIcKontrol where PartiAsamaIcKontrol.PartiSinifId = :ABirlestirilecekPartiSinifId;
  Delete From PartiAsamaAraKontrol where PartiAsamaAraKontrol.PartiSinifId = :ABirlestirilecekPartiSinifId;

/*  FMsg = 'PartiAsamaStokDetay Set GirisTarihi = null, CikisTarihi = null (BASLADI)';
  Fmsg = msgYaz(Fmsg);
*/
  For Select PartiAsamaStok.PartiAsamalariId from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :ABirlestirilecekPartiSinifId and PartiAsamaStok.GirisTarihi is not null
      order by sira Desc into :FPartiAsamalariId do
  begin
/*    FMsg = 'PartiAsamaStokDetay Set (DEVAM) PartiAsamalariID = ' || EgemenIntegerToStr(FPartiAsamalariId);
    Fmsg = msgYaz(Fmsg);
*/
    for Select PartiAsamaStokDetayId from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = :ABirlestirilecekPartiSinifId
      order by PartiAsamaStokDetayId desc
      into :FPartiAsamaStokDetayId do
    begin
      Select Count(*) from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = :ABirlestirilecekPartiSinifId
        into :KayitSayisi;

      if (KayitSayisi = 1) then
        Update PartiAsamaStokDetay Set GirisTarihi = null, CikisTarihi = null where PartiAsamaStokDetay.PartiAsamaStokDetayId = :FPartiAsamaStokDetayId;
      else Delete from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiAsamaStokDetayId = :FPartiAsamaStokDetayId; /* bolenerek basladıgı icin kaydı siliyoruz */
    end
  end
/*
  FMsg = 'PartiAsamaStokDetay Set GirisTarihi = null, CikisTarihi = null  BİTTİ';
  Fmsg = msgYaz(Fmsg);
*/
  Delete from PartiAsamaStok where PartiAsamaStok.PartiSinifId = :ABirlestirilecekPartiSinifId;
  Delete From PartiTamirSinif where PartiTamirSinif.PartiSinifId = :ABirlestirilecekPartiSinifId;

  For Select BoyaSiparisRezervId, RafDetayId, HamId, KapAdedi, Kilo, Metre, Adet, BobinSayisi, PartilemeTarihi
     from PartiStokDetay where PartiStokDetay.PartiSinifId = :ABirlestirilecekPartiSinifId
     into :FBoyaSiparisRezervId, :FRafDetayId, :FHamId, :FKapAdedi, :FKilo, :FMetre, :FAdet, :FBobinSayisi, :FPartilemeTarihi do
  begin
    select count(*) from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId and
                                              PartiStokDetay.RafDetayId = :FrafDetayId and
                                             (PartiStokDetay.BoyaSiparisRezervId = :FBoyaSiparisRezervId or (:FBoyaSiparisRezervId is null and PartiStokDetay.BoyaSiparisRezervId is null))
                                              into :kayitSayisi;
    if (KayitSayisi > 0) then
    begin
      update PartiStokDetay set LinkIslemi = 2 where PartiStokDetay.PartiSinifId = :ABirlestirilecekPartiSinifId and
                                                     PartiStokDetay.RafDetayId = :FRafDetayId and
                                                    (PartiStokDetay.BoyaSiparisRezervId = :FBoyaSiparisRezervId or (:FBoyaSiparisRezervId is null and PartiStokDetay.BoyaSiparisRezervId is null));

      Delete from PartiStokDetay where PartiStokDetay.PartiSinifId = :ABirlestirilecekPartiSinifId and
                                       PartiStokDetay.RafDetayId = :FRafDetayId and
                                      (PartiStokDetay.BoyaSiparisRezervId = :FBoyaSiparisRezervId or (:FBoyaSiparisRezervId is null and PartiStokDetay.BoyaSiparisRezervId is null));

      Update PartiStokDetay Set LinkIslemi = 1, Kilo = Kilo + :FKilo, Metre = Metre + :FMetre, Adet = Adet + :FAdet, KapAdedi = KapAdedi + :FKapAdedi,
                                                BobinSayisi = BobinSayisi + :FBobinSayisi
        where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId and
                                                              (PartiStokDetay.BoyaSiparisRezervId = :FBoyaSiparisRezervId or (:FBoyaSiparisRezervId is null and PartiStokDetay.BoyaSiparisRezervId is null));

    end
    else Update PartiStokDetay set LinkIslemi = 1, PartiSinifId = :APartiSinifId where PartiStokDetay.PartiSinifId = :ABirlestirilecekPartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;

  end

  Delete from PartiStok where PartiStok.PartiSinifId = :ABirlestirilecekPartiSinifId;

  Execute Procedure PartiSinifDurumuHesapla(:APartiKayitId, :APartiSinifId);
end
