CREATE PROCEDURE BOYASIPARISMIKTARBELIRLE(
    ABOYASIPARISDETAYID INTEGER)
AS
declare variable TOplamKilo BoyaMiktarSize;
declare variable ToplamAdet integer;
declare variable ToplamKapAdedi integer;
begin
  if (ABoyaSiparisDetayId is not null) then
  begin
    select sum(Kilo), sum(Adet), sum(KapAdedi) from BoyaSiparisOlcu where BoyaSiparisOlcu.BoyaSiparisDetayId = :ABoyaSiparisDetayId into :ToplamKilo, :ToplamAdet, :ToplamKapAdedi;

    update BoyaSiparisDetay set SiparisKilo = Coalesce(:ToplamKilo, 0), Adet = Coalesce(:ToplamAdet, 0), KapAdedi = Coalesce(:ToplamKapAdedi, 0)
    where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId;
  end
end
