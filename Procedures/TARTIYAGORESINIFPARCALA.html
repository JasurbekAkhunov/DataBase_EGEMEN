CREATE PROCEDURE TARTIYAGORESINIFPARCALA(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    APARTISTOKTARTIIDLERI BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    AKULLANICIID INTEGER)
AS
declare variable HamTipiId  integer;
declare variable TopSayisi  integer;
declare variable RetTopSayisi  integer;
declare variable Kilo       Numeric(12,3);
declare variable Metre      Numeric(12,3);
declare variable RetKilo    Numeric(12,3);
declare variable RetMetre   Numeric(12,3);
declare variable ToplamKilo    Numeric(12,3);
declare variable ToplamNetKilo Numeric(12,3);
declare variable ToplamMetre   Numeric(12,3);
declare variable AMiktar    Numeric(12,3);
declare variable AMiktar2   Numeric(12,3);
declare variable ProxyId   varchar(8);
declare variable YeniPartiSinifID  integer;
declare variable PartiYil integer;
declare variable PartiId integer;
declare variable FirmaAdi varchar(100);
declare variable FBlob             Blob;
declare variable Aciklama          varchar(1024);
declare variable Konu              varchar(50);
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable FPSKaliteKontrolAsamaId integer;
declare variable PartiBarkodTakibi integer;
declare variable OtonomTopParcalayabilir EgeBoolean;
declare variable OtonomTopParcalaniyor EgeBoolean;
declare variable TartiSevkKS integer;
declare variable FPartiStokTartiId integer;
declare variable KayitSayisi integer;
declare variable SecilenPartiStokTartiId integer;
declare variable FPSKaliteKontrolId integer;
declare variable YeniPSKaliteKontrolId integer;
declare variable TartiDara  Type of Column PartiStokTarti.Dara;
declare variable TartiKilo  Type of Column PartiStokTarti.Kilo;
declare variable TartiMetre  Type of Column PartiStokTarti.Metre;
declare variable TartiBrutMetre Type of Column PartiStokTarti.BrutMetre;
declare variable TartiAdet  Type of Column PartiStokTarti.Adet;
declare variable APsKaliteKontrolAsamaId integer;
declare Variable FHSSarfFisiYil integer;
declare Variable FHSSarfFisiNo  integer;
declare Variable FHSSarfFisiId  integer;
declare variable SarfTipiId     integer;
declare variable SarfAdi  varchar(30);
begin
  Select DokumaKumasiKilodanTakipEt, PartiBarkodTakibi from Sistem into :DokumaKumasiKilodanTakipEt, :PartiBarkodTakibi;
  Select OtonomTopParcalayabilir from Kullanici where Kullanici.KullaniciID = :AKullaniciId into :OtonomTopParcalayabilir;

  OtonomTopParcalaniyor = 0;
  if (OtonomTopParcalayabilir = 1) then
  begin
    Select Count(PartiStokTarti.BoyaIrsaliyeDetayId)+ Count(PartiStokTarti.HamStokSarfFisiDetayId) from PartiStokTarti
     Where :APartiStokTartiIdleri like '%*'||PartiStokTarti.PartiStokTartiId||'*%' into :TartiSevkKS;
    if (TartiSevkKS = 0) then OtonomTopParcalaniyor = 1;
    else Exception SevkEdilmisTartilmisBolunemez;
  end

  Select Parti.PartiYil, Parti.PartiId, Firma.Adi, PartiStok.HamTipiId, PartiStok.ProxyId, PartiStok.Kilo, PartiStok.Metre from PartiStok
  left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
  left join Firma on Firma.FirmaID = Parti.FirmaID
  where PartiStok.PartiSinifID = :APartiSinifID
  into :PartiYil, :PartiId, :FirmaAdi, :HamTipiId, :ProxyId, :Kilo, :Metre;

  Select Sum(Dara + Kilo), Sum(Kilo), Sum(Metre), Count(*) from PartiStokTarti Where :APartiStokTartiIdleri like '%*'||PartiStokTarti.PartiStokTartiId||'*%'
      into :ToplamKilo, :ToplamNetKilo, :ToplamMetre, :TopSayisi;

  if (TopSayisi = 0) then exit;

  if (HamTipiID <> 2 or DokumaKumasiKilodanTakipEt = 1) then
  begin
    AMiktar  = ToplamKilo;
    AMiktar2 = ToplamMetre;
  end
  else
  begin
    AMiktar  = ToplamMetre;
    AMiktar2 = ToplamKilo;
  end

  Select Max(PSKaliteKontrol.PSKaliteKontrolAsamaId) from PSKaliteKontrol Where :APartiStokTartiIdleri like '%*'||PSKaliteKontrol.PartiStokTartiId||'*%' into :APsKaliteKontrolAsamaId;

  execute procedure PartininSinifiniBol(:APartiKayitId, :APartiSinifId, :AMiktar, :AKullaniciId, :APsKaliteKontrolAsamaId, :AMiktar2, Null, Null, :OtonomTopParcalaniyor);

  Select Max(PartiStok.PartiSinifID) from PartiStok where PartiStok.PartiKayitID = :APartiKayitID into :YeniPartiSinifId;
  Select PSKaliteKontrolAsamaId from PSKaliteKontrolAsama where PSKaliteKontrolAsama.PartiSinifId = :YeniPartiSinifID into :FPSKaliteKontrolAsamaId;

  if (PartiBarkodTakibi = 1) then
  begin
    Update PartiStok set ProxyID = Null, NetMetre = NetMetre - :ToplamMetre, NetKilo = NetKilo - :ToplamNetKilo where PartiStok.partiSinifID = :APartiSinifID;
    Update PartiStok set ProxyID = :ProxyId where PartiStok.partiSinifID = :YeniPartiSinifID;
  end
  else
    Update PartiStok set NetMetre = NetMetre - :ToplamMetre, NetKilo = NetKilo - :ToplamNetKilo where PartiStok.partiSinifID = :APartiSinifID;



  if (OtonomTopParcalaniyor = 1) then
  begin
    rdb$set_context('USER_TRANSACTION', 'OtonomTopParcalaniyor', 1);
    For Select PartiStokTarti.PartiStokTartiId, Coalesce(PartiStokTarti.BolunduguPartiStokTartiId, PartiStokTarti.PartiStokTartiId), PartiStokTarti.Dara, PartiStokTarti.Kilo, PartiStokTarti.Metre, PartiStokTarti.BrutMetre, PartiStokTarti.Adet
      from PartiStokTarti Where :APartiStokTartiIdleri like '%*'||PartiStokTarti.PartiStokTartiId||'*%'
    into :SecilenPartiStokTartiId, :FPartiStokTartiId, :TartiDara, :TartiKilo, :TartiMetre, :TartiBrutMetre, :TartiAdet do
    begin
      For Select PSKaliteKontrol.PSKaliteKontrolId from PSKaliteKontrol
        where PartiStokTartiID = :SecilenPartiStokTartiId into :FPSKaliteKontrolId do
      begin

        YeniPSKaliteKontrolId = Gen_Id(PSKaliteKontrol_Id_Gen,1);
        Insert Into PSKaliteKontrol (PSKaliteKontrolId, PaketlemeId, Tarih, PartiKayitId, PartiSinifId, Makinaid, Personelid, Kaliteno, TopNo, Kilo, Dara, Metre, Brutmetre, En, Gramaj, Donme, Egrilik, Baslangicen, Ortaen, Bitisen, Ortalamaen, Aciklama,
                                     Hatapuani, Paketlemehatapuani, Insertkullaniciid, Kullaniciid, Puanlisistem, Parcayarim, PsKaliteKontrolAsamaid, Onaydurumu, Onaytarihi, Onaykullaniciid, Parcasayisi, Partirenklotno, Musteriaciklama,
                                     Raporboyu, Parcametre, Encekmezlik, Boycekmezlik, Pacadonmesi, Olcuid, Adet, Anahatatipiid, Anahatabolumid, SartliKabul, LinkIslemi)
                             Select :YeniPSKaliteKontrolId, PaketlemeId, Tarih, PartiKayitId, :YeniPartiSinifID, Makinaid, Personelid, KaliteNo, TopNo, :TartiKilo + :TartiDara, :TartiDara, :TartiMetre, :TartiBrutMetre, En, Gramaj, Donme, Egrilik, Baslangicen, Ortaen, Bitisen, Ortalamaen, Aciklama,
                                     Hatapuani, Paketlemehatapuani, :AKullaniciId, :AKullaniciId, Puanlisistem, Parcayarim, :FPSKaliteKontrolAsamaId, OnayDurumu, OnayTarihi, Onaykullaniciid, Parcasayisi, Partirenklotno, Musteriaciklama,
                                     Raporboyu, Parcametre, Encekmezlik, Boycekmezlik, PacaDonmesi, Olcuid, :TartiAdet, Anahatatipiid, Anahatabolumid, SartliKabul, 1 from PSKaliteKontrol
                             where PSKaliteKontrol.PsKaliteKontrolId = :FPSKaliteKontrolId;

        Insert Into PSKaliteKontrolHatalari (PSKaliteKontrolId, HataTipiId, Adet, InsertKullaniciid, KullaniciId)
                                     Select :YeniPSKaliteKontrolId, HataTipiId, Adet, :AKullaniciid, :AKullaniciId From PSKaliteKontrolHatalari where PsKaliteKontrolId = :FPSKaliteKontrolId;

/*        Insert Into PSKaliteKontrolHataMetre (PSKaliteKontrolId, HataTipiId, BaslangicMetre, BitisMetre, HataPuani, HataYeri, HataYeriCm, LinkIslemi, InsertKullaniciId, KullaniciId)
                                     Select :YeniPSKaliteKontrolId, HataTipiId, BaslangicMetre, BitisMetre, HataPuani, HataYeri, HataYeriCm, 1, :AKullaniciid, :AKullaniciId From PSKaliteKontrolHataMetre where PsKaliteKontrolId = :FPSKaliteKontrolId;
  */
        Update PartiStokTarti set LinkIslemi = 1, KapAdedi = 0, Metre = Metre - :TartiMetre, BrutMetre = BrutMetre - :TartiBrutMetre,
                                   Dara = Dara - :TartiDara, Kilo = Kilo - :TartiKilo, Adet = Adet - :TartiAdet, BrutKilo = 0 where PartiStokTartiId = :SecilenPartiStokTartiId;

        Update PSKaliteKontrol set Metre = Metre - :TartiMetre, Brutmetre = BrutMetre - :TartiBrutMetre,
                                   Dara = Dara - :TartiDara, Kilo = Kilo - (:TartiKilo + :TartiDara), Adet = Adet - :TartiAdet, LinkIslemi = 1 where PSKaliteKontrol.PsKaliteKontrolId = :FPSKaliteKontrolId;


      end

      /*Sarf Fisi*/
      SarfAdi = 'YENÄ° PARTI SINIF';
      SarfTipiId = null;
      select Max(SarfTipi.SarfTipiId) from SarfTipi where SarfTipi.Adi = :SarfAdi into :SarfTipiId;
      if (SarfTipiId is null) then
      begin
        insert into SarfTipi (Adi) values (:SarfAdi);
        SarfTipiId = gen_id(SarfTipi_Id_Gen, 0);
      end

      FHSSarfFisiYil = Extract(year from Current_date);
      select id from HamStokSarfFisiNumaratoruAl(:FHSSarfFisiYil) into :FHSSarfFisiNo;
      FHSSarfFisiId = gen_id(HamStokSarfFisi_Id_Gen, 1);

      insert into HamStokSarfFisi(HamStokSarfFisiId, FisYil, FisNo, FisTarihi, SarfTipiId, Notu, InsertKullaniciId, KullaniciId)
         values(:FHSSarfFisiId, :FHSSarfFisiYil, :FHSSarfFisiNo, Current_date, :SarfTipiId, 'Otomatik Sarf', :AKullaniciId, :AKullaniciId);

      execute procedure PartiStokKapSarfFisiIleSevkEt(:FHSSarfFisiId, :APartiKayitId, :APartiSinifId, Null, :SecilenPartiStokTartiId, :AKullaniciId, Null, Null);
      /*Sarf Fisi*/

    end
  end
end
