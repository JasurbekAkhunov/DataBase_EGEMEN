CREATE PROCEDURE NETSISBYSATISIRSALIYELISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    BOYAIRSALIYEID INTEGER,
    ORMEIRSALIYEID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    TRANSFEREDILDI INTEGER)
AS
declare variable OrmeFirmaKodumuz varchar(20);
declare variable FirmaKodumuz varchar(20);
begin
  Select OrmeFirmaKodumuz, FirmaKodumuz from sistem where sistemid = 1 into :OrmeFirmaKodumuz, :FirmaKodumuz; /* boyahaneden Ã¶rmeye transfer icin kesilen irsaliyeler listeye gelmesin */

  if (OrmeFirmaKodumuz is null) then OrmeFirmaKodumuz = '';

  for Select BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo,
             BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.TransferEdildi from BoyaIrsaliye
    left join Firma On Firma.FirmaId = BoyaIrsaliye.FirmaId
    left join Kullanici TK On TK.KullaniciId = BoyaIrsaliye.TransferKullaniciId
    left join BoyaIrsaliyeDetay On BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
    left join Parti On Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and BoyaIrsaliye.HareketTipiId = 0 and
          (BoyaIrsaliyeDetay.FasonHamIade <> 1) and
           BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId is not null and
          (BoyaIrsaliye.FirmaID <> :OrmeFirmaKodumuz)
    group by BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo,
             BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.TransferEdildi
    order by BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo
  into :BoyaIrsaliyeId, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, FiiliSevkTarihi, :TransferEdildi do
  begin
    suspend;
  end

  BoyaIrsaliyeId = null;
  for Select OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.FirmaId, Firma.Adi, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo,
             OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.FiiliSevkTarihi, OrmeIrsaliye.TransferEdildi from OrmeIrsaliye
    left join Firma On Firma.FirmaId = OrmeIrsaliye.FirmaId
    left join Kullanici TK On TK.KullaniciId = OrmeIrsaliye.TransferKullaniciId
    left join StokHareket On StokHareket.OrmeIrsaliyeId = OrmeIrsaliye.OrmeIrsaliyeId
    where OrmeIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and OrmeIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and OrmeIrsaliye.HareketTipi = 2 and
          StokHareket.HamIade <> 1 and StokHareket.OrmeIrsaliyeId is not null and
          (OrmeIrsaliye.FirmaID <> :FirmaKodumuz and OrmeIrsaliye.FirmaID <> :OrmeFirmaKodumuz)
    group by OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.FirmaId, Firma.Adi, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo,
             OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.FiiliSevkTarihi, OrmeIrsaliye.TransferEdildi
  order by OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo
  into :OrmeIrsaliyeId, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi, :TransferEdildi do
  begin
    suspend;
  end
end
