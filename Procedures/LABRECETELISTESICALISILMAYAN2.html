CREATE PROCEDURE LABRECETELISTESICALISILMAYAN2(
    AFIRMAID VARCHAR(20),
    ABASLANGICTALEPTARIHI TIMESTAMP,
    ABITISTALEPTARIHI TIMESTAMP,
    ALABRECETEGRUPID INTEGER,
    AGRUPIPLIKBOYAMI INTEGER,
    ALABRENKGRUPID INTEGER,
    AFASON INTEGER)
RETURNS (
    LABRECETEID INTEGER,
    LABRECETECALISMAID INTEGER,
    HAMADI VARCHAR(255),
    CALISMATARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    RPTMI INTEGER,
    RPTSTR VARCHAR(3),
    ILAVE INTEGER,
    SOKUMUZERIBOYA INTEGER,
    ILAVESOKUMUZERIBOYASTR VARCHAR(20),
    KRITIKCALISMASI INTEGER,
    KRITIKCALISMASISTR VARCHAR(10),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable LrgIplikBoya           integer;
begin
  Select LrgIplikBoya from LabReceteGrupGetir into :LrgIplikBoya;

  for Select LabRecete.LabReceteId, LabReceteCalisma.LabReceteCalismaID, Ham.Adi, LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi,
             LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, LabReceteCalisma.Notu from LabReceteCalisma
  left join LabRecete On LabRecete.LabReceteId = LabReceteCalisma.LabReceteId
  left join Firma On Firma.FirmaId = LabRecete.FirmaId
  left join Ham On Ham.HamId = LabReceteCalisma.HamId
  left join LabReceteGrup On LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
  left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = LabRecete.LabRenkGrupId
  left join LabReceteAtis On LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
  Left Join Kullanici TK on TK.KullaniciId = Firma.TemsilciId
  Left Join Kullanici IK on IK.KullaniciId = LabRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = LabRecete.KullaniciId
  where LabReceteCalisma.CalismaTarihi >= :ABaslangicTalepTarihi and LabReceteCalisma.CalismaTarihi <= :ABitisTalepTarihi and
        LabReceteAtis.LabReceteAtisID is null and
        (LabRecete.FirmaId = :AFirmaId or :AFirmaId is null) and
        (LabRecete.LabReceteGrupId = :ALabReceteGrupId or
        (:ALabReceteGrupId is null and :AGrupIplikBoyami = 0 and LabRecete.LabReceteGrupId <> :LrgIplikBoya) or
        (:ALabReceteGrupId is null and :AGrupIplikBoyami is null) ) and
        (LabRecete.LabRenkGrupId = :ALabRenkGrupId or :ALabRenkGrupId is null) and
        (LabRecete.Fason = :AFason or :AFason is null)
  into :LabReceteId, :LabReceteCalismaID, :HamAdi, :CalismaTarihi, :TerminTarihi, :Rptmi, :Ilave, :SokumUzeriBoya, :KritikCalismasi, :Notu do
  begin
    if (Rptmi = 1) then RptStr = 'Rpt';
    else RptStr = '';

    if (KritikCalismasi = 1) then KritikCalismasiStr = 'Kritik';
    else KritikCalismasiStr = '';

    if (Ilave = 1) then IlaveSokumUzeriBoyaStr = 'İlave';
    else if (SokumUzeriBoya = 1) then IlaveSokumUzeriBoyaStr = 'Söküm Üzeri Boya';
    else IlaveSokumUzeriBoyaStr = '';

    suspend;
  end
end
