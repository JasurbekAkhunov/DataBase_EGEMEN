CREATE PROCEDURE FASONSIPARISFIYATIHESAPLA(
    AFIRMAID VARCHAR(20),
    ASIPARISPROSESID INTEGER,
    AISLEMGRUBUID INTEGER,
    ARENKDERINLIGIID INTEGER,
    AILAVEISLEMLERIDSTR VARCHAR(255),
    ASONUCDOVIZID VARCHAR(5),
    AGRAMAJ DOUBLE PRECISION)
RETURNS (
    BIRIMFIYAT DOUBLE PRECISION,
    SONUCDOVIZID VARCHAR(5))
AS
declare variable AsamaFiyati Double Precision;
declare variable IlaveIslemFiyati Double Precision;
declare variable RenkDerinligiFiyati Double Precision;

declare variable FDovizId                   varchar(5);
declare variable FBirimFiyat                Double Precision;
declare variable FFDovizId                   varchar(5);
declare variable FFBirimFiyat                Double Precision;

declare variable FDovizKuru                  Double Precision;

declare variable VirgulPos integer;
declare variable IlaveIslemlerId integer;
declare variable Str  varchar(30);
declare variable FasonSiparisFiyatHesapTipi integer;
declare variable IGSiparisFiyatKatsayi Double Precision;
declare variable FYuzdeselArtis Double Precision;
begin
  AsamaFiyati = 0;
  IlaveIslemFiyati = 0;
  RenkDerinligiFiyati = 0;

  if (ASonucDovizId is null) then
    Select SistemDovizId, FasonSiparisFiyatHesapTipi from Sistem where Sistemid = 1 into :ASonucDovizID, :FasonSiparisFiyatHesapTipi;
  else
    Select FasonSiparisFiyatHesapTipi from Sistem where Sistemid = 1 into :FasonSiparisFiyatHesapTipi;


  Select SiparisFiyatKatSayi from IslemGrubu where IslemGrubuID = :AIslemGrubuId into :IGSiparisFiyatKatSayi;
  SonucDovizId = ASonucDovizID;

/* Proses Asamalarinin fiyatÄ± hesaplanacak */
  if (FasonSiparisFiyatHesapTipi = 0) then
  begin
    for Select Asama.BirimFiyat, Asama.DovizId, FirmaAsama.BirimFiyat, FirmaAsama.DovizId from SiparisProsesAsamalari
     left join Asama on Asama.AsamaID = SiparisProsesAsamalari.AsamaId
      left join FirmaAsama On FirmaAsama.FirmaId = :AFirmaId and FirmaAsama.AsamaId = Asama.AsamaId
      where SiparisProsesAsamalari.SiparisProsesId = :ASiparisProsesId
      into :FBirimFiyat, :FDovizId, :FFBirimFiyat, :FFDovizId do
    begin
      if (FFDovizId is not null) then
      begin
        FDovizID = FFDovizId;
        FBirimFiyat = FFBirimFiyat;
      end

      if (ASonucDovizId <> FDovizId) then
      begin
        select DovizKuru from EnyakinDovizKuruGetir(:FDovizId, :ASonucDovizId, Current_date) into :FDovizKuru;
        FBirimFiyat = FBirimFiyat * FDovizKuru;
      end

      AsamaFiyati = AsamaFiyati + FBirimFiyat;
    end
  end
  else if (FasonSiparisFiyatHesapTipi = 1) then
  begin
    for Select FirmaSiparisProses.BirimFiyat, FirmaSiparisProses.DovizId from FirmaSiparisProses
      where FirmaSiparisProses.SiparisProsesId = :ASiparisProsesId and FirmaSiparisProses.FirmaID = :AFirmaID
      into :FBirimFiyat, :FDovizId do
    begin
      if (ASonucDovizId <> FDovizId) then
      begin
        select DovizKuru from EnyakinDovizKuruGetir(:FDovizId, :ASonucDovizId, Current_date) into :FDovizKuru;
        FBirimFiyat = FBirimFiyat * FDovizKuru;
      end

      AsamaFiyati = AsamaFiyati + FBirimFiyat;
    end
  end
  else if (FasonSiparisFiyatHesapTipi = 2) then
  begin
    for Select FirmaSiparisProses.BirimFiyat, FirmaSiparisProses.DovizId from FirmaSiparisProses
      where FirmaSiparisProses.SiparisProsesId = :ASiparisProsesId and FirmaSiparisProses.FirmaID = :AFirmaID and
           (FirmaSiparisProses.IslemGrubuID = :AIslemGrubuId or (:AIslemGrubuId is Null and FirmaSiparisProses.IslemGrubuID is Null)) and
           (FirmaSiparisProses.RenkDerinligiID = :ARenkDerinligiId or (:ARenkDerinligiId is Null and FirmaSiparisProses.RenkDerinligiID is Null))
      into :FBirimFiyat, :FDovizId do
    begin
      if (ASonucDovizId <> FDovizId) then
      begin
        select DovizKuru from EnyakinDovizKuruGetir(:FDovizId, :ASonucDovizId, Current_date) into :FDovizKuru;
        FBirimFiyat = FBirimFiyat * FDovizKuru;
      end

      AsamaFiyati = AsamaFiyati + FBirimFiyat;
    end
  end

/* ilave iÅlemÅlerin fiyatÄ± hesaplanacak */
  VirgulPos = PosBul(AIlaveIslemlerIdStr, ',');

  While (VirgulPos > 1) do
  begin
    IlaveIslemlerId = Null;
    Str = EgemenCopyStr(AIlaveIslemlerIdStr, 1, VirgulPos -1);

    IlaveIslemlerId = EgemenIntegerToStr(Str);

    Select IlaveIslemler.BirimFiyat, IlaveIslemler.DovizId, FirmaIlaveIslem.BirimFiyat, FirmaIlaveIslem.DovizId from IlaveIslemler
      left join FirmaIlaveIslem On FirmaIlaveIslem.FirmaId = :AFirmaId and FirmaIlaveIslem.IlaveIslemlerId = IlaveIslemler.IlaveIslemlerId
      where IlaveIslemler.IlaveIslemlerID = :IlaveIslemlerId
      into FBirimFiyat, :FDovizId, :FFBirimFiyat, :FFDovizId;

    if (FFDovizId is not null) then
    begin
      FDovizID = FFDovizId;
      FBirimFiyat = FFBirimFiyat;
    end

    if (ASonucDovizId <> FDovizId) then
    begin
      select DovizKuru from EnyakinDovizKuruGetir(:FDovizId, :ASonucDovizId, Current_date) into :FDovizKuru;
      FBirimFiyat = FBirimFiyat * FDovizKuru;
    end

    if (FBirimFiyat is null) then FBirimFiyat = 0;
    IlaveIslemFiyati = IlaveIslemFiyati + FBirimFiyat;

    if (Char_Length(AIlaveIslemlerIdStr) > VirgulPos) then AIlaveIslemlerIdStr = EgemenCopyStr(AIlaveIslemlerIdStr, VirgulPos + 1, Char_Length(AIlaveIslemlerIdStr) - VirgulPos);
    else AIlaveIslemlerIdStr = '';

    VirgulPos = PosBul(AIlaveIslemlerIdStr, ',');
  end

/* Renk Derinligi fiyatÄ± hesaplanacak */

  if (ARenkDerinligiId is not null and FasonSiparisFiyatHesapTipi <> 2) then
  begin
    Select RenkDerinligi.BirimFiyat, RenkDerinligi.DovizId, FirmaRenkDerinligi.BirimFiyat, FirmaRenkDerinligi.DovizId from RenkDerinligi
      left join FirmaRenkDerinligi On FirmaRenkDerinligi.FirmaId = :AFirmaId and FirmaRenkDerinligi.RenkDerinligiId = RenkDerinligi.RenkDerinligiId
      where RenkDerinligi.RenkDerinligiID = :ARenkDerinligiId
        into FBirimFiyat, :FDovizId, :FFBirimFiyat, :FFDovizId;

    if (FFDovizId is not null) then
    begin
      FDovizID = FFDovizId;
      FBirimFiyat = FFBirimFiyat;
    end

    if (ASonucDovizId <> FDovizId) then
    begin
      select DovizKuru from EnYakinDovizKuruGetir(:FDovizId, :ASonucDovizId, Current_date) into :FDovizKuru;
      FBirimFiyat = FBirimFiyat * FDovizKuru;
    end

    RenkDerinligiFiyati = FBirimFiyat * IGSiparisFiyatKatSayi;
  end

  if (FasonSiparisFiyatHesapTipi = 1) then /* tekboy */
    BirimFiyat = AsamaFiyati + IlaveIslemFiyati;
  else if (FasonSiparisFiyatHesapTipi = 2) then
  begin
    FBirimFiyat = 0;
    FYuzdeselArtis = 0;
    BirimFiyat = AsamaFiyati + IlaveIslemFiyati;
    Select FirmaSiparisProsesDetay.YuzdeselArtis, FirmaSiparisProsesDetay.TutarsalArtis, FirmaSiparisProses.DovizId from FirmaSiparisProses
      left join FirmaSiparisProsesDetay on FirmaSiparisProsesDetay.FirmaSiparisProsesId = FirmaSiparisProses.FirmaSiparisProsesID
      where FirmaSiparisProses.SiparisProsesId = :ASiparisProsesId and FirmaSiparisProses.FirmaID = :AFirmaID and
           (FirmaSiparisProses.IslemGrubuID = :AIslemGrubuId or (:AIslemGrubuId is Null and FirmaSiparisProses.IslemGrubuID is Null)) and
           (FirmaSiparisProses.RenkDerinligiID = :ARenkDerinligiId or (:ARenkDerinligiId is Null and FirmaSiparisProses.RenkDerinligiID is Null)) and
           (:AGramaj Between FirmaSiparisProsesDetay.BaslangicGramaj and FirmaSiparisProsesDetay.BitisGramaj)
      into :FYuzdeselArtis, :FBirimFiyat, :FDovizId;

    if (FYuzdeselArtis is not Null and FYuzdeselArtis > 0) then BirimFiyat = BirimFiyat * ((FYuzdeselArtis / 100) + 1);
    else if (FBirimFiyat is not Null) then
    begin
      if (ASonucDovizId <> FDovizId) then
      begin
        select DovizKuru from EnyakinDovizKuruGetir(:FDovizId, :ASonucDovizId, Current_date) into :FDovizKuru;
        FBirimFiyat = FBirimFiyat * FDovizKuru;
      end

      BirimFiyat = BirimFiyat + FBirimFiyat;
    end
  end
  else
    BirimFiyat = AsamaFiyati + IlaveIslemFiyati + RenkDerinligiFiyati;

  BirimFiyat = Yuvarla(BirimFiyat, 100);

  suspend;
end
