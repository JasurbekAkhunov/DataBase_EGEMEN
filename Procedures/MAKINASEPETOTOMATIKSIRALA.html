CREATE PROCEDURE MAKINASEPETOTOMATIKSIRALA(
    AMAKINAID VARCHAR(10),
    AORDERMODU INTEGER)
AS
declare variable FSepetNo             integer;
declare variable UretimPlanlamaId     integer;
begin
  FSepetNo = 1;

  if (AOrderModu = 0) then
  begin
    Update UretimPlanlama Set SepetNo = 10000, SiralamaYapilacak = 1 where UretimPlanlama.MakinaID = :AMakinaId;

    For Select UretimPlanlama.UretimPlanlamaId from UretimPlanlama
      left join UretimPlanlamaDetay on UretimPlanlamaDetay.UretimPlanlamaId = UretimPlanlama.UretimPlanlamaId
      left join BolumPlanlama on BolumPlanlama.BolumPlanlamaId = UretimPlanlamaDetay.BolumPlanlamaId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = BolumPlanlama.PartiAsamalariId
      left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
      Left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
      where UretimPlanlama.MakinaId = :AMakinaId
      group by UretimPlanlama.UretimPlanlamaId, Parti.TerminTarihi, Parti.AciliyetDerecesiID
      order by Max(PartiAsamaStok.GirisTarihi) nulls last, Parti.TerminTarihi, Parti.AciliyetDerecesiID desc
      into :UretimPlanlamaId do
    begin
      Update UretimPlanlama Set SepetNO = :FSepetNo, SiralamaYapilacak = 0
        where UretimPlanlama.UretimPlanlamaID = :UretimPlanlamaID and UretimPlanlama.SiralamaYapilacak = 1;
      FSepetNo = FSepetNo + 1;
    end
  end
  else
  begin
    For Select UretimPlanlama.UretimPlanlamaId from UretimPlanlama
      left join UretimPlanlamaDetay on UretimPlanlamaDetay.UretimPlanlamaId = UretimPlanlama.UretimPlanlamaId
      left join BolumPlanlama on BolumPlanlama.BolumPlanlamaId = UretimPlanlamaDetay.BolumPlanlamaId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = BolumPlanlama.PartiAsamalariId
      left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
      Left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId and PartiAsamaStok.PartiSinifId = BolumPlanlama.PartiSinifId
      where UretimPlanlama.MakinaId = :AMakinaId
      group by UretimPlanlama.UretimPlanlamaId, UretimPlanlama.SepetNo
      order by Max(PartiAsamaStok.GirisTarihi) nulls last, UretimPlanlama.SepetNo
      into :UretimPlanlamaId do
    begin
      Update UretimPlanlama Set SepetNO = :FSepetNo
        where UretimPlanlama.UretimPlanlamaID = :UretimPlanlamaID;
      FSepetNo = FSepetNo + 1;
    end
  end
end
