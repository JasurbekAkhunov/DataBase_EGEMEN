CREATE PROCEDURE PARTILEMETARTIGIRISIRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    PARTISTOKPARTILEMEID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    DARA DOUBLE PRECISION,
    KILO NUMERIC(18,3),
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    INSERTKULLANICI VARCHAR(160),
    DEGISTIRENKULLANICI VARCHAR(160),
    HAMTIPIADI VARCHAR(80),
    KUMASPAKETLEMENO VARCHAR(120),
    HAMADITOPLAMI VARCHAR(255),
    BOYASIPARISIDSTR VARCHAR(30),
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    ACILIYETDERECESIADI VARCHAR(20),
    FASONSTR VARCHAR(10),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTIIDSTR VARCHAR(30),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PLANLANANKILO NUMERIC(18,3),
    KALITEID VARCHAR(20),
    KALITEADI VARCHAR(255),
    METRE NUMERIC(18,3))
AS
  declare variable ABoyaSiparisKayitId integer;
  declare variable FFason              integer;
Begin
  ABoyaSiparisKayitId = null;
  if (ABoyaSiparisId is not null) then
    Select BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId into :ABoyaSiparisKayitId;

  For
    Select PartiStokPartileme.PartiStokPartilemeId, PartiStokPartileme.PartiKayitId, PartiStokPartileme.PartiSinifId,
           PartiStokPartileme.Dara, PartiStokPartileme.Kilo, PartiStokPartileme.InsertKullaniciId, PartiStokPartileme.InsertTarihi,
           PartiStokPartileme.KullaniciId, PartiStokPartileme.DegisimTarihi, IK.Adi as InsertKullanici, Dk.Adi as DegistirenKullanici,
           HamTipi.ADI HamTipiAdi, PartiKumasPaketlemeNoGetir.KumasPaketlemeNo, PartiStok.HamAdiToplami, Parti.PartiYil, Parti.PartiId, PartiStok.PlanlananKilo,
           PartiStokPartileme.Metre
    From PartiStokPartileme
    left join PartiStok ON PartiStok.PartiSinifID = PartiStokPartileme.PartiSinifID
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join HamTipi ON HamTipi.HamTipiId = PartiStok.HamTipiId
    left join Kullanici IK ON IK.KullaniciId = PartiStokPartileme.InsertKullaniciId
    left join Kullanici DK ON DK.KullaniciId = PartiStokPartileme.KullaniciId
    left join BoyaSiparisDetay ON BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join PartiKumasPaketlemeNoGetir(PartiStok.PartiKayitId, PartiStok.PartiSinifId) ON PartiKumasPaketlemeNoGetir.KumasPaketlemeNo is not null
    where PartiStokPartileme.inserttarihi >= :ABaslangicTarihi and PartiStokPartileme.inserttarihi <= :ABitisTarihi and
          (PArtiStok.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null)
    order by PartiStok.PartiSinifId
    into :PartiStokPartilemeId, :PartiKayitId, :PartiSinifId, :Dara, :Kilo, :InsertKullaniciId, :InsertTarihi,
         :KullaniciId, :DegisimTarihi, :InsertKullanici, :DegistirenKullanici, :HamTipiAdi, :KumasPaketlemeNo,
         :HamAdiToplami, :PartiYil, :PartiId, :PlanlananKilo, :Metre
  do
  begin
    select PartiBilgisiGetir.BoyaSiparisIdSTR, PartiBilgisiGetir.IslemTipiAdi , PartiBilgisiGetir.IslemGrubuAdi,
           PartiBilgisiGetir.AciliyetDerecesiAdi,PartiBilgisiGetir.Fason, PartiBilgisiGetir.FirmaId, PartiBilgisiGetir.FirmaAdi, PartiBilgisiGetir.PartiIdSTR, PartiBilgisiGetir.KaliteId, PartiBilgisiGetir.KaliteAdi FROM PartiBilgisiGetir(:PartiYil, :PartiId, :AKullaniciId)
      into :BoyaSiparisIdSTR,  :IslemTipiAdi, :IslemGrubuAdi, :AciliyetDerecesiAdi, :FasonStr, :FirmaId, :FirmaAdi, :PartiIdSTR, :KaliteId, :KaliteAdi;
    if (FFason = 1) then FasonStr = 'Fason'; Else FasonStr = 'Satış';
    suspend;
  end
end
