CREATE PROCEDURE BARKODLASTOKIRSALIYESISATISYAP(
    AIRSALIYEID INTEGER,
    AFIRMAID VARCHAR(20),
    ABARKODID VARCHAR(20),
    AKULLANICIID INTEGER)
AS
declare variable PaketId integer;
declare variable IK varchar(2);
declare variable TF integer;
declare variable FirmaKodumuz varchar(20);
declare variable OrmeFirmaKodumuz varchar(20);
declare variable EgemenFirmaId varchar(20);
declare variable FHareketTipi integer;
declare variable FDepoId DepoKoduSize;
declare variable FUrunId UrunKoduSize;
declare variable FUrunBirimID integer;
declare variable FMiktar Double Precision;
declare variable FBirimFiyat Double Precision;
declare variable FLot varchar(150);
declare variable KS integer;
declare variable FKdv Double Precision;
declare variable FKapAdedi integer;
declare variable CikilanMiktar Double Precision;
declare variable CikilanKapAdedi  integer;
begin
  if (Char_Length(ABarkodId) < 3) then exception BarkodHatali1;
  TF = EgemenStrToIntOlurmu(ABarkodId);
  if (TF = 0) then exception BarkodHatali1;
  PaketId = EgemenStrToInt(ABarkodId);

  Select FirmaKodumuz, OrmeFirmaKodumuz, EgemenFirmaId from Sistem where sistemid = 1 into :FirmaKodumuz, :OrmeFirmaKodumuz, :EgemenFirmaId;

  select Irsaliye.HareketTipi from Irsaliye where Irsaliye.IrsaliyeId = :AIrsaliyeId into :FHareketTipi;

  select count(StokDetay.StokDetayID) from StokDetay where StokDetayId = :paketid into :KS;

  if (KS = 0) then exception BarkodHatali1;

  select Coalesce(Sum(StokDetay.Miktar), 0), Coalesce(Sum(StokDetay.KapAdedi), 0) from StokDetay where ReferansStokDetayId = :PaketId into :CikilanMiktar, :CikilanKapAdedi;

  select StokDetay.DepoId, StokDetay.UrunId, StokDetay.UrunBirimId, StokDetay.Miktar, StokDetay.BirimFiyat, StokDetay.Lot, StokDetay.Kdv, StokDetay.KapAdedi
    from StokDetay where StokDetayId = :PaketId
      into :FDepoId, :FUrunId, :FUrunBirimID, :FMiktar, :FBirimFiyat, :FLot, :FKdv, :FKapAdedi;

  if (CikilanMiktar >= FMiktar) then  exception BarkodHatali1 ' Bu Barkod Okutulmu≈ü';

  FMiktar = FMiktar - CikilanMiktar;
  FKapAdedi = FKapAdedi - CikilanKapAdedi;
  if (FKapAdedi < 0) then FKapAdedi = 0;

  insert into StokDetay(IrsaliyeId, FirmaID,Tarih, HareketTipi,  DepoId, UrunId, UrunBirimId, Miktar, BirimFiyat, Lot, Kdv, KapAdedi, ReferansStokDetayId  )
                 values(:AIrsaliyeId, :AFirmaId,current_timestamp, :FHareketTipi,  :FDepoId, :FUrunId, :FUrunBirimID, :FMiktar, :FBirimFiyat, :FLot, :FKdv, :FKapAdedi, :PaketId);

end
