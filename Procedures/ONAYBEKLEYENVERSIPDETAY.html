CREATE PROCEDURE ONAYBEKLEYENVERSIPDETAY(
    AKULLANICIID INTEGER,
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    VERILENSIPARISKAYITID INTEGER,
    VERILENSIPARISDETAYID INTEGER,
    SATINALMATALEPKAYITID INTEGER,
    SATINALMATALEPDETAYID INTEGER,
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    URUNID VARCHAR(20),
    HAMID INTEGER,
    URUNADI VARCHAR(80),
    URUNADI2 VARCHAR(80),
    URUNBIRIMID INTEGER,
    BIRIMID VARCHAR(2),
    BIRIMADI VARCHAR(10),
    MIKTAR DOUBLE PRECISION,
    GELENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    REFERANSKODU VARCHAR(30),
    ODEMESEKLIID VARCHAR(10),
    ODEMESEKLIADI VARCHAR(50),
    ONODEMETUTARI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    KULLANILACAKIPLIK1 VARCHAR(40),
    KULLANILACAKIPLIK2 VARCHAR(40),
    KULLANILACAKIPLIK3 VARCHAR(40),
    IPLIKORANI1 DOUBLE PRECISION,
    IPLIKORANI2 DOUBLE PRECISION,
    IPLIKORANI3 DOUBLE PRECISION,
    IPLIKUZUNLUGU1 VARCHAR(15),
    IPLIKUZUNLUGU2 VARCHAR(15),
    IPLIKUZUNLUGU3 VARCHAR(15),
    DENYE INTEGER,
    LYCRAURETICIID INTEGER,
    LYCRAURETICIADI VARCHAR(20),
    MAKINAMARKASI VARCHAR(20),
    MAKINACINSI VARCHAR(20),
    SISTEMSAYISI INTEGER,
    PUS INTEGER,
    FEINE INTEGER,
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    HAMEN INTEGER,
    HAMGRAMAJ INTEGER,
    MAMULEN INTEGER,
    MAMULGRAMAJ INTEGER,
    KONTROLORANI INTEGER,
    KAPALI INTEGER,
    KAPALISTR VARCHAR(6),
    KENARKONSTRUKSIYONTIPIID INTEGER,
    KENARKONSTRUKSIYONTIPIADI VARCHAR(20),
    AMBALAJSEKLIID INTEGER,
    AMBALAJSEKLIADI VARCHAR(10),
    TALEPNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SATINALMAKULLANIMYERIID INTEGER,
    SATINALMAKULLANIMYERIADI VARCHAR(50),
    TESLIMTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KDVORANI DOUBLE PRECISION,
    SONUCDOVIZID VARCHAR(5),
    SONUCBIRIMFIYAT DOUBLE PRECISION,
    SONUCTUTAR DOUBLE PRECISION,
    SONUCKDVDAHILTUTAR DOUBLE PRECISION,
    RESIMDOSYAADI1 VARCHAR(255),
    RESIMDOSYAADI2 VARCHAR(255),
    TEXTDOSYAADI1 VARCHAR(255),
    TEXTDOSYAADI2 VARCHAR(255),
    RESIMDOSYAYAZDIR1 INTEGER,
    RESIMDOSYAYAZDIR2 INTEGER,
    TEXTDOSYAYAZDIR1 INTEGER,
    TEXTDOSYAYAZDIR2 INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    URUNSTOKMEVCUDU DOUBLE PRECISION,
    DEPARTMANADI VARCHAR(30))
AS
declare variable FUrunAdi        varchar(60);
declare variable FHamAdi         varchar(60);
declare variable FHamAdi2        varchar(60);
declare variable KS              integer;
declare variable DovizKuru       Double precision;
declare variable SiparisTarihi   TimeStamp;
declare variable FirmaId         varchar(20);
declare variable HamTipiId       integer;
begin

  For Select VerilenSiparisDetay.VerilenSiparisDetayId, VerilenSiparisDetay.SatinAlmaTalepDetayId, SatinAlmaTalepDetay.SatinAlmaTalepKayitId,
             SatinAlmatalep.SatinAlmaTalepYil, SatinAlmatalep.SatinAlmaTalepId,
             VerilenSiparisDetay.BoyaSiparisKayitId, VerilenSiparisDetay.BoyaSiparisDetayId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             VerilenSiparisDetay.UrunId, VerilenSiparisDetay.HamId,
             VerilenSiparisDetay.UrunBirimId,UrunBirim.BirimId, Birim.Adi, Urun.Adi, Urun.Adi2, VerilenSiparisDetay.Adi, Ham.Adi, Ham.Adi2, Ham.hamTipiId, Ham.ReferansKodu,
             VerilenSiparisDetay.Miktar, VerilenSiparis.OdemeSekliId, OdemeSekli.Adi,
             VerilenSiparisDetay.KdvOrani,
             VerilenSiparisDetay.KullanilacakIplik1, VerilenSiparisDetay.KullanilacakIplik2, VerilenSiparisDetay.KullanilacakIplik3,
             VerilenSiparisDetay.IplikOrani1, VerilenSiparisDetay.IplikOrani2, VerilenSiparisDetay.IplikOrani3,
             VerilenSiparisDetay.IplikUzunlugu1, VerilenSiparisDetay.IplikUzunlugu2, VerilenSiparisDetay.IplikUzunlugu3,
             VerilenSiparisDetay.Denye, VerilenSiparisDetay.LycraUreticiId, LycraUretici.Adi, VerilenSiparisDetay.MakinaMarkasi, VerilenSiparisDetay.MakinaCinsi,
             VerilenSiparisDetay.SistemSayisi, VerilenSiparisDetay.Pus, VerilenSiparisDetay.Feine, VerilenSiparisDetay.TupAcikEnMayli,
             VerilenSiparisDetay.HamEn, VerilenSiparisDetay.HamGramaj, VerilenSiparisDetay.MamulEn, VerilenSiparisDetay.MamulGramaj,
             VerilenSiparisDetay.KontrolOrani, VerilenSiparisDetay.KenarKonstruksiyonTipiId, KenarKonstruksiyonTipi.Adi,
             VerilenSiparisDetay.AmbalajSekliId, AmbalajSekli.Adi, VerilenSiparisDetay.TalepNotu,
             VerilenSiparisDetay.SatinAlmaKullanimYeriId, SatinAlmaKullanimYeri.Adi,
             VerilenSiparisDetay.GelenMiktar, VerilenSiparisDetay.BirimFiyat, VerilenSiparisDetay.OnOdemeTutari, VerilenSiparisDetay.DovizId, VerilenSiparisDetay.VadeGun, VerilenSiparisDetay.TeslimTarihi,
             VerilenSiparisDetay.Kapali, VerilenSiparisDetay.ResimDosyaAdi1, VerilenSiparisDetay.ResimDosyaAdi2,
             VerilenSiparisDetay.TextDosyaAdi1, VerilenSiparisDetay.TextDosyaAdi2,
             VerilenSiparisDetay.ResimDosyayazdir1, VerilenSiparisDetay.ResimDosyayazdir2, VerilenSiparisDetay.TextDosyayazdir1, VerilenSiparisDetay.TextDosyayazdir2,
             VerilenSiparisDetay.Notu, Ik.Adi, VerilenSiparisDetay.InsertTarihi, SK.Adi, VerilenSiparisDetay.DegisimTarihi,
             VerilenSiparisDetay.VerilenSiparisKayitId, Departman.adi FROM OnayBekleyenVerilenSiparisler(:AKullaniciId, :ARaporTipi, :ABaslangicTarihi, :ABitisTarihi)
    left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = OnayBekleyenVerilenSiparisler.VerilenSiparisKayitId
    left join VerilenSiparisDetay on OnayBekleyenVerilenSiparisler.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
    LEFT JOIN Urun ON Urun.UrunId = VerilenSiparisDetay.UrunId
    left join UrunBirim on UrunBirim.UrunBirimId = VerilenSiparisDetay.UrunBirimId and UrunBirim.UrunId = VerilenSiparisDetay.UrunId
    left join Birim on Birim.BirimId = UrunBirim.BirimId
    LEFT JOIN Ham ON Ham.HamId = VerilenSiparisDetay.HamId
    LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = VerilenSiparisDetay.BoyaSiparisKayitId
    left join LycraUretici on LycraUretici.LycraUreticiId = VerilenSiparisDetay.LycraUreticiId
    left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = VerilenSiparisDetay.AmbalajSekliId
    left join OdemeSekli on OdemeSekli.OdemeSekliId = VerilenSiparis.OdemeSekliId
    left join SatinAlmaKullanimYeri on SatinAlmaKullanimYeri.SatinAlmaKullanimYeriId = VerilenSiparisDetay.SatinAlmaKullanimYeriId
    LEFT JOIN SatinAlmaTalepDetay ON SatinAlmaTalepDetay.SatinAlmaTalepDetayId = VerilenSiparisDetay.SatinAlmaTalepDetayId
    left join KenarKonstruksiyonTipi on KenarKonstruksiyonTipi.KenarKonstruksiyonTipiId = VerilenSiparisDetay.KenarKonstruksiyonTipiId
    LEFT JOIN SatinAlmaTalep ON SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
    Left Join Kullanici IK on IK.KullaniciId = VerilenSiparisDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = VerilenSiparisDetay.KullaniciId
  left join Departman on Departman.departmanid = satinalmatalep.departmanid
    --where OnayBekleyenVerilenSiparisler.OnaylayacakKullaniciID = :AKullaniciId and
    --    (VerilenSiparisOnay.OnayDurumu = :ARaporTipi) and
    --    ((:ARaporTipi = 0 and VerilenSiparisDetay.Miktar - VerilenSiparisDetay.Gelenmiktar <= 0) or (VerilenSiparisOnay.OnayTarihi >= :ABaslangicTarihi and VerilenSiparisOnay.OnayTarihi <= :ABitisTarihi))
    order by VerilenSiparisDetay.VerilenSiparisDetayId
  into :VerilenSiparisDetayId, :SatinAlmatalepDetayId, SatinAlmaTalepKayitId, :SatinAlmaTalepYil, SatinAlmaTalepId,
       :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
       :UrunId, :HamId, :UrunBirimId, :BirimId, :BirimAdi, :UrunAdi, :UrunAdi2, :FUrunAdi, :FHamAdi, :FHamAdi2, :HamTipiId, :ReferansKodu, :Miktar,
       :OdemeSekliId, :OdemeSekliAdi, :KdvOrani,
       :KullanilacakIplik1, :KullanilacakIplik2, :KullanilacakIplik3, :IplikOrani1, :IplikOrani2, :IplikOrani3, :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3,
       :Denye, :LycraUreticiId, :LycraUreticiAdi, :MakinaMarkasi, :MakinaCinsi, :SistemSayisi, :Pus, :Feine, :TupAcikEnMayli,
       :HamEn, :HamGramaj, :MamulEn, :MamulGramaj, :KontrolOrani, :KenarKonstruksiyonTipiId, :KenarKonstruksiyonTipiAdi,
       :AmbalajSekliId, :AmbalajSekliAdi, :TalepNotu, :SatinAlmaKullanimYeriId, :SatinAlmaKullanimYeriAdi,
       :GelenMiktar, :BirimFiyat, :OnOdemeTutari, :DovizId, :VadeGun, :Teslimtarihi,
       :Kapali, :ResimDosyaAdi1, :ResimDosyaAdi2, :TextDosyaAdi1, :TextDosyaAdi2, :ResimDosyayazdir1,
       :ResimDosyayazdir2, :TextDosyayazdir1, :TextDosyayazdir2,
       :Notu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :VerilenSiparisKayitId, :DepartmanAdi do
  begin

  Select Min(VerilenSiparisDetay.DovizId) FROM VerilenSiparisDetay
    WHERE VerilenSiparisDetay.VerilenSiparisKayitId = :VerilenSiparisKayitId
    into :SonucDovizId;

  if (SonucDovizId is not null) then
  begin
    Select Count(*) FROM VerilenSiparisDetay
      WHERE VerilenSiparisDetay.VerilenSiparisKayitId = :VerilenSiparisKayitId and VerilenSiparisDetay.DovizId <> :SonucDovizId
      into :KS;

    if (KS > 0) then Select SistemDovizId from Sistem where sistemid = 1 into :SonucDovizId;
  end
  else Select SistemDovizId from Sistem where sistemid = 1 into :SonucDovizId;

  Select SiparisTarihi, FirmaId from VerilenSiparis where VerilenSiparis.VerilenSiparisKayitId = :VerilenSiparisKayitId into :SiparisTarihi, :FirmaId;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;


    if (UrunId is null and HamId is null) then
    begin
      UrunAdi = FUrunAdi;
      BirimId = '';
    end
    else if (HamId is not null) then
    begin
      UrunAdi = FHamAdi;
      UrunAdi2 = FHamAdi2;

      if (HamTipiId <> 2) then BirimId = 'Kg';
      else BirimId = 'Mt';
    end

    SatinAlmaTalepIdStr = '';
    if (SatinAlmaTalepYil is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId)  || ' ';
    KalanMiktar = Miktar - GelenMiktar;

    if (Kapali = 1) then KapaliStr = 'Kapalı';
    else KapaliStr = '';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    SonucBirimFiyat = BirimFiyat;
    if (SonucDovizId <> DovizId) then
    begin
      DovizKuru = 0;
      select DovizKuru from DovizKuruGetir(:DovizId, :SonucDovizId, :SiparisTarihi) into :DovizKuru;

      if (DovizKuru is null) then DovizKuru = 0;

      SonucBirimFiyat = SonucBirimFiyat * DovizKuru;

    end

    SonucTutar = SonucBirimFiyat * Miktar;
    SonucTutar = Yuvarla(SonucTutar, 100);
    SonucKdvDahilTutar = SonucBirimFiyat * Miktar + SonucBirimFiyat * Miktar * KdvOrani / 100;
    SonucKdvDahilTutar = Yuvarla(SonucKdvDahilTutar, 100);

    select sum(DepoStokMevcudu) from UrunDepoStokGetir(:UrunId, Null) into UrunStokMevcudu;

    suspend;
  end
end
