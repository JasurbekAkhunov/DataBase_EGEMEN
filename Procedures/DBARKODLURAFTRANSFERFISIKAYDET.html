CREATE PROCEDURE DBARKODLURAFTRANSFERFISIKAYDET(
    ARAFTRANSFERFISIID INTEGER,
    ABARKODID VARCHAR(30),
    AHEDEFRAFID RAFKODUSIZE,
    AKILO BOYAMIKTARSIZE,
    ABRUTKILO BOYAMIKTARSIZE,
    AMETRE BOYAMIKTARSIZE,
    AKAPADEDI INTEGER,
    AKULLANICIID INTEGER)
AS
Declare Variable DokumaKaliteKontrolId Integer;
Declare Variable StokHareketDetayId Integer;
Declare Variable Kilo Boyamiktarsize;
Declare Variable BrutKilo Boyamiktarsize;
Declare Variable Metre Boyamiktarsize;
Declare Variable KaliteNo Integer;
Declare Variable OrmeRafId Integer;
Declare Variable KapAdedi Integer;
Declare Variable FOrmeRafTransferFisiDetayId Integer;
declare variable MiktarBirim    Integer;
declare variable KS integer;
declare variable EgemenFirmaId integer;
begin
  Select Sistem.EgemenFirmaId from Sistem into :EgemenFirmaId;

  if (EgemenStrToIntOlurMu(Abarkodid) = 1) then
  begin
    Select First 1 DokumaKaliteKontrol.DokumaKaliteKontrolId, null, OrmeRaf.OrmeRafId, Case When DokumaKaliteKontrol.Sakat = 1 then 2 else 1 end,
      DokumaKaliteKontrol.Kilo, DokumaKaliteKontrol.Kilo, DokumaKaliteKontrol.Miktar, 1 From DokumaKaliteKontrol
      Left Join DokumaPlanlama On DokumaPlanlama.DokumaPlanlamaId = DokumaKaliteKontrol.DokumaPlanlamaId
      Left Join DokumaMakina On DokumaMakina.DokumaMakinaId = DokumaPlanlama.DokumaMakinaId
      Left Join OrmeRaf On OrmeRaf.RafId = DokumaKaliteKontrol.RafId And OrmeRaf.DokumaPlanlamaId = DokumaPlanlama.DokumaPlanlamaId And OrmeRaf.DokumaMakinaId = DokumaMakina.DokumaMakinaId And
      ((OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari) >= DokumaKaliteKontrol.Miktar)
      Where DokumaKaliteKontrol.DokumaKaliteKontrolId = :ABarkodId and DokumaKaliteKontrol.SatisStokHareketId is null and DokumaKaliteKontrol.UretimeCikisStokHareketId is null and DokumaKaliteKontrol.SarfStokHareketId is null
    Into :DokumaKaliteKontrolId, :StokHareketDetayId, :OrmeRafId, :KaliteNo, :Kilo, :BrutKilo, :Metre, :KapAdedi;

    if (DokumaKaliteKontrolId is null) then exception BarkodHatali2;

    Select First 1 OrmeRafTransferFisiDetay.YeniOrmeRafID from OrmeRafTransferFisiDetay where OrmeRafTransferFisiDetay.DokumaKaliteKontrolID = :DokumaKaliteKontrolID
      Order By InsertTarihi Desc
    into :OrmeRafId;
  end
  else
  begin
    Select First 1 null, StokHareketDetay.StokHareketDetayId, OrmeRaf.OrmeRafID, OrmeRaf.KaliteNo, (GirisKilo - CikisKilo), (GirisBrutKilo - CikisBrutKilo),
      (GirisMetre - CikisMetre), (StokHareketDetayMiktar.GirisKapAdedi - StokHareketDetayMiktar.CikisKapAdedi), StokHareket.MiktarBirim from StokHareketDetay
      Left Join StokHareket on StokHareket.StokHareketId = StokHareketDetay.StokHareketId
      Left Join StokHareketDetayMiktar on StokHareketDetayMiktar.Barkod = StokHareketDetay.Barkod
      Left Join OrmeRaf on OrmeRaf.OrmeRafID = StokHareketDetay.OrmeRafID
      where StokHareketDetay.Barkod = :Abarkodid and (StokHareketDetay.HareketTipi in (1, 8, 9)) and
            ((StokHareket.MiktarBirim = 1 and (GirisMetre - CikisMetre > 0)) or (StokHareket.MiktarBirim = 0 and (GirisKilo - CikisKilo > 0)))
    Into :DokumaKaliteKontrolId, :StokHareketDetayId, :OrmeRafId, :KaliteNo, :Kilo, :BrutKilo, :Metre, :KapAdedi, :MiktarBirim;

    if (StokHareketDetayId is null) then exception BarkodHatali2;

    Select count(*) from ormeraftransferfisidetay where  ormeraftransferfisidetay.ormeraftransferfisiid =  :ARAFTRANSFERFISIID and   'S'||OrmeRafTransferFisiDetay.StokHareketDetayId =  :ABARKODID into :KS;
    if(KS>0) then exception kayithatasi+ 'Bu Barkod Bu Fişde Girilmiş';

    Select First 1 OrmeRafTransferFisiDetay.YeniOrmeRafID from OrmeRafTransferFisiDetay where OrmeRafTransferFisiDetay.StokHareketDetayId = :StokHareketDetayId
      Order By InsertTarihi Desc
    into :OrmeRafId;

    if ((MiktarBirim = 0 and AKilo <> Kilo) or (MiktarBirim = 1 and AMetre <> Metre)) then /*Miktar Böleceğiz*/
    begin
      Update StokHareketDetay Set Kilo = Kilo - :AKilo, BrutKilo = BrutKilo - :ABrutKilo, Metre = Metre - :AMetre, KapAdedi = KapAdedi - :AKapAdedi  where StokHareketDetayID = :StokHareketDetayId;

      insert into StokHareketDetay(StokHareketID, HareketTipi, IplikStokID, IplikStokNumarator, Kilo, BrutKilo, Metre, AmbalajNo, PSKaliteKontrolID, IplikPaketlemeID, GirisKaliteKontrolId, KapAdedi, BobinGramaji, KonikRengiId, InsertKullaniciId, BolunduguStokHareketDetayId)
        Select StokHareketID, HareketTipi, IplikStokID, IplikStokNumarator, :AKilo, :ABrutKilo, :AMetre, AmbalajNo, PSKaliteKontrolID, IplikPaketlemeID, GirisKaliteKontrolId, :AKapAdedi, BobinGramaji, KonikRengiId, :AKullaniciId, :StokHareketDetayId from StokHareketDetay
        where StokHareketDetayId = :StokHareketDetayID
      Returning StokHareketDetayID into :StokHareketDetayID;

    end
  end

  FOrmeRafTransferFisiDetayId = gen_ID(OrmeRafTransferFisiDetay_Id_gen, 1);
  if (EgemenFirmaId = 151) then AKapAdedi = 1;   /*Lico Tex*/

  Insert Into OrmeRafTransferFisiDetay(OrmeRafTransferFisiDetayId, OrmeRafTransferFisiId, OrmeRafId, Kilo, Brutkilo, KapAdedi, RafId, KaliteNo, InsertKullaniciId, KullaniciId, DokumaKaliteKontrolId, StokHareketDetayId)
                             Values(:FOrmeRafTransferFisiDetayId, :ARafTransferFisiId, :OrmeRafId, iif (:MiktarBirim = 0, :AKilo, :AMetre), :ABrutkilo, :AKapAdedi, :AHedefRafId, :KaliteNo, :AKullaniciId, :AKullaniciId, :DokumaKaliteKontrolId, :StokHareketDetayId);

  if (StokHareketDetayId is Not Null) then
  begin
    Update StokHareketDetay Set OrmeRafTransferFisiDetayID = :FOrmeRafTransferFisiDetayId where StokHareketDetayID = :StokHareketDetayId;
  end
end
