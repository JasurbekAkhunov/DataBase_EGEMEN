CREATE PROCEDURE BOYASIPARISRAPORU_5(
    AIMALATTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AFIRMAYETKILIID INTEGER,
    AISLEMTIPIID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AMUSTERIKALITENO VARCHAR(20),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AHAMID INTEGER,
    AKULLANICIID INTEGER,
    AKALITEID VARCHAR(20),
    ATALIMATKODU VARCHAR(20),
    AMAKINAID VARCHAR(10),
    ATEKRARURETIM INTEGER,
    ADESENID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AMARKALI INTEGER,
    AKOMISYONLU INTEGER,
    ALAMINASYON INTEGER,
    ARECETETURUID INTEGER,
    ABATCHNO VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ABASKITIPIID INTEGER,
    AHAMGRUP1ID VARCHAR(4),
    AHAMGRUP2ID VARCHAR(4),
    AHAMGRUP3ID VARCHAR(4),
    AHAMGRUP4ID VARCHAR(4),
    ASIPARISRECETEDURUMU INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTISINIFID INTEGER,
    BOYAIRSALIYEID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    BOYAFATURAID INTEGER,
    BIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    TARTIBRUT NUMERIC(12,3),
    TARTIBOBINSAYISI INTEGER,
    BOYAIRSALIYEHAREKETTIPIID INTEGER,
    BOYAIRSALIYEHAREKETTIPI VARCHAR(10),
    KALITE2TARTIKILO BOYAMIKTARSIZE,
    KALITE2TARTIBRUTKILO BOYAMIKTARSIZE)
AS
declare variable PartiKayitId               integer;
declare variable Entegrasyon               integer;
declare variable AltIskontoCarpani         Double Precision;
declare variable Tevkifat                  integer;
declare variable Fason                     integer;
declare variable KdvOrani                  Double Precision;
declare variable SistemDovizID             VarChar(5);
begin
  Select SistemDovizID from Sistem where SistemID = 1 into :SistemDovizID;

  For Select DISTINCT BoyaIrsaliyeDetay.PartiKayitId, BoyaIrsaliyeDetay.PartiSinifId, BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi,
         BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo, BoyaFatura.FaturaTarihi, BoyaFatura.BoyaFaturaId, BoyaFatura.AltIskontoCarpani, BoyaFatura.Tevkifat, BoyaFatura.Entegrasyon,
         BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.Kilo, BoyaIrsaliyeDetay.BrutMetre, BoyaIrsaliyeDetay.Metre, BoyaSiparisRaporu_4.Fason, BoyaSiparisRaporu_4.BoyaSiparisDetayID, BoyaIrsaliye.HareketTipiId, BoyaSiparisRaporu_4.Kalite2TartiKilo, BoyaSiparisRaporu_4.Kalite2TartiBrutKilo
         from BoyaSiparisRaporu_4(:AImalatTipiId, :ABaslangicTarihi, :ABitisTarihi, :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AFirmaYetkiliId, :AIslemTipiId, :AIslemGrubuId, :ASatisTipi, :ABoyaSiparisTipiId, :ALabReceteId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMusteriKaliteNo, :AOnayDurumu, :APartilenen, :AKapaliDurumu, :ASiparisiAlanKullaniciId, :AYurtDisi, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AHamId, :AKullaniciId, :AKaliteId, :ATalimatKodu, :AMakinaId, :ATekrarUretim, :ADesenID, :ATemsilciId, :AMarkali, :AKomisyonlu, :ALaminasyon, :AReceteTuruId, :ABatchNo, :ASertifikaNo, :ABaskiTipiId, :AHamGrup1Id, :AHamGrup2Id, :AHamGrup3Id, :AHamGrup4Id, :ASiparisReceteDurumu)
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = BoyaSiparisRaporu_4.PartiSinifId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    left join BoyaFatura on BoyaFatura.BoyaFaturaId = BoyaIrsaliyeDetay.BoyaFaturaId
    where BoyaIrsaliyeDetay.PartiSinifId is not null
    into :PartiKayitId, :PartiSinifId, :BoyaIrsaliyeId, :BoyaIrsaliyeDetayId, :IrsaliyeSeriNo, :IrsaliyeNo,
         :Irsaliyetarihi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :BoyaFaturaId, :AltIskontoCarpani, :Tevkifat, :Entegrasyon,
         :BrutKilo, :NetKilo, :BrutMetre, :NetMetre, :Fason, :BoyaSiparisDetayID, :BoyaIrsaliyeHareketTipiId, :Kalite2TartiKilo, :Kalite2TartiBrutKilo do
  begin
    if ( (Fason = 0 and ASatisBoyaSiparisFiyatGorebilir = 1) or (Fason = 1 and AFasonBoyaSiparisFiyatGorebilir = 1)) then
    begin
      if (BoyaFaturaId is not null and Entegrasyon = 1) then
      begin

        if (Tevkifat = 1) then
        begin
          Select Sum(BoyaFaturaDetay.BirimFiyat * (Select DovizKuru from EnYakinDovizKuruGetir(BoyaFatura.DovizID, :SistemDovizID, BoyaFatura.FaturaTarihi))),
          Sum(BoyaFaturaDetay.Miktar * BoyaFaturaDetay.BirimFiyat * (Select DovizKuru from EnYakinDovizKuruGetir(BoyaFatura.DovizID, :SistemDovizID, BoyaFatura.FaturaTarihi)) * ((BoyaFaturaDetay.KdvOrani / 100 / 3) + 1)) from BoyaFaturaDetay
          Left Join BoyaFatura on BoyaFatura.BoyaFaturaID = BoyaFaturaDetay.BoyaFaturaID
          where BoyaFaturaDetay.BoyaFaturaID = :BoyaFaturaID and BoyaFaturaDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID
          into :BirimFiyat, :Tutar;
        end
        else
        begin
          Select Sum(BoyaFaturaDetay.BirimFiyat * (Select DovizKuru from EnYakinDovizKuruGetir(BoyaFatura.DovizID, :SistemDovizID, BoyaFatura.FaturaTarihi))),
            Sum(BoyaFaturaDetay.Miktar * BoyaFaturaDetay.BirimFiyat * (Select DovizKuru from EnYakinDovizKuruGetir(BoyaFatura.DovizID, :SistemDovizID, BoyaFatura.FaturaTarihi)) * ((BoyaFaturaDetay.KdvOrani / 100) + 1)) from BoyaFaturaDetay
          Left Join BoyaFatura on BoyaFatura.BoyaFaturaID = BoyaFaturaDetay.BoyaFaturaID
          where BoyaFaturaDetay.BoyaFaturaID = :BoyaFaturaID and BoyaFaturaDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID
          into :BirimFiyat, :Tutar;
        end

        Tutar = (Tutar - Tutar * AltIskontoCarpani);
        Tutar = Yuvarla(Tutar, 100);
      end
      else
      begin
        Tutar = null;
        BirimFiyat = Null;
      end
    end
    else
    begin
      Tutar = null;
      BirimFiyat = Null;
    end

    Select Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara), Sum(BobinSayisi) from PartiStokTarti
      where PartiStokTarti.BoyaIrsaliyeDetayId = :BoyaIrsaliyeDetayId
      into :TartiBrut, :TartiBobinSayisi;

    BoyaIrsaliyeHareketTipi = null;
    if (BoyaIrsaliyeHareketTipiId = 0) then BoyaIrsaliyeHareketTipi = 'Satış';
    if (BoyaIrsaliyeHareketTipiId = 1) then BoyaIrsaliyeHareketTipi = 'Alış İade';
    if (BoyaIrsaliyeHareketTipiId = 2) then BoyaIrsaliyeHareketTipi = 'Satış İade';

    suspend;
  end
end
