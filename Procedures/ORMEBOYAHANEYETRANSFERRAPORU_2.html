CREATE PROCEDURE ORMEBOYAHANEYETRANSFERRAPORU_2
RETURNS (
    IDSTR VARCHAR(20),
    STOKHAREKETID INTEGER,
    IPLIKIRSALIYEDETAYID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    IPLIKTALEPIDSTR VARCHAR(30),
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    BOYASIPARISDETAYID INTEGER,
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    ORMEMAKINAID VARCHAR(10),
    ORMECIMAKINANO VARCHAR(10),
    DOKUMAMAKINAID MAKINAKODUSIZE,
    OLCUADI ADI40,
    ADET INTEGER,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20))
AS
declare Variable BoyaSiparisYil integer;
declare Variable BoyaSiparisId  integer;
declare Variable IpliktalepYil integer;
declare Variable IpliktalepId  integer;
begin
  For Select OrmeBoyahaneyeTransferRaporu_1.IdStr, StokHareket.StokHareketId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             Ham.Adi, Renk.Adi, StokHareket.BoyaSiparisKayitId, StokHareket.OrmeSiparisDetayId,
             OrmeSiparisDetay.SiparisSiraNo, StokHareket.BoyaSiparisDetayId, BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id,
             StokHareket.KapAdedi, StokHareket.Miktar, BoyaSiparis.FirmaId, Firma.Adi, StokHareket.Lot,
             OrmeRaf.OrmeMakinaId, OrmeRaf.OrmeciMakinaNo, OrmeRaf.DokumaMakinaId, Olcu.OlcuAdi, StokHareket.Adet,
             OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3 from OrmeBoyahaneyeTransferRaporu_1
    left join StokHareket on StokHareket.OrmeFisId = OrmeBoyahaneyeTransferRaporu_1.OrmeFisId or StokHareket.OrmeIrsaliyeId = OrmeBoyahaneyeTransferRaporu_1.OrmeIrsaliyeId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join Ham on Ham.HamId = StokHareket.HamId
    left join Renk on Renk.RenkId = StokHareket.RenkId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
    left join Olcu on Olcu.OlcuId = OrmeRaf.OlcuId
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeRaf.OrmePlanlamaId
    where OrmeBoyahaneyeTransferRaporu_1.IplikIrsaliyeID is null and StokHareket.HamStokIrsaliyeID is null
    into :IdStr, :StokHareketId, :BoyaSiparisYil, :BoyaSiparisId, :HamAdi, :RenkAdi, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :BoyaSiparisDetayId, :SiparisGrupId, :SiparisGrup2Id,
         :KapAdedi, :Miktar, :SiparisFirmaId, :SiparisFirmaAdi, :Lot, :OrmeMakinaId, :OrmeciMakinaNo, :DokumaMakinaId, :OlcuAdi, :Adet,
         :IplikBoyu1, :IplikBoyu2, :IplikBoyu3 do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    suspend;
  end

  OrmeMakinaId = Null;
  OrmeciMakinaNo = Null;
  DokumaMakinaId = Null;
  OlcuAdi = Null;
  Adet = Null;
  IplikBoyu1 = Null;
  IplikBoyu2 = Null;
  IplikBoyu3 = Null;

  For Select OrmeBoyahaneyeTransferRaporu_1.IdStr, IplikIrsaliyeDetay.IplikIrsaliyeDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             Ham.Adi, Renk.Adi, IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId,
             OrmeSiparisDetay.SiparisSiraNo, IplikTalepDetay.BoyaSiparisDetayId, BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id,
             IplikIrsaliyeDetay.KapAdedi, IplikIrsaliyeDetay.Miktar, BoyaSiparis.FirmaId, Firma.Adi,
             IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, IplikIrsaliyeDetay.Lot from OrmeBoyahaneyeTransferRaporu_1
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeId = OrmeBoyahaneyeTransferRaporu_1.IplikIrsaliyeId
    left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikIrsaliyeDetay.IplikTalepDetayId
    left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
    left join Renk on Renk.RenkId = IplikIrsaliyeDetay.RenkId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    where OrmeBoyahaneyeTransferRaporu_1.IplikIrsaliyeID is not null and IplikIrsaliyeDetay.HamStokIrsaliyeID is null
    into :IdStr, :IplikIrsaliyeDetayId, :BoyaSiparisYil, :BoyaSiparisId, :HamAdi, :RenkAdi, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :BoyaSiparisDetayId, :SiparisGrupId, :SiparisGrup2Id,
         :KapAdedi, :Miktar, :SiparisFirmaId, :SiparisFirmaAdi, :IplikTalepYil, :IplikTalepId, :Lot do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    IplikTalepIdStr = '';
    if (IplikTalepId is not null) then IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId);
    suspend;
  end
end
