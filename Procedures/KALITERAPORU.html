    left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 On SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join Renk OrmeRenk On OrmeRenk.RenkId = OrmeSiparisDetay.RenkId
    left join SiparisProses on SiparisProses.SiparisProsesId = Kalite.SiparisProsesId
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
    left join Parti on Parti.PartikayitId = PartiStok.PartikayitId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    left join DesenOlcu RaportB on RaportB.DesenOlcuID = Desen.DesenRaportBoyuDesenOlcuID
    left join Ham on Ham.HamId = Kalite.HamId
    Left Join Kullanici IK on IK.KullaniciId = Kalite.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = Kalite.KullaniciId
    Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    Left Join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId

    Left Join BaskiTipi on BaskiTipi.BaskiTipiId = Kalite.BaskiTipiID
    Left Join SiparisGrup KG on KG.SiparisGrupId = Kalite.GrupID
    where (Kalite.KaliteAdi like '%' || :AKaliteAdi || '%' or :AKaliteAdi is Null) and
          (Kalite.HamAdi like '%' || :AHamAdi || '%' or :AHamAdi is Null) and
          (Kalite.ReceteKodu = :AReceteKodu or :AReceteKodu is Null) and

          ((:AHamGrup1Aralikli = 1 and Ham.HamGrup1 >= :ABaslangicGrup1Id  and (Ham.HamGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AHamGrup1Aralikli = 0) or (:AHamGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and

          ((:AHamGrup3Aralikli = 1 and Ham.HamGrup3 >= :ABaslangicGrup3Id and (Ham.HamGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AHamGrup3Aralikli = 0) or (:AHamGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
          ((:AHamGrup4Aralikli = 1 and Ham.HamGrup4 >= :ABaslangicGrup4Id and (Ham.HamGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AHamGrup4Aralikli = 0) or (:AHamGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = '')) and
          (Kalite.DesenKodu = :ADesenKodu or :ADesenKodu is null) and
          (Kalite.SiparisProsesId = :ASiparisProsesId or :ASiparisProsesId is Null) and
          (Kalite.OzelTestler like '%' || :AOzelTestler || '%' or :AOzelTestler is Null) and
          (Kalite.OzelApreler like '%' || :AOzelApreler || '%' or :AOzelApreler is Null) and
          (Kalite.IlaveIslemler like '%' || :AIlaveIslemler || '%' or :AIlaveIslemler is Null) and
          (Kalite.Kapali = :AKapali or :AKapali is null) and
          ((:AEnAralikli = 1 and Kalite.EnIstenen >= :AEnIstenenBaslangic  and (Kalite.EnIstenen <= :AEnIstenenBitis or :AEnIstenenBitis = '')) or (:AEnAralikli = 0 and :AEnIstenenBaslangic = '') or (:AEnAralikli = 0 and :AEnIstenenBaslangic = Kalite.EnIstenen) or (:AEnAralikli = 1 and :AEnIstenenBaslangic = '' and :AEnIstenenBitis = '')) and
          ((:AGramajAralikli = 1 and Kalite.GramajIstenen >= :AGramajIstenenBaslangic and (Kalite.GramajIstenen <= :AGramajIstenenBitis or :AGramajIstenenBitis = '')) or (:AGramajAralikli = 0 and :AGramajIstenenBaslangic = '') or (:AGramajAralikli = 0 and :AGramajIstenenBaslangic = Kalite.GramajIstenen) or (:AGramajAralikli = 1 and :AGramajIstenenBaslangic = '' and :AGramajIstenenBitis = ''))
    into :KaliteId, :KaliteAdi, :PartiSinifId, :PartiYil, :PartiId, :HamAdi, :ReceteKodu, :ReceteAdi,
         :Denye, :SiparisProsesId, :SiparisProsesAdi, :EnIstenen, :GramajIstenen, :CekmezlikEnIstenen,
         :CekmezlikBoyIstenen, :DonmeIstenen, :SansIstenen, :Pus, :Feine, :DinlenmisGramaj, :IplikBoyu1,
         :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :TupAcikEnMayli, :EnVerilen, :GramajVerilen, :DonmeVerilen,
         :CekmezlikEnVerilen, :CekmezlikBoyVerilen, :CekmezlikEnVerilen2, :CekmezlikBoyVerilen2,
         :DesenKodu, :DesenAdi, :DesenVaryantAdi,
         :MetrajNumunesiVar, :KartelaStokMevcudu, :DesenRaportBoyuId, :FlamKodu, :OrmeRenkAdi,
         :Notu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :IlaveIslemler, :SiparisFirmaID, :SiparisFirmaAdi,
         :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi, :SiparisTarihi, :FBoyaSiparisYil, :FBoyaSiparisID, :MusteriSiparisNo, :BoyaSipDetayNotu, :OrmeSipDetayNotu,
         :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :HamAdi2, :HamAdi3, :HamAdi4, :TalimatKodu, :MinSiparisMiktari, :KapaliStr,
         :LabGrupId, :Composition, :BaskiTipiAdi, :Aciklama, :GrupId, :KaliteGrupAdi, :MiktarGun, :OzelApreler, :HamEn, :MakinaUstuGramaj do
  begin
    if (FBoyaSiparisYil is not null) then BoyaSiparisNoStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') ' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
    else BoyaSiparisNoStr = '';

    if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    else PartiIdStr = '';

    if (MetrajNumunesiVar = 1) then MetrajNumunesiVarStr = 'Var';
    else MetrajNumunesiVarStr = 'Yok';

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    select  Sum(SevkeHazirKilo), Sum(SevkeHazirBrutKilo), Sum(SevkeHazirTartiBrutKilo), Sum(iif(Durumu = 7, 1, 0)),
            Sum(UretimdekiKilo), Sum(UretimdekiBrutKilo), Sum(UretimdekiTartiBrutKilo), Sum(iif(Durumu < 7, 1, 0)),
            Sum(Kalite2SevkeHazirKilo), Sum(Kalite2SevkeHazirBrutKilo), Sum(Kalite2SevkeHazirTartiBrutKilo)
            from (
      select /*iif(PartiStok.Durumu = 7, PartiStok.NetKilo, 0) SevkeHazirKilo, iif(PartiStok.Durumu = 7, PartiStok.Kilo, 0) SevkeHazirBrutKilo,*/
             Sum(iif(PartiStok.Durumu = 7 and PartiStokTarti.BoyaIrsaliyeDetayId is null, PartiStokTarti.Kilo, 0)) SevkeHazirKilo,
             Sum(iif(PartiStok.Durumu = 7 and PartiStokTarti.BoyaIrsaliyeDetayId is null, PartiStokTarti.BrutKilo, 0)) SevkeHazirBrutKilo,
             Sum(iif(PartiStok.Durumu = 7 and PartiStokTarti.BoyaIrsaliyeDetayId is null, PartiStokTarti.Kilo + PartiStokTarti.Dara, 0)) SevkeHazirTartiBrutKilo,
             Sum(iif(PartiStok.Durumu = 7 and PartiStokTarti.BoyaIrsaliyeDetayId is null and PartiStokTarti.KaliteNo > 1, PartiStokTarti.Kilo, 0)) Kalite2SevkeHazirKilo,
             Sum(iif(PartiStok.Durumu = 7 and PartiStokTarti.BoyaIrsaliyeDetayId is null and PartiStokTarti.KaliteNo > 1, PartiStokTarti.BrutKilo, 0)) Kalite2SevkeHazirBrutKilo,
             Sum(iif(PartiStok.Durumu = 7 and PartiStokTarti.BoyaIrsaliyeDetayId is null and PartiStokTarti.KaliteNo > 1, PartiStokTarti.Kilo + PartiStokTarti.Dara, 0)) Kalite2SevkeHazirTartiBrutKilo,
             iif(PartiStok.Durumu < 7, PartiStok.NetKilo, 0) UretimdekiKilo, iif(PartiStok.Durumu < 7, PartiStok.Kilo, 0) UretimdekiBrutKilo,
             Sum(iif(PartiStok.Durumu < 7, PartiStokTarti.Kilo + PartiStokTarti.Dara, 0)) UretimdekiTartiBrutKilo,
             PartiStok.Durumu
      from PartiStok
      left join PartiStokTarti on PartiStokTarti.PartiSinifId = PartiStok.PartiSinifId
      left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      where :FirmaKodumuz = Parti.FirmaId and
      PartiStok.KaliteId = :KaliteId and
      (PartiStok.HamId = :AStokHamId or :AStokHamId is null)
      group by PartiStok.PartiSinifId, PartiStok.Durumu, PartiStok.NetKilo, PartiStok.Kilo
    )
    into
    :SevkeHazirKilo, :SevkeHazirBrutKilo, :SevkeHazirTartiBrutKilo, :SevkeHazirPartiSayisi,
    :UretimdekiKilo, :UretimdekiBrutKilo, :UretimdekiTartiBrutKilo, :UretimdekiPartiSayisi,
    :Kalite2SevkeHazirKilo, :Kalite2SevkeHazirBrutKilo, :Kalite2SevkeHazirTartiBrutKilo;

    SiparisSayisi = 0;
    SiparisKilo = 0;
    SiparisBrutKilo = 0;
    PartilenenKilo = 0;
    RezervEdilecekKilo = 0;
    RezervEdilecekBrutKilo = 0;
    RezervKilo = 0;
    RezervBrutKilo = 0;
    RezervTartiBrutKilo = 0;
    RezSevkKilo = 0;
    RezSevkBrutKilo = 0;
    RezSevkTartiBrutKilo = 0;
    BrutSevkEdilenKilo = 0;
    NetSevkEdilenKilo = 0;

    for select BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisBrutKilo, iif(:PartilenenMiktardanIptaliDus = 1, BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo, BoyaSiparisDetay.PartilenenKilo)
    from BoyaSiparisDetay where BoyaSiparisDetay.Kapali = 0 and BoyaSiparisDetay.KaliteId = :KaliteId and
         (BoyaSiparisDetay.HamId = :AStokHamId or :AStokHamId is null) and
         (BoyaSiparisDetay.TalimatKodu = :ATalimatKodu or :ATalimatKodu is Null)
    into :FBoyaSiparisDetayId, :FSiparisKilo, :FSiparisBrutKilo, :FPartilenenKilo do
    begin
      SiparisSayisi = SiparisSayisi + 1;
      SiparisKilo = SiparisKilo + FSiparisKilo;
      SiparisBrutKilo = SiparisBrutKilo + FSiparisBrutKilo;
      PartilenenKilo = PartilenenKilo + FPartilenenKilo;

      Select Sum(PartiStok.BrutSevkKilo), Sum(PartiStok.NetSevkKilo) from BoyaSiparisDetay
      left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
      where BoyaSiparisDetay.BoyaSiparisDetayId = :FBoyaSiparisDetayId
      into :FBrutSevkEdilenKilo, :FNetSevkEdilenKilo;

      BrutSevkEdilenKilo = BrutSevkEdilenKilo + FBrutSevkEdilenKilo;
      NetSevkEdilenKilo = NetSevkEdilenKilo + FNetSevkEdilenKilo;

      for select BoyaRezervFisiDetay.EdilecekKilo, BoyaRezervFisiDetay.EdilecekBrutKilo,
             Sum(iif(PartiStokTarti.BoyaRezervFisiDetayId is not null, PartiStokTarti.Kilo, 0)) ,
             Sum(iif(PartiStokTarti.BoyaRezervFisiDetayId is not null, PartiStokTarti.BrutKilo, 0)) ,
             Sum(iif(PartiStokTarti.BoyaRezervFisiDetayId is not null, PartiStokTarti.Kilo + PartiStokTarti.Dara, 0)) ,
             Sum(iif(PartiStokTarti.BoyaRezervFisiDetayId is not null and PartiStokTarti.BoyaIrsaliyeDetayId is not null, PartiStokTarti.Kilo, 0)) ,
             Sum(iif(PartiStokTarti.BoyaRezervFisiDetayId is not null and PartiStokTarti.BoyaIrsaliyeDetayId is not null, PartiStokTarti.BrutKilo, 0)) ,
             Sum(iif(PartiStokTarti.BoyaRezervFisiDetayId is not null and PartiStokTarti.BoyaIrsaliyeDetayId is not null, PartiStokTarti.Kilo + PartiStokTarti.Dara, 0))
      from BoyaRezervFisiDetay
      left join PartiStokTarti on PartiStokTarti.BoyaRezervFisiDetayId = BoyaRezervFisiDetay.BoyaRezervFisiDetayId
      where BoyaRezervFisiDetay.BoyaSiparisDetayId = :FBoyaSiparisDetayId
      group by BoyaRezervFisiDetay.BoyaRezervFisiDetayId, BoyaRezervFisiDetay.EdilecekKilo, BoyaRezervFisiDetay.EdilecekBrutKilo
      into :FRezervEdilecekKilo, :FRezervEdilecekBrutKilo, :FRezervKilo, :FRezervBrutKilo, :FRezervTartiBrutKilo, :FRezSevkKilo, :FRezSevkBrutKilo, :FRezSevkTartiBrutKilo do
      begin
        RezervEdilecekKilo = RezervEdilecekKilo + FRezervEdilecekKilo;
        RezervEdilecekBrutKilo = RezervEdilecekBrutKilo + FRezervEdilecekBrutKilo;
        RezervKilo = RezervKilo + FRezervKilo;
        RezervBrutKilo = RezervBrutKilo + FRezervBrutKilo;
        RezervTartiBrutKilo = RezervTartiBrutKilo + FRezervTartiBrutKilo;
        RezSevkKilo = RezSevkKilo + FRezSevkKilo;
        RezSevkBrutKilo = RezSevkBrutKilo + FRezSevkBrutKilo;
        RezSevkTartiBrutKilo = RezSevkTartiBrutKilo + FRezSevkTartiBrutKilo;
      end
    end

    suspend;
  end
end
