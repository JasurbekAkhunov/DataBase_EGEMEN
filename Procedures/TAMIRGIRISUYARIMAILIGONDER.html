CREATE PROCEDURE TAMIRGIRISUYARIMAILIGONDER(
    AFIRMAID VARCHAR(20),
    AIRSALIYETARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AHAMID INTEGER,
    ARENK VARCHAR(50),
    AMIKTAR DOUBLE PRECISION,
    AGIRISYERI VARCHAR(50))
AS
declare variable FirmaAdi          varchar(100);
declare variable HamAdi            varchar(255);
declare variable Aciklama          varchar(1024);
declare variable Konu              varchar(50);
begin
    select Adi from Firma where FirmaId = :AFirmaId into :FirmaAdi;
    select Adi from Ham where HamId = :AHamId into :HamAdi;
    Konu = 'Tamir Bilgilendirme İletisi';
    Aciklama = left(Cast(AIrsaliyeNo || ' Nolu ' || TimeStampToDateTimeStr(AIrsaliyeTarihi) || ' tarihli irsaliye ile tamir kaydı girildi.' || Chr(13) as Blob character set UTF8), 1024);
    if (FirmaAdi is Not Null) then Aciklama = LEFT(Aciklama || Cast(Chr(13) || 'Firma  : (' || AFirmaId || ') ' || FirmaAdi as Blob character set UTF8), 1024);
    if (AMiktar is Not Null) then Aciklama = LEFT(Aciklama || Cast(Chr(13) || 'Miktar : ' || Format_Float('', AMiktar) as Blob character set UTF8), 1024);
    if (HamAdi is Not Null) then Aciklama = LEFT(Aciklama || CAst(Chr(13) || 'Ürün Adı : ' || HamAdi as Blob character set UTF8), 1024);
    if (ARenk is Not Null) then Aciklama = LEFT(Aciklama || Cast(' ' as Blob character set UTF8) || ARenk, 1024);
    if (AGirisYeri is Not Null) then Aciklama = LEFT(Aciklama || Cast(Chr(13) || 'Giriş Yeri : ' as Blob character set UTF8) || AGirisYeri, 1024);
    insert into GelenMail(KullaniciId, KimdenKullaniciId, KimdenKullaniciAdi, Konu, Mesaj)
      Select Kullanici.KullaniciId, 1, 'Tamir Uyarısı', :Konu, :Aciklama from Kullanici where Kullanici.TamirGirislerindeMailUyari = 1;
end
