CREATE PROCEDURE RECETEURUNLISTESI(
    ARECETEID VARCHAR(20))
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(60))
AS
begin
  For Select ReceteDetay.UrunId, Urun.Adi from ReceteDetay
  left join Urun on Urun.UrunID = ReceteDetay.UrunId
  where ReceteDetay.ReceteID = :AReceteId and ReceteDetay.UrunID is not null
  group by ReceteDetay.UrunId, Urun.Adi
  into :UrunId, :UrunAdi do suspend;
end
