CREATE PROCEDURE ORMEHAMRAFLISTESIGETIR(
    AHAMID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    AORMESIPARISDETAYID INTEGER,
    ABOYASIPARISDETAYID INTEGER,
    ABOYASIPARISBASKIID INTEGER,
    ASAKATSTOKHAREKETID INTEGER,
    ADEPOID VARCHAR(10))
RETURNS (
    RESULTID VARCHAR(50),
    RAFID VARCHAR(10),
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    KAPADEDI INTEGER,
    RAFMIKTARI DOUBLE PRECISION,
    GIRISMIKTARI DOUBLE PRECISION,
    GIRISKAPADEDI INTEGER,
    CIKISMIKTARI DOUBLE PRECISION,
    CIKISKAPADEDI INTEGER,
    URUNRAFID INTEGER,
    ORMERAFDETAYID INTEGER,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    URETIMPARTINO VARCHAR(30),
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    ORMECIMAKINANO VARCHAR(10),
    FASONCUFIRMAID VARCHAR(20),
    FASONCUFIRMAADI VARCHAR(100),
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAADI VARCHAR(100),
    MUSTERISIPARISNO VARCHAR(50),
    SIPARISPUS INTEGER,
    SIPARISFEINE INTEGER,
    DINLENMISGRAMAJ INTEGER,
    GERCEKLESENGRAMAJ INTEGER,
    ARTIKELNO VARCHAR(20),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    ENDENCEKMEZLIKISTENEN VARCHAR(10),
    BOYDANCEKMEZLIKISTENEN VARCHAR(10),
    FBOYASIPARISKAYITID INTEGER,
    FORMESIPARISDETAYID INTEGER,
    FBOYASIPARISDETAYID INTEGER,
    FBOYASIPARISBASKIID INTEGER,
    FURETIMPARTINO VARCHAR(30),
    FLOT VARCHAR(150),
    FMAKINAID VARCHAR(10),
    FFIRMAID VARCHAR(20),
    FORMERAFDETAYRAFID VARCHAR(10),
    FORMECIMAKINANO VARCHAR(10),
    AYARTOPU INTEGER,
    STOKGIRISNO INTEGER,
    MELANJKODU VARCHAR(20),
    BIRIMMALIYET DOUBLE PRECISION)
AS
declare variable FBoyaSiparisYil   integer;
declare variable FBoyaSiparisId    integer;
declare variable FIptal            integer;
declare variable FDIptal           integer;
declare variable FBaskaSipariseSatilabilir   integer;
declare variable Okey              integer;
declare variable FSiparisFirmaId   varchar(20);
declare variable KayitSayisi       integer;
declare variable astIplikBoyaliKumas integer;
declare variable FOrmeSiparisTipiId integer;
declare variable IpligiBoyalidaDetaysizStokDus integer;

begin
  FBaskaSipariseSatilabilir = 1; /*Boyadaki ormede boyle bir soru yok, koyarsak olacak, simdilik hepsine izin verelim */

  Select astIplikBoyaliKumas from OrmeSiparisTipiGetir into :astIplikBoyaliKumas;
  Select IpligiBoyalidaDetaysizStokDus from Sistem where SistemId = 1 into :IpligiBoyalidaDetaysizStokDus;

  if (ASakatStokHareketId is null) then
  begin
    For Select OrmeRaf.OrmeRafId, OrmeRaf.Sakat, OrmeRaf.OrmeMakinaId, OrmeMakina.Adi, OrmeRaf.OrmeciMakinaNo, OrmeRaf.StokFirmaId, FFirma.Adi, OrmeRaf.UretimPartiNo, OrmeRaf.Lot, OrmeRaf.RafId, (OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari),
               OrmeRaf.NetGirisMiktari, OrmeRaf.NetCikisMiktari, OrmeRaf.GirisKapAdedi, OrmeRaf.CikisKapAdedi,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               OrmeSiparisDetay.MusteriSiparisNo, SFirma.Adi,
               OrmeRaf.BoyaSiparisKayitId, OrmeRaf.OrmeSiparisDetayId, OrmeRaf.BoyaSiparisDetayId, OrmeRaf.BoyaSiparisBaskiId,
               OrmeSiparisDetay.Iptal, OrmeSiparisDetay.Iptal,
               OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmeSiparisDetay.DinlenmisGramaj, OrmeSiparisDetay.GerceklesenGramaj, OrmeSiparisDetay.ArtikelNo,
               BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen,
               OrmeRaf.RenkId, Renk.Adi, OrmeRaf.AyarTopu, BoyaSiparis.OrmeSiparisTipiId, OrmeRaf.MelanjKodu, OrmeRaf.BirimMaliyet from OrmeRaf
      left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmeRaf.OrmeMakinaId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = OrmeRaf.BoyaSiparisDetayId
      left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisBaskiId = OrmeRaf.BoyaSiparisBaskiId
      left join Firma FFirma on FFirma.FirmaId = OrmeRaf.StokFirmaId
      left join Firma SFirma on SFirma.FirmaId = BoyaSiparis.FirmaId
      left join Renk on Renk.RenkId = OrmeRaf.RenkId
      Where OrmeRaf.HamId = :AHamId and
            (OrmeRaf.NetGirisMiktari > 0 or OrmeRaf.NetCikisMiktari > 0)
      ORDER BY OrmeRaf.OrmeSiparisDetayId Desc, OrmeRaf.BoyaSiparisDetayId Desc, OrmeRaf.BoyaSiparisKayitID Desc, RAFID Asc
      into :UrunRafId, :Sakat, :MakinaId, :MakinaAdi, :OrmeciMakinaNo, :FasoncuFirmaId, :FasoncuFirmaAdi, :UretimPartiNo, :Lot, :RafId, :RafMiktari, :GirisMiktari, :CikisMiktari,
           :GirisKapAdedi, :CikisKapAdedi, :FBoyaSiparisYil, :FBoyaSiparisId, :MusteriSiparisNo, :SiparisFirmaAdi,
           :FBoyaSiparisKayitId, :FOrmeSiparisDetayId, :FBoyaSiparisDetayId, :FBoyaSiparisBaskiId,
           :FIptal, :FDIptal,
           :SiparisPus, :SiparisFeine, :DinlenmisGramaj, :GerceklesenGramaj, :ArtikelNo,
           :EnIstenen, :GramajIstenen, :EndenCekmezlikIstenen, :BoydanCekmezlikIstenen, :RenkId, :RenkAdi, :AyarTopu, :FOrmeSiparisTipiId, :MelanjKodu, :BirimMaliyet do
    begin
      if (FBoyaSiparisKayitId is null) then
      begin
        FIptal = 0;
        FDIptal = 0;
        FBaskaSipariseSatilabilir = 0;
      end

      if (FIptal = 1 or FBaskaSipariseSatilabilir = 1) then
      begin
        FBoyaSiparisKayitId = Null;
        FOrmeSiparisDetayId = Null;
        FBoyaSiparisDetayId = Null;
        FBoyaSiparisBaskiId = Null;
      end
      else if (FDIptal = 1) then
      begin
        FOrmeSiparisDetayId = Null;
        FBoyaSiparisDetayId = Null;
        FBoyaSiparisBaskiId = Null;
      end

      Okey = 0;

      if ((FBoyaSiparisKayitId is null or (FBoyaSiparisKayitId = ABoyaSiparisKayitID)) and
          (FOrmeSiparisDetayId is null or FOrmeSiparisDetayId = AOrmeSiparisDETAYID or (FOrmeSiparisTipiId = astIplikBoyaliKumas and IpligiBoyalidaDetaysizStokDus = 1)) and
          (FBoyaSiparisBaskiId is null or FBoyaSiparisBaskiId = ABoyaSiparisBaskiID) and
          (FBoyaSiparisDetayId is null or FBoyaSiparisDetayId = ABoyaSiparisDetayID) ) then Okey = 1;

      BoyaSiparisIdStr = '';
      KapAdedi = GirisKapAdedi - CikisKapAdedi;

      if (FBoyaSiparisYil is not null) then
      begin
        BoyaSiparisIdStr = '(';
        BoyaSiparisIdStr = BoyaSiparisIdStr || Format_Float('#0', FBoyaSiparisYil);
        BoyaSiparisIdStr = BoyaSiparisIdStr || ') ';
        BoyaSiparisIdStr = BoyaSiparisIdStr || FBoyaSiparisId;
        BoyaSiparisIdStr = BoyaSiparisIdStr || ' ';
      end

      SakatStr = null;
      if (Sakat = 1) then SakatStr = 'Sakat';
      if (Okey = 1) then suspend;
    end
  end
  else
  begin
    FSiparisFirmaId = Null;
    if (ABoyaSiparisKayitId is not null) then
      Select FirmaId from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitID  into :FSiparisFirmaId;

    Select UretimPartiNo, Lot, OrmeMakinaId, OrmeciMakinaNo, StokFirmaId, BoyaSiparisKayitID, OrmeSiparisDetayId, BoyaSiparisDetayId, BoyaSiparisBaskiId, StokGirisNo
     from StokHareket
           where StokHareket.StokHareketId = :ASakatStokHareketId
           into :FUretimPartiNo, :FLot, :FMakinaId, :FOrmeciMakinaNo, :FFirmaId, :FBoyaSiparisKayitID, :FOrmeSiparisDetayId, :FBoyaSiparisDetayId, :FBoyaSiparisBaskiId, :StokGirisNo;

    if (FSiparisFirmaId is null) then
    begin
      FBoyaSiparisKayitId    = null;
      FOrmeSiparisDetayId = Null;
      FBoyaSiparisDetayId = Null;
      FBoyaSiparisBaskiId = Null;
    end

    For select OrmeRafDetay.OrmeRafDetayId, OrmeRafDetay.ormerafid from OrmeRafDetay where OrmeRafDetay.StokHareketId = :ASakatStokHareketId into :OrmeRafDetayId, :FOrmeRafDetayRafId do

    For Select OrmeRaf.OrmeRafId, OrmeRaf.OrmeMakinaId, OrmeMakina.Adi, OrmeRaf.OrmeciMakinaNo, OrmeRaf.UretimPartiNo, OrmeRaf.Lot, OrmeRaf.RafId, (OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari),
               OrmeRaf.NetGirisMiktari, OrmeRaf.NetCikisMiktari, OrmeRaf.GirisKapAdedi, OrmeRaf.CikisKapAdedi,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               OrmeSiparisDetay.MusteriSiparisNo, Firma.Adi,
               OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmeSiparisDetay.DinlenmisGramaj, OrmeSiparisDetay.GerceklesenGramaj, OrmeSiparisDetay.ArtikelNo,
               OrmeRaf.RenkId, Renk.Adi, OrmeRaf.AyarTopu, OrmeRaf.MelanjKodu, OrmeRaf.BirimMaliyet from OrmeRaf
      left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmeRaf.OrmeMakinaId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
      left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
      left join Renk on Renk.RenkId = OrmeRaf.RenkId
      Where OrmeRaf.HamId = :AHamId and
            OrmeRaf.ormerafid = :FOrmeRafDetayRafId and OrmeRaf.Sakat = 0 and
            (OrmeRaf.StokGirisNo = :StokGirisNo or :StokGirisNo is null ) and
            (OrmeRaf.BoyaSiparisKayitId = :FBoyaSiparisKayitId or (OrmeRaf.BoyaSiparisKayitId is null and :FBoyaSiparisKayitId is null)) and
            (OrmeRaf.OrmeSiparisDetayId = :FOrmeSiparisDetayId or (OrmeRaf.OrmeSiparisDetayId is null and :FOrmeSiparisDetayId is null)) and
            (OrmeRaf.BoyaSiparisDetayId = :FBoyaSiparisDetayId or (OrmeRaf.BoyaSiparisDetayId is null and :FBoyaSiparisDetayId is null)) and
            (OrmeRaf.BoyaSiparisBaskiId = :FBoyaSiparisBaskiId or (OrmeRaf.BoyaSiparisBaskiId is null and :FBoyaSiparisBaskiId is null)) and
            (OrmeRaf.StokFirmaId = :FFirmaId or (OrmeRaf.StokFirmaId is null and :FFirmaId is null)) and
            (OrmeRaf.OrmeMakinaId = :FMakinaId or (OrmeRaf.OrmeMakinaId is null and :FMakinaId is null)) and
            (StrKarsilastir(OrmeRaf.OrmeciMakinaNo, :FOrmeciMakinaNo) = 1) and
            (OrmeRaf.UretimPartiNo = :FUretimPartiNo or (OrmeRaf.UretimPartiNo is null and :FUretimPartiNo is null)) and
            (OrmeRaf.Lot = :FLot or (OrmeRaf.Lot is null and :FLot is null))
      ORDER BY OrmeRaf.OrmeSiparisDetayId Desc, OrmeRaf.BoyaSiparisDetayId Desc, OrmeRaf.BoyaSiparisKayitID Desc, RAFID Asc
      into :UrunRafId, :MakinaId, :MakinaAdi, :OrmeciMakinaNo, :UretimPartiNo, :Lot ,:RafId, :RafMiktari, :GirisMiktari, :CikisMiktari,
           :GirisKapAdedi, :CikisKapAdedi,
           :FBoyaSiparisYil, :FBoyaSiparisId, :MusteriSiparisNo, :SiparisFirmaAdi,
           :SiparisPus, :SiparisFeine, :DinlenmisGramaj, :GerceklesenGramaj, :ArtikelNo, :RenkId, :RenkAdi, :AyarTopu, :MelanjKodu, :BirimMaliyet do
    begin
      BoyaSiparisIdStr = '';
      KapAdedi = GirisKapAdedi - CikisKapAdedi;

      if (FBoyaSiparisYil is not null) then
      begin
        BoyaSiparisIdStr = '(';
        BoyaSiparisIdStr = BoyaSiparisIdStr || Format_Float('#0', FBoyaSiparisYil);
        BoyaSiparisIdStr = BoyaSiparisIdStr || ') ';
        BoyaSiparisIdStr = BoyaSiparisIdStr || FBoyaSiparisId;
        BoyaSiparisIdStr = BoyaSiparisIdStr || ' ';
      end
      ResultId = EgemenIntegerToStr(OrmeRafDetayId);
      ResultId = ResultId || ',' || EgemenIntegerToStr(UrunRafId);

      KayitSayisi = 1;

      if (ADepoId is not Null) then
      begin
        select count(*) from DepoRaf where DepoId = :ADepoID and RafId = :RafId into :KayitSayisi;
        if (KayitSayisi is Null) then KayitSayisi = 0;
      end
      if (KayitSayisi > 0) then suspend;
    end
  end
end
