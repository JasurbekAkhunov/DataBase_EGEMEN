CREATE PROCEDURE PKARTIICINSONVARDIYAIDGETIR(
    AORMEMAKINAID VARCHAR(10),
    ADOKUMAMAKINAID VARCHAR(10),
    PERSONELKOKUTULDU EGEBOOLEAN)
RETURNS (
    VARDIYAID INTEGER,
    VARDIYANO INTEGER,
    VARDIYATARIHI TIMESTAMP,
    VARDIYASAATI TIMESTAMP)
AS
declare variable T1                     Double Precision;
declare variable T2                     Double Precision;
declare variable T3                     Double Precision;
declare variable VardiyaBaslangicTarihi TimeStamp;
declare variable VardiyaBaslangicSaati  Double Precision;
declare variable OrmeVardiyaSayisi          integer;
declare variable SuankiSaat             Double Precision;
declare variable FNow                   TimeStamp;
declare variable ToleransT              Double Precision;
declare variable FVardiya1              integer;
declare variable FVardiya2              integer;
declare variable FVardiya3              integer;
declare variable EsitOlmayanVardiyaKullan integer;
declare variable Personel1Id            integer;
declare variable OrmeDefaultUniteId     integer;
declare variable OrmeUnitesimi          integer;
declare variable DokumaUnitesimi        integer;
declare variable KS                     integer;
begin
  FNow =  current_timestamp(0);
  SuankiSaat = TimeStampToDouble(TimeStampToTime(FNow));

  Select VardiyaBaslangicSaati, OrmeVardiyaSayisi, PersonelVardiyaToleransDakika, EsitOlmayanVardiyaKullan from Sistem
    into :VardiyaBaslangicTarihi, :OrmeVardiyaSayisi, :ToleransT, :EsitOlmayanVardiyaKullan;

  if (ToleransT is null) then ToleransT = 30;
  if (PersonelKOkutuldu = 1) then ToleransT = 0;

  VardiyaBaslangicSaati = TimeStampToDouble(TimeStampToTime(VardiyaBaslangicTarihi));

  if (TimeStampToDouble(TimeStampToTime(FNow)) + ((1.0000/24.0000/60.0000) * ToleransT) < VardiyaBaslangicSaati) then VardiyaTarihi = cast(Fnow as Date) - 1;
  else VardiyaTarihi = cast(Fnow as Date);


  if (EsitOlmayanVardiyaKullan is Null) then EsitOlmayanVardiyaKullan = 0;

  FVardiya1 = 0;
  FVardiya2 = 0;
  FVardiya3 = 0;

  if (EsitOlmayanVardiyaKullan <> 1) then
  begin
    FVardiya1 = VardiyaHesapla(SuankiSaat - VardiyaBaslangicSaati,3);
    FVardiya2 = VardiyaHesapla(SuankiSaat - ((1.0000/24.0000/60.0000) * ToleransT) - VardiyaBaslangicSaati, OrmeVardiyaSayisi);
    FVardiya3 = VardiyaHesapla(SuankiSaat + ((1.0000/24.0000/60.0000) * ToleransT) - VardiyaBaslangicSaati, OrmeVardiyaSayisi);
  end
  else
  begin
    select VardiyaNo, VardiyaSaati from OrmeVardiyaHesapla(current_timestamp(0)) into :FVardiya1, :VardiyaSaati;
	if (TimeStampToDouble(TimeStampToTime(FNow)) + ((1.0000/24.0000/60.0000) * ToleransT) < VardiyaBaslangicSaati) then VardiyaSaati = VardiyaSaati - 1;
    if (PersonelKOkutuldu = 1) then
      select VardiyaNo from OrmeVardiyaHesapla(current_timestamp(0) - ((1.0000/24.0000/60.0000) * :ToleransT)) into :FVardiya2;
    select VardiyaNo from OrmeVardiyaHesapla(current_timestamp(0) + ((1.0000/24.0000/60.0000) * :ToleransT)) into :FVardiya3;
  end

  if (FVardiya1 is Null) then FVardiya1 = 0;
  if (FVardiya2 is Null) then FVardiya2 = 0;
  if (FVardiya3 is Null) then FVardiya3 = 0;

  VardiyaNo = FVardiya1;

  if ((FVardiya1 <> FVardiya2) and (FVardiya1 = FVardiya3) and (FVardiya2 <> 0)) then
    VardiyaNo = FVardiya2;

  if ((FVardiya1 <> FVardiya3) and (FVardiya3 <> 0)) then
    VardiyaNo = FVardiya3;

  if (AOrmeMakinaId is not null) then
  begin
    For Select OrmeVardiyaPersonel.OrmeVardiyaId from OrmeVardiya
      left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = OrmeVardiya.OrmeVardiyaId and OrmeVardiyaPersonel.OrmeMakinaID = :AOrmeMakinaId
      where OrmeVardiya.Tarih = :VardiyaTarihi and OrmeVardiya.VardiyaNo = :VardiyaNo and OrmeVardiyaPersonel.OrmeMakinaId = :AOrmeMakinaId
      order by tarih desc, vardiyano desc
      into :VardiyaId do
    begin
      suspend;
      exit;
    end
  end
  else
  begin
    For Select OrmeVardiya.OrmeVardiyaId from OrmeVardiya
      where OrmeVardiya.Tarih = :VardiyaTarihi and OrmeVardiya.VardiyaNo = :VardiyaNo and OrmeVardiya.DokumaVardiyasi = 1
      order by tarih desc, vardiyano desc
      into :VardiyaId do
    begin
      suspend;
      exit;
    end
  end

  if (VardiyaId is Null) then
  begin
    select OrmeDefaultUniteId, Unite.OrmeUnitesimi, Unite.DokumaUnitesimi from Sistem
      Left Join Unite on Unite.UniteID = Sistem.OrmeDefaultUniteId
    into :OrmeDefaultUniteId, :OrmeUnitesimi, :DokumaUnitesimi;

    if (OrmeDefaultUniteId is not Null) then
    begin
      select PersonelId from KULLANICI where KULLANICIID = 1 into :Personel1Id;

      if ((Personel1Id is not null) and (Personel1Id > 0)) then
      begin
        select count(*) from OrmeVardiya where OrmeVardiya.Tarih = :VardiyaTarihi and OrmeVardiya.VardiyaNo = :VardiyaNo and OrmeVardiya.UniteId = :OrmeDefaultUniteId
        into :KS;

        if (KS = 0) then
          insert into OrmeVardiya (Tarih, VardiyaNo, VardiyaAmiriPersonelId, UniteId, Aktif, OrmeVardiyasi, DokumaVardiyasi, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
          values(:VardiyaTarihi, :VardiyaNo, :Personel1Id, :OrmeDefaultUniteId, 1, :OrmeUnitesimi, :DokumaUnitesimi, 1, current_timestamp(0), 1, current_timestamp(0));

        if (AOrmeMakinaId is not null) then
        begin
          Select OrmeVardiyaPersonel.OrmeVardiyaId from OrmeVardiya
            left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = OrmeVardiya.OrmeVardiyaId and OrmeVardiyaPersonel.OrmeMakinaID = :AOrmeMakinaId
            where OrmeVardiya.Tarih = :VardiyaTarihi and OrmeVardiya.VardiyaNo = :VardiyaNo and OrmeVardiyaPersonel.OrmeMakinaId = :AOrmeMakinaId
            order by tarih desc, vardiyano desc
            into :VardiyaId;
        end
        else
        begin
          Select OrmeVardiyaPersonel.OrmeVardiyaId from OrmeVardiya
            left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = OrmeVardiya.OrmeVardiyaId and OrmeVardiyaPersonel.DokumaMakinaID = :ADokumaMakinaId
            where OrmeVardiya.Tarih = :VardiyaTarihi and OrmeVardiya.VardiyaNo = :VardiyaNo and OrmeVardiyaPersonel.DokumaMakinaId = :ADokumaMakinaId
            order by tarih desc, vardiyano desc
            into :VardiyaId;
        end

        suspend;
      end
    end
  end
end
