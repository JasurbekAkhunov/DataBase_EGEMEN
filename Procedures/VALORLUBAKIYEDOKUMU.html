CREATE PROCEDURE VALORLUBAKIYEDOKUMU(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    ASINIF VARCHAR(1),
    AFAIZORANI DOUBLE PRECISION,
    AKULLANICIID INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    DOVIZID VARCHAR(5),
    BORCTOPLAM DOUBLE PRECISION,
    ALACAKTOPLAM DOUBLE PRECISION,
    CARIBAKIYE DOUBLE PRECISION,
    CARIBAKIYEBA VARCHAR(1),
    VALORBAKIYE DOUBLE PRECISION,
    VALORBAKIYEBA VARCHAR(1),
    FARKGUN INTEGER,
    TOPLAMBAKIYE DOUBLE PRECISION,
    TOPLAMBAKIYEBA VARCHAR(1))
AS
declare variable FirmaGrup1 varchar(4);
declare variable FirmaGrup2 varchar(4);
declare variable FirmaGrup3 varchar(4);
declare variable Okey integer;
declare variable Personelmi     integer;
declare variable FBorcAlacak    varchar(1);
declare variable FMeblag        Double Precision;

declare variable TFarkGun        Double Precision;
declare variable TAlacak         Double Precision;
declare variable TGFarkGun       Double Precision;
declare variable TGAlacak        Double Precision;

declare variable TB1        Double Precision;
declare variable TB2        Double Precision;

declare variable PersonelCariYetkisiVar integer;
begin
  select PersonelCariYetkisiVar from Kullanici where KullaniciID = :AKullaniciId into :PersonelCariYetkisiVar;

  for Select FirmaId, Adi, Adi2, DovizId, FirmaGrup1, FirmaGrup2, FirmaGrup3, Personelmi  from Firma
  where ((:AFirmaAralikli = 1 and FirmaId >= :ABaslangicFirmaId and (FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0))  and
        ((:AFirmaGrup1Aralikli = 1 and FirmaGrup1 >= :ABaslangicGrup1Id  and (FirmaGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and FirmaGrup2 >= :ABaslangicGrup2Id and (FirmaGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and FirmaGrup3 >= :ABaslangicGrup3Id and (FirmaGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = ''))
  into :FirmaId, :FirmaAdi, :FirmaAdi2, :DovizId, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :Personelmi do
  begin
    Okey = 1;
    if (Personelmi = 1 and PersonelCariYetkisiVar = 0) then Okey = 0;

    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (okey = 1) then
    begin
      BorcToplam = 0;
      AlacakToplam = 0;
      CariBakiye = 0;
      CariBakiyeBA = '';
      ValorBakiye = 0;
      ValorBakiyeBA = '';

      for select  BorcAlacak, sum( Meblag / DovizKuru1 * DovizKuru2)  from cari
           where Cari.FirmaId = :FirmaId and Tarih >= :ABaslangicTarihi and Tarih <= :ABitisTarihi and (Cari.Sinif = :ASinif or :ASinif is null)
           group by Cari.BorcAlacak
           into :FBorcAlacak, :FMeblag do
      begin
        if (FBorcAlacak = 'B') then BorcToplam = FMeblag;
        else if (FBorcAlacak = 'A') then AlacakToplam = FMeblag;
      end

      if (BorcToplam > AlacakToplam) then
      begin
        CariBakiye = BorcToplam - AlacakToplam;
        CariBakiyeBA = 'B';
      end
      else if (BorcToplam < AlacakToplam) then
      begin
        CariBakiye = AlacakToplam - BorcToplam;
        CariBakiyeBA = 'A';
      end

      if (BorcToplam > 0 or AlacakToplam > 0) then
      begin
        TGFarkGun  = 0;
        TGAlacak  = 0;
        for Select BorcAlacak, sum(alacak * farkgun), sum(alacak), Sum(VadeFarki) from ValorluCariKartDokumu(:FirmaId, :FirmaId, '', '', '', '', '', '', :ABaslangicTarihi, :ABitisTarihi, 1,0,0,0, :ASinif, :AFaizOrani, :AKullaniciId)
           where Tarih is not null and Alacak is not null and farkgun <> 0
           group by BorcAlacak
           into :FBorcAlacak, :TFarkGun, :TAlacak, :FMeblag do
        begin
          TGFarkGun = TGFarkGun + TFarkGun;
          TGAlacak  = TGAlacak + TAlacak;

          if (FBorcAlacak = 'B') then ValorBakiye = ValorBakiye + FMeblag;
          else if (FBorcAlacak = 'A') then ValorBakiye = ValorBakiye - FMeblag;
        end

        if (ValorBakiye > 0) then ValorBakiyeBA = 'B';
        else
        begin
          ValorBakiyeBA = 'A';
          ValorBakiye = ValorBakiye * -1;
        end
        if (TGAlacak > 0) then FarkGun = TGFarkGun / TGAlacak;
        else FarkGun = Null;

        if (CariBakiyeBa = 'B') then TB1 = CariBakiye * -1;
        else TB1 = CariBakiye;

        if (ValorBakiyeBa = 'B') then TB2 = ValorBakiye * -1;
        else TB2 = ValorBakiye;

        ToplamBakiye = TB1 + TB2;

        if (ToplamBakiye < 0) then
        begin
          ToplamBakiyeBA = 'B';
          ToplamBakiye = ToplamBakiye * -1;
        end
        else ToplamBakiyeBA = 'A';

        suspend;

      end
    end
  end
end
