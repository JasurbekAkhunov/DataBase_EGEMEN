CREATE PROCEDURE BAKIMDURUMRAPORU(
    AORMEMAKINAGRUPID VARCHAR(4),
    AORMEMAKINAID VARCHAR(10))
RETURNS (
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    ORMEMAKINAGRUPID VARCHAR(4),
    ORMEMAKINAGRUPADI VARCHAR(30),
    IGNESIFIRLAMATARIHI TIMESTAMP,
    IGNESIFIRLAMASONRASIURETIM DOUBLE PRECISION,
    SONGENELBAKIMTARIHI TIMESTAMP,
    BAKIMSONRASIURETIM DOUBLE PRECISION)
AS
declare Variable MakinaGenelBakimTipiID   integer;
declare Variable MakinaIgneDegisimBakimTipiID integer;
declare variable ShtUretim integer;
declare variable uitOrme integer;
declare variable FBakimDurumRaporuSon6Ay EgeBoolean;
Declare Variable FIgneSifirlamaTarihi Timestamp;
Declare Variable FSonGenelBakimTarihi Timestamp;
begin
  Select MakinaGenelBakimTipiID, MakinaIgneDegisimBakimTipiID, BakimDurumRaporuSon6Ay from Sistem where SistemID = 1 into :MakinaGenelBakimTipiID, :MakinaIgneDegisimBakimTipiID, :FBakimDurumRaporuSon6Ay;
  select ShtUretim from OrmeStokHareketTipiGetir into :ShtUretim;
  select uitOrme from UretimIslemTipiGetir into :uitOrme;

  if (FBakimDurumRaporuSon6Ay = 0) then
  begin
    For Select OrmeMakina.OrmeMakinaID, OrmeMakina.Adi, OrmeMakina.MakinaGrupId, MakinaGrup.Adi, Max(IgneDegisimBakim.BakimEndTarihi), Max(GenelBakim.BakimEndTarihi) from OrmeMakina
          Left Join MakinaGrup on MakinaGrup.MakinaGrupID = OrmeMakina.MakinaGrupID
          Left Join MakinaBakimIsEmri GenelBakim on GenelBakim.OrmeMakinaID = OrmeMakina.OrmeMakinaID and GenelBakim.MakinaBakimTipiID = :MakinaGenelBakimTipiID and GenelBakim.BakimEndKullaniciID is not Null
          Left Join MakinaBakimIsEmri IgneDegisimBakim on IgneDegisimBakim.OrmeMakinaID = OrmeMakina.OrmeMakinaID and IgneDegisimBakim.MakinaBakimTipiID = :MakinaIgneDegisimBakimTipiID and IgneDegisimBakim.BakimEndKullaniciID is not Null
          Where (OrmeMakina.OrmeMakinaID = :AOrmeMakinaID or :AOrmeMakinaID is Null) and (OrmeMakina.MakinaGrupID = :AOrmeMakinaGrupID or :AOrmeMakinaGrupID is Null)
          Group By OrmeMakina.OrmeMakinaID, OrmeMakina.Adi, OrmeMakina.MakinaGrupId, MakinaGrup.Adi, OrmeMakina.IgneSifirlamaTarihi
        into :OrmeMakinaID, :OrmeMakinaAdi, :OrmeMakinaGrupId, :OrmeMakinaGrupAdi, :IgneSifirlamaTarihi, :SonGenelBakimTarihi do
    begin
      Select Sum(StokHareket.Miktar) from stokhareket
      Left Join OrmePlanlama on Ormeplanlama.OrmePlanlamaID = Stokhareket.OrmePlanlamaID
      where Stokhareket.OrmeMakinaID = :OrmeMakinaID and Stokhareket.Tarih >= :IgneSifirlamaTarihi and StokHareket.UretimIslemTipiId = :uitOrme and  StokHareket.HamIade=0 and
           (StokHareket.HareketTipi = :shtUretim)
      into :IgneSifirlamaSonrasiUretim;
  
      Select Sum(StokHareket.Miktar) from stokhareket
      Left Join OrmePlanlama on Ormeplanlama.OrmePlanlamaID = Stokhareket.OrmePlanlamaID
      where Stokhareket.OrmeMakinaID = :OrmeMakinaID and Stokhareket.Tarih >= :SonGenelBakimTarihi and StokHareket.UretimIslemTipiId = :uitOrme and  StokHareket.HamIade=0 and
           (StokHareket.HareketTipi = :shtUretim)
      into :BakimSonrasiUretim;
      Suspend;
    end
  end
  else
  begin
    FIgneSifirlamaTarihi = Null;
    FSonGenelBakimTarihi = Null;

    For Select OrmeMakina.OrmeMakinaID, OrmeMakina.Adi, OrmeMakina.MakinaGrupId, MakinaGrup.Adi, IgneDegisimBakim.BakimEndTarihi, GenelBakim.BakimEndTarihi from OrmeMakina
          Left Join MakinaGrup on MakinaGrup.MakinaGrupID = OrmeMakina.MakinaGrupID
          Left Join MakinaBakimIsEmri GenelBakim on GenelBakim.OrmeMakinaID = OrmeMakina.OrmeMakinaID and GenelBakim.MakinaBakimTipiID = :MakinaGenelBakimTipiID and GenelBakim.BakimEndKullaniciID is not Null and GenelBakim.BakimEndTarihi > Current_timestamp(0) - 180
          Left Join MakinaBakimIsEmri IgneDegisimBakim on IgneDegisimBakim.OrmeMakinaID = OrmeMakina.OrmeMakinaID and IgneDegisimBakim.MakinaBakimTipiID = :MakinaIgneDegisimBakimTipiID and IgneDegisimBakim.BakimEndKullaniciID is not Null and IgneDegisimBakim.BakimEndTarihi > Current_timestamp(0) - 180
          Where (OrmeMakina.OrmeMakinaID = :AOrmeMakinaID or :AOrmeMakinaID is Null) and (OrmeMakina.MakinaGrupID = :AOrmeMakinaGrupID or :AOrmeMakinaGrupID is Null)
        into :OrmeMakinaID, :OrmeMakinaAdi, :OrmeMakinaGrupId, :OrmeMakinaGrupAdi, :IgneSifirlamaTarihi, :SonGenelBakimTarihi do
    begin
      IgneSifirlamaSonrasiUretim = Null;
      BakimSonrasiUretim = Null;
      FIgneSifirlamaTarihi = Null;
      FSonGenelBakimTarihi = Null;

      Select First 1 BakimEndTarihi From MakinaBakimIsEmri
      Where OrmeMakinaID = :OrmeMakinaID and MakinaBakimTipiID = :MakinaIgneDegisimBakimTipiID and BakimEndKullaniciID is not Null and BakimEndTarihi > :IgneSifirlamaTarihi
      Order By BakimEndTarihi
      Into :FIgneSifirlamaTarihi;

      if (FIgneSifirlamaTarihi is null) then
        FIgneSifirlamaTarihi = Current_timestamp(0);

      if (IgneSifirlamaTarihi is not null and FIgneSifirlamaTarihi is not null) then
      begin
        Select Sum(StokHareket.Miktar) from StokHareket
        Left Join OrmePlanlama on Ormeplanlama.OrmePlanlamaID = Stokhareket.OrmePlanlamaID
        where StokHareket.OrmeMakinaID = :OrmeMakinaID and StokHareket.Tarih >= :IgneSifirlamaTarihi and  StokHareket.Tarih <= :FIgneSifirlamaTarihi and StokHareket.UretimIslemTipiId = :uitOrme and  StokHareket.HamIade=0 and
             (StokHareket.HareketTipi = :shtUretim)
        into :IgneSifirlamaSonrasiUretim;
      end

      Select First 1 BakimEndTarihi From MakinaBakimIsEmri
      Where OrmeMakinaID = :OrmeMakinaID and MakinaBakimTipiID = :MakinaGenelBakimTipiID and BakimEndKullaniciID is not Null and BakimEndTarihi > :SonGenelBakimTarihi
      Order By BakimEndTarihi
      Into :FSonGenelBakimTarihi;

      if (SonGenelBakimTarihi is not null and FSonGenelBakimTarihi is not null) then
      begin
        Select Sum(StokHareket.Miktar) from StokHareket
        Left Join OrmePlanlama on OrmePlanlama.OrmePlanlamaID = StokHareket.OrmePlanlamaID
        where Stokhareket.OrmeMakinaID = :OrmeMakinaID and Stokhareket.Tarih >= :SonGenelBakimTarihi and Stokhareket.Tarih <= :FSonGenelBakimTarihi and StokHareket.UretimIslemTipiId = :uitOrme and  StokHareket.HamIade=0 and
             (StokHareket.HareketTipi = :shtUretim)
        into :BakimSonrasiUretim;
      end

      Suspend;
    end
  end
end
