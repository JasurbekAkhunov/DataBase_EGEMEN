CREATE PROCEDURE MC_BOYASIPARISRAPORU_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AISLEMTIPIID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AMUSTERIRENKKODUVEADI VARCHAR(30),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISNO VARCHAR(20),
    SIPARISPROSESADI VARCHAR(255),
    ILAVEISLEMLER VARCHAR(255),
    OZELAPRELER VARCHAR(255),
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    IPTALKILO NUMERIC(12,3),
    IPTALMETRE NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    BRUTSEVKEDILENKILO NUMERIC(12,3),
    BRUTSEVKEDILENMETRE NUMERIC(12,3),
    NETSEVKEDILENKILO NUMERIC(12,3),
    NETSEVKEDILENMETRE NUMERIC(12,3),
    HAMADI VARCHAR(255),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    KAPALI INTEGER,
    KAPALISTR VARCHAR(6),
    KAPATMATARIHI TIMESTAMP)
AS
declare variable ABoyaSiparisKayitId integer;
declare variable Okey     integer;
declare variable PartilenenMiktardanIptaliDus EgeBoolean;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  if (ABoyaSiparisYil is not nUll) then
  begin
    Select BoyaSiparis.BoyaSiparisKayitId from BoyaSiparis
      where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId
      into :ABoyaSiparisKayitId;
    if (ABoyaSiparisKayitId is null) then ABoyaSiparisKayitId = -1;
  end

  select PartilenenMiktardanIptaliDus from Sistem where Sistem.SistemId = 1 into :PartilenenMiktardanIptaliDus;

  if (ABoyaSiparisKayitId is null) then
  begin
    if (AKapaliDurumu = 1) then
    begin
      for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo,
                 LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr, BoyaSiparisDetay.SiparisKilo,
                 BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.PartilenenKilo+BoyaSiparisDetay.PartilenenKiloBaski, BoyaSiparisDetay.IptalKilo+BoyaSiparisDetay.IptalKiloBaski, BoyaSiparisDetay.PartilenenMetre+BoyaSiparisDetay.PartilenenMetreBaski, BoyaSiparisDetay.IptalMetre+BoyaSiparisDetay.IptalMetreBaski,
                 Ham.Adi, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi, BoyaSiparisDetay.Kapali,
                 BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
                 SiparisProses.Adi, BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.OzelApreler from BoyaSiparis
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
      left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
      left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
      left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
      left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
        where  (BoyaSiparisDetay.KapatmaTarihi >= :ABaslangicTarihi and BoyaSiparisDetay.KapatmaTarihi <= :ABitisTarihi) and
               (BoyaSiparis.FirmaId = :AFirmaId or :AFirmaId is null) and
               (BoyaSiparis.Fason = :ASatisTipi or :ASatisTipi is null) and
               (BoyaSiparisDetay.IslemTipiId = :AIslemTipiId or :AIslemTipiId is null) and
               (BoyaSiparisDetay.SiparisTipiId = :ABoyaSiparisTipiId or :ABoyaSiparisTipiId is null) and
               (BoyaSiparisDetay.LabReceteId = :ALabReceteId or :ALabReceteId is null) and
               (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
               (BoyaSiparisDetay.MusteriRenkKoduveAdi = :AMusteriRenkKoduveAdi or :AMusteriRenkKoduveAdi is null) and
               (BoyaSiparisDetay.OnayDurumu = :AOnayDurumu or :AOnayDurumu is null) and
               ( ((BoyaSiparisDetay.PartilenenKilo + BoyaSiparisDetay.PartilenenKiloBaski - BoyaSiparisDetay.IptalKilo - BoyaSiparisDetay.IptalKiloBaski) < BoyaSiparisDetay.SiparisKilo and :APartilenen = 0) or
                 ((BoyaSiparisDetay.PartilenenKilo + BoyaSiparisDetay.PartilenenKiloBaski - BoyaSiparisDetay.IptalKilo - BoyaSiparisDetay.IptalKiloBaski) >= BoyaSiparisDetay.SiparisKilo and :APartilenen = 1) or :APartilenen is null ) and
               (BoyaSiparisDetay.Kapali = :AKapaliDurumu)
      order by BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId
      into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :MusteriSiparisNo, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisNo,
           :SiparisKilo, :SiparisMetre, :PartilenenKilo, :IptalKilo, :PartilenenMetre, :IptalMetre,
           :HamAdi, :TerminTarihi, :RevizeTerminTarihi, :Kapali, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
           :SiparisProsesAdi, :IlaveIslemler, :OzelApreler do
      begin
        if (Kapali = 1) then KapaliStr = 'Kapali';
        else if (Kapali = 2) then KapaliStr = 'İptal';
        else KapaliStr = '';

        KalanKilo = SiparisKilo - PartilenenKilo + IptalKilo;
        KalanMetre = SiparisMetre - PartilenenMetre + IptalMetre;

        Select Sum(BrutSevkKilo), Sum(BrutSevkMetre), Sum(NetSevkKilo), Sum(NetSevkMetre) from PartiStok where PartiStok.BoyaSiparisDetayId = :BoyaSiparisDetayId
          into :BrutSevkEdilenKilo, :BrutSevkEdilenMetre, :NetSevkEdilenKilo, :NetSevkEdilenMetre;

        if (PartilenenMiktardanIptaliDus = 1) then
        begin
          PartilenenKilo = PartilenenKilo - IptalKilo;
          PartilenenMetre = PartilenenMetre - IptalMetre;
        end
        suspend;
      end
    end
    else
    begin
      for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo,
                 LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr, BoyaSiparisDetay.SiparisKilo,
                 BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.PartilenenKilo+BoyaSiparisDetay.PartilenenKiloBaski, BoyaSiparisDetay.IptalKilo+BoyaSiparisDetay.IptalKiloBaski, BoyaSiparisDetay.PartilenenMetre+BoyaSiparisDetay.PartilenenMetreBaski, BoyaSiparisDetay.IptalMetre+BoyaSiparisDetay.IptalMetreBaski,
                 Ham.Adi, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi, BoyaSiparisDetay.Kapali,
                 BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
                 SiparisProses.Adi, BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.OzelApreler from BoyaSiparis
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
      left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
      left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
      left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
      left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
        where  (BoyaSiparis.SiparisTarihi >= :ABaslangicTarihi and BoyaSiparis.SiparisTarihi <= :ABitisTarihi) and
               (BoyaSiparis.FirmaId = :AFirmaId or :AFirmaId is null) and
               (BoyaSiparis.Fason = :ASatisTipi or :ASatisTipi is null) and
               (BoyaSiparisDetay.IslemTipiId = :AIslemTipiId or :AIslemTipiId is null) and
               (BoyaSiparisDetay.SiparisTipiId = :ABoyaSiparisTipiId or :ABoyaSiparisTipiId is null) and
               (BoyaSiparisDetay.LabReceteId = :ALabReceteId or :ALabReceteId is null) and
               (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
               (BoyaSiparisDetay.MusteriRenkKoduveAdi = :AMusteriRenkKoduveAdi or :AMusteriRenkKoduveAdi is null) and
               (BoyaSiparisDetay.OnayDurumu = :AOnayDurumu or :AOnayDurumu is null) and
               ( ((BoyaSiparisDetay.PartilenenKilo + BoyaSiparisDetay.PartilenenKiloBaski - BoyaSiparisDetay.IptalKilo - BoyaSiparisDetay.IptalKiloBaski) < BoyaSiparisDetay.SiparisKilo and :APartilenen = 0) or
                 ((BoyaSiparisDetay.PartilenenKilo + BoyaSiparisDetay.PartilenenKiloBaski - BoyaSiparisDetay.IptalKilo - BoyaSiparisDetay.IptalKiloBaski) >= BoyaSiparisDetay.SiparisKilo and :APartilenen = 1) or :APartilenen is null ) and
               (BoyaSiparisDetay.Kapali = :AKapaliDurumu or :AKapaliDurumu is null)
      order by BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId
      into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :MusteriSiparisNo, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisNo,
           :SiparisKilo, :SiparisMetre, :PartilenenKilo, :IptalKilo, :PartilenenMetre, :IptalMetre,
           :HamAdi, :TerminTarihi, :RevizeTerminTarihi, :Kapali, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
           :SiparisProsesAdi, :IlaveIslemler, :OzelApreler do
      begin
        if (Kapali = 1) then KapaliStr = 'Kapali';
        else if (Kapali = 2) then KapaliStr = 'İptal';
        else KapaliStr = '';

        KalanKilo = SiparisKilo - PartilenenKilo + IptalKilo;
        KalanMetre = SiparisMetre - PartilenenMetre + IptalMetre;

        Select Sum(BrutSevkKilo), Sum(BrutSevkMetre), Sum(NetSevkKilo), Sum(NetSevkMetre) from PartiStok where PartiStok.BoyaSiparisDetayId = :BoyaSiparisDetayId
          into :BrutSevkEdilenKilo, :BrutSevkEdilenMetre, :NetSevkEdilenKilo, :NetSevkEdilenMetre;

        if (PartilenenMiktardanIptaliDus = 1) then
        begin
          PartilenenKilo = PartilenenKilo - IptalKilo;
          PartilenenMetre = PartilenenMetre - IptalMetre;
        end

        suspend;
      end
    end
  end
  else
  begin
    for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo,
               LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr, BoyaSiparisDetay.SiparisKilo,
               BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.PartilenenKilo+BoyaSiparisDetay.PartilenenKiloBaski, BoyaSiparisDetay.IptalKilo+BoyaSiparisDetay.IptalKiloBaski, BoyaSiparisDetay.PartilenenMetre+BoyaSiparisDetay.PartilenenMetreBaski, BoyaSiparisDetay.IptalMetre+BoyaSiparisDetay.IptalMetreBaski,
               Ham.Adi, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi, BoyaSiparisDetay.Kapali,
               BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
               SiparisProses.Adi, BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.OzelApreler from BoyaSiparis
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
    left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
    Left Join Kullanici OK on OK.KullaniciId = BoyaSiparisDetay.OnaylayanKullaniciId
    Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisDetay.KullaniciId
      where BoyaSiparis.BoyaSiparisKayitId = :ABoyaSiparisKayitId
    order by BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId
      into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :MusteriSiparisNo, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisNo,
           :SiparisKilo, :SiparisMetre, :PartilenenKilo, :IptalKilo, :PartilenenMetre, :IptalMetre,
           :HamAdi, :TerminTarihi, :RevizeTerminTarihi, :Kapali, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
           :SiparisProsesAdi, :IlaveIslemler, :OzelApreler do
    begin
      if (Kapali = 1) then KapaliStr = 'Kapali';
      else if (Kapali = 2) then KapaliStr = 'İptal';
      else KapaliStr = '';

      KalanKilo = SiparisKilo - PartilenenKilo + IptalKilo;
      KalanMetre = SiparisMetre - PartilenenMetre + IptalMetre;

      Select Sum(BrutSevkKilo), Sum(BrutSevkMetre), Sum(NetSevkKilo), Sum(NetSevkMetre) from PartiStok where PartiStok.BoyaSiparisDetayId = :BoyaSiparisDetayId
        into :BrutSevkEdilenKilo, :BrutSevkEdilenMetre, :NetSevkEdilenKilo, :NetSevkEdilenMetre;

      if (PartilenenMiktardanIptaliDus = 1) then
      begin
        PartilenenKilo = PartilenenKilo - IptalKilo;
        PartilenenMetre = PartilenenMetre - IptalMetre;
      end

      suspend;
    end
  end
