CREATE PROCEDURE BOYASIPARISRAPORU_1(
    AIMALATTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AFIRMAYETKILIID INTEGER,
    AISLEMTIPIID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AMUSTERIKALITENO VARCHAR(20),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AHAMID INTEGER,
    AKULLANICIID INTEGER,
    AKALITEID VARCHAR(20),
    ATALIMATKODU VARCHAR(20),
    AMAKINAID VARCHAR(10),
    ATEKRARURETIM INTEGER,
    ADESENID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AMARKALI INTEGER,
    AKOMISYONLU INTEGER,
    ALAMINASYON INTEGER,
    ARECETETURUID INTEGER,
    ABATCHNO VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ABASKITIPIID INTEGER,
    AHAMGRUP1ID VARCHAR(4),
    AHAMGRUP2ID VARCHAR(4),
    AHAMGRUP3ID VARCHAR(4),
    AHAMGRUP4ID VARCHAR(4),
    ASIPARISRECETEDURUMU INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISNOSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    SIPARISTARIHI TIMESTAMP,
    SOZLESMETARIHI TIMESTAMP,
    SIPARISIVEREN VARCHAR(50),
    SIPARISIVERENEMAIL VARCHAR(80),
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISGRUP3ID VARCHAR(4),
    SIPARISGRUP3ADI VARCHAR(30),
    SIPARISGRUP4ID VARCHAR(4),
    SIPARISGRUP4ADI VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(40),
    TESLIMSEKLIID INTEGER,
    TESLIMSEKLIADI VARCHAR(40),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ISLEMTIPIADI VARCHAR(20),
    SIPARISTIPIID INTEGER,
    YURTDISI INTEGER,
    YURTDISISTR VARCHAR(9),
    TOPLAMORULEN DOUBLE PRECISION,
    PLANLANANMIKTAR DOUBLE PRECISION,
    TOPLAMSAKATORULEN DOUBLE PRECISION,
    ORMESATINALINACAKMIKTAR DOUBLE PRECISION,
    ORMESTOKTANKULLANILACAKMIKTAR DOUBLE PRECISION,
    ORMETOPLAMBOYAYACIKIS DOUBLE PRECISION,
    BOYAHANEHAMDEPOKALANMIKTAR DOUBLE PRECISION,
    ORMESIPARISI INTEGER,
    ORMESIPARISTIPIID INTEGER,
    ORMESIPARISTIPIADI VARCHAR(30),
    DOKUMASIPARISI INTEGER,
    DOKUMASIPARISTIPIID INTEGER,
    DOKUMASIPARISTIPIADI VARCHAR(30),
    IPLIKTALEPIDSTR VARCHAR(255),
    IPLIKTALEPMIKTARI DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    TEMSILCIID INTEGER,
    TEMSILCIADI ADI40,
    MARKALISTR VARCHAR(5),
    KOMISYONLUSTR VARCHAR(5),
    KOMISYONORANI DOUBLE PRECISION)
AS
declare variable FIplikTalepYil         integer;
declare variable FIplikTalepId          integer;
declare variable IadeMiktari            Double Precision;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable uitBoya integer;
declare variable FirmaAdi2yiGorebilir   integer;
declare variable HamStokMasterID        integer;
declare variable HamStokDetayId         integer;
declare variable FMiktar                Numeric(12,3);
declare variable FIptalPartiMiktar      Numeric(12,3);
declare variable FKullanilanMiktar      Numeric(12,3);
declare variable FKalanMiktar           Numeric(12,3);
declare variable FMarkali               Boolean;
declare variable FKomisyonlu            Boolean;
begin
  select uitBoya from UretimIslemTipiGetir into :uitBoya;
  select FirmaAdi2yiGorebilir from Kullanici where KullaniciID = :AKullaniciId into :FirmaAdi2yiGorebilir;

  Select shtFasonUretimeCikis, shtUretimeCikis, shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir
    into :shtFasonUretimeCikis, :shtUretimeCikis, :shtFasonUretim, :shtUretim;

  For Select DISTINCT BoyaSiparisRaporu_2.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, Firma.Adi2,
             BoyaSiparis.SiparisTarihi, BoyaSiparis.SozlesmeTarihi, FirmaYetkili.Adi, BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi,
             BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparis.SiparisGrup3Id, SiparisGrup3.Adi, BoyaSiparis.SiparisGrup4Id, SiparisGrup4.Adi,
             BoyaSiparis.Fason, BoyaSiparis.OdemeSekliId, OdemeSekli.Adi, BoyaSiparis.TeslimSekliId, TeslimSekli.Adi, BoyaSiparis.Notu,
             BoyaSiparis.SiparisTipiId, IslemTipi.Adi, BoyaSiparis.YurtDisi, BoyaSiparis.OrmeSiparisi,
             IK.Adi, BoyaSiparis.InsertTarihi, Sk.Adi, BoyaSiparis.DegisimTarihi, BoyaSiparis.OrmeSiparisTipiId, OrmeSiparisTipi.Adi, FirmaYetkili.Email,
             BoyaSiparis.DokumaSiparisi, BoyaSiparis.DokumaSiparisTipiId, DokumaSiparisTipi.Adi, BoyaSiparisRaporu_2.TemsilciId, BoyaSiparisRaporu_2.TemsilciAdi,
             BoyaSiparis.Markali, BoyaSiparis.Komisyonlu, BoyaSiparis.KomisyonOrani from BoyaSiparisRaporu_2(:AImalatTipiId, :ABaslangicTarihi, :ABitisTarihi, :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AFirmaYetkiliId, :AIslemTipiId, :AIslemGrubuId, :ASatisTipi, :ABoyaSiparisTipiId, :ALabReceteId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMusteriKaliteNo, :AOnayDurumu, :APartilenen, :AKapaliDurumu, 0, 0, :ASiparisiAlanKullaniciId, :AYurtDisi, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AHamId, :AKullaniciId, :AKaliteId, :ATalimatKodu, :AMakinaId, :ATekrarUretim, :ADesenID, :ATemsilciId, :AMarkali, :AKomisyonlu, :ALaminasyon, :AReceteTuruId, :ABatchNo, 1, :ASertifikaNo, :ABaskiTipiId, :AHamGrup1Id, :AHamGrup2Id,  :AHamGrup3Id, :AHamGrup4Id, :ASiparisReceteDurumu)
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisRaporu_2.BoyaSiparisKayitId
    left join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
    left join OdemeSekli On OdemeSekli.OdemeSekliId = BoyaSiparis.OdemeSekliId
    left join TeslimSekli On TeslimSekli.TeslimSekliId = BoyaSiparis.TeslimSekliId
    left join IslemTipi On IslemTipi.IslemTipiId = BoyaSiparis.IslemTipiId
    left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
    Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    Left Join Kullanici IK on IK.KullaniciId = BoyaSiparis.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = BoyaSiparis.KullaniciId
    left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 On SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join SiparisGrup SiparisGrup3 On SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
    left join SiparisGrup SiparisGrup4 On SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
    left join OrmeSiparisTipi on OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
    left join DokumaSiparisTipi on DokumaSiparisTipi.DokumaSiparisTipiId = BoyaSiparis.DokumaSiparisTipiId
   into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :FirmaAdi2, :SiparisTarihi, :SozlesmeTarihi, :SiparisiVeren, :SiparisiAlanKullaniciId,
        :SiparisiAlanKullaniciAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi, :Fason, :OdemeSekliId, :OdemeSekliAdi, :TeslimSekliId, :TeslimSekliAdi, :Notu,
        :SiparisTipiId, :IslemTipiAdi, :YurtDisi, :OrmeSiparisi,
        :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :OrmeSiparisTipiId, :OrmeSiparisTipiAdi, :SiparisiVerenEmail, :DokumaSiparisi, :DokumaSiparisTipiId, :DokumaSiparisTipiAdi,
        :TemsilciId, :TemsilciAdi, :FMarkali, :FKomisyonlu, :KomisyonOrani do
  begin
    if (FMarkali = 1) then MarkaliStr = 'Evet';
    else MarkaliStr = '';

    if (FKomisyonlu = 1) then KomisyonluStr = 'Evet';
    else KomisyonluStr = '';

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (FirmaAdi2yiGorebilir = 0) then FirmaAdi2 = Null;

    if (YurtDisi = 1) then YurtDisiStr = 'Yurt Dışı';
    else YurtDisiStr = '';

    BoyaSiparisNoStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    ToplamOrulen = null;
    ToplamSakatOrulen = Null;
    OrmeSatinAlinacakMiktar = Null;
    OrmeStoktanKullanilacakMiktar = Null;


    if (OrmeSiparisi = 1) then
    begin
      Select Sum(OrmeSiparisDetay.ToplamOrulen), Sum(OrmeSiparisDetay.ToplamSakatOrulen), Sum(OrmeSiparisDetay.SatinAlinacakMiktar), Sum(OrmeSiparisDetay.StoktanKullanilacakMiktar), Sum(OrmeSiparisDetay.PlanlananMiktar) from OrmeSiparisDetay
        where OrmeSiparisDetay.BoyaSiparisKayitID = :BoyaSiparisKayitID into :ToplamOrulen, :ToplamSakatOrulen, :OrmeSatinAlinacakMiktar, :OrmeStoktanKullanilacakMiktar, :PlanlananMiktar;
    end
    else
    begin
      ToplamOrulen=0;
      ToplamSakatOrulen=0;
      OrmeSatinAlinacakMiktar=0;
      OrmeStoktanKullanilacakMiktar =0;
      PlanlananMiktar = 0;
    end

    Select Sum(StokHareket.Miktar) from StokHareket
      where StokHareket.BoyaSiparisKayitId = :BoyaSiparisKayitId and UretimIslemTipiId = :uitBoya and
            (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis)
      into :OrmeToplamBoyayaCikis;

    Select Sum(StokHareket.Miktar) from StokHareket
      left join Ham On Ham.HamID = StokHareket.HamId
      where StokHareket.BoyaSiparisKayitId = :BoyaSiparisKayitId and UretimIslemTipiId = :uitBoya and
            (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and StokHareket.HamIade = 1
      into :IadeMiktari;

    if (IadeMiktari is null) then IadeMiktari = 0;
    if (OrmeToplamBoyayaCikis is null) then OrmeToplamBoyayaCikis = 0;

    OrmeToplamBoyayaCikis = OrmeToplamBoyayaCikis - IadeMiktari;

    IplikTalepMiktari = 0;

    Select Sum(Miktar) from IplikTalepDetay where IplikTalepDetay.BoyaSiparisKayitId = :BoyaSiparisKayitId into :IplikTalepMiktari;

    IplikTalepIdStr = '';

    BoyahaneHamDepoKalanMiktar = Null;

    Select Sum(HamStokMaster.Miktar) from HamStokMaster
      where HamStokMaster.BoyaSiparisKayitID = :BoyaSiparisKayitID and HamStokMaster.OnayDurumu = 0
      into :BoyahaneHamDepoKalanMiktar;

    if (BoyahaneHamDepoKalanMiktar is null) then BoyahaneHamDepoKalanMiktar = 0;

    For Select HamStokDetay.HamStokDetayID, HamStokDetay.Miktar from HamStokMaster
      left join HamStokDetay on HamStokDetay.HamStokMasterID = HamStokMaster.HamStokMasterId
      where HamStokMaster.BoyaSiparisKayitID = :BoyaSiparisKayitID and HamStokDetay.HamStokDetayID is not null
      into :HamStokDetayId, :FMiktar do
    begin
       FKullanilanMiktar = Null;

       Select sum(RafDetay.IadeEdilecekMiktar + RafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.SatisMiktar) from RafDetay
         where RafDetay.HamStokDetayID = :HamStokDetayID
         into :FKullanilanMiktar;

      Select Sum(Miktar) from RafDetay
        where RafDetay.HamStokDetayID = :HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
        into :FIptalPartiMiktar;

      if (FIptalPartiMiktar is null) then FIptalPartiMiktar = 0;
      if (FKullanilanMiktar is null) then FKullanilanMiktar = 0;

      FKalanMiktar = FMiktar + FIptalPartiMiktar - FKullanilanMiktar;

      BoyahaneHamDepoKalanMiktar = BoyahaneHamDepoKalanMiktar + FKalanMiktar;
    end

    for Select IpliKTalep.IplikTalepYil, IpliKTalep.IplikTalepId from IplikTalepDetay
      left join IpliKTalep on IpliKTalep.IpliKTalepKayitID = IpliKTalepDetay.IpliKTalepKayitId
      where IplikTalepDetay.BoyaSiparisKayitId = :BoyaSiparisKayitId
      group by IpliKTalep.IplikTalepYil, IpliKTalep.IplikTalepId
      into :FIplikTalepYil, FIplikTalepId do
    begin
      if (IplikTalepIdStr <> '') then IplikTalepIdStr = IplikTalepIdStr || cast(', ' as varchar(2) character set UTF8);

      IplikTalepIdStr = IplikTalepIdStr ||  '(' || EgemenIntegerToStr(FIplikTalepYil) || ') ' || EgemenIntegerToStr(FIplikTalepId);
    end

    suspend;
  end
end
