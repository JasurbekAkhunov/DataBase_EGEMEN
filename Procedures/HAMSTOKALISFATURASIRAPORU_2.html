CREATE PROCEDURE HAMSTOKALISFATURASIRAPORU_2(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    HAMSTOKFATURAID INTEGER,
    HAMSTOKMASTERID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    RAPORHAMTIPIID INTEGER,
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    MIKTAR NUMERIC(12,3),
    BIRIMFIYAT DOUBLE PRECISION,
    KDV DOUBLE PRECISION,
    KDVHARICFIYAT DOUBLE PRECISION,
    BRUTTUTAR DOUBLE PRECISION,
    BIRIMISKONTOPARASAL DOUBLE PRECISION,
    BIRIMISKONTOORANI1 DOUBLE PRECISION,
    BIRIMISKONTOORANI2 DOUBLE PRECISION,
    BIRIMISKONTOTOPLAMI DOUBLE PRECISION,
    NETBIRIMFIYAT DOUBLE PRECISION,
    KDVTUTARI DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYEGRUPID VARCHAR(4),
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30),
    REFERANSKODU VARCHAR(30))
AS
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;

declare variable FAltIskontoCarpani   Double Precision;
declare variable FKdvYuvarlamaCarpani Double Precision;
declare variable FNetFiyatYuvarlamaCarpani Double Precision;
declare variable T1 Double Precision;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;

  for select HamStokAlisFaturasiRaporu_1.HamStokFaturaId, HamStokMaster.HamStokMasterId, Ham.Adi, Ham.HamTipiId, HamStokMaster.KapAdedi,
             HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Adet, HamStokMaster.Miktar,
             HamStokMaster.Kdv, HamStokMaster.BirimFiyat, HamStokMaster.KdvharicFiyat, HamStokMaster.BirimIskontoOrani1, HamStokMaster.BirimIskontoOrani2,
             HamStokMaster.BirimIskontoToplami, HamStokMaster.BirimIskontoParasal, HamStokMaster.NetBirimFiyat,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.IrsaliyeGrupId,
             HamStokAlisFaturasiRaporu_1.AltIskontoCarpani, HamStokAlisFaturasiRaporu_1.KdvYuvarlamaCarpani,
             HamStokAlisFaturasiRaporu_1.NetFiyatYuvarlamaCarpani, VerilenSiparis.VerilenSiparisId, VerilenSiparis.VerilenSiparisYil, Ham.ReferansKodu
             from HamStokAlisFaturasiRaporu_1(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi)
  left join HamStokMaster on HamStokMaster.HamStokFaturaId = HamStokAlisFaturasiRaporu_1.HamStokFaturaId
  left join Ham on Ham.HamId = HamStokMaster.HamId
  left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = HamStokMaster.VerilenSiparisKayitId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
  into :HamStokFaturaId, :HamStokMasterId, :HamAdi, :HamTipiId,
       :KapAdedi, :Kilo, :Metre, :Adet, :Miktar, :Kdv, :BirimFiyat, :KdvharicFiyat, :BirimIskontoOrani1, :BirimIskontoOrani2,
       :BirimIskontoToplami, :BirimIskontoParasal, :NetBirimFiyat, :IrsaliyeNo, :IrsaliyeTarihi, :IrsaliyeGrupId,
       :FAltIskontoCarpani, :FKdvYuvarlamaCarpani, :FNetFiyatYuvarlamaCarpani, :VerilenSiparisYil, :VerilenSiparisId, :ReferansKodu do
  begin
    if (HamTipiID = hstIplik or HamTipiID = hstElyaf or HamTipiID = hstTelef) then RaporHamTipiId = 0;
    else if (HamTipiID = hstOrmeKumas or HamTipiID = hstRibana or HamTipiID = hstAksesuar) then RaporHamTipiId = 1;
    else if (HamTipiID = hstDokumaKumas) then RaporHamTipiId = 2;
    else RaporHamTipiId = -1;

    BirimIskontoToplami = BirimIskontoToplami * Miktar;
    BrutTutar = KdvHaricFiyat * Miktar;

    T1 = NetBirimFiyat - NetBirimFiyat * FAltIskontoCarpani;
    T1 = Yuvarla(T1, FNetFiyatYuvarlamaCarpani);
    Tutar = T1 * Miktar;

    KdvTutari = T1 * Miktar * Kdv / 100;
    KdvTutari = Yuvarla(:KdvTutari, :FKdvYuvarlamaCarpani);

    VerilenSiparisIdStr = '';
    if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

    Suspend;
  end
end
