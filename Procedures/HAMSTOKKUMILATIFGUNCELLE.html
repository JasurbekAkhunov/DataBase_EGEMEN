CREATE PROCEDURE HAMSTOKKUMILATIFGUNCELLE(
    AFIRMAID VARCHAR(20),
    AFASON INTEGER,
    ANEREDEN INTEGER,
    AOLDHAMID INTEGER,
    ANEWHAMID INTEGER,
    AOLDTARIH TIMESTAMP,
    ANEWTARIH TIMESTAMP,
    AOLDKILO NUMERIC(12,3),
    AOLDMETRE NUMERIC(12,3),
    ANEWKILO NUMERIC(12,3),
    ANEWMETRE NUMERIC(12,3))
AS
declare variable KayitSayisi integer;
declare variable FirmaKodumuz varchar(20);
declare variable AOldMiktar   Numeric(12,3);
declare variable ANewMiktar   Numeric(12,3);
declare variable HamTipiId    integer;
declare variable DokumaKumasiKilodanTakipEt integer;
begin
  Select FirmaKodumuz, DokumaKumasiKilodanTakipEt from Sistem into :FirmaKodumuz, :DokumaKumasiKilodanTakipEt;
  if (AFason = 0) then AFirmaId = FirmaKodumuz;

  AOldMiktar = 0;
  ANewMiktar = 0;

  if (AOldHamId is not null) then
  begin
    Select Ham.HamTipiId from Ham where Ham.HamID = :AOldHamId into :HamTipiId;

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then AOldMiktar = AOldKilo;
    else AOldMiktar = AOldMetre;
  end

  if (ANewHamId is not null) then
  begin
    Select Ham.HamTipiId from Ham where Ham.HamID = :ANewHamId into :HamTipiId;

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then ANewMiktar = ANewKilo;
    else ANewMiktar = ANewMetre;
  end

  KayitSayisi = 0;
  if (ANewHamId is not null) then
    Select Count(*) from HamStokKumilatif where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :ANewHamId and HamStokKumilatif.Tarih = :ANewTarih into :KayitSayisi;

  if (ANereden = 1) then
  begin
    if (AOldHamId is not null) then
    begin
      Update HamStokKumilatif Set GirisMiktar = GirisMiktar - :AOldMiktar
         where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :AOldHamId and HamStokKumilatif.Tarih = :AOldTarih;
    end

    if (ANewHamId is not null) then
    begin
      if (KayitSayisi = 0) then insert into HamStokKumilatif(FirmaId, HamId, Tarih, GirisMiktar) values(:AFirmaId, :ANewHamId, :ANewTarih, :ANewMiktar);
      else Update HamStokKumilatif Set GirisMiktar = GirisMiktar + :ANewMiktar where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :ANewHamId and HamStokKumilatif.Tarih = :ANewTarih;
    end

  end
  else
  if (ANereden = 2) then
  begin
    if (AOldHamId is not null) then
    begin
      Update HamStokKumilatif Set PartilenenMiktar = PartilenenMiktar - :AOldMiktar
         where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :AOldHamId and HamStokKumilatif.Tarih = :AOldTarih;
    end

    if (ANewHamId is not null) then
    begin
      if (KayitSayisi = 0) then insert into HamStokKumilatif(FirmaId, HamId, Tarih, PartilenenMiktar) values(:AFirmaId, :ANewHamId, :ANewTarih, :ANewMiktar);
      else Update HamStokKumilatif Set PartilenenMiktar = PartilenenMiktar + :ANewMiktar where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :ANewHamId and HamStokKumilatif.Tarih = :ANewTarih;
    end
  end
  else
  if (ANereden = 3) then
  begin
    if (AOldHamId is not null) then
    begin
      Update HamStokKumilatif Set IadeEdilenMiktar = IadeEdilenMiktar - :AOldMiktar
         where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :AOldHamId and HamStokKumilatif.Tarih = :AOldTarih;
    end

    if (ANewHamId is not null) then
    begin
      if (KayitSayisi = 0) then insert into HamStokKumilatif(FirmaId, HamId, Tarih, IadeEdilenMiktar) values(:AFirmaId, :ANewHamId, :ANewTarih, :ANewMiktar);
      else Update HamStokKumilatif Set IadeEdilenMiktar = IadeEdilenMiktar + :ANewMiktar where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :ANewHamId and HamStokKumilatif.Tarih = :ANewTarih;
    end
  end
  else
  if (ANereden = 4) then
  begin
    if (AOldHamId is not null) then
    begin
      Update HamStokKumilatif Set SatisMiktar = SatisMiktar - :AOldMiktar
         where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :AOldHamId and HamStokKumilatif.Tarih = :AOldTarih;
    end

    if (ANewHamId is not null) then
    begin
      if (KayitSayisi = 0) then insert into HamStokKumilatif(FirmaId, HamId, Tarih, SatisMiktar) values(:AFirmaId, :ANewHamId, :ANewTarih, :ANewMiktar);
      else Update HamStokKumilatif Set SatisMiktar = SatisMiktar + :ANewMiktar where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :ANewHamId and HamStokKumilatif.Tarih = :ANewTarih;
    end
  end
  else
  if (ANereden = 5) then
  begin
    if (AOldHamId is not null) then
    begin
      Update HamStokKumilatif Set PartiSevkMiktar = PartiSevkMiktar - :AOldMiktar
         where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :AOldHamId and HamStokKumilatif.Tarih = :AOldTarih;
    end

    if (ANewHamId is not null) then
    begin
      if (KayitSayisi = 0) then insert into HamStokKumilatif(FirmaId, HamId, Tarih, PartiSevkMiktar) values(:AFirmaId, :ANewHamId, :ANewTarih, :ANewMiktar);
      else Update HamStokKumilatif Set PartiSevkMiktar = PartiSevkMiktar + :ANewMiktar where HamStokKumilatif.FirmaId = :AFirmaId and HamStokKumilatif.HamId = :ANewHamId and HamStokKumilatif.Tarih = :ANewTarih;
    end
  end
end
