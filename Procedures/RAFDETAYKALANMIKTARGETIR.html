CREATE PROCEDURE RAFDETAYKALANMIKTARGETIR(
    ARAFDETAYID INTEGER)
RETURNS (
    KALANMIKTAR NUMERIC(12,3),
    IADEEDILECEKMIKTAR NUMERIC(12,3),
    IADEEDILENMIKTAR NUMERIC(12,3),
    PARTILENENMIKTAR NUMERIC(12,3),
    RAFTRANSFERMIKTAR NUMERIC(12,3),
    SATISMIKTAR NUMERIC(12,3))
AS
declare variable Miktar                 Numeric(12,3);
declare variable AktarmayaCikanKilo    Numeric(12,3);
declare variable AktarmadanIadeEdienKilo  Numeric(12,3);
begin
  select RafDetay.Miktar, RafDetay.IadeEdilecekMiktar, rafDetay.IadeEdilenMiktar, RafDetay.PartilenenMiktar, RafDetay.SatisMiktar,
         RafDetay.RafTransferMiktar, RafDetay.AktarmayaCikanKilo, RafDetay.AktarmadanIadeEdienKilo from RafDetay
  where RafDetay.RafDetayId = :ARafDetayId
  into :Miktar, :IadeEdilecekMiktar, :IadeEdilenMiktar, :PartilenenMiktar, :SatisMiktar, :RafTransferMiktar, :AktarmayaCikanKilo, :AktarmadanIadeEdienKilo;

  KalanMiktar = Miktar + AktarmadanIadeEdienKilo - AktarmayaCikanKilo - PartilenenMiktar - RafTransferMiktar - IadeEdilecekMiktar - IadeEdilenMiktar - SatisMiktar;
  suspend;
end
