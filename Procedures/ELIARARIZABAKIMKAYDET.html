CREATE PROCEDURE ELIARARIZABAKIMKAYDET(
    AMAKINAID VARCHAR(10),
    APERSONELADI VARCHAR(80),
    ATARIH TIMESTAMP,
    AGELDI INTEGER)
RETURNS (
    STATUS INTEGER,
    MESAJ VARCHAR(255))
AS
declare variable GeldiMakinaDurumuId   integer;
declare variable GittiMakinaDurumuId   integer;
declare variable ArizaIhbarId          integer;
declare variable ArizaBakimId          integer;
declare variable KayitSayisi           integer;
declare variable AsilMakinaID      MakinaKoduSize;
declare variable fPersonelID  Integer;
begin
  Select Coalesce(AsilMakinaID, MakinaID) from Makina where Makina.OtomasyonMakinaNo = :AMakinaId into :AsilMakinaID;
  if (AsilMakinaID is Not Null) then AMakinaID = :AsilMakinaID;

  Select First 1 Personel.PersonelId from Personel where Upper(Personel.Adi) = Upper(:APersonelAdi) into :fPersonelID;

  if (fPersonelID is null) then
  begin
    Select PersonelID from Kullanici where KullaniciID = 1 into :fPersonelID;
    if (fPersonelID Is Null) then
    begin
      Status = 5;
      mesaj = 'Personel Kodu Uyuşmadı ve Sistem Yöneticisine atanmış Personel bulunamadı!';
      suspend;
      Exit;
    end
  end

  if (AGeldi = 1) then
  begin
    Select ArizaIhbarId from ArizaIhbar where ArizaIhbar.MakinaId = :AMakinaId and ArizaIhbar.BittiMakinaDurumuId is null into :ArizaIhbarId;

    if (ArizaIhbarId is not null) then
    begin
      Select Count(*) from ArizaBakim where ArizaBakim.MakinaId = :AMakinaId and ArizaBakim.PersonelId = :fPersonelID and ArizaBakim.GittiMakinaDurumuId is null into :KayitSayisi;
      if (kayitSayisi = 0) then
      begin
        GeldiMakinaDurumuId = gen_Id(MakinaDurumu_id_gen, 1);
        insert into MakinaDurumu(MakinaDurumuId, MakinaId, Tarih, ServisciGeldi, ServisciVar) values(:GeldiMakinaDurumuId, :AMakinaId, :ATarih, 1, 1); /* bakımcı geldi  */

        insert into ArizaBakim(MakinaId, GeldiMakinaDurumuId, ArizaIhbarId, PersonelId) values(:AMakinaId, :GeldiMakinaDurumuId, :ArizaIhbarId, :fPersonelID);
        Status = 1;
        Mesaj = 'Kayıt yapıldı';
      end
      else
      begin
        Status = 2;
        Mesaj = 'Bakımcı Zaten Var';
      end
    end
    else
    begin
      Status = 3;
      Mesaj = 'Ariza Yok';
    end
  end
  else if (AGeldi = 2) then
  begin
    Select ArizaBakimId from ArizaBakim where ArizaBakim.GittiMakinaDurumuId is null and ArizaBakim.PersonelId = :fPersonelID and ArizaBakim.MakinaId = :AMakinaId
    into :ArizaBakimId;

    if (ArizaBakimId is not null) then
    begin
      GittiMakinaDurumuId = gen_Id(MakinaDurumu_id_gen, 1);
/* bu bakımcıdan baska bakımcı varmı bak, servisci(bakımcı)var degiskenine ne koyacagına karar ver.. */
      Select Count(*) from ArizaBakim where ArizaBakim.GittiMakinaDurumuId is null and ArizaBakim.MakinaId = :AMakinaId and ArizaBakim.PersonelId <> :fPersonelID into :KayitSayisi;

      if (KayitSayisi = 0) then  insert into MakinaDurumu(MakinaDurumuId, MakinaId, Tarih, ServisciGeldi, ServisciVar) values(:GittiMakinaDurumuId, :AMakinaId, :ATarih, 2, 0); /* bakımcı gitti  */
      else insert into MakinaDurumu(MakinaDurumuId, MakinaId, Tarih, ServisciGeldi, ServisciVar) values(:GittiMakinaDurumuId, :AMakinaId, :ATarih, 2, 1); /* bakımcı gitti  */

      Update ArizaBakim Set GittiMakinaDurumuId = :GittiMakinaDurumuId where ArizaBakim.ArizaBakimID = :ArizaBakimId;
      Status = 1;
      Mesaj = 'Kayıt Yapıldı..';
    end
    else
    begin
      Status = 4;
      Mesaj = 'Bakımcı Yok';
    end
  end
  else
  begin
    Status = 5;
    mesaj = 'Parametre Hatalı..Egemen Yazılıma Haber ver.. ';
  end
  suspend;
end
