CREATE PROCEDURE KALITEKONTROLRAPORU_4(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    ARECETEID VARCHAR(20),
    AFIRMAID VARCHAR(20),
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AIMALATTIPIID INTEGER,
    AHAMID INTEGER,
    ABOYASIPSIRANO INTEGER,
    AONAYDURUMU INTEGER,
    AAKTARMADAHIL SMALLINT,
    AHAMTIPIID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    PSKALITEKONTROLID INTEGER,
    PSKALITEKONTROLHATAMETREID INTEGER,
    HATATIPIID INTEGER,
    BASLANGICMETRE NUMERIC(12,3),
    BITISMETRE NUMERIC(12,3))
AS
begin
  For Select KaliteKontrolRaporu_3.PartiSinifId, KaliteKontrolRaporu_3.PSKaliteKontrolId, PSKaliteKontrolHataMetre.PSKaliteKontrolHataMetreId, KaliteKontrolRaporu_3.HataTipiId,
         PSKaliteKontrolHataMetre.BaslangicMetre, PSKaliteKontrolHataMetre.BitisMetre from KaliteKontrolRaporu_3(:ABaslangicTarihi, :ABitisTarihi, :APartiYil, :APartiId, :AReceteId, :AFirmaId, :ABoyaSiparisYil, :ABoyaSiparisId, :AImalatTipiId, :AHamId, :ABoyaSipSiraNo, :AOnayDurumu, :AAktarmaDahil, :AHamTipiId)
    left join PSKaliteKontrolHataMetre on PSKaliteKontrolHataMetre.PSKaliteKontrolId = KaliteKontrolRaporu_3.PSKaliteKontrolID and PSKaliteKontrolHataMetre.HataTipiID = KaliteKontrolRaporu_3.HataTipiId
    where PSKaliteKontrolHataMetre.PSKaliteKontrolHataMetreId is not null
    order by KaliteKontrolRaporu_3.PartiSinifId, KaliteKontrolRaporu_3.PSKaliteKontrolId
  into :PartiSinifId, :PSKaliteKontrolId, :PSKaliteKontrolHataMetreId, :HataTipiId, :BaslangicMetre, :BitisMetre do suspend;
end
