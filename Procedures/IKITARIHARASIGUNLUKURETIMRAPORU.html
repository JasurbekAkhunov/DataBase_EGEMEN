AS
declare variable FPartiTamirId    integer;
declare variable FIlaveProses    integer;
declare variable FKilo            Numeric(12,3);
declare variable FMakinaId        varchar(10);
declare variable FMakinaRaporGrupId integer;
declare variable FMakinaGrubu     varchar(10);
declare variable FMakinaGrupAdi   varchar(30);

declare variable FMakinaPlanlamaGrupId varchar(10);
declare variable FAsamaId         integer;
declare variable FRaporHamTipiId  integer;
declare variable FLastRaporHamTipiID integer;
declare variable FCikisTarihi     TimeStamp;
declare variable FLastCikisTarihi TimeStamp;
declare variable FLastMakinaId    varchar(10);
declare variable FLastAsamaId     integer;
declare variable VardiyaBaslangicSaati  Double Precision;
declare variable FSaat                  Double Precision;
Declare variable FDate                  date;
declare variable itTamir          integer;
declare variable FIslemTipiId     integer;
declare variable Vardiya     integer;
declare variable VardiyaSayisi integer;
declare variable T1                Double Precision;
declare variable VardiyaBaslangicTarihi TimeStamp;
Declare variable FFaturaKesilecek   integer;
Declare variable FUcretli           integer;
declare variable FIslemGrubuId integer;
declare variable FPartiId integer;
declare variable FNetMiktar numeric(12,3);
declare variable FNetSevkMiktar numeric(12,3);
begin
  Select itTamir from IslemTipiGetir into :itTamir;

  Select TimeStampToDouble(TimeStampToTime(VardiyaBaslangicSaati)), VardiyaSayisi from sistem into :VardiyaBaslangicSaati, :VardiyaSayisi;

  FLastCikisTarihi = Null;
  FLastMakinaID = Null;
  FLastAsamaID = Null;
  FLastRaporHamTipiID = -1;

  for select PartiAsamaStokDetay.MakinaId, Makina.RaporGrupId, Makina.MakinaGrubu, MakinaGrup.Adi, Makina.MakinaPlanlamaGrupId, PartiAsamaStok.AsamaId, Parti.IslemTipiId, PartiAsamalari.PartiTamirId,
             PartiAsamalari.IlaveProses, PartiAsamaStokDetay.kilo, PartiAsamaStokDetay.Cikistarihi, TamirCozumSekli.Ucretli, Parti.FaturaKesilecek, Parti.IslemGrubuId, Parti.PartiId, PartiStok.NetMiktar, PartiStok.NetSevkMiktar,
             case when (HamTipiId = 0 or HamTipiId = 5 or HamTipiId = 7) then 0 when (HamTipiId = 2) then 2 else 1 end from PartiAsamaStokDetay
  left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
  left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
  left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
  left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
  left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
  left join MakinaGrup on MakinaGrup.MakinaGrupId = Makina.MakinaGrubu
  left join MakinaPlanlamaGrup on MakinaPlanlamaGrup.MakinaPlanlamaGrupId = Makina.MakinaPlanlamaGrupId
  left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
  left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  where PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi < :ABitisTarihi and
       (Makina.UniteId = :AUniteId or :AUniteId is null) and Makina.UretimRaporlarinaDahil = 1 and
       (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null)
  order by PartiAsamaStokDetay.MakinaId, PartiAsamaStok.AsamaId, PartiAsamaStokDetay.CikisTarihi, 12
  into :FMakinaId, :FMakinaRaporGrupId, :FMakinaGrubu, :FMakinaGrupAdi, :FMakinaPlanlamaGrupId, :FAsamaId, :FIslemTipiId, :FPartiTamirId, :FIlaveProses, :FKilo, :FCikistarihi, :FUcretli, :FFaturaKesilecek, :FIslemGrubuId,:FPartiId, :FNetMiktar ,:FNetSevkMiktar ,:FRaporHamTipiId do
  begin
    FDate = FCikisTarihi;
    FSaat = TimeStampToDouble(FCikisTarihi) - TimeStampToDouble(FDate);

    VardiyaBaslangicTarihi = FDate + VardiyaBaslangicSaati;

    if (FCikisTarihi is not null) then
    begin
       T1 = FCikisTarihi - VardiyaBaslangicTarihi;
       if (T1 < 0) then T1 = T1 * -1;

       vardiya = VardiyaHesapla(FSaat, VardiyaSayisi);
    end
    else vardiya = 0;

    FCikisTarihi = FDate;
    if (FSaat < VardiyaBaslangicSaati) then FCikisTarihi = FCikisTarihi - 1;

    if (FLastCikisTarihi is null or FMakinaID is null or FLastAsamaId is null or FCikisTarihi <> FLastCikisTarihi or FMakinaId <> FLastMakinaId or FAsamaId <> FLastAsamaId or FLastRaporHamTipiID <> FRaporHamTipiID) then
    begin
      if (FLastCikisTarihi is not null) then
      begin
        ToplamUretim = NormalUretimToplami + IcTamirUretimToplami + DisTamirUretimToplami + IlaveProsesUretimToplami;
        V1_ToplamUretim = V1_NormalUretimToplami + V1_IcTamirUretimToplami + V1_DisTamirUretimToplami + V1_IlaveProsesUretimToplami;
        V2_ToplamUretim = V2_NormalUretimToplami + V2_IcTamirUretimToplami + V2_DisTamirUretimToplami + V2_IlaveProsesUretimToplami;
        V3_ToplamUretim = V3_NormalUretimToplami + V3_IcTamirUretimToplami + V3_DisTamirUretimToplami + V3_IlaveProsesUretimToplami;
        suspend;
      end

      NormalUretimToplami = 0;
      IlaveProsesUretimToplami = 0;
      IcTamirUretimToplami  = 0;
      UcretliIcTamirUretimToplami  = 0;
      UCretsizIcTamirUretimToplami  = 0;

      Beyaz         = 0;
      Renkli        = 0;
      Yikama        = 0;
      SekerKasar    = 0;
      KuruBolum     = 0;
      CiftBoyama    = 0;
      Siyah         = 0;
      NaylonBoyama  = 0;
      PolyesterBoyama  = 0;
      Diger         = 0;

      IlaveProsesBeyaz            = 0;
      IlaveProsesRenkli           = 0;
      IlaveProsesYikama           = 0;
      IlaveProsesSekerKasar       = 0;
      IlaveProsesKuruBolum        = 0;
      IlaveProsesCiftBoyama       = 0;
      IlaveProsesSiyah            = 0;
      IlaveProsesNaylonBoyama     = 0;
      IlaveProsesPolyesterBoyama  = 0;
      IlaveProsesDiger            = 0;

      DisTamirUretimToplami = 0;
      UcretliDisTamirUretimToplami = 0;
      UcretsizDisTamirUretimToplami = 0;
      ToplamUretim = 0;

      V1_NormalUretimToplami = 0;
      V1_IlaveProsesUretimToplami = 0;
      V1_IcTamirUretimToplami = 0;
      V1_UcretliIcTamirUretimToplami = 0;
      V1_UcretsizIcTamirUretimToplami = 0;
      V1_DisTamirUretimToplami = 0;
      V1_UcretliDisTamirUretimToplami = 0;
      V1_UcretsizDisTamirUretimToplam = 0;
      V1_ToplamUretim = 0;

      V2_NormalUretimToplami = 0;
      V2_IlaveProsesUretimToplami = 0;
      V2_IcTamirUretimToplami = 0;
      V2_UcretliIcTamirUretimToplami = 0;
      V2_UcretsizIcTamirUretimToplami = 0;
      V2_DisTamirUretimToplami = 0;
      V2_UcretliDisTamirUretimToplami = 0;
      V2_UcretsizDisTamirUretimToplam = 0;
      V2_ToplamUretim = 0;

      V3_NormalUretimToplami = 0;
      V3_IlaveProsesUretimToplami = 0;
      V3_IcTamirUretimToplami = 0;
      V3_UcretliIcTamirUretimToplami = 0;
      V3_UcretsizIcTamirUretimToplami = 0;
      V3_DisTamirUretimToplami = 0;
      V3_UcretliDisTamirUretimToplami = 0;
      V3_UcretsizDisTamirUretimToplam = 0;
      V3_ToplamUretim = 0;

      Tarih = FCikisTarihi;
      FLastCikisTarihi = FCikisTarihi;
      MakinaID = FMakinaId;
      IslemGrubuId = FIslemGrubuId;
      FLastMakinaID = FMakinaId;
      MakinaGrubu = FMakinaGrubu;
      MakinaGrupAdi = FMakinaGrupAdi;
      MakinaPlanlamaGrupId = FMakinaPlanlamaGrupId;
      MakinaRaporGrupId = FMakinaRaporGrupId;
      AsamaID = FAsamaId;
      FLastAsamaID = FAsamaId;
      FLastRaporHamTipiID = FRaporHamTipiID;
      RaporHamTipiID = FRaporHamTipiID;
      PartiId = FPartiId;
      NetMiktar = FNetMiktar;
      NetSevkMiktar = FNetSevkMiktar;

      if (RaporHamTipiId = 0) then RaporHamTipiStr = 'iplik+elyaf';
      else if (RaporHamTipiId = 1) then RaporHamTipiStr = 'Kumas+Aksesuar+Ribana';
      else if (RaporHamTipiId = 2) then RaporHamTipiStr = 'Dokuma';
      else RaporHamTipiStr = 'HatalÄ±';
    end

    if (FIlaveProses = 1) then
    begin
      IlaveProsesUretimToplami = IlaveProsesUretimToplami + FKilo;

      if (FIslemGrubuId = 0) then
        IlaveProsesBeyaz = IlaveProsesBeyaz + FKilo;
      else if (FIslemGrubuId = 1) then
        IlaveProsesRenkli = IlaveProsesRenkli + FKilo;
      else if (FIslemGrubuId = 2) then
        IlaveProsesYikama = IlaveProsesYikama + FKilo;
      else if (FIslemGrubuId = 3) then
        IlaveProsesSekerKasar = IlaveProsesSekerKasar + FKilo;
      else if (FIslemGrubuId = 4) then
        IlaveProsesKuruBolum = IlaveProsesKuruBolum + FKilo;
      else if (FIslemGrubuId = 5) then
        IlaveProsesCiftBoyama = IlaveProsesCiftBoyama + FKilo;
      else if (FIslemGrubuId = 6) then
        IlaveProsesSiyah = IlaveProsesSiyah + FKilo;
      else if (FIslemGrubuId = 8) then
        IlaveProsesNaylonBoyama = IlaveProsesNaylonBoyama + FKilo;
      else if (FIslemGrubuId = 20) then
        IlaveProsesPolyesterBoyama = IlaveProsesPolyesterBoyama + FKilo;
      else IlaveProsesDiger = IlaveProsesDiger + FKilo;
    end
    else if (FPartiTamirID is not null) then
    begin
      IcTamirUretimToplami = IcTamirUretimToplami + FKilo;

      if (FUcretli = 1) then
        UcretliIcTamirUretimToplami = UcretliIcTamirUretimToplami + FKilo;
      else
        UcretsizIcTamirUretimToplami = UcretsizIcTamirUretimToplami + FKilo;
    end
    else if (FIslemTipiID = itTamir) then
    begin
      DisTamirUretimToplami = DisTamirUretimToplami + FKilo;
      if (FFaturaKesilecek = 1) then
        UcretliDisTamirUretimToplami = UcretliDisTamirUretimToplami + FKilo;
      else
        UcretsizDisTamirUretimToplami = UcretsizDisTamirUretimToplami + FKilo;
    end
    else
    begin
      NormalUretimToplami = NormalUretimToplami + FKilo;

      if (FIslemGrubuId = 0) then
        Beyaz = Beyaz + FKilo;
      else if (FIslemGrubuId = 1) then
        Renkli = Renkli + FKilo;
      else if (FIslemGrubuId = 2) then
        Yikama = Yikama + FKilo;
      else if (FIslemGrubuId = 3) then
        SekerKasar = SekerKasar + FKilo;
      else if (FIslemGrubuId = 4) then
        KuruBolum = KuruBolum + FKilo;
      else if (FIslemGrubuId = 5) then
        CiftBoyama = CiftBoyama + FKilo;
      else if (FIslemGrubuId = 6) then
        Siyah = Siyah + FKilo;
      else if (FIslemGrubuId = 8) then
        NaylonBoyama = NaylonBoyama + FKilo;
      else if (FIslemGrubuId = 20) then
        PolyesterBoyama = PolyesterBoyama + FKilo;
      else Diger = Diger + FKilo;

    end

    if (Vardiya = 1) then
    begin
      if (FIlaveProses = 1) then V1_IlaveProsesUretimToplami = V1_IlaveProsesUretimToplami + FKilo;
      else if (FPartiTamirID is not null) then
      begin
        V1_IcTamirUretimToplami = V1_IcTamirUretimToplami + FKilo;
        if (FUcretli = 1) then V1_UcretliIcTamirUretimToplami = V1_UcretliIcTamirUretimToplami + FKilo;
        else V1_UcretsizIcTamirUretimToplami = V1_UcretsizIcTamirUretimToplami + FKilo;
      end
      else if (FIslemTipiID = itTamir) then
      begin
        V1_DisTamirUretimToplami = V1_DisTamirUretimToplami + FKilo;
        if (FFaturaKesilecek = 1) then V1_UcretliDisTamirUretimToplami = V1_UcretliDisTamirUretimToplami + FKilo;
        else V1_UcretsizDisTamirUretimToplam = V1_UcretsizDisTamirUretimToplam + FKilo;
      end
      else
      begin
        V1_NormalUretimToplami = V1_NormalUretimToplami + FKilo;

      end
    end

    else if (Vardiya = 2) then
    begin
      if (FIlaveProses = 1) then V2_IlaveProsesUretimToplami = V2_IlaveProsesUretimToplami + FKilo;
      else if (FPartiTamirID is not null) then
      begin
        V2_IcTamirUretimToplami = V2_IcTamirUretimToplami + FKilo;
        if (FUcretli = 1) then V2_UcretliIcTamirUretimToplami = V2_UcretliIcTamirUretimToplami + FKilo;
        else V2_UcretsizIcTamirUretimToplami = V2_UcretsizIcTamirUretimToplami + FKilo;
      end
      else if (FIslemTipiID = itTamir) then
      begin
        V2_DisTamirUretimToplami = V2_DisTamirUretimToplami + FKilo;
        if (FFaturaKesilecek = 1) then V2_UcretliDisTamirUretimToplami = V2_UcretliDisTamirUretimToplami + FKilo;
        else V2_UcretsizDisTamirUretimToplam = V2_UcretsizDisTamirUretimToplam + FKilo;
      end
      else
      begin
        V2_NormalUretimToplami = V2_NormalUretimToplami + FKilo;

      end
    end

    else if (Vardiya = 3) then
    begin
      if (FIlaveProses = 1) then V3_IlaveProsesUretimToplami = V3_IlaveProsesUretimToplami + FKilo;
      else if (FPartiTamirID is not null) then
      begin
        V3_IcTamirUretimToplami = V3_IcTamirUretimToplami + FKilo;
        if (FUcretli = 1) then V3_UcretliIcTamirUretimToplami = V3_UcretliIcTamirUretimToplami + FKilo;
        else V3_UcretsizIcTamirUretimToplami = V3_UcretsizIcTamirUretimToplami + FKilo;
      end
      else if (FIslemTipiID = itTamir) then
      begin
        V3_DisTamirUretimToplami = V3_DisTamirUretimToplami + FKilo;
        if (FFaturaKesilecek = 1) then V3_UcretliDisTamirUretimToplami = V3_UcretliDisTamirUretimToplami + FKilo;
        else V3_UcretsizDisTamirUretimToplam = V3_UcretsizDisTamirUretimToplam + FKilo;
      end
      else
      begin
        V3_NormalUretimToplami = V3_NormalUretimToplami + FKilo;

      end
    end
  end

  if (FLastCikisTarihi is not null) then
  begin
    ToplamUretim = NormalUretimToplami + IcTamirUretimToplami + DisTamirUretimToplami + IlaveProsesUretimToplami;
    V1_ToplamUretim = V1_NormalUretimToplami + V1_IcTamirUretimToplami + V1_DisTamirUretimToplami + V1_IlaveProsesUretimToplami;
    V2_ToplamUretim = V2_NormalUretimToplami + V2_IcTamirUretimToplami + V2_DisTamirUretimToplami + V2_IlaveProsesUretimToplami;
    V3_ToplamUretim = V3_NormalUretimToplami + V3_IcTamirUretimToplami + V3_DisTamirUretimToplami + V3_IlaveProsesUretimToplami;

    suspend;
  end

  MakinaId              = Null;
  MakinaRaporGrupId    =  Null;
  FLastMakinaId          = Null;
  FLastCikisTarihi      = null;
  MakinaGrubu           = Null;
  MakinaGrupAdi         = Null;
  Tarih                 = Null;
  RaporHamTipiId        = Null;
  RaporHamTipiStr       = Null;
  AsamaId               = Null;
  NormalUretimToplami = 0;
  IlaveProsesUretimToplami = 0;
  IcTamirUretimToplami  = 0;
  UcretliIcTamirUretimToplami  = 0;
  UcretsizIcTamirUretimToplami  = 0;
  DisTamirUretimToplami = 0;
  UcretliDisTamirUretimToplami = 0;
  UcretsizDisTamirUretimToplami = 0;
  ToplamUretim = 0;

  V1_NormalUretimToplami = 0;
  V1_IlaveProsesUretimToplami = 0;
  V1_IcTamirUretimToplami = 0;
  V1_UcretliIcTamirUretimToplami = 0;
  V1_UcretsizIcTamirUretimToplami = 0;
  V1_DisTamirUretimToplami = 0;
  V1_UcretliDisTamirUretimToplami = 0;
  V1_UcretsizDisTamirUretimToplam = 0;
  V1_ToplamUretim = 0;

  V2_NormalUretimToplami = 0;
  V2_IlaveProsesUretimToplami = 0;
  V2_IcTamirUretimToplami = 0;
  V2_UcretliIcTamirUretimToplami = 0;
  V2_UcretsizIcTamirUretimToplami = 0;
  V2_DisTamirUretimToplami = 0;
  V2_UcretliDisTamirUretimToplami = 0;
  V2_UcretsizDisTamirUretimToplam = 0;
  V2_ToplamUretim = 0;

  V3_NormalUretimToplami = 0;
  V3_IlaveProsesUretimToplami = 0;
  V3_IcTamirUretimToplami = 0;
  V3_UcretliIcTamirUretimToplami = 0;
  V3_UcretsizIcTamirUretimToplami = 0;
  V3_DisTamirUretimToplami = 0;
  V3_UcretliDisTamirUretimToplami = 0;
  V3_UcretsizDisTamirUretimToplam = 0;
  V3_ToplamUretim = 0;

  RaporHamTipiID = 0;
  RaporHamTipiStr = 'Ä°plik+Elyaf';

  Select Asama.AsamaId from Sistem
    left join Asama on Asama.AsamaId = Sistem.HamAktarmaAsamaId
    where sistemid = 1 into AsamaId;

  if (AsamaId is Null) then exit;

  For Select HAktarmaIsEmri.MakinaId, Makina.RaporGrupId, Makina.MakinaGrubu, MakinaGrup.Adi, HAktarmaIsEmri.BitisTarihi, Sum(HAktarmaIsEmriDetay.Kilo) from HAktarmaIsEmri
    left join HAktarmaIsEmriDetay on HAktarmaIsEmriDetay.HAktarmaIsEmriKayitId = HAktarmaIsEmri.HAktarmaIsEmriKayitId
    left join Makina on Makina.MakinaId = HAktarmaIsEmri.MakinaId
    left join MakinaGrup on MakinaGrup.MakinaGrupId = Makina.MakinaGrubu
    where HAktarmaIsEmri.BitisTarihi >= :ABaslangicTarihi and HAktarmaIsEmri.BitisTarihi < :ABitisTarihi
    group By HAktarmaIsEmri.MakinaId, Makina.RaporGrupId, Makina.MakinaGrubu, MakinaGrup.Adi, HAktarmaIsEmri.BitisTarihi, HAktarmaIsEmri.HAktarmaIsEmriKayitId
    into :FMakinaId, :FMakinaRaporGrupId, :FMakinaGrubu, :FMakinaGrupAdi, :FCikisTarihi, :FKilo do
  begin
    FDate = FCikisTarihi;
    FSaat = TimeStampToDouble(FCikisTarihi) - TimeStampToDouble(FDate);

    VardiyaBaslangicTarihi = FDate + VardiyaBaslangicSaati;

    if (FCikisTarihi is not null) then
    begin
       T1 = FCikisTarihi - VardiyaBaslangicTarihi;
       if (T1 < 0) then T1 = T1 * -1;

       vardiya = VardiyaHesapla(FSaat, VardiyaSayisi);
    end
    else vardiya = 0;

    FCikisTarihi = FDate;
    if (FSaat < VardiyaBaslangicSaati) then FCikisTarihi = FCikisTarihi - 1;

    if (FLastCikisTarihi is null or FMakinaId <> FLastMakinaId or FCikisTarihi <> FLastCikisTarihi) then
    begin
      if (FLastMakinaId <> '') then
      begin
        ToplamUretim = NormalUretimToplami;
        suspend;
      end

      Tarih = FCikisTarihi;
      FLastCikisTarihi = FCikisTarihi;
      MakinaId = FMakinaId;
      FLastMakinaId = FMakinaId;
      MakinaGrubu = FMakinaGrubu;
      MakinaGrupAdi = FMakinaGrupAdi;

      NormalUretimToplami = 0;
      IlaveProsesUretimToplami = 0;
      IcTamirUretimToplami  = 0;
      DisTamirUretimToplami = 0;
      ToplamUretim = 0;
    end
    NormalUretimToplami = NormalUretimToplami + FKilo;

    if (Vardiya = 1) then
    begin
      V1_NormalUretimToplami = V1_NormalUretimToplami + FKilo;
    end
    else if (Vardiya = 2) then
    begin
      V2_NormalUretimToplami = V2_NormalUretimToplami + FKilo;

    end
    else if (Vardiya = 3) then
    begin
      V3_NormalUretimToplami = V3_NormalUretimToplami + FKilo;
    end
  end

  if (FLastCikisTarihi is not null) then
  begin
    ToplamUretim = NormalUretimToplami + IcTamirUretimToplami + DisTamirUretimToplami + IlaveProsesUretimToplami;
    V1_ToplamUretim = V1_NormalUretimToplami + V1_IcTamirUretimToplami + V1_DisTamirUretimToplami + V1_IlaveProsesUretimToplami;
    V2_ToplamUretim = V2_NormalUretimToplami + V2_IcTamirUretimToplami + V2_DisTamirUretimToplami + V2_IlaveProsesUretimToplami;
    V3_ToplamUretim = V3_NormalUretimToplami + V3_IcTamirUretimToplami + V3_DisTamirUretimToplami + V3_IlaveProsesUretimToplami;

    suspend;
  end

end
