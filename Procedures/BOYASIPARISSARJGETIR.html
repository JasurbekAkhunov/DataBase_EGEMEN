CREATE PROCEDURE BOYASIPARISSARJGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISSARJID INTEGER,
    SARJNO INTEGER,
    KAZANKAPASITE INTEGER,
    BOBINSAYISI INTEGER,
    GRAMAJ NUMERIC(12,3),
    MIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    FIREORANI DOUBLE PRECISION,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(255),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable LastBoyaSiparisDetayId integer;
declare variable FPartiIdStr varchar(30);
begin
  SarjNo = 0;
  LastBoyaSiparisDetayId = -1;
  for select Distinct BoyaSiparisSarj.BoyaSiparisDetayId, BoyaSiparisSarj.BoyaSiparisSarjId, BoyaSiparisSarj.KazanKapasite, BoyaSiparisSarj.BobinSayisi,
             BoyaSiparisSarj.Gramaj, BoyaSiparisSarj.FireOrani, BoyaSiparisSarj.SarjKilo, BoyaSiparisSarj.BrutSarjKilo, IK.Adi, BoyaSiparisSarj.InsertTarihi, Sk.Adi, BoyaSiparisSarj.DegisimTarihi from BoyaSiparisSarj
  Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisSarj.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisSarj.KullaniciId
  where BoyaSiparisSarj.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  order by BoyaSiparisSarj.BoyaSiparisDetayId, BoyaSiparisSarj.BoyaSiparisSarjId
  into :BoyaSiparisDetayId, :BoyaSiparisSarjId, :KazanKapasite, :BobinSayisi, :Gramaj, :FireOrani, :Miktar, :BrutMiktar,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    if (LastBoyaSiparisDetayId <> BoyaSiparisDetayId) then
    begin
      LastBoyaSiparisDetayId = BoyaSiparisDetayId;
      SarjNo = 0;
    end
    SarjNo = SarjNo + 1;
    PartiIdStr = '';

    For Select Distinct Parti.PartiYil, Parti.PartiId from PartiStok
      left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      where PartiStok.BoyaSiparisSarjId = :BoyaSiparisSarjId and PartiStok.PartiStokIptalKayitId is null
      into :PartiYil, :PartiId do
    begin
      FPartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (Char_Length(PartiIdStr) + Char_Length(FPartiIdStr) < 253) then
      begin
        if (PartiIdStr <> '' ) then PartiIdStr = PartiIdStr || cast(', ' as varchar(2) character set UTF8);
        PartiIdStr = PartiIdStr || FPartiIdStr;
      end
    end
    PartiYil = Null;
    PartiId = Null;
    suspend;
  end
end
