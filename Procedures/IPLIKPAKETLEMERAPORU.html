CREATE PROCEDURE IPLIKPAKETLEMERAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    ARECETEID VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    AFASON INTEGER,
    ATARTIKAYDIOLMAYANLAR EGEBOOLEAN)
RETURNS (
    TARIH TIMESTAMP,
    TARIHSTR VARCHAR(25),
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMADITOPLAMI VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(50),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    MUSTERISIPARISNO VARCHAR(50),
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    PARTISTOKKILO NUMERIC(12,3),
    PARTISTOKBOBINSAYISI INTEGER,
    PAKETLEMEKAPADEDI INTEGER,
    PAKETLEMEBOBINSAYISI INTEGER,
    PAKETLEMEBRUTKILO NUMERIC(12,3),
    PAKETLEMENETKILO NUMERIC(12,3),
    TAMAMITARTILDI INTEGER,
    FIREORANI DOUBLE PRECISION,
    RECETERENKDERINLIGIADI VARCHAR(20),
    SIPARISRENKDERINLIGIADI VARCHAR(20),
    TAMAMIRAPORLANDI INTEGER,
    GIRISNEM DOUBLE PRECISION,
    CIKISNEM DOUBLE PRECISION,
    SURTUNMEKATSAYISI DOUBLE PRECISION,
    PAKETLEMEFIRE DOUBLE PRECISION,
    HAMSTOKMALIYETI DOUBLE PRECISION,
    ASAMAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    MAKINAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    BOYAKIMYASALMALIYETI DOUBLE PRECISION,
    TEORIKBOYAKIMYASALMALIYETI DOUBLE PRECISION,
    UNITELERASEVKMALIYETI DOUBLE PRECISION,
    KALITEID VARCHAR(20),
    KALITEADI ADI255,
    SATISTRANSFERKODU VARCHAR(55),
    HAMSTOKGIRISBIRIMFIYAT DOUBLE PRECISION,
    HAMSTOKGIRISDOVIZID VARCHAR(10),
    HAMSTOKGIRISFIRMAKISAADI VARCHAR(10),
    HAMSTOKGIRISLOT VARCHAR(150),
    PARTIDURUMU INTEGER,
    SONYAPILANASAMAADI VARCHAR(40),
    SONYAPILANASAMAID INTEGER,
    SIPARISRENKADI VARCHAR(40))
AS
declare variable KS integer;
declare variable MinHamStokDetayId integer;
declare variable MaxHamStokDetayId integer;
declare variable SonKontrolAsamaId integer;
declare variable IplikSonKontrolAsamaId integer;
declare variable MaxSira  integer;
begin
  Select SonKontrolAsamaId, IplikSonKontrolAsamaId from Sistem Where SistemId = 1 into :SonKontrolAsamaId, :IplikSonKontrolAsamaId;
  if (ATartiKaydiOlmayanlar = 0) then
  begin
    For Select F_StripTime(PartiStokTarti.Tarih), PartiStokTarti.PartiKayitId, Parti.PartiYil,
               Parti.PartiId, PartiStokTarti.PartiSinifId, Parti.FirmaId, Firma.Adi, partiStok.HamAdiToplami,
               Parti.ReceteId, Recete.Adi, LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.MusteriSiparisNo,
               Parti.Fason, PartiStok.Kilo, PartiStok.BobinSayisi, PartiStok.TamamiTartildi, PartiStok.FireOrani,
               ReceteRenkDerinligi.Adi, SipRenkDerinligi.Adi, PartiStokFizikKontrol.CikisNemOrani, PartiStokFizikKontrol.SurtunmeKatsayisi,
               Sum(PartiStokTarti.KapAdedi),  Sum(PartiStokTarti.BobinSayisi), Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara),
               Sum(PartiStokTarti.Kilo), Parti.IslemGrubuId, IslemGrubu.Adi, BoyaSiparisDetay.KaliteID, Kalite.KaliteAdi, Ham.SatisTransferKodu,
               Min(PartiStok.Durumu), BoyaSiparisDetay.RenkAdi from PartiStokTarti
    left join PartiStok on PartiStok.PartiSinifID = PartiStokTarti.PartiSinifId
    left join Ham on Ham.HamID = PartiStok.HamId
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
    left join Recete on Recete.ReceteID = Parti.ReceteId
    left join LabRecete on LabRecete.LabReceteID = Recete.LabReceteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join Firma on Firma.FirmaID = Parti.FirmaId
    left join RenkDerinligi SipRenkDerinligi on SipRenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
    left join RenkDerinligi ReceteRenkDerinligi on ReceteRenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    where PartiStokTarti.Tarih >= :ABaslangicTarihi and PartiStokTarti.Tarih < :ABitisTarihi and
          (Parti.PartiTipiID = 0) and
          (PartiStok.HamID = :AHamID or :AHamId is null) and
          (Parti.FirmaID = :AFirmaID or :AFirmaID is null) and
          (Parti.Fason = :AFason or :AFason is null) and
          (Parti.ReceteID = :AReceteID or :AReceteID is null) and
          (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
          (PartiStokTarti.IadeBoyaIrsaliyeDetayId is null)
    group by F_StripTime(PartiStokTarti.Tarih), PartiStokTarti.PartiKayitId, Parti.PartiYil,
               Parti.PartiId, PartiStokTarti.PartiSinifId, Parti.FirmaId, Firma.Adi, partiStok.HamAdiToplami,
               Parti.ReceteId, Recete.Adi, LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.MusteriSiparisNo,
               Parti.Fason, PartiStok.Kilo, PartiStok.BobinSayisi, PartiStok.TamamiTartildi, PartiStok.FireOrani,
               ReceteRenkDerinligi.Adi, SipRenkDerinligi.Adi, PartiStokFizikKontrol.CikisNemOrani,
               PartiStokFizikKontrol.SurtunmeKatsayisi, Parti.IslemGrubuId, IslemGrubu.Adi, BoyaSiparisDetay.KaliteID, Kalite.KaliteAdi, PartiStok.HamID, Ham.SatisTransferKodu, BoyaSiparisDetay.RenkAdi

    into :Tarih, :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :FirmaId, :FirmaAdi, :HamAdiToplami,
         :ReceteId, :ReceteAdi, :LabReceteKodu, :LabReceteAdi, :MusteriSiparisNo, :Fason, :PartiStokKilo , :PartiStokBobinSayisi, :TamamiTartildi, :FireOrani,
         :ReceteRenkDerinligiAdi, :SiparisRenkDerinligiAdi, :CikisNem, :SurtunmeKatsayisi,
         :PaketlemeKapAdedi, :PaketlemeBobinSayisi, :PaketlemeBrutKilo, :PaketlemeNetKilo, :IslemGrubuId, :IslemGrubuAdi, :KaliteID, :KaliteAdi, :SatisTransferKodu,
         :PartiDurumu, :SiparisRenkAdi do
    begin
      TarihStr = cast(Tarih as varchar(25));
      select max(PartiAsamalari.Sira) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiKayitId = :PartiKayitId and PartiAsamaStok.PartiSinifId = :PartiSinifId and
           (PartiAsamalari.AsamaId <> :SonKontrolAsamaId and PartiAsamalari.AsamaId <> :IplikSonKontrolAsamaId ) and
            PartiAsamaStok.Giristarihi is not null
      into :MaxSira;

      SonYapilanAsamaID  = Null;
      SonYapilanAsamaAdi = Null;

      if(MaxSira is not null) then
      begin
        select first 1 Asama.Adi, Asama.Asamaid from PartiAsamalari
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join Recete on Recete.Receteid = PartiAsamalari.Receteid
        left join Asama on (Asama.Asamaid = PartiAsamalari.AsamaId or Asama.Asamaid = Recete.UstAsamaId)
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.sira = :MaxSira and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
        group By Asama.Adi, Asama.Asamaid into :SonYapilanAsamaAdi, :SonYapilanAsamaID;
      end

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      TamamiRaporlandi = 1;

      Select Count(*) from PartiStokTarti
      where PartiStokTarti.PartiSinifID = :PartiSinifID  and
           (PartiStokTarti.Tarih < :ABaslangicTarihi or PartiStokTarti.Tarih >= :ABitisTarihi)
           into :KS;

      if (KS > 0) then TamamiRaporlandi = 0;

      Select Min(RafDetay.HamStokDetayId), Max(RafDetay.HamStokDetayId) from PartiStokDetay
        left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        where PartiStokDetay.PartiSinifId = :PartiSinifId
        into :MinHamStokDetayId, :MaxHamStokDetayId;

      GirisNem = Null;

      if (MinHamStokDetayId is not null) then
      begin
        Select HamStokMaster.NemOrani from HamStokDetay
        left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
        left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = HamStokMaster.HamStokIrsaliyeId
        where HamStokDetay.HamStokDetayID = :MinHamStokDetayId
        into :GirisNem;
      end

      select first 1 HamStokMaliyeti, AsamaUMUzerindenAsamaMaliyeti, MakinaUMUzerindenAsamaMaliyeti, BoyaKimyasalMaliyeti,
           TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti from PartiSinifMaliyetKumulatif
           where PartiSinifId = :PartiSinifId
           into :HamStokMaliyeti, :AsamaUMUzerindenAsamaMaliyeti, :MakinaUMUzerindenAsamaMaliyeti, :BoyaKimyasalMaliyeti,
           :TeorikBoyaKimyasalMaliyeti, :UnitelerASevkMaliyeti;

      if (PaketlemeBrutKilo <> 0) then
        PaketlemeFire = (PaketlemeBrutKilo - PaketlemeNetKilo) * 100 / PaketlemeBrutKilo;
      else
        PaketlemeFire = 0;

    Select first 1 Firma.KisaAdi, HamStokMaster.Lot, HamStokMaster.BirimFiyat, HamStokIrsaliye.DovizId From PartiStokDetay
      Left Join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      Left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      Left Join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      Left Join Firma on Firma.FirmaId = HamStokMaster.FirmaId
      Left Join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
        Where PartiStokDetay.PartiSinifId = :PartiSinifId
      order by PartiStokDetay.Kilo desc
      Into :HamStokGirisFirmaKisaAdi, :HamStokGirisLot, :HamStokGirisBirimFiyat, :HamStokGirisDovizId ;
      suspend;
    end
  end
  else /*iplik paketlemesi olup tarti kaydi olmayanlar*/
  begin
    For Select F_StripTime(IplikPaketleme.PaketlemeTarihi), IplikPaketleme.PartiKayitId, Parti.PartiYil,
               Parti.PartiId, IplikPaketleme.PartiSinifId, Parti.FirmaId, Firma.Adi, PartiStok.HamAdiToplami,
               Parti.ReceteId, Recete.Adi, LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.MusteriSiparisNo,
               Parti.Fason, PartiStok.Kilo, PartiStok.BobinSayisi, PartiStok.TamamiTartildi, PartiStok.FireOrani,
               ReceteRenkDerinligi.Adi, SipRenkDerinligi.Adi, PartiStokFizikKontrol.CikisNemOrani, PartiStokFizikKontrol.SurtunmeKatsayisi,
               Null, Sum(IplikPaketleme.BobinSayisi), Sum(IplikPaketleme.NetKilo + IplikPaketleme.Dara),
               Sum(IplikPaketleme.NetKilo), Parti.IslemGrubuId, IslemGrubu.Adi, BoyaSiparisDetay.KaliteID, Kalite.KaliteAdi, Ham.SatisTransferKodu,
               Min(PartiStok.Durumu), BoyaSiparisDetay.RenkAdi  from IplikPaketleme
    left join PartiStok on PartiStok.PartiSinifID = IplikPaketleme.PartiSinifId
    left join Ham on Ham.HamID = PartiStok.HamId
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
    left join Recete on Recete.ReceteID = Parti.ReceteId
    left join LabRecete on LabRecete.LabReceteID = Recete.LabReceteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join Firma on Firma.FirmaID = Parti.FirmaId
    left join RenkDerinligi SipRenkDerinligi on SipRenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
    left join RenkDerinligi ReceteRenkDerinligi on ReceteRenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    where IplikPaketleme.PartiStokTartiId is null and
          IplikPaketleme.PaketlemeTarihi >= :ABaslangicTarihi and IplikPaketleme.PaketlemeTarihi < :ABitisTarihi and
          (Parti.PartiTipiID = 0) and
          (PartiStok.HamID = :AHamID or :AHamId is null) and
          (Parti.FirmaID = :AFirmaID or :AFirmaID is null) and
          (Parti.Fason = :AFason or :AFason is null) and
          (Parti.ReceteID = :AReceteID or :AReceteID is null) and
          (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null)
    group by F_StripTime(IplikPaketleme.PaketlemeTarihi), IplikPaketleme.PartiKayitId, Parti.PartiYil,
               Parti.PartiId, IplikPaketleme.PartiSinifId, Parti.FirmaId, Firma.Adi, PartiStok.HamAdiToplami,
               Parti.ReceteId, Recete.Adi, LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.MusteriSiparisNo,
               Parti.Fason, PartiStok.Kilo, PartiStok.BobinSayisi, PartiStok.TamamiTartildi, PartiStok.FireOrani,
               ReceteRenkDerinligi.Adi, SipRenkDerinligi.Adi, PartiStokFizikKontrol.CikisNemOrani,
               PartiStokFizikKontrol.SurtunmeKatsayisi, Parti.IslemGrubuId, IslemGrubu.Adi, BoyaSiparisDetay.KaliteID, Kalite.KaliteAdi, PartiStok.HamID, Ham.SatisTransferKodu, BoyaSiparisDetay.RenkAdi

    into :Tarih, :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :FirmaId, :FirmaAdi, :HamAdiToplami,
         :ReceteId, :ReceteAdi, :LabReceteKodu, :LabReceteAdi, :MusteriSiparisNo, :Fason, :PartiStokKilo , :PartiStokBobinSayisi, :TamamiTartildi, :FireOrani,
         :ReceteRenkDerinligiAdi, :SiparisRenkDerinligiAdi, :CikisNem, :SurtunmeKatsayisi,
         :PaketlemeKapAdedi, :PaketlemeBobinSayisi, :PaketlemeBrutKilo, :PaketlemeNetKilo, :IslemGrubuId, :IslemGrubuAdi, :KaliteID, :KaliteAdi, :SatisTransferKodu,
         :PartiDurumu, :SiparisRenkAdi do
    begin
      TarihStr = cast(Tarih as varchar(25));

      select max(PartiAsamalari.Sira) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiKayitId = :PartiKayitId and PartiAsamaStok.PartiSinifId = :PartiSinifId and
           (PartiAsamalari.AsamaId <> :SonKontrolAsamaId and PartiAsamalari.AsamaId <> :IplikSonKontrolAsamaId ) and
            PartiAsamaStok.Giristarihi is not null
      into :MaxSira;

      SonYapilanAsamaID  = Null;
      SonYapilanAsamaAdi = Null;

      if(MaxSira is not null) then
      begin
        select first 1 Asama.Adi, Asama.Asamaid from PartiAsamalari
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join Recete on Recete.Receteid = PartiAsamalari.Receteid
        left join Asama on (Asama.Asamaid = PartiAsamalari.AsamaId or Asama.Asamaid = Recete.UstAsamaId)
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.sira = :MaxSira and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
        group By Asama.Adi, Asama.Asamaid into :SonYapilanAsamaAdi, :SonYapilanAsamaID;
      end

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      TamamiRaporlandi = 1;

      Select Count(*) from IplikPaketleme
      where IplikPaketleme.PartiSinifID = :PartiSinifID  and
           (IplikPaketleme.PaketlemeTarihi < :ABaslangicTarihi or IplikPaketleme.PaketlemeTarihi >= :ABitisTarihi)
           into :KS;

      if (KS > 0) then TamamiRaporlandi = 0;

      Select Min(RafDetay.HamStokDetayId), Max(RafDetay.HamStokDetayId) from PartiStokDetay
        left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        where PartiStokDetay.PartiSinifId = :PartiSinifId
        into :MinHamStokDetayId, :MaxHamStokDetayId;

      GirisNem = Null;

      if (MinHamStokDetayId is not null) then
      begin
        Select HamStokMaster.NemOrani from HamStokDetay
        left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
        left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = HamStokMaster.HamStokIrsaliyeId
        where HamStokDetay.HamStokDetayID = :MinHamStokDetayId
        into :GirisNem;
      end

      select first 1 HamStokMaliyeti, AsamaUMUzerindenAsamaMaliyeti, MakinaUMUzerindenAsamaMaliyeti, BoyaKimyasalMaliyeti,
           TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti from PartiSinifMaliyetKumulatif
           where PartiSinifId = :PartiSinifId
           into :HamStokMaliyeti, :AsamaUMUzerindenAsamaMaliyeti, :MakinaUMUzerindenAsamaMaliyeti, :BoyaKimyasalMaliyeti,
           :TeorikBoyaKimyasalMaliyeti, :UnitelerASevkMaliyeti;

      if (PaketlemeBrutKilo <> 0) then
        PaketlemeFire = (PaketlemeBrutKilo - PaketlemeNetKilo) * 100 / PaketlemeBrutKilo;
      else
        PaketlemeFire = 0;

    Select first 1 Firma.KisaAdi, HamStokMaster.Lot, HamStokMaster.BirimFiyat, HamStokIrsaliye.DovizId From PartiStokDetay
      Left Join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      Left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      Left Join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      Left Join Firma on Firma.FirmaId = HamStokMaster.FirmaId
      Left Join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
        Where PartiStokDetay.PartiSinifId = :PartiSinifId
      order by PartiStokDetay.Kilo desc
      Into :HamStokGirisFirmaKisaAdi, :HamStokGirisLot, :HamStokGirisBirimFiyat, :HamStokGirisDovizId ;
      suspend;
    end
