CREATE PROCEDURE STOKDUSUMDUZENLE(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
AS
declare variable PartiKayitId     integer;
declare variable PartiAsamalariId integer;
begin
  For Select PartiAsamalari.PartiKayitId, PartiAsamalari.PartiAsamalariId from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariID
    Where PartiAsamaStok.CikisTarihi >= :ABaslangicTarihi and PartiAsamaStok.CikisTarihi < :ABitisTarihi and
          PartiAsamalari.ReceteID is not null and PartiAsamalari.StoktanDusuldu = 1
    group by PartiAsamalari.PartiKayitId, PartiAsamalari.PartiAsamalariId
    into :PartiKayitId, :PartiAsamalariId do
  begin

    Execute Procedure StoktanDusmeKarariVer(:PartiKayitId, :PartiAsamalariId, 3, 1);
  end
end
