CREATE PROCEDURE STOKSAYIMFARKINISIFIRLA(
    ACIKISSTOKID INTEGER,
    AGIRISSTOKID INTEGER,
    AURUNID VARCHAR(20),
    AURUNBIRIMID INTEGER,
    AMIKTAR DOUBLE PRECISION,
    ADEPOID VARCHAR(10),
    AKULLANICIID INTEGER,
    ALOT VARCHAR(150),
    ATARIH TIMESTAMP)
RETURNS (
    CIKISSTOKID INTEGER,
    GIRISSTOKID INTEGER)
AS
declare variable FStokTabanDovizId varchar(5);
declare variable FHareketTipi      integer;
declare variable FStokId           integer;
declare variable FFisSeriNo        varchar(2);
declare variable FFisNo            integer;
declare variable FFirmaId          FIRMAKODUSIZE;
begin
  FHareketTipi = 6; --shtUretimeCikis;
  if (AMiktar > 0) then FHareketTipi = 1; --shtAlis;
  AMiktar = ABS(AMiktar);

  Select FirmaKodumuz From Sistem where SISTEMID = 1 into :FFirmaId;

  if (FHareketTipi = 6 and ACikisStokId = -1) then
  begin
    select StokTabanDovizId from SISTEM where SISTEMID = 1 into :FStokTabanDovizId;
    Select FisSeriNo from Kullanici where KullaniciId = :AKullaniciId into :FFisSeriNo;
    if (FFisSeriNo is null) then FFisSeriNo = '';
    select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;

    ACikisStokId = gen_id(Stok_Id_Gen, 1);

    insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, KullaniciId, DengelemeEvraki, FirmaId)
                  values(:ACikisStokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :ATarih, :FStokTabanDovizId, 1, 1, :AKullaniciId, 1, :FFirmaId);
  end
  else if (FHareketTipi = 1 and AGirisStokId = -1) then
  begin
    select StokTabanDovizId from SISTEM where SISTEMID = 1 into :FStokTabanDovizId;
    Select FisSeriNo from Kullanici where KullaniciId = :AKullaniciId into :FFisSeriNo;
    if (FFisSeriNo is null) then FFisSeriNo = '';
    select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;

    AGirisStokId = gen_id(Stok_Id_Gen, 1);

    insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, KullaniciId, DengelemeEvraki, FirmaId)
                  values(:AGirisStokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :ATarih, :FStokTabanDovizId, 1, 1, :AKullaniciId, 1, :FFirmaId);
  end

  if(:ALot = '') then ALot = null;

  if (FHareketTipi = 1) then FStokId = AGirisStokId;
  else  FStokId = ACikisStokId;

  insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, UrunId, UrunBirimId, Miktar, Otomasyon, KullaniciId, Lot, FirmaId)
                 values(:ATarih, :FHareketTipi, :FStokId, :ADepoId, :AUrunId, :AUrunBirimId, :AMiktar, 0, :AKullaniciId, :ALot, :FFirmaId);

  GirisStokId = AGirisStokId;
  CikisStokId = ACikisStokId;
  suspend;
end
