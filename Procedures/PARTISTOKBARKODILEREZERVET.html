CREATE PROCEDURE PARTISTOKBARKODILEREZERVET(
    ABOYAREZERVFISIDETAYID INTEGER,
    ABARKODID VARCHAR(20),
    AKULLANICIID INTEGER)
AS
declare variable PartiKayitId integer;
declare variable PartiSinifId integer;
declare variable DepoId varchar(10);
declare variable PartiStokTartiId integer;
declare variable BoyaRezervDetayId integer;
declare variable HamStokSarfFisiDetayId integer;
declare variable BoyaIrsaliyeDetayId integer;
declare variable PaketId integer;
declare variable IK     varchar(2);
declare variable TF     integer;
declare variable PartiStokTartiKilo     BoyaMiktarSize;
declare variable PartiStokTartiBrutKilo BoyaMiktarSize;
declare variable RezervPartiSinifId integer;
declare variable FBoyaRezervFisiID integer;
declare variable FBarkodID          varchar(30);
declare variable IadeBoyaIrsaliyeId Integer;
declare variable KS Integer;
declare variable TartiRafID  RafKoduSize;
declare variable RezervRafID RafKoduSize;
begin
  if (Char_Length(ABarkodId) < 3) then exception BarkodHatali1;

  FBarkodID = ABarkodId;
  IK = EgemenCopyStr(ABarkodId, 1, 2);

  ABarkodId = EgemenCopyStr(ABarkodId, 3, Char_Length(ABarkodId) - 2);
  TF = EgemenStrToIntOlurmu(ABarkodId);

  if (TF = 0) then exception BarkodHatali1;

  select count(*), Max(PartiStokTartiId) from PartiStokTarti where BarkodId = :FBarkodId into :KS, :PaketId;
  if (KS > 0) then /*iade kaydidir*/
    IK = 'R-';
  else
    PaketId = EgemenStrToInt(ABarkodId);

  TartiRafId = Null;

  if (IK = 'K-') then
  begin
    Select PsKaliteKontrol.PartiKayitId, PsKaliteKontrol.PartiSinifId, PsKaliteKontrol.PartiStokTartiId, PartiStokTarti.DepoId,
           PartiStokTarti.BoyaRezervFisiDetayId, PartiStokTarti.HamStokSarfFisiDetayId, PartiStokTarti.BoyaIrsaliyeDetayId from PsKaliteKontrol
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = PsKaliteKontrol.PartiStokTartiId
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where PsKaliteKontrol.PsKaliteKontrolId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :DepoId, :BoyaRezervDetayId, :HamStokSarfFisiDetayId, :BoyaIrsaliyeDetayId;
  end
  else
  if (IK = 'R-') then /*iade kaydi*/
  begin
    Select PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId, PartiStokTarti.DepoId,
           PartiStokTarti.BoyaRezervFisiDetayId, PartiStokTarti.HamStokSarfFisiDetayId, PartiStokTarti.BoyaIrsaliyeDetayId, MamulDepoTransferFisiDetay.RafId from PartiStokTarti
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId and MamulDepoTransferFisiDetay.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
      where PartiStokTarti.PartiStokTartiId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :DepoId, :BoyaRezervDetayId, :HamStokSarfFisiDetayId, :BoyaIrsaliyeDetayId, :TartiRafId;
  end
  else
  begin
    Select IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, IplikPaketleme.PartiStokTartiId, PartiStokTarti.DepoId,
           PartiStokTarti.BoyaRezervFisiDetayId, PartiStokTarti.HamStokSarfFisiDetayId, PartiStokTarti.BoyaIrsaliyeDetayId, MamulDepoTransferFisiDetay.RafId from IplikPaketleme
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiId
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId and MamulDepoTransferFisiDetay.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
      where IplikPaketleme.IplikPaketlemeId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :DepoId, :BoyaRezervDetayId, :HamStokSarfFisiDetayId, :BoyaIrsaliyeDetayId, :TartiRafId;
  end

  if (PartiSinifId is null) then exception BarkodHatali2;
  if (PartiStokTartiId is null) then exception TartiKaydiYokRezerveEdilemez;
  if (BoyaIrsaliyeDetayId is not Null) then exception BuPaketinIrsaliyesiKesilmis;

  /* Emre kapatti, yarim cuval iade mantigi bunu kapsadi
  if (BoyaIrsaliyeDetayId is not null) then
  begin
    Select PartiStokTartiId, BoyaIrsaliyeDetayId, BoyaRezervFisiDetayId, PartiStokTarti.DepoId, PartiStokTarti.IadeBoyaIrsaliyeId from PartiStokTarti
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where PartiStokTarti.BarkodId = :FBarkodID and
            PartiStokTarti.BoyaIrsaliyeDetayId is Null and
            PartiStokTarti.HamStokSarfFisiDetayId is Null and
            PartiStokTarti.IadeBoyaIrsaliyeDetayId is not null
    into :PartiStokTartiId, :BoyaIrsaliyeDetayId, :BoyaRezervDetayId, :DepoId, :IadeBoyaIrsaliyeId;

    if (BoyaIrsaliyeDetayId is not null) then exception BuPaketinIrsaliyesiKesilmis;
  end*/
  if (BoyaRezervDetayId is not null) then exception BuPaketRezerveEdilmis;
  if (HamStokSarfFisiDetayId is not null) then exception BuPaketSarFisiIleCikmis;

  select PartiSinifId, BoyaRezervFisiID, BoyaRezervFisiDetay.RafId from BoyaRezervFisiDetay
  where BoyaRezervFisiDetayId = :ABoyaRezervFisiDetayId
  into :RezervPartiSinifId, :FBoyaRezervFisiID, :RezervRafId;

  /*if (RezervRafId is not null) then
    if (TartiRafId <> RezervRafId) then
      exception KayitHatasi 'Çuval Rezerv Rafında Değil. (Çuval Rafı:'||TartiRafId||')'; */

  if (RezervPartiSinifId <> PartiSinifId or RezervRafId <> TartiRafId) then
  begin
    ABoyaRezervFisiDetayId = Null;
    Select BoyaRezervFisiDetayID from BoyaRezervFisiDetay where BoyaRezervFisiDetay.BoyaRezervFisiId = :FBoyaRezervFisiID and
    BoyaRezervFisiDetay.PartiSinifId = :PartiSinifId and BoyaRezervFisiDetay.RafId = :TartiRafId and Tamamlandi = 0 into :ABoyaRezervFisiDetayId;
    if (ABoyaRezervFisiDetayId is null) then exception TartiKaydiBuPartiyeAitDegil;
  end

  update PartiStokTarti set PartiStokTarti.BoyaRezervFisiDetayId = :ABoyaRezervFisiDetayId, PartiStokTarti.KullaniciId = :AKullaniciId, PartiStokTarti.DegisimTarihi = current_timestamp(0) where PartiStokTarti.PartiStokTartiId = :PartiStokTartiId;
end
