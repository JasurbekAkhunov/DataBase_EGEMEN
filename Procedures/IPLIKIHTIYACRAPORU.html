CREATE PROCEDURE IPLIKIHTIYACRAPORU(
    AIPLIKLERIDOK INTEGER,
    ALYCRALARIDOK INTEGER,
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    ATARIHEGORESIPRECETETOPLAMI INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    ARAPORTIPI INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    STOKMIKTARI DOUBLE PRECISION,
    ACIKSIPARISTOPLAMI DOUBLE PRECISION,
    ACIKSIPARISRECETET DOUBLE PRECISION,
    ACIKTALIMATRECETET DOUBLE PRECISION,
    ACIKSTALIMATCIKIST DOUBLE PRECISION,
    ACIKSIPARISEKSIGI DOUBLE PRECISION,
    TEORIKSTOKEKSIGI DOUBLE PRECISION,
    MAKINAACIKTALIMATRECETET DOUBLE PRECISION,
    FASONACIKTALIMATRECETET DOUBLE PRECISION,
    FASONACIKTALRECETEKULLANILMIST DOUBLE PRECISION,
    MAKINAACIKTALRECETEKULLANILMIST DOUBLE PRECISION,
    ACIKTALRECETEKULLANILMIST DOUBLE PRECISION,
    ACIKSTALIMATFASONCIKIST DOUBLE PRECISION,
    FASONTEORIKSTOKEKSIGI DOUBLE PRECISION,
    VERILENSIPARISTOPLAMI DOUBLE PRECISION,
    VERILENSIPARISEKSIGI DOUBLE PRECISION,
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIID INTEGER,
    HAMGRUP1 VARCHAR(4),
    HAMGRUP2 VARCHAR(4),
    HAMGRUP3 VARCHAR(4),
    HAMGRUP4 VARCHAR(4))
AS
declare variable hstIplik       integer;
declare variable hstLycra       integer;
declare variable AstIplik               integer;
declare variable astBoyaliIplik         integer;
declare variable astLycra               integer;
declare variable astHamKumas            integer;
declare variable astBoyaliKumas         integer;
declare variable astIplikBoyaliKumas    integer;
declare variable UstHamId       integer;
declare variable KullanimOrani  Double Precision;
declare variable FireOrani      Double Precision;
declare variable ReceteMiktar   Double Precision;
declare variable shtUretimeCikis integer;
declare variable assSatis          integer;
declare variable assFasonArtiSatis integer;
declare variable assFason          integer;
declare variable KaliteRecetesiniKullan integer;
declare variable UstKaliteId         KaliteKoduSize;
declare variable shtFasonUretimeCikis integer;
begin
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  select KaliteRecetesiniKullan from Sistem into KaliteRecetesiniKullan;

  Select astIplik, astBoyaliIplik, AstLycra, AstHamKumas, AstBoyaliKumas, AstIplikBoyaliKumas from OrmeSiparisTipiGetir
    into :astIplik, :astBoyaliIplik, :AstLycra, :AstHamKumas, :AstBoyaliKumas, :AstIplikBoyaliKumas;

  Select assSatis, assFasonArtiSatis, assFason from OrmeSiparisSekliGetir into :assSatis, :assFasonArtiSatis, :assFason;
  select shtUretimeCikis, shtFasonUretimeCikis from OrmeStokHareketTipiGetir into :shtUretimeCikis, :shtFasonUretimeCikis;

  for Select Ham.HamId, Ham.Adi, HamTipiID, Ham.IplikCinsiId,Ham.HamGrup1,Ham.HamGrup2 ,Ham.HamGrup3 ,Ham.HamGrup4, Kalinlik.Adi From Ham
  left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
    where (HamTipiID = :hstIplik or HamTipiID = :hstLycra) and
    (:AIplikleriDok = 1 or HamTipiID <> :hstIplik) and
    (:ALycralariDok = 1 or HamTipiID <> :hstLycra)
    into :HamID, : HamAdi, :HamTipiID, :IplikCinsiId, :HamGrup1, :HamGrup2 , :HamGrup3 , :HamGrup4, :KalinlikAdi do
  begin
    Select Sum(OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari) from OrmeRaf
      --left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
      where OrmeRaf.HamID = :HamID and OrmeRaf.MiktarBirim = 0 and
            ((:AFasonSiparisler = 1 and OrmeRaf.FasonStok = 1) or
             (:ASatisSiparisler = 1 and OrmeRaf.FasonStok = 0 ))
           /*  Ali Revize Mark Formelle eb-2116
            ((BoyaSiparis.OrmeSiparisTipiId = :astIplik) or
            (BoyaSiparis.OrmeSiparisTipiId = :astBoyaliIplik) or
            (BoyaSiparis.OrmeSiparisTipiId = :AstLycra)) and

            (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1)*/
      into :StokMiktari;
    if (StokMiktari is Null) then StokMiktari = 0;


    select sum(OrmeSiparisDetay.HamMiktar) from OrmeSiparisDetay
      left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
      --Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
      where (OrmeSiparisDetay.HamId = :HamId) and
            (OrmeSiparisDetay.Kapali = 0) and
            ((BoyaSiparis.OrmeSiparisTipiId = :astIplik) or
            (BoyaSiparis.OrmeSiparisTipiId = :astBoyaliIplik) or
            (BoyaSiparis.OrmeSiparisTipiId = :AstLycra)) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1)
      into :AcikSiparisToplami;
    if (AcikSiparisToplami is Null) then AcikSiparisToplami = 0;

    if (KaliteRecetesiniKullan = 1) then
    begin
      AcikSiparisReceteT = 0;
      for select KaliteId, KullanimOrani, FireOrani from KaliteRecete
        where ReceteHamId = :HamId and FireOrani < 100
        into :UstKaliteId, :KullanimOrani, :FireOrani do
      begin
        select sum( (OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar - OrmeSiparisDetay.ToplamOrulen) * (:KullanimOrani / 100.00) / ((100.00 - :FireOrani) / 100.00) )  from OrmeSiparisDetay
          left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
          --Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
          where (OrmeSiparisDetay.KaliteId = :UstKaliteId) and
                (OrmeSiparisDetay.Kapali = 0) and
                /*tarih filtresi gir parametreye göre*/
                ((:ATariheGoreSipReceteToplami = 0) or
                (:ATariheGoreSipReceteToplami = 1 and BoyaSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and BoyaSiparis.SiparisTarihi <= :ABitisSiparisTarihi ))   and
                ((BoyaSiparis.OrmeSiparisTipiId = :AstHamKumas) or
                (BoyaSiparis.OrmeSiparisTipiId = :AstBoyaliKumas) or
                (BoyaSiparis.OrmeSiparisTipiId = :AstIplikBoyaliKumas)) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1) and
                (OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar - OrmeSiparisDetay.ToplamOrulen) * (:KullanimOrani / 100.00) / ((100.00 - :FireOrani) / 100.00)  >0 -- can ekledi

        into :ReceteMiktar;
        if (ReceteMiktar is Null) then ReceteMiktar = 0;
        AcikSiparisReceteT = AcikSiparisReceteT + ReceteMiktar;
      end
      AcikSiparisToplami = AcikSiparisToplami + AcikSiparisReceteT;

    end
    else
    begin
      AcikSiparisReceteT = 0;
      for select HamId, KullanimOrani, FireOrani from HamStokRecete
        where ReceteHamId = :HamId and FireOrani < 100
        into :UstHamId, :KullanimOrani, :FireOrani do
      begin
        select sum( (OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar - OrmeSiparisDetay.ToplamOrulen) * (:KullanimOrani / 100.00) / ((100.00 - :FireOrani) / 100.00) )  from OrmeSiparisDetay
          left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
          --Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
          where (OrmeSiparisDetay.HamId = :UstHamId) and
                (OrmeSiparisDetay.Kapali = 0) and
                /*tarih filtresi gir parametreye göre*/
                ((:ATariheGoreSipReceteToplami = 0) or
                (:ATariheGoreSipReceteToplami = 1 and BoyaSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and BoyaSiparis.SiparisTarihi <= :ABitisSiparisTarihi ))   and
                ((BoyaSiparis.OrmeSiparisTipiId = :AstHamKumas) or
                (BoyaSiparis.OrmeSiparisTipiId = :AstBoyaliKumas) or
                (BoyaSiparis.OrmeSiparisTipiId = :AstIplikBoyaliKumas)) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
                (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1)  and
                (OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar - OrmeSiparisDetay.ToplamOrulen) * (:KullanimOrani / 100.00) / ((100.00 - :FireOrani) / 100.00)  >0 -- can ekledi
        into :ReceteMiktar;
        if (ReceteMiktar is Null) then ReceteMiktar = 0;
        AcikSiparisReceteT = AcikSiparisReceteT + ReceteMiktar;
      end
      AcikSiparisToplami = AcikSiparisToplami + AcikSiparisReceteT;
    end
    MakinaAcikTalimatReceteT =  0;
    FasonAcikTalimatReceteT  =  0;
    MakinaAcikTalReceteKullanilmisT =  0;
    FasonAcikTalReceteKullanilmisT  =  0;

    select sum( (OrmePlanlama.PlanlananMiktar - OrmePlanlama.GerceklesenMiktar + OrmePlanlama.ToplamSakatOrulen) * (OrmePlanlamaRecete.KullanimOrani / 100.00) / ((100.00 - OrmePlanlamaRecete.TahminiFireOrani) / 100.00) ),
           sum(OrmePlanlama.GerceklesenMiktar * OrmePlanlamaRecete.KullanimOrani / 100)
       from OrmePlanlamaRecete
      left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmePlanlamaRecete.OrmePlanlamaId
      left join OrmeSiparisDetay On OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
      left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
      where (OrmePlanlamaRecete.HamId = :HamId) and
            (OrmeSiparisDetay.Kapali = 0)  and
            (OrmePlanlama.Bitti = 0)  and
            (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1) and
             (((:ARaporTipi is null)  and
      (OrmePlanlama.OrmeMakinaId is not null) )  or  :ARaporTipi = 0) and
             (OrmePlanlama.OrmeMakinaId is not null) and
             OrmePlanlamaRecete.TahminiFireOrani < 100
      into :MakinaAcikTalimatReceteT, :MakinaAcikTalReceteKullanilmisT;
    if (MakinaAcikTalimatReceteT is Null) then MakinaAcikTalimatReceteT = 0;
    if (MakinaAcikTalReceteKullanilmisT is Null) then MakinaAcikTalReceteKullanilmisT = 0;

    select sum( (OrmePlanlama.PlanlananMiktar - OrmePlanlama.GerceklesenMiktar + OrmePlanlama.ToplamSakatOrulen) * (OrmePlanlamaRecete.KullanimOrani / 100.00) / ((100.00 - OrmePlanlamaRecete.TahminiFireOrani) / 100.00) ),
           sum(OrmePlanlama.GerceklesenMiktar * OrmePlanlamaRecete.KullanimOrani / 100)
       from OrmePlanlamaRecete
      left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmePlanlamaRecete.OrmePlanlamaId
      left join OrmeSiparisDetay On OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
      left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
      where (OrmePlanlamaRecete.HamId = :HamId) and
            (OrmeSiparisDetay.Kapali = 0)  and
            (OrmePlanlama.Bitti = 0)  and
            (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1) and
             (((:ARaporTipi is null)  AND (OrmePlanlama.FirmaId is not null) )  or  :ARaporTipi = 1) and
             (OrmePlanlama.FirmaId is not null) and
             OrmePlanlamaRecete.TahminiFireOrani < 100
      into :FasonAcikTalimatReceteT, :FasonAcikTalReceteKullanilmisT;
    if (FasonAcikTalimatReceteT is Null) then FasonAcikTalimatReceteT = 0;
    if (FasonAcikTalReceteKullanilmisT is Null) then FasonAcikTalReceteKullanilmisT = 0;



    AcikTalimatReceteT        =  MakinaAcikTalReceteKullanilmisT +  FasonAcikTalimatReceteT;
    AcikTalReceteKullanilmisT =  MakinaAcikTalReceteKullanilmisT +  FasonAcikTalReceteKullanilmisT ;

    select sum(StokHareket.Miktar) from StokHareket
      left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = StokHareket.OrmePlanlamaId
      left join OrmeSiparisDetay On OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
      left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
      --Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
      where (StokHareket.OrmePlanlamaId is not Null) and
            (StokHareket.HareketTipi = :shtUretimeCikis) and
            (StokHareket.HamId = :HamId) and
            (OrmeSiparisDetay.Kapali = 0)  and
            (OrmePlanlama.Bitti = 0) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1) and
            (:ARaporTipi is null or
             (OrmePlanlama.OrmeMakinaId is not null and :ARaporTipi = 0) or
             (OrmePlanlama.FirmaId is not null and :ARaporTipi = 1) )
      into :AcikSTalimatCikisT;
    if (AcikSTalimatCikisT is Null) then AcikSTalimatCikisT = 0;


    select sum(StokHareket.Miktar) from StokHareket
      left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = StokHareket.OrmePlanlamaId
      left join OrmeSiparisDetay On OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
      left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
      --Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
      where (StokHareket.OrmePlanlamaId is not Null) and
            (StokHareket.HareketTipi = :shtFasonUretimeCikis) and
            (StokHareket.HamId = :HamId) and
            (OrmeSiparisDetay.Kapali = 0)  and
            (OrmePlanlama.Bitti = 0) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assSatis or :ASatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFasonArtiSatis or :AFasonSatisSiparisler = 1) and
            (BoyaSiparis.OrmeSiparisSekliId <> :assFason or :AFasonSiparisler = 1) and
            (:ARaporTipi is null or
             (OrmePlanlama.OrmeMakinaId is not null and :ARaporTipi = 0) or
             (OrmePlanlama.FirmaId is not null and :ARaporTipi = 1) )
      into :AcikSTalimatFasonCikisT;
    if (AcikSTalimatFasonCikisT is Null) then AcikSTalimatFasonCikisT = 0;


    AcikSiparisEksigi     = AcikSiparisToplami ;-- AcikSTalimatCikisT;
    TeorikStokEksigi      = AcikSiparisEksigi - StokMiktari;
    FasonTeorikStokEksigi = AcikSiparisEksigi - (StokMiktari + (AcikSTalimatFasonCikisT - FasonAcikTalReceteKullanilmisT));

    select sum(VerilenSiparisDetay.Miktar - VerilenSiparisDetay.GelenMiktar + VerilenSiparisDetay.IadeEdilenMiktar) from VerilenSiparisDetay
      where (VerilenSiparisDetay.HamId = :HamId) and
            (VerilenSiparisDetay.Kapali = 0) and
             VerilenSiparisDetay.Miktar - VerilenSiparisDetay.GelenMiktar + VerilenSiparisDetay.IadeEdilenMiktar > 0
      into :VerilenSiparisToplami;
    if (VerilenSiparisToplami is Null) then VerilenSiparisToplami = 0;

    VerilenSiparisEksigi = TeorikStokEksigi - VerilenSiparisToplami;

    if (:ARaporTipi is null) then
      suspend;
    else
      if ((AcikTalimatReceteT <> 0) or (AcikSTalimatCikisT <> 0)) then
        suspend;
  end
end
