CREATE PROCEDURE PARTISABLONRECETEILAVEGETIR(
    APARTISABLONID INTEGER)
RETURNS (
    PARTISABLONID INTEGER,
    PARTISABLONILAVEID INTEGER,
    PARTISABLONRECETEILAVEID INTEGER,
    SABLONNO INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BARKOD VARCHAR(255),
    URUNBIRIMID INTEGER,
    BIRIMID VARCHAR(2),
    MIKTAR DOUBLE PRECISION,
    BIRIMKULLANILANMIKTAR DOUBLE PRECISION,
    KULLANILANMIKTAR DOUBLE PRECISION,
    URETIMECIKISFIYATI DOUBLE PRECISION,
    URETIMECIKISFIYATIBUGUN DOUBLE PRECISION,
    URETIMECIKISFIYATIBIRIMID VARCHAR(2),
    STOKMIKTARI NUMERIC(12,3),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable DesenVaryantId        integer;
declare variable SablonHazirlananMiktar Double Precision;
declare variable AcilisTarihi           TimeStamp;
declare variable DovizKuru                   Double Precision;
declare variable FUretimeCikisFiyati   Double precision;
declare variable FUretimeCikisFiyatiDovizId  varchar(5);
declare variable StokTabanDovizId               varchar(5);
begin
  Select DesenVaryantID, AcilisTarihi from PartiSablon
  left join Parti on Parti.PartiKayitId = PartiSablon.PartiKayitId
  where PartiSablon.PartiSablonID = :APartiSablonId into :DesenVaryantID, :AcilisTarihi;

  select StokTabanDovizId from SISTEM where SISTEMID = 1 into :StokTabanDovizId;

  for select PartiSablonReceteIlave.PartiSablonReceteIlaveId, PartiSablonReceteIlave.PartiSablonIlaveId, PartiSablonIlave.PartiSablonId, PartiSablonReceteIlave.UrunId, Urun.Adi,
             PartiSablonReceteIlave.UrunBirimId, PSRUB.BirimId, PartiSablonReceteIlave.Miktar, PartiSablonReceteIlave.KullanilanMiktar,
             PartiSablonIlave.HazirlananMiktar,
             UrunFiyat.UretimeCikisFiyati, UrunFiyat.UretimeCikisFiyatiDovizId, UBUCF.BirimId,
             IK.Adi, PartiSablonReceteIlave.InsertTarihi, SK.Adi, PartiSablonReceteIlave.DegisimTarihi, PartiSablonIlave.SablonNo, Urun.Barkod from PartiSablonReceteIlave
  left join PartiSablonIlave on PartiSablonIlave.PartiSablonIlaveID = PartiSablonReceteIlave.PartiSablonIlaveID
  left join DesenVaryantSablon on DesenVaryantSablon.DesenVaryantID = :DesenvaryantId and DesenVaryantSablon.SablonNo = PartiSablonIlave.SablonNo
  left join Urun on Urun.UrunId = PartiSablonReceteIlave.UrunId
  left join UrunBirim PSRUB on PSRUB.UrunBirimId = PartiSablonReceteIlave.UrunBirimId and PSRUB.UrunId = PartiSablonReceteIlave.UrunId
  left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
  left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
  left join Kullanici IK on IK.KullaniciId = PartiSablonReceteIlave.KullaniciId
  left join Kullanici SK on SK.KullaniciId = PartiSablonReceteIlave.KullaniciId
  where PartiSablonIlave.PartiSablonId = :APartiSablonId
  order by PartiSablonReceteIlave.PartiKayitId, PartiSablonIlave.SablonNo, Urun.UrunTipiId
  into :PartiSablonReceteIlaveId, :PartiSablonIlaveId, :PartiSablonId, :UrunId, :UrunAdi, :UrunBirimId, :BirimId, :Miktar, :KullanilanMiktar, :SablonHazirlananMiktar,
       :FUretimeCikisFiyati, :FUretimeCikisFiyatiDovizId, :UretimeCikisFiyatiBirimId,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :SablonNo, :Barkod do
  begin
    if (SablonHazirlananMiktar <> 0) then
      BirimKullanilanMiktar = KullanilanMiktar / SablonHazirlananMiktar;
    else
      BirimKullanilanMiktar = 0;

    if (FUretimeCikisFiyatiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :StokTabanDovizId, Current_Date) into :DovizKuru;
    else DovizKuru = 1;

    if (DovizKuru is null) then DovizKuru = 0;
    UretimeCikisFiyatiBugun = FUretimeCikisFiyati * DovizKuru;

    if (FUretimeCikisFiyatiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :StokTabanDovizId, :AcilisTarihi) into :DovizKuru;
    else DovizKuru = 1;

    if (DovizKuru is null) then DovizKuru = 0;
    UretimeCikisFiyati = FUretimeCikisFiyati * DovizKuru;

    StokMiktari = Null;
    Select sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from stokdetay
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      where Stokdetay.Urunid = :UrunId and StokDetay.HareketTipi <> 5
    into :StokMiktari;

    if (StokMiktari is Null) then StokMiktari = 0.0;

    suspend;
  end
end
