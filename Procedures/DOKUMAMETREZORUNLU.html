CREATE PROCEDURE DOKUMAMETREZORUNLU(
    ADEPOID DEPOKODUSIZE,
    AHAMID INTEGER)
RETURNS (
    ZORUNLU EGEBOOLEAN)
AS
declare variable HamTipiId       integer;
declare variable hstDokumaKumas  integer;
declare variable CozguDeposu     integer;
declare variable HasilDeposu     integer;
declare variable IndigoDeposu       integer;
declare variable DokumaProjesiAktif integer;
begin
  Zorunlu = 0;
  select sistem.DokumaProjesiAktif from sistem  into :DokumaProjesiAktif;
  if (AHamId is not null) then
  begin
    Select hstDokumaKumas from HamTipiGetir into :hstDokumaKumas;

    select HamTipiId from Ham where HamId = :AHamId into :HamTipiId;
    if (HamTipiId is not null and HamTipiId = hstDokumaKumas) then
      Zorunlu = 1;
  end

  if (Zorunlu = 0) then /*Dokuma Kumaş ile işlem yapılırken metre zorunlu*/
  begin
    if (ADepoId is not null) then  /*çözgü deposu ile işlem yapılırken metre zorunlu*/
    begin
      select Coalesce(Depo.CozguDeposu,0), Coalesce(Depo.HasilDeposu, 0), Coalesce(Depo.IndigoDeposu, 0) from Depo where DepoId = :ADepoId into :CozguDeposu, :HasilDeposu, :IndigoDeposu;
      if (CozguDeposu = 1 or HasilDeposu = 1 or IndigoDeposu = 1) then
        Zorunlu = 1;
    end
  end

  if (DokumaProjesiAktif = 2) then Zorunlu = 0;

  suspend;
end
