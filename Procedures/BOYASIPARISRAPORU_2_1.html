          into :BoyaGirisTarihi, :BoyaCikisTarihi, :BoyaGirisPersonelId, :BoyaGirisPersonelAdi, :BoyaCikisPersonelId, :BoyaCikisPersonelAdi, :BoyaMakinaId ;

        HamSevkBrutKilo = Null;
        HamSevkNetKilo = Null;

        Select Sum(BoyaIrsaliyeDetay.BrutKilo), Sum(BoyaIrsaliyeDetay.Kilo) from BoyaIrsaliyeDetay
          where BoyaIrsaliyeDetay.RafDetayId is not Null and BoyaIrsaliyeDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
            into :HamSevkBrutKilo, :HamSevkNetKilo;

        ReferansPartiYil = Null; ReferansPartiId = Null; ReferansPartiIdStr = Null; ReferansPartiGrupID = Null; ReferansPartiGrupAdi = Null;
        if (ReferansPartiSinifId is not null) then
        begin
          Select Parti.PartiYil, Parti.PartiId, PartiStok.PartiAsamalariId, PartiStok.Durumu, PartiDurum.Adi, Parti.PartiGrupID, PartiGrup.Adi from PartiStok
            left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
            left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
            left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupID
            where PartiStok.PartiSinifId = :ReferansPartiSinifId
            into :ReferansPartiYil, :ReferansPartiId, :FPartiAsamalariId, :ReferansPartiDurumu, :ReferansPartiDurumuStr, :ReferansPartiGrupID, :ReferansPartiGrupAdi;

          select YilIDStr from YilveIDBirlestir(:ReferansPartiYil, :ReferansPartiId) into :ReferansPartiIdStr;
          Select SonYapilanAsamaAdi from SonAsamaBilgisiGetir(:ReferansPartiSinifId, :FPartiAsamalariId) into :RefPartiSonYapilanAsamaAdi;
        end
        OrmeyeCikanIplik = 0; OrmedenIadeIplik = 0;   Orulen = 0; OrgudenBoyayaIlkSevkTarihi = Null; OrgudenBoyayaSonSevkTarihi = Null;
        if (OrmeSiparisDetayId is not Null) then
        begin
          Select OrgudenBoyayaIlkSevkTarihi, OrgudenBoyayaSonSevkTarihi, OrmeyeCikanIplik, OrmedenIadeIplik, Orulen from BoyaSipRaporuSevkBilgisiGetir(:OrmeSiparisDetayId)
            into :OrgudenBoyayaIlkSevkTarihi, :OrgudenBoyayaSonSevkTarihi, :OrmeyeCikanIplik, :OrmedenIadeIplik, :Orulen;
        end

        select Sum(Miktar), Sum(NetGelenMiktar), Sum(BrutGelenMiktar), Sum(GidenMiktar) from BoyaTalimat where BoyaSiparisDetayId = :BoyaSiparisDetayId into
        :TalimatBoyaMiktar, :TalimatBoyaGelenNetMiktar, :TalimatBoyaGelenBrutMiktar, :TalimatBoyaGidenMiktar;

        if (PartilenenMiktardanIptaliDus = 1) then
        begin
          PartilenenKilo = PartilenenKilo - IptalKilo;
          PartilenenMetre = PartilenenMetre - IptalMetre;
          PartilenenKiloBaski = PartilenenKiloBaski - IptalKiloBaski;
          PartilenenMetreBaski = PartilenenMetreBaski - IptalMetreBaski;
        end
      end
      BukumGerceklesenMiktar = null;
      Select Sum(BukumTalimat.GerceklesenMiktar) From BukumTalimat where BukumTalimat.OrmeSiparisDetayId = :OrmeSiparisDetayID into :BukumGerceklesenMiktar;


      if (UrunFiyatGorebilir = 1) then
      begin
        Select Sum((MiktarYuzde * :FCarpan) * (UretimeCikisFiyati * (Select DovizKuru from EnYakinDovizKuruGetir(UretimeCikisFiyatiDovizID, :FDovizId, Current_Date)))  / FiyatCarpan ) from LabReceteUrun
          left join Urun on Urun.UrunId = LabReceteUrun.UrunId
          left join UrunFiyat on UrunFiyat.UrunId = LabReceteUrun.UrunId
          where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and Urun.UrunTipiID = 0
          into :AtisGuncelBoyaMaliyeti;
      end
      if (Okey = 1) then suspend;
    end
