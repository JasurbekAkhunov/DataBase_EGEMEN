CREATE PROCEDURE BOYASIPARISRAPORU_4(
    AIMALATTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AFIRMAYETKILIID INTEGER,
    AISLEMTIPIID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AMUSTERIKALITENO VARCHAR(20),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AHAMID INTEGER,
    AKULLANICIID INTEGER,
    AKALITEID VARCHAR(20),
    ATALIMATKODU VARCHAR(20),
    AMAKINAID VARCHAR(10),
    ATEKRARURETIM INTEGER,
    ADESENID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AMARKALI INTEGER,
    AKOMISYONLU INTEGER,
    ALAMINASYON INTEGER,
    ARECETETURUID INTEGER,
    ABATCHNO VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ABASKITIPIID INTEGER,
    AHAMGRUP1ID VARCHAR(4),
    AHAMGRUP2ID VARCHAR(4),
    AHAMGRUP3ID VARCHAR(4),
    AHAMGRUP4ID VARCHAR(4),
    ASIPARISRECETEDURUMU INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISIRANO INTEGER,
    FASON INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    DENYE VARCHAR(30),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    PROSESID INTEGER,
    PROSESADI VARCHAR(255),
    SONASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    DURUMU INTEGER,
    DURUMUSTR VARCHAR(20),
    SINIFBOYASIPARISDETAYID INTEGER,
    PARTISTOKIPTALYIL INTEGER,
    PARTISTOKIPTALID INTEGER,
    TARTIBOBINSAYISI INTEGER,
    BOBINSAYISI INTEGER,
    BASKITARIHI TIMESTAMP,
    NETSEVKMETRE BOYAMIKTARSIZE,
    NETSEVKMETRETARTI BOYAMIKTARSIZE,
    BRUTSEVKMETRE BOYAMIKTARSIZE,
    NETSEVKKILO BOYAMIKTARSIZE,
    BRUTSEVKKILO BOYAMIKTARSIZE,
    NETSARFMETRE BOYAMIKTARSIZE,
    BRUTSARFMETRE BOYAMIKTARSIZE,
    NETSARFKILO BOYAMIKTARSIZE,
    BRUTSARFKILO BOYAMIKTARSIZE,
    REZERVKILO BOYAMIKTARSIZE,
    BRUTREZERVKILO BOYAMIKTARSIZE,
    REZERVMETRE BOYAMIKTARSIZE,
    REZERVMIKTAR BOYAMIKTARSIZE,
    BRUTREZERVMIKTAR BOYAMIKTARSIZE,
    IPTALRAFID RAFKODUSIZE,
    IPTALTARIHI TIMESTAMP,
    KALITENO INTEGER,
    HAMSTOKSONALISFIYATIUSD DOUBLE PRECISION,
    PSKKTOPLAMMETRE BOYAMIKTARSIZE,
    PSKKTOPLAMBRUTMETRE BOYAMIKTARSIZE,
    PSKKTOPLAMBRUTKILO BOYAMIKTARSIZE,
    PSKKTOPLAMTOPLAMKILO BOYAMIKTARSIZE,
    SEVKEHAZIRNETMETRE BOYAMIKTARSIZE,
    SEVKEHAZIRNETKILO BOYAMIKTARSIZE,
    LOGBLOBBIRIMFIYAT VARCHAR(1000),
    LOGBLOBKILO VARCHAR(1000),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    PARTISTOKASAMALARI VARCHAR(1024),
    PARTISTOKYAPILANASAMALARI VARCHAR(1024),
    KALITE2TARTIKILO BOYAMIKTARSIZE,
    KALITE2TARTIBRUTKILO BOYAMIKTARSIZE,
    KKBITISTARIHI TIMESTAMP)
AS
declare variable EgemenFirmaId integer;
begin
  select EgemenFirmaId from Sistem where sistemid = 1 into :EgemenFirmaId;
  /*Tekboy için veri güncellemesi olduğundan bir tarihten önceki kayıtların metreleri farklı mantıkta gösterildi.*/
  For Select BoyaSiparisRaporu_3.BoyaSiparisDetayId, BoyaSiparisRaporu_3.PartiKayitId, BoyaSiparisRaporu_3.Fason, PartiStok.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.Denye,
             PartiStok.Kilo, PartiStok.Metre, PartiStok.ProsesId, Proses.Adi, PartiStok.BoyaSiparisDetayId, PartiStok.Durumu, PartiDurum.Adi,
             PartiStokIptal.PartiStokIptalYil, PartiStokIptal.PartiStokIptalId, PartiAsamaStok.AsamaId, Asama.Adi,
             PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi, Sum(PartiStokTarti.BobinSayisi), PartiStok.BobinSayisi, PartiStok.PartiSiraNo,
             PartiStok.NetSevkMetre, PartiStok.BrutSevkMetre, PartiStok.NetSevkKilo, PartiStok.BrutSevkKilo, PartiStok.NetSarfMetre, PartiStok.BrutSarfMetre, PartiStok.NetSarfKilo,
             PartiStok.BrutSarfKilo, PartiStok.RezervKilo, PartiStok.BrutRezervKilo, PartiStok.RezervMetre, PartiStok.RezervMiktar, PartiStok.BrutRezervMiktar,
             PartiStokIptal.RafId, PartiStokIptal.IptalTarihi, PartiStokIptal.KaliteNo,
             Sum(iif(PSKaliteKontrol.InsertTarihi < '11.10.2018 19:00:00' and PSKaliteKontrol.En is not null and PSKaliteKontrol.Gramaj is not null,
             (1000 / (cast(PSKaliteKontrol.En as Double Precision) * Cast(PSKaliteKontrol.Gramaj as Double Precision))) * (PSKaliteKontrol.Kilo-PSKaliteKontrol.Dara) *
             Cast(Coalesce(BoyaSiparisDetay.HesaplamaCarpimDegeri, 1) as Double Precision)
             , PartiStokTarti.Metre)),
             Sum(PartiStokTarti.Kilo),
             Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is not null,
             iif(PSKaliteKontrol.InsertTarihi < '11.10.2018 19:00:00' and PSKaliteKontrol.En is not null and PSKaliteKontrol.Gramaj is not null,
             (1000 / (cast(PSKaliteKontrol.En as Double Precision) * Cast(PSKaliteKontrol.Gramaj as Double Precision))) * (PSKaliteKontrol.Kilo-PSKaliteKontrol.Dara) *
             Cast(Coalesce(BoyaSiparisDetay.HesaplamaCarpimDegeri, 1) as Double Precision)
             , PartiStokTarti.Metre),0)), PartiStok.BirimFiyat, PartiStok.DovizId,
             Sum(iif(/*PartiStok.Durumu = 7 and PartiStokTarti.BoyaIrsaliyeDetayId is null and*/ PartiStokTarti.KaliteNo > 1, PartiStokTarti.Kilo, 0)) Kalite2TartiKilo,
             Sum(iif(/*PartiStok.Durumu = 7 and PartiStokTarti.BoyaIrsaliyeDetayId is null and*/ PartiStokTarti.KaliteNo > 1, PartiStokTarti.BrutKilo, 0)) Kalite2TartiBrutKilo,
       PartiStok.KKBitisTarihi
             from BoyaSiparisRaporu_3(:AImalatTipiId, :ABaslangicTarihi, :ABitisTarihi, :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AFirmaYetkiliId, :AIslemTipiId, :AIslemGrubuId, :ASatisTipi, :ABoyaSiparisTipiId, :ALabReceteId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMusteriKaliteNo, :AOnayDurumu, :APartilenen, :AKapaliDurumu, :ASiparisiAlanKullaniciId, :AYurtDisi, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AHamId, :AKullaniciId, :AKaliteId, :ATalimatKodu, :AMakinaId, :ATekrarUretim, :ADesenID, :ATemsilciId, :AMarkali, :AKomisyonlu, :ALaminasyon, :AReceteTuruId, :ABatchNo, :ASertifikaNo, :ABaskiTipiId, :AHamGrup1Id, :AHamGrup2Id, :AHamGrup3Id, :AHamGrup4Id, :ASiparisReceteDurumu)
    left join PartiStok on PartiStok.PartiKayitId = BoyaSiparisRaporu_3.PartiKayitId
    left join Proses on Proses.ProsesId = PartiStok.ProsesId
    left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
    left join PartiStokIptal on PartiStokIptal.PartiStokIptalKayitId = PartiStok.PartiStokIptalKayitId
    left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiStok.PartiSinifId and PartiAsamaStok.PartiAsamalariId = PartiStok.PartiAsamalariId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left Join PartiStokTarti on PartiStokTarti.PartiSinifId = PartiStok.PartiSinifID
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisRaporu_3.BoyaSiparisDetayId
    left join PSKaliteKontrol on PSKaliteKontrol.PartiSinifId = PartiStok.PartiSinifID and PSKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    where BoyaSiparisRaporu_3.PartiKayitId is not null
    Group By BoyaSiparisRaporu_3.BoyaSiparisDetayId, BoyaSiparisRaporu_3.PartiKayitId, BoyaSiparisRaporu_3.Fason, PartiStok.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.Denye,
             PartiStok.Kilo, PartiStok.Metre, PartiStok.ProsesId, Proses.Adi, PartiStok.BoyaSiparisDetayId, PartiStok.Durumu, PartiDurum.Adi,
             PartiStokIptal.PartiStokIptalYil, PartiStokIptal.PartiStokIptalId, PartiStokIptal.RafId, PartiStokIptal.IptalTarihi, PartiStokIptal.KaliteNo, PartiAsamaStok.AsamaId, Asama.Adi,
             PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi, PartiStok.BobinSayisi, PartiStok.PartiSiraNo, PartiStok.NetSevkMetre, PartiStok.BrutSevkMetre, PartiStok.NetSevkKilo, PartiStok.BrutSevkKilo, PartiStok.NetSarfMetre, PartiStok.BrutSarfMetre, PartiStok.NetSarfKilo,
             PartiStok.BrutSarfKilo, PartiStok.RezervKilo, PartiStok.BrutRezervKilo, PartiStok.RezervMetre, PartiStok.RezervMiktar, PartiStok.BrutRezervMiktar, PartiStok.BirimFiyat,
       PartiStok.DovizId, PartiStok.KKBitisTarihi
    into :BoyaSiparisDetayId, :PartiKayitId, :Fason, :PartiSinifId, :HamAdiToplami, :Denye, :Kilo, :Metre, :ProsesId, :ProsesAdi,
         :SinifBoyaSiparisDetayId, :Durumu, :DurumuStr, :PartiStokIptalYil, :PartiStokIptalId,
         :SonAsamaId, :AsamaAdi, :GirisTarihi, :CikisTarihi, :TartiBobinSayisi, :BobinSayisi, :PartiSiraNo, :NetSevkMetre, :BrutSevkMetre,
         :NetSevkKilo, :BrutSevkKilo, :NetSarfMetre, :BrutSarfMetre, :NetSarfKilo, :BrutSarfKilo, :RezervKilo, :BrutRezervKilo, :RezervMetre, :RezervMiktar, :BrutRezervMiktar,
         :IptalRafId, :IptalTarihi, :KaliteNo, :SevkeHazirNetMetre, :SevkeHazirNetKilo, :NetSevkMetreTarti, :BirimFiyat, :DovizId, :Kalite2TartiKilo, :Kalite2TartiBrutKilo, :KKBitisTarihi do
  begin
    if (PartiStokIptalId is not null) then
    begin
      Durumu = -1;
      DurumuStr = 'Ham Depoya İade';
    end

    Select Max(CikisTarihi) from PartiAsamaStok
      Left Join Asama  on Asama.AsamaID = PartiAsamaStok.AsamaID
      where Asama.BaskiAsamasimi = 1 and PartiAsamaStok.PartiSinifID = :PartiSinifId  into :BaskiTarihi;


    select AlisFiyati from PartiSonAlisFiyatiUSD(:PartiSinifId) into :HamStokSonAlisFiyatiUSD;

    select
    Sum(iif(PSKaliteKontrol.InsertTarihi < '11.10.2018 19:00:00' and PSKaliteKontrol.En is not null and PSKaliteKontrol.Gramaj is not null,
             (1000 / (cast(PSKaliteKontrol.En as Double Precision) * Cast(PSKaliteKontrol.Gramaj as Double Precision))) * (PSKaliteKontrol.Kilo-PSKaliteKontrol.Dara) *
             Cast(Coalesce(BoyaSiparisDetay.HesaplamaCarpimDegeri, 1) as Double Precision)
             , PSKaliteKontrol.Metre)),
             Sum(PSKaliteKontrol.BrutMetre), Sum(PSKaliteKontrol.Kilo), Sum(PSKaliteKontrol.Kilo - PSKaliteKontrol.Dara) from PSKaliteKontrolAsama
    left join PSKaliteKontrol on PSKaliteKontrol.PSKaliteKontrolAsamaId = PSKaliteKontrolAsama.PSKaliteKontrolAsamaId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
    where PSKaliteKontrolAsama.PartiSinifId = :PArtiSinifID and PSKaliteKontrolAsama.SonKayit = 1 into :PsKKToplamMetre, :PsKKToplamBrutMetre, :PsKKToplamBrutKilo, :PsKKToplamToplamKilo;

    Select PartiStokAsamalari, PartiStokYapilanAsamalari from PartiSinifAsamalariGetir(:PartiSinifId) into :PartiStokAsamalari, :PartiStokYapilanAsamalari;

    if (EgemenFirmaId = 92) then /*ideal tekstil*/
      select PartiStokLogGetir.LogBlobBirimFiyat, PartiStokLogGetir.LogBlobKilo from PartiStokLogGetir(:PartiSinifId, Null) into :LogBlobBirimFiyat, :LogBlobKilo;

    suspend;
  end
end
