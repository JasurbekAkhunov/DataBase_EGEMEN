CREATE PROCEDURE HAKTARMAFISILISTESIDETAYI(
    ABASLANGICFISTARIHI TIMESTAMP,
    ABITISFISTARIHI TIMESTAMP,
    ATESLIMDURUMU INTEGER,
    ABASLANGICFISYIL INTEGER,
    ABASLANGICFISNO INTEGER,
    ABITISFISYIL INTEGER,
    ABITISFISNO INTEGER)
RETURNS (
    HAKTARMAFISIID INTEGER,
    HAKTARMAFISIDETAYID INTEGER,
    HAMSTOKDETAYID INTEGER,
    HAMSTOKIRSALIYEID INTEGER,
    RAFID VARCHAR(10),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    LOT VARCHAR(150),
    KILO NUMERIC(12,3),
    TEORIKKILO NUMERIC(12,3),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FASONSTR VARCHAR(5),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    FASON INTEGER)
AS
begin
  For select HAktarmaFisiListesi.HAktarmaFisiId, HAktarmaFisiDetay.HAktarmaFisiDetayId, HAktarmaFisiDetay.HamStokDetayId, HamStokDetay.HamStokIrsaliyeId, HamStokDetay.HamId,
             Ham.Adi, HamStokMaster.Lot, RafDetay.RafId, HAktarmaFisiDetay.Kilo, HAktarmaFisiDetay.TeorikKilo, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, HamStokIrsaliye.Fason from HAktarmaFisiListesi(:ABaslangicFisTarihi, :ABitisFisTarihi, :ATeslimDurumu, :ABaslangicFisYil, :ABaslangicFisNo, :ABitisFisYil, :ABitisFisNo)
  left join HAktarmaFisiDetay on HAktarmaFisiDetay.HAktarmaFisiId = HAktarmaFisiListesi.HAktarmaFisiId
  left join HamStokDetay on HamStokDetay.HamStokDetayid = HAktarmaFisiDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterid = HamStokDetay.HamStokMasterId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
  left join RafDetay on RafDetay.RafDetayid = HAktarmaFisiDetay.RafDetayId
  left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
  left join Ham on Ham.HamId = HamStokDetay.HamId
  where HAktarmaFisiDetay.HAktarmaFisiDetayId is not null
  order by HAktarmaFisiDetay.HAktarmaFisiId, HAktarmaFisiDetay.HAktarmaFisiDetayId
  into :HAktarmaFisiId, :HAktarmaFisiDetayId, :HamStokDetayId, :HamStokIrsaliyeId, :HamId, :HamAdi, :Lot, :RafId, :Kilo, :TeorikKilo, :FirmaId, :FirmaAdi,
       :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi, :Fason do
  begin
    if (Fason = 1) then FasonStr = 'Fason'; else FasonStr = '';

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';
    suspend;
  end
end
