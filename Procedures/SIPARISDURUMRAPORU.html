CREATE PROCEDURE SIPARISDURUMRAPORU(
    AHAMID INTEGER,
    AHAMTIPIID INTEGER,
    ALOT VARCHAR(30),
    AANALOT VARCHAR(50))
RETURNS (
    HAMID INTEGER,
    IPLIKDEPOID VARCHAR(10),
    HAMADI VARCHAR(100),
    HAMADI2 VARCHAR(255),
    HAMTIPIID INTEGER,
    HAMMIKTAR DOUBLE PRECISION,
    NETMIKTAR DOUBLE PRECISION,
    IPLIKLOTID INTEGER,
    LOT VARCHAR(30),
    ANALOT VARCHAR(50),
    KAPALI INTEGER,
    KAPALISTR VARCHAR(6),
    ACIKISEMRI INTEGER,
    ACIKISEMRISTR VARCHAR(6),
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    HARMANADI VARCHAR(50),
    ANALOTBOLUM VARCHAR(20),
    FLAMKODU VARCHAR(20),
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    BOBINSAYISI INTEGER)
AS
begin
  if (AHamTipiId = 0) then Select hstIplik from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 1) then Select hstElyaf from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 2) then Select hstTelef from HamTipiGetir into :AHamTipiId;
  else AHamTipiId = null;

  For Select IplikIrsaliyeDetay.HamId, IplikIrsaliyeDetay.IplikDepoId, Ham.Adi, Ham.HamtipiId,  Sum(IplikStok.Miktar), Sum(IplikStok.NetMiktar),
             Ham.Adi2, IplikIrsaliyeDetay.Lot, IplikLot.Kapali, IplikLot.IplikLotID, Harman.AnaLot, Harman.HarmanYil, Harman.HarmanID,
             IplikIrsaliyeDetay.FlamKodu, IplikIrsaliyeDetay.AmbalajId, Ambalaj.Adi, Sum(IplikStok.BobinSayisi) from IplikStok
  left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikStok.IplikIrsaliyeId
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = IplikStok.IplikIrsaliyeDetayId
  left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
  left join IplikLot on IplikLot.IplikLotID = IplikIrsaliyeDetay.IplikLotID
  left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitID
  left join Ambalaj on Ambalaj.AmbalajId = IplikIrsaliyeDetay.AmbalajId
  where IplikStok.CikisIplikIrsaliyeDetayID is Null and
        (IplikIrsaliyeDetay.HamId = :AHamID or :AHamID is null) and
        (Ham.HamtipiId = :AHamTipiId or :AHamTipiId is Null) and
        (IplikIrsaliyeDetay.Lot = :ALot or :ALot is Null) and
        (Harman.AnaLot = :AAnaLot or :AAnaLot is Null)
  group by IplikIrsaliyeDetay.HamId, IplikIrsaliyeDetay.IplikDepoId, Ham.Adi, Ham.HamtipiId, Ham.Adi2, IplikIrsaliyeDetay.Lot, IplikLot.Kapali,
           IplikLot.IplikLotID, Harman.AnaLot, Harman.HarmanYil, Harman.HarmanID, IplikIrsaliyeDetay.FlamKodu, IplikIrsaliyeDetay.AmbalajId, Ambalaj.Adi
  into :HamId, :IplikDepoId, :HamAdi, :HamtipiId, :HamMiktar, :NetMiktar, :HamAdi2, :Lot, :Kapali, :IplikLotID, :AnaLot, :HarmanYil, :HarmanID,
       :FlamKodu, :AmbalajId, :AmbalajAdi, :BobinSayisi do
  begin

    if (Kapali = 1) then KapaliStr = 'Kapali';
    else KapaliStr = '';

    Select Count(*) from IplikUretimIsEmri where IplikUretimIsEmri.Bitti = 0 and IplikUretimIsEmri.IplikMakinaTipiID = 0 and IplikUretimIsEmri.IplikLotID = :IplikLotID Into :AcikIsEmri;
    if (AcikIsEmri > 0) then AcikIsEmriStr = 'Evet';
    else AcikIsEmriStr = '';

    if (HarmanYil is not null) then
      HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') '  || EgemenIntegerToStr(HarmanId)  || ' ';
    else
      HarmanIdStr = '';

     AnaLotBolum = Null ;
    if (Char_Length(Lot) > 0 and Position('.' in Lot) > 0)  then
      AnaLotBolum = substring(Lot from 1 for (Position('.' in Lot)-1) );

    suspend;
  end
end
