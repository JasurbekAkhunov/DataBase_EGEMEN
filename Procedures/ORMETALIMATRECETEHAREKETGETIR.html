CREATE PROCEDURE ORMETALIMATRECETEHAREKETGETIR(
    ABOYASIPARISKAYITID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    ORMEPLANLAMAID INTEGER,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(150),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    SONALISFIYATI DOUBLE PRECISION,
    SONALISDOVIZID DOVIZSIZE,
    STOKGIRISFIYATI DOUBLE PRECISION,
    STOKGIRISDOVIZID DOVIZSIZE,
    STSONALISFIYATI DOUBLE PRECISION,
    STSONALISDOVIZID DOVIZSIZE)
AS
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable SistemDovizId DovizSize;
declare variable StokTabanDovizId DovizSize;
declare variable VerilenSiparisFiyatGorebilir EgeBoolean;
begin
  select first 1 VerilenSiparisFiyatGorebilir from Kullanici where KullaniciID = :aKullaniciId into :VerilenSiparisFiyatGorebilir;
  Select shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim, ShtUretim from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis,  :shtUretimeCikis, :shtFasonUretim, :ShtUretim;
  select Sistem.SistemDovizId, Sistem.StokTabanDovizId from Sistem where SistemId = 1 into :SistemDovizId, :StokTabanDovizId;

  for Select GStokHareket.OrmePlanlamaId, GStokHareket.HamId, Ham.Adi, Ham.Adi2, GStokHareket.RenkId, Renk.Adi, GStokHareket.MelanjKodu, GStokHareket.Lot,
             GStokHareket.UreticiID, Uretici.Adi, Sum(CikisMiktari), Sum(IadeMiktari), StokGirisNo
      from  (Select iif((HareketTipi = :shtFasonUretimeCikis or HareketTipi = :shtUretimeCikis) and (HamIade = 0), Sum(Miktar),0) CikisMiktari,
                    iif((HareketTipi = :shtFasonUretim or HareketTipi = :shtUretim) and (HamIade = 1), Sum(Miktar), 0) IadeMiktari,
                    OrmePlanlama.OrmePlanlamaId, StokHareket.HamId, StokHareket.RenkId, StokHareket.MelanjKodu, StokHareket.Lot, StokHareket.HareketTipi,
                    StokHareket.HamIade, StokHareket.UreticiID, OrmeRaf.StokGirisNo from OrmePlanlama
          left join StokHareket on StokHareket.OrmePlanlamaID = OrmePlanlama.OrmePlanlamaID
          left join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
          where OrmePlanlama.BoyaSiparisKayitID = :ABoyaSiparisKayitId and
               (((HareketTipi = :shtFasonUretimeCikis or HareketTipi = :shtUretimeCikis) and (HamIade = 0)) or
                ((HareketTipi = :shtFasonUretim or HareketTipi = :shtUretim) and (HamIade = 1)))
          Group By OrmePlanlama.OrmePlanlamaId, StokHareket.HamId, StokHareket.RenkId, StokHareket.MelanjKodu, StokHareket.Lot, StokHareket.HareketTipi,
                   StokHareket.HamIade, StokHareket.UreticiID, OrmeRaf.StokGirisNo
        )  GStokHareket
  Left join Ham on Ham.HamId = GStokHareket.HamID
  Left Join Renk on Renk.RenkID = GStokHareket.RenkID
  left join Uretici on Uretici.UreticiId = GStokHareket.UreticiId
  Group By GStokHareket.OrmePlanlamaId, GStokHareket.HamId, Ham.Adi, Ham.Adi2, Renk.Adi, GStokHareket.RenkId, GStokHareket.MelanjKodu, GStokHareket.Lot,
           GStokHareket.UreticiID, Uretici.Adi, GStokHareket.StokGirisNo
  into :OrmePlanlamaId, :HamID, :HamAdi, :HamAdi2, :RenkId, :RenkAdi, :MelanjKodu, :Lot, :UreticiID, :UreticiAdi, :CikisMiktari, :IadeMiktari, :StokGirisNo do
  begin
    SonAlisFiyati = null;
    SonAlisDovizId = null;
    StokGirisFiyati = Null;
    StokGirisDovizId = Null;

    if (VerilenSiparisFiyatGorebilir = 1) then
    begin
      select OrmeHamStokSonAlisFiyati.SonAlisFiyati, OrmeHamStokSonAlisFiyati.DovizId from OrmeHamStokSonAlisFiyati(:HamId, :RenkId, :SistemDovizId, current_timestamp, 0, Null, :Lot)
      into :SonAlisFiyati, :SonAlisDovizId;

      select OrmeHamStokSonAlisFiyati.SonAlisFiyati, OrmeHamStokSonAlisFiyati.DovizId from OrmeHamStokSonAlisFiyati(:HamId, :RenkId, :StokTabanDovizId, current_timestamp, 0, Null, :Lot)
      into :StSonAlisFiyati, :StSonAlisDovizId;

      if (StokGirisNo is not null) then
        select First 1 StokHareket.BirimFiyat, StokHareket.DovizId from StokHareket where StokHareket.StokGirisNo = :StokGirisNo /*and StokHareket.HamIade = 0 EB-832 */and  StokHareket.GirisCikis = 1 order by StokHareket.Tarih Desc
        into :StokGirisFiyati, :StokGirisDovizId;
    end
    suspend;
  end
end
