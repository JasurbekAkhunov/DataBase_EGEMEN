CREATE PROCEDURE DOKUMAVARDIYAKARTILISTESI(
    ATARIH TIMESTAMP)
RETURNS (
    ORMEVARDIYAID INTEGER,
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    VARDIYANO INTEGER,
    VARDIYAAMIRIPERSONELID INTEGER,
    VARDIYAAMIRIPERSONELADI VARCHAR(40),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select OrmeVardiya.OrmeVardiyaId, OrmeVardiya.UniteId, Unite.Adi, OrmeVardiya.VardiyaNo, OrmeVardiya.VardiyaAmiriPersonelId, VAP.Adi,
             IK.Adi, OrmeVardiya.InsertTarihi, SK.Adi, OrmeVardiya.DegisimTarihi from OrmeVardiya
    left join Unite on Unite.UniteId = OrmeVardiya.UniteId
    left join Personel VAP on VAP.PersonelId = OrmeVardiya.VardiyaAmiriPersonelId
    left join Kullanici IK on IK.KullaniciId = OrmeVardiya.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId = OrmeVardiya.KullaniciId
    where OrmeVardiya.Tarih = :ATarih and OrmeVardiya.DokumaVardiyasi = 1
    into :OrmeVardiyaId, :UniteId, :UniteAdi, :VardiyaNo, :VardiyaAmiriPersonelId, :VardiyaAmiriPersonelAdi,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi  do suspend;
end
