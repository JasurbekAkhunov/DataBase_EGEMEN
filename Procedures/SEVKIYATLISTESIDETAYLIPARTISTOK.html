CREATE PROCEDURE SEVKIYATLISTESIDETAYLIPARTISTOK(
    AFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AARACID VARCHAR(11),
    APARTIGRUPID VARCHAR(4),
    AUNITEID INTEGER,
    APARTITIPI INTEGER,
    AMAMULDEPOSEVKLERIDAHIL INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARECETEID VARCHAR(20),
    AFASON INTEGER,
    AIMALATTIPIID INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    APARTIKAYITID INTEGER,
    ABASKITIPIID INTEGER,
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    MUSTERIYESEVK INTEGER,
    FIRMAID VARCHAR(20),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    ARACID VARCHAR(11),
    SEFERNO INTEGER,
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    KAPADEDI INTEGER,
    CIKISBOBINSAYISI INTEGER,
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    MUSTERIPARTINO VARCHAR(100),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    SIPARISBIRIMFIYAT DOUBLE PRECISION,
    SIPARISDOVIZID VARCHAR(5),
    HAMRENKADI VARCHAR(50),
    PAKETLEMESEKLIID INTEGER,
    PAKETLEMESEKLIADI VARCHAR(40),
    PROSESID INTEGER,
    PROSESADI VARCHAR(255),
    REFERANSKODU HAMKODUSIZE,
    ALTBEZHAMID INTEGER,
    ALTBEZHAMADI ADI255,
    DIGERADI2 VARCHAR(255),
    DIGERADI3 VARCHAR(255),
    DIGERADI4 VARCHAR(255),
    SIPARISRENKADI VARCHAR(40))
AS
declare Variable FMusteriPartiNo varchar(30);
declare Variable Fason  integer;
declare Variable BoyaIrsaliyeDetayID integer;
declare variable biSatisIade integer;
begin
  Select biSatisIade from BoyaIrsaliyeHareketTipiGetir into :biSatisIade;
  MusteriyeSevk = 1;
  For Select Parti.PartiKayitId, Parti.Fason, PartiStok.PartiSinifId,Parti.PartiYil, Parti.PartiId, PartiStok.hamAdiToplami, Boyairsaliye.Irsaliyeno,
     Boyairsaliye.Irsaliyetarihi, Boyairsaliye.SevkTarihi, Boyairsaliye.AracId, Boyairsaliye.SevkSeferno, boyairsaliyedetay.KapAdedi, boyairsaliyedetay.BrutKilo,
     boyairsaliyedetay.Kilo, boyairsaliyedetay.BrutMetre, BoyaIrsaliyeDetay.Metre, PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.ReferansSiparisNo, BoyaSiparisDetay.MusteriRenkKoduveAdi,
     BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
     BoyaSiparisDetay.ProsesId, Proses.Adi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, HamRenk.Adi, BoyaSiparisDetay.BirimFiyat,
     BoyaSiparisDetay.DovizId, PartiStok.PaketlemeSekliId, PaketlemeSekli.Adi,
     BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID, BoyaIrsaliye.FirmaId, Ham.ReferansKodu, BoyaSiparisDetay.AltBezHamId, HamAltBez.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4,
     BoyaSiparisDetay.RenkAdi from BoyaIrsaliye
  left join firma on firma.firmaid = BoyaIrsaliye.FirmaId
  left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
  left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
  left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliId = PartiStok.PaketlemeSekliId
  left join Proses on Proses.ProsesId = PartiStok.ProsesId
  left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
  left join Renk HamRenk on HamRenk.RenkID = OrmeSiparisDetay.RenkId
  left join Ham on Ham.HamID = PartiStok.HamId
  left join Ham HamAltBez on HamAltBez.HamId = BoyaSiparisDetay.AltBezHamId
  where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and
        BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
       (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
       (Parti.PartiTipiID = :APartiTipi or :APartiTipi is null) and
       (Parti.Fason = :AFason or :AFason is null) and
       (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null) and
       (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
       (BoyaIrsaliye.AracId = :AAracId or :AAracId is null) and
       (BoyaIrsaliye.UniteId = :AUniteId or :AUniteId is null) and
       (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and BoyaIrsaliyeDetay.PartiSinifId is not null and
       (Parti.ReceteId = :AReceteId or :AReceteId is null) and
       (Parti.PartiKayitId = :APartiKayitId or :APartiKayitId is null) and
       (Parti.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
       (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
       (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
       (BoyaSiparisDetay.BaskiTipiId = :ABaskiTipiId or :ABaskiTipiId is null) and
       BoyaIrsaliye.HareketTipiId <> :biSatisIade
  order by BoyaIrsaliye.IrsaliyeTarihi
  into :PartiKayitId, :Fason, :PartiSinifId, :PartiYil, :PartiId, :HamAdiToplami, :Irsaliyeno,
     :IrsaliyeTarihi, :FiiliSevkTarihi, :AracId, :Seferno, :KapAdedi, :BrutKilo, :NetKilo, :BrutMetre, :NetMetre,
     :BoyaSiparisDetayId, :MusteriSiparisNo, :ReferansSiparisNo, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
     :ProsesId, :ProsesAdi, :SiparisProsesId, :SiparisProsesAdi, :HamRenkAdi, :SiparisBirimFiyat, :SiparisDovizId,
     :PaketlemeSekliId, :PaketlemeSekliAdi, :BoyaIrsaliyeDetayID, :FirmaId, :ReferansKodu, :AltBezHamId, :AltBezHamAdi,
     :DigerAdi2, :DigerAdi3, :DigerAdi4, :SiparisRenkAdi do
  begin
    MusteriPartiNo = '';

    For Select HamStokMaster.MusteriPartiNo from PartiStokDetay
      left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
      where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.MusteriPartiNo is not null and HamStokMaster.MusteriPartiNo <> ''
      group by HamStokMaster.MusteriPartiNo
      into :FMusteriPartiNo do
    begin
      if (Char_Length(MusteriPartiNo) < 90) then
      begin
        if (FMusteriPartiNo is not null) then
        begin
          if (MusteriPartiNo <> '' ) then MusteriPartiNo = MusteriPartiNo || cast(', ' as varchar(2) character set UTF8);
          MusteriPartiNo = MusteriPartiNo || FMusteriPartiNo;
        end
      end
    end

    if (Fason = 0 and ASatisBoyaSiparisFiyatGorebilir = 0) then
    begin
      SiparisBirimFiyat = null;
      SiparisDovizId    = null;
    end
    else if (Fason = 1 and AFasonBoyaSiparisFiyatGorebilir = 0) then
    begin
      SiparisBirimFiyat = null;
      SiparisDovizId    = null;
    end

    CikisBobinSayisi = null;

    Select Sum(IplikPaketleme.BobinSayisi) from PartiStokTarti
      left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
      where PartiStokTarti.PartiSinifID = :PartiSinifID and PartiStokTarti.BoyaIrsaliyeDetayId = :BoyaIrsaliyeDetayId
      into :CikisBobinSayisi;

    suspend;
  end

  if (AMamulDepoSevkleriDahil = 1 and AUniteId is not null) then
  begin
    MusteriyeSevk = 0;
    CikisBobinSayisi = null;

    For Select Parti.PartiKayitId, Parti.Fason, PartiStok.PartiSinifId, Parti.PartiYil, Parti.PartiId, PartiStok.hamAdiToplami, MamulDepoTransferFisi.Irsaliyeno,
       MamulDepoTransferFisi.Irsaliyetarihi, MamulDepoTransferFisi.SevkTarihi, MamulDepoTransferFisi.AracId, MamulDepoTransferFisi.SevkSeferno,
       PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.ReferansSiparisNo, BoyaSiparisDetay.MusteriRenkKoduveAdi,
       BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
       BoyaSiparisDetay.ProsesId, Proses.Adi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, HamRenk.Adi,
       BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, PartiStok.PaketlemeSekliId, PaketlemeSekli.Adi,
       Parti.FirmaId,
       Sum(PartiStokTarti.Kilo),  sum(PartiStokTarti.BrutKilo), Sum(PartiStokTarti.Metre),  sum(PartiStokTarti.BrutMetre), Sum(PartiStokTarti.KapAdedi), Ham.ReferansKodu,
       BoyaSiparisDetay.AltBezHamId, HamAltBez.Adi, BoyaSiparisDetay.RenkAdi from MamulDepoTransferFisi
    left join Depo D1 on D1.DepoId = MamulDepoTransferFisi.DepoId
    left join Depo D2 on D2.DepoId = MamulDepoTransferFisi.TransferDepoId
    left join Unite U1 on U1.UniteId = D1.UniteId
    left join Unite U2 on U2.UniteId = D2.UniteId
    left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiID = MamulDepoTransferFisi.MamulDepoTransferFisiId
    left join PartiStokTarti on PartiStokTarti.PartiStokTartiID = MamulDepoTransferFisiDetay.PartiStokTartiId
    left join PartiStok on PartiStok.PartiSinifId = MamulDepoTransferFisiDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
    left join firma on firma.firmaid = Parti.FirmaId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliId = PartiStok.PaketlemeSekliId
    left join Proses on Proses.ProsesId = PartiStok.ProsesId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
    left join Renk HamRenk on HamRenk.RenkID = OrmeSiparisDetay.RenkId
    left join Ham on Ham.HamID = PartiStok.HamId
    left join Ham HamAltBez on HamAltBez.HamID = BoyaSiparisDetay.AltBezHamId
    where MamulDepoTransferFisi.IrsaliyeTarihi >= :ABaslangicTarihi and
            MamulDepoTransferFisi.IrsaliyeTarihi <= :ABitisTarihi and MamulDepoTransferFisi.SistemKaydi = 0 and
           (D1.UniteId = :AUniteId or :AUniteId is null) and
           (D2.UniteId <> :AUniteId or :AUniteId is null) and
           (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
           (Parti.PartiTipiID = :APartiTipi or :APartiTipi is null) and
           (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
           (Parti.Fason = :AFason or :AFason is null) and
           (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null) and
           (MamulDepoTransferFisi.AracId = :AAracId or :AAracId is null) and
           (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and MamulDepoTransferFisiDetay.PartiSinifId is not null and
           (Parti.ReceteId = :AReceteId or :AReceteId is null) and
           (Parti.PartiKayitId = :APartiKayitId or :APartiKayitId is null) and
           (Parti.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
           (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
           (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
           (BoyaSiparisDetay.BaskiTipiId = :ABaskiTipiId or :ABaskiTipiId is null)
    group by Parti.PartiKayitId, Parti.Fason, PartiStok.PartiSinifId, Parti.PartiYil, Parti.PartiId, PartiStok.hamAdiToplami, MamulDepoTransferFisi.Irsaliyeno,
             MamulDepoTransferFisi.Irsaliyetarihi, MamulDepoTransferFisi.SevkTarihi, MamulDepoTransferFisi.AracId, MamulDepoTransferFisi.SevkSeferno,
             PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.ReferansSiparisNo, BoyaSiparisDetay.MusteriRenkKoduveAdi,
             BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
             BoyaSiparisDetay.ProsesId, Proses.Adi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, HamRenk.Adi,
             BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, PartiStok.PaketlemeSekliId, PaketlemeSekli.Adi,
             Parti.FirmaId, Ham.ReferansKodu, BoyaSiparisDetay.AltBezHamId, HamAltBez.Adi, BoyaSiparisDetay.RenkAdi
    order by MamulDepoTransferFisi.IrsaliyeTarihi
    into :PartiKayitId, :Fason, :PartiSinifId, :PartiYil, :PartiId, :HamAdiToplami, :Irsaliyeno,
         :Irsaliyetarihi, :FiiliSevkTarihi, :AracId, :Seferno, :BoyaSiparisDetayId, :MusteriSiparisNo, :ReferansSiparisNo, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
         :ProsesId, :ProsesAdi, :SiparisProsesId, :SiparisProsesAdi, :HamRenkAdi, :SiparisBirimFiyat, :SiparisDovizId, :PaketlemeSekliId, :PaketlemeSekliAdi, :FirmaId,
         :NetKilo, :BrutKilo, :NetMetre, :BrutMetre, :KapAdedi, :ReferansKodu, :AltBezHamId, :AltBezHamAdi, :SiparisRenkAdi do
    begin
      MusteriPartiNo = '';

      For Select HamStokMaster.MusteriPartiNo from PartiStokDetay
        left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
        left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
        where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.MusteriPartiNo is not null and HamStokMaster.MusteriPartiNo <> ''
        group by HamStokMaster.MusteriPartiNo
        into :FMusteriPartiNo do
      begin
        if (Char_Length(MusteriPartiNo) < 90) then
        begin
          if (FMusteriPartiNo is not null) then
          begin
            if (MusteriPartiNo <> '' ) then MusteriPartiNo = MusteriPartiNo || cast(', ' as varchar(2) character set UTF8);
            MusteriPartiNo = MusteriPartiNo || FMusteriPartiNo;
          end
        end
      end

      if (Fason = 0 and ASatisBoyaSiparisFiyatGorebilir = 0) then
      begin
        SiparisBirimFiyat = null;
        SiparisDovizId    = null;
      end
      else if (Fason = 1 and AFasonBoyaSiparisFiyatGorebilir = 0) then
      begin
        SiparisBirimFiyat = null;
        SiparisDovizId    = null;
      end

      CikisBobinSayisi = null;

      Select Sum(IplikPaketleme.BobinSayisi) from PartiStokTarti
        left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
        where PartiStokTarti.PartiSinifID = :PartiSinifID and PartiStokTarti.BoyaIrsaliyeDetayId = :BoyaIrsaliyeDetayId
        into :CikisBobinSayisi;

      suspend;
    end
  end
end
