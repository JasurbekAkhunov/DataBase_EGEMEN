CREATE PROCEDURE MC_FIRMADETAYLIHAMSTOKLISTESI_1(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    ATAMIR INTEGER,
    ARAPORHAMTIPI INTEGER,
    ATAMIRTIPIID INTEGER,
    ASIFIROLANLARIDOK INTEGER,
    AHAMID INTEGER,
    ALOT VARCHAR(150),
    AKULLANICIID INTEGER)
RETURNS (
    HAMSTOKDETAYID INTEGER,
    HAMSTOKMASTERID INTEGER,
    TAMIRMI VARCHAR(10),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    HAMADI VARCHAR(255),
    MUSTERIPARTINO VARCHAR(30),
    PUS INTEGER,
    FEINE INTEGER,
    RENK VARCHAR(20),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    MUSTERISIPARISNO VARCHAR(50),
    IADEEDILENKILO NUMERIC(12,3),
    IADEEDILENMETRE NUMERIC(12,3),
    PARTIIPTALKILO NUMERIC(12,3),
    PARTIIPTALMETRE NUMERIC(12,3))
AS
Declare variable Tamir integer;
declare variable HamTipiId integer;
declare variable Okey     integer;
declare variable SarfFisiKilo    Numeric(12,3);
declare variable SarfFisiMetre   Numeric(12,3);
declare variable hstIplik               integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  for select HamStokDetay.HamStokDetayId, HamStokMaster.HamStokMasterId , HamStokMaster.Tamir, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             HamStokIrsaliye.IrsaliyeNo, Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Renk,
             HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo, HamStokDetay.KapAdedi, HamStokDetay.Kilo, HamStokDetay.Metre, Ham.HamTipiId from HamStokIrsaliye
  left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
  left join HamStokDetay on HamStokDetay.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join Ham on Ham.HamId = HamStokDetay.HamId
  where HamStokIrsaliye.FirmaId = :AFirmaId and HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
        HamStokDetay.HamStokDetayId is not null and (HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
        (HamStokMaster.Tamir = :ATamir or :ATamir is null) and
        (DisTamir.TamirTipiID = :ATamirTipiId or :ATamirTipiId is null) and
        (HamStokDetay.Fason = 1) and
        (HamStokDetay.HamId = :AHamId or :AHamId is null) and
        ( (Ham.HamTipiId = 0 and :ARaporHamTipi = 0) or (Ham.HamTipiId <> 0 and :ARaporHamTipi = 1) or :ARaporHamTipi is null ) and
        ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporHamTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporHamTipi = 1) or :ARaporHamTipi is null) and
        (Lot = :ALot or :ALot is null)
  into :HamStokDetayId, :HamStokMasterId , :Tamir, :IrsaliyeTarihi, :EvrakTipi, :IrsaliyeNo, :HamAdi, :Pus, :Feine, :Renk, :MusteriPartiNo,
       :MusteriSiparisNo, :KapAdedi, :Kilo, :Metre, :HamTipiId do
  begin
    select sum(PartiStokDetay.Kilo), sum(PartiStokDetay.Metre) from RafDetay
    left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
    where RafDetay.HamStokDetayId = :HamStokDetayId
    into :PartilenenKilo,  :PartilenenMetre;

    select sum(HamStokSarfFisiDetay.Kilo), sum(HamStokSarfFisiDetay.Metre) from RafDetay
    left join HamStokSarfFisiDetay on HamStokSarfFisiDetay.RafDetayId = RafDetay.RafDetayId
    where RafDetay.HamStokDetayId = :HamStokDetayId
    into :SarfFisiKilo, :SarfFisiMetre;


    select sum(RafDetay.Kilo), sum(RafDetay.Metre) from RafDetay
    where RafDetay.HamStokDetayId = :HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
    into :PartiIptalKilo, :PartiIptalMetre;

    select sum(RafDetay.IadeEdilenKilo), sum(RafDetay.IadeEdilenMetre) from RafDetay
      where RafDetay.HamStokDetayId = :HamStokDetayId
      into :IadeEdilenKilo, :IadeEdilenMetre;

    if (PartilenenKilo is null) then KalanKilo = Kilo;
    else KalanKilo = Kilo - PartilenenKilo;


    if (IadeEdilenKilo is not null) then KalanKilo = KalanKilo - IadeEdilenKilo;

    if (PartilenenMetre is null) then KalanMetre = Metre;
    else KalanMetre = Metre - PartilenenMetre;
    if (IadeEdilenMetre is not null) then KalanMetre = KalanMetre - IadeEdilenMetre;

    if (PartiIptalKilo is null) then PartiIptalKilo = 0;
    if (PartiIptalMetre is null) then PartiIptalMetre = 0;

    if (SarfFisiKilo is null) then SarfFisiKilo = 0;
    if (SarfFisiMetre is null) then SarfFisiMetre = 0;

    KalanKilo = KalanKilo + PartiIptalKilo - SarfFisiKilo;
    KalanMetre = KalanMetre + PartiIptalMetre  - SarfFisiMetre;


    if (Tamir = 1) then Tamirmi = 'Tamir';
    else Tamirmi = null;
    if (KalanKilo is null) then KalanKilo = 0;
    if (KalanMetre is null) then KalanMetre = 0;

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';

    if (ASifirOlanlariDok = 1 or (ASifirOlanlariDok = 0 and KalanKilo > 0) ) then suspend;
  end

  HamStokDetayId = Null;
  HamStokMasterId = Null;
  PartilenenKilo = null;
  PartilenenMetre = null;
  PartiIptalKilo = null;
  PartiIptalMetre = null;

  IadeEdilenKilo        = Null;
  IadeEdilenMetre       = Null;
  SarfFisiKilo = null;
  SarfFisiMetre = null;
  KalanKilo             = Null;
  KalanMetre            = Null;

  for select HamStokIrsaliye.IrsaliyeTarihi, HamStokMaster.HamStokMasterId, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.EvrakTipi, Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Renk,
             HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo, HamStokMaster.Tamir, HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Metre,
             Ham.HamTipiId from HamStokIrsaliye
  left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
  left join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
  left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
  left join Ham on Ham.HamId = HamStokMaster.HamId
  where (HamStokIrsaliye.FirmaId = :AFirmaId) and HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
        (HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
        (HamStokMaster.Tamir = :ATamir or :ATamir is null) and
        (DisTamir.TamirTipiID = :ATamirTipiId or :ATamirTipiId is null) and
        (HamStokMaster.Fason = 1) and
        ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporHamTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporHamTipi = 1) or :ARaporHamTipi is null) and
        (HamStokMaster.Lot = :ALot or :ALot is null) and
        HamStokMaster.OnayDurumu = 0 and HamStokMaster.OnaylayanKullaniciId is null
  into :IrsaliyeTarihi, :HamStokMasterId, :IrsaliyeNo, :EvrakTipi, :HamAdi,
       :Pus, :Feine, :Renk, :MusteriPartiNo, :MusteriSiparisNo, :Tamir, :KapAdedi, :Kilo, :Metre, :HamTipiId do
  begin
    if (Tamir = 1) then Tamirmi = 'Tamir';
    else Tamirmi = null;

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';

    KalanKilo      = Kilo;
    KalanMetre     = Metre;
    suspend;
  end
end
