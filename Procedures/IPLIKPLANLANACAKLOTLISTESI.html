CREATE PROCEDURE IPLIKPLANLANACAKLOTLISTESI(
    AIPLIKMAKINATIPIID INTEGER)
RETURNS (
    IPLIKLOTID INTEGER,
    TARIHI TIMESTAMP,
    LOT VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    HARMANIDSTR VARCHAR(30),
    HARMANADI VARCHAR(50),
    HARMANCINSIADI VARCHAR(50),
    IPLIKISLETMEID INTEGER,
    IPLIKISLETMEADI VARCHAR(30),
    BOBINGRAMAJI NUMERIC(12,3),
    MIKTAR DOUBLE PRECISION,
    PLANLANANMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    PLANLANDIGIMAKINALAR VARCHAR(255))
AS
declare variable CopsPlanlananMiktar Double Precision;
declare variable BobinPlanlananMiktar Double Precision;
declare variable HarmanYil              integer;
declare variable HarmanId               integer;
declare variable FIplikMakinaId         varchar(10);
declare variable FBitti integer;
begin
  For Select IplikLot.IplikLotId, IplikLot.Tarihi, IplikLot.Lot, IplikLot.HamID, Ham.Adi, IplikLot.MelanjKodu, IplikLot.FlamKodu,
             Harman.HarmanYil, Harman.HarmanID, Harman.Adi, HarmanCinsi.Adi, Harman.IplikIsletmeID, IplikIsletme.Adi,
             IplikLot.BobinGramaji, IplikLot.Miktar, IplikLot.CopsPlanlananMiktar, IplikLot.BobinPlanlananMiktar from IplikLot
    left join Ham on Ham.HamID = IplikLot.HamID
    left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitID
    left join HarmanCinsi on HarmanCinsi.HarmanCinsiID = Harman.HarmanCinsiId
    left join IplikIsletme on IplikIsletme.IplikIsletmeID = Harman.IplikIsletmeId
    where IplikLot.Kapali = 0 and IplikLot.Tamir = 0 and
         ( (:AIplikMakinaTipiId = 0 and IplikLot.Miktar > IplikLot.CopsPlanlananMiktar) or
           (:AIplikMakinaTipiId = 1 and IplikLot.Miktar > IplikLot.BobinPlanlananMiktar) )
    into :IplikLotId, :Tarihi, :Lot, :HamID, :HamAdi, :MelanjKodu, :FlamKodu, :HarmanYil, :HarmanID, :HarmanAdi, :HarmanCinsiAdi,
         :IplikIsletmeID, :IplikIsletmeAdi, :BobinGramaji, :Miktar, :CopsPlanlananMiktar, :BobinPlanlananMiktar do
  begin
    if (AIplikMakinaTipiId = 0) then
    begin
      PlanlananMiktar = CopsPlanlananMiktar;
    end
    else if (AIplikMakinaTipiId = 1) then
    begin
      PlanlananMiktar = BobinPlanlananMiktar;
    end
    else PlanlananMiktar = Miktar;

    KalanMiktar = Miktar - PlanlananMiktar;

    PlanlandigiMakinalar  = '';

    for Select IplikUretimIsEmri.IplikMakinaId, Min(Bitti) from IplikUretimIsEmri
    where IplikUretimIsEmri.IplikLotID = :IplikLotID and IplikUretimIsEmri.IplikMakinaTipiId = :AIplikMakinaTipiId
    group by IplikUretimIsEmri.IplikMakinaId
    into :FIplikMakinaId, :FBitti do
    begin
      if ( Char_Length(PlanlandigiMakinalar) + Char_Length(FIplikMakinaId) < 230) then
      begin
        if (PlanlandigiMakinalar <> '') then PlanlandigiMakinalar = PlanlandigiMakinalar || cast(', ' as varchar(2) character set UTF8);
        PlanlandigiMakinalar = PlanlandigiMakinalar || FIplikMakinaId;

        if (FBitti = 0) then PlanlandigiMakinalar = PlanlandigiMakinalar || '(-)';
        else PlanlandigiMakinalar = PlanlandigiMakinalar || '(+)';
      end
    end

    if (KalanMiktar > 0) then suspend;
  end
end
