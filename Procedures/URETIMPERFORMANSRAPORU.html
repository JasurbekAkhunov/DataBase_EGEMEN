CREATE PROCEDURE URETIMPERFORMANSRAPORU(
    ABASLANGICTARIH TIMESTAMP)
RETURNS (
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    MAKINAGRUBU VARCHAR(4),
    HAFTA1URETIM DOUBLE PRECISION,
    HAFTA1HEDEF DOUBLE PRECISION,
    HAFTA2URETIM DOUBLE PRECISION,
    HAFTA2HEDEF DOUBLE PRECISION,
    HAFTA3URETIM DOUBLE PRECISION,
    HAFTA3HEDEF DOUBLE PRECISION,
    HAFTA4URETIM DOUBLE PRECISION,
    HAFTA4HEDEF DOUBLE PRECISION,
    HEDEFLENENURETIM DOUBLE PRECISION,
    HEDEFTARIH TIMESTAMP)
AS
declare variable BasH1 TimeStamp;
declare variable BitH1 TimeStamp;
declare variable BasH2 TimeStamp;
declare variable BitH2 TimeStamp;
declare variable BasH3 TimeStamp;
declare variable BitH3 TimeStamp;
declare variable BasH4 TimeStamp;
declare variable BitH4 TimeStamp;
declare variable FVardiyaSayisi   integer;
declare variable FCikisTarihi      TimeStamp;
declare variable FKilo             Double Precision;
declare variable FHedeflenenUretim  Double Precision;
begin
  BasH1 = ABaslangicTarih;
  BitH1 = BasH1 + 7;

  BasH2 = BitH1;
  BitH2 = BasH2 + 7;

  BasH3 = BitH2;
  BitH3 = BasH3 + 7;

  BasH4 = BitH3;
  BitH4 = BasH4 + 7;

  Select Max(Tarih) from MakinaGunlukUretimHedefi where MakinaGunlukUretimHedefi.Tarih <= :BitH4 into :HedefTarih;

  for select Makina.MakinaId, Makina.Adi, Makina.MakinaGrubu, MakinaGunlukUretimHedefi.HedeflenenUretim from Makina
    left join MakinaGunlukUretimHedefi on MakinaGunlukUretimHedefi.MakinaId = Makina.MakinaId and MakinaGunlukUretimHedefi.Tarih = :HedefTarih
    where Makina.UretimRaporlarinaDahil = 1
    order by Makina.MakinaGrubu, Makina.MakinaId
    into :MakinaId, :MakinaAdi, :MakinaGrubu, :HedeflenenUretim do
  begin
    FHedeflenenUretim = HedeflenenUretim * 7;

    Select Sum(VardiyaSayisi) from MakinaVardiya
      where MakinaVardiya.MakinaId = :MakinaId and MakinaVardiya.Tarih > :BasH1 - 1 and MakinaVardiya.Tarih < :BitH1 - 1 into :FVardiyaSayisi;
    if (FVardiyaSayisi is null) then FVardiyaSayisi = 0;

    Hafta1Hedef = FHedeflenenUretim * FVardiyaSayisi / 21;

    Select Sum(VardiyaSayisi) from MakinaVardiya
      where MakinaVardiya.MakinaId = :MakinaId and MakinaVardiya.Tarih > :BasH2 - 1 and MakinaVardiya.Tarih < :BitH2 - 1into :FVardiyaSayisi;
    if (FVardiyaSayisi is null) then FVardiyaSayisi = 0;

    Hafta2Hedef = FHedeflenenUretim * FVardiyaSayisi / 21;

    Select Sum(VardiyaSayisi) from MakinaVardiya
      where MakinaVardiya.MakinaId = :MakinaId and MakinaVardiya.Tarih > :BasH3 - 1 and MakinaVardiya.Tarih < :BitH3 - 1 into :FVardiyaSayisi;
    if (FVardiyaSayisi is null) then FVardiyaSayisi = 0;

    Hafta3Hedef = FHedeflenenUretim * FVardiyaSayisi / 21;

    Select Sum(VardiyaSayisi) from MakinaVardiya
      where MakinaVardiya.MakinaId = :MakinaId and MakinaVardiya.Tarih > :BasH4 - 1 and MakinaVardiya.Tarih < :BitH4 - 1 into :FVardiyaSayisi;
    if (FVardiyaSayisi is null) then FVardiyaSayisi = 0;

    Hafta4Hedef = FHedeflenenUretim * FVardiyaSayisi / 21;

    Hafta1Uretim = 0;
    Hafta2Uretim = 0;
    Hafta3Uretim = 0;
    Hafta4Uretim = 0;

    for select PartiAsamaStokDetay.CikisTarihi, PartiAsamaStokDetay.Kilo from PartiAsamaStokDetay
      where PartiAsamaStokDetay.MakinaId = :MakinaId and PartiAsamaStokDetay.CikisTarihi >= :BasH1 and
            PartiAsamaStokDetay.CikisTarihi < :BitH4
      into :FCikisTarihi, :FKilo do
    begin
      if (FCikisTarihi >= BasH1 and FCikisTarihi < BitH1) then Hafta1Uretim = Hafta1Uretim + FKilo;
      else if (FCikisTarihi >= BasH2 and FCikisTarihi < BitH2) then Hafta2Uretim = Hafta2Uretim + FKilo;
      else if (FCikisTarihi >= BasH3 and FCikisTarihi < BitH3) then Hafta3Uretim = Hafta3Uretim + FKilo;
      else if (FCikisTarihi >= BasH4 and FCikisTarihi < BitH4) then Hafta4Uretim = Hafta4Uretim + FKilo;
    end

    For Select HAktarmaIsEmri.BitisTarihi, Sum(HAktarmaIsEmriDetay.Kilo) from HAktarmaIsEmri
      left join HAktarmaIsEmriDetay on HAktarmaIsEmriDetay.HAktarmaIsEmriKayitId = HAktarmaIsEmri.HAktarmaIsEmriKayitId
      where HAktarmaIsEmri.MakinaId = :MakinaId and  HAktarmaIsEmri.BitisTarihi >= :BasH1 and HAktarmaIsEmri.BitisTarihi < :BitH4
      group By HAktarmaIsEmri.HAktarmaIsEmriKayitId, HAktarmaIsEmri.BitisTarihi
      into :FCikisTarihi, :FKilo do
    begin
      if (FCikisTarihi >= BasH1 and FCikisTarihi < BitH1) then Hafta1Uretim = Hafta1Uretim + FKilo;
      else if (FCikisTarihi >= BasH2 and FCikisTarihi < BitH2) then Hafta2Uretim = Hafta2Uretim + FKilo;
      else if (FCikisTarihi >= BasH3 and FCikisTarihi < BitH3) then Hafta3Uretim = Hafta3Uretim + FKilo;
      else if (FCikisTarihi >= BasH4 and FCikisTarihi < BitH4) then Hafta4Uretim = Hafta4Uretim + FKilo;
    end
    suspend;
  end
end
