AS
declare variable APartiKayitId integer;
declare variable ABoyaSiparisKayitId integer;
declare variable EnIstenenInt integer;
declare variable APartiYil integer;
declare variable APartiId integer;
declare variable APartiSiraNo integer;
declare variable virgulPos  integer;
declare variable NoktaPos  integer;
declare variable Temp  Varchar(25);
begin
  VirgulPos = PosBul(APartiler, ',');

  While (VirgulPos > 1) do
  begin
    Temp = EgemenCopyStr(APartiler, 1, VirgulPos -1);
    NoktaPos = PosBul(Temp, '.');
    APartiYil = EgemenCopyStr(Temp, 1, NoktaPos -1);
    Temp = EgemenCopyStr(Temp, NoktaPos + 1, Char_Length(Temp) - NoktaPos);
    NoktaPos = PosBul(Temp, '.');
    APartiID = EgemenCopyStr(Temp, 1, NoktaPos -1);
    APartiSiraNo = EgemenCopyStr(Temp, NoktaPos + 1, Char_Length(Temp) - NoktaPos);

    ABoyaSiparisKayitId = NUll;
    APartiKayitId = null;

    For Select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi, Parti.AcilisTarihi, PartiStok.PartiSinifId, PartiStok.BoyaSiparisDetayId,
               PartiStok.HamAdiToplami, PartiStok.KKBitisTarihi, PartiStok.Kilo, PartiStok.Metre, Parti.receteId, Recete.Adi,
               IslemGrubu.Adi, IslemTipi.Adi, Parti.Fason, BoyaSiparisdetay.MusteriRenkKoduveAdi, BoyaSiparisdetay.MusteriSiparisNo,
               Ham.Adi2, Ham.Adi3, Ham.Adi4,
               PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.PartiSinifId, PSKaliteKontrol.Tarih, PSKaliteKontrol.MakinaId,
               PSKaliteKontrol.KaliteNo, PSKaliteKontrol.TopNo, PSKaliteKontrol.Kilo, PSKaliteKontrol.Dara, PSKaliteKontrol.Metre, PSKaliteKontrol.BrutMetre, PSKaliteKontrol.En, PSKaliteKontrol.Gramaj,
               PSKaliteKontrol.BaslangicEn, PSKaliteKontrol.OrtaEn, PSKaliteKontrol.BitisEn, PSKaliteKontrol.OrtalamaEn, PSKaliteKontrol.Donme, PSKaliteKontrol.Egrilik,
               PSKaliteKontrol.PaketlemeId, PSKaliteKontrol.HataPuani, BoyaSiparisDetay.EnIstenen, PSKaliteKontrol.Aciklama, PartiStok.HacimEn, PartiStok.HacimBoy,
               PartiStok.HacimYukseklik,
               BoyaSiparisDetay.IstenenRaporBoyu, BoyaSiparisDetay.IstenenOzelTuse,
               BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.DonmeIstenen, BoyaSiparisDetay.CekmezlikEnIstenen,
               BoyaSiparisDetay.CekmezlikBoyIstenen, BoyaSiparisDetay.YikamaHasligiIstenen, BoyaSiparisDetay.SansIstenen,
               BoyaSiparisDetay.ToleransEn, BoyaSiparisDetay.ToleransGramaj, BoyaSiparisDetay.ToleransEndenCekmezlik,
               BoyaSiparisDetay.ToleransBoydanCekmezlik, BoyaSiparisDetay.ToleransDonme, BoyaSiparisDetay.ToleransSans,
               BoyaSiparisDetay.MaximumFireOrani, BoyaSiparisDetay.SatisFireOngoruOrani, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
               IK.Adi, PSKaliteKontrol.InsertTarihi, SK.Adi, PSKaliteKontrol.DegisimTarihi, PartiStok.PartiSiraNo,
               BoyaSiparisDetay.KKLabel1, BoyaSiparisDetay.KKLabel2, BoyaSiparisDetay.KKLabel3, BoyaSiparisDetay.KKLabel4, BoyaSiparisDetay.KKLabel5, BoyaSiparisDetay.KKLabel6, BoyaSiparisDetay.KKLabel7,
               BoyaSiparisDetay.KKDeger1, BoyaSiparisDetay.KKDeger2, BoyaSiparisDetay.KKDeger3, BoyaSiparisDetay.KKDeger4, BoyaSiparisDetay.KKDeger5, BoyaSiparisDetay.KKDeger6, BoyaSiparisDetay.KKDeger7,
               OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.BelgeHamAdi, DesenVaryant.DesenID, Desen.Adi DesenAdi, DesenVaryant.VaryantAdi, PSKaliteKontrol.PartiRenkLotNo,
               BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.IkinciKaliteBirimFiyat, BoyaSiparisDetay.FasonBirimFiyati, BoyaSiparisDetay.BaskiBirimFiyati,
               PSKaliteKontrol.PSKaliteKontrolAsamaId, PartiStokTarti.BoyaIrsaliyeDetayId, BoyaSiparisDetay.TupAcikEnMayli, BaskiTipi.Adi, SiparisProses.Adi from Parti
      left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
      left join PSKaliteKontrolAsama on PSKaliteKontrolAsama .PartiSinifId = PartiStok.PartiSinifId
      left join PSKaliteKontrol on PSKaliteKontrol.PSKaliteKontrolAsamaId = PSKaliteKontrolAsama.PSKaliteKontrolAsamaId
      left join Firma on Firma.FirmaId = Parti.FirmaId
      left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = BoyaSiparisDetay.OrmeSiparisDetayID
      left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
      left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
      Left Join Kullanici IK on IK.KullaniciId = PSKaliteKontrol.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = PSKaliteKontrol.KullaniciId
      left join DesenVaryant on DesenVaryant.DesenVaryantID = Parti.DesenVaryantID
      left join Desen on Desen.DesenID = DesenVaryant.DesenID
      left join BaskiTipi on BaskiTipi.BaskiTipiId = BoyaSiparisDetay.BaskiTipiId
      left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = PsKaliteKontrol.PartiStokTartiId
      where Parti.PartiYil = :APartiYil and
            Parti.PartiID = :APartiID and
            PartiStok.PartiSiraNo = :APartiSiraNo and
            PSKaliteKontrolAsama.SonKayit = 1
      order by PSKaliteKontrol.PartiSinifId, PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.TopNo
    into :PartiKayitId, :PartiYil, :PartiId, :BoyaSiparisYil, :BoyaSiparisId, :PartigrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi, :AcilisTarihi, :PartiSinifId, :BoyaSiparisDetayId,
           :HamAdiToplami, :KaliteKontrolTarihi, :PartiBrutKilo, :PartiBrutMetre, :ReceteId, :ReceteAdi, :IslemGrubuAdi, :IslemTipiAdi, :Fason,
           :MusteriRenkKoduveAdi, :MusteriSiparisNo, :HamAdi2, :HamAdi3, :HamAdi4,
           :PSKaliteKontrolId, :PartiSinifId, :Tarih, :MakinaId, :KaliteNo, :TopNo, :BrutKilo, :Dara, :Metre, :BrutMetre, :En, :Gramaj,
           :BaslangicEn, :OrtaEn, :BitisEn, :OrtalamaEn, :Donme, :Egrilik, :PaketlemeId, :HataPuani,
           :EnIstenen, :Aciklama, :HacimEn, :HacimBoy, :HacimYukseklik,
           :IstenenRaporBoyu, :IstenenOzelTuse,
           :GramajIstenen, :DonmeIstenen, :CekmezlikEnIstenen,
           :CekmezlikBoyIstenen, :YikamaHasligiIstenen, :SansIstenen,
           :ToleransEn, :ToleransGramaj, :ToleransEndenCekmezlik,
           :ToleransBoydanCekmezlik, :ToleransDonme, :ToleransSans,
           :MaximumFireOrani, :SatisFireOngoruOrani, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :PartiSiraNo,
           :KKLabel1, :KKLabel2, :KKLabel3, :KKLabel4, :KKLabel5, :KKLabel6, :KKLabel7, :KKDeger1, :KKDeger2, :KKDeger3, :KKDeger4, :KKDeger5, :KKDeger6, :KKDeger7,
           :OrmeSiparisSiraNo, :BelgeHamAdi, :DesenId, :desenadi, :VaryantAdi, :PartiRenkLotNo, :SiparisBirimFiyat, :SiparisIkinciKaliteBirimFiyat, :SiparisFasonBirimFiyat, :SiparisBaskiBirimFiyat,
           :PSKaliteKontrolAsamaId, :BoyaIrsaliyeDetayId, :TupAcikEnMayli, :BaskiTipiAdi, :SiparisProsesAdi
           do
    begin
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      else BoyaSiparisIdStr = '';

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      Hacim = HacimEn * HacimBoy * HacimYukseklik;
      Hacim = Hacim / 1000000;

      BarkodID = 'K-' || EgemenIntegerToStr(PSKaliteKontrolId);

      Kilo = BrutKilo - Dara;
      EnSapmasiCm = null;
      EnSapmasiOrani = null;

      if (EgemenStrToIntOlurmu(EnIstenen) = 1) then
      begin
        EnIstenenInt = EgemenStrtoInt(EnIstenen);
        EnSapmasiCm = OrtalamaEn - EnIstenenInt;
        EnSapmasiOrani = EnSapmasiCm / EnIstenenInt * 100;
      end

      SadeceTarih = Tarih;
      suspend;
    end
    if (Char_Length(APartiler) > VirgulPos) then APartiler = EgemenCopyStr(APartiler, VirgulPos + 1, Char_Length(APartiler) - VirgulPos);
    else APartiler = '';

    VirgulPos = PosBul(APartiler, ',');
  end
end
