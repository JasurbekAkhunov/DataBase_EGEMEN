CREATE PROCEDURE KALITEKONTROLRAPORU_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    ARECETEID VARCHAR(20),
    AFIRMAID VARCHAR(20),
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AIMALATTIPIID INTEGER,
    AHAMID INTEGER,
    ABOYASIPSIRANO INTEGER,
    AONAYDURUMU INTEGER,
    AAKTARMADAHIL SMALLINT,
    AHAMTIPIID INTEGER)
RETURNS (
    PSKALITEKONTROLID INTEGER,
    BARKODID VARCHAR(20),
    PARTISINIFID INTEGER,
    MAKINAID VARCHAR(10),
    TARIH TIMESTAMP,
    KALITENO INTEGER,
    TOPNO INTEGER,
    BRUTKILO NUMERIC(12,3),
    DARA NUMERIC(12,3),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    EN VARCHAR(10),
    GRAMAJ VARCHAR(10),
    BASLANGICEN VARCHAR(10),
    ORTAEN VARCHAR(10),
    BITISEN VARCHAR(10),
    ORTALAMAEN INTEGER,
    ENSAPMASICM INTEGER,
    ENSAPMASIORANI INTEGER,
    DONME VARCHAR(10),
    EGRILIK VARCHAR(10),
    PAKETLEMEID VARCHAR(15),
    ACIKLAMA VARCHAR(100),
    HATAPUANI INTEGER,
    HACIMEN INTEGER,
    HACIMBOY INTEGER,
    HACIMYUKSEKLIK INTEGER,
    HACIM DOUBLE PRECISION,
    ISTENENRAPORBOYU VARCHAR(100),
    ISTENENOZELTUSE VARCHAR(25),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    DONMEISTENEN VARCHAR(10),
    CEKMEZLIKENISTENEN DOUBLE PRECISION,
    CEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    YIKAMAHASLIGIISTENEN VARCHAR(10),
    SANSISTENEN VARCHAR(10),
    TOLERANSEN VARCHAR(10),
    TOLERANSGRAMAJ VARCHAR(10),
    TOLERANSENDENCEKMEZLIK VARCHAR(10),
    TOLERANSBOYDANCEKMEZLIK VARCHAR(10),
    TOLERANSDONME VARCHAR(10),
    TOLERANSSANS VARCHAR(10),
    MAXIMUMFIREORANI DOUBLE PRECISION,
    SATISFIREONGORUORANI DOUBLE PRECISION,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SONKAYIT INTEGER,
    PSKALITEKONTROLASAMAID INTEGER,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP,
    MUSTERIACIKLAMA VARCHAR(50),
    LABRECETEKODU VARCHAR(20),
    PARCASAYISI INTEGER,
    PARCAMETRE NUMERIC(12,3),
    PAKETLEMEHATAPUANI DOUBLE PRECISION,
    KKPARTIRENKLOTNO INTEGER)
AS
declare variable EnIstenenInt integer;
begin

  For Select PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.PartiSinifId, PSKaliteKontrol.Tarih, PSKaliteKontrol.MakinaId,
             PSKaliteKontrol.KaliteNo, PSKaliteKontrol.TopNo, PSKaliteKontrol.Kilo, PSKaliteKontrol.Dara, PSKaliteKontrol.Metre, PSKaliteKontrol.BrutMetre, PSKaliteKontrol.En, PSKaliteKontrol.Gramaj,
             PSKaliteKontrol.BaslangicEn, PSKaliteKontrol.OrtaEn, PSKaliteKontrol.BitisEn, PSKaliteKontrol.OrtalamaEn, PSKaliteKontrol.Donme, PSKaliteKontrol.Egrilik,
             PSKaliteKontrol.PaketlemeId, PSKaliteKontrol.HataPuani, BoyaSiparisDetay.EnIstenen, PSKaliteKontrol.Aciklama, PartiStok.HacimEn, PartiStok.HacimBoy,
             PartiStok.HacimYukseklik,
             BoyaSiparisDetay.IstenenRaporBoyu, BoyaSiparisDetay.IstenenOzelTuse,
             BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.DonmeIstenen, BoyaSiparisDetay.CekmezlikEnIstenen,
             BoyaSiparisDetay.CekmezlikBoyIstenen, BoyaSiparisDetay.YikamaHasligiIstenen, BoyaSiparisDetay.SansIstenen,
             BoyaSiparisDetay.ToleransEn, BoyaSiparisDetay.ToleransGramaj, BoyaSiparisDetay.ToleransEndenCekmezlik,
             BoyaSiparisDetay.ToleransBoydanCekmezlik, BoyaSiparisDetay.ToleransDonme, BoyaSiparisDetay.ToleransSans,
             BoyaSiparisDetay.MaximumFireOrani, BoyaSiparisDetay.SatisFireOngoruOrani, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             PSKaliteKontrol.PSKaliteKontrolAsamaId, PSKaliteKontrolAsama.SonKayit, PSKaliteKontrol.OnayDurumu, PSKaliteKontrol.OnayKullaniciId, OnayKullanici.Adi, PSKaliteKontrol.OnayTarihi, PSKaliteKontrol.MusteriAciklama,
             BoyaSiparisDetay.LabReceteKodu, PSKaliteKontrol.ParcaSayisi, PSKaliteKontrol.ParcaMetre, PsKaliteKontrol.PaketlemeHataPuani, PartiStok.KKPartiRenkLotNo
             from KaliteKontrolRaporu_1(:ABaslangicTarihi, :ABitisTarihi, :APartiYil, :APartiId, :AReceteId, :AFirmaId, :ABoyaSiparisYil, :ABoyaSiparisId, :AImalatTipiID, :AHamID, :ABoyaSipSiraNo, :AOnayDurumu, :AHamTipiId)
    left join PSKaliteKontrol on PSKaliteKontrol.PartiSinifID = KaliteKontrolRaporu_1.PartiSinifID
    left join PartiStok on PartiStok.PartiSinifID = PSKaliteKontrol.PartiSinifID
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join PSKaliteKontrolAsama on PSKaliteKontrolAsama.PSKaliteKontrolAsamaID = PSKaliteKontrol.PSKaliteKontrolAsamaID
    Left Join Kullanici OnayKullanici on OnayKullanici.KullaniciId = PSKaliteKontrol.OnayKullaniciId
    where PSKaliteKontrol.PartiSinifId is not null  and
          (PSKaliteKontrol.OnayDurumu = :AOnayDurumu or :AOnayDurumu is Null)  and
          (PSKaliteKontrolAsama.AktarmaKontrol = :AAktarmaDahil)
    order by PSKaliteKontrol.PartiSinifId, PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.TopNo
  into :PSKaliteKontrolId, :PartiSinifId, :Tarih, :MakinaId, :KaliteNo, :TopNo, :BrutKilo, :Dara, :Metre, :BrutMetre, :En, :Gramaj,
       :BaslangicEn, :OrtaEn, :BitisEn, :OrtalamaEn, :Donme, :Egrilik, :PaketlemeId, :HataPuani,
       :EnIstenen, :Aciklama, :HacimEn, :HacimBoy, :HacimYukseklik,
       :IstenenRaporBoyu, :IstenenOzelTuse,
       :GramajIstenen, :DonmeIstenen, :CekmezlikEnIstenen,
       :CekmezlikBoyIstenen, :YikamaHasligiIstenen, :SansIstenen,
       :ToleransEn, :ToleransGramaj, :ToleransEndenCekmezlik,
       :ToleransBoydanCekmezlik, :ToleransDonme, :ToleransSans,
       :MaximumFireOrani, :SatisFireOngoruOrani, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :PSKaliteKontrolAsamaId, :SonKayit,
       :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi, :MusteriAciklama, :LabReceteKodu, :ParcaSayisi, :ParcaMetre, :PaketlemeHataPuani, :KKPartiRenkLotNo do
  begin
    OnayDurumuStr = '';
    if (OnayDurumu = 1) then OnayDurumuStr = 'Okey';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Red';


    Hacim = HacimEn * HacimBoy * HacimYukseklik;
    Hacim = Hacim / 1000000;

    BarkodID = 'K-' || EgemenIntegerToStr(PSKaliteKontrolId);

    Kilo = BrutKilo - Dara;
    EnSapmasiCm = null;
    EnSapmasiOrani = null;

    if (EgemenStrToIntOlurmu(EnIstenen) = 1) then
    begin
      EnIstenenInt = EgemenStrtoInt(EnIstenen);
      EnSapmasiCm = OrtalamaEn - EnIstenenInt;
      EnSapmasiOrani = EnSapmasiCm / EnIstenenInt * 100;
    end
    suspend;
  end
end
