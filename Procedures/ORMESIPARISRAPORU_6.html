CREATE PROCEDURE ORMESIPARISRAPORU_6(
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    AIPLIKSIPARISI INTEGER,
    ABOYALIIPLIKSIPARISI INTEGER,
    ALYCRASIPARISI INTEGER,
    AHAMKUMASSIPARISI INTEGER,
    ABOYALIKUMASSIPARISI INTEGER,
    AIPLIKBOYALIKUMASSIPARISI INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AKUMASCINSIID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    AYURTDISI INTEGER,
    AKUMASTEMINSIPARISI INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ASINIF VARCHAR(1),
    AKALITEID VARCHAR(20),
    ASIPONAYDURUMU SMALLINT,
    AMALIONAYDURUMU SMALLINT,
    AKAPATMATARIHIBASLANGIC TIMESTAMP,
    AKAPATMATARIHIBITIS TIMESTAMP,
    AIMALATTIPIID INTEGER)
RETURNS (
    ORMEPLANLAMARECETEID INTEGER,
    ORMEPLANLAMAID INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(250),
    MELANJKODU VARCHAR(20),
    RECETEFIREORANI DOUBLE PRECISION,
    KULLANIMORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    URETICIID INTEGER,
    URETICIADI ADI40)
AS
declare variable TahminiFireOrani Double Precision;
declare variable UrunHamId Integer;
declare variable OrmeTalimattaReceteFiresiYap Integer;
declare variable OtomatikUCikislardaReceteFire Integer;
begin
  For select OrmeSiparisRaporu_5.OrmePlanlamaId, OrmePlanlamaRecete.TahminiFireOrani, OrmePlanlamaRecete.OrmePlanlamaReceteId, OrmePlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, OrmePlanlamaRecete.RenkId, Renk.Adi, OrmePlanlamaRecete.Lot, OrmePlanlamaRecete.KullanimOrani, OrmePlanlamaRecete.CikisMiktari, OrmePlanlamaRecete.IadeMiktari,
             OrmeSiparisRaporu_5.PlanlananMiktar * OrmePlanlamaRecete.KullanimOrani / 100, OrmePlanlamaRecete.MelanjKodu, OrmeSiparisRaporu_5.HamID, Uretici.UreticiId, Uretici.Adi
             from OrmeSiparisRaporu_5(:ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ABaslangicSiparisTarihi, :ABitisSiparisTarihi, :AFirmaId,
              :AFASONSIPARISLER, :AFASONSATISSIPARISLER, :ASATISSIPARISLER, :AIplikSiparisi, :ABoyaliIplikSiparisi, :ALycraSiparisi, :AHamKumasSiparisi, :ABoyaliKumasSiparisi,
              :AIplikBoyaliKumasSiparisi, :AMusteriSiparisNo, :AReferansSiparisNo, :ARaporTipi, :AKapali, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :ASiparisiAlanKullaniciId,
              :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId, :AYurtDisi, :AKumasTeminSiparisi, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASinif, :AKaliteId, :ASipOnayDurumu, :AMaliOnayDurumu, :AKapatmaTarihiBaslangic, :AKapatmaTarihiBitis, :AImalatTipiId)
  left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = OrmeSiparisRaporu_5.OrmePlanlamaId
  left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
  left join Renk on Renk.RenkId = OrmePlanlamaRecete.RenkId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  Left Join Kullanici IK on IK.KullaniciId = OrmePlanlamaRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = OrmePlanlamaRecete.KullaniciId
  Left Join Uretici on Uretici.UreticiId = OrmePlanlamaRecete.UreticiId
  where OrmePlanlamaRecete.OrmePlanlamaId is not null
  order by OrmeSiparisRaporu_5.OrmePlanlamaId
  into :OrmePlanlamaId, :ReceteFireOrani, :OrmePlanlamaReceteId, :HamId, :hamTipiId, :HamTipiAdi,
       :HamAdi, :RenkId, :RenkAdi, :Lot, :KullanimOrani, :CikisMiktari, :IadeMiktari, :KullanilacakMiktar, :MelanjKodu, :UrunHamId, :UreticiId, :UreticiAdi do
  begin
    if (HamTipiID = 0 or HamTipiID = 6) then
      KullanilacakMiktar = KullanilacakMiktar / ((100 - ReceteFireOrani) / 100);
    suspend;
  end
end
