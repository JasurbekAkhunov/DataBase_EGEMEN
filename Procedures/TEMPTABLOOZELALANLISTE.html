CREATE PROCEDURE TEMPTABLOOZELALANLISTE(
    AOZELALANKAYNAGIID INTEGER NOT NULL,
    ATABLOKAYITIDSTR ADI50 NOT NULL)
RETURNS (
    ROZELALANTIPIID INTEGER,
    ROZELALANTIPIDETAYID INTEGER,
    ROZELALANTIPIDETAYADI ADI255)
AS
begin
  For Select Distinct OzelAlanTipi.OzelAlanTipiId, OzelAlanTipiDetay.OzelAlanTipiDetayId, OzelAlanTipiDetay.Adi as OzelAlanTipiDetayAdi From OzelAlanKaynagi
  Join OzelAlanTipi on OzelAlanKaynagi.OzelAlanKaynagiId = OzelAlanTipi.OzelAlanKaynagiId And OzelAlanTipi.Kapali = 0
  Left Join OzelAlan On OzelAlan.OzelAlanTipiId = OzelAlanTipi.OzelAlanTipiId And OzelAlan.TabloKayitIdStr = :ATabloKayitIdStr
  Left Join OzelAlanTipiDetay On OzelAlanTipiDetay.OzelAlanTipiId = OzelAlanTipi.OzelAlanTipiId
  Left Join OzelAlanKolonTipiGetir On 1 = 1
  Where OzelAlanKaynagi.OzelAlanKaynagiId = :AOzelAlanKaynagiId and OzelAlanTipi.OzelAlanKolonTipiId = OzelAlanKolonTipiGetir.OaktListe and OzelAlanTipiDetay.OzelAlanTipiDetayId is not null
  Into :ROzelAlanTipiId, :ROzelAlanTipiDetayId, :ROzelAlanTipiDetayAdi Do
  Begin
    Suspend;
  End
end
