CREATE PROCEDURE ORMEMAKINASONDURUMUNUGETIR(
    AMAKINAID VARCHAR(10))
RETURNS (
    MAKINADURUMUID INTEGER,
    TARIH TIMESTAMP,
    ARIZA SMALLINT,
    TUR INTEGER,
    MEMINGERKOPTU SMALLINT,
    IGNEKOPTU SMALLINT,
    KAPAKACILDI SMALLINT,
    STARTSTOP SMALLINT,
    ARAVERMETARIHI TIMESTAMP,
    ARAVERMESEBEBIID INTEGER,
    ARAVERMESEBEBIADI VARCHAR(50),
    IMALATAARAVERENPERSONELID INTEGER,
    IMALATAARAVERENPERSONELADI VARCHAR(40),
    PARCABEKLIYOR INTEGER,
    ARIZAIHBARPERSONELID INTEGER,
    ARIZAIHBARPERSONELADI VARCHAR(40),
    BAKIMCIPERSONELID INTEGER,
    BAKIMCIPERSONELADI VARCHAR(40),
    ARIZATIPIADI ADI40,
    ARIZAIHBARID INTEGER,
    ARIZAYERIID INTEGER,
    AYAR SMALLINT)
AS
declare variable ImalataAraVermeId integer;
declare variable ImalataAraVerildi SmallInt;
declare variable FOrmeMakinaId MakinaKoduSize;
begin

  Select MakinaDurumuOzet.OrmeMakinaId, DegisimTarihi, Ariza, AraVerme, Tur, MemingerKoptu, IgneKoptu, KapakAcildi, StartStop, Ayar from MakinaDurumuOzet
    where MakinaDurumuOzet.OrmeMakinaId = :AMakinaId
    into :FOrmeMakinaId, :Tarih, :Ariza, :ImalataAraVerildi, :Tur, :MemingerKoptu, :IgneKoptu, :KapakAcildi, :StartStop, :Ayar;

  if (FOrmeMakinaId is Null) then
  begin
    Ariza                 = 0;
    AraVermeTarihi        = Null;
    AraVermeSebebiId      = Null;
    AraVermeSebebiAdi     = Null;
    ImalataAraVerenPersonelId  = Null;
    ImalataAraVerenPersonelAdi = Null;
    ParcaBekliyor         = 0;
    ArizaIhbarPersonelId  = Null;
    ArizaIhbarPersonelAdi = null;
    ArizaIhbarId          = Null;
    ArizaYeriID           = Null;
    BakimciPersonelId     = Null;
    BakimciPersonelAdi    = Null;
    Tur = Null;
    MemingerKoptu         = 0;
    IgneKoptu             = 0;
    KapakAcildi           = 0;
    StartStop             = 0;
  end
  else
  begin
    if (Ariza = 1 or Ariza = 3) then Ariza = 1;
    else Ariza = 0;

    if (Ayar = 1 or Ayar = 3) then Ayar = 1;
    else Ayar = 0;

    if (ImalataAraVerildi = 1 or ImalataAraVerildi = 3) then ImalataAraVerildi = 1;
    else ImalataAraVerildi = 0;

    ParcaBekliyor = 0;
    ArizaIhbarId = Null;

    if (Ariza = 1) then
    begin
      if (ArizaIhbarId is Null) then
        Select Max(OrmeArizaIhbarId) from OrmeArizaIhbar where OrmeArizaIhbar.OrmeMakinaId = :AMakinaId and OrmeArizaIhbar.BittiMakinaDurumuId is null into :ArizaIhbarId;

        Select OrmeArizaIhbar.ArizaYeriID, OrmeArizaIhbar.BaslatanPersonelId, Personel.Adi, OrmeArizaIhbar.ParcaBekliyor, OrmeArizaTipi.Adi, OrmeArizaBakim.PersonelId, BakimPersonel.Adi from OrmeArizaIhbar
          left join Personel on Personel.PersonelId = OrmeArizaIhbar.BaslatanPersonelId
          left join OrmeArizaTipi on OrmeArizaTipi.OrmeArizaTipiId = OrmeArizaIhbar.OrmeArizaTipiId
        left join OrmeArizaBakim on OrmeArizaBakim.OrmeArizaIhbarID = OrmeArizaIhbar.OrmeArizaIhbarID and OrmeArizaBakim.GittiMakinaDurumuID is Null
          left join Personel BakimPersonel on BakimPersonel.PersonelId = OrmeArizaBakim.PersonelId
          where OrmeArizaIhbar.OrmeArizaIhbarId = :ArizaIhbarId
          into :ArizaYeriID, :ArizaIhbarPersonelId, :ArizaIhbarPersonelAdi, :ParcaBekliyor, :ArizaTipiAdi, :BakimciPersonelId, :BakimciPersonelAdi;

      if (ParcaBekliyor is null) then ParcaBekliyor = 0;
    end

    if (ImalataAraVerildi = 1) then
    begin
      Select Max(ImalataAraVermeId) from ImalataAraVerme
        where ImalataAraVerme.OrmeMakinaId = :AMakinaId and ImalataAraVerme.BittiOrmeMakinaDurumuId is null into :ImalataAraVermeId;


      Select ImalataAraVerme.AraVermeSebebiId, AraVermeSebebi.Adi, ImalataAraVerme.BaslamaTarihi, ImalataAraVerme.AraVerdirenPersonelId,
        Personel.Adi from ImalataAraVerme
        left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
        left join Personel on Personel.PersonelId = ImalataAraVerme.AraVerdirenPersonelId
        where ImalataAraVerme.ImalataAraVermeId = :ImalataAraVermeId
        into :AraVermeSebebiId, :AraVermeSebebiAdi, :AraVermeTarihi, :ImalataAraVerenPersonelId, :ImalataAraVerenPersonelAdi;
    end
  end
  suspend;
end
