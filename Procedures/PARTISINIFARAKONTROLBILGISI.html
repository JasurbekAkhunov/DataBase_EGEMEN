CREATE PROCEDURE PARTISINIFARAKONTROLBILGISI(
    APROXYID VARCHAR(8))
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTITIPIID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BOBINSAYISI INTEGER,
    YAPILACAKARAKONTROLVAR INTEGER,
    ASAMAADI VARCHAR(40),
    ASAMAKISAADI VARCHAR(10),
    ARAKONTROLYAPABILIR INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTISINIFID INTEGER)
AS
declare variable PartiKayitId  integer;
declare variable FSira         integer;
declare variable KayitSayisi   integer;
begin
  AraKontrolYapabilir = 0;
  Select PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.PartiTipiId, PartiStok.HamAdiToplami,
         PartiStok.Kilo, PartiStok.Metre, PartiStok.BobinSayisi from PartiStok
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  where PartiStok.ProxyId = :AProxyId
  into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :PartiTipiId, :HamAdiToplami, :Kilo, :Metre, :BobinSayisi;

  if (PartiSinifId is not null) then
  begin
    Select Min(PartiAsamalari.Sira) from PartiAsamaStok
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamaStok.CikisTarihi is null and PartiAsamalari.OtomasyonaDahilmi = 1 and
            Asama.AraKontrolAsamasimi = 1 and PartiAsamalari.AsamaId is not null
    into :FSira;

    if (FSira is null) then
    begin
      YapilacakAraKontrolVar = 0;
      AraKontrolYapabilir    = 0;
      PartiAsamalariId       = Null;
      AsamaAdi = Null;
      AsamaKisaAdi = Null;
    end
    else
    begin
      YapilacakAraKontrolVar = 1;

      Select PartiAsamalari.PartiAsamalariId, Asama.Adi, Asama.AsamaKisaAdi from PartiAsamalari
        left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.Sira = :FSira
        into :PartiAsamalariId, :AsamaAdi, :AsamaKisaAdi;

      Select Count(*) from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamalari.Sira < :FSira and
              PartiAsamaStok.CikisTarihi is null
        into :KayitSayisi;

      if (KayitSayisi = 0) then AraKontrolYapabilir = 1;
    end
  end
  suspend;
end
