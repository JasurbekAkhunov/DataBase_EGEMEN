      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join FirmaSevkAdresi on FirmaSevkAdresi.FirmaSevkAdresiId = BoyaSiparisDetay.FirmaSevkAdresiId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
      left join Desen on Desen.DesenId = DesenVaryant.DesenId
      left join OdemeSekli on OdemeSekli.OdemeSekliId = BoyaSiparis.OdemeSekliID
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiID
      left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
      left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliId = BoyaSiparisDetay.PaketlemeSekliId
      left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId

              ((:ARaporHamTipi = 2) and (PartiStok.HamTipiId = :hstDokumaKumas)) or
              ((:ARaporHamTipi = 3) and (PartiStok.HamTipiId = :hstDokumaKumas or PartiStok.HamTipiId = :hstOrmeKumas)) or
              (:ARaporHamTipi is null) ) and
            (Parti.Fason = :ASatisTipi or :ASatisTipi is null) and
            (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
            (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
            (:ATamirDurumu = 2 or (:ATamirDurumu = 0 and Parti.IslemTipiId = :itTamir) or (:ATamirDurumu = 1 and Parti.IslemTipiId <> :itTamir)) and
            (Parti.ReceteId = :AReceteId or :AReceteId is null) and
            (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
            (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
            (Desen.DesenId = :ADesenID or :ADesenID is null) and
            (PartiStokTarti.KaliteNo = :AKaliteNo or :AKaliteNo is null) and
            (BoyaSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null)  and
            (PartiStokTarti.PartiKayitId = :APartiKayitId or :APartiKayitId is null) and
            (PartiStok.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
            (coalesce(PartiStokTarti.BarkodId,'K-' || EgemenIntegerToStr(PSKaliteKontrol.PSKaliteKontrolId)) = :ABarkodId or :ABarkodId is null) and
            (BoyaSiparisDetay.BelgeHamAdi = :ABelgeHamAdi or :ABelgeHamAdi is null)
    into :PartiKayitId, :FirmaId, :FirmaAdi, :PartiYil, :PartiId, :Fason, :PartiSinifId, :HamAdiToplami, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
         :ReceteId, :ReceteAdi, :DesenId, :DesenAdi, :VaryantAdi,
         :FirmaSevkAdresiId, :FirmaSevkAdresiAdi, :SevkSehir, :SevkUlke, :SevkOncelikNo, :SevkNotu, :SevkKendiAracimizla,
         :TerminTarihi, :SiparisTerminTarihi, :BirimFiyat, :DovizId, :VadeGun, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisRenkAdi,
         :KapAdedi, :TartiKapAdedi, :Kilo, :Metre, :BrutKilo, :BrutMetre,
         :RezervKapAdedi, :RezervTartiKapAdedi, :RezervKilo, :RezervMetre, :BrutRezervKilo, :BrutRezervMetre, :RezervTartiBrutKilo,
         :MamulDepoTransferNotu, :MusteriSiparisNo, :ReferansSiparisNo, :AciliyetDerecesiID, :AciliyetDerecesiAdi, :KaliteNo, :KKBitisTarihi, :StokNetKilo, :StokKilo,
         :AmbalajId, :KaliteId, :MinSiparisMiktari, :LabReceteKodu, :TartiBrutKilo, :YurtDisiStr, :RezervEdilenKapAdedi, :RezervEdilenKilo, :RezervEdilenBrutKilo, :ReferansKodu,
         :BobinSayisi, :Adet, :PartiNotu, :BelgeHamAdi, :MamulRezervNotu, :MamulRezervTarihi, :BaskaFirmayaSatilabilir, :PaketlemeSekliId, :TasarimNo, :VaryantAciklama, :PaketlemeSekliAdi, :RenkDerinligiAdi, :MusteriRenkKoduVeAdi, :BaskiTipiAdi, :BarkodId,
         :TopNo, :SiparisProsesAdi, :SiparisProsesAdi2, :KaliteAdi, :SiparisMetre, :SiparisiAlanKullaniciAdi, :SiparisPartilenenMetre, :SiparisPartilenenMetreBaski,
         :LRAColor, :OYDosyaAdi, :VaryantResimDosyaAdi, :EnIstenen, :GramajIstenen, :OdemeSekliAdi, :OdemeSekliVadeGun, :OdemeSekliDovizId, :PartiRenkLotNo,
     :HamAdi2, :HamAdi3 , HamAdi4, :PartiSiraNo, :PsKaliteKontrolAciklama do
    begin
      KalanTartiBrutKilo = TartiBrutKilo - RezervTartiBrutKilo;
      KalanKilo = Kilo - RezervKilo;
      HamLot = '';
      for select HamStokMaster.Lot from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
          where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.Lot is not null and HamStokMaster.Lot <> ''
          group by HamStokMaster.Lot
          into :FLot do
      begin
        if (char_length(HamLot) + char_length(FLot) < 255) then
        begin
         if (HamLot <> '' ) then HamLot = HamLot || cast(', ' as varchar(2) character set UTF8);
         HamLot = HamLot || FLot;
        end
      end

      select PartiStokFizikKontrol.GramajVerilen, PartiStokFizikKontrol.EnVerilen from PartiStokFizikKontrol
      where PartiStokFizikKontrol.PartiSinifId = :PartiSinifId and PartiStokFizikKontrol.SonKayit = 1 into :GramajVerilen, :EnVerilen;

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      else BoyaSiparisIdStr = '';

      if (SevkKendiAracimizla = 1) then SevkKendiAracimizlaStr = 'Bizim Aracmz';
      else if (SevkKendiAracimizla = 2) then SevkKendiAracimizlaStr = 'Mteri''nin Arac';
      else SevkKendiAracimizlaStr = '';

      if (SevkNotu <> '') then SevkNotuvar = 1;
      else SevkNotuvar = 0;

      if (Fason = 0 and ASatisBoyaSiparisFiyatGorebilir = 0) then
      begin
        BirimFiyat = null;
        DovizId    = null;
        VadeGun    = null;
      end
      else if (Fason = 1 and AFasonBoyaSiparisFiyatGorebilir = 0) then
      begin
        BirimFiyat = null;
        DovizId    = null;
        VadeGun    = null;
      end

      RafStr = '';
      if (ARafID is null)  then Okey = 1;
        else Okey = 0;
      for select MamulDepoTransferFisiDetay.RafId from PartiStokTarti
        left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiID and MamulDepoTransferFisiDetay.PartiStokTartiID = PartiStokTarti.PartiStoktartiID
        left join Depo on Depo.DepoId = PartiStokTarti.DepoId
        left join PSKaliteKontrol on PSKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
        where PartiStokTarti.PartiSinifID = :PartiSinifID and
              PartiStokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null and
              (:ADepoId is not null and PartiStokTarti.DepoId = :ADepoId or :ADepoId is null and Depo.Sevk = 1) and
              (:TopNo is null or PSKaliteKontrol.TopNo = :TopNo)
        group by MamulDepoTransferFisiDetay.RafId
        into :FRafId do
      begin
        if (char_length(RafStr) + char_length(FRafId) < 48) then
        begin
         if (RafStr <> '' ) then RafStr = RafStr || cast(', ' as varchar(2) character set UTF8);
         RafStr = RafStr || FRafId;
        end
        if (ARafId is not null) then
          if (ARafId = FRafId) then Okey = 1;
      end

      select PartiStokFizikKontrol.EnVerilen,PartiStokFizikKontrol.GramajVerilen
          from PartiStokFizikKontrol
      where PartiSinifId = :PartiSinifID and PartiStokFizikKontrol.SonKayit=1
      into :EnVerilen, :GramajVerilen;

      select first 1 HamStokMaliyeti from PartiSinifMaliyetKumulatif where PartiSinifId = :PartiSinifId into :HamStokMaliyeti;

      select AlisFiyati from PartiSonAlisFiyatiUSD(:PartiSinifId) into :HamStokSonAlisFiyatiUSD;

      IadeGelenFirmalar = '';
      IadeKilo = 0;
      IadeBrutKilo = 0;
      IadeTartiBrutKilo = 0;

      select   sum(PartiStok.BrutSevkMetre), sum(PartiStok.NetSevkMetre), sum(PartiStok.NetSevkKapAdedi),
               sum(PartiStok.BrutSarfMetre),  sum(PartiStok.NetSarfMetre)  from PartiStok
      where PartiStok.BoyaSiparisDetayID = :BoyaSiparisDetayID
      into :SiparisBrutSevkEdilenMetre,  :SiparisNetSevkEdilenMetre, :SiparisNetSevkKapAdedi,
           :SiparisBrutSarfEdilenMetre,  :SiparisNetSarfEdilenMetre ;

      for select Firma.KisaAdi, PartiStokTarti.Kilo, PartiStokTarti.BrutKilo, PartiStokTarti.Dara from PartiStok
        left join PartiStokTarti on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
        left join Depo on Depo.DepoId = PartiStokTarti.DepoId
        left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = PartiStokTarti.IadeBoyaIrsaliyeId
        left join Firma on Firma.FirmaID = BoyaIrsaliye.FirmaId
        where PartiStok.PartiSinifId = :PartiSinifId and PartiStokTarti.IadeBoyaIrsaliyeId is not null and
              PartiStokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null and
              (:ADepoId is not null and Depo.DepoId = :ADepoId or :ADepoId is null and Depo.Sevk = 1) and
              (PartiStokTarti.KaliteNo = :AKaliteNo or :AKaliteNo is null)
        into :FirmaKisaAdi, :FKilo, :FBrutKilo, :FDara do
      begin
        if (FirmaKisaAdi is not null and FirmaKisaAdi <> '' and position(FirmaKisaAdi, IadeGelenFirmalar) = 0) then
          IadeGelenFirmalar = IadeGelenFirmalar || iif(IadeGelenFirmalar = '', '', '-') || :FirmaKisaAdi;
        IadeKilo = IadeKilo + FKilo;
        IadeBrutKilo = IadeBrutKilo + FBrutKilo;
        IadeTartiBrutKilo = IadeTartiBrutKilo + FKilo + FDara;
      end
     select GirisIrsaliyeTarihi, GirisFirmaAdi from PartiStokGirisBilgisiGetir(:PartiSinifID) into :GirisIrsaliyeTarihi, :GirisFirmaAdi;

     select first 1  PSDokumaIsEmriBilgisiGetir.CozguIndigoIsEmriIDStr, PSDokumaIsEmriBilgisiGetir.IndigoReceteId from PSDokumaIsEmriBilgisiGetir(:PartiSinifID, null) into :CozguIndigoIsEmriIDStr, :IndigoReceteId;


     HamStokBirimFiyat =null;
     HamStokBirimFiyatDovizId =null;

      select first 1 HamStokIrsaliye.DovizId, HamStokMaster.NetBirimFiyat from PartiStokDetay
         left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
         left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
         left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
         left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
         where PartiStokDetay.PartiSinifID = :PartiSinifId and HamStokMaster.NetBirimFiyat > 0
         into :HamStokBirimFiyatDovizId, :HamStokBirimFiyat;

     if (okey = 1) then suspend;
    end
  end
