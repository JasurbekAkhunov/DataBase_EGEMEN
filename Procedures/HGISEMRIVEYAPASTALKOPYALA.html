CREATE PROCEDURE HGISEMRIVEYAPASTALKOPYALA(
    AHGISEMRIKAYITID INTEGER,
    AHGISEMRIPASTALID INTEGER,
    AKULLANICIID INTEGER NOT NULL)
AS
Declare Variable FHGIsEmriKayitId      Integer;
Declare Variable FHGIsEmriPastalId     Integer;
Declare Variable FYeniHGIsEmriPastalId Integer;
Begin
  if (AHGIsEmriKayitId is Not Null and AHGIsEmriPastalId is Not Null) then
    Exception KayitHatasi 'AHGIsEmriKayitId ve AHGIsEmriPastalId parametrelerinin ikisi de dolu olamaz...';

  if (AHGIsEmriKayitId is Null and AHGIsEmriPastalId is Null) then
    Exception KayitHatasi 'AHGIsEmriKayitId ve AHGIsEmriPastalId parametrelerinin ikisi de bo≈ü olamaz...';

  FHGIsEmriKayitId = Null;
  FHGIsEmriPastalId = Null;

  if (AHGIsEmriKayitId is not null) then
  begin
    Insert Into HGIsEmri(HGSiparisKayitId, HGSiparisDetayId, Tarih, FirmaId, OrmeMakinaId, ParcaIscilik, KesimTalimati, TerminTarihi,
                            BirimFiyat, DovizId, VadeGun, Notu, Bitti, MiktarGun, Baslamatarihi, Bitistarihi, InsertKullaniciId, KullaniciId)
    Select HGSiparisKayitId, HGSiparisDetayId, Tarih, FirmaId, OrmeMakinaId, ParcaIscilik, KesimTalimati, TerminTarihi,
                            BirimFiyat, DovizId, VadeGun, Notu, Bitti, MiktarGun, Baslamatarihi, Bitistarihi, :AKullaniciId, :AKullaniciId
    From HGIsEmri Where HGIsEmriKayitId = :AHGIsEmriKayitId Returning HGIsEmriKayitId Into :FHGIsEmriKayitId;
  end

  For Select Distinct HGIsEmriPastalId From HGIsEmriPastal
    Where (HGIsEmriPastal.HGIsEmriKayitId = :AHGIsEmriKayitId or :AHGIsEmriKayitId = Null) Or (HGIsEmriPastal.HGIsEmriPastalId = :AHGIsEmriPastalId or :AHGIsEmriPastalId = Null)
  Into :FHGIsEmriPastalId Do
  Begin
    FYeniHGIsEmriPastalId = Null;

    Insert Into HGIsEmriPastal(HGIsEmriKayitId, En, Boy, AnaKumas, Notu, InsertKullaniciId, KullaniciId)
    Select Coalesce(:FHGIsEmriKayitId, HGIsEmriKayitId), En, Boy, AnaKumas, Notu, :AKullaniciId, :AKullaniciId From HGIsEmriPastal
    Where HGIsEmriPastalId = :FHGIsEmriPastalId Returning HGIsEmriPastalId Into :FYeniHGIsEmriPastalId;

    Insert Into HGIsEmriPastalBeden(HGIsEmriPastalId, BedenId, Miktar, InsertKullaniciId, KullaniciId)
    Select :FYeniHGIsEmriPastalId, BedenId, Miktar, :AKullaniciId, :AKullaniciId From HGIsEmriPastalBeden
     Where HGIsEmriPastalId = :FHGIsEmriPastalId;

    Insert Into HGIsEmriPastalKumas(HGIsEmriPastalId, HGSiparisDetayKumasId, KatSayisi, InsertKullaniciId, KullaniciId)
    Select :FYeniHGIsEmriPastalId, HGSiparisDetayKumasId, KatSayisi, :AKullaniciId, :AKullaniciId From HGIsEmriPastalKumas
    Where HGIsEmriPastalId = :FHGIsEmriPastalId;
  end
end
