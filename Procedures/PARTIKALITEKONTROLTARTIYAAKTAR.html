CREATE PROCEDURE PARTIKALITEKONTROLTARTIYAAKTAR(
    APSKALITEKONTROLID INTEGER,
    APSKALITEKONTROLASAMAID INTEGER,
    APARTISINIFID INTEGER,
    APARTIKAYITID INTEGER,
    AKULLANICIID INTEGER,
    ADEPOID DEPOKODUSIZE)
AS
declare variable TamamiTartildi integer;
declare variable KS integer;
declare variable NetKilo        Numeric(12,3);
declare variable Dara           Numeric(12,3);
declare variable NetMetre       Numeric(12,3);
declare variable KapAdedi       integer;
declare variable PartiStokTartiId integer;
declare variable PsKaliteKontrolId integer;
declare variable KaliteNo          integer;
declare variable OlcuId            integer;
begin
  Select TamamiTartildi from PartiStok where PartiStok.PartiSinifId = :APartiSinifId into :TamamiTartildi;
  if (TamamiTartildi = 1) then exception TamamiTartilmisKiloGiremezsiniz;

  Select Count(*) from PSKaliteKontrol where PSKaliteKontrol.PSKaliteKontrolID = :APSKaliteKontrolID and PartiStokTartiID is not Null into :KS;
  if (KS > 0) then exception BuTopTartiyaGonderilmis;

  Select PsKaliteKontrolId, KaliteNo, Dara, (Kilo - Dara), Metre, OlcuId from PSKaliteKontrol
    where PSKaliteKontrol.PSKaliteKontrolID = :APSKaliteKontrolID and PSKaliteKontrol.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID and
          PSKaliteKontrol.PartiSinifID = :APartiSinifID and PSKaliteKontrol.OnayDurumu = 1
    into :PsKaliteKontrolId, :KaliteNo, :Dara, :NetKilo, :NetMetre, :OlcuId;

  if (PsKaliteKontrolId is Null) then exception KayitHatasi ' Okutulan Top İlgili Aşamaya Ait Değil';
  else
  begin
    PartiStokTartiId = null;

    KapAdedi = 1;
    PartiStokTartiId = gen_id(PartiStokTarti_Id_Gen, 1);

    Insert Into PartiStokTarti(PartiStokTartiId, PartiKayitId, PartiSinifId, Kilo, Dara, Metre, KapAdedi, KaliteNo, KaliteKontrol, InsertKullaniciId, OlcuId, DepoId)
                          values(:PartiStokTartiId, :APartiKayitId, :APartiSinifId, :NetKilo, :Dara, :NetMetre, :KapAdedi, :KaliteNo, 1, :AKullaniciId, :OlcuId, :ADepoID);

    Update PsKaliteKontrol set PartiStokTartiId = :PartiStokTartiId, LinkIslemi = 1, KullaniciId = :AKullaniciId where PsKaliteKontrol.PsKaliteKontrolId = :PsKaliteKontrolId;
  end

  Select Count(*) from PSKaliteKontrol where PSKaliteKontrol.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID and PartiStokTartiID is Null into :KS;

  /*Tamamı Tartıya gönderilmiş ise Gerekli flaglar yazılsın*/
  if (KS = 0) then
  begin
    Update PSKaliteKontrolAsama set TartiyaAktarildi = 1 where PSKaliteKontrolAsama.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID;
    Update PartiStok Set TamamiTartildi = 1 where PartiStok.PartiSinifId = :APartiSinifId;

    Execute Procedure PartiStokTartiBrutHesapla(:APartiSinifId, 0);
  end
end
