CREATE PROCEDURE YUZEYKONTROLTALEPLISTESI
RETURNS (
    YUZEYKONTROLTALEPKAYITID INTEGER,
    YUZEYKONTROLTALEPYIL INTEGER,
    YUZEYKONTROLTALEPID INTEGER,
    YUZEYKONTROLTALEPIDSTR VARCHAR(30),
    TALEPTARIHI TIMESTAMP,
    HAMSTOKMASTERID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    KONTROLORANI INTEGER,
    TALEPMIKTARI NUMERIC(12,3),
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select YuzeyKontrolTalep.YuzeyKontrolTalepKayitId, YuzeyKontrolTalep.YuzeyKontrolTalepYil, YuzeyKontrolTalep.YuzeyKontrolTalepId,
             YuzeyKontrolTalep.TalepTarihi, HamStokMaster.HamStokMasterId, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             HamStokMaster.HamId, Ham.Adi, YuzeyKontrolTalep.KontrolOrani, YuzeyKontrolTalep.TalepMiktari, IK.Adi,
             YuzeyKontrolTalep.DegisimTarihi from YuzeyKontrolTalep
    left join HamStokMaster on HamStokMaster.HamStokMasterId = YuzeyKontrolTalep.HamStokMasterID
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeid = HamStokMaster.HamStokIrsaliyeid
    left join ham on ham.hamid = HamStokMaster.hamId
    left join Kullanici IK on Ik.KullaniciId = YuzeyKontrolTalep.KullaniciId
    where YuzeyKontrolTalep.KontrolKullaniciId is null
    into :YuzeyKontrolTalepKayitId, :YuzeyKontrolTalepYil, :YuzeyKontrolTalepId, :TalepTarihi, :HamStokMasterId, :IrsaliyeNo,
         :IrsaliyeTarihi, :EvrakTipi, :HamId, :HamAdi, :KontrolOrani, :TalepMiktari, :KullaniciAdi, :DegisimTarihi do
  begin
    YuzeyKontrolTalepIdStr = '(' || EgemenIntegerToStr(YuzeyKontrolTalepYil) || ') ' || EgemenIntegerToStr(YuzeyKontrolTalepId) || ' ';
    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';

    suspend;
  end
end
