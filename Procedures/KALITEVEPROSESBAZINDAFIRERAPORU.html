CREATE PROCEDURE KALITEVEPROSESBAZINDAFIRERAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMID INTEGER,
    ARAPORHAMTIPI INTEGER,
    AKUMASCINSIID INTEGER,
    AIPLIKCINSIID INTEGER,
    ASATISTIPI INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    TOPLAMBRUTKILO DOUBLE PRECISION,
    TOPLAMNETKILO DOUBLE PRECISION,
    ORTALAMAFIREORANI DOUBLE PRECISION)
AS
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;

  for select BoyaIrsaliyeDetay.HamId, Ham.Adi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaIrsaliye.FirmaId, Firma.Adi, Parti.Fason,
             Sum(BoyaIrsaliyeDetay.BrutKilo), Sum(BoyaIrsaliyeDetay.Kilo) from BoyaIrsaliye
    left join Firma on Firma.FirmaID = BoyaIrsaliye.FirmaID
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
    left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
          BoyaIrsaliyeDetay.PartiSinifID is not null and
          (BoyaIrsaliyeDetay.HamId = :AHamId or :AHamId is null) and
          (Parti.Fason = :ASatisTipi or :ASatisTipi is null) and
          (Ham.KumasCinsiID = :AKumasCinsiId or :AKumasCinsiId is null ) and
          (Ham.IplikCinsiID = :AIplikCinsiId or :AIplikCinsiId is null ) and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporHamTipi = 0) or
            (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporHamTipi = 1) or
            ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :ARaporHamTipi = 2) or
            (Ham.HamTipiId = :hstDokumaKumas and :ARaporHamTipi = 3) or :ARaporHamTipi is null)
    group by BoyaIrsaliyeDetay.HamId, Ham.Adi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaIrsaliye.FirmaId, Firma.Adi, Parti.Fason
    into :HamId, :HamAdi, :SiparisProsesId, :SiparisProsesAdi, :FirmaId, :FirmaAdi, :Fason, :ToplamBrutKilo, :ToplamNetKilo do
  begin
    if (ToplamBrutKilo > 0) then
      OrtalamaFireOrani = ( (ToplamBrutKilo - ToplamNetKilo) / ToplamBrutKilo) * 100;
    else
      OrtalamaFireOrani = 0;

    OrtalamaFireOrani = Yuvarla(OrtalamaFireOrani, 100);

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    suspend;
  end
end
