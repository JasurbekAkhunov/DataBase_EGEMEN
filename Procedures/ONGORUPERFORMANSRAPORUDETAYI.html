CREATE PROCEDURE ONGORUPERFORMANSRAPORUDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    ENOKEY INTEGER,
    GRAMAJOKEY INTEGER,
    ENDENCEKMEZLIKOKEY INTEGER,
    BOYDANCEKMEZLIKOKEY INTEGER,
    DONMEOKEY INTEGER,
    ONGORULENEN INTEGER,
    ONGORULENGRAMAJ INTEGER,
    ONGORULENENDENCEKMEZLIK DOUBLE PRECISION,
    ONGORULENBOYDANCEKMEZLIK DOUBLE PRECISION,
    ONGORULENDONME VARCHAR(10),
    ENVERILEN VARCHAR(10),
    GRAMAJVERILEN VARCHAR(10),
    DONMEVERILEN VARCHAR(10),
    CEKMEZLIKENVERILEN DOUBLE PRECISION,
    CEKMEZLIKBOYVERILEN DOUBLE PRECISION)
AS
declare variable FInt integer;
declare variable EInt integer;
begin
  for Select Parti.PartiYil, Parti.PartiId, PartiStokFizikKontrol.PartiSinifId, PartiStok.HamAdiToplami, PartiStokFizikKontrol.EnVerilen, PartiStokFizikKontrol.GramajVerilen, PartiStokFizikKontrol.DonmeVerilen, PartiStokFizikKontrol.CekmezlikEnVerilen,
             PartiStokFizikKontrol.CekmezlikBoyVerilen, Max(BoyaSiparisDetayOngoru.OngorulenEn), Max(BoyaSiparisDetayOngoru.OngorulenGramaj), max(BoyaSiparisDetayOngoru.OngorulenDonme),
             Max(BoyaSiparisDetayOngoru.OngorulenEndenCekmezlik), Max(BoyaSiparisDetayOngoru.OngorulenBoydanCekmezlik) from PartiStokFizikKontrol
    left join PartiStok on PartiStok.PartiSinifId = PartiStokFizikKontrol.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStokFizikKontrol.PartiSinifId
    left join BoyaSiparisRezerv on BoyaSiparisRezerv.BoyaSiparisRezervId = PartiStokDetay.BoyaSiparisRezervId
    left join BoyaSiparisDetayOngoru on BoyaSiparisDetayOngoru.BoyaSiparisDetayId = BoyaSiparisRezerv.BoyaSiparisDetayId and BoyaSiparisDetayOngoru.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
    where PartiStokFizikKontrol.FKontrol1KontrolTarihi >= :ABaslangicTarihi and PartiStokFizikKontrol.FKontrol1KontrolTarihi <= :ABitisTarihi and
          PartiStokFizikKontrol.SonKayit = 1
    group by Parti.PartiYil, Parti.PartiId, PartiStokFizikKontrol.PartiSinifId, PartiStok.HamAdiToplami, PartiStokFizikKontrol.EnVerilen, PartiStokFizikKontrol.GramajVerilen, PartiStokFizikKontrol.DonmeVerilen, PartiStokFizikKontrol.CekmezlikEnVerilen, PartiStokFizikKontrol.CekmezlikBoyVerilen
    into :PartiYil, :PartiId, :PartiSinifId, :HamAdiToplami, :EnVerilen, :GramajVerilen, :DonmeVerilen, :CekmezlikEnVerilen, :CekmezlikBoyVerilen,
         :OngorulenEn, :OngorulenGramaj, :OngorulenDonme, :OngorulenEndenCekmezlik, :OngorulenBoydanCekmezlik do
  begin
    EnOkey                  = 0;
    GramajOkey              = 0;
    EndenCekmezlikOkey      = 0;
    BoydanCekmezlikOkey     = 0;
    DonmeOkey               = 0;

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (EgemenStrToIntOlurmu(EnVerilen) = 1) then
    begin
      FInt = EgemenStrToInt(EnVerilen);

      if (OngorulenEn > 0 and FInt <> 0) then
      begin
        if ( (FInt >= (OngorulenEn - OngorulenEn * 5 /100)) and (FInt <= (OngorulenEn + OngorulenEn * 5 /100)) ) then EnOkey = 1;
        else EnOkey = -1;
      end
    end

    if (EgemenStrToIntOlurmu(GramajVerilen) = 1) then
    begin
      FInt = EgemenStrToInt(GramajVerilen);

      if (OngorulenGramaj <> 0 and FInt <> 0) then
      begin
        if ( (FInt >= (OngorulenGramaj - OngorulenGramaj * 5 /100)) and (FInt <= (OngorulenGramaj + OngorulenGramaj * 5 /100)) ) then GramajOkey = 1;
        else GramajOkey = -1;
      end
    end

    if (EgemenStrToIntOlurmu(DonmeVerilen) = 1 and EgemenStrToIntOlurmu(OngorulenDonme) = 1) then
    begin
      FInt = EgemenStrToInt(DonmeVerilen);
      EInt = EgemenStrToInt(OngorulenDonme);

      if (EInt <> 0 and FInt <> 0) then
      begin
        if (FInt <= EInt) then DonmeOkey = 1;
        else DonmeOkey = -1;
      end
    end

    if (CekmezlikEnVerilen is not null and CekmezlikEnVerilen <> 0) then
    if (OngorulenEndenCekmezlik is not null and OngorulenEndenCekmezlik <> 0) then
    begin
      if (CekmezlikEnVerilen >= OngorulenEndenCekmezlik * -1 and CekmezlikEnVerilen <= OngorulenEndenCekmezlik) then EndenCekmezlikOkey = 1;
      else EndenCekmezlikOkey = -1;
    end

    if (CekmezlikBoyVerilen is not null and CekmezlikBoyVerilen <> 0) then
    if (OngorulenBoydanCekmezlik is not null and OngorulenBoydanCekmezlik <> 0) then
    begin
      if (CekmezlikBoyVerilen >= OngorulenBoydanCekmezlik * -1 and CekmezlikBoyVerilen <= OngorulenBoydanCekmezlik) then BoydanCekmezlikOkey = 1;
      else BoydanCekmezlikOkey = -1;
    end

    if (EnOkey = -1 or GramajOkey = -1 or EndenCekmezlikOkey = -1 or BoydanCekmezlikOkey = -1 or DonmeOkey = -1) then suspend;
  end
end
