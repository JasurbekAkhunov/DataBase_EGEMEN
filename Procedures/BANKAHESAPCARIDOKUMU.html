CREATE PROCEDURE BANKAHESAPCARIDOKUMU(
    ABANKAHESAPID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1),
    AGRUP1ID BANKAHESAPGRUPKODUSIZE,
    AGRUP2ID BANKAHESAPGRUPKODUSIZE,
    ABANKACARIHAREKETTIPIID BANKACARIHAREKETTIPISIZE,
    ADOVIZID DOVIZSIZE,
    ARAPORDOVIZID DOVIZSIZE)
RETURNS (
    BANKAHESAPID VARCHAR(20),
    BANKAHESAPCARIID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    BORCALACAK VARCHAR(1),
    SINIF VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    DOVIZKURUSTR VARCHAR(40),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    CARIHAREKETTIPIID VARCHAR(10),
    BANKACARIHAREKETTIPIADI VARCHAR(30),
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSYERI INTEGER,
    REFERANSID INTEGER,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    BORCBAKIYE DOUBLE PRECISION,
    ALACAKBAKIYE DOUBLE PRECISION,
    BANKAHESAPGRUP1ADI ADI30,
    BANKAHESAPGRUP2ADI ADI30,
    RAPORDOVIZID DOVIZSIZE,
    RAPORDOVIZMEBLAG DOUBLE PRECISION,
    RAPORDOVIZBORC DOUBLE PRECISION,
    RAPORDOVIZALACAK DOUBLE PRECISION,
    RAPORDOVIZBORCBAKIYE DOUBLE PRECISION,
    RAPORDOVIZALACAKBAKIYE DOUBLE PRECISION)
AS
begin
  For Select BankaHesapId from BankaHesap
      where (BankaHesap.BankaHesapID = :ABankaHesapID or :ABankaHesapID is null)
      and (BankaHesap.BankaHesapGrup1 = :AGrup1Id or :AGrup1Id is null)
      and (BankaHesap.BankaHesapGrup2 = :AGrup2Id or :AGrup2Id is null) into :BankaHesapId do
  begin
    for Select BankaHesapCariId, FirmaId, FirmaAdi, FirmaAdi2, Tarih, Aciklama, BorcAlacak, Sinif, Meblag, CariHareketTipiId, BankaCariHareketTipiAdi, DovizKuruStr, Notu,
               KullaniciAdi, DegisimTarihi, ReferansYeri, ReferansId, Borc, Alacak, BorcBakiye, AlacakBakiye, BankaHesapGrup1Adi, BankaHesapGrup2Adi, RaporDovizMeblag, RaporDovizBorc, RaporDovizAlacak, RaporDovizBorcBakiye, RaporDovizAlacakBakiye, :ARaporDovizId from BankaHesapCariHareketiGetir(:BankaHesapId, :ABaslangicTarihi, :ABitisTarihi, :ASinif, :AGrup1Id, :AGrup2Id, :ABankaCariHareketTipiId, :ADovizId, :ARaporDovizId)
      into :BankaHesapCariId, :FirmaId, :FirmaAdi, :FirmaAdi2, :Tarih, :Aciklama, :BorcAlacak, :Sinif, :Meblag, :CariHareketTipiId, :BankaCariHareketTipiAdi, :DovizKuruStr, :Notu,
            :KullaniciAdi, :DegisimTarihi, :ReferansYeri, :ReferansId, :Borc, :Alacak, :BorcBakiye, :AlacakBakiye, :BankaHesapGrup1Adi, :BankaHesapGrup1Adi, :RaporDovizMeblag, :RaporDovizBorc, :RaporDovizAlacak, :RaporDovizBorcBakiye, :RaporDovizAlacakBakiye, :RaporDovizId do
    begin
      suspend;
    end
  end
end
