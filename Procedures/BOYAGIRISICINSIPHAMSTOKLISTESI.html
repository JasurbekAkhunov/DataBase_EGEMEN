CREATE PROCEDURE BOYAGIRISICINSIPHAMSTOKLISTESI(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    RESULTID VARCHAR(255),
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    BOYASIPARISDETAYRENKADI VARCHAR(40),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    SIPARISKILO NUMERIC(12,3),
    TOPLAMBOYAYACIKIS NUMERIC(12,3),
    MUSTERISIPARISNO VARCHAR(50),
    MELANJKODU VARCHAR(20),
    ARTIKELNO VARCHAR(20),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10))
AS
declare variable FHamId integer;
declare variable FRenkId integer;
declare variable FHamStokBirimId integer;
begin
  for Select OrmeSiparisDetay.OrmeSiparisDetayID, BoyaSiparisDetay.BoyaSiparisDetayId, OrmeSiparisDetay.HamId, BoyaSiparisDetay.HamId, Ham.Adi, BoyaSiparisDetay.RenkId, OrmeSiparisDetay.RenkId, Renk.Adi,
             BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.MusteriRenkKoduveAdi,
             BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.ToplamBoyayaCikis, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo,
             BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi, BoyaSiparisDetay.MelanjKodu, OrmeSiparisDetay.HamStokBirimId, BoyaSiparisDetay.EnIstenen,
             BoyaSiparisDetay.GramajIstenen from OrmeSiparisDetay
  left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join Renk on Renk.RenkId = BoyaSiparisDetay.RenkId
  left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
  where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisDetay.BoyaSiparisDetayId is not null
  order by OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.BoyaSiparisDetayId
  into :OrmeSiparisDetayID, :BoyaSiparisDetayId, :FHamId, :HamId, :HamAdi, :RenkId, :FRenkId, :RenkAdi, :BoyaSiparisDetayRenkAdi, :LabReceteKodu, :LabReceteAdi,
       :MusteriRenkKoduveAdi, :SiparisKilo, :ToplamBoyayaCikis, :ArtikelNo, :MusteriSiparisNo, :IslemTipiId, :IslemTipiAdi, :MelanjKodu, :FHamStokBirimId, :EnIstenen,
       :GramajIstenen do
  begin
    if (HamID = FHamId and FRenkID is not null) then RenkID = FRenkId;

    ResultId = '"' || EgemenIntegerToStr(OrmeSiparisDetayID) || '",' ;
    ResultId = ResultId ||'"' || EgemenIntegerToStr(BoyaSiparisDetayId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",' ;

    if (RenkId is not null) then
      ResultId = ResultId || '"' || EgemenIntegerToStr(RenkId) || '",' ;
    else
      ResultId = ResultId || '"",' ;

    if (BoyaSiparisDetayRenkAdi is null) then BoyaSiparisDetayRenkAdi = '';
    ResultId = ResultId || '"' || BoyaSiparisDetayRenkAdi || '", ' ;

    ResultId = ResultId || '"' || EgemenIntegerToStr(ISLEMTIPIID) || '",' ;

    if (MelanjKodu is not null) then
      ResultId = ResultId || '"' || MelanjKodu || '",' ;
    else
      ResultId = ResultId || '"",' ;

    ResultId = ResultId || '"' || EgemenIntegerToStr(FHamStokBirimID) || '"' ;

    suspend;
  end
end
