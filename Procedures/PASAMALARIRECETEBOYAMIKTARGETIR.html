CREATE PROCEDURE PASAMALARIRECETEBOYAMIKTARGETIR(
    APARTIKAYITID INTEGER,
    RECETEISEMRIPARTILERINIGETIR SMALLINT)
RETURNS (
    PARTIASAMALARIID INTEGER,
    PARTIKAYITID INTEGER,
    RECETEASAMAID INTEGER,
    RECETEASAMAADI VARCHAR(40),
    BOYATIPIID INTEGER,
    BOYATIPIADI VARCHAR(20),
    MIKTAR DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    DOZAJLAMAISEMRIID INTEGER)
AS
Declare Variable FReceteIsEmriYil  Integer;
Declare Variable FReceteIsEmriId   Integer;
begin
  if (ReceteIsEmriPartileriniGetir > 0) then
    Select Count(*) from ReceteIsEmriDetay where ReceteIsEmriDetay.PartiKayitID = :APartiKayitId into :ReceteIsEmriPartileriniGetir;
  if (ReceteIsEmriPartileriniGetir > 0) then
  begin
    for Select ReceteIsEmriYil, ReceteIsEmriId from ReceteIsEmriDetay where ReceteIsEmriDetay.PartiKayitID = :APartiKayitId into :FReceteIsEmriYil, :FReceteIsEmriId do
    begin
        For Select PartiAsamalariReceteBoyaMiktar.PartiAsamalariId, PartiAsamalariReceteBoyaMiktar.PartiKayitID, PartiAsamalariReceteBoyaMiktar.ReceteAsamaID,
                   ReceteAsama.Adi, ReceteAsama.BoyaTipiId, BoyaTipi.Adi, PartiAsamalariReceteBoyaMiktar.Miktar, Sk.Adi,
                   PartiAsamalariReceteBoyaMiktar.DegisimTarihi, IK.Adi, PartiAsamalariReceteBoyaMiktar.InsertTarihi, DozajlamaIsEmri.DozajlamaIsEmriId from ReceteIsEmriDetay RIED
        Left Join PartiAsamalariReceteBoyaMiktar on PartiAsamalariReceteBoyaMiktar.PartiAsamalariId = RIED.PartiAsamalariId
        Left Join ReceteAsama on ReceteAsama.ReceteAsamaID = PartiAsamalariReceteBoyaMiktar.ReceteAsamaID
        Left Join BoyaTipi on BoyaTipi.BoyaTipiID = ReceteAsama.BoyaTipiID
        Left Join Kullanici SK on SK.KullaniciId = PartiAsamalariReceteBoyaMiktar.KullaniciId
        Left Join Kullanici IK on IK.KullaniciId = PartiAsamalariReceteBoyaMiktar.InsertKullaniciId
    Left Join DozajlamaIsEmri on DozajlamaIsEmri.ReceteIsEmriYil = RIED.ReceteIsEmriYil and DozajlamaIsEmri.ReceteIsEmriId = RIED.ReceteIsEmriId
        where RIED.ReceteIsEmriYil = :FReceteIsEmriYil  and RIED.ReceteIsEmriId = :FReceteIsEmriId and PartiAsamalariReceteBoyaMiktar.PartiKayitID is not null
        order by ReceteAsama.BoyaTipiId desc
        into :PartiAsamalariId, :PartiKayitID, :ReceteAsamaID, :ReceteAsamaAdi, :BoyaTipiId, :BoyaTipiAdi, :Miktar, :KullaniciAdi,
             :DegisimTarihi, :InsertKullaniciAdi, :InsertTarihi, :DozajlamaIsEmriId do suspend;
    end
  end
  else
  begin
    For Select PartiAsamalariReceteBoyaMiktar.PartiAsamalariId, PartiAsamalariReceteBoyaMiktar.PartiKayitID, PartiAsamalariReceteBoyaMiktar.ReceteAsamaID,
               ReceteAsama.Adi, ReceteAsama.BoyaTipiId, BoyaTipi.Adi, PartiAsamalariReceteBoyaMiktar.Miktar, Sk.Adi,
               PartiAsamalariReceteBoyaMiktar.DegisimTarihi, IK.Adi, PartiAsamalariReceteBoyaMiktar.InsertTarihi, DozajlamaIsEmri.DozajlamaIsEmriId from PartiAsamalariReceteBoyaMiktar
    Left Join ReceteAsama on ReceteAsama.ReceteAsamaID = PartiAsamalariReceteBoyaMiktar.ReceteAsamaID
    Left Join BoyaTipi on BoyaTipi.BoyaTipiID = ReceteAsama.BoyaTipiID
    Left Join Kullanici SK on SK.KullaniciId = PartiAsamalariReceteBoyaMiktar.KullaniciId
    Left Join Kullanici IK on IK.KullaniciId = PartiAsamalariReceteBoyaMiktar.InsertKullaniciId
  Left Join DozajlamaIsEmri on DozajlamaIsEmri.PartiAsamalariId = PartiAsamalariReceteBoyaMiktar.PartiAsamalariId
    where PartiAsamalariReceteBoyaMiktar.PartiKayitId = :APartiKayitId
    order by ReceteAsama.BoyaTipiId desc
    into :PartiAsamalariId, :PartiKayitID, :ReceteAsamaID, :ReceteAsamaAdi, :BoyaTipiId, :BoyaTipiAdi, :Miktar, :KullaniciAdi,
         :DegisimTarihi, :InsertKullaniciAdi, :InsertTarihi, :DozajlamaIsEmriId do suspend;
  end
end
