CREATE PROCEDURE TERMINTAKIPRAPORUORME(
    ABASLANGICTERMINTARIHI TIMESTAMP,
    ABITISTERMINTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASON INTEGER,
    ASIPARISONAYDURUMU INTEGER,
    AHAMID INTEGER,
    APARTITIPIID INTEGER)
RETURNS (
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    HAMADI VARCHAR(255),
    REFERANSKODU VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    MUSTERISIPARISNO VARCHAR(50),
    SIPARISNOTU VARCHAR(1024),
    SIPARISIALANKULLANICIID INTEGER,
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    SIPARISTARIHI TIMESTAMP,
    SIPARISINSERTTARIHI TIMESTAMP,
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    SATINALINACAKMIKTAR NUMERIC(12,3),
    STOKTANKULLANILACAKMIKTAR NUMERIC(12,3),
    PLANLANANMIKTAR NUMERIC(12,3),
    TOPLAMORULEN NUMERIC(12,3),
    TOPLAMBOYAYACIKIS NUMERIC(12,3),
    TOPLAMSATIS NUMERIC(12,3),
    BOYASIPARISMIKTAR NUMERIC(12,3),
    BOYAREZERVMIKTAR NUMERIC(12,3),
    BASKASIPARISEKAYDIRILAN NUMERIC(12,3),
    BASKASIPARISTENKAYDIRILAN NUMERIC(12,3))
AS
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;

declare variable FSiparisNotu Blob;
declare variable BlobSize    integer;
declare variable Okey integer;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;

  For select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.OrmeSiparisDetayId,
               BoyaSiparis.FirmaId, Firma.Adi, OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.Metre,
               Ham.Adi, Ham.ReferansKodu, BoyaSiparis.Fason, OrmeSiparisDetay.MusteriSiparisNo,
               OrmeSiparisDetay.Notu, OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.RevizeTeslimTarihi,
               BoyaSiparis.SiparisiAlanKullaniciId,
               BoyaSiparis.SiparisTarihi, OrmeSiparisDetay.InsertTarihi, OrmeSiparisDetay.SatinAlinacakMiktar, OrmeSiparisDetay.StoktanKullanilacakMiktar,
               OrmeSiparisDetay.PlanlananMiktar, OrmeSiparisDetay.ToplamOrulen, OrmeSiparisDetay.ToplamBoyayaCikis, OrmeSiparisDetay.ToplamSatis from OrmeSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    where (OrmeSiparisDetay.TeslimTarihi >= :ABaslangicTerminTarihi and OrmeSiparisDetay.TeslimTarihi <= :ABitisTerminTarihi) and
           OrmeSiparisDetay.Kapali = 0 and
          (OrmeSiparisDetay.HamId = :AHamId or :AHamId is null) and
          (BoyaSiparis.Fason = :AFason or :AFason is null) and
          (BoyaSiparis.FirmaId = :AFirmaId or :AFirmaId is null) and
          (OrmeSiparisDetay.OnayDurumu = :ASiparisOnayDurumu or :ASiparisOnayDurumu is null) and
           ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :APartiTipiId = 0) or
             (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :APartiTipiId = 1) or
             ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :APartiTipiId = 2) or
             (Ham.HamTipiId = :hstDokumaKumas and :APartiTipiId = 3) or :APartiTipiId is null)
  into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :OrmeSiparisDetayId,
       :FirmaId, :FirmaAdi, :SiparisKilo, :SiparisMetre,
       :HamAdi, :ReferansKodu, :Fason, :MusteriSiparisNo,
       :FSiparisNotu,  :TerminTarihi, :RevizeTerminTarihi,
       :SiparisiAlanKullaniciId,
       :SiparisTarihi, :SiparisInsertTarihi, :SatinAlinacakMiktar, :StoktanKullanilacakMiktar, :PlanlananMiktar, :ToplamOrulen, :ToplamBoyayaCikis, :ToplamSatis do
  begin
    Okey = 1;

    Select Sum(BoyaSiparisDetay.SiparisKilo), Sum(BoyaSiparisDetay.BaskaSipariseKaydirilan), Sum(BoyaSiparisDetay.BaskaSiparistenKaydirilan) from BoyaSiparisDetay
      where BoyaSiparisDetay.OrmeSiparisDetayID = :OrmeSiparisDetayID
    into :BoyaSiparisMiktar, :BaskaSipariseKaydirilan, :BaskaSiparistenKaydirilan;

    Select Sum(BoyaSiparisRezervDetay.Kilo) from BoyaSiparisDetay
      Left Join BoyaSiparisRezervDetay on BoyaSiparisRezervDetay.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayID
      where BoyaSiparisDetay.OrmeSiparisDetayID = :OrmeSiparisDetayID
    into :BoyaRezervMiktar;

    if (BoyaSiparisMiktar is Null) then BoyaSiparisMiktar = 0;
    if (BoyaRezervMiktar is Null) then BoyaRezervMiktar = 0;
    if (BaskaSipariseKaydirilan is Null) then BaskaSipariseKaydirilan = 0;
    if (BaskaSiparistenKaydirilan is Null) then BaskaSiparistenKaydirilan = 0;


    if (BoyaRezervMiktar >= BoyaSiparisMiktar) then Okey = 0;
    else if ((ToplamBoyayaCikis + BaskaSiparistenKaydirilan - BaskaSipariseKaydirilan) >= (SiparisKilo - StoktanKullanilacakMiktar)) then Okey = 0;

    if (Okey > 0) then
    begin
      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      BlobSize = char_length(FSiparisNotu);
      if (BlobSize > 1023) then SiparisNotu = Left(FSiparisNotu, 1023);
      else if (BlobSize > 0) then SiparisNotu = Left(FSiparisNotu, BlobSize);
      else SiparisNotu = '';

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      suspend;
    end
  end
end
