CREATE PROCEDURE RAFLAMAYISUMILEORGANIZEET(
    AORMERAFID INTEGER)
AS
declare variable NetGirisToplami double precision;
declare variable BrutGirisToplami double precision;
declare variable KapGirisToplami double precision;
declare variable NetCikisToplami double precision;
declare variable BrutCikisToplami double precision;
declare variable KapCikisToplami double precision;

declare variable TrNetGirisToplami double precision;
declare variable TrBrutGirisToplami double precision;
declare variable TrKapGirisToplami double precision;
declare variable TrNetCikisToplami double precision;
declare variable TrBrutCikisToplami double precision;
declare variable TrKapCikisToplami  double precision;
declare variable BobinGirisToplami  double precision;
declare variable BobinCikisToplami  double precision;
declare variable AdetGirisToplami  double precision;
declare variable AdetCikisToplami  double precision;
declare variable TrAdetGirisToplami double precision;
declare variable TrAdetCikisToplami double precision;
declare variable IslemDevamEdiyor integer;
declare variable DigerRafId integer;
begin
  if (AOrmeRafId is Null) then Exit;

  select IslemDevamEdiyor from OrmeRaf where OrmeRafId = :AOrmeRafId into :IslemDevamEdiyor;

  if (IslemDevamEdiyor = 1) then Exit;

  update OrmeRaf set IslemDevamEdiyor = 1 where OrmeRafId = :AOrmeRafId;

  select sum(Miktar), sum(BrutMiktar), sum(KapAdedi), sum(BobinSayisi), sum(Adet) from OrmeRafDetay where OrmeRafId = :AOrmeRafId into :NetGirisToplami, :BrutGirisToplami, :KapGirisToplami, :BobinGirisToplami, :AdetGirisToplami;
  select sum(Kilo), sum(BrutKilo), sum(KapAdedi), sum(Adet) from OrmeRafTransferFisiDetay where YeniOrmeRafId = :AOrmeRafId and OnayDurumu = 1 into :TrNetGirisToplami, :TrBrutGirisToplami, :TrKapGirisToplami, :TrAdetGirisToplami;

  select sum(Miktar), sum(BrutMiktar), sum(KapAdedi), sum(BobinSayisi), sum(Adet) from StokHareket where OrmeRafId = :AOrmeRafId into :NetCikisToplami, :BrutCikisToplami, :KapCikisToplami, :BobinCikisToplami, :AdetCikisToplami;
  select sum(Kilo), sum(BrutKilo), sum(KapAdedi), sum(Adet) from OrmeRafTransferFisiDetay where OrmeRafId = :AOrmeRafId and OnayDurumu = 1 into :TrNetCikisToplami, :TrBrutCikisToplami, :TrKapCikisToplami, :TrAdetCikisToplami;

  if (NetGirisToplami is Null) then NetGirisToplami = 0;
  if (BrutGirisToplami is Null) then BrutGirisToplami = 0;
  if (KapGirisToplami is Null) then KapGirisToplami = 0;
  if (NetCikisToplami is Null) then NetCikisToplami = 0;
  if (BrutCikisToplami is Null) then BrutCikisToplami = 0;
  if (KapCikisToplami is Null) then KapCikisToplami = 0;
  if (BobinGirisToplami is Null) then BobinGirisToplami = 0;
  if (BobinCikisToplami is Null) then BobinCikisToplami = 0;
  if (AdetGirisToplami is Null) then AdetGirisToplami = 0;
  if (AdetCikisToplami is Null) then AdetCikisToplami = 0;


  if (TrNetGirisToplami is Null) then TrNetGirisToplami = 0;
  if (TrBrutGirisToplami is Null) then TrBrutGirisToplami = 0;
  if (TrKapGirisToplami is Null) then TrKapGirisToplami = 0;
  if (TrNetCikisToplami is Null) then TrNetCikisToplami = 0;
  if (TrBrutCikisToplami is Null) then TrBrutCikisToplami = 0;
  if (TrKapCikisToplami is Null) then TrKapCikisToplami = 0;
  if (TrAdetGirisToplami is Null) then TrAdetGirisToplami = 0;
  if (TrAdetCikisToplami is Null) then TrAdetCikisToplami = 0;

  /*if (NetGirisToplami + TrNetGirisToplami < NetCikisToplami + TrNetCikisToplami) then
    Exception KayitHatasi ' AOrmeRafId: ' || AOrmeRafId;*/

  update OrmeRaf set GirisKapAdedi = :KapGirisToplami + :TrKapGirisToplami, BrutGirisMiktari = :BrutGirisToplami + :TrBrutGirisToplami, NetGirisMiktari = :NetGirisToplami + :TrNetGirisToplami,
    CikisKapAdedi = :KapCikisToplami + :TrKapCikisToplami, BrutCikisMiktari = :BrutCikisToplami + :TrBrutCikisToplami, NetCikisMiktari = :NetCikisToplami + :TrNetCikisToplami,
    GirisBobinSayisi = :BobinGirisToplami, CikisBobinSayisi = :BobinCikisToplami, GirisAdet = :AdetGirisToplami + :TrAdetGirisToplami, CikisAdet = :AdetCikisToplami + :TrAdetCikisToplami
  where OrmeRafId = :AOrmeRafId;

  Execute Procedure OrmeRafBosHavuzSil(:AOrmeRafId);

  update OrmeRaf set IslemDevamEdiyor = 0 where OrmeRafId = :AOrmeRafId;
end
