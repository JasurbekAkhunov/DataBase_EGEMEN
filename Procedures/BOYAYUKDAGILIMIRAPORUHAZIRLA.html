CREATE PROCEDURE BOYAYUKDAGILIMIRAPORUHAZIRLA(
    ABOYAHANEBOLUMID INTEGER,
    AFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    ASATISTIPI INTEGER)
RETURNS (
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    FIRMAID VARCHAR(20),
    USTISLEMGRUBU INTEGER,
    SARJKILO NUMERIC(12,3),
    UCRETSIZDISTAMIR INTEGER,
    ONAYDURUMU INTEGER,
    FIRMASARJ INTEGER,
    TEMSILCIID INTEGER,
    TEMSILCIADI VARCHAR(40),
    FASON INTEGER)
AS
declare variable IslemTipiId integer;
declare variable FaturaKesilecek integer;
begin
  For Select BolumPlanlama.MakinaPlanlamaGrupId, Parti.FaturaKesilecek, Parti.IslemTipiId, IslemGrubu.UstIslemGrubu, Parti.FirmaId,
             Parti.OnayDurumu, Sum(PartiStok.Kilo), Parti.Fason, Firma.TemsilciID, TK.Adi from BolumPlanlama
  Left join PartiStok on PartiStok.PartiSinifId = BolumPlanlama.PartiSinifId
  Left join Parti on Parti.PartiKayitId = BolumPlanlama.PartiKayitId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join Firma on Firma.FirmaID = Parti.FirmaID
  Left Join Kullanici TK on TK.KullaniciId = Firma.TemsilciId
  where BolumPlanlama.BoyahaneBolumID = :ABoyahaneBolumID and
        (Firma.TemsilciID = :ATemsilciId or :ATemsilciId is null) and
        (Parti.Fason = :ASatisTipi or :ASatisTipi is null)
  Group By Parti.PartiKayitId, BolumPlanlama.PartiAsamalariId, BolumPlanlama.MakinaPlanlamaGrupId, Parti.FaturaKesilecek, Parti.IslemTipiId, IslemGrubu.UstIslemGrubu, Parti.FirmaId,
           Parti.OnayDurumu, Parti.Fason, Firma.TemsilciID, TK.Adi
  into :MakinaPlanlamaGrupId, :FaturaKesilecek, :IslemTipiId, :UstIslemGrubu, :FirmaId, :OnayDurumu, :SarjKilo, :Fason, :TemsilciId, :TemsilciAdi do
  begin
    if (MakinaPlanlamaGrupId is null) then MakinaPlanlamaGrupId = '';
    if (IslemTipiId = 3 and FaturaKesilecek = 0) then UcretsizDisTamir = 1;
    else UcretsizDisTamir = 0;

    if (FirmaID = AFirmaID) then FirmaSarj = 1;
    else FirmaSarj = 0;

    suspend;
  end
end
