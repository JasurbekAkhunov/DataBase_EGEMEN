CREATE PROCEDURE HAMSTOKDETAYOLUSTUR(
    AHAMSTOKDETAYID INTEGER,
    AKULLANICIID INTEGER)
AS
begin
  Insert Into HamStokDetay(HamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
                           LycraUreticiId, TupAcikEnMayli,
                           FireOrani, BirimFiyat, Aciklama,
                           En, Grm2, FaturaNo, FaturaTarihi, InsertKullaniciId, KullaniciId)
    Select HamStokMasterId, HamStokIrsaliyeId, HamStokHareketTipiId, FirmaId, StokFirmaId, Fason, Tarih, HamId,
           LycraUreticiId, TupAcikEnMayli,

end
