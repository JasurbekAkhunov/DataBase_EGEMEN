CREATE PROCEDURE DEPOMAMULSTOKRAPORU_2(
    ADEPOID VARCHAR(10),
    AFIRMAID VARCHAR(20),
    ARAPORHAMTIPI INTEGER,
    ASATISTIPI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AHAMID INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ATAMIRDURUMU INTEGER,
    ADESENID VARCHAR(20),
    ARECETEID VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AKALITENO INTEGER,
    AYURTDISI EGEBOOLEAN,
    ATOPDETAYLIGOSTER EGEBOOLEAN,
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ABARKODID VARCHAR(30),
    ABELGEHAMADI ADI60,
    ARAFID RAFKODUSIZE)
RETURNS (
    PARTISTOKTARTIID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    TARIH TIMESTAMP,
    KAPADEDI INTEGER,
    DARA BOYAMIKTARSIZE,
    KILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    ADET INTEGER,
    BOBINSAYISI INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    BOYAIRSALIYEID INTEGER,
    HAMSTOKSARFFISIDETAYID INTEGER,
    HAMSTOKSARFFISIID INTEGER,
    MAMULDEPOTRANSFERFISIID INTEGER,
    KALITENO INTEGER,
    KALITEKONTROL INTEGER,
    DEPOID VARCHAR(10),
    BRUTKILO BOYAMIKTARSIZE,
    BRUTMETRE BOYAMIKTARSIZE,
    PALETNO VARCHAR(5),
    KALITEACIKLAMATIPIID INTEGER,
    MAMULPROSESID INTEGER,
    BOYAREZERVFISIDETAYID INTEGER,
    IADEBOYAIRSALIYEDETAYID INTEGER,
    IADEBOYAIRSALIYEID INTEGER,
    BARKODID VARCHAR(30),
    SIPARISPROSESADI VARCHAR(255),
    SIPARISPROSESADI2 VARCHAR(255))
AS
declare variable itTamir integer;
declare variable APartiKayitId  integer;
declare variable ABoyaSiparisKayitId integer;
declare variable Okey EgeBoolean;
declare variable FRafID  varchar(10);
begin
  Select itTamir from IslemTipiGetir into :itTamir;

  if (APartiId is not null) then
  begin
    Select PartiKayitId from Parti where Parti.PartiYil = :APartiYil and Parti.PartiId = :APartiId into :APartiKayitId;
    if (APartiKayitId is null) then APartiKayitId = -1;
  end
  else APartiKayitId = Null;

  if (ABoyaSiparisYil is not Null) then
  begin
    Select BoyaSiparis.BoyaSiparisKayitId from BoyaSiparis
      where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId
      into :ABoyaSiparisKayitId;
    if (ABoyaSiparisKayitId is null) then ABoyaSiparisKayitId = -1;
  end
  else ABoyaSiparisKayitId = null;

  for Select PartiStokTarti.PartiStokTartiId, PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.Tarih, PartiStokTarti.KapAdedi, PartiStokTarti.Dara, PartiStokTarti.Kilo,
             PartiStokTarti.Metre, PartiStokTarti.Adet, PartiStokTarti.BobinSayisi, PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.BoyaIrsaliyeId, PartiStokTarti.HamStokSarfFisiDetayId,
             PartiStokTarti.HamStokSarfFisiId, PartiStokTarti.MamulDepoTransferFisiId, PartiStokTarti.KaliteNo, PartiStokTarti.KaliteKontrol, PartiStokTarti.DepoId,
             PartiStokTarti.BrutKilo, PartiStokTarti.BrutMetre, PartiStokTarti.PaletNo, PartiStokTarti.KaliteAciklamaTipiId, PartiStokTarti.MamulProsesId, PartiStokTarti.BoyaRezervFisiDetayId,
             PartiStokTarti.IadeBoyaIrsaliyeDetayId, PartiStokTarti.IadeBoyaIrsaliyeId, PartiStokTarti.BarkodId, SiparisProses.Adi, SiparisProses.Adi2
             from PartiStokTarti
    left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
    left join Kalite on Kalite.KaliteId = PartiStok.KaliteId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparis on BoyaSiparis.BoyaSipariskayitId = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join FirmaSevkAdresi on FirmaSevkAdresi.FirmaSevkAdresiId = BoyaSiparisDetay.FirmaSevkAdresiId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiID
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    left join Depo on Depo.DepoId = PartiStokTarti.DepoId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    where PartiStokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null and
          (:ADepoId is not null and Depo.DepoId = :ADepoId or :ADepoId is null and Depo.Sevk = 1) and
          (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
          (PartiStok.HamId = :AHamId or :AHamId is null) and
          (BoyaSiparisDetay.SiparisTipiId = :ARaporHamTipi or :ARaporHamTipi is null) and
          (Parti.Fason = :ASatisTipi or :ASatisTipi is null) and
          (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
          (:ATamirDurumu = 2 or (:ATamirDurumu = 0 and Parti.IslemTipiId = :itTamir) or (:ATamirDurumu = 1 and Parti.IslemTipiId <> :itTamir)) and
          (Parti.ReceteId = :AReceteId or :AReceteId is null) and
          (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
          (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
          (Desen.DesenId = :ADesenID or :ADesenID is Null) and
          (PartiStokTarti.KaliteNo = :AKaliteNo or :AKaliteNo is Null) and
          (BoyaSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null) and
          (PartiStokTarti.PartiKayitId = :APartiKayitId or :APartiKayitId is null) and
          (PartiStok.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          (PartiStokTarti.BarkodId = :ABarkodId or :ABarkodId is null) and
          (PartiStokTarti.BarkodId = :ABarkodId or :ABarkodId is null) and
          (BoyaSiparisDetay.BelgeHamAdi = :ABelgeHamAdi or :ABelgeHamAdi is null)
  into :PartiStokTartiId, :PartiKayitId, :PartiSinifId, :Tarih, :KapAdedi, :Dara, :Kilo,
             :Metre, :Adet, :BobinSayisi, :BoyaIrsaliyeDetayId, :BoyaIrsaliyeId, :HamStokSarfFisiDetayId,
             :HamStokSarfFisiId, :MamulDepoTransferFisiId, :KaliteNo, :KaliteKontrol, :DepoId,
             :BrutKilo, :BrutMetre, :PaletNo, :KaliteAciklamaTipiId, :MamulProsesId, :BoyaRezervFisiDetayId,
             :IadeBoyaIrsaliyeDetayId, :IadeBoyaIrsaliyeId, :BarkodId, :SiparisProsesAdi, :SiparisProsesAdi2 do
  begin
    Okey = 1;

    if (ARafID is not null)  then
    begin
      Okey = 0;
      for select MamulDepoTransferFisiDetay.RafId from PartiStokTarti
        left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiID and MamulDepoTransferFisiDetay.PartiStokTartiID = PartiStokTarti.PartiStoktartiID
        left join Depo on Depo.DepoId = PartiStokTarti.DepoId
        where PartiStokTarti.PartiSinifID = :PartiSinifID and
              PartiStokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null and
              (:ADepoId is not null and PartiStokTarti.DepoId = :ADepoId or :ADepoId is null and Depo.Sevk = 1) and
              (MamulDepoTransferFisiDetay.RafId = :ARafId)
        group by MamulDepoTransferFisiDetay.RafId
        into :FRafId do
      begin
        if (ARafId = FRafId) then Okey = 1;
      end
    end
    if (okey = 1) then suspend;
  end
end
