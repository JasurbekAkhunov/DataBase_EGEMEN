CREATE PROCEDURE OSRICINIPLIKBILGILERI(
    ORMESIPARISDETAYID INTEGER)
RETURNS (
    IPLIKSATISIADE NUMERIC(12,3),
    IPLIKSATISIADETAMIR NUMERIC(12,3),
    IPLIKSATIS NUMERIC(12,3),
    IPLIKSATISTAMIR NUMERIC(12,3),
    IPLIKSATISIADENET NUMERIC(12,3),
    IPLIKSATISIADETAMIRNET NUMERIC(12,3),
    IPLIKSATISNET NUMERIC(12,3),
    IPLIKSATISTAMIRNET NUMERIC(12,3),
    IPLIKMUSTERIYESATIS NUMERIC(12,3),
    IPLIKMUSTERIYESATISNET NUMERIC(12,3),
    IPLIKORGUYESATIS NUMERIC(12,3),
    IPLIKORGUYESATISNET NUMERIC(12,3),
    TOPLAMFATURAMIKTAR NUMERIC(12,3),
    TOPLAMBOYALIFATURAMIKTAR NUMERIC(12,3),
    FTOPLAMBOYALIFATURAMIKTAR NUMERIC(12,3),
    URETIMMIKTARI NUMERIC(12,3),
    TOPLAMBOYALISATIS NUMERIC(12,3))
AS
declare variable ShtSatis               integer;
declare variable OrmeFirmaKodumuz  varchar(20);
declare variable FirmaKodumuz  varchar(20);
begin
  IplikSatisIade        = 0;
  IplikSatisIadeTamir   = 0;
  IplikSatis            = 0;
  IplikSatisTamir       = 0;
  IplikSatisIadeNet      = 0;
  IplikSatisIadeTamirNet = 0;
  IplikSatisNet          = 0;
  IplikSatisTamirNet     = 0;

  IplikMusteriyeSatis     = 0;
  IplikMusteriyeSatisNet  = 0;
  IplikOrguyeSatis        = 0;
  IplikOrguyeSatisNet     = 0;
  UretimMiktari = 0;
  ToplamBoyaliSatis = 0;

  Select OrmeFirmaKodumuz, FirmaKodumuz from Sistem where SistemId = 1 into :OrmeFirmaKodumuz, :FirmaKodumuz;

  select Sum(iif(IplikIrsaliye.FirmaID <> :OrmeFirmaKodumuz and IplikIrsaliye.FirmaID <> :FirmaKodumuz, IplikIrsaliyeDetay.Miktar, 0)),
         Sum(iif(IplikIrsaliye.FirmaID <> :OrmeFirmaKodumuz and IplikIrsaliye.FirmaID <> :FirmaKodumuz, IplikIrsaliyeDetay.NetMiktar, 0)),
         Sum(iif(IplikIrsaliye.FirmaID = :OrmeFirmaKodumuz, IplikIrsaliyeDetay.Miktar, 0)),
         Sum(iif(IplikIrsaliye.FirmaID = :OrmeFirmaKodumuz, IplikIrsaliyeDetay.NetMiktar, 0)),
         sum(IplikIrsaliyeDetay.Miktar), sum(IplikIrsaliyeDetay.NetMiktar)
         from IplikTalepDetay
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayId = IplikTalepDetay.IplikTalepDetayId
    left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    left join IplikLot on IplikLot.IplikLotId = IplikIrsaliyeDetay.IplikLotId
    where IplikTalepDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and IplikIrsaliye.HareketTipi = 1 and (IplikLot.Tamir = 0 or IplikLot.Tamir is Null)
    into :IplikMusteriyeSatis, :IplikMusteriyeSatisNet, :IplikOrguyeSatis, :IplikOrguyeSatisNet, :IplikSatis, :IplikSatisNet;

  select sum(IplikIrsaliyeDetay.Miktar), sum(IplikIrsaliyeDetay.NetMiktar) from IplikTalepDetay
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayId = IplikTalepDetay.IplikTalepDetayId
    left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    left join IplikLot on IplikLot.IplikLotId = IplikIrsaliyeDetay.IplikLotId
    where IplikTalepDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and IplikIrsaliye.HareketTipi = 1 and IplikLot.Tamir = 1
    into :IplikSatisTamir , :IplikSatisTamirNet;

  select Sum(iif(IplikSatisIade.Tamir = 0, IplikSatisIade.Miktar, 0)),
         Sum(iif(IplikSatisIade.Tamir = 0, IplikSatisIade.NetMiktar, 0)),
         Sum(iif(IplikSatisIade.Tamir = 1, IplikSatisIade.Miktar, 0)),
         Sum(iif(IplikSatisIade.Tamir = 1, IplikSatisIade.NetMiktar, 0))
         from IplikTalepDetay
    left join IplikSatisIade on IplikSatisIade.IplikTalepDetayId = IplikTalepDetay.IplikTalepDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    where IplikTalepDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId
    into :IplikSatisIade, :IplikSatisIadeNet, :IplikSatisIadeTamir , :IplikSatisIadeTamirNet;

  Select Sum(IplikUretim.Miktar) from IplikTalepDetay
    left Join IplikUretim on IplikUretim.IplikTalepDetayID = IplikTalepDetay.IplikTalepDetayID
    where IplikTalepDetay.OrmeSiparisDetayID = :OrmeSiparisDetayID
  into :UretimMiktari;

  /*Bu bolum yukariyla pek alakali degil, sigmadigindan o da buraya kondu, kafa karistirmasin */
  Select ShtSatis from OrmeStokHareketTipiGetir into :ShtSatis;


  Select Sum(iif(StokHareket.BoyaSiparisDetayID is null, StokHareket.Miktar, 0)),
         Sum(iif(StokHareket.BoyaSiparisDetayID is not null, StokHareket.Miktar, 0)) from OrmeSiparisDetay
    left join StokHareket on StokHareket.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
    where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and StokHareket.StokHareketID is not null and
          StokHareket.HareketTipi = :shtSatis and StokHareket.HamIade = 0
    into :ToplamFaturaMiktar, :ToplamBoyaliFaturaMiktar;

  Select Sum(BoyaFaturaDetay.Miktar), Sum(iif(BoyaIrsaliye.FirmaID <> :OrmeFirmaKodumuz, BoyaIrsaliyeDetay.Miktar, 0)) from OrmeSiparisDetay
    left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
    left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
    where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and BoyaSiparisDetay.BoyaSiparisDetayID is not null and
          PartiStok.PartiSinifID is not null and BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID is not null
    into :FToplamBoyaliFaturaMiktar, :ToplamBoyaliSatis;

  if (ToplamBoyaliSatis is null) then ToplamBoyaliSatis = 0;
  if (ToplamBoyaliFaturaMiktar is null) then ToplamBoyaliFaturaMiktar = 0;
  if (FToplamBoyaliFaturaMiktar is null) then FToplamBoyaliFaturaMiktar = 0;

  ToplamBoyaliFaturaMiktar = ToplamBoyaliFaturaMiktar + FToplamBoyaliFaturaMiktar;
  if (ToplamBoyaliFaturaMiktar = 0) then ToplamBoyaliFaturaMiktar = null;

  Suspend;
end
