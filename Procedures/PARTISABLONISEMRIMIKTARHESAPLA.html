CREATE PROCEDURE PARTISABLONISEMRIMIKTARHESAPLA(
    APARTISABLONISEMRIID INTEGER,
    AREFPARTIKAYITID INTEGER,
    AREFPARTIGUNCELLE INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable RefPartiKayitID Integer;
declare variable Acilim1 Double Precision;
declare variable Acilim2 Double Precision;
declare variable HazirlananMiktar Double Precision;
declare variable RefPartiMiktar Double Precision;
declare variable BirimHazirlananMiktar Double Precision;
declare variable BirimKullanilanMiktar Double Precision;
declare variable SablonNo Integer;
declare variable EmulsiyonUrunId Varchar(20);
declare variable TopPartiMiktar double precision;
declare variable DepoID varchar(10);
declare variable PatMiktari Double Precision;
declare variable FikseSicakligi Integer;
declare variable UrunID Varchar(20);
declare variable UrunBirimId Integer;
declare variable Miktar Double Precision;
declare variable KullanilanMiktar Double Precision;
declare variable Pat SmallInt;
declare variable Lot Varchar(150);
Begin
  Select Sum(PartiStok.Miktar) from PartiStok where PartiStok.PARTISTOKIPTALKAYITID IS NULL AND PartiStok.PartiKayitID = :ARefPartiKayitID into :RefPartiMiktar;

  /*Ref Parti Upd*/
  if (ARefPartiGuncelle = 1) then
  begin
    Select Sum(PartiStok.Miktar) from PartiStok where PartiStok.PARTISTOKIPTALKAYITID IS NULL AND PartiStok.PartiKayitID in
      (select distinct partisablon.PartiKayitID from partisablon where partisablon.PartiSablonIsEmriID = :APartiSablonIsEmriID) into :TopPartiMiktar;

    For Select PartiSablon.SablonNo, PartiSablon.EmulsiyonUrunId, PartiSablon.Acilim1, PartiSablon.Acilim2, PartiSablon.HazirlananMiktar,
               PartiSablon.DepoID, PartiSablon.PatMiktari, PartiSablon.FikseSicakligi from PartiSablon
      where PartiSablon.PartiSablonIsEmriID = :APartiSablonIsEmriID and PartiSablon.PartiKayitID = :ARefPartiKayitID
      into :SablonNo, :EmulsiyonUrunId, :Acilim1, :Acilim2, :HazirlananMiktar, :DepoID, :PatMiktari, :FikseSicakligi do
    begin
      if (HazirlananMiktar > 0 and RefPartiMiktar > 0) then BirimHazirlananMiktar =  HazirlananMiktar / RefPartiMiktar;
      else BirimHazirlananMiktar = 0;

      Update PartiSablon set EmulsiyonUrunId = :EmulsiyonUrunId, Acilim1 = :Acilim1, Acilim2 = :Acilim2, DepoID = :DepoID, PatMiktari = :PatMiktari, FikseSicakligi = :FikseSicakligi,KullaniciID = :AKullaniciID,
                             HazirlananMiktar = :HazirlananMiktar * (Select Sum(PartiStok.Miktar) from Partistok where PartiStok.PartiStokIptalKayitId is Null and PartiStok.PartiKayitID = PartiSablon.PartiKayitID)/:TopPartiMiktar
      where PartiSablon.PartiSablonIsEmriID = :APartiSablonIsEmriID and PartiSablon.PartiKayitID = :ARefPartiKayitID and PartiSablon.SablonNo = :SablonNo ;

    end
  end
  /*Ref Parti Upd*/

  For Select PartiSablon.SablonNo, PartiSablon.EmulsiyonUrunId, PartiSablon.Acilim1, PartiSablon.Acilim2, PartiSablon.HazirlananMiktar,
             PartiSablon.DepoID, PartiSablon.PatMiktari, PartiSablon.FikseSicakligi from PartiSablon
    where PartiSablon.PartiSablonIsEmriID = :APartiSablonIsEmriID and PartiSablon.PartiKayitID = :ARefPartiKayitID
    into :SablonNo, :EmulsiyonUrunId, :Acilim1, :Acilim2, :HazirlananMiktar, :DepoID, :PatMiktari, :FikseSicakligi do
  begin
    if (HazirlananMiktar > 0 and RefPartiMiktar > 0) then BirimHazirlananMiktar =  HazirlananMiktar / RefPartiMiktar;
    else BirimHazirlananMiktar = 0;

    Update PartiSablon set EmulsiyonUrunId = :EmulsiyonUrunId, Acilim1 = :Acilim1, Acilim2 = :Acilim2, DepoID = :DepoID, PatMiktari = :PatMiktari, FikseSicakligi = :FikseSicakligi,KullaniciID = :AKullaniciID,
                           HazirlananMiktar = :BirimHazirlananMiktar * (Select Sum(PartiStok.Miktar) from Partistok where PartiStok.PartiStokIptalKayitId is Null and PartiStok.PartiKayitID = PartiSablon.PartiKayitID)
    where PartiSablon.PartiSablonIsEmriID = :APartiSablonIsEmriID and PartiSablon.PartiKayitID <> :ARefPartiKayitID and PartiSablon.SablonNo = :SablonNo ;

  end

  Delete from PartiSablonRecete where PartiSablonID in (Select PartiSablonID from PartiSablon Where PartiSablonIsEmriID = :APartiSablonIsEmriID and PartiSablon.PartiKayitID <> :ARefPartiKayitID);

  Delete from PartiSablonRecete where PartiKayitID = :ARefPartiKayitID and Pat = 1;

  Execute Procedure PartiSablonAcilimiHesapla(:ARefPartiKayitID);


  For Select PartiSablon.SablonNo, PartiSablonRecete.UrunID, PartiSablonRecete.UrunBirimId, PartiSablonRecete.Miktar, PartiSablonRecete.KullanilanMiktar, PartiSablonRecete.Pat,
            PartiSablonRecete.Lot from PartiSablon
        Left Join PartiSablonRecete on PartiSablonRecete.PartiSablonID = PartiSablon.PartiSablonID
      Where PartiSablonIsEmriID = :APartiSablonIsEmriID and PartiSablon.PartiKayitID = :ARefPartiKayitID and PartiSablonRecete.PartiSablonReceteID is not Null

      into :SablonNo, :UrunID, :UrunBirimId, :Miktar, :KullanilanMiktar, :Pat, :Lot do
  begin
    if (KullanilanMiktar > 0 and RefPartiMiktar > 0) then BirimKullanilanMiktar =  KullanilanMiktar / RefPartiMiktar;
    else BirimKullanilanMiktar = 0;


    Insert Into PartiSablonRecete (PartiSablonID, PartiKayitID, UrunID, UrunBirimID, Miktar, KullanilanMiktar, Pat, InsertKullaniciID, KullaniciID, Lot)
      Select PartiSablonID, PartiKayitID, :UrunId, :UrunBirimId, :Miktar, :BirimKullanilanMiktar * (Select Sum(PartiStok.Miktar) from Partistok where PartiStok.PartiStokIptalKayitId is Null and PartiStok.PartiKayitID = PartiSablon.PartiKayitID),
      :Pat, :AKullaniciID, :AKullaniciID, :Lot from PartiSablon
      where PartiSablon.PartiSablonIsEmriID = :APartiSablonIsEmriID and PartiSablon.PartiKayitID <> :ARefPartiKayitID and PartiSablon.SablonNo = :SablonNo;
  end
end
