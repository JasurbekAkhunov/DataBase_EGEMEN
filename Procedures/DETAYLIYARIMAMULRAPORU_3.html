CREATE PROCEDURE DETAYLIYARIMAMULRAPORU_3(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIH TIMESTAMP,
    ABITISTARIH TIMESTAMP,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    AFASON INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    HAMSTOKSARFFISIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(50),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    PARTIKILO NUMERIC(12,3),
    PARTIMETRE NUMERIC(12,3),
    ISLETMESTOKKULLANILANKILO NUMERIC(12,3),
    ISLETMESTOKKULLANILANMETRE NUMERIC(12,3),
    BOYASIPARISIDSTR VARCHAR(30),
    RAFDETAYID INTEGER)
AS
declare variable FBoyaSiparisYil integer;
declare variable FBoyaSiparisId  integer;
declare variable FFisYil integer;
declare variable FFisNo  integer;
begin
/*yari mamulun hangi partilerde kullanildigini g√∂steriyoruz */
  for select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.FirmaId,
             Firma.Adi, Parti.Fason, PartiStok.Kilo, PartiStok.Metre, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
              sum(PartiStokDetay.Kilo), sum(PartiStokDetay.Metre), DetayliYariMamulRaporu_2.RafDetayId from DetayliYariMamulRaporu_2(:ARaporTipi, :ABaslangicTarih, :ABitisTarih, :ABaslangicPartiYil, :ABaslangicPartiId, :ABitisPartiYil, :ABitisPartiId, :AHamTipiId, :AHamId, :AFason)
    left join PartiStokDetay on PartiStokDetay.RafDetayId = DetayliYariMamulRaporu_2.RafDetayId
    left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
    left join Firma on Firma.FirmaID = Parti.FirmaID
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where PartiStok.PartiSinifId is not null
    group by DetayliYariMamulRaporu_2.RafDetayId, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.FirmaId,
             Firma.Adi, Parti.Fason, PartiStok.Kilo, PartiStok.Metre, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId
    into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :FirmaId, :FirmaAdi, :Fason,
         :PartiKilo, :PartiMetre, :FBoyaSiparisYil, :FBoyaSiparisId,
         :IsletmeStokKullanilanKilo, :IsletmeStokKullanilanMetre, :RafDetayId do
  begin
    BoyaSiparisIdStr = '';

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (FBoyaSiparisId is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') '  || EgemenIntegerToStr(FBoyaSiparisId);

    suspend;
  end

  PartiKayitId          = null;
  PartiYil              = null;
  PartiId               = null;
  PartiIdStr            = null;
  PartiSinifId          = null;
  FirmaId               = null;
  FirmaAdi              = null;
  Fason                 = null;
  FasonStr              = null;
  PartiKilo             = null;
  PartiMetre            = null;
  BoyaSiparisIdStr      = '';

  for select DetayliYariMamulRaporu_2.RafDetayId,  HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo,
             sum(HamStokSarfFisiDetay.Kilo), sum(HamStokSarfFisiDetay.Metre) from DetayliYariMamulRaporu_2(:ARaporTipi, :ABaslangicTarih, :ABitisTarih, :ABaslangicPartiYil, :ABaslangicPartiId, :ABitisPartiYil, :ABitisPartiId, :AHamTipiId, :AHamId, :AFason)
    left join HamStokSarfFisiDetay on HamStokSarfFisiDetay.RafDetayId = DetayliYariMamulRaporu_2.RafDetayId
    left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiID = HamStokSarfFisiDetay.HamStokSarfFisiId
    where HamStokSarfFisiDetay.HamStokSarfFisiId is not null
    group by DetayliYariMamulRaporu_2.RafDetayId, HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo

    into :RafDetayId, :FFisYil, :FFisNo, :IsletmeStokKullanilanKilo, :IsletmeStokKullanilanMetre do
  begin
    HamStokSarfFisiIdStr = '(' || EgemenIntegerToStr(FFisYil)  || ') ' || EgemenIntegerToStr(FFisNo) || ' ';

    suspend;
  end
end
