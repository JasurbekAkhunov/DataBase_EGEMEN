CREATE PROCEDURE FIRMABAKIYEGETIR(
    AFIRMAID VARCHAR(20),
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1),
    ADOVIZID VARCHAR(5))
RETURNS (
    BORCTOPLAM DOUBLE PRECISION,
    ALACAKTOPLAM DOUBLE PRECISION,
    BORCBAKIYE DOUBLE PRECISION,
    ALACAKBAKIYE DOUBLE PRECISION,
    BAKIYE DOUBLE PRECISION,
    BAKIYEBA VARCHAR(1),
    BAKIYENINVADETARIHI TIMESTAMP)
AS
declare variable FMeblag Double Precision;
declare variable FBorcAlacak varchar(1);
begin
  BorcToplam = 0;
  AlacakToplam = 0;
  Bakiye = 0;
  BakiyeBA = ' ';
/*
  select sum( Meblag / DovizKuru1 * DovizKuru2)  from cari where FirmaId = :AFirmaId and (tarih <= :ABitisTarihi or :ABitistarihi is null) and BorcAlacak = 'B' into :BorcToplam;
  select sum( Meblag / DovizKuru1 * DovizKuru2)  from cari where FirmaId = :AFirmaId and (tarih <= :ABitisTarihi or :ABitistarihi is null) and BorcAlacak = 'A' into :AlacakToplam;
*/
  for select  BorcAlacak, iif(:ADovizID is Null, sum( Meblag / DovizKuru1 * DovizKuru2), sum( Meblag))  from cari
       where FirmaId = :AFirmaId and (tarih <= :ABitisTarihi or :ABitistarihi is null) and (Cari.Sinif = :ASinif or :ASinif is null) and
             (DovizID = :ADovizID or :ADovizID is Null)
       group by Cari.BorcAlacak
       into :FBorcAlacak, :FMeblag do
  begin
    if (FBorcAlacak = 'B') then BorcToplam = FMeblag;
    else if (FBorcAlacak = 'A') then AlacakToplam = FMeblag;
  end


  BorcBakiye = null;
  AlacakBakiye = Null;

  if (BorcToplam is Null) then BorcToplam = 0;
  if (AlacakToplam is Null) then AlacakToplam = 0;

  if (BorcToplam > AlacakToplam) then
  begin
    BorcBakiye = BorcToplam - AlacakToplam;
    Bakiye = BorcBakiye;
    BakiyeBA = 'B';
  end
  else
  begin
    AlacakBakiye = AlacakToplam - BorcToplam;
    Bakiye = AlacakBakiye;
    if (AlacakToplam > BorcToplam) then BakiyeBA = 'A';
  end

  if (BorcBakiye = 0) then BorcBakiye = null;
  if (AlacakBakiye = 0) then AlacakBakiye = null;


  select BakiyeVadeTarihi from FirmaCariBakiyeVadesiHesapla(:AFirmaId, :Bakiye, :BakiyeBA, :ABitisTarihi, :ASinif, :ADovizId) into :BakiyeninVadeTarihi;

  suspend;
end
