CREATE PROCEDURE ARIZAIHBARRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAID VARCHAR(10),
    AARIZATIPIID INTEGER,
    ABITTI INTEGER,
    APARCABEKLIYOR INTEGER)
RETURNS (
    ARIZAIHBARID INTEGER,
    MAKINAID VARCHAR(10),
    BASLATANPERSONELID INTEGER,
    BASLATANPERSONELADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENPERSONELID INTEGER,
    BITIRENPERSONELADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    BITTI INTEGER,
    BITTISTR VARCHAR(5),
    ARIZATIPIID INTEGER,
    ARIZATIPIADI VARCHAR(50),
    ARIZAYERIID INTEGER,
    ARIZAYERIADI VARCHAR(20),
    PLANLIBAKIMNO VARCHAR(20),
    NOTU MEMO,
    SERVISNOTU MEMO,
    BAKIMCIVAR EGEBOOLEAN,
    BAKIMYAPANPERSONELADI VARCHAR(550),
    PARCABEKLIYOR INTEGER,
    PARCABEKLIYORSTR VARCHAR(4),
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30),
    BAKIMCIILKGELISTARIHI TIMESTAMP,
    BAKIMCIGECENSURE DOUBLE PRECISION,
    BAKIMCIGECENSURESTR VARCHAR(30),
    BASPERSMAKINABAKIMDAIREID INTEGER,
    BASPERSMAKINABAKIMDAIREADI VARCHAR(255),
    BITPERSMAKINABAKIMDAIREID INTEGER,
    BITPERSMAKINABAKIMDAIREADI VARCHAR(255),
    CALISANPERSONELSAYISI NUMERIC(12,3),
    ARIZASEKLIADI VARCHAR(50),
    MAKINECALISMIYOR EGEBOOLEAN,
    MAKINECALISMIYORSTR VARCHAR(12),
    PARCABEKLEMETARIHI TIMESTAMP,
    PARCAGELISTARIHI TIMESTAMP)
AS
begin
  For Select ArizaIhbar.ArizaIhbarId, ArizaIhbar.MakinaId, ArizaIhbar.BaslatanPersonelId, BasP.Adi, MD1.Tarih,
             ArizaIhbar.BitirenPersonelId, BitP.Adi, MD2.Tarih, ArizaIhbar.ArizaTipiId, ArizaTipi.Adi, ArizaIhbar.ArizaYeriId,
             ArizaIhbar.PlanliBakimNo, ArizaIhbar.Notu, ArizaIhbar.ServisNotu, ArizaIhbar.ParcaBekliyor,
             BasP.BakimDairesiId, BasBakimDairesi.Adi, BitP.BakimDairesiId, BitBakimDairesi.Adi, ArizaIhbar.CalisanPersonelSayisi,
             ArizaSekli.Adi, ArizaIhbar.MakineCalismiyor, ArizaIhbar.ParcaBeklemeTarihi, ArizaIhbar.ParcaGelisTarihi from ArizaIhbar
    left join Personel BasP on BasP.PersonelId = ArizaIhbar.BaslatanPersonelId
    left join Personel BitP on BitP.PersonelId = ArizaIhbar.BitirenPersonelId
    left join BakimDairesi BasBakimDairesi on BasBakimDairesi.BakimDairesiId = BasP.BakimDairesiId
    left join BakimDairesi BitBakimDairesi on BitBakimDairesi.BakimDairesiId = BitP.BakimDairesiId
    left join MakinaDurumu MD1 on MD1.MakinaDurumuId = ArizaIhbar.BasladiMakinaDurumuId
    left join MakinaDurumu MD2 on MD2.MakinaDurumuId = ArizaIhbar.BittiMakinaDurumuId
    left join ArizaTipi on ArizaTipi.ArizaTipiId = ArizaIhbar.ArizaTipiId
    left join ArizaSekli on ArizaIhbar.ArizaSekliId = ArizaSekli.ArizaSekliId
    Where (ArizaIhbar.MakinaId = :AMakinaId or :AMakinaId is null) and
          (ArizaIhbar.ArizaTipiId = :AArizaTipiId or :AArizaTipiId is null) and
          (MD1.Tarih >= :ABaslangicTarihi and MD1.Tarih <= :ABitisTarihi) and
          (:ABitti is null or (ArizaIhbar.BitirenPersonelId is not null and :ABitti = 1) or
            (ArizaIhbar.BitirenPersonelId is null and :ABitti = 0) ) and
          (ArizaIhbar.ParcaBekliyor = :AParcaBekliyor or :AParcaBekliyor is null)
    into :ArizaIhbarId, :MakinaId, :BaslatanPersonelId, :BaslatanPersonelAdi, :BaslamaTarihi, :BitirenPersonelId, :BitirenPersonelAdi,
         :BitisTarihi, :ArizaTipiId, :ArizaTipiAdi, :ArizaYeriId, :PlanliBakimNo, :NOTU, :SERVISNOTU, :ParcaBekliyor,
         :BasPersMakinaBakimDaireID, :BasPersMakinaBakimDaireAdi, :BitPersMakinaBakimDaireID, :BitPersMakinaBakimDaireAdi, :CalisanPersonelSayisi,
         :ArizaSekliAdi, :MakineCalismiyor, :ParcaBeklemeTarihi ,:ParcaGelisTarihi  do
  begin
    if (BitirenPersonelId is not null) then Bitti = 1;
    else Bitti = 0;

    if (Bitti = 1) then BittiStr = 'Bitti';
    else BittiStr = '';

    if (MakineCalismiyor = 1) then MakineCalismiyorStr = 'Çalışmıyor';
    else MakineCalismiyorStr = 'Çalışıyor';

    if (ParcaBekliyor = 1) then ParcaBekliyorStr = 'Evet';
    else ParcaBekliyorStr = '';

    if (ArizaYeriId = 0) then ArizaYeriAdi = 'Mekanik';
    else if (ArizaYeriId = 1) then ArizaYeriAdi = 'Elektrik';
    else if (ArizaYeriId is not null) then ArizaYeriAdi = 'Hatalı';
    else ArizaYeriAdi = '';

    Select Count(*) from ArizaBakim
      where ArizaBakim.ArizaIhbarId = :ArizaIhbarId and
            ArizaBakim.GeldiMakinaDurumuId is Not Null and
            ArizaBakim.GittiMakinaDurumuId is Null
      into :BakimciVar;
    if (BitisTarihi is not null) then GecenSure = BitisTarihi - BaslamaTarihi;
    else GecenSure = (CURRENT_DATE + CURRENT_TIME) - BaslamaTarihi;

    GecenSureStr = TarihToStr(GecenSure);

    BakimYapanPersonelAdi = '';

    Select COALESCE(LEFT(LIST(Personel.Adi, cast(', ' as varchar(2) character set UTF8)), 550), '')
      from ArizaBakim
      left join Personel on Personel.PersonelId = ArizaBakim.PersonelId
      where ArizaBakim.ArizaIhbarId = :ArizaIhbarId
      into :BakimYapanPersonelAdi;

    BakimciIlkGelisTarihi = Null;

    Select Min(MakinaDurumu.Tarih) from ArizaBakim
      left join MakinaDurumu on MakinaDurumu.MakinaDurumuId = ArizaBakim.GeldiMakinaDurumuId
      where ArizaBakim.ArizaIhbarId = :ArizaIhbarId
      into :BakimciIlkGelisTarihi;

    if (BakimciIlkGelisTarihi is not null) then
    begin
      BakimciGecenSure = BakimciIlkGelisTarihi - BaslamaTarihi;
      BakimciGecenSureStr = TarihToStr(BakimciGecenSure);
    end
    else
    begin
      BakimciGecenSure = 0;
      BakimciGecenSureStr = '';
    end

    suspend;
  end
end
