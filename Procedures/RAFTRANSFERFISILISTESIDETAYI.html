CREATE PROCEDURE RAFTRANSFERFISILISTESIDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    RAFTRANSFERFISIID INTEGER,
    RAFTRANSFERFISIDETAYID INTEGER,
    HAMADI VARCHAR(255),
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    RAFDETAYID INTEGER,
    ESKIRAFID VARCHAR(10),
    YENIRAFID VARCHAR(10),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    LOT VARCHAR(150),
    PALETNO VARCHAR(5),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    GIRISEVRAKTIPI INTEGER,
    GIRISEVRAKTIPISTR VARCHAR(8),
    KALITENO INTEGER,
    PUS INTEGER,
    FEINE INTEGER,
    EN INTEGER,
    ORMECIMAKINANO VARCHAR(10),
    IPLIKUZUNLUGU1 VARCHAR(40),
    IPLIKUZUNLUGU2 VARCHAR(40),
    IPLIKUZUNLUGU3 VARCHAR(40),
    GRM2 INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    FIRMAADI ADI100,
    SIPARISFIRMAADI ADI100)
AS
Declare Variable FTupAcikEnMayli Integer;
begin
  For Select RafTransferFisiListesi.RafTransferFisiId, RafTransferFisiListesi.FisNoStr, RafTransferFisiListesi.FisTarihi,
             RafTransferFisiDetay.RafTransferFisiDetayId, Ham.Adi, HamStokDetay.HamStokDetayId, HamStokDetay.HamStokMasterId,
             RafTransferFisiDetay.RafDetayId, RafDetay.RafId, RafTransferFisiDetay.RafId,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             HamStokMaster.Lot, RafTransferFisiDetay.KapAdedi, RafTransferFisiDetay.Kilo, RafTransferFisiDetay.Metre,
             RafTransferFisiDetay.KaliteNo, RafTransferFisiDetay.PaletNo, HamStokMaster.Pus, HamStokMaster.Feine,
             HamStokMaster.En, HamStokMaster.OrmeciMakinaNo, HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3,
             HamStokMaster.Grm2, HamStokMaster.TupAcikEnMayli, F1.Adi, F2.Adi from RafTransferFisiListesi(:ABaslangicTarihi, :ABitisTarihi, :AHamStokMasterID)
  left join RafTransferFisiDetay on RafTransferFisiDetay.RafTransferFisiID = RafTransferFisiListesi.RafTransferFisiId
  left join RafDetay on RafDetay.RafDetayID = RafTransferFisiDetay.RafDetayId
  left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = HamStokDetay.HamStokIrsaliyeId
  left join Ham on Ham.HamId = RafDetay.HamId
  Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = HamStokMaster.BoyaSiparisKayitId
  left join Firma F1 on F1.FirmaId = HamStokMaster.FirmaId
  left join Firma F2 on F2.FirmaId = BoyaSiparis.FirmaId
  where RafTransferFisiDetay.RafTransferFisiDetayId is not null
  into :RafTransferFisiId, :FisNoStr, :FisTarihi, :RafTransferFisiDetayId, :HamAdi,  :HamStokDetayId, :HamStokMasterId, :RafDetayId, :EskiRafId, :YeniRafId,
       :IrsaliyeNo, :IrsaliyeTarihi, :GirisEvrakTipi, :Lot, :KapAdedi, :Kilo, :Metre, :KaliteNo, :PaletNo,
       :Pus, :Feine, :En, :OrmeciMakinaNo, :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3, :Grm2, :FTupAcikEnMayli, :FirmaAdi, :SiparisFirmaAdi do
  begin
    Select TupAcikEnMayliStr From TupAcikEnMayListrGetir(:FTupAcikEnMayli) Into :TupAcikEnMayliStr;
    suspend;
  end
end
