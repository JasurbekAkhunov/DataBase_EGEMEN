CREATE PROCEDURE ORMEANLIKIPLIKSTOKDURUMUSTOK(
    AHAMGRUP1 VARCHAR(4))
RETURNS (

    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    HAMGRUP1 VARCHAR(4),
    TOPLAMMIKTAR NUMERIC(12,3))
AS
declare variable hstIplik integer;
begin
  Select hstIplik from HamTipiGetir into :hstIplik;

  For Select OrmeRaf.HamId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Ham.HamGrup1, Sum(OrmeRaf.NetGirisMiktari-OrmeRaf.NetCikisMiktari) from OrmeRaf
    left join Ham on Ham.HamID = OrmeRaf.HamId
    where Ham.HamTipiId = :hstIplik and (Ham.HamGrup1 = :AHamGrup1 or :AHamGrup1 is null)
    group by OrmeRaf.HamId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Ham.HamGrup1
    into :HamId, :HamAdi, :HamAdi2, :HamAdi3, :HamAdi4, :HamGrup1, :ToplamMiktar do
  begin
    if (ToplamMiktar <> 0) then suspend;
  end
end
