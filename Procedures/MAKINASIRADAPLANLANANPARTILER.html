CREATE PROCEDURE MAKINASIRADAPLANLANANPARTILER(
    AMAKINAID VARCHAR(10))
RETURNS (
    PLANLAMATIPI INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    TOPLAMKILO NUMERIC(12,3),
    TOPLAMMETRE NUMERIC(12,3),
    REZERVNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable SuAnkiSepet integer;
declare variable SiradakiSepetNo integer;
declare variable SonrakiSepetNo integer;
declare variable AsilMakinaID      MakinaKoduSize;
begin
  Select AsilMakinaID from Makina where Makina.MakinaId = :AMakinaId into :AsilMakinaID;
  if (AsilMakinaID is Not Null) then AMakinaID = :AsilMakinaID;

  Select Max(UretimPlanlama.SepetNo) from UretimPlanlama
    left join UretimPlanlamaDetay on UretimPlanlamaDetay.UretimPlanlamaId = UretimPlanlama.UretimPlanlamaId
    left join BolumPlanlama on BolumPlanlama.BolumPlanlamaId = UretimPlanlamaDetay.BolumPlanlamaId
    Left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = BolumPlanlama.PartiAsamalariId
    where UretimPlanlama.MakinaId = :AMakinaId and PartiAsamaStok.GirisTarihi is not null
    into :SuAnkiSepet;

  if (SuAnkiSepet is null) then SuAnkiSepet = -1;

  Select Min(UretimPlanlama.SepetNo) from UretimPlanlama
    where UretimPlanlama.MakinaId = :AMakinaId and UretimPlanlama.SepetNo > :SuAnkiSepet
    into :SiradakiSepetNo;

  if (SiradakiSepetNo is null) then exit;
  else
  begin
    For Select min(UretimPlanlamaDetay.PlanlamaTipi), Parti.PartiYil, Parti.PartiId, Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from UretimPlanlama
      left join UretimPlanlamaDetay on UretimPlanlamaDetay.UretimPlanlamaId = UretimPlanlama.UretimPlanlamaId
      left join BolumPlanlama on BolumPlanlama.BolumPlanlamaId = UretimPlanlamaDetay.BolumPlanlamaId
      left join PartiStok on PartiStok.PartiSinifId = BolumPlanlama.PartiSinifId
      left join Parti on Parti.PartiKayitId = BolumPlanlama.PartiKayitId
      where UretimPlanlama.MakinaId = :AMakinaId and UretimPlanlama.SepetNo = :SiradakiSepetNo
      group by Parti.PartiYil, Parti.PartiId
      into :PlanlamaTipi, :PartiYil, :PartiId, :ToplamKilo, ToplamMetre do
    begin
      if (PlanlamaTipi = 2) then
      begin
        Select UretimPlanlamaDetay.RezervNotu from UretimPlanlama
         left join UretimPlanlamaDetay on UretimPlanlamaDetay.UretimPlanlamaId = UretimPlanlama.UretimPlanlamaId
         where UretimPlanlama.MakinaId = :AMakinaId and UretimPlanlama.SepetNo = :SiradakiSepetNo
         into :RezervNotu;
      end
      suspend;
    end

    RezervNotu = Null;
    if (PlanlamaTipi <> 0) then
    begin
      Select Min(UretimPlanlama.SepetNo) from UretimPlanlama
        where UretimPlanlama.MakinaId = :AMakinaId and UretimPlanlama.SepetNo > :SiradakiSepetNo
        into :SonrakiSepetNo;

      if (SonrakiSepetNo is null) then exit;
      else
      begin
        For Select min(UretimPlanlamaDetay.PlanlamaTipi), Parti.PartiYil, Parti.PartiId, Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from UretimPlanlama
          left join UretimPlanlamaDetay on UretimPlanlamaDetay.UretimPlanlamaId = UretimPlanlama.UretimPlanlamaId
          left join BolumPlanlama on BolumPlanlama.BolumPlanlamaId = UretimPlanlamaDetay.BolumPlanlamaId
          left join PartiStok on PartiStok.PartiSinifId = BolumPlanlama.PartiSinifId
          left join Parti on Parti.PartiKayitId = BolumPlanlama.PartiKayitId
          where UretimPlanlama.MakinaId = :AMakinaId and UretimPlanlama.SepetNo = :SonrakiSepetNo
          group by Parti.PartiYil, Parti.PartiId
          into :PlanlamaTipi, :PartiYil, :PartiId, :ToplamKilo, ToplamMetre do
        begin
          if (PlanlamaTipi = 2) then
          begin
            Select UretimPlanlamaDetay.RezervNotu from UretimPlanlama
             left join UretimPlanlamaDetay on UretimPlanlamaDetay.UretimPlanlamaId = UretimPlanlama.UretimPlanlamaId
             where UretimPlanlama.MakinaId = :AMakinaId and UretimPlanlama.SepetNo = :SonrakiSepetNo
             into :RezervNotu;
          end
          suspend;
        end
      end
    end
  end
end
