CREATE PROCEDURE TOPLUORMETALIMATIOLUSTUR(
    AFIRMAID VARCHAR(20),
    AORMESIPARISDETAYIDSTR VARCHAR(1025),
    AKULLANICIID INTEGER)
AS
declare variable VirgulPos integer;
declare variable Str varchar(20);
declare variable OrmeSiparisDetayId integer;
declare variable BoyaSiparisKayitId integer;
declare variable FPlanlanacakMiktar Numeric(12,3);
declare variable FBirimFiyat    Double Precision;
declare variable FDovizId       varchar(5);
declare variable FVadeGun       integer;
declare variable FBeyazlik      integer;
declare variable DokumaSiparis  integer;
begin
  VirgulPos = PosBul(AOrmeSiparisDetayIdStr, ',');

  While (VirgulPos > 1) do
  begin
    Str = EgemenCopyStr(AOrmeSiparisDetayIdStr, 1, VirgulPos -1);
    OrmeSiparisDetayId = EgemenIntegerToStr(Str);

    Select OrmeSiparisDetay.BoyaSiparisKayitId, BoyaSiparis.DokumaSiparisi, (OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.PlanlananMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar),
           OrmeSiparisDetay.OrmeBirimMaliyet, OrmeSiparisDetay.OrmeBirimMaliyetDovizId, OrmeSiparisDetay.VadeGun,
           OrmeSiparisDetay.Beyazlik from OrmeSiparisDetay
     Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = OrmeSiparisDetay.BoyaSiparisKayitID
     where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId
     into :BoyaSiparisKayitId, :DokumaSiparis, :FPlanlanacakMiktar, :FBirimFiyat, :FDovizId, :FVadeGun, :FBeyazlik;

    if (FPlanlanacakMiktar > 0) then
    begin
      if (DokumaSiparis = 1) then
        insert into DokumaPlanlama(Tarih, FirmaId, Sira, BoyaSiparisKayitId, OrmeSiparisDetayId, Beyazlik, PlanlananMiktar,
                               BirimFiyat, DovizId, VadeGun, MiktarGun, InsertKullaniciId, KullaniciId)
                        values(Current_date, :AFirmaId, -1, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :FBeyazlik, :FPlanlanacakMiktar,
                               :FBirimFiyat, :FDovizId, :FVadeGun, 0, :AKullaniciId, :AKullaniciId);
      else
        insert into OrmePlanlama(Tarih, FirmaId, Sira, BoyaSiparisKayitId, OrmeSiparisDetayId, Beyazlik, PlanlananMiktar,
                               BirimFiyat, DovizId, VadeGun, MiktarGun, InsertKullaniciId, KullaniciId)
                        values(Current_date, :AFirmaId, -1, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :FBeyazlik, :FPlanlanacakMiktar,
                               :FBirimFiyat, :FDovizId, :FVadeGun, 0, :AKullaniciId, :AKullaniciId);

    end
    if (Char_Length(AOrmeSiparisDetayIdStr) > VirgulPos) then AOrmeSiparisDetayIdStr = EgemenCopyStr(AOrmeSiparisDetayIdStr, VirgulPos + 1, Char_Length(AOrmeSiparisDetayIdStr) - VirgulPos);
    else AOrmeSiparisDetayIdStr = '';

    VirgulPos = PosBul(AOrmeSiparisDetayIdStr, ',');
  end
end
