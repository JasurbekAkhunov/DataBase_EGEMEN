CREATE PROCEDURE IMALATAARAVERMERAPORUGECMIS(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AARAVERMESEBEBIID INTEGER,
    AMAKINAGRUPID VARCHAR(4),
    AMAKINAID VARCHAR(10),
    AORMEMAKINAID VARCHAR(10))
RETURNS (
    IMALATAARAVERMEID INTEGER,
    MAKINAID VARCHAR(10),
    ARAVERMESEBEBIID INTEGER,
    ARAVERMESEBEBIADI VARCHAR(50),
    ARAVERMEGRUPID VARCHAR(5),
    ARAVERMEGRUPADI VARCHAR(20),
    BASLADIMAKINADURUMUID INTEGER,
    ARAVERDIRENPERSONELID INTEGER,
    ARAVERDIRENPERSONELADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITTIMAKINADURUMUID INTEGER,
    ARAYIBITIRENPERSONELID INTEGER,
    ARAYIBITIRENPERSONELADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30))
AS
begin
  for select ImalataAraVerme.ImalataAraVermeId, MK.MakinaId, ImalataAraVerme.AraVermeSebebiId, AraVermeSebebi.Adi AraVermeSebebiAdi,
             Imalataaraverme.BasladiMakinaDurumuId, ImalataAraVerme.AraVerdirenPersonelId, Bs.Adi AraVerdirenPersonelAdi, MK.Tarih BaslamaTarihi,
             ImalataAraVerme.BittiMakinaDurumuId, ImalataAraVerme.ArayiBitirenPersonelId, bt.adi ArayiBitirenPersonelAdi, Bmk.tarih Bitistarihi, AraVermeGrup.AraVermeGrupID, AraVermeGrup.Adi from MakinaDurumu MK
  left join Imalataaraverme on Imalataaraverme.BasladiMakinaDurumuId = MK.MakinaDurumuId
  left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
  left join AraVermeGrup on AraVermeGrup.AraVermeGrupId = AraVermeSebebi.AraVermeGrupId
  left join Makina on Makina.MakinaId = MK.MakinaId
  left join Personel bs on bs.PersonelId = ImalataAraVerme.AraVerdirenPersonelId
  left join Personel bt on bt.PersonelId = ImalataAraVerme.ArayiBitirenPersonelId
  left join MakinaDurumu BMK on BMK.MakinaDurumuId = ImalataAraVerme.bittiMakinaDurumuId
  where MK.tarih >= :ABaslangicTarihi and MK.tarih <= :ABitisTarihi and MK.ImalataAraVerildi = 1 and
       (Imalataaraverme.AraVermeSebebiId = :AAraVermeSebebiId or :AAraVermeSebebiId is null) and
       (Mk.MakinaId = :AMakinaId or :AMakinaId is null) and
       (Makina.MakinaGrubu = :AMakinaGrupId or :AMakinaGrupId is null)
  into :ImalataAraVermeId, :MakinaId, :AraVermeSebebiId, :AraVermeSebebiAdi,
       :BasladiMakinaDurumuId, :AraVerdirenPersonelId, :AraVerdirenPersonelAdi, :BaslamaTarihi,
       :BittiMakinaDurumuId, :ArayiBitirenPersonelId, :ArayiBitirenPersonelAdi, :BitisTarihi, :AraVermeGrupID, :AraVermeGrupAdi do
  begin
    if (BitisTarihi is not null) then GecenSure = BitisTarihi - BaslamaTarihi;
    else GecenSure = (CURRENT_DATE + CURRENT_TIME) - BaslamaTarihi;

    GecenSureStr = TarihToStr(GecenSure);
    suspend;
  end
  for select ImalataAraVerme.ImalataAraVermeId, ImalataAraVerme.OrmeMakinaId, ImalataAraVerme.AraVermeSebebiId, AraVermeSebebi.Adi AraVermeSebebiAdi,
             Null, ImalataAraVerme.AraVerdirenPersonelId, Bs.Adi AraVerdirenPersonelAdi, ImalataAraVerme.BaslamaTarihi,
             Null, ImalataAraVerme.ArayiBitirenPersonelId, Bt.adi ArayiBitirenPersonelAdi, ImalataAraVerme.BitisTarihi, AraVermeGrup.AraVermeGrupID, AraVermeGrup.Adi from ImalataAraVerme
  left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
  left join AraVermeGrup on AraVermeGrup.AraVermeGrupId = AraVermeSebebi.AraVermeGrupId
  left join OrmeMakina on OrmeMakina.OrmeMakinaId = ImalataAraVerme.OrmeMakinaId
  left join Personel bs on bs.PersonelId = ImalataAraVerme.AraVerdirenPersonelId
  left join Personel bt on bt.PersonelId = ImalataAraVerme.ArayiBitirenPersonelId
  where ImalataAraVerme.BaslamaTarihi >= :ABaslangicTarihi and ImalataAraVerme.BaslamaTarihi <= :ABitisTarihi and
       (Imalataaraverme.AraVermeSebebiId = :AAraVermeSebebiId or :AAraVermeSebebiId is null) and
       (ImalataAraVerme.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
       (OrmeMakina.Makinagrupid = :AMakinaGrupId or :AMakinaGrupId is null)
  into :ImalataAraVermeId, :MakinaId, :AraVermeSebebiId, :AraVermeSebebiAdi,
       :BasladiMakinaDurumuId, :AraVerdirenPersonelId, :AraVerdirenPersonelAdi, :BaslamaTarihi,
       :BittiMakinaDurumuId, :ArayiBitirenPersonelId, :ArayiBitirenPersonelAdi, :BitisTarihi, :AraVermeGrupID, :AraVermeGrupAdi do
  begin
    if (BitisTarihi is not null) then GecenSure = BitisTarihi - BaslamaTarihi;
    else GecenSure = (CURRENT_DATE + CURRENT_TIME) - BaslamaTarihi;

    GecenSureStr = TarihToStr(GecenSure);
    suspend;
  end
end
