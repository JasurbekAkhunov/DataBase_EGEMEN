CREATE PROCEDURE HGSIPARISRAPORU(
    ABASLANGICSIPARISYIL INTEGER,
    ABASLANGICSIPARISID INTEGER,
    ABITISSIPARISYIL INTEGER,
    ABITISSIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASINIF VARCHAR(1),
    AMODELID MODELKODUSIZE,
    ASIPONAYDURUMU SMALLINT,
    AMALIONAYDURUMU SMALLINT)
RETURNS (
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER,
    HGSIPARISIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAYETKILIID INTEGER,
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    HGSIPARISKAYITID INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISIVERENADI ADI50,
    SIPARISTARIHI TIMESTAMP,
    SOZLESMETARIHI TIMESTAMP,
    NOTU VARCHAR(1024),
    YURTDISI INTEGER,
    YURTDISISTR VARCHAR(9),
    FASON SMALLINT,
    FASONSTR VARCHAR(5),
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI ADI40,
    TESLIMSEKLIID INTEGER,
    TESLIMSEKLIADI ADI40,
    BASKAFIRMAYASATILABILIR EGEBOOLEAN,
    BASKAFIRMAYASATILABILIRSTR VARCHAR(5),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
Declare Variable ModelKayitId Integer;
begin
  Select ModelKayitId From Model Where Model.ModelID = :AModelID into :ModelKayitId;

  for select Hgsiparis.Hgsipariskayitid, Hgsiparis.Hgsiparisyil, Hgsiparis.Hgsiparisid, Hgsiparis.Firmaid, Firma.Adi,
                Hgsiparis.Siparistarihi, Hgsiparis.Sozlesmetarihi, Hgsiparis.Firmayetkiliid, Hgsiparis.Siparisialankullaniciid, SAK.Adi,
                Hgsiparis.Siparisgrupid, SiparisGrup.Adi, Hgsiparis.Siparisgrup2id, SiparisGrup2.Adi, FirmaYetkili.Adi, Hgsiparis.Fason, Hgsiparis.Odemesekliid, OdemeSekli.Adi,
                Hgsiparis.Teslimsekliid, TeslimSekli.Adi, Hgsiparis.Yurtdisi, Hgsiparis.Notu, Hgsiparis.Baskafirmayasatilabilir,
                IK.Adi, Hgsiparis.Inserttarihi, SK.Adi, Hgsiparis.Degisimtarihi
      from HgSiparis
      Left Join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
      left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = HgSiparis.FirmaYetkiliId
      Left Join Firma On Firma.FirmaId = HgSiparis.FirmaId
      left join OdemeSekli on OdemeSekli.OdemeSekliId = HgSiparis.OdemeSekliId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = HgSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = HgSiparis.SiparisGrup2Id
      Left Join TeslimSekli on TeslimSekli.TeslimSekliId = HgSiparis.TeslimSekliId
      Left Join Kullanici SAK on SAK.KullaniciId = HgSiparis.SiparisiAlanKullaniciId
      Left Join Kullanici IK  ON IK.KullaniciId = HgSiparis.InsertKullaniciID
      Left Join Kullanici SK  ON SK.KullaniciId = HgSiparis.KullaniciID
      where (:ARaporTipi = 1 and HgSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and HgSiparis.SiparisTarihi <= :ABitisSiparisTarihi or
             :ARaporTipi = 2 and HgSiparisDetay.TeslimTarihi >= :ABaslangicSiparisTarihi and HgSiparisDetay.TeslimTarihi <= :ABitisSiparisTarihi or
             :ARaporTipi = 0 and
                (HgSiparis.HgSiparisYil > :ABaslangicSiparisYil or (HgSiparis.HgSiparisYil = :ABaslangicSiparisYil and HgSiparis.HgSiparisId >= :ABaslangicSiparisId)) and
                (HgSiparis.HgSiparisYil < :ABitisSiparisYil or (HgSiparis.HgSiparisYil = :ABitisSiparisYil  and HgSiparis.HgSiparisId <= :ABitisSiparisId))
             ) and
            (Firma.FirmaId = :AFirmaId or :AFirmaId is null) and
            (HgSiparisDetay.Kapali = :AKapali or :AKapali is null) and
            (HgSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
            (HgSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
            (HgSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
            (HgSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null) and
            (HgSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
            (HgSiparisDetay.Sinif = :ASinif or :ASinif is null) and
            (HgSiparisDetay.ModelKayitId = :ModelKayitId or :ModelKayitId is Null) and
            (HgSiparisDetay.OnayDurumu = :ASipOnayDurumu or :ASipOnayDurumu is Null) and
            (HgSiparisDetay.MaliOnayDurumu = :AMaliOnayDurumu or :AMaliOnayDurumu is Null) and
            (HgSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null)
      group by Hgsiparis.Hgsipariskayitid, Hgsiparis.Hgsiparisyil, Hgsiparis.Hgsiparisid, Hgsiparis.Firmaid, Firma.Adi,
                Hgsiparis.Siparistarihi, Hgsiparis.Sozlesmetarihi, Hgsiparis.Firmayetkiliid, Hgsiparis.Siparisialankullaniciid, SAK.Adi,
                Hgsiparis.Siparisgrupid, SiparisGrup.Adi, Hgsiparis.Siparisgrup2id, SiparisGrup2.Adi, FirmaYetkili.Adi, Hgsiparis.Fason, Hgsiparis.Odemesekliid, OdemeSekli.Adi,
                Hgsiparis.Teslimsekliid, TeslimSekli.Adi, Hgsiparis.Yurtdisi, Hgsiparis.Notu, Hgsiparis.Baskafirmayasatilabilir,
                IK.Adi, Hgsiparis.Inserttarihi, SK.Adi, Hgsiparis.Degisimtarihi into
    :HgSiparisKayitId, :HgSiparisYil, :HgSiparisId, :FirmaId, :FirmaAdi,
    :SiparisTarihi, :SozlesmeTarihi, :FirmaYetkiliId, :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi,
    :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisiVerenAdi, :Fason, :OdemeSekliId, :OdemeSekliAdi,
    :TeslimSekliId, :TeslimSekliAdi, :YurtDisi, :Notu, :BaskaFirmayaSatilabilir,
    :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    if (YurtDisi = 1) then YurtDisiStr = 'Yurt Dışı';
    else YurtDisiStr = '';

    if (Fason = 1) then FasonStr = 'Evet';
    else FasonStr = 'Hayır';

    if (BaskaFirmayaSatilabilir = 1) then BaskaFirmayaSatilabilirStr = 'Evet';
    else BaskaFirmayaSatilabilirStr = 'Hayır';

    HgSiparisIdStr = '(' || EgemenIntegerToStr(HgSiparisYil) || ') '  || EgemenIntegerToStr(HgSiparisId);

    suspend;
  end
end
