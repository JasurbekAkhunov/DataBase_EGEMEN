CREATE PROCEDURE MODELKUMASKOPYALA(
    AMODELKUMASID INTEGER,
    AYENIHAMID INTEGER,
    AYENIRENKID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable YeniModelKumasId integer;
declare variable FGarniModelKumasId integer;
declare variable YeniGarniModelKumasId integer;
declare variable FYeniRenkAdi Type of Column Renk.Adi;
declare variable FRenkId integer;
begin
  Select Adi from Renk where Renk.RenkId = :AYeniRenkId into :FYeniRenkAdi;
  Select RenkId from ModelKumas where ModelKumas.ModelKumasId = :AModelKumasId into :FRenkId;


  YeniModelKumasId = gen_ID(ModelKumas_Id_gen, 1);
  Insert into ModelKumas(ModelKumasID, ModelKayitId, HamId, Miktar, FireOrani, LabReceteId, LabReceteKodu, LabReceteAtisId, RenkId, BirimFiyat, BirimFiyatDovizId,
                         AnaKumas, AnaKumasModelKumasID, PantoneKodu, RenkAdi, TupAcikEnMayli, BoyaliGramaj, BoyaliEn, Notu, InsertKullaniciId, KullaniciId, TedarikciRenkKodu)
      Select :YeniModelKumasId, ModelKayitId, :AYeniHamId, Miktar, FireOrani, LabReceteId, LabReceteKodu, LabReceteAtisId, :AYeniRenkId, BirimFiyat, BirimFiyatDovizId,
             AnaKumas, AnaKumasModelKumasID, PantoneKodu, :FYeniRenkAdi, TupAcikEnMayli, BoyaliGramaj, BoyaliEn, Notu, :AKullaniciId, :AKullaniciId, TedarikciRenkKodu from ModelKumas where ModelKumas.ModelKumasId = :AModelKumasId;

  Insert into ModelParca(ModelKayitId, ModelKumasID, ParcaId, InsertKullaniciId, KullaniciId)
      Select ModelKayitId, :YeniModelKumasId, ParcaId, :AKullaniciId, :AKullaniciId from ModelParca where ModelParca.ModelKumasId = :AModelKumasId;

  Insert into ModelAksesuar(ModelKayitId, ModelKumasID, UrunID, Miktar, FireOrani, BirimFiyat, BirimFiyatDovizId, Notu, KullanimTipi, BirimId, InsertKullaniciId, KullaniciId)
      Select ModelKayitId, :YeniModelKumasId, UrunID, Miktar, FireOrani, BirimFiyat, BirimFiyatDovizId, Notu, KullanimTipi, BirimId, :AKullaniciId, :AKullaniciId from ModelAksesuar
      where ModelAksesuar.ModelKumasId = :AModelKumasId and ModelAksesuar.ModelKumasId is not Null;

  /*Garni Kumaslar*/
  For Select ModelKumasId from ModelKumas where AnaKumas = 0 and ModelKumas.AnaKumasModelKumasId = :AModelKumasId into :FGarniModelKumasId do
  begin
    YeniGarniModelKumasId = gen_ID(ModelKumas_Id_gen, 1);
    /*iif(:FRenkId = RenkId, :FYeniRenkAdi, RenkAdi) bu koşul kaynak kumas ile garni kumasın rengi aynıydıysa yeni rrenkleride aynı yapmak için eklendi.*/
    Insert into ModelKumas(ModelKumasID, ModelKayitId, HamId, Miktar, FireOrani, LabReceteId, LabReceteKodu, LabReceteAtisId, RenkId, BirimFiyat, BirimFiyatDovizId,
                           AnaKumas, AnaKumasModelKumasID, PantoneKodu, RenkAdi, TupAcikEnMayli, BoyaliGramaj, BoyaliEn, Notu, InsertKullaniciId, KullaniciId, TedarikciRenkKodu)
        Select :YeniGarniModelKumasId, ModelKayitId, HamId, Miktar, FireOrani, LabReceteId, LabReceteKodu, LabReceteAtisId, iif(:FRenkId = RenkId, :AYeniRenkId, RenkId), BirimFiyat, BirimFiyatDovizId,
               AnaKumas, :YeniModelKumasId, PantoneKodu, iif(:FRenkId = RenkId, :FYeniRenkAdi, RenkAdi), TupAcikEnMayli, BoyaliGramaj, BoyaliEn, Notu, :AKullaniciId, :AKullaniciId, TedarikciRenkKodu from ModelKumas
               where ModelKumas.ModelKumasId = :FGarniModelKumasId;

    Insert into ModelParca(ModelKayitId, ModelKumasID, ParcaId, InsertKullaniciId, KullaniciId)
        Select ModelKayitId, :YeniGarniModelKumasId, ParcaId, :AKullaniciId, :AKullaniciId from ModelParca where ModelParca.ModelKumasId = :FGarniModelKumasId;
  end
    /*Garni Kumaslar*/
end
