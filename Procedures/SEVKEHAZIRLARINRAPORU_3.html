             Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and PSKaliteKontrol.KaliteNo = 3 and PartistokTarti.HamStokSarfFisiDetayId is null, PSKaliteKontrol.Kilo - PSKaliteKontrol.Dara, null)),
             Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and PSKaliteKontrol.KaliteNo = 3 and PartistokTarti.HamStokSarfFisiDetayId is null, PSKaliteKontrol.Kilo, null)),
             Sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and PSKaliteKontrol.KaliteNo = 3 and PartistokTarti.HamStokSarfFisiDetayId is null, PSKaliteKontrol.Metre, null)),

    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId

    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join FirmaGrup FG1 on FG1.FirmaGrupId = Firma.FirmaGrup1
    left join FirmaGrup FG2 on FG2.FirmaGrupId = Firma.FirmaGrup2
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
    left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    left join IplikCinsi On IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    Left Join Unite on Unite.UniteId = PartiStok.UniteId
    left join DesenVaryant on DesenVaryant.DesenVaryantID = Parti.DesenVaryantID
    left join Desen on Desen.DesenID = DesenVaryant.DesenID
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabREceteId
    left join PartiStokTarti on PartiStokTarti.PartiSinifID = PartiStok.PartiSinifId
    Left Join PSKaliteKontrol on PSKaliteKontrol.PartiStokTartiID = PartiStokTarti.PartiStokTartiID
    left join SiparisProses on SiparisProses.SiparisProsesid = BoyaSiparisDetay.SiparisProsesid
    where PartiStok.durumu = 7 and PartiStok.PartiStokIptalKayitId is null and (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
        (PartiStok.HamID = :AHamID or :AHamID is null) and
        (Parti.Fason = :AFason or :AFason is null) and
         ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
           ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 1) or
           (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 2) or
           (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf  and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 3) or :AHamTipiId is null) and
        (Firma.TemsilciID = :ATemsilciId or :ATemsilciId is null) and
        (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
        (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
        (BoyaSiparisDetay.KaliteID = :AKaliteId or :AKaliteId is Null) and
        (BoyaSiparisDetay.LabReceteId = :ALabReceteId or :ALabReceteId is Null) and
        (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null) and
        (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is Null) and
        (IplikCinsi.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is Null) and
        ( (PartiStok.UniteID is null and :ADisUniteDurumu = 0) or
          (PartiStok.UniteID is not null and :ADisUniteDurumu = 1) or :ADisUniteDurumu is null) and
        (parti.PartiGrupId = :APartiGrupId or :APartiGrupId is Null) and
        (BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
        (DesenVaryant.DesenId = :ADesenId or :ADesenId is null ) and
        (PartiStok.UniteId = :AUniteId or :AUniteId is Null )
    group by
    PartiStok.PartiKayitId, PartiStok.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.NetKilo,
             PartiStok.BrutSevkKilo, PartiStok.NetSevkKilo, PartiStok.Metre, PartiStok.NetMetre, PartiStok.BrutSevkMetre, PartiStok.NetSevkMetre,
             Parti.BoyaSiparisKayitId, PartiStok.BoyaSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.MusteriSiparisNo,
             BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
             PartiStok.FirmaSevkAdresiId, FirmaSevkAdresi.AdresAdi, PartiStok.BirimFiyat, PartiStok.DovizId,
             PartiStokFizikKontrol.EnVerilen, PartiStokFizikKontrol.GramajVerilen,
             BoyaSiparisDetay.KaliteId, Kalite.KaliteAdi, BoyaSiparisDetay.SiparisKilo, PartiStok.UniteId, Unite.Adi, BoyaSiparisDetay.BelgeHamAdi, OrmeSiparisDetay.OrmeBirimMaliyet,
             OrmeSiparisDetay.OrmeBirimMaliyetDovizId, BoyaSiparisDetay.FasonBirimFiyati, BoyaSiparisDetay.FasonBirimFiyatDovizId, DesenVaryant.DesenID, Desen.Adi, DesenVaryant.VaryantAdi,
             PartiStok.RezervKilo, PartiStok.BrutRezervKilo, PartiStok.RezervMiktar, PartiStok.BrutRezervMiktar,
             FG1.FirmaGrupId, FG1.Adi, FG2.FirmaGrupId, FG2.Adi, Firma.FirmaId, LabRecete.Aciklama, Ham.Adi2, Ham.ReferansKodu, PartiStok.KKBitisTarihi,
             BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.KKDeger2, SiparisProses.Adi,
             BoyaSiparisDetay.MelanjKodu
    into :PartiKayitId, :PartiSinifId, :HamAdiToplami, :Kilo, :NetKilo, :SevkEdilenKilo, :SevkEdilenNetKilo, :SevkeHazirKilo, :SevkeHazirNetKilo,
         :Metre, :NetMetre, :SevkEdilenMetre, :SevkEdilenNetMetre, :SevkeHazirMetre,
         :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :MusteriSiparisNo,
         :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
         :FirmaSevkAdresiId, :FirmaSevkAdresiAdi, :BirimFiyat, :DovizId, :EnVerilen, :GramajVerilen,
         :KaliteId, :KaliteAdi, :SiparisDetayKilo, :UniteId, :UniteAdi, :BelgeHamAdi, :OrmeBirimMaliyet, :OrmeBirimMaliyetDovizId,
         :FasonBirimFiyati, :FasonBirimFiyatDovizId, :DesenID, :DesenAdi, :VaryantAdi, :RezervKilo, :BrutRezervKilo, :RezervMiktar, :BrutRezervMiktar,
         :FirmaGrup1Id, :FirmaGrup1Adi, :FirmaGrup2Id, :FirmaGrup2Adi, :FirmaId, :LabReceteAciklama, :HamAdi2,
         :SevkeHazirKapAdedi, :SevkeHazirBobinSayisi, :SevkeHazirTartiBrut, :SevkTartiBrut, :TartiBrut,
         :KKBirinciKaliteKilo, :KKBirinciKaliteBrutKilo, :KKBirinciKaliteMetre, :KKBirinciKaliteTopSayisi,
         :KKIkinciKaliteKilo, :KKIkinciKaliteBrutKilo, :KKIkinciKaliteMetre, :KKIkinciKaliteTopSayisi,
         :KKUcuncuKaliteKilo, :KKUcuncuKaliteBrutKilo, :KKUcuncuKaliteMetre, :KKUcuncuKaliteTopSayisi,
         :KKDigerKaliteKilo, :KKDigerKaliteBrutKilo, :KKDigerKaliteMetre, :KKDigerKaliteTopSayisi, :ReferansKodu, :KKBitisTarihi, :EnIstenen, :GramajIstenen, :KKDeger2,
         :SevkEdilenKapAdedi, :SevkEdilenAdet, :Adet, :KalanAdet, :SiparisProsesAdi, :MelanjKodu do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    select max(PartiAsamalari.Sira) from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    where PartiAsamaStok.PartiKayitId = :PartiKayitId and PartiAsamaStok.PartiSinifId = :PartiSinifId and
         (PartiAsamalari.AsamaId <> :SonKontrolAsamaId and PartiAsamalari.AsamaId <> :IplikSonKontrolAsamaId ) and
          PartiAsamaStok.Giristarihi is not null
    into :MaxSira;

    SonMakinaId        = '';
    SonYapilanAsamaAdi = Null;

    if(MaxSira is not null) then
    begin
      For select Asama.Adi, PartiAsamaStokDetay.MakinaId from PartiAsamalari
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join Recete on Recete.Receteid = PartiAsamalari.Receteid
      left join Asama on (Asama.Asamaid = PartiAsamalari.AsamaId or Asama.Asamaid = Recete.UstAsamaId)
      where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.sira = :MaxSira and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
      group By Asama.Adi, PartiAsamaStokDetay.MakinaId
      into :SonYapilanAsamaAdi, :FMakinaId do
      begin
        if (Char_Length(SonMakinaId) + Char_Length(FMakinaId) < 250) then
        begin
          if (SonMakinaId <> '') then SonMakinaId = SonMakinaId || ',';

          SonMakinaId = SonMakinaId || FMakinaId;
        end
      end
    end

    HamLot = '';
    for Select HamStokMaster.Lot from PartiStokDetay
        Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.Lot is not null and HamStokMaster.Lot <> ''
        group by HamStokMaster.Lot
        into :FLot do
    begin
      if (Char_Length(HamLot) + Char_Length(FLot) < 255) then
      begin
       if (HamLot <> '' ) then HamLot = HamLot || cast(', ' as varchar(2) character set UTF8);
       HamLot = HamLot || FLot;
      end
    end

    DevirPartiNo = '';
    for Select Distinct HamStokMaster.DevirPartiYil, HamStokMaster.DevirPartiId from PartiStokDetay
        Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.DevirPartiYil is not null and HamStokMaster.DevirPartiId  is not null
        into :FDevirPartiYil, :FDevirPartiID do
    begin
      FPartiNoStr = '(' || EgemenIntegerToStr(FDevirPartiYil)  || ') ' || EgemenIntegerToStr(FDevirPartiID) || ' ';
      if (Char_Length(DevirPartiNo) + Char_Length(FPartiNoStr) < 253) then
      begin
        if (DevirPartiNo <> '' ) then DevirPartiNo = DevirPartiNo || cast(', ' as varchar(2) character set UTF8);
        DevirPartiNo = DevirPartiNo || FPartiNoStr;
      end
    end

    Select Sum(Kilo - Dara) NetKilo, Sum(Kilo) BrutKilo, Sum(Metre) Metre, Count(*) TopSayisi from PSKaliteKontrol
      left join PSKaliteKontrolAsama on PSKaliteKontrolAsama.PSKaliteKontrolAsamaId = PSKalitekontrol.PSKaliteKontrolAsamaId
      where PSKaliteKontrol.PartiSinifId = :PartiSinifId and PSKaliteKontrolAsama.SonKayit = 1
      into :KaliteKontrolKilo, :KaliteKontrolBrutKilo, :KaliteKontrolMetre, :KaliteKontrolTopSayisi;


    select first 1 HamStokMaliyeti, AsamaUMUzerindenAsamaMaliyeti, MakinaUMUzerindenAsamaMaliyeti, BoyaKimyasalMaliyeti,
         TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti from PartiSinifMaliyetKumulatif
         where PartiSinifId = :PartiSinifId
         into :HamStokMaliyeti, :AsamaUMUzerindenAsamaMaliyeti, :MakinaUMUzerindenAsamaMaliyeti, :BoyaKimyasalMaliyeti,
         :TeorikBoyaKimyasalMaliyeti, :UnitelerASevkMaliyeti;

    PartiSinifSevkRezervToplami = Null;
    Select Sum(PartiSinifSevkRezerv.Miktar) from PartiSinifSevkRezerv
      where PartiSinifSevkRezerv.PartiSinifId = :PartiSinifId
       into :PartiSinifSevkRezervToplami;

    suspend;
  end
