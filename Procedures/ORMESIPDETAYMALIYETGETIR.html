CREATE PROCEDURE ORMESIPDETAYMALIYETGETIR(
    AORMESIPARISDETAYID INTEGER,
    ASIPARISTARIHI TIMESTAMP)
RETURNS (
    HAMMALIYET DOUBLE PRECISION,
    IPLIKMALIYETI DOUBLE PRECISION,
    LYCRAMALIYETI DOUBLE PRECISION)
AS
declare variable DovizKuru  Double Precision;
declare variable hstLycra   integer;
declare variable OrmeBirimMaliyet              Double Precision;
declare variable OrmeBirimMaliyetDovizId       varchar(5);
declare variable TahminiFireOrani              Double Precision;
declare variable SistemDovizID varchar(5);
begin
  Select hstLycra from HamTipiGetir into :hstLycra;

  Select SistemDovizID from Sistem where SistemID = 1 into :SistemDovizID;

  Select TahminiFireOrani, OrmeBirimMaliyet, OrmeBirimMaliyetDovizID from OrmeSiparisDetay
    where OrmeSiparisDetay.OrmeSiparisDetayID = :AOrmeSiparisDetayID
    into :TahminiFireOrani, :OrmeBirimMaliyet, :OrmeBirimMaliyetDovizID;

  IplikMaliyeti = 0;
  LycraMaliyeti = 0;

  Select Sum(BirimMaliyet * (select DovizKuru from EnYakinDovizKuruGetir(BirimMaliyetDovizID, :SistemDovizID, :ASiparisTarihi)) * KullanimOrani/100) from OrmeSiparisDetayRecete
    left join Ham On Ham.HamID = OrmeSiparisDetayRecete.HamId
    where OrmeSiparisDetayRecete.OrmeSiparisDetayId = :AOrmeSiparisDetayId and Ham.HamTipiID <> :hstLycra
    into :IplikMaliyeti;

  if (IplikMaliyeti is null) then IplikMaliyeti = 0;
  else IplikMaliyeti = IplikMaliyeti + IplikMaliyeti * TahminiFireOrani / 100;

  Select Sum(BirimMaliyet * (select DovizKuru from EnYakinDovizKuruGetir(BirimMaliyetDovizID, :SistemDovizID, :ASiparisTarihi)) * KullanimOrani/100) from OrmeSiparisDetayRecete
    left join Ham On Ham.HamID = OrmeSiparisDetayRecete.HamId
    where OrmeSiparisDetayRecete.OrmeSiparisDetayId = :AOrmeSiparisDetayId and Ham.HamTipiID = :hstLycra
    into :LycraMaliyeti;

  if (OrmeBirimMaliyetDovizID <> SistemDovizId) then
    select DovizKuru from EnYakinDovizKuruGetir(:OrmeBirimMaliyetDovizID, :SistemDovizId, :ASiparisTarihi) into :DovizKuru;
  else
    DovizKuru = 1;

  HamMaliyet = IplikMaliyeti + LycraMaliyeti + OrmeBirimMaliyet * DovizKuru;
  suspend;
end
