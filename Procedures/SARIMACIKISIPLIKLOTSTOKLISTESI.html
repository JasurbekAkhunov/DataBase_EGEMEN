CREATE PROCEDURE SARIMACIKISIPLIKLOTSTOKLISTESI(
    AIPLIKLOTID INTEGER)
RETURNS (
    IPLIKSTOKID INTEGER,
    BOBINSAYISI INTEGER,
    AMBALAJSEKLIID INTEGER,
    AMBALAJSEKLIADI VARCHAR(10),
    KAPNO INTEGER,
    MIKTAR NUMERIC(12,3))
AS
declare variable Tamir integer;
declare variable Lot   varchar(30);
begin
  Select Tamir from IplikLot where IplikLot.IplikLotID = :AIplikLotID into :Tamir;

  if (Tamir = 0) then
  begin
    for Select IplikStok.IplikStokId, IplikStok.KapNo, IplikUretim.BobinSayisi, IplikStok.Miktar,
               Ambalaj.AmbalajSekliId, AmbalajSekli.Adi from IplikUretim
      left join IplikStok on IplikStok.IplikStokID = IplikUretim.IplikStokId and IplikStok.Numarator = IplikUretim.Numarator
      left join Ambalaj on Ambalaj.AmbalajID = IplikUretim.AmbalajId
      left join AmbalajSekli on AmbalajSekli.AmbalajSekliID = Ambalaj.AmbalajSekliId
      where IplikUretim.IplikLotId = :AIplikLotId and IplikUretim.SarimIsEmriKayitID is null and IplikStok.CikisIplikIrsaliyeDetayID is null
      into :IplikStokId, :KapNo, :BobinSayisi, :Miktar, :AmbalajSekliId, :AmbalajSekliAdi do
    begin
      Suspend;
    end

    for Select IplikSatisIade.YeniIplikStokId, IplikSatisIade.KapNo, IplikSatisIade.BobinSayisi, IplikSatisIade.Miktar,
               Ambalaj.AmbalajSekliId, AmbalajSekli.Adi from IplikSatisIade
      left join IplikStok on IplikStok.IplikStokID = IplikSatisIade.YeniIplikStokId and IplikStok.Numarator = IplikSatisIade.YeniNumarator
      left join Ambalaj on Ambalaj.AmbalajID = IplikSatisIade.AmbalajId
      left join AmbalajSekli on AmbalajSekli.AmbalajSekliID = Ambalaj.AmbalajSekliId
      where IplikSatisIade.IplikLotId = :AIplikLotId and IplikSatisIade.Tamir = 0 and IplikSatisIade.YeniIplikStokId is not null and
            IplikStok.CikisIplikIrsaliyeDetayID is null
      into :IplikStokId, :KapNo, :BobinSayisi, :Miktar, :AmbalajSekliId, :AmbalajSekliAdi do
    begin
      Suspend;
    end
  end
end
