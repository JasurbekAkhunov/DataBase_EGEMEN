CREATE PROCEDURE ORMEMALIYETLERIHESAPLA_IC(
    STOKHAREKETIDLER VARCHAR(1000),
    RAFTRANSFERIDLER VARCHAR(1000))
RETURNS (
    STOKHAREKETID INTEGER,
    ORMERAFTRANSFERFISIDETAYID INTEGER,
    BIRIMMALIYET DOUBLE PRECISION,
    NETBIRIMFIYAT DOUBLE PRECISION)
AS
declare variable DovizKuru         double precision;
declare variable uitBukum       integer;
declare variable uitOrme        integer;
declare variable uitBoya        integer;
declare variable IplikMaliyeti        double precision;
declare variable LycraMaliyeti   double precision;
declare variable HamMaliyeti   double precision;
declare variable LycraKullanimOrani  double precision;
declare variable HamTipiId        integer;
declare variable PartiSinifId        integer;
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable shtKaydirmaCikis        integer;
declare variable shtKaydirmaGiris        integer;
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstLycra        integer;
declare variable hstTelef        integer;
declare variable ReceteHamId        integer;
declare variable ReceteKullanimOrani  double precision;
declare variable ReceteMaliyet  double precision;
declare variable BoyaSiparisKayitId    integer;
Declare Variable OrmeSiparisDetayId    integer;
Declare Variable BoyaSiparisDetayID    integer;
Declare Variable OrmePlanlamaId        integer;
Declare Variable SHOrmeRafId           integer;
Declare Variable Tarih             TimeStamp;
Declare Variable HareketTipi       integer;
Declare Variable UretimIslemTipiId integer;
Declare Variable GirisCikis        integer;
Declare Variable HamId             integer;
Declare Variable RenkId            integer;
Declare Variable BirimFiyat        Double Precision;
Declare Variable Miktar            Double Precision;
Declare Variable TahminiFireOrani  Double Precision;
Declare Variable DovizId              varchar(5);
Declare Variable HamIade              integer;
Declare Variable BoyaIrsaliyeDetayId  integer;
Declare Variable FirmaId              varchar(20);
Declare Variable OrmeMakinaId         varchar(10);
Declare Variable OrmeFaturaId         Integer;
Declare Variable OrmeStokKaydirmaId   Integer;
Declare Variable EgemenFirmaId        Integer;
Declare Variable StokTabanDovizId     Varchar(5);
declare variable Sorgu  varchar(2000);
begin

  if (StokHareketIdler is not Null) then
  begin
    Select EgemenFirmaId, StokTabanDovizId from Sistem where SistemID = 1 into :EgemenFirmaId, :StokTabanDovizId;

    Sorgu = 'Select StokHareket.StokHareketId, StokHareket.OrmeSiparisDetayId, StokHareket.DovizId, StokHareket.Tarih, '
          ||' StokHareket.HareketTipi, StokHareket.UretimIslemTipiId, iif('||EgemenFirmaId||' = 9, StokHareket.BirimFiyat, StokHareket.NetBirimFiyat), StokHareket.HamIade, '
          ||' StokHareket.BoyaSiparisKayitId, StokHareket.BoyaIrsaliyeDetayId, StokHareket.HamId, StokHareket.RenkId, StokHareket.OrmePlanlamaId, StokHareket.BoyaSiparisDetayID, '
          ||' StokHareket.OrmeRafId, StokHareket.GirisCikis, StokHareket.Miktar, StokHareket.BirimFiyat, StokHareket.FirmaId, OrmeMakinaId, '
          ||' StokHareket.OrmeFaturaId, Coalesce(TahminiFireOrani, 0), OrmeFis.OrmeStokKaydirmaId from StokHareket '
          ||' left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId '
          ||' Left Join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = StokHareket.OrmeSiparisDetayID '
          ||' where StokHareket.StokHareketId in ('||StokHareketIdler||') order by StokHareket.Tarih, UretimIslemTipiId, Harekettipi, StokHareketId asc ';


    for execute statement(Sorgu) into :StokHareketId, :OrmeSiparisDetayId, :DovizID, :Tarih, :HareketTipi, :UretimIslemTipiId, :NetBirimFiyat, :HamIade,
           :BoyaSiparisKayitId, :BoyaIrsaliyeDetayId, :HamId, :RenkId, :OrmePlanlamaId, :BoyaSiparisDetayID,
           :SHOrmeRafId, :GirisCikis, :Miktar, :BirimFiyat, :FirmaId, :OrmeMakinaId,
           :OrmeFaturaId, :TahminiFireOrani, :OrmeStokKaydirmaId do
    begin
      select uitBukum, uitOrme, uitBoya from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya;

      select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
             ShtUretim, shtKaydirmaCikis, shtKaydirmaGiris from OrmeStokHareketTipiGetir
             into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
                  :shtFasonUretim, :ShtUretim, :shtKaydirmaCikis, :shtKaydirmaGiris;

      select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstLycra, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstLycra, :hstTelef;

      if (StokTabanDovizId <> DovizID) then
        select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :StokTabanDovizId, :Tarih) into :DovizKuru;
      else
        DovizKuru = 1;

      NetBirimFiyat = NetBirimFiyat * Dovizkuru ;
      BirimMaliyet = 0 ;
      IplikMaliyeti = 0 ;
      LycraMaliyeti = 0 ;
      LycraKullanimOrani = 0 ;
      HamMaliyeti = 0 ;

      if (GirisCikis = 1) then
      begin
        if (HareketTipi = shtKaydirmaGiris) then
        begin
          select StokHareket.BirimMaliyet from STOKHAREKET
          left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
          where OrmeFis.OrmeStokKaydirmaId = :OrmeStokKaydirmaId and StokHareket.HareketTipi = :shtKaydirmaCikis
          into :BirimFiyat;
          if (BirimFiyat is null) then BirimFiyat = 0;
          BirimMaliyet = BirimFiyat;
        end
        else if (UretimIslemTipiID is null) then
        begin
          BirimMaliyet = NetBirimFiyat;
        end
        else
        if (HamIade = 1) then
        begin
          if (BoyaSiparisKayitId is null) then
          begin
            BirimMaliyet = NetBirimFiyat;
          end
          else
          begin
            if (BoyaIrsaliyeDetayId is null) then
            begin
              select sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / sum(StokHareket.Miktar) from StokHareket
                where StokHareket.BoyaSiparisKayitId = :BoyaSiparisKayitId and
                      StokHareket.UretimIslemTipiID = :UretimIslemTipiID and
                      StokHareket.GirisCikis = -1 and
                      StokHareket.HamID = :HamId and
                      IntKarsilastir(StokHareket.RenkId, :RenkId) = 1 and
                      StokHareket.Miktar > 0
                into :BirimMaliyet;

              if (BirimMaliyet is null) then BirimMaliyet = 0;
            end
            else
            begin
              /*BirimMaliyet = NetBirimFiyat;*/
              select iif(:EgemenFirmaId = 9, HamStokMaster.BirimFiyat, HamStokMaster.NetBirimFiyat) from BoyaIrsaliyeDetay
                  left join RafDetay on RafDetay.RafDetayID = BoyaIrsaliyeDetay.RafDetayID
                  left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
                  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = HamStokDetay.HamStokIrsaliyeId
                  left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
                  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = HamStokMaster.BoyaSiparisDetayId
                  where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID
                  into :BirimFiyat;
              if (BirimFiyat is null) then BirimFiyat = 0;
              BirimMaliyet = BirimFiyat;
            end
          end
        end
        else
        begin
          if (BoyaSiparisKayitId is null) then
          begin
            if (UretimIslemTipiId = uitBukum) then
            begin
              BirimMaliyet = 0;
              for select HamId, KullanimOrani from HamStokRecete where HamId = :HamId into :ReceteHamId, :ReceteKullanimOrani do
              begin
                select sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / sum(StokHareket.Miktar) from StokHareket
                  where ((:FirmaId is not null and StokHareket.FirmaId = :FirmaId) or (:OrmeMakinaId is not null and StokHareket.OrmeMakinaId = :OrmeMakinaId)) and
                        StokHareket.HamId = :ReceteHamId and
                        StokHareket.UretimIslemTipiID = :UretimIslemTipiID and
                        StokHareket.GirisCikis = -1 and
                        StokHareket.BirimMaliyet > 0 and
                        StokHareket.Miktar > 0
                  into :ReceteMaliyet;

                if (ReceteMaliyet is null) then ReceteMaliyet = 0;
                if (ReceteKullanimOrani is null) then ReceteKullanimOrani = 0;
                BirimMaliyet = BirimMaliyet + (ReceteMaliyet * ReceteKullanimOrani / 100);
              end
              BirimMaliyet = BirimMaliyet + NetBirimFiyat;
            end
            else
            begin
              BirimMaliyet = NetBirimFiyat;
            end
          end
          else
          begin
            if (UretimIslemTipiId = uitBukum) then
            begin
              select sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / sum(StokHareket.Miktar) from StokHareket
                where StokHareket.BoyaSiparisKayitId = :BoyaSiparisKayitId and
            StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and
                      StokHareket.UretimIslemTipiID = :UretimIslemTipiID and
                      StokHareket.GirisCikis = -1 and
                      StokHareket.BirimMaliyet > 0 and
                      StokHareket.Miktar > 0
                into :BirimMaliyet;

              if (BirimMaliyet is null) then BirimMaliyet = 0;
              BirimMaliyet = BirimMaliyet + NetBirimFiyat;
            end
            else if (UretimIslemTipiId = uitOrme) then
            begin
              if (OrmePlanlamaId is not null) then
              begin
                select sum(OrmePlanlamaRecete.KullanimOrani) from OrmePlanlamaRecete
                left join Ham on Ham.HamID = OrmePlanlamaRecete.HamId
                where OrmePlanlamaRecete.OrmePlanlamaId = :OrmePlanlamaId and
                      Ham.HamTipiId = :hstLycra
                into :LycraKullanimOrani;

                select sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / sum(StokHareket.Miktar) from StokHareket
                  left join Ham on Ham.HamID = StokHareket.HamId
                  where StokHareket.OrmePlanlamaId = :OrmePlanlamaId and
                        StokHareket.UretimIslemTipiID = :UretimIslemTipiID and
                        Ham.HamTipiId = :hstIplik and
                        StokHareket.GirisCikis = -1 and
                        StokHareket.Miktar > 0
                  into :IplikMaliyeti;

                if (IplikMaliyeti is null) then IplikMaliyeti = 0;

                IplikMaliyeti = IplikMaliyeti + IplikMaliyeti * TahminiFireOrani / 100;

                select sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / sum(StokHareket.Miktar) from StokHareket
                  left join Ham on Ham.HamID = StokHareket.HamId
                  where StokHareket.OrmePlanlamaId = :OrmePlanlamaId and
                        StokHareket.UretimIslemTipiID = :UretimIslemTipiID and
                        Ham.HamTipiId = :hstLycra and
                        StokHareket.GirisCikis = -1 and
                        StokHareket.Miktar > 0
                  into :LycraMaliyeti;

                if (LycraMaliyeti is null) then LycraMaliyeti = 0;
                else LycraMaliyeti = LycraMaliyeti * LycraKullanimOrani / 100;

              end
              else
              begin
                LycraKullanimOrani = 0;

                select sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / sum(StokHareket.Miktar) from StokHareket
                  left join Ham on Ham.HamID = StokHareket.HamId
                  where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and
                        StokHareket.UretimIslemTipiID = :UretimIslemTipiID and
                        Ham.HamTipiId = :hstIplik and
                        StokHareket.GirisCikis = -1 and
                        StokHareket.Miktar > 0
                  into :IplikMaliyeti;

                if (IplikMaliyeti is null) then IplikMaliyeti = 0;

                IplikMaliyeti = IplikMaliyeti + IplikMaliyeti * TahminiFireOrani / 100;

                select sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / sum(StokHareket.Miktar) from StokHareket
                  left join Ham on Ham.HamID = StokHareket.HamId
                  where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and
                        StokHareket.UretimIslemTipiID = :UretimIslemTipiID and
                        Ham.HamTipiId = :hstLycra and
                        StokHareket.GirisCikis = -1 and
                        StokHareket.Miktar > 0
                  into :LycraMaliyeti;

                if (LycraMaliyeti is null) then LycraMaliyeti = 0;
                else LycraMaliyeti = LycraMaliyeti * LycraKullanimOrani / 100;
              end

              BirimMaliyet = IplikMaliyeti + LycraMaliyeti + NetBirimFiyat;
            end
            else if (UretimIslemTipiId = uitBoya) then
            begin
              if (BoyaSiparisDetayID is not null) then
              if (BoyaIrsaliyeDetayId is null) then
              begin
                select HamTipiId from Ham where Ham.HamID = :HamId into :HamTipiId;
                /* iplik boya ise ham iplik maliyeti, kumas boya ise ham kumas maliyetini hesaplayacagiz */

                select sum(StokHareket.Miktar * StokHareket.BirimMaliyet) / sum(StokHareket.Miktar) from StokHareket
                  left join Ham on Ham.HamID = StokHareket.HamId
                  where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and
                        StokHareket.UretimIslemTipiID = :UretimIslemTipiID and
                        Ham.HamTipiId = :HamTipiId and
                        StokHareket.GirisCikis = -1 and
                        StokHareket.Miktar > 0
                  into :HamMaliyeti;

                if (HamMaliyeti is null) then HamMaliyeti = 0;

                BirimMaliyet = HamMaliyeti + NetBirimFiyat;
              end
              else
              begin
                BirimMaliyet = 0;
                HamMaliyeti = 0;

                select PartiSinifID from BoyaIrsaliyeDetay where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = :BoyaIrsaliyeDetayID into :PartiSinifID;

                if (PartiSinifID is not null) then
                   select HamStokBirimMaliyeti from SiparisDetayBirimMaliyetGetir(:PartiSinifID, null) into :HamMaliyeti;

                BirimMaliyet = HamMaliyeti + NetBirimFiyat;
                /* kendi boyahaneimizden gelirken NetBirimFiyat sipariş prosesden ve ilave işlenlerden hesaplanmıs gelecek */
              end
            end
          end
        end
      end
      else
        select BirimMaliyet from RafFifoMaliyetGetirTemp(:SHOrmeRafId, :Miktar, :Tarih, :StokHareketId) into :BirimMaliyet;

      if (BirimMaliyet is null) then BirimMaliyet = 0;


      if (OrmeFaturaId is Null) then
        update StokHareket set BirimMaliyet = :BirimMaliyet, BirimFiyat = :BirimFiyat, NeredenUpdate = 4, LinkIslemi = 1 where StokHareket.StokHareketId = :StokHareketId;
      else
        update StokHareket set BirimMaliyet = :BirimMaliyet, NeredenUpdate = 4, LinkIslemi = 1 where StokHareket.StokHareketId = :StokHareketId;

      Select BirimMaliyet from Stokhareket where StokHareket.StokHareketId = :StokHareketId into :BirimMaliyet;
      update HamStokMaster set NeredenUpdate = 3, BirimFiyat = :BirimMaliyet where HamStokMaster.StokhareketID = :StokhareketID;
      suspend;
    end
  end

  StokhareketID = Null;
  OrmeRafTransferFisiDetayId = Null;
  BirimMaliyet  = Null;
  NetBirimFiyat = Null;

  if (RafTransferIdler is Not Null) then
  begin
    Sorgu = 'Select OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId, OrmeRafTransferFisi.FisTarihi, OrmeRafTransferFisiDetay.Kilo, OrmeRafTransferFisiDetay.OrmeRafId from OrmeRafTransferFisiDetay '
          ||'Left Join OrmeRafTransferFisi on OrmeRafTransferFisi.OrmeRafTransferFisiId = OrmeRafTransferFisiDetay.OrmeRafTransferFisiId '
          ||'Left join OrmeRaf on OrmeRaf.OrmeRafId = OrmeRafTransferFisiDetay.OrmeRafId '
          ||'  where OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId in ('||RafTransferIdler||') Order by OrmeRafTransferFisi.FisTarihi, OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId asc';

    for execute statement(Sorgu) into :OrmeRafTransferFisiDetayId, :Tarih, :Miktar, :SHOrmeRafId do
    begin

      Select BirimMaliyet from RafFifoMaliyetGetirTemp(:SHOrmeRafId, :Miktar, :Tarih, :OrmeRafTransferFisiDetayId) into :BirimMaliyet;

      if (BirimMaliyet is null) then BirimMaliyet = 0;
      Update OrmeRafTransferFisiDetay set BirimMaliyet = :BirimMaliyet where OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId = :OrmeRafTransferFisiDetayId;

      suspend;
    end
  end
end
