CREATE PROCEDURE HSSARFFISIICINTARTIGETIR(
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    ADEPOID VARCHAR(10),
    APARTISIRANO INTEGER,
    AMAMULDEPOTRANSFERFISIID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    PARTISIRANO INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    TARIH TIMESTAMP,
    KAPADEDI INTEGER,
    TOPNO INTEGER,
    BRUTKILO NUMERIC(12,3),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    DARA NUMERIC(12,3),
    KALITENO INTEGER,
    DEPOID VARCHAR(10),
    HESAPLANANBRUTKILO NUMERIC(12,3),
    HESAPLANANBRUTMETRE NUMERIC(12,3),
    KALITEACIKLAMATIPIID INTEGER,
    KALITEACIKLAMATIPIADI VARCHAR(20),
    MAMULPROSESID INTEGER,
    MAMULPROSESADI VARCHAR(255),
    PALETNO VARCHAR(5),
    MAMULDEPOTRANSFERFISIID INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    KALITEADI VARCHAR(250),
    DESENNUMARASI VARCHAR(20),
    LABRECETEKODU VARCHAR(20),
    DESENVARYANTADI VARCHAR(30))
AS
declare variable BoyaIrsaliyeDetayId integer;
declare variable HamStokSarfFisiDetayId integer;
begin
  if(:APartiYil > 0 or :APartiId > 0) then
  begin
    For Select PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId, PartiStokTarti.Tarih, PartiStokTarti.KapAdedi, PartiStokTarti.Kilo,
               PartiStokTarti.Metre, PartiStokTarti.Adet, PartiStokTarti.Dara, PartiStokTarti.KaliteNo, PartiStokTarti.DepoId, PsKaliteKontrol.TopNo,
               PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, PartiStokTarti.KaliteAciklamaTipiId, KaliteAciklamaTipi.Adi, PartiStokTarti.MamulProsesId,
               MamulProses.Adi, IK.Adi, PartiStokTarti.InsertTarihi, SK.Adi, PartiStokTarti.DegisimTarihi, PartiStokTarti.BrutKilo, PartiStokTarti.BrutMetre, PartiStokTarti.PaletNo,
               PartiStok.HamAdiToplami, PartiStok.PartiSiraNo, PartiStokTarti.MamulDepoTransferFisiId, Kalite.KaliteAdi, BoyaSiparisDetay.DesenId, BoyaSiparisDetay.LabReceteKodu, DesenVaryant.VaryantAdi from Parti
        left join PartiStok on PartiStok.PartiKayitId = Parti.PArtiKayitId
        left join PartiStokTarti on PartiStokTarti.PartiSinifId = PartiStok.PartiSinifId
        Left Join PsKaliteKontrol on PsKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
        Left Join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = PartiStokTarti.KaliteAciklamaTipiId
        Left Join MamulProses on MamulProses.MamulProsesId = PartiStokTarti.MamulProsesId
        Left Join Kullanici IK on IK.KullaniciId = PartiStokTarti.InsertKullaniciId
        Left Join Kullanici SK on SK.KullaniciId = PartiStokTarti.KullaniciId
        Left Join BoyaSiparisDetay on BoyaSiparisDetay.boyasiparisdetayid = PartiStok.BoyaSiparisDetayId
        Left Join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
        Left Join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    Where Parti.PartiYil = :APartiYil
        and Parti.PartiId = :APartiId
        and PartiStok.Durumu = 7
        and (PartiStok.PartiSiraNo = :APartiSiraNo)
        and (Parti.ParcaliSevk = 1 or (Parti.ParcaliSevk = 0 and Partistok.TamamiTartildi = 1))
        and (PartistokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.BoyaRezervFisiDetayId is null)
    Order by PartiStokTarti.PartiStokTartiId
    into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, Tarih, :KapAdedi, :Kilo, :Metre, :Adet, :Dara, :KaliteNo, :DepoId, :TopNo,
         :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :KaliteAciklamaTipiId, :KaliteAciklamaTipiAdi, :MamulProsesId,
         :MamulProsesAdi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :HesaplananBrutKilo, :HesaplananBrutMetre, :PaletNo,
         :HamAdiToplami, :PartiSiraNo, :MamulDepoTransferFisiId, :KaliteAdi, :DesenNumarasi, :LabReceteKodu, :DesenVaryantAdi do
    begin
      BrutKilo = Kilo + Dara;
      suspend;
    end
  end
  else
  begin
    For Select PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId, PartiStokTarti.Tarih,
               PartiStokTarti.KapAdedi, PartiStokTarti.Kilo, PartiStokTarti.Metre, PartiStokTarti.Adet, PartiStokTarti.Dara,
               PartiStokTarti.KaliteNo, PartiStokTarti.DepoId,
               PsKaliteKontrol.TopNo, PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId,
               PartiStokTarti.KaliteAciklamaTipiId, KaliteAciklamaTipi.Adi, PartiStokTarti.MamulProsesId, MamulProses.Adi,
               IK.Adi, PartiStokTarti.InsertTarihi, SK.Adi, PartiStokTarti.DegisimTarihi,
               PartiStokTarti.BrutKilo, PartiStokTarti.BrutMetre, PartiStokTarti.PaletNo, PartiStok.HamAdiToplami, PartiStok.PartiSiraNo,MamulDepoTransferFisi.MamulDepoTransferFisiId,
               Kalite.KaliteAdi, BoyaSiparisDetay.DesenId, BoyaSiparisDetay.LabReceteKodu, DesenVaryant.VaryantAdi from MamulDepoTransferFisi
         left join PartiStokTarti on PartiStokTarti.MamulDepoTransferFisiId = MamulDepoTransferFisi.MamulDepoTransferFisiId
         left Join PsKaliteKontrol on PsKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
         left Join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = PartiStokTarti.KaliteAciklamaTipiId
         left Join MamulProses on MamulProses.MamulProsesId = PartiStokTarti.MamulProsesId
         left Join Kullanici IK on IK.KullaniciId = PartiStokTarti.InsertKullaniciId
         left Join Kullanici SK on SK.KullaniciId = PartiStokTarti.KullaniciId
         left Join PartiStok on PartiStok.PartiKayitId = PartiStokTarti.PartiKayitId
         left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
         Left Join BoyaSiparisDetay on BoyaSiparisDetay.boyasiparisdetayid = PartiStok.BoyaSiparisDetayId
         Left Join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
         Left Join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    Where  (MamulDepoTransferFisi.FisNo = :AMamulDepoTransferFisiId or :AMamulDepoTransferFisiId is  null)
        and PartiStok.Durumu = 7
        and (PartiStok.PartiSiraNo = :APartiSiraNo or :APartiSiraNo is null)
        and (PartiStokTarti.DepoId = :ADepoId or  :ADepoId is null)
        and (Parti.ParcaliSevk = 1 or (Parti.ParcaliSevk = 0 and Partistok.TamamiTartildi = 1))
        and (PartiStokTarti.BoyaIrsaliyeDetayId is null and PartiStokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.BoyaRezervFisiDetayId is null)
    Order by PartiStokTarti.PartiStokTartiId
    into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, Tarih, :KapAdedi, :Kilo, :Metre, :Adet, :Dara, :KaliteNo, :DepoId, :TopNo,
         :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :KaliteAciklamaTipiId, :KaliteAciklamaTipiAdi, :MamulProsesId, :MamulProsesAdi,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
         :HesaplananBrutKilo, :HesaplananBrutMetre, :PaletNo, :HamAdiToplami, :PartiSiraNo, :MamulDepoTransferFisiId, :KaliteAdi, :DesenNumarasi, :LabReceteKodu, :DesenVaryantAdi do
    begin
      BrutKilo = Kilo + Dara;
      suspend;
    end
  end
end
