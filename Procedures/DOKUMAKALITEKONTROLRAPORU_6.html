CREATE PROCEDURE DOKUMAKALITEKONTROLRAPORU_6(
    ADOKUMAMAKINAID VARCHAR(10),
    ADOKUMAPLANLAMAID INTEGER,
    ASAKAT INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKONTROLDURUMU INTEGER,
    AHAMID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    AKUMASDURUMU INTEGER)
RETURNS (
    HAMID INTEGER,
    RECETEHAMID INTEGER,
    RECETEHAMADI ADI255,
    RENKID INTEGER,
    RENKADI ADI50,
    ATKICOZGU INTEGER,
    HESAPLANANMIKTAR BOYAMIKTARSIZE,
    TALIMATCIKISMIKTARI BOYAMIKTARSIZE,
    SONALISFIYATI DOUBLE PRECISION,
    SONALISDOVIZID DOVIZSIZE,
    FASON SMALLINT,
    HATASTR VARCHAR(500),
    TELSAYISI VARCHAR(20))
AS
declare variable shtFasonUretim Integer;
declare variable uitDokuma Integer;
declare variable SonUretimTarihi  timestamp;
declare variable FHataStr         varchar(500);
declare variable HamId2           integer;
declare variable ReceteMiktari BoyaMiktarSize;
begin
  select shtFasonUretim from OrmeStokHareketTipiGetir into :shtFasonUretim;
  select uitDokuma from UretimIslemTipiGetir into :uitDokuma;

  for select Fason, KumasHamId, HamId, HamAdi, AtkiCozgu, Sum(HesaplananMiktar), Sum(CikilanMiktar), RenkId, RenkAdi, MAx(Tarih), TelSayisi from (
        select 0 as Fason, DokumaKaliteKontrolRaporu_2.KumasHamId, DokumaKaliteKontrolRaporu_2.HamId, DokumaKaliteKontrolRaporu_2.HamAdi, DokumaPlanlamaRecete.AtkiCozgu,
        Sum(DokumaKaliteKontrolRaporu_2.HesaplananMiktar) HesaplananMiktar, Sum(DokumaKaliteKontrolRaporu_2.CikilanMiktar) CikilanMiktar, DokumaKaliteKontrolRaporu_2.RenkId, DokumaKaliteKontrolRaporu_2.RenkAdi,
        Max(DokumaKaliteKontrolRaporu_2.Tarih) Tarih,DokumaPlanlamaRecete.TelSayisi
        from DokumaKaliteKontrolRaporu_2(:ADokumaMakinaId, :ADokumaPlanlamaId, :ASakat, :ABaslangicTarihi, :ABitisTarihi, :AKontrolDurumu, :AHamId, :ABoyaSiparisKayitId, :AKumasDurumu)
        left join DokumaPlanlamaRecete on DokumaPlanlamaRecete.DokumaPlanlamaReceteId = DokumaKaliteKontrolRaporu_2.DokumaPlanlamaReceteId
        group by DokumaKaliteKontrolRaporu_2.KumasHamId, DokumaKaliteKontrolRaporu_2.HamId, DokumaKaliteKontrolRaporu_2.HamAdi, DokumaPlanlamaRecete.AtkiCozgu, DokumaKaliteKontrolRaporu_2.RenkId, DokumaKaliteKontrolRaporu_2.RenkAdi,
        DokumaPlanlamaRecete.TelSayisi
          union all
        select 1 as Fason, StokHareket.HamId, DokumaPlanlamaRecete.HamId, Ham.Adi, DokumaPlanlamaRecete.AtkiCozgu, Sum(DokumaUretimeOtomatikCikis.HesaplananMiktar), Sum(DokumaUretimeOtomatikCikis.CikilanMiktar),
        DokumaPlanlamaRecete.RenkId, Renk.Adi, Max(StokHareket.Tarih), DokumaPlanlamaRecete.TelSayisi
        from StokHareket
        left join DokumaUretimeOtomatikCikis on DokumaUretimeOtomatikCikis.StokHareketId = StokHareket.StokHareketID
        left join DokumaPlanlamaRecete on DokumaPlanlamaRecete.DokumaPlanlamaReceteId = DokumaUretimeOtomatikCikis.DokumaPlanlamaReceteId
        left join Ham on Ham.HamId = DokumaPlanlamaRecete.HamId
        left join Renk on Renk.RenkId = DokumaPlanlamaRecete.RenkId
        where StokHareket.HareketTipi = :shtFasonUretim and
        StokHareket.HamIade = 0 and
        StokHareket.DokumaHareketi = 1 and
        (StokHareket.Tarih >= cast(:ABaslangicTarihi as date) and StokHareket.Tarih <= cast(:ABitistarihi as date)+1) and
        StokHAreket.UretimIslemTipiId = :uitDokuma and
        (StokHareket.DokumaPlanlamaId = :ADokumaPlanlamaId or :ADokumaPlanlamaId is null) and
        (StokHareket.HamId = :AHamId or :AHamId is null) and
        (StokHareket.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
        (StokHareket.Sakat = :ASakat or :ASakat is null)
        group by StokHareket.HamId, DokumaPlanlamaRecete.HamId, Ham.Adi, DokumaPlanlamaRecete.AtkiCozgu, DokumaPlanlamaRecete.RenkId, Renk.Adi, DokumaPlanlamaRecete.TelSayisi
      ) group by Fason, KumasHamId, HamId, HamAdi, AtkiCozgu, RenkId, RenkAdi, TelSayisi
      into :Fason, :HamId, :ReceteHamId, :ReceteHamAdi, :AtkiCozgu, :HesaplananMiktar, :TalimatCikisMiktari, :RenkID, :RenkAdi, :SonUretimTarihi, :TelSayisi do
  begin
    select OrmeHamStokSonAlisFiyati_v2.SonAlisFiyati, OrmeHamStokSonAlisFiyati_v2.SonAlisDovizId from
    OrmeHamStokSonAlisFiyati_v2(:ReceteHamId, :SonUretimTarihi) into :SonAlisFiyati, :SonAlisDovizId;

    ReceteMiktari = HesaplananMiktar;

    suspend;

    SonAlisFiyati= Null;
    SonAlisDovizId = Null;

    HamId2 = ReceteHamId;

    if (AtkiCozgu = 2) then
    begin
      for select HamStokRecete.ReceteHamId, Ham.Adi, 3, DokumaIplikMiktariHesaplama.AgirlikKilo, DokumaIplikMiktariHesaplama.AgirlikKilo, DokumaIplikMiktariHesaplama.HataStr
        from HamStokRecete
        left join DokumaIplikMiktariHesaplama(null, null, null, 0, :ReceteMiktari, HamStokRecete.HamStokReceteId) on 1=1
        left join Ham on HamStokRecete.ReceteHamId = Ham.HamId
        where HamStokRecete.HamId = :HamId2
        into :ReceteHamId, :ReceteHamAdi, :AtkiCozgu, :HesaplananMiktar, :TalimatCikisMiktari, :FHataStr do
        begin
          HataStr = '';
          if (FHataStr <> '') then
          begin
            HataStr = ReceteHamAdi||': '||FHataStr;
          end

          select OrmeHamStokSonAlisFiyati_v2.SonAlisFiyati, OrmeHamStokSonAlisFiyati_v2.SonAlisDovizId from
          OrmeHamStokSonAlisFiyati_v2(:ReceteHamId, :SonUretimTarihi) into :SonAlisFiyati, :SonAlisDovizId;

          suspend;

          SonAlisFiyati= Null;
          SonAlisDovizId = Null;
        end
    end
  end
end
