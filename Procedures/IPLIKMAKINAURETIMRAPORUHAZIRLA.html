CREATE PROCEDURE IPLIKMAKINAURETIMRAPORUHAZIRLA(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAGRUPID VARCHAR(4),
    AIPLIKMAKINATIPIID INTEGER)
RETURNS (
    IPLIKMAKINATIPIID INTEGER,
    IPLIKMAKINAID VARCHAR(10),
    MAKINAIGSAYISI INTEGER,
    HIZMTDK INTEGER,
    MAKINAGRUPID VARCHAR(4),
    TARIH TIMESTAMP,
    HARMANKAYITID INTEGER,
    HARMANIDSTR VARCHAR(30),
    HARMANADI VARCHAR(50),
    HARMANCINSIADI VARCHAR(50),
    IPLIKURETIMISEMRIKAYITID INTEGER,
    IPLIKURETIMISEMRIIDSTR VARCHAR(30),
    KALINLIKNE DOUBLE PRECISION,
    LOT VARCHAR(30),
    BUKUMSAYISI INTEGER,
    VARDIYA1TEORIKURETIMMIKTARI DOUBLE PRECISION,
    VARDIYA1SAYACDEGERI DOUBLE PRECISION,
    VARDIYA1HESAPLANANURETIM DOUBLE PRECISION,
    VARDIYA1MAKINATARTITOPLAMI DOUBLE PRECISION,
    VARDIYA2TEORIKURETIMMIKTARI DOUBLE PRECISION,
    VARDIYA2SAYACDEGERI DOUBLE PRECISION,
    VARDIYA2HESAPLANANURETIM DOUBLE PRECISION,
    VARDIYA2MAKINATARTITOPLAMI DOUBLE PRECISION,
    VARDIYA3TEORIKURETIMMIKTARI DOUBLE PRECISION,
    VARDIYA3SAYACDEGERI DOUBLE PRECISION,
    VARDIYA3HESAPLANANURETIM DOUBLE PRECISION,
    VARDIYA3MAKINATARTITOPLAMI DOUBLE PRECISION,
    BIRTURKACPULSE INTEGER,
    EKSIKIGSAYISI INTEGER,
    HAMADI VARCHAR(255),
    IPLIKMAKINAISLETMEID INTEGER)
AS
declare variable FVardiyaNo            integer;
declare variable FHarmanYil            integer;
declare variable FHarmanId             integer;
declare variable FIplikUretimIsEmriYil integer;
declare variable FIplikUretimIsEmriId  integer;
declare variable FMiktar               Double Precision;
declare variable FSayac                Double Precision;
declare variable FEksikIgSayisi        integer;
begin
  For Select IplikMakina.IplikMakinaID, IplikMakina.IplikMakinaTipiId, IplikMakina.MakinaGrupId, IplikMakina.DefaultHizMtDk, IplikMakina.ToplamIgSayisi, IplikMakina.BirTurKacPulse, IplikMakina.IplikIsletmeId from IplikMakina
    where (IplikMakina.MakinaGrupID = :AMakinaGrupID or :AMakinaGrupID is null) and
          (IplikMakina.IplikMakinaTipiID = :AIplikMakinaTipiID or :AIplikMakinaTipiID is null)
    into :IplikMakinaID, :IplikMakinaTipiId, :MakinaGrupId, :HizMtDk, :MakinaIgSayisi, :BirTurKacPulse, :IplikMakinaIsletmeId  do
  begin
    Tarih                      = NUll;
    HarmanKayitId              = NUll;
    HarmanIdStr                = NUll;
    HarmanAdi                  = NUll;
    HarmanCinsiAdi             = NUll;
    IplikUretimIsEmriKayitId   = NUll;
    IplikUretimIsEmriIdStr     = NUll;
    KalinlikNe                 = NUll;
    Lot                        = NUll;
    BukumSayisi                = NUll;
    Vardiya1TeorikUretimMiktari= NUll;
    Vardiya1SayacDegeri        = NUll;
    Vardiya1HesaplananUretim   = NUll;
    Vardiya1MakinaTartiToplami = NUll;
    Vardiya2TeorikUretimMiktari= NUll;
    Vardiya2SayacDegeri        = NUll;
    Vardiya2HesaplananUretim   = NUll;
    Vardiya2MakinaTartiToplami = NUll;
    Vardiya3TeorikUretimMiktari= NUll;
    Vardiya3SayacDegeri        = NUll;
    Vardiya3HesaplananUretim   = NUll;
    Vardiya3MakinaTartiToplami = NUll;

    For Select IplikMakinaTarti.VardiyaTarihi, IplikMakinaTarti.VardiyaNo, IplikMakinaTarti.HarmanKayitId, Harman.HarmanYil, Harman.HarmanId,
               Harman.Adi, HarmanCinsi.Adi, IplikMakinaTarti.IplikUretimIsEmriKayitId,  IplikUretimIsEmri.IplikUretimIsEmriYil, IplikUretimIsEmri.IplikUretimIsEmriId,
               IplikLot.Lot, IplikLot.BukumSayisi, IplikMakinaTarti.Miktar - IplikMakinaTarti.Dara from IplikMakinaTarti
    left join IplikUretimIsEmri on IplikUretimIsEmri.IplikUretimIsEmriKayitID = IplikMakinaTarti.IplikUretimIsEmriKayitID
    left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotID
    left join Harman on Harman.HarmanKayitId = coalesce(IplikMakinaTarti.HarmanKayitId, IplikLot.HarmanKayitId)
    Left Join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Harman.HarmanCinsiId
    where IplikMakinaTarti.VardiyaTarihi >= :ABaslangicTarihi and IplikMakinaTarti.VardiyaTarihi <= :ABitisTarihi and
          IplikMakinaTarti.IplikMakinaID = :IplikMakinaId
    into :Tarih, :FVardiyaNo, :HarmanKayitId, :FHarmanYil, :FHarmanId, :HarmanAdi, :HarmanCinsiAdi, :IplikUretimIsEmriKayitId,
         :FIplikUretimIsEmriYil, :FIplikUretimIsEmriId, :Lot, :BukumSayisi, :FMiktar do
    begin
      Vardiya1MakinaTartiToplami = 0;
      Vardiya2MakinaTartiToplami = 0;
      Vardiya3MakinaTartiToplami = 0;

      if (FVardiyaNo = 1) then Vardiya1MakinaTartiToplami = :FMiktar;
      else if (FVardiyaNo = 2) then Vardiya2MakinaTartiToplami = :FMiktar;
      else Vardiya3MakinaTartiToplami = :FMiktar;

      if (FHarmanID is not null) then
        HarmanIdStr = '(' || EgemenIntegerToStr(FHarmanYil) || ') ' || EgemenIntegerToStr(FHarmanId) || ' ';
      else
        HarmanIDStr = '';

      if (FIplikUretimIsEmriId is not null) then
        IplikUretimIsEmriIdStr = '(' || EgemenIntegerToStr(FIplikUretimIsEmriYil) || ') ' || EgemenIntegerToStr(FIplikUretimIsEmriId) || ' ';
      else
        IplikUretimIsEmriIdStr = '';
      suspend;
    end

    Tarih                      = NUll;
    HarmanKayitId              = NUll;
    HarmanIdStr                = NUll;
    HarmanAdi                  = NUll;
    HarmanCinsiAdi             = NUll;
    IplikUretimIsEmriKayitId   = NUll;
    IplikUretimIsEmriIdStr     = NUll;
    KalinlikNe                 = NUll;
    Lot                        = NUll;
    BukumSayisi                = NUll;
    Vardiya1TeorikUretimMiktari= NUll;
    Vardiya1SayacDegeri        = NUll;
    Vardiya1HesaplananUretim   = NUll;
    Vardiya1MakinaTartiToplami = NUll;
    Vardiya2TeorikUretimMiktari= NUll;
    Vardiya2SayacDegeri        = NUll;
    Vardiya2HesaplananUretim   = NUll;
    Vardiya2MakinaTartiToplami = NUll;
    Vardiya3TeorikUretimMiktari= NUll;
    Vardiya3SayacDegeri        = NUll;
    Vardiya3HesaplananUretim   = NUll;
    Vardiya3MakinaTartiToplami = NUll;

    For Select IplikMakinaSayac.VardiyaTarihi, IplikMakinaSayac.VardiyaNo, IplikMakinaSayac.HarmanKayitId, Harman.HarmanYil, Harman.HarmanId,
               Harman.Adi, HarmanCinsi.Adi, IplikMakinaSayac.IplikUretimIsEmriKayitId,  IplikUretimIsEmri.IplikUretimIsEmriYil, IplikUretimIsEmri.IplikUretimIsEmriId,
               IplikMakinaSayac.KalinlikNe, IplikLot.Lot, IplikLot.BukumSayisi,
               IplikMakinaSayac.Sayac, IplikMakinaSayac.EksikIgSayisi, Ham.Adi from IplikMakinaSayac
    left join IplikUretimIsEmri on IplikUretimIsEmri.IplikUretimIsEmriKayitID = IplikMakinaSayac.IplikUretimIsEmriKayitID
    left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotID
    left join Ham on Ham.hamid =IplikLot.HamId
    left join Harman on Harman.HarmanKayitId = coalesce(IplikMakinaSayac.HarmanKayitId, IplikLot.HarmanKayitId)
    Left Join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Harman.HarmanCinsiId
    where IplikMakinaSayac.VardiyaTarihi >= :ABaslangicTarihi and IplikMakinaSayac.VardiyaTarihi <= :ABitisTarihi and
          IplikMakinaSayac.IplikMakinaID = :IplikMakinaId
    into :Tarih, :FVardiyaNo, :HarmanKayitId, :FHarmanYil, :FHarmanId, :HarmanAdi, :HarmanCinsiAdi,
         :IplikUretimIsEmriKayitId,  :FIplikUretimIsEmriYil, :FIplikUretimIsEmriId, :KalinlikNe, :Lot, :BukumSayisi,
         :FSayac,  :EksikIgSayisi, :HamAdi do
    begin
      Vardiya1SayacDegeri = 0;
      Vardiya2SayacDegeri = 0;
      Vardiya3SayacDegeri = 0;

      if (FVardiyaNo = 1) then Vardiya1SayacDegeri = :FSayac;
      else if (FVardiyaNo = 2) then Vardiya2SayacDegeri = :FSayac;
      else Vardiya3SayacDegeri = :FSayac;

      if (FHarmanID is not null) then
        HarmanIdStr = '(' || EgemenIntegerToStr(FHarmanYil) || ') ' || EgemenIntegerToStr(FHarmanId) || ' ';
      else
        HarmanIDStr = '';

      if (FIplikUretimIsEmriId is not null) then
        IplikUretimIsEmriIdStr = '(' || EgemenIntegerToStr(FIplikUretimIsEmriYil) || ') ' || EgemenIntegerToStr(FIplikUretimIsEmriId) || ' ';
      else
        IplikUretimIsEmriIdStr = '';

      Vardiya1TeorikUretimMiktari = 0;
      Vardiya2TeorikUretimMiktari = 0;
      Vardiya3TeorikUretimMiktari = 0;
      Vardiya1HesaplananUretim = 0;
      Vardiya2HesaplananUretim = 0;
      Vardiya3HesaplananUretim = 0;

      if (IplikMakinaTipiID = 4) then /* tarak */
      begin
        if (KalinlikNe is not null and KalinlikNe <> 0) then
        begin
          Vardiya1TeorikUretimMiktari = HizMtDk * 480 * 0.59 / KalinlikNe / 1000;
          Vardiya2TeorikUretimMiktari = HizMtDk * 480 * 0.59 / KalinlikNe / 1000;
          Vardiya3TeorikUretimMiktari = HizMtDk * 480 * 0.59 / KalinlikNe / 1000;

          Vardiya1HesaplananUretim = Vardiya1SayacDegeri * 0.59 / KalinlikNE / 1000;
          Vardiya2HesaplananUretim = Vardiya2SayacDegeri * 0.59 / KalinlikNE / 1000;
          Vardiya3HesaplananUretim = Vardiya3SayacDegeri * 0.59 / KalinlikNE / 1000;
        end
      end
      else if (IplikMakinaTipiID = 5) then /* cer-1*/
      begin
        if (KalinlikNe is not null and KalinlikNe <> 0) then
        begin
          Vardiya1TeorikUretimMiktari = HizMtDk * 480 * 0.59 / KalinlikNe / 1000;
          Vardiya2TeorikUretimMiktari = HizMtDk * 480 * 0.59 / KalinlikNe / 1000;
          Vardiya3TeorikUretimMiktari = HizMtDk * 480 * 0.59 / KalinlikNe / 1000;

          Vardiya1HesaplananUretim = Vardiya1SayacDegeri * 0.59 / KalinlikNE / 1000;
          Vardiya2HesaplananUretim = Vardiya2SayacDegeri * 0.59 / KalinlikNE / 1000;
          Vardiya3HesaplananUretim = Vardiya3SayacDegeri * 0.59 / KalinlikNE / 1000;
        end
      end
      else if (IplikMakinaTipiID = 6) then /* cer-2*/
      begin
        if (KalinlikNe is not null and KalinlikNe <> 0) then
        begin
          Vardiya1TeorikUretimMiktari = HizMtDk * 480 * 0.59 / KalinlikNe / 1000;
          Vardiya2TeorikUretimMiktari = HizMtDk * 480 * 0.59 / KalinlikNe / 1000;
          Vardiya3TeorikUretimMiktari = HizMtDk * 480 * 0.59 / KalinlikNe / 1000;

        end
      end
      else
      if (IplikMakinaTipiID = 7) then /* vatka */
      begin
        if (KalinlikNe is not null and KalinlikNe <> 0) then
        begin
          Vardiya1TeorikUretimMiktari = HizMtDk * 480 * KalinlikNe / 1000;
          Vardiya2TeorikUretimMiktari = HizMtDk * 480 * KalinlikNe / 1000;
          Vardiya3TeorikUretimMiktari = HizMtDk * 480 * KalinlikNe / 1000;

          Vardiya1HesaplananUretim = Vardiya1SayacDegeri * KalinlikNE / 1000;
          Vardiya2HesaplananUretim = Vardiya2SayacDegeri * KalinlikNE / 1000;
          Vardiya3HesaplananUretim = Vardiya3SayacDegeri * KalinlikNE / 1000;
        end
      end
      else if (IplikMakinaTipiID = 8) then /* pen√∂z */
      begin
        if (KalinlikNe is not null and KalinlikNe <> 0) then
        begin
          Vardiya1TeorikUretimMiktari = HizMtDk * 480;
          Vardiya2TeorikUretimMiktari = HizMtDk * 480;
          Vardiya3TeorikUretimMiktari = HizMtDk * 480;

          Vardiya1HesaplananUretim = Vardiya1SayacDegeri * 0.59 / KalinlikNe;
          Vardiya2HesaplananUretim = Vardiya2SayacDegeri * 0.59 / KalinlikNe;
          Vardiya3HesaplananUretim = Vardiya3SayacDegeri * 0.59 / KalinlikNe;
        end
      end

      suspend;
    end
  end
end
