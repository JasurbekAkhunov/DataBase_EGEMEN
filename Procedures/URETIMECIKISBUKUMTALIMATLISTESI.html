CREATE PROCEDURE URETIMECIKISBUKUMTALIMATLISTESI(
    AORMEMAKINAID VARCHAR(10),
    AFIRMAID FIRMAKODUSIZE,
    ADOKUMAHAREKETI EGEBOOLEAN)
RETURNS (
    RESULTID VARCHAR(255),
    BUKUMTALIMATID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    SIPARISTIPIID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    URETIMPARTINO VARCHAR(15),
    HAMADI VARCHAR(255))
AS
declare variable Bitti integer;
begin
  if (AOrmeMakinaId is not null) then
  begin
    for Select BukumTalimat.BukumTalimatId, BukumTalimat.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
              BukumTalimat.UretimPartiNo, BoyaSiparis.OrmeSiparisTipiId, Ham.Adi, BukumTalimat.Bitti from BukumTalimat
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BukumTalimat.BoyaSiparisKayitId
    left join Ham on Ham.HamId = BukumTalimat.HamId
    where ((BukumTalimat.OrmeMakinaId = :AOrmeMakinaId and :ADokumaHareketi = 0) or
           (BukumTalimat.DokumaMakinaId = :AOrmeMakinaId and :ADokumaHareketi = 1)) and
            Bitti = 0 and BukumTalimat.DokumaTalimati = :ADokumaHareketi
    into :BukumTalimatId, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :UretimPartiNo, :SiparisTipiId,
         :HamAdi, :Bitti do
    begin
        ResultId = '"' || EgemenIntegerToStr(BukumTalimatId) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisKayitId) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisYil) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisId) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(SiparisTipiId) || '",' ;
        ResultId = ResultId || '"' || UretimPartiNo || '",' ;
        ResultId = ResultId || '"' || HamAdi || '"' ;

        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
        suspend;
    end
  end
  else if (AFirmaId is not null) then
  begin
    for Select BukumTalimat.BukumTalimatId, BukumTalimat.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
              BukumTalimat.UretimPartiNo, BoyaSiparis.OrmeSiparisTipiId, Ham.Adi, BukumTalimat.Bitti from BukumTalimat
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BukumTalimat.BoyaSiparisKayitId
    left join Ham on Ham.HamID = BukumTalimat.HamID
    where BukumTalimat.FirmaId = :AFirmaId and Bitti = 0 and BukumTalimat.DokumaTalimati = :ADokumaHareketi
    into :BukumTalimatId, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :UretimPartiNo, :SiparisTipiId,
         :HamAdi, :Bitti do
    begin
      if (Bitti = 0) then
      begin
        ResultId = '"' || EgemenIntegerToStr(BukumTalimatId) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisKayitId) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisYil) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisId) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(SiparisTipiId) || '",' ;
        ResultId = ResultId || '"' || UretimPartiNo || '",' ;
        ResultId = ResultId || '"' || HamAdi || '"' ;

        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
        suspend;
      end
    end
  end
  else
  begin
    for Select BukumTalimat.BukumTalimatId, BukumTalimat.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
              BukumTalimat.UretimPartiNo, BoyaSiparis.OrmeSiparisTipiId, Ham.Adi, BukumTalimat.Bitti from BukumTalimat
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BukumTalimat.BoyaSiparisKayitId
    left join Ham on Ham.HamID = BukumTalimat.HamID
    where Bitti = 0 and BukumTalimat.DokumaTalimati = :ADokumaHareketi
    into :BukumTalimatId, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :UretimPartiNo, :SiparisTipiId,
         :HamAdi, :Bitti do
    begin
      if (Bitti = 0) then
      begin
        ResultId = '"' || EgemenIntegerToStr(BukumTalimatId) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisKayitId) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisYil) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisId) || '",' ;
        ResultId = ResultId || '"' || EgemenIntegerToStr(SiparisTipiId) || '",' ;
        ResultId = ResultId || '"' || UretimPartiNo || '",' ;
        ResultId = ResultId || '"' || HamAdi || '"' ;

        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
        suspend;
      end
    end
  end
end
