CREATE PROCEDURE SAATEGOREMAKINACALISMARAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAGRUBU VARCHAR(4),
    AMAKINAID VARCHAR(10),
    ASAAT1 TIMESTAMP,
    ASAAT2 TIMESTAMP,
    ASAAT3 TIMESTAMP)
RETURNS (
    TARIH TIMESTAMP,
    MAKINAID VARCHAR(10),
    MAKINAGRUBU VARCHAR(4),
    CALISMASURESI1 DOUBLE PRECISION,
    CALISMASURESI2 DOUBLE PRECISION,
    CALISMASURESI3 DOUBLE PRECISION)
AS
begin
  For Select Tarih, MakinaId, MakinaGrubu, Sum(CalismaSuresi1), Sum(CalismaSuresi2), Sum(CalismaSuresi3) from SaateGoreMakinaCalismaRaporu1(:ABaslangicTarihi, :ABitisTarihi, :AMakinaGrubu, :AMakinaId, :ASaat1, :ASaat2, :ASaat3)
  group by Tarih, MakinaId, MakinaGrubu
  order by MakinaId, Tarih
  into :Tarih, :MakinaId, :MakinaGrubu, :CalismaSuresi1, :CalismaSuresi2, :CalismaSuresi3 do
  begin
    CalismaSuresi1 = CalismaSuresi1 * 24;
    CalismaSuresi2 = CalismaSuresi2 * 24;
    CalismaSuresi3 = CalismaSuresi3 * 24;

    CalismaSuresi1 = Yuvarla(CalismaSuresi1, 100);
    CalismaSuresi2 = Yuvarla(CalismaSuresi2, 100);
    CalismaSuresi3 = Yuvarla(CalismaSuresi3, 100);
    suspend;
  end
end
