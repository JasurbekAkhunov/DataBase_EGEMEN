CREATE PROCEDURE PARTISABLONACILIMIHESAPLA(
    APARTIKAYITID INTEGER)
AS
declare variable EmulsiyonUrunId       varchar(20);
declare variable UrunBirimId           integer;
declare variable PartiSablonId         integer;
declare variable Acilim1               Double precision;
declare variable Acilim2               Double precision;
declare variable AcilimToplami         Double precision;
declare variable HazirlananMiktar      Double Precision;
declare variable BoyaToplami           Double Precision;
declare variable PatToplami            Double Precision;
declare variable B1                    Double Precision;
declare variable X1                    Double Precision;
declare variable BoyaCarpani           Double Precision;
declare variable PatCarpani            Double Precision;
declare variable Carpan                Double Precision;
declare variable SablonReceteHesapSekli integer;
declare variable SablonDolulukOrani    Double Precision;
declare variable DesenVaryantId        integer;
declare variable HesaplananPatMiktari  Double Precision;
declare variable KullanilanBoyaMiktari Double Precision;
begin
  Select SablonReceteHesapSekli from sistem where sistemid = 1 into :SablonReceteHesapSekli;

  Select Parti.DesenvaryantId from parti
    Where Parti.PartiKayitId = :APartiKayitId
  into :DesenvaryantId;

  For Select PartiSablon.EmulsiyonUrunId, PartiSablon.PartiSablonId, PartiSablon.Acilim1, PartiSablon.Acilim2, PartiSablon.PatMiktari, PartiSablon.HazirlananMiktar, PartiSablon.BoyaToplami,
             DesenVaryantSablon.DolulukOrani, UrunBirim.UrunBirimId from PartiSablon
    left join DesenVaryantSablon on DesenVaryantSablon.DesenVaryantID = :DesenvaryantId and DesenVaryantSablon.SablonNo = PartiSablon.SablonNo
    left join UrunBirim on UrunBirim.UrunId = PartiSablon.EmulsiyonUrunId and UrunBirim.ReceteDefaultBirim = 1
    where PartiSablon.PartiKayitId = :APartiKayitId
    into :EmulsiyonUrunId, :PartiSablonId, :Acilim1, :Acilim2, :PatToplami, :HazirlananMiktar, :BoyaToplami, :SablonDolulukOrani, :UrunBirimId do
  begin
    if (SablonReceteHesapSekli = 0) then  /*Tekboy*/
    begin
      AcilimToplami = Acilim1 + Acilim2;

      PatToplami = Acilim2 * 1000;
      B1 = BoyaToplami * Acilim1;
      X1 = Acilim1 * 1000;
      PatToplami = PatToplami + X1 - B1;

      if (Acilim1 + Acilim2 <> 0) then
      begin
        BoyaCarpani = B1 / (Acilim1 + Acilim2); /* 1 kg icin kullan覺lacak boya miktar覺 */
        PatCarpani = PatToplami / (Acilim1 + Acilim2); /* 1 kg icin kullan覺lacak Emulsiyon miktar覺 */
      end
      else
      begin
        PatCarpani = 0;
        BoyaCarpani = 1;
      end

      Update PartiSablonRecete Set KullanilanMiktar = Miktar * :HazirlananMiktar * :BoyaCarpani
         where PartiSablonRecete.PartiSablonId = :PartiSablonId;


      if (PatCarpani is not null) then
      if (EmulsiyonUrunId is null) then exception KullanilacakPatSecilmemis;
      else
      begin
        insert into PartiSablonRecete(PartiKayitId, PartiSablonId, UrunId, UrunBirimId, Miktar, KullanilanMiktar, Pat)
                        values(:APartiKayitId, :PartiSablonId, :EmulsiyonUrunId, :UrunBirimId, 0, :HazirlananMiktar * :PatCarpani, 1);
      end
    end
    else if (SablonReceteHesapSekli = 1) then
    begin
      Update PartiSablonRecete Set KullanilanMiktar = Miktar * :HazirlananMiktar
         where PartiSablonRecete.PartiSablonId = :PartiSablonId;

      Select Sum(KullanilanMiktar) from PartiSablonRecete where PartiSablonRecete.PartiSablonID = :PartiSablonID into :KullanilanBoyaMiktari;

      HesaplananPatMiktari = HazirlananMiktar * 1000 - KullanilanBoyaMiktari;

      if (HesaplananPatMiktari > 0) then
      if (EmulsiyonUrunId is null) then exception KullanilacakPatSecilmemis;
      else
      begin
        insert into PartiSablonRecete(PartiKayitId, PartiSablonId, UrunId, UrunBirimId, Miktar, KullanilanMiktar, Pat)
                        values(:APartiKayitId, :PartiSablonId, :EmulsiyonUrunId, :UrunBirimId, 0, :HesaplananPatMiktari, 1);
      end
    end
    else if (SablonReceteHesapSekli = 2) then /*Orkum falan filan*/
    begin
      if (Acilim1 + Acilim2 <> 0) then
      begin
        AcilimToplami = Acilim1 + Acilim2;
      end
      else
      begin
        AcilimToplami = 1;
        Acilim1 = 1;
      end

      Update PartiSablonRecete Set KullanilanMiktar = Miktar / :AcilimToplami * :Acilim1 * :HazirlananMiktar
         where PartiSablonRecete.PartiSablonId = :PartiSablonId;

      Select Sum(KullanilanMiktar) from PartiSablonRecete where PartiSablonRecete.PartiSablonID = :PartiSablonID into :KullanilanBoyaMiktari;

      HesaplananPatMiktari = HazirlananMiktar * 1000 - KullanilanBoyaMiktari;

      if (HesaplananPatMiktari > 0) then
      if (EmulsiyonUrunId is null) then exception KullanilacakPatSecilmemis;
      else
      begin
        insert into PartiSablonRecete(PartiKayitId, PartiSablonId, UrunId, UrunBirimId, Miktar, KullanilanMiktar, Pat)
                        values(:APartiKayitId, :PartiSablonId, :EmulsiyonUrunId, :UrunBirimId, 0, :HesaplananPatMiktari, 1);
      end
    end
    else if (SablonReceteHesapSekli = 3) then /*Baht */
    begin
      if (Acilim1 + Acilim2 <> 0) then
      begin
        AcilimToplami = Acilim1 + Acilim2;
      end
      else
      begin
        AcilimToplami = 1;
        Acilim1 = 1;
      end

      Update PartiSablonRecete Set KullanilanMiktar = Miktar / :AcilimToplami * :Acilim1 * :HazirlananMiktar
         where PartiSablonRecete.PartiSablonId = :PartiSablonId;


      Select Sum(KullanilanMiktar) from PartiSablonRecete where PartiSablonRecete.PartiSablonID = :PartiSablonID into :KullanilanBoyaMiktari;

      HesaplananPatMiktari = PatToplami * 1000 ;

      if (HesaplananPatMiktari > 0) then
      if (EmulsiyonUrunId is null) then exception KullanilacakPatSecilmemis;
      else
      begin
        insert into PartiSablonRecete(PartiKayitId, PartiSablonId, UrunId, UrunBirimId, Miktar, KullanilanMiktar, Pat)
                        values(:APartiKayitId, :PartiSablonId, :EmulsiyonUrunId, :UrunBirimId, 0, :HesaplananPatMiktari, 1);
      end
    end
  end
end
