CREATE PROCEDURE BOYAFATURAKDVGETIR(
    ABOYAFATURAID INTEGER)
RETURNS (
    KDVORANI DOUBLE PRECISION,
    KDVMATRAHI DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION)
AS
declare variable AltIskontoCarpani Double Precision;
declare variable Tevkifat integer;
declare variable AltIskYuvarlamaCarpani Double Precision;
declare variable KdvYuvarlamaCarpani    Double Precision;
declare variable TevkifatOrani1     integer;
declare variable TevkifatOrani2     integer;
declare variable ToplamKdv          Double Precision;
begin
  Select Tevkifat, AltIskontoCarpani, AltIskYuvarlamaCarpani, KdvYuvarlamaCarpani, TevkifatOrani1, TevkifatOrani2 from BoyaFatura
    where BoyaFatura.BoyaFaturaId = :ABoyaFaturaId
    into :Tevkifat, :AltIskontoCarpani, :AltIskYuvarlamaCarpani, :KdvYuvarlamaCarpani, :TevkifatOrani1, :TevkifatOrani2;

  for select Sum(Yuvarla(BoyaFaturaDetay.BirimFiyat * BoyaFaturaDetay.Miktar, :AltIskYuvarlamaCarpani)), BoyaFaturaDetay.KdvOrani from BoyaFaturaDetay
  where BoyaFaturaDetay.BoyaFaturaId = :ABoyaFaturaId
  Group By BoyaFaturaDetay.KdvOrani
  into :KdvMatrahi, :KdvOrani do
  begin
    KdvMatrahi = (KdvMatrahi - KdvMatrahi * AltIskontoCarpani);
    KdvMatrahi = Yuvarla(KdvMatrahi, KdvYuvarlamaCarpani);

    TahsilEdilecekKdv = 0;
    IndirilecekKdv = 0;

    if (Tevkifat = 0) then
    begin
      TahsilEdilecekKdv = KdvMatrahi * KdvOrani / 100;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, KdvYuvarlamaCarpani);
    end
    else if (Tevkifat = 1) then
    begin
      ToplamKdv = KdvMatrahi * KdvOrani / 100;
      ToplamKdv = Yuvarla(ToplamKdv, KdvYuvarlamaCarpani);

      IndirilecekKdv = ToplamKdv * TevkifatOrani1 / TevkifatOrani2;
      IndirilecekKdv = Yuvarla(IndirilecekKdv, KdvYuvarlamaCarpani);

      TahsilEdilecekKdv = ToplamKdv - IndirilecekKdv;

/*
      TahsilEdilecekKdv = KdvMatrahi * KdvOrani / 100;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, KdvYuvarlamaCarpani);

      TahsilEdilecekKdv = TahsilEdilecekKdv * (TevkifatOrani2 - TevkifatOrani1) / TevkifatOrani2;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, KdvYuvarlamaCarpani);

      IndirilecekKdv = KdvMatrahi * KdvOrani / 100;
      IndirilecekKdv = Yuvarla(IndirilecekKdv, KdvYuvarlamaCarpani);

      IndirilecekKdv = IndirilecekKdv * TevkifatOrani1 / TevkifatOrani2;
      IndirilecekKdv = Yuvarla(IndirilecekKdv, KdvYuvarlamaCarpani);
*/
    end
    suspend;
  end
end
