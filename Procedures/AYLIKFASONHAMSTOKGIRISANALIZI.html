CREATE PROCEDURE AYLIKFASONHAMSTOKGIRISANALIZI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICFIRMAGRUP1ID VARCHAR(4),
    ABITISFIRMAGRUP1ID VARCHAR(4),
    ABASLANGICFIRMAGRUP2ID VARCHAR(4),
    ABITISFIRMAGRUP2ID VARCHAR(4),
    ABASLANGICFIRMAGRUP3ID VARCHAR(4),
    ABITISFIRMAGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER)
RETURNS (
    FIRMAADI VARCHAR(100),
    FIRMAKISAADI VARCHAR(10),
    GUN1 DOUBLE PRECISION,
    GUN2 DOUBLE PRECISION,
    GUN3 DOUBLE PRECISION,
    GUN4 DOUBLE PRECISION,
    GUN5 DOUBLE PRECISION,
    GUN6 DOUBLE PRECISION,
    GUN7 DOUBLE PRECISION,
    GUN8 DOUBLE PRECISION,
    GUN9 DOUBLE PRECISION,
    GUN10 DOUBLE PRECISION,
    GUN11 DOUBLE PRECISION,
    GUN12 DOUBLE PRECISION,
    GUN13 DOUBLE PRECISION,
    GUN14 DOUBLE PRECISION,
    GUN15 DOUBLE PRECISION,
    GUN16 DOUBLE PRECISION,
    GUN17 DOUBLE PRECISION,
    GUN18 DOUBLE PRECISION,
    GUN19 DOUBLE PRECISION,
    GUN20 DOUBLE PRECISION,
    GUN21 DOUBLE PRECISION,
    GUN22 DOUBLE PRECISION,
    GUN23 DOUBLE PRECISION,
    GUN24 DOUBLE PRECISION,
    GUN25 DOUBLE PRECISION,
    GUN26 DOUBLE PRECISION,
    GUN27 DOUBLE PRECISION,
    GUN28 DOUBLE PRECISION,
    GUN29 DOUBLE PRECISION,
    GUN30 DOUBLE PRECISION,
    GUN31 DOUBLE PRECISION,
    NORMALKILO DOUBLE PRECISION,
    UCRETLITAMIRKILO DOUBLE PRECISION,
    UCRETSIZTAMIRKILO DOUBLE PRECISION,
    DEGERLENDIRILMEMISTAMIRKILO DOUBLE PRECISION,
    UCRETLITAMIRORAN DOUBLE PRECISION,
    UCRETSIZTAMIRORAN DOUBLE PRECISION,
    TOPLAMKILO DOUBLE PRECISION)
AS
declare variable FFirmaId               varchar(20);
declare variable LastFirmaId            varchar(20);
declare variable FFirmaAdi              varchar(100);
declare variable FFirmaKisaAdi          varchar(10);
declare variable FIrsaliyeTarihi        TimeStamp;
declare variable FKilo                  Double Precision;
declare variable FarkGun                Double Precision;
declare variable FirmaGrup1            varchar(4);
declare variable FirmaGrup2            varchar(4);
declare variable FirmaGrup3            varchar(4);
declare variable Okey integer;
declare variable FTamir integer;
declare variable MinUcretli integer;
declare variable MaxUcretli integer;
declare variable FHamStokMasterId integer;

begin
  LastFirmaId = '';

  For select HamStokIrsaliye.FirmaId, Firma.Adi, Firma.FirmaGrup1,Firma.FirmaGrup2,Firma.FirmaGrup3, HamStokIrsaliye.IrsaliyeTarihi, HamStokMaster.Tamir, HamStokMaster.Kilo, HamStokMaster.HamStokMasterId from HamStokIrsaliye
    left join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
    left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
    where HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and HamStokIrsaliye.Fason = 1 and HamStokIrsaliye.HamStokHareketTipiId = 0 and
        ((:AFirmaAralikli = 1 and (HamStokIrsaliye.FirmaId >= :ABaslangicFirmaId) and (HamStokIrsaliye.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
        ((:AFirmaGrup1Aralikli = 1 and (Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id or :ABaslangicFirmaGrup1Id = '') and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and (Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id or :ABaslangicFirmaGrup2Id = '') and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and (Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id or :ABaslangicFirmaGrup3Id = '') and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = '')) and
        HamStokMaster.HamStokMasterId is not null and
        HamStokMaster.PartiSinifId is null
    order by HamStokIrsaliye.FirmaId, HamStokIrsaliye.IrsaliyeTarihi
    into :FFirmaId, :FFirmaAdi, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :FIrsaliyeTarihi, :FTamir, :FKilo, :FHamStokMasterId Do
  begin
    if (LastFirmaId <> FFirmaId) then
    begin
      if (LastFirmaId <> '' and Okey = 1) then
      begin
        ToplamKilo = NormalKilo + UcretliTamirKilo + UcretsizTamirKilo + DegerlendirilmemisTamirKilo;

        if (ToplamKilo > 0) then UcretliTamirOran = UcretliTamirKilo * 100 / ToplamKilo;
        if (ToplamKilo > 0) then UcretsizTamirOran = UcretsizTamirKilo * 100 / ToplamKilo;
        suspend;
      end

      FirmaAdi = FFirmaAdi;
      FirmaKisaAdi = FFirmaKisaAdi;

      Gun1 = 0; Gun2 = 0; Gun3 = 0;  Gun4 = 0; Gun5 = 0; Gun6 = 0; Gun7 = 0; Gun8 = 0; Gun9 = 0; Gun10 = 0;
      Gun11 = 0; Gun12 = 0; Gun13 = 0;  Gun14 = 0; Gun15 = 0; Gun16 = 0; Gun17 = 0; Gun18 = 0; Gun19 = 0; Gun20 = 0;
      Gun21 = 0; Gun22 = 0; Gun23 = 0;  Gun24 = 0; Gun25 = 0; Gun26 = 0; Gun27 = 0; Gun28 = 0; Gun29 = 0; Gun30 = 0; Gun31 = 0;
      NormalKilo = 0;
      UcretliTamirKilo  = 0;
      UcretsizTamirKilo = 0;
      DegerlendirilmemisTamirKilo = 0;
      UcretliTamirOran  = 0;
      UcretsizTamirOran = 0;

      LastFirmaId = FFirmaId;
      Okey = 1;
      if (AFirmaAralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaId, FFirmaId) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup1Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup1Id, FirmaGrup1) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup2Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup2Id, FirmaGrup2) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup3Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup3Id, FirmaGrup3) = 0) then okey = 0;
    end

    if (Okey = 1) then
    begin
      if (FTamir = 1) then
      begin
        MinUcretli = -1;
        MaxUcretli = -1;

        Select Min(Ucretli), Max(Ucretli) from DisTamirCozum
          left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliID = DisTamirCozum.DisTamirCozumSekliID
          where DisTamirCozum.HamStokMasterId = :FHamStokMasterId
          into :MinUcretli, :MaxUcretli;

        if (MinUcretli is not null) then
        begin
          if (MaxUcretli = 0) then
            UcretliTamirKilo = UcretliTamirKilo + FKilo;
          else
            UcretsizTamirKilo = UcretsizTamirKilo + FKilo;
        end
        else
          DegerlendirilmemisTamirKilo = DegerlendirilmemisTamirKilo + FKilo;
      end
      else
      begin
        FarkGun = :FIrsaliyeTarihi - ABaslangicTarihi + 1;
        if (FarkGun = 1) then Gun1 = Gun1 + FKilo;
        if (FarkGun = 2) then Gun2 = Gun2 + FKilo;
        if (FarkGun = 3) then Gun3 = Gun3 + FKilo;
        if (FarkGun = 4) then Gun4 = Gun4 + FKilo;
        if (FarkGun = 5) then Gun5 = Gun5 + FKilo;
        if (FarkGun = 6) then Gun6 = Gun6 + FKilo;
        if (FarkGun = 7) then Gun7 = Gun7 + FKilo;
        if (FarkGun = 8) then Gun8 = Gun8 + FKilo;
        if (FarkGun = 9) then Gun9 = Gun9 + FKilo;
        if (FarkGun = 10) then Gun10 = Gun10 + FKilo;
        if (FarkGun = 11) then Gun11 = Gun11 + FKilo;
        if (FarkGun = 12) then Gun12 = Gun12 + FKilo;
        if (FarkGun = 13) then Gun13 = Gun13 + FKilo;
        if (FarkGun = 14) then Gun14 = Gun14 + FKilo;
        if (FarkGun = 15) then Gun15 = Gun15 + FKilo;
        if (FarkGun = 16) then Gun16 = Gun16 + FKilo;
        if (FarkGun = 17) then Gun17 = Gun17 + FKilo;
        if (FarkGun = 18) then Gun18 = Gun18 + FKilo;
        if (FarkGun = 19) then Gun19 = Gun19 + FKilo;
        if (FarkGun = 20) then Gun20 = Gun20 + FKilo;
        if (FarkGun = 21) then Gun21 = Gun21 + FKilo;
        if (FarkGun = 22) then Gun22 = Gun22 + FKilo;
        if (FarkGun = 23) then Gun23 = Gun23 + FKilo;
        if (FarkGun = 24) then Gun24 = Gun24 + FKilo;
        if (FarkGun = 25) then Gun25 = Gun25 + FKilo;
        if (FarkGun = 26) then Gun26 = Gun26 + FKilo;
        if (FarkGun = 27) then Gun27 = Gun27 + FKilo;
        if (FarkGun = 28) then Gun28 = Gun28 + FKilo;
        if (FarkGun = 29) then Gun29 = Gun29 + FKilo;
        if (FarkGun = 30) then Gun30 = Gun30 + FKilo;
        if (FarkGun = 31) then Gun31 = Gun31 + FKilo;
        NormalKilo = NormalKilo + FKilo;
      end
    end
  end
  if (FFirmaId <> '' and Okey = 1) then
  begin
    ToplamKilo = NormalKilo + UcretliTamirKilo + UcretsizTamirKilo + DegerlendirilmemisTamirKilo;

    if (ToplamKilo > 0) then UcretliTamirOran = UcretliTamirKilo * 100 / ToplamKilo;
    if (ToplamKilo > 0) then UcretsizTamirOran = UcretsizTamirKilo * 100 / ToplamKilo;
    suspend;
  end
end
