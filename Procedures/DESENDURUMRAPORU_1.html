CREATE PROCEDURE DESENDURUMRAPORU_1(
    ADESENID VARCHAR(20),
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    KAYITTARIHI TIMESTAMP,
    ILKISTEKTARIHI TIMESTAMP,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(80),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SABLONSAYISI INTEGER,
    MUSTERIDESENNO1 VARCHAR(20),
    MUSTERIDESENADI1 VARCHAR(80),
    MUSTERIDESENNO2 VARCHAR(20),
    MUSTERIDESENADI2 VARCHAR(80),
    DESENTASARIMNO TYPE OF COLUMN DESEN.TASARIMNO)
AS
begin
  if (ADesenId is not null) then
  begin
    for Select Desen.InsertTarihi, Desen.IlkIstekTarihi, Desen.DesenId,
           Desen.Adi, Desen.FirmaId, Firma.Adi, Desen.SablonSayisi, Desen.MusteriDesenNo1, Desen.MusteriDesenAdi1,
           Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2, Desen.TasarimNo from Desen
      left join Firma on Firma.FirmaID = Desen.FirmaId
      where Desen.DesenID = :ADesenId
      into :KayitTarihi, :IlkIstekTarihi, :DesenId, :DesenAdi, :FirmaId, :FirmaAdi,
           :SablonSayisi, :MusteriDesenNo1, :MusteriDesenAdi1, :MusteriDesenNo2, :MusteriDesenAdi2, :DesenTasarimNo do
    begin
      suspend;
    end
  end
  else
  if (ARaporTipi = 0) then
  begin
    for Select Desen.InsertTarihi, Desen.IlkIstekTarihi, Desen.DesenId,
           Desen.Adi, Desen.FirmaId, Firma.Adi, Desen.SablonSayisi, Desen.MusteriDesenNo1, Desen.MusteriDesenAdi1,
           Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2, Desen.TasarimNo from Desen
      left join Firma on Firma.FirmaID = Desen.FirmaId
      where Desen.InsertTarihi >= :ABaslangicTarihi and Desen.InsertTarihi <= :ABitisTarihi and
            (Desen.FirmaID = :AFirmaId or :AFirmaId  is null)
      into :KayitTarihi, :IlkIstekTarihi, :DesenId, :DesenAdi, :FirmaId, :FirmaAdi,
           :SablonSayisi, :MusteriDesenNo1, :MusteriDesenAdi1, :MusteriDesenNo2, :MusteriDesenAdi2, :DesenTasarimNo do
    begin
      suspend;
    end
  end
  else
  begin
    for Select Desen.InsertTarihi, Desen.IlkIstekTarihi, Desen.DesenId,
           Desen.Adi, Desen.FirmaId, Firma.Adi, Desen.SablonSayisi, Desen.MusteriDesenNo1, Desen.MusteriDesenAdi1,
           Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2, Desen.TasarimNo from Desen
      left join Firma on Firma.FirmaID = Desen.FirmaId
      where Desen.IlkIstekTarihi >= :ABaslangicTarihi and Desen.IlkIstekTarihi <= :ABitisTarihi and
            (Desen.FirmaID = :AFirmaId or :AFirmaId  is null)
      into :KayitTarihi, :IlkIstekTarihi, :DesenId, :DesenAdi, :FirmaId, :FirmaAdi,
           :SablonSayisi, :MusteriDesenNo1, :MusteriDesenAdi1, :MusteriDesenNo2, :MusteriDesenAdi2, :DesenTasarimNo do
    begin
      suspend;
    end
  end
end
