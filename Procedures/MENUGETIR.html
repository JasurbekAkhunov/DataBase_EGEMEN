CREATE PROCEDURE MENUGETIR(
    AMENUGRUPID INTEGER)
RETURNS (
    SIRAID VARCHAR(30),
    ADI VARCHAR(50),
    TAG INTEGER,
    ISIMSIRA INTEGER)
AS
declare variable SiraBoy integer;
begin
  IsimSira = 0;
  SiraBoy = 0;
  for select Menu.SiraId, Menu.Adi, Menu.Tag from MenuGrupElemanlari
    left join Menu on Menu.Tag = MenuGrupElemanlari.Tag
    where MenuGrupElemanlari.MenuGrupId = :AMenuGrupId order by SiraId
    into :SiraId, :Adi, :Tag do
  begin
    if (Tag < 0 or Char_Length(SiraId) <> SiraBoy) then IsimSira = IsimSira + 1;
    suspend;
    if (Tag < 0 or Char_Length(SiraId) <> SiraBoy) then IsimSira = IsimSira + 1;

    SiraBoy = Char_Length(SiraId);
  end
end
