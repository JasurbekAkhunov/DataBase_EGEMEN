CREATE PROCEDURE ASAMAIKAZRAPORU(
    AKULLANICIID INTEGER,
    AIKAZDURUMU INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIASAMALARIID INTEGER,
    ASAMAIKAZTIPI INTEGER,
    ASAMAIKAZTIPISTR VARCHAR(30),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    ASAMABASLAMATARIHI TIMESTAMP,
    ASAMABITISTARIHI TIMESTAMP,
    ONCEKIPARTIASAMALARIID INTEGER,
    ONCEKIASAMAID INTEGER,
    ONCEKIASAMAADI VARCHAR(40),
    ONCEKIASAMABITISTARIHI TIMESTAMP,
    IKAZSURESI DOUBLE PRECISION,
    IKAZGOSTERILDI INTEGER,
    IKAZGOSTERILDISTR VARCHAR(15),
    IKAZYERI INTEGER,
    IKAZYERISTR VARCHAR(12),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMADI VARCHAR(1024),
    KILO NUMERIC(12,3),
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    FASON INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    DESENVARYANTID INTEGER,
    VARYANTADI VARCHAR(30),
    DESENKODUADIVARYANT VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40))
AS
declare variable shtUretimeCikis integer;
begin

  for Select PartiAsamaIkazlari.PartiKayitId, PartiAsamaIkazlari.PartiAsamalariId, PartiAsamaIkazlari.AsamaIkazTipi,
             PartiAsamaIkazlari.PartiYil, PartiAsamaIkazlari.PartiId, PartiAsamaIkazlari.AsamaId,
             PartiAsamaIkazlari.AsamaAdi, PartiAsamaIkazlari.AsamaBaslamaTarihi, PartiAsamaIkazlari.AsamaBitisTarihi,
             PartiAsamaIkazlari.OncekiPartiAsamalariId, PartiAsamaIkazlari.OncekiAsamaId, PartiAsamaIkazlari.OncekiAsamaAdi,
             PartiAsamaIkazlari.OncekiAsamaBitisTarihi, PartiAsamaIkazlari.IkazSuresi, PartiAsamaIkazlari.IkazGosterildi,
             PartiAsamaIkazlari.IkazYeri from PartiAsamaIkazlari
      where (PartiAsamaIkazlari.KullaniciID = :AKullaniciID) and (:AIkazDurumu is Null or PartiAsamaIkazlari.IkazGosterildi = :AIkazDurumu)
      into :PartiKayitId, :PartiAsamalariId, :AsamaIkazTipi, :PartiYil, :PartiId, :AsamaId, :AsamaAdi, :AsamaBaslamaTarihi,
           :AsamaBitisTarihi, :OncekiPartiAsamalariId, :OncekiAsamaId, :OncekiAsamaAdi, :OncekiAsamaBitisTarihi, :IkazSuresi, :IkazGosterildi,
           :IkazYeri do
  begin

    if (AsamaIkazTipi = 0) then AsamaIkazTipiStr = 'Baslamadi ise haber ver';
    else if (AsamaIkazTipi = 1) then AsamaIkazTipiStr = 'Bitmedi ise haber ver';
    else if (AsamaIkazTipi = 2) then AsamaIkazTipiStr = 'Baslayinca Haber ver';
    else if (AsamaIkazTipi = 3) then AsamaIkazTipiStr = 'Bitince Haber ver';

    if (IkazGosterildi = 0) then IkazGosterildiStr = 'Gosterilmedi';
    else IkazGosterildiStr = 'Gösterildi';

    if (IkazYeri = 0) then IkazYeriStr = 'Bilgisayar';
    else if (IkazYeri = 1) then IkazYeriStr = 'Mail';
    else if (IkazYeri = 2) then IkazYeriStr = 'Sms';
    else if (IkazYeri = 3) then IkazYeriStr = 'Fax';

    Select PartiIdStr, HamAdi, FirmaID, FirmaAdi, Kilo, BoyaSiparisIdStr, SiparisGrupId, SiparisGrupAdi, SiparisGrup2Id,
           SiparisGrup2Adi, Fason, IslemTipiAdi, IslemGrubuAdi, DesenId, DesenAdi, DesenVaryantId, VaryantAdi, DesenKoduAdiVaryant,
           ReceteId, ReceteAdi from PartiBilgisigetir(:PartiYil, :PartiId, :AKullaniciId)
    into :PartiIdStr, :HamAdi, :FirmaID, :FirmaAdi, :Kilo, :BoyaSiparisIdStr, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id,
           :SiparisGrup2Adi, :Fason, :IslemTipiAdi, :IslemGrubuAdi, :DesenId, :DesenAdi, :DesenVaryantId, :VaryantAdi, :DesenKoduAdiVaryant,
           :ReceteId, :ReceteAdi;


    Suspend;
  end
end
