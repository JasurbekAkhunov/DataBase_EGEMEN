CREATE PROCEDURE MAMULDEPODANHAMDEPOYATRANSFERET(
    APARTISTOKTARTIIDLER VARCHAR(1000),
    AMAMULDEPOTRANSFERFISIID INTEGER,
    ARAFID RAFKODUSIZE,
    AKULLANICIID INTEGER)
RETURNS (
    FHAMID INTEGER)
AS
declare variable FPartiKayitId integer;
declare variable FPartiSinifId integer;
declare variable FPartiStokTartiId integer;
--declare variable FHamId        integer;
declare variable FMetre        BoyaMiktarSize;
declare variable FKilo         BoyaMiktarSize;
declare variable Aciklama           varchar(30);
declare variable FirmaKodumuz       FirmaKoduSize;
declare variable SistemDovizId      DovizSize;
declare variable HamStokIrsaliyeId integer;
declare variable FFisSeriNo         varchar(2);
declare variable FFisNo             integer;
declare variable FFason             EgeBoolean;
declare variable FFirmaID           FirmaKoduSize;
declare variable HamStokMasterId    integer;
declare Variable FHSSarfFisiYil integer;
declare Variable FHSSarfFisiNo  integer;
declare Variable FHSSarfFisiId  integer;
declare variable SarfTipiId     integer;
declare variable SarfAdi  varchar(20);
declare stmt varchar(2000);
declare stmt2 varchar(2000);
declare variable FIlkYikamaOncesiEn       Double Precision;
declare variable FIlkYikamaOncesiGramaj   Double Precision;
declare variable FBirimfiyat  double precision;
declare variable FDovizId varchar(5);
declare variable FKullaniciAdi varchar(40);
declare variable FOrmeciId integer;
begin
  stmt =  'select PartiStok.PartiSinifId, PartiStok.HamId, PartiStok.PartiKayitId, Parti.Fason, Parti.FirmaId, Sum(PartiStokTarti.Metre), Sum(PartiStokTarti.Kilo) from PartiStokTarti
  left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  where PartiStokTarti.PartiStokTartiId in ('||APartiStokTartiIdler||')
  group by PartiStok.PartiSinifId, PartiStok.PartiKayitId, PartiStok.HamId, Parti.Fason, Parti.FirmaId';

  Aciklama = 'Mamul Depodan Transfer';

  Select FirmaKodumuz, SistemDovizId from sistem where sistemid = 1 into :FirmaKodumuz, :SistemDovizId;

  /*sarf tipi olustur*/
  SarfAdi = 'HAMSTOKA TRANSFER';
  SarfTipiId = null;
  select Max(SarfTipi.SarfTipiId) from SarfTipi where SarfTipi.Adi = :SarfAdi into :SarfTipiId;
  if (SarfTipiId is null) then
  begin
    insert into SarfTipi (Adi) values (:SarfAdi);
    SarfTipiId = gen_id(SarfTipi_Id_Gen, 0);
  end
  /*sarf fis olustur*/
  FHSSarfFisiYil = Extract(year from Current_date);
  select id from HamStokSarfFisiNumaratoruAl(:FHSSarfFisiYil) into :FHSSarfFisiNo;
  FHSSarfFisiId = gen_id(HamStokSarfFisi_Id_Gen, 1);
  insert into HamStokSarfFisi(HamStokSarfFisiId, FisYil, FisNo, FisTarihi, SarfTipiId, Notu, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
  values(:FHSSarfFisiId, :FHSSarfFisiYil, :FHSSarfFisiNo, Current_date, :SarfTipiId, :SarfAdi, 1, current_timestamp(0), 1, current_timestamp(0));

  for execute statement(stmt) into :FPartiSinifId, :FHamId, :FPartiKayitId, :FFason, :FFirmaId, :FMetre, :FKilo do
  begin
    if (FFason = 0) then
      FFirmaId = FirmaKodumuz;

    /*Hamstok giris irsaliyesi olustur*/
    HamStokIrsaliyeId = Gen_ID(HamStokIrsaliye_Id_Gen, 1);

    Select FisSeriNo from Kullanici where KullaniciId = 1 into :FFisSeriNo;
    if (FFisSeriNo is null) then FFisSeriNo = '';

    select FisNo from FisNumaratorAl(:FFisSeriNo) into :FFisNo;
    update FisNumarator set FisNo = :FFisNo where SeriNo = :FFisSeriNo;

		 -- Arta İstek
     Select first 1 BoyaSiparisDetay.birimfiyat , BoyaSiparisDetay.DovizId, Kullanici.Adi, PARTISTOKLOTBILGISI.OrmeciId From PartiStok
                    Left Join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayID
                    Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitID
                    Left Join Kullanici on Kullanici.KullaniciID = BoyaSiparis.SiparisiAlanKullaniciId
                    Left Join PARTISTOKLOTBILGISI(PartiStok.PartiSinifId) on 1 = 1
                 where PartiStok.PartiSinifid = :FPartiSinifId
                 into :FBirimfiyat, :FDovizId, :FKullaniciAdi, :FOrmeciId;
		 -- Arta İstek
   insert into HamStokIrsaliye(HamStokIrsaliyeId, HamStokHareketTipiId, FirmaId, EvrakTipi, IrsaliyeSeriNo, IrsaliyeNo, IrsaliyeTarihi, FiiliSevkTarihi, DovizId, Sinif, Notu, Fason, KullaniciId, InsertKullaniciId,OrmeciID)
   values(:HamStokIrsaliyeId, 0, :FFirmaId, 0, :FFisSeriNo, :FFisNo, current_date, current_date, :FDovizId, 'B', :Aciklama, :FFason, :AKullaniciId, :AKullaniciId,:FOrmeciId);

    /*HamStokMaster ekle*/
    HamStokMasterId = gen_Id(HamStokMaster_Id_gen, 1);

    select IlkYikamaOncesiEn, IlkYikamaOncesiGramaj from PartiStokGirisBilgisiGetir(:FPartiSinifId) into :FIlkYikamaOncesiEn, :FIlkYikamaOncesiGramaj;

     insert into HamStokMaster(HamStokIrsaliyeId, HamStokMasterId, StokFirmaId, HamId, Kilo, Metre, RenkId, Renk, Aciklama, KaliteId, KullaniciId, InsertKullaniciId, YikamaOncesiEn, YikamaOncesiGramaj,BirimFiyat,EntegrasyonEkBilgi)
    Select :HamStokIrsaliyeId, :HamStokMasterId, Null, PartiStok.HamId, :FKilo, :FMetre, null, null, :Aciklama, BoyaSiparisDetay.KaliteId, :AKullaniciId, :AKullaniciId, :FIlkYikamaOncesiEn, :FIlkYikamaOncesiGramaj,:FBirimFiyat, :FKullaniciAdi from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiSinifId = :FPartiSinifId;

    /*Gecici kabul rafi olustur*/
    Insert into GeciciKabulRaf(HamStokMasterId, RafId, KapAdedi, Kilo, Metre) Values (:HamStokMasterId, :ARafId, 0, :FKilo, :FMetre);

    /*Ham stok girisi onayla. Bu satir otomatik olarak hamstokdetay ve rafdetay kaydi olusturur*/
    /*select HamStokMaster.Miktar from HamStokMaster where HamStokMaster.HamStokMasterId = :HamStokMasterId into :FKabulMiktar;
    update HamStokMasterKabul set KabulMiktar = :FKabulMiktar, KontrolDurumu = 1,  KontrolKullaniciId = 1 where HamStokMasterId = :HamStokMasterId;*/

    /*tartiya hangi hamstokmasterla ciktigini yaz*/
    stmt2 = 'update PartiStokTarti set HamStokMasterID = '||HamStokMasterId||' where PartiStokTartiId in ('||APartiStokTartiIdler||') and PartiSinifID = '||FPartiSinifId;
    execute statement(stmt2);

    /*Hamstoka gidenleri tek tek fise ekle ve sarf et*/
    stmt2 = 'select PartiStokTartiId from PartiStokTarti where PartiStokTartiId in ('||APartiStokTartiIdler||') and PartiSinifID = '||FPartiSinifId;
    for execute statement(stmt2) into :FPartiStokTartiId do
    begin
      /*transfer fisi detay olustur*/
      insert into MamulDepoTransferFisiDetay(MamulDepoTransferFisiId, PartiSinifId, PartiStokTartiId, RafId, InsertKullaniciId, KullaniciId)
      values(:AMamulDepoTransferFisiId, :FPartiSinifId, :FPartiStokTartiId, :ARafId, :AKullaniciId, :AKullaniciId);

      /*sarf et*/
      execute procedure PartiStokKapSarfFisiIleSevkEt(:FHSSarfFisiId, :FPartiKayitId, :FPartiSinifId, Null, :FPartiStokTartiId, 1, null,null);
    end

    suspend;
  end
end
