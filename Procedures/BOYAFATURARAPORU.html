CREATE PROCEDURE BOYAFATURARAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASERINO SERINOSIZE5,
    ABASLANGICFATURANO FATURANOSIZE,
    ABITISFATURANO FATURANOSIZE,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICFIRMAGRUP1ID VARCHAR(4),
    ABITISFIRMAGRUP1ID VARCHAR(4),
    ABASLANGICFIRMAGRUP2ID VARCHAR(4),
    ABITISFIRMAGRUP2ID VARCHAR(4),
    ABASLANGICFIRMAGRUP3ID VARCHAR(4),
    ABITISFIRMAGRUP3ID VARCHAR(4),
    ABASLANGICFATURAGRUPID VARCHAR(4),
    ABITISFATURAGRUPID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    AFATURAGRUPARALIKLI INTEGER,
    AENTEGRASYON INTEGER,
    ARAPORDOVIZID VARCHAR(5),
    AFIRMAADI2YIGOREBILIR INTEGER,
    AIHRACATFATURASI INTEGER,
    ASATISTIPI INTEGER,
    ASINIF VARCHAR(1))
RETURNS (
    BOYAFATURAID INTEGER,
    FATURATIPI VARCHAR(30),
    TARIH TIMESTAMP,
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    ODEMETARIHI TIMESTAMP,
    FATURAGRUPID VARCHAR(4),
    FATURAGRUPADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    FIRMAGRUP1 VARCHAR(4),
    FIRMAGRUP2 VARCHAR(4),
    FIRMAGRUP3 VARCHAR(4),
    FIRMATRANSFERKODU TRANSFERKODUSIZE55,
    DOVIZID VARCHAR(5),
    FIRMADOVIZKURUSTR VARCHAR(40),
    TOPLAMMIKTAR DOUBLE PRECISION,
    BRUTTOPLAM DOUBLE PRECISION,
    ISKONTOTOPLAMI DOUBLE PRECISION,
    PARASALISKONTO DOUBLE PRECISION,
    ISKONTOORANI DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    FATURATUTARI DOUBLE PRECISION,
    RAPORFATURATUTARI DOUBLE PRECISION,
    ENTEGRASYONSTR VARCHAR(3),
    TEVKIFATSTR VARCHAR(3),
    DOVIZKURUMESAJ VARCHAR(255),
    IRSALIYENOSTR VARCHAR(255),
    NETFIYATYUVARLAMACARPANI DOUBLE PRECISION,
    DOVIZKURU DOUBLE PRECISION,
    IHRACATFATURASI INTEGER,
    IHRACATFATURASISTR VARCHAR(5),
    NAVLUN DOUBLE PRECISION,
    SIGORTA DOUBLE PRECISION,
    PESINAT DOUBLE PRECISION,
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(40),
    AKREDITIFNO VARCHAR(50),
    EORINO VARCHAR(50),
    GTIPNO VARCHAR(50),
    DAHILIISLEMBELGENO VARCHAR(50),
    KOMISYONCUID INTEGER,
    KOMISYONCUADI VARCHAR(20),
    KOMISYONORANI DOUBLE PRECISION,
    FIRMASEVKADRESIID INTEGER,
    FIRMASEVKADRESIADRESADI VARCHAR(50),
    BANKAHESAPID VARCHAR(20),
    BANKAHESAPADI VARCHAR(40),
    SEVKADRESI1 VARCHAR(40),
    SEVKADRESI2 VARCHAR(40),
    SEVKADRESI3 VARCHAR(40),
    SEVKSEMT VARCHAR(20),
    SEVKSEHIR VARCHAR(20),
    SEVKULKE VARCHAR(20),
    TESLIMSEKLIID INTEGER,
    TESLIMSEKLIADI VARCHAR(40),
    FIRMAULKE VARCHAR(20),
    VERGIHESAPNO VARCHAR(20),
    FIRMADOVIZKURU1 DOUBLE PRECISION,
    FIRMADOVIZKURU2 DOUBLE PRECISION)
AS
declare variable Tevkifat         integer;
declare variable Entegrasyon      integer;
declare variable Okey             integer;
declare variable HareketTipi      integer;

declare variable DovizKuruTarihi  Date;
declare variable FirmaDovizId     varchar(5);
declare variable IrsaliyeNo       IrsaliyeNoSize;
begin
  For Select BoyaFatura.BoyaFaturaId, BoyaFatura.HareketTipiId, BoyaFatura.FaturaTarihi, BoyaFatura.OdemeTarihi, BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo, BoyaFatura.FaturaGrupId,
      StokHareketGrup.Adi, BoyaFatura.FirmaId, Firma.Adi, Firma.Adi2, Firma.DovizId, Firma.FirmaGrup1, Firma.FirmaGrup2, Firma.FirmaGrup3,  BoyaFatura.DovizId, BoyaFatura.DovizKuru1,
      BoyaFatura.DovizKuru2, BoyaFatura.BrutToplam, BoyaFatura.ParasalIskonto, BoyaFatura.IskontoOrani,
      BoyaFatura.IskontoToplami, BoyaFatura.IndirilecekKdv, BoyaFatura.TahsilEdilecekKdv,
      BoyaFatura.Entegrasyon, BoyaFatura.Tevkifat, BoyaFatura.NetFiyatYuvarlamaCarpani,
      BoyaFatura.IhracatFaturasi, BoyaFatura.Navlun,BoyaFatura.Sigorta,BoyaFatura.Pesinat,
      BoyaFatura.OdemeSekliId,OdemeSekli.Adi,BoyaFatura.AkreditifNo,
      BoyaFatura.EoriNo,BoyaFatura.GtipNo,BoyaFatura.DahiliIslemBelgeNo,
      BoyaFatura.KomisyoncuId,Komisyoncu.Adi,BoyaFatura.KomisyonOrani,
      BoyaFatura.FirmaSevkAdresiId,FirmaSevkAdresi.AdresAdi,FirmaSevkAdresi.SevkAdresi1,FirmaSevkAdresi.SevkAdresi2,FirmaSevkAdresi.SevkAdresi3,
      FirmaSevkAdresi.SevkSemt,FirmaSevkAdresi.SevkSehir,FirmaSevkAdresi.SevkUlke,
      BoyaFatura.BankaHesapId, BankaHesap.Adi, BoyaFatura.TeslimSekliId, TeslimSekli.Adi,
      Sum(BoyaFaturaDetay.Miktar), Firma.TransferKodu, Firma.Ulke,Firma.VergiHesapNo from BoyaFatura
  left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaFaturaId = BoyaFatura.BoyaFaturaId
  left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = BoyaFatura.FaturaGrupId
  left join Firma on Firma.FirmaId = BoyaFatura.FirmaId
  left join OdemeSekli on OdemeSekli.OdemeSekliId = BoyaFatura.OdemeSekliId
  left join Komisyoncu on Komisyoncu.KomisyoncuId = BoyaFatura.KomisyoncuId
  left join FirmaSevkAdresi on FirmaSevkAdresi.FirmaSevkAdresiId = BoyaFatura.FirmaSevkAdresiId
  left join BankaHesap on BankaHesap.BankaHesapId = BoyaFatura.BankaHesapId
  left join TeslimSekli on TeslimSekli.TeslimSekliId = BoyaFatura.TeslimSekliId
  left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
  left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
  where BoyaFatura.FaturaTarihi >= :ABaslangicTarihi and
        BoyaFatura.FaturaTarihi <= :ABitisTarihi and
       (BoyaFatura.FaturaSeriNo = :ASeriNo or :ASeriNo is null) and
       (BoyaFatura.FaturaNo >= :ABaslangicFaturaNo or :ABaslangicFaturaNo = 0) and
       (BoyaFatura.FaturaNo <= :ABitisFaturaNo or :ABitisFaturaNo = 0) and
       ((:AFirmaAralikli = 1 and (BoyaFatura.FirmaId >= :ABaslangicFirmaId or :ABaslangicFirmaId = '') and (BoyaFatura.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
       ((:AFaturaGrupAralikli = 1 and (BoyaFatura.FaturaGrupId >= :ABaslangicFaturaGrupId or :ABaslangicFaturaGrupId = '') and (BoyaFatura.FaturaGrupId <= :ABitisFaturaGrupId or :ABitisFaturaGrupId = '')) or (:AFaturaGrupAralikli = 0)) and
       ((:AFirmaGrup1Aralikli = 1 and (Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id or :ABaslangicFirmaGrup1Id = '') and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
       ((:AFirmaGrup2Aralikli = 1 and (Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id or :ABaslangicFirmaGrup2Id = '') and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
       ((:AFirmaGrup3Aralikli = 1 and (Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id or :ABaslangicFirmaGrup3Id = '') and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = '')) and
       (Parti.Fason = :ASatisTipi or :ASatisTipi is null) and
       (BoyaFatura.Entegrasyon = :AEntegrasyon or :AEntegrasyon is null) and
       (BoyaFatura.IhracatFaturasi = :AIhracatFaturasi or :AIhracatFaturasi is Null) and
       (BoyaFatura.Sinif = :ASinif or :ASinif is null)
   group by BoyaFatura.BoyaFaturaId, BoyaFatura.HareketTipiId, BoyaFatura.FaturaTarihi, BoyaFatura.OdemeTarihi, BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo, BoyaFatura.FaturaGrupId,
            StokHareketGrup.Adi, BoyaFatura.FirmaId, Firma.Adi, Firma.Adi2, Firma.DovizId, Firma.FirmaGrup1, Firma.FirmaGrup2, Firma.FirmaGrup3,  BoyaFatura.DovizId, BoyaFatura.DovizKuru1,
            BoyaFatura.DovizKuru2, BoyaFatura.BrutToplam, BoyaFatura.ParasalIskonto, BoyaFatura.IskontoOrani,
            BoyaFatura.IskontoToplami, BoyaFatura.IndirilecekKdv, BoyaFatura.TahsilEdilecekKdv,
            BoyaFatura.Entegrasyon, BoyaFatura.Tevkifat, BoyaFatura.NetFiyatYuvarlamaCarpani,
            BoyaFatura.IhracatFaturasi,BoyaFatura.Navlun,BoyaFatura.Sigorta,BoyaFatura.Pesinat,
            BoyaFatura.OdemeSekliId,OdemeSekli.Adi,BoyaFatura.AkreditifNo,
            BoyaFatura.EoriNo,BoyaFatura.GtipNo,BoyaFatura.DahiliIslemBelgeNo,
            BoyaFatura.KomisyoncuId,Komisyoncu.Adi,BoyaFatura.KomisyonOrani,
            BoyaFatura.FirmaSevkAdresiId,FirmaSevkAdresi.AdresAdi,FirmaSevkAdresi.SevkAdresi1,FirmaSevkAdresi.SevkAdresi2,FirmaSevkAdresi.SevkAdresi3,
            FirmaSevkAdresi.SevkSemt,FirmaSevkAdresi.SevkSehir,FirmaSevkAdresi.SevkUlke,
            BoyaFatura.BankaHesapId,BankaHesap.Adi, BoyaFatura.TeslimSekliId, TeslimSekli.Adi, Firma.TransferKodu, Firma.Ulke, Firma.VergiHesapNo
  into :BoyaFaturaId, :HareketTipi, :Tarih, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FaturaGrupId, :FaturaGrupAdi, :FirmaId, :FirmaAdi, :FirmaAdi2, :FirmaDovizId,
       :FirmaGrup1, :FirmaGrup2, :FirmaGrup3,
       :DovizId, :FirmaDovizKuru1, :FirmaDovizKuru2, :BrutToplam, :ParasalIskonto, :IskontoOrani,
       :IskontoToplami, :IndirilecekKdv, :TahsilEdilecekKdv, :Entegrasyon, :Tevkifat, :NetFiyatYuvarlamaCarpani,
       :IhracatFaturasi, :Navlun,:Sigorta,:Pesinat,
       :OdemeSekliId,:OdemeSekliAdi,:AkreditifNo,
       :EoriNo,:GtipNo,:DahiliIslemBelgeNo,
       :KomisyoncuId,:KomisyoncuAdi,:KomisyonOrani,
       :FirmaSevkAdresiId,:FirmaSevkAdresiAdresAdi,:SevkAdresi1,:SevkAdresi2,:SevkAdresi3,
       :SevkSemt,SevkSehir,SevkUlke,
       :BankaHesapId,:BankaHesapAdi, TeslimSekliId, TeslimSekliAdi,
       :ToplamMiktar, :FirmaTransferKodu, :FirmaUlke, :VergiHesapNo do
  begin
    if (HareketTipi = 0) then FaturaTipi = 'Boya Satis';
    else if (HareketTipi = 1) then FaturaTipi = 'Ham Stok Alis Iade';
    else FaturaTipi = '';

    if (IhracatFaturasi = 1) then IhracatFaturasiStr = 'Evet';
    else IhracatFaturasiStr = Null;

    Okey = 1;
    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (AFirmaAdi2yiGorebilir = 0) then FirmaAdi2 = NUll;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (okey = 1) then
    begin
      DovizKuruMesaj = Null;
      if (Entegrasyon = 0) then
      begin
       BrutToplam         = 0;
       IskontoToplami     = 0;
       IndirilecekKdv     = 0;
       TahsilEdilecekKdv  = 0;
       FaturaTutari       = 0;
       RaporFaturaTutari  = 0;
       DovizKuru          = 0;
       EntegrasyonStr     = '';
      end
      else
      begin
        EntegrasyonStr = 'Var';
        FaturaTutari = BrutToplam - IskontoToplami + TahsilEdilecekKdv;
        if (DovizId <> ARaporDovizId) then
        begin
          select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :Tarih) into :DovizKuru, :DovizKuruTarihi;
          if (Tarih <> DovizKuruTarihi) then DovizKuruMesaj = 'Fatura Tarihine Ait Kur Girisi Bulunamadigi icin ' || TimeStampToDateTimeStr(DovizKurutarihi) || ' Tarihli Kur ile Hesaplandi';
        end
        else DovizKuru = 1;

        RaporFaturaTutari = FaturaTutari * DovizKuru;
      end

      if (Tevkifat = 0) then TevkifatStr = '';
      else TevkifatStr = 'Var';

      if (FirmaDovizId <> DovizId) then FirmaDovizKuruStr = Format_Float('', FirmaDovizKuru1) || ' ' || DovizId || '-' || Format_Float('', FirmaDovizKuru2) || ' ' || FirmaDovizId;
      else FirmaDovizKuruStr = '';

      IrsaliyeNoStr = '';
      For Select BoyaIrsaliye.IrsaliyeNo from BoyaFaturaDetay
        Left Join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
        left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
        where BoyaFaturaDetay.BoyaFaturaId = :BoyaFaturaId
        group by BoyaIrsaliye.IrsaliyeNo
        into :IrsaliyeNo do
      begin
        if (Char_Length(IrsaliyeNoStr) + Char_Length(IrsaliyeNo) < 253) then
        begin
          if (IrsaliyeNoStr <> '') then IrsaliyeNoStr = left((IrsaliyeNoStr || cast(', ' as varchar(2) character set UTF8)), 255);
          IrsaliyeNoStr = left(IrsaliyeNoStr || IrsaliyeNo, 255);
        end
      end

      suspend;
    end
  end
end
