CREATE PROCEDURE DOVIZKURLARINIGETIR(
    ADOVIZID VARCHAR(5),
    BDOVIZID VARCHAR(5),
    ATARIH TIMESTAMP)
RETURNS (
    DOVIZKURU1 DOUBLE PRECISION,
    DOVIZKURU2 DOUBLE PRECISION)
AS
declare variable RaporDovizKuruTipi integer;
begin
  RaporDovizKuruTipi = rdb$get_context('USER_TRANSACTION', 'Sistem.RaporDovizKuruTipi');

  if (RaporDovizKuruTipi is Null) then
  begin
    Select RaporDovizKuruTipi from sistem where SistemId = 1 into :RaporDovizKuruTipi;
    if (RaporDovizKuruTipi is null) then RaporDovizKuruTipi = 0;
    rdb$set_context('USER_TRANSACTION', 'Sistem.RaporDovizKuruTipi', RaporDovizKuruTipi);
  end

  if (ADovizId = BDovizId) then
  begin
    DovizKuru1 = 1;
    DovizKuru2 = 1;
  end
  else
  begin
    /* Ã¶nceki
      select alisfiati from dovizkuru where tarih = :ATarih and DovizId = :AdovizId into :DovizKuru2;
      if (DovizKuru2 is null) then DovizKuru2 = 0;

      select alisfiati from dovizkuru where tarih = :ATarih and DovizId = :BDovizId into :DovizKuru1;
      if (DovizKuru1 is null) then DovizKuru1 = 0;
    */
    select case when (:RaporDovizKuruTipi = 0) then AlisFiati else
           case when (:RaporDovizKuruTipi = 1) then Satisfiati else
           case when (:RaporDovizKuruTipi = 2) then SerbestAlisFiati else SerbestSatisFiati end end end
           from dovizkuru where tarih = :ATarih and DovizId = :AdovizId into :DovizKuru2;
    if (DovizKuru2 is null) then DovizKuru2 = 0;

    select case when (:RaporDovizKuruTipi = 0) then AlisFiati else
           case when (:RaporDovizKuruTipi = 1) then Satisfiati else
           case when (:RaporDovizKuruTipi = 2) then SerbestAlisFiati else SerbestSatisFiati end end end
           from dovizkuru where tarih = :ATarih and DovizId = :BDovizId into :DovizKuru1;
    if (DovizKuru1 is null) then DovizKuru1 = 0;

  end

  if (DovizKuru1 <> 0 and DovizKuru2 <> 0) then
  if (DovizKuru1 <> 1 and DovizKuru2 <> 1) then
  begin
    if (DovizKuru1 > DovizKuru2) then
    begin
       DovizKuru1 = DovizKuru1 / DovizKuru2;
       DovizKuru2 = 1;
       DovizKuru1 = DovizKuru1 * 1000000 + 0.5;
       DovizKuru1 = Yuvarla(DovizKuru1, 1);
       DovizKuru1 = DovizKuru1 / 1000000;
    end
    else
    begin
       DovizKuru2 = DovizKuru2 / DovizKuru1;
       DovizKuru1 = 1;
       DovizKuru2 = DovizKuru2 * 1000000 + 0.5;
       DovizKuru2 = Yuvarla(DovizKuru2, 1);
       DovizKuru2 = DovizKuru2 / 1000000;
    end
  end
  suspend;
end
