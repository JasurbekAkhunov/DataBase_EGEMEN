CREATE PROCEDURE BUKUMTALIMATICINOSDETAYLARI(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIIDSTR VARCHAR(20),
    HAMADI VARCHAR(255),
    MIKTAR DOUBLE PRECISION)
AS
begin
  for Select BoyaSiparis.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             OrmeSiparisDetay.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, OrmeSiparisDetay.HamMiktar FROM OrmeSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
    LEFT JOIN Ham ON Ham.HamId = OrmeSiparisDetay.HamId
    LEFT JOIN HamTipi ON HamTipi.HamTipiId = Ham.HamTipiId
    Left Join Kullanici IK on IK.KullaniciId = OrmeSiparisDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = OrmeSiparisDetay.KullaniciId
    WHERE BoyaSiparis.BoyaSiparisKayitId = :ABoyaSiparisKayitId
   into  :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
        :HamID, :HamTipiId, :HamTipiIdStr, :HamAdi, :Miktar do
  begin
    suspend;
  end
end
