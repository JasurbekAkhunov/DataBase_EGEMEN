CREATE PROCEDURE BOYAFATURADANFIYATGETIR(
    ABOYAIRSALIYEDETAYID INTEGER)
RETURNS (
    FATURABIRIMFIYAT DOUBLE PRECISION,
    FATURADOVIZID VARCHAR(5))
AS
begin
  Select BoyaFatura.DovizId, Sum(BoyaFaturaDetay.Miktar * BoyaFaturaDetay.BirimFiyat)/Sum(Miktar) from BoyaFaturaDetay
    left join BoyaFatura on BoyaFatura.BoyaFaturaID = BoyaFaturaDetay.BoyaFaturaID
    where BoyaFaturaDetay.BoyaIrsaliyeDetayID = :ABoyaIrsaliyeDetayID and BoyaFaturaDetay.Miktar > 0
    group by BoyaFatura.DovizId
    into :FaturaDovizId, :FaturaBirimFiyat;
  suspend;
end
