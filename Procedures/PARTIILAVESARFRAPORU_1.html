CREATE PROCEDURE PARTIILAVESARFRAPORU_1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APARTIKAYITID INTEGER,
    ASTOKTANDUSULDU INTEGER,
    APARTIGRUPID VARCHAR(4),
    ASIPARISGRUPID VARCHAR(4),
    ARECETEID VARCHAR(20))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTITIPIID INTEGER,
    PARTITIPIIDSTR VARCHAR(10),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    ISLEMTIPIADI VARCHAR(20),
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    HAMADI VARCHAR(255),
    KILO DOUBLE PRECISION,
    BIRLESTIRILMISPKILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    ILAVESARFSAYISI INTEGER,
    TOPLAMILAVESARFSAYISI INTEGER,
    ILKKEZBOYANACAK INTEGER,
    BOYAMAKINAIDLER VARCHAR(50))
AS
declare variable FHamAdi varchar(255);
declare variable EskiLabReceteKodu varchar(20);
declare variable IlkBoyananPartiKayitId integer;
declare variable Ilave integer;
declare variable SokumUzeriBoya integer;
declare variable FBoyaMakinaID varchar(10);
Declare variable FReceteIsEmriYil integer;
Declare variable FReceteIsEmriId  integer;
begin
  if (APartiKayitId is null) then
  begin
    For select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, Parti.AcilisTarihi, IslemTipi.Adi,
               Parti.PartiGrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi, Parti.ReceteId, Recete.Adi,
               Parti.Fason, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, Labrecete.EskiLabReceteKodu,
               Recete.IlkBoyananPartiKayitId, Recete.SokumUzeriBoya, Recete.Ilave, Count(*),
               ReceteIsEmriDetay.ReceteIsEmriYil, ReceteIsEmriDetay.ReceteIsEmriId, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi from PartiAsamaIlaveSarf
    left join PartiAsamalari On PartiAsamalari.PartiAsamalariId = PartiAsamaIlaveSarf.PartiAsamalariid
    left join Parti On Parti.PartiKayitId = PartiAsamalari.PartiKayitId
    left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join Recete On Recete.ReceteId = Parti.ReceteId
    left join PartiGrup On PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join Firma On Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join LabRecete On LabRecete.LabReceteId = Recete.LabReceteId
    left join ReceteIsEmriDetay on ReceteIsEmriDetay.PartiAsamalariID = PartiAsamalari.PartiAsamalariID
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    where PartiAsamaIlaveSarf.Tarih >= :ABaslangictarihi and PartiAsamaIlaveSarf.Tarih <= :ABitisTarihi and
          (PartiAsamaIlaveSarf.StoktanDusuldu = :AStoktanDusuldu or :AStoktanDusuldu is null) and
          (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is Null) and
          (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (Recete.ReceteId = :AReceteId or :AReceteId is null)

    group by Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, Parti.AcilisTarihi, IslemTipi.Adi,
             Parti.PartiGrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi, Parti.ReceteId, Recete.Adi, Parti.Fason,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Labrecete.EskiLabReceteKodu,  Recete.IlkBoyananPartiKayitId,Recete.SokumUzeriBoya, Recete.Ilave,
             ReceteIsEmriDetay.ReceteIsEmriYil, ReceteIsEmriDetay.ReceteIsEmriId, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi
    into :PartiKayitId, :PartiYil, :PartiId, :PartiTipiId, :AcilisTarihi, :IslemTipiAdi, :PartiGrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi,
         :ReceteId, :ReceteAdi, :Fason, :BoyaSiparisYil, :BoyaSiparisId, :EskiLabReceteKodu, :IlkBoyananPartiKayitId, :SokumUzeriBoya, :Ilave, :IlaveSarfSayisi,
         :FReceteIsEmriYil, :FReceteIsEmriId, :SiparisGrupId, :SiparisGrupAdi do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      Select Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from PartiStok where PartiStok.PartiKayitId = :PartiKayitId into :Kilo, :Metre;

      if (PartiTipiId = 0) then PartiTipiIdStr = 'İplik';
      else PartiTipiIdStr = 'Kumaş';

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      Select Count(*) from PartiAsamaIlaveSarf where PartiAsamaIlaveSarf.PartiKayitId = :PartiKayitId into :ToplamIlaveSarfSayisi;

      IlkKezBoyanacak = 0;

      if (SokumUzeriBoya = 0 and Ilave = 0) then
      if (EskiLabReceteKodu is null or EskiLabReceteKodu = '') then
      if (IlkBoyananPartiKayitId is null or (IlkBoyananPartiKayitId = :PartiKayitId)) then IlkKezBoyanacak = 1;

      select Sum(PartiStok.Kilo) from ReceteIsEmriDetay
        left join PartiStok on PartiStok.PartiKayitId = ReceteIsEmriDetay.PartiKayitId
        where ReceteIsEmriDetay.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :FReceteIsEmriId
        into :BirlestirilmisPKilo;

      HamAdi = '';
      For Select Ham.Adi from PartiStok
        left join Ham on Ham.HamId = PartiStok.HamId
        where PartiStok.PartiKayitId = :PartiKayitId
        group by Ham.Adi
        into :FHamAdi do
      begin
        if (Char_Length(HamAdi) + Char_Length(FHamAdi) < 250) then
        begin
          if (HamAdi <> '' ) then HamAdi = HamAdi || cast(', ' as varchar(2) character set UTF8);
          HamAdi = HamAdi || FHamAdi;
        end
      end
      BoyaMakinaIDler = '';
      For Select PartiAsamaStokDetay.MakinaId From PartiAsamaStokDetay
        Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
        Left Join Asama on Asama.AsamaID = PartiAsamaStok.AsamaID
        Where PartiAsamaStokDetay.PartiKayitID = :PartiKayitID and Asama.BoyaAsamasimi = 1
        Group By PartiAsamaStokDetay.MakinaId
        into :FBoyaMakinaID do
      begin
        if (Char_Length(BoyaMakinaIDler) + Char_Length(FBoyaMakinaID) < 50) then
        begin
          if (BoyaMakinaIDler <> '' ) then BoyaMakinaIDler = BoyaMakinaIDler || cast(', ' as varchar(2) character set UTF8);
          BoyaMakinaIDler = BoyaMakinaIDler || FBoyaMakinaID;
        end
      end

      suspend;
    end
  end
  else
  begin
    For select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, Parti.AcilisTarihi, IslemTipi.Adi,
               Parti.PartiGrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi, Parti.ReceteId, Recete.Adi,
               Parti.Fason, BoyaSiparis.BoyaSiparisYil,
               BoyaSiparis.BoyaSiparisID, Labrecete.EskiLabReceteKodu, Recete.IlkBoyananPartiKayitId, Recete.SokumUzeriBoya, Recete.Ilave,  Count(*),
               ReceteIsEmriDetay.ReceteIsEmriYil, ReceteIsEmriDetay.ReceteIsEmriId, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi from PartiAsamaIlaveSarf
    left join PartiAsamalari On PartiAsamalari.PartiAsamalariId = PartiAsamaIlaveSarf.PartiAsamalariid
    left join Parti On Parti.PartiKayitId = PartiAsamalari.PartiKayitId
    left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join Recete On Recete.ReceteId = Parti.ReceteId
    left join PartiGrup On PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join Firma On Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join LabRecete On LabRecete.LabReceteId = Recete.LabReceteId
    left join ReceteIsEmriDetay on ReceteIsEmriDetay.PartiAsamalariID = PartiAsamalari.PartiAsamalariID
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    where PartiAsamaIlaveSarf.PartiKayitID = :APartiKayitID
    group by Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, Parti.AcilisTarihi, IslemTipi.Adi,
             Parti.PartiGrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi, Parti.ReceteId, Recete.Adi, Parti.Fason,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Recete.IlkBoyananPartiKayitId, Recete.SokumUzeriBoya, Recete.Ilave, Labrecete.EskiLabReceteKodu,
             ReceteIsEmriDetay.ReceteIsEmriYil, ReceteIsEmriDetay.ReceteIsEmriId, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi
    into :PartiKayitId, :PartiYil, :PartiId, :PartiTipiId, :AcilisTarihi, :IslemTipiAdi, :PartiGrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi,
         :ReceteId, :ReceteAdi, :Fason, :BoyaSiparisYil, :BoyaSiparisId, :EskiLabReceteKodu,  :IlkBoyananPartiKayitId, :SokumUzeriBoya, :Ilave, :IlaveSarfSayisi,
         :FReceteIsEmriYil, :FReceteIsEmriId, :SiparisGrupId, :SiparisGrupAdi do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      Select Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from PartiStok where PartiStok.PartiKayitId = :PartiKayitId into :Kilo, :Metre;

      if (PartiTipiId = 0) then PartiTipiIdStr = 'İplik';
      else PartiTipiIdStr = 'Kumaş';

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      Select Count(*) from PartiAsamaIlaveSarf where PartiAsamaIlaveSarf.PartiKayitId = :PartiKayitId into :ToplamIlaveSarfSayisi;

      IlkKezBoyanacak = 0;

      if (SokumUzeriBoya = 0 and Ilave = 0) then
      if (EskiLabReceteKodu is null or EskiLabReceteKodu = '') then
      if (IlkBoyananPartiKayitId is null or (IlkBoyananPartiKayitId = :PartiKayitId)) then IlkKezBoyanacak = 1;

      select Sum(PartiStok.Kilo) from ReceteIsEmriDetay
        left join PartiStok on PartiStok.PartiKayitId = ReceteIsEmriDetay.PartiKayitId
        where ReceteIsEmriDetay.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :FReceteIsEmriId
        into :BirlestirilmisPKilo;

      HamAdi = '';
      For Select Ham.Adi from PartiStok
        left join Ham on Ham.HamId = PartiStok.HamId
        where PartiStok.PartiKayitId = :PartiKayitId
        group by Ham.Adi
        into :FHamAdi do
      begin
        if (Char_Length(HamAdi) + Char_Length(FHamAdi) < 250) then
        begin
          if (HamAdi <> '' ) then HamAdi = HamAdi || cast(', ' as varchar(2) character set UTF8);
          HamAdi = HamAdi || FHamAdi;
        end
      end

      BoyaMakinaIDler = '';
      For Select PartiAsamaStokDetay.MakinaId From PartiAsamaStokDetay
        Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
        Left Join Asama on Asama.AsamaID = PartiAsamaStok.AsamaID
        Where PartiAsamaStokDetay.PartiKayitID = :PartiKayitID and Asama.BoyaAsamasimi = 1
        Group By PartiAsamaStokDetay.MakinaId
        into :FBoyaMakinaID do
      begin
        if (Char_Length(BoyaMakinaIDler) + Char_Length(FBoyaMakinaID) < 50) then
        begin
          if (BoyaMakinaIDler <> '' ) then BoyaMakinaIDler = BoyaMakinaIDler || cast(', ' as varchar(2) character set UTF8);
          BoyaMakinaIDler = BoyaMakinaIDler || FBoyaMakinaID;
        end
      end

      suspend;
    end
  end
end
