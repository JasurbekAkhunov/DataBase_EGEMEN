CREATE PROCEDURE RECETEURUNLERITOPTANDUZENLE(
    AESKIURUNID VARCHAR(20),
    AESKIMIKTAR DOUBLE PRECISION,
    AYENIURUNID VARCHAR(20),
    AYENIMIKTAR DOUBLE PRECISION,
    ASABLONLAR INTEGER,
    ARECETELER INTEGER,
    ARECETEASAMASABLONLAR INTEGER)
RETURNS (
    RECETEDETAYID INTEGER,
    RECETEID VARCHAR(20),
    RECETEASAMASABLONDETAYID INTEGER,
    RECETEASAMASABLONID INTEGER)
AS
declare variable FMiktar Double Precision;
declare variable FBanyo Double Precision;
declare variable FUrunBirimId integer;
declare variable FEskiBirimID varchar(2);

begin
  if (AEskiUrunId = '') then AEskiUrunId = Null;
  if (AYeniUrunId = '') then AYeniUrunId = Null;
  if (AEskiUrunId is Null) then exception KayitHatasi ' Eski urun secilmemis...';
  if (AYeniUrunId is Null) then exception KayitHatasi ' Yeni urun secilmemis...';
  if (AEskiMiktar is Null) then AEskiMiktar = 0;
  if (AYeniMiktar is Null) then AYeniMiktar = 0;

  if (ASablonlar = 1) then
  begin
    if (AEskiMiktar > 0) then
    begin
      for select ReceteDetay.ReceteDetayId, ReceteDetay.ReceteId, ReceteDetay.Miktar, ReceteDetay.Banyo, UrunBirim.BirimID from ReceteDetay
        left join Recete on Recete.ReceteId = ReceteDetay.ReceteId
        left join UrunBirim on UrunBirim.UrunBirimId = ReceteDetay.UrunBirimId
        where ReceteDetay.UrunId = :AEskiUrunId and
        (ReceteDetay.Miktar = :AEskiMiktar or ReceteDetay.Banyo = :AEskiMiktar) and
        Recete.Sablon = 1
        into :ReceteDetayId, :ReceteId, :FMiktar, :FBanyo, :FEskiBirimID do
      begin

        Select UrunBirimID from UrunBirim where UrunID = :AYeniUrunId and BirimID = :FEskiBirimID into :FUrunBirimId;

        if (AYeniMiktar > 0) then
        begin
          if (FMiktar > 0) then
            update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Miktar = :AYeniMiktar where ReceteDetayId = :ReceteDetayId;
          else if (FBanyo > 0) then
            update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Banyo = :AYeniMiktar where ReceteDetayId = :ReceteDetayId;
        end
        else
          update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId where ReceteDetayId = :ReceteDetayId;
        Suspend;
      end
    end
    else
    begin
      for select ReceteDetay.ReceteDetayId, ReceteDetay.ReceteId, ReceteDetay.Miktar, ReceteDetay.Banyo, UrunBirim.BirimID from ReceteDetay
        left join Recete on Recete.ReceteId = ReceteDetay.ReceteId
        left join UrunBirim on UrunBirim.UrunBirimId = ReceteDetay.UrunBirimId
        where ReceteDetay.UrunId = :AEskiUrunId and
        Recete.Sablon = 1
        into :ReceteDetayId, :ReceteId, :FMiktar, :FBanyo, :FEskiBirimID do
      begin

        Select UrunBirimID from UrunBirim where UrunID = :AYeniUrunId and BirimID = :FEskiBirimID into :FUrunBirimId;

        if (AYeniMiktar > 0) then
        begin
          if (FMiktar > 0) then
            update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Miktar = :AYeniMiktar where ReceteDetayId = :ReceteDetayId;
          else if (FBanyo > 0) then
            update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Banyo = :AYeniMiktar where ReceteDetayId = :ReceteDetayId;
        end
        else
          update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId where ReceteDetayId = :ReceteDetayId;
        Suspend;
      end
    end
  end
  if (AReceteler = 1) then
  begin
    if (AEskiMiktar > 0) then
    begin
      for select ReceteDetay.ReceteDetayId, ReceteDetay.ReceteId, ReceteDetay.Miktar, ReceteDetay.Banyo, UrunBirim.BirimID from ReceteDetay
        left join Recete on Recete.ReceteId = ReceteDetay.ReceteId
        left join UrunBirim on UrunBirim.UrunBirimId = ReceteDetay.UrunBirimId
        where ReceteDetay.UrunId = :AEskiUrunId and
        (ReceteDetay.Miktar = :AEskiMiktar or ReceteDetay.Banyo = :AEskiMiktar) and
        Recete.Sablon = 0
        into :ReceteDetayId, :ReceteId, :FMiktar, :FBanyo, :FEskiBirimID do
      begin

        Select UrunBirimID from UrunBirim where UrunID = :AYeniUrunId and BirimID = :FEskiBirimID into :FUrunBirimId;

        if (AYeniMiktar > 0) then
        begin
          if (FMiktar > 0) then
            update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Miktar = :AYeniMiktar where ReceteDetayId = :ReceteDetayId;
          else if (FBanyo > 0) then
            update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Banyo = :AYeniMiktar where ReceteDetayId = :ReceteDetayId;
        end
        else
          update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId where ReceteDetayId = :ReceteDetayId;
        Suspend;
      end
    end
    else
    begin
      for select ReceteDetay.ReceteDetayId, ReceteDetay.ReceteId, ReceteDetay.Miktar, ReceteDetay.Banyo, UrunBirim.BirimID from ReceteDetay
        left join Recete on Recete.ReceteId = ReceteDetay.ReceteId
        left join UrunBirim on UrunBirim.UrunBirimId = ReceteDetay.UrunBirimId
        where ReceteDetay.UrunId = :AEskiUrunId and
        Recete.Sablon = 0
        into :ReceteDetayId, :ReceteId, :FMiktar, :FBanyo, :FEskiBirimID do
      begin

        Select UrunBirimID from UrunBirim where UrunID = :AYeniUrunId and BirimID = :FEskiBirimID into :FUrunBirimId;

        if (AYeniMiktar > 0) then
        begin
          if (FMiktar > 0) then
            update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Miktar = :AYeniMiktar where ReceteDetayId = :ReceteDetayId;
          else if (FBanyo > 0) then
            update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Banyo = :AYeniMiktar where ReceteDetayId = :ReceteDetayId;
        end
        else
          update ReceteDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId where ReceteDetayId = :ReceteDetayId;
        Suspend;
      end
    end
  end
  if (AReceteAsamaSablonlar = 1) then
  begin
    ReceteDetayId = null;
    ReceteId  = null;
    if (AEskiMiktar > 0) then
    begin
      for select ReceteAsamaSablonDetay.ReceteAsamaSablonDetayId, ReceteAsamaSablonDetay.ReceteAsamaSablonId, ReceteAsamaSablonDetay.Miktar, ReceteAsamaSablonDetay.Banyo, UrunBirim.BirimID from ReceteAsamaSablonDetay -- receteDetay
        left join UrunBirim on UrunBirim.UrunBirimId = ReceteAsamaSablonDetay.UrunBirimId
        where ReceteAsamaSablonDetay.UrunId = :AEskiUrunId and
        (ReceteAsamaSablonDetay.Miktar = :AEskiMiktar or ReceteAsamaSablonDetay.Banyo = :AEskiMiktar)
        into :ReceteAsamaSablonDetayId, :ReceteAsamaSablonId, :FMiktar, :FBanyo, :FEskiBirimID do
      begin

        Select UrunBirimID from UrunBirim where UrunID = :AYeniUrunId and BirimID = :FEskiBirimID into :FUrunBirimId;

        if (AYeniMiktar > 0) then
        begin
          if (FMiktar > 0) then
            update ReceteAsamaSablonDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Miktar = :AYeniMiktar where ReceteAsamaSablonDetayId = :ReceteAsamaSablonDetayId;
          else if (FBanyo > 0) then
            update ReceteAsamaSablonDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Banyo = :AYeniMiktar where ReceteAsamaSablonDetayId = :ReceteAsamaSablonDetayId;
        end
        else
          update ReceteAsamaSablonDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId where ReceteAsamaSablonDetayId = :ReceteAsamaSablonDetayId;
        Suspend;
      end
    end
    else
    begin
      for select ReceteAsamaSablonDetay.ReceteAsamaSablonDetayId, ReceteAsamaSablonDetay.ReceteAsamaSablonId, ReceteAsamaSablonDetay.Miktar, ReceteAsamaSablonDetay.Banyo, UrunBirim.BirimID from ReceteAsamaSablonDetay
        left join UrunBirim on UrunBirim.UrunBirimId = ReceteAsamaSablonDetay.UrunBirimId
        where ReceteAsamaSablonDetay.UrunId = :AEskiUrunId
        into :ReceteAsamaSablonDetayId, :ReceteAsamaSablonId, :FMiktar, :FBanyo, :FEskiBirimID do
      begin

        Select UrunBirimID from UrunBirim where UrunID = :AYeniUrunId and BirimID = :FEskiBirimID into :FUrunBirimId;

        if (AYeniMiktar > 0) then
        begin
          if (FMiktar > 0) then
            update ReceteAsamaSablonDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Miktar = :AYeniMiktar where ReceteAsamaSablonDetayId = :ReceteAsamaSablonDetayId;
          else if (FBanyo > 0) then
            update ReceteAsamaSablonDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId, Banyo = :AYeniMiktar where ReceteAsamaSablonDetayId = :ReceteAsamaSablonDetayId;
        end
        else
          update ReceteAsamaSablonDetay set UrunId = :AYeniUrunId, UrunBirimId = :FUrunBirimId where ReceteAsamaSablonDetayId = :ReceteAsamaSablonDetayId;
        Suspend;
      end
    end
  end
