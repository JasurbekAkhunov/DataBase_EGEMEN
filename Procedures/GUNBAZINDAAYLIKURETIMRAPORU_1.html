CREATE PROCEDURE GUNBAZINDAAYLIKURETIMRAPORU_1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMTIPIID INTEGER,
    APARTIGRUPID VARCHAR(4))
RETURNS (
    TARIH TIMESTAMP,
    FASONGIRISI NUMERIC(12,3),
    FASONTAMIRGIRISI NUMERIC(12,3),
    TOPLAMGIRIS NUMERIC(12,3),
    TAMIRORANI DOUBLE PRECISION)
AS
declare variable FTamir integer;
declare variable FKilo  integer;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable FLastIrsaliyeTarihi TimeStamp;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  FLastIrsaliyeTarihi = null;

  For Select HamStokDetay.Tarih, HamStokMaster.Tamir, Sum(HamStokDetay.Kilo) from HamStokDetay
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join Ham on Ham.HamID = HamStokDetay.HamId
    where HamStokDetay.Tarih >= :ABaslangicTarihi and HamStokDetay.Tarih <= :ABitisTarihi and HamStokDetay.Fason = 1 and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null) and
          (HamStokMaster.PartiGrupId = :APartiGrupId or :APartiGrupId is null)
    group by HamStokDetay.Tarih, HamStokMaster.Tamir
    into :FIrsaliyeTarihi, :FTamir, :FKilo do
  begin
    if (FLastIrsaliyeTarihi is null or FIrsaliyeTarihi <> FLastIrsaliyeTarihi) then
    begin
      if (FLastIrsaliyeTarihi is not null) then
      begin
        ToplamGiris = FasonGirisi + FasonTamirGirisi;
        if (FasonTamirGirisi <> 0 and ToplamGiris <> 0 ) then
          TamirOrani = FasonTamirGirisi * 100.00 / ToplamGiris;
        suspend;
      end

      FasonGirisi       = 0;
      FasonTamirGirisi  = 0;
      ToplamGiris       = 0;
      TamirOrani   = 0;
      Tarih = FIrsaliyeTarihi;
      FLastIrsaliyeTarihi = FIrsaliyeTarihi;
    end

    if (FTamir = 0) then FasonGirisi = FasonGirisi + FKilo;
    else FasonTamirGirisi = FasonTamirGirisi + FKilo;
  end

  if (FLastIrsaliyeTarihi is not null) then
  begin
    ToplamGiris = FasonGirisi + FasonTamirGirisi;
    if (FasonTamirGirisi <> 0 and ToplamGiris<>0) then
      TamirOrani = FasonTamirGirisi * 100.00 / ToplamGiris;
    suspend;
  end
end
