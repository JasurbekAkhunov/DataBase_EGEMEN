            (Parti.IslemTipiId = :AIslemTipi or :AIslemTipi is null) and
            (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
            (PartiStok.HamId = :AHamId or :AHamId is null) and
            (BoyaSiparisDetay.SiraNo = :ABoyaSipSiraNo or :ABoyaSipSiraNo is null) and
            (PSKaliteKontrolAsama.AktarmaKontrol = :AAktarmaDahil) and
            (BoyaSiparis.Fason = :ASatisTipi or :ASatisTipi is null) and

        left join PartiTamir on PartiTamir.PartiTamirId = PartiTamirSinif.PartiTamirId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiTamir.Hata3PartiAsamalariId
        left join Recete RR on RR.ReceteId = PartiAsamalari.ReceteId
        left join asama on asama.AsamaId = PartiAsamalari.AsamaId or asama.AsamaId = RR.UstAsamaId
        where PartiTamirSinif.PartiSinifId = :PartiSinifId order by PartiTamir.Tarih into :FSebepAsamaAdi  do

        suspend;
