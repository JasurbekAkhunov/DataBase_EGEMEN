CREATE PROCEDURE YKONTROLTALEPISEMIRLERIGETIR(
    AYUZEYKONTROLTALEPKAYITID INTEGER)
RETURNS (
    YKONTROLISEMRIKAYITID INTEGER,
    YKONTROLISEMRIYIL INTEGER,
    YKONTROLISEMRIID INTEGER,
    YKONTROLISEMRIIDSTR VARCHAR(30),
    YKONTROLCIKISFISIDETAYID INTEGER,
    MIKTAR NUMERIC(12,3),
    PLANLAMATARIHI TIMESTAMP,
    MAKINAID VARCHAR(10),
    BASLATANPERSONELID INTEGER,
    BASLATANPERSONELADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENPERSONELID INTEGER,
    BITIRENPERSONELADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select YKontrolIsEmri.YKontrolIsEmriKayitId, YKontrolIsEmri.YKontrolIsEmriYil, YKontrolIsEmri.YKontrolIsEmriId,
             YKontrolIsEmri.YKontrolCikisFisiDetayId, YKontrolIsEmri.Miktar, YKontrolIsEmri.PlanlamaTarihi,
             YKontrolIsEmri.MakinaId, YKontrolIsEmri.BaslatanPersonelId, BasP.Adi, YKontrolIsEmri.BaslamaTarihi,
             YKontrolIsEmri.BitirenPersonelId, BitP.Adi, YKontrolIsEmri.BitisTarihi,
             YKontrolIsEmri.Notu, Ik.Adi, YKontrolIsEmri.InsertTarihi, Sk.Adi, YKontrolIsEmri.DegisimTarihi from YKontrolIsEmri
  left join Personel BasP on BasP.PersonelId = YKontrolIsEmri.BaslatanPersonelId
  left join Personel BitP on BitP.PersonelId = YKontrolIsEmri.BitirenPersonelId
  Left Join Kullanici IK on IK.KullaniciId = YKontrolIsEmri.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = YKontrolIsEmri.KullaniciId
  where YKontrolIsEmri.YuzeyKontrolTalepKayitId = :AYuzeyKontrolTalepKayitId
  into :YKontrolIsEmriKayitId, :YKontrolIsEmriYil, :YKontrolIsEmriId, :YKontrolCikisFisiDetayId, :Miktar,
       :PlanlamaTarihi, :MakinaId, :BaslatanPersonelId, :BaslatanPersonelAdi, :BaslamaTarihi,
       :BitirenPersonelId, :BitirenPersonelAdi, :BitisTarihi, :Notu,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    YKontrolIsEmriIdStr = '(' || EgemenIntegerToStr(YKontrolIsEmriYil) || ') ' || EgemenIntegerToStr(YKontrolIsEmriId) || ' ';
    suspend;
  end
end
