Definition

CREATE PROCEDURE GIRISCIKISTOPLAMLARIRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    FINORMALGIRISTOPLAMI NUMERIC(12,3),
    FITAMIRGIRISTOPLAMI NUMERIC(12,3),
    FKNORMALGIRISTOPLAMI NUMERIC(12,3),
    FKTAMIRGIRISTOPLAMI NUMERIC(12,3),
    SINORMALGIRISTOPLAMI NUMERIC(12,3),
    SITAMIRGIRISTOPLAMI NUMERIC(12,3),
    SKNORMALGIRISTOPLAMI NUMERIC(12,3),
    SKTAMIRGIRISTOPLAMI NUMERIC(12,3),
    FISEVKTOPLAMI NUMERIC(12,3),
    FKSEVKTOPLAMI NUMERIC(12,3),
    SISEVKTOPLAMI NUMERIC(12,3),
    SKSEVKTOPLAMI NUMERIC(12,3),
    FIIADESEVKTOPLAMI NUMERIC(12,3),
    FKIADESEVKTOPLAMI NUMERIC(12,3),
    SIIADESEVKTOPLAMI NUMERIC(12,3),
    SKIADESEVKTOPLAMI NUMERIC(12,3),
    FINORMALGIRISMETRETOPLAMI NUMERIC(12,3),
    FITAMIRGIRISMETRETOPLAMI NUMERIC(12,3),
    FKNORMALGIRISMETRETOPLAMI NUMERIC(12,3),
    FKTAMIRGIRISMETRETOPLAMI NUMERIC(12,3),
    SINORMALGIRISMETRETOPLAMI NUMERIC(12,3),
    SITAMIRGIRISMETRETOPLAMI NUMERIC(12,3),
    SKNORMALGIRISMETRETOPLAMI NUMERIC(12,3),
    SKTAMIRGIRISMETRETOPLAMI NUMERIC(12,3),
    FISEVKMETRETOPLAMI NUMERIC(12,3),
    FKSEVKMETRETOPLAMI NUMERIC(12,3),
    SISEVKMETRETOPLAMI NUMERIC(12,3),
    SKSEVKMETRETOPLAMI NUMERIC(12,3),
    FIIADESEVKMETRETOPLAMI NUMERIC(12,3),
    FKIADESEVKMETRETOPLAMI NUMERIC(12,3),
    SIIADESEVKMETRETOPLAMI NUMERIC(12,3),
    SKIADESEVKMETRETOPLAMI NUMERIC(12,3))
AS
declare variable FHamTipiId integer;
declare variable FFason     integer;
declare variable FTamir     integer;
declare variable FPartiTipiId integer;
declare variable FKilo      Numeric(12,3);
declare variable FMetre      Numeric(12,3);
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable FHareketTipiId integer;
begin
  FINormalGirisToplami = 0;
  FITamirGirisToplami  = 0;
  FKNormalGirisToplami = 0;
  FKTamirGirisToplami  = 0;

  SINormalGirisToplami = 0;
  SITamirGirisToplami  = 0;
  SKNormalGirisToplami = 0;
  SKTamirGirisToplami  = 0;

  FISevkToplami        = 0;
  FKSevkToplami        = 0;
  SISevkToplami        = 0;
  SKSevkToplami        = 0;

  FIIadeSevkToplami    = 0;
  FKIadeSevkToplami    = 0;
  SIIadeSevkToplami    = 0;
  SKIadeSevkToplami    = 0;

  FINormalGirisMetreToplami = 0;
  FITamirGirisMetreToplami  = 0;
  FKNormalGirisMetreToplami = 0;
  FKTamirGirisMetreToplami  = 0;

  SINormalGirisMetreToplami = 0;
  SITamirGirisMetreToplami  = 0;
  SKNormalGirisMetreToplami = 0;
  SKTamirGirisMetreToplami  = 0;

  FISevkMetreToplami        = 0;
  FKSevkMetreToplami        = 0;
  SISevkMetreToplami        = 0;
  SKSevkMetreToplami        = 0;

  FIIadeSevkMetreToplami    = 0;
  FKIadeSevkMetreToplami    = 0;
  SIIadeSevkMetreToplami    = 0;
  SKIadeSevkMetreToplami    = 0;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

/* giri≈üleri cekiyoruz */

  For Select Ham.HamTipiId, HamStokIrsaliye.Fason, HamStokMaster.Tamir, Sum(HamStokMaster.Kilo), Sum(HamStokMaster.Metre) from HamStokIrsaliye
    left join HamStokMaster on HamStokMaster.HamStokIrsaliyeid = HamStokIrsaliye.HamStokIrsaliyeid
    left join Ham on Ham.Hamid = HamStokMaster.HamId
    where HamStokIrsaliye.Irsaliyetarihi >= :ABaslangicTarihi and HamStokIrsaliye.Irsaliyetarihi <= :ABitisTarihi and HamStokMaster.HamStokMasterId is not null
    group by Ham.HamTipiId, HamStokIrsaliye.Fason, HamStokMaster.Tamir
    into :FHamTipiId, :FFason, :FTamir, :FKilo, :FMetre do
  begin
    if (FFason = 1) then
    begin
      if (FHamTipiId = hstIplik or FHamTipiId = hstElyaf or FHamTipiId = hstTelef) then
      begin
        if(FTamir = 0) then FINormalGirisToplami = FINormalGirisToplami + FKilo;
        else FITamirGirisToplami = FITamirGirisToplami + FKilo;

        if(FTamir = 0) then FINormalGirisMetreToplami = FINormalGirisMetreToplami + FMetre;
        else FITamirGirisMetreToplami = FITamirGirisMetreToplami + FMetre;
      end
      else
      begin
        if(FTamir = 0) then FKNormalGirisToplami = FKNormalGirisToplami + FKilo;
        else FKTamirGirisToplami = FKTamirGirisToplami + FKilo;
        if(FTamir = 0) then FKNormalGirisMetreToplami = FKNormalGirisMetreToplami + FMetre;
        else FKTamirGirisMetreToplami = FKTamirGirisMetreToplami + FMetre;
      end
    end
    else
    begin
      if (FHamTipiId = hstIplik or FHamTipiId = hstElyaf or FHamTipiId = hstTelef) then
      begin
        if(FTamir = 0) then SINormalGirisToplami = SINormalGirisToplami + FKilo;
        else SITamirGirisToplami = SITamirGirisToplami + FKilo;
        if(FTamir = 0) then SINormalGirisMetreToplami = SINormalGirisMetreToplami + FMetre;
        else SITamirGirisMetreToplami = SITamirGirisMetreToplami + FMetre;
      end
      else
      begin
        if(FTamir = 0) then SKNormalGirisToplami = SKNormalGirisToplami + FKilo;
        else SKTamirGirisToplami = SKTamirGirisToplami + FKilo;
        if(FTamir = 0) then SKNormalGirisMetreToplami = SKNormalGirisMetreToplami + FMetre;
        else SKTamirGirisMetreToplami = SKTamirGirisMetreToplami + FMetre;
      end
    end
  end

/* Sevkleri cekiyoruz */

  For Select Parti.PartiTipiId, Parti.Fason, Sum(BoyaIrsaliyeDetay.Kilo), Sum(BoyaIrsaliyeDetay.Metre) from BoyaIrsaliye
    left join BoyaIrsaliyedetay on BoyaIrsaliyedetay.BoyaIrsaliyeid = BoyaIrsaliye.BoyaIrsaliyeid
    left join Parti on Parti.PartiKayitId = BoyaIrsaliyedetay.PartiKayitId
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.irsaliyetarihi <= :ABitisTarihi and
          BoyaIrsaliyeDetay.PartiSinifId is not null
    group by Parti.PartiTipiId, Parti.Fason
    into :FPartiTipiId, :FFason, :FKilo, :FMetre do
  begin
    if (FFason = 1) then
    begin
      if (FPartiTipiId = 0) then FISevkToplami = FISevkToplami + FKilo;
      else FKSevkToplami = FKSevkToplami + FKilo;
      if (FPartiTipiId = 0) then FISevkMetreToplami = FISevkMetreToplami + FMetre;
      else FKSevkMetreToplami = FKSevkMetreToplami + FMetre;
    end
    else
    begin
      if (FPartiTipiId = 0) then SISevkToplami = SISevkToplami + FKilo;
      else SKSevkToplami = SKSevkToplami + FKilo;
      if (FPartiTipiId = 0) then SISevkMetreToplami = SISevkMetreToplami + FMetre;
      else SKSevkMetreToplami = SKSevkMetreToplami + FMetre;
    end
  end


/* iade edilenleri cekiyoruz */

  For Select BoyaIrsaliye.HareketTipiId, Ham.HamTipiId, Sum(BoyaIrsaliyeDetay.Kilo), Sum(BoyaIrsaliyeDetay.Metre) from BoyaIrsaliye
    left join BoyaIrsaliyedetay on BoyaIrsaliyedetay.BoyaIrsaliyeid = BoyaIrsaliye.BoyaIrsaliyeid
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.Hamid
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.irsaliyetarihi <= :ABitisTarihi and
          BoyaIrsaliyeDetay.HamIadeRezervId is not null
    group by BoyaIrsaliye.HareketTipiId, Ham.HamTipiId
    into :FHareketTipiId, :FHamTipiId, :FKilo, :FMetre do
  begin
    if (FHareketTipiId = 0) then
    begin
      if (FHamTipiId = HstIplik or FHamTipiId = HstElyaf or FHamTipiId = HstTelef) then FIIadeSevkToplami = FIIadeSevkToplami + FKilo;
      else FKIadeSevkToplami = FKIadeSevkToplami + FKilo;
      if (FHamTipiId = HstIplik or FHamTipiId = HstElyaf or FHamTipiId = HstTelef) then FIIadeSevkMetreToplami = FIIadeSevkMetreToplami + FMetre;
      else FKIadeSevkMetreToplami = FKIadeSevkMetreToplami + FMetre;
    end
    else
    begin
      if (FHamTipiId = HstIplik or FHamTipiId = HstElyaf or FHamTipiId = HstTelef) then SIIadeSevkToplami = SIIadeSevkToplami + FKilo;
      else SKIadeSevkToplami = SKIadeSevkToplami + FKilo;
      if (FHamTipiId = HstIplik or FHamTipiId = HstElyaf or FHamTipiId = HstTelef) then SIIadeSevkMetreToplami = SIIadeSevkMetreToplami + FMetre;
      else SKIadeSevkMetreToplami = SKIadeSevkMetreToplami + FMetre;
    end
  end
  suspend;
end
