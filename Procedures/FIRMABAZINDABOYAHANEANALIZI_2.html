CREATE PROCEDURE FIRMABAZINDABOYAHANEANALIZI_2(
    AFIRMAID VARCHAR(20))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    KISLETMENORMALKILOTOPLAM DOUBLE PRECISION,
    KISLETMETAMIRKILOTOPLAM DOUBLE PRECISION,
    KISLETMENUMUNEKILOTOPLAM DOUBLE PRECISION,
    KISLETMEKONTROLKILOTOPLAM DOUBLE PRECISION,
    KISLETMERPTKILOTOPLAM DOUBLE PRECISION,
    KISLETMENORMALMETRETOPLAM DOUBLE PRECISION,
    KISLETMETAMIRMETRETOPLAM DOUBLE PRECISION,
    KISLETMENUMUNEMETRETOPLAM DOUBLE PRECISION,
    KISLETMEKONTROLMETRETOPLAM DOUBLE PRECISION,
    KISLETMERPTMETRETOPLAM DOUBLE PRECISION,
    IISLETMENORMALKILOTOPLAM DOUBLE PRECISION,
    IISLETMETAMIRKILOTOPLAM DOUBLE PRECISION,
    IISLETMENUMUNEKILOTOPLAM DOUBLE PRECISION,
    IISLETMEKONTROLKILOTOPLAM DOUBLE PRECISION,
    IISLETMERPTKILOTOPLAM DOUBLE PRECISION)
AS
declare variable FlastFirmaId  varchar(20);
declare variable FFirmaId     varchar(20);
declare variable FFirmaAdi    varchar(100);
declare variable FIslemTipiId integer;
declare variable FPartiTipiID integer;
declare variable FKilo        Numeric(12,3);
declare variable FMetre       Numeric(12,3);
begin
  FlastFirmaId = '';

  For Select Parti.FirmaId, Firma.Adi, Parti.IslemTipiId, Parti.PartiTipiId, Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    where (Firma.AnaHesapFirmaID = :AFirmaId or :AFirmaID is null) and
          PartiStok.Durumu <= 7 and PartiStok.PartiStokIptalKayitID is null
    group by Parti.FirmaId, Firma.Adi, Parti.IslemTipiId, Parti.PartiTipiId
    into :FFirmaId, :FFirmaAdi, :FIslemTipiId, :FPartiTipiId, :FKilo, :FMetre do
  begin
    if (FLastFirmaID <> FFirmaID) then
    begin
      if (FLastFirmaID <> '') then suspend;
      FirmaID = FFirmaId;
      FLastFirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;

      KIsletmeNormalKiloToplam   = 0;
      KIsletmeRptKiloToplam      = 0;
      KIsletmeKontrolKiloToplam  = 0;
      KIsletmeTamirKiloToplam    = 0;
      KIsletmeNumuneKiloToplam   = 0;

      KIsletmeNormalMetreToplam   = 0;
      KIsletmeRptMetreToplam      = 0;
      KIsletmeKontrolMetreToplam  = 0;
      KIsletmeTamirMetreToplam    = 0;
      KIsletmeNumuneMetreToplam   = 0;

      IIsletmeNormalKiloToplam   = 0;
      IIsletmeRptKiloToplam      = 0;
      IIsletmeKontrolKiloToplam  = 0;
      IIsletmeTamirKiloToplam    = 0;
      IIsletmeNumuneKiloToplam   = 0;
    end

    if (FPartiTipiID = 0) then
    begin
      if (FIslemTipiID = 0) then IIsletmeNormalKiloToplam = IIsletmeNormalKiloToplam + FKilo;
      else if (FIslemTipiID = 1) then IIsletmeRptKiloToplam = IIsletmeRptKiloToplam + FKilo;
      else if (FIslemTipiID = 2) then IIsletmeKontrolKiloToplam = IIsletmeKontrolKiloToplam + FKilo;
      else if (FIslemTipiID = 3) then IIsletmeTamirKiloToplam = IIsletmeTamirKiloToplam + FKilo;
      else if (FIslemTipiID = 4) then IIsletmeNumuneKiloToplam = IIsletmeNumuneKiloToplam + FKilo;
    end
    else
    begin
      if (FIslemTipiID = 0) then KIsletmeNormalKiloToplam = KIsletmeNormalKiloToplam + FKilo;
      else if (FIslemTipiID = 1) then KIsletmeRptKiloToplam = KIsletmeRptKiloToplam  + FKilo;
      else if (FIslemTipiID = 2) then KIsletmeKontrolKiloToplam = KIsletmeKontrolKiloToplam + FKilo;
      else if (FIslemTipiID = 3) then KIsletmeTamirKiloToplam = KIsletmeTamirKiloToplam + FKilo;
      else if (FIslemTipiID = 4) then KIsletmeNumuneKiloToplam = KIsletmeNumuneKiloToplam + FKilo;

      if (FIslemTipiID = 0) then KIsletmeNormalMetreToplam = KIsletmeNormalMetreToplam + FMetre;
      else if (FIslemTipiID = 1) then KIsletmeRptMetreToplam = KIsletmeRptMetreToplam + FMetre;
      else if (FIslemTipiID = 2) then KIsletmeKontrolMetreToplam = KIsletmeKontrolMetreToplam + FMetre;
      else if (FIslemTipiID = 3) then KIsletmeTamirMetreToplam = KIsletmeTamirMetreToplam + FMetre;
      else if (FIslemTipiID = 4) then KIsletmeNumuneMetreToplam = KIsletmeNumuneMetreToplam + FMetre;
    end
  end

  if (FLastFirmaID <> '') then suspend;
end
