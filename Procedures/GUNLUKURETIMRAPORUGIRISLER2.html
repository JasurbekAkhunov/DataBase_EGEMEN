Definition

CREATE PROCEDURE GUNLUKURETIMRAPORUGIRISLER2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FASON INTEGER,
    ORMEGIRISIRSALIYEDOLU INTEGER,
    IPLIKGIRISI NUMERIC(12,3),
    IPLIKGIRISIUCRETLITAMIR NUMERIC(12,3),
    IPLIKGIRISIUCRETSIZTAMIR NUMERIC(12,3),
    KUMASGIRISI NUMERIC(12,3),
    KUMASGIRISIUCRETLITAMIR NUMERIC(12,3),
    KUMASGIRISIUCRETSIZTAMIR NUMERIC(12,3),
    AYLIKIPLIKGIRISI NUMERIC(12,3),
    AYLIKIPLIKGIRISIUCRETLITAMIR NUMERIC(12,3),
    AYLIKIPLIKGIRISIUCRETSIZTAMIR NUMERIC(12,3),
    AYLIKKUMASGIRISI NUMERIC(12,3),
    AYLIKKUMASGIRISIUCRETLITAMIR NUMERIC(12,3),
    AYLIKKUMASGIRISIUCRETSIZTAMIR NUMERIC(12,3),
    IAIPLIKGIRISI NUMERIC(12,3),
    IAIPLIKGIRISIUCRETLITAMIR NUMERIC(12,3),
    IAIPLIKGIRISIUCRETSIZTAMIR NUMERIC(12,3),
    IAKUMASGIRISI NUMERIC(12,3),
    IAKUMASGIRISIUCRETLITAMIR NUMERIC(12,3),
    IAKUMASGIRISIUCRETSIZTAMIR NUMERIC(12,3),
    IAAYLIKIPLIKGIRISI NUMERIC(12,3),
    IAAYLIKIPLIKGIRISIUCRETLITAMIR NUMERIC(12,3),
    IAAYLIKIPLIKGIRISIUCRETSIZTAMIR NUMERIC(12,3),
    IAAYLIKKUMASGIRISI NUMERIC(12,3),
    IAAYLIKKUMASGIRISIUCRETLITAMIR NUMERIC(12,3),
    IAAYLIKKUMASGIRISIUCRETSIZTAMIR NUMERIC(12,3))
AS
declare variable FFirmaId               varchar(20);
declare variable FFirmaAdi              varchar(100);
declare variable FIrsaliyeTarihi        TimeStamp;
declare variable FHamTipiId             integer;
declare variable FFason                 integer;
declare variable FTamir                 integer;
declare variable FUcretli               integer;
declare variable FKilo                  Numeric(12,3);
declare variable FLastFirmaId           varchar(20);
declare variable FLastFason             integer;
declare variable FOrmeGIrsaliyeDolu integer;
declare variable FLastOrmeGIrsaliyeDolu integer;
declare variable FHamStokIadeSebebiId   integer;
declare variable FHamStokHareketTipiId   integer;
begin
  FLastFirmaId = '@@@@@';
  FLastFason = -1;
  FLastOrmeGIrsaliyeDolu = -1;

  For Select HamStokMaster.FirmaId, Firma.Adi, HamStokMaster.Fason, HamStokMaster.Tarih, Ham.HamTipiId, HamStokMaster.Tamir, HamStokMaster.Kilo,
      iif(Char_Length(HamStokMasterOrmeGirisiGetir.OrmeGirisIrsaliyeNo) > 0, 1, 0), iif(HamStokMaster.HamStokIadeSebebiId is not Null, 1, 0),
      Max(DisTamirCozumSekli.Ucretli), HamStokMaster.HamStokHareketTipiId from HamStokMaster
    left join Ham on Ham.HamID = HamStokMaster.HamId
    left join Firma on Firma.FirmaID = HamStokMaster.FirmaId
    left join DisTamirCozum on DisTamirCozum.HamStokMasterID = HamStokMaster.HamStokMasterId
    left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliID = DisTamirCozum.DisTamirCozumSekliId
    left join HamStokMasterOrmeGirisiGetir(HamStokMaster.HamStokMasterId) on HamStokMasterOrmeGirisiGetir.OrmeGirisIrsaliyeNo is not Null
    where HamStokMaster.Tarih >= :AAylikBaslangic and HamStokMaster.Tarih <= :AAylikBitis
    group by HamStokMaster.FirmaId, Firma.Adi, HamStokMaster.Fason, HamStokMaster.HamStokMasterId,  HamStokMaster.Tarih, Ham.HamTipiId, HamStokMaster.Tamir, HamStokMaster.Kilo,
      iif(Char_Length(HamStokMasterOrmeGirisiGetir.OrmeGirisIrsaliyeNo) > 0, 1, 0), iif(HamStokMaster.HamStokIadeSebebiId is not Null, 1, 0), HamStokMaster.HamStokHareketTipiId
    order by HamStokMaster.FirmaId, HamStokMaster.Fason, iif(Char_Length(HamStokMasterOrmeGirisiGetir.OrmeGirisIrsaliyeNo) > 0, 1, 0)
    into :FFirmaId, FFirmaAdi, :FFason, :FIrsaliyeTarihi, :FHamTipiId, :FTamir, :FKilo,
      :FOrmeGIrsaliyeDolu, :FHamStokIadeSebebiId,
      :FUcretli, :FHamStokHareketTipiId do
  begin
    if (FFirmaID <> FLastFirmaID or FFason <> FLastFason or FOrmeGIrsaliyeDolu <> FLastOrmeGIrsaliyeDolu) then
    begin
      if (FirmaID is not null) then suspend;

      IplikGirisi               = 0;
      IplikGirisiUcretliTamir   = 0;
      IplikGirisiUcretsizTamir  = 0;

      KumasGirisi               = 0;
      KumasGirisiUcretliTamir   = 0;
      KumasGirisiUcretsizTamir  = 0;

      AylikIplikGirisi              = 0;
      AylikIplikGirisiUcretliTamir  = 0;
      AylikIplikGirisiUcretsizTamir = 0;

      AylikKumasGirisi              = 0;
      AylikKumasGirisiUcretliTamir  = 0;
      AylikKumasGirisiUcretsizTamir = 0;

      IaIplikGirisi               = 0;
      IaIplikGirisiUcretliTamir   = 0;
      IaIplikGirisiUcretsizTamir  = 0;

      IaKumasGirisi               = 0;
      IaKumasGirisiUcretliTamir   = 0;
      IaKumasGirisiUcretsizTamir  = 0;

      IaAylikIplikGirisi              = 0;
      IaAylikIplikGirisiUcretliTamir  = 0;
      IaAylikIplikGirisiUcretsizTamir = 0;

      IaAylikKumasGirisi              = 0;
      IaAylikKumasGirisiUcretliTamir  = 0;
      IaAylikKumasGirisiUcretsizTamir = 0;

      FirmaID = FFirmaID;
      FirmaAdi = FFirmaAdi;
      FLastFirmaID = FFirmaID;
      Fason = FFason;
      FLastFason = FFason;
      OrmeGirisIrsaliyeDolu = FOrmeGIrsaliyeDolu;
      FLastOrmeGIrsaliyeDolu = FOrmeGIrsaliyeDolu;
    end

    if (FHamTipiId = 0) then
    begin
      if (FTamir = 0) then
      begin
        AylikIplikGirisi = AylikIplikGirisi + FKilo;
        if (FHamStokIadeSebebiId = 1 or FHamStokHareketTipiId = 1) then IaAylikIplikGirisi = IaAylikIplikGirisi + FKilo;
      end
      else
      begin
        if (FUcretli = 0) then AylikIplikGirisiUcretsizTamir = AylikIplikGirisiUcretsizTamir + FKilo;
        else AylikIplikGirisiUcretliTamir = AylikIplikGirisiUcretliTamir + FKilo;
        if (FHamStokIadeSebebiId = 1 or FHamStokHareketTipiId = 1) then
        begin
          if (FUcretli = 0) then IaAylikIplikGirisiUcretsizTamir = IaAylikIplikGirisiUcretsizTamir + FKilo;
          else IaAylikIplikGirisiUcretliTamir = IaAylikIplikGirisiUcretliTamir + FKilo;
        end
      end
    end
    else
    begin
      if (FTamir = 0) then
      begin
        AylikKumasGirisi = AylikKumasGirisi + FKilo;
        if (FHamStokIadeSebebiId = 1 or FHamStokHareketTipiId = 1) then IaAylikKumasGirisi = IaAylikKumasGirisi + FKilo;
      end
      else
      begin
        if (FUcretli = 0) then AylikKumasGirisiUcretsizTamir = AylikKumasGirisiUcretsizTamir + FKilo;
        else AylikKumasGirisiUcretliTamir = AylikKumasGirisiUcretliTamir + FKilo;
        if (FHamStokIadeSebebiId = 1 or FHamStokHareketTipiId = 1) then
        begin
          if (FUcretli = 0) then IaAylikKumasGirisiUcretsizTamir = IaAylikKumasGirisiUcretsizTamir + FKilo;
          else IaAylikKumasGirisiUcretliTamir = IaAylikKumasGirisiUcretliTamir + FKilo;
        end
      end
    end

    if (FIrsaliyeTarihi >= ABaslangicTarihi and FIrsaliyeTarihi <= ABitisTarihi) then
    begin
      if (FHamTipiId = 0) then
      begin
        if (FTamir = 0) then
        begin
          IplikGirisi = IplikGirisi + FKilo;
          if (FHamStokIadeSebebiId = 1 or FHamStokHareketTipiId = 1) then IaIplikGirisi = IaIplikGirisi + FKilo;
        end
        else
        begin
          if (FUcretli = 0) then IplikGirisiUcretsizTamir = IplikGirisiUcretsizTamir + FKilo;
          else IplikGirisiUcretliTamir = IplikGirisiUcretliTamir + FKilo;
          if (FHamStokIadeSebebiId = 1 or FHamStokHareketTipiId = 1) then
          begin
            if (FUcretli = 0) then IaIplikGirisiUcretsizTamir = IaIplikGirisiUcretsizTamir + FKilo;
            else IaIplikGirisiUcretliTamir = IaIplikGirisiUcretliTamir + FKilo;
          end
        end
      end
      else
      begin
        if (FTamir = 0) then
        begin
          KumasGirisi = KumasGirisi + FKilo;
          if (FHamStokIadeSebebiId = 1 or FHamStokHareketTipiId = 1) then IaKumasGirisi = IaKumasGirisi + FKilo;
        end
        else
        begin
          if (FUcretli = 0) then KumasGirisiUcretsizTamir = KumasGirisiUcretsizTamir + FKilo;
          else KumasGirisiUcretliTamir = KumasGirisiUcretliTamir + FKilo;
          if (FHamStokIadeSebebiId = 1 or FHamStokHareketTipiId = 1) then
          begin
            if (FUcretli = 0) then IaKumasGirisiUcretsizTamir = IaKumasGirisiUcretsizTamir + FKilo;
            else IaKumasGirisiUcretliTamir = IaKumasGirisiUcretliTamir + FKilo;
          end
        end
      end
    end
  end

  if (FirmaID is not null) then suspend;
end
