    ABARKODID VARCHAR(20),
    APARTIYIL INTEGER,

    TALIMATKODU VARCHAR(20),
    BOYASIPARISNOSTR VARCHAR(30),
    HAMADI ADI255,
    RENKADI VARCHAR(40),
    SIPARISKILO BOYAMIKTARSIZE,
    SIPARISBRUTKILO BOYAMIKTARSIZE,
    FIRMAADI ADI100,
    FIRMAGRUP1 FIRMAGRUPKODUSIZE,
    FISTARIHI TIMESTAMP,
    FISNO INTEGER,
    RAFID RAFKODUSIZE,
    EDILECEKKILO BOYAMIKTARSIZE,
    EDILECEKBRUTKILO BOYAMIKTARSIZE,
    EDILENKILO BOYAMIKTARSIZE,
    EDILENBRUTKILO BOYAMIKTARSIZE,
    PARTIIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    PARTIID INTEGER,
    PARTIYIL INTEGER,
    LABRECETEKODU VARCHAR(20),
    FIRMAGRUP1ADI VARCHAR(30),
    NOTU VARCHAR(1024))
AS
begin
  for select BoyaSiparisSevkRezervRaporu.talimatkodu, BoyaSiparisSevkRezervRaporu.boyasiparisnostr,
    BoyaSiparisSevkRezervRaporu.hamadi, BoyaSiparisSevkRezervRaporu.renkadi,
    BoyaSiparisSevkRezervRaporu.sipariskilo, BoyaSiparisSevkRezervRaporu.siparisbrutkilo,
    BoyaSiparisSevkRezervRaporu.firmaadi, BoyaSiparisSevkRezervRaporu.firmagrup1,
    BoyaRezervFisi.fistarihi, BoyaRezervFisi.fisno, BoyaRezervFisiDetay.rafid, BoyaRezervFisiDetay.edilecekkilo, BoyaRezervFisiDetay.edilecekbrutkilo,
    BoyaRezervFisiDetay.edilenkilo, BoyaRezervFisiDetay.edilenbrutkilo, Parti.PartiId, Parti.PartiYil, BoyaSiparisSevkRezervRaporu.BoyaSiparisId, BoyaSiparisSevkRezervRaporu.BoyaSiparisId, BoyaSiparisSevkRezervRaporu.LabReceteKodu, BoyaSiparisSevkRezervRaporu.Notu, BoyaSiparisSevkRezervRaporu.FirmaGrup1Adi
  from BoyaSiparisSevkRezervRaporu(:AImalatTipiId, :ABaslangicTarihi, :ABitisTarihi, :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AFirmaYetkiliId, :AIslemTipiId, :ASatisTipi, :ABoyaSiparisTipiId, :ALabReceteId,
                 :AMusteriSiparisNo, :AReferansSiparisNo, :AMusteriKaliteNo, :AOnayDurumu, :APartilenen, :AKapaliDurumu, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASiparisiAlanKullaniciId, :AYurtDisi, :ASiparisGrupId, :ASiparisGrup2Id,
                 :AHamId, :AKullaniciId, :AKaliteId, :ATalimatKodu, :AMakinaId, :ATekrarUretim, :ADesenID, :ARezervDurumu, :ABarkodId, :APartiYil, :APartiId)
  left join BoyaRezervFisiDetay on BoyaRezervFisiDetay.BoyaSiparisDetayId = BoyaSiparisSevkRezervRaporu.BoyaSiparisDetayId
  left join BoyaRezervFisi on BoyaRezervFisi.BoyaRezervFisiId = BoyaRezervFisiDetay.BoyaRezervFisiId
  left join PartiStok on PartiStok.PartiSinifId = BoyaRezervFisiDetay.PartiSinifId
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  into TalimatKodu, BoyaSiparisNoStr, HamAdi, RenkAdi, SiparisKilo, SiparisBrutKilo, FirmaAdi,
  FirmaGrup1, FisTarihi, FisNo, RafId, EdilecekKilo, EdilecekBrutKilo, EdilenKilo, EdilenBrutKilo, PartiId, PartiYil, BoyaSiparisId, BoyaSiparisId, LabReceteKodu, Notu, FirmaGrup1Adi  do
  begin
    PartiIdStr = '(' || PartiYil || ') ' || PartiId;
    suspend;
  end
end
