CREATE PROCEDURE RAFLOKASYONHAREKETRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ARAFID RAFKODUSIZE,
    ARAFLOKASYONID VARCHAR(10))
RETURNS (
    RAFLOKASYONHAREKETID INTEGER,
    RAFID RAFKODUSIZE,
    RAFADI ADI30,
    RAFLOKASYONID VARCHAR(10),
    RAFLOKASYONADI ADI50,
    KULLANICIADI ADI40,
    TARIH TIMESTAMP)
AS
declare variable FBitisTarihi TimeStamp;
begin
    FBitisTarihi = Dateadd(1 day to ABitisTarihi);

    For Select RafLokasyonHareket.RafLokasyonHareketId, RafLokasyonHareket.RafId, Raf.Adi, RafLokasyonHareket.RafLokasyonId, RafLokasyon.Adi,
               Kullanici.Adi, RafLokasyonHareket.InsertTarihi From RafLokasyonHareket
    Left Join  Raf on RafLokasyonHareket.RafId = Raf.RafId
    Left Join  RafLokasyon on RafLokasyon.RafLokasyonId = RafLokasyonHareket.RafLokasyonId
    Left Join  Kullanici on Kullanici.KullaniciId  = RafLokasyonHareket.InsertKullaniciId
    Where (RafLokasyonHareket.InsertTarihi >= :ABaslangicTarihi and RafLokasyonHareket.InsertTarihi <= :FBitisTarihi) and
    (RafLokasyonHareket.RafId = :ARafId or :ARafId is null) and
    (RafLokasyonHareket.RafLokasyonId = :ARafLokasyonId or :ARafLokasyonId is null)
    Order By RafLokasyonHareket.InsertTarihi Desc
    Into :RafLokasyonHareketId, :RafId, :RafAdi, :RafLokasyonId, :RafLokasyonAdi, :KullaniciAdi, :Tarih do
    begin
        suspend;
    end
end
