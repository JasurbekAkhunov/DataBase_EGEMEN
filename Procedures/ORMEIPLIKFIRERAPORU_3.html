CREATE PROCEDURE ORMEIPLIKFIRERAPORU_3(
    ABASLANGICGIRISTARIHI TIMESTAMP,
    ABITISGIRISTARIHI TIMESTAMP,
    AHAMID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ARAFID VARCHAR(10),
    ADEPOID VARCHAR(10),
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    ASTOKGIRISNO INTEGER,
    ALOT VARCHAR(150))
RETURNS (
    ORMERAFID INTEGER,
    ORMEPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    TARIH TIMESTAMP,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    MAKINAUSTUGRAMAJ INTEGER,
    DINLENMISGRAMAJ INTEGER,
    PUS INTEGER,
    FEINE INTEGER,
    MELANJKODU VARCHAR(20),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    SIPTUPACIKENMAYLI INTEGER,
    SIPTUPACIKENMAYLISTR VARCHAR(10),
    TALIMATMIKTARI DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    SAKATURETIMTOPLAMI DOUBLE PRECISION,
    TAHMINIFIREORANI DOUBLE PRECISION,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    BITTI INTEGER,
    BITTISTR VARCHAR(4),
    TOPLAMURETIMECIKISIPLIK DOUBLE PRECISION,
    TOPLAMURETIMECIKISLYCRA DOUBLE PRECISION,
    IPLIKFIREORANI DOUBLE PRECISION,
    TALIMATKULLANIMORANI DOUBLE PRECISION,
    TALIMATCIKISMIKTARI DOUBLE PRECISION,
    TALIMATIADEMIKTARI DOUBLE PRECISION,
    LOT VARCHAR(150))
AS
declare variable shtUretimeCikis       integer;
declare variable shtFasonUretimeCikis  integer;
declare variable hstIplik  integer;
declare variable hstLycra  integer;
declare variable FHamID     integer;
declare variable FRenkID    integer;
begin
/* Kullanıldıgı Talimatlar */
  Select shtUretimeCikis, shtFasonUretimeCikis from OrmeStokHareketTipiGetir into :shtUretimeCikis, :shtFasonUretimeCikis;

  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  For Select OrmeIplikFireRaporu_1.OrmeRafId, StokHareket.OrmePlanlamaID,
             OrmePlanlama.Tarih, OrmePlanlama.UretimPartiNo, OrmePlanlama.FirmaId, PlanlamaFirma.Adi, OrmePlanlama.OrmeMakinaId, OrmeMakina.Adi,
             OrmePlanlama.BoyaSiparisKayitId, OrmePlanlama.OrmeSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             OrmeSiparisDetay.HamId, Ham.Adi, Renk.Adi, OrmePlanlama.PlanlananMiktar, OrmePlanlama.GerceklesenMiktar, OrmePlanlama.ToplamSakatOrulen,
             OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, BoyaSiparis.FirmaId, ASFirma.Adi,
             OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.ArtikelNo, OrmePlanlama.MakinaUstuGramaj, OrmePlanlama.DinlenmisGramaj,
             OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmePlanlama.TupAcikEnMayli, OrmeSiparisDetay.TupAcikEnMayli,
             OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.TahminiFireOrani, OrmePlanlama.Bitti,
             OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3,
             OrmePlanlama.IplikBoyu4, OrmeIplikFireRaporu_1.HamID, OrmeIplikFireRaporu_1.RenkId, OrmeIplikFireRaporu_1.Lot from OrmeIplikFireRaporu_1(:ABaslangicGirisTarihi, :ABitisGirisTarihi, :AHamId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id, :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id, :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ARafId, :ADepoId, :AIplikCinsiId, :AKalinlikId, :AStokGirisNo, :ALot)
    left join StokHareket on StokHareket.OrmeRafID = OrmeIplikFireRaporu_1.OrmeRafId
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaID = StokHareket.OrmePlanlamaID
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
    left join Firma PlanlamaFirma on PlanlamaFirma.FirmaId = OrmePlanlama.FirmaId
    left join Firma ASFirma on ASFirma.FirmaId = BoyaSiparis.FirmaId
    where (StokHareket.HareketTipi = :shtUretimeCikis or StokHareket.HareketTipi = :shtFasonUretimeCikis) and StokHareket.OrmePlanlamaID is not null
    group by OrmeIplikFireRaporu_1.OrmeRafId, StokHareket.OrmePlanlamaID, OrmePlanlama.Tarih, OrmePlanlama.UretimPartiNo, OrmePlanlama.FirmaId, PlanlamaFirma.Adi, OrmePlanlama.OrmeMakinaId, OrmeMakina.Adi,
             OrmePlanlama.BoyaSiparisKayitId, OrmePlanlama.OrmeSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             OrmeSiparisDetay.HamId, Ham.Adi, Renk.Adi, OrmePlanlama.PlanlananMiktar, OrmePlanlama.GerceklesenMiktar, OrmePlanlama.ToplamSakatOrulen,
             OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, BoyaSiparis.FirmaId, ASFirma.Adi,
             OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.ArtikelNo, OrmePlanlama.MakinaUstuGramaj, OrmePlanlama.DinlenmisGramaj,
             OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmePlanlama.TupAcikEnMayli, OrmeSiparisDetay.TupAcikEnMayli,
             OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.TahminiFireOrani, OrmePlanlama.Bitti,
             OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3,
             OrmePlanlama.IplikBoyu4, OrmeIplikFireRaporu_1.HamID, OrmeIplikFireRaporu_1.RenkId, OrmeIplikFireRaporu_1.Lot
  into :OrmeRafID, :OrmePlanlamaID, :Tarih, :UretimPartiNo, :FirmaId, :FirmaAdi, :OrmeMakinaId, :OrmeMakinaAdi,
       :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :HamId, :HamAdi, :RenkAdi, :TalimatMiktari, :GerceklesenMiktar, :SakatUretimToplami,
       :MusteriSiparisNo, :ReferansSiparisNo, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisSiraNo, :ArtikelNo, :MakinaUstuGramaj, :DinlenmisGramaj,
       :Pus, :Feine, :TupAcikEnMayli, :SipTupAcikEnMayli, :MelanjKodu, :TahminiFireOrani, :Bitti, :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4,
       :FHamID, :FRenkId, :Lot do
  begin
    ToplamUretimeCikisIplik = null;
    ToplamUretimeCikisLycra = Null;
    IplikFireOrani = Null;
    TalimatKullanimOrani = Null;
    TalimatCikisMiktari  = Null;
    TalimatIadeMiktari   = Null;

    Select Sum(CikisMiktari - IadeMiktari) from OrmePlanlamaRecete
       left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
       where OrmePlanlamaRecete.OrmePlanlamaId = :OrmePlanlamaId  and Ham.HamTipiID = :hstLycra
       into :ToplamUretimeCikisLycra;

    Select Sum(CikisMiktari - IadeMiktari) from OrmePlanlamaRecete
       left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
       where OrmePlanlamaRecete.OrmePlanlamaId = :OrmePlanlamaId and Ham.HamTipiID = :hstIplik
       into :ToplamUretimeCikisIplik;


    if (ToplamUretimeCikisLycra is null) then ToplamUretimeCikisLycra = 0;
    if (ToplamUretimeCikisIplik is null) then ToplamUretimeCikisIplik = 0;

    if (Bitti = 1) then
    begin
      if (GerceklesenMiktar > 0) then
      begin
        IplikFireOrani = ((ToplamUretimeCikisIplik + ToplamUretimeCikisLycra - GerceklesenMiktar) / GerceklesenMiktar) * 100;

        IplikFireOrani = Yuvarla(IplikFireOrani, 100);
      end
    end

    if (Bitti = 0) then BittiStr = '';
    else BittiStr = 'Evet';

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:SipTupAcikenMayli) into :SipTupAcikEnMayliStr;

    BoyaSiparisIdStr = '';
    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';


    Select Sum(OrmePlanlamaRecete.KullanimOrani), Sum(OrmePlanlamaRecete.CikisMiktari), Sum(OrmePlanlamaRecete.IadeMiktari) from OrmePlanlamaRecete
      where OrmePlanlamaRecete.OrmePlanlamaID = :OrmePlanlamaID and OrmePlanlamaRecete.hamID = :FHamID and
            IntKarsilastir(OrmePlanlamaRecete.RenkID, :FRenkID) = 1  and
            (OrmePlanlamaRecete.Lot =:Lot or :Lot is null)
      into :TalimatKullanimOrani, TalimatCikisMiktari, :TalimatIadeMiktari;


    suspend;
  end
end
