CREATE PROCEDURE PROXYOKUKAYDET(
    APROXYID VARCHAR(8))
AS
declare variable KayitSayisi integer;
begin
  select count(*) from proxy where proxyid = :AProxyId into :KayitSayisi;

  if (KayitSayisi = 0) then  insert into Proxy(ProxyId) values(:AProxyId);
end
