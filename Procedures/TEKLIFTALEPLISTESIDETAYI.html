CREATE PROCEDURE TEKLIFTALEPLISTESIDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ATEKLIFTALEPDETAYID INTEGER)
RETURNS (
    TEKLIFTALEPKAYITID INTEGER,
    TEKLIFTALEPDETAYID INTEGER,
    SATINALMATALEPDETAYID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    URUNID VARCHAR(20),
    HAMID INTEGER,
    URUNADI VARCHAR(80),
    MIKTAR DOUBLE PRECISION,
    TEKLIFGELDI INTEGER,
    TEKLIFGELDISTR VARCHAR(5),
    TEKLIFGELISTARIHI TIMESTAMP,
    TEKLIFDOSYAADI VARCHAR(255),
    BIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    TERMINTARIHI TIMESTAMP,
    YUKLEMETARIHI TIMESTAMP,
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(40),
    TESLIMSEKLIID INTEGER,
    TESLIMSEKLIADI VARCHAR(40),
    VADEGUN INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(4))
AS
declare variable FUrunAdi        varchar(80);
declare variable FHamAdi         varchar(255);
begin
  for select TeklifTalepListesi.TeklifTalepKayitId, TeklifTalepDetay.TeklifTalepDetayId, TeklifTalepDetay.SatinAlmaTalepDetayId,
             TeklifTalepDetay.FirmaId, Firma.Adi,
             SatinAlmaTalepDetay.UrunId, SatinAlmaTalepDetay.HamId, Urun.Adi, SatinAlmaTalepDetay.Adi, Ham.Adi,
             TeklifTalepDetay.Miktar, TeklifTalepDetay.TeklifGeldi, TeklifTalepDetay.TeklifGelisTarihi, TeklifTalepDetay.TeklifDosyaAdi, TeklifTalepDetay.BirimFiyat,
             TeklifTalepDetay.DovizId, TeklifTalepDetay.TerminTarihi, TeklifTalepDetay.YuklemeTarihi, TeklifTalepDetay.TeslimSekliId, TeslimSekli.Adi,
             TeklifTalepDetay.OdemeSekliId, OdemeSekli.Adi, TeklifTalepDetay.VadeGun, TeklifTalepDetay.Notu,
             TeklifTalepDetay.OnayDurumu,
             VerilenSiparis.VerilenSiparisyil, VerilenSiparis.VerilenSiparisId  from TeklifTalepListesi(:ABaslangicTarihi, :ABitisTarihi, :AFirmaId, :ATeklifTalepDetayId)
  left join TeklifTalepDetay on TeklifTalepDetay.TeklifTalepKayitId = TeklifTalepListesi.TeklifTalepKayitId
  left join SatinAlmaTalepDetay on SatinAlmaTalepDetay.SatinAlmaTalepDetayId = TeklifTalepDetay.SatinAlmaTalepDetayId
  LEFT JOIN Urun ON Urun.UrunId = SatinAlmaTalepDetay.UrunId
  LEFT JOIN Ham ON Ham.HamId = SatinAlmaTalepDetay.HamId
  left join OdemeSekli On OdemeSekli.OdemeSekliId = TeklifTalepDetay.OdemeSekliId
  left join TeslimSekli On TeslimSekli.TeslimSekliId = TeklifTalepDetay.TeslimSekliId
  left join Firma On Firma.FirmaId = TeklifTalepDetay.FirmaId
  left join VerilenSiparisDetay On VerilenSiparisDetay.VerilenSiparisDetayId = TeklifTalepDetay.VerilenSiparisDetayId
  left join VerilenSiparis On VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
  order by TeklifTalepListesi.TeklifTalepKayitId, TeklifTalepDetay.TeklifTalepDetayId
  into :TeklifTalepKayitId, :TeklifTalepDetayId, :SatinAlmaTalepDetayId, :FirmaId, :FirmaAdi,
       :UrunId, :HamId, :UrunAdi, :FUrunAdi, :FHamAdi, :Miktar, :TeklifGeldi,
       :TeklifGelisTarihi, :TeklifDosyaAdi, :BirimFiyat, :DovizId, :TerminTarihi, :YuklemeTarihi, :TeslimSekliId, :TeslimSekliAdi,
       :OdemeSekliId, :OdemeSekliAdi, :VadeGun, :Notu, :OnayDurumu, :VerilenSiparisyil, :VerilenSiparisId do
  begin
    if (UrunId is null and HamId is null) then
      UrunAdi = FUrunAdi;
    else if (HamId is not null) then
      UrunAdi = FHamAdi;

    if (TeklifGeldi = 1) then TeklifGeldiStr = 'Evet';
    else TeklifGeldiStr = '';

    if (OnayDurumu = 1) then OnayDurumuStr = 'Okey';
    else OnayDurumuStr = '';

    Tutar = BirimFiyat * Miktar;
    VerilenSiparisIdStr = '';
    if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || EgemenIntegerToStr(VerilenSiparisYil) || ') '  || EgemenIntegerToStr(VerilenSiparisId)  || ' ';

    suspend;
  end
end
