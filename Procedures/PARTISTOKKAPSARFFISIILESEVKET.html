CREATE PROCEDURE PARTISTOKKAPSARFFISIILESEVKET(
    AHAMSTOKSARFFISIID INTEGER,
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    ADEPOID VARCHAR(10),
    APARTISTOKTARTIID INTEGER,
    AKULLANICIID INTEGER,
    APARTIASAMASTOKDETAYID INTEGER,
    AMAMULDEPOTRANSFERFISIID INTEGER)
AS
declare variable KayitSayisi integer;
declare variable FBoyaIrsaliyeDetayId integer;
declare variable FHamStokSarfFisiDetayId integer;
declare variable ParcaliSevk integer;
declare variable FNetKilo numeric(12,3);
declare variable FNetMetre numeric(12,3);
declare variable FTahminiBrutKilo numeric(12,3);
declare variable FTahminiBrutMetre numeric(12,3);
declare variable BrutKilo numeric(12,3);
declare variable BrutMetre numeric(12,3);
declare variable TamamiTartildi integer;
declare variable FMsg varchar(255);
declare variable FPartiStokTartiId integer;
begin
  Select Parti.ParcaliSevk, PartiStok.Kilo, PartiStok.Metre, PartiStok.TamamiTartildi from PartiStok
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayID
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  left join Ham On Ham.HamId = PartiStok.HamId
  where PartiStok.PartiSinifId = :APartiSinifId
  into :ParcaliSevk, :BrutKilo, :BrutMetre, :TamamiTartildi;


  if (APartiStokTartiId is null) then
  begin
    For Select Sum(BrutKilo), Sum(BrutMetre), Sum(Kilo), Sum(Metre) from PartiStokTarti
       where PartiStokTarti.PartiSinifId = :APartiSinifId and PartiStokTarti.BoyaIrsaliyeId is null
          and   PartistokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.DepoId = :ADepoId
          and PartiStokTarti.MamulDepoTransferFisiId = :AMamulDepoTransferFisiId
       into :FTahminiBrutKilo, :FTahminiBrutMetre, :FNetKilo, :FNetMetre do
    begin
      Select Min(HamStokSarfFisiDetayId) from HamStokSarfFisiDetay
      left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiId = HamStokSarfFisiDetay.HamStokSarfFisiId
        where HamStokSarfFisiDetay.HamStokSarfFisiId = :AHamStokSarfFisiId and HamStokSarfFisiDetay.PartiSinifId = :APartiSinifId
        into :FHamStokSarfFisiDetayId;

      if (FHamStokSarfFisiDetayId is not Null) then
      begin
        Update HamStokSarfFisiDetay Set BrutKilo = BrutKilo + :FTahminiBrutKilo,
                                     Kilo = Kilo + :FNetKilo,
                                     BrutMetre = BrutMetre + :FTahminiBrutMetre,
                                     Metre = Metre + :FNetMetre
            where HamStokSarfFisiDetay.HamStokSarfFisiDetayId = :FHamStokSarfFisiDetayId;
      end
      else
      begin
        FHamStokSarfFisiDetayId = gen_id(HamStokSarfFisiDetay_Id_Gen, 1);

        insert into HamStokSarfFisiDetay(HamStokSarfFisiDetayId, HamStokSarfFisiId, PartiSinifId, BrutKilo, Kilo, BrutMetre, Metre, InsertKullaniciId, KullaniciId)
                      values(:FHamStokSarfFisiDetayId, :AHamStokSarfFisiId, :APartiSinifId, :FTahminiBrutKilo, :FNetKilo, :FTahminiBrutMetre, :FNetMetre, :AKullaniciId, :AKullaniciId);

      end

      Update PartiStokTarti Set HamStokSarfFisiID = :AHamStokSarfFisiId, HamStokSarfFisiDetayId = :FHamStokSarfFisiDetayId
         where PartiStokTarti.PartiSinifId = :APartiSinifId and PartiStokTarti.BoyaIrsaliyeId is null
          and  PartistokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.DepoId = :ADepoId
          and PartiStokTarti.MamulDepoTransferFisiId = :AMamulDepoTransferFisiId;

    end
  end
  else
  begin


    Select BrutKilo, BrutMetre, Kilo, Metre, BoyaIrsaliyeDetayId, HamStokSarfFisiDetayId from PartiStokTarti
    where PartiStokTarti.PartiStokTartiId = :APartiStokTartiId
    and (PartiStokTarti.MamulDepoTransferFisiId = :AMamulDepoTransferFisiId or :AMamulDepoTransferFisiId is Null)
    into :FTahminiBrutKilo, :FTahminiBrutMetre, :FNetKilo, :FNetMetre, :FBoyaIrsaliyeDetayId, :FHamStokSarfFisiDetayId;

    if (FBoyaIrsaliyeDetayId is not Null) then exception BuPaketinIrsaliyesiKesilmis;
    if (FHamStokSarfFisiDetayId is not Null) then exception BuPaketSarFisiIleCikmis;

    Select Distinct HamStokSarfFisiDetayId from HamStokSarfFisiDetay
      left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiId = HamStokSarfFisi.HamStokSarfFisiId
      where HamStokSarfFisiDetay.HamStokSarfFisiId = :AHamStokSarfFisiId and HamStokSarfFisiDetay.PartiSinifId = :APartiSinifId
      into :FHamStokSarfFisiDetayId;

    if (FHamStokSarfFisiDetayId is not Null) then
    begin
      Update HamStokSarfFisiDetay Set BrutKilo = BrutKilo + :FTahminiBrutKilo,
                                   Kilo = Kilo + :FNetKilo,
                                   BrutMetre = BrutMetre + :FTahminiBrutMetre,
                                   Metre = Metre + :FNetMetre
        where HamStokSarfFisiDetay.HamStokSarfFisiDetayId = :FHamStokSarfFisiDetayId;
    end
    else
    begin
      FHamStokSarfFisiDetayId = gen_id(HamStokSarfFisiDetay_Id_Gen, 1);

      insert into HamStokSarfFisiDetay(HamStokSarfFisiDetayId, HamStokSarfFisiId, PartiSinifId, BrutKilo, Kilo, BrutMetre, Metre, InsertKullaniciId, KullaniciId, PartiAsamaStokDetayId)
                    values(:FHamStokSarfFisiDetayId, :AHamStokSarfFisiId, :APartiSinifId, :FTahminiBrutKilo, :FNetKilo, :FTahminiBrutMetre, :FNetMetre, :AKullaniciId, :AKullaniciId, :APartiAsamaStokDetayId);
    end

    Update PartiStokTarti Set HamStokSarfFisiID = :AHamStokSarfFisiId, HamStokSarfFisiDetayId = :FHamStokSarfFisiDetayId
       where PartiStokTarti.PartiStokTartiId = :APartiStokTartiId
     and (PartiStokTarti.MamulDepoTransferFisiId = :AMamulDepoTransferFisiId or :AMamulDepoTransferFisiId is Null);

  end
end
