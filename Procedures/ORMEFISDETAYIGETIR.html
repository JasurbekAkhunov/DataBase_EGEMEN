    UretimFirmaId = Null;
    UretimFirmaAdi = Null;

    if (HarmanYil is not null) then
      HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') '  || EgemenIntegerToStr(HarmanId)  || ' ';
    else
      HarmanIdStr = '';

    IlkIslemGrubuId = IslemGrubuId;

    if (TamirBoyaSiparisDetayId is not null) then
      Select IslemGrubuId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = :TamirBoyaSiparisDetayId into :IlkIslemGrubuId;

    if (OrmeSiparisDetayID is null and BoyaSiparisDetayId is not null) then
    begin
      MusteriSiparisNo = FMusteriSiparisNo;
      ReferansSiparisNo = FReferansSiparisNo;
    end

    if (BoyaTalimatKayitID is not null) then
    begin
      Select Renk.Adi, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, BoyaTalimat.ReceteKodu from BoyaTalimat
        left join Renk On Renk.RenkID = BoyaTalimat.RenkId
        left join BoyaSiparisBaski On BoyaSiparisBaski.BoyaSiparisBaskiID = BoyaTalimat.BoyaSiparisBaskiId
        left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
        left join Desen On Desen.DesenID = DesenVaryant.DesenId
        where BoyaTalimat.BoyaTalimatKayitID = :BoyaTalimatKayitID
        into :BoyaTalimatRenkAdi, :BoyaTalimatDesenId, :BoyaTalimatDesenAdi, :BoyaTalimatVaryantAdi, :BoyaTalimatReceteKodu;
    end
    else
    begin
      BoyaTalimatRenkAdi    = Null;
      BoyaTalimatDesenId    = Null;
      BoyaTalimatDesenAdi   = Null;
      BoyaTalimatVaryantAdi = Null;
      BoyaTalimatReceteKodu = Null;
    end

    if (FGirisCikis = -1) then
    begin
      Select OrmeRaf.StokGirisNo, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.KaliteNo, OrmeRaf.KaliteAciklamaTipiId, KaliteAciklamaTipi.Adi,
             OrmeRaf.MamulProsesId, MamulProses.Adi, OrmeRaf.CikisPartiNo,
             OrmeRaf.UretimPartiNo, OrmeRaf.OrmeMakinaId, OrmeMakina.Adi, OrmeRaf.DokumaMakinaId, DokumaMakina.Adi, OrmeRaf.UreticiFirmaId, Firma.Adi from OrmeRaf
        Left join Renk on Renk.RenkID = OrmeRaf.RenkId
        Left Join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = OrmeRaf.KaliteAciklamaTipiId
        Left Join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmeRaf.OrmeMakinaId
        Left Join DokumaMakina on DokumaMakina.DokumaMakinaId = OrmeRaf.DokumaMakinaId
        Left Join MamulProses on MamulProses.MamulProsesId = OrmeRaf.MamulProsesId
          Left join Firma on Firma.FirmaID = OrmeRaf.UreticiFirmaId
        where OrmeRaf.OrmeRafID = :OrmeRafId
        into :StokGirisNo, :RenkId, :RenkAdi, :KaliteNo, :KaliteAciklamaTipiId, :KaliteAciklamaTipiAdi, :MamulProsesId, :MamulProsesAdi,
             :FCikisPartiNo, :UretimPartiNo, :UretimOrmeMakinaId, :UretimOrmeMakinaAdi, :UretimDokumaMakinaId, :UretimDokumaMakinaAdi, :UretimFirmaId, :UretimFirmaAdi;

        Select CikisPartiNo from BoyahaneCikisPartiNo where BoyahaneCikisPartiNo.StokHareketID = :StokHareketID into :CikisPartiNo;
        if (CikisPartiNo is null) then CikisPartiNo = FCikisPartiNo;
    end

    if (Tevkifat = 1) then TevkifatStr = 'Var';
    else TevkifatStr = '';

    if (Sakat = 1) then SakatStr = 'Sakat';
    else SakatStr = '';

    if (EskiPartiNo is not null) then EskiPartiNoStr = '(' || EgemenIntegerToStr(EskiPartiYil)  || ') ' || EgemenIntegerToStr(EskiPartiNo) || ' ';
    else EskiPartiNoStr = '';

    NeredenGirildiStr = '';

    if (NeredenGirildi <> AEvrakTipi) then
    if (NeredenGirildi = 0) then NeredenGirildiStr = 'Fiş''ten ';
    else if (NeredenGirildi = 1) then NeredenGirildiStr = 'İrsaliye''den';

    if (HamIade = 1) then HamIadeStr = 'İade';
    else HamIadeStr = '';

    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';
    else
      BoyaSiparisIdStr = '';

    if (FVerilenSiparisYil is not null) then
      VerilenSiparisIdStr = '(' || EgemenIntegerToStr(FVerilenSiparisYil) || ') '  || EgemenIntegerToStr(FVerilenSiparisId)  || ' ';
    else
      VerilenSiparisIdStr = '';

    if (OrmePlanlamaId is not null) then
    begin
      IplikBoyu1 = TIplikBoyu1;
      IplikBoyu2 = TIplikBoyu2;
      IplikBoyu3 = TIplikBoyu3;
      IplikBoyu4 = TIplikBoyu4;
    end

    BirimIskontoToplami = BirimIskontoToplami * FaturaMiktar;
    BrutTutar = KdvHaricFiyat * FaturaMiktar;

    if (FAltIskontoCarpani is null) then FAltIskontoCarpani = 0;
    if (FFKdvYuvarlamaCarpani is not null) then FKdvYuvarlamaCarpani = FFKdvYuvarlamaCarpani; /* Faturanin carpanini kullansin */
    if (FFNetFiyatYuvarlamaCarpani is not null) then FNetFiyatYuvarlamaCarpani = FFNetFiyatYuvarlamaCarpani; /* Faturanin carpanini kullansin */
    if (FFBrutToplamYuvarlamaCarpani is not null) then FBrutToplamYuvarlamaCarpani = FFBrutToplamYuvarlamaCarpani; /* Faturanin carpanini kullansin */

    BrutTutar = Yuvarla(BrutTutar, FBrutToplamYuvarlamaCarpani);

    T1 = NetBirimFiyat - NetBirimFiyat * FAltIskontoCarpani;
    T1 = Yuvarla(T1, FNetFiyatYuvarlamaCarpani);

    Tutar = T1 * FaturaMiktar;

    IndirilecekKdv = 0;
    TahsilEdilecekKdv = 0;


    if (Tevkifat = 1) then
    begin
      ToplamKdv = (T1 * FaturaMiktar * Kdv / 100);

      IndirilecekKdv = ToplamKdv * TevkifatOrani1 / TevkifatOrani2;
      IndirilecekKdv = Yuvarla(IndirilecekKdv, :FKdvYuvarlamaCarpani);

      TahsilEdilecekKdv = ToplamKdv - IndirilecekKdv;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, :FKdvYuvarlamaCarpani);


    end
    else
    begin
      TahsilEdilecekKdv = T1 * FaturaMiktar * Kdv / 100;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, :FKdvYuvarlamaCarpani);
    end

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:PlanlamaTupAcikenMayli) into :PlanlamaTupAcikEnMayliStr;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:SipTupAcikenMayli) into :SipTupAcikEnMayliStr;

    SipMayli = 0;
    SipAcikEn = 0;

    if (SipTupAcikEnMayli = 2) then SipMayli = 1;
    else if (SipTupAcikEnMayli = 1) then SipAcikEn = 1;

    if (SipMayli = 1) then SipMayliStr = 'Evet'; else SipMayliStr = '';
    if (SipAcikEn = 1) then SipAcikEnStr = 'Evet'; else SipAcikEnStr = '';


    if (FGirisCikis = 1) then
    begin
      RafID = '';
      For Select OrmeRafDetay.RafID from OrmeRafDetay
        where OrmeRafDetay.StokHareketID = :StokHareketID
        group by OrmeRafDetay.RafID
        into :FRafId do
      begin
        if (Char_Length(RafId) + Char_Length(FRafId) < 49) then
        begin
          if (RafId <> '') then RafId = RafId || ',';
          RafId = RafId || FRafId;
        end
      end
    end

    if (OrmePlanlamaID is not null) then
    begin
      if (HareketTipi = shtFasonUretimeCikis or HareketTipi = shtUretimeCikis) then
      begin
        Select sum(OrmePlanlamaRecete.CikisMiktari - OrmePlanlamaRecete.IadeMiktari) from OrmePlanlamaRecete
            where OrmePlanlamaRecete.OrmePlanlamaId = :OrmePlanlamaId and OrmePlanlamaRecete.HamId = :HamId and
                  IntKarsilastir(OrmePlanlamaRecete.RenkId, :RenkID) = 1
        into :OrmeTalimatCikisMiktari;
      end
      else
      begin
        Select sum(OrmePlanlamaRecete.CikisMiktari - OrmePlanlamaRecete.IadeMiktari) from OrmePlanlamaRecete
            where OrmePlanlamaRecete.OrmePlanlamaId = :OrmePlanlamaId
        into :OrmeTalimatCikisMiktari;
      end

      Select OrmePlanlama.GerceklesenMiktar from OrmePlanlama
          where OrmePlanlama.OrmePlanlamaId = :OrmePlanlamaId
      into :OrmeTalimatGerceklesenMiktar;
    end
    else
    begin
      OrmeTalimatCikisMiktari = Null;
      OrmeTalimatGerceklesenMiktar = Null;
    end

   if (DokumaHareketi = 1) then
    begin
      Select sum(iif(:MiktarBirim = 1, StokHareketDetay.Metre, StokHareketDetay.Kilo)) from StokHareketDetay where StokHareketDetay.Stokhareketid = :StokHareketId into :StokHareketDetayMiktar;
    end

    suspend;
  end
