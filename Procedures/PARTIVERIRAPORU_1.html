CREATE PROCEDURE PARTIVERIRAPORU_1(
    APARTIYIL INTEGER,
    APARTIID INTEGER)
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTIASAMASTOKDETAYID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    SINIFKILO NUMERIC(12,3),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    CIKISKILO NUMERIC(12,3),
    CIKISMETRE NUMERIC(12,3),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    MAKINAID VARCHAR(20),
    MAKINASARJID INTEGER,
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP)
AS
declare variable PartiKayitID integer;
begin
  Select PartiKayitId from Parti where Parti.PartiYil = :APartiYil and Parti.PartiId = :APartiId into :PartiKayitId;

  if (PartiKayitID is null) then exit;

  PartiYil = APartiYil;
  PartiId = APartiId;

  For Select PartiAsamaStokDetay.PartiAsamaStokDetayID, PartiAsamaStokDetay.PartiSinifId, PartiAsamaStokDetay.PartiAsamalariId, partiStok.HamAdiToplami, PartiStok.Kilo,
             PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.Metre, PartiAsamaStokDetay.CikisKilo, PartiAsamaStokDetay.CikisMetre,
             PartiAsamaStok.AsamaId, Asama.Adi, PartiAsamaStokDetay.MakinaId, PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.CikisTarihi,
             PartiAsamaStokDetay.MakinaSarjId from PartiAsamaStokDetay
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Asama on Asama.AsamaID = PartiAsamaStok.AsamaId
    left join partiStok on PartiStok.PartiSinifID = PartiAsamaStokDetay.PartiSinifId
    where PartiAsamaStokDetay.PartiKayitID = :PartiKayitID
    into :PartiAsamaStokDetayID, :PartiSinifId, :PartiAsamalariId, :HamAdiToplami, :SinifKilo, :Kilo, :Metre, :CikisKilo, :CikisMetre,
         :AsamaId, :AsamaAdi, :MakinaId, :GirisTarihi, :CikisTarihi, :MakinaSarjId do
  begin
    suspend;
  end
end
