CREATE PROCEDURE ORMEMAKINACALISMARAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AORMEMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4))
RETURNS (
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    SURE DOUBLE PRECISION,
    SURESTR VARCHAR(30),
    PLANLAMALICALISMASURESI DOUBLE PRECISION,
    PLANLAMALICALISMASURESISTR VARCHAR(30),
    PLANLAMALICALISMASURESIORANI DOUBLE PRECISION,
    PLANLAMASIZCALISMASURESI DOUBLE PRECISION,
    PLANLAMASIZCALISMASURESISTR VARCHAR(30),
    PLANLAMASIZCALISMASURESIORANI DOUBLE PRECISION,
    TOPLAMCALISMASURESI DOUBLE PRECISION,
    TOPLAMCALISMASURESISTR VARCHAR(30),
    TOPLAMCALISMASURESIORANI DOUBLE PRECISION,
    DURUSSURESI DOUBLE PRECISION,
    DURUSSURESISTR VARCHAR(30),
    DURUSSURESIORANI DOUBLE PRECISION,
    ENYUKSEKTUR INTEGER,
    ENYUKSEKTURSURESI DOUBLE PRECISION,
    ENYUKSEKTURSURESISTR VARCHAR(30),
    ARIZASURESI DOUBLE PRECISION,
    ARIZASURESISTR VARCHAR(30),
    ARIZASURESIORANI DOUBLE PRECISION,
    SERVISSURESI DOUBLE PRECISION,
    SERVISSURESISTR VARCHAR(30),
    SERVISSURESIORANI DOUBLE PRECISION,
    AYARSURESI DOUBLE PRECISION,
    AYARSURESISTR VARCHAR(30),
    AYARSURESIORANI DOUBLE PRECISION,
    MEMINGERKOPTUSURESI DOUBLE PRECISION,
    MEMINGERKOPTUSURESISTR VARCHAR(30),
    MEMINGERKOPTUSURESIORANI DOUBLE PRECISION,
    MEMINGERKOPTUSAYISI INTEGER,
    ALTMEMINGERKOPTUSURESI DOUBLE PRECISION,
    ALTMEMINGERKOPTUSURESISTR VARCHAR(30),
    ALTMEMINGERKOPTUSURESIORANI DOUBLE PRECISION,
    ALTMEMINGERKOPTUSAYISI INTEGER,
    LYCRAKOPTUSURESI DOUBLE PRECISION,
    LYCRAKOPTUSURESISTR VARCHAR(30),
    LYCRAKOPTUSURESIORANI DOUBLE PRECISION,
    LYCRAKOPTUSAYISI INTEGER,
    IGNEKOPTUSURESI DOUBLE PRECISION,
    IGNEKOPTUSURESISTR VARCHAR(30),
    IGNEKOPTUSURESIORANI DOUBLE PRECISION,
    IGNEKOPTUSAYISI INTEGER,
    KAPAKACILDISURESI DOUBLE PRECISION,
    KAPAKACILDISURESISTR VARCHAR(30),
    KAPAKACILDISURESIORANI DOUBLE PRECISION,
    KAPAKACILDISAYISI INTEGER,
    STARTSTOPSURESI DOUBLE PRECISION,
    STARTSTOPSURESISTR VARCHAR(30),
    STARTSTOPSURESIORANI DOUBLE PRECISION,
    STARTSTOPSAYISI INTEGER,
    TOPLAMTUR INTEGER,
    KAPALI INTEGER)
AS
declare variable Tarih  TimeStamp;
declare variable Tur integer;
declare variable OrmePlanlamaId integer;

declare variable LastTarih     TimeStamp;
declare variable LastTur integer;
declare variable LastOrmePlanlamaId integer;
declare variable FarkTarih  Double Precision;
declare variable Sanal      integer;
declare variable LastAriza integer;
declare variable LastServis integer;
declare variable LastAyar integer;
declare variable Ariza integer;
declare variable Servisci integer;
declare variable Ayar integer;

declare variable LastMemingerKoptu      SmallInt;
declare variable LastAltMemingerKoptu   SmallInt;
declare variable LastLycraKoptu         SmallInt;
declare variable LastIgneKoptu          SmallInt;
declare variable LastKapakAcildi        SmallInt;
declare variable LastStartStop          SmallInt;
declare variable MemingerKoptu      SmallInt;
declare variable AltMemingerKoptu   SmallInt;
declare variable IgneKoptu          SmallInt;
declare variable LycraKoptu         SmallInt;
declare variable KapakAcildi        SmallInt;
declare variable StartStop          SmallInt;

begin
  if (ABitisTarihi > Current_TimeStamp(0)) then ABitisTarihi = Current_TimeStamp(0);

  Sure = ABitistarihi - ABaslangicTarihi;
  if (Sure > 0) then SureStr = TarihToStr(Sure);

  For Select OrmeMakinaId, Adi, OrmeMakina.Kapali from OrmeMakina
    where (OrmeMakina.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
          (OrmeMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and OrmeMakina.OtomasyonaDahil = 1
    into :OrmeMakinaId, :OrmeMakinaAdi, :Kapali do
  begin
    PlanlamaliCalismaSuresi       = 0;
    PlanlamaliCalismaSuresiOrani  = 0;
    PlanlamasizCalismaSuresi      = 0;
    PlanlamasizCalismaSuresiOrani = 0;
    ToplamCalismaSuresi           = 0;
    ToplamCalismaSuresiOrani      = 0;
    DurusSuresi                   = 0;
    DurusSuresiOrani              = 0;
    EnYuksekTur                   = 0;
    EnYuksekTurSuresi             = 0;
    ArizaSuresi                   = 0;
    ArizaSuresiOrani              = 0;
    ServisSuresi                   = 0;
    ServisSuresiOrani              = 0;
    AyarSuresi                   = 0;
    AyarSuresiOrani              = 0;

    MemingerKoptuSuresi      = 0;
    MemingerKoptuSuresiOrani = 0;
    MemingerKoptuSayisi      = 0;
    AltMemingerKoptuSuresi      = 0;
    AltMemingerKoptuSuresiOrani = 0;
    AltMemingerKoptuSayisi      = 0;
    IgneKoptuSuresi          = 0;
    IgneKoptuSuresiOrani     = 0;
    IgneKoptuSayisi          = 0;
    LycraKoptuSuresi         = 0;
    LycraKoptuSuresiOrani    = 0;
    LycraKoptuSayisi         = 0;
    KapakAcildiSuresi        = 0;
    KapakAcildiSuresiOrani   = 0;
    KapakAcildiSayisi        = 0;
    StartStopSuresi          = 0;
    StartStopSuresiOrani     = 0;
    StartStopSayisi          = 0;


    LastTur = -1;
    LastOrmePlanlamaId = -1;
    LastTarih = null;
    LastAriza = -1;
    LastServis = -1;
    LastAyar = -1;
    LastMemingerKoptu = -1;
    LastAltMemingerKoptu = -1;
    LastLycraKoptu    = -1;
    LastIgneKoptu     = -1;
    LastKapakAcildi   = -1;
    LastStartStop     = -1;

    ToplamTur = 0;


    For select Tarih, Tur, OrmePlanlamaId, Sanal, Ariza, Servisci, Ayar, MemingerKoptu, AltMemingerKoptu, LycraKoptu, IgneKoptu, KapakAcildi, StartStop from MakinaCalismaBilgisi(:OrmeMakinaId, :ABaslangicTarihi, :ABitisTarihi, Null) order by Tarih
         into :Tarih, :Tur, :OrmePlanlamaId, :Sanal, :Ariza, :Servisci, :Ayar, :MemingerKoptu, :AltMemingerKoptu, :LycraKoptu, :IgneKoptu, :KapakAcildi, :StartStop do

    begin
      ToplamTur = ToplamTur + Tur;

      if (LastTarih is null) then LastTarih = Tarih;
      if (LastTur = -1) then LastTur = Tur;
      FarkTarih = Tarih - LastTarih;

      if (Sanal = 0) then
      if (Tur > EnYuksekTur) then EnYuksekTur = Tur;

      if (Tur > 0 and MemingerKoptu = 0 and AltMemingerKoptu = 0 and LycraKoptu = 0 and KapakAcildi = 0 and IgneKoptu = 0 and StartStop = 0) then Tur = 1;
      else Tur = 0;

      if (LastTur = -1) then LastTur = Tur;
      if (LastAriza = -1) then LastAriza = Ariza;
      if (LastServis = -1) then LastServis = Servisci;
      if (LastAyar = -1) then LastAyar = Ayar;
      if (LastMemingerKoptu = -1) then LastMemingerKoptu = MemingerKoptu;
      if (LastAltMemingerKoptu = -1) then LastAltMemingerKoptu = AltMemingerKoptu;
      if (LastLycraKoptu = -1) then LastLycraKoptu = LycraKoptu;
      if (LastIgneKoptu = -1) then LastIgneKoptu = IgneKoptu;
      if (LastKapakAcildi = -1) then LastKapakAcildi = KapakAcildi;
      if (LastStartStop = -1) then LastStartStop = StartStop;

      if (LastTur = 0) then DurusSuresi = DurusSuresi + FarkTarih;
      else if (LastTur = 1) then
      begin
        if (OrmePlanlamaId is null) then PlanlamasizCalismaSuresi = PlanlamasizCalismaSuresi + FarkTarih;
        else PlanlamaliCalismaSuresi = PlanlamaliCalismaSuresi + FarkTarih;

        ToplamCalismaSuresi = ToplamCalismaSuresi + FarkTarih;
      end

      if (LastAriza = 1) then ArizaSuresi = ArizaSuresi + FarkTarih;
      if (LastServis = 1) then ServisSuresi = ServisSuresi + FarkTarih;
      if (LastAyar = 1) then AyarSuresi = AyarSuresi + FarkTarih;

      if (LastMemingerKoptu = 1) then MemingerKoptuSuresi = MemingerKoptuSuresi + FarkTarih;
      if (LastAltMemingerKoptu = 1) then AltMemingerKoptuSuresi = AltMemingerKoptuSuresi + FarkTarih;
      if (LastLycraKoptu = 1) then LycraKoptuSuresi = LycraKoptuSuresi + FarkTarih;
      if (LastIgneKoptu = 1) then IgneKoptuSuresi = IgneKoptuSuresi + FarkTarih;
      if (LastKapakAcildi = 1) then KapakAcildiSuresi = KapakAcildiSuresi + FarkTarih;
      if (LastStartStop = 1 and LastMemingerKoptu = 0 and LastAltMemingerKoptu = 0 and LastLycraKoptu = 0 and LastIgneKoptu = 0 and LastKapakAcildi = 0) then StartStopSuresi = StartStopSuresi + FarkTarih;

      if (LastMemingerKoptu = 0 and MemingerKoptu = 1) then
        MemingerKoptuSayisi = MemingerKoptuSayisi + 1;
      if (LastAltMemingerKoptu = 0 and AltMemingerKoptu = 1) then
        AltMemingerKoptuSayisi = AltMemingerKoptuSayisi + 1;
      if (LastLycraKoptu = 0 and LycraKoptu = 1) then
        LycraKoptuSayisi = LycraKoptuSayisi + 1;
      if (LastIgneKoptu = 0 and IgneKoptu = 1) then
        IgneKoptuSayisi = IgneKoptuSayisi + 1;
      if (LastKapakAcildi = 0 and KapakAcildi = 1) then
        KapakAcildiSayisi = KapakAcildiSayisi + 1;
      if (LastStartStop = 0 and StartStop = 1) then
        StartStopSayisi = StartStopSayisi + 1;

      LastTarih = Tarih;
      LastTur = Tur;
      LastAriza = Ariza;
      LastServis = Servisci;
      LastAyar = Ayar;
      LastMemingerKoptu = MemingerKoptu;
      LastAltMemingerKoptu = AltMemingerKoptu;
      LastLycraKoptu = LycraKoptu;
      LastIgneKoptu = IgneKoptu;
      LastKapakAcildi = KapakAcildi;
      LastStartStop = StartStop;


    end

    if (Sure > 0) then
    begin
      PlanlamaliCalismaSuresiOrani  = (PlanlamaliCalismaSuresi / Sure) * 100;
      PlanlamasizCalismaSuresiOrani = (PlanlamasizCalismaSuresi/ Sure) * 100;
      ToplamCalismaSuresiOrani = (ToplamCalismaSuresi / Sure) * 100;
      DurusSuresiOrani         = (DurusSuresi / Sure) * 100;
      ArizaSuresiOrani         = (ArizaSuresi / Sure) * 100;
      ServisSuresiOrani        = (ServisSuresi / Sure) * 100;
      AyarSuresiOrani          = (AyarSuresi / Sure) * 100;
      MemingerKoptuSuresiOrani = (MemingerKoptuSuresi / Sure) * 100;
      AltMemingerKoptuSuresiOrani = (AltMemingerKoptuSuresi / Sure) * 100;
      LycraKoptuSuresiOrani     = (LycraKoptuSuresi / Sure) * 100;
      IgneKoptuSuresiOrani     = (IgneKoptuSuresi / Sure) * 100;
      KapakAcildiSuresiOrani   = (KapakAcildiSuresi / Sure) * 100;
      StartStopSuresiOrani     = (StartStopSuresi / Sure) * 100;

      PlanlamaliCalismaSuresiOrani = Yuvarla(PlanlamaliCalismaSuresiOrani, 100);
      PlanlamasizCalismaSuresiOrani = Yuvarla(PlanlamasizCalismaSuresiOrani, 100);
      ToplamCalismaSuresiOrani = Yuvarla(ToplamCalismaSuresiOrani, 100);
      DurusSuresiOrani = Yuvarla(DurusSuresiOrani, 100);
      ArizaSuresiOrani = Yuvarla(ArizaSuresiOrani, 100);
      ServisSuresiOrani = Yuvarla(ServisSuresiOrani, 100);
      AyarSuresiOrani = Yuvarla(AyarSuresiOrani, 100);
      MemingerKoptuSuresiOrani = Yuvarla(MemingerKoptuSuresiOrani, 100);
      AltMemingerKoptuSuresiOrani = Yuvarla(AltMemingerKoptuSuresiOrani, 100);
      LycraKoptuSuresiOrani     = Yuvarla(LycraKoptuSuresiOrani, 100);
      IgneKoptuSuresiOrani     = Yuvarla(IgneKoptuSuresiOrani, 100);
      KapakAcildiSuresiOrani   = Yuvarla(KapakAcildiSuresiOrani, 100);
      StartStopSuresiOrani     = Yuvarla(StartStopSuresiOrani, 100);

    end

    if (PlanlamaliCalismaSuresi = 0) then PlanlamaliCalismaSuresiStr = '';
    else PlanlamaliCalismaSuresiStr = TarihToStr(PlanlamaliCalismaSuresi);

    if (PlanlamasizCalismaSuresi = 0) then PlanlamasizCalismaSuresiStr = '';
    else PlanlamasizCalismaSuresiStr = TarihToStr(PlanlamasizCalismaSuresi);

    if (ToplamCalismaSuresi = 0) then ToplamCalismaSuresiStr = '';
    else ToplamCalismaSuresiStr = TarihToStr(ToplamCalismaSuresi);

    if (DurusSuresi = 0) then DurusSuresiStr = '';
    else DurusSuresiStr = TarihToStr(DurusSuresi);

    if (ArizaSuresi = 0) then ArizaSuresiStr = '';
    else ArizaSuresiStr = TarihToStr(ArizaSuresi);

    if (ServisSuresi = 0) then ServisSuresiStr = '';
    else ServisSuresiStr = TarihToStr(ServisSuresi);

    if (AyarSuresi = 0) then AyarSuresiStr = '';
    else AyarSuresiStr = TarihToStr(AyarSuresi);

    if (MemingerKoptuSuresi = 0) then MemingerKoptuSuresiStr = '';
    else MemingerKoptuSuresiStr = TarihToStr(MemingerKoptuSuresi);

    if (AltMemingerKoptuSuresi = 0) then AltMemingerKoptuSuresiStr = '';
    else AltMemingerKoptuSuresiStr = TarihToStr(AltMemingerKoptuSuresi);

    if (LycraKoptuSuresi = 0) then LycraKoptuSuresiStr = '';
    else LycraKoptuSuresiStr = TarihToStr(LycraKoptuSuresi);

    if (IgneKoptuSuresi = 0) then IgneKoptuSuresiStr = '';
    else IgneKoptuSuresiStr = TarihToStr(IgneKoptuSuresi);

    if (KapakAcildiSuresi = 0) then KapakAcildiSuresiStr = '';
    else KapakAcildiSuresiStr = TarihToStr(KapakAcildiSuresi);

    if (StartStopSuresi = 0) then StartStopSuresiStr = '';
    else StartStopSuresiStr = TarihToStr(StartStopSuresi);

    suspend;
  end
end
