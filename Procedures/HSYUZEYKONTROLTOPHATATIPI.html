CREATE PROCEDURE HSYUZEYKONTROLTOPHATATIPI(
    AYKONTROLISEMRIKAYITID INTEGER)
RETURNS (
    HSYUZEYKONTROLID INTEGER,
    HAMHATATIPIID INTEGER,
    HAMHATATIPIADI VARCHAR(20),
    NOKTASALHATA INTEGER,
    NOKTASALHATASTR VARCHAR(4),
    ADET INTEGER)
AS
begin
  for Select HSYuzeyKontrol.HSYuzeyKontrolId, HamHataTipi.HamHataTipiId, HamHataTipi.Adi, HamHataTipi.NoktasalHata,
             HSYuzeyKontrolHatalari.Adet from HSYuzeyKontrol
  left join HSYuzeyKontrolHatalari on HSYuzeyKontrolHatalari.HSYuzeyKontrolId = HSYuzeyKontrol.HSYuzeyKontrolId
  left join HamHataTipi on HamHataTipi.HamHataTipiId = HSYuzeyKontrolHatalari.HamHataTipiId
  where HSYuzeyKontrol.YKontrolIsEmriKayitId = :AYKontrolIsEmriKayitId and HSYuzeyKontrolHatalari.HSYuzeyKontrolId is not null
  into :HSYuzeyKontrolId, :HamHataTipiId, :HamHataTipiAdi, :NoktasalHata, :Adet do
  begin
    if (NoktasalHata = 1) then NoktasalHataStr = 'Evet';
    else NoktasalHataStr = '';

    suspend;
  end
end
