CREATE PROCEDURE BOYAREZERVFISIDETAYCEKILISTESI(
    ABOYAREZERVFISIID INTEGER,
    ABOYASIPARISDETAYID INTEGER)
RETURNS (
    BOYAREZERVFISIDETAYID INTEGER,
    BOYAREZERVFISIID INTEGER,
    IPLIKPAKETLEMEID INTEGER,
    PSKALITEKONTROLID INTEGER,
    HAMSTOKMASTERTOPID INTEGER,
    BARKODID VARCHAR(20),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    KOLINO INTEGER,
    KAPNO INTEGER,
    KALITENO INTEGER,
    EN VARCHAR(10),
    GRAMAJ VARCHAR(10),
    DARA NUMERIC(12,3),
    AMBALAJAGIRLIGI NUMERIC(12,3),
    BOBINSAYISI INTEGER,
    BOBINDARASI NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    HACIMEN INTEGER,
    HACIMBOY INTEGER,
    HACIMYUKSEKLIK INTEGER,
    HACIM DOUBLE PRECISION,
    PALETNO VARCHAR(5),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIIDSTR VARCHAR(20),
    ACIKLAMA VARCHAR(100),
    PARTIRENKLOTNO INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    LABRECETEKODU RECETEKODUSIZE,
    KALITEID KALITEKODUSIZE)
AS
begin
  En = null;
  Gramaj = Null;
  HacimEn = 0;
  HacimBoy = 0;
  HacimYukseklik = 0;

  For select IplikPaketleme.IplikPaketlemeId, IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PartiStokTarti.BoyaRezervFisiDetayId, PartiStokTarti.KaliteNo, IplikPaketleme.KoliNo,
             PartiStokTarti.Dara, PartiStokTarti.BobinSayisi, IplikPaketleme.BobinDarasi,
             IplikPaketleme.AmbalajAgirligi, PartiStokTarti.BrutKilo,
             PartiStokTarti.Kilo, PartiStokTarti.PaletNo,
             IplikPaketleme.AmbalajId, Ambalaj.Adi, IplikPaketleme.KonikCinsiId, KonikCinsi.Adi,
             LabRecete.RenkDerinligiId, RenkDerinligi.Adi, PartiStokTarti.BoyaIrsaliyeDetayId, BoyaRezervFisiDetay.BoyaRezervFisiId, LabRecete.LabReceteKodu,
             PartiStok.KaliteId, PartiStokTarti.BarkodId from BoyaRezervFisiDetay
  left join PartiStokTarti on BoyaRezervFisiDetay.BoyaRezervFisiDetayId = PartiStokTarti.BoyaRezervFisiDetayId
  left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
  left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId or
  IplikPaketleme.IplikPaketlemeID = Replace(Replace(Replace(PartiStokTarti.BarkodID, 'I-', ''), 'Y-', ''), 'R-', '-1')
  Left Join Ambalaj on Ambalaj.AmbalajId = IplikPaketleme.AmbalajId
  Left Join KonikCinsi on KonikCinsi.KonikCinsiId = IplikPaketleme.KonikCinsiId
  left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
  where (BoyaRezervFisiDetay.BoyaRezervFisiId = :ABoyaRezervFisiId or :ABoyarezervFisiId is null) and
        (BoyaRezervFisiDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId or :ABoyaSiparisDetayId is null)
        and IplikPaketleme.IplikPaketlemeId is not null and (PartiStokTarti.IadeBoyaIrsaliyeId is null or PartiStokTarti.BarkodID like 'I-%' or PartiStokTarti.BarkodID like 'Y-%')
  into :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaRezervFisiDetayId, :KaliteNo, :KoliNo,
       :Dara, :BobinSayisi, :BobinDarasi, :AmbalajAgirligi,
       :BrutKilo, :NetKilo, :PaletNo, :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :RenkDerinligiId, :RenkDerinligiIdStr, :BoyaIrsaliyeDetayId,
       :BoyaRezervFisiId, :LabReceteKodu, :KaliteId, :BarkodId do
  begin
    if (BarkodId is null) then
      BarkodID = 'I-' || EgemenIntegerToStr(IplikPaketlemeId);
    Hacim = 0;
    suspend;
  end

  IplikPaketlemeId = Null;
  KoliNo = null;
  AmbalajId = Null;
  AmbalajAdi= Null;
  KonikCinsiId= Null;
  KonikCinsiAdi= Null;
  AmbalajAgirligi = Null;
  BobinSayisi     = Null;
  BobinDarasi     = Null;

  En = null;
  Gramaj = Null;
  HacimEn = 0;
  HacimBoy = 0;
  HacimYukseklik = 0;

  For select Null, PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PartiStokTarti.BoyaRezervFisiDetayId, PartiStokTarti.KaliteNo, Null,
             PartiStokTarti.Dara, PartiStokTarti.BobinSayisi, Null,
             Null, PartiStokTarti.BrutKilo,
             PartiStokTarti.Kilo, PartiStokTarti.PaletNo,
             Null, Null, Null, Null,
             LabRecete.RenkDerinligiId, RenkDerinligi.Adi, PartiStokTarti.BoyaIrsaliyeDetayId, BoyaRezervFisiDetay.BoyaRezervFisiId, LabRecete.LabReceteKodu,
             PartiStok.KaliteId, PartiStokTarti.BarkodId from BoyaRezervFisiDetay
  left join PartiStokTarti on BoyaRezervFisiDetay.BoyaRezervFisiDetayId = PartiStokTarti.BoyaRezervFisiDetayId
  left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
  left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
  where (BoyaRezervFisiDetay.BoyaRezervFisiId = :ABoyaRezervFisiId or :ABoyarezervFisiId is null) and
        (BoyaRezervFisiDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId or :ABoyaSiparisDetayId is null)
        and PartiStokTarti.IadeBoyaIrsaliyeId is not null and PartiStokTarti.BarkodID not like 'I-%' and PartiStokTarti.BarkodID not like 'Y-%'
  into :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaRezervFisiDetayId, :KaliteNo, :KoliNo,
       :Dara, :BobinSayisi, :BobinDarasi, :AmbalajAgirligi,
       :BrutKilo, :NetKilo, :PaletNo, :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :RenkDerinligiId, :RenkDerinligiIdStr, :BoyaIrsaliyeDetayId,
       :BoyaRezervFisiId, :LabReceteKodu, :KaliteId, :BarkodID do
  begin
    Hacim = 0;
    suspend;
  end

  IplikPaketlemeId = Null;
  KoliNo = null;
  AmbalajId = Null;
  AmbalajAdi= Null;
  KonikCinsiId= Null;
  KonikCinsiAdi= Null;
  AmbalajAgirligi = Null;
  BobinSayisi     = Null;
  BobinDarasi     = Null;

  For select PsKaliteKontrol.PsKaliteKontrolId, PsKaliteKontrol.PartiKayitId, PsKaliteKontrol.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PartiStokTarti.BoyaRezervFisiDetayId, PsKaliteKontrol.TopNo, PartiStokTarti.BrutKilo, PartiStokTarti.Dara, PartiStokTarti.Metre,
             PartiStokTarti.KaliteNo, PsKaliteKontrol.En, PsKaliteKontrol.Gramaj,
             PartiStok.HacimEn, PartiStok.HacimBoy, PartiStok.HacimYukseklik, PartiStokTarti.PaletNo,
             LabRecete.RenkDerinligiId, RenkDerinligi.Adi, PsKaliteKontrol.Aciklama, PsKaliteKontrol.PartiRenkLotNo, PartiStokTarti.BoyaIrsaliyeDetayId,
             BoyaRezervFisiDetay.BoyaRezervFisiId, LabRecete.LabReceteKodu, PartiStok.KaliteId from BoyaRezervFisiDetay
    left join PartiStokTarti on BoyaRezervFisiDetay.BoyaRezervFisiDetayId = PartiStokTarti.BoyaRezervFisiDetayId
    left join PsKaliteKontrol IadePsKaliteKontrol on PartiStokTarti.BarkodID is not Null and  ('K-' || IadePsKaliteKontrol.PSKaliteKontrolId = PartiStokTarti.BarkodID)
    left join PsKaliteKontrol on PsKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId or PsKaliteKontrol.PartiStokTartiId = IadePsKaliteKontrol.PartiStokTartiId
    left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
    left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
  where (BoyaRezervFisiDetay.BoyaRezervFisiId = :ABoyaRezervFisiId or :ABoyarezervFisiId is null) and
        (BoyaRezervFisiDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId or :ABoyaSiparisDetayId is null)
        and PsKaliteKontrol.PsKaliteKontrolId is not null and PartiStokTarti.IadeBoyaIrsaliyeId is null
  into :PsKaliteKontrolId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaRezervFisiDetayId, :KapNo, :BrutKilo, :Dara, :Metre,
       :KaliteNo, :En, :Gramaj, :HacimEn, :HacimBoy, :HacimYukseklik, :PaletNo, :RenkDerinligiId, :RenkDerinligiIdStr, :Aciklama, :PartiRenkLotNo, :BoyaIrsaliyeDetayId,
       :BoyaRezervFisiId, :LabReceteKodu, :KaliteId do
  begin
    NetKilo = BrutKilo - Dara;
    BarkodID = 'K-' || EgemenIntegerToStr(PSKaliteKontrolId);
    Hacim = HacimEn * HacimBoy * HacimYukseklik;
    Hacim = Hacim / 1000000;
    suspend;
  end
end
