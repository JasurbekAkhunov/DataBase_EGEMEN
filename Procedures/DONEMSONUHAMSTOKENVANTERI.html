CREATE PROCEDURE DONEMSONUHAMSTOKENVANTERI(
    ABITISTARIHI TIMESTAMP,
    AHAMTIPIID INTEGER,
    AFASON INTEGER,
    AUNITEID INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    TRANSFERKODU VARCHAR(55),
    HAMDEPOSTOKMIKTARI NUMERIC(12,3),
    YARIMAMULSTOKMIKTARI NUMERIC(12,3),
    MAMULSTOKMIKTARI NUMERIC(12,3))
AS
declare variable FasonTransferKodu  varchar(55);
declare variable SatisTransferKodu  varchar(55);
declare variable AFirmaId           varchar(20);
declare variable PartiSinifId       integer;
declare variable KayitSayisi        integer;
declare variable HamTipiId          integer;
declare variable Miktar             Numeric(12,3);
declare variable Kilo               Numeric(12,3);
declare variable Metre              Numeric(12,3);
declare variable TartiKilo               Numeric(12,3);
declare variable TartiMetre              Numeric(12,3);
declare variable DokumaKumasiKilodanTakipEt integer;
begin
  Select FirmaKodumuz, DokumaKumasiKilodanTakipEt from Sistem into :AFirmaId, :DokumaKumasiKilodanTakipEt;

  For Select Ham.HamId, Ham.Adi, Ham.HamTipiId, Ham.FasonTransferKodu, Ham.SatisTransferKodu from Ham
    where ((Ham.HamTipiId = 0 and :AHamTipiId = 0) or (Ham.HamTipiId <> 0 and :AHamTipiId = 1))
    into :HamId, :HamAdi, :HamTipiId, :FasonTransferKodu, :SatisTransferKodu do
  begin
    if (AFason = 0) then TransferKodu = SatisTransferKodu;
    else TransferKodu = FasonTransferKodu;

    YariMamulStokMiktari = 0;
    MamulStokMiktari     = 0;

    if (AFason = 0) then
    begin
      select sum(girisMiktar) - sum(PartilenenMiktar) - sum(iadeedilenMiktar) - sum(satisMiktar) from HamStokKumilatif
        where HamStokKumilatif.Firmaid = :AFirmaId and HamStokKumilatif.HamId = :HamId and HamStokKumilatif.Tarih <= :ABitisTarihi
        into :HamDepoStokMiktari;

      For Select Distinct PartiStokDetay.PartiSinifId, PartiStok.Kilo, PartiStok.Metre from PartiStokDetay
        left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
        left join PartiStokIptal on PartiStokIptal.PartiStokIptalKayitId = PartiStok.PartiStokIptalKayitId
        where PartiStokDetay.HamId = :HamId and PartiStokDetay.PartilemeTarihi <= :ABitisTarihi and Parti.Fason = :AFason and
             (PartiStokIptal.IptalTarihi is null or PartiStokIptal.IptalTarihi > :ABitisTarihi + 1)
        into :PartiSinifId, :Kilo, :Metre do
      begin
        if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
          Miktar = Kilo;
        else
          Miktar = Metre;

        Select Count(*) from PartiStokTarti
          where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.Tarih < :ABitisTarihi + 1 into :KayitSayisi;

        if (KayitSayisi = 0) then YariMamulStokMiktari = YariMamulStokMiktari + Miktar;
        else
        begin
          TartiKilo = Null;
          TartiMetre = Null;

          Select Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.Metre) from PartiStokTarti
            left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId
            left join Depo on Depo.DepoId = MamulDepoTransferFisi.TransferDepoId
            left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = PartiStokTarti.BoyaIrsaliyeId
            left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiId = PartiStokTarti.HamStokSarfFisiId
            where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.Tarih < :ABitisTarihi + 1 and
                 (PartiStokTarti.BoyaIrsaliyeId is null or BoyaIrsaliye.IrsaliyeTarihi > :ABitisTarihi) and
                 (PartiStokTarti.HamStokSarfFisiId is null or HamStokSarfFisi.FisTarihi > :ABitisTarihi) and
                 (Depo.UniteId = :AUniteId or :AUniteId is null)
            into :TartiKilo, :TartiMetre;

          if (TartiKilo is null) then TartiKilo = 0;
          if (TartiMetre is null) then TartiMetre = 0;

          if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
            MamulStokMiktari = MamulStokMiktari + TartiKilo;
          else
            MamulStokMiktari = MamulStokMiktari + TartiMetre;
        end
      end
      suspend;
    end
    else
    begin
      For select HamStokKumilatif.FirmaId, Firma.Adi, sum(girisMiktar) - sum(PartilenenMiktar) - sum(iadeedilenMiktar) - sum(satisMiktar) from HamStokKumilatif
        left join Firma On Firma.FirmaId = HamStokKumilatif.FirmaId
        where HamStokKumilatif.Firmaid <> :AFirmaId and HamStokKumilatif.HamId = :HamId and HamStokKumilatif.Tarih <= :ABitisTarihi
        group by HamStokKumilatif.FirmaId, Firma.Adi
        into :FirmaId, :FirmaAdi, :HamDepoStokMiktari do
      begin
        if (HamDepoStokMiktari is null) then HamDepoStokMiktari = 0;
        YariMamulStokMiktari = 0;
        MamulStokMiktari     = 0;
        For Select Distinct PartiStokDetay.PartiSinifId, PartiStok.Kilo, PartiStok.Metre from PartiStokDetay
          left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
          left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
          where PartiStokDetay.HamId = :HamId and Parti.FirmaId = :FirmaId and PartiStokDetay.PartilemeTarihi <= :ABitisTarihi and Parti.Fason = :AFason
          into :PartiSinifId, :Kilo, :Metre do
        begin
          if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
            Miktar = Kilo;
          else
            Miktar = Metre;

          Select Count(*) from PartiStokTarti
            where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.Tarih < :ABitisTarihi + 1 into :KayitSayisi;

          if (KayitSayisi = 0) then YariMamulStokMiktari = YariMamulStokMiktari + Miktar;
          else
          begin
            TartiKilo = Null;
            TartiMetre = Null;

            Select Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.Metre) from PartiStokTarti
              left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId
              left join Depo on Depo.DepoId = MamulDepoTransferFisi.TransferDepoId
              left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = PartiStokTarti.BoyaIrsaliyeId
              left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiId = PartiStokTarti.HamStokSarfFisiId
              where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.Tarih < :ABitisTarihi + 1 and
                   (PartiStokTarti.BoyaIrsaliyeId is null or BoyaIrsaliye.IrsaliyeTarihi > :ABitisTarihi) and
                   (PartiStokTarti.HamStokSarfFisiId is null or HamStokSarfFisi.FisTarihi > :ABitisTarihi) and
                   (Depo.UniteId = :AUniteId or :AUniteId is null)
              into :TartiKilo, :TartiMetre;

            if (TartiKilo is null) then TartiKilo = 0;
            if (TartiMetre is null) then TartiMetre = 0;

            if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
              MamulStokMiktari = MamulStokMiktari + TartiKilo;
            else
              MamulStokMiktari = MamulStokMiktari + TartiMetre;
          end
        end
        if (HamDepoStokMiktari <> 0 or YariMamulStokMiktari <> 0 or MamulStokMiktari <> 0) then suspend;
      end
    end
  end
end
