CREATE PROCEDURE IADEEDILEBILECEKCEKSENETLISTESI(
    AAV VARCHAR(1),
    ACS VARCHAR(1),
    AFIRMAID VARCHAR(20))
RETURNS (
    BORDROID INTEGER,
    CEKSENETID INTEGER,
    GENELKAYITID INTEGER,
    ACIKLAMA VARCHAR(40),
    VADETARIHI TIMESTAMP,
    MEBLAG DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    DOVIZKURU1 DOUBLE PRECISION,
    DOVIZKURU2 DOUBLE PRECISION,
    DOVIZKURUSTR VARCHAR(40),
    CEKSENETNO VARCHAR(30),
    HESAPNO VARCHAR(40),
    ILKBORCLU VARCHAR(80),
    KESIDEYERI VARCHAR(20),
    AITOLDUGUBANKAADI VARCHAR(40),
    BANKAHESAPID VARCHAR(20),
    CEKSENETGRUP1 VARCHAR(5),
    CEKSENETGRUP2 VARCHAR(5),
    CEKSENETGRUP3 VARCHAR(5),
    KARSILIKSIZ INTEGER)
AS
declare variable SonBordroTipi Integer;
declare variable KayitSayisi Integer;
declare variable TahsilEdilenMeblag Double Precision;
declare variable FirmaDovizId varchar(5);
begin
  Select DovizId from Firma where Firma.FirmaId = :AFirmaId into :FirmaDovizId;

  for select CekSenetDetay.BordroId, CekSenetDetay.GenelKayitId, CekSenetDetay.CekSenetId, CekSenetDetay.Aciklama,
             CekSenetDetay.Vadetarihi, CekSenetDetay.Meblag, CekSenetDetay.TahsilEdilenMeblag, CekSenetDetay.DovizId, CekSenetDetay.DovizKuru1, CekSenetDetay.DovizKuru2,
             Banka.Adi, CekSenetDetay.BankaHesapId, CekSenetDetay.CekSenetno, CekSenetDetay.HesapNo, CekSenetDetay.IlkBorclu,
             CekSenetDetay.KesideYeri, CekSenetDetay.CekSenetGrup1, CekSenetDetay.CekSenetGrup2, CekSenetDetay.CekSenetGrup3,
             Karsiliksiz, SonBordroTipi  from CekSenet
  left join CekSenetDetay on CekSenetDetay.CsBordroKayitId = CekSenet.CsBordroKayitId
  left join Bankahesap on Bankahesap.BankaHesapId = CekSenetDetay.BankaHesapId
  left join Banka on (Banka.bankaId = CekSenetDetay.AitOlduguBankaId or Banka.BankaId = BankaHesap.BankaId)
  where CekSenet.FirmaId = :AFirmaId and CekSenet.AV = :AAv and CekSenet.CS = :ACs and TahsilEdilenMeblag = 0 and
       (CekSenetDetay.SonBordroTipi = 0 or CekSenetDetay.SonBordroTipi = 2 or CekSenetDetay.SonBordroTipi = 4 or CekSenetDetay.SonBordroTipi = 6)
  into :BordroId, :GenelKayitId, :CekSenetId, :Aciklama, :Vadetarihi, :Meblag, :TahsilEdilenMeblag, :DovizId, :DovizKuru1, :DovizKuru2,
       :AitOlduguBankaAdi, :BankaHesapId, :CekSenetno, :HesapNo,
       :IlkBorclu, :KesideYeri, :CekSenetGrup1, :CekSenetGrup2, :CekSenetGrup3, :Karsiliksiz, :SonBordroTipi do
  begin
    if (FirmaDovizId <> DovizId) then
    begin
      DovizKuruStr = Format_Float('', DovizKuru1) || ' ' || DovizId || '-' || Format_Float('', DovizKuru2) || ' ' || FirmaDovizId;
    end
    else DovizKuruStr = '';
    suspend;
  end
end
