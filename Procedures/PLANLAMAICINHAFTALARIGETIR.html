CREATE PROCEDURE PLANLAMAICINHAFTALARIGETIR
RETURNS (
    HAFTAKAYITID INTEGER,
    HAFTAYIL INTEGER,
    HAFTANO INTEGER,
    BASLANGICTARIH TIMESTAMP,
    BITISTARIH TIMESTAMP,
    SISTEMTARIHI TIMESTAMP)
AS
declare variable BulundugumHafta integer;
declare variable BulundugumYil integer;
declare variable PlanlanacakHaftaSayisi integer;
declare variable FBaslangicTarih    TimeStamp;
declare variable FBitisTarih       TimeStamp;
declare variable Yilgun integer;
begin
  PlanlanacakHaftaSayisi = 4;
  BulundugumHafta = f_WeekOfYear(Current_date);

  BulundugumYil   = f_Year(Current_date);
  SistemTarihi = Current_date;

  if (BulundugumHafta = 1) then
  begin
    Yilgun = f_DayOfYear(Current_date);
    if (YilGun > 360) then BulundugumYil = BulundugumYil + 1;
  end


  if (BulundugumHafta >= 53) then
  begin
    Yilgun = f_DayOfYear(Current_date);
    if (YilGun < 10) then BulundugumHafta = 1;
  end



  Select BaslangicTarih, BitisTarih from Hafta where Hafta.Yil = :BulundugumYil and Hafta.HaftaNo = :BulundugumHafta into :FBaslangicTarih, :FBitisTarih;

  FBitisTarih = FBitisTarih + PlanlanacakHaftaSayisi * 7;

  for Select Hafta.HaftaKayitId, Hafta.Yil, Hafta.HaftaNo, Hafta.BaslangicTarih, Hafta.BitisTarih from Hafta
    where Hafta.BaslangicTarih >= :FBaslangicTarih and Hafta.BitisTarih <= :FBitisTarih
    order by Hafta.Yil, Hafta.HaftaNo
    into :HaftaKayitId, :HaftaYil, :HaftaNo, :BaslangicTarih, :BitisTarih do
  begin
    suspend;
  end

  for Select Hafta.HaftaKayitId, Hafta.Yil, Hafta.HaftaNo, Hafta.BaslangicTarih, Hafta.BitisTarih from Hafta
    left join DesenPlanlama on DesenPlanlama.HaftaKayitID = Hafta.HaftaKayitId
    where DesenPlanlama.HaftaKayitId is not null and Hafta.BaslangicTarih < :FBaslangicTarih
    group by Hafta.HaftaKayitId, Hafta.Yil, Hafta.HaftaNo, Hafta.BaslangicTarih, Hafta.BitisTarih
    order by Hafta.Yil, Hafta.HaftaNo
    into :HaftaKayitId, :HaftaYil, :HaftaNo, :BaslangicTarih, :BitisTarih do
  begin
    suspend;
  end
end
