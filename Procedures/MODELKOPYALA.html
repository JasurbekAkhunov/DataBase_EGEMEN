CREATE PROCEDURE MODELKOPYALA(
    AMODELID MODELKODUSIZE,
    AYENIMODELID MODELKODUSIZE,
    AYENIMODELADI ADI40,
    AKULLANICIID INTEGER)
RETURNS (
    MODELID MODELKODUSIZE)
AS
declare variable FModelBedenId integer;
declare variable YeniModelBedenId integer;
declare variable FModelKumasId integer;
declare variable YeniModelKumasId integer;
declare variable FGarniModelKumasId integer;
declare variable YeniGarniModelKumasId integer;
declare variable AModelKayitId Integer;
declare variable AYeniModelKayitId Integer;
begin
  Select ModelKayitId From Model Where Model.ModelID = :AModelID  into :AModelKayitId;

  Insert into Model(ModelId, Adi, KisaAdi, SezonId, KoleksiyonId, ModelGrupId, YikamaTalimatiID, MusteriModelNo, FirmaID, Notu, OrmeNotu, DokumaNotu, BoyahaneNotu,
                    AksesuarNotu, NumuneNotu, DikisTalimati, SerilemeHazir, BedenSetiDikildi, KalipHazir, DikisTalimatResmi, ResimDosyaAdi1, ResimDosyaAdi2,
                    BedenOlcuResimDosyaAdi1, BedenOlcuResimDosyaAdi2, DigerResimDosyaAdi1, DigerResimDosyaAdi2, InsertKullaniciId, KullaniciId)
      Select :AYeniModelId, :AYeniModelAdi, KisaAdi, SezonId, KoleksiyonId, ModelGrupId, YikamaTalimatiID, MusteriModelNo, FirmaID, Notu, OrmeNotu, DokumaNotu, BoyahaneNotu,
                    AksesuarNotu, NumuneNotu, DikisTalimati, SerilemeHazir, BedenSetiDikildi, KalipHazir, DikisTalimatResmi, ResimDosyaAdi1, ResimDosyaAdi2,
                    BedenOlcuResimDosyaAdi1, BedenOlcuResimDosyaAdi2, DigerResimDosyaAdi1, DigerResimDosyaAdi2, :AKullaniciId, :AKullaniciId from Model
             where Model.ModelKayitId = :AModelKayitId
  Returning ModelKayitId, ModelId into :AYeniModelKayitId, :ModelId;

  For Select ModelBedenId from ModelBeden where ModelBeden.ModelKayitId = :AModelKayitId into :FModelBedenId do
  begin
    YeniModelBedenId = gen_ID(ModelBeden_Id_gen, 1);
    Insert into ModelBeden(ModelBedenId, ModelKayitId, BedenId, InsertKullaniciId, KullaniciId)
        Select :YeniModelBedenId, :AYeniModelKayitId, BedenId, :AKullaniciId, :AKullaniciId from ModelBeden where ModelBeden.ModelBedenId = :FModelBedenId;

    Insert into ModelBedenOlcu(ModelKayitId, BedenOlcuTipiId, ModelBedenId, YikamaOncesi, YikamaSonrasi, InsertKullaniciId, KullaniciId)
        Select :AYeniModelKayitId, BedenOlcuTipiId, :YeniModelBedenId, YikamaOncesi, YikamaSonrasi, :AKullaniciId, :AKullaniciId from ModelBedenOlcu where ModelBedenOlcu.ModelBedenId = :FModelBedenId;
  end

  Insert into ModelUretimIslemleri(ModelKayitId, ModelUretimIslemTipiID, Sira, BirimFiyat, BirimFiyatDovizId, Notu, InsertKullaniciId, KullaniciId)
      Select :AYeniModelKayitId, ModelUretimIslemTipiID, Sira, BirimFiyat, BirimFiyatDovizId, Notu, :AKullaniciId, :AKullaniciId from ModelUretimIslemleri
             where ModelUretimIslemleri.ModelKayitId = :AModelKayitId;


  For Select ModelKumasId from ModelKumas where ModelKumas.ModelKayitId = :AModelKayitId and AnaKumas = 1 into :FModelKumasId do
  begin
    YeniModelKumasId = gen_ID(ModelKumas_Id_gen, 1);
    Insert into ModelKumas(ModelKumasID, ModelKayitId, HamId, Miktar, FireOrani, LabReceteId, LabReceteKodu, LabReceteAtisId, RenkId, BirimFiyat, BirimFiyatDovizId,
                           AnaKumas, AnaKumasModelKumasID, PantoneKodu, RenkAdi, TupAcikEnMayli, BoyaliGramaj, BoyaliEn, Notu, InsertKullaniciId, KullaniciId, TedarikciRenkKodu)
        Select :YeniModelKumasId, :AYeniModelKayitId, HamId, Miktar, FireOrani, LabReceteId, LabReceteKodu, LabReceteAtisId, RenkId, BirimFiyat, BirimFiyatDovizId,
               AnaKumas, AnaKumasModelKumasID, PantoneKodu, RenkAdi, TupAcikEnMayli, BoyaliGramaj, BoyaliEn, Notu, :AKullaniciId, :AKullaniciId, TedarikciRenkKodu from ModelKumas where ModelKumas.ModelKumasId = :FModelKumasId;

    Insert into ModelParca(ModelKayitId, ModelKumasID, ParcaId, InsertKullaniciId, KullaniciId)
        Select :AYeniModelKayitId, :YeniModelKumasId, ParcaId, :AKullaniciId, :AKullaniciId from ModelParca where ModelParca.ModelKumasId = :FModelKumasId;

    Insert into ModelAksesuar(ModelKayitId, ModelKumasID, UrunID, Miktar, FireOrani, BirimFiyat, BirimFiyatDovizId, Notu, KullanimTipi, BirimId, InsertKullaniciId, KullaniciId)
        Select :AYeniModelKayitId, :YeniModelKumasId, UrunID, Miktar, FireOrani, BirimFiyat, BirimFiyatDovizId, Notu, KullanimTipi, BirimId, :AKullaniciId, :AKullaniciId from ModelAksesuar where ModelAksesuar.ModelKumasId = :FModelKumasId and ModelAksesuar.ModelKumasId is not Null;

    /*Garni Kumaslar*/
    For Select ModelKumasId from ModelKumas where ModelKumas.ModelKayitId = :AModelKayitId and AnaKumas = 0 and ModelKumas.AnaKumasModelKumasId = :FModelKumasId into :FGarniModelKumasId do
    begin
      YeniGarniModelKumasId = gen_ID(ModelKumas_Id_gen, 1);
      Insert into ModelKumas(ModelKumasID, ModelKayitId, HamId, Miktar, FireOrani, LabReceteId, LabReceteKodu, LabReceteAtisId, RenkId, BirimFiyat, BirimFiyatDovizId,
                             AnaKumas, AnaKumasModelKumasID, PantoneKodu, RenkAdi, TupAcikEnMayli, BoyaliGramaj, BoyaliEn, Notu, InsertKullaniciId, KullaniciId, TedarikciRenkKodu)
          Select :YeniGarniModelKumasId, :AYeniModelKayitId, HamId, Miktar, FireOrani, LabReceteId, LabReceteKodu, LabReceteAtisId, RenkId, BirimFiyat, BirimFiyatDovizId,
                 AnaKumas, :YeniModelKumasId, PantoneKodu, RenkAdi, TupAcikEnMayli, BoyaliGramaj, BoyaliEn, Notu, :AKullaniciId, :AKullaniciId, TedarikciRenkKodu from ModelKumas where ModelKumas.ModelKumasId = :FGarniModelKumasId;

      Insert into ModelParca(ModelKayitId, ModelKumasID, ParcaId, InsertKullaniciId, KullaniciId)
          Select :AYeniModelKayitId, :YeniGarniModelKumasId, ParcaId, :AKullaniciId, :AKullaniciId from ModelParca where ModelParca.ModelKumasId = :FGarniModelKumasId;
    end
    /*Garni Kumaslar*/
  end

  Insert into ModelAksesuar(ModelKayitId, UrunID, Miktar, FireOrani, BirimFiyat, BirimFiyatDovizId, Notu, KullanimTipi, BirimId, InsertKullaniciId, KullaniciId)
      Select :AYeniModelKayitId, UrunID, Miktar, FireOrani, BirimFiyat, BirimFiyatDovizId, Notu, KullanimTipi, BirimId, :AKullaniciId, :AKullaniciId from ModelAksesuar where ModelAksesuar.ModelKayitId = :AModelKayitId and ModelAksesuar.ModelKumasId is Null;
  Suspend;
end
