CREATE PROCEDURE HAMSTOKLOTLISTESI(
    AHAMID INTEGER,
    ARENKID INTEGER)
RETURNS (
    LOT VARCHAR(150),
    BRUT VARCHAR(4),
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    FIRMAADI VARCHAR(100),
    URETICIADI VARCHAR(40),
    TELADEDI INTEGER,
    IPLIKUZAMADEGERI DOUBLE PRECISION,
    RESULTID VARCHAR(170),
    RAFID VARCHAR(10),
    URETIMPARTINO VARCHAR(30),
    NOTU MEMO)
AS
  declare variable BRUTID integer;
  declare variable Sys_BrutNetSecebilir EgeBoolean;
  declare variable DokumaProjesiAktif integer;
begin
  -- NOT : Kullanıcı eğer SISTEM tablosundaki "OrmeTalimatBrutNetSecebilir" parametresini değiştirir ise OrmeRaf tablosundaki LOT bilgisi GÜNCELLENMEZ

  SELECT OrmeTalimatBrutNetSecebilir FROM SISTEM WHERE SistemID = 1 INTO :Sys_BrutNetSecebilir;
  select sistem.DokumaProjesiAktif from sistem  into :DokumaProjesiAktif;

  if (DokumaProjesiAktif = 2) then
  begin
    For Select Trim(OrmeRaf.Lot), BrutNetListesiGetir.BrutNetStr, OrmeRaf.Brut, OrmeRaf.StokGirisNo,Firma.Adi ,Uretici.Adi, StokHareket.TelAdedi, StokHareket.IplikUzamaDegeri, OrmeRaf.OrmeRafId, Ormeraf.RafId, OrmeRaf.Notu from OrmeRaf
    left join BrutNetListesiGetir ON BrutNetListesiGetir.BrutNet = OrmeRaf.Brut
    left join uretici on uretici.ureticiid = ormeRaf.ureticiid
    left join StokHareket on StokHareket.StokGirisNo = OrmeRaf.StokGirisNo
    left join Firma on Firma.FirmaId = StokHareket.FirmaId
      Where StokHareket.DokumaHareketi = 1 and OrmeRaf.HamId = :AHamId AND IntKarsilastir(OrmeRaf.RenkId, :ARenkId) = 1 AND OrmeRaf.StokVar = 1 AND Trim(OrmeRaf.Lot) <> ''
      Group By Trim(OrmeRaf.Lot), BrutNetListesiGetir.BrutNetStr, OrmeRaf.Brut, OrmeRaf.StokGirisNo,Firma.Adi ,Uretici.Adi, StokHareket.TelAdedi, StokHareket.IplikUzamaDegeri, OrmeRaf.OrmeRafId, Ormeraf.RafId, OrmeRaf.Notu
      INTO      :Lot, :Brut, :BrutID, :StokGirisNo, :FirmaAdi, :UreticiAdi, :TelAdedi, :IplikUzamaDegeri, :OrmeRafId, :RafId, :Notu DO
    begin
      ResultId = egemenintegertostr(OrmeRafId);

      suspend;
    end
  end
  else if (:Sys_BrutNetSecebilir = 1) then
  begin
    For Select Trim(OrmeRaf.Lot), BrutNetListesiGetir.BrutNetStr, OrmeRaf.Brut, OrmeRaf.UretimPartiNo, OrmeRaf.Notu from OrmeRaf
    left join BrutNetListesiGetir ON BrutNetListesiGetir.BrutNet = OrmeRaf.Brut
      Where OrmeRaf.HamId = :AHamId AND IntKarsilastir(OrmeRaf.RenkId, :ARenkId) = 1 AND OrmeRaf.StokVar = 1 AND Trim(OrmeRaf.Lot) <> ''
      Group By Trim(OrmeRaf.Lot), BrutNetListesiGetir.BrutNetStr, OrmeRaf.Brut, OrmeRaf.UretimPartiNo, OrmeRaf.Notu
      INTO      :Lot, :Brut, :BrutID, :UretimPartiNo, :Notu DO
    begin
      if (Lot is not null) then
        ResultId = '"' || Lot || '",' ;
      else
        ResultId = '"",' ;

      if (BrutID is not null) then
        ResultId = ResultId || '"' || BrutID || '"' ;
      else
        ResultId = ResultId || '""' ;

      suspend;
    end
  end
  else
  begin
    Brut = Null;
    For Select Lot from OrmeRaf
      where (OrmeRaf.HamId = :AHamId) AND (IntKarsilastir(OrmeRaf.RenkId, :ARenkId) = 1) AND (OrmeRaf.Lot is not null) AND ( OrmeRaf.StokVar = 1 )
      group by Lot
      into :Lot do
    begin
      ResultId = :Lot;
      suspend;
    end
  end
end
