CREATE PROCEDURE ODESIDOZAJLAMAISEMRIBOYAGETIR(
    ADOZAJLAMAISEMRIID INTEGER,
    ADOZAJLAMASISTEMIID INTEGER)
RETURNS (
    MAKINAID VARCHAR(10),
    OTOMASYONMAKINANO VARCHAR(10),
    URUNID VARCHAR(20),
    MIKTAR DOUBLE PRECISION,
    BOYAGRUPID URUNGRUPKODUSIZE)
AS
declare variable FPartiKayitId         integer;
declare variable FReceteId             varchar(20);
declare variable FPartiAsamalariId     integer;
declare variable FReceteIsEmriYil      Integer;
declare variable FReceteIsEmriId       Integer;
declare variable EgemenFirmaId         integer;
begin
/*ASUS icin yapildi */
  select EgemenFirmaId from Sistem where Sistem.SistemId = 1 into :EgemenFirmaId;

  Select PartiAsamalariId, ReceteIsEmriYil, ReceteIsEmriId from DozajlamaIsEmri
    where DozajlamaIsEmri.DozajlamaIsEmriID = :ADozajlamaIsEmriID
    into :FPartiAsamalariId, FReceteIsEmriYil, FReceteIsEmriId;

  if (FPartiAsamalariID is null) then
  begin
    Select Min(PartiAsamalariId) from ReceteIsEmriDetay
      where ReceteIsEmriDetay.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :FReceteIsEmriId
      into :FPartiAsamalariId;
  end

  Select PartiAsamalari.PartiKayitId, PartiAsamalari.ReceteId from PartiAsamalari
    where PartiAsamalari.PartiAsamalariID = :FPartiAsamalariId
    into :FPartiKayitId, :FReceteId;

  for Select PartiRecetesiDokumu_2.MakinaID, DozajlamaSistemiUrun.BoyaTartimUrunID, iif(Miktar = 0, PartiRecetesiDokumu_2.Banyo, Miktar), PartiRecetesiDokumu_2.OtomasyonMakinaNo, Urun.BoyaGrupId from PartiRecetesiDokumu_2(:FPartiKayitId, :FReceteId, :FPartiAsamalariId, 1)
    left join DozajlamaSistemiUrun On DozajlamaSistemiUrun.UrunID = PartiRecetesiDokumu_2.UrunId and DozajlamaSistemiUrun.DozajlamaSistemiID = :ADozajlamaSistemiID
    left join Urun on Urun.UrunId = PartiRecetesiDokumu_2.UrunId
    where DozajlamaSistemiUrun.BoyaTartimUrunID is not null and
    (PartiRecetesiDokumu_2.UrunTipiId = 0 or
     PartiRecetesiDokumu_2.UrunTipiId = 1 and Urun.TartimYapilacak = 1 and :EgemenFirmaId = 76 /*new jacard*/)
    into :MakinaID, UrunId, :Miktar, :OtomasyonMakinaNo, :BoyaGrupId do
  begin
    suspend;
  end
end
