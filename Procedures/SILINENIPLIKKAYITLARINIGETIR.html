CREATE PROCEDURE SILINENIPLIKKAYITLARINIGETIR(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ALOT VARCHAR(30))
RETURNS (
    LOGTARIHI TIMESTAMP,
    IPTALKULLANICIID INTEGER,
    IPTALKULLANICIADI VARCHAR(40),
    IPLIKLOTID INTEGER,
    IPLIKMAKINAID MAKINAKODUSIZE,
    IPLIKMAKINAADI ADI40,
    IPLIKDEPOID DEPOKODUSIZE,
    IPLIKDEPOADI ADI30,
    GIRISTIPI INTEGER,
    URETIMTARIHI TIMESTAMP,
    SARIMSEKLIID INTEGER,
    IPLIKSTOKID INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI ADI20,
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(30),
    KONIKRENGIID INTEGER,
    KONIKCINSIID INTEGER,
    BOBINSAYISI INTEGER,
    AMBALAJAGIRLIGI DOUBLE PRECISION,
    KAPNO INTEGER,
    PALETAGIRLIGI DOUBLE PRECISION,
    MIKTAR BOYAMIKTARSIZE,
    NETMIKTAR BOYAMIKTARSIZE)
AS
begin

  For Select IplikUretimLog.LogTarihi, IplikUretimLog.IptalKullaniciId, IplikUretimLog.IplikLotId, IplikUretimLog.IplikMakinaId, IplikMakina.Adi, IplikUretimLog.IplikDepoId, IplikDepo.Adi, IplikUretimLog.GirisTipi, IplikUretimLog.UretimTarihi, IplikUretimLog.SarimSekliId,
             IplikUretimLog.IplikStokId, IplikUretimLog.AmbalajId, Ambalaj.Adi, IplikUretimLog.MelanjKodu, IplikUretimLog.Lot, IplikUretimLog.KonikRengiId, IplikUretimLog.KonikCinsiId,
             IplikUretimLog.BobinSayisi, IplikUretimLog.AmbalajAgirligi, IplikUretimLog.KapNo, IplikUretimLog.PaletAgirligi, IplikUretimLog.Miktar, IplikUretimLog.NetMiktar, Kullanici.Adi from IplikUretimLog
  left join IplikMakina on IplikMakina.IplikMakinaId = IplikUretimLog.IplikMakinaId
  left join IplikDepo on IplikDepo.IplikDepoId = IplikUretimLog.IplikDepoId
  left join Ambalaj on Ambalaj.AmbalajId = IplikUretimLog.AmbalajId
  left join Kullanici on Kullanici.KullaniciId = IplikUretimLog.IptalKullaniciId
  where IplikUretimLog.LogTarihi >= :ABaslangicTarihi and  IplikUretimLog.LogTarihi  <= :ABitisTarihi  and (IplikUretimLog.Lot = :ALot or :ALot is Null )
  into :LogTarihi, :IptalKullaniciId, :IplikLotId, :IplikMakinaId, :IplikMakinaAdi, :IplikDepoId, :IplikDepoAdi, :GirisTipi, :UretimTarihi, :SarimSekliId,
       :IplikStokId, :AmbalajId, :AmbalajAdi, :MelanjKodu, :Lot, :KonikRengiId, :KonikCinsiId,
       :BobinSayisi, :AmbalajAgirligi, :KapNo, :PaletAgirligi, :Miktar, :NetMiktar, :IptalKullaniciAdi do

  begin
    suspend;
  end
end
