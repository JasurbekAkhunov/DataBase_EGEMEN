CREATE PROCEDURE SEVKIYATCALISMAGUNGETIR(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AUNITEID INTEGER)
RETURNS (
    CALISMAGUNSAYISI INTEGER)
AS
declare variable FIrsaliyeTarihi    TimeStamp;
begin
  CalismaGunSayisi = 0;

  For Select BoyaIrsaliye.IrsaliyeTarihi from BoyaIrsaliye
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
         (BoyaIrsaliye.UniteId = :AUniteId or :AUniteId is null)
    group by BoyaIrsaliye.IrsaliyeTarihi
    into :FIrsaliyeTarihi do
  begin
    CalismaGunSayisi = CalismaGunSayisi + 1;
  end
  suspend;
end
