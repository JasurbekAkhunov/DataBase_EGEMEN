CREATE PROCEDURE DOKUMABOLUNMUSTOPOLUSTUR(
    ADOKUMAKALITEKONTROLID INTEGER,
    ADOKUMAPLANLAMAID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    TOPNO INTEGER,
    HATASTR VARCHAR(200))
AS
declare variable DokumaKaliteKontrolIdRef integer;
begin
  select DokumaKaliteKontrolIdRef from DokumaKaliteKontrol where DokumaKaliteKontrol.DokumaKaliteKontrolId = :ADokumaKaliteKontrolId into :DokumaKaliteKontrolIdRef;

  if (DokumaKaliteKontrolIdRef is not null) then
  begin
    HataStr = Cast('Bu Top, Zaten Başka Bir Toptan Bölünerek Oluşturulmuş.'||Chr(13)||Chr(10)||'Yine Aynı Topu Bölerek Yeni Top Oluşturabilirsiniz.' as Blob character set UTF8);
  end
  else
  begin
    HataStr = Null;
    Select TopNo from DokumaKKTopNoGetir(:ADokumaPlanlamaId) into :TopNo;

    insert into Dokumakalitekontrol (Dokumaplanlamaid, Tarih, Topno, Miktar, Kapadedi, Dokumamakinaid, Depoid, Rafid, Sakat, Ayartopu, Lot, Sakathamhatatipiid, Stokhareketid,
             Ormerafdetayid, Ormeirsaliyeid, Satisstokhareketid, Ormeucikisirsaliyeid, Uretimecikisstokhareketid, Insertkullaniciid, Inserttarihi, Kullaniciid, Degisimtarihi,
             Linkislemi, Leventno, Leventbitti, Depopartino, Kilo, Ormevardiyaid, Vardiyatarihi, Vardiya, Vardiyaamiripersonelid, Personelid, Atkilotu, Cozguisemrikayitid,
             Noktasalhatahatapuani, Sureklihatahatapuani, Bitirenkullaniciid, Bitistarihi, Kontrolbitti, Notu, Ormerafid, Cozgulotu, Dokumakalitekontrolidref)
          select Dokumaplanlamaid, current_timestamp(0), :TopNo, 1, Kapadedi, Dokumamakinaid, Depoid, Rafid, Sakat, Ayartopu, Lot, Sakathamhatatipiid, Stokhareketid, Ormerafdetayid, Ormeirsaliyeid, Satisstokhareketid,
         Ormeucikisirsaliyeid, Uretimecikisstokhareketid, :AKullaniciId, current_timestamp(0), :AKullaniciId, Degisimtarihi, 0, Leventno, Leventbitti, Depopartino, Null, Ormevardiyaid, Vardiyatarihi, Vardiya,
         Vardiyaamiripersonelid, Personelid, Atkilotu, Cozguisemrikayitid, Noktasalhatahatapuani, Sureklihatahatapuani, null, null, 0, Notu, Ormerafid, Cozgulotu, :ADokumaKaliteKontrolId from Dokumakalitekontrol
         where DokumaKaliteKontrolId = :ADokumaKaliteKontrolId;
  end

  suspend;
end
