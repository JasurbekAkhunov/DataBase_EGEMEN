CREATE PROCEDURE BOYADANORMEYETRANSFERRAPORU_1(
    ADOKUMAHAREKETI SMALLINT)
RETURNS (
    IDSTR VARCHAR(30),
    SEVKYERI VARCHAR(20),
    BOYAIRSALIYEID INTEGER,
    IPLIKIRSALIYEID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    SINIF VARCHAR(1),
    HAREKETTIPIID INTEGER)
AS
declare variable OrmeFirmaKodumuz    varchar(20);
declare variable DokumaFirmaKodumuz  varchar(20);
declare variable shtSatisIplik integer;
declare variable EgemenFirmaID integer;
begin
  Select OrmeFirmaKodumuz, DokumaFirmaKodumuz, EgemenFirmaID from Sistem where SistemID = 1 into :OrmeFirmaKodumuz, :DokumaFirmaKodumuz, :EgemenFirmaID;
  Select shtSatis from IplikStokHareketTipiGetir into :shtSatisIplik;

  if (ADokumaHareketi = 1) then
    OrmeFirmaKodumuz = DokumaFirmaKodumuz;

  for select BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.Sinif, BoyaIrsaliye.HareketTipiId from BoyaIrsaliyeDetay
  left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeID
  left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
  left join RafDetay on RafDetay.RafDetayId = BoyaIrsaliyeDetay.RafDetayId
  left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
  where BoyaIrsaliyeDetay.FirmaId = :OrmeFirmaKodumuz and BoyaIrsaliyeDetay.OrmeIrsaliyeID is null and
        (:EgemenFirmaID <> 84 or
         ((Parti.Fason is null or Parti.Fason = 0) and
         (HamStokDetay.Fason is null or HamStokDetay.Fason = 0))
        )
  group by BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.Sinif, BoyaIrsaliye.HareketTipiId
  into :IrsaliyeNo, :IrsaliyeTarihi, :BoyaIrsaliyeId, :FiiliSevkTarihi, :Sinif, :HareketTipiId  do
  begin
    IdStr = 'B' || EgemenIntegerToStr(BoyaIrsaliyeId);
    SevkYeri = 'Boyahane';
    suspend;
  end

  BoyaIrsaliyeId  = Null;
  IplikIrsaliyeId = Null;
  IrsaliyeNo      = Null;
  IrsaliyeTarihi  = Null;
  FiiliSevkTarihi = Null;
  Sinif           = Null;
  HareketTipiId   = Null;


  for select IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.IplikIrsaliyeId, IplikIrsaliye.FiiliSevkTarihi, IplikIrsaliye.Sinif, IplikIrsaliye.HareketTipi from IplikIrsaliyeDetay
  left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikIrsaliyeDetay.IplikIrsaliyeID
  where IplikIrsaliyeDetay.FirmaId = :OrmeFirmaKodumuz and IplikIrsaliyeDetay.OrmeIrsaliyeID is null and IplikIrsaliye.TransferEdilebilir = 1 and
        IplikIrsaliye.HareketTipi = :shtSatisIplik
  group by IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.IplikIrsaliyeId, IplikIrsaliye.FiiliSevkTarihi, IplikIrsaliye.Sinif, IplikIrsaliye.HareketTipi
  into :IrsaliyeNo, :IrsaliyeTarihi, :IplikIrsaliyeId, :FiiliSevkTarihi, :Sinif, :HareketTipiId  do
  begin
    IdStr = 'I' || EgemenIntegerToStr(IplikIrsaliyeId);
    SevkYeri = 'İplik';

    suspend;
  end
end
