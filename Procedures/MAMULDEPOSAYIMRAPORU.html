CREATE PROCEDURE MAMULDEPOSAYIMRAPORU(
    ATARIH TIMESTAMP,
    ADEPOID DEPOKODUSIZE,
    ARAFID RAFKODUSIZE,
    ATIP INTEGER)
RETURNS (
    DEPOID DEPOKODUSIZE,
    RAFID RAFKODUSIZE,
    PARTISAYISI INTEGER,
    TARTISAYISI INTEGER,
    SAYILANTARTISAYISI INTEGER,
    SAYIMTAMAM EGEBOOLEAN)
AS
declare variable StoktaOlmayanSayisi integer;
declare variable SayilmayanSayisi integer;
begin
  for select Depo.DepoID, Raf.RafId from Depo
      left join DepoRaf on DepoRaf.DepoId = Depo.DepoId
      left join Raf on Raf.RafId = DepoRaf.RafId
    where Depo.Sevk = 1 and
         (Depo.DepoId = :ADepoId or :ADepoId is null) and
         (Raf.RafId = :ARafId or :ARafId is null)
    into :DepoId, :RafId do
  begin
    select count(Distinct PartiStokTarti.PartiSinifId), count(*),
          sum(iif((select count(*) from DepoSayimDetay
            join DepoSayim on DepoSayimDetay.DepoSayimId = DepoSayim.DepoSayimId
            where DepoSayimDetay.PartiStokTartiId = PartiStokTarti.PartiStokTartiId and DepoSayim.FisTarihi = :ATarih and DepoSayimDetay.RafId = :RafId) = 0, 1, 0))
    from MamulDepoTransferFisiDetay
    left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = MamulDepoTransferFisiDetay.PartiStokTartiId and PartiStokTarti.MamulDepoTransferFisiId = MamulDepoTransferFisiDetay.MamulDepoTransferFisiId
    where MamulDepoTransferFisiDetay.RafId = :RafId and
    PartiStokTarti.PartiStokTartiId is not null and
    PartiStokTarti.BoyaIrsaliyeDetayId is null and
    PartiStokTarti.HamStokSarfFisiId is null into :PartiSayisi, :TartiSayisi, :SayilmayanSayisi;

    select count(*), sum(iif(PartiStokTarti.BoyaIrsaliyeDetayId is null and PartiStokTarti.HamStokSarfFisiId is null and MamulDepoTransferFisiDetay.RafId = :RafId , 0, 1))
    from DepoSayim
    join DepoSayimDetay on DepoSayimDetay.DepoSayimId = DepoSayim.DepoSayimID
    join PartiStokTarti on PartiStokTarti.PartiStokTartiId = DepoSayimDetay.PartiStokTartiId
    join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.PartiStokTartiId = PartiStokTarti.PartiStokTartiId and MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId
    where DepoSayim.FisTarihi = :ATarih and DepoSayimDetay.RafId = :RafId and PartiStokTarti.PartiStokTartiId is not null
    into :SayilanTartiSayisi, :StoktaOlmayanSayisi;

    SayimTamam = 1;
    if (StoktaOlmayanSayisi > 0 or SayilmayanSayisi > 0) then SayimTamam = 0;

    if (ATip = 0 and SayimTamam = 0) then suspend;
    else if(ATip is null) then suspend;
  end
end
