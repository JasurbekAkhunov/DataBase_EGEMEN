CREATE PROCEDURE HAMSTOKFATURADETAYIGETIR(
    AHAMSTOKFATURAID INTEGER)
RETURNS (
    HAMSTOKMASTERID INTEGER,
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    MIKTAR NUMERIC(12,3),
    KDV DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    KDVHARICFIYAT DOUBLE PRECISION,
    BRUTTUTAR DOUBLE PRECISION,
    BIRIMISKONTOPARASAL DOUBLE PRECISION,
    BIRIMISKONTOORANI1 DOUBLE PRECISION,
    BIRIMISKONTOORANI2 DOUBLE PRECISION,
    BIRIMISKONTOTOPLAMI DOUBLE PRECISION,
    NETBIRIMFIYAT DOUBLE PRECISION,
    KDVTUTARI DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(50),
    VERILENSIPARISDETAYID INTEGER,
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30))
AS
declare variable FAltIskontoCarpani   Double Precision;
declare variable FKdvYuvarlamaCarpani Double Precision;
declare variable FNetFiyatYuvarlamaCarpani Double Precision;
declare variable T1 Double Precision;
begin
  Select AltIskontoCarpani, KdvYuvarlamaCarpani, NetFiyatYuvarlamaCarpani from HamStokFatura where HamStokFatura.HamStokFaturaId = :AHamStokFaturaId
     into :FAltIskontoCarpani, :FKdvYuvarlamaCarpani, :FNetFiyatYuvarlamaCarpani;

  for select HamStokMaster.HamStokMasterId, Ham.HamTipiId, Ham.Adi, HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Miktar,
             HamStokMaster.Kdv, HamStokMaster.BirimFiyat, HamStokMaster.KdvharicFiyat, HamStokMaster.BirimIskontoOrani1, HamStokMaster.BirimIskontoOrani2,
             HamStokMaster.BirimIskontoToplami, HamStokMaster.BirimIskontoParasal, HamStokMaster.NetBirimFiyat,
             VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, HamStokMaster.VerilenSiparisDetayId,
             HamStokMaster.OdemeSekliId, OdemeSekli.Adi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi from HamStokMaster
    left join Ham on Ham.HamId = HamStokMaster.HamId
    left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = HamStokMaster.VerilenSiparisDetayId
    left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
    left join OdemeSekli on OdemeSekli.OdemeSekliId = HamStokMaster.OdemeSekliId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
    where HamStokMaster.HamStokFaturaId = :AHamStokFaturaId
    into :HamStokMasterId, :HamTipiId, :HamAdi, :KapAdedi, :Kilo, :Metre, :Miktar, :Kdv, :BirimFiyat, :KdvharicFiyat, :BirimIskontoOrani1,
         :BirimIskontoOrani2, :BirimIskontoToplami, :BirimIskontoParasal, :NetBirimFiyat, :VerilenSiparisYil, :VerilenSiparisId, :VerilenSiparisDetayId,
         :OdemeSekliId, :OdemeSekliAdi, :IrsaliyeNo, :IrsaliyeTarihi do
  begin
    BirimIskontoToplami = BirimIskontoToplami * Miktar;
    BrutTutar = KdvHaricFiyat * Miktar;

    T1 = NetBirimFiyat - NetBirimFiyat * FAltIskontoCarpani;
    T1 = Yuvarla(T1, FNetFiyatYuvarlamaCarpani);
    Tutar = T1 * Miktar;

    KdvTutari = T1 * Miktar * Kdv / 100;
    KdvTutari = Yuvarla(:KdvTutari, :FKdvYuvarlamaCarpani);

    VerilenSiparisIdStr = '';
    if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || EgemenIntegerToStr(VerilenSiparisYil) || ') '  || EgemenIntegerToStr(VerilenSiparisId)  || ' ';

    suspend;
  end
end
