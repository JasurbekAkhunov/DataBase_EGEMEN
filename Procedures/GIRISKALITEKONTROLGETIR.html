CREATE PROCEDURE GIRISKALITEKONTROLGETIR(
    ASTOKHAREKETID INTEGER)
RETURNS (
    GIRISKALITEKONTROLID INTEGER,
    TARIH TIMESTAMP,
    TOPNO INTEGER,
    BRUTKILO NUMERIC(12,3),
    DARA NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET NUMERIC(12,3),
    KALITENO INTEGER,
    EN INTEGER,
    GRAMAJ INTEGER,
    CAP VARCHAR(10),
    TOPFIRE DOUBLE PRECISION,
    KONTROLGRAMAJ DOUBLE PRECISION,
    ACIKLAMA VARCHAR(50),
    NOKTASALHATAHATAPUANI DOUBLE PRECISION,
    SUREKLIHATAHATAPUANI DOUBLE PRECISION,
    LOT VARCHAR(150),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    PARTIRENKLOTNO INTEGER,
    BARKOD VARCHAR(12),
    OZELKOD VARCHAR(50))
AS
declare variable FToplamMetre           Double Precision;
declare variable FMetreTul              Double Precision;
begin
  For Select GirisKaliteKontrol.GirisKaliteKontrolId, GirisKaliteKontrol.Tarih,
             GirisKaliteKontrol.TopNo, GirisKaliteKontrol.BrutKilo, GirisKaliteKontrol.Dara, GirisKaliteKontrol.NetKilo,
             GirisKaliteKontrol.Metre, GirisKaliteKontrol.Adet, GirisKaliteKontrol.En, GirisKaliteKontrol.Gramaj, GirisKaliteKontrol.Cap, GirisKaliteKontrol.TopFire, GirisKaliteKontrol.Aciklama,
             GirisKaliteKontrol.KaliteNo, GirisKaliteKontrol.NoktasalHataHataPuani, GirisKaliteKontrol.SurekliHataHataPuani,
             IK.Adi, GirisKaliteKontrol.InsertTarihi, SK.Adi, GirisKaliteKontrol.DegisimTarihi,
             StokHareket.Lot, GirisKaliteKontrol.PartiRenkLotNo, GirisKaliteKontrol.OzelKod from GirisKaliteKontrol
    left join Kullanici OK on OK.KullaniciId  = GirisKaliteKontrol.OnayKullaniciId
    left join Kullanici IK on IK.KullaniciId  = GirisKaliteKontrol.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = GirisKaliteKontrol.KullaniciId
    left join StokHareket on StokHareket.StokHareketId  = GirisKaliteKontrol.StokHareketId
    where GirisKaliteKontrol.StokHareketId = :AStokHareketId
    order by GirisKaliteKontrol.GirisKaliteKontrolId, GirisKaliteKontrol.TopNo
    into :GirisKaliteKontrolId, :Tarih, :TopNo, :BrutKilo, :Dara, :NetKilo, :Metre, :Adet, :En, :Gramaj, :Cap, :TopFire,
         :Aciklama, :KaliteNo, :NoktasalHataHataPuani, :SurekliHataHataPuani,
         :InsertKullaniciAdi, InsertTarihi, :KullaniciAdi, :DegisimTarihi, :Lot, :PartiRenkLotNo, :OzelKod do
  begin
/******** Kontrol Gramaj hesaplaniyor *********/
    if (Metre > 0) then
      FMetreTul = NetKilo / Metre;
    else
     FMetreTul = 0;

    if (En is not null and En <> 0) then
      FMetreTul = FMetreTul * 100 / En;
    else
      FMetreTul = 0;

    KontrolGramaj = FMetreTul * 1000;
    KontrolGramaj = Yuvarla(KontrolGramaj, 100);

    Barkod = 'G-' || GirisKaliteKontrolId;

    suspend;
  end
end
