            left join DokumaPlanlamaRecete on DokumaPlanlamaRecete.DokumaPlanlamaID = DokumaPlanlama.DokumaPlanlamaID
            left join Ham on Ham.HamID = DokumaPlanlamaRecete.HamID
            where DokumaPlanlama.OrmeSiparisDetayId = :OrmeSiparisDetayId and DokumaPlanlamaRecete.DokumaPlanlamaID is not null
            group by Ham.HamTipiID
            into :FHamTipiId, :FMiktar do
          begin
            if (FHamTipiId = hstLycra) then DokumayaCikanNetLycra = DokumayaCikanNetLycra + FMiktar;
            else DokumayaCikanNetIplik = DokumayaCikanNetIplik + FMiktar;
          end

          DokumaFireMiktari = DokumayaCikanNetLycra + DokumayaCikanNetIplik - ToplamDokunan;

          if (FireHesapTipi = 0) then
            if ((ToplamDokunan - DokumayaCikanNetLycra)> 0) then DokumaFireOrani = (DokumaFireMiktari / (ToplamDokunan - DokumayaCikanNetLycra)) * 100;
          else
            if (DokumayaCikanNetIplik  > 0) then DokumaFireOrani = (DokumaFireMiktari / DokumayaCikanNetIplik ) * 100;

          DokumaFireOrani = Yuvarla(DokumaFireOrani, 100);
        end
        else
        begin
          DokumaFireMiktari = 0;
          DokumaFireOrani =  0;
        end

        IlkSevkTarihi = Null;
        if (ToplamSatis > 0) then
        begin
          Select Min(StokHareket.Tarih) from OrmeSiparisDetay
            left join StokHareket on StokHareket.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
            where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and StokHareket.HareketTipi = 2
          into :IlkSevkTarihi;
        end

        IlkTalimatTarihi = Null;
        IlkDokumaUretimTarihi = Null;
        SonDokumaUretimTarihi = Null;

        Select Min(Tarih) from DokumaPlanlama where DokumaPlanlama.OrmeSiparisDetayID = :OrmeSiparisDetayID into :IlkTalimatTarihi;

        Select Min(Tarih), max(Tarih) from StokHareket
          where StokHareket.OrmeSiparisDetayID = :OrmeSiparisDetayID and
                (StokHareket.HareketTipi = :shtUretim or StokHareket.HareketTipi = :shtFasonUretim) and
                StokHareket.UretimIslemTipiID = :uitDokuma and StokHareket.HamIade = 0
          into :IlkDokumaUretimTarihi, :SonDokumaUretimTarihi;

        GlobalSatinAlinanIplik  = 0;
        GlobalIadeEdilenIplik   = 0;

        BYSatinAlinanIplik        = 0;
        BYGlobalSatinAlinanIplik  = 0;

        ToplamFaturaMiktar        = 0;
        ToplamBoyaliFaturaMiktar  = 0;

        Select SatinAlinanIplik, IadeEdilenIplik, ToplamBoyayaCikis, ToplamIplikBoyayaCikis from OrmeSiparisDetayBoyaCikisGetir(:OrmeSiparisDetayID)
              into :SatinAlinanIplik, :IadeEdilenIplik, :ToplamBoyayaCikis, :ToplamIplikBoyayaCikis;

        KumasTeminBoyaSiparisYil = null;
        KumasTeminBoyaSiparisId  = null;

        if (KTBoyaSiparisKayitId is not null) then
        begin
          Select BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId from BoyaSiparis
            where BoyaSiparis.BoyaSiparisKayitID = :KTBoyaSiparisKayitId
            into :KumasTeminBoyaSiparisYil, :KumasTeminBoyaSiparisId;
        end


        select GlobalSatinAlinanIplik, GlobalIadeEdilenIplik from OSRaporu_GlobalIplikMiktarlari(:BoyaSiparisKayitId) into :GlobalSatinAlinanIplik, :GlobalIadeEdilenIplik;

        Select BYSatinAlinanKumas, OrmeSatinAlinanKumas from OrmeSipDetaySatinAlinan(:OrmeSiparisDetayID) into :BYSatinAlinanKumas, :OrmeSatinAlinanKumas;

        FToplamSatis = Null;

        Select Sum(IplikIrsaliyeDetay.Miktar) from IplikTalepDetay
          left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayID = IplikTalepDetay.IplikTalepDetayId
          left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
          where IplikTalepDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and IplikIrsaliyeDetay.IplikIrsaliyeDetayID is not null and
                IplikIrsaliyeDetay.hareketTipi = :ShtSatisIplik and
                (IplikIrsaliye.FirmaID <> :DokumaFirmaKodumuz and IplikIrsaliye.FirmaID <> :FirmaKodumuz)
          into :FToplamSatis;

        if (FToplamSatis is null) then FToplamSatis = 0;
        ToplamSatis = ToplamSatis + FToplamSatis;

        FToplamSatisIade = Null;

        Select Sum(IplikIrsaliyeDetay.Miktar) from IplikTalepDetay
          left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayID = IplikTalepDetay.IplikTalepDetayId
          left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
          where IplikTalepDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and IplikIrsaliyeDetay.IplikIrsaliyeDetayID is not null and
                IplikIrsaliyeDetay.hareketTipi = :ShtSatisIadeIplik and IplikIrsaliye.FirmaID <> :DokumaFirmaKodumuz
          into :FToplamSatisIade;

        if (FToplamSatisIade is null) then FToplamSatisIade = 0;
        ToplamSatisIade = ToplamSatisIade + FToplamSatisIade;

        Select Sum(BoyaIrsaliyeDetay.Kilo) from OrmeSiparisDetay
        left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayID = OrmeSiparisDetay.OrmeSiparisDetayId
        left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
        left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
        left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
        where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and BoyaSiparisDetay.BoyaSiparisDetayID is not null and
              PartiStok.PartiSinifID is not null and BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID is not null and
              BoyaIrsaliye.FirmaID <> :DokumaFirmaKodumuz
        into :FToplamBoyaliSatis;

        if (ToplamBoyaliSatis is null) then ToplamBoyaliSatis = 0;
        if (FToplamBoyaliSatis is null) then FToplamBoyaliSatis = 0;
        ToplamBoyaliSatis = ToplamBoyaliSatis + FToplamBoyaliSatis;

        select IplikSatisIade, IplikSatisIadeTamir, IplikSatis, IplikSatisTamir,
          IplikSatisIadeNet, IplikSatisIadeTamirNet, IplikSatisNet, IplikSatisTamirNet,
          ToplamFaturaMiktar, ToplamBoyaliFaturaMiktar, FToplamBoyaliFaturaMiktar from OSRIcinIplikBilgileri(:OrmeSiparisDetayId)
          into :IplikSatisIade, :IplikSatisIadeTamir, :IplikSatis, :IplikSatisTamir,
          :IplikSatisIadeNet, :IplikSatisIadeTamirNet, :IplikSatisNet, :IplikSatisTamirNet,
          :ToplamFaturaMiktar, :ToplamBoyaliFaturaMiktar, :FToplamBoyaliFaturaMiktar;

        if (KumasTemin = 1) then KumasTeminStr = 'Kumaş Temin Siparişi';
        else KumasTeminStr = '';


        BlobSize = char_length(FMusteriTalimati);

        if (BlobSize > 1023) then MusteriTalimati = Left(FMusteriTalimati, 1023);
        else if (BlobSize > 0) then MusteriTalimati = Left(FMusteriTalimati, BlobSize);
        else MusteriTalimati = '';

        BlobSize = char_length(FNotu);

        if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
        else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
        else Notu = '';

        BlobSize = char_length(FUretimNotu);

        if (BlobSize > 1023) then UretimNotu = Left(FUretimNotu, 1023);
        else if (BlobSize > 0) then UretimNotu = Left(FUretimNotu, BlobSize);
        else UretimNotu = '';

        Select IplikTalepMiktari, IplikTalepIdStr, BYGlobalSatinAlinanIplik, BYSatinAlinanIplik from OrmeSiparisIplikTalepBilgisi(:BoyaSiparisKayitId, :OrmeSiparisDetayId) into :IplikTalepMiktari, :IplikTalepIdStr, :BYGlobalSatinAlinanIplik, :BYSatinAlinanIplik;

        Select HamMaliyet, IplikMaliyeti, LycraMaliyeti from OrmeSipDetayMaliyetGetir(:OrmeSiparisDetayID, :SiparisTarihi) into :HamMaliyet, :IplikMaliyeti, :LycraMaliyeti;

        ToplamBoyaliSiparisMiktari  = 0;
        BoyaliSatisIade             = 0;
        BoyaliSatisEksigi           = 0;

        Select Sum(BoyaSiparisDetay.SiparisKilo) from OrmeSiparisDetay
          left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
          where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and BoyaSiparisDetay.HamID = OrmeSiparisDetay.HamId
        into :ToplamBoyaliSiparisMiktari;

        Select Sum(Miktar) from StokHareket
          where StokHareket.OrmeSiparisDetayID = :OrmeSiparisDetayID and
          BoyaSiparisDetayID is not null and HareketTipi = :shtSatisIade
        into :BoyaliSatisIade;

        if (ToplamBoyaliSiparisMiktari is Null) then ToplamBoyaliSiparisMiktari = 0;
        if (BoyaliSatisIade is Null) then BoyaliSatisIade = 0;

        SatisEksigi       = (HamMiktar + ToplamSatisIade - BoyaliSatisIade) - ToplamSatis;
        BoyaliSatisEksigi = (ToplamBoyaliSiparisMiktari + BoyaliSatisIade) - ToplamBoyaliSatis;


        suspend;
      end
    end
