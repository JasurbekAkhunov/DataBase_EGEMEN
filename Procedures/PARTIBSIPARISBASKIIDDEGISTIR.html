CREATE PROCEDURE PARTIBSIPARISBASKIIDDEGISTIR(
    APARTIKAYITID INTEGER,
    ABOYASIPARISBASKIID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable KayitSayisi             integer;
declare variable PBoyaSiparisKayitId integer;
declare variable BoyaSiparisKayitId      integer;
declare variable PartiStokDesenVaryantId integer;
declare variable OldBoyaSiparisBaskiId   integer;
declare variable Kilo                    Numeric(12,3);
declare variable Metre                   Numeric(12,3);
declare variable PartiSinifId            integer;
declare variable BoyaSiparisDetayId      integer;
begin
  Select BoyaSiparisKayitId from Parti where Parti.PartiKayitID = :APartiKayitID into :PBoyaSiparisKayitId;

  Select BoyaSiparisBaski.BoyaSiparisKayitId, BoyaSiparisBaski.DesenVaryantId from BoyaSiparisBaski
     where BoyaSiparisBaski.BoyaSiparisBaskiId = :ABoyaSiparisBaskiId
     into :BoyaSiparisKayitId, :PartiStokDesenVaryantId;

  if (BoyaSiparisKayitId <> PBoyaSiparisKayitId) then exception KayitHatasi 'Siparişler eşleşmiyor';

  Select Count(*) from PartiStok where PartiStok.PartiKayitID = :APartiKayitID and Durumu >= 8 into :KayitSayisi;

  if (KayitSayisi > 0) then exception KayitHatasi 'Sevk Edilmiş, Varyant Değişemez';

  Update Parti Set DesenVaryantId = :PartiStokDesenVaryantId where Parti.PartiKayitId = :APartiKayitId;

  for Select PartiSinifId, BoyaSiparisDetayId, BoyaSiparisBaskiId, Kilo, Metre from PartiStok
     where PartiStok.PartiKayitID = :APartiKayitID
      into :PartiSinifId, :BoyaSiparisDetayId, :OldBoyaSiparisBaskiId, :Kilo, :Metre do
  begin
    ABoyaSiparisBaskiID = Null;

    Select BoyaSiparisBaskiID from BoyaSiparisBaski
      where BoyaSiparisBaski.BoyaSiparisDetayID = :BoyaSiparisDetayID and BoyaSiparisBaski.DesenVaryantID = :PartiStokDesenVaryantId
      into :ABoyaSiparisBaskiID;

    if (ABoyaSiparisBaskiID is null) then exception KayitHatasi 'BoyaSiparişDetay''a a Desen Varyant Bulunamadı..';

    Update PartiStok Set BoyaSiparisBaskiID = :ABoyaSiparisBaskiID where PartiStok.PartiSinifID = :PartiSinifID;

    Update BoyaSiparisBaski Set PartilenenKilo = PartilenenKilo - :Kilo, PartilenenMetre = PartilenenMetre - :Metre where BoyaSiparisBaski.BoyaSiparisBaskiID = :OldBoyaSiparisBaskiId;
    Update BoyaSiparisBaski Set PartilenenKilo = PartilenenKilo + :Kilo, PartilenenMetre = PartilenenMetre + :Metre where BoyaSiparisBaski.BoyaSiparisBaskiID = :ABoyaSiparisBaskiId;
  end
