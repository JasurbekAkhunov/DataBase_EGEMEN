CREATE PROCEDURE DOKUMAKARMASTOKHAREKETRECETE(
    ATARIHTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    ASIPARISFIRMAID VARCHAR(20),
    ADOKUMAMAKINAID VARCHAR(10),
    AALIS INTEGER,
    ASATIS INTEGER,
    AALISIADE INTEGER,
    ASATISIADE INTEGER,
    AFASONALIS INTEGER,
    AFASONURETIMECIKIS INTEGER,
    AFASONURETIMECIKISIADE INTEGER,
    AURETIMECIKIS INTEGER,
    AURETIMECIKISIADE INTEGER,
    AFASONURETIM INTEGER,
    AURETIM INTEGER,
    AKAYDIRMA INTEGER,
    ASARF INTEGER,
    AHSTIPLIK INTEGER,
    AHSTLYCRA INTEGER,
    AHSTKUMAS INTEGER,
    ADOKUMA INTEGER,
    ACOZGU INTEGER,
    AHASIL INTEGER,
    AHAMID INTEGER,
    ARENKID INTEGER,
    AURETIMPARTINO VARCHAR(30),
    ACIKISPARTINO VARCHAR(30),
    AKUMASCINSIID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    AARACID VARCHAR(11),
    ASEVKSEFERNO INTEGER,
    ASTOKISLEMTIPIID INTEGER,
    AKALITENO INTEGER,
    ALOT VARCHAR(150),
    AFIRMAADI2YIGOREBILIR INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AMUSTERISIPARISNO VARCHAR(50),
    ADEPOID VARCHAR(10),
    ARAFTRANSFERFISI INTEGER,
    AREFERANSNO VARCHAR(30),
    ADEPOPARTINO VARCHAR(30),
    AATKILOTU VARCHAR(30),
    ACOZGULOTU VARCHAR(30),
    ABUKUM INTEGER,
    ABOYA INTEGER,
    AKESIM INTEGER,
    ARAFID VARCHAR(10),
    AINDIGO INTEGER,
    ALOTFILTRE INTEGER)
RETURNS (
    DOKUMAPLANLAMARECETEID INTEGER,
    DOKUMAPLANLAMAID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(30),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    KULLANIMORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    KULLANILMISMIKTAR DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(15),
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    RECETEFIREORANI DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable TahminiFireOrani Double Precision;
declare variable FTalimatReceteFireOrani Double Precision;
declare variable UrunHamId Integer;
declare variable OrmeTalimattaReceteFiresiYap Integer;
declare variable OtomatikUCikislardaReceteFire integer;
begin
  For Select DokumaPlanlamaId From DokumaKarmaStokHareketRaporu(:ATarihTipi, :ABASLANGICTARIHI, :ABITISTARIHI, :ABASLANGICBOYASIPARISYIL, :ABASLANGICBOYASIPARISID, :ABITISBOYASIPARISYIL,
    :ABITISBOYASIPARISID, :AFIRMAID, :ASIPARISFIRMAID, :ADOKUMAMAKINAID, :AALIS, :ASATIS, :AALISIADE, :ASATISIADE, :AFASONALIS, :AFASONURETIMECIKIS, :AFASONURETIMECIKISIADE,
    :AURETIMECIKIS, :AURETIMECIKISIADE, :AFASONURETIM, :AURETIM, :AKAYDIRMA, :ASARF, :AHSTIPLIK, :AHSTLYCRA, :AHSTKUMAS, :ADOKUMA, :ACOZGU, :AHASIL, :AHAMID, :ARENKID, :AURETIMPARTINO, :ACIKISPARTINO, :AKUMASCINSIID, :AIPLIKCINSIID, :AKALINLIKID, :AFASONSIPARISLER, :AFASONSATISSIPARISLER, :ASATISSIPARISLER,
    :AARACID, :ASEVKSEFERNO, :ASTOKISLEMTIPIID, :AKALITENO, :ALot, :AFirmaAdi2yiGorebilir, :ASiparisGrupId, :ASiparisGrup2Id, :AMusteriSiparisNo, :ADepoId, :ARafTransferFisi, :AReferansNo, :ADepoPartiNo, :AAtkiLotu, :ACozguLotu, :ABukum, :ABoya, :AKesim, :ARafID, :AIndigo, :ALotFiltre)
  where DokumaPlanlamaId is not Null
  group by DokumaPlanlamaId
  into :DokumaPlanlamaId do
  begin
    For Select DokumaPlanlamaRecete.DokumaPlanlamaReceteId, DokumaPlanlama.OrmeSiparisDetayId,
               DokumaPlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
               Ham.Adi, Ham.Adi2, DokumaPlanlamaRecete.RenkId, Renk.Adi, DokumaPlanlamaRecete.Lot, DokumaPlanlamaRecete.UreticiId, Uretici.Adi, DokumaPlanlamaRecete.Fason,
               DokumaPlanlamaRecete.KullanimOrani, DokumaPlanlamaRecete.CikisMiktari, DokumaPlanlamaRecete.IadeMiktari,
               DokumaPlanlama.PlanlananMiktar * DokumaPlanlamaRecete.KullanimOrani / 100,
               DokumaPlanlama.GerceklesenMiktar * DokumaPlanlamaRecete.KullanimOrani / 100,
               Ik.Adi, DokumaPlanlamaRecete.InsertTarihi, Sk.Adi, DokumaPlanlamaRecete.DegisimTarihi,
               OrmeSiparisDetay.TahminiFireOrani, OrmeSiparisDetay.HamId, DokumaPlanlamaRecete.TahminiFireOrani From DokumaPlanlama
    left join DokumaPlanlamaRecete on DokumaPlanlamaRecete.DokumaPlanlamaId = DokumaPlanlama.DokumaPlanlamaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    left join Ham on Ham.HamId = DokumaPlanlamaRecete.HamId
    left join Renk on Renk.RenkId = DokumaPlanlamaRecete.RenkId
    left join Uretici on Uretici.UreticiId = DokumaPlanlamaRecete.UreticiId
    left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
    Left Join Kullanici IK on IK.KullaniciId = DokumaPlanlamaRecete.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = DokumaPlanlamaRecete.KullaniciId
    where DokumaPlanlama.DokumaPlanlamaID = :DokumaPlanlamaID
    into :DokumaPlanlamaReceteId, :OrmeSiparisDetayId, :HamId, :HamTipiId, :HamTipiAdi,
         :HamAdi, :HamAdi2, :RenkID, :RenkAdi,  :Lot, :UreticiId, :UreticiAdi, :Fason,
         :KullanimOrani, :CikisMiktari, :IadeMiktari, :KullanilacakMiktar, :KullanilmisMiktar,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :TahminiFireOrani, :UrunHamId, :FTalimatReceteFireOrani do
    begin
      if (HamTipiID = 0 or HamTipiID = 6) then
      begin
        select first 1 OrmeTalimattaReceteFiresiYap, OtomatikUCikislardaReceteFire from Sistem into :OrmeTalimattaReceteFiresiYap, :OtomatikUCikislardaReceteFire;
        select first 1 FireOrani from HamStokRecete where HamId = :UrunHamId and ReceteHamId = :HamId into :ReceteFireOrani;
        if (OtomatikUCikislardaReceteFire = 1) then
        begin
          KullanilacakMiktar = KullanilacakMiktar / ((100 - FTalimatReceteFireOrani) / 100);
          ReceteFireOrani    = FTalimatReceteFireOrani;
        end
        else if (OrmeTalimattaReceteFiresiYap <> 1) then
        begin
          KullanilacakMiktar = KullanilacakMiktar / ((100 - TahminiFireOrani) / 100);
          ReceteFireOrani    = TahminiFireOrani;
        end
        else
          KullanilacakMiktar = KullanilacakMiktar / ((100 - ReceteFireOrani) / 100);
      end

      KullanilacakMiktar = Yuvarla(KullanilacakMiktar, 100);

      if (Fason = 1) then FasonStr = 'Fason';
      else if (Fason = 0) then FasonStr = '';
      else FasonStr = 'Bilinmiyor';
      suspend;
    end
  end
end
