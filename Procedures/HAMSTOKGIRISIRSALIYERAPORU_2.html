CREATE PROCEDURE HAMSTOKGIRISIRSALIYERAPORU_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AGIRISTIPI INTEGER,
    ATAMIR INTEGER,
    AIRSALIYENO IRSALIYENOSIZE,
    AHAMTIPI INTEGER,
    AHAMSTOKHAREKETTIPIID INTEGER,
    AIRSALIYEGRUPID VARCHAR(4))
RETURNS (
    HAMSTOKIRSALIYEID INTEGER,
    HAMSTOKMASTERID INTEGER,
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    KILOX DOUBLE PRECISION,
    METRE NUMERIC(12,3),
    ADET INTEGER,
    MIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    LOT VARCHAR(150),
    ACIKLAMA BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TAMIR INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    RAPORBIRIMFIYAT DOUBLE PRECISION,
    RAPORDOVIZID VARCHAR(5),
    ESKIPARTINO INTEGER,
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    MUSTERIPARTINO VARCHAR(30),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    STOKFIRMAID VARCHAR(20),
    STOKFIRMAADI VARCHAR(100),
    REFERANSFIRMAID VARCHAR(20),
    REFERANSFIRMAADI VARCHAR(100),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    TRANSFERKODU VARCHAR(55),
    IRSALIYETARIHI TIMESTAMP,
    ORMEGIRISIRSALIYENO VARCHAR(400),
    ORMEGIRISIRSALIYETARIHI VARCHAR(400),
    ORMEGIRISFIRMAID VARCHAR(400),
    ORMEGIRISFIRMAADI VARCHAR(400),
    RENKADI VARCHAR(20),
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMSTOKHAREKETTIPIID INTEGER,
    HAMSTOKHAREKETTIPIIDSTR VARCHAR(10),
    UCRETLITAMIR INTEGER,
    UCRETLITAMIRSTR VARCHAR(30),
    HAMSATISTRANSFERKODU VARCHAR(55),
    SATISTRANSFERKODU VARCHAR(55),
    ALISTRANSFERKODU VARCHAR(55),
    RENKLIALISTRANSFERKODU VARCHAR(55),
    SATISIADETRANSFERKODU VARCHAR(55),
    BOYALITRANSFERKODU VARCHAR(55),
    SABOYALITRANSFERKODU VARCHAR(55),
    ALISIADETRANSFERKODU VARCHAR(55),
    ALISBOYALITRANSFERKODU VARCHAR(55),
    ALISBASKILITRANSFERKODU VARCHAR(55),
    HAMSTOKFATURAID INTEGER,
    REFERANSKODU HAMKODUSIZE)
AS
declare variable FHamSatisTransferKodu varchar(55);
declare variable FSatisTransferKodu varchar(55);
declare variable FFasonTransferKodu varchar(55);
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable KullanilanMiktar Double Precision;
declare variable IptalPartiMiktar Double Precision;
declare variable DovizKuru        Double Precision;
declare variable EgemenFirmaId integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  Select SistemDovizId, EgemenFirmaId from Sistem where Sistemid = 1 into :RaporDovizId, :EgemenFirmaId;

  if (AIrsaliyeNo is null) then
  begin
    for select HamStokIrsaliye.HamStokIrsaliyeId, HamStokMaster.HamStokMasterId, HamStokMaster.Fason, Ham.HamTipiId, Ham.Adi,
               HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Miktar, HamStokMaster.Adet, HamStokMaster.BirimFiyat, HamStokIrsaliye.DovizId,
               HamStokMaster.Lot, HamStokMaster.Aciklama, HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartino,
               VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               BoyaSiparis.SiparisGrupID, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2ID, SiparisGrup2.Adi, BoyaSiparis.FirmaId, SipFirma.Adi,
               VerilenSiparisDetay.BoyaSiparisDetayId, HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo,
               HamStokMaster.ReferansSiparisNo, HamStokMaster.StokFirmaId, StokFirma.Adi, HamStokMaster.ReferansFirmaId, ReferansFirma.Adi,
               Ham.HamSatisTransferKodu, Ham.SatisTransferKodu, Ham.FasonTransferKodu, HamStokIrsaliye.IrsaliyeTarihi,
               Ham.Adi2, Ham.Adi3, Ham.Adi4, HamStokMaster.Renk, HamStokFatura.FaturaSeriNo, HamStokFatura.FaturaNo, HamStokFatura.FaturaTarihi, HamStokIrsaliye.HamStokHareketTipiId ,
               IrsaliyeFirma.Firmaid, IrsaliyeFirma.Adi, Ham.HamSatisTransferKodu, Ham.SatisTransferKodu, Ham.AlisTransferKodu, Ham.RenkliAlisTransferKodu, Ham.SatisIadeTransferKodu,
               Ham.BoyaliTransferKodu, Ham.SABoyaliTransferKodu, Ham.AlisIadeTransferKodu, Ham.AlisBoyaliTransferKodu, Ham.AlisBaskiliTransferKodu, HamStokMaster.HamStokFaturaId,
               Ham.ReferansKodu from HamStokIrsaliye
      Left Join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
      left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = HamStokMaster.VerilenSiparisDetayId
      left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = VerilenSiparisDetay.BoyaSiparisKayitId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join Ham on Ham.HamId = HamStokMaster.HamId
      left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
      left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
      left join Firma SipFirma on SipFirma.FirmaId = BoyaSiparis.FirmaId
      left join Firma ReferansFirma on ReferansFirma.FirmaId = HamStokMaster.ReferansFirmaId
      left join Firma StokFirma on StokFirma.FirmaId = HamStokMaster.StokFirmaId
      left join Firma IrsaliyeFirma on IrsaliyeFirma.FirmaId = HamStokIrsaliye.FirmaId
      left join HamStokFatura on HamStokFatura.HamStokFaturaId = HamStokMaster.HamStokFaturaId
      where HamStokMaster.PartiSinifId is null and HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
            (HamStokIrsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (HamStokIrsaliye.Fason = :AGirisTipi or :AGirisTipi is null) and
            (HamStokIrsaliye.HamStokHareketTipiId = :AHamStokHareketTipiId or :AHamStokHareketTipiId is null) and
            (HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
            (HamStokMaster.Tamir = :ATamir or :ATamir is null) and
            ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipi = 0) or ( Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipi = 1) or :AHamTipi is null ) and
            HamStokMaster.HamStokMasterId is not null and
            (HamStokIrsaliye.IrsaliyeGrupID = :AIrsaliyeGrupID or :AIrsaliyeGrupID is null)
      into :HamStokIrsaliyeId, :HamStokMasterId, :Fason, :HamTipiId, :HamAdi, :KapAdedi, :Kilox, :Metre, :Miktar, :Adet, :BirimFiyat, :DovizId,
           :Lot, :Aciklama, :Tamir, :TamirTipiAdi, :EskiPartino, :VerilenSiparisYil, :VerilenSiparisId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
           :SiparisFirmaId, :SiparisFirmaAdi, :BoyaSiparisDetayId, :MusteriPartiNo, :MusteriSiparisNo, :ReferansSiparisNo, :StokFirmaId, :StokFirmaAdi, :ReferansFirmaId, :ReferansFirmaAdi,
           :FHamSatisTransferKodu, :FSatisTransferKodu, :FFasonTransferKodu, :IrsaliyeTarihi,
           :HamAdi2, :HamAdi3, :HamAdi4, :RenkAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :HamStokHareketTipiId ,:FirmaId, :FirmaAdi, :HamSatisTransferKodu, :SatisTransferKodu, :AlisTransferKodu,
           :RenkliAlisTransferKodu, :SatisIadeTransferKodu, :BoyaliTransferKodu, :SABoyaliTransferKodu, :AlisIadeTransferKodu, :AlisBoyaliTransferKodu, :AlisBaskiliTransferKodu, :HamStokFaturaId, :ReferansKodu do
    begin
      VerilenSiparisIdStr = '';
      if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      if (Fason = 1) then
      begin
        FasonStr = 'Fason';
        TransferKodu = FFasonTransferKodu;
      end
      else
      begin
        FasonStr = '';
        TransferKodu = FHamSatisTransferKodu;
      end

      if (EgemenFirmaId = 34) then
      begin
        if (HamStokFaturaId is not null) then
            select TransferKodu from TransferIcinHSFatDetayGetir(:HamStokFaturaId) where HamStokMasterId = :HamStokMasterId into :TransferKodu ;
        else
           TransferKodu = Null ;
      end

      Kilo = Kilox;

      KullanilanMiktar = 0;

      Select Sum(rafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.SatisMiktar) from HamStokDetay
      left join RafDetay On RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId
      where HamStokDetay.HamStokMasterId = :HamStokMasterId
      into :KullanilanMiktar;

      if (KullanilanMiktar is null) then KullanilanMiktar = 0;

      IptalPartiMiktar = 0;

      Select Sum(RafDetay.Miktar) from HamStokDetay
         left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId
         where HamStokDetay.HamStokMasterId = :HamStokMasterId and RafDetay.PartiStokIptalKayitId is not null
         into :IptalPartiMiktar;

      if (IptalPartiMiktar is null) then IptalPartiMiktar = 0;

      KalanMiktar = Miktar - KullanilanMiktar + IptalPartiMiktar;

      if (DovizID <> RaporDovizID) then
      begin
        select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :RaporDovizID, Current_Date) into :DovizKuru;
      end
      else DovizKuru = 1;

      RaporBirimFiyat = BirimFiyat * DovizKuru;

      Select OrmeGirisIrsaliyeNo, OrmeGirisIrsaliyeTarihi, GirisOrmeFirmaId, GirisOrmeFirmaAdi from HamStokMasterOrmeGirisiGetir(:HamStokMasterID) into :OrmeGirisIrsaliyeNo, OrmeGirisIrsaliyeTarihi, :OrmeGirisFirmaId, :OrmeGirisFirmaAdi;


      suspend;
    end
  end
  else
  begin
    for select HamStokIrsaliye.HamStokIrsaliyeId, HamStokMaster.HamStokMasterId, HamStokMaster.Fason, Ham.HamTipiId, Ham.Adi,
               HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Miktar, HamStokMaster.Adet, HamStokMaster.BirimFiyat, HamStokIrsaliye.DovizId,
               HamStokMaster.Lot, HamStokMaster.Aciklama,
               HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartino,
               VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               BoyaSiparis.SiparisGrupID, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2ID, SiparisGrup2.Adi, BoyaSiparis.FirmaId, SipFirma.Adi,
               VerilenSiparisDetay.BoyaSiparisDetayId, HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo,
               HamStokMaster.ReferansSiparisNo, HamStokMaster.StokFirmaId, StokFirma.Adi, HamStokMaster.ReferansFirmaId, ReferansFirma.Adi,
               Ham.HamSatisTransferKodu, Ham.SatisTransferKodu, Ham.FasonTransferKodu, HamStokIrsaliye.IrsaliyeTarihi,
               Ham.Adi2, Ham.Adi3, Ham.Adi4, HamStokMaster.Renk, HamStokFatura.FaturaSeriNo, HamStokFatura.FaturaNo, HamStokFatura.FaturaTarihi,
               HamStokIrsaliye.HamStokHareketTipiId , IrsaliyeFirma.Firmaid, IrsaliyeFirma.Adi, Ham.HamSatisTransferKodu, Ham.SatisTransferKodu,
               Ham.AlisTransferKodu, Ham.RenkliAlisTransferKodu, Ham.SatisIadeTransferKodu, Ham.BoyaliTransferKodu, Ham.SABoyaliTransferKodu,
               Ham.AlisIadeTransferKodu, Ham.AlisBoyaliTransferKodu, Ham.AlisBaskiliTransferKodu, HamStokMaster.HamStokFaturaId, Ham.ReferansKodu from HamStokIrsaliye
      Left Join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
      left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = HamStokMaster.VerilenSiparisDetayId
      left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = VerilenSiparisDetay.BoyaSiparisKayitId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join Ham on Ham.HamId = HamStokMaster.HamId
      left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
      left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
      left join Firma SipFirma on SipFirma.FirmaId = BoyaSiparis.FirmaId
      left join Firma ReferansFirma on ReferansFirma.FirmaId = HamStokMaster.ReferansFirmaId
      left join Firma StokFirma on StokFirma.FirmaId = HamStokMaster.StokFirmaId
      left join Firma IrsaliyeFirma on IrsaliyeFirma.FirmaId = HamStokIrsaliye.FirmaId
      left join HamStokFatura on HamStokFatura.HamStokFaturaId = HamStokMaster.HamStokFaturaId
      where HamStokMaster.PartiSinifId is null and (HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
            (HamStokIrsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (HamStokIrsaliye.HamStokHareketTipiId = :AHamStokHareketTipiId or :AHamStokHareketTipiId is null) and
            (HamStokIrsaliye.Fason = :AGirisTipi or :AGirisTipi is null) and
            (HamStokMaster.Tamir = :ATamir or :ATamir is null) and
            ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipi = 0) or ( Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipi = 1) or :AHamTipi is null ) and
            (HamStokIrsaliye.IrsaliyeGrupID = :AIrsaliyeGrupID or :AIrsaliyeGrupID is null)
      into :HamStokIrsaliyeId, :HamStokMasterId, :Fason, :HamTipiId, :HamAdi, :KapAdedi, :Kilo, :Metre, :Miktar, :Adet, :BirimFiyat, :DovizId, :Lot, :Aciklama,
           :Tamir, :TamirTipiAdi, :EskiPartino, :VerilenSiparisYil, :VerilenSiparisId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
           :SiparisFirmaId, :SiparisFirmaAdi, :BoyaSiparisDetayId, :MusteriPartiNo, :MusteriSiparisNo, :ReferansSiparisNo, :StokFirmaId, :StokFirmaAdi, :ReferansFirmaId, :ReferansFirmaAdi,
           :FHamSatisTransferKodu, :FSatisTransferKodu, :FFasonTransferKodu, :IrsaliyeTarihi,
           :HamAdi2, :HamAdi3, :HamAdi4, :RenkAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :HamStokHareketTipiId ,:FirmaId, :FirmaAdi, :HamSatisTransferKodu, :SatisTransferKodu, :AlisTransferKodu,
           :RenkliAlisTransferKodu, :SatisIadeTransferKodu, :BoyaliTransferKodu, :SABoyaliTransferKodu, :AlisIadeTransferKodu, :AlisBoyaliTransferKodu, :AlisBaskiliTransferKodu, :HamStokFaturaId, :ReferansKodu do
    begin
      VerilenSiparisIdStr = '';
      if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      if (Fason = 1) then
      begin
        FasonStr = 'Fason';
        TransferKodu = FFasonTransferKodu;
      end
      else
      begin
        FasonStr = '';
        TransferKodu = FHamSatisTransferKodu;
      end

      if (EgemenFirmaId = 34) then
      begin
        if (HamStokFaturaId is not null) then
           select TransferKodu from TransferIcinHSFatDetayGetir(:HamStokFaturaId) where HamStokMasterId = :HamStokMasterId into :TransferKodu ;
        else
           TransferKodu = Null ;
      end

      KullanilanMiktar = 0;

      Select Sum(rafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.SatisMiktar) from HamStokDetay
      left join RafDetay On RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId
      where HamStokDetay.HamStokMasterId = :HamStokMasterId
      into :KullanilanMiktar;

      if (KullanilanMiktar is null) then KullanilanMiktar = 0;

      IptalPartiMiktar = 0;

      Select Sum(RafDetay.Miktar) from HamStokDetay
         left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId
         where HamStokDetay.HamStokMasterId = :HamStokMasterId and RafDetay.PartiStokIptalKayitId is not null
         into :IptalPartiMiktar;

      if (IptalPartiMiktar is null) then IptalPartiMiktar = 0;

      KalanMiktar = Miktar - KullanilanMiktar + IptalPartiMiktar;

      if (DovizID <> RaporDovizID) then
      begin
        select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :RaporDovizID, Current_Date) into :DovizKuru;
      end
      else DovizKuru = 1;

      RaporBirimFiyat = BirimFiyat * DovizKuru;

      Select OrmeGirisIrsaliyeNo, OrmeGirisIrsaliyeTarihi, GirisOrmeFirmaId, GirisOrmeFirmaAdi from HamStokMasterOrmeGirisiGetir(:HamStokMasterID) into :OrmeGirisIrsaliyeNo, OrmeGirisIrsaliyeTarihi, :OrmeGirisFirmaId, :OrmeGirisFirmaAdi;

      suspend;
    end
  end
