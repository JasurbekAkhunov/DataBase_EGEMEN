CREATE PROCEDURE DEPOSAYIMRAPORU(
    ADEPOSAYIMID INTEGER,
    AHAMTIPIID INTEGER)
RETURNS (
    DEPOSAYIMID INTEGER,
    FISNO VARCHAR(20),
    FISTARIHI TIMESTAMP,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    KALITENO INTEGER,
    RAFID VARCHAR(10),
    KILO BOYAMIKTARSIZE,
    BRUTKILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    KAPADEDI INTEGER,
    DESENVARYANTID INTEGER,
    VARYANTADI VARCHAR(30),
    DESENID DESENKODUSIZE,
    DESENADI VARCHAR(50),
    GERCEKMIKTAR NUMERIC(12,3),
    GERCEKBRUTMIKTAR NUMERIC(12,3),
    GERCEKKAPADEDI INTEGER,
    DIGERRAFLARDAKIMIKTAR NUMERIC(12,3))
AS
declare hstIplik        integer;
declare hstOrmeKumas    integer;
declare hstDokumaKumas  integer;
declare hstRibana   integer;
declare hstAksesuar integer;
declare hstElyaf    integer;
declare hstLycra    integer;
declare hstTelef    integer;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstLycra, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstLycra, :hstTelef;

  for select DepoSayim.DepoSayimID, DepoSayim.SeriNo || DepoSayim.FisNo, DepoSayim.FisTarihi, DepoSayim.RafID, DepoSayimDetay.HamID, Ham.Adi, DepoSayimDetay.RenkId, Renk.Adi, DepoSayimDetay.Lot, DepoSayimDetay.MelanjKodu,
         DepoSayimDetay.FlamKodu, DepoSayimDetay.KaliteNo, DepoSayimDetay.DesenId, SHDesen.Adi, SHDesenVaryant.VaryantAdi, DepoSayimDetay.DesenVaryantId
          from DepoSayimDetay
    left join DepoSayim on DepoSayim.DepoSayimID = DepoSayimDetay.DepoSayimID
    left join Ham on Ham.HamId = DepoSayimDetay.HamId
    Left join Renk on Renk.RenkID = DepoSayimDetay.RenkId
    left join Desen ShDesen on ShDesen.DesenId = DepoSayimDetay.DesenId
    left join DesenVaryant SHDesenVaryant on SHDesenVaryant.DesenVaryantId = DepoSayimDetay.DesenVaryantId
    where DepoSayim.DepoSayimID = :ADepoSayimID and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
          (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or
          ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 2) or
          (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 3) or :AHamTipiId is null)
  Union
  Select DepoSayim.DepoSayimID, DepoSayim.SeriNo || DepoSayim.FisNo, DepoSayim.FisTarihi, DepoSayim.RafID, OrmeRaf.HamID, Ham.Adi, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.Lot, OrmeRaf.MelanjKodu,
         iif(OrmeRaf.FlamKodu = '', Null, OrmeRaf.FlamKodu), OrmeRaf.KaliteNo, OrmeRaf.DesenId, SHDesen.Adi, SHDesenVaryant.VaryantAdi, OrmeRaf.DesenVaryantId
        from DepoSayim
    Left Join OrmeRaf on OrmeRaf.RafID = DepoSayim.RafID
    left join Ham on Ham.HamId = OrmeRaf.HamId
    Left join Renk on Renk.RenkID = OrmeRaf.RenkId
    left join Desen ShDesen on ShDesen.DesenId = OrmeRaf.DesenId
    left join DesenVaryant SHDesenVaryant on SHDesenVaryant.DesenVaryantId = OrmeRaf.DesenVaryantId
    where DepoSayim.DepoSayimID = :ADepoSayimID and OrmeRaf.StokVar = 1 and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
          (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or
          ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 2) or
          (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 3) or :AHamTipiId is null)
  into :DepoSayimID, :FisNo, FisTarihi, :RafID, :HamId, :HamAdi, :RenkId, :RenkAdi,
       :Lot, :MelanjKodu, :FlamKodu, :KaliteNo, :DesenId, :DesenAdi, :VaryantAdi, :DesenVaryantId do
  begin
    GercekMiktar = 0;
    GercekBrutMiktar = 0;
    GercekKapAdedi = 0;
    Kilo = 0;
    BrutKilo = 0;
    Metre = 0;
    KapAdedi = 0;
    DigerRaflardakiMiktar = Null;

    select Sum(DepoSayimDetay.Kilo), Sum(DepoSayimDetay.BrutKilo), Sum(DepoSayimDetay.Metre), Sum(DepoSayimDetay.KapAdedi) from DepoSayimDetay
      where DepoSayimDetay.DepoSayimID = :ADepoSayimID and DepoSayimDetay.HamID = :HamId and
            IntKarsilastir(DepoSayimDetay.RenkID, :RenkId) = 1 and
            StrKarsilastir(DepoSayimDetay.Lot, :Lot) = 1 and
            StrKarsilastir(DepoSayimDetay.MelanjKodu, :MelanjKodu) = 1 and
            StrKarsilastir(DepoSayimDetay.FlamKodu, :FlamKodu) = 1 and
            StrKarsilastir(DepoSayimDetay.DesenId, :DesenId) = 1 and
            IntKarsilastir(DepoSayimDetay.DesenVaryantId, :DesenVaryantId) = 1 and
            IntKarsilastir(DepoSayimDetay.KaliteNo, :KaliteNo) = 1
    into :Kilo, :BrutKilo, :Metre, :KapAdedi;

    Select Sum(OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari),
           Sum(OrmeRaf.BrutGirisMiktari - OrmeRaf.BrutCikisMiktari),
           Sum(OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi)
            from OrmeRaf
    where  OrmeRaf.RafID = :RafID and OrmeRaf.HamID = :HamId and OrmeRaf.StokVar = 1 and
            IntKarsilastir(OrmeRaf.RenkID, :RenkId) = 1 and
            StrKarsilastir(OrmeRaf.Lot, :Lot) = 1 and
            StrKarsilastir(OrmeRaf.MelanjKodu, :MelanjKodu) = 1 and
            StrKarsilastir(OrmeRaf.FlamKodu, :FlamKodu) = 1 and
            StrKarsilastir(OrmeRaf.DesenId, :DesenId) = 1 and
            IntKarsilastir(OrmeRaf.DesenVaryantId, :DesenVaryantId) = 1 and
            IntKarsilastir(OrmeRaf.KaliteNo, :KaliteNo) = 1
    into :GercekMiktar, :GercekBrutMiktar, :GercekKapAdedi;

    Select Sum(OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari) from OrmeRaf
    where  OrmeRaf.RafID <> :RafID and OrmeRaf.HamID = :HamId and OrmeRaf.StokVar = 1 and
            IntKarsilastir(OrmeRaf.RenkID, :RenkId) = 1 and
            StrKarsilastir(OrmeRaf.Lot, :Lot) = 1 and
            StrKarsilastir(OrmeRaf.MelanjKodu, :MelanjKodu) = 1 and
            StrKarsilastir(OrmeRaf.FlamKodu, :FlamKodu) = 1 and
            StrKarsilastir(OrmeRaf.DesenId, :DesenId) = 1 and
            IntKarsilastir(OrmeRaf.DesenVaryantId, :DesenVaryantId) = 1 and
            IntKarsilastir(OrmeRaf.KaliteNo, :KaliteNo) = 1
    into :DigerRaflardakiMiktar;

    suspend;
  end
end
