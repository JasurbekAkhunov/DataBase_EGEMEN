CREATE PROCEDURE HAMSTOKHAREKETKARTI(
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    HAREKETTIPISTR VARCHAR(50),
    TARIH TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYENO IRSALIYENOSIZE,
    FISNOSTR VARCHAR(30),
    PARTIIDSTR VARCHAR(30),
    MIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    ACIKLAMA1 VARCHAR(75),
    ACIKLAMA2 VARCHAR(75),
    GIRISCIKIS INTEGER)
AS
Declare variable FFirmaId varchar(20);
Declare variable PartiYil integer;
Declare variable PartiId  integer;
Declare variable FIrsaliyeNo  IrsaliyeNoSize;
Declare variable FIrsaliyeTarihi TimeStamp;
Declare variable FHareketTipiId integer;
Declare variable FHamIadeRezervId integer;
Declare variable FTamir integer;
Declare variable FFason integer;
Declare variable FisYil integer;
Declare variable FisNo  integer;
Declare variable EvrakTipi integer;
begin
  Select FirmaKodumuz from sistem where sistemid = 1 into :FFirmaId;

  if (AFirmaId is null) then
  begin
    AFirmaId = FFirmaId;
    FFason = 0;
  end
  else FFason = 1;

  select sum(girisMiktar) - sum(PartilenenMiktar) - sum(iadeedilenMiktar) - sum(satisMiktar) from HamStokKumilatif
    where HamStokKumilatif.Firmaid = :AFirmaId and HamStokKumilatif.HamId = :AHamId and HamStokKumilatif.Tarih < :ABaslangicTarihi
    into :Miktar;

  if (Miktar is null) then Miktar = 0;

  Tarih = ABaslangicTarihi - 1;
  HareketTipiStr = TimeStampToDateStr(ABaslangicTarihi) || ' Tarihi Ã–ncesi Devreden ';
  GirisCikis = 1;
  suspend;


  For Select HamStokIrsaliye.EvrakTipi, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokDetay.Miktar,
             HamStokMaster.Tamir from HamStokDetay
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeid = HamStokDetay.HamStokIrsaliyeid
    left join Firma On Firma.FirmaId = HamStokIrsaliye.FirmaId
    where HamStokMaster.PartiSinifId is null and HamStokDetay.HamId = :AHamId and HamStokDetay.Tarih >= :ABaslangicTarihi and HamStokDetay.Tarih <= :ABitisTarihi and
          HamStokDetay.StokFirmaID = :AFirmaId
    into :EvrakTipi, :Tarih, :IrsaliyeNo, :FirmaId, :FirmaAdi, :Miktar, :FTamir do
  begin
    if (FTamir = 1) then HareketTipiStr = 'Tamir GiriÅŸ';
    else HareketTipiStr = 'GiriÅŸ';

    if (EvrakTipi = 0) then
    begin
      HareketTipiStr = HareketTipiStr || ' (FiÅŸ)';
      FisNoStr = IrsaliyeNo;
      IrsaliyeNo = Null;
    end

    GirisCikis = 1;
    suspend;
  end

  IrsaliyeNo = Null;
  Aciklama1 = '';
  NetMiktar = 0;

  For Select PartiStokDetay.PartilemeTarihi, Parti.PartiYil, Parti.PartiId, Parti.FirmaId, Firma.Adi, PartiStokDetay.Miktar from PartiStokDetay
    left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma On Firma.FirmaId = Parti.FirmaId
    where PartiStokDetay.HamId = :AHamId and PartiStokDetay.PartilemeTarihi >= :ABaslangicTarihi and PartiStokDetay.PartilemeTarihi <= :ABitisTarihi and
         ( (Parti.FirmaID = :AFirmaId and Parti.Fason = 1 and :FFason = 1) or (Parti.Fason = 0 and :FFason = 0) )
         /*and partiStok.partiStokIptalKayitId is null
         Partilenen ayri gosterildi, Leman istedi, oynama
         */
    into :Tarih, :PartiYil, :PartiId, :FirmaId, :FirmaAdi, :Miktar do
  begin
    HareketTipiStr = 'Partileme';

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId)  || ' ';
    GirisCikis = -1;
    suspend;
  end

  IrsaliyeNo = Null;
  Aciklama1 = '';
  NetMiktar = 0;

  For Select PartiStokDetay.PartilemeTarihi, Parti.PartiYil, Parti.PartiId, Parti.FirmaId, Firma.Adi, PartiStokDetay.Miktar from PartiStokIptal
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStokIptal.PartiSinifId
    left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma On Firma.FirmaId = Parti.FirmaId
    where PartiStokDetay.HamId = :AHamId and PartiStokIptal.IptalTarihi >= :ABaslangicTarihi and PartiStokIptal.IptalTarihi <= :ABitisTarihi and
         ( (Parti.FirmaID = :AFirmaId and Parti.Fason = 1 and :FFason = 1) or (Parti.Fason = 0 and :FFason = 0) ) and PartiStok.partiStokIptalKayitId is not null
    into :Tarih, :PartiYil, :PartiId, :FirmaId, :FirmaAdi, :Miktar do
  begin
    HareketTipiStr = 'Ä°ptal';

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId)  || ' ';
    GirisCikis = 1;
    suspend;
  end

  PartiIdStr = Null;
  NetMiktar = 0;

  For Select BoyaIrsaliyeDetay.HareketTipiId, BoyaIrsaliyeDetay.HamIadeRezervId, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliyeDetay.Miktar,
             HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.IrsaliyeNo, HamStokIadeSebebi.Adi from BoyaIrsaliyeDetay
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeid = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    left join RafDetay on RafDetay.RafDetayId = BoyaIrsaliyeDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId /* hangi irsaliyeden iade etti */
    left join Firma On Firma.FirmaId = BoyaIrsaliye.FirmaId /* kime gitti */
    left join HamIadeRezerv on HamIadeRezerv.HamIadeRezervid = BoyaIrsaliyeDetay.HamIadeRezervId
    left join HamStokIadeSebebi on HamStokIadeSebebi.HamStokIadeSebebiid = HamIadeRezerv.HamStokIadeSebebiid /* iade ise sebebi icin */
    where BoyaIrsaliyeDetay.HamId = :AHamId and BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliyeDetay.IrsaliyeTarihi <= :ABitisTarihi and
          BoyaIrsaliyeDetay.RafDetayId is not null and HamStokDetay.StokFirmaID = :AFirmaId
    into :FHareketTipiId, :FHamIadeRezervId, :Tarih, :IrsaliyeNo, :FirmaId, :FirmaAdi, :Miktar, :FIrsaliyeTarihi, :FIrsaliyeNo, :Aciklama2 do
  begin
    if (FHareketTipiId = 0 and FHamIadeRezervId is not null) then HareketTipiStr = 'Fason Ä°ade';
    else if (FHareketTipiId = 0 and FHamIadeRezervId is null) then HareketTipiStr = 'SatÄ±ÅŸ';
    else if (FHareketTipiId = 1) then HareketTipiStr = 'AlÄ±ÅŸ Ä°ade';

    GirisCikis = -1;
    Aciklama1 = EgemenIntegerToStr(FIrsaliyeNo) || ' Nolu ' || TimeStampToDateStr(FIrsaliyeTarihi) || ' Tar. Ä°rsaliye''den ';
    suspend;
  end

  Aciklama2 = Null;
  Aciklama1 = Null;
  FIrsaliyeTarihi = NUll;
  FIrsaliyeNo = NUll;
  FirmaId = Null;
  FirmaAdi = null;
  PartiIdStr = Null;
  IrsaliyeNo = Null;
  NetMiktar = 0;

  For Select HamStokSarfFisi.FisTarihi, HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo, HamStokSarfFisiDetay.Miktar,
             HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.IrsaliyeNo from HamStokSarfFisiDetay
    left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiid = HamStokSarfFisiDetay.HamStokSarfFisiId
    left join RafDetay on RafDetay.RafDetayId = HamStokSarfFisiDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
    where HamStokSarfFisiDetay.HamId = :AHamId and HamStokSarfFisiDetay.FisTarihi >= :ABaslangicTarihi and HamStokSarfFisiDetay.FisTarihi <= :ABitisTarihi and
          HamStokSarfFisiDetay.RafDetayId is not null and HamStokDetay.StokFirmaID = :AFirmaId
    into :Tarih, :FisYil, :FisNo, :Miktar, :FIrsaliyeTarihi, :FIrsaliyeNo do
  begin
    GirisCikis = -1;
    HareketTipiStr = 'Sarf FiÅŸi (HamStok)';

    FisNoStr = '(' || EgemenIntegerToStr(FisYil)  || ') ' || EgemenIntegerToStr(FisNo)  || ' ';

    Aciklama1 = EgemenIntegerToStr(FIrsaliyeNo) || ' Nolu ' || TimeStampToDateStr(FIrsaliyeTarihi) || ' Tar. Ä°rsaliye''den ';
    suspend;
  end
end
