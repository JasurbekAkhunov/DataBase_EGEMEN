CREATE PROCEDURE MODELURUNGETIR(
    AHGSIPARISKAYITID INTEGER,
    AHGSIPARISDETAYASORTIID INTEGER)
RETURNS (
    MODELID MODELKODUSIZE,
    URUNID URUNKODUSIZE,
    MODELKUMASID INTEGER,
    MODELBEDENID INTEGER,
    RENKADI ADI40,
    KISAKODU VARCHAR(7),
    RESIMDOSYASI VARCHAR(255),
    RENKID INTEGER,
    BEDENID INTEGER,
    BEDENADI ADI30,
    MIKTAR BOYAMIKTARSIZE,
    HGSIPARISDETAYURUNID INTEGER,
    HGSIPARISDETAYID INTEGER)
AS
Declare Variable FDevam              EgeBoolean;
Declare Variable FOnayDurumu         EgeBoolean;
Declare Variable FModelBedenKapali   EgeBoolean;
Declare Variable FModelKumasKapali   EgeBoolean;
Declare Variable FModelBedenMiktar   BoyaMiktarSize;
Declare Variable FModelKumasMiktar   BoyaMiktarSize;
begin
  For Select Model.ModelId, Urun.UrunId, Urun.ModelKumasId, Urun.ModelBedenId, ModelKumas.RenkAdi, Beden.KisaKodu, Urun.ResimDosyasi,
             ModelKumas.RenkId, Beden.BedenId, Iif(:AHGSiparisDetayAsortiId is Null, Beden.Adi, Beden.KisaKodu), HGSiparisDetayUrun.Miktar,
             HGSiparisDetayUrunId, HGSiparisDetay.HGSiparisDetayId, ModelBeden.Kapali, ModelKumas.Kapali, HGSiparisDetay.OnayDurumu
    From HGSiparisDetay
    Join Urun on Urun.ModelKayitId = HGSiparisDetay.ModelKayitId and Urun.ModelParcaId is null and Urun.ModelBedenId is not null
    Left Join Model on Model.ModelKayitId = HGSiparisDetay.ModelKayitId
    Left Join ModelKumas on ModelKumas.ModelKumasId = Urun.ModelKumasId
    Left Join ModelBeden on ModelBeden.ModelBedenId = Urun.ModelBedenId
    Left Join Beden on Beden.BedenId = ModelBeden.BedenId
    Left Join HGSiparisDetayUrun on HGSiparisDetayUrun.ModelBedenId = ModelBeden.ModelBedenId and HGSiparisDetayUrun.ModelKumasId = ModelKumas.ModelKumasId and HGSiparisDetayUrun.HGSiparisDetayId = HGSiparisDetay.HGSiparisDetayId
              And Iif(:AHGSiparisDetayAsortiId is Null, Iif(HGSiparisDetayUrun.HGSiparisDetayAsortiId Is Null, 1, 0), Iif(HGSiparisDetayUrun.HGSiparisDetayAsortiId = :AHGSiparisDetayAsortiId, 1, 0)) = 1
    Where HGSiparisDetay.HGSiparisKayitId = :AHGSiparisKayitId
    Order By HGSiparisDetay.HGSiparisDetayId, ModelKumas.ModelKumasId, ModelBeden.ModelBedenId
    Into :ModelId, :UrunId, :ModelKumasId, :ModelBedenId, :RenkAdi, :KisaKodu, :ResimDosyasi, :RenkId, :BedenId, :BedenAdi, :Miktar, :HGSiparisDetayUrunId, :HGSiparisDetayId, :FModelBedenKapali, :FModelKumasKapali, :FOnayDurumu do
  Begin
    FDevam = 1;

    if  (((FOnayDurumu = 0) and (FModelBedenKapali = 1 or FModelKumasKapali = 1)) or (FOnayDurumu = 1)) then
    begin
      Select Sum(Miktar) From HGSiparisDetayUrun
      Where HGSiparisDetayUrun.ModelBedenId = :ModelBedenId and HGSiparisDetayUrun.HGSiparisDetayId = :HGSiparisDetayId and Iif(:AHGSiparisDetayAsortiId is Null, Iif(HGSiparisDetayUrun.HGSiparisDetayAsortiId Is Null, 1, 0), Iif(HGSiparisDetayUrun.HGSiparisDetayAsortiId Is Not Null, 1, 0)) = 1
      Into :FModelBedenMiktar;

      Select Sum(Miktar) From HGSiparisDetayUrun
      Where HGSiparisDetayUrun.ModelKumasId = :ModelKumasId and HGSiparisDetayUrun.HGSiparisDetayId = :HGSiparisDetayId and Iif(:AHGSiparisDetayAsortiId is Null, Iif(HGSiparisDetayUrun.HGSiparisDetayAsortiId Is Null, 1, 0), Iif(HGSiparisDetayUrun.HGSiparisDetayAsortiId Is Not Null, 1, 0)) = 1
      Into :FModelKumasMiktar;

      FModelBedenMiktar = Coalesce(FModelBedenMiktar, 0);
      FModelKumasMiktar = Coalesce(FModelKumasMiktar, 0);

      if (FOnayDurumu = 0) then
      begin
        if (FModelBedenKapali = 1 And FModelKumasKapali = 0) then
          FDevam = Iif(FModelBedenMiktar > 0, 1, 0);
        else
          if (FModelBedenKapali = 0 And FModelKumasKapali = 1) then
            FDevam = Iif(FModelKumasMiktar > 0, 1, 0);
          else
            if (FModelBedenKapali = 1 And FModelKumasKapali = 1) then
              FDevam = Iif(FModelKumasMiktar > 0 And FModelBedenMiktar > 0, 1, 0);
      end
      Else
        FDevam = Iif(FModelKumasMiktar > 0 And FModelBedenMiktar > 0, 1, 0);
    end

    if (FDevam = 1) then Suspend;
  End
end
