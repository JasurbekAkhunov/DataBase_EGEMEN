CREATE PROCEDURE ZANCANERBOBINICIETIKETGETIR(
    APARTISINIFID INTEGER)
RETURNS (
    DEGER VARCHAR(100),
    SIRASI INTEGER)
AS
declare variable BEPartiNoSirasi                 integer;
declare variable BESiparisNoSirasi               integer;
declare variable BEFirmaAdiSirasi                integer;
declare variable BELabReceteKoduSirasi           integer;
declare variable BELabReceteAdiSirasi            integer;
declare variable BEReceteKoduSirasi              integer;
declare variable BEReceteAdiSirasi               integer;
declare variable BEEskiLabReceteKoduSirasi       integer;
declare variable BELabMusteriRenkNoSirasi        integer;
declare variable BESipMusteriRenkKoduAdiSirasi   integer;
declare variable BEMusteriSiparisNoSirasi        integer;
declare variable BEHamAdiSirasi                  integer;
declare variable BEHamAdi2Sirasi                 integer;
declare variable BELotSirasi                     integer;
declare variable BETarihSirasi                   integer;
declare variable BESaatSirasi                    integer;
declare variable BEIplikCinsiSirasi              integer;
declare variable BEIplikCinsiDigerAdiSirasi      integer;
declare variable BEKalinlikSirasi                integer;
declare variable BEOzellikSirasi                 integer;
declare variable BEOzellikDigerAdiSirasi         integer;
declare variable BEElyafSinifSirasi              integer;
declare variable BELabReceteKoduveAdiSirasi      integer;
declare variable BEKonikCinsiSirasi              integer;
declare variable BEKonikRengiSirasi              integer;

declare variable PartiYil         integer;
declare variable PartiId          integer;
declare variable BoyaSiparisYil   integer;
declare variable BoyaSiparisId    integer;
declare variable FirmaAdi         varchar(100);
declare variable LabReceteKodu    varchar(20);
declare variable LabReceteAdi     varchar(40);
declare variable ReceteId         varchar(20);
declare variable ReceteAdi        varchar(40);
declare variable EskiLabReceteKodu    varchar(20);
declare variable LabMusteriRenkKodu   varchar(30);
declare variable MusteriRenkKoduveAdi varchar(30);
declare variable MusteriSiparisNo varchar(50);
declare variable HamAdi  varchar(255);
declare variable HamAdi2 varchar(255);
declare variable Lot    varchar(100);
declare variable PSKKEtiketTipiId   integer;
declare variable KalinlikAdi        varchar(50);
declare variable IplikCinsiAdi      varchar(80);
declare variable IplikCinsiDigerAdi varchar(30);
declare variable OzellikAdi         varchar(40);
declare variable OzellikDigerAdi    varchar(80);
declare variable ElyafSinifAdi      varchar(20);
declare variable KonikCinsiAdi      varchar(20);
declare variable KonikRengiAdi      varchar(20);
declare variable DefaultIplikPsKKEtiketTipiId  integer;
declare variable EgemenFirmaId     integer;
declare variable PartiSiraNo       integer;
declare variable BELGEHAMADI varchar(255);
declare variable BEFirmaAdimizSirasi integer;
declare variable FirmaKodumuz varchar(20);
declare variable FirmaAdimiz varchar(100);
begin
  Select DefaultIplikPsKKEtiketTipiId, EgemenFirmaId, FirmaKodumuz from sistem where sistemID = 1 into :DefaultIplikPsKKEtiketTipiId, :EgemenFirmaId, :FirmaKodumuz;

  if (DefaultIplikPsKKEtiketTipiId = 0) then DefaultIplikPsKKEtiketTipiId = Null;

  select Parti.PartiYil, Parti.PartiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
         Firma.Adi, BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, Parti.ReceteId, Recete.Adi,
         LabRecete.EskiLabReceteKodu, LabRecete.MusteriRenkKodu, BoyaSiparisDetay.MusteriRenkKoduveAdi,
         BoyaSiparisDetay.MusteriSiparisNo, Ham.Adi, Ham.Adi2, BoyaSiparis.PSKKEtiketTipiId,
         Kalinlik.Adi, IplikCinsi.Adi, Ozellik.Adi, ElyafSinif.Adi, KonikCinsi.Adi, KonikRengi.Adi, PartiStok.PartiSiraNo, IplikCinsi.DigerAdi, Ozellik.DigerAdi,
         case when coalesce(BoyaSiparisDetay.BelgeHamAdi,'')<>'' then BoyaSiparisDetay.BelgeHamAdi else left(Ham.Adi, 255) end from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join Ozellik on Ozellik.OzellikId = Ham.OzellikId
    left join ElyafSinif on ElyafSinif.ElyafSinifId = Ham.ElyafSinifId
    left join KonikCinsi on KonikCinsi.KonikCinsiId = PartiStok.KonikCinsiId
    left join KonikRengi on KonikRengi.KonikRengiId = PartiStok.KonikRengiId
    where PartiStok.PartiSinifId = :APartiSinifId
    into :PartiYil, :PartiId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaAdi, :LabReceteKodu, :LabReceteAdi, :ReceteId, :ReceteAdi,
         :EskiLabReceteKodu, :LabMusteriRenkKodu, :MusteriRenkKoduveAdi, :MusteriSiparisNo, :HamAdi, :HamAdi2, :PSKKEtiketTipiId,
         :KalinlikAdi, :IplikCinsiAdi, :OzellikAdi, :ElyafSinifAdi, :KonikCinsiAdi, :KonikRengiAdi, :PartiSiraNo, :IplikCinsiDigerAdi, :OzellikDigerAdi, :BelgeHamAdi;

  if (PartiId is null) then exit;

  if (PSKKEtiketTipiId is null) then PSKKEtiketTipiId = DefaultIplikPsKKEtiketTipiId;

  if (PSKKEtiketTipiId is null) then exit;

  Select BEPartiNoSirasi, BESiparisNoSirasi, BEFirmaAdiSirasi, BELabReceteKoduSirasi, BELabReceteAdiSirasi, BEReceteKoduSirasi,
         BEReceteAdiSirasi,  BEEskiLabReceteKoduSirasi, BELabMusteriRenkNoSirasi, BESipMusteriRenkKoduAdiSirasi, BEMusteriSiparisNoSirasi,
         BEHamAdiSirasi, BELotSirasi, BETarihSirasi, BESaatSirasi, BEIplikCinsiSirasi, BEKalinlikSirasi, BEOzellikSirasi,
         BEElyafSinifSirasi, BELabReceteKoduveAdiSirasi, BEKonikCinsiSirasi, BEKonikRengiSirasi, BEIplikCinsiDigerAdiSirasi, BEOzellikDigerAdiSirasi,
         BEHamAdi2Sirasi, BEFirmaAdimizSirasi from PSKKEtiketTipi
    where PSKKEtiketTipi.PSKKEtiketTipiID = :PSKKEtiketTipiId
    into :BEPartiNoSirasi, :BESiparisNoSirasi, :BEFirmaAdiSirasi, :BELabReceteKoduSirasi, :BELabReceteAdiSirasi, :BEReceteKoduSirasi,
         :BEReceteAdiSirasi,  :BEEskiLabReceteKoduSirasi, :BELabMusteriRenkNoSirasi, :BESipMusteriRenkKoduAdiSirasi, :BEMusteriSiparisNoSirasi,
         :BEHamAdiSirasi, :BELotSirasi, :BETarihSirasi, :BESaatSirasi, :BEIplikCinsiSirasi, :BEKalinlikSirasi, :BEOzellikSirasi,
         :BEElyafSinifSirasi, :BELabReceteKoduveAdiSirasi, :BEKonikCinsiSirasi, :BEKonikRengiSirasi, :BEIplikCinsiDigerAdiSirasi, :BEOzellikDigerAdiSirasi,
     :BEHamAdi2Sirasi, :BEFirmaAdimizSirasi;

  Select BobinEtiketiLot from PartiStokLotBilgisi(:APartiSinifId) into :Lot;


  if (BEPartiNoSirasi > 0) then
  begin
    Sirasi = BEPartiNoSirasi;
    if (EgemenFirmaId = 19) then Deger = EgemenIntegerToStr(PartiId);
    else if (EgemenFirmaId = 34) then Deger = EgemenIntegerToStr(PartiYil)  || '-' || EgemenIntegerToStr(PartiId) || '.' || EgemenIntegerToStr(PartiSiraNo);
    else if (EgemenFirmaId = 75) then Deger = EgemenIntToLzStr((PartiYil-2000),2)  || '-' || EgemenIntegerToStr(PartiId);
    else Deger = EgemenIntegerToStr(PartiYil)  || '-' || EgemenIntegerToStr(PartiId);

    suspend;
  end

  if (BESiparisNoSirasi > 0) then
  begin
    Sirasi = BESiparisNoSirasi;
    Deger = EgemenIntegerToStr(BoyaSiparisYil)  || '-' || EgemenIntegerToStr(BoyaSiparisId);
    suspend;
  end

  if (BEFirmaAdiSirasi > 0) then
  begin
    Sirasi = BEFirmaAdiSirasi;
    if (EgemenFirmaId = 34) then Deger = Substring(FirmaAdi From 1 for 25);
    else Deger = Substring(FirmaAdi From 1 for 30);
    suspend;
  end

  if (BELabReceteKoduSirasi > 0) then
  begin
    Sirasi = BELabReceteKoduSirasi;
    Deger = LabReceteKodu;
    suspend;
  end

  if (BELabReceteAdiSirasi > 0) then
  begin
    Sirasi = BELabReceteAdiSirasi;
    Deger = LabReceteAdi;
    suspend;
  end

  if (BEReceteKoduSirasi > 0) then
  begin
    Sirasi = BEReceteKoduSirasi;
    if (EgemenFirmaId = 46) then Deger = SubString(ReceteId from 1 for (position('.', ReceteId)+1));   /*Başyazıcı*/
    else Deger = ReceteId;
    suspend;
  end

  if (BEReceteAdiSirasi > 0) then
  begin
    Sirasi = BEReceteAdiSirasi;
    Deger = ReceteAdi;
    suspend;
  end


  if (BEEskiLabReceteKoduSirasi > 0) then
  begin
    Sirasi = BEEskiLabReceteKoduSirasi;
    Deger = EskiLabReceteKodu;
    suspend;
  end

  if (BELabMusteriRenkNoSirasi > 0) then
  begin
    Sirasi = BELabMusteriRenkNoSirasi;
    Deger = labMusteriRenkKodu;
    suspend;
  end

  if (BESipMusteriRenkKoduAdiSirasi > 0) then
  begin
    Sirasi = BESipMusteriRenkKoduAdiSirasi;
    Deger = MusteriRenkKoduveAdi;
    suspend;
  end

  if (BEMusteriSiparisNoSirasi > 0) then
  begin
    Sirasi = BEMusteriSiparisNoSirasi;
    Deger = MusteriSiparisNo;
    suspend;
  end

  if (BEHamAdiSirasi > 0) then
  begin
    Sirasi = BEHamAdiSirasi;
  if (EgemenFirmaId = 142) then Deger = Substring(BelgeHamAdi From 1 for 24);  /*ABM Cotton*/
    else Deger = Substring(HamAdi From 1 for 24);
    suspend;
  end

  if (BEHamAdi2Sirasi > 0) then
  begin
    Sirasi = BEHamAdi2Sirasi;
    Deger = Substring(HamAdi2 From 1 for 24);
    suspend;
  end

  if (BELotSirasi > 0) then
  begin
    Sirasi = BELotSirasi;
    if (EgemenFirmaId = 46) then Deger = Substring(Lot From 5 for Char_Length(Lot) - 5); /*Başyazıcı*/
    else Deger = Lot;
    suspend;
  end

  if (BETarihSirasi > 0) then
  begin
    Sirasi = BETarihSirasi;
    Deger = TimeStampToDateStr(Current_Date);
    suspend;
  end

  if (BESaatSirasi > 0) then
  begin
    Sirasi = BESaatSirasi;
    Deger = current_TimeStamp(0);
    Deger = Substring(Deger From 1 for 16);
    suspend;
  end

  if (BEIplikCinsiSirasi > 0) then
  begin
    Sirasi = BEIplikCinsiSirasi;
    if (EgemenFirmaId = 0) then Deger = Substring(IplikCinsiAdi From 1 for 8);/*NY*/
    else Deger = IplikCinsiAdi;
    suspend;
  end

  if (BEKalinlikSirasi > 0) then
  begin
    Sirasi = BEKalinlikSirasi;
    Deger = KalinlikAdi;
    suspend;
  end

  if (BEOzellikSirasi > 0) then
  begin
    if (OzellikAdi is null) then OzellikAdi = '';
    Sirasi = BEOzellikSirasi;
    Deger = OzellikAdi;
    suspend;
  end

  if (BEOzellikDigerAdiSirasi > 0) then
  begin
    if (OzellikDigerAdi is null) then OzellikDigerAdi = '';
    Sirasi = BEOzellikDigerAdiSirasi;
    Deger = OzellikDigerAdi;
    suspend;
  end

  if (BEIplikCinsiDigerAdiSirasi > 0) then
  begin
    if (IplikCinsiDigerAdi is null) then IplikCinsiDigerAdi = '';
    Sirasi = BEIplikCinsiDigerAdiSirasi;
    Deger = IplikCinsiDigerAdi;
    suspend;
  end

  if (BEElyafSinifSirasi > 0) then
  begin
    if (ElyafSinifAdi is null) then ElyafSinifAdi = '';
    Sirasi = BEElyafSinifSirasi;
    Deger = ElyafSinifAdi;
    suspend;
  end

  if (BELabReceteKoduveAdiSirasi > 0) then
  begin
    Sirasi = BELabReceteKoduveAdiSirasi;
    Deger = LabReceteKodu || '/' || LabReceteAdi;
    suspend;
  end

  if (BEKonikRengiSirasi > 0) then
  begin
    if (KonikRengiAdi is null) then KonikRengiAdi = '';

    Sirasi = BEKonikRengiSirasi;
    Deger = KonikRengiAdi;
    suspend;
  end

  if (BEKonikCinsiSirasi > 0) then
  begin
    if (KonikCinsiAdi is null) then KonikCinsiAdi = '';

    Sirasi = BEKonikCinsiSirasi;
    Deger = KonikCinsiAdi;
    suspend;
  end
  if (BEFirmaAdimizSirasi > 0) then
  begin
    select Adi from Firma where Firma.FirmaId = :FirmaKodumuz into  :FirmaAdimiz;

    if (FirmaAdimiz is null) then FirmaAdimiz = '';

    Sirasi = BEFirmaAdimizSirasi;
    Deger = FirmaAdimiz;
    suspend;
  end
end
