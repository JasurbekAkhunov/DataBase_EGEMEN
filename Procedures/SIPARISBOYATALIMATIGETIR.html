CREATE PROCEDURE SIPARISBOYATALIMATIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYATALIMATKAYITID INTEGER,
    TARIH TIMESTAMP,
    TALIMATTIPIID INTEGER,
    TALIMATTIPIIDSTR VARCHAR(12),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER,
    SIPARISSIRANO INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPFIRMAID VARCHAR(20),
    SIPFIRMAADI VARCHAR(100),
    URETIMECIKISIRSNO VARCHAR(30),
    MIKTAR DOUBLE PRECISION,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    RECETEKODU VARCHAR(20),
    HAMRENKADI VARCHAR(50),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    TERMINTARIHI TIMESTAMP,
    BOYATALIMATNO VARCHAR(30),
    GIDENMIKTAR DOUBLE PRECISION,
    IADEALINANMIKTAR DOUBLE PRECISION,
    BRUTGELENMIKTAR DOUBLE PRECISION,
    NETGELENMIKTAR DOUBLE PRECISION,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KAPALI INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For select BoyaTalimat.BoyaTalimatKayitId, BoyaTalimat.TalimatTipiId, BoyaTalimat.Tarih, BoyaTalimat.BoyaSiparisKayitId, BoyaTalimat.BoyaSiparisDetayId, BoyaTalimat.BoyaSiparisBaskiId,
             OrmeSiparisDetay.SiparisSiraNo, BoyaTalimat.FirmaId, Firma.Adi, BoyaTalimat.Miktar, BoyaTalimat.RenkId, TalimatRenk.Adi, BoyaTalimat.ReceteKodu,
             DesenVaryant.VaryantAdi, HamRenk.Adi, BoyaTalimat.BoyaTalimatNo, BoyaTalimat.TerminTarihi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaTalimat.BirimFiyat, BoyaTalimat.DovizId, BoyaTalimat.VadeGun, BoyaTalimat.GidenMiktar, BoyaTalimat.IadeAlinanMiktar,
             BoyaTalimat.BrutGelenMiktar, BoyaTalimat.NetGelenMiktar, BoyaTalimat.Notu, BoyaTalimat.Kapali,
             Ik.Adi, BoyaTalimat.InsertTarihi, Sk.Adi, BoyaTalimat.DegisimTarihi, BoyaSiparis.FirmaID, SipFirma.Adi,
             (OrmeIrsaliye.SeriNo || OrmeIrsaliye.IrsaliyeNo), BoyaSiparis.BoyaSiparisId, BoyaSiparis.BoyaSiparisYil from BoyaTalimat
    left join Firma on Firma.FirmaId = BoyaTalimat.FirmaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaTalimat.BoyaSiparisKayitId
    left join Firma SipFirma on SipFirma.FirmaId = BoyaSiparis.FirmaId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaTalimat.BoyaSiparisDetayId
    left join OrmesiparisDetay on OrmesiparisDetay.OrmesiparisDetayId = BoyaSiparisDetay.OrmesiparisDetayId
    left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisBaskiId = BoyaTalimat.BoyaSiparisBaskiId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaTalimat.SiparisProsesId
    left join Renk TalimatRenk on TalimatRenk.RenkId = BoyaTalimat.RenkId
    left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
    left join StokHareket on StokHareket.StokHareketID = BoyaTalimat.UretimeCikisStokHareketID
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeID = StokHareket.OrmeIrsaliyeID
    Left Join Kullanici IK on IK.KullaniciId = BoyaTalimat.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = BoyaTalimat.KullaniciId
    where BoyaTalimat.Boyasipariskayitid = :ABoyasipariskayitid
    order by BoyaTalimat.BoyaSiparisDetayId, BoyaTalimat.BoyaTalimatKayitId
    into :BoyaTalimatKayitId, :TalimatTipiId, :Tarih, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :SiparisSiraNo, :FirmaId, :FirmaAdi,
         :Miktar, :RenkId, :RenkAdi, :ReceteKodu, :VaryantAdi, :HamRenkAdi, :BoyaTalimatNo, :TerminTarihi, :SiparisProsesId, :SiparisProsesAdi, :BirimFiyat, :DovizId, :VadeGun,
         :GidenMiktar, :IadeAlinanMiktar, :BrutGelenMiktar, :NetGelenMiktar, :Notu, :Kapali,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :SipFirmaID, :SipFirmaAdi, :UretimeCikisIrsNo, :BoyaSiparisId, :BoyaSiparisYil do
  begin
    if (TalimatTipiId = 0) then TalimatTipiIdStr = 'Boya';
    else if (TalimatTipiId = 1) then TalimatTipiIdStr = 'Boya+Baskı';
    else if (TalimatTipiId = 2) then TalimatTipiIdStr = 'Baskı';

    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    suspend;
  end
end
