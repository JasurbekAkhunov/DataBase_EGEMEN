CREATE PROCEDURE BOLEREKMAMULDEPOTRANSFERIYAP(
    APARTISTOKTARTIID INTEGER,
    ABOBINSAYISI INTEGER,
    AKILO BOYAMIKTARSIZE,
    ABRUTKILO BOYAMIKTARSIZE,
    AMETRE BOYAMIKTARSIZE,
    ABRUTMETRE BOYAMIKTARSIZE,
    AKULLANICIID INTEGER,
    AMAMULDEPOTRANSFERFISIID INTEGER,
    ADEPOID VARCHAR(10),
    ARAFID VARCHAR(10),
    ABARKODID VARCHAR(20))
AS
declare variable KayitSayisi integer;
declare variable YeniIplikPaketlemeId Integer;
declare variable EskiIplikPaketlemeId Integer;
declare variable BarkodId  varchar(30);
declare variable YeniBarkodId varchar(30);
declare variable DepoId DepoKoduSize;
declare variable PartiSinifId   integer;
declare variable YeniPartiStokTartiId integer;

declare variable IK varchar(2);
declare variable PsKaliteKontrolID integer;
begin

  IK = SubString(ABarkodId from 1 for 2);
  if (IK = 'K-') then
  begin
    select PartiStokTartiBol.YeniBarkodId, PartiStokTartiBol.PsKaliteKontrolID, PartiStokTartiBol.YeniPartiStokTartiId
    from PartiStokTartiBol(:APartiStokTartiId, :AKilo, :AKullaniciId, 0) into :YeniBarkodId, :PsKaliteKontrolID, :YeniPartiStokTartiId;
  end

  select count(*) from PartiStokTarti where PartiStokTarti.PartiStokTartiId = :APartiStokTartiId and
  (PartiStokTarti.BoyaIrsaliyeId is not null or PartiStokTarti.HamStokSarfFisiId is not null) into :KayitSayisi;

  if (KayitSayisi > 0) then Exception KayitHatasi 'Stok Sevk yada Sarf Edilmi≈ü. Transfer Edilemez.';

  select BarkodId, DepoId, PartiSinifId from PartiStokTarti where PartiStokTarti.PartiStokTartiId = :APartiStokTartiId into :BarkodId, :DepoId, :PartiSinifId;

  if (DepoId <> :ADepoId) then exception OkutulanPaketinDeposuUygunDegil + ',Deposu :' || Depoid;

  if (IK <> 'K-') then
  begin
    update PartiStokTarti set LinkIslemi = 1,
           PartiStokTarti.BobinSayisi = PartiStokTarti.BobinSayisi - :ABobinSayisi,
           PartiStokTarti.Kilo = PartiStokTarti.Kilo - :AKilo,
           PartiStokTarti.BrutKilo = iif(PartiStokTarti.BrutKilo - :ABrutKilo >= 0, PartiStokTarti.BrutKilo - :ABrutKilo, 0),
           PartiStokTarti.Metre = PartiStokTarti.Metre - :AMetre,
           PartiStokTarti.Dara = iif(Dara - (:ABrutKilo - :AKilo) < 0, 0, Dara - (:ABrutKilo - :AKilo)),
           PartiStokTarti.BrutMetre = PartiStokTarti.BrutMetre - :ABrutMetre where PartiStokTarti.PartiStokTartiId = :APartiStokTartiId;

    YeniPartiStokTartiId = gen_id(PartiStokTarti_Id_Gen, 1);

    if (BarkodId is null) then /*iade degil, normal akis islemesi*/
    begin
      select IplikPaketlemeId from IplikPaketleme where PartiStokTartiId = :APartiStokTartiId into :EskiIplikPaketlemeId;

      update IplikPaketleme set BobinSayisi = BobinSayisi - :ABobinSayisi, NetKilo = NetKilo - :AKilo, BrutKilo = BrutKilo - :ABrutKilo,
      Dara = iif(Dara - (:ABrutKilo - :AKilo) < 0, 0, Dara - (:ABrutKilo - :AKilo)), LinkIslemi = 1 where IplikPaketlemeId = :EskiIplikPaketlemeId;

      YeniBarkodId = Null;
    end
    else
    begin
        YeniBarkodId = 'R-'||YeniPartiStokTartiId;
    end

    insert into PartiStokTarti (PartiStokTartiId, PartiKayitId, PartiSinifId, Tarih, KapAdedi, Dara, Kilo, BrutKilo, Metre, BrutMetre, Adet, BobinSayisi, KaliteNo, KaliteKontrol, DepoId, PaletNo, KaliteAciklamaTipiId, MamulProsesId, InsertKullaniciId, InsertTarihi, KullaniciId, LinkIslemi, BolunduguPartiStokTartiId, IadeBoyaIrsaliyeId, IadeBoyaIrsaliyeDetayId, BarkodId, OlcuId)
    select :YeniPartiStokTartiId, PartiKayitId, PartiSinifId, current_timestamp(0), 1, (:ABrutKilo - :AKilo), :AKilo, :ABrutKilo, :AMetre, :ABrutMetre, Adet, :ABobinSayisi, KaliteNo, KaliteKontrol, DepoId, PaletNo, KaliteAciklamaTipiId, MamulProsesId, :AKullaniciId, current_timestamp(0), :AKullaniciId, 1, :APartiStokTartiId, IadeBoyaIrsaliyeId, IadeBoyaIrsaliyeDetayId, :YeniBarkodId, OlcuId
    from PartiStokTarti where PartiStokTartiId = :APartiStokTartiId;
  end

  if (BarkodId is null) then /*iade degil, normal akis islemesi*/
  begin
    YeniIplikPaketlemeId = gen_id(IplikPaketleme_Id_Gen, 1);

    insert into IplikPaketleme (IplikPaketlemeId, PartiKayitId, PartiSinifId, MakinaId, PaketlemeTarihi, PaketlemeId,
                                 AmbalajId, AmbalajAgirligi, KoliNo, BobinSayisi, BobinDarasi, Dara, PaletAgirligi, BrutKilo,
                                 NetKilo, KaliteNo, KonikCinsiId, KonikRengiId, PartiStokTartiId, InsertKullaniciId,
                                 InsertTarihi, KullaniciId, DegisimTarihi, LinkIslemi)
     select :YeniIplikPaketlemeId, PartiKayitId, PartiSinifId, MakinaId, PaketlemeTarihi, PaketlemeId, AmbalajId, AmbalajAgirligi,
            KoliNo, :ABobinSayisi, BobinDarasi, :ABrutKilo - :AKilo, PaletAgirligi, :ABrutKilo, :AKilo, KaliteNo, KonikCinsiId, KonikRengiId,
            :YeniPartiStokTartiId, :AKullaniciId, Current_TimeStamp(0), :AKullaniciId, DegisimTarihi, 1
     from IplikPaketleme where IplikPaketlemeId = :EskiIplikPaketlemeId;
  end

  Insert into MamulDepoTransferFisiDetay(MamulDepoTransferFisiId, PartiSinifId, PartiStokTartiId, RafId, InsertKullaniciId, KullaniciId)
              values(:AMamulDepoTransferFisiId, :PartiSinifId, :YeniPartiStokTartiId, :ARafId, :AKullaniciId, :AKullaniciId);
end
