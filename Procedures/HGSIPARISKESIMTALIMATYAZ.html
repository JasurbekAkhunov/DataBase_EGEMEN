CREATE PROCEDURE HGSIPARISKESIMTALIMATYAZ(
    AHGSIPARISKAYITID INTEGER)
RETURNS (
    HGSIPARISDETAYID INTEGER,
    URUNID URUNKODUSIZE,
    RENKID INTEGER,
    BEDENID INTEGER,
    BEDENADI ADI30,
    RENKADI ADI40,
    MODELKUMASID INTEGER,
    MODELBEDENID INTEGER,
    SIPARISMIKTAR INTEGER,
    PLANLANANMIKTAR DOUBLE PRECISION,
    METOMIKTAR INTEGER,
    TASNIFMIKTAR INTEGER,
    KESILENMIKTAR DOUBLE PRECISION,
    URETILENMIKTAR DOUBLE PRECISION)
AS
begin
  For Select HGSiparisDetay.HGSiparisDetayId, Urun.UrunID, Urun.ModelKumasID, Urun.ModelBedenID, ModelKumas.RenkID, Beden.BedenID,
             ModelKumas.RenkAdi, Beden.Adi BedenAdi, HGSiparisDetayUrun.Miktar From HGSiparisDetay
   Left Join Urun ON Urun.ModelKayitId = HGSiparisDetay.ModelKayitId and Urun.ModelPARCAID IS NULL and Urun.ModelBedenID IS NOT NULL
   Left Join Model ON Model.ModelKayitId = HGSiparisDetay.ModelKayitId
   Left Join ModelKumas ON ModelKumas.ModelKumasID = Urun.ModelKumasID
   Left Join ModelBeden ON ModelBeden.ModelBedenID = Urun.ModelBedenID
   Left Join Beden ON Beden.BedenID = ModelBeden.BedenID
   Left Join HGSiparisDetayUrun ON HGSiparisDetayUrun.ModelBedenid = ModelBeden.ModelBedenID and HGSiparisDetayUrun.ModelKumasid = ModelKumas.ModelKumasid and HGSiparisDetayUrun.HGSiparisDetayId = HGSiparisDetay.HGSiparisDetayId
             and HGSiparisDetayUrun.HGSiparisDetayAsortiId IS NULL
    Where HGSiparisDetay.HGSIPARISKayitId = :AHgSiparisKayitId and
          (((Select Sum(Miktar) From HGSiparisDetayUrun Where HGSiparisDetayUrun.ModelBedenID = ModelBeden.ModelBedenID and HGSiparisDetayUrun.HGSiparisDetayId = HGSiparisDetay.HGSiparisDetayId AND HGSiparisDetayASORTIID IS NULL) > 0
            and
           (Select Sum(Miktar) From HGSiparisDetayUrun Where HGSiparisDetayUrun.ModelKumasid = ModelKumas.ModelKumasid and HGSiparisDetayUrun.HGSiparisDetayId = HGSiparisDetay.HGSiparisDetayId and HGSiparisDetayAsortiId is null) > 0)
          )
    ORDER BY HGSiparisDetay.HGSiparisDetayID, ModelKumas.ModelKumasID, ModelBeden.ModelBedenID
  into :HGSiparisDetayId, :UrunId, :ModelKumasId,  :ModelBedenId, :RenkId, :BedenId, :RenkAdi, :BedenAdi, :SiparisMiktar do
  begin
    PlanlananMiktar = 0;
    KesilenMiktar   = 0;
    MetoMiktar      = 0;
    TasnifMiktar    = 0;
    UretilenMiktar  = 0;

    Select Sum(HGIsemriPastalUrun.PlanlananMiktar), sum(HGIsemriPastalUrun.GerceklesenMiktar),
           Sum(HGIsemriPastalUrun.MetoMiktar), Sum(HGIsemriPastalUrun.TasnifMiktar) From HgIsEmri
      Left Join HGISEmriPastal on HGISEmriPastal.HgIsEmriKayitId = HgIsEmri.HgIsemriKayitId
      Left Join HGIsemriPastalUrun on HGIsemriPastalUrun.HGIsEmriPastalID = HGISEmriPastal.HGIsEmriPastalID
      Where HgIsEmri.HGSiparisDetayId = :HGSiparisDetayId and HGIsemriPastalUrun.UrunId = :UrunId
    into :PlanlananMiktar, :KesilenMiktar, :MetoMiktar, :TasnifMiktar;

    Select Sum(HgIsEmriUrun.GerceklesenMiktar) from HgIsEmri
      Left Join HgIsEmriUrun on HgIsEmriUrun.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
      where HgIsEmri.HGSiparisDetayId = :HGSiparisDetayId and HgIsEmriUrun.UrunId = :UrunId
    into :UretilenMiktar;

    suspend;
  end

end
