CREATE PROCEDURE IPLIKTALEPRAPORUELYAFBOYA(
    ARAPORTIPI INTEGER,
    ATALEPYERIID INTEGER,
    ABASLANGICIPLIKTALEPYIL INTEGER,
    ABASLANGICIPLIKTALEPID INTEGER,
    ABITISIPLIKTALEPYIL INTEGER,
    ABITISIPLIKTALEPID INTEGER,
    ABASLANGICTALEPTARIHI TIMESTAMP,
    ABITISTALEPTARIHI TIMESTAMP,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    AKAPALI INTEGER,
    ATAMIR INTEGER,
    AONAYDURUMU INTEGER,
    ATESLIMDURUMU INTEGER)
RETURNS (
    IPLIKTALEPDETAYID INTEGER,
    IPLIKIRSALIYEID INTEGER,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(20),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable shtUretimCikisBoya    integer;
declare variable shtUretimBoya         integer;
begin
  Select shtUretimCikisBoya, shtUretimBoya from IplikStokHareketTipiGetir into :shtUretimCikisBoya, :shtUretimBoya;

  For Select IplikTalepRaporu.IplikTalepDetayID, IplikIrsaliyeDetay.IplikIrsaliyeId, IplikIrsaliyeDetay.HareketTipi,
             IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.FirmaId, Firma.Adi,
             IplikIrsaliyeDetay.HamId, Ham.Adi, IplikIrsaliyeDetay.RenkId, Renk.Adi,
             IplikIrsaliyeDetay.KapAdedi, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.Notu from IplikTalepRaporu(:ARaporTipi, :ATalepYeriId, :ABaslangicIplikTalepYil, :ABaslangicIplikTalepId, :ABitisIplikTalepYil, :ABitisIplikTalepId,
                               :ABaslangicTalepTarihi, :ABitisTalepTarihi, :ASiparisGrupId, :ASiparisFirmaId, :AHamId, :AKapali,  :ATamir,:AOnayDurumu, :ATeslimDurumu)
         left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayID = IplikTalepRaporu.IplikTalepDetayID
         left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikIrsaliyeDetay.IplikIrsaliyeId
         left join Firma on Firma.FirmaID = IplikIrsaliye.FirmaId
         left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
         left join Renk on Renk.RenkID = IplikIrsaliyeDetay.RenkId
  where IplikIrsaliyeDetay.IplikIrsaliyeDetayId is not null and
        (IplikIrsaliyeDetay.HareketTipi = :shtUretimCikisBoya or IplikIrsaliyeDetay.HareketTipi = :shtUretimBoya)
  into :IplikTalepDetayID, :IplikIrsaliyeId, :HareketTipi, :IrsaliyeNo, :IrsaliyeTarihi, :FirmaId, :FirmaAdi,
       :HamId, :HamAdi, :RenkId, :RenkAdi, :KapAdedi, :Miktar, :Notu do
  begin
    if (HareketTipi = shtUretimCikisBoya) then HareketTipiStr = 'Çıkış';
    else if (HareketTipi = shtUretimBoya) then HareketTipiStr = 'Giriş';
    else HareketTipiStr = 'Hatalı...';

    suspend;
  end
end
