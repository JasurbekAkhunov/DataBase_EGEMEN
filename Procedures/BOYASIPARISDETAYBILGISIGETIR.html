Definition

CREATE PROCEDURE BOYASIPARISDETAYBILGISIGETIR(
    BOYASIPARISDETAYID INTEGER)
RETURNS (
    SIPARISKILO DOUBLE PRECISION,
    SIPARISMETRE DOUBLE PRECISION,
    SIPARISBRUTKILO DOUBLE PRECISION,
    SIPARISBRUTMETRE DOUBLE PRECISION,
    PARTILENENKILO DOUBLE PRECISION,
    IPTALKILO DOUBLE PRECISION,
    PARTILENENMETRE DOUBLE PRECISION,
    IPTALMETRE DOUBLE PRECISION,
    REZERVKILO DOUBLE PRECISION,
    REZERVMETRE DOUBLE PRECISION)
AS
  declare variable SipBrutMiktarHesaplamaSekli  integer;
  declare variable SiparisMaximumFireOrani      Double precision;
  declare variable PartilenenMiktardanIptaliDus EgeBoolean;
begin
  Select SipBrutMiktarHesaplamaSekli, PartilenenMiktardanIptaliDus from sistem where sistemid = 1 into :SipBrutMiktarHesaplamaSekli, :PartilenenMiktardanIptaliDus;

  Select SiparisKilo, SiparisMetre, PartilenenKilo, IptalKilo, PartilenenMetre, IptalMetre, MaximumFireOrani, SiparisBrutKilo from BoyaSiparisDetay
  where BoyaSiparisDetay.BoyaSiparisDetayID = :BoyaSiparisDetayID
  into :SiparisKilo, :SiparisMetre, :PartilenenKilo, :IptalKilo, :PartilenenMetre, :IptalMetre, :SiparisMaximumFireOrani, :SiparisBrutKilo;

  if (SiparisMaximumFireOrani is not null) then
  begin
    if (SipBrutMiktarHesaplamaSekli = 0) then
    begin
      SiparisBrutMetre = SiparisMetre / ((100 - SiparisMaximumFireOrani) / 100);
      SiparisBrutMetre = Yuvarla(SiparisBrutMetre, 1);
    end
    else
    begin
      SiparisBrutMetre = SiparisMetre + SiparisMetre * SiparisMaximumFireOrani / 100;
      SiparisBrutMetre = Yuvarla(SiparisBrutMetre, 1);
    end
  end
  else
  begin
    SiparisBrutMetre = SiparisMetre;
  end

  Select Sum(Kilo), Sum(Metre) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayID = :BoyaSiparisDetayID into :RezervKilo, :RezervMetre;

  if (PartilenenMiktardanIptaliDus = 1) then
  begin
    PartilenenKilo = PartilenenKilo - IptalKilo;
    PartilenenMetre = PartilenenMetre - IptalMetre;
  end

  Suspend;
end
