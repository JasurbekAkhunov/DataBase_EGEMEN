CREATE PROCEDURE TRANSFERICINHAMSTOKFATURAGETIR(
    AHAMSTOKFATURAID INTEGER)
RETURNS (
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    ODEMETARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    DOVIZID VARCHAR(5),
    TRANSFERKODU VARCHAR(55),
    ATRANSFERKODU VARCHAR(55),
    MUHASEBEKODU VARCHAR(20),
    BRUTTOPLAM DOUBLE PRECISION,
    ISKONTOTOPLAMI DOUBLE PRECISION,
    PARASALISKONTO DOUBLE PRECISION,
    ISKONTOORANI DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    ARATOPLAM DOUBLE PRECISION,
    GENELTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZKURU1 DOUBLE PRECISION,
    ISLEMDOVIZKURU2 DOUBLE PRECISION,
    ISLEMDOVIZTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZTAHSILEDILECEKKDV DOUBLE PRECISION,
    TEVKIFAT INTEGER,
    TEVKIFATORANI DOUBLE PRECISION,
    SINIF VARCHAR(1),
    IRSALIYENO VARCHAR(255),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTTARIHI TIMESTAMP)
AS
declare variable FIrsaliyeNo varchar(20);
begin
  For Select HamStokFatura.FaturaTarihi, HamStokFatura.OdemeTarihi, HamStokFatura.FaturaSeriNo, HamStokFatura.FaturaNo, HamStokFatura.FirmaId,
         Coalesce(DovizliFirmaTransfer.TransferKodu,Firma.TransferKodu), Firma.ATransferKodu, Firma.MuhasebeKodu, HamStokFatura.DovizId,
         HamStokFatura.BrutToplam, HamStokFatura.ParasalIskonto, (HamStokFatura.Iskonto1OranTutari + HamStokFatura.Iskonto2OranTutari),
         HamStokFatura.KDVToplami, 0, HamStokFatura.IslemDovizID, HamStokFatura.IslemDovizToplam , HamStokFatura.IslemDovizTahsilEdilecekKdv,
         HamStokFatura.IslemDovizKuru1, HamStokFatura.IslemDovizKuru2, 0, HamStokFatura.Sinif,
         (HamStokFatura.BrutToplam + HamStokFatura.KDVToplami - HamStokFatura.Iskonto1OranTutari - HamStokFatura.Iskonto2OranTutari) GenelToplam,
         (HamStokFatura.BrutToplam + HamStokFatura.KDVToplami - HamStokFatura.Iskonto1OranTutari - HamStokFatura.Iskonto2OranTutari) AraToplam,
         1, HamStokFatura.Notu, HamStokFatura.InsertTarihi from HamStokFatura
  left join Firma On Firma.FirmaID = HamStokFatura.FirmaId
  left join DovizliFirmaTransfer On DovizliFirmaTransfer.FirmaID = HamStokFatura.FirmaId and DovizliFirmaTransfer.DovizID = HamStokFatura.DovizID
  where HamStokFatura.HamStokFaturaID = :AHamStokFaturaId
  into :FaturaTarihi, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FirmaId, TransferKodu, :ATransferKodu, :MuhasebeKodu, :DovizId,
       :BrutToplam, :ParasalIskonto, :IskontoToplami, :TahsilEdilecekKdv, :IndirilecekKdv, :IslemDovizID,
       :IslemDovizToplam, :IslemDovizTahsilEdilecekKdv, :IslemDovizKuru1, :IslemDovizKuru2, :Tevkifat, :Sinif, :GenelToplam, :AraToplam, :TevkifatOrani, :Notu, :InsertTarihi do
  begin
    IrsaliyeNo = '';
    For Select HamStokIrsaliye.IrsaliyeSeriNo || HamStokIrsaliye.IrsaliyeNo from HamStokMaster
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
      where HamStokMaster.HamStokFaturaID = :AHamStokFaturaID
      Group By HamStokIrsaliye.IrsaliyeSeriNo,HamStokIrsaliye.IrsaliyeNo
    into :FIrsaliyeNo do
    begin
      if (Char_Length(IrsaliyeNo) + Char_Length(FIrsaliyeNo) < 240) then
      begin
        if (IrsaliyeNo <> '' ) then IrsaliyeNo = IrsaliyeNo || cast(', ' as varchar(2) character set UTF8);
        IrsaliyeNo = IrsaliyeNo || FIrsaliyeNo;
      end
    end
    suspend;
  end
end
