CREATE PROCEDURE DOKUMAURETIMPNOBILGISIGETIR(
    ADOKUMAPLANLAMAID INTEGER)
RETURNS (
    DOKUMAPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    PLANLANANMIKTAR NUMERIC(12,3),
    GERCEKLESENMIKTAR NUMERIC(12,3),
    TOPLAMSAKATDOKUNAN NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    TARIH TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAIDSTR VARCHAR(75),
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    DOKUMAMAKINAIDSTR VARCHAR(55),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    HAMADI VARCHAR(255),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    BITTI SMALLINT,
    BITTISTR VARCHAR(10),
    BEKLEMEDE SMALLINT,
    BEKLEMEDESTR VARCHAR(10),
    ARAVERMESEBEBIID INTEGER,
    ARAVERMESEBEBIADI ADI50,
    DOKUMATALIMATLOT VARCHAR(150),
    DOKUNANTOPSAYISI INTEGER,
    KALITEADI ADI255)
AS
begin

  Select DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlama.UretimPartiNo, DokumaPlanlama.PlanlananMiktar, DokumaPlanlama.Tarih,
         DokumaPlanlama.DokumaMakinaId, DokumaMakina.Adi, DokumaPlanlama.FirmaId, PFirma.Adi, BoyaSiparis.BoyaSiparisYil,
         BoyaSiparis.BoyaSiparisId, Ham.Adi, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo,
         OrmeSiparisDetay.ArtikelNo, SFirma.Adi, DokumaPlanlama.GerceklesenMiktar, DokumaPlanlama.ToplamSakatDokunan,
         DokumaPlanlama.Bitti, DokumaPlanlama.Beklemede, DokumaPlanlama.AraVermeSebebiId, AraVermeSebebi.Adi, DokumaPlanlama.lot, Kalite.KaliteAdi from DokumaPlanlama
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = DokumaPlanlama.DokumaMakinaId
    left join Firma PFirma on PFirma.FirmaId = DokumaPlanlama.FirmaId
    left join Firma SFirma on SFirma.FirmaId = BoyaSiparis.FirmaId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join Kalite on Kalite.KaliteId = OrmeSiparisDetay.KaliteId
    left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = DokumaPlanlama.AraVermeSebebiId
    where DokumaPlanlama.DokumaPlanlamaId = :ADokumaPlanlamaId
    into :DokumaPlanlamaId, :UretimPartiNo, :PlanlananMiktar, :Tarih,
         :DokumaMakinaId, :DokumaMakinaAdi, :FirmaId, :FirmaAdi, :BoyaSiparisYil,
         :BoyaSiparisId, :HamAdi, :MusteriSiparisNo, :ReferansSiparisNo,
         :ArtikelNo, :SiparisFirmaAdi, :GerceklesenMiktar, :ToplamSakatDokunan,
         :Bitti, :Beklemede, :AraVermeSebebiId, :AraVermeSebebiAdi, :DokumaTalimatLot, :KaliteAdi;

    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    if (FirmaId is not null) then FirmaIdStr = '(' || FirmaId || ') ' || FirmaAdi;
    if (DokumaMakinaId is not null) then DokumaMakinaIdStr = '(' || DokumaMakinaId || ') ' || DokumaMakinaAdi;

    KalanMiktar = PlanlananMiktar - GerceklesenMiktar + ToplamSakatDokunan;
    if (Bitti = 1) then BittiStr = 'Bitti';
    if (Beklemede = 1) then BeklemedeStr = 'Beklemede';

    Select count(*) from Dokumakalitekontrol where Dokumakalitekontrol.DokumaPlanlamaId = :ADokumaPlanlamaId into :DokunanTopSayisi;

    suspend;
end
