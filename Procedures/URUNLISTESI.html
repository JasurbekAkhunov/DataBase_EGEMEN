CREATE PROCEDURE URUNLISTESI(
    ADEPOID VARCHAR(10))
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNADI2 VARCHAR(80),
    URUNGRUP1 VARCHAR(4),
    MUTFAKDEPONO VARCHAR(10),
    BIRIMID VARCHAR(2),
    RAF VARCHAR(15),
    DEPOID VARCHAR(10),
    DEPOSTOKMEVCUDU DOUBLE PRECISION,
    STOKMEVCUDU DOUBLE PRECISION,
    URUNTIPIID INTEGER,
    URUNTIPIADI VARCHAR(20),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    URUNGRUP1ADI VARCHAR(30),
    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30))
AS
declare variable DefaultBirimCarpan double precision;
begin
  if (ADepoId = '') then ADepoId = Null;

  DepoId = :ADepoId;

  For Select Urun.UrunId, Urun.Adi, Urun.Adi2, Urun.UrunGrup1, Urun.MutfakDepoNo, UrunBirim.BirimId, UrunBirim.Carpan, Urun.Raf, Urun.UrunTipiId, UrunTipi.Adi,
               Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi from Urun
    left join UrunBirim on UrunBirim.UrunId = Urun.UrunId and UrunBirim.DefaultBirim = 1
    left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
    left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
    left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
    left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
    left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
    where Kapali = 0
   into :UrunId, :UrunAdi, :UrunAdi2, :UrunGrup1, :MutfakDepoNo, :BirimId, :DefaultBirimCarpan, :Raf,
           :UrunTipiId, :UrunTipiAdi, :UrunGrup2, :UrunGrup3, :UrunGrup4, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi do
  begin
    DepoStokMevcudu = null;
    StokMevcudu = Null;

    if (ADepoId is not null) then
      Select Devir*:DefaultBirimCarpan + (Alis-AlisIadesi)*:DefaultBirimCarpan -
            (Satis-SatisIadesi)*:DefaultBirimCarpan - DepoyaCikan*:DefaultBirimCarpan +
             DepodanGelen * :DefaultBirimCarpan - UretimeCikis*:DefaultBirimCarpan + Uretim*:DefaultBirimCarpan - FasonUretimeCikis*:DefaultBirimCarpan + FasonUretim*:DefaultBirimCarpan + FasonUretimIade*:DefaultBirimCarpan from DepoStok
          where DepoStok.UrunId = :UrunId and DepoStok.DepoId = :ADepoId
         into :DepoStokMevcudu;

    Select Sum(Devir)*:DefaultBirimCarpan + Sum(Alis-AlisIadesi)*:DefaultBirimCarpan -
           Sum(Satis-SatisIadesi)*:DefaultBirimCarpan - Sum(UretimeCikis)*:DefaultBirimCarpan + Sum(Uretim)*:DefaultBirimCarpan - Sum(FasonUretimeCikis)*:DefaultBirimCarpan + Sum(FasonUretim)*:DefaultBirimCarpan + Sum(FasonUretimIade)*:DefaultBirimCarpan from DepoStok
        where DepoStok.UrunId = :UrunId
        into :StokMevcudu;
    suspend;
  end
end
