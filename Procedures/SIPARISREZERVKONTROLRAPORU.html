CREATE PROCEDURE SIPARISREZERVKONTROLRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AHAMTIPIID INTEGER,
    AISLEMTIPIID INTEGER,
    ASATISTIPI INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AHAMID INTEGER,
    AREZERVDURUMU INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AKALITEID VARCHAR(20))
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISNOSTR VARCHAR(30),
    FASONSTR VARCHAR(5),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPARISTARIHI TIMESTAMP,
    BOYASIPARISDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    MUSTERISIPARISNO VARCHAR(50),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISNO VARCHAR(20),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    SIPARISMIKTAR NUMERIC(12,3),
    SIPARISBRUTKILO NUMERIC(12,3),
    SIPARISBRUTMETRE NUMERIC(12,3),
    PARTILENENMIKTAR NUMERIC(12,3),
    IPTALMIKTAR NUMERIC(12,3),
    SIPARISKALANMIKTAR NUMERIC(12,3),
    REZERVMIKTAR NUMERIC(12,3),
    REZERVKALANMIKTAR NUMERIC(12,3),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    SIPARISMAXIMUMFIREORANI DOUBLE PRECISION,
    MAXIMUMFIREORANI DOUBLE PRECISION,
    SATISFIREONGORUORANI DOUBLE PRECISION,
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    SIPARISTIPIID INTEGER,
    SIPILAVEISLEMLER VARCHAR(255),
    ORMEPUS INTEGER,
    ORMEFEINE INTEGER,
    TUPACIKENMAYLI INTEGER,
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    UCRETLITAMIR INTEGER,
    BRUTSEVKEDILENKILO NUMERIC(12,3),
    BRUTSEVKEDILENMETRE NUMERIC(12,3),
    NETSEVKEDILENKILO NUMERIC(12,3),
    NETSEVKEDILENMETRE NUMERIC(12,3),
    SEVKEHAZIRBRUTKILO NUMERIC(12,3),
    SEVKEHAZIRNETKILO NUMERIC(12,3),
    ONAYDURUMU INTEGER,
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    KALITEID VARCHAR(20),
    STOKTANKULLANILACAKKILO NUMERIC(12,3),
    STOKTANKULLANILACAKMETRE NUMERIC(12,3),
    BASKITIPIID INTEGER,
    BASKITIPIADI VARCHAR(40),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    FASONBIRIMFIYATI DOUBLE PRECISION,
    FASONBIRIMFIYATDOVIZID VARCHAR(5),
    YURTDISI INTEGER,
    YURTDISISTR VARCHAR(10),
    KAPALI INTEGER,
    KAPALISTR VARCHAR(6),
    RISKHESAPLANMASIN INTEGER,
    ILKSEVKTARIHI TIMESTAMP,
    MUSTERIRENKKODUVEADI VARCHAR(30),
    PLANLAMANOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    DEGISIMLOG BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LABRECETEACIKLAMA VARCHAR(40))
AS
declare variable Fason integer;
declare variable HamTipiId integer;

declare variable FSiparisKilo           Numeric(12,3);
declare variable FSiparisMetre          Numeric(12,3);
declare variable FPartilenenKilo        Numeric(12,3);
declare variable FPartilenenMetre       Numeric(12,3);
declare variable FIptalKilo             Numeric(12,3);
declare variable FIptalMetre            Numeric(12,3);
declare variable Okey           integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable SatSiparisKalanBruttenHesapla integer;
begin
  Select DokumaKumasiKilodanTakipEt, SatSiparisKalanBruttenHesapla from Sistem
    into :DokumaKumasiKilodanTakipEt, :SatSiparisKalanBruttenHesapla;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.Fason, BoyaSiparis.FirmaId, Firma.Adi,
             BoyaSiparis.SiparisTarihi, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.HamId, Ham.Adi, BoyaSiparisDetay.MusteriSiparisNo,
             LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr,
             BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisMetre,
             (BoyaSiparisDetay.PartilenenKilo + BoyaSiparisDetay.PartilenenKiloBaski - BoyaSiparisDetay.IptalKilo - BoyaSiparisDetay.IptalKiloBaski),
             (BoyaSiparisDetay.PartilenenMetre + BoyaSiparisDetay.PartilenenMetreBaski - BoyaSiparisDetay.IptalMetre - BoyaSiparisDetay.IptalMetreBaski),
             BoyaSiparisDetay.IptalKilo + BoyaSiparisDetay.IptalKiloBaski, BoyaSiparisDetay.IptalMetre + BoyaSiparisDetay.IptalMetreBaski,
             BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi,
             BoyaSiparisDetay.MaximumFireOrani, BoyaSiparisDetay.SatisFireOngoruOrani, Ham.MaximumFireOrani, Ham.HamTipiId, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi,
             BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi, BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi, BoyaSiparisDetay.IslemgrubuId, Islemgrubu.Adi,
             BoyaSiparisDetay.SiparisTipiId, BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.UcretliTamir,
             BoyaSiparisDetay.DesenId, Desen.Adi,
             BoyaSiparisDetay.OnayDurumu, BoyaSiparisDetay.OnayTarihi, BoyaSiparisDetay.OnaylayanKullaniciId, Ok.Adi,
             BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             OrmeSiparisDetay.Pus, ormeSiparisDetay.Feine, OrmeSiparisDetay.TupAcikEnMayli, Ham.Adi2, Ham.Adi3, Ham.Adi4,
             BoyaSiparisDetay.KaliteId, BoyaSiparisDetay.StoktanKullanilacakKilo, BoyaSiparisDetay.StoktanKullanilacakMetre,
             BoyaSiparisDetay.BaskiTipiId, BaskiTipi.Adi, BoyaSiparisDetay.SiparisBrutKilo, BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.FasonBirimFiyati,
             BoyaSiparisDetay.FasonBirimFiyatDovizId, BoyaSiparisDetay.DovizId, BoyaSiparis.YurtDisi, BoyaSiparisDetay.Kapali, Firma.RiskHesaplanmasin, BoyaSiparisDetay.SiparisBrutMetre,
             BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.PlanlamaNotu, BoyaSiparisDetay.DegisimLog, LabRecete.Aciklama from BoyaSiparis
  Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
  left join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
  left join IslemTipi On IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
  left join IslemGrubu On IslemGrubu.IslemGrubuId = BoyaSiparisDetay.IslemGrubuId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
  left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
  Left Join Kullanici OK on OK.KullaniciId = BoyaSiparisDetay.OnaylayanKullaniciId
  left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  left join BaskiTipi on BaskiTipi.BaskiTipiId = BoyaSiparisDetay.BaskiTipiId
  where (BoyaSiparis.SiparisTarihi >= :ABaslangicTarihi and BoyaSiparis.SiparisTarihi <= :ABitisTarihi) and
        (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
        (BoyaSiparis.Fason = :ASatisTipi or :ASatisTipi is null) and
        (BoyaSiparisDetay.IslemTipiId = :AIslemTipiId or :AIslemTipiId is null) and
        (BoyaSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
        (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
        (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
        ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null) and
        (BoyaSiparisDetay.HamId = :AHamId or :AHamId is null) and
        (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
        (BoyaSiparisDetay.OnayDurumu = :AOnayDurumu or :AOnayDurumu is null) and
        (BoyaSiparisDetay.Kapali = :AKapaliDurumu or :AKapaliDurumu is null) and
        (BoyaSiparisDetay.KaliteId = :AKaliteId or :AKaliteId is Null)
  order by BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId
  into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :Fason, :FirmaId, :FirmaAdi, :SiparisTarihi, :BoyaSiparisDetayId, :HamId,
       :HamAdi, :MusteriSiparisNo, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisNo,
       :FSiparisKilo, :FSiparisMetre, :FPartilenenKilo, :FPartilenenMetre, :FIptalKilo, :FIptalMetre,
       :SiparisProsesId, :SiparisProsesAdi, :SiparisMaximumFireOrani, :SatisFireOngoruOrani, :MaximumFireOrani, :HamTipiId, :TerminTarihi, :RevizeTerminTarihi,
       :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi, :IslemTipiId, :IslemTipiAdi, :IslemgrubuId, :IslemgrubuAdi,
       :SiparisTipiId, :SipIlaveIslemler, EnIstenen, :GramajIstenen, :UcretliTamir, :DesenId, :DesenAdi,
       :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
       :OrmePus, :OrmeFeine, :TupAcikEnMayli, :HamAdi2, :HamAdi3, :HamAdi4,
       :KaliteId, :StoktanKullanilacakKilo, :StoktanKullanilacakMetre, :BaskiTipiId, :BaskiTipiAdi, :SiparisBrutKilo, :BirimFiyat, :FasonBirimFiyati,
       :FasonBirimFiyatDovizId, :DovizId, :YurtDisi, :Kapali, :RiskHesaplanmasin, :SiparisBrutMetre, :MusteriRenkKoduveAdi, :PlanlamaNotu, :DegisimLog, :LabReceteAciklama do
  begin
    BoyaSiparisNoStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (YurtDisi = 1) then YurtDisiStr = 'Yurt Dışı';
    else YurtDisiStr = '';

    if (Kapali = 1) then KapaliStr = 'Kapalı';
    else KapaliStr = '';

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
    begin
      SiparisMiktar = FSiparisKilo;
      PartilenenMiktar = FPartilenenKilo;
      IptalMiktar = FIptalKilo;

      if (Fason = 0 and SatSiparisKalanBruttenHesapla = 1) then
        SiparisKalanMiktar = SiparisBrutKilo - FPartilenenKilo;
      else
        SiparisKalanMiktar = FSiparisKilo - FPartilenenKilo;
    end
    else
    begin
      SiparisMiktar = FSiparisMetre;
      PartilenenMiktar = FPartilenenMetre;
      IptalMiktar = FIptalMetre;

      SiparisKalanMiktar = SiparisMiktar - PartilenenMiktar;
    end

    Okey = 1;

    if (APartilenen = 0 and SiparisKalanMiktar <= 0) then Okey = 0;
    else if (APartilenen = 1 and SiparisKalanMiktar > 0) then Okey = 0;


    RezervMiktar = 0;
    RezervKalanMiktar = 0;

    if (Okey = 1) then
    begin
      Select Sum(BoyaSiparisRezerv.Miktar), Sum(BoyaSiparisRezerv.Miktar) - Sum(BoyaSiparisRezerv.KullanilanMiktar), min(HamStokIrsaliye.IrsaliyeTarihi) from BoyaSiparisRezerv
        left join HamStokDetay on  HamStokDetay.HamStokDetayID = BoyaSiparisRezerv.HamStokDetayID
        left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = hamstokdetay.HamStokIrsaliyeID
        where BoyaSiparisRezerv.BoyaSiparisDetayId = :BoyaSiparisDetayId
        into :RezervMiktar, :RezervKalanMiktar, :IlkSevkTarihi;



      if (RezervMiktar is Null) then RezervMiktar = 0;
      if (RezervKalanMiktar is Null) then RezervKalanMiktar = 0;

      if (ARezervDurumu = 0 and RezervKalanMiktar = 0) then Okey = 0; /* rezerv yok */
      else if (ARezervDurumu = 1 and RezervKalanMiktar > 0) then Okey = 0; /* rezerv var */

      if (Okey = 1) then
      begin
        BrutSevkEdilenKilo = Null;
        BrutSevkEdilenMetre = Null;
        NetSevkEdilenKilo = Null;
        NetSevkEdilenMetre = Null;
        SevkeHazirBrutKilo = Null;
        SevkeHazirNetKilo = Null;

        Select Sum(PartiStok.BrutSevkKilo), Sum(PartiStok.BrutSevkMetre), Sum(PartiStok.NetSevkKilo), Sum(PartiStok.NetSevkMetre) from BoyaSiparisDetay
          left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
          where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
            into :BrutSevkEdilenKilo, :BrutSevkEdilenMetre, :NetSevkEdilenKilo, :NetSevkEdilenMetre;

        Select Sum(PartiStok.Kilo - PartiStok.BrutSevkKilo), Sum(PartiStok.NetKilo - PartiStok.NetSevkKilo) from PartiStok
          where PartiStok.BoyaSiparisDetayID = :BoyaSiparisDetayID and PartiStok.Durumu = 7 and PartiStok.TamamiTartildi = 1
          into :SevkeHazirBrutKilo, :SevkeHazirNetKilo;

        suspend;
      end
    end
  end
end
