CREATE PROCEDURE IMALATAARAVERMERAPORUANLIK
RETURNS (
    IMALATAARAVERMEID INTEGER,
    MAKINAID VARCHAR(10),
    BASLADIMAKINADURUMUID INTEGER,
    ARAVERDIRENPERSONELID INTEGER,
    ARAVERMESEBEBIID INTEGER,
    BITTIMAKINADURUMUID INTEGER,
    ARAYIBITIRENPERSONELID INTEGER,
    TARIH TIMESTAMP,
    PERSONELADI VARCHAR(40),
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30))
AS
begin
  For Select ImalataAraVerme.ImalataAraVermeId, ImalataAraVerme.AraVermeSebebiId, Coalesce(ImalataAraVerme.MakinaId, ImalataAraVerme.OrmeMakinaId), ImalataAraVerme.BasladiMakinaDurumuId,
             ImalataAraVerme.AraVerdirenPersonelId, ImalataAraVerme.AraVermeSebebiId, ImalataAraVerme.BittiMakinaDurumuId,
             ImalataAraVerme.ArayiBitirenPersonelId, Coalesce(MakinaDurumu.Tarih, ImalataAraVerme.BaslamaTarihi), Personel.Adi, (CURRENT_DATE + CURRENT_TIME-Coalesce(MakinaDurumu.Tarih, ImalataAraVerme.BaslamaTarihi)) from ImalataAraVerme
  left join MakinaDurumu on MakinaDurumu.MakinaDurumuId = ImalataAraVerme.BasladiMakinaDurumuId
  left join Personel on Personel.PersonelId = ImalataAraVerme.AraVerdirenPersonelId
  where (ImalataAraVerme.MakinaId is Not Null and ImalataAraVerme.BittiMakinaDurumuId is null) or (ImalataAraVerme.OrmeMakinaId is Not Null and ImalataAraVerme.BittiOrmeMakinaDurumuId is null)
  Order By ImalataAraVerme.AraVermeSebebiId, ImalataAraVerme.MakinaId, ImalataAraVerme.OrmeMakinaId
  into :ImalataAraVermeId, :AraVermeSebebiId, :MakinaId, :BasladiMakinaDurumuId, :AraVerdirenPersonelId, :AraVermeSebebiId, :BittiMakinaDurumuId,
       :ArayiBitirenPersonelId, :Tarih, :PersonelAdi, :GecenSure do
  begin
    GecenSureStr = TarihToStr(GecenSure);
    suspend;
  end
end
