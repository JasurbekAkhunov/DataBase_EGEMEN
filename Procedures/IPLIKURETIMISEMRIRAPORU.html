CREATE PROCEDURE IPLIKURETIMISEMRIRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIPLIKMAKINATIPIID INTEGER,
    AIPLIKMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4),
    ALOT VARCHAR(30),
    ABITTI INTEGER)
RETURNS (
    IPLIKURETIMISEMRIKAYITID INTEGER,
    IPLIKURETIMISEMRIYIL INTEGER,
    IPLIKURETIMISEMRIID INTEGER,
    IPLIKURETIMISEMRIIDSTR VARCHAR(30),
    TARIH TIMESTAMP,
    IPLIKMAKINAID VARCHAR(10),
    MAKINAGRUPID VARCHAR(4),
    IPLIKMAKINATIPIID INTEGER,
    LOT VARCHAR(30),
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    HARMANADI VARCHAR(50),
    HARMANCINSIID INTEGER,
    HARMANCINSIADI VARCHAR(50),
    BUKUMTIPIID INTEGER,
    BUKUMTIPIADI VARCHAR(10),
    COPSTIPIADI VARCHAR(20),
    KALINLIKNE INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    SIRA INTEGER,
    MIKTARGUN DOUBLE PRECISION,
    BASLAMATARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    BITTI INTEGER,
    BITTISTR VARCHAR(4))
AS
begin
  if (ABitti = 1) then
  begin
    for Select IplikUretimIsEmri.IplikUretimIsEmriKayitId, IplikUretimIsEmri.IplikUretimIsEmriyil, IplikUretimIsEmri.IplikUretimIsEmriId,
               IplikUretimIsEmri.PlanlamaTarihi, IplikUretimIsEmri.IplikMakinaId, IplikLot.Lot, IplikUretimIsEmri.PlanlananMiktar,
               IplikUretimIsEmri.GerceklesenMiktar, Harman.Adi, Harman.HarmanCinsiId, HarmanCinsi.Adi, IplikLot.BukumTipiId, BukumTipi.Adi,
               IplikLot.KalinlikNE, IplikCinsi.Adi, CopsTipi.Adi, IplikUretimIsEmri.Sira, IplikUretimIsEmri.MiktarGun, IplikUretimIsEmri.BaslamaTarihi,
               IplikUretimIsEmri.BitisTarihi, IplikUretimIsEmri.Bitti, IplikMakina.MakinaGrupId, IplikUretimIsEmri.IplikMakinaTipiId from IplikUretimIsEmri
      left join IplikMakina on IplikMakina.IplikMakinaID = IplikUretimIsEmri.IplikMakinaId
      left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotId
      left join IplikCinsi on IplikCinsi.IplikCinsiID = IplikLot.IplikCinsiId
      left join BukumTipi on BukumTipi.BukumTipiID = IplikLot.BukumTipiId
      left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitID
      left join HarmanCinsi on HarmanCinsi.HarmanCinsiID = Harman.HarmanCinsiID
      left join CopsTipi on CopsTipi.CopsTipiID = IplikUretimIsEmri.CopsTipiID
      where (IplikUretimIsEmri.BitisTarihi >= :ABaslangicTarihi and IplikUretimIsEmri.BitisTarihi < :ABitisTarihi + 1) and
            IplikUretimIsEmri.Bitti = 1 and
            (IplikUretimIsEmri.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null) and
            (IplikLot.Lot = :ALot or :ALot is null) and
            (IplikMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
            (IplikMakina.IplikMakinaTipiId = :AIplikMakinaTipiId or :AIplikMakinaTipiId is null)
      order by IplikUretimIsEmri.IplikMakinaId, IplikUretimIsEmri.Sira
      into :IplikUretimIsEmriKayitId, :IplikUretimIsEmriyil, :IplikUretimIsEmriId, :Tarih, :IplikMakinaId, :Lot, :PlanlananMiktar,
           :GerceklesenMiktar, :HarmanAdi, :HarmanCinsiId, :HarmanCinsiAdi, :BukumTipiId, :BukumTipiAdi, :KalinlikNE, :IplikCinsiAdi, :CopsTipiAdi,
           :Sira, :MiktarGun, :BaslamaTarihi, :BitisTarihi, :Bitti, :MakinaGrupId, :IplikMakinaTipiId do
    begin
      if (Bitti = 0) then BittiStr = '';
      else BittiStr = 'Evet';

      IplikUretimIsEmriIdStr = '(' || EgemenIntegerToStr(IplikUretimIsEmriYil) || ') ' || EgemenIntegerToStr(IplikUretimIsEmriId) || ' ';

      KalanMiktar = PlanlananMiktar - GerceklesenMiktar;

      suspend;
    end
  end
  else
  begin
    for Select IplikUretimIsEmri.IplikUretimIsEmriKayitId, IplikUretimIsEmri.IplikUretimIsEmriyil, IplikUretimIsEmri.IplikUretimIsEmriId,
               IplikUretimIsEmri.PlanlamaTarihi, IplikUretimIsEmri.IplikMakinaId, IplikLot.Lot, IplikUretimIsEmri.PlanlananMiktar,
               IplikUretimIsEmri.GerceklesenMiktar, Harman.Adi, Harman.HarmanCinsiId, HarmanCinsi.Adi, IplikLot.BukumTipiId, BukumTipi.Adi,
               IplikLot.KalinlikNE, IplikCinsi.Adi, CopsTipi.Adi, IplikUretimIsEmri.Sira, IplikUretimIsEmri.MiktarGun, IplikUretimIsEmri.BaslamaTarihi,
               IplikUretimIsEmri.BitisTarihi, IplikUretimIsEmri.Bitti, IplikMakina.MakinaGrupId, IplikUretimIsEmri.IplikMakinaTipiId from IplikUretimIsEmri
      left join IplikMakina on IplikMakina.IplikMakinaID = IplikUretimIsEmri.IplikMakinaId
      left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotId
      left join IplikCinsi on IplikCinsi.IplikCinsiID = IplikLot.IplikCinsiId
      left join BukumTipi on BukumTipi.BukumTipiID = IplikLot.BukumTipiId
      left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitID
      left join HarmanCinsi on HarmanCinsi.HarmanCinsiID = Harman.HarmanCinsiID
      left join CopsTipi on CopsTipi.CopsTipiID = IplikUretimIsEmri.CopsTipiID
      where (IplikUretimIsEmri.PlanlamaTarihi >= :ABaslangicTarihi and IplikUretimIsEmri.PlanlamaTarihi < :ABitisTarihi + 1) and
            (IplikUretimIsEmri.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null) and
            (IplikLot.Lot = :ALot or :ALot is null) and
            (IplikMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
            (IplikMakina.IplikMakinaTipiId = :AIplikMakinaTipiId or :AIplikMakinaTipiId is null) and
            (IplikUretimIsEmri.Bitti = :ABitti or :ABitti is null)
      order by IplikUretimIsEmri.IplikMakinaId, IplikUretimIsEmri.Sira
      into :IplikUretimIsEmriKayitId, :IplikUretimIsEmriyil, :IplikUretimIsEmriId, :Tarih, :IplikMakinaId, :Lot, :PlanlananMiktar,
           :GerceklesenMiktar, :HarmanAdi, :HarmanCinsiId, :HarmanCinsiAdi, :BukumTipiId, :BukumTipiAdi, :KalinlikNE, :IplikCinsiAdi, :CopsTipiAdi,
           :Sira, :MiktarGun, :BaslamaTarihi, :BitisTarihi, :Bitti, :MakinaGrupId, :IplikMakinaTipiId do
    begin
      if (Bitti = 0) then BittiStr = '';
      else BittiStr = 'Evet';

      IplikUretimIsEmriIdStr = '(' || EgemenIntegerToStr(IplikUretimIsEmriYil) || ') ' || EgemenIntegerToStr(IplikUretimIsEmriId) || ' ';

      KalanMiktar = PlanlananMiktar - GerceklesenMiktar;

      suspend;
    end
  end
end
