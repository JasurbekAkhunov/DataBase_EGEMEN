CREATE PROCEDURE FATURASIKESILMEMISIRSALIYELER(
    AFIRMAID VARCHAR(20),
    AIRSALIYEFATURALASTIRMASURESI INTEGER,
    ARAPORTIPI INTEGER,
    AFIRMAADI2YIGOREBILIR INTEGER,
    AHAREKETTIPI INTEGER,
    AEFATURADURUMU INTEGER,
    AFASON EGEBOOLEAN Default Null)
RETURNS (
    BOYAIRSALIYEID INTEGER,
    ORMEIRSALIYEID INTEGER,
    IPLIKIRSALIYEID INTEGER,
    STOKIRSALIYEID INTEGER,
    HAMSTOKIRSALIYEID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    FIRMAVERGIHESAPNO VARCHAR(20),
    HAREKETTIPIID INTEGER,
    HAREKETTIPIIDSTR VARCHAR(30),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    SINIF VARCHAR(1),
    DOVIZID VARCHAR(50),
    EFATURA INTEGER,
    EFATURASTR VARCHAR(6))
AS
declare variable FNow Date;
declare variable FAyIlkGun Date;
declare variable IrsaliyeFaturalastirmaSuresi integer;
declare variable shtSatis integer;
declare variable shtAlis integer;
declare variable shtAlisIade integer;
declare variable shtFasonUretim integer;
declare variable shtSatisIade integer;
declare variable OrmeFirmaKodumuz varchar(20);
declare variable FirmaKodumuz varchar(20);
declare variable IplikFirmaKodumuz varchar(20);
declare variable FKendiOrmeFirmamizaFatKesilsin integer;
declare variable FDovizID varchar(5);
declare variable IslemKaynagi SmallInt;
declare variable sikHazirGiyimKesim SmallInt;
declare variable shtDepoTransfer integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtFasonAlis integer;
begin
  Select OrmeFirmaKodumuz, FirmaKodumuz, IplikFirmaKodumuz, KendiOrmeFirmamizaFatKesilsin from sistem where sistemid = 1 into :OrmeFirmaKodumuz, :FirmaKodumuz, :IplikFirmaKodumuz, :FKendiOrmeFirmamizaFatKesilsin; /* boyahaneye veya merkeze kesilen irsaliyeler listeye gelmesin */

  Select shtSatis, ShtAlisIade, shtFasonAlis from OrmeStokHareketTipiGetir into :shtSatis, :ShtAlisIade, :shtFasonAlis;

  select sikHazirGiyimKesim from StokIslemKaynagiGetir into :sikHazirGiyimKesim;

  if (IplikFirmaKodumuz is null) then IplikFirmaKodumuz = '';

/* Irsaliyede faturasi kesilmemis Parti varsa faturalastirme süresi gecmis ise  listelenir,  */
  if (AIrsaliyeFaturalastirmaSuresi is null) then
  begin
    Select IrsaliyeFaturalastirmaSuresi from Sistem where sistemid = 1 into :IrsaliyeFaturalastirmaSuresi;
    if (IrsaliyeFaturalastirmaSuresi is null) then IrsaliyeFaturalastirmaSuresi = 0;
  end
  else IrsaliyeFaturalastirmaSuresi = AIrsaliyeFaturalastirmaSuresi;

  FNow = Current_Date;
  FAyIlkGun = TimeStampToAyIlkGun(FNow);
  FNow = FNow - IrsaliyeFaturalastirmaSuresi;

  if (ARaporTipi = 0) then
  begin

    for Select BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.FirmaId, Firma.Adi, Firma.Adi2, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi,
               BoyaIrsaliye.HareketTipiId, BoyaIrsaliyeHareketTipi.Adi, BoyaIrsaliye.Sinif, Firma.EFaturaKullanimda, Firma.VergiHesapNo From BoyaIrsaliyeDetay
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
      left join BoyaIrsaliyeHareketTipi on BoyaIrsaliyeHareketTipi.HareketTipiId = BoyaIrsaliye.HareketTipiId
      where BoyaIrsaliyeDetay.BoyaFaturaId is null and (BoyaIrsaliyeDetay.PartiKayitId is not null or BoyaIrsaliyeDetay.FasonHamIade = 0) and
            BoyaIrsaliye.FaturaKesilecek = 1 and
            ((:FKendiOrmeFirmamizaFatKesilsin = 1 or BoyaIrsaliye.FirmaID <> :OrmeFirmaKodumuz) and BoyaIrsaliye.FirmaID <> :IplikFirmaKodumuz) and
            BoyaIrsaliyeDetay.FaturaKesilecek = 1 and (BoyaIrsaliyeDetay.IrsaliyeTarihi <= :FNow or BoyaIrsaliyeDetay.IrsaliyeTarihi < :FAyIlkGun) and
            (BoyaIrsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (BoyaIrsaliye.HareketTipiID = :AHareketTipi or :AHareketTipi is null) and
            (Firma.EFaturaKullanimda = :AEFaturaDurumu or :AEFaturaDurumu is null)
      group by BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.FirmaId, Firma.Adi, Firma.Adi2, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi,
               BoyaIrsaliye.HareketTipiId, BoyaIrsaliyeHareketTipi.Adi, BoyaIrsaliye.Sinif, Firma.EFaturaKullanimda, Firma.VergiHesapNo
      into :BoyaIrsaliyeId, :FirmaId, :FirmaAdi, :FirmaAdi2, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi,
            :HareketTipiId, :HareketTipiIdStr, :Sinif, :EFatura, :FirmaVergiHesapNo do
      begin
        if (AFirmaAdi2yiGorebilir = 0) then FirmaAdi2 = NUll;

        if (EFatura = 1) then EFaturaStr = 'Evet';
        else EFaturaStr = 'Hayır';

        for Select PartiStok.DovizID From BoyaIrsaliyeDetay
              Left Join PartiStok on PartiStok.PartiSinifID = BoyaIrsaliyeDetay.PartiSinifID
            Where BoyaIrsaliyeDetay.BoyaIrsaliyeID = :BoyaIrsaliyeId Group By PartiStok.DovizID into :FDovizID do
        begin
          if (FDovizID is not Null) then
          begin
            if (Char_Length(FDovizID) + Char_Length(DovizID) < 50) then
            begin
              if (DovizID <> '' ) then DovizID = DovizID || cast(', ' as varchar(2) character set UTF8);
              DovizID = DovizID || FDovizID;
            end
          end
        end

        suspend;
      end
  end
  else if (ARaporTipi = 1) then
  begin
    for Select OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.FirmaId, Firma.Adi, Firma.Adi2, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
               OrmeIrsaliye.HareketTipi, OrmeStokHareketTipi.Adi, OrmeIrsaliye.Sinif, OrmeIrsaliye.DovizID, Firma.EFaturaKullanimda, Firma.VergiHesapNo From OrmeIrsaliye
      left join Firma on Firma.FirmaId = OrmeIrsaliye.FirmaId
      left join StokHareket on StokHareket.OrmeIrsaliyeId = OrmeIrsaliye.OrmeIrsaliyeId
      left join OrmeStokHareketTipi on OrmeStokHareketTipi.OrmeStokHareketTipiId = OrmeIrsaliye.HareketTipi
      where OrmeIrsaliye.DokumaHareketi = 0 and (OrmeIrsaliye.IrsaliyeTarihi <= :FNow or OrmeIrsaliye.IrsaliyeTarihi < :FAyIlkGun) and
            (OrmeIrsaliye.HareketTipi <> :shtFasonAlis) and
            OrmeIrsaliye.FirmaID <> :IplikFirmaKodumuz and
            StokHareket.OrmeFaturaId is null and OrmeIrsaliye.TransferYeri = 0 and StokHareket.HamIade = 0 and
            StokHareket.FaturaKesilecek = 1 and
            (OrmeIrsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (OrmeIrsaliye.HareketTipi = :AHareketTipi or :AHareketTipi is null) and
            (Firma.EFaturaKullanimda = :AEFaturaDurumu or :AEFaturaDurumu is null)
      group by OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.FirmaId, Firma.Adi, Firma.Adi2, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
               OrmeIrsaliye.HareketTipi, OrmeStokHareketTipi.Adi, OrmeIrsaliye.Sinif, OrmeIrsaliye.DovizID, Firma.EFaturaKullanimda, Firma.VergiHesapNo
      into :OrmeIrsaliyeId, :FirmaId, :FirmaAdi, :FirmaAdi2, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi,
            :HareketTipiId, :HareketTipiIdStr, :Sinif, :DovizID, :EFatura, :FirmaVergiHesapNo do
    begin
      if (AFirmaAdi2yiGorebilir = 0) then FirmaAdi2 = NUll;

      if (EFatura = 1) then EFaturaStr = 'Evet';
      else EFaturaStr = 'Hayır';

      suspend;
    end
  end
  else if (ARaporTipi = 2) then
  begin
    Select shtSatis, shtAlis from IplikStokHareketTipiGetir into :shtSatis, :shtAlis;

    for Select IplikIrsaliye.IplikIrsaliyeId, IplikIrsaliye.FirmaId, Firma.Adi, Firma.Adi2, IplikIrsaliye.SeriNo, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi,
               IplikIrsaliye.HareketTipi, IplikIrsaliye.Sinif, IplikIrsaliye.DovizID, Firma.EFaturaKullanimda, Firma.VergiHesapNo From IplikIrsaliye
      left join Firma on Firma.FirmaId = IplikIrsaliye.FirmaId
      where (IplikIrsaliye.IrsaliyeTarihi <= :FNow or IplikIrsaliye.IrsaliyeTarihi < :FAyIlkGun) and
            IplikIrsaliye.FaturaKesilecek = 1 and
/*            IplikIrsaliye.HareketTipi = :shtSatis and*/
            IplikIrsaliye.IplikFaturaId is null and
            (IplikIrsaliye.FirmaID <> :OrmeFirmaKodumuz and IplikIrsaliye.FirmaID <> :FirmaKodumuz) and
            (IplikIrsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (IplikIrsaliye.HareketTipi = :AHareketTipi or :AHareketTipi is null) and
            (Firma.EFaturaKullanimda = :AEFaturaDurumu or :AEFaturaDurumu is null)
      into :IplikIrsaliyeId, :FirmaId, :FirmaAdi, :FirmaAdi2, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi,
            :HareketTipiId, :Sinif, :DovizID, :EFatura, :FirmaVergiHesapNo do
    begin
      if (HareketTipiID = :shtSatis) then HareketTipiIdStr = 'Satış';
      else if (HareketTipiID = :shtAlis) then HareketTipiIdStr = 'Alış';

      if (EFatura = 1) then EFaturaStr = 'Evet';
      else EFaturaStr = 'Hayır';

      if (AFirmaAdi2yiGorebilir = 0) then FirmaAdi2 = NUll;
      suspend;
    end
  end
  else if (ARaporTipi = 3) then
  begin
    Select shtSatis, shtAlis, shtSatisIade, shtAlisIade, shtFasonUretim, shtDepoTransfer, shtUretimeCikis, shtFasonUretimeCikis
    from StokHareketTipiGetir into :shtSatis, :shtAlis, :shtSatisIade, :shtAlisIade, :shtFasonUretim, :shtDepoTransfer, :shtUretimeCikis, :shtFasonUretimeCikis;

    for Select Irsaliye.IrsaliyeId, Irsaliye.FirmaId, Firma.Adi, Firma.Adi2, Irsaliye.SeriNo, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi,
               Irsaliye.HareketTipi, Irsaliye.Sinif, Irsaliye.DovizID, Firma.VergiHesapNo, Irsaliye.IslemKaynagi From Irsaliye
      left join Firma on Firma.FirmaId = Irsaliye.FirmaId
      where (Irsaliye.IrsaliyeTarihi <= :FNow or Irsaliye.IrsaliyeTarihi < :FAyIlkGun) and
            Irsaliye.FaturaId is null and
            Irsaliye.Faturakesilecek = 1 and
            (Irsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (Irsaliye.HareketTipi = :AHareketTipi or :AHareketTipi is null) and
            Irsaliye.HareketTipi not in (:shtDepoTransfer, :shtUretimeCikis, :shtFasonUretimeCikis)
      into :StokIrsaliyeId, :FirmaId, :FirmaAdi, :FirmaAdi2, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi,
            :HareketTipiId, :Sinif, :DovizID, :FirmaVergiHesapNo, :IslemKaynagi do
    begin
      HareketTipiIdStr = '';
      if (HareketTipiID = :shtSatis) then HareketTipiIdStr = 'Satış';
      else if (HareketTipiID = :shtAlis) then HareketTipiIdStr = 'Alış';
      else if (HareketTipiID = :shtAlisIade) then HareketTipiIdStr = 'Alış İade';
      else if (HareketTipiID = :shtSatisIade) then HareketTipiIdStr = 'Satış İade';
      else if (HareketTipiID = :shtFasonUretim) then HareketTipiIdStr = 'Fason Üretim';

      if (IslemKaynagi = sikHazirGiyimKesim) then
        HareketTipiIdStr = HareketTipiIdStr || ' (Kesim)';

      if (AFirmaAdi2yiGorebilir = 0) then FirmaAdi2 = NUll;
      suspend;
    end
  end
  else if (ARaporTipi = 4) then
  begin
    for Select OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.FirmaId, Firma.Adi, Firma.Adi2, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
               OrmeIrsaliye.HareketTipi, OrmeStokHareketTipi.Adi, OrmeIrsaliye.Sinif, OrmeIrsaliye.DovizID, Firma.EFaturaKullanimda, Firma.VergiHesapNo From OrmeIrsaliye
      left join Firma on Firma.FirmaId = OrmeIrsaliye.FirmaId
      left join StokHareket on StokHareket.OrmeIrsaliyeId = OrmeIrsaliye.OrmeIrsaliyeId
      left join OrmeStokHareketTipi on OrmeStokHareketTipi.OrmeStokHareketTipiId = OrmeIrsaliye.HareketTipi
      where OrmeIrsaliye.DokumaHareketi = 1 and (OrmeIrsaliye.IrsaliyeTarihi <= :FNow or OrmeIrsaliye.IrsaliyeTarihi < :FAyIlkGun) and
/*            (OrmeIrsaliye.HareketTipi = :shtSatis or OrmeIrsaliye.HareketTipi = :ShtAlisIade) and*/
            OrmeIrsaliye.FirmaID <> :IplikFirmaKodumuz and
            StokHareket.OrmeFaturaId is null and OrmeIrsaliye.TransferYeri = 0 and StokHareket.HamIade = 0 and
            StokHareket.FaturaKesilecek = 1 and
            (OrmeIrsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (OrmeIrsaliye.HareketTipi = :AHareketTipi or :AHareketTipi is null) and
            (Firma.EFaturaKullanimda = :AEFaturaDurumu or :AEFaturaDurumu is null)
      group by OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.FirmaId, Firma.Adi, Firma.Adi2, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
               OrmeIrsaliye.HareketTipi, OrmeStokHareketTipi.Adi, OrmeIrsaliye.Sinif, OrmeIrsaliye.DovizID, Firma.EFaturaKullanimda, Firma.VergiHesapNo
      into :OrmeIrsaliyeId, :FirmaId, :FirmaAdi, :FirmaAdi2, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi,
            :HareketTipiId, :HareketTipiIdStr, :Sinif, :DovizID, :EFatura, :FirmaVergiHesapNo do
    begin
      if (AFirmaAdi2yiGorebilir = 0) then FirmaAdi2 = NUll;

      if (EFatura = 1) then EFaturaStr = 'Evet';
      else EFaturaStr = 'Hayır';

      suspend;
    end
  end
  else
  begin
    for Select HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.FirmaId, Firma.Adi, Firma.Adi2, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi,
               HamStokIrsaliye.HamStokHareketTipiID, HamStokIrsaliye.Sinif, HamStokIrsaliye.DovizID, Firma.VergiHesapNo From HamStokIrsaliye
      left join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
      left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
      where HamStokMaster.PartiSinifId is null and (HamStokIrsaliye.IrsaliyeTarihi <= :FNow or HamStokIrsaliye.IrsaliyeTarihi < :FAyIlkGun) and
            HamStokMaster.HamStokFaturaId is null and HamStokIrsaliye.EvrakTipi = 1 and
            (HamStokIrsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (HamStokIrsaliye.HamStokHareketTipiID = :AHareketTipi or :AHareketTipi is null) and
            (HamStokIrsaliye.Fason = :AFason or :AFason is null)
      Group by HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.FirmaId, Firma.Adi, Firma.Adi2, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi,
               HamStokIrsaliye.HamStokHareketTipiID, HamStokIrsaliye.Sinif, HamStokIrsaliye.DovizID, Firma.VergiHesapNo
      into :HamStokIrsaliyeId, :FirmaId, :FirmaAdi, :FirmaAdi2, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi,
            :HareketTipiId, :Sinif, :DovizID, :FirmaVergiHesapNo do
    begin
      if (HareketTipiID = 0) then HareketTipiIdStr = 'Giriş';
      else if (HareketTipiID = 1) then HareketTipiIdStr = 'Satış İade';
      else HareketTipiID = '';

      if (AFirmaAdi2yiGorebilir = 0) then FirmaAdi2 = NUll;
      suspend;
    end
  end
end
