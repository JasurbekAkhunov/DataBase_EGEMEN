CREATE PROCEDURE DOZAJLAMAISEMRIBILGISIGETIR(
    ADOZAJLAMAISEMRIID INTEGER)
RETURNS (
    DOZAJLAMAISEMRIID INTEGER,
    RECETEISEMRIYIL INTEGER,
    RECETEISEMRIID INTEGER,
    RECETEISEMRIIDSTR VARCHAR(30),
    PARTIKAYITID INTEGER,
    PARTIIDSTR VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(120),
    TOPLAMKILO NUMERIC(12,3),
    PLANLANANMAKINAID VARCHAR(10),
    PARTIASAMALARIID INTEGER,
    PARTISABLONISEMRIID INTEGER,
    STOKTANDUSULDU SMALLINT,
    BASKITARTIM EGEBOOLEAN,
    SIPARISGRUPID SIPARISGRUPKODUSIZE,
    AKUPLE INTEGER,
    AKUPLEPLANLANANMAKINAID VARCHAR(10),
    TOPLAMMETRE NUMERIC(12,3),
    ISEMRIDESENVARYANTSTR VARCHAR(250))
AS
declare variable PartiYil              integer;
declare variable PartiId               integer;
declare variable FPartiIdStr           varchar(30);
begin
/*ASUS icin yapildi */
  Select DozajlamaIsEmri.DozajlamaIsEmriId, DozajlamaIsEmri.ReceteIsEmriYil, DozajlamaIsEmri.ReceteIsEmriId,
         Parti.PartiYil, Parti.PartiId, DozajlamaIsEmri.PartiAsamalariId, DozajlamaIsEmri.PartiKayitId,
         DozajlamaIsEmri.PartiSablonIsEmriId, DozajlamaIsEmri.StoktanDusuldu, BoyaSiparis.SiparisGrupId from DozajlamaIsEmri
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = DozajlamaIsEmri.PartiAsamalariId
    left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
    left join BoyaSiparis on Boyasiparis.boyasipariskayitid = parti.boyasipariskayitid
     where DozajlamaIsEmri.DozajlamaIsEmriID = :ADozajlamaIsEmriId
  into :DozajlamaIsEmriId, :ReceteIsEmriYil, :ReceteIsEmriId, :PartiYil, :PartiId, :PartiAsamalariId,
       :PartiKayitId, :PartiSablonIsEmriId, :StoktanDusuldu, :SiparisGrupId;

  if (DozajlamaIsEmriId is null) then exit;

  if (PartiSablonIsEmriId is not null or PartiKayitId is not null) then BaskiTartim = 1;
  else BaskiTartim = 0;

  if (PartiSablonIsEmriId is not null) then
  begin
    Select PartiSablonIsEmri.DesenId, Desen.Adi || '  ' || DesenVaryant.VaryantAdi, '(' || ReceteIsEmriYil || ') ' || ReceteIsEmriId,
    :DozajlamaIsEmriId||'-'||PartiSablonIsEmri.DesenId||'-'||DesenVaryant.VaryantAdi from PartiSablonIsEmri
    Left Join Desen on Desen.DesenId = PartiSablonIsEmri.DesenId
    Left Join DesenVaryant on DesenVaryant.DesenVaryantId = PartiSablonIsEmri.DesenVaryantId
    where PartiSablonIsEmri.PartiSablonIsEmriId = :PartiSablonIsEmriId
    into :ReceteId, :ReceteAdi, :ReceteIsEmriIdStr, :IsEmriDesenVaryantStr;

    Select Sum(PartiStok.SonCikisMetre), Sum(PartiStok.SonCikisKilo) from PartiSablon
    left join PartiStok on PartiStok.PartiKayitId = PartiSablon.PartiKayitId
    where PartiSablon.PartiSablonIsEmriId = :PartiSablonIsEmriId and PartiSablon.Sira = 0
    into :ToplamMetre, :ToplamKilo;

    PartiIdStr = '';
    select List('(' || Parti.PartiYil || ') ' || Parti.PartiId, ', ') from PartiSablon
      left join Parti on Parti.PartiKayitId = PartiSablon.PartiKayitId
      where PartiSablon.PartiSablonIsEmriId = :PartiSablonIsEmriId and PartiSablon.Sira = 0
    into :PartiIdStr;

  end
  else if (PartiKayitId is not null) then
  begin
    PartiIdStr = '';

    Select '(' || Parti.PartiYil || ') ' || Parti.PartiId, Desen.DesenId, Desen.Adi || '  ' || DesenVaryant.VaryantAdi,
    :DozajlamaIsEmriId||'-'||Desen.DesenId||'-'||DesenVaryant.VaryantAdi from Parti
    Left Join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    Left Join Desen on Desen.DesenId = DesenVaryant.DesenId
    where Parti.PartiKayitId = :PartiKayitId
    into :PartiIdStr ,:ReceteId, :ReceteAdi, :IsEmriDesenVaryantStr;

    Select Sum(PartiStok.SonCikisMetre), Sum(PartiStok.SonCikisKilo) from PartiStok
      where PartiStok.PartiKayitId = :PartiKayitId into :ToplamMetre, :ToplamKilo;
  end
  else if (ReceteIsEmriId is not null) then
  begin
    ReceteIsEmriIdStr = '(' || EgemenIntegerToStr(ReceteIsEmriYil) || ') ' || EgemenIntegerToStr(ReceteIsEmriId) || ' ';

    Select ReceteIsEmri.ReceteId, Recete.Adi from ReceteIsEmri
      left join Recete on Recete.ReceteID = ReceteIsEmri.ReceteId
      where ReceteIsEmri.ReceteIsEmriYil = :ReceteIsEmriYil and ReceteIsEmri.ReceteIsEmriId = :ReceteIsEmriId
      into :ReceteId, :ReceteAdi;

    Select Max(UretimPlanlama.MakinaId) from ReceteIsEmriDetay
     left join BolumPlanlama on BolumPlanlama.PartiAsamalariId = ReceteIsEmriDetay.PartiAsamalariId
     left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
     left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
      where ReceteIsEmriDetay.ReceteIsEmriYil = :ReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :ReceteIsEmriId
     into :PlanlananMakinaId;

    Select Sum(PartiAsamaStok.GirisKilo), Sum(PartiAsamaStok.GirisMetre) from ReceteIsEmriDetay
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariID = ReceteIsEmriDetay.PartiAsamalariId
    where ReceteIsEmriDetay.ReceteIsEmriYil = :ReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :ReceteIsEmriId and
          PartiAsamaStok.AsamayaDahil = 1
    into :ToplamKilo, :ToplamMetre;

    PartiIdStr = '';
    for select Parti.PartiYil, Parti.PartiId from ReceteIsEmriDetay
      left join Parti on Parti.PartiKayitId = ReceteIsEmriDetay.PartiKayitId
      where ReceteIsEmriDetay.ReceteIsEmriYil = :ReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :ReceteIsEmriId
      group by Parti.PartiYil, Parti.PartiId
      into :PartiYil, :PartiId do
    begin
      FPartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (Char_Length(PartiIdStr) + Char_Length(FPartiIdStr) < 90) then
      begin
        if (PartiIdStr <> '' ) then PartiIdStr = PartiIdStr || cast(', ' as varchar(2) character set UTF8);
        PartiIdStr = PartiIdStr || FPartiIdStr;
      end
    end
  end
  else if (PartiId is not null) then
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    Select PartiAsamalari.ReceteId, Recete.Adi from PartiAsamalari
      left join Recete on Recete.ReceteID = PartiAsamalari.ReceteId
      where PartiAsamalari.PartiAsamalariId = :PartiAsamalariId
      into :ReceteId, :ReceteAdi;

    Select Max(UretimPlanlama.MakinaId) from BolumPlanlama
     left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
     left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
     where BolumPlanlama.PartiAsamalariId = :PartiAsamalariId
     into :PlanlananMakinaId;

    Select PartiAsamalari.Akuple from PartiAsamalari
      where PartiAsamalari.PartiAsamalariID = :PartiAsamalariId
      into  :Akuple;

    select m.makinaid from makina
      left join Makina M on M.otomasyonMakinaNo = cast(coalesce(makina.slavemachine,-1) as varchar(10))
      where Makina.MakinaId= :PlanlananMakinaId
      into AkuplePlanlananMakinaId;


    Select Sum(PartiAsamaStok.GirisKilo), Sum(PartiAsamaStok.GirisMetre) from PartiAsamalari
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariID = PartiAsamalari.PartiAsamalariId
    where PartiAsamalari.PartiAsamalariId = :PartiAsamalariId and  PartiAsamaStok.AsamayaDahil = 1
    into :ToplamKilo, :ToplamMetre;

  end
  suspend;
end
