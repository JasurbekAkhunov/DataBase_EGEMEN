   left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
   left join HamStokBirim on HamStokBirim.HamStokBirimId = BoyaSiparisDetay.HamStokBirimId
   left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
   left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
   left join SiparisGrup SiparisGrup3 on SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
   left join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
   left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
   left join BaskiTipi on BaskiTipi.BaskiTipiID = BoyaSiparisDetay.BaskiTipiID
   left join PaketlemeSekli On PaketlemeSekli.PaketlemeSekliID = BoyaSiparisDetay.PaketlemeSekliID
   left join TeslimSekli on TeslimSekli.TeslimSekliID = BoyaSiparis.TeslimSekliID
   left join PartiGrup on PartiGrup.PartiGrupID = Parti.PartiGrupID
   where BoyaFaturaDetay.BoyaFaturaId is not null
   into :BoyaFaturaId, :FaturaTipi, :Tarih, :FaturaSeriNo, :FaturaNo, :OdemeTarihi, :FaturaGrupId, :FaturaGrupAdi,
        :FirmaId, :FirmaAdi, :FirmaAdi2, :DovizId, :FirmaDovizKuruStr, :EntegrasyonStr, :TevkifatStr, :ParasalIskonto, :IskontoOrani, :DovizKuru,
        :FHareketTipiId, :BoyaFaturaDetayId, :PartiKayitId, :PartiSinifId, :HamAdiToplami, :PartiYil, :PartiId, :Fason,
        :ReceteId, :ReceteAdi, :SiparisProsesId, :SiparisProsesAdi, :SiparisProsesAdi2, :SipIlaveIslemler, :SipOzelTestler, :SipOzelApreler,
        :YapilanIslemler, :Aciklama, :PaketAciklama1, :PaketAciklama2,
        :SiparisRenkAdi, :MusteriRenkKoduVeAdi, :KaliteNo, :Miktar, :BirimFiyat, :VadeGun,
        :SiparisBirimFiyat, :KdvOrani, :SiparisDovizId,
        :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :BoyaIrsaliyeId, :BoyaIrsaliyeDetayId, :BoyaSiparisYil, :BoyaSiparisId,
        :BoyaSiparisDetayId, :GirisIrsaliyeNoStr, :GirisIrsaliyeTarihiStr, :HamStokBirimId, :BirimId, :FHamTipiId,
        :IslemTipiId, :IslemTipiAdi, :IslemGrubuId, :IslemGrubuAdi, :PartiPaketId, :FNetFiyatYuvarlamaCarpani, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
       :IhracatFaturasi, :Navlun, :Sigorta, :Pesinat,
       :OdemeSekliId,:OdemeSekliAdi,:AkreditifNo,
       :EoriNo,:GtipNo,:DahiliIslemBelgeNo, :KomisyoncuId,:KomisyoncuAdi,:KomisyonOrani,
       :FirmaSevkAdresiId,:FirmaSevkAdresiAdresAdi,:SevkAdresi1,:SevkAdresi2,:SevkAdresi3,
       :SevkSemt,SevkSehir,SevkUlke, :BankaHesapId,:BankaHesapAdi, :HamAdi2, :HamAdi3, :HamAdi4, :BelgeHamAdi, :LabReceteCalismaBirimFiyat, :LabReceteCalismaDovizID,
       :RenkDerinligiId, :RenkDerinligiIdStr, :BaskiTipiID, :BaskiTipiAdi, :FirmaTransferKodu, :SiparisFasonBirimFiyat, :SiparisFasonDovizId, :PaketlemeSekliAdi, :TeslimSekliAdi, :Markali,
       :Komisyonlu, :FiyatKomisyon, :FiyatKomisyonOrani, :FAturaMiktarTipi, :FirmaUlke, :BaskiBirimFiyati, :BaskiBirimFiyatDovizId, :PartiStokMiktar, :PartiGrupID, :PartiGrupAdi,
       :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi, :FirmaDovizKuru1, :FirmaDovizKuru2, :FirmaVergiHesapNo, :PartiStokFasonBirimFiyat, :PartiStokFasonDovizId,
       :PartiSiraNo do
   begin
     Tutar = Miktar * BirimFiyat;
     Tutar = Yuvarla(Tutar, FNetFiyatYuvarlamaCarpani);

     TutarFason = Miktar * SiparisFasonBirimFiyat;
     TutarFason = Yuvarla(TutarFason, FNetFiyatYuvarlamaCarpani);

     if (SiparisFasonDovizId <> RaporDovizID) then  select DovizKuru from EnYakinDovizKuruGetir(:SiparisFasonDovizId, :RaporDovizID, :Tarih) into :FDovizKuruFason;
     else FDovizKuruFason = 1;

     RaporTutarFason = TutarFason * FDovizKuruFason ;

     if (IhracatFaturasi = 1) then IhracatFaturasiStr = 'Evet';
     else IhracatFaturasiStr = null;

     if (PartiSinifId is null) then
     begin
        Fason = 0;
        if (FHareketTipiId = 0) then PartiIdStr = 'Ham Stok Sat';
        else if (FHareketTipiId = 1) then PartiIdStr = 'Ham Stok Al ade';
        else exception Kayithatasi;
        BoyaSiparisIdStr = '';

        if (FHamTipiId <> 2) then
        begin
          HamStokBirimId = 0;
          BirimId = 'Kg';
        end
        else
        begin
          HamStokBirimId = 1;
          BirimId = 'Mt';
        end
     end
     else
     begin
       PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
       BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
     end

     RaporTutar = Tutar * DovizKuru;

     if (SiparisDovizId is not null and FFaturaDovizId <> SiparisDovizId) then
       DovizKuruStr = Format_Float('', DovizKuru1) || ' ' || SiparisDovizId || '-' || Format_Float('', DovizKuru2) || ' ' || FFaturaDovizId;
     else
       DovizKuruStr = '';

     Select Sum(Kilo) from PartiStok where PartiStok.PartiKayitID = :PartiKayitID and PartiStok.PartiStokIptalKayitid is null into :FToplamPartiKilo;
     Select BoyaKGMaliyeti, KimyasalKGMaliyeti from PartiBoyaKimyasalMaliyetiGetir(:PartiKayitId, :FToplamPartiKilo) into :BoyaKGMaliyeti, :KimyasalKGMaliyeti;
     select first 1 AsamaUMUzerindenAsamaMaliyeti from PartiSinifMaliyetKumulatif where PartiSinifId = :PartiSinifId into  :AsamaUMUzerindenAsamaMaliyeti;

     KomisyonTutari = KomisyonOrani * Miktar;

     suspend;
   end
end
