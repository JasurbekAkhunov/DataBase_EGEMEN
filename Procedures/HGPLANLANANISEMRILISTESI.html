CREATE PROCEDURE HGPLANLANANISEMRILISTESI(
    AMODELURETIMISLEMTIPIID MODELGRUPKODUSIZE,
    ABASLAMATARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    HGISEMRIKAYITID INTEGER,
    HGSIPARISKAYITID INTEGER,
    HGSIPARISDETAYID INTEGER,
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER,
    SIPARISFIRMAID FIRMAKODUSIZE,
    SIPARISFIRMAADI ADI100,
    SIPARISTARIHI TIMESTAMP,
    MODELID MODELKODUSIZE,
    MODELADI ADI40,
    TESLIMTARIHI TIMESTAMP,
    ORMEMAKINAID MAKINAKODUSIZE,
    MAKINANO INTEGER,
    URETIMPARTINO VARCHAR(30),
    TERMINTARIHI TIMESTAMP,
    MIKTARGUN INTEGER,
    PLANLAMATARIHI TIMESTAMP,
    BASLAMATARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    PLANLAYANKULLANICIID INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PLANLANANTOPLAMMIKTAR INTEGER,
    GERCEKLESENTOPLAMMIKTAR INTEGER,
    MUSTERISIPARISNO VARCHAR(50))
AS
declare variable KesimIslemi EgeBoolean;
begin
  select ModelUretimIslemTipi.KesimIslemi from ModelUretimIslemTipi where ModelUretimIslemTipi.ModelUretimIslemTipiId = :AModelUretimIslemTipiId into :KesimIslemi;

  if (KesimIslemi = 0) then
    for select HgIsEmri.HgIsEmriKayitId, HgSiparisDetay.HgSiparisKayitId, HgSiparisDetay.HgSiparisDetayId, HgSiparis.HgSiparisYil, HgSiparis.HgSiparisId,
        Firma.FirmaId, Firma.Adi, HgSiparis.SiparisTarihi, Model.ModelId, Model.Adi, HgSiparisDetay.TeslimTarihi, HgIsEmri.OrmeMakinaId, HgIsEmri.UretimPartiNo,
        HgIsEmri.TerminTarihi, OrmeMakina.MakinaNo, HgIsEmri.Notu, HgIsEmri.MiktarGun, HgIsEmri.PlanlamaTarihi, HgIsEmri.BaslamaTarihi, HgIsEmri.BitisTarihi, HgIsEmri.PlanlayanKullaniciId, HgSiparisDetay.MusteriSiparisNo
        from HgIsEmri
        left join HgIsEmriUretimIslemleri on HgIsEmriUretimIslemleri.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
        left join HgSipDetayUretimIslemleri on HgSipDetayUretimIslemleri.HgSipDetayUretimIslemleriId = HgIsEmriUretimIslemleri.HgSipDetayUretimIslemleriId
        left join HgSiparisDetay on HgSiparisDetay.HgSiparisDetayId = HgSipDetayUretimIslemleri.HgSiparisDetayId
        left join Model on Model.ModelKayitId = HgSiparisDetay.ModelKayitId
        left join HgSiparis on HgSiparis.HgSiparisKayitId = HgSiparisDetay.HgSiparisKayitId
        left join Firma on Firma.FirmaId = HgSiparis.FirmaId
        left join OrmeMakina on OrmeMakina.OrmeMakinaId = HgIsEmri.OrmeMakinaId
        where HgSipDetayUretimIslemleri.ModelUretimIslemTipiId = :AModelUretimIslemTipiId and
        HgIsEmri.Bitti = 0 and
        HgIsEmri.Planlandi = 1 and
        HgIsEmri.KesimTalimati = 0 and
        (HgIsEmri.BaslamaTarihi >= :ABaslamaTarihi and HgIsEmri.BaslamaTarihi <= :ABitisTarihi or
         HgIsEmri.BitisTarihi >= :ABaslamaTarihi and HgIsEmri.BaslamaTarihi <= :ABitisTarihi or
         HgIsEmri.BaslamaTarihi > :ABaslamaTarihi and HgIsEmri.BitisTarihi > :ABitisTarihi)
        into
        :HgIsEmriKayitId, :HgSiparisKayitId, :HgSiparisDetayId, :HgSiparisYil, :HgSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisTarihi, :ModelId, :ModelAdi,
        :TeslimTarihi, :OrmeMakinaId, :UretimPartiNo, :TerminTarihi, :MakinaNo, :Notu, :MiktarGun, :PlanlamaTarihi, :BaslamaTarihi, :BitisTarihi, :PlanlayanKullaniciId, :MusteriSiparisNo do
    begin
      select Sum(HgIsEmriUrun.PlanlananMiktar), Sum(HgIsEmriUrun.GerceklesenMiktar) from HgIsEmriUrun where HgIsEmriUrun.HgIsEmriKayitId = :HgIsEmriKayitId into :PlanlananToplamMiktar, :GerceklesenToplamMiktar;

      suspend;
    end
  else
    for select HgIsEmri.HgIsEmriKayitId, HgSiparisDetay.HgSiparisKayitId, HgSiparisDetay.HgSiparisDetayId, HgSiparis.HgSiparisYil, HgSiparis.HgSiparisId,
        Firma.FirmaId, Firma.Adi, HgSiparis.SiparisTarihi, Model.ModelId, Model.Adi, HgSiparisDetay.TeslimTarihi, HgIsEmri.OrmeMakinaId, HgIsEmri.UretimPartiNo,
        HgIsEmri.TerminTarihi, OrmeMakina.MakinaNo, HgIsEmri.Notu, HgIsEmri.MiktarGun, HgIsEmri.PlanlamaTarihi, HgIsEmri.BaslamaTarihi, HgIsEmri.BitisTarihi, HgIsEmri.PlanlayanKullaniciId, HgSiparisDetay.MusteriSiparisNo
        from HgIsEmri
        left join HgSiparisDetay on HgSiparisDetay.HgSiparisDetayId = HgIsEmri.HgSiparisDetayId
        left join Model on Model.ModelKayitId = HgSiparisDetay.ModelKayitId
        left join HgSiparis on HgSiparis.HgSiparisKayitId = HgSiparisDetay.HgSiparisKayitId
        left join Firma on Firma.FirmaId = HgSiparis.FirmaId
        left join OrmeMakina on OrmeMakina.OrmeMakinaId = HgIsEmri.OrmeMakinaId
        where HgIsEmri.Planlandi = 1 and
        HgIsEmri.Bitti = 0 and
        HgIsEmri.KesimTalimati = 1 and
        (HgIsEmri.BaslamaTarihi >= :ABaslamaTarihi and HgIsEmri.BaslamaTarihi <= :ABitisTarihi or
         HgIsEmri.BitisTarihi >= :ABaslamaTarihi and HgIsEmri.BaslamaTarihi <= :ABitisTarihi or
         HgIsEmri.BaslamaTarihi > :ABaslamaTarihi and HgIsEmri.BitisTarihi > :ABitisTarihi)
        into
        :HgIsEmriKayitId, :HgSiparisKayitId, :HgSiparisDetayId, :HgSiparisYil, :HgSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisTarihi, :ModelId, :ModelAdi,
        :TeslimTarihi, :OrmeMakinaId, :UretimPartiNo, :TerminTarihi, :MakinaNo, :Notu, :MiktarGun, :PlanlamaTarihi, :BaslamaTarihi, :BitisTarihi, :PlanlayanKullaniciId, :MusteriSiparisNo do
    begin
      select Sum(HgIsEmriPastalUrun.PlanlananMiktar), Sum(HgIsEmriPastalUrun.GerceklesenMiktar) from HgIsEmriPastal
      Left Join HgIsEmriPastalUrun on HgIsEmriPastalUrun.HgIsEmriPastalId = HgIsEmriPastal.HgIsEmriPastalId
      where HgIsEmriPastal.HgIsEmriKayitId = :HgIsEmriKayitId into :PlanlananToplamMiktar, :GerceklesenToplamMiktar;

      suspend;
    end
end
