CREATE PROCEDURE PARTIASAMALARIKARTIGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIASAMALARIID INTEGER,
    SIRA INTEGER,
    RECETEID VARCHAR(20),
    SABLONRECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    RENKDERINLIGIADI VARCHAR(20),
    RECETEKAPALI INTEGER,
    LABRECETEGRUPID INTEGER,
    LABRENKGRUPID INTEGER,
    LABRENKGRUPADI VARCHAR(30),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    GRUPPROSESNO VARCHAR(10),
    GRUPPROSESADI VARCHAR(80),
    PROSESNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    OTOMASYONADAHILMI INTEGER,
    OTOMASYONADAHILMISTR VARCHAR(5),
    SERBESTASAMA INTEGER,
    BANYOCARPANI DOUBLE PRECISION,
    BANYOCARPANI2 DOUBLE PRECISION,
    BANYOCARPANI3 DOUBLE PRECISION,
    BANYOCARPANI4 DOUBLE PRECISION,
    HAMSTOKFLOTTECARPANI DOUBLE PRECISION,
    DOZAJLAMAISEMRIID INTEGER,
    DOZAJLAMAYAGONDERILDI INTEGER,
    DOZAJLAMAYAGONDERILDISTR VARCHAR(20),
    KOYULTMA DOUBLE PRECISION,
    DURULAMASUYU INTEGER,
    BOYASUYU INTEGER,
    BOYAMASEVIYESI INTEGER,
    ILAVEPROSES INTEGER,
    ILAVEPROSESSTR VARCHAR(4),
    ILAVEPROSESSEBEBIID INTEGER,
    ILAVEPROSESSEBEBIADI VARCHAR(20),
    POMPADEGERI INTEGER,
    TUR DOUBLE PRECISION,
    DUZE INTEGER,
    KULEHIZI DOUBLE PRECISION,
    LABRECETEATISCOLOR INTEGER,
    PARTITAMIRID INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    FIKSETIPIADI VARCHAR(20),
    RECETEBOYAASAMASIMI INTEGER,
    LABRECETEATISOKEY INTEGER,
    ILAVESARFVAR INTEGER,
    MINPARTISTOKDURUMU INTEGER,
    SARFFISISERINO VARCHAR(2),
    SARFFISINO INTEGER,
    SARFFISITARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    STOKGRUPID VARCHAR(4),
    LABRECETEKODU RECETEKODUSIZE,
    MAKINABILGIRECETEADI VARCHAR(50),
    RECETEISEMRIIDSTR VARCHAR(30),
    BIRLESTIRILENPARTILER VARCHAR(100),
    LABRECETEKALITEID KALITEKODUSIZE)
AS
declare variable FikseTipiId integer;
declare variable KS integer;
declare variable ReceteIsEmriYil integer;
declare variable ReceteIsEmriId integer;
begin
  for select PartiAsamalariId, Sira, PartiAsamalari.ReceteId, Recete.Adi, Recete.ReceteKapatma, Recete.LabreceteGrupId, LabRecete.FikseTipiId, PartiAsamalari.AsamaId, PAsama.Adi,
             PartiAsamalari.MakinaPlanlamaGrupId, PartiAsamalari.GrupProsesNo, MakinaPlanlamaGrupProses.Adi, PartiAsamalari.ProsesNotu,
             PartiAsamalari.OtomasyonaDahilmi, PartiAsamalari.SerbestAsama, PartiAsamalari.PartiTamirId,
             TamirTipi.Adi, PartiAsamalari.BanyoCarpani, PartiAsamalari.BanyoCarpani2, PartiAsamalari.BanyoCarpani3, PartiAsamalari.BanyoCarpani4,
             PartiAsamalari.HamStokFlotteCarpani, PartiAsamalari.Koyultma, PartiAsamalari.DurulamaSuyu, PartiAsamalari.BoyaSuyu, PartiAsamalari.Tur, PartiAsamalari.PompaDegeri, PartiAsamalari.Duze,
             PartiAsamalari.BoyamaSeviyesi, IK.Adi, PartiAsamalari.InsertTarihi, SK.Adi, PartiAsamalari.DegisimTarihi,
             RAsama.BoyaAsamasimi, LabReceteAtis.Okey, PartiAsamalari.IlaveProses, PartiAsamalari.IlaveProsesSebebiId, IlaveProsesSebebi.Adi,
             RenkDerinligi.Adi, LabReceteAtis.Color, Recete.LabRenkGrupId, LabRenkGrup.Adi, Recete.SablonReceteId, PartiAsamalari.StokGrupId, PartiAsamalari.KuleHizi, LabRecete.Labrecetekodu, MakinaBilgiRecete.Adi, LabRecete.KaliteId from PartiAsamalari
  left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = Recete.LabReceteAtisId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
  left join MakinaBilgiRecete on MakinaBilgiRecete.MakinaBilgiReceteId = PartiAsamalari.MakinaBilgiReceteId
  left join MakinaPlanlamaGrupProses on MakinaPlanlamaGrupProses.MakinaPlanlamaGrupId = PartiAsamalari.MakinaPlanlamaGrupId and MakinaPlanlamaGrupProses.GrupProsesNo = PartiAsamalari.GrupProsesNo
  left join Asama PAsama on PAsama.AsamaId = PartiAsamalari.AsamaId
  left join Asama RAsama on RAsama.AsamaId = Recete.UstAsamaId
  left join Kullanici IK on IK.KullaniciId = PartiAsamalari.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId = PartiAsamalari.KullaniciId
  left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
  left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
  left join IlaveProsesSebebi on IlaveProsesSebebi.IlaveProsesSebebiId = PartiAsamalari.IlaveProsesSebebiId
  left join LabRenkGrup on LabRenkGrup.LabRenkGrupId = Recete.LabRenkGrupId
  where PartiAsamalari.PartiKayitId = :APartiKayitId order by sira
  into :PartiAsamalariId, :Sira, :ReceteId, :ReceteAdi, :ReceteKapali, :LabreceteGrupId, :FikseTipiId, :AsamaId, :AsamaAdi,
       :MakinaPlanlamaGrupId, :GrupProsesNo, :GrupProsesAdi, :ProsesNotu, :OtomasyonaDahilmi, :SerbestAsama,
       :PartiTamirId, :TamirTipiAdi, :BanyoCarpani, :BanyoCarpani2, :BanyoCarpani3, :BanyoCarpani4, :HamStokFlotteCarpani,
       :Koyultma, :DurulamaSuyu, :BoyaSuyu, :Tur, :PompaDegeri, :Duze, :BoyamaSeviyesi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :ReceteBoyaAsamasimi, :LabReceteAtisOkey,
       :IlaveProses, :IlaveProsesSebebiId, :IlaveProsesSebebiAdi, :RenkDerinligiAdi, :LabReceteAtisColor,
       :LabRenkGrupId, :LabRenkGrupAdi, :SablonReceteId, :StokGrupId, :KuleHizi, :Labrecetekodu, :MakinaBilgiReceteAdi, :LabReceteKaliteId do
  begin
    DozajlamaIsEmriId = null;
    ReceteIsEmriYil = null;
    ReceteIsEmriId = null;

    select ReceteIsEmriYil, ReceteIsEmriId from ReceteIsEmriDetay where ReceteIsEmriDetay.PartiAsamalariId = :PartiAsamalariId
      into :ReceteIsEmriYil, :ReceteIsEmriId;

    if (ReceteIsEmriYil is null) then
      select DozajlamaIsEmriId, Gonderildi from DozajlamaIsEmri where DozajlamaIsEmri.PartiAsamalariId = :PartiAsamalariId into :DozajlamaIsEmriId, :DozajlamayaGonderildi;
    else
      select DozajlamaIsEmriId, Gonderildi from DozajlamaIsEmri where DozajlamaIsEmri.ReceteIsEmriYil = :ReceteIsEmriYil and DozajlamaIsEmri.ReceteIsEmriId = :ReceteIsEmriId into :DozajlamaIsEmriId, :DozajlamayaGonderildi;

    if (DozajlamaIsEmriId is null) then DozajlamayaGonderildi = 0;

    if (DozajlamayaGonderildi = 1) then
       DozajlamayaGonderildiStr = 'Gönderildi';
    else
       DozajlamayaGonderildiStr = '';

    if (IlaveProses = 1) then
      IlaveProsesStr = 'Evet';
    else
      IlaveProsesStr = '';

    OtomasyonaDahilmiStr = null;
    if (OtomasyonaDahilmi = 1) then OtomasyonaDahilmiStr = 'Dahil';

    FikseTipiAdi = '';

    if (FikseTipiId = 0) then FikseTipiAdi = 'Ön Fikse';
    else if (FikseTipiId = 1) then FikseTipiAdi = 'Kasar Sonu Fikse';

    select count(*) from PartiAsamaIlaveSarf where PartiAsamaIlaveSarf.PartiAsamalariId = :PartiAsamalariId into :KS;
    if (KS = 0) then IlaveSarfVar = 0;
    else IlaveSarfVar = 1;

    select min(PartiStok.Durumu) from PartiAsamaStok
      left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
      where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
    into :MinPartiStokDurumu;
    SarfFisiSeriNo = null;
    SarfFisiNo = null;
    SarfFisiTarihi = null;

    select first 1 Stok.SeriNO, Stok.FisNo, Stok.FisTarihi from StokDetay
      left join Stok on Stok.StokId = StokDetay.StokID
      where StokDetay.PartiAsamalariId = :PartiAsamalariId
      into :SarfFisiSeriNo, :SarfFisiNo, :SarfFisiTarihi;

    ReceteIsEmriIdStr = Null;
    BirlestirilenPartiler = '';

    if (ReceteId is not null) then
      Select ReceteIsEmriIdStr, BirlestirilenPartiler from PAReceteIsEmriBilgisiGetir(:APartikayitId, :ReceteId, :PartiAsamalariId) into :ReceteIsEmriIdStr, :BirlestirilenPartiler;

    suspend;
  end
end
