CREATE PROCEDURE IPLIKIMALATAARAVERBASLAT(
    AIPLIKMAKINAID VARCHAR(10),
    APERSONELID INTEGER,
    AIPLIKARAVERMESEBEBIID INTEGER,
    ABASLADIBITTI INTEGER)
RETURNS (
    STATUS INTEGER,
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    IPLIKARAVERMESEBEBIID INTEGER,
    IPLIKARAVERMESEBEBIADI VARCHAR(50),
    TARIH TIMESTAMP,
    MESAJ VARCHAR(255))
AS
declare variable BasladiIplikMakinaDurumuId  integer;
declare variable BittiIplikMakinaDurumuId    integer;
declare variable IplikImalataAraVermeId      integer;
declare variable KayitSayisi            integer;
begin
  if (ABasladiBitti = 1) then
  begin
    Select max(IplikImalataAraVermeId) from IplikImalataAraVerme
      where IplikImalataAraVerme.IplikMakinaId = :AIplikMakinaId and BittiIplikMakinaDurumuId is null into :IplikImalataAraVermeId;

    if (IplikImalataAraVermeId is null) then
    begin
      Select IplikAraVermeSebebi.Adi from IplikAraVermeSebebi where IplikAraVermeSebebi.IplikAraVermeSebebiId = :AIplikAraVermeSebebiId into :IplikAraVermeSebebiAdi;

      if (IplikAraVermeSebebiAdi is null) then
      begin
        Status = 5;
        mesaj = 'Ara Verme Sebebi Kayitli Degil..';
      end
      else
      begin
        BasladiIplikMakinaDurumuId = gen_Id(IplikMakinaDurumu_id_gen, 1);

        Tarih = Current_TimeStamp(0);

        insert into IplikMakinaDurumu(IplikMakinaDurumuId, IplikMakinaId, Tarih, ImalataAraVerildi) values(:BasladiIplikMakinaDurumuId, :AIplikMakinaId, Current_TimeStamp(0), 1); /* ara verildi */

        insert into IplikImalataAraVerme(IplikMakinaId, BasladiIplikMakinaDurumuId, AraVerdirenPersonelId, IplikAraVermeSebebiId)
                          values(:AIplikMakinaId, :BasladiIplikMakinaDurumuId, :APersonelId, :AIplikAraVermeSebebiId);
        Status = 1;
        mesaj = 'Kayit Yapıldı';
      end
    end
    else
    begin
      Status = 2;
      mesaj = 'imalata daha once ara verilmis';

      Select IplikImalataAraVerme.AraVerdirenPersonelId, Personel.Adi, IplikImalataAraVerme.IplikAraVermeSebebiId, IplikAraVermeSebebi.Adi, IplikMakinaDurumu.Tarih from IplikImalataAraVerme
        left join Personel on Personel.PersonelId = IplikImalataAraVerme.AraVerdirenPersonelId
        left join IplikAraVermeSebebi on IplikAraVermeSebebi.IplikAraVermeSebebiId = IplikImalataAraVerme.IplikAraVermeSebebiId
        left join IplikMakinaDurumu on IplikMakinaDurumu.IplikMakinaDurumuId = IplikImalataAraVerme.BasladiIplikMakinaDurumuId
        where IplikImalataAraVerme.IplikImalataAraVermeId = :IplikImalataAraVermeId
        into :PersonelId, :PersonelAdi, :IplikAraVermeSebebiId, :IplikAraVermeSebebiAdi, :Tarih;
    end
  end
  else if (ABasladiBitti = 2) then
  begin
    Select IplikImalataAraVermeId from IplikImalataAraVerme
        where IplikImalataAraVerme.IplikMakinaId = :AIplikMakinaId and IplikImalataAraVerme.BittiIplikMakinaDurumuId is null
        into :IplikImalataAraVermeId;

    if (IplikImalataAraVermeId is not null) then
    begin
      BittiIplikMakinaDurumuId = gen_Id(IplikMakinaDurumu_id_gen, 1);

      insert into IplikMakinaDurumu(IplikMakinaDurumuId, IplikMakinaId, Tarih, ImalataAraVerildi) values(:BittiIplikMakinaDurumuId, :AIplikMakinaId, Current_TimeStamp(0), 2); /* ara bitirildi */

      Update IplikImalataAraVerme Set BittiIplikMakinaDurumuId = :BittiIplikMakinaDurumuId, ArayiBitirenPersonelId = :APersonelId where IplikImalataAraVerme.IplikImalataAraVermeID = :IplikImalataAraVermeId;

      Status = 1;
      mesaj = 'Kayit Yapıldı';
      Tarih = Current_TimeStamp(0);

    end
    else
    begin
      Status = 3;
      mesaj = 'imalata ara verilmemiş';
    end
  end
  else
  begin
    Status = 4;
    mesaj = 'Parametre Hatalı..Egemen Yazılıma Haber ver.. ';
  end
  suspend;
end
