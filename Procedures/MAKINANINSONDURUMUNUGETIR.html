CREATE PROCEDURE MAKINANINSONDURUMUNUGETIR(
    AMAKINAID VARCHAR(10))
RETURNS (
    MAKINADURUMUID INTEGER,
    TARIH TIMESTAMP,
    CALISIYOR SMALLINT,
    ARIZA SMALLINT,
    ARIZASAYISI SMALLINT,
    TEMIZLIK SMALLINT,
    ARAVERMETARIHI TIMESTAMP,
    ARAVERMESEBEBIID INTEGER,
    ARAVERMESEBEBIADI VARCHAR(50),
    IMALATAARAVERENPERSONELID INTEGER,
    IMALATAARAVERENPERSONELADI VARCHAR(40),
    PARCABEKLIYOR INTEGER,
    ARIZAIHBARPERSONELID INTEGER,
    ARIZAIHBARPERSONELADI VARCHAR(40),
    BAKIMCIPERSONELID INTEGER,
    BAKIMCIPERSONELADI VARCHAR(255),
    ARIZATIPIADI ADI20,
    ARIZAIHBARID INTEGER,
    ARIZAYERIID INTEGER,
    MAKINABILGIID INTEGER,
    MAKINECALISMIYOR EGEBOOLEAN,
    NEM DOUBLE PRECISION,
    ISI DOUBLE PRECISION,
    HIZ DOUBLE PRECISION,
    BACANEM DOUBLE PRECISION)
AS
declare variable ImalataAraVermeId integer;
declare variable ImalataAraVerildi SmallInt;
declare variable FBakimciPersonelAdi varchar(255);
begin
  Select MakinaDurumuId, MakinaBilgiID from Makina where Makina.MakinaID = :AMakinaId into :MakinaDurumuId, :MakinaBilgiID;

  Calisiyor             = 0;
  Ariza                 = 0;
  ArizaSayisi           = 0;
  Temizlik              = 0;
  MakineCalismiyor      = 0;
  AraVermeTarihi        = null;
  AraVermeSebebiId      = Null;
  AraVermeSebebiAdi     = NUll;
  ImalataAraVerenPersonelId = NUll;
  ImalataAraVerenPersonelAdi = NUll;
  ParcaBekliyor         = 0;
  ArizaIhbarPersonelId  = Null;
  ArizaIhbarPersonelAdi = null;
  ArizaIhbarId          = Null;
  ArizaYeriID           = Null;
  BakimciPersonelId     = Null;
  BakimciPersonelAdi    = Null;


  if (MakinaDurumuId is not null) then
  begin
    Select Tarih, Calisiyor, Ariza, Temizlik, ImalataAraVerildi from MakinaDurumu
      where MakinaDurumu.MakinaDurumuID = :MakinaDurumuId
      into :Tarih, :Calisiyor, :Ariza, :Temizlik, :ImalataAraVerildi;

    if (Ariza = 1 or Ariza = 3) then Ariza = 1;
    else Ariza = 0;

    if (ImalataAraVerildi = 1 or ImalataAraVerildi = 3) then ImalataAraVerildi = 1;
    else ImalataAraVerildi = 0;

    ParcaBekliyor = 0;

    if (Ariza = 1) then
    begin
      Select Max(ArizaIhbarId), Count(*) from ArizaIhbar where ArizaIhbar.MakinaId = :AMakinaId and ArizaIhbar.BittiMakinaDurumuId is null into :ArizaIhbarId, :ArizaSayisi;
      BakimciPersonelId = null;
      BakimciPersonelAdi = '';
      For Select ArizaIhbar.ArizaYeriID, ArizaIhbar.BaslatanPersonelId, Personel.Adi, ArizaIhbar.ParcaBekliyor, ArizaTipi.Adi, ArizaBakim.PersonelId, BakimPersonel.Adi, ArizaIhbar.MakineCalismiyor from ArizaIhbar
        left join Personel on Personel.PersonelId = ArizaIhbar.BaslatanPersonelId
        left join ArizaTipi on ArizaTipi.ArizaTipiId = ArizaIhbar.ArizaTipiId
        left join ArizaBakim on ArizaBakim.ArizaIhbarID = ArizaIhbar.ArizaIhbarID and ArizaBakim.GittiMakinaDurumuID is Null
        left join Personel BakimPersonel on BakimPersonel.PersonelId = ArizaBakim.PersonelId
        where ArizaIhbar.ArizaIhbarId = :ArizaIhbarId
        into :ArizaYeriID, :ArizaIhbarPersonelId, :ArizaIhbarPersonelAdi, :ParcaBekliyor, :ArizaTipiAdi, :BakimciPersonelId, :FBakimciPersonelAdi, :MakineCalismiyor do
        begin
          if(BakimciPersonelId is not null)  then
          begin
            if(BakimciPersonelAdi = '') then BakimciPersonelAdi =  substring(FBakimciPersonelAdi from 1 for 6) ;
            else  BakimciPersonelAdi = 'Birden Fazla Bakımcı';
          end
        end

      if (ParcaBekliyor is null) then ParcaBekliyor = 0;
    end

    if (ImalataAraVerildi = 1) then
    begin
      Select Max(ImalataAraVermeId) from ImalataAraVerme
        where ImalataAraVerme.MakinaId = :AMakinaId and ImalataAraVerme.BittiMakinaDurumuId is null into :ImalataAraVermeId;


      Select ImalataAraVerme.AraVermeSebebiId, AraVermeSebebi.Adi, MakinaDurumu.Tarih, ImalataAraVerme.AraVerdirenPersonelId,
             Personel.Adi from ImalataAraVerme
        left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
        left join Personel on Personel.PersonelId = ImalataAraVerme.AraVerdirenPersonelId
        left join MakinaDurumu on MakinaDurumu.MakinaDurumuId = ImalataAraVerme.BasladiMakinaDurumuId
        where ImalataAraVerme.ImalataAraVermeId = :ImalataAraVermeId
        into :AraVermeSebebiId, :AraVermeSebebiAdi, :AraVermeTarihi, :ImalataAraVerenPersonelId, :ImalataAraVerenPersonelAdi;
    end

  end

  if (MakinaBilgiID is not null) then
    Select Nem, Isi, Hiz, BacaNem from MakinaBilgi where MakinaBilgi.MakinaBilgiID = :MakinaBilgiID into :Nem, Isi, Hiz, :BacaNem;

  suspend;
end
