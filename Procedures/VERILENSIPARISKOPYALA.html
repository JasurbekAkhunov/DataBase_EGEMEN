CREATE PROCEDURE VERILENSIPARISKOPYALA(
    AVERILENSIPARISKAYITID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER)
AS
declare variable VerilenSiparisKayitId  integer;
declare variable AVerilenSiparisDetayId integer;
begin
  VerilenSiparisYil = f_Year(Current_date);

  VerilenSiparisId = Null;

  Select Id from VerilenSiparisNumaratoruAl(:VerilenSiparisYil) into :VerilenSiparisId;

  insert into VerilenSiparis(VerilenSiparisYil, VerilenSiparisId, VerilenSiparisTipiId, SatinAlmaBirimId, SatinAlmaKabulKosullariId, FirmaId, SiparisTarihi,
              SiparisGrupId, Notu, OdemeSekliId, TeslimSekliId, InsertKullaniciId, KullaniciId)
             Select :VerilenSiparisYil, :VerilenSiparisId, VerilenSiparisTipiId, SatinAlmaBirimId, SatinAlmaKabulKosullariId, FirmaId, current_date,
                     SiparisGrupId, Notu,  OdemeSekliId, TeslimSekliId, :AKullaniciId, :AKullaniciId from VerilenSiparis
               where VerilenSiparis.VerilenSiparisKayitID = :AVerilenSiparisKayitID;

  Select VerilenSiparisKayitId from VerilenSiparis where VerilenSiparis.VerilenSiparisYil = :VerilenSiparisYil and VerilenSiparis.VerilenSiparisID = :VerilenSiparisID into :VerilenSiparisKayitId;

  For Select VerilenSiparisDetayId from VerilenSiparisDetay where VerilenSiparisDetay.VerilenSiparisKayitID = :AVerilenSiparisKayitID into :AVerilenSiparisDetayId do
  begin
    Insert into VerilenSiparisDetay(VerilenSiparisKayitId, UrunId, HamId, RenkId, Adi, Miktar, UrunBirimId, BirimFiyat, KdvOrani, DovizId,
                                 VadeGun, TeslimTarihi, KullanilacakIplik1, KullanilacakIplik2, KullanilacakIplik3, IplikOrani1, IplikOrani2,
                                 IplikOrani3, IplikUzunlugu1, IplikUzunlugu2, IplikUzunlugu3, YikamaDereceId, KurutmaSekliId, Denye, LycraUreticiId,
                                 MakinaMarkasi, MakinaCinsi, SistemSayisi, Pus, Feine, MelanjKodu, TupAcikEnMayli, KenarKonstruksiyonTipiId,
                                 HamEn, HamGramaj, MamulEn, MamulGramaj, IstenenDonme, KontrolOrani, AmbalajSekliId, TalepNotu, SatinAlmaKullanimYeriId,
                                 Notu, ResimDosyaAdi1, ResimDosyayazdir1, ResimDosyaAdi2, ResimDosyayazdir2, TextDosyaAdi1, TextDosyayazdir1,
                                 TextDosyaAdi2, TextDosyayazdir2, ReferansFirmaId, ReferansSiparisNo, InsertKullaniciId, KullaniciId)
      Select :VerilenSiparisKayitId, UrunId, HamId, RenkId, Adi, Miktar, UrunBirimId, BirimFiyat, KdvOrani, DovizId,
                                 VadeGun, TeslimTarihi, KullanilacakIplik1, KullanilacakIplik2, KullanilacakIplik3, IplikOrani1, IplikOrani2,
                                 IplikOrani3, IplikUzunlugu1, IplikUzunlugu2, IplikUzunlugu3, YikamaDereceId, KurutmaSekliId, Denye, LycraUreticiId,
                                 MakinaMarkasi, MakinaCinsi, SistemSayisi, Pus, Feine, MelanjKodu, TupAcikEnMayli, KenarKonstruksiyonTipiId,
                                 HamEn, HamGramaj, MamulEn, MamulGramaj, IstenenDonme, KontrolOrani, AmbalajSekliId, TalepNotu, SatinAlmaKullanimYeriId,
                                 Notu, ResimDosyaAdi1, ResimDosyayazdir1, ResimDosyaAdi2, ResimDosyayazdir2, TextDosyaAdi1, TextDosyayazdir1,
                                 TextDosyaAdi2, TextDosyayazdir2, ReferansFirmaId, ReferansSiparisNo, :AKullaniciId, :AKullaniciId from VerilenSiparisDetay
      where VerilenSiparisDetay.VerilenSiparisDetayId = :AVerilenSiparisDetayId;

  end
  suspend;
end
