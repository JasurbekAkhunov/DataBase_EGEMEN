CREATE PROCEDURE KARAKTERDUZELT(
    ASTR VARCHAR(100),
    ADEGISECEK VARCHAR(1),
    AYENI VARCHAR(1))
RETURNS (
    STR VARCHAR(100))
AS
declare variable I integer;
begin
  Str = AStr;
  i = PosBul(Str, ADegisecek);

  While (I > 0) do
  begin
    Str = SubString(Str From 1 for  i - 1) || AYeni || substring(Str from  i + 1 for StrLength(AStr));
    i = PosBul(Str, ADegisecek);
  end
  suspend;
end
