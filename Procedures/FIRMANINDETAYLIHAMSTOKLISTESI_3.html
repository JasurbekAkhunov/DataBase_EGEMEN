          into :FaturaBirimFiyat, :FaturaDovizId;
      end

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    end

    if (FIrsaliyeNo is not null) then
    begin
      Str = Cast(FIrsaliyeNo as Varchar(18));

      if (Str <> '') then
      if (PosBul(IrsaliyeNo, Str) = 0) then
      if (Char_Length(IrsaliyeNo) + Char_Length(Str) < 250) then
      begin
        if (IrsaliyeNo <> '') then IrsaliyeNo = IrsaliyeNo ||  cast(', ' as varchar(2) character set UTF8);
        IrsaliyeNo = IrsaliyeNo || Str;

        Str = TimeStampToDateStr(FIrsaliyeTarihi);

        if (Str <> '') then
        if (Char_Length(IrsaliyeTarihi) + Char_Length(Str) < 250) then
        begin
          if (IrsaliyeTarihi <> '') then IrsaliyeTarihi = IrsaliyeTarihi ||  cast(', ' as varchar(2) character set UTF8);
          IrsaliyeTarihi = IrsaliyeTarihi || Str;
        end

        Str = TimeStampToDateStr(FSevkTarihi);

        if (Str <> '') then
        if (Char_Length(SevkTarihi) + Char_Length(Str) < 250) then
        begin
          if (SevkTarihi <> '') then SevkTarihi = SevkTarihi ||  cast(', ' as varchar(2) character set UTF8);
          SevkTarihi = SevkTarihi || Str;
        end

        if (FSevkAracId is not null) then Str = FSevkAracId;
        else Str = '';

        if (Str <> '') then
        if (PosBul(SevkAracId, Str) = 0) then
        if (Char_Length(SevkAracId) + Char_Length(Str) < 250) then
        begin
          if (SevkAracId <> '') then SevkAracId = SevkAracId ||  cast(', ' as varchar(2) character set UTF8);
          SevkAracId = SevkAracId || Str;
        end

        if (FFirmaKisaAdi is not null) then Str = FFirmaKisaAdi;
        else Str = '';

        if (Str <> '') then
        if (PosBul(IrsaliyeFirmaAdi, Str) = 0) then
        if (Char_Length(IrsaliyeFirmaAdi) + Char_Length(Str) < 100) then
        begin
          if (IrsaliyeFirmaAdi <> '') then IrsaliyeFirmaAdi = IrsaliyeFirmaAdi ||  cast(', ' as varchar(2) character set UTF8);
          IrsaliyeFirmaAdi = IrsaliyeFirmaAdi || Str;
        end

      end
    end

    if (FFaturaNo is not null) then
    begin
      Str = Format_Float('',FFaturaNo);
      if (Str <> '') then
      if (PosBul(FaturaNo, Str) = 0) then
      if (Char_Length(FaturaNo) + Char_Length(Str) < 250) then
      begin
        if (FaturaNo <> '') then FaturaNo = FaturaNo ||  cast(', ' as varchar(2) character set UTF8);
        FaturaNo = FaturaNo || Str;

        Str = TimeStampToDateStr(FFaturaTarihi);

        if (Str <> '') then
        if (Char_Length(FaturaTarihi) + Char_Length(Str) < 250) then
        begin
          if (FaturaTarihi <> '') then FaturaTarihi = FaturaTarihi ||  cast(', ' as varchar(2) character set UTF8);
          FaturaTarihi = FaturaTarihi || Str;
        end
      end
    end


    Select Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara), Sum(PartiStokTarti.Kilo) from PartiStokTarti
          where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.IadeBoyaIrsaliyeId is null
           into :TartiBrutKilo, :TartiNetKilo;

    Select Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara) from PartiStokTarti
          where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.BoyaIrsaliyeDetayId is not Null
           into :TartiSevkBrutKilo;

    Select Sum(PartiStokTarti.Kilo) from PartiStokTarti
          where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.BoyaIrsaliyeDetayId is not Null
           into :TartiSevkNetKilo;

    select max(PartiAsamalari.Sira) from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    where PartiAsamaStok.PartiKayitId = :PartiKayitId and PartiAsamaStok.PartiSinifId = :PartiSinifId and
         (PartiAsamalari.AsamaId <> :SonKontrolAsamaId and PartiAsamalari.AsamaId <> :IplikSonKontrolAsamaId ) and
          PartiAsamaStok.Giristarihi is not null
    into :MaxSira;

    if(MaxSira is not null) then
    begin
      select first 1 Asama.Adi, Asama.Asamaid from PartiAsamalari
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join Recete on Recete.Receteid = PartiAsamalari.Receteid
      left join Asama on (Asama.Asamaid = PartiAsamalari.AsamaId or Asama.Asamaid = Recete.UstAsamaId)
      where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.sira = :MaxSira and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
      group By Asama.Adi, Asama.Asamaid into :SonYapilanAsamaAdi, :SonYapilanAsamaID;
    end

    select First 1 PartiAsamaStokDetay.CikisMiktar, PartiAsamaStokDetay.CikisMetre, PartiAsamaStokDetay.CikisKilo from PartiAsamalari
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
    left join Asama on Asama.Asamaid = PartiAsamalari.AsamaId
    where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId and Asama.PartilemeMi = 1
    order by PartiAsamalari.Sira
    into :PartilemeCikisMiktari, :PartilemeCikisMetre, :PartilemeCikisKilo;
  end

  if (LastPartiSinifId <> -1) then
  begin
    if (netKilo <> 0 and netKilo is not null and Kilo <> 0 and Kilo is not null and Kilo <> netKilo) then Fire = (Kilo - netKilo) / Kilo *  100;
                                                                                 else Fire = 0;
    suspend;
  end
