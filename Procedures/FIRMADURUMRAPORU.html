CREATE PROCEDURE FIRMADURUMRAPORU
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BIRIMID VARCHAR(2),
    SIPARISMIKTAR NUMERIC(12,3),
    PARTILENENMIKTAR NUMERIC(12,3),
    DEPOCIKISMIKTAR NUMERIC(12,3),
    ISLETMEMIKTAR NUMERIC(12,3),
    SEVKEHAZIRMIKTAR NUMERIC(12,3),
    SEVKEHAZIRNETMIKTAR NUMERIC(12,3),
    BRUTSEVKEDILENMIKTAR NUMERIC(12,3),
    NETSEVKEDILENMIKTAR NUMERIC(12,3),
    HAMDEPOMIKTAR NUMERIC(12,3))
AS
begin
/* has örme tüm stokları kg olarak baktıgı icin metreyi dikkate almadık */
  for Select FirmaID, FirmaAdi, BirimId, Sum(SiparisMiktar), Sum(PartilenenMiktar), Sum(DepoCikisMiktar),
             Sum(SevkeHazirMiktar), Sum(SevkeHazirNetMiktar), Sum(BrutSevkEdilenMiktar),
             Sum(NetSevkEdilenMiktar), Sum(HamDepoMiktar)
             from FirmaDurumRaporuHazirla
    Group by FirmaID, FirmaAdi, BirimId
    into :FirmaId, :FirmaAdi, :BirimId, :SiparisMiktar, :PartilenenMiktar, :DepoCikisMiktar, :SevkeHazirMiktar,
         :SevkeHazirNetMiktar, :BrutSevkEdilenMiktar, :NetSevkEdilenMiktar, :HamDepoMiktar do
  begin
    IsletmeMiktar = DepoCikisMiktar - BrutSevkEdilenMiktar - SevkeHazirMiktar;

    if (SiparisMiktar <> 0 or HamDepoMiktar <> 0 or IsletmeMiktar <> 0) then suspend;
  end

end
