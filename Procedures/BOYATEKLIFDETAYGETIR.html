CREATE PROCEDURE BOYATEKLIFDETAYGETIR(
    ABOYATEKLIFKAYITID INTEGER)
RETURNS (
    BOYATEKLIFDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    SIPARISPROSESASAMALARI VARCHAR(500),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETECALISMATARIHI TIMESTAMP,
    LABRECETEATISNO VARCHAR(20),
    GLOBALTEKLIF INTEGER,
    GLOBALTEKLIFSTR VARCHAR(6),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI VARCHAR(20),
    HESAPLANANFIYAT DOUBLE PRECISION,
    TEKLIFEDILENFIYAT DOUBLE PRECISION,
    RECETEMALIYETI DOUBLE PRECISION,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    VADEGUN INTEGER,
    HAMSTOKBEDELI DOUBLE PRECISION,
    TEKLIFSURESI INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    KALITEID VARCHAR(20),
    KOMISYON DOUBLE PRECISION,
    NAVLUN DOUBLE PRECISION,
    FASONBIRIMFIYAT DOUBLE PRECISION,
    FASONFIREORANI DOUBLE PRECISION,
    ORMEBIRIMMALIYET DOUBLE PRECISION,
    IPLIKFIREORANI DOUBLE PRECISION,
    TEKLIFFIREORANI DOUBLE PRECISION,
    SATISKARORANI DOUBLE PRECISION,
    MODELID MODELKODUSIZE,
    URUNID URUNKODUSIZE,
    MODELADI ADI40,
    URUNADI ADI80,
    RENKADI ADI40,
    BOYALIEN INTEGER,
    BOYALIGRAMAJ INTEGER,
    TALEPEDILENFIYAT DOUBLE PRECISION,
    TALEPEDILENKABULEDILDI EGEBOOLEAN,
    SIPARISPROSESFIYATI DOUBLE PRECISION,
    SIPARISPROSESDOVIZID DOVIZSIZE,
    FIRMAID FIRMAKODUSIZE,
    COMPOSITION VARCHAR(255),
    KALITEADI VARCHAR(255))
AS
declare variable FAsamaAdi    Varchar(40);
declare variable BirimMaliyet Double Precision;
declare variable DovizId      DovizSize;
begin
  for Select BoyaTeklifDetay.BoyaTeklifDetayId, LabRecete.LabReceteKodu, BoyaTeklifDetay.GlobalTeklif, BoyaTeklifDetay.RenkDerinligiId, RenkDerinligi.Adi, BoyaTeklifDetay.HesaplananFiyat,
             BoyaTeklifDetay.TeklifEdilenFiyat, BoyaTeklifDetay.ReceteMaliyeti, BoyaTeklifDetay.Kilo, BoyaTeklifDetay.Metre, BoyaTeklifDetay.Adet,
             Coalesce(BoyaTeklifDetay.HamId, ModelKumas.HamId), Coalesce(Ham.Adi, ModelHam.Adi), BoyaTeklifDetay.SiparisProsesId, SiparisProses.Adi, BoyaTeklifDetay.VadeGun, BoyaTeklifDetay.HamStokBedeli,
             BoyaTeklifDetay.TeklifSuresi, BoyaTeklifDetay.Notu,
             LabRecete.Adi, LabReceteAtis.AtisNoStr, LabReceteCalisma.CalismaTarihi,
             Ik.Adi, BoyaTeklifDetay.Inserttarihi, Sk.Adi, BoyaTeklifDetay.DegisimTarihi, BoyaTeklifDetay.EnIstenen, BoyaTeklifDetay.GramajIstenen, Boyateklifdetay.KaliteId,
             BoyaTeklifDetay.Komisyon, BoyaTeklifDetay.Navlun, BoyaTeklifDetay.FasonBirimFiyat, BoyaTeklifDetay.FasonFireOrani, BoyaTeklifDetay.OrmeBirimMaliyet, BoyaTeklifDetay.IplikFireOrani, BoyaTeklifDetay.TeklifFireOrani,
             BoyaTeklifDetay.SatisKarOrani, Model.ModelID,  BoyaTeklifDetay.UrunID, Model.Adi, Urun.Adi, ModelKumas.RenkAdi,
             ModelKumas.BoyaliGramaj, ModelKumas.BoyaliEn, BoyaTeklifDetay.TalepEdilenFiyat, BoyaTeklifDetay.TalepEdilenKabulEdildi, BoyaTeklif.FirmaId,
             SiparisProses.BirimMaliyet, SiparisProses.DovizId, Kalite.Composition, Kalite.KaliteAdi from BoyaTeklifDetay
    left join LabReceteAtis On LabReceteAtis.LabReceteAtisId = BoyaTeklifDetay.LabReceteAtisId
    left join LabRecete On LabRecete.LabReceteId = LabReceteAtis.LabReceteId
    left join LabReceteCalisma On LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
    left join RenkDerinligi On RenkDerinligi.RenkDerinligiId = BoyaTeklifDetay.RenkDerinligiId
    left join Ham On Ham.HamId = BoyaTeklifDetay.HamId
    left join SiparisProses On SiparisProses.SiparisProsesId = BoyaTeklifDetay.SiparisProsesId
    Left Join Kullanici IK on IK.KullaniciId = BoyaTeklifDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = BoyaTeklifDetay.KullaniciId
    Left Join Kalite on Kalite.KaliteId = BoyaTeklifDetay.KaliteId
    Left Join Model on Model.ModelKayitID = BoyaTeklifDetay.ModelKayitID
    Left Join Urun on Urun.UrunId = BoyaTeklifDetay.UrunId
    Left Join ModelKumas on ModelKumas.ModelKumasID = Urun.ModelKumasID
    left join Ham ModelHam On ModelHam.HamId = ModelKumas.HamId
    left join BoyaTeklif on BoyaTeklif.BoyaTeklifkayitId = BoyaTeklifDetay.BoyaTeklifKayitId
    where BoyaTeklifDetay.BoyaTeklifKayitId = :ABoyaTeklifKayitId
   into :BoyaTeklifDetayId, :LabReceteKodu, :GlobalTeklif, :RenkDerinligiId, :RenkDerinligiAdi, :HesaplananFiyat, :TeklifEdilenFiyat, :ReceteMaliyeti, :Kilo, :Metre,
        :Adet, :HamId, :HamAdi, :SiparisProsesId, :SiparisProsesAdi, :VadeGun, :HamStokBedeli, :TeklifSuresi, :Notu,
        :LabReceteAdi,  :LabReceteAtisNo, :LabReceteCalismaTarihi,
        :InsertKullaniciAdi, :Inserttarihi, :KullaniciAdi, :DegisimTarihi, :EnIstenen, :GramajIstenen, :KaliteId, :Komisyon, :Navlun, :FasonBirimFiyat, :FasonFireOrani, :OrmeBirimMaliyet, :IplikFireOrani, :TeklifFireOrani,
        :SatisKarOrani, :ModelID, :UrunID, :ModelAdi, :UrunAdi, :RenkAdi, :BoyaliGramaj, :BoyaliEn, :TalepEdilenFiyat, :TalepEdilenKabulEdildi, :FirmaId, :BirimMaliyet, :DovizId, :Composition, :KaliteAdi do
  begin
    if (GlobalTeklif = 1) then GlobalTeklifStr = 'Global'; else GlobalTeklifStr = '';

    SiparisProsesAsamalari = '';
    for Select  Asama.Adi from SiparisProsesAsamalari
          left join Asama on Asama.Asamaid = SiparisProsesAsamalari.Asamaid
          where SiparisProsesAsamalari.SiparisProsesId = :SiparisProsesId Order BY Sira
    into :FAsamaAdi do
    begin
      if (FAsamaAdi is not null) then
      if (Char_Length(SiparisProsesAsamalari) + Char_Length(FAsamaAdi) < 461) then
      begin
       if (SiparisProsesAsamalari <> '' ) then SiparisProsesAsamalari = SiparisProsesAsamalari || '+';
       SiparisProsesAsamalari = SiparisProsesAsamalari || FAsamaAdi;
      end
    end

    SiparisProsesFiyati = Null;
    SiparisProsesDovizId = Null;

    Select FirmaSiparisProses.BirimFiyat, FirmaSiparisProses.DovizId from FirmaSiparisProses
    where FirmaSiparisProses.FirmaId = :FirmaId and FirmaSiparisProses.SiparisProsesId = :SiparisProsesId
    into :SiparisProsesFiyati, :SiparisProsesDovizId;

    if (SiparisProsesFiyati is null) then
    begin
      SiparisProsesFiyati = BirimMaliyet;
      SiparisProsesDovizId = DovizId;
    end
    suspend;
  end
end
