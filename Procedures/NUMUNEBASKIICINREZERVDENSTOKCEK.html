CREATE PROCEDURE NUMUNEBASKIICINREZERVDENSTOKCEK(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    AMIKTAR NUMERIC(12,3),
    AKULLANICIID INTEGER)
AS
declare variable KS   integer;
declare variable BoyaSiparisKayitId   integer;
declare variable BoyaSiparisDetayId   integer;
declare variable BoyaSiparisBaskiId   integer;
declare variable BoyaSiparisRezervId   integer;
declare variable RezervKalanMiktar    Numeric(12,3);
declare variable RafKalanMiktar    Numeric(12,3);
declare variable Miktar               Numeric(12,3);
declare variable PartiSinifId         integer;
declare variable ImalatTipiId         integer;
declare variable IslemTipiID          integer;
declare variable HamTipiId            integer;
declare variable FarkMiktar           Numeric(12,3);
declare variable FKilo                Numeric(12,3);
declare variable FMetre               Numeric(12,3);
declare variable FMiktar              Numeric(12,3);

declare variable RafDetayId           integer;
declare variable IsletmeStokRafId     integer;
declare variable ProxyId   varchar(8);
declare variable FProsesId  integer;
declare variable FBoyaSiparisSarjId integer;
declare variable DokumaKumasiKilodanTakipEt integer;
begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  Select BoyaSiparisKayitId, ImalatTipiId, IslemTipiId from Parti where Parti.PartiKayitID = :APartiKayitID
     into :BoyaSiparisKayitId, :ImalatTipiId, :IslemTipiId;

  Select Ham.HamtipiId, PartiStok.BoyaSiparisDetayId, PartiStok.ProxyId, PartiStok.BoyaSiparisBaskiId, PartiStok.ProsesId, PartiStok.BoyaSiparisSarjId from PartiStok
    left join Ham on Ham.HamId = PartiStok.HamId
    where PartiStok.PartiSinifId = :APartiSinifId
    into :HamTipiId, :BoyaSiparisDetayId, :ProxyId, :BoyaSiparisBaskiId, :FProsesId, :FBoyaSiparisSarjId;

  if (ImalatTipiId <> 1 or IslemTipiId <> 5) then exception NumuneBaskiPartisiDegil;

  Miktar = AMiktar;

  FarkMiktar = Miktar;
  FKilo = 0;
  FMetre = 0;

  PartiSinifId = Gen_ID(PartiSinif_Id_Gen, 1);
  Insert Into PartiStok(PartiKayitId, PartiSinifId, ProsesId, BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisBaskiId, BoyaSiparisSarjId, LinkIslemi, InsertKullaniciId, KullaniciId)
              values(:APartiKayitId, :PartiSinifId, :FProsesId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :FBoyaSiparisSarjId, 1, :AKullaniciId, :AKullaniciId);


  For Select BoyaSiparisRezervId, KalanMiktar from SiparisRezervListesiGetir(:BoyaSiparisKayitId) where BoyaSiparisDetayID = :BoyaSiparisDetayID
    into :BoyaSiparisRezervId, :RezervKalanMiktar do
  begin
    For Select RafDetayId, KalanMiktar from SiparisRezervRafListesiGetir(:BoyaSiparisKayitId) where BoyaSiparisRezervId = :BoyaSiparisRezervId
       into :RafDetayId, :RafKalanMiktar do
    begin
      if (FarkMiktar > 0) then
      begin
        if (RezervKalanMiktar > RafKalanMiktar) then FMiktar = RafKalanMiktar;
        else FMiktar = RezervKalanMiktar;

        if (FMiktar > FarkMiktar) then FMiktar = FarkMiktar;

        RezervKalanMiktar = RezervKalanMiktar - FMiktar;
        FarkMiktar = FarkMiktar - FMiktar;

        KS = 0;
        Select Count(*) from PartiStokDetay
          where PartiStokDetay.PartiSinifId = :PartiSinifId and IntKarsilastir(PartiStokDetay.RafDetayId, :RafDetayId) = 1 into :KS;

        if (HamTipiID <> 2 or DokumaKumasiKilodanTakipEt = 1) then FKilo = FMiktar;
        else FMetre = FMiktar;

        if (KS > 0) then
          update PartiStokDetay Set Kilo = Kilo + :FKilo, Metre = Metre + :FMetre, LinkIslemi = 1
            where PartiStokDetay.PartiSinifId = :PartiSinifId and
                  IntKarsilastir(PartiStokDetay.RafDetayId, :RafDetayId) = 1;
        else
          insert into PartiStokDetay(PartiKayitId, PartiSinifId, RafDetayId, Kilo, Metre, BoyaSiparisRezervId, InsertKullaniciId, KullaniciId, LinkIslemi)
                      values(:APartiKayitId, :PartiSinifId, :RafDetayId, :FKilo, :FMetre, :BoyaSiparisRezervId, :AKullaniciId, :AKullaniciId, 1);
      end
    end
  end
  if (FarkMiktar <> 0) then exception RezervIcinYeterliKiloYok;


  insert into PartiAsamaStok(PartiKayitId, PartiSinifId, PartiAsamalariId, AsamayaDahil, AsamaId)
      Select :APartiKayitId, :PartiSinifId, PartiAsamaStok.PartiAsamalariId, PartiAsamaStok.AsamayaDahil, PartiAsamaStok.AsamaId from PartiAsamaStok
      where PartiAsamaStok.PartiSinifId = :APartiSinifId;

  Update PartiStok set ProxyID = Null where PartiStok.partiSinifID = :APartiSinifID and partiStokIptalKayitId is null;
  Update PartiStok set ProxyID = :ProxyId, LinkIslemi = 0 where PartiStok.partiSinifID = :PartiSinifID;

  execute Procedure PSSablonRecetesiKopyala(:APartiKayitId, :APartiSinifId, :PartiSinifID, :AKullaniciId);
end
