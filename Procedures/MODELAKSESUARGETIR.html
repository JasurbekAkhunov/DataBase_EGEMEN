CREATE PROCEDURE MODELAKSESUARGETIR(
    AMODELKAYITID INTEGER)
RETURNS (
    MODELAKSESUARID INTEGER,
    MODELKAYITID INTEGER,
    MODELID MODELKODUSIZE,
    MODELKUMASID INTEGER,
    URUNID URUNKODUSIZE,
    URUNADI ADI80,
    MIKTAR NUMERIC(12,4),
    FIREORANI DOUBLE PRECISION,
    RENKID INTEGER,
    RENKADI ADI50,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATDOVIZID DOVIZSIZE,
    NOTU MEMO,
    KULLANIMTIPI VARCHAR(30),
    RESIMDOSYASI VARCHAR(255),
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40),
    BIRIMI ADI10)
AS
begin

  For Select ModelAksesuar.ModelAksesuarID, ModelAksesuar.ModelKayitID, Model.ModelId, ModelAksesuar.ModelKumasID, ModelAksesuar.UrunId, Urun.Adi, ModelAksesuar.FireOrani, Urun.RenkId, Renk.Adi, ModelAksesuar.BirimFiyat,
             ModelAksesuar.BirimFiyatDovizId, ModelAksesuar.InsertKullaniciId, ModelAksesuar.InsertTarihi, ModelAksesuar.KullaniciId, ModelAksesuar.DegisimTarihi,
             IK.Adi, SK.Adi, ModelAksesuar.Notu, ModelAksesuar.Miktar, ModelKullanimTipiListesi.Adi, Urun.ResimDosyasi, Coalesce(Birim.Adi, UrunDefaultBirimi.Adi) From ModelAksesuar
    left join Model on Model.ModelKayitID = ModelAksesuar.ModelKayitID
    left join Urun on Urun.UrunID = ModelAksesuar.UrunId
    left join Renk on Renk.RenkId = Urun.RenkId
    left join Kullanici IK on IK.KullaniciId  = ModelAksesuar.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = ModelAksesuar.KullaniciId
    left join ModelKullanimTipiListesi on ModelKullanimTipiListesi.KullanimTipiId = ModelAksesuar.KullanimTipi
    left join Birim on Birim.BirimId = ModelAksesuar.BirimId
    left join UrunBirim on UrunBirim.Urunid = ModelAksesuar.UrunId and UrunBirim.DefaultBirim = 1
    left join Birim UrunDefaultBirimi on UrunDefaultBirimi.BirimId = UrunBirim.BirimId
    Where ModelAksesuar.ModelKayitID = :AModelKayitID
    into :ModelAksesuarID, :ModelKayitID, :ModelId, :ModelKumasID, :UrunId, :UrunAdi, :FireOrani, :RenkId, :RenkAdi, :BirimFiyat, :BirimFiyatDovizId,
         :InsertKullaniciId, :InsertTarihi, :KullaniciId, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi, :Notu, :Miktar, :KullanimTipi, :ResimDosyasi, :Birimi do
  begin
    Suspend;
  end
end
