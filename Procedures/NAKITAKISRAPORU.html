CREATE PROCEDURE NAKITAKISRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABASLANGICHAFTA INTEGER,
    ABITISHAFTA INTEGER,
    ABASLANGICFIRMAGRUP1ID VARCHAR(4),
    ABITISFIRMAGRUP1ID VARCHAR(4),
    ABASLANGICFIRMAGRUP2ID VARCHAR(4),
    ABITISFIRMAGRUP2ID VARCHAR(4),
    ABASLANGICFIRMAGRUP3ID VARCHAR(4),
    ABITISFIRMAGRUP3ID VARCHAR(4),
    ABASLANGICFIRMAGRUP4ID VARCHAR(4),
    ABITISFIRMAGRUP4ID VARCHAR(4),
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    AFIRMAGRUP4ARALIKLI INTEGER,
    ABASLANGICCSGRUP1ID VARCHAR(4),
    ABITISCSGRUP1ID VARCHAR(4),
    ABASLANGICCSGRUP2ID VARCHAR(4),
    ABITISCSGRUP2ID VARCHAR(4),
    ABASLANGICCSGRUP3ID VARCHAR(4),
    ABITISCSGRUP3ID VARCHAR(4),
    ACSGRUP1ARALIKLI INTEGER,
    ACSGRUP2ARALIKLI INTEGER,
    ACSGRUP3ARALIKLI INTEGER,
    ARAPORDOVIZID VARCHAR(5))
RETURNS (
    HAFTA INTEGER,
    BOLUM INTEGER,
    ACIKLAMA1 VARCHAR(100),
    VADETARIHI TIMESTAMP,
    DOVIZID VARCHAR(5),
    MEBLAG DOUBLE PRECISION,
    DOVIZKURU DOUBLE PRECISION,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION)
AS
declare variable Aciklamax              varchar(40);
declare variable ARaporTarihi           Date;
declare variable FBorcAlacak            varchar(1);
declare variable FMeblag                Double Precision;
declare variable AlinanCeklerToplami    Double Precision;
declare variable VerilenCeklerToplami   Double Precision;
Declare variable HaftaBaslangicTarihi   Date;
Declare variable HaftaBitisTarihi       Date;
declare variable FHafta                 integer;
declare variable HaftaToplami           Double Precision;
declare variable HaftaToplamlari        Double Precision;
Declare Variable GenelToplam            Double Precision;
begin
  ARaportarihi = current_TimeStamp(0);
  HaftaBaslangicTarihi = ABaslangicTarihi;
  HaftaBitisTarihi = HaftaBaslangicTarihi + 6;
  Hafta = ABaslangicHafta;
  FHafta = ABaslangicHafta;
  Borc = Null;
  Alacak = Null;
  Meblag = Null;
  VadeTarihi = Null;
  DovizId = Null;
  Aciklama1 = EgemenIntegerToStr(Hafta) || '.Hafta (' || TimeStampToDateStr(HaftaBaslangicTarihi) || '-'  || TimeStampToDateStr(HaftaBitisTarihi) || ')';
  HaftaToplami = 0;
  HaftaToplamlari = 0;
  Bolum = 0;
  suspend;
  For Select KasaId, KasaAdi, DovizId, Borc, Alacak from NakitAkisRaporuKasaDok
    into :Aciklama1, :Aciklamax, :DovizId, :Borc, :Alacak do
  begin
    if (Borc > 0) then Meblag = Borc;
    else Meblag = Alacak;

    if (DovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
    else DovizKuru = 1;

    Borc = Borc * DovizKuru;
    Alacak = Alacak * DovizKuru;
    HaftaToplami = HaftaToplami + Alacak - Borc;
    Aciklama1 = '(' || Aciklama1 || ')-';
    Aciklama1 = Aciklama1 || Aciklamax;
    Bolum = 1;  /* Kasa dökümü */
    if (Meblag <> 0) then suspend;
  end

  For Select BankaHesapId, BankaHesapAdi, DovizId, Borc, Alacak from NakitAkisRaporuBankaDok
    into :Aciklama1, :Aciklamax, :DovizId, :Borc, :Alacak do
  begin
    if (Borc > 0) then Meblag = Borc;
    else Meblag = Alacak;

    if (DovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
    else DovizKuru = 1;

    Aciklama1 = '(' || Aciklama1 || ')-';
    Aciklama1 = Aciklama1 || Aciklamax;

    Borc = Borc * DovizKuru;
    Alacak = Alacak * DovizKuru;
    HaftaToplami = HaftaToplami + Alacak - Borc;
    Bolum = 2;  /* Bankalar */
    if (Meblag <> 0) then suspend;
  end
  Hafta = FHafta;

  While (FHafta <= ABitisHafta) do
  begin
    DovizId = Null;
    DovizKuru = Null;
    vadeTarihi = Null;
    Meblag = Null;
/* alinan/verilen Cekler toplami */
    for Select AlinanCeklerToplami, VerilenCeklerToplami from NakitAkisRaporuCek(:HaftaBaslangicTarihi, :HaftaBitisTarihi, :ARaportarihi, :ARaporDovizId)
      into :AlinanCeklerToplami, :VerilenCeklerToplami do
    begin
      Aciklama1 = 'Çekler Toplamı';
      Borc = VerilenCeklerToplami;
      Alacak = AlinanCeklerToplami;
      HaftaToplami = HaftaToplami + Alacak - Borc;
      Bolum = 3;  /* alınan-verilen çekler toplamı */
      suspend;
    end

    Aciklama1 = 'Alınan Senetler';
    VadeTarihi = Null;
    DovizId = '';
    DovizKuru = Null;
    Meblag = Null;
    Borc  = Null;
    Alacak = Null;
    Bolum = 4;  /* alınan senetler */
    Suspend;

    For select Aciklama, VadeTarihi, KalanMeblag, DovizId, DovizKalanMeblag from VadeTarihineGoreCSRaporu('','',:HaftaBaslangicTarihi, :HaftaBitisTarihi,'A','S',0,0,0,0,NULL,0,NULL,'','','','','','',0,0,0,0,Null, NULL, NULL, :ARaporDovizId, null)
    into :Aciklama1, :VadeTarihi, :Meblag, :DovizId, :Alacak do
    begin
      HaftaToplami = HaftaToplami + Alacak;
      Bolum = 4;  /* alınan senetler */
      if (DovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
      else DovizKuru = 1;
      suspend;
    end

    Aciklama1 = 'Ödenecek Senetler';
    VadeTarihi = Null;
    DovizId = '';
    DovizKuru = Null;
    Meblag = Null;
    Borc  = Null;
    Alacak = Null;
    Bolum = 5;  /* verilen senetler */
    Suspend;

    For select Aciklama, VadeTarihi, KalanMeblag, DovizId, DovizKalanMeblag from VadeTarihineGoreCSRaporu('','',:HaftaBaslangicTarihi, :HaftaBitisTarihi,'V','S',0,NULL,NULL,0,NULL,NULL,NULL,'','','','','','',0,0,0,0,null, NULL, NULL, :ARaporDovizId, null)
    into :Aciklama1, :VadeTarihi, :Meblag, :DovizId, :Borc do
    begin
      HaftaToplami = HaftaToplami - Borc;
      if (DovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
      else DovizKuru = 1;
      Bolum = 5;  /* verilen senetler */
      suspend;
    end

    Borc = Null;
    Alacak = Null;
    Meblag = Null;
    VadeTarihi = Null;
    DovizId = Null;
    DovizKuru = Null;
    Aciklama1 = 'Hafta Toplamı';
    Alacak = HaftaToplami;
    Bolum = 6;  /* Hafta Toplami */
    suspend;


    FHafta = FHafta + 1;
    Hafta = FHafta;
    HaftaBaslangicTarihi = HaftaBaslangicTarihi + 7;
    Borc = Null;
    Alacak = Null;
    Meblag = Null;
    VadeTarihi = Null;
    DovizId = Null;
    HaftaToplamlari = HaftaToplamlari + HaftaToplami;
    HaftaToplami = 0;
    if (FHafta <= ABitisHafta) then
    begin
      HaftaBitisTarihi = HaftaBaslangicTarihi + 6;
      Aciklama1 = EgemenIntegerToStr(FHafta) || '.Hafta (' || TimeStampToDateStr(HaftaBaslangicTarihi) || '-'  || TimeStampToDateStr(HaftaBitisTarihi) || ')';
      Bolum = 0;  /* Hafta Baslangici */
      suspend;
    end
  end

  Hafta = Null;
  Aciklama1 = EgemenIntegerToStr(ABaslangicHafta) || '-' || EgemenIntegerToStr(ABitisHafta) || '.Haftalar Toplamı =========>';
  Alacak = HaftaToplamlari;
  Bolum = 7; /* Tüm Haftalar Toplamı */
  suspend;
  HaftaBitistarihi = HaftaBitistarihi + 1;
  GenelToplam = HaftaToplamlari;
/* Kalan Cekleri Dok */
  for Select AlinanCeklerToplami, VerilenCeklerToplami from NakitAkisRaporuCek(:HaftaBitisTarihi, Null, :ARaportarihi, :ARaporDovizId)
    into :AlinanCeklerToplami, :VerilenCeklerToplami do
  begin
    Aciklama1 = 'Sonraki Çekler Toplamı';
    Borc = VerilenCeklerToplami;
    Alacak = AlinanCeklerToplami;
    GenelToplam = GenelToplam + Alacak - Borc;
    Bolum = 8;
    suspend;
  end
/* Kalan Alınan Senetleri Dok */
  Borc = Null;
  Alacak = Null;
  Meblag = Null;
  VadeTarihi = Null;
  DovizId = Null;
  DovizKuru = Null;

  For select Aciklama, VadeTarihi, KalanMeblag, DovizId, DovizKalanMeblag from VadeTarihineGoreCSRaporu('','',:HaftaBitisTarihi, Null,'A','S',0,0,0,0,NULL,0,NULL, :ABaslangicCsGrup1Id, :ABitisCsGrup1Id, :ABaslangicCsGrup2Id, :ABitisCsGrup2Id, :ABaslangicCsGrup3Id, :ABitisCsGrup3Id,0, :ACSGrup1Aralikli, :ACSGrup2Aralikli, :ACSGrup3Aralikli,null, NULL, NULL, :ARaporDovizId, null)
  into :Aciklama1, :VadeTarihi, :Meblag, :DovizId, :Alacak do
  begin
    GenelToplam = GenelToplam + Alacak;
    if (DovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
    else DovizKuru = 1;
    Bolum = 10;
    suspend;
  end
/* Kalan Verilen Senetleri Dok */
  Borc = Null;
  Alacak = Null;
  Meblag = Null;
  VadeTarihi = Null;
  DovizId = Null;
  DovizKuru = Null;
  For select Aciklama, VadeTarihi, KalanMeblag, DovizId, DovizKalanMeblag from VadeTarihineGoreCSRaporu('','', :HaftaBitisTarihi, Null, 'V','S',0,NULL,NULL,0,NULL,NULL,NULL, :ABaslangicCsGrup1Id, :ABitisCsGrup1Id, :ABaslangicCsGrup2Id, :ABitisCsGrup2Id, :ABaslangicCsGrup3Id, :ABitisCsGrup3Id,0, :ACSGrup1Aralikli, :ACSGrup2Aralikli, :ACSGrup3Aralikli, null, NULL, NULL, :ARaporDovizId, Null )
  into :Aciklama1, :VadeTarihi, :Meblag, :DovizId, :Borc do
  begin
    GenelToplam = GenelToplam - Borc;
    if (DovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
    else DovizKuru = 1;
    Bolum = 11;
    suspend;
  end
/* Cari Bakiye  Dok */
  Borc = 0;
  Alacak = 0;
  Meblag = Null;
  VadeTarihi = Null;
  DovizId = Null;
  DovizKuru = Null;
  Aciklama1 = 'Cari Bakiyeleri Toplamı';


  For Select BakiyeBa, Sum(DovizBakiye) from BakiyeListesi('','',:ABaslangicFirmaGrup1Id, :ABitisFirmaGrup1Id, :ABaslangicFirmaGrup2Id, :ABitisFirmaGrup2Id, :ABaslangicFirmaGrup3Id, :ABitisFirmaGrup3Id, :ABaslangicFirmaGrup4Id, :ABitisFirmaGrup4Id, NULL, 0, :AFirmaGrup1Aralikli, :AFirmaGrup2Aralikli, :AFirmaGrup3Aralikli, :AFirmaGrup4Aralikli, NULL,0,:ARaporDovizId, :ARaporTarihi, Null, Null)
      where DovizBakiye > 0
      group by BakiyeBA into :FBorcAlacak, :FMeblag do
  begin
    if (FBorcAlacak = 'B') then Alacak = FMeblag;
    else Borc = FMeblag;
  end
  Bolum = 9;
  suspend;
  GenelToplam = GenelToplam + Alacak - Borc;

  VadeTarihi = Null;
  For Select Aciklama, meblag, Borc, Alacak, DovizId from NakitAkisEkiGetir
  into :Aciklama1, :Meblag, :Borc, :Alacak, :DovizId do
  begin
    if (DovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
    else DovizKuru = 1;

    Borc = Borc * DovizKuru;
    Alacak = Alacak * DovizKuru;

    if (Borc is not null) then GenelToplam = GenelToplam - Borc;
    if (Alacak is not null) then GenelToplam = GenelToplam + Alacak;
    Bolum = 12;
    suspend;
  end

/* Genel Toplami Yaz */
  Borc = Null;
  Meblag = Null;
  VadeTarihi = Null;
  DovizId = Null;
  DovizKuru = Null;
  Aciklama1 = 'GENEL TOPLAM ';
  Alacak = GenelToplam;
  Bolum = 13;
  suspend;
end
