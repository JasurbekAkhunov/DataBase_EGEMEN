CREATE PROCEDURE COZGUHASILURETIMGIRISFISKAYDET(
    AISEMRIKAYITID INTEGER,
    AURETIMISLEMTIPIID INTEGER,
    ADOVIZID VARCHAR(5),
    ASINIF VARCHAR(2),
    AKULLANICIID INTEGER)
RETURNS (
    ORMEFISID INTEGER)
AS
Declare Variable  FisSeriNo   varchar(5);
Declare Variable  FisNo       integer;
Declare Variable  shtUretim   integer;
Declare Variable  uitCozgu    integer;
declare variable  uitIndigo    integer;
Declare Variable  uitHasil    integer;
Declare Variable  KayitSayisi integer;
begin
  Select shtUretim from OrmeStokHareketTipiGetir into :shtUretim;
  select uitCozgu, uitHasil, uitIndigo from UretimIslemTipiGetir into :uitCozgu, :uitHasil, :uitIndigo;

  if (AUretimIslemTipiId <> uitHasil and AUretimIslemTipiId <> uitCozgu and AUretimIslemTipiId <> uitIndigo) then exception KayitHatasi 'Üretim İşlem Tipi Hatalı';

  if (AUretimIslemTipiId = uitIndigo) then
  begin
    Select count(*) from OrmeFis where IndigoIsEmriKayitID = :AIsEmriKayitId into :KayitSayisi;
    if (KayitSayisi > 0) then exception KayitHatasi 'Bu Girişin Fişi Oluşturulmuş..';
  end  else if (AUretimIslemTipiId = uitHasil) then
  begin
    Select count(*) from OrmeFis where HasilIsEmriKayitID = :AIsEmriKayitId into :KayitSayisi;
    if (KayitSayisi > 0) then exception KayitHatasi 'Bu Girişin Fişi Oluşturulmuş..';
  end
  else if (AUretimIslemTipiId = uitCozgu) then
  begin
    Select count(*) from OrmeFis where CozguIsEmriKayitID = :AIsEmriKayitId into :KayitSayisi;
    if (KayitSayisi > 0) then exception KayitHatasi 'Bu Girişin Fişi Oluşturulmuş..';
  end

  Select DokumaUretimSeriNo from Kullanici where KullaniciId = :AKullaniciId into :FisSeriNo;
  if (FisSeriNo is null) then FisSeriNo = '';

  Select FisNo from FisNumaratorAL(:FisSeriNo) into :FisNo;

  OrmeFisId = gen_id(OrmeFis_Id_Gen, 1);

  if (AUretimIslemTipiId = uitIndigo) then
  begin
    insert into OrmeFis(OrmeFisId, HareketTipi, UretimIslemTipiId, IndigoIsEmriKayitID, SeriNo, FisNo, FisTarihi, DovizId, DokumaHareketi, InsertKullaniciId, KullaniciId, Sinif)
             values(:OrmeFisId, :shtUretim, :AUretimIslemTipiId, :AIsEmriKayitId, :FisSeriNo, :FisNo, Current_Date, :ADovizID, 1, :AKullaniciId, :AKullaniciId, :ASinif);

    Update IndigoIsEmri set BaslamaTarihi = Current_Date where  IndigoIsEmriKayitID = :AIsEmriKayitId;
  end else if (AUretimIslemTipiId = uitHasil) then
  begin
    insert into OrmeFis(OrmeFisId, HareketTipi, UretimIslemTipiId, HasilIsEmriKayitID, SeriNo, FisNo, FisTarihi, DovizId, DokumaHareketi, InsertKullaniciId, KullaniciId, Sinif)
             values(:OrmeFisId, :shtUretim, :AUretimIslemTipiId, :AIsEmriKayitId, :FisSeriNo, :FisNo, Current_Date, :ADovizID, 1, :AKullaniciId, :AKullaniciId, :ASinif);

    Update HasilIsEmri set BaslamaTarihi = Current_Date where  HasilIsEmriKayitID = :AIsEmriKayitId;
  end
  else
  begin
    insert into OrmeFis(OrmeFisId, HareketTipi, UretimIslemTipiId, CozguIsEmriKayitID, SeriNo, FisNo, FisTarihi, DovizId, DokumaHareketi, InsertKullaniciId, KullaniciId, Sinif)
             values(:OrmeFisId, :shtUretim, :AUretimIslemTipiId, :AIsEmriKayitId, :FisSeriNo, :FisNo, Current_Date, :ADovizID, 1, :AKullaniciId, :AKullaniciId, :ASinif);
    Update CozguIsEmri set BaslamaTarihi = Current_Date where  CozguIsEmriKayitID = :AIsEmriKayitId;
  end

  Suspend;
end
