CREATE PROCEDURE DESENISTEKLERIGETIR(
    ADESENID VARCHAR(20))
RETURNS (
    DESENISTEKLERIID INTEGER,
    SIRA INTEGER,
    ISTEKTARIHI TIMESTAMP,
    ISTEKYAPANKULLANICIID INTEGER,
    ISTEKYAPANKULLANICIADI VARCHAR(40),
    TERMINTARIHI TIMESTAMP,
    DESENISTEKTIPIID VARCHAR(5),
    DESENISTEKTIPIADI VARCHAR(50),
    ACIKLAMA VARCHAR(100),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ISTEKBEKLEMEDE INTEGER,
    BASLATANKULLANICIADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENKULLANICIADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    PLANLANANDESENISTASYONADI VARCHAR(30),
    PLANLANANHAFTANO INTEGER,
    SONUCDOSYAADI1 VARCHAR(255),
    SONUCDOSYAADI2 VARCHAR(255),
    SONUCDOSYAADI3 VARCHAR(255),
    SONUCDOSYAADI4 VARCHAR(255),
    SONUCDOSYAADI5 VARCHAR(255),
    DUZELTMEISTEMIDOSYAADI1 VARCHAR(255),
    DUZELTMEISTEMIDOSYAADI2 VARCHAR(255),
    STOKTANDUSULDU INTEGER,
    STOKTANDUSULDUSTR VARCHAR(4),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTIPI INTEGER,
    ONAYTIPISTR VARCHAR(30))
AS
begin
  For Select DesenIstekleri.DesenIstekleriId, DesenIstekleri.Sira, DesenIstekleri.IstekTarihi, DesenIstekleri.IstekYapanKullaniciId, IYK.Adi,
             DesenIstekleri.TerminTarihi, DesenIstekleri.DesenIstekTipiId, DesenIstekTipi.Adi, DesenIstekleri.Aciklama, DesenIstekleri.Notu, DesenIstekleri.Beklemede,
             BasK.Adi, DesenIstekleri.BaslamaTarihi, BitK.Adi, DesenIstekleri.BitisTarihi, DesenIstekleri.UniteId, Unite.Adi,
             DesenIstekleri.SonucDosyaAdi1, DesenIstekleri.SonucDosyaAdi2, DesenIstekleri.SonucDosyaAdi3, DesenIstekleri.SonucDosyaAdi4,
             DesenIstekleri.SonucDosyaAdi5, DesenIstekleri.DuzeltmeIstemiDosyaAdi1, DesenIstekleri.DuzeltmeIstemiDosyaAdi2,
             DesenIstasyon.Adi, Hafta.HaftaNo, DesenIstekleri.OnayDurumu, DesenIstekleri.OnayTarihi, OK.Adi, DesenIstekleri.OnayTipi,
             DesenIstekleri.StoktanDusuldu from DesenIstekleri
  Left Join Kullanici BasK on BasK.KullaniciId = DesenIstekleri.BaslatanKullaniciId
  Left Join Kullanici BitK on BitK.KullaniciId = DesenIstekleri.BitirenKullaniciId
  Left Join Unite on Unite.UniteId = DesenIstekleri.UniteId
  Left Join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiId
  left join Kullanici IYK on IYK.KullaniciId = DesenIstekleri.IstekYapanKullaniciId
  Left Join DesenPlanlama on DesenPlanlama.DesenIstekleriId = DesenIstekleri.DesenIstekleriId
  Left Join DesenIstasyon on DesenIstasyon.DesenIstasyonId = DesenPlanlama.DesenIstasyonId
  Left Join Hafta on Hafta.HaftaKayitId = DesenPlanlama.HaftaKayitId
  left join Kullanici OK on OK.KullaniciId = DesenIstekleri.OnaylayanKullaniciId
  where DesenIstekleri.DesenId = :ADesenId
  order by DesenIstekleri.DesenId, DesenIstekleri.Sira
  into :DesenIstekleriId, :Sira, :IstekTarihi, :IstekYapanKullaniciId, :IstekYapanKullaniciAdi, :TerminTarihi, :DesenIstekTipiId, :DesenIstekTipiAdi,
       :Aciklama, :Notu, :IstekBeklemede, :BaslatanKullaniciAdi, :BaslamaTarihi, :BitirenKullaniciAdi, :BitisTarihi, :UniteId, :UniteAdi,
       :SonucDosyaAdi1, :SonucDosyaAdi2, :SonucDosyaAdi3, :SonucDosyaAdi4, :SonucDosyaAdi5, :DuzeltmeIstemiDosyaAdi1, :DuzeltmeIstemiDosyaAdi2,
       :PlanlananDesenIstasyonAdi, :PlanlananHaftaNo, :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciAdi, :OnayTipi, :StoktanDusuldu do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    if (OnayTipi = 1) then OnayTipiStr = 'Başlamadan Onay Ver';
    else OnayTipiStr = 'Bitince Onay Ver';

    if (StoktanDusuldu = 1) then StoktanDusulduStr = 'Evet';
    else StoktanDusulduStr = '';
    suspend;
  end
end
