AS
declare variable TamirGirisKayitId      integer;
declare variable IplikSatisIadeId       integer;
declare variable shtSatis               integer;
declare variable shtUretimeCikis        integer;
declare variable shtUretimCikisBoya     integer;
declare variable ShtSarimaCikis         integer;
declare variable shtBukumeCikis         integer;
declare variable shtDepoTransfer        integer;
declare variable OrmeSiparisDetayID     integer;
declare variable FasonMu                integer;

begin
  Select shtSatis, shtUretimeCikis, shtUretimCikisBoya, shtSarimaCikis, shtBukumeCikis, ShtDepoTransfer from IplikStokHareketTipiGetir
         into :shtSatis, :shtUretimeCikis, :shtUretimCikisBoya, :shtSarimaCikis, :shtBukumeCikis, :ShtDepoTransfer;

  For Select IplikIrsaliye.IplikIrsaliyeId, IplikIrsaliye.IplikFaturaId, IplikIrsaliyeDetay.IplikIrsaliyeDetayId,
             IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.SevkSeferNo, IplikIrsaliye.IrsaliyeGrupId, IplikIrsaliye.FirmaId,
             IrsaliyeFirma.Adi, IplikIrsaliye.AracId, Sofor.Adi, IplikIrsaliyeDetay.HamId, Ham.Adi, Ham.HamTipiId,
             coalesce(IplikIrsaliyeDetay.RenkId,HarmanRenk.RenkID), coalesce(Renk.Adi, HarmanRenk.Adi), IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu, IplikIrsaliyeDetay.Lot,
             IplikIrsaliyeDetay.KapAdedi, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.NetMiktar, IplikIrsaliyeDetay.BobinSayisi, IplikIrsaliyeDetay.AmbalajId, Ambalaj.Adi,

             Coalesce(BoyaSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.MusteriSiparisNo),
             Coalesce(BoyaSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.ReferansSiparisNo),
             BoyaSiparisDetay.MusteriKaliteNo, OrmeSiparisDetay.ArtikelNo, IplikLot.Kapali,

             IplikIrsaliyeDetay.Tevkifat, OrmeSiparisDetay.SatisFiyati, OrmeSiparisDetay.DovizId, OrmeSiparisDetay.VadeGun, IplikIrsaliyeDetay.IplikDepoID, IplikIrsaliye.IplikDepoID,

  left join Sofor on Sofor.SoforId = IplikIrsaliye.SoforId
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeID = IplikIrsaliye.IplikIrsaliyeID

  left join IplikLot on IplikLot.Lot = IplikIrsaliyeDetay.Lot and IplikLot.HamID = IplikIrsaliyeDetay.HamId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
  left join IplikFatura on IplikFatura.IplikFaturaId = IplikIrsaliyeDetay.IplikFaturaId
  left join Harman on Harman.HarmanKayitId = IplikLot.HarmanKayitId
  left join Renk HarmanRenk on HarmanRenk.RenkId = Harman.RenkId
  left join IplikIsletme on IplikIsletme.IplikIsletmeId = Harman.IplikIsletmeId
  left join Harman IrsaliyeDetayHarman on IrsaliyeDetayHarman.HarmanKayitId = IplikIrsaliyeDetay.HarmanKayitId
  left join SevkFirma On SevkFirma.SevkFirmaId = IplikIrsaliye.SevkFirmaID
  left join Arac on Arac.AracId =IplikIrsaliye.AracId
  where Cast(IplikIrsaliye.FiiliSevkTarihi as Date) >= :ABaslangicTarihi and Cast(IplikIrsaliye.FiiliSevkTarihi as Date) <= :ABitisTarihi and
        (
         (:ASatisGoster = 1 and IplikIrsaliye.HareketTipi = :shtSatis) or
         (:AUretimeCikisGoster = 1 and IplikIrsaliye.HareketTipi = :shtUretimeCikis) or
         (:ASarimaCikisGoster = 1 and IplikIrsaliye.HareketTipi = :shtSarimaCikis) or
         (:ABukumeCikisGoster = 1 and IplikIrsaliye.HareketTipi = :shtBukumeCikis) or
         (:ABoyahaneyeCikislariGoster = 1 and IplikIrsaliye.HareketTipi = :shtUretimCikisBoya) or
         (:ADepoTransferGoster = 1 and IplikIrsaliye.HareketTipi = :shtDepoTransfer)
        ) and
        IplikIrsaliyeDetay.IplikIrsaliyeDetayId is not null and
        (IplikIrsaliye.FirmaID = :AFirmaId or :AFirmaId is null) and
        (IplikIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
        (IplikIrsaliye.SevkSeferNo = :ASevkSeferNo or :ASevkSeferNo is null) and
        (IplikIrsaliye.AracId = :AAracId or :AAracId is null) and
        (IplikIrsaliye.SoforID = :ASoforID or :ASoforID is null) and
        (IplikIrsaliyeDetay.HamID = :AHamId or :AHamId is null) and
        (IplikIrsaliyeDetay.IplikTalepKayitId = :AIplikTalepKayitId or :AIplikTalepKayitId is null) and
        (IplikIrsaliyeDetay.Lot = :ALot or :ALot is null) and
        (Ham.KalinlikId = :AKalinlikId or :AKalinlikId is null) and
        (Ham.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is null) and
        (BoyaSiparis.FirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
        (IplikLot.Tamir = :ATamirDurumu or :ATamirDurumu is null) and
        (IplikIrsaliyeDetay.IplikDepoID = :AIplikDepoID or :AIplikDepoID is null) and
        (Harman.AnaLot = :AAnaLot or :AAnaLot is Null) and
        (Harman.RenkId = :ARenkId or :ARenkId is Null)

  into :IplikIrsaliyeId, :IplikFaturaId, :IplikIrsaliyeDetayId, :IrsaliyeNo, :IrsaliyeTarihi, :SevkSeferNo, :IrsaliyeGrupId, :FirmaId,
       :FirmaAdi, :AracId, :SoforAdi, :HamId, :HamAdi, :HamTipiId, :RenkId, :RenkAdi, :MelanjKodu, :FlamKodu,  :Lot,
       :KapAdedi, :Miktar, :NetMiktar, :BobinSayisi,  :AmbalajId, :AmbalajAdi, :KonikRengiId, :KonikRengiAdi, :KonikCinsiId, :KonikCinsiAdi,
       :TamirGirisKayitId, :IplikTalepYil, :IplikTalepId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisGrupId,
       :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrupAdi, :FaturaNo, :FaturaTarihi, :MusteriSiparisNo, :ReferansSiparisNo, :MusteriKaliteNo, :ArtikelNo, :LotKapali,
       :HarmanYil, :HarmanId, :HarmanAdi, :LotElyafDegerlendirme,
       :FaturaMiktar, :Kdv, :Tevkifat, :SatisFiyati, :DovizId, :VadeGun, :IplikDepoID, :GeldigiIplikDepoID, :HareketTipi,
       :OrmeSiparisDetayID, :SevkFirmaID, :TeslimTarihi, :RevizeTeslimTarihi, :HamSatisTransferKodu, :SatisTransferKodu, :AlisTransferKodu,
       :TalepBirimMaliyet, :LotBirimMaliyet, :LotIscilikMaliyeti, :LotDigerMaliyet, :HarmanKayitID,
       :HarmanElyafBirimMaliyet, :HarmanIscilikBirimMaliyeti, :HarmanDigerBirimMaliyet, :SevkFirmaAdi, :HarmanIplikIsletmeId, :HarmanIplikIsletmeAdi, :VergiHesapNo, :NetBirimFiyat,
       :IrsaliyeSeriNo, :FaturaSeriNo, :TalepTarihi, :FasonMu, :AracKapasite do
  begin
    if (FasonMu = 1) then
      FasonStr = 'Fason';
    else
      FasonStr = 'Satış';

    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';
    else
      BoyaSiparisIdStr = '';

    if (HarmanYil is not null) then
      HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') '  || EgemenIntegerToStr(HarmanId)  || ' ';
    else
      HarmanIdStr = '';

    if (IplikTalepYil is not null) then
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';
    else
      IplikTalepIdStr = '';

    if (TamirGirisKayitId is not null) then
    begin
      Tamir = 1;
      TamirStr = 'Evet';
      select IplikSatisIadeId from TamirGiris where TamirGirisKayitId = :TamirGirisKayitId into :IplikSatisIadeId;
      if (IplikSatisIadeId is not Null) then
      begin
        select EskiPartiYil, EskiPartiNo from IplikSatisIade where IplikSatisIadeId = :IplikSatisIadeId
          into :EskiPartiYil, :EskiPartiNo;

        if (EskiPartiNo is not null) then
        begin
          EskiPartiNoStr = '(' || EgemenIntegerToStr(EskiPartiYil)  || ') ' || EgemenIntegerToStr(EskiPartiNo) || ' ';
          select Parti.PartiKayitId, Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
            BoyaSiparis.FirmaId, Firma.Adi from Parti
            left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
            left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
            where Parti.PartiYil = :EskiPartiYil and Parti.PartiId = :EskiPartiNo
            into :EskiPartiKayitId, :EskiSiparisKayitId, :EskiSiparisYil, :EskiSiparisNo,
            :EskiSiparisFirmaId, :EskiSiparisFirmaAdi;

          if (EskiSiparisNo is not Null) then
            EskiSiparisNoStr = '(' || EgemenIntegerToStr(EskiSiparisYil)  || ') ' || EgemenIntegerToStr(EskiSiparisNo) || ' ';
        end
        else EskiPartiNoStr = '';
      end
    end
    else
    begin
      Tamir = 0;
      TamirStr = '';
    end

    if (LotKapali = 1) then LotKapaliStr = 'Evet';
    else LotKapaliStr = '';

    if (HareketTipi = :shtSatis) then HareketTipiStr = 'Satış';
    else if (HareketTipi = :shtUretimeCikis) then HareketTipiStr = 'Üretime Çıkış';
    else if (HareketTipi = :ShtDepoTransfer) then HareketTipiStr = 'Depo Transfer';
    else if (HareketTipi = :shtUretimCikisBoya) then HareketTipiStr = 'Üretim Çıkış Boya';
    else if (HareketTipi = :ShtSarimaCikis) then HareketTipiStr = 'Sarima Çıkış';
    else if (HareketTipi = :shtBukumeCikis) then HareketTipiStr = 'Büküme Çıkış';
    else HareketTipiStr = 'Hatalı!';

    IplikIlkSevkTarihi = Null;
    IplikSonSevkTarihi = Null;
    Select Min(IplikIrsaliyeDetay.Tarih),MAx(IplikIrsaliyeDetay.tarih) from IplikTalepDetay
         left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayID= IplikTalepDetay.IplikTalepDetayID
         where IplikTalepDetay.OrmeSiparisDetayID=:OrmeSiparisDetayID and IplikIrsaliyeDetay.IplikIrsaliyeDetayID is not null and
               IplikIrsaliyeDetay.HareketTipi = :shtSatis
    into :IplikIlkSevkTarihi, :IplikSonSevkTarihi;


    suspend;
  end
