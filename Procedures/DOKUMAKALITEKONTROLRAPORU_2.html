CREATE PROCEDURE DOKUMAKALITEKONTROLRAPORU_2(
    ADOKUMAMAKINAID VARCHAR(10),
    ADOKUMAPLANLAMAID INTEGER,
    ASAKAT INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKONTROLDURUMU INTEGER,
    AHAMID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    AKUMASDURUMU INTEGER)
RETURNS (
    DOKUMAKALITEKONTROLID INTEGER,
    DOKUMAPLANLAMARECETEID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    LOT VARCHAR(30),
    RENKID INTEGER,
    RENKADI ADI50,
    ORMEFISID INTEGER,
    FISNO INTEGER,
    HESAPLANANMIKTAR BOYAMIKTARSIZE,
    CIKILANMIKTAR BOYAMIKTARSIZE,
    TAMAMICIKILDI EGEBOOLEAN,
    ACIKLAMA VARCHAR(500),
    KUMASHAMID INTEGER,
    TARIH TIMESTAMP,
    ADET INTEGER)
AS
declare variable uitBoya integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable FirmaKodumuz varchar(30);
begin
  Select Sistem.FirmaKodumuz  from Sistem Where sistemid = 1 into :FirmaKodumuz;
  select UretimIslemTipiGetir.uitboya from UretimIslemTipiGetir into :uitBoya;
  select OrmeStokHareketTipiGetir.shtFasonUretimeCikis, OrmeStokHareketTipiGetir.shtUretimeCikis from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis, :shtUretimeCikis;

  For Select DokumaKaliteKontrol.DokumaKaliteKontrolId, DokumaUretimeOtomatikCikis.DokumaPlanlamaReceteId, DokumaUretimeOtomatikCikis.OrmeFisId, DokumaUretimeOtomatikCikis.HesaplananMiktar,
  DokumaUretimeOtomatikCikis.CikilanMiktar, DokumaUretimeOtomatikCikis.TamamiCikildi, DokumaUretimeOtomatikCikis.Aciklama, DokumaPlanlamaRecete.HamId, Ham.Adi,
  DokumaPlanlamaRecete.Lot, DokumaPlanlamaRecete.RenkId, Renk.Adi, OrmeFis.FisNo, HamKumas.HamId, DokumaKaliteKontrol.Tarih, DokumaKaliteKontrol.Adet from DokumaKaliteKontrol
    left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaKaliteKontrol.DokumaPlanlamaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = DokumaKaliteKontrol.OrmeUCikisIrsaliyeId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    left join DokumaUretimeOtomatikCikis on DokumaUretimeOtomatikCikis.DokumaKaliteKontrolId = DokumaKaliteKontrol.DokumaKaliteKontrolId
    left join DokumaPlanlamaRecete on DokumaPlanlamaRecete.DokumaPlanlamaReceteId = DokumaUretimeOtomatikCikis.DokumaPlanlamaReceteId
    left join Ham on Ham.HamId = DokumaPlanlamaRecete.HamId
    left join Ham HamKumas on HamKumas.HamId = OrmeSiparisDetay.HamId
    left join Renk on Renk.RenkId = DokumaPlanlamaRecete.RenkId
    left join OrmeFis on OrmeFis.OrmeFisId = DokumaUretimeOtomatikCikis.OrmeFisId
    where  DokumaKaliteKontrol.Tarih >= :ABaslangicTarihi and DokumaKaliteKontrol.Tarih <= :ABitisTarihi and
          (DokumaKaliteKontrol.DokumaPlanlamaId = :ADokumaPlanlamaId or :ADokumaPlanlamaId is null) and
          (DokumaKaliteKontrol.Sakat = :ASakat or :ASakat is null) and
          (DokumaPlanlama.DokumaMakinaId = :ADokumaMakinaId or :ADokumaMakinaId is null) and
          (DokumaKaliteKontrol.KontrolBitti = :AKontrolDurumu or :AKontrolDurumu is null) and
          (OrmeSiparisDetay.HamId = :AHamId or :AHamId is null) and
          (BoyaSiparis.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          ((:AKumasDurumu is null) or
           (:AKumasDurumu = 0 and DokumaKaliteKontrol.OrmeUCikisIrsaliyeId is not null and OrmeIrsaliye.FirmaId = :FirmaKodumuz and
             OrmeIrsaliye.HareketTipi in (:shtFasonUretimeCikis, :shtUretimeCikis) and OrmeIrsaliye.UretimIslemtipiId = :uitBoya ) or
           (:AKumasDurumu = 1 and DokumaKaliteKontrol.OrmeUCikisIrsaliyeId is not null and OrmeIrsaliye.FirmaId <> :FirmaKodumuz and
             OrmeIrsaliye.HareketTipi in (:shtFasonUretimeCikis, :shtUretimeCikis) and OrmeIrsaliye.UretimIslemtipiId = :uitBoya ) or
           (:AKumasDurumu = 2 and DokumaKaliteKontrol.OrmeUCikisIrsaliyeId is null)
          )
    into :DokumaKaliteKontrolId, :DokumaPlanlamaReceteId, :OrmeFisId, :HesaplananMiktar, :CikilanMiktar, :TamamiCikildi, :Aciklama, :HamId, :HamAdi, :Lot, :RenkId, :RenkAdi, :FisNo, :KumasHamId,
    :Tarih, :Adet do
  begin
    suspend;
  end
end
