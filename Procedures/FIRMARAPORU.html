CREATE PROCEDURE FIRMARAPORU(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    AKAPALI INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    FIRMAGRUP1 VARCHAR(4),
    FIRMAGRUP2 VARCHAR(4),
    FIRMAGRUP3 VARCHAR(4),
    FIRMAGRUP1ADI VARCHAR(30),
    FIRMAGRUP2ADI VARCHAR(30),
    FIRMAGRUP3ADI VARCHAR(30),
    TEMSILCIID INTEGER,
    TEMSILCIADI VARCHAR(40),
    ADRESI1 VARCHAR(40),
    ADRESI2 VARCHAR(40),
    ADRESI3 VARCHAR(40),
    SEMT VARCHAR(20),
    SEHIR VARCHAR(20),
    VERGIDAIRESI VARCHAR(20),
    VERGIHESAPNO VARCHAR(20),
    TELEFON1 VARCHAR(15),
    TELEFONTIPIADI1 VARCHAR(10),
    TELEFON2 VARCHAR(15),
    TELEFONTIPIADI2 VARCHAR(10),
    TELEFON3 VARCHAR(15),
    TELEFONTIPIADI3 VARCHAR(10),
    TELEFON4 VARCHAR(15),
    TELEFONTIPIADI4 VARCHAR(10),
    TELEFON5 VARCHAR(15),
    TELEFONTIPIADI5 VARCHAR(10),
    EMAIL VARCHAR(80),
    WEB VARCHAR(80),
    ISKONTOORANI DOUBLE PRECISION,
    ACIKHESAPRISKI DOUBLE PRECISION,
    CEKSENETRISKI DOUBLE PRECISION,
    SIPARISRISKI DOUBLE PRECISION,
    VADEGUN INTEGER,
    DOVIZID VARCHAR(5),
    TAHSILATHAFTA1 INTEGER,
    TAHSILATHAFTA2 INTEGER,
    TAHSILATHAFTA3 INTEGER,
    TAHSILATHAFTA4 INTEGER,
    TAHSILATGUN1 INTEGER,
    TAHSILATGUN2 INTEGER,
    TAHSILATGUN3 INTEGER,
    TAHSILATGUN4 INTEGER,
    TAHSILATGUN5 INTEGER,
    TAHSILATGUN6 INTEGER,
    TAHSILATGUNU VARCHAR(20),
    SATICI INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TRANSFERKODU VARCHAR(55),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    ULKE VARCHAR(20),
    KURULUSTARIHI DATE,
    FIRMAGRUP4 VARCHAR(4),
    FIRMAGRUP4ADI VARCHAR(30))
AS
declare variable Okey integer;
declare variable PersonelCariYetkisiVar integer;
declare variable FirmaAdi2yiGorebilir   integer;
begin
  select PersonelCariYetkisiVar, FirmaAdi2yiGorebilir from Kullanici where KullaniciID = :AKullaniciId into :PersonelCariYetkisiVar, :FirmaAdi2yiGorebilir;

  for Select Firma.FirmaId, Firma.Adi, Firma.Adi2,Firma.FirmaGrup1, Firma.FirmaGrup2, Firma.FirmaGrup3, FG1.Adi, FG2.Adi, Fg3.Adi, Firma.TemsilciId, T.Adi,
    Firma.Adresi1, Firma.Adresi2, Firma.Adresi3, Firma.Semt, Firma.Sehir, Firma.VergiDairesi, Firma.VergiHesapNo,
    Firma.Telefon1, Firma.Telefon2, Firma.Telefon3, Firma.Telefon4, Firma.Telefon5, TT1.Adi, TT2.Adi, TT3.Adi, TT4.Adi, TT5.Adi,
    Firma.Email, Firma.Web, Firma.IskontoOrani, Firma.AcikHesapRiski, Firma.CekSenetRiski, Firma.SiparisRiski, Firma.VadeGun,
    Firma.DovizId, Firma.TahsilatHafta1, Firma.TahsilatHafta2, Firma.TahsilatHafta3, Firma.TahsilatHafta4, Firma.TahsilatGun1, Firma.TahsilatGun2, Firma.TahsilatGun3,
    Firma.TahsilatGun4, Firma.TahsilatGun5, Firma.TahsilatGun6, Firma.TahsilatGunu, Firma.Notu,
    Firma.TransferKodu, IK.Adi, Firma.InsertTarihi, SK.Adi, Firma.DegisimTarihi, Firma.Satici, Firma.Ulke, Firma.KurulusTarihi, Firma.FirmaGrup4, FG4.Adi  from Firma
  left join FirmaGrup FG1 on Fg1.FirmaGrupId = Firma.FirmaGrup1
  left join FirmaGrup FG2 on Fg2.FirmaGrupId = Firma.FirmaGrup2
  left join FirmaGrup FG3 on Fg3.FirmaGrupId = Firma.FirmaGrup3
  left join Kullanici T on T.KullaniciId = Firma.TemsilciId
  left join TelefonTipi TT1 on TT1.TelefonTipiId = Firma.TelefonTipi1
  left join TelefonTipi TT2 on TT2.TelefonTipiId = Firma.TelefonTipi2
  left join TelefonTipi TT3 on TT3.TelefonTipiId = Firma.TelefonTipi3
  left join TelefonTipi TT4 on TT4.TelefonTipiId = Firma.TelefonTipi4
  left join TelefonTipi TT5 on TT5.TelefonTipiId = Firma.TelefonTipi5
  left join Kullanici IK on IK.KullaniciId = Firma.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId = Firma.KullaniciId
  left join FirmaGrup FG4 on Fg4.FirmaGrupId = Firma.FirmaGrup4
  where ((:AFirmaAralikli = 1 and Firma.FirmaId >= :ABaslangicFirmaId and (Firma.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0))  and
        ((:AFirmaGrup1Aralikli = 1 and Firma.FirmaGrup1 >= :ABaslangicGrup1Id  and (Firma.FirmaGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and Firma.FirmaGrup2 >= :ABaslangicGrup2Id and (Firma.FirmaGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and Firma.FirmaGrup3 >= :ABaslangicGrup3Id and (Firma.FirmaGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
        (Firma.Personelmi = 0 or (Firma.Personelmi = 1 and :PersonelCariYetkisiVar = 1)) and
        (Firma.Kapali = :AKapali or :AKapali is null)
  into :FirmaId, :FirmaAdi, :FirmaAdi2, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :FirmaGrup1Adi, :FirmaGrup2Adi, :FirmaGrup3Adi,
       :TemsilciId, :TemsilciAdi, :Adresi1, :Adresi2, :Adresi3, :Semt, :Sehir, :VergiDairesi, :VergiHesapNo,
       :Telefon1, :Telefon2, :Telefon3, :Telefon4, :Telefon5, :TelefonTipiAdi1, :TelefonTipiAdi2, :TelefonTipiAdi3, :TelefonTipiAdi4, :TelefonTipiAdi5,
       :Email, :Web, :IskontoOrani, :AcikHesapRiski, :CekSenetRiski, :SiparisRiski, :VadeGun,
       :DovizId, :TahsilatHafta1, :TahsilatHafta2, :TahsilatHafta3, :TahsilatHafta4, :TahsilatGun1, :TahsilatGun2, :TahsilatGun3,
       :TahsilatGun4, :TahsilatGun5, :TahsilatGun6, :TahsilatGunu, :Notu,
       :TransferKodu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :Satici, :Ulke, :KurulusTarihi, :FirmaGrup4, :FirmaGrup4Adi  do
  begin
    if (FirmaAdi2yiGorebilir = 0) then FirmaAdi2 = Null;

    Okey = 1;
    if (AFirmaAralikli= 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (okey = 1) then suspend;
  end
end
