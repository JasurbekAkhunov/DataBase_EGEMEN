declare variable FIslemTipiId  integer;

declare variable FMiktar        Double Precision;
declare variable FCikisTarihi TimeStamp;
declare variable itTamir        integer;
declare variable FFaturaKesilecek integer;
declare variable FUcretliIcTamir  integer;
declare variable FGunlukKapasite Double Precision;
declare variable VardiyaSayisi      integer;
declare variable Vardiya           integer;
declare variable T1                Double Precision;
declare variable KatSayi  Double Precision;
declare variable EgemenFirmaID integer;
declare variable Metremi SmallInt;

begin
/*Bu Raporda Field isimlerinde Kilo geçiyor ama biz Miktar bilgisini select ediyoruz. Yani Metre olarakta çıktı verebilir.
  Eski yapılan dizaynlar bozulmasın diye alan isimlerini ...Miktar olarak değiştirmedik*/

  Select itTamir from IslemTipiGetir into :itTamir;

  Select VardiyaSayisi, EgemenFirmaID from sistem where sistemid = 1 into :VardiyaSayisi, :EgemenFirmaID;

  if (EgemenFirmaID = 64 or EgemenFirmaID = 5 or EgemenFirmaID = 37 or EgemenFirmaID = 77) then Metremi = 1;
  else Metremi = 0;

  if (VardiyaSayisi = 0 or VardiyaSayisi is null) then VardiyaSayisi = 3;

  For select PartiAsamaStokDetay.MakinaId, Makina.Adi, Makina.GunlukKapasite, Makina.MakinaGrubu, MakinaGrup.Adi,Makina.RaporGrupId,
             Sum(iif(:Metremi = 0, PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.Metre)), Sum(iif(:Metremi = 0, PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.Metre) * IslemGrubu.KatSayi) from PartiAsamaStokDetay
  left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
  left join MakinaGrup on MakinaGrup.MakinaGrupId = Makina.MakinaGrubu
  left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join PartiAsamaStok On PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
  left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
  where PartiAsamaStokDetay.Cikistarihi >= :AAylikBaslangic and PartiAsamaStokDetay.Cikistarihi <= :AAylikBitis and
        (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
        (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null) and
        Asama.GunlukUretimRaporunaDahil = 1
  group by PartiAsamaStokDetay.MakinaId, Makina.Adi, Makina.GunlukKapasite, Makina.MakinaGrubu, MakinaGrup.Adi,Makina.RaporGrupId
  into :MakinaId, :MakinaAdi, :GunlukKapasite, :MakinaGrubu, :MakinaGrupAdi, :MakinaRaporGrupId, :AylikUretimToplami, :AylikEsdegerUretimToplami do
  begin
    UretimToplami = 0;
    AylikIcTamirUretimToplami = 0;
    AylikUcretliIcTamirToplami  = 0;
    AylikUcretsizIcTamirToplami = 0;
    AylikUcretliDisTamirToplami = 0;
    AylikUcretsizDisTamirToplami = 0;
    AylikIcTamirEsdegerUretim = 0;

    IcTamirSiyah         = 0;
    IcTamirNaylonBoyama  = 0;
    IcTamirPolyesterBoyama  = 0;

    UcretliIcTamirSekerKasar    = 0;

    UcretliIcTamirCiftBoyama    = 0;
    UcretliIcTamirSiyah         = 0;
    UcretliIcTamirNaylonBoyama  = 0;
    UcretliIcTamirPolyesterBoyama  = 0;
    UcretliIcTamirDiger         = 0;

    UcretsizIcTamirBeyaz         = 0;
    UcretsizIcTamirRenkli        = 0;
    UcretsizIcTamirYikama        = 0;
    UcretsizIcTamirSekerKasar    = 0;
    UcretsizIcTamirKuruBolum     = 0;
    UcretsizIcTamirCiftBoyama    = 0;
    UcretsizIcTamirSiyah         = 0;
    UcretsizIcTamirNaylonBoyama  = 0;
    UcretsizIcTamirPolyesterBoyama  = 0;
    UcretsizIcTamirDiger         = 0;

    DisTamirBeyaz           = 0;
    DisTamirRenkli          = 0;
    DisTamirYikama          = 0;
    DisTamirSekerKasar      = 0;
    DisTamirKuruBolum       = 0;
    DisTamirCiftBoyama      = 0;
    DisTamirSiyah           = 0;
    DisTamirNaylonBoyama    = 0;
    DisTamirPolyesterBoyama = 0;
    DisTamirDiger           = 0;

    UcretliDisTamirBeyaz           = 0;
    UcretliDisTamirRenkli          = 0;
    UcretliDisTamirYikama          = 0;
    UcretliDisTamirSekerKasar      = 0;
    UcretliDisTamirKuruBolum       = 0;
    UcretliDisTamirCiftBoyama      = 0;
    UcretliDisTamirSiyah           = 0;
    UcretliDisTamirNaylonBoyama    = 0;
    UcretliDisTamirPolyesterBoyama = 0;
    UcretliDisTamirDiger           = 0;

    UcretsizDisTamirBeyaz           = 0;
    UcretsizDisTamirRenkli          = 0;
    UcretsizDisTamirYikama          = 0;
    UcretsizDisTamirSekerKasar      = 0;

    UcretsizDisTamirSiyah           = 0;

    UcretsizDisTamirPolyesterBoyama = 0;

    Vardiya1Kilo         = 0;

    Vardiya3Kilo         = 0;

    for select Parti.IslemTipiId, Parti.FaturaKesilecek, TamirCozumSekli.Ucretli, PartiAsamalari.PartiTamirId,
               iif(:Metremi = 0, PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.Metre), IslemGrubu.KatSayi from PartiAsamaStokDetay

    where PartiAsamaStokDetay.MakinaID = :MakinaId and
          PartiAsamaStokDetay.Cikistarihi >= :AAylikBaslangic and PartiAsamaStokDetay.Cikistarihi <= :AAylikBitis and
          Asama.GunlukUretimRaporunaDahil = 1
    into :FIslemTipiId, :FFaturaKesilecek, :FUcretliIcTamir, :PartiTamirId, :FMiktar, :KatSayi do
    begin
      if (PartiTamirID is not null) then
      begin
        if (FUcretliIcTamir = 1) then AylikUcretliIcTamirToplami = AylikUcretliIcTamirToplami + FMiktar;
        else AylikUcretsizIcTamirToplami = AylikUcretsizIcTamirToplami + FMiktar;

        AylikIcTamirUretimToplami = AylikIcTamirUretimToplami + FMiktar;
        AylikIcTamirEsdegerUretim = AylikIcTamirEsdegerUretim + FMiktar * KatSayi;
      end
      else if (FIslemTipiId = itTamir) then
      begin
        if (FFaturaKesilecek = 1) then AylikUcretliDisTamirToplami = AylikUcretliDisTamirToplami + FMiktar;
        else AylikUcretsizDisTamirToplami = AylikUcretsizDisTamirToplami + FMiktar;

        AylikDisTamirEsdegerUretim = AylikDisTamirEsdegerUretim + FMiktar * KatSayi;
      end

    end

    for select parti.IslemGrubuId, parti.IslemTipiId, Parti.FaturaKesilecek, TamirCozumSekli.Ucretli, PartiAsamalari.PartiTamirId,
               iif(:Metremi = 0, PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.Metre), PartiAsamaStokDetay.Cikistarihi from PartiAsamaStokDetay
    left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
    left join PartiAsamaStok On PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    where PartiAsamaStokDetay.MakinaID = :MakinaId and
          PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi <= :ABitisTarihi and
          Asama.GunlukUretimRaporunaDahil = 1
    order by PartiAsamaStokDetay.MakinaId
