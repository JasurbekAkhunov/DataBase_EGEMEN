CREATE PROCEDURE HGSIPARISRAPORU_3(
    ABASLANGICSIPARISYIL INTEGER,
    ABASLANGICSIPARISID INTEGER,
    ABITISSIPARISYIL INTEGER,
    ABITISSIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASINIF VARCHAR(1),
    AMODELID MODELKODUSIZE,
    ASIPONAYDURUMU SMALLINT,
    AMALIONAYDURUMU SMALLINT)
RETURNS (
    HGSIPARISKAYITID INTEGER,
    HGSIPARISDETAYID INTEGER,
    URUNTIPIADI VARCHAR(20),
    HAMID INTEGER,
    URUNID URUNKODUSIZE,
    URUNADI ADI80,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    FIREORANI DOUBLE PRECISION,
    FAZLAKESIMORANI DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    BIRIMMIKTAR NUMERIC(12,3),
    SIPARISMIKTAR NUMERIC(12,3),
    STOKMIKTARI NUMERIC(12,3),
    STOKKAPADEDI NUMERIC(12,3),
    TALEPMIKTARI DOUBLE PRECISION,
    VERILENSIPARISMIKTAR DOUBLE PRECISION,
    GELENSIPARISMIKTAR DOUBLE PRECISION)
AS
declare variable SiparisFirmaId varchar(20);
declare variable StokFirmaId varchar(20);
declare variable OrmeFirmaKodumuz varchar(20);
Declare Variable ModelKayitId Integer;
declare variable ModelKumasID integer;
declare variable AnaKumasModelKumasID integer;
begin
  Select ModelKayitId From Model Where Model.ModelID = :AModelID into :ModelKayitId;
  Select OrmeFirmaKodumuz from sistem where sistemid = 1 into :OrmeFirmaKodumuz;

  For Select HgSiparisDetay.HgSiparisDetayId, HgSiparisDetay.HGSiparisKayitID, HGSiparisDetayKumas.HamId, HamTipi.Adi, Ham.Adi,
             HGSiparisDetayKumas.RenkId, Renk.Adi, HGSiparisDetayKumas.Fireorani, HGSiparisDetayKumas.FazlaKesimOrani,
             ((HGSiparisDetayKumas.Miktar * iif(HGSiparisDetayKumas.FireOrani > 0, HGSiparisDetayKumas.FireOrani / 100 + 1, 1)) *
             iif(HGSiparisDetayKumas.FazlaKesimOrani > 0, HGSiparisDetayKumas.FazlaKesimOrani / 100 + 1, 1)),
             HgSiparis.FirmaId, HgSiparis.Fason, HGSiparisDetayKumas.ModelKumasID, HGSiparisDetayKumas.AnaKumasModelKumasID from HgSiparis
    left join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
    left join HGSiparisDetayKumas on HGSiparisDetayKumas.HgSiparisDetayId = HgSiparisDetay.HgSiparisDetayId
    left join Ham on Ham.HamId = HGSiparisDetayKumas.HamId
    Left Join HamTipi On HamTipi.HamTipiId = Ham.HamTipiId
    left join Renk on Renk.RenkId = HGSiparisDetayKumas.RenkId
    where HGSiparisDetayKumas.HGSiparisDetayKumasId is not Null and
          (:ARaporTipi = 1 and HgSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and HgSiparis.SiparisTarihi <= :ABitisSiparisTarihi or
           :ARaporTipi = 2 and HgSiparisDetay.TeslimTarihi >= :ABaslangicSiparisTarihi and HgSiparisDetay.TeslimTarihi <= :ABitisSiparisTarihi or
           :ARaporTipi = 0 and
              (HgSiparis.HgSiparisYil > :ABaslangicSiparisYil or (HgSiparis.HgSiparisYil = :ABaslangicSiparisYil and HgSiparis.HgSiparisId >= :ABaslangicSiparisId)) and
              (HgSiparis.HgSiparisYil < :ABitisSiparisYil or (HgSiparis.HgSiparisYil = :ABitisSiparisYil  and HgSiparis.HgSiparisId <= :ABitisSiparisId))
          ) and
          (HgSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (HgSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
          (HgSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
          (HgSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null) and
          (HgSiparis.FirmaId = :AFirmaId or :AFirmaId is null) and
          (HgSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
          (HgSiparisDetay.Sinif = :ASinif or :ASinif is null) and
          (HgSiparisDetay.ModelKayitId = :ModelKayitId or :ModelKayitId is Null) and
          (HgSiparisDetay.OnayDurumu = :ASipOnayDurumu or :ASipOnayDurumu is Null) and
          (HgSiparisDetay.MaliOnayDurumu = :AMaliOnayDurumu or :AMaliOnayDurumu is Null) and
          (HgSiparisDetay.Kapali = :AKapali or :AKapali is null) and
          (HgSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null)

  into  :HgSiparisDetayId, :HGSiparisKayitID ,:HamId, :UrunTipiAdi, :UrunAdi, :RenkId, :RenkAdi, :FireOrani, :FazlaKesimOrani, :BirimMiktar, :SiparisFirmaId, :Fason, :ModelKumasID, :AnaKumasModelKumasID do
  begin
    if (Fason = 1) then
    begin
      FasonStr = 'Fason';
      StokFirmaID = SiparisFirmaId;
    end
    else
    begin
      FasonStr = '';
      StokFirmaID = OrmeFirmaKodumuz;
    end

    SiparisMiktar = 0;
    Select Sum(HGSiparisDetayUrun.Miktar) from HGSiparisDetayUrun where HGSiparisDetayUrun.HGSiparisDetayId = :HGSiparisDetayId and (HGSiparisDetayUrun.ModelKumasID = :ModelKumasID or HGSiparisDetayUrun.ModelKumasID = :AnaKumasModelKumasID)
      into :SiparisMiktar;
    SiparisMiktar = SiparisMiktar * (BirimMiktar / 1000); /*Siparişe Birim Gram girilen veri kg ye çevriliyor*/

    StokMiktari  = 0;
    StokKapAdedi = 0;

    Select Sum(OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari), Sum(OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi) from OrmeRaf
      where OrmeRaf.HamID = :HamId and OrmeRaf.StokFirmaId = :StokFirmaID and OrmeRaf.StokVar = 1 and
            StrKarsilastir(OrmeRaf.RenkID, :RenkId) = 1 and OrmeRaf.FasonStok = :Fason
      into :StokMiktari, :StokKapAdedi;


    TalepMiktari = 0;
    VerilenSiparisMiktar = 0;
    GelenSiparisMiktar = 0;

    Select Sum(Miktar) from SatinAlmaTalepDetay where SatinAlmaTalepDetay.HgSiparisDetayId = :HgSiparisDetayId and SatinAlmaTalepDetay.HamId = :HamId into :TalepMiktari;

    Select Sum(VerilenSiparisDetay.Miktar), Sum(VerilenSiparisDetay.GelenMiktar - VerilenSiparisDetay.IadeEdilenMiktar)  from SatinAlmaTalepDetay
      Left Join VerilenSiparisDetay on VerilenSiparisDetay.SatinAlmaTalepDetayId = SatinAlmaTalepDetay.SatinAlmaTalepDetayId
    where SatinAlmaTalepDetay.HgSiparisDetayId = :HgSiparisDetayId and SatinAlmaTalepDetay.HamId = :HamId into :VerilenSiparisMiktar, :GelenSiparisMiktar;

    suspend;
  end

  FazlaKesimOrani = Null;
  HamId           = Null;
  UrunTipiAdi     = Null;
  UrunAdi         = Null;
  StokKapAdedi    = Null;
  For Select HgSiparisDetay.HgSiparisDetayId, HgSiparisDetay.HGSiparisKayitID, Urun.RenkId, UrunRenk.Adi, HGSiparisDetayAksesuar.FireOrani,
             (HGSiparisDetayAksesuar.Miktar * iif(HGSiparisDetayAksesuar.FireOrani > 0, HGSiparisDetayAksesuar.FireOrani / 100 + 1, 1)),
             HgSiparis.FirmaId, HgSiparis.Fason, HGSiparisDetayAksesuar.UrunId, Urun.Adi, UrunTipi.Adi from HgSiparis
    left join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
    left join HGSiparisDetayAksesuar on HGSiparisDetayAksesuar.HgSiparisDetayId = HgSiparisDetay.HgSiparisDetayId
    left join Urun on Urun.UrunID = HGSiparisDetayAksesuar.UrunId
    left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
    left join Renk UrunRenk on UrunRenk.RenkId = Urun.RenkId
    where HGSiparisDetayAksesuar.HGSiparisDetayAksesuarId is not Null and
          (:ARaporTipi = 1 and HgSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and HgSiparis.SiparisTarihi <= :ABitisSiparisTarihi or
           :ARaporTipi = 2 and HgSiparisDetay.TeslimTarihi >= :ABaslangicSiparisTarihi and HgSiparisDetay.TeslimTarihi <= :ABitisSiparisTarihi or
           :ARaporTipi = 0 and
              (HgSiparis.HgSiparisYil > :ABaslangicSiparisYil or (HgSiparis.HgSiparisYil = :ABaslangicSiparisYil and HgSiparis.HgSiparisId >= :ABaslangicSiparisId)) and
              (HgSiparis.HgSiparisYil < :ABitisSiparisYil or (HgSiparis.HgSiparisYil = :ABitisSiparisYil  and HgSiparis.HgSiparisId <= :ABitisSiparisId))
          ) and
          (HgSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (HgSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
          (HgSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
          (HgSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null) and
          (HgSiparis.FirmaId = :AFirmaId or :AFirmaId is null) and
          (HgSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
          (HgSiparisDetay.Sinif = :ASinif or :ASinif is null) and
          (HgSiparisDetay.ModelKayitId = :ModelKayitId or :ModelKayitId is Null) and
          (HgSiparisDetay.OnayDurumu = :ASipOnayDurumu or :ASipOnayDurumu is Null) and
          (HgSiparisDetay.MaliOnayDurumu = :AMaliOnayDurumu or :AMaliOnayDurumu is Null) and
          (HgSiparisDetay.Kapali = :AKapali or :AKapali is null)
  into  :HgSiparisDetayId, :HGSiparisKayitID, :RenkId, :RenkAdi, :FireOrani, :BirimMiktar, :SiparisFirmaId, :Fason,
        :UrunId, :UrunAdi, :UrunTipiAdi do
  begin
    if (Fason = 1) then
    begin
      FasonStr = 'Fason';
      StokFirmaID = SiparisFirmaId;
    end
    else
    begin
      FasonStr = '';
      StokFirmaID = OrmeFirmaKodumuz;
    end

    SiparisMiktar = 0;
    Select Sum(HgSiparisDetayUrun.Miktar) from HgSiparisDetayUrun where HgSiparisDetayUrun.HgSiparisDetayId = :HgSiparisDetayId and HgSiparisDetayUrun.HgSiparisDetayAsortiId is null
      into :SiparisMiktar;
    SiparisMiktar = SiparisMiktar * BirimMiktar;

    StokMiktari  = 0;
    TalepMiktari = 0;
    VerilenSiparisMiktar = 0;
    GelenSiparisMiktar = 0;

    Select Sum(UrunDepoStokGetir.DepoStokMevcudu) from UrunDepoStokGetir(:UrunId, Null) into :StokMiktari;

    Select Sum(Miktar) from SatinAlmaTalepDetay where SatinAlmaTalepDetay.HgSiparisDetayId = :HgSiparisDetayId and SatinAlmaTalepDetay.UrunId = :UrunId into :TalepMiktari;

    Select Sum(VerilenSiparisDetay.Miktar), Sum(VerilenSiparisDetay.GelenMiktar - VerilenSiparisDetay.IadeEdilenMiktar)  from SatinAlmaTalepDetay
      Left Join VerilenSiparisDetay on VerilenSiparisDetay.SatinAlmaTalepDetayId = SatinAlmaTalepDetay.SatinAlmaTalepDetayId
    where SatinAlmaTalepDetay.HgSiparisDetayId = :HgSiparisDetayId and SatinAlmaTalepDetay.UrunId = :UrunId into :VerilenSiparisMiktar, :GelenSiparisMiktar;

    suspend;
  end
end
