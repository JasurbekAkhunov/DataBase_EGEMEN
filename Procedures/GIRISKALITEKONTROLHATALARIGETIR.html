CREATE PROCEDURE GIRISKALITEKONTROLHATALARIGETIR(
    ASTOKHAREKETID INTEGER)
RETURNS (
    GIRISKALITEKONTROLHATALARIID INTEGER,
    GIRISKALITEKONTROLID INTEGER,
    HATATIPIID INTEGER,
    HATATIPIADI VARCHAR(20),
    NOKTASALHATA INTEGER,
    ADET DOUBLE PRECISION,
    BASLANGICBITISMETRESTR VARCHAR(1024),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable BaslangicMetre Double Precision;
declare variable BitisMetre     Double Precision;
begin
  For Select GirisKaliteKontrolHatalari.GirisKaliteKontrolHatalariId, GirisKaliteKontrol.GirisKaliteKontrolId, GirisKaliteKontrolHatalari.HataTipiId, HamHataTipi.Adi,
             GirisKaliteKontrolHatalari.Adet, HamHataTipi.NoktasalHata, IK.Adi, GirisKaliteKontrolHatalari.InsertTarihi, SK.Adi, GirisKaliteKontrolHatalari.DegisimTarihi from GirisKaliteKontrol
    left join GirisKaliteKontrolHatalari on GirisKaliteKontrolHatalari.GirisKaliteKontrolId  = GirisKaliteKontrol.GirisKaliteKontrolId
    left join HamHataTipi on HamHataTipi.HamHataTipiId = GirisKaliteKontrolHatalari.HataTipiId
    left join Kullanici IK on IK.KullaniciId  = GirisKaliteKontrolHatalari.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = GirisKaliteKontrolHatalari.KullaniciId
    where GirisKaliteKontrol.StokHareketId = :AStokHareketId and GirisKaliteKontrolHatalari.GirisKaliteKontrolHatalariID is not Null
  into :GirisKaliteKontrolHatalariId, :GirisKaliteKontrolId, :HataTipiId, :HataTipiAdi, :Adet, :NoktasalHata, :InsertKullaniciAdi,
       :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    BaslangicBitisMetreStr = '';

    if (Adet > 0) then
    begin
      For Select GirisKaliteKontrolHataMetre.BaslangicMetre, GirisKaliteKontrolHataMetre.BitisMetre from GirisKaliteKontrolHataMetre
        where GirisKaliteKontrolHataMetre.GirisKaliteKontrolId = :GirisKaliteKontrolId and GirisKaliteKontrolHataMetre.HataTipiId = :HataTipiId
        order by GirisKaliteKontrolHataMetre.BaslangicMetre, GirisKaliteKontrolHataMetre.BitisMetre
        into :BaslangicMetre, :BitisMetre do
      begin
        if (Char_Length(BaslangicBitisMetreStr) < 100) then
        begin
          if (BaslangicBitisMetreStr <> '' ) then BaslangicBitisMetreStr = BaslangicBitisMetreStr || ' , ';

          BaslangicBitisMetreStr = BaslangicBitisMetreStr || Format_Float('', BaslangicMetre);

          if (NoktasalHata = 0) then
            BaslangicBitisMetreStr = BaslangicBitisMetreStr || '-' || Format_Float('', BitisMetre);

        end
      end
    end

    suspend;
  end
end
