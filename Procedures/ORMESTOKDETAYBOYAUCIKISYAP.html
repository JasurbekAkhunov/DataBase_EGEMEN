CREATE PROCEDURE ORMESTOKDETAYBOYAUCIKISYAP(
    ABARKODNO VARCHAR(30),
    AFISID INTEGER,
    AIRSALIYEID INTEGER,
    ABOYATALIMATID INTEGER,
    AHAREKETTIPI INTEGER,
    AKILO DOUBLE PRECISION,
    ABRUTKILO DOUBLE PRECISION,
    AMETRE DOUBLE PRECISION,
    AKAPADEDI INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable shtAlis smallInt;
declare variable shtFasonUretim SmallInt;
declare variable shtUretim      SmallInt;
declare variable FHamID integer;
declare variable FBoyaSiparisKayitID integer;
declare variable FBoyaSiparisDetayID integer;
declare variable FBoyaSiparisBaskiId integer;
declare variable FOrmeSiparisDetayID integer;
declare variable FLot       varchar(150);
declare variable FRenkID    integer;
declare variable FUreticiId integer;
declare variable FMelanjKodu varchar(20);
declare variable FDepoId     varchar(10);
declare variable FStokHareketID  integer;
declare variable IplikStokNumarator integer;
declare variable Kilo  Double Precision;
declare variable BrutKilo Double Precision;
declare variable Sakat integer;
declare variable StokHareketID integer;
declare variable OrmeRafId integer;
declare variable KayitStokHareketId integer;
declare variable PBoyaSiparisKayitID integer;
declare variable POrmeSiparisDetayID integer;
declare variable PBoyaSiparisDetayID integer;
declare variable PBoyaSiparisBaskiId integer;
declare variable PRenkID    integer;
declare variable IK     varchar(2);
declare variable TF     integer;
declare variable PaketId integer;
declare variable DokumaHareketi Integer;
declare variable MiktarBirim    Integer;
declare variable DovizId DovizSize;
declare variable KaynakDovizId DovizSize;
declare variable Ambalajno varchar(10);
declare variable Notu blob sub_type 0 segment size 80;
declare variable KapAdedi integer;
declare variable BobinGramaji double precision;
declare variable KonikRengiId integer;
declare variable StokHareketDetayId integer;
declare variable KaliteNo integer;
declare variable Metre double precision;
begin
  Select shtAlis, shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir into :shtAlis, :shtFasonUretim, :shtUretim;

  Select First 1 StokHareket.HamID, StokHareket.BoyaSiparisKayitID, StokHareket.BoyaSiparisDetayID, StokHareket.BoyaSiparisBaskiId, StokHareket.OrmeSiparisDetayID, StokHareket.Lot, StokHareket.RenkID,
         StokHareket.UreticiId, StokHareket.MelanjKodu, StokHareket.DepoId, StokHareket.Sakat, StokHareket.DokumaHareketi, StokHareket.MiktarBirim, StokHAreket.DovizId,
          StokHareketDetay.Ambalajno, StokHareketDetay.Notu, StokHareketDetay.KapAdedi, StokHareketDetay.BobinGramaji, StokHareketDetay.KonikRengiId,
          Coalesce(OrmeRafTransferFisiDetay.YeniOrmeRafId, StokHareketDetay.OrmeRafId) from StokHareketDetay
  left join OrmeRafTransferFisiDetay on OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId = StokHareketDetay.OrmeRafTransferFisiDetayID
  left join StokHareket on StokHareket.StokHareketID = StokHareketDetay.StokHareketID
  where StokHareketDetay.Barkod = :ABarkodNo and (StokHareket.HareketTipi = :shtAlis or StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim)
  into :FHamID, :FBoyaSiparisKayitID, :FBoyaSiparisDetayID, :FBoyaSiparisBaskiId, :FOrmeSiparisDetayID, :FLot, :FRenkID, :FUreticiId, :FMelanjKodu, :FDepoId, :Sakat,
  :DokumaHareketi, :MiktarBirim, :KaynakDovizId, :Ambalajno, :Notu, :KapAdedi, :BobinGramaji, :KonikRengiId, :OrmeRafId;


  /*Girişi Yapılmamışsa Yapalım*/
  IK = EgemenCopyStr(ABarkodNo, 1, 2);
  if (FHamID is Null) then
  begin


    if (IK = 'K-') then
    begin
      TF = EgemenStrToIntOlurmu(EgemenCopyStr(ABarkodNo, 3, StrLength(ABarkodNo) - 2));
      if (TF = 0) then exception BarkodHatali1;
      PaketId = EgemenStrToInt(EgemenCopyStr(ABarkodNo, 3, StrLength(ABarkodNo) - 2));

      Select first 1 StokHareket.StokHareketID, PSKaliteKontrol.Kilo, PSKaliteKontrol.Kilo + PSKaliteKontrol.Dara, StokHareket.HamID, StokHareket.BoyaSiparisKayitID, StokHareket.BoyaSiparisDetayID,
           StokHareket.BoyaSiparisBaskiId, StokHareket.OrmeSiparisDetayID, StokHareket.Lot, StokHareket.RenkID, StokHareket.UreticiId, StokHareket.MelanjKodu, StokHareket.DepoId, StokHareket.Sakat from PsKaliteKontrol
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = PsKaliteKontrol.PartiStokTartiId
      Left Join StokHareket on StokHareket.BoyaIrsaliyeDetayID = PartiStokTarti.BoyaIrsaliyeDetayID
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where PsKaliteKontrol.PsKaliteKontrolId = :PaketId
      into :StokHareketID, :Kilo, :BrutKilo, :FHamID, :FBoyaSiparisKayitID, :FBoyaSiparisDetayID, :FBoyaSiparisBaskiId, :FOrmeSiparisDetayID,
           :FLot, :FRenkID, :FUreticiId, :FMelanjKodu, :FDepoId, :Sakat;

      if (StokHareketID is not Null) then
      begin
        insert into StokHareketDetay(StokHareketID, HareketTipi, PsKaliteKontrolID, Kilo, BrutKilo, InsertKullaniciId, KullaniciId) Values
                                    (:StokHareketID, :shtAlis, :PaketId, :Kilo, :BrutKilo, :AKullaniciId, :AKullaniciId);
      end
    end
    else
    if (IK = 'I-') then
    begin
      TF = EgemenStrToIntOlurmu(EgemenCopyStr(ABarkodNo, 3, StrLength(ABarkodNo) - 2));
      if (TF = 0) then exception BarkodHatali1;
      PaketId = EgemenStrToInt(EgemenCopyStr(ABarkodNo, 3, StrLength(ABarkodNo) - 2));

      Select first 1 StokHareket.StokHareketID, IplikPaketleme.NetKilo, IplikPaketleme.BrutKilo, StokHareket.HamID, StokHareket.BoyaSiparisKayitID, StokHareket.BoyaSiparisDetayID,
           StokHareket.BoyaSiparisBaskiId, StokHareket.OrmeSiparisDetayID, StokHareket.Lot, StokHareket.RenkID, StokHareket.UreticiId, StokHareket.MelanjKodu, StokHareket.DepoId, StokHareket.Sakat from IplikPaketleme
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiId
      Left Join StokHareket on StokHareket.BoyaIrsaliyeDetayID = PartiStokTarti.BoyaIrsaliyeDetayID
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where IplikPaketleme.IplikPaketlemeID = :PaketId
      into :StokHareketID, :Kilo, :BrutKilo, :FHamID, :FBoyaSiparisKayitID, :FBoyaSiparisDetayID, :FBoyaSiparisBaskiId, :FOrmeSiparisDetayID,
           :FLot, :FRenkID, :FUreticiId, :FMelanjKodu, :FDepoId, :Sakat;

      if (StokHareketID is not Null) then
      begin
        insert into StokHareketDetay(StokHareketID, HareketTipi, IplikPaketlemeID, Kilo, BrutKilo, InsertKullaniciId, KullaniciId) Values
                                    (:StokHareketID, :shtAlis, :PaketId, :Kilo, :BrutKilo, :AKullaniciId, :AKullaniciId);
      end
    end
    else
    begin
      Select first 1 StokHareket.StokHareketID, IplikStok.Numarator, IplikStok.NetMiktar, IplikStok.Miktar, StokHareket.HamID, StokHareket.BoyaSiparisKayitID, StokHareket.BoyaSiparisDetayID,
           StokHareket.BoyaSiparisBaskiId, StokHareket.OrmeSiparisDetayID, StokHareket.Lot, StokHareket.RenkID, StokHareket.UreticiId, StokHareket.MelanjKodu, StokHareket.DepoId, StokHareket.Sakat from IplikStok
      Left Join StokHareket on StokHareket.IplikIrsaliyeDetayID = IplikStok.CikisIplikIrsaliyeDetayID
      where IplikStok.IplikStokID = :ABarkodNo and StokHareket.StokHareketID is not Null
      Order by IplikStok.Numarator Desc
      into :StokHareketID, :IplikStokNumarator, :Kilo, :BrutKilo, :FHamID, :FBoyaSiparisKayitID, :FBoyaSiparisDetayID, :FBoyaSiparisBaskiId, :FOrmeSiparisDetayID,
           :FLot, :FRenkID, :FUreticiId, :FMelanjKodu, :FDepoId, :Sakat;

      if (StokHareketID is not Null) then
      begin
        insert into StokHareketDetay(StokHareketID, HareketTipi, IplikStokID, IplikStokNumarator, Kilo, BrutKilo, InsertKullaniciId, KullaniciId) Values
                                    (:StokHareketID, :shtAlis, :ABarkodNo, :IplikStokNumarator, :Kilo, :BrutKilo, :AKullaniciId, :AKullaniciId);
      end
    end



    if (StokHareketID is not Null) then
    begin
      if (AKilo = 0) then
      begin
        AKilo      = Kilo;
        ABrutKilo  = BrutKilo;
      end
    end
  end

  if (FHamID is Null) then exception KayitHatasi ' Çıkışı Yapılmak İstenen Barkod Bulunamadı!';
  if (MiktarBirim = 0 and (AKilo is Null or AKilo = 0)) then exception KayitHatasi ' Üretime Çıkılacak Miktar 0 dan Büyük Olmalı';
  if (MiktarBirim = 1 and (AMetre is Null or AMetre = 0)) then exception KayitHatasi ' Üretime Çıkılacak Miktar 0 dan Büyük Olmalı';

  Select OrmeSiparisDetay.OrmeSiparisDetayID, BoyaTalimat.BoyaSiparisKayitID, BoyaTalimat.BoyaSiparisDetayId, BoyaTalimat.BoyaSiparisBaskiId, OrmeSiparisDetay.RenkID from BoyaTalimat
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaTalimat.BoyaSiparisDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
    where BoyaTalimat.BoyaTalimatKayitID = :ABoyaTalimatID
  into :POrmeSiparisDetayID, :PBoyaSiparisKayitID, :PBoyaSiparisDetayId, :PBoyaSiparisBaskiId, :PRenkID;


  if (FOrmeSiparisDetayID>0) then
  begin
    if ((POrmeSiparisDetayID is not null and IntKarsilastir(:POrmeSiparisDetayID, :FOrmeSiparisDetayID) = 0) or (POrmeSiparisDetayID is not null and IntKarsilastir(:PBoyaSiparisKayitID, :FBoyaSiparisKayitID) = 0)) then
      Exception KayitHatasi ' Sipariş Bilgileri Uyuşmuyor';
  end

--  FBoyaSiparisKayitID = PBoyaSiparisKayitID;
--  FOrmeSiparisDetayID = POrmeSiparisDetayID;
  FBoyaSiparisDetayId = PBoyaSiparisDetayId;
  FBoyaSiparisBaskiId = PBoyaSiparisBaskiId;

  if (AFisId is not null) then
    select OrmeFis.DovizId from OrmeFis where OrmeFis.OrmeFisId = :AFisId into :DovizId;
  if (AIrsaliyeId is not null) then
    select OrmeIrsaliye.DovizId from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = :AIrsaliyeId into :DovizId;

  if (DovizId is null) then
    DovizId = KaynakDovizId;

  if (OrmeRafId is Null) then
  begin
    if (Sakat = 0) then Select Max(UrunRafId) from OrmeHamRafListesiGetir(:FHamID, :FBoyaSiparisKayitId, :FOrmeSiparisDetayId, :FBoyaSiparisDetayId, Null, Null, :FDepoID)
                          where StrKarsilastir(Lot, :FLot) = 1 and IntKarsilastir(RenkID, :FRenkID) = 1 and RafMiktari > 0  into :OrmeRafId;
    else Select Max(UrunRafId) from OrmeHamRafListesiGetir(:FHamID, :FBoyaSiparisKayitId, :FOrmeSiparisDetayId, :FBoyaSiparisDetayId, Null, Null, :FDepoID)
                          where StrKarsilastir(Lot, :FLot) = 1 and IntKarsilastir(RenkID, :FRenkID) = 1 and  Sakat = 1 and RafMiktari > 0 into :OrmeRafId;
  end

  AKapAdedi = coalesce(AKapAdedi,0);
  if (AKapAdedi = 0) then AKapAdedi = coalesce(KapAdedi, 1);

    -- Bölme İşi İçin Girildi
  Select First 1  StokHareketDetay.StokHareketDetayId, OrmeRaf.OrmeRafID, OrmeRaf.KaliteNo, (GirisKilo - CikisKilo), (GirisBrutKilo - CikisBrutKilo),
      (GirisMetre - CikisMetre), (StokHareketDetayMiktar.GirisKapAdedi - StokHareketDetayMiktar.CikisKapAdedi), StokHareket.MiktarBirim,
      StokHareketDetay.Ambalajno, StokHareketDetay.Notu,  StokHareketDetay.BobinGramaji, StokHareketDetay.KonikRengiId from StokHareketDetay
      Left Join StokHareket on StokHareket.StokHareketId = StokHareketDetay.StokHareketId
      Left Join StokHareketDetayMiktar on StokHareketDetayMiktar.Barkod = StokHareketDetay.Barkod
      Left Join OrmeRaf on OrmeRaf.OrmeRafID = StokHareketDetay.OrmeRafID
      where StokHareketDetay.Barkod = :ABARKODNO and (StokHareketDetay.HareketTipi in (1, 8, 9)) and
            ((StokHareket.MiktarBirim = 1 and (GirisMetre - CikisMetre > 0)) or (StokHareket.MiktarBirim = 0 and (GirisKilo - CikisKilo > 0)))
    Into  :StokHareketDetayId, :OrmeRafId, :KaliteNo, :Kilo, :BrutKilo, :Metre, :KapAdedi, :MiktarBirim, :Ambalajno, :Notu,  :BobinGramaji, :KonikRengiId;

    if (StokHareketDetayId is null) then exception BarkodHatali2;

    Select First 1 OrmeRafTransferFisiDetay.YeniOrmeRafID from OrmeRafTransferFisiDetay where OrmeRafTransferFisiDetay.StokHareketDetayId = :StokHareketDetayId
      Order By InsertTarihi Desc
    into :OrmeRafId;

    if ((MiktarBirim = 0 and AKilo <> Kilo) or (MiktarBirim = 1 and AMetre <> Metre)) then /*Miktar Böleceğiz*/
    begin
      Update StokHareketDetay Set Kilo = Kilo - :AKilo, BrutKilo = BrutKilo - :ABrutKilo, Metre = Metre - :AMetre, KapAdedi = KapAdedi - :AKapAdedi  where StokHareketDetayID = :StokHareketDetayId;

      insert into StokHareketDetay(StokHareketID, HareketTipi, IplikStokID, IplikStokNumarator, Kilo, BrutKilo, Metre, PSKaliteKontrolID, IplikPaketlemeID, GirisKaliteKontrolId, KapAdedi, InsertKullaniciId, Ambalajno, Notu,  BobinGramaji, KonikRengiId, BolunduguStokHareketDetayId)
        Select StokHareketID, HareketTipi, IplikStokID, IplikStokNumarator, :AKilo, :ABrutKilo, :AMetre,  PSKaliteKontrolID, IplikPaketlemeID, GirisKaliteKontrolId, :AKapAdedi , :AKullaniciId, :Ambalajno, :Notu,  :BobinGramaji, :KonikRengiId, :StokHareketDetayId from StokHareketDetay
        where StokHareketDetayId = :StokHareketDetayID
      Returning StokHareketDetayID into :StokHareketDetayID;

    end

   -- Bölme İşi İçin Girildi

  Select StokHareketId from StokHareket
    where (StokHareket.OrmeIrsaliyeId = :AIrsaliyeId or :AIrsaliyeId is Null) and
          (StokHareket.OrmeFisId = :AFisId or :AFisId is Null) and
           StokHareket.OrmeRafId = :OrmeRafId and (StokHareket.BoyaTalimatKayitID = :ABoyaTalimatID or :ABoyaTalimatID is null) and
           (StrKarsilastir(StokHareket.Lot, :FLot) = 1)
    into :KayitStokHareketId;


  if (KayitStokHareketId is not Null) then
  begin
    if (MiktarBirim = 0) then
      Update StokHareket Set KapAdedi = KapAdedi + :AKapAdedi, Miktar = Miktar + :AKilo, BrutMiktar = BrutMiktar + :ABrutKilo, Metre = Metre + :AMetre, NeredenUpdate = 3 where StokHareket.StokHareketId = :KayitStokHareketId;
    else
      Update StokHareket Set KapAdedi = KapAdedi + :AKapAdedi, Metre = Metre + :AMetre, Kilo = Kilo + :AKilo, NeredenUpdate = 3 where StokHareket.StokHareketId = :KayitStokHareketId;
  end
  else
  begin
    KayitStokHareketId = gen_id(StokHareket_Id_Gen, 1);

    if (MiktarBirim = 0) then
      insert into Stokhareket(StokHareketId, DovizId, Tarih, FiiliSevkTarihi, OrmeFisId, OrmeIrsaliyeId, HareketTipi, Iade, HamIade, OrmeSiparisDetayId, BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisBaskiId, BoyaTalimatKayitID, HamId, OrmeRafId,
                            KapAdedi, Miktar, Kilo, BrutMiktar, Metre, RenkId, Lot, UreticiId, MelanjKodu, DepoId, InsertKullaniciId, KullaniciId, DokumaHareketi) Values
                           (:KayitStokHareketId, :DovizId, current_timestamp(0), current_timestamp(0), :AFisId, :AIrsaliyeId, :AHareketTipi, 0, 0, :FOrmeSiparisDetayId, :FBoyaSiparisKayitId, :FBoyaSiparisDetayId, :FBoyaSiparisBaskiId, :ABoyaTalimatId, :FHamId, :OrmeRafId,
                            :AKapAdedi, :AKilo, :AKilo, :ABrutKilo, :AMetre, :FRenkId, :FLot, :FUreticiId, :FMelanjKodu, :FDepoId, :AKullaniciId, :AKullaniciId, :DokumaHareketi);
    else
      insert into Stokhareket(StokHareketId, DovizId, Tarih, FiiliSevkTarihi, OrmeFisId, OrmeIrsaliyeId, HareketTipi, Iade, HamIade, OrmeSiparisDetayId, BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisBaskiId, BoyaTalimatKayitID, HamId, OrmeRafId,
                            KapAdedi, Metre, Kilo, RenkId, Lot, UreticiId, MelanjKodu, DepoId, InsertKullaniciId, KullaniciId, DokumaHareketi) Values
                           (:KayitStokHareketId, :DovizId, current_timestamp(0), current_timestamp(0), :AFisId, :AIrsaliyeId, :AHareketTipi, 0, 0, :FOrmeSiparisDetayId, :FBoyaSiparisKayitId, :FBoyaSiparisDetayId, :FBoyaSiparisBaskiId, :ABoyaTalimatId, :FHamId, :OrmeRafId,
                            :AKapAdedi, :AMetre, :AKilo, :FRenkId, :FLot, :FUreticiId, :FMelanjKodu, :FDepoId, :AKullaniciId, :AKullaniciId, :DokumaHareketi);
  end

  insert into StokHareketDetay(StokHareketID, HareketTipi, Barkod, Kilo, BrutKilo, Metre, InsertKullaniciId, KullaniciId, Ambalajno, Notu, KapAdedi, BobinGramaji, KonikRengiId) Values
                              (:KayitStokHareketId, :AHareketTipi, case when (Coalesce(:IK,'') <> 'G-') then  'S'||:StokHareketDetayID else :ABarkodNo end, :AKilo, :ABrutKilo, :AMetre, :AKullaniciId, :AKullaniciId, :Ambalajno, :Notu, :AKapAdedi, :BobinGramaji, :KonikRengiId);

end
