CREATE PROCEDURE DOKUMABOYAURETIMCIKISTOPLISTESI(
    AMINTOPNO INTEGER,
    AMAXTOPNO INTEGER,
    AMINMIKTAR NUMERIC(12,3),
    AMAXMIKTAR NUMERIC(12,3),
    ALOT VARCHAR(150),
    AATKILOTU VARCHAR(30),
    AHAMID INTEGER,
    AFIRMAID VARCHAR(20),
    AMAKINAGRUPID MAKINAGRUPKODUSIZE,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ACOZGULOTU VARCHAR(30),
    AHATADURUMU SMALLINT,
    AGIRISKALITEKONTROL SMALLINT)
RETURNS (
    DOKUMAKALITEKONTROLID INTEGER,
    STOKHAREKETDETAYID INTEGER,
    GIRISKALITEKONTROLID INTEGER,
    DOKUMAPLANLAMAID INTEGER,
    ORMEVARDIYAID INTEGER,
    STOKHAREKETID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    TARIH TIMESTAMP,
    VARDIYATARIHI TIMESTAMP,
    VARDIYA INTEGER,
    VARDIYAAMIRIPERSONELADI VARCHAR(40),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    AYARTOPU INTEGER,
    AYARTOPUSTR VARCHAR(10),
    SAKATHAMHATATIPIID INTEGER,
    SAKATHAMHATATIPIADI VARCHAR(20),
    RAFID VARCHAR(10),
    KKLOT VARCHAR(150),
    FISSERINO VARCHAR(2),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    TOPNO INTEGER,
    MIKTAR DOUBLE PRECISION,
    KAPADEDI INTEGER,
    SIPARISFIRMAADI VARCHAR(100),
    TALIMATLOT VARCHAR(150),
    PLANLAMANOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    ISTENENKUMASENI INTEGER,
    KILO BOYAMIKTARSIZE,
    NETKILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    SEVKIRSALIYENO IRSALIYENOSIZE,
    SEVKIRSALIYETARIHI TIMESTAMP,
    ATKILOTU VARCHAR(30),
    COZGULOTU VARCHAR(30),
    KONTROLBITTI SMALLINT,
    KONTROLBITTISTR VARCHAR(10),
    KONTROLBITISTARIHI TIMESTAMP,
    BITIRENKULLANICIADI ADI40,
    NOTU VARCHAR(4000),
    URETIMPARTINO VARCHAR(30),
    DOKUMAMAKINAID MAKINAKODUSIZE,
    DOKUMAMAKINAADI ADI40,
    HATALAR VARCHAR(4000),
    GRAMAJ DOUBLE PRECISION,
    GRAMAJSTR VARCHAR(20),
    KALITENO INTEGER)
AS
declare variable TUPACIKENMAYLI integer;
declare variable FHATATIPIADI ADI20;
declare variable RAFMIKTARI numeric(12,3);
declare variable FADET integer;
declare variable FHATATOPLAMMETRE BOYAMIKTARSIZE;
begin
  Hatalar = '';
  if (AGirisKaliteKontrol = 0) then
  begin
    For Select DokumaKaliteKontrol.DokumaKaliteKontrolId, DokumaKaliteKontrol.DokumaPlanlamaId, DokumaKaliteKontrol.OrmeVardiyaId, DokumaKaliteKontrol.StokHareketId,
               DokumaKaliteKontrol.Tarih, DokumaKaliteKontrol.VardiyaTarihi, DokumaKaliteKontrol.Vardiya, VAP.Adi, DokumaKaliteKontrol.PersonelId, P.Adi, DokumaKaliteKontrol.RafId,
               DokumaKaliteKontrol.TopNo, DokumaKaliteKontrol.Sakat, DokumaKaliteKontrol.AyarTopu, DokumaKaliteKontrol.Miktar, IK.Adi, DokumaKaliteKontrol.InsertTarihi, SK.Adi, DokumaKaliteKontrol.DegisimTarihi,
               OrmeFis.Serino, OrmeFis.FisNo, OrmeFis.Fistarihi, Firma.Adi, DokumaKaliteKontrol.SakatHamHataTipiId, HamHataTipi.Adi, DokumaPlanlama.Lot, DokumaPlanlama.Notu, OrmeSiparisDetay.IstenenKumasEni,
               DokumaKaliteKontrol.Lot, DokumaKaliteKontrol.KapAdedi, DokumaKaliteKontrol.Kilo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, DokumaKaliteKontrol.AtkiLotu, DokumaKaliteKontrol.CozguLotu,
               DokumaKaliteKontrol.KontrolBitti, DokumaKaliteKontrol.BitisTarihi, BK.Adi, DokumaKaliteKontrol.Notu,
               Ham.HamId, Ham.Adi, Ham.HamTipiId, DokumaPlanlama.UretimPartiNo, DokumaMakina.DokumaMakinaId, DokumaMakina.Adi, DokumaKaliteKontrol.KaliteNo
                from DokumaKaliteKontrol
      left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaKaliteKontrol.DokumaPlanlamaId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
      left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
      left join StokHareket on StokHareket.StokHareketId = DokumaKaliteKontrol.StokHareketId
      left join OrmeFis on OrmeFis.OrmeFisId  = StokHareket.OrmeFisId
      left join Personel VAP on VAP.PersonelId  = DokumaKaliteKontrol.VardiyaAmiriPersonelId
      left join Personel P on P.PersonelId  = DokumaKaliteKontrol.PersonelId
      left join HamHataTipi on HamHataTipi.HamHataTipiId  = DokumaKaliteKontrol.SakatHamHataTipiId
      left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = DokumaKaliteKontrol.OrmeIrsaliyeId
      left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
      left join DokumaMakina on DokumaMakina.DokumaMakinaId = DokumaPlanlama.DokumaMakinaId
      left join Kullanici IK on IK.KullaniciId  = DokumaKaliteKontrol.InsertKullaniciId
      left join Kullanici SK on SK.KullaniciId  = DokumaKaliteKontrol.KullaniciId
      left join Kullanici BK on BK.KullaniciId  = DokumaKaliteKontrol.BitirenKullaniciId
      where DokumaKaliteKontrol.KontrolBitti = 1 and DokumaKaliteKontrol.OrmeIrsaliyeId is null and DokumaKaliteKontrol.OrmeUCikisIrsaliyeId is null and DokumaKaliteKontrol.SarfOrmeFisId is null and
            (OrmeSiparisDetay.HamId = :AHamId or :AHamId is null) and
            (DokumaKaliteKontrol.Lot = :ALot or :ALot is null) and
            (DokumaKaliteKontrol.AtkiLotu = :AAtkiLotu or :AAtkiLotu is null) and
            (DokumaKaliteKontrol.TopNo <= :AMaxTopNo or :AMaxTopNo is null) and
            (DokumaKaliteKontrol.TopNo >= :AMinTopNo or :AMinTopNo is null) and
            (DokumaKaliteKontrol.Miktar <= :AMaxMiktar or :AMaxMiktar is null) and
            (DokumaKaliteKontrol.Miktar >= :AMinMiktar or :AMinMiktar is null) and
            (DokumaMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
            DokumaKaliteKontrol.InsertTarihi >= :ABaslangicTarihi and DokumaKaliteKontrol.InsertTarihi < DateAdd(DAY, 1, :ABitisTarihi) and
            (DokumaKaliteKontrol.CozguLotu = :ACozguLotu or :ACozguLotu is null)
      order by DokumaKaliteKontrol.DokumaPlanlamaId, DokumaKaliteKontrol.Tarih
    into :DokumaKaliteKontrolId, :DokumaPlanlamaId, :OrmeVardiyaId, :StokHareketId, :Tarih,
         :VardiyaTarihi, :Vardiya, :VardiyaAmiriPersonelAdi, :PersonelId, :PersonelAdi, :RafId, :TopNo, :Sakat, :AyarTopu,  :Miktar, :InsertKullaniciAdi,
         :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :FisSerino, :FisNo, :Fistarihi, :SiparisFirmaAdi, :SakatHamHataTipiId, :SakatHamHataTipiAdi,
         :TalimatLot, :PlanlamaNotu, :IstenenKumasEni, :KKLot, :KapAdedi, :Kilo, :SevkIrsaliyeNo, :SevkIrsaliyeTarihi, :AtkiLotu, :CozguLotu, :KontrolBitti,
         :KontrolBitisTarihi, :BitirenKullaniciAdi, :Notu, :HamId, :HamAdi, :HamTipiId, :UretimPartiNo, :DokumaMakinaId, :DokumaMakinaAdi, :KaliteNo do
    begin
      if (Kilo is null or Miktar is null or Miktar = 0) then
        Gramaj = Null;
      else
      begin
        Gramaj = Yuvarla(Kilo / Miktar, 1000);
        GramajStr = Replace(Format_Float('0.000', Gramaj), ',', '.');
      end

      if (KontrolBitti = 1) then KontrolBittiStr = 'Bitti';
      else KontrolBittiStr = '';

      if (Sakat = 1) then SakatStr = 'Sakat';
      else SakatStr = '';

      if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
      else AyarTopuStr = '';

      Hatalar = '';

      for select DokumaHataTipi.Adi, DokumaKaliteKontrolHatalari.Adet  from DokumaKaliteKontrolHatalari
        left join DokumaHataTipi on DokumaHataTipi.DokumaHataTipiId = DokumaKaliteKontrolHatalari.DokumaHataTipiId
        where DokumaKaliteKontrolHatalari.DokumaKaliteKontrolId = :DokumaKaliteKontrolId into :FHataTipiAdi, :FAdet do
      begin
        select Coalesce(Sum(DokumaKaliteKontrolHataMetre.BitisMetre - DokumaKaliteKontrolHataMetre.BaslangicMetre),0) from DokumaKaliteKontrolHataMetre
        where DokumaKaliteKontrolHataMetre.DokumaKaliteKontrolId = :DokumaKaliteKontrolId into :FHataToplamMetre;

        Hatalar = Hatalar||FHataTipiAdi||'('||FAdet||','||FHataToplamMetre||')'||' ';
      end

      select Coalesce(OrmeRaf.netgirismiktari - ormeraf.netcikismiktari, 0) from DokumaKaliteKontrol
      left join OrmeRafDetay on OrmeRafDetay.stokhareketid = DokumaKaliteKontrol.StokHAreketId
      left join OrmeRaf on OrmeRaf.OrmeRafId = OrmeRafDetay.ormerafid
      where DokumaKaliteKontrol.dokumakalitekontrolid = :DokumaKaliteKontrolId
      into :RafMiktari;


      if (AHataDurumu = 0 or (AHataDurumu = 1 and Hatalar <> '') or (AHataDurumu = 2 and Hatalar = '')) then
          if (RafMiktari > 0) then
                suspend;
    end
  end

  Hatalar = '';
  For Select Null, Null, Null, StokHareket.StokHareketId,
             GirisKaliteKontrol.Tarih, Null, Null, Null, Null, Null, OrmeRafDetay.RafId,
             GirisKaliteKontrol.TopNo, StokHareket.Sakat, StokHareket.AyarTopu, GirisKaliteKontrol.Metre, IK.Adi, GirisKaliteKontrol.InsertTarihi, SK.Adi, GirisKaliteKontrol.DegisimTarihi,
             OrmeFis.Serino, OrmeFis.FisNo, OrmeFis.Fistarihi, Firma.Adi, Null, Null, StokHareket.Lot, Null, OrmeSiparisDetay.IstenenKumasEni,
             StokHareket.Lot, GirisKaliteKontrol.Adet, GirisKaliteKontrol.BrutKilo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, StokHareket.AtkiLotu, StokHareket.CozguLotu,
             1, StokHareket.GirisKKBitisTarihi, BK.Adi, Null,
             Ham.HamId, Ham.Adi, Ham.HamTipiId, Coalesce(DokumaPlanlama.UretimPartiNo, StokHareket.UretimPartiNo), StokHareket.DokumaMakinaId, DokumaMakina.Adi, GirisKaliteKontrol.KaliteNo,
             StokHareketDetay.StokHareketDetayId, GirisKaliteKontrol.GirisKaliteKontrolId, GirisKaliteKontrol.NetKilo, GirisKaliteKontrol.Metre from GirisKaliteKontrol
    left join StokHareketDetay on StokHareketDetay.GirisKaliteKontrolId = GirisKaliteKontrol.GirisKaliteKontrolId
    left join StokHareket on StokHareket.StokHareketId = StokHareketDetay.StokHareketId
    left join OrmeRafDetay on OrmeRafDetay.StokHareketId = StokHareket.StokHareketId
    left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = StokHareket.DokumaPlanlamaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join Firma on Firma.FirmaId = StokHareket.FirmaId
    left join OrmeFis on OrmeFis.OrmeFisId  = StokHareket.OrmeFisId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
    left join Ham on Ham.HamId = StokHareket.HamId
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = StokHareket.DokumaMakinaId
    left join Kullanici IK on IK.KullaniciId  = GirisKaliteKontrol.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = GirisKaliteKontrol.KullaniciId
    left join Kullanici BK on BK.KullaniciId = StokHareket.GirisKKBitirenKullaniciId
    left join StokHareketDetayMiktar on StokHareketDetayMiktar.Barkod = StokHareketDetay.Barkod
    where StokHareket.GirisKKBitirenKullaniciId is not null and StokHareketDetayMiktar.GirisMetre - StokHareketDetayMiktar.CikisMetre > 0 and
          (StokHareket.HamId = :AHamId or :AHamId is null) and
          (StokHareket.Lot = :ALot or :ALot is null) and
          (StokHareket.AtkiLotu = :AAtkiLotu or :AAtkiLotu is null) and
          (GirisKaliteKontrol.TopNo <= :AMaxTopNo or :AMaxTopNo is null) and
          (GirisKaliteKontrol.TopNo >= :AMinTopNo or :AMinTopNo is null) and
          (GirisKaliteKontrol.Metre <= :AMaxMiktar or :AMaxMiktar is null) and
          (GirisKaliteKontrol.Metre >= :AMinMiktar or :AMinMiktar is null) and
          (DokumaMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
          GirisKaliteKontrol.InsertTarihi >= :ABaslangicTarihi and GirisKaliteKontrol.InsertTarihi < DateAdd(DAY, 1, :ABitisTarihi) and
          (StokHareket.CozguLotu = :ACozguLotu or :ACozguLotu is null)
    order by GirisKaliteKontrol.Tarih, GirisKaliteKontrol.InsertTarihi
  into :DokumaKaliteKontrolId, :DokumaPlanlamaId, :OrmeVardiyaId, :StokHareketId, :Tarih,
       :VardiyaTarihi, :Vardiya, :VardiyaAmiriPersonelAdi, :PersonelId, :PersonelAdi, :RafId, :TopNo, :Sakat, :AyarTopu,  :Miktar, :InsertKullaniciAdi,
       :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :FisSerino, :FisNo, :Fistarihi, :SiparisFirmaAdi, :SakatHamHataTipiId, :SakatHamHataTipiAdi,
       :TalimatLot, :PlanlamaNotu, :IstenenKumasEni, :KKLot, :KapAdedi, :Kilo, :SevkIrsaliyeNo, :SevkIrsaliyeTarihi, :AtkiLotu, :CozguLotu, :KontrolBitti,
       :KontrolBitisTarihi, :BitirenKullaniciAdi, :Notu, :HamId, :HamAdi, :HamTipiId, :UretimPartiNo, :DokumaMakinaId, :DokumaMakinaAdi, :KaliteNo, :StokHareketDetayId,
       :GirisKaliteKontrolId, :NetKilo, :Metre do
  begin
    /*if (Kilo is null or Miktar is null or Miktar = 0) then
      Gramaj = Null;
    else
    begin
      Gramaj = Yuvarla(Kilo / Miktar, 1000);
      GramajStr = Replace(Format_Float('0.000', Gramaj), ',', '.');
    end*/
    GramajStr = Null;
    GramajStr = Null;

    if (KontrolBitti = 1) then KontrolBittiStr = 'Bitti';
    else KontrolBittiStr = '';

    if (Sakat = 1) then SakatStr = 'Sakat';
    else SakatStr = '';

    if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
    else AyarTopuStr = '';

    Hatalar = '';

    for select HataTipi.Adi, GirisKaliteKontrolHatalari.Adet  from GirisKaliteKontrolHatalari
      left join HataTipi on HataTipi.HataTipiId = GirisKaliteKontrolHatalari.HataTipiId
      where GirisKaliteKontrolHatalari.GirisKaliteKontrolId = :GirisKaliteKontrolId into :FHataTipiAdi, :FAdet do
    begin
      select Coalesce(Sum(GirisKaliteKontrolHataMetre.BitisMetre - GirisKaliteKontrolHataMetre.BaslangicMetre),0) from GirisKaliteKontrolHataMetre
      where GirisKaliteKontrolHataMetre.GirisKaliteKontrolId = :GirisKaliteKontrolId into :FHataToplamMetre;

      Hatalar = Hatalar||FHataTipiAdi||'('||FAdet||','||FHataToplamMetre||')'||' ';
    end

    if (AHataDurumu = 0 or (AHataDurumu = 1 and Hatalar <> '') or (AHataDurumu = 2 and Hatalar = '')) then
      suspend;
  end
end
