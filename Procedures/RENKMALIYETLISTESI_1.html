CREATE PROCEDURE RENKMALIYETLISTESI_1(
    ARECETELER VARCHAR(200),
    ARAPORDOVIZID VARCHAR(5),
    AFIYATTIPI INTEGER)
RETURNS (
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMADI VARCHAR(255),
    MUSTERIRENKKODU VARCHAR(30),
    RECETETURUID INTEGER,
    RECETETURUADI VARCHAR(30),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI VARCHAR(20),
    GENELGIDER DOUBLE PRECISION,
    BOYAMIKTARI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    ACIKLAMA VARCHAR(40),
    ACIKLAMA2 VARCHAR(40))
AS
declare variable AtisNoStr varchar(20);
declare variable GenelGiderDovizId  varchar(5);
declare variable VirgulPos  integer;
declare variable Str        varchar(20);
declare variable DovizKuru  Double Precision;
begin
  VirgulPos = PosBul(AReceteler, ',');
  DovizID = ARaporDovizId;

  While (VirgulPos > 1) do
  begin
    Str = EgemenCopyStr(AReceteler, 1, VirgulPos -1);
    LabReceteAtisId = EgemenIntegerToStr(Str);

    Select LabRecete.LabReceteId, LabRecete.LabReceteKodu, LabRecete.Adi,  LabRecete.FirmaId, Firma.Adi, Ham.Adi, LabRecete.ReceteTuruId,  ReceteTuru.Adi,
           LabRecete.RenkDerinligiId, RenkDerinligi.Adi, ReceteTuru.GenelGider, ReceteTuru.GenelGiderDovizId, LabReceteAtis.BoyaYuzdesi,
           LabReceteAtis.AtisNoStr,
           LabRecete.Aciklama, LabRecete.Aciklama2, LabRecete.MusteriRenkKodu from LabReceteAtis
      left join LabRecete on LabRecete.LabReceteID = LabReceteAtis.LabReceteID
      left join Firma on Firma.FirmaID = LabRecete.FirmaID
      left join ReceteTuru on ReceteTuru.ReceteTuruID = LabRecete.ReceteTuruID
      left join RenkDerinligi on RenkDerinligi.RenkDerinligiID = LabRecete.RenkDerinligiID
      left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaID = LabReceteAtis.LabReceteCalismaID
      left join Ham on Ham.HamID = LabReceteCalisma.HamID
      where LabReceteAtis.LabReceteAtisID = :LabReceteAtisID
      into :LabReceteId, :LabReceteKodu, :LabReceteAdi,  :FirmaId, :FirmaAdi, :HamAdi, :ReceteTuruId,  :ReceteTuruAdi, :RenkDerinligiId, :RenkDerinligiAdi,
           :GenelGider, :GenelGiderDovizId, :BoyaMiktari, :AtisNoStr, :Aciklama, :Aciklama2, :MusteriRenkKodu;

    LabReceteKodu = LabReceteKodu || '.' || AtisNoStr;


    if (GenelGiderDovizID <> ARaporDovizId) then
    begin
      select DovizKuru from FirmaDovizKurlariniGetir(:FirmaId, :GenelGiderDovizID, :ARaporDovizId, Current_date) into :DovizKuru;
      GenelGider = GenelGider * DovizKuru;
    end

    suspend;
    if (Char_Length(AReceteler) > VirgulPos) then AReceteler = EgemenCopyStr(AReceteler, VirgulPos + 1, Char_Length(AReceteler) - VirgulPos);
    else AReceteler = '';

    VirgulPos = PosBul(AReceteler, ',');
  end
end
