  where BoyaSiparisDetay.BoyaSiparisDetayid is not null and
       (OrmeSiparisRaporu_1.MusteriSiparisNo = :AMusteriSiparisNo or BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null)
  into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :FOrmeSiparisSekliId,  :HamId, :HamAdi, :HamTipiId, :RenkId, :RenkAdi, :FRenkAdi, :SiparisKilo, :LabReceteKodu,
       :LabReceteAtisNo, :LabReceteAdi, :IslemTipiId, :IslemTipiAdi, :SiparisProsesId, :SiparisProsesAdi, :IlaveIslemler, :PartilenenKilo, :IptalKilo, :FirmaId,
       :BirimFiyat, :DovizId, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi, :MelanjKodu, :ToplamBoyayaCikis, :ToplamBoyaIade,
       :BaskaSipariseKaydirilan, :BaskaSiparistenKaydirilan,
       :Kapali, :UcretliTamir, :MaximumFireOrani, :SatisFireOngoruOrani,  :TerminTarihi, :RevizeTerminTarihi, :IstenenRaporBoyu, :IstenenOzelTuse, :EnIstenen, :GramajIstenen, :DonmeIstenen, :SansIstenen,
       :CekmezlikEnIstenen, :CekmezlikBoyIstenen, :YikamaHasligiIstenen, :BoyaSiparisRenkNo, :BoyaSiparisRenkRenkAdi,
       :MusteriSiparisNo, :ReferansSiparisNo, :FTerminNotu, :FNotu, :FPlanlamaNotu, :DesenId, :DesenAdi,
       :ProsesBirimMaliyet, :RenkDerinligiBirimMaliyet, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :SiparisBrutKilo,
       :YikamaDereceId, :YikamaDereceAdi, :KurutmaSekliId, :KurutmaSekliAdi, :YikamaHasligiTestiId, :YikamaHasligiTestiAdi, :BaskiBirimFiyati, :BaskiBirimFiyatDovizID, :IkinciKaliteBirimFiyat,
       :FasonBirimFiyati, :FasonBirimFiyatDovizID, :FaturaEdilecekMiktarTipi, :HamStokBirimId, :KapatmaTarihi, :BaskiTipiID, :BaskiTipiAdi, :PartilenenKiloBaski, :IptalKiloBaski,
       :SiparisDetayGrupId, :SiparisGrupAdi do
  begin
    KapaliStr = '';
    if (Kapali = 1) then KapaliStr = 'Kapal';
    else if (Kapali = 2) then KapaliStr = 'ptal';

    if (RenkId is null) then RenkAdi = FRenkAdi;

    if (IslemTipiID = itTamir and UcretliTamir = 1) then UcretliTamirStr = 'Evet';
    else UcretliTamirStr = '';

    if (FOrmeSiparisSekliId = 0 and ASatisBoyaSiparisFiyatGorebilir = 0) then
    begin
      BirimFiyat = null;
      DovizId    = null;
      BaskiBirimFiyati       = Null;
      BaskiBirimFiyatDovizID = Null;
      IkinciKaliteBirimFiyat = Null;
      FasonBirimFiyati       = Null;
      FasonBirimFiyatDovizID = Null;
    end
    else if (FOrmeSiparisSekliId <> 0 and AFasonBoyaSiparisFiyatGorebilir = 0) then
    begin
      BirimFiyat = null;
      DovizId    = null;
      BaskiBirimFiyati       = Null;
      BaskiBirimFiyatDovizID = Null;
      IkinciKaliteBirimFiyat = Null;
      FasonBirimFiyati       = Null;
      FasonBirimFiyatDovizID = Null;
    end

    BoyahaneBrutSevkKilo = 0;
    BoyahaneNetSevkKilo = 0;

    BoyahanedenGelenBrutToplam = 0;
    BoyahanedenGelenNetToplam = 0;
    BoyahanedenGelenBrutToplamK1 = 0;
    BoyahanedenGelenNetToplamK1 = 0;

    BskBoyahanedenGelenBrutToplam = 0;
    BskBoyahanedenGelenNetToplam = 0;
    BskBoyahanedenGelenBrutToplamK1 = 0;
    BskBoyahanedenGelenNetToplamK1 = 0;
    BrutSevkKilo = 0;
    NetSevkKilo = 0;
    BoyahaneSevkeHazirKilo = 0;

    Select Sum(BoyaIrsaliyeDetay.BrutKilo), Sum(BoyaIrsaliyeDetay.Kilo) from BoyaSiparisDetay
    left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and BoyaIrsaliye.FirmaID <> :OrmeFirmaKodumuz
    into :BoyahaneBrutSevkKilo, :BoyahaneNetSevkKilo;

    if (BoyahaneBrutSevkKilo is null) then BoyahaneBrutSevkKilo = 0;
    if (BoyahaneNetSevkKilo is null) then BoyahaneNetSevkKilo = 0;

    Select sum(PartiStok.Kilo - PartiStok.BrutSevkKilo) from BoyaSiparisDetay
    left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
    where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and PartiStok.Durumu = 7 and PartiStok.PartiStokIptalKayitID is null
    into :BoyahaneSevkeHazirKilo;

    Select Sum(StokHareket.BrutMiktar), Sum(StokHareket.Miktar) from BoyaSiparisDetay
    left join StokHareket on StokHareket.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
    where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and
          StokHareket.HareketTipi = :shtSatis and StokHareket.HamIade = 0
    into :BrutSevkKilo, :NetSevkKilo;

    if (BrutSevkKilo is null) then BrutSevkKilo = 0;
    if (NetSevkKilo is null) then NetSevkKilo = 0;

    BrutSevkKilo = BoyahaneBrutSevkKilo + BrutSevkKilo;
    NetSevkKilo = BoyahaneNetSevkKilo + NetSevkKilo;

    Select Sum(StokHareket.Miktar) from BoyaSiparisDetay
    left join StokHareket on StokHareket.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
    where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and
          StokHareket.HareketTipi = :shtSatisIade
    into :ToplamSatisIade;

    ToplamSatisIadeIplik = Null;

    Select Sum(IplikIrsaliyeDetay.Miktar) from IplikTalepDetay
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayID = IplikTalepDetay.IplikTalepDetayId
      left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
      where IplikTalepDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and IplikIrsaliyeDetay.IplikIrsaliyeDetayID is not null and
            IplikIrsaliyeDetay.hareketTipi = :ShtSatisIadeIplik
      into :ToplamSatisIadeIplik;

    for Select StokHareket.KaliteNo, Sum(StokHareket.BrutMiktar), Sum(StokHareket.Miktar) from BoyaSiparisDetay
    left join StokHareket on StokHareket.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
    where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and StokHareket.BoyaSiparisBaskiId is null and
          (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :ShtUretim) and StokHareket.HamIade = 0
    group by StokHareket.KaliteNo
    into :FKaliteNo, :FBrutToplam, :FNetToplam do
    begin
      BoyahanedenGelenBrutToplam = BoyahanedenGelenBrutToplam + FBrutToplam;
      BoyahanedenGelenNetToplam = BoyahanedenGelenNetToplam + FNetToplam;

      if (FKaliteNo = 1) then
      begin
        BoyahanedenGelenBrutToplamK1 = BoyahanedenGelenBrutToplamK1  + FBrutToplam;
        BoyahanedenGelenNetToplamK1 = BoyahanedenGelenNetToplamK1 + FNetToplam;
      end
    end

    for Select StokHareket.KaliteNo, Sum(StokHareket.BrutMiktar), Sum(StokHareket.Miktar) from BoyaSiparisDetay
    left join StokHareket on StokHareket.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
    where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and StokHareket.BoyaSiparisBaskiId is not null and
          (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :ShtUretim) and StokHareket.HamIade = 0
    group by StokHareket.KaliteNo
    into :FKaliteNo, :FBrutToplam, :FNetToplam do
    begin
      BskBoyahanedenGelenBrutToplam = BskBoyahanedenGelenBrutToplam + FBrutToplam;
      BskBoyahanedenGelenNetToplam = BskBoyahanedenGelenNetToplam + FNetToplam;

      if (FKaliteNo = 1) then
      begin
        BskBoyahanedenGelenBrutToplamK1 = BskBoyahanedenGelenBrutToplamK1  + FBrutToplam;
        BskBoyahanedenGelenNetToplamK1 = BskBoyahanedenGelenNetToplamK1 + FNetToplam;
      end
    end

    Select sum(Miktar), sum(NetMiktar), sum(Kilo), sum(NetKilo) from BoyaSiparisDetay
       left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
       where BoyaSiparisDetay.BoyaSiparisDetayID = :BoyaSiparisDetayID and PartiStok.TamamiTartildi = 1
       into :FBrutMiktar, :FNetMiktar, :FBrutKilo, :FNetKilo;

    if (FNetMiktar <> 0 and FBrutMiktar <> 0) then
    begin
      FSiparisFire = FBrutMiktar - FNetMiktar;

      if (FSiparisFire <> 0) then
      begin
         SiparisFireOrani = (FSiparisFire / FBrutMiktar) * 100;
         SiparisFireOrani = Yuvarla(SiparisFireOrani, 100);
      end
      else SiparisFireOrani = 0;
    end else SiparisFireOrani = Null;

    IlkSevkTarihi = Null;
    FIlkSevkTarihi = Null;

    if (NetSevkKilo > 0) then
    begin
      Select Min(StokHareket.Tarih) from BoyaSiparisDetay
         left join StokHareket on StokHareket.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
         where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and
               StokHareket.HareketTipi = 2 into :IlkSevkTarihi;

      Select Min(BoyaIrsaliyeDetay.IrsaliyeTarihi) from BoyaSiparisDetay
       left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
       left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
       where BoyaSiparisDetay.BoyaSiparisDetayID = :BoyaSiparisDetayID
       into :FIlkSevkTarihi;

      if (IlkSevkTarihi is null or IlkSevkTarihi > FIlkSevkTarihi) then IlkSevkTarihi = FIlkSevkTarihi;
    end

    if (BoyaSiparisRenkNo is not null) then
      BoyaSiparisRenkNoStr = '(' || EgemenIntToLzStr(BoyaSiparisRenkNo, 2) || ') - ' || BoyaSiparisRenkRenkAdi;
    else
      BoyaSiparisRenkNoStr = '';

    BlobSize = char_length(FTerminNotu);

    if (BlobSize > 1023) then TerminNotu = Left(FTerminNotu, 1023);
    else if (BlobSize > 0) then TerminNotu = Left(FTerminNotu, BlobSize);
    else TerminNotu = '';

    BlobSize = char_length(FNotu);

    if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
    else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
    else Notu = '';

    BlobSize = char_length(FPlanlamaNotu);

    if (BlobSize > 1023) then PlanlamaNotu = Left(FPlanlamaNotu, 1023);
    else if (BlobSize > 0) then PlanlamaNotu = Left(FPlanlamaNotu, BlobSize);
    else PlanlamaNotu = '';

    IplikTalepMiktari = 0;

    Select Sum(Miktar) from IplikTalepDetay where IplikTalepDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId into :IplikTalepMiktari;

    IplikTalepIdStr = '';

    for Select IpliKTalep.IplikTalepYil, IpliKTalep.IplikTalepId from IplikTalepDetay
      left join IpliKTalep on IpliKTalep.IpliKTalepKayitID = IpliKTalepDetay.IpliKTalepKayitId
      where IplikTalepDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
      group by IpliKTalep.IplikTalepYil, IpliKTalep.IplikTalepId
      into :FIplikTalepYil, FIplikTalepId do
    begin
      if (IplikTalepIdStr <> '') then IplikTalepIdStr = IplikTalepIdStr || cast(', ' as varchar(2) character set UTF8);

      IplikTalepIdStr = IplikTalepIdStr ||  '(' || EgemenIntegerToStr(FIplikTalepYil) || ') ' || EgemenIntegerToStr(FIplikTalepId);
    end

    IlaveIslemlerMaliyeti = Null;

    Select Sum(BirimMaliyet) from SiparisIlaveIslemler where SiparisIlaveIslemler.BoyaSiparisDetayId = :BoyaSiparisDetayId into :IlaveIslemlerMaliyeti;

    SatisEksigi = SiparisKilo - NetSevkKilo;


    IplikIlkSevkTarihi = Null;
    IplikSonSevkTarihi = Null;
    Select Min(IplikIrsaliyeDetay.Tarih),MAx(IplikIrsaliyeDetay.tarih) from IplikTalepDetay
         left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayID= IplikTalepDetay.IplikTalepDetayID
         where IplikTalepDetay.OrmeSiparisDetayID=:OrmeSiparisDetayID and IplikIrsaliyeDetay.IplikIrsaliyeDetayID is not null and
               IplikIrsaliyeDetay.HareketTipi = :shtSatisIplik
    into :IplikIlkSevkTarihi, :IplikSonSevkTarihi;

    if (PartilenenMiktardanIptaliDus = 1) then
    begin
      PartilenenKilo = PartilenenKilo - IptalKilo;
      PartilenenKiloBaski = PartilenenKiloBaski - IptalKiloBaski;
    end

    suspend;
  end
end
