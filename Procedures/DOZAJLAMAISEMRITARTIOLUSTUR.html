CREATE PROCEDURE DOZAJLAMAISEMRITARTIOLUSTUR(
    ADOZAJLAMAISEMRIID INTEGER,
    FURUNTIPIID INTEGER)
AS
declare variable FPartiKayitId         integer;
declare variable FReceteId             varchar(20);
declare variable FPartiAsamalariId     integer;
declare variable FReceteIsEmriYil      Integer;
declare variable FReceteIsEmriId       Integer;
declare variable FPartiSablonIsEmriId  Integer;
declare variable ReferansPartiKayitId  Integer;
declare variable FPartiSablonIlaveId   Integer;
declare variable FSablonNo             Integer;
declare variable FStoktanDusuldu       SmallInt;

declare variable FUrunId          varchar(20);
declare variable FEmulsiyonUrunId varchar(20);
declare variable FUrunBirimId     integer;
declare variable FReceteMiktar    double precision;
declare variable FPlanlananMiktar double precision;
declare variable FBidonSayisi     integer;
declare variable FBidonKapasite     double precision;
declare variable Say              integer;
declare variable FIslem          numeric(15,2);
declare variable FIslemOran      numeric(15,2);

declare variable ToplamBrMiktar double precision;
declare variable ToplamKullanilanMiktar double precision;
declare variable FHazirlananMiktar double precision;

declare variable BaskiTartimAktif integer;
declare variable FSablonId integer;
declare variable SablonHazirlananMiktar double precision;
declare variable BirimKullanilanMiktar double precision;
declare variable KullanilanMiktar double precision;
declare variable FUrunBirimdId integer;
declare variable FAkuple integer;
declare variable FBolum integer;
begin
  select sistem.BaskiTartimAktif  from sistem into :BaskiTartimAktif;

  Select PartiAsamalariId, ReceteIsEmriYil, ReceteIsEmriId, PartiKayitId, PartiSablonIsEmriId, StoktanDusuldu from DozajlamaIsEmri
    where DozajlamaIsEmri.DozajlamaIsEmriID = :ADozajlamaIsEmriID
    into :FPartiAsamalariId, FReceteIsEmriYil, FReceteIsEmriId, :FPartiKayitId, :FPartiSablonIsEmriId, :FStoktanDusuldu;

  if (FPartiAsamalariId is Not Null or FReceteIsEmriYil is Not Null or FReceteIsEmriId is Not Null) then
  begin
    if (FPartiAsamalariID is null) then
    begin
      Select Min(PartiAsamalariId) from ReceteIsEmriDetay
        where ReceteIsEmriDetay.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :FReceteIsEmriId
        into :FPartiAsamalariId;
    end

    Select PartiAsamalari.PartiKayitId, PartiAsamalari.ReceteId, PartiAsamalari.Akuple from PartiAsamalari
      where PartiAsamalari.PartiAsamalariID = :FPartiAsamalariId
      into :FPartiKayitId, :FReceteId, :FAkuple;

    FBolum = 1;
    if (FAkuple = 1) then FBolum = 2;

    if (FReceteID is not null) then
      insert into DozajlamaIsEmriTarti(DozajlamaIsEmriId, UrunId, UrunBirimId, ReceteMiktar, PlanlananMiktar)
       select :ADozajlamaIsEmriId, PartiRecetesiDokumu_2.UrunId, PartiRecetesiDokumu_2.UrunBirimId, Coalesce(ReceteMiktar, 0)+Coalesce(ReceteBanyo, 0), (Coalesce(Miktar, 0)+Coalesce(Banyo, 0)) / :FBolum
       from PartiRecetesiDokumu_2(:FPartiKayitId, :FReceteId, :FPartiAsamalariId, 1)
         left join Urun on Urun.UrunID = PartiRecetesiDokumu_2.UrunID
         where PartiRecetesiDokumu_2.UrunId is not null and (PartiRecetesiDokumu_2.UrunTipiId = :FUrunTipiId or Urun.TartimYapilacak = 1);

   if (FReceteID is not null and FAkuple =1) then
      insert into DozajlamaIsEmriTarti(DozajlamaIsEmriId, UrunId, UrunBirimId, ReceteMiktar, PlanlananMiktar)
       select :ADozajlamaIsEmriId, PartiRecetesiDokumu_2.UrunId, PartiRecetesiDokumu_2.UrunBirimId, Coalesce(ReceteMiktar, 0)+Coalesce(ReceteBanyo, 0), (Coalesce(Miktar, 0)+Coalesce(Banyo, 0)) / :FBolum
       from PartiRecetesiDokumu_2(:FPartiKayitId, :FReceteId, :FPartiAsamalariId, 1)
         left join Urun on Urun.UrunID = PartiRecetesiDokumu_2.UrunID
         where PartiRecetesiDokumu_2.UrunId is not null and (PartiRecetesiDokumu_2.UrunTipiId = :FUrunTipiId or Urun.TartimYapilacak = 1);
  end
  else
  begin
    if (FStoktanDusuldu = 0) then
    begin
      if (FPartiSablonIsEmriId is Not Null) then
      begin
        Select PartiSablonIsEmri.ReferansPartiKayitId from PartiSablonIsEmri Where PartiSablonIsEmri.PartiSablonIsEmriId = :FPartiSablonIsEmriId into :ReferansPartiKayitId;

        Execute Procedure PARTISABLONISEMRIMIKTARHESAPLA(:FPartiSablonIsEmriId, :ReferansPartiKayitId,0 , 1);
      end
      else
      begin
        Execute Procedure PARTISABLONACILIMIHESAPLA(FPartiKayitId);
      end

      if (BaskiTartimAktif = 0) then
      begin
        insert into DozajlamaIsEmriTarti(DozajlamaIsEmriId, UrunId, UrunBirimId, ReceteMiktar, PlanlananMiktar, SablonNo)
        select :ADozajlamaIsEmriId, PartiSablonRecete.UrunId, PartiSablonRecete.UrunBirimId, Sum(PartiSablonRecete.Miktar), Sum(PartiSablonRecete.Kullanilanmiktar), PartiSablon.SablonNo from PartiSablon
          left join PartiSablonRecete on PartiSablonRecete.PartiSablonId = PartiSablon.PartiSablonId
          where (PartiSablon.PartiKayitId = :FPartiKayitId or (PartiSablon.PartiSablonIsEmriID = :FPartiSablonIsEmriId)) and PartiSablonRecete.PartiSablonId is not null
          group by PartiSablonRecete.UrunId, PartiSablonRecete.UrunBirimId, PartiSablon.SablonNo;
      end
      else
      begin
        if (FPartiKayitId is not null) then
        begin
          for select PartiSablon.partisablonid, sum(PartiSablonRecete.Miktar), Sum(PartiSablonRecete.Kullanilanmiktar),PartiSablon.hazirlananmiktar, PartiSablon.bidonsayisi,
                 PartiSablon.bidonkapasite from PartiSablon
            left join PartiSablonRecete on PartiSablonRecete.PartiSablonId = PartiSablon.PartiSablonId
            where (PartiSablon.PartiKayitId = :FPartiKayitId or (PartiSablon.PartiSablonIsEmriID = :FPartiSablonIsEmriId)) and PartiSablonRecete.PartiSablonId is not null
            group by PartiSablon.partisablonid, PartiSablon.hazirlananmiktar, PartiSablon.bidonsayisi, PartiSablon.bidonkapasite
            into :FSablonId, :ToplamBrMiktar, :ToplamKullanilanMiktar, :FHazirlananMiktar, :FBidonSayisi, :FBidonKapasite do
        begin

        say = 0;

        while (say<FBidonSayisi) do
        begin
          say = say +1;

          if (FHazirlananMiktar>FBidonKapasite) then
          begin
            FhazirlananMiktar = FhazirlananMiktar -FBidonKapasite;
          end
          else FBidonKapasite =FhazirlananMiktar;

          insert into DozajlamaIsEmriTarti(DozajlamaIsEmriId, UrunId, UrunBirimId, ReceteMiktar, PlanlananMiktar, SablonNo,BidonNo)
            select :ADozajlamaIsEmriId, PartiSablonRecete.UrunId, PartiSablonRecete.UrunBirimId, Sum(PartiSablonRecete.Miktar),Sum(PartiSablonRecete.Kullanilanmiktar)/ :ToplamKullanilanMiktar *:FBidonKapasite*1000, PartiSablon.SablonNo,:Say from PartiSablon
            left join PartiSablonRecete on PartiSablonRecete.PartiSablonId = PartiSablon.PartiSablonId
            where (PartiSablon.PartiKayitId = :FPartiKayitId or (PartiSablon.PartiSablonIsEmriID = :FPartiSablonIsEmriId)) and PartiSablonRecete.PartiSablonId is not null
               and PartiSablon.partisablonid = :FSablonId
            group by PartiSablonRecete.UrunId, PartiSablonRecete.UrunBirimId, PartiSablon.SablonNo;
        end
        end
      end
        else
        if (FPartiSablonIsEmriId is not null) then
        begin
          for select PartiSablon.Sablonno, sum(PartiSablonRecete.Miktar), Sum(PartiSablonRecete.Kullanilanmiktar) from PartiSablon
                 left join PartiSablonRecete on PartiSablonRecete.PartiSablonId = PartiSablon.PartiSablonId
                 where (PartiSablon.PartiKayitId = :FPartiKayitId or (PartiSablon.PartiSablonIsEmriID = :FPartiSablonIsEmriId)) and PartiSablonRecete.PartiSablonId is not null
                 group by PartiSablon.Sablonno
                 into :FSablonNo, :ToplamBrMiktar, :ToplamKullanilanMiktar do
          begin

            select PartiSablon.hazirlananmiktar, PartiSablon.bidonsayisi, PartiSablon.bidonkapasite from PartiSablon
              where (PartiSablon.PartiKayitId = :FPartiKayitId or (PartiSablon.PartiSablonIsEmriID = :FPartiSablonIsEmriId))
                     and PartiSablon.bidonsayisi> 0 and  PartiSablon.Sablonno = :FSablonNo
                     into :FHazirlananMiktar, :FBidonSayisi, :FBidonKapasite;
            say=0;
            FHazirlananMiktar =ToplamKullanilanMiktar/1000;
            while (say<FBidonSayisi) do
            begin
              if (FHazirlananMiktar>FBidonKapasite) then
                FhazirlananMiktar = FhazirlananMiktar -FBidonKapasite;
              else FBidonKapasite =FhazirlananMiktar;
                say = say +1;

             insert into DozajlamaIsEmriTarti(DozajlamaIsEmriId, UrunId, UrunBirimId, ReceteMiktar, PlanlananMiktar, SablonNo,BidonNo)
             select :ADozajlamaIsEmriId,PartiSablonRecete.UrunId, PartiSablonRecete.UrunBirimId, PartiSablonRecete.Miktar,
                        Sum(PartiSablonRecete.Kullanilanmiktar)/ :ToplamKullanilanMiktar *:FBidonKapasite * 1000,:FSablonNo,:say from PartiSablon
                    left join PartiSablonRecete on PartiSablonRecete.PartiSablonId = PartiSablon.PartiSablonId
                    where (PartiSablon.PartiKayitId = :FPartiKayitId or (PartiSablon.PartiSablonIsEmriID = :FPartiSablonIsEmriId)) and PartiSablonRecete.PartiSablonId is not null
                    and PartiSablon.sablonNo = :FSablonNo
                    group by PartiSablonRecete.UrunId, PartiSablonRecete.UrunBirimId,PartiSablonRecete.Miktar, PartiSablon.SablonNo;
            end
          end
        end
      end
    end
    else
    begin
      for Select PartiSablonIlaveId, SablonNo from PartiSablonIlave
        where ((PartiSablonIlave.PartiKayitId = :FPartiKayitId) or (PartiSablonIlave.PartiSablonIsEmriID = :FPartiSablonIsEmriId)) and PartiSablonIlave.StoktanDusuldu = 0
        into :FPartiSablonIlaveId, :FSablonNo do
      begin
        Execute Procedure PartiSablonIlaveAcilimiHesapla(FPartiKayitId, FPartiSablonIlaveId);

        Delete From DozajlamaIsEmriTarti where DozajlamaIsEmriTarti.PartiSablonIlaveId = :FPartiSablonIlaveId;

        insert into DozajlamaIsEmriTarti(DozajlamaIsEmriId, UrunId, UrunBirimId, ReceteMiktar, PlanlananMiktar, SablonNo, Ilave, PartiSablonIlaveId)
          select :ADozajlamaIsEmriId, PartiSablonReceteIlave.UrunId, PartiSablonReceteIlave.UrunBirimId, Sum(PartiSablonReceteIlave.Miktar), Sum(PartiSablonReceteIlave.Kullanilanmiktar), :FSablonNo, 1, :FPartiSablonIlaveId from PartiSablonReceteIlave
            where PartiSablonIlaveId = :FPartiSablonIlaveId
            group by UrunId, UrunBirimId;

      end

    end
  end
end
