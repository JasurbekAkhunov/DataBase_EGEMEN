CREATE PROCEDURE LABRECETEATISLISTESI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTALEPTARIHI TIMESTAMP,
    ABITISTALEPTARIHI TIMESTAMP,
    ALABRECETEGRUPID INTEGER,
    AGRUPIPLIKBOYAMI INTEGER,
    ALABRENKGRUPID INTEGER,
    AFASON INTEGER,
    AESKILABRECETEKODU VARCHAR(20),
    APANTONEKODU VARCHAR(20),
    AHAMID INTEGER,
    AKALITEID VARCHAR(20),
    AURUNFIYATGOREBILIR SMALLINT,
    ALABRECETEKODU VARCHAR(20),
    ALABRECETETALEPYIL INTEGER,
    ALABRECETETALEPID INTEGER)
RETURNS (
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEKODUSTR VARCHAR(30),
    LABRECETEATISID INTEGER,
    CALISMATARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    CALISMABITISTARIHI TIMESTAMP,
    KARTELATARIHI TIMESTAMP,
    ATISNOSTR VARCHAR(20),
    RPTMI INTEGER,
    RPTSTR VARCHAR(3),
    ILAVE INTEGER,
    SOKUMUZERIBOYA INTEGER,
    ILAVESOKUMUZERIBOYASTR VARCHAR(20),
    KRITIKCALISMASI INTEGER,
    KRITIKCALISMASISTR VARCHAR(10),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    BOYAMALIYETI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    SDBOYAMALIYETI DOUBLE PRECISION,
    SISTEMDOVIZID VARCHAR(5),
    BOYAYUZDESI DOUBLE PRECISION,
    HESAPBIRIMID VARCHAR(5),
    GIDISTARIHI TIMESTAMP,
    DURUMUSTR VARCHAR(6),
    TARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    RECETEASAMAADI VARCHAR(40),
    ISIKHASLIGI VARCHAR(50))
AS
declare variable LastDovizId        varchar(5);
declare variable DovizKuru          Double precision;
begin
  Select STOKTABANDOVIZID from Sistem where SistemId = 1 into :SistemDovizId;
  LastDovizId = '';

  for Select LabReceteListesi.LabReceteId, LabReceteListesi.LabReceteKodu, LabReceteAtis.LabReceteAtisId, LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi, LabReceteCalisma.BitisTarihi,
    LabReceteAtis.AtisNoStr, LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi,
    LabReceteCalisma.HamId, Ham.Adi, LabReceteAtis.GidisTarihi, LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi,
    LabReceteAtis.KartelaTarihi, LabReceteAtis.BoyaMaliyeti, LabReceteAtis.BoyaYuzdesi, LabReceteGrup.Caption,
    IK.Adi, LabReceteAtis.InsertTarihi, SK.Adi, LabReceteAtis.DegisimTarihi, ReceteAsama.Adi,
    LabReceteListesi.DovizId, LabReceteAtis.IsikHasligi from LabReceteListesi(:AFirmaId, :ABaslangicTalepTarihi, :ABitisTalepTarihi, :ALabReceteGrupid, :AGrupIplikBoyami, :ALabRenkGrupid, :AFason, :AEskiLabReceteKodu, :APantoneKodu, :AHamId, :AKaliteId, :ALabReceteKodu, :ALabReceteTalepYil, :ALabReceteTalepId)
  left join LabReceteGrup on LabReceteGrup.LabReceteGrupId = LabReceteListesi.LabReceteGrupId
  Left Join LabReceteAtis on LabReceteAtis.LabReceteId = LabReceteListesi.LabReceteId
  Left Join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
  left join Ham On Ham.HamId = LabReceteCalisma.HamId
  Left Join Kullanici IK on IK.KullaniciId = LabReceteAtis.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = LabReceteAtis.KullaniciId
  left join ReceteAsama on ReceteAsama.ReceteAsamaId = LabReceteAtis.ReceteAsamaId
  where LabReceteAtis.LabReceteAtisId is not null
  into :LabReceteId, :LabReceteKodu, :LabReceteAtisId, :CalismaTarihi, :TerminTarihi, :CalismaBitisTarihi, :AtisNoStr, :Rptmi, :Ilave, :SokumUzeriBoya,
       :KritikCalismasi, :HamId, :HamAdi, :GidisTarihi,
       :DurumuStr, :Tarihi, :KartelaTarihi, :BoyaMaliyeti, :BoyaYuzdesi, :HesapBirimId, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :ReceteAsamaAdi,
       :DovizId, :IsikHasligi do
  begin
    LabReceteKoduStr = LabReceteKodu || '.' || AtisNoStr;

    if (Rptmi = 1) then RptStr = 'Rpt';
    else RptStr = '';

    if (KritikCalismasi = 1) then KritikCalismasiStr = 'Kritik';
    else KritikCalismasiStr = '';

    if (Ilave = 1) then IlaveSokumUzeriBoyaStr = 'İlave';
    else if (SokumUzeriBoya = 1) then IlaveSokumUzeriBoyaStr = 'Söküm Üzeri Boya';
    else IlaveSokumUzeriBoyaStr = '';

    if (LastDovizId <> DovizId) then
    begin
      if (DovizId <> SistemDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :SistemDovizId, Current_Date) into :DovizKuru;
          else DovizKuru = 1;

      if (DovizKuru is null) then DovizKuru = 0;
      LastDovizId = DovizId;
    end
    SDBoyaMaliyeti = BoyaMaliyeti * DovizKuru;

    if (AUrunFiyatGorebilir = 0) then
    begin
      SDBoyaMaliyeti = Null;
      BoyaMaliyeti = Null;
    end
    suspend;
  end
end
