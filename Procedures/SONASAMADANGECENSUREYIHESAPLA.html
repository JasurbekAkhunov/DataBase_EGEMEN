CREATE PROCEDURE SONASAMADANGECENSUREYIHESAPLA(
    APARTISINIFID INTEGER)
RETURNS (
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30))
AS
begin
  GecenSure    = null;
  GecenSureStr = null;

  select current_timestamp - Max(CikisTarihi), TarihToStr(current_timestamp - Max(CikisTarihi)) from PartiAsamaStok
  where  PartiAsamaStok.PartiSinifId = :APartiSinifID
  into   :GecenSure, :GecenSureStr;

  suspend;
end
