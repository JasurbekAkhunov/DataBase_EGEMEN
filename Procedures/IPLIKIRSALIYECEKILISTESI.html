CREATE PROCEDURE IPLIKIRSALIYECEKILISTESI(
    AIPLIKIRSALIYEID INTEGER)
RETURNS (
    IPLIKSTOKID INTEGER,
    NUMARATOR INTEGER,
    CIKISIPLIKIRSALIYEDETAYID INTEGER,
    KAPNO INTEGER,
    BOBINSAYISI INTEGER,
    MIKTAR DOUBLE PRECISION,
    NETMIKTAR DOUBLE PRECISION,
    BOLUNDUGUIPLIKSTOKID INTEGER)
AS
begin
  For Select IplikStok.IplikStokId, IplikStok.Numarator, IplikStok.CikisIplikIrsaliyeDetayId, IplikStok.KapNo, IplikStok.BobinSayisi, IplikStok.Miktar,
             IplikStok.NetMiktar, IplikStok.BolunduguIplikStokID from IplikIrsaliyeDetay
    left join IplikStok on IplikStok.CikisIplikIrsaliyeDetayId = IplikIrsaliyeDetay.IplikIrsaliyeDetayId
    left join Ambalaj on Ambalaj.AmbalajID = IplikIrsaliyeDetay.AmbalajId
    where IplikIrsaliyeDetay.IplikIrsaliyeId = :AIplikIrsaliyeId and IplikStok.IplikStokID is not null
    order by IplikStok.CikisIplikIrsaliyeDetayId, IplikStok.KapNo
  into :IplikStokId, :Numarator, :CikisIplikIrsaliyeDetayId, :KapNo, :BobinSayisi,  :Miktar, :NetMiktar, :BolunduguIplikStokID do
  begin

    suspend;
  end
end
