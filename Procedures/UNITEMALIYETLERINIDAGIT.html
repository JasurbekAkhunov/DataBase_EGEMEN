CREATE PROCEDURE UNITEMALIYETLERINIDAGIT(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    UNITEID INTEGER,
    UNITEADI ADI30,
    PARTIASAMASTOKDETAYID INTEGER,
    MAKINAID MAKINAKODUSIZE,
    PARTIASAMALARIID INTEGER,
    SIRA INTEGER,
    ASAMACIKISTARIHI TIMESTAMP,
    ASAMAID INTEGER,
    ASAMAADI ADI40,
    PARTITAMIRID INTEGER,
    RECETEID RECETEKODUSIZE,
    GENELURETIMGIDERINDENPAYALIR INTEGER,
    ELEKTRIKMALIYETINDENPAYALIR INTEGER,
    DOGALGAZMALIYETINDENPAYALIR INTEGER,
    BUHARMALIYETINDENPAYALIR INTEGER,
    SUMALIYETINDENPAYALIR INTEGER,
    ISCILIKMALIYETINDENPAYALIR INTEGER,
    GENELURETIMGIDERDAGITIMTUTARI DOUBLE PRECISION,
    ELEKTRIKDAGITIMTUTARI DOUBLE PRECISION,
    DOGALGAZDAGITIMTUTARI DOUBLE PRECISION,
    BUHARDAGITIMTUTARI DOUBLE PRECISION,
    SUDAGITIMTUTARI DOUBLE PRECISION,
    ISCILIKDAGITIMTUTARI DOUBLE PRECISION,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    IMALATTIPIID INTEGER,
    IMALATTIPIADI VARCHAR(5),
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI ADI20,
    FASON INTEGER,
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI ADI20,
    FATURAEDILECEK INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    PSKILO NUMERIC(12,3),
    SIPARISBIRIMFIYAT DOUBLE PRECISION,
    SIPARISBIRIMFIYATDOVIZID VARCHAR(5),
    SIPARISTARIHI TIMESTAMP,
    FATURABIRIMFIYAT DOUBLE PRECISION,
    FATURABIRIMFIYATDOVIZID VARCHAR(5),
    FATURATARIHI TIMESTAMP,
    BENZERPARTIKAYITID INTEGER,
    BENZERPARTIYIL INTEGER,
    BENZERPARTIID INTEGER,
    BENZERPARTISINIFID INTEGER,
    BENZERFATURABIRIMFIYAT DOUBLE PRECISION,
    BENZERFATURABIRIMFIYATDOVIZID VARCHAR(5),
    BENZERFATURATARIHI TIMESTAMP,
    DOVIZKURU DOUBLE PRECISION,
    DOVIZKURUTARIH TIMESTAMP,
    HESAPBIRIMFIYAT DOUBLE PRECISION,
    HESAPBIRIMFIYATDOVIZID VARCHAR(5),
    HESAPBIRIMFIYATTARIHI TIMESTAMP,
    STOKTABANDOVIZID VARCHAR(5),
    STOKTABANDOVIZTUTAR DOUBLE PRECISION,
    UNITEAYLIKGENELURETIMGIDERI DOUBLE PRECISION,
    UNITEAYLIKELEKTRIKGIDERI DOUBLE PRECISION,
    UNITEAYLIKDOGALGAZGIDERI DOUBLE PRECISION,
    UNITEAYLIKBUHARGIDERI DOUBLE PRECISION,
    UNITEAYLIKSUGIDERI DOUBLE PRECISION,
    UNITEAYLIKISCILIKGIDERI DOUBLE PRECISION,
    UNITEAYLIKGUGTOPLAM DOUBLE PRECISION,
    UNITEAYLIKELEKTRIKGIDERITOPLAM DOUBLE PRECISION,
    UNITEAYLIKDOGALGAZGIDERITOPLAM DOUBLE PRECISION,
    UNITEAYLIKBUHARGIDERITOPLAM DOUBLE PRECISION,
    UNITEAYLIKSUGIDERITOPLAM DOUBLE PRECISION,
    UNITEAYLIKISCILIKGIDERITOPLAM DOUBLE PRECISION,
    UNITEASAMAAYLIKGUG DOUBLE PRECISION,
    UNITEASAMAAYLIKGUGPAYITUTARI DOUBLE PRECISION,
    URETIMGUGPAYIORANI DOUBLE PRECISION,
    URETIMGUGPAYITUTARI DOUBLE PRECISION,
    UNITEASAMAAYLIKELEKTRIK DOUBLE PRECISION,
    UNITEASAMAAYLIKELEKPAYITUTARI DOUBLE PRECISION,
    URETIMELEKTRIKPAYIORANI DOUBLE PRECISION,
    URETIMELEKTRIKPAYITUTARI DOUBLE PRECISION,
    UNITEASAMAAYLIKDOGALGAZ DOUBLE PRECISION,
    UNITEASAMAAYLIKDGAZPAYITUTARI DOUBLE PRECISION,
    URETIMDOGALGAZPAYIORANI DOUBLE PRECISION,
    URETIMDOGALGAZPAYITUTARI DOUBLE PRECISION,
    UNITEASAMAAYLIKBUHAR DOUBLE PRECISION,
    UNITEASAMAAYLIKBUHARPAYITUTARI DOUBLE PRECISION,
    URETIMBUHARPAYIORANI DOUBLE PRECISION,
    URETIMBUHARPAYITUTARI DOUBLE PRECISION,
    UNITEASAMAAYLIKSU DOUBLE PRECISION,
    UNITEASAMAAYLIKSUPAYITUTARI DOUBLE PRECISION,
    URETIMSUPAYIORANI DOUBLE PRECISION,
    URETIMSUPAYITUTARI DOUBLE PRECISION,
    UNITEASAMAAYLIKISCILIK DOUBLE PRECISION,
    UNITEASAMAAYLIKISPAYITUTARI DOUBLE PRECISION,
    URETIMISCILIKPAYIORANI DOUBLE PRECISION,
    URETIMISCILIKPAYITUTARI DOUBLE PRECISION,
    PSTOPLAMTAMAMLANMAPUANI DOUBLE PRECISION,
    PSTOPLAMTAMAMLANAN DOUBLE PRECISION,
    BOYAMALIYETI DOUBLE PRECISION,
    KIMYASALMALIYETI DOUBLE PRECISION,
    BOYAKIMYASALMALIYETI DOUBLE PRECISION,
    ASAMATOPLAMKILO DOUBLE PRECISION,
    SISTEMDOVIZID DOVIZSIZE)
AS
Declare Variable FSayac Integer;
Declare Variable FKayitSayisi Integer;
Declare Variable FYil Integer;
Declare Variable FAy Integer;
Declare Variable FFisTarihi TimeStamp;
Declare Variable TempTarih1 TimeStamp;
Declare Variable TempTarih2 TimeStamp;
Declare Variable FUrunTipiId integer;
Declare Variable FMiktar Double Precision;
Declare Variable FUretimeCikisFiyati Double Precision;
Declare Variable FStokDovizId Dovizsize;
Declare Variable FDovizKuru Double Precision;
Declare Variable Kilo Double Precision;
Begin
  /*Aylık ünite giderleri tablosuna yazılan maliyetler SistemDovizID biriminden olmalı*/
  /*Birim maliyetler ise StokTabanDovizi olarak hesaplanıyor*/
  Select SistemDovizId, StokTabanDovizId From Sistem Where Sistem.Sistemid = 1 Into :SistemDovizId, :StokTabanDovizId;

  Select count(*) from UrunSarfiyatFiyatlariDuzenle(:ABaslangicTarihi, :ABitisTarihi, Null) into :FMiktar;

  For Select extract(year From PartiAsamaStokDetay.CikisTarihi) , extract(month From PartiAsamaStokDetay.CikisTarihi), Makina.UniteId
        From PartiAsamaStokDetay
        Left Join Makina on makina.MakinaId = PartiAsamaStokDetay.MakinaId
        Where PartiAsamaStokDetay.CikisTarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.CikisTarihi <= :ABitisTarihi and Makina.UniteId Is Not Null
        Group by extract(year From PartiAsamaStokDetay.CikisTarihi) , extract(month From PartiAsamaStokDetay.CikisTarihi), Makina.UniteId
  Into :FYil, :FAy, :UniteId do
  Begin
    if (SistemDovizId <> StokTabanDovizId) Then
      Select DovizKuru From EnYakinDovizKuruGetir(:SistemDovizId, :StokTabanDovizId, Current_Date) Into :FDovizKuru;
    else
      FDovizKuru = 1;

    UniteAylikGenelUretimGideri = 0 ;
    UniteAylikElektrikGideri    = 0 ;
    UniteAylikDogalGazGideri    = 0 ;
    UniteAylikBuharGideri       = 0 ;
    UniteAylikSuGideri          = 0 ;
    UniteAylikIscilikGideri     = 0 ;

    Select Sum(AylikUniteGiderleri.GenelUretimGideri) * :FDovizKuru, Sum(AylikUniteGiderleri.ElektrikGideri) * :FDovizKuru, Sum(AylikUniteGiderleri.Dogalgazgideri) * :FDovizKuru,
           Sum(AylikUniteGiderleri.BuharGideri) * :FDovizKuru, Sum(AylikUniteGiderleri.SuGideri) * :FDovizKuru, Sum(AylikUniteGiderleri.IscilikGideri) * :FDovizKuru From AylikUniteGiderleri
      Where Yil = :FYil and Ay = :FAy and UniteId = :UniteId
    Into :UniteAylikGenelUretimGideri, :UniteAylikElektrikGideri, :UniteAylikDogalGazGideri, :UniteAylikBuharGideri, :UniteAylikSuGideri, :UniteAylikIscilikGideri  ;

    If (UniteAylikGenelUretimGideri Is Null) then UniteAylikGenelUretimGideri = 0;
    If (UniteAylikElektrikGideri Is Null) then UniteAylikElektrikGideri = 0 ;
    If (UniteAylikDogalGazGideri Is Null) then UniteAylikDogalGazGideri = 0 ;
    If (UniteAylikBuharGideri Is Null) then UniteAylikGenelUretimGideri = 0 ;
    If (UniteAylikSuGideri Is Null) then UniteAylikSuGideri = 0 ;
    If (UniteAylikIscilikGideri Is Null) then UniteAylikIscilikGideri = 0 ;

    TempTarih1 = Cast('01.'||:FAy||'.'|| :FYil As Timestamp) ;
    TempTarih2 = Dateadd(1 Month To TempTarih1) ;
    Kilo = 0;
    GenelUretimGiderDagitimTutari   = 0;
    ElektrikDagitimTutari           = 0;
    DogalgazDagitimTutari           = 0;
    BuharDagitimTutari              = 0;
    SuDagitimTutari                 = 0;
    IscilikDagitimTutari            = 0;

    for Select Sum(((Select Sum(AsamaTamamlanmaPuani.TamamlanmaPuani) From PartiAsamaStok
                       Left Join AsamaTamamlanmaPuani on AsamaTamamlanmaPuani.UniteId = Makina.UniteId and AsamaTamamlanmaPuani.AsamaId = PartiAsamaStok.AsamaId
                       Left Join PartiAsamalari FF on FF.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
                       Where PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and FF.Sira <= PartiAsamalari.Sira)
                   /
                    (Select Sum(AsamaTamamlanmaPuani.TamamlanmaPuani) From PartiAsamaStok
                       Left Join AsamaTamamlanmaPuani on AsamaTamamlanmaPuani.UniteId = Makina.UniteId and AsamaTamamlanmaPuani.AsamaId = PartiAsamaStok.AsamaId
                        Where PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId and PartiAsamaStok.AsamayaDahil = 1)) * PartiAsamaStokDetay.Kilo),
       Asama.GenelUretimGiderindenPayAlir, Asama.ElektrikMaliyetindenPayAlir, Asama.DogalgazMaliyetindenPayAlir, Asama.BuharMaliyetindenPayAlir,
       Asama.SuMaliyetindenPayAlir, Asama.IscilikMaliyetindenPayAlir From PartiAsamaStokDetay
      Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
      Left Join Makina  on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
      Left Join Makina PP on PP.MakinaId = PartiAsamaStokDetay.MakinaId
      Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      Left Join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      Where PartiAsamaStokDetay.CikisTarihi >= :TempTarih1 and PartiAsamaStokDetay.CikisTarihi <= :TempTarih2 and
            Makina.UniteId = :UniteId
    Group by Asama.GenelUretimGiderindenPayAlir, Asama.ElektrikMaliyetindenPayAlir, Asama.DogalgazMaliyetindenPayAlir, Asama.BuharMaliyetindenPayAlir, Asama.SuMaliyetindenPayAlir,
             Asama.IscilikMaliyetindenPayAlir
    into :StokTabanDovizTutar, :GenelUretimGiderindenPayAlir, :ElektrikMaliyetindenPayAlir, :DogalGazMaliyetindenPayAlir, :BuharMaliyetindenPayAlir, :SuMaliyetindenPayAlir, :IscilikMaliyetindenPayAlir do
    begin
      GenelUretimGiderDagitimTutari   = GenelUretimGiderDagitimTutari + StokTabanDovizTutar * GenelUretimGiderindenPayAlir ;
      ElektrikDagitimTutari           = ElektrikDagitimTutari + StokTabanDovizTutar * ElektrikMaliyetindenPayAlir;
      DogalgazDagitimTutari           = DogalgazDagitimTutari + StokTabanDovizTutar * DogalgazMaliyetindenPayAlir;
      BuharDagitimTutari              = BuharDagitimTutari + StokTabanDovizTutar * BuharmaliyetindenPayAlir;
      SuDagitimTutari                 = SuDagitimTutari + StokTabanDovizTutar * SuMaliyetindenPayAlir;
      IscilikDagitimTutari            = IscilikDagitimTutari + StokTabanDovizTutar * IscilikMaliyetindenPayAlir;
    end

    For Select PartiAsamaStokDetay.PartiAsamaStokDetayId, Makina.UniteId, Unite.Adi, PartiAsamaStokDetay.MakinaId, PartiAsamaStokDetay.PartiAsamalariId, PartiAsamaStokDetay.CikisTarihi,
          PartiAsamalari.Sira, PartiAsamaStok.AsamaId, Asama.Adi, PartiAsamalari.PartiTamirId, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiAsamaStokDetay.PartiSinifId, Parti.ImalatTipiId, Parti.IslemGrubuId,
          Parti.Fason, Parti.IslemTipiId, PartiAsamalari.ReceteId, Parti.FaturaKesilecek, PartiStok.HamId, Ham.Adi, PartiAsamaStokDetay.Kilo,
          Asama.GenelUretimGiderindenPayAlir, Asama.ElektrikMaliyetindenPayAlir, Asama.DogalgazMaliyetindenPayAlir, Asama.BuharMaliyetindenPayAlir, Asama.SuMaliyetindenPayAlir, Asama.IscilikMaliyetindenPayAlir
      From PartiAsamaStokDetay
      Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
      Left Join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
      Left Join Unite on Unite.UniteId = Makina.UniteId
      Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      Left Join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      Left Join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      Left Join Ham on Ham.HamId = PartiStok.HamId
      Left Join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
      Where PartiAsamaStokDetay.CikisTarihi >= :TempTarih1 and PartiAsamaStokDetay.CikisTarihi <= :TempTarih2 and
            Makina.UniteId = :UniteId
      Into :PartiAsamaStokDetayId, :UniteId, :UniteAdi, :MakinaId, :PartiAsamalariId, :AsamaCikisTarihi, :Sira, :AsamaId, :AsamaAdi, :PartiTamirID, :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :ImalatTipiId,
           :IslemGrubuAdi, :Fason, :IslemTipiId, :ReceteId, :FaturaEdilecek, :HamId, :Hamadi, :PSKilo,
           :GenelUretimGiderindenPayAlir, :ElektrikMaliyetindenPayAlir, :DogalGazMaliyetindenPayAlir,
           :BuharMaliyetindenPayAlir, :SuMaliyetindenPayAlir, :IscilikMaliyetindenPayAlir do
    begin

      PSToplamTamamlanan = 0 ;
      PSToplamTamamlanmaPuani = 0;
      UretimGUGPayiTutari = 0 ;
      UretimElektrikPayiTutari  = 0 ;
      UretimDogalgazPayiTutari  = 0 ;
      UretimBuharPayiTutari     = 0 ;
      UretimSuPayiTutari        = 0 ;
      UretimIscilikPayiTutari   = 0 ;
      StokTabanDovizTutar       = 0 ;

      Select Sum(AsamaTamamlanmaPuani.TamamlanmaPuani), Sum(iif(PartiAsamalari.Sira <=:Sira, AsamaTamamlanmaPuani.TamamlanmaPuani, 0)) From PartiAsamaStok
      Left Join AsamaTamamlanmaPuani on AsamaTamamlanmaPuani.UniteId = :UniteId and AsamaTamamlanmaPuani.AsamaId = PartiAsamaStok.AsamaId
      Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      Where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1
      Into :PSToplamTamamlanmaPuani, :PSToplamTamamlanan ;

      If (PSToplamTamamlanan Is Null) then PSToplamTamamlanan = 0 ;
      If (PSToplamTamamlanmaPuani Is Null) then PSToplamTamamlanmaPuani = 0;

      If (PSToplamTamamlanan<> 0 and PSToplamTamamlanmaPuani<>0 ) then
      Begin
        StokTabanDovizTutar = (PSToplamTamamlanan / PSToplamTamamlanmaPuani) * PSKilo;
      End
      If (StokTabanDovizTutar Is Null) then StokTabanDovizTutar = 0 ;


      If (GenelUretimGiderDagitimTutari > 0) then
        UretimGUGPayiTutari          = (StokTabanDovizTutar / GenelUretimGiderDagitimTutari) * UniteAylikGenelUretimGideri * GenelUretimGiderindenPayAlir ;
      If (ElektrikDagitimTutari > 0) then
        UretimElektrikPayiTutari     = (StokTabanDovizTutar / ElektrikDagitimTutari) * UniteAylikElektrikGideri * ElektrikMaliyetindenPayAlir;
      If (DogalgazDagitimTutari > 0) then
        UretimDogalgazPayiTutari     = (StokTabanDovizTutar / DogalgazDagitimTutari) * UniteAylikDogalgazGideri * DogalgazMaliyetindenPayAlir;
      If (BuharDagitimTutari > 0) then
        UretimBuharPayiTutari        = (StokTabanDovizTutar / BuharDagitimTutari) * UniteAylikBuharGideri * BuharmaliyetindenPayAlir;
      If (SuDagitimTutari > 0) then
        UretimSuPayiTutari           = (StokTabanDovizTutar / SuDagitimTutari) * UniteAylikSuGideri * SuMaliyetindenPayAlir;
      If (IscilikDagitimTutari > 0) then
        UretimIscilikPayiTutari      = (StokTabanDovizTutar / IscilikDagitimTutari) * UniteAylikIscilikGideri * IscilikMaliyetindenPayAlir;

      /* basla - boya kimyasal maliyetini bul*/
      KimyasalMaliyeti = 0 ;
      BoyaMaliyeti = 0 ;
      BoyaKimyasalMaliyeti = 0 ;
      FDovizKuru = 0 ;
      FFisTarihi = Null ;

      If (ReceteId Is Not Null) then
      Begin
        AsamaToplamKilo = 0 ;
        Select Sum(PartiAsamaStok.GirisKilo) From PartiAsamaStok
        Where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId And PartiAsamaStok.AsamayaDahil = 1
        Into :AsamaToplamKilo ;

        For Select Stok.Fistarihi, Urun.UrunTipiId, StokDetay.Miktar, StokDetay.BirimFiyat, Stok.DovizId from StokDetay
          Left Join Stok on Stok.Stokid = StokDetay.StokId
          Left Join Urun On Urun.UrunId = StokDetay.UrunId
          Where StokDetay.PartiKayitId = :PartiKayitId And StokDetay.Partiasamalariid = :PartiAsamalariId And StokDetay.HareketTipi = 6
          Into :FFisTarihi, :FUrunTipiId, :FMiktar, :FUretimeCikisFiyati, :FStokDovizId do
        begin
          if (FStokDovizId <> StokTabanDovizId) Then
            Select DovizKuru From EnYakinDovizKuruGetir(:FStokDovizId, :StokTabanDovizId, :FFisTarihi) Into :FDovizKuru;
          else
            FDovizKuru = 1;

          If (FUretimeCikisFiyati Is Null) then FUretimeCikisFiyati = 0;

          FUretimeCikisFiyati = FUretimeCikisFiyati * FDovizKuru;

          If (FUrunTipiId = 0) then
            BoyaMaliyeti = BoyaMaliyeti + FMiktar * FUretimeCikisFiyati;
          Else If (FUrunTipiId = 1) then
            KimyasalMaliyeti = KimyasalMaliyeti + FMiktar * FUretimeCikisFiyati;

          BoyaKimyasalMaliyeti = BoyaKimyasalMaliyeti + FMiktar * FUretimeCikisFiyati;
        end

        BoyaMaliyeti = BoyaMaliyeti / AsamaToplamKilo * PSKilo ;
        KimyasalMaliyeti = KimyasalMaliyeti / AsamaToplamKilo * PSKilo ;
        BoyaKimyasalMaliyeti = BoyaKimyasalMaliyeti / AsamaToplamKilo * PSKilo ;
      End
      /* bitir - boya kimyasal maliyetini bul*/

      Update PartiAsamaStokDetay
        Set GenelUretimGiderPayiTutari = :UretimGUGPayiTutari,
          ElektrikPayiTutari = :UretimElektrikPayiTutari,
          DogalgazPayiTutari = :UretimDogalgazPayiTutari,
          BuharPayiTutari = :UretimBuharPayiTutari,
          SuPayiTutari = :UretimSuPayiTutari,
          IscilikPayiTutari = :UretimIscilikPayiTutari,
          BoyaMaliyeti = :BoyaMaliyeti,
          KimyasalMaliyeti = :KimyasalMaliyeti,
          LinkIslemi = 2
        Where PartiAsamaStokDetay.PartiAsamaStokDetayId = :PartiAsamaStokDetayId ;
      Suspend;
    end
  end

  for Select PartiKayitId From PartiAsamaStokDetay where PartiAsamaStokDetay.CikisTarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.CikisTarihi <= :ABitisTarihi
      Group By PartiKayitId
  into :PartiKayitId do
  begin
    Select First 1 Kilo from PartiMaliyetiKOlustur(:PartiKayitId) into :FMiktar;
  end

End
