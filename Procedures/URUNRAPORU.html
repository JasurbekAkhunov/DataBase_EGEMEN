CREATE PROCEDURE URUNRAPORU(
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BIRIMID VARCHAR(2),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    URUNGRUP1ADI VARCHAR(30),
    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30),
    BOYAGRUPID VARCHAR(4),
    BOYAGRUPADI VARCHAR(30),
    URUNKATEGORIID VARCHAR(4),
    URUNKATEGORIADI VARCHAR(30),
    MUTFAKDEPONO VARCHAR(10),
    MUTFAKDEPOADI VARCHAR(30),
    BOYARENK INTEGER,
    KIMYASALTIPIID INTEGER,
    KIMYASALTIPIADI VARCHAR(20),
    KDVORANI DOUBLE PRECISION,
    PESINISKONTOSU DOUBLE PRECISION,
    VADELIISKONTOSU DOUBLE PRECISION,
    USTSEVIYE DOUBLE PRECISION,
    ALTSEVIYE DOUBLE PRECISION,
    USTSEVIYENINUSTUNECIKMA INTEGER,
    ALTSEVIYENINALTINAINME INTEGER,
    VADESI INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ALISFIYATI DOUBLE PRECISION,
    ALISFIYATIDOVIZID VARCHAR(5),
    ALISFIYATIBIRIMID VARCHAR(2),
    SATISFIYATI DOUBLE PRECISION,
    SATISFIYATIDOVIZID VARCHAR(5),
    SATISFIYATIBIRIMID VARCHAR(2),
    URETIMECIKISFIYATI DOUBLE PRECISION,
    URETIMECIKISFIYATIDOVIZID VARCHAR(5),
    URETIMECIKISFIYATIBIRIMID VARCHAR(2),
    ENVANTEREDAHIL INTEGER,
    ENVANTEREDAHILSTR VARCHAR(4),
    URUNTIPIID INTEGER,
    URUNTIPIADI VARCHAR(20),
    TEMINSURESI DOUBLE PRECISION,
    TRANSFERKODU VARCHAR(55),
    KULLANICIADI VARCHAR(40),
    INSERTKULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    URUNRAF VARCHAR(15),
    MUHASEBEHESAPKODU ADI30,
    KAPALI SMALLINT,
    CASNO VARCHAR(255),
    EKOLOJINO VARCHAR(30),
    MRSL VARCHAR(30),
    THELIST VARCHAR(30),
    ZDHC VARCHAR(30),
    GOTS VARCHAR(30),
    GRS VARCHAR(30),
    MSDS VARCHAR(30),
    TDS VARCHAR(30),
    KOI VARCHAR(30),
    CTW VARCHAR(30),
    TEHLIKETANITIMI VARCHAR(150),
    KONSANTRASYON VARCHAR(30),
    MARKA VARCHAR(40),
    URETICIFIRMA VARCHAR(250),
    RESIMDOSYASI VARCHAR(255),
    PIPETLEMETRANSFERKODU VARCHAR(55),
    RAFOMRU INTEGER)
AS
declare variable Okey integer;
begin
  for Select Urun.UrunId, Urun.Adi, Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4,
             UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi, Urun.MutfakDepoNo, Depo.Adi, Urun.BoyaRenk, urun.KimyasalTipiId,
             KimyasalTipi.Adi, Urun.KdvOrani, Urun.PesinIskontosu, Urun.VadeliIskontosu, Urun.UstSeviye, Urun.AltSeviye,
             Urun.UstSeviyeninUstuneCikma, Urun.AltSeviyeninAltinaInme, Urun.Vadesi,
             UrunFiyat.AlisFiyati, urunFiyat.AlisFiyatiDovizId, UBAF.BirimId,
             UrunFiyat.SatisFiyati, urunFiyat.SatisFiyatiDovizId, UBSF.BirimId,
             UrunFiyat.UretimeCikisFiyati, UrunFiyat.UretimeCikisFiyatiDovizId, UBUCF.BirimId,
             UBD.BirimId, Urun.TransferKodu, Kullanici.Adi, IK.adi,  Urun.EnvantereDahil, Urun.BoyaGrupId, BoyaGrup.Adi,
             Urun.UrunKategoriId, UrunKategori.adi, Urun.UrunTipiId, UrunTipi.Adi, Urun.TeminSuresi, Urun.Raf, Urun.MuhasebeHesapKodu, Urun.Kapali,
             Urun.CasNo, Urun.EkolojiNo, Urun.MRSL, Urun.TheList, Urun.Zdhc, Urun.Gots, Urun.Grs, Urun.Msds, Urun.Tds, Urun.Koi, Urun.Ctw,
             Urun.TehlikeTanitimi, Urun.Konsantrasyon, UrunMarka.adi Marka, StokUretici.adi UreticiFirma, Urun.ResimDosyasi,Urun.PipetlemeTransferKodu, Urun.Notu, Urun.RafOmru from Urun
    left join UrunBirim UBD on UBD.UrunId = urun.Urunid and UBD.DefaultBirim = 1
    left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
    left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
    left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
    left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
    left join UrunFiyat on UrunFiyat.UrunId = urun.Urunid
    left join UrunBirim UBAF on UBAF.UrunBirimId = urunFiyat.AlisFiyatiUrunBirimId
    left join UrunBirim UBSF on UBSF.UrunBirimId = urunFiyat.SatisFiyatiUrunBirimId
    left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
    left join Depo on Depo.DepoId = urun.MutfakDepoNo
    left join BoyaGrup on BoyaGrup.BoyaGrupId = urun.BoyaGrupId
    left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = urun.KimyasalTipiId
    left join Kullanici on Kullanici.KullaniciId = urun.kullaniciid
    left join Kullanici IK on IK.KullaniciId = Urun.Insertkullaniciid
    left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
    left join UrunKategori on UrunKategori.UrunKategoriId = Urun.UrunKategoriId
    left join UrunMarka on UrunMarka.UrunMarkaId = Urun.UrunMarkaId
    left join StokUretici on StokUretici.StokUreticiId = Urun.StokUreticiId
    where (:AUrunAralikli = 0 or (:AUrunAralikli = 1 and Urun.UrunId >= :ABaslangicUrunId) and (Urun.UrunId <= :ABitisUrunId or :ABitisUrunId  = '')) and
           ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicGrup1Id  and (UrunGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
           ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicGrup2Id and (UrunGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
           ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicGrup3Id and (UrunGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
           ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicGrup4Id and (UrunGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = ''))
     into :UrunId, :UrunAdi, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4,
          :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi, :MutfakDepoNo, :MutfakDepoAdi, :BoyaRenk, :KimyasalTipiId,
          :KimyasalTipiAdi, :KdvOrani, :PesinIskontosu, :VadeliIskontosu, :UstSeviye, :AltSeviye,
          :UstSeviyeninUstuneCikma, :AltSeviyeninAltinaInme, :Vadesi,
          :AlisFiyati, :ALisFiyatiDovizId, :AlisFiyatiBirimId,
          :SatisFiyati, :SatisFiyatiDovizId, :SatisFiyatiBirimId,
          :UretimeCikisFiyati, :UretimeCikisFiyatiDovizId, :UretimeCikisFiyatiBirimId,
          :BirimId, :TransferKodu, :KullaniciAdi, :InsertKullaniciAdi, :EnvantereDahil,:BoyaGrupId, :BoyaGrupAdi,
          :UrunKategoriId, :UrunKategoriAdi, :UrunTipiId, :UrunTipiAdi, :TeminSuresi, :UrunRaf, :MuhasebeHesapKodu, :Kapali,
          :CasNo, :EkolojiNo, :MRSL, :TheList, :Zdhc, :Gots, :Grs, :Msds, :Tds, :Koi, :Ctw,
          :TehlikeTanitimi,:Konsantrasyon, :Marka, :UreticiFirma, :ResimDosyasi, :PipetlemeTransferKodu, :Notu, :RafOmru do
  begin
    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, UrunGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      if (EnvantereDahil = 1) then EnvantereDahilStr = 'Evet';
      else EnvantereDahilStr = '';

      suspend;
    end
  end
end
