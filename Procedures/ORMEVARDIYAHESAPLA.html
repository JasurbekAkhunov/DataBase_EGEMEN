CREATE PROCEDURE ORMEVARDIYAHESAPLA(
    ATARIH TIMESTAMP)
RETURNS (
    VARDIYATARIHI DATE,
    VARDIYASAATI TIMESTAMP,
    VARDIYANO INTEGER)
AS
declare variable OrmeV1BaslangicSaati Double Precision;
declare variable OrmeV2BaslangicSaati Double Precision;
declare variable OrmeV3BaslangicSaati Double Precision;
declare variable SuAnkiSaat           Double Precision;
declare variable OrmeVardiyaSayisi    Integer;
begin
  Select TimeStampToDouble(TimeStampToTime(OrmeV1BaslangicSaati)), TimeStampToDouble(TimeStampToTime(OrmeV2BaslangicSaati)), TimeStampToDouble(TimeStampToTime(OrmeV3BaslangicSaati)), OrmeVardiyaSayisi from Sistem
    into :OrmeV1BaslangicSaati, :OrmeV2BaslangicSaati, :OrmeV3BaslangicSaati, :OrmeVardiyaSayisi;


  VardiyaTarihi = ATarih;

  OrmeV1BaslangicSaati = TimeStampToDouble(VardiyaTarihi) + OrmeV1BaslangicSaati;
  OrmeV2BaslangicSaati = TimeStampToDouble(VardiyaTarihi) + OrmeV2BaslangicSaati;
  if (OrmeVardiyaSayisi = 3) then
    OrmeV3BaslangicSaati = TimeStampToDouble(VardiyaTarihi) + OrmeV3BaslangicSaati;
  else OrmeV3BaslangicSaati = 0;

  if (OrmeV2BaslangicSaati < OrmeV1BaslangicSaati) then OrmeV2BaslangicSaati = OrmeV2BaslangicSaati + 1;
  if (OrmeV3BaslangicSaati < OrmeV2BaslangicSaati and OrmeVardiyaSayisi > 2) then OrmeV3BaslangicSaati = OrmeV3BaslangicSaati + 1;

  SuankiSaat = TimeStampToDouble(ATarih);

  if (OrmeV2BaslangicSaati = 0) then
  begin
    VardiyaNo = 1;
  end
  else
  begin
    if (SuAnkiSaat >= OrmeV1BaslangicSaati and SuAnkiSaat < OrmeV2BaslangicSaati) then
    begin
      VardiyaNo = 1;
    end
    else
    begin
      if (OrmeV3BaslangicSaati = 0) then VardiyaNo = 2;
      else if (SuAnkiSaat >= OrmeV2BaslangicSaati and SuAnkiSaat < OrmeV3BaslangicSaati) then VardiyaNo = 2;
      else
      begin
        VardiyaNo = 3;
        if (SuAnkiSaat < OrmeV1BaslangicSaati) then VardiyaTarihi = VardiyaTarihi - 1;
      end
    end
  end

  if (VardiyaNo = 1) then VardiyaSaati = Doubletotimestamp(OrmeV1BaslangicSaati);
  else if (VardiyaNo = 2) then VardiyaSaati = Doubletotimestamp(OrmeV2BaslangicSaati);
  else VardiyaSaati = Doubletotimestamp(OrmeV3BaslangicSaati);

  suspend;
end
