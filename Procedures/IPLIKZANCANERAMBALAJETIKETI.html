CREATE PROCEDURE IPLIKZANCANERAMBALAJETIKETI(
    AIPLIKLOTID INTEGER,
    AIPLIKURETIMID INTEGER)
RETURNS (
    DEGER VARCHAR(100),
    SIRASI INTEGER)
AS
declare variable LotSirasi                     integer;
declare variable SiparisNoSirasi               integer;
declare variable FirmaAdiSirasi                integer;
declare variable MusteriSiparisNoSirasi        integer;
declare variable HamAdiSirasi                  integer;
declare variable HamAdi2Sirasi                 integer;
declare variable HamAdi3Sirasi                 integer;
declare variable MelanjSirasi                  integer;
declare variable FlamSirasi                    integer;
declare variable IplikCinsiSirasi              integer;
declare variable IplikCinsiDigerAdiSirasi      integer;
declare variable KalinlikSirasi                integer;
declare variable OzellikSirasi                 integer;
declare variable OzellikDigerAdiSirasi         integer;
declare variable ElyafSinifSirasi              integer;
declare variable ElyafTipiSirasi               integer;
declare variable HarmanAdiSirasi               integer;
declare variable TalepNoSirasi                 integer;
declare variable BobinGramajiSirasi            integer;
declare variable KonikRengiSirasi              integer;
declare variable KonikCinsiSirasi              integer;
declare variable AmbalajAdiSirasi              integer;
declare variable BarkodNoSirasi                integer;
declare variable BrutKiloSirasi                integer;
declare variable NetKiloSirasi                 integer;
declare variable BobinSayisiSirasi             integer;
declare variable KoliNoSirasi                  integer;
declare variable TarihSirasi                   integer;
declare variable SaatSirasi                    integer;
declare variable LotNotuSirasi                 integer;
declare variable LotNotu2Sirasi                integer;
declare variable DaraSirasi                    integer;

declare variable Lot  varchar(30);
declare variable BoyaSiparisYil  integer;
declare variable BoyaSiparisId   integer;
declare variable IplikTalepYil  integer;
declare variable IplikTalepId   integer;
declare variable FirmaAdi        varchar(100);
declare variable MusteriSiparisNo varchar(50);
declare variable HamAdi           varchar(255);
declare variable HamAdi2          varchar(255);
declare variable HamAdi3          varchar(255);
declare variable MelanjKodu       varchar(20);
declare variable FlamKodu         varchar(20);
declare variable IplikCinsiAdi    varchar(80);
declare variable IplikCinsiDigerAdi  varchar(30);
declare variable KalinlikAdi      varchar(50);
declare variable OZellikAdi       varchar(40);
declare variable OzellikDigerAdi  varchar(40);
declare variable ElyafSinifAdi    varchar(20);
declare variable ElyafTipiAdi     varchar(30);
declare variable BobinGramaji     varchar(20);
declare variable AmbalajAdi       varchar(20);
declare variable KonikRengiAdi    varchar(20);
declare variable KonikCinsiAdi    varchar(20);
declare variable HarmanAdi        varchar(50);
declare variable LotNotu          varchar(250);
declare variable LotNotu2          varchar(250);
declare variable IplikStokId      integer;
declare variable EtiketTipiId     integer;
declare variable BrutKilo         Numeric(12,2);
declare variable NetKilo          Numeric(12,2);
declare variable BobinSayisi      integer;
declare variable KapNo            integer;
declare variable Fnow             TimeStamp;
declare variable EgemenFirmaId    integer;
declare variable HamAdiKisim2Sirasi  integer;
declare variable HamAdi2Kisim2Sirasi integer;
begin
  Select EgemenFirmaId from sistem where sistemID = 1 into :EgemenFirmaId;

  select IplikLot.Lot, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Firma.Adi, coalesce(BoyaSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.MusteriSiparisNo),
         Ham.Adi, Ham.Adi2, Ham.Adi3, IplikLot.MelanjKodu, IplikLot.FlamKodu, IplikCinsi.Adi, Kalinlik.Adi, Ozellik.Adi, ElyafSinif.Adi,
         ElyafTipi.Adi, Harman.Adi, IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, IplikUretim.BobinGramaji, IplikLot.EtiketTipiId,
         KonikRengi.Adi, KonikCinsi.Adi, Ambalaj.Adi, IplikUretim.IplikStokId, IplikUretim.Miktar,  IplikUretim.NetMiktar,
         IplikUretim.BobinSayisi, IplikUretim.KapNo, IplikLot.Notu, IplikLot.Notu2, IplikCinsi.DigerAdi, Ozellik.DigerAdi from IplikUretim
    left join IplikLot on IplikLot.IplikLotId = IplikUretim.IplikLotID
    left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikUretim.IplikTalepDetayID
    left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikUretim.IplikTalepKayitID
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join Ham on Ham.HamId = IplikLot.HamId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
    left join Ozellik on Ozellik.OzellikId = Ham.OzellikId
    left join ElyafSinif on ElyafSinif.ElyafSinifId = Ham.ElyafSinifId
    left join ElyafTipi on ElyafTipi.ElyafTipiId = IplikLot.ElyafTipiId
    left join Harman on Harman.HarmanKayitId = IplikLot.HarmanKayitId
    left join KonikRengi on KonikRengi.KonikRengiId = IplikUretim.KonikRengiId
    left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikUretim.KonikCinsiId
    left join Ambalaj on Ambalaj.AmbalajId = IplikUretim.AmbalajId
    where IplikUretim.IplikUretimId = :AIplikUretimId
    into :Lot, :BoyaSiparisYil, :BoyaSiparisId, :FirmaAdi, :MusteriSiparisNo, :HamAdi, :HamAdi2, :HamAdi3, :MelanjKodu, :FlamKodu,
         :IplikCinsiAdi, :KalinlikAdi, :OzellikAdi, :ElyafSinifAdi, :ElyafTipiAdi, :HarmanAdi, :IplikTalepYil, :IplikTalepId,
         :BobinGramaji, :EtiketTipiId, :KonikRengiAdi, :KonikCinsiAdi, :AmbalajAdi, :IplikStokId, :BrutKilo,  :NetKilo,
         :BobinSayisi, :KapNo, :LotNotu, :LotNotu2, :IplikCinsiDigerAdi, :OzellikDigerAdi;

  if (EtiketTipiId is null) then exit;


  Select LotSirasi, SiparisNoSirasi, FirmaAdiSirasi, MusteriSiparisNoSirasi, HamAdiSirasi, HamAdi2Sirasi, HamAdi3Sirasi, MelanjSirasi, FlamSirasi,
         IplikCinsiSirasi, KalinlikSirasi, OzellikSirasi, ElyafSinifSirasi, ElyafTipiSirasi, HarmanAdiSirasi, TalepNoSirasi, BobinGramajiSirasi,
         KonikRengiSirasi, KonikCinsiSirasi, AmbalajAdiSirasi, BarkodNoSirasi, BrutKiloSirasi, NetKiloSirasi, BobinSayisiSirasi, KoliNoSirasi,
         TarihSirasi, SaatSirasi, LotNotuSirasi, LotNotu2Sirasi, DaraSirasi, HamAdiKisim2Sirasi, HamAdi2Kisim2Sirasi, IplikCinsiDigerAdiSirasi, OzellikDigerAdiSirasi from EtiketTipi
    where EtiketTipi.EtiketTipiID = :EtiketTipiId
    into :LotSirasi, :SiparisNoSirasi, :FirmaAdiSirasi, :MusteriSiparisNoSirasi, :HamAdiSirasi, :HamAdi2Sirasi, :HamAdi3Sirasi, :MelanjSirasi, :FlamSirasi,
         :IplikCinsiSirasi, :KalinlikSirasi, :OzellikSirasi, :ElyafSinifSirasi, :ElyafTipiSirasi, :HarmanAdiSirasi, :TalepNoSirasi, :BobinGramajiSirasi,
         :KonikRengiSirasi, :KonikCinsiSirasi, :AmbalajAdiSirasi, :BarkodNoSirasi, :BrutKiloSirasi, :NetKiloSirasi, :BobinSayisiSirasi, :KoliNoSirasi,
         :TarihSirasi, :SaatSirasi, :LotNotuSirasi, :LotNotu2Sirasi, :DaraSirasi, :HamAdiKisim2Sirasi, :HamAdi2Kisim2Sirasi, :IplikCinsiDigerAdiSirasi, :OzellikDigerAdiSirasi;

  if (EgemenFirmaId <> 46) then
  begin
    Deger = '';
    Sirasi = -2;
    suspend;

    Deger = '';
    Sirasi = -1;
    suspend;
  end

  if (LotSirasi > 0) then
  begin
    Sirasi = LotSirasi;
    if (EgemenFirmaId = 46) then Deger = SubString(Lot from 5 for Char_Length(Lot) - 4); /*Başyazıcı*/
    else Deger = Lot;
    suspend;
  end

  if (SiparisNoSirasi > 0) then
  begin
    Sirasi = SiparisNoSirasi;

    if (BoyaSiparisYil is not null) then
      Deger = EgemenIntegerToStr(BoyaSiparisYil)  || '-' || EgemenIntegerToStr(BoyaSiparisId);
    else
      Deger = '';
    suspend;
  end

  if (FirmaAdiSirasi > 0) then
  begin
    if (FirmaAdi is null) then FirmaAdi = '';

    Sirasi = FirmaAdiSirasi;
    Deger = SubString(FirmaAdi from 1 for 20);
    suspend;
  end


  if (MusteriSiparisNoSirasi > 0) then
  begin
    if (MusteriSiparisNo is null) then MusteriSiparisNo = '';

    Sirasi = MusteriSiparisNoSirasi;
    Deger = MusteriSiparisNo;
    suspend;
  end

  if (HamAdiSirasi > 0) then
  begin
    Sirasi = HamAdiSirasi;
    Deger  = SubString(HamAdi from 1 for 22);
    suspend;
  end

  if (HamAdi2Sirasi > 0) then
  begin
    if (HamAdi2 is null) then HamAdi2 = '';

    Sirasi = HamAdi2Sirasi;
    Deger = SubString(HamAdi2 from 1 for 22);
    suspend;
  end

  if (EgemenFirmaId = 46) then
  begin
    if (HamAdiKisim2Sirasi > 0) then
    begin
      Sirasi = HamAdiKisim2Sirasi;
      Deger  = SubString(HamAdi from 23 for 22);
      suspend;
    end

    if (HamAdi2Kisim2Sirasi > 0) then
    begin
      if (HamAdi2 is null) then HamAdi2 = '';

      Sirasi = HamAdi2Kisim2Sirasi;
      Deger = SubString(HamAdi2 from 23 for 22);
      suspend;
    end

  end


  if (HamAdi3Sirasi > 0) then
  begin
    if (HamAdi3 is null) then HamAdi3 = '';

    Sirasi = HamAdi3Sirasi;
    Deger = SubString(HamAdi3 from 1 for 22);
    suspend;
  end

  if (MelanjSirasi > 0) then
  begin
    if (MelanjKodu is null) then MelanjKodu = '';

    Sirasi = MelanjSirasi;
    Deger = MelanjKodu;
    suspend;
  end

  if (FlamSirasi > 0) then
  begin
    if (FlamKodu is null) then FlamKodu = '';

    Sirasi = FlamSirasi;
    Deger = FlamKodu;
    suspend;
  end


  if (IplikCinsiSirasi > 0) then
  begin
    if (IplikCinsiAdi is null) then IplikCinsiAdi = '';

    Sirasi = IplikCinsiSirasi;
    Deger = IplikCinsiAdi;
    suspend;
  end

  if (KalinlikSirasi > 0) then
  begin
    if (KalinlikAdi is null) then KalinlikAdi = '';

    Sirasi = KalinlikSirasi;
    Deger = KalinlikAdi;
    suspend;
  end

  if (OzellikSirasi > 0) then
  begin
    if (OzellikAdi is null) then OzellikAdi = '';

    Sirasi = OzellikSirasi;
    Deger = OzellikAdi;
    suspend;
  end

  if (OzellikDigerAdiSirasi > 0) then
  begin
    if (OzellikDigerAdi is null) then OzellikDigerAdi = '';
    Sirasi = OzellikDigerAdiSirasi;
    Deger = OzellikDigerAdi;
    suspend;
  end

  if (IplikCinsiDigerAdiSirasi > 0) then
  begin
    if (IplikCinsiDigerAdi is null) then IplikCinsiDigerAdi = '';
    Sirasi = IplikCinsiDigerAdiSirasi;
    Deger = IplikCinsiDigerAdi;
    suspend;
  end

  if (ElyafSinifSirasi > 0) then
  begin
    if (ElyafSinifAdi is null) then ElyafSinifAdi = '';

    Sirasi = ElyafSinifSirasi;
    Deger = ElyafSinifAdi;
    suspend;
  end

  if (ElyafTipiSirasi > 0) then
  begin
    if (ElyafTipiAdi is null) then ElyafTipiAdi = '';

    Sirasi = ElyafTipiSirasi;
    Deger = ElyafTipiAdi;
    suspend;
  end

  if (BobinGramajiSirasi > 0) then
  begin
    if (BobinGramaji is null) then BobinGramaji = '';

    Sirasi = BobinGramajiSirasi;
    Deger = ElyafTipiAdi;
    suspend;
  end

  if (HarmanAdiSirasi > 0) then
  begin
    if (HarmanAdi is null) then HarmanAdi = '';

    Sirasi = HarmanAdiSirasi;
    Deger = HarmanAdi;
    suspend;
  end

  if (TalepNoSirasi > 0) then
  begin
    Sirasi = TalepNoSirasi;

    if (IplikTalepYil is not null) then
      Deger = EgemenIntegerToStr(IplikTalepYil)  || '-' || EgemenIntegerToStr(IplikTalepId);
    else
      Deger = '';
    suspend;
  end


  if (KonikRengiSirasi > 0) then
  begin
    if (KonikRengiAdi is null) then KonikRengiAdi = '';

    Sirasi = KonikRengiSirasi;
    Deger = KonikRengiAdi;
    suspend;
  end

  if (KonikCinsiSirasi > 0) then
  begin
    if (KonikCinsiAdi is null) then KonikCinsiAdi = '';

    Sirasi = KonikCinsiSirasi;
    Deger = KonikCinsiAdi;
    suspend;
  end

  if (LotNotuSirasi > 0) then
  begin
    if (LotNotu is null) then LotNotu = '';

    Sirasi = LotNotuSirasi;
    Deger = SubString(LotNotu from 1 for 24);
    suspend;
  end

  if (LotNotu2Sirasi > 0) then
  begin
    if (LotNotu2 is null) then LotNotu2 = '';

    Sirasi = LotNotu2Sirasi;
    Deger = SubString(LotNotu2 from 1 for 24);
    suspend;
  end

  if (AmbalajAdiSirasi > 0) then
  begin
    if (AmbalajAdi is null) then AmbalajAdi = '';

    Sirasi = AmbalajAdiSirasi;
    Deger = AmbalajAdi;
    suspend;
  end

  if (BarkodNoSirasi > 0) then
  begin
    Sirasi = BarkodNoSirasi;
    Deger = IplikStokId;
    suspend;
  end

  if (BrutKiloSirasi > 0) then
  begin
    Sirasi = BrutKiloSirasi;
    deger = cast(BrutKilo as varchar(20));
    suspend;
  end

  if (NetKiloSirasi > 0) then
  begin
    Sirasi = NetKiloSirasi;
    deger = cast(netKilo as varchar(20));
    suspend;
  end

  if (DaraSirasi > 0) then
  begin
    Sirasi = DaraSirasi;
    Deger = cast((BrutKilo - NetKilo) as varchar(20));
    suspend;
  end

  if (BobinSayisiSirasi > 0) then
  begin
    Sirasi = BobinSayisiSirasi;
    Deger = EgemenIntegerToStr(BobinSayisi);
    suspend;
  end

  if (KoliNoSirasi > 0) then
  begin
    Sirasi = KoliNoSirasi;
    Deger = EgemenIntegerToStr(KapNo);
    suspend;
  end

  if (TarihSirasi > 0) then
  begin
    Sirasi = TarihSirasi;
    Deger = Current_Date;
    suspend;
  end

  if (SaatSirasi > 0) then
  begin
    Sirasi = SaatSirasi;
    Deger = current_TimeStamp(0);
    Deger = SubString(Deger from 1 for 16);
    suspend;
  end

  if (BrutKiloSirasi > 0) then
  begin
    Sirasi = 999;
    deger = '0.0';
    suspend;

    Sirasi = 1000;
    deger = '0.0';
    suspend;
  end
end
