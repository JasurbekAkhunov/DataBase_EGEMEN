CREATE PROCEDURE HGURETIMICINURUNLISTESI(
    AHGISEMRIKAYITID INTEGER,
    AISLEMKAYNAGI INTEGER)
RETURNS (
    HGISEMRIKAYITID INTEGER,
    HGISEMRIPASTALID INTEGER,
    HGISEMRIURUNID INTEGER,
    URUNID TYPE OF COLUMN URUN.URUNID,
    URUNADI TYPE OF COLUMN URUN.ADI,
    PLANLANANMIKTAR INTEGER,
    GERCEKLESENMIKTAR INTEGER,
    URETIMECIKILANMIKTAR INTEGER,
    KALANMIKTAR INTEGER)
AS
declare variable sikHazirGiyim     integer;
declare variable sikHazirGiyimKesim integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretimeCikis integer;
begin
  select sikHazirGiyim, sikHazirGiyimKesim from StokIslemKaynagiGetir into :sikHazirGiyim, :sikHazirGiyimKesim;
  select shtUretimeCikis, shtFasonUretimeCikis from StokHareketTipiGetir into :shtUretimeCikis, :shtFasonUretimeCikis;

  if (AIslemKaynagi = sikHazirGiyim) then
  begin
    For Select HgIsEmriUrun.HgIsEmriUrunId, HgIsEmriUrun.HGIsEmriKayitId, HgIsEmriUrun.UrunId, Urun.Adi, HgIsEmriUrun.PlanlananMiktar,
               HgIsEmriUrun.GerceklesenMiktar from HgIsEmriUrun
      Left Join Urun On Urun.UrunId = HgIsEmriUrun.UrunId
      Left Join HgIsEmri On HgIsEmri.HGIsEmriKayitId = HgIsEmriUrun.HGIsEmriKayitId
      Where HgIsEmriUrun.HGIsEmriKayitId = :AHGIsEmriKayitId and HgIsEmri.Parcaiscilik = 0
        Union
      Select HgIsEmriDetay.HgIsEmriUrunId, HgIsEmriDetay.HGIsEmriKayitId, HgIsEmriDetay.UrunId, Urun.Adi, HgIsEmriUrun.Planlananmiktar,
             HgIsEmriUrun.GerceklesenMiktar From HgIsEmriDetay
      Left Join HgIsEmriUrun On HgIsEmriUrun.HgIsEmriUrunId = HgIsEmriDetay.HgIsEmriUrunId
      Left Join HgIsEmri On HgIsEmri.HGIsEmriKayitId = HgIsEmriDetay.HGIsEmriKayitId
      Left Join Urun On Urun.UrunId = HgIsEmriDetay.UrunId
      Where HgIsEmriDetay.HGIsEmriKayitId = :AHGIsEmriKayitId And HgIsEmri.Parcaiscilik = 1
    Into :HgIsEmriUrunId, :HGIsEmriKayitId, :UrunId, :UrunAdi, :PlanlananMiktar, :GerceklesenMiktar do
    begin
      UretimeCikilanMiktar = 0;
      Select Coalesce(Sum(StokDetay.Miktar), 0) from StokDetay
        where StokDetay.UrunID = :UrunId and HGIsEmriKayitId = :AHGIsEmriKayitId and (StokDetay.HareketTipi = :shtFasonUretimeCikis or StokDetay.HareketTipi = :shtUretimeCikis)
      into :UretimeCikilanMiktar;
      KalanMiktar = UretimeCikilanMiktar - GerceklesenMiktar;
      if (KalanMiktar > 0) then suspend;
    end
  end
  else
  begin
    For Select HgIsEmriPastal.HgIsEmriPastalId, HgIsEmriPastal.HGIsEmriKayitId, HgIsEmriPastalUrun.UrunId, Urun.Adi,
             HgIsEmriPastalUrun.PlanlananMiktar, HgIsEmriPastalUrun.GerceklesenMiktar From HgIsEmriPastal
      Left Join HgIsEmriPastalUrun On HgIsEmriPastalUrun.HgisemripastalId = HgIsEmriPastal.HgIsEmriPastalId
      Left Join Urun On Urun.Urunid = HgIsEmriPastalUrun.Urunid
      Where HgIsEmriPastal.HGIsEmriKayitId = :AHGIsEmriKayitId and HgIsEmriPastalUrun.HgIsEmriPastalId Is Not Null
    Into :HgIsEmriPastalId, :HGIsEmriKayitId, :UrunId, :UrunAdi, :PlanlananMiktar, :GerceklesenMiktar do
    begin
      KalanMiktar = PlanlananMiktar - GerceklesenMiktar;
      suspend;
    end
  end

end
