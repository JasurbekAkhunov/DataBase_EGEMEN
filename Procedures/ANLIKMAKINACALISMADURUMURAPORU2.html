CREATE PROCEDURE ANLIKMAKINACALISMADURUMURAPORU2(
    AUNITEID INTEGER,
    AMAKINAGRUBU VARCHAR(4),
    AMAKINAID VARCHAR(10))
RETURNS (
    MAKINAID VARCHAR(20),
    MAKINASARJID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    RECETETURUID INTEGER,
    RECETETURUADI VARCHAR(15),
    LABRENKGRUPID INTEGER,
    LABRENKGRUPADI VARCHAR(30),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    ASAMARECETEID VARCHAR(20),
    ASAMARECETEADI VARCHAR(40),
    GIRISTARIHI TIMESTAMP,
    GIRISPERSONELADI VARCHAR(40),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ARAVERMESURESI DOUBLE PRECISION,
    ARAVERMESURESISTR VARCHAR(20),
    DISTAMIRTIPIADI VARCHAR(20),
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    DENYE VARCHAR(30),
    ICTAMIR INTEGER,
    DISTAMIR INTEGER,
    ICTAMIRTIPIADI VARCHAR(20),
    ICTAMIRSEBEBIADI VARCHAR(20),
    ILAVEPROSES INTEGER,
    ILAVEPROSESSTR VARCHAR(4),
    ILAVEPROSESSEBEBIID INTEGER,
    ILAVEPROSESSEBEBIADI VARCHAR(20),
    RECETEASAMATOPLAMSURE DOUBLE PRECISION,
    RECETEASAMATOPLAMSURESTR VARCHAR(20),
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(20))
AS
declare variable PartiTamirID integer;
declare variable IslemTipiID integer;
begin
  for select PartiAsamaStokDetay.MakinaId, PartiAsamaStokDetay.MakinaSarjId, Parti.Partiyil, Parti.PartiId, Parti.PartiKayitId, Parti.FirmaId, Firma.adi, Parti.ReceteId, PartiRecete.adi,
             PartiRecete.ReceteTuruId, ReceteTuru.Adi, PartiRecete.LabRenkGrupId, LabRenkGrup.Adi,
             PartiStok.HamAdiToplami, Gp.Adi, PartiAsamaStokDetay.kilo, PartiAsamaStokDetay.Metre,
             PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.AraVermeSuresi,
             DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, PartiAsamaStok.AsamaID, Asama.Adi,
             IslemTipi.Adi, IslemGrubu.Adi, PartiStok.Denye, DisTamirTipi.Adi, IcTamirTipi.Adi, TamirSebebi.Adi,
             PartiAsamalari.ReceteId, AsamaRecete.Adi, PartiAsamalari.PartiTamirId, Parti.IslemTipiId,
             PartiAsamalari.IlaveProses, PartiAsamalari.IlaveProsesSebebiId, IlaveProsesSebebi.Adi from PartiAsamaStokDetay
  left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
  left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
  left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
  left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
  left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
  left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join TamirTipi DisTamirTipi on DisTamirTipi.TamirTipiId = PartiStok.TamirTipiId
  left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
  left join TamirTipi IcTamirTipi on IcTamirTipi.TamirTipiId = PartiTamir.TamirTipiId
  left join TamirSebebi on TamirSebebi.TamirSebebiid = Partitamir.TamirSebebiid
  left join IlaveProsesSebebi on IlaveProsesSebebi.IlaveProsesSebebiId = PartiAsamalari.IlaveProsesSebebiId
  left join Firma on Firma.FirmaId = Parti.FirmaId
  left join Recete PartiRecete on PartiRecete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = PartiRecete.LabReceteId
  left join LabRenkGrup on LabRenkGrup.LabRenkGrupId = PartiRecete.LabRenkGrupId
  left join ReceteTuru on ReceteTuru.ReceteTuruId = PartiRecete.ReceteTuruId
  left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  left join Desen on Desen.DesenId = DesenVaryant.DesenId
  left join personel GP on GP.personelId = PartiAsamaStokDetay.GirispersonelId
  left join Recete AsamaRecete on AsamaRecete.ReceteId = PartiAsamalari.ReceteId
  where PartiAsamaStokDetay.CikisTarihi is null and PartiAsamaStokDetay.GirisTarihi is not null and
       (PartiAsamaStokDetay.MakinaId = :AMakinaId or :AMakinaId is null) and
       (Makina.uniteId = :AUniteId or :AUniteId is null) and
       (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and Makina.OtomasyonaDahil = 1 and
       PartiAsamaStok.AsamayaDahil = 1
  into :MakinaId, :MakinaSarjId, :Partiyil, :PartiId, :PartiKayitId, :FirmaId, :FirmaAdi, :ReceteId, :ReceteAdi,
       :ReceteTuruId, :ReceteTuruAdi, :LabRenkGrupId, :LabRenkGrupAdi, :HamAdiToplami,
       :GirisPersonelAdi, :Kilo, :Metre, :GirisTarihi, :AraVermeSuresi,
       :DesenId, :DesenAdi, :VaryantAdi, :AsamaID, :AsamaAdi,
       :IslemTipiAdi, :IslemGrubuAdi, :Denye, :DisTamirTipiAdi, :IcTamirTipiAdi, :IcTamirSebebiAdi,
       :AsamaReceteId, :AsamaReceteAdi, :PartiTamirId, :IslemTipiId, :IlaveProses, :IlaveProsesSebebiId, :IlaveProsesSebebiAdi do
  begin
    if (AraVermeSuresi <> 0) then AraVermeSuresiStr = TarihtoStr(AraVermeSuresi);
    else AraVermeSuresiStr = '';

    PartiIDStr = '(' || EgemenIntegerToStr(PartiYil) || ') '  || EgemenIntegerToStr(PartiId)  || ' ';


    ReceteAsamaToplamSure = Null;
    ReceteAsamaToplamSureStr = '';

    if (AsamaReceteID is not null) then
    begin
      Select Sum(coalesce(MakinaReceteAsamalari.Sure, 0)) from ReceteDetay
        left join MakinaReceteAsamalari on MakinaReceteAsamalari.MakinaID = :MakinaId and MakinaReceteAsamalari.ReceteAsamaId = ReceteDetay.ReceteAsamaId
        where ReceteDetay.ReceteID = :AsamaReceteID and ReceteDetay.ReceteAsamaId is not null into :ReceteAsamaToplamSure;

      if (ReceteAsamaToplamSure is not null) then
         ReceteAsamaToplamSureStr = TarihtoStr(ReceteAsamaToplamSure / 1440.0);

    end
    DisTamir = 0;
    IcTamir = 0;

    if (IslemTipiID = 3) then DisTamir = 1;
    else if (PartiTamirID is not null) then IcTamir = 1;

    if (IlaveProses = 1) then
      IlaveProsesStr = 'Evet';
    else
      IlaveProsesStr = '';

    GecenSure = Current_TimeStamp - GirisTarihi;
    GecenSure = GecenSure - AraVermeSuresi;

    GecenSureStr = TarihtoStr(GecenSure);

    suspend;
  end
end
