CREATE PROCEDURE CSGIRISBORDROSULISTESIGETIR(
    AAV VARCHAR(1),
    ACS VARCHAR(1),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    BORDROID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMADOVIZID VARCHAR(5),
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    MAKBUZNO VARCHAR(20),
    TOPLAMMEBLAG DOUBLE PRECISION,
    EVRAKSAYISI INTEGER)
AS
begin
  for select CekSenet.BordroId, CekSenet.FirmaId, Firma.Adi, Firma.DovizId, CekSenet.Tarih, CekSenet.Aciklama, CekSenet.MakbuzNo,
      (select count(*) from CekSenetDetay where CekSenetDetay.Av = CekSenet.AV and CekSenetDetay.Cs = CekSenet.Cs and CekSenetDetay.BordroId = CekSenet.BordroId),
      (select sum(CekSenetDetay.Meblag / CekSenetDetay.DovizKuru1 * CekSenetDetay.DovizKuru2) from CekSenetDetay where CekSenetDetay.Av = CekSenet.AV and CekSenetDetay.Cs = CekSenet.Cs and CekSenetDetay.BordroId = CekSenet.BordroId)
  from CekSenet
  left join firma on firma.FirmaId = CekSenet.FirmaId
    where CekSenet.AV = :AAv and CekSenet.CS = :ACs and CekSenet.Tarih >= :ABaslangicTarihi and CekSenet.Tarih <= :ABitisTarihi and
     (CekSenet.FirmaId = :AFirmaId or :AfirmaId = '')
     into :BordroId, :FirmaId, :FirmaAdi, :FirmaDovizId, :Tarih, :Aciklama, :MakbuzNo, :EvrakSayisi,:ToplamMeblag  do suspend;
end
