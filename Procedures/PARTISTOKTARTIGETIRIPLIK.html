CREATE PROCEDURE PARTISTOKTARTIGETIRIPLIK(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    PAKETLEMEID INTEGER,
    TARIH TIMESTAMP,
    KAPADEDI INTEGER,
    BOBINSAYISI INTEGER,
    KOLINO INTEGER,
    BRUTKILO NUMERIC(12,3),
    KILO NUMERIC(12,3),
    DARA NUMERIC(12,3),
    KALITENO INTEGER,
    DEPOID VARCHAR(10),
    SEVKDURUMU VARCHAR(25),
    PAKETLEMETARIHI TIMESTAMP,
    HESAPLANANBRUTKILO NUMERIC(12,3),
    KALITEACIKLAMATIPIID INTEGER,
    KALITEACIKLAMATIPIADI VARCHAR(20),
    MAMULPROSESID INTEGER,
    MAMULPROSESADI VARCHAR(255),
    PALETNO VARCHAR(5),
    IPLIKPAKETLEMEID INTEGER,
    BARKODID VARCHAR(30),
    IADEBOYAIRSALIYEDETAYID INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    IADESTR VARCHAR(5))
AS
declare variable BoyaIrsaliyeDetayId integer;
declare variable HamStokSarfFisiDetayId integer;
begin
  For Select PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId, PartiStokTarti.Tarih,
             PartiStokTarti.KapAdedi, PartiStokTarti.BobinSayisi, PartiStokTarti.Kilo, PartiStokTarti.Dara,
             PartiStokTarti.KaliteNo, PartiStokTarti.DepoId, IplikPaketleme.KoliNo, IplikPaketleme.PaketlemeId, IplikPaketleme.PaketlemeTarihi,
             PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId,
             PartiStokTarti.KaliteAciklamaTipiId, KaliteAciklamaTipi.Adi, PartiStokTarti.MamulProsesId, MamulProses.Adi,
             IplikPaketleme.IplikPaketlemeID,
             IK.Adi, PartiStokTarti.InsertTarihi, SK.Adi, PartiStokTarti.DegisimTarihi,
             PartiStokTarti.BrutKilo, PartiStokTarti.PaletNo, PartiStokTarti.BarkodId, PartiStokTarti.IadeBoyaIrsaliyeDetayId from PartiStokTarti
  Left Join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
  Left Join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = PartiStokTarti.KaliteAciklamaTipiId
  Left Join MamulProses on MamulProses.MamulProsesId = PartiStokTarti.MamulProsesId
  Left Join Kullanici IK on IK.KullaniciId = PartiStokTarti.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = PartiStokTarti.KullaniciId
  Where PartiStokTarti.PartiKayitId  = :APartiKayitId
  Order by PartiStokTarti.PartiStokTartiId
  into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, Tarih, :KapAdedi, :BobinSayisi, :Kilo, :Dara, :KaliteNo, :DepoId, :KoliNo, :PaketlemeId,
       :PaketlemeTarihi, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :KaliteAciklamaTipiId, :KaliteAciklamaTipiAdi, :MamulProsesId, :MamulProsesAdi,
       :IplikPaketlemeID,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :HesaplananBrutKilo, :PaletNo, :BarkodId, :IadeBoyaIrsaliyeDetayId do
  begin
    if (BoyaIrsaliyeDetayId is not null) then SevkDurumu = 'Sevk Edilmiş';
    else if (HamStokSarfFisiDetayId is not null) then SevkDurumu = 'Sarf Fişi İle Çıkış';
    else SevkDurumu = '';

    IadeStr = '';
    if (IadeBoyaIrsaliyeDetayId is not null) then
      IadeStr = 'Evet';

    BrutKilo = Kilo + Dara;
    suspend;
  end
end
