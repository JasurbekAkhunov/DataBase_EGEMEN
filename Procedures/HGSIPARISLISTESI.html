CREATE PROCEDURE HGSIPARISLISTESI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKAPALI INTEGER,
    AMODELID MODELKODUSIZE,
    AMUSTERISIPARISNO VARCHAR(50))
RETURNS (
    HGSIPARISKAYITID INTEGER,
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER,
    HGSIPARISNOSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPARISTARIHI TIMESTAMP,
    SIPARISIVEREN VARCHAR(50),
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(40),
    TESLIMSEKLIID INTEGER,
    TESLIMSEKLIADI VARCHAR(40),
    YURTDISISTR VARCHAR(5),
    NOTU MEMO,
    BASKAFIRMAYASATILABILIRSTR VARCHAR(5),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
Declare Variable FGoster EgeBoolean;
Declare Variable ModelKayitId Integer;
begin
  if (AModelId is Not Null) then Select ModelKayitId From Model Where Model.ModelID = :AModelID  into :ModelKayitId;

  For Select HGSiparis.HGSiparisKayitId, HGSiparis.HGSiparisYil, HGSiparis.HGSiparisId, HGSiparis.FirmaId, Firma.Adi,
             HGSiparis.SiparisTarihi, FirmaYetkili.Adi, HGSiparis.SiparisiAlanKullaniciId, SAK.Adi, HGSiparis.SiparisGrupId, SiparisGrup.Adi, HGSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             HGSiparis.Fason, HGSiparis.OdemeSekliId, OdemeSekli.Adi, HGSiparis.TeslimSekliId, TeslimSekli.Adi,
             IK.Adi, HGSiparis.InsertTarihi, Sk.Adi, HGSiparis.DegisimTarihi from HGSiparis
    left join Firma On Firma.FirmaId = HGSiparis.FirmaId
    left join OdemeSekli On OdemeSekli.OdemeSekliId = HGSiparis.OdemeSekliId
    left join TeslimSekli On TeslimSekli.TeslimSekliId = HGSiparis.TeslimSekliId
    left join SiparisGrup On SiparisGrup.SiparisGrupId = HGSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 On SiparisGrup2.SiparisGrupId = HGSiparis.SiparisGrup2Id
    left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = HGSiparis.FirmaYetkiliId
    Left Join Kullanici SAK on SAK.KullaniciId = HGSiparis.SiparisiAlanKullaniciId
    Left Join Kullanici IK on IK.KullaniciId = HGSiparis.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = HGSiparis.KullaniciId
    left join HGSiparisDetay On HGSiparisDetay.HGSiparisKayitId = HGSiparis.HGSiparisKayitId
    where HGSiparis.SiparisTarihi >= :ABaslangicTarihi and HGSiparis.SiparisTarihi <= :ABitisTarihi and
          ((Firma.AnaHesapFirmaId = :AFirmaId) or :AFirmaId is null) and
          (:AKapali is null or HGSiparisDetay.Kapali = :AKapali)
    group By HGSiparis.HGSiparisKayitId, HGSiparis.HGSiparisYil, HGSiparis.HGSiparisId, HGSiparis.FirmaId, Firma.Adi,
             HGSiparis.SiparisTarihi, FirmaYetkili.Adi, HGSiparis.SiparisiAlanKullaniciId, SAK.Adi, HGSiparis.SiparisGrupId, SiparisGrup.Adi, HGSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             HGSiparis.Fason, HGSiparis.OdemeSekliId, OdemeSekli.Adi, HGSiparis.TeslimSekliId, TeslimSekli.Adi,
             IK.Adi, HGSiparis.InsertTarihi, Sk.Adi, HGSiparis.DegisimTarihi
    into :HGSiparisKayitId, :HGSiparisYil, :HGSiparisId, :FirmaId, :FirmaAdi, :SiparisTarihi, :SiparisiVeren, :SiparisiAlanKullaniciId,
         :SiparisiAlanKullaniciAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :Fason, :OdemeSekliId, :OdemeSekliAdi, :TeslimSekliId,
         :TeslimSekliAdi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi
         do
   begin
     if (Fason = 1) then FasonStr = 'Fason';
     else FasonStr = '';

     HGSiparisNoStr = '(' || EgemenIntegerToStr(HGSiparisYil) || ') ' || EgemenIntegerToStr(HGSiparisId) || ' ';

     FGoster = 1;

     if (AModelId is Not Null) then
     begin
        Select Iif(Count(*) > 0, 1, 0) From HGSiparisDetay
        Where HGSiparisDetay.HGSiparisKayitId = :HGSiparisKayitId and HGSiparisDetay.ModelKayitId = :ModelKayitId
        Into :FGoster;
     end

     if (AMusteriSiparisNo is Not Null) then
     begin
        Select Iif(Count(*) > 0, 1, 0) From HGSiparisDetay
        Where HGSiparisDetay.HGSiparisKayitId = :HGSiparisKayitId and HGSiparisDetay.MusteriSiparisNo Like '%'||:AMusteriSiparisNo||'%'
        Into :FGoster;
     end

     if (FGoster = 1) then
        suspend;
   end
end
