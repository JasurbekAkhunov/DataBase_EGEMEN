CREATE PROCEDURE PARTISABLONGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTISABLONID INTEGER,
    PARTISINIFID INTEGER,
    SABLONNO INTEGER,
    SABLONADI VARCHAR(1),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    ACILIM VARCHAR(10),
    HAZIRLANANMIKTAR DOUBLE PRECISION,
    KULLANIMMIKTARI DOUBLE PRECISION,
    ILAVEMIKTARI BOYAMIKTARSIZE,
    DUZELTMEMIKTARI BOYAMIKTARSIZE,
    DEPOID VARCHAR(10),
    EMULSIYONURUNID VARCHAR(20),
    PATMIKTARI DOUBLE PRECISION,
    SABLONRENKID INTEGER,
    SABLONRENKADI VARCHAR(30),
    MESHID INTEGER,
    DOLULUKORANI DOUBLE PRECISION,
    SIRA INTEGER,
    RAKLE INTEGER,
    PRES INTEGER,
    FIKSESICAKLIGI INTEGER,
    MUSTERISIPARISNO VARCHAR(255),
    PARTISABLONISEMRIID INTEGER,
    STOKTANDUSULDU INTEGER,
    STOKDUSUMSTOKID INTEGER,
    STOKDUSUMFISNO VARCHAR(100),
    OKEYLEYENKULLANICIID INTEGER,
    OKEYLEYENKULLANICIADI VARCHAR(40),
    OKEYTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    BIDONSAYISI INTEGER,
    BIDONKAPASITE DOUBLE PRECISION)
AS
declare variable Acilim1               Double Precision;
declare variable Acilim2               Double Precision;
declare variable DesenVaryantId        integer;
declare variable DesenId               varchar(20);
declare variable FBoyaSiparisKayitId   integer;
declare variable FMusteriSiparisNo     varchar(50);
begin
  Select Parti.DesenVaryantID, DesenVaryant.DesenId from Parti
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    where Parti.PartiKayitID = :APartiKayitID
    into :DesenVaryantID, :DesenId;

  for select PartiSablon.PartiSablonId, PartiSablon.SablonNo, PartiSablon.Acilim1, PartiSablon.Acilim2,
             PartiSablon.HazirlananMiktar, PartiSablon.KullanimMiktari,
             PartiSablon.DepoId, PartiSablon.StoktanDusuldu, DesenVaryantSablon.SablonRenkId, SablonRenk.Adi,
             DesenVaryantSablon.DolulukOrani, DesenVaryantSablon.MeshId, PartiSablon.PatMiktari, PartiSablon.EmulsiyonUrunId,
             PartiSablon.OkeyleyenKullaniciId, Ok.Adi, PartiSablon.OkeyTarihi, PartiSablon.PartiSinifId, PartiSablon.Sira, DesenSablon.SablonAdi,
             DesenSablon.RenkId, Renk.Adi, PartiSablon.Rakle, PartiSablon.Pres,
             IK.Adi, PartiSablon.InsertTarihi, SK.Adi, PartiSablon.DegisimTarihi, PartiSablon.FikseSicakligi, BoyaSiparisDetay.MusteriSiparisNo,
             Parti.BoyaSiparisKayitId, PartiSablon.PartiSablonIsEmriID, PartiSablon.BidonSayisi, PartiSablon.BidonKapasite from PartiSablon
  left join DesenSablon on DesenSablon.DesenID = :DesenID and DesenSablon.SablonSiraNo = PartiSablon.SablonNo
  left join Renk on Renk.RenkId = DesenSablon.RenkId
  left join DesenVaryantSablon on DesenVaryantSablon.DesenVaryantID = :DesenvaryantId and DesenVaryantSablon.SablonNo = PartiSablon.SablonNo
  left join SablonRenk on SablonRenk.SablonRenkID = DesenVaryantSablon.SablonRenkId
  left join Kullanici OK on OK.KullaniciId = PartiSablon.OkeyleyenKullaniciId
  left join Kullanici IK on IK.KullaniciId = PartiSablon.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId = PartiSablon.KullaniciId
  left join PartiStok on PartiStok.PartiSinifID = PartiSablon.PartiSinifID
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join Parti on Parti.PartiKayitId = PartiSablon.PartiKayitId
  where PartiSablon.PartiKayitId = :APartiKayitId
  order by PartiSablon.PartiKayitId, PartiSablon.SablonNo
  into :PartiSablonId, :SablonNo, :Acilim1, :Acilim2, :HazirlananMiktar, :KullanimMiktari, :DepoId, :StoktanDusuldu,
       :SablonRenkId, :SablonRenkAdi, :DolulukOrani, :MeshId, :PatMiktari, :EmulsiyonUrunId,
       :OkeyleyenKullaniciId, :OkeyleyenKullaniciAdi, :OkeyTarihi, :PartiSinifId, :Sira, :SablonAdi, :RenkId, :RenkAdi, :Rakle, :Pres,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :FikseSicakligi, :MusteriSiparisNo, :FBoyaSiparisKayitId, :PartiSablonIsEmriID, :BidonSayisi, :BidonKapasite do
  begin
    Acilim = Format_Float('',Acilim1) || ' / ' || Format_Float('',Acilim2);

    if (MusteriSiparisNo is null) then
    begin
      for select Distinct MusteriSiparisNo from BoyaSiparisDetay where BoyaSiparisKayitId = :FBoyaSiparisKayitId into :FMusteriSiparisNo do
      begin
        if (MusteriSiparisNo is null) then
          MusteriSiparisNo = FMusteriSiparisNo;
        else if (Char_Length(MusteriSiparisNo ||'-'|| FMusteriSiparisNo) < 256) then
          MusteriSiparisNo = MusteriSiparisNo ||'-'|| FMusteriSiparisNo;
      end
    end

    select Sum(iif(PartiSablonIlave.IlaveTipi = 1, PartiSablonIlave.HazirlananMiktar, 0)), Sum(iif(PartiSablonIlave.IlaveTipi = 2, PartiSablonIlave.HazirlananMiktar, 0))
    from PartiSablonIlave where PartiSablonIlave.PartiSablonId = :PartiSablonId into :IlaveMiktari, :DuzeltmeMiktari;

    StokDusumStokId = Null;
    StokDusumFisNo  = Null;
    Select Max(Stok.StokId), List(Stok.SeriNo||Stok.FisNo , ',') from Stok where Stok.PartiKayitID = :APartiKayitId and SablonRecetesi = 1 into :StokDusumStokId, :StokDusumFisNo;

    suspend;
  end
end
