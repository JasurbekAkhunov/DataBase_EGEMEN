CREATE PROCEDURE LABDANSABLONILERECETEOLUSTUR(
    ALABRECETEATISID INTEGER,
    ASABLONRECETEID VARCHAR(20),
    AKULLANICIID INTEGER,
    AMAKINAPLANLAMAGRUPID VARCHAR(10),
    AGRUPPROSESNO VARCHAR(10))
AS
declare variable LabReceteKodu varchar(20);
declare variable AtisNoStr     Varchar(20);
declare variable ReceteId      varchar(20);
declare variable Str           varchar(50);
declare variable Adi           varchar(40);
declare variable FirmaId       varchar(20);
declare variable UstAsamaId    integer;
declare variable ReceteTuruId  integer;
declare variable LabReceteGrupId          integer;
declare variable LabRenkGrupId          integer;
declare variable RenkDerinligiId        integer;
declare variable CalismaTarihi TimeStamp;
declare variable RenkOkey      integer;
declare variable KayitSayisi   integer;
declare variable LabReceteId          integer;
declare variable ReceteGrafikId           varchar(10);
declare variable Migrasyon  integer;

declare variable ReceteAsamaId            integer;
declare variable ReceteAltAsamaId         integer;
declare variable ReceteAsamaSablonId      integer;
declare variable UrunId                   varchar(20);
declare variable UrunBirimId              integer;
declare variable Miktar                   Double precision;
declare variable Banyo                    Double precision;
declare variable MiktarYuzde    Double Precision;

declare variable LUrunId                   varchar(20);
declare variable LUrunBirimId              integer;
declare variable LMiktarYuzde              Double precision;
declare variable LUrunTipiId               integer;

declare variable MinSira                   integer;
declare variable Sira                      integer;
declare variable Ilave              integer;
declare variable SokumUzeriBoya     integer;

declare variable FLabReceteKoduTipi    integer;
declare variable CalismaReceteGrafikID        varchar(10);
declare variable BanyoSayisi integer;
declare variable ToplamSicaklikDegisimi integer;
declare variable BoyaTipiId integer;
declare variable FAtisRenkSira  integer;
declare variable FAtisRenkColor integer;
declare variable FAtisRenkKodu        varchar(50);
declare variable FAtisRenkAdi         varchar(120);
Declare variable FNotu    Blob ;
Declare variable FNotu2    Blob ;
Declare variable Aciklama varchar(255);
declare variable BoyaKimyasalUrunTipiId integer;
declare variable OncekiReceteAltAsamaId integer;
declare variable FBoyaAsamasimi integer;
declare variable FMinSira integer;
declare variable FMaxSira integer;
declare variable FReceteAltAsamaId integer;
declare variable FWReceteAltAsamaId integer;
declare variable FBoyaEklendi integer;
declare variable FReceteOtomatikIstekOlustur integer;
declare variable FKayitSayisi integer;
begin
  if (AMakinaPlanlamaGrupId = '') then AMakinaPlanlamaGrupId = Null;
  if (AGrupProsesNo = '') then AGrupProsesNo = Null;

  BoyaKimyasalUrunTipiId = 0;

  select LabReceteKoduTipi, ReceteOtomatikIstekOlustur from SISTEM where SISTEMID = 1 into :FLabReceteKoduTipi, :FReceteOtomatikIstekOlustur;

  Select ReceteId from Recete where Recete.LabReceteAtisId = :ALabReceteAtisId into :ReceteId;

  Select LabRecete.LabReceteGrupId, LabReceteCalisma.ReceteGrafikID from LabReceteAtis
    left join LabRecete on LabRecete.LabReceteId = LabReceteAtis.LabReceteId
    left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
    where LabReceteAtis.LabReceteAtisId = :ALabReceteAtisId
    into :LabReceteGrupId, :CalismaReceteGrafikID;

  if (ReceteId is null) then
  begin
    Select LabRecete.LabReceteId, LabRecete.LabReceteKodu, LabReceteAtis.AtisNoStr, LabRecete.Adi, LabRecete.FirmaId, Coalesce(LabReceteCalisma.UstAsamaId, LabRecete.UstAsamaId), LabRecete.LabReceteGrupId,
           LabRecete.LabRenkGrupId, LabRecete.RenkDerinligiId, LabReceteCalisma.CalismaTarihi, LabReceteAtis.Okey, LabRecete.ReceteTuruId,
           LabReceteAtis.Migrasyon, LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya, LabRecete.Notu from LabReceteAtis
    left join LabRecete on LabRecete.LabReceteId = LabReceteAtis.LabReceteId
    left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
    where LabReceteAtis.LabReceteAtisId = :ALabReceteAtisId
    into :LabReceteId, :LabReceteKodu, :AtisNoStr, :Adi, :FirmaId, :UstAsamaId, :LabReceteGrupId, :LabRenkGrupId, :RenkDerinligiId,  :CalismaTarihi,
         :RenkOkey, :ReceteTuruId, :Migrasyon, :Ilave, :SokumUzeriBoya, :FNotu;

    Str = LabReceteKodu || '.' || AtisNoStr;

    if ( Char_Length(Str) > 20) then exception ReceteKoduAlaniYetersiz;

    ReceteId = Str;

    Select Count(*) from Recete where Recete.ReceteId = :ReceteId into :KayitSayisi;
    if (KayitSayisi = 1) then Exception LabReceteManuelGirilmis;

    Select Recete.ReceteGrafikId, Recete.BanyoSayisi, Recete.ToplamSicaklikDegisimi, Recete.Notu from Recete where Recete.ReceteId = :ASablonReceteId into :ReceteGrafikId, :BanyoSayisi, :ToplamSicaklikDegisimi, :FNotu2;
    if (FNotu is Null) then FNotu = '';
    if (FNotu2 is Null) then FNotu2 = '';

    insert into Recete(ReceteId, SablonReceteId, ReceteTuruId, Adi, FirmaId, UstAsamaId, CalismaTarihi, RenkOkey, LabReceteId, LabReceteAtisId, LabReceteGrupId, LabRenkGrupId, RenkDerinligiId, ReceteGrafikId, Migrasyon, Ilave, SokumUzeriBoya, BanyoSayisi, ToplamSicaklikDegisimi, InsertKullaniciId, KullaniciId, MakinaPlanlamaGrupId, GrupProsesNo, Notu)
          values(:ReceteId, :ASablonReceteId, :ReceteTuruId, :Adi, :FirmaId, :UstAsamaId, :CalismaTarihi, :RenkOkey, :LabReceteId, :ALabReceteAtisId, :LabReceteGrupId, :LabRenkGrupId, :RenkDerinligiId,  :ReceteGrafikId, :Migrasyon, :Ilave, :SokumUzeriBoya, :BanyoSayisi, :ToplamSicaklikDegisimi, :AKullaniciId, :AKullaniciId, :AMakinaPlanlamaGrupId, :AGrupProsesNo, :FNotu || '  ' || :FNotu2 );
  end
  else
  begin
    Select count(*) from ReceteDetay where ReceteDetay.ReceteId = :ReceteId into :KayitSayisi;
    if (KayitSayisi > 0) then exception UretimRecetesiZatenVar;
    Update Recete Set SablonReceteId = :ASablonReceteId where Recete.ReceteId = :ReceteId;
  end

  Update Recete Set OzelDurumLinkIslemi = 1 where ReceteId = :ReceteId;

  if (ASablonReceteId is not null or ASablonReceteId = '') then
  begin
    OncekiReceteAltAsamaId = null;
    For Select ReceteDetay.ReceteAsamaId,  ReceteDetay.ReceteAltAsamaId, ReceteDetay.ReceteAsamaSablonId, ReceteDetay.UrunId, ReceteDetay.UrunBirimId, ReceteDetay.Miktar,
               ReceteDetay.Banyo, ReceteDetay.ReceteGrafikId, ReceteDetay.AtisRenkSira, ReceteDetay.AtisRenkColor, ReceteDetay.AtisRenkKodu, ReceteDetay.AtisRenkAdi,
               ReceteDetay.Aciklama from ReceteDetay
      left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
      where ReceteDetay.ReceteId = :ASablonReceteId
      order by ReceteDetay.ReceteId, ReceteDetay.Sira
      into :ReceteAsamaId, :ReceteAltAsamaId, :ReceteAsamaSablonId, :UrunId, :UrunBirimId, :Miktar, :Banyo, :ReceteGrafikId, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu,
           :FAtisRenkAdi, :Aciklama do
    begin
      insert into ReceteDetay(ReceteId, Sira, ReceteAsamaId, ReceteAltAsamaId, ReceteAsamaSablonId, UrunId, UrunBirimId, Miktar, Banyo, ReceteGrafikId, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, Aciklama, InsertKullaniciId, KullaniciId)
                  values(:ReceteId, -1, :ReceteAsamaId, :ReceteAltAsamaId, :ReceteAsamaSablonId, :UrunId, :UrunBirimId, :Miktar, :Banyo, :ReceteGrafikId, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi, :Aciklama, :AKullaniciId, :AKullaniciId);
      OncekiReceteAltAsamaId =  :ReceteAltAsamaId;
    end


    for Select ReceteAsama.BoyaTipiId, Min(Sira),ReceteDetay.ReceteAsamaId from ReceteDetay
      left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
      where ReceteDetay.ReceteId = :ReceteId and
            ReceteDetay.ReceteAsamaId is not null and
            ReceteAsama.BoyaAsamasimi = 1
      group by BoyaTipiId,ReceteDetay.ReceteAsamaId
      order by Min(Sira) desc
      into :BoyaTipiId, :MinSira, :ReceteAsamaId do
    begin
      if (MinSira is not null) then
      begin
        if (CalismaReceteGrafikID is not null) then
          Update ReceteDetay Set ReceteGrafikID = :CalismaReceteGrafikID where ReceteDetay.ReceteID = :ReceteId and ReceteDetay.Sira = :MinSira;

        Sira = null;
        if (FLabReceteKoduTipi <> 8 and FLabReceteKoduTipi <> 1) then /* ege boya, boybo ve selcuk dısındakiler icin */
        begin
          Select Min(Sira) from ReceteDetay
            left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
            where ReceteDetay.ReceteId = :ReceteId and ReceteDetay.Sira > :MinSira and
                  ReceteDetay.ReceteAsamaId is not null
            into :Sira;
        end
        else /* ege boyaya ve selcuk iplik icin özel recete asamasının hemen altına boyaları koyuyoruz */
        begin
          Select Min(Sira) from ReceteDetay
            left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
            where ReceteDetay.ReceteId = :ReceteId and ReceteDetay.Sira > :MinSira
            into :Sira;
        end

        if (Sira is null) then Sira = -1;
        FBoyaEklendi = 0;
        For Select LabReceteUrun.UrunId, LabReceteUrun.UrunBirimId, LabReceteUrun.MiktarYuzde, Urun.UrunTipiId, LabReceteAtisRenk.Sira, LabReceteAtisRenk.Color, LabReceteAtisRenk.RenkKodu, LabReceteAtisRenk.RenkAdi from LabReceteUrun
          left join Urun on Urun.UrunId = LabReceteUrun.UrunId
          left join LabReceteAtisRenk on LabReceteAtisRenk.LabReceteAtisRenkId = LabReceteUrun.LabReceteAtisRenkId
          where LabReceteUrun.LabReceteAtisId = :ALabReceteAtisId and
          ((:BoyaTipiId = 0 and (Urun.UrunTipiId <> 0 or Urun.BoyaTipiId = 0)) or
          (:BoyaTipiId <> 0 and (Urun.BoyaTipiId = :BoyaTipiId)))
          order by LabReceteAtisRenk.Sira, LabReceteUrun.Sira
          into :LUrunId, :LUrunBirimId, :LMiktarYuzde, :LUrunTipiId, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi do
        begin
           -- STEP 2
          select ReceteAsama.BoyaAsamasimi from ReceteAsama where ReceteAsama.receteasamaid = :ReceteAsamaId into :FBoyaAsamasimi;
          if (FBoyaAsamasimi=1) then
          begin
            if (LUrunTipiId = 0 and LabReceteGrupId <> 3 and LabReceteGrupId <> 0 and LabReceteGrupId <> 5) then
            begin
              select Min(ReceteDetay.Sira), Count(*) from ReceteDetay where ReceteDetay.ReceteID = :ReceteId and ReceteDetay.ReceteAsamaId= :ReceteAsamaId into :FMinSira, :FKayitSayisi;
              if (FKayitSayisi > 1) then Exception KayitHatasi 'Şablonun İçerisinde Aynı Türdeki Boyama Aşaması Birden Fazla Olamaz !';
              
			  select min(ReceteDetay.Sira) from ReceteDetay where  ReceteDetay.ReceteID = :ReceteId and sira > :FMinSira and ReceteDetay.ReceteAsamaId is not null into :FMaxSira;
              select max(ReceteDetay.ReceteAltAsamaId) from ReceteDetay  where ReceteDetay.ReceteID = :ReceteID  and ReceteDetay.Sira > :FMinSira and ReceteDetay.sira < :FMaxSira into :FReceteAltAsamaId;
              if (FReceteAltAsamaId is null) then FReceteAltAsamaId = 0;
              select min(ReceteAltAsama.ReceteAltAsamaId) from ReceteAltAsama where ReceteAltAsama.ReceteAltAsamaId > :FReceteAltAsamaId into :FWReceteAltAsamaId;
              if ((FBoyaEklendi = 0) or (LUrunTipiId <> BoyaKimyasalUrunTipiId)) then
              begin
                if (FReceteOtomatikIstekOlustur = 1) then
                begin
                  insert into ReceteDetay(ReceteId, Sira, ReceteAltAsamaId, Miktar, InsertKullaniciId, KullaniciId)
                  values(:ReceteId, :Sira, :FWReceteAltAsamaId, 0, :AKullaniciId, :AKullaniciId);
                  if (Sira <> -1) then Sira = Sira + 1;
                end
                if (LUrunTipiId = BoyaKimyasalUrunTipiId) then FBoyaEklendi = 1;
              end
              FBoyaEklendi = 1;
              insert into ReceteDetay(ReceteId, Sira, UrunId, UrunBirimId, Miktar, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, InsertKullaniciId, KullaniciId)
              values(:ReceteId, :Sira, :LUrunId, :LUrunBirimId, :LMiktarYuzde, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi, :AKullaniciId, :AKullaniciId);
            end
            else
            begin
              select ReceteDetay.Sira from ReceteDetay where ReceteDetay.ReceteID = :ReceteId  and ReceteDetay.ReceteAsamaId= :ReceteAsamaId into :FMinSira;
              select min(ReceteDetay.Sira) from ReceteDetay where  ReceteDetay.ReceteID = :ReceteId and sira > :FMinSira and ReceteDetay.ReceteAsamaId is not null into :FMaxSira;
              select max(ReceteDetay.ReceteAltAsamaId) from ReceteDetay  where ReceteDetay.ReceteID = :ReceteID  and ReceteDetay.Sira > :FMinSira and ReceteDetay.sira < :FMaxSira into :FReceteAltAsamaId;
              if (FReceteAltAsamaId is null) then FReceteAltAsamaId = 0;
              select min(ReceteAltAsama.ReceteAltAsamaId) from ReceteAltAsama where ReceteAltAsama.ReceteAltAsamaId > :FReceteAltAsamaId into :FWReceteAltAsamaId;
              if ((FBoyaEklendi = 0) or (LUrunTipiId <> BoyaKimyasalUrunTipiId)) then
              begin
                if (FReceteOtomatikIstekOlustur = 1) then
                begin
                  insert into ReceteDetay(ReceteId, Sira, ReceteAltAsamaId, Miktar, InsertKullaniciId, KullaniciId)
                  values(:ReceteId, :Sira, :FWReceteAltAsamaId, 0, :AKullaniciId, :AKullaniciId);
                  if (Sira <> -1) then Sira = Sira + 1;
                end
                if (LUrunTipiId = BoyaKimyasalUrunTipiId) then FBoyaEklendi = 1;
              end
              if (LUrunTipiId <> BoyaKimyasalUrunTipiId) then FBoyaEklendi = 0;
              insert into ReceteDetay(ReceteId, Sira, UrunId, UrunBirimId, Banyo, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, InsertKullaniciId, KullaniciId)
              values(:ReceteId, :Sira, :LUrunId, :LUrunBirimId, :LMiktarYuzde, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi, :AKullaniciId, :AKullaniciId);
            end
          end
          else
          begin
            if (LUrunTipiId = 0 and LabReceteGrupId <> 3 and LabReceteGrupId <> 0 and LabReceteGrupId <> 5) then insert into ReceteDetay(ReceteId, Sira, UrunId, UrunBirimId, Miktar, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, InsertKullaniciId, KullaniciId) values(:ReceteId, :Sira, :LUrunId, :LUrunBirimId, :LMiktarYuzde, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi, :AKullaniciId, :AKullaniciId);
            else insert into ReceteDetay(ReceteId, Sira, UrunId, UrunBirimId, Banyo, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, InsertKullaniciId, KullaniciId) values(:ReceteId, :Sira, :LUrunId, :LUrunBirimId, :LMiktarYuzde, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi, :AKullaniciId, :AKullaniciId);
          end
           --Step 2
          if (Sira <> -1) then Sira = Sira + 1;
        end
      end
    end
  end

  if (FLabReceteKoduTipi = 8) then /* Ege boya icin boya miktarına göre kimyasalları hesaplıyoruz */
  begin
    for Select ReceteDetay.ReceteAsamaId, ReceteDetay.Sira from ReceteDetay
      left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
      where ReceteDetay.ReceteID = :ReceteId and ReceteDetay.ReceteAsamaID is not null and ReceteAsama.BoyaAsamasimi = 1
      into :ReceteAsamaId, :Sira do
    begin
      execute procedure ReceteDetayKimyasalHesapla(:ReceteId, :ReceteAsamaId, :Sira);
    end
  end

  Update Recete Set OzelDurumLinkIslemi = 0 where ReceteId = :ReceteId;
end
