CREATE PROCEDURE PARTIKUMASBILGIRAPORU(
    ARAPORTIPI INTEGER,
    AASAMAID INTEGER,
    ABASLANGICINSERTTARIHI TIMESTAMP,
    ABITISINSERTTARIHI TIMESTAMP,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER)

    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,

    KAPADEDI INTEGER,
    KILO BOYAMIKTARSIZE,
    ADET INTEGER,
    DENYE VARCHAR(30),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    ISTENENRAPORBOYU ADI100,
    ISTENENOZELTUSE VARCHAR(25),
    DONMEISTENEN VARCHAR(10),
    CEKMEZLIKENISTENEN DOUBLE PRECISION,
    CEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    YIKAMAHASLIGIISTENEN VARCHAR(10),
    ENVERILEN VARCHAR(10),
    GRAMAJVERILEN VARCHAR(10),
    DONMEVERILEN VARCHAR(10),
    CEKMEZLIKENVERILEN DOUBLE PRECISION,
    CEKMEZLIKBOYVERILEN DOUBLE PRECISION,
    CEKMEZLIKENVERILEN2 DOUBLE PRECISION,
    CEKMEZLIKBOYVERILEN2 DOUBLE PRECISION,
    HAMKONTROLNOTU MEMO,
    HAMKONTROLDURUMU INTEGER,
    HAMKONTROLKULLANICIID INTEGER,
    HAMKONTROLKULLANICIADI ADI40,
    HAMKONTROLTARIHI TIMESTAMP,
    HAMKONTROLONAYDURUMU INTEGER,
    HAMKONTROLONAYKULLANICIID INTEGER,
    HAMKONTROLONAYKULLANICIADI ADI40,
    HAMKONTROLONAYTARIHI TIMESTAMP,
    HAMTIPIID INTEGER,
    CEKMEZLIKSTANDARDIADI ADI50,
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI ADI40,
    YAPILACAKLAR VARCHAR(1024),
    ILAVEIISLEMLERADI VARCHAR(1024),
    PARTISIRANO INTEGER)
AS
begin
  if (ARaporTipi = 0) then
  begin
    for
      select distinct Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.KapAdedi, PartiStok.Kilo, PartiStok.Adet,
             PartiStok.Denye, PartiStok.EnIstenen, PartiStok.GramajIstenen, PartiStok.IstenenRaporBoyu, PartiStok.IstenenOzelTuse, PartiStok.DonmeIstenen, PartiStok.CekmezlikEnIstenen,
             PartiStok.CekmezlikBoyIstenen, PartiStok.YikamaHasligiIstenen, PartiStokFiziKKontrol.EnVerilen, PartiStokFiziKKontrol.GramajVerilen, PartiStokFiziKKontrol.DonmeVerilen,

             BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi , PartiStokYapilacaklar.Yapilacaklar, PartiStokYapilacaklar.IlaveIslemlerAdi, PartiStok.PartiSiraNo FROM Parti
      left join PartiKumasKBilgi ON PartiKumasKBilgi.PartiKayitId = Parti.PartiKayitId
      left join PartiStok ON PartiStok.PartiSinifId = PartiKumasKBilgi.PartiSinifId
      left join PartiStokFiziKKontrol ON PartiStokFiziKKontrol.PARTISINIFID = PartiStok.PARTISINIFID and PartiStokFiziKKontrol.SonKayit = 1
      left join Kullanici KK ON KK.KULLANICIID = PartiStok.HamKontrolKullaniciId
      left join Kullanici OK ON OK.KULLANICIID = PartiStok.HamKontrolOnayKullaniciId
      left join BoyaSiparis ON BoyaSiparis.BOYASIPARISKAYITID = Parti.BOYASIPARISKAYITID
      left join Kullanici SAK ON SAK.KULLANICIID = BoyaSiparis.SIPARISIALANKULLANICIID
      left join CekmezlikStandardi ON CekmezlikStandardi.CEKMEZLIKSTANDARDIID = PartiStokFiziKKontrol.CEKMEZLIKSTANDARDIID
      left join PartiStokYapilacaklar(PartiStok.partiSinifId) on PartiStokYapilacaklar.PartiSinifID = PartiStok.PartiSinifID
      left join Asama ON Asama.ASAMAID = PARTIKUMASKBILGI.ASAMAID
      where PartiKumasKBilgi.PartiKumasKBilgiId is not null and
           (Parti.PartiYil > :ABaslangicPartiYil or (Parti.PartiYil = :ABaslangicPartiYil and Parti.PartiId >= :ABaslangicPartiId)) and
           (Parti.PartiYil < :ABitisPartiYil or (Parti.PartiYil = :ABitisPartiYil  and Parti.PartiId <= :ABitisPartiId)) and
           (Asama.AsamaId = :AAsamaId or :AAsamaId is null) and
           (:ABaslangicBoyaSiparisYil = 0 or
            ((BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
            (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId))))
      into  :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :HamAdiToplami, :KapAdedi, :Kilo, :Adet,
            :Denye, :EnIstenen,  :GramajIstenen, :IstenenRaporBoyu, :IstenenOzelTuse, :DonmeIstenen,  :CekmezlikEnIstenen,
            :CekmezlikBoyIstenen, :YikamaHasligiIstenen, :EnVerilen, :GramajVerilen, :DonmeVerilen,
            :CekmezlikEnVerilen, :CekmezlikBoyVerilen, :CekmezlikEnVerilen2, :CekmezlikBoyVerilen2,
            :HamKontrolNotu,   :HamKontrolDurumu, :HamKontrolKullaniciId, :HamKontrolKullaniciADI, :HamKontrolTarihi, :HamKontrolOnayDurumu,
            :HamKontrolOnayKullaniciId, :HamKontrolOnayKullaniciADI, :HamKontrolOnayTarihi, :HamTipiId, :CekmezlikStandardiAdi,
            :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi, :Yapilacaklar, :IlaveIislemlerAdi, :PartiSiraNo
    do
    begin
       suspend;
    end
    end
  else
   begin
   For
     select distinct  Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.KapAdedi, PartiStok.Kilo, PartiStok.Adet,
             PartiStok.Denye, PartiStok.EnIstenen, PartiStok.GramajIstenen, PartiStok.IstenenRaporBoyu, PartiStok.IstenenOzelTuse, PartiStok.DonmeIstenen, PartiStok.CekmezlikEnIstenen,
             PartiStok.CekmezlikBoyIstenen, PartiStok.YikamaHasligiIstenen, PartiStokFiziKKontrol.EnVerilen, PartiStokFiziKKontrol.GramajVerilen, PartiStokFiziKKontrol.DonmeVerilen,
             PartiStokFiziKKontrol.CekmezlikEnVerilen, PartiStokFiziKKontrol.CekmezlikBoyVerilen, PartiStokFiziKKontrol.CekmezlikEnVerilen2, PartiStokFiziKKontrol.CekmezlikBoyVerilen2,
             PartiStok.HamKontrolNotu, PartiStok.HamKontrolDurumu, PartiStok.HamKontrolKullaniciId, KK.ADI HamKontrolKullaniciADI, PartiStok.HamKontrolTarihi, PartiStok.HamKontrolOnayDurumu,
             PartiStok.HamKontrolOnayKullaniciId, OK.ADI HamKontrolOnayKullaniciADI, PartiStok.HamKontrolOnayTarihi, PartiStok.HAMTIPIID, CekmezlikStandardi.ADI CekmezlikStandardiAdi,
             BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi , PartiStokYapilacaklar.Yapilacaklar, PartiStokYapilacaklar.IlaveIslemlerAdi, PartiStok.PartiSiraNo FROM  PartiKumasKBilgi
      left join Parti ON  Parti.PartiKayitId = PartiKumasKBilgi.PartiKayitId
      left join Asama ON Asama.ASAMAID = PARTIKUMASKBILGI.ASAMAID
      left join PartiStok ON PartiStok.PartiSinifId = PartiKumasKBilgi.PartiSinifId
      left join PartiStokFiziKKontrol ON PartiStokFiziKKontrol.PartiSinifId = PARTISTOK.PartiSinifId and PartiStokFiziKKontrol.SonKayit = 1
      left join Kullanici KK ON KK.KULLANICIID = PARTISTOK.HamKontrolKullaniciId
      left join Kullanici OK ON OK.KULLANICIID = PARTISTOK.HamKontrolOnayKullaniciId
      left join BoyaSiparis ON BoyaSiparis.BOYASIPARISKAYITID = Parti.BOYASIPARISKAYITID
      left join Kullanici SAK ON SAK.KULLANICIID = BoyaSiparis.SiparisiAlanKullaniciId
      left join CekmezlikStandardi ON CekmezlikStandardi.CEKMEZLIKSTANDARDIID = PartiStokFiziKKontrol.CEKMEZLIKSTANDARDIID
      left join PartiStokYapilacaklar(PArtiStok.partiSinifId) on PartiStokYapilacaklar.PartiSinifID = PartiStok.PartiSinifID
      where  PartiKumasKBilgi.PartiKumasKBilgiId is not null and
            (partikumaskbilgi.inserttarihi >= :ABaslangicInsertTarihi and partikumaskbilgi.inserttarihi <= :ABitisInsertTarihi ) and
            (Asama.AsamaId = :AAsamaId or :AAsamaId is null) and
            (:ABaslangicBoyaSiparisYil = 0 or
            ((BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
            (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId))))
     into  :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :HamAdiToplami, :KapAdedi, :Kilo, :Adet,
           :Denye, :EnIstenen,  :GramajIstenen, :IstenenRaporBoyu, :IstenenOzelTuse, :DonmeIstenen,  :CekmezlikEnIstenen,
           :CekmezlikBoyIstenen, :YikamaHasligiIstenen, :EnVerilen, :GramajVerilen, :DonmeVerilen,
           :CekmezlikEnVerilen, :CekmezlikBoyVerilen, :CekmezlikEnVerilen2, :CekmezlikBoyVerilen2,
           :HamKontrolNotu,   :HamKontrolDurumu, :HamKontrolKullaniciId, :HamKontrolKullaniciADI, :HamKontrolTarihi, :HamKontrolOnayDurumu,
           :HamKontrolOnayKullaniciId, :HamKontrolOnayKullaniciADI, :HamKontrolOnayTarihi, :HamTipiId, :CekmezlikStandardiAdi,
           :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi, :Yapilacaklar, :IlaveIislemlerAdi, :PartiSiraNo
    do
    begin
       suspend;
    end
   end
