CREATE PROCEDURE MUSTERIRENKCALISMASITAKIPRAPORU(
    ARAPORTIPI INTEGER,
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    LABRECETEID INTEGER,
    LABRECETECALISMAID INTEGER,
    LABRECETEATISID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    LABRECETETALEPTARIHI TIMESTAMP,
    TEMSILCIID INTEGER,
    TEMSILCIADI VARCHAR(40),
    FASON INTEGER,
    FASONSATISSTR VARCHAR(5),
    HAMADI VARCHAR(255),
    CALISMATARIHI TIMESTAMP,
    ILAVE INTEGER,
    SOKUMUZERIBOYA INTEGER,
    RPTSTR VARCHAR(3),
    ILAVESOKUMUZERIBOYASTR VARCHAR(20),
    KRITIKCALISMASI INTEGER,
    KRITIKCALISMASISTR VARCHAR(10),
    TERMINTARIHI TIMESTAMP,
    GIDISTARIHI TIMESTAMP,
    DURUMUSTR VARCHAR(6),
    OKEYKRITIKTARIHI TIMESTAMP,
    LABRECETENOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    NOTU VARCHAR(1023),
    LABRECETEGRUPID INTEGER,
    DOVIZID DOVIZSIZE,
    CAPTION VARCHAR(5))
AS
declare variable FAtisNoStr Varchar(20);
declare variable Rptmi   integer;
declare variable FAtisNotu    Blob;
begin

  if (ARaporTipi = 0) then
  begin
    For Select LabReceteCalisma.LabReceteId, LabReceteAtis.LabReceteCalismaId, LabReceteAtis.LabReceteAtisId, LabRecete.LabReceteKodu, LabRecete.FirmaId,
        Firma.Adi, LabRecete.TalepTarihi, LabReceteAtis.AtisNoStr, LabRecete.Adi, LabRecete.Fason, Ham.Adi, LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi,
               LabReceteAtis.GidisTarihi, LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi, LabRecete.Notu, LabReceteAtis.Notu, LabReceteCalisma.Ilave,
               LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, LabReceteCalisma.Rptmi, Firma.TemsilciId, Kullanici.Adi, LabRecete.labReceteGrupId,
               LabRecete.DovizID, LabReceteGrup.Caption from LabReceteCalisma
      left join LabRecete on LabRecete.LabReceteID = LabReceteCalisma.LabReceteId
      left join LabReceteGrup on LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
      left join Firma on Firma.FirmaID = LabRecete.FirmaId
      left join LabReceteAtis on LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
      left join Ham on Ham.HamId = LabReceteCalisma.HamId
      left join Kullanici on Kullanici.KullaniciId = Firma.TemsilciId
      where LabReceteCalisma.CalismaTarihi >= :ABaslangicTarihi and LabReceteCalisma.CalismaTarihi <= :ABitisTarihi and
            (LabReceteCalisma.FirmaId = :AFirmaId or :AFirmaId is null) and LabReceteAtis.LabReceteAtisID is not null
      into :LabReceteId, :LabReceteCalismaId, :LabReceteAtisId, :LabReceteKodu, :FirmaId, :FirmaAdi, :LabReceteTalepTarihi, :FAtisNoStr,  :LabReceteAdi, :Fason, :HamAdi, :CalismaTarihi, :TerminTarihi, :GidisTarihi,
           :DurumuStr, :OkeyKritikTarihi, :LabReceteNotu, :FAtisNotu, :Ilave, :SokumUzeriBoya, :KritikCalismasi, :Rptmi, :TemsilciId, :TemsilciAdi, :LabReceteGrupId, :DovizID, :Caption do
    begin
      LabReceteKodu = LabReceteKodu || '.' || FAtisNoStr;

      if (Fason = 0) then FasonSatisStr = 'Satış';
      else FasonSatisStr = 'Fason';

      if (Rptmi = 1) then RptStr = 'Rpt';
      else RptStr = '';

      if (Ilave = 1) then IlaveSokumUzeriBoyaStr = 'İlave';
      else if (SokumUzeriBoya = 1) then IlaveSokumUzeriBoyaStr = 'Söküm Üzeri Boya';
      else IlaveSokumUzeriBoyaStr = '';

      if (KritikCalismasi = 1) then KritikCalismasiStr = 'Kritik';
      else KritikCalismasiStr = '';

      if (char_length(FAtisNotu) > 1023) then Notu = Left(FAtisNotu, 1023);
      else Notu = Left(FAtisNotu, char_length(FAtisNotu));

      if (AFirmaID is not null or GidisTarihi is null) then suspend;
    end
  end
  else
  begin
    For Select LabReceteCalisma.LabReceteId, LabReceteAtis.LabReceteCalismaId, LabReceteAtis.LabReceteAtisId, LabRecete.LabReceteKodu, LabRecete.FirmaId, Firma.Adi, LabRecete.TalepTarihi, LabReceteAtis.AtisNoStr, LabRecete.Adi, LabRecete.Fason, Ham.Adi, LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi,
               LabReceteAtis.GidisTarihi, LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi, LabRecete.Notu, LabReceteAtis.Notu, LabReceteCalisma.Ilave,
               LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, LabReceteCalisma.Rptmi, Firma.TemsilciId, Kullanici.Adi, LabRecete.labReceteGrupId, LabRecete.DovizID, LabReceteGrup.Caption from LabReceteAtis
      left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaID = LabReceteAtis.LabReceteCalismaID
      left join LabRecete on LabRecete.LabReceteID = LabReceteCalisma.LabReceteId
      left join LabReceteGrup on LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
      left join Firma on Firma.FirmaID = LabRecete.FirmaId
      left join Ham on Ham.HamId = LabReceteCalisma.HamId
      left join Kullanici on Kullanici.KullaniciId = Firma.TemsilciId
      where LabReceteAtis.GidisTarihi >= :ABaslangicTarihi and LabReceteAtis.GidisTarihi <= :ABitisTarihi and
            (LabReceteCalisma.FirmaId = :AFirmaId or :AFirmaId is null)
      into :LabReceteId, :LabReceteCalismaId, :LabReceteAtisId, :LabReceteKodu, :FirmaId, :FirmaAdi, :LabReceteTalepTarihi, :FAtisNoStr, :LabReceteAdi, :Fason, :HamAdi, :CalismaTarihi, :TerminTarihi, :GidisTarihi,
           :DurumuStr, :OkeyKritikTarihi, :LabReceteNotu, :FAtisNotu, :Ilave, :SokumUzeriBoya, :KritikCalismasi, :Rptmi, :TemsilciId, :TemsilciAdi, :LabReceteGrupId, :DovizID, :Caption do
    begin
      LabReceteKodu = LabReceteKodu || '.' || FAtisNoStr;

      if (Fason = 0) then FasonSatisStr = 'Satış';
      else FasonSatisStr = 'Fason';

      if (Rptmi = 1) then RptStr = 'Rpt';
      else RptStr = '';

      if (KritikCalismasi = 1) then KritikCalismasiStr = 'Kritik';
      else KritikCalismasiStr = '';

      if (Ilave = 1) then IlaveSokumUzeriBoyaStr = 'İlave';
      else if (SokumUzeriBoya = 1) then IlaveSokumUzeriBoyaStr = 'Söküm Üzeri Boya';
      else IlaveSokumUzeriBoyaStr = '';

      if (char_length(FAtisNotu) > 1023) then Notu = Left(FAtisNotu, 1023);
      else Notu = Left(FAtisNotu, char_length(FAtisNotu));

      suspend;
    end
  end
end
