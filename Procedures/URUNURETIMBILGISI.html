CREATE PROCEDURE URUNURETIMBILGISI(
    AURUNURETIMID INTEGER)
RETURNS (
    URUNURETIMID INTEGER,
    TARIH TIMESTAMP,
    URUNRECETEURUNID VARCHAR(20),
    ADI VARCHAR(320),
    MIKTAR DOUBLE PRECISION,
    DEPOID VARCHAR(40),
    DEPOADI VARCHAR(30),
    URUNBIRIMID INTEGER,
    BIRIMID VARCHAR(8),
    KULLANICIID INTEGER,
    KULLANICIADI VARCHAR(40),
    STOKTANDUSULDU INTEGER)
AS
declare variable ks integer;
Begin
  for
    select UrunUretim.UrunUretimId,UrunUretim.Tarih,UrunUretim.UrunReceteUrunId, Urun.Adi, UrunUretim.Miktar,
           UrunUretim.DepoId,Depo.Adi DepoAdi, UrunBirim.UrunBirimId,UrunBirim.BirimId, UrunUretim.KullaniciId, Kullanici.Adi KullaniciAdi
    from UrunUretim
      left join Urun on Urun.urunid = UrunUretim.Urunreceteurunid
      left join UrunBirim on UrunBirim.UrunBirimid = UrunUretim.UrunBirimId
      left join Depo on Depo.DepoId = UrunUretim.DepoId
      left join Kullanici on kullanici.KullaniciId = UrunUretim.KullaniciId
    where UrunUretim.UrunUretimId = :AUrunUretimId
    into :UrunUretimId, :Tarih, :UrunReceteUrunId, :Adi, :Miktar, :DepoId, :DepoAdi, :UrunBirimId, :BirimId,
         :KullaniciId, :KullaniciAdi
  do
  begin
    StoktanDusuldu = 0;
    select count(*) from Stok where Stok.UrunUretimId = :AUrunUretimId into :KS;
    if (ks>0) then StoktanDusuldu = 1;

    suspend;
  end
end
