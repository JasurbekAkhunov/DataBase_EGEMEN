CREATE PROCEDURE HGISLEMAKISIRAPORU2(
    AHGSIPARISYIL INTEGER,
    AHGSIPARISID INTEGER,
    AMODELURETIMISLEMTIPIID MODELGRUPKODUSIZE)
RETURNS (
    HGSIPARISDETAYID INTEGER,
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    FIREORANI DOUBLE PRECISION,
    FAZLAKESIMORANI DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    RAFID VARCHAR(10),
    DEPOID DEPOKODUSIZE,
    HESAPLANANMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI NUMERIC(12,3),
    KALITENO INTEGER,
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    URETICIADI VARCHAR(40),
    URETIMPARTINO VARCHAR(30),
    ROWNUM INTEGER)
AS
declare variable SiparisFirmaId varchar(20);
declare variable StokFirmaId varchar(20);
declare variable OrmeFirmaKodumuz varchar(20);
declare variable KS integer;
begin
  RowNum = 0;
  select count(*) from ModelUretimIslemTipi where ModelUretimIslemTipi.ModelUretimIslemTipiId = :AModelUretimIslemTipiId and ModelUretimIslemTipi.KesimIslemi = 1 into :KS;
  if (KS = 0 and AModelUretimIslemTipiId is not null) then Exit;

  Select OrmeFirmaKodumuz from sistem where sistemid = 1 into :OrmeFirmaKodumuz;

  For Select HGSiparisDetayKumas.HamId, Ham.HamTipiId, HamTipi.Adi, Ham.Adi, HGSiparisDetayKumas.RenkId, Renk.Adi, HGSiparisDetayKumas.Fireorani,
             HGSiparisDetayKumas.FazlaKesimOrani,
             ((HGSiparisDetayKumas.Miktar * iif(HGSiparisDetayKumas.FireOrani > 0, HGSiparisDetayKumas.FireOrani / 100 + 1, 1)) *
                iif(HGSiparisDetayKumas.FazlaKesimOrani > 0, HGSiparisDetayKumas.FazlaKesimOrani / 100 + 1, 1)),
      HgSiparis.FirmaId, HgSiparis.Fason, HgSiparisDetay.HgSiparisDetayId
 from HgSiparis
    left join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
    left join HGSiparisDetayKumas on HGSiparisDetayKumas.HgSiparisDetayId = HgSiparisDetay.HgSiparisDetayId
    left join Ham on Ham.HamId = HGSiparisDetayKumas.HamId
    Left Join HamTipi On HamTipi.HamTipiId = Ham.HamTipiId
    left join Renk on Renk.RenkId = HGSiparisDetayKumas.RenkId
    where
      HgSiparisDetay.Kapali = 0 and
      (HgSiparis.HgSiparisYil = :AHgSiparisYil or :AHgSiparisYil is null) and
      (HgSiparis.HgSiparisId = :AHgSiparisId or :AHgSiparisId is null)
  into  :HamId, :HamTipiId, :HamTipiAdi, :HamAdi, :RenkId, :RenkAdi, :FireOrani, :FazlaKesimOrani, :HesaplananMiktar, :SiparisFirmaId, :Fason, :HgSiparisDetayId do
  begin
    if (Fason = 1) then
    begin
      FasonStr = 'Fason';
      StokFirmaID = SiparisFirmaId;
    end
    else
    begin
      FasonStr = '';
      StokFirmaID = OrmeFirmaKodumuz;
    end

    OrmeRafId = Null;
    RafId = Null;
    StokGirisNo = Null;
    KalanMiktar = Null;
    KalanKapAdedi = Null;
    Lot = Null;
    MelanjKodu = Null;
    KaliteNo = Null;
    UreticiAdi = Null;
    UretimPartiNo = Null;
    DepoId = Null;

    For Select OrmeRaf.OrmeRafId, OrmeRaf.RafId, OrmeRaf.StokGirisNo, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari, OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi,
               OrmeRaf.Lot, OrmeRaf.MelanjKodu, OrmeRaf.KaliteNo, Uretici.Adi, UretimPartiNo, DepoRaf.DepoId from OrmeRaf
      left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      where OrmeRaf.HamID = :HamId and OrmeRaf.StokFirmaId = :StokFirmaID and OrmeRaf.StokVar = 1 and
            StrKarsilastir(OrmeRaf.RenkID, :RenkId) = 1 and OrmeRaf.FasonStok = :Fason
      into :OrmeRafId, :RafId, :StokGirisNo, :KalanMiktar, :KalanKapAdedi, :Lot, :MelanjKodu,  :KaliteNo, :UreticiAdi, :UretimPartiNo, :DepoId do
    begin
      RowNum = RowNum + 1;
      suspend;
    end

    if (OrmeRafId is null) then
    begin
      RowNum = RowNum + 1;
      suspend;
    end
  end
end
