CREATE PROCEDURE ORMESTOKKAYDIRMAFISILISTESI(
    ABASLANGICFISTARIHI TIMESTAMP,
    ABITISFISTARIHI TIMESTAMP,
    ABASLANGICFISYIL INTEGER,
    ABASLANGICFISNO INTEGER,
    ABITISFISYIL INTEGER,
    ABITISFISNO INTEGER)
RETURNS (
    ORMESTOKKAYDIRMAID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    ESKIHAMADI VARCHAR(255),
    YENIHAMADI VARCHAR(255),
    ESKIBOYASIPARISIDSTR VARCHAR(30),
    YENIBOYASIPARISIDSTR VARCHAR(30),
    HAMRENKADI VARCHAR(50),
    RENKADI VARCHAR(50),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    CIKISPARTINO VARCHAR(30),
    ACIKLAMA VARCHAR(255),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP)
AS
declare variable FBoyaSiparisYil integer;
declare variable FBoyaSiparisId  integer;
declare variable BoyaSiparisYil integer;
declare variable BoyaSiparisId  integer;
begin
  For Select OrmeStokKaydirma.OrmeStokKaydirmaId, OrmeStokKaydirma.FisYil, OrmeStokKaydirma.FisNo, OrmeStokKaydirma.Fistarihi,
             EskiHam.Adi, YeniHam.Adi, EskiBoyaSiparis.BoyaSiparisYil, EskiBoyaSiparis.BoyaSiparisId,
             YeniBoyaSiparis.BoyaSiparisYil, YeniBoyaSiparis.BoyaSiparisId, OrmeStokKaydirma.Aciklama,
             SonRenk.Adi, HamRenk.Adi, OrmeRaf.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, OrmeRaf.CikisPartiNo,
             Ik.Adi, OrmeStokKaydirma.InsertTarihi from OrmeStokKaydirma
    Left Join OrmeRaf on OrmeRaf.OrmeRafId = OrmeStokKaydirma.OrmeRafId
    Left Join Ham EskiHam on EskiHam.HamId = OrmeRaf  .HamId
    Left Join BoyaSiparis EskiBoyaSiparis on EskiBoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
    Left Join Ham YeniHam on YeniHam.HamId = OrmeStokKaydirma.HamId
    Left Join BoyaSiparis YeniBoyaSiparis on YeniBoyaSiparis.BoyaSiparisKayitId = OrmeStokKaydirma.BoyaSiparisKayitId
    Left Join Kullanici IK on IK.KullaniciId = OrmeStokKaydirma.InsertKullaniciId
    left join Renk SonRenk on SonRenk.RenkId = OrmeRaf.Renkid
    left join Desen on Desen.DesenId = OrmeRaf.DesenId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = OrmeRaf.DesenVaryantId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
    left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
    where OrmeStokKaydirma.FisTarihi >= :ABaslangicFisTarihi and OrmeStokKaydirma.FisTarihi <= :ABitisFistarihi and
         (:ABaslangicFisYil is null or (OrmeStokKaydirma.FisYil > :ABaslangicFisYil or (OrmeStokKaydirma.FisYil = :ABaslangicFisYil and OrmeStokKaydirma.FisNo >= :ABaslangicFisNo))) and
         (:ABitisFisYil is null or (OrmeStokKaydirma.FisYil < :ABitisFisYil or (OrmeStokKaydirma.FisYil = :ABitisFisYil and OrmeStokKaydirma.FisNo <= :ABitisFisNo)))
    into :OrmeStokKaydirmaId, :FisYil, :FisNo, :Fistarihi,
         :EskiHamAdi, :YeniHamAdi, :FBoyaSiparisYil, :FBoyaSiparisId, :BoyaSiparisYil, :BoyaSiparisId, :Aciklama,
         :RenkAdi, :HamRenkAdi, :DesenId, :DesenAdi, :VaryantAdi, :CikisPartiNo, :InsertKullaniciAdi, :InsertTarihi do
  begin
    FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';

    if (FBoyaSiparisYil is not null) then
      EskiBoyaSiparisIdStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') ' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
    else
      EskiBoyaSiparisIdStr = '';

    if (BoyaSiparisYil is not null) then
      YeniBoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    else
      YeniBoyaSiparisIdStr = '';

    suspend;
  end
end
