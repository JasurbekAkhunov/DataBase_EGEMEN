    PARTIONAYLAYANKULLANICIADI VARCHAR(40),

    KALITEADI ADI255,

    SIPARISGRUPADI VARCHAR(30),

AS
declare variable FPartiAsamalariId integer;
declare variable SonYapilanSira integer;
declare variable SonrakiSira integer;
declare variable FPartiTamirSinifId     integer;
declare variable Sira   integer;
declare variable FIlaveIslemlerAdi      varchar(30);
declare variable FDegerlendirme integer;
declare variable FLot           varchar(150);
declare variable FMusteriPartiNo varchar(150);
declare variable FPartilememi         integer;
declare variable IlkSira       integer;
declare variable itDesenTasarim integer;
begin
  Select itDesenTasarim from IslemTipiGetir into :itDesenTasarim;

  For Select BolumPlanlama.MakinaPlanlamaGrupId, BolumPlanlama.BolumPlanlamaId, GrupPlanlama.GrupPlanlamaId, GrupPlanlama.SiraNo,
            BolumPlanlama.PartiKayitId, BolumPlanlama.PartiSinifId, BolumPlanlama.PartiAsamalariId, Parti.PartiYil, Parti.PartiId, Parti.AciliyetDerecesiId,
            Parti.IslemTipiId, IslemTipi.Adi, Parti.FirmaId, Firma.Adi, PartiAsamalari.ReceteId, ARecete.Adi, Asama.AsamaId, Asama.Adi, Asama.BoyaAsamasimi,
            Parti.AcilisTarihi, Parti.TerminTarihi, PartiStok.Kilo, PartiStok.Metre, PartiStok.SonCikisKilo, PartiStok.SonCikisMetre, PartiStok.HamAdiToplami,
            BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId, PartiStok.PartiAsamalariId,
            Parti.ReceteId, PRecete.Adi, PartiStokBekle.PartiStokBekleID, PartiStokBekle.Bitti, PartiAsamalari.Sira,
            BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.Lot, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, PartiStok.Denye,
            PartiAsamalari.IlaveProses, PartiAsamalari.IlaveProsesSebebiId, IlaveProsesSebebi.Adi,
            PartiRpt.Degerlendirme, PartiRpt.BaslamaTarihi, PartiRpt.BitisTarihi,
            Parti.OnayDurumu, Parti.OnayTarihi, Parti.OnaylayanKullaniciId, POK.Adi,
            Parti.ReceteYazildi, Parti.RefakatKartiYazildi, BoyaSiparisDetay.KaliteId, Kalite.KaliteAdi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
            Max(HamStokMaster.YikamaSonrasiEn), Max(HamStokMaster.YikamaSonrasiGramaj), PartiStok.ProxyID, PartiStok.PartiSiraNo, BoyaSiparisDetay.BelgeHamAdi, PartiStok.GenelKontrolNotu,
            Ham.Adi2, Ham.Adi3, Ham.Adi4, BolumPlanlama.DegisimLog, Desen.TasarimNo, DesenGravurBoyu.Olcu, DesenSablonEni.Olcu, Ham.ReferansKodu, Desen.SablonSayisi, Parti.AciliyetNotu, AciliyetDerecesi.Adi
            from BolumPlanlama
  Left join GrupPlanlama on GrupPlanlama.BolumPlanlamaId = BolumPlanlama.BolumPlanlamaId
  Left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = BolumPlanlama.PartiAsamalariId
  Left join PartiStok on PartiStok.PartiSinifId = BolumPlanlama.PartiSinifId
  left join PartiStokBekle on PartiStokBekle.PartiSinifId = BolumPlanlama.PartiSinifId and PartiStokBekle.PartiAsamalariId = BolumPlanlama.PartiAsamalariId
  Left join Parti on Parti.PartiKayitId = BolumPlanlama.PartiKayitId
  Left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
  Left join PartiRpt on PartiRpt.PartiKayitId = Parti.PartiKayitId and PartiRpt.SonKayit = 1
  Left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
  Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  Left join Firma on Firma.FirmaId = Parti.FirmaId
  left join IlaveProsesSebebi on IlaveProsesSebebi.IlaveProsesSebebiId = PartiAsamalari.IlaveProsesSebebiId
  left join Recete ARecete on ARecete.ReceteId = PartiAsamalari.ReceteId
  left join Asama on (Asama.AsamaId = PartiAsamalari.AsamaId or Asama.AsamaId = ARecete.UstAsamaId)
  left join PartiTamirSinif on PartiTamirSinif.PartiSinifId = PartiStok.PartiSinifId and PartiTamirSinif.Bitti = 0
  Left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
  Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
  Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
  Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join Recete PRecete on PRecete.ReceteId = Parti.ReceteId
  Left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  left join Desen on Desen.DesenId = DesenVaryant.DesenId
  left join DesenOlcu DesenGravurBoyu on DesenGravurBoyu.DesenOlcuId = Desen.GravurBoyuDesenOlcuId
  left join DesenOlcu DesenSablonEni on DesenSablonEni.DesenOlcuId = Desen.SablonEniDesenOlcuId
  left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
  left join Kullanici POK on POK.KullaniciId  = Parti.OnaylayanKullaniciId
  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  Left join Ham on Ham.HamID = PartiStok.HamID
  where BolumPlanlama.BoyahaneBolumID = :ABoyahaneBolumID and GrupPlanlama.UretimPlanlamaId is null and BolumPlanlama.MakinaPlanlamagrupId is not null
  group By BolumPlanlama.MakinaPlanlamaGrupId, BolumPlanlama.BolumPlanlamaId, GrupPlanlama.GrupPlanlamaId, GrupPlanlama.SiraNo,
            BolumPlanlama.PartiKayitId, BolumPlanlama.PartiSinifId, BolumPlanlama.PartiAsamalariId, Parti.PartiYil, Parti.PartiId, Parti.AciliyetDerecesiId,
            Parti.IslemTipiId, IslemTipi.Adi, Parti.FirmaId, Firma.Adi, PartiAsamalari.ReceteId, ARecete.Adi, Asama.AsamaId, Asama.Adi, Asama.BoyaAsamasimi,
            Parti.AcilisTarihi, Parti.TerminTarihi, PartiStok.Kilo, PartiStok.Metre, PartiStok.SonCikisKilo, PartiStok.SonCikisMetre, PartiStok.HamAdiToplami,
            BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId, PartiStok.PartiAsamalariId,
            Parti.ReceteId, PRecete.Adi, PartiStokBekle.PartiStokBekleID, PartiStokBekle.Bitti, PartiAsamalari.Sira,
            BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.Lot, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, PartiStok.Denye,
            PartiAsamalari.IlaveProses, PartiAsamalari.IlaveProsesSebebiId, IlaveProsesSebebi.Adi,
            PartiRpt.Degerlendirme, PartiRpt.BaslamaTarihi, PartiRpt.BitisTarihi,
            Parti.OnayDurumu, Parti.OnayTarihi, Parti.OnaylayanKullaniciId, POK.Adi, Parti.ReceteYazildi, Parti.RefakatKartiYazildi,
            BoyaSiparisDetay.KaliteId, Kalite.KaliteAdi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, PartiStok.ProxyID, PartiStok.PartiSiraNo,
            BoyaSiparisDetay.BelgeHamAdi, PartiStok.GenelKontrolNotu, Ham.Adi2, Ham.Adi3, Ham.Adi4, BolumPlanlama.DegisimLog, Desen.TasarimNo, DesenGravurBoyu.Olcu, DesenSablonEni.Olcu, Ham.ReferansKodu, Desen.SablonSayisi, Parti.AciliyetNotu, AciliyetDerecesi.Adi

  order by BolumPlanlama.MakinaPlanlamaGrupId, GrupPlanlama.SiraNo, PartiStok.PartiSiraNo
  into :MakinaPlanlamaGrupId, :BolumPlanlamaId, :GrupPlanlamaId, :SiraNo, :PartiKayitId, :PartiSinifId, :PartiAsamalariId, :PartiYil, :PartiId, :AciliyetDerecesiId, :IslemTipiId, :IslemTipiAdi,  :FirmaId, :FirmaAdi,
       :ReceteId, :ReceteAdi, :AsamaId, :AsamaAdi, :BoyaAsamasimi, :PartiAcilisTarihi, :TerminTarihi, :BrutKilo, :BrutMetre, :Kilo, :Metre, :HamAdiToplami,
       :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId, :FPartiAsamalariId, :PartiReceteId, :PartiReceteAdi,
       :PartiStokBekleID, :PartiStokBekleBitti, :Sira, :SiparisRenkAdi, :MusteriRenkKoduveAdi, :MusteriSiparisNo, :SiparisLot, :DesenId, :DesenAdi, :VaryantAdi, :Denye,
       :IlaveProses, :IlaveProsesSebebiId, :IlaveProsesSebebiAdi,
       :FDegerlendirme, :PartiRptBaslamaTarihi, :PartiRptBitisTarihi, :PartiOnayDurumu, :PartiOnayTarihi, :PartiOnaylayanKullaniciId, :PartiOnaylayanKullaniciAdi,
       :ReceteYazildi, :RefakatKartiYazildi, :KaliteId, :KaliteAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
       :YikamaSonrasiEn, :YikamaSonrasiGramaj, :ProxyID, :PartiSiraNo, :BelgeHamAdi, :GenelKontrolNotu, :HamAdi2, :HamAdi3, :HamAdi4, :DegisimLog,
       :DesenTasarimNo, :DesenGravurBoyu, :DesenSablonEni, :ReferansKodu, :SablonSayisi, :AciliyetNotu, :AciliyetDerecesiAdi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    else BoyaSiparisIdStr = '';

    PartiRptDegerlendirmeDurumu = 1;

    if (FDegerlendirme = 1) then
      PartiRptDegerlendirme = 'Çalışılmayacak';
    else if (FDegerlendirme = 2) then
    begin
      PartiRptDegerlendirme = 'Çalışılacak';
      PartiRptDegerlendirmeDurumu = 0;
    end
    else if (FDegerlendirme = 3) then
    begin
      PartiRptDegerlendirme = 'Çalışılacak (Hammadde Bekleniyor';
      PartiRptDegerlendirmeDurumu = 0;
    end
    else PartiRptDegerlendirme = '';

    if (PartiRptBitisTarihi is not null) then
    begin
      PartiRptDegerlendirmeDurumu = 1;
      if (FDegerlendirme = 2 or FDegerlendirme = 3) then PartiRptDegerlendirme = 'Çalışıldı';
    end

    if (PartiOnayDurumu = 1) then PartiOnayDurumuStr = 'Evet';
    else PartiOnayDurumuStr = '';

    if (IlaveProses = 1) then
      IlaveProsesStr = 'Evet';
    else
      IlaveProsesStr = '';

    Select Notu, TerminNotu, PlanlamaNotu, SiparisBrutKilo, SiparisKilo from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = :BoyaSiparisDetayId into :SiparisNotu, :SiparisTerminNotu, :SiparisPlanlamaNotu, :SiparisBrutKilo, :SiparisKilo;

    Select Sum(PartiStok.SonCikisKilo), Sum(PartiStok.SonCikisMetre) from PartiStok where PartiStok.PartiKayitID = :PartiKayitID and PartiStok.PartiStokIptalKayitId is null
      into :PartiKilo, :PartiMetre;

    if (FPartiAsamalariId is not null) then
    begin
      select Asama.Adi, PartiAsamalari.Sira from PartiAsamaStok
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId
        into :SonYapilanAsamaAdi, :SonYapilanSira;
    end
    else
    begin
      SonYapilanSira = -1;
      SonYapilanAsamaAdi = Null;
    end

    select Min(PartiAsamalari.Sira) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.Sira > :SonYapilanSira
      into :SonrakiSira;

    if (SonrakiSira is not null) then
    begin
      select Asama.Adi, PartiAsamalari.PartiAsamalariId from PartiAsamaStok
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :SonrakiSira
        into :SonrakiAsamaAdi, :SiradakiPartiAsamalariId;
    end
    else
    begin
      SonrakiAsamaAdi = Null;
      SiradakiPartiAsamalariId = Null;
    end
    if (SiradakiPartiAsamalariId is null) then ImalataHazir = 0;
    else if (SiradakiPartiAsamalariId = PartiAsamalariId) then ImalataHazir = 1;
    else ImalataHazir = 0;

    if (PartiStokBekleId is null and ImalataHazir = 0) then
    begin
      Select Min(PartiStokBekleId) from PartiStokBekle
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiStokBekle.PartiAsamalariId
        where PartiStokBekle.PartiSinifId = :PartiSinifId and PartiStokBekle.BitirenKullaniciId is null and
              PartiAsamalari.Sira < :Sira
        into :PartiStokBekleId;

      if (PartiStokBekleId is not null) then PartiStokBekleBitti = 0;
    end

    ReceteIsEmriIdStr = Null;
    BirlestirilenPartiler = '';

    if (ReceteId is not null) then
      Select ReceteIsEmriIdStr, BirlestirilenPartiler from PAReceteIsEmriBilgisiGetir(:PartikayitId, :ReceteId, :PartiAsamalariId) into :ReceteIsEmriIdStr, :BirlestirilenPartiler;

    AsamaIlaveIslemler = '';
    for Select IlaveIslemler.Adi from PartiAsamaStok
     left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
     left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
     where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.Asamayadahil = 1 and IlaveIslemler.Adi is not null
     into :FIlaveIslemlerAdi do
    begin
      if (Char_Length(AsamaIlaveIslemler) + Char_Length(FIlaveIslemlerAdi) < 1020) then
      begin
       if (AsamaIlaveIslemler <> '' ) then AsamaIlaveIslemler = AsamaIlaveIslemler || cast(', ' as varchar(2) character set UTF8);
       AsamaIlaveIslemler = AsamaIlaveIslemler || FIlaveIslemlerAdi;
      end
    end

    Lot = '';
    for Select HamStokMaster.Lot from PartiStokDetay
        Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.Lot is not null and HamStokMaster.Lot <> ''
        group by HamStokMaster.Lot
        into :FLot do
    begin
      if (Char_Length(Lot) + Char_Length(FLot) < 148) then
      begin
       if (Lot <> '' ) then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
       Lot = Lot || FLot;
      end
    end

    MusteriPartiNo = '';
    for Select HamStokMaster.MusteriPartiNo from PartiStokDetay
        Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.MusteriPartiNo is not null and HamStokMaster.MusteriPartiNo <> ''
        group by HamStokMaster.MusteriPartiNo
        into :FMusteriPartiNo do
    begin
      if (Char_Length(MusteriPartiNo) + Char_Length(FMusteriPartiNo) < 148) then
      begin
       if (MusteriPartiNo <> '' ) then MusteriPartiNo = MusteriPartiNo || cast(', ' as varchar(2) character set UTF8);
       MusteriPartiNo = MusteriPartiNo || FMusteriPartiNo;
      end
    end

    VaryantTasarimFullProses = 0;
    if (IslemTipiID = itDesenTasarim) then
    begin
      IlkSira = Null;

      select Min(PartiAsamalari.Sira) from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1
        into :IlkSira;

      if (IlkSira is not null) then
      begin
        select Asama.Partilememi from PartiAsamaStok
          left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :IlkSira
          into :FPartilememi;
        if (FPartilememi = 1) then VaryantTasarimFullProses = 1;
      end
    end

    Select GecenSureStr from SonAsamadanGecenSureyiHesapla(:PartiSinifId) into :GecenSure;

    suspend;
  end
