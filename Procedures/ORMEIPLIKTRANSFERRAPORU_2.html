CREATE PROCEDURE ORMEIPLIKTRANSFERRAPORU_2
RETURNS (
    IDSTR VARCHAR(20),
    STOKHAREKETID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(150),
    ESKIPARTIYIL INTEGER,
    ESKIPARTINO INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    IPLIKTALEPIDSTR VARCHAR(30),
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    BOYASIPARISDETAYID INTEGER,
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100))
AS
declare Variable BoyaSiparisYil integer;
declare Variable BoyaSiparisId  integer;
declare Variable IpliktalepYil integer;
declare Variable IpliktalepId  integer;
begin
  For Select OrmeIplikTransferRaporu_1.IdStr, StokHareket.StokHareketId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             Ham.Adi, Renk.Adi, StokHareket.BoyaSiparisKayitId, StokHareket.OrmeSiparisDetayId,
             OrmeSiparisDetay.SiparisSiraNo, StokHareket.BoyaSiparisDetayId, BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id,
             StokHareket.KapAdedi, StokHareket.Miktar, BoyaSiparis.FirmaId, Firma.Adi, StokHareket.Lot,
             StokHareket.MelanjKodu, StokHareket.EskiPartiYil, StokHareket.EskiPartiNo  from OrmeIplikTransferRaporu_1
    left join StokHareket on StokHareket.OrmeIrsaliyeId = OrmeIplikTransferRaporu_1.OrmeIrsaliyeId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join Ham on Ham.HamId = StokHareket.HamId
    left join Renk on Renk.RenkId = StokHareket.RenkId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    where OrmeIplikTransferRaporu_1.OrmeIrsaliyeID is not null
    into :IdStr, :StokHareketId, :BoyaSiparisYil, :BoyaSiparisId, :HamAdi, :RenkAdi, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :BoyaSiparisDetayId, :SiparisGrupId, :SiparisGrup2Id,
         :KapAdedi, :Miktar, :SiparisFirmaId, :SiparisFirmaAdi, :Lot, :MelanjKodu, :EskiPartiYil, :EskiPartiNo do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    suspend;
  end
end
