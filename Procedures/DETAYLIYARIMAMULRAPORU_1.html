CREATE PROCEDURE DETAYLIYARIMAMULRAPORU_1(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIH TIMESTAMP,
    ABITISTARIH TIMESTAMP,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    AFASON INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(50),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    HAMADITOPLAMI VARCHAR(255),
    PARTIKILO NUMERIC(12,3),
    PARTIMETRE NUMERIC(12,3),
    TOPLAMKILO NUMERIC(12,3),
    TOPLAMMETRE NUMERIC(12,3),
    BOYASIPARISIDSTR VARCHAR(30))
AS
declare variable FBoyaSiparisYil integer;
declare variable FBoyaSiparisId  integer;
begin
  for select DetayliYariMamulRaporu_2.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.FirmaId, Firma.Adi,
             Parti.Fason, PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.Metre,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             sum(DetayliYariMamulRaporu_2.Kilo), sum(DetayliYariMamulRaporu_2.Metre) from DetayliYariMamulRaporu_2(:ARaporTipi, :ABaslangicTarih, :ABitisTarih, :ABaslangicPartiYil, :ABaslangicPartiId, :ABitisPartiYil, :ABitisPartiId, :AHamTipiId, :AHamId, :AFason)
    left join PartiStok on PartiStok.PartiSinifID = DetayliYariMamulRaporu_2.PartiSinifId
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
    left join Firma on Firma.FirmaID = Parti.FirmaID
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    group by DetayliYariMamulRaporu_2.PartiKayitId, Parti.PartiYil, Parti.PartiId, partiStok.PartiSinifId, Parti.FirmaId, Firma.Adi,
          Parti.Fason, PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.Metre,
          BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId
    into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :FirmaId, :FirmaAdi, :Fason, :HamAdiToplami,
         :PartiKilo, :PartiMetre, :FBoyaSiparisYil, :FBoyaSiparisId, :ToplamKilo, :ToplamMetre do
  begin
    BoyaSiparisIdStr = '';

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (FBoyaSiparisId is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') '  || EgemenIntegerToStr(FBoyaSiparisId);

    suspend;
  end
end
