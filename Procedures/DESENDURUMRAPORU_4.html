CREATE PROCEDURE DESENDURUMRAPORU_4(
    ADESENID VARCHAR(20),
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    DESENID VARCHAR(20),
    DESENISTEKLERIID INTEGER,
    DESENISTEKTIPIID VARCHAR(5),
    DESENISTEKSARFIYATID INTEGER,
    DEPOID VARCHAR(10),
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    MIKTAR DOUBLE PRECISION)
AS
begin
  For select DesenDurumRaporu_2.DesenId, DesenIstekSarfiyat.DesenIstekleriId, DesenIStekSarfiyat.DesenIstekSarfiyatID, DesenIStekSarfiyat.DepoID, DesenIstekSarfiyat.UrunID, Urun.Adi UrunAdi, DesenIStekSarfiyat.Miktar from DesenDurumRaporu_2(:ADesenId, :ARaporTipi, :ABaslangicTarihi, :ABitisTarihi, :AFirmaId)

    left join Urun on Urun.UrunID = DesenIStekSarfiyat.UrunID
    into :DesenID, :DesenIstekleriId, :DesenIstekSarfiyatID, :DepoID, :UrunID, :UrunAdi, Miktar do
  begin
    suspend;
  end
end
