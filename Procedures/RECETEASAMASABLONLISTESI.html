CREATE PROCEDURE RECETEASAMASABLONLISTESI(
    ARECETEASAMAID INTEGER)
RETURNS (
    RECETEASAMASABLONID INTEGER,
    ADI VARCHAR(40),
    RECETEGRAFIKID VARCHAR(10),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
begin
  For Select ReceteAsamaSablon.ReceteAsamaSablonId, ReceteAsamaSablon.Adi, ReceteAsamaSablon.ReceteGrafikId, ReceteAsamaSablon.Notu from ReceteAsamaSablon
    where ReceteAsamaSablon.ReceteAsamaId = :AReceteAsamaId and ReceteAsamaSablon.Kapali = 0
    into :ReceteAsamaSablonId, :Adi, :ReceteGrafikId, :Notu do suspend;
end
