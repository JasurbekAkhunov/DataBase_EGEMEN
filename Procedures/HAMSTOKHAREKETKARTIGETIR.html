CREATE PROCEDURE HAMSTOKHAREKETKARTIGETIR(
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    HAREKETTIPISTR VARCHAR(50),
    TARIH TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYENO IRSALIYENOSIZE,
    FISNOSTR VARCHAR(30),
    PARTIIDSTR VARCHAR(30),
    MIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    FIRE NUMERIC(12,3),
    ACIKLAMA1 VARCHAR(75),
    ACIKLAMA2 VARCHAR(75),
    GIRISCIKIS INTEGER,
    STOKMEVCUDU NUMERIC(12,3))
AS
begin
  StokMevcudu = 0.0;

  For Select HareketTipiStr, Tarih, FirmaId, FirmaAdi, IrsaliyeNo, FisNoStr, PartiIdStr, Miktar, NetMiktar, Aciklama1, Aciklama2, GirisCikis from HamStokHareketKarti(:AFirmaId, :AHamId, :ABaslangicTarihi, :ABitisTarihi) order by tarih, giriscikis desc
   into :HareketTipiStr, :Tarih, :FirmaId, :FirmaAdi, :IrsaliyeNo, :FisNoStr, :PartiIdStr, :Miktar, :NetMiktar, :Aciklama1, :Aciklama2, :GirisCikis do
  begin
    if (GirisCikis = 1) then StokMevcudu = StokMevcudu + Miktar;
    else StokMevcudu = StokMevcudu - Miktar;

    if (NetMiktar is not null and NetMiktar <> 0) then Fire = Miktar - NetMiktar;
    else Fire = Null;
    suspend;
  end
end
