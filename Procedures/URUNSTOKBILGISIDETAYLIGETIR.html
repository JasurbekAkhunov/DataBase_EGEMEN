CREATE PROCEDURE URUNSTOKBILGISIDETAYLIGETIR(
    AURUNID VARCHAR(20),
    ADEPOID VARCHAR(10))
RETURNS (
    DEPOID VARCHAR(10),
    DEPOADI VARCHAR(30),
    DEPODEVIR DOUBLE PRECISION,
    DEPOALIS DOUBLE PRECISION,
    DEPOURETIM DOUBLE PRECISION,
    DEPOSATIS DOUBLE PRECISION,
    DEPODEPOLARDANGELEN DOUBLE PRECISION,
    DEPODEPOLARACIKIS DOUBLE PRECISION,
    DEPOURETIMECIKIS DOUBLE PRECISION,
    DEPOSTOKMEVCUDU DOUBLE PRECISION,
    GENELDEVIR DOUBLE PRECISION,
    GENELALISTOPLAMI DOUBLE PRECISION,
    GENELURETIMTOPLAMI DOUBLE PRECISION,
    GENELSATISTOPLAMI DOUBLE PRECISION,
    GENELURETIMECIKISTOPLAMI DOUBLE PRECISION,
    TOPLAMSTOKMEVCUDU DOUBLE PRECISION,
    SONALISTARIHI TIMESTAMP,
    SONALISFISNO INTEGER,
    SONALISIRSALIYENO IRSALIYENOSIZE,
    SONALISFATURANO FATURANOSIZE,
    SONALISFIRMAID VARCHAR(20),
    SONALISFIRMAADI VARCHAR(100),
    SONALISFIYATI DOUBLE PRECISION,
    SONALISDOVIZID VARCHAR(5),
    SONALISURUNBIRIMID INTEGER,
    SONALISBIRIMID VARCHAR(2),
    SONALISBIRIMCARPAN DOUBLE PRECISION,
    DEFAULTBIRIMID VARCHAR(2))
AS
  declare variable MaxStokDetayId integer;
  declare variable DefaultBirimCarpan double precision;
  declare variable SDovizId varchar(5);
  declare variable IDovizId varchar(5);
  declare variable FDovizId varchar(5);
  declare variable FAltIskontoCarpani double precision;
begin
  DepoId = ADepoId;
  select BirimId, Carpan from UrunBirim where UrunBirim.UrunId = :AUrunId and DefaultBirim = 1 into :DefaultBirimId, :DefaultBirimCarpan;

  select Depo.Adi from Depo where DepoId = :ADepoId into DepoAdi;

  if (ADepoId is not null) then
  begin
    select Devir*:DefaultBirimCarpan, (Alis-AlisIadesi)*:DefaultBirimCarpan, (Satis-SatisIadesi)*:DefaultBirimCarpan,
                  DepoyaCikan*:DefaultBirimCarpan, DepodanGelen * :DefaultBirimCarpan,
                  UretimeCikis*:DefaultBirimCarpan + FasonUretimeCikis*:DefaultBirimCarpan - FasonUretimIade*:DefaultBirimCarpan ,
                  Uretim*:DefaultBirimCarpan + FasonUretim*:DefaultBirimCarpan from DepoStok where DepoStok.DepoId = :ADepoId and DepoStok.UrunId = :AUrunId
      into :DepoDevir, :DepoAlis, :DepoSatis, :DepoDepolaraCikis, :DepoDepolardanGelen, :DepoUretimeCikis, :DepoUretim;

    DepoStokMevcudu = DepoDevir + DepoAlis + DepoUretim - DepoSatis + DepoDepolardanGelen - DepoDepolaraCikis - DepoUretimeCikis;

    if (DepoDevir = 0) then DepoDevir = null;
    if (DepoAlis = 0) then DepoAlis = null;
    if (DepoUretim = 0) then DepoUretim = null;
    if (Deposatis = 0) then DepoSatis = null;
    if (DepoDepolardanGelen = 0) then DepoDepolardanGelen = null;
    if (DepoDepolaraCikis = 0) then DepoDepolaraCikis = null;
    if (DepoUretimeCikis = 0) then DepoUretimeCikis = null;
  end

  select sum(Devir)*:DefaultBirimCarpan, sum(Alis-AlisIadesi)*:DefaultBirimCarpan, sum(Satis-SatisIadesi)*:DefaultBirimCarpan, sum(UretimeCikis)*:DefaultBirimCarpan + sum(FasonUretimeCikis)*:DefaultBirimCarpan - sum(FasonUretimIade)*:DefaultBirimCarpan, sum(Uretim)*:DefaultBirimCarpan + sum(FasonUretim)*:DefaultBirimCarpan from DepoStok where DepoStok.UrunId = :AUrunId
    into :GenelDevir, :GenelAlisToplami, :GenelSatisToplami, :GenelUretimeCikisToplami, :GenelUretimToplami;

  ToplamStokMevcudu =   GenelDevir + GenelAlisToplami + GenelUretimToplami - GenelSatisToplami - GenelUretimeCikisToplami;

  if (ToplamStokMevcudu = 0) then ToplamStokMevcudu = null;
  if (GenelDevir = 0) then GenelDevir = null;
  if (GenelAlisToplami = 0) then GenelAlisToplami = null;
  if (GenelUretimToplami = 0) then GenelUretimToplami = null;
  if (GenelSatisToplami = 0) then GenelSatisToplami = null;
  if (GenelUretimeCikisToplami = 0) then GenelUretimeCikisToplami = null;

  select SonAlisTarihi, SonAlisFisNo, SonAlisIrsaliyeNo, SonAlisFaturaNo, SonAlisFirmaId, SonAlisFirmaAdi, SonAlisFiyati,
         SonAlisDovizId, SonAlisUrunBirimId, SonAlisBirimId, SonAlisBirimCarpan from UrunSonAlisBilgisiGetir(:AUrunId)
         into :SonAlisTarihi, :SonAlisFisNo, :SonAlisIrsaliyeNo, :SonAlisFaturaNo, :SonAlisFirmaId, :SonAlisFirmaAdi,
              :SonAlisFiyati, :SonAlisDovizId, :SonAlisUrunBirimId, :SonAlisBirimId, :SonAlisBirimCarpan;
  suspend;
end
