CREATE PROCEDURE OSRAPORU_GLOBALIPLIKMIKTARLARI(
    BOYASIPARISKAYITID INTEGER)
RETURNS (
    GLOBALSATINALINANIPLIK DOUBLE PRECISION,
    GLOBALIADEEDILENIPLIK DOUBLE PRECISION)
AS
declare variable hstIplik               integer;
declare variable ShtAlis                integer;
declare variable shtAlisIade                integer;
begin
  Select hstIplik from HamTipiGetir into :hstIplik;

  Select ShtAlis, shtAlisIade from OrmeStokHareketTipiGetir into :ShtAlis, :shtAlisIade;

  Select Sum(iif(StokHareket.HareketTipi = :shtAlis, StokHareket.Miktar, 0)),
         Sum(iif(StokHareket.HareketTipi = :shtAlisIade, StokHareket.Miktar, 0)) from BoyaSiparis
    left join StokHareket on StokHareket.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
    left join Ham On Ham.HamID = StokHareket.HamId
    where BoyaSiparis.BoyaSiparisKayitId = :BoyaSiparisKayitId and
          StokHareket.StokHareketID is not null and
          StokHareket.OrmeSiparisDetayID is null and Ham.HamTipiID = :hstIplik
    into :GlobalSatinAlinanIplik, :GlobalIadeEdilenIplik;

    suspend;
end
