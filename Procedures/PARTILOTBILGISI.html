CREATE PROCEDURE PARTILOTBILGISI(
    APARTIKAYITID INTEGER)
RETURNS (
    LOT VARCHAR(150),
    ORMECIMAKINANO VARCHAR(100),
    MUSTERIPARTINO VARCHAR(100),
    BUYUKSINIFHAMADITOPLAMI VARCHAR(255),
    BUYUKSINIFSIPARISNOTU VARCHAR(1024),
    BUYUKSINIFSIPARISPLANLAMANOTU VARCHAR(1024),
    BUYUKSINIFMUSTERIRENKKODUVEADI VARCHAR(30),
    HAMSTOKMASTERID VARCHAR(255),
    REFERANSSIPARISNO VARCHAR(255),
    SIPARISPROSESADI VARCHAR(1024),
    SIPARISENISTENEN VARCHAR(255),
    SIPARISGRAMAJISTENEN VARCHAR(255),
    SIPARISILAVEISLEMLER VARCHAR(1024))
AS
declare variable FLot varchar(150);
declare variable BuyukPartiSinifId             integer;
declare variable MaxKilo                    Numeric(12,3);
declare variable BlobSize    integer;
declare variable FSiparisNotu Blob;
declare variable FPlanlamaNotu Blob;
declare variable FOrmeciMakinaNo varchar(10);
declare variable FMusteriPartiNo varchar(30);
declare variable FHamStokMasterId varchar(30);
declare variable FReferansSiparisNo varchar(30);
declare variable FSiparisProsesAdi      varchar(255);
declare variable FSiparisEnIstenen      varchar(10);
declare variable FSiparisGramajIstenen  varchar(10);
declare variable EgemenFirmaId integer;
declare variable FSiparisIlaveIslemler  varchar(255);
begin
  select EgemenFirmaID from Sistem where SistemId = 1 into :EgemenFirmaId ;

  select  coalesce(left(list(HamStokMaster.Lot), 150), '') as Lot,
          coalesce(left(list(HamStokMaster.MusteriPartiNo), 100), '') as MusteriPartiNo,
          coalesce(left(list(HamStokMaster.OrmeciMakinaNo), 100), '') as OrmeciMakinaNo from PartiStokDetay
  left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
  left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  where PartiStokDetay.PartiKayitId = :APartiKayitId and
    ( HamStokMaster.Lot is not null or HamStokMaster.MusteriPartiNo is not null or HamStokMaster.OrmeciMakinaNo is not null )
  into :Lot, :MusteriPartiNo, :OrmeciMakinaNo;

  Select Max(PartiStok.Kilo) from PartiStok where PartiStok.PartiKayitId = :APartiKayitId into :MaxKilo;

  Select Max(PartiStok.PartiSinifId) from PartiStok
    where PartiStok.PartiKayitId = :APartiKayitId and PartiStok.Kilo = :MaxKilo into :BuyukPartiSinifId;

  Select PartiStok.HamAdiToplami, BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu, BoyaSiparisDetay.MusteriRenkKoduveAdi from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiSinifId = :BuyukPartiSinifId
    into :BuyukSinifHamAdiToplami, :FSiparisNotu, :FPlanlamaNotu, :BuyukSinifMusteriRenkKoduveAdi;

  BlobSize = char_length(FSiparisNotu);
  if (BlobSize > 1023) then BuyukSinifSiparisNotu = Left(FSiparisNotu, 1023);
  else if (BlobSize > 0) then BuyukSinifSiparisNotu = Left(FSiparisNotu, BlobSize);
  else BuyukSinifSiparisNotu = '';

  BlobSize = char_length(FPlanlamaNotu);
  if (BlobSize > 1023) then BuyukSinifSiparisPlanlamaNotu = Left(FPlanlamaNotu, 1023);
  else if (BlobSize > 0) then BuyukSinifSiparisPlanlamaNotu = Left(FPlanlamaNotu, BlobSize);
  else BuyukSinifSiparisPlanlamaNotu = '';


  SiparisProsesAdi = '';
  SiparisEnIstenen = '';
  SiparisGramajIstenen = '';
  SiparisIlaveIslemler = '';

  for Select SiparisProses.Adi, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisdetay.IlaveIslemler from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    where PartiStok.PartiSinifId = :BuyukPartiSinifId
    Group By SiparisProses.Adi, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisdetay.IlaveIslemler
    into :FSiparisProsesAdi, :FSiparisEnIstenen, :FSiparisGramajIstenen, :FSiparisIlaveIslemler do
  begin
    if (FSiparisProsesAdi is not Null) then
    if (Char_Length(SiparisProsesAdi) + Char_Length(FSiparisProsesAdi) < 1020) then
    begin
      if (SiparisProsesAdi <> '' ) then SiparisProsesAdi = SiparisProsesAdi || cast(', ' as varchar(2) character set UTF8);
      SiparisProsesAdi = SiparisProsesAdi || FSiparisProsesAdi;
    end

    if (FSiparisIlaveIslemler is not Null) then
    if (Char_Length(SiparisIlaveIslemler) + Char_Length(FSiparisIlaveIslemler) < 1020) then
    begin
      if (FSiparisIlaveIslemler <> '' ) then FSiparisIlaveIslemler = FSiparisIlaveIslemler || cast(', ' as varchar(2) character set UTF8);
      SiparisIlaveIslemler = SiparisIlaveIslemler || FSiparisIlaveIslemler;
    end

    if (FSiparisEnIstenen is not Null) then
    if (Char_Length(SiparisEnIstenen) + Char_Length(FSiparisEnIstenen) < 250) then
    begin
      if (SiparisEnIstenen <> '' ) then SiparisEnIstenen = SiparisEnIstenen || cast(', ' as varchar(2) character set UTF8);
      SiparisEnIstenen = SiparisEnIstenen || FSiparisEnIstenen;
    end

    if (FSiparisGramajIstenen is not Null) then
    if (Char_Length(SiparisGramajIstenen) + Char_Length(FSiparisGramajIstenen) < 250) then
    begin
      if (SiparisGramajIstenen <> '' ) then SiparisGramajIstenen = SiparisGramajIstenen || cast(', ' as varchar(2) character set UTF8);
      SiparisGramajIstenen = SiparisGramajIstenen || FSiparisGramajIstenen;
    end
  end

  HamStokMasterId = '';


  for select HamStokMaster.HamStokMasterId from PartiStokDetay
  left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
  left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  where PartiStokDetay.PartiSinifId = :BuyukPartiSinifId and ((:EgemenFirmaId = 120) or (HamStokMaster.Lot is not null))
  group by HamStokMaster.HamStokMasterId
  into :FHamStokMasterId do
  begin
    if (Char_Length(HamStokMasterId) + Char_Length(HamStokMasterId) < 225) then
    begin
      if (HamStokMasterId <> '' ) then HamStokMasterId = HamStokMasterId || cast(', ' as varchar(2) character set UTF8);
      HamStokMasterId = HamStokMasterId || FHamStokMasterId;
    end
  end

  ReferansSiparisNo = '';

  for select HamStokMaster.ReferansSiparisNo from PartiStokDetay
  left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
  left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  where PartiStokDetay.PartiKayitId = :APartiKayitId and HamStokMaster.Lot is not null
  group by HamStokMaster.ReferansSiparisNo
  into :FReferansSiparisNo do
  begin
    if (Char_Length(ReferansSiparisNo) + Char_Length(ReferansSiparisNo) < 225) then
    begin
      if (ReferansSiparisNo <> '' ) then ReferansSiparisNo = ReferansSiparisNo || cast(', ' as varchar(2) character set UTF8);
      ReferansSiparisNo = ReferansSiparisNo || FReferansSiparisNo;
    end
  end

  suspend;
end
