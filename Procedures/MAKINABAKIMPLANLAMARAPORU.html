CREATE PROCEDURE MAKINABAKIMPLANLAMARAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    MAKINABAKIMISEMRIID INTEGER,
    ADI VARCHAR(255),
    PLANLANANBASLANGIC TIMESTAMP,
    PLANLANANBITIS TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    MAKINANO INTEGER,
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    IPLIKMAKINAID VARCHAR(10),
    IPLIKMAKINAADI VARCHAR(40),
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    ISEMRITAMAMLANDI EGEBOOLEAN)
AS
declare variable FMakinaNo integer;
declare variable FOrmeMakinaNo integer;
declare variable FIplikMakinaNo integer;
declare variable FDokumaMakinaNo integer;
declare variable FIEEndTarihi TimeStamp;
begin
  for select MakinaBakimIsEmri.MakinaBakimIsEmriId, MakinaBakimIsEmri.PlanlananBaslangic, MakinaBakimIsEmri.PlanlananBaslangic + (MakinaBakimIsEmri.TahminiSaat / 24),
             MakinaBakimIsEmri.Notu, MakinaBakimIsEmri.Adi, MakinaBakimIsEmri.MakinaId, MakinaBakimIsEmri.OrmeMakinaId, MakinaBakimIsEmri.IplikMakinaId, MakinaBakimIsEmri.DokumaMakinaId,
             Makina.Adi, OrmeMakina.Adi, IplikMakina.Adi, DokumaMakina.Adi, Makina.MakinaNo, OrmeMakina.MakinaNo, IplikMakina.MakinaNo, DokumaMakina.MakinaNo,
             MakinaBakimIsEmri.IEEndTarihi FROM MakinaBakimIsEmri
  left join Makina on Makina.MakinaId = MakinaBakimIsEmri.MakinaId
  left join OrmeMakina on OrmeMakina.OrmeMakinaId = MakinaBakimIsEmri.OrmeMakinaId
  left join IplikMakina on IplikMakina.IplikMakinaId = MakinaBakimIsEmri.IplikMakinaId
  left join DokumaMakina on DokumaMakina.DokumaMakinaId = MakinaBakimIsEmri.DokumaMakinaId
  where ((MakinaBakimIsEmri.PlanlananBaslangic >= :ABaslangicTarihi
           or (MakinaBakimIsEmri.PlanlananBaslangic + (MakinaBakimIsEmri.TahminiSaat / 24) <= :ABaslangicTarihi)
           or (MakinaBakimIsEmri.PlanlananBaslangic + (MakinaBakimIsEmri.TahminiSaat / 24) >= :ABaslangicTarihi and MakinaBakimIsEmri.PlanlananBaslangic <= :ABaslangicTarihi)
         )
         or :ABaslangicTarihi is Null
        ) and
        ((MakinaBakimIsEmri.PlanlananBaslangic + (MakinaBakimIsEmri.TahminiSaat / 24) <= :ABitisTarihi
           or (MakinaBakimIsEmri.PlanlananBaslangic <= :ABitisTarihi)
           or (MakinaBakimIsEmri.PlanlananBaslangic >= :ABitisTarihi and MakinaBakimIsEmri.PlanlananBaslangic <= :ABitisTarihi)
         )
        or :ABitisTarihi is Null)
  into :MakinaBakimIsEmriId, :PlanlananBaslangic, :PlanlananBitis, :Notu, :Adi, :MakinaId, :OrmeMakinaId, :IplikMakinaId, :DokumaMakinaId,
       :MakinaAdi, :OrmeMakinaAdi, :IplikMakinaAdi, :DokumaMakinaAdi, :FMakinaNo, :FOrmeMakinaNo, :FIplikMakinaNo, :FDokumaMakinaNo, :FIEEndTarihi  do
  begin
    Adi = 'Is Emri No: ' || EgemenIntegertoStr(:MakinaBakimIsEmriId) || '  ' || Adi;

    if (MakinaId is not Null) then MakinaNo = FMakinaNo;
    if (OrmeMakinaId is not Null) then MakinaNo = FOrmeMakinaNo;
    if (IplikMakinaId is not Null) then MakinaNo = FIplikMakinaNo;
    if (DokumaMakinaId is not Null) then MakinaNo = FDokumaMakinaNo;

    IsEmriTamamlandi = 0;
    if (FIEEndTarihi is not null) then
      IsEmriTamamlandi = 1;

    suspend;
  end
end
