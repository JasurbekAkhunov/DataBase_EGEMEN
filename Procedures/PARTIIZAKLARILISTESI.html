CREATE PROCEDURE PARTIIZAKLARILISTESI
RETURNS (
    KULLANICIID INTEGER,
    KULLANICIADI ADI40,
    PARTIKAYITID INTEGER,
    PARTINOSTR VARCHAR(40),
    PARTIIKAZTIPI INTEGER,
    PARTIIKAZTIPISTR VARCHAR(20),
    PARTIBASLAMATARIHI TIMESTAMP,
    PARTIBITISTARIHI TIMESTAMP,
    IKAZSURESIGUN INTEGER,
    IKAZSURESISAAT INTEGER,
    IKAZSURESIDAKIKA INTEGER,
    IKAZGONDERILDISTR VARCHAR(5),
    IKAZYERI INTEGER,
    IKAZYERISTR VARCHAR(20),
    IKAZSURESI DOUBLE PRECISION)
AS
declare variable pitBitmezIse      integer;
declare variable pitBitince        integer;
declare variable iyBilgisayar      integer;
declare variable iySirketIciMail   integer;
declare variable iyEmail           integer;
declare variable iySms             integer;
declare variable iyFax             integer;
begin
  select pitBitmezIse, pitBitince from PartiIkazTipiGetir into :pitBitmezIse, :pitBitince;
  select iyBilgisayar, iySirketIciMail, iyEmail, iySms, iyFax from PartiIkazYeriGetir into :iyBilgisayar, :iySirketIciMail, :iyEmail, :iySms, :iyFax;

  for select PartiIkazlari.KullaniciId, Kullanici.Adi, PartiIkazlari.PartiKayitId, '('||Parti.Partiyil||')'||' '||Parti.PartiId, PartiIkazlari.PartiIkazTipi,
             PartiIkazlari.PartiBaslamaTarihi, PartiIkazlari.PartiBitisTarihi, PartiIkazlari.IkazSuresi, Decode(PartiIkazlari.IkazGosterildi, 0, 'Hayır', 'Evet'), PartiIkazlari.IkazYeri from PartiIkazlari
      left join Kullanici on Kullanici.KullaniciId = PartiIkazlari.KullaniciId
      left join Parti on Parti.PartiKayitId = PartiIkazlari.PartiKayitId
      into :KullaniciId, :KullaniciAdi, :PartiKayitId, :PartiNoStr, :PartiIkazTipi,
           :PartiBaslamaTarihi, :PartiBitisTarihi, :IkazSuresi, :IkazGonderildiStr, :IkazYeri do
  begin
    IkazSuresiGun = Null;
    IkazSuresiSaat = Null;
    IkazSuresiDakika = Null;

    if (PartiIkazTipi = pitBitmezIse) then
      PartiIkazTipiStr = 'Bitmezse';
    else if (PartiIkazTipi = pitBitince) then
      PartiIkazTipiStr = 'Bitince';

    if (IkazYeri = iyBilgisayar) then
      IkazYeriStr = 'Bilgisayar';
    else if (IkazYeri = iySirketIciMail) then
      IkazYeriStr = 'Şirket İçi Mail';
    else if (IkazYeri = iyEmail) then
      IkazYeriStr = 'E-mail';
    else if (IkazYeri = iySms) then
      IkazYeriStr = 'Sms';
    else if (IkazYeri = iyFax) then
      IkazYeriStr = 'Fax';

    /*if (IkazSuresi > 0 ) then
    begin
      IkazSuresiGun = Trunc(IkazSuresi);
      IkazSuresiSaat = Extract(hour from cast('01.01.01 ' || dateadd(hour, IkazSuresi - IkazSuresiGun, cast('00:00' as time)) as TimeStamp));
      IkazSuresiDakika = Extract(minute from cast('01.01.01 ' || dateadd(hour, IkazSuresi - IkazSuresiGun, cast('00:00' as time)) as TimeStamp));
    end*/

    suspend;
  end
end
