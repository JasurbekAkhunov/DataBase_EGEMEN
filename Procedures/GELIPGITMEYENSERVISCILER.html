CREATE PROCEDURE GELIPGITMEYENSERVISCILER(
    AMAKINAID VARCHAR(10))
RETURNS (
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    TARIH TIMESTAMP)
AS
begin
  For Select ArizaBakim.PersonelId, Personel.Adi, MakinaDurumu.Tarih from ArizaBakim
    left join Personel on Personel.PersonelId = ArizaBakim.PersonelId
    left join MakinaDurumu on MakinaDurumu.MakinaDurumuId = ArizaBakim.GeldiMakinaDurumuId
    where ArizaBakim.MakinaId = :AMakinaId and ArizaBakim.GittiMakinaDurumuId is null
    into :PersonelId, :PersonelAdi, :Tarih do suspend;
end
