CREATE PROCEDURE BOYAIRSALIYERAPORU_1(
    AIRSALIYETIPI INTEGER,
    AFIRMAID VARCHAR(20),
    ASIPARISFIRMAID VARCHAR(20),
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1),
    ASATISTIPI INTEGER,
    ARAPORTIPI INTEGER,
    AISLEMTIPI INTEGER,
    ARAPORTARIHTIPI INTEGER,
    AUNITEID INTEGER,
    APARTIGRUPID VARCHAR(4),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AMAMULSEVK INTEGER,
    AHAMSATIS INTEGER,
    AHAMIADE INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    ASEVKSEFERNO INTEGER,
    AIRSALIYENO INTEGER,
    AFATURADURUMU INTEGER,
    AFATURAKESILECEK INTEGER)
RETURNS (
    RAPORID VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    ALICIUNITEID INTEGER,
    ALICIUNITEADI VARCHAR(30),
    SEVKFIRMAID VARCHAR(20),
    SEVKFIRMAADI VARCHAR(50),
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    SEVKTARIHI TIMESTAMP,
    IRSALIYEGRUPID VARCHAR(4),
    IRSALIYEGRUPADI VARCHAR(30),
    ARACID VARCHAR(11),
    SOFORADI VARCHAR(40),
    SINIF VARCHAR(1),
    SEVKSEFERNO INTEGER,
    KANTARKAYITID INTEGER,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    SIRALAMATARIHI TIMESTAMP,
    BOYAIRSALIYEID INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TRANSFEREDILEBILIR EGEBOOLEAN,
    TRANSFEREDILEBILIRSTR VARCHAR(5),
    FATURANOSTR VARCHAR(500),
    FATURAKESILECEKSTR VARCHAR(20))
AS
declare variable MamulDepoTransferFisiId integer;
declare variable FaturaNo FaturaNoSize;
begin
  for select BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeSeriNo,
             BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.UniteId, Unite.Adi, BoyaIrsaliye.SevkFirmaId, SevkFirma.Adi,
             BoyaIrsaliye.IrsaliyeGrupId, StokHareketGrup.Adi, BoyaIrsaliye.AracId, Sofor.Adi, BoyaIrsaliye.SevkSeferNo,
             BoyaIrsaliye.Sinif, BoyaIrsaliye.KantarKayitId, Kullanici.Adi, BoyaIrsaliye.Degisimtarihi, BoyaIrsaliyeRaporu_2.SiralamaTarihi,
             BoyaIrsaliye.Notu, BoyaIrsaliye.TransferEdilebilir, BoyaIrsaliyeRaporu_2.FaturaKesilecekStr
             from BoyaIrsaliyeRaporu_2(:AIrsaliyeTipi, :AFirmaId, :ASiparisFirmaId, :ABoyaSiparisYil, :ABoyaSiparisId, :ABaslangicTarihi, :ABitisTarihi, :ASinif, :ASatisTipi, :ARaporTipi, :AIslemTipi, :ARaporTarihTipi, :AUniteId, :APartiGrupId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMamulSevk, :AHamSatis, :AHamIade, :ASiparisiAlanKullaniciId, :ASevkSeferNo, :AIrsaliyeNo, :AFaturaDurumu, :AFaturaKesilecek)
    left join BoyaIrsaliye On BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeRaporu_2.BoyaIrsaliyeId
    left join Firma On Firma.FirmaId = BoyaIrsaliye.FirmaId
    left join Kullanici On Kullanici.KullaniciId = BoyaIrsaliye.KullaniciId
    left join Unite On Unite.UniteId = BoyaIrsaliye.UniteId
    left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = BoyaIrsaliye.IrsaliyeGrupId
    left join SevkFirma On SevkFirma.SevkFirmaId = BoyaIrsaliye.SevkFirmaId
    left join Sofor On Sofor.SoforId = BoyaIrsaliye.SoforId
    where BoyaIrsaliyeRaporu_2.BoyaIrsaliyeId is not null
    group By BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.FirmaId, Firma.Adi, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeSeriNo,
             BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.UniteId, Unite.Adi, BoyaIrsaliye.SevkFirmaId, SevkFirma.Adi,
             BoyaIrsaliye.IrsaliyeGrupId, StokHareketGrup.Adi, BoyaIrsaliye.AracId, Sofor.Adi, BoyaIrsaliye.SevkSeferNo,
             BoyaIrsaliye.Sinif, BoyaIrsaliye.KantarKayitId,
             Kullanici.Adi, BoyaIrsaliye.Degisimtarihi, BoyaIrsaliyeRaporu_2.SiralamaTarihi, BoyaIrsaliye.Notu, BoyaIrsaliye.TransferEdilebilir, BoyaIrsaliyeRaporu_2.FaturaKesilecekStr
    order by BoyaIrsaliyeRaporu_2.SiralamaTarihi, BoyaIrsaliye.IrsaliyeNo
    into :BoyaIrsaliyeId, :FirmaId, :FirmaAdi, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :SevkTarihi, :UniteId, :UniteAdi,
         :SevkFirmaId, :SevkFirmaAdi,
         :IrsaliyeGrupId, :IrsaliyeGrupAdi, :AracId, :SoforAdi, :SevkSeferNo, :Sinif, :KantarKayitId, :KullaniciAdi, :Degisimtarihi, :SiralamaTarihi, :Notu, :TransferEdilebilir, :FaturaKesilecekStr do
  begin
    RaporId = 'BI' || EgemenIntegertoStr(BoyaIrsaliyeId);

    TransferEdilebilirStr = '';
    if (TransferEdilebilir = 1) then
      TransferEdilebilirStr = 'Evet';

    FaturaNoStr = '';
    for select BoyaFatura.FaturaNo from BoyaIrsaliyeDetay
        join BoyaFatura on BoyaFatura.BoyaFaturaId = BoyaIrsaliyeDetay.BoyaFaturaId
        where BoyaIrsaliyeDetay.BoyaIrsaliyeId = :BoyaIrsaliyeId into :FaturaNo do
    begin
      if (FaturaNoStr = '') then FaturaNoStr = FaturaNo;
      else
      begin
        if (Position(',', FaturaNoStr) = 0) then
        begin
          if (FaturaNoStr <> FaturaNo) then
            FaturaNoStr = FaturaNoStr ||cast(', ' as varchar(2) character set UTF8)||FaturaNo;
        end
        else
          if (Position(FaturaNo||',', FaturaNoStr) = 0 and Position(', '||FaturaNo, FaturaNoStr) = 0) then
            FaturaNoStr = FaturaNoStr ||cast(', ' as varchar(2) character set UTF8)||FaturaNo;
      end
    end

    suspend;
  end

  TransferEdilebilir = Null;
  TransferEdilebilirStr = Null;
  FirmaID = Null;
  FirmaAdi = NUll;
  SevkFirmaID = Null;
  SevkFirmaAdi = NUll;
  BoyaIrsaliyeId = NUll;
  Sinif = Null;
  IrsaliyeGrupId = Null;
  UniteId     = Null;
  UniteAdi    = Null;
  FaturaNoStr = Null;
  FaturaKesilecekStr = null;

  if (AMamulSevk = 1) then
  for select MamulDepoTransferFisi.MamulDepoTransferFisiId, GD.UniteId, GU.Adi, AD.UniteId, AU.Adi, MamulDepoTransferFisi.IrsaliyeTarihi, MamulDepoTransferFisi.IrsaliyeSeriNo,
             MamulDepoTransferFisi.IrsaliyeNo, MamulDepoTransferFisi.SevkTarihi, MamulDepoTransferFisi.AracId, Sofor.Adi, MamulDepoTransferFisi.SevkSeferNo,
             MamulDepoTransferFisi.KantarKayitId, Kullanici.Adi, MamulDepoTransferFisi.Degisimtarihi, BoyaIrsaliyeRaporu_2.SiralamaTarihi, MamulDepoTransferFisi.Notu from BoyaIrsaliyeRaporu_2(:AIrsaliyeTipi, :AFirmaId, :ASiparisFirmaId, :ABoyaSiparisYil, :ABoyaSiparisId, :ABaslangicTarihi, :ABitisTarihi, :ASinif, :ASatisTipi, :ARaporTipi, :AIslemTipi, :ARaporTarihTipi, :AUniteId, :APartiGrupId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMamulSevk, :AHamSatis, :AHamIade, :ASiparisiAlanKullaniciId, :ASevkSeferNo, :AIrsaliyeNo, :AFaturaDurumu, :AFaturaKesilecek)
    left join MamulDepoTransferFisi On MamulDepoTransferFisi.MamulDepoTransferFisiId = BoyaIrsaliyeRaporu_2.MamulDepoTransferFisiId
    left join Depo GD on GD.DepoId = MamulDepoTransferFisi.DepoId
    left join Depo AD on AD.DepoId = MamulDepoTransferFisi.TransferDepoId
    left join Unite GU on GU.UniteId = GD.UniteId
    left join Unite AU on AU.UniteId = AD.UniteId
    left join Sofor On Sofor.SoforId = MamulDepoTransferFisi.SoforId
    left join Kullanici On Kullanici.KullaniciId = MamulDepoTransferFisi.KullaniciId
    where BoyaIrsaliyeRaporu_2.MamulDepoTransferFisiId is not null
    group By MamulDepoTransferFisi.MamulDepoTransferFisiId, GD.UniteId, GU.Adi, AD.UniteId, AU.Adi, MamulDepoTransferFisi.IrsaliyeTarihi, MamulDepoTransferFisi.IrsaliyeSeriNo,
             MamulDepoTransferFisi.IrsaliyeNo, MamulDepoTransferFisi.SevkTarihi, MamulDepoTransferFisi.AracId, Sofor.Adi, MamulDepoTransferFisi.SevkSeferNo,
             MamulDepoTransferFisi.KantarKayitId, Kullanici.Adi, MamulDepoTransferFisi.Degisimtarihi, BoyaIrsaliyeRaporu_2.SiralamaTarihi, MamulDepoTransferFisi.Notu
    order by BoyaIrsaliyeRaporu_2.SiralamaTarihi, MamulDepoTransferFisi.IrsaliyeNo
    into :MamulDepoTransferFisiId, :UniteId, :UniteAdi, :AliciUniteId, :AliciUniteAdi, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :SevkTarihi,
         :AracId, :SoforAdi, :SevkSeferNo, :KantarKayitId, :KullaniciAdi, :Degisimtarihi, :SiralamaTarihi, :Notu do
  begin
    RaporId = 'MDI' || EgemenIntegertoStr(MamulDepoTransferFisiId);
    suspend;
  end
end
