CREATE PROCEDURE KALITEASAMADEGERLERINIOLUSTUR(
    AKALITEID KALITEKODUSIZE,
    ASIPARISPROSESID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable FAsamaID integer;
declare variable FDegerTipiId integer;
declare variable FSira integer;
begin

    Delete From KaliteAsamaDegerTipi where KaliteAsamaDegerTipi.KaliteId = :AKaliteId;
    for select distinct  Asama.AsamaID, AsamaDegerTipi.DegerTipiId, ProsesAsamalari.Sira  from Siparisproses
        left join proses on Proses.Prosesid = Siparisproses.prosesid
        left join ProsesAsamalari ON ProsesAsamalari.ProsesId = Proses.ProsesId
        left join Asama ON Asama.AsamaID =  ProsesAsamalari.AsamaID
        left join AsamaDegerTipi ON AsamaDegerTipi.AsamaId =  Asama.AsamaID
        where Siparisproses.siparisprosesid = :ASIPARISPROSESID  and Asama.AsamaID is not null and  AsamaDegerTipi.DegerTipiId is not null
    into :FAsamaID, :FDegerTipiId, :FSira do
    begin
      insert into KaliteAsamaDegerTipi(KaliteId, AsamaId, DegerTipiId, Sira, ProsesAsamalariSira, INSERTKULLANICIID, KULLANICIID)
      values(:AKaliteId, :FAsamaID, :FDegerTipiId, -1, :FSira, :AKullaniciId, :AKullaniciId);
    end
end
