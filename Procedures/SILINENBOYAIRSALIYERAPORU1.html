CREATE PROCEDURE SILINENBOYAIRSALIYERAPORU1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    BOYAIRSALIYEID INTEGER,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    SEVKTARIHI TIMESTAMP,
    FIRMAID FIRMAKODUSIZE,
    IPTALNOTU MEMO,
    IPTALTARIHI TIMESTAMP,
    IPTALKULLANICIID INTEGER,
    FIRMAADI ADI100,
    IPTALKULLANICIADI ADI40)
AS
begin
  for select BoyaIrsaliyeIptal.BoyaIrsaliyeId, BoyaIrsaliyeIptal.IrsaliyeSeriNo, BoyaIrsaliyeIptal.IrsaliyeNo, BoyaIrsaliyeIptal.IrsaliyeTarihi, BoyaIrsaliyeIptal.SevkTarihi,
      BoyaIrsaliyeIptal.FirmaId, BoyaIrsaliyeIptal.IptalNotu, BoyaIrsaliyeIptal.IptalTarihi, BoyaIrsaliyeIptal.IptalKullaniciId,
      Firma.Adi, Kullanici.Adi from BoyaIrsaliyeIptal
      left join Firma on Firma.FirmaId = BoyaIrsaliyeIptal.FirmaId
      left join Kullanici on Kullanici.KullaniciId = BoyaIrsaliyeIptal.IptalKullaniciId
      where BoyaIrsaliyeIptal.IptalTarihi >= :ABaslangicTarihi and BoyaIrsaliyeIptal.IptalTarihi <= :ABitisTarihi
      union
      select iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.BoyaIrsaliyeId, BoyaIrsaliye.BoyaIrsaliyeId),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeSeriNo),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.IrsaliyeNo, BoyaIrsaliye.IrsaliyeNo),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeTarihi),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.SevkTarihi, BoyaIrsaliye.SevkTarihi),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.FirmaId, BoyaIrsaliye.FirmaId),
      BoyaIrsaliyeIptal.IptalNotu, BoyaIrsaliyeIptal.IptalTarihi,  BoyaIrsaliyeIptal.IptalKullaniciId,
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, IptalFirma.Adi, Firma.FirmaId),
      IptalKullanici.Adi
      from LogMaster
      left join LogDetay on LogDetay.LogMasterId = LogMaster.LogMasterId and LogDetay.FieldName = 'BOYAIRSALIYEID'
      left join BoyaIrsaliyeIptal on BoyaIrsaliyeIptal.BoyaIrsaliyeId = LogDetay.OldValue
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = LogDetay.OldValue
      left join Firma IptalFirma on IptalFirma.FirmaId = BoyaIrsaliyeIptal.FirmaId
      left join Kullanici IptalKullanici on IptalKullanici.KullaniciId = BoyaIrsaliyeIptal.IptalKullaniciId
      left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
      where LogMaster.Tip = 'D' and /*silinenler*/
      LogMaster.TabloAdi = 'BOYAIRSALIYEDETAY' and
      LogMaster.Tarih >= :ABaslangicTarihi and LogMaster.Tarih <= :ABitisTarihi+1
      union
      select iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.BoyaIrsaliyeId, BoyaIrsaliye.BoyaIrsaliyeId),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeSeriNo),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.IrsaliyeNo, BoyaIrsaliye.IrsaliyeNo),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeTarihi),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.SevkTarihi, BoyaIrsaliye.SevkTarihi),
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, BoyaIrsaliyeIptal.FirmaId, BoyaIrsaliye.FirmaId),
      BoyaIrsaliyeIptal.IptalNotu, BoyaIrsaliyeIptal.IptalTarihi,  BoyaIrsaliyeIptal.IptalKullaniciId,
      iif(BoyaIrsaliyeIptal.BoyaIrsaliyeID is not null, IptalFirma.Adi, Firma.FirmaId),
      IptalKullanici.Adi
      from LogMaster
      left join LogDetay on LogDetay.LogMasterId = LogMaster.LogMasterId and LogDetay.FieldName = 'BOYAIRSALIYEID'
      left join BoyaIrsaliyeIptal on BoyaIrsaliyeIptal.BoyaIrsaliyeId = LogDetay.OldValue
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = LogDetay.OldValue
      left join Firma IptalFirma on IptalFirma.FirmaId = BoyaIrsaliyeIptal.FirmaId
      left join Kullanici IptalKullanici on IptalKullanici.KullaniciId = BoyaIrsaliyeIptal.IptalKullaniciId
      left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
      where LogMaster.Tip = 'U' and /*silinen tartiler*/
      LogMaster.TabloAdi = 'PARTISTOKTARTI' and
      LogMaster.Tarih >= :ABaslangicTarihi and LogMaster.Tarih <= :ABitisTarihi+1 and
      LogDetay.OldValue is not null and LogDetay.NewValue is null
      into :BoyaIrsaliyeId, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :SevkTarihi, :FirmaId, :IptalNotu, :IptalTarihi, :IptalKullaniciId,
      :FirmaAdi, :IptalKullaniciAdi do
  begin
    suspend;
  end
end
