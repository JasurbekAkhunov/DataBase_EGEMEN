declare variable AsilRecetePartiAsamalariId integer;
Declare variable FReceteIsEmriYil integer;
Declare variable FReceteIsEmriId  integer;
declare variable FPartiIkazTipi integer;
declare variable FIkazSuresi double precision;
declare variable KayitSayisi integer;
declare variable FHamAdi varchar(255);
declare variable FMusteriRenkKoduveAdi varchar(30);
declare variable DefaultIplikPsKKEtiketTipiId  integer;
begin

  Select DefaultIplikPsKKEtiketTipiId from sistem where sistemID = 1 into :DefaultIplikPsKKEtiketTipiId;

  AnaPartiKayitId = Null;
  AnaReceteId = Null;
  for SELECT PartiYil, PartiId, PartiKayitId, PartiSinifId, ReceteId FROM BOYAIRSALIYELISTESI
    (:ABaslangicTarihi, :ABitisTarihi, :ASatisTipi, :ARaporTipi, :AHamId, :AFirmaId,
    :AFirmaYetkiliId, :ASevkFirmaId, :ASiparisFirmaId, :AReceteId, :AIrsaliyeNo, :AFiyatOnay,
    :AUniteId, :APartiGrupId, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AMusteriSiparisNo, :AReferansSiparisNo, :AIslemTipiId,
    :AIslemGrubuId, :ASiparisiAlanKullaniciId, :AKalinlikId, :AIplikCinsiId, :AOzellikId, :AKumasCinsiId,
    :ALycraTipiId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id,
    :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id, :AHamGrup1Aralikli, :AHamGrup2Aralikli,
    :AHamGrup3Aralikli, :AHamGrup4Aralikli, :AYurtDisi, :AMamulSevk, :AHamSatis, :AHamIade, :ABoyaSiparisYil,
    :ABoyaSiparisId, :ASinif, :ABaslangicPartiMiktar, :ABitisPartiMiktar,
    :AOrmeOlmayanlar, :AIplikSiparisi, :ABoyaliIplikSiparisi, :ALycraSiparisi, :AHamKumasSiparisi, :ABoyaliKumasSiparisi, :AIplikBoyaliKumasSiparisi,
    :AImalatTipiId, :AKaliteId, :AAracId, :ADesenId, :ADesenVaryantId, :AHareketTipiID, :ARaporAlanKullaniciID, :APartiKayitId, :ATemsilciId, :AReceteTuruId, :ADepoIdleri, :AMusteriKaliteNo, :ABaskiTipiId)
    into :AnaPartiYil, :AnaPartiId, :AnaPartiKayitId, :AnaPartiSinifId, :AnaReceteId do
  begin
    if (AnaPartiKayitId is not null and AnaReceteId is not null) then
    begin
      AsilRecetePartiAsamalariId = Null;
      Select PartiAsamalariId from PartiAsamalari where PartiAsamalari.PartiKayitId = :AnaPartiKayitId and PartiAsamalari.PartininAsilRecetesimi = 1 into :AsilRecetePartiAsamalariId;

      if (AsilRecetePartiAsamalariId is not null) then
      begin
        FReceteIsEmriId = Null;
        Select ReceteIsEmriYil, ReceteIsEmriId from ReceteIsEmriDetay where ReceteIsEmriDetay.PartiAsamalariId = :AsilRecetePartiAsamalariId into :FReceteIsEmriYil, :FReceteIsEmriId;
        if (FReceteIsEmriId is not null) then
        begin
          ReceteIsEmriIdStr = '(' || EgemenIntegerToStr(FReceteIsEmriYil) || ') ' || EgemenIntegerToStr(FReceteIsEmriId) || ' ';

          for select Parti.PartiYil, Parti.PartiId from ReceteIsEmriDetay
            left join Parti on Parti.PartiKayitId = ReceteIsEmriDetay.PartiKayitId
            where ReceteIsEmriDetay.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :FReceteIsEmriId and ReceteIsEmriDetay.PartiKayitId <> :AnaPartiKayitId
            group by Parti.PartiYil, Parti.PartiId
            into :PartiYil, :Partiid do
          begin
            select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.ImalatTipiId, Parti.AcilisTarihi, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Parti.ModelNo, Parti.IslemGrubuId, Parti.IslemTipiId,
                   Parti.AciliyetDerecesiId, Parti.PlanlananImalatTarihi, Parti.TerminTarihi, Parti.ReceteOkey, Parti.Talimat, Parti.TalimatOkey,
                   Parti.Notu, Parti.ReceteYazildi, Parti.ReceteTarihi, Parti.ReceteYazilisTarihi, Parti.RefakatKartiYazildi, Parti.Sinif, Parti.RafId, PartiGecikme.PartiGecikmeId, Firma.Adi, GecikmeTipi.Adi, PartiGecikme.Notu,
                   Parti.ParcaliSevk, Parti.FaturaKesilecek, IslemTipi.Adi, IslemGrubu.Adi, AciliyetDerecesi.Adi,
                   BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.SiparisTipiId, BoyaSiparis.Notu, Parti.VadeGun, Parti.ReceteId, Recete.Adi, Recete.RenkOkey,
                   Parti.Fason, Parti.BoyaSiparisKayitId, Parti.PartiTipiId,
                   DesenVaryant.DesenId, Desen.Adi, DesenVaryant.DesenVaryantId, DesenVaryant.VaryantAdi, Parti.EmulsiyonUrunId, Parti.PatMiktari, LabReceteAtis.Color, BoyaSiparis.PsKKEtiketTipiId, PsKKEtiketTipi.Adi,
                   PsKKEtiketTipi.PaketlemeEtiketNo,
                   PartiRpt.Degerlendirme, PartiRpt.BaslamaTarihi, PartiRpt.BitisTarihi, Desen.OzelYerlesimDosyaAdi, Desen.OzelYerlesimNotu,
                   Firma.Adresi1, Firma.Adresi2, Firma.Adresi3, Firma.Semt, Firma.Sehir, Firma.Ulke,
                   Firma.SevkAdresi1, Firma.SevkAdresi2, Firma.SevkAdresi3, Firma.SevkSemt, Firma.SevkSehir, Firma.SevkUlke,
                   BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparis.SiparisGrup3Id, SiparisGrup3.Adi, BoyaSiparis.SiparisGrup4Id, SiparisGrup4.Adi, BoyaSiparis.YurtDisi, Parti.SevkiyatNotu,
                   LabRecete.UretimNotu from Parti
              left join Firma on Firma.FirmaId = Parti.FirmaId
              left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
              left join Recete on Recete.ReceteId = Parti.ReceteId
              left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = Recete.LabReceteAtisId
              left join LabRecete on LabRecete.LabReceteId = LabReceteAtis.LabReceteId
              left join PartiGecikme on PartiGecikme.PartiKayitId = Parti.PartiKayitId and PartiGecikme.Bitti = 0
              left join GecikmeTipi on GecikmeTipi.GecikmeTipiId = PartiGecikme.GecikmeTipiId
              left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
              left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
              left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
              left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
              left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
              left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
              left join SiparisGrup SiparisGrup3 on SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
              left join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
              left join PsKKEtiketTipi on PsKKEtiketTipi.PsKKEtiketTipiID = BoyaSiparis.PsKKEtiketTipiId
              left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
              left join Desen on Desen.DesenId = DesenVaryant.DesenId
              Left join PartiRpt on PartiRpt.PartiKayitId = Parti.PartiKayitId and PartiRpt.SonKayit = 1
