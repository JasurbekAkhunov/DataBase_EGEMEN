CREATE PROCEDURE DESENINKULLANILDIGIPARTILER(
    ADESENID DESENKODUSIZE)
RETURNS (
    DESENVARYANTID INTEGER,
    DESENID DESENKODUSIZE,
    VARYANTADI VARCHAR(30),
    RENKCALISILDI INTEGER,
    HAMID INTEGER,
    RECETEID RECETEKODUSIZE,
    ACIKLAMA VARCHAR(255),
    MUSTERIBARKODNO VARCHAR(20),
    SIRANO INTEGER,
    RESIMDOSYAADI VARCHAR(255),
    KAPALI INTEGER,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40),
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    PARTISIRANO INTEGER,
    KILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    KAPADEDI INTEGER,
    PARTIIDSTR VARCHAR(30))
AS
begin
  For select DesenVaryant.DesenVaryantId, DesenVaryant.DesenId, DesenVaryant.VaryantAdi, DesenVaryant.RenkCalisildi, DesenVaryant.HamId, DesenVaryant.ReceteId,
             DesenVaryant.Aciklama, DesenVaryant.MusteriBarkodNo, DesenVaryant.SiraNo, DesenVaryant.ResimDosyaAdi, DesenVaryant.Kapali, DesenVaryant.InsertKullaniciId,
             DesenVaryant.InsertTarihi, DesenVaryant.KullaniciId, DesenVaryant.DegisimTarihi,
             Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId,
             PartiStok.PartiSiraNo, PartiStok.Kilo, PartiStok.Metre, PartiStok.KapAdedi, IK.Adi, SK.Adi from DesenVaryant
  left join Parti on Parti.DesenVaryantId = DesenVaryant.DesenVaryantId
  left join PartiStok on PartiStok.PartiKayitId = Parti.PArtiKayitId
  Left Join Kullanici IK on IK.KullaniciId = DesenVaryant.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = DesenVaryant.KullaniciId
  where DesenVaryant.DesenId = :ADesenId
  into :DesenVaryantId, :DesenId, :VaryantAdi, :RenkCalisildi, :HamId, :ReceteId, :Aciklama, :MusteriBarkodNo, :SiraNo, :ResimDosyaAdi, :Kapali, :InsertKullaniciId,
       :InsertTarihi, :KullaniciId, :DegisimTarihi, :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :PartiSiraNo, :Kilo, :Metre, :KapAdedi,
       :InsertKullaniciAdi, :KullaniciAdi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    suspend;
  end
end
