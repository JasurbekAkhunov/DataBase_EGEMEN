CREATE PROCEDURE DEPOSAYIMBARKODLUDETAYEKLE(
    ABARKOD VARCHAR(40),
    ADEPOSAYIMID INTEGER,
    AKILO DOUBLE PRECISION,
    ABRUTKILO DOUBLE PRECISION,
    AMETRE DOUBLE PRECISION,
    AKAPADEDI INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable Renkid integer;
declare variable DESENVARYANTID integer;
declare variable KaliteNo integer;
declare variable DesenID DESENKODUSIZE;
declare variable Notu    Memo;
declare variable FlamKodu   varchar(20);
declare variable MelanjKodu varchar(20);
declare variable Lot        varchar(150);
declare variable HamID      integer;
declare variable shtAlis        SmallInt;
declare variable shtFasonUretim SmallInt;
declare variable shtUretim      SmallInt;

declare variable FBoyaSiparisKayitID integer;
declare variable FBoyaSiparisDetayID integer;
declare variable FBoyaSiparisBaskiId integer;
declare variable FOrmeSiparisDetayID integer;
declare variable FUreticiId integer;
declare variable FDepoId     varchar(10);
declare variable IplikStokNumarator integer;
declare variable Sakat integer;
declare variable StokHareketID integer;
declare variable OrmeRafId integer;
declare variable IK     varchar(2);
declare variable TF     integer;
declare variable PaketId integer;
declare variable Kilo  Double Precision;
declare variable BrutKilo Double Precision;
begin
  if (AKilo is Null) then AKilo = 0;

  Select shtAlis, shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir into :shtAlis, :shtFasonUretim, :shtUretim;

  Select StokHareket.HamID, StokHareket.RenkID, StokHareket.Lot, StokHareket.MelanjKodu, StokHareket.FlamKodu, StokHareket.Notu,
         StokHareket.DesenID, StokHareket.KaliteNo, StokHareket.DesenVaryantID from StokHareketDetay
  Left Join StokHareket on StokHareket.StokHareketId = StokHareketDetay.StokHareketID
  where StokHareketDetay.Barkod = :ABarkod  and (StokHareketDetay.HareketTipi = :shtAlis or StokHareketDetay.HareketTipi = :shtFasonUretim or StokHareketDetay.HareketTipi = :shtUretim)
  into :HamID, :RenkID, :Lot, :MelanjKodu, :FlamKodu, :Notu, :DesenID, :KaliteNo, :DesenVaryantID;


  if (HamID is Null) then
  begin
    IK = EgemenCopyStr(ABarkod, 1, 2);

    if (IK = 'K-') then
    begin
      TF = EgemenStrToIntOlurmu(EgemenCopyStr(ABarkod, 3, Char_Length(ABarkod) - 2));
      if (TF = 0) then exception BarkodHatali1;
      PaketId = EgemenStrToInt(EgemenCopyStr(ABarkod, 3, Char_Length(ABarkod) - 2));

      Select first 1 StokHareket.StokHareketID, PSKaliteKontrol.Kilo, PSKaliteKontrol.Kilo + PSKaliteKontrol.Dara, StokHareket.HamID, StokHareket.BoyaSiparisKayitID, StokHareket.BoyaSiparisDetayID,
           StokHareket.BoyaSiparisBaskiId, StokHareket.OrmeSiparisDetayID, StokHareket.Lot, StokHareket.RenkID, StokHareket.UreticiId, StokHareket.MelanjKodu, StokHareket.DepoId, StokHareket.Sakat,
           StokHareket.Notu, StokHareket.DesenID, StokHareket.KaliteNo, StokHareket.DesenVaryantID, StokHareket.FlamKodu from PsKaliteKontrol
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = PsKaliteKontrol.PartiStokTartiId
      Left Join StokHareket on StokHareket.BoyaIrsaliyeDetayID = PartiStokTarti.BoyaIrsaliyeDetayID
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where PsKaliteKontrol.PsKaliteKontrolId = :PaketId
      into :StokHareketID, :Kilo, :BrutKilo, :HamID, :FBoyaSiparisKayitID, :FBoyaSiparisDetayID, :FBoyaSiparisBaskiId, :FOrmeSiparisDetayID,
           :Lot, :RenkID, :FUreticiId, :MelanjKodu, :FDepoId, :Sakat, :Notu, :DesenID, :KaliteNo, :DesenVaryantID, :FlamKodu;


      if (StokHareketID is not Null) then
      begin
        insert into StokHareketDetay(StokHareketID, HareketTipi, PsKaliteKontrolID, Kilo, BrutKilo, InsertKullaniciId, KullaniciId) Values
                                    (:StokHareketID, :shtAlis, :PaketId, :Kilo, :BrutKilo, :AKullaniciId, :AKullaniciId);
      end

    end
    else
    if (IK = 'I-') then
    begin
      TF = EgemenStrToIntOlurmu(EgemenCopyStr(ABarkod, 3, Char_Length(ABarkod) - 2));
      if (TF = 0) then exception BarkodHatali1;
      PaketId = EgemenStrToInt(EgemenCopyStr(ABarkod, 3, Char_Length(ABarkod) - 2));

      Select first 1 StokHareket.StokHareketID, IplikPaketleme.NetKilo, IplikPaketleme.BrutKilo, StokHareket.HamID, StokHareket.BoyaSiparisKayitID, StokHareket.BoyaSiparisDetayID,
           StokHareket.BoyaSiparisBaskiId, StokHareket.OrmeSiparisDetayID, StokHareket.Lot, StokHareket.RenkID, StokHareket.UreticiId, StokHareket.MelanjKodu, StokHareket.DepoId, StokHareket.Sakat,
           StokHareket.Notu, StokHareket.DesenID, StokHareket.KaliteNo, StokHareket.DesenVaryantID, StokHareket.FlamKodu from IplikPaketleme
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiId
      Left Join StokHareket on StokHareket.BoyaIrsaliyeDetayID = PartiStokTarti.BoyaIrsaliyeDetayID
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      where IplikPaketleme.IplikPaketlemeID = :PaketId
      into :StokHareketID, :Kilo, :BrutKilo, :HamID, :FBoyaSiparisKayitID, :FBoyaSiparisDetayID, :FBoyaSiparisBaskiId, :FOrmeSiparisDetayID,
           :Lot, :RenkID, :FUreticiId, :MelanjKodu, :FDepoId, :Sakat, :Notu, :DesenID, :KaliteNo, :DesenVaryantID, :FlamKodu;

      if (StokHareketID is not Null) then
      begin
        insert into StokHareketDetay(StokHareketID, HareketTipi, IplikPaketlemeID, Kilo, BrutKilo, InsertKullaniciId, KullaniciId) Values
                                    (:StokHareketID, :shtAlis, :PaketId, :Kilo, :BrutKilo, :AKullaniciId, :AKullaniciId);
      end

    end
    else
    begin
      Select first 1 StokHareket.StokHareketID, IplikStok.Numarator, IplikStok.NetMiktar, IplikStok.Miktar, StokHareket.HamID, StokHareket.BoyaSiparisKayitID, StokHareket.BoyaSiparisDetayID,
           StokHareket.BoyaSiparisBaskiId, StokHareket.OrmeSiparisDetayID, StokHareket.Lot, StokHareket.RenkID, StokHareket.UreticiId, StokHareket.MelanjKodu, StokHareket.DepoId, StokHareket.Sakat,
           StokHareket.Notu, StokHareket.DesenID, StokHareket.KaliteNo, StokHareket.DesenVaryantID, StokHareket.FlamKodu from IplikStok
      Left Join StokHareket on StokHareket.IplikIrsaliyeDetayID = IplikStok.CikisIplikIrsaliyeDetayID
      where IplikStok.IplikStokID = :ABarkod and StokHareket.StokHareketID is not Null
      Order by IplikStok.Numarator Desc
      into :StokHareketID, :IplikStokNumarator, :Kilo, :BrutKilo, :HamID, :FBoyaSiparisKayitID, :FBoyaSiparisDetayID, :FBoyaSiparisBaskiId, :FOrmeSiparisDetayID,
           :Lot, :RenkID, :FUreticiId, :MelanjKodu, :FDepoId, :Sakat, :Notu, :DesenID, :KaliteNo, :DesenVaryantID, :FlamKodu;

      if (StokHareketID is not Null) then
      begin
        insert into StokHareketDetay(StokHareketID, HareketTipi, IplikStokID, IplikStokNumarator, Kilo, BrutKilo, InsertKullaniciId, KullaniciId) Values
                                    (:StokHareketID, :shtAlis, :ABarkod, :IplikStokNumarator, :Kilo, :BrutKilo, :AKullaniciId, :AKullaniciId);
      end
    end
    if (HamID is Null) then
    begin
      Select StokHareket.HamID, StokHareket.RenkID, StokHareket.Lot, StokHareket.MelanjKodu, StokHareket.FlamKodu, StokHareket.Notu,
             StokHareket.DesenID, StokHareket.KaliteNo, StokHareket.DesenVaryantID, OrmeKaliteKontrol.Kilo, OrmeKaliteKontrol.Kilo, OrmeKaliteKontrol.Metre, 1 from OrmeKaliteKontrol
        Left Join StokHareket on StokHareket.StokHareketId = OrmeKaliteKontrol.StokHareketID
        where OrmeKaliteKontrol.OrmeKaliteKontrolId = :ABarkod
      into :HamID, :RenkID, :Lot, :MelanjKodu, :FlamKodu, :Notu, :DesenID, :KaliteNo, :DesenVaryantID, :AKilo, :ABrutKilo, :AMetre, :AKapAdedi;

    end

  end


  if (HamID is Null) then Exception BarkodHatali2;

  if (AKilo = 0) then
  begin
    select (GirisKilo - CikisKilo), (GirisBrutKilo - CikisBrutKilo), (GirisMetre - CikisMetre), (GirisKapAdedi - CikisKapAdedi) from StokHareketDetayMiktar
      where StokHareketDetayMiktar.Barkod = :ABarkod
    into :AKilo, :ABrutKilo, :AMetre, :AKapAdedi;

    if (AKilo is Null) then AKilo = 0;
  end

  insert into DepoSayimDetay(DepoSayimID, Barkod, HamID, RenkID, Lot, MelanjKodu, FlamKodu, Notu, DesenID, KaliteNo, DesenVaryantID, Kilo, BrutKilo, Metre, KapAdedi, KullaniciID, InsertKullaniciID)
                   Values (:ADepoSayimID, :ABarkod, :HamID, :RenkID, :Lot, :MelanjKodu, :FlamKodu, :Notu, :DesenID, :KaliteNo, :DesenVaryantID,:AKilo, :ABrutKilo, :AMetre, 1, :AKullaniciID, :AKullaniciID);
end
