CREATE PROCEDURE HAMSTOGUNAKIBETI(
    AHAMSTOKMASTERID INTEGER,
    AHAMSTOKDETAYID INTEGER)
RETURNS (
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    IPTAL INTEGER,
    IPTALSTR VARCHAR(5),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SEVKIRSALIYENOVETARIHI VARCHAR(100),
    SEVKFATURANOVETARIHI VARCHAR(100),
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISSARJID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MODELNO VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    KAPADEDI INTEGER,
    KULLANILANMIKTAR NUMERIC(12,3),
    ADET INTEGER,
    HAKTARMAFISIDSTR VARCHAR(30),
    HAMSTOKSARFFISIIDSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    IPTALTARIHI TIMESTAMP,
    IPTALFISNOSTR VARCHAR(30),
    IPTALNOTU MEMO,
    IPTALRAFID RAFKODUSIZE,
    AKTARMAYACIKANKILO NUMERIC(12,3),
    PARTINOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    IRSALIYEKILO BOYAMIKTARSIZE,
    SERTIFIKAPARTIIDSTR VARCHAR(255),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    RENKADI VARCHAR(40))
AS
declare variable BoyaSiparisYil integer;
declare variable BoyaSiparisId integer;
declare variable HamIadeRezervId integer;
declare variable FisYil integer;
declare variable FisNo integer;
declare variable FPartiStokIptalKayitId integer;
declare variable FIrsaliyeNo     IrsaliyeNoSize;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable FFaturaNo     FaturaNoSize;
declare variable FFaturaTarihi TimeStamp;
declare variable FPartiStokIptalYil integer;
declare variable FPartiStokIptalID integer;
declare variable FSertifikaNo varchar(30);
declare variable FBoyaSiparisRezervId integer;
declare variable FPartiYil integer;
declare variable FPartiId integer;
declare variable FPartiSiraNo integer;
declare variable FPartiIdStr varchar(255);
begin
  if (AHamStokDetayId is not null) then Select HamStokMasterId from HamStokDetay where HamStokDetay.HamStokDetayID = :AHamStokDetayID into :AHamStokMasterId;

  HamStokMasterId = :AHamStokMasterId;
  select HamStokMaster.SertifikaNo from HamStokMaster where HamStokMaster.HamStokMasterID = :HamStokMasterID into :FSertifikaNo;

  For Select HamStokDetayId from HamStokDetay where HamStokDetay.HamStokMasterId = :AHamStokMasterId and (HamStokDetay.HamStokDetayId = :AHamStokDetayId or :AHamStokDetayId is null)
    into :HamStokDetayId do
  begin
    SertifikaPartiIdStr = '';
    for select Parti.PartiYil, Parti.PartiId, PartiStok.PartiSiraNo from HamStokMaster
          left join Hamstokdetay on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterID
          left join BoyaSiparisRezervSertifika on HamStokDetay.HamStokDetayID = BoyaSiparisRezervSertifika.HamStokDetayId
          left join BoyaSiparisRezerv on boyaSiparisRezervSertifika.BoyaSiparisRezervId = BoyaSiparisRezerv.BoyaSiparisRezervId

          left join PartiStokDetay on PartiStokDetay.BoyaSiparisRezervId = boyasiparisrezervsertifika.BoyaSiparisRezervId
          left join PartiStok on PartiStok.PartiSinifID = PartiStokDetay.PartiSinifID
          left join Parti on Parti.PartiKayitId = PartiStokDetay.PartiKayitId
        where HamStokMaster.HAmStokMasterID = :AHamStokMasterId  and BoyaSiparisRezerv.BoyaSiparisRezervId is not null
        group by Parti.PartiYil, Parti.PartiId, PartiStok.PartiSiraNo
        order by 1,2,3 into :FPartiYil, :FPartiId, :FPartiSiraNo do
     begin
         FPartiIdStr = '(' || EgemenIntegerToStr(FPartiYil)  || ') ' || EgemenIntegerToStr(FPartiId) || '.' || EgemenIntegerToStr(FPartiSiraNo) || ' ';

         if (Char_Length(SertifikaPartiIdStr) + Char_Length(FPartiIdStr) < 250) then
         begin
           if (SertifikaPartiIdStr <> '' ) then SertifikaPartiIdStr = SertifikaPartiIdStr || cast(', ' as varchar(2) character set UTF8);
           SertifikaPartiIdStr = SertifikaPartiIdStr || FPartiIdStr;
         end
       end



    For select Parti.PartiKayitId, PartiYil, PartiId, Parti.FirmaId, Firma.Adi, Parti.Modelno, Parti.ReceteId, Recete.Adi, PartiStokDetay.KapAdedi, PartiStokDetay.Miktar,
               PartiStokDetay.Adet, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo,
               PartiStok.PartiSinifId, PartiStok.PartiStokIptalKayitId, PartiStokIptal.IptalTarihi, PartiStokIptal.PartiStokIptalYil, PartiStokIptal.PartiStokIptalID, Parti.Notu,
               PartiStokIptal.IptalNotu, PartiStokIptal.RafId, BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.RenkAdi from rafdetay
      left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
      left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
      left join Parti on Parti.PartiKayitId = PartiStokDetay.PartiKayitId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join Firma on Firma.FirmaId = Parti.FirmaId
      left join BoyaSiparis on BoyaSiparis.BoyaSipariskayitid = Parti.BoyaSiparisKayitId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayid = PartiStok.BoyaSiparisDetayId
      left join PartiStokIptal on PartiStokIptal.PartiStokIptalKayitId = PartiStok.PartiStokIptalKayitId
      left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
      where RafDetay.hamstokdetayid = :HamStokDetayId and Parti.PartiKayitId is not null
      into :PartiKayitId, :PartiYil, :PartiId, :FirmaId, :FirmaAdi, :ModelNo, :Receteid, :ReceteAdi, :KapAdedi, :KullanilanMiktar, :Adet,
           :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayID, :MusteriSiparisNo, :PartiSinifId, :FPartiStokIptalKayitId, :IptalTarihi, :FPartiStokIptalYil, :FPartiStokIptalID, :PartiNotu,
           :IptalNotu, :IptalRafId, :LabReceteKodu,  :LabReceteAdi, :RenkAdi do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      else BoyaSiparisIdStr = '';

      if (FPartiStokIptalID is not null) then IptalFisNoStr = '(' || EgemenIntegerToStr(FPartiStokIptalYil)  || ') ' || EgemenIntegerToStr(FPartiStokIptalID) || ' ';
      else IptalFisNoStr = '';

      SevkIrsaliyeNoveTarihi = '';
      SevkFaturaNoveTarihi = '';
      if (FPartiStokIptalKayitId is not null) then
      begin
        Iptal    = 1;
        IptalStr = 'İptal';
      end
      else
      begin
        Iptal    = 0;
        IptalStr = '';

        for Select BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeNo from BoyaIrsaliyeDetay
          left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliyeDetay.BoyaIrsaliyeId
          where BoyaIrsaliyeDetay.PartiSinifID = :PartiSinifID
          group by BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeNo
          into :FIrsaliyeTarihi, :FIrsaliyeNo do
        begin
          if (Char_Length(SevkIrsaliyeNoveTarihi) < 100) then
          begin
            if (SevkIrsaliyeNoveTarihi <> '') then SevkIrsaliyeNoveTarihi = SevkIrsaliyeNoveTarihi || cast(', ' as varchar(2) character set UTF8);
            SevkIrsaliyeNoveTarihi = TimeStampToDateStr(FIrsaliyeTarihi) || ' - ' || Cast(FIrsaliyeNo as varchar(20));
          end
        end
        for Select BoyaFatura.FaturaTarihi, BoyaFatura.FaturaNo from BoyaIrsaliyeDetay
          left join BoyaFatura on BoyaFatura.BoyaFaturaID = BoyaIrsaliyeDetay.BoyaFaturaId
          where BoyaIrsaliyeDetay.PartiSinifID = :PartiSinifID and BoyaIrsaliyeDetay.BoyaFaturaId is not null
          group by BoyaFatura.FaturaTarihi, BoyaFatura.FaturaNo
          into :FFaturaTarihi, :FFaturaNo do
        begin
          if (Char_Length(SevkFaturaNoveTarihi) < 70) then
          begin
            if (SevkFaturaNoveTarihi <> '') then SevkFaturaNoveTarihi = SevkFaturaNoveTarihi || cast(', ' as varchar(2) character set UTF8);
            SevkFaturaNoveTarihi = TimeStampToDateStr(FFaturaTarihi) || ' - ' || Format_Float('',FFaturaNo);
          end
        end
      end
      suspend;
    end

    IptalFisNoStr = Null;
    IptalTarihi = Null;
    Iptal    = 0;
    IptalStr = '';
    IptalNotu = Null;
    IptalRafId = Null;
    PartiKayitId   = null;
    PartiYil       = null;
    PartiId        = null;
    PartiSinifId   = null;
    MusteriSiparisNo = Null;
    PartiIdStr     = null;
    ModelNo        = null;
    FirmaId        = null;
    FirmaAdi       = null;
    ReceteId       = null;
    ReceteAdi      = null;
    IrsaliyeSeriNo = null;
    IrsaliyeNo     = null;
    IrsaliyeTarihi = null;
    KapAdedi       = null;
    KullanilanMiktar = null;
    Adet           = null;
    BoyaSiparisIdStr = Null;
    BoyaSiparisDetayId = Null;
    AktarmayaCikanKilo = Null;
    PartiNotu = Null;
    SevkIrsaliyeNoveTarihi = '';
    SevkFaturaNoveTarihi = '';

    For select BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi, BoyaFatura.FaturaNo, BoyaFatura.FaturaTarihi,
               BoyaIrsaliyeDetay.KapAdedi, BoyaIrsaliyeDetay.Miktar, BoyaIrsaliyeDetay.Adet, BoyaIrsaliyeDetay.HamIadeRezervId, Firma.FirmaId, Firma.Adi,
               BoyaIrsaliyeDetay.Kilo, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId from RafDetay
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.rafdetayId = RafDetay.rafDetayId
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaIrsaliyeDetay.BoyaSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
      left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
      left join BoyaFatura on BoyaFatura.BoyaFaturaID = BoyaIrsaliyeDetay.BoyaFaturaId
      where RafDetay.HamStokDetayId = :HamStokDetayId and BoyaIrsaliyeDetay.RafDetayId is not null
      into :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FFaturaNo, :FFaturaTarihi, :KapAdedi, :KullanilanMiktar, :Adet, :HamIadeRezervId, :FirmaId, :FirmaAdi,
      :IrsaliyeKilo, :BoyaSiparisYil, :BoyaSiparisId do
    begin
      BoyaSiparisIdStr = Null;
      if (HamIadeRezervId is not null) then PartiIdStr = 'İade';
      else
      begin
        PartiIdStr = 'Satış';
        if (BoyaSiparisYil is not null) then
        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      end
      SevkIrsaliyeNoveTarihi = TimeStampToDateStr(IrsaliyeTarihi) || ' - ' || cast(IrsaliyeNo as varchar(20));

      if (FFaturaNo is not null) then SevkFaturaNoveTarihi = TimeStampToDateStr(FFaturaTarihi) || ' - ' || Format_Float('',FFaturaNo);
      else SevkFaturaNoveTarihi = '';

      suspend;
    end

    PartiKayitId   = null;
    PartiIdStr     = null;
    FirmaId        = null;
    FirmaAdi       = null;
    ModelNo        = null;
    ReceteId       = null;
    IrsaliyeSeriNo = null;
    IrsaliyeNo     = null;
    IrsaliyeTarihi = null;
    KapAdedi       = null;
    KullanilanMiktar = null;
    Adet           = null;
    BoyaSiparisIdStr = Null;
    BoyaSiparisDetayId = Null;
    AktarmayaCikanKilo = Null;
    SevkIrsaliyeNoveTarihi = '';
    SevkFaturaNoveTarihi = '';
    IrsaliyeKilo = Null;

    For select BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisRezerv.BoyaSiparisDetayId,
  BoyaSiparisRezerv.BoyaSiparisSarjId, BoyaSiparisRezerv.Miktar - BoyaSiparisRezerv.KullanilanMiktar, BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.RenkAdi from BoyaSiparisRezerv
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayid = BoyaSiparisRezerv.BoyaSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSipariskayitid = BoyaSiparisDetay.BoyaSiparisKayitId
      left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
      where BoyaSiparisRezerv.HamStokDetayId = :HamStokDetayId
      into :BoyaSiparisYil, :BoyaSiparisId, :MusteriSiparisNo, :BoyaSiparisDetayId, :BoyaSiparisSarjId, :KullanilanMiktar, :LabReceteKodu,  :LabReceteAdi, :RenkAdi do
    begin
      PartiIdStr = 'Rezerv';
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      if (KullanilanMiktar > 0 ) then suspend;
    end

    PartiKayitId   = null;
    PartiIdStr     = null;
    MusteriSiparisNo = Null;
    ModelNo        = null;
    ReceteId       = null;
    IrsaliyeSeriNo = null;
    IrsaliyeNo     = null;
    IrsaliyeTarihi = null;
    KapAdedi       = null;
    KullanilanMiktar = null;
    Adet           = null;
    BoyaSiparisIdStr = Null;
    BoyaSiparisDetayId = Null;
    AktarmayaCikanKilo = Null;

    For Select HAktarmaFisi.FisYil, HAktarmaFisi.FisNo, HAktarmaFisi.FisTarihi, HAktarmaFisiDetay.Kilo from HAktarmaFisiDetay
      left join HAktarmaFisi on HAktarmaFisi.HAktarmaFisiid = HAktarmaFisiDetay.HAktarmaFisiId
      where HAktarmaFisiDetay.HamStokDetayId = :HamStokDetayId
    into :FisYil, :FisNo, :FisTarihi, :AktarmayaCikanKilo do
    begin
      PartiIdStr = 'Aktarmaya Çıkış';
      HAktarmaFisIdStr = '(' || EgemenIntegerToStr(FisYil)  || ') ' || EgemenIntegerToStr(FisNo) || ' ';
      suspend;
    end

    HAktarmaFisIdStr = '';
    FisYil = Null;
    FisNo = Null;
    FisTarihi = NUll;
    AktarmayaCikanKilo = 0;

    For Select HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo, HamStokSarfFisi.FisTarihi, HamStokSarfFisiDetay.Miktar from rafdetay
      left join HamStokSarfFisiDetay on HamStokSarfFisiDetay.rafdetayId = RafDetay.rafDetayId
      left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiid = HamStokSarfFisiDetay.HamStokSarfFisiId
      where rafdetay.HamStokDetayId = :HamStokDetayId and HamStokSarfFisiDetay.rafdetayId is not null
    into :FisYil, :FisNo, :FisTarihi, :KullanilanMiktar do
    begin
      PartiIdStr = 'Sarf Fişi ile Çıkış';
      HamStokSarfFisiIdStr = '(' || EgemenIntegerToStr(FisYil)  || ') ' || EgemenIntegerToStr(FisNo) || ' ';
      suspend;
    end

    FisYil = Null;
    FisNo = Null;
    FisTarihi = NUll;
    HamStokSarfFisiIdStr = '';

    PartiIdStr = 'İade Rezerv';
    KullanilanMiktar = 0;
    Select Sum(RafDetay.IadeEdilecekMiktar) from RafDetay where RafDetay.HamStokDetayID  = :HamStokDetayId into :KullanilanMiktar;

    if (KullanilanMiktar is not null and KullanilanMiktar > 0) then suspend;
  end
end
