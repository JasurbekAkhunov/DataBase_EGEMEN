CREATE PROCEDURE KASAHAREKETIGETIR(
    AKASAID VARCHAR(10),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1))
RETURNS (
    KASAHAREKETID INTEGER,
    KASAID VARCHAR(10),
    TARIH TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    BANKAHESAPID VARCHAR(20),
    BANKAHESAPADI VARCHAR(50),
    ACIKLAMA VARCHAR(50),
    KASAHARCAMATIPIID VARCHAR(10),
    KASAHARCAMATIPIADI VARCHAR(30),
    BORCALACAK VARCHAR(1),
    SINIF VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    DOVIZKURUSTR VARCHAR(40),
    ISLEMDOVIZMEBLAG DOUBLE PRECISION,
    ISLEMDOVIZID VARCHAR(5),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    REFERANSYERI INTEGER,
    REFERANSID INTEGER)
AS
declare Variable ToplamBorc double precision;
declare variable ToplamAlacak double precision;
declare variable CsReferansId integer;
declare variable StReferansId integer;
declare variable AcReferansId integer;
declare variable NfReferansId integer;
declare variable DovizKuru1 Double Precision;
declare variable DovizKuru2 Double Precision;
declare variable IslemDovizKuru1 Double Precision;
declare variable IslemDovizKuru2 Double Precision;
declare variable FirmaDovizId varchar(5);
declare variable BankaDovizId varchar(5);
declare variable HedefDovizId varchar(5);
declare variable KasaDovizId varchar(5);
begin
  ToplamBorc = 0;
  ToplamAlacak = 0;
  select  sum(Meblag)  from KasaHareket where KasaId = :AKasaId and tarih < :ABaslangicTarihi and BorcAlacak = 'B' and (KasaHareket.Sinif = :ASinif or :ASinif is null) into :ToplamBorc;
  select  sum(Meblag)  from KasaHareket where KasaId = :AKasaId and tarih < :ABaslangicTarihi and BorcAlacak = 'A' and (KasaHareket.Sinif = :ASinif or :ASinif is null) into :ToplamAlacak;

  select DovizId from Kasa Where Kasa.KasaId = :AKasaId into :KasaDovizId;

  KasaHareketId = -1;
  Tarih  = null;
  KasaId = AKasaId;
  FirmaId = null;
  FirmaAdi = null;
  Sinif = Null;
  BankaHesapId = Null;
  BankaHesapAdi = Null;
  Aciklama = 'Devir';
  KasaHarcamaTipiId = null;
  BorcAlacak = null;
  Meblag = null;
  IslemDovizMeblag = Null;
  IslemDovizID = Null;
  Notu = null;
  KullaniciAdi = null;
  DegisimTarihi = null;
  Borc = 0;
  Alacak = 0;

  if (ToplamBorc > 0) then Borc = ToplamBorc;
  if (ToplamAlacak > 0) then Alacak = ToplamAlacak;

  if (Borc > Alacak) then
  begin
    Borc = Borc - Alacak;
    Alacak = Null;
  end
  else
  begin
    Alacak = Alacak - Borc;
    Borc = Null;
  end
  suspend;

  for select KasaHareket.KasaId, KasaHareket.Tarih, KasaHareket.KasaHareketId, KasaHareket.FirmaId, Firma.Adi, Firma.Adi2, Firma.DovizId, KasaHareket.BankaHesapId, BankaHesap.Adi, BankaHesap.DovizId, KasaHareket.Aciklama,
             KasaHareket.BorcAlacak, KasaHareket.Sinif, KasaHareket.Meblag, KasaHareket.DovizKuru1, KasaHareket.DovizKuru2, KasaHareket.Notu, KasaHareket.KasaHarcamaTipiId,
             KasaHarcamaTipi.Adi, Kullanici.Adi, KasaHareket.DegisimTarihi,CSReferansId, STReferansId, AcReferansId, NfReferansId,
             KasaHareket.IslemDovizMeblag, KasaHareket.IslemDovizID, KasaHareket.IslemDovizKuru1, KasaHareket.IslemDovizKuru2 from KasaHareket
  left join Firma on Firma.FirmaId = KasaHareket.FirmaId
  left join BankaHesap on BankaHesap.BankaHesapId = KasaHareket.BankaHesapId
  left join KasaHarcamaTipi on KasaHarcamaTipi.KasaHarcamaTipiId = KasaHareket.KasaHarcamaTipiId
  left join Kullanici on Kullanici.KullaniciId = KasaHareket.KullaniciId
  where KasaHareket.KasaId = :AKasaId and
        (KasaHareket.Tarih >= :ABaslangicTarihi and KasaHareket.Tarih <= :ABitisTarihi) and
        (KasaHareket.Sinif = :ASinif or :ASinif is null)
  order by KasaHareket.KasaHareketId
  into :KasaId, :Tarih, :KasaHareketId, :FirmaId, :FirmaAdi, :FirmaAdi2, :FirmaDovizId, :BankaHesapId, :BankaHesapAdi, :BankaDovizId, :Aciklama,
       :BorcAlacak, :Sinif,  :Meblag, :DovizKuru1, :DovizKuru2, :Notu, :KasaHarcamaTipiId, :KasaHarcamaTipiAdi, :KullaniciAdi, :DegisimTarihi, :CSReferansId,
       :STReferansId, :AcReferansId , :NfReferansId, :IslemDovizMeblag, :IslemDovizID, :IslemDovizKuru1, :IslemDovizKuru2 do
  begin
    ReferansYeri = null;

    if (CsReferansId is not null) then
    begin
      ReferansYeri = 1;     /* Ceksenet tahsilat */
      ReferansId = CsReferansId;
    end
    else if (StReferansId is not null) then
    begin
      ReferansYeri = 2;     /* Stok */
      ReferansId = StReferansId;
    end
    else if (AcReferansId is not null) then
    begin
      ReferansYeri = 3;     /* Avukat Cari */
      ReferansId = AcReferansId;
    end
    else if (NfReferansId is not null) then
    begin
      ReferansYeri = 4;     /* Nakit Fatura  */
      ReferansId = NfReferansId;
    end

    Borc = Null;
    Alacak = Null;

    if (BorcAlacak = 'B') then Borc = Meblag;
    else Alacak = Meblag;

    if (BankaDovizId is not null) then HedefDovizId = BankaDovizId;
    else if (FirmaDovizId is not null) then HedefDovizId = FirmaDovizId;
    else HedefDovizId = KasaDovizId;

    if (KasaDovizID <> IslemDovizId) then
    begin
      DovizKuruStr = Format_Float('', IslemDovizKuru1) || ' ' || KasaDovizId || '-' || Format_Float('', ISlemDovizKuru2) || ' ' || IslemDovizId;
    end
    else DovizKuruStr = '';

    suspend;
  end
end
