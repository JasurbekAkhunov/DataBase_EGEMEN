CREATE PROCEDURE PARTIREFAKATKARTIONGORU(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    HAMSTOKDETAYID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    YIKAMAONCESIEN DOUBLE PRECISION,
    YIKAMASONRASIEN DOUBLE PRECISION,
    ENISTENEN VARCHAR(10),
    ONGORULENEN INTEGER,
    YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    YIKAMASONRASIGRAMAJ DOUBLE PRECISION,
    GRAMAJISTENEN VARCHAR(10),
    ONGORULENGRAMAJ INTEGER,
    ENDENCEKMEZLIK DOUBLE PRECISION,
    BOYDANCEKMEZLIK DOUBLE PRECISION,
    CEKMEZLIKENISTENEN DOUBLE PRECISION,
    CEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    ONGORULENENDENCEKMEZLIK DOUBLE PRECISION,
    ONGORULENBOYDANCEKMEZLIK DOUBLE PRECISION,
    YIKAMAHASLIGIISTENEN VARCHAR(10),
    HAMDONME DOUBLE PRECISION,
    DONMEISTENEN VARCHAR(10),
    ONGORULENDONME VARCHAR(10),
    YIKAMAONCESIRAPORBOYU VARCHAR(100),
    YIKAMASONRASIRAPORBOYU VARCHAR(100),
    ISTENENRAPORBOYU VARCHAR(100),
    ONGORULENRAPORBOYU VARCHAR(100),
    YIKAMAHASLIGIYAPILDI INTEGER,
    MERSERIZEHASLIGIYAPILDI INTEGER,
    YIKAMAHASLIGIKABUL INTEGER,
    MERSERIZEHASLIGIKABUL INTEGER,
    DEMIRTESTIYAPILDI INTEGER,
    DEMIRTESTIKABUL INTEGER,
    NEMORANI DOUBLE PRECISION,
    ONGORUNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ONAYNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    HAMSTOKMASTERID INTEGER)
AS
begin
  For Select DISTINCT PartiStok.PartiSinifId, HamStokDetay.HamStokDetayId, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaSonrasiEn, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetayOngoru.OngorulenEn,
             HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiGramaj, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetayOngoru.OngorulenGramaj,
             HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen, BoyaSiparisDetay.YikamaHasligiIstenen,
             BoyaSiparisDetayOngoru.OngorulenEndenCekmezlik, BoyaSiparisDetayOngoru.OngorulenBoydanCekmezlik,
             HamStokMaster.HamDonme, BoyaSiparisDetay.DonmeIstenen, BoyaSiparisDetayOngoru.OngorulenDonme,
             HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu, BoyaSiparisDetay.IstenenRaporBoyu, BoyaSiparisDetayOngoru.OngorulenRaporBoyu,
             HamStokMaster.NemOrani, HamStokMaster.YikamaHasligiYapildi, HamStokMaster.MerserizeHasligiYapildi,
             HamStokMaster.YikamaHasligiKabul, HamStokMaster.MerserizeHasligiKabul, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi,
             BoyaSiparisDetayOngoru.OnayNotu, BoyaSiparisDetayOngoru.OngoruNotu,
             HamStokMaster.DemirTestiYapildi, HamStokMaster.DemirTestiKabul, HamStokMaster.HamStokMasterId from PartiStok
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparisDetayOngoru on BoyaSiparisDetayOngoru.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId and BoyaSiparisDetayOngoru.HamStokDetayId = RafDetay.HamStokDetayId
    where PartiStok.PartiKayitId = :APartiKayitId
    into :PartiSinifId, :HamStokDetayId, :YikamaOncesiEn, :YikamaSonrasiEn, :EnIstenen, :OngorulenEn, :YikamaOncesiGramaj, :YikamaSonrasiGramaj, :GramajIstenen, :OngorulenGramaj,
         :EndenCekmezlik, :BoydanCekmezlik, :CekmezlikEnIstenen, :CekmezlikBoyIstenen, :YikamaHasligiIstenen, :OngorulenEndenCekmezlik, :OngorulenBoydanCekmezlik,
         :HamDonme, :DonmeIstenen, :OngorulenDonme, :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu, :IstenenRaporBoyu, :OngorulenRaporBoyu,
         :NemOrani, :YikamaHasligiYapildi, :MerserizeHasligiYapildi, :YikamaHasligiKabul, :MerserizeHasligiKabul,
         :IrsaliyeNo, :IrsaliyeTarihi, :OnayNotu, :OngoruNotu, :DemirTestiYapildi, :DemirTestiKabul, :HamStokMasterId do
  begin
    suspend;
  end
end
