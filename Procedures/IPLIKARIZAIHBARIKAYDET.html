CREATE PROCEDURE IPLIKARIZAIHBARIKAYDET(
    AIPLIKMAKINAID VARCHAR(10),
    APERSONELID INTEGER,
    AARIZAYERIID INTEGER,
    AARIZABASLADI INTEGER)
RETURNS (
    STATUS INTEGER,
    MESAJ VARCHAR(255))
AS
declare variable KayitSayisi                integer;
declare variable BasladiIplikMakinaDurumuId integer;
declare variable BittiIplikMakinaDurumuId   integer;
declare variable IplikArizaIhbarId          integer;
declare variable FStatus                    integer;
declare variable IplikArizaBakimId          integer;
declare variable GittiIplikMakinaDurumuId   integer;
declare variable BakimPersonelId            integer;
begin
  if (AArizaBasladi = 1) then
  begin
    Select Count(*) from IplikArizaIhbar where IplikArizaIhbar.IplikMakinaId = :AIplikMakinaId and IplikArizaIhbar.BittiIplikMakinaDurumuId is null into :KayitSayisi;

    if (KayitSayisi = 0) then
    begin
      BasladiIplikMakinaDurumuId = gen_Id(IplikMakinaDurumu_id_gen, 1);
      insert into IplikMakinaDurumu(IplikMakinaDurumuId, IplikMakinaId, Tarih, Ariza) values(:BasladiIplikMakinaDurumuId, :AIplikMakinaId, Current_TimeStamp(0), 1); /* ariza basladi  */

      insert into IplikArizaIhbar(IplikMakinaId, ArizaYeriId, BasladiIplikMakinaDurumuId, BaslatanPersonelId) values(:AIplikMakinaId, :AArizaYeriId, :BasladiIplikMakinaDurumuId, :APersonelId);
      Status = 1;
      mesaj = 'Arıza Kaydı Yapıldı';
    end
    else
    begin
      Status = 2;
      mesaj = 'Daha Önce Arıza kaydı alınmış, Arıza bitirilmemiş..';
    end
  end
  else if (AArizaBasladi = 2) then
  begin
    Select IplikArizaIhbarId from IplikArizaIhbar where IplikArizaIhbar.BittiIplikMakinaDurumuId is null and IplikArizaIhbar.IplikMakinaId = :AIplikMakinaId into :IplikArizaIhbarId;

    if (IplikArizaIhbarId is not null) then
    begin
      BittiIplikMakinaDurumuId = gen_Id(IplikMakinaDurumu_id_gen, 1);

      insert into IplikMakinaDurumu(IplikMakinaDurumuId, IplikMakinaId, Tarih, Ariza) values(:BittiIplikMakinaDurumuId, :AIplikMakinaId, Current_TimeStamp(0), 2); /* ariza giderildi  */

      Update IplikArizaIhbar Set BittiIplikMakinaDurumuId = :BittiIplikMakinaDurumuId, BitirenPersonelId = :APersonelId where IplikArizaIhbar.IplikArizaIhbarID = :IplikArizaIhbarId;

      for Select PersonelId, IplikArizaBakimId from IplikArizaBakim where IplikArizaBakim.IplikArizaIhbarId = :IplikArizaIhbarId and GittiIplikMakinaDurumuId is null into :BakimPersonelId, :IplikArizaBakimId do
      begin
        GittiIplikMakinaDurumuId = gen_Id(IplikMakinaDurumu_id_gen, 1);

        Select Count(*) from IplikArizaBakim where IplikArizaBakim.GittiIplikMakinaDurumuId is null and IplikArizaBakim.IplikMakinaId = :AIplikMakinaId and IplikArizaBakim.PersonelId <> :BakimPersonelId into :KayitSayisi;

        if (KayitSayisi = 0) then  insert into IplikMakinaDurumu(IplikMakinaDurumuId, IplikMakinaId, Tarih, ServisciGeldi, ServisciVar) values(:GittiIplikMakinaDurumuId, :AIplikMakinaId, Current_TimeStamp(0), 2, 0);
        else insert into IplikMakinaDurumu(IplikMakinaDurumuId, IplikMakinaId, Tarih, ServisciGeldi, ServisciVar) values(:GittiIplikMakinaDurumuId, :AIplikMakinaId, Current_TimeStamp(0), 2, 1);

        Update IplikArizaBakim Set GittiIplikMakinaDurumuId = :GittiIplikMakinaDurumuId where IplikArizaBakim.IplikArizaBakimID = :IplikArizaBakimId;
      end

      Status = 1;
      mesaj = 'Kayıt Yapıldı';
    end
    else
    begin
      Status = 3;
      mesaj = 'Bitmemiş Arıza İhbar kaydı yok.. ';
    end
  end
  else
  begin
    Status = 4;
    mesaj = 'Parametre Hatalı..Egemen Yazılıma Haber ver.. ';
  end
  suspend;
end
