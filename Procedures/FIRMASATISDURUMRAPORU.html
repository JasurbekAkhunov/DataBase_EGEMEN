CREATE PROCEDURE FIRMASATISDURUMRAPORU
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BIRIMID VARCHAR(2),
    SATISSIPARISMIKTAR NUMERIC(12,3),
    SATISPARTILENENMIKTAR NUMERIC(12,3),
    SATISDEPOCIKISMIKTAR NUMERIC(12,3),
    SATISSEVKEHAZIRMIKTAR NUMERIC(12,3),
    SATISSEVKEHAZIRNETMIKTAR NUMERIC(12,3),
    SATISBRUTSEVKEDILENMIKTAR NUMERIC(12,3),
    SATISNETSEVKEDILENMIKTAR NUMERIC(12,3),
    SATISISLETMEMIKTAR NUMERIC(12,3))
AS
begin
/* has örme tüm stokları kg olarak baktıgı icin metreyi dikkate almadık */
  for Select FirmaID, FirmaAdi, BirimId,
             sum(SatisSiparisMiktar), sum(SatisPartilenenMiktar), sum(SatisDepoCikisMiktar), sum(SatisSevkeHazirMiktar),
             sum(SatisSevkeHazirNetMiktar), sum(SatisBrutSevkEdilenMiktar), sum(SatisNetSevkEdilenMiktar)
             from FirmaSatisDurumRaporuHazirla
    Group by FirmaID, FirmaAdi, BirimId
    into :FirmaId, :FirmaAdi, :BirimId,
         :SatisSiparisMiktar, :SatisPartilenenMiktar, :SatisDepoCikisMiktar, :SatisSevkeHazirMiktar,
         :SatisSevkeHazirNetMiktar, :SatisBrutSevkEdilenMiktar, :SatisNetSevkEdilenMiktar do
  begin
    SatisIsletmeMiktar = SatisDepoCikisMiktar - SatisBrutSevkEdilenMiktar - SatisSevkeHazirMiktar;

    if (SatisSiparisMiktar <> 0 or SatisIsletmeMiktar <> 0) then suspend;
  end

end
