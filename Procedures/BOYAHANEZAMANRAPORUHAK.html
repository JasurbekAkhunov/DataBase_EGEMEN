    begin
      HamAdiToplami = '';

        left join HAktarmaFisiDetay on HAktarmaFisiDetay.HAktarmaFisiDetayId = HAktarmaIsEmriDetay.HAktarmaFisiDetayId
        left join RafDetay on RafDetay.RafDetayId = HAktarmaFisiDetay.RafDetayId
        left join Ham on Ham.HamId = RafDetay.HamId

        group By Ham.Adi
