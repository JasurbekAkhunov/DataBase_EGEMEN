CREATE PROCEDURE FIRMACALISMAANALIZIRAPORU(
    ABASLANGICTARIH TIMESTAMP,
    ABITISTARIH TIMESTAMP,
    APARTITIPIID INTEGER,
    ANUMUNEKILO NUMERIC(12,3))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IMALATTOPLAMI DOUBLE PRECISION,
    IMALATPARTIADEDI DOUBLE PRECISION,
    NUMUNETOPLAMI DOUBLE PRECISION,
    NUMUNEPARTIADEDI DOUBLE PRECISION,
    NUMUNEKILOIMALATAKILOORANI DOUBLE PRECISION,
    NADETIMALATADETORANI DOUBLE PRECISION,
    IMALATKILONPARTIADETORANI DOUBLE PRECISION,
    ORTALAMAPARTIKILOSU DOUBLE PRECISION,
    RENKCALISMAADEDI INTEGER)
AS
declare variable FPartiKayitId integer;
declare variable FIslemTipiId integer;
declare variable FKilo Numeric(12,3);
declare variable itNumune       integer;
begin
  Select itNumune from IslemTipiGetir into :itNumune;

  if (ANumuneKilo is null) then ANumuneKilo = 0;
  For Select Firma.FirmaId, Firma.Adi from Firma into :FirmaId, :FirmaAdi do
  begin
    ImalatToplami                     = 0;
    ImalatPartiAdedi                  = 0;
    NumuneToplami                     = 0;
    NumunePartiAdedi                  = 0;

    NumuneKiloImalataKiloOrani        = 0;
    NAdetImalatAdetOrani              = 0;
    ImalatKiloNPartiAdetOrani         = 0;
    OrtalamaPartiKilosu               = 0;
    RenkCalismaAdedi                  = 0;

    For Select Parti.PartiKayitId, Parti.IslemTipiId, Sum(PartiStok.Kilo) from Parti
      left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
      where Parti.FirmaID = :FirmaId and Parti.AcilisTarihi >= :ABaslangicTarih and Parti.Acilistarihi <= :ABitisTarih and
            (Parti.PartiTipiId = :APartiTipiId or :APartiTipiId is null) and PartiStok.PartiKayitId is not null and PartiStok.PartiStokIptalKayitId is null
      group by Parti.PartiKayitId, Parti.IslemTipiId
      into :FPartiKayitId, :FIslemTipiId, :FKilo do
    begin
      if (FIslemTipiID = itNumune or FKilo <= ANumuneKilo) then
      begin
        NumuneToplami    = NumuneToplami + FKilo;
        NumunePartiAdedi = NumunePartiAdedi + 1;
      end
      else
      begin
        ImalatToplami    = ImalatToplami + FKilo;
        ImalatPartiAdedi = ImalatPartiAdedi + 1;
      end
    end

    if (ImalatToplami <> 0) then NumuneKiloImalataKiloOrani = (NumuneToplami / ImalatToplami) * 100;
    if (ImalatPartiAdedi <> 0) then NAdetImalatAdetOrani = (NumunePartiAdedi / ImalatPartiAdedi) * 100;
    if (ImalatToplami <> 0) then ImalatKiloNPartiAdetOrani = (NumunePartiAdedi / ImalatToplami) * 100;

    if ((NumunePartiAdedi + ImalatPartiAdedi) <> 0) then OrtalamaPartiKilosu = (ImalatToplami + NumuneToplami) / (NumunePartiAdedi + ImalatPartiAdedi);

    Select Count(*) from LabReceteCalisma
      left join LabReceteAtis on LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
      where LabReceteCalisma.FirmaId = :FirmaId and LabReceteCalisma.CalismaTarihi >= :ABaslangicTarih and LabReceteCalisma.CalismaTarihi <= :ABitisTarih and
            LabReceteAtis.LabReceteAtisID is not null
      into :RenkCalismaAdedi;

    if (NumunePartiAdedi <> 0 or ImalatPartiAdedi <> 0 or RenkCalismaAdedi <> 0) then suspend;

  end
end
