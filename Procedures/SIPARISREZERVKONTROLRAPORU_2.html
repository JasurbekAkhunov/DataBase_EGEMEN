CREATE PROCEDURE SIPARISREZERVKONTROLRAPORU_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AHAMTIPIID INTEGER,
    AISLEMTIPIID INTEGER,
    ASATISTIPI INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AHAMID INTEGER,
    AREZERVDURUMU INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AKALITEID VARCHAR(20))
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISSARJID INTEGER,
    BOYASIPARISREZERVID INTEGER,
    MIKTAR BOYAMIKTARSIZE,
    KULLANILANMIKTAR BOYAMIKTARSIZE,
    TRANSFERKODU VARCHAR(55),
    INSERTTARIHI TIMESTAMP,
    INSERTKULLANICIID INTEGER,
    INSERTKULLANICIADI ADI40,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(10),
    ONGORUNOTU MEMO)
AS
declare variable HamStokDetayId integer;
declare variable EgemenFirmaID         integer;
begin
  select EgemenFirmaId from Sistem where SistemId = 1 into :EgemenFirmaId;

  for select BoyaSiparisRezerv.BoyaSiparisRezervId, BoyaSiparisRezerv.BoyaSiparisDetayId, BoyaSiparisRezerv.BoyaSiparisSarjId, BoyaSiparisRezerv.BoyaSiparisKayitId,
    BoyaSiparisRezerv.Miktar, BoyaSiparisRezerv.KullanilanMiktar, BoyaSiparisRezerv.InsertTarihi, BoyaSiparisRezerv.InsertKullaniciId, Kullanici.Adi, BoyaSiparisRezerv.HamStokDetayId,
    BoyaSiparisDetayOngoru.OnayDurumu, IIF(BoyaSiparisDetayOngoru.OnayDurumu is Null,  '', IIF(BoyaSiparisDetayOngoru.OnayDurumu = 1,'Kabul', 'Ret')), BoyaSiparisDetayOngoru.OngoruNotu
    from SiparisRezervKontrolRaporu(:ABaslangicTarihi, :ABitisTarihi, :AFirmaId, :AHamTipiId, :AIslemTipiId, :ASatisTipi, :AMusteriSiparisNo, :AOnayDurumu, :APartilenen, :AKapaliDurumu, :ASiparisGrupId, :ASiparisGrup2Id, :AHamId, :ARezervDurumu, :ASiparisiAlanKullaniciId ,:AKaliteId)
    join BoyaSiparisRezerv on SiparisRezervKontrolRaporu.BoyaSiparisDetayId = BoyaSiparisRezerv.BoyaSiparisDetayId
    left join Kullanici on Kullanici.KullaniciId = BoyaSiparisRezerv.InsertKullaniciId
    left join BoyaSiparisDetayOngoru On BoyaSiparisDetayOngoru.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId and BoyaSiparisDetayOngoru.BoyaSiparisDetayId = BoyaSiparisRezerv.BoyaSiparisDetayId
    into :BoyaSiparisRezervId, :BoyaSiparisDetayId, :BoyaSiparisSarjId, :BoyaSiparisKayitId, :Miktar, :KullanilanMiktar, :InsertTarihi, :InsertKullaniciId, :InsertKullaniciAdi, :HamStokDetayId, :OnayDurumu, :OnayDurumuStr, :OngoruNotu do
  begin
    select TransferKodu from BoyaIrsaliyeFaturaTransferKodu(Null, :HamStokDetayId, :EgemenFirmaId) into :TransferKodu;

    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    suspend;
  end
end
