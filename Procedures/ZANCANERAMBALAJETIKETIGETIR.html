CREATE PROCEDURE ZANCANERAMBALAJETIKETIGETIR(
    APARTISINIFID INTEGER,
    AIPLIKPAKETLEMEID INTEGER)
RETURNS (
    DEGER VARCHAR(100),
    SIRASI INTEGER)
AS
declare variable PartiNoSirasi                 integer;
declare variable SiparisNoSirasi               integer;
declare variable FirmaAdiSirasi                integer;
declare variable LabReceteKoduSirasi           integer;
declare variable LabReceteAdiSirasi            integer;
declare variable LabRecetePantoneKoduSirasi    integer;
declare variable ReceteKoduSirasi              integer;
declare variable ReceteAdiSirasi               integer;
declare variable EskiLabReceteKoduSirasi       integer;
declare variable LabMusteriRenkNoSirasi        integer;
declare variable SipMusteriRenkKoduveAdiSirasi integer;
declare variable MusteriSiparisNoSirasi        integer;
declare variable HamAdiSirasi                  integer;
declare variable HamAdi2Sirasi                 integer;
declare variable HamAdi3Sirasi                 integer;
declare variable HamAdi4Sirasi                 integer;
declare variable LotSirasi                     integer;
declare variable BrutKiloSirasi                integer;
declare variable NetKiloSirasi                 integer;
declare variable BobinSayisiSirasi             integer;
declare variable KoliNoSirasi                  integer;
declare variable TarihSirasi                   integer;
declare variable SaatSirasi                    integer;
declare variable IplikCinsiSirasi              integer;
declare variable IplikCinsiDigerAdiSirasi      integer;
declare variable KalinlikSirasi                integer;
declare variable OzellikSirasi                 integer;
declare variable OzellikDigerAdiSirasi         integer;
declare variable ElyafSinifSirasi              integer;
declare variable LabReceteKoduveAdiSirasi      integer;
declare variable BarkodSirasi                  integer;
declare variable KonikCinsiSirasi              integer;
declare variable KonikRengiSirasi              integer;
declare variable FirmaAdimizSirasi             integer;
declare variable FirmaAdres1Sirasi             integer;
declare variable PartiYil         integer;
declare variable PartiId          integer;
declare variable BoyaSiparisYil   integer;
declare variable BoyaSiparisId    integer;
declare variable FirmaAdi         varchar(100);
declare variable LabReceteKodu    varchar(20);
declare variable LabReceteAdi     varchar(40);
declare variable ReceteId         varchar(20);
declare variable ReceteAdi        varchar(40);
declare variable EskiLabReceteKodu    varchar(20);
declare variable LabMusteriRenkKodu   varchar(30);
declare variable MusteriRenkKoduveAdi varchar(30);
declare variable MusteriSiparisNo     varchar(50);
declare variable HamAdi       varchar(255);
declare variable HamAdi2      varchar(255);
declare variable HamAdi3      varchar(255);
declare variable HamAdi4      varchar(255);
declare variable Lot          varchar(100);
declare variable BrutKilo    numeric(12,2);
declare variable NetKilo     numeric(12,2);
declare variable BobinSayisi            integer;
declare variable KoliNo                 integer;
declare variable PSKKEtiketTipiId       integer;
declare variable KalinlikAdi            varchar(50);
declare variable IplikCinsiAdi          varchar(80);
declare variable IplikCinsiDigerAdi     varchar(30);
declare variable OzellikAdi             varchar(40);
declare variable OzellikDigerAdi        varchar(40);
declare variable ElyafSinifAdi          varchar(20);
declare variable KonikCinsiAdi          varchar(20);
declare variable KonikRengiAdi          varchar(20);
declare variable Fnow                   TimeStamp;
declare variable DefaultIplikPsKKEtiketTipiId  integer;
declare variable EgemenFirmaId          integer;
declare variable PantoneKodu            varchar(20);
declare variable FirmaAdimiz            Adi100;
declare variable FirmaAdres1            varchar(40);
declare variable PSBobinSayisi          Double precision;
declare variable BobinTopSayisi         Double precision;
declare variable FirmaKodumuz           varchar(20);
begin
  Select DefaultIplikPsKKEtiketTipiId, EgemenFirmaId, FirmaKodumuz from sistem where sistemID = 1 into :DefaultIplikPsKKEtiketTipiId, :EgemenFirmaId, :FirmaKodumuz;

  select Firma.Adi from Firma where Firma.FirmaId = :FirmaKodumuz into :FirmaAdimiz ;

  if (DefaultIplikPsKKEtiketTipiId = 0) then DefaultIplikPsKKEtiketTipiId = Null;

  select Parti.PartiYil, Parti.PartiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
         Firma.Adi, BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, Parti.ReceteId, Recete.Adi,
         LabRecete.EskiLabReceteKodu, LabRecete.MusteriRenkKodu, BoyaSiparisDetay.MusteriRenkKoduveAdi,
         BoyaSiparisDetay.MusteriSiparisNo, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, BoyaSiparis.PSKKEtiketTipiId,
         Kalinlik.Adi, IplikCinsi.Adi, Ozellik.Adi, IplikCinsi.DigerAdi, Ozellik.DigerAdi, ElyafSinif.Adi, LabRecete.PantoneKodu, PartiStok.BobinSayisi,Firma.adresi1 from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join Ozellik on Ozellik.OzellikId = Ham.OzellikId
    left join ElyafSinif on ElyafSinif.ElyafSinifId = Ham.ElyafSinifId
    where PartiStok.PartiSinifId = :APartiSinifId
    into :PartiYil, :PartiId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaAdi, :LabReceteKodu, :LabReceteAdi, :ReceteId, :ReceteAdi,
         :EskiLabReceteKodu, :LabMusteriRenkKodu, :MusteriRenkKoduveAdi, :MusteriSiparisNo, :HamAdi, :HamAdi2, :HamAdi3, :HamAdi4,
         :PSKKEtiketTipiId, :KalinlikAdi, :IplikCinsiAdi, :OzellikAdi, :IplikCinsiDigerAdi, :OzellikDigerAdi, :ElyafSinifAdi, :PantoneKodu, :PSBobinSayisi, :FirmaAdres1;

  if (PartiId is null) then exit;

  if (PSKKEtiketTipiId is null) then PSKKEtiketTipiId = DefaultIplikPsKKEtiketTipiId;

  if (PSKKEtiketTipiId is null) then exit;

  Select PartiNoSirasi, SiparisNoSirasi, FirmaAdiSirasi, LabReceteKoduSirasi, LabReceteAdiSirasi, ReceteKoduSirasi,
         ReceteAdiSirasi,  EskiLabReceteKoduSirasi, LabMusteriRenkNoSirasi, SipMusteriRenkKoduveAdiSirasi, MusteriSiparisNoSirasi,
         HamAdiSirasi, HamAdi2Sirasi, HamAdi3Sirasi, HamAdi4Sirasi, LotSirasi, BrutKiloSirasi, NetKiloSirasi, BobinSayisiSirasi, KoliNoSirasi, TarihSirasi,
         SaatSirasi, IplikCinsiSirasi, KalinlikSirasi, OzellikSirasi, ElyafSinifSirasi, LabReceteKoduveAdiSirasi,
         BarkodSirasi, KonikCinsiSirasi, KonikRengiSirasi, IplikCinsiDigerAdiSirasi, OzellikDigerAdiSirasi, LabRecetePantoneKoduSirasi,
         FirmaAdimizSirasi, FirmaAdres1Sirasi from PSKKEtiketTipi
    where PSKKEtiketTipi.PSKKEtiketTipiID = :PSKKEtiketTipiId
    into :PartiNoSirasi, :SiparisNoSirasi, :FirmaAdiSirasi, :LabReceteKoduSirasi, :LabReceteAdiSirasi, :ReceteKoduSirasi,
         :ReceteAdiSirasi,  :EskiLabReceteKoduSirasi, :LabMusteriRenkNoSirasi, :SipMusteriRenkKoduveAdiSirasi, :MusteriSiparisNoSirasi,
         :HamAdiSirasi, :HamAdi2Sirasi, :HamAdi3Sirasi, :HamAdi4Sirasi, :LotSirasi, :BrutKiloSirasi, :NetKiloSirasi, :BobinSayisiSirasi, :KoliNoSirasi, :TarihSirasi,
         :SaatSirasi, :IplikCinsiSirasi, :KalinlikSirasi, :OzellikSirasi, :ElyafSinifSirasi, :LabReceteKoduveAdiSirasi,
         :BarkodSirasi, :KonikCinsiSirasi, :KonikRengiSirasi, :IplikCinsiDigerAdiSirasi, :OzellikDigerAdiSirasi, :LabRecetePantoneKoduSirasi,
         :FirmaAdimizSirasi, :FirmaAdres1Sirasi;

  Select BobinEtiketiLot from PartiStokLotBilgisi(:APartiSinifId) into :Lot;

  Select IplikPaketleme.BrutKilo, IplikPaketleme.NetKilo, IplikPaketleme.BobinSayisi, IplikPaketleme.KoliNo,
         KonikCinsi.Adi, KonikRengi.Adi, Ambalaj.BobinTopSayisi from IplikPaketleme
    left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikPaketleme.KonikCinsiId
    left join KonikRengi on KonikRengi.KonikRengiId = IplikPaketleme.KonikRengiId
    left join Ambalaj on Ambalaj.AmbalajId = IplikPaketleme.AmbalajId
    where IplikPaketlemeID = :AIplikPaketlemeID
    into :BrutKilo, :NetKilo, :BobinSayisi, :KoliNo, :KonikCinsiAdi, :KonikRengiAdi, :BobinTopSayisi;

  if (EgemenFirmaId = 4) then
  begin
    Deger = 'GÜLLE';
    Sirasi = -2;
    suspend;

    Deger = 'ENTEGRE';
    Sirasi = -1;
    suspend;
  end

  if (PartiNoSirasi > 0) then
  begin
    Sirasi = PartiNoSirasi;
    if (EgemenFirmaId <> 19) then Deger = PartiYil  || '-' || PartiId;
    if (EgemenFirmaId = 19) then Deger = PartiId;
    suspend;
  end

  if (SiparisNoSirasi > 0) then
  begin
    Sirasi = SiparisNoSirasi;
    Deger = BoyaSiparisYil  || '-' || BoyaSiparisId;
    suspend;
  end

  if (FirmaAdiSirasi > 0) then
  begin
    Sirasi = FirmaAdiSirasi;
    Deger = SubString(FirmaAdi from 1 for 20);
    suspend;
  end

  if (LabReceteKoduSirasi > 0) then
  begin
    Sirasi = LabReceteKoduSirasi;
    Deger = LabReceteKodu;
    suspend;
  end

  if (LabReceteAdiSirasi > 0) then
  begin
    Sirasi = LabReceteAdiSirasi;
    Deger = LabReceteAdi;
    suspend;
  end

  if (LabRecetePantoneKoduSirasi > 0) then
  begin
    Sirasi = LabRecetePantoneKoduSirasi;
    Deger = PantoneKodu;
    suspend;
  end


  if (ReceteKoduSirasi > 0) then
  begin
    Sirasi = ReceteKoduSirasi;
    Deger = ReceteId;
    suspend;
  end

  if (ReceteAdiSirasi > 0) then
  begin
    Sirasi = ReceteAdiSirasi;
    Deger = ReceteAdi;
    suspend;
  end


  if (EskiLabReceteKoduSirasi > 0) then
  begin
    Sirasi = EskiLabReceteKoduSirasi;
    Deger = EskiLabReceteKodu;
    suspend;
  end

  if (LabMusteriRenkNoSirasi > 0) then
  begin
    Sirasi = LabMusteriRenkNoSirasi;
    Deger = labMusteriRenkKodu;
    suspend;
  end

  if (SipMusteriRenkKoduveAdiSirasi > 0) then
  begin
    Sirasi = SipMusteriRenkKoduveAdiSirasi;
    Deger = MusteriRenkKoduveAdi;
    suspend;
  end

  if (MusteriSiparisNoSirasi > 0) then
  begin
    Sirasi = MusteriSiparisNoSirasi;
    Deger = MusteriSiparisNo;
    suspend;
  end

  if (HamAdiSirasi > 0) then
  begin
    Sirasi = HamAdiSirasi;
    Deger = SubString(HamAdi from 1 for 22);
    suspend;
  end

  if (HamAdi2Sirasi > 0) then
  begin
    Sirasi = HamAdi2Sirasi;
    Deger = SubString(HamAdi2 from 1 for 22);
    suspend;
  end

  if (HamAdi3Sirasi > 0) then
  begin
    Sirasi = HamAdi3Sirasi;
    Deger = SubString(HamAdi3 from 1 for 22);
    suspend;
  end

  if (HamAdi4Sirasi > 0) then
  begin
    Sirasi = HamAdi4Sirasi;
    Deger = SubString(HamAdi4 from 1 for 22);
    suspend;
  end

  if (LotSirasi > 0) then
  begin
    Sirasi = LotSirasi;
    if (EgemenFirmaId = 46) then Deger = SubString(Lot from 5 for Char_Length(Lot) - 4); /*Başyazıcı*/
    else Deger = Lot;
    suspend;
  end

  if (BrutKiloSirasi > 0) then
  begin
    Sirasi = BrutKiloSirasi;
    deger = cast(BrutKilo as varchar(20));
    suspend;
  end

  if (NetKiloSirasi > 0) then
  begin
    Sirasi = NetKiloSirasi;
    deger = cast(netKilo as varchar(20));
    suspend;
  end

  if (BobinSayisiSirasi > 0) then
  begin
    Sirasi = BobinSayisiSirasi;
    Deger = EgemenIntegerToStr(BobinSayisi);
    suspend;
  end

  if (KoliNoSirasi > 0) then
  begin
    Sirasi = KoliNoSirasi;
    if (EgemenFirmaID = 75) then
    begin
      Deger = EgemenIntegerToStr(KoliNo) || '/' || EgemenIntegerToStr(Ceil(PSBobinSayisi/BobinTopSayisi));
    end
    else Deger = EgemenIntegerToStr(KoliNo);
    suspend;
  end

  if (TarihSirasi > 0) then
  begin
    Sirasi = TarihSirasi;
    Deger = TimeStampToDateStr(Current_Date);
    suspend;
  end

  if (SaatSirasi > 0) then
  begin
    Sirasi = SaatSirasi;
    Fnow = current_TimeStamp(0);
    Deger = TimeStampToDateTimeStr(FNow);
    Deger = SubString(Deger from 1 for 16);
    suspend;
  end

  if (IplikCinsiSirasi > 0) then
  begin
    Sirasi = IplikCinsiSirasi;
    Deger = IplikCinsiAdi;
    suspend;
  end

  if (KalinlikSirasi > 0) then
  begin
    Sirasi = KalinlikSirasi;
    Deger = KalinlikAdi;
    suspend;
  end

  if (OzellikSirasi > 0) then
  begin
    Sirasi = OzellikSirasi;
    Deger = OzellikAdi;
    suspend;
  end

  if (OzellikDigerAdiSirasi > 0) then
  begin
    if (OzellikDigerAdi is null) then OzellikDigerAdi = '';
    Sirasi = OzellikDigerAdiSirasi;
    Deger = OzellikDigerAdi;
    suspend;
  end

  if (IplikCinsiDigerAdiSirasi > 0) then
  begin
    if (IplikCinsiDigerAdi is null) then IplikCinsiDigerAdi = '';
    Sirasi = IplikCinsiDigerAdiSirasi;
    Deger = IplikCinsiDigerAdi;
    suspend;
  end


  if (ElyafSinifSirasi > 0) then
  begin
    Sirasi = ElyafSinifSirasi;
    Deger = ElyafSinifAdi;
    suspend;
  end

  if (LabReceteKoduveAdiSirasi > 0) then
  begin
    Sirasi = LabReceteKoduveAdiSirasi;
    Deger = LabReceteKodu || '/' || LabReceteAdi;
    suspend;
  end

  if (BarkodSirasi > 0) then
  begin
    Sirasi = BarkodSirasi;
    Deger = 'I-' || EgemenIntegerToStr(AIplikPaketlemeId);
/*    Deger = EgemenIntegerToStr(AIplikPaketlemeId); */
    suspend;
  end

  if (KonikRengiSirasi > 0) then
  begin
    if (KonikRengiAdi is null) then KonikRengiAdi = '';

    Sirasi = KonikRengiSirasi;
    Deger = KonikRengiAdi;
    suspend;
  end

  if (KonikCinsiSirasi > 0) then
  begin
    if (KonikCinsiAdi is null) then KonikCinsiAdi = '';

    Sirasi = KonikCinsiSirasi;
    Deger = KonikCinsiAdi;
    suspend;
  end

  if (BrutKiloSirasi > 0) then
  begin
    Sirasi = 999;
    deger = '0.0';
    suspend;

    Sirasi = 1000;
    deger = '0.0';
    suspend;
  end

  if (FirmaAdimizSirasi > 0) then
  begin
    if (FirmaAdimiz is null) then FirmaAdimiz = '';

    Sirasi = FirmaAdimizSirasi;
    Deger = FirmaAdimiz;
    suspend;
  end

  if (FirmaAdres1Sirasi > 0) then
  begin
    if (FirmaAdres1 is null) then FirmaAdres1 = '';

    Sirasi = FirmaAdres1Sirasi;
    Deger = FirmaAdres1;
    suspend;
  end
end
