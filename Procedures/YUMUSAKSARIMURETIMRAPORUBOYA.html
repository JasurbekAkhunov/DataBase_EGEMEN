CREATE PROCEDURE YUMUSAKSARIMURETIMRAPORUBOYA(
    ATARIH TIMESTAMP)
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    KALINLIKID INTEGER,
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIID INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    LOT VARCHAR(30),
    KONIKCINSIADI VARCHAR(20),
    KONIKRENGIADI VARCHAR(20),
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION)
AS
declare variable ABaslangicTarihi  TimeStamp;
declare variable FirmaKodumuz      varchar(20);
declare variable hstIplik          integer;
begin
/* Boyahaneye sevk edilenleri listeler */
  ABaslangicTarihi = F_StripTime(ATarih);

  Select FirmaKodumuz from Sistem where Sistemid = 1 into :FirmaKodumuz;
  Select hstIplik from HamTipiGetir into :hstIplik;

  For Select IplikIrsaliyeDetay.HamId, Ham.Adi, IplikIrsaliyeDetay.Lot, KonikRengi.Adi, KonikCinsi.Adi,
             Ham.KalinlikId, Kalinlik.Adi, Ham.IplikCinsiId, IplikCinsi.Adi,
             Sum(IplikIrsaliyeDetay.KapAdedi), Sum(IplikIrsaliyeDetay.Miktar)  from IplikIrsaliye
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeID = IplikIrsaliye.IplikIrsaliyeID
  left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
  left join KonikRengi on KonikRengi.KonikRengiId = IplikIrsaliyeDetay.KonikRengiId
  left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikIrsaliyeDetay.KonikCinsiId
  left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
  left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
  where IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and IplikIrsaliye.IrsaliyeTarihi < :ABaslangicTarihi + 1 and
        IplikIrsaliye.FirmaId = :FirmaKodumuz and (Ham.HamTipiID = :hstIplik)
  group by IplikIrsaliyeDetay.HamId, Ham.Adi, IplikIrsaliyeDetay.Lot, KonikRengi.Adi, KonikCinsi.Adi,
           Ham.KalinlikId, Kalinlik.Adi, Ham.IplikCinsiId, IplikCinsi.Adi
  order by Kalinlik.Adi, IplikCinsi.Adi, IplikIrsaliyeDetay.Lot, KonikCinsi.Adi, KonikRengi.Adi
  into :HamId, :HamAdi, :Lot, :KonikRengiAdi, :KonikCinsiAdi, :KalinlikId, :KalinlikAdi, :IplikCinsiId, :IplikCinsiAdi, :KapAdedi, :Miktar do suspend;

end
