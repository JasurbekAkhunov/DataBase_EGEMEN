CREATE PROCEDURE PARTININSINIFINIBOL(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    AMIKTAR NUMERIC(12,3),
    AKULLANICIID INTEGER,
    APSKALITEKONTROLASAMAID INTEGER,
    AMIKTAR2 NUMERIC(12,3),
    ARAFDETAYIDLER VARCHAR(100),
    ARAFMIKTARLARI VARCHAR(100),
    AOTONOMTOPPARCALANIYOR SMALLINT)
AS
declare variable FHamTipiId integer;
declare variable DokumaKumasiKilodanTakipEt integer;
begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;
  Select HamTipiId from PartiStok where PartiStok.PartiSinifID = :APartiSinifId into :FHamTipiId;

  if (FHamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then execute procedure PartininSinifKiloBol(:APartiKayitId, :APartiSinifId, :AMiktar, :AKullaniciId, :APsKaliteKontrolAsamaId, :AMiktar2, :ARafDetayIdler, :ARafMiktarlari, :AOtonomTopParcalaniyor);
  else execute procedure PartininSinifMetreBol(:APartiKayitId, :APartiSinifId, :AMiktar, :AKullaniciId, :APsKaliteKontrolAsamaId, :AMiktar2, :ARafDetayIdler, :ARafMiktarlari, :AOtonomTopParcalaniyor);
end
