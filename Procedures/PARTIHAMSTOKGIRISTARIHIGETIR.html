CREATE PROCEDURE PARTIHAMSTOKGIRISTARIHIGETIR(
    APARTISINIFID INTEGER)
RETURNS (
    HAMSTOKGIRISTARIHI TIMESTAMP)
AS
begin
  Select Max(HamStokIrsaliye.IrsaliyeTarihi) from PartiStok
  left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
  Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
  left Join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
  where PartiStok.PartiSinifId = :APartiSinifId
  into :HamStokGirisTarihi;
  suspend;
end
