CREATE PROCEDURE PARTISTOKONGORUGETIR(
    APARTISINIFID INTEGER)
RETURNS (
    HAMSTOKDETAYID INTEGER,
    ONGORULENDONME VARCHAR(10),
    ONGORULENEN INTEGER,
    ONGORULENGRAMAJ INTEGER,
    ONGORULENENDENCEKMEZLIK DOUBLE PRECISION,
    ONGORULENBOYDANCEKMEZLIK DOUBLE PRECISION,
    ONGORULENRAPORBOYU VARCHAR(100),
    ONGORUNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    YIKAMAONCESIEN DOUBLE PRECISION,
    YIKAMASONRASIEN DOUBLE PRECISION,
    YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    YIKAMASONRASIGRAMAJ DOUBLE PRECISION,
    ENDENCEKMEZLIK DOUBLE PRECISION,
    BOYDANCEKMEZLIK DOUBLE PRECISION,
    HAMDONME DOUBLE PRECISION,
    YIKAMAONCESIRAPORBOYU VARCHAR(100),
    YIKAMASONRASIRAPORBOYU VARCHAR(100),
    NEMORANI DOUBLE PRECISION)
AS
begin
  For Select DISTINCT HamStokDetay.HamStokDetayId, BoyaSiparisDetayOngoru.OngorulenDonme, BoyaSiparisDetayOngoru.OngorulenEn,
             BoyaSiparisDetayOngoru.OngorulenGramaj, BoyaSiparisDetayOngoru.OngorulenEndenCekmezlik,
             BoyaSiparisDetayOngoru.OngorulenBoydanCekmezlik, BoyaSiparisDetayOngoru.OngorulenRaporBoyu, BoyaSiparisDetayOngoru.OngoruNotu,
             HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiGramaj,
             HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
             HamStokMaster.NemOrani from PartiStok
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
    left join BoyaSiparisDetayOngoru on BoyaSiparisDetayOngoru.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId and BoyaSiparisDetayOngoru.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisDetayOngoru.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    where PartiStok.PartiSinifId = :APartiSinifId
    into :HamStokDetayId, :OngorulenDonme, :OngorulenEn, :OngorulenGramaj, :OngorulenEndenCekmezlik, :OngorulenBoydanCekmezlik, :OngorulenRaporBoyu,
         :OngoruNotu,
         :YikamaOncesiEn, :YikamaSonrasiEn, :YikamaOncesiGramaj, :YikamaSonrasiGramaj,
         :EndenCekmezlik, :BoydanCekmezlik, :HamDonme, :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu, :NemOrani do
  begin
    suspend;
  end
end
