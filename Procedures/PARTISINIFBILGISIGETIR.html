CREATE PROCEDURE PARTISINIFBILGISIGETIR(
    APARTISINIFID INTEGER)
RETURNS (
    KAPADEDI INTEGER,
    MIKTAR NUMERIC(12,3),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    BOBINSAYISI INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    TAMIRTIPIID INTEGER,
    DENYE VARCHAR(30))
AS
declare variable HamAdi varchar(255);
declare variable FDenye varchar(30);
begin
  HamAdiToplami = '';
  for select distinct Ham.Adi from PartiStokDetay
  left join Ham on Ham.Hamid = PartiStokDetay.Hamid
  where PartiStokDetay.PartiSinifId = :APartiSinifId into :HamAdi do
  begin
     if (Char_Length(HamAdiToplami) + Char_Length(HamAdi) < 253) then
     begin
       if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set UTF8);
       HamAdiToplami = HamAdiToplami || HamAdi;
     end
  end

  select Sum(Miktar), sum(kilo), sum(Metre), Sum(KapAdedi), Sum(Adet), sum(BobinSayisi) from PartiStokDetay
  where PartiSinifId = :APartiSinifId
  into :Miktar, :Kilo, :Metre, :KapAdedi, :Adet, :BobinSayisi;

  select Max(DisTamir.TamirTipiId) from PartiStokDetay
      left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
      where PartiStokDetay.PartiSinifId = :APartiSinifId
      into :TamirTipiId;

  Denye = '';

  for Select HamStokMaster.Denye from PartiStokDetay
   left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
   left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
   left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
   where PartiStokDetay.PartiSinifID = :APartiSinifId and HamStokMaster.Denye is not null and HamStokMaster.Denye <> 0
   group by HamStokMaster.Denye
   into :FDenye do
  begin
    if (Char_Length(Denye) + Char_Length(FDenye) < 28) then
    begin
      if (Denye <> '') then Denye = Denye || cast(', ' as varchar(2) character set UTF8);
      Denye = Denye || FDenye;
    end
  end

  suspend;
end
