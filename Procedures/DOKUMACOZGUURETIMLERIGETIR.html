CREATE PROCEDURE DOKUMACOZGUURETIMLERIGETIR(
    ADOKUMAPLANLAMAID INTEGER)
RETURNS (
    URETIMPARTINO VARCHAR(30),
    TARIH TIMESTAMP,
    HAMADI ADI255,
    COZGUHAMID INTEGER,
    COZGUHAMADI ADI255,
    ORMERAFID INTEGER,
    RAFID RAFKODUSIZE,
    LEVENTNO INTEGER,
    ISEMRIID VARCHAR(15),
    TOPNO INTEGER,
    KILO BOYAMIKTARSIZE,
    MIKTAR BOYAMIKTARSIZE,
    DOKUMATALIMATLOT VARCHAR(150))
AS
begin

  for select DokumaPlanlama.UretimPartiNo, DokumaKaliteKontrol.Tarih, Ham.Adi HamAdi, OrmeRaf.OrmeRafId, OrmeRaf.RafId, OrmeRaf.LeventNo, CozguIsEmri.IsEmriId,
    DokumaKaliteKontrol.TopNo, DokumaKaliteKontrol.Kilo, DokumaKaliteKontrol.Miktar, DokumaPlanlama.Lot, CozguIsEmri.IplikHamId, IplikHam.Adi
    from DokumaStokPlanlama
    join DokumaKaliteKontrol on DokumaKaliteKontrol.OrmeRafId = DokumaStokPlanlama.OrmeRafId
    left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaKaliteKontrol.DokumaPlanlamaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join OrmeRaf on OrmeRaf.OrmeRafId = DokumaKaliteKontrol.OrmeRafId
    left join CozguIsEmri on CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId
    left join Ham IplikHam on IplikHam.HamId = CozguIsEmri.IplikHamId
    where DokumaStokPlanlama.DokumaPlanlamaId = :ADokumaPlanlamaId
    into :UretimPartiNo, :Tarih, :HamAdi, :OrmeRafId, :RafId, :LeventNo, :IsEmriId, :TopNo, :Kilo, :Miktar, :DokumaTalimatLot, :CozguHamId, :CozguHamAdi do
      suspend;

end
