     OrmeSiparisDetay.HasilMaliyetHamStokBirimId, OrmeSiparisDetay.HasilKdvOrani, Ham.TelSayisi, Ham.SolKenarTelSayisi, Ham.SagKenarTelSayisi,
     OrmeSiparisDetay.DokumaHazirlikMaliyet, OrmeSiparisDetay.DokumaHazirlikMaliyetDovizId, OrmeSiparisDetay.HamEn, OrmeSiparisDetay.BaslamaTarihi, OrmeSiparisDetay.BukumBirimMaliyet, OrmeSiparisDetay.BukumBirimMaliyetDovizId,
     OrmeSiparisDetay.IstenenKumasBoyu, OrmeSiparisDetay.BelgeHamAdi, OrmeSiparisDetay.PlanlananMiktar, OrmeSiparisDetay.ToplamDokunan, OrmeSiparisDetay.ToplamSakatDokunan from OrmeSiparisDetay
  Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
  Left Join Renk ON Renk.RenkId = OrmeSiparisDetay.RenkId
  Left Join HamTipi ON HamTipi.HamTipiId = Ham.HamTipiId
  left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
  left join Uretici on Uretici.UreticiId  = Ham.UreticiId
  Left Join KumasCinsi ON KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
  left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliID = OrmeSiparisDetay.PaketlemeSekliID
  left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
  left join BoyaSiparis KumasTeminSiparis on KumasTeminSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.KTBoyaSiparisKayitId
  Left Join Kullanici IK  ON IK.KullaniciId = OrmeSiparisDetay.InsertKullaniciID
  Left Join Kullanici SK  ON SK.KullaniciId = OrmeSiparisDetay.KullaniciID
  Left Join Kullanici OK  ON OK.KullaniciId = OrmeSiparisDetay.OnaylayanKullaniciId
  left join OdemeSekli on OdemeSekli.OdemeSekliId  = OrmeSiparisDetay.OdemeSekliId
  left join Kalite on Kalite.KaliteId  = OrmeSiparisDetay.KaliteId
  Left Join IplikLot On IplikLot.IplikLotID = OrmeSiparisDetay.IplikLotID
  Left Join KullaniciImza On KullaniciImza.KullaniciID = OrmeSiparisDetay.OnaylayanKullaniciId
  where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  order by OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.SiparisSiraNo
  into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :HamId, :Sinif, :HamAdi, :HamAdi2, :MelanjKodu, :FlamKodu,  :HamTipiId, :HamTipiAdi,
       :RenkId, :RenkAdi, :HamMiktar, :Adet, :SatinAlinacakMiktar, :StoktanKullanilacakMiktar,
       :SatisFiyati, :DovizId, :OrmeBirimMaliyet, :OrmeBirimMaliyetDovizId, :Teslimtarihi, :RevizeTeslimTarihi,
       :VadeGun, :OdemeSekliId, :OdemeSekliAdi, :MakinaUstuGramaj, :DinlenmisGramaj, :IgneSayisi, :MaySayisi, :Pus, :Feine,
       :TupAcikEnMayli, :ArtikelNo, :MusteriSiparisNo, :ReferansSiparisNo, :TahminiFireOrani,
       :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4,
       :Kapali, :KapatmaTarihi, :ToplamOrulen, :ToplamSakatOrulen,  :ToplamSatis, :ToplamBoyaliSatis, :ToplamSatisIade,
       :MusteriTalimati, :Notu, :UretimNotu,  :IplikCinsiAdi, :UreticiAdi, :KalinlikAdi, :KumasCinsiAdi,
       :PaketlemeSekliId, :PaketlemeSekliAdi, :Beyazlik, :KaliteId, :KaliteAdi, :KumasTeminBoyaSiparisYil, :KumasTeminBoyaSiparisId,
       :ResimDosyaAdi1, :ResimDosyayazdir1, :ResimDosyaAdi2, :ResimDosyayazdir2,
       :KPOrmeyeCikanIplikMiktari, :KPOrmeyeCikanIplikBirimMaliyet, :KPOrmeyeCikanLycraMiktari,
       :KPOrmeyeCikanLycraBirimMaliyet, :KPOrmeMiktari, :KPOrmeBirimMaliyet,
       :KPIplikBoyayaCikanBrutMiktar, :KPIplikBoyayaCikanBirimMaliyet, :KPIplikBoyaBrutMiktar,
       :KPIplikBoyaNetMiktar, :KPIplikBoyaBirimMaliyet, :KPKumasBoyayaCikanBrutMiktar,
       :KPKumasBoyayaCikanBirimMaliyet, :KPKumasBoyaBrutMiktar, :KPKumasBoyaNetMiktar,
       :KPKumasBoyaBirimMaliyet, :KPToplamBirimMaliyet, :KPSatisMiktari,
       :KPBrutSatisMiktari, :KPSatisBirimFiyati, :KPDovizId, :Planlanabilir,
       :InsertKullaniciID, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :HamAdi3, :HamAdi4,
       :RaporBoyu, :SevkYeri, :DegisimLog, :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi,
       :IplikLotID, :IplikLotAdi, :HedefSatisKarOrani, :OnaylayanKullaniciImza, :Organik, :DarPus,
       :DokumaBirimMaliyet, :DokumaBirimMaliyetDovizId, :CozguBirimMaliyet, :CozguBirimMaliyetDovizId, :HasilBirimMaliyet, :HasilBirimMaliyetDovizId, :Kilo,
       :HamEnCm, :HamEnInch, :IstenenKumasEni, :CozguIplikCm, :CozguIplikInch, :TarakNumarasi, :TarakDisindekiTelSayisi, :TarakEni, :CozguArtiHasilIsEmri,
       :DokumaMaliyetHamStokBirimId, :DokumaKdvOrani, :CozguMaliyetHamStokBirimId, :CozguKdvOrani, :HasilMaliyetHamStokBirimId, :HasilKdvOrani, :HamTelSayisi, :HamSolKenarTelSayisi, :HamSagKenarTelSayisi,
       :DokumaHazirlikMaliyet, :DokumaHazirlikMaliyetDovizId, :HamEn, :BaslamaTarihi, :BukumBirimMaliyet, :BukumBirimMaliyetDovizId,
       :IstenenKumasBoyu, :BelgeHamAdi, :PlanlananMiktar, :ToplamDokunan, :ToplamSakatDokunan do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    KapaliStr = '';
    if (Kapali = 1) then KapaliStr = 'Evet';

    BeyazlikStr = '';
    if (Beyazlik = 1) then BeyazlikStr = 'Evet';

    OrganikStr = '';
    if (Organik = 1) then OrganikStr = 'Organik';

    PlanlanabilirStr = '';
    if (Planlanabilir = 1) then PlanlanabilirStr = 'Evet';

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    if (KumasTeminBoyaSiparisId is not null) then
      KumasTeminSiparisIdStr = '(' || EgemenintegerToStr(KumasTeminBoyaSiparisYil) || ') ' || EgemenintegerToStr(KumasTeminBoyaSiparisId);
    else
      KumasTeminSiparisIdStr = '';


    BoyayaGidenMiktar = 0;
    Select Sum(StokHareket.GirisCikis * StokHareket.Miktar * -1) from StokHareket
      where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and
            StokHareket.UretimIslemTipiId = :uitBoya and (StokHareket.GirisCikis = -1 or (StokHareket.GirisCikis = 1 and StokHareket.HamIade = 1) )
      into :BoyayaGidenMiktar;


    Select Sum(BoyaIrsaliyeDetay.Kilo) from OrmeSiparisDetay
      left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayID = OrmeSiparisDetay.OrmeSiparisDetayId
      left join PartiStok on PartiStok.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = PartiStok.PartiSinifId
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId and BoyaSiparisDetay.BoyaSiparisDetayID is not null and
            PartiStok.PartiSinifID is not null and BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID is not null and
            BoyaIrsaliye.FirmaID <> :OrmeFirmaKodumuz
      into :FToplamBoyaliSatis;

    if (ToplamBoyaliSatis is null) then ToplamBoyaliSatis = 0;
    if (FToplamBoyaliSatis is null) then FToplamBoyaliSatis = 0;
    ToplamBoyaliSatis = ToplamBoyaliSatis + FToplamBoyaliSatis;

    if (Kapali = 1) then
    begin
      select DovizKuru from EnYakinDovizKuruGetir('USD', :KPDovizId, :KapatmaTarihi) into :KapanisDolarKuru;
      select DovizKuru from EnYakinDovizKuruGetir('EURO', :KPDovizId, :KapatmaTarihi) into :KapanisEuroKuru;
    end
    else
    begin
      KapanisDolarKuru = Null;
      KapanisEuroKuru = Null;
    end


    IplikTalepMiktari = 0;

    Select Sum(Miktar) from IplikTalepDetay where IplikTalepDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId  into :IplikTalepMiktari;

    IplikTalepIdStr = '';

    for Select IpliKTalep.IplikTalepYil, IpliKTalep.IplikTalepId from IplikTalepDetay
      left join IpliKTalep on IpliKTalep.IpliKTalepKayitID = IpliKTalepDetay.IpliKTalepKayitId
      where IplikTalepDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId
      group by IpliKTalep.IplikTalepYil, IpliKTalep.IplikTalepId
      into :FIplikTalepYil, FIplikTalepId do
    begin
      if (IplikTalepIdStr <> '') then IplikTalepIdStr = IplikTalepIdStr || cast(', ' as varchar(2) character set UTF8);

      IplikTalepIdStr = IplikTalepIdStr ||  '(' || EgemenIntegerToStr(FIplikTalepYil) || ') ' || EgemenIntegerToStr(FIplikTalepId);
    end

    FIplikMaliyeti = 0;

    Select Sum(BirimMaliyet * (select DovizKuru from EnYakinDovizKuruGetir(BirimMaliyetDovizID, :SistemDovizID, :SiparisTarihi)) * KullanimOrani/100) from OrmeSiparisDetayRecete
      left join Ham On Ham.HamID = OrmeSiparisDetayRecete.HamId
      where OrmeSiparisDetayRecete.OrmeSiparisDetayId = :OrmeSiparisDetayId and Ham.HamTipiID <> :hstLycra
      into :FIplikMaliyeti;

    if (FIplikMaliyeti is null) then FIplikMaliyeti = 0;
    else FIplikMaliyeti = FIplikMaliyeti + FIplikMaliyeti * TahminiFireOrani / 100;

    Select Sum(BirimMaliyet * (select DovizKuru from EnYakinDovizKuruGetir(BirimMaliyetDovizID, :SistemDovizID, :SiparisTarihi)) * KullanimOrani/100) from OrmeSiparisDetayRecete
      left join Ham On Ham.HamID = OrmeSiparisDetayRecete.HamId
      where OrmeSiparisDetayRecete.OrmeSiparisDetayId = :OrmeSiparisDetayId and Ham.HamTipiID = :hstLycra
      into :FLycraMaliyeti;

    if (FLycraMaliyeti is null) then FLycraMaliyeti = 0;
    else FLycraMaliyeti = FLycraMaliyeti + FLycraMaliyeti * TahminiFireOrani / 100;

    if (DovizId <> SistemDovizId) then
      select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :SistemDovizId, :SiparisTarihi) into :DovizKuru;
    else
      DovizKuru = 1;

    HamMaliyet = FIplikMaliyeti + FLycraMaliyeti + OrmeBirimMaliyet * DovizKuru;

    Ormeciler = '';
    for select Firma.KisaAdi from OrmePlanlama
      left join Firma on Firma.FirmaId = OrmePlanlama.FirmaId
      where OrmePlanlama.OrmeSiparisDetayId = :OrmeSiparisDetayId
      into :FOrmeciAdi do
    begin
      if (Char_Length(Ormeciler) < 210) then
      begin
        if (Ormeciler <> '') then Ormeciler = Ormeciler || cast(', ' as varchar(2) character set UTF8);
        Ormeciler = Ormeciler || FOrmeciAdi;
      end
    end

    select coalesce(sum(StokHareket.Adet), 0), coalesce(sum(iif(StokHareket.Sakat = 1, StokHareket.Adet, 0)), 0) from StokHareket
    where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and
    StokHareket.HamIade = 0 and
    StokHareket.UretimIslemTipiId = :uitOrme and
    (StokHareket.HareketTipi = :shtUretim or StokHareket.HareketTipi = :shtFasonUretim)
    into :ToplamOrulenAdet, :ToplamSakatOrulenAdet;

    suspend;
  end
