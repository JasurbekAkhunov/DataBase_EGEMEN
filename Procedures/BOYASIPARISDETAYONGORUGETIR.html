    HAMKONTROLIPLIKUZUNLUGU3 VARCHAR(40),

    BOYASIPARISID INTEGER,
    BOYASIPARISNOSTR VARCHAR(30))
AS
declare variable AtisNoStr      varchar(20);
begin
  Select BoyaSiparisDetayOngoru.BoyaSiparisDetayId, BoyaSiparisDetayOngoru.HamStokDetayId, HamStokDetay.HamStokIrsaliyeId, HamStokDetay.Fason,
             HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             Ormeci.Adi, HamStokMaster.OrmeciMakinaNo, Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, KumasCinsi.Adi,
             HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaSonrasiEn, BoyaSiparisDetay.EnIstenen,
             HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiGramaj, BoyaSiparisDetay.GramajIstenen,
             HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, BoyaSiparisDetay.CekmezlikEnIstenen, BoyaSiparisDetay.CekmezlikBoyIstenen, BoyaSiparisDetay.YikamaHasligiIstenen,
             HamStokMaster.HamDonme, BoyaSiparisDetay.DonmeIstenen, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
             HamStokMaster.YikamaOncesiCPI, HamStokMaster.YikamaSonrasiCPI, HamStokMaster.YikamaOncesiWPI, HamStokMaster.YikamaSonrasiWPI,
             HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2, HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4,
             BoyaSiparisDetay.IstenenRaporBoyu, HamStokMaster.NemOrani, HamStokDetay.KapAdedi, HamStokDetay.Kilo, HamStokDetay.Metre, HamStokDetay.Adet,
             BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaSiparisDetay.LabReceteKodu, LabReceteAtis.AtisNoStr,
             BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu, HamStokMaster.YikamaHasligiYapildi, HamStokMaster.MerserizeHasligiYapildi,
             HamStokMaster.YikamaHasligiKabul, HamStokMaster.MerserizeHasligiKabul, BoyaSiparisDetay.IslemTipiId,
             CekmezlikStandardi.Adi,
             BoyaSiparisDetay.IlaveIslemler, HamStokmaster.HamStokHareketTipiId, BoyaSiparisDetay.OrmeSiparisDetayId,
             BoyaSiparisDetay.YikamaDereceId, YikamaDerece.Derece, BoyaSiparisDetay.KurutmaSekliId, KurutmaSekli.Adi,
             BoyaSiparisDetay.YikamaHasligiTestiId, YikamaHasligiTesti.Adi,
             HamStokMaster.DemirTestiYapildi, HamStokMaster.DemirTestiKabul, HamStokMaster.HamStokMasterId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId from BoyaSiparisDetayOngoru
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetayOngoru.BoyaSiparisKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisDetayOngoru.BoyaSiparisDetayId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisDetayOngoru.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
    left join Firma on Firma.Firmaid = hamstokIrsaliye.Firmaid
    left join Ormeci on Ormeci.Ormeciid = HamstokIrsaliye.Ormeciid
    left join Ham on Ham.Hamid = HamStokDetay.Hamid
    left join KumasCinsi on KumasCinsi.KumasCinsiid = Ham.KumasCinsiId
    left join CekmezlikStandardi on CekmezlikStandardi.CekmezlikStandardiid = HamStokMaster.CekmezlikStandardiid
    left join YikamaDerece on YikamaDerece.YikamaDereceid = BoyaSiparisDetay.YikamaDereceId
    left join KurutmaSekli on KurutmaSekli.KurutmaSekliid = BoyaSiparisDetay.KurutmaSekliId
    left join YikamaHasligiTesti on YikamaHasligiTesti.YikamaHasligiTestiid = BoyaSiparisDetay.YikamaHasligiTestiId
    where BoyaSiparisDetayOngoru.BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisDetayOngoru.HamStokDetayId = :AHamStokDetayId
    into :BoyaSiparisDetayId, :HamStokDetayId, :HamStokIrsaliyeId, :Fason, :FirmaId, :FirmaAdi, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi,
         :OrmeciAdi, :OrmeciMakinaNo, :HamAdi, :Pus, :Feine, :KumasCinsiAdi, :YikamaOncesiEn, :YikamaSonrasiEn, :EnIstenen,
         :YikamaOncesiGramaj, :YikamaSonrasiGramaj, :GramajIstenen, :EndenCekmezlik, :BoydanCekmezlik, :CekmezlikEnIstenen,
         :CekmezlikBoyIstenen, :YikamaHasligiIstenen, :HamDonme, :DonmeIstenen, :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu,
         :YikamaOncesiCPI, :YikamaSonrasiCPI, :YikamaOncesiWPI, :YikamaSonrasiWPI, :HamKontrolIplikUzunlugu1, :HamKontrolIplikUzunlugu2,
         :HamKontrolIplikUzunlugu3, :HamKontrolIplikUzunlugu4,
         :IstenenRaporBoyu, :NemOrani, :KapAdedi, :Kilo, :Metre, :Adet,
         :SiparisProsesId, :SiparisProsesAdi, :LabReceteKodu, :AtisNoStr, :SiparisNotu, :PlanlamaNotu,
         :YikamaHasligiYapildi, :MerserizeHasligiYapildi, :YikamaHasligiKabul, :MerserizeHasligiKabul, :IslemTipiId,
         :CekmezlikStandardiAdi, :IlaveIslemler, :HamStokHareketTipiId, :OrmeSiparisDetayId,
         :YikamaDereceId, :YikamaDereceAdi, :KurutmaSekliId, :KurutmaSekliAdi, :YikamaHasligiTestiId, :YikamaHasligiTestiAdi,
         :DemirTestiYapildi, :DemirTestiKabul, :HamStokMasterId, :BoyaSiparisYil, :BoyaSiparisId;

    LabReceteKodu = LabReceteKodu || '.' || AtisNoStr;

   if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
   else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
   else EvrakTipiStr = 'Hatalı';

   BoyaSiparisNoStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
   suspend;
end
