CREATE PROCEDURE FASONIRSALIYEDENSIPARISOLUSTUR(
    AHAMSTOKIRSALIYEID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASIPARISPROSESID INTEGER,
    ALABRECETEID INTEGER,
    ALABRECETEATISID INTEGER,
    ARENKDERINLIGIID INTEGER,
    ABIRIMFIYAT DOUBLE PRECISION,
    ADOVIZID VARCHAR(5),
    AODEMESEKLIID INTEGER,
    AVADEGUN INTEGER,
    ATERMINTARIHI TIMESTAMP,
    AKULLANICIID INTEGER)
RETURNS (
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    IPLIKBOYASIPARISYIL INTEGER,
    IPLIKBOYASIPARISID INTEGER)
AS
declare variable BoyaSiparisKayitID  integer;
declare variable BoyaSiparisDetayID  integer;
declare variable HamStokMasterId     integer;
declare variable FirmaId             varchar(20);
declare variable LabReceteKodu       varchar(20);
declare variable RenkAdi             varchar(40);
declare variable IplikVar            integer;
declare variable KumasVar            integer;
declare variable IrsaliyeFason       integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable FirmaSevkAdresiId integer;
declare variable FasonSiparisteOngoruYapilacak integer;
declare variable TamirSayisi  integer;
declare variable NormalSayisi integer;
declare variable FIslemTipiId integer;
begin
  Select Sum(HamStokMaster.Tamir), Sum(Decode(HamStokMaster.Tamir, 1, 0, 0, 1)) from HamStokMaster where HamStokMaster.HamStokIrsaliyeID = :AHamStokIrsaliyeID into :TamirSayisi, :NormalSayisi;
  if (NormalSayisi > 0 and TamirSayisi > 0) then
    Exception KayitHatasi 'Bu İrsaliye Tamir ve Normal Kayıtları Birlikte İçerdiği İçin Sipariş Oluşturulamadı.';

  FIslemTipiId = 0;
  if (TamirSayisi > 0) then
    FIslemTipiId = 3;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  select FasonSiparisteOngoruYapilacak from Sistem into :FasonSiparisteOngoruYapilacak;
  if (FasonSiparisteOngoruYapilacak is Null) then FasonSiparisteOngoruYapilacak = 0;

  Select HamStokIrsaliye.FirmaId, HamStokIrsaliye.Fason from HamStokIrsaliye where HamStokIrsaliye.HamStokIrsaliyeID = :AHamStokIrsaliyeID into :FirmaId, :IrsaliyeFason;

  Select FirmaSevkAdresiId from FirmaSevkAdresi where FirmaSevkAdresi.FirmaID = :FirmaID And FirmaSevkAdresi.DefaultAdres = 1 into :FirmaSevkAdresiId;

  if (IrsaliyeFason = 0) then exception KayitHatasi 'Secilen İrsaliye Fason Değil';

  Select Count(*) from HamStokMaster
    left join Ham on Ham.HamId = HamStokMaster.HamId
    where HamStokMaster.HamStokIrsaliyeID = :AHamStokIrsaliyeID and
          (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef)
    into :IplikVar;

  Select Count(*) from HamStokMaster
    left join Ham on Ham.HamId = HamStokMaster.HamId
    where HamStokMaster.HamStokIrsaliyeID = :AHamStokIrsaliyeID and
          (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef)
    into :KumasVar;

  if (KumasVar > 0) then
  begin
    BoyaSiparisYil = f_Year(Current_date);

    BoyaSiparisId = Null;

    Select Id from BoyaSiparisNumaratoruAl(:BoyaSiparisYil) into :BoyaSiparisId;

    insert into BoyaSiparis(BoyaSiparisYil, BoyaSiparisId, FirmaId, SiparisTarihi, SiparisiAlanKullaniciId, IslemTipiId, Fason,
                            SiparisTipiId, InsertKullaniciId, KullaniciId)
               values(:BoyaSiparisYil, :BoyaSiparisId, :FirmaId, Current_date, :AKullaniciId, :FIslemTipiId, 1, 1, :AKullaniciId, :AKullaniciId);

    Select BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :BoyaSiparisYil and BoyaSiparis.BoyaSiparisID = :BoyaSiparisID into :BoyaSiparisKayitId;

    Select LabRecete.LabReceteKodu, LabRecete.Adi from LabRecete where LabRecete.LabReceteID = :ALabReceteId into :LabReceteKodu, :RenkAdi;


    For Select HamStokMaster.HamStokMasterId from HamStokMaster
      left join Ham on Ham.HamId = HamStokMaster.HamId
      where HamStokMaster.HamStokIrsaliyeID = :AHamStokIrsaliyeID and (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf)
      into :HamStokMasterId do
    begin
      BoyaSiparisDetayId = Gen_ID(BoyaSiparisDetay_Id_gen, 1);

      Insert into BoyaSiparisDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
                                   RenkAdi, SiparisProsesId, SiparisKilo, SiparisMetre, RenkDerinligiId, BirimFiyat, DovizId,
                                   VadeGun, OdemeSekliId, HamId, TerminTarihi, IslemGrubuId, OnayDurumu, OnaylayanKullaniciId,
                                   FirmaSevkAdresiId, FasonBirimFiyatDovizId, BaskiBirimFiyatDovizId, InsertKullaniciId, KullaniciId, OngoruYapilacak)
          Select :BoyaSiparisKayitId, :BoyaSiparisDetayId, HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo, :ALabReceteId, :LabReceteKodu,
                 :ALabReceteAtisId, :RenkAdi, :ASiparisProsesId, HamStokMaster.Kilo, HamStokMaster.Metre, :ARenkDerinligiId,
                 :ABirimFiyat, :ADovizId, :AVadeGun, :AOdemeSekliId, HamStokMaster.HamId, :ATerminTarihi, :AIslemGrubuId,
                 1, :AKullaniciId, :FirmaSevkAdresiId, :ADovizId, :ADovizId, :AKullaniciId, :AKullaniciId, :FasonSiparisteOngoruYapilacak from HamStokMaster
            where HamStokMaster.HamStokMasterID = :HamStokMasterID;

      Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, HamStokDetayId, Kilo, Metre, Adet, KapAdedi)
              Select :BoyaSiparisKayitId, :BoyaSiparisDetayId, HamStokDetay.HamStokDetayId, HamStokDetay.Kilo, HamStokDetay.Metre, HamStokDetay.Adet, HamStokDetay.KapAdedi from HamStokMaster
                left join HamStokDetay on HamStokDetay.HamStokMasterId = HamStokMaster.HamStokMasterId
                where HamStokMaster.HamStokMasterID = :HamStokMasterID and HamStokDetay.HamStokMasterID is not null;
    end
  end

  if (IplikVar > 0) then
  begin
    IplikBoyaSiparisYil = f_Year(Current_date);

    IplikBoyaSiparisId = Null;

    Select Id from BoyaSiparisNumaratoruAl(:IplikBoyaSiparisYil) into :IplikBoyaSiparisId;

    insert into BoyaSiparis(BoyaSiparisYil, BoyaSiparisId, FirmaId, SiparisTarihi, SiparisiAlanKullaniciId, IslemTipiId, Fason,
                            SiparisTipiId, InsertKullaniciId, KullaniciId)
               values(:IplikBoyaSiparisYil, :IplikBoyaSiparisId, :FirmaId, Current_date, :AKullaniciId, :FIslemTipiId, 1, 0, :AKullaniciId, :AKullaniciId);

    Select BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :IplikBoyaSiparisYil and BoyaSiparis.BoyaSiparisID = :IplikBoyaSiparisID into :BoyaSiparisKayitId;

    Select LabRecete.LabReceteKodu, LabRecete.Adi from LabRecete where LabRecete.LabReceteID = :ALabReceteId into :LabReceteKodu, :RenkAdi;


    For Select HamStokMaster.HamStokMasterId from HamStokMaster
      left join Ham on Ham.HamId = HamStokMaster.HamId
      where HamStokMaster.HamStokIrsaliyeID = :AHamStokIrsaliyeID and (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf)
      into :HamStokMasterId do
    begin
      BoyaSiparisDetayId = Gen_ID(BoyaSiparisDetay_Id_gen, 1);

      Insert into BoyaSiparisDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
                                   RenkAdi, SiparisProsesId, SiparisKilo, SiparisMetre, RenkDerinligiId, BirimFiyat, DovizId,
                                   VadeGun, OdemeSekliId, HamId, TerminTarihi, IslemGrubuId, OnayDurumu, OnaylayanKullaniciId,
                                   FirmaSevkAdresiId, FasonBirimFiyatDovizId, BaskiBirimFiyatDovizId, InsertKullaniciId, KullaniciId, OngoruYapilacak)
          Select :BoyaSiparisKayitId, :BoyaSiparisDetayId, HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo, :ALabReceteId, :LabReceteKodu,
                 :ALabReceteAtisId, :RenkAdi, :ASiparisProsesId, HamStokMaster.Kilo, HamStokMaster.Metre, :ARenkDerinligiId,
                 :ABirimFiyat, :ADovizId, :AVadeGun, :AOdemeSekliId, HamStokMaster.HamId, :ATerminTarihi, :AIslemGrubuId,
                 1, :AKullaniciId, :FirmaSevkAdresiId, :ADovizId, :ADovizId, :AKullaniciId, :AKullaniciId, :FasonSiparisteOngoruYapilacak from HamStokMaster
            where HamStokMaster.HamStokMasterID = :HamStokMasterID;
    end
  end

  suspend;
end
