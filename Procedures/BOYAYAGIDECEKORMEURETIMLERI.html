CREATE PROCEDURE BOYAYAGIDECEKORMEURETIMLERI(
    AORMEPLANLAMAID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    ASIPARISFIRMAID FIRMAKODUSIZE,
    ASAKAT INTEGER,
    ARAFFIRMAID FIRMAKODUSIZE,
    ADOKUMAPLANLAMAID INTEGER,
    ADOKUMAHAREKETI EGEBOOLEAN)
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISSIRANO INTEGER,
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    RAFID VARCHAR(10),
    MIKTAR NUMERIC(12,3),
    KAPADEDI INTEGER,
    KALITENO INTEGER,
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    URETIMPARTINO VARCHAR(30),
    URETIMORMEMAKINAID VARCHAR(10),
    URETIMDOKUMAMAKINAID VARCHAR(10),
    URETIMFIRMAID VARCHAR(20),
    URETIMFIRMAADI VARCHAR(100),
    ORMESIPARISSEKLIID INTEGER,
    ORMESIPARISSEKLIADI VARCHAR(15),
    ORMESIPARISTIPIID INTEGER,
    ORMESIPARISTIPIADI VARCHAR(30),
    DOKUMASIPARISTIPIID INTEGER,
    DOKUMASIPARISTIPIADI VARCHAR(30),
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    AYARTOPU INTEGER,
    AYARTOPUSTR VARCHAR(10),
    ADET INTEGER)
AS
declare variable FFason               integer;
declare variable FSiparisFirmaId      varchar(20);
declare variable astBoyaliKumas       integer;
declare variable astIplikBoyaliKumas  integer;
declare variable FirmaKodumuz FirmaKoduSize;
declare variable dstBoyaliDokuma      integer;
declare variable dstIplikBoyaliDokuma integer;
begin
  Select astBoyaliKumas, astIplikBoyaliKumas from OrmeSiparisTipiGetir into :AstBoyaliKumas, :AstIplikBoyaliKumas;
  select dstBoyaliDokuma, dstIplikBoyaliDokuma from DokumaSiparisTipiGetir into :dstBoyaliDokuma, :dstIplikBoyaliDokuma;

  Select FirmaKodumuz from Sistem where Sistem.SistemID = 1 into :FirmaKodumuz;
  if (FirmaKodumuz = ARafFirmaId) then ARafFirmaId = Null;

  For Select OrmeRaf.HamId, Ham.Adi, OrmeRaf.StokGirisNo, OrmeRaf.BoyaSiparisKayitId, OrmeRaf.OrmeSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             BoyaSiparis.FirmaId, SiparisFirma.Adi, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.UretimPartiNo, OrmeRaf.RafId, OrmeRaf.OrmeRafId,
             OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari, OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo,
             OrmeRaf.KaliteNo, OrmeRaf.Lot, OrmeRaf.MelanjKodu, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.SiparisSiraNo, coalesce(OrmeRaf.OrmeMakinaId, OrmeRaf.OrmeciMakinaNo),
             OrmeRaf.DokumaMakinaId, OrmeRaf.UreticiFirmaId, UreticiFirma.Adi,
             BoyaSiparis.OrmeSiparisSekliId, OrmeSiparisSekli.Adi, BoyaSiparis.OrmeSiparisTipiId, OrmeSiparisTipi.Adi,
             OrmeRaf.Sakat, OrmeRaf.AyarTopu, BoyaSiparis.DokumaSiparisTipiId, DokumaSiparisTipi.Adi, OrmeRaf.GirisAdet - OrmeRaf.CikisAdet from OrmeRaf
    left join Raf on Raf.RafId = OrmeRaf.RafId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayid = OrmeRaf.OrmeSiparisDetayid
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
    left join Firma SiparisFirma on SiparisFirma.Firmaid = BoyaSiparis.Firmaid
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join Ham on Ham.Hamid = OrmeRaf.Hamid
    left join Firma UreticiFirma on UreticiFirma.Firmaid = OrmeRaf.UreticiFirmaid
    left join OrmeSiparisSekli On OrmeSiparisSekli.OrmeSiparisSekliId = BoyaSiparis.OrmeSiparisSekliId
    left join OrmeSiparisTipi On OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
    left join DokumaSiparisTipi on DokumaSiparisTipi.DokumaSiparisTipiId = BoyaSiparis.DokumaSiparisTipiId
    where OrmeRaf.StokVar = 1 and
          ((OrmeRaf.OrmePlanlamaId is not null and :ADokumaHareketi = 0 or OrmeRaf.DokumaPlanlamaId is not null and :ADokumaHareketi = 1) and OrmeRaf.OrmeSiparisDetayID is not null) and
          (OrmeRaf.Sakat = :ASakat or :ASakat is null) and
          (OrmeRaf.OrmePlanlamaId = :AOrmePlanlamaId or :AOrmePlanlamaId is null) and
          (OrmeRaf.DokumaPlanlamaId = :ADokumaPlanlamaId or :ADokumaPlanlamaId is null) and
          (OrmeRaf.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          (StrKarsilastir(Raf.FirmaId, :ARafFirmaId) = 1 ) and
          (BoyaSiparis.FirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
          ((:ADokumaHareketi = 0 and (BoyaSiparis.OrmeSiparisTipiID = :AstBoyaliKumas or BoyaSiparis.OrmeSiparisTipiID = :AstIplikBoyaliKumas)) or
          (:ADokumaHareketi = 1 and (BoyaSiparis.DokumaSiparisTipiID = :dstBoyaliDokuma or BoyaSiparis.DokumaSiparisTipiID = :dstIplikBoyaliDokuma)))
    into :HamId, :HamAdi, :StokGirisNo, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi,
         :RenkId, :RenkAdi, :UretimPartiNo, :RafId, :OrmeRafId, :Miktar, :KapAdedi, :MusteriSiparisNo, :ReferansSiparisNo,
         :KaliteNo, :Lot, :MelanjKodu, :ArtikelNo, :SiparisSiraNo, :UretimOrmeMakinaId, :UretimDokumaMakinaId, :UretimFirmaId, :UretimFirmaAdi,
         :OrmeSiparisSekliId, :OrmeSiparisSekliAdi, :OrmeSiparisTipiId, :OrmeSiparisTipiAdi, :Sakat, :AyarTopu, :DokumaSiparisTipiId, :DokumaSiparisTipiAdi, :Adet do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    if (Sakat = 1) then SakatStr = 'Sakat';
    else SakatStr = '';

    if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
    else AyarTopuStr = '';
    suspend;
  end
end
