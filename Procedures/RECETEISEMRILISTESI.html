CREATE PROCEDURE RECETEISEMRILISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ARECETEID VARCHAR(20))
RETURNS (
    RECETEISEMRIYIL INTEGER,
    RECETEISEMRIID INTEGER,
    RECETEISEMRIIDSTR VARCHAR(30),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    TARIH TIMESTAMP,
    BANYOCARPANI DOUBLE PRECISION,
    BANYOCARPANI2 DOUBLE PRECISION,
    BANYOCARPANI3 DOUBLE PRECISION,
    BANYOCARPANI4 DOUBLE PRECISION,
    HAMSTOKFLOTTECARPANI DOUBLE PRECISION,
    FULARBASINCI DOUBLE PRECISION,
    KOYULTMA DOUBLE PRECISION,
    DURULAMASUYU INTEGER,
    BOYASUYU INTEGER,
    BOYAMASEVIYESI INTEGER,
    TUR DOUBLE PRECISION,
    POMPADEGERI INTEGER,
    DUZE INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select ReceteIsEmri.ReceteIsEmriYil, ReceteIsEmri.ReceteIsEmriId, ReceteIsEmri.ReceteId, Recete.Adi, ReceteIsEmri.Tarih,
      ReceteIsEmri.BanyoCarpani, ReceteIsEmri.BanyoCarpani2, ReceteIsEmri.BanyoCarpani3, ReceteIsEmri.BanyoCarpani4,
      ReceteIsEmri.HamStokFlotteCarpani, ReceteIsEmri.FularBasinci, ReceteIsEmri.Koyultma, ReceteIsEmri.DurulamaSuyu, ReceteIsEmri.BoyaSuyu, ReceteIsEmri.Tur,
      ReceteIsEmri.BoyamaSeviyesi, Ik.Adi, ReceteIsEmri.InsertTarihi, Sk.Adi, ReceteIsEmri.DegisimTarihi from ReceteIsEmri
    left join Recete on Recete.ReceteId = ReceteIsEmri.ReceteId
    left join Kullanici IK on IK.KullaniciId = ReceteIsEmri.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId = ReceteIsEmri.KullaniciId
    where ReceteIsEmri.Tarih >= :ABaslangicTarihi and ReceteIsEmri.Tarih <= :ABitisTarihi and
          ( ReceteIsEmri.ReceteId = :AReceteId or :AReceteId is null)
    into :ReceteIsEmriYil, :ReceteIsEmriId, :ReceteId, :ReceteAdi, :Tarih, :BanyoCarpani, :BanyoCarpani2, :BanyoCarpani3, :BanyoCarpani4, :HamStokFlotteCarpani,
         :FularBasinci, :Koyultma, :DurulamaSuyu, :BoyaSuyu, :Tur, :BoyamaSeviyesi,
         :InsertKullaniciAdi, InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    ReceteIsEmriIdStr = '(' || EgemenIntegerToStr(ReceteIsEmriYil) || ') ' || EgemenIntegerToStr(ReceteIsEmriId) || ' ';
    suspend;
  end
end
