CREATE PROCEDURE ORMEPLANLAMABILGISIGETIR(
    AORMEMAKINAID VARCHAR(10),
    AFIRMAID VARCHAR(20))
RETURNS (
    ORMEPLANLAMAID INTEGER,
    TARIH TIMESTAMP,
    URETIMPARTINO VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    SIRA INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    HAMRENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    SAKATURETIMTOPLAMI DOUBLE PRECISION,
    MIKTARGUN DOUBLE PRECISION,
    SURE DOUBLE PRECISION,
    BASLAMATARIHI TIMESTAMP,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    MAKINAUSTUGRAMAJ INTEGER,
    DINLENMISGRAMAJ INTEGER,
    PUS INTEGER,
    FEINE INTEGER,
    TUPACIKENMAYLI INTEGER,
    BEKLEMEDE INTEGER,
    SIPARISKALANMIKTAR DOUBLE PRECISION,
    PLANLANDIGIMAKINALAR VARCHAR(255),
    PLANLANDIGIFASONCULAR VARCHAR(255),
    RECETELOT VARCHAR(255),
    KULLANILANIPLIKLOT VARCHAR(255),
    KALITEADI ADI255,
    ONAYDURUMU SMALLINT,
    KALITEID VARCHAR(20),
    KALITEGRUPID VARCHAR(4),
    KALITEGRUPADI VARCHAR(30),
    KALITEMIKTARGUN BOYAMIKTARSIZE,
    TOPLAMORULEN BOYAMIKTARSIZE)
AS
declare variable KalanMiktar Double precision;
declare variable ToplamSakatUretimToplami  Double Precision;
declare variable ELF                       integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable FOrmeMakinaId  varchar(10);
declare variable FFirmaKisaAdi varchar(10);
declare variable FBitti integer;
declare variable FLot   varchar(250);
begin
  Select shtFasonUretim, ShtUretim from OrmeStokHareketTipiGetir into :shtFasonUretim, :ShtUretim;

  For Select OrmePlanlama.OrmePlanlamaId, OrmePlanlama.Tarih, OrmePlanlama.FirmaId, PlanlamaFirma.Adi, case when OrmeMakina.OrmeMakinaId is null then OrmeMakina.OrmeMakinaId else f_rtrim(OrmeMakina.OrmeMakinaId) end,
         OrmeMakina.Adi, OrmePlanlama.Sira, OrmePlanlama.BoyaSiparisKayitId,
         OrmePlanlama.OrmeSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Ham.Adi, Renk.Adi, OrmePlanlama.PlanlananMiktar,
         OrmePlanlama.GerceklesenMiktar, OrmePlanlama.ToplamSakatOrulen, OrmeSiparisDetay.ToplamSakatOrulen, OrmePlanlama.MiktarGun, OrmePlanlama.BaslamaTarihi, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo,
         OrmePlanlama.UretimPartiNo, BoyaSiparis.FirmaId, SiparisFirma.Adi, OrmePlanlama.Beklemede,
         OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.ArtikelNo, OrmePlanlama.BirimFiyat, OrmePlanlama.DovizId, OrmePlanlama.vadeGun,
         OrmeSiparisDetay.MakinaUstuGramaj, OrmeSiparisDetay.DinlenmisGramaj, OrmeSiparisDetay.Pus,
         OrmeSiparisDetay.Feine, OrmeSiparisDetay.TupAcikEnMayli, OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.ToplamOrulen - OrmeSiparisDetay.SatinAlinacakMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar,
         OrmeSiparisDetay.MelanjKodu, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
         BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi, Kalite.KaliteAdi, OrmePlanlama.OnayDurumu, Kalite.KaliteId, Kalite.GrupId, KG.Adi, Kalite.MiktarGun, OrmeSiparisDetay.ToplamOrulen from OrmePlanlama
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
    left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
    left join Firma PlanlamaFirma on PlanlamaFirma.FirmaId = OrmePlanlama.FirmaId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    left join Kalite on Kalite.KaliteId = OrmeSiparisDetay.KaliteId
    Left Join SiparisGrup KG on KG.SiparisGrupId = Kalite.GrupID
    where (OrmePlanlama.FirmaId = :AFirmaId or :AFirmaId is null) and
          (OrmePlanlama.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and OrmePlanlama.Bitti = 0
          order by OrmePlanlama.OrmeMakinaId, OrmePlanlama.FirmaId, Sira
    into :OrmePlanlamaId, :Tarih, :FirmaId, :FirmaAdi, :OrmeMakinaId, :OrmeMakinaAdi, :Sira, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
         :hamAdi, :RenkAdi, :PlanlananMiktar, :GerceklesenMiktar, :SakatUretimToplami, :ToplamSakatUretimToplami, :MiktarGun, :BaslamaTarihi, :MusteriSiparisNo, :ReferansSiparisNo,
         :UretimPartiNo, :SiparisFirmaId, :SiparisFirmaAdi, :Beklemede,
         :SiparisSiraNo, :ArtikelNo, :BirimFiyat, :DovizId, :vadeGun, :MakinaUstuGramaj, :DinlenmisGramaj, :Pus, :Feine, :TupAcikEnMayli, :SiparisKalanMiktar,
         :MelanjKodu, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi, :KaliteAdi, :OnayDurumu, :KaliteId,
         :KaliteGrupId, :KaliteGrupAdi, :KaliteMiktarGun, :ToplamOrulen do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    KalanMiktar = PlanlananMiktar - GerceklesenMiktar + SakatUretimToplami;
    SiparisKalanMiktar = SiparisKalanMiktar + ToplamSakatUretimToplami;

    if (MiktarGun > 0 and KalanMiktar > 0) then Sure = KalanMiktar / MiktarGun;
    else Sure = 0;

    PlanlandigiMakinalar  = '';
    PlanlandigiFasoncular = '';
    ReceteLot             = '';
    KullanilanIplikLot    = '';

    for Select OrmePlanlama.OrmeMakinaId, Firma.KisaAdi, min(Bitti) from OrmePlanlama
    left join Firma on Firma.FirmaID = OrmePlanlama.FirmaId
    where OrmePlanlama.OrmeSiparisDetayID = :OrmeSiparisDetayID
    group by OrmePlanlama.OrmeMakinaId, Firma.KisaAdi
    order by OrmePlanlama.OrmeMakinaId, Firma.KisaAdi
    into :FOrmeMakinaId, :FFirmaKisaAdi, :FBitti do
    begin
      if (FOrmeMakinaID is not null) then
      begin
        if ( Char_Length(PlanlandigiMakinalar) + Char_Length(FOrmeMakinaID) < 230) then
        begin
          if (PlanlandigiMakinalar <> '') then PlanlandigiMakinalar = PlanlandigiMakinalar || cast(', ' as varchar(2) character set UTF8);
          PlanlandigiMakinalar = PlanlandigiMakinalar || FOrmeMakinaID;

          if (FBitti = 0) then PlanlandigiMakinalar = PlanlandigiMakinalar || cast('(-)' as varchar(3) character set UTF8);
          else PlanlandigiMakinalar = PlanlandigiMakinalar || cast('(+)' as varchar(3) character set UTF8);
        end
      end
      else if (FFirmaKisaAdi is not null) then
      begin
        if (Char_Length(PlanlandigiFasoncular) + Char_Length(FFirmaKisaAdi) < 230) then
        begin
          if (PlanlandigiFasoncular <> '') then PlanlandigiFasoncular = PlanlandigiFasoncular || cast(', ' as varchar(2) character set UTF8);
          PlanlandigiFasoncular = PlanlandigiFasoncular || FFirmaKisaAdi;

          if (FBitti = 0) then PlanlandigiFasoncular = PlanlandigiFasoncular || cast('(-)' as varchar(3) character set UTF8);
          else PlanlandigiFasoncular = PlanlandigiFasoncular || cast('(+)' as varchar(3) character set UTF8);
        end
      end
    end

    ReceteLot             = '';
    KullanilanIplikLot    = '';

    For Select OrmePlanlamaRecete.Lot from OrmePlanlamaRecete
      left join Ham on Ham.HamID = OrmePlanlamaRecete.HamId
      where OrmePlanlamaRecete.OrmePlanlamaID = :OrmePlanlamaID and Ham.HamTipiID = 0
      group by Lot
      into :FLot do
    begin
      if (Char_Length(ReceteLot) + Char_Length(FLot) < 200) then
      begin
       if (ReceteLot <> '' ) then ReceteLot = ReceteLot || cast(', ' as varchar(2) character set UTF8);
       ReceteLot = ReceteLot || FLot;
      end
    end

    For Select OrmeRaf.Lot from StokHareket
      left join Ham on Ham.HamID = StokHareket.HamId
      left join OrmeRaf on OrmeRaf.OrmeRafID = StokHareket.OrmeRafId
      where StokHareket.OrmePlanlamaID = :OrmePlanlamaID and StokHareket.OrmeRafId is not null and StokHareket.GirisCikis = -1
      group by OrmeRaf.Lot
      into :FLot do
    begin
      if (Char_Length(KullanilanIplikLot) + Char_Length(FLot) < 200) then
      begin
       if (KullanilanIplikLot <> '' ) then KullanilanIplikLot = KullanilanIplikLot || cast(', ' as varchar(2) character set UTF8);
       KullanilanIplikLot = KullanilanIplikLot || FLot;
      end
    end

    suspend;
  end
end
