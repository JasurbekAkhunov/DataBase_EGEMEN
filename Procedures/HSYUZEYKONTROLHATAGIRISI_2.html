CREATE PROCEDURE HSYUZEYKONTROLHATAGIRISI_2(
    AHSYUZEYKONTROLHATAMETREID INTEGER,
    AHSYUZEYKONTROLID INTEGER,
    AHAMHATATIPIID INTEGER,
    AMETRE DOUBLE PRECISION,
    ABITIS INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    HSYUZEYKONTROLHATAMETREID INTEGER)
AS
begin
  if (ABitis = 0) then
  begin
    if (AHSYuzeyKontrolHataMetreId is not null) then Update HSYuzeyKontrolHataMetre Set BaslangicMetre = :AMetre where HSYuzeyKontrolHataMetre.HSYuzeyKontrolHataMetreID = :AHSYuzeyKontrolHataMetreID;
    else
    begin
       HSYuzeyKontrolHataMetreId = Gen_Id(HSYuzeyKontrolHataMetre_Id_Gen,1);

      insert into HSYuzeyKontrolHataMetre(HSYuzeyKontrolHataMetreId, HSYuzeyKontrolid, HamHataTipiId, BaslangicMetre, InsertKullaniciId, KullaniciId)
                                  values(:HSYuzeyKontrolHataMetreId, :AHSYuzeyKontrolId, :AHamHataTipiId, :AMetre, :AKullaniciId, :AKullaniciId);
    end
  end
  else Update HSYuzeyKontrolHataMetre Set BitisMetre = :AMetre where HSYuzeyKontrolHataMetre.HSYuzeyKontrolHataMetreID = :AHSYuzeyKontrolHataMetreID;
  suspend;
end
