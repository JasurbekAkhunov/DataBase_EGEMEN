CREATE PROCEDURE IPLIKTELEFISEMRILISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKAPALI INTEGER)
RETURNS (
    IPLIKTELEFISEMRIKAYITID INTEGER,
    IPLIKTELEFISEMRIYIL INTEGER,
    IPLIKTELEFISEMRIID INTEGER,
    IPLIKTELEFISEMRIIDSTR VARCHAR(30),
    TARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KAPALI INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select IplikTelefIsEmri.IplikTelefIsEmriKayitId, IplikTelefIsEmri.IplikTelefIsEmriYil, IplikTelefIsEmri.IplikTelefIsEmriId, IplikTelefIsEmri.Tarihi,
             IplikTelefIsEmri.Notu, IplikTelefIsEmri.Kapali, Ik.Adi, IplikTelefIsEmri.InsertTarihi, SK.Adi, IplikTelefIsEmri.DegisimTarihi from IplikTelefIsEmri
  Left Join Kullanici IK on IK.KullaniciId = IplikTelefIsEmri.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = IplikTelefIsEmri.KullaniciId
  where IplikTelefIsEmri.Tarihi >= :ABaslangicTarihi and IplikTelefIsEmri.Tarihi <= :ABitisTarihi and
        (IplikTelefIsEmri.Kapali = :AKapali or :AKapali is null)
  into :IplikTelefIsEmriKayitId, :IplikTelefIsEmriYil, :IplikTelefIsEmriId, :Tarihi, :Notu, :Kapali,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    IplikTelefIsEmriIdStr = '(' || EgemenIntegerToStr(IplikTelefIsEmriYil) || ') ' || EgemenIntegerToStr(IplikTelefIsEmriId) || ' ';
    suspend;
  end
end
