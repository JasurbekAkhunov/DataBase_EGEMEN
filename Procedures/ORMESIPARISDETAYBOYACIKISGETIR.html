CREATE PROCEDURE ORMESIPARISDETAYBOYACIKISGETIR(
    AORMESIPARISDETAYID INTEGER)
RETURNS (
    TOPLAMBOYAYACIKIS DOUBLE PRECISION,
    TOPLAMIPLIKBOYAYACIKIS DOUBLE PRECISION,
    SATINALINANIPLIK DOUBLE PRECISION,
    IADEEDILENIPLIK DOUBLE PRECISION)
AS
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable shtAlis   integer;
declare variable shtAlisIade integer;

declare variable uitBoya integer;
declare variable hstIplik       integer;
declare variable IadeMiktari    Double Precision;
declare variable IadeIplikMiktari Double Precision;
begin
  Select shtFasonUretimeCikis, shtUretimeCikis, shtFasonUretim, shtUretim, shtAlis, shtAlisIade from OrmeStokHareketTipiGetir
    into :shtFasonUretimeCikis, :shtUretimeCikis, :shtFasonUretim, :shtUretim, :shtAlis, :shtAlisIade;

  select uitBoya from UretimIslemTipiGetir into :uitBoya;
  Select hstIplik from HamTipiGetir into :hstIplik;

  Select Sum(iif(StokHareket.HareketTipi = :shtAlis, StokHareket.Miktar, 0)),
         Sum(iif(StokHareket.HareketTipi = :shtAlisIade, StokHareket.Miktar, 0)) from OrmeSiparisDetay
    left join StokHareket on StokHareket.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
    left join Ham On Ham.HamID = StokHareket.HamId
    where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId and
          StokHareket.StokHareketID is not null and
          Ham.HamTipiID = :hstIplik
    into :SatinAlinanIplik, :IadeEdilenIplik;

  Select Sum(iif(Ham.HamTipiId <> :hstIplik, StokHareket.Miktar,0)),
         Sum(iif(Ham.HamTipiId = :hstIplik, StokHareket.Miktar,0)) from OrmeSiparisDetay
    left join StokHareket on StokHareket.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
    left join Ham On Ham.HamID = StokHareket.HamId
    where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId and
          StokHareket.StokHareketID is not null and UretimIslemTipiId = :uitBoya and
          (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis)
    into :ToplamBoyayaCikis, :ToplamIplikBoyayaCikis;


  Select Sum(iif(Ham.HamTipiId <> :hstIplik, StokHareket.Miktar,0)),
         Sum(iif(Ham.HamTipiId = :hstIplik, StokHareket.Miktar,0)) from OrmeSiparisDetay
    left join StokHareket on StokHareket.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
    left join Ham On Ham.HamID = StokHareket.HamId
    where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId and
          StokHareket.StokHareketID is not null and UretimIslemTipiId = :uitBoya and
          (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and StokHareket.HamIade = 1
    into :IadeMiktari, :IadeIplikMiktari;

  if (IadeMiktari is null) then IadeMiktari = 0;
  if (IadeIplikMiktari is null) then IadeIplikMiktari = 0;

  if (ToplamBoyayaCikis is null) then ToplamBoyayaCikis = 0;
  if (ToplamIplikBoyayaCikis is null) then ToplamIplikBoyayaCikis = 0;

  ToplamBoyayaCikis = ToplamBoyayaCikis - IadeMiktari;
  ToplamIplikBoyayaCikis = ToplamIplikBoyayaCikis - IadeIplikMiktari;
  suspend;
end
