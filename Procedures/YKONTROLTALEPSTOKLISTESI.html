CREATE PROCEDURE YKONTROLTALEPSTOKLISTESI(
    AYUZEYKONTROLTALEPKAYITID INTEGER)
RETURNS (
    YKONTROLCIKISFISIDETAYID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(20),
    FISTARIHI TIMESTAMP,
    GELENMIKTAR NUMERIC(12,3),
    ISEMRIHAZIRLANANMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3))
AS
begin
  for select YKontrolCikisFisiDetay.YKontrolCikisFisiDetayId, YKontrolCikisFisi.FisTarihi, YKontrolCikisFisi.FisYil, YKontrolCikisFisi.FisNo,
             YKontrolCikisFisiDetay.Miktar, YKontrolCikisFisiDetay.IsEmriHazirlananMiktar from YKontrolCikisFisiDetay
  left join YKontrolCikisFisi on YKontrolCikisFisi.YKontrolCikisFisiId = YKontrolCikisFisiDetay.YKontrolCikisFisiId
  where YKontrolCikisFisiDetay.YuzeyKontrolTalepKayitId = :AYuzeyKontrolTalepKayitId
  into :YKontrolCikisFisiDetayId, :FisTarihi, :FisYil, :FisNo, :GelenMiktar, :IsEmriHazirlananMiktar do
  begin
     KalanMiktar = GelenMiktar - IsEmriHazirlananMiktar;
     FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';
     if (KalanMiktar <> 0) then suspend;
  end
end
