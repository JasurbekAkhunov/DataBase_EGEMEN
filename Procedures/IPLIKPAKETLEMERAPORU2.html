CREATE PROCEDURE IPLIKPAKETLEMERAPORU2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    ARECETEID VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    AFASON INTEGER,
    ATARTIKAYDIOLMAYANLAR EGEBOOLEAN)
RETURNS (
    TARIH TIMESTAMP,
    TARIHSTR VARCHAR(25),
    IPLIKPAKETLEMEID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    HAMSTOKSARFFISIDETAYID INTEGER,
    MAKINAID VARCHAR(10),
    PAKETLEMEID INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    KONIKRENGIID INTEGER,
    KONIKRENGIADI VARCHAR(20),
    KALITENO INTEGER,
    KOLINO INTEGER,
    BOBINSAYISI INTEGER,
    DARA NUMERIC(12,3),
    BOBINDARASI NUMERIC(12,3),
    AMBALAJAGIRLIGI DOUBLE PRECISION,
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    PAKETLEMETARIHI TIMESTAMP,
    SEVKDURUMU VARCHAR(25),
    PALETNO VARCHAR(5),
    BARKODID VARCHAR(30),
    KAPADEDI INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable KS integer;
declare variable MinHamStokDetayId integer;
declare variable MaxHamStokDetayId integer;
declare variable SonKontrolAsamaId integer;
declare variable IplikSonKontrolAsamaId integer;
declare variable MaxSira  integer;
begin
  Select SonKontrolAsamaId, IplikSonKontrolAsamaId from Sistem Where SistemId = 1 into :SonKontrolAsamaId, :IplikSonKontrolAsamaId;
  if (ATartiKaydiOlmayanlar = 0) then
  begin
    For Select F_StripTime(PartiStokTarti.Tarih), PartiStokTarti.PartiStokTartiId, IplikPaketleme.IplikPaketlemeId, PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, IplikPaketleme.MakinaId, IplikPaketleme.PaketlemeId,
      PartiStokTarti.KaliteNo, IplikPaketleme.KoliNo, PartiStokTarti.BobinSayisi, PartiStokTarti.Dara, IplikPaketleme.BobinDarasi, (PartiStokTarti.Kilo + PartiStokTarti.Dara), PartiStokTarti.Kilo, IplikPaketleme.PaketlemeTarihi,
      IplikPaketleme.AmbalajId, Ambalaj.Adi, IplikPaketleme.KonikCinsiId, KonikCinsi.Adi, IplikPaketleme.KonikRengiId, KonikRengi.Adi, IplikPaketleme.AmbalajAgirligi,
      PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, IK.Adi, PartiStokTarti.InsertTarihi, SK.Adi, PartiStokTarti.DegisimTarihi, PartiStokTarti.PaletNo, PartiStokTarti.BarkodId,
      PartiStokTarti.KapAdedi
      from PartiStokTarti
    left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    left join PartiStok on PartiStok.PartiSinifID = PartiStokTarti.PartiSinifId
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    Left Join Ambalaj on Ambalaj.AmbalajId = IplikPaketleme.AmbalajId
    Left Join KonikCinsi on KonikCinsi.KonikCinsiId = IplikPaketleme.KonikCinsiId
    Left Join KonikRengi on KonikRengi.KonikRengiId = IplikPaketleme.KonikRengiId
    Left Join Kullanici IK on IK.KullaniciId = IplikPaketleme.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = IplikPaketleme.KullaniciId
    where PartiStokTarti.Tarih >= :ABaslangicTarihi and PartiStokTarti.Tarih < :ABitisTarihi and
          (Parti.PartiTipiID = 0) and
          (PartiStok.HamID = :AHamID or :AHamId is null) and
          (Parti.FirmaID = :AFirmaID or :AFirmaID is null) and
          (Parti.Fason = :AFason or :AFason is null) and
          (Parti.ReceteID = :AReceteID or :AReceteID is null) and
          (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
      (PartiStokTarti.IadeBoyaIrsaliyeDetayId is null)
    into :Tarih, :PartiStokTartiId, :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :MakinaId, :PaketlemeId, :KaliteNo, :KoliNo, :BobinSayisi, :Dara, :BobinDarasi, :BrutKilo, :NetKilo, :PaketlemeTarihi,
       :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :KonikRengiId, :KonikRengiAdi, :AmbalajAgirligi,
       :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :PaletNo, :BarkodId, :KapAdedi do
    begin
      TarihStr = cast(Tarih as varchar(25));

      if (BoyaIrsaliyeDetayId is not null) then SevkDurumu = 'Sevk Edilmiş';
      else if (HamStokSarfFisiDetayId is not null) then SevkDurumu = 'Sarf Fişi ile Çıkış';
      else SevkDurumu = '';

      if (BarkodId is null) then
        if (IplikPaketlemeId is not null) then
          BarkodId = 'I-'||IplikPaketlemeId;

      suspend;
    end
  end
  else /*iplik paketlemesi olup tarti kaydi olmayanlar*/
  begin
    For Select F_StripTime(IplikPaketleme.PaketlemeTarihi), IplikPaketleme.IplikPaketlemeId, IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, IplikPaketleme.MakinaId, IplikPaketleme.PaketlemeId,
      IplikPaketleme.KaliteNo, IplikPaketleme.KoliNo, IplikPaketleme.BobinSayisi, IplikPaketleme.Dara, IplikPaketleme.BobinDarasi, (IplikPaketleme.NetKilo + IplikPaketleme.Dara), IplikPaketleme.NetKilo, IplikPaketleme.PaketlemeTarihi,
      IplikPaketleme.AmbalajId, Ambalaj.Adi, IplikPaketleme.KonikCinsiId, KonikCinsi.Adi, IplikPaketleme.KonikRengiId, KonikRengi.Adi, IplikPaketleme.AmbalajAgirligi,
      IplikPaketleme.PartiStokTartiId, PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, IK.Adi, IplikPaketleme.InsertTarihi, SK.Adi, IplikPaketleme.DegisimTarihi
      from IplikPaketleme
    Left Join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiId
    left join PartiStok on PartiStok.PartiSinifID = IplikPaketleme.PartiSinifId
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    Left Join Ambalaj on Ambalaj.AmbalajId = IplikPaketleme.AmbalajId
    Left Join KonikCinsi on KonikCinsi.KonikCinsiId = IplikPaketleme.KonikCinsiId
    Left Join KonikRengi on KonikRengi.KonikRengiId = IplikPaketleme.KonikRengiId
    Left Join Kullanici IK on IK.KullaniciId = IplikPaketleme.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = IplikPaketleme.KullaniciId
    where IplikPaketleme.PartiStokTartiId is null and
          IplikPaketleme.PaketlemeTarihi >= :ABaslangicTarihi and IplikPaketleme.PaketlemeTarihi < :ABitisTarihi and
          (Parti.PartiTipiID = 0) and
          (PartiStok.HamID = :AHamID or :AHamId is null) and
          (Parti.FirmaID = :AFirmaID or :AFirmaID is null) and
          (Parti.Fason = :AFason or :AFason is null) and
          (Parti.ReceteID = :AReceteID or :AReceteID is null) and
          (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null)
    into :Tarih, :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :MakinaId, :PaketlemeId, :KaliteNo, :KoliNo, :BobinSayisi, :Dara, :BobinDarasi, :BrutKilo, :NetKilo, :PaketlemeTarihi,
       :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :KonikRengiId, :KonikRengiAdi, :AmbalajAgirligi,
       :PartiStokTartiId, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
    begin
      TarihStr = cast(Tarih as varchar(25));

      if (BoyaIrsaliyeDetayId is not null) then SevkDurumu = 'Sevk Edilmiş';
      else if (HamStokSarfFisiDetayId is not null) then SevkDurumu = 'Sarf Fişi ile Çıkış';
      else SevkDurumu = '';

      BarkodId = 'I-'||IplikPaketlemeId;

      suspend;
    end
  end
end
