CREATE PROCEDURE PARTIVERIRAPORU_5(
    APARTIYIL INTEGER,
    APARTIID INTEGER)
RETURNS (
    MAKINAID VARCHAR(10),
    MAKINABILGIID INTEGER,
    MAKINAKAMARABILGIID INTEGER,
    KAMARANO INTEGER,
    SAGISIDEGERI DOUBLE PRECISION,
    SAGISIDEGERISET DOUBLE PRECISION,
    SOLISIDEGERI DOUBLE PRECISION,
    SOLISIDEGERISET DOUBLE PRECISION,
    ALTFANCALISMAYUZDESI DOUBLE PRECISION,
    USTFANCALISMAYUZDESI DOUBLE PRECISION,
    ENDEGERI DOUBLE PRECISION)
AS
declare variable GirisTarihi  TimeStamp;
declare variable CikisTarihi  TimeStamp;
begin
  for Select MakinaId, GirisTarihi, CikisTarihi from PartiVeriRaporu_1(:APartiYil, :APartiId) order by GirisTarihi into :MakinaId, :GirisTarihi, :CikisTarihi do
  begin
    for Select MakinaKamaraBilgi.MakinaBilgiId, MakinaKamaraBilgi.MakinaKamaraBilgiID, MakinaKamaraBilgi.KamaraNo, MakinaKamaraBilgi.SagIsiDegeri,
               MakinaKamaraBilgi.SagIsiDegeriSet, MakinaKamaraBilgi.SolIsiDegeri, MakinaKamaraBilgi.SolIsiDegeriSet, MakinaKamaraBilgi.AltFanCalismaYuzdesi,
               MakinaKamaraBilgi.UstFanCalismaYuzdesi, MakinaKamaraBilgi.EnDegeri from MakinaBilgi
      left Join MakinaKamaraBilgi on MakinaKamaraBilgi.MakinaBilgiID = MakinaBilgi.MakinaBilgiID
      where MakinaBilgi.Tarih >= :GirisTarihi and MakinaBilgi.Tarih <= :CikisTarihi and
            MakinaBilgi.MakinaId = :MakinaId and MakinaKamaraBilgi.MakinaBilgiID is not Null
      into :MakinaBilgiId, :MakinaKamaraBilgiID, :KamaraNo, :SagIsiDegeri, :SagIsiDegeriSet, :SolIsiDegeri, :SolIsiDegeriSet, :AltFanCalismaYuzdesi,
           :UstFanCalismaYuzdesi, :EnDegeri do
    begin
      suspend;
    end
  end
end
