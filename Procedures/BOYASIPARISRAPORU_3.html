CREATE PROCEDURE BOYASIPARISRAPORU_3(
    AIMALATTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AFIRMAYETKILIID INTEGER,
    AISLEMTIPIID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AMUSTERIKALITENO VARCHAR(20),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AHAMID INTEGER,
    AKULLANICIID INTEGER,
    AKALITEID VARCHAR(20),
    ATALIMATKODU VARCHAR(20),
    AMAKINAID VARCHAR(10),
    ATEKRARURETIM INTEGER,
    ADESENID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AMARKALI INTEGER,
    AKOMISYONLU INTEGER,
    ALAMINASYON INTEGER,
    ARECETETURUID INTEGER,
    ABATCHNO VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ABASKITIPIID INTEGER,
    AHAMGRUP1ID VARCHAR(4),
    AHAMGRUP2ID VARCHAR(4),
    AHAMGRUP3ID VARCHAR(4),
    AHAMGRUP4ID VARCHAR(4),
    ASIPARISRECETEDURUMU INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    ISLEMGRUBUADI VARCHAR(20),
    ISLEMTIPIADI VARCHAR(20),
    ACILIYETDERECESIADI VARCHAR(20),
    PLANLANANIMALATTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    FASON INTEGER,
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    NOTU VARCHAR(1024),
    NUMUNEONAYTARIHI DATE,
    NUMUNENOTU VARCHAR(100),
    NUMUNEGONDERITARIHI TIMESTAMP,
    TASARIMNO VARCHAR(20),
    VARYANTACIKLAMA VARCHAR(255))
AS
declare variable FNotu  Memo;
begin
  for select DISTINCT  BoyaSiparisRaporu_2.BoyaSiparisDetayId, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.ReceteId,
             Recete.Adi, IslemGrubu.Adi, IslemTipi.Adi, AciliyetDerecesi.Adi, Parti.PlanlananImalatTarihi, Parti.TerminTarihi, Parti.Fason,
             Parti.Notu, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, Parti.PartiGrupId, PartiGrup.Adi, Parti.NumuneOnayTarihi,
             Parti.numunenotu, Parti.NumuneGonderiTarihi, Desen.TasarimNo, DesenVaryant.Aciklama
             from BoyaSiparisRaporu_2(:AImalatTipiId, :ABaslangicTarihi, :ABitisTarihi, :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AFirmaYetkiliId, :AIslemTipiId, :AIslemGrubuId, :ASatisTipi, :ABoyaSiparisTipiId, :ALabReceteId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMusteriKaliteNo, :AOnayDurumu, :APartilenen, :AKapaliDurumu, 0, 0, :ASiparisiAlanKullaniciId, :AYurtDisi, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AHamId, :AKullaniciId, :AKaliteId, :ATalimatKodu, :AMakinaId, :ATekrarUretim, :ADesenID, :ATemsilciId, :AMarkali, :AKomisyonlu, :ALaminasyon, :AReceteTuruId, :ABatchNo, 1, :ASertifikaNo, :ABaskiTipiId, :AHamGrup1Id, :AHamGrup2Id, :AHamGrup3Id, :AHamGrup4Id, :ASiparisReceteDurumu)
    left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisRaporu_2.BoyaSiparisDetayId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    where Parti.PartiKayitId is not null
    into :BoyaSiparisDetayId, :PartiKayitId, :PartiYil, :PartiId, :AcilisTarihi, :ReceteId, :ReceteAdi, :IslemGrubuAdi, :IslemTipiAdi,
         :AciliyetDerecesiAdi, :PlanlananImalatTarihi, :TerminTarihi, :Fason, :FNotu, :DesenId, :DesenAdi, :VaryantAdi, :PartiGrupId,
         :PartiGrupAdi, :NumuneOnayTarihi, :NumuneNotu, :NumuneGonderiTarihi, :TasarimNo, :VaryantAciklama do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (char_length(FNotu) > 1023) then Notu = F_BlobLeft(FNotu, 1023); --F_BlobLeft Türkçe Karakter Problemi için kullanıldı
    else if (char_length(FNotu) > 0) then Notu = F_BlobLeft(FNotu, char_length(FNotu));
    else Notu = '';

    suspend;
  end
end
