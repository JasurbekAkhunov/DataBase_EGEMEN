CREATE PROCEDURE CTARIHINEGORECSRAPORUAYTOPLAMI(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ACS VARCHAR(1),
    ACIROFIRMAID VARCHAR(20),
    ABASLANGICCSGRUP1ID VARCHAR(4),
    ABITISCSGRUP1ID VARCHAR(4),
    ABASLANGICCSGRUP2ID VARCHAR(4),
    ABITISCSGRUP2ID VARCHAR(4),
    ABASLANGICCSGRUP3ID VARCHAR(4),
    ABITISCSGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    ACSGRUP1ARALIKLI INTEGER,
    ACSGRUP2ARALIKLI INTEGER,
    ACSGRUP3ARALIKLI INTEGER,
    ADOVIZID VARCHAR(5),
    ARAPORDOVIZID VARCHAR(5),
    ASINIF VARCHAR(1))
RETURNS (
    RAPORAY TIMESTAMP,
    CIROEDILEN DOUBLE PRECISION)
AS
declare variable LastCiroay     timestamp;
declare variable FCiroay        timestamp;
declare variable FMeblag        double precision;
begin
  LastCiroAy = Null;
  For select CiroTarihiAyToplam, Sum(DovizMeblag) from CiroTarihineGoreCSRaporu(:ABaslangicFirmaId, :ABitisFirmaId, :ABaslangicTarihi, :ABitisTarihi, :ACs,
           :ACiroFirmaId, :ABaslangicCsGrup1Id, :ABitisCsGrup1Id,
           :ABaslangicCsGrup2Id, :ABitisCsGrup2Id, :ABaslangicCsGrup3Id, :ABitisCsGrup3Id, :AFirmaAralikli, :ACSGrup1Aralikli,
           :ACSGrup2Aralikli, :ACSGrup3Aralikli, :ADovizId, :ARaporDovizId, :ASinif)
    group by CiroTarihiAyToplam
    into :FCiroAy, :FMeblag do
  begin
    if (LastCiroAy is null or LastCiroay <> FCiroay) then
    begin
      if (LastCiroay is not null) then suspend;
      LastCiroay = FCiroay;
      RaporAy = FCiroay;
      CiroEdilen = 0;
    end
    CiroEdilen = CiroEdilen + FMeblag;
  end
  if (LastCiroay is not null) then suspend;
end
