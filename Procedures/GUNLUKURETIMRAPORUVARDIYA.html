CREATE PROCEDURE GUNLUKURETIMRAPORUVARDIYA(
    AMAKINAGRUBU VARCHAR(4),
    AMAKINARAPORGRUPID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    VARDIYANO INTEGER,
    BEYAZ DOUBLE PRECISION,
    RENKLI DOUBLE PRECISION,
    YIKAMA DOUBLE PRECISION,
    SEKERKASAR DOUBLE PRECISION,
    KURUBOLUM DOUBLE PRECISION,
    CIFTBOYAMA DOUBLE PRECISION,
    NAYLONBOYAMA DOUBLE PRECISION,
    SIYAH DOUBLE PRECISION,
    POLYESTERBOYAMA DOUBLE PRECISION,
    DIGER DOUBLE PRECISION,
    URETIMTOPLAMI DOUBLE PRECISION,
    TOPLAMURETIMEORANI DOUBLE PRECISION)
AS
declare variable VardiyaSayisi      integer;
declare variable Vardiya           integer;
declare variable T1                Double Precision;
declare variable FIslemGrubuId integer;
declare variable FKilo        Double Precision;
declare variable FCikisTarihi TimeStamp;
declare variable RaporToplami Double Precision;
begin
  Select VardiyaSayisi from sistem where sistemid = 1 into :VardiyaSayisi;

  if (VardiyaSayisi = 0 or VardiyaSayisi is null) then VardiyaSayisi = 3;

  VardiyaNo = -1;

  select Sum(PartiAsamaStokDetay.kilo) from PartiAsamaStokDetay
  left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
  left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
  where PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi <= :ABitisTarihi and
        (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
        (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null)
  into :RaporToplami;

  if (RaporToplami is null) then RaporToplami = 0;

  for select parti.IslemGrubuId, PartiAsamaStokDetay.kilo, PartiAsamaStokDetay.Cikistarihi from PartiAsamaStokDetay
  left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
  left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
  where PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi <= :ABitisTarihi and
        (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
        (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null)
  order by PartiAsamaStokDetay.Cikistarihi
  into :FIslemGrubuId, :FKilo, :FCikistarihi do
  begin
    T1 = FCikisTarihi - ABaslangicTarihi;
    Vardiya = VardiyaHesapla(T1, VardiyaSayisi);

    if (Vardiya <> VardiyaNo) then
    begin
      if (VardiyaNo <> -1) then
      begin
        ToplamUretimeOrani = UretimToplami / RaporToplami * 100;
        suspend;
      end
      VardiyaNo = Vardiya;
      Beyaz               = 0;
      Renkli              = 0;
      Yikama              = 0;
      SekerKasar          = 0;
      KuruBolum           = 0;
      CiftBoyama          = 0;
      Siyah               = 0;
      NaylonBoyama        = 0;
      PolyesterBoyama     = 0;
      Diger               = 0;
      UretimToplami       = 0;
      ToplamUretimeOrani  = 0;
    end

    if (FIslemGrubuId = 0) then Beyaz = Beyaz + FKilo;
    else if (FIslemGrubuId = 1) then Renkli = Renkli + FKilo;
    else if (FIslemGrubuId = 2) then Yikama = Yikama + FKilo;
    else if (FIslemGrubuId = 3) then SekerKasar = SekerKasar + FKilo;
    else if (FIslemGrubuId = 4) then KuruBolum = KuruBolum + FKilo;
    else if (FIslemGrubuId = 5) then CiftBoyama = CiftBoyama + FKilo;
    else if (FIslemGrubuId = 6) then Siyah = Siyah + FKilo;
    else if (FIslemGrubuId = 8) then NaylonBoyama = NaylonBoyama + FKilo;
    else if (FIslemGrubuId = 20) then PolyesterBoyama = PolyesterBoyama + FKilo;
    else Diger = Diger + FKilo;

    UretimToplami = UretimToplami + FKilo;
  end

  if (VardiyaNo <> -1) then
  begin
    ToplamUretimeOrani = UretimToplami / RaporToplami * 100;
    suspend;
  end
end
