CREATE PROCEDURE RECETEILEBOYANANSONBESPARTI(
    ARECETEID VARCHAR(20))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    URETIMTARIHI TIMESTAMP,
    KILO NUMERIC(12,3),
    HAMSTOKMASTERIDSTR VARCHAR(50),
    GIRISLOT VARCHAR(150))
AS
declare variable FLot Varchar(150);
declare variable FHamStokMasterID integer;
declare variable Str varchar(30);

begin
  For Select First 5 Parti.PartiKayitID, Parti.PartiYil, Parti.PartiID, Parti.FirmaId, Firma.Adi, Sum(PartiAsamaStok.GirisKilo), Max(PartiAsamaStok.CikisTarihi) from PartiAsamalari
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
    left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaID
    where PartiAsamalari.ReceteID = :AReceteID and PartiAsamaStok.CikisTarihi is not null and PartiAsamaStok.AsamayaDahil = 1 and
          PartiAsamalari.OtomasyonaDahilmi = 1
    group by Parti.PartiKayitID, Parti.PartiYil, Parti.PartiID, Parti.FirmaId, Firma.Adi
    order by 6 desc
    into :PartiKayitID, :PartiYil, :PartiID, :FirmaId, :FirmaAdi, :Kilo, :UretimTarihi do
  begin
    for Select HamStokMaster.Lot, HamStokDetay.HamStokMasterID from PartiStokDetay
      left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
      where PartiStokDetay.PartiKayitId = :PartiKayitID
      Group By HamStokMaster.Lot, HamStokDetay.HamStokMasterID
      into :FLot, :FHamStokMasterID do
    begin
      HamStokMasterIDStr = '';
      GirisLot = '';

      Str = EgemenIntegerToStr(FHamStokMasterID);
      if (Str is not null) then
      if (PosBul(HamStokMasterIDStr, Str) = 0) then
      if (Char_Length(HamStokMasterIDStr) + Char_Length(Str) < 47) then
      begin
        if (HamStokMasterIDStr <> '') then HamStokMasterIDStr = HamStokMasterIDStr || cast(', ' as varchar(2) character set UTF8);
        HamStokMasterIDStr = HamStokMasterIDStr || Str;
      end

      if (FLot is not null) then
      if (PosBul(GirisLot, FLot) = 0) then
      if (Char_Length(GirisLot) + Char_Length(FLot) < 148) then
      begin
        if (GirisLot <> '' ) then GirisLot = GirisLot || cast(', ' as varchar(2) character set UTF8);
        GirisLot = GirisLot || FLot;
      end
    end
    suspend;
  end
end
