CREATE PROCEDURE DOKUMAMAKINAURETIMBILGISIOKU(
    ADOKUMAMAKINAID VARCHAR(10))
RETURNS (
    DOKUMAMAKINAID VARCHAR(10),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISKAYITID INTEGER,
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ADI VARCHAR(30),
    DOKUMAPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    HAMADI VARCHAR(255),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    MIKTARGUN DOUBLE PRECISION,
    MAYLI INTEGER,
    ACIKEN INTEGER,
    ORMESIPARISTIPIID INTEGER,
    ORMESIPARISSEKLIID INTEGER,
    GERCEKLESENTURSAYISI INTEGER,
    GERCEKLESENTOPKESME INTEGER,
    GERCEKLESENTEMIZLIKYAPILDI INTEGER,
    PLANLAMABIRIMFIYAT DOUBLE PRECISION,
    PLANLAMADOVIZID DOVIZSIZE,
    PLANLANANTURSAYISI INTEGER,
    PERSONELTOPLAMTURSAYISI INTEGER,
    SONDOKUMAKALITEKONTROLID INTEGER,
    SONDOKUMAKKTARIH TIMESTAMP)
AS
declare variable MinSira            integer;
declare variable TupAcikEnMayli     integer;
declare variable OrmeDefaultUniteId integer;
begin
  select Sistem.OrmeDefaultUniteId from Sistem into :OrmeDefaultUniteId;

  for Select DokumaMakina.DokumaMakinaId, min(Sira) from DokumaMakina
    left join DokumaPlanlama on DokumaPlanlama.DokumaMakinaId = DokumaMakina.DokumaMakinaId and Bitti = 0 and Beklemede = 0
    where DokumaMakina.OtomasyonaDahil = 1 and (DokumaMakina.DokumaMakinaId = :ADokumaMakinaId or :ADokumaMakinaId is null)
    Group By DokumaMakina.DokumaMakinaId into :DokumaMakinaId, :MinSira do
  begin
    select OrmeVardiyaPersonel.PersonelId, Personel.Adi from DokumaVardiyaHesapla(current_timestamp)
    left join OrmeVardiya on OrmeVardiya.Tarih = DokumaVardiyaHesapla.VardiyaTarihi and OrmeVardiya.VardiyaNo = DokumaVardiyaHesapla.VardiyaNo
                             and OrmeVardiya.UniteId = :OrmeDefaultUniteId and OrmeVardiya.DokumaVardiyasi = 1
    left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = OrmeVardiya.OrmeVardiyaId
    left join Personel on Personel.PersonelId = OrmeVardiyaPersonel.PersonelId
    where OrmeVardiyaPersonel.DokumaMakinaId = :DokumaMakinaId into :PersonelId, :PersonelAdi;


    if (MinSira is null) then
    begin
      BoyaSiparisYil      = Null;
      BoyaSiparisId       = Null;
      BoyaSiparisIdStr    = Null;
      BoyaSiparisKayitId  = Null;
      SiparisFirmaId      = Null;
      SiparisFirmaAdi     = Null;
      DokumaPlanlamaId    = Null;
      UretimPartiNo       = Null;
      HamAdi              = Null;
      PlanlananMiktar     = Null;
      GerceklesenMiktar   = Null;
      KalanMiktar         = Null;
      MiktarGun           = Null;
      Mayli               = Null;
      AcikEn              = Null;
      OrmeSiparisTipiId   = Null;
      OrmeSiparisSekliId  = Null;
      SiparisGrupAdi      = Null;
      SiparisGrup2Adi      = Null;
      PlanlananTurSayisi      = Null;
      GerceklesenTurSayisi    = Null;
      GerceklesenTopKesme     = Null;
      GerceklesenTemizlikYapildi = Null;
      PersonelToplamTurSayisi = Null;
      PlanlamaBirimFiyat      = Null;
      PlanlamaDovizID         = Null;
      SonDokumaKaliteKontrolId = Null;
    end
    else
    begin
      Select first 1 BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.FirmaId, Firma.Adi, SiparisGrup.Adi, SiparisGrup2.Adi, DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlama.UretimPartiNo,
             Ham.Adi, DokumaPlanlama.PlanlananMiktar,
             DokumaPlanlama.GerceklesenMiktar, DokumaPlanlama.MiktarGun, BoyaSiparis.OrmeSiparisTipiId, BoyaSiparis.OrmeSiparisSekliId,
             OrmeSiparisDetay.TupAcikEnMayli, DokumaPlanlama.PlanlananTurSayisi, DokumaPlanlama.DovizID, DokumaPlanlama.BirimFiyat from DokumaPlanlama
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
        left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
        left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
        left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
        left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
        left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
        where DokumaPlanlama.DokumaMakinaId = :DokumaMakinaId and DokumaPlanlama.Sira = :MinSira and DokumaPlanlama.Bitti = 0 and DokumaPlanlama.Beklemede = 0
        into :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisKayitId, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisGrupAdi, :SiparisGrup2Adi, :DokumaPlanlamaId, :UretimPartiNo,
        :HamAdi, :PlanlananMiktar, :GerceklesenMiktar, :MiktarGun, :OrmeSiparisTipiId, :OrmeSiparisSekliId, :TupAcikEnMayli, :PlanlananTurSayisi, :PlanlamaDovizID, :PlanlamaBirimFiyat;

      Mayli = 0;
      AcikEn = 0;
      if (TupAcikEnMayli = 1) then AcikEn = 1;
      else if (TupAcikEnMayli = 2) then Mayli = 1;

      KalanMiktar = PlanlananMiktar - GerceklesenMiktar;

      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      Select Sum(Tur), Sum(TopKesme), Sum(TemizlikYapildi) from OrmeMakinaDurumu
        where OrmeMakinaDurumu.DokumaMakinaID = :DokumaMakinaId and
              OrmeMakinaDurumu.DokumaPlanlamaId = :DokumaPlanlamaId
      into :GerceklesenTurSayisi, :GerceklesenTopKesme, :GerceklesenTemizlikYapildi;

      Select Sum(Tur) from OrmeMakinaDurumu
        where OrmeMakinaDurumu.DokumaMakinaID = :DokumaMakinaId and
              OrmeMakinaDurumu.DokumaPlanlamaId = :DokumaPlanlamaId and
              OrmeMakinaDurumu.PersonelId = :PersonelId
      into :PersonelToplamTurSayisi;

      select First 1 DokumaKaliteKontrol.DokumaKaliteKontrolId, DokumaKaliteKontrol.Tarih from DokumaKaliteKontrol where DokumaKaliteKontrol.DokumaPlanlamaId = :DokumaPlanlamaId and OtomasyondanGeldi = 1 Order By DokumaKaliteKontrolID Desc into :SonDokumaKaliteKontrolId, :SonDokumaKKTarih;

    end
    suspend;
  end
end
