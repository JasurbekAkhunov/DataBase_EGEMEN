CREATE PROCEDURE ORMEMAKINAANIMASYONBILGISIOKU(
    AORMEMAKINAID VARCHAR(10))
RETURNS (
    ORMEMAKINAID VARCHAR(10),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ADI VARCHAR(30),
    ORMEPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    HAMADI VARCHAR(255),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    TOPLAMSAKATORULEN DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    MIKTARGUN DOUBLE PRECISION,
    SIPPUS INTEGER,
    SIPFEINE INTEGER,
    MAYLI INTEGER,
    ACIKEN INTEGER,
    ORMESIPARISTIPIID INTEGER,
    ORMESIPARISSEKLIID INTEGER,
    GERCEKLESENTURSAYISI INTEGER,
    BINTURDAORULECEKKG DOUBLE PRECISION,
    PLANLANANTURSAYISI INTEGER,
    TUR INTEGER,
    CALISIYOR SMALLINT,
    ARIZA INTEGER,
    SERVISCI INTEGER,
    AYAR INTEGER,
    MEMINGERKOPTU SMALLINT,
    ALTMEMINGERKOPTU SMALLINT,
    IGNEKOPTU SMALLINT,
    LYCRAKOPTU SMALLINT,
    KAPAKACILDI SMALLINT,
    STARTSTOP SMALLINT,
    TOPKESME SMALLINT,
    TEMIZLIKYAPILDI SMALLINT)
AS
declare variable MinSira integer;
declare variable TupAcikEnMayli integer;
declare variable EskiKayit SmallInt;
begin
  for Select OrmeMakina.OrmeMakinaId, min(sira) from OrmeMakina
    left join OrmePlanlama on OrmePlanlama.OrmeMakinaId = OrmeMakina.OrmeMakinaId and Bitti = 0
    where OrmeMakina.OtomasyonaDahil = 1 and (OrmeMakina.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null)
    Group By OrmeMakina.OrmeMakinaId into :OrmeMakinaId, :MinSira do
  begin
    Select Tur, Ariza, Bakim, Ayar, MemingerKoptu, AltMemingerKoptu, LycraKoptu, TemizlikYapildi, IgneKoptu, KapakAcildi, StartStop, TopKesme, iif(Current_TimeStamp(2) - DegisimTarihi > 0.0014, 1, 0) from MakinaDurumuOzet
      where MakinaDurumuOzet.OrmeMakinaId = :OrmeMakinaId
    into :Tur, :Ariza, :Servisci, :Ayar, :MemingerKoptu, :AltMemingerKoptu, :LycraKoptu, :TemizlikYapildi, :IgneKoptu, :KapakAcildi, :StartStop, :TopKesme, :EskiKayit;

    if (Ariza = 0 and Servisci = 0 and Ayar = 0 and MemingerKoptu = 0 and AltMemingerKoptu = 0 and LycraKoptu = 0 and IgneKoptu = 0 and KapakAcildi = 0 and StartStop = 0 and EskiKayit = 0) then Calisiyor = 1;
    else Calisiyor = 0;

    Select OrmeVardiyaPersonel.PersonelId, Personel.Adi from OrmeSonVardiyaIdGetir(:OrmeMakinaId, null)
      left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = OrmeSonVardiyaIdGetir.OrmeVardiyaId
      left join Personel on Personel.PersonelId = OrmeVardiyaPersonel.PersonelId
      where OrmeVardiyaPersonel.OrmeMakinaId = :OrmeMakinaId
      into :PersonelId, :PersonelAdi;

    if (MinSira is null) then
    begin
      BoyaSiparisYil      = Null;
      BoyaSiparisId       = Null;
      BoyaSiparisIdStr    = Null;
      SiparisFirmaId        = Null;
      SiparisFirmaAdi       = Null;
      OrmePlanlamaId        = Null;
      UretimPartiNo         = Null;
      HamAdi                = Null;
      PlanlananMiktar       = Null;
      GerceklesenMiktar     = Null;
      ToplamSakatOrulen     = Null;
      KalanMiktar           = Null;
      MiktarGun             = Null;
      SipPus                = Null;
      SipFeine              = Null;
      Mayli                 = Null;
      AcikEn                = Null;
      OrmeSiparisTipiId   = Null;
      OrmeSiparisSekliId  = Null;
      SiparisGrupAdi      = Null;
      SiparisGrup2Adi      = Null;
      BinTurdaOrulecekKg      = Null;
      PlanlananTurSayisi      = Null;
      GerceklesenTurSayisi    = Null;

    end
    else
    begin
      Select first 1 BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, SiparisGrup.Adi, SiparisGrup2.Adi, OrmePlanlama.OrmePlanlamaId, OrmePlanlama.UretimPartiNo,
             OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, Ham.Adi, OrmePlanlama.PlanlananMiktar,
             OrmePlanlama.GerceklesenMiktar, OrmePlanlama.ToplamSakatOrulen, OrmePlanlama.MiktarGun, BoyaSiparis.OrmeSiparisTipiId, BoyaSiparis.OrmeSiparisSekliId,
             OrmeSiparisDetay.TupAcikEnMayli, OrmePlanlama.BinTurdaOrulecekKg, OrmePlanlama.PlanlananTurSayisi from OrmePlanlama
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
        left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
        left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
        left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
        left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
        left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
        where OrmePlanlama.OrmeMakinaId = :OrmeMakinaId and OrmePlanlama.Sira = :MinSira and OrmePlanlama.Bitti = 0
        into :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisGrupAdi, :SiparisGrup2Adi, :OrmePlanlamaId, :UretimPartiNo, :SipPus, :SipFeine,
        :HamAdi, :PlanlananMiktar, :GerceklesenMiktar, :ToplamSakatOrulen, :MiktarGun, :OrmeSiparisTipiId, :OrmeSiparisSekliId, :TupAcikEnMayli, :BinTurdaOrulecekKg, :PlanlananTurSayisi;

      Mayli = 0;
      AcikEn = 0;
      if (TupAcikEnMayli = 1) then AcikEn = 1;
      else if (TupAcikEnMayli = 2) then Mayli = 1;

      KalanMiktar = PlanlananMiktar - GerceklesenMiktar + ToplamSakatOrulen;

      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    end
    suspend;
  end
end
