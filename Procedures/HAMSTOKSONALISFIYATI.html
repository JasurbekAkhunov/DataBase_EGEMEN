CREATE PROCEDURE HAMSTOKSONALISFIYATI(
    AHAMID INTEGER,
    ADOVIZID VARCHAR(5),
    ATARIH TIMESTAMP)
RETURNS (
    SONALISFIYATI DOUBLE PRECISION,
    DOVIZID VARCHAR(5))
AS
declare variable FSonAlisTarihi     TimeStamp;
declare variable FMaxHamStokDetayId integer;
declare variable DovizKuru          Double Precision;
declare variable UretimeCikisFiyatiHesapOran  Double Precision;
begin
  select max(Tarih) from HamStokDetay
     where HamStokDetay.HamId = :AHamId and HamStokDetay.HamStokHareketTipiId = 0 and HamStokDetay.BirimFiyat > 0 into :FSonAlisTarihi;

  select max(HamStokDetayId) from HamStokDetay
    where HamStokDetay.HamId = :AHamId and HamStokDetay.HamStokHareketTipiId = 0 and HamStokDetay.Tarih = :FSonAlisTarihi and HamStokDetay.BirimFiyat > 0
    into :FMaxHamStokDetayId;

  if (FMaxHamStokDetayId is not null) then
  begin
    Select HamStokDetay.BirimFiyat, HamStokIrsaliye.DovizId, HamTipi.UretimeCikisFiyatiHesapOran from HamStokDetay
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
      left join Ham on Ham.HamId = HamStokDetay.HamId
      left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
      where HamStokDetay.HamStokDetayId = :FMaxHamStokDetayId
      into :SonAlisFiyati, :DovizId, :UretimeCikisFiyatiHesapOran;

    if (ADovizId is not null) then
    begin
      if (ADovizId <> DovizId) then
      begin
         select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ADovizId, :ATarih) into :DovizKuru;
         SonAlisFiyati = SonAlisFiyati * DovizKuru;

         Select Fiyat from FormataGoreFiyatYuvarla(:SonAlisFiyati, 'BOYA', 'BIRIMFIYAT') into :SonAlisFiyati;
         DovizID = ADovizId;
      end
    end
    SonAlisFiyati = SonAlisFiyati + SonAlisFiyati * UretimeCikisFiyatiHesapOran / 100;
    Suspend;
  end
end
