CREATE PROCEDURE DETAYLISTOKENVANTERRAPORUSARF(
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    ABASLANGICURUNKATEGORIID VARCHAR(4),
    ABITISURUNKATEGORIID VARCHAR(4),
    AURUNKATEGORIARALIKLI INTEGER,
    ADEPOID VARCHAR(10),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ARAPORDOVIZID VARCHAR(5),
    ASIFIROLANLARDOKULSUNMU INTEGER,
    ABOYAURUNTIPI INTEGER,
    AKIMYASALURUNTIPI INTEGER,
    ADIGERURUNTIPI INTEGER,
    AYEDEKPARCAURUNTIPI INTEGER,
    ASARFMALZEMESIURUNTIPI INTEGER,
    AHIZMETURUNTIPI INTEGER,
    ASABLONURUNTIPI INTEGER,
    AAKSESUARURUNTIPI INTEGER,
    ATICARIMALURUNTIPI INTEGER,
    ABASLANGICRAF VARCHAR(15),
    ABITISRAF VARCHAR(15),
    ARAFARALIKLI INTEGER,
    AKAPALIDURUMU INTEGER,
    ASINIF VARCHAR(1),
    AMODELID MODELKODUSIZE,
    AURUNPARCADURUMU INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    STOKGRUPID VARCHAR(4),
    STOKGRUPADI VARCHAR(30),
    YATIRIM INTEGER,
    MIKTAR DOUBLE PRECISION,
    TOPLAMMALIYET DOUBLE PRECISION,
    DOVIZID VARCHAR(5))
AS
declare variable StokTabanDovizId      varchar(5);
declare variable RaporTarihi           TimeStamp;
declare variable UrunGrup1                     varchar(4);
declare variable UrunGrup2                     varchar(4);
declare variable UrunGrup3                     varchar(4);
declare variable UrunGrup4                     varchar(4);
declare variable UrunKategoriId                varchar(4);
declare variable Raf                           varchar(15);
declare variable DovizKuru DOuble Precision;
declare variable Okey integer;
declare variable shtUretimeCikis integer;
begin
  DovizID = ARaporDovizId;
  Raportarihi = Current_date;
  select STOKTABANDOVIZID from SISTEM where SISTEMID = 1 into :StokTabanDovizId;

  ABitisTarihi = f_stripTime(ABitisTarihi) || ' 23:59:59';

  if (StokTabanDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:StokTabanDovizId, :ARaporDovizId, :RaporTarihi) into :DovizKuru;
  else DovizKuru = 1;

  select shtUretimeCikis from StokHareketTipiGetir into :shtUretimeCikis;

  for Select Urun.UrunId, Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, Urun.UrunKategoriId, Urun.Raf, StokDetay.StokGrupId, StokHareketGrup.Adi, StokDetay.Yatirim,
             sum(StokDetay.Miktar / UrunBirim.Carpan),
             sum((StokDetay.Miktar * StokDetay.NetBirimFiyat) / StokDetay.TabanDovizKuru1 * StokDetay.TabanDovizKuru2 )  from StokDetay
    left join Urun on Urun.UrunId = StokDetay.UrunId
    left join Model on Model.ModelKayitID = Urun.ModelKayitID
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = StokDetay.StokGrupId
    where StokDetay.Tarih >= :ABaslangicTarihi and StokDetay.Tarih <= :ABitisTarihi and
          (StokDetay.Sinif = :ASinif or :ASinif is null) and
          (StokDetay.HareketTipi = :shtUretimeCikis) and
          (:AUrunAralikli = 0 or (:AUrunAralikli = 1 and StokDetay.UrunId >= :ABaslangicUrunId) and (StokDetay.UrunId <= :ABitisUrunId or :ABitisUrunId  = '')) and
          ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicGrup1Id  and (UrunGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
          ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicGrup2Id and (UrunGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
          ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicGrup3Id and (UrunGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
          ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicGrup4Id and (UrunGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = '')) and
          ((:AUrunKategoriAralikli = 1 and UrunKategoriId >= :ABaslangicUrunKategoriId and (UrunKategoriId <= :ABitisUrunKategoriId or :ABitisUrunKategoriId = '')) or (:AUrunKategoriAralikli = 0) or (:AUrunKategoriAralikli = 1 and :ABaslangicUrunKategoriId = '' and :ABitisUrunKategoriId = '')) and
          ((:ARafAralikli = 1 and Urun.Raf >= :ABaslangicRaf and (Urun.Raf <= :ABitisRaf or :ABitisRaf = '')) or (:ARafAralikli = 0) or (:ARafAralikli = 1 and :ABaslangicRaf = '' and :ABitisRaf = '')) and
          (:ADepoId is null or (:ADepoId is not null and StokDetay.DepoId = :ADepoId)) and
          (:AModelId is null or (:AModelId is not null and Model.ModelId = :AModelId)) and
          (Urun.Kapali = :AKapaliDurumu or :AKapaliDurumu is null) and
          (:AUrunParcaDurumu is null or (Iif(:AUrunParcaDurumu = 0, Iif(Urun.ModelParcaId Is Null, 1, 0), Iif(:AUrunParcaDurumu = 1, Iif(Urun.ModelParcaId Is Not Null, 1, 0), 1)) = 1)) and
          ((:ABoyaUrunTipi = 1 or Urun.UrunTipiId <> 0) and
           (:AKimyasalUrunTipi = 1 or Urun.UrunTipiId <> 1) and
           (:ADigerUrunTipi = 1 or Urun.UrunTipiId <> 2) and
           (:AYedekParcaUrunTipi = 1 or Urun.UrunTipiId <> 3) and
           (:ASarfMalzemesiUrunTipi = 1 or Urun.UrunTipiId <> 4) and
           (:AHizmetUrunTipi = 1 or Urun.UrunTipiId <> 5) and
            (:ASablonUrunTipi = 1 or Urun.UrunTipiId <> 6) and
            (:AAksesuarUrunTipi = 1 or Urun.UrunTipiId <> 7) and
            (:ATicariMalUrunTipi = 1 or Urun.UrunTipiId <> 8) )
 and
          Urun.EnvantereDahil = 1
    group by Urun.UrunId, Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, Urun.UrunKategoriId, Urun.Raf, StokDetay.StokGrupId, StokHareketGrup.Adi, StokDetay.Yatirim
    into :UrunId, :urunGrup1, :urunGrup2, :UrunGrup3, :UrunGrup4, :UrunKategoriId, :Raf,  :StokGrupId, :StokGrupAdi, :Yatirim, :Miktar, :ToplamMaliyet do
  begin
    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, UrunGrup4) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunKategoriAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunKategoriId, UrunKategoriId) = 0) then okey = 0;

    if (okey = 1) then
    if (ARafAralikli = 0) then
    if (KutuKarsilastir(ABaslangicRaf, Raf) = 0) then okey = 0;


    if (okey = 1) then
    begin
      ToplamMaliyet = ToplamMaliyet * DovizKuru;
      suspend;
    end
  end
end
