Definition

CREATE PROCEDURE GUNLUKPARTIDISTAMIRTTIPIANALIZI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP)
RETURNS (
    TAMIRTIPIID INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    TOPLAMKILO NUMERIC(12,3),
    TOPLAMMETRE NUMERIC(12,3),
    UCRETLITOPLAMKILO NUMERIC(12,3),
    UCRETLITOPLAMMETRE NUMERIC(12,3),
    UCRETSIZTOPLAMKILO NUMERIC(12,3),
    UCRETSIZTOPLAMMETRE NUMERIC(12,3),
    AYTOPLAMKILO NUMERIC(12,3),
    AYTOPLAMMETRE NUMERIC(12,3),
    UCRETLIAYTOPLAMKILO NUMERIC(12,3),
    UCRETLIAYTOPLAMMETRE NUMERIC(12,3),
    UCRETSIZAYTOPLAMKILO NUMERIC(12,3),
    UCRETSIZAYTOPLAMMETRE NUMERIC(12,3),
    ORMETOPLAMKILO NUMERIC(12,3),
    DOKUMATOPLAMKILO NUMERIC(12,3),
    ORMEAYTOPLAMKILO NUMERIC(12,3),
    DOKUMAAYTOPLAMKILO NUMERIC(12,3))
AS
declare variable AcilisTarihi TimeStamp;
declare variable FlastTamirTipiId  integer;
declare variable FHamTipiId  integer;
declare variable FTamirTipiId  integer;
declare variable FTamirTipiAdi varchar(20);
declare variable FKilo   Numeric(12,3);
declare variable FMetre  Numeric(12,3);
declare variable FFaturaKesilecek integer;
begin
/*  ToplamKilo  ==>  OrmeToplamKilo  +  DokumaToplamKilo
    AyToplamKilo  ==>  OrmeAyToplamKilo  +  DokumaAyToplamKilo   detaylı yazdıranlar icin sonradan örme ve dokuma diye ayırdık,
    eskileri etkilenmesin diye ToplamKilo ve AyToplamKilo vs  fieldlarını cıkarmadım
*/

  FlastTamirTipiId = -1;

  for Select Parti.AcilisTarihi, PartiStok.TamirTipiId, TamirTipi.Adi, PartiStok.HamTipiId, Parti.FaturaKesilecek,
             Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from Parti
    left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
    left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
    where Parti.AcilisTarihi >= :AAylikBaslangic and Parti.AcilisTarihi <= :AAylikBitis and Parti.IslemTipiID = 3
  group by PartiStok.TamirTipiId, TamirTipi.Adi, Parti.AcilisTarihi, PartiStok.HamTipiId, Parti.FaturaKesilecek
  order by PartiStok.TamirTipiId /* degistirme rapor düzgün calısmaz */
  into :AcilisTarihi, :FTamirTipiId, :FTamirTipiAdi, :FHamTipiId, :FFaturaKesilecek, :FKilo, :FMetre do
  begin
    if (FtamirTipiID is null) then FTamirTipiId = 0;

    if (FLastTamirTipiID <> FtamirTipiID) then
    begin
      if (FLastTamirTipiID <> -1) then suspend;
      FLastTamirTipiID = FtamirTipiID;
      TamirTipiId    = FTamirTipiId;
      TamirTipiAdi   = FTamirTipiAdi;
      ToplamKilo     = 0;
      ToplamMetre    = 0;
      UcretliToplamKilo    = 0;
      UcretliToplamMetre   = 0;
      UcretsizToplamKilo   = 0;
      UcretsizToplamMetre  = 0;


      AyToplamKilo     = 0;
      AyToplamMetre    = 0;
      UcretliAyToplamKilo   = 0;
      UcretliAyToplamMetre  = 0;
      UcretsizAyToplamKilo  = 0;
      UcretsizAyToplamMetre = 0;

      OrmeToplamKilo    = 0;
      DokumaToplamKilo  = 0;
      OrmeAyToplamKilo  = 0;
      DokumaAyToplamKilo = 0;
    end

    AyToplamKilo  = AyToplamKilo + FKilo;
    AyToplamMetre = AyToplamMetre + FMetre;

    if (FFaturaKesilecek = 1) then
    begin
      UcretliAyToplamKilo  = UcretliAyToplamKilo + FKilo;
      UcretliAyToplamMetre = UcretliAyToplamMetre + FMetre;
    end
    else
    begin
      UcretsizAyToplamKilo  = UcretsizAyToplamKilo + FKilo;
      UcretsizAyToplamMetre = UcretsizAyToplamMetre + FMetre;
    end

    if (FHamTipiId <> 2) then
      OrmeAyToplamKilo  = OrmeAyToplamKilo + FKilo;
    else
      DokumaAyToplamKilo  = DokumaAyToplamKilo + FKilo;

    if (AcilisTarihi >= ABaslangicTarihi and AcilisTarihi <= ABitisTarihi) then
    begin
      ToplamKilo  = ToplamKilo + FKilo;
      ToplamMetre = ToplamMetre + FMetre;

      if (FFaturaKesilecek = 1) then
      begin
        UcretliToplamKilo  = UcretliToplamKilo + FKilo;
        UcretliToplamMetre = UcretliToplamMetre + FMetre;
      end
      else
      begin
        UcretsizToplamKilo  = UcretsizToplamKilo + FKilo;
        UcretsizToplamMetre = UcretsizToplamMetre + FMetre;
      end

      if (FHamTipiId <> 2) then
        OrmeToplamKilo  = OrmeToplamKilo + FKilo;
      else
        DokumaToplamKilo  = DokumaToplamKilo + FKilo;

    end
  end
  if (FLastTamirTipiID <> -1) then suspend;
end
