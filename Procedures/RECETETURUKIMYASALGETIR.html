CREATE PROCEDURE RECETETURUKIMYASALGETIR
RETURNS (
    RECETETURUID INTEGER,
    RECETETURUADI VARCHAR(15),
    RECETEASAMAID INTEGER,
    RECETEASAMAADI VARCHAR(40),
    KIMYASALTIPIID INTEGER,
    KIMYASALTIPIADI VARCHAR(20),
    BASLANGICMIKTAR DOUBLE PRECISION,
    BITISMIKTAR DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION)
AS
declare variable KayitVar integer;
begin
  For Select ReceteTuru.ReceteTuruId, ReceteTuru.Adi from ReceteTuru
    into :ReceteTuruId, :ReceteTuruAdi do
  begin
    For Select ReceteAsama.ReceteAsamaId, ReceteAsama.Adi from ReceteAsama
      into :ReceteAsamaId, :ReceteAsamaAdi do
    begin
      For Select KimyasalTipi.KimyasalTipiId, KimyasalTipi.Adi from KimyasalTipi
        into :KimyasalTipiId, :KimyasalTipiAdi do
      begin
        KayitVar = 0;
        BaslangicMiktar = Null;
        BitisMiktar = Null;
        Miktar = Null;

        For Select BaslangicMiktar, BitisMiktar, Miktar from ReceteTuruKimyasal
          where ReceteTuruKimyasal.ReceteTuruID = :ReceteTuruID and
                ReceteTuruKimyasal.ReceteAsamaId = :ReceteAsamaId and
                ReceteTuruKimyasal.KimyasalTipiId = :KimyasalTipiId
          order by ReceteTuruKimyasal.BaslangicMiktar
          into :BaslangicMiktar, :BitisMiktar, :Miktar do
        begin
          KayitVar = 1;
          suspend;
        end

        if (KayitVar = 0) then suspend;
      end
    end
  end
end
