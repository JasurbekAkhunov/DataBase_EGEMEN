CREATE PROCEDURE LABRECETEURUNLISTESI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTALEPTARIHI TIMESTAMP,
    ABITISTALEPTARIHI TIMESTAMP,
    ALABRECETEGRUPID INTEGER,
    AGRUPIPLIKBOYAMI INTEGER,
    ALABRENKGRUPID INTEGER,
    AFASON INTEGER,
    AESKILABRECETEKODU VARCHAR(20),
    APANTONEKODU VARCHAR(20),
    AHAMID INTEGER,
    AKALITEID VARCHAR(20),
    AURUNFIYATGOREBILIR SMALLINT,
    ALABRECETEKODU VARCHAR(20),
    ALABRECETETALEPYIL INTEGER,
    ALABRECETETALEPID INTEGER)
RETURNS (
    LABRECETEURUNID INTEGER,
    LABRECETEATISID INTEGER,
    LABRECETEATISRENKID INTEGER,
    LABRECETEID INTEGER,
    SIRA INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    KIMYASALTIPIID INTEGER,
    KIMYASALTIPIADI VARCHAR(20),
    PIPETLEMETRANSFERKODU VARCHAR(55),
    MIKTARYUZDE DOUBLE PRECISION,
    HESAPBIRIMID VARCHAR(5),
    BIRIMID VARCHAR(2),
    PIPETLEMEBIRIMKODU VARCHAR(10),
    URUNTIPIID INTEGER,
    BIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select LabReceteUrun.LabReceteUrunId, LabReceteUrun.LabReceteAtisId, LabReceteAtisListesi.LabReceteId,
   LabReceteUrun.UrunId, Urun.Adi, Urun.PipetlemeTransferKodu, Urun.KimyasalTipiId, KimyasalTipi.Adi, LabReceteUrun.MiktarYuzde, LabReceteUrun.BirimFiyat, LabReceteGrup.Caption, UrunBirim.BirimId,
   ((MiktarYuzde * LabReceteGrup.Carpan) * BirimFiyat  / FiyatCarpan),
   IK.Adi, LabReceteAtisListesi.InsertTarihi, SK.Adi, LabReceteAtisListesi.DegisimTarihi, Birim.PipetlemeBirimKodu, Urun.UrunTipiId, LabReceteUrun.Sira, LabReceteUrun.LabReceteAtisRenkId  from
   LabReceteAtisListesi(:AFirmaId, :ABaslangicTalepTarihi, :ABitisTalepTarihi, :ALabReceteGrupid, :AGrupIplikBoyami, :ALabRenkGrupid, :AFason, :AEskiLabReceteKodu, :APantoneKodu, :AHamId, :AKaliteId, :AUrunFiyatGorebilir, :ALabReceteKodu, :ALabReceteTalepYil, :ALabReceteTalepId)
  left join LabRecete on LabRecete.LabReceteId = LabReceteAtisListesi.LabReceteId
  left join LabReceteGrup on LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
  left join LabReceteUrun on LabReceteUrun.LabReceteAtisId = LabReceteAtisListesi.LabReceteAtisId
  left join Urun on Urun.UrunId = LabReceteUrun.UrunId
  left join UrunBirim on UrunBirim.UrunBirimId = LabReceteUrun.UrunBirimId
  left join Birim on Birim.BirimId = UrunBirim.BirimId
  left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = urun.KimyasalTipiid
  Left Join Kullanici IK on IK.KullaniciId = LabReceteUrun.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = LabReceteUrun.KullaniciId
  Order By LabReceteUrun.Sira
  into :LabReceteUrunId, :LabReceteAtisId, :LabReceteId, :UrunId, :UrunAdi, :PipetlemeTransferKodu,
       :KimyasalTipiId, :KimyasalTipiAdi, :MiktarYuzde, :BirimFiyat, :HesapBirimId, :BirimId,
       :Tutar, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :PipetlemeBirimKodu, :UrunTipiId, :Sira, :LabReceteAtisRenkId do
  begin
    if (AUrunFiyatGorebilir = 0) then
    begin
      Tutar = Null;
      BirimFiyat = Null;
    end

    suspend;
  end
end
