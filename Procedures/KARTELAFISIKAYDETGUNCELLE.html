CREATE PROCEDURE KARTELAFISIKAYDETGUNCELLE(
    AKARTELAFISIID INTEGER,
    AFIRMAADI VARCHAR(100),
    AYETKILIKISI VARCHAR(100),
    AEMAIL VARCHAR(100),
    ATELEFON VARCHAR(100),
    ANOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    AKOMISYON INTEGER,
    ANAVLUN DOUBLE PRECISION,
    AKULLANICIID INTEGER,
    ARESIMDOSYAADI VARCHAR(255),
    AFIRMAID FIRMAKODUSIZE)
RETURNS (
    DURUMMSG VARCHAR(50),
    KARTELAFISIID INTEGER)
AS
declare variable FFirmaID   varchar(20);
declare variable FFisYil integer;
declare variable FFisNo integer;
declare variable FBoyaTeklifKayitID integer;
begin
  if (AKartelaFisiId = -1) then AKartelaFisiId = Null;
  KartelaFisiId = :AKartelaFisiId;
  if (AFirmaID = '') then AFirmaID = Null;

  if (AKartelaFisiId is Null) then
  begin
    Select first 1 FirmaID, BoyaTeklifKayitID from BoyaTeklif where FirmaID = (Select Sistem.FirmaKodumuz from Sistem where Sistem.SistemId = 1) Order By TeklifTarihi Desc, BoyaTeklifKayitID desc into :FFirmaID, :FBoyaTeklifKayitID;

    if (AFirmaID is not Null) then FFirmaID = AFirmaID;

    KartelaFisiID = gen_ID(KartelaFisi_Id_gen, 1);
    FFisYil = f_Year(Current_date);
    Select Id from KartelaFisiNumaratorAl(-1, :FFisYil) into :FFisNo;

    Insert Into KartelaFisi(KartelaFisiID, FisYil, FisNo, Fistarihi, FirmaId, Giris, Notu,
                             InsertkullaniciID, Kullaniciid, BoyaTeklifKayitID, MusteriAdi, MusteriEMail, MusteriTelefon, Komisyon, Navlun, ResimDosyaAdi, YetkiliKisi)
                     Values (:KartelaFisiID, :FFisYil, :FFisNo, Current_Date, :FFirmaID, -1, :ANotu,
                             :AKullaniciID, :AKullaniciID, :FBoyaTeklifKayitID, :AFirmaAdi, :AEMail, :ATelefon, :AKomisyon, :ANavlun, :AResimDosyaAdi, :AYetkiliKisi);
  end
  else
  begin
    Update KartelaFisi set Notu = :ANotu, FirmaID = :AFirmaID, YetkiliKisi = :AYetkiliKisi, MusteriAdi = :AFirmaAdi, MusteriEMail = :AEMail, MusteriTelefon = :ATelefon, Komisyon = :AKomisyon, Navlun = :ANavlun, ResimDosyaAdi = :AResimDosyaAdi, Kullaniciid = :AKullaniciID where KartelaFisi.KartelaFisiID = :AKartelaFisiID;
  end

  DurumMsg = 'OK';
  Suspend;
end
