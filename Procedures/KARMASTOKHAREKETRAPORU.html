            (Stok.IlgiliKullaniciID = :AIlgiliKullaniciID or :AIlgiliKullaniciID is null) and
            (Stok.IlgiliPersonelID = :AIlgiliPersonelID or :AIlgiliPersonelID is null) and
            ((:ASiparisNoAralikli = 1 and (HGSiparis.HGSiparisYil > :ABaslangicHGSiparisYil or (HGSiparis.HGSiparisYil = :ABaslangicHGSiparisYil and HGSiparis.HGSiparisId >= :ABaslangicHGSiparisId)) and
            (HGSiparis.HGSiparisYil < :ABitisHGSiparisYil or (HGSiparis.HGSiparisYil = :ABitisHGSiparisYil and HGSiparis.HGSiparisId <= :ABitisHGSiparisId))) or :ASiparisNoAralikli = 0) and
            (HGIsEmri.UretimPartiNo = :AUretimPartiNo or :AUretimPartiNo is null)
          into :StokDetayId, :StokId, :IrsaliyeId, :FaturaId, :HareketTipi, :FirmaId, :FirmaAdi, :FisNo, :FisTarihi, :IrsaliyeNo,
               :IrsaliyeTarihi, :FiiliSevkTarihi, :FaturaNo, :FaturaTarihi, :DepoId, :DepoAdi, :TransferDepoId, :TransferDepoAdi, :FisGrupId, :IrsaliyeGrupId, :FaturaGrupId,
               :FisGrupAdi, :IrsaliyeGrupAdi, :FaturaGrupAdi, :Miktar, :NetBirimFiyat, :UrunBirimId, :BirimId, :Notu, :Yatirim,
               :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :Tarih,
               :SDovizId, :IDovizId, :FDovizId, :VerilenSiparisYil, :VerilenSiparisId, :UniteID, :UniteAdi, :MasrafMerkeziId, :MasrafMerkeziAdi, :DepartmanID, :DepartmanAdi,
               :TabanDovizKuru1, :TabanDovizKuru2, :MakinaId, :MakinaAdi, :FBirimCarpan, :Sinif,
               :FiyatAciklama, :DengelemeEvraki, :OrmeMakinaId, :OrmeMakinaAdi, :IplikMakinaId, :IplikMakinaAdi, :Lot,
               :PartiYil, :PartiID, :PartiFirmaID, :PartiFirmaAdi, :PartiReceteID, :PartiReceteAdi, :IlgiliKullaniciID, :IlgiliKullaniciAdi, :MakinaGrupID, :MakinaGrupAdi,
               :OrmeMakinaGrupID, :OrmeMakinaGrupAdi, :IplikMakinaGrupID, :IplikMakinaGrupAdi, :DokumaMakinaID, :DokumaMakinaAdi, :DokumaMakinaGrupID, :DokumaMakinaGrupAdi,
               :RaporMakinaID, :RaporMakinaAdi, :RaporMakinaGrupID, :RaporMakinaGrupAdi, :IlgiliPersonelID, :IlgiliPersonelAdi, :ReferansStokDetayID, :DesenID, :DesenAdi,
               :HGIsEmriKayitID, :HGIsEmriUrunID, :HGIsEmriPastalID, :HGSiparisKayitId, :HGSiparisYil, :HGSiparisId, :HGSiparisDetayId, :UretimPartiNo, :KesimTalimati, :IslemKaynagi, :VergiHesapNo, :FirmaDovizId, :FirmaDovizKuru1, :FirmaDovizKuru2,
               :BrutTutar, :KDV, :Vadesi, :EntegrasyonEkBilgi, :FPartiKayitId, :FPartiAsamalariId, :FisSeriNo , IrsaliyeSeriNo, :FaturaSeriNo, :UretimeCikisTarihi, :UretimTarihi,
               :IrsaliyeNotu, :FisNotu, :MusteriSiparisNo do

        begin

          Okey = 1;

          if (FisGrupId is not null) then
          if (AFisGrupAralikli = 0) then
          if (KutuKarsilastir(ABaslangicFisGrupId, FisGrupId) = 0) then okey = 0;

          if (Okey = 1) then
          if (IrsaliyeGrupId is not null) then
          if (AIrsaliyeGrupAralikli = 0) then
          if (KutuKarsilastir(ABaslangicIrsaliyeGrupId, IrsaliyeGrupId) = 0) then okey = 0;

          if (Okey = 1) then
          if (FaturaGrupId is not null) then
          if (AFaturaGrupAralikli = 0) then
          if (KutuKarsilastir(ABaslangicFaturaGrupId, FaturaGrupId) = 0) then okey = 0;

          if (okey = 1) then
          begin

            HGSiparisIdStr = '';
            if (HGSiparisId is not null) then HGSiparisIdStr = '(' || EgemenIntegerTostr(HGSiparisYil) || ') '  || EgemenIntegerTostr(HGSiparisId)  || ' ';
            UretimIslemleriID = '';
            UretimIslemleriAdi = '';
            Select UretimIslemleriID, UretimIslemleriAdi from HGIsEmriUretimIslemiGetir(:HGIsEmriKayitID, :KesimTalimati) into :UretimIslemleriId, :UretimIslemleriAdi;


            if (DengelemeEvraki = 1) then DengelemeEvrakiStr = 'Evet';
            else DengelemeEvrakiStr = '';

            VerilenSiparisIdStr = '';
            if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

            if (FDovizId is not null) then DovizId = FDovizId;
            else if (IDovizId is not null) then DovizId = IDovizId;
            else if (SDovizId is not null) then DovizId = SDovizId;

            if (HareketTipi = shtAlis) then HareketTipiStr = 'Alış';
            else if (HareketTipi = shtSatis) then HareketTipiStr = 'Satış';
            else if (HareketTipi = shtAlisIade) then HareketTipiStr = 'Alış İade';
            else if (HareketTipi = shtSatisIade) then HareketTipiStr = 'Satış İade';
            else if (HareketTipi = shtUretimeCikis) then HareketTipiStr = 'Üretime Çıkış';
            else if (HareketTipi = ShtDepoTransfer) then HareketTipiStr = 'Depo Transfer';
            else if (HareketTipi = ShtUretim) then HareketTipiStr = 'Üretim';
            else if (HareketTipi = shtFasonUretimeCikis) then HareketTipiStr = 'Fason Üretime Çıkış';
            else if (HareketTipi = ShtFasonUretim) then HareketTipiStr = 'Fason Üretim';
            else HareketTipiStr = 'Hatalı!';


            if (Yatirim = 1) then YatirimStr = 'Yatırım';
            else YatirimStr = '';

            RaporNetBirimFiyat = NetBirimFiyat;


            ReferansIrsFirmaID = Null;
            ReferansIrsTarihi = Null;
            ReferansIrsNo = Null;
            if (ReferansStokDetayID is not Null) then
            begin
              Select First 1 Irsaliye.FirmaID, Irsaliye.IrsaliyeTarihi, Irsaliye.SeriNo || Irsaliye.IrsaliyeNo from StokDetay
                left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeID
                where StokDetay.StokDetayID = :ReferansStokDetayID
              into :ReferansIrsFirmaID, :ReferansIrsTarihi, :ReferansIrsNo;
            end

            if (DovizId <> RaporDovizId and RaporNetBirimFiyat <> 0) then
            begin
              Select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :RaporDovizId, Current_Date) into :FDovizKuru;

              RaporNetBirimFiyat = RaporNetBirimFiyat * FDovizKuru;
            end

            Tutar = NetBirimFiyat * Miktar;
            RaporTutar = RaporNetBirimFiyat * Miktar;

            if (FirmaDovizId <> DovizId) then FirmaDovizKuruStr = Format_Float('', FirmaDovizKuru1) || ' ' || DovizId || '-' || Format_Float('', FirmaDovizKuru2) || ' ' || FirmaDovizId;
            else FirmaDovizKuruStr = '';

            if (FirmaDovizId <> DovizId) then FirmaTutar = NetBirimFiyat * Miktar * :FirmaDovizKuru2;
            else FirmaTutar = NetBirimFiyat * Miktar ;

            if (DovizId <> StokTabanDovizId) then
            begin
              select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :StokTabanDovizId, :Tarih) into :FDovizKuru, :DovizKuruTarihi;
              if (Tarih <> DovizKuruTarihi) then DovizKuruMesaj = 'Fatura Tarihine Ait Kur Girisi Bulunamadigi icin ' || TimeStampToDateTimeStr(DovizKuruTarihi) || ' Tarihli Kur ile Hesaplandi';
            end
            else FDovizKuru = 1;

            DefaultBirimMiktar = Miktar;
            DefaultBirimFiyat = NetBirimFiyat;

            if (FBirimCarpan is Null) then FBirimCarpan = 0;
            if (DefaultBirimCarpan is Null) then DefaultBirimCarpan = 0;
            if (FBirimCarpan <> 0 and DefaultBirimCarpan <> 0) then
            begin
              if (FBirimCarpan <> DefaultBirimCarpan) then
              begin
                 DefaultBirimMiktar = DefaultBirimMiktar / FBirimCarpan * DefaultBirimCarpan;
                 DefaultBirimFiyat = DefaultBirimFiyat / FBirimCarpan * DefaultBirimCarpan;
              end
            end
            DefaultBirimFiyatTL = DefaultBirimFiyat * FDovizKuru;

            IlaveSarfAciklamalari = '';
            for select PartiAsamaIlaveSarf.Aciklama from PartiAsamaIlaveSarf where PartiAsamaIlaveSarf.PartiKayitId = :FPartiKayitId into :FIlaveSarfAciklamasi do
            begin
              if (char_length(IlaveSarfAciklamalari) + char_length(FIlaveSarfAciklamasi) < 250) then
              begin
                if (IlaveSarfAciklamalari <> '' ) then IlaveSarfAciklamalari = IlaveSarfAciklamalari || cast(', ' as varchar(2) character set UTF8);
                IlaveSarfAciklamalari = IlaveSarfAciklamalari || FIlaveSarfAciklamasi;
              end
            end

            IlaveTipiAdiAciklamalari = '';
            for select distinct IlaveTipi.Adi from PartiAsamaIlaveSarf
              left join IlaveTipi on IlaveTipi.IlaveTipiID = PartiAsamaIlaveSarf.IlaveTipiID
            where PartiAsamaIlaveSarf.PartiKayitId = :FPartiKayitId into :FIlaveTipiAdi do
            begin
              if (char_length(IlaveTipiAdiAciklamalari) + char_length(FIlaveTipiAdi) < 250) then
              begin
                if (IlaveTipiAdiAciklamalari <> '' ) then IlaveTipiAdiAciklamalari = IlaveTipiAdiAciklamalari || cast(', ' as varchar(2) character set UTF8);
                IlaveTipiAdiAciklamalari = IlaveTipiAdiAciklamalari || FIlaveTipiAdi;
              end
            end
            HamAdiToplami = '';
            PartiKilo     = 0;

            For Select PartiStok.HamAdiToplami from PartiAsamaStok
                  left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
                  where PartiAsamaStok.PartiKayitId = :FPartiKayitId and PartiAsamaStok.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
                  Group BY PartiStok.HamAdiToplami
                  into :FHamAdiToplami do
            begin
              if (Char_Length(HamAdiToplami) + Char_Length(FHamAdiToplami) < 995) then
              begin
                if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set utf8);
                HamAdiToplami = cast(HamAdiToplami as blob character set utf8) || FHamAdiToplami;

              end
            end

            Select sum(PartiStok.Kilo) from PartiAsamaStok
                 left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
                 where PartiAsamaStok.PartiKayitId = :FPartiKayitId and PartiAsamaStok.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
                 into :PartiKilo;

            if (Lot is null) then
               select SonAlisTarihi, SonAlisFiyati, SonAlisDovizId from EnvanterIcinSonAlisBilgisiGetir(:UrunId, null) into :SonAlisTarihi, :SonAlisFiyati, :SonAlisDovizId;
            else
              select SonAlisTarihi, SonAlisFiyati, SonAlisDovizId from EnvanterIcinSonAlisBilgisiGetir(:UrunId, null) into :SonAlisTarihi, :SonAlisFiyati, :SonAlisDovizId;

            RaporSonAlisFiyati = SonAlisFiyati;
            if (SonAlisDovizId <> RaporDovizId and SonAlisFiyati <> 0) then
            begin
              Select DovizKuru from EnYakinDovizKuruGetir(:SonAlisDovizId, :RaporDovizId, Current_Date) into :FDovizKuru;

              RaporSonAlisFiyati = RaporSonAlisFiyati * FDovizKuru;
            end

            TakipBirimFiyatiTLToplam = DefaultBirimFiyatTL * Miktar;
            suspend;
          end
        end
      end
