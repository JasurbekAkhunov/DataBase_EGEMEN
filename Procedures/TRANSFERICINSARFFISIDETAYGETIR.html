CREATE PROCEDURE TRANSFERICINSARFFISIDETAYGETIR(
    ASTOKID INTEGER)
RETURNS (
    STOKDETAYID INTEGER,
    URUNID VARCHAR(20),
    TRANSFERKODU VARCHAR(55),
    MIKTAR DOUBLE PRECISION,
    BIRIMID VARCHAR(2),
    BIRIMTRANSFERKODU VARCHAR(10),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    BIRIMFIYATTL DOUBLE PRECISION,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    DEPOTRANSFERKODU VARCHAR(55),
    MASRAFMERTRANSFERKODU VARCHAR(30))
AS
declare variable FisTarihi       TimeStamp;
declare variable DovizKuru          Double Precision;
declare variable SistemDovizId      varchar(5);
declare variable EgemenFirmaID  integer;
begin
  Select SistemDovizId, EgemenFirmaID from Sistem where sistemId = 1 into :SistemDovizId, :EgemenFirmaID; /* YTL*/

  for select StokDetay.StokDetayId, Urun.UrunId, Urun.TransferKodu, StokDetay.Miktar, UrunBirim.BirimId, Birim.TransferKodu, StokDetay.BirimFiyat,
             Stok.FisTarihi, Stok.DovizId, Parti.PartiYil, Parti.PartiId, Depo.TransferKodu, iif(MasrafMerkezi.TransferKodu is Not Null, MasrafMerkezi.TransferKodu, UrunKategori.OzelKod) from StokDetay
    left join Stok on Stok.StokId = StokDetay.StokId
    left join Urun on Urun.UrunId = StokDetay.UrunId
    left join UrunKategori On UrunKategori.UrunKategoriID = Urun.UrunKategoriID
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join Birim on Birim.BirimId = UrunBirim.BirimId
    left join Parti on Parti.PartiKayitId = StokDetay.PartiKayitId
    left join Depo on Depo.DepoId = StokDetay.DepoId
    left join MasrafMerkezi on MasrafMerkezi.MasrafMerkeziId = Stok.MasrafMerkeziId
    where StokDetay.StokId = :AStokId order by StokDetayId
    into :StokDetayId, :UrunId, :TransferKodu, :Miktar, :BirimId, :BirimTransferKodu, :BirimFiyat, :FisTarihi, :DovizId,
         :PartiYil, :PartiId, :DepoTransferKodu, :MasrafMerTransferKodu do
  begin
    if (DovizID <> SistemDovizId) then
    begin
      select DovizKuru from DovizKuruGetir(:DovizId, :SistemDovizId, :FisTarihi) into :DovizKuru;
    end
    else DovizKuru = 1;

    BirimFiyatTL = BirimFiyat * DovizKuru;


    if (EgemenFirmaID = 46 and BirimId = 'Gr') then
    begin
      Miktar = Miktar / 1000;
      BirimId = 'KG';
      BirimTransferKodu = 'KG';
    end

    suspend;
  end
end
