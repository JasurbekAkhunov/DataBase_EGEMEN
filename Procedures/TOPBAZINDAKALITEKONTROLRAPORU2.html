CREATE PROCEDURE TOPBAZINDAKALITEKONTROLRAPORU2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    ARECETEID VARCHAR(20),
    AFIRMAID VARCHAR(20),
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AHAMID INTEGER,
    ABOYASIPSIRANO INTEGER,
    AONAYDURUMU INTEGER,
    ABARKODID VARCHAR(20),
    AICTAMIR SMALLINT,
    AKALITENO INTEGER,
    ATUMTOPLAR INTEGER,
    AIMALATTIPIID INTEGER,
    APARTIGRUPID SIPARISGRUPKODUSIZE)
RETURNS (
    PSKALITEKONTROLID INTEGER,
    PSKALITEKONTROLHATAMETREID INTEGER,
    HATATIPIID INTEGER,
    HATATIPIADI ADI20,
    BASLANGICMETRE BOYAMIKTARSIZE,
    BITISMETRE BOYAMIKTARSIZE,
    HATAPUANI DOUBLE PRECISION,
    HATAYERI INTEGER,
    HATAYERICM INTEGER,
    HATAYERISTR VARCHAR(4))
AS
begin
  for select TopBazindaKaliteKontrolRaporu.PsKaliteKontrolId, PsKaliteKontrolHataMetre.PsKaliteKontrolHataMetreId, PsKaliteKontrolHataMetre.HataTipiId, HataTipi.Adi,
      PsKaliteKontrolHataMetre.BaslangicMetre, PsKaliteKontrolHataMetre.BitisMetre, PsKaliteKontrolHataMetre.HataPuani, PsKaliteKontrolHataMetre.HataYeri,
      PsKaliteKontrolHataMetre.HataYeriCm from TopBazindaKaliteKontrolRaporu(:ABaslangicTarihi, :ABitisTarihi, :APartiYil, :APartiId, :AReceteId, :AFirmaId,
                    :ABoyaSiparisYil, :ABoyaSiparisId, :AHamId, :ABoyaSipSiraNo, :AOnayDurumu, :ABarkodId, :AIcTamir, :AKaliteNo, :ATumToplar,
                    :AImalatTipiId, :APartiGrupId, 0, null, 0, null, Null, null, null,null,null,null,null)
      left join PsKaliteKontrolHataMetre on PsKaliteKontrolHataMetre.PsKaliteKontrolId = TopBazindaKaliteKontrolRaporu.PsKaliteKontrolId
      left join HataTipi on HataTipi.HataTipiId = PsKaliteKontrolHataMetre.HataTipiId
      into :PsKaliteKontrolId, :PsKaliteKontrolHataMetreId, :HataTipiId, :HataTipiAdi, :BaslangicMetre, :BitisMetre, :HataPuani, :HataYeri, :HataYeriCm do
  begin
    if (HataYeri is null) then HataYeriStr = '';
    else if (HataYeri = 1) then HataYeriStr = 'Sol';
    else if (HataYeri = 2) then HataYeriStr = 'Orta';
    else if (HataYeri = 3) then HataYeriStr = 'Sağ';

    suspend;
  end
end
