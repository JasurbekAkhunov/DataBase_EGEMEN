CREATE PROCEDURE HGSIPARISRAPORU_1(
    ABASLANGICSIPARISYIL INTEGER,
    ABASLANGICSIPARISID INTEGER,
    ABITISSIPARISYIL INTEGER,
    ABITISSIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASINIF VARCHAR(1),
    AMODELID MODELKODUSIZE,
    ASIPONAYDURUMU SMALLINT,
    AMALIONAYDURUMU SMALLINT)
RETURNS (
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER,
    HGSIPARISIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAYETKILIID INTEGER,
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    HGSIPARISKAYITID INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISIVERENADI ADI50,
    SIPARISTARIHI TIMESTAMP,
    SOZLESMETARIHI TIMESTAMP,
    NOTU VARCHAR(1024),
    YURTDISI INTEGER,
    YURTDISISTR VARCHAR(9),
    FASON SMALLINT,
    FASONSTR VARCHAR(5),
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI ADI40,
    TESLIMSEKLIID INTEGER,
    TESLIMSEKLIADI ADI40,
    BASKAFIRMAYASATILABILIR EGEBOOLEAN,
    BASKAFIRMAYASATILABILIRSTR VARCHAR(5),
    HGSIPARISDETAYID INTEGER,
    MODELID MODELKODUSIZE,
    MODELADI ADI40,
    SIPARISSIRANO INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIMODELNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    SINIF VARCHAR(1),
    HAMSTOKBIRIMID INTEGER,
    SATISFIYATI DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    SATISTIPIID SMALLINT,
    SATISTIPIADI VARCHAR(10),
    TESLIMTARIHI TIMESTAMP,
    VADEGUN INTEGER,
    MODELNOTU MEMO,
    ORMENOTU MEMO,
    DOKUMANOTU MEMO,
    BOYAHANENOTU MEMO,
    AKSESUARNOTU MEMO,
    NUMUNENOTU MEMO,
    DIKISTALIMATI MEMO,
    DIKISTALIMATRESMI VARCHAR(500),
    RESIMDOSYAADI1 VARCHAR(255),
    RESIMDOSYAYAZDIR1 EGEBOOLEAN,
    RESIMDOSYAADI2 VARCHAR(255),
    RESIMDOSYAYAZDIR2 EGEBOOLEAN,
    IPTAL EGEBOOLEAN,
    IPTALSTR VARCHAR(5),
    KAPALI EGEBOOLEAN,
    KAPALISTR VARCHAR(5),
    KAPATMATARIHI TIMESTAMP,
    KUMASSATINALMATALEPKAYITID INTEGER,
    AKSESUARSATINALMATALEPKAYITID INTEGER,
    ONAYDURUMU EGEBOOLEAN,
    ONAYDURUMUSTR VARCHAR(20),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI ADI40,
    MALIONAYDURUMU SMALLINT,
    MALIONAYDURUMUSTR VARCHAR(20),
    MALIONAYTARIHI TIMESTAMP,
    MALIONAYLAYANKULLANICIID INTEGER,
    MALIONAYLAYANKULLANICIADI ADI40,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
Declare Variable ModelKayitId Integer;
begin
  Select ModelKayitId From Model Where Model.ModelID = :AModelID into :ModelKayitId;

  for select Hgsiparis.Hgsipariskayitid, Hgsiparis.Hgsiparisyil, Hgsiparis.Hgsiparisid, Hgsiparis.Firmaid, Firma.Adi,
             Hgsiparis.Siparistarihi, Hgsiparis.Sozlesmetarihi, Hgsiparis.Firmayetkiliid, Hgsiparis.Siparisialankullaniciid, SAK.Adi,
             Hgsiparis.Siparisgrupid, SiparisGrup.Adi, Hgsiparis.Siparisgrup2id, SiparisGrup2.Adi, FirmaYetkili.Adi, Hgsiparis.Fason, HgSiparisDetay.Odemesekliid, OdemeSekli.Adi,
             Hgsiparis.Teslimsekliid, TeslimSekli.Adi, Hgsiparis.Yurtdisi, HgsiparisDetay.Notu, Hgsiparis.Baskafirmayasatilabilir,
             IK.Adi, HgsiparisDetay.Inserttarihi, SK.Adi, HgsiparisDetay.Degisimtarihi,
             Hgsiparisdetay.Hgsiparisdetayid, Hgsiparisdetay.Hgsipariskayitid, Model.Modelid, Model.Adi,
             Hgsiparisdetay.Siparissirano, Hgsiparisdetay.Musterisiparisno, Hgsiparisdetay.Musterimodelno,
             Hgsiparisdetay.Referanssiparisno, Hgsiparisdetay.Sinif, Hgsiparisdetay.Hamstokbirimid,
             Hgsiparisdetay.Satisfiyati, Hgsiparisdetay.Dovizid, Hgsiparisdetay.Satistipiid, HgSatisTipiGetir.Adi, Hgsiparisdetay.Teslimtarihi,
             Hgsiparisdetay.Vadegun, Hgsiparisdetay.Resimdosyaadi1, Hgsiparisdetay.Resimdosyayazdir1, Hgsiparisdetay.Resimdosyaadi2,
             Hgsiparisdetay.Resimdosyayazdir2, Hgsiparisdetay.Iptal, Hgsiparisdetay.Kapali, Hgsiparisdetay.Kapatmatarihi,
             Hgsiparisdetay.Onaydurumu, Hgsiparisdetay.Onaytarihi,
             Hgsiparisdetay.Onaylayankullaniciid, Ok.Adi, Hgsiparisdetay.Malionaydurumu, Hgsiparisdetay.Malionaytarihi,
             Hgsiparisdetay.Malionaylayankullaniciid, MOK.Adi, Hgsiparisdetay.Kumassatinalmatalepkayitid,
             Hgsiparisdetay.Aksesuarsatinalmatalepkayitid, Hgsiparisdetay.Modelnotu, Hgsiparisdetay.Ormenotu,
             Hgsiparisdetay.Dokumanotu, Hgsiparisdetay.Boyahanenotu, Hgsiparisdetay.Aksesuarnotu, Hgsiparisdetay.Numunenotu,
             Hgsiparisdetay.Dikistalimati, Hgsiparisdetay.Dikistalimatresmi
      from HgSiparis
      Left Join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
      left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = HgSiparis.FirmaYetkiliId
      Left Join Firma On Firma.FirmaId = HgSiparis.FirmaId
      left join OdemeSekli on OdemeSekli.OdemeSekliId = HgSiparisDetay.OdemeSekliId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = HgSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = HgSiparis.SiparisGrup2Id
      Left Join TeslimSekli on TeslimSekli.TeslimSekliId = HgSiparis.TeslimSekliId
      Left Join Model on Model.ModelKayitID = HgSiparisDetay.ModelKayitID
      Left Join HgSatisTipiGetir on HgSatisTipiGetir.SatisTipiId = HgSiparisDetay.SatisTipiId
      Left Join Kullanici SAK on SAK.KullaniciId = HgSiparis.SiparisiAlanKullaniciId
      Left Join Kullanici IK  ON IK.KullaniciId = HgSiparisDetay.InsertKullaniciID
      Left Join Kullanici SK  ON SK.KullaniciId = HgSiparisDetay.KullaniciID
      Left Join Kullanici OK  ON OK.KullaniciId = HgSiparisDetay.OnaylayanKullaniciId
      Left Join Kullanici MOK ON MOK.KullaniciId = HgSiparisDetay.MaliOnaylayanKullaniciId
      where (:ARaporTipi = 1 and HgSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and HgSiparis.SiparisTarihi <= :ABitisSiparisTarihi or
             :ARaporTipi = 2 and HgSiparisDetay.TeslimTarihi >= :ABaslangicSiparisTarihi and HgSiparisDetay.TeslimTarihi <= :ABitisSiparisTarihi or
             :ARaporTipi = 0 and
                (HgSiparis.HgSiparisYil > :ABaslangicSiparisYil or (HgSiparis.HgSiparisYil = :ABaslangicSiparisYil and HgSiparis.HgSiparisId >= :ABaslangicSiparisId)) and
                (HgSiparis.HgSiparisYil < :ABitisSiparisYil or (HgSiparis.HgSiparisYil = :ABitisSiparisYil  and HgSiparis.HgSiparisId <= :ABitisSiparisId))
             ) and
            (Firma.FirmaId = :AFirmaId or :AFirmaId is null) and
            (HgSiparisDetay.Kapali = :AKapali or :AKapali is null) and
            (HgSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
            (HgSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
            (HgSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
            (HgSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null) and
            (HgSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
            (HgSiparisDetay.Sinif = :ASinif or :ASinif is null) and
            (HgSiparisDetay.ModelKayitId = :ModelKayitId or :ModelKayitId is Null) and
            (HgSiparisDetay.OnayDurumu = :ASipOnayDurumu or :ASipOnayDurumu is Null) and
            (HgSiparisDetay.MaliOnayDurumu = :AMaliOnayDurumu or :AMaliOnayDurumu is Null) and
            (HgSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null) into
    :HgSiparisKayitId, :HgSiparisYil, :HgSiparisId, :FirmaId, :FirmaAdi,
    :SiparisTarihi, :SozlesmeTarihi, :FirmaYetkiliId, :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi,
    :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisiVerenAdi, :Fason, :OdemeSekliId, :OdemeSekliAdi,
    :TeslimSekliId, :TeslimSekliAdi, :YurtDisi, :Notu, :BaskaFirmayaSatilabilir,
    :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
    :Hgsiparisdetayid, :Hgsipariskayitid, :Modelid, :ModelAdi, :Siparissirano, :Musterisiparisno, :Musterimodelno,
    :Referanssiparisno, :Sinif, :Hamstokbirimid, :Satisfiyati, :Dovizid, :Satistipiid, :SatisTipiAdi, :Teslimtarihi, :Vadegun,
    :Resimdosyaadi1, :Resimdosyayazdir1, :Resimdosyaadi2, :Resimdosyayazdir2,
    :Iptal, :Kapali, :Kapatmatarihi, :Onaydurumu, :Onaytarihi, :Onaylayankullaniciid, :OnaylayanKullaniciAdi, :Malionaydurumu,
    :Malionaytarihi, :Malionaylayankullaniciid, :MalionaylayankullaniciAdi, :Kumassatinalmatalepkayitid, :Aksesuarsatinalmatalepkayitid, :Modelnotu, :Ormenotu,
    :Dokumanotu, :Boyahanenotu, :Aksesuarnotu, :Numunenotu, :Dikistalimati, :Dikistalimatresmi do
  begin
    if (YurtDisi = 1) then YurtDisiStr = 'Yurt Dışı';
    else YurtDisiStr = '';

    if (Fason = 1) then FasonStr = 'Evet';
    else FasonStr = 'Hayır';

    if (BaskaFirmayaSatilabilir = 1) then BaskaFirmayaSatilabilirStr = 'Evet';
    else BaskaFirmayaSatilabilirStr = 'Hayır';

    HgSiparisIdStr = '(' || EgemenIntegerToStr(HgSiparisYil) || ') '  || EgemenIntegerToStr(HgSiparisId);

    if (Iptal = 1) then IptalStr = 'Evet';
    else IptalStr = 'Hayır';

    if (Kapali = 1) then KapaliStr = 'Evet';
    else KapaliStr = 'Hayır';

    OnayDurumuStr = '';
    if (OnayDurumu = 1) then
      OnayDurumuStr = 'Onaylandı';
    else if (OnayDurumu = 2) then
      OnayDurumuStr = 'Reddedildi';

    MaliOnayDurumuStr = '';
    if (MaliOnayDurumu = 1) then
      MaliOnayDurumuStr = 'Onaylandı';
    else if (MaliOnayDurumu = 2) then
      MaliOnayDurumuStr = 'Reddedildi';

    suspend;
  end
end
