CREATE PROCEDURE IMALATPARTIASAMADEGISTIR(
    AMAKINAID VARCHAR(10),
    APARTIASAMASTOKDETAYID INTEGER,
    AASAMADEGISIMSEBEBIID INTEGER,
    APERSONELID INTEGER)
RETURNS (
    STATUS INTEGER,
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    PARTISINIFID INTEGER,
    STR VARCHAR(50))
AS
declare variable SuAnkiSira         integer;
declare variable KS                 integer;
declare variable FPartiAsamalariId  integer;
declare variable PartiKayitId       integer;
declare variable GirisPersonelId    integer;
declare variable GirisTarihi        timeStamp;
declare variable CikisTarihi        timeStamp;
declare variable MakinaId           varchar(10);
declare variable EskiPartiAsamalariId  integer;
declare variable ImalatMakinalaraBolunebilir integer;
declare variable AsilMakinaID      MakinaKoduSize;
begin
  Select AsilMakinaID from Makina where Makina.MakinaId = :AMakinaId into :AsilMakinaID;
  if (AsilMakinaID is Not Null) then AMakinaID = :AsilMakinaID;

/* imalat makinalara bölünmüş ise yapılamaz*/
/* imalat bitmiş  ise veya önceki asama bitmeden baslayaniliyor ise, veya bu asama bitmeden sonraki baslayabiliyor ise yapılamaz*/
  if (AAsamaDegisimSebebiId is null) then
  begin
    Status = 10;
    Str = 'Değiştirme Sebebini Girmediniz..';
    suspend;
    exit;
  end
  else
  begin
    Select Count(*) from AsamaDegisimSebebi where AsamaDegisimSebebi.AsamaDegisimSebebiID = :AAsamaDegisimSebebiID into :KS;

    if (KS = 0) then
    begin
      Status = 11;
      Str = 'Girilen Asama Değiştirme Sebebi Kayıtlı Değil';
      suspend;
      exit;
    end

  end

  Select PartiAsamalari.Sira, PartiAsamaStokDetay.PartiSinifId, PartiAsamaStokDetay.PartikayitId, PartiAsamaStokDetay.PartiAsamalariId,
         PartiAsamaStokDetay.MakinaId, PartiAsamaStokDetay.GirisPersonelId, PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.CikisTarihi,
         EskiAsama.DegistirilebilecegiAsamaId, YeniAsama.Adi, EskiAsama.ImalatMakinalaraBolunebilir from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStokDetay.PartiAsamalariID
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Asama EskiAsama on EskiAsama.AsamaId = PartiAsamaStok.AsamaId
    left join Asama YeniAsama on YeniAsama.AsamaId = EskiAsama.DegistirilebilecegiAsamaId
    where PartiAsamaStokDetay.PartiAsamaStokDetayId = :APartiAsamaStokDetayId
    into :SuAnkiSira, :PartiSinifId, :PartikayitId, :EskiPartiAsamalariId, :MakinaId, :GirisPersonelId, :GirisTarihi, :CikisTarihi,
         :AsamaId, :AsamaAdi, :ImalatMakinalaraBolunebilir;

  if (SuAnkiSira is null) then
  begin
    Status = 2;
    Str = 'Kayıt bulunamadı';
    suspend;
    exit;
  end
  else if (CikisTarihi is not null) then
  begin
    Status = 3;
    Str = 'Makinada İmalatta Değil-3';
    suspend;
    exit;
  end
  else if (GirisTarihi is null) then
  begin
    Status = 4; /* Giriş yok */
    Str = 'Makinada İmalatta Değil-4';
    suspend;
    exit;
  end
  else if (MakinaId <> :AMakinaId) then
  begin
    Status = 5; /* Baska Makinada imalatta */
    Str = 'Parti Başka Makinada İmalatta-5';
    suspend;
    exit;
  end
  else if (AsamaID is null) then
  begin
    Status = 6;
    Str = 'Şu An Yapılan Aşamanın Değiştirilebileceği Aşama tanımlanmamış';
    suspend;
    exit;
  end
  else
  begin
    Select count(*) from PartiAsamaStok
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and
            PartiAsamaStok.PartiAsamalariId <> :EskiPartiAsamalariId and
            PartiAsamaStok.GirisTarihi is not null and
            PartiAsamaStok.CikisTarihi is null
      into :KS;

    if (KS > 0) then
    begin
      Status = 7; /* Partinin Başka Makinada imalatı var, bu işlem yapılamaz */
      Str = 'Parti Başka Makinada da İmalatta-7';
      suspend;
      exit;
    end

    Select count(*) from PartiAsamaStokDetay
      where PartiAsamaStokDetay.PartiSinifId = :PartiSinifId and
            PartiAsamaStokDetay.GirisTarihi is not null and
            PartiAsamaStokDetay.CikisTarihi is null and
            PartiAsamaStokDetay.MakinaId <> :AMakinaId
      into :KS;

    if (KS > 0) then
    begin
      Status = 8;
      Str = 'Partinin İmalatı Bölünmüş, Aşama Değiştirilemez-8';
      suspend;
      exit;
    end

    Select count(*) from PartiAsamaStokDetay
      where PartiAsamaStokDetay.PartiSinifId = :PartiSinifId and PartiAsamaStokDetay.PartiAsamalariID = :EskiPartiAsamalariID
      into :KS;

    if (KS > 1 or ImalatMakinalaraBolunebilir = 1) then
    begin
      Status = 9;
      Str = 'Partinin İmalatı Bölünmüş, Aşama Değiştirilemez-9';
      suspend;
      exit;
    end
  end

  Update PartiAsamaStokDetay Set MakinaId = Null, GirisPersonelId = Null, GirisTarihi = null
     where PartiAsamaStokDetay.PartiAsamaStokDetayID = :APartiAsamaStokDetayID;

  FPartiAsamalariId = gen_id(PartiAsamalari_Id_Gen, 1);
  insert into partiAsamalari(PartiAsamalariId, PartiKayitId, Sira, AsamaId, OtomasyonaDahilmi, SerbestAsama, AsamaDegistirildi, AsamaDegisimSebebiId, AsamaDegistirenPersonelId) values(:FPartiAsamalariId, :PartiKayitId, :SuAnkiSira, :AsamaId, 1, 1, 1, :AAsamaDegisimSebebiId, :APersonelId);

  insert into PartiAsamaStok(PartiAsamalariId, PartiKayitId, PartiSinifId, AsamayaDahil, AsamaId) values(:FPartiAsamalariId, :PartiKayitId, :PartiSinifId, 1, :AsamaId);

  Update PartiAsamaStokDetay Set MakinaId = :AMakinaId, GirisPersonelId = :GirisPersonelId, GirisTarihi = :GirisTarihi, GirisOtomatik = 1
      where PartiAsamaStokDetay.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId;

  Status = 1; /* basarılı */
  suspend;
end
