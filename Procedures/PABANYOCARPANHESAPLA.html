CREATE PROCEDURE PABANYOCARPANHESAPLA(
    APARTIASAMALARIID INTEGER,
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    ARECETEISEMRIYIL INTEGER,
    ARECETEISEMRIID INTEGER)
RETURNS (
    BANYONO INTEGER,
    DEGER DOUBLE PRECISION)
AS
declare variable IlaveProsesSebebiAdi Varchar(20);
declare variable SiviKostikKimyasalTipiID integer;
declare variable HidrojenPerOksitKimyasalTipiID integer;
declare variable SodaKimyasalTipiID integer;
declare variable AlkaliKimyasalTipiID integer;
declare variable FBanyo Double Precision;
declare variable FBanyoCarpanNo integer;
declare variable FLastBanyoCarpanNo integer;
declare variable FKimyasalTipiID integer;
declare variable FMakinaID varchar(10);
declare variable PompaSuyu double precision;
declare variable FKumastakiSuCarpan double precision;
declare variable FMakinaOtomasyonGrupID varchar(4);
declare variable FIlaveTankHacmi double precision;
declare variable FFloteOrani double precision;
declare variable FPartiMiktar double precision;
declare variable FKimyasalKatsayi double precision;
declare variable FKimyasalMiktar double precision;
declare variable FKumastakiSu double precision;
declare variable BoyaAsamasimi integer;
declare variable BoyaGrubuIlaveMiktarOrani double precision;
declare variable BoyaSuyu double precision;
declare variable KumasKilosuEtkili integer;
declare variable SiraNo integer;
declare variable EgemenFirmaID integer;
declare variable TuzKimyasalTipiID integer;
begin
  if (AReceteId is not null) then
  begin
    FLastBanyoCarpanNo = -1;
    SiraNo = 0;

    Select EgemenFirmaID, SiviKostikKimyasalTipiID, HidrojenPerOksitKimyasalTipiID, SodaKimyasalTipiID, BoyaGrubuIlaveMiktarOrani, AlkaliKimyasalTipiID, TuzKimyasalTipiID from sistem where sistemId = 1
      into :EgemenFirmaID, :SiviKostikKimyasalTipiID, :HidrojenPerOksitKimyasalTipiID, :SodaKimyasalTipiID, :BoyaGrubuIlaveMiktarOrani, :AlkaliKimyasalTipiID, :TuzKimyasalTipiID ;

    Select first 1 Makina.MakinaId, Makina.PompaSuyu, Makina.MakinaOtomasyonGrupID, Makina.IlaveTankHacmi, Makina.FloteOrani from BolumPlanlama
    left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
    left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
    left join Makina on Makina.MakinaID = UretimPlanlama.MakinaId
    where BolumPlanlama.PartiAsamalariId = :APartiAsamalariId
    into :FMakinaID, :PompaSuyu, :FMakinaOtomasyonGrupID, :FIlaveTankHacmi, :FFloteOrani;

    Select first 1 ElyafCinsiMOGruplari.KumastakiSu, MakinaOtomasyonGrup.KumasKilosuEtkili from PArtiStok
    Left join Ham on Ham.HamID = PartiStok.HamID
    Left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    Left join ElyafCinsiMOGruplari on ElyafCinsiMOGruplari.ElyafCinsiID = IplikCinsi.ElyafCinsiID
    Left join MakinaOtomasyonGrup on MakinaOtomasyonGrup.MakinaOtomasyonGrupID = ElyafCinsiMOGruplari.MakinaOtomasyonGrupID
    where PartiKayitID = :APartiKayitId and ElyafCinsiMOGruplari.MakinaOtomasyonGrupID = :FMakinaOtomasyonGrupID
    order by PartiStok.Miktar Desc
    into :FKumastakiSuCarpan, :KumasKilosuEtkili;

    if (AReceteIsEmriID is Not Null and AReceteIsEmriYil is Not Null) then
    begin
      Select Sum(PartiAsamaStok.GirisKilo) PartiKilo From ReceteIsEmriDetay
      Left join PartiAsamaStok on PartiAsamaStok.PartiKayitID = ReceteIsEmriDetay.PartiKayitID and PartiAsamaStok.PartiAsamalariId = ReceteIsEmriDetay.PartiAsamalariId
      where ReceteIsEmriDetay.ReceteIsEmriYil = :AReceteIsEmriYil  and ReceteIsEmriDetay.ReceteIsEmriID = :AReceteIsEmriID  into :FPartiMiktar;
    end
    else Select Sum(PartiStok.Miktar) from PArtiStok where PartiKayitID = :APartiKayitId into :FPartiMiktar;

    if (EgemenFirmaID =128 ) then
     Select partiasamalari.islaklikorani from partiasamalari where partiasamalari.partiasamalariid =:APARTIASAMALARIID into :FKumastakiSuCarpan;


    FKumastakiSu = FPartiMiktar * FKumastakiSuCarpan;
    if (EgemenFirmaID <> 128 ) then FKumastakiSu = FKumastakiSu / 100.0;

    FKimyasalMiktar = 0;
    BoyaSuyu = 0;

    for Select ReceteDetay.Banyo, ReceteAsama.BanyoCarpanNo, Urun.KimyasalTipiID, KimyasalTipi.Katsayi, ReceteAsama.BoyaAsamasimi from ReceteDetay
    Left Join ReceteAsama on ReceteAsama.ReceteAsamaID = ReceteDetay.ReceteAsamaID
    left join urun on Urun.urunid = ReceteDetay.UrunID
    left join KimyasalTipi on KimyasalTipi.KimyasalTipiID = Urun.KimyasalTipiID
    Where ReceteID = :AReceteId Order By Sira
    into :FBanyo, :FBanyoCarpanNo, :FKimyasalTipiID, :FKimyasalKatsayi, :BoyaAsamasimi do
    begin
      if (FLastBanyoCarpanNo = -1 ) then FLastBanyoCarpanNo = FBanyoCarpanNo;
      if (FLastBanyoCarpanNo <> FBanyoCarpanNo and FBanyoCarpanNo is not Null) then
      begin
        BanyoNo = FLastBanyoCarpanNo;
        Deger = Round(FKimyasalMiktar + FKumasTakiSu + PompaSuyu + BoyaSuyu);
        SiraNo = SiraNo + 1;

        if (AReceteIsEmriID is Not Null and AReceteIsEmriYil is Not Null) then
        begin
          if (Deger is Null) then Deger = 0;

          if (SiraNo = 1) then
            update ReceteIsEmri set BanyoCarpani = :Deger where ReceteIsEmri.ReceteIsEmriYil = :AReceteIsEmriYil  and ReceteIsEmri.ReceteIsEmriID = :AReceteIsEmriID;
          else if (SiraNo = 2) then
            update ReceteIsEmri set BanyoCarpani2 = :Deger where ReceteIsEmri.ReceteIsEmriYil = :AReceteIsEmriYil  and ReceteIsEmri.ReceteIsEmriID = :AReceteIsEmriID;
          else if (SiraNo = 3) then
            update ReceteIsEmri set BanyoCarpani3 = :Deger where ReceteIsEmri.ReceteIsEmriYil = :AReceteIsEmriYil  and ReceteIsEmri.ReceteIsEmriID = :AReceteIsEmriID;
          else if (SiraNo = 4) then
            update ReceteIsEmri set BanyoCarpani4 = :Deger where ReceteIsEmri.ReceteIsEmriYil = :AReceteIsEmriYil  and ReceteIsEmri.ReceteIsEmriID = :AReceteIsEmriID;
        end
        else
        begin
          if (SiraNo = 1) then
            update PartiAsamalari set BanyoCarpani = :Deger where PartiAsamalariId = :APartiAsamalariId;
          else if (SiraNo = 2) then
            update PartiAsamalari set BanyoCarpani2 = :Deger where PartiAsamalariId = :APartiAsamalariId;
          else if (SiraNo = 3) then
            update PartiAsamalari set BanyoCarpani3 = :Deger where PartiAsamalariId = :APartiAsamalariId;
          else if (SiraNo = 4) then
            update PartiAsamalari set BanyoCarpani4 = :Deger where PartiAsamalariId = :APartiAsamalariId;
        end


        FLastBanyoCarpanNo = FBanyoCarpanNo;
        FKimyasalMiktar = 0;
        BoyaSuyu = 0;
        Suspend;
      end

      if (BoyaAsamasimi = 1) then
        BoyaSuyu =  (FIlaveTankHacmi * (BoyaGrubuIlaveMiktarOrani / 100) );

      if (FKimyasalTipiID = SodaKimyasalTipiID) then
      begin
        FKimyasalMiktar =  FKimyasalMiktar + ((FPartiMiktar * FFloteOrani * FBanyo) / 1000);
        if (FKimyasalTipiID = SodaKimyasalTipiID or (EgemenFirmaID = 128 and (FKimyasalTipiID = TuzKimyasalTipiID or FKimyasalTipiID = SiviKostikKimyasalTipiID ))) then
        begin
          FKimyasalMiktar =  FKimyasalMiktar + ((FPartiMiktar * FFloteOrani * FBanyo) / 1000);

         end
         else if (FKimyasalTipiID = SiviKostikKimyasalTipiID or FKimyasalTipiID = HidrojenPerOksitKimyasalTipiID or FKimyasalTipiID = AlkaliKimyasalTipiID) then
         begin
          FKimyasalMiktar =  FKimyasalMiktar + (FKimyasalKatsayi * FIlaveTankHacmi);
         end

       end
    end
    BanyoNo = FLastBanyoCarpanNo;
    Deger = Round(FKimyasalMiktar + FKumasTakiSu + PompaSuyu + BoyaSuyu);
    SiraNo = SiraNo + 1;

    if (AReceteIsEmriID is Not Null and AReceteIsEmriYil is Not Null) then
    begin
      if (Deger is Null) then Deger = 0;

      if (SiraNo = 1) then
        update ReceteIsEmri set BanyoCarpani = :Deger where ReceteIsEmri.ReceteIsEmriYil = :AReceteIsEmriYil  and ReceteIsEmri.ReceteIsEmriID = :AReceteIsEmriID;
      else if (SiraNo = 2) then
        update ReceteIsEmri set BanyoCarpani2 = :Deger where ReceteIsEmri.ReceteIsEmriYil = :AReceteIsEmriYil  and ReceteIsEmri.ReceteIsEmriID = :AReceteIsEmriID;
      else if (SiraNo = 3) then
        update ReceteIsEmri set BanyoCarpani3 = :Deger where ReceteIsEmri.ReceteIsEmriYil = :AReceteIsEmriYil  and ReceteIsEmri.ReceteIsEmriID = :AReceteIsEmriID;
      else if (SiraNo = 4) then
        update ReceteIsEmri set BanyoCarpani4 = :Deger where ReceteIsEmri.ReceteIsEmriYil = :AReceteIsEmriYil  and ReceteIsEmri.ReceteIsEmriID = :AReceteIsEmriID;
    end
    else
    begin
      if (SiraNo = 1) then
        update PartiAsamalari set BanyoCarpani = :Deger where PartiAsamalariId = :APartiAsamalariId;
      else if (SiraNo = 2) then
        update PartiAsamalari set BanyoCarpani2 = :Deger where PartiAsamalariId = :APartiAsamalariId;
      else if (SiraNo = 3) then
        update PartiAsamalari set BanyoCarpani3 = :Deger where PartiAsamalariId = :APartiAsamalariId;
      else if (SiraNo = 4) then
        update PartiAsamalari set BanyoCarpani4 = :Deger where PartiAsamalariId = :APartiAsamalariId;
    end

    Suspend;
  end
end
