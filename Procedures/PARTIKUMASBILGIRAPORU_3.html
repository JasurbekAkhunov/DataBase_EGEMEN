CREATE PROCEDURE PARTIKUMASBILGIRAPORU_3(
    ARAPORTIPI INTEGER,
    AASAMAID INTEGER,
    ABASLANGICINSERTTARIHI TIMESTAMP,
    ABITISINSERTTARIHI TIMESTAMP,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER)
RETURNS (
    PARTIKUMASKBILGIID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    ASAMAID INTEGER,
    EN VARCHAR(10),
    GRAMAJ VARCHAR(10),
    CEKMEZLIKEN DOUBLE PRECISION,
    CEKMEZLIKBOY DOUBLE PRECISION,
    DONME VARCHAR(10),
    HIZ VARCHAR(20),
    SICAKLIK VARCHAR(20),
    ACIKLAMA VARCHAR(100),
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    SONEN VARCHAR(10),
    ASAMAADI VARCHAR(40),
    INSERTKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40),
    CEKMEZLIKENSTR VARCHAR(30),
    CEKMEZLIKBOYSTR VARCHAR(30))
AS
begin
  if (ARaporTipi = 0) then
  begin
    for
    SELECT PartiKumasKBilgi.PartiKumasKBilgiId, PartiKumasKBilgi.PartiKayitId, PartiKumasKBilgi.PartiSinifId, PartiKumasKBilgi.AsamaID, PartiKumasKBilgi.En,
           PartiKumasKBilgi.Gramaj, PartiKumasKBilgi.CekmezlikEn, PartiKumasKBilgi.CekmezlikBoy, PartiKumasKBilgi.Donme, PartiKumasKBilgi.Hiz,  PartiKumasKBilgi.Sicaklik,
           PartiKumasKBilgi.Aciklama, PartiKumasKBilgi.InsertKullaniciId, PartiKumasKBilgi.InsertTarihi, PartiKumasKBilgi.KullaniciId, PartiKumasKBilgi.DegisimTarihi,
           PartiKumasKBilgi.SonEn, Asama.Adi AsamaAdi, IK.Adi InsertKullaniciAdi, SK.Adi KullaniciAdi,
           cast (case when (PartiKumasKBilgi.CekmezlikEn is null) then ''
                 else format_float('0.##', PartiKumasKBilgi.CekmezlikEn)
                 end as varchar (30)) CekmezlikEnStr,
          cast (case when (PartiKumasKBilgi.CekmezlikBoy is null) then ''
                 else format_float('0.##', PartiKumasKBilgi.CekmezlikBoy)
                 end as varchar (30)) CekmezlikBoyStr FROM Parti
    left join PartiKumasKBilgi on PartiKumasKBilgi.PartiKayitId = Parti.PartiKayitId
    left join Asama ON Asama.AsamaID = PartiKumasKBilgi.AsamaID
    left join Kullanici IK ON IK.KullaniciId = PartiKumasKBilgi.InsertKullaniciId
    left join Kullanici SK ON SK.KullaniciId = PartiKumasKBilgi.KullaniciId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where  PartiKumasKBilgi.PartiKumasKBilgiId is not null and
           (Parti.PartiYil > :ABaslangicPartiYil or (Parti.PartiYil = :ABaslangicPartiYil and Parti.PartiId >= :ABaslangicPartiId)) and
           (Parti.PartiYil < :ABitisPartiYil or (Parti.PartiYil = :ABitisPartiYil  and Parti.PartiId <= :ABitisPartiId)) and
           (Asama.AsamaId = :AAsamaId or :AAsamaId is null) and
           (:ABaslangicBoyaSiparisYil = 0 or
            ((BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
            (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId))))
    ORDER BY PartiKumasKBilgiId
    into  :PartiKumasKBilgiId, :PartiKayitId, :PartiSinifId, :AsamaID, :En, :Gramaj, :CekmezlikEn, :CekmezlikBoy,
          :Donme, :Hiz, :Sicaklik, :Aciklama, :InsertKullaniciId, :InsertTarihi,  :KullaniciId, :DegisimTarihi,
          :SonEn, :AsamaAdi, :InsertKullaniciAdi, :KullaniciAdi, :CekmezlikEnStr, :CekmezlikBoyStr
       do
       begin
         suspend;
       end
    end
  else
   begin
   For
   SELECT PartiKumasKBilgi.PartiKumasKBilgiId, PartiKumasKBilgi.PartiKayitId, PartiKumasKBilgi.PartiSinifId, PartiKumasKBilgi.AsamaID, PartiKumasKBilgi.En,
           PartiKumasKBilgi.Gramaj, PartiKumasKBilgi.CekmezlikEn, PartiKumasKBilgi.CekmezlikBoy, PartiKumasKBilgi.Donme, PartiKumasKBilgi.Hiz,  PartiKumasKBilgi.Sicaklik,
           PartiKumasKBilgi.Aciklama, PartiKumasKBilgi.InsertKullaniciId, PartiKumasKBilgi.InsertTarihi, PartiKumasKBilgi.KullaniciId, PartiKumasKBilgi.DegisimTarihi,
           PartiKumasKBilgi.SonEn, Asama.Adi AsamaAdi, IK.Adi InsertKullaniciAdi, SK.ADI KullaniciAdi,
           cast (case when (PartiKumasKBilgi.CekmezlikEn is null) then ''
                 else format_float('0.##', PartiKumasKBilgi.CekmezlikEn)
                 end as varchar (30)) CekmezlikEnStr,
          cast (case when (PartiKumasKBilgi.CekmezlikBoy is null) then ''
                 else format_float('0.##', PartiKumasKBilgi.CekmezlikBoy)
                 end as varchar (30)) CekmezlikBoyStr FROM Parti
    left join PartiKumasKBilgi on PartiKumasKBilgi.PartiKayitId = Parti.PartiKayitId
    left join Asama ON Asama.AsamaID = PartiKumasKBilgi.AsamaID
    left join Kullanici IK ON IK.KullaniciId = PartiKumasKBilgi.InsertKullaniciId
    left join Kullanici SK ON SK.KullaniciId = PartiKumasKBilgi.KullaniciId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where  PartiKumasKBilgi.PartiKumasKBilgiId is not null and
           (PartiKumasKBilgi.InsertTarihi >= :ABaslangicInsertTarihi and PartiKumasKBilgi.InsertTarihi <= :ABitisInsertTarihi ) and
           (Asama.AsamaId = :AAsamaId or :AAsamaId is null) and
           (:ABaslangicBoyaSiparisYil = 0 or
           ((BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
           (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId))))
    ORDER BY PartiKumasKBilgiId
    into  :PartiKumasKBilgiId, :PartiKayitId, :PartiSinifId, :AsamaID, :En, :Gramaj, :CekmezlikEn, :CekmezlikBoy,
          :Donme, :Hiz, :Sicaklik, :Aciklama, :InsertKullaniciId, :InsertTarihi,  :KullaniciId, :DegisimTarihi,
          :SonEn, :AsamaAdi, :InsertKullaniciAdi, :KullaniciAdi, :CekmezlikEnStr, :CekmezlikBoyStr
    do
    begin
       suspend;
    end
   end
end
