CREATE PROCEDURE BARKODLAPARTILEMEYAP(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    ABARKODID VARCHAR(30),
    AKULLANICIID INTEGER)
AS
declare variable PartiStokPartilemeId integer;
declare variable HamStokMasterTopId   integer;
declare variable HamStokMasterId      integer;
declare variable KS integer;
declare variable RafDetayId    integer;
declare variable BoyaIrsaliyeDetayId    integer;
declare variable RezervKilo    Double Precision;
declare variable OkutulanToplamKilo    Double Precision;
declare variable Kilo    Double Precision;
declare variable Metre   Double Precision;
declare variable Dara    Double Precision;
declare variable OkutulanTopKilo    Double Precision;
begin
  if (Char_Length(ABarkodId) < 1) then exception BarkodHatali1;

  Select HamStokMasterTopId, BoyaIrsaliyeDetayId, HamStokMasterId, Kilo, Dara, Metre From HamStokMasterTop where HamStokMasterTop.BarkodId = :ABarkodId into :HamStokMasterTopId, :BoyaIrsaliyeDetayId, :HamStokMasterId, :Kilo, :Dara, :Metre;

  if (BoyaIrsaliyeDetayId is not null) then exception BuTartiSevkEdilmis;
  if (HamStokMasterTopId is null) then exception BarkodHatali2;

  Select PartiStokPartileme.PartiStokPartilemeId from PartiStokPartileme
    where PartiStokPartileme.HamStokMasterTopId = :HamStokMasterTopId
  into :PartiStokPartilemeId;

  if (PartiStokPartilemeId is not null) then exception KayitHatasi ' Bu top zaten partilenmiş.';

  Select Count(*), Sum(BoyaSiparisRezerv.Kilo) from PartiStokDetay
    left join BoyaSiparisRezerv on BoyaSiparisRezerv.BoyaSiparisRezervId = PartiStokDetay.BoyaSiparisRezervId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
    where PartiStokDetay.PartiSinifId = :APartiSinifId and HamStokDetay.HamStokMasterId = :HamStokMasterId
    Group By HamStokDetay.HamStokMasterId
  into :KS, :RezervKilo;

  if (KS <= 0 or KS is Null) then exception KayitHatasi ' okutulan top rezerv arasında yok';
 /*
  Select Sum(PartiStokPartileme.Kilo) from PartiStokPartileme
    Left Join HamStokMasterTop on HamStokMasterTop.HamStokMasterTopId = PartiStokPartileme.HamStokMasterTopId
    where PartiSinifId = :APartiSinifId and HamStokMasterTop.HamStokMasterId = :HamStokMasterId
  into :OkutulanTopKilo;

  if (OkutulanTopKilo is Null) then OkutulanTopKilo = 0;

  if (OkutulanTopKilo + Kilo > RezervKilo) then exception KayitHatasi ' okutulan topların toplamı rezerv miktarını aşamaz!';
*/

  Insert Into PartiStokPartileme(PartiKayitId, PartiSinifId, Kilo, Metre, InsertKullaniciId, KullaniciId, HamStokMasterTopId)
    Values (:APartiKayitId, :APartiSinifId, :Kilo, :Metre, :AKullaniciId, :AKullaniciId, :HamStokMasterTopId);

end
