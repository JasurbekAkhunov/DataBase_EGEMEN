CREATE PROCEDURE IPLIKBARKODLUSATISYAP(
    AIPLIKSTOKID INTEGER,
    ANUMARATOR INTEGER,
    AIPLIKTALEPDETAYID INTEGER,
    AIPLIKIRSALIYEID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable KayitSayisi    integer;
declare variable FIplikStokId   integer;
declare variable FIplikIrsaliyeId integer;
declare variable FHamId           integer;
declare variable FTalepHamId      integer;
declare variable FIplikTalepKayitId integer;
declare variable FKayitIplikIrsaliyeDetayId integer;
declare variable FMiktar                Numeric(12,3);
declare variable FNetMiktar             Numeric(12,3);
declare variable FBobinSayisi integer;
declare variable FCikisIplikIrsaliyeDetayId integer;

declare variable FRenkID                integer;
declare variable FLot                  varchar(30);
declare variable FMelanjKodu           varchar(20);
declare variable FFlamKodu             varchar(20);
declare variable FKonikRengiId          integer;
declare variable FKonikCinsiId          integer;
declare variable FSarimSekliId          integer;
declare variable FBobinGramaji          Numeric(12,3);
declare variable FAmbalajId             integer;
declare variable FHarmanKayitId         integer;
declare variable HareketTipi integer;
declare variable FIplikDepoID           varchar(10);
declare variable FIplikIsletmeId        integer;
declare variable FUretimIplikTalepDetayId integer;
declare variable FUretimBoyaSiparisKayitId integer;
declare variable FTamirGirisKayitId        integer;
declare variable MaxNumarator              integer;
declare variable HamTipiId                 integer;
declare variable hstIplik       integer;
declare variable SiparisFirmaId varchar(20);
declare variable IrsaliyeFirmaId varchar(20);
declare variable OrmeFirmaKodumuz varchar(20);
declare variable FirmaKodumuz varchar(20);
declare variable KdvOrani     Double Precision;
declare variable TalepMiktari Double Precision;
declare variable SatisMiktari Double Precision;
declare variable IplikUretimMiktarTolerans Double Precision;
declare variable TalepMiktariAsanSatisaIzinVer integer;
declare variable Tamir integer;
begin
  Select OrmeFirmaKodumuz, FirmaKodumuz, TalepMiktariAsanSatisaIzinVer from sistem where sistemid = 1 into :OrmeFirmaKodumuz, :FirmaKodumuz, :TalepMiktariAsanSatisaIzinVer;

  Select hstIplik from HamTipiGetir into :hstIplik;

  Select shtSatis from IplikStokHareketTipiGetir into :HareketTipi;

  if (AIplikIrsaliyeId is not null) then
  begin
    Select Max(Numarator) from IplikStok where IplikStok.IplikStokID = :AIplikStokId into :MaxNumarator;
  end
  else MaxNumarator = ANumarator;

  if (MaxNumarator is null) then MaxNumarator = -1;

/*  Select DovizId from IplikIrsaliye where IplikIrsaliye.IplikIrsaliyeID = :AIplikIrsaliyeId into :DovizId; */

  Select IplikStok.IplikStokId, IplikStok.Miktar, IplikStok.NetMiktar, IplikStok.BobinSayisi, IplikStok.CikisIplikIrsaliyeDetayId, IplikStok.HamId,
         IplikIrsaliyeDetay.RenkId, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu,
         IplikIrsaliyeDetay.KonikRengiId, IplikIrsaliyeDetay.KonikCinsiId,
         IplikIrsaliyeDetay.BobinGramaji, IplikIrsaliyeDetay.AmbalajId, IplikStok.IplikDepoId, IplikStok.IplikIsletmeId,
         IplikUretim.IplikTalepDetayId, IplikTalepDetay.BoyaSiparisKayitId, IplikLot.HarmanKayitId,
         IplikIrsaliyeDetay.TamirGirisKayitId, IplikIrsaliyeDetay.SarimSekliId, Ham.KdvOrani, IplikStok.Tamir from IplikStok
    left join IplikUretim On IplikUretim.IplikStokID = IplikStok.IplikStokId and IplikUretim.Numarator = 1
    left join IplikLot On IplikLot.IplikLotID = IplikUretim.IplikLotId
    left join IplikTalepDetay On IplikTalepDetay.IplikTalepDetayID = IplikUretim.IplikTalepDetayId
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = IplikStok.IplikIrsaliyeDetayId
    left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
    where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator
    into :FIplikStokId, :FMiktar, :FNetMiktar, :FBobinSayisi, :FCikisIplikIrsaliyeDetayId, :FHamId, :FRenkId, :FLot, :FMelanjKodu, :FFlamKodu, :FKonikRengiId, :FKonikCinsiId,
         :FBobinGramaji, :FAmbalajId, :FIplikDepoID, :FIplikIsletmeId, :FUretimIplikTalepDetayId, :FUretimBoyaSiparisKayitId, :FHarmanKayitId,
         :FTamirGirisKayitId, :FSarimSekliId, :KdvOrani, :Tamir;

  if (FIplikStokId is null) then exception IplikStokKaydiBulunamadi;

  if (AIplikIrsaliyeId is not null) then
  begin
    if (AIplikTalepDetayID is not null) then
    begin
      Select HamId from IplikTalepDetay where IplikTalepDetay.IplikTalepDetayId = :AIplikTalepDetayID into :FTalepHamId;
      if (FTalepHamId <> FHamId) then exception OkutulanStokTalebeUymuyor;

      if (FUretimIplikTalepDetayId <> AIplikTalepDetayID and FUretimBoyaSiparisKayitId is not null) then exception BuStokBaskaSipariseAit;
    end
    else
    begin
      if (FUretimIplikTalepDetayId is not null) then AIplikTalepDetayID = FUretimIplikTalepDetayId;
      else if (HamTipiID = hstIplik) then exception TalepSecmedenSevkYapilamaz;
    end
  end

  Select IplikTalepDetay.IplikTalepKayitId, BoyaSiparis.FirmaId, IplikTalepDetay.Miktar from IplikTalepDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = IplikTalepDetay.BoyaSiparisKayitID
    where IplikTalepDetay.IplikTalepDetayId = :AIplikTalepDetayID
    into :FIplikTalepKayitId, :SiparisFirmaId, :TalepMiktari;


  if (TalepMiktariAsanSatisaIzinVer = 0 and AIplikTalepDetayID is not null) then
  begin
/* tamir edilip baska talep icin satılıyor ise tamir bilgisine bakmadan kontrol ediyoruız, aynı talep icin sevk ediliyorsa tamir sevkimi diye bakıyoruz */
    if (FUretimIplikTalepDetayId <> AIplikTalepDetayID) then
    begin
      Select Sum(NetMiktar) from IplikIrsaliyeDetay
        left Join IplikLot on IplikLot.IplikLotID = IplikIrsaliyeDetay.IplikLotID
        where IplikIrsaliyeDetay.IplikTalepDetayID = :AIplikTalepDetayID and IplikIrsaliyeDetay.HareketTipi = :HareketTipi and IplikLot.Tamir = 0
        into :SatisMiktari;
    end
    else
    begin
      Select Sum(NetMiktar) from IplikIrsaliyeDetay
        left Join IplikLot on IplikLot.IplikLotID = IplikIrsaliyeDetay.IplikLotID
        where IplikIrsaliyeDetay.IplikTalepDetayID = :AIplikTalepDetayID and IplikIrsaliyeDetay.HareketTipi = :HareketTipi and IplikLot.Tamir = 0 and IplikIrsaliyeDetay.Tamir = :Tamir
        into :SatisMiktari;
    end

    if (SatisMiktari is null) then SatisMiktari = 0;

    SatisMiktari = SatisMiktari + FMiktar;

    Select Min(Miktar) from IplikUretimTolerans
      where :TalepMiktari > IplikUretimTolerans.BaslangicMiktar and :TalepMiktari <= IplikUretimTolerans.BitisMiktar
      into :IplikUretimMiktarTolerans;

    if (IplikUretimMiktarTolerans is null) then IplikUretimMiktarTolerans = 0;

    if (SatisMiktari > (TalepMiktari + IplikUretimMiktarTolerans)) then exception TalepMiktariniAstiniz;
  end

  if (SiparisFirmaId is not null) then
  begin
     Select FirmaId from IplikIrsaliye where IplikIrsaliye.IplikIrsaliyeID = :AIplikIrsaliyeId into :IrsaliyeFirmaId;

     if (IrsaliyeFirmaId <> OrmeFirmaKodumuz and IrsaliyeFirmaId <> FirmaKodumuz) then
     if (IrsaliyeFirmaId <> SiparisFirmaID) then exception SevkFirmaSipFirmayaUygunDegil;
  end

  if (AIplikIrsaliyeId is Not null and FCikisIplikIrsaliyeDetayId is not null) then exception BuStokDahaOnceStoktanCikmis;

  if (AIplikIrsaliyeId is not null) then
  begin
    Select max(IplikIrsaliyeDetayId) from IplikIrsaliyeDetay
      where IplikIrsaliyeDetay.IplikIrsaliyeId = :AIplikIrsaliyeId and
            IplikIrsaliyeDetay.HamId = :FHamId and
            IplikIrsaliyeDetay.IplikDepoId = :FIplikDepoId and
            IntKarsilastir(IplikIrsaliyeDetay.IplikTalepDetayId, :AIplikTalepDetayId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.IplikIsletmeId, :FIplikIsletmeId) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.Lot, :FLot) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.MelanjKodu, :FMelanjKodu) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.FlamKodu, :FFlamKodu) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.AmbalajId, :FAmbalajID) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.HarmanKayitId, :FHarmanKayitId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.RenkId, :FRenkId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.KonikRengiId, :FKonikRengiId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.KonikCinsiId, :FKonikCinsiId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.SarimSekliId, :FSarimSekliId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.TamirGirisKayitId, :FTamirGirisKayitId) = 1 and
            IplikIrsaliyeDetay.BobinGramaji = :FBobinGramaji
      into :FKayitIplikIrsaliyeDetayId;

    if (FKayitIplikIrsaliyeDetayId is not Null) then
    begin
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi + 1, Miktar = Miktar + :FMiktar, NetMiktar = NetMiktar + :FNetMiktar, BobinSayisi = BobinSayisi + :FBobinSayisi, KullaniciID = :AKullaniciId, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :FKayitIplikIrsaliyeDetayId;
    end
    else
    begin
      FKayitIplikIrsaliyeDetayId = gen_id(IplikIrsaliyeDetay_Id_Gen, 1);

      insert into IplikIrsaliyeDetay(IplikIrsaliyeDetayId, HareketTipi, IplikIsletmeId, IplikDepoId, IplikIrsaliyeId, NeredenUpdate, IplikTalepKayitId, IplikTalepDetayId,
                                     TamirGirisKayitId, HamId, RenkId, HarmanKayitId, Lot, MelanjKodu, FlamKodu, SarimSekliId, KonikCinsiId, KonikRengiId, AmbalajId, BobinGramaji, KapAdedi, Miktar, NetMiktar, BobinSayisi,
                                     Kdv, InsertKullaniciId, KullaniciId)
                    values(:FKayitIplikIrsaliyeDetayId, :HareketTipi, :FIplikIsletmeId,  :FIplikDepoId, :AIplikIrsaliyeId, 1, :FIplikTalepKayitId, :AIplikTalepDetayId,
                            :FTamirGirisKayitId, :FHamId, :FRenkId, :FHarmanKayitId, :FLot, :FMelanjKodu, :FFlamKodu,  :FSarimSekliId, :FKonikCinsiId, :FKonikRengiId, :FAmbalajId, :FBobinGramaji, 1, :FMiktar, :FNetMiktar, :FBobinSayisi,
                            :KdvOrani, :AKullaniciId, :AKullaniciId);

    end
    Update IplikStok Set CikisIplikIrsaliyeDetayId = :FKayitIplikIrsaliyeDetayId where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;
  end
  else
  begin
    if (FCikisIplikIrsaliyeDetayId is Null) then exception BuStokSatisiZatenYapilmamis;

    Select count(*) from IplikStok where IplikStok.CikisIplikIrsaliyeDetayId = :FCikisIplikIrsaliyeDetayId into :KayitSayisi;

    Update IplikStok Set CikisIplikIrsaliyeDetayId = Null where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;

    if (KayitSayisi = 1) then delete from IplikIrsaliyeDetay where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :FCikisIplikIrsaliyeDetayId;
    else
    begin
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi - 1, Miktar = Miktar - :FMiktar, NetMiktar = NetMiktar - :FNetMiktar, BobinSayisi = BobinSayisi - :FBobinSayisi, KullaniciID = :AKullaniciId, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :FCikisIplikIrsaliyeDetayId;
    end
  end
end
