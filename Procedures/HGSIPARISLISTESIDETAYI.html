CREATE PROCEDURE HGSIPARISLISTESIDETAYI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKAPALI INTEGER,
    AMODELID MODELKODUSIZE,
    AMUSTERISIPARISNO VARCHAR(50))
RETURNS (
    HGSIPARISDETAYID INTEGER,
    HGSIPARISKAYITID INTEGER,
    MODELID MODELKODUSIZE,
    SIPARISSIRANO INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIMODELNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    SINIF VARCHAR(1),
    HAMSTOKBIRIMID INTEGER,
    SATISFIYATI DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    SATISTIPIID SMALLINT,
    SATISTIPISTR VARCHAR(20),
    TESLIMTARIHI TIMESTAMP,
    VADEGUN INTEGER,
    NOTU MEMO,
    RESIMDOSYAADI1 VARCHAR(255),
    RESIMDOSYAYAZDIR1 EGEBOOLEAN,
    RESIMDOSYAADI2 VARCHAR(255),
    RESIMDOSYAYAZDIR2 EGEBOOLEAN,
    IPTAL EGEBOOLEAN,
    IPTALSTR VARCHAR(5),
    KAPALI EGEBOOLEAN,
    KAPALISTR VARCHAR(5),
    KAPATMATARIHI TIMESTAMP,
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(40),
    ONAYDURUMU EGEBOOLEAN,
    ONAYDURUMUSTR VARCHAR(20),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI ADI40,
    MALIONAYDURUMU SMALLINT,
    MALIONAYDURUMUSTR VARCHAR(20),
    MALIONAYTARIHI TIMESTAMP,
    MALIONAYLAYANKULLANICIID INTEGER,
    MALIONAYLAYANKULLANICIADI ADI40,
    LINKISLEMI INTEGER,
    DEGISIMLOG BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    INSERTKULLANICIADI ADI40,
    KULLANICIID INTEGER,
    KULLANICIADI ADI40,
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select HGSiparisDetay.HGSiparisDetayid, HGSiparisDetay.HGSiparisKayitId, Model.ModelId, HGSiparisDetay.SiparisSiraNo, HGSiparisDetay.MusteriSiparisNo,
    HGSiparisDetay.MusteriModelNo, HGSiparisDetay.ReferansSiparisNo, HGSiparisDetay.Sinif, HGSiparisDetay.HamStokBirimId, HGSiparisDetay.SatisFiyati, HGSiparisDetay.DovizId,
    HGSiparisDetay.SatisTipiID, Decode(HGSiparisDetay.SatisTipiID, 0, 'Tek Ürün', 1, 'Asorti Seti', 2, 'Düzine', ''), HGSiparisDetay.TeslimTarihi, HGSiparisDetay.VadeGun, HGSiparisDetay.Notu,
    HGSiparisDetay.ResimDosyaAdi1, HGSiparisDetay.ResimDosyayazdir1, HGSiparisDetay.ResimDosyaAdi2, HGSiparisDetay.ResimDosyaYazdir2, HGSiparisDetay.Iptal, Decode(HGSiparisDetay.Iptal, 1, 'Evet', ''), HGSiparisDetay.Kapali,
    Decode(HGSiparisDetay.Kapali, 1, 'Evet', ''), HGSiparisDetay.KapatmaTarihi, HGSiparisDetay.OdemeSekliId, HGSiparisDetay.OnayDurumu, Decode(HGSiparisDetay.OnayDurumu, 0, 'Onay Bekleniyor', 1, 'Onaylandı', 'Reddedildi', ''), HGSiparisDetay.OnayTarihi, HGSiparisDetay.OnaylayanKullaniciId,
    HGSiparisDetay.MaliOnayDurumu, Decode(HGSiparisDetay.MaliOnayDurumu, 0, 'Onay Bekleniyor', 1, 'Onaylandı', 'Reddedildi', ''), HGSiparisDetay.MaliOnayTarihi, HGSiparisDetay.MaliOnaylayanKullaniciId, HGSiparisDetay.LinkIslemi, HGSiparisDetay.DegisimLog, HGSiparisDetay.InsertKullaniciId,
    HGSiparisDetay.InsertTarihi, HGSiparisDetay.KullaniciId, HGSiparisDetay.DegisimTarihi,
    OK.Adi, IK.Adi, SK.Adi, MOK.Adi, OdemeSekli.Adi from HGSiparisListesi(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :AKapali, :AModelId, :AMusteriSiparisNo)
  left join HGSiparisDetay on HGSiparisDetay.HGSiparisKayitId = HGSiparisListesi.HGSiparisKayitId
  left join Model on Model.ModelKayitID = HGSiparisDetay.ModelKayitID
  Left Join Kullanici OK on OK.KullaniciId = HGSiparisDetay.OnaylayanKullaniciId
  Left Join Kullanici IK on IK.KullaniciId = HGSiparisDetay.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HGSiparisDetay.KullaniciId
  Left Join Kullanici MOK on MOK.KullaniciId = HGSiparisDetay.MaliOnaylayanKullaniciId
  Left Join OdemeSekli on OdemeSekli.OdemeSekliId = HGSiparisDetay.OdemeSekliId
  Where ((HGSiparisDetay.MusteriSiparisNo Like '%'||:AMusteriSiparisNo||'%') or (:AMusteriSiparisNo is null))
  order by HGSiparisListesi.HGSiparisKayitId, HGSiparisDetay.HGSiparisDetayId
  into :HGSiparisDetayid, :HGSiparisKayitId, :ModelId, :SiparisSiraNo, :MusteriSiparisNo,
        :MusteriModelNo, :ReferansSiparisNo, :Sinif, :HamStokBirimId, :SatisFiyati, :DovizId,
        :SatisTipiID, :SatisTipiStr, :TeslimTarihi, :VadeGun, :Notu,
        :ResimDosyaAdi1, :ResimDosyayazdir1, :ResimDosyaAdi2, :ResimDosyaYazdir2, :Iptal, :IptalStr, :Kapali, :KapaliStr,
        :KapatmaTarihi, :OdemeSekliId, :OnayDurumu, :OnayDurumuStr, :OnayTarihi, :OnaylayanKullaniciId, :MaliOnayDurumu, :MaliOnayDurumuStr,
        :MaliOnayTarihi, :MaliOnaylayanKullaniciId, :LinkIslemi, :DegisimLog, :InsertKullaniciId,
        :InsertTarihi, :KullaniciId, :DegisimTarihi, :OnaylayanKullaniciAdi, :InsertKullaniciAdi, :KullaniciAdi, :MaliOnaylayanKullaniciAdi, :OdemeSekliAdi do
  begin

    suspend;
  end
end
