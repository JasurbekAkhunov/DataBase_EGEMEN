CREATE PROCEDURE IPLIKTARTIKILOGIRISI(
    APARTISINIFID INTEGER,
    AKILO NUMERIC(12,3),
    APERSONELID INTEGER)
AS
declare variable KayitSayisi         integer;
declare variable FKilo               Numeric(12,3); /*eski kilo */
declare variable FarkKilo            Numeric(12,3);
declare variable CikisKilo            Numeric(12,3);
declare variable FRafDetayId         integer;
declare variable FHamStokDetayId     integer;
declare variable FPartiKayitId       integer;
declare variable FBoyaSiparisKayitId integer;
declare variable FBoyaSiparisDetayId integer;
declare variable FBoyaSiparisSarjId  integer;
declare variable FRezervKilo         Numeric(12,3);
declare variable FBoyaSiparisRezervId integer;
declare variable FRafKilo             Numeric(12,3);
declare variable K1                   Numeric(12,3);
declare variable K2                   Numeric(12,3);
declare variable K3                   Numeric(12,3);
declare variable AKullaniciId          integer;
declare variable TartiMinMaxFarkOrani  Double Precision;
declare variable HamTipiId      integer;
declare variable HSDKalanMiktar Numeric(12,3);
declare variable RezervEdilecekKilo Numeric(12,3);
declare variable IplikTartidanCikisKiloKaydet  integer;
declare variable IplikTartiAsamaId             integer;
declare variable IplikTartiMakinaId            varchar(10);
declare variable FSira                         integer;
declare variable FGirisTarihi                  TimeStamp;
declare variable FCikisTarihi                  TimeStamp;
declare variable FPartiAsamalariId             integer;
declare variable FFireOrani                    Double precision;
begin
  Select PartiStok.HamTipiId, BoyaSiparisSarj.FireOrani from PartiStok
    left join BoyaSiparisSarj on BoyaSiparisSarj.BoyaSiparisSarjID = PartiStok.BoyaSiparisSarjID
    where PartiStok.PartiSinifId = :APartiSinifId
    into :HamTipiId, :FFireOrani;

  if (FFireOrani is null) then FFireOrani = 0;

  if (AKilo is null or AKilo <= 0) then exception TartiKilosuSifirOlamaz;

  CikisKilo = AKilo;

  if (HamTipiId <> 0 and HamTipiId <> 5) then Exception PartiIplikPartisiDegil;

  Select Max(KullaniciId) from Kullanici where Kullanici.PersonelId = :APersonelId into :AKullaniciId;

  if (AKullaniciId is null) then exception PersonelKullaniciDegil;

  Select IplikTartidanCikisKiloKaydet, IplikTartiAsamaId, IplikTartiMakinaId, TartiMinMaxFarkOrani from sistem where SistemId = 1
    into :IplikTartidanCikisKiloKaydet, :IplikTartiAsamaId, :IplikTartiMakinaId, :TartiMinMaxFarkOrani;

  if (IplikTartidanCikisKiloKaydet >= 1) then
  begin
    if (IplikTartiAsamaId is null or IplikTartiMakinaId is null) then exception KayitHatasi 'Sistem de tartı aşaması, makinası tanıtılmamış.';
    Select Count(*) from PartiAsamaStok where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is null and PartiAsamaStok.AsamaID <> :IplikTartiAsamaId into :KayitSayisi;
    if (KayitSayisi > 0) then exception PartiImalattaKiloDegisemez;

    Select Min(Sira) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamaStok.AsamayaDahil = 1 and
            PartiAsamaStok.AsamaId = :IplikTartiAsamaId
      into :FSira;

    if (FSira is null) then exception KayitHatasi 'Parti nin tartı aşaması yok.Tartı Yapılamaz..';

    Select Count(*) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.Sira < :FSira and
            PartiAsamaStok.CikisTarihi is null
      into :KayitSayisi;

    if (KayitSayisi > 0) then exception OnceYapilmasiGerekenAsamaVar;

    Select PartiAsamaStok.PartiAsamalariId, PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamalari.Sira = :FSira and PartiAsamaStok.AsamayaDahil = 1 and
            PartiAsamaStok.AsamaId = :IplikTartiAsamaId
      into :FPartiAsamalariId, :FGirisTarihi, :FCikisTarihi;

    if (FCikisTarihi is not null) then exception KayitHatasi 'Parti nin tartı işlemi tamamlanmış.Değiştirilemez.';

    if (FGirisTarihi is null) then
    begin
      Update PartiAsamaStokDetay Set Giristarihi = Current_TimeStamp(0), MakinaID = :IplikTartiMakinaId, GirisPersonelID = :APersonelId, GirisOtomatik = 1
         where PartiAsamaStokDetay.PartiAsamalariID = :FPartiAsamalariID and PartiAsamaStokDetay.PartiSinifID = :APartiSinifID;
    end

    AKilo = AKilo + AKilo * FFireOrani /100;
    AKilo = Yuvarla(AKilo, 100);
  end
  else
  begin
    Select Count(*) from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiSinifId = :APartiSinifId and PartiAsamaStokDetay.GirisTarihi is not null and PartiAsamaStokDetay.CikisTarihi is null into :KayitSayisi;
    if (KayitSayisi > 0) then exception PartiImalattaKiloDegisemez;
  end

  if (IplikTartidanCikisKiloKaydet <= 1) then
  begin
    Select PartiStok.PartiKayitId, Parti.BoyaSiparisKayitId, PartiStok.BoyaSiparisDetayId, PartiStok.BoyaSiparisSarjId, PartiStok.Kilo from PartiStok
      left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
      where PartiStok.PartiSinifId = :APartiSinifId
      into :FPartiKayitId, :FBoyaSiparisKayitId, :FBoyaSiparisDetayId, :FBoyaSiparisSarjId, :FKilo;

    if (FKilo > AKilo) then K1 = FKilo - AKilo;
    else K1 = AKilo - FKilo;

    if (TartiMinMaxFarkOrani is null) then TartiMinMaxFarkOrani = 0;
    if (K1 > (FKilo * TartiMinMaxFarkOrani/ 100)) then exception TartiFarkiIzinVerilendenCok;

    if (FKilo >= AKilo) then
    begin
      FarkKilo = FKilo - AKilo;

      For Select RafDetayId, Kilo from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId into :FRafDetayId, :FKilo do
      begin
        if (FarkKilo > 0) then
        if (FarkKilo < FKilo) then
        begin
          Update PartiStokDetay Set Kilo = Kilo - :FarkKilo, KullaniciId = :AKullaniciId, KiloDegistirenPersonelId = :APersonelId where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;
          FarkKilo = 0;
        end
        else
        begin
          Update PartiStokDetay Set Kilo = 0, KullaniciId = :AKullaniciId, KiloDegistirenPersonelId = :APersonelId, LinkIslemi = 2 where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;

          Delete from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;
          FarkKilo = FarkKilo - FKilo;
        end
      end
    end else
    begin
      FarkKilo = AKilo - FKilo;

      FRezervKilo = 0;

      For Select RafDetay.HamStokDetayId from PartiStokDetay
        left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        where PartiStokDetay.PartiSinifId = :APartiSinifId
        group by RafDetay.HamStokDetayId
        into :FHamStokDetayId do
      begin
        K1 = 0;
        Select Kilo - KullanilanKilo from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisSarjId = :FBoyaSiparisSarjId and BoyaSiparisRezerv.HamStokDetayID = :FHamStokDetayId and Kilo - KullanilanKilo > 0 into :K1;
        if (K1 is null) then K1 = 0;
        FRezervKilo = FRezervKilo + K1;
      end
      RezervEdilecekKilo = 0;

      if (FRezervKilo < FarkKilo) then
      begin
        RezervEdilecekKilo = FarkKilo - FRezervKilo;

        For Select RafDetay.HamStokDetayId from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          where PartiStokDetay.PartiSinifId = :APartiSinifId
          group by RafDetay.HamStokDetayId
          into :FHamStokDetayId do
        begin
          HSDKalanMiktar = 0;
          select PartilenebilirMiktar from HamStokDetayKalaniGetir(:FHamStokDetayId) into :HSDKalanMiktar;

          if (RezervEdilecekKilo > HSDKalanMiktar) then
            K2 = HSDKalanMiktar;
          else
            K2 = RezervEdilecekKilo;

          RezervEdilecekKilo = RezervEdilecekKilo - K2;

          if (K2 > 0) then
          begin
            Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, HamStokDetayId, Kilo, InsertKullaniciId, KullaniciId)
                                   values(:FBoyaSiparisKayitId, :FBoyaSiparisDetayId, :FBoyaSiparisSarjId, :FHamStokDetayId, :K2, :AKullaniciId, :AKullaniciId);
          end
        end
      end

      if (RezervEdilecekKilo > 0) then exception TartiKilosuDegisitirlemedi;
      else
      begin
        For Select BoyaSiparisRezervId, KalanMiktar from SiparisRezervListesiGetir(:FBoyaSiparisKayitId) where BoyaSiparisSarjID = :FBoyaSiparisSarjID
          into :FBoyaSiparisRezervId, :K1 do
        begin
          if (FarkKilo > 0) then
          begin
            For Select RafDetayId, KalanMiktar from SiparisRezervRafListesiGetir(:FBoyaSiparisKayitId) where BoyaSiparisRezervId = :FBoyaSiparisRezervId
               into :FRafDetayId, :K2 do
            begin
              if (FarkKilo > 0) then
              begin
                if (K1 > K2) then K3 = K2;
                else K3 = K1;

                if (K3 > FarkKilo) then K3 = FarkKilo;

                K1 = K1 - K3;
                FarkKilo = FarkKilo - K3;

                KayitSayisi = 0;
                Select Count(*) from PartiStokDetay where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId into :KayitSayisi;

                if (KayitSayisi > 0) then
                  update PartiStokDetay Set Kilo = Kilo + :K3, LinkIslemi = 1 where PartiStokDetay.PartiSinifId = :APartiSinifId and PartiStokDetay.RafDetayId = :FRafDetayId;
                else
                 insert into PartiStokDetay(PartiKayitId, PartiSinifId, RafDetayId, Kilo, BoyaSiparisRezervId, LinkIslemi)
                             values(:FPartiKayitId, :APartiSinifId, :FRafDetayId, :K3, :FBoyaSiparisRezervId, 1);
              end
            end
          end
        end
      end
    end

    if (FarkKilo <> 0) then exception TartiKilosuDegisitirlemedi;
    else
    begin
      Select PartiStok.Kilo from PartiStok where PartiStok.PartiSinifId = :APartiSinifId into :FKilo;
      if (FKilo <> AKilo) then exception TartiKilosuDegisimHatasi;
    end
  end

  if (IplikTartidanCikisKiloKaydet >= 1) then
  begin
    Update PartiAsamaStokDetay Set CikisKilo = :CikisKilo
       where PartiAsamaStokDetay.PartiAsamalariID = :FPartiAsamalariID and PartiAsamaStokDetay.PartiSinifID = :APartiSinifID;
  end
end
