CREATE PROCEDURE ENERJISARFIYATRAPORUHAZIRLA(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAID VARCHAR(10))
RETURNS (
    URETIMTOPLAMI NUMERIC(12,3),
    ICTAMIRTOPLAMI NUMERIC(12,3),
    DISTAMIRTOPLAMI NUMERIC(12,3),
    IMALATSURESI DOUBLE PRECISION)
AS
declare variable FIcTamir   integer;
declare variable FDisTamir  integer;
declare variable Kilo           Numeric(12,3);
declare variable GirisTarihi    Timestamp;
declare variable CikisTarihi    TimeStamp;
declare variable AraVermeSuresi Double Precision;
begin
  ImalatSuresi = 0;
  UretimToplami = 0;
  IcTamirToplami = 0;
  DisTamirToplami = 0;

  for select IcTamir, DisTamir, Sum(kilo), Min(GirisTarihi), max(Cikistarihi), max(AraVermeSuresi) from MakinaSarjRaporuPartiler(null, :ABaslangicTarihi, :ABitisTarihi, Null, :AMakinaId)
  group By MakinaSarjId, IcTamir, DisTamir
  into :FICTamir, :FDisTamir, :Kilo, :GirisTarihi, :Cikistarihi, :AraVermeSuresi do
  begin
    ImalatSuresi = ImalatSuresi + CikisTarihi - GirisTarihi;
    ImalatSuresi = ImalatSuresi - AraVermeSuresi;

    UretimToplami = UretimToplami + Kilo;

    if (FDisTamir = 1) then
      DisTamirToplami = DisTamirToplami + Kilo;
    else if (FIcTamir = 1) then
      IcTamirToplami = IcTamirToplami + Kilo;
  end
  suspend;
end
