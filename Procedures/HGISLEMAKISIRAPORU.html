CREATE PROCEDURE HGISLEMAKISIRAPORU(
    AHGSIPARISYIL INTEGER,
    AHGSIPARISID INTEGER,
    AMODELURETIMISLEMTIPIID MODELGRUPKODUSIZE)
RETURNS (
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER,
    HGSIPARISKAYITID INTEGER,
    HGSIPARISDETAYID INTEGER,
    SIPARISFIRMAID FIRMAKODUSIZE,
    SIPARISFIRMAADI ADI100,
    SIPARISTARIHI TIMESTAMP,
    MODELID MODELKODUSIZE,
    MODELADI ADI40,
    MODELURETIMISLEMTIPIID MODELGRUPKODUSIZE,
    MODELURETIMISLEMTIPIADI ADI20,
    SIPARISMIKTAR BOYAMIKTARSIZE,
    PLANLANANMIKTAR BOYAMIKTARSIZE,
    GERCEKLESENMIKTAR BOYAMIKTARSIZE,
    CIKILANMIKTAR BOYAMIKTARSIZE,
    CIKILABILECEKMIKTAR BOYAMIKTARSIZE,
    KESIMISLEMI EGEBOOLEAN)
AS
declare variable HgSipDetayUretimIslemleriId Integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable OncekiGerceklesenMiktar BoyaMiktarSize;
declare variable OncekiHgSiparisDetayId  Integer;
begin
  select shtFasonUretimeCikis, shtUretimeCikis from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis,  :shtUretimeCikis;
  OncekiGerceklesenMiktar = Null;
  OncekiHgSiparisDetayId = 0;

  for select HgSiparis.HgSiparisYil, HgSiparis.HgSiparisId, HgSiparis.HgSiparisKayitId, HgSiparisDetay.HgSiparisDetayId, HgSiparis.FirmaId, Firma.Adi,
             HgSiparis.SiparisTarihi, Model.ModelId, Model.Adi, HgSipDetayUretimIslemleri.ModelUretimIslemTipiId, ModelUretimIslemTipi.Adi, ModelUretimIslemTipi.KesimIslemi,
             HgSipDetayUretimIslemleri.HgSipDetayUretimIslemleriId from HgSiparis
      left join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
      left join HgSipDetayUretimIslemleri on  HgSipDetayUretimIslemleri.HgSiparisDetayId = HgSiparisDetay.HgSiparisDetayId
      left join Model on Model.ModelKayitId = HgSiparisDetay.ModelKayitId
      left join Firma on Firma.FirmaId = HgSiparis.FirmaId
      left join ModelUretimIslemTipi on ModelUretimIslemTipi.ModelUretimIslemTipiId = HgSipDetayUretimIslemleri.ModelUretimIslemTipiId
    where HgSiparisDetay.Kapali = 0 and
      (HgSiparis.HgSiparisYil = :AHgSiparisYil or :AHgSiparisYil is null) and
      (HgSiparis.HgSiparisId = :AHgSiparisId or :AHgSiparisId is null)
    order by HgSipDetayUretimIslemleri.HgSiparisDetayId, HgSipDetayUretimIslemleri.Sira
      into :HgSiparisYil, :HgSiparisId, :HgSiparisKayitId, :HgSiparisDetayId, :SiparisFirmaId, :SiparisFirmaAdi,
           :SiparisTarihi, :ModelId, :ModelAdi, :ModelUretimIslemTipiId, :ModelUretimIslemTipiAdi, :KesimIslemi, :HgSipDetayUretimIslemleriId do
   begin
     /*Talimatlara bakılmadan miktarları siparişten al*/
     select Sum(HgSiparisDetayUrun.Miktar) from HgSiparisDetayUrun
     where HgSiparisDetayUrun.HgSiparisDetayId = :HgSiparisDetayId and
     HgSiparisDetayUrun.HgSiparisDetayAsortiId is null into :SiparisMiktar;
     PlanlananMiktar = 0;
     GerceklesenMiktar = 0;
     CikilanMiktar = 0;

     /*Sipariş için oluşan talimatlardan miktarları al*/
     if (KesimIslemi = 1) then
     begin
       select Coalesce(Sum(HgIsEmriPastalUrun.PlanlananMiktar), 0), Coalesce(Sum(HgIsEmriPastalUrun.GerceklesenMiktar), 0) from HgIsEmri
       left join HgIsEmriPastal on HgIsEmriPastal.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
       left join HgIsEmriPastalUrun on HgIsEmriPastalUrun.HgIsEmriPastalId = HgIsEmriPastal.HgIsEmriPastalId
       where HgIsEmri.KesimTalimati = 1 and
       HgIsEmri.HgSiparisDetayId = :HgSiparisDetayId
       into :PlanlananMiktar, :GerceklesenMiktar;

       select Coalesce(Sum(StokHareket.Miktar), 0) from HgIsEmri
       left join HgIsEmriPastal on HgIsEmriPastal.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
       left join StokHareket on StokHareket.HgIsEmriPastalId = HgIsEmriPastal.HgIsEmriPastalId
       where HgIsEmri.KesimTalimati = 1 and
       HgIsEmri.HgSiparisDetayId = :HgSiparisDetayId and
       (StokHareket.HareketTipi = :shtUretimeCikis or StokHareket.HareketTipi = :shtFasonUretimeCikis)
       into :CikilanMiktar;
     end
     else
     begin
       select Coalesce(Sum(HgIsEmriUrun.PlanlananMiktar), 0), Coalesce(Sum(HgIsEmriUrun.GerceklesenMiktar), 0) from HgIsEmriUretimIslemleri
       left join HgIsEmri on HgIsEmri.HgIsEmriKayitId = HgIsEmriUretimIslemleri.HgIsEmriKayitId
       left join HgIsEmriUrun on HgIsEmriUrun.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
       where HgIsEmriUretimIslemleri.HgSipDetayUretimIslemleriId = :HgSipDetayUretimIslemleriId and
       HgIsEmri.KesimTalimati = 0
       into :PlanlananMiktar, :GerceklesenMiktar;

       select Coalesce(Sum(StokHareket.Miktar), 0) from HgIsEmriUretimIslemleri
       left join HgIsEmri on HgIsEmri.HgIsEmriKayitId = HgIsEmriUretimIslemleri.HgIsEmriKayitId
       left join StokHareket on StokHareket.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
       where HgIsEmriUretimIslemleri.HgSipDetayUretimIslemleriId = :HgSipDetayUretimIslemleriId and
       HgIsEmri.KesimTalimati = 0 and
       (StokHareket.HareketTipi = :shtUretimeCikis or StokHareket.HareketTipi = :shtFasonUretimeCikis)
       into :CikilanMiktar;
     end

     if (OncekiHgSiparisDetayId <> HgSiparisDetayId) then
       CikilabilecekMiktar = Null;
     else
       CikilabilecekMiktar = OncekiGerceklesenMiktar - CikilanMiktar;

     OncekiHgSiparisDetayId = HgSiparisDetayId;
     OncekiGerceklesenMiktar = GerceklesenMiktar;

     if (ModelUretimIslemTipiId = :AModelUretimIslemTipiId or :AModelUretimIslemTipiId is null) then
       suspend;
   end
end
