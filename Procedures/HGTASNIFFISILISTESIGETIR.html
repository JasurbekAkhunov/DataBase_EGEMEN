CREATE PROCEDURE HGTASNIFFISILISTESIGETIR(
    ABASLAMATARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    HGTASNIFFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    TASNIFMETO SMALLINT,
    TASNIFMETOSTR VARCHAR(20),
    NOTU MEMO,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40))
AS
begin
  for select HGTasnifFisi.HGTasnifFisiId, HGTasnifFisi.FisYil, HGTasnifFisi.FisNo, HGTasnifFisi.FisTarihi, HGTasnifFisi.TasnifMeto,
            HGTasnifFisi.Notu, HGTasnifFisi.InsertKullaniciId, HGTasnifFisi.InsertTarihi, HGTasnifFisi.KullaniciId,
            HGTasnifFisi.DegisimTarihi, IK.Adi, SK.Adi from HgTasnifFisi
      left join Kullanici IK on IK.KullaniciId  = HgTasnifFisi.InsertKullaniciId
      left join Kullanici SK on SK.KullaniciId  = HgTasnifFisi.KullaniciId
      where (HgTasnifFisi.FisTarihi >= :ABaslamaTarihi or :ABaslamaTarihi is null) and
            (HgTasnifFisi.FisTarihi <= :ABitisTarihi or :ABitisTarihi is null)
      into :HGTasnifFisiId, :FisYil, :FisNo, :FisTarihi, :TasnifMeto, :Notu, :InsertKullaniciId, :InsertTarihi, :KullaniciId,
           :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi do
  begin
    if (TasnifMeto = 0) then TasnifMetoStr = 'Meto';
    else if (TasnifMeto = 1) then TasnifMetoStr = 'Tasnif';
    else TasnifMetoStr = 'Meto + Tasnif';

    suspend;
  end
end
