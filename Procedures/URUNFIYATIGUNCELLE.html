CREATE PROCEDURE URUNFIYATIGUNCELLE(
    AURUNID VARCHAR(20))
AS
declare variable AYAlisFiyatiGuncelle           integer;
declare variable AYUretimeCikisFiyatiGuncelle   integer;
declare variable UretimeCikisFiyatiHesapOran    Double Precision;
declare variable StokDetayId integer;

declare variable FisDovizId       varchar(5);
declare variable IrsaliyeDovizId  varchar(5);
declare variable FaturaDovizId    varchar(5);

declare variable HNetBirimFiyat          Double Precision;
declare variable HUrunBirimCarpan        Double Precision;
declare variable HDovizId                varchar(5);
declare variable FDovizKuru              Double Precision;
declare variable HUrunBirimId            integer;

declare variable AFUrunBirimCarpan  Double Precision;
declare variable AlisFiyatiDovizId          varchar(5);
declare variable UCFUrunBirimCarpan  Double Precision;
declare variable UretimeCikisFiyatiDovizId  varchar(5);

declare variable AlisFiyati          Double Precision;
declare variable UretimeCikisFiyati  Double Precision;
declare variable UrunFiyatVar        integer;
declare variable YuvarlamaCarpani    Double Precision;
declare variable UrunTipiId          integer;
begin
  Select AYAlisFiyatiGuncelle, AYUretimeCikisFiyatiGuncelle, UretimeCikisFiyatiHesapOran from Sistem Where SistemId = 1 into :AYAlisFiyatiGuncelle, :AYUretimeCikisFiyatiGuncelle, :UretimeCikisFiyatiHesapOran;

  Select urun.UrunTipiId from Urun where Urun.UrunID = :AUrunID into :UrunTipiId;

  if (AYAlisFiyatiGuncelle = 0 and AYUretimeCikisFiyatiGuncelle = 0) then exit;

  Select UrunFiyat.AlisFiyatiDovizId, UBAF.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId, UBUCF.Carpan from UrunFiyat
    left join UrunBirim UBAF on UBAF.UrunBirimId = urunFiyat.AlisFiyatiUrunBirimId
    left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
    where UrunFiyat.UrunId = :AUrunId
    into :AlisFiyatiDovizId, :AFUrunBirimCarpan, :UretimeCikisFiyatiDovizId, UCFUrunBirimCarpan;

  if (AlisFiyatiDovizId is null) then
    UrunFiyatVar = 0;
  else
    UrunFiyatVar = 1;

  Select SonAlisStokDetayId from Urun where Urun.UrunId = :AUrunId into :StokDetayId;

  if (StokDetayId is null) then exit;

  Select StokDetay.NetBirimFiyat, StokDetay.UrunBirimId, UrunBirim.Carpan, Stok.DovizId, Irsaliye.DovizId, Fatura.DovizId from StokDetay
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join Stok on Stok.StokId = StokDetay.StokId
    left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
    left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
    where StokDetay.StokDetayID = :StokDetayId
    into :HNetBirimFiyat, :HUrunBirimId, :HUrunBirimCarpan, :FisDovizId, :IrsaliyeDovizId, :FaturaDovizId;

  if (FaturaDovizId is not null) then HDovizId = FaturaDovizId;
  else if (IrsaliyeDovizId is not null) then HDovizId = IrsaliyeDovizId;
  else HDovizID = FisDovizId;

  if (AlisFiyatiDovizId is null) then AlisFiyatiDovizId = HDovizId;
  if (UretimeCikisFiyatiDovizId is null) then UretimeCikisFiyatiDovizId = HDovizId;

  if (UrunFiyatVar = 0) then
  begin
    AFUrunBirimCarpan = HUrunBirimCarpan;
    UCFUrunBirimCarpan = HUrunBirimCarpan;
  end

  AlisFiyati = HNetBirimFiyat * HUrunBirimCarpan / AFUrunBirimCarpan;

  UretimeCikisFiyati = HNetBirimFiyat * HUrunBirimCarpan / UCFUrunBirimCarpan;

  if (UrunTipiId = 0 or UrunTipiId = 1) then
    UretimeCikisFiyati = UretimeCikisFiyati + UretimeCikisFiyati * UretimeCikisFiyatiHesapOran / 100;

  if (AlisFiyatiDovizId <> HDovizId) then
  begin
    select DovizKuru from EnYakinDovizKuruGetir(:HDovizId, :AlisFiyatiDovizId, Current_Date) into :FDovizKuru;
    AlisFiyati = AlisFiyati * FDovizKuru;
  end

  Select NetFiyatYuvarlamaCarpani from Doviz where Doviz.DovizID = :AlisFiyatiDovizId into :YuvarlamaCarpani;

  if (YuvarlamaCarpani is null) then YuvarlamaCarpani = 100;

  AlisFiyati = Yuvarla(AlisFiyati, YuvarlamaCarpani);

  if (UretimeCikisFiyatiDovizId <> HDovizId) then
  begin
    if (UretimeCikisFiyatiDovizId <> AlisFiyatiDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:HDovizId, :UretimeCikisFiyatiDovizId, Current_Date) into :FDovizKuru;
    UretimeCikisFiyati = UretimeCikisFiyati * FDovizKuru;
  end

  Select NetFiyatYuvarlamaCarpani from Doviz where Doviz.DovizID = :UretimeCikisFiyatiDovizId into :YuvarlamaCarpani;

  UretimeCikisFiyati = Yuvarla(UretimeCikisFiyati, YuvarlamaCarpani);

  if (UrunFiyatVar = 0) then
  begin
    insert into UrunFiyat(UrunId, AlisFiyatiDovizId, AlisFiyatiUrunBirimId, SatisFiyatiDovizId, SatisFiyatiUrunBirimId, UretimeCikisFiyatiDovizId, UretimeCikisFiyatiUrunBirimId)
                   values(:AUrunId, :HDovizId, :HUrunBirimId, :HDovizId, :HUrunBirimId, :HDovizId, :HUrunBirimId);
  end

  if (AYAlisFiyatiGuncelle = 1) then
    Update UrunFiyat Set AlisFiyati = :AlisFiyati where UrunFiyat.UrunId = :AUrunId;

  if (AYUretimeCikisFiyatiGuncelle = 1 and UretimeCikisFiyati > 0) then
    Update UrunFiyat Set UretimeCikisFiyati = :UretimeCikisFiyati where UrunFiyat.UrunId = :AUrunId;
end
