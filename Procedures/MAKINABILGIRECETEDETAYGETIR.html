CREATE PROCEDURE MAKINABILGIRECETEDETAYGETIR(
    AMAKINABILGIRECETEID INTEGER)
RETURNS (
    MAKINAKAMARABILGIRECETEID INTEGER,
    MAKINABILGIRECETEID INTEGER,
    KAMARANO INTEGER,
    SAGISIDEGERI DOUBLE PRECISION,
    SAGISIDEGERISET DOUBLE PRECISION,
    SOLISIDEGERI DOUBLE PRECISION,
    SOLISIDEGERISET DOUBLE PRECISION,
    ALTFANCALISMAYUZDESI DOUBLE PRECISION,
    USTFANCALISMAYUZDESI DOUBLE PRECISION,
    ENDEGERI DOUBLE PRECISION,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP)
AS
begin
     for select MakinaKamaraBilgiRecete.MakinaKamaraBilgiReceteid, MakinaKamaraBilgiRecete.makinabilgireceteid,
             MakinaKamaraBilgiRecete.kamarano, MakinaKamaraBilgiRecete.sagisidegeri,
             MakinaKamaraBilgiRecete.sagisidegeriset, MakinaKamaraBilgiRecete.solisidegeri,
             MakinaKamaraBilgiRecete.solisidegeriset, MakinaKamaraBilgiRecete.altfancalismayuzdesi,
             MakinaKamaraBilgiRecete.ustfancalismayuzdesi, MakinaKamaraBilgiRecete.endegeri,
             MakinaKamaraBilgiRecete.insertkullaniciid, MakinaKamaraBilgiRecete.inserttarihi,
             MakinaKamaraBilgiRecete.kullaniciid, MakinaKamaraBilgiRecete.degisimtarihi
      from MakinaKamaraBilgiRecete where MakinaKamaraBilgiRecete.MakinaBilgiReceteId = :AMakinaBilgiReceteId
      into :MakinaKamaraBilgiReceteid, :makinabilgireceteid, :kamarano, :sagisidegeri, :sagisidegeriset, :solisidegeri,
           :solisidegeriset, :altfancalismayuzdesi, :ustfancalismayuzdesi, :endegeri, :insertkullaniciid, :inserttarihi,
           :kullaniciid, :degisimtarihi do suspend;
end
