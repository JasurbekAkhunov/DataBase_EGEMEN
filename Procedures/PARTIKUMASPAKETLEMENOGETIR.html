CREATE PROCEDURE PARTIKUMASPAKETLEMENOGETIR(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER)
RETURNS (
    KUMASPAKETLEMENO VARCHAR(30))
AS
declare variable KumasPaketlemeAsamaId integer;
declare variable PartiAsamalariId      integer;
declare variable MaxSira               integer;
declare variable PaketlemeYil          varchar(4);
declare variable PaketlemeId           varchar(10);
begin
  KumasPaketlemeNo = '';

  select KumasPaketlemeAsamaId from Sistem where SistemId = 1 into :KumasPaketlemeAsamaId;

  Select Max(PartiAsamalari.Sira) from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariID
    where PartiAsamaStok.PartiKayitID = :APartiKayitID and PartiAsamaStok.PartiSinifID = :APartiSinifID and
          PartiAsamalari.AsamaID = :KumasPaketlemeAsamaId and
          PartiAsamaStok.AsamayaDahil = 1
    into :MaxSira;

  if (MaxSira is not null) then
  begin
    Select PartiAsamalari.PartiAsamalariId from PartiAsamalari
      where PartiAsamalari.PartiKayitID = :APartiKayitID and PartiAsamalari.Sira = :MaxSira
      into :PartiAsamalariId;

    if (PartiAsamalariId is null) then exception KayitHatasi 'Paketleme No Bulunamadi..';

    Select PaketlemeYil, PaketlemeId from PartiAsamaStok
      where PartiAsamaStok.PartiSinifID = :APartiSinifID and PartiAsamaStok.PartiAsamalariID = :PartiAsamalariID
      into :PaketlemeYil, :PaketlemeId;

    if (PaketlemeYil is not null) then
      KumasPaketlemeNo = EgemenCopyStr(PaketlemeYil, 3, 2) || '-' || PaketlemeId;
  end

  suspend;
end
