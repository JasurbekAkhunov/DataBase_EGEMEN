CREATE PROCEDURE SARIMSTOKLISTESI
RETURNS (
    IPLIKLOTID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    KALINLIKID INTEGER,
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIID INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    LOT VARCHAR(30),
    MELANJKODU VARCHAR(30),
    FLAMKODU VARCHAR(20),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    ESKILOT VARCHAR(30),
    KAPADEDI INTEGER,
    MIKTAR NUMERIC(12,3),
    ISEMRIHAZIRLANANMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    TAMIR INTEGER,
    IPLIKTAMIRTIPIID INTEGER,
    IPLIKTAMIRTIPIADI VARCHAR(20))
AS
begin
  For Select SarimStok.IplikLotId, IplikLot.HamId, Ham.Adi, IplikLot.Lot, IplikLot.MelanjKodu, IplikLot.FlamKodu, IplikLot.EskiLot,
             IplikSatisIade.RenkId, Renk.Adi, SarimStok.KapAdedi, SarimStok.Miktar, SarimStok.IsEmriHazirlananMiktar,
             IplikSatisIade.IplikTamirTipiId, IplikTamirTipi.Adi, IplikLot.Tamir,
             Ham.KalinlikId, Kalinlik.Adi, Ham.IplikCinsiId, IplikCinsi.Adi from SarimStok
    left join IplikLot on IplikLot.IplikLotID = SarimStok.IplikLotId
    left join TamirGiris on TamirGiris.TamirGirisKayitID = IplikLot.TamirGirisKayitId
    left join IplikSatisIade on IplikSatisIade.IplikSatisIadeID = TamirGiris.IplikSatisIadeId
    left join Renk on Renk.RenkID = IplikSatisIade.RenkId
    left join Ham on Ham.HamID = IplikLot.HamId
    left join IplikTamirTipi on IplikTamirTipi.IplikTamirTipiID = IplikSatisIade.IplikTamirTipiId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
    where SarimStok.StokVar = 1
    into :IplikLotId, :HamId, :HamAdi, :Lot, :MelanjKodu, :FlamKodu, :EskiLot, :RenkId, RenkAdi, :KapAdedi, :Miktar, :IsEmriHazirlananMiktar,
         :IplikTamirTipiId, :IplikTamirTipiAdi, :Tamir, :KalinlikId, :KalinlikAdi, :IplikCinsiId, :IplikCinsiAdi do
  begin
    KalanMiktar = Miktar - IsEmriHazirlananMiktar;
    suspend;
  end
end
