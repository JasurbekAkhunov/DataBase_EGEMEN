CREATE PROCEDURE PARTIRECETEDUSUMKONTROL(
    ABASLANGICPATARIHI TIMESTAMP,
    ABITISPATARIHI TIMESTAMP,
    AIMALATTIPIID INTEGER,
    APARTITIPIID INTEGER,
    AISLEMTIPIID INTEGER,
    ASADECEDUSUMYAPILMASIGEREKENLER INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIACILISTARIHI TIMESTAMP,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    PARTIFIRMAID FIRMAKODUSIZE,
    PARTIFIRMAADI ADI100,
    IMALATTIPIID INTEGER,
    IMALATTIPIADI VARCHAR(10),
    PARTITIPIID INTEGER,
    PARTITIPIADI ADI20,
    HAMADITOPLAMI VARCHAR(255),
    RECETEID RECETEKODUSIZE,
    RECETEADI ADI40,
    PARTITOPLAMKILO BOYAMIKTARSIZE,
    RECETESIOLANASAMASAYISI INTEGER,
    RECETEDUSUMYAPILMISASAMASAYISI INTEGER,
    RECETEDUSUMUYAPILMALISTR VARCHAR(15))
AS
declare variable FPartiKayitId integer;
declare variable FHamAdiToplami varchar(255);
declare variable FPartiAsamalariId integer;
begin
  For Select Parti.Partikayitid, Parti.Partiyil, Parti.Partiid, Parti.Acilistarihi, Boyasiparis.Boyasiparisyil, Boyasiparis.Boyasiparisid,
             Parti.Fason, Parti.Firmaid , Firma.Adi, Parti.Imalattipiid, Parti.Partitipiid, Parti.Receteid, Recete.Adi
      From parti
      Left Join Boyasiparis on Boyasiparis.Boyasipariskayitid = Parti.Boyasipariskayitid
      Left join Firma on Firma.FirmaId = Parti.FirmaId
      Left Join Recete on Recete.ReceteId = Parti.Receteid
      Where Parti.Acilistarihi >= :Abaslangicpatarihi and
            Parti.Acilistarihi <= :Abitispatarihi and
            (Parti.Imalattipiid = :Aimalattipiid or :Aimalattipiid is null ) and
            (Parti.Partitipiid = :Apartitipiid or :Apartitipiid is null ) and
            (Parti.Islemtipiid = :Aislemtipiid or :Aislemtipiid is null )
      Order By Parti.Partiyil, parti.Partiid
      Into :Partikayitid, :Partiyil , :Partiid, :PartiAcilisTarihi , :BoyaSiparisYil, :BoyaSiparisId, :Fason, :PartiFirmaId , :PartiFirmaAdi, :Imalattipiid, :Partitipiid, :Receteid, :ReceteAdi do
  Begin
    FasonStr = '' ;

    if (Fason = 0) then FasonStr = 'Satış' ;
    else if (Fason = 1) then FasonStr = 'Fason' ;

    ImalatTipiAdi = '' ;
    if (ImalatTipiId = 0 ) then ImalatTipiAdi = 'Boya' ;
    else if (ImalatTipiId = 1 ) then ImalatTipiAdi = 'Baskı' ;

    Partitipiadi = '' ;
    if (PartiTipiId = 0 ) then Partitipiadi = 'İplik' ;
    else Partitipiadi = 'Kumaş' ;

    HamAdiToplami = '' ;
    For Select PartiStok.HamAdiToplami from PartiStok
        Where PartiStok.PartiKayitId = :PartiKayitId and PartiStok.PartiStokIptalKayitID is null
        Group by PartiStok.HamAdiToplami
        Into :FHamAdiToplami do
    Begin
      if (Char_Length(HamAdiToplami) + Char_Length(FHamAdiToplami) < 252) then
      Begin
        if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set UTF8);
        HamAdiToplami = HamAdiToplami || FHamAdiToplami;
      end
    end

    RecetesiOlanAsamaSayisi = 0 ;
    ReceteDusumYapilmisAsamaSayisi = 0 ;

    Select sum(Partistok.Kilo) From Partistok Where Partistok.Partikayitid = :Partikayitid Into :PartiToplamKilo;

    Select count(Partiasamalari.Partiasamalariid) From Partiasamalari Where Partiasamalari.Partikayitid = :Partikayitid and partiasamalari.Receteid is not null
    into :RecetesiOlanAsamaSayisi ;


    Select Stokdetay.Partikayitid, Count(distinct stokdetay.Partiasamalariid)
    From Stokdetay
    Where Stokdetay.Partikayitid = :Partikayitid and Stokdetay.Partiasamalariid is not null and Stokdetay.PartiAsamaIlaveSarfid is null
    Group by StokDetay.Partikayitid
    Into :FPartiKayitId, :ReceteDusumYapilmisAsamaSayisi ;


    if (RecetesiOlanAsamaSayisi > ReceteDusumYapilmisAsamaSayisi ) then
      ReceteDusumuYapilmaliStr = 'Düşüm yapılmalı' ;
    else
      ReceteDusumuYapilmaliStr = '' ;

    if (ASadeceDusumYapilmasiGerekenler is null) then ASadeceDusumYapilmasiGerekenler = 0 ;
    if ( ( ASadeceDusumYapilmasiGerekenler = 1 and RecetesiOlanAsamaSayisi > ReceteDusumYapilmisAsamaSayisi ) or (ASadeceDusumYapilmasiGerekenler = 0 )  ) then
      suspend ;
  end
end
