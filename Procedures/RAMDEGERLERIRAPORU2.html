    APARTIYIL INTEGER,

    AGRAMAJBASLANGIC VARCHAR(10),
    AGRAMAJBITIS VARCHAR(10),
    AKUMASCINSIID INTEGER,
    AMAKINABILGIRECETEID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    MAKINABILGIID INTEGER,
    TARIH TIMESTAMP,
    MAKINAID MAKINAKODUSIZE,
    NEM DOUBLE PRECISION,
    ISI DOUBLE PRECISION,
    HIZ DOUBLE PRECISION,
    BACANEM DOUBLE PRECISION,
    DOGALGAZ INTEGER,
    ELEKTIRIK INTEGER,
    RAMUSTBESLEME DOUBLE PRECISION,
    RAMALTBESLEME DOUBLE PRECISION,
    RAMCALISMAHIZI DOUBLE PRECISION,
    RAMGIRISEN DOUBLE PRECISION,
    RAMBIRBACAFANCALISMAYUZDESI DOUBLE PRECISION,
    RAMIKIBACAFANCALISMAYUZDESI DOUBLE PRECISION,
    RAMSOGUTMAFANIUST DOUBLE PRECISION,
    RAMSOGUTMAFANIALT DOUBLE PRECISION,
    RAMSOGUTMAFANIUSTMAN DOUBLE PRECISION,
    RAMSOGUTMAFANIALTOTO DOUBLE PRECISION,
    RAMBACANEMIOTOMATIK EGEBOOLEAN,
    RAMBACANEMI DOUBLE PRECISION,
    RAMCIKISEN DOUBLE PRECISION,
    RAMCIKISNEMDEGERI DOUBLE PRECISION,
    RAMBESLEMEHIZI DOUBLE PRECISION,
    RAMKALIPGENISLIGI DOUBLE PRECISION,
    RAMKUMASCEKMEZLIKDEGERI DOUBLE PRECISION,
    RAMKUMASMAYDEGERI DOUBLE PRECISION,
    RAMKUMASNEMI DOUBLE PRECISION,
    RAMMAKINEHIZI DOUBLE PRECISION,
    RAMMAKINESILINDIRSICAKLIGI DOUBLE PRECISION,
    MAKINAKAMARABILGIID INTEGER,
    KAMARANO INTEGER,
    SAGISIDEGERI DOUBLE PRECISION,
    SAGISIDEGERISET DOUBLE PRECISION,
    SOLISIDEGERI DOUBLE PRECISION,
    SOLISIDEGERISET DOUBLE PRECISION,
    ALTFANCALISMAYUZDESI DOUBLE PRECISION,
    USTFANCALISMAYUZDESI DOUBLE PRECISION,
    ENDEGERI DOUBLE PRECISION)
AS
begin
  for select RamDegerleriRaporu.PartiSinifId,
      makinabilgi.makinabilgiid, makinabilgi.tarih, makinabilgi.makinaid, makinabilgi.nem, makinabilgi.isi, makinabilgi.hiz, makinabilgi.bacanem, makinabilgi.dogalgaz,
      makinabilgi.elektirik, makinabilgi.ramustbesleme, makinabilgi.ramaltbesleme, makinabilgi.ramcalismahizi, makinabilgi.ramgirisen, makinabilgi.rambirbacafancalismayuzdesi,
      makinabilgi.ramikibacafancalismayuzdesi, makinabilgi.ramsogutmafaniust, makinabilgi.ramsogutmafanialt, makinabilgi.ramsogutmafaniustman, makinabilgi.ramsogutmafanialtoto,
      makinabilgi.rambacanemiotomatik, makinabilgi.rambacanemi, makinabilgi.ramcikisen, makinabilgi.ramcikisnemdegeri, makinabilgi.rambeslemehizi, makinabilgi.ramkalipgenisligi,
      makinabilgi.ramkumascekmezlikdegeri, makinabilgi.ramkumasmaydegeri, makinabilgi.ramkumasnemi, makinabilgi.rammakinehizi, makinabilgi.rammakinesilindirsicakligi,
      makinakamarabilgi.makinakamarabilgiid, makinakamarabilgi.kamarano, makinakamarabilgi.sagisidegeri, makinakamarabilgi.sagisidegeriset,
      makinakamarabilgi.solisidegeri, makinakamarabilgi.solisidegeriset, makinakamarabilgi.altfancalismayuzdesi, makinakamarabilgi.ustfancalismayuzdesi, makinakamarabilgi.endegeri
    from RamDegerleriRaporu(:APartiYil, :APartiId, :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AHamId, :AProsesId, :AEnBaslangic, :AEnBitis, :AGramajBaslangic, :AGramajBitis, :AKumasCinsiId, :AMakinaBilgiReceteId)
    join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiSinifId = RamDegerleriRaporu.PartiSinifId
    join MakinaBilgi on MakinaBilgi.MakinaId = PartiAsamaStokDetay.MakinaId and MakinaBilgi.Tarih >= PartiAsamaStokDetay.GirisTarihi and MakinaBilgi.Tarih <= PartiAsamaStokDetay.CikisTarihi
    left join MakinaKamaraBilgi on MakinaKamaraBilgi.MakinaBilgiId = MakinaBilgi.MakinaBilgiId
    order by MakinaBilgi.Tarih desc into
    :PartiSinifId, :makinabilgiid, :tarih, :makinaid, :nem, :isi, :hiz, :bacanem, :dogalgaz,
    :elektirik, :ramustbesleme, :ramaltbesleme, :ramcalismahizi, :ramgirisen, :rambirbacafancalismayuzdesi,
    :ramikibacafancalismayuzdesi, :ramsogutmafaniust, :ramsogutmafanialt, :ramsogutmafaniustman, :ramsogutmafanialtoto,
    :rambacanemiotomatik, :rambacanemi, :ramcikisen, :ramcikisnemdegeri, :rambeslemehizi, :ramkalipgenisligi,
    :ramkumascekmezlikdegeri, :ramkumasmaydegeri, :ramkumasnemi, :rammakinehizi, :rammakinesilindirsicakligi,
    :makinakamarabilgiid, :kamarano, :sagisidegeri, :sagisidegeriset, :solisidegeri, :solisidegeriset, :altfancalismayuzdesi, :ustfancalismayuzdesi, :endegeri do
  begin
    suspend;
  end
end
