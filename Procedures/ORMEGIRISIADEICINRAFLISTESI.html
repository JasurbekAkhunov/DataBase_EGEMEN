CREATE PROCEDURE ORMEGIRISIADEICINRAFLISTESI(
    AHAMID INTEGER,
    ARENKID INTEGER,
    ADEPOID VARCHAR(10))
RETURNS (
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    RAFID VARCHAR(10),
    KALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI NUMERIC(12,3),
    LOT VARCHAR(150),
    KALITENO INTEGER,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    URETICIADI VARCHAR(40),
    DEPOPARTINO VARCHAR(30))
AS
begin
  For Select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.RafId, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
             OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi, OrmeRaf.Lot, OrmeRaf.KaliteNo, Uretici.Adi, OrmeRaf.DepoPartiNo from OrmeRaf
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
    left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
    where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and
          (OrmeRaf.RenkId = :ARenkId or :ARenkId is null) and OrmeRaf.FasonStok = 0 and OrmeRaf.BoyahaneFasonIade = 0 and
          OrmeRaf.DesenID is null and
          (DepoRaf.DepoId = :ADepoID or :ADepoID is null)
    into :OrmeRafId, :StokGirisNo, :RafId, :RenkId, :RenkAdi, :KalanMiktar, :KalanKapAdedi, :Lot, :KaliteNo,  :UreticiAdi, :DepoPartiNo do
  begin
    suspend;
  end
end
