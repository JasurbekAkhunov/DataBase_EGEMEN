CREATE PROCEDURE BANKACARIBAKIYERAPORU(
    ATARIH TIMESTAMP)
RETURNS (
    BANKAHESAPID VARCHAR(20),
    BANKAADI VARCHAR(40),
    HESAPADI VARCHAR(40),
    DOVIZID VARCHAR(5),
    BORCTOPLAM DOUBLE PRECISION,
    ALACAKTOPLAM DOUBLE PRECISION,
    BORCBAKIYE DOUBLE PRECISION,
    ALACAKBAKIYE DOUBLE PRECISION,
    BAKIYE DOUBLE PRECISION,
    BAKIYEBA VARCHAR(1))
AS
declare variable FBorcAlacak varchar(1);
declare variable FMeblag Double Precision;
begin
  for Select BankaHesapId, Banka.Adi, BankaHesap.Adi, BankaHesap.DovizId  from BankaHesap
    left join Banka on Banka.Bankaid = BankaHesap.BankaId
    into :BankaHesapId, :BankaAdi, :HesapAdi, :DovizId do
  begin
    BorcToplam = 0;
    AlacakToplam = 0;
    for select  BorcAlacak, sum(Meblag)  from BankaHesapCari  where BankaHesapId = :BankaHesapId and BankaHesapCari.Tarih <= :ATarih
         group by BankaHesapCari.BorcAlacak into :FBorcAlacak, :FMeblag do
    begin
      if (FBorcAlacak = 'B') then BorcToplam = FMeblag;
      else if (FBorcAlacak = 'A') then AlacakToplam = FMeblag;
    end
    BorcBakiye = null;
    AlacakBakiye = Null;

    if (BorcToplam is Null) then BorcToplam = 0;
    if (AlacakToplam is Null) then AlacakToplam = 0;

    if (BorcToplam > AlacakToplam) then
    begin
      BorcBakiye = BorcToplam - AlacakToplam;
      Bakiye = BorcBakiye;
      BakiyeBA = 'B';
    end
    else
    begin
      AlacakBakiye = AlacakToplam - BorcToplam;
      Bakiye = AlacakBakiye;
      if (AlacakToplam > BorcToplam) then BakiyeBA = 'A';
    end
    suspend;
  end
end
