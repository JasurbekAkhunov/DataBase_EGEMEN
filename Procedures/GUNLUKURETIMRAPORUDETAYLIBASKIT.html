CREATE PROCEDURE GUNLUKURETIMRAPORUDETAYLIBASKIT(
    AMAKINAGRUBU VARCHAR(4),
    AMAKINARAPORGRUPID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP)
RETURNS (
    MAKINAID VARCHAR(10),
    BASKITIPIID INTEGER,
    BASKITIPIADI ADI40,
    MIKTAR NUMERIC(12,3),
    VARDIYA1MIKTAR NUMERIC(12,3),
    VARDIYA2MIKTAR NUMERIC(12,3),
    VARDIYA3MIKTAR NUMERIC(12,3),
    AYTOPLAMMIKTAR NUMERIC(12,3),
    MAKINARAPORGRUPID INTEGER)
AS
begin
-- AyToplamMiktar = 0;
  For  select PartiAsamaStokDetay.MakinaID, BoyaSiparisDetay.BaskiTipiId, BaskiTipi.Adi, Sum(PartiAsamaStokDetay.Kilo), Makina.RaporGrupId from PartiAsamaStokDetay
    left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join BaskiTipi on BaskiTipi.BaskiTipiId = BoyaSiparisDetay.BaskiTipiId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    where
          PartiAsamaStokDetay.Cikistarihi >= :AAylikBaslangic and PartiAsamaStokDetay.Cikistarihi <= :AAylikBitis and
          (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
          (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null) and
          Asama.GunlukUretimRaporunaDahil = 1
          -- and BoyaSiparisDetay.BaskiTipiId = :BaskiTipiId
    Group By PartiAsamaStokDetay.MakinaID, BoyaSiparisDetay.BaskiTipiId, BaskiTipi.Adi, Makina.RaporGrupId
    order by PartiAsamaStokDetay.MakinaId, BaskiTipi.Adi
    into :MakinaID, :BaskiTipiId, :BaskiTipiAdi, :AyToplamMiktar, :MakinaRaporGrupId  do
  begin
    Miktar = 0 ;
    Vardiya1Miktar = 0 ;
    Vardiya2Miktar = 0 ;
    Vardiya3Miktar = 0 ;
    select Sum(PartiAsamaStokDetay.Kilo),
           Sum(iif(VardiyaHesapla(Timestamptodouble(PartiAsamaStokDetay.CikisTarihi) - Timestamptodouble(:ABaslangicTarihi), 3) = 1, PartiAsamaStokDetay.Kilo, 0 )),
           Sum(iif(VardiyaHesapla(Timestamptodouble(PartiAsamaStokDetay.CikisTarihi) - Timestamptodouble(:ABaslangicTarihi), 3) = 2, PartiAsamaStokDetay.Kilo, 0 )),
           Sum(iif(VardiyaHesapla(Timestamptodouble(PartiAsamaStokDetay.CikisTarihi) - Timestamptodouble(:ABaslangicTarihi), 3) = 3, PartiAsamaStokDetay.Kilo, 0 ))
    from PartiAsamaStokDetay
      left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      left join BaskiTipi on BaskiTipi.BaskiTipiId = BoyaSiparisDetay.BaskiTipiId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    where  PartiAsamaStokDetay.MakinaID = :MakinaID and
           PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi <= :ABitisTarihi and
          (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
          (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null) and
          Asama.GunlukUretimRaporunaDahil = 1 and
          BoyaSiparisDetay.BaskiTipiId = :BaskiTipiId
    Group By PartiAsamaStokDetay.MakinaID, BoyaSiparisDetay.BaskiTipiId, BaskiTipi.Adi, Makina.RaporGrupId
    order by PartiAsamaStokDetay.MakinaId, BaskiTipi.Adi
  into :Miktar, :Vardiya1Miktar, :Vardiya2Miktar, :Vardiya3Miktar ;
    suspend;
  end

end
