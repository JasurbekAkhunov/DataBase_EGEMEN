CREATE PROCEDURE ORMEIPLIKTRANSFERRAPORU_1
RETURNS (
    IDSTR VARCHAR(20),
    EVRAKTIPI VARCHAR(20),
    ORMEIRSALIYEID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    HAREKETTIPI INTEGER,
    SINIF VARCHAR(1))
AS
declare variable shtAlisIade integer;
declare variable IplikFirmaKodumuz    varchar(20);
begin
  Select shtAlisIade from OrmeStokHareketTipiGetir into :shtAlisIade;

  Select IplikFirmaKodumuz from Sistem where SistemID = 1 into :IplikFirmaKodumuz;

  for select OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.FiiliSevkTarihi, OrmeIrsaliye.Sinif,
             OrmeIrsaliye.HareketTipi from StokHareket
  left join OrmeIrsaliye On OrmeIrsaliye.OrmeIrsaliyeID = StokHareket.OrmeIrsaliyeId
  where StokHareket.FirmaId = :IplikFirmaKodumuz and StokHareket.HareketTipi = :shtAlisIade and StokHareket.IplikIrsaliyeID is null and StokHareket.Tamir = 1 and
        OrmeIrsaliye.BoyahaneTransferEdebilir = 1
  group by OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.FiiliSevkTarihi, OrmeIrsaliye.Sinif,
           OrmeIrsaliye.HareketTipi
  into :IrsaliyeNo, :IrsaliyeTarihi, :OrmeIrsaliyeId, :FiiliSevkTarihi, :Sinif, :HareketTipi do
  begin
    EvrakTipi = 'Merkez Ä°rsaliye';
    IDStr = 'I' || EgemenIntegerToStr(OrmeIrsaliyeId);
    suspend;
  end
end
