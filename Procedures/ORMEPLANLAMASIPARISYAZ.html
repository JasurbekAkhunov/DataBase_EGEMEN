CREATE PROCEDURE ORMEPLANLAMASIPARISYAZ(
    AORMEPLANLAMAGRUPID VARCHAR(10),
    AORMEMAKINAID VARCHAR(10),
    AFIRMAID VARCHAR(20))
RETURNS (
    ORMEPLANLAMAID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPARISIVEREN VARCHAR(50),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    SIPARISMIKTARI NUMERIC(12,3),
    SIPARISKALANMIKTAR NUMERIC(12,3),
    MAKINAUSTUGRAMAJ INTEGER,
    DINLENMISGRAMAJ INTEGER,
    IGNESAYISI INTEGER,
    MAYSAYISI VARCHAR(10),
    PUS INTEGER,
    FEINE INTEGER,
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    ARTIKELNO VARCHAR(20),
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    MUSTERITALIMATI BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    BEYAZLIK INTEGER,
    RAPORBOYU VARCHAR(100),
    URETIMNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
begin
  For Select OrmePlanlamaYaz.OrmePlanlamaId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
     OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.HamMiktar,
     OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.ToplamOrulen + OrmeSiparisDetay.ToplamSakatOrulen,
     OrmeSiparisDetay.MakinaUstuGramaj, OrmeSiparisDetay.DinlenmisGramaj, OrmeSiparisDetay.IgneSayisi,
     OrmeSiparisDetay.MaySayisi, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
     OrmeSiparisDetay.TupAcikEnMayli, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.IplikBoyu1, OrmeSiparisDetay.IplikBoyu2,
     OrmeSiparisDetay.IplikBoyu3, OrmeSiparisDetay.IplikBoyu4, OrmeSiparisDetay.MusteriTalimati, OrmeSiparisDetay.Notu,
     OrmeSiparisDetay.Beyazlik, OrmeSiparisDetay.RaporBoyu, OrmeSiparisDetay.UretimNotu, FirmaYetkili.Adi  from OrmePlanlamaYaz(:AOrmePlanlamaGrupId, :AOrmeMakinaId, :AFirmaId)
  Left Join BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = OrmePlanlamaYaz.BoyaSiparisKayitId
  Left Join SiparisGrup ON SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  Left Join Firma ON Firma.FirmaId = BoyaSiparis.FirmaId
  left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
  Left Join OrmeSiparisDetay ON OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlamaYaz.OrmeSiparisDetayId
  into :OrmePlanlamaId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :MusteriSiparisNo, :ReferansSiparisNo, :SiparisMiktari,
       :SiparisKalanMiktar, :MakinaUstuGramaj, :DinlenmisGramaj, :IgneSayisi, :MaySayisi, :Pus, :Feine, :TupAcikEnMayli, :ArtikelNo, :IplikBoyu1,
       :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :MusteriTalimati, :Notu, :Beyazlik, :RaporBoyu, :UretimNotu, :SiparisiVeren  do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    suspend;
  end
end
