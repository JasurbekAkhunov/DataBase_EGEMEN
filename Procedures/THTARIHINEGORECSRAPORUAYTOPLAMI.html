CREATE PROCEDURE THTARIHINEGORECSRAPORUAYTOPLAMI(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAV VARCHAR(1),
    ACS VARCHAR(1),
    ATAHSIL INTEGER,
    APROTESTO INTEGER,
    AAVUKAT INTEGER,
    ABANKACIKIS INTEGER,
    ABANKAHESAPID VARCHAR(20),
    ABASLANGICCSGRUP1ID VARCHAR(4),
    ABITISCSGRUP1ID VARCHAR(4),
    ABASLANGICCSGRUP2ID VARCHAR(4),
    ABITISCSGRUP2ID VARCHAR(4),
    ABASLANGICCSGRUP3ID VARCHAR(4),
    ABITISCSGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    ACSGRUP1ARALIKLI INTEGER,
    ACSGRUP2ARALIKLI INTEGER,
    ACSGRUP3ARALIKLI INTEGER,
    ADOVIZID VARCHAR(5),
    ARAPORDOVIZID VARCHAR(5),
    ASINIF VARCHAR(1))
RETURNS (
    RAPORAY TIMESTAMP,
    PORTFOYDEBEKLEYEN DOUBLE PRECISION,
    BANKADABEKLEYEN DOUBLE PRECISION,
    TAHSILEDILEN DOUBLE PRECISION)
AS
declare variable LastTahsilAy             timestamp;
declare variable FTahsilAy                timestamp;
declare variable FSonBordroTipi          integer;
declare variable FMeblag                 double precision;
declare variable FTahsilEdilenMeblag     double precision;
begin
  LastTahsilAy = Null;
  For select TahsilTarihiAyToplam, SonBordroTipi, Sum(DovizMeblag), Sum(DovizTahsiledilenMeblag) from TahsilTarihineGoreCSRaporu(:ABaslangicFirmaId, :ABitisFirmaId, :ABaslangicTarihi, :ABitisTarihi, :AAv, :ACs, :ATahsil,
           :AProtesto, :AAvukat, :ABankaCikis, :ABankaHesapId, :ABaslangicCsGrup1Id, :ABitisCsGrup1Id,
           :ABaslangicCsGrup2Id, :ABitisCsGrup2Id, :ABaslangicCsGrup3Id, :ABitisCsGrup3Id, :AFirmaAralikli, :ACSGrup1Aralikli,
           :ACSGrup2Aralikli, :ACSGrup3Aralikli, :ADovizId, :ARaporDovizId, :ASinif)
    group by TahsilTarihiAyToplam,  SonBordroTipi
    into :FTahsilAy, :FSonBordroTipi, :FMeblag, :FTahsilEdilenMeblag do
  begin
    if (LastTahsilAy is null or LastTahsilAy <> FTahsilAy) then
    begin
      if (LastTahsilAy is not null) then suspend;
      LastTahsilAy = FTahsilAy;
      RaporAy = FTahsilAy;
      PortfoydeBekleyen = 0;
      BankadaBekleyen   = 0;
      TahsilEdilen      = 0;
    end
    if (FSonBordroTipi = 0 or FSonBordroTipi = 2 or FSonBordroTipi = 4 or FSonBordroTipi = 6) then PortfoydeBekleyen = PortfoydeBekleyen + FMeblag - FTahsilEdilenMeblag;
    else if (FSonBordroTipi = 3) then BankadaBekleyen = BankadaBekleyen + FMeblag - FTahsilEdilenMeblag;
    TahsilEdilen = TahsilEdilen + FTahsilEdilenMeblag;
  end
  if (LastTahsilAy is not null) then suspend;
end
