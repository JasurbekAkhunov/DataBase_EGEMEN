CREATE PROCEDURE ORMESIPDETAYLISTESI(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    SIPARISSIRANO INTEGER,
    ORMESIPARISDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MUSTERISIPARISNO VARCHAR(50),
    SIPARISKILO NUMERIC(12,3),
    MELANJKODU VARCHAR(20))
AS
declare variable hstIplik        integer;
begin
  for select OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.MusteriSiparisNo,
             OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.MelanjKodu from OrmeSiparisDetay
  left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
  where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  order by OrmeSiparisDetay.OrmeSiparisDetayId
  into :OrmeSiparisDetayId, :HamId, :HamAdi, :MusteriSiparisNo, :SiparisSiraNo, :SiparisKilo, :MelanjKodu do
  begin
    suspend;
  end
end
