CREATE PROCEDURE DETAYLIHAMSTOKANALIZI(
    AFASON INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    ATEMSILCIID INTEGER,
    AHAMSTOKTIPI INTEGER,
    AKALANISIFIROLANLARIDOK INTEGER,
    AONAYVERILMEMISLERDAHIL INTEGER,
    ATUTARGOSTER INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    DEVREDENMIKTAR NUMERIC(12,3),
    DEVREDENMIKTARTUTAR DOUBLE PRECISION,
    GIRISMIKTAR NUMERIC(12,3),
    GIRISMIKTARTUTAR DOUBLE PRECISION,
    PARTILENENMIKTAR NUMERIC(12,3),
    PARTILENENMIKTARTUTAR DOUBLE PRECISION,
    IADEEDILENMIKTAR NUMERIC(12,3),
    IADEEDILENMIKTARTUTAR DOUBLE PRECISION,
    SATISMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    KALANMIKTARTUTAR DOUBLE PRECISION)
AS
declare variable Okey         integer;
declare variable FirmaGrup1   varchar(4);
declare variable FirmaGrup2   varchar(4);
declare variable FirmaGrup3   varchar(4);
declare variable FirmaKodumuz varchar(20);
declare variable hstIplik     integer;
declare variable hstElyaf     integer;
declare variable hstTelef     integer;
declare variable biAlisIade   integer;
declare variable Miktar       Numeric(12,3);
declare variable Tutar        Double Precision;
declare variable DevirTutar   Double Precision;
declare variable GirisDevirMiktar BoyaMiktarSize;
declare variable GirisDevirMiktarTutar Double Precision;
declare variable PartilenenDevirMiktar BoyaMiktarSize;
declare variable PartilenenDevirMiktarTutar Double Precision;
declare variable IadeEdilenDevirMiktar BoyaMiktarSize;
declare variable IadeEdilenDevirMiktarTutar Double Precision;
declare variable SatisDevirMiktar BoyaMiktarSize;
declare variable SatisDevirMiktarTutar Double Precision;
declare variable SatisMiktarTutar Double Precision;
declare variable SarfDevirMiktar BoyaMiktarSize;
declare variable SarfDevirMiktarTutar Double Precision;
declare variable SarfMiktar BoyaMiktarSize;
declare variable SarfMiktarTutar Double Precision;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  Select biAlisIade from BoyaIrsaliyeHareketTipiGetir into :biAlisIade;

  Select FirmaKodumuz from Sistem into :FirmaKodumuz;

  if (AFason = 0) then
  begin
    for Select Firma.FirmaId, Firma.Adi, Firma.FirmaGrup1, Firma.FirmaGrup2, Firma.FirmaGrup3 from Firma
    where ((:AFirmaAralikli = 1 and Firma.FirmaId >= :ABaslangicFirmaId and (Firma.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0))  and
          ((:AFirmaGrup1Aralikli = 1 and FirmaGrup1 >= :ABaslangicGrup1Id  and (FirmaGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
          ((:AFirmaGrup2Aralikli = 1 and FirmaGrup2 >= :ABaslangicGrup2Id and (FirmaGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
          ((:AFirmaGrup3Aralikli = 1 and FirmaGrup3 >= :ABaslangicGrup3Id and (FirmaGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
          (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null)
    into :FirmaId, :FirmaAdi, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3  do
    begin
      Okey = 1;

      if (FirmaId = FirmaKodumuz) then Okey = 0;

      if (okey = 1) then
      if (AFirmaAralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup1Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicGrup1Id, FirmaGrup1) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup2Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicGrup2Id, FirmaGrup2) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup3Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicGrup3Id, FirmaGrup3) = 0) then okey = 0;

      if (okey = 1) then
      begin
        For Select ham.HamId, Ham.Adi from Ham
           where ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamStokTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamStokTipi = 1) or :AHamStokTipi is null) and
                 Ham.Kapali = 0
           into :HamId, :HamAdi do
        begin
          if (ATutarGoster = 1) then
          begin
            select Sum(iif(HamStokMaster.Tarih < :ABaslangicTarihi, HamStokMaster.Miktar, 0)),
            Sum(iif(HamStokMaster.Tarih < :ABaslangicTarihi, HamStokMaster.Miktar * HamStokMaster.BirimFiyat, 0)),
            Sum(iif(HamStokMaster.Tarih >= :ABaslangicTarihi, HamStokMaster.Miktar, 0)),
            Sum(iif(HamStokMaster.Tarih >= :ABaslangicTarihi, HamStokMaster.Miktar * HamStokMaster.BirimFiyat, 0))  from HamStokMaster
            where HamStokMaster.PartiSinifId is null and HamStokMaster.StokFirmaId = :FirmaId and HamStokMaster.HamId = :HamId and HamStokMaster.Tarih <= :ABitisTarihi and
            (HamStokMaster.OnayDurumu <> 0 or :AOnayVerilmemislerDahil = 1 and HamStokMaster.OnayDurumu = 0)
            into :GirisDevirMiktar, :GirisDevirMiktarTutar, :GirisMiktar, :GirisMiktarTutar;

            select Coalesce(sum(iif(partistokiptal.iptaltarihi < :ABaslangicTarihi, partistok.kilo, 0)), 0) + Coalesce(:GirisDevirMiktar, 0),
            :GirisDevirMiktarTutar,
            Coalesce(sum(iif(partistokiptal.iptaltarihi >= :ABaslangicTarihi, partistok.kilo, 0)), 0) + Coalesce(:GirisMiktar, 0),
            :GirisMiktarTutar from partistokiptal, partistok, parti
            where partistokiptal.iptaltarihi < :ABitisTarihi and partistokiptal.partisinifid = partistok.partisinifid
            and partistok.hamid = :HamId and parti.partikayitid = partistok.partikayitid
            and (parti.fason = 1 and parti.firmaid = :FirmaId)
            and partistokiptal.DesenTasarimIptal = 0
            into :GirisDevirMiktar, :GirisDevirMiktarTutar, :GirisMiktar, :GirisMiktarTutar;

            select Sum(iif(PartiStokDetay.PartilemeTarihi < :ABaslangicTarihi, PartiStokDetay.Miktar, 0)),
            Sum(iif(PartiStokDetay.PartilemeTarihi < :ABaslangicTarihi, PartiStokDetay.Miktar * HamStokMaster.BirimFiyat, 0)),
            Sum(iif(PartiStokDetay.PartilemeTarihi >= :ABaslangicTarihi, PartiStokDetay.Miktar, 0)),
            Sum(iif(PartiStokDetay.PartilemeTarihi >= :ABaslangicTarihi, PartiStokDetay.Miktar * HamStokMaster.BirimFiyat, 0)) from PartiStokDetay
            left join Parti on Parti.PartiKayitId = PartiStokDetay.PartiKayitId
            left join PartiStokIptal on PartiStokIptal.PartiSinifId =  PartiStokDetay.PartiSinifId
            left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
            left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
            left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
            where HamStokMaster.PartiSinifId is null and PartiStokDetay.RafDetayId is not null and (Parti.FirmaId = :FirmaId and Parti.Fason = 1)  and PartiStokDetay.HamId = :HamId and
            PartiStokDetay.PartilemeTarihi <= :ABitisTarihi
            --and PartiStokIptal.PartiStokIptalId is null
            into :PartilenenDevirMiktar, :PartilenenDevirMiktarTutar, :PartilenenMiktar, :PartilenenMiktarTutar;


            select Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi < :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar, 0)),
            Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi < :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar * BoyaIrsaliyeDetay.BirimFiyat, 0)),
            Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar, 0)),
            Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar * BoyaIrsaliyeDetay.BirimFiyat, 0)) from BoyaIrsaliyeDetay
            left join RafDetay on RafDetay.RafDetayId = BoyaIrsaliyeDetay.RafDetayId
            left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
            left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
            where HamStokDetay.StokFirmaId = :FirmaId and BoyaIrsaliyeDetay.HamId = :HamId and BoyaIrsaliyeDetay.IrsaliyeTarihi <= :ABitisTarihi
            and (BoyaIrsaliyeDetay.FasonHamIade = 1)
            and BoyaIrsaliyeDetay.rafdetayid is not null
            into :IadeEdilenDevirMiktar, :IadeEdilenDevirMiktarTutar, :IadeEdilenMiktar, :IadeEdilenMiktarTutar;

            /*select Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi < :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar, 0)),
            Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi < :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar * BoyaIrsaliyeDetay.BirimFiyat, 0)),
            Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar, 0)),
            Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar * BoyaIrsaliyeDetay.BirimFiyat, 0)) from BoyaIrsaliyeDetay
            where BoyaIrsaliyeDetay.FirmaId = :FirmaId and BoyaIrsaliyeDetay.HamId = :HamId and BoyaIrsaliyeDetay.IrsaliyeTarihi <= :ABitisTarihi
            and (BoyaIrsaliyeDetay.FasonHamIade = 0 and BoyaIrsaliyeDetay.HareketTipiId <> :biAlisIade)
            and BoyaIrsaliyeDetay.rafdetayid is not null
            into :SatisDevirMiktar, :SatisDevirMiktarTutar, :SatisMiktar, :SatisMiktarTutar;*/

            select Sum(iif(HamStokSarfFisiDetay.FisTarihi < :ABaslangicTarihi, HamStokSarfFisiDetay.Miktar, 0)),
            Sum(iif(HamStokSarfFisiDetay.FisTarihi < :ABaslangicTarihi, HamStokSarfFisiDetay.Miktar * HamStokDetay.BirimFiyat, 0)),
            Sum(iif(HamStokSarfFisiDetay.FisTarihi >= :ABaslangicTarihi, HamStokSarfFisiDetay.Miktar, 0)),
            Sum(iif(HamStokSarfFisiDetay.FisTarihi >= :ABaslangicTarihi, HamStokSarfFisiDetay.Miktar * HamStokDetay.BirimFiyat, 0)) from HamStokSarfFisiDetay
            left join RafDetay on RafDetay.RafDetayId = HamStokSarfFisiDetay.RafDetayId
            left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayID where HamStokSarfFisiDetay.HamId = :HamId and HamStokSarfFisiDetay.FisTarihi <= :ABitisTarihi
            and HamStokDetay.Fason = 1 and HamStokDetay.StokFirmaId = :FirmaId
            and HamStokSarfFisiDetay.rafdetayid is not null
            --into :SatisDevirMiktar, :SatisDevirMiktarTutar, :SatisMiktar, :SatisMiktarTutar;
            into :SarfDevirMiktar, :SarfDevirMiktarTutar, :SarfMiktar, :SarfMiktarTutar;

            select Coalesce(Sum(iif(HamStokSarfFisiDetay.FisTarihi < :ABaslangicTarihi, HamStokSarfFisiDetay.BrutMiktar, 0)), 0),
            coalesce(Sum(iif(HamStokSarfFisiDetay.FisTarihi < :ABaslangicTarihi, 0, 0)), 0),
            coalesce(Sum(iif(HamStokSarfFisiDetay.FisTarihi >= :ABaslangicTarihi, HamStokSarfFisiDetay.BrutMiktar, 0)), 0),
            coalesce(Sum(iif(HamStokSarfFisiDetay.FisTarihi >= :ABaslangicTarihi, 0, 0)), 0) from HamStokSarfFisiDetay
            left join PartiStok on PartiStok.PartiSinifID = HamStokSarfFisiDetay.PartiSinifId
            left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitID
            where HamStokSarfFisiDetay.HamId = :HamId and HamStokSarfFisiDetay.FisTarihi <= :ABitisTarihi
            and Parti.Fason = 1
            and Parti.FirmaId = :FirmaId
            and HamStokSarfFisiDetay.PartiSinifId is not null
            into :SatisDevirMiktar, :SatisDevirMiktarTutar, :SatisMiktar, :SatisMiktarTutar;
            --into :SarfDevirMiktar, :SarfDevirMiktarTutar, :SarfMiktar, :SarfMiktarTutar;

            if (GirisMiktar is null) then GirisMiktar = 0.0;
            if (PartilenenMiktar is null) then PartilenenMiktar = 0.0;
            if (IadeEdilenMiktar is null) then IadeEdilenMiktar = 0.0;
            if (SatisMiktar is null) then SatisMiktar = 0.0;
            if (GirisMiktarTutar is null) then GirisMiktarTutar = 0.0;
            if (PartilenenMiktarTutar is null) then PartilenenMiktarTutar = 0.0;
            if (IadeEdilenMiktarTutar is null) then IadeEdilenMiktarTutar = 0.0;
            if (SatisMiktarTutar is null) then SatisMiktarTutar = 0.0;
            if (GirisDevirMiktar is null) then GirisDevirMiktar = 0.0;
            if (PartilenenDevirMiktar is null) then PartilenenDevirMiktar = 0.0;
            if (IadeEdilenDevirMiktar is null) then IadeEdilenDevirMiktar = 0.0;
            if (SatisDevirMiktar is null) then SatisDevirMiktar = 0.0;
            if (GirisDevirMiktarTutar is null) then GirisDevirMiktarTutar = 0.0;
            if (PartilenenDevirMiktarTutar is null) then PartilenenDevirMiktarTutar = 0.0;
            if (IadeEdilenDevirMiktarTutar is null) then IadeEdilenDevirMiktarTutar = 0.0;
            if (SatisDevirMiktarTutar is null) then SatisDevirMiktarTutar = 0.0;
            if (SarfDevirMiktar is null) then SarfDevirMiktar = 0.0;
            if (SarfDevirMiktarTutar is null) then SarfDevirMiktarTutar = 0.0;
            if (SarfMiktar is null) then SarfMiktar = 0.0;
            if (SarfMiktarTutar is null) then SarfMiktarTutar = 0.0;

            DevredenMiktar = GirisDevirMiktar - PartilenenDevirMiktar  - IadeEdilenDevirMiktar - SatisDevirMiktar - SarfDevirMiktar;
            DevredenMiktarTutar = GirisDevirMiktarTutar - PartilenenDevirMiktarTutar  - IadeEdilenDevirMiktarTutar - SatisDevirMiktarTutar - SarfDevirMiktarTutar;
            KalanMiktar = DevredenMiktar + GirisMiktar - (PartilenenMiktar + IadeEdilenMiktar + SatisMiktar + SarfMiktar);
            KalanMiktarTutar = DevredenMiktarTutar + GirisMiktarTutar - (PartilenenMiktarTutar + IadeEdilenMiktartutar + SatisMiktarTutar + SarfMiktarTutar);

            if (DevredenMiktar = 0 and GirisMiktar = 0 and PartilenenMiktar = 0 and IadeEdilenMiktar = 0 and SatisMiktar = 0 and SarfMiktar = 0) then Okey = 0;
            else Okey = 1;

            if (Okey = 1) then
            if (AKalaniSifirOlanlariDok = 1 or KalanMiktar <> 0) then suspend;
          end
          else
          begin
            select sum(girisMiktar), sum(PartilenenMiktar), sum(iadeedilenMiktar), sum(satisMiktar) from HamStokKumilatif
            where HamStokKumilatif.Firmaid = :FirmaId and HamStokKumilatif.HamId = :HamId and HamStokKumilatif.Tarih < :ABaslangicTarihi
            into :GirisMiktar, :PartilenenMiktar, :IadeEdilenMiktar, :SatisMiktar;

            if (GirisMiktar is null) then GirisMiktar = 0.0;
            if (PartilenenMiktar is null) then PartilenenMiktar = 0.0;
            if (IadeEdilenMiktar is null) then IadeEdilenMiktar = 0.0;
            if (SatisMiktar is null) then SatisMiktar = 0.0;

            DevredenMiktar = GirisMiktar - PartilenenMiktar  - IadeEdilenMiktar - SatisMiktar;

            if (AOnayVerilmemislerDahil = 1) then
            begin
              Miktar = 0;

              Select Sum(HamStokMaster.Miktar) from HamStokMaster
                where HamStokMaster.StokFirmaId = :FirmaId and HamStokMaster.HamID = :HamId and HamStokMaster.Tarih < :ABaslangicTarihi and
                      HamStokMaster.OnayDurumu = 0
                into :Miktar;

              if (Miktar is not null) then DevredenMiktar = DevredenMiktar + Miktar;
            end

            select sum(girisMiktar), Sum(PartilenenMiktar), Sum(iadeedilenMiktar), Sum(SatisMiktar) from HamStokKumilatif
              where HamStokKumilatif.FirmaId = :FirmaId and HamStokKumilatif.HamId = :HamId and HamStokKumilatif.Tarih >= :ABaslangicTarihi and HamStokKumilatif.Tarih <= :ABitisTarihi
              into :GirisMiktar, :PartilenenMiktar, :IadeEdilenMiktar, :SatisMiktar;

            if (GirisMiktar is null) then GirisMiktar = 0.0;
            if (PartilenenMiktar is null) then PartilenenMiktar = 0.0;
            if (IadeEdilenMiktar is null) then IadeEdilenMiktar = 0.0;
            if (SatisMiktar is null) then SatisMiktar = 0.0;

            if (AOnayVerilmemislerDahil = 1) then
            begin
              Miktar = 0;

              Select Sum(HamStokMaster.Miktar) from HamStokMaster
                where HamStokMaster.StokFirmaId = :FirmaId and HamStokMaster.HamID = :HamId and HamStokMaster.Tarih >= :ABaslangicTarihi and HamStokMaster.Tarih <= :ABitisTarihi and
                      HamStokMaster.OnayDurumu = 0
                into :Miktar;

              if (Miktar is not null) then GirisMiktar = GirisMiktar + Miktar;
            end

            KalanMiktar = DevredenMiktar + GirisMiktar - (PartilenenMiktar + IadeEdilenMiktar + SatisMiktar);

            if (DevredenMiktar = 0 and GirisMiktar = 0 and PartilenenMiktar = 0 and IadeEdilenMiktar = 0 and SatisMiktar = 0) then Okey = 0;
            else Okey = 1;

            if (Okey = 1) then
            if (AKalaniSifirOlanlariDok = 1 or KalanMiktar <> 0) then suspend;
          end
        end
      end
    end
  end
  else
  begin
    For Select ham.HamId, Ham.Adi from Ham
       where ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamStokTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamStokTipi = 1) or :AHamStokTipi is null) and
             Ham.Kapali = 0
       into :HamId, :HamAdi do
    begin
      if (ATutarGoster = 1) then
      begin
        select Sum(iif(HamStokMaster.Tarih < :ABaslangicTarihi, HamStokMaster.Miktar, 0)),
        Sum(iif(HamStokMaster.Tarih < :ABaslangicTarihi, HamStokMaster.Miktar * HamStokMaster.BirimFiyat, 0)),
        Sum(iif(HamStokMaster.Tarih >= :ABaslangicTarihi, HamStokMaster.Miktar, 0)),
        Sum(iif(HamStokMaster.Tarih >= :ABaslangicTarihi, HamStokMaster.Miktar * HamStokMaster.BirimFiyat, 0))  from HamStokMaster
        where HamStokMaster.PartiSinifId is null and HamStokMaster.StokFirmaId = :FirmaKodumuz and HamStokMaster.HamId = :HamId and HamStokMaster.Tarih <= :ABitisTarihi and
        (HamStokMaster.OnayDurumu <> 0 or :AOnayVerilmemislerDahil = 1 and HamStokMaster.OnayDurumu = 0)
        into :GirisDevirMiktar, :GirisDevirMiktarTutar, :GirisMiktar, :GirisMiktarTutar;

        select Coalesce(sum(iif(partistokiptal.iptaltarihi < :ABaslangicTarihi, partistok.kilo, 0)), 0) + Coalesce(:GirisDevirMiktar, 0),
        :GirisDevirMiktarTutar,
        Coalesce(sum(iif(partistokiptal.iptaltarihi >= :ABaslangicTarihi, partistok.kilo, 0)), 0) + Coalesce(:GirisMiktar, 0),
        :GirisMiktarTutar from partistokiptal, partistok, parti
        where partistokiptal.iptaltarihi < :ABitisTarihi and partistokiptal.partisinifid = partistok.partisinifid
        and partistok.hamid = :HamId and parti.partikayitid = partistok.partikayitid
        and (parti.fason = 0 or parti.firmaid = :FirmaKodumuz)
        and partistokiptal.DesenTasarimIptal = 0
        into :GirisDevirMiktar, :GirisDevirMiktarTutar, :GirisMiktar, :GirisMiktarTutar;

        select Sum(iif(PartiStokDetay.PartilemeTarihi < :ABaslangicTarihi, PartiStokDetay.Miktar, 0)),
        Sum(iif(PartiStokDetay.PartilemeTarihi < :ABaslangicTarihi, PartiStokDetay.Miktar * HamStokMaster.BirimFiyat, 0)),
        Sum(iif(PartiStokDetay.PartilemeTarihi >= :ABaslangicTarihi, PartiStokDetay.Miktar, 0)),
        Sum(iif(PartiStokDetay.PartilemeTarihi >= :ABaslangicTarihi, PartiStokDetay.Miktar * HamStokMaster.BirimFiyat, 0)) from PartiStokDetay
        left join Parti on Parti.PartiKayitId = PartiStokDetay.PartiKayitId
        left join PartiStokIptal on PartiStokIptal.PartiSinifId =  PartiStokDetay.PartiSinifId
        left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where HamStokMaster.PartiSinifId is null and PartiStokDetay.RafDetayId is not null and (Parti.FirmaId = :FirmaKodumuz or Parti.Fason = 0)  and PartiStokDetay.HamId = :HamId and
        PartiStokDetay.PartilemeTarihi <= :ABitisTarihi
        --and PartiStokIptal.PartiStokIptalId is null
        into :PartilenenDevirMiktar, :PartilenenDevirMiktarTutar, :PartilenenMiktar, :PartilenenMiktarTutar;

        select Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi < :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar, 0)),
        Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi < :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar * BoyaIrsaliyeDetay.BirimFiyat, 0)),
        Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar, 0)),
        Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar * BoyaIrsaliyeDetay.BirimFiyat, 0)) from BoyaIrsaliyeDetay
        where BoyaIrsaliyeDetay.HamId = :HamId and BoyaIrsaliyeDetay.IrsaliyeTarihi <= :ABitisTarihi
        and BoyaIrsaliyeDetay.FasonHamIade = 0
        and BoyaIrsaliyeDetay.HareketTipiId = :biAlisIade
        and BoyaIrsaliyeDetay.rafdetayid is not null
        into :IadeEdilenDevirMiktar, :IadeEdilenDevirMiktarTutar, :IadeEdilenMiktar, :IadeEdilenMiktarTutar;

        select Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi < :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar, 0)),
        Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi < :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar * BoyaIrsaliyeDetay.BirimFiyat, 0)),
        Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar, 0)),
        Sum(iif(BoyaIrsaliyeDetay.IrsaliyeTarihi >= :ABaslangicTarihi, BoyaIrsaliyeDetay.Miktar * BoyaIrsaliyeDetay.BirimFiyat, 0)) from BoyaIrsaliyeDetay
        where BoyaIrsaliyeDetay.HamId = :HamId and BoyaIrsaliyeDetay.IrsaliyeTarihi <= :ABitisTarihi
        and (BoyaIrsaliyeDetay.FasonHamIade <> 1 and BoyaIrsaliyeDetay.HareketTipiId <> :biAlisIade)
        and BoyaIrsaliyeDetay.rafdetayid is not null
        --and BoyaIrsaliyeDetay.FirmaId = :FirmaKodumuz
        into :SatisDevirMiktar, :SatisDevirMiktarTutar, :SatisMiktar, :SatisMiktarTutar;

        select Sum(iif(HamStokSarfFisiDetay.FisTarihi < :ABaslangicTarihi, HamStokSarfFisiDetay.Miktar, 0)),
        Sum(iif(HamStokSarfFisiDetay.FisTarihi < :ABaslangicTarihi, HamStokSarfFisiDetay.Miktar * HamStokDetay.BirimFiyat, 0)),
        Sum(iif(HamStokSarfFisiDetay.FisTarihi >= :ABaslangicTarihi, HamStokSarfFisiDetay.Miktar, 0)),
        Sum(iif(HamStokSarfFisiDetay.FisTarihi >= :ABaslangicTarihi, HamStokSarfFisiDetay.Miktar * HamStokDetay.BirimFiyat, 0)) from HamStokSarfFisiDetay
        left join RafDetay on RafDetay.RafDetayId = HamStokSarfFisiDetay.RafDetayId
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayID
        where HamStokSarfFisiDetay.HamId = :HamId and HamStokSarfFisiDetay.FisTarihi <= :ABitisTarihi
        and HamStokDetay.Fason = 0
        --and HamStokDetay.StokFirmaId = :FirmaKodumuz
        and HamStokSarfFisiDetay.rafdetayid is not null
        --into :SatisDevirMiktar, :SatisDevirMiktarTutar, :SatisMiktar, :SatisMiktarTutar;
        into :SarfDevirMiktar, :SarfDevirMiktarTutar, :SarfMiktar, :SarfMiktarTutar;

        select Coalesce(Sum(iif(HamStokSarfFisiDetay.FisTarihi < :ABaslangicTarihi, HamStokSarfFisiDetay.BrutMiktar, 0)), 0) + Coalesce(:SarfDevirMiktar, 0),
        coalesce(Sum(iif(HamStokSarfFisiDetay.FisTarihi < :ABaslangicTarihi, 0, 0)), 0) + coalesce(:SarfDevirMiktarTutar, 0),
        Coalesce(Sum(iif(HamStokSarfFisiDetay.FisTarihi >= :ABaslangicTarihi, HamStokSarfFisiDetay.BrutMiktar, 0)), 0) + Coalesce(:SarfMiktar, 0),
        coalesce(Sum(iif(HamStokSarfFisiDetay.FisTarihi >= :ABaslangicTarihi, 0, 0)), 0) + coalesce(:SarfMiktarTutar, 0) from HamStokSarfFisiDetay
        left join PartiStok on PartiStok.PartiSinifID = HamStokSarfFisiDetay.PartiSinifId
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitID
        where HamStokSarfFisiDetay.HamId = :HamId and HamStokSarfFisiDetay.FisTarihi <= :ABitisTarihi
        and Parti.Fason = 0
        --and Parti.FirmaId = :FirmaKodumuz
        and HamStokSarfFisiDetay.PartiSinifId is not null
        --into :SatisDevirMiktar, :SatisDevirMiktarTutar, :SatisMiktar, :SatisMiktarTutar;
        into :SarfDevirMiktar, :SarfDevirMiktarTutar, :SarfMiktar, :SarfMiktarTutar;


        if (GirisMiktar is null) then GirisMiktar = 0.0;
        if (PartilenenMiktar is null) then PartilenenMiktar = 0.0;
        if (IadeEdilenMiktar is null) then IadeEdilenMiktar = 0.0;
        if (SatisMiktar is null) then SatisMiktar = 0.0;
        if (GirisMiktarTutar is null) then GirisMiktarTutar = 0.0;
        if (PartilenenMiktarTutar is null) then PartilenenMiktarTutar = 0.0;
        if (IadeEdilenMiktarTutar is null) then IadeEdilenMiktarTutar = 0.0;
        if (SatisMiktarTutar is null) then SatisMiktarTutar = 0.0;
        if (GirisDevirMiktar is null) then GirisDevirMiktar = 0.0;
        if (PartilenenDevirMiktar is null) then PartilenenDevirMiktar = 0.0;
        if (IadeEdilenDevirMiktar is null) then IadeEdilenDevirMiktar = 0.0;
        if (SatisDevirMiktar is null) then SatisDevirMiktar = 0.0;
        if (GirisDevirMiktarTutar is null) then GirisDevirMiktarTutar = 0.0;
        if (PartilenenDevirMiktarTutar is null) then PartilenenDevirMiktarTutar = 0.0;
        if (IadeEdilenDevirMiktarTutar is null) then IadeEdilenDevirMiktarTutar = 0.0;
        if (SatisDevirMiktarTutar is null) then SatisDevirMiktarTutar = 0.0;
        if (SarfDevirMiktar is null) then SarfDevirMiktar = 0.0;
        if (SarfDevirMiktarTutar is null) then SarfDevirMiktarTutar = 0.0;
        if (SarfMiktar is null) then SarfMiktar = 0.0;
        if (SarfMiktarTutar is null) then SarfMiktarTutar = 0.0;

        DevredenMiktar = GirisDevirMiktar - PartilenenDevirMiktar - IadeEdilenDevirMiktar - SatisDevirMiktar - SarfDevirMiktar;
        DevredenMiktarTutar = GirisDevirMiktarTutar - PartilenenDevirMiktarTutar  - IadeEdilenDevirMiktarTutar - SatisDevirMiktarTutar - SarfDevirMiktarTutar;
        KalanMiktar = DevredenMiktar + GirisMiktar - (PartilenenMiktar + IadeEdilenMiktar + SatisMiktar + SarfMiktar);
        KalanMiktarTutar = DevredenMiktarTutar + GirisMiktarTutar - (PartilenenMiktarTutar + IadeEdilenMiktartutar + SatisMiktarTutar + SarfMiktarTutar);

        if (DevredenMiktar = 0 and GirisMiktar = 0 and PartilenenMiktar = 0 and IadeEdilenMiktar = 0 and SatisMiktar = 0 and SarfMiktar = 0) then Okey = 0;
        else Okey = 1;

        if (Okey = 1) then
        if (AKalaniSifirOlanlariDok = 1 or KalanMiktar <> 0) then suspend;
      end
      else
      begin
        select sum(girisMiktar), sum(PartilenenMiktar), sum(iadeedilenMiktar), sum(satisMiktar) from HamStokKumilatif
        where HamStokKumilatif.Firmaid = :FirmaKodumuz and HamStokKumilatif.HamId = :HamId and HamStokKumilatif.Tarih < :ABaslangicTarihi
        into :GirisMiktar, :PartilenenMiktar, :IadeEdilenMiktar, :SatisMiktar;

        if (GirisMiktar is null) then GirisMiktar = 0.0;
        if (PartilenenMiktar is null) then PartilenenMiktar = 0.0;
        if (IadeEdilenMiktar is null) then IadeEdilenMiktar = 0.0;
        if (SatisMiktar is null) then SatisMiktar = 0.0;

        DevredenMiktar = GirisMiktar - PartilenenMiktar  - IadeEdilenMiktar - SatisMiktar;

        if (AOnayVerilmemislerDahil = 1) then
        begin
          Miktar = 0;

          Select Sum(HamStokMaster.Miktar) from HamStokMaster
            where HamStokMaster.StokFirmaId = :FirmaKodumuz and HamStokMaster.HamID = :HamId and HamStokMaster.Tarih < :ABaslangicTarihi and
                  HamStokMaster.OnayDurumu = 0
            into :Miktar;

          if (Miktar is not null) then DevredenMiktar = DevredenMiktar + Miktar;
        end

        select sum(girisMiktar), Sum(PartilenenMiktar), Sum(iadeedilenMiktar), Sum(SatisMiktar) from HamStokKumilatif
          where HamStokKumilatif.FirmaId = :FirmaKodumuz and HamStokKumilatif.HamId = :HamId and HamStokKumilatif.Tarih >= :ABaslangicTarihi and HamStokKumilatif.Tarih <= :ABitisTarihi
          into :GirisMiktar, :PartilenenMiktar, :IadeEdilenMiktar, :SatisMiktar;

        if (GirisMiktar is null) then GirisMiktar = 0.0;
        if (PartilenenMiktar is null) then PartilenenMiktar = 0.0;
        if (IadeEdilenMiktar is null) then IadeEdilenMiktar = 0.0;
        if (SatisMiktar is null) then SatisMiktar = 0.0;

        if (AOnayVerilmemislerDahil = 1) then
        begin
          Miktar = 0;

          Select Sum(HamStokMaster.Miktar) from HamStokMaster
            where HamStokMaster.PartiSinifId is null and HamStokMaster.StokFirmaId = :FirmaKodumuz and HamStokMaster.HamID = :HamId and HamStokMaster.Tarih >= :ABaslangicTarihi and HamStokMaster.Tarih <= :ABitisTarihi and
                  HamStokMaster.OnayDurumu = 0
            into :Miktar;

          if (Miktar is not null) then GirisMiktar = GirisMiktar + Miktar;
        end

        KalanMiktar = DevredenMiktar + GirisMiktar - (PartilenenMiktar + IadeEdilenMiktar + SatisMiktar);

        if (DevredenMiktar = 0 and GirisMiktar = 0 and PartilenenMiktar = 0 and IadeEdilenMiktar = 0 and SatisMiktar = 0) then Okey = 0;
        else Okey = 1;

        if (Okey = 1) then
        if (AKalaniSifirOlanlariDok = 1 or KalanMiktar <> 0) then suspend;
      end
    end
  end
