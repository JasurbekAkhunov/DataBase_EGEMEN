CREATE PROCEDURE BUKUMTALIMATRECETEURUNLERI(
    ABUKUMTALIMATID INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMTIPIIDSTR VARCHAR(15),
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    PUS INTEGER,
    FEINE INTEGER,
    LOT VARCHAR(20),
    RENKID INTEGER,
    RESULTID VARCHAR(255),
    CIKISMIKTARI BOYAMIKTARSIZE,
    IADEMIKTARI BOYAMIKTARSIZE)
AS
begin
  For Select Ham.HamId, Ham.HamTipiId, HamTipi.Adi, Ham.Adi, Ham.Pus, Ham.Feine, BukumTalimatRecete.Lot, BukumTalimatRecete.RenkID, BukumTalimatRecete.CikisMiktari, BukumTalimatRecete.IadeMiktari from BukumTalimatRecete
    Left Join Ham On Ham.HamId = BukumTalimatRecete.HamId
    Left Join HamTipi On HamTipi.HamTipiId = Ham.HamTipiId
    where BukumTalimatRecete.BukumTalimatId = :ABukumTalimatId
  into  :HamId, :HamTipiId, :HamTipiIdStr, :HamAdi, :Pus, :Feine, :Lot, :RenkID, :CikisMiktari, :IadeMiktari do
  begin
    ResultId = '"' || EgemenIntegerToStr(HamId) || '",' ;

    if (Lot is not null) then
      ResultId = ResultId || '"' || Lot || '",' ;
    else
      ResultId = ResultId || '"",' ;

    if (RenkId is null) then
      ResultId = ResultId || '""' ;
    else
      ResultId = ResultId || '"' || RenkId || '"' ;
    suspend;
  end
end
