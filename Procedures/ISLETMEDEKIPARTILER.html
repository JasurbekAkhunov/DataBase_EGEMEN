CREATE PROCEDURE ISLETMEDEKIPARTILER(
    ADURUMU INTEGER,
    AFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AISLEMGRUBUID INTEGER,
    AISLEMTIPIID INTEGER,
    AACILIYETDERECESIID INTEGER,
    ARECETEID VARCHAR(20),
    AMODELNO VARCHAR(20),
    ABASLANGICGECENSURE INTEGER,
    ABITISGECENSURE INTEGER,
    AISLETMEICITAMIR INTEGER,
    AASAMAID INTEGER,
    ARECETEYAZILDI INTEGER,
    APARTIGRUPID VARCHAR(4))
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIKAYITID INTEGER,
    PARTITIPIID INTEGER,
    PARTITIPIIDSTR VARCHAR(10),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FASON INTEGER,
    FASONSATISSTR VARCHAR(5),
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ISLEMTIPIID INTEGER,
    ISLEMGRUBUID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    ACILIYETDERECESIID INTEGER,
    ACILIYETDERECESIADI VARCHAR(20),
    ASAMAVAR VARCHAR(3),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    PANTONEKODU VARCHAR(20),
    RECETEYAZILDI INTEGER,
    RECETEYAZILDISTR VARCHAR(4),
    RECETEDETAYVAR INTEGER,
    RECETEDETAYVARSTR VARCHAR(3),
    PARTIKAPADEDI INTEGER,
    PARTIKILO NUMERIC(12,3),
    PARTIMETRE NUMERIC(12,3),
    MODELNO VARCHAR(20),
    ACILISTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30),
    GECIKMETIPIADI VARCHAR(30),
    GECIKMENOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    OTOMATIKPOSTA INTEGER,
    OTOMATIKPOSTASTR VARCHAR(4),
    RAFID VARCHAR(10),
    PARTIRPTDEGERLENDIRME INTEGER,
    PARTIRPTDEGERLENDIRMESTR VARCHAR(40),
    PARTIRPTBASLAMATARIHI TIMESTAMP,
    PARTIRPTBITISTARIHI TIMESTAMP,
    DEGISIMLOG BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable T1 timestamp;
declare variable T2 timestamp;
declare variable KayitSayisi integer;
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable GDurumu        integer;
declare variable Bos            integer;
begin
  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  if (ADurumu is null) then
  begin
    for select distinct Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi,Parti.IslemTipiId, IslemTipi.Adi, Parti.IslemGrubuId, IslemGrubu.Adi, Parti.AciliyetDerecesiId,
          AciliyetDerecesi.Adi,  Parti.ReceteId, Recete.Adi, Parti.ReceteYazildi, Recete.ReceteDetayVar, Parti.Modelno, Parti.Acilistarihi, Parti.Termintarihi, PartiGecikme.Notu, GecikmeTipi.Adi, Parti.RafId,
           BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.Fason,
           BoyaSiparisBaski.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, LabRecete.PantoneKodu,
           PartiRpt.Degerlendirme, PartiRpt.BaslamaTarihi, PartiRpt.BitisTarihi, Parti.DegisimLog from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
    left join PartiGecikme on PartiGecikme.PartiKayitId = Parti.PartiKayitId and PartiGecikme.Bitti = 0
    left join GecikmeTipi on GecikmeTipi.GecikmeTipiId = PartiGecikme.GecikmeTipiId
    left join PartiTamirSinif on PartiTamirSinif.PartiSinifId = PartiStok.PartiSinifId and PartiTamirSinif.Bitti = 0
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisBaskiId = PartiStok.BoyaSiparisBaskiId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    Left join PartiRpt on PartiRpt.PartiKayitId = Parti.PartiKayitId and PartiRpt.SonKayit = 1
    where PartiStok.Durumu <= 6 and
           (:AFirmaId is null or Firma.AnaHesapFirmaId = :AFirmaId) and
           (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
           (PartiStok.PartiStokIptalKayitId is null) and
           (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and
           (:AIslemGrubuId is null or Parti.IslemGrubuId = :AIslemGrubuId) and
           (:AIslemTipiId is null or (Parti.IslemTipiId = :AIslemTipiId) or (:AIslemTipiId = -1 and Parti.IslemTipiId <> :itTamir)) and
           (:AAciliyetDerecesiId is null or Parti.AciliyetDerecesiId >= :AAciliyetDerecesiId) and
           (:AReceteId is null or Parti.ReceteId = :AReceteId) and
           (:AModelNo is null or Parti.ModelNo = :AModelNo) and
           (:AIsletmeIciTamir is null or (:AIsletmeIciTamir = 1 and PartiTamirSinif.PartiTamirSinifId is not null and Parti.IslemTipiId <> :itTamir) or (:AIsletmeIciTamir = 0 and PartiTamirSinif.PartiTamirSinifId is null)) and
           (:AReceteYazildi is null or (Parti.ReceteId is not null and Parti.ReceteYazildi = :AReceteYazildi))
      into :PartiKayitId, :PartiYil, :PartiId, :PartiTipiId, :PartigrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi, :IslemTipiId, :IslemTipiAdi, :IslemGrubuId, :IslemGrubuAdi, :AciliyetDerecesiId,:AciliyetDerecesiAdi,
           :ReceteId, :ReceteAdi, :ReceteYazildi, :ReceteDetayVar, :Modelno, :Acilistarihi, :Termintarihi, :GecikmeNotu, :GecikmeTipiAdi, :RafId,
           :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :Fason, :DesenId, :DesenAdi, :VaryantAdi, :PantoneKodu,
           :PartiRptDegerlendirme, :PartiRptBaslamaTarihi, :PartiRptBitisTarihi, :DegisimLog do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      else BoyaSiparisIdStr = '';

      if (Fason = 0) then FasonSatisStr = 'Satış';
      else FasonSatisStr = 'Fason';

      if (ReceteYazildi = 1) then ReceteYazildiStr = 'Evet';
      else ReceteYazildiStr = '';

      if (ReceteDetayVar = 0) then ReceteDetayVarStr = 'Boş';
      else ReceteDetayVarStr = '';

      if (PartiTipiId = 0) then PartiTipiIdStr = 'İplik';
      else PartiTipiIdStr = 'Kumaş';

      if (OtomatikPosta = 1) then OtomatikPostaStr = 'Evet';
      else OtomatikPostaStr = '';

      KayitSayisi = 1;
      if (AAsamaId is not null) then /* icinde xx asamasi olanları dok demiş ise */
      begin
        Select count(*) from PartiAsamalari where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.AsamaId = :AAsamaId into :KayitSayisi;
      end

      if (KayitSayisi > 0) then
      begin
        KayitSayisi = 0;
        Select count(*) from PartiAsamalari where PartiAsamalari.PartiKayitId = :PartiKayitId into :KayitSayisi;

        if (KayitSayisi = 1) then AsamaVar = 'Var';
        else AsamaVar = '';

        select sum(kilo), sum(Metre), Sum(KapAdedi) from Partistok where Partistok.PartiKayitId = :PartiKayitId into :PartiKilo, :PartiMetre, :PartiKapAdedi;

        select Min(PartiAsamaStok.PartiKayitID), Min(GirisTarihi) from PartiAsamaStok
          where PartiAsamaStok.PartiKayitId = :PartiKayitId and PartiAsamaStok.GirisTarihi is not null
          into :Bos, :T1;

        T2 = Current_TimeStamp(0);
        if (T1 is not null) then GecenSure = T2 - T1;
        else GecenSure = null;

        if ((ABaslangicGecenSure = 0 and ABitisGecenSure = 0) or
            ((GecenSure is not null) and (GecenSure >= ABaslangicGecenSure and GecenSure <= ABitisGecenSure ))) then
        begin
          if (GecenSure = 0 or GecenSure is null) then GecenSureStr = '';
          else GecenSureStr = TarihToStr(GecenSure);

          Select PartiRptDegerlendirmeStr from PartiRptDegerlendirmeStrGetir(:PartiRptDegerlendirme) into :PartiRptDegerlendirmeStr;

          if (PartiRptBitisTarihi is not null) then
          if (PartiRptDegerlendirme = 2 or PartiRptDegerlendirme = 3) then PartiRptDegerlendirmeStr = 'Çalışıldı';

          suspend;
        end
      end
    end
  end
  else
  begin
    for select distinct Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi,Parti.IslemTipiId, IslemTipi.Adi, Parti.IslemGrubuId, IslemGrubu.Adi, Parti.AciliyetDerecesiId,
          AciliyetDerecesi.Adi,  Parti.ReceteId, Recete.Adi, Parti.ReceteYazildi, Recete.ReceteDetayVar, Parti.Modelno, Parti.Acilistarihi, Parti.Termintarihi, PartiGecikme.Notu, GecikmeTipi.Adi, Parti.RafId,
           BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.Fason, LabRecete.PantoneKodu,
           PartiRpt.Degerlendirme, PartiRpt.BaslamaTarihi, PartiRpt.BitisTarihi, Parti.DegisimLog from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
    left join PartiGecikme on PartiGecikme.PartiKayitId = Parti.PartiKayitId and PartiGecikme.Bitti = 0
    left join GecikmeTipi on GecikmeTipi.GecikmeTipiId = PartiGecikme.GecikmeTipiId
    left join PartiTamirSinif on PartiTamirSinif.PartiSinifId = PartiStok.PartiSinifId and PartiTamirSinif.Bitti = 0
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    Left join PartiRpt on PartiRpt.PartiKayitId = Parti.PartiKayitId and PartiRpt.SonKayit = 1
    where PartiStok.Durumu = :ADurumu and
           (:AFirmaId is null or Firma.AnaHesapFirmaId = :AFirmaId) and
           (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
           (PartiStok.PartiStokIptalKayitId is null) and
           (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and
           (:AIslemGrubuId is null or Parti.IslemGrubuId = :AIslemGrubuId) and
           (:AIslemTipiId is null or (Parti.IslemTipiId = :AIslemTipiId) or (:AIslemTipiId = -1 and Parti.IslemTipiId <> :itTamir)) and
           (:AAciliyetDerecesiId is null or Parti.AciliyetDerecesiId >= :AAciliyetDerecesiId) and
           (:AReceteId is null or Parti.ReceteId = :AReceteId) and
           (:AModelNo is null or Parti.ModelNo = :AModelNo) and
           (:AIsletmeIciTamir is null or (:AIsletmeIciTamir = 1 and PartiTamirSinif.PartiTamirSinifId is not null and Parti.IslemTipiId <> :itTamir) or (:AIsletmeIciTamir = 0 and PartiTamirSinif.PartiTamirSinifId is null)) and
           (:AReceteYazildi is null or (Parti.ReceteId is not null and Parti.ReceteYazildi = :AReceteYazildi))
      into :PartiKayitId, :PartiYil, :PartiId, :PartiTipiId, :PartigrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi, :IslemTipiId, :IslemTipiAdi, :IslemGrubuId, :IslemGrubuAdi, :AciliyetDerecesiId,:AciliyetDerecesiAdi,
           :ReceteId, :ReceteAdi, :ReceteYazildi, :ReceteDetayVar, :Modelno, :Acilistarihi, :Termintarihi, :GecikmeNotu, :GecikmeTipiAdi, :RafId,
           :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :Fason, :PantoneKodu,
           :PartiRptDegerlendirme, :PartiRptBaslamaTarihi, :PartiRptBitisTarihi, :DegisimLog do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      else BoyaSiparisIdStr = '';

      if (Fason = 0) then FasonSatisStr = 'Satış';
      else FasonSatisStr = 'Fason';

      if (ReceteYazildi = 1) then ReceteYazildiStr = 'Evet';
      else ReceteYazildiStr = '';

      if (ReceteDetayVar = 0) then ReceteDetayVarStr = 'Boş';
      else ReceteDetayVarStr = '';

      if (PartiTipiId = 0) then PartiTipiIdStr = 'İplik';
      else PartiTipiIdStr = 'Kumaş';

      if (OtomatikPosta = 1) then OtomatikPostaStr = 'Evet';
      else OtomatikPostaStr = '';

      KayitSayisi = 1;
      if (AAsamaId is not null) then /* icinde xx asamasi olanları dok demiş ise */
      begin
        Select count(*) from PartiAsamalari where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.AsamaId = :AAsamaId into :KayitSayisi;
      end

      if (KayitSayisi > 0) then
      begin
        KayitSayisi = 0;
        Select count(*) from PartiAsamalari where PartiAsamalari.PartiKayitId = :PartiKayitId into :KayitSayisi;

        if (KayitSayisi = 1) then AsamaVar = 'Var';
        else AsamaVar = '';

        select sum(kilo), sum(Metre), Sum(KapAdedi) from Partistok where Partistok.PartiKayitId = :PartiKayitId into :PartiKilo, :PartiMetre, :PartiKapAdedi;

        select Min(PartiAsamaStok.PartiKayitID), Min(GirisTarihi) from PartiAsamaStok
          where PartiAsamaStok.PartiKayitId = :PartiKayitId and PartiAsamaStok.GirisTarihi is not null
          into :Bos, :T1;

        T2 = Current_TimeStamp(0);
        if (T1 is not null) then GecenSure = T2 - T1;
        else GecenSure = null;

        if ((ABaslangicGecenSure = 0 and ABitisGecenSure = 0) or
            ((GecenSure is not null) and (GecenSure >= ABaslangicGecenSure and GecenSure <= ABitisGecenSure ))) then
        begin
          if (GecenSure = 0 or GecenSure is null) then GecenSureStr = '';
          else GecenSureStr = TarihToStr(GecenSure);

          Select PartiRptDegerlendirmeStr from PartiRptDegerlendirmeStrGetir(:PartiRptDegerlendirme) into :PartiRptDegerlendirmeStr;

          if (PartiRptBitisTarihi is not null) then
          if (PartiRptDegerlendirme = 2 or PartiRptDegerlendirme = 3) then PartiRptDegerlendirmeStr = 'Çalışıldı';

          suspend;
        end
      end
    end
  end
end
