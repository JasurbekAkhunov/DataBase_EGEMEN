CREATE PROCEDURE HSYUZEYKONTROLTOPHATALARIGETIR(
    AYKONTROLISEMRIKAYITID INTEGER)
RETURNS (
    HSYUZEYKONTROLHATAMETREID INTEGER,
    HSYUZEYKONTROLID INTEGER,
    HAMHATATIPIID INTEGER,
    BASLANGICMETRE DOUBLE PRECISION,
    BITISMETRE DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for Select HSYuzeyKontrol.HSYuzeyKontrolId, HSYuzeyKontrolHataMetre.HSYuzeyKontrolHataMetreId, HSYuzeyKontrolHataMetre.HamHataTipiId,
         HSYuzeyKontrolHataMetre.BaslangicMetre, HSYuzeyKontrolHataMetre.BitisMetre, IK.Adi, HSYuzeyKontrolHataMetre.InsertTarihi, SK.Adi, HSYuzeyKontrolHataMetre.DegisimTarihi from HSYuzeyKontrol
  left join HSYuzeyKontrolHataMetre on HSYuzeyKontrolHataMetre.HSYuzeyKontrolId = HSYuzeyKontrol.HSYuzeyKontrolId
  left join Kullanici IK on IK.KullaniciId  = HSYuzeyKontrolHataMetre.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = HSYuzeyKontrolHataMetre.KullaniciId
  where HSYuzeyKontrol.YKontrolIsEmriKayitId = :AYKontrolIsEmriKayitId and HSYuzeyKontrolHataMetre.HSYuzeyKontrolId is not null
  into :HSYuzeyKontrolId, :HSYuzeyKontrolHataMetreId, :HamHataTipiId, :BaslangicMetre, :BitisMetre, :InsertKullaniciAdi, :InsertTarihi,
       :KullaniciAdi, :DegisimTarihi do suspend;
end
