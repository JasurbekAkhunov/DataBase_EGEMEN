CREATE PROCEDURE TERMINGECIKMEANALIZI(
    ABASLANGICSEVKTARIHI TIMESTAMP,
    ABITISSEVKTARIHI TIMESTAMP,
    APARTITIPI INTEGER,
    AFASON INTEGER,
    AISLEMTIPIID INTEGER,
    AFIRMAID VARCHAR(20))
RETURNS (
    GECIKMETIPIID INTEGER,
    GECIKMETIPIADI VARCHAR(30),
    KILOORANI DOUBLE PRECISION,
    ADETORANI DOUBLE PRECISION)
AS
declare variable Kilo               Numeric(12,3);
declare variable KayitSayisi        Double Precision;
declare variable ToplamKilo         Numeric(12,3);
declare variable ToplamKayitSayisi  Double Precision;
begin
  Select Sum(BrutKilo), Count(*) from TerminPerformansRaporuDetayi(:ABaslangicSevkTarihi, :ABitisSevkTarihi, :APartiTipi, :AFason, :AIslemTipiId, :AFirmaId, Null) where Sonuc = -1
   into :ToplamKilo, :ToplamKayitSayisi;

  For Select GecikmeTipiId, GecikmeTipiAdi, Sum(BrutKilo), Count(*) from TerminPerformansRaporuDetayi(:ABaslangicSevkTarihi, :ABitisSevkTarihi, :APartiTipi, :AFason, :AIslemTipiId, :AFirmaId, Null) where Sonuc = -1
    Group By GecikmeTipiId, GecikmeTipiAdi
    Order by GecikmeTipiAdi
    into :GecikmeTipiId, :GecikmeTipiAdi, :Kilo, :KayitSayisi do
  begin
    AdetOrani = 0;
    KiloOrani = 0;

    if (ToplamKayitSayisi > 0) then AdetOrani = KayitSayisi * 100 / ToplamKayitSayisi;

    if (ToplamKilo > 0) then KiloOrani = Kilo * 100 / ToplamKilo;
    suspend;
  end
end
