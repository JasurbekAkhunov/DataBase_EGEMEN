CREATE PROCEDURE ORMERAFLANMAMISHAMSTOKRAPORU
RETURNS (
    ORMEFISID INTEGER,
    ORMEIRSALIYEID INTEGER,
    ORMEFATURAID INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    STOKHAREKETID INTEGER,
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    KAPADEDI INTEGER,
    MIKTAR NUMERIC(12,3),
    RAFLANANMIKTAR NUMERIC(12,3),
    FASONSTOK INTEGER,
    FASONSTOKSTR VARCHAR(5),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP)
AS
begin
  for select StokHareket.OrmeFisId, StokHareket.OrmeIrsaliyeId, StokHareket.OrmeFaturaId, StokHareket.HareketTipi, OrmeStokHareketTipi.Adi, StokHareket.FirmaId, Firma.Adi,
             OrmeFis.FisNo, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, StokHareket.StokHareketId, Ham.HamTipiId, Ham.Adi,
             StokHareket.KapAdedi, StokHareket.Miktar, StokHareket.RaflananMiktar, StokHareket.FasonStok, Renk.Adi from StokHareket
  left join OrmeStokHareketTipi on OrmeStokHareketTipi.OrmeStokHareketTipiId = StokHareket.HareketTipi
  left join Ham on Ham.HamId = StokHareket.HamId
  left join Renk on Renk.RenkId = StokHareket.RenkId
  left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeid = StokHareket.OrmeIrsaliyeId
  left join OrmeFis on OrmeFis.OrmeFisid = StokHareket.OrmeFisId
  left join OrmeFatura on OrmeFatura.OrmeFaturaid = StokHareket.OrmeFaturaId
  left join Firma on Firma.Firmaid = StokHareket.Firmaid
  where StokHareket.GirisCikis = 1 and StokHareket.Miktar > StokHareket.RaflananMiktar
  into :OrmeFisId, :OrmeIrsaliyeId, :OrmeFaturaId, :HareketTipi, :HareketTipiStr, :FirmaId, :FirmaAdi, :FisNo, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi, :StokHareketId, :HamTipiId, :HamAdi,
       :KapAdedi, :Miktar, :RaflananMiktar, :FasonStok, :RenkAdi do
  begin
    if (FasonStok= 1) then FasonStokStr = 'Fason';
    else FasonStokStr = '';

    if (OrmeFisId is not null) then EvrakTipiStr = 'Fiş';
    else if (OrmeIrsaliyeId is not null) then EvrakTipiStr = 'İrsaliye';
    else if (OrmeFaturaId is not null) then EvrakTipiStr = 'Fatura';
    else EvrakTipiStr = 'Hatalı';
    suspend;
  end
end
