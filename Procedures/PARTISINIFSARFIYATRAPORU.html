CREATE PROCEDURE PARTISINIFSARFIYATRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI VARCHAR(20),
    AMASRAFMERKEZIID INTEGER,
    ASINIF VARCHAR(1))
RETURNS (
    PARTIASAMALARIID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    SINIFISLEMKILO NUMERIC(12,3),
    ASAMAISLEMKILO NUMERIC(12,3),
    URUNTIPIID INTEGER,
    URUNTIPIADI VARCHAR(15),
    SINIFMIKTAR NUMERIC(12,3),
    SINIFMALIYET NUMERIC(12,3),
    SINIFBOYAMIKTAR NUMERIC(12,3),
    SINIFBOYAMALIYET NUMERIC(12,3),
    SINIFKIMYASALMIKTAR NUMERIC(12,3),
    SINIFKIMYASALMALIYET NUMERIC(12,3),
    SINIFDIGERMIKTAR NUMERIC(12,3),
    SINIFDIGERMALIYET NUMERIC(12,3),
    TOPLAMMIKTAR NUMERIC(12,3),
    TOPLAMMALIYET NUMERIC(12,3),
    BOYAMIKTAR NUMERIC(12,3),
    BOYAMALIYET NUMERIC(12,3),
    KIMYASALMIKTAR NUMERIC(12,3),
    KIMYASALMALIYET NUMERIC(12,3),
    DIGERMIKTAR NUMERIC(12,3),
    DIGERMALIYET NUMERIC(12,3),
    MAKINAID VARCHAR(10),
    MAKINAGRUPID VARCHAR(10),
    MAKINAGRUPADI VARCHAR(50),
    FASON INTEGER,
    FASONSTR VARCHAR(10),
    ISLEMTIPIID INTEGER,
    SIPARISTIPIID INTEGER,
    SIPARISGRUPID VARCHAR(10),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    YURTDISI INTEGER,
    ICTAMIR INTEGER,
    ICTAMIRSTR VARCHAR(10),
    HAMADI VARCHAR(100),
    HAMGRUP1 VARCHAR(4),
    HAMGRUP2 VARCHAR(4),
    HAMGRUP3 VARCHAR(4),
    HAMGRUP4 VARCHAR(4),
    HAMGRUPADI1 VARCHAR(30),
    HAMGRUPADI2 VARCHAR(30),
    HAMGRUPADI3 VARCHAR(30),
    HAMGRUPADI4 VARCHAR(30),
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIADI VARCHAR(80),
    HAMRENKADI VARCHAR(30),
    OZELLIKADI VARCHAR(40),
    LYCRATIPIADI VARCHAR(40),
    KUMASCINSIADI VARCHAR(40),
    ISLEMTIPIADI VARCHAR(20),
    ASAMAADI VARCHAR(40),
    TEORIKBOYAKIMYASALMALIYETI DOUBLE PRECISION,
    BOYAKIMYASALMALIYETI DOUBLE PRECISION,
    HAMSTOKMALIYETI DOUBLE PRECISION,
    ASAMAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    MAKINAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    UNITELERASEVKMALIYETI DOUBLE PRECISION)
AS
declare variable PartiTamirId Double Precision;
declare variable ReceteAsamaAdi varchar(40);
declare variable FMiktar Numeric(12,3);
declare variable FMaliyet Numeric(12,3);
begin
  ABitisTarihi = f_stripTime(ABitisTarihi) || ' 23:59:59';

  for select StokDetay.PartiAsamalariId, PartiAsamalari.PartiKayitId, Parti.PartiYil, Parti.PartiID, sum(StokDetay.Miktar / UrunBirim.Carpan),
    sum(StokDetay.Miktar  * StokDetay.NetBirimFiyat) from stokdetay
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = StokDetay.PartiAsamalariId
    left join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
    left join Stok on Stok.StokId = StokDetay.StokId
    left join Urun on Urun.UrunId = StokDetay.UrunId
    where StokDetay.UretimeCikisTarihi >= :ABaslangicTarihi and StokDetay.UretimeCikisTarihi <= :ABitisTarihi and
          (StokDetay.Sinif = :ASinif or :ASinif is null) and
          StokDetay.HareketTipi = 6 and StokDetay.PartiKayitId is not null and StokDetay.EasPartiNo is null and
         (Stok.MasrafMerkeziId = :AMasrafMerkeziId or :AMasrafMerkeziId is null)
    group by StokDetay.PartiAsamalariId, PartiAsamalari.PartiKayitId, Parti.PartiYil, Parti.PartiID
    into :PartiAsamalariId, :PartiKayitId, :PartiYil, :PartiID, :ToplamMiktar, :ToplamMaliyet do
  begin
    BoyaMiktar = 0;
    BoyaMaliyet = 0;
    KimyasalMiktar = 0;
    KimyasalMaliyet = 0;
    DigerMiktar = 0;
    DigerMaliyet = 0;
    for select Urun.UrunTipiId, sum(StokDetay.Miktar / UrunBirim.Carpan), sum(StokDetay.Miktar  * StokDetay.NetBirimFiyat) from stokdetay
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      left join Urun on Urun.UrunId = StokDetay.UrunId
      left join Stok on Stok.StokId = StokDetay.StokId
      where StokDetay.PartiAsamalariId = :PartiAsamalariId and StokDetay.UretimeCikisTarihi >= :ABaslangicTarihi and
          StokDetay.UretimeCikisTarihi <= :ABitisTarihi and
          (StokDetay.Sinif = :ASinif or :ASinif is null) and
          StokDetay.HareketTipi = 6 and StokDetay.PartiKayitId is not null and StokDetay.EasPartiNo is null and
          (Stok.MasrafMerkeziId = :AMasrafMerkeziId or :AMasrafMerkeziId is null)
      group by Urun.UrunTipiId
      into :UrunTipiId, :FMiktar, :FMaliyet do
    begin
      if (UrunTipiId = 0) then
      begin
        BoyaMiktar = FMiktar;
        BoyaMaliyet = FMaliyet;
      end
      if (UrunTipiId = 1) then
      begin
        KimyasalMiktar = FMiktar;
        KimyasalMaliyet = FMaliyet;
      end
      if (UrunTipiId >= 2) then
      begin
        DigerMiktar = FMiktar;
        DigerMaliyet = FMaliyet;
      end
    end
    select sum(PartiAsamaStokDetay.Kilo) from PartiAsamaStokDetay
      left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId and PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
      where PartiAsamaStokDetay.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
      into :AsamaIslemKilo;
    SinifBoyaMiktar = 0;
    SinifBoyaMaliyet = 0;
    SinifKimyasalMiktar = 0;
    SinifKimyasalMaliyet = 0;
    SinifDigerMiktar = 0;
    SinifDigerMaliyet = 0;
    for select PartiAsamaStokDetay.PartiSinifId, PartiAsamaStokDetay.MakinaId, PartiAsamaStokDetay.Kilo,
      BoyaSiparis.Fason, BoyaSiparis.IslemTipiId, BoyaSiparis.SiparisTipiId, BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id, BoyaSiparis.YurtDisi,
      PartiAsamalari.PartiTamirId, Makina.MakinaGrubu,
      Kalinlik.Adi, IplikCinsi.Adi,
      HamRenk.Adi, Ozellik.Adi, LycraTipi.Adi, KumasCinsi.Adi, Ham.Adi, SiparisGrup.Adi, SiparisGrup2.Adi,
      MakinaGrup.Adi, IslemTipi.Adi, Asama.Adi, ReceteAsama.Adi, Ham.HamGrup1, Ham.HamGrup2, Ham.HamGrup3, Ham.HamGrup4,
      HG1.Adi, HG2.Adi, HG3.Adi, HG4.Adi from PartiAsamaStokDetay
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
      left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
      left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId and PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = PartiStok.BoyaSiparisKayitId
      left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
      left join Ham on Ham.HamId = PartiStok.HamId
      left join HamGrup HG1 on HG1.HamGrupId = Ham.HamGrup1
      left join HamGrup HG2 on HG2.HamGrupId = Ham.HamGrup2
      left join HamGrup HG3 on HG3.HamGrupId = Ham.HamGrup3
      left join HamGrup HG4 on HG4.HamGrupId = Ham.HamGrup4
      left join Kalinlik On Kalinlik.KalinlikId = Ham.KalinlikId
      left join IplikCinsi On IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
      left join HamRenk On HamRenk.HamRenkId = Ham.HamRenkId
      left join Ozellik On Ozellik.OzellikId = Ham.OzellikId
      left join LycraTipi On LycraTipi.LycraTipiId = Ham.LycraTipiId
      left join KumasCinsi On KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join MakinaGrup on MakinaGrup.MakinaGrupId = Makina.MakinaGrubu
      left join IslemTipi On IslemTipi.IslemTipiId = BoyaSiparis.IslemTipiId
      left join Asama On Asama.AsamaId = PartiAsamalari.AsamaId
      left join Recete On Recete.ReceteId = PartiAsamalari.ReceteId
      left join Asama ReceteAsama On ReceteAsama.AsamaId = Recete.UstAsamaId
      where PartiAsamaStokDetay.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
      into :PartiSinifId, :MakinaId, :SinifIslemKilo,
      :Fason, :IslemTipiId, :SiparisTipiId, :SiparisGrupId, :SiparisGrup2Id, :YurtDisi,
      :PartiTamirId, :MakinaGrupId,
      :KalinlikAdi, :IplikCinsiAdi,
      :HamRenkAdi, :OzellikAdi, :LycraTipiAdi, :KumasCinsiAdi, :HamAdi, :SiparisGrupAdi, :SiparisGrup2Adi,
      :MakinaGrupAdi, :IslemTipiAdi, :AsamaAdi, :ReceteAsamaAdi,
      :HamGrup1, :HamGrup2, :HamGrup3, :HamGrup4, :HamGrupAdi1, :HamGrupAdi2, :HamGrupAdi3, :HamGrupAdi4 do
    begin
      if (PartiTamirId is not Null) then IcTamir = 1;
      else IcTamir = 0;
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';
      if (IcTamir = 1) then IcTamirStr = 'İç Tamir';
      else IcTamirStr = '';
      if (AsamaIslemKilo > 0) then
      begin
        SinifMiktar = (ToplamMiktar * SinifIslemKilo) / AsamaIslemKilo;
        SinifMaliyet = (ToplamMaliyet * SinifIslemKilo) / AsamaIslemKilo;

        SinifBoyaMiktar = (BoyaMiktar * SinifIslemKilo) / AsamaIslemKilo;
        SinifBoyaMaliyet = (BoyaMaliyet * SinifIslemKilo) / AsamaIslemKilo;

        SinifKimyasalMiktar = (KimyasalMiktar * SinifIslemKilo) / AsamaIslemKilo;
        SinifKimyasalMaliyet = (KimyasalMaliyet * SinifIslemKilo) / AsamaIslemKilo;

        SinifDigerMiktar = (DigerMiktar * SinifIslemKilo) / AsamaIslemKilo;
        SinifDigerMaliyet = (DigerMaliyet * SinifIslemKilo) / AsamaIslemKilo;
      end

      if (AsamaAdi is Null) then AsamaAdi = ReceteAsamaAdi;

      select first 1 HamStokMaliyeti, AsamaUMUzerindenAsamaMaliyeti, MakinaUMUzerindenAsamaMaliyeti, BoyaKimyasalMaliyeti,
           TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti from PartiSinifMaliyetKumulatif
           where PartiSinifID = :PartiSinifID
           into :HamStokMaliyeti, :AsamaUMUzerindenAsamaMaliyeti, :MakinaUMUzerindenAsamaMaliyeti, :BoyaKimyasalMaliyeti,
           :TeorikBoyaKimyasalMaliyeti, :UnitelerASevkMaliyeti;
      Suspend;
    end
  end
end
