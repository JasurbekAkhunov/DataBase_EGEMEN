CREATE PROCEDURE ORMEMALIYETLERIHESAPLA(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMID INTEGER,
    AISLEMTIPI SMALLINT)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ORMEPLANLAMAID INTEGER,
    SHORMERAFID INTEGER,
    STOKHAREKETID INTEGER,
    ORMERAFTRANSFERFISIDETAYID INTEGER,
    TARIH TIMESTAMP,
    HAREKETTIPI INTEGER,
    URETIMISLEMTIPIID INTEGER,
    GIRISCIKIS INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    BIRIMFIYAT DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    NETBIRIMFIYAT DOUBLE PRECISION,
    BIRIMMALIYET DOUBLE PRECISION,
    TAHMINIFIREORANI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    HAMIADE INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    FIRMAID VARCHAR(20),
    ORMEMAKINAID VARCHAR(10),
    ORMEFATURAID INTEGER,
    ORMESTOKKAYDIRMAID INTEGER,
    EGEMENFIRMAID INTEGER,
    STOKTABANDOVIZID VARCHAR(5))
AS
declare variable SiparisMaliyetTakibiVar integer;
declare variable Sayac SmallInt;
begin
  Delete From MaliyetTemp;

  ABaslangicTarihi = :ABaslangicTarihi - 3;
  ABitisTarihi     = :ABitisTarihi + 3;
  if (AIslemTipi = 0) then Sayac = 3; /*Eğer clienttan çağırılmış ise cliet tarafı zaten 3 defa çağırıyor*/
  else Sayac = 1;
  Select SiparisMaliyetTakibiVar, EgemenFirmaId, StokTabanDovizId from Sistem where SistemID = 1 into :SiparisMaliyetTakibiVar, :EgemenFirmaId, :StokTabanDovizId;
  if (SiparisMaliyetTakibiVar = 1) then exception KayitHatasi ' Siparis maliyet takibini kapatmalisiniz...';

  While (Sayac <= 3) do
  begin
    for select StokHareket.StokHareketId, StokHareket.OrmeSiparisDetayId, StokHareket.DovizId, StokHareket.Tarih,
          StokHareket.HareketTipi, StokHareket.UretimIslemTipiId, iif(:EgemenFirmaId = 9, StokHareket.BirimFiyat, StokHareket.NetBirimFiyat), StokHareket.HamIade,
          StokHareket.BoyaSiparisKayitId, StokHareket.BoyaIrsaliyeDetayId, StokHareket.HamId, StokHareket.RenkId,
          StokHareket.OrmePlanlamaId, StokHareket.BoyaSiparisDetayID,
          StokHareket.OrmeRafId, StokHareket.GirisCikis, StokHareket.Miktar, StokHareket.BirimFiyat, StokHareket.FirmaId, OrmeMakinaId,
          Ham.Adi, Renk.Adi, StokHareket.OrmeFaturaId, Coalesce(TahminiFireOrani, 0),
          iif(BoyaSiparisID is not Null, '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisID), Null),
          OrmeFis.OrmeStokKaydirmaId from StokHareket
      left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
      left join Ham on Ham.HamId = StokHareket.HamID
      left join Renk on Renk.RenkId = StokHareket.RenkID
      Left Join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = StokHareket.OrmeSiparisDetayID
      where StokHareket.Tarih >= :ABaslangicTarihi and StokHareket.Tarih <= :ABitisTarihi and (StokHareket.HamId = :AHamId or :AHamId is null)
      order by StokHareket.Tarih, UretimIslemTipiId, Harekettipi, StokHareketId asc
      into :StokHareketId, :OrmeSiparisDetayId, :DovizID, :Tarih, :HareketTipi, :UretimIslemTipiId, :NetBirimFiyat, :HamIade,
           :BoyaSiparisKayitId, :BoyaIrsaliyeDetayId, :HamId, :RenkId, :OrmePlanlamaId, :BoyaSiparisDetayID,
           :SHOrmeRafId, :GirisCikis, :Miktar, :BirimFiyat, :FirmaId, :OrmeMakinaId, :HamAdi, :RenkAdi,
           :OrmeFaturaId, :TahminiFireOrani, :BoyaSiparisIdStr, :OrmeStokKaydirmaId do
    begin
      if (AIslemTipi = 1) then
      begin
        Select BirimMaliyet, NetBirimFiyat from OrmeMaliyetleriHesapla_Ic(:StokHareketId, Null) into :BirimMaliyet, :NetBirimFiyat;
      end

      if (Sayac = 3) then suspend;
    end

    OrmeSiparisDetayId = null;
    BoyaSiparisDetayID = null;
    BoyaSiparisIdStr   = null;
    StokHareketId = null;
    Tarih = null;
    HareketTipi = null;
    UretimIslemTipiId = null;
    HamId   = null;
    HamAdi  = null;
    RenkId  = null;
    RenkAdi = null;
    BirimMaliyet = null;
    Miktar = null;
    NetBirimFiyat = null;

    for select OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId, OrmeRafTransferFisi.FisTarihi, OrmeRafTransferFisiDetay.Kilo, OrmeRafTransferFisiDetay.OrmeRafId from OrmeRafTransferFisi
      left join OrmeRafTransferFisiDetay on OrmeRafTransferFisiDetay.OrmeRafTransferFisiId = OrmeRafTransferFisi.OrmeRafTransferFisiId
      left join OrmeRaf on OrmeRaf.OrmeRafId = OrmeRafTransferFisiDetay.OrmeRafId
      where OrmeRafTransferFisi.FisTarihi >= :ABaslangicTarihi and OrmeRafTransferFisi.FisTarihi <= :ABitisTarihi and (OrmeRaf.HamId = :AHamId or :AHamId is null)
      order by OrmeRafTransferFisi.FisTarihi, OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId asc
      into :OrmeRafTransferFisiDetayId, :Tarih, :Miktar, :SHOrmeRafId do
    begin
      if (AIslemTipi = 1) then
      begin
        Select BirimMaliyet, NetBirimFiyat from OrmeMaliyetleriHesapla_Ic(Null, :OrmeRafTransferFisiDetayId) into :BirimMaliyet, :NetBirimFiyat;
      end

      if (Sayac = 3) then suspend;
      if (Sayac = 3) then suspend;
    end
    Sayac = Sayac + 1;
  end

end
