CREATE PROCEDURE DESENTAKIMLISTESIDETAYI
RETURNS (
    DESENTAKIMID INTEGER,
    INSERTTARIHI TIMESTAMP,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(80),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DESENGRUPID VARCHAR(5),
    MUSTERIDESENNO1 VARCHAR(20),
    MUSTERIDESENADI1 VARCHAR(80),
    MUSTERIDESENNO2 VARCHAR(20),
    MUSTERIDESENADI2 VARCHAR(80))
AS
begin
  For Select Desen.DesenTakimId, Desen.InsertTarihi, Desen.DesenId, Desen.Adi,
             Desen.FirmaId, Firma.Adi, Desen.DesenGrupId, Desen.MusteriDesenNo1, Desen.MusteriDesenAdi1, Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2 from DesenTakimListesi
  left join Desen on Desen.DesenTakimId = DesenTakimListesi.DesenTakimId
  left join Firma on Firma.FirmaId = Desen.FirmaId
  into :DesenTakimId, :InsertTarihi, :DesenId, :DesenAdi, :FirmaId, :FirmaAdi, :DesenGrupId, :MusteriDesenNo1,
       :MusteriDesenAdi1, :MusteriDesenNo2, :MusteriDesenAdi2 do
  begin
    suspend;
  end
end
