CREATE PROCEDURE TALIMATICINFIRMASTOKLISTELE(
    AORMEPLANLAMAID INTEGER,
    AHAMID INTEGER,
    AFASON INTEGER,
    ARENKID INTEGER,
    AFIRMAID VARCHAR(20))
RETURNS (
    ORMERAFID INTEGER,
    RAFID VARCHAR(10),
    STOKGIRISNO INTEGER,
    LOT VARCHAR(150),
    URETICIADI VARCHAR(40),
    KALANMIKTAR NUMERIC(12,3),
    BEKLEYENMIKTAR NUMERIC(12,3),
    KULLANIMAUYGUNMIKTAR NUMERIC(12,3),
    URETIMPARTINO VARCHAR(30),
    MELANJKODU VARCHAR(20),
    KALITENO INTEGER,
    KALANKAPADEDI INTEGER,
    BOYASIPARISNOSTR VARCHAR(30),
    RESULTID VARCHAR(70))
AS
declare variable SiparisFirmaId varchar(20);
declare variable StokFirmaId varchar(20);
declare variable OrmeFirmaKodumuz varchar(20);
declare variable OrmeUretimeCikistaLotTakibiVar integer;
declare variable FUreticiId integer;
declare variable SiparisYil Integer;
declare variable SiparisId Integer;
begin
  Select BoyaSiparis.FirmaId from OrmePlanlama
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
  where OrmePlanlama.OrmePlanlamaId = :AOrmePlanlamaId
  into :SiparisFirmaId;

  Select OrmeFirmaKodumuz from sistem where sistemid = 1 into :OrmeFirmaKodumuz;

  if (AFason = 1) then
    StokFirmaID = SiparisFirmaId;
  else
    StokFirmaID = OrmeFirmaKodumuz;

  For Select OrmeRaf.OrmeRafId, OrmeRaf.RafId, OrmeRaf.StokGirisNo, OrmeRaf.Lot, Uretici.Adi, OrmeRaf.UretimPartiNo,
             OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari, OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi,
             OrmeRaf.KaliteNo, OrmeRaf.MelanjKodu, OrmeRaf.UreticiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId  from OrmeRaf
    left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
    left join Raf on Raf.RafId = OrmeRaf.RafId
    Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitID
    where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and OrmeRaf.StokFirmaId = :StokFirmaID and OrmeRaf.BoyahaneFasonIade = 0 and
          StrKarsilastir(OrmeRaf.RenkID, :ARenkId) = 1 and OrmeRaf.FasonStok = :AFason and Raf.FirmaId = : AFirmaID
    into :OrmeRafId, :RafId, :StokGirisNo, :Lot, :UreticiAdi, :UretimPartiNo, :KalanMiktar, :KalanKapAdedi, :KaliteNo, :MelanjKodu, :FUreticiId, :SiparisYil, :SiparisId do
  begin

    BoyaSiparisNoStr = '';
    if (SiparisId is not Null) then BoyaSiparisNoStr = '(' || EgemenIntegerToStr(SiparisYil) || ') ' || EgemenIntegerToStr(SiparisId) || ' ';


    if (MelanjKodu is not null) then
      ResultId = '"' || MelanjKodu || '",' ;
    else
      ResultId = '"",' ;

    if (FUreticiId is not null) then
      ResultId = ResultId || '"' || EgemenIntegerToStr(FUreticiId) || '",' ;
    else
      ResultId = ResultId || '"",' ;

    if (Lot is not null) then
      ResultId = ResultId || '"' || Lot || '"' ;
    else
      ResultId = ResultId || '""' ;


    KullanimaUygunMiktar = KalanMiktar;
    Select Sum((OrmePlanlama.PlanlananMiktar * OrmePlanlamaRecete.KullanimOrani / 100) * ((100 - OrmePlanlamaRecete.TahminiFireOrani) / 100) - (OrmePlanlamaRecete.CikisMiktari - OrmePlanlamaRecete.IadeMiktari)) BekleyenMiktar from OrmePlanlamaRecete
      Left Join OrmePlanlama on OrmePlanlama.OrmePlanlamaID = OrmePlanlamaRecete.OrmePlanlamaID
      where OrmePlanlamaRecete.HamID = :AHamId and OrmePlanlama.Bitti = 0 and
            IntKarsilastir(OrmePlanlamaRecete.RenkID, :ARenkId) = 1 and
            StrKarsilastir(OrmePlanlamaRecete.Lot, :Lot) = 1 and
            StrKarsilastir(OrmePlanlamaRecete.MelanjKodu, :MelanjKodu) = 1 and
            IntKarsilastir(OrmePlanlamaRecete.UreticiId, :FUreticiId) = 1 and
            OrmePlanlamaRecete.OrmePlanlamaID <> :AOrmePlanlamaId and
            (OrmePlanlama.FirmaID = :AFirmaID)
      into :BekleyenMiktar;

    if (BekleyenMiktar > 0 ) then KullanimaUygunMiktar = KalanMiktar - BekleyenMiktar;
    else BekleyenMiktar = 0;

    suspend;
  end
end
