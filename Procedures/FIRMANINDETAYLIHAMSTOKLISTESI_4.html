CREATE PROCEDURE FIRMANINDETAYLIHAMSTOKLISTESI_4(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AORMEGIRISIRSALIYENO IRSALIYENOSIZE,
    AEVRAKNO VARCHAR(10),
    AHAMSTOKMASTERID INTEGER,
    ATAMIR INTEGER,
    AFASON INTEGER,
    ARAPORHAMTIPI INTEGER,
    ATAMIRTIPIID INTEGER,
    ASIFIROLANLARIDOK INTEGER,
    AHAMID INTEGER,
    ALOT VARCHAR(150),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AREFERANSFIRMAID VARCHAR(20),
    AORMECIID INTEGER,
    ASINIF VARCHAR(1),
    AHAMSTOKHAREKETTIPIID INTEGER,
    AIRSALIYEGRUPID VARCHAR(4),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ASIPARISFIRMAID VARCHAR(20),
    AIPLIKTALEPYIL INTEGER,
    AIPLIKTALEPID INTEGER,
    AURETIMPARTINO VARCHAR(30),
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AMUSTERIPARTINO VARCHAR(30),
    AVERILENSIPARISKAYITID INTEGER,
    AREFERANSKODU VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ASERTIFIKAGOSTER INTEGER,
    ATARIHFILTRETIPI INTEGER,
    AKALITEID VARCHAR(20))
RETURNS (
    HAMSTOKMASTERID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    RAFDETAYID INTEGER,
    HAMADI VARCHAR(255),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    SINIF VARCHAR(1),
    GIRISIRSALIYENO IRSALIYENOSIZE,
    PARTILEMECIKISMIKTARI BOYAMIKTARSIZE,
    PARTILEMECIKISMETRE BOYAMIKTARSIZE,
    PARTILEMECIKISKILO BOYAMIKTARSIZE)
AS
declare variable Oran Double Precision;
begin
  for select distinct HamStokDetay.HamStokMasterID, PartiStokDetay.PartiKayitId, PartiStokDetay.PartiSinifId, RafDetay.RafDetayId, Ham.Adi,
             PartiStokDetay.KapAdedi, PartiStokDetay.Kilo, PartiStokDetay.Metre, PartiStokDetay.Adet, HamStokIrsaliye.Sinif, HamStokIrsaliye.IrsaliyeNo from FirmaninDetayliHamStokListesi_2(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :AIrsaliyeNo, :AOrmeGirisIrsaliyeNo, :AEvrakNo, :AHamStokMasterId, :ATamir, :AFason, :ARaporHamTipi,
                                     :ATamirTipiId, :ASifirOlanlariDok, :AHamId, :ALot, :AMusteriSiparisNo, :AReferansSiparisNo, :AReferansFirmaId, :AOrmeciId, :ASinif, :AHamStokHareketTipiId, :AIrsaliyeGrupId,
                                     :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id, :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id,
                                     :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ABoyaSiparisYil, :ABoyaSiparisId, :ASiparisFirmaId, :AIplikTalepYil, :AIplikTalepId, :AUretimPartiNo, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AMusteriPartiNo, :AVerilenSiparisKayitId, :AReferansKodu, :ASertifikaNo, :ASertifikaGoster, :ATarihFiltreTipi, :AKaliteId)
    left join PartiStokDetay on PartiStokDetay.PartiKayitId = FirmaninDetayliHamStokListesi_2.PartiKayitId
    left join RafDetay on RafDetay.RafDetayid = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayid = RafDetay.HamStokDetayId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
    left join Ham on Ham.HamId = PartiStokDetay.HamId
    where PartiStokDetay.PartiSinifId is not null
    into :HamStokMasterID, :PartiKayitId, :PartiSinifId, :RafDetayId, :HamAdi, :KapAdedi, :Kilo, :Metre, :Adet, :Sinif, :GirisIrsaliyeNo do
    begin
      select First 1 PartiAsamaStokDetay.CikisMiktar, PartiAsamaStokDetay.CikisMetre, PartiAsamaStokDetay.CikisKilo from PartiAsamalari
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join Asama on Asama.Asamaid = PartiAsamalari.AsamaId
      where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId and Asama.PartilemeMi = 1
      order by PartiAsamalari.Sira
      into :PartilemeCikisMiktari, :PartilemeCikisMetre, :PartilemeCikisKilo;

      select iif(Sum(PartiStokDetay.Miktar) <= 0, 0, Sum(iif(RafDetay.RafDetayid = :RafDetayId, PartiStokDetay.Miktar, 0)) / Sum(PartiStokDetay.Miktar))
      from PartiStokDetay
      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      where PartiStokDetay.PartiSinifId = :PartiSinifId into :Oran;

      PartilemeCikisMiktari = PartilemeCikisMiktari * Oran;
      PartilemeCikisMetre = PartilemeCikisMetre * Oran;
      PartilemeCikisKilo = PartilemeCikisKilo * Oran;

      suspend;
    end
end
