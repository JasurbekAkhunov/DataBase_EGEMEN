CREATE PROCEDURE TEKLIFICINRENKDERINLIKMALIYETI(
    ARENKDERINLIGIID INTEGER,
    AFIRMAID VARCHAR(20),
    ADOVIZID VARCHAR(5),
    ATEKLIFTARIHI TIMESTAMP)
RETURNS (
    RECETEMALIYETI DOUBLE PRECISION)
AS
declare variable RDovizId         varchar(5);
declare variable FDovizId         varchar(5);
declare variable FReceteMaliyeti  Double Precision;
declare variable FDovizKuru                  Double Precision;
begin
  Select RenkDerinligi.BirimFiyat, RenkDerinligi.DovizId, FirmaRenkDerinligi.BirimFiyat, FirmaRenkDerinligi.DovizId from RenkDerinligi
    left join FirmaRenkDerinligi On FirmaRenkDerinligi.FirmaId = :AFirmaId and FirmaRenkDerinligi.RenkDerinligiId = RenkDerinligi.RenkDerinligiId
    where RenkDerinligi.RenkDerinligiID = :ARenkDerinligiId
    into :ReceteMaliyeti, :RDovizId, :FReceteMaliyeti, :FDovizId;

  if (FDovizId is not null) then
  begin
    RDovizID = FDovizId;
    ReceteMaliyeti = FReceteMaliyeti;
  end

  if (RDovizId <> ADovizId) then
  begin
    select DovizKuru from DovizKuruGetir(:RDovizId, :ADovizId, :ATeklifTarihi) into :FDovizKuru;

    if (FDovizKuru = 0) then exception DovizKuruGirisiYapilmamis;
    ReceteMaliyeti = ReceteMaliyeti * FDovizKuru;
  end

  suspend;
end
