CREATE PROCEDURE PSKALITEKONTROLHATAGIRISI_2(
    APSKALITEKONTROLHATAMETREID INTEGER,
    APSKALITEKONTROLID INTEGER,
    AHATATIPIID INTEGER,
    AMETRE DOUBLE PRECISION,
    AHATAYERI INTEGER,
    AHATAYERICM INTEGER,
    ABITIS INTEGER,
    AHATAPUANI DOUBLE PRECISION,
    AKULLANICIID INTEGER)
RETURNS (
    PSKALITEKONTROLHATAMETREID INTEGER)
AS
begin
  if (AHataYeriCm = 0) then AHataYeriCm = Null;
  if (ABitis = 0) then
  begin
    if (APSKaliteKontrolHataMetreId is not null) then Update PSKaliteKontrolHataMetre Set BaslangicMetre = :AMetre, KullaniciId = :AKullaniciId  where PSKaliteKontrolHataMetre.PSKaliteKontrolHataMetreID = :APSKaliteKontrolHataMetreID;
    else
    begin
      PSKalitekontrolHataMetreID = Gen_Id(PSKaliteKontrolHataMetre_Id_Gen,1);
      insert into PSKalitekontrolHataMetre(PSKalitekontrolHataMetreID, PSKalitekontrolid, HataTipiId, BaslangicMetre, HataYeri, HataYeriCm, HataPuani, InsertKullaniciId, KullaniciId) values(:PSKalitekontrolHataMetreID, :APSKaliteKontrolId, :AHataTipiId, :AMetre, :AHataYeri, :AHataYeriCm, :AHataPuani, :AKullaniciId, :AKullaniciId);
    end
  end
  else Update PSKaliteKontrolHataMetre Set BitisMetre = :AMetre where PSKaliteKontrolHataMetre.PSKaliteKontrolHataMetreID = :APSKaliteKontrolHataMetreID;
  suspend;
end
