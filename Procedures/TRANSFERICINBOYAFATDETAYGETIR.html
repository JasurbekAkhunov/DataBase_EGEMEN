             Ham.AlisSekerKasarTransferKodu, Ham.AlisOptikBeyazTransferKodu, Ham.AlisYikamaTransferKodu, Parti.PartiKayitID,
             iif(BoyaFatura.IslemDovizKuru1 > BoyaFatura.IslemDovizKuru2, BoyaFatura.IslemDovizKuru1, BoyaFatura.IslemDovizKuru2), BoyaFaturaDetay.SiparisBirimFiyat, BoyaIrsaliyeDetay.Kilo,

             BoyaSiparisDetay.BelgeHamAdi, BoyaSiparis.SiparisiAlanKullaniciId, Ham.Adi2, StokHareketGrup.TransferKodu, BaskiTipi.DijitalBaski,
             iif(Parti.Fason = 1, BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.Kilo), BoyaFaturaDetay.FaturaMiktarTipi from BoyaFaturaDetay
    left join BoyaFatura on BoyaFatura.BoyaFaturaId = BoyaFaturaDetay.BoyaFaturaId
    Left Join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
    Left Join PartiPaket on PartiPaket.PartiPaketId = BoyaFaturaDetay.PartiPaketId

    Left Join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId

    Left Join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId

  begin
    if (FBoyaSiparisYil is not null) then BoyaSiparisNoStr = EgemenCopyStr(EgemenIntegerToStr(FBoyaSiparisYil),3,2) || '-' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
    else BoyaSiparisNoStr = '';

    if (FPartiID is not null) then PartiNoStr = EgemenCopyStr(EgemenIntegerToStr(FPartiYil),3,2) || '-' || EgemenIntegerToStr(FPartiID) || ' ';
    else PartiNoStr = '';


    BirimTransferKodu = '';
    if (FaturaBirim = 0) then Select TransferKodu from Birim where Birim.BirimID = 'Kg' into :BirimTransferKodu;
    else Select TransferKodu from Birim where Birim.BirimID = 'Mt' into :BirimTransferKodu;

    if (EgemenFirmaId = 34 and FaturaBirim = 1) then /*Orkum ve Metre ise*/
    begin
      if (Fason = 0) then Aciklama = Format_Float('', FKilo) || 'Kg';
      else
      begin
        if (FaturaMiktarTipi = 0) then
          Aciklama = Format_Float('', FKilo) || 'Kg';  /* ali değiştirdi*/
        else
          Aciklama = Format_Float('', FBrutKilo) || 'Kg';
      end
    end


    if (DovizID <> SistemDovizId) then
    begin
      select DovizKuru from FirmaDovizKurlariniGetir(:FirmaId, :DovizId, :SistemDovizId, :FaturaTarihi) into :DovizKuru;
    end
    else DovizKuru = 1;

    Tutar = BirimFiyat * Miktar;

    BirimFiyatTL = BirimFiyat * DovizKuru;
    BirimFiyatTL = Yuvarla(BirimFiyatTL, FNetFiyatYuvarlamaCarpani);

    TutarTL = BirimFiyatTL * Miktar;
    TutarTL = Yuvarla(TutarTL, FNetFiyatYuvarlamaCarpani);
    IslemDovizToplam =  (FSiparisBirimFiyat  * Miktar);

    FireMiktari = 0;
    TransferKodu = Null;

    IlaveIslemler = '';
    if (PartiSinifID is not null) then
    begin
      if (TamirBoyaSiparisDetayId is not null) then
        Select IslemGrubuId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = :TamirBoyaSiparisDetayId into :SiparisIslemGrubuId;

      if (EgemenFirmaID = 14 or EgemenFirmaID = 22 or EgemenFirmaID = 48) then /* has örme, fon boya, Karem */
      begin
        if (Fason = 0) then
        begin
          Select SatisKumasTransferKodu, SatisKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
        end
        else
        begin
          TransferKodu = SiparisProsesTransferKodu;
          MuhGelirHesapKodu = FSipFasonMuhGelirHesapKodu;
        end
      end
      else if (EgemenFirmaID = 4 or EgemenFirmaID = 15 or EgemenFirmaID = 16 or EgemenFirmaId = 6 or EgemenFirmaId = 26 or EgemenFirmaId = 50 or EgemenFirmaId = 32) then /* gülle, ozak, tekboy, MArsan(Megasan), Ottoman(Biessici) */
      begin
        if (Fason = 0) then
        begin
          TransferKodu = Null;
          MuhGelirHesapKodu = Null;
          if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
            Select SatisIplikTransferKodu, SatisIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
          else
          begin
            if (FaturaBirim = 0) then /*kg*/
            begin
              if (ImalatTipiId = 1) then /* baski ise */
                Select BaskiSatisKumasTransferKodu, BaskiSKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
              else /* boya ise */
                Select SatisKumasTransferKodu, SatisKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
            end
            else if (FaturaBirim = 1) then /*mt*/
            begin
              if (ImalatTipiId = 1) then /* baski ise */
                Select BaskiSatisKumasMtTransferKodu, BaskiSKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
              else /* boya ise */
                Select SatisKumasMtTransferKodu, SatisKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
            end
            else if (FaturaBirim = 2) then /*ad*/
            begin
              if (ImalatTipiId = 1) then /* baski ise */
                Select BaskiSatisKumasAdTransferKodu, BaskiSKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
              else /* boya ise */
                Select SatisKumasAdTransferKodu, SatisKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
            end
          end
        end
        else
        begin
          TransferKodu = Null;
          MuhGelirHesapKodu = Null;
          if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
            Select FasonIplikTransferKodu, FasonIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
          else
          begin
            if (FaturaBirim = 0) then /*kg*/
            begin
              if (ImalatTipiId = 1 or IslemGrubuId = 7) then /* baski ise */
                Select BaskiFasonKumasTransferKodu, BaskiFKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
              else
                Select FasonKumasTransferKodu, FasonKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
            end
            else if (FaturaBirim = 1) then /*mt*/
            begin
              if (ImalatTipiId = 1 or IslemGrubuId = 7) then /* baski ise */
                Select BaskiFasonKumasMtTransferKodu, BaskiFKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
              else
                Select FasonKumasMtTransferKodu, FasonKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
            end
            else if (FaturaBirim = 2) then /*ad*/
            begin
              if (ImalatTipiId = 1 or IslemGrubuId = 7) then /* baski ise */
                Select BaskiFasonKumasAdTransferKodu, BaskiFKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
              else
                Select FasonKumasAdTransferKodu, FasonKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
            end
          end
        end
      end
      else if (EgemenFirmaID = 46) then /* Başyazıcı*/
      begin
        if (Fason = 0) then
        begin
          TransferKodu = FSatisTransferKodu;
          MuhGelirHesapKodu = HamStokMuhSatisHesapKodu;
        end
        else
        begin
          TransferKodu = Null;
          MuhGelirHesapKodu = Null;
          if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
            Select FasonIplikTransferKodu, FasonIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
          else
          begin
            if (ImalatTipiId = 1 or IslemGrubuId = 7) then /* baski ise */
              Select BaskiFasonKumasTransferKodu, BaskiFKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
            else
              Select FasonKumasTransferKodu, FasonKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuID into :TransferKodu, :MuhGelirHesapKodu;
          end
        end
      end
      else if (EgemenFirmaId = 7) then /* imaj(Ottoman) */
      begin
        if (Fason = 0) then
        begin
          TransferKodu      = FSatisTransferKodu;
          MuhGelirHesapKodu = HamStokMuhSatisHesapKodu;
        end
        else
        begin
          TransferKodu      = SiparisProsesTransferKodu;
          MuhGelirHesapKodu = FSipFasonMuhGelirHesapKodu;
        end
      end
      else if (EgemenFirmaID = 34) then /* Örkum*/
      begin
        select TransferKodu, IslemGrubuId, HareketSekli, Null from BoyaIrsaliyeFaturaTransferKodu(:BoyaIrsaliyeDetayId, Null, :EgemenFirmaId) into :TransferKodu, :IslemGrubuId, :HareketSekli, :MuhGelirHesapKodu;
      end

      if (PartiPaketId is null and Fason = 0) then FireMiktari = BrutMiktar - NetMiktar;
      if (FireMiktari < 0) then FireMiktari = 0;

      for Select IlaveIslemler.Adi from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1 and PartiAsamalari.SerbestAsama = 0 and IlaveIslemler.Adi is not null
      into :FIlaveIslemlerAdi do
      begin
        if (Char_Length(IlaveIslemler) + Char_Length(FIlaveIslemlerAdi) < 1020) then
        begin
         if (IlaveIslemler <> '' ) then IlaveIslemler = IlaveIslemler || cast(', ' as varchar(2) character set UTF8);
         IlaveIslemler = IlaveIslemler || FIlaveIslemlerAdi;
        end
      end

      if (EgemenFirmaId = 3 or EgemenFirmaId = 11) then /* modern kumas ve iplik */
      begin
        LoginErpTransferKodu = Null;
        if (HamTipiId = hstIplik and Fason = 0) then LoginErpTransferKodu = FSatisTransferKodu;
        else if (HamTipiId = hstIplik and Fason = 1) then Select FasonIplikTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuID into :LoginErpTransferKodu;
        else if (HamTipiId <> hstIplik and Fason = 0) then Select SatisKumasTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuID into :LoginErpTransferKodu;
        else if (HamTipiId <> hstIplik and Fason = 1) then Select FasonKumasTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuID into :LoginErpTransferKodu;
      end
    end
    else
    begin
      if (EgemenFirmaID = 4 or EgemenFirmaID = 15 or EgemenFirmaId = 7 or EgemenFirmaId = 26) then /* gülle , ozak ve imaj(Ottoman)*/
      begin
        TransferKodu = FHamSatisTransferKodu;
        MuhGelirHesapKodu = HamStokMuhHamSatisHesapKodu;
      end
      else if (EgemenFirmaID = 34) then /* örkum */
      begin
        select TransferKodu, HareketSekli, Null from BoyaIrsaliyeFaturaTransferKodu(:BoyaIrsaliyeDetayId, Null, :EgemenFirmaId) into :TransferKodu, :HareketSekli, :MuhGelirHesapKodu;
      end
      else
      begin
        Select HamStokMaster.Renk from RafDetay
          left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
          where RafDetay.RafDetayId = :RafDetayID
        into :GirisRenk;

        if (GirisRenk is not null and GirisRenk <> '') then TransferKodu = FSatisTransferKodu;
        else TransferKodu = FHamSatisTransferKodu;

        MuhGelirHesapKodu = HamStokMuhSatisHesapKodu;
      end

/*      if (HamTipiId = 2) then FaturaBirim = 1;  */
/*      else FaturaBirim = 0;*/
    end

    KdvMuhasebeHesapKodu = Null;

    Select MuhSatisHesapKodu from KdvHesapTablosu where KdvHesapTablosu.KdvOrani = :KdvOrani into :KdvMuhasebeHesapKodu;

    ToplamKdv = 0;
    IndirilecekKdv = 0;
    TahsilEdilecekKdv = 0;

    if (Tevkifat = 0) then
    begin
      TahsilEdilecekKdv = TutarTL * KdvOrani / 100;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, FKdvYuvarlamaCarpani);
    end
    else
    begin
      ToplamKdv = (TutarTL * KdvOrani / 100);

      IndirilecekKdv = ToplamKdv * TevkifatOrani1 / TevkifatOrani2;
      IndirilecekKdv = Yuvarla(IndirilecekKdv, FKdvYuvarlamaCarpani);

      TahsilEdilecekKdv = ToplamKdv - IndirilecekKdv;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, FKdvYuvarlamaCarpani);
    end
    suspend;
  end
