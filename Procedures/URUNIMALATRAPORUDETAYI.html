CREATE PROCEDURE URUNIMALATRAPORUDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICURUNGRUP1ID VARCHAR(4),
    ABITISURUNGRUP1ID VARCHAR(4),
    ABASLANGICURUNGRUP2ID VARCHAR(4),
    ABITISURUNGRUP2ID VARCHAR(4),
    ABASLANGICURUNGRUP3ID VARCHAR(4),
    ABITISURUNGRUP3ID VARCHAR(4),
    ABASLANGICURUNGRUP4ID VARCHAR(4),
    ABITISURUNGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER)
RETURNS (
    URUNURETIMID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BIRIMID VARCHAR(2),
    MIKTAR DOUBLE PRECISION,
    LOT VARCHAR(150))
AS
begin
  for select UrunImalatRaporu.UrunUretimId, StokDetay.UrunId, Urun.Adi, UrunBirim.BirimId, StokDetay.Miktar, StokDetay.Lot from UrunImalatRaporu(:ABaslangicTarihi, :ABitisTarihi, :ABaslangicUrunId, :ABitisUrunId, :ABaslangicUrunGrup1Id,
   :ABitisUrunGrup1Id, :ABaslangicUrunGrup2Id, :ABitisUrunGrup2Id, :ABaslangicUrunGrup3Id, :ABitisUrunGrup3Id, :ABaslangicUrunGrup4Id,
   :ABitisUrunGrup4Id, :AUrunAralikli, :AUrunGrup1Aralikli, :AUrunGrup2Aralikli, :AUrunGrup3Aralikli, :AUrunGrup4Aralikli)
  left join Stok on Stok.UrunUretimId = UrunImalatRaporu.UrunUretimId
  left join StokDetay on StokDetay.StokId = Stok.StokId
  left join Urun on Urun.UrunId = StokDetay.UrunId
  left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
  where StokDetay.GirisCikis = -1
  into :UrunUretimId, :UrunId, :UrunAdi, :BirimId, :Miktar, :Lot do suspend;
end
