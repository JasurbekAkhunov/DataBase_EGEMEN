      left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
      left join Makina on Makina.MakinaID = UretimPlanlama.MakinaId
      where BolumPlanlama.PartiAsamalariId = :Boyaasamasipartiasamalariid and BolumPlanlama.PartiSinifId = :PartiSinifId
      into :PlanlandigiMakinaId, :PlanlandigiMakinaKapasite;
    end

    HamStokMasterIDStr = '';
    ReferansSiparisNo = '';

    if ((SonAsamaId is null or SonAsamaId <> FTamirAsamaId) or FTamirAsamaId is null) then
    begin
      SonrakiSiraId = null;

      select min(Sira) from PartiAsamaStok
         left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
         where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and
               PartiAsamalari.Sira > :SonSira and PartiAsamalari.OtomasyonaDahilmi = 1
         into :SonrakiSiraId;         /* and PartiAsamalari.SerbestAsama = 0  */

      if (SonrakiSiraId is not null) then
      begin
        select Asama.AsamaId, Asama.Adi, PartiStokBekle.Bitti, PartiStokBekle.Notu, Asama.BoyahaneBolumId, Asama.GuRaporuSiraNo from PartiAsamaStok
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
          left join PartiStokBekle on PartiStokBekle.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiStokBekle.PartiSinifId = PartiAsamaStok.PartiSinifId and PartiStokBekle.Bitti = 0
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :SonrakiSiraId
          into :SonrakiAsamaId, :SonrakiAsamaAdi, :Bitti, :FBlob, :SonrakiAsamaBolumId, :SonrakiGuRaporuSiraNo;

        if (Bitti is not null) then
        begin
          Beklemede = 1;
          if (char_length(FBlob) > 1023) then BeklemeNotu = left(FBlob, 1023);
          else BeklemeNotu = left(FBlob, char_length(FBlob));
        end
      end
    end

    if (GirisTarihi is not null) then GirisVar = 'Var';
    else GirisVar = '';
    if (CikisTarihi is not null) then CikisVar = 'Var';
    else CikisVar = '';

    if (BoyaSiparisYil is null) then BoyaSiparisIdStr = '';
    else BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (PartiOnayDurumu = 1) then PartiOnayDurumuStr = 'Evet';
    else PartiOnayDurumuStr = '';

    if ( (ASadeceSiradakiAsamalar = 0 or (AsamaId = SonrakiAsamaId and AsamaSiraNo = SonrakiSiraId) ) and
    (ABoyaAsamasiPlanlanmislar=0 or (ABoyaAsamasiPlanlanmislar =1 and (PlanlandigiMakinaId is not null or BoyaAsamasiPartiAsamalariId is null and AciliyetDerecesiId = 1)) ) ) then
    begin
      if (ABoyaAsamasiPlanlanmislar = 1 and BoyaAsamasiPartiAsamalariId is null and AciliyetDerecesiId = 1) then
        Sira = 0;

      if (AsamaId = SonrakiAsamaId) then Hazir = 1;
      else Hazir = 0;

      for select HamStokMaster.HamStokMasterID, HamStokMaster.ReferansSiparisNo from PartiStok
        left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiStok.PartiSinifID
        left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where PartiStok.PartiSinifId = :PartiSinifId
        group by HamStokMaster.HamStokMasterID, HamStokMaster.ReferansSiparisNo
        into :FHamStokMasterID, :FReferansSiparisNo do
      begin
        Str = EgemenIntegerToStr(FHamStokMasterID);

        if (char_length(HamStokMasterIDStr) + char_length(Str) < 47) then
        begin
          if (HamStokMasterIDStr <> '') then HamStokMasterIDStr = HamStokMasterIDStr || cast(', ' as varchar(2) character set UTF8);
          HamStokMasterIDStr = HamStokMasterIDStr || Str;
        end

        if (char_length(ReferansSiparisNo) + char_length(FReferansSiparisNo) < 57) then
        begin
          if (ReferansSiparisNo <> '') then ReferansSiparisNo = ReferansSiparisNo || cast(', ' as varchar(2) character set UTF8);
          ReferansSiparisNo = ReferansSiparisNo || FReferansSiparisNo;
        end
      end


      YapilacakAsamalar = '';
      for select coalesce(AsamaKisaAdi, '') from PartiStok
        left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = partistok.PartiSinifId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        where PartiStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.Giristarihi is null and
              PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1
        order by PartiAsamalari.Sira
        into :AsamaKisaAdi do
      begin
        if (char_length(YapilacakAsamalar) + char_length(AsamaKisaAdi) < 250) then
        begin
          if (YapilacakAsamalar <> '' ) then YapilacakAsamalar = YapilacakAsamalar || cast(', ' as varchar(2) character set UTF8);
          YapilacakAsamalar = YapilacakAsamalar || AsamaKisaAdi;
        end
      end
      CikistanSonraGecenSureDouble = current_timestamp - CikisTarihi;
      CikistanSonraGecenSure = TarihToStr(current_timestamp - CikisTarihi);

      select first 1 PartiAsamaStokDetay.CikisRafId from PartiAsamaStokDetay
      where PartiAsamaStokDetay.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId into :CikisRafId;


      AraKontrolSonuclar = '';
      For Select  AraKontrolTipi.Adi, coalesce(PartiAsamaAraKontrol.Sonuc, '')  from PartiAsamaAraKontrol
        Left Join AraKontrolTipi on AraKontrolTipi.AraKontrolTipiId = PartiAsamaAraKontrol.AraKontrolTipiId
        Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaAraKontrol.PartiAsamalariId
        Left Join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
        where PartiAsamaAraKontrol.PartiKayitId = :PartiKayitId and PartiAsamaAraKontrol.PartiSinifId = :PartiSinifId
        into :AraKontrolTipiAdi, :Sonuc do
      begin
        if (char_length(AraKontrolSonuclar) + char_length(AraKontrolTipiAdi)  + char_length(Sonuc) < 250) then
        begin
          if (AraKontrolSonuclar <> '' ) then AraKontrolSonuclar = AraKontrolSonuclar || cast(', ' as varchar(2) character set UTF8);
          AraKontrolSonuclar = AraKontrolSonuclar || AraKontrolTipiAdi || cast('-' as varchar(1) character set UTF8) || Sonuc ;
        end
      end

      BlobSize = char_length(FSipPlanlamaNotu);

      if (BlobSize > 2000) then SipPlanlamaNotu = Left(FSipPlanlamaNotu, 2000);
      else if (BlobSize > 0) then SipPlanlamaNotu = Left(FSipPlanlamaNotu, BlobSize);
      else SipPlanlamaNotu = '';

      suspend;
    end
  end
