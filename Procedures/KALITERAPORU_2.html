CREATE PROCEDURE KALITERAPORU_2(
    AKALITEADI ADI255,
    ARECETEKODU VARCHAR(20),
    AHAMID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ADESENKODU VARCHAR(20),
    ASIPARISPROSESID INTEGER,
    AOZELTESTLER VARCHAR(255),
    AOZELAPRELER VARCHAR(255),
    AILAVEISLEMLER VARCHAR(255),
    AENISTENENBASLANGIC VARCHAR(10),
    AENISTENENBITIS VARCHAR(10),
    AGRAMAJISTENENBASLANGIC VARCHAR(10),
    AGRAMAJISTENENBITIS VARCHAR(10),
    AENARALIKLI INTEGER,
    AGRAMAJARALIKLI INTEGER,
    AHAMADI ADI255,
    ASTOKHAMID INTEGER,
    ATALIMATKODU VARCHAR(20),
    AKAPALI EGEBOOLEAN)
RETURNS (
    KALITEID KALITEKODUSIZE,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    FIRMAID FIRMAKODUSIZE,
    FIRMAADI ADI100,
    TALIMATKODU VARCHAR(20),
    SIPARISTARIHI TIMESTAMP,
    SIPARISKILO BOYAMIKTARSIZE,
    SIPARISBRUTKILO BOYAMIKTARSIZE,
    REZERVEDILECEKKILO BOYAMIKTARSIZE,
    REZERVEDILECEKBRUTKILO BOYAMIKTARSIZE,
    REZERVEDILENKILO BOYAMIKTARSIZE,
    REZERVEDILENBRUTKILO BOYAMIKTARSIZE,
    REZERVEDILENTARTIBRUTKILO BOYAMIKTARSIZE,
    SEVKEDILENKILO BOYAMIKTARSIZE,
    SEVKEDILENBRUTKILO BOYAMIKTARSIZE,
    REZSEVKEDILENTARTIBRUTKILO BOYAMIKTARSIZE)
AS
begin
  for select KaliteRaporu.KaliteId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.HamId, Ham.Adi, BoyaSiparis.FirmaId, Firma.Adi, BoyaSiparisDetay.TalimatKodu, BoyaSiparis.SiparisTarihi,
    BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisBrutKilo
    from KaliteRaporu(:AKaliteAdi, :AReceteKodu, :AHamId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id, :ABitisGrup3Id,
                               :ABaslangicGrup4Id, :ABitisGrup4Id, :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ADesenKodu, :ASiparisProsesId,
                               :AOzelTestler, :AOzelApreler, :AIlaveIslemler, :AEnIstenenBaslangic, :AEnIstenenBitis, :AGramajIstenenBaslangic, :AGramajIstenenBitis, :AEnAralikli,
                               :AGramajAralikli, :AHamAdi, :AStokHamId, :ATalimatKodu, :AKapali)
    join BoyaSiparisDetay on BoyaSiparisDetay.KaliteId = KaliteRaporu.KaliteId and BoyaSiparisDetay.Kapali = 0
    join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    where BoyaSiparisDetay.HamId = :AStokHamId or :AStokHamId is null into
    :KaliteId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :HamId, :HamAdi, :FirmaId, :FirmaAdi, :TalimatKodu, :SiparisTarihi, :SiparisKilo, :SiparisBrutKilo do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    select sum(EdilecekKilo), sum(EdilenKilo), sum(EdilecekBrutKilo), sum(EdilenBrutKilo), sum(RezervEdilenTartiBrutKilo), sum(RezSevkEdilenBrutKilo)
    from (
      select BoyaRezervFisiDetay.EdilecekKilo, BoyaRezervFisiDetay.EdilenKilo, BoyaRezervFisiDetay.EdilecekBrutKilo, BoyaRezervFisiDetay.EdilenBrutKilo,
      Coalesce(Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara), 0) RezervEdilenTartiBrutKilo,
      Coalesce(Sum(iif(PartiStokTarti.BoyaIrsaliyeId is not null, PartiStokTarti.Kilo + PartiStokTarti.Dara, 0)), 0) RezSevkEdilenBrutKilo
      from BoyaRezervFisiDetay
      left join PartiStokTarti on PartiStokTarti.BoyaRezervFisiDetayId = BoyaRezervFisiDetay.BoyaRezervFisiDetayId
      where BoyaRezervFisiDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
      group by BoyaRezervFisiDetay.BoyaRezervFisiDetayId, BoyaRezervFisiDetay.EdilecekKilo, BoyaRezervFisiDetay.EdilenKilo, BoyaRezervFisiDetay.EdilecekBrutKilo, BoyaRezervFisiDetay.EdilenBrutKilo
    ) into :RezervEdilecekKilo, :RezervEdilenKilo, :RezervEdilecekBrutKilo, :RezervEdilenBrutKilo, :RezervEdilenTartiBrutKilo, :RezSevkEdilenTartiBrutKilo;

    Select Coalesce(Sum(PartiStok.BrutSevkKilo), 0), Coalesce(Sum(PartiStok.NetSevkKilo), 0) from BoyaSiparisDetay
      left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
      where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
      into :SevkEdilenBrutKilo, :SevkEdilenKilo;

    suspend;
  end
end
