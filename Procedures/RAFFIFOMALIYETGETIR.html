CREATE PROCEDURE RAFFIFOMALIYETGETIR(
    AORMERAFID INTEGER,
    AMIKTAR DOUBLE PRECISION,
    ATARIH TIMESTAMP,
    ASTOKHAREKETID INTEGER)
RETURNS (
    BIRIMMALIYET DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    NETGIRISTOPLAMI DOUBLE PRECISION,
    NETCIKISTOPLAMI DOUBLE PRECISION,
    TRNETGIRISTOPLAMI DOUBLE PRECISION,
    TRNETCIKISTOPLAMI DOUBLE PRECISION,
    KALANCIKISMIKTARI DOUBLE PRECISION,
    FMIKTAR DOUBLE PRECISION,
    FBIRIMMALIYET DOUBLE PRECISION)
AS
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable FTarih TimeStamp;
declare variable RafCikisMiktari Double Precision;
begin

  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim;

  select sum(OrmeRafDetay.Miktar) from OrmeRafDetay
    left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketId
    where OrmeRafDetay.OrmeRafId = :AOrmeRafId and StokHareket.Tarih <= :ATarih
    into :NetGirisToplami;
  select sum(Kilo) from OrmeRafTransferFisiDetay
    left join OrmeRafTransferFisi on OrmeRafTransferFisi.OrmeRafTransferFisiId = OrmeRafTransferFisiDetay.OrmeRafTransferFisiId
    where OrmeRafTransferFisiDetay.YeniOrmeRafId = :AOrmeRafId and OrmeRafTransferFisi.FisTarihi <= :ATarih
    into :TrNetGirisToplami;

  select sum(Miktar) from StokHareket
    where OrmeRafId = :AOrmeRafId and StokHareket.Tarih <= :ATarih  and
    (StokHareket.Tarih < :ATarih or StokHareket.StokHareketId < :AStokHareketId)
    into :NetCikisToplami;
  select sum(Kilo) from OrmeRafTransferFisiDetay
    left join OrmeRafTransferFisi on OrmeRafTransferFisi.OrmeRafTransferFisiId = OrmeRafTransferFisiDetay.OrmeRafTransferFisiId
    where OrmeRafTransferFisiDetay.OrmeRafId = :AOrmeRafId and OrmeRafTransferFisi.FisTarihi <= :ATarih  and
    (OrmeRafTransferFisi.FisTarihi < :ATarih or OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId < :AStokHareketId)
    into :TrNetCikisToplami;

  if (NetGirisToplami is Null) then NetGirisToplami = 0;
  if (TrNetGirisToplami is Null) then TrNetGirisToplami = 0;
  if (NetCikisToplami is Null) then NetCikisToplami = 0;
  if (TrNetCikisToplami is Null) then TrNetCikisToplami = 0;
  NetGirisToplami = NetGirisToplami + TrNetGirisToplami;
  NetCikisToplami = NetCikisToplami + TrNetCikisToplami;

  KalanMiktar = AMiktar;
  KalanCikisMiktari = NetCikisToplami;
  BirimMaliyet = 0;
  For Select StokHareket.Tarih, StokHareket.Miktar, StokHareket.BirimMaliyet from OrmeRafDetay
    left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketId
    where OrmeRafDetay.OrmeRafId = :AOrmeRafId and StokHareket.Tarih <= :ATarih and StokHareket.GirisCikis = 1
    order by StokHareket.Tarih, StokHareket.UretimIslemTipiId, StokHareket.HareketTipi, StokHareket.StokHareketId
    into :FTarih, :FMiktar, :FBirimMaliyet do
  begin
    if (KalanMiktar > 0) then
    begin
      if (KalanCikisMiktari >= FMiktar) then
      begin
        KalanCikisMiktari = KalanCikisMiktari - FMiktar;
      end
      else
      begin
        KalanCikisMiktari = KalanCikisMiktari - FMiktar;
        if ((KalanCikisMiktari + FMiktar) > 0) then FMiktar = FMiktar - (KalanCikisMiktari + FMiktar);
        if (FMiktar >= KalanMiktar) then
        begin
          BirimMaliyet = BirimMaliyet + ((KalanMiktar * FBirimMaliyet) / AMiktar);
        end
        else
        begin
          BirimMaliyet = BirimMaliyet + ((FMiktar * FBirimMaliyet) / AMiktar);
        end
        KalanMiktar = KalanMiktar - FMiktar;
      end
    end
  end

  if (KalanMiktar > 0) then
  begin
    For Select OrmeRafTransferFisi.FisTarihi, OrmeRafTransferFisiDetay.Kilo, OrmeRafTransferFisiDetay.BirimMaliyet from OrmeRafTransferFisiDetay
      left join OrmeRafTransferFisi on OrmeRafTransferFisi.OrmeRafTransferFisiId = OrmeRafTransferFisiDetay.OrmeRafTransferFisiId
      where OrmeRafTransferFisiDetay.YeniOrmeRafId = :AOrmeRafId and OrmeRafTransferFisi.FisTarihi <= :ATarih
      order by OrmeRafTransferFisi.FisTarihi, OrmeRafTransferFisiDetay.OrmeRafTransferFisiDetayId
      into :FTarih, :FMiktar, :FBirimMaliyet do
    begin
      if (KalanMiktar > 0) then
      begin
        if (KalanCikisMiktari >= FMiktar) then
        begin
          KalanCikisMiktari = KalanCikisMiktari - FMiktar;
        end
        else
        begin
          KalanCikisMiktari = KalanCikisMiktari - FMiktar;
          if ((KalanCikisMiktari + FMiktar) > 0) then FMiktar = FMiktar - (KalanCikisMiktari + FMiktar);
          if (FMiktar >= KalanMiktar) then
          begin
            BirimMaliyet = BirimMaliyet + ((KalanMiktar * FBirimMaliyet) / AMiktar);
          end
          else
          begin
            BirimMaliyet = BirimMaliyet + ((FMiktar * FBirimMaliyet) / AMiktar);
          end
          KalanMiktar = KalanMiktar - FMiktar;
        end
      end
    end
  end

  Suspend;
end
