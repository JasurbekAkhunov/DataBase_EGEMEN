CREATE PROCEDURE FIYATLANDIRILACAKPARTILER(
    AFIRMAID VARCHAR(10),
    ARAPORTIPI INTEGER)
RETURNS (
    FIRMAID VARCHAR(10),
    FIRMAADI VARCHAR(100),
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    FASONSTR VARCHAR(10),
    PARTISINIFID INTEGER,
    MODELNO VARCHAR(30),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    KILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION,
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    IRSALIYENO VARCHAR(255),
    BASLAMIS VARCHAR(4))
AS
declare variable KayitSayisi integer;
declare variable PaketKayitSayisi integer;
declare variable FIrsaliyeNo IrsaliyeNoSize;
declare variable LastPartiSinifId integer;
declare variable FFirmaId       varchar(10);
declare variable FFirmaAdi      varchar(100);
declare variable FPartiKayitId  integer;
declare variable FPartiYil      integer;
declare variable FPartiId       integer;
declare variable FPartiSinifId  integer;
declare variable FModelNo       varchar(30);
declare variable FBoyaSiparisYil          integer;
declare variable FBoyaSiparisId           integer;
declare variable FBoyaSiparisIdStr        varchar(30);
declare variable FBoyaSiparisDetayId      integer;
declare variable FReceteId      varchar(20);
declare variable FReceteAdi     varchar(40);
declare variable FKilo          double Precision;
declare variable FMetre         double Precision;
declare variable FFason         integer;
declare variable FIslemTipi     integer;
declare variable FIslemGrubu    integer;
declare variable FIslemTipiAdi  varchar(20);
declare variable FIslemGrubuAdi varchar(20);
declare variable SartlarOkey    integer;
declare variable Str            varchar(20);

begin
  LastPartiSinifId = -1;
  SartlarOkey = 0;
  for select Parti.FirmaId, Firma.Adi, parti.partikayitid, parti.partiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.ModelNo,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId, Parti.ReceteId, Recete.Adi, Parti.fason,
             Parti.IslemTipiId, Parti.IslemGrubuID, IslemTipi.Adi, IslemGrubu.Adi, PartiStok.Kilo, PartiStok.Metre, BoyaIrsaliye.IrsaliyeNo from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiKayitId = Parti.PartiKayitId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BOyaIrsaliyeDetay.BoyaIrsaliyeId
    where (Parti.firmaid = :AFirmaId or :AFirmaId is null) and Parti.faturaKesilecek = 1 and Parti.faturakesildi = 0 and PartiStok.PartiPakettenFiyatlandirildi = 0 and
          PartiStok.PartiStokIptalKayitId is null and PartiStok.FiyatOnayiIstensin = 1 and PartiStok.FiyatOnaylayanKullaniciId is null
    Group By Parti.FirmaId, Firma.Adi, parti.partikayitid, parti.partiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.ModelNo,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId, Parti.ReceteId, Recete.Adi, Parti.fason,
             Parti.IslemTipiId, Parti.IslemGrubuID, IslemTipi.Adi, IslemGrubu.Adi, PartiStok.Kilo, PartiStok.Metre, BoyaIrsaliye.IrsaliyeNo
    order by Parti.FirmaId, Parti.PartiKayitId
    into :FFirmaId, :FFirmaAdi, :Fpartikayitid, :FpartiYil, :FPartiId, :FPartiSinifId, :FModelNo,
         :FBoyaSiparisYil, :FBoyaSiparisId, :FBoyaSiparisDetayId,:FReceteId, :FReceteAdi, :Ffason,
         :FIslemTipi, :FIslemGrubu, :FIslemTipiAdi, :FIslemGrubuAdi, :FKilo, :FMetre, :FIrsaliyeNo do
  begin
    if (LastPartiSinifId <> FPartiSinifId) then
    begin
      if (LastPartiSinifId <> -1 and SartlarOkey = 1) then
      begin
        BoyaSiparisIdStr = '';
        if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

        PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
        suspend;
      end
      LastPartiSinifId  = FPartiSinifId;
      FirmaId       = FFirmaId;
      FirmaAdi      = FFirmaAdi;
      PartiKayitId  = FPartiKayitId;
      PartiSinifId  = FPartiSinifId;
      PartiYil      = FPartiYil;
      PartiId       = FPartiId;
      BoyaSiparisYil = FBoyaSiparisYil;
      BoyaSiparisId  = FBoyaSiparisId;
      BoyaSiparisDetayId  = FBoyaSiparisDetayId;
      ModelNo       = FModelNo;
      ReceteId      = FReceteId;
      ReceteAdi     = FReceteAdi;
      Kilo          = FKilo;
      Metre         = FMetre;
      IslemTipiAdi  = FIslemTipiAdi;
      IslemGrubuAdi = FIslemGrubuAdi;

      if (FFason = 1) then FasonStr = 'Fason';
      else FasonStr = '';
      IrsaliyeNo    = '';
      SartlarOkey = 1;

      Select Count(*) from PartiPaket where PartiPaket.PartiSinifId = :PartiSinifId into :PaketKayitSayisi;
      if (PaketKayitSayisi > 0) then Baslamis = 'Evet';
      else Baslamis = '';

      if (ARaporTipi is not null) then
      if (ARaporTipi = 0 and PaketKayitSayisi > 0) then SartlarOkey = 0;
      else if (ARaporTipi = 1 and PaketKayitSayisi = 0) then SartlarOkey = 0;
    end

    if (FIrsaliyeNo is not null) THen Str = Format_Float('#0', FIrsaliyeNo);
    else Str = '';

    if (Str <> '') then
    if (PosBul(IrsaliyeNo, Str) = 0) then
    if (Char_Length(IrsaliyeNo) + Char_Length(Str) < 250) then
    begin
      if (IrsaliyeNo <> '') then IrsaliyeNo = IrsaliyeNo || cast(', ' as varchar(2) character set UTF8);
      IrsaliyeNo = IrsaliyeNo || Str;
    end
  end

  if (LastPartiSinifId <> -1 and SartlarOkey = 1) then
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    suspend;
  end
end
