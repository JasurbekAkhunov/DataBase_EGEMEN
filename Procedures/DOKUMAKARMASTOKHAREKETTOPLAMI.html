AS
declare variable Tamir   integer;
declare variable FMiktar Double Precision;
begin
  Select StokTabanDovizId from Sistem where SistemId = 1 into :DovizId;

  For Select HareketTipi, HareketTipiStr, UretimIslemTipiId, UretimIslemTipiIdStr, HamIade, sum(Miktar), sum(BrutMiktar), sum(KdvsizTutar), sum(TahsilEdilecekKdv), Sum(KdvliTutar) From DokumaKarmaStokHareketRaporu(:ATarihTipi, :ABASLANGICTARIHI, :ABITISTARIHI, :ABASLANGICBOYASIPARISYIL, :ABASLANGICBOYASIPARISID, :ABITISBOYASIPARISYIL,
