    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ALABRENKGRUPID INTEGER,

    LABRECETEATISID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),

    PARTISAYISI INTEGER,
    PARTITOPLAMKILO DOUBLE PRECISION,
    BOYAYUZDESI DOUBLE PRECISION,
    BOYAMALIYETI DOUBLE PRECISION,
    KIMYASALMALIYETI DOUBLE PRECISION,
    BOYAMALIYETITL DOUBLE PRECISION,
    KIMYASALMALIYETITL DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    TUZMIKTARI DOUBLE PRECISION,
    SODAMIKTARI DOUBLE PRECISION,
    RECETEONISLEMTIPIADI VARCHAR(30),
    LABPROGRAMNO INTEGER,
    LABPROGRAMNO2 INTEGER,
    ACILPIPETLENECEK INTEGER,
    ACILPIPETLENECEKSTR VARCHAR(5),
    PIPETLEMEYEGONDERILDI INTEGER,
    PIPETLEMEYEGONDERILDISTR VARCHAR(5),
    PIPETLEMETARIHI TIMESTAMP,
    ETIKETYAZDIRILDI INTEGER,
    ETIKETYAZDIRILDISTR VARCHAR(5),
    ETIKETYAZDIRMATARIHI TIMESTAMP,
    KALITEID VARCHAR(20),
    CALISMABIRIMFIYAT DOUBLE PRECISION,
    CALISMADOVIZID VARCHAR(5),
    RECETETALEPGRUP1ID VARCHAR(4),
    RECETETALEPGRUP1ADI VARCHAR(20),
    FIYATFARKI DOUBLE PRECISION,
    LABGRUPID VARCHAR(20),
    LABGRUPADI VARCHAR(60))
AS
declare variable FCalismaNotu Blob;
declare variable BlobSize integer;
declare variable ReceteId varchar(20);
declare variable TuzKimyasalTipiId integer;
declare variable SodaKimyasalTipiID integer;
declare variable SistemDovizId  varchar(5);
declare variable DovizKuru  Double Precision;
begin
  Select TuzKimyasalTipiID, SodaKimyasalTipiID, SistemDovizId from sistem where sistemid = 1 into :TuzKimyasalTipiID, :SodaKimyasalTipiID, :SistemDovizId;

  if (AraporTipi = 0) then
  begin
    for Select LabReceteAtis.LabReceteAtisId, LabReceteCalisma.LabReceteCalismaId, LabRecete.LabReceteId, LabRecete.LabReceteKodu, LabRecete.Adi, LabRecete.FirmaId, Firma.Adi, LabRecete.TalepTarihi,

               LabRecete.ReceteTuruId, ReceteTuru.Adi, LabRecete.Fason, LabRecete.MusteriRenkKodu, LabReceteCalisma.HamId, Ham.Adi,
               LabRecete.Aciklama, LabRecete.Aciklama2, LabRecete.EskiLabReceteKodu, LabRecete.PantoneKodu,
               LabReceteCalisma.CalismaTarihi, LabReceteCalisma.OrjinalDegisimTarihi, LabReceteCalisma.TerminTarihi, LabReceteCalisma.BitisTarihi, LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave,
               LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, LabReceteAtis.AtisNoStr, LabReceteAtis.AtisTarihi, LabReceteAtis.GidisTarihi, LabReceteAtis.KartelaTarihi,
               LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi, LabReceteAtis.Okey, LabReceteAtis.Kritik,
               LabReceteCalisma.Notu, LabReceteAtis.BoyaYuzdesi, LabReceteAtis.BoyaMaliyeti, LabReceteAtis.KimyasalMaliyeti, LabRecete.DovizId,
               LabReceteAtis.EtiketYazdirildi, LabReceteAtis.EtiketYazdirmaTarihi, ReceteOnIslemTipi.Adi,
               LabReceteCalisma.LabProgramNo, LabReceteCalisma.LabProgramNo2, LabReceteAtis.AcilPipetlenecek, LabReceteAtis.PipetlemeyeGonderildi,
               LabReceteAtis.PipetlemeTarihi, LabRecete.KaliteID, LabReceteCalisma.BirimFiyat, LabReceteCalisma.DovizID, LabReceteTalep.ReceteTalepGrup1,
               ReceteTalepGrup.Adi, LabReceteCalisma.FiyatFarki, LabGrup.LabGrupId, LabGrup.Adi from LabRecete
    left join Firma On Firma.FirmaId = LabRecete.FirmaId
    left join LabReceteGrup On LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
    left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = LabRecete.LabRenkGrupId
    left join ReceteTuru On ReceteTuru.ReceteTuruId = LabRecete.ReceteTuruId
    left join RenkDerinligi On RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
    left join LabReceteCalisma On LabReceteCalisma.LabReceteId = LabRecete.LabReceteId
    left join Ham On Ham.HamId = LabReceteCalisma.HamId
    left join LabReceteAtis On LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
    left join ReceteOnIslemTipi on ReceteOnIslemTipi.ReceteOnIslemTipiID = LabReceteCalisma.ReceteOnIslemTipiId
    left join LabReceteTalep on LabReceteTalep.LabReceteTalepKayitId = LabReceteCalisma.LabReceteTalepKayitId
    left join ReceteTalepGrup on ReceteTalepGrup.ReceteTalepGrupId = LabReceteTalep.ReceteTalepGrup1
    left join LabGrup On LabGrup.LabGrupId = LabRecete.LabGrupId
    where LabRecete.TalepTarihi >= :ABaslangicTarihi and LabRecete.TalepTarihi <= :ABitisTarihi and
          (LabRecete.FirmaId = :AFirmaId or :AFirmaId is null) and
          (LabRecete.LabRenkGrupId = :ALabRenkGrupId or :ALabRenkGrupId is null) and
          (LabRecete.LabReceteGrupId = :ALabReceteGrupId or :ALabReceteGrupId is null) and
          (LabRecete.ReceteTuruId = :AReceteTuruId or :AReceteTuruId is null) and
          (LabRecete.RenkDerinligiId = :ARenkDerinligiId or :ARenkDerinligiId is null) and
          (LabRecete.LabReceteKodu = :AReceteKodu or :AReceteKodu is null) and
          (LabRecete.Adi like '%' || :AReceteAdi || '%' or :AReceteAdi is null) and
          (LabReceteCalisma.LabProgramNo >= :ABaslangicLabProgramNo or :ABaslangicLabProgramNo is null) and
          (LabReceteCalisma.LabProgramNo <= :ABitisLabProgramNo or :ABitisLabProgramNo is null) and
          (LabReceteCalisma.LabProgramNo2 >= :ABaslangicLabProgramNo2 or :ABaslangicLabProgramNo2 is null) and
          (LabReceteCalisma.LabProgramNo2 <= :ABitisLabProgramNo2 or :ABitisLabProgramNo2 is null) and
          ( (:ACalismaDurumu = 1 and LabReceteCalisma.Rptmi = 1) or
            (:ACalismaDurumu = 2 and LabReceteCalisma.Ilave = 1) or
            (:ACalismaDurumu = 3 and LabReceteCalisma.SokumUzeriBoya = 1) or
            (:ACalismaDurumu = 0 and (LabReceteCalisma.Rptmi = 0 and LabReceteCalisma.Ilave = 0 and LabReceteCalisma.SokumUzeriBoya = 0) ) or
            :ACalismaDurumu is null) and
          (LabReceteAtis.Okey = 1 or :ASadeceOkeyliAtislar = 0) and
          (LabReceteAtis.EtiketYazdirildi = 0 or :ASadeceEtiketYazdirilmayan = 0) and
          (LabReceteAtis.PipetlemeyeGonderildi = 0 or :APipetlemeyeGonderilmeyenler= 0)
    into :LabReceteAtisId, :LabReceteCalismaId, :LabReceteId, :LabReceteKodu, :LabReceteAdi, :FirmaId, :FirmaAdi, :TalepTarihi, :LabReceteGrupId, :LabReceteGrupAdi, :LabRenkGrupId, :LabRenkGrupAdi,
         :RenkDerinligiId, :RenkDerinligiAdi, :ReceteTuruId, :ReceteTuruAdi, :Fason, :MusteriRenkKodu, :HamId, :HamAdi,
         :Aciklama, :Aciklama2, :EskiLabReceteKodu, :PantoneKodu, :CalismaTarihi, :OrjinalDegisimTarihi, :TerminTarihi, :CalismaBitisTarihi,  :Rptmi, :Ilave, :SokumUzeriBoya, :KritikCalismasi,
         :AtisNoStr, :AtisTarihi, :GidisTarihi, :KartelaTarihi, :DurumuStr, :Tarihi, :Okey, :Kritik, :FCalismaNotu,
         :BoyaYuzdesi, :BoyaMaliyeti, :KimyasalMaliyeti, :DovizId, :EtiketYazdirildi,  :EtiketYazdirmaTarihi, :ReceteOnIslemTipiAdi,
         :LabProgramNo, :LabProgramNo2, :AcilPipetlenecek, :PipetlemeyeGonderildi, :PipetlemeTarihi, :KaliteID, :CalismaBirimFiyat, :CalismaDovizID, :ReceteTalepGrup1Id, :ReceteTalepGrup1Adi,
         :FiyatFarki ,:LabGrupId, :LabGrupAdi do
    begin
      if (Fason = 0) then FasonStr = 'Satış';
      else FasonStr = 'Fason';

      if (EtiketYazdirildi = 1) then EtiketYazdirildiStr = 'Evet';
      else EtiketYazdirildiStr = '';

      if (AcilPipetlenecek = 1) then AcilPipetlenecekStr = 'Evet';
      else AcilPipetlenecekStr = '';

      if (PipetlemeyeGonderildi = 1) then PipetlemeyeGonderildiStr = 'Evet';
      else PipetlemeyeGonderildiStr = '';

      BlobSize = char_length(FCalismaNotu);

      if (BlobSize > 1023) then CalismaNotu = Left(FCalismaNotu, 1023);
      else if (BlobSize > 0) then CalismaNotu = Left(FCalismaNotu, BlobSize);
      else CalismaNotu = '';

      CalismaTipiId = 0;

      if (Rptmi = 1) then CalismaTipiId = 1;
      else if (Ilave = 1) then CalismaTipiId = 2;
      else if (SokumUzeriBoya = 1) then CalismaTipiId = 3;
      else if (KritikCalismasi = 1) then CalismaTipiId = 4;

      PartiToplamKilo = Null;
      PartiSayisi = Null;
      TuzMiktari = Null;
      SodaMiktari = null;

      if (AtisNoStr is not null) then
      begin
        ReceteId = LabReceteKodu || '.' || AtisNoStr;


        Select Count(*) from Parti where Parti.ReceteID = :ReceteId into :PartiSayisi;

        if (PartiSayisi > 0) then
        begin
          Select Sum(PartiStok.Kilo) from Parti
            left join PartiStok on PartiStok.PartiKayitID = Parti.PartiKayitId
            where Parti.ReceteID = :ReceteId and PartiStok.PartiStokIptalKayitID is null
            into :PartiToplamKilo;
        end

        if (PartiSayisi = 0) then PartiSayisi = Null;

        if (TuzKimyasalTipiID is not null) then
        begin
          Select Sum(LabReceteUrun.MiktarYuzde) from LabReceteUrun
          left join Urun On Urun.UrunID = LabReceteUrun.UrunId
          where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and LabReceteUrun.UrunID is not null and Urun.KimyasalTipiID = :TuzKimyasalTipiID
          into :TuzMiktari;
        end

        if (SodaKimyasalTipiID is not null) then
        begin
          Select Sum(LabReceteUrun.MiktarYuzde) from LabReceteUrun
          left join Urun On Urun.UrunID = LabReceteUrun.UrunId
          where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and LabReceteUrun.UrunID is not null and Urun.KimyasalTipiID = :SodaKimyasalTipiID
          into :SodaMiktari;
        end
      end

      select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :SistemDovizId, :AtisTarihi) into :DovizKuru;

      BoyaMaliyetiTL = BoyaMaliyeti * DovizKuru;
      KimyasalMaliyetiTL = KimyasalMaliyeti * DovizKuru;

      if (AUrunFiyatGorebilir = 0) then
      begin
        BoyaMaliyeti     = Null;
        KimyasalMaliyeti = Null;
        BoyaMaliyetiTL   = Null;
        KimyasalMaliyetiTL = Null;
      end
      suspend;
    end
  end
  else if (AraporTipi = 1) then /* calisma tarihine göre */
  begin
    for Select LabReceteAtis.LabReceteAtisId, LabReceteCalisma.LabReceteCalismaId, LabRecete.LabReceteId, LabRecete.LabReceteKodu, LabRecete.Adi, LabRecete.FirmaId, Firma.Adi, LabRecete.TalepTarihi,
               LabRecete.LabReceteGrupId, LabReceteGrup.Adi, LabRecete.LabRenkGrupId, LabRenkGrup.Adi,  LabRecete.RenkDerinligiId, RenkDerinligi.Adi,
               LabRecete.ReceteTuruId, ReceteTuru.Adi, LabRecete.Fason, LabRecete.MusteriRenkKodu, LabReceteCalisma.HamId, Ham.Adi,
               LabRecete.Aciklama, LabRecete.Aciklama2, LabRecete.EskiLabReceteKodu, LabRecete.PantoneKodu,
               LabReceteCalisma.CalismaTarihi, LabReceteCalisma.OrjinalDegisimTarihi, LabReceteCalisma.TerminTarihi, LabReceteCalisma.BitisTarihi, LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave,
               LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, LabReceteAtis.AtisNoStr, LabReceteAtis.AtisTarihi, LabReceteAtis.GidisTarihi, LabReceteAtis.KartelaTarihi,
               LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi, LabReceteAtis.Okey, LabReceteAtis.Kritik, LabReceteCalisma.Notu,
               LabReceteAtis.BoyaYuzdesi, LabReceteAtis.BoyaMaliyeti, LabReceteAtis.KimyasalMaliyeti, LabRecete.DovizId,
               LabReceteAtis.EtiketYazdirildi, LabReceteAtis.EtiketYazdirmaTarihi, ReceteOnIslemTipi.Adi,
               LabReceteCalisma.LabProgramNo, LabReceteCalisma.LabProgramNo2, LabReceteAtis.AcilPipetlenecek, LabReceteAtis.PipetlemeyeGonderildi,
               LabReceteAtis.PipetlemeTarihi, LabRecete.KaliteID, LabReceteCalisma.BirimFiyat, LabReceteCalisma.DovizID, LabReceteTalep.ReceteTalepGrup1,
               ReceteTalepGrup.Adi, LabReceteCalisma.FiyatFarki, LabGrup.Labgrupid, LabGrup.Adi from LabReceteCalisma
    left join LabRecete On LabRecete.LabReceteId = LabReceteCalisma.LabReceteId
    left join Firma On Firma.FirmaId = LabRecete.FirmaId
    left join LabReceteGrup On LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
    left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = LabRecete.LabRenkGrupId
    left join ReceteTuru On ReceteTuru.ReceteTuruId = LabRecete.ReceteTuruId
    left join RenkDerinligi On RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
    left join Ham On Ham.HamId = LabReceteCalisma.HamId
    left join LabReceteAtis On LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
    left join ReceteOnIslemTipi on ReceteOnIslemTipi.ReceteOnIslemTipiID = LabReceteCalisma.ReceteOnIslemTipiId
    left join LabReceteTalep on LabReceteTalep.LabReceteTalepKayitId = LabReceteCalisma.LabReceteTalepKayitId
    left join ReceteTalepGrup on ReceteTalepGrup.ReceteTalepGrupId = LabReceteTalep.ReceteTalepGrup1
    left join LabGrup On LabGrup.LabGrupId = LabRecete.LabGrupId
    where LabReceteCalisma.CalismaTarihi >= :ABaslangicTarihi and LabReceteCalisma.CalismaTarihi <= :ABitisTarihi and
          (LabRecete.FirmaId = :AFirmaId or :AFirmaId is null) and
          (LabRecete.LabRenkGrupId = :ALabRenkGrupId or :ALabRenkGrupId is null) and
          (LabRecete.LabReceteGrupId = :ALabReceteGrupId or :ALabReceteGrupId is null) and
          (LabRecete.ReceteTuruId = :AReceteTuruId or :AReceteTuruId is null) and
          (LabRecete.RenkDerinligiId = :ARenkDerinligiId or :ARenkDerinligiId is null) and
          (LabRecete.LabReceteKodu = :AReceteKodu or :AReceteKodu is null) and
          (LabRecete.Adi like '%' || :AReceteAdi || '%' or :AReceteAdi is null) and
          (LabReceteCalisma.LabProgramNo >= :ABaslangicLabProgramNo or :ABaslangicLabProgramNo is null) and
          (LabReceteCalisma.LabProgramNo <= :ABitisLabProgramNo or :ABitisLabProgramNo is null) and
          (LabReceteCalisma.LabProgramNo2 >= :ABaslangicLabProgramNo2 or :ABaslangicLabProgramNo2 is null) and
          (LabReceteCalisma.LabProgramNo2 <= :ABitisLabProgramNo2 or :ABitisLabProgramNo2 is null) and
          ( (:ACalismaDurumu = 1 and LabReceteCalisma.Rptmi = 1) or
            (:ACalismaDurumu = 2 and LabReceteCalisma.Ilave = 1) or
            (:ACalismaDurumu = 3 and LabReceteCalisma.SokumUzeriBoya = 1) or
            (:ACalismaDurumu = 0 and (LabReceteCalisma.Rptmi = 0 and LabReceteCalisma.Ilave = 0 and LabReceteCalisma.SokumUzeriBoya = 0) ) or
            :ACalismaDurumu is null) and

        end

        if (SodaKimyasalTipiID is not null) then
        begin
          Select Sum(LabReceteUrun.MiktarYuzde) from LabReceteUrun
          left join Urun On Urun.UrunID = LabReceteUrun.UrunId
          where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and LabReceteUrun.UrunID is not null and Urun.KimyasalTipiID = :SodaKimyasalTipiID
          into :SodaMiktari;
        end
      end

      select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :SistemDovizId, :AtisTarihi) into :DovizKuru;

      BoyaMaliyetiTL = BoyaMaliyeti * DovizKuru;
      KimyasalMaliyetiTL = KimyasalMaliyeti * DovizKuru;

      if (AUrunFiyatGorebilir = 0) then
      begin
        BoyaMaliyeti     = Null;
        KimyasalMaliyeti = Null;
        BoyaMaliyetiTL   = Null;
        KimyasalMaliyetiTL = Null;
      end

      suspend;
    end
  end
  else if (AraporTipi = 4) then /* orjinal Degisim tarihine göre */
  begin
    for Select LabReceteAtis.LabReceteAtisId, LabReceteCalisma.LabReceteCalismaId, LabRecete.LabReceteId, LabRecete.LabReceteKodu, LabRecete.Adi, LabRecete.FirmaId, Firma.Adi, LabRecete.TalepTarihi,
               LabRecete.LabReceteGrupId, LabReceteGrup.Adi, LabRecete.LabRenkGrupId, LabRenkGrup.Adi,  LabRecete.RenkDerinligiId, RenkDerinligi.Adi,
               LabRecete.ReceteTuruId, ReceteTuru.Adi, LabRecete.Fason, LabRecete.MusteriRenkKodu, LabReceteCalisma.HamId, Ham.Adi,
               LabRecete.Aciklama, LabRecete.Aciklama2, LabRecete.EskiLabReceteKodu, LabRecete.PantoneKodu,
               LabReceteCalisma.CalismaTarihi, LabReceteCalisma.OrjinalDegisimTarihi, LabReceteCalisma.TerminTarihi, LabReceteCalisma.BitisTarihi, LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave,
               LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, LabReceteAtis.AtisNoStr, LabReceteAtis.AtisTarihi, LabReceteAtis.GidisTarihi, LabReceteAtis.KartelaTarihi,
               LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi, LabReceteAtis.Okey, LabReceteAtis.Kritik, LabReceteCalisma.Notu,
               LabReceteAtis.BoyaYuzdesi, LabReceteAtis.BoyaMaliyeti, LabReceteAtis.KimyasalMaliyeti, LabRecete.DovizId,
               LabReceteAtis.EtiketYazdirildi, LabReceteAtis.EtiketYazdirmaTarihi, ReceteOnIslemTipi.Adi,
               LabReceteCalisma.LabProgramNo, LabReceteCalisma.LabProgramNo2, LabReceteAtis.AcilPipetlenecek, LabReceteAtis.PipetlemeyeGonderildi,
               LabReceteAtis.PipetlemeTarihi, LabRecete.KaliteID, LabReceteCalisma.BirimFiyat, LabReceteCalisma.DovizID, LabReceteTalep.ReceteTalepGrup1,
               ReceteTalepGrup.Adi, LabReceteCalisma.FiyatFarki, LabGrup.LabGrupId, LabGrup.Adi from LabReceteCalisma
    left join LabRecete On LabRecete.LabReceteId = LabReceteCalisma.LabReceteId
    left join Firma On Firma.FirmaId = LabRecete.FirmaId
    left join LabReceteGrup On LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
    left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = LabRecete.LabRenkGrupId
    left join ReceteTuru On ReceteTuru.ReceteTuruId = LabRecete.ReceteTuruId
    left join RenkDerinligi On RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
    left join Ham On Ham.HamId = LabReceteCalisma.HamId
    left join LabReceteAtis On LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
    left join ReceteOnIslemTipi on ReceteOnIslemTipi.ReceteOnIslemTipiID = LabReceteCalisma.ReceteOnIslemTipiId
    left join LabReceteTalep on LabReceteTalep.LabReceteTalepKayitId = LabReceteCalisma.LabReceteTalepKayitId
    left join ReceteTalepGrup on ReceteTalepGrup.ReceteTalepGrupId = LabReceteTalep.ReceteTalepGrup1
    left join LabGrup On LabGrup.LabGrupId = LabRecete.LabGrupId
    where LabReceteCalisma.OrjinalDegisimTarihi >= :ABaslangicTarihi and LabReceteCalisma.OrjinalDegisimTarihi <= :ABitisTarihi and
          (LabRecete.FirmaId = :AFirmaId or :AFirmaId is null) and
          (LabRecete.LabRenkGrupId = :ALabRenkGrupId or :ALabRenkGrupId is null) and
          (LabRecete.LabReceteGrupId = :ALabReceteGrupId or :ALabReceteGrupId is null) and
          (LabRecete.ReceteTuruId = :AReceteTuruId or :AReceteTuruId is null) and
          (LabRecete.RenkDerinligiId = :ARenkDerinligiId or :ARenkDerinligiId is null) and
          (LabRecete.LabReceteKodu = :AReceteKodu or :AReceteKodu is null) and
          (LabRecete.Adi like '%' || :AReceteAdi || '%' or :AReceteAdi is null) and
          (LabReceteCalisma.LabProgramNo >= :ABaslangicLabProgramNo or :ABaslangicLabProgramNo is null) and
          (LabReceteCalisma.LabProgramNo <= :ABitisLabProgramNo or :ABitisLabProgramNo is null) and
          (LabReceteCalisma.LabProgramNo2 >= :ABaslangicLabProgramNo2 or :ABaslangicLabProgramNo2 is null) and
          (LabReceteCalisma.LabProgramNo2 <= :ABitisLabProgramNo2 or :ABitisLabProgramNo2 is null) and
          ( (:ACalismaDurumu = 1 and LabReceteCalisma.Rptmi = 1) or
            (:ACalismaDurumu = 2 and LabReceteCalisma.Ilave = 1) or
            (:ACalismaDurumu = 3 and LabReceteCalisma.SokumUzeriBoya = 1) or
            (:ACalismaDurumu = 0 and (LabReceteCalisma.Rptmi = 0 and LabReceteCalisma.Ilave = 0 and LabReceteCalisma.SokumUzeriBoya = 0) ) or
            :ACalismaDurumu is null) and
          (LabReceteAtis.Okey = 1 or :ASadeceOkeyliAtislar = 0) and
          (LabReceteAtis.EtiketYazdirildi = 0 or :ASadeceEtiketYazdirilmayan = 0) and
          (LabReceteAtis.PipetlemeyeGonderildi = 0 or :APipetlemeyeGonderilmeyenler= 0)
    into :LabReceteAtisId, :LabReceteCalismaId,  :LabReceteId, :LabReceteKodu, :LabReceteAdi, :FirmaId, :FirmaAdi, :TalepTarihi, :LabReceteGrupId, :LabReceteGrupAdi, :LabRenkGrupId, :LabRenkGrupAdi,
         :RenkDerinligiId, :RenkDerinligiAdi, :ReceteTuruId, :ReceteTuruAdi, :Fason, :MusteriRenkKodu, :HamId, :HamAdi,
         :Aciklama, :Aciklama2, :EskiLabReceteKodu, :PantoneKodu, :CalismaTarihi, :OrjinalDegisimTarihi, :TerminTarihi, :CalismaBitisTarihi,  :Rptmi, :Ilave, :SokumUzeriBoya, :KritikCalismasi,
         :AtisNoStr, :AtisTarihi, :GidisTarihi, :KartelaTarihi, :DurumuStr, :Tarihi, :Okey, :Kritik, :FCalismaNotu,
         :BoyaYuzdesi, :BoyaMaliyeti, :KimyasalMaliyeti, :DovizId, :EtiketYazdirildi, :EtiketYazdirmaTarihi, :ReceteOnIslemTipiAdi,
         :LabProgramNo, :LabProgramNo2, :AcilPipetlenecek, :PipetlemeyeGonderildi, :PipetlemeTarihi, :KaliteID, :CalismaBirimFiyat, :CalismaDovizID, :ReceteTalepGrup1Id, :ReceteTalepGrup1Adi,
         :FiyatFarki ,:LabGrupId, :LabGrupAdi do
    begin
      if (Fason = 0) then FasonStr = 'Satış';
      else FasonStr = 'Fason';

      if (EtiketYazdirildi = 1) then EtiketYazdirildiStr = 'Evet';
      else EtiketYazdirildiStr = '';

      if (AcilPipetlenecek = 1) then AcilPipetlenecekStr = 'Evet';
      else AcilPipetlenecekStr = '';

      if (PipetlemeyeGonderildi = 1) then PipetlemeyeGonderildiStr = 'Evet';
      else PipetlemeyeGonderildiStr = '';

      BlobSize = char_length(FCalismaNotu);

      if (BlobSize > 1023) then CalismaNotu = Left(FCalismaNotu, 1023);
      else if (BlobSize > 0) then CalismaNotu = Left(FCalismaNotu, BlobSize);
      else CalismaNotu = '';

      CalismaTipiId = 0;

      if (Rptmi = 1) then CalismaTipiId = 1;
      else if (Ilave = 1) then CalismaTipiId = 2;
      else if (SokumUzeriBoya = 1) then CalismaTipiId = 3;
      else if (KritikCalismasi = 1) then CalismaTipiId = 4;

      PartiToplamKilo = Null;
      PartiSayisi = Null;
      TuzMiktari = Null;
      SodaMiktari = null;

      if (AtisNoStr is not null) then
      begin
        ReceteId = LabReceteKodu || '.' || AtisNoStr;


        Select Count(*) from Parti where Parti.ReceteID = :ReceteId into :PartiSayisi;

        if (PartiSayisi > 0) then
        begin
          Select Sum(PartiStok.Kilo) from Parti
            left join PartiStok on PartiStok.PartiKayitID = Parti.PartiKayitId
            where Parti.ReceteID = :ReceteId and PartiStok.PartiStokIptalKayitID is null
            into :PartiToplamKilo;
        end

        if (PartiSayisi = 0) then PartiSayisi = Null;

        if (TuzKimyasalTipiID is not null) then
        begin
          Select Sum(LabReceteUrun.MiktarYuzde) from LabReceteUrun
          left join Urun On Urun.UrunID = LabReceteUrun.UrunId
          where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and LabReceteUrun.UrunID is not null and Urun.KimyasalTipiID = :TuzKimyasalTipiID
          into :TuzMiktari;
        end

        if (SodaKimyasalTipiID is not null) then
        begin
          Select Sum(LabReceteUrun.MiktarYuzde) from LabReceteUrun
          left join Urun On Urun.UrunID = LabReceteUrun.UrunId
          where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and LabReceteUrun.UrunID is not null and Urun.KimyasalTipiID = :SodaKimyasalTipiID
          into :SodaMiktari;
        end
      end

      select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :SistemDovizId, :AtisTarihi) into :DovizKuru;

      BoyaMaliyetiTL = BoyaMaliyeti * DovizKuru;
      KimyasalMaliyetiTL = KimyasalMaliyeti * DovizKuru;

      if (AUrunFiyatGorebilir = 0) then
      begin
        BoyaMaliyeti     = Null;
        KimyasalMaliyeti = Null;
        BoyaMaliyetiTL   = Null;
        KimyasalMaliyetiTL = Null;
      end
      suspend;
    end
  end
