CREATE PROCEDURE HGTASNIFSTOKOLUSTUR(
    AHGTASNIFFISIID INTEGER,
    AHGTASNIFFISIDETAYID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable KesimDepoID            DepoKoduSize;
declare variable TasnifDepoID           DepoKoduSize;
declare variable FStokTabanDovizId      varchar(5);
declare variable FStokId                integer;
declare variable FHareketTipi           integer;
declare variable FFisSeriNo             varchar(2);
declare variable FFisNo                 integer;
declare variable FTarih                 TimeStamp;
declare variable FNow date;
declare variable FKalanMiktar           BoyaMiktarSize;
declare variable FCikilanMiktar         BoyaMiktarSize;
declare variable FCikilacakMiktar       BoyaMiktarSize;
declare variable HGSiparisDetayID integer;
declare variable HGIsEmriKayitID  integer;
declare variable HGIsEmriPastalID integer;
declare variable HGSiparisKayitID integer;
declare variable IEUretimPartiNo  Type of Column HGIsEmri.UretimPartiNo;
declare variable UrunId           UrunKoduSize;
declare variable TasnifMiktar     integer;
declare variable DusulmusMiktar   BoyaMiktarSize;
declare variable FUrunBirimID     Type of Column UrunBirim.UrunBirimId;
begin
  FHareketTipi = 6;  /* Uretime Cikis */
  Select StokId, FisTarihi From Stok Where HGTasnifFisiId = :AHGTasnifFisiId and HareketTipi = :FHareketTipi into :FStokId, :FTarih;

  if (FStokId is Null) then
  begin
    FNow = current_TimeStamp(0);
    FTarih =  FNow;

    select StokTabanDovizId from Sistem where SistemId = 1 into :FStokTabanDovizId;
    Select FisSeriNo from Kullanici where KullaniciId = :AKullaniciID into :FFisSeriNo;

    if (FFisSeriNo is null) then FFisSeriNo = '';

    select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;
    FStokId = gen_id(Stok_Id_Gen, 1);

    insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, HGTasnifFisiId, IslemKaynagi, KullaniciID, InsertKullaniciID)
              values(:FStokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :FTarih, :FStokTabanDovizId, 1, 1, :AHGTasnifFisiId, 1, :AKullaniciID, :AKullaniciID);
  end

  Select HGIsEmri.HGSiparisKayitID, HGIsEmri.UretimPartiNo, HGTasnifFisiDetay.DepoId, HGTasnifFisiDetay.HGSiparisDetayID, HGTasnifFisiDetay.HGIsEmriKayitID,
         HGTasnifFisiDetay.HGIsEmriPastalID, HGTasnifFisiDetay.UrunId, HGTasnifFisiDetay.TasnifMiktar, UrunBirimID from HGTasnifFisiDetay
  Left Join HGIsEmri on HGIsEmri.HGIsEmriKayitId = HGTasnifFisiDetay.HGIsEmriKayitId
  left join UrunBirim on UrunBirim.UrunId = HGTasnifFisiDetay.UrunId and UrunBirim.DefaultBirim = 1
    where HGTasnifFisiDetay.HGTasnifFisiDetayId = :AHGTasnifFisiDetayID
    into :HGSiparisKayitID, :IEUretimPartiNo, :TasnifDepoID, :HGSiparisDetayID, :HGIsEmriKayitID, :HGIsEmriPastalID,  :UrunId, :TasnifMiktar, :FUrunBirimID;

  Select Sum(Miktar) from StokDetay Where StokDetay.HGTasnifFisiDetayId = :AHGTasnifFisiDetayId and HareketTipi = :FHareketTipi into :DusulmusMiktar;

  if (DusulmusMiktar <> TasnifMiktar) then Delete From StokDetay where StokDetay.HGTasnifFisiDetayId = :AHGTasnifFisiDetayId;

  Select First 1 ModelUretimIslemTipi.DefaultDepoId from ModelUretimIslemTipi where ModelUretimIslemTipi.KesimIslemi = 1 into :KesimDepoId;

  FCikilanMiktar = 0;
  FCikilacakMiktar = 0;

  For Select Sum(Miktar * GirisCikis) from StokDetay
          Left Join Depo on Depo.DepoID = StokDetay.DepoID
        where StokDetay.UrunID = :UrunID and StokDetay.DepoID = :KesimDepoId and StokDetay.HGIsEmriPastalID = :HGIsEmriPastalID and HGYariMamul = 1
  into :FKalanMiktar do
  begin
    if (FKalanMiktar < TasnifMiktar - FCikilanMiktar) then
      FCikilacakMiktar = FKalanMiktar;
    else
      FCikilacakMiktar = TasnifMiktar - FCikilanMiktar;

    insert into StokDetay(Tarih, HareketTipi, HGSiparisKayitID, HGSiparisDetayID, HGIsEmriKayitID, HGIsEmriPastalID, StokId, DepoId, UrunId, Miktar, UrunBirimID, Otomasyon, PastalNo, HGTasnifFisiDetayId, KullaniciID, InsertKullaniciID)
                 values(:FTarih, :FHareketTipi, :HGSiparisKayitID, :HGSiparisDetayID, :HGIsEmriKayitID, :HGIsEmriPastalID, :FStokId, :KesimDepoId, :UrunId, :FCikilacakMiktar, :FUrunBirimID, 1, :IEUretimPartiNo, :AHGTasnifFisiDetayId, :AKullaniciID, :AKullaniciID);


    FCikilanMiktar = FCikilanMiktar + FCikilacakMiktar;
    if (FCikilanMiktar = TasnifMiktar) then break;
  end


  /***************Üretim Girişi Yapalım***************/

  FStokId    = Null;
  FFisSeriNo = Null;
  FFisNo     = Null;
  FHareketTipi = 7;  /* Uretim */
  Select StokId, FisTarihi From Stok Where HGTasnifFisiId = :AHGTasnifFisiId and HareketTipi = :FHareketTipi into :FStokId, :FTarih;

  if (FStokId is Null) then
  begin
    FNow = current_TimeStamp(0);
    FTarih =  FNow;

    Select UretimFisSeriNo from Kullanici where KullaniciId = :AKullaniciID into :FFisSeriNo;

    if (FFisSeriNo is null) then FFisSeriNo = '';

    select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;
    FStokId = gen_id(Stok_Id_Gen, 1);

    insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, HGTasnifFisiId, IslemKaynagi, KullaniciID, InsertKullaniciID)
              values(:FStokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :FTarih, :FStokTabanDovizId, 1, 1, :AHGTasnifFisiId, 1, :AKullaniciID, :AKullaniciID);
  end

  Insert Into StokDetay(StokID, HareketTipi, HGSiparisKayitID, HGSiparisDetayID, HGIsEmriKayitID, HGIsEmriPastalID, DepoId,  UrunID, UrunBirimID, Miktar, PastalNo, HGTasnifFisiDetayId, KullaniciID, InsertKullaniciID, Otomasyon) values
                      (:FStokId, :FHareketTipi, :HGSiparisKayitID, :HGSiparisDetayID, :HGIsEmriKayitID, :HGIsEmriPastalID, :TasnifDepoId, :UrunId, :FUrunBirimID, :TasnifMiktar, :IEUretimPartiNo, :AHGTasnifFisiDetayId, :AKullaniciID, :AKullaniciID, 1);

end
