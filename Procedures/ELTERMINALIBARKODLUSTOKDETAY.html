CREATE PROCEDURE ELTERMINALIBARKODLUSTOKDETAY(
    AEVRAKKAYITID INTEGER,
    AHAREKETTIPI INTEGER,
    ADEPOID VARCHAR(10),
    ATRANSFERDEPOID VARCHAR(10),
    AURUNID VARCHAR(20),
    AURUNBIRIMID INTEGER,
    AVERILENSIPARISKAYITID INTEGER,
    AVERILENSIPARISDETAYID INTEGER,
    AMIKTAR DOUBLE PRECISION,
    AKDV DOUBLE PRECISION,
    ABIRIMFIYAT DOUBLE PRECISION,
    AKULLANICIID INTEGER)
RETURNS (
    DETAYKAYITID INTEGER)
AS
declare variable shtAlis         integer;
declare variable shtSatis        integer;
declare variable shtDepoTransfer integer;
begin
  select shtAlis, shtSatis, ShtDepoTransfer from StokHareketTipiGetir into :shtAlis, :shtSatis, :ShtDepoTransfer;

  DetayKayitID = gen_id(StokDetay_Id_Gen, 1);

  if (AHareketTipi = shtAlis or AHareketTipi = shtSatis) then
  begin
    Insert Into StokDetay(FaturaID, HareketTipi, DepoId,  UrunID, UrunBirimID, VerilenSiparisKayitID, VerilenSiparisDetayID, Miktar, Kdv, BirimFiyat, InsertKullaniciID) values
                        (:AEvrakKayitID, :AHareketTipi, :ADepoID, :AUrunID, :AUrunBirimID, :AVerilenSiparisKayitID, :AVerilenSiparisDetayID, :AMiktar, :AKdv, :ABirimFiyat, :AKullaniciID);

  end
  else if (AHareketTipi = ShtDepoTransfer) then
  begin
    Insert Into StokDetay(StokID, HareketTipi, DepoId, TransferDepoID, UrunID, UrunBirimID, Miktar, InsertKullaniciID) values
                        (:AEvrakKayitID, :AHareketTipi, :ATransferDepoID, :ADepoId, :AUrunID, :AUrunBirimID, :AMiktar, :AKullaniciID);

  end
  else DetayKayitID = -1;

  suspend;
end
