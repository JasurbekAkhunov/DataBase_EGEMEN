CREATE PROCEDURE SILINENBOYAIRSALIYERAPORU3(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    BOYAIRSALIYEDETAYID INTEGER,
    SILINMETARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    KULLANICIADI ADI40,
    PARTISTOKTARTIID INTEGER,
    KAPADEDI INTEGER,
    KILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    DEPOID DEPOKODUSIZE,
    DARA BOYAMIKTARSIZE)
AS
begin
  for select LogDetay.OldValue, LogMaster.Tarih, LogMaster.KullaniciId, Kullanici.Adi, LogMaster.KeyValue,
      PartiStokTarti.KapAdedi, PartiStokTarti.Kilo, PartiStokTarti.Metre, PartiStokTarti.DepoId, PartiStokTarti.Dara
      from LogMaster
      left join LogDetay on LogDetay.LogMasterId = LogMaster.LogMasterId and LogDetay.FieldName = 'BOYAIRSALIYEDETAYID'
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = LogMaster.KeyValue
      left join Kullanici on Kullanici.KullaniciId = LogMaster.KullaniciId
      where LogMaster.Tip = 'U' and /*silinen tartiler*/
      LogMaster.TabloAdi = 'PARTISTOKTARTI' and
      LogMaster.Tarih >= :ABaslangicTarihi and LogMaster.Tarih <= :ABitisTarihi+1 and
      LogDetay.OldValue is not null and LogDetay.NewValue is null
      into :BoyaIrsaliyeDetayId, :SilinmeTarihi, :KullaniciID, :KullaniciAdi, :PartiStokTartiID,
      :KapAdedi, :Kilo, :Metre, :DepoId, :Dara do
  begin
    suspend;
  end
end
