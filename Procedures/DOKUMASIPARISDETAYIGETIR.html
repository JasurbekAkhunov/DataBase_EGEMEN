CREATE PROCEDURE DOKUMASIPARISDETAYIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    SINIF VARCHAR(1),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    FLAMKODU VARCHAR(20),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    KALITEID VARCHAR(20),
    KALITEADI ADI255,
    HAMMIKTAR DOUBLE PRECISION,
    KILO DOUBLE PRECISION,
    SATINALINACAKMIKTAR DOUBLE PRECISION,
    STOKTANKULLANILACAKMIKTAR DOUBLE PRECISION,
    SATISFIYATI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    TESLIMTARIHI TIMESTAMP,
    REVIZETESLIMTARIHI TIMESTAMP,
    VADEGUN INTEGER,
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(40),
    KAPALI INTEGER,
    KAPALISTR VARCHAR(4),
    KAPATMATARIHI TIMESTAMP,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    SEVKYERI VARCHAR(50),
    TOPLAMDOKUNAN DOUBLE PRECISION,
    TOPLAMSAKATDOKUNAN DOUBLE PRECISION,
    TOPLAMSATIS DOUBLE PRECISION,
    TOPLAMBOYALISATIS DOUBLE PRECISION,
    TOPLAMSATISIADE DOUBLE PRECISION,
    MUSTERITALIMATI BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TAHMINIFIREORANI DOUBLE PRECISION,
    BOYAYAGIDENMIKTAR DOUBLE PRECISION,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    URETIMNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KALINLIKADI VARCHAR(50),
    URETICIADI VARCHAR(40),
    IPLIKCINSIADI VARCHAR(80),
    KUMASCINSIADI VARCHAR(40),
    LOT VARCHAR(30),
    BEYAZLIK INTEGER,
    BEYAZLIKSTR VARCHAR(8),
    KUMASTEMINSIPARISIDSTR VARCHAR(30),
    RESIMDOSYAADI1 VARCHAR(255),
    RESIMDOSYAYAZDIR1 INTEGER,
    IPLIKTALEPMIKTARI DOUBLE PRECISION,
    IPLIKTALEPIDSTR VARCHAR(100),
    DOKUMACILAR VARCHAR(255),
    DEGISIMLOG BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PLANLANABILIR INTEGER,
    PLANLANABILIRSTR VARCHAR(5),
    HAMMALIYET DOUBLE PRECISION,
    SISTEMDOVIZID VARCHAR(5),
    IPLIKLOTID INTEGER,
    IPLIKLOTADI VARCHAR(30),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    PLANLANANMIKTAR DOUBLE PRECISION,
    TAHARRAPORNOTU MEMO)
AS
declare variable KumasTeminBoyaSiparisYil integer;
declare variable KumasTeminBoyaSiparisId integer;
declare variable uitBoya integer;
declare variable FIplikTalepYil                integer;
declare variable FIplikTalepId                 integer;
declare variable FIplikMaliyeti                Double Precision;
declare variable FLycraMaliyeti                Double Precision;
declare variable DovizKuru                     Double Precision;
declare variable SiparisTarihi                  TimeStamp;
declare variable hstLycra        integer;
declare variable FDokumaciAdi varchar(50);
begin
  Select hstLycra from HamTipiGetir into :hstLycra;

  select uitBoya from UretimIslemTipiGetir into :uitBoya;
  Select SistemDovizID from Sistem where SistemID = 1 into :SistemDovizID;

  Select SiparisTarihi from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitID into :SiparisTarihi;

  For Select OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.HamId, OrmeSiparisDetay.Sinif,
     Ham.Adi, Ham.Adi2, OrmeSiparisDetay.FlamKodu, Ham.HamTipiId, HamTipi.Adi, OrmeSiparisDetay.RenkId, Renk.Adi, OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.SatinAlinacakMiktar,
     OrmeSiparisDetay.StoktanKullanilacakMiktar, OrmeSiparisDetay.SatisFiyati, OrmeSiparisDetay.DovizId, OrmeSiparisDetay.Teslimtarihi, OrmeSiparisDetay.RevizeTeslimTarihi,
     OrmeSiparisDetay.VadeGun, OrmeSiparisDetay.OdemeSekliId, OdemeSekli.Adi, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo,
     OrmeSiparisDetay.TahminiFireOrani, OrmeSiparisDetay.Kapali, OrmeSiparisDetay.KapatmaTarihi,
     OrmeSiparisDetay.ToplamDokunan, OrmeSiparisDetay.ToplamSakatDokunan, OrmeSiparisDetay.ToplamSatis, OrmeSiparisDetay.ToplamBoyaliSatis, OrmeSiparisDetay.ToplamSatisIade,
     OrmeSiparisDetay.MusteriTalimati, OrmeSiparisDetay.Notu, OrmeSiparisDetay.UretimNotu, IplikCinsi.Adi, Uretici.Adi, Kalinlik.Adi, KumasCinsi.Adi,
     OrmeSiparisDetay.Beyazlik, OrmeSiparisDetay.KaliteId, Kalite.kaliteAdi,
     KumasTeminSiparis.BoyaSiparisYil, KumasTeminSiparis.BoyaSiparisId, OrmeSiparisDetay.ResimDosyaAdi1, OrmeSiparisDetay.ResimDosyayazdir1,
     OrmeSiparisDetay.Planlanabilir, IK.Adi, OrmeSiparisDetay.InsertTarihi, SK.Adi, OrmeSiparisDetay.DegisimTarihi, Ham.Adi3, Ham.Adi4,
     OrmeSiparisDetay.SevkYeri, OrmeSiparisDetay.DegisimLog, OrmeSiparisDetay.OnayDurumu, OrmeSiparisDetay.OnayTarihi, OrmeSiparisDetay.OnaylayanKullaniciId,
     Ok.Adi, OrmeSiparisDetay.IplikLotID, IplikLot.Lot, OrmeSiparisDetay.PlanlananMiktar, OrmeSiparisDetay.Kilo, Ham.TaharRaporNotu from OrmeSiparisDetay
  Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
  Left Join Renk ON Renk.RenkId = OrmeSiparisDetay.RenkId
  Left Join HamTipi ON HamTipi.HamTipiId = Ham.HamTipiId
  left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
  left join Uretici on Uretici.UreticiId  = Ham.UreticiId
  Left Join KumasCinsi ON KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
  left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
  left join BoyaSiparis KumasTeminSiparis on KumasTeminSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.KTBoyaSiparisKayitId
  Left Join Kullanici IK  ON IK.KullaniciId = OrmeSiparisDetay.InsertKullaniciID
  Left Join Kullanici SK  ON SK.KullaniciId = OrmeSiparisDetay.KullaniciID
  Left Join Kullanici OK  ON OK.KullaniciId = OrmeSiparisDetay.OnaylayanKullaniciId
  left join OdemeSekli on OdemeSekli.OdemeSekliId  = OrmeSiparisDetay.OdemeSekliId
  left join Kalite on Kalite.KaliteId  = OrmeSiparisDetay.KaliteId
  Left Join IplikLot On IplikLot.IplikLotID = OrmeSiparisDetay.IplikLotID
  where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  order by OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.SiparisSiraNo
  into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :HamId, :Sinif, :HamAdi, :HamAdi2, :FlamKodu,  :HamTipiId, :HamTipiAdi,
       :RenkId, :RenkAdi, :hamMiktar, :SatinAlinacakMiktar, :StoktanKullanilacakMiktar,
       :SatisFiyati, :DovizId, :Teslimtarihi, :RevizeTeslimTarihi,
       :VadeGun, :OdemeSekliId, :OdemeSekliAdi, :ArtikelNo, :MusteriSiparisNo, :ReferansSiparisNo, :TahminiFireOrani,
       :Kapali, :KapatmaTarihi, :ToplamDokunan, :ToplamSakatDokunan,  :ToplamSatis, :ToplamBoyaliSatis, :ToplamSatisIade,
       :MusteriTalimati, :Notu, :UretimNotu,  :IplikCinsiAdi, :UreticiAdi, :KalinlikAdi, :KumasCinsiAdi,
       :Beyazlik, :KaliteId, :KaliteAdi, :KumasTeminBoyaSiparisYil, :KumasTeminBoyaSiparisId,
       :ResimDosyaAdi1, :ResimDosyayazdir1, :Planlanabilir,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :HamAdi3, :HamAdi4,
       :SevkYeri, :DegisimLog, :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :IplikLotID, :IplikLotAdi,
       :PlanlananMiktar, :Kilo, :TaharRaporNotu do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    KapaliStr = '';
    if (Kapali = 1) then KapaliStr = 'Evet';

    BeyazlikStr = '';
    if (Beyazlik = 1) then BeyazlikStr = 'Evet';

    PlanlanabilirStr = '';
    if (Planlanabilir = 1) then PlanlanabilirStr = 'Evet';

    if (KumasTeminBoyaSiparisId is not null) then
      KumasTeminSiparisIdStr = '(' || EgemenintegerToStr(KumasTeminBoyaSiparisYil) || ') ' || EgemenintegerToStr(KumasTeminBoyaSiparisId);
    else
      KumasTeminSiparisIdStr = '';


    BoyayaGidenMiktar = 0;

    Select Sum(StokHareket.GirisCikis * StokHareket.Miktar * -1) from StokHareket
      where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and
            StokHareket.UretimIslemTipiId = :uitBoya and (StokHareket.GirisCikis = -1 or (StokHareket.GirisCikis = 1 and StokHareket.HamIade = 1) )
      into :BoyayaGidenMiktar;



    IplikTalepMiktari = 0;

    Select Sum(Miktar) from IplikTalepDetay where IplikTalepDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId  into :IplikTalepMiktari;

    IplikTalepIdStr = '';

    for Select IpliKTalep.IplikTalepYil, IpliKTalep.IplikTalepId from IplikTalepDetay
      left join IpliKTalep on IpliKTalep.IpliKTalepKayitID = IpliKTalepDetay.IpliKTalepKayitId
      where IplikTalepDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId
      group by IpliKTalep.IplikTalepYil, IpliKTalep.IplikTalepId
      into :FIplikTalepYil, FIplikTalepId do
    begin
      if (IplikTalepIdStr <> '') then IplikTalepIdStr = IplikTalepIdStr || cast(', ' as varchar(2) character set UTF8);

      IplikTalepIdStr = IplikTalepIdStr ||  '(' || EgemenIntegerToStr(FIplikTalepYil) || ') ' || EgemenIntegerToStr(FIplikTalepId);
    end

    FIplikMaliyeti = 0;

    Select Sum(BirimMaliyet * (select DovizKuru from EnYakinDovizKuruGetir(BirimMaliyetDovizID, :SistemDovizID, :SiparisTarihi)) * KullanimOrani/100) from OrmeSiparisDetayRecete
      left join Ham On Ham.HamID = OrmeSiparisDetayRecete.HamId
      where OrmeSiparisDetayRecete.OrmeSiparisDetayId = :OrmeSiparisDetayId and Ham.HamTipiID <> :hstLycra
      into :FIplikMaliyeti;

    if (FIplikMaliyeti is null) then FIplikMaliyeti = 0;
    else FIplikMaliyeti = FIplikMaliyeti + FIplikMaliyeti * TahminiFireOrani / 100;

    Select Sum(BirimMaliyet * (select DovizKuru from EnYakinDovizKuruGetir(BirimMaliyetDovizID, :SistemDovizID, :SiparisTarihi)) * KullanimOrani/100) from OrmeSiparisDetayRecete
      left join Ham On Ham.HamID = OrmeSiparisDetayRecete.HamId
      where OrmeSiparisDetayRecete.OrmeSiparisDetayId = :OrmeSiparisDetayId and Ham.HamTipiID = :hstLycra
      into :FLycraMaliyeti;

    if (FLycraMaliyeti is null) then FLycraMaliyeti = 0;
    else FLycraMaliyeti = FLycraMaliyeti + FLycraMaliyeti * TahminiFireOrani / 100;

    if (DovizId <> SistemDovizId) then
      select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :SistemDovizId, :SiparisTarihi) into :DovizKuru;
    else
      DovizKuru = 1;

    HamMaliyet = FIplikMaliyeti + FLycraMaliyeti;

    Dokumacilar = '';
    for select Firma.Adi from DokumaPlanlama
      left join Firma on Firma.FirmaId = DokumaPlanlama.FirmaId
      where DokumaPlanlama.OrmeSiparisDetayId = :OrmeSiparisDetayId
      into :FDokumaciAdi do
    begin
      if (Char_Length(Dokumacilar) < 210) then
      begin
        if (Dokumacilar <> '') then Dokumacilar = Dokumacilar || cast(', ' as varchar(2) character set UTF8);
        Dokumacilar = Dokumacilar || FDokumaciAdi;
      end
    end


    suspend;
  end
end
