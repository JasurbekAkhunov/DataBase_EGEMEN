    PSELEKTRIKPAYITUTARI DOUBLE PRECISION,
    PSDOGALGAZPAYITUTARI DOUBLE PRECISION,
    PSBUHARPAYITUTARI DOUBLE PRECISION,
    PSSUPAYITUTARI DOUBLE PRECISION,
    PSISCILIKPAYITUTARI DOUBLE PRECISION,

AS
Declare Variable OrmeFirmaKodumuz Varchar(20);
Declare Variable SistemDovizid    Varchar(5);
Declare Variable FOrmeRafID       Integer;
Declare Variable FSayac           Integer;
Begin
  Select Sistem.Ormefirmakodumuz, Sistem.Sistemdovizid  From Sistem where Sistem.Sistemid = 1 into :OrmeFirmaKodumuz, :SistemDovizid;

  if (AHazirlik = 1) then
  begin
/* 2 giriş çıkışlar - stok hareket tablosu*/
/* 3 giriş çıkışlar - örme raftransferi*/
    FSayac = 2;
    while (FSayac <= 3) do
    begin
      For Select OrmeStokHareket.Stokhareketid, OrmeStokHareket.Tarih, OrmeStokHareket.FirmaID,
                 OrmeStokHareket.Firmaadi, OrmeStokHareket.OrmeRafID, OrmeStokHareket.YeniOrmeRafID,
                 OrmeStokHareket.FasonStok, OrmeStokHareket.Giriscikis, OrmeStokHareket.HareketTipiID,
                 OrmeStokHareket.HareketTipiAdi, OrmeStokHareket.Uretimislemtipiid, OrmeStokHareket.UretimIslemTipiAdi,
                 OrmeStokHareket.HamIade, OrmeStokHareket.Hamid, OrmeStokHareket.Hamadi, OrmeStokHareket.HamTipiID,
                 OrmeStokHareket.HamTipiAdi, OrmeStokHareket.Lot, OrmeStokHareket.Melanjkodu, OrmeStokHareket.FlamKodu,
                 OrmeStokHareket.RenkID, OrmeStokHareket.RenkAdi, OrmeStokHareket.DesenID,
                 OrmeStokHareket.DesenVaryantID, OrmeStokHareket.MiktarBirim, OrmeStokHareket.BrutMiktar,
                 OrmeStokHareket.Miktar, OrmeStokHareket.Birimfiyat, OrmeStokHareket.Dovizid,
                 OrmeStokHareket.BirimFiyatSistemDoviz, OrmeStokHareket.BirimMaliyet, OrmeStokHareket.TabanDovizKuru1,
                 OrmeStokHareket.TabanDovizKuru2, OrmeStokHareket.InsertTarihi, OrmeStokHareket.DegisimTarihi,
                 OrmeStokHareket.OrmeSiparisDetayID, OrmeStokHareket.SiparisTahminiFireOrani,
                 OrmeStokHareket.Boyasipariskayitid, OrmeStokHareket.Boyasiparisyil, OrmeStokHareket.Boyasiparisid,
                 OrmeStokHareket.Boyairsaliyedetayid, OrmeStokHareket.Partikayitid, OrmeStokHareket.Partisinifid,
                 OrmeStokHareket.Partiyil, OrmeStokHareket.Partiid, OrmeStokHareket.PSHamStokMaliyeti,
                 OrmeStokHareket.Psgugpayitutari, OrmeStokHareket.PsElektrikPayiTutari,
                 OrmeStokHareket.Psdogalgazpayitutari, OrmeStokHareket.PsbuharPayiTutari,
                 OrmeStokHareket.Pssupayitutari, OrmeStokHareket.Psiscilikpayitutari, OrmeStokHareket.Psboyamaliyeti,
                 OrmeStokHareket.Pskimyasalmaliyeti, OrmeStokHareket.Psboyakimyasalmaliyeti,
                 OrmeStokHareket.Pstoplammaliyet, OrmeStokHareket.Pskgmaliyeti, OrmeStokHareket.Ormeplanlamaid,
                 OrmeStokHareket.Uretimpartino, OrmeStokHareket.Ormefisid, OrmeStokHareket.Ormefisno,
                 OrmeStokHareket.Ormefistarih, OrmeStokHareket.Ormeirsaliyeid, OrmeStokHareket.Ormeirsaliyeno,
                 OrmeStokHareket.Ormeirsaliyetarih, OrmeStokHareket.Ormeraftransferfisid,
                 OrmeStokHareket.Ormeraftransferfisidetayid, OrmeStokHareket.Ormeraftransferfisyil,
                 OrmeStokHareket.Ormeraftransferfisno, OrmeStokHareket.Ormeraftransferfistarih,
                 OrmeStokHareket.Ormestokkaydirmaid, OrmeStokHareket.Ormestokkaydirmafisyil,
                 OrmeStokHareket.Ormestokkaydirmafisno, OrmeStokHareket.Ormestokkaydirmafistarihi,
                 OrmeStokHareket.Ormestokkaydirmaormerafid
        From Ormestokhareket(:FSayac, :Ahamid, :Abaslangictarihi, :Abitistarihi, :Aormerafid)
        Order By OrmeStokHareket.Tarih Asc, OrmeStokHareket.Giriscikis Desc
        into :Stokhareketid, :Tarih, :Firmaid, :Firmaadi, :Ormerafid, :Yeniormerafid, :Fasonstok, :Giriscikis,
             :Harekettipiid, :Harekettipiadi, :Uretimislemtipiid, :Uretimislemtipiadi, :Hamiade, :Hamid, :Hamadi,
             :Hamtipiid, :Hamtipiadi, :Lot, :Melanjkodu, :Flamkodu, :Renkid, Renkadi, :Desenid, :Desenvaryantid,
             :miktarbirim, :Brutmiktar, :Miktar, :Birimfiyat, :Dovizid, :Birimfiyatsistemdoviz, :Birimmaliyet,
             :Tabandovizkuru1, :Tabandovizkuru2, :Inserttarihi, :Degisimtarihi, :Ormesiparisdetayid,
             :Siparistahminifireorani, :Boyasipariskayitid, :Boyasiparisyil, :Boyasiparisid, :Boyairsaliyedetayid,
             :Partikayitid, :Partisinifid, :Partiyil, :Partiid, :Pshamstokmaliyeti, :PSGUGPayiTutari,
             :PSElektrikPayiTutari, :Psdogalgazpayitutari, :PSBuharPayiTutari, :PSSuPayiTutari, :PSIscilikPayiTutari,
             :PSBoyaMaliyeti, :PSKimyasalMaliyeti, :PSBoyaKimyasalMaliyeti, :PSToplamMaliyet, :PSKgMaliyeti,
             :Ormeplanlamaid, :Uretimpartino, :Ormefisid, :Ormefisno, :Ormefistarih, :Ormeirsaliyeid, :Ormeirsaliyeno,
             :Ormeirsaliyetarih, :Ormeraftransferfisid, :Ormeraftransferfisidetayid, :Ormeraftransferfisyil,
             :Ormeraftransferfisno, :Ormeraftransferfistarih, :Ormestokkaydirmaid, :Ormestokkaydirmafisyil,
             :Ormestokkaydirmafisno, :Ormestokkaydirmafistarihi, :Ormestokkaydirmaormerafid
        do Suspend;
      FSayac = FSayac + 1;
    end
  end
  else
  if (AHazirlik = 2) then
  begin
    For Select Stokhareket.Stokhareketid, Stokhareket.Tarih, Stokhareket.Firmaid, Firma.Adi,
               Coalesce(Girisormeraf.Ormerafid, Cikisormeraf.Ormerafid),
               Coalesce(Girisormeraf.Fasonstok, Cikisormeraf.Fasonstok), Stokhareket.Giriscikis,
               Stokhareket.Harekettipi, Ormestokharekettipi.Adi, Stokhareket.Uretimislemtipiid, Uretimislemtipi.Adi,
               Stokhareket.Hamiade, Stokhareket.Hamid, Ham.Adi, Ham.Hamtipiid, Hamtipi.Adi, Stokhareket.Lot,
               Stokhareket.Melanjkodu, Stokhareket.Flamkodu, Stokhareket.Renkid, Renk.Adi, Stokhareket.Desenid,
               Stokhareket.Desenvaryantid, Stokhareket.Miktarbirim, Stokhareket.Brutmiktar, Stokhareket.Miktar,
               Stokhareket.Birimfiyat, Stokhareket.Dovizid, Stokhareket.Birimmaliyet,
               coalesce(Ormefatura.Tabandovizkuru1, Stokhareket.Tabandovizkuru1),
               coalesce(Ormefatura.Tabandovizkuru2, Stokhareket.Tabandovizkuru2), Stokhareket.Inserttarihi,
               Stokhareket.Degisimtarihi, StokHareket.Ormesiparisdetayid, Ormesiparisdetay.Tahminifireorani,
               Boyasiparis.Boyasipariskayitid, Boyasiparis.Boyasiparisyil, Boyasiparis.Boyasiparisid,
               Stokhareket.Boyairsaliyedetayid, Boyairsaliyedetay.Partikayitid, Boyairsaliyedetay.Partisinifid,
               Parti.Partiyil, Parti.Partiid, PartiSinifMaliyetKumulatif.HamStokMaliyeti, Stokhareket.Ormeplanlamaid,
               Ormeplanlama.Uretimpartino, Ormefis.Ormefisid, Ormefis.Fisno, Ormefis.Fistarihi,
               Ormeirsaliye.Ormeirsaliyeid, Ormeirsaliye.Irsaliyeno, Ormeirsaliye.Irsaliyetarihi,
               Ormefis.Ormestokkaydirmaid, Ormestokkaydirma.Fisyil, Ormestokkaydirma.Fisno, Ormestokkaydirma.Fistarihi,
               Ormestokkaydirma.Ormerafid
      From Stokhareket
      Left Join Ormefis On Ormefis.Ormefisid = Stokhareket.Ormefisid
      Left Join Ormeirsaliye On Ormeirsaliye.Ormeirsaliyeid = Stokhareket.Ormeirsaliyeid
      Left Join Ormefatura On Ormefatura.Ormefaturaid = Stokhareket.Ormefaturaid
      Left Join Ormeplanlama on Ormeplanlama.Ormeplanlamaid = Stokhareket.Ormeplanlamaid
      Left Join Firma on Firma.Firmaid = Stokhareket.Firmaid
      Left Join Ormerafdetay Girisormerafdetay on Girisormerafdetay.Stokhareketid = Stokhareket.Stokhareketid
      Left Join ormeraf Girisormeraf On Girisormeraf.Ormerafid = Girisormerafdetay.Ormerafid
      Left Join ormeraf Cikisormeraf on Cikisormeraf.Ormerafid = Stokhareket.Ormerafid
      Left Join Ormestokharekettipi On Ormestokharekettipi.Ormestokharekettipiid = Stokhareket.Harekettipi
      Left Join Ham On Ham.Hamid = Stokhareket.Hamid
      Left Join Hamtipi On Hamtipi.Hamtipiid = Ham.Hamtipiid
      Left Join Renk On Renk.Renkid = Stokhareket.Renkid
      Left Join Ormesiparisdetay On Ormesiparisdetay.Ormesiparisdetayid = Stokhareket.Ormesiparisdetayid
      Left Join Boyasiparis On Boyasiparis.Boyasipariskayitid = Stokhareket.Boyasipariskayitid
      Left Join Boyairsaliyedetay On Boyairsaliyedetay.Boyairsaliyedetayid = Stokhareket.Boyairsaliyedetayid
      Left Join Partistok On Partistok.Partisinifid = Boyairsaliyedetay.Partisinifid
      Left Join Parti On Parti.Partikayitid = Partistok.Partikayitid
      Left Join Uretimislemtipi On Uretimislemtipi.Uretimislemtipiid = Stokhareket.Uretimislemtipiid
      Left Join PartiSinifMaliyetKumulatif on PartiSinifMaliyetKumulatif.PartiSinifId = PartiStok.Partisinifid
      Left Join Ormestokkaydirma on Ormestokkaydirma.Ormestokkaydirmaid = Ormefis.Ormestokkaydirmaid
      Where (Stokhareket.Hamid = :Ahamid Or :Ahamid Is Null) And
            Stokhareket.Tarih >= :Abaslangictarihi And
            Stokhareket.Tarih <= :Abitistarihi and
            (Girisormeraf.Fasonstok = 0 or Cikisormeraf.Fasonstok = 0) and
            (Girisormeraf.Ormerafid = :Aormerafid or Cikisormeraf.Ormerafid = :Aormerafid or :Aormerafid Is Null)
      Into :Stokhareketid, :Tarih, :Firmaid, :Firmaadi, :Ormerafid, :Fasonstok, :Giriscikis, :Harekettipiid,
           :Harekettipiadi, :Uretimislemtipiid, :Uretimislemtipiadi, :Hamiade, :Hamid, :Hamadi, :Hamtipiid, :Hamtipiadi,
           :Lot, :Melanjkodu, :Flamkodu, :Renkid, :Renkadi, :Desenid, :Desenvaryantid, :Miktarbirim, :Brutmiktar,
           :Miktar, :Birimfiyat, :Dovizid, :Birimmaliyet, :Tabandovizkuru1, :Tabandovizkuru2, :Inserttarihi,
           :Degisimtarihi, :Ormesiparisdetayid, :SiparisTahminifireorani, :Boyasipariskayitid, :Boyasiparisyil,
           :Boyasiparisid, :Boyairsaliyedetayid, :Partikayitid, :Partisinifid, :Partiyil, :Partiid, :Pshamstokmaliyeti,
           :Ormeplanlamaid, :Uretimpartino, :Ormefisid, :Ormefisno, :Ormefistarih, :Ormeirsaliyeid, :Ormeirsaliyeno,
           :Ormeirsaliyetarih, :Ormestokkaydirmaid, :Ormestokkaydirmafisyil, :Ormestokkaydirmafisno,
           :Ormestokkaydirmafistarihi, :Ormestokkaydirmaormerafid do
    begin
      if (Birimmaliyet Is Null) Then
        Birimmaliyet = 0;
      Birimfiyatsistemdoviz = Birimfiyat * Tabandovizkuru2 / Tabandovizkuru1;

      PSGUGPayiTutari = 0;
      PSElektrikPayiTutari = 0;
      PSDogalgazPayiTutari = 0;
      PSBuharPayiTutari = 0;
      PSSuPayiTutari = 0;
      PSIscilikPayiTutari = 0;
      PSBoyaMaliyeti = 0;
      PSKimyasalMaliyeti = 0;
      PSBoyaKimyasalMaliyeti = 0;
      PSToplamMaliyet = 0;
      Pskgmaliyeti = 0;

      If (PartiSinifId Is Not Null) Then
      Begin
        Select Coalesce(Sum(PartiAsamaStokDetay.GenelUretimGiderPayiTutari), 0),
               Coalesce(sum(PartiAsamaStokDetay.ElektrikPayiTutari), 0),
               Coalesce(Sum(PartiAsamaStokDetay.DogalgazPayiTutari), 0),
               Coalesce(Sum(PartiAsamaStokDetay.BuharPayiTutari), 0),
               Coalesce(Sum(PartiAsamaStokDetay.SuPayiTutari), 0),
               Coalesce(Sum(PartiAsamaStokDetay.IscilikPayiTutari), 0),
               Coalesce(Sum(PartiAsamaStokDetay.BoyaMaliyeti), 0),
               Coalesce(Sum(PartiAsamaStokDetay.KimyasalMaliyeti), 0),
               Coalesce(Sum(PartiAsamaStokDetay.BoyaMaliyeti + PartiAsamaStokDetay.KimyasalMaliyeti), 0)
        From PartiAsamaStokDetay
        Where PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
        Into :PSGUGPayiTutari, :PSElektrikPayiTutari, :PSDogalgazPayiTutari, :PSBuharPayiTutari, :PSSuPayiTutari,
             :PSIscilikPayiTutari, :PSBoyaMaliyeti, :PSKimyasalMaliyeti, :PSBoyaKimyasalMaliyeti;

        PSToplamMaliyet = Pshamstokmaliyeti + PSGUGPayiTutari + PSElektrikPayiTutari + PSDogalgazPayiTutari + PSBuharPayiTutari + PSSuPayiTutari + PSIscilikPayiTutari + PSBoyaKimyasalMaliyeti;
        If ((Pstoplammaliyet > 0) And
            (Brutmiktar > 0)) Then
          Pskgmaliyeti = PSToplamMaliyet / Brutmiktar;
      End

      Suspend;
    end
  end
  else
  if (Ahazirlik = 3) Then
  begin
/* rafdan giriş çıkışlar*/

    Hamiade = 0;
    Birimfiyat = 0;
    Dovizid = Sistemdovizid;
    Tabandovizkuru1 = 1;
    Tabandovizkuru2 = 1;
