CREATE PROCEDURE STOKENVANTERRAPORU(
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    ADEPOID VARCHAR(10),
    ASIFIROLANLARDOKULSUNMU INTEGER,
    AKAPALIDURUMU INTEGER,
    ABOYAURUNTIPI INTEGER,
    AKIMYASALURUNTIPI INTEGER,
    ADIGERURUNTIPI INTEGER,
    AYEDEKPARCAURUNTIPI INTEGER,
    ASARFMALZEMESIURUNTIPI INTEGER,
    AHIZMETURUNTIPI INTEGER,
    ASABLONURUNTIPI INTEGER,
    AAKSESUARURUNTIPI INTEGER,
    ATICARIMALURUNTIPI INTEGER,
    AMODELID MODELKODUSIZE,
    AURUNPARCADURUMU INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BIRIMID VARCHAR(2),
    KIMYASALTIPIID INTEGER,
    KIMYASALTIPIADI VARCHAR(20),
    RAF VARCHAR(15),
    URUNBIRIMID INTEGER,
    KDVORANI DOUBLE PRECISION,
    DEVIRMIKTARI DOUBLE PRECISION,
    ALISMIKTARI DOUBLE PRECISION,
    ALISIADESIMIKTARI DOUBLE PRECISION,
    URETIMMIKTARI DOUBLE PRECISION,
    SATISMIKTARI DOUBLE PRECISION,
    SATISIADESIMIKTARI DOUBLE PRECISION,
    URETIMECIKISMIKTARI DOUBLE PRECISION,
    DEPOLARACIKANMIKTAR DOUBLE PRECISION,
    DEPOLARDANGELENMIKTAR DOUBLE PRECISION,
    STOKMIKTARI DOUBLE PRECISION,
    TRANSFERKODU VARCHAR(55),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    KAPALI INTEGER,
    ALISFIYATI DOUBLE PRECISION,
    ALISFIYATIDOVIZID VARCHAR(5),
    SATISFIYATI DOUBLE PRECISION,
    SATISFIYATIDOVIZID VARCHAR(5),
    URETIMECIKISFIYATI DOUBLE PRECISION,
    URETIMECIKISFIYATIDOVIZID VARCHAR(5),
    FASONURETIMECIKISMIKTARI DOUBLE PRECISION,
    FASONURETIMMIKTARI DOUBLE PRECISION,
    FASONURETIMIADEMIKTARI DOUBLE PRECISION,
    TEMINSURESI DOUBLE PRECISION,
    MODELID MODELKODUSIZE,
    MODELADI ADI40)
AS
declare variable Okey           integer;
begin
  for Select Urun.UrunId, Urun.Adi, Urun.Raf, Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, Urun.KdvOrani, UrunBirim.BirimId, Urun.TransferKodu,
             Urun.KimyasalTipiId, KimyasalTipi.Adi, Urun.Kapali,
             UrunFiyat.AlisFiyati,  UrunFiyat.AlisFiyatiDovizId, UrunFiyat.SatisFiyati,  UrunFiyat.SatisFiyatiDovizId, UrunFiyat.UretimeCikisFiyati,  UrunFiyat.UretimeCikisFiyatiDovizId,
             Sum(DepoStok.Devir * UrunBirim.Carpan), Sum(DepoStok.Alis * UrunBirim.Carpan), Sum(DepoStok.AlisIadesi * UrunBirim.Carpan),
             Sum(DepoStok.Satis * UrunBirim.Carpan), Sum(DepoStok.SatisIadesi * UrunBirim.Carpan),
             Sum(DepoStok.UretimeCikis * UrunBirim.Carpan), Sum(DepoStok.DepoyaCikan * UrunBirim.Carpan),
             Sum(DepoStok.DepodanGelen * UrunBirim.Carpan), Sum(DepoStok.Uretim * UrunBirim.Carpan), UrunBirim.UrunBirimId, Urun.TeminSuresi,
             Sum(DepoStok.FasonUretimeCikis * UrunBirim.Carpan), Sum(DepoStok.FasonUretim * UrunBirim.Carpan), Sum(DepoStok.FasonUretimIade * UrunBirim.Carpan), Model.ModelId, Model.Adi from DepoStok
    left join Urun on Urun.UrunId = DepoStok.UrunId
    left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = urun.KimyasalTipiid
    left join UrunBirim on UrunBirim.urunId = DepoStok.UrunId and UrunBirim.DefaultBirim = 1
    left join UrunFiyat on UrunFiyat.urunId = Urun.UrunId
    left join Model on Model.ModelKayitID = Urun.ModelKayitID
    where (:AUrunAralikli = 0 or (:AUrunAralikli = 1 and DepoStok.UrunId >= :ABaslangicUrunId) and (DepoStok.UrunId <= :ABitisUrunId or :ABitisUrunId  = '')) and
           ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicGrup1Id and (UrunGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
           ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicGrup2Id and (UrunGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
           ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicGrup3Id and (UrunGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
           ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicGrup4Id and (UrunGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = '')) and
           Urun.EnvantereDahil = 1 and
           (:ADepoId is null or (:ADepoId is not null and DepoStok.DepoId = :ADepoId)) and
           (:AModelId is null or (:AModelId is not null and Model.ModelId = :AModelId)) and
           (:AUrunParcaDurumu is null or (Iif(:AUrunParcaDurumu = 0, Iif(Urun.ModelParcaId Is Null, 1, 0), Iif(:AUrunParcaDurumu = 1, Iif(Urun.ModelParcaId Is Not Null, 1, 0), 1)) = 1)) and
           ((:ABoyaUrunTipi = 1 or Urun.UrunTipiId <> 0) and
            (:AKimyasalUrunTipi = 1 or Urun.UrunTipiId <> 1) and
            (:ADigerUrunTipi = 1 or Urun.UrunTipiId <> 2) and
            (:AYedekParcaUrunTipi = 1 or Urun.UrunTipiId <> 3) and
            (:ASarfMalzemesiUrunTipi = 1 or Urun.UrunTipiId <> 4) and
            (:AHizmetUrunTipi = 1 or Urun.UrunTipiId <> 5) and
            (:ASablonUrunTipi = 1 or Urun.UrunTipiId <> 6) and
            (:AAksesuarUrunTipi = 1 or Urun.UrunTipiId <> 7) and
            (:ATicariMalUrunTipi = 1 or Urun.UrunTipiId <> 8)
           )  and
           (Urun.Kapali = :AKapaliDurumu or :AKapaliDurumu is null)
     group by Urun.UrunId, Urun.Adi, Urun.Raf, Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, Urun.KdvOrani,UrunBirim.BirimId,
              UrunBirim.UrunBirimId, Urun.TransferKodu, Urun.KimyasalTipiId, KimyasalTipi.Adi, Urun.Kapali,
              UrunFiyat.AlisFiyati,  UrunFiyat.AlisFiyatiDovizId, UrunFiyat.SatisFiyati,  UrunFiyat.SatisFiyatiDovizId, UrunFiyat.UretimeCikisFiyati,  UrunFiyat.UretimeCikisFiyatiDovizId,
              UrunBirim.UrunBirimId, Urun.TeminSuresi, Model.ModelId, Model.Adi
     into :UrunId, :UrunAdi, :Raf, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :KdvOrani, :BirimId, :TransferKodu, :KimyasalTipiId, :KimyasalTipiAdi,
          :Kapali, :AlisFiyati,  :AlisFiyatiDovizId, :SatisFiyati,  :SatisFiyatiDovizId, :UretimeCikisFiyati,  :UretimeCikisFiyatiDovizId,
          :DevirMiktari, :AlisMiktari, :AlisIadesiMiktari, :SatisMiktari, :SatisIadesiMiktari, :UretimeCikisMiktari,
          :DepolaraCikanMiktar, :DepolardanGelenMiktar, :UretimMiktari, :UrunBirimId, :TeminSuresi,
          :FasonUretimeCikisMiktari, :FasonUretimMiktari, :FasonUretimIadeMiktari, :ModelId, :ModelAdi do

  begin
    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, UrunGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      StokMiktari = DevirMiktari + AlisMiktari + UretimMiktari + SatisIadesiMiktari + DepolardanGelenMiktar + FasonUretimMiktari + FasonUretimIadeMiktari -
                   (SatisMiktari + AlisIadesiMiktari + UretimeCikisMiktari + DepolaraCikanMiktar + FasonUretimeCikisMiktari);
      if (ASifirOlanlarDokulsunmu = 1 or StokMiktari <> 0) then suspend;
    end
  end
end
