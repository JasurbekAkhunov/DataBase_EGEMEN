CREATE PROCEDURE BAKIMURUNKRITIKSEVIYERAPORU
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    TOPLAMBAKIMMIKTARI INTEGER,
    OPSIYONELBAKIMMIKTARI INTEGER,
    ZORUNLUBAKIMMIKTARI INTEGER,
    STOKMEVCUDU DOUBLE PRECISION,
    TALEPMIKTARI DOUBLE PRECISION,
    SIPARISMIKTARI DOUBLE PRECISION,
    SIPARISGELENMIKTAR DOUBLE PRECISION,
    SIPARISKALANMIKTAR DOUBLE PRECISION,
    KRITIKGUN TIMESTAMP,
    KRBAKIMMIKTARI INTEGER,
    KROPSIYONELBAKIMMIKTARI INTEGER,
    KRZORUNLUBAKIMMIKTARI INTEGER)
AS
declare Variable FToplam Integer;
declare Variable TahminiBaslangic TimeStamp;
declare Variable FMiktar Integer;
begin
  for select MakinaBakimIEUrun.UrunId, Urun.Adi, sum(Miktar)
    from MakinaBakimIEUrun
    left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
    left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
    left join MakinaBakimIsEmri on MakinaBakimIsEmri.MakinaBakimIsEmriId = MakinaBakimIEHizmet.MakinaBakimIsEmriId
    where MakinaBakimIsEmri.Kapali = 0 and
      MakinaBakimIsEmri.BakimEndKullaniciId is Null and
      MakinaBakimIEHizmet.Kapali = 0
    group by MakinaBakimIEUrun.UrunId, Urun.Adi, MakinaBakimIEUrun.Opsiyonel
    into :UrunId, :UrunAdi, :ToplamBakimMiktari do
  begin
    if (ToplamBakimMiktari is Null) then ToplamBakimMiktari = 0;

    Select Sum(Devir) + Sum(Alis-AlisIadesi) - Sum(Satis-SatisIadesi) + Sum(Uretim - UretimeCikis) + Sum(FasonUretim + FasonUretimIade - FasonUretimeCikis) FROM DepoStok
    WHERE DepoStok.UrunID = :UrunId
    into :StokMevcudu;
    if (StokMevcudu is Null) then StokMevcudu = 0;

    if (ToplamBakimMiktari > StokMevcudu) then
    begin
      select sum(Miktar)
      from MakinaBakimIEUrun
      left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
      left join MakinaBakimIsEmri on MakinaBakimIsEmri.MakinaBakimIsEmriId = MakinaBakimIEHizmet.MakinaBakimIsEmriId
      left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
      where MakinaBakimIsEmri.Kapali = 0 and
        MakinaBakimIsEmri.BakimEndKullaniciId is Null and
        MakinaBakimIEHizmet.Kapali = 0 and
        MakinaBakimIEUrun.UrunId = :UrunId and
        MakinaBakimIEUrun.Opsiyonel = 0
      into :ZorunluBakimMiktari;
      if (ZorunluBakimMiktari is Null) then ZorunluBakimMiktari = 0;
      OpsiyonelBakimMiktari = ToplamBakimMiktari - ZorunluBakimMiktari;

      select sum(SatinAlmaTalepDetay.Miktar) from SatinAlmaTalepDetay
        left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
        where (SatinAlmaTalepDetay.UrunId is not Null and SatinAlmaTalepDetay.UrunId = :UrunId) and
          SatinAlmaTalepDetay.Kapali = 0 and
          SatinAlmaTalep.OnaylayanKullaniciId is not Null
        into :TalepMiktari;
      if (TalepMiktari is Null) then TalepMiktari = 0;

      select sum(VerilenSiparisDetay.Miktar),
      sum(VerilenSiparisDetay.Miktar  - (VerilenSiparisDetay.GelenMiktar - VerilenSiparisDetay.IadeEdilenMiktar)),
      sum(VerilenSiparisDetay.GelenMiktar - VerilenSiparisDetay.IadeEdilenMiktar) from SatinAlmaTalepDetay
        left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
        left join VerilenSiparisDetay on VerilenSiparisDetay.SatinAlmaTalepDetayId = SatinAlmaTalepDetay.SatinAlmaTalepDetayId
        left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
        where (VerilenSiparisDetay.UrunId is not Null and VerilenSiparisDetay.UrunId = :UrunId) and
          VerilenSiparisDetay.Kapali = 0 and
          VerilenSiparis.Kapali = 0 and
          VerilenSiparis.OnaylayanKullaniciId is not Null
        into :SiparisMiktari, :SiparisKalanMiktar, :SiparisGelenMiktar;
      if (SiparisMiktari is Null) then SiparisMiktari = 0;
      if (SiparisKalanMiktar is Null) then SiparisKalanMiktar = 0;
      if (SiparisGelenMiktar is Null) then SiparisGelenMiktar = 0;

      FToplam = 0;
      KrBakimMiktari = 0;
      for select TahminiBaslangic, Miktar
        from MakinaBakimIEUrun
        left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
        left join MakinaBakimIsEmri on MakinaBakimIsEmri.MakinaBakimIsEmriId = MakinaBakimIEHizmet.MakinaBakimIsEmriId
        left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
        where MakinaBakimIsEmri.Kapali = 0 and
          MakinaBakimIsEmri.BakimEndKullaniciId is Null and
          MakinaBakimIEHizmet.Kapali = 0 and
          MakinaBakimIEUrun.UrunId = :UrunId
        order by TahminiBaslangic
        into :TahminiBaslangic, :FMiktar do
      begin
        FToplam = FToplam + FMiktar;
        if (FToplam > StokMevcudu and KrBakimMiktari = 0) then
        begin
          KritikGun = TahminiBaslangic;

          select sum(Miktar)
            from MakinaBakimIEUrun
            left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
            left join MakinaBakimIsEmri on MakinaBakimIsEmri.MakinaBakimIsEmriId = MakinaBakimIEHizmet.MakinaBakimIsEmriId
            left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
            where MakinaBakimIsEmri.TahminiBaslangic <= :KritikGun and
              MakinaBakimIsEmri.Kapali = 0 and
              MakinaBakimIsEmri.BakimEndKullaniciId is Null and
              MakinaBakimIEHizmet.Kapali = 0 and
              MakinaBakimIEUrun.UrunId = :UrunId and
              MakinaBakimIEUrun.Opsiyonel = 0
            into :KrZorunluBakimMiktari;
          if (KrZorunluBakimMiktari is Null) then KrZorunluBakimMiktari = 0;

          select sum(Miktar)
            from MakinaBakimIEUrun
            left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
            left join MakinaBakimIsEmri on MakinaBakimIsEmri.MakinaBakimIsEmriId = MakinaBakimIEHizmet.MakinaBakimIsEmriId
            left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
            where MakinaBakimIsEmri.TahminiBaslangic <= :KritikGun and
              MakinaBakimIsEmri.Kapali = 0 and
              MakinaBakimIsEmri.BakimEndKullaniciId is Null and
              MakinaBakimIEHizmet.Kapali = 0 and
              MakinaBakimIEUrun.UrunId = :UrunId and
              MakinaBakimIEUrun.Opsiyonel = 1
            into :KrOpsiyonelBakimMiktari;
          if (KrOpsiyonelBakimMiktari is Null) then KrOpsiyonelBakimMiktari = 0;

          KrBakimMiktari = KrZorunluBakimMiktari + KrOpsiyonelBakimMiktari;
        end
      end
      suspend;
    end
  end
end
