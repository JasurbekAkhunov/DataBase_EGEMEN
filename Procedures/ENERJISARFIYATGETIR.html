CREATE PROCEDURE ENERJISARFIYATGETIR(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAID MAKINAKODUSIZE,
    AUNITEID INTEGER,
    AIPLIKMAKINAID MAKINAKODUSIZE)
RETURNS (
    MAKINAENERJISARFIYATID INTEGER,
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    IPLIKMAKINAID MAKINAKODUSIZE,
    IPLIKMAKINAADI ADI40,
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    TARIH DATE,
    DOGALGAZPULSE INTEGER,
    DOGALGAZSARF DOUBLE PRECISION,
    ELEKTIRIKPULSE INTEGER,
    ELEKTIRIKSARF DOUBLE PRECISION,
    OKUNANSUPULSE INTEGER,
    OKUNANSUSARF DOUBLE PRECISION,
    BUHARSARF DOUBLE PRECISION,
    KOMURSARF DOUBLE PRECISION,
    BUHARURETIMI DOUBLE PRECISION,
    URETIMMIKTARI DOUBLE PRECISION,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    URETIMMIKTARIMETRE DOUBLE PRECISION,
    KIRATUTARI DOUBLE PRECISION,
    KIMYASALMALIYETI DOUBLE PRECISION,
    CALISANSAYISI INTEGER,
    ISCILIKMALIYETI DOUBLE PRECISION,
    ATIKSU DOUBLE PRECISION,
    KONDENSSARF DOUBLE PRECISION,
    ISITMASUYUSARF DOUBLE PRECISION,
    SICAKSUSARF DOUBLE PRECISION,
    IPLIKMAKINAGRUPID MAKINAGRUPKODUSIZE,
    IPLIKMAKINAGRUPADI ADI30)
AS
begin
  For Select MakinaEnerjiSarfiyatID, MakinaEnerjiSarfiyat.MakinaId, Makina.Adi, MakinaEnerjiSarfiyat.UniteId, Unite.Adi, Tarih, DogalgazPulse,
             DogalgazSarf, ElektirikPulse, ElektirikSarf, OkunanSuPulse, OkunanSuSarf, BuharSarf, KomurSarf, BuharUretimi, UretimMiktari, Notu,
             UretimMiktariMetre, KiraTutari, KimyasalMaliyeti, CalisanSayisi, IscilikMaliyeti, IplikMakina.IplikMakinaId, IplikMakina.Adi,
             AtikSu, KondensSarf, IsitmaSuyuSarf, SicakSuSarf, IplikMakina.MakinaGrupId, IMG.Adi from MakinaEnerjiSarfiyat
    Left Join Makina On Makina.MakinaID = MakinaEnerjiSarfiyat.MakinaID
    Left Join Unite On Unite.UniteID = MakinaEnerjiSarfiyat.UniteID
    Left Join IplikMakina On IplikMakina.IplikMakinaId = MakinaEnerjiSarfiyat.IplikMakinaId
    Left Join MakinaGrup IMG on IMG.MakinaGrupId = IplikMakina.MakinaGrupId
    Where MakinaEnerjiSarfiyat.Tarih >= :ABaslangicTarihi and MakinaEnerjiSarfiyat.Tarih <= :ABitisTarihi and
          (MakinaEnerjiSarfiyat.MakinaId = :AMakinaID or :AMakinaID is Null) and
          (MakinaEnerjiSarfiyat.UniteID = :AUniteID or :AUniteID is Null) and
          (MakinaEnerjiSarfiyat.IplikMakinaId = :AIplikMakinaID or :AIplikMakinaID is Null)
    Order By Tarih
    into :MakinaEnerjiSarfiyatID, :MakinaId, :MakinaAdi, :UniteId, :UniteAdi, :Tarih, :DogalgazPulse,
         :DogalgazSarf, :ElektirikPulse, :ElektirikSarf, :OkunanSuPulse, :OkunanSuSarf, BuharSarf, KomurSarf, BuharUretimi, UretimMiktari, Notu,
         :UretimMiktariMetre, :KiraTutari, :KimyasalMaliyeti, :CalisanSayisi, :IscilikMaliyeti, :IplikMakinaId, :IplikMakinaAdi, :AtikSu,
         :KondensSarf, :IsitmaSuyuSarf, :SicakSuSarf, :IplikMakinaGrupId, :IplikMakinaGrupAdi do suspend ;
end
