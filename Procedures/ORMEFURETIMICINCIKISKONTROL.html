CREATE PROCEDURE ORMEFURETIMICINCIKISKONTROL(
    AIRSALIYEID INTEGER,
    AORMEFISID INTEGER,
    AORMEPLANLAMAID INTEGER,
    AMIKTAR BOYAMIKTARSIZE,
    ASTOKHAREKETID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    RAFMIKTAR BOYAMIKTARSIZE,
    SARFMIKTAR BOYAMIKTARSIZE,
    ONCEKICIKISMIKTARI BOYAMIKTARSIZE,
    ONCEKIMIKTAR BOYAMIKTARSIZE,
    LOT VARCHAR(150),
    RENKID INTEGER,
    RENKADI ADI50,
    MELANJKODU VARCHAR(20),
    RECETEVAR EGEBOOLEAN)
AS
declare variable hstIplik              integer;
declare variable AlinanSiparisKayitId  integer;
declare variable TahminiFireOrani      Double Precision;
declare variable UrunKayitId           integer;
declare variable UrunTipiId            integer;
declare variable KullanimOrani         Double precision;
declare variable DepoId                varchar(10);
declare variable FTahminiFireOrani     Double Precision;
declare variable FirmaId               FirmaKoduSize;
declare variable OtomatikUCikislardaReceteFire integer;
declare variable OUCikisRafMikCikisMikAzOlamaz integer;
declare variable shtFasonUretimeCikis  integer;
declare variable shtFasonUretim        integer;
declare variable shtUretim             integer;
declare variable uitOrme               integer;
begin
  Select hstIplik from HamTipiGetir into :hstIplik;

  Select ShtFasonUretimeCikis, shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis, :shtFasonUretim, :shtUretim;

  Select uitOrme from UretimIslemTipiGetir into :uitOrme;

  Select OtomatikUCikislardaReceteFire, OUCikisRafMikCikisMikAzOlamaz from Sistem where sistemid = 1 into :OtomatikUCikislardaReceteFire, :OUCikisRafMikCikisMikAzOlamaz;

  if (AOrmeFisId is not Null) then
    FirmaId = Null;
  else
    Select FirmaId from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = :AIrsaliyeId into :FirmaId;

  Select Unite.SarfiyatDepoID from OrmePlanlama
    left join OrmeMakina on OrmeMakina.OrmeMakinaID = OrmePlanlama.OrmeMakinaID
    left join Unite on Unite.UniteID = OrmeMakina.UniteID
    where OrmePlanlama.OrmePlanlamaId = :AOrmePlanlamaId
    into :DepoId;

  ReceteVar = 0;

  For Select OrmePlanlamaRecete.HamId, Ham.HamTipiId, OrmePlanlamaRecete.Lot, OrmePlanlamaRecete.MelanjKodu, OrmePlanlamaRecete.KullanimOrani,
    OrmePlanlamaRecete.TahminiFireOrani, Ham.Adi, OrmePlanlamaRecete.RenkId, Renk.Adi,
    OrmeSiparisDetay.TahminiFireOrani, OrmePlanlama.BoyaSiparisKayitId, OrmePlanlama.OrmeSiparisDetayId, OrmePlanlamaRecete.CikisMiktari from OrmePlanlamaRecete
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmePlanlamaRecete.OrmePlanlamaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
    left join Renk on Renk.RenkId = OrmePlanlamaRecete.RenkId
    where OrmePlanlamaRecete.OrmePlanlamaId = :AOrmePlanlamaID
    into :UrunKayitId, :UrunTipiId, :Lot, :MelanjKodu,  :KullanimOrani, :FTahminiFireOrani, :HamAdi, :RenkID, :RenkAdi,
         :TahminiFireOrani, :AlinanSiparisKayitID, :OrmeSiparisDetayId, :OncekiCikisMiktari do
  begin
    ReceteVar = 1;

    /*
    if (AStokHareketId > 0) then select Sum(StokHareket.Miktar) from StokHareket where StokHareket.StokHareketId = :AStokHareketId into :OncekiMiktar;

    */
    if (OncekiMiktar is null) then OncekiMiktar = 0;
    SarfMiktar = (OncekiMiktar + AMiktar) * KullanimOrani / 100;

    if (OtomatikUCikislardaReceteFire = 1 and FTahminiFireOrani is not Null) then TahminiFireOrani = FTahminiFireOrani;

    if (UrunTipiID = hstIplik) then
      SarfMiktar = SarfMiktar + SarfMiktar * TahminiFireOrani / 100;


    if (Lot = '') then Lot = Null;


    Select Sum(RafMiktari) from OrmeHamRafListesiGetir(:UrunKayitId, :AlinanSiparisKayitId, :OrmeSiparisDetayId, Null, Null, Null, :DepoId)
      left join Raf on Raf.RafId = OrmeHamRafListesiGetir.RafId
      where StrKarsilastir(OrmeHamRafListesiGetir.Lot, :Lot) = 1 and
            StrKarsilastir(OrmeHamRafListesiGetir.MelanjKodu, :MelanjKodu) = 1 and
            IntKarsilastir(OrmeHamRafListesiGetir.RenkId, :RenkID) = 1 and RafMiktari > 0 and StrKarsilastir(Raf.FirmaId, FirmaId) = 1
      into :RafMiktar;

    if (RafMiktar is null) then RafMiktar = 0;
    if (OncekiCikisMiktari is null) then OncekiCikisMiktari = 0;

    if (SarfMiktar > RafMiktar - 10) then /*chustta  -10 u sildik*/
    begin
      BoyaSiparisKayitId = AlinanSiparisKayitId;
      OrmeSiparisDetayId = OrmeSiparisDetayId;
      HamId = UrunKayitId;

      suspend;
    end

  end

  if (ReceteVar = 0) then
    suspend;
end
