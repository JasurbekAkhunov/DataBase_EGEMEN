CREATE PROCEDURE IPLIKLOTURETIMIGETIR(
    AIPLIKLOTID INTEGER)
RETURNS (
    IPLIKURETIMID INTEGER,
    IPLIKSTOKID INTEGER,
    KAPNO INTEGER,
    URETIMTARIHI TIMESTAMP,
    IPLIKIRSALIYEID INTEGER,
    IPLIKIRSALIYEDETAYID INTEGER,
    BOBINGRAMAJI NUMERIC(12,3),
    MIKTAR NUMERIC(12,3),
    BOBINSAYISI INTEGER,
    AMBALAJAGIRLIGI DOUBLE PRECISION,
    PALETAGIRLIGI DOUBLE PRECISION,
    NETMIKTAR NUMERIC(12,3),
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKRENGIID INTEGER,
    KONIKRENGIADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    IPLIKMAKINAID VARCHAR(10),
    IPLIKDEPOID VARCHAR(10),
    HAREKETTIPISTR VARCHAR(50),
    CIKISIPLIKIRSALIYEDETAYID INTEGER,
    SEVKTARIHI TIMESTAMP,
    IPLIKTALEPYIL INTEGER,
    IPLIKTALEPID INTEGER,
    IPLIKTALEPIDSTR VARCHAR(30),
    IPLIKTALEPKAYITID INTEGER,
    IPLIKTALEPDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable HareketTipi            integer;
declare variable shtSatis               integer;
declare variable shtSarimaCikis         integer;
declare variable shtBukumeCikis         integer;
declare variable EgemenFirmaId         integer;
begin
  select EgemenFirmaId from Sistem into :EgemenFirmaId;

  Select shtSatis, ShtSarimaCikis, shtBukumeCikis from IplikStokHareketTipiGetir into :shtSatis, :ShtSarimaCikis, :shtBukumeCikis;

  if (EgemenFirmaId = 4) then /* GULLE IPLIK ICIN */
  begin
    For Select IplikUretim.IplikUretimID, IplikUretim.IplikStokId, IplikUretim.KapNo, IplikUretim.UretimTarihi, IplikUretim.IplikIrsaliyeId,
               IplikUretim.IplikIrsaliyeDetayId, IplikUretim.BobinGramaji, IplikUretim.Miktar, IplikUretim.BobinSayisi,
               IplikUretim.AmbalajAgirligi, IplikUretim.PaletAgirligi, IplikUretim.NetMiktar, IplikUretim.AmbalajId, Ambalaj.Adi,
               IplikUretim.KonikRengiId, KonikRengi.Adi, IplikUretim.KonikCinsiId, KonikCinsi.Adi, IplikUretim.IplikMakinaId,
               IplikUretim.IplikDepoId, IplikStok.CikisIplikIrsaliyeDetayId, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.HareketTipi,
               IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, IplikUretim.IplikTalepKayitID, IplikUretim.IplikTalepDetayID, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               BoyaSiparis.FirmaId, SiparisFirma.Adi,
               Ik.adi, IplikUretim.InsertTarihi, Sk.Adi, IplikUretim.DegisimTarihi from IplikUretim
      left join Ambalaj on Ambalaj.AmbalajId = IplikUretim.AmbalajId
      left join KonikRengi on KonikRengi.KonikRengiId = IplikUretim.KonikRengiId
      left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikUretim.KonikCinsiId
      Left Join Kullanici IK on IK.KullaniciId = IplikUretim.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = IplikUretim.KullaniciId
      left join IplikStok on IplikStok.IplikStokId = IplikUretim.IplikStokId and IplikStok.Numarator = IplikUretim.Numarator
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.CikisIplikIrsaliyeDetayId
      left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikIrsaliyeDetay.IplikIrsaliyeId
      left join IplikTalep on IplikTalep.IplikTalepKayitID = IplikUretim.IplikTalepKayitID
      left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayID = IplikUretim.IplikTalepDetayID
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = IplikTalepDetay.BoyaSiparisKayitId
      left join Firma SiparisFirma on SiparisFirma.FirmaID = BoyaSiparis.FirmaId
      where IplikUretim.IplikLotID = :AIplikLotID and IplikUretim.GirisTipi = 0
      order by IplikUretim.AmbalajId, IplikUretim.KapNo
      into :IplikUretimID, :IplikStokId, :KapNo, :UretimTarihi, :IplikIrsaliyeId, :IplikIrsaliyeDetayId, :BobinGramaji, :Miktar,
           :BobinSayisi, :AmbalajAgirligi, :PaletAgirligi, :NetMiktar, :AmbalajId, :AmbalajAdi, :KonikRengiId, :KonikRengiAdi,
           :KonikCinsiId, :KonikCinsiAdi, :IplikMakinaId, :IplikDepoId, :CikisIplikIrsaliyeDetayId, :SevkTarihi, :HareketTipi,
           :IplikTalepYil, :IplikTalepId, :IplikTalepKayitID, :IplikTalepDetayID, :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
    begin
      if (HareketTipi is null) then HareketTipiStr = '';
      else if (HareketTipi = shtSatis) then HareketTipiStr = 'Satış';
      else if (HareketTipi = ShtSarimaCikis) then HareketTipiStr = 'Yumuşak Sarım';
      else if (HareketTipi = ShtBukumeCikis) then HareketTipiStr = 'Büküme Çıkış';
      else HareketTipiStr = 'Hatalı';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      if (IplikTalepYil is not null) then
        IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';
      else
        IplikTalepIdStr = '';

      suspend;
    end
  end
  else
  begin
    For Select IplikUretim.IplikUretimID, IplikUretim.IplikStokId, IplikUretim.KapNo, IplikUretim.UretimTarihi, IplikUretim.IplikIrsaliyeId,
               IplikUretim.IplikIrsaliyeDetayId, IplikUretim.BobinGramaji, IplikUretim.Miktar, IplikUretim.BobinSayisi,
               IplikUretim.AmbalajAgirligi, IplikUretim.PaletAgirligi, IplikUretim.NetMiktar, IplikUretim.AmbalajId, Ambalaj.Adi,
               IplikUretim.KonikRengiId, KonikRengi.Adi, IplikUretim.KonikCinsiId, KonikCinsi.Adi, IplikUretim.IplikMakinaId,
               IplikUretim.IplikDepoId, IplikStok.CikisIplikIrsaliyeDetayId, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.HareketTipi,
               IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, IplikUretim.IplikTalepKayitID, IplikUretim.IplikTalepDetayID, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               BoyaSiparis.FirmaId, SiparisFirma.Adi,
               Ik.adi, IplikUretim.InsertTarihi, Sk.Adi, IplikUretim.DegisimTarihi from IplikUretim
      left join Ambalaj on Ambalaj.AmbalajId = IplikUretim.AmbalajId
      left join KonikRengi on KonikRengi.KonikRengiId = IplikUretim.KonikRengiId
      left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikUretim.KonikCinsiId
      Left Join Kullanici IK on IK.KullaniciId = IplikUretim.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = IplikUretim.KullaniciId
      left join IplikStok on IplikStok.IplikStokId = IplikUretim.IplikStokId and IplikStok.Numarator = IplikUretim.Numarator
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.CikisIplikIrsaliyeDetayId
      left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikIrsaliyeDetay.IplikIrsaliyeId
      left join IplikTalep on IplikTalep.IplikTalepKayitID = IplikUretim.IplikTalepKayitID
      left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayID = IplikUretim.IplikTalepDetayID
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = IplikTalepDetay.BoyaSiparisKayitId
      left join Firma SiparisFirma on SiparisFirma.FirmaID = BoyaSiparis.FirmaId
      where IplikUretim.IplikLotID = :AIplikLotID and IplikUretim.GirisTipi = 0
      order by IplikUretim.KapNo
      into :IplikUretimID, :IplikStokId, :KapNo, :UretimTarihi, :IplikIrsaliyeId, :IplikIrsaliyeDetayId, :BobinGramaji, :Miktar,
           :BobinSayisi, :AmbalajAgirligi, :PaletAgirligi, :NetMiktar, :AmbalajId, :AmbalajAdi, :KonikRengiId, :KonikRengiAdi,
           :KonikCinsiId, :KonikCinsiAdi, :IplikMakinaId, :IplikDepoId, :CikisIplikIrsaliyeDetayId, :SevkTarihi, :HareketTipi,
           :IplikTalepYil, :IplikTalepId, :IplikTalepKayitID, :IplikTalepDetayID, :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
    begin
      if (HareketTipi is null) then HareketTipiStr = '';
      else if (HareketTipi = shtSatis) then HareketTipiStr = 'Satış';
      else if (HareketTipi = ShtSarimaCikis) then HareketTipiStr = 'Yumuşak Sarım';
      else if (HareketTipi = ShtBukumeCikis) then HareketTipiStr = 'Büküme Çıkış';
      else HareketTipiStr = 'Hatalı';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      if (IplikTalepYil is not null) then
        IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';
      else
        IplikTalepIdStr = '';

      suspend;
    end
  end
end
