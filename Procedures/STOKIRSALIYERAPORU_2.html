CREATE PROCEDURE STOKIRSALIYERAPORU_2(
    AHAREKETTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AIRSALIYENO IRSALIYENOSIZE,
    AURUNTIPIID INTEGER)
RETURNS (
    IRSALIYEID INTEGER,
    STOKDETAYID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNBIRIMID INTEGER,
    BIRIMID VARCHAR(2),
    CARPAN DOUBLE PRECISION,
    KDV DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    NETBIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LOT VARCHAR(150))
AS
begin
  if (AIrsaliyeNo is null) then
  begin
    for select Irsaliye.IrsaliyeId, StokDetay.StokDetayId, StokDetay.UrunId, Urun.Adi, StokDetay.UrunBirimId, UrunBirim.BirimId, UrunBirim.Carpan,
               StokDetay.Kdv, StokDetay.Miktar, StokDetay.NetBirimFiyat,

               VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, StokDetay.Notu, StokDetay.Lot from Irsaliye
      Left Join StokDetay on StokDetay.IrsaliyeId = Irsaliye.IrsaliyeId
      left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = StokDetay.VerilenSiparisDetayId
      left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
      left join Urun on Urun.UrunId = StokDetay.UrunId
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      where Irsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and Irsaliye.IrsaliyeTarihi <= :ABitisTarihi and
            (Irsaliye.HareketTipi = :AHareketTipiId) and
            (Irsaliye.FirmaId = :AFirmaId or :AFirmaId is null) and
            (Urun.UrunTipiId = :AUrunTipiId or :AUrunTipiId is null)
      into :IrsaliyeId, :StokDetayId, :UrunId, :UrunAdi, :UrunBirimId, :BirimId, :Carpan, :Kdv, :Miktar, :NetBirimFiyat,

           :VerilenSiparisYil, :VerilenSiparisId, :Notu, :Lot do
    begin
      VerilenSiparisIdStr = '';
      if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

      Tutar = NetBirimFiyat * Miktar;
      suspend;
    end
  end
  else
  begin
    for select Irsaliye.IrsaliyeId, StokDetay.StokDetayId, StokDetay.UrunId, Urun.Adi, StokDetay.UrunBirimId, UrunBirim.BirimId, UrunBirim.Carpan,
               StokDetay.Kdv, StokDetay.Miktar, StokDetay.NetBirimFiyat,

               VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, StokDetay.Notu, StokDetay.Lot from Irsaliye
      Left Join StokDetay on StokDetay.IrsaliyeId = Irsaliye.IrsaliyeId
      left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = StokDetay.VerilenSiparisDetayId
      left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
      left join Urun on Urun.UrunId = StokDetay.UrunId
      left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
      where (Irsaliye.IrsaliyeNo = :AIrsaliyeNo) and (Irsaliye.HareketTipi = :AHareketTipiId) and
            (Urun.UrunTipiId = :AUrunTipiId or :AUrunTipiId is null)
      into :IrsaliyeId, :StokDetayId, :UrunId, :UrunAdi, :UrunBirimId, :BirimId, :Carpan, :Kdv, :Miktar, :NetBirimFiyat,

           :VerilenSiparisYil, :VerilenSiparisId, :Notu, :Lot do
    begin
      VerilenSiparisIdStr = '';
      if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

      Tutar = NetBirimFiyat * Miktar;
      suspend;
    end
  end
end
