CREATE PROCEDURE PARTIGENELRAPORU_2(
    AIMALATTIPIID INTEGER,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    AFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    ARECETEID VARCHAR(20),
    AMODELNO VARCHAR(20),
    ABASLANGICACILISTARIHI TIMESTAMP,
    ABITISACILISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ASINIF VARCHAR(1),
    AISLEMGRUBU INTEGER,
    AISLEMTIPI INTEGER,
    AFATURAKESILECEK INTEGER,
    AACILIYETDERECESI INTEGER,
    AFASON INTEGER,
    ARECETEOKEY INTEGER,
    ATALIMAT INTEGER,
    ATALIMATOKEY INTEGER,
    ARAPORTIPI INTEGER,
    ARECETEYAZILDI INTEGER,
    APARTIGRUPID VARCHAR(4))
RETURNS (
    CIKILANPARTIKAYITID INTEGER,
    CIKILANPARTISINIFID INTEGER,
    CIKILANHAMADI ADI255,
    HAMSTOKSARFFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    HAMSTOKSARFFISIDETAYID INTEGER,
    PARTISINIFID INTEGER,
    RAFDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    METRE BOYAMIKTARSIZE,
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    LOT VARCHAR(150),
    RAFID RAFKODUSIZE,
    PARTISTOKTARTIID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER)
AS
begin
  for select PartiStok.PartiKayitID, PartiStok.PartiSinifId, Ham.Adi, PArtiAltBezCikislariGetir.HamStokSarfFisiId, PArtiAltBezCikislariGetir.FisYil, PArtiAltBezCikislariGetir.FisNo, PArtiAltBezCikislariGetir.FisTarihi, PArtiAltBezCikislariGetir.HamStokSarfFisiDetayId, PArtiAltBezCikislariGetir.PartiSinifId, PArtiAltBezCikislariGetir.RafDetayId, PArtiAltBezCikislariGetir.HamId, PArtiAltBezCikislariGetir.HamAdi, PArtiAltBezCikislariGetir.Metre, PArtiAltBezCikislariGetir.HamStokMasterId,
             PArtiAltBezCikislariGetir.HamStokDetayId, PArtiAltBezCikislariGetir.Lot, PArtiAltBezCikislariGetir.RafId, PArtiAltBezCikislariGetir.PartiStokTartiId, PArtiAltBezCikislariGetir.PartiYil, PArtiAltBezCikislariGetir.PartiId
      from PartiGenelRaporu(:AImalatTipiId, :ABaslangicPartiYil, :ABaslangicPartiId, :ABitisPartiYil, :ABitisPartiId, :AFirmaId, :ATemsilciId, :AReceteId, :AModelNo, :ABaslangicAcilisTarihi,
                            :ABitisAcilisTarihi, :ABoyaSiparisYil, :ABoyaSiparisId, :ASinif, :AIslemGrubu, :AIslemTipi, :AFaturaKesilecek, :AAciliyetDerecesi, :AFason, :AReceteOkey,
                            :ATalimat, :ATalimatOkey, :ARaporTipi, :AReceteYazildi, :APartiGrupId)
      left join PartiStok on PartiStok.PartiKayitID = PartiGenelRaporu.PartiKayitId
      left join Ham on Ham.HamId = PartiStok.HamId
      left join PartiAltBezCikislariGetir(PartiStok.PartiSinifId) on 1=1
      into :CikilanPartiKayitID, :CikilanPartiSinifId, :CikilanHamAdi, :HamStokSarfFisiId, :FisYil, :FisNo, :FisTarihi, :HamStokSarfFisiDetayId, :PartiSinifId, :RafDetayId, :HamId, :HamAdi, :Metre, :HamStokMasterId,
           :HamStokDetayId, :Lot, :RafId, :PartiStokTartiId, :PartiYil, :PartiId do suspend;
end
