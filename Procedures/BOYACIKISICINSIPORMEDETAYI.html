CREATE PROCEDURE BOYACIKISICINSIPORMEDETAYI(
    ABOYASIPARISKAYITID INTEGER,
    ADOKUMAHAREKETI EGEBOOLEAN)
RETURNS (
    RESULTID VARCHAR(2550),
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    MUSTERISIPARISNO VARCHAR(50),
    ARTIKELNO VARCHAR(20),
    MUSTERIKALITENO VARCHAR(20),
    RAFIDSTR VARCHAR(40),
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20))
AS
declare variable FRenkId integer;
declare variable FRafId        varchar(10);
declare variable Organik       SmallInt;
begin
  for Select OrmeSiparisDetay.OrmeSiparisDetayID, OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.HamId, Ham.Adi,
             OrmeSiparisDetay.RenkId, Renk.Adi, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.Organik,
             OrmeSiparisDetay.iplikboyu1, OrmeSiparisDetay.iplikboyu2, OrmeSiparisDetay.iplikboyu3, OrmeSiparisDetay.iplikboyu4 from OrmeSiparisDetay
  left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
  left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
  where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  order by OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.SiparisSiraNo
  into :OrmeSiparisDetayID, :SiparisSiraNo, :HamId, :HamAdi, :FRenkId, :RenkAdi, :ArtikelNo, :MusteriSiparisNo, :Organik, :iplikboyu1, :iplikboyu2, :iplikboyu3, :iplikboyu4 do
  begin
    ResultId = '"' || EgemenIntegerToStr(OrmeSiparisDetayID) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(SiparisSiraNo) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",' ;

    if (FRenkId is not null) then
      ResultId = ResultId || '"' || EgemenIntegerToStr(FRenkId) || '",' ;
    else
      ResultId = ResultId || '"",' ;

    ResultId = ResultId || '"' || EgemenIntegerToStr(Organik) || '"' ;


    RafIdStr = '';

    for Select RafId from BoyayaCikisIcinStokListesi(:OrmeSiparisDetayId, Null, :HamId, :FRenkId, Null, :ADokumaHareketi)
      group by RafId
      into :FRafId do
    begin
      if (Char_Length(RafIdStr) + Char_Length(FRafId) < 38) then
      begin
        if (RafIdStr <> '' ) then RafIdStr = RafIdStr || cast(', ' as varchar(2) character set UTF8);
        RafIdStr = RafIdStr || FRafId;
      end
    end

    suspend;
  end

  for Select OrmeSiparisDetay.OrmeSiparisDetayID, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.HamId, Ham.Adi,
             OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.Organik from OrmeSiparisDetay
  left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
  where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and OrmeSiparisDetay.HamId <> BoyaSiparisDetay.HamId
  group by OrmeSiparisDetay.OrmeSiparisDetayID, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.HamId, Ham.Adi,
           OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.Organik
  order by OrmeSiparisDetay.SiparisSiraNo
  into :OrmeSiparisDetayID, :SiparisSiraNo, :HamId, :HamAdi, :ArtikelNo, :MusteriSiparisNo, :Organik do
  begin
    ResultId = '"' || EgemenIntegerToStr(OrmeSiparisDetayID) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(SiparisSiraNo) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",' ;

    ResultId = ResultId || '"",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(Organik) || '"' ;

    RafIdStr = '';

    for Select RafId from BoyayaCikisIcinStokListesi(:OrmeSiparisDetayId, Null, :HamId, Null, Null, :ADokumaHareketi)
      group by RafId
      into :FRafId do
    begin
      if (Char_Length(RafIdStr) + Char_Length(FRafId) < 38) then
      begin
        if (RafIdStr <> '' ) then RafIdStr = RafIdStr || cast(', ' as varchar(2) character set UTF8);
        RafIdStr = RafIdStr || FRafId;
      end
    end

    suspend;
  end
end
