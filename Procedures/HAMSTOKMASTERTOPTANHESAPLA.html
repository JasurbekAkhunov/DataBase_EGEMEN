CREATE PROCEDURE HAMSTOKMASTERTOPTANHESAPLA(
    AHAMSTOKMASTERID INTEGER,
    SADECERAFLARISIL INTEGER)
AS
declare variable RafId varchar(10);
declare variable Kilo double precision;
declare variable Metre double precision;
declare variable KapAdedi integer;
declare variable GeciciKabulRafId integer;
declare variable KayitSayisi integer;
declare variable EgemenFirmaId integer;
begin
  select sistem.EgemenFirmaId from sistem into :EgemenFirmaId;

  if (EgemenFirmaId <> 133) then
  begin

    update HamStokMaster set OnayDurumu = 0, NeredenUpdate = 1, TopRaflamaYapildi = 0 where HamStokMaster.HamStokMasterId = :AHamStokMasterId;
    UPDATE HAMSTOKMASTERKABUL SET KABULMIKTAR=0, KONTROLDURUMU = 0,  KONTROLKULLANICIID = Null WHERE HAMSTOKMASTERID = :AHamStokMasterId;

    for select GeciciKabulRafId from GeciciKabulRaf where HamStokMasterId = :AHamStokMasterId into :GeciciKabulRafId do
    begin
      delete from RafDetay where GeciciKabulRafId = :GeciciKabulRafId;
    end

    delete from GeciciKabulRaf where HamStokMasterId = :AHamStokMasterId;

    if (SadeceRaflariSil <> 1) then
    begin
      select count(*) from HamStokMasterTop where RafId is Null and HamStokMasterId = :AHamStokMasterId into :KayitSayisi;

      if (KayitSayisi > 0) then exception TumToplarRaflanmamis;

      Select sum(Kilo), sum(Metre), count(*) from HamStokMasterTop where HamStokMasterTop.HamStokMasterId = :AHamStokMasterId into :Kilo, :Metre, :KapAdedi;

      update HamStokMaster set OnayDurumu = 0, NeredenUpdate = 1 where HamStokMaster.HamStokMasterId = :AHamStokMasterId;
      update HamStokMaster set Kilo = :Kilo, Metre = :Metre, KapAdedi = :KapAdedi, NeredenUpdate = 1, TopRaflamaYapildi = 1 where HamStokMaster.HamStokMasterId = :AHamStokMasterId;

      for select RafId, sum(Kilo), sum(Metre), count(*) from HamStokMasterTop
        where HamStokMasterId = :AHamStokMasterId
        group by RafId
        into :RafId, :Kilo, :Metre, :KapAdedi do
      begin
        if (RafId is not Null) then
        begin
          insert into GeciciKabulRaf (HamStokMasterId, RafId, KapAdedi, Kilo, Metre, Adet)
            values(:AHamStokMasterId, :RafId, :KapAdedi, :Kilo, :Metre, 0);
        end
      end
    end
  end
  else
  begin
    /*bteks*/
    update HamStokMaster set OnayDurumu = 0, NeredenUpdate = 1, TopRaflamaYapildi = 0 where HamStokMaster.HamStokMasterId = :AHamStokMasterId;
    UPDATE HAMSTOKMASTERKABUL SET KABULMIKTAR=0, KONTROLDURUMU = 0,  KONTROLKULLANICIID = Null WHERE HAMSTOKMASTERID = :AHamStokMasterId;

    if (SadeceRaflariSil <> 1) then
    begin
      select count(*) from HamStokMasterTop where RafId is Null and HamStokMasterId = :AHamStokMasterId into :KayitSayisi;

      if (KayitSayisi > 0) then exception TumToplarRaflanmamis;

      Select sum(Kilo), sum(Metre), count(*) from HamStokMasterTop where HamStokMasterTop.HamStokMasterId = :AHamStokMasterId into :Kilo, :Metre, :KapAdedi;

      update HamStokMaster set OnayDurumu = 0, NeredenUpdate = 1 where HamStokMaster.HamStokMasterId = :AHamStokMasterId;
      update HamStokMaster set Miktar2 = :Kilo, NeredenUpdate = 1, TopRaflamaYapildi = 1 where HamStokMaster.HamStokMasterId = :AHamStokMasterId;

     end
  end
end
