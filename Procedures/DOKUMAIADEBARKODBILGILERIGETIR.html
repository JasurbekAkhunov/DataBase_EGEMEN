CREATE PROCEDURE DOKUMAIADEBARKODBILGILERIGETIR(
    ABARKODID VARCHAR(30))
RETURNS (
    DOKUMAPLANLAMAID INTEGER,
    DOKUMAMAKINAID MAKINAKODUSIZE,
    DEPOID DEPOKODUSIZE,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    URETIMPARTINO VARCHAR(30),
    HAMID INTEGER,
    TELADEDI INTEGER,
    LOT VARCHAR(150),
    RENKID INTEGER,
    LEVENTNO INTEGER,
    COZGUISEMRIKAYITID INTEGER,
    ORMERAFID INTEGER,
    MIKTAR BOYAMIKTARSIZE,
    LEVENTTIPIID INTEGER,
    KALITENO INTEGER,
    URETICIID INTEGER)
AS
declare variable oshtUretimeCikis      Integer;
declare variable oshtFasonUretimeCikis Integer;
begin
  select OrmeStokHareketTipiGetir.shtUretimeCikis, OrmeStokHareketTipiGetir.shtFasonUretimeCikis from OrmeStokHareketTipiGetir into :oshtUretimeCikis, :oshtFasonUretimeCikis;

  select OrmeStokHareketTipiGetir.shtUretimeCikis, OrmeStokHareketTipiGetir.shtFasonUretimeCikis from OrmeStokHareketTipiGetir into :oshtUretimeCikis, :oshtFasonUretimeCikis;
  for select First 1 StokHareket.DokumaPlanlamaId, StokHareket.DokumaMakinaId, DepoRaf.DepoId, StokHareket.BoyaSiparisKayitId, StokHareket.OrmeSiparisDetayId,
      DokumaPlanlama.UretimPartiNo, OrmeRaf.HamId, OrmeRaf.TelAdedi, OrmeRaf.Lot, OrmeRaf.RenkId, OrmeRaf.LeventNo, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeRaf.CozguIsEmriKayitId,
      OrmeRaf.OrmeRafId, StokHareket.Miktar, OrmeRaf.LeventTipiId, OrmeRaf.KaliteNo, OrmeRaf.UreticiId
      from OrmeRaf
      left join StokHareket on StokHareket.OrmeRafId = OrmeRaf.OrmeRafId
      left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = StokHareket.DokumaPlanlamaId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitID
      left join Raf on Raf.RafId = OrmeRaf.RafId
      left join DepoRaf on DepoRaf.RafId = Raf.RafId
      where OrmeRaf.BarkodId = :ABarkodId and
      StokHareket.DokumaHareketi = 1 and
      StokHareket.HamIade = 0 and
      (StokHareket.HareketTipi = :oshtUretimeCikis or StokHareket.HareketTipi = :oshtFasonUretimeCikis)
      order by StokHareket.InsertTarihi desc nulls last
      into :DokumaPlanlamaId, :DokumaMakinaId, :DepoId, :BoyaSiparisKayitId, :OrmeSiparisDetayId,
      :UretimPartiNo, :HamId, :TelAdedi, :Lot, :RenkId, :LeventNo, :BoyaSiparisYil, :BoyaSiparisId, :CozguIsEmriKayitId, :OrmeRafId, :Miktar, :LeventTipiId, :KaliteNo, :UreticiId do
  begin
    suspend;
  end
end
