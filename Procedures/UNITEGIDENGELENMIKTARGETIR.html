CREATE PROCEDURE UNITEGIDENGELENMIKTARGETIR(
    AGONDERENUNITEID INTEGER,
    AALICIUNITEID INTEGER,
    APARTISINIFID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    GIDENKILO DOUBLE PRECISION,
    GIDENNETKILO DOUBLE PRECISION,
    GIDENMETRE DOUBLE PRECISION,
    GELENKILO DOUBLE PRECISION,
    GELENNETKILO DOUBLE PRECISION,
    GELENMETRE DOUBLE PRECISION)
AS
declare variable GonderenFirmaDisiUnite integer;
declare variable FKilo        Double Precision;
declare variable FNetKilo     Double Precision;
declare variable FMetre       Double Precision;
begin
  GelenKilo     = 0;
  GelenNetKilo  = 0;
  GelenMetre    = 0;
  GidenKilo     = 0;
  GidenNetKilo  = 0;
  GidenMetre    = 0;
  PartiSinifID  = :APartiSinifID;

  Select FirmaDisiUnite from Unite where Unite.UniteID = :AGonderenUniteID into :GonderenFirmaDisiUnite;

  Select Sum(UniteIrsaliyeDetay.Kilo), Sum(UniteIrsaliyeDetay.NetKilo), Sum(UniteIrsaliyeDetay.Metre) from UniteIrsaliye
    Left Join UniteIrsaliyeDetay on UniteIrsaliyeDetay.UniteIrsaliyeID = UniteIrsaliye.UniteIrsaliyeID
  where UniteIrsaliye.GonderenUniteID = :AGonderenUniteID and UniteIrsaliye.AliciUniteID = :AAliciUniteID and UniteIrsaliyeDetay.PartiSinifID = :APartiSinifID
  into :FKilo, :FNetKilo, :FMetre;

  if (GonderenFirmaDisiUnite = 0) then
  begin
    GidenKilo     = FKilo;
    GidenNetKilo  = FNetKilo;
    GidenMetre    = FMetre;
  end
  else
  begin
    GelenKilo     = FKilo;
    GelenNetKilo  = FNetKilo;
    GelenMetre    = FMetre;
  end

  Select Sum(UniteIrsaliyeDetay.Kilo), Sum(UniteIrsaliyeDetay.NetKilo), Sum(UniteIrsaliyeDetay.Metre) from UniteIrsaliye
    Left Join UniteIrsaliyeDetay on UniteIrsaliyeDetay.UniteIrsaliyeID = UniteIrsaliye.UniteIrsaliyeID
  where UniteIrsaliye.AliciUniteID = :AGonderenUniteID and UniteIrsaliye.GonderenUniteID = :AAliciUniteID and UniteIrsaliyeDetay.PartiSinifID = :APartiSinifID
  into :FKilo, :FNetKilo, :FMetre;

  if (GonderenFirmaDisiUnite = 0) then
  begin
    GelenKilo     = FKilo;
    GelenNetKilo  = FNetKilo;
    GelenMetre    = FMetre;
  end
  else
  begin
    GidenKilo     = FKilo;
    GidenNetKilo  = FNetKilo;
    GidenMetre    = FMetre;
  end
  Suspend;
end
