CREATE PROCEDURE ORMESIPARISKOPYALA(
    ABOYASIPARISKAYITID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER)
AS
declare variable BoyaSiparisKayitID  integer;
declare variable BoyaSiparisDetayId integer;
declare variable OrmeSiparisDetayId integer;
declare variable AOrmeSiparisDetayId integer;
declare variable ABoyaSiparisDetayId integer;
declare variable EskiBoyaSiparisRenkNo  integer;
declare variable YeniBoyaSiparisRenkNo  integer;
declare variable BoyaSiparisRenkNoKullanimda integer;
declare variable SiparisKopyalarkenFiyatiTasi integer;
declare variable SatisFiyati Double Precision;
declare variable FFasonBirimFiyati Double Precision;
declare variable FBaskiBirimFiyati Double Precision;
declare variable BirimFiyat Double Precision;
declare variable OTOlanOSiparisKopyalanabilir integer;
declare variable TalimatVar integer;
declare variable OrmeBirimMaliyet Double Precision;
begin
  Select BoyaSiparisRenkNoKullanimda, SiparisKopyalarkenFiyatiTasi, OTOlanOSiparisKopyalanabilir from sistem
    where sistemid = 1
    into :BoyaSiparisRenkNoKullanimda, :SiparisKopyalarkenFiyatiTasi, :OTOlanOSiparisKopyalanabilir;

  if (OTOlanOSiparisKopyalanabilir = 0) then
  begin
    Select Count(*) from OrmePlanlama where OrmePlanlama.BoyaSiparisKayitId = :ABoyaSiparisKayitId into :TalimatVar;
    if (TalimatVar > 0) then exception KayitHatasi ' Örme Talimatı Olan Sipariş Kopyalanamaz.';

    Select Count(*) from OrmeSiparisDetay
      where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and OrmeSiparisDetay.KTBoyaSiparisKayitId is not null
      into :TalimatVar;

    if (TalimatVar > 0) then exception KayitHatasi ' Kumaş Temin Siparişi Açılmış, Sipariş Kopyalanamaz.';
  end


  BoyaSiparisYil = f_Year(Current_date);

  BoyaSiparisId = Null;

  Select Id from BoyaSiparisNumaratoruAl(:BoyaSiparisYil) into :BoyaSiparisId;

  insert into BoyaSiparis(BoyaSiparisYil, BoyaSiparisId, FirmaId, SiparisTarihi, SiparisiAlanKullaniciId, FirmaYetkiliId, IslemTipiId,
                          SiparisGrupId, SiparisGrup2Id, Fason, OdemeSekliId, TeslimSekliId, YurtDisi, Notu, SiparisTipiId, PSKKEtiketTipiId,
                          OrmeSiparisSekliId, OrmeSiparisTipiId, OrmeSiparisi, DokumaSiparisTipiId, DokumaSiparisi, InsertKullaniciId, KullaniciId)
             Select :BoyaSiparisYil, :BoyaSiparisId, FirmaId, Current_date, :AKullaniciId, FirmaYetkiliId, IslemTipiId,
                     SiparisGrupId, SiparisGrup2Id, Fason, OdemeSekliId, TeslimSekliId, YurtDisi, Notu, SiparisTipiId, PSKKEtiketTipiId,
                     OrmeSiparisSekliId, OrmeSiparisTipiId, OrmeSiparisi, DokumaSiparisTipiId, DokumaSiparisi, :AKullaniciId, :AKullaniciId from BoyaSiparis
               where BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitID;

  Select BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :BoyaSiparisYil and BoyaSiparis.BoyaSiparisID = :BoyaSiparisID into :BoyaSiparisKayitId;


  if (BoyaSiparisRenkNoKullanimda = 1) then
  begin
    insert into BoyaSiparisRenk(BoyaSiparisKayitId, LabReceteId, LabReceteKodu, LabReceteAtisId, RenkId, RenkAdi, MusteriRenkKoduveAdi)
       Select :BoyaSiparisKayitId, LabReceteId, LabReceteKodu, LabReceteAtisId, RenkAdi, RenkId, MusteriRenkKoduveAdi from BoyaSiparisRenk where BoyaSiparisRenk.BoyaSiparisKayitId = :ABoyaSiparisKayitId;
  end

  For Select OrmeSiparisDetayId, SatisFiyati, OrmeBirimMaliyet from OrmeSiparisDetay
  where OrmeSiparisDetay.BoyaSiparisKayitID = :ABoyaSiparisKayitID
  order by OrmeSiparisDetay.OrmeSiparisDetayId
  into :AOrmeSiparisDetayId, :SatisFiyati, :OrmeBirimMaliyet do
  begin
    OrmeSiparisDetayId = gen_id(OrmeSiparisDetay_Id_Gen, 1);

    if (SiparisKopyalarkenFiyatiTasi = 0) then
    begin
      SatisFiyati = 0;
      OrmeBirimMaliyet = 0;
    end

    Insert Into OrmeSiparisDetay(BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, MelanjKodu, FlamKodu, HamMiktar, Metre, SatinAlinacakMiktar, StoktanKullanilacakMiktar, SatisFiyati,
                                   DovizId, OrmeBirimMaliyet, OrmeBirimMaliyetDovizId, TeslimTarihi, RevizeTeslimTarihi, MusteriSiparisNo, ReferansSiparisNo,
                                   VadeGun, OdemeSekliId, MakinaUstuGramaj, DinlenmisGramaj, GerceklesenGramaj, IgneSayisi, MaySayisi, Pus, Feine, TupAcikEnMayli, ArtikelNo,
                                   Beyazlik, TahminiFireOrani, IplikBoyu1, IplikBoyu2, IplikBoyu3, IplikBoyu4, RaporBoyu, SevkYeri, MusteriTalimati, Notu, KaliteId, KopyaId,
                                   InsertKullaniciId, KullaniciId, HamStokBirimID)
         Select :BoyaSiparisKayitId, :OrmeSiparisDetayId, HamId, RenkId, MelanjKodu, FlamKodu, HamMiktar, Metre, SatinAlinacakMiktar, StoktanKullanilacakMiktar, :SatisFiyati,
                 DovizId, :OrmeBirimMaliyet, OrmeBirimMaliyetDovizId, TeslimTarihi, RevizeTeslimTarihi,
                 MusteriSiparisNo, ReferansSiparisNo, VadeGun, OdemeSekliId, MakinaUstuGramaj, DinlenmisGramaj, GerceklesenGramaj, IgneSayisi, MaySayisi, Pus, Feine, TupAcikEnMayli, ArtikelNo,
                 Beyazlik, TahminiFireOrani, IplikBoyu1, IplikBoyu2, IplikBoyu3, IplikBoyu4, RaporBoyu, SevkYeri, MusteriTalimati, Notu, KaliteId, OrmeSiparisDetayId, :AKullaniciId, :AKullaniciId, HamStokBirimID from OrmeSiparisDetay
             where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId;

    For Select BoyaSiparisDetayId, BoyaSiparisRenkNo, BirimFiyat, FasonBirimFiyati, BaskiBirimFiyati from BoyaSiparisDetay
       where BoyaSiparisDetay.OrmeSiparisDetayID = :AOrmeSiparisDetayID
       order by BoyaSiparisDetay.BoyaSiparisDetayId
    into :ABoyaSiparisDetayId, :EskiBoyaSiparisRenkNo, :BirimFiyat, :FFasonBirimFiyati, :FBaskiBirimFiyati do
    begin
      if (SiparisKopyalarkenFiyatiTasi = 0) then
      begin
        BirimFiyat = 0;
        FFasonBirimFiyati = 0;
        FBaskiBirimFiyati = 0;
      end

      BoyaSiparisDetayId = gen_id(BoyaSiparisDetay_Id_Gen, 1);

      YeniBoyaSiparisRenkNo = Null;

      if (BoyaSiparisRenkNoKullanimda = 1) then
      begin
        Select Max(YeniRenk.BoyaSiparisRenkNo) from BoyaSiparisRenk EskiRenk
          left join BoyaSiparisRenk YeniRenk on YeniRenk.BoyaSiparisKayitID = :BoyaSiparisKayitID
          where EskiRenk.BoyaSiparisKayitId = :ABoyaSiparisKayitId and EskiRenk.BoyaSiparisRenkNo = :EskiBoyaSiparisRenkNo and
                IntKarsilastir(EskiRenk.LabReceteID, YeniRenk.LabReceteID) = 1 and
                IntKarsilastir(EskiRenk.LabReceteAtisID, YeniRenk.LabReceteAtisID) = 1 and
                StrKarsilastir(EskiRenk.RenkAdi, YeniRenk.RenkAdi) = 1 and
                StrKarsilastir(EskiRenk.MusteriRenkKoduveAdi, YeniRenk.MusteriRenkKoduveAdi) = 1
          into :YeniBoyaSiparisRenkNo;
      end

      Insert into BoyaSiparisDetay(BoyaSiparisKayitId, OrmeSiparisDetayId, BoyaSiparisDetayId, BoyaSiparisRenkNo, FirmaSevkAdresiId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
                                   RenkId, RenkAdi, ProsesId, SiparisProsesId, SiparisKilo, SiparisMetre, PaketlemeSekliId, RenkDerinligiId, BirimFiyat, HamStokBirimId,
                                   FaturaEdilecekMiktarTipi, DovizId, UcretliTamir, FaturaKesilecek, VadeGun, OdemeSekliId, HamId, Pus, Feine, Lot, MusteriRenkKoduveAdi,
                                   DesenId, DesenKenarYazisi, DesenIlkBaski, MelanjKodu, FlamKodu,
                                   KaliteId, MusteriKaliteNo, MusteriKaliteAdi, Denye, IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen,
                                   CekmezlikBoyIstenen, YikamaHasligiIstenen, SansIstenen, ToleransEn, ToleransGramaj, ToleransEndenCekmezlik, ToleransBoydanCekmezlik,
                                   ToleransDonme, ToleransSans, TerminTarihi, RevizeTerminTarihi, Notu,
                                   YikamaDereceId, KurutmaSekliId, YikamaHasligiTestiId, OngoruYapilacak, FirmaAdimizEtiketteCiksinmi, MaximumFireOrani, MinimumFireOrani, IslemGrubuId, FasonBirimFiyati, FasonBirimFiyatDovizId, BaskiBirimFiyati, BaskiBirimFiyatDovizId, InsertKullaniciId, KullaniciId, KopyaSonuUpdateYok, SiparisteISBrutNetDaradan, PSKKEtiketTipiId)
        Select :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :YeniBoyaSiparisRenkNo, FirmaSevkAdresiId, MusteriSiparisNo, ReferansSiparisNo, LabReceteId, LabReceteKodu, LabReceteAtisId,
               RenkId, RenkAdi, ProsesId, SiparisProsesId, SiparisKilo, SiparisMetre, PaketlemeSekliId, RenkDerinligiId, :BirimFiyat, HamStokBirimId,
               FaturaEdilecekMiktarTipi, DovizId, UcretliTamir, FaturaKesilecek, VadeGun, OdemeSekliId, HamId, Pus, Feine, Lot, MusteriRenkKoduveAdi,
               DesenId, DesenKenarYazisi, DesenIlkBaski, MelanjKodu, FlamKodu,
               KaliteId, MusteriKaliteNo, MusteriKaliteAdi, Denye, IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen,
               CekmezlikBoyIstenen, YikamaHasligiIstenen, SansIstenen, ToleransEn, ToleransGramaj, ToleransEndenCekmezlik, ToleransBoydanCekmezlik,
               ToleransDonme, ToleransSans, TerminTarihi, RevizeTerminTarihi, Notu,
               YikamaDereceId, KurutmaSekliId, YikamaHasligiTestiId, OngoruYapilacak, FirmaAdimizEtiketteCiksinmi, MaximumFireOrani, MinimumFireOrani, IslemGrubuId, :FFasonBirimFiyati, FasonBirimFiyatDovizId, :FBaskiBirimFiyati, BaskiBirimFiyatDovizId, :AKullaniciId, :AKullaniciId, 1, SiparisteISBrutNetDaradan, PSKKEtiketTipiId from BoyaSiparisDetay
        where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

      Insert into SiparisIlaveIslemler(BoyaSiparisDetayId, BoyaSiparisKayitId, BirimFiyat, DovizId, IlaveIslemlerId, InsertKullaniciId, KullaniciId)
        Select :BoyaSiparisDetayId, :BoyaSiparisKayitId, BirimFiyat, DovizId, IlaveIslemlerId, :AKullaniciId, :AKullaniciId from SiparisIlaveIslemler
        where SiparisIlaveIslemler.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

      Insert into SiparisOzelTestler(BoyaSiparisDetayId, BoyaSiparisKayitId, OzelTestId, InsertKullaniciId, KullaniciId)
        Select :BoyaSiparisDetayId, :BoyaSiparisKayitId, OzelTestId, :AKullaniciId, :AKullaniciId from SiparisOzelTestler
        where SiparisOzelTestler.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

      Insert into SiparisOzelApreler(BoyaSiparisDetayId, BoyaSiparisKayitId, OzelApreId, InsertKullaniciId, KullaniciId)
        Select :BoyaSiparisDetayId, :BoyaSiparisKayitId, OzelApreId, :AKullaniciId, :AKullaniciId from SiparisOzelApreler
        where SiparisOzelApreler.BoyaSiparisDetayId = :ABoyaSiparisDetayId;

      update BoyaSiparisDetay set KopyaSonuUpdateYok = 0 where BoyaSiparisDetayId = :BoyaSiparisDetayId;
    end
  end

  suspend;
end
