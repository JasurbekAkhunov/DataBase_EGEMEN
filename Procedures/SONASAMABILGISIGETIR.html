CREATE PROCEDURE SONASAMABILGISIGETIR(
    APARTISINIFID INTEGER,
    APARTIASAMALARIID INTEGER)
RETURNS (
    SONYAPILANASAMAADI VARCHAR(40),
    SONYAPILANASAMAGIRISTARIHI TIMESTAMP,
    SONYAPILANASAMACIKISTARIHI TIMESTAMP,
    SONRAKIASAMAADI VARCHAR(40),
    SONCIKISTARIHI TIMESTAMP,
    SONRAKIASAMAID INTEGER,
    SONRAKIASAMACOLOR INTEGER)
AS
declare variable SonYapilanSira integer;
declare variable SonrakiSira integer;
begin
  if (APartiAsamalariId is not null) then
  begin
    select Asama.Adi, PartiAsamalari.Sira, PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi from PartiAsamaStok
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiAsamalariId = :APartiAsamalariId and PartiAsamaStok.PartiSinifId = :APartiSinifId
      into :SonYapilanAsamaAdi, :SonYapilanSira, :SonYapilanAsamaGirisTarihi, :SonYapilanAsamaCikisTarihi;
  end
  else
  begin
    SonYapilanAsamaAdi = Null;
    SonYapilanSira = -1;
  end

  select Min(PartiAsamalari.Sira) from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.Sira > :SonYapilanSira
    into :SonrakiSira;

  if (SonrakiSira is not null) then
  begin
    select Asama.Adi, Asama.AsamaId, Asama.Color from PartiAsamaStok
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamalari.Sira = :SonrakiSira
      into :SonrakiAsamaAdi, :SonrakiAsamaID, :SonrakiAsamaColor;
  end


  select Max(CikisTarihi) from PartiAsamaStok
    where PartiAsamaStok.PartiSinifId = :APartiSinifId
    into :SonCikisTarihi;
  suspend;
end
