CREATE PROCEDURE HGKESIMUCIKISICINSTOKLISTESI(
    AHGISEMRIKAYITID INTEGER,
    ADEPOID VARCHAR(10),
    ARAFID VARCHAR(10),
    AHGSIPARISDETAYID INTEGER)
RETURNS (
    HGISEMRIPASTALID INTEGER,
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    FIREORANI DOUBLE PRECISION,
    FAZLAKESIMORANI DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    RAFID VARCHAR(10),
    HESAPLANANMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI NUMERIC(12,3),
    KALITENO INTEGER,
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    URETICIADI VARCHAR(40),
    URETIMPARTINO VARCHAR(30))
AS
declare variable SiparisFirmaId varchar(20);
declare variable StokFirmaId varchar(20);
declare variable OrmeFirmaKodumuz varchar(20);
begin
  Select HGSiparis.FirmaId, HGSiparis.Fason from HGIsEmri
  left join HGSiparis on HGSiparis.HGSiparisKayitId = HGIsEmri.HGSiparisKayitId
  where HGIsEmri.HGIsEmriKayitId = :AHGIsEmriKayitId
  into :SiparisFirmaId, :Fason;

  Select OrmeFirmaKodumuz from sistem where sistemid = 1 into :OrmeFirmaKodumuz;

  For Select HGIsEmriPastal.HGIsEmriPastalID, HGSiparisDetayKumas.HamId, Ham.HamTipiId, HamTipi.Adi, Ham.Adi, HGSiparisDetayKumas.RenkId, Renk.Adi, HGSiparisDetayKumas.Fireorani,
             HGSiparisDetayKumas.FazlaKesimOrani,
            ( (((HGSiparisDetayKumas.Miktar / 1000) * iif(HGSiparisDetayKumas.FireOrani > 0, HGSiparisDetayKumas.FireOrani / 100 + 1, 1)) *
                iif(HGSiparisDetayKumas.FazlaKesimOrani > 0, HGSiparisDetayKumas.FazlaKesimOrani / 100 + 1, 1)
              ) * (HGIsEmriPastalKumas.KatSayisi * (select Sum(HGIsEmriPastalBeden.Miktar) from HGIsEmriPastalBeden where HGIsEmriPastalBeden.HGIsEmriPastalID = HGIsEmriPastal.HGIsEmriPastalID))   )
 from HGIsEmri
    left join HGIsEmriPastal on HGIsEmriPastal.HGIsEmriKayitID = HGIsEmri.HGIsEmriKayitID
    left join HGIsEmriPastalKumas on HGIsEmriPastalKumas.HGIsEmriPastalID = HGIsEmriPastal.HGIsEmriPastalID
    left join HGSiparisDetayKumas on HGSiparisDetayKumas.HGSiparisDetayKumasID = HGIsEmriPastalKumas.HGSiparisDetayKumasID
    left join Ham on Ham.HamId = HGSiparisDetayKumas.HamId
    Left Join HamTipi On HamTipi.HamTipiId = Ham.HamTipiId
    left join Renk on Renk.RenkId = HGSiparisDetayKumas.RenkId
    where (HGIsEmri.HGIsEmriKayitId = :AHGIsEmriKayitId or HgIsEmri.HgSiparisDetayId = :AHgSiparisDetayId)
  into  :HGIsEmriPastalID, :HamId, :HamTipiId, :HamTipiAdi, :HamAdi, :RenkId, :RenkAdi, :FireOrani, :FazlaKesimOrani, :HesaplananMiktar do
  begin

    if (Fason = 1) then
    begin
      FasonStr = 'Fason';
      StokFirmaID = SiparisFirmaId;
    end
    else
    begin
      FasonStr = '';
      StokFirmaID = OrmeFirmaKodumuz;
    end

    For Select OrmeRaf.OrmeRafId, OrmeRaf.RafId, OrmeRaf.StokGirisNo, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari, OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi,
               OrmeRaf.Lot, OrmeRaf.MelanjKodu, OrmeRaf.KaliteNo, Uretici.Adi, UretimPartiNo from OrmeRaf
      left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      where OrmeRaf.HamID = :HamId and OrmeRaf.StokFirmaId = :StokFirmaID and OrmeRaf.StokVar = 1 and
            StrKarsilastir(OrmeRaf.RenkID, :RenkId) = 1 and OrmeRaf.FasonStok = :Fason and
           (DepoRaf.DepoId = :ADepoID or :ADepoID is null) and
           (OrmeRaf.RafID = :ARafID or :ARafID is null)
      into :OrmeRafId, :RafId, :StokGirisNo, :KalanMiktar, :KalanKapAdedi, :Lot, :MelanjKodu,  :KaliteNo, :UreticiAdi, :UretimPartiNo do
    begin
      suspend;
    end
  end
end
