CREATE PROCEDURE BAKIMTIPINDENISEMRITALEPYARAT(
    AMAKINABAKIMISEMRIID INTEGER,
    ASADECEZORUNLULAR INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    SATINALMATALEPKAYITID INTEGER,
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER)
AS
declare variable BakimSatinAlmaDepartmanId          integer;
declare variable BakimSatinAlmaTalepYeriId          integer;
declare variable BakimSatinAlmaBirimId              integer;
declare variable BakimSatinAlmaKullanimYeriId       integer;
declare variable KayitSayisi       Double Precision;
declare variable UrunId                    varchar(20);
declare variable UrunAdi                   varchar(80);
declare variable SistemDovizId                    varchar(5);
declare variable DefaultBirimId                   integer;
declare variable IsEmriTarihi TimeStamp;
begin
  SatinAlmaTalepKayitId = -1;
  /*if (SatinAlmaTalepKayitId is not Null) then Exception KayitHatasi 'Bu iş emri için daha önceden talep oluşturulmuş...';*/

  select TahminiBaslangic from MakinaBakimIsEmri where MakinaBakimIsEmriId = :AMakinaBakimIsEmriId into :IsEmriTarihi;

  select BakimSatinAlmaDepartmanId, BakimSatinAlmaTalepYeriId, BakimSatinAlmaBirimId, BakimSatinAlmaKullanimYeriId, SistemDovizId from Sistem
    into :BakimSatinAlmaDepartmanId, :BakimSatinAlmaTalepYeriId, :BakimSatinAlmaBirimId, :BakimSatinAlmaKullanimYeriId, :SistemDovizId;

  if (BakimSatinAlmaDepartmanId is Null) then Exception KayitHatasi 'Sistemde BakimSatinAlmaDepartmanId bilgisi girilmemiş...';
  if (BakimSatinAlmaTalepYeriId is Null) then Exception KayitHatasi 'Sistemde BakimSatinAlmaTalepYeriId bilgisi girilmemiş...';
  if (BakimSatinAlmaBirimId is Null) then Exception KayitHatasi 'Sistemde BakimSatinAlmaBirimId bilgisi girilmemiş...';
  if (BakimSatinAlmaKullanimYeriId is Null) then Exception KayitHatasi 'Sistemde BakimSatinAlmaKullanimYeriId bilgisi girilmemiş...';

  SatinAlmaTalepKayitId = gen_id(SatinAlmaTalepKayit_Id_Gen, 1);
  SatinAlmaTalepYil = f_Year(Current_date);
  select Id from SatinAlmaTalepNumaratoruAl(:SatinAlmaTalepYil) into :SatinAlmaTalepId;

  if (ASadeceZorunlular = 1) then
  begin
    select sum(Miktar)
        from MakinaBakimIEUrun
        left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
        where MakinaBakimIEHizmet.MakinaBakimIsEmriId = :AMakinaBakimIsEmriId and MakinaBakimIEHizmet.Kapali = 0 and MakinaBakimIEUrun.Opsiyonel = 0
        into :KayitSayisi;
  end
  else
  begin
    select sum(Miktar)
        from MakinaBakimIEUrun
        left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
        where MakinaBakimIEHizmet.MakinaBakimIsEmriId = :AMakinaBakimIsEmriId and MakinaBakimIEHizmet.Kapali = 0
        into :KayitSayisi;
  end

  if (KayitSayisi is Null) then KayitSayisi = 0;

  if (KayitSayisi > 0) then
  begin
    insert into SatinAlmaTalep(SatinAlmaTalepKayitId, SatinAlmaTalepYil, SatinAlmaTalepId, TalepKullaniciId, Tarih, DepartmanId,
        SatinAlmaTalepYeriId, SadeceFiyatArastirmasi, SatinAlmaBirimId, IstekYeri, VerilenSiparisTipiId,
        OnayDurumu, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
        values(
        :SatinAlmaTalepKayitId, :SatinAlmaTalepYil, :SatinAlmaTalepId, :AKullaniciId, current_date, :BakimSatinAlmaDepartmanId,
        :BakimSatinAlmaTalepYeriId, 0, :BakimSatinAlmaBirimId, 1, 3,
        0, :AKullaniciId, current_timestamp(0), :AKullaniciId, current_timestamp(0));

    if (ASadeceZorunlular = 1) then
    begin
      for select MakinaBakimIEUrun.UrunId, Urun.Adi, sum(Miktar)
          from MakinaBakimIEUrun
          left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
          left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
          where MakinaBakimIEHizmet.MakinaBakimIsEmriId = :AMakinaBakimIsEmriId and MakinaBakimIEHizmet.Kapali = 0 and MakinaBakimIEUrun.Opsiyonel = 0
          group by MakinaBakimIEUrun.UrunId, Urun.Adi, MakinaBakimIEUrun.Opsiyonel
          into :UrunId, :UrunAdi, :KayitSayisi do
      begin
        select first 1 UrunBirimId from UrunBirim where UrunBirim.UrunId = :UrunId and DefaultBirim = 1 into :DefaultBirimId;
        insert into SatinAlmaTalepDetay(SatinAlmaTalepKayitId, VerilenSiparisTipiId, UrunId,  Miktar, UrunBirimId, DovizId,
          TerminTarihi, SatinAlmaKullanimYeriId, SiparisiVerilenMiktar, TalepNotu,
          Notu, Kapali, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
          values(:SatinAlmaTalepKayitId, 3, :UrunId, :KayitSayisi, :DefaultBirimId, :SistemDovizId,
          :IsEmriTarihi, :BakimSatinAlmaKullanimYeriId, 0, :AMakinaBakimIsEmriId || ' nolu bakım iş emri talebi',
          :AMakinaBakimIsEmriId || ' nolu bakım iş emri talebi', 0, :AKullaniciId, current_timestamp(0), :AKullaniciId, current_timestamp(0));
        update MakinaBakimIsEmri set SatinAlmaTalepKayitId = :SatinAlmaTalepKayitId where MakinaBakimIsEmriId = :AMakinaBakimIsEmriId;
      end
    end
    else
    begin
      for select MakinaBakimIEUrun.UrunId, Urun.Adi, sum(Miktar)
          from MakinaBakimIEUrun
          left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIEHizmetId = MakinaBakimIEUrun.MakinaBakimIEHizmetId
          left join Urun on Urun.UrunId = MakinaBakimIEUrun.UrunId
          where MakinaBakimIEHizmet.MakinaBakimIsEmriId = :AMakinaBakimIsEmriId and MakinaBakimIEHizmet.Kapali = 0 and MakinaBakimIEUrun.Opsiyonel = 0
          group by MakinaBakimIEUrun.UrunId, Urun.Adi, MakinaBakimIEUrun.Opsiyonel
          into :UrunId, :UrunAdi, :KayitSayisi do
      begin
        select first 1 UrunBirimId from UrunBirim where UrunBirim.UrunId = :UrunId and DefaultBirim = 1 into :DefaultBirimId;
        insert into SatinAlmaTalepDetay(SatinAlmaTalepKayitId, VerilenSiparisTipiId, UrunId,  Miktar, UrunBirimId, DovizId,
          TerminTarihi, SatinAlmaKullanimYeriId, SiparisiVerilenMiktar, TalepNotu,
          Notu, Kapali, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
          values(:SatinAlmaTalepKayitId, 3, :UrunId, :KayitSayisi, :DefaultBirimId, :SistemDovizId,
          :IsEmriTarihi, :BakimSatinAlmaKullanimYeriId, 0, :AMakinaBakimIsEmriId || ' nolu bakım iş emri talebi',
          :AMakinaBakimIsEmriId || ' nolu bakım iş emri talebi', 0, :AKullaniciId, current_timestamp(0), :AKullaniciId, current_timestamp(0));
        update MakinaBakimIsEmri set SatinAlmaTalepKayitId = :SatinAlmaTalepKayitId where MakinaBakimIsEmriId = :AMakinaBakimIsEmriId;
      end
    end
  end
  Suspend;
end
