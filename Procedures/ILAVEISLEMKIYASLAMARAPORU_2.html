CREATE PROCEDURE ILAVEISLEMKIYASLAMARAPORU_2(
    ABASLANGICACILISTARIHI TIMESTAMP,
    ABITISACILISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AISLEMGRUBU INTEGER,
    AISLEMTIPI INTEGER,
    AFASON INTEGER,
    AILAVEISLEMLERID INTEGER,
    RAPORTIPI INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ILAVEISLEMLERID INTEGER,
    ADI VARCHAR(30),
    SIPARISDEVAR INTEGER,
    PARTIDEVAR INTEGER,
    SIPARISSTR VARCHAR(3),
    PARTISTR VARCHAR(3))
AS
begin
  for select Parti.PartiKayitId, PartiStok.PartiSinifId, PartiStok.BoyaSiparisDetayId from Parti
  left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  where (Parti.AcilisTarihi >= :ABaslangicAcilisTarihi and Parti.AcilisTarihi <= :ABitisAcilisTarihi ) and
        (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
        (BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil or :ABoyaSiparisYil is null) and
        (BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId or :ABoyaSiparisId is null) and
        (:AIslemGrubu is null or Parti.IslemGrubuId = :AIslemGrubu) and
        (:AIslemTipi is null or Parti.IslemTipiId = :AIslemTipi) and
        (Parti.Fason = :AFason or :AFason is null)
  into :PartiKayitId, :PartiSinifId, :BoyaSiparisDetayId do
  begin
    for Select IlaveIslemler.IlaveIslemlerId, IlaveIslemler.Adi from IlaveIslemler
      where (IlaveIslemler.IlaveIslemlerID = :AIlaveIslemlerId or :AIlaveIslemlerId is null)
      into :IlaveIslemlerId, :Adi do
    begin
      Select Count(*) from PartiIlaveIslemler
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiIlaveIslemler.PartiAsamalariId
      where PartiIlaveIslemler.PartiKayitId = :PartiKayitId and PartiIlaveIslemler.IlaveIslemlerId = :IlaveIslemlerId and
            PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1
       into :PartideVar;

      Select Count(*) from SiparisIlaveIslemler
      where SiparisIlaveIslemler.BoyaSiparisDetayId = :BoyaSiparisDetayId and SiparisIlaveIslemler.IlaveIslemlerId = :IlaveIslemlerId
       into :SiparisdeVar;

      if (SiparisdeVar > 0 or Partidevar > 0) then
      begin
        if (SiparisdeVar > 0) then SiparisStr = 'Var';
        else SiparisStr = '';

        if (PartideVar > 0) then PartiStr = 'Var';
        else PartiStr = '';
        suspend;
      end
    end
  end
end
