CREATE PROCEDURE DOKUMAMAKINACALISMARAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ADOKUMAMAKINAID MAKINAKODUSIZE,
    AMAKINAGRUPID VARCHAR(4))
RETURNS (
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    SURE DOUBLE PRECISION,
    SURESTR VARCHAR(30),
    PLANLAMALICALISMASURESI DOUBLE PRECISION,
    PLANLAMALICALISMASURESISTR VARCHAR(30),
    PLANLAMALICALISMASURESIORANI DOUBLE PRECISION,
    PLANLAMASIZCALISMASURESI DOUBLE PRECISION,
    PLANLAMASIZCALISMASURESISTR VARCHAR(30),
    PLANLAMASIZCALISMASURESIORANI DOUBLE PRECISION,
    TOPLAMCALISMASURESI DOUBLE PRECISION,
    TOPLAMCALISMASURESISTR VARCHAR(30),
    TOPLAMCALISMASURESIORANI DOUBLE PRECISION,
    DURUSSURESI DOUBLE PRECISION,
    DURUSSURESISTR VARCHAR(30),
    DURUSSURESIORANI DOUBLE PRECISION,
    ENYUKSEKTUR INTEGER,
    ENYUKSEKTURSURESI DOUBLE PRECISION,
    ENYUKSEKTURSURESISTR VARCHAR(30),
    ARIZASURESI DOUBLE PRECISION,
    ARIZASURESISTR VARCHAR(30),
    ARIZASURESIORANI DOUBLE PRECISION,
    SERVISSURESI DOUBLE PRECISION,
    SERVISSURESISTR VARCHAR(30),
    SERVISSURESIORANI DOUBLE PRECISION,
    AYARSURESI DOUBLE PRECISION,
    AYARSURESISTR VARCHAR(30),
    AYARSURESIORANI DOUBLE PRECISION,
    LEVENTSONUSURESI DOUBLE PRECISION,
    LEVENTSONUSURESISTR VARCHAR(30),
    LEVENTSONUSURESIORANI DOUBLE PRECISION,
    LEVENTSONUSAYISI INTEGER,
    ATKIKOPTUSURESI DOUBLE PRECISION,
    ATKIKOPTUSURESISTR VARCHAR(30),
    ATKIKOPTUSURESIORANI DOUBLE PRECISION,
    ATKIKOPTUSAYISI INTEGER,
    COZGUKOPTUSURESI DOUBLE PRECISION,
    COZGUKOPTUSURESISTR VARCHAR(30),
    COZGUKOPTUSURESIORANI DOUBLE PRECISION,
    COZGUKOPTUSAYISI INTEGER,
    STARTSTOPSURESI DOUBLE PRECISION,
    STARTSTOPSURESISTR VARCHAR(30),
    STARTSTOPSURESIORANI DOUBLE PRECISION,
    STARTSTOPSAYISI INTEGER,
    TOPKESMESURESI DOUBLE PRECISION,
    TOPKESMESURESISTR VARCHAR(30),
    TOPKESMESURESIORANI DOUBLE PRECISION,
    TOPLAMTUR INTEGER,
    KAPALI INTEGER,
    VERIM DOUBLE PRECISION,
    MAKINAHIZI INTEGER)
AS
declare variable Tarih  TimeStamp;
declare variable Tur integer;
declare variable DokumaPlanlamaId integer;

declare variable LastTopKesme         integer;
declare variable LastTarih            TimeStamp;
declare variable LastTur              integer;
declare variable LastDokumaPlanlamaId integer;
declare variable FarkTarih            Double Precision;
declare variable Sanal                integer;
declare variable LastAriza            integer;
declare variable LastServis           integer;
declare variable LastAyar             integer;
declare variable Ariza                integer;
declare variable Servisci             integer;
declare variable Ayar                 integer;

declare variable LastLeventSonu  SmallInt;
declare variable LastAtkiKoptu   SmallInt;
declare variable LastCozguKoptu  SmallInt;
declare variable LastStartStop   SmallInt;
declare variable LeventSonu      SmallInt;
declare variable AtkiKoptu       SmallInt;
declare variable CozguKoptu      SmallInt;
declare variable StartStop       SmallInt;

begin
  if (ABitisTarihi > Current_TimeStamp(0)) then ABitisTarihi = Current_TimeStamp(0);

  Sure = ABitistarihi - ABaslangicTarihi;
  if (Sure > 0) then SureStr = TarihToStr(Sure);

  For Select DokumaMakinaId, Adi, DokumaMakina.Kapali, MakinaHizi from DokumaMakina
    where (DokumaMakina.DokumaMakinaId = :ADokumaMakinaId or :ADokumaMakinaId is null) and
          (DokumaMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and DokumaMakina.OtomasyonaDahil = 1
    into :DokumaMakinaId, :DokumaMakinaAdi, :Kapali, :MakinaHizi do
  begin
    PlanlamaliCalismaSuresi       = 0;
    PlanlamaliCalismaSuresiOrani  = 0;
    PlanlamasizCalismaSuresi      = 0;
    PlanlamasizCalismaSuresiOrani = 0;
    ToplamCalismaSuresi           = 0;
    ToplamCalismaSuresiOrani      = 0;
    DurusSuresi                   = 0;
    DurusSuresiOrani              = 0;
    EnYuksekTur                   = 0;
    EnYuksekTurSuresi             = 0;
    ArizaSuresi                   = 0;
    ArizaSuresiOrani              = 0;
    ServisSuresi                   = 0;
    ServisSuresiOrani              = 0;
    AyarSuresi                   = 0;
    AyarSuresiOrani              = 0;

    TopKesmeSuresi      = 0;
    TopKesmeSuresiOrani = 0;
    LeventSonuSuresi      = 0;
    LeventSonuSuresiOrani = 0;
    AtkiKoptuSuresi          = 0;
    AtkiKoptuSuresiOrani     = 0;
    CozguKoptuSuresi        = 0;
    CozguKoptuSuresiOrani   = 0;
    StartStopSuresi          = 0;
    StartStopSuresiOrani     = 0;


    LastTur = -1;
    LastDokumaPlanlamaId = -1;
    LastTarih = null;
    LastAriza = -1;
    LastServis = -1;
    LastAyar = -1;
    LastLeventSonu = -1;
    LastAtkiKoptu     = -1;
    LastCozguKoptu   = -1;
    LastStartStop     = -1;

    ToplamTur = 0;
    Verim = 0;
    AtkiKoptuSayisi = 0;
    CozguKoptuSayisi = 0;
    LeventSonuSayisi = 0;
    StartStopSayisi = 0;


    For select Tarih, Tur, DokumaPlanlamaId, Sanal, Ariza, Servisci, Ayar, LeventSonu, AtkiKoptu, CozguKoptu, StartStop from MakinaCalismaBilgisi(Null, :ABaslangicTarihi, :ABitisTarihi, :DokumaMakinaId) order by Tarih
         into :Tarih, :Tur, :DokumaPlanlamaId, :Sanal, :Ariza, :Servisci, :Ayar, :LeventSonu, :AtkiKoptu, :CozguKoptu, :StartStop do

    begin

      ToplamTur = ToplamTur + Tur;

      if (LastTarih is null) then LastTarih = Tarih;
      if (LastTur = -1) then LastTur = Tur;
      FarkTarih = Tarih - LastTarih;

      if (Sanal = 0) then
      if (Tur > EnYuksekTur) then EnYuksekTur = Tur;

      if (Tur > 0 and LeventSonu = 0 and CozguKoptu = 0 and AtkiKoptu = 0 and StartStop = 0) then Tur = 1;
      else Tur = 0;

      if (LastTur = -1) then LastTur = Tur;
      if (LastAriza = -1) then LastAriza = Ariza;
      if (LastServis = -1) then LastServis = Servisci;
      if (LastAyar = -1) then LastAyar = Ayar;
      if (LastLeventSonu = -1) then LastLeventSonu = LeventSonu;
      if (LastAtkiKoptu = -1) then LastAtkiKoptu = AtkiKoptu;
      if (LastCozguKoptu = -1) then LastCozguKoptu = CozguKoptu;
      if (LastStartStop = -1) then LastStartStop = StartStop;

      if (LastTur = 0) then DurusSuresi = DurusSuresi + FarkTarih;
      else if (LastTur = 1) then
      begin
        if (DokumaPlanlamaId is null) then PlanlamasizCalismaSuresi = PlanlamasizCalismaSuresi + FarkTarih;
        else PlanlamaliCalismaSuresi = PlanlamaliCalismaSuresi + FarkTarih;

        ToplamCalismaSuresi = ToplamCalismaSuresi + FarkTarih;
      end

      if (LastAriza = 1) then ArizaSuresi = ArizaSuresi + FarkTarih;
      if (LastServis = 1) then ServisSuresi = ServisSuresi + FarkTarih;
      if (LastAyar = 1) then AyarSuresi = AyarSuresi + FarkTarih;

      if (LastTopKesme = 1) then TopKesmeSuresi = TopKesmeSuresi + FarkTarih;
      if (LastLeventSonu = 1) then LeventSonuSuresi = LeventSonuSuresi + FarkTarih;
      if (LastAtkiKoptu = 1) then AtkiKoptuSuresi = AtkiKoptuSuresi + FarkTarih;
      if (LastCozguKoptu = 1) then CozguKoptuSuresi = CozguKoptuSuresi + FarkTarih;
      if (LastStartStop = 1 and LastLeventSonu = 0 and LastAtkiKoptu = 0 and LastCozguKoptu = 0) then StartStopSuresi = StartStopSuresi + FarkTarih;

      if (LastAtkiKoptu = 0 and AtkiKoptu = 1) then
        AtkiKoptuSayisi = AtkiKoptuSayisi + 1;
      if (LastCozguKoptu = 0 and CozguKoptu = 1) then
        CozguKoptuSayisi = CozguKoptuSayisi + 1;
      if (LastLeventSonu = 0 and LeventSonu = 1) then
        LeventSonuSayisi = LeventSonuSayisi + 1;
      if (LastStartStop = 0 and StartStop = 1) then
        StartStopSayisi = StartStopSayisi + 1;

      LastTarih = Tarih;
      LastTur = Tur;
      LastAriza = Ariza;
      LastServis = Servisci;
      LastAyar = Ayar;
      LastLeventSonu = LeventSonu;
      LastAtkiKoptu = AtkiKoptu;
      LastCozguKoptu = CozguKoptu;
      LastStartStop = StartStop;

    end

    if (Sure > 0) then
    begin
      PlanlamaliCalismaSuresiOrani  = (PlanlamaliCalismaSuresi / Sure) * 100;
      PlanlamasizCalismaSuresiOrani = (PlanlamasizCalismaSuresi/ Sure) * 100;
      ToplamCalismaSuresiOrani = (ToplamCalismaSuresi / Sure) * 100;
      DurusSuresiOrani         = (DurusSuresi / Sure) * 100;
      ArizaSuresiOrani         = (ArizaSuresi / Sure) * 100;
      ServisSuresiOrani        = (ServisSuresi / Sure) * 100;
      AyarSuresiOrani          = (AyarSuresi / Sure) * 100;
      TopKesmeSuresiOrani = (TopKesmeSuresi / Sure) * 100;
      LeventSonuSuresiOrani = (LeventSonuSuresi / Sure) * 100;
      AtkiKoptuSuresiOrani     = (AtkiKoptuSuresi / Sure) * 100;
      CozguKoptuSuresiOrani   = (CozguKoptuSuresi / Sure) * 100;
      StartStopSuresiOrani     = (StartStopSuresi / Sure) * 100;

      PlanlamaliCalismaSuresiOrani = Yuvarla(PlanlamaliCalismaSuresiOrani, 100);
      PlanlamasizCalismaSuresiOrani = Yuvarla(PlanlamasizCalismaSuresiOrani, 100);
      ToplamCalismaSuresiOrani = Yuvarla(ToplamCalismaSuresiOrani, 100);
      DurusSuresiOrani = Yuvarla(DurusSuresiOrani, 100);
      ArizaSuresiOrani = Yuvarla(ArizaSuresiOrani, 100);
      ServisSuresiOrani = Yuvarla(ServisSuresiOrani, 100);
      AyarSuresiOrani = Yuvarla(AyarSuresiOrani, 100);
      TopKesmeSuresiOrani = Yuvarla(TopKesmeSuresiOrani, 100);
      LeventSonuSuresiOrani = Yuvarla(LeventSonuSuresiOrani, 100);
      AtkiKoptuSuresiOrani     = Yuvarla(AtkiKoptuSuresiOrani, 100);
      CozguKoptuSuresiOrani   = Yuvarla(CozguKoptuSuresiOrani, 100);
      StartStopSuresiOrani     = Yuvarla(StartStopSuresiOrani, 100);

      Verim = (ToplamTur / (Sure*24*60)) * 100 / MakinaHizi;
    end

    if (PlanlamaliCalismaSuresi = 0) then PlanlamaliCalismaSuresiStr = '';
    else PlanlamaliCalismaSuresiStr = TarihToStr(PlanlamaliCalismaSuresi);

    if (PlanlamasizCalismaSuresi = 0) then PlanlamasizCalismaSuresiStr = '';
    else PlanlamasizCalismaSuresiStr = TarihToStr(PlanlamasizCalismaSuresi);

    if (ToplamCalismaSuresi = 0) then ToplamCalismaSuresiStr = '';
    else ToplamCalismaSuresiStr = TarihToStr(ToplamCalismaSuresi);

    if (DurusSuresi = 0) then DurusSuresiStr = '';
    else DurusSuresiStr = TarihToStr(DurusSuresi);

    if (ArizaSuresi = 0) then ArizaSuresiStr = '';
    else ArizaSuresiStr = TarihToStr(ArizaSuresi);

    if (ServisSuresi = 0) then ServisSuresiStr = '';
    else ServisSuresiStr = TarihToStr(ServisSuresi);

    if (AyarSuresi = 0) then AyarSuresiStr = '';
    else AyarSuresiStr = TarihToStr(AyarSuresi);

    if (TopKesmeSuresi = 0) then TopKesmeSuresiStr = '';
    else TopKesmeSuresiStr = TarihToStr(TopKesmeSuresi);

    if (LeventSonuSuresi = 0) then LeventSonuSuresiStr = '';
    else LeventSonuSuresiStr = TarihToStr(LeventSonuSuresi);

    if (AtkiKoptuSuresi = 0) then AtkiKoptuSuresiStr = '';
    else AtkiKoptuSuresiStr = TarihToStr(AtkiKoptuSuresi);

    if (CozguKoptuSuresi = 0) then CozguKoptuSuresiStr = '';
    else CozguKoptuSuresiStr = TarihToStr(CozguKoptuSuresi);

    if (StartStopSuresi = 0) then StartStopSuresiStr = '';
    else StartStopSuresiStr = TarihToStr(StartStopSuresi);

    suspend;
  end
end
