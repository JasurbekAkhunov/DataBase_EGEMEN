CREATE PROCEDURE DESENVARYANTSABLONRECETEGETIR(
    ADESENID VARCHAR(20),
    ADESENVARYANTID INTEGER)
RETURNS (
    DESENVARYANTID INTEGER,
    DESENVARYANTSABLONID INTEGER,
    DESENVARYANTSABLONRECETEID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNBIRIMID INTEGER,
    BIRIMID VARCHAR(2),
    MIKTAR DOUBLE PRECISION,
    URETIMECIKISFIYATI DOUBLE PRECISION,
    URETIMECIKISFIYATIBIRIMID VARCHAR(2),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable DovizKuru                   Double Precision;
declare variable FUretimeCikisFiyati   Double precision;
declare variable FUretimeCikisFiyatiDovizId  varchar(5);
declare variable StokTabanDovizId               varchar(5);
begin
  select StokTabanDovizId from SISTEM where SISTEMID = 1 into :StokTabanDovizId;

  for select DesenVaryantSablonRecete.DesenVaryantId, DesenVaryantSablonRecete.DesenVaryantSablonId, DesenVaryantSablonRecete.DesenVaryantSablonReceteId, DesenVaryantSablonRecete.UrunId, Urun.Adi,
             DesenVaryantSablonRecete.UrunBirimId, UrunBirim.BirimId, DesenVaryantSablonRecete.Miktar,
             IK.Adi, DesenVaryantSablonRecete.InsertTarihi, SK.Adi, DesenVaryantSablonRecete.DegisimTarihi,
             UrunFiyat.UretimeCikisFiyati, UrunFiyat.UretimeCikisFiyatiDovizId, UBUCF.BirimId from DesenVaryant
  left join DesenVaryantSablonRecete on DesenVaryantSablonRecete.DesenVaryantID = DesenVaryant.DesenVaryantId
  left join Urun on Urun.UrunId = DesenVaryantSablonRecete.UrunId
  left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
  left join UrunBirim on UrunBirim.UrunBirimId = DesenVaryantSablonRecete.UrunBirimId and UrunBirim.UrunId = DesenVaryantSablonRecete.UrunId
  left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
  left join Kullanici IK on IK.KullaniciId = DesenVaryantSablonRecete.KullaniciId
  left join Kullanici SK on SK.KullaniciId = DesenVaryantSablonRecete.KullaniciId
  where DesenVaryant.DesenID = :ADesenID and (DesenVaryant.DesenVaryantId = :ADesenVaryantId or :ADesenVaryantId is null) and
        DesenVaryantSablonRecete.DesenVaryantID is not null
  order by DesenVaryantSablonRecete.DesenVaryantSablonId
  into :DesenVaryantId, :DesenVaryantSablonId, :DesenVaryantSablonReceteId, :UrunId, :UrunAdi, :UrunBirimId, :BirimId, :Miktar,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :FUretimeCikisFiyati, :FUretimeCikisFiyatiDovizId, :UretimeCikisFiyatiBirimId do
  begin

    if (FUretimeCikisFiyatiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :StokTabanDovizId, Current_Date) into :DovizKuru;
    else DovizKuru = 1;

    if (DovizKuru is null) then DovizKuru = 0;
    UretimeCikisFiyati = FUretimeCikisFiyati * DovizKuru;

    suspend;
  end
end
