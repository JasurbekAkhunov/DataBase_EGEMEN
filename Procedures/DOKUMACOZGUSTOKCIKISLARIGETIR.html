CREATE PROCEDURE DOKUMACOZGUSTOKCIKISLARIGETIR(
    ADOKUMAPLANLAMAID INTEGER)
RETURNS (
    ORMERAFID INTEGER,
    RAFID RAFKODUSIZE,
    LEVENTNO INTEGER,
    ISEMRIID VARCHAR(15),
    MIKTAR BOYAMIKTARSIZE)
AS
declare variable shtFasonUretimeCikis  integer;
declare variable shtUretimeCikis       integer;
begin
  Select shtFasonUretimeCikis, shtUretimeCikis from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis, :shtUretimeCikis;

  for select OrmeRaf.OrmeRafId, OrmeRaf.RafId, OrmeRaf.LeventNo, CozguIsEmri.IsEmriId, StokHareket.Miktar from StokHareket
      join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
      left join CozguIsEmri on CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId
      join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
      join Depo on Depo.DepoId = DepoRaf.DepoId
      where StokHareket.DokumaHareketi = 1 and StokHareket.HamIade = 0 and StokHareket.GirisCikis = -1 and
      (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis) and
      StokHareket.DokumaPlanlamaId = :ADokumaPlanlamaId and
      (Depo.CozguDeposu = 1 or Depo.HasilDeposu = 1 or Depo.IndigoDeposu = 1)
      order by StokHareket.InsertTarihi desc into :OrmeRafId, :RafId, :LeventNo, :IsEmriId, :Miktar do
   suspend;
end
