CREATE PROCEDURE HAMSTOKGECICIKABULRAFKARTI(
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    HAMSTOKMASTERID INTEGER,
    GECICIKABULRAFID INTEGER,
    HAMSTOKIRSALIYEID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RAFID VARCHAR(10),
    KAPADEDI INTEGER,
    MIKTAR NUMERIC(12,3),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select GeciciKabulRaf.GeciciKabulRafId, GeciciKabulRaf.HamStokIrsaliyeId, GeciciKabulRaf.HamStokMasterId, GeciciKabulRaf.HamId, Ham.Adi,
             GeciciKabulRaf.RafId, GeciciKabulRaf.Miktar, GeciciKabulRaf.KapAdedi, GeciciKabulRaf.Kilo, GeciciKabulRaf.Metre, GeciciKabulRaf.Adet,
             IK.Adi, GeciciKabulRaf.InsertTarihi, Sk.Adi, GeciciKabulRaf.DegisimTarihi FROM HamStokMaster
    left join GeciciKabulRaf on GeciciKabulRaf.HamStokMasterId = HamStokMaster.HamStokMasterId
    left join Ham on Ham.HamId = GeciciKabulRaf.HamId
    Left Join Kullanici IK on IK.KullaniciId = GeciciKabulRaf.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = GeciciKabulRaf.KullaniciId
    Where HamStokMaster.HamStokMasterId = :AHamStokMasterId and GeciciKabulRaf.GeciciKabulRafId is not null
    into :GeciciKabulRafId, :HamStokIrsaliyeId, :HamStokMasterId, :HamId, :HamAdi, :RafId, :Miktar, :KapAdedi, :Kilo, :Metre, :Adet,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    suspend;
  end
end
