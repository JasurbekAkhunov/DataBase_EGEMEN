CREATE PROCEDURE SARIMACIKISICINTAMIRSTOKLISTESI
RETURNS (
    IPLIKLOTID INTEGER,
    YENILOT VARCHAR(30),
    TAMIRGIRISKAYITID INTEGER,
    TAMIRGIRISYIL INTEGER,
    TAMIRGIRISID INTEGER,
    TAMIRGIRISIDSTR VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    ESKILOT VARCHAR(30),
    ESKIPARTIYIL INTEGER,
    ESKIPARTINO INTEGER,
    ESKIPARTINOSTR VARCHAR(30),
    IRSALIYETARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IPLIKTAMIRTIPIID INTEGER,
    IPLIKTAMIRTIPIADI VARCHAR(20),
    TAMIRGIRISMIKTAR NUMERIC(12,3),
    SARIMACIKANMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3))
AS
begin
  For Select IplikLot.IplikLotId, IplikLot.Lot, TamirGiris.TamirGirisKayitId, TamirGiris.TamirGirisYil, TamirGiris.TamirGirisId, TamirGiris.HamId,
             Ham.Adi, IplikSatisIade.RenkId, Renk.Adi, IplikLot.MelanjKodu, IplikLot.FlamKodu, IplikLot.EskiLot, IplikIrsaliye.FirmaId,
             Firma.Adi, IplikSatisIade.IplikTamirTipiId, IplikTamirTipi.Adi,
             IplikSatisIade.EskiPartiYil, IplikSatisIade.EskiPartiNo, IplikIrsaliye.IrsaliyeTarihi,
             TamirGiris.Miktar, TamirGiris.SarimaCikanMiktar,
             TamirGiris.Miktar - TamirGiris.SarimaCikanMiktar from TamirGiris
    left join IplikLot on IplikLot.TamirGirisKayitId = TamirGiris.TamirGirisKayitId
    left join IplikSatisIade on IplikSatisIade.IplikSatisIadeId = TamirGiris.IplikSatisIadeId
    left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikSatisIade.IplikIrsaliyeId
    left join Ham on Ham.HamId = TamirGiris.HamId
    left join Renk on Renk.RenkId = IplikSatisIade.RenkId
    left join Firma on Firma.FirmaId = IplikIrsaliye.FirmaId
    left join IplikTamirTipi on IplikTamirTipi.IplikTamirTipiID = IplikSatisIade.IplikTamirTipiId
    where TamirGiris.StokVar = 1
    into :IplikLotId, :YeniLot, :TamirGirisKayitId, :TamirGirisYil, :TamirGirisId, :HamId, :HamAdi, :RenkId, :RenkAdi, :MelanjKodu, :FlamKodu, :EskiLot, :FirmaId,
         :FirmaAdi, :IplikTamirTipiId, :IplikTamirTipiAdi, :EskiPartiYil, :EskiPartiNo, :IrsaliyeTarihi,
         :TamirGirisMiktar, :SarimaCikanMiktar, :KalanMiktar do
    begin
      TamirGirisIdStr = '(' || EgemenIntegerToStr(TamirGirisYil) || ') '  || EgemenIntegerToStr(TamirGirisId)  || ' ';
      if (EskiPartiNo is not null) then EskiPartiNoStr = '(' || EgemenIntegerToStr(EskiPartiYil)  || ') ' || EgemenIntegerToStr(EskiPartiNo) || ' ';
      else EskiPartiNoStr = '';

      if (KalanMiktar > 0) then suspend;
    end
end
