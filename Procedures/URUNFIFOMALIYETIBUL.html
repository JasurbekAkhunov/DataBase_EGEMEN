CREATE PROCEDURE URUNFIFOMALIYETIBUL(
    URUNID VARCHAR(20),
    BIRIMCARPAN INTEGER,
    TARIH TIMESTAMP,
    DOVIZID VARCHAR(5),
    MIKTAR DOUBLE PRECISION,
    STOKDETAYCARPANI DOUBLE PRECISION)
RETURNS (
    FIFOORTALAMAMALIYET DOUBLE PRECISION,
    FIFOORTALAMAMALIYET_ASIL DOUBLE PRECISION)
AS
declare variable CikisMiktari           double precision;
declare variable KalanMaliyet     double precision;
declare variable KalanMiktar     double precision;
declare variable ToplamKalanMaliyet     double precision;
declare variable ToplamKalanMiktar     double precision;
declare variable FMiktar           double precision;
declare variable NetBirimFiyat           double precision;
declare variable F1 Double precision;
begin
  ToplamKalanMaliyet = 0;
  /*ToplamKalanMiktar = 0;*/
  if (Miktar is null) then Miktar = 0;
  CikisMiktari = Miktar;
  ToplamKalanMiktar = CikisMiktari;

  for Select NetBirimFiyat, KalanMiktar from UrunFifoMaliyetiGetir(:UrunId, :BirimCarpan, :Tarih, :DovizId)
    into :NetBirimFiyat, :FMiktar do
  begin
    if (CikisMiktari >= FMiktar) then
    begin
      KalanMaliyet = FMiktar * NetBirimFiyat;
      ToplamKalanMaliyet = ToplamKalanMaliyet + KalanMaliyet;
      CikisMiktari = CikisMiktari - FMiktar;
    end
    else
    begin
      KalanMaliyet = CikisMiktari * NetBirimFiyat;
      ToplamKalanMaliyet = ToplamKalanMaliyet + KalanMaliyet;
      CikisMiktari = 0;
    end
    /*ToplamKalanMiktar = ToplamKalanMiktar + FMiktar;*/
  end

  if (ToplamKalanMiktar <> 0) then
    FifoOrtalamaMaliyet_Asil = ToplamKalanMaliyet / ToplamKalanMiktar;
  else
    FifoOrtalamaMaliyet_Asil = 0;

  F1 = BirimCarpan / StokDetayCarpani;

  if (BirimCarpan <> StokDetayCarpani) then
     FifoOrtalamaMaliyet = FifoOrtalamaMaliyet_Asil  * F1;
  else
     FifoOrtalamaMaliyet = FifoOrtalamaMaliyet_Asil;

  suspend;
end
