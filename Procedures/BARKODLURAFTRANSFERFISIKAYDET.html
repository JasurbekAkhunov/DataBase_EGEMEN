CREATE PROCEDURE BARKODLURAFTRANSFERFISIKAYDET(
    ARAFTRANSFERFISIID INTEGER,
    ABARKODID VARCHAR(30),
    AHEDEFRAFID RAFKODUSIZE,
    AKULLANICIID INTEGER)
AS
Declare Variable  OrmeKaliteKontrolID integer;
Declare Variable  StokHareketDetayId Integer;
Declare Variable  Miktar BoyaMiktarSize;
Declare Variable  Kilo BoyaMiktarSize;
Declare Variable  Brutkilo BoyaMiktarSize;
Declare Variable  Metre BoyaMiktarSize;
Declare Variable  KaliteNo integer;
Declare Variable  OrmeRafId integer;
Declare Variable  KapAdedi Integer;
Declare Variable  FOrmeRafTransferFisiDetayId Integer;
begin
    Select first 1 OrmeKaliteKontrol.OrmeKaliteKontrolID, null, OrmeRaf.OrmeRafId, case when OrmeKaliteKontrol.Sakat = 1 then 2 else 1 end,
    OrmeKaliteKontrol.Miktar, OrmeKaliteKontrol.Miktar, OrmeKaliteKontrol.Metre, 1 from OrmeKaliteKontrol
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeKaliteKontrol.OrmePlanlamaId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
    left join ORmeRaf on ORmeRaf.Rafid = OrmeKaliteKontrol.RafID and OrmeRaf.Ormeplanlamaid = OrmePlanlama.Ormeplanlamaid
    and OrmeRaf.OrmeMakinaId = OrmeMakina.OrmeMakinaId and ((OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari) >= OrmeKaliteKontrol.Miktar)
    where OrmeKaliteKontrol.OrmeKaliteKontrolId = :ABarkodId and OrmeKaliteKontrol.SatisStokHareketId is null and OrmeKaliteKontrol.UretimeCikisStokHareketId is null
    into :OrmeKaliteKontrolID, :StokHareketDetayId, :OrmeRafId, :KaliteNo, :Kilo, :BrutKilo, :Metre, :KapAdedi;

    if (OrmeKaliteKontrolID is null) then
    begin
        Select First 1 null, StokHareketDetay.StokHareketDetayId, OrmeRaf.OrmeRafID, OrmeRaf.KaliteNo, (GirisKilo - CikisKilo), (GirisBrutKilo - CikisBrutKilo),
        (GirisMetre - CikisMetre), (StokHareketDetayMiktar.GirisKapAdedi - StokHareketDetayMiktar.CikisKapAdedi) from StokHareketDetay
        Left Join StokHareketDetayMiktar on StokHareketDetayMiktar.Barkod = StokHareketDetay.Barkod
        Left Join OrmeRaf on OrmeRaf.OrmeRafID = StokHareketDetay.OrmeRafID
        where StokHareketDetay.Barkod = :Abarkodid and (StokHareketDetay.HareketTipi in (1, 8, 9)) and (IIF(EgemenStrToIntOlurMu(:Abarkodid) = 1, 1, IIF((GirisMetre - CikisMetre) > 0, 1, 0)) = 1)
        Into :OrmeKaliteKontrolID, :StokHareketDetayId, :OrmeRafId, :KaliteNo, :Kilo, :BrutKilo, :Metre, :KapAdedi;

        if (StokHareketDetayId is null) then exception BarkodHatali2;

        Select First 1 OrmeRafTransferFisiDetay.YeniOrmeRafID from OrmeRafTransferFisiDetay where OrmeRafTransferFisiDetay.StokHareketDetayId = :StokHareketDetayId
        Order By InsertTarihi Desc
        into :OrmeRafId;
    end
    else
    begin
        Select First 1 OrmeRafTransferFisiDetay.YeniOrmeRafID from OrmeRafTransferFisiDetay where OrmeRafTransferFisiDetay.OrmeKaliteKontrolID = :OrmeKaliteKontrolID
        Order By InsertTarihi Desc
        into :OrmeRafId;
    end

    FOrmeRafTransferFisiDetayId = gen_ID(OrmeRafTransferFisiDetay_Id_gen, 1);

    Insert Into OrmeRafTransferFisiDetay(OrmeRafTransferFisiDetayId, OrmeRafTransferFisiId, OrmeRafId, Kilo, Brutkilo, KapAdedi, RafId, KaliteNo, InsertKullaniciId, KullaniciId, OrmeKaliteKontrolId, StokHareketDetayId)
                                    Values(:FOrmeRafTransferFisiDetayId, :ARafTransferFisiId, :OrmeRafId, :Kilo, :Brutkilo, :KapAdedi, :AHedefRafId, :KaliteNo, :AKullaniciId, :AKullaniciId, :OrmeKaliteKontrolId, :StokHareketDetayId);

    if (StokHareketDetayId is Not Null) then
      Update StokHareketDetay Set OrmeRafTransferFisiDetayID = :FOrmeRafTransferFisiDetayId where StokHareketDetayID = :StokHareketDetayId;
end
