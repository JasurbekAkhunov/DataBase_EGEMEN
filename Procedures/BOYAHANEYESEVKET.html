Definition

CREATE PROCEDURE BOYAHANEYESEVKET(
    AORMEFISID INTEGER,
    AORMEIRSALIYEID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    STATUS INTEGER,
    FISSERINO VARCHAR(2),
    FISNO INTEGER,
    ORMEFISID INTEGER)
AS
declare variable UretimIslemTipiId      integer;
declare variable uitOrme                integer;
declare variable uitBoya                integer;
declare variable shtUretimeCikis        integer;
declare variable SistemDovizId          varchar(4);
declare variable Aciklama               varchar(1000);
declare variable UretimFisNo            integer;
declare variable UretimFisTarihi        Timestamp;
declare variable UretimIrsaliyeNo       IrsaliyeNoSize;
declare variable UretimIrsaliyeTarihi   Timestamp;
declare variable KS                     integer;
declare variable XLot                   varchar(150);
declare variable XMelanjKodu            varchar(20);
declare variable XBoyaSiparisKayitId    integer;
declare variable XOrmeSiparisDetayId    integer;
declare variable XRenkId                integer;
declare variable XHamId                 integer;
declare variable XOrmeRafId             integer;
declare variable XDepoId                varchar(10);
declare variable XKapAdedi              integer;
declare variable XAdet                  integer;
declare variable XOlcuId                integer;
declare variable XMiktar                Numeric(12,3);
declare variable SatirAciklama          varchar(100);
declare variable FisTarihi              Timestamp;
declare variable AracId                 varchar(11);
declare variable SoforId                integer;
declare variable Sinif                  varchar(1);
declare variable FKumasTeminSiparisi    integer;
declare variable TeminToplamMiktar      Numeric(12,3);
declare variable DagitimMiktar          Numeric(12,3);
declare variable TeminMiktar            Numeric(12,3);
declare variable TBoyaSiparisKayitId    integer;
declare variable ToplamBoyayaCikis      Numeric(12,3);
declare variable OrmeRafKalanMiktar     Numeric(12,3);
declare variable OrmeRafId              Integer;
declare variable FSeriNo                SeriNoSize5;
declare variable FSeriFis               varchar(20);
begin
  select uitOrme, uitBoya from UretimIslemTipiGetir into :uitOrme, :uitBoya;

  Select shtUretimeCikis from OrmeStokHareketTipiGetir into :shtUretimeCikis;

  Select SistemDovizId from Sistem into SistemDovizId;

  if (AOrmeFisID is not null) then
  begin
    Select count(*) from StokHareket where StokHareket.OrmeFisID = :AOrmeFisID and StokHareket.HamIade = 0 into :KS;
    Select UretimIslemTipiID, FisNo, FisTarihi, AracId, SoforId, Sinif, coalesce(SeriNo,'') from OrmeFis where OrmeFis.OrmeFisID = :AOrmeFisID into :UretimIslemTipiID, :UretimFisNo, :UretimFisTarihi, :AracId, :SoforId, :Sinif, :FSeriNo;
    FisTarihi = UretimFisTarihi;

    Aciklama = FSeriNo ||'-'|| EgemenIntegerToStr(UretimFisNo) || ' Nolu ' || TimeStampToDateTimeStr(UretimFisTarihi) || ' Tarihli Fiş''in Boyahaneye Çıkışı';
  end
  else
  begin
    Select count(*) from StokHareket where StokHareket.OrmeIrsaliyeID = :AOrmeIrsaliyeID and StokHareket.HamIade = 0 into :KS;
    Select UretimIslemTipiID, IrsaliyeNo, IrsaliyeTarihi, AracId, SoforId, Sinif, coalesce(SeriNo,'') from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeID = :AOrmeIrsaliyeID into :UretimIslemTipiID, :UretimIrsaliyeNo, :UretimIrsaliyeTarihi, :AracId, :SoforId, :Sinif, :FSeriNo;

    FisTarihi = UretimIrsaliyeTarihi;

    Aciklama = FSeriNo ||'-'|| format_float('',UretimIrsaliyeNo) || ' Nolu ' || TimeStampToDateTimeStr(UretimIrsaliyeTarihi) || ' Tarihli İrsaliye''nin Boyahaneye Çıkışı';
    SatirAciklama = 'Giriş İrsaliye NO =' || format_float('',UretimIrsaliyeNo);
  end

  if (UretimIslemTipiID <> uitOrme) then
  begin
    Status = -1;
    suspend;
    exit;
  end

  if (KS = 0) then
  begin
    Status = -2;
    suspend;
    exit;
  end

  if (AOrmeFisID is not null) then
  begin
    Select count(*) from StokHareket where StokHareket.OrmeFisID = :AOrmeFisID and StokHareket.HamIade = 0 and StokHareket.Miktar <> StokHareket.RaflananMiktar into :KS;
  end
  else
  begin
    Select count(*) from StokHareket where StokHareket.OrmeIrsaliyeID = :AOrmeIrsaliyeID and StokHareket.HamIade = 0 and StokHareket.Miktar <> StokHareket.RaflananMiktar into :KS;
  end

  if (KS > 0) then
  begin
    Status = -3;
    suspend;
    exit;
  end

  Select FisSeriNo from Kullanici where KullaniciId = :AKullaniciId into :FisSeriNo;
  if (FisSeriNo is null) then FisSeriNo = '';

  Select FisNo from FisNumaratorAL(:FisSeriNo) into :FisNo;

  OrmeFisId = gen_id(OrmeFis_Id_Gen, 1);
  insert into OrmeFis(OrmeFisId, HareketTipi, UretimIslemTipiId, SeriNo, FisNo, FisTarihi, DovizId, Notu, InsertKullaniciId, KullaniciId, AracId, SoforId, Sinif)
           values(:OrmeFisId, :shtUretimeCikis, :uitBoya, :FisSeriNo, :FisNo, :FisTarihi, :SistemDovizId, :Aciklama, :AKullaniciId, :AKullaniciId, :AracId, :SoforId, :Sinif);

  if (AOrmeFisID is not null) then
  begin
    For Select StokHareket.BoyaSiparisKayitId, StokHareket.OrmeSiparisDetayId, StokHareket.RenkId, StokHareket.HamId, OrmeRafDetay.OrmeRafId,
               OrmeRafDetay.KapAdedi, OrmeRafDetay.Miktar, StokHareket.Lot, StokHareket.MelanjKodu, DepoRaf.DepoId, OrmeRafDetay.OrmeRafId, coalesce(StokHareket.Adet,0), StokHareket.OlcuId from StokHareket
      left join OrmeRafDetay on OrmeRafDetay.StokHareketID = StokHareket.StokHareketId
      left join OrmeRaf on OrmeRaf.OrmeRafID = OrmeRafDetay.OrmeRafId
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      where StokHareket.OrmeFisID = :AOrmeFisID and StokHareket.HamIade = 0
      into :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :XRenkId, :XHamId, :XOrmeRafId, :XKapAdedi, :XMiktar, :XLot, :XMelanjKodu, :XDepoId, :OrmeRafId, :XAdet, :XOlcuId do
    begin
      select OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari from OrmeRaf where OrmeRafId = :OrmeRafId into :OrmeRafKalanMiktar;
      if (OrmeRafKalanMiktar < XMiktar) then
        XMiktar = OrmeRafKalanMiktar;

      if (XMiktar > 0) then
      begin
        Select BoyaSiparis.KumasTemin from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitID = :XBoyaSiparisKayitId into :FKumasTeminSiparisi;

        if (FKumasTeminSiparisi = 1) then
        begin
          TBoyaSiparisKayitId = XBoyaSiparisKayitId;
          SELECT sum(OrmeSiparisDetay.HamMiktar) FROM OrmeSiparisDetay
            WHERE OrmeSiparisDetay.KtBoyaSiparisKayitId = :TBoyaSiparisKayitId
            into :TeminToplamMiktar;

          DagitimMiktar = XMiktar;

          for SELECT OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.ToplamBoyayaCikis FROM OrmeSiparisDetay
            LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
            WHERE OrmeSiparisDetay.KtBoyaSiparisKayitId = :TBoyaSiparisKayitId
            order by OrmeSiparisDetay.OrmeSiparisDetayId
            into :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :TeminMiktar, :ToplamBoyayaCikis do
          begin
            if (TeminMiktar is Null) then TeminMiktar = 0;
            if (ToplamBoyayaCikis is Null) then ToplamBoyayaCikis = 0;
            TeminMiktar = TeminMiktar - ToplamBoyayaCikis;
            if (TeminMiktar < 0) then TeminMiktar = 0;
            if (TeminMiktar >= DagitimMiktar) then XMiktar = DagitimMiktar;
            else XMiktar = TeminMiktar;
            if (XMiktar > 0) then
            begin
              insert into StokHareket(OrmeFisId, HareketTipi, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, Lot, MelanjKodu, OrmeRafId, DepoId, KapAdedi, Miktar, OrmeGirisIrsaliyeNo, Notu, InsertKullaniciId, KullaniciId, OlcuId, Adet)
                             values(:OrmeFisId, :shtUretimeCikis, :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :XHamId, :XRenkId, :Xlot, :XMelanjKodu, :XOrmeRafId, :XDepoId, :XKapAdedi, :XMiktar, :UretimIrsaliyeNo, :SatirAciklama, :AKullaniciId, :AKullaniciId, :XOlcuId, :XAdet);
              DagitimMiktar = DagitimMiktar - XMiktar;
            end
          end
          if (DagitimMiktar > 0) then
          begin
            XMiktar = DagitimMiktar;
            if (XMiktar > 0) then
            begin
              insert into StokHareket(OrmeFisId, HareketTipi, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, Lot, MelanjKodu, OrmeRafId, DepoId, KapAdedi, Miktar, OrmeGirisIrsaliyeNo, Notu, InsertKullaniciId, KullaniciId, OlcuId, Adet)
                             values(:OrmeFisId, :shtUretimeCikis, :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :XHamId, :XRenkId, :Xlot, :XMelanjKodu, :XOrmeRafId, :XDepoId, :XKapAdedi, :XMiktar, :UretimIrsaliyeNo, :SatirAciklama, :AKullaniciId, :AKullaniciId, :XOlcuId, :XAdet);
              DagitimMiktar = DagitimMiktar - XMiktar;
            end
          end
        end
        else
          insert into StokHareket(OrmeFisId, HareketTipi, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, Lot, MelanjKodu, OrmeRafId, DepoId, KapAdedi, Miktar, OrmeGirisIrsaliyeNo, Notu, InsertKullaniciId, KullaniciId, OlcuId, Adet)
                           values(:OrmeFisId, :shtUretimeCikis, :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :XHamId, :XRenkId, :Xlot, :XMelanjKodu, :XOrmeRafId, :XDepoId, :XKapAdedi, :XMiktar, :UretimIrsaliyeNo, :SatirAciklama, :AKullaniciId, :AKullaniciId, :XOlcuId, :XAdet);
      end
    end
  end
  else
  begin
    For Select StokHareket.BoyaSiparisKayitId, StokHareket.OrmeSiparisDetayId, StokHareket.RenkId, StokHareket.HamId, OrmeRafDetay.OrmeRafId,
               OrmeRafDetay.KapAdedi, OrmeRafDetay.Miktar, StokHareket.Lot, StokHareket.MelanjKodu, DepoRaf.DepoID, OrmeRafDetay.OrmeRafId, StokHareket.OlcuId, StokHareket.Adet from StokHareket
      left join OrmeRafDetay on OrmeRafDetay.StokHareketID = StokHareket.StokHareketId
      left join OrmeRaf on OrmeRaf.OrmeRafID = OrmeRafDetay.OrmeRafId
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      where StokHareket.OrmeIrsaliyeID = :AOrmeIrsaliyeID and StokHareket.HamIade = 0
      into :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :XRenkId, :XHamId, :XOrmeRafId, :XKapAdedi, :XMiktar, :XLot, :XMelanjKodu, :XDepoID, :OrmeRafId, :XOlcuId, :XAdet do
    begin
      select OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari from OrmeRaf where OrmeRafId = :OrmeRafId into :OrmeRafKalanMiktar;
      if (OrmeRafKalanMiktar < XMiktar) then
        XMiktar = OrmeRafKalanMiktar;

      if (XMiktar > 0) then
      begin
        Select BoyaSiparis.KumasTemin from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitID = :XBoyaSiparisKayitId into :FKumasTeminSiparisi;

        if (FKumasTeminSiparisi = 1) then
        begin
          TBoyaSiparisKayitId = XBoyaSiparisKayitId;
          SELECT sum(OrmeSiparisDetay.HamMiktar) FROM OrmeSiparisDetay
            WHERE OrmeSiparisDetay.KtBoyaSiparisKayitId = :TBoyaSiparisKayitId
            into :TeminToplamMiktar;

          DagitimMiktar = XMiktar;

          for SELECT OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.ToplamBoyayaCikis FROM OrmeSiparisDetay
            LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
            WHERE OrmeSiparisDetay.KtBoyaSiparisKayitId = :TBoyaSiparisKayitId
            into :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :TeminMiktar, :ToplamBoyayaCikis do
          begin
            if (TeminMiktar is Null) then TeminMiktar = 0;
            if (ToplamBoyayaCikis is Null) then ToplamBoyayaCikis = 0;
            TeminMiktar = TeminMiktar - ToplamBoyayaCikis;
            if (TeminMiktar < 0) then TeminMiktar = 0;
            if (TeminMiktar >= DagitimMiktar) then XMiktar = DagitimMiktar;
            else XMiktar = TeminMiktar;
            if (XMiktar > 0) then
            begin
              insert into StokHareket(OrmeFisId, HareketTipi, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, Lot, MelanjKodu, DepoID, OrmeRafId, KapAdedi, Miktar, OrmeGirisIrsaliyeNo, Notu, InsertKullaniciId, KullaniciId, OlcuId, Adet)
                           values(:OrmeFisId, :shtUretimeCikis, :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :XHamId, :XRenkId, :XLot, :XMelanjKodu, :XDepoID, :XOrmeRafId, :XKapAdedi, :XMiktar, :UretimIrsaliyeNo, :SatirAciklama, :AKullaniciId, :AKullaniciId, :XOlcuId, :XAdet);
              DagitimMiktar = DagitimMiktar - XMiktar;
            end
          end
          if (DagitimMiktar > 0) then
          begin
            XMiktar = DagitimMiktar;
            if (XMiktar > 0) then
            begin
              insert into StokHareket(OrmeFisId, HareketTipi, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, Lot, MelanjKodu, DepoID, OrmeRafId, KapAdedi, Miktar, OrmeGirisIrsaliyeNo, Notu, InsertKullaniciId, KullaniciId, OlcuId, Adet)
                           values(:OrmeFisId, :shtUretimeCikis, :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :XHamId, :XRenkId, :XLot, :XMelanjKodu, :XDepoID, :XOrmeRafId, :XKapAdedi, :XMiktar, :UretimIrsaliyeNo, :SatirAciklama, :AKullaniciId, :AKullaniciId, :XOlcuId, :XAdet);
              DagitimMiktar = DagitimMiktar - XMiktar;
            end
          end
        end
        else
          insert into StokHareket(OrmeFisId, HareketTipi, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, RenkId, Lot, MelanjKodu, DepoID, OrmeRafId, KapAdedi, Miktar, OrmeGirisIrsaliyeNo, Notu, InsertKullaniciId, KullaniciId, OlcuId, Adet)
                         values(:OrmeFisId, :shtUretimeCikis, :XBoyaSiparisKayitId, :XOrmeSiparisDetayId, :XHamId, :XRenkId, :XLot, :XMelanjKodu, :XDepoID, :XOrmeRafId, :XKapAdedi, :XMiktar, :UretimIrsaliyeNo, :SatirAciklama, :AKullaniciId, :AKullaniciId, :XOlcuId, :XAdet);
      end
    end
  end

  Status = 1;
  suspend;
end
