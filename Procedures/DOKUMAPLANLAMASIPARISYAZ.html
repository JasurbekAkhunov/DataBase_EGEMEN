CREATE PROCEDURE DOKUMAPLANLAMASIPARISYAZ(
    ADOKUMAPLANLAMAGRUPID VARCHAR(10),
    ADOKUMAMAKINAID VARCHAR(10),
    AFIRMAID VARCHAR(20))
RETURNS (
    DOKUMAPLANLAMAID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPARISIVEREN VARCHAR(50),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    SIPARISMIKTARI NUMERIC(12,3),
    SIPARISKALANMIKTAR NUMERIC(12,3),
    ARTIKELNO VARCHAR(20),
    MUSTERITALIMATI BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    BEYAZLIK INTEGER,
    URETIMNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
begin
  For Select DokumaPlanlamaYaz.DokumaPlanlamaId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
     OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.HamMiktar,
     OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.ToplamDokunan + OrmeSiparisDetay.ToplamSakatDokunan,
     OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriTalimati, OrmeSiparisDetay.Notu,
     OrmeSiparisDetay.Beyazlik, OrmeSiparisDetay.UretimNotu, FirmaYetkili.Adi  from DokumaPlanlamaYaz(:ADokumaPlanlamaGrupId, :ADokumaMakinaId, :AFirmaId)
  Left Join BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlamaYaz.BoyaSiparisKayitId
  Left Join SiparisGrup ON SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  Left Join Firma ON Firma.FirmaId = BoyaSiparis.FirmaId
  left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
  Left Join OrmeSiparisDetay ON OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlamaYaz.OrmeSiparisDetayId
  into :DokumaPlanlamaId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :MusteriSiparisNo, :ReferansSiparisNo, :SiparisMiktari,
       :SiparisKalanMiktar, :ArtikelNo, :MusteriTalimati, :Notu, :Beyazlik, :UretimNotu, :SiparisiVeren  do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';

    suspend;
  end
end
