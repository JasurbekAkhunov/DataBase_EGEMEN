CREATE PROCEDURE TRANSFERICINBOYASATISIRSDETAY(
    ABOYAIRSALIYEID INTEGER)
RETURNS (
    BOYAIRSALIYEID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    HAREKETTIPIID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTITIPIID INTEGER,
    HAMTIPIID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    VADEGUN INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISIRANO INTEGER,
    PARTIFASON INTEGER,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ISLEMTIPIID INTEGER,
    RAFDETAYID INTEGER,
    KAPADEDI INTEGER,
    TRANSFERMIKTARI DOUBLE PRECISION,
    BRUTMIKTAR DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATTL DOUBLE PRECISION,
    TUTARTL DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    SIPARISPROSESADI VARCHAR(255),
    TRANSFERKODU VARCHAR(55),
    KDVORANI DOUBLE PRECISION,
    DOVIZKURU DOUBLE PRECISION,
    MAMULKODU VARCHAR(20),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    ILAVEISLEMLER VARCHAR(1024),
    MUSTERISIPARISNO VARCHAR(50),
    BIRIMTRANSFERKODU VARCHAR(10),
    ODEMESEKLITRANSFERKODU VARCHAR(10),
    MUHGELIRHESAPKODU VARCHAR(30),
    KDVMUHASEBEHESAPKODU VARCHAR(30),
    SIPARISISLEMGRUBUID INTEGER,
    SPFASONMUHGELIRHESAPKODU VARCHAR(30),
    SPSATISMUHGELIRHESAPKODU VARCHAR(30),
    LABRECETEDERINLIKID VARCHAR(2),
    SIPDETAYRENKDERINLIGIID INTEGER,
    SIPDETAYRENKDERINLIGIADI VARCHAR(20),
    SIPDETAYTRANSFERKODU VARCHAR(35),
    HAMSTOKMALIYETI DOUBLE PRECISION,
    BOYAKIMYASALMALIYETI DOUBLE PRECISION,
    ASAMAUMUZERINDENPARTIMALIYETI DOUBLE PRECISION,
    UNITELERASEVKMALIYETI DOUBLE PRECISION,
    TOPLAMPARTIMALIYETI DOUBLE PRECISION,
    HAMTRANSFERKODU VARCHAR(55),
    YURTDISI INTEGER,
    HAMSTOKBIRIMID INTEGER,
    BIRIMADI VARCHAR(2),
    FATURAEDILECEKMIKTARTIPI INTEGER,
    BRUTKILO DOUBLE PRECISION,
    NETKILO DOUBLE PRECISION,
    BRUTMETRE DOUBLE PRECISION,
    NETMETRE DOUBLE PRECISION,
    ADET INTEGER,
    RENKADI VARCHAR(40),
    KALITEID VARCHAR(20),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    IMALATTIPIID INTEGER)
AS
declare variable FHamSatisTransferKodu varchar(55);
declare variable FSatisTransferKodu varchar(55);
declare variable FFasonTransferKodu varchar(55);
declare variable FYariMamulSatisTransferKodu varchar(55);
declare variable IrsaliyeTarihi     TimeStamp;
declare variable SistemDovizId  varchar(5);
declare variable FIlaveIslemlerAdi  varchar(30);
declare variable PartiStokFasonUretimmi integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable HamStokMuhSatisHesapKodu varchar(30);
declare variable HamStokMuhFasonSatisHesapKodu varchar(30);
declare variable HamStokMuhHamSatisHesapKodu varchar(30);
declare variable HamStokMuhFasonOrmeHesapKodu varchar(30);
declare variable EgemenFirmaID         integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable FirmaId varchar(20);
declare variable SiparisProsesId integer;
declare variable PartiSinifKilo       Double Precision;

begin
  BoyaIrsaliyeId = ABoyaIrsaliyeId;
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  Select SistemDovizId, EgemenFirmaID from Sistem where sistemId = 1 into :SistemDovizId, :EgemenFirmaID; /* YTL*/

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  Select IrsaliyeTarihi, FirmaId from BoyaIrsaliye where BoyaIrsaliye.BoyaIrsaliyeID = :ABoyaIrsaliyeID into :IrsaliyeTarihi, :FirmaId;

  if (IrsaliyeTarihi < '01.01.2005') then SistemDovizId = 'TL';

  For select BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliyeDetay.HareketTipiId, BoyaIrsaliyeDetay.PartiKayitId,
         BoyaIrsaliyeDetay.PartiSinifId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSiraNo,
         Parti.PartiTipiId, Parti.Fason, Parti.ReceteId, Recete.Adi, Parti.IslemTipiId, Parti.ImalatTipiId, BoyaIrsaliyeDetay.RafDetayId,
         BoyaIrsaliyeDetay.KapAdedi, BoyaIrsaliyeDetay.Adet,
         BoyaIrsaliyeDetay.BrutMiktar, BoyaIrsaliyeDetay.Miktar, BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.Kilo,
         BoyaIrsaliyeDetay.BrutMetre, BoyaIrsaliyeDetay.Metre, PartiStok.BirimFiyat, PartiStok.DovizId,
         BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, Ham.HamTipiId, Ham.HamSatisTransferKodu, Ham.SatisTransferKodu, Ham.FasonTransferKodu, Ham.YariMamulSatisTransferKodu, BoyaSiparisDetay.VadeGun,
         Ham.MuhSatisHesapKodu, Ham.MuhHamSatisHesapKodu, Ham.MuhFasonSatisHesapKodu, Ham.MuhFasonOrmeHesapKodu,
         PartiStok.MamulKodu, BoyaIrsaliyeDetay.HamId, Ham.Adi, BoyaSiparisDetay.HamStokBirimId, HamStokBirim.BirimId, Coalesce(RS.FaturaEdilecekMiktarTipi, BoyaSiparisDetay.FaturaEdilecekMiktarTipi), Ham.KdvOrani,
         BoyaSiparisDetay.IslemGrubuId, OdemeSekli.TransferKodu,
         SiparisProses.FasonMuhGelirHesapKodu, SiparisProses.SatisMuhGelirHesapKodu, RenkDerinligi.DerinlikId,
         Coalesce(RS.MusteriSiparisNo, BoyaSiparisDetay.MusteriSiparisNo), BoyaSiparisDetay.SipDetayTransferKodu, PartiStok.NetKilo, BoyaSiparis.YurtDisi, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.KaliteId, BoyaSiparisDetay.RenkDerinligiId,
         SipRenkDerinligi.Adi, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparis.SiparisTarihi from BoyaIrsaliyeDetay
    left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
    left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join RenkDerinligi SipRenkDerinligi on SipRenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join HamStokBirim on HamStokBirim.HamStokBirimID = BoyaSiparisDetay.HamStokBirimID
    Left Join OdemeSekli on OdemeSekli.OdemeSekliId = BoyaSiparisDetay.OdemeSekliId
    left join BoyaRezervFisiDetay on BoyaRezervFisiDetay.BoyaRezervFisiDetayId = BoyaIrsaliyeDetay.BoyaRezervFisiDetayId
    left join BoyaSiparisDetay RS on RS.BoyaSiparisDetayId = BoyaRezervFisiDetay.BoyaSiparisDetayId
    where BoyaIrsaliyeId = :ABoyaIrsaliyeID and BoyaIrsaliyeDetay.FasonHamIade = 0 and BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId is not null
    order by Parti.PartiKayitId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliyeDetay.BrutMiktar
    into :BoyaIrsaliyeDetayId, :HareketTipiId, :PartiKayitId, :PartiSinifId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
         :PartiYil, :PartiId, :PartiSiraNo, :PartiTipiId, :PartiFason, :ReceteId, :ReceteAdi, :IslemTipiId, :ImalatTipiId, :RafDetayId, :KapAdedi, :Adet, :BrutMiktar, :Miktar,
         :BrutKilo, :NetKilo, :BrutMetre, :NetMetre, :BirimFiyat, :DovizId,
         :SiparisProsesId, :SiparisProsesAdi, :HamTipiId, :FHamSatisTransferKodu, :FSatisTransferKodu, :FFasonTransferKodu, :FYariMamulSatisTransferKodu, :VadeGun,
         :HamStokMuhSatisHesapKodu, :HamStokMuhHamSatisHesapKodu, :HamStokMuhFasonSatisHesapKodu, :HamStokMuhFasonOrmeHesapKodu,
         :MamulKodu, :HamId, :HamAdi,
         :HamStokBirimId, :BirimAdi, :FaturaEdilecekMiktarTipi, :KdvOrani, :SiparisIslemGrubuId, :OdemeSekliTransferKodu,
         :SPFasonMuhGelirHesapKodu, :SPSatisMuhGelirHesapKodu, :LabReceteDerinlikId, :MusteriSiparisNo, :SipDetayTransferKodu, :PartiSinifKilo, :YurtDisi, :RenkAdi, :KaliteId,
         :SipDetayRenkDerinligiId, :SipDetayRenkDerinligiAdi, :MusteriRenkKoduveAdi, :SiparisTarihi do
  begin
    if (MamulKodu is null) then MamulKodu = '';

    IlaveIslemler = '';

    if (PartiSinifId is not null) then
    begin
      if (EgemenFirmaID = 0) then /* nuryildiz */
      begin
        if (PartiFason = 0) then
        begin
          PartiStokFasonUretimmi = 0;

          Select Count(*) from PartiStokDetay
            left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
            left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayID
            left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterID
            left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = RafDetay.HamStokIrsaliyeID
            where PartiStokDetay.PartiSinifID = :PartiSinifID and
                 (HamStokIrsaliye.GirisTipi = 1 or HamStokMaster.UreticiFirmaID is not null)
            into :PartiStokFasonUretimmi;

          if (PartiStokFasonUretimmi = 0) then TransferKodu = FSatisTransferKodu;
          else TransferKodu = FYariMamulSatisTransferKodu;
        end
        else
        begin
          TransferKodu = FFasonTransferKodu;
        end
      end
      else if (EgemenFirmaID = 4 or EgemenFirmaID = 26) then /* gülle, Ünteks  */
      begin
        if (PartiFason  = 0) then
        begin
          TransferKodu = Null;
          MuhGelirHesapKodu = Null;

          if (HamTipiId = 0 or HamTipiId = 5) then
            Select SatisIplikTransferKodu, SatisIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
          else
          begin
            if (ImalatTipiId = 1) then /* baski ise */
              Select BaskiSatisKumasTransferKodu, BaskiSKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
            else /* boya ise */
              Select SatisKumasTransferKodu, SatisKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
          end
        end
        else
        begin
          TransferKodu = Null;
          MuhGelirHesapKodu = Null;

          if ((HamTipiId = 0 or HamTipiId = 5)) then
            Select FasonIplikTransferKodu, FasonIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
          else
          begin
            if (ImalatTipiId = 1) then /* baski ise */
              Select BaskiFasonKumasTransferKodu, BaskiFKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
            else
              Select FasonKumasTransferKodu, FasonKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
          end
        end
      end
      else if (EgemenFirmaID = 16) then /* tekboy */
      begin
        TransferKodu = Null;
        MuhGelirHesapKodu = Null;

        Select TransferKodu from SiparisProses where SiparisProses.SiparisProsesID = :SiparisProsesId into :TransferKodu;
      end
      else if (EgemenFirmaID = 19) then /* osborn */
      begin
        TransferKodu = Null;
        MuhGelirHesapKodu = Null;

        Select TransferKodu, MuhasebeTransferKodu from RenkDerinlikHamTransferKod
          where RenkDerinlikHamTransferKod.HamID = :HamID and RenkDerinlikHamTransferKod.RenkDerinligiID = :SipDetayRenkDerinligiId
          into :TransferKodu, MuhGelirHesapKodu;
      end
      else /* diger firmalar  */
      begin
        if (PartiFason  = 0) then
        begin
          TransferKodu = Null;
          MuhGelirHesapKodu = Null;

          if (HamTipiId = 0 or HamTipiId = 5) then
            Select SatisIplikTransferKodu, SatisIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
          else
          begin
            if (ImalatTipiId = 1) then /* baski ise */
              Select BaskiSatisKumasTransferKodu, BaskiSKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
            else /* boya ise */
              Select SatisKumasTransferKodu, SatisKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
          end

          PartiStokFasonUretimmi = 0;

          Select Count(*) from PartiStokDetay
            left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
            left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayID
            left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterID
            left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = RafDetay.HamStokIrsaliyeID
            where PartiStokDetay.PartiSinifID = :PartiSinifID and
                 (HamStokIrsaliye.GirisTipi = 1 or HamStokMaster.UreticiFirmaID is not null)
            into :PartiStokFasonUretimmi;

          if (PartiStokFasonUretimmi = 0) then HamTransferKodu = FSatisTransferKodu;
          else HamTransferKodu = FYariMamulSatisTransferKodu;
        end
        else
        begin
          TransferKodu = Null;
          MuhGelirHesapKodu = Null;

          if ((HamTipiId = 0 or HamTipiId = 5)) then
            Select FasonIplikTransferKodu, FasonIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
          else
          begin
            if (ImalatTipiId = 1) then /* baski ise */
              Select BaskiFasonKumasTransferKodu, BaskiFKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
            else
              Select FasonKumasTransferKodu, FasonKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
          end

          HamTransferKodu = FFasonTransferKodu;
        end
      end

      if (PartiFason = 0) then
        Select KdvOrani from Sistem where sistemid = 1 into :KdvOrani;
      else
        Select FasonKdvOrani from Sistem where sistemid = 1 into :KdvOrani;

      if (HamStokBirimId = 0) then /* Kg*/
      begin
        if (FaturaEdilecekMiktarTipi = 0) then
        begin
          TransferMiktari = NetKilo;  /* net Kg*/
          if (TransferMiktari = 0 and BrutKilo <> 0 ) then
            TransferMiktari = BrutKilo; /* Brüt Kg*/
          else if (TransferMiktari = 0 and NetMetre <> 0 ) then
            TransferMiktari = NetMetre; /* Net metre*/
        end
        else
        begin
          TransferMiktari = BrutKilo; /* Brüt Kg*/
          if (TransferMiktari = 0 and NetKilo <> 0 ) then
            TransferMiktari = NetKilo; /* Brüt Kg*/
          else if (TransferMiktari = 0 and BrutMetre <> 0 ) then
            TransferMiktari = BrutMetre; /* Net metre*/
        end
      end
      else if (HamStokBirimId = 1) then /* Mt*/
      begin /* metre */
        if (FaturaEdilecekMiktarTipi = 0) then
        begin
          TransferMiktari = NetMetre; /* Net metre*/
          if (TransferMiktari = 0 and BrutMetre <> 0 ) then
            TransferMiktari = BrutMetre; /* Brüt Kg*/
          else if (TransferMiktari = 0 and NetKilo <> 0 ) then
            TransferMiktari = NetKilo; /* Net metre*/
        end
        else
        begin
          TransferMiktari = BrutMetre; /* Brüt metre*/
          if (TransferMiktari = 0 and NetMetre <> 0 ) then
            TransferMiktari = NetMetre; /* Brüt Kg*/
          else if (TransferMiktari = 0 and BrutKilo <> 0 ) then
            TransferMiktari = BrutKilo; /* Net metre*/
        end
      end
      else if (HamStokBirimId = 2) then /* Adet*/
      begin /* metre */
        TransferMiktari = Adet;
      end

      if (ReceteId is null) then ReceteId = '';
      if (ReceteAdi is null) then ReceteAdi = '';
      if (VadeGun is null) then VadeGun = 0;

      if (DovizID <> SistemDovizId) then
      begin
        select DovizKuru from FirmaDovizKurlariniGetir(:FirmaId, :DovizId, :SistemDovizId, :IrsaliyeTarihi) into :DovizKuru;
      end
      else DovizKuru = 1;

      if (BirimFiyat is null) then BirimFiyat = 0;

      BirimFiyatTL = BirimFiyat * DovizKuru;
      TutarTL = BirimFiyatTL * Miktar;

      for Select IlaveIslemler.Adi from PartiAsamaStok
       left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
       left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
       left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
       where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1 and PartiAsamalari.SerbestAsama = 0 and IlaveIslemler.Adi is not null
       into :FIlaveIslemlerAdi do
      begin
        if (Char_Length(IlaveIslemler) + Char_Length(FIlaveIslemlerAdi) < 1020) then
        begin
         if (IlaveIslemler <> '' ) then IlaveIslemler = IlaveIslemler || cast(', ' as varchar(2) character set UTF8);
         IlaveIslemler = IlaveIslemler || FIlaveIslemlerAdi;
        end
      end
    end
    else
    begin
      VadeGun = 0;
      PartiTipiId = -1;
      TransferKodu = FHamSatisTransferKodu;
      MuhGelirHesapKodu = HamStokMuhHamSatisHesapKodu;
      TransferMiktari = BrutMiktar;
      PartiYil = null;
      PartiId = Null;
      BoyaSiparisYil = Null;
      BoyaSiparisId = Null;
      BirimFiyat = 0;
      ReceteId = '';
      ReceteAdi = '';
      BirimFiyat = 0;
      BirimFiyatTL = 0;
      DovizKuru = 1;
    end

    if (VadeGun is null) then VadeGun = 0;

    BirimTransferKodu = '';

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then Select TransferKodu from Birim where Birim.BirimID = 'Kg' into :BirimTransferKodu;
    else Select TransferKodu from Birim where Birim.BirimID = 'Mt' into :BirimTransferKodu;

    KdvMuhasebeHesapKodu = Null;

    Select MuhSatisHesapKodu from KdvHesapTablosu where KdvHesapTablosu.KdvOrani = :KdvOrani into :KdvMuhasebeHesapKodu;


    ToplamPartiMaliyeti = 0;
    select first 1 HamStokMaliyeti, BoyaKimyasalMaliyeti, AsamaUMUzerindenAsamaMaliyeti, UnitelerASevkMaliyeti from PartiSinifMaliyetKumulatif
         where PartiSinifId = :PartiSinifId
         into :HamStokMaliyeti, :BoyaKimyasalMaliyeti, :AsamaUMUzerindenPartiMaliyeti, :UnitelerASevkMaliyeti;
    ToplamPartiMaliyeti = HamStokMaliyeti + BoyaKimyasalMaliyeti + AsamaUMUzerindenPartiMaliyeti + UnitelerASevkMaliyeti;

    if (ToplamPartiMaliyeti is Null) then ToplamPartiMaliyeti = 0;

    if (PartiSinifKilo is not Null) then
    if (PartiSinifKilo > 0) then
    begin
       ToplamPartiMaliyeti = (ToplamPartiMaliyeti / PartiSinifKilo);
    end

    suspend;
  end
end
