    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATAMIRTIPIGRUPID VARCHAR(4),
    ATAMIRTIPIID INTEGER,
    ARAPORTIPI INTEGER,
    ABITTI INTEGER,
    APARTITIPI INTEGER,
    AFIRMAID VARCHAR(20),
    AUCRETDURUMU INTEGER,
    ATAMIRDUZELTMETIPIID VARCHAR(10),
    ATAMIRCOZUMSEKLIID INTEGER,
    AISLEMEALINACAK INTEGER,
    APARTIGRUPID VARCHAR(4),
    ASIPARISGRUPID VARCHAR(4),
    ARECETETURUID INTEGER,
    AHATA1ASAMAID INTEGER,
    AHATA2ASAMAID INTEGER,
    AHATA3ASAMAID INTEGER,
    AONLEYICIFAALIYETTIPIID INTEGER)
RETURNS (
    PARTITAMIRID INTEGER,
    TARIH TIMESTAMP,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    TAMIRTIPIID VARCHAR(20),
    TAMIRTIPIADI VARCHAR(20),
    TAMIRTIPIGRUPID VARCHAR(4),
    TAMIRTIPIGRUPADI VARCHAR(30),
    HAMADITOPLAMI VARCHAR(1024),
    TAMIRCOZUMSEKLIID INTEGER,
    TAMIRCOZUMSEKLIADI VARCHAR(30),
    ISLEMEALINACAK INTEGER,
    ISLEMEALINACAKSTR VARCHAR(20),
    UCRETLI INTEGER,
    HATAPARTIASAMALARIID INTEGER,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    ASAMATAMIRRAPORSIRANO INTEGER,
    TAMIRDUZELTMETIPIID VARCHAR(10),
    TAMIRDUZELTMETIPIADI VARCHAR(30),
    BITTI INTEGER,
    BITTISTR VARCHAR(5),
    SEBEPTESPITEDILDI INTEGER,
    TAMIRSEBEBIID INTEGER,
    TAMIRSEBEBIADI VARCHAR(20),
    TAMIRSEBEBINOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ONLEYICIFAALIYETTIPIID INTEGER,
    ONLEYICIFAALIYETTIPIADI VARCHAR(40),
    ONLEYICIFAALIYETI BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    UCRETLISTR VARCHAR(4),
    SEBEPTESPITEDILDISTR VARCHAR(4),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    MODELNO VARCHAR(20),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    TAMIRNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    HAMTIPIID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    TAMIRONAYKULLANICIID INTEGER,
    TAMIRONAYKULLANICIADI VARCHAR(40),
    TAMIRONAYTARIHI TIMESTAMP,
    ILKKEZBOYANACAK INTEGER,
    ILKKEZBOYANACAKSTR VARCHAR(20),
    BOYAKIMYASALMALIYETI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    RECETETURUADI ADI15,
    RECETETURUTURID VARCHAR(4),
    PARTININTOPLAMKILOSU NUMERIC(12,3),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    HAMTIPIIDTOPLAMI VARCHAR(100),
    HATA2PARTIASAMALARIID INTEGER,
    SEBEPASAMA2ID INTEGER,
    SEBEPASAMA2ADI VARCHAR(40),
    HATA3PARTIASAMALARIID INTEGER,
    SEBEPASAMA3ID INTEGER,
    SEBEPASAMA3ADI VARCHAR(40),
    KALITEID VARCHAR(20),
    KALITEADI ADI255)
AS
declare variable EskiLabReceteKodu varchar(20);
declare variable IlkBoyananPartiKayitId integer;
declare variable Ilave integer;
declare variable SokumUzeriBoya integer;
declare variable FHamTipiId integer;
declare variable hstIplik integer;
declare variable hstOrmeKumas integer;
declare variable hstDokumaKumas integer;
declare variable hstRibana integer;
declare variable hstAksesuar integer;
declare variable hstElyaf integer;
declare variable hstLycra integer;
declare variable hstTelef integer;
declare variable Durum varchar(5);
declare variable AltBez integer;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstLycra, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstLycra, :hstTelef;
  Select StokTabanDovizId from sistem where sistemid = 1 into :DovizId;
  if (ARaporTipi = 0) then
  begin
    For select PartiTamir.PartiTamirId, PartiTamir.Tarih, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.Fason, Parti.PartigrupId, PartiGrup.Adi, PartiTamir.TamirTipiId, TamirTipi.Adi,
           TamirTipi.TamirTipiGrupId, TamirTipiGrup.Adi,
           PartiTamir.Bitti, PartiTamir.TamirCozumSekliId, TamirCozumSekli.Adi, TamirCozumSekli.IslemeAlinacak, TamirCozumSekli.Ucretli, PartiTamir.SebepTespitEdildi,  Parti.FirmaId, Firma.Adi, Parti.ReceteId, PR.Adi, Parti.ModelNo,
           PartiTamir.Notu, PartiTamir.TamirSebebiId, TamirSebebi.Adi, PartiTamir.TamirSebebiNotu, PartiTamir.OnleyiciFaaliyeti,
           PartiTamir.OnleyiciFaaliyetTipiId, OnleyiciFaaliyetTipi.Adi,
           Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
           PartiTamir.TamirOnayKullaniciId, PTOK.Adi, PartiTamir.TamirOnayTarihi,
           PartiTamir.HataPartiAsamalariId, Asama.AsamaId, Asama.Adi, Asama.TamirRaporSiraNo,
           PartiTamir.TamirDuzeltmeTipiId, TamirDuzeltmeTipi.Adi, Labrecete.EskiLabReceteKodu, RR.IlkBoyananPartiKayitId,
           RR.SokumUzeriBoya, RR.Ilave, ReceteTuru.Adi, ReceteTuru.TurId, PartiTamir.Hata2PartiAsamalariId, A2.AsamaId, A2.Adi, PartiTamir.Hata3PartiAsamalariId, A3.AsamaId, A3.Adi, Kalite.KaliteAdi, Kalite.KaliteId from PartiTamir
    left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
    left join TamirSebebi on TamirSebebi.TamirSebebiId = PartiTamir.TamirSebebiId
    left join Parti on Parti.PartiKayitId = PartiTamir.PartiKayitId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join Recete PR on PR.ReceteId = Parti.ReceteId
    left join ReceteTuru on ReceteTuru.ReceteTuruId = PR.ReceteTuruId
    left join LabRecete on LabRecete.LabReceteId = PR.LabReceteId
    left join Kalite on Kalite.KaliteId = LabRecete.KaliteId

    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiTamir.HataPartiAsamalariId
    left join Recete RR on RR.ReceteId = PartiAsamalari.ReceteId
    left join asama on asama.AsamaId = PartiAsamalari.AsamaId or asama.AsamaId = RR.UstAsamaId
    -- hata2

    left join PartiAsamalari PA2 on PA2.PartiAsamalariId = PartiTamir.Hata2PartiAsamalariId
    left join Recete RR2 on RR2.ReceteId = PA2.ReceteId
    left join asama A2 on A2.AsamaId = PA2.AsamaId or A2.AsamaId = RR2.UstAsamaId
    -- hata3

    left join PartiAsamalari PA3 on PA3.PartiAsamalariId = PartiTamir.Hata3PartiAsamalariId
    left join Recete RR3 on RR3.ReceteId = PA3.ReceteId
    left join asama A3 on A3.AsamaId = PA3.AsamaId or A3.AsamaId = RR2.UstAsamaId


    Left Join TamirDuzeltmeTipi on TamirDuzeltmeTipi.TamirDuzeltmeTipiId = PartiTamir.TamirDuzeltmeTipiId
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
    left join OnleyiciFaaliyetTipi on OnleyiciFaaliyetTipi.OnleyiciFaaliyetTipiid = Partitamir.OnleyiciFaaliyetTipiId
    left join TamirTipiGrup on TamirTipiGrup.TamirTipiGrupId = TamirTipi.TamirTipiGrupId
    Left Join Kullanici PTOK on PTOK.KullaniciId = PartiTamir.TamirOnayKullaniciId
    where PartiTamir.TamirOnayKullaniciId is null and
          (PartiTamir.TamirTipiId = :ATamirTipiId or :ATamirTipiId is null) and
          (TamirTipi.TamirTipiGrupId = :ATamirTipiGrupId or :ATamirTipiGrupId is null) and
          (PartiTamir.TamirDuzeltmeTipiId = :ATamirDuzeltmeTipiId or :ATamirDuzeltmeTipiId is null) and
          (PartiTamir.TamirCozumSekliID = :ATamirCozumSekliID or :ATamirCozumSekliID is null) and
          (TamirCozumSekli.IslemeAlinacak = :AIslemeAlinacak or :AIslemeAlinacak is null) and
          (PartiTamir.Bitti = :ABitti or :ABitti is null) and
          (TamirCozumSekli.Ucretli = :AUcretDurumu or :AUcretDurumu is null) and
          --(Parti.PartiTipiID = :APartiTipi or :APartiTipi is null) and
          (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
          (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is Null) and
          (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (Asama.AsamaId = :AHata1AsamaId or :AHata1AsamaId is null) and
          (A2.AsamaId = :AHata2AsamaId or :AHata2AsamaId is null) and
          (A3.AsamaId = :AHata3AsamaId or :AHata3AsamaId is null) and
          (OnleyiciFaaliyetTipi.OnleyiciFaaliyetTipiId = :AOnleyiciFaaliyetTipiId or :AOnleyiciFaaliyetTipiId is null) and
          (ReceteTuru.ReceteTuruId = :AReceteTuruId or :AReceteTuruId is null)
    order by tarih
    into :PartiTamirId, :Tarih, :PartiKayitId, :PartiYil, :PartiId, :Fason, :PartigrupId, :PartiGrupAdi, :TamirTipiId, :TamirTipiAdi,
         :TamirTipiGrupId, :TamirTipiGrupAdi,
         :Bitti,  :TamirCozumSekliId, :TamirCozumSekliAdi, :IslemeAlinacak, :Ucretli, :SebepTespitEdildi,  :FirmaId, :FirmaAdi, :ReceteId, :ReceteAdi, :ModelNo, :TamirNotu,
         :TamirSebebiId, :TamirSebebiAdi, :TamirSebebiNotu, :OnleyiciFaaliyeti, :OnleyiciFaaliyetTipiId, :OnleyiciFaaliyetTipiAdi,
         :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :TamirOnayKullaniciId, :TamirOnayKullaniciAdi, :TamirOnayTarihi,
         :HataPartiAsamalariId, :AsamaId, :AsamaAdi, :AsamaTamirRaporSiraNo, :TamirDuzeltmeTipiId, :TamirDuzeltmeTipiAdi,
         :EskiLabReceteKodu, :IlkBoyananPartiKayitId, :SokumUzeriBoya, :Ilave, :ReceteTuruAdi, :ReceteTuruTurId, :Hata2PartiAsamalariId, :SebepAsama2Id, :SebepAsama2Adi,
        :Hata3PartiAsamalariId, :SebepAsama3Id, :SebepAsama3Adi, :KaliteAdi, :KaliteId
 do
 begin
      PartiIdStr = '(' || (PartiYil)  || ') ' || (PartiId) || ' ';
      -- Hamtipi Filtreleme
      Durum = '';
      for select PartiStok.HamTipiId, Ham.AltBez from PartiTamirSinif
        left join PartiStok on PartiStok.PartiSinifID = PartiTamirSinif.PartiSinifId
        left join Ham on Ham.HamID = PartiStok.PartiSinifId
        where PartiTamirSinif.PartiTamirId = :PartiTamirId
        group by PartiStok.HamTipiId, Ham.AltBez
        into :FHamTipiId, :AltBez do
      begin
        if (((FHamTipiId = hstIplik or FHamTipiId = hstElyaf or FHamTipiId = hstTelef) and APartiTipi = 0) or
           (FHamTipiId <> hstIplik and FHamTipiId <> hstElyaf and FHamTipiId <> hstTelef and APartiTipi = 1) or
           ((FHamTipiId = hstOrmeKumas or FHamTipiId = hstRibana or FHamTipiId = hstAksesuar) and APartiTipi = 2) or
           (FHamTipiId = hstDokumaKumas and APartiTipi = 3) or
           (AltBez = 1 and APartiTipi = 5) or
           (APartiTipi is null)) then Durum = 'Ok';
      end
      -- HamTipi Filtreleme

      if (Durum = 'Ok') then
      begin
        select PartiBilgisigetir.DesenId, PartiBilgisigetir.DesenAdi, PartiBilgisigetir.VaryantAdi from PartiBilgisigetir(:PartiYil, :PartiId , null) into :DesenId, :DesenAdi, :VaryantAdi;

        if (BoyaSiparisYil is null) then BoyaSiparisIdStr = '';
        else BoyaSiparisIdStr = '(' || (BoyaSiparisYil) || ') ' || (BoyaSiparisId) || ' ';

        if (Ucretli = 1) then UcretliStr = 'Evet';
        else UcretliStr = Null;

        if (SebepTespitEdildi = 1) then SebepTespitEdildiStr = 'Evet';
        else SebepTespitEdildiStr = Null;

        Select Min(HamTipiId), sum(PartiStok.kilo), sum(PartiStok.Metre) from PartiTamirSinif
        left join PartiStok on PartiStok.PartiSinifId = PartiTamirSinif.PartiSinifId
        where PartiTamirSinif.PartiTamirId = :PartiTamirId
        into :HamTipiId, :Kilo, :Metre;

        PartininToplamKilosu = 0;
        select sum(PartiStok.Kilo) from Partistok
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
        where PartiStok.PartiKayitId = :PartiKayitId
        into :PartininToplamKilosu;

        if (Bitti = 1) then BittiStr = 'Bitti';
        else BittiStr = '';

        if (IslemeAlinacak = 1) then IslemeAlinacakStr = 'İşleme Alınacak';
        else IslemeAlinacakStr = '';

        select HamAdiToplami from PartiTamirHamAdiHesapla(:PartiTamirID) into :HamAdiToplami;

        HamTipiIdToplami = '';
        for select PartiStok.HamTipiId from PartiTamirSinif
          left join PartiStok on PartiStok.PartiSinifID = PartiTamirSinif.PartiSinifId
          where PartiTamirSinif.PartiTamirId = :PartiTamirId
          group by PartiStok.HamTipiId
          into :FHamTipiId do
        begin
          if (Char_Length(HamTipiIdToplami) + Char_Length(cast(FHamTipiId as varchar(1)) ) < 100) then
          begin
            if (HamTipiIdToplami <> '' ) then HamTipiIdToplami = HamTipiIdToplami || cast(', ' as varchar(2) character set UTF8);
            HamTipiIdToplami = HamTipiIdToplami || cast(FHamTipiId as varchar(1));
          end
        end

        IlkKezBoyanacak = 0;
        IlkKezBoyanacakStr = '';

        if (SokumUzeriBoya = 0 and Ilave = 0) then
        if (EskiLabReceteKodu is null or EskiLabReceteKodu = '') then
        if (IlkBoyananPartiKayitId is null or (IlkBoyananPartiKayitId = :PartiKayitId)) then IlkKezBoyanacak = 1;

        if (IlkKezBoyanacak = 1) then IlkKezBoyanacakStr = 'İlk Kez Boyandı';

        BoyaKimyasalMaliyeti = 0;

        Select sum(StokDetay.Miktar * StokDetay.BirimFiyat) from PartiAsamalari
          left join StokDetay on StokDetay.PartiAsamalariID = PartiAsamalari.PartiAsamalariId
          where PartiAsamalari.PartiTamirId = :PartiTamirId and StokDetay.PartiAsamalariId is not null and StokDetay.HareketTipi = 6
          into :BoyaKimyasalMaliyeti;

        if (Fason = 1) then FasonStr = 'Fason';
        else FasonStr = '';

        suspend;
      end
    end
  end
  else if (ARaporTipi = 1) then
  begin
    For select PartiTamir.PartiTamirId, PartiTamir.Tarih, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.Fason, Parti.PartigrupId, PartiGrup.Adi, PartiTamir.TamirTipiId, TamirTipi.Adi,
           TamirTipi.TamirTipiGrupId, TamirTipiGrup.Adi,
           PartiTamir.Bitti, PartiTamir.TamirCozumSekliId, TamirCozumSekli.Adi, TamirCozumSekli.IslemeAlinacak, TamirCozumSekli.Ucretli, PartiTamir.SebepTespitEdildi,  Parti.FirmaId, Firma.Adi, Parti.ReceteId, PR.Adi, Parti.ModelNo,

        end

        BoyaKimyasalMaliyeti = 0;

          left join StokDetay on StokDetay.PartiAsamalariID = PartiAsamalari.PartiAsamalariId

          into :BoyaKimyasalMaliyeti;

        else FasonStr = '';
        suspend;
      end
