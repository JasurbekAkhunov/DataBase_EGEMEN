declare variable IslemTipiId      integer;
declare variable ATamir         integer;

declare variable itNormal       integer;

declare variable KullanilanMiktar      Numeric(12,3);
declare variable IptalPartiKilo        Numeric(12,3);
declare variable IptalPartiMetre       Numeric(12,3);
declare variable IptalPartiMiktar      Numeric(12,3);

declare variable IptalKapAdedi        integer;
declare variable FRafID                varchar(10);
declare variable SiparisSecimiOk       integer;
declare variable RafKapAdedi           integer;
declare variable KapAdediSatis         integer;
declare variable YariMamulPartiYil integer;
declare variable YariMamulPartiId  integer;
declare variable DokumaTopTakibiDevrede integer;
begin
  select DokumaTopTakibiDevrede from sistem where sistem.sistemid = 1 into :DokumaTopTakibiDevrede;
  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  Select BoyaSiparisDetay.IslemTipiId, BoyaSiparis.FirmaId, Fason from BoyaSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId
    into :IslemTipiId, :FFirmaId, :Fason;

  if (IslemTipiId = itTamir) then ATamir = 1;
  else ATamir = 0;

  if (Fason = 0) then Select FirmaKodumuz from Sistem into :FFirmaId;
  else AReferansFirmaId = null;

  ASatinAlmaTalepKayitId = Null;
  if (ASatinAlmaTalepYil is not null and ASatinAlmaTalepId is not null) then
  begin
    ASatinAlmaTalepKayitId = -1;

  end

         HamStokMaster.Denye, LycraUretici.Adi,
         HamStokMaster.Renk, HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartiNo,
         HamStokDetay.RezervEdilenMiktar, HamStokDetay.RezervEdilenKilo, HamStokDetay.RezervEdilenMetre, HamStokDetay.RezervEdilenAdet, HamStokDetay.RezervEdilenKapAdedi,
         HamStokDetay.KapAdedi, HamStokDetay.Miktar, HamStokDetay.Kilo, HamStokDetay.Metre, HamStokDetay.Adet,
         BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, coalesce(VerilenSiparisDetay.BoyaSiparisDetayId, HamStokMaster.BoyaSiparisDetayID),
         SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId, SatinAlmaTalep.TalepKullaniciId, TalepKullanici.Adi,
         HamStokIrsaliye.HamStokHareketTipiId, HamStokMaster.HamKontrolKullanimYeri,
         OrmeSiparisDetay.SiparisSiraNo, OrmePlanlama.Beyazlik, HamStokMaster.Brut,
         BoyaSiparis.SiparisGrupID, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2ID, SiparisGrup2.Adi, BoyaSiparis.FirmaId, BoyaSiparisFirma.Adi,

         Sum(RafDetay.IadeEdilecekKilo + RafDetay.IadeEdilenKilo + RafDetay.PartilenenKilo + RafDetay.SatisKilo),
         Sum(RafDetay.IadeEdilecekMetre + RafDetay.IadeEdilenMetre + RafDetay.PartilenenMetre + RafDetay.SatisMetre),
         Sum(RafDetay.IadeEdilecekAdet + RafDetay.IadeEdilenAdet + RafDetay.PartilenenAdet), HamStokMaster.HamKontrolNotu, Raf.YariMamul, iif(OrmePlanlama.UretimPartiNo is not null, OrmePlanlama.UretimPartiNo, DokumaPlanlama.UretimPartiNo),
         sum(RafDetay.IadeEdilecekKapAdedi + RafDetay.IadeEdilenKapAdedi), Uretici.UreticiId, Uretici.Adi, HamStokMaster.Aciklama,
         Parti.PartiYil, Parti.PartiId, Olcu.OlcuId, Olcu.OlcuAdi, SiparisGrup3.SiparisGrupId, SiparisGrup3.Adi, SiparisGrup4.SiparisGrupId, SiparisGrup4.Adi, Kalite.KaliteId, Kalite.KaliteAdi,
         HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj, HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik,
         HamStokMaster.BirimFiyat, HamStokMaster.BirimFiyat2, PSDokumaIsEmriBilgisiGetir.CozguIndigoIsEmriIdStr, PSDokumaIsEmriBilgisiGetir.TopNo  from BoyaSiparisDetay
  left join ham on ham.hamid = BoyaSiparisDetay.Hamid
  left join HamTipi on HamTipi.HamTipiid = Ham.HamTipiId
  join HamStokDetay on HamStokDetay.HamId = BoyaSiparisDetay.HamId
  join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  join HamstokIrsaliye on hamstokIrsaliye.HamStokIrsaliyeid = HamstokDetay.HamStokIrsaliyeId
  left join LycraUretici on LycraUretici.LycraUreticiId = HamStokMaster.LycraUreticiId
  left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = HamstokMaster.VerilenSiparisDetayId
  left join SatinAlmaTalepDetay on SatinAlmaTalepDetay.SatinAlmaTalepDetayId = VerilenSiparisDetay.SatinAlmaTalepDetayId
  left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = SatinAlmaTalepDetay.SatinAlmaTalepKayitId
  left join Kullanici TalepKullanici on TalepKullanici.Kullaniciid = SatinAlmaTalep.TalepKullaniciId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = coalesce(VerilenSiparisDetay.BoyaSiparisKayitId, HamStokMaster.BoyaSiparisKayitId)
  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  left join SiparisGrup SiparisGrup3 on SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
  left join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = coalesce(VerilenSiparisDetay.OrmeSiparisDetayId, HamStokMaster.OrmeSiparisDetayId)
  left join Firma IrsaliyeFirma on IrsaliyeFirma.FirmaId = HamStokIrsaliye.FirmaId
  left join Firma ReferansFirma on ReferansFirma.FirmaId = HamStokMaster.ReferansFirmaId
  left join Firma UreticiFirma on UreticiFirma.FirmaId = HamStokMaster.UreticiFirmaId
  left join Firma BoyaSiparisFirma on BoyaSiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join Ormeci on Ormeci.OrmeciId = HamstokIrsaliye.OrmeciId
  left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
  left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  left join OrmePlanlama on OrmePlanlama.OrmePlanlamaid = HamStokMaster.OrmePlanlamaid
  left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaid = HamStokMaster.DokumaPlanlamaid
  left join Uretici on Uretici.UreticiId = HamStokMaster.UreticiId
  left join PartiStok on PartiStok.PartiSinifId = HamStokMaster.PartiSinifId
  left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
  left join Olcu on Olcu.OlcuId = HamStokMaster.OlcuId
  left join RafDetay on RafDetay.HamStokDetayid = HamStokDetay.HamStokDetayid
  left join Raf on Raf.RafId = RafDetay.RafId
  left join Kalite on  Kalite.KaliteId = HamStokMaster.KaliteId
  left join PSDokumaIsEmriBilgisiGetir(null,HamStokMaster.hamstokmasterid) ON 1=1  and HamStokMaster.dokumaplanlamaid is not null
  where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId and
        HamstokDetay.StokFirmaId = :FFirmaId and HamStokMaster.Tamir = :ATamir and
       (HamStokMaster.HamStokMasterId = :AHamStokMasterId or :AHamStokMasterId is null) and
       (HamStokMaster.ReferansFirmaId = :AReferansFirmaId or :AReferansFirmaId is null or HamStokMaster.ReferansFirmaId is null) and
       (HamStokMaster.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
       (HamStokMaster.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
       ((:AOrmeciId > -1 and HamstokIrsaliye.OrmeciId = :AOrmeciId) or :AOrmeciId = -1) and
       ((:AIrsaliyeNo is not null and HamstokIrsaliye.IrsaliyeNo = :AIrsaliyeNo) or :AIrsaliyeNo is null) and
       ((:AIrsaliyeNo is null and HamstokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi) or :AIrsaliyeNo is not null) and
       (SatinAlmaTalep.SatinAlmaTalepKayitId = :ASatinAlmaTalepKayitId or :ASatinAlmaTalepKayitId is null) and
       (SatinAlmaTalep.TalepKullaniciId = :ATalepKullaniciId or :ATalepKullaniciId is null) and
       (:APartilemesizRaflar = 1 or Raf.PartilemeYapilabilir = 1) and
       (:AYariMamul = 2 or Raf.YariMamul = :AYariMamul) and
       HamStokMaster.OnayDurumu = 1 and HamStokDetay.HamStokKabulTipiId <> 5 /*and
       (:ABoyaSiparisYil is Null or :ABoyaSiparisYil = BoyaSiparis.BoyaSiparisYil) and
       (:ABoyaSiparisId is Null or :ABoyaSiparisId = BoyaSiparis.BoyaSiparisId) and
       (BoyaSiparis.FirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null)*/
  Group By HamStokDetay.HamStokDetayId, HamStokDetay.HamStokMasterId, HamStokIrsaliye.Sinif, HamStokIrsaliye.FirmaId, IrsaliyeFirma.Adi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi,
         HamStokIrsaliye.EvrakTipi, Ormeci.Adi, UreticiFirma.Adi, BoyaSiparisDetay.HamId, Ham.Adi, Ham.HamTipiId, HamTipi.Adi,
         HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.En, HamStokMaster.Grm2, HamStokMaster.OrmeciMakinaNo, HamStokMaster.Lot, HamStokMaster.MelanjKodu,
         HamStokMaster.BobinGramaji,
         HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo,
         HamStokMaster.ReferansFirmaId, ReferansFirma.Adi,
         HamStokMaster.TupAcikEnMayli, HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3,
         HamStokMaster.Denye, LycraUretici.Adi,
         HamStokMaster.Renk, HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartiNo,
         HamStokDetay.RezervEdilenMiktar, HamStokDetay.RezervEdilenKilo, HamStokDetay.RezervEdilenMetre, HamStokDetay.RezervEdilenAdet, HamStokDetay.RezervEdilenKapAdedi,
         HamStokDetay.KapAdedi, HamStokDetay.Miktar, HamStokDetay.Kilo, HamStokDetay.Metre, HamStokDetay.Adet,
         BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, coalesce(VerilenSiparisDetay.BoyaSiparisDetayId, HamStokMaster.BoyaSiparisDetayID),
         SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId, SatinAlmaTalep.TalepKullaniciId, TalepKullanici.Adi,
         HamStokIrsaliye.HamStokHareketTipiId, HamStokMaster.HamKontrolKullanimYeri, OrmeSiparisDetay.SiparisSiraNo,
         OrmePlanlama.Beyazlik, HamStokMaster.Brut, BoyaSiparis.SiparisGrupID, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2ID, SiparisGrup2.Adi, BoyaSiparis.FirmaId, BoyaSiparisFirma.Adi,
         HamStokMaster.HamKontrolNotu, Raf.YariMamul, OrmePlanlama.OrmePlanlamaId, OrmePlanlama.UretimPartiNo, DokumaPlanlama.UretimPartiNo, Uretici.UreticiId, Uretici.Adi, HamStokMaster.Aciklama,
         Parti.PartiYil, Parti.PartiId, Olcu.OlcuId, Olcu.OlcuAdi, SiparisGrup3.SiparisGrupId, SiparisGrup3.Adi, SiparisGrup4.SiparisGrupId, SiparisGrup4.Adi, Kalite.KaliteId, Kalite.KaliteAdi,
         HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj, HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik,
         HamStokMaster.BirimFiyat, HamStokMaster.BirimFiyat2, PSDokumaIsEmriBilgisiGetir.CozguIndigoIsEmriIdStr, PSDokumaIsEmriBilgisiGetir.TopNo
    into :HamStokDetayId, :HamStokMasterId, :Sinif, :FirmaId, :FirmaAdi, :IrsaliyeNo, :Tarih, :EvrakTipi, :OrmeciAdi,
         :UreticiFirmaAdi,  :HamId, :HamAdi, :HamTipiId, :HamTipiAdi,
         :Pus, :Feine, :En, :Grm2, :OrmeciMakinaNo, :Lot, :MelanjKodu, :BobinGramaji, :MusteriPartiNo, :MusteriSiparisNo, :ReferansSiparisNo,
         :ReferansFirmaId, :ReferansFirmaAdi, :TupAcikEnMayli, :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3, :Denye, :LycraUreticiAdi,
         :Renk, :Tamirmi, :TamirTipiAdi,
         :EskiPartiNo, :RezervEdilenMiktar, :RezervEdilenKilo, :RezervEdilenMetre, :RezervEdilenAdet, :RezervEdilenKapAdedi, :KapAdedi,
         :Miktar, :Kilo, :Metre, :Adet, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
         :SatinAlmaTalepYil, :SatinAlmaTalepId, :TalepKullaniciId, :TalepKullaniciAdi, :HamStokHareketTipiId, :HamKontrolKullanimYeri, :OrmeSiparisSiraNo, :Beyazlik, :BrutNet,
         :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisFirmaId, :SiparisFirmaAdi,
         :KullanilanMiktar, :KullanilanKilo, :KullanilanMetre, :KullanilanAdet, :HamKontrolNotu, :YariMamul, :OrmeUretimPartiNo, :RafKapAdedi, :UreticiId, :UreticiAdi, :Aciklama,
         :YariMamulPartiYil, :YariMamulPartiId, :OlcuId, :OlcuAdi, :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi, :KaliteId, :KaliteAdi,
         :YikamaOncesiEn, :YikamaOncesiGramaj, :YikamaSonrasiEn, :YikamaSonrasiGramaj, :EndenCekmezlik, :BoydanCekmezlik, :BirimFiyat, :BirimFiyat2, :DokumaOzelKodStr,:DokumaTopNo do
  begin
    SiparisSecimiOk = 1;
    RezervBilgiKilo = 0;

    YariMamulPartiNoStr = '';
    if (YariMamulPartiYil is not null) then
      YariMamulPartiNoStr = '('||YariMamulPartiYil||') '||YariMamulPartiId;

    if (YariMamul = 1) then YariMamulStr = 'Yar Mamul';
    else YariMamulStr = 'Ham';

    select Sum(BoyaIrsaliyeDetay.KapAdedi) from RafDetay
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.RafDetayID =  RafDetay.RafDetayID
      where RafDetay.HamStokDetayID = :HamStokDetayID and BoyaIrsaliyeDetay.HareketTipiID = 0 into :KapAdediSatis;

    if (RafKapAdedi is Null) then RafKapAdedi = 0;
    if (KapAdediSatis is Null) then KapAdediSatis = 0;


    select Sum(BoyaSiparisRezerv.KullanilanKapAdedi + :RafKapAdedi + :KapAdediSatis) from BoyaSiparisRezerv
      where BoyaSiparisRezerv.HamStokDetayID = :HamStokDetayID into :KullanilanKapAdedi;

      if (RafKapAdedi is Null) then RafKapAdedi = 0;
      if (KullanilanKapAdedi is Null) then KullanilanKapAdedi = 0;
      if (KapAdediSatis is Null) then KapAdediSatis = 0;

      KalanKapAdedi = RafKapAdedi - KullanilanKapAdedi -  KapAdediSatis;


    if (ASiparisFirmaId is not Null) then
    begin
      select sum(HSDRezervBilgi.Kilo) from HSDRezervBilgi
        left join HamStokDetay on HamStokDetay.HamStokDetayId = HSDRezervBilgi.HamStokDetayId
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = HSDRezervBilgi.BoyaSiparisKayitId
        where HSDRezervBilgi.HamStokDetayId = :HamStokDetayId and BoyaSiparis.FirmaId = :ASiparisFirmaId
        into :RezervBilgiKilo;
      if (RezervBilgiKilo is Null) then RezervBilgiKilo = 0;

      select sum(Kilo) from BoyaSiparisRezervDetay
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisRezervDetay.BoyaSiparisDetayId
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisRezervDetay.BoyaSiparisKayitId
        where BoyaSiparisRezervDetay.HamStokDetayId = :HamStokDetayId and BoyaSiparis.FirmaId = :ASiparisFirmaId
        into :BuSipariseRezerv;
      if (BuSipariseRezerv is Null) then BuSipariseRezerv = 0;

      RezervBilgiKilo = RezervBilgiKilo - BuSipariseRezerv;
      if (RezervBilgiKilo <= 0) then SiparisSecimiOk = 0;
    end

    if (ABoyaSiparisYil is not Null and ABoyaSiparisId is not Null) then
    begin
      select sum(HSDRezervBilgi.Kilo) from HSDRezervBilgi
        left join HamStokDetay on HamStokDetay.HamStokDetayId = HSDRezervBilgi.HamStokDetayId
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = HSDRezervBilgi.BoyaSiparisKayitId
        where HSDRezervBilgi.HamStokDetayId = :HamStokDetayId and
        BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId
        into :RezervBilgiKilo;
      if (RezervBilgiKilo is Null) then RezervBilgiKilo = 0;

      select sum(Kilo) from BoyaSiparisRezervDetay
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisRezervDetay.BoyaSiparisDetayId
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisRezervDetay.BoyaSiparisKayitId
        where BoyaSiparisRezervDetay.HamStokDetayId = :HamStokDetayId and
        BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId
        into :BuSipariseRezerv;
      if (BuSipariseRezerv is Null) then BuSipariseRezerv = 0;

      RezervBilgiKilo = RezervBilgiKilo - BuSipariseRezerv;
      if (RezervBilgiKilo <= 0) then SiparisSecimiOk = 0;
    end

    if (Tamirmi = 1) then Tamir = 'Tamir';
    else Tamir = '';

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fi';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'rsaliye';
    else EvrakTipiStr = 'Hatal';

    if (HamKontrolKullanimYeri = 0) then HamKontrolKullanimYeriStr = 'Beyazlk';
    else if (HamKontrolKullanimYeri = 1) then HamKontrolKullanimYeriStr = 'Siyah';
    else if (HamKontrolKullanimYeri = 2) then HamKontrolKullanimYeriStr = 'Bask Alt';
    else HamKontrolKullanimYeriStr = '';

    if (HamStokHareketTipiId = 0) then HamStokHareketTipiIdStr = 'Giri';
    else if (HamStokHareketTipiId = 1) then HamStokHareketTipiIdStr = 'Sat ade';
    else HamStokHareketTipiIdStr = '';

    BeyazlikStr = '';
    if (Beyazlik = 1) then BeyazlikStr = 'Evet';

    if (BrutNet = 0) then BrutNetStr = 'Net';
    else if (BrutNet = 1) then BrutNetStr = 'Brt';
    else BrutNetStr = '';

    IptalPartiKilo = 0;
    IptalPartiMetre = 0;
    IptalPartiMiktar = 0;
    PartilenebilirRafKilo = 0;
    PartilenebilirRafAdet = 0;
    PartilenebilirRafKapAdedi = 0;

    Select Sum(Kilo), Sum(Metre), Sum(Miktar), Sum(Adet), Sum(KapAdedi) from RafDetay
      where RafDetay.HamStokDetayID = :HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
      into :IptalPartiKilo, :IptalPartiMetre, :IptalPartiMiktar, :IptalPartiAdet, :IptalKapAdedi;

    if (IptalPartiKilo is null) then IptalPartiKilo = 0;
    if (IptalPartiMetre is null) then IptalPartiMetre = 0;
    if (IptalPartiMiktar is null) then IptalPartiMiktar = 0;
    if (IptalPartiAdet is null) then IptalPartiAdet = 0;
    if (IptalKapAdedi is null) then IptalKapAdedi = 0;

    KalanKilo = Kilo + IptalPartiKilo - KullanilanKilo - RezervEdilenKilo;
    KalanMetre = Metre + IptalPartiMetre - KullanilanMetre - RezervEdilenMetre;
    KalanMiktar = Miktar + IptalPartiMiktar - KullanilanMiktar - RezervEdilenMiktar;
    KalanAdet = Adet + IptalPartiAdet - KullanilanAdet - RezervEdilenAdet;
    KalanKapAdedi = KapAdedi + IptalKapAdedi - KullanilanKapAdedi - RezervEdilenKapAdedi;

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    SatinAlmaTalepIdStr = '';
    if (SatinAlmaTalepYil is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId)  || ' ';

    Select TupAcikEnMayliStrGetir.tupacikenmaylistr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    RafIDStr = '';
    PartilenebilirRafKilo = Null;

    if (KalanMiktar > 0) then
    begin
      For select distinct RafDetay.RafId from RafDetay
       where RafDetay.HamStokDetayId = :HamStokDetayId and (RafDetay.Miktar + AktarmadanIadeEdienKilo - RafDetay.IadeEdilecekMiktar - RafDetay.IadeEdilenMiktar - RafDetay.AktarmayaCikanKilo - RafDetay.PartilenenMiktar - RafDetay.RafTransferMiktar - RafDetay.SatisMiktar) > 0
       into :FRafID do
      begin
        if (Char_Length(FRafId) + Char_Length(RafIDStr) < 48) then
        begin
          if (RafIDStr <> '' ) then RafIDStr = RafIDStr || cast(', ' as varchar(2) character set UTF8);
          RafIDStr = RafIDStr || FRafId;
        end
      end

      Select Sum((Kilo + AktarmadanIadeEdienKilo - AktarmayaCikanKilo - PartilenenKilo - RafTransferKilo - IadeEdilecekKilo - IadeEdilenKilo - SatisKilo)),
             Sum(Metre - PartilenenMetre - RafTransferMetre - IadeEdilecekMetre - IadeEdilenMetre - SatisMetre),
             Sum(Adet - PartilenenAdet - IadeEdilecekAdet - IadeEdilenAdet),
             Sum(KapAdedi - PartilenenKapAdedi - RafTransferKapAdedi - IadeEdilecekKapAdedi - IadeEdilenKapAdedi) from RafDetay
        left join Raf on Raf.RafID = RafDetay.RafId
        where RafDetay.HamStokDetayID = :HamStokDetayId and Raf.PartilemeYapilabilir = 1
        into :PartilenebilirRafKilo, :PartilenebilirRafMetre, :PartilenebilirRafAdet, :PartilenebilirRafKapAdedi;

      if ((SiparisSecimiOk = 1) and (PartilenebilirRafKilo> 0 or PartilenebilirRafMetre > 0) ) then suspend;
    end
  end
