CREATE PROCEDURE BOYATEKLIFRAPORU_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ASATISTIPI INTEGER,
    AGLOBAL INTEGER,
    AMUSTERIREFERANSI_1 VARCHAR(30),
    AMUSTERIREFERANSI_2 VARCHAR(30),
    AKONTROLDURUMU INTEGER,
    AONAYDURUMU INTEGER,
    ASIPARISPROSESID INTEGER,
    AHAMID INTEGER,
    AKALITEID VARCHAR(20),
    AISLEMKAYNAGI SMALLINT)
RETURNS (
    BOYATEKLIFDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEATISID INTEGER,
    LABRECETEATISNO VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    RENKADI VARCHAR(40),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIIDSTR VARCHAR(20),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(80),
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40))
AS
begin
  for Select BoyaSiparisDetay.BoyaTeklifDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo,
             BoyaSiparisDetay.LabReceteId, LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.LabReceteAtisId, LabReceteAtis.AtisNoStr,
             BoyaSiparisDetay.RenkAdi, Desen.DesenId, Desen.Adi, BoyaSiparisDetay.RenkDerinligiId, RenkDerinligi.Adi,
             BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.HamId, Ham.Adi,
             BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo,
             BoyaSiparisDetay.MusteriKaliteAdi,
             BoyaSiparisDetay.OnayDurumu, BoyaSiparisDetay.OnayTarihi, BoyaSiparisDetay.OnaylayanKullaniciId, Ok.Adi from BoyaTeklifRaporu_1(:ABASLANGICTARIHI, :ABITISTARIHI, :AFIRMAID, :ASATISTIPI, :AGLOBAL, :AMUSTERIREFERANSI_1, :AMUSTERIREFERANSI_2, :AKONTROLDURUMU, :AONAYDURUMU, :ASIPARISPROSESID, :AHAMID, :AKALITEID, :AISLEMKAYNAGI)
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaTeklifDetayId = BoyaTeklifRaporu_1.BoyaTeklifDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
    left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
    Left Join Kullanici OK on OK.KullaniciId = BoyaSiparisDetay.OnaylayanKullaniciId
    where BoyaSiparisDetay.BoyaTeklifKayitId is not null
    into :BoyaTeklifDetayId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId, :MusteriSiparisNo, :LabReceteId, :LabReceteKodu, :LabReceteAdi,
         :LabReceteAtisId, :LabReceteAtisNo, :RenkAdi, :DesenId, :DesenAdi, :RenkDerinligiId, :RenkDerinligiIdStr, :SiparisKilo, :SiparisMetre,
         :HamId, :HamAdi, :SiparisProsesId, :SiparisProsesAdi, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
         :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    suspend;
  end
end
