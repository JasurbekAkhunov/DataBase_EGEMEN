CREATE PROCEDURE FIRMAKALITESONBESPARTI(
    AACILISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    APARTISINIFID INTEGER)
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(50),
    HK_YIKAMAONCESIEN DOUBLE PRECISION,
    HK_YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    HK_ENDENCEKMEZLIK DOUBLE PRECISION,
    HK_BOYDANCEKMEZLIK DOUBLE PRECISION,
    HK_HAMDONME DOUBLE PRECISION,
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    ENVERILEN VARCHAR(10),
    GRAMAJVERILEN VARCHAR(10),
    DONMEVERILEN VARCHAR(10),
    CEKMEZLIKENVERILEN DOUBLE PRECISION,
    CEKMEZLIKBOYVERILEN DOUBLE PRECISION,
    CEKMEZLIKENVERILEN2 DOUBLE PRECISION,
    CEKMEZLIKBOYVERILEN2 DOUBLE PRECISION)
AS
declare variable HamStokDetayID integer;
declare variable AHamID          integer;
begin
  Select HamId from PartiStok where PartiStok.PartiSinifID = :APartiSinifID into :AHamId;

  For Select First 5 Parti.PartiYil, Parti.PartiID, PartiStok.PartiSinifId, PartiStok.BoyaSiparisDetayID, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             Parti.ReceteID, Recete.Adi, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen,
             PartiStokFizikKontrol.EnVerilen, PartiStokFizikKontrol.GramajVerilen, PartiStokFizikKontrol.DonmeVerilen,
             PartiStokFizikKontrol.CekmezlikEnVerilen, PartiStokFizikKontrol.CekmezlikBoyVerilen, PartiStokFizikKontrol.CekmezlikEnVerilen2, PartiStokFizikKontrol.CekmezlikBoyVerilen2 from Parti
    join PartiStok on PartiStok.PartiKayitID = Parti.PartiKayitID and PartiStok.HamID = :AHamId and PartiStok.PartiStokIptalKayitID is null
    join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = Parti.BoyaSiparisKayitID
    join Recete on Recete.ReceteID = Parti.ReceteID
    join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
    where Parti.FirmaID = :AFirmaId and Parti.AcilisTarihi < :AAcilisTarihi
    order by Parti.AcilisTarihi desc
    into :PartiYil, :PartiID, :PartiSinifId, :BoyaSiparisDetayID, :BoyaSiparisYil, :BoyaSiparisId, :ReceteID, :ReceteAdi, :EnIstenen, :GramajIstenen,
         :EnVerilen, :GramajVerilen, :DonmeVerilen, :CekmezlikEnVerilen, :CekmezlikBoyVerilen, :CekmezlikEnVerilen2, :CekmezlikBoyVerilen2 do
  begin
    HamStokDetayID = Null;
    HK_YikamaOncesiEn = Null;
    HK_YikamaOncesiGramaj = Null;
    HK_EndenCekmezlik = Null;
    HK_BoydanCekmezlik = Null;
    HK_HamDonme = Null;

    Select Min(RafDetay.HamStokDetayID) from PartiStokDetay
      left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
      where PartiStokDetay.PartiSinifId = :PartiSinifId
      into :HamStokDetayID;

    Select HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj,
           HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme from HamStokDetay
      left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
      where HamStokDetay.HamStokDetayID = :HamStokDetayId
      into :HK_YikamaOncesiEn, :HK_YikamaOncesiGramaj, :HK_EndenCekmezlik, :HK_BoydanCekmezlik, :HK_HamDonme;

    suspend;
  end
end
