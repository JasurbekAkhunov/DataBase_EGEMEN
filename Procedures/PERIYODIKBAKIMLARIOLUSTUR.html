CREATE PROCEDURE PERIYODIKBAKIMLARIOLUSTUR(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINABAKIMTIPIID INTEGER,
    AMAKINAGRUBU VARCHAR(4),
    AMAKINAID VARCHAR(10),
    AORMEMAKINAGRUBU VARCHAR(4),
    AORMEMAKINAID VARCHAR(10),
    AIPLIKMAKINAGRUBU VARCHAR(4),
    AIPLIKMAKINAID VARCHAR(10),
    ADOKUMAMAKINAGRUBU VARCHAR(4),
    ADOKUMAMAKINAID VARCHAR(10),
    AKULLANICIID INTEGER)
RETURNS (
    MAKINABAKIMISEMRIID INTEGER,
    PERIYOTTIPI INTEGER,
    PERIYOT DOUBLE PRECISION,
    TAHMINIBASLANGIC TIMESTAMP,
    MAKINABAKIMTIPIID INTEGER,
    ADI VARCHAR(255),
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40))
AS
declare variable EklenecekPeriyot               Double Precision;
declare variable HaftalikOrtCalismaSaati        integer;
begin
  select Periyot, Adi, PeriyotTipi from MakinaBakimTipi where MakinaBakimTipiId = :AMakinaBakimTipiId into :Periyot, :Adi, :PeriyotTipi;
  TahminiBaslangic = Null;

  if (AMakinaId is not Null or AMakinaGrubu is not Null) then
  begin
    For select MakinaId, Adi, HaftalikOrtCalismaSaati from Makina where (MakinaId = :AMakinaId or :AMakinaId is Null) and (MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is Null)
        into :MakinaId, :MakinaAdi, :HaftalikOrtCalismaSaati do
    begin
      EklenecekPeriyot = 0;
      TahminiBaslangic = Null;

      if (HaftalikOrtCalismaSaati is Null) then HaftalikOrtCalismaSaati = 0;
      if (HaftalikOrtCalismaSaati = 0) then Exception KayitHatasi ' Makinanin haftalik ortalama calisma saati girilmemis...';

      if (PeriyotTipi = 1) then EklenecekPeriyot = Periyot / (HaftalikOrtCalismaSaati / 7.000000);
      else EklenecekPeriyot = Periyot;

      select first 1 TahminiBaslangic from MakinaBakimIsEmri
        where MakinaBakimIsEmri.MakinaBakimTipiId = :AMakinaBakimTipiId and
          MakinaBakimIsEmri.MakinaId = :MakinaId and
          MakinaBakimIsEmri.TahminiBaslangic >= :ABaslangicTarihi and
          MakinaBakimIsEmri.TahminiBaslangic <= :ABitisTarihi and
          MakinaBakimIsEmri.ArizaIsEmri = 0 and MakinaBakimIsEmri.Kapali = 0
        order by TahminiBaslangic desc into :TahminiBaslangic;

      if (TahminiBaslangic is Null) then
        TahminiBaslangic = ABaslangicTarihi;
      else
        TahminiBaslangic = TahminiBaslangic + EklenecekPeriyot;

      if (TahminiBaslangic is not null) then
      begin
        While (TahminiBaslangic <= ABitisTarihi) do
        begin
          select MakinaBakimIsEmriId, MakinaBakimTipiId from BakimTipindenIsEmriOlustur(:AMakinaBakimTipiId, :MakinaId, :AOrmeMakinaId, :AIplikMakinaId, :ADokumaMakinaId, :TahminiBaslangic, :AKullaniciId) into :MakinaBakimIsEmriId, :MakinaBakimTipiId;
          TahminiBaslangic = TahminiBaslangic + EklenecekPeriyot;
          Suspend;
        end
      end
    end
  end
  else if (AOrmeMakinaId is not Null or AOrmeMakinaGrubu is not Null) then
  begin
    For select OrmeMakinaId, Adi, HaftalikOrtCalismaSaati from OrmeMakina where (OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is Null) and (MakinaGrupID = :AOrmeMakinaGrubu or :AOrmeMakinaGrubu is Null)
        into :MakinaId, :MakinaAdi, :HaftalikOrtCalismaSaati do
    begin
      EklenecekPeriyot = 0;
      TahminiBaslangic = Null;

      if (HaftalikOrtCalismaSaati is Null) then HaftalikOrtCalismaSaati = 0;
      if (HaftalikOrtCalismaSaati = 0) then Exception KayitHatasi ' Makinanin haftalik ortalama calisma saati girilmemis...';

      if (PeriyotTipi = 1) then EklenecekPeriyot = Periyot / (HaftalikOrtCalismaSaati / 7.000000);
      else EklenecekPeriyot = Periyot;

      select first 1 TahminiBaslangic from MakinaBakimIsEmri
        where MakinaBakimIsEmri.MakinaBakimTipiId = :AMakinaBakimTipiId and
          MakinaBakimIsEmri.OrmeMakinaId = :MakinaId and
          MakinaBakimIsEmri.TahminiBaslangic >= :ABaslangicTarihi and
          MakinaBakimIsEmri.TahminiBaslangic <= :ABitisTarihi and
          MakinaBakimIsEmri.ArizaIsEmri = 0 and MakinaBakimIsEmri.Kapali = 0
        order by TahminiBaslangic desc into :TahminiBaslangic;

      if (TahminiBaslangic is Null) then
        TahminiBaslangic = ABaslangicTarihi;
      else
        TahminiBaslangic = TahminiBaslangic + EklenecekPeriyot;

      if (TahminiBaslangic is not null) then
      begin
        While (TahminiBaslangic <= ABitisTarihi) do
        begin
          select MakinaBakimIsEmriId, MakinaBakimTipiId from BakimTipindenIsEmriOlustur(:AMakinaBakimTipiId, :AMakinaId, :MakinaId, :AIplikMakinaId, :ADokumaMakinaId, :TahminiBaslangic, :AKullaniciId) into :MakinaBakimIsEmriId, :MakinaBakimTipiId;
          TahminiBaslangic = TahminiBaslangic + EklenecekPeriyot;
          Suspend;
        end
      end
    end
  end
  else if (AIplikMakinaId is not Null or AIplikMakinaGrubu is not Null) then
  begin
    For select IplikMakinaId, Adi, HaftalikOrtCalismaSaati from IplikMakina where (IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is Null) and (MakinaGrupID = :AIplikMakinaGrubu or :AIplikMakinaGrubu is Null)
        into :MakinaId, :MakinaAdi, :HaftalikOrtCalismaSaati do
    begin
      EklenecekPeriyot = 0;
      TahminiBaslangic = Null;

      if (HaftalikOrtCalismaSaati is Null) then HaftalikOrtCalismaSaati = 0;
      if (HaftalikOrtCalismaSaati = 0) then Exception KayitHatasi ' Makinanin haftalik ortalama calisma saati girilmemis...';

      if (PeriyotTipi = 1) then EklenecekPeriyot = Periyot / (HaftalikOrtCalismaSaati / 7.000000);
      else EklenecekPeriyot = Periyot;

      select first 1 TahminiBaslangic from MakinaBakimIsEmri
        where MakinaBakimIsEmri.MakinaBakimTipiId = :AMakinaBakimTipiId and
          MakinaBakimIsEmri.IplikMakinaId = :MakinaId and
          MakinaBakimIsEmri.TahminiBaslangic >= :ABaslangicTarihi and
          MakinaBakimIsEmri.TahminiBaslangic <= :ABitisTarihi and
          MakinaBakimIsEmri.ArizaIsEmri = 0 and MakinaBakimIsEmri.Kapali = 0
        order by TahminiBaslangic desc into :TahminiBaslangic;

      if (TahminiBaslangic is Null) then
        TahminiBaslangic = ABaslangicTarihi;
      else
        TahminiBaslangic = TahminiBaslangic + EklenecekPeriyot;

      if (TahminiBaslangic is not null) then
      begin
        While (TahminiBaslangic <= ABitisTarihi) do
        begin
          select MakinaBakimIsEmriId, MakinaBakimTipiId from BakimTipindenIsEmriOlustur(:AMakinaBakimTipiId, :AMakinaId, :AOrmeMakinaId, :MakinaId, :ADokumaMakinaId, :TahminiBaslangic, :AKullaniciId) into :MakinaBakimIsEmriId, :MakinaBakimTipiId;
          TahminiBaslangic = TahminiBaslangic + EklenecekPeriyot;
          Suspend;
        end
      end
    end
  end
  else if (ADokumaMakinaId is not Null or ADokumaMakinaGrubu is not Null) then
  begin
    For select DokumaMakinaId, Adi, HaftalikOrtCalismaSaati from DokumaMakina where (DokumaMakinaId = :ADokumaMakinaId or :ADokumaMakinaId is Null) and (MakinaGrupID = :ADokumaMakinaGrubu or :ADokumaMakinaGrubu is Null)
        into :MakinaId, :MakinaAdi, :HaftalikOrtCalismaSaati do
    begin
      EklenecekPeriyot = 0;
      TahminiBaslangic = Null;

      if (HaftalikOrtCalismaSaati is Null) then HaftalikOrtCalismaSaati = 0;
      if (HaftalikOrtCalismaSaati = 0) then Exception KayitHatasi ' Makinanin haftalik ortalama calisma saati girilmemis...';

      if (PeriyotTipi = 1) then EklenecekPeriyot = Periyot / (HaftalikOrtCalismaSaati / 7.000000);
      else EklenecekPeriyot = Periyot;

      select first 1 TahminiBaslangic from MakinaBakimIsEmri
        where MakinaBakimIsEmri.MakinaBakimTipiId = :AMakinaBakimTipiId and
          MakinaBakimIsEmri.DokumaMakinaId = :MakinaId and
          MakinaBakimIsEmri.TahminiBaslangic >= :ABaslangicTarihi and
          MakinaBakimIsEmri.TahminiBaslangic <= :ABitisTarihi and
          MakinaBakimIsEmri.ArizaIsEmri = 0 and MakinaBakimIsEmri.Kapali = 0
        order by TahminiBaslangic desc into :TahminiBaslangic;

      if (TahminiBaslangic is Null) then
        TahminiBaslangic = ABaslangicTarihi;
      else
        TahminiBaslangic = TahminiBaslangic + EklenecekPeriyot;

      if (TahminiBaslangic is not null) then
      begin
        While (TahminiBaslangic <= ABitisTarihi) do
        begin
          select MakinaBakimIsEmriId, MakinaBakimTipiId from BakimTipindenIsEmriOlustur(:AMakinaBakimTipiId, :AMakinaId, :AOrmeMakinaId, :AIplikMakinaId, :MakinaId, :TahminiBaslangic, :AKullaniciId) into :MakinaBakimIsEmriId, :MakinaBakimTipiId;
          TahminiBaslangic = TahminiBaslangic + EklenecekPeriyot;
          Suspend;
        end
      end
    end
  end
