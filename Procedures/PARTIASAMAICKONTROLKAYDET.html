CREATE PROCEDURE PARTIASAMAICKONTROLKAYDET(
    APARTIASAMALARIID INTEGER,
    APARTISINIFID INTEGER,
    AICKONTROLTIPIID INTEGER,
    APERSONELID INTEGER,
    AMAKINAID VARCHAR(10),
    ASONUC VARCHAR(15))
AS
declare variable SonucTumPartilerIcinAynimi integer;
declare variable AsilMakinaID      MakinaKoduSize;
begin
  Select AsilMakinaID from Makina where Makina.MakinaId = :AMakinaId into :AsilMakinaID;
  if (AsilMakinaID is Not Null) then AMakinaID = :AsilMakinaID;
  select SonucTumPartilerIcinAynimi from IcKontrolTipi where IcKontrolTipi.IcKontrolTipiId = :AIcKontrolTipiId into :SonucTumPartilerIcinAynimi;

  if (SonucTumPartilerIcinAynimi = 0) then
  begin
    Update PartiAsamaIcKontrol set Sonuc = :ASonuc, PersonelId = :APersonelId
      where PartiAsamaIcKontrol.PartiAsamalariID = :APartiAsamalariID and PartiAsamaIcKontrol.PartiSinifId = :APartiSinifId and
            PartiAsamaIcKontrol.IcKontrolTipiId = :AIcKontrolTipiId;
  end
  else
  begin
    For Select PartiSinifId from PartiAsamaStokDetay
      where PartiAsamaStokDetay.PartiAsamalariID = :APartiAsamalariID and
            PartiAsamaStokDetay.MakinaId = :AMakinaId and PartiAsamaStokDetay.CikisTarihi is null
      into :APartiSinifId do
    begin
      Update PartiAsamaIcKontrol set Sonuc = :ASonuc, PersonelId = :APersonelId
        where PartiAsamaIcKontrol.PartiAsamalariID = :APartiAsamalariID and PartiAsamaIcKontrol.PartiSinifId = :APartiSinifId and
              PartiAsamaIcKontrol.IcKontrolTipiId = :AIcKontrolTipiId;
    end
  end
end
