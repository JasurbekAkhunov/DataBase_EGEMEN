CREATE PROCEDURE IPIMALATAARAVERMERAPORUANLIK
RETURNS (
    IPLIKIMALATAARAVERMEID INTEGER,
    IPLIKMAKINAID VARCHAR(10),
    BASLADIIPLIKMAKINADURUMUID INTEGER,
    ARAVERDIRENPERSONELID INTEGER,
    IPLIKARAVERMESEBEBIID INTEGER,
    BITTIIPLIKMAKINADURUMUID INTEGER,
    ARAYIBITIRENPERSONELID INTEGER,
    TARIH TIMESTAMP,
    PERSONELADI VARCHAR(40),
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30))
AS
begin
  For Select IplikImalataAraVerme.IplikImalataAraVermeId, IplikImalataAraVerme.IplikMakinaId, IplikImalataAraVerme.BasladiIplikMakinaDurumuId,
             IplikImalataAraVerme.AraVerdirenPersonelId, IplikImalataAraVerme.IplikAraVermeSebebiId, IplikImalataAraVerme.BittiIplikMakinaDurumuId,
             IplikImalataAraVerme.ArayiBitirenPersonelId, IplikMakinaDurumu.Tarih, Personel.Adi, (CURRENT_DATE + CURRENT_TIME-IPLIKMAKINADURUMU.TARIH) from IplikImalataAraVerme
  left join IplikMakinaDurumu on IplikMakinaDurumu.IplikMakinaDurumuId = IplikImalataAraVerme.BasladiIplikMakinaDurumuId
  left join Personel on Personel.PersonelId = IplikImalataAraVerme.AraVerdirenPersonelId
  where IplikImalataAraVerme.BittiIplikMakinaDurumuId is null
  Order By IplikImalataAraVerme.IplikAraVermeSebebiId, IplikImalataAraVerme.IplikMAKINAID
  into :IplikImalataAraVermeId, :IplikMakinaId, :BasladiIplikMakinaDurumuId, :AraVerdirenPersonelId, :IplikAraVermeSebebiId, :BittiIplikMakinaDurumuId,
       :ArayiBitirenPersonelId, :Tarih, :PersonelAdi, :GecenSure do
  begin
    GecenSureStr = TarihToStr(GecenSure);
    suspend;
  end
end
