CREATE PROCEDURE RENKDERINLIKHAMTRANSFERGETIR
RETURNS (
    RENKDERINLIKHAMTRANSFERKODID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI VARCHAR(20),
    MUHASEBETRANSFERKODU VARCHAR(55),
    TRANSFERKODU VARCHAR(55))
AS
begin
  for Select Ham.HamId, Ham.Adi, HamTipiID, RenkDerinligi.RenkDerinligiID, RenkDerinligi.Adi From RenkDerinligi, Ham
    into :HamID, : HamAdi, :HamTipiID, :RenkDerinligiID, :RenkDerinligiAdi do
  begin
    RenkDerinlikHamTransferKodID = Null;
    MuhasebeTransferKodu = Null;
    TransferKodu = Null;
    Select RenkDerinlikHamTransferKodID, MuhasebeTransferKodu, TransferKodu from RenkDerinlikHamTransferKod
      where HamId = : HamID and RenkDerinligiID = :RenkDerinligiID
      into :RenkDerinlikHamTransferKodID, :MuhasebeTransferKodu, :TransferKodu;

    suspend;
  end
end
