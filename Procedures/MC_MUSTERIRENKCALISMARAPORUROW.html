CREATE PROCEDURE MC_MUSTERIRENKCALISMARAPORUROW(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKULLANICIID INTEGER,
    ALABRECETEATISID INTEGER)
RETURNS (
    LABRECETEATISID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    MUSTERIRENKKODU VARCHAR(30),
    FASON INTEGER,
    FASONSATISSTR VARCHAR(5),
    HAMADI VARCHAR(255),
    CALISMATARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    GIDISTARIHI TIMESTAMP,
    DURUMUSTR VARCHAR(6),
    OKEYKRITIKTARIHI TIMESTAMP,
    NOTU VARCHAR(1023))
AS
declare variable FAtisNoStr         Varchar(20);
declare variable FAtisNotu    Blob;
declare variable Okey     integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  if (ALabReceteAtisID is Null) then
  begin
    For Select LabRecete.LabReceteKodu, LabReceteAtis.AtisNoStr, LabRecete.Adi, LabRecete.MusteriRenkKodu,
               LabRecete.Fason, Ham.Adi, LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi,
               LabReceteAtis.GidisTarihi, LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi, LabReceteAtis.Notu,
               LabReceteAtis.LabReceteAtisID from LabReceteCalisma
      left join LabRecete on LabRecete.LabReceteID = LabReceteCalisma.LabReceteId
      left join LabReceteAtis on LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
      left join Ham on Ham.HamId = LabReceteCalisma.HamId
      where LabReceteCalisma.FirmaId = :AFirmaId and LabReceteCalisma.CalismaTarihi >= :ABaslangicTarihi and
            LabReceteCalisma.CalismaTarihi <= :ABitisTarihi and LabReceteAtis.LabReceteAtisID is not null
      into :LabReceteKodu, :FAtisNoStr, :LabReceteAdi, :MusteriRenkKodu, :Fason, :HamAdi, :CalismaTarihi, :TerminTarihi, :GidisTarihi,
            :DurumuStr, :OkeyKritikTarihi, :FAtisNotu, :LabReceteAtisID do
    begin
      LabReceteKodu = LabReceteKodu || '.' || FAtisNoStr;

      if (Fason = 0) then FasonSatisStr = 'Satış';
      else FasonSatisStr = 'Fason';

      if (char_length(FAtisNotu) > 1023) then Notu = Left(FAtisNotu, 1023);
      else Notu = Left(FAtisNotu, char_length(FAtisNotu));

      suspend;
    end
  end
  else
  begin
    For Select LabRecete.LabReceteKodu, LabReceteAtis.AtisNoStr, LabRecete.Adi, LabRecete.MusteriRenkKodu,
               LabRecete.Fason, Ham.Adi, LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi,
               LabReceteAtis.GidisTarihi, LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi, LabReceteAtis.Notu,
               LabReceteAtis.LabReceteAtisID from LabReceteCalisma
      left join LabRecete on LabRecete.LabReceteID = LabReceteCalisma.LabReceteId
      left join LabReceteAtis on LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
      left join Ham on Ham.HamId = LabReceteCalisma.HamId
      where LabReceteAtis.LabReceteAtisID = :ALabReceteAtisID
      into :LabReceteKodu, :FAtisNoStr, :LabReceteAdi, :MusteriRenkKodu, :Fason, :HamAdi, :CalismaTarihi, :TerminTarihi, :GidisTarihi,
            :DurumuStr, :OkeyKritikTarihi, :FAtisNotu, :LabReceteAtisID do
    begin
      LabReceteKodu = LabReceteKodu || '.' || FAtisNoStr;

      if (Fason = 0) then FasonSatisStr = 'Satış';
      else FasonSatisStr = 'Fason';

      if (char_length(FAtisNotu) > 1023) then Notu = Left(FAtisNotu, 1023);
      else Notu = Left(FAtisNotu, char_length(FAtisNotu));

      suspend;
    end
  end

end
