CREATE PROCEDURE HAMSTOKMAMULYARIMAMULSARFRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFASON INTEGER,
    AFIRMAID VARCHAR(20),
    AHAMTIPIID INTEGER,
    ASARFTIPIID INTEGER,
    ARAPORTIPI INTEGER,
    AREFERANSKODU VARCHAR(30),
    AIRSALIYENO IRSALIYENOSIZE,
    AKALINLIKID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKUMASCINSIID INTEGER,
    ALOT VARCHAR(150),
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    SARFYERI VARCHAR(20),
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    HAMSTOKMASTERID INTEGER,
    SARFTIPIID INTEGER,
    SARFTIPIADI VARCHAR(30),
    HAMADI VARCHAR(255),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    HAMTIPIID INTEGER,
    REFERANSKODU VARCHAR(30),
    RAFID VARCHAR(10),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYEGRUPID VARCHAR(4),
    LOT VARCHAR(150),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LAMINASYONPARTIYIL INTEGER,
    LAMINASYONPARTIID INTEGER,
    LAMINASYONPARTIIDSTR VARCHAR(30),
    LAMINASYONHAMADI VARCHAR(255),
    DESENID VARCHAR(20),
    VARYANTADI VARCHAR(30),
    DESENVARYANTID INTEGER,
    LABRECETEKODU VARCHAR(20),
    ATISNOSTR VARCHAR(20),
    KALITEADI VARCHAR(255),
    BOYASIPARISID INTEGER)
AS
declare variable PartiSinifId integer;
declare variable HamStokFason integer;
declare variable PartiFason integer;
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;

  For Select HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo, HamStokSarfFisi.FisTarihi, HamStokSarfFisi.SarfTipiId, SarfTipi.Adi, Ham.Adi,
             Ham.ReferansKodu, HamStokSarfFisiDetay.Kilo, HamStokSarfFisiDetay.Metre,
             HamStokSarfFisiDetay.PartiSinifId, Parti.PartiYil, Parti.PartiId, HamStokIrsaliye.IrsaliyeGrupId, HamStokIrsaliye.IrsaliyeNo,
             HamStokIrsaliye.Fason, Parti.Fason, Firma.FirmaId, Firma.Adi, RafDetay.RafId, HamStokMaster.Lot, HamStokMaster.HamStokMasterID, HamStokSarfFisi.Notu,
             LaminasyonP.PartiYil, LaminasyonP.PartiId, LaminasyonHam.Adi,
             BoyaSiparisDetay.DesenId, DesenVaryant.VaryantAdi, DesenVaryant.DesenVaryantId, BoyaSiparisDetay.LabReceteKodu,LabReceteAtis.AtisNoStr, Kalite.KaliteAdi, BoyaSiparisDetay.BoyaSiparisKayitId from HamStokSarfFisi
  left join SarfTipi on SarfTipi.SarfTipiId = HamStokSarfFisi.SarfTipiId
  left join HamStokSarfFisiDetay on HamStokSarfFisiDetay.HamStokSarfFisiId = HamStokSarfFisi.HamStokSarfFisiId
  left join Ham on Ham.HamId = HamStokSarfFisiDetay.HamId
  left join PartiStok on PartiStok.PartiSinifId = HamStokSarfFisiDetay.PartiSinifId
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  left join RafDetay on RafDetay.RafDetayID = HamStokSarfFisiDetay.RafDetayId
  left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = RafDetay.HamStokIrsaliyeId
  left join Firma on Firma.FirmaId = Parti.FirmaId or Firma.FirmaId = HamStokIrsaliye.FirmaId
  left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamaStokDetayId = HamStokSarfFisiDetay.PartiAsamaStokDetayId
  left join PartiStok LaminasyonPS on LaminasyonPS.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
  left join Parti LaminasyonP on LaminasyonP.PartiKayitId = LaminasyonPs.PartiKayitId
  left join Ham LaminasyonHam on LaminasyonHam.HamId = LaminasyonPs.HamId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join Kalite on Kalite.KaliteId = PartiStok.Kaliteid
  left join DesenVaryant on DesenVaryant.desenvaryantid = Parti.DesenVaryantId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisID = BoyaSiparisDetay.LabReceteAtisId
  where HamStokSarfFisi.FisTarihi >= :ABaslangicTarihi and HamStokSarfFisi.FisTarihi <= :ABitisTarihi and
        (HamStokSarfFisi.SarfTipiId = :ASarfTipiId or :ASarfTipiId is null) and
        ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
          (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or
          ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 2) or
          (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 3) or :AHamTipiId is null) and
        (Ham.KalinlikId = :AKalinlikId or :AKalinlikId is null) and
        (Ham.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is null) and
        (Ham.KumasCinsiId = :AKumasCinsiId or :AKumasCinsiId is null) and
        (HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo = 0) and
        (HamStokMaster.Lot = :ALot or :ALot is null) and
        (HamStokSarfFisiDetay.HamStokSarfFisiDetayId is not null) and
        (Ham.ReferansKodu = :AReferansKodu or :AReferansKodu is null) and
        ( (:ARaporTipi = 0 and HamStokSarfFisiDetay.RafDetayId is not null) or
          (:ARaporTipi = 1 and HamStokSarfFisiDetay.PartiSinifId is not null) or :ARaporTipi is null) and
        (Firma.FirmaID = :AFirmaId or :AFirmaID is null) and
        (HamStokMaster.HamStokMasterID = :AHamStokMasterID or :AHamStokMasterID = 0)
  order by HamStokSarfFisi.FisTarihi, HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo
  into :FisYil, :FisNo, :FisTarihi, :SarfTipiId, :SarfTipiAdi, :HamAdi, :ReferansKodu, :Kilo, :Metre,
       :PartiSinifId, :PartiYil, :PartiId, :IrsaliyeGrupId, :IrsaliyeNo, :HamStokFason, :PartiFason,
       :FirmaId, :FirmaAdi, :RafId, :Lot, :HamStokMasterID, :Notu, :LaminasyonPartiYil, :LaminasyonPartiId, LaminasyonHamAdi,:DesenId, :VaryantAdi,
       :DesenVaryantId, :LabReceteKodu,:AtisNoStr,:KaliteAdi, :BoyaSiparisId do

  begin
    if (PartiId is not null) then
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') '  || EgemenIntegerToStr(PartiId);
      Fason = PartiFason;
    end
    else
    begin
      PartiIdStr = '';
      Fason = HamStokFason;
    end

    if (LaminasyonPartiId is not null) then
      LaminasyonPartiIdStr = '(' || EgemenIntegerToStr(LaminasyonPartiYil) || ') '  || EgemenIntegerToStr(LaminasyonPartiId);
    else
      LaminasyonPartiIdStr = '';

    if (PartiSinifId is not null) then
      SarfYeri = 'Mamul';
    else
      SarfYeri = 'Ham Stok';

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';

    if (AFason = Fason or AFason is null) then suspend;
  end
end
