CREATE PROCEDURE URUNSARFIYATRAPORUPARTILI_3(
    ARAPORTIPI INTEGER,
    AFIRMAID VARCHAR(20),
    ABASLANGICACILISTARIHI TIMESTAMP,
    ABITISACILISTARIHI TIMESTAMP,
    ABASLANGICRECETEYAZILISTARIHI TIMESTAMP,
    ABITISRECETEYAZILISTARIHI TIMESTAMP,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICURUNGRUP1ID VARCHAR(4),
    ABITISURUNGRUP1ID VARCHAR(4),
    ABASLANGICURUNGRUP2ID VARCHAR(4),
    ABITISURUNGRUP2ID VARCHAR(4),
    ABASLANGICURUNGRUP3ID VARCHAR(4),
    ABITISURUNGRUP3ID VARCHAR(4),
    ABASLANGICURUNGRUP4ID VARCHAR(4),
    ABITISURUNGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    ARAPORDOVIZID VARCHAR(5),
    AIMALATIBITMISOLANLAR INTEGER,
    APARTITIPIID INTEGER,
    AFASON INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTITAMIRID INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    TEORIKIMALATSURESI DOUBLE PRECISION,
    TEORIKIMALATSURESISTR VARCHAR(30),
    URETIMMALIYETI DOUBLE PRECISION,
    URETIMMALIYETIDOVIZID VARCHAR(5),
    IMALATSURESI DOUBLE PRECISION,
    IMALATSURESISTR VARCHAR(30),
    MAKINAID VARCHAR(10),
    BANYOCARPANI1 DOUBLE PRECISION,
    BANYOCARPANI2 DOUBLE PRECISION,
    BANYOCARPANI3 DOUBLE PRECISION,
    BANYOCARPANI4 DOUBLE PRECISION,
    ILAVEPROSES INTEGER,
    ILAVEPROSESSTR VARCHAR(4),
    ILAVEPROSESSEBEBIID INTEGER,
    ILAVEPROSESSEBEBIADI VARCHAR(20),
    DOGALGAZSARFIYATI DOUBLE PRECISION,
    ELEKTIRIKSARFIYATI DOUBLE PRECISION,
    OKUNANSUSARFIYATI DOUBLE PRECISION,
    HESAPLANANELEKTIRIKSARFIYATI DOUBLE PRECISION,
    HESAPLANANDOGALGAZSARFIYATI DOUBLE PRECISION,
    HESAPLANANSUSARFIYATI DOUBLE PRECISION,
    HESAPLANANBUHARSARFIYATI DOUBLE PRECISION,
    ASAMAKILO DOUBLE PRECISION,
    GIRISKILO DOUBLE PRECISION,
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP)
AS
declare variable AraVermeSuresi        Double Precision;
declare variable FUretimMaliyeti        Double Precision;
declare variable FUretimMaliyetiDovizId varchar(5);
declare variable FUretimMaliyetiKg        Double Precision;
declare variable FUretimMaliyetiKgDovizId varchar(5);
declare variable PartiAsamaMaliyetHesapTipi integer;
declare variable PartiKilo                  Double Precision;
declare variable BoyaAsamasimi              integer;
declare variable ToplamSicaklikDegisimi     Double Precision;
declare variable BanyoSayisi                Double Precision;
begin
  Select PartiAsamaMaliyetHesapTipi from sistem where sistemid = 1 into :PartiAsamaMaliyetHesapTipi;


  for select UrunSarfiyatRaporuPartili_1.PartiKayitId, PartiAsamalari.PartiAsamalariId, PartiAsamaStokDetay.MakinaId, PartiAsamalari.ReceteId, Recete.Adi, PartiAsamalari.AsamaId, Asama.Adi,
         PartiAsamalari.UretimMaliyeti, PartiAsamalari.UretimMaliyetiDovizId, PartiAsamalari.UretimMaliyetiKg, PartiAsamalari.UretimMaliyetiKgDovizId, PartiAsamalari.PartiTamirId,
         TamirTipi.Adi, PartiAsamalari.BanyoCarpani, PartiAsamalari.BanyoCarpani2, PartiAsamalari.BanyoCarpani3, PartiAsamalari.BanyoCarpani4,
         PartiAsamalari.IlaveProses, PartiAsamalari.IlaveProsesSebebiId, IlaveProsesSebebi.Adi, Asama.BoyaAsamasimi,
         Recete.ToplamSicaklikDegisimi, Recete.BanyoSayisi, sum(PartiAsamaStokDetay.Kilo), sum(PartiAsamaStok.GirisKilo),
         Max(PartiAsamaStok.TeorikImalatSuresi), Min(PartiAsamaStok.GirisTarihi), Max(PartiAsamaStok.CikisTarihi), Max(PartiAsamaStokDetay.AraVermeSuresi) from UrunSarfiyatRaporuPartili_1(:ARaporTipi, :AFirmaId, :ABaslangicAcilisTarihi, :ABitisAcilisTarihi, :ABaslangicReceteYazilisTarihi, :ABitisReceteYazilisTarihi, :ABaslangicPartiYil, :ABaslangicPartiId, :ABitisPartiYil, :ABitisPartiId, :ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ABaslangicUrunId, :ABitisUrunId,
                                     :ABaslangicUrunGrup1Id, :ABitisUrunGrup1Id, :ABaslangicUrunGrup2Id, :ABitisUrunGrup2Id, :ABaslangicUrunGrup3Id, :ABitisUrunGrup3Id, :ABaslangicUrunGrup4Id, :ABitisUrunGrup4Id, :AUrunAralikli, :AUrunGrup1Aralikli,
                                     :AUrunGrup2Aralikli, :AUrunGrup3Aralikli, :AUrunGrup4Aralikli, :ARaporDovizID, :AImalatiBitmisOlanlar, :APartiTipiId, :AFason)
        left join PartiAsamalari on PartiAsamalari.PartiKayitId = UrunSarfiyatRaporuPartili_1.PartiKayitId
        left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
        left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
        left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join IlaveProsesSebebi on IlaveProsesSebebi.IlaveProsesSebebiId = PartiAsamalari.IlaveProsesSebebiID
        where PartiAsamaStok.AsamayaDahil = 1
        group by UrunSarfiyatRaporuPartili_1.PartiKayitId, PartiAsamalari.Sira, PartiAsamalari.PartiAsamalariId, PartiAsamaStokDetay.MakinaId, PartiAsamalari.ReceteId, Recete.Adi, PartiAsamalari.AsamaId, Asama.Adi,
                 PartiAsamalari.UretimMaliyeti, PartiAsamalari.UretimMaliyetiDovizId, PartiAsamalari.UretimMaliyetiKg, PartiAsamalari.UretimMaliyetiKgDovizId, PartiAsamalari.PartiTamirId,
                 TamirTipi.Adi, PartiAsamalari.BanyoCarpani, PartiAsamalari.BanyoCarpani2, PartiAsamalari.BanyoCarpani3, PartiAsamalari.BanyoCarpani4,
                 PartiAsamalari.IlaveProses, PartiAsamalari.IlaveProsesSebebiId, IlaveProsesSebebi.Adi, Asama.BoyaAsamasimi,
                 Recete.ToplamSicaklikDegisimi, Recete.BanyoSayisi
        order by UrunSarfiyatRaporuPartili_1.PartiKayitId, PartiAsamalari.Sira
        into :PartiKayitId, :PartiAsamalariId, :MakinaId, :ReceteId, :ReceteAdi, :AsamaId, :AsamaAdi, :FUretimMaliyeti, :FUretimMaliyetiDovizId, :FUretimMaliyetiKg, :FUretimMaliyetiKgDovizId,
             :PartiTamirId, :TamirTipiAdi, :BanyoCarpani1, :BanyoCarpani2, :BanyoCarpani3, :BanyoCarpani4,
             :IlaveProses, :IlaveProsesSebebiId, :IlaveProsesSebebiAdi, :BoyaAsamasimi, :ToplamSicaklikDegisimi, :BanyoSayisi,
             :AsamaKilo, :GirisKilo,
             :TeorikImalatSuresi, :GirisTarihi, :CikisTarihi, :AraVermeSuresi do
  begin
    DogalgazSarfiyati    = Null;
    ElektirikSarfiyati   = Null;
    OkunanSuSarfiyati    = Null;


    if (PartiAsamalariId = -1) then
    begin
      PartiAsamalariId = -1;
      AsamaAdi = 'Manuel Sarfiyat (Aşama Bilinmiyor)';
      ImalatSuresi = Null;
      ImalatSuresiStr = '';
      TeorikImalatSuresi = Null;
      TeorikImalatSuresiStr = '';
    end
    else
    begin
      if (PartiAsamaMaliyetHesapTipi = 0) then
      begin
        UretimMaliyeti = FUretimMaliyeti;
        UretimMaliyetiDovizId = FUretimMaliyetiDovizId;
      end
      else
      begin
        UretimMaliyeti = FUretimMaliyetiKg;
        UretimMaliyetiDovizId = FUretimMaliyetiKgDovizId;
      end

      ImalatSuresi = CikisTarihi - GirisTarihi;
      ImalatSuresi = ImalatSuresi - AraVermeSuresi;

      if (ImalatSuresi <> 0) then ImalatSuresiStr = TarihtoStr(ImalatSuresi);
      else ImalatSuresiStr = '';

      TeorikImalatSuresi = TeorikImalatSuresi / 1440;
      if (TeorikImalatSuresi <> 0) then TeorikImalatSuresiStr = TarihtoStr(TeorikImalatSuresi);
      else TeorikImalatSuresiStr = '';
    end

    if (IlaveProses = 1) then
      IlaveProsesStr = 'Evet';
    else
      IlaveProsesStr = '';

    Select DogalgazSarfiyati, ElektirikSarfiyati, OkunanSuSarfiyati from PartiEnerjiSarfiyatiHesapla2(:PartiKayitID, :PartiAsamalariID, :MakinaId) into :DogalgazSarfiyati, :ElektirikSarfiyati, :OkunanSuSarfiyati;


    Select Sum(Makina.KiloWatt * (PartiAsamaStokDetay.CikisTarihi - PartiAsamaStokDetay.GirisTarihi - PartiAsamaStokDetay.AraVermeSuresi)* 24),
             Sum(Makina.DogalgazM3 * PartiAsamaStokDetay.Kilo),
             Sum(Makina.BuharSarfiyatiKg * PartiAsamaStokDetay.Kilo) from PartiAsamaStokDetay
        left join Makina on Makina.MakinaID = PartiAsamaStokDetay.MakinaId
        where PartiAsamaStokDetay.PartiAsamalariID = :PartiAsamalariID and PartiAsamaStokDetay.CikisTarihi is not null
        into :HesaplananElektirikSarfiyati, :HesaplananDogalgazSarfiyati, :HesaplananBuharSarfiyati;

    if (BoyaAsamasimi = 1) then
      HesaplananBuharSarfiyati = ToplamSicaklikDegisimi * BanyoCarpani1 / 540;

    HesaplananElektirikSarfiyati = Yuvarla(HesaplananElektirikSarfiyati, 1000);
    HesaplananSuSarfiyati = BanyoSayisi * BanyoCarpani1;
    suspend;
  end

  for select UrunSarfiyatRaporuPartili_2.PartiKayitId, UrunSarfiyatRaporuPartili_2.PartiAsamalariId from UrunSarfiyatRaporuPartili_2(:ARaporTipi, :AFirmaId, :ABaslangicAcilisTarihi, :ABitisAcilisTarihi, :ABaslangicReceteYazilisTarihi, :ABitisReceteYazilisTarihi, :ABaslangicPartiYil, :ABaslangicPartiId, :ABitisPartiYil, :ABitisPartiId, :ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ABaslangicUrunId, :ABitisUrunId,
                                     :ABaslangicUrunGrup1Id, :ABitisUrunGrup1Id, :ABaslangicUrunGrup2Id, :ABitisUrunGrup2Id, :ABaslangicUrunGrup3Id, :ABitisUrunGrup3Id, :ABaslangicUrunGrup4Id, :ABitisUrunGrup4Id, :AUrunAralikli, :AUrunGrup1Aralikli,
                                     :AUrunGrup2Aralikli, :AUrunGrup3Aralikli, :AUrunGrup4Aralikli, :ARaporDovizID, :AImalatiBitmisOlanlar, :APartiTipiId, :AFason)


        where UrunSarfiyatRaporuPartili_2.PartiAsamalariId = -1
        group by UrunSarfiyatRaporuPartili_2.PartiKayitId, UrunSarfiyatRaporuPartili_2.PartiAsamalariId
        into :PartiKayitId, :PartiAsamalariId do
  begin
    MakinaId = Null;
    ReceteId  = Null;
    ReceteAdi = Null;
    AsamaId  = Null;
    UretimMaliyeti = Null;
    UretimMaliyetiDovizId = Null;

    BanyoCarpani1  = Null;
    BanyoCarpani2  = Null;
    BanyoCarpani3  = Null;
    BanyoCarpani4  = Null;

    PartiTamirId = Null;
    PartiAsamalariId = -1;
    AsamaAdi = 'Manuel Sarfiyat (Aşama Bilinmiyor)';
    ImalatSuresi = Null;
    ImalatSuresiStr = '';
    TeorikImalatSuresi = Null;
    TeorikImalatSuresiStr = '';
    IlaveProses           = Null;
    IlaveProsesStr        = Null;
    IlaveProsesSebebiId   = Null;
    IlaveProsesSebebiAdi  = Null;
    DogalgazSarfiyati     = Null;
    ElektirikSarfiyati    = Null;
    OkunanSuSarfiyati     = Null;
    HesaplananElektirikSarfiyati = Null;
    HesaplananDogalgazSarfiyati  = Null;
    HesaplananBuharSarfiyati     = Null;
    HesaplananSuSarfiyati = null;
    GirisTarihi = null;
    CikisTarihi = null;

    suspend;
  end
end
