CREATE PROCEDURE LABDANRECETEYEBOYALARICEK(
    ARECETEID VARCHAR(20),
    ALABRECETEATISID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable LabReceteGrupId          integer;
declare variable MinSira                   integer;
declare variable Sira                      integer;

declare variable LUrunId                   varchar(20);
declare variable LUrunBirimId              integer;
declare variable LMiktarYuzde              Double precision;
declare variable LUrunTipiId               integer;
declare variable BoyaTipiId               integer;
declare variable FAtisRenkSira  integer;
declare variable FAtisRenkColor integer;
declare variable FAtisRenkKodu        varchar(50);
declare variable FAtisRenkAdi         varchar(120);

declare variable BoyaKimyasalUrunTipiId      integer;
declare variable FBoyaEklendi                integer;
declare variable FReceteOtomatikIstekOlustur integer;
declare variable FWReceteAltAsamaId          integer;
declare variable ReceteAsamaId               integer;
declare variable FMinSira                    integer;
declare variable FMaxSira                    integer;
declare variable FMaxSira2                    integer;
declare variable FReceteAltAsamaId integer;
declare variable FBoyaAsamasimi smallint;

begin

  select ReceteOtomatikIstekOlustur from SISTEM where SISTEMID = 1 into :FReceteOtomatikIstekOlustur;
  BoyaKimyasalUrunTipiId = 0;

  Select LabRecete.LabReceteGrupId from LabReceteAtis
    left join LabRecete on LabRecete.LabReceteId = LabReceteAtis.LabReceteId
    where LabReceteAtis.LabReceteAtisId = :ALabReceteAtisId
    into :LabReceteGrupId;


  for Select BoyaTipiId, Min(Sira),ReceteDetay.ReceteAsamaId, ReceteAsama.BoyaAsamasimi from ReceteDetay
    left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
    where ReceteDetay.ReceteId = :AReceteId and
          ReceteDetay.ReceteAsamaId is not null and
          ReceteAsama.BoyaAsamasimi = 1
    group by BoyaTipiId,ReceteDetay.ReceteAsamaId, ReceteAsama.BoyaAsamasimi
    order by Min(Sira) desc
    into :BoyaTipiId, :MinSira, :ReceteAsamaId, :FBoyaAsamasimi  do
  begin
    if (MinSira is not null) then
    begin
      /*SİLME İŞLEMİ*/
      Select Min(Sira) from ReceteDetay
        left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
        where ReceteDetay.ReceteId = :AReceteId and
          ReceteDetay.ReceteAsamaId is not null and
          ReceteDetay.Sira > :MinSira
        Into :FMaxSira2;
       if (FMaxSira2 is null) then FMaxSira2 = 99999;

       if (FBoyaAsamasimi = 1 and FReceteOtomatikIstekOlustur = 1) then  Delete From ReceteDetay where ReceteDetay.ReceteId = :AReceteId and sira >:MinSira and Sira < :FMaxSira2;
      /*SİLME İŞLEMİ*/

      Sira = null;
      Select Min(Sira) from ReceteDetay
        left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
        where ReceteDetay.ReceteId = :AReceteId and ReceteDetay.Sira > :MinSira and ReceteDetay.ReceteAsamaId is not null
        into :Sira;

      if (Sira is null) then Sira = -1;
      FBoyaEklendi = 0;
      if (BoyaTipiId = 0) then
      begin
        For Select LabReceteUrun.UrunId, LabReceteUrun.UrunBirimId, LabReceteUrun.MiktarYuzde, Urun.UrunTipiId, LabReceteAtisRenk.Sira, LabReceteAtisRenk.Color, LabReceteAtisRenk.RenkKodu, LabReceteAtisRenk.RenkAdi from LabReceteUrun
          left join Urun on Urun.UrunId = LabReceteUrun.UrunId
          left join LabReceteAtisRenk on LabReceteAtisRenk.LabReceteAtisRenkId = LabReceteUrun.LabReceteAtisRenkId
          where LabReceteUrun.LabReceteAtisId = :ALabReceteAtisId and (Urun.UrunTipiId <> 0 or Urun.BoyaTipiId = :BoyaTipiId)
          order by LabReceteAtisRenk.Sira, LabReceteUrun.Sira
          into :LUrunId, :LUrunBirimId, :LMiktarYuzde, :LUrunTipiId, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi do
        begin
          /*1*/
          select ReceteDetay.Sira from ReceteDetay where ReceteDetay.ReceteID = :AReceteId  and ReceteDetay.ReceteAsamaId= :ReceteAsamaId into :FMinSira;
          select min(ReceteDetay.Sira) from ReceteDetay where  ReceteDetay.ReceteID = :AReceteId and sira > :FMinSira and ReceteDetay.ReceteAsamaId is not null into :FMaxSira;
          select max(ReceteDetay.ReceteAltAsamaId) from ReceteDetay  where ReceteDetay.ReceteID = :AReceteID  and ReceteDetay.Sira > :FMinSira and ReceteDetay.sira < :FMaxSira into :FReceteAltAsamaId;
          if (FReceteAltAsamaId is null) then FReceteAltAsamaId = 0;
          select min(ReceteAltAsama.ReceteAltAsamaId) from ReceteAltAsama where ReceteAltAsama.ReceteAltAsamaId > :FReceteAltAsamaId into :FWReceteAltAsamaId;
          if ((FBoyaEklendi = 0) or (LUrunTipiId <> BoyaKimyasalUrunTipiId)) then
          begin
            if (FReceteOtomatikIstekOlustur = 1) then
            begin
              insert into ReceteDetay(ReceteId, Sira, ReceteAltAsamaId, Miktar, InsertKullaniciId, KullaniciId)
              values(:AReceteId, :Sira, :FWReceteAltAsamaId, 0, :AKullaniciId, :AKullaniciId);
              if (Sira <> -1) then Sira = Sira + 1;
            end
            if (LUrunTipiId = BoyaKimyasalUrunTipiId) then FBoyaEklendi = 1;
          end
          if (LUrunTipiId <> BoyaKimyasalUrunTipiId) then FBoyaEklendi = 0;
          /*1*/
          if (LUrunTipiId = 0 and LabReceteGrupId <> 3 and LabReceteGrupId <> 0) then insert into ReceteDetay(ReceteId, Sira, UrunId, UrunBirimId, Miktar, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, InsertKullaniciId, KullaniciId) values(:AReceteId, :Sira, :LUrunId, :LUrunBirimId, :LMiktarYuzde, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi, :AKullaniciId, :AKullaniciId);
          else insert into ReceteDetay(ReceteId, Sira, UrunId, UrunBirimId, Banyo, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, InsertKullaniciId, KullaniciId) values(:AReceteId, :Sira, :LUrunId, :LUrunBirimId, :LMiktarYuzde, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi, :AKullaniciId, :AKullaniciId);
          if (Sira <> -1) then Sira = Sira + 1;
        end
      end
      else
      begin
        For Select LabReceteUrun.UrunId, LabReceteUrun.UrunBirimId, LabReceteUrun.MiktarYuzde, Urun.UrunTipiId, LabReceteAtisRenk.Sira, LabReceteAtisRenk.Color, LabReceteAtisRenk.RenkKodu, LabReceteAtisRenk.RenkAdi from LabReceteUrun
          left join Urun on Urun.UrunId = LabReceteUrun.UrunId
          left join LabReceteAtisRenk on LabReceteAtisRenk.LabReceteAtisRenkId = LabReceteUrun.LabReceteAtisRenkId
          where LabReceteUrun.LabReceteAtisId = :ALabReceteAtisId and Urun.BoyaTipiId = :BoyaTipiId
          order by LabReceteAtisRenk.Sira, LabReceteUrun.Sira
          into :LUrunId, :LUrunBirimId, :LMiktarYuzde, :LUrunTipiId, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi do
        begin
         /*2*/
          select ReceteDetay.Sira from ReceteDetay where ReceteDetay.ReceteID = :AReceteId  and ReceteDetay.ReceteAsamaId= :ReceteAsamaId into :FMinSira;
          select min(ReceteDetay.Sira) from ReceteDetay where  ReceteDetay.ReceteID = :AReceteId and sira > :FMinSira and ReceteDetay.ReceteAsamaId is not null into :FMaxSira;
          select max(ReceteDetay.ReceteAltAsamaId) from ReceteDetay  where ReceteDetay.ReceteID = :AReceteID  and ReceteDetay.Sira > :FMinSira and ReceteDetay.sira < :FMaxSira into :FReceteAltAsamaId;
          if (FReceteAltAsamaId is null) then FReceteAltAsamaId = 0;
          select min(ReceteAltAsama.ReceteAltAsamaId) from ReceteAltAsama where ReceteAltAsama.ReceteAltAsamaId > :FReceteAltAsamaId into :FWReceteAltAsamaId;
          if ((FBoyaEklendi = 0) or (LUrunTipiId <> BoyaKimyasalUrunTipiId)) then
          begin
            if (FReceteOtomatikIstekOlustur = 1) then
            begin
              insert into ReceteDetay(ReceteId, Sira, ReceteAltAsamaId, Miktar, InsertKullaniciId, KullaniciId)
              values(:AReceteId, :Sira, :FWReceteAltAsamaId, 0, :AKullaniciId, :AKullaniciId);
              if (Sira <> -1) then Sira = Sira + 1;
            end
            if (LUrunTipiId = BoyaKimyasalUrunTipiId) then FBoyaEklendi = 1;
          end
          if (LUrunTipiId <> BoyaKimyasalUrunTipiId) then FBoyaEklendi = 0;
          /*2*/
          if (LUrunTipiId <> 1) then insert into ReceteDetay(ReceteId, Sira, UrunId, UrunBirimId, Miktar, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, InsertKullaniciId, KullaniciId) values(:AReceteId, :Sira, :LUrunId, :LUrunBirimId, :LMiktarYuzde, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi, :AKullaniciId, :AKullaniciId);
          else insert into ReceteDetay(ReceteId, Sira, UrunId, UrunBirimId, Banyo, AtisRenkSira, AtisRenkColor, AtisRenkKodu, AtisRenkAdi, InsertKullaniciId, KullaniciId) values(:AReceteId, :Sira, :LUrunId, :LUrunBirimId, :LMiktarYuzde, :FAtisRenkSira, :FAtisRenkColor, :FAtisRenkKodu, :FAtisRenkAdi, :AKullaniciId, :AKullaniciId);
          if (Sira <> -1) then Sira = Sira + 1;
        end
      end
    end
  end

end
