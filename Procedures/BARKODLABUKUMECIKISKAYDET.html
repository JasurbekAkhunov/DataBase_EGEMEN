CREATE PROCEDURE BARKODLABUKUMECIKISKAYDET(
    AIPLIKSTOKID INTEGER,
    ANUMARATOR INTEGER,
    AIPLIKIRSALIYEID INTEGER,
    ABUKUMISEMRIKAYITID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    STATUS INTEGER)
AS
declare variable GirisIplikIrsaliyeDetayId integer;
declare variable CikisIplikIrsaliyeDetayId integer;
declare variable HamId integer;
declare variable IplikDepoId    varchar(10);
declare variable IplikIrsaliyeDetayId integer;
declare variable Lot                  varchar(30);
declare variable RenkId               integer;
declare variable Miktar               numeric(12,3);
declare variable NetMiktar            numeric(12,3);
declare variable KS integer;
declare variable HareketTipi integer;
declare variable MaxNumarator integer;
declare variable hstIplik integer;
declare variable HamTipiId integer;

declare variable MelanjKodu           varchar(20);
declare variable FlamKodu             varchar(20);
declare variable KonikRengiId          integer;
declare variable KonikCinsiId          integer;
declare variable SarimSekliId          integer;
declare variable BobinGramaji          Numeric(12,3);
declare variable AmbalajId             integer;
declare variable HarmanKayitId         integer;
declare variable BukumIsEmriLot        varchar(30);
begin
  select shtBukumeCikis from IplikStokHareketTipiGetir into :HareketTipi;
  Select hstIplik from HamTipiGetir into :hstIplik;

  if (AIplikIrsaliyeId is not null) then
    Select Max(Numarator) from IplikStok where IplikStok.IplikStokID = :AIplikStokId into :MaxNumarator;
  else
    MaxNumarator = ANumarator;

  if (MaxNumarator is null) then
  begin
    Status = -1;
    suspend;
    exit;
  end

  Select IplikStok.CikisIplikIrsaliyeDetayId, IplikStok.IplikIrsaliyeDetayId, IplikStok.HamId, IplikStok.IplikDepoId,
         IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.RenkId, IplikStok.Miktar, IplikStok.NetMiktar, Ham.HamTipiId,
         IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu, IplikIrsaliyeDetay.KonikRengiID, IplikIrsaliyeDetay.KonikCinsiId,
         IplikIrsaliyeDetay.SarimSekliId, IplikIrsaliyeDetay.BobinGramaji, IplikIrsaliyeDetay.AmbalajId, IplikIrsaliyeDetay.HarmanKayitId from IplikStok
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.IplikIrsaliyeDetayId
    left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
    where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator
    into :CikisIplikIrsaliyeDetayId, :GirisIplikIrsaliyeDetayId, :HamId, :IplikDepoId, :Lot, :RenkId, :Miktar, :NetMiktar, :HamTipiId,
         :MelanjKodu, :FlamKodu, :KonikRengiID, :KonikCinsiId, :SarimSekliId, :BobinGramaji, :AmbalajId, :HarmanKayitId;

  if (GirisIplikIrsaliyeDetayId is null) then
  begin
    Status = -1;
    suspend;
    exit;
  end

  if (AIplikIrsaliyeId is not null) then
  if (CikisIplikIrsaliyeDetayId is not null) then
  begin
    Status = -2;
    suspend;
    exit;
  end

  if (AIplikIrsaliyeId is not null) then
  if (HamTipiId <> hstIplik) then
  begin
    Status = -3;
    suspend;
    exit;
  end

  Select coalesce(IplikLot.Lot, BukumIsEmri.Lot) from BukumIsEmri
    left join IplikLot on IplikLot.IplikLotID = BukumIsEmri.IplikLotId
    where BukumIsEmri.BukumIsEmriKayitId = :ABukumIsEmriKayitId
    into :BukumIsEmriLot;

  if (BukumIsEmriLot <> Lot) then
  begin
    Status = -4;
    suspend;
    exit;
  end

  if (AIplikIrsaliyeId is not null) then
  begin
    Status = 1;

    Select Min(IplikIrsaliyeDetay.IplikIrsaliyeDetayId) from IplikIrsaliyeDetay
      left join IplikStok on IplikStok.CikisIplikIrsaliyeDetayId = IplikIrsaliyeDetay.IplikIrsaliyeDetayId
      where IplikIrsaliyeDetay.IplikIrsaliyeId = :AIplikIrsaliyeId and IplikIrsaliyeDetay.BukumIsEmriKayitId = :ABukumIsEmriKayitId and
            IplikIrsaliyeDetay.HamID = :HamId and IplikStok.IplikIrsaliyeDetayID = :GirisIplikIrsaliyeDetayID and
            IplikIrsaliyeDetay.IplikDepoId = :IplikDepoId and
            StrKarsilastir(IplikIrsaliyeDetay.Lot, :Lot) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.MelanjKodu, :MelanjKodu) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.FlamKodu, :FlamKodu) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.AmbalajId, :AmbalajID) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.HarmanKayitId, :HarmanKayitId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.RenkId, :RenkId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.KonikRengiId, :KonikRengiId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.KonikCinsiId, :KonikCinsiId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.SarimSekliId, :SarimSekliId) = 1 and
            IplikIrsaliyeDetay.BobinGramaji = :BobinGramaji
      into :IplikIrsaliyeDetayId;

    if (IplikIrsaliyeDetayId is null) then
    begin
      IplikIrsaliyeDetayId = gen_Id(IplikIrsaliyeDetay_Id_gen, 1);
      insert into IplikIrsaliyeDetay(IplikIrsaliyeId, IplikIrsaliyeDetayId, HareketTipi, IplikDepoId, BukumIsEmriKayitId, HamId, RenkId,
                                     HarmanKayitId, Lot, MelanjKodu, FlamKodu, SarimSekliId, KonikCinsiId, KonikRengiId, AmbalajId, BobinGramaji,
                                     KapAdedi, Miktar, NetMiktar, InsertKullaniciId, KullaniciId)
                  values(:AIplikIrsaliyeId, :IplikIrsaliyeDetayId, :HareketTipi, :IplikDepoId, :ABukumIsEmriKayitId, :HamId, :RenkId, :HarmanKayitId,
                                     :Lot, :MelanjKodu, :FlamKodu, :SarimSekliId, :KonikCinsiId, :KonikRengiId, :AmbalajId, :BobinGramaji,
                                     1, :Miktar, :NetMiktar, :AKullaniciID, :AKullaniciID);

    end
    else
    begin
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi + 1, Miktar = Miktar + :Miktar, NetMiktar = NetMiktar + :NetMiktar, NeredenUpdate = 1, KullaniciID = :AKullaniciID where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :IplikIrsaliyeDetayId;
    end

    Update IplikStok set CikisIplikIrsaliyeDetayId = :IplikIrsaliyeDetayId where IplikStok.IplikStokID = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;
    suspend;
  end
  else
  begin
    if (CikisIplikIrsaliyeDetayId is Null) then
    begin
      Status = -1;
      Suspend;
      exit;
    end

    Update IplikStok set CikisIplikIrsaliyeDetayId = Null where IplikStok.IplikStokID = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;

    Select count(*) from IplikStok where IplikStok.CikisIplikIrsaliyeDetayId = :CikisIplikIrsaliyeDetayId into :KS;

    if (KS = 0) then delete from IplikIrsaliyeDetay where IplikIrsaliyeDetay.IplikIrsaliyeDetayID = :CikisIplikIrsaliyeDetayId;
    else
    begin
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi - 1, Miktar = Miktar - :Miktar, NetMiktar = NetMiktar - :NetMiktar, KullaniciID = :AKullaniciId, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :CikisIplikIrsaliyeDetayId;
    end

    Status = 1;
    suspend;
  end
end
