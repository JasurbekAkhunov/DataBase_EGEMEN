CREATE PROCEDURE TRANSFERICINSTOKFATURALISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AEVRAKTIPI INTEGER,
    ATRANSFERDURUMU INTEGER)
RETURNS (
    FATURAID INTEGER,
    HAREKETTIPI INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    TRANSFEREDILDI INTEGER,
    TRANSFEREDILDISTR VARCHAR(7),
    TRANSFERTARIHI TIMESTAMP,
    TRANSFERKULLANICIID INTEGER,
    TRANSFERKULLANICIADI VARCHAR(40))
AS
begin
  if (AEvrakTipi = 0) then AEvrakTipi = 1;
  else if (AEvrakTipi = 1) then AEvrakTipi = 3;
  else if (AEvrakTipi = 2) then AEvrakTipi = 2;
  else if (AEvrakTipi = 3) then AEvrakTipi = 4;
  else AEvrakTipi = -1;

  HareketTipi = :AEvrakTipi;

  for Select Fatura.FaturaId, Fatura.FirmaId, Firma.Adi, Fatura.SeriNo, Fatura.FaturaNo, Fatura.FaturaTarihi,
             Fatura.TransferEdildi, Fatura.TransferTarihi, Fatura.TransferKullaniciId,
             TK.Adi from Fatura
    left join Firma On Firma.FirmaId = Fatura.FirmaId
    left join Kullanici TK On TK.KullaniciId = Fatura.TransferKullaniciId
    where Fatura.FaturaTarihi >= :ABaslangicTarihi and Fatura.FaturaTarihi <= :ABitisTarihi and Fatura.HareketTipi = :AEvrakTipi and
          Fatura.Entegrasyon = 1 and (Fatura.TransferEdildi = :ATransferDurumu or :ATransferDurumu is null)
  into :FaturaId, :FirmaId, :FirmaAdi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :TransferEdildi, :TransferTarihi, :TransferKullaniciId,
       :TransferKullaniciAdi do
  begin
    if (TransferEdildi = 1) then TransferEdildiStr = 'Yapıldı';
    else TransferEdildiStr = '';
    suspend;
  end
end
