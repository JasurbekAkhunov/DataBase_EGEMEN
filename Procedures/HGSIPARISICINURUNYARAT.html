CREATE PROCEDURE HGSIPARISICINURUNYARAT(
    AANAURUNID URUNKODUSIZE,
    ARENKID INTEGER,
    ABEDENID INTEGER,
    AURUNTIPIID SMALLINT,
    AKULLANICIID INTEGER)
RETURNS (
    URUNID URUNKODUSIZE)
AS
Declare Variable BedenKodu Varchar(7);
Declare Variable RenkAdi Varchar(50);
Declare Variable MaxNumara integer;
Declare Variable FUrunID Varchar(20);
Declare Variable FUrunAdi Varchar(80);
Declare Variable DefaultUniteID  Integer;
Declare Variable ModelKumasID    Integer;
Declare Variable ModelBedenID    Integer;
Declare Variable AnaKumasModelKumasID Integer;
Declare Variable KDVOrani Double Precision;
Declare Variable ModelParcaID    Integer;
Declare Variable KayitliUrunID Varchar(20);
Declare Variable FModelKayitId Integer;
Declare Variable FUrunGrup1                UrunGrupKoduSize;
Declare Variable FUrunGrup2                UrunGrupKoduSize;
Declare Variable FUrunGrup3                UrunGrupKoduSize;
Declare Variable FUrunGrup4                UrunGrupKoduSize;
Declare Variable FMutfakDepoNo             DepoKoduSize;

begin
  Select DefaultUniteID from Kullanici where Kullanici.KullaniciID = :AKullaniciID into :DefaultUniteID;
  Select KdvOrani from Sistem where Sistem.SistemID = 1 into :KDVOrani;

  if (DefaultUniteID is Null) then Exception KayitHatasi 'Kullanıcının Varsayılarn Unite Kodu Tanımlanmamış..';


  Select Urun.Adi, Urun.ModelKayitId, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, MutfakDepoNo from Urun where Urun.UrunID = :AAnaUrunID
    into :FUrunAdi, :FModelKayitId, :FUrunGrup1, :FUrunGrup2, :FUrunGrup3, :FUrunGrup4, :FMutfakDepoNo;
  Select Renk.Adi from Renk where Renk.RenkId = :ARenkId into :RenkAdi;
  Select Beden.KisaKodu from Beden where Beden.BedenId = :ABedenID into :BedenKodu;

  if (BedenKodu is Null) then BedenKodu = '';
  if (RenkAdi is Null) then RenkAdi = '';

  Select Max(EgemenStrToInt(SubString(UrunID From Char_Length(:AAnaUrunID) + 2 for 3)))  from Urun where Urun.AnaUrunID = :AAnaUrunID into :MaxNumara;

  if (MaxNumara is Null) then MaxNumara = 1;
  else MaxNumara = MaxNumara + 1;

  FUrunID = AAnaUrunID || '.' || EgemenInttoLzStr(MaxNumara, 2);
  FUrunAdi = FUrunAdi || ' / ' || RenkAdi ||' ' || BedenKodu;

  insert into Urun(UrunId, Adi, UrunTipiId, UniteID, KDVOrani, ModelKayitId, RenkID, BedenID, AnaUrunID, InsertKullaniciId, KullaniciId, UrunGrup1, UrunGrup2, UrunGrup3, UrunGrup4, MutfakDepoNo)
           Values (:FUrunID, :FUrunAdi, :AUrunTipiID, :DefaultUniteID, :KdvOrani, :FModelKayitId, :ARenkId, :ABedenID, :AAnaUrunID, :AKullaniciID, :AKullaniciID, :FUrunGrup1, :FUrunGrup2, :FUrunGrup3, :FUrunGrup4, :FMutfakDepoNo);

  insert into UrunBirim (UrunID, BirimID, DefaultBirim, InsertKullaniciId, KullaniciId) Values (:FUrunID, 'Ad', 1, :AKullaniciID, :AKullaniciID);

  UrunID = FUrunID;

  Suspend;
end
