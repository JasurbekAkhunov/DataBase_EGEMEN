CREATE PROCEDURE ORMEURETIMTALIMATLISTESI(
    AORMEMAKINAID VARCHAR(20),
    AFIRMAID VARCHAR(20))
RETURNS (
    RESULTID VARCHAR(255),
    ORMEPLANLAMAID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    URETIMPARTINO VARCHAR(30),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    MELANJKODU VARCHAR(20),
    SIPARISMIKTARI NUMERIC(12,3),
    SIPARISTOPLAMORULEN NUMERIC(12,3),
    SIPARISTOPLAMSAKATORULEN NUMERIC(12,3),
    SIPARISORULECEKMIKTAR NUMERIC(12,3),
    PLANLANANMIKTAR NUMERIC(12,3),
    GERCEKLESENMIKTAR NUMERIC(12,3),
    TOPLAMSAKATORULEN NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    MUSTERISIPARISNO VARCHAR(50),
    PUS INTEGER,
    FEINE INTEGER,
    LOT VARCHAR(50),
    TUPACIKENMAYLISTR VARCHAR(30),
    ORMESIPARISSEKLIID INTEGER,
    ORMESIPARISSEKLIADI VARCHAR(15),
    ORMESIPARISTIPIID INTEGER,
    ORMESIPARISTIPIADI VARCHAR(30),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    MAKINAUSTUGRAMAJ INTEGER)
AS
declare variable Bitti integer;
declare variable TupAcikEnMayli integer;
declare variable FHamStokBirimID        integer;
begin

  for Select OrmePlanlama.OrmePlanlamaId, OrmePlanlama.BoyaSiparisKayitId, OrmePlanlama.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             OrmePlanlama.UretimPartiNo, OrmeSiparisDetay.HamId, Ham.Adi, OrmePlanlama.Bitti,
             OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.ToplamOrulen, OrmeSiparisDetay.ToplamSakatOrulen,
             (OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.ToplamOrulen - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar + OrmeSiparisDetay.ToplamSakatOrulen),
             OrmePlanlama.PlanlananMiktar, OrmePlanlama.GerceklesenMiktar, OrmePlanlama.ToplamSakatOrulen,
             OrmeSiparisDetay.MusteriSiparisNo,
             OrmeSiparisDetay.RenkId, Renk.Adi, OrmeSiparisDetay.MelanjKodu, OrmePlanlama.Lot, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
             OrmePlanlama.TupAcikEnMayli, BoyaSiparis.OrmeSiparisSekliId, OrmeSiparisSekli.Adi, BoyaSiparis.OrmeSiparisTipiId, OrmeSiparisTipi.Adi, OrmeSiparisDetay.HamStokBirimID,
             OrmePlanlama.Notu, OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4, OrmeSiparisDetay.MakinaUstuGramaj from OrmePlanlama
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join OrmeSiparisSekli On OrmeSiparisSekli.OrmeSiparisSekliId = BoyaSiparis.OrmeSiparisSekliId
    left join OrmeSiparisTipi On OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
  where (OrmePlanlama.FirmaId = :AFirmaId or :AFirmaId is null) and
        (OrmePlanlama.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
        OrmeSiparisDetay.Kapali = 0
  into :OrmePlanlamaId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :BoyaSiparisYil, :BoyaSiparisId,
       :UretimPartiNo, :HamId, :HamAdi, :Bitti, :SiparisMiktari, :SiparisToplamOrulen, :SiparisToplamSakatOrulen, :SiparisOrulecekMiktar,
       :PlanlananMiktar, :GerceklesenMiktar, :ToplamSakatOrulen, :MusteriSiparisNo, :RenkId, :RenkAdi, :MelanjKodu, :Lot, :Pus, :Feine, :TupAcikEnMayli,
       :OrmeSiparisSekliId, :OrmeSiparisSekliAdi, :OrmeSiparisTipiId, :OrmeSiparisTipiAdi, :FHamStokBirimID, :Notu,
       :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :MakinaUstuGramaj do
  begin
    if (Bitti = 0) then
    begin
      ResultId = '"' || EgemenIntegerToStr(OrmePlanlamaId) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisKayitId) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(OrmeSiparisDetayId) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisYil) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisId) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(OrmeSiparisTipiId) || '",' ;
      ResultId = ResultId || '"' || UretimPartiNo || '",' ;
      if (RenkId is null) then
        ResultId = ResultId || '"",' ;
      else
        ResultId = ResultId || '"' || RenkId || '",' ;

      ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",' ;

      if (MelanjKodu is not null) then
        ResultId = ResultId || '"' || MelanjKodu || '",' ;
      else
        ResultId = ResultId || '"",' ;

      if (Lot is not null) then
        ResultId = ResultId || '"' || Lot || '",' ;
      else
        ResultId = ResultId || '"",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(FHamStokBirimID) || '"' ;

      KalanMiktar = PlanlananMiktar - GerceklesenMiktar + ToplamSakatOrulen;
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

      suspend;
    end
  end
end
