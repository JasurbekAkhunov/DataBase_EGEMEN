CREATE PROCEDURE MC_KALITEKONTROLRAPORUTOP(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKULLANICIID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    PARTIVESINIFID VARCHAR(40),
    PSKALITEKONTROLID INTEGER,
    TOPNO INTEGER,
    TARIH TIMESTAMP,
    KILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION,
    EN VARCHAR(10),
    GRAMAJ VARCHAR(10),
    INSERTKULLANICIADI VARCHAR(40),
    MAKINAID VARCHAR(10),
    KALITENO INTEGER,
    HATAPUANI INTEGER,
    BASLANGICEN VARCHAR(10),
    ORTAEN VARCHAR(10),
    BITISEN VARCHAR(10),
    ORTALAMAEN INTEGER,
    TOPLAMHATAADEDI INTEGER)
AS
begin
  For Select MC_KaliteKontrolRaporuParti.PartiSinifId, MC_KaliteKontrolRaporuParti.PartiveSinifID,
             PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.TopNo, PSKaliteKontrol.Tarih, PSKaliteKontrol.Kilo,
             PSKaliteKontrol.Metre, PSKaliteKontrol.En, PSKaliteKontrol.Gramaj, Kullanici.Adi, PSKaliteKontrol.MakinaId,
             PSKaliteKontrol.KaliteNo, PSKaliteKontrol.HataPuani, PSKaliteKontrol.BaslangicEn, PSKaliteKontrol.OrtaEn,
             PSKaliteKontrol.BitisEn, PSKaliteKontrol.OrtalamaEn from MC_KaliteKontrolRaporuParti(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :AKullaniciID)
    left join PSKaliteKontrol on PSKaliteKontrol.PartiSinifID = MC_KaliteKontrolRaporuParti.PartiSinifId
    left join Kullanici on Kullanici.KullaniciID = PSKaliteKontrol.InsertKullaniciId
    where PSKaliteKontrol.PSKaliteKontrolID is not null
    into :PartiSinifId, :PartiveSinifID, :PSKaliteKontrolId, :TopNo, :Tarih, :Kilo, :Metre, :En, :Gramaj, :InsertKullaniciAdi, :MakinaId,
         :KaliteNo, :HataPuani, :BaslangicEn, :OrtaEn, :BitisEn, :OrtalamaEn do
  begin
    ToplamHataAdedi = Null;

    Select Count(*) from PSKaliteKontrolHataMetre where PSKaliteKontrolHataMetre.PSKaliteKontrolID = :PSKaliteKontrolID into :ToplamHataAdedi;
    suspend;
  end
end
