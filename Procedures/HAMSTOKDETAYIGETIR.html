CREATE PROCEDURE HAMSTOKDETAYIGETIR(
    AHAMSTOKIRSALIYEID INTEGER)
RETURNS (
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    HAMTIPIID INTEGER,
    GIRISPARTINOSTR VARCHAR(40),
    HAMADI VARCHAR(255),
    PUS INTEGER,
    FEINE INTEGER,
    ORMECIMAKINANO VARCHAR(10),
    EN INTEGER,
    GRM2 INTEGER,
    RENK VARCHAR(20),
    TUPACIKENMAYLISTR VARCHAR(10),
    LOT VARCHAR(150),
    DENYE INTEGER,
    LYCRAURETICIADI VARCHAR(20),
    IPLIKUZUNLUGU1 VARCHAR(40),
    IPLIKUZUNLUGU2 VARCHAR(40),
    IPLIKUZUNLUGU3 VARCHAR(40),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    MIKTAR NUMERIC(12,3),
    RAFLANANKILO NUMERIC(12,3),
    RAFLANANMETRE NUMERIC(12,3),
    RAFLANANMIKTAR NUMERIC(12,3),
    RAFLANMAYANMIKTAR NUMERIC(12,3),
    ACIKLAMA BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TAMIR INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    ESKIPARTINO INTEGER,
    MUSTERIPARTINO VARCHAR(30),
    NEMORANI DOUBLE PRECISION,
    YIKAMAONCESIEN DOUBLE PRECISION,
    YIKAMASONRASIEN DOUBLE PRECISION,
    YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    YIKAMASONRASIGRAMAJ DOUBLE PRECISION,
    ENDENCEKMEZLIK DOUBLE PRECISION,
    BOYDANCEKMEZLIK DOUBLE PRECISION,
    HAMDONME DOUBLE PRECISION,
    YIKAMAONCESIRAPORBOYU VARCHAR(100),
    YIKAMASONRASIRAPORBOYU VARCHAR(100),
    BOYAMAKABILIYETI VARCHAR(10),
    YIKAMAHASLIGI VARCHAR(10),
    NUMARATIPIADI VARCHAR(10),
    NUMARA VARCHAR(10),
    BUKUM VARCHAR(10),
    MUKAVEMET VARCHAR(10),
    UZAMA VARCHAR(10),
    UVKONTROL INTEGER,
    KONTROLKULLANICIADI VARCHAR(40),
    KONTROLTARIHI TIMESTAMP,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(4),
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    HAMSTOKKABULTIPIID INTEGER,
    HAMSTOKKABULTIPIADI VARCHAR(20),
    BOYASIPARISFIRMAID VARCHAR(20),
    BOYASIPARISFIRMAADI VARCHAR(100),
    URETICIID INTEGER,
    URETICIADI ADI40)
AS
declare variable TupAcikEnMayli  integer;
declare variable GirisPartiYil  integer;
declare variable GirisPartiId   integer;
begin
  for select HSD.HamStokMasterId, HSD.HamStokDetayId, Ham.HamTipiId, Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.OrmeciMakinaNo, HamStokMaster.En, HamStokMaster.Grm2,  HamStokMaster.Renk, HamStokMaster.Lot, HamStokMaster.TupAcikEnMayli,
             HamStokMaster.Denye, LycraUretici.Adi, HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3,
             HSD.KapAdedi, HSD.Miktar, (HSD.Miktar - HSD.RaflananMiktar), HSD.RaflananKilo, HSD.RaflananMetre, HSD.RaflananMiktar,  HSD.Kilo, HSD.Metre, HSD.Adet, HSD.Aciklama,  HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartino, HamStokMaster.MusteriPartiNo,
             HSd.GirisPartiYil, HSD.GirisPartiId, HamStokMaster.OnayDurumu, HamStokMaster.OnaylayanKullaniciId, OK.Adi, HamStokMaster.OnayTarihi,
             HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiGramaj, HamStokMaster.EndenCekmezlik,
             HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
             HamStokMaster.BoyamaKabiliyeti, HamStokMaster.YikamaHasligi, NumaraTipi.Adi, HamStokMaster.Numara, HamStokMaster.Bukum, HamStokMaster.Mukavemet, HamStokMaster.NemOrani, HamStokMaster.Uzama, HamStokMaster.UVKontrol,
             KK.Adi, HamStokMaster.Kontroltarihi, IK.Adi, HSD.InsertTarihi, Sk.Adi, HSD.DegisimTarihi,
             HSD.HamStokKabulTipiId, HamStokKabulTipi.Adi,
             BoyaSiparis.FirmaId, SiparisFirma.Adi, Uretici.UreticiId, Uretici.Adi from HamStokDetay HSD
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HSD.HamStokMasterId
  left join Ham on Ham.HamId = HSD.HamId
  left join NumaraTipi on NumaraTipi.NumaraTipiId = HamStokMaster.NumaraTipiId
  left join HamStokKabulTipi on HamStokKabulTipi.HamStokKabulTipiId = HSD.HamStokKabulTipiId
  left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
  left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  left join LycraUretici on LycraUretici.LycraUreticiId = HSD.LycraUreticiId
  Left Join Kullanici OK on OK.KullaniciId = HamStokMaster.OnaylayanKullaniciId
  Left Join Kullanici KK on KK.KullaniciId = HamStokMaster.KontrolKullaniciId
  Left Join Kullanici IK on IK.KullaniciId = HSD.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HSD.KullaniciId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = HamStokMaster.BoyaSiparisKayitId
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join Uretici on Uretici.UreticiId = HamStokMaster.UreticiId
  where HSD.HamStokIrsaliyeId = :AHamStokIrsaliyeId order by HSD.HamStokDetayId
  into :HamStokMasterId, :HamStokDetayId, :HamTipiId, :HamAdi, :Pus, :Feine, :OrmeciMakinaNo, :En, :Grm2, :Renk, :Lot, :TupAcikEnMayli, :Denye, :LycraUreticiAdi,
       :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3,
       :KapAdedi, :Miktar, :RaflanmayanMiktar, :RaflananKilo, :RaflananMetre, :RaflananMiktar, :Kilo, :Metre, :Adet, :Aciklama, :Tamir, :TamirTipiAdi, :EskiPartino, :MusteriPartiNo,
       :GirisPartiYil, :GirisPartiId,
       :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi,
       :YikamaOncesiEn, :YikamaSonrasiEn, :YikamaOncesiGramaj, :YikamaSonrasiGramaj, :EndenCekmezlik,
       :BoydanCekmezlik, :HamDonme, :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu,
       :BoyamaKabiliyeti, :YikamaHasligi, :NumaraTipiAdi, :Numara, :Bukum, :Mukavemet, :NemOrani, :Uzama, :UVKontrol,
       :KontrolKullaniciAdi, :Kontroltarihi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :HamStokKabulTipiId, :HamStokKabulTipiAdi, :BoyaSiparisFirmaId, :BoyaSiparisFirmaAdi, :UreticiId, :UreticiAdi  do
  begin
    Select TupAcikEnMayliStrGetir.tupacikenmaylistr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    if (OnayDurumu = 1) then OnayDurumuStr = 'Okey';
    else OnayDurumuStr = '';

    GirisPartiNoStr = '(' || EgemenIntegerToStr(GirisPartiYil) || ') ' || EgemenIntegerToStr(GirisPartiId) || ' ';

    suspend;
  end
end
