CREATE PROCEDURE MC_KALITEKONTROLRAPORUPARTI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKULLANICIID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    PARTIVESINIFID VARCHAR(40),
    KKBITISTARIHI TIMESTAMP,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAKISAADI VARCHAR(10),
    FIRMAKODUMUZ VARCHAR(20),
    FIRMAADIMIZ VARCHAR(50),
    FIRMAKISAADIMIZ VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(50),
    TOPLAMTOPADEDI INTEGER,
    TOPLAMKILO DOUBLE PRECISION,
    TOPLAMMETRE DOUBLE PRECISION,
    ORTALAMAEN DOUBLE PRECISION,
    ORTALAMAGRAMAJ DOUBLE PRECISION,
    TOPLAMHATAADEDI INTEGER)
AS
declare variable Okey     integer;
declare variable ToplamEn Double Precision;
declare variable ToplamGramaj Double Precision;
declare variable FTopAdedi integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  Select FirmaKodumuz from Sistem into :FirmaKodumuz;

  Select Firma.Adi, Firma.KisaAdi from Firma where Firma.FirmaID = :FirmaKodumuz into :FirmaAdimiz, :FirmaKisaAdimiz;


  For Select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifID, PartiStok.KKBitisTarihi, BoyaSiparis.SiparisGrupId,
             SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.ReferansSiparisNo, Parti.FirmaId, Firma.Adi, Firma.KisaAdi,
             Parti.ReceteID, Recete.Adi from PartiStok
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaID = Parti.FirmaId
    left join Recete on Recete.ReceteID = Parti.ReceteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitId
    left join SiparisGrup on SiparisGrup.SiparisGrupID = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    where PartiStok.KKBitisTarihi >= :ABaslangicTarihi and PartiStok.KKBitisTarihi < :ABitisTarihi + 1 and
          Parti.FirmaID = :AFirmaId and PartiStok.PartiStokIptalKayitID is null
    into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifID, :KKBitisTarihi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :MusteriSiparisNo, :ReferansSiparisNo,
         :FirmaId, :FirmaAdi, :FirmaKisaAdi, :ReceteID, :ReceteAdi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId);
    PartiveSinifID = PartiIdStr || '-' || EgemenIntegerToStr(PartiSinifId);

    ToplamTopAdedi = 0;
    ToplamKilo     = Null;
    ToplamMetre    = Null;
    OrtalamaEn     = 0;
    OrtalamaGramaj = 0;
    ToplamHataAdedi = 0;

    Select Count(*), Sum(Kilo), sum(Metre) from PSKaliteKontrol
      where PSKaliteKontrol.PartiSinifID = :PartiSinifID
      into :ToplamTopAdedi, :ToplamKilo , :ToplamMetre;



    if (ToplamTopAdedi > 0) then
    begin
      FTopAdedi = 0;
      ToplamEn = 0;
      ToplamGramaj = 0;

      Select Count(*), Sum( EgemenStrToInt(En) ) from PSKaliteKontrol
        where PSKaliteKontrol.PartiSinifID = :PartiSinifID and EgemenStrToIntOlurmu(PSKaliteKontrol.En) = 1 and EgemenStrToInt(En) > 0
        into :FTopAdedi, :ToplamEn;

      if (FTopAdedi <> 0) then OrtalamaEn = ToplamEn / FTopAdedi;

      FTopAdedi = 0;

      Select Count(*), Sum( EgemenStrToInt(Gramaj) ) from PSKaliteKontrol
        where PSKaliteKontrol.PartiSinifID = :PartiSinifID and EgemenStrToIntOlurmu(PSKaliteKontrol.Gramaj) = 1 and EgemenStrToInt(Gramaj) > 0
        into :FTopAdedi, :ToplamGramaj;


      if (FTopAdedi <> 0) then OrtalamaGramaj = ToplamGramaj / FTopAdedi;

      ToplamHataAdedi = Null;

      Select Count(*) from PSKaliteKontrol
        left join PSKaliteKontrolHataMetre on PSKaliteKontrolHataMetre.PSKaliteKontrolID = PSKaliteKontrol.PSKaliteKontrolId
        where PSKaliteKontrol.PartiSinifID = :PartiSinifID and PSKaliteKontrolHataMetre.PSKaliteKontrolHataMetreID is not null
        into :ToplamHataAdedi;

      suspend;
    end
  end
end
