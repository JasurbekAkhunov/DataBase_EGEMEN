CREATE PROCEDURE IPLIKIHTIYACRAPORURAF(
    AIPLIKLERIDOK INTEGER,
    ALYCRALARIDOK INTEGER,
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    ATARIHEGORESIPRECETETOPLAMI INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    ARAPORTIPI INTEGER)
RETURNS (
    HAMID INTEGER,
    STOKGIRISNO INTEGER,
    RENKID INTEGER,
    RENKADI ADI50,
    LOT TYPE OF COLUMN ORMERAF.LOT,
    SERTIFIKA TYPE OF COLUMN ORMERAF.SERTIFIKA,
    MIKTAR DOUBLE PRECISION,
    KAPADEDI INTEGER,
    MELANJKODU TYPE OF COLUMN ORMERAF.MELANJKODU,
    URETICIADI TYPE OF COLUMN URETICI.ADI,
    FASONSTOKSTR VARCHAR(10),
    STOKFIRMAID TYPE OF COLUMN ORMERAF.STOKFIRMAID,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISTIPIADI VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISFIRMAID FIRMAKODUSIZE,
    SIPARISFIRMAADI ADI100,
    SIPARISSIRANO INTEGER)
AS
declare variable hstIplik       integer;
declare variable hstLycra       integer;
declare variable AstIplik               integer;
declare variable astBoyaliIplik         integer;
declare variable astLycra               integer;
declare variable astHamKumas            integer;
declare variable astBoyaliKumas         integer;
declare variable astIplikBoyaliKumas    integer;
declare variable KullanimOrani  Double Precision;
declare variable FireOrani      Double Precision;
declare variable assSatis          integer;
declare variable assFasonArtiSatis integer;
declare variable assFason          integer;
begin
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  Select astIplik, astBoyaliIplik, AstLycra, AstHamKumas, AstBoyaliKumas, AstIplikBoyaliKumas from OrmeSiparisTipiGetir
    into :astIplik, :astBoyaliIplik, :AstLycra, :AstHamKumas, :AstBoyaliKumas, :AstIplikBoyaliKumas;

  Select assSatis, assFasonArtiSatis, assFason from OrmeSiparisSekliGetir into :assSatis, :assFasonArtiSatis, :assFason;

  for Select OrmeRaf.StokGirisNo, OrmeRaf.HamId,  OrmeRaf.RenkId, Renk.Adi, sum(OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari),
       sum(OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi), OrmeRaf.Lot, OrmeRaf.MelanjKodu, Uretici.Adi, iif(OrmeRaf.FasonStok = 1, 'Fason',''),
       OrmeRaf.StokFirmaId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID,
       OrmeRaf.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.FirmaId,
       SiparisFirma.Adi, OrmeSiparisTipi.Adi, OrmeRaf.Sertifika from OrmeRaf
    left join Ham on Ham.HamId = OrmeRaf.HamId
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
    left join OrmeSiparisTipi on OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
    left join Firma SiparisFirma on SiparisFirma.Firmaid = BoyaSiparis.FirmaId
    where OrmeRaf.StokVar = 1 and(HamTipiID = :hstIplik or HamTipiID = :hstLycra) and
    (:AIplikleriDok = 1 or HamTipiID <> :hstIplik) and
    (:ALycralariDok = 1 or HamTipiID <> :hstLycra)
    Group by
    OrmeRaf.StokGirisNo, OrmeRaf.HamId,  OrmeRaf.RenkId, Renk.Adi,
       OrmeRaf.Lot, OrmeRaf.MelanjKodu, Uretici.Adi, iif(OrmeRaf.FasonStok = 1, 'Fason',''),
       OrmeRaf.StokFirmaId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID,
       OrmeRaf.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.FirmaId,
       SiparisFirma.Adi, OrmeSiparisTipi.Adi, OrmeRaf.Sertifika
  into  :StokGirisNo, :HamId,  :RenkId, :RenkAdi, :Miktar,
       :KapAdedi, :Lot, :MelanjKodu, :UreticiAdi, :FasonStokStr,
       :StokFirmaId, :BoyaSiparisYil, :BoyaSiparisID, :OrmeSiparisDetayId, :SiparisSiraNo, :SiparisFirmaId,
       :SiparisFirmaAdi, :SiparisTipiAdi, :Sertifika do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';


    suspend;
  end
end
