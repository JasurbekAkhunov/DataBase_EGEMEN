CREATE PROCEDURE DESENSABLONGETIR(
    ADESENID VARCHAR(20))
RETURNS (
    DESENSABLONID INTEGER,
    SABLONSIRANO INTEGER,
    ILAVESABLONNO INTEGER,
    SABLONADI VARCHAR(1),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    COLOR INTEGER,
    MESHID INTEGER,
    RASTERVARMI INTEGER,
    RASTERVARMISTR VARCHAR(4),
    RASTERDEGERLERI1 INTEGER,
    RASTERDEGERLERI2 INTEGER,
    RASTERDEGERLERI3 INTEGER,
    RASTERDEGERLERI4 INTEGER,
    RASTERDEGERLERI5 INTEGER,
    RASTERDEGERLERI6 INTEGER,
    RASTERDEGERLERI7 INTEGER,
    RASTERDEGERLERI8 INTEGER,
    RASTERDEGERLERI9 INTEGER,
    RASTERDEGERLERI10 INTEGER,
    RASTERDEGERLERI11 INTEGER,
    ORTAKSABLON INTEGER,
    ORTAKSABLONSTR VARCHAR(4),
    ORTAKSABLONISEDESENID VARCHAR(20),
    ORTAKSABLONISESABLONSIRANO INTEGER,
    ACIKLAMA VARCHAR(100),
    SABLONAAITSEPAREDOSYAADI VARCHAR(100),
    LASERPOWER VARCHAR(20),
    FOCUS VARCHAR(20),
    SPEET VARCHAR(20),
    VREZ VARCHAR(20),
    HREZ VARCHAR(20),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    TASARIMNO VARCHAR(20))
AS
begin
  For Select DesenSablon.DesenSablonId, DesenSablon.SablonSiraNo, DesenSablon.SablonAdi, DesenSablon.IlaveSablonNo, DesenSablon.RenkId, Renk.Adi, Renk.Color,
             DesenSablon.MeshId, DesenSablon.RasterVarmi, DesenSablon.RasterDegerleri1, DesenSablon.RasterDegerleri2, DesenSablon.RasterDegerleri3,
             DesenSablon.RasterDegerleri4, DesenSablon.RasterDegerleri5, DesenSablon.RasterDegerleri6,
             DesenSablon.RasterDegerleri7, DesenSablon.RasterDegerleri8, DesenSablon.RasterDegerleri9,
             DesenSablon.RasterDegerleri10, DesenSablon.RasterDegerleri11,
             DesenSablon.OrtakSablon, DesenSablon.OrtakSabloniseDesenId, DesenSablon.OrtakSabloniseSablonSiraNo, DesenSablon.Aciklama,
             DesenSablon.SablonaAitSepareDosyaAdi, Ik.Adi, DesenSablon.InsertTarihi, Sk.Adi, DesenSablon.DegisimTarihi,
             DesenSablon.LaserPower, DesenSablon.Focus, DesenSablon.Speet, DesenSablon.VRez, DesenSablon.HRez, Desen.TasarimNo  from DesenSablon
  left join Renk on Renk.RenkID = DesenSablon.RenkID
  left join Desen on Desen.DesenId = DesenSablon.OrtakSabloniseDesenId
  Left Join Kullanici IK on IK.KullaniciId = DesenSablon.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = DesenSablon.KullaniciId
  Where DesenSablon.DesenID = :ADesenID
  order by DesenSablon.DesenID, DesenSablon.SablonSiraNo, DesenSablon.SablonAdi, DesenSablon.IlaveSablonNo
  into :DesenSablonId, :SablonSiraNo, :SablonAdi, :IlaveSablonNo, :RenkId, :RenkAdi, :Color, MeshId, :RasterVarmi,
       :RasterDegerleri1, :RasterDegerleri2, :RasterDegerleri3, :RasterDegerleri4, :RasterDegerleri5, :RasterDegerleri6,
       :RasterDegerleri7, :RasterDegerleri8, :RasterDegerleri9, :RasterDegerleri10, :RasterDegerleri11,
       :OrtakSablon, :OrtakSabloniseDesenId, :OrtakSabloniseSablonSiraNo, :Aciklama,
       :SablonaAitSepareDosyaAdi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :LaserPower, :Focus, :Speet, :VRez, :HRez, :TasarimNo do
  begin

    OrtakSablonStr = '';
    if (OrtakSablon = 1) then OrtakSablonStr = 'Evet';

    RasterVarmiStr = '';
    if (RasterVarmi = 1) then RasterVarmiStr = 'Evet';

    if (IlaveSablonNo = 0) then IlaveSablonNo = Null;
    suspend;
  end
end
