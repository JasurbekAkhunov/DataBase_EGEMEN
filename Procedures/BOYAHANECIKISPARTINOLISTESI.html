CREATE PROCEDURE BOYAHANECIKISPARTINOLISTESI(
    AFIRMAID VARCHAR(20))
RETURNS (
    CIKISPARTINO VARCHAR(30),
    MIKTAR NUMERIC(12,3),
    BRUTGIRISMIKTAR NUMERIC(12,3))
AS
begin
  for Select CikisPartiNo, Miktar, BrutGirisMiktar from BoyahaneCikisPartiNo where FirmaID = :AFirmaId and KalanVar = 1

end
