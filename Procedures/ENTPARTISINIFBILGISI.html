CREATE PROCEDURE ENTPARTISINIFBILGISI(
    APARTISINIFID INTEGER)
RETURNS (
    HAMADITOPLAMI VARCHAR(253),
    MIKTAR DOUBLE PRECISION,
    PATRIIDSTR VARCHAR(30),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    DESENKODUADIVARYANT VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    FIRMAIDVEADI VARCHAR(150),
    FIRMAKISAADI VARCHAR(10),
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISNOTU VARCHAR(80),
    MUSTERIRENKKODUVEADI VARCHAR(1000),
    SIPARISTARIHI TIMESTAMP,
    PARTIIDSTR VARCHAR(255),
    DOZAJLAMAISEMRIIDSTR VARCHAR(255))
AS
declare variable VBoyaSiparisYil        integer;
declare variable VPartiYil              integer;
declare variable VPartiId               integer;
declare variable VBoyaSiparisId         integer;
declare variable VDesenTasarimNo        varchar(20);
declare variable VPartiKayitId          integer;
declare variable VMusteriRenkKoduveAdi  varchar(1000);
declare variable VHamAdi                varchar(255);
declare variable FDozajlamaIsEmriId     varchar(255);
declare variable ReceteIsEmriYil        integer;
declare variable ReceteIsEmriId         integer;
declare variable FPartiAsamalariId      integer;
begin

  HamAdiToplami = '';
  for select distinct Ham.Adi from PartiStokDetay
  left join Ham on Ham.Hamid = PartiStokDetay.Hamid
  where PartiStokDetay.PartiSinifId = :APartiSinifId into :vHamAdi do
  begin
     if (Char_Length(HamAdiToplami) + Char_Length(vHamAdi) < 253) then
     begin
       if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set UTF8);
       HamAdiToplami = HamAdiToplami || vHamAdi;
     end
  end

  select Sum(Miktar) from PartiStokDetay where PartiSinifId = :APartiSinifId into :Miktar;

  Select Distinct Parti.PartiId, Parti.PartiKayitId, Parti.PartiYil, Parti.ReceteId, Parti.FirmaId, DesenVaryant.DesenId, DesenVaryant.VaryantAdi, Desen.Adi, Desen.TasarimNo,
         Recete.ADi, Firma.ADI, Firma.ADI2, Firma.KISAADI, BoyaSiparis.BoyaSiparisId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id,
         BoyaSiparis.Notu, BoyaSiparis.SiparisTarihi, SiparisGrup.Adi, SiparisGrup2.Adi
  from PartiStokDetay
  Left Join Parti On Parti.PartiKayitId = PartiStokDetay.PartiKayitId
  Left Join DesenVaryant On DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  Left join Desen On Desen.DesenId = DesenVaryant.DesenId
  Left Join Recete On Recete.ReceteId = Parti.ReceteId
  Left Join Firma On Firma.FirmaId = Parti.FirmaId
  Left Join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  Left Join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  Left Join SiparisGrup AS SiparisGrup2 On SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  Where PartiStokDetay.PartiSinifId = :APartiSinifId
  into  :vPartiId, :vPartiKayitId, :vPartiYil, :ReceteId, :FirmaId, :DesenId, :VaryantAdi, :DesenAdi, :vDesenTasarimNo, :ReceteAdi, :FirmaAdi, :FirmaAdi2, :FirmaKisaAdi,
        :vBoyaSiparisId, :vBoyaSiparisYil, :SiparisGrupId, :SiparisGrup2Id, :SiparisNotu, :SiparisTarihi, :SiparisGrupAdi, :SiparisGrup2Adi;


  BoyaSiparisIdStr = '';
  if (vBoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(vBoyaSiparisYil) || ') '  || EgemenIntegerToStr(vBoyaSiparisId);

  PartiIdStr = '(' || EgemenIntegerToStr(vPartiYil)  || ') ' || EgemenIntegerToStr(vPartiId) || ' ';

  DesenKoduAdiVaryant = cast( '(' || Coalesce(DesenId, '') || ') ' || Coalesce(vDesenTasarimNo, '') || iif(coalesce(vDesenTasarimNo, '') <> '', ' - ', '') || DesenAdi || ' - ' || VaryantAdi as blob character set UTF8);

  FirmaIdveAdi = '(' || FirmaId;
  FirmaIdveAdi = FirmaIdveAdi || ') ';
  FirmaIdveAdi = FirmaIdveAdi || FirmaAdi;



  MusteriRenkKoduveAdi = '';

  for Select BoyaSiparisDetay.MusteriRenkKoduveAdi from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiKayitId = :vPartiKayitId
    group by BoyaSiparisDetay.MusteriRenkKoduveAdi
    into :vMusteriRenkKoduveAdi do
  begin
    if (Char_Length(cast(MusteriRenkKoduveAdi as blob character set UTF8)) + Char_Length(cast(vMusteriRenkKoduveAdi as blob character set UTF8)) < 1000) then
    begin
      if (MusteriRenkKoduveAdi <> '' ) then MusteriRenkKoduveAdi = MusteriRenkKoduveAdi || cast(', ' as blob character set UTF8);
      MusteriRenkKoduveAdi = cast(MusteriRenkKoduveAdi as blob character set UTF8) || vMusteriRenkKoduveAdi;
    end
  end

  DozajlamaIsEmriIDStr=null;

  for select PartiAsamaStok.PartiAsamalariID from PartiAsamaStok where PartiAsamaStok.PartiSinifID = :APartisinifID and PartiAsamaStok.AsamayaDahil = 1 into :FPartiAsamalariID do
  begin
    FDozajlamaIsEmriId = Null;
    ReceteIsEmriYil = Null;
    ReceteIsEmriId = Null;
    Select ReceteIsEmriYil, ReceteIsEmriId from ReceteIsEmriDetay where ReceteIsEmriDetay.PartiAsamalariId = :FPartiAsamalariId
      into :ReceteIsEmriYil, :ReceteIsEmriId;

    if (ReceteIsEmriYil is null) then
      Select EgemenIntegerToStr(DozajlamaIsEmriId) from DozajlamaIsEmri where DozajlamaIsEmri.PartiAsamalariId = :FPartiAsamalariId into :FDozajlamaIsEmriId;
    else
      Select EgemenIntegerToStr(DozajlamaIsEmriId) from DozajlamaIsEmri where DozajlamaIsEmri.ReceteIsEmriYil = :ReceteIsEmriYil and DozajlamaIsEmri.ReceteIsEmriId = :ReceteIsEmriId into :FDozajlamaIsEmriId;

    if (FDozajlamaIsEmriId is not null) then
    begin
      if (DozajlamaIsEmriIDStr is null) then  DozajlamaIsEmriIDStr =  coalesce(FDozajlamaIsEmriID,'') || cast(', ' as blob character set UTF8);
      else DozajlamaIsEmriIDStr = coalesce(DozajlamaIsEmriIDStr,'')  || coalesce(FDozajlamaIsEmriID,'');
    end
   end

  suspend;
end
