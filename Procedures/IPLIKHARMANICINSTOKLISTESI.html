CREATE PROCEDURE IPLIKHARMANICINSTOKLISTESI(
    AMINSTOKID INTEGER,
    AMAXSTOKID INTEGER,
    AMINMIKTAR NUMERIC(12,3),
    AMAXMIKTAR NUMERIC(12,3),
    ALOT VARCHAR(30),
    AHAMID INTEGER,
    AFIRMAID VARCHAR(20),
    AMINKAPNO INTEGER,
    AMAXKAPNO INTEGER,
    AHAMTIPI INTEGER,
    ARENKID INTEGER)
RETURNS (
    IPLIKSTOKID INTEGER,
    BOBINSAYISI INTEGER,
    KAPNO INTEGER,
    MIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    LOT VARCHAR(30),
    FLAMKODU VARCHAR(20),
    IPLIKDEPOID VARCHAR(20),
    RENKADI VARCHAR(50))
AS
declare variable hstTelef       integer;
declare variable hstElyaf       integer;
begin
  Select hstTelef, hstElyaf from HamTipiGetir into :hstTelef, :hstElyaf;

  for Select IplikStok.IplikStokId, IplikStok.Miktar, IplikStok.NetMiktar, IplikStok.HamId, Ham.Adi,
         IplikIrsaliyeDetay.Lot, coalesce(IplikIrsaliyeDetay.FlamKodu, IplikUretim.FlamKodu), IplikStok.KapNo, IplikStok.BobinSayisi, IplikStok.IplikDepoId,Renk.Adi from IplikStok
    left join Ham on Ham.HamId = IplikStok.HamId
    left join IplikUretim On IplikUretim.IplikStokID = IplikStok.IplikStokId
    left join IplikLot On IplikLot.IplikLotID = IplikUretim.IplikLotId
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = IplikStok.IplikIrsaliyeDetayId
    left join Renk on Renk.Renkid = IplikIrsaliyeDetay.RenkId
    where (:AMinStokId is Null or IplikStok.IplikStokId >= :AMinStokId) and
          (:AMaxStokId is Null or IplikStok.IplikStokId <= :AMaxStokId) and
          (:AMinKapNo is Null or IplikStok.KapNo >= :AMinKapNo) and
          (:AMaxKapNo is Null or IplikStok.KapNo <= :AMaxKapNo) and
          (:AMinMiktar is Null or IplikStok.Miktar >= :AMinMiktar) and
          (:AMaxMiktar is Null or IplikStok.Miktar <= :AMaxMiktar) and
          ((:ALot is Null or IplikUretim.Lot = :ALot) or (:ALot is Null or IplikIrsaliyeDetay.Lot = :ALot))  and
          (:AHamId is Null or IplikStok.HamId = :AHamId) and
          (:AFirmaId is Null or IplikIrsaliyeDetay.FirmaId = :AFirmaId) and
          ( IplikIrsaliyeDetay.RenkId = :ARenkId or :ARenkId is null) and
          (
           (:AHamTipi is Null and (Ham.HamTipiID = :hstTelef or Ham.HamTipiID = :hstElyaf))
           or
           (:AHamTipi = 1 and (Ham.HamTipiID = :hstElyaf))
           or
           (:AHamTipi = 0 and (Ham.HamTipiID = :hstTelef))
          )
          and
          IplikStok.CikisIplikIrsaliyeDetayId is Null
    into :IplikStokId, :Miktar, :NetMiktar, :HamId, :HamAdi, :Lot, :FlamKodu, :KapNo, :BobinSayisi, :IplikDepoId, :RenkAdi do
  begin
    suspend;
  end
end
