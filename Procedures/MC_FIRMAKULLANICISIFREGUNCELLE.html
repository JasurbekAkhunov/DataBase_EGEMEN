CREATE PROCEDURE MC_FIRMAKULLANICISIFREGUNCELLE(
    AFIRMAID VARCHAR(20),
    AKULLANICIID INTEGER,
    AESKISIFRE VARCHAR(10),
    AYENISIFRE VARCHAR(10))
AS
declare variable FEskiSifre    varchar(10);
declare variable Okey          integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exception KayitHatasi;

  Select Sifre from Kullanici where Kullanici.KullaniciId = :AKullaniciId into :FEskiSifre;

  if (FEskiSifre <> AEskiSifre) then exception KayitHatasi;
  Update Kullanici Set Sifre = :AYeniSifre, SifreDegistirilsin = 0 where Kullanici.KullaniciId = :AKullaniciId;
end
