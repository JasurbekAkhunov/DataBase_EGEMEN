CREATE PROCEDURE PARTIASAMAICIKONTROLBILGISI(
    APROXYID VARCHAR(8),
    AMAKINAID VARCHAR(10))
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTITIPIID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    PARTIASAMALARIID INTEGER,
    PARTISINIFID INTEGER,
    STATUS INTEGER)
AS
declare variable PartiKayitId  integer;
declare variable KayitSayisi   integer;
declare variable AsilMakinaID      MakinaKoduSize;
begin
  Select AsilMakinaID from Makina where Makina.MakinaId = :AMakinaId into :AsilMakinaID;
  if (AsilMakinaID is Not Null) then AMakinaID = :AsilMakinaID;
  Status = 0;

  Select PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.PartiTipiId, PartiStok.HamAdiToplami,
         PartiStok.Kilo, PartiStok.Metre from PartiStok
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  where PartiStok.ProxyId = :AProxyId
  into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :PartiTipiId, :HamAdiToplami, :Kilo, :Metre;

  if (PartiSinifId is not null) then
  begin
    Select Min(PartiAsamalariId) from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiSinifId = :PartiSinifId and
                                                   PartiAsamaStokDetay.MakinaId = :AMakinaId and
                                                   PartiAsamaStokDetay.GirisTarihi is not null and
                                                   PartiAsamaStokDetay.CikisTarihi is null into :PartiAsamalariId;
    if (PartiAsamalariId is null) then Status = 2; /* makinada imalatta degil */
    else
    begin
      Select Count(*) from PartiAsamaIcKontrol where PartiAsamaIcKontrol.PartiSinifId = :PartiSinifId and PartiAsamaIcKontrol.PartiAsamalariId = :PartiAsamalariId into :KayitSayisi;

      if (KayitSayisi > 0) then Status = 0;
      else Status = 3; /* Asamanin yapÄ±lacak ic kontrolu yok */
    end
  end
  else Status = 1; /* proxy Partiye ait degil */
  suspend;
end
