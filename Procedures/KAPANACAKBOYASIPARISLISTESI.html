CREATE PROCEDURE KAPANACAKBOYASIPARISLISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    AORAN DOUBLE PRECISION,
    AMUSTERIKALITENO VARCHAR(20),
    ASIPARISIALANKULLANICIID INTEGER,
    ATALIMATKODU VARCHAR(20))
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISNOSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPARISTARIHI TIMESTAMP,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISTIPIID INTEGER,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIKALITENO VARCHAR(20),
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISNO VARCHAR(20),
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    PARTILENENKILOBASKI NUMERIC(12,3),
    PARTILENENMETREBASKI NUMERIC(12,3),
    IPTALKILO NUMERIC(12,3),
    IPTALMETRE NUMERIC(12,3),
    IPTALKILOBASKI NUMERIC(12,3),
    IPTALMETREBASKI NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    BRUTSEVKEDILENKILO NUMERIC(12,3),
    BRUTSEVKEDILENMETRE NUMERIC(12,3),
    NETSEVKEDILENKILO NUMERIC(12,3),
    NETSEVKEDILENMETRE NUMERIC(12,3),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    MUSTERIRENKKODUVEADI VARCHAR(30),
    REZERVIVAR INTEGER,
    SEVKEDILMEMISPARTIVAR INTEGER,
    SEVKEDILMEMISPARTIIDLER VARCHAR(250),
    KAPATILABILIR INTEGER,
    TALIMATKODU VARCHAR(20),
    REZSEVKEDILENKILO BOYAMIKTARSIZE,
    REZSEVKEDILENBRUTKILO BOYAMIKTARSIZE)
AS
declare variable KayitSayisi  integer;
declare variable FarkMiktar   Double Precision;
declare variable FarkOran     Double Precision;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable PartilenenMiktardanIptaliDus EgeBoolean;
begin
  Select DokumaKumasiKilodanTakipEt, PartilenenMiktardanIptaliDus from Sistem into :DokumaKumasiKilodanTakipEt, :PartilenenMiktardanIptaliDus;

  AOran = 100 - AOran;

  for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.BoyaSiparisDetayId,
             BoyaSiparis.FirmaId, Firma.Adi, BoyaSiparis.SiparisTarihi, BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id, BoyaSiparis.Fason,
             BoyaSiparisDetay.SiparisTipiId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.LabReceteId, LabRecete.LabReceteKodu, LabRecete.Adi,
             LabReceteAtis.AtisNoStr, BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisMetre,
             BoyaSiparisDetay.PartilenenKilo, BoyaSiparisDetay.IptalKilo, BoyaSiparisDetay.PartilenenMetre, BoyaSiparisDetay.IptalMetre,
             BoyaSiparisDetay.HamId, Ham.Adi, Ham.HamTipiId, BoyaSiparisDetay.MusteriRenkKoduveAdi,
             BoyaSiparisDetay.PartilenenKiloBaski, BoyaSiparisDetay.IptalKiloBaski, BoyaSiparisDetay.PartilenenMetreBaski, BoyaSiparisDetay.IptalMetreBaski,
             BoyaSiparisDetay.TalimatKodu from BoyaSiparis
  left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
  where  (BoyaSiparis.SiparisTarihi >= :ABaslangicTarihi and BoyaSiparis.SiparisTarihi <= :ABitisTarihi) and BoyaSiparisDetay.Kapali = 0 and
         (Firma.AnahesapFirmaId = :AFirmaId or :AFirmaId is null) and
         (BoyaSiparis.Fason = :ASatisTipi or :ASatisTipi is null) and
         (BoyaSiparis.siparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
         (BoyaSiparisDetay.SiparisTipiId = :ABoyaSiparisTipiId or :ABoyaSiparisTipiId is null) and
         (BoyaSiparisDetay.MusteriKaliteNo = :AMusteriKaliteNo or :AMusteriKaliteNo is null) and
         (BoyaSiparisDetay.TalimatKodu = :ATalimatKodu or :ATalimatKodu is null)
  order by BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId
  into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId, :FirmaId, :FirmaAdi, :SiparisTarihi,
       :SiparisGrupId, :SiparisGrup2Id, :Fason, :SiparisTipiId, :MusteriSiparisNo, :MusteriKaliteNo, :LabReceteId, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisNo,
       :SiparisKilo, :SiparisMetre, :PartilenenKilo, :IptalKilo, :PartilenenMetre, :IptalMetre,
       :HamId, :HamAdi, :HamTipiId, :MusteriRenkKoduveAdi, :PartilenenKiloBaski, :IptalKiloBaski, :PartilenenMetreBaski, :IptalMetreBaski, :TalimatKodu do
  begin
     Kapatilabilir = 1;
     if (Fason = 1) then FasonStr = 'Fason';
     else FasonStr = '';

     BoyaSiparisNoStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

     Select Count(*) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayId = :BoyaSiparisDetayId and BoyaSiparisRezerv.RezervVar = 1 into :KayitSayisi;

     if (KayitSayisi > 0) then RezerviVar = 1;
     else RezerviVar = 0;


     SevkEdilmemisPartiIdler = Null;
     Select Left(List('(' || Parti.PartiYil || ') ' || Parti.PartiId, ', '), 240) from PartiStok
     Left Join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
     where PartiStok.BoyaSiparisDetayId = :BoyaSiparisDetayId and PartiStok.Durumu <> 8 and PartiStok.PartiStokIptalKayitId is null
     into :SevkEdilmemisPartiIdler;
     if (SevkEdilmemisPartiIdler is not Null) then SevkEdilmemisPartiVar = 1;
     else SevkEdilmemisPartiVar = 0;

     if (SevkEdilmemisPartiVar = 1 or RezerviVar = 1) then Kapatilabilir = 0;

     if (AOran < 100) then
       FarkOran = AOran + 1;
     else
       FarkOran = 0;

     BrutSevkEdilenKilo = Null;
     BrutSevkEdilenMetre = Null;
     NetSevkEdilenKilo = Null;
     NetSevkEdilenMetre = Null;
     RezSevkEdilenKilo = Null;
     RezSevkEdilenBrutKilo = Null;

     Select Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara) from BoyaRezervFisiDetay
     left join PartiStokTarti on BoyaRezervFisiDetay.BoyaRezervFisiDetayId = PartiStokTarti.BoyaRezervFisiDetayId
     where BoyaRezervFisiDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
     and PartiStokTarti.BoyaIrsaliyeDetayId is not null
     into :RezSevkEdilenKilo, :RezSevkEdilenBrutKilo;

     Select Sum(BrutSevkKilo), Sum(BrutSevkMetre), Sum(NetSevkKilo), Sum(NetSevkMetre) from PartiStok where PartiStok.BoyaSiparisDetayId = :BoyaSiparisDetayId
       into :BrutSevkEdilenKilo, :BrutSevkEdilenMetre, :NetSevkEdilenKilo, :NetSevkEdilenMetre;

     if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
     begin
       if (BrutSevkEdilenKilo is not null) then
       begin
         if (Fason = 1) then FarkMiktar = SiparisKilo - BrutSevkEdilenKilo;
         else FarkMiktar = SiparisKilo - NetSevkEdilenKilo;
         if (SiparisKilo <> 0) then FarkOran = FarkMiktar / SiparisKilo * 100;
       end
       else if(RezSevkEdilenBrutKilo is not null) then
       begin
         if (Fason = 1) then FarkMiktar = SiparisKilo - RezSevkEdilenBrutKilo;
         else FarkMiktar = SiparisKilo - RezSevkEdilenKilo;
         if (SiparisKilo <> 0) then FarkOran = FarkMiktar / SiparisKilo * 100;
       end
     end
     else
     begin
       if (BrutSevkEdilenMetre is not null) then
       begin
         if (Fason = 1) then FarkMiktar = SiparisMetre - BrutSevkEdilenMetre;
         else FarkMiktar = SiparisMetre - NetSevkEdilenMetre;
         if (SiparisMetre <> 0) then FarkOran = FarkMiktar / SiparisMetre * 100;
       end
     end

     if (Fason = 1) then
     begin
       KalanKilo = SiparisKilo - BrutSevkEdilenKilo;
       KalanMetre = SiparisMetre - BrutSevkEdilenMetre;
     end
     else
     begin
       KalanKilo = SiparisKilo - NetSevkEdilenKilo;
       KalanMetre = SiparisMetre - NetSevkEdilenMetre;
     end

     if (PartilenenMiktardanIptaliDus = 1) then
     begin
       PartilenenKilo = PartilenenKilo - IptalKilo;
       PartilenenMetre = PartilenenMetre - IptalMetre;
       PartilenenKiloBaski = PartilenenKiloBaski - IptalKiloBaski;
       PartilenenMetre = PartilenenMetreBaski - IptalMetreBaski;
     end

     if (FarkOran <= AOran) then suspend;
  end
end
