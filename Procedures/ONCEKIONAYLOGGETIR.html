CREATE PROCEDURE ONCEKIONAYLOGGETIR(
    AONAYTIPI INTEGER,
    AHAREKETID INTEGER)
RETURNS (
    ONAYLOGID INTEGER,
    HAREKETID INTEGER,
    ONAYDEGERI INTEGER,
    ONAYDEGERISTR VARCHAR(10),
    ONAYTIPI INTEGER,
    ONAYTIPISTR VARCHAR(25),
    ONAYNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40))
AS
declare variable Sorgu    varchar(1000);
begin
  Sorgu = ' select OnayLog.OnayLogID, ' ||
          '   Case OnayLog.OnayTipi ' ||
          '   when 0 then  OnayLog.OrmeSiparisDetayID ' ||
          '   when 1 then  OnayLog.OrmeSiparisDetayID ' ||
          '   when 2 then  OnayLog.BoyaSiparisDetayID ' ||
          '   when 3 then  OnayLog.OrmePlanlamaID ' ||
          '   when 4 then  OnayLog.VerilenSiparisKayitID ' ||
          '   when 5 then  OnayLog.SatinAlmaTalepKayitId ' ||
          '   when 6 then  OnayLog.VerilenSiparisDetayID ' ||
          '   when 7 then  OnayLog.HGSiparisDetayID ' ||
          '   when 8 then  OnayLog.HGSiparisDetayID ' ||
          '   when 9 then  OnayLog.HGIsEmriKayitId ' ||
          '   else 0 end as HareketID, ' ||
          '   OnayLog.OnayDegeri, OnayLog.OnayTipi, OnayLog.OnayNotu, OnayLog.OnayTarihi, OnayLog.OnaylayanKullaniciID, Kullanici.Adi from OnayLog ' ||
          ' Left Join Kullanici on Kullanici.KullaniciId = OnayLog.OnaylayanKullaniciID ' ||
          ' where OnayLog.OnayTipi = '||:AOnayTipi||' and  ';

      if ((:AOnayTipi = 0) or (:AOnayTipi = 1)) then Sorgu = Sorgu || ' OnayLog.OrmeSiparisDetayID = ';
      else if (:AOnayTipi = 2) then Sorgu = Sorgu || ' OnayLog.BoyaSiparisDetayID = ';
      else if (:AOnayTipi = 3) then Sorgu = Sorgu || ' OnayLog.OrmePlanlamaID = ';
      else if (:AOnayTipi = 4) then Sorgu = Sorgu || ' OnayLog.VerilenSiparisKayitID = ';
      else if (:AOnayTipi = 5) then Sorgu = Sorgu || ' OnayLog.SatinAlmaTalepKayitId = ';
      else if (:AOnayTipi = 6) then Sorgu = Sorgu || ' OnayLog.VerilenSiparisDetayID = ';
      else if (:AOnayTipi = 7 or :AOnayTipi = 8) then Sorgu = Sorgu || ' OnayLog.HGSiparisDetayID = ';
      else if (:AOnayTipi = 9) then Sorgu = Sorgu || ' OnayLog.HGIsEmriKayitId = ';
      Sorgu = Sorgu || :AHareketID;

  for execute statement (Sorgu)
      into :OnayLogID, :HareketID, :OnayDegeri, :OnayTipi, :OnayNotu, :OnayTarihi, :OnaylayanKullaniciID, :OnaylayanKullaniciAdi do
  begin
    if (OnayDegeri = 0) then OnayDegeriStr = 'İptal';
    else if (OnayDegeri = 1) then OnayDegeriStr = 'Onay';
    else if (OnayDegeri = 2) then OnayDegeriStr = 'Red';
    else OnayDegeriStr = 'Bilinmiyor';

    OnayTipiStr = '';
    if (OnayTipi = 0) then OnayTipiStr = 'Örme Sip. Onay';
    else if (OnayTipi = 1) then OnayTipiStr = 'Örme Sip. Mali Onay';
    else if (OnayTipi = 2) then OnayTipiStr = 'Boya Sipariş Onay';
    else if (OnayTipi = 3) then OnayTipiStr = 'Talimat Onayi';
    else if (OnayTipi = 4) then OnayTipiStr = 'Verilen Sipariş Onayı';
    else if (OnayTipi = 5) then OnayTipiStr = 'Satin Alma Talep Onayı';
    else if (OnayTipi = 6) then OnayTipiStr = 'Verilen Sip. Detay Onayı';
    else if (OnayTipi = 7) then OnayTipiStr = 'Hazır Giyim Sip. Onay';
    else if (OnayTipi = 8) then OnayTipiStr = 'Hazır G. Sip. Mali Onay';
    else if (OnayTipi = 9) then OnayTipiStr = 'Hazır G. İş Emri Onay';
    else OnayTipiStr = 'Bilinmiyor';

    suspend;
  end
end
