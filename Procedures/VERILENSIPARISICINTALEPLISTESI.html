CREATE PROCEDURE VERILENSIPARISICINTALEPLISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AVERILENSIPARISTIPIID INTEGER,
    ASATINALMABIRIMID INTEGER,
    AFIRMAID VARCHAR(20),
    AURUNTIPIID INTEGER,
    ATEKLIFDURUMU INTEGER)
RETURNS (
    SATINALMATALEPDETAYID INTEGER,
    ONAYDURUMUSTR VARCHAR(15),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    HAMID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30),
    URUNID VARCHAR(20),
    URUNADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    RENKDIGERADI VARCHAR(50),
    MIKTAR DOUBLE PRECISION,
    SIPARISIVERILENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    URUNBIRIMID INTEGER,
    BIRIMID VARCHAR(2),
    BIRIMFIYAT DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TEKLIFTALEPDETAYID INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSKODU HAMKODUSIZE,
    URUNADI2 VARCHAR(80))
AS
declare variable FUrunAdi        varchar(80);
declare variable FHamAdi        varchar(255);
declare variable FSatinAlmaTalepOnayIstensinmi  integer;
declare variable FOnayDurumu integer;
declare variable Okey integer;
declare variable KS integer;
declare variable FSatinAlmaTalepYil      integer;
declare variable FSatinAlmaTalepId      integer;
begin
  Select SatinAlmaTalepOnayIstensinmi from sistem where SistemId = 1 into :FSatinAlmaTalepOnayIstensinmi;

  For Select SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId, SatinAlmaTalep.OnayDurumu, SatinAlmaTalepDetay.SatinAlmaTalepDetayId, SatinAlmaTalepDetay.BoyaSiparisKayitId,
             SatinAlmaTalepDetay.BoyaSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             SatinAlmaTalepDetay.HamId, SatinAlmaTalepDetay.UrunId, Urun.Adi, SatinAlmaTalepDetay.Adi, Ham.Adi,
             SatinAlmaTalepDetay.RenkId, Renk.Adi, Renk.DigerAdi,
             SatinAlmaTalepDetay.Miktar, SatinAlmaTalepDetay.urunBirimId, UrunBirim.BirimId,
             SatinAlmaTalepDetay.BirimFiyat,SatinAlmaTalepDetay.DovizId, SatinAlmaTalepDetay.SiparisiVerilenMiktar, SatinAlmaTalepDetay.Notu,
             TeklifTalepDetay.TeklifTalepDetayId,
             Ik.Adi, SatinAlmaTalepDetay.InsertTarihi, SK.Adi, SatinAlmaTalepDetay.DegisimTarihi, Ham.ReferansKodu, Urun.Adi2 FROM SatinAlmaTalep
    left join SatinAlmaTalepDetay on SatinAlmaTalepDetay.SatinAlmaTalepKayitId = SatinAlmaTalep.SatinAlmaTalepKayitId
    left join TeklifTalepDetay on TeklifTalepDetay.SatinAlmaTalepDetayId = SatinAlmaTalepDetay.SatinAlmaTalepDetayId and TeklifTalepDetay.FirmaID = :AFirmaId and TeklifTalepDetay.OnayDurumu = 1
    LEFT JOIN Urun ON Urun.UrunId = SatinAlmaTalepDetay.UrunId
    left join UrunBirim on UrunBirim.UrunBirimId = SatinAlmaTalepDetay.UrunBirimId and UrunBirim.UrunId = SatinAlmaTalepDetay.UrunId
    LEFT JOIN Ham ON Ham.HamId = SatinAlmaTalepDetay.HamId
    LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = SatinAlmaTalepDetay.BoyaSiparisKayitId
    LEFT JOIN Renk ON Renk.RenkId = SatinAlmaTalepDetay.RenkId
    Left Join Kullanici IK on IK.KullaniciId = SatinAlmaTalepDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = SatinAlmaTalepDetay.KullaniciId
    WHERE SatinAlmaTalep.Tarih >= :ABaslangicTarihi and SatinAlmaTalep.Tarih <= :ABitisTarihi and
          SatinAlmaTalep.SatinAlmaBirimID = :ASatinAlmaBirimID and
          SatinAlmaTalep.SadeceFiyatArastirmasi = 0 and SatinAlmaTalep.IstekYeri = 1 and
          SatinAlmaTalepDetay.SatinAlmaTalepDetayId is not null and
          SatinAlmaTalepDetay.VerilenSiparisTipiId = :AVerilenSiparisTipiId and
          SatinAlmaTalepDetay.Miktar > SatinAlmaTalepDetay.SiparisiVerilenMiktar and
          SatinAlmaTalepDetay.Kapali = 0 and
          (Urun.UrunTipiId = :AUrunTipiId or :AUrunTipiId is null)
  into :FSatinAlmaTalepYil, :FSatinAlmaTalepId, :FOnayDurumu, :SatinAlmaTalepDetayId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
       :HamId, :UrunId, :UrunAdi, :FUrunAdi, :FHamAdi, :RenkId, :RenkAdi, :RenkDigerAdi, :Miktar, :UrunBirimId, :BirimId, :BirimFiyat,
       :DovizId, :SiparisiVerilenMiktar, :Notu, :TeklifTalepDetayId, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, ReferansKodu, :UrunAdi2 do
  begin
    Okey = 1;
    if ((FSatinAlmaTalepOnayIstensinmi = 1 and FOnayDurumu <> 1) or FOnayDurumu = 2) then Okey = 0;

    if (Okey = 1) then
    begin
      if (UrunId is null and HamId is null) then UrunAdi = FUrunAdi;
      else if (HamId is not null) then UrunAdi = FHamAdi;

      Tutar = Miktar * BirimFiyat;

      if (FOnayDurumu = 1) then OnayDurumuStr = 'Onaylandı';
      else if (FOnayDurumu = 2) then OnayDurumuStr = 'Reddedildi';
      else OnayDurumuStr = '';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);


      SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(FSatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(FSatinAlmaTalepId);

      KalanMiktar = Miktar - SiparisiVerilenMiktar;

      Okey = 1;

      if (ATeklifDurumu is not null) then
      begin
        Select Count(*) from TeklifTalepDetay where TeklifTalepDetay.SatinAlmaTalepDetayID = :SatinAlmaTalepDetayID into :KS;

        if (ATeklifDurumu = 0 and KS > 0) then Okey = 0;
        else if (ATeklifDurumu = 1 and KS = 0) then Okey = 0;
      end

      if (Okey = 1) then suspend;
    end
  end
end
