CREATE PROCEDURE BOYAREZERVSEVKEHAZIRLISTESI(
    AFIRMAID VARCHAR(20),
    AKENDIPARTIMIZ INTEGER,
    ADEPOID VARCHAR(10),
    APARTITIPIID INTEGER,
    AHAMID INTEGER,
    AKALITEID KALITEKODUSIZE,
    ALABRECETEKODU RECETEKODUSIZE,
    AKALITENO INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTITIPIID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    RECETEID VARCHAR(20),
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MODELNO VARCHAR(20),
    HAMID INTEGER,
    HAMADI ADI255,
    HAMADITOPLAMI VARCHAR(255),
    LABRECETEKODU RECETEKODUSIZE,
    KALITEID KALITEKODUSIZE,
    TAMAMITARTILDI INTEGER,
    FIREONAYIISTENSIN INTEGER,
    FIREORANI DOUBLE PRECISION,
    FIREONAYIKULLANICIID INTEGER,
    FIREONAYIKULLANICIADI VARCHAR(40),
    FIREONAYTARIHI TIMESTAMP,
    FIREONAYNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    FIRESEBEPTIPIID INTEGER,
    FIRESEBEPTIPIADI VARCHAR(30),
    PARCALISEVK INTEGER,
    SEVKEHAZIRKAPADEDI INTEGER,
    SEVKEHAZIRKILO NUMERIC(12,3),
    SEVKEHAZIRMETRE NUMERIC(12,3),
    SEVKEHAZIRADET INTEGER,
    REZERVADET INTEGER,
    REZERVKAPADEDI INTEGER,
    REZERVMETRE BOYAMIKTARSIZE,
    REZERVKILO BOYAMIKTARSIZE,
    REZERVMIKTAR BOYAMIKTARSIZE,
    BRUTREZERVKILO BOYAMIKTARSIZE,
    BRUTREZERVMIKTAR BOYAMIKTARSIZE,
    NETMIKTAR BOYAMIKTARSIZE,
    NETSARFMIKTAR BOYAMIKTARSIZE,
    NETIADEMIKTAR BOYAMIKTARSIZE,
    NETIADEMETRE BOYAMIKTARSIZE,
    NETIADEKILO BOYAMIKTARSIZE,
    NETSEVKMIKTAR BOYAMIKTARSIZE,
    NETKALANMIKTAR BOYAMIKTARSIZE,
    ISLEMTIPISTR VARCHAR(20),
    ISLEMGRUBUSTR VARCHAR(20),
    TERMINTARIHI TIMESTAMP,
    SEVKEHAZIRTAHMINIBRUTKILO NUMERIC(12,3),
    SEVKEHAZIRTAHMINIBRUTMETRE NUMERIC(12,3),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    PAKETLEMEVAR INTEGER,
    FIRMASEVKADRESIID INTEGER,
    FIRMASEVKADRESIADI VARCHAR(50),
    BASKAFIRMAYASATILABILIR INTEGER,
    IADEALINANFIRMALAR VARCHAR(1000),
    RAFID RAFKODUSIZE,
    RAFADI ADI30,
    REZERVDENSEVKKILO NUMERIC(12,3))
AS
declare variable KALANTARTIKILO numeric(12,3);
declare variable KS integer;
declare variable KAYITSAYISI integer;
declare variable FIREONAYNOTUSTR varchar(250);
declare variable ORMEFIRMAKODUMUZ varchar(20);
declare variable FIRMAKISAADI ADI10;
begin
  Select OrmeFirmaKodumuz from Sistem where sistemid = 1 into :OrmeFirmaKodumuz;

  if (AKendiPartimiz = 1) then
    Select FirmaKodumuz from Sistem where sistemid = 1 into :AFirmaId;

    for select Partistok.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, PartiStok.PartiSinifId, Parti.ReceteId, Parti.ModelNo, PartiStok.HamAdiToplami, PartiStok.TamamiTartildi,
               Parti.ParcaliSevk,
               IslemTipi.Adi, IslemGrubu.Adi, Parti.TerminTarihi,
               Parti.BoyaSiparisKayitId, PartiStok.BoyaSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.MusteriSiparisNo,
               PartiStok.FirmaSevkAdresiId, FirmaSevkAdresi.AdresAdi,
               PartiStok.FireOnayiIstensin, PartiStok.FireOrani, PartiStok.FireOnayiKullaniciId, FOK.Adi, PartiStok.FireOnayTarihi, Left(PartiStok.FireOnayNotu, char_length(PartiStok.FireOnayNotu)),
               PartiStok.FireSebepTipiId, FireSebepTipi.Adi, Max(PartiStokTarti.KaliteKontrol),
               Sum(iif(PartistokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null, PartiStokTarti.Kilo + PartiStokTarti.Dara, null)),
               Sum(iif(PartistokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null, PartiStokTarti.BrutMetre, null)),
               Sum(iif(PartistokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null, PartiStokTarti.Kilo, null)),
               Sum(iif(PartistokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null, PartiStokTarti.Metre, null)),
               Sum(iif(PartistokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null, PartiStokTarti.KapAdedi, null)),
               Sum(iif(PartistokTarti.BoyaIrsaliyeDetayId is null and PartistokTarti.HamStokSarfFisiDetayId is null, PartiStokTarti.Adet, null)), BoyaSiparis.BaskaFirmayaSatilabilir,
               Sum(PartiStokTarti.Kilo),
               Sum(iif(PartiStokTarti.HamStokSarfFisiId is not null, PartiStokTarti.Kilo, Null)),
               Sum(iif(PartiStokTarti.IadeBoyaIrsaliyeId is not null, PartiStokTarti.Kilo, null)),
               Sum(iif(PartiStokTarti.IadeBoyaIrsaliyeId is not null, PartiStokTarti.Metre, null)),
               Sum(iif(PartiStokTarti.IadeBoyaIrsaliyeId is not null, PartiStokTarti.Kilo, null)),
               Sum(iif(PartiStokTarti.BoyaIrsaliyeId is not null, PartiStokTarti.Kilo, Null)),
               Ham.HamId, Ham.Adi, PartiStok.KaliteId, BoyaSiparisDetay.LabReceteKodu,
               Raf.RafId ,Raf.Adi,
               Sum(iif(PartiStokTarti.BoyaIrsaliyeId is not null and PartiStokTarti.BoyaRezervFisiDetayId is not null, PartiStokTarti.Kilo, Null)),
               Sum(iif(PartiStokTarti.BoyaIrsaliyeId is  null and PartiStokTarti.hamstoksarffisidetayid is null, PartiStokTarti.Kilo, Null))
                from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join PartiStokTarti on PartiStokTarti.PartiSinifId = PartiStok.PartiSinifId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join Kullanici FOK on FOK.KullaniciId = PartiStok.FireOnayiKullaniciId
    left join FireSebepTipi on FireSebepTipi.FireSebepTipiId = PartiStok.FireSebepTipiId
    left join FirmaSevkAdresi on FirmaSevkAdresi.FirmaSevkAdresiId = PartiStok.FirmaSevkAdresiId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join Depo on Depo.DepoId = PartiStokTarti.DepoId
    left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId and MamulDepoTransferFisiDetay.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    left join Raf on Raf.RafId = MamulDepoTransferFisiDetay.RafId
    where PartiStok.Durumu = 7 and
         (Parti.FirmaId = :AFirmaId or :AFirmaID = :OrmeFirmaKodumuz or (:AKendiPartimiz = 1 and BoyaSiparis.BaskaFirmayaSatilabilir = 1 and Parti.Fason = 0) or (:AKendiPartimiz = 1 and PartistokTarti.Iadeboyairsaliyeid is Not Null)) and
         Partistok.NetMiktar > (Partistok.NetSevkMiktar + Partistok.NetSarfMiktar + PartiStok.RezervMiktar - PartiStok.NetIadeMiktar) and
         (Parti.PartiTipiId = :APartiTipiId or :APartiTipiId is null) and
         (Parti.ParcaliSevk = 1 or (Parti.ParcaliSevk = 0 and Partistok.TamamiTartildi = 1)) and
         PartistokTarti.PartistokTartiId is not null and
         (PartiStok.HamId = :AHamId or :AHamId is null) and
         (PartiStok.KaliteId = :AKaliteId or :AKaliteId is null) and
         (:AKaliteId is Null and (BoyaSiparisDetay.LabReceteKodu = :ALabReceteKodu or :ALabReceteKodu is null) or :AKaliteId is not Null) and
         (PartiStokTarti.KaliteNo = :AKaliteNo or :AKaliteNo is null) and
         (:ADepoId is not null and Depo.DepoId = :ADepoId or :ADepoId is null and Depo.Sevk = 1) and
         Raf.RafId is not null
    group by Partistok.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, PartiStok.PartiSinifId, Parti.ReceteId, Parti.ModelNo, PartiStok.HamAdiToplami, PartiStok.TamamiTartildi,
             Parti.ParcaliSevk, IslemTipi.Adi, IslemGrubu.Adi, Parti.TerminTarihi,
             Parti.BoyaSiparisKayitId, PartiStok.BoyaSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.MusteriSiparisNo,
             PartiStok.FirmaSevkAdresiId, FirmaSevkAdresi.AdresAdi,
             PartiStok.FireOnayiIstensin, PartiStok.FireOrani, PartiStok.FireOnayiKullaniciId, FOK.Adi, PartiStok.FireOnayTarihi, Left(PartiStok.FireOnayNotu, char_length(PartiStok.FireOnayNotu)),
             PartiStok.FireSebepTipiId, FireSebepTipi.Adi, BoyaSiparis.BaskaFirmayaSatilabilir, Ham.HamId, Ham.Adi, PartiStok.KaliteId, BoyaSiparisDetay.LabReceteKodu,
             Raf.RafId, Raf.Adi
    having sum(iif(PartistokTarti.boyairsaliyeid is null and PartistokTarti.hamstoksarffisidetayid is null, 1, 0)) > 0
    order by Parti.PartiYil, Parti.PartiId
    into :PartiKayitId, :PartiYil, :PartiId, :PartiTipiId, :PartiSinifId, :ReceteId, :ModelNo, :HamAdiToplami, :TamamiTartildi, :ParcaliSevk,
         :IslemTipiStr, :IslemGrubuStr, :TerminTarihi,
         :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :MusteriSiparisNo,
         :FirmaSevkAdresiId, :FirmaSevkAdresiAdi,
         :FireOnayiIstensin, :FireOrani, :FireOnayiKullaniciId, :FireOnayiKullaniciAdi, :FireOnayTarihi, :FireOnayNotuStr,
         :FireSebepTipiId, :FireSebepTipiAdi, :PaketlemeVar,
         :SevkeHazirTahminiBrutKilo, :SevkeHazirTahminiBrutMetre, :SevkeHazirKilo, :SevkeHazirMetre, :SevkeHazirKapAdedi, :SevkeHazirAdet, :BaskaFirmayaSatilabilir,
         :NetMiktar, :NetSarfMiktar, :NetIadeMiktar, :NetIadeMetre, :NetIadeKilo, :NetSevkMiktar, :HamId, :HamAdi, :KaliteId, :LabReceteKodu, :RafId, :RafAdi, :RezervdenSevkKilo,
         :KALANTARTIKILO do
    begin
      BoyaSiparisIdStr = '';
      IadeAlinanFirmalar = '';

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      select count(distinct mamuldepotransferfisidetay.rafid) from PartiStokTarti
      join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.PartiStokTartiId = PartiStokTarti.PartiSTokTartiId and MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = PartiStokTarti.MAmulDepoTransferFisiId
      where partistoktarti.partisinifid = :PartiSinifId
      into :kayitsayisi;

      select count(*) from BoyaRezervFisiDetay where BoyaRezervFisiDetay.PartiSinifId = :PartiSinifId and BoyaRezervFisiDetay.rafid is null into :KS;

      if (kayitsayisi = 1 or KS = 0) then
      begin
          select sum(BoyaRezervFisiDetay.EdilecekAdet), sum(BoyaRezervFisiDetay.EdilecekKapAdedi), sum(BoyaRezervFisiDetay.EdilecekMetre), sum(BoyaRezervFisiDetay.EdilecekKilo),
          sum(BoyaRezervFisiDetay.EdilecekMiktar), sum(BoyaRezervFisiDetay.EdilecekBrutKilo), sum(BoyaRezervFisiDetay.EdilecekBrutMiktar) from BoyaRezervFisiDetay
          where BoyaRezervFisiDetay.PartiSinifId = :PartiSinifId and
          coalesce(BoyaRezervFisiDetay.RafId, :RafId) = :RafId
          into :RezervAdet, :RezervKapAdedi, :RezervMetre, :RezervKilo, :RezervMiktar, :BrutRezervKilo, :BrutRezervMiktar;

          NetKalanMiktar = Coalesce(NetMiktar,0) - Coalesce(NetSevkMiktar,0) - Coalesce(NetSarfMiktar,0) - Coalesce(RezervMiktar,0) + Coalesce(RezervdenSevkKilo, 0); /*rezervden sevk, hem rezervmiktarda hem de sevk miktarinda var oldugundan 2 kere cikarilmis olmasin diye eklendi*/
      end
      else
      begin
          select sum(BoyaRezervFisiDetay.EdilecekAdet), sum(BoyaRezervFisiDetay.EdilecekKapAdedi), sum(BoyaRezervFisiDetay.EdilecekMetre), sum(BoyaRezervFisiDetay.EdilecekKilo),
          sum(BoyaRezervFisiDetay.EdilecekMiktar), sum(BoyaRezervFisiDetay.EdilecekBrutKilo), sum(BoyaRezervFisiDetay.EdilecekBrutMiktar) from BoyaRezervFisiDetay
          where BoyaRezervFisiDetay.PartiSinifId = :PartiSinifId and
          BoyaRezervFisiDetay.tamamlandi = 0
          into :RezervAdet, :RezervKapAdedi, :RezervMetre, :RezervKilo, :RezervMiktar, :BrutRezervKilo, :BrutRezervMiktar;

          NetKalanMiktar = Coalesce(KALANTARTIKILO,0) - Coalesce(RezervMiktar,0);
      end


     /* if (partisinifid = 26056) then
      exception kayithatasi 'NetMiktar:'||Coalesce(NetMiktar,0) ||' NetSevkMiktar:'|| Coalesce(NetSevkMiktar,0) ||' NetSarfMiktar: '|| Coalesce(NetSarfMiktar,0) ||' RezervMiktar:'|| Coalesce(RezervMiktar,0) ||' RezervdenSevkKilo:'|| Coalesce(RezervdenSevkKilo, 0)||' RafId:'||RafId;
          */
      for select distinct coalesce(Firma.KisaAdi, Firma.FirmaId) from PartiStokTarti
      join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = PartiStokTarti.BoyaIrsaliyeId
      join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId where PartiStokTarti.PartiSinifId = :PartiSinifId and PartiStokTarti.IadeBoyaIRsaliyeID is not null
      order by Firma.KisaAdi
      into :FirmaKisaAdi do /*bu partide iade gelenler*/
      begin
        IadeAlinanFirmalar = IadeAlinanFirmalar || iif(IadeAlinanFirmalar = '', '', ', ') || FirmaKisaAdi;
      end

      if (NetKalanMiktar > 0) then
      suspend;
    end
end
