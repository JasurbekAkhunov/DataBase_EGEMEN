Definition

CREATE PROCEDURE ACIKSIPARISTALIMATLISTESI(
    ABOYASIPARISKAYITID INTEGER,
    ADOKUMAHAREKETI INTEGER)
RETURNS (
    RESULTID VARCHAR(50),
    ORMEPLANLAMAID INTEGER,
    DOKUMAPLANLAMAID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISSIRANO INTEGER,
    URETIMPARTINO VARCHAR(30),
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    FASONFIRMAID VARCHAR(20),
    FASONFIRMAADI VARCHAR(100),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40))
AS
declare variable BoyaSiparisYil integer;
declare variable BoyaSiparisId integer;
declare variable BoyaSiparisKayitId integer;
begin
/* boyahaneye cıkısta special listte kullaniyoruz */
  if (ABoyaSiparisKayitId is not null) then
  begin
    For Select OrmePlanlama.OrmePlanlamaId, DokumaPlanlama.DokumaPlanlamaId, BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.SiparisSiraNo,
               Coalesce(OrmePlanlama.UretimPartiNo, DokumaPlanlama.UretimPartiNo), Renk.Adi, Ham.Adi, BoyaSiparis.Firmaid, SiparisFirma.Adi, Coalesce(OrmePlanlama.FirmaId, DokumaPlanlama.FirmaId), FasonFirma.Adi,
               OrmePlanlama.OrmeMakinaId, DokumaPlanlama.DokumaMakinaId, OrmeMakina.Adi, DokumaMakina.Adi  from OrmeSiparisDetay
      left join OrmePlanlama on OrmePlanlama.OrmeSiparisDetayID = OrmeSiparisDetay.OrmeSiparisDetayId
      left join DokumaPlanlama on DokumaPlanlama.OrmeSiparisDetayID = OrmeSiparisDetay.OrmeSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = OrmeSiparisDetay.BoyaSiparisKayitId
      left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
      left join Firma FasonFirma on FasonFirma.FirmaId = OrmePlanlama.FirmaId
      left join OrmeMakina on OrmeMakina.OrmeMakinaID = OrmePlanlama.OrmeMakinaId
      left join DokumaMakina on DokumaMakina.DokumaMakinaID = DokumaPlanlama.DokumaMakinaId
      left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
      left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
      where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and
          (:ADokumaHareketi = 0 and OrmePlanlama.OrmeSiparisDetayID is not null or :ADokumaHareketi = 1 and DokumaPlanlama.OrmeSiparisDetayID is not null)
    into  :OrmePlanlamaId, :DokumaPlanlamaId, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisSiraNo, :UretimPartiNo, :RenkAdi, :HamAdi, :SiparisFirmaid, :SiparisFirmaAdi,
          :FasonFirmaId, :FasonFirmaAdi, :OrmeMakinaId, :DokumaMakinaId, :OrmeMakinaAdi, :DokumaMakinaAdi do
    begin
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      ResultId = '"' || EgemenIntegerToStr(Coalesce(OrmePlanlamaId, DokumaPlanlamaId)) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisKayitId) || '",' ;
      ResultId = ResultId || '"' || UretimPartiNo || '"' ;

      suspend;
    end
  end
  else
  begin
    For Select OrmePlanlama.OrmePlanlamaId, DokumaPlanlama.DokumaPlanlamaId, BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.SiparisSiraNo,
               Coalesce(OrmePlanlama.UretimPartiNo, DokumaPlanlama.UretimPartiNo), Renk.Adi, Ham.Adi, BoyaSiparis.Firmaid, SiparisFirma.Adi, Coalesce(OrmePlanlama.FirmaId, DokumaPlanlama.FirmaId), FasonFirma.Adi,
               OrmePlanlama.OrmeMakinaId, DokumaPlanlama.DokumaMakinaId, OrmeMakina.Adi, DokumaMakina.Adi  from OrmeSiparisDetay
      left join OrmePlanlama on OrmePlanlama.OrmeSiparisDetayID = OrmeSiparisDetay.OrmeSiparisDetayId
      left join DokumaPlanlama on DokumaPlanlama.OrmeSiparisDetayID = OrmeSiparisDetay.OrmeSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = OrmeSiparisDetay.BoyaSiparisKayitId
      left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
      left join Firma FasonFirma on FasonFirma.FirmaId = OrmePlanlama.FirmaId
      left join OrmeMakina on OrmeMakina.OrmeMakinaID = OrmePlanlama.OrmeMakinaId
      left join DokumaMakina on DokumaMakina.DokumaMakinaID = DokumaPlanlama.DokumaMakinaId
      left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
      left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
      where OrmeSiparisDetay.Kapali = 0 and
            (:ADokumaHareketi = 0 and OrmePlanlama.OrmeSiparisDetayID is not null or :ADokumaHareketi = 1 and DokumaPlanlama.OrmeSiparisDetayID is not null)
    into   :OrmePlanlamaId, :DokumaPlanlamaId, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisSiraNo, :UretimPartiNo, :RenkAdi, :HamAdi, :SiparisFirmaid, :SiparisFirmaAdi,
          :FasonFirmaId, :FasonFirmaAdi, :OrmeMakinaId, :DokumaMakinaId, :OrmeMakinaAdi, :DokumaMakinaAdi do
    begin
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      ResultId = '"' || EgemenIntegerToStr(Coalesce(OrmePlanlamaId, DokumaPlanlamaId)) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisKayitId) || '",' ;
      ResultId = ResultId || '"' || UretimPartiNo || '"' ;

      suspend;
    end
  end
end
