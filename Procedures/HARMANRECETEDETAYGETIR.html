CREATE PROCEDURE HARMANRECETEDETAYGETIR(
    AHARMANKAYITID INTEGER)
RETURNS (
    HARMANRECETEID INTEGER,
    HARMANKAYITID INTEGER,
    HARMANRECETEDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    IPLIKSTOKID INTEGER,
    NUMARATOR INTEGER,
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION,
    KULLANIMORANI DOUBLE PRECISION,
    LOT VARCHAR(30),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    INCELIK DOUBLE PRECISION,
    PARLAKLIK DOUBLE PRECISION,
    SARILIK DOUBLE PRECISION,
    UZUNLUK DOUBLE PRECISION,
    KISAELYAFINDEKSI DOUBLE PRECISION,
    MUKAVEMET DOUBLE PRECISION,
    NEMLILIK DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable HarmanMiktari Numeric(12,3);
begin
  Select Harman.HarmanMiktari from Harman where Harman.HarmanKayitId = :AHarmanKayitId into :HarmanMiktari;

  For select HarmanReceteDetay.HarmanKayitId, HarmanReceteDetay.HarmanReceteId, HarmanReceteDetay.HarmanReceteDetayID, HarmanReceteDetay.HamId,
             Ham.Adi, HarmanReceteDetay.KapAdedi, HarmanReceteDetay.Miktar, HarmanReceteDetay.IplikStokId, HarmanReceteDetay.Numarator,
             IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.RenkId, Renk.Adi,
             IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliyeDetay.HareketTipi,
             Ik.Adi, HarmanReceteDetay.InsertTarihi, Sk.Adi, HarmanReceteDetay.DegisimTarihi, IplikStok.Incelik, IplikStok.Parlaklik,
             IplikStok.Sarilik, IplikStok.Uzunluk, IplikStok.KisaElyafIndeksi, IplikStok.Mukavemet, IplikStok.Nemlilik from HarmanReceteDetay
  left join Ham on Ham.HamId = HarmanReceteDetay.HamId
  left join IplikStok on IplikStok.IplikStokId = HarmanReceteDetay.IplikStokId and IplikStok.Numarator = HarmanReceteDetay.Numarator
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = IplikStok.IplikIrsaliyeDetayId
  left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
  left join Renk on Renk.RenkId = IplikIrsaliyeDetay.RenkId
  Left Join Kullanici IK on IK.KullaniciId = HarmanReceteDetay.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HarmanReceteDetay.KullaniciId
  where HarmanReceteDetay.HarmanKayitId = :AHarmanKayitId
  into :HarmanKayitId, :HarmanReceteId, :HarmanReceteDetayID, :HamId,
       :HamAdi, :KapAdedi, :Miktar, :IplikStokId, :Numarator, :Lot, :RenkId, :RenkAdi, :IrsaliyeNo, :IrsaliyeTarihi, :HareketTipi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :Incelik, :Parlaklik,
       :Sarilik, :Uzunluk, :KisaElyafIndeksi, :Mukavemet, :Nemlilik do
  begin
    if (HarmanMiktari > 0) then
    begin
      KullanimOrani = (Miktar / HarmanMiktari) * 100;
      KullanimOrani = Yuvarla(KullanimOrani, 100);
    end

    if (HareketTipi = 0) then HareketTipiStr = 'Alış';
    else if (HareketTipi = 5) then HareketTipiStr = 'Üretim İade';
    else if (HareketTipi = 6) then HareketTipiStr = 'Telef Giriş Fişi';
    else if (HareketTipi = 8) then HareketTipiStr = 'Üretim (boya)';
    else if (HareketTipi = 12) then HareketTipiStr = 'Depo Transfer';
    else if (HareketTipi = 13) then HareketTipiStr = 'Fason Alış';
    else HareketTipiStr = '????';

    suspend;
  end
end
