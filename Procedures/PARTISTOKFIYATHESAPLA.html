CREATE PROCEDURE PARTISTOKFIYATHESAPLA(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    ASONUCDOVIZID VARCHAR(5))
RETURNS (
    HESAPLANANFIYAT DOUBLE PRECISION,
    HESAPLANANUCRETLITAMIRFIYATI DOUBLE PRECISION,
    HESAPLANANUCRETSIZTAMIRFIYATI DOUBLE PRECISION)
AS
declare variable FirmaId        varchar(20);

declare variable AsamaId                    integer;
declare variable ADovizId                   varchar(5);
declare variable ABirimFiyat                Double Precision;

declare variable FADovizId                   varchar(5);
declare variable FABirimFiyat                Double Precision;

declare variable FDovizKuru                  Double Precision;

declare variable PartiTamirId                integer;
declare variable Ucretli                     integer;
declare variable Kilo                        numeric(12,3);

declare variable ReceteId                    varchar(20);
declare variable FToplamKilo                 Double Precision;
declare variable FBanyoCarpani               Double Precision;
declare variable FKoyultma                   Double Precision;

declare variable UrunDovizId                 varchar(5);
declare variable UrunBirimFiyat              Double Precision;
declare variable UrunId                      varchar(20);
declare variable Miktar                      double precision;
declare variable Banyo                       double precision;
declare variable UrunBirimId                 integer;
declare variable BirimCarpan                 Double precision;
declare variable BoyaAsamasimi               integer;
declare variable FLabReceteGrupId            integer;
declare variable FBoyaAsamasimi              integer;
declare variable FKimyasalTipiReceteOrani    DOuble Precision;
declare variable FReceteFiyat                DOuble Precision;
begin
  HesaplananFiyat               = 0;
  HesaplananUcretliTamirFiyati  = 0;
  HesaplananUCretsizTamirFiyati = 0;

  Select Parti.FirmaId, PartiStok.Kilo, Parti.ReceteId from PartiStok
    left join Parti on Parti.partikayitId = PartiStok.PartiKayitId
     where PartiStok.PartiSinifID = :APartiSinifID
     into :FirmaId, :Kilo, :ReceteId;

  for Select PartiAsamalari.PartiTamirId, TamirCozumSekli.Ucretli, Asama.BirimFiyat, Asama.DovizId, FirmaAsama.BirimFiyat, FirmaAsama.DovizId from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join FirmaAsama On FirmaAsama.FirmaId = :FirmaId and FirmaAsama.AsamaId = Asama.AsamaId
    where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.Asamayadahil = 1 and PartiAsamalari.PartininAsilRecetesimi = 0 and
          Asama.ParasalDegeriVar = 1
    into :PartiTamirId, :Ucretli, :ABirimFiyat, :ADovizId, :FABirimFiyat, :FaDovizId do
  begin
    if (FADovizId is not null) then
    begin
      ADovizID = FADovizId;
      ABirimFiyat = FABirimFiyat;
    end

    if (ABirimFiyat = 0) then
    begin
      HesaplananFiyat               = -1;
      HesaplananUcretliTamirFiyati  = -1;
      HesaplananUCretsizTamirFiyati = -1;
      suspend;
      exit;
    end

    if (ASonucDovizId <> ADovizId) then
    begin
      select DovizKuru from DovizKuruGetir(:ADovizId, :ASonucDovizId, Current_date) into :FDovizKuru;

      if (FDovizKuru = 0) then
      begin
        HesaplananFiyat               = -1;
        HesaplananUcretliTamirFiyati  = -1;
        HesaplananUCretsizTamirFiyati = -1;
        suspend;
        exit;
      end
      ABirimFiyat = ABirimFiyat * FDovizKuru;
    end

    if (PartiTamirId is not null) then
    begin
      if (Ucretli = 1) then
        HesaplananUcretliTamirFiyati = HesaplananUcretliTamirFiyati + ABirimFiyat;
      else
        HesaplananUCretsizTamirFiyati = HesaplananUCretsizTamirFiyati + ABirimFiyat;
    end
    else HesaplananFiyat = HesaplananFiyat + ABirimFiyat;
  end

  for Select PartiAsamalari.PartiTamirId, TamirCozumSekli.Ucretli, IlaveIslemler.BirimFiyat, IlaveIslemler.DovizId, FirmaIlaveIslem.BirimFiyat, FirmaIlaveIslem.DovizId from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
    left join PartiIlaveIslemler on PartiIlaveIslemler.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
    left join FirmaIlaveIslem On FirmaIlaveIslem.FirmaId = :FirmaId and FirmaIlaveIslem.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
    where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.Asamayadahil = 1 and PartiAsamalari.PartininAsilRecetesimi = 0 and
          Asama.ParasalDegeriVar = 1 and PartiIlaveIslemler.PartiAsamalariId is not null
    into :PartiTamirId, :Ucretli, :ABirimFiyat, :ADovizId, :FABirimFiyat, :FaDovizId do
  begin
    if (FADovizId is not null) then
    begin
      ADovizID = FADovizId;
      ABirimFiyat = FABirimFiyat;
    end

    if (ABirimFiyat = 0) then
    begin
      HesaplananFiyat               = -1;
      HesaplananUcretliTamirFiyati  = -1;
      HesaplananUCretsizTamirFiyati = -1;
      suspend;
      exit;
    end

    if (ASonucDovizId <> ADovizId) then
    begin
      select DovizKuru from DovizKuruGetir(:ADovizId, :ASonucDovizId, Current_date) into :FDovizKuru;

      if (FDovizKuru = 0) then
      begin
        HesaplananFiyat               = -1;
        HesaplananUcretliTamirFiyati  = -1;
        HesaplananUCretsizTamirFiyati = -1;
        suspend;
        exit;
      end
      ABirimFiyat = ABirimFiyat * FDovizKuru;
    end

    if (PartiTamirId is not null) then
    begin
      if (Ucretli = 1) then
        HesaplananUcretliTamirFiyati = HesaplananUcretliTamirFiyati + ABirimFiyat;
      else
        HesaplananUCretsizTamirFiyati = HesaplananUCretsizTamirFiyati + ABirimFiyat;
    end
    else HesaplananFiyat = HesaplananFiyat + ABirimFiyat;
  end

  Select PartiAsamalari.BanyoCarpani, PartiAsamalari.Koyultma, Recete.LabReceteGrupId, sum(PartiStok.Kilo) from PartiAsamalari
    left join Recete on Recete.ReceteID = PartiAsamalari.ReceteId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariID = PartiAsamalari.PartiAsamalariId
    left join PartiStok on PartiStok.partiSinifId = PartiAsamaStok.partiSinifId
    where PartiAsamalari.PartikayitID = :APartiKayitId and PartiAsamalari.ReceteId = :ReceteId and PartiAsamaStok.AsamayaDahil = 1 and
          PartiAsamalari.PartininAsilRecetesimi = 1
    group by PartiAsamalari.BanyoCarpani, PartiAsamalari.Koyultma, Recete.LabReceteGrupId
    into :FBanyoCarpani, :FKoyultma, :FLabReceteGrupId, :FToplamKilo;

  if (FKoyultma = 0 or FKoyultma is null) then FKoyultma = 1;

  FReceteFiyat = 0;
  for select ReceteDetay.UrunId, (Miktar * :FToplamKilo * :FKoyultma),
         iif(:FLabReceteGrupId = 0, Banyo * :FBanyoCarpani * :FKoyultma, Banyo * :FBanyoCarpani),
         KimyasalTipi.ReceteOrani, ReceteAsama.BoyaAsamasimi,
         UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / RUB.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId from ReceteDetay
  left join Urun on Urun.UrunId = ReceteDetay.UrunId
  left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = Urun.KimyasalTipiId
  left join UrunBirim RUB on RUB.urunBirimId = ReceteDetay.urunBirimId
  left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
  left join UrunFiyat on UrunFiyat.UrunId = ReceteDetay.UrunId
  left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
  where Receteid = :ReceteId
  order by sira
  into :UrunId, :Miktar, :Banyo, :FKimyasalTipiReceteOrani, :FBoyaAsamasimi, :UrunBirimFiyat, :UrunDovizId do
  begin
    if (FBoyaAsamasimi is not null) then BoyaAsamasimi = FBoyaAsamasimi;

    if (FKimyasalTipiReceteOrani = 0 or FKimyasalTipiReceteOrani is null) then FKimyasalTipiReceteOrani = 1;

    if (BoyaAsamasimi = 1) then
    if (FKimyasalTipiReceteOrani is not null) then Banyo = Banyo / FKimyasalTipiReceteOrani;

    if (UrunId is not null) then
    begin
      if (UrunBirimFiyat = 0) then
      begin
        HesaplananFiyat               = -1;
        HesaplananUcretliTamirFiyati  = -1;
        HesaplananUCretsizTamirFiyati = -1;
        suspend;
        exit;
      end

      if (ASonucDovizId <> UrunDovizId) then
      begin
        select DovizKuru from dovizkurugetir(:UrunDovizId, :ASonucDovizId, Current_date) into :FDovizKuru;

        if (FDovizKuru = 0) then
        begin
          HesaplananFiyat               = -1;
          HesaplananUcretliTamirFiyati  = -1;
          HesaplananUCretsizTamirFiyati = -1;
          suspend;
          exit;
        end

        UrunBirimFiyat = UrunBirimFiyat * FDovizKuru;
      end

      if (Miktar > 0) then FReceteFiyat = FReceteFiyat + UrunBirimFiyat * Miktar;
      else if (Banyo > 0) then FReceteFiyat = FReceteFiyat + UrunBirimFiyat * Banyo;
    end
  end

  if (FReceteFiyat <> 0) then
    HesaplananFiyat = HesaplananFiyat + FReceteFiyat / FToplamKilo;

  if (HesaplananFiyat > 0) then
    Select Fiyat from FormataGoreFiyatYuvarla(:HesaplananFiyat, 'BOYA', 'BIRIMFIYAT') into :HesaplananFiyat;

  if (HesaplananUcretliTamirFiyati > 0) then
    Select Fiyat from FormataGoreFiyatYuvarla(:HesaplananUcretliTamirFiyati, 'BOYA', 'BIRIMFIYAT') into :HesaplananUcretliTamirFiyati;

  if (HesaplananUCretsizTamirFiyati > 0) then
    Select Fiyat from FormataGoreFiyatYuvarla(:HesaplananUCretsizTamirFiyati, 'BOYA', 'BIRIMFIYAT') into :HesaplananUCretsizTamirFiyati;

  suspend;
end
