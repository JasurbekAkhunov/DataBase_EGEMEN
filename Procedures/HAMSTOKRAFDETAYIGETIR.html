CREATE PROCEDURE HAMSTOKRAFDETAYIGETIR(
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    RAFDETAYID INTEGER,
    HAMSTOKIRSALIYEID INTEGER,
    HAMSTOKDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RAFID VARCHAR(10),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    IADEEDILECEKKAPADEDI INTEGER,
    IADEEDILECEKKILO NUMERIC(12,3),
    IADEEDILECEKMETRE NUMERIC(12,3),
    IADEEDILECEKADET INTEGER,
    IADEEDILENKAPADEDI INTEGER,
    IADEEDILENKILO NUMERIC(12,3),
    IADEEDILENMETRE NUMERIC(12,3),
    IADEEDILENADET INTEGER,
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    PARTILENENADET INTEGER,
    RAFTRANSFERKILO NUMERIC(12,3),
    RAFTRANSFERMETRE NUMERIC(12,3),
    SATISKILO NUMERIC(12,3),
    SATISMETRE NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    KALANADET INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    AKTARMA INTEGER)
AS
begin
  For Select RafDetay.RafDetayId, RafDetay.HamStokIrsaliyeId, RafDetay.HamStokDetayId, RafDetay.HamId, Ham.Adi, RafDetay.RafId,
             RafDetay.KapAdedi, RafDetay.Kilo, RafDetay.Metre, RafDetay.Adet, RafDetay.IadeEdilecekKapAdedi, RafDetay.IadeEdilecekKilo,
             RafDetay.IadeEdilecekMetre, RafDetay.IadeEdilecekAdet, RafDetay.IadeEdilenKapAdedi, RafDetay.IadeEdilenKilo,
             RafDetay.IadeEdilenMetre, RafDetay.IadeEdilenAdet, RafDetay.PartilenenKilo, RafDetay.PartilenenMetre, RafDetay.PartilenenAdet,
             RafDetay.RafTransferKilo, RafDetay.RafTransferMetre,
             RafDetay.SatisKilo, RafDetay.SatisMetre, RafDetay.Aktarma,
             IK.Adi, RafDetay.InsertTarihi, Sk.Adi, RafDetay.DegisimTarihi FROM HamStokDetay
    left join RafDetay on RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId
    left join Ham on Ham.HamId = RafDetay.HamId
    Left Join Kullanici IK on IK.KullaniciId = RafDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = RafDetay.KullaniciId
    Where HamStokDetay.HamStokMasterId = :AHamStokMasterId and
          RafDetay.RafDetayID is not null and
          RafDetay.RafTransferFisiId is null and RafDetay.PartiStokIptalKayitId is null and
          RafDetay.Aktarma = 0
    into :RafDetayId, :HamStokIrsaliyeId, :HamStokDetayId, :HamId, :HamAdi, :RafId, :KapAdedi, :Kilo, :Metre, :Adet, :IadeEdilecekKapAdedi,
         :IadeEdilecekKilo, :IadeEdilecekMetre, :IadeEdilecekAdet, :IadeEdilenKapAdedi, :IadeEdilenKilo, :IadeEdilenMetre, :IadeEdilenAdet,
         :PartilenenKilo, :PartilenenMetre, :PartilenenAdet,
         :RafTransferKilo, :RafTransferMetre, :SatisKilo, :SatisMetre,
         :Aktarma, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    Select KalanKilo, KalanMetre, KalanAdet From RafDetayHamStokKalaniGetir(:RafDetayId) into :KalanKilo, :KalanMetre, :KalanAdet;
    suspend;
  end
end
