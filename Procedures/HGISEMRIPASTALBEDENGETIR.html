CREATE PROCEDURE HGISEMRIPASTALBEDENGETIR(
    AHGISEMRIPASTALID INTEGER)
RETURNS (
    HGISEMRIPASTALBEDENID INTEGER,
    HGISEMRIPASTALID INTEGER,
    BEDENID INTEGER,
    BEDENADI VARCHAR(30),
    BEDENKISAKODU VARCHAR(7),
    MIKTAR INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    TOPLAMMIKTAR INTEGER)
AS
Declare Variable FToplamKatSayisi Integer;
begin
  Select Sum(HGIsEmriPastalKumas.KatSayisi) From HGIsEmriPastalKumas Where HGIsEmriPastalKumas.HGIsEmriPastalID = :AHGIsEmriPastalID Into :FToplamKatSayisi;

  For select HGIsEmriPastalBeden.HGIsEmriPastalID, HGIsEmriPastalBeden.HGIsEmriPastalBedenID, HGIsEmriPastalBeden.Miktar, HGIsEmriPastalBeden.BedenID,
             Beden.Adi, Beden.KisaKodu, Ik.Adi, HGIsEmriPastalBeden.InsertTarihi, Sk.Adi, HGIsEmriPastalBeden.DegisimTarihi, :FToplamKatSayisi * HGIsEmriPastalBeden.Miktar From HGIsEmriPastalBeden
    Left Join Beden on Beden.BedenId = HGIsEmriPastalBeden.Bedenid
    Left Join Kullanici IK on IK.KullaniciId = HGIsEmriPastalBeden.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = HGIsEmriPastalBeden.KullaniciId
    where HGIsEmriPastalBeden.HGIsEmriPastalID = :AHGIsEmriPastalID
    into :HGIsEmriPastalID, :HGIsEmriPastalBedenID, :Miktar, :BedenID, :BedenAdi, :BedenKisaKodu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :ToplamMiktar do
  begin
    suspend;
  end
end
