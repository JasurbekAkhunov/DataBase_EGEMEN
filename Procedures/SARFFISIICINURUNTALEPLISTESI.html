CREATE PROCEDURE SARFFISIICINURUNTALEPLISTESI(
    AURUNID VARCHAR(20),
    ASATINALMATALEPYIL INTEGER,
    ASATINALMATALEPID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    SATINALMATALEPKAYITID INTEGER,
    SATINALMATALEPDETAYID INTEGER,
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30),
    ONAYDURUMUSTR VARCHAR(15),
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNBIRIMID INTEGER,
    BIRIMID VARCHAR(2),
    MIKTAR DOUBLE PRECISION,
    TESLIMEDILENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable FUrunAdi        varchar(80);
declare variable Okey integer;
begin
  if (AUrunID is not null) then
  begin
    for Select SatinAlmaTalep.SatinAlmaTalepKayitId, SatinAlmaTalepDetay.SatinAlmaTalepDetayId,
               SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId, SatinAlmaTalepDetay.UrunId, Urun.Adi,
               SatinAlmaTalepDetay.Adi, SatinAlmaTalepDetay.Miktar, SatinAlmaTalepDetay.UrunBirimId, UrunBirim.BirimId,
               SatinAlmaTalepDetay.SiparisiVerilenMiktar, SatinAlmaTalepDetay.Notu FROM SatinAlmaTalep
      left join SatinAlmaTalepDetay on SatinAlmaTalepDetay.SatinAlmaTalepKayitId = SatinAlmaTalep.SatinAlmaTalepKayitId
      LEFT JOIN Urun ON Urun.UrunId = SatinAlmaTalepDetay.UrunId
      left join UrunBirim on UrunBirim.UrunBirimId = SatinAlmaTalepDetay.UrunBirimId and UrunBirim.UrunId = SatinAlmaTalepDetay.UrunId
      Left Join Kullanici IK on IK.KullaniciId = SatinAlmaTalepDetay.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = SatinAlmaTalepDetay.KullaniciId
      WHERE (SatinAlmaTalep.Tarih >= :ABaslangicTarihi or :ASatinAlmaTalepId is not null) and
           (SatinAlmaTalep.Tarih <= :ABitisTarihi or :ASatinAlmaTalepId is not null) and
           (SatinAlmaTalep.SatinAlmaTalepYil = :ASatinAlmaTalepYil or :ASatinAlmaTalepYil is null) and
           (SatinAlmaTalep.SatinAlmaTalepId = :ASatinAlmaTalepId or :ASatinAlmaTalepId is null) and
            SatinAlmaTalepDetay.SatinAlmaTalepDetayId is not null and
            SatinAlmaTalep.VerilenSiparisTipiID = 3 and
            (SatinAlmaTalepDetay.UrunId = :AUrunId or SatinAlmaTalepDetay.UrunId is null) and
            SatinAlmaTalepDetay.Miktar > SatinAlmaTalepDetay.SiparisiVerilenMiktar and
            SatinAlmaTalep.OnayDurumu = 1
     into :SatinAlmaTalepKayitId, :SatinAlmaTalepDetayId, :SatinAlmaTalepYil, :SatinAlmaTalepId,
          :UrunId, :UrunAdi, :FUrunAdi, :Miktar, :UrunBirimId, :BirimId, :TeslimEdilenMiktar, :Notu do
    begin
      if (UrunId is null) then UrunAdi = FUrunAdi;

      KalanMiktar = Miktar - TeslimEdilenMiktar;

      SatinAlmaTalepIdStr = '';
      if (SatinAlmaTalepYil is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId)  || ' ';
      suspend;
    end
  end
  else
  begin
    for Select SatinAlmaTalep.SatinAlmaTalepKayitId, SatinAlmaTalepDetay.SatinAlmaTalepDetayId,
               SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId, SatinAlmaTalepDetay.UrunId, Urun.Adi,
               SatinAlmaTalepDetay.Adi, SatinAlmaTalepDetay.Miktar, SatinAlmaTalepDetay.UrunBirimId, UrunBirim.BirimId,
               SatinAlmaTalepDetay.SiparisiVerilenMiktar, SatinAlmaTalepDetay.Notu FROM SatinAlmaTalep
      left join SatinAlmaTalepDetay on SatinAlmaTalepDetay.SatinAlmaTalepKayitId = SatinAlmaTalep.SatinAlmaTalepKayitId
      LEFT JOIN Urun ON Urun.UrunId = SatinAlmaTalepDetay.UrunId
      left join UrunBirim on UrunBirim.UrunBirimId = SatinAlmaTalepDetay.UrunBirimId and UrunBirim.UrunId = SatinAlmaTalepDetay.UrunId
      Left Join Kullanici IK on IK.KullaniciId = SatinAlmaTalepDetay.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = SatinAlmaTalepDetay.KullaniciId
      WHERE (SatinAlmaTalep.Tarih >= :ABaslangicTarihi or :ASatinAlmaTalepId is not null) and
           (SatinAlmaTalep.Tarih <= :ABitisTarihi or :ASatinAlmaTalepId is not null) and
           (SatinAlmaTalep.SatinAlmaTalepYil = :ASatinAlmaTalepYil or :ASatinAlmaTalepYil is null) and
           (SatinAlmaTalep.SatinAlmaTalepId = :ASatinAlmaTalepId or :ASatinAlmaTalepId is null) and
            SatinAlmaTalepDetay.SatinAlmaTalepDetayId is not null and
            SatinAlmaTalep.VerilenSiparisTipiID = 3 and
            SatinAlmaTalepDetay.Miktar > SatinAlmaTalepDetay.SiparisiVerilenMiktar and
            SatinAlmaTalep.OnayDurumu = 1
     into :SatinAlmaTalepKayitId, :SatinAlmaTalepDetayId, :SatinAlmaTalepYil, :SatinAlmaTalepId,
          :UrunId, :UrunAdi, :FUrunAdi, :Miktar, :UrunBirimId, :BirimId, :TeslimEdilenMiktar, :Notu do
    begin
      if (UrunId is null) then UrunAdi = FUrunAdi;

      KalanMiktar = Miktar - TeslimEdilenMiktar;

      SatinAlmaTalepIdStr = '';
      if (SatinAlmaTalepYil is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId)  || ' ';
      suspend;
    end
  end
end
