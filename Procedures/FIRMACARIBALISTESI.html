CREATE PROCEDURE FIRMACARIBALISTESI(
    AFIRMAID VARCHAR(20),
    ABORCALACAK VARCHAR(1),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1))
RETURNS (
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    VALORTARIHI TIMESTAMP,
    MEBLAG DOUBLE PRECISION,
    SATIRSAYISI INTEGER,
    HESAPKAPATMA INTEGER)
AS
Declare variable Toplam         Double precision;
Declare variable RaporToplam    Double precision;
declare variable FDate          Date;
Declare variable Bakiye         Double precision;
Declare variable BakiyeBa       varchar(1);
Declare variable BakiyeVar      integer;
begin
  FDate = current_TimeStamp(0);
  HesapKapatma = 0;
  select sum(Meblag / dovizkuru1 * DovizKuru2)  from cari  where FirmaId = :AFirmaId and Tarih >= :ABaslangicTarihi and tarih <= :ABitisTarihi and  (Cari.Sinif = :ASinif or :ASinif is null) and BorcAlacak <> :ABorcAlacak into :Toplam;

  if (Toplam is null) then Toplam = 0;

  RaporToplam = 0;
  SatirSayisi = 0;
  BakiyeVar = 0;

  for Select Tarih, Aciklama, ValorTarihi, Meblag / dovizkuru1 * DovizKuru2 from Cari where FirmaId = :AFirmaId and tarih >= :ABaslangicTarihi  and tarih <= :ABitisTarihi and  (Cari.Sinif = :ASinif or :ASinif is null) and BorcAlacak = :ABorcAlacak
     order by tarih
     into :Tarih, :Aciklama, :ValorTarihi, :Meblag do
  begin
    if (RaporToplam >= Toplam) then
    begin
      RaporToplam = RaporToplam + Meblag;
      BakiyeVar = 1;
    end
    else
    begin
      if (RaporToplam + Meblag > Toplam) then
      begin
        Bakiye = Meblag;
        Meblag = Toplam - RaporToplam;
        RaporToplam = RaporToplam + Bakiye;
        SatirSayisi = SatirSayisi + 1;
        BakiyeVar = 1;
        suspend;
      end
      else
      begin
        RaporToplam = RaporToplam + Meblag;
        SatirSayisi = SatirSayisi + 1;
        suspend;
      end
    end
  end

  if (Toplam <> 0 or RaporToplam <> 0) then
  begin
    if (RaporToplam < Toplam) then
    begin
      if (ABorcAlacak = 'A') then BakiyeBa = 'B';
      else BakiyeBa = 'A';

      Bakiye = Toplam - RaporToplam;
      select BakiyeVadeTarihi from FirmaCariBakiyeVadesiHesapla(:AFirmaId, :Bakiye, :BakiyeBa, :ABitisTarihi, :ASinif, Null) into :ValorTarihi;
      if (ValorTarihi < FDate) then ValorTarihi = FDate;

      Tarih = FDate;
      Aciklama = 'Guncel Hesap Bakiyesi Kapatma';
      Meblag = Bakiye;

      SatirSayisi = SatirSayisi + 1;
      HesapKapatma = 1;
      suspend;
    end
    else if (Bakiyevar = 1) then
    begin
      Bakiye = RaporToplam - Toplam;
      select BakiyeVadeTarihi from FirmaCariBakiyeVadesiHesapla(:AFirmaId, :Bakiye, :ABorcAlacak, :ABitisTarihi, :ASinif, Null) into :ValorTarihi;

      Tarih = FDate;
      Aciklama = 'Bakiye';
      Meblag = Bakiye;

      SatirSayisi = SatirSayisi + 1;
      HesapKapatma = 0;
      suspend;
    end
  end
end
