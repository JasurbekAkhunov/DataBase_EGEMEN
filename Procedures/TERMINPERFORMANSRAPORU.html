CREATE PROCEDURE TERMINPERFORMANSRAPORU(
    ABASLANGICSEVKTARIHI TIMESTAMP,
    ABITISSEVKTARIHI TIMESTAMP,
    APARTITIPI INTEGER,
    AFASON INTEGER,
    AISLEMTIPIID INTEGER,
    AFIRMAID VARCHAR(20),
    AGECIKMETIPIID INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ORTALAMATERMINGUN INTEGER,
    ERKENSEVK NUMERIC(12,3),
    ZAMANINDASEVK NUMERIC(12,3),
    GECSEVK NUMERIC(12,3),
    MUSTERIKAYNAKLIGECSEVK NUMERIC(12,3),
    BOYAHANEKAYNAKLIGECSEVK NUMERIC(12,3),
    TOPLAMSEVK NUMERIC(12,3),
    PERFORMANS DOUBLE PRECISION,
    PERFORMANSADET DOUBLE PRECISION,
    BOYAHANEPERFORMANS DOUBLE PRECISION,
    BOYAHANEPERFORMANSADET DOUBLE PRECISION)
AS
declare variable FLastFirmaId   varchar(20);

declare variable FSonuc      integer;
declare variable FFirmaId   varchar(20);
declare variable FFirmaAdi  varchar(100);
declare variable FKilo      Numeric(12,3);

declare variable ErkenSevkSayisi      integer;
declare variable ZamanindaSevkSayisi  integer;
declare variable GecSevkSayisi        integer;
declare variable MusteriKaynakliGecSevkSayisi        integer;
declare variable BoyahaneKaynakliGecSevkSayisi        integer;
declare variable ToplamSevkSayisi     integer;
declare variable KayitSayisi          integer;
declare variable MusteriKaynakli      integer;
declare variable TerminGunToplami     integer;
declare variable PartiSayisi          integer;
begin
  FLastFirmaId = '';
  For Select FirmaId, FirmaAdi, Sonuc, MusteriKaynakli, Sum(BrutKilo), Count(*) from TerminPerformansRaporuDetayi(:ABaslangicSevkTarihi, :ABitisSevkTarihi, :APartiTipi, :AFason, :AIslemTipiId, :AFirmaId, :AGecikmeTipiId)
    Group By FirmaId, FirmaAdi, Sonuc, MusteriKaynakli
    Order by FirmaId
    into :FFirmaId, :FFirmaAdi, :FSonuc, :MusteriKaynakli, :FKilo, :KayitSayisi do
  begin
    if (FFirmaId <> FLastFirmaId) then
    begin
      if (FLastFirmaId <> '') then
      begin
        if (ToplamSevk > 0) then Performans = (ErkenSevk + ZamanindaSevk) * 100 / ToplamSevk;
        else Performans = 0;

        PerformansAdet = (ErkenSevkSayisi + ZamanindaSevkSayisi) * 100.0 / ToplamSevkSayisi;

        if (ToplamSevk > 0) then BoyahanePerformans = (ErkenSevk + ZamanindaSevk + MusteriKaynakliGecSevk) * 100 / ToplamSevk;
        else Performans = 0;

        BoyahanePerformansAdet = (ErkenSevkSayisi + ZamanindaSevkSayisi + MusteriKaynakliGecSevkSayisi) * 100.0 / ToplamSevkSayisi;

        OrtalamaTerminGun = 0;
        TerminGunToplami = 0;
        PartiSayisi = 0;
        Select Sum(TerminTarihi - AcilisTarihi), Count(*)  from TerminPerformansRaporuDetayi(:ABaslangicSevkTarihi, :ABitisSevkTarihi, :APartiTipi, :AFason, :AIslemTipiId, :FirmaId, :AGecikmeTipiId)
          into :TerminGunToplami, :PartiSayisi;

        if (PartiSayisi <> 0) then
          OrtalamaTerminGun = TerminGunToplami / PartiSayisi;

        suspend;
      end
      FLastFirmaId = FFirmaId;
      FirmaId  = FFirmaId;
      FirmaAdi = FFirmaAdi;
      ErkenSevk    = 0;
      ZamanindaSevk = 0;
      GecSevk       = 0;
      MusteriKaynakliGecSevk  = 0;
      BoyahaneKaynakliGecSevk = 0;
      ToplamSevk = 0;
      Performans = 0;
      ErkenSevkSayisi = 0;
      ZamanindaSevkSayisi = 0;
      GecSevkSayisi = 0;
      MusteriKaynakliGecSevkSayisi = 0;
      BoyahaneKaynakliGecSevkSayisi = 0;
      ToplamSevkSayisi = 0;
    end

    if (FSonuc = 1) then
    begin
      ErkenSevk = ErkenSevk + FKilo;
      ErkenSevkSayisi = ErkenSevkSayisi + KayitSayisi;
    end
    else if (FSonuc = 0) then
    begin
      ZamanindaSevk = ZamanindaSevk + FKilo;
      ZamanindaSevkSayisi = ZamanindaSevkSayisi + KayitSayisi;
    end
    else if (FSonuc = -1) then
    begin
      if (MusteriKaynakli is null or MusteriKaynakli = 0) then
      begin
        BoyahaneKaynakliGecSevk = BoyahaneKaynakliGecSevk + Fkilo;
        BoyahaneKaynakliGecSevkSayisi = BoyahaneKaynakliGecSevkSayisi + KayitSayisi;
      end
      else
      begin
        MusteriKaynakliGecSevk = MusteriKaynakliGecSevk + Fkilo;
        MusteriKaynakliGecSevkSayisi = MusteriKaynakliGecSevkSayisi + KayitSayisi;
      end

      GecSevk = GecSevk + FKilo;
      GecSevkSayisi = GecSevkSayisi + KayitSayisi;
    end

    ToplamSevk = ToplamSevk + FKilo;
    ToplamSevkSayisi = ToplamSevkSayisi + KayitSayisi;

  end
  if (FLastFirmaId <> '') then
  begin
    if (ToplamSevk > 0) then Performans = (ErkenSevk + ZamanindaSevk) * 100 / ToplamSevk;
    else Performans = 0;

    PerformansAdet = (ErkenSevkSayisi + ZamanindaSevkSayisi) * 100.0 / ToplamSevkSayisi;

    if (ToplamSevk > 0) then BoyahanePerformans = (ErkenSevk + ZamanindaSevk + MusteriKaynakliGecSevk) * 100 / ToplamSevk;
    else Performans = 0;

    BoyahanePerformansAdet = (ErkenSevkSayisi + ZamanindaSevkSayisi + MusteriKaynakliGecSevkSayisi) * 100.0 / ToplamSevkSayisi;

    OrtalamaTerminGun = 0;
    TerminGunToplami = 0;
    PartiSayisi = 0;
    Select Sum(TerminTarihi - AcilisTarihi), Count(*)  from TerminPerformansRaporuDetayi(:ABaslangicSevkTarihi, :ABitisSevkTarihi, :APartiTipi, :AFason, :AIslemTipiId, :FirmaId, :AGecikmeTipiId)
      into :TerminGunToplami, :PartiSayisi;

    if (PartiSayisi <> 0) then
      OrtalamaTerminGun = TerminGunToplami / PartiSayisi;

    suspend;
  end
end
