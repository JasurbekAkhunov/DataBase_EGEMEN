CREATE PROCEDURE DOVIZGRUPLUCARIKARTTOPLAMI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1))
RETURNS (
    DOVIZID VARCHAR(5),
    BORCTOPLAM DOUBLE PRECISION,
    ALACAKTOPLAM DOUBLE PRECISION,
    BAKIYE DOUBLE PRECISION,
    BAKIYEBA VARCHAR(1))
AS
declare variable  FLastDovizId  varchar(5);
declare variable  FDovizId      varchar(5);
declare variable  FBorcAlacak   varchar(1);
declare variable  FMeblag       Double precision;
begin
  FLastDovizId = '';
  for select DovizId, BorcAlacak, Sum(Meblag) from Cari
    where Cari.FirmaId = :AFirmaId and Cari.Tarih >= :ABaslangicTarihi and Cari.Tarih <= :ABitisTarihi and (Cari.Sinif = :ASinif or :ASinif is null)
    Group BY DovizId, BorcAlacak
    into :FDovizId, :FBorcAlacak, :FMeblag do
  begin
    if (FDovizId <> FLastDovizId) then
    begin
      if (FLastDovizId <> '') then
      begin
        if (BorcToplam > AlacakToplam) then
        begin
          Bakiye = BorcToplam - AlacakToplam;
          BakiyeBA = 'B';
        end
        else
        begin
          Bakiye = AlacakToplam - BorcToplam;
          BakiyeBA = 'A';
        end
        suspend;
      end
      DovizId = FDovizId;
      FLastDovizId = FDovizId;
      BorcToplam = 0;
      AlacakToplam = 0;
      Bakiye = 0;
      BakiyeBa = '';
    end
    if (FBorcAlacak = 'B') then BorcToplam = FMeblag;
    else AlacakToplam = FMeblag;
  end
  if (FLastDovizId <> '') then
  begin
    if (BorcToplam > AlacakToplam) then
    begin
      Bakiye = BorcToplam - AlacakToplam;
      BakiyeBA = 'B';
    end
    else
    begin
      Bakiye = AlacakToplam - BorcToplam;
      BakiyeBA = 'A';
    end
    suspend;
  end
end
