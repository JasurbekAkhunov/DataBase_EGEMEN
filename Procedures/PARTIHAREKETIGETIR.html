CREATE PROCEDURE PARTIHAREKETIGETIR(
    APARTIKAYITID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    SIRA INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTIASAMASTOKDETAYID INTEGER,
    PARTISINIFID INTEGER,
    HAMID INTEGER,
    PARTISTOKIPTALKAYITID INTEGER,
    PARTISIRANO INTEGER,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    CIKISKILO NUMERIC(12,3),
    CIKISMETRE NUMERIC(12,3),
    CIKISRAFID VARCHAR(10),
    RECETEISEMRIIDSTR VARCHAR(30),
    BIRLESTIRILENPARTILER VARCHAR(100),
    ASAMAIKAZTIPISTR VARCHAR(50),
    MAKINAPLANLAMAGRUPVEPROSES VARCHAR(110),
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    GRUPPROSESNO VARCHAR(10),
    GRUPPROSESADI VARCHAR(80),
    PROSESNOTU VARCHAR(3000),
    MAKINAID VARCHAR(100),
    PLANLANANMAKINAID VARCHAR(10),
    GIRISPERSONELADI VARCHAR(40),
    KILODEGISTIRENPERSONELADI VARCHAR(40),
    GIRISTARIHI TIMESTAMP,
    GIRISOTOMATIK INTEGER,
    CIKISPERSONELADI VARCHAR(40),
    CIKISTARIHI TIMESTAMP,
    CIKISOTOMATIK INTEGER,
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30),
    OTOMASYONADAHILMI INTEGER,
    SERBESTASAMA INTEGER,
    ASAMAACIKLAMATIPIADI VARCHAR(3000),
    ILAVEISLEMLERADI VARCHAR(3000),
    ILAVEPROSES INTEGER,
    ILAVEPROSESSTR VARCHAR(4),
    ILAVEPROSESSEBEBIID INTEGER,
    ILAVEPROSESSEBEBIADI VARCHAR(20),
    PARTITAMIRID INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    TAMIRDUZELTMETIPIID VARCHAR(10),
    TAMIRDUZELTMETIPIADI VARCHAR(30),
    GIRISVAR VARCHAR(3),
    CIKISVAR VARCHAR(3),
    HAREKETKULLANICIADI VARCHAR(40),
    ASAMADEGISTIRENPERSONELADI VARCHAR(40),
    ASAMADEGISIMSEBEBIADI VARCHAR(15),
    ARAKONTROLASAMASIMI INTEGER,
    ORNEKPARTISINIFMI INTEGER,
    ORNEKPARTISINIFMISTR VARCHAR(5),
    ORNEKONAY INTEGER,
    ORNEKPARTIYIL INTEGER,
    ORNEKPARTIID INTEGER,
    ORNEKPARTIINSERTBILGISI VARCHAR(100),
    ORNEKPARTIONAYBILGISI VARCHAR(100),
    PARTISTOKBEKLEID INTEGER,
    PARTISTOKBEKLEBITTI INTEGER,
    CIKISTAKILOZORUNLU INTEGER,
    CIKISTAMETREZORUNLU INTEGER,
    CIKISTAARABAZORUNLU INTEGER,
    FASAMAID INTEGER,
    BEKLEBASLAMATARIHI TIMESTAMP,
    BEKLENOTU VARCHAR(3000),
    BEKLEKULLANICIID INTEGER,
    BEKLEKULLANICIADI VARCHAR(40),
    DOZAJLAMAISEMRIID INTEGER,
    RECETEISEMRIYIL INTEGER,
    RECETEISEMRIID INTEGER,
    MAKINAADI VARCHAR(40),
    KALITEADI VARCHAR(250),
    TASINANPARTIIDSTR VARCHAR(20))
AS
declare variable FIlaveIslemlerAdi    varchar(30);
declare variable Virgul               integer;
declare variable FBlob                varchar(1000);
declare variable CikisYok             integer;
declare variable FHamAdiToplami       varchar(255);
declare variable FGirisTarihi         timestamp;
declare variable FCikisTarihi         timestamp;
declare variable FPASCikisTarihi      timestamp;
declare variable FGirisPersonelAdi    varchar(40);
declare variable FCikisPersonelAdi    varchar(40);
declare variable FMakinaId            varchar(10);
declare variable FGirisKilo           Numeric(12,3);
declare variable FGirisMetre          Numeric(12,3);
declare variable FKilo                Numeric(12,3);
declare variable FMetre               Numeric(12,3);
declare variable FCikisKilo           Numeric(12,3);
declare variable FCikisMetre          Numeric(12,3);
declare variable FHamTipiId           integer;
declare variable FOrnekTarihi         TimeStamp;
declare variable FOrnekKullaniciAdi   varchar(40);
declare variable FOrnekOnayTarihi     TimeStamp;
declare variable FOrnekOnaylayanKullaniciAdi varchar(40);
declare variable FAsamaAciklamaTipiAdi  varchar(50);
Declare variable FReceteIsEmriYil     integer;
Declare variable FReceteIsEmriId      integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable FAsamaIkazTipi       integer;
declare variable FAsamaIkazTipiStr    varchar(30);
declare variable FBeklemeBitti        integer;

begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  Sira = 0;
  FBlob = '';

  For Select PartiAsamalari.PartiAsamalariId, PartiAsamalari.ReceteId, Recete.Adi, PartiAsamalari.OtomasyonaDahilmi, PartiAsamalari.SerbestAsama,
             PartiAsamalari.PartiTamirId, TamirTipi.Adi, PartiTamir.TamirDuzeltmeTipiId, TamirDuzeltmeTipi.Adi, cast(PartiAsamalari.ProsesNotu as blob character set utf8), PartiAsamalari.CikistaKiloZorunlu,
             PartiAsamalari.CikistaMetreZorunlu, PartiAsamalari.CikistaArabaZorunlu, Personel.Adi, AsamaDegisimSebebi.Adi,
             PartiAsamalari.IlaveProses, PartiAsamalari.IlaveProsesSebebiId, IlaveProsesSebebi.Adi from PartiAsamalari
      left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
      left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
      left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
      Left Join TamirDuzeltmeTipi on TamirDuzeltmeTipi.TamirDuzeltmeTipiId = PartiTamir.TamirDuzeltmeTipiId
      left join Personel on Personel.PersonelId = PartiAsamalari.AsamaDegistirenPersonelID
      left join AsamaDegisimSebebi on AsamaDegisimSebebi.AsamaDegisimSebebiId = PartiAsamalari.AsamaDegisimSebebiID
      left join IlaveProsesSebebi on IlaveProsesSebebi.IlaveProsesSebebiId = PartiAsamalari.IlaveProsesSebebiID
      where PartiAsamalari.PartiKayitId = :APartiKayitId
      Order by PartiAsamalari.Sira
      into :PartiAsamalariId, :ReceteId, :ReceteAdi, :OtomasyonaDahilmi, :SerbestAsama, :PartiTamirId, :TamirTipiAdi, :TamirDuzeltmeTipiId, :TamirDuzeltmeTipiAdi,
           :ProsesNotu, :CikistaKilozorunlu, :CikistaMetreZorunlu, :CikistaArabaZorunlu, :AsamaDegistirenPersonelAdi, :AsamaDegisimSebebiAdi,
           :IlaveProses, :IlaveProsesSebebiId, :IlaveProsesSebebiAdi do
  begin
    select First 1 Asama.AsamaId, Asama.Adi, Asama.AraKontrolAsamasiMi from PartiAsamaStok
    Left Join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and
    PartiAsamaStok.AsamaId is not null into :AsamaId, :AsamaAdi, :AraKontrolAsamasimi;

    Virgul = 0;

    FBlob = '';
    PartiAsamaStokDetayId = null;
    PartiSinifId          = Null;
    PartiSiraNo           = Null;
    PartiStokIptalkayitId = Null;
    MakinaPlanlamaGrupveProses = Null;
    MakinaPlanlamaGrupId   = Null;
    GrupProsesNo           = Null;
    GrupProsesAdi          = Null;

    GirisOtomatik          = Null;
    CikisOtomatik          = Null;
    AsamaAciklamaTipiAdi   = FBlob;

    HareketKullaniciAdi    = Null;
    OrnekPartiSinifmi      = Null;
    OrnekPartiSinifmiStr   = Null;
    OrnekOnay              = Null;
    OrnekPartiYil          = Null;
    OrnekPartiId           = Null;
    OrnekPartiInsertBilgisi = Null;
    OrnekPartiOnayBilgisi   = Null;
    PartiStokBekleId = Null;
    PartiStokBekleBitti = Null;

    HamAdiToplami = '';
    MakinaId = '';
    PlanlananMakinaId = '';
    GirisPersonelAdi = '';
    CikisPersonelAdi = '';
    KiloDegistirenPersonelAdi = '';
    GirisTarihi = Null;
    CikisTarihi = Null;
    FPASCikisTarihi = Null;
    PartiStokBekleId = Null;
    PartiStokBekleBitti = Null;
    Kilo = 0;
    Metre = 0;
    CikisKilo = 0;
    CikisMetre = 0;
    CikisRafID = Null;
    BekleBaslamaTarihi = Null;
    BekleNotu          = Null;
    BekleKullaniciID   = Null;
    BekleKullaniciAdi  = Null;
    TasinanPartiIdStr  = Null;

    if (IlaveProses = 1) then
      IlaveProsesStr = 'Evet';
    else
      IlaveProsesStr = '';

    ReceteIsEmriIdStr = Null;
    BirlestirilenPartiler = '';

    if (ReceteId is not null) then
      Select ReceteIsEmriIdStr, BirlestirilenPartiler from PAReceteIsEmriBilgisiGetir(:APartikayitId, :ReceteId, :PartiAsamalariId) into :ReceteIsEmriIdStr, :BirlestirilenPartiler;

    ReceteIsEmriYil = Null;

    Select ReceteIsEmriDetay.ReceteIsEmriYil, ReceteIsEmriDetay.ReceteIsEmriId from ReceteIsEmriDetay
    where ReceteIsEmriDetay.PartiAsamalariId = :PartiAsamalariId
    into :ReceteIsEmriYil, :ReceteIsEmriId;

    if (ReceteIsEmriYil is null) then
      select DozajlamaIsEmriId from DozajlamaIsEmri where DozajlamaIsEmri.PartiAsamalariId = :PartiAsamalariId into :DozajlamaIsEmriId;
    else
      Select DozajlamaIsEmriId from DozajlamaIsEmri where DozajlamaIsEmri.ReceteIsEmriYil = :ReceteIsEmriYil and DozajlamaIsEmri.ReceteIsEmriId = :ReceteIsEmriId into :DozajlamaIsEmriId;

    For Select PartiStok.HamAdiToplami, Sum(PartiAsamaStok.GirisKilo), Sum(PartiAsamaStok.GirisMetre) from PartiAsamaStok
      left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
      where PartiAsamaStok.PartiKayitId = :APartiKayitId and PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
      Group BY PartiStok.PartiSinifId, PartiStok.HamAdiToplami
      into :FHamAdiToplami, :FKilo, :FMetre do
    begin
      if (Char_Length(HamAdiToplami) + Char_Length(FHamAdiToplami) < 252) then
      begin
        if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set utf8);
        HamAdiToplami = cast(HamAdiToplami as blob character set utf8) || FHamAdiToplami;
      end
      Kilo = Kilo + FKilo;
      Metre = Metre + FMetre;
    end
    CikisYok = 0;
    MakinaID = '';
    For Select PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.CikisTarihi, GP.Adi, CP.Adi,
             PartiAsamaStokDetay.MakinaId, PartiAsamaStok.CikisTarihi, PartiAsamaStokDetay.CikisKilo, PartiAsamaStokDetay.CikisMetre,
             Makina.Adi from PartiAsamaStok
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
      left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
      left join Personel Gp on Gp.PersonelId = PartiAsamaStokDetay.GirisPersonelId
      left join Personel Cp on Cp.PersonelId = PartiAsamaStokDetay.CikisPersonelId
      left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
      where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
      order by PartiAsamaStok.PartiAsamalariId, PartiAsamaStokDetay.GirisTarihi
      into :FGirisTarihi, :FCikisTarihi, :FGirisPersonelAdi, :FCikisPersonelAdi, :FMakinaId, :FPASCikisTarihi, :FCikisKilo, :FCikisMetre,
      :MakinaAdi do
    begin
      if (FCikisTarihi is not null) then
      begin
        CikisKilo = CikisKilo + FCikisKilo;
        CikisMetre = CikisMetre + FCikisMetre;
      end

      if (FGirisTarihi is not null) then
      if ((GirisTarihi is null) or (FGirisTarihi < GirisTarihi)) then
      begin
        GirisTarihi = FGirisTarihi;
        GirisPersonelAdi = FGirisPersonelAdi;
      end


      if (FMakinaId is not null) then
      begin
        if (PosBul(MakinaId, FMakinaId) = 0) then
        if (Char_Length(MakinaId) + Char_Length(FMakinaId) < 97) then
        begin
          if (MakinaId <> '' ) then MakinaId = MakinaId || cast(', ' as varchar(2) character set utf8);
          MakinaId = cast(MakinaId as blob character set utf8) || FMakinaId;
        end
      end

      if (FPASCikisTarihi is not null) then
      begin
        if (CikisYok = 0) then
        if ((FCikisTarihi is not null) or (FCikisTarihi > CikisTarihi) ) then
        begin
          CikisTarihi = FCikistarihi;
          CikisPersonelAdi = FCikisPersonelAdi;
        end
      end
      else
      begin
        CikisYok = 1;
        CikisTarihi = Null;
        CikisPersonelAdi = Null;
      end
    end

    if (GirisTarihi is not null) then GirisVar = 'Var';
    else GirisVar = '';
    if (CikisTarihi is not null) then CikisVar = 'Var';
    else CikisVar = '';

    for select IlaveIslemler.Adi from PartiIlaveIslemler
      left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
      where PartiIlaveIslemler.PartiKayitId = :APartiKayitId and PartiIlaveIslemler.PartiAsamalariId = :PartiAsamalariId
      into :FIlaveIslemlerAdi do
    begin
      if (Virgul = 1) then FBlob = FBlob||cast(', ' as varchar(2) character set UTF8);
      FBlob = FBlob||FIlaveIslemlerAdi;
      Virgul = 1;
    end

    IlaveIslemlerAdi = FBlob;

    if (GirisTarihi is not null) then
    if (CikisTarihi is not null) then GecenSure = CikisTarihi - GirisTarihi;
    else GecenSure = (CURRENT_DATE + CURRENT_TIME) - GirisTarihi;
    else GecenSure = 0;

    if (GecenSure = 0) then GecenSureStr = '';
    else GecenSureStr = TarihToStr(GecenSure);


    FAsamaIkazTipi = Null;
    AsamaIkazTipiStr = '';

    for Select AsamaIkazTipi from PartiAsamaIkazlari
      where PartiAsamaIkazlari.partiAsamalariID = :partiAsamalariID and PartiAsamaIkazlari.KullaniciID = :AKullaniciId
      into :FAsamaIkazTipi do
    begin
      if (FAsamaIkazTipi = 0) then FAsamaIkazTipiStr = 'Başlamadı ise';
      else if (FAsamaIkazTipi = 1) then FAsamaIkazTipiStr = 'Bitmedi İse';
      else if (FAsamaIkazTipi = 2) then FAsamaIkazTipiStr = 'Başlayınca ';
      else if (FAsamaIkazTipi = 3) then FAsamaIkazTipiStr = 'Bitince';
      else FAsamaIkazTipiStr = '';

      if ( Char_Length(AsamaIkazTipiStr) + Char_Length(FAsamaIkazTipiStr) < 48) then
      begin
        if (AsamaIkazTipiStr <> '') then AsamaIkazTipiStr = AsamaIkazTipiStr || cast(',' as varchar(1) character set utf8);
        AsamaIkazTipiStr = cast(AsamaIkazTipiStr as blob character set utf8) || FAsamaIkazTipiStr;
      end
    end

    FAsamaId = AsamaId;
    suspend;


    AsamaIkazTipiStr = '';
    FBlob = '';
    IlaveIslemlerAdi = FBlob;

    AsamaDegistirenPersonelAdi = Null;
    AsamaDegisimSebebiAdi = Null;
    ReceteId = Null;
    ReceteAdi = Null;
    ReceteIsEmriIdStr = Null;
    BirlestirilenPartiler = '';

    AsamaId = Null;
    AsamaAdi = Null;
    TamirTipiAdi = Null;
    ProsesNotu = FBlob;
    TamirDuzeltmeTipiId = null;
    TamirDuzeltmeTipiAdi = Null;
    IlaveProsesStr        = null;
    IlaveProsesSebebiId   = null;
    IlaveProsesSebebiAdi  = null;

    Sira = Sira + 1;

    For Select PartiAsamaStok.PartiSinifId, PartiStok.HamId, PartiStok.PartiSiraNo, PartiStok.PartiStokIptalKayitId, PartiStok.HamAdiToplami, PartiAsamaStok.GirisKilo,
               PartiAsamaStok.GirisMetre, PartiStokBekle.PartiStokBekleId, PartiStokBekle.Bitti, Ham.HamTipiId, PartiStokBekle.BaslamaTarihi, PartiStokBekle.BaslatanKullaniciId,
               Kullanici.Adi ,PartiStokBekle.Notu, PartiStokBekle.Bitti, iif(PartiBolunen.PartiYil is null, null, '('||PartiBolunen.PartiYil||') ' || PartiBolunen.PartiId)  from PartiAsamaStok
      left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
      left join Ham on Ham.HamId = PartiStok.HamId
      left join PartiStokBekle on PartiStokBekle.PartiSinifId = PartiAsamaStok.PartiSinifId and PartiStokBekle.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join Kullanici on Kullanici.KullaniciId = PartiStokBekle.BaslatanKullaniciId
      left join Parti PartiBolunen on PartiBolunen.PartiKayitId = PartiStok.TasindigiPartiKayitId
      where PartiAsamaStok.PartiKayitId = :APartiKayitId and PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
      into :PartiSinifId, :HamId, :PartiSiraNo, :PartiStokIptalKayitId, :HamAdiToplami, :FGirisKilo, :FGirisMetre, :PartiStokBekleId, :PartiStokBekleBitti, :FHamTipiId,
           :BekleBaslamaTarihi, :BekleKullaniciID, :BekleKullaniciAdi, :BekleNotu, :FBeklemeBitti, :TasinanPartiIdStr
      do
    begin

      if (FBeklemeBitti = 1) then
      begin
        BekleBaslamaTarihi = Null;
        BekleNotu          = Null;
        BekleKullaniciID   = Null;
        BekleKullaniciAdi  = Null;
      end

      For Select PartiAsamaStokDetay.PartiAsamaStokDetayId, PartiAsamaStok.PartiSinifId, PS.HamAdiToplami, PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.Metre,
                 PartiAsamaStokDetay.MakinaId, GP.Adi, PartiAsamaStokDetay.GirisTarihi, KDP.Adi, PartiAsamaStokDetay.GirisOtomatik,
                 Cp.Adi, PartiAsamaStokDetay.CikisTarihi, PartiAsamaStokDetay.CikisOtomatik,  HK.Adi,
                 PartiAsamaStokDetay.CikisKilo, PartiAsamaStokDetay.CikisMetre, PartiAsamaStokDetay.CikisRafId,
                 PartiAsamaStok.MakinaPlanlamaGrupId, PartiAsamaStok.GrupProsesNo, MakinaPlanlamaGrupProses.Adi,
                 PartiAsamaStok.OrnekPartiSinifmi, BoyaSiparisOrnekParti.OnayDurumu,
                 BoyaSiparisOrnekParti.InsertTarihi, OPIK.Adi, BoyaSiparisOrnekParti.OnayTarihi, OPOK.Adi, OParti.PartiYil, OParti.PartiId,
                 Coalesce(UretimPlanlama.MakinaId, PartiAsamaStokDetay.PlanlanandigiMakinaID), Kalite.KaliteAdi from PartiAsamaStok
          left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
          left join PartiStok PS on PS.PartiSinifId = PartiAsamaStok.PartiSinifId
          left join MakinaPlanlamaGrupProses on MakinaPlanlamaGrupProses.MakinaPlanlamaGrupId = PartiAsamaStok.MakinaPlanlamaGrupId and MakinaPlanlamaGrupProses.GrupProsesNo = PartiAsamaStok.GrupProsesNo
          left join Personel Gp on Gp.PersonelId = PartiAsamaStokDetay.GirisPersonelId
          left join Personel Cp on Cp.PersonelId = PartiAsamaStokDetay.CikisPersonelId
          left join Personel KDP on KDP.PersonelId = PartiAsamaStokDetay.KiloDegistirenPersonelId
          left join Kullanici HK on HK.KullaniciId = PartiAsamaStokDetay.HareketKullaniciId
          left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PS.BoyaSiparisDetayId
          left join BoyaSiparisOrnekParti on
                       (( (BoyaSiparisOrnekParti.PartiSinifId = BoyaSiparisDetay.OrnekPartiSinifId and BoyaSiparisOrnekParti.PartiAsamalariId = BoyaSiparisDetay.OrnekPartiAsamalariId) and PartiAsamaStok.OrnekPartiSinifmi = 0) or
                        ((BoyaSiparisOrnekParti.PartiSinifId = PS.PartiSinifId and BoyaSiparisOrnekParti.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId ) and PartiAsamaStok.OrnekPartiSinifmi = 1))
          left join PartiStok OrnekPS on OrnekPS.PartiSinifId = BoyaSiparisOrnekParti.PartiSinifId
          left join Parti OParti on OParti.PartiKayitId = OrnekPS.PartiKayitId
          left join BolumPlanlama on BolumPlanlama.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and BolumPlanlama.PartiSinifId = PartiAsamaStok.PartiSinifId and PartiAsamaStokDetay.GirisTarihi is null
          left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
          left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
          left join Kullanici OPIK on OPIK.KullaniciId = BoyaSiparisOrnekParti.InsertKullaniciId
          left join Kullanici OPOK on OPOK.KullaniciId = BoyaSiparisOrnekParti.OnaylayanKullaniciId
          left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
          where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1
          order by PartiAsamaStok.PartiAsamalariId, PartiAsamaStokDetay.GirisTarihi
          into :PartiAsamaStokDetayId, :PartiSinifId, :HamAdiToplami, :Kilo, :Metre,  :MakinaId, :GirisPersonelAdi, :GirisTarihi,
               :KiloDegistirenPersonelAdi, :GirisOtomatik, :CikisPersonelAdi, :CikisTarihi, :CikisOtomatik, :HareketKullaniciAdi,
               :CikisKilo, :CikisMetre, :CikisRafId,
               :MakinaPlanlamaGrupId, :GrupProsesNo, :GrupProsesAdi,
               :OrnekPartiSinifmi, :OrnekOnay, :FOrnekTarihi, :FOrnekKullaniciAdi, :FOrnekOnayTarihi,
               :FOrnekOnaylayanKullaniciAdi,  :OrnekPartiYil, :OrnekPartiId, :PlanlananMakinaId, :KaliteAdi do
      begin

        Fblob = '';
        IlaveIslemlerAdi = FBlob;


        MakinaPlanlamaGrupveProses = '';
        if (MakinaPlanlamaGrupId is not null) then
        begin
          MakinaPlanlamaGrupveProses = '(' || MakinaPlanlamaGrupId || ')';
          if (GrupProsesNo is not null) then MakinaPlanlamaGrupveProses = MakinaPlanlamaGrupveProses || '-' || GrupProsesNo || ' ' || GrupProsesAdi;
        end

        if (OrnekPartiSinifmi = 1) then OrnekPartiSinifmiStr = 'Örnek';
        else OrnekPartiSinifmiStr = '';

        if (FOrnekKullaniciAdi is not null) then OrnekPartiInsertBilgisi = FOrnekKullaniciAdi || ' - ' || TimeStampToDateTimeStr(FOrnekTarihi);

        if (FOrnekOnaylayanKullaniciAdi is not null) then OrnekPartiOnayBilgisi = FOrnekOnaylayanKullaniciAdi || ' - ' || TimeStampToDateTimeStr(FOrnekOnayTarihi);

        if (GirisTarihi is not null) then
        if (CikisTarihi is not null) then GecenSure = CikisTarihi - GirisTarihi;
        else GecenSure = (CURRENT_DATE + CURRENT_TIME) - GirisTarihi;
        else GecenSure = 0;

        if (GecenSure = 0) then GecenSureStr = '';
        else GecenSureStr = TarihToStr(GecenSure);

        if (HamAdiToplami is not null) then
        begin
          if (GirisTarihi is not null) then GirisVar = 'Var';
          else GirisVar = '';
          if (CikisTarihi is not null) then CikisVar = 'Var';
          else CikisVar = '';

          FBlob = '';
          Virgul = 0;

          For Select AsamaAciklamaTipi.Adi from PartiAsamaStokAciklama
           left join AsamaAciklamaTipi on AsamaAciklamaTipi.AsamaAciklamaTipiId = PartiAsamaStokAciklama.AsamaAciklamaTipiId
           where PartiAsamaStokAciklama.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStokAciklama.PartiSinifId = :PartiSinifId
           into :FAsamaAciklamaTipiAdi do
          begin
            if (Virgul = 1) then FBlob = FBlob||cast(', ' as varchar(2) character set UTF8);
            FBlob = FBlob||FAsamaAciklamaTipiAdi;
            Virgul = 1;
          end

          AsamaAciklamaTipiAdi = FBlob;

          FGirisKilo = FGirisKilo - Kilo;
          FGirisMetre = FGirisMetre - Metre;

          suspend;
          FBlob = '';
          AsamaAciklamaTipiAdi = FBlob;

          Sira = Sira + 1;

        end
      end
      PartiAsamaStokDetayId = Null;

      if ( ( (FHamTipiID <> 2 or DokumaKumasiKilodanTakipEt = 1) and FGirisKilo <> 0) or ( (FHamTipiID = 2 and DokumaKumasiKilodanTakipEt = 0) and FGirisMetre <> 0)) then
      begin
        Kilo = FGirisKilo;
        Metre = FGirisMetre;
        MakinaId = Null;
        GirisPersonelAdi = '';
        KiloDegistirenPersonelAdi = '';
        GirisTarihi = Null;
        CikisPersonelAdi = Null;
        CikisTarihi = Null;
        CikisOtomatik = Null;
        GirisOtomatik = Null;
        GirisVar = Null;
        CikisVar = Null;
        HareketKullaniciAdi = Null;
        PartiStokBekleId = Null;
        PartiStokBekleBitti = null;
        PlanlananMakinaId = Null;
        GecenSure = 0;
        GecenSureStr = '';
        CikisKilo = Null;
        CikisMetre = Null;
        CikisRafId = NUll;
        BekleBaslamaTarihi = Null;
        BekleNotu          = Null;
        BekleKullaniciID   = Null;
        BekleKullaniciAdi  = Null;

        suspend;
        Sira = Sira + 1;
      end
    end
  end
