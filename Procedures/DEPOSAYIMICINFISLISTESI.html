CREATE PROCEDURE DEPOSAYIMICINFISLISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFISNO VARCHAR(30))
RETURNS (
    DEPOSAYIMID INTEGER,
    FISNO VARCHAR(50),
    SERINO VARCHAR(2),
    SINIF VARCHAR(1),
    FISTARIHI TIMESTAMP,
    NOTU VARCHAR(80),
    RAFID RAFKODUSIZE,
    DEPOID DEPOKODUSIZE)
AS
begin
  for Select DepoSayim.DepoSayimID, DepoSayim.FisNo, DepoSayim.SeriNo, DepoSayim.Sinif, DepoSayim.FisTarihi, DepoSayim.RafID,
             DepoSayim.Notu, DepoSayim.DepoID from DepoSayim
  where (DepoSayim.fistarihi >= :ABaslangicTarihi and DepoSayim.fistarihi <= :ABitisTarihi) and
        ((DepoSayim.FisNo Like '%'||:AFisNo||'%' and :AFisNo is not null)or (:AFisNo is  null))
  Order By DepoSayim.DepoSayimID
  into :DepoSayimID, :FisNo, :SeriNo, :Sinif, :FisTarihi, :RafID, :Notu, :DepoID do
  begin
    suspend;
  end
end
