CREATE PROCEDURE REZERVDENHIZLIPARTIAC(
    ABOYASIPARISKAYITID INTEGER,
    ABOYASIPARISDETAYID INTEGER,
    ABOYASIPARISSARJID INTEGER,
    ABOYASIPARISBASKIID INTEGER,
    ABOYASIPARISREZERVID INTEGER,
    AKILO NUMERIC(12,3),
    AMETRE NUMERIC(12,3),
    AKAPADEDI INTEGER,
    AISLEMGRUBUID INTEGER,
    AACILIYETDERECESIID INTEGER,
    APARTIGRUPID VARCHAR(4),
    ATERMINTARIHI TIMESTAMP,
    APROSESID INTEGER,
    ANOTU VARCHAR(255),
    AMAKINAID VARCHAR(10),
    ABOBINSAYISI INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIKAYITID INTEGER)
AS
declare variable PartiSinifId integer;
declare variable AMiktar      Numeric(12,3);
declare variable RafDetayId   integer;
declare variable hamTipiId    integer;
declare variable IslemTipiId  integer;
declare variable RezervKalanMiktar  Numeric(12,3);
declare variable KalanMiktar  Numeric(12,3);
declare variable Kilo         Numeric(12,3);
declare variable Metre        Numeric(12,3);
declare variable ParcaliSevk  integer;
declare variable PartiTipiId  integer;
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable PartiAsamalariId integer;
declare variable ImalatTipiId     integer;
declare variable FKaliteId    KaliteKoduSize;
begin
  ImalatTipiId = 0;

  if (ABoyaSiparisBaskiID is not null) then ImalatTipiId = 1;

  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  Select Ham.HamTipiId, BoyaSiparisDetay.IslemTipiId, BoyaSiparisDetay.KaliteId from BoyaSiparisDetay
    left join Ham on Ham.HamID = BoyaSiparisDetay.HamId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitId
    where BoyaSiparisDetay.BoyaSiparisDetayID = :ABoyaSiparisDetayID
    into :HamTipiId, :IslemTipiId, :FKaliteId;

  if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then AMiktar = AKilo;
  else AMiktar = AMetre;

  if (HamTipiID = 0 or HamTipiID = 5 or HamTipiID = 7) then PartiTipiID = 0; /* iplik */
  else PartiTipiID = 1; /* kumas */

  Select (Miktar - KullanilanMiktar) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisRezervID = :ABoyaSiparisRezervID into :RezervKalanMiktar;

  if (RezervKalanMiktar < AMiktar or AMiktar = 0) then exit;

  PartiYil = TimeStampToYil(Current_Date);
  Select ID FROM PartiNumaratorAL(:PartiYil) into :PartiId;

  if (PartiTipiId = 0) then
    Select IplikDefaultParcaliSevkDurumu from Sistem where SistemID = 1 into :ParcaliSevk;
  else
    Select DefaultParcaliSevkDurumu from Sistem where SistemID = 1 into :ParcaliSevk;

  insert into Parti(PartiYil, PartiId, AcilisTarihi, FirmaId, PartiGrupId, Fason, ImalatTipiId, IslemGrubuId, IslemTipiId, AciliyetDerecesiId, TerminTarihi,
                    BoyaSiparisKayitId, PartiTipiId, PlanlananImalatTarihi, ReceteOkey, Talimat, TalimatOkey, ParcaliSevk, Notu,
                    OnayDurumu, InsertKullaniciId, KullaniciId)
            Select :PartiYil, :PartiId, Current_Date, BoyaSiparis.FirmaId, :APartiGrupId, BoyaSiparis.Fason, :ImalatTipiId, :AIslemGrubuId, :IslemTipiId, :AAciliyetDerecesiId,
                   :ATerminTarihi, :ABoyaSiparisKayitId, :PartiTipiId, Current_date, 1, 1, 1, :ParcaliSevk, :ANotu, 0, :AKullaniciId, :AKullaniciId from BoyaSiparis
             where BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitID;

  Select Parti.PartiKayitID from Parti where Parti.PartiYil = :PartiYil and Parti.PartiID = :PartiId into :PartiKayitId;

  PartiSinifId = gen_id(PartiSinif_Id_gen, 1);

  Insert Into PartiStok(PartiKayitId, PartiSinifId, ProsesId, BoyaSiparisDetayId, BoyaSiparisBaskiId, BoyaSiparisSarjId, KaliteId, InsertKullaniciId, KullaniciId)
              values(:PartiKayitId, :PartiSinifId, :AProsesId, :ABoyaSiparisDetayId, :ABoyaSiparisBaskiId, :ABoyaSiparisSarjId, :FKaliteId, :AKullaniciId, :AKullaniciId);

  if (HamTipiID <> 2 or DokumaKumasiKilodanTakipEt = 1) then
  begin
    For Select RafDetayId, KalanMiktar from SiparisRezervRafListesiGetir(:ABoyaSiparisKayitId) where BoyaSiparisRezervId = :ABoyaSiparisRezervId
       into :RafDetayId, :KalanMiktar do
    begin
      if (AMiktar > 0) then
      begin
        if (AMiktar > KalanMiktar) then Kilo = KalanMiktar;
        else Kilo = AMiktar;


        if (AKilo <> 0) then
          Metre = AMetre * Kilo / AKilo;
        else
          Metre = 0;

        AMiktar = AMiktar - Kilo;

        insert into PartiStokDetay(PartiKayitId, PartiSinifId, RafDetayId, KapAdedi, Kilo, Metre, BoyaSiparisRezervId, InsertKullaniciId, KullaniciId, BobinSayisi)
                      values(:PartiKayitId, :PartiSinifId, :RafDetayId, :AKapAdedi, :Kilo, :Metre, :ABoyaSiparisRezervId, :AKullaniciId, :AKullaniciId, :ABobinSayisi);
        AKapAdedi = 0;
      end
    end
  end
  else
  begin
    For Select RafDetayId, KalanMiktar from SiparisRezervRafListesiGetir(:ABoyaSiparisKayitId) where BoyaSiparisRezervId = :ABoyaSiparisRezervId
       into :RafDetayId, :KalanMiktar do
    begin
      if (AMiktar > 0) then
      begin
        if (AMiktar > KalanMiktar) then Metre = KalanMiktar;
        else Metre = AMiktar;

        AMiktar = AMiktar - Metre;

        insert into PartiStokDetay(PartiKayitId, PartiSinifId, RafDetayId, KapAdedi, Kilo, Metre, BoyaSiparisRezervId, InsertKullaniciId, KullaniciId, BobinSayisi)
                     values(:PartiKayitId, :PartiSinifId, :RafDetayId, :AKapAdedi, :AKilo, :Metre, :ABoyaSiparisRezervId, :AKullaniciId, :AKullaniciId, :ABobinSayisi);
        AKilo = 0;
        AKapAdedi = 0;
      end
    end
  end

  if (AMiktar > 0) then exception KayitHatasi 'Uygun Stok yok Parti Olusturulamadi..';


  if (AMakinaId is not null) then
  begin
    Execute Procedure PartiAsamalariOlustur(:PartiKayitId, :AKullaniciId);

    Select PartiAsamalariId from PartiAsamalari where PartiAsamalari.PartiKayitID = :PartiKayitID and PartiAsamalari.PartininAsilRecetesimi = 1 into :PartiAsamalariId;

    if (PartiAsamalariId is not null) then
      Execute Procedure AsamaGirisindenPlanlamaYap(:PartiAsamalariId, :AMakinaId, :AKullaniciId);
  end

  suspend;
end
