CREATE PROCEDURE SIPARISFIRERAPORU(
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    ABOYASIPARISTIPIID INTEGER,
    ASATISTIPI INTEGER,
    AHAMID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    SIPARISADET INTEGER,
    SIPARISKILO NUMERIC(12,3),
    SIPARISBRUTKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    PARTIADET INTEGER,
    SEVKEDILENPARTISAYISI INTEGER,
    PARTILENENMIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    IPTALKILO NUMERIC(12,3),
    IPTALMETRE NUMERIC(12,3),
    FIREORANI DOUBLE PRECISION)
AS
begin
  For Select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.Fason,
             BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparis.SiparisTarihi, BoyaSiparis.FirmaId, Firma.Adi, BoyaSiparisDetayGetir.HamId, BoyaSiparisDetayGetir.HamAdi,
             Sum(BoyaSiparisDetayGetir.SiparisBrutKilo), Sum(BoyaSiparisDetayGetir.SiparisKilo), Sum(BoyaSiparisDetayGetir.SiparisMetre), Count(*) from BoyaSiparis
    left join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
    left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 On SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join BoyaSiparisDetayGetir(BoyaSiparis.BoyaSiparisKayitId, null) On BoyaSiparisDetayGetir.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
    where (BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
          (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil  and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId)) and
          (BoyaSiparisDetayGetir.OnayDurumu = 1) and
          (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
          (BoyaSiparis.Fason = :ASatisTipi or :ASatisTipi is null) and
          (BoyaSiparisDetayGetir.SiparisTipiId = :ABoyaSiparisTipiId or :ABoyaSiparisTipiId is null) and
          (BoyaSiparisDetayGetir.HamId = :AHamId or :AHamId is null)
    group by BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.Fason,
             BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparis.SiparisTarihi, BoyaSiparis.FirmaId, Firma.Adi, BoyaSiparisDetayGetir.HamId, BoyaSiparisDetayGetir.HamAdi
    into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :Fason, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisTarihi, :FirmaId, :FirmaAdi, :HamId, :HamAdi,
         :SiparisBrutKilo, :SiparisKilo, :SiparisMetre, :SiparisAdet do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    if (Fason = 0) then FasonStr = 'Satış';
    else FasonStr = 'Fason';

    Select Count(*) from PartiStok
      left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
      where PartiStok.BoyaSiparisKayitID = :BoyaSiparisKayitID and PartiStok.HamId = :HamId and PartiStok.partiStokIptalKayitId is null
      into :PartiAdet;

    Select Count(*) from PartiStok
      left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
      where PartiStok.BoyaSiparisKayitID = :BoyaSiparisKayitID and PartiStok.HamId = :HamId and PartiStok.partiStokIptalKayitId is null and PartiStok.Durumu = 8
      into :SevkEdilenPartiSayisi;

    Select Sum(iif(PartiStok.partiStokIptalKayitId is null, PartiStok.Miktar, null)), Sum(iif(PartiStok.partiStokIptalKayitId is null, PartiStok.BrutSevkMiktar, null)),
           Sum(iif(PartiStok.partiStokIptalKayitId is null, PartiStok.BrutSevkKilo, null)), Sum(iif(PartiStok.partiStokIptalKayitId is null, PartiStok.BrutSevkMetre, null)),
           Sum(iif(PartiStok.partiStokIptalKayitId is null, PartiStok.NetSevkMiktar, null)), Sum(iif(PartiStok.partiStokIptalKayitId is null, PartiStok.NetSevkKilo, null)),
           Sum(iif(PartiStok.partiStokIptalKayitId is null, PartiStok.NetSevkMetre, null)),
           Sum(iif(PartiStok.PartiStokIptalKayitId is not null, PartiStok.Kilo, null)), Sum(iif(PartiStok.PartiStokIptalKayitId is not null, PartiStok.Metre, null)) from PartiStok
      left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
      where PartiStok.BoyaSiparisKayitID = :BoyaSiparisKayitID and PartiStok.HamId = :HamId
      into :PartilenenMiktar, :BrutMiktar, :BrutKilo, :BrutMetre, :NetMiktar, NetKilo, :NetMetre, :IptalKilo, :IptalMetre;

    FireOrani = BrutMiktar - NetMiktar;
    if (FireOrani > 0) then
      FireOrani = (FireOrani / BrutMiktar) * 100;
    FireOrani = Yuvarla(FireOrani, 100);
    suspend;
  end
end
