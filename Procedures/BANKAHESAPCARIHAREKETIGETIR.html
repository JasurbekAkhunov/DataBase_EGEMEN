CREATE PROCEDURE BANKAHESAPCARIHAREKETIGETIR(
    ABANKAHESAPID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1),
    AGRUP1ID BANKAHESAPGRUPKODUSIZE,
    AGRUP2ID BANKAHESAPGRUPKODUSIZE,
    ABANKACARIHAREKETTIPIID BANKACARIHAREKETTIPISIZE,
    ADOVIZID DOVIZSIZE,
    ARAPORDOVIZID DOVIZSIZE)
RETURNS (
    BANKAHESAPCARIID INTEGER,
    BANKAHESAPID VARCHAR(20),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    BORCALACAK VARCHAR(1),
    SINIF VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    CARIHAREKETTIPIID VARCHAR(10),
    BANKACARIHAREKETTIPIADI VARCHAR(30),
    DOVIZKURUSTR VARCHAR(40),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSYERI INTEGER,
    REFERANSID INTEGER,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    BORCBAKIYE DOUBLE PRECISION,
    ALACAKBAKIYE DOUBLE PRECISION,
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZMEBLAG DOUBLE PRECISION,
    BANKAHESAPGRUP1ADI ADI30,
    BANKAHESAPGRUP2ADI ADI30,
    RAPORDOVIZMEBLAG DOUBLE PRECISION,
    RAPORDOVIZBORC DOUBLE PRECISION,
    RAPORDOVIZALACAK DOUBLE PRECISION,
    RAPORDOVIZBORCBAKIYE DOUBLE PRECISION,
    RAPORDOVIZALACAKBAKIYE DOUBLE PRECISION)
AS
declare Variable ToplamBorc double precision;
declare variable ToplamAlacak double precision;
declare variable CsReferansId integer;
declare variable StReferansId integer;
declare variable KsReferansId integer;
declare variable BHCReferansId integer;
declare variable DovizKuru1 Double Precision;
declare variable DovizKuru2 Double Precision;
declare variable IslemDovizKuru1 Double Precision;
declare variable IslemDovizKuru2 Double Precision;
declare variable FirmaDovizId varchar(5);
declare variable BankaHesapDovizId varchar(5);
declare variable RaporDovizKuru Double Precision;
begin
  ToplamBorc = 0;
  ToplamAlacak = 0;

  select  sum(Meblag)  from BankaHesapCari
  left join BankaHesap on BankaHesap.BankaHesapId = BankaHesapCari.BankaHesapId
  where BankaHesapCari.BankaHesapId = :ABankaHesapId and tarih < :ABaslangicTarihi and BorcAlacak = 'A' and (BankaHesapCari.Sinif = :ASinif or :ASinif is null)
  and (BankaHesap.DovizId = :ADovizId or :ADovizId is null)
  and (BankaHesap.BankaHesapGrup1 = :AGrup1Id or :AGrup1Id is null)
  and (BankaHesap.BankaHesapGrup2 = :AGrup2Id or :AGrup2Id is null) into :ToplamAlacak;

  select  sum(Meblag)  from BankaHesapCari
  left join BankaHesap on BankaHesap.BankaHesapId = BankaHesapCari.BankaHesapId
  where BankaHesapCari.BankaHesapId = :ABankaHesapId and tarih < :ABaslangicTarihi and BorcAlacak = 'B' and (BankaHesapCari.Sinif = :ASinif or :ASinif is null)
  and (BankaHesap.DovizId = :ADovizId or :ADovizId is null)
  and (BankaHesap.BankaHesapGrup1 = :AGrup1Id or :AGrup1Id is null)
  and (BankaHesap.BankaHesapGrup2 = :AGrup2Id or :AGrup2Id is null) into :ToplamBorc;

  select DovizId from BankaHesap Where BankaHesap.BankaHesapId = :ABankaHesapId into :BankaHesapDovizId;

  BankaHesapCariId = -1;
  ReferansYeri = -1;
  ReferansId = -1;
  BankaHesapId = ABankaHesapId;
  FirmaId = null;
  FirmaAdi = null;
  Tarih  = null;
  Aciklama = 'Devir';
  BorcAlacak = null;
  Sinif = Null;
  Meblag = null;
  Notu = null;
  KullaniciAdi = null;
  DegisimTarihi = null;
  Borc = null;
  Alacak = Null;
  BorcBakiye = null;
  AlacakBakiye = Null;
  IslemDovizId = Null;
  IslemDovizMeblag = Null;
  if (ToplamBorc > 0) then Borc = ToplamBorc;
  if (ToplamAlacak > 0) then Alacak = ToplamAlacak;

  if (toplamBorc is Null) then ToplamBorc = 0;
  if (ToplamAlacak is Null) then ToplamAlacak = 0;
  if (ToplamBorc > ToplamAlacak) then BorcBakiye = ToplamBorc - ToplamAlacak;
  else AlacakBakiye = ToplamAlacak - ToplamBorc;
  if (BorcBakiye = 0) then BorcBakiye = null;
  if (AlacakBakiye = 0) then AlacakBakiye = null;

  suspend;

  for select BankaHesapCari.BankaHesapCariId,  BankaHesapCari.BankaHesapId, BankaHesapCari.FirmaId, Firma.Adi, Firma.Adi2, Firma.DovizId, BankaHesapCari.Tarih, BankaHesapCari.Aciklama,
             BankaHesapCari.BorcAlacak, BankaHesapCari.Sinif, BankaHesapCari.Meblag, BankaHesapCari.CariHareketTipiId, BankaCariHareketTipi.Adi, DovizKuru1, DovizKuru2, BankaHesapCari.Notu,
             BankaHesapCari.DegisimTarihi, Kullanici.Adi, CSReferansId, STReferansId, KSReferansId, BHCReferansId, BankaHesapCari.IslemDovizMeblag, BankaHesapCari.IslemDovizID, BankaHesapCari.IslemDovizKuru1, BankaHesapCari.IslemDovizKuru2,
             BankaHesapGrup.Adi, BG2.Adi from BankaHesapCari
  left join BankaHesap on BankaHesap.BankaHesapId = BankaHesapCari.BankaHesapId
  left join BankaHesapGrup on BankaHesapGrup.BankaHesapGrupId = BankaHesap.BankaHesapGrup1
  left join BankaHesapGrup BG2 on BG2.BankaHesapGrupId = BankaHesap.BankaHesapGrup2
  left join Firma on Firma.FirmaId = BankaHesapCari.FirmaId
  left join Kullanici on Kullanici.KullaniciId = BankaHesapCari.KullaniciId
  left join BankaCariHareketTipi on BankaCariHareketTipi.BankaCariHareketTipiID = BankaHesapCari.CariHareketTipiId
  where BankaHesapCari.BankaHesapId = :ABankaHesapId and BankaHesapCari.Tarih >= :ABaslangicTarihi and BankaHesapCari.Tarih <= :ABitisTarihi and (BankaHesapCari.Sinif = :ASinif or :ASinif is null)
  and (BankaHesap.DovizId = :ADovizId or :ADovizId is null)
  and (BankaHesap.BankaHesapGrup1 = :AGrup1Id or :AGrup1Id is null) and (BankaHesap.BankaHesapGrup2 = :AGrup2Id or :AGrup2Id is null) and (BankaHesapCari.CariHareketTipiId = :ABankaCariHareketTipiId or :ABankaCariHareketTipiId is null) order by BankaHesapCari.Tarih, BankaHesapCari.BankaHesapCariId
  into :BankaHesapCariId, :BankaHesapId, :FirmaId, :FirmaAdi, :FirmaAdi2, :FirmaDovizId, :Tarih, :Aciklama, :BorcAlacak, :Sinif, :Meblag, :CariHareketTipiId, :BankaCariHareketTipiAdi,
       :DovizKuru1, :DovizKuru2, :Notu, :DegisimTarihi, :KullaniciAdi, :CsReferansId, :StReferansId, :KsReferansId, :BHCReferansId, :IslemDovizMeblag, :IslemDovizID, :IslemDovizKuru1, :IslemDovizKuru2,
       :BankaHesapGrup1Adi, :BankaHesapGrup2Adi do
  begin
    Borc = Null;
    Alacak = Null;
    BorcBakiye = Null;
    AlacakBakiye = Null;
    ReferansYeri = Null;
    RaporDovizMeblag = Null;
    RaporDovizBorc = Null;
    RaporDovizAlacak = Null;
    RaporDovizBorcBakiye = Null;
    RaporDovizAlacakBakiye = Null;
    RaporDovizKuru = Null;

    if (CsReferansId is not null) then
    begin
      ReferansYeri = 1;
      ReferansId = CsReferansId;
    end
    else if (StReferansId is not null) then
    begin
      ReferansYeri = 2;
      ReferansId = StReferansId;
    end
    else if (KsReferansId is not null) then
    begin
      ReferansYeri = 3;
      ReferansId = KsReferansId;
    end
    else if (BhcReferansId is not null) then
    begin
      ReferansYeri = 4;
      ReferansId = BhcReferansId;
    end

    if (BorcAlacak = 'B') then
    begin
      Borc = Meblag;
      ToplamBorc = ToplamBorc + Borc;
    end
    else
    begin
      Alacak = Meblag;
      ToplamAlacak = ToplamAlacak + Alacak;
    end

    if (toplamBorc is Null) then ToplamBorc = 0;
    if (ToplamAlacak is Null) then ToplamAlacak = 0;
    if (ToplamBorc > ToplamAlacak) then BorcBakiye = ToplamBorc - ToplamAlacak;
    else AlacakBakiye = ToplamAlacak - ToplamBorc;
    if (BorcBakiye = 0) then BorcBakiye = null;
    if (AlacakBakiye = 0) then AlacakBakiye = null;

    if (BankaHesapDovizID <> IslemDovizId) then
    begin
      DovizKuruStr = Format_Float('', IslemDovizKuru1) || ' ' || BankaHesapDovizID || '-' || Format_Float('', ISlemDovizKuru2) || ' ' || IslemDovizId;
    end
    else DovizKuruStr = '';

    if (BankaHesapDovizId <> ARaporDovizId) then
        select DovizKuru from EnYakinDovizKuruGetir(:BankaHesapDovizId, :ARaporDovizId, :Tarih) into :RaporDovizKuru;
    else
        RaporDovizKuru = 1;

    RaporDovizMeblag = Meblag * RaporDovizKuru;
    RaporDovizBorc = Borc * RaporDovizKuru;
    RaporDovizAlacak = Alacak * RaporDovizKuru;
    RaporDovizBorcBakiye = BorcBakiye * RaporDovizKuru;
    RaporDovizAlacakBakiye = AlacakBakiye * RaporDovizKuru;

    suspend;
  end
end
