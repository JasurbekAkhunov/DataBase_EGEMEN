CREATE PROCEDURE KASADOKUMU(
    AKASAID VARCHAR(10),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1))
RETURNS (
    KASAHAREKETID INTEGER,
    KASAID VARCHAR(10),
    TARIH TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    BANKAHESAPID VARCHAR(20),
    BANKAHESAPADI VARCHAR(50),
    ACIKLAMA VARCHAR(50),
    KASAHARCAMATIPIID VARCHAR(10),
    BORCALACAK VARCHAR(1),
    SINIF VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    DOVIZKURUSTR VARCHAR(40),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    KASAHARCAMATIPIADI VARCHAR(30))
AS
declare variable ATarih         TimeStamp;
declare variable KayitVar       integer;
begin
  For Select KasaId from Kasa where Kasa.KasaID = :AKasaID or :AKasaID is null into :KasaId do
  begin
    KayitVar = 0;

    For Select Tarih from KasaHareket
        where KasaHareket.KasaId = :KasaId and KasaHareket.Tarih >= :ABAslangicTarihi and KasaHareket.Tarih <= :ABitisTarihi and (KasaHareket.Sinif = :ASinif or :Asinif is null)
        group by Tarih  into :ATarih do
    begin
      KayitVar = 1;
      for Select KasaHareketId, Tarih, FirmaId, FirmaAdi, FirmaAdi2, BankaHesapId, BankaHesapAdi, Aciklama, KasaHarcamaTipiId, BorcAlacak,
             Sinif, Meblag, DovizKuruStr, Notu, KullaniciAdi, DegisimTarihi, Borc, Alacak, KasaHarcamaTipiAdi from KasaHareketigetir(:KasaId, :ATarih, :ATarih, :ASinif)
        into :KasaHareketId, :Tarih, :FirmaId, :FirmaAdi, :FirmaAdi2, :BankaHesapId, :BankaHesapAdi, :Aciklama, :KasaHarcamaTipiId, :BorcAlacak,
             :Sinif, :Meblag, :DovizKuruStr, :Notu, :KullaniciAdi, :DegisimTarihi, :Borc, :Alacak, :KasaHarcamaTipiAdi do
      begin
        if (Tarih is null) then Tarih = :ATarih;
        suspend;
      end
    end

    if (KayitVar = 0) then
    begin
      ATarih = ABitisTarihi;

      for Select KasaHareketId, Tarih, FirmaId, FirmaAdi, FirmaAdi2, BankaHesapId, BankaHesapAdi, Aciklama, KasaHarcamaTipiId, BorcAlacak,
             Sinif, Meblag, DovizKuruStr, Notu, KullaniciAdi, DegisimTarihi, Borc, Alacak, KasaHarcamaTipiAdi from KasaHareketigetir(:KasaId, :ATarih, :ATarih, :ASinif)
        into :KasaHareketId, :Tarih, :FirmaId, :FirmaAdi, :FirmaAdi2, :BankaHesapId, :BankaHesapAdi, :Aciklama, :KasaHarcamaTipiId, :BorcAlacak,
             :Sinif, :Meblag, :DovizKuruStr, :Notu, :KullaniciAdi, :DegisimTarihi, :Borc, :Alacak, :KasaHarcamaTipiAdi do
      begin
        if (Tarih is null) then Tarih = :ATarih;
        suspend;
      end
    end
  end
end
