    /*PartiSinifId = Null;*/

      SiparisBirimFiyat = null;

    AtisNoHaricAsamaReceteId = AsamaReceteId;

    begin

        AtisNoHaricReceteId = EgemenCopyStr(ReceteId, 1, NoktaPos -1);

    begin
      NoktaPos = PosBul(AsamaReceteId, '.');

    Select Personel.Adi from Personel where Personel.PersonelID = :CikisPersonelID into :CikisPersonelAdi;

         if (T1 < 0) then T1 = T1 * -1;
