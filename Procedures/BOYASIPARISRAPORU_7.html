CREATE PROCEDURE BOYASIPARISRAPORU_7(
    AIMALATTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AFIRMAYETKILIID INTEGER,
    AISLEMTIPIID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AMUSTERIKALITENO VARCHAR(20),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AHAMID INTEGER,
    AKULLANICIID INTEGER,
    AKALITEID VARCHAR(20),
    ATALIMATKODU VARCHAR(20),
    AMAKINAID VARCHAR(10),
    ATEKRARURETIM INTEGER,
    ADESENID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AMARKALI INTEGER,
    AKOMISYONLU INTEGER,
    ALAMINASYON INTEGER,
    ARECETETURUID INTEGER,
    ABATCHNO VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ABASKITIPIID INTEGER,
    AHAMGRUP1ID VARCHAR(4),
    AHAMGRUP2ID VARCHAR(4),
    AHAMGRUP3ID VARCHAR(4),
    AHAMGRUP4ID VARCHAR(4),
    ASIPARISRECETEDURUMU INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    IRSALIYETIPI VARCHAR(22),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    UNITEIRSALIYEID INTEGER,
    MAMULDEPOTRANSFERFISIID INTEGER,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    NEREDENUNITEID INTEGER,
    NEREDENUNITEADI VARCHAR(30),
    NEREYEUNITEID INTEGER,
    NEREYEUNITEADI VARCHAR(30),
    KAPADEDI INTEGER,
    AMBALAJLIBRUTKILO NUMERIC(12,3),
    KILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    FATURATARIHI TIMESTAMP,
    FATURANO INTEGER,
    YAPILACAKISLEMLER VARCHAR(255),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    TABANDOVIZKURU1 DOUBLE PRECISION,
    TABANDOVIZKURU2 DOUBLE PRECISION,
    PALET VARCHAR(10),
    ACIKLAMA VARCHAR(60))
AS
begin
  For Select DISTINCT BoyaSiparisRaporu_5.PartiSinifID from BoyaSiparisRaporu_5(:AImalatTipiId,:ABaslangicTarihi,:ABitisTarihi,
    :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AFirmaYetkiliId, :AIslemTipiId, :AIslemGrubuId, :ASatisTipi,
    :ABoyaSiparisTipiId, :ALabReceteId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMusteriKaliteNo, :AOnayDurumu, :APartilenen,
    :AKapaliDurumu, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASiparisiAlanKullaniciId,
    :AYurtDisi, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AHamId, :AKullaniciId, :AKaliteId, :ATalimatKodu, :AMakinaId, :ATekrarUretim, :ADesenID, :ATemsilciId, :AMarkali, :AKomisyonlu, :ALaminasyon, :AReceteTuruId, :ABatchNo, :ASertifikaNo, :ABaskiTipiId, :AHamGrup1Id, :AHamGrup2Id, :AHamGrup3Id, :AHamGrup4Id, :ASiparisReceteDurumu)
    into :PartiSinifID do
  begin
    MamulDepoTransferFisiId = null;

    IrsaliyeTipi = 'Üniteler Arası Sevk';
    for Select  UniteIrsaliyeDetay.PartiSinifId, UniteIrsaliye.UniteIrsaliyeId, UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo, UniteIrsaliye.IrsaliyeTarihi, UniteIrsaliye.GonderenUniteId, U1.Adi, UniteIrsaliye.AliciUniteId, U2.Adi,
                 UniteIrsaliyeDetay.KapAdedi, UniteIrsaliyeDetay.AmbalajliBrutKilo, UniteIrsaliyeDetay.Kilo, UniteIrsaliyeDetay.NetKilo, UniteIrsaliyeDetay.Palet, UniteIrsaliyeDetay.Aciklama, PartiStok.BoyaSiparisDetayID, UniteFatura.FaturaTarihi,
                 UniteFatura.FaturaNo, UniteIrsaliyeDetay.BirimFiyat, UniteIrsaliyeDetay.DovizID, UniteIrsaliyeDetay.YapilacakIslemler, PartiAsamalari.AsamaID, Asama.Adi, UniteFatura.TabanDovizKuru1, UniteFatura.TabanDovizKuru2 from UniteIrsaliyeDetay
      left join UniteIrsaliye on UniteIrsaliye.UniteIrsaliyeId = UniteIrsaliyeDetay.UniteIrsaliyeId
      left join UniteFatura on UniteFatura.UniteFaturaId = UniteIrsaliyeDetay.UniteFaturaId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = UniteIrsaliyeDetay.PartiAsamalariId
      left join PartiStok on PartiStok.PartiSinifID = UniteIrsaliyeDetay.PartiSinifID
      left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
      left join Unite U1 on U1.UniteId = UniteIrsaliye.GonderenUniteId
      left join Unite U2 on U2.UniteId = UniteIrsaliye.AliciUniteId
                         where UniteIrsaliyeDetay.PartiSinifID = :PartiSinifID
      into :PartiSinifId, :UniteIrsaliyeId, :IrsaliyeSeriNo, :IrsaliyeNo, :Irsaliyetarihi, :NeredenUniteId, :NeredenUniteAdi, :NereyeUniteId, :NereyeUniteAdi,
           :KapAdedi, :AmbalajliBrutKilo, :Kilo, :NetKilo, :Palet, :Aciklama, :BoyaSiparisDetayID,
           :FaturaTarihi, :FaturaNo, :BirimFiyat, :DovizID, :YapilacakIslemler, :AsamaID, :AsamaAdi, :TabanDovizKuru1, :TabanDovizKuru2 do suspend;

    UniteIrsaliyeId = Null;

    Palet = '';
    Aciklama = '';
    AmbalajliBrutKilo = Null;
    FaturaTarihi = Null;
    FaturaNo     = Null;
    BoyaSiparisDetayID = Null;
    YapilacakIslemler  = Null;
    AsamaID            = Null;
    AsamaAdi           = Null;
    TabanDovizKuru1 = Null;
    TabanDovizKuru2 = Null;

    IrsaliyeTipi = 'Mamul Depo Transferi';
    for Select MamulDepoTransferFisiDetay.PartiSinifId, MamulDepoTransferFisi.MamulDepoTransferFisiId, MamulDepoTransferFisi.IrsaliyeSeriNo,
               MamulDepoTransferFisi.IrsaliyeNo, MamulDepoTransferFisi.IrsaliyeTarihi,
               MamulDepoTransferFisi.FisNo, MamulDepoTransferFisi.FisTarihi, D1.UniteId, U1.Adi, D2.UniteId, U2.Adi,
               Sum(PartiStokTarti.KapAdedi), Sum(PartiStokTarti.Kilo + Dara), Sum(PartiStokTarti.Kilo) from PartiStok
      left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.PartiSinifId = PartiStok.PartiSinifId
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiID = MamulDepoTransferFisiDetay.PartiStokTartiID
      left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiId = MamulDepoTransferFisiDetay.MamulDepoTransferFisiId
      left join Depo D1 on D1.DepoId = MamulDepoTransferFisi.DepoId
      left join Depo D2 on D2.DepoId = MamulDepoTransferFisi.TransferDepoId
      left join Unite U1 on U1.UniteId = D1.UniteId
      left join Unite U2 on U2.UniteId = D2.UniteId
      where PartiStok.PartiSinifID = :PartiSinifID and MamulDepoTransferFisiDetay.PartiSinifId is not null and MamulDepoTransferFisi.SistemKaydi = 0
      group by MamulDepoTransferFisiDetay.PartiSinifId, MamulDepoTransferFisi.MamulDepoTransferFisiId, MamulDepoTransferFisi.IrsaliyeSeriNo,
               MamulDepoTransferFisi.IrsaliyeNo, MamulDepoTransferFisi.IrsaliyeTarihi,
               MamulDepoTransferFisi.FisNo, MamulDepoTransferFisi.FisTarihi, D1.UniteId, U1.Adi, D2.UniteId, U2.Adi
        into :PartiSinifId, :MamulDepoTransferFisiId, :IrsaliyeSeriNo, :IrsaliyeNo, :Irsaliyetarihi, :FisNo, :FisTarihi, :NeredenUniteId, :NeredenUniteAdi,
             :NereyeUniteId, :NereyeUniteAdi, :KapAdedi, :Kilo, :NetKilo do suspend;

  end
end
