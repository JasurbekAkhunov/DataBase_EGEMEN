CREATE PROCEDURE TESLATOPBILGISIKAYDET(
    APSKALITEKONTROLASAMAID INTEGER,
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    APAKETLEMEID INTEGER,
    AOTOMASYONMAKINANO VARCHAR(10),
    AKILO NUMERIC(12,3),
    AMETRE NUMERIC(12,3),
    ABRUTMETRE NUMERIC(12,3),
    AKULLANICIID INTEGER)
RETURNS (
    PSKALITEKONTROLID INTEGER,
    PARTISTOKTARTIID INTEGER,
    STATUS INTEGER,
    MESAJ VARCHAR(200))
AS
declare variable MakinaId          varchar(10);
begin
  Select Max(MakinaId) from Makina where PaketlemeMakinasimi = 1 and OtomasyonMakinaNo = :AOtomasyonMakinaNo into :MakinaId;

  if (MakinaID is null) then
  begin
    Status = 0;
    Mesaj = AOtomasyonMakinaNo || ' nolu Makina''nın Egemen de karşılıgı bulunamadı.';
    suspend;
    exit;
  end

  Select PsKaliteKontrolId, PSKaliteKontrolId from PsKaliteKontrol
    where PsKaliteKontrol.PartiKayitId = :APartiKayitId and
          PsKaliteKontrol.PartiSinifId = :APartiSinifId and
          PsKaliteKontrol.MakinaId = :MakinaId and
          PsKaliteKontrol.PaketlemeId = :APaketlemeId and
          (PsKaliteKontrol.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID or PsKaliteKontrol.PSKaliteKontrolAsamaID is Null)
    into :PsKaliteKontrolId, :PSKaliteKontrolId;

  if (PsKaliteKontrolId is not null) then
  begin
    Status = 2;
    suspend;
    exit;
  end

  PSKaliteKontrolId = gen_Id(PSKaliteKontrol_Id_Gen, 1);

  insert into PSKaliteKontrol(PSKaliteKontrolId, PSKaliteKontrolAsamaID, Tarih, PartiKayitId, PartiSinifId, MakinaId, Kilo, Metre, BrutMetre, PaketlemeId, InsertKullaniciId, KullaniciId)
            values(:PSKaliteKontrolId, :APSKaliteKontrolAsamaID, current_TimeStamp(0), :APartiKayitId, :APartiSinifId, :MakinaId, :AKilo, :AMetre, :ABrutMetre, :APaketlemeId,  :AKullaniciId, :AKullaniciId);

  Status = 1;
  suspend;
end
