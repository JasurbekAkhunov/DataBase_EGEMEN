CREATE PROCEDURE CTARIHINEGORECSRAPORUTOPLAMI(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ACS VARCHAR(1),
    ACIROFIRMAID VARCHAR(20),
    ABASLANGICCSGRUP1ID VARCHAR(4),
    ABITISCSGRUP1ID VARCHAR(4),
    ABASLANGICCSGRUP2ID VARCHAR(4),
    ABITISCSGRUP2ID VARCHAR(4),
    ABASLANGICCSGRUP3ID VARCHAR(4),
    ABITISCSGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    ACSGRUP1ARALIKLI INTEGER,
    ACSGRUP2ARALIKLI INTEGER,
    ACSGRUP3ARALIKLI INTEGER,
    ADOVIZID VARCHAR(5),
    ARAPORDOVIZID VARCHAR(5),
    ASINIF VARCHAR(1))
RETURNS (
    TAHSILEDILENLERHEPSI DOUBLE PRECISION,
    TAHSILEDILMEMISLERPORTFOYDE DOUBLE PRECISION,
    CIROEDILENLER DOUBLE PRECISION,
    BANKAYAVERILENLERHEPSI DOUBLE PRECISION,
    BANKADAKITAHSILEDILMEMISLER DOUBLE PRECISION,
    BANKADANTAHSILEDILENLER DOUBLE PRECISION,
    PROTESTOOLANLARHEPSI DOUBLE PRECISION,
    BANKADANPROTESTOOLANLAR DOUBLE PRECISION,
    AVUKATAVERILENLERHEPSI DOUBLE PRECISION,
    AVUKATTATAHSILEDILMEYENLER DOUBLE PRECISION,
    AVUKATTANTAHSILOLANLAR DOUBLE PRECISION,
    IADEEDILENLER DOUBLE PRECISION)
AS
declare variable Meblag Double Precision;
declare variable TahsilEdilenMeblag Double Precision;
declare variable SonBordroTipi integer;
declare variable Karsiliksiz integer;
declare variable TahsilEdilmis integer;
begin
  TahsilEdilenlerHepsi          = 0;
  TahsilEdilmemislerPortfoyde   = 0;
  CiroEdilenler                 = 0;
  BankayaVerilenlerHepsi        = 0;
  BankadakiTahsilEdilmemisler   = 0;
  BankadanTahsilEdilenler       = 0;
  ProtestoOlanlarHepsi          = 0;
  BankadanProtestoOlanlar       = 0;
  AvukataVerilenlerHepsi        = 0;
  AvukattaTahsilEdilmeyenler    = 0;
  AvukattanTahsilOlanlar        = 0;
  IadeEdilenler                 = 0;
  TahsilEdilenMeblag = 0;
  For select karsiliksiz, SonBordroTipi, TahsilEdilmis, Sum(DovizMeblag) from
           CiroTarihineGoreCSRaporu(:ABaslangicFirmaId, :ABitisFirmaId, :ABaslangicTarihi, :ABitisTarihi, :ACs,
           :ACiroFirmaId, :ABaslangicCsGrup1Id, :ABitisCsGrup1Id,
           :ABaslangicCsGrup2Id, :ABitisCsGrup2Id, :ABaslangicCsGrup3Id, :ABitisCsGrup3Id, :AFirmaAralikli, :ACSGrup1Aralikli,
           :ACSGrup2Aralikli, :ACSGrup3Aralikli, :ADovizId, :ARaporDovizId, :ASinif)
    group by karsiliksiz, SonBordroTipi, TahsilEdilmis
    into :Karsiliksiz, :SonBordroTipi, :TahsilEdilmis, :Meblag do
  begin
    TahsilEdilenlerHepsi = TahsilEdilenlerHepsi + TahsilEdilenMeblag;

    if (SonBordroTipi = 0 or SonBordroTipi = 2 or SonBordroTipi = 4 or SonBordroTipi = 6) then TahsilEdilmemislerPortfoyde = TahsilEdilmemislerPortfoyde + Meblag;

    if (SonBordroTipi = 1) then CiroEdilenler = CiroEdilenler + meblag;

    if (SonBordroTipi = 3) then
    begin
      BankayaVerilenlerHepsi = BankayaVerilenlerHepsi + Meblag;
      BankadakiTahsilEdilmemisler = BankadakiTahsilEdilmemisler + (Meblag - TahsilEdilenMeblag);
      BankadanTahsilEdilenler = BankadanTahsilEdilenler + TahsilEdilenMeblag;
      if (Karsiliksiz = 1) then BankadanProtestoOlanlar = BankadanProtestoOlanlar + Meblag;
    end
    if (Karsiliksiz = 1) then ProtestoOlanlarHepsi = ProtestoOlanlarHepsi + Meblag;
    if (SonBordroTipi = 5) then
    begin
      AvukataVerilenlerHepsi = AvukataVerilenlerHepsi + meblag;
      AvukattaTahsilEdilmeyenler = AvukattaTahsilEdilmeyenler + Meblag - TahsilEdilenMeblag;
      AvukattanTahsilOlanlar = AvukattanTahsilOlanlar + TahsilEdilenMeblag;
    end
    if (SonBordroTipi = 7 or SonBordroTipi = 7 ) then IadeEdilenler = IadeEdilenler + Meblag;
  end
  suspend;
end
