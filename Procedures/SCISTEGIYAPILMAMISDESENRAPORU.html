CREATE PROCEDURE SCISTEGIYAPILMAMISDESENRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    DESENID VARCHAR(10),
    DESENADI VARCHAR(80),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DESENGRUPID VARCHAR(4),
    MUSTERIDESENNO1 VARCHAR(20),
    MUSTERIDESENADI1 VARCHAR(80),
    MUSTERIDESENNO2 VARCHAR(20),
    MUSTERIDESENADI2 VARCHAR(80))
AS
declare variable DesenIstekTipiId varchar(5);
begin
  For Select Desen.DesenId, Desen.Adi, Desen.FirmaId, Firma.Adi, Desen.DesenGrupId,
             Desen.MusteriDesenNo1, Desen.MusteriDesenAdi1, Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2,
             Max(DesenIstekTipi.DesenIstekTipiId)  from Desen
    Left Join Firma on Firma.FirmaId = Desen.FirmaId
    Left Join DesenIstekleri on DesenIstekleri.DesenId = Desen.DesenId
    Left Join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiID and DesenIstekTipi.SablonCekimmi = 1
    where Desen.TasarimOnayDurumu = 0 and
         (Desen.InsertTarihi >= :ABaslangicTarihi and Desen.InsertTarihi <= :ABitisTarihi) and
         (Desen.FirmaId = :AFirmaId or :AFirmaId is null)
  group by Desen.DesenId, Desen.Adi, Desen.FirmaId, Firma.Adi,  Desen.DesenGrupId,
           Desen.MusteriDesenNo1, Desen.MusteriDesenAdi1, Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2
  into :DesenId, :DesenAdi, :FirmaId, :FirmaAdi, :DesenGrupId,
       :MusteriDesenNo1, :MusteriDesenAdi1, :MusteriDesenNo2, :MusteriDesenAdi2, :DesenIstekTipiId do
  begin
    if (DesenIstekTipiId is null) then
    begin
      suspend;
    end
  end
end
