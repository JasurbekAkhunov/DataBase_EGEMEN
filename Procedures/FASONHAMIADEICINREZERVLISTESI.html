CREATE PROCEDURE FASONHAMIADEICINREZERVLISTESI(
    AFIRMAID VARCHAR(20))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMSTOKMASTERID INTEGER,
    RAFDETAYID INTEGER,
    HAMIADEREZERVID INTEGER,
    SINIF VARCHAR(1),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(10),
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    RAFID VARCHAR(10),
    IADEEDILECEKKAPADEDI INTEGER,
    IADEEDILECEKKILO NUMERIC(12,3),
    IADEEDILECEKMETRE NUMERIC(12,3),
    IADEEDILECEKADET INTEGER,
    IADEEDILENKAPADEDI INTEGER,
    IADEEDILENKILO NUMERIC(12,3),
    IADEEDILENMETRE NUMERIC(12,3),
    IADEEDILENADET INTEGER)
AS
declare variable OrmeFirmaKodumuz varchar(20);
begin
  Select OrmeFirmaKodumuz from Sistem where sistemid = 1 into :OrmeFirmaKodumuz;

  if (AFirmaID = OrmeFirmaKodumuz) then
  begin
    for select hamstokdetay.HamStokMasterId, HamIadeRezerv.RafDetayId, HamIadeRezerv.HamIadeRezervId, HamStokIrsaliye.Sinif, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
               Ham.Adi, Ham.HamTipiId, RafDetay.RafId, HamIadeRezerv.KapAdedi - HamIadeRezerv.IadeEdilenKapAdedi,
               HamIadeRezerv.Kilo - HamIadeRezerv.IadeEdilenKilo, HamIadeRezerv.Metre - HamIadeRezerv.IadeEdilenMetre,
               HamIadeRezerv.Adet - HamIadeRezerv.IadeEdilenAdet,
               HamIadeRezerv.IadeEdilenKapAdedi, HamIadeRezerv.IadeEdilenKilo, HamIadeRezerv.IadeEdilenMetre,
               HamIadeRezerv.IadeEdilenAdet, HamIadeRezerv.FirmaId, Firma.Adi from HamIadeRezerv
      left join Rafdetay on RafDetay.RafDetayId = HamIadeRezerv.RafDetayId
      left join hamstokdetay on hamstokdetay.hamstokDetayId = RafDetay.HamStokDetayid
      left join HamstokIrsaliye on HamstokIrsaliye.HamstokIrsaliyeid = HamStokDetay.HamstokIrsaliyeid
      left join Ham on Ham.HamId = RafDetay.HamId
      left join Firma on Firma.FirmaId = HamIadeRezerv.FirmaId
      where HamstokDetay.Fason = 1 and HamIadeRezerv.Miktar > HamIadeRezerv.IadeEdilenMiktar
      into :HamStokMasterId,:RafDetayId, :HamIadeRezervId, :Sinif, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi, :HamAdi, :HamTipiId, :RafId, :IadeEdilecekKapAdedi,
           :IadeEdilecekKilo, :IadeEdilecekMetre, :IadeEdilecekAdet, :IadeEdilenKapAdedi, :IadeEdilenKilo, :IadeEdilenMetre,
           :IadeEdilenAdet, :FirmaId, :FirmaAdi do
    begin
      if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
      else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
      else EvrakTipiStr = 'Hatalı';

      suspend;
    end

  end
  else
  begin
    for select hamstokdetay.HamStokMasterId, HamIadeRezerv.RafDetayId, HamIadeRezerv.HamIadeRezervId, HamStokIrsaliye.Sinif, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
               Ham.Adi, Ham.HamTipiId, RafDetay.RafId, HamIadeRezerv.KapAdedi - HamIadeRezerv.IadeEdilenKapAdedi,
               HamIadeRezerv.Kilo - HamIadeRezerv.IadeEdilenKilo, HamIadeRezerv.Metre - HamIadeRezerv.IadeEdilenMetre,
               HamIadeRezerv.Adet - HamIadeRezerv.IadeEdilenAdet,
               HamIadeRezerv.IadeEdilenKapAdedi, HamIadeRezerv.IadeEdilenKilo, HamIadeRezerv.IadeEdilenMetre,
               HamIadeRezerv.IadeEdilenAdet, HamIadeRezerv.FirmaId, Firma.Adi from HamIadeRezerv
      left join Rafdetay on RafDetay.RafDetayId = HamIadeRezerv.RafDetayId
      left join hamstokdetay on hamstokdetay.hamstokDetayId = RafDetay.HamStokDetayid
      left join HamstokIrsaliye on HamstokIrsaliye.HamstokIrsaliyeid = HamStokDetay.HamstokIrsaliyeid
      left join ham on ham.hamid = RafDetay.hamid
      left join Firma on Firma.FirmaId = HamIadeRezerv.FirmaId
      where HamIadeRezerv.FirmaId = :AFirmaId and HamstokDetay.Fason = 1 and HamIadeRezerv.Miktar > HamIadeRezerv.IadeEdilenMiktar
      into :HamStokMasterId, :RafDetayId, :HamIadeRezervId, :Sinif, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi, :HamAdi, :HamTipiId, :RafId, :IadeEdilecekKapAdedi,
           :IadeEdilecekKilo, :IadeEdilecekMetre, :IadeEdilecekAdet, :IadeEdilenKapAdedi, :IadeEdilenKilo, :IadeEdilenMetre,
           :IadeEdilenAdet, :FirmaId, :FirmaAdi do
    begin
      if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
      else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
      else EvrakTipiStr = 'Hatalı';

      suspend;
    end
  end
end
