CREATE PROCEDURE CARIVIRMANGETIR(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1))
RETURNS (
    CARIVIRMANID INTEGER,
    BORCLUFIRMAID VARCHAR(20),
    BORCLUFIRMAADI VARCHAR(100),
    ALACAKLIFIRMAID VARCHAR(20),
    ALACAKLIFIRMAADI VARCHAR(100),
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    CARITIPIADI VARCHAR(20),
    SINIF VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    BORCLUDOVIZID VARCHAR(5),
    ALACAKLIDOVIZID VARCHAR(5),
    DOVIZKURUSTR VARCHAR(40),
    VALORTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable DovizKuru1 Double Precision;
declare variable DovizKuru2 Double Precision;
begin
  for select CariVirman.CariVirmanId, CariVirman.BorcluFirmaId, BorcluFirma.Adi, CariVirman.AlacakliFirmaId, AlacakliFirma.Adi,
             CariVirman.Tarih, CariVirman.Aciklama, CariTipi.Adi, CariVirman.Sinif, CariVirman.Meblag,
             BorcluFirma.DovizId, AlacakliFirma.DovizId, CariVirman.DovizKuru1, CariVirman.DovizKuru2, CariVirman.ValorTarihi, CariVirman.Notu,
             Ik.Adi, CariVirman.InsertTarihi, SK.Adi, CariVirman.DegisimTarihi from CariVirman
    left join Firma BorcluFirma on BorcluFirma.FirmaId = CariVirman.BorcluFirmaId
    left join Firma AlacakliFirma on AlacakliFirma.FirmaId = CariVirman.AlacakliFirmaId
    left join CariTipi on CariTipi.CariTipiId = CariVirman.CariTipiId
    left join Kullanici IK on IK.KullaniciId = CariVirman.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId = CariVirman.KullaniciId
    where CariVirman.Tarih >= :ABaslangicTarihi and CariVirman.Tarih <= :ABitisTarihi and (CariVirman.Sinif = :ASinif or :ASinif is null)
    order by CariVirman.Tarih, CariVirman.CariVirmanId
    into :CariVirmanId, :BorcluFirmaId, :BorcluFirmaAdi, :AlacakliFirmaId, :AlacakliFirmaAdi, :Tarih, :Aciklama, :CariTipiAdi,
         :Sinif, :Meblag, :BorcluDovizId, :AlacakliDovizId, :DovizKuru1, :DovizKuru2, :ValorTarihi, :Notu,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    if (BorcluDovizId <> AlacakliDovizId) then
    begin
      DovizKuruStr = Format_Float('', DovizKuru1) || ' ' ||  BorcluDovizId || '-' || Format_Float('', DovizKuru2) || ' ' || AlacakliDovizId;
    end
    else DovizKuruStr = '';


    suspend;
  end
end
