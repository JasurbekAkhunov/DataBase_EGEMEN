CREATE PROCEDURE ANALOTNUMARATORGETIR(
    AHARMANCINSIID INTEGER,
    AHARMANKARISIMID INTEGER,
    AISLETMEID VARCHAR(1))
RETURNS (
    ANALOTID VARCHAR(50))
AS
Declare Variable IkiHaneYil  Varchar(2);
Declare Variable HCinsiNumarator integer;
Declare Variable HKarisimNumarator integer;
Declare Variable HarmanCinsiKisaKod    varchar(2);
Declare Variable HarmanKarisimKisaKod  varchar(2);
begin
  IkiHaneYil = EgemenCopyStr(TimeStampToYil(Current_TimeStamp), 3, 2);


  Select HarmanCinsi.KisaKod, AnaLotNumarator.ID + 1 from HarmanCinsi
    left Join AnaLotNumarator on AnaLotNumarator.KisaKod = HarmanCinsi.KisaKod and AnaLotNumarator.Bolum = 1
    where HarmanCinsi.HarmanCinsiID = :AHarmanCinsiID
  into :HarmanCinsiKisaKod, :HCinsiNumarator;

  if (AHarmanKarisimID is Not Null ) then
  begin
    Select HarmanKarisim.KisaKod, AnaLotNumarator.ID + 1 from HarmanKarisim
      left Join AnaLotNumarator on AnaLotNumarator.KisaKod = HarmanKarisim.KisaKod and AnaLotNumarator.Bolum = 2
      where HarmanKarisim.HarmanKarisimID = :AHarmanKarisimID
    into :HarmanKarisimKisaKod, :HKarisimNumarator;
  end
  else
  begin
    HarmanKarisimKisaKod = '00';
    HKarisimNumarator = 0;
  end

  AnaLotID = IkiHaneYil || EgemenIntegerToStr(AIsletmeID) || '-' || HarmanCinsiKisaKod || EgemenIntToLzStr(HCinsiNumarator, 3) || HarmanKarisimKisaKod || EgemenIntToLzStr(HKarisimNumarator, 3) ;

  Update AnaLotNumarator set ID =  :HCinsiNumarator where Bolum = 1 and KisaKod = :HarmanCinsiKisaKod;
  Update AnaLotNumarator set ID =  :HKarisimNumarator where Bolum = 2 and KisaKod = :HarmanKarisimKisaKod;

  Suspend;
end
