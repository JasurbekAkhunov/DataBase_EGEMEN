CREATE PROCEDURE HSSARFFISIICINHAMSTOKLISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AHAMID INTEGER,
    AHAMSTOKMASTERID INTEGER,
    ARAFID VARCHAR(10))
RETURNS (
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    RAFDETAYID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    LOT VARCHAR(150),
    RAFID VARCHAR(10),
    MIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    KAPADEDI INTEGER,
    KALANKAPADEDI INTEGER)
AS
declare variable TamirX integer;
declare variable HamStokDetayKalanMiktar   numeric(12,3);
declare variable LastHamStokDetayId        integer;
declare variable KullanilanMiktar          Numeric(12,3);
declare variable KullanilanKapAdedi        integer;
declare variable KapAdediSatis             integer;
declare variable RezervKapAdedi integer;
begin
  if (AHamStokMasterId is not null) then
  begin
    Select HamId from HamStokMaster where HamStokMaster.HamStokMasterID = :AHamStokMasterID into :AHamID;
    if (AHamID is null) then AHamID = -1;
  end

  LastHamStokDetayId = -1;
  for select Hamstokdetay.HamStokMasterId, RafDetay.HamStokDetayId, RafDetay.RafDetayId, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
         HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.Fason,
         HamStokdetay.HamId, Ham.Adi, Ham.HamTipiId, HamStokMaster.Lot, RafDetay.RafId, RafDetay.Miktar, RafDetay.KapAdedi,
         (RafDetay.IadeEdilecekMiktar + RafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.RafTransferMiktar + RafDetay.SatisMiktar + RafDetay.AktarmayaCikanKilo - RafDetay.AktarmadanIadeEdienKilo),
         (RafDetay.IadeEdilecekKapAdedi + RafDetay.IadeEdilenKapAdedi + RafDetay.PartilenenKapAdedi) from HamStokDetay
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeid = HamStokDetay.HamStokIrsaliyeId
  left join RafDetay on RafDetay.HamStokDetayid = HamStokDetay.HamStokDetayid
  left join Ham on Ham.HamId = hamstokdetay.hamid
  left join Firma on Firma.Firmaid = HamStokIrsaliye.FirmaId
  where (HamStokDetay.HamId = :AHamId) and RafDetay.RafDetayID is not null and
       ((:AIrsaliyeNo is not null and HamstokIrsaliye.IrsaliyeNo = :AIrsaliyeNo) or :AIrsaliyeNo is null) and
       (HamStokDetay.HamStokMasterID = :AHamStokMasterID or :AHamStokMasterID is null) and
       (RafDetay.RafId = :ARafId or :ARafId is null) and
       ( (:AIrsaliyeNo is null And :AHamStokMasterID is null and HamstokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamstokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi ) or
          :AIrsaliyeNo is not null or :AHamStokMasterID is not null)
       order by RafDetay.HamStokDetayId, RafDetay.RafDetayId
    into :HamStokMasterId, :HamStokDetayId, :RafDetayId, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi, :FirmaId, :FirmaAdi, :Fason, :HamId, :HamAdi, :HamTipiId, :Lot, :RafId,
         :Miktar, :KapAdedi, :KullanilanMiktar, :KullanilanKapAdedi do
  begin
    select Sum(BoyaSiparisRezerv.KullanilanKapAdedi) from BoyaSiparisRezerv
      where BoyaSiparisRezerv.HamStokDetayID = :HamStokDetayId into :RezervKapAdedi;

    select Sum(BoyaIrsaliyeDetay.KapAdedi) from BoyaIrsaliyeDetay
      where BoyaIrsaliyeDetay.RafDetayID = :RafDetayID and HareketTipiID = 0 into :KapAdediSatis;

    if (RezervKapAdedi is Null) then RezervKapAdedi = 0;
    if (KullanilanKapAdedi is Null) then KullanilanKapAdedi = 0;
    if (KapAdediSatis is Null) then KapAdediSatis = 0;

    KalanKapAdedi = KapAdedi - KullanilanKapAdedi - KapAdediSatis - RezervKapAdedi;


    if (HamStokDetayID <> LastHamStokDetayID) then
    begin
      select KalanMiktar from HamStokDetayKalaniGetir(:HamStokDetayId) into :HamStokDetayKalanMiktar;
      LastHamStokDetayID = HamStokDetayId;
    end

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (KullanilanMiktar is null) then KullanilanMiktar = 0;

    KalanMiktar = Miktar - KullanilanMiktar;

    if (KalanMiktar > 0 and HamStokDetayKalanMiktar > 0) then
    begin
      if (KalanMiktar > HamStokDetayKalanMiktar) then
      begin
        KalanMiktar = HamStokDetayKalanMiktar;
        HamStokDetayKalanMiktar = 0;
      end
      else HamStokDetayKalanMiktar = HamStokDetayKalanMiktar - KalanMiktar;

      if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
      else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
      else EvrakTipiStr = 'Hatalı';
      suspend;
    end
  end
end
