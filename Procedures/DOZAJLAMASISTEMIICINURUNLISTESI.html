CREATE PROCEDURE DOZAJLAMASISTEMIICINURUNLISTESI(
    ADOZAJLAMASISTEMIID INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    DOZAJLAMAURUNID VARCHAR(15),
    MUTFAKDEPONO VARCHAR(10),
    URUNBIRIMID INTEGER,
    BIRIMCARPAN DOUBLE PRECISION,
    URETIMECIKISFIYATI DOUBLE PRECISION)
AS
declare variable FStokTabanDovizId          varchar(5);
declare variable FUretimeCikisFiyatiDovizId Varchar(5);
declare variable FDovizKuru             Double Precision;
begin
  Select STOKTABANDOVIZID from SISTEM where SISTEMID = 1 into :FStokTabanDovizId;

  for Select DozajlamaSistemiUrun.UrunId, DozajlamaSistemiUrun.DozajlamaUrunId, urun.MutfakDepoNo, UBD.urunBirimId, UBD.Carpan, UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / UBD.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId from DozajlamaSistemiUrun
    left join Urun on Urun.UrunID = DozajlamaSistemiUrun.UrunId
    left join UrunBirim UBD on UBD.UrunId = urun.Urunid and UBD.ReceteDefaultBirim = 1
    left join UrunFiyat on UrunFiyat.UrunId = urun.Urunid
    left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
    where DozajlamaSistemiUrun.DozajlamaSistemiID = :ADozajlamaSistemiID
    into :UrunId, :DozajlamaUrunId, :MutfakDepoNo, :urunBirimId, :BirimCarpan, :UretimeCikisFiyati, :FUretimeCikisFiyatiDovizId do
  begin
    if (FStokTabanDovizId <> FUretimeCikisFiyatiDovizId) then
    begin
      select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :FStokTabanDovizId, Current_Date) into :FDovizKuru;

      UretimeCikisFiyati = UretimeCikisFiyati * FDovizKuru;
    end

    UretimeCikisFiyati = Yuvarla(UretimeCikisFiyati, 1000000);
    suspend;
  end
end
