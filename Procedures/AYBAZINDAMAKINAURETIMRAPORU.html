CREATE PROCEDURE AYBAZINDAMAKINAURETIMRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAGRUBU1 VARCHAR(4))
RETURNS (
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    MAKINAGRUBU VARCHAR(4),
    AY1 NUMERIC(12,3),
    AY2 NUMERIC(12,3),
    AY3 NUMERIC(12,3),
    AY4 NUMERIC(12,3),
    AY5 NUMERIC(12,3),
    AY6 NUMERIC(12,3),
    AY7 NUMERIC(12,3),
    AY8 NUMERIC(12,3),
    AY9 NUMERIC(12,3),
    AY10 NUMERIC(12,3),
    AY11 NUMERIC(12,3),
    AY12 NUMERIC(12,3))
AS
declare variable FMakinaID      varchar(10);
declare variable FMakinaGrubu   varchar(4);
declare variable Ay     integer;
declare variable Miktar   Numeric(12,3);
begin
  MakinaID = '';

  For Select Makina.MakinaID, Makina.MakinaGrubu, Makina.Adi from Makina
    where (Makina.MakinaGrubu = :AMakinaGrubu1 or :AMakinaGrubu1 is null)
    into :MakinaID, :MakinaGrubu, :MakinaAdi do
  begin
    Ay1           = 0;
    Ay2           = 0;
    Ay3           = 0;
    Ay4           = 0;
    Ay5           = 0;
    Ay6           = 0;
    Ay7           = 0;
    Ay8           = 0;
    Ay9           = 0;
    Ay10          = 0;
    Ay11          = 0;
    Ay12          = 0;

    For Select f_Month(PartiAsamaStokDetay.Cikistarihi), Sum(PartiAsamaStokDetay.Miktar) from PartiAsamaStokDetay
      where PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi < :ABitisTarihi and
            PartiAsamaStokDetay.MakinaID = :MakinaId
      group by f_Month(PartiAsamaStokDetay.Cikistarihi)
      into :Ay, :Miktar do
    begin
      if (Ay = 1) then Ay1 = Miktar;
      else if (Ay = 2) then Ay2 = Miktar;
      else if (Ay = 3) then Ay3 = Miktar;
      else if (Ay = 4) then Ay4 = Miktar;
      else if (Ay = 5) then Ay5 = Miktar;
      else if (Ay = 6) then Ay6 = Miktar;
      else if (Ay = 7) then Ay7 = Miktar;
      else if (Ay = 8) then Ay8 = Miktar;
      else if (Ay = 9) then Ay9 = Miktar;
      else if (Ay = 10) then Ay10 = Miktar;
      else if (Ay = 11) then Ay11 = Miktar;
      else if (Ay = 12) then Ay12 = Miktar;
    end
    suspend;
  end
end
