CREATE PROCEDURE IKINCIKALITETOPLUSTOKKAYDIR(
    AORMERAFIDSTR VARCHAR(1025),
    AKULLANICIID INTEGER)
AS
declare variable VirgulPos integer;
declare variable Str varchar(20);
declare variable OrmeRafId integer;
declare variable FisYil integer;
declare variable FisNo integer;
declare variable HamId integer;
declare variable HamTipiId integer;
declare variable IkinciKaliteIplikHamId                integer;
declare variable IkinciKaliteKumasHamId                integer;
declare variable OrmeDefaultIkinciKaliteRafId          varchar(10);
begin
  Select IkinciKaliteIplikHamId, IkinciKaliteKumasHamId, OrmeDefaultIkinciKaliteRafId from Sistem where sistemid = 1 into :IkinciKaliteIplikHamId, :IkinciKaliteKumasHamId, :OrmeDefaultIkinciKaliteRafId;

  VirgulPos = PosBul(AOrmeRafIDStr, ',');
  FisYil = f_Year(Current_date);

  While (VirgulPos > 1) do
  begin
    Str = EgemenCopyStr(AOrmeRafIDStr, 1, VirgulPos -1);
    OrmeRafId = EgemenIntegerToStr(Str);

    SELECT ID FROM OrmeStokKaydirmaNumaratorAl(:FisYil) into :FisNo;

    Select Ham.HamTipiID from OrmeRaf
      left join Ham on Ham.HamID = OrmeRaf.HamId
      where OrmeRaf.OrmeRafID = :OrmeRafId
      into :HamTipiID;

    if (HamTipiId = 0) then HamID = IkinciKaliteIplikHamId;
    else HamID = IkinciKaliteKumasHamId;

    insert into OrmeStokKaydirma(FisYil, FisNo, FisTarihi, OrmeRafId, RafId, HamId, KaliteNo, BrutMiktar, NetMiktar, InsertKullaniciId)
             Select :FisYil, :FisNo, Current_date, :OrmeRafId, :OrmeDefaultIkinciKaliteRafId, :HamId, 2,
                    OrmeRaf.BrutGirisMiktari - OrmeRaf.BrutCikisMiktari, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari, :AKullaniciId from OrmeRaf
               where OrmeRaf.OrmeRafID = :OrmeRafId;


    if (Char_Length(AOrmeRafIDStr) > VirgulPos) then AOrmeRafIDStr = EgemenCopyStr(AOrmeRafIDStr, VirgulPos + 1, Char_Length(AOrmeRafIDStr) - VirgulPos);
    else AOrmeRafIDStr = '';

    VirgulPos = PosBul(AOrmeRafIDStr, ',');
  end
end
