CREATE PROCEDURE PARTISINIFHAMSTOKBILGILERIGETIR(
    APARTISINIFID INTEGER)
RETURNS (
    PUS VARCHAR(255),
    FEINE VARCHAR(255),
    LOT VARCHAR(150),
    ORMECI VARCHAR(255),
    IPLIKUZUNLUGU1 VARCHAR(40),
    IPLIKUZUNLUGU2 VARCHAR(40),
    IPLIKUZUNLUGU3 VARCHAR(40))
AS
declare variable FPus integer;
declare variable FFeine integer;
declare variable FLot varchar(150);
declare variable FOrmeci varchar(30);
begin
  Pus = '';
  Feine = '';
  Lot = '';
  Ormeci = '';
  for select HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Lot, Ormeci.Adi,
        HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3 from PartiStokDetay
    left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
    left join Ormeci on Ormeci.OrmeciId = HamStokIrsaliye.OrmeciId
    where PartiStokDetay.PartiSinifId = :APartiSinifId
    group by HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Lot, Ormeci.Adi,
            HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3
    into :FPus, :FFeine, :FLot, :FOrmeci,
        :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3 do
  begin
    if (Char_Length(Pus) + Char_Length(FPus) < 48) then
    begin
     if (Pus <> '' ) then Pus = Pus || cast(', ' as varchar(2) character set UTF8);
     Pus = Pus || FPus;
    end

    if (Char_Length(Feine) + Char_Length(FFeine) < 48) then
    begin
     if (Feine <> '' ) then Feine = Feine || cast(', ' as varchar(2) character set UTF8);
     Feine = Feine || FFeine;
    end

    if (Char_Length(Lot) + Char_Length(FLot) < 148) then
    begin
     if (Lot <> '' ) then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
     Lot = Lot || FLot;
    end

    if (Char_Length(Ormeci) + Char_Length(FOrmeci) < 48) then
    begin
     if (Ormeci <> '' ) then Ormeci = Ormeci || cast(', ' as varchar(2) character set UTF8);
     Ormeci = Ormeci || FOrmeci;
    end
  end

  suspend;
end
