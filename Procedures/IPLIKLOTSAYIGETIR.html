CREATE PROCEDURE IPLIKLOTSAYIGETIR(
    AYIL INTEGER,
    AELYAFTIPIID INTEGER)
RETURNS (
    LOTSAYI1 INTEGER,
    LOTSAYI2 INTEGER,
    LOTSAYI3 INTEGER)
AS
declare variable FLotSayi1 integer;
begin
  Select first 1 LotSayi1, LotSayi2, LotSayi3 from IplikLotSayi
    /*where IplikLotSayi.Yil = :AYil and IplikLotSayi.ElyafTipiID = :AElyafTipiId */
    /*YIL basinda sifirlamasin dediler diye AYIL kriterini kaldirdik */
    where IplikLotSayi.ElyafTipiID = :AElyafTipiId
    order by IplikLotSayi.Yil desc
    into :LotSayi1, :LotSayi2, :LotSayi3;

  if (LotSayi1 is null) then
  begin
    LotSayi1 = 65;
    LotSayi2 = 65;
    LotSayi3 = 65;

    insert into IplikLotSayi(Yil, ElyafTipiId, LotSayi1, LotSayi2, LotSayi3) values(:AYil, :AElyafTipiId, :LotSayi1, :LotSayi2, :LotSayi3);
    suspend;
  end
  else
  begin
    if (AElyafTipiID = 1) then
    begin
      if (LotSayi2 >= 90) then
      begin
        LotSayi1 = LotSayi1 + 1;
        LotSayi2 = 65;

        if (LotSayi1 > 90) then LotSayi1 = 65;
      end
      else
      begin
        LotSayi2 = LotSayi2 + 1;
      end

      LotSayi3 = 1;
    end
    else if (AElyafTipiID = 2) then
    begin
      if (LotSayi3 >= 90) then
      begin
       if (LotSayi2 < 90) then
       begin
         LotSayi2 = LotSayi2 + 1;
         LotSayi3 = 65;
       end
       else
       begin
         LotSayi1 = LotSayi1 + 1;
         LotSayi2 = 65;
         LotSayi3 = 65;
         if (LotSayi1 > 90) then LotSayi1 = 65;
       end
      end
      else LotSayi3 = LotSayi3 + 1;

    end
    else
    begin
      LotSayi1 = LotSayi1 + 1;
      if (LotSayi1 > 999) then LotSayi1 = 1;

      LotSayi2 = 1;
      LotSayi3 = 1;
    end

    Select first 1 LotSayi1 from IplikLotSayi
    where IplikLotSayi.Yil = :AYil and IplikLotSayi.ElyafTipiID = :AElyafTipiId
    order by IplikLotSayi.Yil desc
    into :FLotSayi1;

    if (FLotSayi1 is Null) then
      insert into IplikLotSayi(Yil, ElyafTipiId, LotSayi1, LotSayi2, LotSayi3) values(:AYil, :AElyafTipiId, :LotSayi1, :LotSayi2, :LotSayi3);
    else
      Update IplikLotSayi Set LotSayi1 = :LotSayi1, LotSayi2 = :LotSayi2, LotSayi3 = :LotSayi3 where IplikLotSayi.Yil = :AYil and IplikLotSayi.ElyafTipiID = :AElyafTipiID;
    suspend;
  end
end
