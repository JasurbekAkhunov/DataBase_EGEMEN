CREATE PROCEDURE FIRMABAZINDABOYAHANEANALIZI_1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    KACILANPARTINORMALKILOTOPLAM DOUBLE PRECISION,
    KACILANPARTITAMIRKILOTOPLAM DOUBLE PRECISION,
    KACILANPARTINUMUNEKILOTOPLAM DOUBLE PRECISION,
    KACILANPARTIKONTROLKILOTOPLAM DOUBLE PRECISION,
    KACILANPARTIRPTKILOTOPLAM DOUBLE PRECISION,
    KACILANPARTINORMALMETRETOPLAM DOUBLE PRECISION,
    KACILANPARTITAMIRMETRETOPLAM DOUBLE PRECISION,
    KACILANPARTINUMUNEMETRETOPLAM DOUBLE PRECISION,
    KACILANPARTIKONTROLMETRETOPLAM DOUBLE PRECISION,
    KACILANPARTIRPTMETRETOPLAM DOUBLE PRECISION,
    IACILANPARTINORMALKILOTOPLAM DOUBLE PRECISION,
    IACILANPARTITAMIRKILOTOPLAM DOUBLE PRECISION,
    IACILANPARTINUMUNEKILOTOPLAM DOUBLE PRECISION,
    IACILANPARTIKONTROLKILOTOPLAM DOUBLE PRECISION,
    IACILANPARTIRPTKILOTOPLAM DOUBLE PRECISION)
AS
declare variable FlastFirmaId  varchar(20);
declare variable FFirmaId     varchar(20);
declare variable FFirmaAdi    varchar(100);
declare variable FIslemTipiId integer;
declare variable FPartiTipiId integer;
declare variable FKilo        Numeric(12,3);
declare variable FMetre       Numeric(12,3);
begin
  FlastFirmaId = '';

  For Select Parti.FirmaId, Firma.Adi, Parti.IslemTipiId, Parti.PartiTipiId, Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from Parti
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
    where Parti.AcilisTarihi >= :ABaslangicTarihi and Parti.AcilisTarihi <= :ABitisTarihi and
          (Firma.AnaHesapFirmaID = :AFirmaId or :AFirmaID is null) and PartiStok.PartiSinifID is not null and PartiStok.PartiStokIptalKayitID is null
    group by Parti.FirmaId, Firma.Adi, Parti.IslemTipiId, Parti.PartiTipiId
    into :FFirmaId, :FFirmaAdi, :FIslemTipiId, :FPartiTipiId, :FKilo, :FMetre do
  begin
    if (FLastFirmaID <> FFirmaID) then
    begin
      if (FLastFirmaID <> '') then suspend;
      FirmaID = FFirmaId;
      FLastFirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;

      KAcilanPartiNormalKiloToplam         = 0;
      KAcilanPartiTamirKiloToplam          = 0;
      KAcilanPartiNumuneKiloToplam         = 0;
      KAcilanPartiKontrolKiloToplam        = 0;
      KAcilanPartiRptKiloToplam            = 0;

      KAcilanPartiNormalMetreToplam        = 0;
      KAcilanPartiTamirMetreToplam         = 0;
      KAcilanPartiNumuneMetreToplam        = 0;
      KAcilanPartiKontrolMetreToplam       = 0;
      KAcilanPartiRptMetreToplam           = 0;

      IAcilanPartiNormalKiloToplam         = 0;
      IAcilanPartiTamirKiloToplam          = 0;
      IAcilanPartiNumuneKiloToplam         = 0;
      IAcilanPartiKontrolKiloToplam        = 0;
      IAcilanPartiRptKiloToplam            = 0;
    end
    if (FPartiTipiID = 0) then
    begin
      if (FIslemTipiID = 0) then IAcilanPartiNormalKiloToplam = FKilo;
      else if (FIslemTipiID = 1) then IAcilanPartiRptKiloToplam = FKilo;
      else if (FIslemTipiID = 2) then IAcilanPartiKontrolKiloToplam = FKilo;
      else if (FIslemTipiID = 3) then IAcilanPartiTamirKiloToplam = FKilo;
      else if (FIslemTipiID = 4) then IAcilanPartiNumuneKiloToplam = FKilo;
    end
    else
    begin
      if (FIslemTipiID = 0) then KAcilanPartiNormalKiloToplam = FKilo;
      else if (FIslemTipiID = 1) then KAcilanPartiRptKiloToplam = FKilo;
      else if (FIslemTipiID = 2) then KAcilanPartiKontrolKiloToplam = FKilo;
      else if (FIslemTipiID = 3) then KAcilanPartiTamirKiloToplam = FKilo;
      else if (FIslemTipiID = 4) then KAcilanPartiNumuneKiloToplam = FKilo;

      if (FIslemTipiID = 0) then KAcilanPartiNormalMetreToplam = FMetre;
      else if (FIslemTipiID = 1) then KAcilanPartiRptMetreToplam = FMetre;
      else if (FIslemTipiID = 2) then KAcilanPartiKontrolMetreToplam = FMetre;
      else if (FIslemTipiID = 3) then KAcilanPartiTamirMetreToplam = FMetre;
      else if (FIslemTipiID = 4) then KAcilanPartiNumuneMetreToplam = FMetre;
    end
  end

  if (FLastFirmaID <> '') then suspend;
end
