CREATE PROCEDURE DOKUMAPLANLAMASURUKLEBIRAK(
    AORMESIPARISDETAYID INTEGER,
    ADOKUMAMAKINAID VARCHAR(10),
    AFIRMAID VARCHAR(20),
    ADOKUMAPLANLAMAID INTEGER,
    AALTDOKUMAPLANLAMAID INTEGER,
    AKULLANICIID INTEGER,
    AMIKTARGUN DOUBLE PRECISION)
AS
declare variable FBoyaSiparisKayitId  integer;
declare variable FSira                  integer;
declare variable FPlanlanacakMiktar     Numeric(12,3);
declare variable OPId           integer;
declare variable ASira          integer;
declare variable FTarih         Date;
declare variable FBirimFiyat    Double Precision;
declare variable FDovizId       varchar(5);
declare variable FVadeGun       integer;
declare variable FBeyazlik      integer;
declare variable FTezgahtaAtilacakAtki Double Precision;
begin
  FSira = -1;
  FTarih = current_TimeStamp(0);

  if (ADokumaPlanlamaId is null) then
  begin
    select BoyaSiparisKayitId, (OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.PlanlananMiktar - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar) from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId
     into :FBoyaSiparisKayitId, :FPlanlanacakMiktar;

    if (FPlanlanacakMiktar <= 0) then exception SiparisinTamamiPlanlanmis;
  end

  if (ADokumaPlanlamaId is Not null) then
  begin
    if (AFirmaId is null and ADokumaMakinaId is null) then
    begin
      delete from DokumaPlanlama where DokumaPlanlama.DokumaPlanlamaID = :ADokumaPlanlamaId;
      exit;
    end
  end

  if (AAltDokumaPlanlamaId is not null) then
  begin
    Select Sira from DokumaPlanlama where DokumaPlanlama.DokumaPlanlamaId = :AAltDokumaPlanlamaId into :FSira;
    if (FSira is null) then exception OrmePlanlamaYerBulunamadi;
  end

  if (ADokumaPlanlamaId is null) then
  begin
    select OrmeSiparisDetay.OrmeBirimMaliyet, OrmeSiparisDetay.OrmeBirimMaliyetDovizId, OrmeSiparisDetay.VadeGun, OrmeSiparisDetay.Beyazlik, Ham.TezgahtaAtilacakAtki from OrmeSiparisDetay
      Left Join Ham on Ham.HamID = OrmeSiparisDetay.HamID
    where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId
     into :FBirimFiyat, :FDovizId, :FVadeGun, :FBeyazlik, :FTezgahtaAtilacakAtki;

    insert into DokumaPlanlama(Tarih, FirmaId, DokumaMakinaId, Sira, BoyaSiparisKayitId, OrmeSiparisDetayId, Beyazlik, PlanlananMiktar,
                             BirimFiyat, DovizId, VadeGun, MiktarGun, TezgahtaAtilacakAtki, InsertKullaniciId, KullaniciId)
                      values(:FTarih, :AFirmaId, :ADokumaMakinaId, :FSira, :FBoyaSiparisKayitId, :AOrmeSiparisDetayId, :FBeyazlik, :FPlanlanacakMiktar,
                             :FBirimFiyat, :FDovizId, :FVadeGun, :AMiktarGun, :FTezgahtaAtilacakAtki, :AKullaniciId, :AKullaniciId);
  end
  else
  begin
    if (ADokumaPlanlamaId is not null) then
    begin
      if (ADokumaMakinaId is not null) then
      begin
        if (FSira > -1) then
        begin
          ASira = FSira;
          for select DokumaPlanlamaId from DokumaPlanlama where DokumaPlanlama.DokumaMakinaId = :ADokumaMakinaId and Sira >= :FSira order by Sira into :OPId do
          begin
            ASira = ASira + 1;
            update DokumaPlanlama set Sira = :ASira where DokumaPlanlamaId = :OPId;
          end
        end
        else
        begin
          select max(Sira) from DokumaPlanlama where DokumaPlanlama.DokumaMakinaId = :ADokumaMakinaId into :ASira;
          if (ASira is null) then ASira = 0;
          else ASira = ASira + 1;
          FSira = ASira;
        end
      end
      else
      begin
        if (FSira > -1) then
        begin
          ASira = FSira;
          for select DokumaPlanlamaId from DokumaPlanlama where DokumaPlanlama.FirmaId = :AFirmaId and Sira >= :FSira order by Sira into :OPId do
          begin
            ASira = ASira + 1;
            update DokumaPlanlama set Sira = :ASira where DokumaPlanlamaId = :OPId;
          end
        end
        else
        begin
          select max(Sira) from DokumaPlanlama where DokumaPlanlama.FirmaId = :AFirmaId into :ASira;
          if (ASira is null) then ASira = 0;
          else ASira = ASira + 1;
          FSira = ASira;
        end
      end
    end

    Update DokumaPlanlama Set DokumaMakinaId = :ADokumaMakinaId, FirmaId = :AFirmaId, Sira = :FSira where DokumaPlanlama.DokumaPlanlamaId = :ADokumaPlanlamaId;
  end
end
