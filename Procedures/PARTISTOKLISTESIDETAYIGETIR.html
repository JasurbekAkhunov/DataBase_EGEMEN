declare variable LastHamStokMasterID  integer;
declare variable FKullanilanMiktar    Numeric(12,3);
declare variable FIptalPartiMiktar    Numeric(12,3);
declare variable YariMamulPartiYil    integer;
declare variable YariMamulPartiId     integer;
declare variable FPartiKayitId         integer;
declare variable DokumaTopTakibiDevrede integer;
declare variable EgemenFirmaId integer;
begin
  LastPartiSinifId = -1;
  LastHamStokMasterID = -1;
  select DokumaTopTakibiDevrede, EgemenFirmaId from sistem where sistem.sistemid = 1 into :DokumaTopTakibiDevrede, :EgemenFirmaId;

  if (coalesce(AReceteId,'') <> '' and coalesce(APartiAsamalariId,0) <> 0 ) then
  begin
      for select PartiKayitId from PartiRecetesiDokumu_1(:APartiKayitId, :AReceteId, :APartiAsamalariId)
        into :FPartiKayitId do
      begin
        For select PartiStok.PartiSinifId, Ham.Adi, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.FirmaID, IrsaliyeFirma.Adi, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, HamStokDetay.HamStokMasterId, RafDetay.rafId,
               PartiStokDetay.BobinSayisi, PartiStokDetay.KapAdedi, PartiStokDetay.Kilo, PartiStokDetay.Metre, PartiStokDetay.Adet,
               Ormeci.Adi, UreticiFirma.Adi, HamStokMaster.OrmeciMakinaNo, HamStokMaster.BobinGramaji, HamStokMaster.HamKontrolBukumTuru,
               HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Lot, HamStokMaster.MelanjKodu, HamStokMaster.OrmeStokGirisNo, HamStokMaster.OrmeGirisIrsaliyeNo,
               HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo,
               HamStokMaster.En, HamStokMaster.Grm2, HamStokMaster.Renk,
               HamStokMaster.Denye, HamStokDetay.LycraUreticiId, LycraUretici.Adi, HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2,
               HamStokMaster.IplikUzunlugu3, TamirTipi.Adi, HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Miktar,
               HamStokMaster.TupAcikEnMayli, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, HamStokMaster.BoyaSiparisDetayId,
               HamStokMaster.Aciklama, HamStokMaster.Brut, PartiStok.HamId, DisTamirCozumSekli.Adi, DisTamirCozumSekli.Ucretli, BoyaSiparis.YurtDisi,
               Coalesce(OrmePlanlama.UretimPartiNo, DokumaPlanlama.UretimPartiNo),
               HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2,
               HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4,
               HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj, HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik,
               HamStokMaster.ElyafIcerigiAnalizi, HamStokMaster.BatchNo, Uretici.UreticiId, Uretici.Adi, StokHareketGrup.Adi, HamStokMaster.HamKontrolKarisimOrani,
               HamStokMaster.HamKontrolLikraOrani, HamStokMaster.NemOrani, Parti.PartiYil, Parti.PartiId, OrmeMakina.Adi,
               OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4,
               OrmePlanlama.DinlenmisGramaj, OrmePlanlama.MakinaUstuGramaj,
               OrmePlanlama.IgneSayisi, OrmePlanlama.MaySayisi, OrmePlanlama.PlanlananTurSayisi, OrmePlanlama.BinTurdaOrulecekKG, HamStokMaster.HamDonme,
               OrmeSiparisDetay.HamEn, HamStokMaster.KaliteId,Kalite.KaliteAdi,   PSDokumaIsEmriBilgisiGetir.CozguIndigoIsEmriIdStr,
         PSDokumaIsEmriBilgisiGetir.TopNo, HamStokMaster.HarmanAdi, OrmeRaf.Notu, Parti.ReceteId, Recete.Adi From Partistok
          left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
          left join RafDetay on RafDetay.RafDetayid = PartiStokDetay.RafDetayId
          left join HamStokDetay on HamStokDetay.HamStokDetayid = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterid = HamStokDetay.HamStokMasterId
          left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
          left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = HamStokIrsaliye.IrsaliyeGrupId
          left join Ormeci on Ormeci.OrmeciId = HamStokIrsaliye.OrmeciId
          left join Firma UreticiFirma on UreticiFirma.FirmaId = HamStokMaster.UreticiFirmaId
          left join Firma IrsaliyeFirma on IrsaliyeFirma.FirmaId = HamStokIrsaliye.FirmaId
          left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
          left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
          left join DisTamirCozum on DisTamirCozum.HamStokMasterId = HamStokMaster.HamStokMasterId
          left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliId = DisTamirCozum.DisTamirCozumSekliId
          left join Ham on Ham.HamId = RafDetay.HamId
          left join LycraUretici on LycraUretici.LycraUreticiId = HamStokDetay.LycraUreticiId
          left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitid = HamStokMaster.BoyaSiparisKayitid
          left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = HamStokMaster.OrmePlanlamaId
          left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = HamStokMaster.DokumaPlanlamaId
          left join Uretici on Uretici.UreticiId = HamStokMaster.UreticiId
          left join PartiStok YariMamulPartiStok on YariMamulPartiStok.PArtiSinifId = HamStokMaster.PartiSinifId
          left join Parti on Parti.PartiKayitID = YariMamulPartiStok.PartiKayitId
          left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
          left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
          left join Kalite on Kalite.KaliteId = HamStokMaster.KaliteId
          left join PSDokumaIsEmriBilgisiGetir(null, HamStokMaster.HamStokMasterId) ON 1=1 and HamStokMaster.DokumaPlanlamaId is not null
          Left Join StokHareket on StokHareket.StokHareketId = HamStokMaster.StokHareketId
          Left Join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
          Left join Recete on Recete.ReceteId = Parti.ReceteId
          /*
          left join DokumaStokPlanlama On DokumaStokPlanlama.Dokumaplanlamaid = HAmStokMAster.Dokumaplanlamaid and :EgemenFirmaId = 139
          left join OrmeRaf On OrmeRaf.OrmeRafId = DokumaStokPlanlama.Ormerafid
          left join IndigoIsEmri On IndigoIsEmri.IndigoIsEmriKayitId = OrmeRaf.IndigoIsEmriKayitId
          left join CozguIsEmri On CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId

          left join StokHareket On StokHareket.StokHareketId = HAmStokMAster.StokHareketId and :DokumaTopTakibiDevrede = 1
          left join OrmeRafDetay on OrmeRafDetay.Ormerafid = StokHareket.Ormerafid and :EgemenFirmaId = 139
          left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketID = OrmeRafDetay.StokHareketId
          */

          order by PartiStok.PartiSinifId, HamStokDetay.HamStokMasterId
          into :PartiSinifId, :HamAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeFirmaID, :IrsaliyeFirmaAdi, :IrsaliyeTarihi, :EvrakTipi, :HamStokMasterId, :RafId,
               :BobinSayisi, :KapAdedi, :Kilo, :Metre, :Adet, :OrmeciAdi, :UreticiFirmaAdi, :OrmeciMakinaNo, :BobinGramaji, :HamKontrolBukumTuru,  :Pus, :Feine, :Lot, :MelanjKodu, :OrmeStokGirisNo, :OrmeGirisIrsaliyeNo,
               :MusteriPartiNo, :MusteriSiparisNo, :ReferansSiparisNo, :En, :Grm2,
               :Renk, :Denye, :LycraUreticiId, :LycraUreticisiAdi, :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3, :TamirTipiAdi,
               :IrsaliyeKapAdedi, :IrsaliyeKilo, :IrsaliyeMiktar, :TupAcikEnMayli, :YikamaOncesiEn, :YikamaOncesiGramaj,
               :GirisBoyaSiparisYil, :GirisBoyaSiparisId, :GirisBoyaSiparisDetayId, :GirisIrsaliyeNotu, :Brut, :HamId,
               :DisTamirCozumSekliAdi, :DisTamirCozumUcretli, :SiparisYurtDisi, :UretimPartiNo,
               :HamKontrolIplikUzunlugu1, :HamKontrolIplikUzunlugu2, :HamKontrolIplikUzunlugu3, :HamKontrolIplikUzunlugu4,
               :YikamaSonrasiEn, :YikamaSonrasiGramaj, :EndenCekmezlik, :BoydanCekmezlik, :ElyafIcerigiAnalizi, :BatchNo, :UreticiId, :UreticiAdi, :IrsaliyeGrupAdi,
               :HamKontrolKarisimOrani, :HamKontrolLikraOrani, :NemOrani, :YariMamulPartiYil, :YariMamulPartiId, :OrmeMakinaAdi,
               :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :DinlenmisGramaj, :MakinaUstuGramaj,
               :IgneSayisi, :MaySayisi, :PlanlananTurSayisi, :BinTurdaOrulecekKG, :HamDonme, :HamEn, :KaliteId, :KaliteAdi, :DokumaOzelKodStr, :DokumaTopNo, :HarmanAdi, :OrmeRafNotu,
         :YariMamulPartiReceteId, :YariMamulPartiReceteAdi do
        begin
          if (LastPartiSinifId <> PartiSinifID) then
          begin
            SiraNo = 1;
            LastPartiSinifId = PartiSinifID;
            LastHamStokMasterID = HamStokMasterId;

            select Sum(RafDetay.IadeEdilecekMiktar + rafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.SatisMiktar) from HamStokDetay
            left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayID
            where HamStokDetay.HamStokMasterId = :HamStokMasterId
            into :FKullanilanMiktar;

            Select Sum(RafDetay.Miktar) from HamStokDetay
               left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
               where HamStokDetay.HamStokMasterId = :HamStokMasterId
               into :FIptalPartiMiktar;

            if (FIptalPartiMiktar is null) then FIptalPartiMiktar = 0;

            if (FKullanilanMiktar is null) then FKullanilanMiktar = 0;

            IrsaliyeKalanMiktar = IrsaliyeMiktar +  FIptalPartiMiktar - FKullanilanMiktar;

          end

          if (LastHamStokMasterID <> HamStokMasterId) then
          begin
            SiraNO = SiraNO + 1;

            select Sum(RafDetay.IadeEdilecekMiktar + rafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.SatisMiktar) from HamStokDetay
            left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayID
            where HamStokDetay.HamStokMasterId = :HamStokMasterId
            into :FKullanilanMiktar;

            Select Sum(RafDetay.Miktar) from HamStokDetay
               left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
               where HamStokDetay.HamStokMasterId = :HamStokMasterId
               into :FIptalPartiMiktar;

            if (FIptalPartiMiktar is null) then FIptalPartiMiktar = 0;

            if (FKullanilanMiktar is null) then FKullanilanMiktar = 0;


            IrsaliyeKalanMiktar = IrsaliyeMiktar +  FIptalPartiMiktar - FKullanilanMiktar;
          end

          Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

          if (EvrakTipi = 0) then EvrakTipiStr = 'Fi';
          else if (EvrakTipi = 1) then EvrakTipiStr = 'rsaliye';
          else EvrakTipiStr = 'Hatal';

          if (Brut is null) then BrutStr = '';
          else if (Brut = 1) then BrutStr = 'Brt';
          else if (Brut = 0) then BrutStr = 'Net';

          YariMamulPartiNoStr = '';
          if (YariMamulPartiYil is not null) then
            YariMamulPartiNoStr = '('||YariMamulPartiYil||') ' || YariMamulPartiId;
          suspend;
        end
      end
  end
  else
  begin
  For select PartiStok.PartiSinifId, Ham.Adi, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.FirmaID, IrsaliyeFirma.Adi, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, HamStokDetay.HamStokMasterId, RafDetay.rafId,
               PartiStokDetay.BobinSayisi, PartiStokDetay.KapAdedi, PartiStokDetay.Kilo, PartiStokDetay.Metre, PartiStokDetay.Adet,
               Ormeci.Adi, UreticiFirma.Adi, HamStokMaster.OrmeciMakinaNo, HamStokMaster.BobinGramaji, HamStokMaster.HamKontrolBukumTuru,
               HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Lot, HamStokMaster.MelanjKodu, HamStokMaster.OrmeStokGirisNo, HamStokMaster.OrmeGirisIrsaliyeNo,
               HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo,
               HamStokMaster.En, HamStokMaster.Grm2, HamStokMaster.Renk,
               HamStokMaster.Denye, HamStokDetay.LycraUreticiId, LycraUretici.Adi, HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2,
               HamStokMaster.IplikUzunlugu3, TamirTipi.Adi, HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Miktar,
               HamStokMaster.TupAcikEnMayli, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, HamStokMaster.BoyaSiparisDetayId,
               HamStokMaster.Aciklama, HamStokMaster.Brut, PartiStok.HamId, DisTamirCozumSekli.Adi, DisTamirCozumSekli.Ucretli, BoyaSiparis.YurtDisi,
               Coalesce(OrmePlanlama.UretimPartiNo, DokumaPlanlama.UretimPartiNo),
               HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2,
               HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4,
               HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj, HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik,
               HamStokMaster.ElyafIcerigiAnalizi, HamStokMaster.BatchNo, Uretici.UreticiId, Uretici.Adi, StokHareketGrup.Adi, HamStokMaster.HamKontrolKarisimOrani,
               HamStokMaster.HamKontrolLikraOrani, HamStokMaster.NemOrani, Parti.PartiYil, Parti.PartiId, OrmeMakina.Adi,
               OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4,
               OrmePlanlama.DinlenmisGramaj, OrmePlanlama.MakinaUstuGramaj,
               OrmePlanlama.IgneSayisi, OrmePlanlama.MaySayisi, OrmePlanlama.PlanlananTurSayisi, OrmePlanlama.BinTurdaOrulecekKG, HamStokMaster.HamDonme,
               OrmeSiparisDetay.HamEn, HamStokMaster.KaliteId,Kalite.KaliteAdi,  PSDokumaIsEmriBilgisiGetir.CozguIndigoIsEmriIdStr, PSDokumaIsEmriBilgisiGetir.TopNo,
               HamStokMaster.HarmanAdi, OrmeRaf.Notu, Parti.ReceteId, Recete.Adi From Partistok
          left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
          left join RafDetay on RafDetay.RafDetayid = PartiStokDetay.RafDetayId
          left join HamStokDetay on HamStokDetay.HamStokDetayid = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterid = HamStokDetay.HamStokMasterId
          left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
          left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = HamStokIrsaliye.IrsaliyeGrupId
          left join Ormeci on Ormeci.OrmeciId = HamStokIrsaliye.OrmeciId
          left join Firma UreticiFirma on UreticiFirma.FirmaId = HamStokMaster.UreticiFirmaId
          left join Firma IrsaliyeFirma on IrsaliyeFirma.FirmaId = HamStokIrsaliye.FirmaId
          left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
          left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
          left join DisTamirCozum on DisTamirCozum.HamStokMasterId = HamStokMaster.HamStokMasterId
          left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliId = DisTamirCozum.DisTamirCozumSekliId
          left join Ham on Ham.HamId = RafDetay.HamId
          left join LycraUretici on LycraUretici.LycraUreticiId = HamStokDetay.LycraUreticiId
          left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitid = HamStokMaster.BoyaSiparisKayitid
          left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = HamStokMaster.OrmePlanlamaId
          left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = HamStokMaster.DokumaPlanlamaId
          left join Uretici on Uretici.UreticiId = HamStokMaster.UreticiId
          left join PartiStok YariMamulPartiStok on YariMamulPartiStok.PArtiSinifId = HamStokMaster.PartiSinifId
          left join Parti on Parti.PartiKayitID = YariMamulPartiStok.PartiKayitId
          left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
          left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
          left join Kalite on Kalite.KaliteId = HamStokMaster.KaliteId
          left join PSDokumaIsEmriBilgisiGetir(null,HamStokMaster.HamStokMasterId) ON 1=1 and HamStokMaster.DokumaPlanlamaId is not null
          Left Join StokHareket on StokHareket.StokHareketId = HamStokMaster.StokHareketId
          Left Join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
          Left join Recete on Recete.ReceteId = Parti.ReceteId
           /*
          left join DokumaStokPlanlama On DokumaStokPlanlama.Dokumaplanlamaid = HAmStokMAster.Dokumaplanlamaid and :EgemenFirmaId = 139
          left join OrmeRaf On OrmeRaf.OrmeRafId = DokumaStokPlanlama.Ormerafid
          left join IndigoIsEmri On IndigoIsEmri.IndigoIsEmriKayitId = OrmeRaf.IndigoIsEmriKayitId
          left join CozguIsEmri On CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId

          left join StokHareket On StokHareket.StokHareketId = HAmStokMAster.StokHareketId and :DokumaTopTakibiDevrede = 1
          left join OrmeRafDetay on OrmeRafDetay.Ormerafid = StokHareket.Ormerafid   and :EgemenFirmaId = 139
          left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketID = HamStokMaster.StokHareketId
          */
          where PartiStok.PartiKayitId = :APartiKayitId and PartiStok.PartiStokIptalKayitId is null
          order by PartiStok.PartiSinifId, HamStokDetay.HamStokMasterId
          into :PartiSinifId, :HamAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeFirmaID, :IrsaliyeFirmaAdi, :IrsaliyeTarihi, :EvrakTipi, :HamStokMasterId, :RafId,
               :BobinSayisi, :KapAdedi, :Kilo, :Metre, :Adet, :OrmeciAdi, :UreticiFirmaAdi, :OrmeciMakinaNo, :BobinGramaji, :HamKontrolBukumTuru,  :Pus, :Feine, :Lot, :MelanjKodu, :OrmeStokGirisNo, :OrmeGirisIrsaliyeNo,
               :MusteriPartiNo, :MusteriSiparisNo, :ReferansSiparisNo, :En, :Grm2,
               :Renk, :Denye, :LycraUreticiId, :LycraUreticisiAdi, :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3, :TamirTipiAdi,
               :IrsaliyeKapAdedi, :IrsaliyeKilo, :IrsaliyeMiktar, :TupAcikEnMayli, :YikamaOncesiEn, :YikamaOncesiGramaj,
               :GirisBoyaSiparisYil, :GirisBoyaSiparisId, :GirisBoyaSiparisDetayId, :GirisIrsaliyeNotu, :Brut, :HamId,
               :DisTamirCozumSekliAdi, :DisTamirCozumUcretli, :SiparisYurtDisi, :UretimPartiNo,
               :HamKontrolIplikUzunlugu1, :HamKontrolIplikUzunlugu2, :HamKontrolIplikUzunlugu3, :HamKontrolIplikUzunlugu4,
               :YikamaSonrasiEn, :YikamaSonrasiGramaj, :EndenCekmezlik, :BoydanCekmezlik, :ElyafIcerigiAnalizi, :BatchNo, :UreticiId, :UreticiAdi, :IrsaliyeGrupAdi,

               :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :DinlenmisGramaj, :MakinaUstuGramaj,
               :IgneSayisi, :MaySayisi, :PlanlananTurSayisi, :BinTurdaOrulecekKG, :HamDonme, :HamEn, :KaliteId, :KaliteAdi, :DokumaOzelKodStr,
         :DokumaTopNo, :HarmanAdi, :OrmeRafNotu, :YariMamulPartiReceteId, :YariMamulPartiReceteAdi do
        begin
          if (LastPartiSinifId <> PartiSinifID) then
          begin
            SiraNo = 1;
            LastPartiSinifId = PartiSinifID;
            LastHamStokMasterID = HamStokMasterId;

            select Sum(RafDetay.IadeEdilecekMiktar + rafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.SatisMiktar) from HamStokDetay
            left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayID
            where HamStokDetay.HamStokMasterId = :HamStokMasterId
            into :FKullanilanMiktar;

            Select Sum(RafDetay.Miktar) from HamStokDetay
               left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
               where HamStokDetay.HamStokMasterId = :HamStokMasterId
               into :FIptalPartiMiktar;

            if (FIptalPartiMiktar is null) then FIptalPartiMiktar = 0;

            if (FKullanilanMiktar is null) then FKullanilanMiktar = 0;

            IrsaliyeKalanMiktar = IrsaliyeMiktar +  FIptalPartiMiktar - FKullanilanMiktar;

          end

          if (LastHamStokMasterID <> HamStokMasterId) then
          begin
            SiraNO = SiraNO + 1;

            select Sum(RafDetay.IadeEdilecekMiktar + rafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.SatisMiktar) from HamStokDetay
            left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayID
            where HamStokDetay.HamStokMasterId = :HamStokMasterId
            into :FKullanilanMiktar;

            Select Sum(RafDetay.Miktar) from HamStokDetay
               left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
               where HamStokDetay.HamStokMasterId = :HamStokMasterId
               into :FIptalPartiMiktar;

            if (FIptalPartiMiktar is null) then FIptalPartiMiktar = 0;

            if (FKullanilanMiktar is null) then FKullanilanMiktar = 0;


            IrsaliyeKalanMiktar = IrsaliyeMiktar +  FIptalPartiMiktar - FKullanilanMiktar;
          end

          Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

          if (EvrakTipi = 0) then EvrakTipiStr = 'Fi';
          else if (EvrakTipi = 1) then EvrakTipiStr = 'rsaliye';
          else EvrakTipiStr = 'Hatal';

          if (Brut is null) then BrutStr = '';
          else if (Brut = 1) then BrutStr = 'Brt';
          else if (Brut = 0) then BrutStr = 'Net';

          YariMamulPartiNoStr = '';
          if (YariMamulPartiYil is not null) then
            YariMamulPartiNoStr = '('||YariMamulPartiYil||') ' || YariMamulPartiId;
          suspend;
        end
  end
