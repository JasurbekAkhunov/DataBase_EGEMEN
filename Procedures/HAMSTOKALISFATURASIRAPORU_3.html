CREATE PROCEDURE HAMSTOKALISFATURASIRAPORU_3(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    FATURATARIHI TIMESTAMP,
    DOVIZID VARCHAR(5),
    ALISFIATI DOUBLE PRECISION,
    SATISFIATI DOUBLE PRECISION,
    SERBESTALISFIATI DOUBLE PRECISION,
    SERBESTSATISFIATI DOUBLE PRECISION)
AS
begin
  for select HamStokAlisFaturasiRaporu_1.FaturaTarihi, DovizKuru.DovizId, DovizKuru.AlisFiati, DovizKuru.SatisFiati,
             DovizKuru.SerbestAlisFiati, DovizKuru.SerbestSatisFiati from HamStokAlisFaturasiRaporu_1(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi)
  left join DovizKuru on DovizKuru.Tarih = HamStokAlisFaturasiRaporu_1.FaturaTarihi
  group by HamStokAlisFaturasiRaporu_1.FaturaTarihi, DovizKuru.DovizId, DovizKuru.AlisFiati, DovizKuru.SatisFiati,
           DovizKuru.SerbestAlisFiati, DovizKuru.SerbestSatisFiati
  into :FaturaTarihi, :DovizId, :AlisFiati, :SatisFiati, :SerbestAlisFiati, :SerbestSatisFiati do
  begin
    suspend;
  end
end
