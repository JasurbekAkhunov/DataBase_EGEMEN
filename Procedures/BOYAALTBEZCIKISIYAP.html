CREATE PROCEDURE BOYAALTBEZCIKISIYAP(
    APARTIASAMASTOKDETAYID INTEGER)
RETURNS (
    ISLEMBASARILI EGEBOOLEAN,
    MESAJ VARCHAR(255))
AS
declare variable AltBezHamId    integer;
declare variable BoyaSiparisDetayId integer;
declare variable PartiSinifId    integer;
declare variable PartiKayitId    integer;
declare variable PartiStokTartiId integer;
declare variable CikilacakMetre BoyaMiktarSize;
declare variable Metre          BoyaMiktarSize;
declare variable DetayMetre     BoyaMiktarSize;
declare Variable FHSSarfFisiYil integer;
declare Variable FHSSarfFisiNo  integer;
declare Variable FHSSarfFisiId  integer;
declare variable SarfTipiId     integer;
declare variable PartiVar       EgeBoolean;
declare variable RafDetayId     integer;
declare variable YeterliStokVar EgeBoolean;
declare variable MevcutToplamStok BoyaMiktarSize;
begin
  IslemBasarili = 1;
  Mesaj = '';
  select BoyaSiparisDetay.AltBezHamId, BoyaSiparisDetay.BoyaSiparisDetayId, PartiAsamaStokDetay.CikisMetre from PartiAsamaStokDetay
  left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  where PartiAsamaStokDetay.PartiAsamaStokDetayId = :APartiAsamaStokDetayId into :AltBezHamId, :BoyaSiparisDetayId, :CikilacakMetre;

  if (AltBezHamId is not null) then /*laminasyon siparisi*/
  begin
    /*sarf tipi olustur*/
    SarfTipiId = null;
    select Max(SarfTipi.SarfTipiId) from SarfTipi where SarfTipi.Adi = 'ALT BEZ SARF' into :SarfTipiId;
    if (SarfTipiId is null) then
    begin
      insert into SarfTipi (Adi) values ('ALT BEZ SARF');
      SarfTipiId = gen_id(SarfTipi_Id_Gen, 0);
    end

    /*fise parti cikislarini gir*/
    if (CikilacakMetre > 0) then
    begin
      PartiVar = 0;

      for select BoyaSiparisDetayAltBez.PartiSinifId, PartiStok.PartiKayitId from BoyaSiparisDetayAltBez
          left join PartiStok on PartiStok.PartiSinifId = BoyaSiparisDetayAltBez.PartiSinifId
          where BoyaSiparisDetayAltBez.BoyaSiparisDetayId = :BoyaSiparisDetayId order by BoyaSiparisDetayAltBez.BoyaSiparisDetayAltBezId into :PartiSinifId, :PartiKayitId do
      begin
        /*siparise alt bez partisi varsa buraaya girer*/
        if (PartiVar = 0) then /*ilk giriste total stok kontrolu yap.*/
        begin
          select Coalesce(Sum(PartiStokTarti.Metre), 0) from BoyaSiparisDetayAltBez
          left join PartiStok on PartiStok.PartiSinifId = BoyaSiparisDetayAltBez.PartiSinifId
          left join PartiStokTarti on PartiStokTarti.PartiSinifId = PartiStok.PartiSinifId
            where BoyaSiparisDetayAltBez.BoyaSiparisDetayId = :BoyaSiparisDetayId and
                  PartiStokTarti.HamStokSarfFisiId is null and
                  PartiStokTarti.BoyaIrsaliyeId is null into :MevcutToplamStok;

          if (MevcutToplamStok < CikilacakMetre) then
          begin
            IslemBasarili = 0;
            Mesaj = 'Yeterli Parti Stok Yok';
            suspend;
            Exit;
          end

          /*fis olustur*/
          FHSSarfFisiYil = Extract(year from Current_date);
          select id from HamStokSarfFisiNumaratoruAl(:FHSSarfFisiYil) into :FHSSarfFisiNo;
          FHSSarfFisiId = gen_id(HamStokSarfFisi_Id_Gen, 1);
          insert into HamStokSarfFisi(HamStokSarfFisiId, FisYil, FisNo, FisTarihi, SarfTipiId, Notu, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
          values(:FHSSarfFisiId, :FHSSarfFisiYil, :FHSSarfFisiNo, Current_date, :SarfTipiId, 'Otomatik Sarf', 1, current_timestamp(0), 1, current_timestamp(0));
        end
        PartiVar = 1;
        for select PartiStokTarti.PartiStokTartiId, PartiStokTarti.Metre from PartiStokTarti
            where PartiStokTarti.PartiSinifId = :PartiSinifId and
                  PartiStokTarti.HamStokSarfFisiId is null and
                  PartiStokTarti.BoyaIrsaliyeId is null
                  order by PartiStokTarti.Metre into :PartiStokTartiId, :Metre do
        begin
          /*tarti bolunecek mi diye bakiliyor*/
          if (CikilacakMetre < Metre) then
            select PartiStokTartiBol.YeniPartiStokTartiId from PartiStokTartiBol(:PartiStokTartiId, :CikilacakMetre, 1, 1) into :PartiStokTartiId;

          /*tartiyi cik*/
          execute procedure PartiStokKapSarfFisiIleSevkEt(:FHSSarfFisiId, :PartiKayitId, :PartiSinifId, Null, :PartiStokTartiId, 1, :APartiAsamaStokDetayId, Null);

          CikilacakMetre = CikilacakMetre - Metre;
          if (CikilacakMetre <= 0) then
             break;
        end
      end

      /*parti baglantisi yok, fise ham stok cikisi yap*/
      if (PartiVar = 0) then
      begin
        select Coalesce(Sum(RafDetay.Miktar + RafDetay.AktarmadanIadeEdienKilo - RafDetay.AktarmayaCikanKilo - RafDetay.PartilenenMiktar - RafDetay.RafTransferMiktar - RafDetay.IadeEdilenMiktar - RafDetay.SatisMiktar), 0)
        from RafDetay where RafDetay.StokVar = 1 and RafDetay.HamId = :AltBezHamId into :MevcutToplamStok;

        if (MevcutToplamStok < CikilacakMetre) then
        begin
          IslemBasarili = 0;
          Mesaj = 'Yeterli Ham Stok Yok';
          suspend;
          Exit;
        end

        /*fis olustur*/
        FHSSarfFisiYil = Extract(year from Current_date);
        select id from HamStokSarfFisiNumaratoruAl(:FHSSarfFisiYil) into :FHSSarfFisiNo;
        FHSSarfFisiId = gen_id(HamStokSarfFisi_Id_Gen, 1);
        insert into HamStokSarfFisi(HamStokSarfFisiId, FisYil, FisNo, FisTarihi, SarfTipiId, Notu, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
        values(:FHSSarfFisiId, :FHSSarfFisiYil, :FHSSarfFisiNo, Current_date, :SarfTipiId, 'Otomatik Sarf', 1, current_timestamp(0), 1, current_timestamp(0));

        for select RafDetay.Miktar + RafDetay.AktarmadanIadeEdienKilo - RafDetay.AktarmayaCikanKilo - RafDetay.PartilenenMiktar - RafDetay.RafTransferMiktar - RafDetay.IadeEdilenMiktar - RafDetay.SatisMiktar,
          RafDetay.RafDetayId
          from RafDetay where RafDetay.StokVar = 1 and RafDetay.HamId = :AltBezHamId into :Metre, :RafDetayId do
        begin
          if (CikilacakMetre <= Metre) then
            DetayMetre = CikilacakMetre;
          else
            DetayMetre = Metre;

          insert into HamStokSarfFisiDetay(HamStokSarfFisiId, RafDetayId, Metre, PartiAsamaStokDetayId) values (:FHSSarfFisiId, :RafDetayId, :DetayMetre, :APartiAsamaStokDetayId);

          CikilacakMetre = CikilacakMetre - DetayMetre;
          if (CikilacakMetre <= 0) then
             break;
        end
      end
    end
  end
  else
  begin
    IslemBasarili = 0;
    Mesaj = 'Sipariste Alt Bez Tanimli Degil';
  end
  suspend;
end
