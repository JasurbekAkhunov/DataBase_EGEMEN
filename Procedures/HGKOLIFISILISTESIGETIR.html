CREATE PROCEDURE HGKOLIFISILISTESIGETIR(
    ABASLAMATARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHGKOLIFISIID INTEGER,
    AHGSIPARISYIL INTEGER,
    AHGSIPARISID INTEGER)
RETURNS (
    HGKOLIFISIID INTEGER,
    FISTARIHI TIMESTAMP,
    FIRMAID FIRMAKODUSIZE,
    FIRMAADI ADI100,
    HGSIPARISKAYITID INTEGER,
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER)
AS
begin
  for select HgKoliFisi.HgKoliFisiId, HgKoliFisi.FisTarihi, HgSiparis.FirmaId, Firma.Adi, HgSiparis.HgSiparisKayitId, HgSiparis.HgSiparisYil, HgSiparis.HgSiparisId from HgSiparis
      left join HgKoliFisi on HgKoliFisi.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
      left join Firma on Firma.FirmaId = HgSiparis.FirmaId
      where (HgSiparis.HgSiparisYil = :AHgSiparisYil or :AHgSiparisYil is null) and
      (HgSiparis.HgSiparisId = :AHgSiparisId or :AHgSiparisId is null) and
      (HgKoliFisi.HgKoliFisiId = :AHgKoliFisiId or :AHgKoliFisiId is null) and
      (HgKoliFisi.FisTarihi >= :ABaslamaTarihi or :ABaslamaTarihi is null) and
      (HgKoliFisi.FisTarihi <= :ABitisTarihi or :ABitisTarihi is null)
      into :HgKoliFisiId, :FisTarihi, :FirmaId, :FirmaAdi, :HgSiparisKayitId, :HgSiparisYil, :HgSiparisId do
  begin
    suspend;
  end
end
