            where OrmeSiparisRezerv.OrmeRafID = :OrmeRafID into :FBoyaSiparisYil, :FBoyaSiparisId, :FOrmeSipRezervMiktar do

        FOrmeSiparisRezervSiparisStr = '('|| EgemenIntegerToStr(FBoyaSiparisYil) || ' - '  || EgemenIntegerToStr(FBoyaSiparisId)  || ' Miktar : '|| Format_float('', FOrmeSipRezervMiktar) || ')';
        if (Char_Length(FOrmeSiparisRezervSiparisStr) + Char_Length(OrmeSiparisRezervSiparisStr) < 1000) then
        begin
          if (OrmeSiparisRezervSiparisStr <> '' ) then OrmeSiparisRezervSiparisStr = OrmeSiparisRezervSiparisStr || cast(', ' as varchar(2) character set UTF8);
          OrmeSiparisRezervSiparisStr = OrmeSiparisRezervSiparisStr || FOrmeSiparisRezervSiparisStr;
        end
      end

      select Sum(StokHareket.Kilo), Sum(StokHareket.BrutMiktar), Sum(StokHareket.Metre)
      from OrmeRafDetay
      left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketId
      where OrmeRafDetay.OrmeRafId = :OrmeRafId into :Kilo,  :BrutKilo, :Metre;

         Select Sum(Kilo), Sum(BrutKilo)  from OrmeRafTransferFisiDetay
      where OrmeRafTransferFisiDetay.YeniOrmeRafId = :OrmeRafId into :RafMiktar, :RafBrutMiktar;

      TakipBirimi = 'kg';
      if ((HamTipiID = hstDokumaKumas and dokumakumasikilodantakipet <> 1) or ((HamTipiID = hstIplik or HamTipiID = hstElyaf or HamTipiID = hstTelef) and CozguIsEmriId is not Null)) then TakipBirimi = 'mt';


      if (TakipBirimi = 'mt') then Metre = Metre + Coalesce(RafMiktar, 0);
      else
      begin
        Kilo = Coalesce(Kilo,0) + Coalesce(RafMiktar, 0);
        BrutKilo =  Coalesce(BrutKilo,0) + Coalesce(RafBrutMiktar, 0);
      end

      select Sum(StokHareket.Kilo), Sum(StokHareket.BrutMiktar), Sum(StokHareket.Metre), Max(StokHareket.Tarih)
      from StokHareket
      where StokHareket.OrmeRafId = :OrmeRafId into :FKilo,  :FBrutKilo, :FMetre, :SonCikisHareketTarihi;

        RafMiktar = 0;
      RafBrutMiktar = 0;
      Select Sum(Kilo), Sum(BrutKilo) from OrmeRafTransferFisiDetay
      where OrmeRafTransferFisiDetay.OrmeRafId = :OrmeRafId into :RafMiktar, :RafBrutMiktar;

      if (TakipBirimi = 'mt') then FMetre = FMetre + Coalesce(RafMiktar, 0);
      else
      begin
        FKilo =  Coalesce(FKilo,0) + Coalesce(RafMiktar, 0);
        FBrutKilo =  Coalesce(FBrutKilo,0) + Coalesce(RafBrutMiktar, 0);
      end


      select list(distinct HamHataTipi.Adi,',')
      from OrmeKaliteKontrol
      left join HamHataTipi on HamHataTipi.HamHataTipiId = OrmeKaliteKontrol.SakatHamHataTipiId
      where OrmeKaliteKontrol.OrmePlanlamaId = :OrmePlanlamaId and OrmeKaliteKontrol.Sakat = 1 and OrmeKaliteKontrol.RafId = :RafId into :SakatHataAdiStr;

      KalanKilo = Coalesce(Kilo, 0) - Coalesce(FKilo, 0);
      BrutKalanKilo = Coalesce(BrutKilo, 0) - Coalesce(FBrutKilo, 0);
      KalanMetre = Coalesce(Metre, 0) - Coalesce(FMetre, 0);

      RezervSonrasiKalanMiktar = KalanMiktar - OrmeSipRezervKalanMiktar;
      Select BrutNetStr From BrutNetListesiGetir Where BrutNet = :FBrut Into :BrutNetStr;

      Suspend;
    end
  end
