CREATE PROCEDURE HGURETIMRAPORU_2(
    ABASLAMATARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHGSIPARISYIL INTEGER,
    AHGSIPARISID INTEGER,
    AMODELID MODELKODUSIZE)
RETURNS (
    MODELID MODELKODUSIZE,
    HAMID INTEGER,
    URUNID URUNKODUSIZE,
    SARFADI ADI255,
    AKSESUARSTR VARCHAR(5),
    MIKTAR BOYAMIKTARSIZE)
AS
declare variable shtUretim             Integer;
declare variable shtFasonUretim        Integer;
declare variable shtUretimeCikis       Integer;
declare variable shtFasonUretimeCikis  Integer;
declare variable oshtUretimeCikis      Integer;
declare variable oshtFasonUretimeCikis Integer;
declare variable utAksesuar            Integer;
declare variable HgSiparisDetayId      Integer;
declare variable AModelKayitId Integer;
begin
  if (AModelId is Not Null) then Select ModelKayitId From Model Where Model.ModelID = :AModelID  into :AModelKayitId;

  select StokHareketTipiGetir.shtUretim, StokHareketTipiGetir.shtFasonUretim, StokHareketTipiGetir.shtUretimeCikis, StokHareketTipiGetir.shtFasonUretimeCikis from StokHareketTipiGetir
  into :shtUretim, :shtFasonUretim, :shtUretimeCikis, :shtFasonUretimeCikis;

  select OrmeStokHareketTipiGetir.shtUretimeCikis, OrmeStokHareketTipiGetir.shtFasonUretimeCikis from OrmeStokHareketTipiGetir into :oshtUretimeCikis, :oshtFasonUretimeCikis;

  select UrunTipiGetir.utAksesuar from UrunTipiGetir into :utAksesuar;
  UrunId = Null;
  AksesuarStr = '';

  /*kesim icin kumas cikislari*/
  for select Model.ModelId, Ham.HamId, Ham.Adi, Sum(StokHareket.Miktar) from
    (select StokDetay.HgSiparisDetayId
    from StokDetay
    left join Depo on Depo.DepoId = StokDetay.DepoId
    left join HgSiparis on HgSiparis.HgSiparisKayitId = StokDetay.HgSiparisKayitId
    left join Urun on Urun.UrunId = StokDetay.UrunId
    where (StokDetay.HareketTipi = :shtUretim or StokDetay.HareketTipi = :shtFasonUretim) and
    StokDetay.HgIsEmriKayitId is not null and
    Depo.HgYariMamul = 0 and
    (:ABaslamaTarihi is null or StokDetay.Tarih >= :ABaslamaTarihi and StokDetay.Tarih <= :ABitisTarihi) and
    (:AHgSiparisYil is null or HgSiparis.HgSiparisYil = :AHgSiparisYil) and
    (:AHgSiparisId is null or HgSiparis.HgSiparisId = :AHgSiparisId) and
    (:AModelKayitId is null or Urun.ModelKayitId = :AModelKayitId)
    group by StokDetay.HgSiparisDetayId) Siparisler
    left join HgSiparisDetay on HgSiparisDetay.HgSiparisDetayId = Siparisler.HgSiparisDetayId
    left join HgIsEmri on HgIsEmri.HgSiparisDetayId = HgSiparisDetay.HgSiparisDetayId
    left join StokHareket on StokHareket.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
    left join Ham on Ham.HamId = StokHareket.HamId
    left join Model on Model.ModelKayitId = HgSiparisDetay.ModelKayitId
    where HgIsEmri.KesimTalimati = 1 and
    (StokHareket.HareketTipi = :oshtUretimeCikis or StokHareket.HareketTipi = :oshtFasonUretimeCikis)
    group by Model.ModelId, Ham.HamId, Ham.Adi into :ModelId, :HamId, :SarfAdi, :Miktar
  do
  begin
    suspend;
  end

  HamId = Null;
  AksesuarStr = 'Evet';

  /*diger icin urun cikislari*/
  for select Model.ModelId, UrunCikis.UrunId, UrunCikis.Adi, Sum(SDCikis.Miktar) from
    (select StokDetay.HgSiparisDetayId
    from StokDetay
    left join Depo on Depo.DepoId = StokDetay.DepoId
    left join HgSiparis on HgSiparis.HgSiparisKayitId = StokDetay.HgSiparisKayitId
    left join Urun on Urun.UrunId = StokDetay.UrunId
    where (StokDetay.HareketTipi = :shtUretim or StokDetay.HareketTipi = :shtFasonUretim) and
    StokDetay.HgIsEmriKayitId is not null and
    Depo.HgYariMamul = 0 and
    (:ABaslamaTarihi is null or StokDetay.Tarih >= :ABaslamaTarihi and StokDetay.Tarih <= :ABitisTarihi) and
    (:AHgSiparisYil is null or HgSiparis.HgSiparisYil = :AHgSiparisYil) and
    (:AHgSiparisId is null or HgSiparis.HgSiparisId = :AHgSiparisId) and
    (:AModelKayitId is null or Urun.ModelKayitId = :AModelKayitId)
    group by StokDetay.HgSiparisDetayId) Siparisler
    left join HgSiparisDetay on HgSiparisDetay.HgSiparisDetayId = Siparisler.HgSiparisDetayId
    left join HgIsEmri on HgIsEmri.HgSiparisDetayId = HgSiparisDetay.HgSiparisDetayId
    left join StokDetay SDCikis on SDCikis.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
    left join Urun UrunCikis on UrunCikis.UrunId = SDCikis.UrunId
    left join Model on Model.ModelKayitId = HgSiparisDetay.ModelKayitId
    where HgIsEmri.KesimTalimati = 0 and
    (SDCikis.HareketTipi = :shtUretimeCikis or SDCikis.HareketTipi = :shtFasonUretimeCikis) and
    UrunCikis.UrunTipiId = :utAksesuar
    group by Model.ModelId, UrunCikis.UrunId, UrunCikis.Adi into :ModelId, :UrunId, :SarfAdi, :Miktar
  do
  begin
    suspend;
  end
end
