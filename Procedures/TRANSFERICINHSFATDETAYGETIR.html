CREATE PROCEDURE TRANSFERICINHSFATDETAYGETIR(
    AHAMSTOKFATURAID INTEGER)
RETURNS (
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    TRANSFERKODU VARCHAR(55),
    ALISTRANSFERKODU VARCHAR(55),
    DEKONTTRANSFERKODU VARCHAR(55),
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZKURU DOUBLE PRECISION,
    KDVORANI DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    BIRIMFIYATTL DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    TUTARTL DOUBLE PRECISION,
    VADEGUN INTEGER,
    BIRIMISKONTOPARASAL DOUBLE PRECISION,
    BIRIMISKONTOORANI1 DOUBLE PRECISION,
    BIRIMISKONTOORANI2 DOUBLE PRECISION,
    BIRIMTRANSFERKODU VARCHAR(10),
    MUHASEBEHESAPKODU VARCHAR(30),
    KDVMUHASEBEHESAPKODU VARCHAR(30),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    DEKONT INTEGER,
    ACIKLAMA VARCHAR(255),
    DOVIZKURU DOUBLE PRECISION,
    IRSALIYETARIHI TIMESTAMP,
    HAREKETSEKLI VARCHAR(20),
    HAMSTOKMASTERID INTEGER,
    ODEMESEKLITRANSFERKODU VARCHAR(10),
    FATURABIRIM INTEGER,
    BOYASIPARISNOSTR VARCHAR(30),
    PARTINOSTR VARCHAR(30),
    SIPARISIALANKULLANICIID INTEGER)
AS
declare variable FaturaTarihi       TimeStamp;
declare variable SistemDovizId      varchar(5);
declare variable HamTipiId          integer;
declare variable HamStokHareketTipiId integer;
declare variable MuhAlisHesapKodu       varchar(30);
declare variable MuhSatisIadeHesapKodu  varchar(30);
declare variable DokumaKumasiKilodanTakipEt integer;
declare variable FDevirPartiYil  Integer;
declare variable FDevirPartiID  Integer;
declare variable FDevirIslemGrubuID  Integer;
declare variable FDevirDesenID varchar(20);
declare variable EgemenFirmaID  Integer;
declare variable BoyaliIplikMSatisTipi   varchar(20);
declare variable HamKumasMSatisTipi      varchar(20);
declare variable BoyaliKumasMSatisTipi   varchar(20);
declare variable BaskiliKumasMSatisTipi  varchar(20);

declare variable AlisBoyaliTransferKodu        varchar(55);
declare variable AlisSekerKasarTransferKodu    varchar(55);
declare variable AlisOptikBeyazTransferKodu    varchar(55);
declare variable AlisYikamaTransferKodu        varchar(55);
declare variable FRenk varchar(20);
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable FIslemDovizKuru  Double Precision;
declare variable FFirmaID  varchar(20);
declare variable IplikFirmaKodumuz varchar(20);
declare variable FBoyaSiparisYil Integer;
declare variable FBoyaSiparisID Integer;
begin
  Select SistemDovizId, DokumaKumasiKilodanTakipEt, EgemenFirmaID, BoyaliIplikMSatisTipi, HamKumasMSatisTipi, BoyaliKumasMSatisTipi, BaskiliKumasMSatisTipi, IplikFirmaKodumuz from Sistem where sistemId = 1
  into :SistemDovizId, :DokumaKumasiKilodanTakipEt, EgemenFirmaID, :BoyaliIplikMSatisTipi, :HamKumasMSatisTipi, :BoyaliKumasMSatisTipi, :BaskiliKumasMSatisTipi, :IplikFirmaKodumuz;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  for select Ham.Adi, Ham.HamSatisTransferKodu, HamStokMaster.Miktar, HamStokMaster.BirimFiyat, HamStokFatura.DovizId, HamStokMaster.Kdv,
             HamStokFatura.HamStokHareketTipiId, HamStokFatura.FaturaTarihi, HamStokFatura.DovizId, HamStokMaster.Vadesi, HamStokMaster.BirimIskontoParasal,
             HamStokMaster.BirimIskontoOrani1, HamStokMaster.BirimIskontoOrani2, Ham.HamTipiId,
             Ham.MuhAlisHesapKodu, Ham.MuhSatisIadeHesapKodu, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo,
             Ham.AlisTransferKodu, HamStokFatura.Dekont, Ham.DekontTransferKodu, HamStokMaster.DevirPartiYil, HamStokMaster.DevirPartiID,
             HamStokMaster.IslemGrubuID, HamStokMaster.DesenID, Ham.AlisBoyaliTransferKodu, Ham.AlisSekerKasarTransferKodu,
             Ham.AlisOptikBeyazTransferKodu, Ham.AlisYikamaTransferKodu, HamStokMaster.Renk, HamStokMaster.Aciklama, HamStokIrsaliye.IrsaliyeTarihi, HamStokFatura.IslemDovizId,
             iif(HamStokFatura.IslemDovizKuru1 > HamStokFatura.IslemDovizKuru2, HamStokFatura.IslemDovizKuru1, HamStokFatura.IslemDovizKuru2), (HamStokFatura.IslemDovizKuru2 / HamStokFatura.IslemDovizKuru1),
             HamStokMaster.HamStokMasterId, HamStokFatura.FirmaID, BoyaSiparisDetay.HamStokBirimId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, BoyaSiparis.SiparisiAlanKullaniciId, Ham.Adi2 from HamStokMaster
    left join HamStokFatura on HamStokFatura.HamStokFaturaId = HamStokMaster.HamStokFaturaId
    Left Join Ham on Ham.HamId = HamStokMaster.HamId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = HamStokMaster.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = HamStokMaster.BoyaSiparisKayitID
    where HamStokMaster.HamStokFaturaId = :AHamStokFaturaId order by HamStokMasterId
    into :HamAdi, :TransferKodu, :Miktar, :BirimFiyat, :DovizId, :KdvOrani, :HamStokHareketTipiId, :FaturaTarihi, :DovizId, :VadeGun, :BirimIskontoParasal,
         :BirimIskontoOrani1, :BirimIskontoOrani2, :HamTipiId,
         :MuhAlisHesapKodu, :MuhSatisIadeHesapKodu, :IrsaliyeSeriNo, :IrsaliyeNo,
         :AlisTransferKodu, :Dekont, :DekontTransferKodu, :FDevirPartiYil, :FDevirPartiID, :FDevirIslemGrubuID, :FDevirDesenID,
         :AlisBoyaliTransferKodu, :AlisSekerKasarTransferKodu, :AlisOptikBeyazTransferKodu, :AlisYikamaTransferKodu, FRenk, :Aciklama, :IrsaliyeTarihi, :IslemDovizId, :IslemDovizKuru ,
         :FIslemDovizKuru, :HamStokMasterId, :FFirmaID, :FaturaBirim, :FBoyaSiparisYil, :FBoyaSiparisID, :SiparisiAlanKullaniciId, :HamAdi2 do
  begin
    if (FBoyaSiparisYil is not null) then BoyaSiparisNoStr = EgemenCopyStr(EgemenIntegerToStr(FBoyaSiparisYil),3,2) || '-' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
    else BoyaSiparisNoStr = '';


    if (DovizID <> SistemDovizId) then
    begin
      select DovizKuru from DovizKuruGetir(:DovizId, :SistemDovizId, :FaturaTarihi) into :DovizKuru;
    end
    else DovizKuru = 1;

    BirimFiyatTL = BirimFiyat * DovizKuru;

    Tutar = BirimFiyat * Miktar;

    IslemDovizToplam = Tutar * FIslemDovizKuru;

    TutarTL = BirimFiyatTL * Miktar;
    BirimTransferKodu = '';

    IndirilecekKdv = 0;
    TahsilEdilecekKdv = 0;

    TahsilEdilecekKdv = TutarTL * KdvOrani / 100;

    if (HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then Select TransferKodu from Birim where Birim.BirimID = 'Kg' into :BirimTransferKodu;
    else Select TransferKodu from Birim where Birim.BirimID = 'Mt' into :BirimTransferKodu;

    KdvMuhasebeHesapKodu = Null;
    MuhasebeHesapKodu = Null;
    HareketSekli = Null;

    if (EgemenFirmaID = 34) then /* örkum */
    begin
      TransferKodu = AlisTransferKodu;
      if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
      begin
        HareketSekli = 'HAMIPLIK';
      end
      else if (FRenk <> Null and FRenk <> '') then
      begin
        HareketSekli = 'HAMBOYALIKUMAS';
      end
      else
      begin
        HareketSekli = 'HAMKUMAS';
      end
    end

    if (EgemenFirmaID = 46) then /* Başyazıcı */
    begin
      if (IplikFirmaKodumuz <> FFirmaID) then  TransferKodu = AlisTransferKodu;
    end

    if (HamStokHareketTipiId = 0) then
    begin
      MuhasebeHesapKodu = MuhAlisHesapKodu;
      Select MuhAlisHesapKodu from KdvHesapTablosu where KdvHesapTablosu.KdvOrani = :KdvOrani into :KdvMuhasebeHesapKodu;
    end
    else if (HamStokHareketTipiId = 1) then
    begin
      MuhasebeHesapKodu = MuhSatisIadeHesapKodu;
      Select MuhSatisIadeHesapKodu from KdvHesapTablosu where KdvHesapTablosu.KdvOrani = :KdvOrani into :KdvMuhasebeHesapKodu;
    end

    if (Dekont = 1) then
      AlisTransferKodu = DekontTransferKodu;

    suspend;
  end
end
