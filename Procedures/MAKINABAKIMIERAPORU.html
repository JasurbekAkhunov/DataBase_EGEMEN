CREATE PROCEDURE MAKINABAKIMIERAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINABAKIMTIPIID INTEGER,
    AMAKINABAKIMBOLUMUID INTEGER,
    AMAKINAGRUPID VARCHAR(4),
    AMAKINAID VARCHAR(10),
    AORMEMAKINAID VARCHAR(10),
    AIPLIKMAKINAID VARCHAR(10),
    ADOKUMAMAKINAID VARCHAR(10),
    ABOYA INTEGER,
    AORME INTEGER,
    AIPLIK INTEGER,
    ADOKUMA INTEGER,
    AIESTARTBASLANGICTARIHI TIMESTAMP,
    AIESTARTBITISTARIHI TIMESTAMP,
    ABSTARTBASLANGICTARIHI TIMESTAMP,
    ABSTARTBITISTARIHI TIMESTAMP,
    ABENDBASLANGICTARIHI TIMESTAMP,
    ABENDBITISTARIHI TIMESTAMP,
    AIEENDBASLANGICTARIHI TIMESTAMP,
    AIEENDBITISTARIHI TIMESTAMP,
    ATAHMINIBASLANGICTARIHI TIMESTAMP,
    ATAHMINIBITISTARIHI TIMESTAMP,
    ABASLAYANLAR INTEGER,
    ABAKIMBASLAYANLAR INTEGER,
    ABAKIMBITENLER INTEGER,
    ABITENLER INTEGER,
    AARIZAISEMRI INTEGER,
    ASISTEMOLUSTURDU INTEGER,
    ABASLANGICTAHMINISAAT INTEGER,
    ABITISTAHMINISAAT INTEGER,
    AKAPALILARIDOK INTEGER,
    ABAKIMDAIRESIID INTEGER,
    AHIZMETID VARCHAR(20),
    AURUNID VARCHAR(20))
RETURNS (
    MAKINABAKIMISEMRIID INTEGER,
    ADI VARCHAR(255),
    MAKINABAKIMTIPIID INTEGER,
    MAKINABAKIMTIPIADI VARCHAR(255),
    MAKINABAKIMBOLUMUID INTEGER,
    MAKINABAKIMBOLUMUADI VARCHAR(255),
    TAHMINIBASLANGIC TIMESTAMP,
    TAHMINISAAT DOUBLE PRECISION,
    IESTARTKULLANICIID INTEGER,
    IESTARTKULLANICIADI VARCHAR(40),
    IESTARTTARIHI TIMESTAMP,
    BAKIMSTARTKULLANICIID INTEGER,
    BAKIMSTARTKULLANICIADI VARCHAR(40),
    BAKIMSTARTTARIHI TIMESTAMP,
    BAKIMENDKULLANICIID INTEGER,
    BAKIMENDKULLANICIADI VARCHAR(40),
    BAKIMENDTARIHI TIMESTAMP,
    IEENDKULLANICIID INTEGER,
    IEENDKULLANICIADI VARCHAR(40),
    IEENDTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    IPLIKMAKINAID VARCHAR(10),
    IPLIKMAKINAADI VARCHAR(40),
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    ARIZAISEMRI INTEGER,
    ARIZAISEMRISTR VARCHAR(20),
    SISTEMOLUSTURDU INTEGER,
    SISTEMOLUSTURDUSTR VARCHAR(20),
    KAPALI INTEGER,
    KAPALISTR VARCHAR(6),
    INSERTKULLANICIID INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    BOLUMSTR VARCHAR(6),
    PLANLANANBASLANGIC TIMESTAMP,
    TOLERANS DOUBLE PRECISION,
    BAKIMDAIRESIID INTEGER,
    BAKIMDAIRESIADI VARCHAR(255),
    SATINALMATALEPKAYITID INTEGER,
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30))
AS
begin
  for select Distinct MIE.MakinaBakimIsEmriId, MIE.Adi, MIE.MakinaBakimTipiId, MIE.MakinaBakimBolumuId, MIE.TahminiBaslangic, MIE.TahminiSaat,
            MIE.IEStartKullaniciId, MIE.IEStartTarihi, MIE.BakimStartKullaniciId, MIE.BakimStartTarihi, MIE.BakimEndKullaniciId, MIE.BakimEndTarihi,
            MIE.IEEndKullaniciId, MIE.IEEndTarihi, MIE.Notu, MIE.MakinaId, MIE.OrmeMakinaId, MIE.IplikMakinaId, MIE.DokumaMakinaId, MIE.ArizaIsEmri, MIE.SistemOlusturdu, MIE.Kapali, MIE.InsertKullaniciId,
            MIE.InsertTarihi, MIE.KullaniciId, MIE.DegisimTarihi,
            MakinaBakimTipi.Adi, MakinaBakimBolumu.Adi, IEStartKullanici.Adi, BakimStartKullanici.Adi, BakimEndKullanici.Adi, IEEndKullanici.Adi,
            Makina.Adi, OrmeMakina.Adi, IplikMakina.Adi, DokumaMakina.Adi, InsertKullanici.Adi, Kullanici.Adi, MIE.PlanlananBaslangic, MIE.Tolerans,
            MIE.BakimDairesiId, BakimDairesi.Adi, MIE.SatinAlmaTalepKayitId, SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId
            from MakinaBakimIsEmri MIE
  left join MakinaBakimTipi on MakinaBakimTipi.MakinaBakimTipiId = MIE.MakinaBakimTipiId
  left join MakinaBakimBolumu on MakinaBakimBolumu.MakinaBakimBolumuId = MIE.MakinaBakimBolumuId
  left join BakimDairesi on BakimDairesi.BakimDairesiId = MIE.BakimDairesiId
  left join Makina on Makina.MakinaId = MIE.MakinaId
  left join OrmeMakina on OrmeMakina.OrmeMakinaId = MIE.OrmeMakinaId
  left join IplikMakina on IplikMakina.IplikMakinaId = MIE.IplikMakinaId
  left join DokumaMakina on DokumaMakina.DokumaMakinaId = MIE.DokumaMakinaId
  left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = MIE.SatinAlmaTalepKayitId
  left join Kullanici IEStartKullanici on IEStartKullanici.KullaniciId = MIE.IEStartKullaniciId
  left join Kullanici BakimStartKullanici on BakimStartKullanici.KullaniciId = MIE.BakimStartKullaniciId
  left join Kullanici BakimEndKullanici on BakimEndKullanici.KullaniciId = MIE.BakimEndKullaniciId
  left join Kullanici IEEndKullanici on IEEndKullanici.KullaniciId = MIE.IEEndKullaniciId
  left join Kullanici InsertKullanici on InsertKullanici.KullaniciId = MIE.InsertKullaniciId
  left join Kullanici on Kullanici.KullaniciId = MIE.KullaniciId
  left join MakinaBakimIEHizmet on MakinaBakimIEHizmet.MakinaBakimIsEmriId = MIE.MakinaBakimIsEmriId
  left join MakinaBakimIEUrun on MakinaBakimIEUrun.MakinaBakimIEHizmetID = MakinaBakimIEHizmet.MakinaBakimIEHizmetID
  where ((MIE.InsertTarihi >= :ABaslangicTarihi) or :ABaslangicTarihi is Null) and
        ((MIE.InsertTarihi <= :ABitisTarihi) or :ABitisTarihi is Null) and
        (MIE.MakinaBakimTipiId = :AMakinaBakimTipiId or :AMakinaBakimTipiId is null) and
        (MIE.MakinaBakimBolumuId = :AMakinaBakimBolumuId or :AMakinaBakimBolumuId is null) and
        (MIE.MakinaId is Null or Makina.MakinaGrubu = :AMakinaGrupId or :AMakinaGrupId is null) and
        (MIE.OrmeMakinaId is Null or OrmeMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
        (MIE.IplikMakinaId is Null or IplikMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
        (MIE.DokumaMakinaId is Null or DokumaMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
        (MIE.MakinaId = :AMakinaId or :AMakinaId is null) and
        (MIE.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
        (MIE.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null) and
        (MIE.DokumaMakinaId = :ADokumaMakinaId or :ADokumaMakinaId is null) and
        ((MIE.MakinaId is Null and :ABoya = 0) or :ABoya = 1) and
        ((MIE.OrmeMakinaId is Null and :AOrme = 0) or :AOrme = 1) and
        ((MIE.IplikMakinaId is Null and :AIplik = 0) or :AIplik = 1) and
        ((MIE.DokumaMakinaId is Null and :ADokuma = 0) or :ADokuma = 1) and
        ((MIE.IEStartTarihi >= :AIEStartBaslangicTarihi) or :AIEStartBaslangicTarihi is Null) and
        ((MIE.IEStartTarihi <= :AIEStartBitisTarihi) or :AIEStartBitisTarihi is Null) and
        ((MIE.BakimStartTarihi >= :ABStartBaslangicTarihi) or :ABStartBaslangicTarihi is Null) and
        ((MIE.BakimStartTarihi <= :ABStartBitisTarihi) or :ABStartBitisTarihi is Null) and
        ((MIE.BakimEndTarihi >= :ABEndBaslangicTarihi) or :ABEndBaslangicTarihi is Null) and
        ((MIE.BakimEndTarihi <= :ABEndBitisTarihi) or :ABEndBitisTarihi is Null) and
        ((MIE.IEEndTarihi >= :AIEEndBaslangicTarihi) or :AIEEndBaslangicTarihi is Null) and
        ((MIE.IEEndTarihi <= :AIEEndBitisTarihi) or :AIEEndBitisTarihi is Null) and
        ((MIE.TahminiBaslangic >= :ATahminiBaslangicTarihi) or :ATahminiBaslangicTarihi is Null) and
        ((MIE.TahminiBaslangic <= :ATahminiBitisTarihi) or :ATahminiBitisTarihi is Null) and
        ((MIE.IEStartKullaniciId is not Null and :ABaslayanlar = 1) or (MIE.IEStartKullaniciId is Null and :ABaslayanlar = 0) or :ABaslayanlar is null) and
        ((MIE.BakimStartKullaniciId is not Null and :ABakimBaslayanlar = 1) or (MIE.BakimStartKullaniciId is Null and :ABakimBaslayanlar = 0) or :ABakimBaslayanlar is null) and
        ((MIE.BakimEndKullaniciId is not Null and :ABakimBitenler = 1) or (MIE.BakimEndKullaniciId is Null and :ABakimBitenler = 0) or :ABakimBitenler is null) and
        ((MIE.IEEndKullaniciId is not Null and :ABitenler = 1) or (MIE.IEEndKullaniciId is Null and :ABitenler = 0) or :ABitenler is null) and
        (MIE.ArizaIsEmri = :AArizaIsEmri or :AArizaIsEmri is null) and
        (MIE.SistemOlusturdu = :ASistemOlusturdu or :ASistemOlusturdu is null) and
        ((MIE.TahminiSaat >= :ABaslangicTahminiSaat) or :ABaslangicTahminiSaat is Null) and
        ((MIE.TahminiSaat <= :ABitisTahminiSaat) or :ABitisTahminiSaat is Null) and
        ((MIE.Kapali = :AKapalilariDok) or :AKapalilariDok is Null) and
        (MakinaBakimIEHizmet.UrunID = :AHizmetID or :AHizmetID is null) and
        (MakinaBakimIEUrun.UrunID = :AUrunID or :AUrunID is null)
  into :MakinaBakimIsEmriId, :Adi, :MakinaBakimTipiId, :MakinaBakimBolumuId, :TahminiBaslangic, :TahminiSaat,
          :IEStartKullaniciId, :IEStartTarihi, :BakimStartKullaniciId, :BakimStartTarihi, :BakimEndKullaniciId, :BakimEndTarihi,
          :IEEndKullaniciId, :IEEndTarihi, :Notu, :MakinaId, :OrmeMakinaId, :IplikMakinaId, :DokumaMakinaId, :ArizaIsEmri, :SistemOlusturdu, :Kapali, :InsertKullaniciId,
          :InsertTarihi, :KullaniciId, :DegisimTarihi,
          :MakinaBakimTipiAdi, :MakinaBakimBolumuAdi, :IEStartKullaniciAdi, :BakimStartKullaniciAdi, :BakimEndKullaniciAdi, IEEndKullaniciAdi,
          :MakinaAdi, :OrmeMakinaAdi, :IplikMakinaAdi, :DokumaMakinaAdi, :InsertKullaniciAdi, :KullaniciAdi, :PlanlananBaslangic, :Tolerans,
          :BakimDairesiId, :BakimDairesiAdi, :SatinAlmaTalepKayitId, :SatinAlmaTalepYil, :SatinAlmaTalepId
          do
  begin
    if (ArizaIsEmri = 1) then ArizaIsEmriStr = 'Ariza';
    else ArizaIsEmriStr = 'Planlı bakım';

    if (SistemOlusturdu = 1) then SistemOlusturduStr = 'Sistem oluşturdu';
    else ArizaIsEmriStr = 'Elle oluşturuldu';

    if (Kapali = 1) then KapaliStr = 'Kapali';
    else KapaliStr = '';

    if (MakinaId is not Null) then BolumStr = 'Boya';
    if (OrmeMakinaId is not Null) then BolumStr = 'Örme';
    if (IplikMakinaId is not Null) then BolumStr = 'İplik';
    if (DokumaMakinaId is not Null) then BolumStr = 'Dokuma';

    SatinAlmaTalepIdStr = '';
    if (SatinAlmaTalepYil is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId)  || ' ';

    suspend;
  end
end
