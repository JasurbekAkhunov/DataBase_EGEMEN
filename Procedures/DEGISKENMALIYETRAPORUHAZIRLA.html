CREATE PROCEDURE DEGISKENMALIYETRAPORUHAZIRLA(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNTIPIID INTEGER,
    MIKTAR DOUBLE PRECISION,
    BIRIMID VARCHAR(2),
    BIRIMFIYAT DOUBLE PRECISION)
AS
declare variable PartiKayitId integer;
begin
  For Select BoyaIrsaliyeDetay.PartiKayitId from BoyaFatura
    left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaFaturaId = BoyaFatura.BoyaFaturaId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayId
    where BoyaFatura.FaturaTarihi >= :ABaslangicTarihi and BoyaFatura.FaturaTarihi <= :ABitisTarihi and
         (BoyaFatura.FirmaID = :AFirmaId or :AFirmaId is null) and BoyaIrsaliyeDetay.PartiKayitId is not null
    group by BoyaIrsaliyeDetay.PartiKayitId
    into :PartiKayitId do
  begin
    for select StokDetay.UrunId, Urun.Adi, Urun.UrunTipiId, DUB.BirimId, StokDetay.Miktar / HUB.Carpan * DUB.Carpan, StokDetay.BirimFiyat * HUB.Carpan / DUB.Carpan from StokDetay
      left join Urun on Urun.UrunId = StokDetay.UrunId
      left join UrunBirim HUB on HUB.UrunBirimId = StokDetay.UrunBirimId
      left join UrunBirim DUB on DUB.UrunId = stokdetay.Urunid and DUB.DefaultBirim = 1
      where StokDetay.PartiKayitId = :PartiKayitId and StokDetay.HareketTipi = 6
      into :UrunId, :UrunAdi, :UrunTipiId, :BirimId, :Miktar, :BirimFiyat do
    begin
      suspend;
    end
  end
end
