CREATE PROCEDURE AVUKATCARIHAREKETIGETIR(
    AAVUKATID VARCHAR(10),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1))
RETURNS (
    AVUKATCARIID INTEGER,
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    BORCALACAK VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    SINIF VARCHAR(1),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSID INTEGER,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    BORCBAKIYE DOUBLE PRECISION,
    ALACAKBAKIYE DOUBLE PRECISION)
AS
declare Variable ToplamBorc double precision;
declare variable ToplamAlacak double precision;
declare variable CsHReferansId integer;
declare variable CsTReferansId integer;
begin
  ToplamBorc = 0;
  ToplamAlacak = 0;
  select  sum(Meblag)  from AvukatCari where AvukatId = :AAvukatId and tarih < :ABaslangicTarihi and BorcAlacak = 'A' and (AvukatCari.Sinif = :ASinif or :ASinif is null) into :ToplamAlacak;
  select  sum(Meblag)  from AvukatCari where AvukatId = :AAvukatId and tarih < :ABaslangicTarihi and BorcAlacak = 'B' and (AvukatCari.Sinif = :ASinif or :ASinif is null) into :ToplamBorc;


  ReferansId = -1;
  Tarih  = null;
  Aciklama = 'Devir';
  BorcAlacak = null;
  Meblag = null;
  Sinif = Null;
  Notu = null;
  KullaniciAdi = null;
  DegisimTarihi = null;
  Borc = null;
  Alacak = Null;
  BorcBakiye = null;
  AlacakBakiye = Null;
  if (ToplamBorc > 0) then Borc = ToplamBorc;
  if (ToplamAlacak > 0) then Alacak = ToplamAlacak;

  if (toplamBorc is Null) then ToplamBorc = 0;
  if (ToplamAlacak is Null) then ToplamAlacak = 0;
  if (ToplamBorc > ToplamAlacak) then BorcBakiye = ToplamBorc - ToplamAlacak;
  else AlacakBakiye = ToplamAlacak - ToplamBorc;
  if (BorcBakiye = 0) then BorcBakiye = null;
  if (AlacakBakiye = 0) then AlacakBakiye = null;

  suspend;

  for select AvukatCariId,  Tarih, Aciklama, BorcAlacak, Sinif, Meblag, Notu, DegisimTarihi, Kullanici.Adi, CSHReferansId, CSTReferansId from AvukatCari
  left join Kullanici on Kullanici.KullaniciId = AvukatCari.KullaniciId
  where AvukatCari.AvukatId = :AAvukatId and AvukatCari.Tarih >= :ABaslangicTarihi and AvukatCari.Tarih <= :ABitisTarihi and (AvukatCari.Sinif = :ASinif or :ASinif is null)
  order by AvukatCari.Tarih, AvukatCari.AvukatCariId
  into :AvukatCariId, :Tarih, :Aciklama, :BorcAlacak, :Sinif, :Meblag, :Notu, :DegisimTarihi, :KullaniciAdi, :CshReferansId, :CstReferansId  do
  begin
    Borc = Null;
    Alacak = Null;
    BorcBakiye = Null;
    AlacakBakiye = Null;
    if (CshReferansId is not null) then ReferansId = CshReferansId;
    else if (CstReferansId is not null) then ReferansId = CstReferansId;

    if (BorcAlacak = 'B') then
    begin
      Borc = Meblag;
      ToplamBorc = ToplamBorc + Borc;
    end
    else
    begin
      Alacak = Meblag;
      ToplamAlacak = ToplamAlacak + Alacak;
    end
    if (toplamBorc is Null) then ToplamBorc = 0;
    if (ToplamAlacak is Null) then ToplamAlacak = 0;
    if (ToplamBorc > ToplamAlacak) then BorcBakiye = ToplamBorc - ToplamAlacak;
    else AlacakBakiye = ToplamAlacak - ToplamBorc;
    if (BorcBakiye = 0) then BorcBakiye = null;
    if (AlacakBakiye = 0) then AlacakBakiye = null;
    suspend;
  end
end
