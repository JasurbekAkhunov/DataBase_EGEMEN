CREATE PROCEDURE MC_BOYASIPARISRAPORU_1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AISLEMTIPIID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AMUSTERIRENKKODUVEADI VARCHAR(30),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISNOSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPARISTARIHI TIMESTAMP,
    SIPARISIVEREN VARCHAR(50),
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    ODEMESEKLIADI VARCHAR(40),
    TESLIMSEKLIADI VARCHAR(40),
    ISLEMTIPIADI VARCHAR(20),
    SIPARISTIPIID INTEGER)
AS
begin
  For Select DISTINCT MC_BoyaSiparisRaporu_2.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi,
             BoyaSiparis.SiparisTarihi, FirmaYetkili.Adi, BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi,
             BoyaSiparis.Fason, OdemeSekli.Adi, TeslimSekli.Adi,
             BoyaSiparis.SiparisTipiId, IslemTipi.Adi from MC_BoyaSiparisRaporu_2(:ABaslangicTarihi, :ABitisTarihi, :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AIslemTipiId, :ASatisTipi, :ABoyaSiparisTipiId, :ALabReceteId, :AMusteriSiparisNo, :AMusteriRenkKoduveAdi, :AOnayDurumu, :APartilenen, :AKapaliDurumu, :AKullaniciId)
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = MC_BoyaSiparisRaporu_2.BoyaSiparisKayitId
    left join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
    left join OdemeSekli On OdemeSekli.OdemeSekliId = BoyaSiparis.OdemeSekliId
    left join TeslimSekli On TeslimSekli.TeslimSekliId = BoyaSiparis.TeslimSekliId
    left join IslemTipi On IslemTipi.IslemTipiId = BoyaSiparis.IslemTipiId
    Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
   into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :SiparisTarihi, :SiparisiVeren, :SiparisiAlanKullaniciId,
        :SiparisiAlanKullaniciAdi, :Fason, :OdemeSekliAdi, :TeslimSekliAdi,
        :SiparisTipiId, :IslemTipiAdi do
   begin
     if (Fason = 1) then FasonStr = 'Fason';
     else FasonStr = '';

     BoyaSiparisNoStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
     suspend;
   end
end
