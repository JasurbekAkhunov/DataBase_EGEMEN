CREATE PROCEDURE DEPARTMANONAYIBITMISHSKABUL(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AORMEGIRISIRSALIYENO IRSALIYENOSIZE,
    AHAMSTOKMASTERID INTEGER,
    AFIRMAID VARCHAR(20),
    AREFERANSSIPARISNO VARCHAR(30),
    AREFERANSFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    AFASON INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER)
RETURNS (
    HAMSTOKMASTERKABULID INTEGER,
    HAMSTOKMASTERID INTEGER,
    DEPARTMANID INTEGER,
    DEPARTMANADI VARCHAR(30),
    KABUL INTEGER,
    KABULSTR VARCHAR(5),
    KABULMIKTAR NUMERIC(12,3),
    RET INTEGER,
    RETSTR VARCHAR(3),
    RETMIKTAR NUMERIC(12,3),
    SARTLIKABUL INTEGER,
    SARTLIKABULSTR VARCHAR(13),
    SARTLIKABULMIKTAR NUMERIC(12,3),
    ZORUNLUKABUL INTEGER,
    ZORUNLUKABULSTR VARCHAR(18),
    ZORUNLUKABULMIKTAR NUMERIC(12,3),
    KONTROLDURUMU INTEGER,
    KONTROLKULLANICIADI VARCHAR(40),
    KONTROLTARIHI TIMESTAMP,
    KONTROLNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
begin
  for select HamStokMasterKabul.HamStokMasterKabulId, HamStokMasterKabul.HamStokMasterId, HamStokMasterKabul.DepartmanId, Departman.Adi,
             HamStokMasterKabul.Kabul, HamStokMasterKabul.KabulMiktar, HamStokMasterKabul.Ret, HamStokMasterKabul.RetMiktar,
             HamStokMasterKabul.SartliKabul, HamStokMasterKabul.SartliKabulMiktar, HamStokMasterKabul.ZorunluKabul, HamStokMasterKabul.ZorunluKabulMiktar,
             HamStokMasterKabul.KontrolDurumu, KK.Adi, HamStokMasterKabul.KontrolTarihi, HamStokMasterKabul.KontrolNotu from DepartmanOnayiBitmisHS(:ARaporTipi, :ABaslangicTarihi, :ABitisTarihi, :AIrsaliyeNo, :AOrmeGirisIrsaliyeNo, :AHamStokMasterId, :AFirmaId, :AReferansSiparisNo, :AReferansFirmaId, :AHamId, :AFason, :ABoyaSiparisYil, :ABoyaSiparisId)
  Left join HamStokMasterKabul on HamStokMasterKabul.HamStokMasterId = DepartmanOnayiBitmisHS.HamStokMasterId
  Left join Departman on Departman.DepartmanId = HamStokMasterKabul.DepartmanId
  Left Join Kullanici KK on KK.KullaniciId = HamStokMasterKabul.KontrolKullaniciId
  order by HamStokMasterKabul.HamStokMasterId, HamStokMasterKabul.SonKarar
  into :HamStokMasterKabulId, :HamStokMasterId, :DepartmanId, :DepartmanAdi, :Kabul, :KabulMiktar, :Ret, :RetMiktar, :SartliKabul, :SartliKabulMiktar,
       :ZorunluKabul, :ZorunluKabulMiktar, :KontrolDurumu, :KontrolKullaniciAdi, :KontrolTarihi, :KontrolNotu do
  begin
    if (Kabul = 1) then KabulStr = 'Kabul';
    else KabulStr = '';

    if (Ret = 1) then RetStr = 'Ret';
    else RetStr = '';

    if (SartliKabul = 1) then SartliKabulStr = 'Şartlı Kabul';
    else SartliKabulStr = '';

    if (ZorunluKabul = 1) then ZorunluKabulStr = 'Zorunlu Kabul';
    else ZorunluKabulStr = '';

    if (KabulMiktar = 0) then KabulMiktar = Null;
    if (RetMiktar = 0) then RetMiktar = Null;
    if (SartliKabulMiktar = 0) then SartliKabulMiktar = Null;
    if (ZorunluKabulMiktar = 0) then ZorunluKabulMiktar = Null;

    suspend;
  end
end
