CREATE PROCEDURE RECETESANALPARTIDOKUMU(
    ARECETESANALPARTIKAYITID INTEGER)
RETURNS (
    RECETEASAMAID INTEGER,
    RECETEASAMAADI VARCHAR(40),
    DOZAJLAMAPROGRAMNO INTEGER,
    SABLONDOZAJLAMAPROGRAMNO INTEGER,
    RECETEALTASAMAID INTEGER,
    RECETEALTASAMAADI VARCHAR(40),
    RECETEGRAFIKADI VARCHAR(40),
    RECETEASAMASABLONADI VARCHAR(40),
    BANYOCARPANNO INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNBIRIMID INTEGER,
    URETIMECIKISFIYATI DOUBLE PRECISION,
    URETIMECIKISFIYATIBUGUN DOUBLE PRECISION,
    URETIMECIKISFIYATIBIRIMID VARCHAR(2),
    URETIMECIKISFIYATIURUNBIRIMID INTEGER,
    URUNTIPIID INTEGER,
    BOYATIPIID INTEGER,
    KIMYASALTIPIID INTEGER,
    KIMYASALTIPIADI VARCHAR(20),
    BIRIMID VARCHAR(2),
    RECETEMIKTAR DOUBLE PRECISION,
    RECETEBANYO DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    MIKTARKG DOUBLE PRECISION,
    MIKTARGR DOUBLE PRECISION,
    MIKTARMG DOUBLE PRECISION,
    BOYASULUMIKTARKG DOUBLE PRECISION,
    BOYASULUMIKTARGR DOUBLE PRECISION,
    BOYASULUMIKTARMG DOUBLE PRECISION,
    BANYO DOUBLE PRECISION,
    BANYOKG DOUBLE PRECISION,
    BANYOGR DOUBLE PRECISION,
    BANYOMG DOUBLE PRECISION,
    RESIM BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    HESAPSTR VARCHAR(30),
    ACIKLAMA VARCHAR(255),
    BOYAASAMASIMI INTEGER,
    ASAMABOYATIPIID INTEGER)
AS
declare variable ReceteId        varchar(20);
declare variable FToplamKilo     Double Precision;
declare variable FBanyoCarpani   Double Precision;
declare variable FBoyaSuyu       Double Precision;
declare variable FKoyultma       Double Precision;
declare variable FReceteMiktarOrani     Double Precision;
declare variable FKimyasalTipiReceteOrani DOuble Precision;
declare variable FBoyaAsamasimi          integer;
declare variable FAsamaBoyaTipiId integer;
declare variable FReceteAsamaSablonAdi varchar(40);
declare variable FUrunTipiId            integer;
declare variable FBoyaTipiId            integer;
declare variable FFBoyaAsamasimi          integer;
declare variable FBanyoToplami          Double precision;
declare variable FReceteAsamaId           integer;
declare variable FReceteAltAsamaId        integer;
declare variable FLastReceteAsamaId           integer;
declare variable FLastReceteAltAsamaId        integer;

declare variable FReceteAsamaAdi      varchar(40);
declare variable FReceteAltAsamaAdi   varchar(40);
declare variable FReceteGrafikAdi     varchar(40);
declare variable FKimyasalTipiId      integer;
declare variable FKimyasalTipiAdi     varchar(20);
declare variable FUrunId              varchar(20);
declare variable FUrunAdi             varchar(80);
declare variable FBirimId              varchar(2);
declare variable FUrunBirimId          integer;
declare variable FReceteMiktar         Double precision;
declare variable FReceteBanyo          Double precision;
declare variable FMiktar               Double precision;
declare variable FBanyo                Double precision;
declare variable FResim                Blob;
declare variable Hesap                 Double precision;
declare variable FN                    Numeric(12,3);
declare variable FUretimeCikisFiyati   Double precision;
declare variable FUretimeCikisFiyatiDovizId  varchar(5);
declare variable FUretimeCikisFiyatiBirimId  varchar(2);
declare variable FUretimeCikisFiyatiUrunBirimId integer;
declare variable StokTabanDovizId               varchar(5);
declare variable SistemDovizId               varchar(5);
declare variable DovizKuru                   Double Precision;
declare variable ReceteTarihi                TimeStamp;
declare variable FAciklama  varchar(255);
begin
  FToplamKilo = 0;
  select RECETEMIKTARORANI, SistemDovizId, StokTabanDovizId from SISTEM where SISTEMID = 1 into :FReceteMiktarOrani, :SistemDovizId, :StokTabanDovizId;

  if (FReceteMiktarOrani is null or FReceteMiktarOrani = 0) then exception RecetemiktarOraniGirilmemis;

  Select PartiKilo, BanyoCarpani, BoyaSuyu, Koyultma, ReceteId, Tarih from ReceteSanalParti
    where ReceteSanalParti.ReceteSanalPartiKayitId = :AReceteSanalPartiKayitId
    into :FToplamKilo, :FBanyoCarpani, :FBoyaSuyu, :FKoyultma, :ReceteId, :ReceteTarihi;

  if (FBanyoCarpani is null or FBanyoCarpani = 0) then exception BanyoCarpaniGirilmemis;

  FToplamKilo = FToplamKilo / FReceteMiktarOrani;

  FBanyoToplami = 0;

  FLastReceteAsamaId = null;
  FLastReceteAltAsamaId = null;

  if (FKoyultma = 0 or FKoyultma is null) then FKoyultma = 1;
  if (FBoyaSuyu is null or FBoyaSuyu = 0) then FBoyaSuyu = 1;

  for select ReceteDetay.ReceteAsamaId, ReceteDetay.ReceteAltAsamaId, ReceteAsama.Adi,
         ReceteAsama.BoyaAsamasimi, ReceteAsama.BoyaTipiId,
         ReceteAsama.DozajlamaProgramNo, ReceteAsamaSablon.DozajlamaProgramNo,
         ReceteAltAsama.Adi, ReceteGrafik.Adi, ReceteAsamaSablon.Adi, ReceteGrafik.Resim, ReceteDetay.UrunId, Urun.Adi, ReceteDetay.UrunBirimId,
         RUB.BirimId, Urun.UrunTipiId, Urun.BoyaTipiId, Miktar,  (Miktar * :FToplamKilo * :FKoyultma) Miktar, Banyo,
         iif(Recete.LabReceteGrupId = 0, Banyo * :FBanyoCarpani * :FKoyultma, Banyo * :FBanyoCarpani) Banyo,
         Urun.KimyasalTipiId, KimyasalTipi.Adi, KimyasalTipi.ReceteOrani,
         UrunFiyat.UretimeCikisFiyati, UrunFiyat.UretimeCikisFiyatiDovizId, UrunFiyat.UretimeCikisFiyatiUrunBirimId, UBUCF.BirimId, ReceteDetay.Aciklama from ReceteDetay
  left join Recete on Recete.ReceteId = ReceteDetay.ReceteId
  left join Urun on Urun.UrunId = ReceteDetay.UrunId
  left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
  left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
  left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = Urun.KimyasalTipiId
  left join UrunBirim RUB on RUB.UrunBirimId = ReceteDetay.UrunBirimId and RUB.UrunId = ReceteDetay.UrunId
  left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
  left join ReceteAltAsama on ReceteAltAsama.ReceteAltAsamaId = ReceteDetay.ReceteAltAsamaId
  left join ReceteGrafik on ReceteGrafik.ReceteGrafikId = ReceteDetay.ReceteGrafikId
  left join ReceteAsamaSablon on ReceteAsamaSablon.ReceteAsamaSablonId = ReceteDetay.ReceteAsamaSablonId
  where ReceteDetay.ReceteID = :ReceteId
  order by sira
  into :FReceteAsamaId, :FReceteAltAsamaId, :ReceteAsamaAdi, :BoyaAsamasimi, :AsamaBoyaTipiId, :DozajlamaProgramNo, :SablonDozajlamaProgramNo, :ReceteAltAsamaAdi, :ReceteGrafikAdi, :ReceteAsamaSablonAdi, :Resim, :UrunId, :UrunAdi, :UrunBirimId, :BirimId,
       :UrunTipiId, :BoyaTipiId, :ReceteMiktar, :Miktar, :ReceteBanyo, :Banyo,
       :KimyasalTipiId, :KimyasalTipiAdi, :FKimyasalTipiReceteOrani,
       :UretimeCikisFiyati, :FUretimeCikisFiyatiDovizId, :UretimeCikisFiyatiUrunBirimId, :UretimeCikisFiyatiBirimId, :Aciklama do
  begin
    if (FReceteAsamaId is not null) then
    begin
      if (FLastReceteAsamaId is not null) then
      begin
        FReceteAsamaAdi = ReceteAsamaAdi;
        FReceteAltAsamaAdi = ReceteAltAsamaAdi;
        FBoyaAsamasimi = BoyaAsamasimi;
        FAsamaBoyaTipiId = AsamaBoyaTipiId;
        FReceteGrafikAdi = ReceteGrafikAdi;
        FReceteAsamaSablonAdi = ReceteAsamaSablonAdi;
        FUrunId = UrunId;
        FUrunAdi = UrunAdi;
        FUrunBirimId = UrunBirimId;
        FUrunTipiId = UrunTipiId;
        FBoyaTipiId = BoyaTipiId;
        FKimyasalTipiId = KimyasalTipiId;
        FKimyasalTipiAdi = KimyasalTipiAdi;
        FBirimId = BirimId;
        FReceteMiktar = ReceteMiktar;
        FReceteBanyo = ReceteBanyo;
        FMiktar = Miktar;
        FBanyo = Banyo;
        FResim = Resim;
        FUretimeCikisFiyati = UretimeCikisFiyati;
        FUretimeCikisFiyatiBirimId = UretimeCikisFiyatiBirimId;
        FUretimeCikisFiyatiUrunBirimId = UretimeCikisFiyatiUrunBirimId;
        FAciklama = Aciklama;

        ReceteAsamaAdi = Null;
        ReceteAltAsamaAdi = Null;
        BoyaAsamasimi = Null;
        AsamaBoyaTipiId = Null;

        ReceteGrafikAdi = Null;
        ReceteAsamaSablonAdi = Null;
        UrunId = Null;
        UrunAdi = Null;
        BirimId = Null;
        UrunBirimId = Null;
        UrunTipiID = Null;
        BoyaTipiID = Null;
        KimyasalTipiId = Null;
        KimyasalTipiAdi = Null;
        ReceteMiktar = Null;
        ReceteBanyo = Null;
        Miktar = Null;
        MiktarKg = Null;
        MiktarGr = Null;
        MiktarMg = Null;
        Banyo = Null;
        BanyoKg = Null;
        BanyoGr = Null;
        BanyoMg = Null;
        Resim = Null;
        UretimeCikisFiyati = Null;
        UretimeCikisFiyatiBugun = Null;
        UretimeCikisFiyatiBirimId = Null;
        UretimeCikisFiyatiUrunBirimId = Null;
        Aciklama = Null;
        ReceteAltAsamaId = FLastReceteAltAsamaId;
        ReceteAsamaId = FLastReceteAsamaId;
        suspend;

        ReceteAsamaAdi = FReceteAsamaAdi;
        ReceteAltAsamaAdi = FReceteAltAsamaAdi;
        BoyaAsamasimi = FBoyaAsamasimi;
        AsamaBoyaTipiId = FAsamaBoyaTipiId;
        ReceteGrafikAdi = FReceteGrafikAdi;
        ReceteAsamaSablonAdi = FReceteAsamaSablonAdi;
        UrunId = FUrunId;
        UrunAdi = FUrunAdi;
        UrunBirimId = FUrunBirimId;
        UrunTipiId = FUrunTipiId;
        BoyaTipiId = FBoyaTipiId;
        KimyasalTipiId = FKimyasalTipiId;
        KimyasalTipiAdi = FKimyasalTipiAdi;
        BirimId = FBirimId;
        ReceteMiktar = FReceteMiktar;
        ReceteBanyo = FReceteBanyo;
        Miktar = FMiktar;
        Banyo = FBanyo;
        Resim = FResim;
        UretimeCikisFiyati = FUretimeCikisFiyati;
        UretimeCikisFiyatiBugun = FUretimeCikisFiyati;
        UretimeCikisFiyatiBirimId = FUretimeCikisFiyatiBirimId;
        UretimeCikisFiyatiUrunBirimId = FUretimeCikisFiyatiUrunBirimId;
        Aciklama = FAciklama;
      end
      FLastReceteAsamaId = FReceteAsamaId;
    end
    ReceteAltAsamaId = FReceteAltAsamaId;
    ReceteAsamaId = FReceteAsamaId;

    if (FUretimeCikisFiyatiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :StokTabanDovizId, Current_Date) into :DovizKuru;
    else DovizKuru = 1;

    if (DovizKuru is null) then DovizKuru = 0;
    UretimeCikisFiyatiBugun = UretimeCikisFiyati * DovizKuru;

    if (FUretimeCikisFiyatiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :StokTabanDovizId, :ReceteTarihi) into :DovizKuru;
    else DovizKuru = 1;

    if (DovizKuru is null) then DovizKuru = 0;
    UretimeCikisFiyati = UretimeCikisFiyati * DovizKuru;

    MiktarKg = 0;
    MiktarGr = 0;
    MiktarMg = 0;

    FN = Miktar;
    FN = FN * 1000;
    Hesap = FN;
    Hesap = KusuratSil(Hesap,1);
    HesapStr = Format_float('', Hesap);

    if (Char_Length(HesapStr) >= 3) then
    begin
      MiktarMg = EgemenStrToInt(EgemenCopyStr(HesapStr, Char_Length(HesapStr) - 2, 3));

      if (Char_Length(HesapStr) >= 6) then
      begin
        MiktarGr = EgemenStrToInt(EgemenCopyStr(HesapStr, Char_Length(HesapStr) - 5, 3));

        if (Char_Length(HesapStr) > 6) then MiktarKg = EgemenStrToInt(EgemenCopyStr(HesapStr, 1, Char_Length(HesapStr) - 6));
      end
      else if (Char_Length(HesapStr) > 3) then MiktarGr = EgemenStrToInt(EgemenCopyStr(HesapStr, 1, Char_Length(HesapStr) - 3));
    end
    else MiktarMg = EgemenStrToInt(EgemenCopyStr(HesapStr, 1, Char_Length(HesapStr)));


    BanyoKg = 0;
    BanyoGr = 0;
    BanyoMg = 0;

    FN = Banyo;
    FN = FN * 1000;
    Hesap = FN;

    Hesap = KusuratSil(Hesap,1);

    HesapStr = Format_float('', Hesap);

    if (Char_Length(HesapStr) >= 3) then
    begin
      BanyoMg = EgemenStrToInt(EgemenCopyStr(HesapStr, Char_Length(HesapStr) - 2, 3));

      if (Char_Length(HesapStr) >= 6) then
      begin
        BanyoGr = EgemenStrToInt(EgemenCopyStr(HesapStr, Char_Length(HesapStr) - 5, 3));

        if (Char_Length(HesapStr) > 6) then BanyoKg = EgemenStrToInt(EgemenCopyStr(HesapStr, 1, Char_Length(HesapStr) - 6));
      end
      else if (Char_Length(HesapStr) > 3) then BanyoGr = EgemenStrToInt(EgemenCopyStr(HesapStr, 1, Char_Length(HesapStr) - 3));
    end
    else BanyoMg = EgemenStrToInt(EgemenCopyStr(HesapStr, 1, Char_Length(HesapStr)));

    suspend;
  end

  if (FLastReceteAsamaId is not null) then
  begin
    ReceteAsamaAdi = Null;
    ReceteAltAsamaAdi = Null;
    BoyaAsamasimi = Null;
    AsamaBoyaTipiId = Null;
    ReceteGrafikAdi = Null;
    ReceteAsamaSablonAdi = Null;
    UrunId = Null;
    UrunAdi = Null;
    UrunBirimId = Null;
    UrunTipiID = Null;
    BoyaTipiID = Null;
    KimyasalTipiId = Null;
    KimyasalTipiAdi = Null;
    BirimId = Null;
    ReceteMiktar = Null;
    ReceteBanyo = Null;
    Miktar = Null;
    MiktarKg = Null;
    MiktarGr = Null;
    MiktarMg = Null;
    Banyo = Null;
    BanyoKg = Null;
    BanyoGr = Null;
    BanyoMg = Null;
    Resim = Null;
    UretimeCikisFiyati = Null;
    UretimeCikisFiyatiBirimId = Null;
    UretimeCikisFiyatiBugun = Null;
    UretimeCikisFiyatiUrunBirimId = Null;
    Aciklama = Null;
    ReceteAltAsamaId = FLastReceteAltAsamaId;
    ReceteAsamaId = FLastReceteAsamaId;
    suspend;
  end
end
