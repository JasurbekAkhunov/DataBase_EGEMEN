CREATE PROCEDURE MC_SEVKEHAZIRLARINRAPORUROW_3(
    AFIRMAID VARCHAR(20),
    AKULLANICIID INTEGER,
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    SEVKEDILENKILO NUMERIC(12,3),
    SEVKEHAZIRKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    SEVKEDILENMETRE NUMERIC(12,3),
    SEVKEHAZIRMETRE NUMERIC(12,3),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    FIRMASEVKADRESIID INTEGER,
    FIRMASEVKADRESIADI VARCHAR(50))
AS
declare variable Okey     integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;
  for select PartiStok.PartiKayitId, PartiStok.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.BrutSevkKilo, (PartiStok.Kilo - PartiStok.BrutSevkKilo),
             PartiStok.Metre, PartiStok.BrutSevkMetre, (PartiStok.Metre - PartiStok.BrutSevkMetre),
             Parti.BoyaSiparisKayitId, PartiStok.BoyaSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.MusteriSiparisNo,
             BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
             PartiStok.FirmaSevkAdresiId, FirmaSevkAdresi.AdresAdi from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join FirmaSevkAdresi on FirmaSevkAdresi.FirmaSevkAdresiId = PartiStok.FirmaSevkAdresiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where (PartiStok.PartiKayitID = :APartiKayitID or :APartiKayitID is Null) and Parti.FirmaId = :AFirmaId and PartiStok.durumu = 7
    into :PartiKayitId, :PartiSinifId, :HamAdiToplami, :Kilo, :SevkEdilenKilo, :SevkeHazirKilo,
         :Metre, :SevkEdilenMetre, :SevkeHazirMetre,
         :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :MusteriSiparisNo,
         :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
         :FirmaSevkAdresiId, :FirmaSevkAdresiAdi do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    suspend;
  end
end
