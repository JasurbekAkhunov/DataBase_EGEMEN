CREATE PROCEDURE BOYASIPARISLISTESI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASIPARISSEKLI INTEGER,
    AISLEMTIPIID INTEGER,
    ASIPARISTIPIID INTEGER,
    AKAPALI INTEGER,
    AORME INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AIMALATTIPIID INTEGER,
    AORMESIPARISTIPIID INTEGER,
    ADOKUMASIPARISTIPIID INTEGER,
    AREFERANSSIPARISNO VARCHAR(30))
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISNOSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ORMESIPARISI INTEGER,
    ORMESIPARISISTR VARCHAR(20),
    DOKUMASIPARISI INTEGER,
    DOKUMASIPARISISTR VARCHAR(20),
    SIPARISTARIHI TIMESTAMP,
    SIPARISIVEREN VARCHAR(50),
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI VARCHAR(40),
    TESLIMSEKLIID INTEGER,
    TESLIMSEKLIADI VARCHAR(40),
    ISLEMTIPIADI VARCHAR(20),
    SIPARISTIPIID INTEGER,
    ORMESIPARISTIPISTR VARCHAR(30),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    SIPARISGRUP3ID VARCHAR(4),
    SIPARISGRUP3ADI VARCHAR(30),
    SIPARISGRUP4ID VARCHAR(4),
    SIPARISGRUP4ADI VARCHAR(30))
AS
begin
  For Select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi,
             BoyaSiparis.SiparisTarihi, FirmaYetkili.Adi, BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             BoyaSiparis.Fason, BoyaSiparis.OdemeSekliId, OdemeSekli.Adi, BoyaSiparis.TeslimSekliId, TeslimSekli.Adi, BoyaSiparis.OrmeSiparisi,
             BoyaSiparis.SiparisTipiId, IslemTipi.Adi, OrmeSiparisTipi.Adi, IK.Adi, BoyaSiparis.InsertTarihi, Sk.Adi, BoyaSiparis.DegisimTarihi,
             BoyaSiparis.DokumaSiparisi, BoyaSiparis.SiparisGrup3Id, SiparisGrup3.Adi, BoyaSiparis.SiparisGrup4Id, SiparisGrup4.Adi from BoyaSiparis
    left join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
    left join OdemeSekli On OdemeSekli.OdemeSekliId = BoyaSiparis.OdemeSekliId
    left join TeslimSekli On TeslimSekli.TeslimSekliId = BoyaSiparis.TeslimSekliId
    left join OrmeSiparisTipi On OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
    left join IslemTipi On IslemTipi.IslemTipiId = BoyaSiparis.IslemTipiId
    left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 On SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join SiparisGrup SiparisGrup3 On SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
    left join SiparisGrup SiparisGrup4 On SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
    left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
    Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    Left Join Kullanici IK on IK.KullaniciId = BoyaSiparis.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = BoyaSiparis.KullaniciId
    left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
    left join OrmeSiparisDetay On OrmeSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
    where BoyaSiparis.SiparisTarihi >= :ABaslangicTarihi and BoyaSiparis.SiparisTarihi <= :ABitisTarihi and
          ( BoyaSiparis.SiparisTipiId = :ASiparisTipiId or BoyaSiparis.SiparisTipiId = 2 or :ASiparisTipiId is null) and
          ((Firma.AnaHesapFirmaId = :AFirmaId) or :AFirmaId is null) and
          (BoyaSiparis.Fason = :ASiparisSekli or :ASiparisSekli is null) and
          (BoyaSiparis.IslemTipiId = :AIslemTipiId or :AIslemTipiId is null) and
          ((:AKapali is null) or (BoyaSiparisDetay.Kapali = :AKapali and :AOrme = 0) or (OrmeSiparisDetay.Kapali = :AKapali and (:AOrme = 1 or :AOrme = 2)) or
           (:AOrme is null and ((BoyaSiparisDetay.Kapali is not Null and BoyaSiparisDetay.Kapali = :AKapali) or (OrmeSiparisDetay.Kapali is not Null and OrmeSiparisDetay.Kapali = :AKapali))) ) and
          ((:AMusteriSiparisNo is null) or (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo and :AOrme = 0) or ( OrmeSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo and (:AOrme = 1 or :AOrme = 2)) or
           (:AOrme is null and ((BoyaSiparisDetay.MusteriSiparisNo is not Null and BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo) or (OrmeSiparisDetay.MusteriSiparisNo is not Null and OrmeSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo) )) ) and

          ((:AReferansSiparisNo is null) or (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo and :AOrme = 0) or ( OrmeSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo and (:AOrme = 1 or :AOrme = 2)) or
           (:AOrme is null and ((BoyaSiparisDetay.ReferansSiparisNo is not Null and BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo) or (OrmeSiparisDetay.ReferansSiparisNo is not Null and OrmeSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo) )) ) and
          ( (BoyaSiparis.OrmeSiparisi = 1 and :AOrme = 1) or
            (BoyaSiparis.DokumaSiparisi = 1 and :AOrme = 2) or
            ((BoyaSiparis.OrmeSiparisi = 0 or BoyaSiparisDetay.BoyaSiparisDetayID is not null) and :AOrme = 0) or
            :AOrme is null ) and
          ((:AImalatTipiID = 0 and BoyaSiparisDetay.DesenID is Null) or (:AImalatTipiID = 1 and BoyaSiparisDetay.DesenID is not Null) or  (:AImalatTipiID is Null)) and
          ( BoyaSiparis.OrmeSiparisTipiId = :AOrmeSiparisTipiId or :AOrmeSiparisTipiId is null) and
          ( BoyaSiparis.DokumaSiparisTipiId = :ADokumaSiparisTipiId or :ADokumaSiparisTipiId is null)
    group By BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi,
             BoyaSiparis.SiparisTarihi, FirmaYetkili.Adi, BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             BoyaSiparis.Fason, BoyaSiparis.OdemeSekliId, OdemeSekli.Adi, BoyaSiparis.TeslimSekliId, TeslimSekli.Adi, BoyaSiparis.OrmeSiparisi,
             BoyaSiparis.SiparisTipiId, IslemTipi.Adi, OrmeSiparisTipi.Adi, IK.Adi, BoyaSiparis.InsertTarihi, Sk.Adi, BoyaSiparis.DegisimTarihi, BoyaSiparis.DokumaSiparisi,
             BoyaSiparis.SiparisGrup3Id, SiparisGrup3.Adi, BoyaSiparis.SiparisGrup4Id, SiparisGrup4.Adi
    into :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :SiparisTarihi, :SiparisiVeren, :SiparisiAlanKullaniciId,
         :SiparisiAlanKullaniciAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :Fason, :OdemeSekliId, :OdemeSekliAdi, :TeslimSekliId,
         :TeslimSekliAdi, :OrmeSiparisi, :SiparisTipiId, :IslemTipiAdi, :OrmeSiparisTipiStr, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
         :DokumaSiparisi, :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi do
   begin
     if (Fason = 1) then FasonStr = 'Fason';
     else FasonStr = '';

     if (OrmeSiparisi = 1) then OrmeSiparisiStr = 'Örme Siparişi';
     else OrmeSiparisiStr = '';

     if (DokumaSiparisi = 1) then DokumaSiparisiStr = 'Dokuma Siparişi';
     else DokumaSiparisiStr = '';

     BoyaSiparisNoStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
     suspend;
   end
end
