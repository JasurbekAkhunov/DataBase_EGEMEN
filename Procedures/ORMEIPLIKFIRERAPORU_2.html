CREATE PROCEDURE ORMEIPLIKFIRERAPORU_2(
    ABASLANGICGIRISTARIHI TIMESTAMP,
    ABITISGIRISTARIHI TIMESTAMP,
    AHAMID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ARAFID VARCHAR(10),
    ADEPOID VARCHAR(10),
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    ASTOKGIRISNO INTEGER,
    ALOT VARCHAR(150))
RETURNS (
    ORMERAFID INTEGER,
    STOKHAREKETID INTEGER,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    MIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    BIRIMFIYAT DOUBLE PRECISION,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISSIRANO INTEGER,
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
/* girisleri */
  For Select OrmeIplikFireRaporu_1.OrmeRafId, StokHareket.StokHareketId, StokHareket.HareketTipi, StokHareket.FirmaId, SHFirma.Adi,
             OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeFatura.Faturano, OrmeFatura.FaturaTarihi,
             OrmeIrsaliye.FiiliSevkTarihi, OrmeStokHareketTipi.Adi,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.FirmaId, SiparisFirma.Adi,
             StokHareket.BirimFiyat, Ik.Adi, StokHareket.InsertTarihi, Sk.Adi, StokHareket.DegisimTarihi,
             Sum(StokHareket.Miktar), Sum(StokHareket.BrutMiktar) from OrmeIplikFireRaporu_1(:ABaslangicGirisTarihi, :ABitisGirisTarihi, :AHamId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id,
                                 :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id, :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ARafId, :ADepoId, :AIplikCinsiId, :AKalinlikId, :AStokGirisNo, :ALot)
  left join OrmeRafDetay on OrmeRafDetay.OrmeRafId = OrmeIplikFireRaporu_1.OrmeRafID
  left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketID
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitID
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayID
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaID
  left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
  left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
  left join OrmeFatura on OrmeFatura.OrmeFaturaId = StokHareket.OrmeFaturaId
  left join Firma SHFirma on SHFirma.Firmaid = StokHareket.Firmaid
  left join OrmeStokHareketTipi on OrmeStokHareketTipi.OrmeStokHareketTipiId = StokHareket.HareketTipi
  left join Kullanici IK on IK.KullaniciId = StokHareket.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId = StokHareket.KullaniciId
  where StokHareket.StokHareketId is not null
  group by OrmeIplikFireRaporu_1.OrmeRafId, StokHareket.StokHareketId, StokHareket.HareketTipi, StokHareket.FirmaId, SHFirma.Adi,
           OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeFatura.Faturano, OrmeFatura.FaturaTarihi,
           OrmeIrsaliye.FiiliSevkTarihi, OrmeStokHareketTipi.Adi,
           BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.FirmaId, SiparisFirma.Adi,
           StokHareket.BirimFiyat, Ik.Adi, StokHareket.InsertTarihi, Sk.Adi, StokHareket.DegisimTarihi
  into :OrmeRafID, :StokHareketId, :HareketTipi, :FirmaId, :FirmaAdi, :Fisno, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi,
       :Faturano, :FaturaTarihi, :FiiliSevkTarihi, :HareketTipiStr,
       :BoyaSiparisYil, :BoyaSiparisID, :SiparisSiraNo, :SiparisFirmaId, :SiparisFirmaAdi, :BirimFiyat,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :Miktar, :BrutMiktar do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    suspend;
  end
end
