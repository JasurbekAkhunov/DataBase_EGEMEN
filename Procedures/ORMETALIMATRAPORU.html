        (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and

  end
  else

    for Select OrmePlanlama.OrmePlanlamaId, OrmePlanlama.Tarih, OrmePlanlama.UretimPartiNo, OrmePlanlama.FirmaId, OPF.Adi, OrmePlanlama.OrmeMakinaId, OrmeMakina.Adi,

           BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4,
           FirmaYetkili.Adi, FirmaYetkili.Telefon, FirmaYetkili.DefaultYetkili,
           OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.RevizeTeslimTarihi, OrmePlanlama.KdvDahil, OrmePlanlama.MakinaSayisi, OrmePlanlama.BaslamaTarihi,
           OrmePlanlama.TerminTarihi, OrmePlanlama.OnayDurumu, OrmePlanlama.OnayTarihi, OrmePlanlama.OnaylayanKullaniciID, OnaylayanKullanici.Adi, OrmePlanlama.Aciklama,
           BoyaSiparis.OrmeSiparisSekliId, OrmeSiparisSekli.Adi, BoyaSiparis.OrmeSiparisTipiId, OrmeSiparisTipi.Adi, OrmeSiparisDetay.OnMaliyetYapildi, OrmePlanlama.Beyazlik,
           OrmeSiparisDetay.Kaliteid, Kalite.KaliteAdi from OrmePlanlama
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
      left join OrmeSiparisSekli On OrmeSiparisSekli.OrmeSiparisSekliId = BoyaSiparis.OrmeSiparisSekliId
      left join OrmeSiparisTipi On OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
      left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
      left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
      left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
      left join Firma OPF on OPF.FirmaId = OrmePlanlama.FirmaId
      left join Firma ASF on ASF.FirmaId = BoyaSiparis.FirmaId
      Left Join Kullanici IK on IK.KullaniciId = OrmePlanlama.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = OrmePlanlama.KullaniciId
      Left Join Kullanici BK on BK.KullaniciId = OrmePlanlama.BitirenKullaniciId
      Left Join Kullanici OnaylayanKullanici on OnaylayanKullanici.KullaniciId = OrmePlanlama.OnaylayanKullaniciID
      left join FirmaYetkili on FirmaYetkili.FirmaId = OPF.FirmaId and FirmaYetkili.DefaultYetkili = 1
      left join Kalite on Kalite.Kaliteid=OrmeSiparisDetay.Kaliteid
      where (OrmePlanlama.Tarih >= :ABaslangicTarihi and OrmePlanlama.Tarih <= :ABitisTarihi) and
            (OPF.AnaHesapFirmaId = :APlanlamaFirmaId or :APlanlamaFirmaId is null) and
            (OrmePlanlama.OrmeMakinaId = :APlanlamaMakinaId or :APlanlamaMakinaId is null) and
            (OrmePlanlama.MakinaAyar = 0) and
            (ASF.AnaHesapFirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
            (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
            (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
            (BoyaSiparis.OrmeSiparisSekliId = :AOrmeSiparisSekliId or :AOrmeSiparisSekliId is null) and
            (OrmeMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
            (OrmeMakina.Pus = :AMakinaPus or :AMakinaPus is null) and
            (OrmeMakina.Feine = :AMakinaFeine or :AMakinaFeine is null) and
            (OrmePlanlama.Bitti = :ABitti or :ABitti is null) and
            (:ARaporTipi is null or
             (OrmePlanlama.OrmeMakinaId is not null and :ARaporTipi = 0) or
             (OrmePlanlama.FirmaId is not null and :ARaporTipi = 1) ) and
            (BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
            (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil  and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId)) and
            (OrmePlanlama.KdvDahil = :AKdvDahil or :AKdvDahil is Null) and
            (OrmePlanlama.OnayDurumu = :AOnayDurumu or :AOnayDurumu is Null) and
            (Ham.HamId = :AHamId or :AHamId is Null)
            order by OrmePlanlama.OrmeMakinaId, OrmePlanlama.FirmaId, Sira
      into :OrmePlanlamaId, :Tarih, :UretimPartiNo, :FirmaId, :FirmaAdi, :OrmeMakinaId, :OrmeMakinaAdi,
           :Sira, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
           :HamId, :HamAdi, :RenkAdi, :TalimatMiktari, :GerceklesenMiktar, :SakatUretimToplami, :MiktarGun, :MusteriSiparisNo, :ReferansSiparisNo,
           :SiparisFirmaId, :SiparisFirmaAdi, :SiparisSiraNo, :HamMiktar,
           :ArtikelNo, :BirimFiyat, :DovizId, :vadeGun, :Notu, :MakinaUstuGramaj, :DinlenmisGramaj, :Pus, :Feine, :TupAcikEnMayli, :SipTupAcikEnMayli,
           :MelanjKodu, :MakinaPus, :MakinaFeine, :TahminiFireOrani, :Bitti, :BitirenKullaniciId, BitirenKullaniciAdi, :BitirmeTarihi,
           :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4,
           :InsertTarihi, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi,
           :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :HamAdi2, :HamAdi3, :HamAdi4,
           :YetkiliAdi, :Telefon, :DefaultYetkili,
           :SipTeslimTarihi, :RevizeTeslimTarihi, :KdvDahil, :MakinaSayisi, :BaslamaTarihi, :TerminTarihi,
           :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciID, :OnaylayanKullaniciAdi, :Aciklama,
           :OrmeSiparisSekliId, :OrmeSiparisSekliIdStr, :OrmeSiparisTipiId, :OrmeSiparisTipiIdStr, :OnMaliyetYapildi, :Beyazlik, :KaliteId, :KaliteAdi do
    begin
      select first 1 OrmePlanlamaGrupDetayi.OrmePlanlamaGrupId  from  OrmePlanlamaGrupDetayi where OrmePlanlamaGrupDetayi.OrmeMakinaId = :OrmeMakinaId into :OrmePlanlamaGrupId;

      if (OnMaliyetYapildi = 1) then OnMaliyetYapildiStr = 'Evet';
      else OnMaliyetYapildiStr = 'Hayır';

      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
      if (Bitti = 0) then BittiStr = '';
      else BittiStr = 'Evet';

      if (KdvDahil = 1) then KdvDahilStr = 'Evet';
      else KdvDahilStr = '';

      if (Beyazlik = 1) then BeyazlikStr = 'Beyazlık';
      else BeyazlikStr = '';

      if (OnayDurumu = 1) then OnayDurumuStr = 'Onaylı';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Red';
      else OnayDurumuStr = '';

      Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

      Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:SipTupAcikenMayli) into :SipTupAcikEnMayliStr;

      KalanMiktar = TalimatMiktari - GerceklesenMiktar + SakatUretimToplami;

      Select Sum(StokHareket.KapAdedi) from StokHareket
        Where StokHareket.OrmePlanlamaId = :OrmePlanlamaId and
        (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and
        StokHareket.HamIade = 0  into :GercceklessenKapAdedi;

      suspend;
    end
  end
