CREATE PROCEDURE FIRMABAZINDABOYAHANEANALIZI_5(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DISTAMIRKILOTOPLAM DOUBLE PRECISION,
    DISTAMIRMETRETOPLAM DOUBLE PRECISION)
AS
begin
  For Select HamStokMaster.FirmaId, Firma.Adi, Sum(HamStokMaster.Kilo), Sum(HamStokMaster.Metre) from DisTamir
    left join HamStokMaster on HamStokMaster.HamStokMasterId = DisTamir.HamStokMasterId
    left join Firma on Firma.FirmaId = HamStokMaster.FirmaId
    where DisTamir.BeyanTarihi >= :ABaslangicTarihi and DisTamir.BeyanTarihi <= :ABitisTarihi and
          (Firma.AnaHesapFirmaID = :AFirmaId or :AFirmaID is null)
    group by HamStokMaster.FirmaId, Firma.Adi
    into :FirmaId, :FirmaAdi, :DisTamirKiloToplam, :DisTamirMetreToplam do
  begin
    suspend;
  end
end
