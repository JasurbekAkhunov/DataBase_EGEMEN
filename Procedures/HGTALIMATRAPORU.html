CREATE PROCEDURE HGTALIMATRAPORU(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASIPARISFIRMAID VARCHAR(20),
    ABASLANGICHGSIPARISYIL INTEGER,
    ABASLANGICHGSIPARISID INTEGER,
    ABITISHGSIPARISYIL INTEGER,
    ABITISHGSIPARISID INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    APLANLAMAFIRMAID VARCHAR(20),
    APLANLAMAMAKINAID VARCHAR(10),
    ABITTI INTEGER,
    AONAYDURUMU INTEGER,
    AMODELID MODELKODUSIZE,
    AKESIMTALIMATI SMALLINT,
    APARCAISCILIK SMALLINT,
    ADIKIMTALIMATI SMALLINT)
RETURNS (
    HGISEMRIKAYITID INTEGER,
    TARIH TIMESTAMP,
    HGSIPARISKAYITID INTEGER,
    HGSIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    URETIMPARTINO VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    SIRA INTEGER,
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER,
    HGSIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPTESLIMTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    MODELID MODELKODUSIZE,
    MODELADI ADI40,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ACIKLAMA VARCHAR(150),
    BITTI INTEGER,
    BITTISTR VARCHAR(4),
    KESIMTALIMATI SMALLINT,
    PARCAISCILIK SMALLINT,
    DIKIMTALIMATI SMALLINT,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    ONAYDURUMU SMALLINT,
    ONAYDURUMUSTR VARCHAR(7),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    KESIMTALIMATISTR VARCHAR(5),
    PARCAISCILIKSTR VARCHAR(5),
    DIKIMTALIMATISTR VARCHAR(5))
AS
declare variable shtSatis integer;
declare variable shtSatisIade integer;
begin

  select shtSatis, ShtSatisIade from OrmeStokHareketTipiGetir into :shtSatis, :ShtSatisIade;

  if (ADikimTalimati = 1) then
    ADikimTalimati = 0;
  else
    ADikimTalimati = -1;

  for Select HGIsEmri.HGIsEmriKayitId, HGIsEmri.Tarih, HGIsEmri.UretimPartiNo, HGIsEmri.FirmaId, OPF.Adi, HGIsEmri.OrmeMakinaId, OrmeMakina.Adi,
             HGIsEmri.HGSiparisKayitId, HGIsEmri.HGSiparisDetayId, HGSiparis.HGSiparisYil, HGSiparis.HGSiparisId, Model.ModelId, Model.Adi,
             HGSiparisDetay.MusteriSiparisNo, HGSiparisDetay.ReferansSiparisNo, HGSiparis.FirmaId, ASF.Adi,
             HGSiparisDetay.SiparisSiraNo, HGIsEmri.BirimFiyat, HGIsEmri.DovizId, HGIsEmri.vadeGun, HGIsEmri.Notu,
             HGIsEmri.Bitti, HGIsEmri.InsertTarihi, HGIsEmri.DegisimTarihi, Ik.Adi, Sk.Adi,
             HGSiparis.SiparisGrupId, SiparisGrup.Adi, HGSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             HGSiparisDetay.TeslimTarihi, HGIsEmri.TerminTarihi, HGIsEmri.OnayDurumu, HGIsEmri.OnayTarihi,
             HGIsEmri.OnaylayanKullaniciID, OnaylayanKullanici.Adi, HGIsEmri.KesimTalimati, HGIsEmri.ParcaIscilik from HGIsEmri
      left join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayId = HGIsEmri.HGSiparisDetayId
      left join HGSiparis on HGSiparis.HGSiparisKayitId = HGIsEmri.HGSiparisKayitId
      left join SiparisGrup on SiparisGrup.SiparisGrupId = HGSiparis.SiparisGrupId
      left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = HGSiparis.SiparisGrup2Id
      left join Model on Model.ModelKayitID = HGSiparisDetay.ModelKayitID
      left join OrmeMakina on OrmeMakina.OrmeMakinaId = HGIsEmri.OrmeMakinaId
      left join Firma OPF on OPF.FirmaId = HGIsEmri.FirmaId
      left join Firma ASF on ASF.FirmaId = HGSiparis.FirmaId
      Left Join Kullanici IK on IK.KullaniciId = HGIsEmri.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = HGIsEmri.KullaniciId
      Left Join Kullanici OnaylayanKullanici on OnaylayanKullanici.KullaniciId = HGIsEmri.OnaylayanKullaniciID
      where (HGIsEmri.Tarih >= :ABaslangicTarihi and HGIsEmri.Tarih <= :ABitisTarihi) and
            (OPF.AnaHesapFirmaId = :APlanlamaFirmaId or :APlanlamaFirmaId is null) and
            (HGIsEmri.OrmeMakinaId = :APlanlamaMakinaId or :APlanlamaMakinaId is null) and
            (ASF.AnaHesapFirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
            (HGSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
            (HGSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
            (HGIsEmri.Bitti = :ABitti or :ABitti is null) and
            (:ARaporTipi is null or
             (HGIsEmri.OrmeMakinaId is not null and :ARaporTipi = 0) or
             (HGIsEmri.FirmaId is not null and :ARaporTipi = 1) ) and
            (HGSiparis.HGSiparisYil > :ABaslangicHGSiparisYil or (HGSiparis.HGSiparisYil = :ABaslangicHGSiparisYil and HGSiparis.HGSiparisId >= :ABaslangicHGSiparisId)) and
            (HGSiparis.HGSiparisYil < :ABitisHGSiparisYil or (HGSiparis.HGSiparisYil = :ABitisHGSiparisYil  and HGSiparis.HGSiparisId <= :ABitisHGSiparisId)) and
            (HGIsEmri.OnayDurumu = :AOnayDurumu or :AOnayDurumu is Null) and
            (Model.ModelId = :AModelId or :AModelId is Null) and
            (HGIsEmri.KesimTalimati = :AKesimTalimati or
            HGIsEmri.ParcaIscilik = :AParcaIscilik or
            HGIsEmri.KesimTalimati = :ADikimTalimati)
            order by HGIsEmri.OrmeMakinaId, HGIsEmri.FirmaId
      into :HGIsEmriKayitId, :Tarih, :UretimPartiNo, :FirmaId, :FirmaAdi, :OrmeMakinaId, :OrmeMakinaAdi,
           :HGSiparisKayitId, :HGSiparisDetayId, :HGSiparisYil, :HGSiparisId,
           :ModelId, :ModelAdi, :MusteriSiparisNo, :ReferansSiparisNo,
           :SiparisFirmaId, :SiparisFirmaAdi, :SiparisSiraNo, :BirimFiyat, :DovizId, :VadeGun, :Notu, :Bitti,
           :InsertTarihi, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi,
           :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SipTeslimTarihi, :TerminTarihi,
           :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciID, :OnaylayanKullaniciAdi, :KesimTalimati, :ParcaIscilik do
    begin
      HGSiparisIdStr = '(' || EgemenIntegerToStr(HGSiparisYil) || ') '  || EgemenIntegerToStr(HGSiparisId);
      if (Bitti = 0) then BittiStr = '';
      else BittiStr = 'Evet';

      if (OnayDurumu = 1) then OnayDurumuStr = 'Onayl覺';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Red';
      else OnayDurumuStr = '';

      KesimTalimatiStr = 'Hay覺r';
      DikimTalimatiStr = 'Hay覺r';
      if (KesimTalimati = 1) then
        KesimTalimatiStr = 'Evet';
      else
        DikimTalimatiStr = 'Evet';

      ParcaIscilikStr = 'Hay覺r';
      if (ParcaIscilik = 1) then
        ParcaIscilikStr = 'Evet';

      suspend;
    end

end
