CREATE PROCEDURE IRSALIYENINFATURASINIKES(
    ABOYAIRSALIYEID INTEGER,
    ABOYAIRSALIYEDETAYID INTEGER,
    ABOYAFATURAID INTEGER,
    AKULLANICIID INTEGER,
    ABOYAFATURANO NUMERIC(18,0))
AS
declare variable PartiKayitId         integer;
declare variable PartiSinifId         integer;
declare variable FaturaTarihi          TimeStamp;
declare variable FHedefDovizId         varchar(5);
declare variable SiparisBirimFiyat     Double Precision;
declare variable BirimFiyat            Double Precision;
declare variable FDovizId              varchar(5);
declare variable DovizKuru1            Double Precision;
declare variable DovizKuru2            Double Precision;
declare variable Fason                 integer;
declare variable HamTipiId             integer;
declare variable Kilo                  Double Precision;
declare variable Metre                 Double Precision;
declare variable BrutKilo              Double Precision;
declare variable BrutMetre             Double Precision;
declare variable Dara                  Double Precision;
declare variable KaliteNo              integer;
declare variable KayitSayisi           integer;
declare variable HamStokBirimId        integer;
declare variable FNetFiyatYuvarlamaCarpani Double Precision;
declare variable FBoyaIrsaliyeDetayId  integer;
declare variable FaturaMiktar          Numeric(12,3);
declare variable HareketTipiId         integer;
declare variable RafDetayId            integer;
declare variable FirmaId               varchar(20);
declare variable FaturaEdilecekMiktarTipi integer;
declare variable FiyatOnayiIstensin       integer;
declare variable FiyatOnaylayanKullaniciId integer;
declare variable PartiPakettenFiyatlandirildi integer;
declare variable PartiPaketId         integer;
declare variable Brut                 integer;
declare variable IrsaliyeTarihi       TimeStamp;
declare variable EgemenFirmaID        integer;
declare variable SiparisteISBrutNetDaradan integer;
declare variable FMiktarBirim         SmallInt;
declare variable OrmeFirmaKodumuz     FirmaKoduSize;
declare variable FaturaSeriNo         SeriNoSize5;
declare variable SistemDovizId        DovizSize;
declare variable  FEFaturaKullanimda smallint;
declare variable  FEFatura smallint;
declare variable  EFaturaSeriNo SERINOSIZE5;
declare variable  FaturaDovizID varchar(5);
declare variable  Adet Type of Column BoyaIrsaliyeDetay.Adet;
begin
  Select EgemenFirmaID, Coalesce(OrmeFirmaKodumuz, ''), SistemDovizId,EFaturaSeriNo from Sistem where sistemId = 1 into :EgemenFirmaID, :OrmeFirmaKodumuz, :SistemDovizId, :EFaturaSeriNo;

  select Firma.EFaturaKullanimda from Firma ,BoyaIrsaliye where Firma.FirmaId = BoyaIrsaliye.FirmaId and BoyaIrsaliye.BoyaIrsaliyeId = :ABoyaIrsaliyeId into :FEFaturaKullanimda;

  if (ABoyaFaturaID is null) then
  begin
    ABoyaFaturaID = gen_id(BoyaFatura_Id_Gen, 1);
   if ((FEFaturaKullanimda=1) and (coalesce(EFaturaSeriNo,'')<>'')) then
    begin
      FaturaSeriNo = :EFaturaSeriNo;
      FEFatura = 1;
      end
    else
    begin
      FEFatura = 0;
      Select Coalesce(FaturaSeriNo, '') from Kullanici where KullaniciId = :AKullaniciId into :FaturaSeriNo;
    end

    Select first 1  PartiStok.DovizID from BoyaIrsaliyeDetay
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    left join PartiStok on partiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join BoyaRezervFisiDetay on BoyaRezervFisiDetay.BoyaRezervFisiDetayId = BoyaIrsaliyeDetay.BoyaRezervFisiDetayId
    left join BoyaSiparisDetay RS on RS.BoyaSiparisDetayId = BoyaRezervFisiDetay.BoyaSiparisDetayId
    left join MiktarTipi On MiktarTipi.MiktarTipiId = Coalesce(RS.FaturaEdilecekMiktarTipi, BoyaSiparisDetay.FaturaEdilecekMiktarTipi)
    where BoyaIrsaliyeDetay.BoyaIrsaliyeId = :ABoyaIrsaliyeId into :FaturaDovizID ;

    if (EgemenFirmaID = 120 ) then FaturaTarihi = current_date;

    if (FaturaTarihi is not null) then
    begin
      Select DovizKuru from EnYakinDovizKuruGetir( :FaturaDovizID, :SistemDovizId,   :FaturaTarihi) into :DovizKuru2 ;
    end
    else
      DovizKuru2 = 1;

    insert into BoyaFatura (BoyaFaturaId, FaturaSeriNo, FaturaNo, FaturaTarihi, FirmaId, OdemeTarihi, DovizId, Sinif,
                            InsertKullaniciId, HareketTipiId, IslemDovizId, FirmaSevkAdresiId, eFatura, DovizKuru2, IslemDovizKuru2)
    select :ABoyaFaturaId, :FaturaSeriNo, :ABoyaFaturaNo,coalesce(:FaturaTarihi, BoyaIrsaliye.IrsaliyeTarihi), BoyaIrsaliye.FirmaId, BoyaIrsaliye.IrsaliyeTarihi, :FaturaDovizID, BoyaIrsaliye.Sinif,
                            :AKullaniciId, BoyaIrsaliye.HareketTipiId, :SistemDovizId, BoyaIrsaliye.FirmaSevkAdresiId, :FeFatura, :DovizKuru2, :DovizKuru2
    from BoyaIrsaliye where BoyaIrsaliye.BoyaIrsaliyeId = :ABoyaIrsaliyeId;
  end



  Select FirmaId, FaturaTarihi, DovizId, NetFiyatYuvarlamaCarpani From BoyaFatura where BoyaFatura.BoyaFaturaId = :ABoyaFaturaID
    into :FirmaId, :FaturaTarihi, :FHedefDovizId, :FNetFiyatYuvarlamaCarpani;

  for Select BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliyeDetay.PartiKayitId, BoyaIrsaliyeDetay.PartiSinifId, BoyaIrsaliyeDetay.Kilo, BoyaIrsaliyeDetay.Metre,
             BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.BrutMetre, BoyaIrsaliyeDetay.KaliteNo,
             BoyaSiparisDetay.HamStokBirimId, MiktarTipi.MiktarTipiId, Parti.Fason, Ham.HamTipiId,
             BoyaIrsaliyeDetay.HareketTipiId, BoyaIrsaliyeDetay.RafDetayId,
             PartiStok.FiyatOnayiIstensin, PartiStok.FiyatOnaylayanKullaniciId, PartiStok.PartiPakettenFiyatlandirildi,
             BoyaIrsaliye.IrsaliyeTarihi, BoyaSiparisDetay.SiparisteISBrutNetDaradan, BoyaIrsaliyeDetay.Adet from BoyaIrsaliyeDetay
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    left join PartiStok on partiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join BoyaRezervFisiDetay on BoyaRezervFisiDetay.BoyaRezervFisiDetayId = BoyaIrsaliyeDetay.BoyaRezervFisiDetayId
    left join BoyaSiparisDetay RS on RS.BoyaSiparisDetayId = BoyaRezervFisiDetay.BoyaSiparisDetayId
    left join MiktarTipi On MiktarTipi.MiktarTipiId = Coalesce(RS.FaturaEdilecekMiktarTipi, BoyaSiparisDetay.FaturaEdilecekMiktarTipi)
    where BoyaIrsaliyeDetay.BoyaIrsaliyeId = :ABoyaIrsaliyeId and
          (BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = :ABoyaIrsaliyeDetayId or :ABoyaIrsaliyeDetayId is null) and
          BoyaIrsaliyeDetay.FaturaKesilecek = 1 and BoyaIrsaliyeDetay.BoyaFaturaId is null
    into :FBoyaIrsaliyeDetayId, :PartiKayitId, :PartiSinifId, :Kilo, :Metre, :BrutKilo, :BrutMetre, :KaliteNo, :HamStokBirimId,
         :FaturaEdilecekMiktarTipi, :Fason, :HamTipiId,  :HareketTipiId, :RafDetayId,
         :FiyatOnayiIstensin, :FiyatOnaylayanKullaniciId, :PartiPakettenFiyatlandirildi,
         :IrsaliyeTarihi, :SiparisteISBrutNetDaradan, :Adet do
  begin
    if (OrmeFirmaKodumuz = FirmaID) then /*Orme firmamiza satisda fason muamelesi yapilacak*/
      FaturaEdilecekMiktarTipi = 1; /*Brüt*/

    if (SiparisteISBrutNetDaradan is Null) then SiparisteISBrutNetDaradan = 0;

    Dara = 0;
    Select Sum(PartiStokTarti.Dara)  from PartiStokTarti
      where PartiStokTarti.BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId
      into :Dara;

    if (HamTipiId = 0 and SiparisteISBrutNetDaradan = 1) then BrutKilo = Kilo + Dara;

    BirimFiyat = 0;
    if (PartiSinifId is not null) then
    begin
      if (FiyatOnayiIstensin = 1 and FiyatOnaylayanKullaniciId is null) then exception PartiStokFiyatOnayVerilmemis;

      if (OrmeFirmaKodumuz <> FirmaID) then
      begin
        if (KaliteNo = 1) then
        begin
          Select PartiStok.BirimFiyat, PartiStok.DovizId from PartiStok
            left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
            where PartiStok.PartiSinifId = :PartiSinifId
            into :SiparisBirimFiyat, :FDovizId;
        end
        else if (KaliteNo >= 2) then
        begin
          SiparisBirimFiyat = 0;

          Select BoyaSiparisDetay.IkinciKaliteBirimFiyat, BoyaSiparisDetay.DovizId from PartiStok
            left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
            where PartiStok.PartiSinifId = :PartiSinifId
            into :SiparisBirimFiyat, :FDovizId;

          if (EgemenFirmaId = 78 and KaliteNo > 2) then /*istanbul boya*/
            SiparisBirimFiyat = 0;
        end
      end
      else
      begin
        Select BoyaSiparisDetay.FasonBirimFiyati, BoyaSiparisDetay.FasonBirimFiyatDovizId from PartiStok
          left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
          where PartiStok.PartiSinifId = :PartiSinifId
          into :SiparisBirimFiyat, :FDovizId;
      end

      if (EgemenFirmaID = 4) then
        Select DovizKuru1, DovizKuru2 from FirmaDovizKurlariniGetir(:FirmaId, :FDovizId, :FHedefDovizId, :IrsaliyeTarihi) into :DovizKuru1, :DovizKuru2;
      else
        Select DovizKuru1, DovizKuru2 from FirmaDovizKurlariniGetir(:FirmaId, :FDovizId, :FHedefDovizId, :FaturaTarihi) into :DovizKuru1, :DovizKuru2;

      if (DovizKuru1 = 0 or DovizKuru2 = 0) then exception DovizKuruGirisiYapilmamis;

      FMiktarBirim = HamStokBirimId;

      if (HamStokBirimId = 0) then /* Kg*/
      begin
        if (FaturaEdilecekMiktarTipi = 0) then FaturaMiktar = Kilo;  /* net Kg*/
        else FaturaMiktar = BrutKilo; /* Brüt Kg*/
      end
      else if (HamStokBirimId = 2) then /* adet*/
      begin
        FaturaMiktar = Adet;
      end
      else
      begin /* metre */
        if (FaturaEdilecekMiktarTipi = 0) then FaturaMiktar = Metre; /* Net metre*/
        else FaturaMiktar = BrutMetre; /* Brüt metre*/
      end
    end
    else
    begin
      FaturaEdilecekMiktarTipi = 0; /*Net*/
      SiparisBirimFiyat = 0;
      BirimFiyat = 0;
      DovizKuru1 = 1;
      DovizKuru2 = 1;

      if (HareketTipiId = 1) then
      begin
        Select HamStokMaster.NetBirimFiyat, HamStokFatura.DovizId from RafDetay
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        left join HamStokFatura on HamStokFatura.HamStokFaturaId = HamStokMaster.HamStokFaturaId
        where RafDetay.RafDetayID = :RafDetayId
        into :BirimFiyat, :FDovizId;

        if (EgemenFirmaID = 4) then
          select DovizKuru1, DovizKuru2 from FirmaDovizKurlariniGetir(:FirmaId, :FDovizId, :FHedefDovizId, :IrsaliyeTarihi) into :DovizKuru1, :DovizKuru2;
        else
          select DovizKuru1, DovizKuru2 from FirmaDovizKurlariniGetir(:FirmaId, :FDovizId, :FHedefDovizId, :FaturaTarihi) into :DovizKuru1, :DovizKuru2;

        if (DovizKuru1 = 0) then DovizKuru1 = 1;
        if (DovizKuru2 = 0) then DovizKuru2 = 1;

        BirimFiyat = BirimFiyat / DovizKuru1 * DovizKuru2;
      end
      else
      begin
        FDovizid = SistemDovizId;
        select DovizKuru1, DovizKuru2 from FirmaDovizKurlariniGetir(:FirmaId, :FDovizId, :FHedefDovizId, :FaturaTarihi) into :DovizKuru1, :DovizKuru2;

        if (DovizKuru1 = 0) then DovizKuru1 = 1;
        if (DovizKuru2 = 0) then DovizKuru2 = 1;
      end

      if (HamTipiId = 2) then
      begin
        FaturaMiktar = Metre;
        FMiktarBirim = 1;
      end
      else
      begin
        FaturaMiktar = Kilo;
        FMiktarBirim = 0;
      end
    end

    if (PartiSinifId is null or PartiPakettenFiyatlandirildi = 0) then
      insert into BoyaFaturaDetay(BoyaFaturaId, BoyaIrsaliyeDetayId, FaturaTarihi, Miktar, FaturaMiktarTipi, MiktarBirim, IplikSipBrutNetDaradan, SiparisBirimFiyat, BirimFiyat, DovizKuru1, DovizKuru2, InsertKullaniciId, KullaniciId)
                  Values(:ABoyaFaturaId, :FBoyaIrsaliyeDetayId, :FaturaTarihi, :FaturaMiktar, :FaturaEdilecekMiktarTipi, :FMiktarBirim, :SiparisteISBrutNetDaradan, :SiparisBirimFiyat, :BirimFiyat, :DovizKuru1, :DovizKuru2, :AKullaniciId, :AKullaniciId);
    else
    begin
      for Select PartiPaket.PartiPaketId, PartiPaket.Brut, PartiPaket.BirimFiyat, PartiPaket.DovizId from PartiPaket where PartiPaket.PartiSinifId = :PartiSinifId
      into :PartiPaketId, :Brut, :SiparisBirimFiyat, :FDovizId do
      begin
        if (EgemenFirmaID = 4) then
          select DovizKuru1, DovizKuru2 from FirmaDovizKurlariniGetir(:FirmaId, :FDovizId, :FHedefDovizId, :IrsaliyeTarihi) into :DovizKuru1, :DovizKuru2;
        else
          select DovizKuru1, DovizKuru2 from FirmaDovizKurlariniGetir(:FirmaId,:FDovizId, :FHedefDovizId, :FaturaTarihi) into :DovizKuru1, :DovizKuru2;

        if (DovizKuru1 = 0 or DovizKuru2 = 0) then exception DovizKuruGirisiYapilmamis;

        FaturaEdilecekMiktarTipi = Brut;

        FMiktarBirim = HamStokBirimId;
        if (HamStokBirimId = 0) then /* Kg*/
        begin
          if (Brut = 0) then FaturaMiktar = Kilo;  /* net Kg*/
          else FaturaMiktar = BrutKilo; /* Brüt Kg*/
        end
        else if (HamStokBirimId = 2) then /* adet*/
        begin
          FaturaMiktar = Adet;
        end
        else
        begin /* metre */
          if (Brut = 0) then FaturaMiktar = Metre; /* Net metre*/
          else FaturaMiktar = BrutMetre; /* Brüt metre*/
        end


        insert into BoyaFaturaDetay(BoyaFaturaId, BoyaIrsaliyeDetayId, FaturaTarihi, PartiPaketId, Miktar, FaturaMiktarTipi, MiktarBirim, IplikSipBrutNetDaradan, SiparisBirimFiyat, BirimFiyat, DovizKuru1, DovizKuru2, InsertKullaniciId, KullaniciId)
                    values(:ABoyaFaturaId, :FBoyaIrsaliyeDetayId, :FaturaTarihi, :PartiPaketId, :FaturaMiktar, :FaturaEdilecekMiktarTipi, :FMiktarBirim, :SiparisteISBrutNetDaradan, :SiparisBirimFiyat, :BirimFiyat, :DovizKuru1, :DovizKuru2, :AKullaniciId, :AKullaniciId);
      end
    end

    update BoyaIrsaliyeDetay Set BoyaFaturaId = :ABoyaFaturaId where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId;
  end
end
