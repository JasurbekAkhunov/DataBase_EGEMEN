    DOKUMAMAKINAADI VARCHAR(40),
    SIRA INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    MAKINASAYISI INTEGER,
    SIPARISSIRANO INTEGER,

    TARAKENI DOUBLE PRECISION,
    RENKID INTEGER,

    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    TOPLAMSAKATDOKUNAN DOUBLE PRECISION,
    MIKTARGUN DOUBLE PRECISION,
    SURE DOUBLE PRECISION,
    BASLAMATARIHI TIMESTAMP,
    BIRIMFIYAT DOUBLE PRECISION,

declare variable KalanMiktar Double precision;
declare variable KS integer;
declare variable LastDokumaMakinaveFirmaId varchar(20);
declare variable DokumaMakinaveFirmaId varchar(20);
declare variable BoyaSiparisYil varchar(20);
declare variable BoyaSiparisID varchar(20);
begin
  if (ADokumaPlanlamaGrupId = '') then ADokumaPlanlamaGrupId = Null;

  LastDokumaMakinaveFirmaId = '@@@';

  if (ADokumaPlanlamaGrupId is null) then
  begin

      else
        CozguKalanGun = Null;

      else
        OrtalamaTur = Null;

      suspend;
    end
  end

    Select count(*) from DokumaPlanlamaGrupDetayi where DokumaPlanlamaGrupDetayi.DokumaPlanlamaGrupId = :ADokumaPlanlamaGrupId and DokumaPlanlamaGrupDetayi.FirmaId is not null into :Ks;

             :BirimFiyat, :DovizId, :vadeGun, :TahminiFireOrani,

        else DokumaMakinaveFirmaId = FirmaId;

        if (LastDokumaMakinaveFirmaId <> DokumaMakinaveFirmaId) then
        begin
          SepetNO = 0;
          LastDokumaMakinaveFirmaId = DokumaMakinaveFirmaId;
        end

        SepetNO = SepetNo + 1;

    BoyaSiparisIdStr = BoyaSiparisYil || '-' ||BoyaSiparisId;

        KalanMiktar = PlanlananMiktar - GerceklesenMiktar + ToplamSakatDokunan;

        if (MiktarGun > 0 and KalanMiktar > 0) then Sure = KalanMiktar / MiktarGun;
        else Sure = 0;

        select Sum(DokumaStokPlanlama.Miktar) from DokumaStokPlanlama where DokumaStokPlanlama.DokumaPlanlamaId = :DokumaPlanlamaId into :PlanlananCozguMiktari;
        CozguKalanMetre = Coalesce(PlanlananCozguMiktari, 0) - Coalesce(HesaplananUretim, 0);
        if (MiktarGun > 0) then
          CozguKalanGun = CozguKalanMetre / MiktarGun;
        else
          CozguKalanGun = Null;

        if (DokumaMakinaId is not null) then
          select Coalesce(Sum(OrmeMakinaDurumu.Tur),0) / 1440 from OrmeMakinaDurumu where OrmeMakinaDurumu.DokumaMakinaId = :DokumaMakinaId and OrmeMakinaDurumu.Tarih >= DateAdd(day, -1, current_timestamp) into :OrtalamaTur;
        else
          OrtalamaTur = Null;

        suspend;
      end
    end
    else /* fasoncu planlama grubu */
    begin
      for Select DokumaPlanlamaGrupDetayi.DokumaPlanlamaGrupId, DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlama.Tarih, DokumaPlanlama.FirmaId, Firma.Adi, Firma.KisaAdi,
           DokumaPlanlama.DokumaMakinaId, DokumaMakina.Adi, DokumaPlanlama.Sira, DokumaPlanlama.BoyaSiparisKayitId,
           DokumaPlanlama.OrmeSiparisDetayId, Ham.Adi, Ham.TarakEni, OrmeSiparisDetay.RenkId, Renk.Adi, DokumaPlanlama.PlanlananMiktar,
           DokumaPlanlama.GerceklesenMiktar, DokumaPlanlama.ToplamSakatDokunan, DokumaPlanlama.MiktarGun, DokumaPlanlama.BaslamaTarihi,
           DokumaPlanlama.UretimPartiNo, DokumaPlanlama.Beklemede, DokumaPlanlama.BirimFiyat, DokumaPlanlama.DovizId, DokumaPlanlama.vadeGun,

        left join DokumaPlanlama on DokumaPlanlama.FirmaId = DokumaPlanlamaGrupDetayi.FirmaId
        left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
        left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
        left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
        left join Firma on Firma.FirmaId = DokumaPlanlama.FirmaId
        left join DokumaMakina on DokumaMakina.DokumaMakinaId = DokumaPlanlama.DokumaMakinaId
        where DokumaPlanlamaGrupDetayi.DokumaPlanlamaGrupId = :ADokumaPlanlamaGrupId and
              (DokumaPlanlama.FirmaId = :AFirmaId or :AFirmaId is null) and DokumaPlanlama.Bitti = 0  and DokumaPlanlama.Beklemede = 0
        order by DokumaPlanlamaGrupDetayi.Sira, DokumaPlanlama.DokumaMakinaId, DokumaPlanlama.FirmaId, DokumaPlanlama.Sira

      begin

        if (LastDokumaMakinaveFirmaId <> DokumaMakinaveFirmaId) then
        begin
          SepetNO = 0;
          LastDokumaMakinaveFirmaId = DokumaMakinaveFirmaId;
        end

        SepetNO = SepetNo + 1;

        KalanMiktar = PlanlananMiktar - GerceklesenMiktar + ToplamSakatDokunan;

        if (MiktarGun > 0 and KalanMiktar > 0) then Sure = KalanMiktar / MiktarGun;

          OrtalamaTur = Null;

        suspend;
      end
    end
  end
