CREATE PROCEDURE KALITEKONTROLORMETALIMATLISTESI(
    AORMEMAKINAID VARCHAR(10),
    AHAMID INTEGER,
    ABITTI INTEGER)
RETURNS (
    RESULTID VARCHAR(255),
    ORMEPLANLAMAID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    URETIMPARTINO VARCHAR(30),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    MUSTERISIPARISNO VARCHAR(50))
AS
declare OrmeSiparisTipiId integer;
begin
  for Select OrmePlanlama.OrmePlanlamaId, OrmePlanlama.BoyaSiparisKayitId, OrmePlanlama.OrmeSiparisDetayId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.OrmeSiparisTipiId, OrmePlanlama.UretimPartiNo,
             OrmeSiparisDetay.HamId, Ham.Adi, OrmePlanlama.OrmeMakinaId, OrmeMakina.Adi, OrmeSiparisDetay.MusteriSiparisNo,
             Renk.adi from OrmePlanlama
  left join  OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
  left join  BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
  left join  Ham on Ham.HamId = OrmeSiparisDetay.HamId
  left join  Renk on Renk.RenkID = OrmeSiparisDetay.RenkID
  left join  OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
  where (OrmePlanlama.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and OrmePlanlama.FirmaId is null and
        (OrmeSiparisDetay.HamId = :AHamId or :AHamId is null) and OrmePlanlama.OnayDurumu = 1 and
        (OrmePlanlama.Bitti = :ABitti or :ABitti is null)
  Order By OrmePlanlama.Sira
  into :OrmePlanlamaId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
       :OrmeSiparisTipiId, :UretimPartiNo, :HamId, :HamAdi, :OrmeMakinaId, :OrmeMakinaAdi, :MusteriSiparisNo, :RenkAdi do
  begin
    ResultId  = '"' || coalesce(OrmePlanlamaId    ,'') || '",'
             || '"' || coalesce(BoyaSiparisKayitId,'') || '",'
             || '"' || coalesce(OrmeSiparisDetayId,'') || '",'
             || '"' || coalesce(BoyaSiparisYil    ,'') || '",'
             || '"' || coalesce(BoyaSiparisId     ,'') || '",'
             || '"' || coalesce(OrmeSiparisTipiId ,'') || '",'
             || '"' || coalesce(UretimPartiNo     ,'') || '",'
             || '"' || coalesce(HamId             ,'') || '"';
    BoyaSiparisIdStr = '(' || BoyaSiparisYil || ') '  || BoyaSiparisId;
    suspend;
  end
end
