CREATE PROCEDURE ORMERAFTRANSFERFISILISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AREFERANSNO VARCHAR(30),
    ADOKUMAHAREKETI INTEGER,
    AIRSALIYENO IRSALIYENOSIZE)
RETURNS (
    ORMERAFTRANSFERFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    REFERANSNO VARCHAR(30),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    IRSALIYENO IRSALIYENOSIZE,
    HEDEFRAF VARCHAR(10),
    RAFFIRMAID VARCHAR(20),
    RAFFIRMAADI VARCHAR(100),
    RAFADI VARCHAR(30))
AS
begin
  For Select OrmeRafTransferFisi.OrmeRafTransferFisiId, OrmeRafTransferFisi.FisYil, OrmeRafTransferFisi.FisNo, OrmeRafTransferFisi.FisTarihi,
             OrmeRafTransferFisi.Notu, IK.Adi, OrmeRafTransferFisi.InsertTarihi, Sk.Adi, OrmeRafTransferFisi.DegisimTarihi, OrmeRafTransferFisi.ReferansNo,
             OrmeRafTransferFisi.IrsaliyeNo, OrmeRafTransferFisi.RafId HedefRaf, Raf.FirmaId, Firma.Adi, Raf.Adi from OrmeRafTransferFisi
    Left join Raf on Raf.RafId = OrmeRafTransferFisi.RafId
    Left join Firma on Firma.FirmaId = Raf.FirmaId
    Left Join Kullanici IK on IK.KullaniciId = OrmeRafTransferFisi.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = OrmeRafTransferFisi.KullaniciId
    where OrmeRafTransferFisi.FisTarihi >= :ABaslangicTarihi and OrmeRafTransferFisi.FisTarihi <= :ABitisTarihi and
          (OrmeRafTransferFisi.ReferansNo = :AReferansNo or :AReferansNo is Null) and
          (OrmeRafTransferFisi.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
          OrmeRafTransferFisi.DokumaHareketi = :ADokumaHareketi
    into :OrmeRafTransferFisiId, :FisYil, :FisNo, :FisTarihi, :Notu,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :ReferansNo, :IrsaliyeNo, :HedefRaf, :RafFirmaId, :RafFirmaAdi, :RafAdi do
   begin
     FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';
     suspend;
   end
end
