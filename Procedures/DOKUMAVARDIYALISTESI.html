CREATE PROCEDURE DOKUMAVARDIYALISTESI(
    AAKTIF INTEGER)
RETURNS (
    RESULTID VARCHAR(100),
    ORMEVARDIYAID INTEGER,
    VARDIYATARIHI TIMESTAMP,
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    VARDIYANO INTEGER)
AS
begin
  For Select OrmeVardiya.OrmeVardiyaId, OrmeVardiya.Tarih, OrmeVardiya.VardiyaNo, OrmeVardiya.UniteId, Unite.Adi from OrmeVardiya
    left join Unite on Unite.UniteId = OrmeVardiya.UniteId
    where (OrmeVardiya.Aktif = :AAktif or :AAktif is null) and OrmeVardiya.DokumaVardiyasi = 1
    Order by OrmeVardiya.tarih, OrmeVardiya.UniteId, OrmeVardiya.VardiyaNo
    into :OrmeVardiyaId, :VardiyaTarihi, :VardiyaNo, :UniteId, :UniteAdi do
  begin
    ResultId = '"' || EgemenIntegerToStr(OrmeVardiyaId) || '",' ;
    ResultId = ResultId || '"' || TimeStampToDateStr(VardiyaTarihi) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(VardiyaNo) || '",' ;
    ResultId = ResultId || '"' || UniteAdi || '"' ;
    suspend;
  end
end
