CREATE PROCEDURE DOKUMAKALITEKONTROLRAPORU(
    ADOKUMAMAKINAID VARCHAR(10),
    ADOKUMAPLANLAMAID INTEGER,
    ASAKAT INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKONTROLDURUMU INTEGER,
    AHAMID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    AKUMASDURUMU INTEGER)
RETURNS (
    DOKUMAKALITEKONTROLID INTEGER,
    ORMEVARDIYAID INTEGER,
    TARIH TIMESTAMP,
    TARIH2 DATE,
    VARDIYATARIHI TIMESTAMP,
    VARDIYA INTEGER,
    DOKUMAPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    VARDIYAAMIRIPERSONELID INTEGER,
    VARDIYAAMIRIPERSONELADI VARCHAR(40),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    HAMID INTEGER,
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    AYARTOPU INTEGER,
    AYARTOPUSTR VARCHAR(10),
    HAMADI VARCHAR(255),
    LOT VARCHAR(150),
    ATKILOTU VARCHAR(30),
    COZGULOTU VARCHAR(30),
    HAMSTOKALISFIYATI DOUBLE PRECISION,
    SIPARISSATISFIYATI DOUBLE PRECISION,
    RAPORDOVIZID VARCHAR(5),
    KUMASCINSIID INTEGER,
    KUMASCINSIADI VARCHAR(40),
    TOPNO INTEGER,
    MIKTAR DOUBLE PRECISION,
    KILO DOUBLE PRECISION,
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    RAFID VARCHAR(10),
    SAKATHAMHATATIPIID INTEGER,
    SAKATHAMHATATIPIADI VARCHAR(20),
    KONTROLBITTI INTEGER,
    KONTROLDURUMUSTR VARCHAR(10),
    KONTROLBITISTARIHI TIMESTAMP,
    KONTROLBITIREN ADI40,
    PLANLAMABITTI SMALLINT,
    PLANLAMABITTISTR VARCHAR(10),
    PLANLAMABEKLEMEDE SMALLINT,
    PLANLAMABEKLEMEDESTR VARCHAR(10),
    ARAVERMESEBEBIID INTEGER,
    ARAVERMESEBEBIADI ADI50,
    MAKINAGRUPID MAKINAGRUPKODUSIZE,
    MAKINAGRUPADI ADI30,
    BOMUSTUTOPENI VARCHAR(10),
    RULOUSTUTOPENI VARCHAR(10),
    HESAPLANANMETRE BOYAMIKTARSIZE,
    HESAPLANANBOYDANCEKMEORANI BOYAMIKTARSIZE,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    KALITENO INTEGER,
    DOVIZID VARCHAR(5),
    ADET INTEGER,
    KALITEID VARCHAR(20),
    KALITEADI VARCHAR(250))
AS
Declare Variable HamStokAlisFiyatiDovizID VarChar(5);
Declare Variable SiparisDovizID VarChar(5);
declare Variable DovizKuru Double Precision;
declare variable uitBoya integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable FirmaKodumuz varchar(30);
begin
  Select SistemDovizId, sistem.FirmaKodumuz  from Sistem Where sistemid = 1 into :RaporDovizId, :FirmaKodumuz;
  select UretimIslemTipiGetir.uitboya from UretimIslemTipiGetir into :uitBoya;
  select OrmeStokHareketTipiGetir.shtFasonUretimeCikis, OrmeStokHareketTipiGetir.shtUretimeCikis from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis, :shtUretimeCikis;

  For Select DokumaKaliteKontrol.DokumaKaliteKontrolId, DokumaKaliteKontrol.OrmeVardiyaId, DokumaKaliteKontrol.Tarih, DokumaKaliteKontrol.VardiyaTarihi, DokumaKaliteKontrol.Vardiya, DokumaKaliteKontrol.DokumaPlanlamaId, DokumaPlanlama.UretimPartiNo,
             DokumaKaliteKontrol.VardiyaAmiriPersonelId, VAmiri.Adi, DokumaKaliteKontrol.PersonelId, VPersonel.Adi, DokumaPlanlama.BoyaSiparisKayitId, DokumaPlanlama.OrmeSiparisDetayid,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi,
             OrmeSiparisDetay.HamId, Ham.Adi, DokumaKaliteKontrol.TopNo, DokumaKaliteKontrol.Miktar, DokumaKaliteKontrol.Sakat, DokumaKaliteKontrol.AyarTopu, DokumaKaliteKontrol.RafId, IK.Adi,
             DokumaKaliteKontrol.InsertTarihi, SK.Adi, DokumaKaliteKontrol.DegisimTarihi, Ham.KumasCinsiId, KumasCinsi.Adi,
             DokumaPlanlama.DokumaMakinaId, DokumaMakina.Adi, Ham.AlisFiyati, Ham.AlisFiyatiDovizId,
             OrmeSiparisDetay.SatisFiyati, OrmeSiparisDetay.DovizId, DokumaKaliteKontrol.SakatHamHataTipiId, HamHataTipi.Adi, DokumaKaliteKontrol.Lot,
             DokumaKaliteKontrol.KontrolBitti, DokumaKaliteKontrol.BitisTarihi, KY.Adi, DokumaKaliteKontrol.AtkiLotu, DokumaKaliteKontrol.CozguLotu, DokumaKaliteKontrol.Kilo,
             DokumaPlanlama.Bitti, DokumaPlanlama.Beklemede, DokumaPlanlama.AraVermeSebebiId, AraVermeSebebi.Adi, DokumaMakina.MakinaGrupID, MakinaGrup.Adi,
             DokumaKaliteKontrol.BomUstuTopEni, DokumaKaliteKontrol.RuloUstuTopEni, DokumaKaliteKontrol.HesaplananMetre, DokumaKaliteKontrol.KaliteNo, OrmeSiparisDetay.DovizId, DokumaKaliteKontrol.Adet,
             Kalite.KaliteId, Kalite.KaliteAdi  from DokumaKaliteKontrol
    left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaKaliteKontrol.DokumaPlanlamaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = DokumaKaliteKontrol.OrmeUCikisIrsaliyeId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    left join Kalite on Kalite.KaliteId=OrmeSiparisDetay.KaliteID
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = DokumaPlanlama.DokumaMakinaId
    left join KumasCinsi on KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
    left join MakinaGrup on MakinaGrup.MakinaGrupId = DokumaMakina.MakinaGrupID
    Left Join Kullanici IK on IK.KullaniciId = DokumaKaliteKontrol.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = DokumaKaliteKontrol.KullaniciId
    Left Join Kullanici KY on KY.KullaniciId = DokumaKaliteKontrol.BitirenKullaniciId
    left join Personel VAmiri on VAmiri.PersonelId  = DokumaKaliteKontrol.VardiyaAmiriPersonelId
    left join Personel VPersonel on VPersonel.PersonelId  = DokumaKaliteKontrol.PersonelId
    left join HamHataTipi on HamHataTipi.HamHataTipiId  = DokumaKaliteKontrol.SakatHamHataTipiId
    left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = DokumaPlanlama.AraVermeSebebiId
    where  DokumaKaliteKontrol.Tarih >= :ABaslangicTarihi and DokumaKaliteKontrol.Tarih <= :ABitisTarihi and
          (DokumaKaliteKontrol.DokumaPlanlamaId = :ADokumaPlanlamaId or :ADokumaPlanlamaId is null) and
          (DokumaKaliteKontrol.Sakat = :ASakat or :ASakat is null) and
          (DokumaPlanlama.DokumaMakinaId = :ADokumaMakinaId or :ADokumaMakinaId is null) and
          (DokumaKaliteKontrol.KontrolBitti = :AKontrolDurumu or :AKontrolDurumu is null) and
          (OrmeSiparisDetay.HamId = :AHamId or :AHamId is null) and
          (BoyaSiparis.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          ((:AKumasDurumu is null) or
           (:AKumasDurumu = 0 and DokumaKaliteKontrol.OrmeUCikisIrsaliyeId is not null and OrmeIrsaliye.FirmaId = :FirmaKodumuz and
             OrmeIrsaliye.HareketTipi in (:shtFasonUretimeCikis, :shtUretimeCikis) and OrmeIrsaliye.UretimIslemtipiId = :uitBoya ) or
           (:AKumasDurumu = 1 and DokumaKaliteKontrol.OrmeUCikisIrsaliyeId is not null and OrmeIrsaliye.FirmaId <> :FirmaKodumuz and
             OrmeIrsaliye.HareketTipi in (:shtFasonUretimeCikis, :shtUretimeCikis) and OrmeIrsaliye.UretimIslemtipiId = :uitBoya ) or
           (:AKumasDurumu = 2 and DokumaKaliteKontrol.OrmeUCikisIrsaliyeId is null)
          )

    Order By DokumaPlanlama.UretimPartiNo, VardiyaTarihi, Vardiya
    into :DokumaKaliteKontrolId, :OrmeVardiyaId, :Tarih, :VardiyaTarihi, :Vardiya, :DokumaPlanlamaId, :UretimPartiNo, :VardiyaAmiriPersonelId, :VardiyaAmiriPersonelAdi,
         :PersonelId, :PersonelAdi, :BoyaSiparisKayitId, :OrmeSiparisDetayid, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :SiparisFirmaAdi,
         :HamId, :HamAdi, :TopNo, :Miktar, :Sakat, :AyarTopu, :RafId, :InsertKullaniciAdi, InsertTarihi, :KullaniciAdi, :DegisimTarihi,
         :KumasCinsiId, :KumasCinsiAdi, :DokumaMakinaId, :DokumaMakinaAdi, :HamStokAlisFiyati, :HamStokAlisFiyatiDovizId,
         :SiparisSatisFiyati, :SiparisDovizId, :SakatHamHataTipiId, :SakatHamHataTipiAdi, :Lot, :KontrolBitti, :KontrolBitisTarihi, :KontrolBitiren, :AtkiLotu, :CozguLotu, :Kilo,
         :PlanlamaBitti, :PlanlamaBeklemede, :AraVermeSebebiId, :AraVermeSebebiAdi, :MakinaGrupID, :MakinaGrupAdi, :BomUstuTopEni, :RuloUstuTopEni, :HesaplananMetre, :KaliteNo, :DovizId, :Adet,
         :KaliteId, :KaliteAdi  do
  begin
    if (PlanlamaBeklemede = 1) then PlanlamaBeklemedeStr = 'Beklemede';
    else PlanlamaBeklemedeStr = '';

    if (PlanlamaBitti = 1) then PlanlamaBittiStr = 'Bitti';
    else PlanlamaBittiStr = '';

    if (KontrolBitti = 1) then KontrolDurumuStr = 'Bitti';
    else KontrolDurumuStr = '';

    Tarih2 = Tarih;
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);


    if (Sakat = 1) then SakatStr = 'Sakat';
    else SakatStr = '';

    if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
    else AyarTopuStr = '';


    if (HamStokAlisFiyatiDovizId <> RaporDovizId) then
      select DovizKuru from EnYakinDovizKuruGetir(:HamStokAlisFiyatiDovizId, :RaporDovizId, Current_Date) into :DovizKuru;
    else
     DovizKuru = 1;

    HamStokAlisFiyati = HamStokAlisFiyati * DovizKuru;

    /*
    if (SiparisDovizId <> RaporDovizId) then
      select DovizKuru from EnYakinDovizKuruGetir(:SiparisDovizId, :RaporDovizId, Current_Date) into :DovizKuru;
    else
      DovizKuru = 1;   */

    if (HesaplananMetre > 0) then
      HesaplananBoydanCekmeOrani = ((HesaplananMetre - Miktar) / HesaplananMetre) * 100;
    else
      HesaplananBoydanCekmeOrani = 0;

    suspend;
  end
end
