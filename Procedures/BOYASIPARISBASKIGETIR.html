CREATE PROCEDURE BOYASIPARISBASKIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISBASKIID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    DESENVARYANTID INTEGER,
    VARYANTADI VARCHAR(40),
    VARYANTILKBASKI INTEGER,
    OZELYERLESIM INTEGER,
    OZELYERLESIMDOSYAADI VARCHAR(255),
    YANRAPORTVARMI INTEGER,
    KENARBEYAZLIGI INTEGER,
    KONFEKSIYON INTEGER,
    TASMA INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    VARYANTACIKLAMA VARCHAR(255),
    RESIMDOSYAADI VARCHAR(255))
AS
begin
  For Select BoyaSiparisBaski.BoyaSiparisBaskiId, BoyaSiparisBaski.BoyaSiparisDetayId, BoyaSiparisBaski.DesenVaryantId, DesenVaryant.VaryantAdi,
             BoyaSiparisBaski.Kilo, BoyaSiparisBaski.Metre, BoyaSiparisBaski.VaryantIlkBaski, BoyaSiparisBaski.OzelYerlesim,
             BoyaSiparisBaski.OzelYerlesimDosyaAdi, BoyaSiparisBaski.YanRaportVarmi, BoyaSiparisBaski.KenarBeyazligi, BoyaSiparisBaski.Konfeksiyon,
             BoyaSiparisBaski.Tasma, IK.ADI, BoyaSiparisBaski.Inserttarihi, SK.ADI, BoyaSiparisBaski.DegisimTarihi, DesenVaryant.Aciklama, DesenVaryant.ResimDosyaAdi From BoyaSiparisBaski
  Left Join DesenVaryant ON DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
  Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisBaski.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisBaski.KullaniciId
  Where BoyaSiparisBaski.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  order by BoyaSiparisBaski.BoyaSiparisKayitId, BoyaSiparisBaski.BoyaSiparisDetayId, BoyaSiparisBaski.BoyaSiparisBaskiId
  into :BoyaSiparisBaskiId, :BoyaSiparisDetayId, :DesenVaryantId, :VaryantAdi, :Kilo, :Metre, :VaryantIlkBaski, :OzelYerlesim,
       :OzelYerlesimDosyaAdi, :YanRaportVarmi, :KenarBeyazligi, :Konfeksiyon, :Tasma,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :VaryantAciklama, :ResimDosyaAdi do suspend;
end
