CREATE PROCEDURE ORMESONVARDIYAIDGETIR(
    AORMEMAKINAID VARCHAR(10),
    ADOKUMAMAKINAID VARCHAR(10))
RETURNS (
    ORMEVARDIYAID INTEGER)
AS
begin
  if (AOrmeMakinaId is not null) then
  begin
    For Select OrmeVardiyaPersonel.OrmeVardiyaId from OrmeVardiyaPersonel
      left join OrmeVardiya on Ormevardiya.OrmeVardiyaId = OrmeVardiyaPersonel.OrmeVardiyaId
      where OrmeMakinaId = :AOrmeMakinaId order by tarih desc, vardiyano desc
      into :OrmeVardiyaId do
    begin
      suspend;
      exit;
    end
  end
  else
  begin
    For Select OrmeVardiyaPersonel.OrmeVardiyaId from OrmeVardiyaPersonel
      left join OrmeVardiya on Ormevardiya.OrmeVardiyaId = OrmeVardiyaPersonel.OrmeVardiyaId
      where DokumaMakinaId = :ADokumaMakinaId order by tarih desc, vardiyano desc
      into :OrmeVardiyaId do
    begin
      suspend;
      exit;
    end
  end
end
