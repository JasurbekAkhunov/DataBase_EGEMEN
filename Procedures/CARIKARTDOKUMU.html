CREATE PROCEDURE CARIKARTDOKUMU(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    ASINIF VARCHAR(1),
    AKULLANICIID INTEGER,
    AMINBAKIYE DOUBLE PRECISION,
    AMINBAKIYEDOVIZID VARCHAR(5),
    AHAREKETTIPI INTEGER)
RETURNS (
    CARIID INTEGER,
    FIRMAID VARCHAR(20),
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    CARITIPIADI VARCHAR(20),
    BORCALACAK VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    DOVIZKURU1 DOUBLE PRECISION,
    DOVIZKURU2 DOUBLE PRECISION,
    VALORTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    BORCBAKIYE DOUBLE PRECISION,
    ALACAKBAKIYE DOUBLE PRECISION)
AS
declare variable FirmaGrup1 varchar(4);
declare variable FirmaGrup2 varchar(4);
declare variable FirmaGrup3 varchar(4);
declare variable okey integer;
declare variable KayitSayisi integer;
declare Variable ToplamBorc double precision;
declare variable ToplamAlacak double precision;
declare variable Personelmi integer;
declare variable PersonelCariYetkisiVar integer;
begin
  select PersonelCariYetkisiVar from Kullanici where KullaniciID = :AKullaniciId into :PersonelCariYetkisiVar;

  For Select FirmaId, Personelmi from CariKartFirmaDokumu(:ABaslangicFirmaId, :ABitisFirmaId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id, :ABitisGrup3Id,
                                         :ABitisTarihi, :AFirmaAralikli, :AFirmaGrup1Aralikli, :AFirmaGrup2Aralikli, :AFirmaGrup3Aralikli, :ASinif, :AMinBakiye, :AMinBakiyeDovizId)
  into :FirmaId, :Personelmi do
  begin
    Okey = 1;
    if (Personelmi = 1 and PersonelCariYetkisiVar = 0) then Okey = 0;

    if (okey = 1) then
    begin
      ToplamBorc = 0;
      ToplamAlacak = 0;

      for select  BorcAlacak, sum( Meblag / DovizKuru1 * DovizKuru2)  from cari
           where FirmaId = :FirmaId and tarih < :ABaslangicTarihi and (Cari.Sinif = :ASinif or :ASinif is null)
           group by Cari.BorcAlacak
           into :BorcAlacak, :Meblag do
      begin
        if (BorcAlacak = 'B') then ToplamBorc = Meblag;
        else if (BorcAlacak = 'A') then ToplamAlacak = Meblag;
      end


      CariId = -1;
      Tarih  = null;
      Aciklama = 'Devir';
      CaritipiAdi = null;
      BorcAlacak = null;
      Meblag = null;
      DovizId = null;
      DovizKuru1 = null;
      DovizKuru2 = null;
      ValorTarihi = null;
      Notu = null;
      KullaniciAdi = null;
      DegisimTarihi = null;
      Borc = null;
      Alacak = Null;
      BorcBakiye = null;
      AlacakBakiye = Null;
      if (ToplamBorc > 0) then Borc = ToplamBorc;
      if (ToplamAlacak > 0) then Alacak = ToplamAlacak;
      if (toplamBorc is Null) then ToplamBorc = 0;
      if (ToplamAlacak is Null) then ToplamAlacak = 0;
      if (ToplamBorc > ToplamAlacak) then BorcBakiye = ToplamBorc - ToplamAlacak;
      else AlacakBakiye = ToplamAlacak - ToplamBorc;
      if (BorcBakiye = 0) then BorcBakiye = null;
      if (AlacakBakiye = 0) then AlacakBakiye = null;

      if ((ToplamBorc > 0) or (ToplamAlacak > 0)) then suspend;

      for select CariId, Cari.FirmaId, Tarih, Aciklama, BorcAlacak, Meblag, DovizId, DovizKuru1, DovizKuru2, ValorTarihi, Notu, DegisimTarihi, CariTipi.Adi, Kullanici.Adi from Cari
      left join CariTipi on CariTipi.CariTipiId = Cari.CariTipiId
      left join Kullanici on Kullanici.KullaniciId = Cari.KullaniciId
      where Cari.FirmaId = :FirmaId and
            Cari.Tarih >= :ABaslangicTarihi and
            Cari.Tarih <= :ABitisTarihi and
            (Cari.Sinif = :ASinif or :ASinif is null) and
            (Cari.CariTipiId = :AHareketTipi or :AHareketTipi is null)
      order by Cari.Tarih, Cari.CariId
      into :CariId, :FirmaId, :Tarih, :Aciklama, :BorcAlacak, :Meblag, :DovizId, :DovizKuru1, :DovizKuru2, :ValorTarihi, :Notu, :DegisimTarihi, :CariTipiAdi, :KullaniciAdi do
      begin
        Borc = Null;
        Alacak = Null;
        BorcBakiye = Null;
        AlacakBakiye = Null;

        if (BorcAlacak = 'B') then
        begin
          Borc = Meblag/DovizKuru1 * DovizKuru2;
          ToplamBorc = ToplamBorc + Borc;
        end
        else
        begin
          Alacak = Meblag / DovizKuru1 * DovizKuru2;
          ToplamAlacak = ToplamAlacak + Alacak;
        end
        if (toplamBorc is Null) then ToplamBorc = 0;
        if (ToplamAlacak is Null) then ToplamAlacak = 0;
        if (ToplamBorc > ToplamAlacak) then BorcBakiye = ToplamBorc - ToplamAlacak;
        else AlacakBakiye = ToplamAlacak - ToplamBorc;
        if (BorcBakiye = 0) then BorcBakiye = null;
        if (AlacakBakiye = 0) then AlacakBakiye = null;
        suspend;
      end
    end
  end
end
