CREATE PROCEDURE PARTISTOKIPTALRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATESLIMDURUMU INTEGER,
    ARAFID VARCHAR(10))
RETURNS (
    PARTISTOKIPTALYIL INTEGER,
    PARTISTOKIPTALID INTEGER,
    PARTISTOKIPTALIDSTR VARCHAR(30),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(20),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    SONCIKISKILO NUMERIC(12,3),
    SONCIKISMETRE NUMERIC(12,3),
    IPTALEDENKULLANICIID INTEGER,
    IPTALEDENKULLANICIADI VARCHAR(40),
    IPTALTARIHI TIMESTAMP,
    IPTALNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    YAPILANASAMALAR VARCHAR(1024),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    RAFID VARCHAR(10),
    TESLIMALANKULLANICIID INTEGER,
    TESLIMALANKULLANICIADI VARCHAR(40),
    TESLIMALINMATARIHI TIMESTAMP)
AS
declare variable FReceteIdAsamaAdi varchar(50);
begin
  For Select PartiStokIptal.PartiStokIptalYil, PartiStokIptal.PartiStokIptalId, Parti.PartiYil, Parti.PartiId, PartiStokIptal.PartiSinifId,
             Parti.FirmaId, Firma.Adi,
             PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.Metre, PartiStok.SonCikisKilo, PartiStok.SonCikisMetre,
             PartiStokIptal.IptalEdenKullaniciId, IPK.Adi, PartiStokIptal.IptalTarihi,
             PartiStokIptal.IptalNotu, PartiStokIptal.RafId, PartiStokIptal.TeslimAlanKullaniciId, TAK.Adi, PartiStokIptal.TeslimAlinmaTarihi,
             Parti.IslemTipiId, IslemTipi.Adi from PartiStokIptal
    Left Join PartiStok on PartiStok.PartiSinifId = PartiStokIptal.PartiSinifId
    Left Join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    Left Join Firma on Firma.FirmaId = Parti.FirmaId
    Left Join Kullanici IPK on IPK.KullaniciId = PartiStokIptal.IptalEdenKullaniciId
    Left Join Kullanici TAK on TAK.KullaniciId = PartiStokIptal.TeslimAlanKullaniciId
    Left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    where PartiStokIptal.IptalTarihi >= :ABaslangicTarihi and PartiStokIptal.IptalTarihi <= :ABitisTarihi  and PartiStokIptal.DesenTasarimIptal = 0 and
        (:ATeslimDurumu is null or (PartiStokIptal.TeslimAlanKullaniciId is null and :ATeslimDurumu = 0) or
                                   (PartiStokIptal.TeslimAlanKullaniciId is not null and :ATeslimDurumu = 1)) and (PartiStokIptal.RafId = :ARafId or :ARafId is Null)
    into :PartiStokIptalYil, :PartiStokIptalId, :PartiYil, :PartiId, :PartiSinifId, :FirmaId, :FirmaAdi, :HamAdiToplami, :Kilo, :Metre, :SonCikisKilo, :SonCikisMetre,
         :IptalEdenKullaniciId, :IptalEdenKullaniciAdi, :IptalTarihi, :IptalNotu, :RafId, :TeslimAlanKullaniciId, :TeslimAlanKullaniciAdi, :TeslimAlinmaTarihi,
         :IslemTipiId, :IslemTipiAdi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    PartiStokIptalIdStr = '(' || EgemenIntegerToStr(PartiStokIptalYil)  || ') ' || EgemenIntegerToStr(PartiStokIptalId) || ' ';

    YapilanAsamalar = '';

    for Select coalesce(PartiAsamalari.ReceteId, Asama.Adi) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamalari.SerbestAsama = 0 and
            PartiAsamaStok.GirisTarihi is not null and Asama.ParasalDegeriVar = 1
      order by PartiAsamalari.Sira
      into :FReceteIdAsamaAdi do
    begin
      if (Char_Length(YapilanAsamalar) + Char_Length(FReceteIdAsamaAdi) < 1020) then
      begin
        if (YapilanAsamalar <> '') then YapilanAsamalar = YapilanAsamalar || ',';
        YapilanAsamalar = YapilanAsamalar || FReceteIdAsamaAdi;
      end
    end

    suspend;
  end
end
