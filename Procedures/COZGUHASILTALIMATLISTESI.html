CREATE PROCEDURE COZGUHASILTALIMATLISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID FIRMAKODUSIZE,
    ACOZGUHASIL INTEGER)
RETURNS (
    COZGUISEMRIKAYITID INTEGER,
    HASILISEMRIKAYITID INTEGER,
    COZGUARTIHASILISEMRI SMALLINT,
    COZGUARTIHASILISEMRISTR VARCHAR(5),
    ISEMRIID VARCHAR(15),
    TARIH TIMESTAMP,
    DOKUMAHAMID INTEGER,
    DOKUMAHAMADI ADI255,
    IPLIKHAMID INTEGER,
    IPLIKHAMADI ADI255,
    BOYASIPARISKAYITID INTEGER,
    SIPARISNO VARCHAR(40),
    ORMESIPARISDETAYID INTEGER,
    LOT VARCHAR(150),
    MAKINAHIZI INTEGER,
    BASKIGUCU INTEGER,
    KONIK INTEGER,
    PARTIBOYU DOUBLE PRECISION,
    LEVENTENI INTEGER,
    MAKINALAR VARCHAR(50),
    NOTU MEMO,
    BASLAMATARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    KAPALI VARCHAR(7),
    FIRMAID FIRMAKODUSIZE,
    INDIGOISEMRIKAYITID INTEGER,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    IPLIKHAMADI2 ADI255,
    IPLIKHAMADI3 ADI255,
    IPLIKHAMADI4 ADI255,
    FIRMAADI VARCHAR(100),
    COZGUTALIMATMIKTARI DOUBLE PRECISION,
    COZGUURETIMMIKTARI DOUBLE PRECISION,
    COZGUTALIMATKALANMIKTAR DOUBLE PRECISION)
AS
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
begin

  select shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir into :shtFasonUretim, :shtUretim;
  if (ACozguHasil = 7) then/*İndigo*/
  begin
    For Select IndigoIsEmri.IndigoIsEmriKayitID, IndigoIsEmri.IsEmriID, IndigoIsEmri.Tarih, IndigoIsEmri.DokumaHamId, DokumaHam.Adi,
               IndigoIsEmri.IplikHamId, IplikHam.Adi, IndigoIsEmri.BoyaSiparisKayitID, '(' || BoyaSiparis.BoyaSiparisYil || ') ' || BoyaSiparis.BoyaSiparisID,
               IndigoIsEmri.OrmeSiparisDetayId, IndigoIsEmri.Lot,
               IndigoIsEmri.Notu, IndigoIsEmri.BaslamaTarihi, IndigoIsEmri.BitisTarihi, IK.ADI InsertKullaniciAdi, IndigoIsEmri.InsertTarihi,
               SK.ADI KullaniciAdi, IndigoIsEmri.DegisimTarihi, iif(IndigoIsEmri.Kapali = 1, 'Kapalı', ''), IndigoIsEmri.FirmaID, IndigoIsEmri.ReceteID, Recete.Adi,
               IplikHam.Adi2, IplikHam.Adi3, IplikHam.Adi4, Firma.Adi FROM IndigoIsEmri
      Left Join Ham IplikHam On IplikHam.HamId = IndigoIsEmri.IplikHamId
      Left Join Ham DokumaHam On DokumaHam.HamId = IndigoIsEmri.DokumaHamId
      Left Join Kullanici IK ON IK.KullaniciID  = IndigoIsEmri.InsertKullaniciID
      Left Join Kullanici SK ON SK.KullaniciID  = IndigoIsEmri.KullaniciID
      Left Join BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitID  = IndigoIsEmri.BoyaSiparisKayitID
      Left Join Recete  ON Recete.ReceteID  = IndigoIsEmri.ReceteID
      left join Firma on Firma.FirmaId = IndigoIsEmri.FirmaId
      Where IndigoIsEmri.Tarih >= :ABaslangicTarihi and IndigoIsEmri.Tarih <= :ABitisTarihi and
           (StrKarsilastir(IndigoIsEmri.FirmaID,:AFirmaID) = 1 or :AFirmaID = '-1')
      into :IndigoIsemriKayitID, :IsEmriID, :Tarih, :DokumaHamId, :DokumaHamAdi, :IplikHamId, :IplikHamAdi, :BoyaSiparisKayitID, :SiparisNo,
           :OrmeSiparisDetayId, :Lot, :Notu, :BaslamaTarihi, :BitisTarihi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi,
           :DegisimTarihi, :Kapali, :FirmaID, :ReceteId, :ReceteAdi, :IplikHamAdi2, :IplikHamAdi3, :IplikHamAdi4, :FirmaAdi do
    begin
      suspend;
    end

  end if (ACozguHasil = 0) then/*Çözgü*/
  begin
    For Select CozguIsEmri.CozguIsEmriKayitID, CozguIsEmri.IsEmriID, CozguIsEmri.Tarih, CozguIsEmri.DokumaHamId, DokumaHam.Adi,
               CozguIsEmri.IplikHamId, IplikHam.Adi, CozguIsEmri.BoyaSiparisKayitID, '(' || BoyaSiparis.BoyaSiparisYil || ') ' || BoyaSiparis.BoyaSiparisID,
               CozguIsEmri.OrmeSiparisDetayId, CozguIsEmri.Lot, CozguIsEmri.PartiBoyu, CozguIsEmri.LeventEni, CozguIsEmri.Makinalar,
               CozguIsEmri.Notu, CozguIsEmri.BaslamaTarihi, CozguIsEmri.BitisTarihi, IK.ADI InsertKullaniciAdi, CozguIsEmri.InsertTarihi,
               SK.ADI KullaniciAdi, CozguIsEmri.DegisimTarihi, iif(CozguIsEmri.Kapali = 1, 'Kapalı', ''), CozguIsEmri.FirmaID,
               CozguIsEmri.MakinaHizi, CozguIsEmri.BaskiGucu, CozguIsEmri.Konik, CozguIsEmri.CozguArtiHasilIsEmri ,
               IplikHam.Adi2, IplikHam.Adi3, IplikHam.Adi4, Firma.Adi  FROM CozguIsEmri
      Left Join Ham IplikHam On IplikHam.HamId = CozguIsEmri.IplikHamId
      Left Join Ham DokumaHam On DokumaHam.HamId = CozguIsEmri.DokumaHamId
      Left Join Kullanici IK ON IK.KullaniciID  = CozguIsEmri.InsertKullaniciID
      Left Join Kullanici SK ON SK.KullaniciID  = CozguIsEmri.KullaniciID
      Left Join BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitID  = CozguIsEmri.BoyaSiparisKayitID
      left join Firma On Firma.FirmaId = CozguIsemri.FirmaId
      Where CozguIsEmri.Tarih >= :ABaslangicTarihi and CozguIsEmri.Tarih <= :ABitisTarihi and
           (StrKarsilastir(CozguIsEmri.FirmaID,:AFirmaID) = 1 or :AFirmaID = '-1')
      into :CozguIsemriKayitID, :IsEmriID, :Tarih, :DokumaHamId, :DokumaHamAdi, :IplikHamId, :IplikHamAdi, :BoyaSiparisKayitID, :SiparisNo,
           :OrmeSiparisDetayId, :Lot, :PartiBoyu, :LeventEni, :Makinalar, :Notu, :BaslamaTarihi, :BitisTarihi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi,
           :DegisimTarihi, :Kapali, :FirmaID, :MakinaHizi, :BaskiGucu, :Konik, :CozguArtiHasilIsEmri,
           :IplikHamAdi2, :IplikHamAdi3, :IplikHamAdi4, :FirmaAdi do
    begin
      if (CozguArtiHasilIsEmri = 1) then CozguArtiHasilIsEmriStr = 'Evet';
      else CozguArtiHasilIsEmriStr = '';

      select  sum(CozguIsEmriDetay.CozguLeventMetre * CozguIsEmriDetay.LeventAdedi) from CozguIsEmriDetay where CozguIsEmriDetay.CozguIsemriKayitID = :CozguIsemriKayitID
        into :CozguTalimatMiktari;
      Select sum(Miktar) from StokHareket where StokHareket.CozguIsemriKayitID = :CozguIsemriKayitID and
        (StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim) and StokHareket.HamIade = 0
        into :CozguUretimMiktari;

        CozguTalimatKalanMiktar = CozguTalimatMiktari - CozguUretimMiktari;

      suspend;
    end
  end
  else/*Hasil*/
  begin
    For Select HasilIsEmri.HasilIsEmriKayitID, HasilIsEmri.IsEmriID, HasilIsEmri.Tarih, HasilIsEmri.DokumaHamId, DokumaHam.Adi,
               HasilIsEmri.IplikHamId, IplikHam.Adi, HasilIsEmri.BoyaSiparisKayitID, '(' || BoyaSiparis.BoyaSiparisYil || ') ' || BoyaSiparis.BoyaSiparisID,
               HasilIsEmri.OrmeSiparisDetayId, HasilIsEmri.Lot, HasilIsEmri.PartiBoyu, HasilIsEmri.LeventEni, HasilIsEmri.Makinalar,
               HasilIsEmri.Notu, HasilIsEmri.BaslamaTarihi, HasilIsEmri.BitisTarihi, IK.ADI InsertKullaniciAdi, HasilIsEmri.InsertTarihi,
               SK.ADI KullaniciAdi, HasilIsEmri.DegisimTarihi, iif(HasilIsEmri.Kapali = 1, 'Kapalı', ''), HasilIsEmri.FirmaID,
               IplikHam.Adi2, IplikHam.Adi3, IplikHam.Adi4, Firma.Adi FROM HasilIsEmri
      Left Join Ham IplikHam On IplikHam.HamId = HasilIsEmri.IplikHamId
      Left Join Ham DokumaHam On DokumaHam.HamId = HasilIsEmri.DokumaHamId
      Left Join Kullanici IK ON IK.KullaniciID  = HasilIsEmri.InsertKullaniciID
      Left Join Kullanici SK ON SK.KullaniciID  = HasilIsEmri.KullaniciID
      Left Join BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitID  = HasilIsEmri.BoyaSiparisKayitID
      left join Firma On Firma.FirmaId = HasilIsEmri.FirmaId
      Where HasilIsEmri.Tarih >= :ABaslangicTarihi and HasilIsEmri.Tarih <= :ABitisTarihi and
           (StrKarsilastir(HasilIsEmri.FirmaID,:AFirmaID) = 1 or :AFirmaID = '-1')
      into :HasilIsemriKayitID, :IsEmriID, :Tarih, :DokumaHamId, :DokumaHamAdi, :IplikHamId, :IplikHamAdi, :BoyaSiparisKayitID, :SiparisNo,
           :OrmeSiparisDetayId, :Lot, :PartiBoyu, :LeventEni, :Makinalar, :Notu, :BaslamaTarihi, :BitisTarihi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi,
           :DegisimTarihi, :Kapali, :FirmaID, :IplikHamAdi2, :IplikHamAdi3, :IplikHamAdi4, :FirmaAdi do
    begin
      suspend;
    end

  end
end
