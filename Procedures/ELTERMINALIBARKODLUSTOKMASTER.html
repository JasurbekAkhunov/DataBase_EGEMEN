CREATE PROCEDURE ELTERMINALIBARKODLUSTOKMASTER(
    AHAREKETTIPI INTEGER,
    AFIRMAID VARCHAR(20),
    ADEPOID VARCHAR(10),
    ATRANSFERDEPOID VARCHAR(10),
    AEVRAKSERINO VARCHAR(2),
    AEVRAKNO INTEGER,
    ADOVIZID VARCHAR(5),
    AODEMETARIHI TIMESTAMP,
    AFATURATARIHI TIMESTAMP,
    AKULLANICIID INTEGER)
RETURNS (
    EVRAKKAYITID INTEGER)
AS
declare variable shtAlis         integer;
declare variable shtSatis        integer;
declare variable shtDepoTransfer integer;
begin
  select shtAlis, shtSatis, ShtDepoTransfer from StokHareketTipiGetir into :shtAlis, :shtSatis, :ShtDepoTransfer;

  if ((AHareketTipi = shtAlis) or (AHareketTipi = shtSatis))  then
  begin
    EvrakKayitID = gen_id(Fatura_Id_Gen, 1);

    Insert Into Fatura(FaturaID, HareketTipi, FirmaID, DepoId, SeriNo, FaturaNo, DovizID, OdemeTarihi, FaturaTarihi, InsertKullaniciID) values
                      (:EvrakKayitID, :AHareketTipi, :AFirmaID, :ADepoId, :AEvrakSeriNo, :AEvrakNo, :ADovizID, :AOdemeTarihi, :AFaturaTarihi, :AKullaniciID);

  end
  else if (AHareketTipi = ShtDepoTransfer) then
  begin
    EvrakKayitID = gen_id(Stok_Id_Gen, 1);

    Insert Into Stok(StokID, HareketTipi, FirmaID, DepoId, TransferDepoID, SeriNo, FisNo, DovizID, FisTarihi, InsertKullaniciID) values
                    (:EvrakKayitID, :AHareketTipi, :AFirmaID, :ADepoId, :ATransferDepoID, :AEvrakSeriNo, :AEvrakNo, :ADovizID, :AFaturaTarihi, :AKullaniciID);

  end
  else EvrakKayitID = -1;

  suspend;
end
