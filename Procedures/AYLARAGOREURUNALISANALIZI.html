CREATE PROCEDURE AYLARAGOREURUNALISANALIZI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    AURUNTIPIID INTEGER,
    AFIRMAID VARCHAR(20))
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    BIRIMID VARCHAR(2),
    URUNKATEGORIID VARCHAR(4),
    DEVIRMIKTARI DOUBLE PRECISION,
    FIFODEVIRMALIYETI DOUBLE PRECISION,
    DEVIRTUTAR DOUBLE PRECISION,
    AY1MIKTAR DOUBLE PRECISION,
    AY1TUTAR DOUBLE PRECISION,
    AY2MIKTAR DOUBLE PRECISION,
    AY2TUTAR DOUBLE PRECISION,
    AY3MIKTAR DOUBLE PRECISION,
    AY3TUTAR DOUBLE PRECISION,
    AY4MIKTAR DOUBLE PRECISION,
    AY4TUTAR DOUBLE PRECISION,
    AY5MIKTAR DOUBLE PRECISION,
    AY5TUTAR DOUBLE PRECISION,
    AY6MIKTAR DOUBLE PRECISION,
    AY6TUTAR DOUBLE PRECISION,
    AY7MIKTAR DOUBLE PRECISION,
    AY7TUTAR DOUBLE PRECISION,
    AY8MIKTAR DOUBLE PRECISION,
    AY8TUTAR DOUBLE PRECISION,
    AY9MIKTAR DOUBLE PRECISION,
    AY9TUTAR DOUBLE PRECISION,
    AY10MIKTAR DOUBLE PRECISION,
    AY10TUTAR DOUBLE PRECISION,
    AY11MIKTAR DOUBLE PRECISION,
    AY11TUTAR DOUBLE PRECISION,
    AY12MIKTAR DOUBLE PRECISION,
    AY12TUTAR DOUBLE PRECISION,
    STOKMIKTARI DOUBLE PRECISION,
    FIFOSTOKMALIYETI DOUBLE PRECISION,
    ALISFIYATI DOUBLE PRECISION,
    ALISFIYATIDOVIZID VARCHAR(5),
    RAPORALISFIYATI DOUBLE PRECISION,
    RAPORDOVIZID VARCHAR(5),
    URUNTIPIID INTEGER,
    URUNTIPIADI VARCHAR(20),
    URUNGRUP1ADI VARCHAR(30),
    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30))
AS
declare variable FMiktar        Double Precision;
declare variable FTutar         Double Precision;
declare variable FUrunId                    varchar(20);
declare variable FLastUrunId                varchar(20);
declare variable FUrunAdi                   varchar(80);
declare variable FUrunGrup1                 varchar(4);
declare variable FUrunGrup2                 varchar(4);
declare variable FUrunGrup3                 varchar(4);
declare variable FUrunGrup4                 varchar(4);
declare variable FBirimId                   varchar(2);
declare variable FUrunKategoriId            varchar(4);
declare variable FAy                        integer;
declare variable Okey                       integer;
declare variable FDate                      Date;
declare variable shtAlis                    integer;
declare variable shtAlisIade                integer;
declare variable HareketVar                 integer;
declare variable DovizKuru                  Double Precision;
declare variable LastDovizId                varchar(5);
declare variable DefaultBirimCarpan     double precision;
begin
  select shtAlis, ShtAlisIade from StokHareketTipiGetir into :shtAlis, :ShtAlisIade;

  FLastUrunId = '';
  LastDovizId = '';
  Okey = 0;

  Select StokTabanDovizId from sistem where sistemid = 1 into :RaporDovizId;

  for Select Urun.UrunId, Urun.Adi, Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, UBD.BirimId, UBD.Carpan,
             Urun.UrunKategoriId, UrunFiyat.AlisFiyati, UrunFiyat.AlisFiyatiDovizId, Urun.UrunTipiId, UrunTipi.Adi,
             UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi from Urun
    left join UrunBirim UBD on UBD.UrunId = urun.Urunid and UBD.DefaultBirim = 1
    left join UrunFiyat on UrunFiyat.UrunId = urun.Urunid
    left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
    left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
    left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
    left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
    left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
    where (:AUrunAralikli = 0 or (:AUrunAralikli = 1 and Urun.UrunId >= :ABaslangicUrunId) and (Urun.UrunId <= :ABitisUrunId or :ABitisUrunId  = '')) and
           ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicGrup1Id  and (UrunGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
           ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicGrup2Id and (UrunGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
           ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicGrup3Id and (UrunGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
           ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicGrup4Id and (UrunGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = '')) and
           (Urun.UrunTipiId = :AUrunTipiId or :AUrunTipiId is null)
     into :UrunId, :UrunAdi, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :BirimId, :DefaultBirimCarpan, :UrunKategoriId, :AlisFiyati, :AlisFiyatiDovizId,
       :UrunTipiId, :UrunTipiAdi, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi do
  begin
    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, UrunGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      DevirMiktari = 0;
      Ay1Miktar    = 0;
      Ay2Miktar    = 0;
      Ay3Miktar    = 0;
      Ay4Miktar    = 0;
      Ay5Miktar    = 0;
      Ay6Miktar    = 0;
      Ay7Miktar    = 0;
      Ay8Miktar    = 0;
      Ay9Miktar    = 0;
      Ay10Miktar   = 0;
      Ay11Miktar   = 0;
      Ay12Miktar   = 0;
      Ay1Tutar    = 0;
      Ay2Tutar    = 0;
      Ay3Tutar    = 0;
      Ay4Tutar    = 0;
      Ay5Tutar    = 0;
      Ay6Tutar    = 0;
      Ay7Tutar    = 0;
      Ay8Tutar    = 0;
      Ay9Tutar    = 0;
      Ay10Tutar   = 0;
      Ay11Tutar   = 0;
      Ay12Tutar   = 0;
      StokMiktari  = 0;
      HareketVar = 0;

      RaporAlisFiyati = AlisFiyati;

      if (AlisFiyatiDovizId <> RaporDovizId and AlisFiyati <> 0) then
      begin
        if (AlisFiyatiDovizId <> LastDovizId) then
          Select DovizKuru from EnYakinDovizKuruGetir(:AlisFiyatiDovizId, :RaporDovizId, Current_Date) into :DovizKuru;

        RaporAlisFiyati = RaporAlisFiyati * DovizKuru;
        LastDovizId = AlisFiyatiDovizId;
      end

      Select sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from stokdetay
        left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
        where Stokdetay.Urunid = :UrunId and StokDetay.Tarih < :ABaslangicTarihi and StokDetay.HareketTipi <> 5
      into :DevirMiktari;

      Select sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from stokdetay
        left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
        where Stokdetay.Urunid = :UrunId and StokDetay.Tarih <= :ABitisTarihi and StokDetay.HareketTipi <> 5
      into :StokMiktari;

      For Select f_Month(StokDetay.Tarih), sum(StokDetay.Miktar * StokDetay.GirisCikis / HUB.Carpan * DUB.Carpan),
                 sum(StokDetay.Miktar * StokDetay.NetBirimFiyat * StokDetay.GirisCikis / TabanDovizKuru1 * TabanDovizKuru2) from stokdetay
      left join UrunBirim HUB on HUB.UrunId = stokdetay.Urunid and HUB.urunbirimid = stokdetay.urunbirimid
      left join UrunBirim DUB on DUB.UrunId = stokdetay.Urunid and DUB.DefaultBirim = 1
      where Stokdetay.UrunID = :UrunId and  (Stokdetay.HareketTipi = :shtAlis or Stokdetay.HareketTipi = :shtAlisIade) and
            StokDetay.Tarih >= :ABaslangicTarihi and StokDetay.Tarih <= :ABitisTarihi and
           (StokDetay.FirmaId = :AFirmaID or :AFirmaID is null)
      group by f_Month(StokDetay.Tarih)
      order by 1
      into :FAy, :FMiktar, :FTutar do
      begin
        HareketVar = 1;

        if (FAY = 1) then
        begin
          Ay1Miktar = Yuvarla(FMiktar,1000);
          Ay1Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 2) then
        begin
          Ay2Miktar = Yuvarla(FMiktar,1000);
          Ay2Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 3) then
        begin
          Ay3Miktar = Yuvarla(FMiktar,1000);
          Ay3Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 4) then
        begin
          Ay4Miktar = Yuvarla(FMiktar,1000);
          Ay4Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 5) then
        begin
          Ay5Miktar = Yuvarla(FMiktar,1000);
          Ay5Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 6) then
        begin
          Ay6Miktar = Yuvarla(FMiktar,1000);
          Ay6Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 7) then
        begin
          Ay7Miktar = Yuvarla(FMiktar,1000);
          Ay7Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 8) then
        begin
          Ay8Miktar = Yuvarla(FMiktar,1000);
          Ay8Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 9) then
        begin
          Ay9Miktar = Yuvarla(FMiktar,1000);
          Ay9Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 10) then
        begin
          Ay10Miktar = Yuvarla(FMiktar,1000);
          Ay10Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 11) then
        begin
          Ay11Miktar = Yuvarla(FMiktar,1000);
          Ay11Tutar = Yuvarla(FTutar, 100);
        end
        else if (FAY = 12) then
        begin
          Ay12Miktar = Yuvarla(FMiktar,1000);
          Ay12Tutar = Yuvarla(FTutar, 100);
        end
      end

      if (HareketVar = 1 or DevirMiktari <> 0 or StokMiktari <> 0) then
      begin
        Select Sum(KalanMaliyet) from UrunFifoMaliyetiGetir(:UrunId, :DefaultBirimCarpan, :ABaslangicTarihi-1, :RaporDovizId) into :FifoDevirMaliyeti;
        Select Sum(KalanMaliyet) from UrunFifoMaliyetiGetir(:UrunId, :DefaultBirimCarpan, :ABitisTarihi, :RaporDovizId) into :FifoStokMaliyeti;
        suspend;
      end
    end
  end
end
