CREATE PROCEDURE SEVKEHAZIRLARINRAPORU_1(
    AFIRMAID VARCHAR(20),
    AHAMTIPIID INTEGER,
    ATEMSILCIID INTEGER,
    AFASON INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AHAMID INTEGER,
    AKALITEID VARCHAR(20),
    ALABRECETEID INTEGER,
    AIPLIKCINSIID INTEGER,
    APARTIGRUPID VARCHAR(4),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ADISUNITEDURUMU INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ADESENID VARCHAR(20),
    AUNITEID INTEGER,
    AYURTICIYURTDISI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAGRUP1ID FIRMAGRUPKODUSIZE,
    FIRMAGRUP2ID FIRMAGRUPKODUSIZE,
    FIRMAGRUP1ADI ADI30,
    FIRMAGRUP2ADI ADI30,
    HAMKILO NUMERIC(12,3),
    HAMMETRE NUMERIC(12,3),
    TAMIRKILO NUMERIC(12,3),
    TAMIRMETRE NUMERIC(12,3),
    HAMORAN DOUBLE PRECISION,
    TAMIRORAN DOUBLE PRECISION,
    YURTICIYURTDISI INTEGER)
AS
declare variable TamirKiloToplam        Numeric(12,3);
declare variable HamKiloToplam          Numeric(12,3);
declare variable Kilo                   Numeric(12,3);
declare variable TamirMetreToplam       Numeric(12,3);
declare variable HamMetreToplam         Numeric(12,3);
declare variable Metre                  Numeric(12,3);
declare variable IslemTipi              integer;
declare variable FFirmaId               varchar(20);
declare variable FFirmaAdi              varchar(100);
declare variable SonFirmaId             varchar(20);
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef       integer;
declare variable ABoyaSiparisKayitId integer;
declare variable hstOrmeKumas   integer;
declare variable hstRibana      integer;
declare variable hstAksesuar    integer;
declare variable hstDokumaKumas integer;

begin
  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  Select hstIplik, hstElyaf, hstTelef, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar;

  if (ABoyaSiparisYil is not Null) then
  begin
    Select BoyaSiparis.BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId into :ABoyaSiparisKayitId;
    if (ABoyaSiparisKayitId is null) then ABoyaSiparisKayitId = -1;
  end

  TamirKiloToplam = 0;
  HamKiloToplam = 0;
  TamirMetreToplam = 0;
  HamMetreToplam = 0;
  SonFirmaId = '';

  select sum(PartiStok.Kilo-PartiStok.BrutSevkKilo), sum(PartiStok.Metre-PartiStok.BrutSevkMetre) from PartiStok
    left join Ham on Ham.HamId = PartiStok.HamId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitID = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join IplikCinsi On IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join DesenVaryant on DesenVaryant.DesenVaryantID = Parti.DesenVaryantID
    where PartiStok.durumu = 7 and PartiStok.PartiStokIptalKayitId is null and (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and Parti.IslemTipiId = :itTamir and
         (partiStok.HamId = :AHamId or :AHamId is null) and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
            ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 1) or
            (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 2) or
            (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 3) or :AHamTipiId is null) and
          (Firma.TemsilciID = :ATemsilciId or :ATemsilciId is null) and
          (parti.Fason = :AFason or :AFason is null) and
          (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
          (BoyaSiparisDetay.KaliteID = :AKaliteId or :AKaliteId is Null) and
          (BoyaSiparisDetay.LabReceteId = :ALabReceteId or :ALabReceteId is Null) and
          (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null) and
          (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is Null) and
          (IplikCinsi.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is Null) and
          ( (PartiStok.UniteID is null and :ADisUniteDurumu = 0) or
            (PartiStok.UniteID is not null and :ADisUniteDurumu = 1) or :ADisUniteDurumu is null) and
          (parti.PartiGrupId = :APartiGrupId or :APartiGrupId is Null) and
          (BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          (DesenVaryant.DesenId = :ADesenId or :ADesenId is null) and
          (PartiStok.UniteID = :AUniteID or :AUniteID is null)and
          (BoyaSiparis.Yurtdisi = :AYurticiYurtdisi or :AYurticiYurtdisi is null)

    into :TamirKiloToplam, :TamirMetreToplam;

  select sum(PartiStok.Kilo-PartiStok.BrutSevkKilo), sum(PartiStok.Metre-PartiStok.BrutSevkMetre) from PartiStok
    left join Ham on Ham.HamId = PartiStok.HamId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitID = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join IplikCinsi On IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join DesenVaryant on DesenVaryant.DesenVaryantID = Parti.DesenVaryantID
  where PartiStok.durumu = 7 and PartiStok.PartiStokIptalKayitId is null and (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and Parti.IslemTipiId <> :itTamir and
       (partiStok.HamId = :AHamId or :AHamId is null) and
       ( ((Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
         ((Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 1) or
         (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 2) or
         ( Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null) and
       (Firma.TemsilciID = :ATemsilciId or :ATemsilciId is null) and
       (parti.Fason = :AFason or :AFason is null) and
       (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
       (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
       (BoyaSiparisDetay.KaliteID = :AKaliteId or :AKaliteId is Null) and
       (BoyaSiparisDetay.LabReceteId = :ALabReceteId or :ALabReceteId is Null) and
       (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null) and
       (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is Null) and
       (IplikCinsi.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is Null) and
       ( (PartiStok.UniteID is null and :ADisUniteDurumu = 0) or
         (PartiStok.UniteID is not null and :ADisUniteDurumu = 1) or :ADisUniteDurumu is null) and
       (parti.PartiGrupId = :APartiGrupId or :APartiGrupId is Null) and
       (BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
       (DesenVaryant.DesenId = :ADesenId or :ADesenId is null) and
       (PartiStok.UniteID  =:AUniteID or :AUniteID is null) and
       (BoyaSiparis.Yurtdisi = :AYurticiYurtdisi or :AYurticiYurtdisi is null)

  into :HamKiloToplam, :HamMetreToplam;

  for select Parti.FirmaId, Firma.Adi, Parti.IslemTipiId, sum(PartiStok.Kilo-PartiStok.BrutSevkKilo), sum(PartiStok.Metre-PartiStok.BrutSevkMetre), FG1.FirmaGrupId, FG1.Adi, FG2.FirmaGrupId, FG2.Adi,
             BoyaSiparis.Yurtdisi from PartiStok
    left join Ham on Ham.HamId = PartiStok.HamId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join FirmaGrup FG1 on FG1.FirmaGrupId = Firma.FirmaGrup1
    left join FirmaGrup FG2 on FG2.FirmaGrupId = Firma.FirmaGrup2
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitID = Parti.BoyaSiparisKayitId
    left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join IplikCinsi On IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join DesenVaryant on DesenVaryant.DesenVaryantID = Parti.DesenVaryantID
    where PartiStok.durumu = 7 and PartiStok.PartiStokIptalKayitId is null and (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
       (partiStok.HamId = :AHamId or :AHamId is null) and
       ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
         ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 1) or
         (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 2) or
         ( Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 1) or :AHamTipiId is null) and
       (Firma.TemsilciID = :ATemsilciId or :ATemsilciId is null) and
       (parti.Fason = :AFason or :AFason is null) and
       (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
       (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
       (BoyaSiparisDetay.KaliteID = :AKaliteId or :AKaliteId is Null) and
       (BoyaSiparisDetay.LabReceteId = :ALabReceteId or :ALabReceteId is Null) and
       (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null) and
       (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is Null) and
       (IplikCinsi.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is Null) and
       ( (PartiStok.UniteID is null and :ADisUniteDurumu = 0) or
         (PartiStok.UniteID is not null and :ADisUniteDurumu = 1) or :ADisUniteDurumu is null) and
       (parti.PartiGrupId = :APartiGrupId or :APartiGrupId is Null) and
       (BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
       (DesenVaryant.DesenId = :ADesenId or :ADesenId is null) and
       (PartiStok.UniteID  = :AUniteID or :AUniteID is null) and
       (BoyaSiparis.Yurtdisi = :AYurticiYurtdisi or :AYurticiYurtdisi is null)
    group by Parti.FirmaId, Firma.Adi, FG1.FirmaGrupId, FG1.Adi, FG2.FirmaGrupId, FG2.Adi, Parti.islemtipiid, BoyaSiparis.Yurtdisi
    into :FFirmaId, :FFirmaAdi, :IslemTipi, :Kilo, :Metre, :FirmaGrup1Id, :FirmaGrup1Adi, :FirmaGrup2Id, :FirmaGrup2Adi,
         :YurticiYurtdisi  do
  begin
    if (SonFirmaId <> FFirmaId) then
    begin
      if (SonFirmaId <> '') then
      begin
        HamOran = 0;
        TamirOran = 0;
        if (HamKiloToplam <> 0) then HamOran = HamKilo * 100 / HamKiloToplam;
        if (TamirKiloToplam <> 0) then TamirOran = TamirKilo * 100 / TamirKiloToplam;
        suspend;
      end
      SonFirmaId = FFirmaId;
      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      HamKilo = 0;
      TamirKilo = 0;
      HamMetre = 0;
      TamirMetre = 0;
      HamOran = 0;
      TamirOran = 0;
    end
    if (IslemTipi <> itTamir) then
    begin
      HamKilo = HamKilo + Kilo;
      HamMetre = HamMetre + Metre;
    end
    else
    begin
      TamirKilo = TamirKilo + Kilo;
      TamirMetre = TamirMetre + Metre;
    end
  end

  if (SonFirmaId <> '') then
  begin
    HamOran = 0;
    TamirOran = 0;
    if (HamKiloToplam <> 0) then HamOran = HamKilo * 100 / HamKiloToplam;
    if (TamirKiloToplam <> 0) then TamirOran = TamirKilo * 100 / TamirKiloToplam;
    suspend;
  end
end
