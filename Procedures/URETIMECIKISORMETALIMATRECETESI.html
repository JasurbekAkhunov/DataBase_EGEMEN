CREATE PROCEDURE URETIMECIKISORMETALIMATRECETESI(
    AORMEMAKINAID VARCHAR(10),
    AFIRMAID VARCHAR(20))
RETURNS (
    ORMEPLANLAMAID INTEGER,
    ORMEPLANLAMARECETEID INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    KULLANIMORANI DOUBLE PRECISION,
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(150),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION)
AS
declare variable TahminiFireOrani Double Precision;
declare variable hstIplik       integer;
begin
  Select hstIplik from HamTipiGetir into :hstIplik;

  For select UretimeCikisOrmeTalimatListesi.OrmePlanlamaId, OrmePlanlamaRecete.OrmePlanlamaReceteId, OrmePlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, OrmePlanlamaRecete.RenkId, Renk.Adi, OrmePlanlamaRecete.KullanimOrani, OrmePlanlamaRecete.CikisMiktari, OrmePlanlamaRecete.IadeMiktari,
             UretimeCikisOrmeTalimatListesi.PlanlananMiktar * OrmePlanlamaRecete.KullanimOrani / 100, OrmeSiparisDetay.TahminiFireOrani,
             OrmePlanlamaRecete.MelanjKodu, OrmePlanlamaRecete.Lot, OrmePlanlamaRecete.UreticiId, Uretici.Adi from UretimeCikisOrmeTalimatListesi(:AOrmeMakinaId, :AFirmaId)
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = UretimeCikisOrmeTalimatListesi.OrmeSiparisDetayId
  left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = UretimeCikisOrmeTalimatListesi.OrmePlanlamaId
  left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
  left join Renk on Renk.RenkId = OrmePlanlamaRecete.RenkId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  left join Uretici on Uretici.UreticiId = OrmePlanlamaRecete.UreticiId
  where OrmePlanlamaRecete.OrmePlanlamaId is not null
  into :OrmePlanlamaId, :OrmePlanlamaReceteId, :HamId, :HamTipiId, :HamTipiAdi,
       :HamAdi, :RenkId, :RenkAdi, :KullanimOrani, :CikisMiktari, :IadeMiktari, :KullanilacakMiktar, :TahminiFireOrani,
       :MelanjKodu, :Lot, :UreticiId, :UreticiAdi do
  begin
    if (TahminiFireOrani is null) then TahminiFireOrani = 0;

    if (HamTipiId = hstIplik) then KullanilacakMiktar = KullanilacakMiktar / ( (100 - TahminiFireOrani) / 100);
    suspend;
  end
end
