CREATE PROCEDURE ORMEKALITEKONTROLGETIR(
    AORMEPLANLAMAID INTEGER,
    AORMEVARDIYAID INTEGER)
RETURNS (
    ORMEKALITEKONTROLID INTEGER,
    ORMEPLANLAMAID INTEGER,
    ORMEVARDIYAID INTEGER,
    STOKHAREKETID INTEGER,
    TARIH TIMESTAMP,
    VARDIYATARIHI TIMESTAMP,
    VARDIYA INTEGER,
    VARDIYAAMIRIPERSONELADI VARCHAR(40),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    AYARTOPU INTEGER,
    AYARTOPUSTR VARCHAR(10),
    SAKATHAMHATATIPIID INTEGER,
    SAKATHAMHATATIPIADI VARCHAR(20),
    RAFID VARCHAR(10),
    FISSERINO VARCHAR(2),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    TOPNO INTEGER,
    MIKTAR DOUBLE PRECISION,
    SIPARISFIRMAADI VARCHAR(100),
    PUS INTEGER,
    FEINE INTEGER,
    LOT VARCHAR(50),
    MAKINAUSTUGRAMAJ INTEGER,
    DINLENMISGRAMAJ INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    URETIMECIKISIRSALIYE IRSALIYENOSIZE,
    ETIKETYAZDIRILDI EGEBOOLEAN,
    ETIKETYAZDIRMATARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    SIPARISGRUPADI VARCHAR(30),
    MELANJKODU VARCHAR(20),
    URETIMNOTU MEMO,
    RENKADI VARCHAR(50),
    KALITEID VARCHAR(20),
    KILO BOYAMIKTARSIZE,
    METRE BOYAMIKTARSIZE,
    PALETNO INTEGER,
    SATISIRSALIYE IRSALIYENOSIZE)
AS
declare variable TupAcikEnMayli integer;
begin
  For Select OrmeKaliteKontrol.OrmeKaliteKontrolId, OrmeKaliteKontrol.OrmePlanlamaId, OrmeKaliteKontrol.OrmeVardiyaId, OrmeKaliteKontrol.StokHareketId,
             OrmeKaliteKontrol.Tarih, OrmeKaliteKontrol.vardiyaTarihi, OrmeKaliteKontrol.Vardiya, VAP.Adi, OrmeKaliteKontrol.PersonelId, P.Adi, OrmeKaliteKontrol.RafId,
             OrmeKaliteKontrol.TopNo, OrmeKaliteKontrol.Sakat, OrmeKaliteKontrol.AyarTopu, OrmeKaliteKontrol.Miktar, IK.Adi, OrmeKaliteKontrol.InsertTarihi, SK.Adi, OrmeKaliteKontrol.DegisimTarihi,
             OrmeFis.Serino, OrmeFis.FisNo, OrmeFis.Fistarihi, Firma.Adi, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
             OrmeSiparisDetay.TupAcikEnMayli, OrmePlanlama.DinlenmisGramaj, OrmePlanlama.MakinaUstuGramaj,
             OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4,
             OrmeKaliteKontrol.SakatHamHataTipiId, HamHataTipi.Adi, OrmePlanlama.Lot, OrmeIrsaliye.IrsaliyeNo,
             OrmeKaliteKontrol.EtiketYazdirildi, OrmeKaliteKontrol.EtiketYazdirmaTarihi, SiparisGrup.Adi, Coalesce(OrmeKaliteKontrol.MelanjKodu, OrmeSiparisDetay.MelanjKodu), OrmeKaliteKontrol.UretimNotu, Renk.Adi, OrmeSiparisDetay.KaliteId,
             OrmeKaliteKontrol.Metre, OrmeKaliteKontrol.Kilo, OrmeKaliteKontrol.PaletNo, SI.OrmeIrsaliyeId from OrmeKaliteKontrol
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeKaliteKontrol.OrmePlanlamaId
    left join OrmeIrsaliye on OrmeIrsaliye.ormeirsaliyeid = OrmeKaliteKontrol.OrmeUCikisIrsaliyeId
    left join OrmeIrsaliye SI on SI.ormeirsaliyeid = OrmeKaliteKontrol.OrmeIrsaliyeId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join StokHareket on StokHareket.StokHareketId = OrmeKaliteKontrol.StokHareketId
    left join OrmeFis on OrmeFis.OrmeFisId  = StokHareket.OrmeFisId
    left join Personel VAP on VAP.PersonelId  = OrmeKaliteKontrol.VardiyaAmiriPersonelId
    left join Personel P on P.PersonelId  = OrmeKaliteKontrol.PersonelId
    left join HamHataTipi on HamHataTipi.HamHataTipiId  = OrmeKaliteKontrol.SakatHamHataTipiId
    left join Kullanici IK on IK.KullaniciId  = OrmeKaliteKontrol.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = OrmeKaliteKontrol.KullaniciId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
    where OrmeKaliteKontrol.OrmePlanlamaId = :AOrmePlanlamaId and
          (OrmeKaliteKontrol.OrmeVardiyaId = :AOrmeVardiyaId or :AOrmeVardiyaId is null)
    order by OrmeKaliteKontrol.OrmePlanlamaId, OrmeKaliteKontrol.Tarih
  into :OrmeKaliteKontrolId, :OrmePlanlamaId, :OrmeVardiyaId, :StokHareketId, :Tarih, :VardiyaTarihi, :Vardiya, :VardiyaAmiriPersonelAdi,
       :PersonelId, :PersonelAdi, :RafId, :TopNo, :Sakat, :AyarTopu,  :Miktar, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :FisSerino, :FisNo, :Fistarihi, :SiparisFirmaAdi, :Pus, :Feine, :TupAcikEnMayli, :DinlenmisGramaj, :MakinaUstuGramaj,
       :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :SakatHamHataTipiId, :SakatHamHataTipiAdi, :Lot, :UretimeCikisIrsaliye,
       :EtiketYazdirildi, :EtiketYazdirmaTarihi, :SiparisGrupAdi, :MelanjKodu, :UretimNotu, :RenkAdi, :KaliteId, :Metre, :Kilo, :PaletNo, :SatisIrsaliye do
  begin
  Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;
    SakatStr    = iif(Sakat = 1, 'Sakat', '');
    AyarTopuStr = iif(AyarTopu=1, 'Ayar Topu', '');
    suspend;
  end
end
