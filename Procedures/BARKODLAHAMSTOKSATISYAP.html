CREATE PROCEDURE BARKODLAHAMSTOKSATISYAP(
    ABOYAIRSALIYEID INTEGER,
    AFIRMAID VARCHAR(20),
    ABARKODID VARCHAR(20),
    AKULLANICIID INTEGER,
    ABOYASIPARISDETAYID INTEGER)
AS
declare variable BoyaIrsaliyeDetayId integer;
declare variable PaketId integer;
declare variable IK     varchar(2);
declare variable TF     integer;
declare variable FirmaId      varchar(20);
declare variable FirmaKodumuz varchar(20);
declare variable OrmeFirmaKodumuz varchar(20);
declare variable EgemenFirmaId varchar(20);
declare variable ParcaliSevk    integer;
declare variable RaftakiKilo    double precision;
declare variable RafDetayId    integer;
declare variable TopKilo    double precision;
declare variable TopBrutKilo    double precision;
declare variable UygunBoyaIrsaliyeDetayId    integer;
declare variable TopMetre    double precision;
begin
  if (Char_Length(ABarkodId) < 3) then exception BarkodHatali1;

  IK = EgemenCopyStr(ABarkodId, 1, 2);

  ABarkodId = EgemenCopyStr(ABarkodId, 3, Char_Length(ABarkodId) - 2);
  TF = EgemenStrToIntOlurmu(ABarkodId);

  if (TF = 0) then exception BarkodHatali1;

  PaketId = EgemenStrToInt(ABarkodId);

  Select FirmaKodumuz, OrmeFirmaKodumuz, EgemenFirmaId from Sistem where sistemid = 1 into :FirmaKodumuz, :OrmeFirmaKodumuz, :EgemenFirmaId;

  if (IK = 'H-') then
  begin
    Select HamStokMasterTop.BoyaIrsaliyeDetayId from HamStokMasterTop
      where HamStokMasterTop.HamStokMasterTopId = :PaketId
      into :BoyaIrsaliyeDetayId;
  end

  if (BoyaIrsaliyeDetayId is not null) then exception BuPaketinIrsaliyesiKesilmis;

  select first 1 RafDetay.RafDetayId,
    (RafDetay.Kilo + RafDetay.AktarmadanIadeEdienKilo - RafDetay.IadeEdilecekKilo - RafDetay.AktarmayaCikanKilo - RafDetay.IadeEdilenKilo - RafDetay.PartilenenKilo - RafDetay.RafTransferKilo - RafDetay.SatisKilo),
    HamStokMasterTop.Kilo, HamStokMasterTop.BrutKilo, HamStokMasterTop.Metre
    from HamStokMasterTop
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokMasterTop.HamStokMasterId
    left join HamStokDetay on HamStokDetay.HamStokMasterId = HamStokMaster.HamStokMasterId
    left join RafDetay on RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId and RafDetay.RafId = HamStokMasterTop.RafId
    where HamStokMasterTop.HamStokMasterTopId = :PaketId
    into :RafDetayId, :RaftakiKilo, :TopKilo, :TopBrutKilo, :TopMetre;

  if (RafDetayId is Null) then exception KayitHatasi 'Rafta ürün bulunamadı.';
  if (TopKilo > RaftakiKilo) then exception RaftakiKiloTopKilosundanAz;

  if (ABoyaSiparisDetayId is not Null) then
  begin
    select BoyaIrsaliyeDetayId from BoyaIrsaliyeDetay
      where BoyaIrsaliyeId = :ABoyaIrsaliyeId and
      RafDetayId = :RafDetayId and BoyaIrsaliyeDetay.FasonHamIade = 0 and BoyaIrsaliyeDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId
      into :UygunBoyaIrsaliyeDetayId;
  end
  else
  begin
    select BoyaIrsaliyeDetayId from BoyaIrsaliyeDetay
      where BoyaIrsaliyeId = :ABoyaIrsaliyeId and
      RafDetayId = :RafDetayId and BoyaIrsaliyeDetay.FasonHamIade = 0 and BoyaIrsaliyeDetay.BoyaSiparisDetayId is Null
      into :UygunBoyaIrsaliyeDetayId;
  end

  if (UygunBoyaIrsaliyeDetayId is Null) then
  begin
    UygunBoyaIrsaliyeDetayId = gen_id(BoyaIrsaliyeDetay_Id_Gen, 1);
    insert into boyairsaliyedetay(boyairsaliyedetayid, boyairsaliyeid, rafdetayid, fasonhamiade, kapadedi, brutkilo, kilo, brutmetre, metre, adet, insertkullaniciid, kullaniciid, boyasiparisdetayid)
    values(:UygunBoyaIrsaliyeDetayId, :ABoyaIrsaliyeId, :RafDetayId, 0, 1, :TopBrutKilo, :TopKilo, :TopMetre, :TopMetre, 0, :AKullaniciId, :AKullaniciId, :ABoyaSiparisDetayId);
  end
  else
  begin
    update BoyaIrsaliyeDetay
      set BRUTKILO = BRUTKILO + :TopBrutKilo,
      KILO = KILO + :TopKilo,
      BRUTMETRE = BRUTMETRE + :TopMetre,
      METRE = METRE + :TopMetre,
      KAPADEDI = KAPADEDI + 1
      where BoyaIrsaliyeDetayId = :UygunBoyaIrsaliyeDetayId;
  end

  update HamStokMasterTop set BoyaIrsaliyeDetayID = :UygunBoyaIrsaliyeDetayId
    where HamStokMasterTop.HamStokMasterTopId = :PaketId;
end
