CREATE PROCEDURE FATURAICMALI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASERINO SERINOSIZE5,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICFIRMAGRUP1ID VARCHAR(4),
    ABITISFIRMAGRUP1ID VARCHAR(4),
    ABASLANGICFIRMAGRUP2ID VARCHAR(4),
    ABITISFIRMAGRUP2ID VARCHAR(4),
    ABASLANGICFIRMAGRUP3ID VARCHAR(4),
    ABITISFIRMAGRUP3ID VARCHAR(4),
    ABASLANGICFATURAGRUPID VARCHAR(4),
    ABITISFATURAGRUPID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    AFATURAGRUPARALIKLI INTEGER,
    AENTEGRASYON INTEGER,
    ATEVKIFAT INTEGER,
    AALIS INTEGER,
    ASATIS INTEGER,
    AALISIADE INTEGER,
    ASATISIADE INTEGER,
    ABOYAFATURA INTEGER,
    AORMEURETIM INTEGER,
    ARAPORDOVIZID VARCHAR(5),
    ASATISTIPI INTEGER,
    ASINIF VARCHAR(1))
RETURNS (
    FATURAID INTEGER,
    FATURATIPI VARCHAR(30),
    TARIH TIMESTAMP,
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    ODEMETARIHI TIMESTAMP,
    FATURAGRUPID VARCHAR(4),
    FATURAGRUPADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    FIRMAGRUP1 VARCHAR(4),
    FIRMAGRUP2 VARCHAR(4),
    FIRMAGRUP3 VARCHAR(4),
    DOVIZID VARCHAR(5),
    FIRMADOVIZKURUSTR VARCHAR(40),
    BRUTTOPLAM DOUBLE PRECISION,
    ISKONTOTOPLAMI DOUBLE PRECISION,
    ISKONTO1TUTARI DOUBLE PRECISION,
    ISKONTO2TUTARI DOUBLE PRECISION,
    ISKONTO1ORANI DOUBLE PRECISION,
    ISKONTO2ORANI DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    FATURATUTARI DOUBLE PRECISION,
    RAPORFATURATUTARI DOUBLE PRECISION,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ENTEGRASYONSTR VARCHAR(3),
    TEVKIFATSTR VARCHAR(3),
    DOVIZKURUMESAJ VARCHAR(255),
    FIRMADOVIZKURU1 DOUBLE PRECISION,
    FIRMADOVIZKURU2 DOUBLE PRECISION,
    FIRMADOVIZID VARCHAR(5),
    FATURAMIKTAR DOUBLE PRECISION,
    VERGIDAIRESI VARCHAR(20),
    VERGIHESAPNO VARCHAR(20),
    MODULID VARCHAR(10),
    FIRMAFATURATUTARI DOUBLE PRECISION)
AS
declare variable Tevkifat integer;
declare variable Entegrasyon integer;
declare variable Okey integer;
declare variable HareketTipi integer;
declare variable DovizKuru Double Precision;
declare variable DovizKuruTarihi Date;
begin
  if (ABoyaFatura = 1) then
  begin
    For Select distinct BoyaFatura.BoyaFaturaId, BoyaFatura.HareketTipiId, BoyaFatura.FaturaTarihi, BoyaFatura.OdemeTarihi, BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo, BoyaFatura.FaturaGrupId,
        StokHareketGrup.Adi, BoyaFatura.FirmaId, Firma.Adi, Firma.Adi2, Firma.DovizId, Firma.FirmaGrup1, Firma.FirmaGrup2, Firma.FirmaGrup3,  BoyaFatura.DovizId, BoyaFatura.DovizKuru1,
        BoyaFatura.DovizKuru2, BoyaFatura.BrutToplam, BoyaFatura.ParasalIskonto, BoyaFatura.IskontoOrani,
        BoyaFatura.IskontoToplami, BoyaFatura.IndirilecekKdv, BoyaFatura.TahsilEdilecekKdv,
        BoyaFatura.Notu, BoyaFatura.Entegrasyon, BoyaFatura.Tevkifat, Firma.VergiDairesi, Firma.VergiHesapNo, 'Boya' from BoyaFatura
      left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = BoyaFatura.FaturaGrupId
      left join Firma on Firma.FirmaId = BoyaFatura.FirmaId
      left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaFaturaID = BoyaFatura.BoyaFaturaID
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayID = BoyaFaturaDetay.BoyaIrsaliyeDetayID
      left join Parti on Parti.PartiKayitID = BoyaIrsaliyeDetay.PartiKayitID
    where BoyaFatura.FaturaTarihi >= :ABaslangicTarihi and
          BoyaFatura.FaturaTarihi <= :ABitisTarihi and
         (BoyaFatura.FaturaSeriNo = :ASeriNo or :ASeriNo is null) and
         ((:AFirmaAralikli = 1 and (BoyaFatura.FirmaId >= :ABaslangicFirmaId or :ABaslangicFirmaId = '') and (BoyaFatura.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
         ((:AFaturaGrupAralikli = 1 and (BoyaFatura.FaturaGrupId >= :ABaslangicFaturaGrupId or :ABaslangicFaturaGrupId = '') and (BoyaFatura.FaturaGrupId <= :ABitisFaturaGrupId or :ABitisFaturaGrupId = '')) or (:AFaturaGrupAralikli = 0)) and
         ((:AFirmaGrup1Aralikli = 1 and (Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id or :ABaslangicFirmaGrup1Id = '') and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
         ((:AFirmaGrup2Aralikli = 1 and (Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id or :ABaslangicFirmaGrup2Id = '') and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
         ((:AFirmaGrup3Aralikli = 1 and (Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id or :ABaslangicFirmaGrup3Id = '') and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = '')) and
         (BoyaFatura.Entegrasyon = :AEntegrasyon or :AEntegrasyon is null) and
         (Parti.Fason = :ASatisTipi or :ASatisTipi is null) and
         (BoyaFatura.Tevkifat = :ATevkifat or :ATevkifat is null) and
         (BoyaFatura.Sinif = :ASinif or :ASinif is null)
    into :FaturaId, :HareketTipi, :Tarih, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FaturaGrupId, :FaturaGrupAdi, :FirmaId, :FirmaAdi, :FirmaAdi2, :FirmaDovizId,
         :FirmaGrup1, :FirmaGrup2, :FirmaGrup3,
         :DovizId, :FirmaDovizKuru1, :FirmaDovizKuru2, :BrutToplam, :Iskonto1Tutari, :Iskonto1Orani,
         :IskontoToplami, :IndirilecekKdv, :TahsilEdilecekKdv, :Notu, :Entegrasyon, :Tevkifat, :VergiDairesi, :VergiHesapNo, :ModulId do
    begin
      if (HareketTipi = 0) then FaturaTipi = 'Boya Satış';
      else if (HareketTipi = 1) then FaturaTipi = 'Ham Stok Alış Iade';
      else FaturaTipi = '';

      Select Sum(Miktar) from BoyaFaturaDetay where BoyaFaturaDetay.BoyaFaturaID = :FaturaId into :FaturaMiktar;

      Okey = 1;
      if (AFirmaAralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup1Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup1Id, FirmaGrup1) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup2Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup2Id, FirmaGrup2) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup3Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup3Id, FirmaGrup3) = 0) then okey = 0;

      if (okey = 1) then
      begin
        DovizKuruMesaj = Null;
        if (Entegrasyon = 0) then
        begin
         BrutToplam = 0;
         IskontoToplami = 0;
         IndirilecekKdv = 0;
         TahsilEdilecekKdv = 0;
         FaturaTutari = 0;
         RaporFaturaTutari = 0;
         EntegrasyonStr = '';
        end
        else
        begin
          EntegrasyonStr = 'Var';
          FaturaTutari = BrutToplam - IskontoToplami + TahsilEdilecekKdv;
          if (DovizId <> ARaporDovizId) then
          begin
            select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :Tarih) into :DovizKuru, :DovizKuruTarihi;
            if (Tarih <> DovizKuruTarihi) then DovizKuruMesaj = 'Fatura Tarihine Ait Kur Girisi Bulunamadigi icin ' || TimeStampToDateTimeStr(DovizKurutarihi) || ' Tarihli Kur ile Hesaplandi';
          end
          else DovizKuru = 1;

          RaporFaturaTutari = FaturaTutari * DovizKuru;
        end

        if (Tevkifat = 0) then TevkifatStr = '';
        else TevkifatStr = 'Var';
/*        DovizKuruMesaj = TrkOemConvert(DovizKuruMesaj);
        FaturaTipi = TrkOemConvert(FaturaTipi);
        EntegrasyonStr = TrkOemConvert(EntegrasyonStr);
*/
        if (FirmaDovizId <> DovizId) then FirmaDovizKuruStr = Format_Float('', FirmaDovizKuru1) || ' ' || DovizId || '-' || Format_Float('', FirmaDovizKuru2) || ' ' || FirmaDovizId;
        else FirmaDovizKuruStr = '';

        if (FirmaDovizId <> DovizId) then FirmaFaturaTutari = FaturaTutari * (:FirmaDovizKuru2);
        else FirmaFaturaTutari = FaturaTutari ;

        suspend;
      end
    end
  end
  TevkifatStr = '';

  if ((AAlis = 1 or ASatis = 1 or  AAlisIade = 1 or  ASatisIade = 1) and ASatisTipi is Null)then
  begin
    For Select Fatura.FaturaId, Fatura.HareketTipi, Fatura.FaturaTarihi, Fatura.OdemeTarihi, Fatura.SeriNo, Fatura.FaturaNo, Fatura.FaturaGrupId,
        StokHareketGrup.Adi, Fatura.FirmaId, Firma.Adi, Firma.Adi2, Firma.DovizId, Firma.FirmaGrup1, Firma.FirmaGrup2, Firma.FirmaGrup3, Fatura.DovizId,
        Fatura.FirmaDovizKuru1, Fatura.FirmaDovizKuru2, Fatura.Notu, Fatura.Entegrasyon,
        Fatura.BrutToplam, (Fatura.Iskonto1Tutari + Fatura.Iskonto2Tutari + Fatura.Iskonto1OranTutari + Fatura.Iskonto2OranTutari),
        Fatura.IndirilecekKdv, Fatura.TahsilEdilecekKdv, Fatura.Iskonto1Tutari, Fatura.Iskonto2Tutari, Fatura.Iskonto1Orani, Fatura.Iskonto2Orani,
        Firma.VergiDairesi, Firma.VergiHesapNo, 'Stok' from Fatura
      left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = Fatura.FaturaGrupId
      left join Firma on Firma.FirmaId = Fatura.FirmaId
    where Fatura.FaturaTarihi >= :ABaslangicTarihi and
          Fatura.FaturaTarihi <= :ABitisTarihi and
         (Fatura.SeriNo = :ASeriNo or :ASeriNo is null) and
         ((:AFirmaAralikli = 1 and Fatura.FirmaId >= :ABaslangicFirmaId and (Fatura.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
         ((:AFaturaGrupAralikli = 1 and (Fatura.FaturaGrupId >= :ABaslangicFaturaGrupId or :ABaslangicFaturaGrupId = '') and (Fatura.FaturaGrupId <= :ABitisFaturaGrupId or :ABitisFaturaGrupId = '')) or (:AFaturaGrupAralikli = 0)) and
         ((:AFirmaGrup1Aralikli = 1 and (Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id or :ABaslangicFirmaGrup1Id = '') and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
         ((:AFirmaGrup2Aralikli = 1 and (Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id or :ABaslangicFirmaGrup2Id = '') and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
         ((:AFirmaGrup3Aralikli = 1 and (Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id or :ABaslangicFirmaGrup3Id = '') and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = '')) and
         ((Fatura.HareketTipi = 1 and :AAlis = 1) or (Fatura.HareketTipi = 2 and :ASatis = 1) or (Fatura.HareketTipi = 3 and :AAlisIade = 1) or (Fatura.HareketTipi = 4 and :ASatisIade = 1)) and
         (Fatura.Entegrasyon = :AEntegrasyon or :AEntegrasyon is null) and
         (Fatura.Sinif = :ASinif or :ASinif is null)
    into :FaturaId, :HareketTipi, :Tarih, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FaturaGrupId, :FaturaGrupAdi, :FirmaId, :FirmaAdi, :FirmaAdi2, :FirmaDovizId,
         :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :DovizId, :FirmaDovizKuru1, :FirmaDovizKuru2, :Notu, :Entegrasyon, :BrutToplam, :IskontoToplami,
         :IndirilecekKdv, :TahsilEdilecekKdv, :Iskonto1Tutari, :Iskonto2Tutari, :Iskonto1Orani, :Iskonto2Orani, :VergiDairesi, :VergiHesapNo, :ModulId do
    begin
      Okey = 1;
      if (AFirmaAralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

      Select Sum(Miktar) from StokDetay where StokDetay.FaturaId = :FaturaId into :FaturaMiktar;

      if (okey = 1) then
      if (AFirmaGrup1Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup1Id, FirmaGrup1) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup2Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup2Id, FirmaGrup2) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup3Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup3Id, FirmaGrup3) = 0) then okey = 0;

      if (okey = 1) then
      begin
        if (HareketTipi = 1) then FaturaTipi = 'Alış';
        else if (HareketTipi = 2) then FaturaTipi = 'Satış';
        else if (HareketTipi = 3) then FaturaTipi = 'Alış İade';
        else if (HareketTipi = 4) then FaturaTipi = 'Satış İade';
        else FaturaTipi = '';

        DovizKuruMesaj = Null;

        if (Entegrasyon = 0) then
        begin
         BrutToplam = 0;
         IskontoToplami = 0;
         IndirilecekKdv = 0;
         TahsilEdilecekKdv = 0;
         FaturaTutari = 0;
         RaporFaturaTutari = 0;
         EntegrasyonStr = '';
        end
        else
        begin
          EntegrasyonStr = 'Var';
          FaturaTutari = BrutToplam - IskontoToplami + TahsilEdilecekKdv;
          if (DovizId <> ARaporDovizId) then
          begin
            select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :Tarih) into :DovizKuru, :DovizKuruTarihi;
            if (Tarih <> DovizKuruTarihi) then DovizKuruMesaj = 'Fatura Tarihine Ait Kur Girişi Bulunamadığı için ' || TimeStampToDateTimeStr(DovizKurutarihi) || ' Tarihli Kur ile Hesaplandı';
          end
          else DovizKuru = 1;
        end
/*        DovizKuruMesaj = TrkOemConvert(DovizKuruMesaj);
        FaturaTipi = TrkOemConvert(FaturaTipi);
        EntegrasyonStr = TrkOemConvert(EntegrasyonStr);
*/
        RaporFaturaTutari = FaturaTutari * DovizKuru;

        if (FirmaDovizId <> DovizId) then FirmaDovizKuruStr = Format_Float('', FirmaDovizKuru1) || ' ' || DovizId || '-' || Format_Float('', FirmaDovizKuru2) || ' ' || FirmaDovizId;
        else FirmaDovizKuruStr = '';

        if (FirmaDovizId <> DovizId) then FirmaFaturaTutari = FaturaTutari * (:FirmaDovizKuru2);
        else FirmaFaturaTutari = FaturaTutari ;

        suspend;
      end
    end
  end

  if (AAlis = 1 or ASatis = 1 or  AAlisIade = 1 or  ASatisIade = 1 or AOrmeUretim = 1) then
  begin
    Iskonto2Tutari = Null;
    Iskonto2Orani = Null;

    For Select distinct OrmeFatura.OrmeFaturaId, OrmeFatura.HareketTipi, OrmeFatura.FaturaTarihi, OrmeFatura.OdemeTarihi, OrmeFatura.SeriNo, OrmeFatura.FaturaNo, OrmeFatura.FaturaGrupId,
        StokHareketGrup.Adi, OrmeFatura.FirmaId, Firma.Adi, Firma.Adi2, Firma.DovizId, Firma.FirmaGrup1, Firma.FirmaGrup2, Firma.FirmaGrup3, OrmeFatura.DovizId,
        OrmeFatura.FirmaDovizKuru1, OrmeFatura.FirmaDovizKuru2, OrmeFatura.Notu, OrmeFatura.Entegrasyon,
        OrmeFatura.BrutToplam, OrmeFatura.IskontoToplami,
        OrmeFatura.IndirilecekKdv, OrmeFatura.TahsilEdilecekKdv, OrmeFatura.ParasalIskonto, OrmeFatura.FaturaIskontosuOrani, Firma.VergiDairesi, Firma.VergiHesapNo, 'Orgu' from OrmeFatura
      left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = OrmeFatura.FaturaGrupId
      left join Firma on Firma.FirmaId = OrmeFatura.FirmaId
      left join StokHareket on StokHareket.OrmeFaturaID = OrmeFatura.OrmeFaturaID
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = StokHareket.BoyaSiparisKayitID
    where OrmeFatura.FaturaTarihi >= :ABaslangicTarihi and
          OrmeFatura.FaturaTarihi <= :ABitisTarihi and
         (OrmeFatura.SeriNo = :ASeriNo or :ASeriNo is null) and
         ((:AFirmaAralikli = 1 and OrmeFatura.FirmaId >= :ABaslangicFirmaId and (OrmeFatura.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
         ((:AFaturaGrupAralikli = 1 and (OrmeFatura.FaturaGrupId >= :ABaslangicFaturaGrupId or :ABaslangicFaturaGrupId = '') and (OrmeFatura.FaturaGrupId <= :ABitisFaturaGrupId or :ABitisFaturaGrupId = '')) or (:AFaturaGrupAralikli = 0)) and
         ((:AFirmaGrup1Aralikli = 1 and (Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id or :ABaslangicFirmaGrup1Id = '') and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
         ((:AFirmaGrup2Aralikli = 1 and (Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id or :ABaslangicFirmaGrup2Id = '') and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
         ((:AFirmaGrup3Aralikli = 1 and (Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id or :ABaslangicFirmaGrup3Id = '') and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = '')) and
         ((OrmeFatura.HareketTipi = 1 and :AAlis = 1) or (OrmeFatura.HareketTipi = 2 and :ASatis = 1) or (OrmeFatura.HareketTipi = 3 and :AAlisIade = 1) or (OrmeFatura.HareketTipi = 4 and :ASatisIade = 1) or (OrmeFatura.HareketTipi = 8 and :AOrmeUretim = 1)) and
         (BoyaSiparis.Fason = :ASatisTipi or :ASatisTipi is null) and
         (OrmeFatura.Entegrasyon = :AEntegrasyon or :AEntegrasyon is null) and
         (OrmeFatura.Sinif = :ASinif or :ASinif is null)
    into :FaturaId, :HareketTipi, :Tarih, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FaturaGrupId, :FaturaGrupAdi, :FirmaId, :FirmaAdi, :FirmaAdi2, :FirmaDovizId,
         :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :DovizId, :FirmaDovizKuru1, :FirmaDovizKuru2, :Notu, :Entegrasyon, :BrutToplam, :IskontoToplami,
         :IndirilecekKdv, :TahsilEdilecekKdv, :Iskonto1Tutari, :Iskonto1Orani, :VergiDairesi, :VergiHesapNo, :ModulId do
    begin
      Okey = 1;
      if (AFirmaAralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

      Select Sum(FaturaMiktar) from StokHareket where StokHareket.OrmeFaturaId = :FaturaId into :FaturaMiktar;

      if (okey = 1) then
      if (AFirmaGrup1Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup1Id, FirmaGrup1) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup2Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup2Id, FirmaGrup2) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup3Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup3Id, FirmaGrup3) = 0) then okey = 0;

      if (okey = 1) then
      begin
        if (HareketTipi = 1) then FaturaTipi = 'Alış';
        else if (HareketTipi = 2) then FaturaTipi = 'Satış';
        else if (HareketTipi = 3) then FaturaTipi = 'Alış İade';
        else if (HareketTipi = 4) then FaturaTipi = 'Satış İade';
        else if (HareketTipi = 8) then FaturaTipi = 'Üretim';
        else FaturaTipi = '';

        DovizKuruMesaj = Null;

        if (Entegrasyon = 0) then
        begin
         BrutToplam = 0;
         IskontoToplami = 0;
         IndirilecekKdv = 0;
         TahsilEdilecekKdv = 0;
         FaturaTutari = 0;
         RaporFaturaTutari = 0;
         EntegrasyonStr = '';
        end
        else
        begin
          EntegrasyonStr = 'Var';
          FaturaTutari = BrutToplam - IskontoToplami + TahsilEdilecekKdv;
          if (DovizId <> ARaporDovizId) then
          begin
            select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :Tarih) into :DovizKuru, :DovizKuruTarihi;
            if (Tarih <> DovizKuruTarihi) then DovizKuruMesaj = 'Fatura Tarihine Ait Kur Girişi Bulunamadığı için ' || TimeStampToDateTimeStr(DovizKurutarihi) || ' Tarihli Kur ile Hesaplandı';
          end
          else DovizKuru = 1;
        end
/*        DovizKuruMesaj = TrkOemConvert(DovizKuruMesaj);
        FaturaTipi = TrkOemConvert(FaturaTipi);
        EntegrasyonStr = TrkOemConvert(EntegrasyonStr);
*/
        RaporFaturaTutari = FaturaTutari * DovizKuru;

        if (FirmaDovizId <> DovizId) then FirmaDovizKuruStr = Format_Float('', FirmaDovizKuru1) || ' ' || DovizId || '-' || Format_Float('', FirmaDovizKuru2) || ' ' || FirmaDovizId;
        else FirmaDovizKuruStr = '';

        if (FirmaDovizId <> DovizId) then FirmaFaturaTutari = FaturaTutari * (:FirmaDovizKuru2);
        else FirmaFaturaTutari = FaturaTutari ;

        suspend;
      end
    end
  end
end
