CREATE PROCEDURE PFSIPARISPROSESVEISLEMLERIGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    ADI VARCHAR(60),
    ASAMAID INTEGER,
    ILAVEISLEMLERID INTEGER,
    SIPARISILAVEISLEMLERID INTEGER)
AS
begin
/* fiyatlandirma ekrani icin yapıldı */
  for Select PartiStok.PartiSinifId, Asama.Adi, SiparisProsesAsamalari.AsamaID from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayID
    left join SiparisProsesAsamalari on SiparisProsesAsamalari.SiparisProsesID = BoyaSiparisDetay.SiparisProsesID
    left join Asama on Asama.AsamaID = SiparisProsesAsamalari.AsamaID
    where PartiStok.PartiKayitId = :APartiKayitId and PartiStok.PartiStokIptalKayitID is null and SiparisProsesAsamalari.AsamaID is not null and Asama.ParasalDegeriVar = 1
    order by PartiStok.PartiSinifId, SiparisProsesAsamalari.Sira
    into :PartiSinifId, :Adi, :AsamaId do
  begin
    SiparisIlaveIslemlerId = null;
    IlaveIslemlerId = Null;
    suspend;
  end

  for Select PartiStok.PartiSinifId, IlaveIslemler.Adi, SiparisIlaveIslemler.IlaveIslemlerId, SiparisIlaveIslemler.SiparisIlaveIslemlerId FROM PartiStok
    left join SiparisIlaveIslemler on SiparisIlaveIslemler.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayID
    LEFT JOIN IlaveIslemler ON IlaveIslemler.IlaveIslemlerId = SiparisIlaveIslemler.IlaveIslemlerId
    where PartiStok.PartiKayitId = :APartiKayitId and PartiStok.PartiStokIptalKayitID is null and SiparisIlaveIslemler.IlaveIslemlerId is not null
    order by PartiStok.PartiSinifId, IlaveIslemler.Adi
    into :PartiSinifId, :Adi, :IlaveIslemlerId, :SiparisIlaveIslemlerId do
  begin
    AsamaID = Null;
    suspend;
  end
end
