CREATE PROCEDURE MAKINABILGIKAYDET(
    AMAKINAID VARCHAR(10),
    ANEM DOUBLE PRECISION,
    AISI DOUBLE PRECISION,
    AHIZ DOUBLE PRECISION,
    ABACANEM DOUBLE PRECISION,
    AELEKTIRIK DOUBLE PRECISION,
    ADOGALGAZ DOUBLE PRECISION)
AS
declare variable MakinaBilgiID integer;
declare variable Farkvar       integer;
declare variable FNem          Double Precision;
declare variable FIsi          Double Precision;
declare variable FHiz          Double Precision;
declare variable FBacaNem      Double Precision;
declare variable FElektirik    Double Precision;
declare variable FDogalgaz     Double Precision;
declare variable FTarih        TimeStamp;
declare variable FNow          TimeStamp;
declare variable ZamanFarki    Double Precision;
declare variable YTarih varchar(10);
declare variable YSaat varchar(8);
declare variable YSonTarih varchar(20);
begin
  FNow = Current_TimeStamp(0);

  if (ANem is null) then ANem = 0;
  if (AIsi is null) then AIsi = 0;
  if (AHiz is null) then AHiz = 0;
  if (ABacaNem is null) then ABacaNem = 0;

  Select MakinaBilgiID from Makina where Makina.MakinaID = :AMakinaID into :MakinaBilgiID;

  if (MakinaBilgiID is not null) then
  begin
    Select Tarih, Nem, Isi, Hiz, BacaNem, Elektirik, Dogalgaz from MakinaBilgi
      where MakinaBilgi.MakinaBilgiID = :MakinaBilgiID
      into :FTarih, :FNem, :FIsi, :FHiz, :FBacaNem, :FElektirik, :FDogalgaz;

    if (FNem is null) then FNem = 0;
    if (FIsi is null) then FIsi = 0;
    if (FHiz is null) then FHiz = 0;
    if (FBacaNem is null) then FBacaNem = 0;
    if (FElektirik is null) then FElektirik = 0;
    if (FDogalgaz is null) then FDogalgaz = 0;
    if (AElektirik is null) then AElektirik = FElektirik;
    if (ADogalgaz is null) then ADogalgaz = FDogalgaz;
    if (FTarih is null) then FTarih = '01.01.2000 00:00:00';

    ZamanFarki = FNow - FTarih;


    if (ANem <> FNem or AIsi <> FIsi or AHiz <> FHiz or ABacaNem <> FBacaNem or AElektirik <> FElektirik or ADogalgaz <> FDogalgaz or ZamanFarki > 0.020834) then FarkVar = 1;
    else FarkVar = 0;
  end
  else FTarih = '01.01.2000 00:00:00';

  YTarih = current_date;
  select substring(current_time(0) from 1 for 5) from sistem into :YSaat;
  YSonTarih = YTarih || ' ' || YSaat || ':00';
  if (FTarih <> YSonTarih) then
    insert into MakinaBilgi(MakinaId, Nem, ISi, Hiz, BacaNem, Elektirik, Dogalgaz, Tarih)
                     Values(:AMakinaId, :ANem, :AISi, :AHiz, :ABacaNem, :AElektirik, :ADogalgaz, :YSonTarih);
  else
  begin
    /*if (FarkVar = 1) then update MakinaBilgi set MakinaId = :AMakinaId, Nem = :ANem, Isi = :AIsi, BacaNem = :ABacaNem, Elektirik = :AElektirik, Dogalgaz = :ADogalgaz where Tarih = :FTarih;*/
    if (FarkVar = 1) then
    begin
      update MakinaBilgi set MakinaId = :AMakinaId, Nem = :ANem, Isi = :AIsi, BacaNem = :ABacaNem, Elektirik = :AElektirik, Dogalgaz = :ADogalgaz where MakinaBilgiID = :MakinaBilgiID;
    end
  end
end
