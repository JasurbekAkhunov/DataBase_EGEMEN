CREATE PROCEDURE HGSIPARISDETAYAKSESUARGETIR(
    AHGSIPARISKAYITID INTEGER)
RETURNS (
    HGSIPARISDETAYAKSESUARID INTEGER,
    HGSIPARISKAYITID INTEGER,
    HGSIPARISDETAYID INTEGER,
    MODELKUMASID INTEGER,
    MODELBEDENID INTEGER,
    URUNID URUNKODUSIZE,
    URUNADI ADI80,
    MIKTAR BOYAMIKTARSIZE,
    FIREORANI DOUBLE PRECISION,
    SIPARISMIKTARI BOYAMIKTARSIZE,
    IHTIYACMIKTARI BOYAMIKTARSIZE,
    RENKID INTEGER,
    RENKADI ADI50,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATDOVIZID DOVIZSIZE,
    NOTU MEMO,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40),
    BIRIMI ADI10)
AS
begin

  For Select HGSiparisDetayAksesuar.HGSiparisDetayAksesuarID, HGSiparisDetayAksesuar.HGSiparisKAyitID, HGSiparisDetayAksesuar.HGSiparisDetayID, HGSiparisDetayAksesuar.ModelKumasID, HGSiparisDetayAksesuar.UrunId, Urun.Adi,
             HGSiparisDetayAksesuar.FireOrani, Urun.RenkId, Renk.Adi, HGSiparisDetayAksesuar.BirimFiyat, HGSiparisDetayAksesuar.BirimFiyatDovizId, HGSiparisDetayAksesuar.InsertKullaniciId,
             HGSiparisDetayAksesuar.InsertTarihi, HGSiparisDetayAksesuar.KullaniciId, HGSiparisDetayAksesuar.DegisimTarihi,
             IK.Adi, SK.Adi, HGSiparisDetayAksesuar.Notu, HGSiparisDetayAksesuar.Miktar, HGSiparisDetayAksesuar.ModelBedenID, Coalesce(Birim.Adi, UrunDefaultBirimi.Adi) From HGSiparisDetayAksesuar
    left join Urun on Urun.UrunID = HGSiparisDetayAksesuar.UrunId
    left join Renk on Renk.RenkId = Urun.RenkId
    left join Kullanici IK on IK.KullaniciId  = HGSiparisDetayAksesuar.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = HGSiparisDetayAksesuar.KullaniciId
    left join Birim on Birim.BirimId = HGSiparisDetayAksesuar.BirimId
    left join UrunBirim on UrunBirim.Urunid = HGSiparisDetayAksesuar.UrunId and UrunBirim.DefaultBirim = 1
    left join Birim UrunDefaultBirimi on UrunDefaultBirimi.BirimId = UrunBirim.BirimId
    Where HGSiparisDetayAksesuar.HGSiparisKayitID = :AHGSiparisKayitID
    into :HGSiparisDetayAksesuarID, :HGSiparisKAyitID, :HGSiparisDetayID, :ModelKumasID, :UrunId, :UrunAdi, :FireOrani, :RenkId, :RenkAdi, :BirimFiyat, :BirimFiyatDovizId,
         :InsertKullaniciId, :InsertTarihi, :KullaniciId, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi, :Notu, :Miktar, :ModelBedenID, :Birimi do
  begin
    SiparisMiktari = 0;
    IhtiyacMiktari = 0;
    Select Sum(Miktar) from HGSiparisDetayUrun
      where HGSiparisDetayUrun.HGSiparisDetayId = :HGSiparisDetayId and (HGSiparisDetayUrun.ModelKumasID = :ModelKumasID or :ModelKumasID is Null) and (HGSiparisDetayUrun.ModelBedenId = :ModelBedenId or :ModelBedenId is Null)
    into :SiparisMiktari;

    IhtiyacMiktari = Round((Miktar * iif(FireOrani > 0, FireOrani / 100.0 + 1, 1)) * SiparisMiktari, 2);

    Suspend;
  end
end
