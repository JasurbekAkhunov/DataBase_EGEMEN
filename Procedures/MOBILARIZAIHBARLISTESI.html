CREATE PROCEDURE MOBILARIZAIHBARLISTESI(
    AKULLANICIID INTEGER,
    AFILTERBUGUN SMALLINT,
    AFILTERDUN SMALLINT,
    AFILTERHAFTALIK SMALLINT)
RETURNS (
    ARIZAIHBARID INTEGER,
    MAKINAID VARCHAR(10),
    BASLATANPERSONELID INTEGER,
    BASLATANPERSONELADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENPERSONELID INTEGER,
    BITIRENPERSONELADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    BITTI INTEGER,
    BITTISTR VARCHAR(5),
    ARIZATIPIID INTEGER,
    ARIZATIPIADI VARCHAR(20),
    PLANLIBAKIMNO VARCHAR(20),
    NOTU VARCHAR(1023),
    SERVISNOTU VARCHAR(1023),
    PARCABEKLIYOR INTEGER,
    PARCABEKLIYORSTR VARCHAR(4),
    ARIZAYIGIDERENPERSONELID INTEGER,
    ARIZAYIGIDERENPERSONELADI VARCHAR(40),
    SONCALISANPERSONELID INTEGER,
    SONCALISANPERSONELADI VARCHAR(40),
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30),
    BAKIMCISURE DOUBLE PRECISION,
    BAKIMCISURESTR VARCHAR(30),
    DEPARTMANID INTEGER,
    DEPARTMANADI ADI30,
    DURUMU INTEGER,
    GELDIMAKINADURUMUID INTEGER,
    GITTIMAKINADURUMUID INTEGER,
    ATANANPERSONELID INTEGER,
    ATANANPERSONELADI VARCHAR(255),
    BAKIMBASLAMATARIHI TIMESTAMP,
    BAKIMBITISTARIHI TIMESTAMP,
    BAKIMCIGECENSURE DOUBLE PRECISION,
    BAKIMCIGECENSURESTR VARCHAR(30))
AS
declare variable FNotu blob sub_type 0 segment size 80;
declare variable FServisNotu blob sub_type 0 segment size 80;
declare variable BlobSize integer;
declare variable MaxMakinaPersonelId integer;
declare variable FPostaBasi integer;
declare variable ADepartmanId integer;
declare variable APersonelId integer;
declare variable FFilterTarih timestamp;
declare variable BakimciIlkGelisTarihi timestamp;
declare variable FAtananPersonelAdi varchar(40);
declare variable FArizayiGiderenPersonelAdi varchar(40);
declare variable FBakimBaslamaTarihi timestamp;
declare variable FBakimBitisTarihi timestamp;
declare variable FGeldiMakinaDurumuId integer;
declare variable FGittiMakinaDurumuId integer;
begin

 /* Select First 1 Personel.PostaBasi, DepartmanId, Personel.PersonelId from Kullanici
         left Join Personel on Personel.PersonelId = Kullanici.PersonelId
         where Kullanici.KullaniciId = :AKullaniciId
         into :FPostaBasi, :ADepartmanId, :APersonelId;   */
  FFilterTarih  = cast(current_timestamp(0) as Date);
  if(:AFilterBugun = 1) then  FFilterTarih  = cast(current_timestamp(0) as Date);
  if(:AFilterDun = 1) then  FFilterTarih  = cast(current_timestamp(0) as Date) - 1;
  if(:AFilterHaftalik = 1) then  FFilterTarih  = cast(current_timestamp(0) as Date) - 7;


  For Select ArizaIhbar.ArizaIhbarId, ArizaIhbar.MakinaId, ArizaIhbar.BaslatanPersonelId, BasP.Adi, MD1.Tarih,
             ArizaIhbar.BitirenPersonelId, BitP.Adi, MD2.Tarih, ArizaIhbar.ArizaTipiId, ArizaTipi.Adi,
             ArizaIhbar.PlanliBakimNo, ArizaIhbar.Notu, ArizaIhbar.ServisNotu,
             ArizaIhbar.ParcaBekliyor, BasP.DepartmanId, Departman.Adi from ArizaIhbar
    left join Personel BasP on BasP.PersonelId = ArizaIhbar.BaslatanPersonelId
    left join Personel BitP on BitP.PersonelId = ArizaIhbar.BitirenPersonelId
    left join Departman on Departman.DepartmanId = BasP.DepartmanId
    left join Makina on Makina.MakinaId = ArizaIhbar.MakinaId
    left join MakinaDurumu MD1 on MD1.MakinaDurumuId = ArizaIhbar.BasladiMakinaDurumuId
    left join MakinaDurumu MD2 on MD2.MakinaDurumuId = ArizaIhbar.BittiMakinaDurumuId
    left join ArizaTipi on ArizaTipi.ArizaTipiId = ArizaIhbar.ArizaTipiId
    Where ( MD1.Tarih > :FFilterTarih) and
          ArizaIhbar.BittiMakinaDurumuId is not null
    into :ArizaIhbarId, :MakinaId, :BaslatanPersonelId, :BaslatanPersonelAdi, :BaslamaTarihi, :BitirenPersonelId, :BitirenPersonelAdi,
         :BitisTarihi, :ArizaTipiId, :ArizaTipiAdi,
         :PlanliBakimNo, :FNotu, :FServisNotu, :ParcaBekliyor,:DepartmanId, :DepartmanAdi do
  begin
    BakimciIlkGelisTarihi = null;
    ArizayiGiderenPersonelAdi = '';
    GeldiMakinaDurumuId = null;
    GittiMakinaDurumuId = null;
    BakimBaslamaTarihi = null;
    BakimBitisTarihi = null;
    For Select ArizaBakim.PersonelId , ABP.Adi ,ArizaBakim.GeldiMakinaDurumuId, ArizaBakim.GittiMakinaDurumuId  ,MD3.Tarih,MD4.Tarih  from ArizaBakim
      left Join Personel ABP ON ABP.PersonelId = ArizaBakim.PersonelId
      left join MakinaDurumu MD3 on MD3.MakinaDurumuId = ArizaBakim.GeldiMakinaDurumuId
      left join MakinaDurumu MD4 on MD4.MakinaDurumuId = ArizaBakim.GittiMakinaDurumuId
      where ArizaBakim.ArizaIhbarId = :ArizaIhbarId
      into :ArizayiGiderenPersonelId, :FArizayiGiderenPersonelAdi , :FGeldiMakinaDurumuId, :FGittiMakinaDurumuId,:FBakimBaslamaTarihi,:FBakimBitisTarihi do
    begin
      if(GeldiMakinaDurumuId is null ) then
      begin
        GeldiMakinaDurumuId = FGeldiMakinaDurumuId;
        BakimBaslamaTarihi = FBakimBaslamaTarihi;
      end
      else if (GeldiMakinaDurumuId > FGeldiMakinaDurumuId  ) then
      begin
        GeldiMakinaDurumuId = FGeldiMakinaDurumuId;
        BakimBaslamaTarihi = FBakimBaslamaTarihi;
      end
      if(GittiMakinaDurumuId is null ) then
      begin
        GittiMakinaDurumuId = FGittiMakinaDurumuId;
        BakimBitisTarihi = FBakimBitisTarihi;
      end
      else if (GittiMakinaDurumuId < FGittiMakinaDurumuId  ) then
      begin
        GittiMakinaDurumuId = FGittiMakinaDurumuId;
        BakimBitisTarihi = FBakimBitisTarihi;
      end

      if(ArizayiGiderenPersonelAdi = '') then ArizayiGiderenPersonelAdi = FArizayiGiderenPersonelAdi;
      else  ArizayiGiderenPersonelAdi = ArizayiGiderenPersonelAdi || ',' || FArizayiGiderenPersonelAdi;
    end
      Select Min(MakinaDurumu.Tarih) from ArizaBakim
        left join MakinaDurumu on MakinaDurumu.MakinaDurumuId = ArizaBakim.GeldiMakinaDurumuId
        where ArizaBakim.ArizaIhbarId = :ArizaIhbarId
        into :BakimciIlkGelisTarihi;

      if (BakimciIlkGelisTarihi is not null) then
      begin
        BakimciGecenSure = BakimciIlkGelisTarihi - BaslamaTarihi;
        BakimciGecenSureStr = TarihToStr(BakimciGecenSure);
      end
      else
      begin
        BakimciGecenSure = 0;
        BakimciGecenSureStr = '';
      end

      Durumu = 3;
      if (BitirenPersonelId is not null) then Bitti = 1;
      else Bitti = 0;

      if (Bitti = 1) then BittiStr = 'Bitti';
      else BittiStr = '';

      if (ParcaBekliyor = 1) then ParcaBekliyorStr = 'Evet';
      else ParcaBekliyorStr = '';


      BlobSize = f_BlobSize(FNotu);

      if (BlobSize > 1020) then Notu = f_BlobLeft(FNotu, 1020);
      else if (BlobSize > 0) then Notu = f_BlobLeft(FNotu, BlobSize);
      else Notu = '';

      BlobSize = f_BlobSize(FServisNotu);
      if (BlobSize > 1020) then ServisNotu = f_BlobLeft(FServisNotu, 1020);
      else if (BlobSize > 0) then ServisNotu = f_BlobLeft(FServisNotu, BlobSize);
      else ServisNotu = '';

      if (BitisTarihi is not null) then GecenSure = BitisTarihi - BaslamaTarihi;
      else GecenSure = (CURRENT_DATE + CURRENT_TIME) - BaslamaTarihi;

    if (GecenSure is not null) then
      begin
        GecenSureStr = TarihToStr(GecenSure);
      end
      else
      begin
        GecenSure = 0;
        BakimciGecenSureStr = '';
      end

      MaxMakinaPersonelId = Null;
      SonCalisanPersonelId  = Null;
      SonCalisanPersonelAdi = Null;

      BakimciSure           = 0;
      BakimciSureStr        = '';

      Select Sum(coalesce(MD2.Tarih, Current_Date+Current_Time) - MD1.Tarih) from ArizaBakim
        left join MakinaDurumu MD1 on MD1.MakinaDurumuId = ArizaBakim.GeldiMakinaDurumuId
        left join MakinaDurumu MD2 on MD2.MakinaDurumuId = ArizaBakim.GittiMakinaDurumuId
      where ArizaBakim.ArizaIhbarId = :ArizaIhbarId
      into :BakimciSure;

      if (BakimciSure is null) then BakimciSure = 0;
      Durumu = 3;
      AtananPersonelId = null;
      AtananPersonelAdi = '';
      For Select Personel.PersonelId , Personel.Adi from ArizaAtama
      left join Personel on Personel.PersonelId = ArizaAtama.AtananPersonelId
      where ArizaAtama.ArizaIhbarId = :ArizaIhbarId
      into :AtananPersonelId,:FAtananPersonelAdi  do
      begin
        if(AtananPersonelAdi = '') then AtananPersonelAdi = FAtananPersonelAdi;
        else 
        begin
          if (strlength(AtananPersonelAdi) > 255) then
             AtananPersonelAdi = f_left(AtananPersonelAdi, 255)  ;
          else
           AtananPersonelAdi = AtananPersonelAdi || ',' || FAtananPersonelAdi;
        end
      end
      suspend;
  end
end
