CREATE PROCEDURE HAMLISTESI(
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ATALIMATLARDANGETIR EGEBOOLEAN,
    ADOKUMAHAREKETI INTEGER)
RETURNS (
    HAMID INTEGER,
    ADI ADI255,
    ADI2 ADI255,
    ADI3 ADI255,
    ADI4 ADI255,
    REFERANSKODU HAMKODUSIZE,
    HAMTIPIID INTEGER,
    HAMTIPIADI ADI20,
    KUMASCINSIADI ADI40,
    KAPALI EGEBOOLEAN,
    PUS INTEGER,
    FEINE INTEGER,
    GRAMAJ INTEGER,
    TUPACIKENMAYLI INTEGER,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    INSERTKULLANICIADI ADI40,
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI ADI40,
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable FBoyaSiparisKayitId Integer;
begin
  if (ATalimatlardanGetir = 0) then
  begin
    if (ABoyaSiparisYil is null and ABoyaSiparisId is null) then
    begin
      For Select Ham.HamId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Ham.ReferansKodu, Ham.HamTipiId, HamTipi.Adi, KumasCinsi.Adi, Ham.Kapali,
                Ham.Pus, Ham.Feine, Ham.Gramaj, Ham.TupAcikEnMayli, Ham.IplikBoyu1, Ham.IplikBoyu2, Ham.IplikBoyu3, Ham.IplikBoyu4,
                IK.Adi, Ham.InsertTarihi, SK.Adi, Ham.DegisimTarihi from Ham
         left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
         left join KumasCinsi on KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
         left join Kullanici IK on IK.KullaniciId  = Ham.InsertKullaniciId
         left join Kullanici SK on SK.KullaniciId  = Ham.KullaniciId
         where Ham.Kapali = 0
      into :HAMID, :ADI, :ADI2, :ADI3, :ADI4, :REFERANSKODU, :HAMTIPIID, :HAMTIPIADI, :KUMASCINSIADI, :KAPALI, :PUS, :FEINE, :GRAMAJ,
           :TUPACIKENMAYLI, :IPLIKBOYU1, :IPLIKBOYU2, :IPLIKBOYU3, :IPLIKBOYU4, :INSERTKULLANICIADI, :INSERTTARIHI, :KULLANICIADI, :DEGISIMTARIHI do
           suspend;
     end
     else
     begin
      For Select distinct Ham.HamId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Ham.ReferansKodu, Ham.HamTipiId, HamTipi.Adi, KumasCinsi.Adi, Ham.Kapali,
                Ham.Pus, Ham.Feine, Ham.Gramaj, Ham.TupAcikEnMayli, Ham.IplikBoyu1, Ham.IplikBoyu2, Ham.IplikBoyu3, Ham.IplikBoyu4,
                IK.Adi, Ham.InsertTarihi, SK.Adi, Ham.DegisimTarihi from BoyaSiparis
         left join OrmeSiparisDetay on OrmeSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
         left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
         left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
         left join KumasCinsi on KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
         left join Kullanici IK on IK.KullaniciId  = Ham.InsertKullaniciId
         left join Kullanici SK on SK.KullaniciId  = Ham.KullaniciId
         where Ham.Kapali = 0
              and (BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil or :ABoyaSiparisYil is null)
              and (BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId or :ABoyaSiparisId is null)
              and (:ADokumaHareketi = 0 and BoyaSiparis.OrmeSiparisi=1 or :ADokumaHareketi = 1 and BoyaSiparis.DokumaSiparisi=1)
      into :HAMID, :ADI, :ADI2, :ADI3, :ADI4, :REFERANSKODU, :HAMTIPIID, :HAMTIPIADI, :KUMASCINSIADI, :KAPALI, :PUS, :FEINE, :GRAMAJ,
           :TUPACIKENMAYLI, :IPLIKBOYU1, :IPLIKBOYU2, :IPLIKBOYU3, :IPLIKBOYU4, :INSERTKULLANICIADI, :INSERTTARIHI, :KULLANICIADI, :DEGISIMTARIHI do
           suspend;
     end
  end
  else
  begin
    if (ABoyaSiparisYil is not null and ABoyaSiparisId is not null) then
    begin
      select BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId and
      (:ADokumaHareketi = 0 and BoyaSiparis.OrmeSiparisi=1 or :ADokumaHareketi = 1 and BoyaSiparis.DokumaSiparisi=1) into :FBoyaSiparisKayitId;
      if (FBoyaSiparisKayitId is not null) then
      begin
        if (ADokumaHareketi = 0) then
          For Select distinct Ham.HamId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Ham.ReferansKodu, Ham.HamTipiId, HamTipi.Adi, KumasCinsi.Adi, Ham.Kapali,
                    Ham.Pus, Ham.Feine, Ham.Gramaj, Ham.TupAcikEnMayli, Ham.IplikBoyu1, Ham.IplikBoyu2, Ham.IplikBoyu3, Ham.IplikBoyu4,
                    IK.Adi, Ham.InsertTarihi, SK.Adi, Ham.DegisimTarihi from ORMETALIMATRECETESIGETIR(:FBoyaSiparisKayitId)
             left join Ham on Ham.HamId = ORMETALIMATRECETESIGETIR.HamId
             left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
             left join KumasCinsi on KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
             left join Kullanici IK on IK.KullaniciId  = Ham.InsertKullaniciId
             left join Kullanici SK on SK.KullaniciId  = Ham.KullaniciId
             where Ham.Kapali = 0
          into :HAMID, :ADI, :ADI2, :ADI3, :ADI4, :REFERANSKODU, :HAMTIPIID, :HAMTIPIADI, :KUMASCINSIADI, :KAPALI, :PUS, :FEINE, :GRAMAJ,
               :TUPACIKENMAYLI, :IPLIKBOYU1, :IPLIKBOYU2, :IPLIKBOYU3, :IPLIKBOYU4, :INSERTKULLANICIADI, :INSERTTARIHI, :KULLANICIADI, :DEGISIMTARIHI do
               suspend;
         else
          For Select distinct Ham.HamId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, Ham.ReferansKodu, Ham.HamTipiId, HamTipi.Adi, KumasCinsi.Adi, Ham.Kapali,
                    Ham.Pus, Ham.Feine, Ham.Gramaj, Ham.TupAcikEnMayli, Ham.IplikBoyu1, Ham.IplikBoyu2, Ham.IplikBoyu3, Ham.IplikBoyu4,
                    IK.Adi, Ham.InsertTarihi, SK.Adi, Ham.DegisimTarihi from DOKUMATALIMATRECETESIGETIR(:FBoyaSiparisKayitId)
             left join Ham on Ham.HamId = DOKUMATALIMATRECETESIGETIR.HamId
             left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
             left join KumasCinsi on KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
             left join Kullanici IK on IK.KullaniciId  = Ham.InsertKullaniciId
             left join Kullanici SK on SK.KullaniciId  = Ham.KullaniciId
             where Ham.Kapali = 0
          into :HAMID, :ADI, :ADI2, :ADI3, :ADI4, :REFERANSKODU, :HAMTIPIID, :HAMTIPIADI, :KUMASCINSIADI, :KAPALI, :PUS, :FEINE, :GRAMAJ,
               :TUPACIKENMAYLI, :IPLIKBOYU1, :IPLIKBOYU2, :IPLIKBOYU3, :IPLIKBOYU4, :INSERTKULLANICIADI, :INSERTTARIHI, :KULLANICIADI, :DEGISIMTARIHI do
               suspend;
      end
      else
        suspend;
    end
    else
      suspend;
  end
end
