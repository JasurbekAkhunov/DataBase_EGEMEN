CREATE PROCEDURE PARTISTOKASAMALARIGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    PARTIASAMALARIID INTEGER,
    SIRA INTEGER,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    MAKINAPLANLAMAGRUPADI VARCHAR(30),
    PLANLANANMAKINAID VARCHAR(10),
    BOYAHANEBOLUMADI VARCHAR(30),
    GRUPPROSESNO VARCHAR(10),
    GRUPPROSESADI VARCHAR(80),
    PROSESNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ASAMAACIKLAMATIPIADI VARCHAR(3000),
    PARTITAMIRID INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    PARASALDEGERIVAR INTEGER,
    KUMASPAKETLEMENO VARCHAR(30),
    UCRETLITAMIR INTEGER,
    OTOMASYONADAHILMI SMALLINT,
    GIRISPERSONELID INTEGER,
    GIRISPERSONELADI VARCHAR(40),
    CIKISPERSONELID INTEGER,
    CIKISPERSONELADI VARCHAR(40),
    CIKISRAFID RAFKODUSIZE,
    RECETEISEMRIYIL INTEGER,
    RECETEISEMRIID INTEGER,
    DOZAJLAMAISEMRIID INTEGER)
AS
declare variable FAsamaAciklamaTipiAdi  varchar(50);
declare variable Virgul                 integer;
declare variable PaketlemeYil  varchar(4);
declare variable PaketlemeId   varchar(10);
declare variable KumasPaketlemeAsamaId integer;
begin
  select KumasPaketlemeAsamaId from Sistem where SistemId = 1 into :KumasPaketlemeAsamaId;

  For Select PartiAsamaStok.PartiSinifId, PartiAsamaStok.PartiAsamalariId, PartiAsamalari.Sira, PartiAsamalari.ReceteId, Recete.Adi, PartiAsamaStok.AsamaId, Asama.Adi,
             PartiAsamaStok.MakinaPlanlamaGrupId, MakinaPlanlamaGrup.Adi, PartiAsamaStok.GrupProsesNo, MakinaPlanlamaGrupProses.Adi, PartiAsamalari.ProsesNotu,
             BoyahaneBolum.Adi, PartiAsamalari.PartiTamirId, TamirTipi.Adi, PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi, Asama.ParasalDegeriVar,
             PartiAsamaStok.PaketlemeYil, PartiAsamaStok.PaketlemeId, TamirCozumSekli.Ucretli, PartiAsamalari.OtomasyonaDahilmi,
             PartiASamaStokDetay.GirisPersonelID, PartiAsamaStokDetay.CikisPersonelId, GP.Adi, CP.Adi, PartiASamaStokDetay.CikisRafId from PartiAsamalari
  left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
  left join PartiASamaStokDetay on PartiASamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiASamaStokDetay.PartiSinifID = PartiAsamaStok.PartiSinifID
  left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
  left join MakinaPlanlamaGrup on MakinaPlanlamaGrup.MakinaPlanlamaGrupId = PartiAsamaStok.MakinaPlanlamaGrupId
  left join MakinaPlanlamaGrupProses on MakinaPlanlamaGrupProses.MakinaPlanlamaGrupId = PartiAsamaStok.MakinaPlanlamaGrupId and MakinaPlanlamaGrupProses.GrupProsesNo = PartiAsamaStok.GrupProsesNo
  left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
  left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
  left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
  left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliId = PartiTamir.TamirCozumSekliId
  left join BoyahaneBolum on BoyahaneBolum.BoyahaneBolumId = Asama.BoyahaneBolumId
  left join Personel Gp on Gp.PersonelId = PartiAsamaStokDetay.GirisPersonelId
  left join Personel Cp on Cp.PersonelId = PartiAsamaStokDetay.CikisPersonelId
  where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamaStok.AsamayaDahil = 1
  order by PartiAsamaStok.PartiSinifId, PartiAsamalari.Sira
  into :PartiSinifId, :PartiAsamalariId, :Sira, :ReceteId, :ReceteAdi, :AsamaId, :AsamaAdi, :MakinaPlanlamaGrupId, :MakinaPlanlamaGrupAdi, :GrupProsesNo, :GrupProsesAdi,
       :ProsesNotu, :BoyahaneBolumAdi, :PartiTamirId, :TamirTipiAdi, :GirisTarihi, :CikisTarihi, :ParasalDegeriVar,
       :PaketlemeYil, :PaketlemeId, :UcretliTamir, :OtomasyonaDahilmi, :GirisPersonelID, :CikisPersonelID, :GirisPersonelAdi, :CikisPersonelAdi, :CikisRafId do
  begin
    AsamaAciklamaTipiAdi = '';
    Virgul = 0;

    For Select AsamaAciklamaTipi.Adi from PartiAsamaStokAciklama
     left join AsamaAciklamaTipi on AsamaAciklamaTipi.AsamaAciklamaTipiId = PartiAsamaStokAciklama.AsamaAciklamaTipiId
     where PartiAsamaStokAciklama.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStokAciklama.PartiSinifId = :PartiSinifId
     into :FAsamaAciklamaTipiAdi do
    begin
     if (Virgul = 1) then AsamaAciklamaTipiAdi =  AsamaAciklamaTipiAdi || cast(', ' as varchar(2) character set UTF8);
     AsamaAciklamaTipiAdi = AsamaAciklamaTipiAdi || FAsamaAciklamaTipiAdi;
     Virgul = 1;
    end

    PlanlananMakinaId = Null;

    Select Max(UretimPlanlama.MakinaId) from BolumPlanlama
      left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
      left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
      where BolumPlanlama.PartiAsamalariId = :PartiAsamalariId and BolumPlanlama.PartiSinifId = :PartiSinifId
      into :PlanlananMakinaId;

    KumasPaketlemeNo = '';

    if (AsamaId = KumasPaketlemeAsamaId) then
    begin
      if (PaketlemeYil is not null) then
      begin
        KumasPaketlemeNo = EgemenCopyStr(PaketlemeYil, 3, 2) || '-' || PaketlemeId;
      end
    end

    ReceteIsEmriYil = Null;
    ReceteIsEmriId  = Null;
    DozajlamaIsEmriId = Null;
    Select ReceteIsEmriDetay.ReceteIsEmriYil, ReceteIsEmriDetay.ReceteIsEmriId from ReceteIsEmriDetay
    where ReceteIsEmriDetay.PartiAsamalariId = :PartiAsamalariId
    into :ReceteIsEmriYil, :ReceteIsEmriId;

    if (ReceteIsEmriYil is null) then
      select DozajlamaIsEmriId from DozajlamaIsEmri where DozajlamaIsEmri.PartiAsamalariId = :PartiAsamalariId into :DozajlamaIsEmriId;
    else
      Select DozajlamaIsEmriId from DozajlamaIsEmri where DozajlamaIsEmri.ReceteIsEmriYil = :ReceteIsEmriYil and DozajlamaIsEmri.ReceteIsEmriId = :ReceteIsEmriId into :DozajlamaIsEmriId;


    suspend;
  end
end
