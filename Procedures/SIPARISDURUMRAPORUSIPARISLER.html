CREATE PROCEDURE SIPARISDURUMRAPORUSIPARISLER(
    AHAMID INTEGER,
    AHAMTIPIID INTEGER,
    AFIRMAID VARCHAR(20),
    AYURTDISI INTEGER,
    AKAPALI INTEGER)
RETURNS (
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISIVEREN VARCHAR(50),
    SIPARISMIKTAR DOUBLE PRECISION,
    SIPARISNETMIKTAR DOUBLE PRECISION,
    TOPLAMSATIS DOUBLE PRECISION,
    YURTDISI INTEGER,
    YURTDISISTR VARCHAR(9),
    LOT VARCHAR(30),
    KAPALI INTEGER,
    KAPALISTR VARCHAR(5),
    ORMESIPARISDETAYID INTEGER,
    TESLIMTARIHI TIMESTAMP,
    REVIZETESLIMTARIHI TIMESTAMP,
    IPLIKSATISIADE NUMERIC(12,3),
    IPLIKSATIS NUMERIC(12,3),
    IPLIKSATISIADENET NUMERIC(12,3),
    IPLIKSATISNET NUMERIC(12,3),
    ANALOTBOLUM VARCHAR(20))
AS
begin
  For Select BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, FirmaYetkili.Adi,
             BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             OrmeSiparisDetay.HamMiktar, IplikLot.Lot, BoyaSiparis.YurtDisi, OrmeSiparisDetay.HamID, Ham.Adi, OrmeSiparisDetay.ToplamSatis,
             IplikLot.Kapali, OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.RevizeTeslimTarihi from OrmeSiparisDetay
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
  left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
  Left Join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  left join IplikLot on IplikLot.IplikLotId = OrmeSiparisDetay.IplikLotId
  left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
  where BoyaSiparis.BoyaSiparisKayitId is not Null and BoyaSiparis.SiparisTipiID is Null and BoyaSiparis.OrmeSiparisTipiID = 0 and
        (Firma.FirmaId = :AFirmaId or :AFirmaId is null) and (BoyaSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null) and
        (OrmeSiparisDetay.Kapali = :AKapali Or :AKapali is Null) and
        (OrmeSiparisDetay.HamId = :AHamID or :AHamID is null) and
        (Ham.HamTipiID = :AHamTipiId or :AHamTipiId is Null)
  into :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi,  :SiparisiVeren, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
       :SiparisMiktar, :Lot, :YurtDisi, :HamID, :HamAdi, :ToplamSatis, :Kapali, :OrmeSiparisDetayId, :TeslimTarihi, :RevizeTeslimTarihi do
  begin
    if (YurtDisi = 1) then
      YurtDisiStr = 'Yurt Dışı';
    else
      YurtDisiStr = 'Yurt İçi';

    if (Kapali = 1) then KapaliStr = 'Evet';
    else KapaliStr = 'Hayır';

    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    select IplikSatisIade, IplikSatis, IplikSatisIadeNet, IplikSatisNet from OSRIcinIplikBilgileri(:OrmeSiparisDetayId)
            into :IplikSatisIade, :IplikSatis, :IplikSatisIadeNet, :IplikSatisNet;

    AnaLotBolum = Null ;
    if (Char_Length(Lot) > 0 and Position('.' in Lot) > 0)  then
      AnaLotBolum = substring(Lot from 1 for (Position('.' in Lot)-1) );

    Suspend;
  end
end
