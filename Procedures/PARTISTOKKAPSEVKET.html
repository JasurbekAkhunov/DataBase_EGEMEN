CREATE PROCEDURE PARTISTOKKAPSEVKET(
    ABOYAIRSALIYEID INTEGER,
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    ADEPOID VARCHAR(10),
    APARTISTOKTARTIID INTEGER,
    AKULLANICIID INTEGER,
    AKALITENO INTEGER)
AS
declare variable KayitSayisi            integer;
declare variable FBoyaIrsaliyeId        integer;
declare variable FBoyaIrsaliyeDetayId   integer;
declare variable FHamStokSarfFisiDetayId integer;
declare variable FKapAdedi              integer;
declare variable FAdet                  integer;
declare variable ParcaliSevk    integer;
declare variable FNetKilo               Numeric(12,3);
declare variable FNetMetre              Numeric(12,3);
declare variable FTahminiBrutKilo       numeric(12,3);
declare variable FTahminiBrutMetre       numeric(12,3);
declare variable BrutKilo               Numeric(12,3);
declare variable BrutMetre               Numeric(12,3);
declare variable TamamiTartildi         integer;
declare variable FKaliteNo              integer;
declare variable FMsg                   varchar(255);
declare variable HamTipiId   integer;
declare variable FPartiStokTartiId       integer;
declare variable KumasSevkteTartiyiAyriGoster integer;
declare variable FKaliteAciklamaTipiId   integer;
declare variable FMamulProsesId   integer;
declare variable FBoyaRezervFisiDetayId integer;
declare variable FLinkIslemi integer;
declare variable FOlcuId integer;
begin
  Select KumasSevkteTartiyiAyriGoster from sistem where sistemid = 1 into :KumasSevkteTartiyiAyriGoster;

  Select Parti.ParcaliSevk, PartiStok.HamTipiId, PartiStok.Kilo, PartiStok.Metre, PartiStok.TamamiTartildi from PartiStok
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayID
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  left join Ham On Ham.HamId = PartiStok.HamId
  where PartiStok.PartiSinifId = :APartiSinifId
  into :ParcaliSevk, :HamTipiId, :BrutKilo, :BrutMetre, :TamamiTartildi;

  if (APartiStokTartiId is null) then
  begin
    if (HamTipiId = 0 or KumasSevkteTartiyiAyriGoster = 0) then
    begin
      For Select KaliteNo, KaliteAciklamaTipiId, MamulProsesId, Sum(BrutKilo), Sum(BrutMetre), Sum(Kilo), Sum(Metre), Sum(KapAdedi), Sum(Adet), OlcuId from PartiStokTarti
         where PartiStokTarti.PartiSinifId = :APartiSinifId and PartiStokTarti.BoyaIrsaliyeId is null and
               PartistokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.DepoId = :ADepoId and
               (PartiStokTarti.KaliteNo = :AKaliteNo or  :AKaliteNo is null)  and
               PartiStokTarti.BoyaRezervFisiDetayId is null
        group By KaliteNo, KaliteAciklamaTipiId, MamulProsesId, OlcuId
        into :FKaliteNo, :FKaliteAciklamaTipiId, :FMamulProsesId, :FTahminiBrutKilo, :FTahminiBrutMetre, :FNetKilo, :FNetMetre, :FKapAdedi, :FAdet, :FOlcuId do
      begin
        Select Min(BoyaIrsaliyeDetayId) from BoyaIrsaliyeDetay
          where BoyaIrsaliyeDetay.BoyaIrsaliyeId = :ABoyaIrsaliyeId and BoyaIrsaliyeDetay.PartiSinifId = :APartiSinifId and
                BoyaIrsaliyeDetay.KaliteNo = :FKaliteNo and IntKarsilastir(BoyaIrsaliyeDetay.KaliteAciklamaTipiId, :FKaliteAciklamaTipiId) = 1 and
                IntKarsilastir(BoyaIrsaliyeDetay.OlcuId, :FOlcuId) = 1
          into :FBoyaIrsaliyeDetayId;

        if (FBoyaIrsaliyeDetayId is not Null) then
        begin
          Update BoyaIrsaliyeDetay Set KapAdedi = KapAdedi + :FKapAdedi,
                                       BrutKilo = BrutKilo + :FTahminiBrutKilo,
                                       Kilo = Kilo + :FNetKilo,
                                       BrutMetre = BrutMetre + :FTahminiBrutMetre,
                                       Metre = Metre + :FNetMetre,
                                       Adet = Adet + :FAdet,
                                       BarkodLinkIslemi = 1
              where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId;
        end
        else
        begin
          FBoyaIrsaliyeDetayId = gen_id(BoyaIrsaliyeDetay_Id_Gen, 1);

          insert into BoyaIrsaliyeDetay(BoyaIrsaliyeDetayId, BoyaIrsaliyeId, FasonHamIade, PartiSinifId, PartiKayitId, KapAdedi, BrutKilo, Kilo, BrutMetre, Metre, KaliteNo, KaliteAciklamaTipiId, MamulProsesId, InsertKullaniciId, KullaniciId, OlcuId, Adet)
                        values(:FBoyaIrsaliyeDetayId, :ABoyaIrsaliyeId, 0, :APartiSinifId, :APartiKayitId,  :FkapAdedi, :FTahminiBrutKilo, :FNetKilo, :FTahminiBrutMetre, :FNetMetre, :FKaliteNo, :FKaliteAciklamaTipiId, :FMamulProsesId, :AKullaniciId, :AKullaniciId, :FOlcuId, :FAdet);
        end

        Update PartiStokTarti Set BoyaIrsaliyeID = :ABoyaIrsaliyeId, BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId
           where PartiStokTarti.PartiSinifId = :APartiSinifId and PartiStokTarti.BoyaIrsaliyeId is null and
                 PartistokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.DepoId = :ADepoId and PartiStokTarti.KaliteNo = :FKaliteNo
                 and IntKarsilastir(PartiStokTarti.OlcuId, :FOlcuId) = 1;
      end
    end
    else
    begin /* her tartıyı ayrı gösteriyoruz KumasSevkteTartiyiAyriGoster = 1 */
      For Select KaliteNo, PartiStoktartiId, KaliteAciklamaTipiId, MamulProsesId, BrutKilo, BrutMetre, Kilo, Metre, KapAdedi, OlcuId, Adet from PartiStokTarti
         where PartiStokTarti.PartiSinifId = :APartiSinifId and PartiStokTarti.BoyaIrsaliyeId is null and
               PartistokTarti.HamStokSarfFisiDetayId is null and PartiStokTarti.DepoId = :ADepoId and
               PartiStokTarti.BoyaRezervFisiDetayId is null and
               (PartiStokTarti.KaliteNo = :AKaliteNo or  :AKaliteNo is null)
        into :FKaliteNo, :FPartiStoktartiId, :FKaliteAciklamaTipiId, :FMamulProsesId, :FTahminiBrutKilo, :FTahminiBrutMetre, :FNetKilo, :FNetMetre, :FKapAdedi, :FOlcuId, :FAdet do
      begin
        FBoyaIrsaliyeDetayId = gen_id(BoyaIrsaliyeDetay_Id_Gen, 1);

        insert into BoyaIrsaliyeDetay(BoyaIrsaliyeDetayId, BoyaIrsaliyeId, FasonHamIade, PartiSinifId, PartiKayitId, KapAdedi, BrutKilo, Kilo, BrutMetre, Metre, KaliteNo, KaliteAciklamaTipiId, MamulProsesId, InsertKullaniciId, KullaniciId, OlcuId, Adet)
                      values(:FBoyaIrsaliyeDetayId, :ABoyaIrsaliyeId, 0, :APartiSinifId, :APartiKayitId,  :FkapAdedi, :FTahminiBrutKilo, :FNetKilo, :FTahminiBrutMetre, :FNetMetre, :FKaliteNo, :FKaliteAciklamaTipiId, :FMamulProsesId, :AKullaniciId, :AKullaniciId, :FOlcuId, :FAdet);

        Update PartiStokTarti Set BoyaIrsaliyeID = :ABoyaIrsaliyeId, BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId
           where PartiStokTarti.PartiStoktartiId = :FPartiStoktartiId;
      end
    end
  end
  else
  begin
    Select KaliteNo, BrutKilo, BrutMetre, Kilo, Metre, KaliteAciklamaTipiId, BoyaIrsaliyeDetayId, HamStokSarfFisiDetayId, KapAdedi, BoyaRezervFisiDetayID, Adet, OlcuId
      from PartiStokTarti where PartiStokTarti.PartiStokTartiId = :APartiStokTartiId
      into :FKaliteNo, :FTahminiBrutKilo, :FTahminiBrutMetre, :FNetKilo, :FNetMetre, :FKaliteAciklamaTipiId, :FBoyaIrsaliyeDetayId, :FHamStokSarfFisiDetayId, :FKapAdedi, :FBoyaRezervFisiDetayId, :FAdet, :FOlcuId;
    FLinkIslemi = 0;
    if (FBoyaRezervFisiDetayId is not null) then
      FLinkIslemi = 1;

    if (FBoyaIrsaliyeDetayId is not Null) then exception BuPaketinIrsaliyesiKesilmis;
    if (FHamStokSarfFisiDetayId is not Null) then exception BuPaketSarFisiIleCikmis;

    FBoyaIrsaliyeDetayId = Null;

    if (HamTipiId = 0 or KumasSevkteTartiyiAyriGoster = 0) then
    begin
      Select BoyaIrsaliyeDetayId from BoyaIrsaliyeDetay
        where BoyaIrsaliyeDetay.BoyaIrsaliyeId = :ABoyaIrsaliyeId and BoyaIrsaliyeDetay.PartiSinifId = :APartiSinifId and
              BoyaIrsaliyeDetay.KaliteNo = :FKaliteNo and IntKarsilastir(BoyaIrsaliyeDetay.KaliteAciklamaTipiId, :FKaliteAciklamaTipiId) = 1 and
              IntKarsilastir(BoyaIrsaliyeDetay.OlcuId, :FOlcuId) = 1 and IntKarsilastir(BoyaIrsaliyeDetay.BoyaRezervFisiDetayID, :FBoyaRezervFisiDetayID) = 1
        into :FBoyaIrsaliyeDetayId;

        -- burasi
    end

    if (FBoyaIrsaliyeDetayId is not Null) then
    begin
        Update BoyaIrsaliyeDetay Set KapAdedi = KapAdedi + :FKapAdedi,
                                     BrutKilo = BrutKilo + :FTahminiBrutKilo,
                                     Kilo = Kilo + :FNetKilo,
                                     BrutMetre = BrutMetre + :FTahminiBrutMetre,
                                     Metre = Metre + :FNetMetre,
                                     Adet = Adet + :FAdet,
                                     BarkodLinkIslemi = 1,
                                     LinkIslemi = :FLinkIslemi,
                                     BoyaRezervFisiDetayId = Coalesce(BoyaRezervFisiDetayId, :FBoyaRezervFisiDetayId)
          where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId;

      if (FBoyaRezervFisiDetayId is not null) then
        update PartiStok set PartiStok.NetSevkKilo = PartiStok.NetSevkKilo + :FNetKilo ,
                             PartiStok.NetSevkMetre = PartiStok.NetSevkMetre + :FNetMetre,
                             PartiStok.BrutSevkMetre = PartiStok.BrutSevkMetre + :FTahminiBrutMetre,
                             PartiStok.NetSevkKapAdedi = PartiStok.NetSevkKapAdedi + :FkapAdedi,
                             PartiStok.BrutSevkKilo = PartiStok.BrutSevkKilo + :FTahminiBrutKilo,
                             PartiStok.NetSevkAdet = PartiStok.NetSevkAdet + :FAdet,
                             PartiStok.RezervKilo = PartiStok.RezervKilo - :FNetKilo,
                             PartiStok.BrutRezervKilo = PartiStok.BrutRezervKilo - :FTahminiBrutKilo,
                             PartiStok.RezervMiktar = PartiStok.RezervMiktar - :FNetKilo,
                             PartiStok.BrutRezervMiktar = PartiStok.BrutRezervMiktar - :FTahminiBrutKilo
               where PartiStok.PartiSinifId = :APartiSinifId;
    end
    else
    begin
      FBoyaIrsaliyeDetayId = gen_id(BoyaIrsaliyeDetay_Id_Gen, 1);

      insert into BoyaIrsaliyeDetay(BoyaIrsaliyeDetayId, BoyaIrsaliyeId, FasonHamIade, PartiSinifId, PartiKayitId, KapAdedi, BrutKilo, Kilo, BrutMetre, Metre, KaliteNo, KaliteAciklamaTipiId, InsertKullaniciId, KullaniciId, LinkIslemi,
                                    BoyaRezervFisiDetayId, OlcuId)
                    values(:FBoyaIrsaliyeDetayId, :ABoyaIrsaliyeId, 0, :APartiSinifId, :APartiKayitId,  :FkapAdedi, :FTahminiBrutKilo, :FNetKilo, :FTahminiBrutMetre, :FNetMetre, :FKaliteNo, :FKaliteAciklamaTipiId, :AKullaniciId, :AKullaniciId, :FLinkIslemi,
                           :FBoyaRezervFisiDetayId, :FOlcuId);
      update BoyaIrsaliyeDetay set LinkIslemi = 0 where BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId;
      if (FBoyaRezervFisiDetayId is not null) then
        update PartiStok set PartiStok.NetSevkKilo = PartiStok.NetSevkKilo + :FNetKilo ,
                             PartiStok.NetSevkMetre = PartiStok.NetSevkMetre + :FNetMetre,
                             PartiStok.BrutSevkMetre = PartiStok.BrutSevkMetre + :FTahminiBrutMetre,
                             PartiStok.NetSevkKapAdedi = PartiStok.NetSevkKapAdedi + :FkapAdedi,
                             PartiStok.BrutSevkKilo = PartiStok.BrutSevkKilo + :FTahminiBrutKilo,
                             PartiStok.NetSevkAdet = PartiStok.NetSevkAdet + :FAdet,
                             PartiStok.RezervKilo = PartiStok.RezervKilo - :FNetKilo,
                             PartiStok.BrutRezervKilo = PartiStok.BrutRezervKilo - :FTahminiBrutKilo,
                             PartiStok.RezervMiktar = PartiStok.RezervMiktar - :FNetKilo,
                             PartiStok.BrutRezervMiktar = PartiStok.BrutRezervMiktar - :FTahminiBrutKilo
               where PartiStok.PartiSinifId = :APartiSinifId;


    end

    Update PartiStokTarti Set BoyaIrsaliyeID = :ABoyaIrsaliyeId, BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId
       where PartiStokTarti.PartiStokTartiId = :APartiStokTartiId;
  end
end
