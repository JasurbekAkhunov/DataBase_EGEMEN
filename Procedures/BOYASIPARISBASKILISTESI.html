CREATE PROCEDURE BOYASIPARISBASKILISTESI(
    ABOYASIPARISKAYITID INTEGER,
    AHAMID INTEGER)
RETURNS (
    BOYASIPARISBASKIID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    DESENVARYANTID INTEGER,
    VARYANTADI VARCHAR(40),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3))
AS
begin
  For Select BoyaSiparisBaski.BoyaSiparisBaskiId, BoyaSiparisBaski.BoyaSiparisDetayId,
             BoyaSiparisDetay.DesenId, Desen.Adi, BoyaSiparisBaski.DesenVaryantId, DesenVaryant.VaryantAdi,
             BoyaSiparisBaski.Kilo, BoyaSiparisBaski.Metre FROM BoyaSiparisBaski
  LEFT JOIN BoyaSiparisDetay ON BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisBaski.BoyaSiparisDetayId
  LEFT JOIN Desen ON Desen.DesenId = BoyaSiparisDetay.DesenId
  LEFT JOIN DesenVaryant ON DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
  WHERE BoyaSiparisBaski.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisDetay.HamID = :AHamId
  order by BoyaSiparisBaski.BoyaSiparisKayitId, BoyaSiparisBaski.BoyaSiparisDetayId, BoyaSiparisBaski.BoyaSiparisBaskiId
  into :BoyaSiparisBaskiId, :BoyaSiparisDetayId, :DesenId, :DesenAdi, :DesenVaryantId, :VaryantAdi, :Kilo, :Metre do suspend;
end
