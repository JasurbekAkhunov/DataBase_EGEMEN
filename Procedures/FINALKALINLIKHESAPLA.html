CREATE PROCEDURE FINALKALINLIKHESAPLA(
    ABUKUMTALIMATID INTEGER,
    CEKIMHIZI DOUBLE PRECISION,
    CEKIM DOUBLE PRECISION,
    TALIMATKALINLIKBIRIM VARCHAR(10))
RETURNS (
    SONUC DOUBLE PRECISION)
AS
declare variable KalinlikToplami         double precision;
declare variable LycraKalinlik           double precision;
declare variable KalinlikNE              double precision;
declare variable KalinlikDegeri          double precision;
declare variable KalinlikCarpani         double precision;
declare variable hstIplik                integer;
declare variable hstLycra                integer;
declare variable TalimatKalinlikCarpani  double precision;
begin
  KalinlikToplami = 0;
  LycraKalinlik = 0;
  KalinlikCarpani = 0;
  KalinlikDegeri = 0;
  KalinlikNE = 0;

  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  select Carpan from IplikKalinlikBirim where IplikKalinlikBirim = :TalimatKalinlikBirim into :TalimatKalinlikCarpani;

  for select Kalinlik.KalinlikNE, IplikKalinlikBirim.Carpan from BukumTalimatRecete
  left join Ham on Ham.HamId = BukumTalimatRecete.HamId
  left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
  left join IplikKalinlikBirim on IplikKalinlikBirim.IplikKalinlikBirim = Kalinlik.IplikKalinlikBirim
  where BukumTalimatRecete.BukumTalimatId = :ABukumTalimatId and Ham.HamTipiId = :hstIplik
  into :KalinlikNE, :KalinlikCarpani do
  begin
    if (KalinlikCarpani is not Null) then
    if (KalinlikNE is not Null) then
    begin
      KalinlikDegeri = (KalinlikNE * TalimatKalinlikCarpani) / KalinlikCarpani;
    end
    else
      KalinlikDegeri = 0;

    KalinlikToplami = KalinlikToplami + KalinlikDegeri;
  end

  for select Kalinlik.KalinlikNE, IplikKalinlikBirim.Carpan from BukumTalimatRecete
  left join Ham on Ham.HamId = BukumTalimatRecete.HamId
  left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
  left join IplikKalinlikBirim on IplikKalinlikBirim.IplikKalinlikBirim = Kalinlik.IplikKalinlikBirim
  where BukumTalimatRecete.BukumTalimatId = :ABukumTalimatId and Ham.HamTipiId = :hstLycra
  into :KalinlikNE, :KalinlikCarpani do
  begin
    if (KalinlikCarpani is not Null) then
    if (KalinlikNE is not Null) then
    begin
      KalinlikDegeri = (KalinlikNE * TalimatKalinlikCarpani) / KalinlikCarpani;
    end
    else
      KalinlikDegeri = 0;

    if (Cekim > 0) then
      LycraKalinlik = KalinlikDegeri / Cekim;

    KalinlikToplami = KalinlikToplami + (LycraKalinlik);
  end

  if (CekimHizi > 0) then
    Sonuc = CekimHizi * KalinlikToplami;

  Suspend;
end
