CREATE PROCEDURE NAKITAKISEKIGETIR
RETURNS (
    NAKITAKISEKIID INTEGER,
    ACIKLAMA VARCHAR(50),
    BORCALACAK VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select NakitAkisEkiId,  Aciklama, BorcAlacak, Meblag, DovizId, DegisimTarihi, Kullanici.Adi from NakitAkisEki
  left join Kullanici on Kullanici.KullaniciId = NakitAkisEki.KullaniciId
  into :NakitAkisEkiId, :Aciklama, :BorcAlacak, :Meblag, :DovizId, :DegisimTarihi, :KullaniciAdi do
  begin
    Borc = Null;
    Alacak = Null;
    if (BorcAlacak = 'B') then Borc = Meblag;
    else  Alacak = Meblag;
    suspend;
  end
end
