CREATE PROCEDURE BOYAFIRERAPORU2(
    ABASLANGICSEVKTARIHI TIMESTAMP,
    ABITISSEVKTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ASIPARISFIRMAID VARCHAR(20),
    APARTIKAYITID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    AFASON INTEGER,
    APARTITIPIID INTEGER,
    AHAMID INTEGER,
    ABASLANGICPARTIMIKTAR NUMERIC(12,3),
    ABITISPARTIMIKTAR NUMERIC(12,3),
    AORANTIPI INTEGER,
    AFIRESEBEPTIPIID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AASAMAID INTEGER,
    AFIRMAADI2YIGOREBILIR INTEGER,
    AIMALATTIPIID INTEGER,
    ARECETEID RECETEKODUSIZE)
RETURNS (
    PARTISINIFID INTEGER,
    KALITENO INTEGER,
    KKBRUTKILO NUMERIC(12,3),
    KKNETKILO NUMERIC(12,3),
    KKNETMETRE NUMERIC(12,3),
    KKBRUTMETRE NUMERIC(12,3))
AS
begin
  for select distinct PartiSinifId from BoyaFireRaporu(:ABaslangicSevkTarihi, :ABitisSevkTarihi, :AFirmaId, :ASiparisFirmaId, :APartiKayitId, :ABoyaSiparisKayitId, :AFason,
    :APartiTipiId, :AHamId, :ABaslangicPartiMiktar, :ABitisPartiMiktar, :AOranTipi, :AFireSebepTipiId, :AMusteriSiparisNo, :AReferansSiparisNo, :AAsamaId, :AFirmaAdi2yiGorebilir,
    :AImalatTipiId, :AReceteId) into :PartiSinifId do
  begin
    for Select PSKaliteKontrol.KaliteNo, Sum(PSKaliteKontrol.Kilo), Sum(PSKaliteKontrol.Kilo - PSKaliteKontrol.Dara), Sum(PSKaliteKontrol.Metre), Sum(PSKaliteKontrol.BrutMetre) from PSKaliteKontrolAsama
      Left Join PSKaliteKontrol on PSKaliteKontrol.PSKaliteKontrolAsamaID = PSKaliteKontrolAsama.PSKaliteKontrolAsamaID
      where PSKaliteKontrolAsama.PartiSinifId = :PartiSinifId and PSKaliteKontrolAsama.SonKayit = 1 group by PSKaliteKontrol.KaliteNo
    into :KaliteNo, :KKBrutKilo, :KKNetKilo, :KKNetMetre, :KKBrutMetre do
      suspend;
  end
end
