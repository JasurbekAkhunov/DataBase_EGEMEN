CREATE PROCEDURE IPLIKMAKINANINSONDURUMUNUGETIR(
    AIPLIKMAKINAID VARCHAR(10))
RETURNS (
    IPLIKMAKINADURUMUID INTEGER,
    TARIH TIMESTAMP,
    ARIZA SMALLINT,
    ARAVERMETARIHI TIMESTAMP,
    IPLIKARAVERMESEBEBIID INTEGER,
    IPLIKARAVERMESEBEBIADI VARCHAR(50),
    IMALATAARAVERENPERSONELID INTEGER,
    IMALATAARAVERENPERSONELADI VARCHAR(40),
    PARCABEKLIYOR INTEGER,
    ARIZAIHBARPERSONELID INTEGER,
    ARIZAIHBARPERSONELADI VARCHAR(40),
    IPLIKARIZAIHBARID INTEGER,
    ARIZAYERIID INTEGER,
    IPLIKURETIMISEMRIKAYITID INTEGER,
    IPLIKURETIMISEMRIYIL INTEGER,
    IPLIKURETIMISEMRIID INTEGER,
    LOT VARCHAR(30),
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    BUKUMTIPIADI VARCHAR(10),
    HARMANADI VARCHAR(50),
    HARMANCINSIADI VARCHAR(50),
    KALINLIKNE INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    IPLIKCINSIURETIMPARTINOLABEL VARCHAR(10),
    COPSTIPIADI VARCHAR(20),
    BUKUMSAYISI INTEGER,
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    BASLAMATARIHI TIMESTAMP,
    TAHMINIBITISTARIHI TIMESTAMP,
    TOPLAMIGSAYISI INTEGER,
    EKSIKIGSAYISI INTEGER)
AS
declare variable IplikImalataAraVermeId integer;
declare variable ImalataAraVerildi SmallInt;
begin
  Select IplikMakinaDurumuId, ToplamIgSayisi, EksikIgSayisi from IplikMakina where IplikMakina.IplikMakinaID = :AIplikMakinaId into :IplikMakinaDurumuId, :ToplamIgSayisi, :EksikIgSayisi;

  Ariza                 = 0;
  AraVermeTarihi        = null;
  IplikAraVermeSebebiId      = Null;
  IplikAraVermeSebebiAdi     = NUll;
  ImalataAraVerenPersonelId = NUll;
  ImalataAraVerenPersonelAdi = NUll;
  ParcaBekliyor         = 0;
  ArizaIhbarPersonelId  = Null;
  ArizaIhbarPersonelAdi = null;
  IplikArizaIhbarId     = Null;
  ArizaYeriID           = Null;

  if (IplikMakinaDurumuId is not null) then
  begin
    Select Tarih, Ariza, ImalataAraVerildi from IplikMakinaDurumu
      where IplikMakinaDurumu.IplikMakinaDurumuID = :IplikMakinaDurumuId
      into :Tarih, :Ariza, :ImalataAraVerildi;

    if (Ariza = 1 or Ariza = 3) then Ariza = 1;
    else Ariza = 0;

    if (ImalataAraVerildi = 1 or ImalataAraVerildi = 3) then ImalataAraVerildi = 1;
    else ImalataAraVerildi = 0;

    ParcaBekliyor = 0;

    if (Ariza = 1) then
    begin
      Select Max(IplikArizaIhbarId) from IplikArizaIhbar
        where IplikArizaIhbar.IplikMakinaId = :AIplikMakinaId and IplikArizaIhbar.BittiIplikMakinaDurumuId is null into :IplikArizaIhbarId;

      Select IplikArizaIhbar.ArizaYeriID, IplikArizaIhbar.BaslatanPersonelId, Personel.Adi, IplikArizaIhbar.ParcaBekliyor from IplikArizaIhbar
        left join Personel on Personel.PersonelId = IplikArizaIhbar.BaslatanPersonelId
        where IplikArizaIhbar.IplikArizaIhbarId = :IplikArizaIhbarId
        into :ArizaYeriID, :ArizaIhbarPersonelId, :ArizaIhbarPersonelAdi, :ParcaBekliyor;

      if (ParcaBekliyor is null) then ParcaBekliyor = 0;
    end

    if (ImalataAraVerildi = 1) then
    begin
      Select Max(IplikImalataAraVermeId) from IplikImalataAraVerme
        where IplikImalataAraVerme.IplikMakinaId = :AIplikMakinaId and IplikImalataAraVerme.BittiIplikMakinaDurumuId is null
        into :IplikImalataAraVermeId;


      Select IplikImalataAraVerme.IplikAraVermeSebebiId, IplikAraVermeSebebi.Adi, IplikMakinaDurumu.Tarih, IplikImalataAraVerme.AraVerdirenPersonelId,
             Personel.Adi from IplikImalataAraVerme
        left join IplikAraVermeSebebi on IplikAraVermeSebebi.IplikAraVermeSebebiId = IplikImalataAraVerme.IplikAraVermeSebebiId
        left join Personel on Personel.PersonelId = IplikImalataAraVerme.AraVerdirenPersonelId
        left join IplikMakinaDurumu on IplikMakinaDurumu.IplikMakinaDurumuId = IplikImalataAraVerme.BasladiIplikMakinaDurumuId
        where IplikImalataAraVerme.IplikImalataAraVermeId = :IplikImalataAraVermeId
        into :IplikAraVermeSebebiId, :IplikAraVermeSebebiAdi, :AraVermeTarihi, :ImalataAraVerenPersonelId, :ImalataAraVerenPersonelAdi;
    end

  end

  Select First 1 IplikUretimIsEmri.IplikUretimIsEmriKayitId, IplikUretimIsEmri.IplikUretimIsEmriYil, IplikUretimIsEmri.IplikUretimIsEmriId,
        IplikLot.Lot, IplikLot.KalinlikNe, IplikCinsi.Adi, IplikCinsi.UretimPartiNoLabel, IplikUretimIsEmri.PlanlananMiktar, IplikUretimIsEmri.GerceklesenMiktar,
        Harman.HarmanYil, Harman.HarmanId, Harman.Adi, HarmanCinsi.Adi, IplikUretimIsEmri.BaslamaTarihi, BukumTipi.Adi, IplikLot.BukumSayisi,
        CopsTipi.Adi from IplikUretimIsEmri
    left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotID
    left join BukumTipi on BukumTipi.BukumTipiID = IplikLot.BukumTipiID
    left join IplikCinsi on IplikCinsi.IplikCinsiID = IplikLot.IplikCinsiID
    left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitID
    left join HarmanCinsi on HarmanCinsi.HarmanCinsiID = Harman.HarmanCinsiID
    left join CopsTipi on CopsTipi.CopsTipiID = IplikUretimIsEmri.CopsTipiID
    where IplikUretimIsEmri.BitisTarihi is null and IplikUretimIsEmri.IplikMakinaId = :AIplikMakinaId and IplikUretimIsEmri.BaslamaTarihi is not null
    order by IplikUretimIsEmri.Sira
    into :IplikUretimIsEmriKayitId, :IplikUretimIsEmriYil, :IplikUretimIsEmriId, :Lot, :KalinlikNe, :IplikCinsiAdi, :IplikCinsiUretimPartiNoLabel,
         :PlanlananMiktar, :GerceklesenMiktar, :HarmanYil, :HarmanId, :HarmanAdi, :HarmanCinsiAdi, :BaslamaTarihi, :BukumTipiAdi, :BukumSayisi,
         :CopsTipiAdi;

  if (IplikUretimIsEmriKayitId is not null) then
  begin
    if (BaslamaTarihi is not null) then TahminiBitisTarihi = BaslamaTarihi + 1;
  end

  suspend;
end
