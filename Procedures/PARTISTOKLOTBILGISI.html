CREATE PROCEDURE PARTISTOKLOTBILGISI(
    APARTISINIFID INTEGER)
RETURNS (
    LOT VARCHAR(150),
    GIRISRENKADI VARCHAR(50),
    GIRISHAMKALINLIK VARCHAR(100),
    MUSTERIPARTINO VARCHAR(100),
    ORMECIADI VARCHAR(100),
    ORMECIMAKINANO VARCHAR(100),
    BOBINETIKETILOT VARCHAR(150),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    LABRECETEKODU VARCHAR(20),
    RENKADI VARCHAR(40),
    HAMID INTEGER,
    KAZANAILKGIRISTARIHI TIMESTAMP,
    SONGIRDIGIKAZANMAKINAID VARCHAR(10),
    SONGIRDIGIKAZANCIKISTARIHI TIMESTAMP,
    ESKIPARTINO VARCHAR(100),
    IRSALIYESERINO VARCHAR(100),
    IRSALIYENO VARCHAR(100),
    IRSALIYETARIHI VARCHAR(100),
    ORMECIID INTEGER)
AS
declare variable FLot varchar(150);
declare variable FMusteriPartiNo varchar(30);
declare variable FRenk varchar(20);
declare variable FOrmeciAdi   varchar(30);
declare variable FOrmeciMakinaNo varchar(10);
declare variable FEskiPartiNo  varchar(100);
declare variable FIrsaliyeSeriNo  SeriNoSize5;
declare variable FIrsaliyeNo      IrsaliyeNoSize;
declare variable FIrsaliyeTarihi  TimeSTamp;
begin
  Select HamId from PartiStok where PartiStok.PartiSinifID = :APartiSinifID into :HamID;

  Select Left(List(Distinct HamStokMaster.Lot, cast(', ' as blob character set UTF8)), 150) AS Lot,
        Left(List(Distinct HamStokMaster.Lot, cast('-' as blob character set UTF8)), 150) AS BobinEtiketiLot,
        Left(List(Distinct HamStokMaster.MusteriPartiNo, cast(', ' as blob character set UTF8)), 100) AS MusteriPartiNo,
        Left(List(Distinct HamStokIrsaliye.IrsaliyeSeriNo, cast(', ' as blob character set UTF8)), 100) AS IrsaliyeSeriNo,
        Left(List(Distinct HamStokIrsaliye.IrsaliyeNo, cast(', ' as blob character set UTF8)), 100) AS IrsaliyeNo,
        Left(List(Distinct Cast(HamStokIrsaliye.IrsaliyeTarihi as Date), cast(', ' as VarChar(2) character set UTF8)), 100) AS IrsaliyeTarihi,
        Left(List(Distinct HamStokMaster.OrmeciMakinaNo, cast(', ' as blob character set UTF8)), 100) AS OrmeciMakinaNo,
        Left(List(Distinct iif(HamStokMaster.EskiPartiYil is not null, HamStokMaster.EskiPartiYil || cast('-' as blob character set UTF8) ||HamStokMaster.EskiPartiNo, HamStokMaster.EskiPartiNo), cast(', ' as blob character set UTF8)), 100) AS EskiPartiNo,
        Left(List(Distinct iif(HamStokMaster.Renk = '', Null, HamStokMaster.Renk) , cast(', ' as blob character set UTF8)), 50) AS GirisRenkAdi,
        Left(List(Distinct Ormeci.Adi, cast(', ' as blob character set UTF8)), 100) AS OrmeciAdi,
        Left(List(Distinct Kalinlik.KalinlikNe, ', '), 100) from PartiStokDetay
  left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
  left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = RafDetay.HamStokIrsaliyeId
  left join Ormeci on Ormeci.OrmeciID = HamStokIrsaliye.OrmeciId
  left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = HamStokMaster.OrmePlanlamaID
  left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
  left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
  where PartiStokDetay.PartiSinifId = :APartiSinifId
  into :Lot, BobinEtiketiLot, :MusteriPartiNo, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :OrmeciMakinaNo, :EskiPartiNo, :GirisRenkAdi, :OrmeciAdi, :GirisHamKalinlik;

  Select coalesce(BoyaSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.MusteriSiparisNo),
         coalesce(BoyaSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.ReferansSiparisNo),
         OrmeSiparisDetay.ArtikelNo, BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.RenkAdi from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = BoyaSiparisDetay.OrmeSiparisDetayId
    where PartiSinifID = :APartiSinifID
    into :MusteriSiparisNo, :ReferansSiparisNo, :ArtikelNo, :LabReceteKodu, :RenkAdi;

  SonGirdigiKazanMakinaId    = Null;
  SonGirdigiKazanCikisTarihi = Null;
  Select Min(PartiAsamaStokDetay.GirisTarihi) from PartiAsamaStokDetay
    left join Makina On Makina.MakinaID = PartiAsamaStokDetay.MakinaId
    where PartiAsamaStokDetay.PartiSinifID = :APartiSinifID and PartiAsamaStokDetay.GirisTarihi is not null and Makina.Kazan = 1
    into :KazanaIlkGirisTarihi;

  Select first 1 PartiAsamaStokDetay.MakinaID, PartiAsamaStokDetay.CikisTarihi from PartiAsamaStokDetay
    left join Makina On Makina.MakinaID = PartiAsamaStokDetay.MakinaId
    where PartiAsamaStokDetay.PartiSinifID = :APartiSinifID and PartiAsamaStokDetay.CikisTarihi is not null and Makina.Kazan = 1
    order by PartiAsamaStokDetay.CikisTarihi Desc
    into :SonGirdigiKazanMakinaId, :SonGirdigiKazanCikisTarihi;

  Select first 1 Ormeci.OrmeciId from PartiStokDetay
    left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = RafDetay.HamStokIrsaliyeId
    left join Ormeci on Ormeci.OrmeciID = HamStokIrsaliye.OrmeciId
    where PartiStokDetay.PartiSinifId = :APartiSinifId and Ormeci.OrmeciId is not null into :OrmeciId;

  suspend;
end
