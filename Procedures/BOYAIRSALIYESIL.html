CREATE PROCEDURE BOYAIRSALIYESIL(
    ABOYAIRSALIYEID INTEGER,
    AKULLANICIID INTEGER)
AS
begin
  insert into BoyaIrsaliyeIptal(BoyaIrsaliyeId, IrsaliyeSeriNo, IrsaliyeNo, IrsaliyeTarihi, SevkTarihi, FirmaId, IptalKullaniciId)
      Select BoyaIrsaliyeId, IrsaliyeSeriNo, IrsaliyeNo, IrsaliyeTarihi, SevkTarihi, FirmaId, :AKullaniciId from BoyaIrsaliye where BoyaIrsaliyeId = :ABoyaIrsaliyeId;

  delete from BoyaIrsaliye where BoyaIrsaliyeId = :ABoyaIrsaliyeId;
end
