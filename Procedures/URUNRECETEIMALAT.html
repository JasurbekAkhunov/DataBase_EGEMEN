CREATE PROCEDURE URUNRECETEIMALAT(
    AURUNID VARCHAR(20),
    AMIKTAR DOUBLE PRECISION,
    ADEPOID VARCHAR(10),
    AKULLANICIID INTEGER,
    ABOME DOUBLE PRECISION,
    ACOKELTME1 DOUBLE PRECISION,
    ACOKELTME2 DOUBLE PRECISION,
    ATARIH TIMESTAMP)
RETURNS (
    URUNURETIMID INTEGER)
AS
declare variable FURUNID varchar(20);
declare variable FURUNBIRIMID integer;
declare variable FURUNDEPOID varchar(10);
declare variable FBIRIMFIYAT double precision;
declare variable FURUNDOVIZID varchar(5);
declare variable FSTOKTABANDOVIZID varchar(5);
declare variable FSTOKID integer;
declare variable KAYITBASLADI integer;
declare variable FHAREKETTIPI integer;
declare variable FFISSERINO varchar(2);
declare variable FURETIMFISSERINO SeriNoSize5;
declare variable FFISNO integer;
declare variable FURUNMIKTAR double precision;
declare variable FMIKTAR double precision;
declare variable FTARIH timestamp;
declare variable FURUNDEFAULTBIRIMCARPAN double precision;
declare variable FSONALISBIRIMCARPAN double precision;
declare variable FDOVIZKURU double precision;
declare variable FNOW date;
declare variable FBIRIMMALIYET double precision;
declare variable FURUNRECETEBIRIMID integer;
declare variable FURUNURETIMID integer;
declare variable FLOT varchar(150);
declare variable FKALANMIKTAR double precision;
declare variable FCIKILANMIKTAR double precision;
declare variable FCIKILACAKMIKTAR double precision;
declare variable GIRISYOK EgeBoolean;
declare variable FTOPLAMLOT varchar(20);
declare variable BOYAKIMYASALLOTTAKIBIFIFO EgeBoolean;
declare variable FEGEMENFIRMAID integer;
declare variable FTUZKIMYASALTIPIID integer;
declare variable FURUNKIMYASALTIPIID integer;
declare variable FURUNRECETEKIMYASALTIPIID integer;
declare variable TF integer;
declare variable FKATIMADDE double precision;
begin
  FHareketTipi = 6;  /* Uretime Cikis */
  FNow = current_TimeStamp(0);
  FTarih =  FNow;
  if (ATarih is not null) then
  begin
    FTarih = ATarih;
    FNow = ATarih;
  end
  FBirimMaliyet = 0;

  if (AMiktar is null or AMiktar = 0) then exception UretilecekMiktarSifirOlamaz;

  KayitBasladi = 0;

  select STOKTABANDOVIZID, BoyaKimyasalLotTakibiFifo, EgemenFirmaID, TuzKimyasalTipiID from SISTEM where SISTEMID = 1 into :FStokTabanDovizId, :BoyaKimyasalLotTakibiFifo, :FEgemenFirmaID, :FTuzKimyasalTipiID ;
  Select UretimFisSeriNo, FisSeriNo  from Kullanici where KullaniciId = :AKullaniciId into :FUretimFisSeriNo, :FFisSeriNo;

  select UrunRecete.UrunBirimId from UrunRecete where UrunRecete.UrunReceteUrunId = :AUrunId into :FUrunReceteBirimId;
  select Urun.kimyasaltipiid from Urun where Urun.UrunId = :AUrunId into :FUrunKimyasalTipiID;

  FUrunUretimId = gen_id(UrunUretim_Id_Gen, 1);
  if (FEgemenFirmaID = 100 and  FUrunKimyasalTipiID = FTuzKimyasalTipiID)  then
  begin
    AMiktar = (:Amiktar - ACOKELTME2);
  end

   insert into UrunUretim(UrunUretimId, Tarih, UrunReceteUrunId, Miktar, UrunBirimId, DepoId, KullaniciId)
                  values(:FUrunUretimId, :FTarih, :AUrunId, :AMiktar, :FUrunReceteBirimId, :ADepoId, :AKullaniciId);


  FToplamLot = '';
  for select UrunReceteDetay.UrunId, UrunDefaultBirim.UrunBirimId, UrunReceteDetay.DepoId,
    UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / UrunDefaultBirim.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId,
    UrunDefaultBirim.Carpan, UrunReceteDetay.Miktar / RUB.Carpan * UrunDefaultBirim.Carpan, Urun.KimyasalTipiID from UrunReceteDetay
    left join UrunBirim RUB on RUB.urunBirimId = UrunReceteDetay.urunBirimId
    left join UrunBirim UrunDefaultBirim on UrunDefaultBirim.urunId = UrunReceteDetay.UrunId and UrunDefaultBirim.DefaultBirim = 1
    left join Urun on Urun.UrunId = UrunReceteDetay.UrunId
    left join UrunFiyat on UrunFiyat.UrunId = UrunReceteDetay.UrunId
    left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
    where UrunReceteDetay.UrunReceteUrunId = :AUrunId and UrunReceteDetay.Miktar > 0
    into :FUrunId, :FUrunBirimId, :FUrunDepoId, :FBirimFiyat, :FUrunDovizId, :FUrunDefaultBirimCarpan, :FUrunMiktar, :FUrunReceteKimyasalTipiID do
  begin
    if (KayitBasladi = 0) then
    begin
      select FisNo from FisNumaratorAl(:FUretimFisSeriNo) into :FFisNo;

      FStokId = gen_id(Stok_Id_Gen, 1);

      insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, KullaniciId, UrunUretimId)
                  values(:FStokId, :FHareketTipi, :FUretimFisSeriNo, :FFisNo, :FTarih, :FStokTabanDovizId, 1, 1, :AKullaniciId, :FUrunUretimId);

      KayitBasladi = 1;
    end

    if (FUrunReceteKimyasalTipiID = FTuzKimyasalTipiID) then
    begin
      if (:ABome > 0 and ACokeltme1 > 0) then TF = 1;
      if (TF =1) then
      begin
        select Bome.KatiMadde from Bome where Bome.BomeID = :ABome and Bome.KimyasalTipiID = :FUrunReceteKimyasalTipiID into :FKatiMadde;

        FMiktar =  (AMiktar + :Acokeltme2) * FKatiMadde / 1000 /(1-(:ACokeltme1/100));
      end
    end
    else
    begin
      FMiktar = FUrunMiktar * AMiktar;
    end

    if (FMiktar is null) then FMiktar = 0;

    if (FBirimFiyat is null) then FBirimFiyat = 0;

    if (FUrunDovizId is not null and FBirimFiyat > 0) then
    if (FStokTabanDovizId <> FUrunDovizId) then
    begin
      select DovizKuru from dovizkurugetir(:FUrunDovizId, :FStokTabanDovizId, :FTarih) into :FDovizKuru;
      FBirimFiyat = FBirimFiyat * FDovizKuru;
    end

    FBirimMaliyet = FBirimMaliyet + FBirimFiyat * FUrunMiktar;

    GirisYok = 1;
    FCikilanMiktar = 0;
    FCikilacakMiktar = 0;
    FLot = Null;
    if (BoyaKimyasalLotTakibiFifo = 1) then
    begin
      for select UrunFifoBilgisi.Lot, UrunFifoBilgisi.KalanMiktar from UrunFifoBilgisi(:FUrunId, Null)
          order by GirisTarihi into :FLot, FKalanMiktar do
      begin
        GirisYok = 0;

        if (FKalanMiktar < FMiktar - FCikilanMiktar) then
          FCikilacakMiktar = FKalanMiktar;
        else
          FCikilacakMiktar = FMiktar - FCikilanMiktar;

        insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, UrunId, Lot, UrunBirimId, Miktar, BirimFiyat, KullaniciId)
                      values(:FTarih, :FHareketTipi, :FStokId, :FUrunDepoId, :FUrunId, :FLot, :FUrunBirimId, :FCikilacakMiktar, :FBirimFiyat, :AKullaniciId);

        FCikilanMiktar = FCikilanMiktar + FCikilacakMiktar;

        if (FEgemenFirmaID = 131 or FEgemenFirmaID = 133 or FEgemenFirmaID=146 )  then /*Ünver , BTEKS Çorlu, Özsar*/
        begin
          FLot = case when Extract(day  FROM cast(FNow as date))<10 then '0'||Extract(day  FROM cast(FNow as date)) else Extract(day  FROM cast(FNow as date)) end||  /* gün */
                   case when Extract(month FROM cast(FNow as date))<10 then '0'||Extract(month FROM cast(FNow as date)) else Extract(month FROM cast(FNow as date)) end|| /* ay */
                   Extract(year FROM cast(FNow as date)); /* yil */
          if (FToplamLot like Coalesce(FLot, '')||'%' ) then FLot = '';
        end

        if (FToplamLot = '') then FToplamLot = Coalesce(FLot, '');
        else if (Coalesce(FLot, '') <> '' and Char_Length(FToplamLot||'+'||Coalesce(FLot, '')) < 21) then FToplamLot = FToplamLot||'+'||Coalesce(FLot, '');

        if (FCikilanMiktar = FMiktar) then break;
      end
    end
    FMiktar = FMiktar - FCikilanMiktar;
    if (FMiktar > 0 or GirisYok = 1) then
      insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, UrunId, Lot, UrunBirimId, Miktar, BirimFiyat, KullaniciId)
                    values(:FTarih, :FHareketTipi, :FStokId, :FUrunDepoId, :FUrunId, '', :FUrunBirimId, :FMiktar, :FBirimFiyat, :AKullaniciId);

  end
  if (KayitBasladi = 1) then
  begin
    FHareketTipi = 7;  /* Uretim */

    select FisNo from FisNumaratorAl(:FFisSeriNo) into :FFisNo;

    FStokId = gen_id(Stok_Id_Gen, 1);

    insert into Stok(StokId, HareketTipi, SeriNo, FisNo, DepoId, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, KullaniciId, UrunUretimId)
                values(:FStokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :ADepoId, :FTarih, :FStokTabanDovizId, 1, 1, :AKullaniciId, :FUrunUretimId);


    insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, UrunId, Lot, UrunBirimId, Miktar, BirimFiyat, KullaniciId)
                values(:FTarih, :FHareketTipi, :FStokId, :ADepoId, :AUrunId, :FToplamLot, :FUrunReceteBirimId, :AMiktar, :FBirimMaliyet, :AKullaniciId);

    UrunUretimId = FUrunUretimId;
    Suspend;
  end
  else exception UrununRecetesiYok;
end
