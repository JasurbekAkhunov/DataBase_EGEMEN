CREATE PROCEDURE TRANSFERICINORMEIRSDETAYGETIR(
    AORMEIRSALIYEID INTEGER)
RETURNS (
    BOYAIRSALIYEID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    DOVIZKURU DOUBLE PRECISION,
    KDVORANI DOUBLE PRECISION,
    TRANSFERKODU VARCHAR(55),
    BIRIMTRANSFERKODU VARCHAR(10),
    BIRIMFIYATTL DOUBLE PRECISION,
    TUTARTL DOUBLE PRECISION,
    VADEGUN INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    SIPDETAYTRANSFERKODU VARCHAR(35),
    SIPARISTARIHI TIMESTAMP)
AS
declare variable IrsaliyeTarihi     TimeStamp;
declare variable SistemDovizId      varchar(5);
declare variable FHamSatisTransferKodu varchar(55);
declare variable FSatisTransferKodu     varchar(55);
declare variable FFasonOrmeTransferKodu varchar(55);
declare variable FYariMamulTransferKodu varchar(55);
declare variable FFasonTransferKodu     varchar(55);
declare variable OrmeSiparisSekliID integer;
declare variable Fason              integer;

declare variable assSatis               integer;
declare variable assFasonArtiSatis      integer;
declare variable assFason               integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable SiparisIslemGrubuId    integer;
declare variable HareketTipiId   integer;
declare variable FirmaId         varchar(20);
declare variable EgemenFirmaID         integer;
declare variable RenkId integer;
declare variable DesenId varchar(20);
declare variable uitOrme integer;
declare variable uitBukum integer;
declare variable shtFasonUretim integer;
declare variable shtSatis integer;
declare variable shtAlis integer;
declare variable shtAlisIade integer;
declare variable FUretimIslemTipiID integer;
declare variable FAlisTransferKodu varchar(55);
declare variable FRenkliAlisTransferKodu varchar(55);
declare variable FAlisIadeTransferKodu varchar(55);
declare variable FFasonBukumTransferKodu varchar(55);
begin
  BoyaIrsaliyeID = Null;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  Select assSatis, assFasonArtiSatis, assFason from OrmeSiparisSekliGetir into :assSatis, :assFasonArtiSatis, :assFason;
  select uitBukum, uitOrme from UretimIslemTipiGetir into :uitBukum, :uitOrme;
  Select shtFasonUretim, shtSatis, shtAlis, ShtAlisIade from OrmeStokHareketTipiGetir into :shtFasonUretim, :shtSatis, :shtAlis, :ShtAlisIade;

  Select SistemDovizId, EgemenFirmaID from Sistem where sistemId = 1 into :SistemDovizId, :EgemenFirmaID;

  for select OrmeIrsaliye.HareketTipi, OrmeIrsaliye.FirmaId, StokHareket.HamId, Ham.Adi, Ham.HamTipiId, StokHareket.KapAdedi, StokHareket.Miktar,
             StokHareket.BirimFiyat, OrmeIrsaliye.DovizId, StokHareket.Kdv,
             Ham.HamSatisTransferKodu, Ham.SatisTransferKodu, Ham.AlisTransferKodu, Ham.RenkliAlisTransferKodu, Ham.AlisIadeTransferKodu, Ham.FasonTransferKodu, Ham.FasonOrmeTransferKodu, Ham.FasonBukumTransferKodu, Ham.YariMamulTransferKodu,
             OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.DovizId, StokHareket.VadeGun,
             BoyaSiparis.OrmeSiparisSekliID, BoyaSiparis.Fason, StokHareket.OrmeSiparisDetayId, StokHareket.BoyaSiparisDetayId,
             StokHareket.BoyaSiparisBaskiId, StokHareket.RenkId, StokHareket.DesenId,
             BoyaSiparisDetay.IslemGrubuId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             coalesce(OrmeSiparisDetay.MusterisiparisNo, BoyaSiparisDetay.MusteriSiparisNo), BoyaSiparisDetay.SipDetayTransferKodu, OrmeIrsaliye.UretimIslemTipiID, BoyaSiparis.SiparisTarihi from StokHareket
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = StokHareket.BoyaSiparisDetayId
    Left Join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    Left Join Ham on Ham.HamId = StokHareket.HamId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    where StokHareket.OrmeIrsaliyeId = :AOrmeIrsaliyeId and StokHareket.HamIade = 0
    order by StokHareketId
    into :HareketTipiId, :FirmaId, :HamId, :HamAdi, :HamTipiId, :KapAdedi, :Miktar, :BirimFiyat, :DovizId, :KdvOrani,
         :FHamSatisTransferKodu, :FSatisTransferKodu, :FAlisTransferKodu, :FRenkliAlisTransferKodu, :FAlisIadeTransferKodu, :FFasonTransferKodu, :FFasonOrmeTransferKodu, :FFasonBukumTransferKodu, :FYariMamulTransferKodu,
         :IrsaliyeTarihi, :DovizId, :VadeGun, :OrmeSiparisSekliID, :Fason, :OrmeSiparisDetayId, :BoyaSiparisDetayId,
         :BoyaSiparisBaskiId, :RenkId, :DesenId, :SiparisIslemGrubuId,
         :BoyaSiparisYil, :BoyaSiparisId, :MusteriSiparisNo, :SipDetayTransferKodu, :FUretimIslemTipiID, :SiparisTarihi do
  begin
    if (HareketTipiId = shtSatis) then
    begin
      if (DovizID <> SistemDovizId) then
      begin
        select DovizKuru from FirmaDovizKurlariniGetir(:FirmaId, :DovizId, :SistemDovizId, :IrsaliyeTarihi) into :DovizKuru;
      end
      else DovizKuru = 1;

      BirimFiyatTL = BirimFiyat * DovizKuru;
      TutarTL = BirimFiyatTL * Miktar;

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      TransferKodu = Null;

      if (EgemenFirmaID = 32) then/* Biessici ise */
      begin
        if (RenkId is not null) then/*Renkli Satıs*/
          TransferKodu = FSatisTransferKodu;
        else
          TransferKodu = FHamSatisTransferKodu; /*Satıs*/
      end
      else
      begin
        if (BoyaSiparisBaskiId is not null or DesenId is not null) then
        begin
          if (Fason = 0 or OrmeSiparisSekliId = assSatis) then
          begin
            TransferKodu = Null;

            Select BaskiSatisKumasTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu;
          end
          else
          begin
            TransferKodu = Null;

            Select BaskiFasonKumasTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu;
          end
        end
        else if (BoyaSiparisDetayId is not null or RenkId is not null) then
        begin
          if (Fason = 0 or OrmeSiparisSekliId = assSatis) then
          begin
            TransferKodu = Null;

            if ( HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
              Select SatisIplikTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu;
            else
              Select SatisKumasTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu;
          end
          else
          begin
            TransferKodu = Null;

            if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
              Select FasonIplikTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu;
            else
              Select FasonKumasTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu;

          end
        end
        else
        begin
          if (OrmeSiparisDetayID is not null and OrmeSiparisSekliID = 2) then
            TransferKodu = FFasonOrmeTransferKodu;
          else
            TransferKodu = FHamSatisTransferKodu;
        end
      end
      BirimTransferKodu = '';

      if (HamTipiId <> 2) then Select TransferKodu from Birim where Birim.BirimID = 'Kg' into :BirimTransferKodu;
      else Select TransferKodu from Birim where Birim.BirimID = 'Mt' into :BirimTransferKodu;

      suspend;
    end
    else
    begin
      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      BirimFiyatTL = BirimFiyat;
      TutarTL = BirimFiyatTL * Miktar;

      if (EgemenFirmaID <> 0 and EgemenFirmaID <> 32) then
      begin
        TransferKodu = FHamSatisTransferKodu;
        BirimTransferKodu = '';
      end
      else if (EgemenFirmaID = 0) then/* Nuryildiz ise */
      begin
        if (HareketTipiID = shtFasonUretim) then TransferKodu = FYariMamulTransferKodu;
        else TransferKodu = FHamSatisTransferKodu;
      end
      else if (EgemenFirmaID = 32) then/* Biessici ise */
      begin
        if (HareketTipiId = shtFasonUretim and FUretimIslemTipiId = uitOrme) then /*Fason Üretim Giriş Örme ise*/
          TransferKodu = FFasonOrmeTransferKodu;
        else if (HareketTipiId = shtFasonUretim and FUretimIslemTipiId = uitBukum) then /*Fason Üretim Giriş Büküm ise*/
          TransferKodu = FFasonBukumTransferKodu;
        else if (HareketTipiId = shtAlisIade) then /*Alış İade ise*/
          TransferKodu = FAlisIadeTransferKodu;
        else if (HareketTipiId = shtAlis and RenkId is not null) then /*Renkli Alış ise*/
          TransferKodu = FRenkliAlisTransferKodu;
        else TransferKodu = FAlisTransferKodu; /*Alış ise*/

        BirimTransferKodu = '';
      end

      if (HamTipiId <> 2) then Select TransferKodu from Birim where Birim.BirimID = 'Kg' into :BirimTransferKodu;
      else Select TransferKodu from Birim where Birim.BirimID = 'Mt' into :BirimTransferKodu;

      suspend;
    end
  end
end
