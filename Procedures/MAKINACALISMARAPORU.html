CREATE PROCEDURE MAKINACALISMARAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AUNITEID INTEGER,
    AMAKINAGRUBU VARCHAR(4),
    AMAKINAID VARCHAR(10))
RETURNS (
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    UNITEADI VARCHAR(30),
    TOPLAMSURE DOUBLE PRECISION,
    TOPLAMSURESTR VARCHAR(30),
    CALISMASURESI DOUBLE PRECISION,
    CALISMASURESISTR VARCHAR(30),
    DURUSSURESI DOUBLE PRECISION,
    DURUSSURESISTR VARCHAR(30),
    CALISMAORANI DOUBLE PRECISION)
AS
declare variable DKtarihi TimeStamp;
declare variable DKDurumu integer;
declare variable RaporBitisTarihi timestamp;
declare variable VardiyaBaslangicSaati   Double Precision;
begin
  ABitisTarihi = ABitisTarihi + 1;

  Select TimeStampToDouble(TimeStampToTime(VardiyaBaslangicSaati)) from sistem into :VardiyaBaslangicSaati;

  RaporBitisTarihi = ABitisTarihi + VardiyaBaslangicSaati;
  if (RaporBitisTarihi > Current_TimeStamp(0)) then RaporBitisTarihi = Current_TimeStamp(0);

  ToplamSure = RaporBitisTarihi - (ABaslangicTarihi + VardiyaBaslangicSaati);
  ToplamSureStr = TarihToStr(ToplamSure);

  for select Makina.MakinaId, Makina.Adi, Unite.Adi, Sum(CalismaSuresi) from makina
    Left join Unite On Unite.UniteId = Makina.UniteId
    Left join MakinaCalismaSuresi On MakinaCalismaSuresi.MakinaId = Makina.MakinaId and MakinaCalismaSuresi.VardiyaTarihi >= :ABaslangicTarihi and MakinaCalismaSuresi.VardiyaTarihi < :ABitisTarihi
    where (Makina.UniteId = :AUniteId or :AUniteId is null) and Makina.UretimRaporlarinaDahil = 1 /*Ali kapattÄ±.and Makina.OtomasyonaDahil = 1*/ and Makina.CalisiyorSinyaliVarmi = 1 and
          (Makina.MakinaId = :AMakinaId or :AMakinaId is null) and
          (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null)
    group by Makina.MakinaId, Makina.Adi,  Unite.Adi
    order by Makina.MakinaId
    into :MakinaId, :MakinaAdi, :UniteAdi, :CalismaSuresi do
  begin
    select MakinaCalismaSuresi.DKTarihi,DkDurumu from MakinaCalismaSuresi   where MakinaCalismaSuresi.MakinaId=:MakinaId and MakinaCalismaSuresi.VardiyaTarihi= (Select  MAX(VardiyaTarihi) from  MakinaCalismaSuresi where  MakinaCalismaSuresi.MakinaId =:MakinaId) into  :DkTarihi, :DkDurumu;
    if (CalismaSuresi is null) then CalismaSuresi = 0;

    if (DKTarihi is not null) then
    if (DKDurumu = 1) then
    if (RaporBitisTarihi > DKTarihi) then
    begin
      if (DkTarihi > (ABaslangicTarihi + VardiyaBaslangicSaati)) then CalismaSuresi = CalismaSuresi + RaporBitisTarihi - DkTarihi;
      else CalismaSuresi = CalismaSuresi + RaporBitisTarihi - (ABaslangicTarihi + VardiyaBaslangicSaati);
    end

    if (CalismaSuresi = 0 or CalismaSuresi is null) then CalismaSuresiStr = '';
    else CalismaSuresiStr = TarihToStr(CalismaSuresi);

    DurusSuresi = ToplamSure - CalismaSuresi;

    if (DurusSuresi = 0 or DurusSuresi is null) then DurusSuresiStr = '';
    else DurusSuresiStr = TarihToStr(DurusSuresi);

    ToplamSureStr = TarihToStr(ToplamSure);

    if (ToplamSure > 0) then CalismaOrani = CalismaSuresi * 100 / ToplamSure;
    suspend;
  end
end
