CREATE PROCEDURE FIRMAFASONHAMSTOKKODULISTESI(
    AFIRMAID VARCHAR(20),
    ARAPORHAMTIPI INTEGER)
RETURNS (
    HAMID INTEGER,
    ADI VARCHAR(255),
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20))
AS
begin
  for select DISTINCT RafDetay.HamId, Ham.Adi, Ham.HamTipiId, HamTipi.Adi from HamStokDetay
  left join RafDetay on RafDetay.HamStokDetayid = HamStokDetay.HamStokDetayid
  left join ham on ham.hamid = hamstokdetay.hamid
  left join HamTipi on HamTipi.HamTipiid = Ham.HamTipiId
  where HamStokDetay.StokFirmaId = :AFirmaId and HamStokDetay.Fason = 1 and
       ( ( ( Ham.HamTipiId = 0 or Ham.HamTipiId = 5) and :ARaporHamTipi = 0) or (Ham.HamTipiId <> 0 and Ham.HamTipiId <> 5 and :ARaporHamTipi <> 0)) and
       RafDetay.Miktar > (RafDetay.AktarmayaCikanKilo - RafDetay.AktarmadanIadeEdienKilo + RafDetay.IadeEdilecekMiktar + RafDetay.IadeEdilenMiktar + RafDetay.PartilenenMiktar + RafDetay.RafTransferMiktar + RafDetay.SatisMiktar)
  into :HamId, :Adi, :HamTipiId, :HamTipiAdi do suspend;
end
