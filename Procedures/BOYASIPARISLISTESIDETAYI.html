CREATE PROCEDURE BOYASIPARISLISTESIDETAYI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASIPARISSEKLI INTEGER,
    AISLEMTIPIID INTEGER,
    ASIPARISTIPIID INTEGER,
    AKAPALI INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AIMALATTIPIID INTEGER,
    AORMESIPARISTIPIID INTEGER,
    ADOKUMASIPARISTIPIID INTEGER,
    AREFERANSSIPARISNO VARCHAR(30))
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISRENKNO INTEGER,
    BOYASIPARISRENKNOSTR VARCHAR(50),
    RENKADI VARCHAR(40),
    MUSTERISIPARISNO VARCHAR(50),
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEATISID INTEGER,
    LABRECETEATISNO VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    ISLEMTIPIADI VARCHAR(20),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    KAPALI INTEGER,
    OZELTESTLER VARCHAR(255),
    OZELAPRELER VARCHAR(255),
    ILAVEISLEMLER VARCHAR(255),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSSIPARISNO VARCHAR(30))
AS
declare variable Fason integer;
begin
  for select BoyaSiparisListesi.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.OrmeSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.LabReceteId,
             LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.LabReceteAtisId, LabReceteAtis.AtisNoStr,
             BoyaSiparisDetay.BoyaSiparisRenkNo, BoyaSiparisDetay.RenkAdi, IslemTipi.Adi, BoyaSiparisDetay.SiparisKilo,
             BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.HamId, Ham.Adi, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi,
             BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, BoyaSiparisDetay.VadeGun,
             BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi, BoyaSiparisDetay.Kapali,
             IK.Adi, BoyaSiparisDetay.InsertTarihi, Sk.Adi, BoyaSiparisDetay.DegisimTarihi,
             BoyaSiparisDetay.OzelTestler, BoyaSiparisDetay.OzelApreler, BoyaSiparisDetay.IlaveIslemler,
             BoyaSiparisListesi.Fason, BoyaSiparisDetay.FlamKodu, BoyaSiparisDetay.MelanjKodu, BoyaSiparisDetay.ReferansSiparisNo from BoyaSiparisListesi(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :ASiparisSekli, :AIslemTipiId, :ASiparisTipiId, :AKapali, 0, :AMusteriSiparisNo, :AImalatTipiID, :AOrmeSiparisTipiID, :ADokumaSiparisTipiID, :AReferansSiparisNo)
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparisListesi.BoyaSiparisKayitId
  left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
  Left Join Kullanici OK on OK.KullaniciId = BoyaSiparisDetay.OnaylayanKullaniciId
  Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisDetay.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisDetay.KullaniciId
  where BoyaSiparisDetay.SiparisTipiId = :ASiparisTipiId or :ASiparisTipiId is null
  order by BoyaSiparisListesi.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId
  into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :OrmeSiparisDetayId, :MusteriSiparisNo, :LabReceteId, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisId, :LabReceteAtisNo,
       :BoyaSiparisRenkNo, :RenkAdi,  :IslemTipiAdi, :SiparisKilo, :SiparisMetre, :HamId, :HamAdi, :SiparisProsesId, :SiparisProsesAdi, :BirimFiyat, :DovizId, :VadeGun,
       :TerminTarihi, :RevizeTerminTarihi, :Kapali,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :OzelTestler, :OzelApreler, :IlaveIslemler, :Fason, :FlamKodu, :MelanjKodu, :ReferansSiparisNo do
  begin
    if (Fason = 0 and ASatisBoyaSiparisFiyatGorebilir = 0) then
    begin
      BirimFiyat = null;
      DovizId    = null;
      VadeGun    = null;
    end
    else if (Fason = 1 and AFasonBoyaSiparisFiyatGorebilir = 0) then
    begin
      BirimFiyat = null;
      DovizId    = null;
      VadeGun    = null;
    end

    if (BoyaSiparisRenkNo is not null) then
      BoyaSiparisRenkNoStr = '(' || EgemenIntToLzStr(BoyaSiparisRenkNo, 2) || ') - ' || RenkAdi;
    else
      BoyaSiparisRenkNoStr = '';

    suspend;
  end
end
