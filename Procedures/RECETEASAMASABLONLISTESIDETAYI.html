CREATE PROCEDURE RECETEASAMASABLONLISTESIDETAYI(
    ARECETEASAMAID INTEGER)
RETURNS (
    RECETEASAMASABLONID INTEGER,
    RECETEALTASAMAID INTEGER,
    RECETEALTASAMAADI VARCHAR(40),
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNTIPIADI VARCHAR(20),
    BOYATIPIADI VARCHAR(20),
    BIRIMID VARCHAR(2),
    MIKTAR DOUBLE PRECISION,
    BANYO DOUBLE PRECISION)
AS
begin
  For Select ReceteAsamaSablonDetay.ReceteAltAsamaId, ReceteAltAsama.Adi, ReceteAsamaSablonDetay.UrunId, Urun.Adi,
             UrunBirim.BirimId, ReceteAsamaSablonDetay.Miktar, ReceteAsamaSablonDetay.Banyo from ReceteAsamaSablon
    left join ReceteAsamaSablonDetay on ReceteAsamaSablonDetay.ReceteAsamaSablonId = ReceteAsamaSablon.ReceteAsamaSablonId
    left join Urun on Urun.UrunId = ReceteAsamaSablonDetay.UrunId
    left join UrunBirim on UrunBirim.UrunBirimId = ReceteAsamaSablonDetay.UrunBirimId and UrunBirim.UrunId = ReceteAsamaSablonDetay.UrunId
    left join ReceteAltAsama on ReceteAltAsama.ReceteAltAsamaId = ReceteAsamaSablonDetay.ReceteAltAsamaId
    where ReceteAsamaSablon.ReceteAsamaId = :AReceteAsamaId and ReceteAsamaSablonDetay.ReceteAsamaSablonDetayId is not null
    order by ReceteAsamaSablon.ReceteAsamaSablonID, ReceteAsamaSablonDetay.Sira
    into :ReceteAltAsamaId, :ReceteAltAsamaAdi, :UrunId, :UrunAdi, :BirimId, :Miktar, :Banyo do suspend;
end
