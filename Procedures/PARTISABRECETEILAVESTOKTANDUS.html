CREATE PROCEDURE PARTISABRECETEILAVESTOKTANDUS(
    APARTIKAYITID INTEGER,
    APARTISABLONISEMRIID INTEGER,
    APARTISABLONILAVEID INTEGER)
AS
declare variable FFisSeriNo             varchar(2);
declare variable FFisNo                 integer;
declare variable FStokId                integer;
declare variable FHareketTipi           integer;
declare variable FStokTabanDovizId      varchar(5);
declare variable PartiSinif  varchar(1);
declare variable FTarih                 TimeStamp;
declare variable FUrunMutfakDepoNo      varchar(10);
declare variable FDepoStokMevcudu       Double Precision;
declare variable FUrunId                varchar(20);
declare variable FUrunBirimId           integer;
declare variable FUretimeCikisFiyati    Double Precision;
declare variable FUretimeCikisFiyatiDovizId  varchar(5);
declare Variable FMiktar                Double Precision;
declare Variable FDovizKuru             Double Precision;
declare variable FBaskiReceteStokKontrolu integer;
declare variable SablonReceteSarfStokGrupId varchar(4);
declare variable FPartiKayitId integer;
declare variable FLot                       varchar(50);
declare variable GirisYok                   EgeBoolean;
declare variable FCikilanMiktar             BoyaMiktarSize;
declare variable FCikilacakMiktar           BoyaMiktarSize;
declare variable FToplamLot                 varchar(50);
declare variable BoyaKimyasalLotTakibiFifo  EgeBoolean;
declare variable FUrunBirimCarpan           double precision;
declare variable FKalanMiktar               BoyaMiktarSize;
declare variable FPartiAsamalariId integer;
begin
  FHareketTipi = 6;  /* Uretime Cikis */
  FTarih = Current_date;

  Select FisSeriNo from Kullanici where KullaniciId = 1 into :FFisSeriNo;

  select FisNo from FisNumaratorAl(:FFisSeriNo) into :FFisNo;

  FStokId = gen_id(Stok_Id_Gen, 1);

  select STOKTABANDOVIZID, SablonReceteSarfStokGrupId from SISTEM where SISTEMID = 1 into :FStokTabanDovizId, :SablonReceteSarfStokGrupId;

  select Parti.Sinif from Parti where Parti.PartikayitId = :APartiKayitId into :PartiSinif;

  select first 1 PartiAsamalari.PartiAsamalariId from Parti
    left join PartiAsamalari on PartiAsamalari.PartiKayitId = Parti.PartiKayitId
    left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
    where Parti.PartikayitId = :APartiKayitId and Asama.BaskiAsamasimi = 1 into :FPartiAsamalariId;

      insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, SablonRecetesi, PartiKayitId, PartiSablonIsEmriID, Sinif, KullaniciId, PartiSablonIlaveId)
                      values(:FStokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :FTarih, :FStokTabanDovizId, 1, 1, 2, :APartiKayitId, :APartiSablonIsEmriID, :PartiSinif, 1, :APartiSablonIlaveId);

    for select PartiSablonIlave.PartiKayitId, PartiSablonIlave.DepoId, PartiSablonReceteIlave.UrunId, PartiSablonReceteIlave.UrunBirimId,
               UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / UBPSB.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId, Sum(PartiSablonReceteIlave.KullanilanMiktar),
               UBPSB.Carpan from PartiSablonIlave
      left join PartiSablonReceteIlave on PartiSablonReceteIlave.PartiSablonIlaveId = PartiSablonIlave.PartiSablonIlaveId
      left join UrunFiyat on UrunFiyat.UrunId = PartiSablonReceteIlave.UrunId
      left join UrunBirim UBPSB on UBPSB.UrunBirimId = PartiSablonReceteIlave.UrunBirimId
      left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
      where (PartiSablonIlave.PartiSablonIlaveId = :APartiSablonIlaveId or PartiSablonIlave.PartiSablonIsEmriID = :APartiSablonIsEmriID) and PartiSablonReceteIlave.PartiSablonIlaveId is not null
      group by PartiSablonIlave.PartiKayitId, PartiSablonIlave.DepoId, PartiSablonReceteIlave.UrunId, PartiSablonReceteIlave.UrunBirimId,
               UrunFiyat.UretimeCikisFiyati, UrunFiyat.UretimeCikisFiyatiDovizId, UBUCF.Carpan, UBPSB.Carpan
      into :FPartiKayitId, FUrunMutfakDepoNo, :FUrunId, :FUrunBirimId, :FUretimeCikisFiyati, :FUretimeCikisFiyatiDovizId, :FMiktar, :FUrunBirimCarpan do
    begin

      if (FStokTabanDovizId <> FUretimeCikisFiyatiDovizId) then
      begin
        select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :FStokTabanDovizId, Current_Date) into :FDovizKuru;

        FUretimeCikisFiyati = FUretimeCikisFiyati * FDovizKuru;
      end

     /*MUSTAFA*/
      select BoyaKimyasalLotTakibiFifo from SISTEM where SISTEMID = 1  into  :BoyaKimyasalLotTakibiFifo;

      FLot = Null;
      GirisYok = 1;
      FCikilanMiktar = 0;
      FCikilacakMiktar = 0;
      FToplamLot = '';

      if (BoyaKimyasalLotTakibiFifo = 1) then
      begin
        for select UrunFifoBilgisi.Lot, UrunFifoBilgisi.KalanMiktar * (:FUrunBirimCarpan / UrunBirim.Carpan) from UrunFifoBilgisi(:FUrunId, :FUrunMutfakDepoNo)
            left join UrunBirim on UrunBirim.UrunBirimID = UrunFifoBilgisi.UrunBirimId
            order by GirisTarihi into :FLot, FKalanMiktar do
        begin
          GirisYok = 0;

          if (FKalanMiktar < FMiktar - FCikilanMiktar) then
            FCikilacakMiktar = FKalanMiktar;
          else
            FCikilacakMiktar = FMiktar - FCikilanMiktar;

          insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, StokGrupId, UrunId,Lot ,UrunBirimId, Miktar, BirimFiyat, PartiKayitId, Otomasyon, KullaniciId, PartiAsamalariId)
                  values(:FTarih, :FHareketTipi, :FStokId, :FUrunMutfakDepoNo, :SablonReceteSarfStokGrupId, :FUrunId, :FLot,  :FUrunBirimId, :FCikilacakMiktar, :FUretimeCikisFiyati, :FPartiKayitId, 1, 1, :FPartiAsamalariId);


          FCikilanMiktar = FCikilanMiktar + FCikilacakMiktar;

          if (FToplamLot = '') then FToplamLot = FLot;
          else if (Char_Length(FToplamLot||'+'||Flot) < 21) then FToplamLot = FToplamLot||'+'||Flot;

          if (FCikilanMiktar = FMiktar) then break;
        end

        FMiktar = FMiktar - FCikilanMiktar;
        if (FMiktar > 0 or GirisYok = 1) then
          insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, StokGrupId, UrunId ,UrunBirimId, Miktar, BirimFiyat, PartiKayitId, Otomasyon, KullaniciId, PartiAsamalariId)
                  values(:FTarih, :FHareketTipi, :FStokId, :FUrunMutfakDepoNo, :SablonReceteSarfStokGrupId, :FUrunId,  :FUrunBirimId, :FMiktar, :FUretimeCikisFiyati, :FPartiKayitId, 1, 1, :FPartiAsamalariId);

      end
      else
      insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, StokGrupId, UrunId, UrunBirimId, Miktar, BirimFiyat, PartiKayitId, Otomasyon, KullaniciId, PartiAsamalariId)
                  values(:FTarih, :FHareketTipi, :FStokId, :FUrunMutfakDepoNo, :SablonReceteSarfStokGrupId, :FUrunId, :FUrunBirimId, :FMiktar, :FUretimeCikisFiyati, :FPartiKayitId, 1, 1, :FPartiAsamalariId);
    end
end
