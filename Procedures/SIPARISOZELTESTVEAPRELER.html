CREATE PROCEDURE SIPARISOZELTESTVEAPRELER(
    ABOYASIPARISDETAYID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    OZELTESTLER VARCHAR(3000),
    OZELAPRELER VARCHAR(3000),
    ILAVEISLEMLER VARCHAR(3000))
AS
declare variable FOzelTestler varchar(3000);
declare variable FOzelApreler varchar(3000);
declare variable FIlaveIslemler varchar(3000);
declare variable FAdi Adi80;
begin
  FOzelTestler = '';
  FOzelApreler = '';
  FIlaveIslemler = '';

  BoyaSiparisDetayId = :ABoyaSiparisDetayId;

  For Select OzelTest.Adi from SiparisOzelTestler
    left join OzelTest on OzelTest.OzelTestId = SiparisOzelTestler.OzelTestId
    where SiparisOzelTestler.BoyaSiparisDetayId = :ABoyaSiparisDetayId
    order By OzelTest.Adi
    into :FAdi do
  begin
    if (FOzelTestler <> '') then FOzelTestler = FOzelTestler || chr(13) || chr(10);
    FOzelTestler = FOzelTestler || FAdi;
  end

  For Select OzelApre.Adi from SiparisOzelApreler
    left join OzelApre on OzelApre.OzelApreId = SiparisOzelApreler.OzelApreId
    where SiparisOzelApreler.BoyaSiparisDetayId = :ABoyaSiparisDetayId
    order By OzelApre.Adi
    into :FAdi do
  begin
    if (FOzelApreler <> '') then FOzelApreler = FOzelApreler || chr(13) || chr(10);
    FOzelApreler = FOzelApreler || FAdi;
  end

  For Select IlaveIslemler.Adi from SiparisIlaveIslemler
    left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = SiparisIlaveIslemler.IlaveIslemlerId
    where SiparisIlaveIslemler.BoyaSiparisDetayId = :ABoyaSiparisDetayId
    order By IlaveIslemler.Adi
    into :FAdi do
  begin
    if (FIlaveIslemler <> '') then FIlaveIslemler = FIlaveIslemler || chr(13) || chr(10);
    FIlaveIslemler = FIlaveIslemler || FAdi;
  end

  if (FOzelTestler <> '' or FOzelApreler <> '' or FIlaveIslemler <> '') then
  begin
    OzelTestler = FOzelTestler;
    OzelApreler = FOzelApreler;
    IlaveIslemler = FIlaveIslemler;
  end
  suspend;
end
