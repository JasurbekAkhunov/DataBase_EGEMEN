CREATE PROCEDURE URETIMECIKISFISIDETAYIGETIR(
    ASTOKID INTEGER)
RETURNS (
    STOKDETAYID INTEGER,
    DEPOID VARCHAR(10),
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    PARTIIDSTR VARCHAR(30),
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30),
    BIRIMID VARCHAR(2),
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    REFERANSSTOKDETAYID INTEGER,
    TUTAR DOUBLE PRECISION,
    YATIRIM INTEGER,
    YATIRIMSTR VARCHAR(10),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    URETIMECIKISTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    LOT VARCHAR(150),
    HGSIPARISKAYITID INTEGER,
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER,
    HGSIPARISIDSTR VARCHAR(30),
    HGSIPARISDETAYID INTEGER,
    HGISEMRIKAYITID INTEGER,
    URETIMPARTINO VARCHAR(30),
    MODELID MODELKODUSIZE,
    URETIMISLEMLERI VARCHAR(100),
    ORMEMAKINAID MAKINAKODUSIZE,
    MUSTERISIPARISNO VARCHAR(50))
AS
declare variable FPartiYil Integer;
declare variable FPartiId Integer;
begin
  for select StokDetayId, StokDetay.DepoId, StokDetay.UrunId, Urun.Adi,  Parti.PartiYil, Parti.PartiId, SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId,
     UrunBirim.BirimId, StokDetay.Miktar, StokDetay.BirimFiyat, StokDetay.ReferansStokDetayId, StokDetay.Notu,
     Kullanici.Adi, StokDetay.DegisimTarihi, StokDetay.Yatirim, StokDetay.UretimeCikisTarihi, StokDetay.Lot, StokDetay.HGIsEmriKayitID, Model.ModelID, StokDetay.HGSiparisKayitId, HGSiparis.HGSiparisYil,
               HGSiparis.HGSiparisId, StokDetay.HGSiparisDetayId, HGIsEmri.UretimPartiNo, HgIsEmri.OrmeMakinaId, HGSiparisDetay.MusteriSiparisNo from StokDetay
  left join Urun on Urun.UrunId = StokDetay.UrunId
  left join Kullanici on Kullanici.KullaniciId  = StokDetay.KullaniciId
  left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
  left join Stok on Stok.StokId = StokDetay.StokId
  left join Parti on Parti.PartiKayitId = StokDetay.PartiKayitId
  left join SatinAlmaTalep on SatinAlmaTalep.SatinAlmaTalepKayitId = StokDetay.SatinAlmaTalepKayitId
  left join HGIsEmri on HGIsEmri.HGIsEmriKayitId = StokDetay.HGIsEmriKayitId
  left join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayId = StokDetay.HGSiparisDetayId
  left join Model on Model.ModelKayitID = HGSiparisDetay.ModelKayitID
  left join HGSiparis on HGSiparis.HGSiparisKayitId = StokDetay.HGSiparisKayitId
  where StokDetay.StokId = :AStokId order by StokDetayId
  into :StokDetayId, :DepoId, :UrunId, :UrunAdi, :FPartiYil, :FPartiId, :SatinAlmaTalepYil, :SatinAlmaTalepId,
       :BirimId, :Miktar, :BirimFiyat, :ReferansStokDetayId, :Notu, :KullaniciAdi, :DegisimTarihi, :Yatirim, :UretimeCikisTarihi, :Lot,
       :HGIsEmriKayitID, :ModelID, :HGSiparisKayitId, :HGSiparisYil, :HGSiparisId, :HGSiparisDetayId, :UretimPartiNo, :OrmeMakinaId, :MusteriSiparisNo do
  begin
    Tutar = BirimFiyat * Miktar;

    PartiIdStr = '';
    if (FPartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(FPartiYil) || ') ' || EgemenIntegerToStr(FPartiId) || ' ';

    SatinAlmaTalepIdStr = '';
    if (SatinAlmaTalepYil is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId)  || ' ';

    if (Yatirim = 1) then YatirimStr = 'Yatırım';
    else YatirimStr = '';

    HGSiparisIdStr = '';
    if (HGSiparisId is not null) then HGSiparisIdStr = '(' || EgemenIntegerTostr(HGSiparisYil) || ') '  || EgemenIntegerTostr(HGSiparisId)  || ' ';
    UretimIslemleri = '';
    Select UretimIslemleriID from HGIsEmriUretimIslemiGetir(:HGIsEmriKayitID, 0) into :UretimIslemleri;

    suspend;
  end
end
