CREATE PROCEDURE TURPOMPADEGERIDUZEHESAPLA(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER,
    APARTIASAMALARIEKRANI INTEGER)
RETURNS (
    STATUS INTEGER,
    MESAJ VARCHAR(255),
    TUR DOUBLE PRECISION,
    POMPADEGERI INTEGER,
    DUZE INTEGER)
AS
declare variable MaxKilo        Numeric(12,3);
declare variable PartiSinifId   integer; /* miktarı en büyük olan parti sinif id, yani ana kumas */
declare variable HamStokDetayId integer;
declare variable Lycrali        integer;
declare variable LycraTipiId    integer;
declare variable TupFikseVar    integer;
declare variable AcikEnFikseVar integer;
declare variable ElyafCinsiId   varchar(5);
declare variable AnaKumasYikamaOncesiGramaj    Double Precision;
declare variable PartiKumasKBilgiGramaj        varchar(10);
declare variable IntPartiKumasKBilgiGramaj     integer;
declare variable Gramaj                        Double Precision;
declare variable FDuze                         integer;
declare variable FLycraliDuze                  integer;
begin
  Select max(PartiRecetesiDokumu_3.Kilo) from PartiRecetesiDokumu_3(:APartiKayitId, :AReceteId, :APartiAsamalariId) into :MaxKilo;

  Select Max(PartiRecetesiDokumu_3.PartiSinifID) from PartiRecetesiDokumu_3(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiRecetesiDokumu_3.PartiSinifID
    where PartiRecetesiDokumu_3.Kilo = :MaxKilo
    into :PartiSinifId;

  Select Ham.LycraTipiId, IplikCinsi.ElyafCinsiId, ElyafCinsi.PompaDegeri, ElyafCinsi.Tur from PartiStok
    left join Ham on Ham.HamID = PartiStok.HamId
    left join IplikCinsi on IplikCinsi.IplikCinsiID = Ham.IplikCinsiId
    left join ElyafCinsi on ElyafCinsi.ElyafCinsiID = IplikCinsi.ElyafCinsiID
    where PartiStok.PartiSinifID = :PartiSinifID
    into :LycraTipiId, :ElyafCinsiID, :PompaDegeri, :Tur;


  if (ElyafCinsiId is not null) then
  begin
    if (LycraTipiId is not null) then Lycrali = 1;
    else Lycrali = 0;

    Select max(RafDetay.HamStokDetayId) from PartiRecetesiDokumu_3(:APartiKayitId, :AReceteId, :APartiAsamalariId)
      left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiRecetesiDokumu_3.PartiSinifID
      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      where PartiRecetesiDokumu_3.Kilo = :MaxKilo
      into :HamStokDetayId;

    Select HamStokMaster.YikamaOncesiGramaj from HamStokDetay
      left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterID
      where HamStokDetay.HamStokDetayId = :HamStokDetayId
      into :AnaKumasYikamaOncesiGramaj;

    Select max(PartiKumasKBilgi.Gramaj) from PartiKumasKBilgi
      where PartiKumasKBilgi.PartiSinifID = :PartiSinifID
       into :PartiKumasKBilgiGramaj;

    if (EgemenStrToIntOlurmu(PartiKumasKBilgiGramaj) = 1) then
    begin
      IntPartiKumasKBilgiGramaj = EgemenStrToInt(PartiKumasKBilgiGramaj);
    end
    else
    begin
      IntPartiKumasKBilgiGramaj = -1;
    end

    Select count(*) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamaStok.PartiSinifID = :PartiSinifId and Asama.TupFikseAsamasimi = 1 and PartiAsamaStok.AsamayaDahil = 1
      into :TupFikseVar;

    Select count(*) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamaStok.PartiSinifID = :PartiSinifId and Asama.AcikEnFikseAsamasimi = 1 and PartiAsamaStok.AsamayaDahil = 1
      into :AcikEnFikseVar;

    if (TupFikseVar >= 1 or AcikEnFikseVar = 0) then /* tüp fikse asaması var ise veya fikse asaması hic yok ise */
    begin
      Gramaj = AnaKumasYikamaOncesiGramaj;
    end
    else Gramaj = IntPartiKumasKBilgiGramaj;

    Select max(Duze), max(LycraliDuze) from ElyafCinsiDetayi
      where ElyafCinsiDetayi.ElyafCinsiID = :ElyafCinsiID and
            :Gramaj > ElyafCinsiDetayi.BaslangicGramaj and :Gramaj <= ElyafCinsiDetayi.BitisGramaj
      into :FDuze, :FLycraliDuze;

    if (Lycrali = 0) then Duze = FDuze;
    else Duze = FLycraliDuze;

    if (APartiAsamalariEkrani = 1) then
    begin
      Update PartiAsamalari Set Tur = :Tur, PompaDegeri = :PompaDegeri, Duze = :Duze where PartiAsamalari.PartiAsamalariID = :APartiAsamalariID;
    end
    Status = 1;
  end
  else
  begin
    Status = -1;
    Mesaj = 'Elyaf Cinsi Girilmedigi icin Hesaplanamadı...';
  end

  suspend;
end
