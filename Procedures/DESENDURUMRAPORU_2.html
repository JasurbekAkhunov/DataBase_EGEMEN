CREATE PROCEDURE DESENDURUMRAPORU_2(
    ADESENID VARCHAR(20),
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    DESENID VARCHAR(20),
    DESENISTEKLERIID INTEGER,
    DESENISTEKTIPIID VARCHAR(5),
    DESENISTEKTIPIADI VARCHAR(50),
    ISTEKTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    ISTEKYAPANKULLANICIID INTEGER,
    ISTEKYAPANKULLANICIADI VARCHAR(40),
    BASLATANKULLANICIADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENKULLANICIADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    DEPARTMANID INTEGER,
    DEPARTMANADI VARCHAR(30),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    PLANLANANDESENISTASYONID INTEGER,
    PLANLANANDESENISTASYONADI VARCHAR(30),
    PLANLANANHAFTAKAYITID INTEGER,
    PLANLANANHAFTAYIL INTEGER,
    PLANLANANHAFTANO INTEGER,
    PLANLANANHAFTANOSTR VARCHAR(30),
    HAFTABASLANGICTARIH TIMESTAMP,
    HAFTABITISTARIH TIMESTAMP)
AS
begin
  For Select DesenDurumRaporu_1.DesenId, DesenIstekleri.DesenIstekleriId,  DesenIstekleri.DesenIstekTipiId, DesenIstekTipi.Adi,
             DesenIstekleri.IstekTarihi, DesenIstekleri.TerminTarihi, DesenIstekleri.IstekYapanKullaniciId, IYK.Adi, BasK.Adi, DesenIstekleri.BaslamaTarihi, BitK.Adi, DesenIstekleri.BitisTarihi,
             DesenIstekTipi.YapacakDepartmanId, Departman.Adi, DesenIstekleri.OnayDurumu, DesenIstekleri.OnayTarihi, OK.Adi,
             DesenIstekleri.PlanlananDesenIstasyonId, PDI.Adi, DesenIstekleri.PlanlananHaftaKayitId, Hafta.Yil, Hafta.HaftaNo,
             Hafta.BaslangicTarih, Hafta.BitisTarih from DesenDurumRaporu_1(:ADesenId, :ARaporTipi, :ABaslangicTarihi, :ABitisTarihi, :AFirmaId)
  left join DesenIstekleri on DesenIstekleri.DesenId = DesenDurumRaporu_1.DesenId
  Left Join Kullanici BasK on BasK.KullaniciId = DesenIstekleri.BaslatanKullaniciId
  Left Join Kullanici BitK on BitK.KullaniciId = DesenIstekleri.BitirenKullaniciId
  Left Join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiId
  Left Join Departman on Departman.DepartmanId = DesenIstekTipi.YapacakDepartmanId
  left join Kullanici IYK on IYK.KullaniciId = DesenIstekleri.IstekYapanKullaniciId
  left join Kullanici OK on OK.KullaniciId = DesenIstekleri.OnaylayanKullaniciId
  Left Join Hafta on Hafta.HaftaKayitId = DesenIstekleri.PlanlananHaftaKayitId
  Left Join DesenIstasyon PDI on PDI.DesenIstasyonId = DesenIstekleri.PlanlananDesenIstasyonId
  into :DesenId, :DesenIstekleriId, :DesenIstekTipiId, :DesenIstekTipiAdi, :IstekTarihi, :TerminTarihi, :IstekYapanKullaniciId, :IstekYapanKullaniciAdi,
       :BaslatanKullaniciAdi, :BaslamaTarihi, :BitirenKullaniciAdi, :BitisTarihi, :DepartmanId, :DepartmanAdi, :OnayDurumu,
       :OnayTarihi, :OnaylayanKullaniciAdi, :PlanlananDesenIstasyonId, :PlanlananDesenIstasyonAdi, :PlanlananHaftaKayitId,
       :PlanlananHaftaYil, :PlanlananHaftaNo, :HaftaBaslangicTarih, :HaftaBitisTarih do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    if (PlanlananHaftaNo is not null) then PlanlananHaftaNoStr = '(' || EgemenIntegerToStr(PlanlananHaftaYil) || ') '  || EgemenIntegerToStr(PlanlananHaftaNo);
    else PlanlananHaftaNoStr = '';

    suspend;
  end
end
