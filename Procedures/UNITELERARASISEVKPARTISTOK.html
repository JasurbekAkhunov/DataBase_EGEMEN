CREATE PROCEDURE UNITELERARASISEVKPARTISTOK(
    AFIRMAID VARCHAR(20))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    RECETEID VARCHAR(20),
    MODELNO VARCHAR(20),
    HAMADITOPLAMI VARCHAR(255),
    TAMAMITARTILDI INTEGER,
    PARCALISEVK INTEGER,
    BRUTKILO DOUBLE PRECISION,
    NETKILO DOUBLE PRECISION,
    SEVKEDILENKILO DOUBLE PRECISION,
    SEVKEHAZIRKILO DOUBLE PRECISION,
    ISLEMTIPISTR VARCHAR(20),
    ISLEMGRUBUSTR VARCHAR(20),
    TERMINTARIHI TIMESTAMP,
    SEVKEHAZIRTAHMINIBRUTKILO DOUBLE PRECISION)
AS
declare variable NetinBruteOrani Double Precision;
declare variable BrutSevkKilo Double Precision;
begin
  Select NetinBruteOrani from Sistem Where SistemID = 1 into :NetinBruteOrani;

  for select Partistok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, Parti.ReceteId, Parti.ModelNo, PartiStok.HamAdiToplami, PartiStok.TamamiTartildi,
             Parti.ParcaliSevk, PartiStok.Kilo, PartiStok.NetKilo, PartiStok.BrutSevkKilo, PartiStok.NetSevkKilo, (PartiStok.NetKilo - PartiStok.NetSevkKilo),
             IslemTipi.Adi, IslemGrubu.Adi, Parti.TerminTarihi from Parti
  left join Partistok on Partistok.PartiKayitId = Parti.PartiKayitId
  left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  where Parti.FirmaId = :AFirmaId and (PartiStok.Durumu = 7 or Parti.ParcaliSevk = 1) and Partistok.netkilo > Partistok.netsevkkilo
    and (Parti.ParcaliSevk = 1 or (Parti.ParcaliSevk = 0 and Partistok.TamamiTartildi = 1))
    order by Parti.PartiYil, Parti.PartiId
  into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :ReceteId, :ModelNo, :HamAdiToplami, :TamamiTartildi, :ParcaliSevk, :BrutKilo, :NetKilo, :BrutSevkKilo, :SevkEdilenKilo, :SevkeHazirKilo,
       :IslemTipiStr, :IslemGrubuStr, :TerminTarihi do
  begin

    SevkeHazirTahminiBrutKilo = 0;

    if (ParcaliSevk = 1) then
    begin
      SevkeHazirTahminiBrutKilo = SevkeHazirKilo + SevkeHazirKilo * NetinBruteOrani / 100;

      if (BrutSevkKilo is null) then BrutSevkKilo = 0;

      if ((BrutSevkKilo + SevkeHazirTahminiBrutKilo) > BrutKilo or TamamiTartildi = 1)
         then SevkeHazirTahminiBrutKilo = BrutKilo - BrutSevkKilo;
    end

    suspend;
  end
end
