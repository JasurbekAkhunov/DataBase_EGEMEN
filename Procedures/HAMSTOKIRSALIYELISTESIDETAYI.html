CREATE PROCEDURE HAMSTOKIRSALIYELISTESIDETAYI(
    AHAMSTOKHAREKETTIPIID INTEGER,
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AORMECIID INTEGER,
    AFASON INTEGER,
    ALOT VARCHAR(150),
    AHAMSTOKMASTERID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AREFERANSFIRMAID VARCHAR(20),
    AEVRAKTIPI INTEGER)
RETURNS (
    HAMSTOKIRSALIYEID INTEGER,
    HAMSTOKMASTERID INTEGER,
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    PUS INTEGER,
    FEINE INTEGER,
    ORMECIMAKINANO VARCHAR(10),
    EN INTEGER,
    GRM2 INTEGER,
    RENK VARCHAR(20),
    TUPACIKENMAYLISTR VARCHAR(10),
    LOT VARCHAR(150),
    DENYE INTEGER,
    LYCRAURETICIADI VARCHAR(20),
    IPLIKUZUNLUGU1 VARCHAR(40),
    IPLIKUZUNLUGU2 VARCHAR(40),
    IPLIKUZUNLUGU3 VARCHAR(40),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    ACIKLAMA BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TAMIR INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    ESKIPARTINO INTEGER,
    MUSTERIPARTINO VARCHAR(30),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    REFERANSFIRMAID VARCHAR(20),
    REFERANSFIRMAADI VARCHAR(100),
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    RAFLANMAYANMIKTAR NUMERIC(12,3),
    HAMSTOKIADESEBEBIID INTEGER,
    HAMSTOKIADESEBEBIADI VARCHAR(30),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(4),
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP,
    HAMSTOKFATURAID INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable Miktar  numeric(12,3);
declare variable TupAcikEnMayli  integer;
begin
  For Select HamStokIrsaliyeListesi.HamStokIrsaliyeId, HamStokMaster.HamStokMasterId, Ham.HamTipiId, Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine,
             HamStokMaster.OrmeciMakinaNo, HamStokMaster.En, HamStokMaster.Grm2,  HamStokMaster.Renk, HamStokMaster.Lot,
             HamStokMaster.TupAcikEnMayli, HamStokMaster.Denye, LycraUretici.Adi, HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2,
             HamStokMaster.IplikUzunlugu3, HamStokMaster.KapAdedi, HamStokMaster.Miktar, (HamStokMaster.Miktar - HamStokMaster.RaflananMiktar),
             HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Adet, HamStokMaster.Aciklama,  HamStokMaster.Tamir, TamirTipi.Adi,
             HamStokMaster.EskiPartino, HamStokMaster.MusteriPartiNo, HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo,
              HamStokMaster.ReferansFirmaId, ReferansFirma.Adi, IK.Adi, HamStokMaster.InsertTarihi, Sk.Adi, HamStokMaster.DegisimTarihi,
              VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             VerilenSiparisDetay.BoyaSiparisDetayId, HamStokMaster.OnayDurumu, HamStokMaster.OnaylayanKullaniciId, OK.Adi, HamStokMaster.OnayTarihi,
             HamStokMaster.HamStokFaturaId, HamStokMaster.HamStokIadeSebebiId,
             HamStokIadeSebebi.Adi from HamStokIrsaliyeListesi(:AHamStokHareketTipiId, :AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :AIrsaliyeNo, :AOrmeciId, :AFason, :ALot, :AHamStokMasterId, :AMusteriSiparisNo, :AReferansSiparisNo, :AReferansFirmaId, :AEvrakTipi)
  left join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliyeListesi.HamStokIrsaliyeId
  left join Ham on Ham.HamId = HamStokMaster.HamId
  left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
  left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  left join LycraUretici on LycraUretici.LycraUreticiId = HamStokMaster.LycraUreticiId
  left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = HamStokMaster.VerilenSiparisDetayId
  left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = HamStokMaster.BoyaSiparisKayitId
  left join Firma ReferansFirma on ReferansFirma.FirmaId = HamStokMaster.ReferansFirmaId
  left join HamStokIadeSebebi on HamStokIadeSebebi.HamStokIadeSebebiId = HamStokMaster.HamStokIadeSebebiId
  Left Join Kullanici OK on OK.KullaniciId = HamStokMaster.OnaylayanKullaniciId
  Left Join Kullanici IK on IK.KullaniciId = HamStokMaster.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HamStokMaster.KullaniciId
  where HamStokMaster.HamStokMasterId is not null order by HamStokMaster.HamStokMasterId
  into :HamStokIrsaliyeId, :HamStokMasterId, :HamTipiId, :HamAdi, :Pus, :Feine, :OrmeciMakinaNo, :En, :Grm2, :Renk, :Lot, :TupAcikEnMayli, :Denye, :LycraUreticiAdi,
       :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3,
       :KapAdedi, :Miktar, :RaflanmayanMiktar, :Kilo, :Metre, :Adet, :Aciklama, :Tamir, :TamirTipiAdi, :EskiPartino, :MusteriPartiNo,
       :MusteriSiparisNo, :ReferansSiparisNo, :ReferansFirmaId, :ReferansFirmaAdi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :VerilenSiparisYil, :VerilenSiparisId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
       :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi, :HamStokFaturaId, :HamStokIadeSebebiId, :HamStokIadeSebebiAdi do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Okey';
    else OnayDurumuStr = '';

    Select TupAcikEnMayliStrGetir.tupacikenmaylistr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    VerilenSiparisIdStr = '';
    if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    suspend;
  end
end
