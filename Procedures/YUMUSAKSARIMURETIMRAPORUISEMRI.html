CREATE PROCEDURE YUMUSAKSARIMURETIMRAPORUISEMRI
RETURNS (
    IPLIKMAKINAID VARCHAR(10),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    KALINLIKID INTEGER,
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIID INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    LOT VARCHAR(30),
    ESKILOT VARCHAR(30),
    TAMIR INTEGER,
    KONIKCINSIADI VARCHAR(20),
    KONIKRENGIADI VARCHAR(20),
    ISEMRIMIKTARI DOUBLE PRECISION)
AS
begin
  For Select SarimIsEmri.IplikMakinaId, IplikLot.HamId, Ham.Adi, IplikLot.Lot, IplikLot.EskiLot, IplikLot.Tamir, KonikCinsi.Adi, KonikRengi.Adi,
             Ham.KalinlikId, Kalinlik.Adi, Ham.IplikCinsiId, IplikCinsi.Adi, SarimIsemri.Miktar from SarimIsEmri
  left join IplikLot on IplikLot.IplikLotId = SarimIsEmri.IplikLotId
  left join Ham on Ham.Hamid = IplikLot.HamId
  left join KonikRengi on KonikRengi.KonikRengiId = IplikLot.KonikRengiId
  left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikLot.KonikCinsiId
  left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
  left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
  where SarimIsEmri.BitisTarihi is null and SarimIsEmri.SarimSekliID = 1
  order by SarimIsEmri.IplikMakinaId, Kalinlik.Adi, IplikCinsi.Adi, IplikLot.Lot, KonikCinsi.Adi, KonikRengi.Adi
  into :IplikMakinaId, :HamId, :HamAdi, :Lot, :EskiLot, :Tamir,  :KonikCinsiAdi, :KonikRengiAdi,
       :KalinlikId, :KalinlikAdi, :IplikCinsiId, :IplikCinsiAdi, :IsEmriMiktari do
  begin
    suspend;
  end
end
