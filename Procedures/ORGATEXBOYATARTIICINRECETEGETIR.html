CREATE PROCEDURE ORGATEXBOYATARTIICINRECETEGETIR(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER)
RETURNS (
    SIRANO INTEGER,
    URUNID VARCHAR(20),
    DOZAJLAMAURUNID VARCHAR(15),
    DOZAJLAMAPROGRAMNO INTEGER,
    URUNTIPIID INTEGER,
    MIKTAR DOUBLE PRECISION,
    BANYO DOUBLE PRECISION,
    OTOMASYONMIKTAR DOUBLE PRECISION,
    BDRCTP VARCHAR(3),
    BDSIRA INTEGER,
    BDAADM INTEGER,
    BDPANO INTEGER,
    BDISTNO INTEGER,
    BDGINO INTEGER)
AS
declare variable ReceteAsamaId         integer;
declare variable ReceteAltAsamaId      integer;
declare variable LastUrunTipiId integer;
declare variable LastReceteAsamaId integer;
declare variable LastReceteAltAsamaId integer;
declare variable OncekiSatirReceteAsamaId integer;
declare variable LastBDSira  integer;
declare variable KimyasalBDAADM integer;
declare variable BoyaBDAADM     integer;
begin
  LastUrunTipiId = -1;
  LastReceteAsamaId = -2;
  OncekiSatirReceteAsamaId = -2;
  LastBdSira = -1;
  BDSIRA = 0;
  BDAADM = 0;
  BDPANO = 0;
  BDISTNO = 0;
  BDGINO = 0;
  SiraNo = 0;

  for select DozajlamaIcinReceteHazirla.SiraNo, DozajlamaIcinReceteHazirla.ReceteAsamaId, DozajlamaIcinReceteHazirla.ReceteAltAsamaId, DozajlamaIcinReceteHazirla.UrunId, DozajlamaIcinReceteHazirla.DozajlamaUrunId,
             DozajlamaIcinReceteHazirla.UrunTipiId, DozajlamaIcinReceteHazirla.Miktar, DozajlamaIcinReceteHazirla.Banyo,
             DozajlamaIcinReceteHazirla.DozajlamaProgramNo from DozajlamaIcinReceteHazirla(:APartiKayitId, :AReceteId, :APartiAsamalariId, 3)
    where UrunID is not null
    into :SiraNo, :ReceteAsamaId, :ReceteAltAsamaId, :UrunId, :DozajlamaUrunId, :UrunTipiId, :Miktar, :Banyo, :DozajlamaProgramNo do
  begin
    if (ReceteAsamaId <> LastReceteAsamaID) then
    begin
      LastReceteAsamaId = ReceteAsamaId;
      LastReceteAltAsamaId = -2;
      BDSIRA = BDSIRA + 1;
      BDISTNO = 0; /* her recete asamasında birden baslıyor */
      BDPANO = 0;
      BDAADM = 0;
      LastUrunTipiId = -1;
      KimyasalBDAADM = 0;
      BoyaBDAADM = 0;
    end

    if (ReceteAltAsamaId <> LastReceteAltAsamaId) then
    begin
      LastUrunTipiId = -1;
      BDPANO = 0;

      BDISTNO = BDISTNO + 1;
      LastReceteAltAsamaId = ReceteAltAsamaId;
    end

    if (UrunTipiID = 0) then BDRCTP = 'BOY'; else BDRCTP = 'KIM';

    if (LastUrunTipiId <> UrunTipiId) then
    begin
      if (UrunTipiID = 0) then
      begin
        BoyaBDAADM = BoyaBDAADM + 1;
        BDAADM = BoyaBDAADM;
        BDPANO = 0;
        if (LastUrunTipiId <> -1) then BDISTNO = BDISTNO + 1;
      end
      else
      begin
        KimyasalBDAADM = KimyasalBDAADM + 1;
        BDAADM = KimyasalBDAADM;
        BDPANO = 0;
        if (LastUrunTipiId <> -1) then BDISTNO = BDISTNO + 1;
      end
    end

    if (LastBDSira <> BDSira or LastUrunTipiId <> UrunTipiId) then BDGINO = BDGINO + 1; /* BDsira veya urun tipi degisince 1 artıyor */
    BDPANO = BDPANO + 1;

    LastUrunTipiId = UrunTipiId;
    LastBDSira = BDSira;
    if (Miktar > 0) then OtomasyonMiktar = Miktar;
    else OtomasyonMiktar = Banyo;

    SiraNo = SiraNo + 1;
    suspend;
  end
end
