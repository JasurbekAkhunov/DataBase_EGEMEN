CREATE PROCEDURE MC_BOYAIRSALIYERAPORUROW_2(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1),
    ASATISTIPI INTEGER,
    ARAPORTIPI INTEGER,
    AISLEMTIPI INTEGER,
    ARAPORTARIHTIPI INTEGER,
    AKULLANICIID INTEGER,
    ABOYAIRSALIYEID INTEGER)
RETURNS (
    BOYAIRSALIYEID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIFASONSTR VARCHAR(5),
    ISLEMTIPIADI VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    HAMADITOPLAMI VARCHAR(255),
    PARTISINIFID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    KAPADEDI INTEGER,
    BRUTKILO NUMERIC(12,3),
    KILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    SIPARISPROSESADI VARCHAR(255),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    LABRECETEKODU VARCHAR(20),
    RENKADI VARCHAR(40),
    ILAVEISLEMLER VARCHAR(255))
AS
declare variable FHamAdi            varchar(255);
declare variable HamId              integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable FasonHamIade         integer;
declare variable PartiFason           integer;
declare variable Okey                 integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  if (ARaporTarihTipi = 0) then
  begin
    for select BoyaIrsaliyeDetay.BoyaIrsaliyeId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliyeDetay.PartiKayitId, Parti.PartiYil,
               Parti.PartiId, Parti.Fason, IslemTipi.Adi, Parti.ReceteId, Recete.Adi, PartiStok.HamAdiToplami, Ham.Adi, BoyaIrsaliyeDetay.PartiSinifId,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId,
               BoyaIrsaliyeDetay.KapAdedi, BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.Kilo, BoyaIrsaliyeDetay.Adet,
               BoyaIrsaliyeDetay.BrutMetre, BoyaIrsaliyeDetay.Metre, BoyaIrsaliyeDetay.HamId, BoyaIrsaliyeDetay.FasonHamIade,
               BoyaSiparisDetay.MusteriSiparisNo, SiparisProses.Adi, BoyaSiparisDetay.MusteriRenkKoduveAdi,
               BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.IlaveIslemler from BoyaIrsaliye
      left join BoyaIrsaliyeDetay On BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
      left join Parti On Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
      left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
      left join Recete On Recete.ReceteId = Parti.ReceteId
      left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      left join PartiStok On PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
      left join Ham On (Ham.HamId = BoyaIrsaliyeDetay.HamId or Ham.HamId = PartiStok.HamId)
      where (BoyaIrsaliye.BoyaIrsaliyeId = :ABoyaIrsaliyeID or :ABoyaIrsaliyeID is Null) and
            BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
            (BoyaIrsaliye.FirmaId = :AFirmaId) and
            (BoyaIrsaliye.HareketTipiId = 0) and
            BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId is not null and
            (Parti.Fason = :ASatisTipi or :ASatisTipi is null) and
            ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporTipi = 1) or :ARaporTipi is null) and
            ( (:AIslemTipi = 0 and Parti.IslemTipiID <> 3) or
              (:AIslemTipi = 1 and Parti.IslemTipiID = 3) or
              (:AIslemTipi is null) )
      order by BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.BoyaIrsaliyeId
      into :BoyaIrsaliyeId, :BoyaIrsaliyeDetayId, :PartiKayitId, :PartiYil, :PartiId, :PartiFason, :IslemTipiAdi,
           :ReceteId, :ReceteAdi,  :HamAdiToplami, :FHamAdi, :PartiSinifId,
           :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
           :KapAdedi, :BrutKilo, :Kilo, :Adet, :BrutMetre, :Metre,
           :HamId, :FasonHamIade, :MusteriSiparisNo, :SiparisProsesAdi, :MusteriRenkKoduveAdi, :LabReceteKodu, :RenkAdi, :IlaveIslemler  do
    begin
      if (PartiYil is null) then PartiIdStr = '';
      else PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (BoyaSiparisYil is null) then BoyaSiparisIdStr = '';
      else BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      if (PartiKayitId is not null) then
      begin
        if (PartiFason = 1) then PartiFasonStr = 'Fason';
        PartiFasonStr = '';
        if (PartiFason = 0) then
        begin
          BrutKilo = 0;
          BrutMetre = 0;
        end
      end
      else
      begin
        PartiFasonStr = '';
        HamAdiToplami = FHamAdi;
        if (FasonHamIade = 1) then PartiIdStr = 'Fason Ham İade';
        else PartiIdStr = 'Ham Stok Satış';
      end

      suspend;
    end
  end
  else
  begin
    ABitisTarihi = ABitisTarihi + 1;
    for select BoyaIrsaliyeDetay.BoyaIrsaliyeId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliyeDetay.PartiKayitId, Parti.PartiYil,
               Parti.PartiId, Parti.Fason, IslemTipi.Adi, Parti.ReceteId, Recete.Adi, PartiStok.HamAdiToplami, Ham.Adi, BoyaIrsaliyeDetay.PartiSinifId,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId,
               BoyaIrsaliyeDetay.KapAdedi, BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.Kilo, BoyaIrsaliyeDetay.Adet,
               BoyaIrsaliyeDetay.BrutMetre, BoyaIrsaliyeDetay.Metre, BoyaIrsaliyeDetay.HamId, BoyaIrsaliyeDetay.FasonHamIade,
               BoyaSiparisDetay.MusteriSiparisNo, SiparisProses.Adi, BoyaSiparisDetay.MusteriRenkKoduveAdi,
               BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.IlaveIslemler from BoyaIrsaliye
      left join BoyaIrsaliyeDetay On BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
      left join Parti On Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
      left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
      left join Recete On Recete.ReceteId = Parti.ReceteId
      left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      left join PartiStok On PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join Ham On (Ham.HamId = BoyaIrsaliyeDetay.HamId or Ham.HamId = PartiStok.HamId)
      left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
      where (BoyaIrsaliye.BoyaIrsaliyeId = :ABoyaIrsaliyeID or :ABoyaIrsaliyeID is Null) and
             BoyaIrsaliye.SevkTarihi >= :ABaslangicTarihi and BoyaIrsaliye.SevkTarihi < :ABitisTarihi and
            (BoyaIrsaliye.FirmaId = :AFirmaId) and
            (BoyaIrsaliye.HareketTipiId = 0) and
            BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId is not null and
            (Parti.Fason = :ASatisTipi or :ASatisTipi is null) and
            ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporTipi = 1) or :ARaporTipi is null) and
            ( (:AIslemTipi = 0 and Parti.IslemTipiID <> 3) or
              (:AIslemTipi = 1 and Parti.IslemTipiID = 3) or
              (:AIslemTipi is null) )
      order by BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.BoyaIrsaliyeId
      into :BoyaIrsaliyeId, :BoyaIrsaliyeDetayId, :PartiKayitId, :PartiYil, :PartiId, :PartiFason, :IslemTipiAdi,
           :ReceteId, :ReceteAdi,  :HamAdiToplami, :FHamAdi, :PartiSinifId,
           :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
           :KapAdedi, :BrutKilo, :Kilo, :Adet, :BrutMetre, :Metre,
           :HamId, :FasonHamIade, :MusteriSiparisNo, :SiparisProsesAdi, :MusteriRenkKoduveAdi, :LabReceteKodu, :RenkAdi, :IlaveIslemler do
    begin
      if (PartiYil is null) then PartiIdStr = '';
      else PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (BoyaSiparisYil is null) then BoyaSiparisIdStr = '';
      else BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';


      if (PartiKayitId is not null) then
      begin
        if (PartiFason = 1) then PartiFasonStr = 'Fason';
        PartiFasonStr = '';

        if (PartiFason = 0) then
        begin
          BrutKilo = 0;
          BrutMetre = 0;
        end

      end
      else
      begin
        PartiFasonStr = '';
        HamAdiToplami = FHamAdi;
        if (FasonHamIade = 1) then PartiIdStr = 'Fason Ham İade';
        else PartiIdStr = 'Ham Stok Satış';
      end
      suspend;
    end
  end
end
