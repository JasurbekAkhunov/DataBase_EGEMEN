CREATE PROCEDURE DESENDURUMRAPORU_3(
    ADESENID VARCHAR(20),
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    DESENID VARCHAR(20),
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTISINIFID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    DURUMUSTR VARCHAR(20),
    MUSTERISIPARISNO VARCHAR(30),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    VARYANTADI VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    SONASAMAID INTEGER,
    SONASAMAADI VARCHAR(40),
    SONASAMAGIRISTARIHI TIMESTAMP,
    SONASAMACIKISTARIHI TIMESTAMP)
AS
begin
  for select DesenDurumRaporu_1.DesenId, Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi,
         PartiStok.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.Metre, PartiDurum.Adi, BoyaSiparisDetay.MusteriSiparisNo,
         PartiStok.BoyaSiparisDetayId, Parti.receteId, Recete.Adi, DesenVaryant.VaryantAdi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.SiparisTarihi,
         Asama.AsamaId, Asama.Adi, PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi from DesenDurumRaporu_1(:ADesenId, :ARaporTipi, :ABaslangicTarihi, :ABitisTarihi, :AFirmaId)
  left join DesenVaryant on DesenVaryant.DesenId = DesenDurumRaporu_1.DesenId
  left join Parti on Parti.DesenVaryantID = DesenVaryant.DesenVaryantId
  left join Firma on Firma.FirmaId = Parti.FirmaId
  left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  left join PartiStok on PartiStok.PartiKayitId = Parti.PartiKayitId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
  left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiStok.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiStok.PartiSinifId
  left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiStok.PartiAsamalariId
  Left Join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
  where Parti.PartiKayitID is not null and PartiStok.PartiStokIptalKayitID is null and Parti.IslemTipiID = 5
  into :DesenId, :PartiKayitId, :PartiYil, :PartiId, :AcilisTarihi, :PartigrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi,
       :PartiSinifId, :HamAdiToplami, :Kilo, :Metre, :DurumuStr, :MusteriSiparisNo, :BoyaSiparisDetayId, :receteId, :ReceteAdi,
       :VaryantAdi, :BoyaSiparisYil, :BoyaSiparisId, :SiparisTarihi, :SonAsamaId, :SonAsamaAdi, :SonAsamaGirisTarihi, :SonAsamaCikisTarihi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    suspend;
  end
end
