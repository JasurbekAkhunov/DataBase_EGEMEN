CREATE PROCEDURE ORMEURETIMECIKISRAFLISTESI(
    ADEPOID VARCHAR(10))
RETURNS (
    RESULTID VARCHAR(255),
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    URETIMPARTINO VARCHAR(30),
    RAFID VARCHAR(10),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    FASONSTOK INTEGER,
    FASONSTOKSTR VARCHAR(5),
    MIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    KAPADEDI INTEGER,
    KALANKAPADEDI INTEGER,
    KALITENO INTEGER,
    MELANJKODU VARCHAR(20),
    ORMEMAKINAID VARCHAR(10),
    KALITEACIKLAMATIPIADI VARCHAR(20),
    BOYASIPARISIDSTR VARCHAR(30),
    URETICIFIRMAID VARCHAR(20),
    URETICIFIRMAADI VARCHAR(100),
    IPLIKREZERVMIKTARI NUMERIC(12,3),
    BARKODID VARCHAR(30))
AS
declare variable CikisMiktari  Numeric(12,3);
declare variable CikisKapAdedi integer;
declare variable hstIplik       integer;
declare variable hstLycra       integer;
declare variable FBoyaSiparisYil   integer;
declare variable FBoyaSiparisId    integer;
begin
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  for select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.BarkodId, OrmeRaf.HamId, Ham.Adi, OrmeRaf.UretimPartiNo,
             OrmeRaf.RafId, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.Lot, OrmeRaf.FasonStok,
             OrmeRaf.GirisKapAdedi, OrmeRaf.NetGirisMiktari, OrmeRaf.NetCikisMiktari, OrmeRaf.CikisKapAdedi,
             OrmeRaf.KaliteNo, OrmeRaf.MelanjKodu, coalesce(OrmeRaf.OrmeMakinaId, OrmeRaf.OrmeciMakinaNo),
             KaliteAciklamaTipi.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             OrmeRaf.UreticiFirmaId, UreticiFirma.Adi from OrmeRaf
  left join Ham on Ham.HamId = OrmeRaf.hamid
  left join Firma UreticiFirma on UreticiFirma.Firmaid = OrmeRaf.UreticiFirmaId
  left join Renk on Renk.RenkId = OrmeRaf.Renkid
  Left Join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = OrmeRaf.KaliteAciklamaTipiId
  Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
  Left join DepoRaf On DepoRaf.RafID = OrmeRaf.RafId
  where OrmeRaf.StokVar = 1 and (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstLycra) and
        (DepoRaf.DepoId = :ADepoID or :ADepoID is null)
  into :OrmeRafId, :StokGirisNo, :BarkodId, :HamId, :HamAdi, :UretimPartiNo, :RafId, :RenkId, :RenkAdi, :Lot, :FasonStok,
       :KapAdedi, :Miktar, :CikisMiktari, :CikisKapAdedi, :KaliteNo, :MelanjKodu, :OrmeMakinaId,
       :KaliteAciklamaTipiAdi,  :FBoyaSiparisYil, :FBoyaSiparisId,
       :UreticiFirmaId, :UreticiFirmaAdi do
  begin
    if (FasonStok = 1) then FasonStokStr = 'Fason';
    else FasonStokStr = '';

    KalanMiktar = Miktar - CikisMiktari;
    KalanKapAdedi = KapAdedi - CikisKapAdedi;

    if (FBoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') '  || EgemenIntegerToStr(FBoyaSiparisId);
    else
      BoyaSiparisIdStr = '';


    ResultId = '"' || EgemenIntegerToStr(HamId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(OrmeRafId) || '"' ;

    Select Sum(OrmeSiparisRezerv.Miktar - OrmeSiparisRezerv.KullanilanMiktar) from OrmeSiparisRezerv where OrmeSiparisRezerv.OrmeRafID = :OrmeRafID into :IplikRezervMiktari;
    if (IplikRezervMiktari is Null) then IplikRezervMiktari = 0;

    if (KalanMiktar > 0) then suspend;
  end
end
