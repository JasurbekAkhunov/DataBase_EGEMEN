CREATE PROCEDURE REZERVICINHAMSTOKLISTESIGETIR2(
    ABOYASIPARISDETAYID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    AORMECIID INTEGER,
    AIRSALIYENO IRSALIYENOSIZE,
    ASATINALMATALEPYIL INTEGER,
    ASATINALMATALEPID INTEGER,
    ATALEPKULLANICIID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AREFERANSFIRMAID VARCHAR(20),
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    APARTILEMESIZRAFLAR INTEGER,
    ASIPARISFIRMAID VARCHAR(20),
    AYARIMAMUL INTEGER,
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    HAMSTOKDETAYID INTEGER,
    RAFDETAYID INTEGER,
    RAFID VARCHAR(10),
    KALANKILO NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    ACIKLAMA VARCHAR(30),
    KALITENO INTEGER,
    IPTALPARTIIDSTR VARCHAR(30),
    IPTALPARTIYAPILANASAMALAR VARCHAR(1024),
    IPTALNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable PartiStokIptalKayitId integer;
declare variable PartiSinifID          integer;
declare variable PartiYil              integer;
declare variable PartiId               integer;
begin
  For Select RezervIcinHamStokListesiGetir.HamStokDetayId, RafDetay.RafDetayId, RafDetay.RafId, RafDetay.KaliteNo, RafDetay.PartiStokIptalKayitId,
            (RafDetay.Miktar + RafDetay.AktarmadanIadeEdienKilo - RafDetay.IadeEdilecekMiktar - RafDetay.IadeEdilenMiktar - RafDetay.AktarmayaCikanKilo - RafDetay.PartilenenMiktar - RafDetay.RafTransferMiktar - RafDetay.SatisMiktar),
            (RafDetay.Kilo + RafDetay.AktarmadanIadeEdienKilo - RafDetay.IadeEdilecekKilo - RafDetay.IadeEdilenKilo - RafDetay.AktarmayaCikanKilo - RafDetay.PartilenenKilo - RafDetay.RafTransferKilo - RafDetay.SatisKilo)
            from RezervIcinHamStokListesiGetir(:ABoyaSiparisDetayId, :ABaslangicTarihi, :AOrmeciId, :AIrsaliyeNo, :ASatinAlmaTalepYil, :ASatinAlmaTalepId, :ATalepKullaniciId, :AMusteriSiparisNo, :AReferansSiparisNo, :AReferansFirmaId, :ABoyaSiparisYil, :ABoyaSiparisId, :APartilemesizRaflar, :ASiparisFirmaId, :AYariMamul, :AHamStokMasterId)
    join RafDetay on RafDetay.HamStokDetayID = RezervIcinHamStokListesiGetir.HamStokDetayID and RafDetay.StokVar = 1
    join Raf on Raf.RafID = RafDetay.RafID
    where (:APartilemesizRaflar = 1 or Raf.PartilemeYapilabilir = 1)
    into :HamStokDetayId, :RafDetayId, :RafId, :KaliteNo, :PartiStokIptalKayitId, :KalanKilo, :KalanMiktar do
  begin
    IptalPartiIdStr              = Null;
    IptalPartiYapilanAsamalar    = Null;
    IptalNotu                    = Null;

    if (PartiStokIptalKayitId is not null) then
    begin
      Aciklama = 'Parti İptal';

      Select PartiStokIptal.IptalNotu, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId from PartiStokIptal
      left join PartiStok on PartiStok.PartiStokIptalKayitId = PartiStokIptal.PartiStokIptalKayitId
      left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      where PartiStokIptal.PartiStokIptalKayitId = :PartiStokIptalKayitId
      into :IptalNotu, :PartiYil, :PartiId, :PartiSinifId;

      IptalPartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      Select PartiStokYapilanAsamalari from PartiSinifAsamalariGetir(:PartiSinifId) into :IptalPartiYapilanAsamalar;
    end
    else Aciklama = '';

    suspend;
  end
end
