CREATE PROCEDURE RECETEDETAYIGETIR(
    ARECETEID VARCHAR(20))
RETURNS (
    RECETEDETAYID INTEGER,
    SIRA INTEGER,
    RECETEID VARCHAR(20),
    RECETEASAMAID INTEGER,
    RECETEASAMAADI VARCHAR(40),
    RECETEASAMASABLONID INTEGER,
    RECETEASAMASABLONADI VARCHAR(40),
    RECETEALTASAMAID INTEGER,
    RECETEALTASAMAADI VARCHAR(40),
    RECETEGRAFIKID VARCHAR(10),
    RECETEGRAFIKADI VARCHAR(40),
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNTIPIID INTEGER,
    URUNTIPIADI VARCHAR(20),
    BOYATIPIADI VARCHAR(20),
    URUNBIRIMID INTEGER,
    BIRIMID VARCHAR(2),
    MIKTAR DOUBLE PRECISION,
    BANYO DOUBLE PRECISION,
    ATISRENKSIRA INTEGER,
    ATISRENKCOLOR INTEGER,
    ATISRENKKODU VARCHAR(50),
    ATISRENKADI VARCHAR(120),
    ACIKLAMA VARCHAR(255),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select ReceteDetay.ReceteId, ReceteDetay.ReceteDetayId, ReceteDetay.Sira, ReceteDetay.ReceteAsamaId, ReceteAsama.Adi,
             ReceteDetay.ReceteAsamaSablonId, ReceteAsamaSablon.Adi,
             ReceteDetay.ReceteAltAsamaId, ReceteAltAsama.Adi, ReceteDetay.ReceteGrafikId, ReceteGrafik.Adi,
             ReceteDetay.UrunId, Urun.Adi, ReceteDetay.UrunBirimId,
             UrunBirim.BirimId, ReceteDetay.Miktar, ReceteDetay.Banyo, Urun.UrunTipiId, UrunTipi.Adi, BoyaTipi.Adi,
             IK.Adi, ReceteDetay.InsertTarihi, SK.Adi, ReceteDetay.DegisimTarihi, ReceteDetay.Aciklama,
             ReceteDetay.AtisRenkSira, ReceteDetay.AtisRenkColor, ReceteDetay.AtisRenkKodu, ReceteDetay.AtisRenkAdi from ReceteDetay
  left join Urun on Urun.UrunId = ReceteDetay.UrunId
  left join UrunBirim on UrunBirim.UrunBirimId = ReceteDetay.UrunBirimId and UrunBirim.UrunId = ReceteDetay.UrunId
  left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
  left join ReceteAltAsama on ReceteAltAsama.ReceteAltAsamaId = ReceteDetay.ReceteAltAsamaId
  left join ReceteGrafik on ReceteGrafik.ReceteGrafikId = ReceteDetay.ReceteGrafikId
  left join ReceteAsamaSablon on ReceteAsamaSablon.ReceteAsamaSablonId = ReceteDetay.ReceteAsamaSablonId
  left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
  left join BoyaTipi on BoyaTipi.BoyaTipiId = Urun.BoyaTipiId
  left join Kullanici IK on IK.KullaniciId = ReceteDetay.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId = ReceteDetay.KullaniciId
  where ReceteDetay.ReceteId = :AreceteId order by sira
  into :ReceteId, :ReceteDetayId, :Sira, :ReceteAsamaId, :ReceteAsamaAdi, :ReceteAsamaSablonId, :ReceteAsamaSablonAdi, :ReceteAltAsamaId, :ReceteAltAsamaAdi,
       :ReceteGrafikId, :ReceteGrafikAdi, :UrunId, :UrunAdi, :UrunBirimId, :BirimId, :Miktar, :Banyo, :UrunTipiId, :UrunTipiAdi, :BoyaTipiAdi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :Aciklama, :AtisRenkSira, :AtisRenkColor, :AtisRenkKodu, :AtisRenkAdi do suspend;
end
