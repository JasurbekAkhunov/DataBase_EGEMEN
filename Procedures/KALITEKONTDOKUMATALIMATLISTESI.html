CREATE PROCEDURE KALITEKONTDOKUMATALIMATLISTESI(
    ADOKUMAMAKINAID VARCHAR(10),
    AHAMID INTEGER)
RETURNS (
    RESULTID VARCHAR(255),
    DOKUMAPLANLAMAID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    URETIMPARTINO VARCHAR(30),
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    MUSTERISIPARISNO VARCHAR(50),
    BEKLEMEDE SMALLINT,
    KALITEADI ADI255)
AS
declare variable DokumaSiparisTipiId integer;
begin
  for Select DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlama.BoyaSiparisKayitId, DokumaPlanlama.OrmeSiparisDetayId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.DokumaSiparisTipiId, DokumaPlanlama.UretimPartiNo,
             OrmeSiparisDetay.HamId, Ham.Adi, DokumaPlanlama.DokumaMakinaId, DokumaMakina.Adi, OrmeSiparisDetay.MusteriSiparisNo, DokumaPlanlama.Beklemede, Kalite.KaliteAdi from DokumaPlanlama
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join Kalite on Kalite.KaliteId = OrmeSiparisDetay.KaliteId
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = DokumaPlanlama.DokumaMakinaId
  where (DokumaPlanlama.DokumaMakinaId = :ADokumaMakinaId or :ADokumaMakinaId is null) and DokumaPlanlama.FirmaId is null and DokumaPlanlama.Bitti = 0 and
        (OrmeSiparisDetay.HamId = :AHamId or :AHamId is null)
        order by Sira
  into :DokumaPlanlamaId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
       :DokumaSiparisTipiId, :UretimPartiNo, :HamId, :HamAdi, :DokumaMakinaId, :DokumaMakinaAdi, :MusteriSiparisNo, :Beklemede, :KaliteAdi do
  begin
    ResultId = '"' || EgemenIntegerToStr(DokumaPlanlamaId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisKayitId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(OrmeSiparisDetayId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisYil) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(DokumaSiparisTipiId) || '",' ;
    ResultId = ResultId || '"' || UretimPartiNo || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '"' ;

    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    suspend;
  end
end
