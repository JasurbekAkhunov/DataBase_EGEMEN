CREATE PROCEDURE IPLIKTALEPRAPORUSEVKLER(
    ARAPORTIPI INTEGER,
    ATALEPYERIID INTEGER,
    ABASLANGICIPLIKTALEPYIL INTEGER,
    ABASLANGICIPLIKTALEPID INTEGER,
    ABITISIPLIKTALEPYIL INTEGER,
    ABITISIPLIKTALEPID INTEGER,
    ABASLANGICTALEPTARIHI TIMESTAMP,
    ABITISTALEPTARIHI TIMESTAMP,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    AKAPALI INTEGER,
    ATAMIR INTEGER,
    AONAYDURUMU INTEGER,
    ATESLIMDURUMU INTEGER)
RETURNS (
    IPLIKTALEPDETAYID INTEGER,
    IPLIKIRSALIYEID INTEGER,
    HAREKETTIPI INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION,
    NETMIKTAR DOUBLE PRECISION,
    BOBINSAYISI INTEGER,
    TAMIR INTEGER,
    LOT VARCHAR(30),
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP)
AS
declare variable shtSatis integer;
begin
  Select shtSatis from IplikStokHareketTipiGetir into :shtSatis;

  For Select IplikTalepRaporu.IplikTalepDetayID, IplikIrsaliyeDetay.IplikIrsaliyeId, IplikIrsaliyeDetay.HareketTipi,
             IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.FirmaId, Firma.Adi,
             IplikIrsaliyeDetay.KapAdedi, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.NetMiktar, IplikIrsaliyeDetay.BobinSayisi, IplikIrsaliyeDetay.Tamir,
             IplikIrsaliyeDetay.Lot, IplikFatura.FaturaNo, IplikFatura.FaturaTarihi from IplikTalepRaporu(:ARaporTipi, :ATalepYeriId, :ABaslangicIplikTalepYil, :ABaslangicIplikTalepId, :ABitisIplikTalepYil, :ABitisIplikTalepId,
                               :ABaslangicTalepTarihi, :ABitisTalepTarihi, :ASiparisGrupId, :ASiparisFirmaId, :AHamId, :AKapali, :ATamir, :AOnayDurumu, :ATeslimDurumu)
         left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikTalepDetayID = IplikTalepRaporu.IplikTalepDetayID
         left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikIrsaliyeDetay.IplikIrsaliyeId
         left join IplikFatura on IplikFatura.IplikFaturaId = IplikIrsaliyeDetay.IplikFaturaId
         left join Firma on Firma.FirmaID = IplikIrsaliye.FirmaId
  where IplikIrsaliyeDetay.IplikIrsaliyeDetayId is not null and IplikIrsaliyeDetay.HareketTipi = :shtSatis
  into :IplikTalepDetayID, :IplikIrsaliyeId, :HareketTipi, :IrsaliyeNo, :IrsaliyeTarihi, :FirmaId, :FirmaAdi, :KapAdedi, :Miktar,
       :NetMiktar, :BobinSayisi, :Tamir, :Lot, :FaturaNo, :FaturaTarihi do
  begin
    suspend;
  end
end
