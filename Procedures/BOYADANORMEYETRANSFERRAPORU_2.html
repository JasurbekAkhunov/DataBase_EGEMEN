CREATE PROCEDURE BOYADANORMEYETRANSFERRAPORU_2(
    ADOKUMAHAREKETI SMALLINT)
RETURNS (
    IDSTR VARCHAR(30),
    BOYAIRSALIYEID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    IPLIKIRSALIYEID INTEGER,
    IPLIKIRSALIYEDETAYID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    BOYASIPARISDETAYID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    IPLIKTALEPIDSTR VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    KAPADEDI INTEGER,
    BRUTMIKTAR DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    FIRE DOUBLE PRECISION,
    FIREORANI DOUBLE PRECISION,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISDOVIZID VARCHAR(5))
AS
declare Variable BoyaSiparisYil integer;
declare Variable BoyaSiparisId  integer;
declare Variable IplikTalepYil integer;
declare Variable IplikTalepId  integer;
declare Variable RafDetayId  integer;
declare Variable PartiSinifId integer;
begin
  For Select BoyadanOrmeyeTransferRaporu_1.IdStr, BoyadanOrmeyeTransferRaporu_1.BoyaIrsaliyeId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Parti.BoyaSiparisKayitId, BoyaSiparisDetay.OrmeSiparisDetayId,
             OrmeSiparisDetay.SiparisSiraNo, PartiStok.BoyaSiparisDetayId, Parti.PartiYil, Parti.PartiId, BoyaIrsaliyeDetay.HamId, Ham.Adi, Renk.RenkId, Renk.Adi,
             BoyaIrsaliyeDetay.KapAdedi, BoyaIrsaliyeDetay.BrutMiktar, BoyaIrsaliyeDetay.Miktar, BoyaSiparis.FirmaId, Firma.Adi,
             BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id, BoyaIrsaliyeDetay.RafDetayId, BoyaSiparisDetay.DovizId, BoyaIrsaliyeDetay.PartiSinifId from BoyadanOrmeyeTransferRaporu_1(:ADokumaHareketi)
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyadanOrmeyeTransferRaporu_1.BoyaIrsaliyeId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
    left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join Renk on Renk.LabReceteId = Recete.LabReceteId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    where BoyadanOrmeyeTransferRaporu_1.BoyaIrsaliyeId is not null and BoyaIrsaliyeDetay.OrmeIrsaliyeID is null
    into :IdStr, :BoyaIrsaliyeId, :BoyaIrsaliyeDetayId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :BoyaSiparisDetayId,
         :PartiYil, :PartiId, :HamId, :HamAdi, :RenkId, :RenkAdi,
         :KapAdedi, :BrutMiktar, :Miktar, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisGrupId, :SiparisGrup2Id, :RafDetayId, :SiparisDovizId, :PartiSinifId do
  begin
    PartiIdStr = '';
    Lot = Null;

    if (PartiYil is not null) then
    begin
      Select Lot from PartiStokLotBilgisi(:PartiSinifID) into :Lot;

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    end
    else
    begin
      Select BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, HamStokMaster.Lot from RafDetay
        left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = HamStokMaster.BoyaSiparisKayitId
        left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
        where RafDetay.RafDetayID = :RafDetayID and HamStokMaster.OrmeSiparisDetayID is not null
      into :BoyaSiparisYil, :BoyaSiparisKayitId, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :Lot;
   end

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    Fire = BrutMiktar - Miktar;
    FireOrani = Null;

    if (BrutMiktar <> 0) then
    begin
      FireOrani = Fire / BrutMiktar;
      FireOrani = Yuvarla(FireOrani, 100);
    end
    suspend;
  end

  BoyaIrsaliyeId        = Null;
  BoyaIrsaliyeDetayId   = Null;
  IplikIrsaliyeId       = Null;
  IplikIrsaliyeDetayId  = Null;
  BoyaSiparisKayitId    = Null;
  BoyaSiparisIdStr      = Null;
  OrmeSiparisDetayId    = Null;
  SiparisSiraNo         = Null;
  BoyaSiparisDetayId    = Null;
  PartiYil              = Null;
  PartiId               = Null;
  PartiIdStr            = Null;
  HamId                 = Null;
  HamAdi                = Null;
  RenkId                = Null;
  RenkAdi               = Null;
  KapAdedi              = Null;
  BrutMiktar            = Null;
  Miktar                = Null;
  Fire                  = Null;
  FireOrani             = Null;
  SiparisGrupId         = Null;
  SiparisGrup2Id         = Null;
  SiparisFirmaId        = Null;
  SiparisFirmaAdi       = Null;
  SiparisDovizId        = Null;
  Lot                   = Null;

  For Select BoyadanOrmeyeTransferRaporu_1.IdStr, BoyadanOrmeyeTransferRaporu_1.IplikIrsaliyeId, IplikIrsaliyeDetay.IplikIrsaliyeDetayId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             IplikTalepDetay.BoyaSiparisKayitId, IplikTalepDetay.OrmeSiparisDetayId,
             OrmeSiparisDetay.SiparisSiraNo, IplikTalepDetay.BoyaSiparisDetayId, IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId,
             IplikIrsaliyeDetay.HamId, Ham.Adi, Renk.RenkId, Renk.Adi,
             IplikIrsaliyeDetay.KapAdedi, IplikIrsaliyeDetay.Miktar, IplikIrsaliyeDetay.NetMiktar, BoyaSiparis.FirmaId, Firma.Adi,
             BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id, IplikIrsaliyeDetay.Lot from BoyadanOrmeyeTransferRaporu_1(:ADokumaHareketi)
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeId = BoyadanOrmeyeTransferRaporu_1.IplikIrsaliyeId
    left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
    left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
    left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikIrsaliyeDetay.IplikTalepDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    left join Renk on Renk.RenkId = IplikIrsaliyeDetay.RenkId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    where BoyadanOrmeyeTransferRaporu_1.IplikIrsaliyeId is not null and IplikIrsaliyeDetay.OrmeIrsaliyeID is null
    into :IdStr, :IplikIrsaliyeId, :IplikIrsaliyeDetayId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisKayitId, :OrmeSiparisDetayId,
         :SiparisSiraNo, :BoyaSiparisDetayId, :IplikTalepYil, :IplikTalepId, :HamId, :HamAdi, :RenkId, :RenkAdi,
         :KapAdedi, :BrutMiktar, :Miktar, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisGrupId, :SiparisGrup2Id, :Lot do
  begin
    IplikTalepIdStr = '';

    if (IplikTalepYil is not null) then
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil)  || ') ' || EgemenIntegerToStr(IplikTalepId) || ' ';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    suspend;
  end
end
