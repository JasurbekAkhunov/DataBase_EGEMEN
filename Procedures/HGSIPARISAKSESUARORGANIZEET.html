CREATE PROCEDURE HGSIPARISAKSESUARORGANIZEET(
    AHGSIPARISDETAYID INTEGER,
    AMODELKUMASID INTEGER,
    AMODELBEDENID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    HGSIPARISDETAYID INTEGER,
    MODELKUMASID INTEGER,
    MODELBEDENID INTEGER,
    URUNID URUNKODUSIZE,
    MIKTAR NUMERIC(12,4),
    FIREORANI DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATDOVIZID DOVIZSIZE,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
Declare Variable FUrunTipiID SmallInt;

Declare Variable KullanimTipi Integer;
Declare Variable FUrunID UrunKoduSize;
Declare Variable FRenkID Integer;
Declare Variable FBedenID Integer;
Declare Variable KS Integer;
Declare Variable FModelAksesuarID Integer;
begin
  FUrunTipiID = 7; /*Aksesuar*/

  Select ModelKumas.RenkID from ModelKumas where ModelKumas.ModelKumasID = :AModelKumasID into :FRenkID;
  Select ModelBeden.BedenID from ModelBeden where ModelBeden.ModelBedenID = :AModelBedenID into :FBedenID;

  for Select ModelAksesuar.ModelAksesuarID, ModelAksesuar.KullanimTipi, ModelAksesuar.UrunID, ModelAksesuar.Miktar, ModelAksesuar.FireOrani, ModelAksesuar.BirimFiyat, ModelAksesuar.BirimFiyatDovizId, ModelAksesuar.Notu,
             ModelAksesuarDetay.KullanilacakRenkID, ModelAksesuarDetay.KullanilacakBedenID from HGSiparisDetay
        left Join ModelAksesuar on ModelAksesuar.ModelKayitId = HGSiparisDetay.ModelKayitId
        left Join ModelAksesuarDetay on ModelAksesuarDetay.ModelAksesuarID = ModelAksesuar.ModelAksesuarID and ModelAksesuar.KullanimTipi <> 4 and ((ModelAksesuarDetay.RenkAktif = 1 and ModelAksesuarDetay.ModelKumasID = :AModelKumasID) or (ModelAksesuarDetay.BedenAktif = 1 and ModelAksesuarDetay.ModelBedenID = :AModelBedenID))
      where HGSiparisDetay.HGSiparisDetayId = :AHGSiparisDetayID
      Order By KullanimTipi
  into :FModelAksesuarID, :KullanimTipi, :FUrunID, :Miktar, :FireOrani, :BirimFiyat, :BirimFiyatDovizId, :Notu, :FRenkID, :FBedenID do
  begin
    HGSiparisDetayID = null;
    ModelKumasID     = null;
    ModelBedenID     = null;
    UrunID           = Null;

    if (KullanimTipi = 1) then /*ModelAksesuarı*/
    begin
      Select Count(*) from HGSiparisDetayAksesuar
        where HGSiparisDetayAksesuar.HGSiparisDetayID = :AHGSiparisDetayID and
              HGSiparisDetayAksesuar.UrunID = :FUrunID and
              HGSiparisDetayAksesuar.ModelKumasId is Null and
              HGSiparisDetayAksesuar.ModelBedenId is Null
      into :KS;

      if (KS = 0) then
      begin
        HGSiparisDetayID = AHGSiparisDetayID;
        ModelKumasID     = null;
        ModelBedenID     = null;
        UrunID           = FUrunID;
        Suspend;
      end
      else
      begin
        HGSiparisDetayID = null;
        ModelKumasID     = null;
        ModelBedenID     = null;
        UrunID           = Null;
        Suspend;
      end
    end
    else if (KullanimTipi = 2) then /*Renk Aksesuarı*/
    begin
      Select Urun.UrunId from Urun where
        StrKarsilastir(Urun.AnaUrunId,:FUrunID) = 1 and
        IntKarsilastir(Urun.RenkID, :FRenkID) = 1 and
        Urun.Bedenid is Null
      into :UrunId;

      if (UrunId is Null) then
      begin
        Select UrunId from HGSiparisIcinUrunYarat(:FUrunID, :FRenkID, Null, :FUrunTipiID, :AKullaniciID) into :UrunId;

        if (UrunId is Null) then Exception KayitHatasi ' Ürün oluşturulamadı...';
      end

      Select Count(*) from HGSiparisDetayAksesuar
        where HGSiparisDetayAksesuar.HGSiparisDetayID = :AHGSiparisDetayID and
              HGSiparisDetayAksesuar.UrunID = :UrunID and
              HGSiparisDetayAksesuar.ModelKumasId = :AModelKumasID and
              HGSiparisDetayAksesuar.ModelBedenId is Null
      into :KS;

      if (KS = 0) then
      begin
        HGSiparisDetayID = AHGSiparisDetayID;
        ModelKumasID     = AModelKumasID;
        ModelBedenID     = null;
        Suspend;
      end
      else
      begin
        HGSiparisDetayID = null;
        ModelKumasID     = null;
        ModelBedenID     = null;
        UrunID           = Null;
        Suspend;
      end

    end
    else if (KullanimTipi = 3) then /*Beden Aksesuarı*/
    begin
      Select Urun.UrunId from Urun where
        StrKarsilastir(Urun.AnaUrunId,:FUrunID) = 1 and
        IntKarsilastir(Urun.BedenId, :FBedenID) = 1 and
        Urun.RenkId is Null
      into :UrunId;

      if (UrunId is Null) then
      begin
        Select UrunId from HGSiparisIcinUrunYarat(:FUrunID, Null, :FBedenID, :FUrunTipiID, :AKullaniciID) into :UrunId;

        if (UrunId is Null) then Exception KayitHatasi ' Ürün oluşturulamadı...';
      end

      Select Count(*) from HGSiparisDetayAksesuar
        where HGSiparisDetayAksesuar.HGSiparisDetayID = :AHGSiparisDetayID and
              HGSiparisDetayAksesuar.UrunID = :UrunID and
              HGSiparisDetayAksesuar.ModelKumasId is Null and
              HGSiparisDetayAksesuar.ModelBedenId = :AModelBedenID
      into :KS;

      if (KS = 0) then
      begin
        HGSiparisDetayID = AHGSiparisDetayID;
        ModelKumasID     = Null;
        ModelBedenID     = AModelBedenID;
        Suspend;
      end
      else
      begin
        HGSiparisDetayID = null;
        ModelKumasID     = null;
        ModelBedenID     = null;
        UrunID           = Null;
        Suspend;
      end

    end
    else if (KullanimTipi = 4) then /*Renk + Beden Aksesuarı*/
    begin
      Select ModelAksesuarDetay.KullanilacakRenkID from ModelAksesuarDetay where ModelAksesuarDetay.ModelAksesuarID = :FModelAksesuarID and ModelAksesuarDetay.RenkAktif = 1 and ModelAksesuarDetay.ModelKumasID = :AModelKumasID into :FRenkID;
      Select ModelAksesuarDetay.KullanilacakBedenID from ModelAksesuarDetay where ModelAksesuarDetay.ModelAksesuarID = :FModelAksesuarID and ModelAksesuarDetay.BedenAktif = 1 and ModelAksesuarDetay.ModelBedenID = :AModelBedenID into :FBedenID;

      Select Urun.UrunId from Urun where
        StrKarsilastir(Urun.AnaUrunId,:FUrunID) = 1 and
        IntKarsilastir(Urun.RenkID, :FRenkID) = 1 and
        IntKarsilastir(Urun.Bedenid, :FBedenID) = 1
      into :UrunId;

      if (UrunId is Null) then
      begin
        Select UrunId from HGSiparisIcinUrunYarat(:FUrunID, :FRenkID, :FBedenID, :FUrunTipiID, :AKullaniciID) into :UrunId;

        if (UrunId is Null) then Exception KayitHatasi ' Ürün oluşturulamadı...';
      end

      Select Count(*) from HGSiparisDetayAksesuar
        where HGSiparisDetayAksesuar.HGSiparisDetayID = :AHGSiparisDetayID and
              HGSiparisDetayAksesuar.UrunID = :UrunID and
              HGSiparisDetayAksesuar.ModelKumasId = :AModelKumasID and
              HGSiparisDetayAksesuar.ModelBedenId = :AModelBedenID
      into :KS;

      if (KS = 0) then
      begin
        HGSiparisDetayID = AHGSiparisDetayID;
        ModelKumasID     = AModelKumasID;
        ModelBedenID     = AModelBedenID;
        Suspend;
      end
      else
      begin
        HGSiparisDetayID = null;
        ModelKumasID     = null;
        ModelBedenID     = null;
        UrunID           = Null;
        Suspend;
      end

    end
  end
end
