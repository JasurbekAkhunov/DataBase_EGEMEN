CREATE PROCEDURE ORMEFISLISTESIDETAYI(
    AHAREKETTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFISNO INTEGER,
    ASINIF VARCHAR(1),
    ADOKUMAHAREKETI INTEGER)
RETURNS (
    STOKHAREKETID INTEGER,
    HAMIADE INTEGER,
    HAMIADESTR VARCHAR(4),
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ORMEPLANLAMAID INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    MELANJKODU VARCHAR(20),
    MAKINAUSTUGRAMAJ INTEGER,
    DINLENMISGRAMAJ INTEGER,
    SIPPUS INTEGER,
    SIPFEINE INTEGER,
    STOKGIRISNO INTEGER,
    SIPTUPACIKENMAYLI INTEGER,
    SIPTUPACIKENMAYLISTR VARCHAR(10),
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    KDV DOUBLE PRECISION,
    TEVKIFAT INTEGER,
    TEVKIFATSTR VARCHAR(3),
    KAPADEDI DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    RAFLANANMIKTAR DOUBLE PRECISION,
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    DOKUMAMAKINAID MAKINAKODUSIZE,
    DOKUMAMAKINAADI ADI40,
    URETIMPARTINO VARCHAR(30),
    TALIMATURETIMPARTINO VARCHAR(30),
    BIRIMFIYAT DOUBLE PRECISION,
    ORMEFISID INTEGER,
    ORMEIRSALIYEID INTEGER,
    ORMEFATURAID INTEGER,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    BARKODID VARCHAR(30))
AS
declare variable FBoyaSiparisYil      integer;
declare variable FBoyaSiparisId       integer;
declare variable NeredenGirildi         integer;

declare variable TIplikBoyu1            varchar(20);
declare variable TIplikBoyu2            varchar(20);
declare variable TIplikBoyu3            varchar(20);
declare variable TIplikBoyu4            varchar(20);

declare variable uitBukum integer;
declare variable uitOrme integer;
declare variable uitBoya integer;
begin
  select uitBukum, uitOrme, uitBoya from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya;

  for select StokHareket.StokHareketId, StokHareket.StokGirisNo, StokHareket.NeredenGirildi, HamIade, StokHareket.HamId, Ham.HamTipiId, HamTipi.Adi, Ham.Adi,
             Ham.Adi2, StokHareket.MelanjKodu, OrmeSiparisDetay.MakinaUstuGramaj, OrmeSiparisDetay.DinlenmisGramaj,
             OrmeSiparisDetay.TupAcikEnMayli, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
             StokHareket.Kdv, StokHareket.Tevkifat, StokHareket.KapAdedi, StokHareket.Miktar, StokHareket.RaflananMiktar, StokHareket.OrmeMakinaId,
             OrmeMakina.Adi, StokHareket.UretimPartiNo,
             StokHareket.BirimFiyat, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             Coalesce(HGSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.MusteriSiparisNo), OrmeSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.ArtikelNo, OrmePlanlama.UretimPartiNo,
             OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeFatura.FaturaNo, OrmeFatura.FaturaTarihi,
             StokHareket.OrmeFisId, StokHareket.OrmeIrsaliyeId, StokHareket.OrmeFaturaId, IK.Adi,
             StokHareket.InsertTarihi, SK.Adi, StokHareket.DegisimTarihi,
             OrmeSiparisDetay.IplikBoyu1, OrmeSiparisDetay.IplikBoyu2, OrmeSiparisDetay.IplikBoyu3, OrmeSiparisDetay.IplikBoyu4,
             OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4,
             StokHareket.OrmePlanlamaId, StokHareket.BoyaSiparisKayitId, StokHareket.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo,
             StokHareket.BoyaSiparisDetayId, DokumaMakina.DokumaMakinaId, DokumaMakina.Adi, StokHareket.BarkodId from OrmeFis
  left join StokHareket on StokHareket.OrmeFisId = OrmeFis.OrmeFisId
  left join Ham on Ham.HamId = StokHareket.HamId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  left join OrmeMakina on OrmeMakina.OrmeMakinaId = StokHareket.OrmeMakinaId
  left join DokumaMakina on DokumaMakina.DokumaMakinaId = StokHareket.DokumaMakinaId
  left join Kullanici IK on IK.KullaniciId  = StokHareket.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = StokHareket.KullaniciId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
  left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = StokHareket.OrmePlanlamaId
  left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
  left join OrmeFatura on OrmeFatura.OrmeFaturaId = StokHareket.OrmeFaturaId
  left join HGSiparisDetay on HGSiparisDetay.HGSiparisDetayId = StokHareket.HGSiparisDetayId
  where OrmeFis.HareketTipi = :AHareketTipi And StokHareket.OrmeFisId is not null and
        ( (OrmeFis.FisTarihi >= :ABaslangicTarihi and OrmeFis.FisTarihi <= :ABitisTarihi) or (OrmeFis.FisNo = :AFisNo and :AFisNo is not null)) and
        (OrmeFis.FirmaId = :AFirmaId or :AFirmaId is null) and
        (OrmeFis.Sinif = :ASinif or :ASinif is null) and
        (OrmeFis.DokumaHareketi = :ADokumaHareketi)
  into :StokHareketId, :StokGirisNo, :NeredenGirildi, :HamIade, :HamId, :HamTipiId, :HamTipiAdi, :HamAdi, :HamAdi2, :MelanjKodu,
       :MakinaUstuGramaj, :DinlenmisGramaj, :SipTupAcikEnMayli, :SipPus, :SipFeine,
       :Kdv, :Tevkifat, :KapAdedi, :Miktar, :RaflananMiktar, :OrmeMakinaId, :OrmeMakinaAdi, :UretimPartiNo,
       :BirimFiyat, :FBoyaSiparisYil, :FBoyaSiparisId, :MusteriSiparisNo, :ReferansSiparisNo, :ArtikelNo, :TalimatUretimPartiNo,
       :Fisno, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi, :FaturaNo, :FaturaTarihi, :OrmeFisId, :OrmeIrsaliyeId, :OrmeFaturaId,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :TIplikBoyu1, :TIplikBoyu2, :TIplikBoyu3, :TIplikBoyu4,
       :OrmePlanlamaId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :BoyaSiparisDetayId, :DokumaMakinaId, :DokumaMakinaAdi, :BarkodId do
  begin

    if (Tevkifat = 1) then TevkifatStr = 'Var';
    else TevkifatStr = '';

    if (HamIade = 1) then HamIadeStr = 'Ä°ade'; else HamIadeStr = '';


    if (FBoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') ' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
    else
      BoyaSiparisIdStr = '';

    if (OrmePlanlamaId is not null) then
    begin
      IplikBoyu1 = TIplikBoyu1;
      IplikBoyu2 = TIplikBoyu2;
      IplikBoyu3 = TIplikBoyu3;
      IplikBoyu4 = TIplikBoyu4;
    end

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:SipTupAcikenMayli) into :SipTupAcikEnMayliStr;
    suspend;
  end
end
