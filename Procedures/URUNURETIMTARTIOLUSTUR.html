CREATE PROCEDURE URUNURETIMTARTIOLUSTUR(
    ATARIH TIMESTAMP,
    AURUNRECETEURUNID VARCHAR(20),
    ADEPOID VARCHAR(10),
    AMIKTAR DOUBLE PRECISION,
    AKULLANICIID INTEGER)
RETURNS (
    URUNURETIMID INTEGER)
AS
declare variable FUrunBirimId       integer;
declare variable FUrunReceteDetayId integer;
declare variable FUrunId            varchar(20);
declare variable FDepoId            varchar(10);
declare variable FUrunMiktar        double precision;
declare variable FAdi               varchar(80);
declare variable FMiktar            double precision;
declare variable Miktar             double precision;
begin

  UrunUretimId = gen_id(UrunUretim_Id_Gen, 1);

  select UrunRecete.UrunBirimId from UrunRecete where UrunRecete.UrunReceteUrunId = :AUrunReceteUrunId into :FUrunBirimId;

  insert into UrunUretim(UrunUretimId, Tarih, UrunReceteUrunId, Miktar, UrunBirimId, DepoId, KullaniciId)
                  values(:UrunUretimId, :ATarih, :AUrunReceteUrunId, :AMiktar, :FUrunBirimId, :ADepoId, :AKullaniciId);


  for select UrunReceteDetay.UrunReceteDetayId, UrunReceteDetay.UrunId, UrunReceteDetay.UrunBirimId, UrunReceteDetay.DepoId,
             UrunReceteDetay.Miktar / RUB.Carpan * UrunDefaultBirim.Carpan, Urun.Adi, UrunUretim.Miktar from UrunUretim
      left join UrunReceteDetay on UrunReceteDetay.UrunReceteUrunId = UrunUretim.UrunReceteUrunId
      left join UrunBirim RUB on RUB.urunBirimId = UrunReceteDetay.urunBirimId
      left join UrunBirim UrunDefaultBirim on UrunDefaultBirim.urunId = UrunReceteDetay.UrunId and UrunDefaultBirim.DefaultBirim = 1
      left join Urun on Urun.UrunId = UrunReceteDetay.UrunId
      left join UrunFiyat on UrunFiyat.UrunId = UrunReceteDetay.UrunId
      left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
      where UrunUretim.UrunUretimId = :UrunUretimId and UrunReceteDetay.Miktar > 0
  into :FUrunReceteDetayId, :FUrunId, :FUrunBirimId, :FDepoId, :FUrunMiktar, :FAdi, :FMiktar do
  begin
    Miktar = FUrunMiktar * FMiktar;
    insert into UrunUretimTarti (UrunUretimID, UrunId, UrunBirimId, Miktar, ReceteMiktar, PlanlananMiktar,
                                 Ilave, MiktarTipi, BirimMiktar, InsertKullaniciId, InsertTarihi)
                          values(:UrunUretimId, :FUrunId, :FUrunBirimId, 0, :FUrunMiktar, :Miktar,
                                 0,0,0,1,current_timestamp(0));

  end

  suspend;
end
