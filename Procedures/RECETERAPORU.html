CREATE PROCEDURE RECETERAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ARENKOKEY INTEGER,
    ASABLON INTEGER,
    AORTAKRECETE INTEGER,
    AUSTASAMAID INTEGER,
    ARECETETIPI INTEGER,
    ARECETETURUID INTEGER)
RETURNS (
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    USTASAMAID INTEGER,
    USTASAMAADI VARCHAR(40),
    RECETETIPI INTEGER,
    CALISMATARIHI TIMESTAMP,
    RECETETURUADI VARCHAR(15),
    RENKOKEYSTR VARCHAR(3),
    RECETEKAPATMA INTEGER,
    RECETEKAPATMASTR VARCHAR(10),
    SABLON VARCHAR(4),
    ORTAKRECETE VARCHAR(4),
    BOYANANKILO DOUBLE PRECISION,
    MIGRASYON INTEGER,
    MIGRASYONSTR VARCHAR(5),
    LABRECETEGRUPID INTEGER,
    LABRECETEGRUPADI VARCHAR(20),
    LABRENKGRUPID INTEGER,
    LABRENKGRUPADI VARCHAR(30),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI VARCHAR(20),
    ILAVE INTEGER,
    ILAVESTR VARCHAR(10),
    SOKUMUZERIBOYA INTEGER,
    SOKUMUZERIBOYASTR VARCHAR(20))
AS
declare variable FRenkOkey       integer;
declare variable FSablon         integer;
declare variable FOrtakRecete    integer;
begin
  for select Recete.ReceteId, Recete.Adi, Recete.FirmaId, Firma.Adi, Recete.UstAsamaId, Asama.Adi, Recete.ReceteTipi, Recete.CalismaTarihi,
             ReceteTuru.Adi, Recete.RenkOkey, Recete.ReceteKapatma, Recete.Sablon, Recete.OrtakRecete, Recete.Migrasyon,
             Recete.LabReceteGrupId, LabReceteGrup.Adi, Recete.LabRenkGrupId, LabRenkGrup.Adi,
             Recete.RenkDerinligiId, RenkDerinligi.Adi, Recete.Ilave, Recete.SokumUzeriBoya from recete
    left join Firma on Firma.FirmaId = Recete.FirmaId
    left join Asama on Asama.AsamaId = Recete.UstAsamaId
    left join LabReceteGrup On LabReceteGrup.LabReceteGrupId = Recete.LabReceteGrupId
    left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = Recete.LabRenkGrupId
    left join ReceteTuru On ReceteTuru.ReceteTuruId = Recete.ReceteTuruId
    left join RenkDerinligi On RenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
    where (Recete.FirmaID = :AFirmaID or :AFirmaId is null) and
          (Recete.RenkOkey = :ARenkOkey or :ARenkOkey is null) and
          (Recete.Sablon = :ASablon or :ASablon is null) and
          (Recete.OrtakRecete = :AOrtakRecete or :AOrtakRecete is null) and
          (Recete.UstAsamaId = :AUstAsamaId or :AUstAsamaId is null) and
          (Recete.ReceteTipi = :AReceteTipi or :AReceteTipi is null) and
          (Recete.ReceteTuruId = :AReceteTuruId or :AReceteTuruId is null) and
          Recete.CalismaTarihi >= :ABaslangicTarihi and
          Recete.CalismaTarihi <= :ABitisTarihi
    into :ReceteId, :ReceteAdi, :FirmaId, :FirmaAdi, :UstAsamaId, :UstAsamaAdi, :ReceteTipi, :CalismaTarihi,
          :ReceteTuruAdi, :FRenkOkey, :ReceteKapatma, :FSablon, :FOrtakRecete, :Migrasyon,
          :LabReceteGrupId, :LabReceteGrupAdi, :LabRenkGrupId, :LabRenkGrupAdi, :RenkDerinligiId, :RenkDerinligiAdi,
          :Ilave, :SokumUzeriBoya do
  begin
    if (FRenkOkey = 1) then RenkOkeyStr = 'Var';
    else RenkOkeyStr = '';

    if (FSablon = 1) then Sablon = 'Evet';
    else Sablon = '';

    if (FOrtakRecete = 1) then OrtakRecete = 'Evet';
    else OrtakRecete = '';

    if (Ilave = 1) then IlaveStr = 'İlave';
    else IlaveStr = '';

    if (SokumUzeriBoya = 1) then SokumUzeriBoyaStr = 'Söküm Üzeri Boya';
    else SokumUzeriBoyaStr = '';

    if (ReceteKapatma = 1) then ReceteKapatmaStr = 'Kapalı';
    else ReceteKapatmaStr = '';

    if (Migrasyon = 0) then MigrasyonStr = '';
    else if (Migrasyon = 1) then MigrasyonStr = 'Soğuk';
    else if (Migrasyon = 2) then MigrasyonStr = 'Sıcak';
    else MigrasyonStr = '??';

    select sum(PartiStok.Kilo) from PartiAsamalari
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and AsamayaDahil = 1
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    where PartiAsamalari.ReceteId = :ReceteId into :BoyananKilo;
    suspend;
  end
end
