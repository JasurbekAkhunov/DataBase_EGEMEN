CREATE PROCEDURE IPLIKLOTBILGISIGETIR(
    ALOT VARCHAR(30))
RETURNS (
    IPLIKLOTID INTEGER,
    LOT VARCHAR(30),
    ELYAFTIPIID INTEGER,
    TARIHI TIMESTAMP,
    HARMANKAYITID INTEGER,
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    ETIKETTIPIID INTEGER,
    ETIKETTIPIADI VARCHAR(30),
    PAKETLEMEETIKETNO INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    NOTU VARCHAR(250),
    TAMIR INTEGER,
    IPLIKTAMIRTIPIID INTEGER,
    IPLIKTAMIRTIPIADI VARCHAR(20),
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYEFIRMAID VARCHAR(20),
    IRSALIYEFIRMAADI VARCHAR(100),
    TAMIRGIRISMIKTAR NUMERIC(12,3),
    URETIMKAPADEDI INTEGER,
    URETIMBRUTMIKTAR NUMERIC(12,3),
    URETIMNETMIKTAR NUMERIC(12,3),
    SEVKEDILMISNETMIKTAR NUMERIC(12,3),
    SEVKEDILMISBRUTMIKTAR NUMERIC(12,3),
    KALANNETMIKTAR NUMERIC(12,3),
    KALANBRUTMIKTAR NUMERIC(12,3),
    SARIMURETIMMIKTARI NUMERIC(12,3),
    KAPALI INTEGER,
    IPLIKTALEPDETAYID INTEGER,
    HARMANRENKADI VARCHAR(50))
AS
begin
  Select IplikLot.IplikLotId, IplikLot.Lot, IplikLot.ElyafTipiId, IplikLot.Tarihi, IplikLot.HarmanKayitId,
         Harman.HarmanYil, Harman.HarmanId, IplikLot.HamId, Ham.Adi, IplikLot.MelanjKodu, IplikLot.FlamKodu,
         IplikLot.KonikCinsiId, KonikCinsi.Adi, IplikLot.AmbalajId, Ambalaj.Adi,
         IplikLot.EtiketTipiId, EtiketTipi.Adi, EtiketTipi.PaketlemeEtiketNo, IplikLot.Notu, IplikLot.Tamir, IplikLot.Kapali,
         IplikSatisIade.IplikTamirTipiId, IplikTamirTipi.Adi, TamirGiris.Miktar,
         IplikLot.SarimUretimMiktari, IplikSatisIade.RenkId, Renk.Adi,
         IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.FirmaId, IrsaliyeFirma.Adi, IplikLot.IplikTalepDetayID, HarmanRenk.Adi from IplikLot
    left join Ham on Ham.HamID = IplikLot.HamId
    left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitId
    left join Renk HarmanRenk on HarmanRenk.RenkId = Harman.RenkId
    left join EtiketTipi on EtiketTipi.EtiketTipiId = IplikLot.EtiketTipiId
    left join Ambalaj on Ambalaj.AmbalajId = IplikLot.AmbalajId
    left join KonikCinsi on KonikCinsi.KonikCinsiId = IplikLot.KonikCinsiId
    left join TamirGiris on TamirGiris.TamirGirisKayitID = IplikLot.TamirGirisKayitId
    left join IplikSatisIade on IplikSatisIade.IplikSatisIadeID = TamirGiris.IplikSatisIadeId
    left join IplikTamirTipi on IplikTamirTipi.IplikTamirTipiID = IplikSatisIade.IplikTamirTipiId
    left join Renk on Renk.RenkID = IplikSatisIade.RenkId
    left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikSatisIade.IplikIrsaliyeId
    left join Firma IrsaliyeFirma on IrsaliyeFirma.FirmaID = IplikIrsaliye.FirmaId
    where IplikLot.Lot = :ALot
    into :IplikLotId, :Lot, :ElyafTipiId, :Tarihi, :HarmanKayitId, :HarmanYil, :HarmanId, :HamId, :HamAdi, :MelanjKodu, :FlamKodu,
         :KonikCinsiId, :KonikCinsiAdi, :AmbalajId, :AmbalajAdi,
         :EtiketTipiId, :EtiketTipiAdi, :PaketlemeEtiketNo, :Notu, :Tamir, :Kapali,
         :IplikTamirTipiId, :IplikTamirTipiAdi, :TamirGirisMiktar, :SarimUretimMiktari, :RenkId, :RenkAdi,
         :IrsaliyeTarihi, :IrsaliyeFirmaId, :IrsaliyeFirmaAdi, :IplikTalepDetayID, :HarmanRenkAdi;

  if (IplikLotId is not null) then
  begin
    if (HarmanYil is not null) then
      HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') ' || EgemenIntegerToStr(HarmanId) || ' ';

    Select Count(*), Sum(Miktar), Sum(NetMiktar) from IplikUretim
      where IplikUretim.IplikLotID = :IplikLotID and IplikUretim.GirisTipi = 0
      into :UretimKapAdedi, :UretimBrutMiktar, :UretimNetMiktar;

    Select Sum(IplikStok.Miktar), Sum(IplikStok.NetMiktar) from IplikUretim
      left join IplikStok on IplikStok.IplikStokId = IplikUretim.IplikStokId and IplikStok.Numarator = IplikUretim.Numarator
      where IplikUretim.IplikLotID = :IplikLotID and IplikUretim.GirisTipi = 0 and IplikStok.CikisIplikIrsaliyeDetayID is not Null
      into :SevkEdilmisBrutMiktar, :SevkEdilmisNetMiktar;

    KalanNetMiktar  =  UretimNetMiktar - SevkEdilmisNetMiktar;
    KalanBrutMiktar =  UretimBrutMiktar - SevkEdilmisBrutMiktar;
    suspend;
  end
end
