CREATE PROCEDURE SIPARISURETIMPARTINOURET(
    AORMESIPARISDETAYID INTEGER,
    AMAKINAID VARCHAR(10),
    AFIRMAID VARCHAR(20),
    ABEYAZLIK INTEGER)
RETURNS (
    URETIMPARTINO VARCHAR(30),
    NUMARATOR INTEGER)
AS
declare variable BoyaSiparisYil         integer;
declare variable BoyaSiparisId          integer;
Declare variable SiparisSiraNo          integer;
Declare variable FirmaKisaAdi           varchar(10);
Declare Variable OrmeSiparisi           integer;
Declare Variable DokumaSiparisi         integer;
begin
  Select BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.OrmeSiparisi, BoyaSiparis.DokumaSiparisi from OrmeSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
    where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId
    into :BoyaSiparisYil, :BoyaSiparisId, :SiparisSiraNo, :OrmeSiparisi, :DokumaSiparisi;

  if (DokumaSiparisi = 1) then
  begin
    Select Max(Numarator) from DokumaPlanlama
      where DokumaPlanlama.OrmeSiparisDetayID = :AOrmeSiparisDetayID and
            (DokumaPlanlama.DokumaMakinaID = :AMakinaId or DokumaPlanlama.FirmaId = :AFirmaId)
      into :Numarator;
  end
  else
  begin
    Select Max(Numarator) from OrmePlanlama
      where OrmePlanlama.OrmeSiparisDetayID = :AOrmeSiparisDetayID and
            (OrmePlanlama.OrmeMakinaID = :AMakinaId or OrmePlanlama.FirmaId = :AFirmaId)
      into :Numarator;
  end

  if (Numarator is null) then Numarator = 0;
  Numarator = Numarator + 1;

  if (AMakinaId is not null) then
  begin
    UretimPartiNo = EgemenIntegerToStr(BoyaSiparisYil) || '-'  || EgemenIntegerToStr(BoyaSiparisId) || '-'  || EgemenIntegerToStr(SiparisSiraNo)  || '-'  || AMakinaId   || '-'  || EgemenIntegerToStr(Numarator);
  end
  else if (AFirmaId is not null) then
  begin
    Select Firma.KisaAdi from Firma where Firma.FirmaID = :AFirmaID into :FirmaKisaAdi;
    UretimPartiNo = EgemenIntegerToStr(BoyaSiparisYil) || '-'  || EgemenIntegerToStr(BoyaSiparisId) || '-'  || EgemenIntegerToStr(SiparisSiraNo)  || '-'  || FirmaKisaAdi || '-'  || EgemenIntegerToStr(Numarator);
  end

  if (ABeyazlik = 1) then UretimPartiNo = UretimPartiNo || '(B)';
  suspend;
end
