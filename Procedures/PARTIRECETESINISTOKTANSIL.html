CREATE PROCEDURE PARTIRECETESINISTOKTANSIL(
    APARTIKAYITID INTEGER,
    APARTIASAMALARIID INTEGER)
AS
declare variable FStokId integer;
declare variable KayitSayisi integer;
declare variable FReceteStoktanDusmeSekli integer;
begin
  select ReceteStoktanDusmeSekli from Sistem Where SistemId = 1 into :FReceteStoktanDusmeSekli;

  if (FReceteStoktanDusmeSekli = 2) then
  begin
    select count(*) from PartiAsamaStok where PartiAsamaStok.PartiKayitID = :APartiKayitId and
                                       PartiAsamaStok.PartiAsamalariId = :APartiAsamalariId and
                                       PartiAsamaStok.GirisTarihi is not null into :KayitSayisi;
  end
  else if (FReceteStoktanDusmeSekli = 3) then
  begin
    select count(*) from PartiAsamaStok where PartiAsamaStok.PartiKayitID = :APartiKayitId and
                                      PartiAsamaStok.PartiAsamalariId = :APartiAsamalariId and
                                      PartiAsamaStok.CikisTarihi is not null into :KayitSayisi;
  end
  else KayitSayisi = 0;

  if (KayitSayisi > 0) then exception ReceteStoklariIptalOlamaz;

  Select Min(StokId) from StokDetay where StokDetay.PartiKayitId = :APartiKayitId and
                                     StokDetay.PartiAsamalariId = :APartiAsamalariId and StokDetay.PartiAsamaIlaveSarfID is null into :FStokId;

  delete from StokDetay where StokDetay.PartiKayitId = :APartiKayitId and StokDetay.PartiAsamalariId = :APartiAsamalariId and StokDetay.PartiAsamaIlaveSarfID is null;

  select Count(*) from StokDetay where StokDetay.StokId = :FStokId into :KayitSayisi;

  if (KayitSayisi = 0) then delete from Stok where Stok.StokId = :FStokId;

  Update PartiAsamalari Set StoktanDusuldu = 0
   where PartiAsamalari.PartiKayitID = :APartiKayitId and PartiAsamalari.PartiAsamalariId = :APartiAsamalariID;

  delete from PartiReceteDetay where PartiReceteDetay.PartiAsamalariId = :APartiAsamalariId;
end
