            else if (KayitSayisi = 0) then KayitSayisi = 1;

          else KayitSayisi = 0;

        else if (:ARezervDurumu = 3) then /*rezervi tamamlanmayan*/
          select count(*) from BoyaRezervFisiDetay where BoyaSiparisDetayId = :BoyaSiparisDetayId and Tamamlandi = 0 into :KayitSayisi;
      end

      if (KayitSayisi <> 0) then
      begin
        KS = 1;
        if (FPartiKayitId is not null) then
        begin
          select count(*) from BoyaRezervFisiDetay
          where BoyaRezervFisiDetay.PartiKayitId = :FPartiKayitId and
                BoyaRezervFisiDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId into :KS;
        end

        if (KS > 0) then
        begin
          if (Fason = 0 and SatSiparisKalanBruttenHesapla = 1) then
            KalanKilo = SiparisBrutKilo - PartilenenKilo + IptalKilo;
          else KalanKilo = SiparisKilo - PartilenenKilo + IptalKilo;

          if ((APartilenen is Null) or (APartilenen = 0 and KalanKilo > 0) or (APartilenen = 1 and KalanKilo <= 0)) then
          begin
            MakinaKoduUygun = 1;
            if (AMakinaId is not Null) then
            begin
              select count(*) from BoyaSiparisSarj where BoyaSiparisDetayId = :BoyaSiparisDetayId and MakinaId = :AMakinaId into :MakinaKoduUygun;
              if (MakinaKoduUygun is not Null) then
              begin
                if (MakinaKoduUygun > 0) then
                  MakinaKoduUygun = 1;
                else
                  MakinaKoduUygun = 0;
              end
              else
                MakinaKoduUygun = 0;
            end

            if (MakinaKoduUygun > 0) then
            begin

              if (YurtDisi = 1) then YurtDisiStr = 'Yurt Dışı';
              else YurtDisiStr = '';

              select Sum(BoyaRezervFisiDetay.EdilecekMiktar), Sum(BoyaRezervFisiDetay.EdilecekKilo), Sum(BoyaRezervFisiDetay.EdilecekMetre),
                     Sum(BoyaRezervFisiDetay.EdilenMiktar), Sum(BoyaRezervFisiDetay.EdilenKilo), Sum(BoyaRezervFisiDetay.EdilenMetre),
                     Sum(BoyaRezervFisiDetay.EdilecekBrutMiktar), Sum(BoyaRezervFisiDetay.EdilecekBrutKilo), Sum(BoyaRezervFisiDetay.EdilenBrutMiktar), Sum(BoyaRezervFisiDetay.EdilenBrutKilo),

                     where PartiStokTarti.BoyaRezervFisiDetayId = BoyaRezervFisiDetay.BoyaRezervFisiDetayId)), Max(BoyaRezervFisi.FisNo)
              from BoyaRezervFisiDetay
              left join BoyaRezervFisi on BoyaRezervFisi.BoyaRezervFisiId = BoyaRezervFisiDetay.BoyaRezervFisiId where BoyaRezervFisiDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId into
              :SevkRezervEdilecekMiktar, :SevkRezervEdilecekKilo, :SevkRezervEdilecekMetre, :SevkRezervEdilenMiktar, :SevkRezervEdilenKilo, :SevkRezervEdilenMetre,
              :SevkRezervEdilecekBrutMiktar, :SevkRezervEdilecekBrutKilo, :SevkRezervEdilenBrutMiktar, :SevkRezervEdilenBrutKilo, :SevkRezervEdilenTartiBrutKilo, :FisNo;

              select Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.BrutKilo), Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara)
              from BoyaRezervFisiDetay
              left join PartiStokTarti on PartiStokTarti.BoyaRezervFisiDetayId = BoyaRezervFisiDetay.BoyaRezervFisiDetayId and PartiStokTarti.BoyaIrsaliyeId is not null
              where BoyaRezervFisiDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId into :RezervdenSevkKilo, :RezervdenSevkBrutKilo, :RezervdenSevkTartiBrutKilo;

              Select Count(*) from Makina where Makina.MakinaPlanlamaGrupId = :MakinaPlanlamaGrupId into :PlanlamaGrubuMakinaSayisi;

              HamStokBedeli  = 0;
              ReceteMaliyeti = 0;
              AsamaMaliyeti  = 0;

              Select BT.HamStokBedeli, BT.ReceteMaliyeti,
                    (Select Sum(BoyaTeklifAsama.BirimFiyat) from BoyaTeklifAsama where BoyaTeklifAsama.BoyaTeklifDetayId = Bt.BoyaTeklifDetayId) AsamaMaliyeti
                From BoyaTeklifDetay BT
                Where BT.BoyaTeklifDetayID = :BoyaTeklifDetayID
              into :HamStokBedeli, :ReceteMaliyeti, :AsamaMaliyeti;


              if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
              else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
              else OnayDurumuStr = '';

              if (OrmeOnayDurumu = 1) then OrmeOnayDurumuStr = 'Kabul';
              else if (OrmeOnayDurumu = 2) then OrmeOnayDurumuStr = 'Ret';
              else OrmeOnayDurumuStr = '';

              Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:BoyaTupAcikEnMayli) into :BoyaTupAcikEnMayliStr;

              KapaliStr = '';
              if (Kapali = 1) then KapaliStr = 'Kapali';
              else if (Kapali = 2) then KapaliStr = 'İptal';

              ReferansPartiIdStr = Null;

                  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
                  left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
                  left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupID
                  where PartiStok.PartiSinifId = :ReferansPartiSinifId
                  into :ReferansPartiYil, :ReferansPartiId, :FPartiAsamalariId, :ReferansPartiDurumu, :ReferansPartiDurumuStr, :ReferansPartiGrupID, :ReferansPartiGrupAdi;

                Select SonYapilanAsamaAdi from SonAsamaBilgisiGetir(:ReferansPartiSinifId, :FPartiAsamalariId) into :RefPartiSonYapilanAsamaAdi;

              OrmeyeCikanIplik   = 0;

              Orulen             = 0;

              OrgudenBoyayaSonSevkTarihi = Null;
              if (OrmeSiparisDetayId is not Null) then
              begin
                Select OrgudenBoyayaIlkSevkTarihi, OrgudenBoyayaSonSevkTarihi, OrmeyeCikanIplik, OrmedenIadeIplik, Orulen from BoyaSipRaporuSevkBilgisiGetir(:OrmeSiparisDetayId)
                  into :OrgudenBoyayaIlkSevkTarihi, :OrgudenBoyayaSonSevkTarihi, :OrmeyeCikanIplik, :OrmedenIadeIplik, :Orulen;
              end

              select Sum(Miktar), Sum(NetGelenMiktar), Sum(BrutGelenMiktar), Sum(GidenMiktar) from BoyaTalimat where BoyaSiparisDetayId = :BoyaSiparisDetayId into
              :TalimatBoyaMiktar, :TalimatBoyaGelenNetMiktar, :TalimatBoyaGelenBrutMiktar, :TalimatBoyaGidenMiktar;

              if (PartilenenMiktardanIptaliDus = 1) then
              begin
                PartilenenKilo = PartilenenKilo - IptalKilo;
                PartilenenMetre = PartilenenMetre - IptalMetre;
                PartilenenKiloBaski = PartilenenKiloBaski - IptalKiloBaski;
                PartilenenMetreBaski = PartilenenMetreBaski - IptalMetreBaski;
              end

              suspend;
            end
          end
        end
      end
