CREATE PROCEDURE HAMSTOKRAPORU
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    GECICIKABULSTOKMEVCUDU NUMERIC(12,3),
    BOYAHANESTOKMEVCUDU NUMERIC(12,3),
    MERKEZSTOKMEVCUDU NUMERIC(12,3),
    ACIKSIPARISVAR INTEGER,
    ISLETMEDEPARTIVAR INTEGER,
    VERILENSIPARISVAR INTEGER,
    ORMETALIMATIVAR INTEGER)
AS
declare variable Ks integer;
begin
  For Select Ham.HamID, Ham.Adi, Ham.HamTipiId from Ham
  order by Ham.HamTipiID, Ham.Adi
  into :HamID, :HamAdi, :HamTipiId do
  begin
    GeciciKabulStokMevcudu = 0;
    BoyahaneStokMevcudu    = 0;
    MerkezStokMevcudu      = 0;
    AcikSiparisVar         = 0;
    IsletmedePartiVar      = 0;
    VerilenSiparisVar      = 0;
    OrmeTalimatiVar        = 0;

    if (HamTipiId = 2) then
    begin
      Select Sum(GeciciKabulRaf.Metre - GeciciKabulRaf.KararMetre) from GeciciKabulRaf
        where GeciciKabulRaf.HamID = :HamID
        into :GeciciKabulStokMevcudu;

      Select Sum(RafDetay.Metre - RafDetay.IadeEdilenMetre - RafDetay.PartilenenMetre -  RafDetay.RafTransferMetre -  RafDetay.SatisMetre) from RafDetay
        where RafDetay.HamID = :HamID
        into :BoyahaneStokMevcudu;
    end
    else
    begin
      Select Sum(GeciciKabulRaf.Kilo - GeciciKabulRaf.KararKilo) from GeciciKabulRaf
        where GeciciKabulRaf.HamID = :HamID
        into :GeciciKabulStokMevcudu;

      Select Sum(RafDetay.Kilo - RafDetay.IadeEdilenKilo - RafDetay.AktarmayaCikanKilo + RafDetay.AktarmadanIadeEdienKilo - RafDetay.PartilenenKilo -  RafDetay.RafTransferKilo -  RafDetay.SatisKilo) from RafDetay
        where RafDetay.HamID = :HamID
        into :BoyahaneStokMevcudu;
    end

    Select Sum(OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari) from OrmeRaf
      where OrmeRaf.HamID = :HamID  and (OrmeRaf.MiktarBirim = 0)
      into :MerkezStokMevcudu;


    Select Count(*) from BoyaSiparisDetay where BoyaSiparisDetay.HamID = :HamID and BoyaSiparisDetay.Kapali = 0 into :KS;
    if (KS > 0) then AcikSiparisVar = 1;

    Select Count(*) from PartiStok where PartiStok.HamID = :HamID and PartiStok.PartiStokIptalKayitID is null and PartiStok.Durumu < 7 into :KS;
    if (KS > 0) then IsletmedePartiVar = 1;

    Select Count(*) from VerilenSiparisDetay where VerilenSiparisDetay.HamID = :HamID and VerilenSiparisDetay.Kapali = 0 into :KS;
    if (KS > 0) then VerilenSiparisVar = 1;

    Select Count(*) from OrmeSiparisDetay
      left join OrmePlanlama On OrmePlanlama.OrmeSiparisDetayID = OrmeSiparisDetay.OrmeSiparisDetayID
      where OrmeSiparisDetay.HamID = :HamID and OrmePlanlama.OrmePlanlamaID is not null and OrmePlanlama.Bitti = 0
      into :KS;

    if (KS > 0) then OrmeTalimatiVar = 1;
    suspend;
  end
end
