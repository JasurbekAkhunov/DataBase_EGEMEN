CREATE PROCEDURE YILLIKORMEIPLIKSTOKANALIZI_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMID INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    AY INTEGER,
    AYSTR VARCHAR(10),
    DEVREDEN DOUBLE PRECISION,
    GIRIS DOUBLE PRECISION,
    GIRISIADE DOUBLE PRECISION,
    HAMSATIS DOUBLE PRECISION,
    BOYALISATIS DOUBLE PRECISION,
    HAMORGUCIKIS DOUBLE PRECISION,
    BOYALIORGUCIKIS DOUBLE PRECISION,
    BUKUMECIKIS DOUBLE PRECISION,
    BOYAHANEYECIKIS DOUBLE PRECISION,
    DIGERCIKISLAR DOUBLE PRECISION,
    DIGERGIRISLER DOUBLE PRECISION)
AS
declare variable AyBaslangicTarihi TimeStamp;
declare variable OrmeFirmaKodumuz Varchar(20);
begin
  Select OrmeFirmaKodumuz from Sistem where sistemid = 1 into :OrmeFirmaKodumuz;

  For Select HamId, HamAdi, HamAdi2, HamAdi3, HamAdi4, Ay, AyStr, AyIlkGun, Sum(Giris), Sum(GirisIade), Sum(HamSatis), Sum(BoyaliSatis), Sum(HamOrguCikis),
             Sum(BoyaliOrguCikis), Sum(BukumeCikis), Sum(BoyahaneyeCikis), Sum(DigerCikislar), Sum(DigerGirisler) from YillikOrmeIplikStokAnalizi_1(:ABaslangicTarihi, :ABitisTarihi, :AHamId)
             group by HamId, HamAdi, HamAdi2, HamAdi3, HamAdi4, Ay, AyStr, AyIlkGun
    into :HamId, :HamAdi, :HamAdi2, :HamAdi3, :HamAdi4, :Ay, :AyStr, :AyBaslangicTarihi, :Giris, :GirisIade, :HamSatis, :BoyaliSatis, :HamOrguCikis, :BoyaliOrguCikis, :BukumeCikis,
         :BoyahaneyeCikis, :DigerCikislar, :DigerGirisler do
  begin
    Devreden = Null;

    Select sum(StokHareket.Miktar * StokHareket.GirisCikis) from StokHareket
      left join OrmeRaf on OrmeRaf.OrmeRafID = StokHareket.OrmeRafId
      where StokHareket.HamId = :HamId and StokHareket.Tarih < :AyBaslangicTarihi and
            (StokHareket.StokFirmaID = :OrmeFirmaKodumuz or OrmeRaf.StokFirmaID = :OrmeFirmaKodumuz)
      into :Devreden;

    if (Devreden is null) then Devreden = 0;
    suspend;
  end
end
