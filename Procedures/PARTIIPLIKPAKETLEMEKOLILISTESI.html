CREATE PROCEDURE PARTIIPLIKPAKETLEMEKOLILISTESI(
    APARTIKAYITID INTEGER,
    ATARTISIAKTARILMIS INTEGER)
RETURNS (
    IPLIKPAKETLEMEID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    HAMSTOKSARFFISIDETAYID INTEGER,
    MAKINAID VARCHAR(10),
    PAKETLEMEID INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    KONIKRENGIID INTEGER,
    KONIKRENGIADI VARCHAR(20),
    KALITENO INTEGER,
    KOLINO INTEGER,
    BOBINSAYISI INTEGER,
    DARA NUMERIC(12,3),
    BOBINDARASI NUMERIC(12,3),
    AMBALAJAGIRLIGI DOUBLE PRECISION,
    TOPLAMDARA NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    PAKETLEMETARIHI TIMESTAMP,
    SEVKDURUMU VARCHAR(25),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select IplikPaketleme.IplikPaketlemeId, IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, IplikPaketleme.MakinaId, IplikPaketleme.PaketlemeId,
      IplikPaketleme.KaliteNo, IplikPaketleme.KoliNo, IplikPaketleme.BobinSayisi, IplikPaketleme.Dara, IplikPaketleme.BobinDarasi, IplikPaketleme.BrutKilo, IplikPaketleme.NetKilo, IplikPaketleme.PaketlemeTarihi,
      IplikPaketleme.AmbalajId, Ambalaj.Adi, IplikPaketleme.KonikCinsiId, KonikCinsi.Adi, IplikPaketleme.KonikRengiId, KonikRengi.Adi, IplikPaketleme.AmbalajAgirligi,
      IplikPaketleme.PartiStokTartiId, PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.HamStokSarfFisiDetayId, IK.Adi, IplikPaketleme.InsertTarihi, SK.Adi, IplikPaketleme.DegisimTarihi from IplikPaketleme
  Left Join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiId
  Left Join Ambalaj on Ambalaj.AmbalajId = IplikPaketleme.AmbalajId
  Left Join KonikCinsi on KonikCinsi.KonikCinsiId = IplikPaketleme.KonikCinsiId
  Left Join KonikRengi on KonikRengi.KonikRengiId = IplikPaketleme.KonikRengiId
  Left Join Kullanici IK on IK.KullaniciId = IplikPaketleme.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = IplikPaketleme.KullaniciId
  where IplikPaketleme.PartiKayitId = :APartiKayitId and
        ( (IplikPaketleme.PartiStokTartiId is null and :ATartisiAktarilmis = 0) or
          (IplikPaketleme.PartiStokTartiId is not null and :ATartisiAktarilmis = 1))
  into :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :MakinaId, :PaketlemeId, :KaliteNo, :KoliNo, :BobinSayisi, :Dara, :BobinDarasi, :BrutKilo, :NetKilo, :PaketlemeTarihi,
       :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi, :KonikRengiId, :KonikRengiAdi, :AmbalajAgirligi,
       :PartiStokTartiId, :BoyaIrsaliyeDetayId, :HamStokSarfFisiDetayId, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    if (BoyaIrsaliyeDetayId is not null) then SevkDurumu = 'Sevk Edilmiş';
    else if (HamStokSarfFisiDetayId is not null) then SevkDurumu = 'Sarf Fişi ile Çıkış';
    else SevkDurumu = '';

    ToplamDara = Dara + BobinSayisi * BobinDarasi;
    suspend;
  end
end
