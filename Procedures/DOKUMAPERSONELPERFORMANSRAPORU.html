CREATE PROCEDURE DOKUMAPERSONELPERFORMANSRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APERSONELIDLER VARCHAR(255),
    AMAKINAGRUPID MAKINAGRUPKODUSIZE,
    ADOKUMAMAKINAIDLER VARCHAR(255))
RETURNS (
    PERSONELID INTEGER,
    DOKUMAMAKINAID MAKINAKODUSIZE,
    DOKUMAPLANLAMAID INTEGER,
    TURGEREKEN INTEGER,
    TURATILAN INTEGER,
    TURORAN DOUBLE PRECISION)
AS
declare variable SonMakinaHizi  integer;
declare variable MakinaHizi     integer;
declare variable SonPersonelId  integer;
declare variable SonDokumaMakinaId MakinaKoduSize;
declare variable IlkTarih       timestamp;
declare variable SonTarih       timestamp;
declare variable Tarih          timestamp;
declare variable Tur            integer;
declare variable SonDokumaPlanlamaId integer;

declare variable TmpPersonelId  integer;
declare variable TmpDokumaMakinaId MakinaKoduSize;
declare variable TmpDokumaPlanlamaId integer;
begin
  if (ABaslangicTarihi > current_timestamp) then exit;


  TurAtilan = 0;
  SonPersonelId = null;
  SonDokumaMakinaId = null;

  for select OrmeMakinaDurumu.PersonelId, OrmeMakinaDurumu.Tarih, Tur, OrmeMakinaDurumu.DokumaPlanlamaId, DokumaMakina.DokumaMakinaId, DokumaMakina.MakinaHizi
      from OrmeMakinaDurumu
      left join Personel on Personel.PersonelId = OrmeMakinaDurumu.PersonelId
      left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = OrmeMakinaDurumu.DokumaPlanlamaId
      left join DokumaMakina on DokumaMakina.DokumaMakinaId = OrmeMakinaDurumu.DokumaMakinaId
      where OrmeMakinaDurumu.Tarih >= :ABaslangicTarihi and OrmeMakinaDurumu.Tarih <= :ABitisTarihi and
      (:APersonelIdler is null or position('-'||OrmeMakinaDurumu.PersonelId||'-', :APersonelIdler) > 0) and
      (:ADokumaMakinaIdler is null or position('-'||OrmeMakinaDurumu.DokumaMakinaId||'-', :ADokumaMakinaIdler) > 0) and
      (:AMakinaGrupId is null or DokumaMakina.MakinaGrupId = :AMakinaGrupId)
      order by OrmeMakinaDurumu.DokumaMakinaId, OrmeMakinaDurumu.PersonelId, OrmeMakinaDurumu.DokumaPlanlamaId, OrmeMakinaDurumu.Tarih
      into :PersonelId, :Tarih, :Tur, :DokumaPlanlamaId, :DokumaMakinaId, :MakinaHizi do
  begin
    if (SonPersonelId <> PersonelId or SonDokumaMakinaId <> DokumaMakinaId or SonDokumaPlanlamaId <> DokumaPlanlamaId) then
    begin
      TmpPersonelId = PersonelId;
      TmpDokumaMakinaId = DokumaMakinaId;
      TmpDokumaPlanlamaId = DokumaPlanlamaId;

      PersonelId = SonPersonelId;
      DokumaMakinaId = SonDOkumaMakinaId;
      DokumaPlanlamaId = SonDokumaPlanlamaId;

      TurGereken = (SonTarih - IlkTarih) * 24 * 60 * SonMakinaHizi;
      if (coalesce(TurGereken,0) > 0) then
        TurOran = TurAtilan * 100 / TurGereken;
      suspend;

      PersonelId = TmpPersonelId;
      DokumaMakinaId = TmpDokumaMakinaId;
      DokumaPlanlamaId = TmpDokumaPlanlamaId;

      TurAtilan = 0;
      IlkTarih = Tarih;
    end

    if (SonDokumaMakinaId is null) then
    begin
      TurAtilan = 0;
      IlkTarih = Tarih;
    end

    TurAtilan = TurAtilan + Tur;

    SonPersonelId = PersonelId;
    SonTarih = Tarih;
    SonMakinaHizi = MakinaHizi;
    SonDokumaMakinaId = DokumaMakinaId;
    SonDokumaPlanlamaId = DokumaPlanlamaId;
  end


  if (SonDokumaMakinaId is not null) then
  begin
    if (SonTarih - IlkTarih = 0) then
      TurGereken = SonMakinaHizi;
    else
      TurGereken = (SonTarih - IlkTarih) * 24 * 60 * SonMakinaHizi;

    if (coalesce(TurGereken,0) > 0) then
      TurOran = TurAtilan * 100 / TurGereken;
  end

  suspend;
end
