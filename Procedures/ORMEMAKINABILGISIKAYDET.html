CREATE PROCEDURE ORMEMAKINABILGISIKAYDET(
    AMAKINAID VARCHAR(10),
    ATURSAYISI INTEGER,
    AMEMINGERKOPTU SMALLINT,
    AIGNEKOPTU SMALLINT,
    AKAPAKACILDI SMALLINT,
    ASTARTSTOP SMALLINT,
    ATARIH TIMESTAMP,
    AALTMEMINGERKOPTU SMALLINT,
    ALYCRAKOPTU SMALLINT)
RETURNS (
    STATUS INTEGER,
    MESAJ VARCHAR(255))
AS
declare variable OrmePlanlamaID  integer;

begin
  if (AMakinaId is Null) then
  begin
    Status = 0;
    Mesaj = 'Makina Kodu zorunludur!';
  end

  if (ATarih is Null) then ATarih = Current_TimeStamp(0);

  if (ATurSayisi is Null) then ATurSayisi = 0;

  Select Max(OrmePlanlama.OrmePlanlamaID) from OrmePlanlama
    where OrmePlanlama.OrmeMakinaId = :AMakinaId and
          OrmePlanlama.Bitti = 0 and
          OrmePlanlama.Sira = (Select Min(sira) from OrmePlanlama where OrmePlanlama.Bitti = 0 and OrmePlanlama.OrmeMakinaId = :AMakinaId )
  into :OrmePlanlamaID;

  insert into OrmeMakinaDurumu(OrmeMakinaId, OrmePlanlamaID, Tarih, Tur, MemingerKoptu, AltMemingerKoptu, LycraKoptu, IgneKoptu, KapakAcildi, StartStop)
                        values(:AMakinaId, :OrmePlanlamaID, :ATarih, :ATurSayisi, :AMemingerKoptu, :AAltMemingerKoptu, :ALycraKoptu, :AIgneKoptu, :AKapakAcildi, :AStartStop);
  Status = 1;
  Mesaj = 'Kayıt yapıldı';

  suspend;
end
