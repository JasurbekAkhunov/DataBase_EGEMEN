CREATE PROCEDURE EFATURAENTICINFATURAGETIR(
    BASLANGICTARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    GONDERILMEDURUMU INTEGER,
    EARSIVAKTIF SMALLINT = 0)
RETURNS (
    BOYAFATURAID INTEGER,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(17),
    FATURANOSTR VARCHAR(23),
    FATURATARIHI TIMESTAMP,
    FATURASERINO SERINOSIZE5,
    FATURANO NUMERIC(18,0),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DETAYCOUNT INTEGER,
    MODULID INTEGER,
    ODEMETARIHI TIMESTAMP,
    DOVIZID VARCHAR(5),
    FIRMADOVIZKURU1 DOUBLE PRECISION,
    FIRMADOVIZKURU2 DOUBLE PRECISION,
    TABANDOVIZKURU1 DOUBLE PRECISION,
    TABANDOVIZKURU2 DOUBLE PRECISION,
    GENELTOPLAM NUMERIC(18,2),
    BRUTTOPLAM NUMERIC(18,2),
    ISKONTOTOPLAMI NUMERIC(18,2),
    PARASALISKONTO DOUBLE PRECISION,
    ISKONTOORANI DOUBLE PRECISION,
    TAHSILEDILECEKKDV NUMERIC(18,2),
    INDIRILECEKKDV NUMERIC(18,2),
    ARATOPLAM NUMERIC(18,2),
    TEVKIFATMATRAHI NUMERIC(18,2),
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZKURU1 DOUBLE PRECISION,
    ISLEMDOVIZKURU2 DOUBLE PRECISION,
    ISLEMDOVIZTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZTAHSILEDILECEKKDV DOUBLE PRECISION,
    FIRMASEMT VARCHAR(20),
    FIRMABINANO VARCHAR(30),
    FIRMASOKAK VARCHAR(200),
    FIRMAKAPINO VARCHAR(30),
    FIRMASEHIR VARCHAR(20),
    FIRMAULKE VARCHAR(20),
    FIRMAVERGIDAIRESI VARCHAR(20),
    FIRMAVERGIHESAPNO VARCHAR(20),
    FIRMATELEFON VARCHAR(15),
    FIRMAFAX VARCHAR(15),
    FIRMAEMAIL VARCHAR(80),
    FIRMAWEB VARCHAR(80),
    FIRMAYETKILI VARCHAR(50),
    TEVKIFATORANI DOUBLE PRECISION,
    IRSALIYENO VARCHAR(50),
    IRSALIYETARIH TIMESTAMP,
    NOTU VARCHAR(1024),
    EFATURAGONDERIMTIPI VARCHAR(13),
    VADEGUN VARCHAR(10),
    FIRMADOVIZID VARCHAR(5),
    IHRACKAYITLI SMALLINT,
    EFATURA SMALLINT,
    KAYITTARIHI TIMESTAMP,
    GIBENTEGRASYONID VARCHAR(50),
    ODEMESEKLIADI ADI40,
    ISTISNAKODU INTEGER,
    ISTISNAADI VARCHAR(255))
AS
declare variable MinDovizid varchar(5);
declare variable MaxDovizid varchar(5);
declare variable MinDovizKuru1 double precision;
declare variable MinDovizKuru2 double precision;
declare variable MaxDovizKuru1 double precision;
declare variable MaxDovizKuru2 double precision;
declare variable KdvYuvarlamaCarpani  double precision;
declare variable FNotu Blob Sub_Type Text;
declare variable BlobSize Integer;
declare variable FDokumaHareketi smallint;
begin
    ModulID = Null;
    For Select BoyaFatura.BoyaFaturaId, BoyaFatura.FaturaTarihi, BoyaFatura.OdemeTarihi, BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo, BoyaFatura.FirmaID, BoyaFatura.HareketTipiID,
         BoyaFatura.dovizid, BoyaFatura.islemdovizkuru1, BoyaFatura.islemdovizkuru2,
         Firma.Adi, Firma.Semt, firma.Sokak, Firma.Sehir, Firma.Kapino, Firma.Email, Firma.VergiDairesi, firma.vergihesapno,
         Firma.Telefon1, firma.telefon2, firma.ulke, Firma.BinaNo,
         BoyaFatura.BrutToplam, BoyaFatura.ParasalIskonto, BoyaFatura.IskontoOrani, BoyaFatura.IskontoToplami, --iskonto orani olmadığından 2 kez iskontotoplami uygulandı
         BoyaFatura.tahsiledilecekkdv, BoyaFatura.indirilecekkdv,  BoyaFatura.islemdovizid, BoyaFatura.islemdovizkuru1, BoyaFatura.IslemDovizKuru2,
         BoyaFatura.islemdoviztoplam, BoyaFatura.islemdoviztahsiledilecekkdv,
         (BoyaFatura.BrutToplam + BoyaFatura.tahsiledilecekkdv - BoyaFatura.iskontotoplami) GenelToplam,
         (BoyaFatura.BrutToplam + BoyaFatura.TahsilEdilecekKDV - BoyaFatura.IskontoToplami + BoyaFatura.IndirilecekKdv) AraToplam, (cast(BoyaFatura.TevkifatOrani1 as Double Precision) / BoyaFatura.TevkifatOrani2) TevkifatOrani,
         Firma.Web, BoyaFatura.KdvYuvarlamaCarpani, BoyaFatura.Notu,
           CASE Egemenintegertostr(BoyaFatura.EFatura) || Egemenintegertostr(Firma.EFaturaGonderimTipi)
            WHEN '10' THEN 'TICARIFATURA'
            WHEN '11' THEN 'TEMELFATURA'
            ELSE 'EARSIVFATURA'
         END,
         Firma.DovizId, BoyaFatura.DovizKuru1, BoyaFatura.DovizKuru2, Coalesce(IstisnaKodlari.IhracKayitli,0), BoyaFatura.EFatura, BoyaFatura.InsertTarihi, BoyaFatura.GibEntegrasyonId, OdemeSekli.Adi, BoyaFatura.IstisnaKodu, IstisnaKodlari.Adi
   from BoyaFatura
   left join Firma On Firma.FirmaID = BoyaFatura.firmaid
   left join OdemeSekli on OdemeSekli.OdemeSekliId = BoyaFatura.OdemeSekliId
   left join IstisnaKodlari on IstisnaKodlari.IstisnaKodu = BoyaFatura.IstisnaKodu
   where (BoyaFatura.EFatura=1 or :EArsivAktif = 1) and (BoyaFatura.Entegrasyon = 1)and (BoyaFatura.HareketTipiID = 0 or BoyaFatura.HareketTipiID = 1) and
         (BoyaFatura.FaturaTarihi >= :BaslangicTarihi and BoyaFatura.FaturaTarihi <= :BitisTarihi) and
         ((BoyaFatura.GIBEntegrasyonKullaniciID is null and :GonderilmeDurumu = 0) or
          (BoyaFatura.GIBEntegrasyonKullaniciID is not null and :GonderilmeDurumu = 1) or
          (:GonderilmeDurumu = 2))
   into :BoyaFaturaid, :FaturaTarihi, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FirmaId, :HareketTipi, :DovizId,
       :FirmaDovizKuru1, :FirmaDovizKuru2, :Firmaadi, :Firmasemt, :Firmasokak, :FirmaSehir, :firmakapino, :firmaemail,
       :FirmaVergiDairesi, :Firmavergihesapno, :FirmaTelefon, :FirmaFax,  :Firmaulke, :Firmabinano, :BrutToplam,
       :ParasalIskonto, :IskontoOrani, :IskontoToplami, :TahsilEdilecekKdv, :IndirilecekKdv, :IslemDovizID,
       :IslemDovizKuru1, :IslemDovizKuru2, :IslemDovizToplam, :IslemDovizTahsilEdilecekKdv,
       :GenelToplam, :AraToplam, :TevkifatOrani, :FirmaWeb, :KdvYuvarlamaCarpani, :FNotu, :EFaturaGonderimTipi, :FirmaDovizId,
       :TabanDovizKuru1, :TabanDovizKuru2, :IhracKayitli, :EFatura,:KayitTarihi, :GibEntegrasyonId, :OdemeSekliAdi, :IstisnaKodu, :IstisnaAdi do
   begin
     select count(*) from BoyaFaturaDetay where BoyaFaturaDetay.BoyaFaturaID = :BoyaFaturaID into :DetayCount;

     FirmaYetkili = '';
     Select First 1 FirmaYetkili.Adi from FirmaYetkili where FirmaYetkili.FirmaID = :FirmaID order by DefaultYetkili Desc into :FirmaYetkili;

     ModulID = 0; /*Boya Faturasi*/

     IrsaliyeNo    = Null;
     IrsaliyeTarih = Null;
     select First 1 BoyaIrsaliye.IrsaliyeSeriNo || BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi from BoyaIrsaliyeDetay
       left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeID = BoyaIrsaliye.BoyaIrsaliyeID
     where BoyaIrsaliyeDetay.BoyaFaturaID = :BoyaFaturaID into :IrsaliyeNo, :IrsaliyeTarih;

     TevkifatMatrahi = (BrutToplam - IskontoToplami);
     TevkifatOrani = TevkifatOrani * 100;
     FaturaNoStr = FaturaSeriNo || Format_Float('', FaturaNo);

     if (HareketTipi = 0) then HareketTipiStr = 'Satış';
     else if (HareketTipi = 1) then HareketTipiStr = 'Alış İade';

     BlobSize = char_length(FNotu);

     if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
     else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
     else Notu = '';

     GenelToplam = Yuvarla(GenelToplam, 100);
     BrutToplam = Yuvarla(BrutToplam, 100);
     AraToplam = Yuvarla(AraToplam, 100);
     IskontoToplami = Yuvarla(IskontoToplami, 100);

     VadeGun = EgemenIntegertoStr(OdemeTarihi - FaturaTarihi);
     if (VadeGun = '0') then VadeGun = '-';
     suspend;
   end

   ModulID = Null;
   OdemeSekliAdi = Null;

   For Select OrmeFatura.OrmeFaturaId, OrmeFatura.FaturaTarihi, OrmeFatura.OdemeTarihi, OrmeFatura.SeriNo, OrmeFatura.FaturaNo, OrmeFatura.FirmaID, OrmeFatura.HareketTipi,
         OrmeFatura.dovizid, OrmeFatura.islemdovizkuru1, OrmeFatura.islemdovizkuru2,
         Firma.Adi, Firma.Semt, firma.Sokak, Firma.Sehir, Firma.Kapino, Firma.Email, Firma.VergiDairesi, firma.vergihesapno,
         Firma.Telefon1, firma.telefon2, firma.ulke, Firma.BinaNo,
         OrmeFatura.BrutToplam, OrmeFatura.ParasalIskonto, OrmeFatura.FaturaIskontosuOrani, OrmeFatura.IskontoToplami, --iskonto orani olmadığından 2 kez iskontotoplami uygulandı
         OrmeFatura.tahsiledilecekkdv, OrmeFatura.indirilecekkdv,  OrmeFatura.islemdovizid, OrmeFatura.islemdovizkuru1, OrmeFatura.IslemDovizKuru2,
         OrmeFatura.islemdoviztoplam, OrmeFatura.islemdoviztahsiledilecekkdv,
         (OrmeFatura.BrutToplam + OrmeFatura.tahsiledilecekkdv - OrmeFatura.iskontotoplami) GenelToplam,
         (OrmeFatura.BrutToplam + OrmeFatura.TahsilEdilecekKDV - OrmeFatura.IskontoToplami + OrmeFatura.IndirilecekKdv) AraToplam, (cast(OrmeFatura.TevkifatOrani1 as Double Precision) / OrmeFatura.TevkifatOrani2) TevkifatOrani,
         Firma.Web, OrmeFatura.KdvYuvarlamaCarpani, OrmeFatura.Notu,
         CASE Egemenintegertostr(OrmeFatura.EFatura) || Egemenintegertostr(Firma.EFaturaGonderimTipi)
            WHEN '10' THEN 'TICARIFATURA'
            WHEN '11' THEN 'TEMELFATURA'
            ELSE 'EARSIVFATURA'
         END, OrmeFatura.DokumaHareketi, Firma.DovizId,
         OrmeFatura.TabanDovizKuru1, OrmeFatura.TabanDovizKuru2, OrmeFatura.EFatura, Coalesce(IstisnaKodlari.IhracKayitli,0), OrmeFatura.InsertTarihi, OrmeFatura.GibEntegrasyonId, OrmeFatura.IstisnaKodu, IstisnaKodlari.Adi
   from OrmeFatura
   left join Firma On Firma.FirmaID = OrmeFatura.FirmaID
   left join IstisnaKodlari on IstisnaKodlari.IstisnaKodu = OrmeFatura.IstisnaKodu
   where (OrmeFatura.EFatura=1 or :EArsivAktif = 1) and (OrmeFatura.Entegrasyon = 1)and (OrmeFatura.HareketTipi = 2 or OrmeFatura.HareketTipi = 3 or OrmeFatura.HareketTipi = 14) and
         (OrmeFatura.FaturaTarihi >= :BaslangicTarihi and OrmeFatura.FaturaTarihi <= :BitisTarihi) and
         ((OrmeFatura.GIBEntegrasyonKullaniciID is null and :GonderilmeDurumu = 0) or
          (OrmeFatura.GIBEntegrasyonKullaniciID is not null and :GonderilmeDurumu = 1) or
          (:GonderilmeDurumu = 2))
   into :BoyaFaturaID, :FaturaTarihi, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FirmaId, :HareketTipi, :DovizId,
       :FirmaDovizKuru1, :FirmaDovizKuru2, :Firmaadi, :Firmasemt, :Firmasokak, :FirmaSehir, :firmakapino, :firmaemail,
       :FirmaVergiDairesi, :Firmavergihesapno, :FirmaTelefon, :FirmaFax,  :Firmaulke, :Firmabinano, :BrutToplam,
       :ParasalIskonto, :IskontoOrani, :IskontoToplami, :TahsilEdilecekKdv, :IndirilecekKdv, :IslemDovizID,
       :IslemDovizKuru1, :IslemDovizKuru2, :IslemDovizToplam, :IslemDovizTahsilEdilecekKdv,
       :GenelToplam, :AraToplam, :TevkifatOrani, :FirmaWeb, :KdvYuvarlamaCarpani, :FNotu, :EFaturaGonderimTipi, :FDokumaHareketi,
       :FirmaDovizId, :TabanDovizKuru1, :TabanDovizKuru2, :EFatura, :IhracKayitli, :KayitTarihi, :GibEntegrasyonId, :IstisnaKodu, :IstisnaAdi do
   begin
     FirmaYetkili = '';
     Select First 1 FirmaYetkili.Adi from FirmaYetkili where FirmaYetkili.FirmaID = :FirmaID order by DefaultYetkili Desc into :FirmaYetkili;
     select count(*) from StokHareket where StokHareket.OrmeFaturaID = :BoyaFaturaID into :DetayCount;


     ModulID = FDokumaHareketi + 1 ; /*Örme veya Dokuma Faturasi*/
     IrsaliyeNo    = Null;
     IrsaliyeTarih = Null;

     select First 1 OrmeIrsaliye.SeriNo || OrmeIrsaliye.IrsaliyeNo, IrsaliyeTarihi from StokHareket
     left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeID = StokHareket.OrmeIrsaliyeId
     where StokHareket.OrmeFaturaID = :BoyaFaturaID into :IrsaliyeNo, :IrsaliyeTarih;

     TevkifatMatrahi = (BrutToplam - IskontoToplami);
     TevkifatOrani = TevkifatOrani * 100;
     FaturaNoStr = FaturaSeriNo || Format_Float('', FaturaNo);

     if (HareketTipi = 2) then HareketTipiStr = 'Satış';
     else if (HareketTipi = 3) then HareketTipiStr = 'Alış İade';
     else if (HareketTipi = 14) then HareketTipiStr = 'Reklamasyon Çıkış';

     BlobSize = char_length(FNotu);

     if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
     else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
     else Notu = '';

     GenelToplam = Yuvarla(GenelToplam, 100);
     BrutToplam = Yuvarla(BrutToplam, 100);
     AraToplam = Yuvarla(AraToplam, 100);
     IskontoToplami = Yuvarla(IskontoToplami, 100);
     VadeGun = EgemenIntegertoStr(OdemeTarihi - FaturaTarihi);
     if (VadeGun = '0') then VadeGun = '-';
     suspend;
   end

   ModulID = Null;
   For Select  Fatura.FaturaId, Fatura.FaturaTarihi, Fatura.OdemeTarihi, Fatura.SeriNo, Fatura.FaturaNo, Fatura.FirmaID, Fatura.HareketTipi,
        Fatura.dovizid, Fatura.islemdovizkuru1, Fatura.islemdovizkuru2,
        Firma.Adi, Firma.Semt, firma.Sokak, Firma.Sehir, Firma.Kapino, Firma.Email, Firma.VergiDairesi, firma.vergihesapno,
        Firma.Telefon1, firma.telefon2, firma.ulke, Firma.BinaNo, Fatura.BrutToplam,
        Fatura.ISKONTO1TUTARI + Fatura.ISKONTO2TUTARI + Fatura.ISKONTO1ORANTUTARI + Fatura.ISKONTO2ORANTUTARI , /* Fatura.ParasalIskonto*/
        100 - (Fatura.BrutToplam + Fatura.tahsiledilecekkdv - (Fatura.ISKONTO1TUTARI + Fatura.ISKONTO2TUTARI + Fatura.ISKONTO1ORANTUTARI+ Fatura.ISKONTO2ORANTUTARI )) / Fatura.BrutToplam * 100, /*IskontoOrani*/
        Fatura.ISKONTO1TUTARI + Fatura.ISKONTO2TUTARI + Fatura.ISKONTO1ORANTUTARI + Fatura.ISKONTO2ORANTUTARI, /*IskontoToplami*/
        Fatura.tahsiledilecekkdv, Fatura.indirilecekkdv,  Fatura.islemdovizid, Fatura.islemdovizkuru1, Fatura.IslemDovizKuru2,
        Fatura.islemdoviztoplam, Fatura.islemdoviztahsiledilecekkdv,
        (Fatura.BrutToplam + Fatura.tahsiledilecekkdv - (Fatura.ISKONTO1TUTARI+Fatura.ISKONTO2TUTARI+Fatura.ISKONTO1ORANTUTARI+Fatura.ISKONTO2ORANTUTARI)) GenelToplam,
        (Fatura.BrutToplam + Fatura.TahsilEdilecekKDV - (Fatura.ISKONTO1TUTARI+Fatura.ISKONTO2TUTARI+Fatura.ISKONTO1ORANTUTARI+Fatura.ISKONTO2ORANTUTARI) + Fatura.IndirilecekKdv) AraToplam, (cast(Fatura.TevkifatOrani1 as Double Precision) / Fatura.TevkifatOrani2) TevkifatOrani,
        Firma.Web, Fatura.KdvYuvarlamaCarpani, Fatura.Notu,
        CASE Egemenintegertostr(Fatura.EFatura) || Egemenintegertostr(Firma.EFaturaGonderimTipi)
          WHEN '10' THEN 'TICARIFATURA'
          WHEN '11' THEN 'TEMELFATURA'
          ELSE 'EARSIVFATURA'
        END,
        Firma.DovizId, Fatura.FirmaDovizKuru1, Fatura.FirmaDovizKuru2, Coalesce(IstisnaKodlari.IhracKayitli,0), Fatura.EFatura, Fatura.InsertTarihi, Fatura.GibEntegrasyonId, OdemeSekli.Adi, Fatura.IstisnaKodu, IstisnaKodlari.Adi
   from Fatura
   left join Firma On Firma.FirmaID = Fatura.firmaid
   left join OdemeSekli on OdemeSekli.OdemeSekliId = Fatura.OdemeSekliId
   left join IstisnaKodlari on IstisnaKodlari.IstisnaKodu = Fatura.IstisnaKodu
   where (Fatura.EFatura=1 or :EArsivAktif = 1) and (Fatura.Entegrasyon = 1)and (Fatura.HareketTipi = 2 or Fatura.HareketTipi = 3 ) and
         (Fatura.FaturaTarihi >= :BaslangicTarihi and Fatura.FaturaTarihi <= :BitisTarihi) and
         ((Fatura.GIBEntegrasyonKullaniciID is null and :GonderilmeDurumu = 0) or
          (Fatura.GIBEntegrasyonKullaniciID is not null and :GonderilmeDurumu = 1) or
          (:GonderilmeDurumu = 2))
   into :BoyaFaturaid, :FaturaTarihi, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FirmaId, :HareketTipi, :DovizId,
        :FirmaDovizKuru1, :FirmaDovizKuru2, :Firmaadi, :Firmasemt, :Firmasokak, :FirmaSehir, :firmakapino, :firmaemail,
        :FirmaVergiDairesi, :Firmavergihesapno, :FirmaTelefon, :FirmaFax,  :Firmaulke, :Firmabinano, :BrutToplam,
        :ParasalIskonto, :IskontoOrani, :IskontoToplami, :TahsilEdilecekKdv, :IndirilecekKdv, :IslemDovizID,
        :IslemDovizKuru1, :IslemDovizKuru2, :IslemDovizToplam, :IslemDovizTahsilEdilecekKdv,
        :GenelToplam, :AraToplam, :TevkifatOrani, :FirmaWeb, :KdvYuvarlamaCarpani, :FNotu, :EFaturaGonderimTipi, :FirmaDovizId,

   begin
     select count(*) from STOKDETAYIGETIR(2, NULL, NULL, :BoyaFaturaID) into :DetayCount;
     FirmaYetkili = '';
     Select First 1 FirmaYetkili.Adi from FirmaYetkili where FirmaYetkili.FirmaID = :FirmaID order by DefaultYetkili Desc into :FirmaYetkili;

     ModulID = 3; /*Stok Faturasi*/

     IrsaliyeNo    = Null;
     IrsaliyeTarih = Null;

     select first 1 Irsaliye.SeriNo || Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi from StokDetay
     left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
     where StokDetay.FaturaId =:BoyaFaturaID into :IrsaliyeNo, :IrsaliyeTarih;

     TevkifatMatrahi = (BrutToplam - IskontoToplami);
     TevkifatOrani = TevkifatOrani * 100;
     FaturaNoStr = FaturaSeriNo || Format_Float('', FaturaNo);

     if (HareketTipi = 2) then HareketTipiStr = 'Satış';
     else if (HareketTipi = 3) then HareketTipiStr = 'Alış İade';

     BlobSize = char_length(FNotu);

     if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
     else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
     else Notu = '';

     GenelToplam = Yuvarla(GenelToplam, 100);
     BrutToplam = Yuvarla(BrutToplam, 100);
     AraToplam = Yuvarla(AraToplam, 100);
     IskontoToplami = Yuvarla(IskontoToplami, 100);

     VadeGun = EgemenIntegertoStr(OdemeTarihi - FaturaTarihi);
     if (VadeGun = '0') then VadeGun = '-';
     suspend;
   end
