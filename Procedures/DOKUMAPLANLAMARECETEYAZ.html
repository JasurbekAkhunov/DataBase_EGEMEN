CREATE PROCEDURE DOKUMAPLANLAMARECETEYAZ(
    ADOKUMAPLANLAMAGRUPID VARCHAR(10),
    ADOKUMAMAKINAID VARCHAR(20),
    AFIRMAID VARCHAR(20))
RETURNS (
    DOKUMAPLANLAMAID INTEGER,
    DOKUMAPLANLAMARECETEID INTEGER,
    DOKUMAMAKINAID MAKINAKODUSIZE,
    DOKUMAPLANLAMAHAMADI ADI255,
    DOKUMAPLANLAMASIRA INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    KULLANIMORANI DOUBLE PRECISION,
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    TAHMINIFIREORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    ATKICOZGU SMALLINT,
    LOT VARCHAR(30),
    TELSAYISI INTEGER,
    AGIRLIKKILO BOYAMIKTARSIZE,
    UZUNLUKMETRE BOYAMIKTARSIZE,
    HATASTR VARCHAR(500),
    URETIMPARTINO VARCHAR(30),
    ATKILOT VARCHAR(255),
    COZGULOT VARCHAR(255),
    SEPETNO INTEGER,
    DISKADI ADI100,
    DISKNO ADI100,
    DISKROTASYONU ADI255,
    DISKTEMP ADI255,
    DISKGAUGE ADI255,
    UZAMA DOUBLE PRECISION,
    TANSIYONAYARI DOUBLE PRECISION,
    MMRACK INTEGER)
AS
declare variable hstIplik       integer;
declare variable FLot           varchar(30);
declare variable FAtkiCozgu     integer;
declare variable OncekiMakinaId MakinaKoduSize;
begin
  Select hstIplik from HamTipiGetir into :hstIplik;
  OncekiMakinaId = Null;

  For select DokumaPlanlamaYaz.DokumaPlanlamaId, DokumaPlanlamaRecete.DokumaPlanlamaReceteId, DokumaPlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, DokumaPlanlamaRecete.RenkId, Renk.Adi, DokumaPlanlamaRecete.KullanimOrani, DokumaPlanlamaRecete.CikisMiktari, DokumaPlanlamaRecete.IadeMiktari,
             DokumaPlanlamaYaz.PlanlananMiktar * DokumaPlanlamaRecete.KullanimOrani / 100, DokumaPlanlamaYaz.TahminiFireOrani,
             DokumaPlanlamaRecete.UreticiID, Uretici.Adi, DokumaPlanlamaRecete.AtkiCozgu, DokumaPlanlamaRecete.Lot, DokumaPlanlamaRecete.TelSayisi,
             DokumaPlanlamaYaz.DokumaMakinaId, DokumaPlanlamaYaz.HamAdi, DokumaPlanlamaYaz.Sira, DokumaPlanlamaYaz.UretimPartiNo,
             DiskTanitim.Adi, DiskTanitim.DiskNo, DiskTanitim.Rotasyonu, DiskTanitim.Temp, DiskTanitim.Gauge, DokumaPlanlamaRecete.Uzama, DokumaPlanlamaRecete.TansiyonAyari, DokumaPlanlamaRecete.MmRack
             from DokumaPlanlamaYaz(:ADokumaPlanlamaGrupId, :ADokumaMakinaId, :AFirmaId)
  left join DokumaPlanlamaRecete on DokumaPlanlamaRecete.DokumaPlanlamaId = DokumaPlanlamaYaz.DokumaPlanlamaId
  left join Ham on Ham.HamId = DokumaPlanlamaRecete.HamId
  left join Renk on Renk.RenkId = DokumaPlanlamaRecete.RenkId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  left join Uretici on Uretici.UreticiID = DokumaPlanlamaRecete.UreticiID
  left join DiskTanitim On DiskTanitim.DiskTanitimId = DokumaPlanlamaRecete.DiskTanitimId
  where DokumaPlanlamaRecete.DokumaPlanlamaId is not null
  order by DokumaPlanlamaYaz.DokumaMakinaId, DokumaPlanlamaYaz.Sira
  into :DokumaPlanlamaId, :DokumaPlanlamaReceteId, :HamId, :HamTipiId, :HamTipiAdi,
       :HamAdi, :RenkId, :RenkAdi, :KullanimOrani, :CikisMiktari, :IadeMiktari, :KullanilacakMiktar,
       :TahminiFireOrani, :UreticiID, :UreticiAdi ,:AtkiCozgu, :Lot, :TelSayisi, :DokumaMakinaId, :DokumaPlanlamaHamAdi, :DokumaPlanlamaSira, :UretimPartiNo,
       :DiskAdi, :DiskNo, :DiskRotasyonu, :DiskTemp, :DiskGauge, :Uzama, :TansiyonAyari, :MmRack do
  begin
    if (OncekiMakinaId is null or OncekiMakinaId <> DokumaMakinaId) then
      SepetNo = 1;
    else
      SepetNo = SepetNo + 1;

    if (TahminiFireOrani is null) then TahminiFireOrani = 0;

    if (HamTipiId = hstIplik) then KullanilacakMiktar = KullanilacakMiktar / ( (100 - TahminiFireOrani) / 100);

    select AgirlikKilo, UzunlukMetre, HataStr from DokumaIplikMiktariHesaplama(Null, :DokumaPlanlamaReceteId, Null, 1, Null, Null) into :AgirlikKilo, :UzunlukMetre, :HataStr;

    AtkiLot = '';
    CozguLot = '';

    For Select DokumaPlanlamaRecete.Lot, DokumaPlanlamaRecete.AtkiCozgu from DokumaPlanlamaRecete
      left join Ham on Ham.HamID = DokumaPlanlamaRecete.HamId
      where DokumaPlanlamaRecete.DokumaPlanlamaID = :DokumaPlanlamaID
      order by DokumaPlanlamaRecete.AtkiCozgu, DokumaPlanlamaRecete.SiraNo
      into :FLot, :FAtkiCozgu do
    begin
      if (FAtkiCozgu = 1) then
      begin
        if (Char_Length(AtkiLot) + Char_Length(FLot) < 255) then
        begin
         if (AtkiLot <> '' ) then AtkiLot = AtkiLot || cast(', ' as varchar(2) character set UTF8);
         AtkiLot = AtkiLot || FLot;
        end
      end
      else
      begin
        if (Char_Length(CozguLot) + Char_Length(FLot) < 255) then
        begin
         if (CozguLot <> '' ) then CozguLot = CozguLot || cast(', ' as varchar(2) character set UTF8);
         CozguLot = CozguLot || FLot;
        end
      end
    end

    OncekiMakinaId = DokumaMakinaId;

    suspend;
  end
end
