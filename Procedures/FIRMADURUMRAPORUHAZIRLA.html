CREATE PROCEDURE FIRMADURUMRAPORUHAZIRLA
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BIRIMID VARCHAR(2),
    SIPARISMIKTAR NUMERIC(12,3),
    PARTILENENMIKTAR NUMERIC(12,3),
    DEPOCIKISMIKTAR NUMERIC(12,3),
    SEVKEHAZIRMIKTAR NUMERIC(12,3),
    SEVKEHAZIRNETMIKTAR NUMERIC(12,3),
    BRUTSEVKEDILENMIKTAR NUMERIC(12,3),
    NETSEVKEDILENMIKTAR NUMERIC(12,3),
    HAMDEPOMIKTAR NUMERIC(12,3))
AS
begin
/* has örme tüm stokları kg olarak baktıgı icin metreyi dikkate almadık */
  SiparisMiktar         = 0;
  PartilenenMiktar      = 0;
  DepoCikisMiktar       = 0;
  SevkeHazirMiktar      = 0;
  SevkeHazirNetMiktar   = 0;
  BrutSevkEdilenMiktar  = 0;
  NetSevkEdilenMiktar   = 0;
  HamDepoMiktar         = 0;
  BirimID = 'Kg';

  for Select HamStokRafEnvanterRaporu.FirmaId, HamStokRafEnvanterRaporu.FirmaAdi,  Sum(HamStokRafEnvanterRaporu.KalanKilo) from HamStokRafEnvanterRaporu(1, null, null,null, null, Null, null, null, null, null, null, 1, 1, null, Null, 2, 0, null, null, null,
  null, null, null, null, null, null, null, null, null, null, null, null)
    Group by FirmaId, FirmaAdi
    into :FirmaId, :FirmaAdi, :HamDepoMiktar do suspend;

  HamDepoMiktar = 0;

  For Select BoyaSiparis.FirmaId, Firma.Adi, Sum(BoyaSiparisDetay.SiparisKilo), Sum(BoyaSiparisDetay.PartilenenKilo - BoyaSiparisDetay.IptalKilo) from BoyaSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaID = BoyaSiparis.FirmaId
    left join Ham on Ham.HamID = BoyaSiparisDetay.HamId
    left join HamTipi on HamTipi.HamTipiID = Ham.HamTipiId
    where BoyaSiparisDetay.Kapali = 0 and BoyaSiparis.Fason = 1
    group by BoyaSiparis.FirmaId, Firma.Adi
    into :FirmaId, :FirmaAdi, :SiparisMiktar, :PartilenenMiktar do suspend;

  SiparisMiktar = 0;
  PartilenenMiktar = 0;

  for Select BoyaSiparis.FirmaId, Firma.Adi, Sum(PartiStok.BrutSevkKilo), Sum(PartiStok.NetSevkKilo)  from BoyaSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaID = BoyaSiparis.FirmaId
    left join Ham on Ham.HamID = BoyaSiparisDetay.HamId
    left join HamTipi on HamTipi.HamTipiID = Ham.HamTipiId
    left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
    where BoyaSiparisDetay.Kapali = 0 and BoyaSiparis.Fason = 1
    group by BoyaSiparis.FirmaId, Firma.Adi
    into :FirmaId, :FirmaAdi, :BrutSevkEdilenMiktar, :NetSevkEdilenMiktar do suspend;

  BrutSevkEdilenMiktar = 0;
  NetSevkEdilenMiktar = 0;

  for Select BoyaSiparis.FirmaId, Firma.Adi, Sum(PartiStok.Kilo - PartiStok.BrutSevkKilo), Sum(PartiStok.NetKilo - PartiStok.NetSevkKilo)  from BoyaSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaID = BoyaSiparis.FirmaId
    left join Ham on Ham.HamID = BoyaSiparisDetay.HamId
    left join HamTipi on HamTipi.HamTipiID = Ham.HamTipiId
    left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
    where BoyaSiparisDetay.Kapali = 0 and BoyaSiparis.Fason = 1 and PartiStok.Durumu = 7 and PartiStok.PartiStokIptalKayitID is null
    group by BoyaSiparis.FirmaId, Firma.Adi
    into :FirmaId, :FirmaAdi, :SevkeHazirMiktar, :SevkeHazirNetMiktar do suspend;

  SevkeHazirMiktar = 0;
  SevkeHazirNetMiktar = 0;

  for Select BoyaSiparis.FirmaId, Firma.Adi, Sum(PartiStok.Kilo) from BoyaSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaID = BoyaSiparis.FirmaId
    left join Ham on Ham.HamID = BoyaSiparisDetay.HamId
    left join HamTipi on HamTipi.HamTipiID = Ham.HamTipiId
    left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
    where BoyaSiparisDetay.Kapali = 0 and BoyaSiparis.Fason = 1 and PartiStok.Durumu > 0 and PartiStok.PartiStokIptalKayitID is null
    group by BoyaSiparis.FirmaId, Firma.Adi
    into :FirmaId, :FirmaAdi, :DepoCikisMiktar do suspend;

  DepoCikisMiktar = 0;
end
