CREATE PROCEDURE HGKOLIICINURUNASORTILISTESI(
    AHGSIPARISKAYITID INTEGER,
    AKOLILENENLERIGOSTERME EGEBOOLEAN)
RETURNS (
    URUNID URUNKODUSIZE,
    URUNADI ADI80,
    HGSIPARISDETAYASORTIID INTEGER,
    HGSIPARISDETAYASORTIADI ADI100,
    ASORTISTR VARCHAR(10),
    RENKID INTEGER,
    RENKADI ADI50,
    BEDENID INTEGER,
    BEDENADI ADI30,
    KOLILENENMIKTAR INTEGER,
    SIPARISMIKTARI BOYAMIKTARSIZE)
AS
begin
  for select HgSiparisDetayUrun.UrunId, Urun.Adi, Null, Null,
             '', Sum(iif(HgSiparisDetayUrun.HgSiparisDetayAsortiId is null, HgSiparisDetayUrun.Miktar, HgSiparisDetayUrun.Miktar * HgSiparisDetayAsorti.Miktar * -1)), Renk.RenkId, Renk.Adi, Beden.BedenId, Beden.Adi
             from HgSiparis
      left join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HGSiparisKayitId
      left join HgSiparisDetayUrun on HgSiparisDetayUrun.HgSiparisDetayId = HgSiparisDetay.HgSiparisDetayId
      left join Urun on Urun.UrunId = HgSiparisDetayUrun.UrunId
      left join ModelKumas on ModelKumas.ModelKumasId = Urun.ModelKumasId
      left join Renk on Renk.RenkId = ModelKumas.RenkId
      left join ModelBeden on ModelBeden.ModelBedenId = Urun.ModelBedenId
      left join Beden on Beden.BedenId = ModelBeden.BedenId
      left join HgSiparisDetayAsorti on HgSiparisDetayAsorti.HgSiparisDetayAsortiId = HgSiparisDetayUrun.HgSiparisDetayAsortiId
      where (HgSiparis.HgSiparisKayitId = :AHgSiparisKayitId or :AHgSiparisKayitId is null) and
      HgSiparisDetay.Kapali = 0
      group by
        HgSiparisDetayUrun.UrunId, Urun.Adi, Renk.RenkId, Renk.Adi, Beden.BedenId, Beden.Adi
      having Sum(iif(HgSiparisDetayUrun.HgSiparisDetayAsortiId is null, HgSiparisDetayUrun.Miktar, HgSiparisDetayUrun.Miktar * HgSiparisDetayAsorti.Miktar * -1)) > 0
      order by HgSiparisDetayUrun.UrunId
      into :UrunId, :UrunAdi, :HgSiparisDetayAsortiId, :HgSiparisDetayAsortiAdi,
             :AsortiStr, :SiparisMiktari, :RenkId, :RenkAdi, :BedenId, :BedenAdi do
  begin
      select Coalesce(Sum(HgKoliDetay.Miktar), 0) from HgSiparis
      left join HgKoliFisi on HgKoliFisi.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
      left join HgKoli on HgKoli.HgKoliFisiId = HgKoliFisi.HgKoliFisiId
      left join HgKoliDetay on HgKoliDetay.HgKoliId = HgKoli.HgKoliId
      left join HgSiparisDetayUrun on HgSiparisDetayUrun.HgSiparisKayitId = HgSiparis.HgSiparisKayitId and HgSiparisDetayUrun.UrunId = HgKoliDetay.UrunId
      left join HgSiparisDetay on HgSiparisDetay.HgSiparisDetayId = HgSiparisDetayUrun.HgSiparisDetayId
      where (HgSiparis.HgSiparisKayitId = :AHgSiparisKayitId or :AHgSiparisKayitId is null) and
      HgSiparisDetay.Kapali = 0 and
      HgSiparisDetayUrun.HgSiparisDetayAsortiId is null and
      HgKoliDetay.UrunId = :UrunId into :KolilenenMiktar;

      if (AKolilenenleriGosterme = 0 or AKolilenenleriGosterme = 1 and (SiparisMiktari - KolilenenMiktar) > 0)  then
        suspend;
  end

  for select Null, Null, HgSiparisDetayAsorti.HgSiparisDetayAsortiId, HgSiparisDetayAsorti.Adi,
             'Evet', Sum(HgSiparisDetayAsorti.Miktar), Null, Null, Null, Null
             from HgSiparis
      left join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HGSiparisKayitId
      left join HgSiparisDetayAsorti on HgSiparisDetayAsorti.HgSiparisDetayId = HgSiparisDetay.HgSiparisDetayID
      where (HgSiparis.HgSiparisKayitId = :AHgSiparisKayitId or :AHgSiparisKayitId is null) and
      HgSiparisDetay.Kapali = 0 and
      HgSiparisDetayAsorti.HgSiparisDetayAsortiId is not null
      group by
        HgSiparisDetayAsorti.HgSiparisDetayAsortiId, HgSiparisDetayAsorti.Adi
      order by HgSiparisDetayAsorti.Adi
      into :UrunId, :UrunAdi, :HgSiparisDetayAsortiId, :HgSiparisDetayAsortiAdi,
             :AsortiStr, :SiparisMiktari, :RenkId, :RenkAdi, :BedenId, :BedenAdi do
  begin
      select Coalesce(Sum(HgKoliDetay.Miktar), 0) from HgSiparis
      left join HgKoliFisi on HgKoliFisi.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
      left join HgKoli on HgKoli.HgKoliFisiId = HgKoliFisi.HgKoliFisiId
      left join HgKoliDetay on HgKoliDetay.HgKoliId = HgKoli.HgKoliId
      left join HgSiparisDetayAsorti on HgSiparisDetayAsorti.HgSiparisKayitId = HgSiparis.HgSiparisKayitId and HgSiparisDetayAsorti.HgSiparisDetayAsortiId = HgKoliDetay.HgSiparisDetayAsortiId
      left join HgSiparisDetay on HgSiparisDetay.HgSiparisDetayId = HgSiparisDetayAsorti.HgSiparisDetayId
      where (HgSiparis.HgSiparisKayitId = :AHgSiparisKayitId or :AHgSiparisKayitId is null) and
      HgSiparisDetay.Kapali = 0 and
      HgKoliDetay.HgSiparisDetayAsortiId = :HgSiparisDetayAsortiId into :KolilenenMiktar;

      if (AKolilenenleriGosterme = 0 or AKolilenenleriGosterme = 1 and (SiparisMiktari - KolilenenMiktar) > 0)  then
        suspend;
  end
end
