CREATE PROCEDURE VALORLUCARIKARTDOKUMU(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    ASINIF VARCHAR(1),
    AFAIZORANI DOUBLE PRECISION,
    AKULLANICIID INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    VALORTARIHI TIMESTAMP,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    FARKGUN INTEGER,
    VADEFARKI DOUBLE PRECISION,
    BORCALACAK VARCHAR(1),
    HESAPKAPATMA INTEGER)
AS
declare variable FirmaGrup1 varchar(4);
declare variable FirmaGrup2 varchar(4);
declare variable FirmaGrup3 varchar(4);
declare variable Okey integer;
declare variable Personelmi integer;
declare variable PersonelCariYetkisiVar integer;

declare variable FBorc         Double Precision;

declare variable FBorcValorTarihi   TimeStamp;

declare variable ETarih         TimeStamp;
declare variable EAciklama      varchar(50);
declare variable EValorTarihi   TimeStamp;
declare variable EAlacak        Double Precision;
declare variable ESatirSayisi   integer;
declare variable EHesapKapatma  integer;
declare variable SonOkunan      integer;


declare variable EATarih         TimeStamp;
declare variable EAAciklama      varchar(50);
declare variable EAValorTarihi   TimeStamp;
declare variable EAAlacak        Double Precision;
declare variable EAHesapKapatma  integer;

declare variable BorcToplam             Double Precision;
declare variable AlacakToplam           Double Precision;
declare variable VadeFarkiToplami       Double Precision;

declare variable Ondalik integer;
declare variable YuvarlamaCarpani Double Precision;
declare variable I integer;
begin

  Select Ondalik from FieldFormat
    left join ModulFormat on ModulFormat.ModulFormatId = FieldFormat.ModulFormatId
    where FieldFormat.TableAdi = 'CARI' and FieldFormat.FieldAdi = 'MEBLAG'
    into :Ondalik;

  I = 1;
  YuvarlamaCarpani = 1.0;

  if (Ondalik is not null) then
  begin
    While (I <= Ondalik) do
    begin
      YuvarlamaCarpani = YuvarlamaCarpani * 10.0;
      I = I + 1;
    end
  end

/*
Select * from ValorluCariKartDokumu('M004', '', '','','','','','','01.01.2002', '31.12.2003', 0,0,0,0, null, 8, 1)
*/
  select PersonelCariYetkisiVar from Kullanici where KullaniciID = :AKullaniciId into :PersonelCariYetkisiVar;


  for Select FirmaId ,FirmaGrup1, FirmaGrup2, FirmaGrup3, Personelmi  from Firma
  where ((:AFirmaAralikli = 1 and FirmaId >= :ABaslangicFirmaId and (FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0))  and
        ((:AFirmaGrup1Aralikli = 1 and FirmaGrup1 >= :ABaslangicGrup1Id  and (FirmaGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and FirmaGrup2 >= :ABaslangicGrup2Id and (FirmaGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and FirmaGrup3 >= :ABaslangicGrup3Id and (FirmaGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = ''))

  into :FirmaId, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :Personelmi do
  begin
    Okey = 1;
    if (Personelmi = 1 and PersonelCariYetkisiVar = 0) then Okey = 0;

    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (okey = 1) then
    begin
      FBorc = 0;
      EAlacak = 0;
      SonOkunan = 0;
      EValorTarihi = Null;
      For Select Tarih, Aciklama, Meblag, ValorTarihi, HesapKapatma from FirmaCariBAListesi(:FirmaId, 'B', :ABaslangicTarihi, :ABitisTarihi, :ASinif)
         into :Tarih, :Aciklama, :FBorc, :ValorTarihi, :HesapKapatma do
      begin
        BorcAlacak = Null;
        Alacak = Null;
        Borc = FBorc;
        BorcToplam = FBorc;
        AlacakToplam = 0;
        VadeFarki = 0;
        VadeFarkiToplami = 0;
        FBorcValorTarihi = ValorTarihi;
        suspend;
        Borc = 0;
        if (EAAlacak > 0) then
        begin
          if (EAAlacak > FBorc) then
          begin
            Tarih = EATarih;
            ValorTarihi = EAValorTarihi;
            HesapKapatma = EAHesapKapatma;
            Aciklama = EAAciklama;
            EAAlacak = EAAlacak - FBorc;
            Alacak  = FBorc;
            FBorc = 0;
    /* Gunfark ve Vade Fark覺 hesapla*/
            FarkGun = TimeStampToDouble(EAValorTarihi) - TimeStampToDouble(FBorcValorTarihi);
            if (FarkGun < 0) then BorcAlacak = 'A';
            else BorcAlacak = 'B';
            VadeFarki = ((Alacak * AFaizOrani / 3000) * FarkGun);
            if (VadeFarki < 0) then VadeFarki = VadeFarki * -1;
            VadeFarki = Yuvarla(VadeFarki, YuvarlamaCarpani);
            suspend;
          end
          else
          begin
            Tarih = EATarih;
            ValorTarihi = EAValorTarihi;
            HesapKapatma = EAHesapKapatma;
            Aciklama = EAAciklama;
            FBorc = FBorc - EAAlacak;
            Alacak  = EAAlacak;
            EAAlacak = 0;
    /* Gunfark ve Vade Fark覺 hesapla*/
            FarkGun = TimeStampToDouble(EAValorTarihi) - TimeStampToDouble(FBorcValorTarihi);
            if (FarkGun < 0) then BorcAlacak = 'A';
            else BorcAlacak = 'B';
            VadeFarki = ((Alacak * AFaizOrani / 3000) * FarkGun);
            if (VadeFarki < 0) then VadeFarki = VadeFarki * -1;
            VadeFarki = Yuvarla(VadeFarki, YuvarlamaCarpani);
            suspend;
          end
          AlacakToplam = AlacakToplam + Alacak;
          if (BorcAlacak = 'B') then VadeFarkiToplami = VadeFarkiToplami - VadeFarki;
          else VadeFarkiToplami = VadeFarkiToplami + VadeFarki;
        end

        if (FBorc > 0) then
        begin
          For Select Tarih, Aciklama, Meblag, ValorTarihi, SatirSayisi, HesapKapatma from FirmaCariBAListesi(:FirmaId, 'A', :ABaslangicTarihi, :ABitisTarihi, :ASinif) where SatirSayisi > :SonOkunan
           into :ETarih, :EAciklama, :EAlacak, :EValorTarihi, :ESatirSayisi, :EHesapKapatma do
          begin
            if (FBorc > 0 and ESatirSayisi > SonOkunan) then
            begin
              EATarih = ETarih;
              EAAciklama = EAciklama;
              EAValorTarihi = EValorTarihi;
              EAHesapKapatma = EHesapKapatma;

              Tarih    = ETarih;
              Aciklama = EAciklama;
              ValorTarihi = EValorTarihi;
              SonOkunan = ESatirSayisi;
              HesapKapatma = EHesapKapatma;
              if (EAlacak > FBorc) then
              begin
                EAlacak = EAlacak - FBorc;
                Alacak  = FBorc;
                FBorc = 0;
    /* Gunfark ve Vade Fark覺 hesapla*/
                FarkGun = TimeStampToDouble(EValorTarihi) - TimeStampToDouble(FBorcValorTarihi);
                if (FarkGun < 0) then BorcAlacak = 'A';
                else BorcAlacak = 'B';
                VadeFarki = ((Alacak * AFaizOrani / 3000) * FarkGun);
                if (VadeFarki < 0) then VadeFarki = VadeFarki * -1;
                VadeFarki = Yuvarla(VadeFarki, YuvarlamaCarpani);
                suspend;
              end
              else
              begin
                FBorc = FBorc - EAlacak;
                Alacak  = EAlacak;
                EAlacak = 0;
    /* Gunfark ve Vade Fark覺 hesapla*/
                FarkGun = TimeStampToDouble(EValorTarihi) - TimeStampToDouble(FBorcValorTarihi);
                if (FarkGun < 0) then BorcAlacak = 'A';
                else BorcAlacak = 'B';
                VadeFarki = ((Alacak * AFaizOrani / 3000) * FarkGun);
                if (VadeFarki < 0) then VadeFarki = VadeFarki * -1;
                VadeFarki = Yuvarla(VadeFarki, YuvarlamaCarpani);
                suspend;
              end
              AlacakToplam = AlacakToplam + Alacak;
              if (BorcAlacak = 'B') then VadeFarkiToplami = VadeFarkiToplami - VadeFarki;
              else VadeFarkiToplami = VadeFarkiToplami + VadeFarki;
              EAAlacak = EAlacak;
            end
          end
        end
        Tarih         = Null;
        Aciklama      = '';
        ValorTarihi   = Null;
        Borc          = BorcToplam;
        Alacak        = AlacakToplam;
        FarkGun       = Null;
        BorcAlacak    = 'A';

        if (VadeFarkiToplami < 0) then
        begin
          VadeFarkiToplami = VadeFarkiToplami * -1;
          BorcAlacak = 'B';
        end
        HesapKapatma = 0;
        VadeFarki     = VadeFarkiToplami;
        VadeFarki = Yuvarla(VadeFarki, YuvarlamaCarpani);
        suspend;
      end
    end
  end
end
