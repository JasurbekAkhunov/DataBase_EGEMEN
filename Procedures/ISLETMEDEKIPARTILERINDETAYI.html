CREATE PROCEDURE ISLETMEDEKIPARTILERINDETAYI(
    ADURUMU INTEGER,
    AFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AISLEMGRUBUID INTEGER,
    AISLEMTIPIID INTEGER,
    AACILIYETDERECESIID INTEGER,
    ARECETEID VARCHAR(20),
    AMODELNO VARCHAR(20),
    ARECETEYAZILDI INTEGER,
    APARTIGRUPID VARCHAR(4))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    PARTISIRANO INTEGER,
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    SIPARISRENKADI VARCHAR(40),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    HAMADI VARCHAR(255),
    MELANJKODU VARCHAR(20),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    SONASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    GIRISVAR VARCHAR(3),
    CIKISVAR VARCHAR(3),
    LOT VARCHAR(150),
    DURUMU INTEGER,
    DURUMUSTR VARCHAR(20),
    ISLETMEICITAMIR INTEGER,
    ISLETMEICITAMIRSTR VARCHAR(8),
    SONRAKIASAMAADI VARCHAR(40),
    SONRAKIASAMAID INTEGER,
    BEKLEMEDE INTEGER,
    BEKLEMENOTU VARCHAR(1023),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    BASKITIPIID INTEGER,
    BASKITIPIADI VARCHAR(40),
    ACILISTARIHI TIMESTAMP,
    ALTBEZSTR VARCHAR(4))
AS
declare variable SonSira Integer;
declare variable SonrakiSiraId Integer;
declare variable PartiAsamalariId integer;
declare variable FTamirAsamaId integer;

declare variable PAAsamaId integer;
declare variable PAAsamaAdi varchar(40);
declare variable RUAAsamaId integer;
declare variable RUAAsamaAdi varchar(40);
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable FPartiTamirSinifId     integer;
declare variable FPartiStokBekleId      integer;
declare variable Bitti          integer;
declare variable FBlob          Blob;
begin
  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;
  select TamirAsamaId from Sistem where SistemId = 1 into :FTamirAsamaId;

  for select PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, PartiStok.PartiSiraNo, PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.ReferansSiparisNo,
             BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
             BoyaSiparisDetay.MelanjKodu, PartiStok.HamAdiToplami, PartiStok.KapAdedi, PartiStok.Kilo, PartiStok.Metre, PartiStok.Adet, PartiStok.Durumu, PartiDurum.Adi,
             PartiStok.PartiAsamalariId, PA.AsamaId, PA.Adi, RUA.AsamaId, RUA.Adi, PartiAsamalari.Sira,
             PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi, PartiTamirSinif.PartiTamirSinifId, PartiStok.BirimFiyat,
             PartiStok.DovizId, BoyaSiparisDetay.BaskiTipiId, BaskiTipi.Adi, Min(PartiStokBekleId), Parti.AcilisTarihi from PartiStok
      left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      left join PartiTamirSinif on PartiTamirSinif.PartiSinifId = PartiStok.PartiSinifId and PartiTamirSinif.Bitti = 0
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join Firma on Firma.FirmaId = Parti.FirmaId
      left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiStok.PartiSinifId and PartiAsamaStok.PartiAsamalariId = PartiStok.PartiAsamalariId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiStok.PartiAsamalariId
      left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
      left join Asama  PA on PA.AsamaId = PartiAsamalari.AsamaId
      left join Asama RUA on RUA.AsamaId = Recete.UstAsamaId
      left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
      left join PartiStokBekle on PartiStokBekle.PartiSinifId = PartiStok.PartiSinifId and PartiStokBekle.Bitti = 0
      left join BaskiTipi on BaskiTipi.BaskiTipiId = BoyaSiparisDetay.BaskiTipiId
      where ((:ADurumu is null and Durumu <= 6) or (:ADurumu is not null and Durumu = :ADurumu)) and
                                  (:AFirmaId is null or Firma.AnaHesapFirmaId = :AFirmaId) and
                                  (PartiStok.PartiStokIptalKayitId is null) and
                                  (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
                                  (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and
                                  (:AIslemGrubuId is null or Parti.IslemGrubuId = :AIslemGrubuId) and
                                  (:AIslemTipiId is null or (Parti.IslemTipiId = :AIslemTipiId) or (:AIslemTipiId = -1 and Parti.IslemTipiId <> :itTamir)) and
                                  (:AAciliyetDerecesiId is null or Parti.AciliyetDerecesiId >= :AAciliyetDerecesiId) and
                                  (:AReceteId is null or Parti.ReceteId = :AReceteId) and
                                  (:AModelNo is null or Parti.ModelNo = :AModelNo) and
                                  (:AReceteYazildi is null or (Parti.ReceteId is not null and Parti.ReceteYazildi = :AReceteYazildi))
      Group By PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, PartiStok.PartiSiraNo, PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.ReferansSiparisNo,
             BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
             BoyaSiparisDetay.MelanjKodu, PartiStok.HamAdiToplami, PartiStok.KapAdedi, PartiStok.Kilo, PartiStok.Metre, PartiStok.Adet, PartiStok.Durumu, PartiDurum.Adi,
             PartiStok.PartiAsamalariId, PA.AsamaId, PA.Adi, RUA.AsamaId, RUA.Adi, PartiAsamalari.Sira,
             PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi, PartiTamirSinif.PartiTamirSinifId, PartiStok.BirimFiyat,
             PartiStok.DovizId, BoyaSiparisDetay.BaskiTipiId, BaskiTipi.Adi, Parti.AcilisTarihi
        order by PartiStok.PartiKayitId, PartiStok.PartiSinifId, PartiStok.PartiSiraNo
        into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :PartiSiraNo, :BoyaSiparisDetayId, :MusteriSiparisNo, :ReferansSiparisNo,  :SiparisRenkAdi, :MusteriRenkKoduveAdi, :MusteriKaliteNo,
             :MusteriKaliteAdi, :MelanjKodu, :HamAdi, :KapAdedi, :Kilo, :Metre, :Adet, :Durumu, :DurumuStr, :PartiAsamalariId,
             :PAAsamaId, :PAAsamaAdi, :RUAAsamaId, :RUAAsamaAdi, :SonSira, :GirisTarihi, :CikisTarihi, :FPartiTamirSinifId,
             :BirimFiyat, :DovizId, :BaskiTipiId, :BaskiTipiAdi, :FPartiStokBekleId, :AcilisTarihi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    if (PAAsamaId is not null) then
    begin
      SonAsamaId = PAAsamaId;
      AsamaAdi = PAAsamaAdi;
    end
    else
    begin
      SonAsamaId = RUAAsamaId;
      AsamaAdi = RUAAsamaAdi;
    end

    SonrakiAsamaAdi = null;
    SonrakiAsamaId = null;
    if ((SonAsamaId is null or SonAsamaId <> FTamirAsamaId) or FTamirAsamaId is null) then
    begin
      SonrakiSiraId = null;
      if (SonSira is null) then SonSira = -1;

      select min(Sira) from PartiAsamalari
         left join PartiAsamaStok on PartiAsamaStok.PartiKayitId = PartiAsamalari.PartiKayitId and PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
         where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.Sira > :SonSira
              and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1
         into :SonrakiSiraId;         /* and PartiAsamalari.SerbestAsama = 0  */


      if (SonrakiSiraId is not null) then
      begin
        Select PA1.AsamaId, PA1.Adi, RUA1.AsamaId, RUA1.Adi from PartiAsamalari
          left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId and PartiAsamalari.ReceteId is not null
          left join Asama  PA1 on PA1.AsamaId = PartiAsamalari.AsamaId
          left join Asama RUA1 on RUA1.AsamaId = Recete.UstAsamaId
          where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.Sira = :SonrakiSiraId
          into :PAAsamaId, :PAAsamaAdi, :RUAAsamaId, :RUAAsamaAdi;
        if (PAAsamaId is not null) then
        begin
          SonrakiAsamaId = PAAsamaId;
          SonrakiAsamaAdi = PAAsamaAdi;
        end
        else
        begin
          SonrakiAsamaId = RUAAsamaId;
          SonrakiAsamaAdi = RUAAsamaAdi;
        end
      end
    end

    Select Lot from PartiStokLotBilgisi(:PartiSinifID) into :Lot;

    if (GirisTarihi is not null) then GirisVar = 'Var';
    else GirisVar = '';
    if (CikisTarihi is not null) then CikisVar = 'Var';
    else CikisVar = '';

    if (FPartiTamirSinifId is not null) then
    begin
      IsletmeIciTamir = 1;
      IsletmeIciTamirStr = 'İç Tamir';
    end
    else
    begin
      IsletmeIciTamir = 0;
      IsletmeIciTamirStr = '';
    end

    Beklemede = 0;
    BeklemeNotu = '';

    if (FPartiStokBekleId is not null) then
    begin
      Beklemede = 1;
      Select Notu From PartiStokBekle where PartiStokBekle.PartiStokBekleID = :FPartiStokBekleID into :FBlob;

      if (char_length(FBlob) > 1023) then BeklemeNotu = Left(FBlob, 1023);
      else BeklemeNotu = Left(FBlob, char_length(FBlob));

    end
    suspend;
  end
end
