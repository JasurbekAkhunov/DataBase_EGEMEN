CREATE PROCEDURE DOKUMAURETIMRAPORU_2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AURETIMISLEMTIPIID INTEGER,
    ABASLANGICMAKINAID VARCHAR(10),
    ABITISMAKINAID VARCHAR(10),
    AMAKINAKODUARALIKLI INTEGER,
    AMAKINAGRUPID VARCHAR(4),
    AHAMID INTEGER)
RETURNS (
    STOKHAREKETID INTEGER,
    COZGUISEMRIRECETEID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    LOT VARCHAR(30),
    RENKID INTEGER,
    RENKADI ADI50,
    ORMEFISID INTEGER,
    FISNO INTEGER,
    HESAPLANANMIKTAR BOYAMIKTARSIZE,
    CIKILANMIKTAR BOYAMIKTARSIZE,
    TAMAMICIKILDI EGEBOOLEAN,
    ACIKLAMA VARCHAR(500))
AS
declare variable uitCozgu integer;
declare variable shtUretim integer;
declare variable shtFasonUretim integer;
begin
  select ShtUretim, shtFasonUretim from OrmeStokHareketTipiGetir into :ShtUretim, :shtFasonUretim;

  select uitCozgu from UretimIslemTipiGetir into :uitCozgu;

  For Select StokHareket.StokHareketId, DokumaUretimeOtomatikCikis.CozguIsEmriReceteId, DokumaUretimeOtomatikCikis.OrmeFisId, DokumaUretimeOtomatikCikis.HesaplananMiktar,
    DokumaUretimeOtomatikCikis.CikilanMiktar, DokumaUretimeOtomatikCikis.TamamiCikildi, DokumaUretimeOtomatikCikis.Aciklama,
    coalesce(CozguIsEmriRecete.HamId, dokumaplanlamarecete.hamid), coalesce(Ham.Adi, hamr.adi),
    CozguIsEmriRecete.Lot, coalesce(CozguIsEmriRecete.RenkId, dokumaPlanlamarecete.renkid),coalesce(Renk.Adi, renkr.adi), OrmeFis.FisNo from StokHareket
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = StokHareket.DokumaMakinaId
    left join DokumaUretimeOtomatikCikis on DokumaUretimeOtomatikCikis.StokHareketId = StokHareket.StokHareketId
    left join OrmeFis on OrmeFis.OrmeFisId = DokumaUretimeOtomatikCikis.OrmeFisId
    left join CozguIsEmriRecete on CozguIsEmriRecete.CozguIsEmriReceteId = DokumaUretimeOtomatikCikis.CozguIsEmriReceteId
    left join Ham on Ham.HamId = CozguIsEmriRecete.HamId
    left join Renk on Renk.RenkId = CozguIsEmriRecete.RenkId
    left join DokumaPlanlamaRecete on DokumaPlanlamaRecete.DokumaPlanlamaReceteId = DokumaUretimeOtomatikCikis.DokumaPlanlamaReceteId
    left join Ham hamr on Hamr.HamId = DokumaPlanlamaRecete.hamid
    left join Renk renkr on Renkr.RenkId = DokumaPlanlamaRecete.RenkId
    where StokHareket.Tarih >= :ABaslangicTarihi and StokHareket.Tarih <= :ABitistarihi and
         ((StokHareket.HareketTipi = :ShtUretim or StokHareket.HareketTipi = :shtFasonUretim) and StokHareket.HamIade = 0) and
         /*(StokHareket.UretimIslemTipiId = :uitCozgu) and*/
         ((:AMakinaKoduAralikli = 1 and StokHareket.DokumaMakinaId >= :ABaslangicMakinaId and (StokHareket.DokumaMakinaId <= :ABitisMakinaId or :ABitisMakinaId = '')) or
          (:AMakinaKoduAralikli = 0) or (:AMakinaKoduAralikli = 1 and :ABaslangicMakinaId = '' and :ABitisMakinaId = '')) and
         (DokumaMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
         (StokHareket.HamId = :AHamId or :AHamId is null)
    into :StokHareketId, :CozguIsEmriReceteId, :OrmeFisId, :HesaplananMiktar, :CikilanMiktar, :TamamiCikildi, :Aciklama, :HamId, :HamAdi, :Lot, :RenkId, :RenkAdi, :FisNo do
  begin
    suspend;
  end
end
