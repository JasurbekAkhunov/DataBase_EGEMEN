CREATE PROCEDURE MAKINAANIMASYONBILGISIOKU(
    AMAKINAID VARCHAR(10))
RETURNS (
    MAKINAID VARCHAR(10),
    CALISIYOR EGEBOOLEAN,
    ARIZA EGEBOOLEAN,
    SERVISCIGELDI EGEBOOLEAN,
    PARCABEKLIYOR EGEBOOLEAN,
    TEMIZLIK EGEBOOLEAN,
    ARIZAYERIID INTEGER,
    IMALATAARAVERMESEBEBIADI ADI50,
    IMALATAARAVERENPERSONELADI ADI40,
    IMALATAARAVERMETARIHI TIMESTAMP)
AS
Declare Variable DKDurumu Integer;
begin
  for select Makina.MakinaId, Coalesce(MakinaDurumu.Calisiyor, 0), Coalesce(MakinaDurumu.Ariza, 0), Coalesce(MakinaDurumu.ServisciGeldi, 0),
      Coalesce(MakinaDurumu.Temizlik, 0), Coalesce(ArizaIhbar.ParcaBekliyor,0), ArizaYeriListesi.ArizaYeriId, AraVermeSebebi.Adi,

      Personel.Adi, iif(ImalataAraVerme.ImalataAraVermeId is not null, MakinaDurumu.Tarih, null)
      from Makina
      /*left join PartiAsamaStokDetay on PartiAsamaStokDetay.MakinaId = Makina.MakinaId and PartiAsamaStokDetay.CikisTarihi is null*/
      left join MakinaDurumu on MakinaDurumu.MakinaDurumuId = Makina.MakinaDurumuId
      left join ArizaIhbar on ArizaIhbar.MakinaId = Makina.MakinaId and ArizaIhbar.BittiMakinaDurumuId is null
      left join ArizaYeriListesi on ArizaYeriListesi.ArizaYeriId = ArizaIhbar.ArizaYeriId
      left join ImalataAraVerme on ImalataAraVerme.MakinaId = Makina.MakinaId and ImalataAraVerme.BittiMakinaDurumuId is null
      left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
      left join Personel on ImalataAraVerme.AraVerdirenPersonelId = Personel.PersonelId
      where Makina.OtomasyonaDahil = 1 and (Makina.MakinaId = :AMakinaId or :AMakinaId is null)
      into :MakinaId, :Calisiyor, :Ariza, :ServisciGeldi, :Temizlik, :ParcaBekliyor, :ArizaYeriId, :ImalataAraVermeSebebiAdi,
      :ImalataAraVerenPersonelAdi, :ImalataAraVermeTarihi do
  begin
    Select  DKDurumu from  MakinaCalismaSuresi
      where MakinaCalismaSuresi.MakinaId = :MakinaId and
            MakinaCalismaSuresi.VardiyaTarihi = (Select  MAX(VardiyaTarihi) from  MakinaCalismaSuresi where  MakinaCalismaSuresi.MakinaId =:MakinaId)
    into :DKDurumu;

    if (DKDurumu is Null) then DKDurumu = 0;
    Calisiyor = DKDurumu;


    suspend;
  end
end
