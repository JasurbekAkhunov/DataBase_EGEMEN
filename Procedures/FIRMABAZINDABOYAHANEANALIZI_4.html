CREATE PROCEDURE FIRMABAZINDABOYAHANEANALIZI_4(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    KICTAMIRUCRETLIKILOTOPLAM DOUBLE PRECISION,
    KICTAMIRUCRETSIZKILOTOPLAM DOUBLE PRECISION,
    KICTAMIRUCRETLIMETRETOPLAM DOUBLE PRECISION,
    KICTAMIRUCRETSIZMETRETOPLAM DOUBLE PRECISION,
    IICTAMIRUCRETLIKILOTOPLAM DOUBLE PRECISION,
    IICTAMIRUCRETSIZKILOTOPLAM DOUBLE PRECISION)
AS
declare variable FlastFirmaId  varchar(20);
declare variable FFirmaId     varchar(20);
declare variable FFirmaAdi    varchar(100);
declare variable FPartiTipiId integer;
declare variable FUcretli     integer;
declare variable FKilo        Numeric(12,3);
declare variable FMetre       Numeric(12,3);
begin
  FlastFirmaId = '';

  For Select Parti.FirmaId, Firma.Adi, Parti.PartiTipiId, TamirCozumSekli.Ucretli, Sum(PartiStok.Kilo), Sum(PartiStok.Metre) from PartiTamir
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
    left join PartiTamirSinif on PartiTamirSinif.PartiTamirId = PartiTamir.PartiTamirId
    left join PartiStok on PartiStok.PartiSinifId = PartiTamirSinif.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    where PartiTamir.Tarih >= :ABaslangicTarihi and PartiTamir.Tarih <= :ABitisTarihi and
          (Firma.AnaHesapFirmaID = :AfirmaId or :AFirmaID is null)
    group by Parti.FirmaId, Firma.Adi, Parti.PartiTipiId, TamirCozumSekli.Ucretli
    into :FFirmaId, :FFirmaAdi, :FPartiTipiId, :FUcretli, :FKilo, :FMetre do
  begin
    if (FLastFirmaID <> FFirmaID) then
    begin
      if (FLastFirmaID <> '') then suspend;
      FirmaID = FFirmaId;
      FLastFirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;

      KIcTamirUcretliKiloToplam       = 0;
      KIcTamirUcretsizKiloToplam      = 0;
      KIcTamirUcretliMetreToplam      = 0;
      KIcTamirUcretsizMetreToplam     = 0;
      IIcTamirUcretliKiloToplam       = 0;
      IIcTamirUcretsizKiloToplam      = 0;
    end

    if (FPartiTipiID = 0) then
    begin
      if (FUcretli = 1) then IIcTamirUcretliKiloToplam = IIcTamirUcretliKiloToplam + FKilo;
      else IIcTamirUcretsizKiloToplam = IIcTamirUcretsizKiloToplam + FKilo;
    end
    else
    begin
      if (FUcretli = 1) then KIcTamirUcretliKiloToplam = KIcTamirUcretliKiloToplam + FKilo;
      else KIcTamirUcretsizKiloToplam = KIcTamirUcretsizKiloToplam + FKilo;

      if (FUcretli = 1) then KIcTamirUcretliMetreToplam = KIcTamirUcretliMetreToplam + FMetre;
      else KIcTamirUcretsizMetreToplam = KIcTamirUcretsizMetreToplam + FMetre;
    end
  end

  if (FLastFirmaID <> '') then suspend;
end
