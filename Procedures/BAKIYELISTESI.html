CREATE PROCEDURE BAKIYELISTESI(
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    ABITISTARIHI TIMESTAMP,
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    AFIRMAGRUP4ARALIKLI INTEGER,
    ABAKIYETIPI INTEGER,
    AMINIMUMBAKIYE DOUBLE PRECISION,
    ARAPORDOVIZID VARCHAR(10),
    ARAPORTARIHI TIMESTAMP,
    ASINIF VARCHAR(1),
    AISLEMDOVIZID VARCHAR(5))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    FIRMAGRUP1 VARCHAR(4),
    FIRMAGRUP2 VARCHAR(4),
    FIRMAGRUP3 VARCHAR(4),
    FIRMAGRUP4 VARCHAR(4),
    TEMSILCIID INTEGER,
    ADRESI1 VARCHAR(40),
    ADRESI2 VARCHAR(40),
    ADRESI3 VARCHAR(40),
    SEMT VARCHAR(20),
    SEHIR VARCHAR(20),
    VERGIDAIRESI VARCHAR(20),
    VERGIHESAPNO VARCHAR(20),
    TELEFON1 VARCHAR(15),
    TELEFONTIPI1 INTEGER,
    TELEFON2 VARCHAR(15),
    TELEFONTIPI2 INTEGER,
    TELEFON3 VARCHAR(15),
    TELEFONTIPI3 INTEGER,
    TELEFON4 VARCHAR(15),
    TELEFONTIPI4 INTEGER,
    TELEFON5 VARCHAR(15),
    TELEFONTIPI5 INTEGER,
    EMAIL VARCHAR(80),
    WEB VARCHAR(80),
    ISKONTOORANI DOUBLE PRECISION,
    ACIKHESAPRISKI DOUBLE PRECISION,
    CEKSENETRISKI DOUBLE PRECISION,
    SIPARISRISKI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    BORCTOPLAM DOUBLE PRECISION,
    ALACAKTOPLAM DOUBLE PRECISION,
    BORCBAKIYE DOUBLE PRECISION,
    ALACAKBAKIYE DOUBLE PRECISION,
    BAKIYE DOUBLE PRECISION,
    DOVIZBORCTOPLAM DOUBLE PRECISION,
    DOVIZALACAKTOPLAM DOUBLE PRECISION,
    DOVIZBORCBAKIYE DOUBLE PRECISION,
    DOVIZALACAKBAKIYE DOUBLE PRECISION,
    DOVIZBAKIYE DOUBLE PRECISION,
    BAKIYEBA VARCHAR(1),
    BAKIYEYAZIILE VARCHAR(255),
    DOVIZBAKIYEYAZIILE VARCHAR(255),
    ACIKSIPRISKTOPLAMI DOUBLE PRECISION,
    ORMESIPRISKTOPLAMI DOUBLE PRECISION,
    BOYASIPRISKTOPLAMI DOUBLE PRECISION,
    TRANSFEREDILMEMISFATURATUTARI DOUBLE PRECISION,
    TOPLAMRISK DOUBLE PRECISION,
    SATICI INTEGER,
    MUSTERI INTEGER,
    KAPALI INTEGER,
    FIRMAGRUP1ADI VARCHAR(30),
    FIRMAGRUP2ADI VARCHAR(30),
    FIRMAGRUP3ADI VARCHAR(30),
    FIRMAGRUP4ADI VARCHAR(30))
AS
declare variable okey           integer;
declare variable DovizKuru      Double Precision;
declare variable XBakiye        Double Precision;
declare variable DovizCinsi     Varchar(20);
declare variable DovizAltBirim  Varchar(10);
declare variable Kurus          Integer;
begin
  if (AMinimumBakiye is null) then AMinimumBakiye = 0;
  Select Doviz.Cinsi, Doviz.AltBirim From Doviz where Doviz.DovizId=:ARaporDovizId into :DovizCinsi, :DovizAltBirim;
  for Select FirmaId, Firma.Adi, Adi2, FirmaGrup1, FirmaGrup2, FirmaGrup3, FirmaGrup4, Adresi1, Adresi2, Adresi3, Semt, Sehir, VergiDairesi, VergiHesapNo,
    Telefon1, Telefon2, Telefon3, Telefon4, Telefon5, Email, Web, IskontoOrani, AcikHesapRiski, CekSenetRiski, SiparisRiski, DovizId,
  Satici, Musteri, Kapali, FG1.Adi, FG2.Adi, FG3.Adi, FG4.Adi  from Firma
  left join FirmaGrup FG1 on FG1.FirmaGrupId = Firma.FirmaGrup1
  left join FirmaGrup FG2 on FG2.FirmaGrupId = Firma.FirmaGrup2
  left join FirmaGrup FG3 on FG3.FirmaGrupId = Firma.FirmaGrup3
  left join FirmaGrup FG4 on FG4.FirmaGrupId = Firma.FirmaGrup4
  where ((:AFirmaAralikli = 1 and FirmaId >= :ABaslangicFirmaId and (FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
        ((:AFirmaGrup1Aralikli = 1 and FirmaGrup1 >= :ABaslangicGrup1Id and (FirmaGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and FirmaGrup2 >= :ABaslangicGrup2Id and (FirmaGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and FirmaGrup3 >= :ABaslangicGrup3Id and (FirmaGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
        ((:AFirmaGrup4Aralikli = 1 and FirmaGrup4 >= :ABaslangicGrup4Id and (FirmaGrup4 <= :ABitisGrup4Id or :ABitisGrup4Id = '')) or (:AFirmaGrup4Aralikli = 0) or (:AFirmaGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = ''))
  into :FirmaId, :FirmaAdi, :FirmaAdi2, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :FirmaGrup4, :Adresi1, :Adresi2, :Adresi3, :Semt, :Sehir, :VergiDairesi, :VergiHesapNo,
  :Telefon1, :Telefon2, :Telefon3, :Telefon4, :Telefon5, :Email, :Web, :IskontoOrani, :AcikHesapRiski, :CekSenetRiski, :SiparisRiski, :DovizId,
  :Satici, :Musteri, :Kapali, :FirmaGrup1Adi, :FirmaGrup2Adi, :FirmaGrup3Adi, :FirmaGrup4Adi do
  begin
    if (AIslemDovizId is not  Null) then
      DovizId = AIslemDovizId;

    Okey = 1;
    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, FirmaGrup4) = 0) then okey = 0;

    if (okey = 1) then
    begin
      Select AcikSipRiskToplami, OrmeSipRiskToplami, BoyaSipRiskToplami from SiparisRiskiniGetir(:FirmaID) into :AcikSipRiskToplami, :OrmeSipRiskToplami, :BoyaSipRiskToplami;
      Select TransferEdilmemisFaturaTutari from TransferEdilmemisFaturalari(:FirmaID) into :TransferEdilmemisFaturaTutari;
      if (AcikSipRiskToplami is Null) then AcikSipRiskToplami = 0;
      if (TransferEdilmemisFaturaTutari is Null) then TransferEdilmemisFaturaTutari = 0;
      ToplamRisk = TransferEdilmemisFaturaTutari + AcikSipRiskToplami;


      select BorcToplam, AlacakToplam, BorcBakiye, AlacakBakiye, Bakiye, BakiyeBa from FirmaBakiyeGetir(:FirmaId, :ABitisTarihi, :ASinif, :AIslemDovizId)
         into :BorcToplam, :AlacakToplam, :BorcBakiye, :AlacakBakiye, :Bakiye, :BakiyeBa;

      if (Bakiye is null) then Bakiye = 0;

      if (DovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then exception DovizKuruGirisiYapilmamis;

      DovizBorcToplam    = BorcToplam * DovizKuru;
      DovizAlacakToplam  = AlacakToplam  * DovizKuru;
      DovizBorcBakiye    = BorcBakiye * DovizKuru;
      DovizAlacakBakiye  = AlacakBakiye * DovizKuru;
      DovizBakiye        = Bakiye * DovizKuru;

      if (ABakiyeTipi = 0) then
      begin
        if (DovizBakiye < AMinimumBakiye) then okey = 0;
      end
      else  if (ABakiyeTipi = 1) then
      begin
        if (DovizBakiye > 0) then
        if (BakiyeBa <> 'B') then okey = 0;
        else
        begin
          if (DovizBakiye < AMinimumBakiye) then Okey = 0;
        end
        else if (AMinimumBakiye > 0) then okey = 0;
      end
      else if (ABakiyeTipi = 2) then
      begin
        if (DovizBakiye > 0) then
        if (BakiyeBa <> 'A') then okey = 0;
        else
        begin
          if (DovizBakiye < AMinimumBakiye) then Okey = 0;
        end
        else if (AMinimumBakiye > 0) then okey = 0;
      end
      if (okey = 1) then
      begin
         if (bakiye<>0) then
         begin
           xbakiye=bakiye-cast(bakiye as bigint);
           kurus = cast(xbakiye*100 as integer);
           if (kurus<0) then kurus= kurus +100;
         end
         else
           kurus = 0;

        if (kurus=0) then
          BakiyeYaziIle = YaziIle(Bakiye) ||' '|| DovizCinsi;
        else
          BakiyeYaziIle = YaziIle(Bakiye) ||' '|| DovizCinsi||' '||YaziIle(Kurus)||' '||DovizAltBirim;

         DovizBakiyeYaziIle = YaziIle(DovizBakiye);
         suspend;
      end
    end
  end
end
