CREATE PROCEDURE PARTISTOKTARTIBRUTHESAPLA(
    APARTISINIFID INTEGER,
    AEKRANDAN INTEGER)
AS
declare variable PartiStokTartiId integer;
declare variable TamamiTartildi integer;
declare variable KayitSayisi integer;
declare variable PartiBrutKilo   Numeric(12,3);
declare variable PartiBrutMetre  Numeric(12,3);
declare variable ToplamBrutKilo   Numeric(12,3);
declare variable ToplamBrutMetre  Numeric(12,3);
declare variable BrutKilo   Numeric(12,3);
declare variable BrutMetre  Numeric(12,3);
declare variable BrutuHesaplanmiyacakKaliteNolar varchar(15);
declare variable SevkEdilenBrutKilo numeric(12,3);
declare variable SevkEdilenMetre numeric(12,3);
declare variable SevkEdilmeyenKilo numeric(12,3);
declare variable SevkEdilmeyenMetre numeric(12,3);
declare variable SevktenKalanKilo numeric(12,3);
declare variable KiloKatSayi double precision;
declare variable SevktenKalanMetre numeric(12,3);
declare variable MetreKatSayi double precision;
begin

  Select Sistem.BrutuHesaplanmiyacakKaliteNolar from sistem into :BrutuHesaplanmiyacakKaliteNolar;

  select PartiStok.Kilo, PartiStok.Metre from PartiStok
    left join Ham On Ham.HamId = PartiStok.HamId
    left join parti On Parti.PartiKayitId = PartiStok.PartiKayitId
    where PartiStok.PartiSinifId = :APartiSinifId
    into :PartiBrutKilo, :PartiBrutMetre;

  Select Sum(PartiStokTarti.BrutKilo), Sum(PartiStokTarti.BrutMetre) from PartiStokTarti
    where PartiStokTarti.PartiSinifId = :APartiSinifId
    into :ToplamBrutKilo, :ToplamBrutMetre;

  Select sum(PartiStokTarti.Kilo), sum(PartiStokTarti.Metre) from PartiStokTarti
      left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId
      where PartiStokTarti.PartiSinifID = :APartiSinifID and MamulDepoTransferFisi.SistemKaydi = 1 and
            PartiStokTarti.BoyaIrsaliyeDetayId is null and PartiStokTarti.HamStokSarfFisiDetayId is null and
            (Position(KaliteNo, :BrutuHesaplanmiyacakKaliteNolar) = 0 or Coalesce(:BrutuHesaplanmiyacakKaliteNolar, '') = '') into :SevkEdilmeyenKilo, :SevkEdilmeyenMetre;


 Select sum(PartiStokTarti.BrutKilo), sum(PartiStokTarti.BrutMetre) from PartiStokTarti
      left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId
      where PartiStokTarti.PartiSinifID = :APartiSinifId and MamulDepoTransferFisi.SistemKaydi = 1 and
          (not  ( PartiStokTarti.BoyaIrsaliyeDetayId is null and PartiStokTarti.HamStokSarfFisiDetayId is null)) and
            (Position(KaliteNo, :BrutuHesaplanmiyacakKaliteNolar) = 0 or Coalesce(:BrutuHesaplanmiyacakKaliteNolar, '') = '') into  :SevkEdilenBrutKilo, :SevkedilenMetre;

  KiloKatSayi = 1;
  SevktenKalanKilo = PartiBrutKilo - coalesce(SevkEdilenBrutKilo,0);
  if (SevkEdilmeyenKilo = 0 ) then SevkEdilmeyenKilo = null;
  if (SevkEdilmeyenKilo is null)   then SevkEdilmeyenKilo = SevktenKalanKilo;
  BrutKilo = PartiBrutKilo - ToplamBrutKilo;
  if ((SevktenKalanKilo) > 0 and (SevkEdilmeyenKilo > 0)) then KiloKatSayi = SevktenKalanKilo / SevkEdilmeyenKilo;


  MetreKatSayi = 1;
  SevktenKalanMetre = PartiBrutMetre - coalesce(SevkEdilenMetre,0);
  if (SevkEdilmeyenMetre = 0 ) then SevkEdilmeyenMetre = null;
  if (SevkEdilmeyenMetre is null)   then SevkEdilmeyenMetre = SevktenKalanMetre;
  BrutMetre = PartiBrutMetre - ToplamBrutMetre;
  if (SevktenKalanMetre > 0 and SevkEdilmeyenMetre > 0) then MetreKatSayi = SevktenKalanMetre / SevkEdilmeyenMetre;


  --if (MetreKatSayi > 1 or KiloKatSayi > 1) then
  --begin

    for Select PartiStokTartiId from PartiStokTarti
      left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId
      where PartiStokTarti.PartiSinifID = :APartiSinifID and MamulDepoTransferFisi.SistemKaydi = 1 and
            PartiStokTarti.BoyaIrsaliyeDetayId is null and PartiStokTarti.HamStokSarfFisiDetayId is null and
            (Position(KaliteNo, :BrutuHesaplanmiyacakKaliteNolar) = 0 or Coalesce(:BrutuHesaplanmiyacakKaliteNolar, '') = '')
         into :PartiStokTartiId do
      begin
        if (StrLength(BrutuHesaplanmiyacakKaliteNolar) = 0) then
           if (PartiStokTartiId is null) then exception Kayithatasi 'Tamami tartildi bilgisi kaydedilemedi..';

        Update PartiStokTarti Set BrutKilo = Kilo * :KILOKatSayi , BrutMetre = Metre * :MetreKatsayi where PartiStokTarti.PartiStokTartiId = :PartiStokTartiId;

      end
  --end

  if (AEkrandan = 1) then
    Update PartiStok Set TamamiTartildi = 1 where PartiStok.PartiSinifId = :APartiSinifId;
end
