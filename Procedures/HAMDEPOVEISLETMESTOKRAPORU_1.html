CREATE PROCEDURE HAMDEPOVEISLETMESTOKRAPORU_1(
    ASTOKFIRMAID VARCHAR(20),
    ASIPARISFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATAMIR INTEGER,
    AFASON INTEGER,
    ARAPORHAMTIPI INTEGER,
    AHAMID INTEGER)
RETURNS (
    STOKFIRMAID VARCHAR(20),
    STOKFIRMAADI VARCHAR(100),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    HAMTIPIID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    TAMIRMI VARCHAR(10),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    MELANJKODU VARCHAR(20),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3))
AS
Declare variable Tamir integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable Kilo      Numeric(12,3);
declare variable Metre     Numeric(12,3);
declare variable KullanilanKilo   Numeric(12,3);
declare variable KullanilanMetre  Numeric(12,3);
declare variable IptalKilo        Numeric(12,3);
declare variable IptalMetre       Numeric(12,3);
declare variable HamStokMasterId  integer;
declare variable DokumaKumasiKilodanTakipEt integer;
begin

  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  for select HamStokMaster.StokFirmaId, StokFirma.Adi, BoyaSiparis.FirmaId, SiparisFirma.Adi, BoyaSiparis.BoyaSiparisYil,
             BoyaSiparis.BoyaSiparisId, HamStokMaster.BoyaSiparisDetayId, HamStokMaster.Fason,
             HamStokMaster.HamStokMasterId, HamStokMaster.HamId, Ham.Adi, ham.HamTipiId, HamStokMaster.Tamir, HamStokMaster.Kilo, HamStokMaster.Metre,
             BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.LabReceteKodu, labRecete.Adi, BoyaSiparisDetay.MelanjKodu from HamStokIrsaliye
  left join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
  left join Ham on Ham.HamId = HamStokMaster.HamId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = HamStokMaster.BoyaSiparisKayitId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = HamStokMaster.BoyaSiparisDetayId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join Firma StokFirma on StokFirma.FirmaId = HamStokMaster.StokFirmaId
  where HamStokMaster.PartiSinifId is null and HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
        (HamStokMaster.StokFirmaId = :AStokFirmaId or :AStokFirmaId is null) and
        (HamStokMaster.Tamir = :ATamir or :ATamir is null) and
         HamStokMaster.Fason = :AFason and
        (HamStokMaster.HamId = :AHamId or :AHamId is null) and
        ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporHamTipi = 0) or
            ( Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporHamTipi = 1) or :ARaporHamTipi is null ) and
        (BoyaSiparis.FirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null)
  into :StokFirmaId, :StokFirmaAdi, :SiparisFirmaId, :SiparisFirmaAdi, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId, :Fason,
       :HamStokMasterId, :HamId, :HamAdi, :HamTipiId, :Tamir, :Kilo, :Metre,
       :MusteriRenkKoduveAdi, :LabReceteKodu, :labReceteAdi, :MelanjKodu do
  begin
    KalanKilo = Kilo;
    KalanMetre = Metre;
    KullanilanKilo = Null;
    KullanilanMetre = Null;

    if (Tamir = 1) then Tamirmi = 'Tamir';
    else Tamirmi = null;

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    Select Sum(RafDetay.IadeEdilecekKilo + RafDetay.IadeEdilenKilo + RafDetay.PartilenenKilo + RafDetay.SatisKilo),
           Sum(RafDetay.IadeEdilecekMetre + RafDetay.IadeEdilenMetre + RafDetay.PartilenenMetre + RafDetay.SatisMetre) from HamStokDetay
      left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId
      where HamStokDetay.HamStokMasterId = :HamStokMasterId and RafDetay.RafDetayID is not null
     into :KullanilanKilo, :KullanilanMetre;

     if (KullanilanKilo is not null) then KalanKilo = KalanKilo - KullanilanKilo;
     if (KullanilanMetre is not null) then KalanMetre = KalanMetre - KullanilanMetre;

    Select Sum(RafDetay.IadeEdilecekKilo + RafDetay.IadeEdilenKilo + RafDetay.PartilenenKilo + RafDetay.SatisKilo),
           Sum(RafDetay.IadeEdilecekMetre + RafDetay.IadeEdilenMetre + RafDetay.PartilenenMetre + RafDetay.SatisMetre) from HamStokDetay
      left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId
      where HamStokDetay.HamStokMasterId = :HamStokMasterId and RafDetay.RafDetayID is not null and  rafDetay.PartiStokIptalKayitId is not null
      into :IptalKilo, :IptalMetre;

    if (IptalKilo is not null) then KalanKilo = KalanKilo + IptalKilo;
    if (IptalMetre is not null) then KalanMetre = KalanMetre + IptalMetre;

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    if ((HamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) and KalanKilo > 0) then suspend;
    else if (KalanMetre > 0) then suspend;
  end
end
