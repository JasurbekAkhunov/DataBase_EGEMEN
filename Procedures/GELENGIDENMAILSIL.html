CREATE PROCEDURE GELENGIDENMAILSIL(
    AKULLANICIID INTEGER,
    AMAILTIPI INTEGER,
    ATARIH TIMESTAMP)
AS
begin
  if (AMailTipi = 1) then
    delete from GidenMail where KullaniciId = :AKullaniciId and Tarih <= :ATarih;
  else if (AMailTipi = 2) then
    delete from GelenMail where KullaniciId = :AKullaniciId and Tarih <= :ATarih and Okundu = 1;
end
