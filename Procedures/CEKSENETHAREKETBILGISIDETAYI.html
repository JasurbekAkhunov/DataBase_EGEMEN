CREATE PROCEDURE CEKSENETHAREKETBILGISIDETAYI(
    AGENELKAYITID INTEGER)
RETURNS (
    CSHAREKETDETAYID INTEGER,
    TARIH TIMESTAMP,
    BORDROTIPI INTEGER,
    BORDROID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BANKAHESAPID VARCHAR(20),
    BANKAHESAPADI VARCHAR(40),
    AVUKATID VARCHAR(10),
    AVUKATADI VARCHAR(50),
    ACIKLAMA VARCHAR(50),
    MAKBUZNO VARCHAR(20),
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
   For Select ceksenethareketDetay.CshareketDetayId, ceksenethareket.BordroTipi, ceksenethareket.BordroId, ceksenethareket.FirmaId,
              Firma.Adi, ceksenethareket.BankaHesapId, BankaHesap.Adi, ceksenethareket.AvukatId, Avukat.Adi, ceksenethareket.Tarih,
              ceksenethareket.Aciklama, ceksenethareket.MakbuzNo, Kullanici.Adi, CekSenetHareketDetay.DegisimTarihi
               from CekSenetHareketDetay
      left join ceksenethareket on ceksenethareket.CshBordroKayitId = CekSenetHareketDetay.CshBordroKayitId
      left join firma on firma.FirmaId = CekSenetHareket.FirmaId
      left join BankaHesap on BankaHesap.BankaHesapId = CekSenetHareket.BankaHesapId
      left join Avukat on Avukat.AvukatId = CekSenetHareket.AvukatId
      left join Kullanici on Kullanici.KullaniciId = CekSenetHareketDetay.KullaniciId
     where CekSenetHareketDetay.GenelKayitId = :AGenelKayitId
     order by CekSenetHareketDetay.CsHareketDetayid
   into :CshareketDetayId, :BordroTipi, :BordroId, :FirmaId, :FirmaAdi, :BankaHesapId, :BankaHesapAdi, :AvukatId, :AvukatAdi,
        :Tarih, :Aciklama, :MakbuzNo, :KullaniciAdi, :DegisimTarihi do suspend;
end
