CREATE PROCEDURE MC_BOYASIPARISRAPORU_4(
    ABOYASIPARISDETAYID INTEGER,
    AFIRMAID VARCHAR(20),
    AKULLANICIID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    DURUMU INTEGER,
    DURUMUSTR VARCHAR(30),
    SINIFBOYASIPARISDETAYID INTEGER)
AS
begin
  For Select MC_BoyaSiparisRaporu_3.BoyaSiparisDetayId, MC_BoyaSiparisRaporu_3.PartiKayitId, PartiStok.PartiSinifId, PartiStok.HamAdiToplami,
             BoyaSiparisDetay.MusteriRenkKoduveAdi,
             PartiStok.Kilo, PartiStok.Metre, PartiStok.BoyaSiparisDetayId, PartiStok.Durumu  from MC_BoyaSiparisRaporu_3(:ABoyaSiparisDetayId, :AFirmaId, :AKullaniciId)
    left join PartiStok on PartiStok.PartiKayitId = MC_BoyaSiparisRaporu_3.PartiKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
    where MC_BoyaSiparisRaporu_3.PartiKayitid is not null and PartiStok.PartiStokIptalKayitId is null
    into :BoyaSiparisDetayId, :PartiKayitId, :PartiSinifId, :HamAdiToplami, :MusteriRenkKoduveAdi, :Kilo, :Metre,
         :SinifBoyaSiparisDetayId, :Durumu do
  begin
    Select DurumuStr from MC_PartiDurumHesapla(:PartiSinifId, :Durumu) into :DurumuStr;

    suspend;
  end
end
