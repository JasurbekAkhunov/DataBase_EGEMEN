CREATE PROCEDURE SEVKIYATLISTESIDETAYLIPARTI(
    AFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AARACID VARCHAR(11),
    APARTIGRUPID VARCHAR(4),
    AUNITEID INTEGER,
    APARTITIPI INTEGER,
    AMAMULDEPOSEVKLERIDAHIL INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARECETEID VARCHAR(20),
    AFASON INTEGER,
    AIMALATTIPIID INTEGER,
    APARTIKAYITID INTEGER,
    ABASKITIPIID INTEGER,
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    MUSTERIYESEVK INTEGER,
    FIRMAID VARCHAR(20),
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    PARTIIDSTR VARCHAR(30),
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    ISLEMGRUBUADI VARCHAR(20),
    ISLEMTIPIADI VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ESKILABRECETEKODU VARCHAR(20),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    MODELNO VARCHAR(20),
    PARTILENENKAPADEDI INTEGER,
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    BRUTSEVKKILO NUMERIC(12,3),
    NETSEVKKILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    BRUTSEVKMETRE NUMERIC(12,3),
    NETSEVKMETRE NUMERIC(12,3),
    FIRE DOUBLE PRECISION,
    FIREMIKTARI DOUBLE PRECISION,
    FIREKARARMIKTARI DOUBLE PRECISION,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    RENKDERINLIGIADI VARCHAR(20),
    PARCALISEVK INTEGER,
    OZELYERLESIM EGEBOOLEAN,
    OZELYERLESIMNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    OZELYERLESIMDOSYAADI VARCHAR(255))
AS
declare variable FTamamiTartildi integer;
declare variable FirmaKodumuz varchar(20);
declare variable FBrutKilo               Numeric(12,3);
declare variable FNetKilo                Numeric(12,3);
declare variable FBrutMetre               Numeric(12,3);
declare variable FNetMetre                Numeric(12,3);
declare variable biSatisIade    integer;
begin

  Select biSatisIade from BoyaIrsaliyeHareketTipiGetir into :biSatisIade;
  Select FirmaKodumuz from Sistem into :FirmaKodumuz;

  MusteriyeSevk = 1;
  For Select BoyaIrsaliye.FirmaId, Parti.PartiKayitId, Parti.Fason, Parti.PartigrupId, PartiGrup.Adi, Parti.PartiYil, Parti.PartiId, islemgrubu.adi, IslemTipi.Adi, Parti.ReceteId, Recete.Adi,
             DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, Parti.ModelNo, Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             Parti.FirmaId, PartiFirma.Adi, LabRecete.EskiLabReceteKodu, RenkDerinligi.Adi, Parti.ParcaliSevk,
             Sum(BoyaIrsaliyeDetay.BrutKilo), Sum(BoyaIrsaliyeDetay.Kilo), Sum(BoyaIrsaliyeDetay.BrutMetre), Sum(BoyaIrsaliyeDetay.Metre),
   (Select sum(PS0.KapAdedi) from Partistok PS0 where PS0.PartiKayitId = Parti.PartiKayitId),
   (Select sum(PS1.Kilo) from Partistok PS1 where PS1.PartiKayitId = Parti.PartiKayitId),
   (Select sum(PS2.netKilo) from Partistok PS2 where PS2.PartiKayitId = Parti.PartiKayitId),
   (Select sum(PS3.BrutSevkKilo) from Partistok PS3 where PS3.PartiKayitId = Parti.PartiKayitId),
   (Select sum(PS4.netSevkKilo) from Partistok PS4 where PS4.PartiKayitId = Parti.PartiKayitId),

   (Select sum(PS1.Metre) from Partistok PS1 where PS1.PartiKayitId = Parti.PartiKayitId),
   (Select sum(PS2.NetMetre) from Partistok PS2 where PS2.PartiKayitId = Parti.PartiKayitId),
   (Select sum(PS3.BrutSevkMetre) from Partistok PS3 where PS3.PartiKayitId = Parti.PartiKayitId),
   (Select sum(PS4.netSevkMetre) from Partistok PS4 where PS4.PartiKayitId = Parti.PartiKayitId),
   (Select Min(PS5.TamamiTartildi) from Partistok PS5 where PS5.PartiKayitId = Parti.PartiKayitId),
   (Select sum(PS6.FireKararMiktari) from Partistok PS6 where PS6.PartiKayitId = Parti.PartiKayitId),
   Desen.OzelYerlesim, Desen.OzelYerlesimNotu, Desen.OzelYerlesimDosyaAdi from BoyaIrsaliye
  left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
  left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
  left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  left join firma PartiFirma on PartiFirma.firmaid = Parti.FirmaId
  left join firma IrsFirma on IrsFirma.firmaid = BoyaIrsaliye.FirmaId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
  left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  left join Desen on Desen.DesenId = DesenVaryant.DesenId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
  where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and
        BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
       (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
       (Parti.PartiTipiID = :APartiTipi or :APartiTipi is null) and
       (Parti.Fason = :AFason or :AFason is null) and
       (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null) and
       (Parti.ReceteId = :AReceteId or :AReceteId is null) and
       (Parti.PartiKayitId = :APartiKayitId or :APartiKayitId is null) and
       (Parti.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
       (IrsFirma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
       (BoyaIrsaliye.AracId = :AAracId or :AAracId is null) and
       (BoyaIrsaliye.UniteId = :AUniteId or :AUniteId is null) and
       (PartiFirma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and BoyaIrsaliyeDetay.PartiSinifId is not null and
       (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
       (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
       (BoyaSiparisDetay.BaskiTipiId = :ABaskiTipiId or :ABaskiTipiId is null) and
       BoyaIrsaliye.HareketTipiId <> :biSatisIade
  Group By BoyaIrsaliye.FirmaId, Parti.PartiKayitId, Parti.Fason, Parti.PartigrupId, PartiGrup.Adi, Parti.PartiYil, Parti.PartiId, Islemgrubu.Adi, IslemTipi.Adi, Parti.ReceteId, Recete.Adi,
           DesenVaryant.DesenId, Desen.Adi, Desen.OzelYerlesim, Desen.OzelYerlesimNotu, Desen.OzelYerlesimDosyaAdi, DesenVaryant.VaryantAdi, Parti.ModelNo, Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
           Parti.FirmaId, PartiFirma.Adi, LabRecete.EskiLabReceteKodu, RenkDerinligi.Adi, Parti.ParcaliSevk
  order by boyairsaliye.firmaid
  into :FirmaId, :PartiKayitId, :Fason, :PartigrupId, :PartiGrupAdi, :PartiYil, :PartiId, :IslemGrubuAdi, :IslemTipiAdi, :ReceteId, :ReceteAdi,
       :DesenId, :DesenAdi, :VaryantAdi, :ModelNo, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi,
       :EskiLabReceteKodu, :RenkDerinligiAdi, :ParcaliSevk,
       :FBrutKilo, :FNetKilo, :FBrutMetre, :FNetMetre,
       :PartilenenKapAdedi, :Brutkilo, :Netkilo, :Brutsevkkilo, :Netsevkkilo, :BrutMetre, :NetMetre, :BrutSevkMetre, :NetSevkMetre, :FTamamiTartildi, :FireKararMiktari,
       :OzelYerlesim, :OzelYerlesimNotu, :OzelYerlesimDosyaAdi do
  begin
    if (SiparisFirmaID <> FirmaId and SiparisFirmaId = FirmaKodumuz) then
    begin
      BrutKilo = FBrutKilo;
      NetKilo = FNetKilo;
      Brutsevkkilo = FBrutKilo;
      NetSevkKilo = FNetKilo;

      BrutMetre = FBrutMetre;
      NetMetre = FNetMetre;
      BrutsevkMetre = FBrutMetre;
      NetSevkMetre = FNetMetre;
    end

    if (FTamamiTartildi = 1) then
    begin
      FireMiktari = BrutKilo - NetKilo;

      if (BrutKilo > 0) then
      begin
        Fire = (BrutKilo - NetKilo) * 100 / BrutKilo;
      end
      else
      begin
        Fire = 0;
      end
    end
    else
    begin
      Fire = Null;
      FireMiktari = Null;
    end

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') '  || EgemenIntegerToStr(PartiId);
    suspend;
  end

  PartilenenKapAdedi = Null;


  if (AMamulDepoSevkleriDahil = 1 and AUniteId is not null) then
  begin
    MusteriyeSevk = 0;

    For Select Parti.FirmaId, Parti.PartiKayitId, Parti.Fason, Parti.PartigrupId, PartiGrup.Adi, Parti.PartiYil, Parti.PartiId, islemgrubu.adi, IslemTipi.Adi, Parti.ReceteId, Recete.Adi, Parti.ModelNo,
               DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, LabRecete.EskiLabReceteKodu, RenkDerinligi.Adi,
               Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
               BoyaSiparis.FirmaId, Firma.Adi, Parti.ParcaliSevk,
     (Select sum(PS1.Kilo) from Partistok PS1 where PS1.PartiKayitId = Parti.PartiKayitId),
     (Select sum(PS2.netKilo) from Partistok PS2 where PS2.PartiKayitId = Parti.PartiKayitId),
     (Select sum(PS3.BrutSevkKilo) from Partistok PS3 where PS3.PartiKayitId = Parti.PartiKayitId),
     (Select sum(PS4.netSevkKilo) from Partistok PS4 where PS4.PartiKayitId = Parti.PartiKayitId),

     (Select sum(PS1.Metre) from Partistok PS1 where PS1.PartiKayitId = Parti.PartiKayitId),
     (Select sum(PS2.netMetre) from Partistok PS2 where PS2.PartiKayitId = Parti.PartiKayitId),
     (Select sum(PS3.BrutSevkMetre) from Partistok PS3 where PS3.PartiKayitId = Parti.PartiKayitId),
     (Select sum(PS4.netSevkMetre) from Partistok PS4 where PS4.PartiKayitId = Parti.PartiKayitId),
     (Select Min(PS5.TamamiTartildi) from Partistok PS5 where PS5.PartiKayitId = Parti.PartiKayitId),
     (Select sum(PS6.FireKararMiktari) from Partistok PS6 where PS6.PartiKayitId = Parti.PartiKayitId),
     Desen.OzelYerlesim, Desen.OzelYerlesimNotu, Desen.OzelYerlesimDosyaAdi from MamulDepoTransferFisi
    left join Depo D1 on D1.DepoId = MamulDepoTransferFisi.DepoId
    left join Depo D2 on D2.DepoId = MamulDepoTransferFisi.TransferDepoId
    left join Unite U1 on U1.UniteId = D1.UniteId
    left join Unite U2 on U2.UniteId = D2.UniteId
    left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiID = MamulDepoTransferFisi.MamulDepoTransferFisiId
    left join PartiStokTarti on PartiStokTarti.PartiStokTartiID = MamulDepoTransferFisiDetay.PartiStokTartiId
    left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join firma on firma.firmaid = BoyaSiparis.FirmaId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join LabRecete on LabRecete.LabReceteId = Recete.LabReceteId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    where MamulDepoTransferFisi.IrsaliyeTarihi >= :ABaslangicTarihi and
          MamulDepoTransferFisi.IrsaliyeTarihi <= :ABitisTarihi and MamulDepoTransferFisi.SistemKaydi = 0 and
         (D1.UniteId = :AUniteId or :AUniteId is null) and
         (D2.UniteId <> :AUniteId or :AUniteId is null) and
         (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
         (Parti.PartiTipiID = :APartiTipi or :APartiTipi is null) and
         (Parti.Fason = :AFason or :AFason is null) and
         (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
         (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null) and
         (MamulDepoTransferFisi.AracId = :AAracId or :AAracId is null) and
         (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and MamulDepoTransferFisiDetay.PartiSinifId is not null and
         (Parti.ReceteId = :AReceteId or :AReceteId is null) and
         (Parti.PartiKayitId = :APartiKayitId or :APartiKayitId is null) and
         (Parti.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
         (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
         (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
         (BoyaSiparisDetay.BaskiTipiId = :ABaskiTipiId or :ABaskiTipiId is null)
    Group By Parti.FirmaId, Parti.PartiKayitId, Parti.Fason, Parti.PartigrupId, PartiGrup.Adi, Parti.PartiYil, Parti.PartiId, Islemgrubu.Adi, IslemTipi.Adi, Parti.ReceteId, Recete.Adi, Parti.ModelNo,
             DesenVaryant.DesenId, Desen.Adi, Desen.OzelYerlesim, Desen.OzelYerlesimNotu, Desen.OzelYerlesimDosyaAdi, DesenVaryant.VaryantAdi, LabRecete.EskiLabReceteKodu, RenkDerinligi.Adi, Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             BoyaSiparis.FirmaId, Firma.Adi, Parti.ParcaliSevk
    order by Parti.firmaid
    into :FirmaId, :PartiKayitId, :Fason, :PartigrupId, :PartiGrupAdi, :PartiYil, :PartiId, :IslemGrubuAdi, :IslemTipiAdi, :ReceteId, :ReceteAdi, :ModelNo,
         :DesenId, :DesenAdi, :VaryantAdi, :EskiLabReceteKodu, :RenkDerinligiAdi, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi,
         :ParcaliSevk, :Brutkilo, :Netkilo, :Brutsevkkilo, :Netsevkkilo, :BrutMetre, :NetMetre, :BrutsevkMetre, :NetsevkMetre, :FTamamiTartildi, :FireKararMiktari,
         :OzelYerlesim, :OzelYerlesimNotu, :OzelYerlesimDosyaAdi  do
    begin
      if (FTamamiTartildi = 1) then
      begin
        FireMiktari = BrutKilo - NetKilo;
        if (BrutKilo > 0) then Fire = (BrutKilo - NetKilo) * 100 / BrutKilo;
        else Fire = 0;
      end
      else
      begin
        Fire = Null;
        FireMiktari = Null;
      end

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') '  || EgemenIntegerToStr(PartiId);
      suspend;
    end
  end
end
