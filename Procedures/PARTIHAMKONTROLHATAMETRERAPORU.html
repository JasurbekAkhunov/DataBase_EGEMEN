CREATE PROCEDURE PARTIHAMKONTROLHATAMETRERAPORU(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    PSHAMKONTROLID INTEGER,
    TOPNO INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    HAMHATATIPIID INTEGER,
    HAMHATATIPIADI VARCHAR(20),
    BASLANGICMETRE DOUBLE PRECISION,
    BITISMETRE DOUBLE PRECISION)
AS
begin
  for Select PSHamKontrol.PartiSinifId, PSHamKontrol.PSHamKontrolId, PSHamKontrol.TopNo, PSHamKontrol.Kilo, PSHamKontrol.Metre,
             PSHamKontrolHataMetre.HamHataTipiId, HamHataTipi.Adi, PSHamKontrolHataMetre.BaslangicMetre, PSHamKontrolHataMetre.BitisMetre from PSHamKontrol
  left join PSHamKontrolHataMetre on PSHamKontrolHataMetre.PSHamKontrolId = PSHamKontrol.PSHamKontrolId
  left join HamHataTipi on HamHataTipi.HamHataTipiId  = PSHamKontrolHataMetre.HamHataTipiId
  where PSHamKontrol.PartiKayitId = :APartiKayitId and (PSHamKontrol.PartiSinifId = :APartiSinifId or :APartiSinifId is null)
  order by PSHamKontrol.PartiSinifId, PSHamKontrol.PSHamKontrolId, PSHamKontrol.TopNo, PSHamKontrolHataMetre.BaslangicMetre
  into :PartiSinifId, :PSHamKontrolId, :TopNo, :Kilo, :Metre, :HamHataTipiId, :HamHataTipiAdi, :BaslangicMetre, :BitisMetre do suspend;
end
