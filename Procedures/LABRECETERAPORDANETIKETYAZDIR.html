CREATE PROCEDURE LABRECETERAPORDANETIKETYAZDIR(
    ALABRECETEATISIDSTR VARCHAR(250))
RETURNS (
    LABRECETEID INTEGER,
    LABRECETECALISMAID INTEGER,
    LABRECETEATISID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TALEPTARIHI TIMESTAMP,
    LABRECETEGRUPID INTEGER,
    LABRECETEGRUPADI VARCHAR(20),
    LABRENKGRUPID INTEGER,
    LABRENKGRUPADI VARCHAR(30),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI VARCHAR(20),
    RECETETURUID INTEGER,
    RECETETURUADI VARCHAR(15),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    MUSTERIRENKKODU VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    ACIKLAMA VARCHAR(40),
    ACIKLAMA2 VARCHAR(40),
    ESKILABRECETEKODU VARCHAR(20),
    PANTONEKODU VARCHAR(20),
    CALISMATARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    RPTMI INTEGER,
    ILAVE INTEGER,
    KRITIKCALISMASI INTEGER,
    SOKUMUZERIBOYA INTEGER,
    ATISNOSTR VARCHAR(20),
    GIDISTARIHI TIMESTAMP,
    KARTELATARIHI TIMESTAMP,
    DURUMUSTR VARCHAR(6),
    TARIHI TIMESTAMP,
    OKEY INTEGER,
    KRITIK INTEGER,
    CALISMANOTU VARCHAR(1024),
    BOYAYUZDESI DOUBLE PRECISION,
    BOYAMALIYETI DOUBLE PRECISION,
    KIMYASALMALIYETI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    LABPROGRAMNO INTEGER,
    LABPROGRAMNO2 INTEGER,
    TUZMIKTARI DOUBLE PRECISION,
    SODAMIKTARI DOUBLE PRECISION,
    RECETEONISLEMTIPIADI VARCHAR(30),
    PIPETLEMEKODU VARCHAR(30),
    BOYATIPIFARKLIURUNVAR INTEGER,
    PIPETLEMEYAPANKULLANICIID INTEGER,
    PIPETLEMEYAPANKULLANICIADI VARCHAR(40))
AS
declare variable VirgulPos integer;
declare variable Str varchar(30);

declare variable FCalismaNotu Blob;
declare variable BlobSize integer;
declare variable ReceteId varchar(20);
declare variable TuzKimyasalTipiId integer;
declare variable SodaKimyasalTipiID integer;
declare variable MinBoyaTipiId  integer;
declare variable MaxBoyaTipiId  integer;
begin

  Select TuzKimyasalTipiID, SodaKimyasalTipiID from sistem where sistemid = 1 into :TuzKimyasalTipiID, :SodaKimyasalTipiID;

  VirgulPos = PosBul(ALabReceteAtisIdStr, ',');

  While (VirgulPos > 1) do
  begin
    Str = EgemenCopyStr(ALabReceteAtisIdStr, 1, VirgulPos -1);
    LabReceteAtisId = EgemenIntegerToStr(Str);

    Select LabRecete.LabReceteId, LabReceteCalisma.LabReceteCalismaId, LabRecete.LabReceteKodu, LabRecete.Adi, LabRecete.FirmaId, Firma.Adi, LabRecete.TalepTarihi,
                   LabRecete.LabReceteGrupId, LabReceteGrup.Adi, LabRecete.LabRenkGrupId, LabRenkGrup.Adi,  LabRecete.RenkDerinligiId, RenkDerinligi.Adi,
                   LabRecete.ReceteTuruId, ReceteTuru.Adi, LabRecete.Fason, LabRecete.MusteriRenkKodu, LabReceteCalisma.HamId, Ham.Adi,
                   LabRecete.Aciklama, LabRecete.Aciklama2, LabRecete.EskiLabReceteKodu, LabRecete.PantoneKodu,
                   LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi, LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave,
                   LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, LabReceteAtis.AtisNoStr, LabReceteAtis.GidisTarihi, LabReceteAtis.KartelaTarihi,
                   LabReceteAtis.DurumuStr, LabReceteAtis.Tarihi, LabReceteAtis.Okey, LabReceteAtis.Kritik,
                   LabReceteCalisma.Notu, LabReceteAtis.BoyaYuzdesi, LabReceteAtis.BoyaMaliyeti, LabReceteAtis.KimyasalMaliyeti, LabRecete.DovizId,
                   LabReceteCalisma.LabProgramNo, LabReceteCalisma.LabProgramNo2, ReceteOnIslemTipi.Adi,
                   LabReceteCalisma.PipetlemeYapanKullaniciId, PYK.Adi from LabReceteAtis
        left join LabRecete On LabRecete.LabReceteId = LabReceteAtis.LabReceteId
        left join LabReceteCalisma On LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
        left join Firma On Firma.FirmaId = LabRecete.FirmaId
        left join LabReceteGrup On LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
        left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = LabRecete.LabRenkGrupId
        left join ReceteTuru On ReceteTuru.ReceteTuruId = LabRecete.ReceteTuruId
        left join RenkDerinligi On RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
        left join Ham On Ham.HamId = LabReceteCalisma.HamId
        left join ReceteOnIslemTipi on ReceteOnIslemTipi.ReceteOnIslemTipiID = LabReceteCalisma.ReceteOnIslemTipiId
        Left Join Kullanici PYK on PYK.KullaniciId = LabReceteCalisma.PipetlemeYapanKullaniciId
        where LabReceteAtis.LabReceteAtisID = :LabReceteAtisID
        into :LabReceteId, :LabReceteCalismaId, :LabReceteKodu, :LabReceteAdi, :FirmaId, :FirmaAdi, :TalepTarihi, :LabReceteGrupId, :LabReceteGrupAdi, :LabRenkGrupId, :LabRenkGrupAdi,
             :RenkDerinligiId, :RenkDerinligiAdi, :ReceteTuruId, :ReceteTuruAdi, :Fason, :MusteriRenkKodu, :HamId, :HamAdi,
             :Aciklama, :Aciklama2, :EskiLabReceteKodu, :PantoneKodu, :CalismaTarihi, :TerminTarihi, :Rptmi, :Ilave, :SokumUzeriBoya, :KritikCalismasi,
             :AtisNoStr, :GidisTarihi, :KartelaTarihi, :DurumuStr, :Tarihi, :Okey, :Kritik, :FCalismaNotu,
             :BoyaYuzdesi, :BoyaMaliyeti, :KimyasalMaliyeti, :DovizId, :LabProgramNo, :LabProgramNo2, :ReceteOnIslemTipiAdi,
             PipetlemeYapanKullaniciId, :PipetlemeYapanKullaniciAdi;

    if (Fason = 0) then FasonStr = 'Satış';
    else FasonStr = 'Fason';

    BlobSize = char_length(FCalismaNotu);

    if (BlobSize > 1023) then CalismaNotu = Left(FCalismaNotu, 1023);
    else if (BlobSize > 0) then CalismaNotu = Left(FCalismaNotu, BlobSize);
    else CalismaNotu = '';

    if (TuzKimyasalTipiID is not null) then
    begin
      Select Sum(LabReceteUrun.MiktarYuzde) from LabReceteUrun
      left join Urun On Urun.UrunID = LabReceteUrun.UrunId
      where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and LabReceteUrun.UrunID is not null and Urun.KimyasalTipiID = :TuzKimyasalTipiID
      into :TuzMiktari;
    end

    if (SodaKimyasalTipiID is not null) then
    begin
      Select Sum(LabReceteUrun.MiktarYuzde) from LabReceteUrun
      left join Urun On Urun.UrunID = LabReceteUrun.UrunId
      where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and LabReceteUrun.UrunID is not null and Urun.KimyasalTipiID = :SodaKimyasalTipiID
      into :SodaMiktari;
    end

    PipetlemeKodu = EgemenIntegerToStr(LabReceteCalismaId) || '.' || AtisNoStr;

    BoyaTipiFarkliUrunVar = 0;


    MinBoyaTipiId = 0;
    MaxBoyaTipiId = 0;

    Select Min(BoyaTipiId), Max(BoyaTipiId) from LabReceteUrun
      left join Urun on Urun.UrunId = LabReceteUrun.UrunId
      where LabReceteUrun.LabReceteAtisId = :LabReceteAtisId and Urun.UrunTipiID = 0
      into :MinBoyaTipiId, :MaxBoyaTipiId;

    if (MinBoyaTipiId <> MaxBoyaTipiId) then BoyaTipiFarkliUrunVar = 1;


    suspend;

    Update LabReceteAtis Set EtiketYazdirildi = 1 where LabReceteAtis.LabReceteAtisID = :LabReceteAtisID and LabReceteAtis.EtiketYazdirildi = 0;

    if (Char_Length(ALabReceteAtisIdStr) > VirgulPos) then ALabReceteAtisIdStr = EgemenCopyStr(ALabReceteAtisIdStr, VirgulPos + 1, Char_Length(ALabReceteAtisIdStr) - VirgulPos);
    else ALabReceteAtisIdStr = '';

    VirgulPos = PosBul(ALabReceteAtisIdStr, ',');
  end
end
