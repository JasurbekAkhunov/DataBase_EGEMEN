CREATE PROCEDURE KARTELAFISILISTESI(
    AGIRIS INTEGER,
    ABASLANGICFISTARIHI TIMESTAMP,
    ABITISFISTARIHI TIMESTAMP,
    ABASLANGICFISYIL INTEGER,
    ABASLANGICFISNO INTEGER,
    ABITISFISYIL INTEGER,
    ABITISFISNO INTEGER)
RETURNS (
    KARTELAFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select KartelaFisi.KartelaFisiId, KartelaFisi.FisYil, KartelaFisi.FisNo, KartelaFisi.Fistarihi, KartelaFisi.FirmaId, Firma.Adi, KartelaFisi.Notu,
             Ik.Adi, KartelaFisi.InsertTarihi, Sk.Adi, KartelaFisi.DegisimTarihi from KartelaFisi
    Left Join Firma on Firma.FirmaId = KartelaFisi.FirmaId
    Left Join Kullanici IK on IK.KullaniciId = KartelaFisi.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = KartelaFisi.KullaniciId
    where KartelaFisi.FisTarihi >= :ABaslangicFisTarihi and KartelaFisi.FisTarihi <= :ABitisFistarihi and
         (:ABaslangicFisYil is null or (KartelaFisi.FisYil > :ABaslangicFisYil or (KartelaFisi.FisYil = :ABaslangicFisYil and KartelaFisi.FisNo >= :ABaslangicFisNo))) and
         (:ABitisFisYil is null or (KartelaFisi.FisYil < :ABitisFisYil or (KartelaFisi.FisYil = :ABitisFisYil and KartelaFisi.FisNo <= :ABitisFisNo))) and
         KartelaFisi.Giris = :AGiris
    into :KartelaFisiId, :FisYil, :FisNo, :Fistarihi, :FirmaId, :FirmaAdi,
         :Notu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';
    suspend;
  end
end
