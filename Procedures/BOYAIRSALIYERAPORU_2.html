declare variable HamId              integer;
declare variable hstIplik           integer;
declare variable FBrutMiktar        Numeric(12,3);
declare variable FMiktar            Numeric(12,3);
declare variable IrsaliyeTarihi     TimeStamp;
declare variable SevkTarihi         TimeStamp;
declare variable hstElyaf           integer;
declare variable hstTelef           integer;
declare variable ABoyaSiparisKayitId integer;
declare variable FFaturaKesilecek   integer;
declare variable FTutar             Double Precision;
declare variable FKilo              Double Precision;
declare variable FToplamTutar       Double Precision;

declare variable HamStokAlisDovizId DovizSize;
declare variable DovizKuru          Double Precision;
declare variable SistemDovizId      DovizSize;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;
  Select SistemDovizId from Sistem where sistemId = 1 into :SistemDovizId;

  FirmaId  = Null;
  FirmaAdi = NUll;

  if (ABoyaSiparisYil is not Null) then
  begin
    Select BoyaSiparis.BoyaSiparisKayitId from BoyaSiparis
      where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId
      into :ABoyaSiparisKayitId;
    if (ABoyaSiparisKayitId is null) then ABoyaSiparisKayitId = -1;
  end

  if (ARaporTarihTipi <> 0) then ABitisTarihi = ABitisTarihi + 1;

  if (ARaporTarihTipi = 0) then
  begin
    for select BoyaIrsaliyeDetay.BoyaIrsaliyeId, BoyaIrsaliyeDetay.HareketTipiId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliyeDetay.PartiKayitId, Parti.PartiYil,

               BoyaSiparisDetay.FasonBirimFiyati, BoyaSiparisDetay.FasonBirimFiyatDovizId, BoyaSiparisDetay.AltBezHamId, HamAltBez.Adi,
               BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.OzelTestler, BoyaSiparisDetay.OzelApreler, SiparisProses.Adi, Olcu.OlcuAdi, PartiStok.PartiSirano, BoyaSiparisDetay.VadeGun from BoyaIrsaliye

      left join Firma IrsaliyeFirma On IrsaliyeFirma.FirmaId = BoyaIrsaliye.FirmaId

      left join Recete On Recete.ReceteId = Parti.ReceteId

      left join Renk OrmeRenk On OrmeRenk.RenkId = OrmeSiparisDetay.RenkId
      left join BoyaRezervFisiDetay on BoyaRezervFisiDetay.BoyaRezervFisiDetayId = BoyaIrsaliyeDetay.BoyaRezervFisiDetayId
      left join BoyaSiparisDetay RS on RS.BoyaSiparisDetayId = BoyaRezervFisiDetay.BoyaSiparisDetayId
      left join MiktarTipi On MiktarTipi.MiktarTipiId = Coalesce(RS.FaturaEdilecekMiktarTipi, BoyaSiparisDetay.FaturaEdilecekMiktarTipi)
      left join Ham HamAltBez on HamAltBez.HamId = BoyaSiparisDetay.AltBezHamId

      left join Olcu on Olcu.OlcuId = BoyaIrsaliyeDetay.OlcuId

            (BoyaIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
            (:AFaturaDurumu is null or (:AFaturaDurumu = 0 and BoyaIrsaliyeDetay.BoyaFaturaId is null) or (:AFaturaDurumu = 1 and BoyaIrsaliyeDetay.BoyaFaturaId is not null)) and

            (BoyaIrsaliye.HareketTipiId = :AIrsaliyeTipi or :AIrsaliyeTipi is null) and
            (BoyaIrsaliye.Sinif = :ASinif or :ASinif is null) and
            (BoyaIrsaliye.FaturaKesilecek = :AFaturaKesilecek or :AFaturaKesilecek is Null) and
            (Coalesce(RS.MusteriSiparisNo, BoyaSiparisDetay.MusteriSiparisNo) = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
            (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
            BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId is not null and
            (Parti.Fason = :ASatisTipi or :ASatisTipi is null) and
            (Parti.BoyaSiparisKayitID = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
            (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
            (BoyaSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
            ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporTipi = 1) or :ARaporTipi is null) and
            ( (:AIslemTipi = 0 and Parti.IslemTipiID <> 3) or
              (:AIslemTipi = 1 and Parti.IslemTipiID = 3) or
              (:AIslemTipi is null) ) and
            ( (BoyaIrsaliyeDetay.PartiSinifID is not null and :AMamulSevk = 1) or
              (BoyaIrsaliyeDetay.HamIadeRezervId is not null and :AHamIade = 1) or
              (BoyaIrsaliyeDetay.PartiSinifID is null and BoyaIrsaliyeDetay.HamIadeRezervId is null and :AHamSatis = 1))
      order by BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.BoyaIrsaliyeId
      into :BoyaIrsaliyeId, :HareketTipiId, :BoyaIrsaliyeDetayId, :PartiKayitId, :PartiYil, :PartiId, :PartiFason, :FirmaId, :FirmaAdi,
           :IslemTipiId, :IslemTipiAdi, :TamirTipiAdi,
           :IslemGrubuId, :IslemGrubuAdi, ImalatTipiId, :ReceteId, :ReceteAdi,  :PartiSinifId, :HamAdiToplami, :FHamAdi, :HamAdi2, :HamAdi3, :HamAdi4,
           :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :BoyaSiparisDetayId,
           :BrutKapAdedi, :BrutKilo, :KapAdedi, :Kilo, :Adet, :FBrutMiktar, :FMiktar, :BoyaFaturaId, :YapilanIslemler, :Metre, :BrutMetre,
           :HamId, :HamIadeRezervId, :FasonHamIade, :MusteriSiparisNo, :ReferansSiparisNo, :KullaniciAdi, :Degisimtarihi,
           :IrsaliyeTarihi, :SevkTarihi, :PartiGrupId, :LabReceteKodu, :RenkAdi, :LabAtisRenkDerinligiAdi, :LabReceteRenkDerinligiAdi,
           :OrmeRenkId, :OrmeRenkAdi, :MelanjKodu, :FFaturaKesilecek, :PartiStokTamamiTartildi, :SarimAciklama, :RecetePantoneKodu, :FaturaEdilecekMiktarTipi,
           :FaturaEdilecekMiktarTipiStr, :PartiBirimFiyat, :PartiDovizId, :SiparisFasonBirimFiyati, :SiparisFasonBFDovizId, :AltBezHamId, :AltBezHamAdi,
           :SiparisIlaveIslemler, :SiparisOzelTestler, :SiparisOzelApreler, :SiparisProsesAdi, :OlcuAdi, :PartiSirano, :Vadegun do
    begin
      Dara = Null;
      BobinSayisi = Null;

      select DovizKuru from EnYakinDovizKuruGetir(:partidovizid, :SistemDovizId, :IrsaliyeTarihi) into :irsaliyetarihidovizkuru;

      FaturaDurumuStr = 'Faturalanmam';
      if (BoyaFaturaId is not null) then
        FaturaDurumuStr = 'Faturalanm';

      if (FFaturaKesilecek = 1) then
        FaturaKesilecekStr = 'Fatura Kesilecek';
