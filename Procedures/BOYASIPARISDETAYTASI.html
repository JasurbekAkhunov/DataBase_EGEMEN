CREATE PROCEDURE BOYASIPARISDETAYTASI(
    ABOYASIPARISDETAYID INTEGER,
    AYENIBOYASIPARISKAYITID INTEGER,
    AKULLANICIID INTEGER)
AS
Declare Variable FOrmeSiparisDetayId Integer;
Declare Variable FEskiBoyaSiparisKayitId Integer;
Declare Variable FKayitSayisi Integer;
Declare Variable FKapali Smallint;
begin
    Select BoyaSiparisDetay.BoyaSiparisKayitId, BoyaSiparisDetay.OrmeSiparisDetayId, BoyaSiparis.Kapali From BoyaSiparisDetay
    Join BoyaSiparis On BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
    Where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId
    Into :FEskiBoyaSiparisKayitId, :FOrmeSiparisDetayId, :FKapali;

    If (FEskiBoyaSiparisKayitId is null) Then Exception KayitHatasi 'Sipariş detay kaydı bulunamadı...';
    If (FOrmeSiparisDetayId is not null) Then Exception KayitHatasi 'Örme siparişine bu işlem uygulanamaz...';
    If (FKapali = 1) Then Exception KayitHatasi 'Kapalı siparişe bu işlem uygulanamaz...';

    Select Count(*) From PartiStok Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId Into :FKayitSayisi;
    if (FKayitSayisi > 0) Then Exception KayitHatasi 'Partisi açılmış sipariş taşınamaz...';

    Select Count(*) From BoyaSiparisDetay Where BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId Into :FKayitSayisi;
    if (FKayitSayisi = 0) Then Exception KayitHatasi 'Aktarılacak sipariş kaydı bulunamadı...';

    Select Kapali From BoyaSiparis Where BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId Into :FKapali;
    If (FKapali = 1) Then Exception KayitHatasi 'Aktarılacak sipariş kapalı olduğu için işlem uygulanamaz...';

    Update BoyaSiparisDetay Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update BoyaSiparisDetayMasraf Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update BoyaSiparisDetayOngoru Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update BoyaSiparisRezervDetay Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update BoyaSiparisRezerv Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update BoyaSiparisRezervSertifika Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update BoyaSiparisSarj Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update BoyaTalimat Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update HamStokMaster Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update IplikTalepDetay Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update OrmeRaf Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update OrmeStokKaydirma Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update SatinAlmaTalepDetay Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update StokHareket Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update VerilenSiparisDetay Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update BoyaSiparisBaski Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update SiparisIlaveIslemler Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update SiparisOzelTestler Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
    Update SiparisOzelApreler Set BoyaSiparisKayitId = :AYeniBoyaSiparisKayitId, KullaniciId = :AKullaniciId Where BoyaSiparisDetayId = :ABoyaSiparisDetayId and BoyaSiparisKayitId = :FEskiBoyaSiparisKayitId;
end
