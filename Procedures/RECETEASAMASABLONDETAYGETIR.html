CREATE PROCEDURE RECETEASAMASABLONDETAYGETIR(
    ARECETEASAMASABLONID INTEGER)
RETURNS (
    RECETEASAMASABLONDETAYID INTEGER,
    SIRA INTEGER,
    RECETEALTASAMAID INTEGER,
    RECETEALTASAMAADI VARCHAR(40),
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    URUNTIPIADI VARCHAR(20),
    BOYATIPIADI VARCHAR(20),
    URUNBIRIMID INTEGER,
    BIRIMID VARCHAR(2),
    MIKTAR DOUBLE PRECISION,
    BANYO DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select ReceteAsamaSablonDetay.ReceteAsamaSablonDetayId, ReceteAsamaSablonDetay.Sira, ReceteAsamaSablonDetay.ReceteAltAsamaId,
             ReceteAltAsama.Adi, ReceteAsamaSablonDetay.UrunId, Urun.Adi,
             ReceteAsamaSablonDetay.UrunBirimId, UrunBirim.BirimId, ReceteAsamaSablonDetay.Miktar, ReceteAsamaSablonDetay.Banyo,
             UrunTipi.Adi, BoyaTipi.Adi, IK.Adi, ReceteAsamaSablonDetay.InsertTarihi, SK.Adi, ReceteAsamaSablonDetay.DegisimTarihi from ReceteAsamaSablonDetay
  left join Urun on Urun.UrunId = ReceteAsamaSablonDetay.UrunId
  left join UrunBirim on UrunBirim.UrunBirimId = ReceteAsamaSablonDetay.UrunBirimId and UrunBirim.UrunId = ReceteAsamaSablonDetay.UrunId
  left join ReceteAltAsama on ReceteAltAsama.ReceteAltAsamaId = ReceteAsamaSablonDetay.ReceteAltAsamaId
  left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
  left join BoyaTipi on BoyaTipi.BoyaTipiId = Urun.BoyaTipiId
  left join Kullanici IK on IK.KullaniciId = ReceteAsamaSablonDetay.KullaniciId
  left join Kullanici SK on SK.KullaniciId = ReceteAsamaSablonDetay.KullaniciId
  where ReceteAsamaSablonDetay.ReceteAsamaSablonId = :AReceteAsamaSablonId order by sira
  into :ReceteAsamaSablonDetayId, :Sira, :ReceteAltAsamaId, :ReceteAltAsamaAdi,
       :UrunId, :UrunAdi, :UrunBirimId, :BirimId, :Miktar, :Banyo, :UrunTipiAdi, :BoyaTipiAdi,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do suspend;
end
