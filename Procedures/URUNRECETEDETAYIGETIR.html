CREATE PROCEDURE URUNRECETEDETAYIGETIR(
    AURUNID VARCHAR(20),
    ATARIH TIMESTAMP)
RETURNS (
    URUNRECETEDETAYID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    DEPOID VARCHAR(10),
    DEPOADI VARCHAR(30),
    BIRIMID VARCHAR(2),
    MIKTAR DOUBLE PRECISION,
    URETIMECIKISFIYATI DOUBLE PRECISION,
    URETIMECIKISFIYATIBUGUN DOUBLE PRECISION,
    URETIMECIKISFIYATIBIRIMID VARCHAR(2),
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable DovizKuru                   Double Precision;
declare variable FUretimeCikisFiyati   Double precision;
declare variable FUretimeCikisFiyatiDovizId  varchar(5);
declare variable StokTabanDovizId               varchar(5);
begin
/* PartiSablonRecetesiTanitimi ekranından da kullanıyoruz, oradan Emulsiyonunun recetesini cekiyoruz */
  select StokTabanDovizId from SISTEM where SISTEMID = 1 into :StokTabanDovizId;

  for select UrunReceteDetay.UrunReceteDetayId, UrunReceteDetay.UrunId, urun.Adi, URDUB.BirimId,
             UrunReceteDetay.DepoId, Depo.Adi, UrunReceteDetay.Miktar,
             UrunFiyat.UretimeCikisFiyati, UrunFiyat.UretimeCikisFiyatiDovizId, UBUCF.BirimId,
             Kullanici.Adi, UrunReceteDetay.DegisimTarihi  from UrunReceteDetay
    left join UrunBirim URDUB on URDUB.urunBirimId = UrunReceteDetay.urunBirimId
    left join Urun on Urun.UrunId = UrunReceteDetay.UrunId
    left join Depo on Depo.DepoId = UrunReceteDetay.DepoId
    left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
    left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
    left join Kullanici on Kullanici.KullaniciId = UrunReceteDetay.KullaniciId
    where UrunReceteDetay.UrunReceteUrunId = :AUrunId
    into :UrunReceteDetayId, :UrunId, :UrunAdi, :BirimId, :DepoId, :DepoAdi, :Miktar,
         :FUretimeCikisFiyati, :FUretimeCikisFiyatiDovizId, :UretimeCikisFiyatiBirimId, :KullaniciAdi, :DegisimTarihi do
  begin
    if (FUretimeCikisFiyatiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :StokTabanDovizId, Current_Date) into :DovizKuru;
    else DovizKuru = 1;

    if (DovizKuru is null) then DovizKuru = 0;
    UretimeCikisFiyatiBugun = FUretimeCikisFiyati * DovizKuru;

    if (ATarih is not null) then
    begin
      if (FUretimeCikisFiyatiDovizId <> StokTabanDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :StokTabanDovizId, :ATarih) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null) then DovizKuru = 0;
      UretimeCikisFiyati = FUretimeCikisFiyati * DovizKuru;
    end
    else UretimeCikisFiyati = UretimeCikisFiyatiBugun;

    suspend;
  end
end
