CREATE PROCEDURE ORMETALIMATRECETEURUNLERI(
    AORMEPLANLAMAID INTEGER)
RETURNS (
    RESULTID VARCHAR(255),
    ORMEPLANLAMAID INTEGER,
    ORMEPLANLAMARECETEID INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(150),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION)
AS
begin
  For select OrmePlanlamaRecete.OrmePlanlamaId, OrmePlanlamaRecete.OrmePlanlamaReceteId, OrmePlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, OrmePlanlamaRecete.RenkId, Renk.Adi, OrmePlanlamaRecete.CikisMiktari, OrmePlanlamaRecete.IadeMiktari,
             OrmePlanlamaRecete.MelanjKodu, OrmePlanlamaRecete.Lot from OrmePlanlamaRecete
  left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
  left join Renk on Renk.RenkId = OrmePlanlamaRecete.RenkId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  where OrmePlanlamaRecete.OrmePlanlamaId = :AOrmePlanlamaId
  into :OrmePlanlamaId, :OrmePlanlamaReceteId, :HamId, :HamTipiId, :HamTipiAdi, :HamAdi, :RenkId, :RenkAdi,
       :CikisMiktari, :IadeMiktari, :MelanjKodu, :Lot do
  begin
    ResultId = '"' || EgemenIntegerToStr(HamId) || '",' ;

    if (RenkId is null) then
      ResultId = ResultId || '"",' ;
    else
      ResultId = ResultId || '"' || RenkId || '",' ;

    if (MelanjKodu is null) then
      ResultId = ResultId || '"",' ;
    else
      ResultId = ResultId || '"' || MelanjKodu || '",' ;

    if (Lot is null or Lot = '') then
      ResultId = ResultId || '""' ;
    else
      ResultId = ResultId || '"' || Lot || '"' ;

    suspend;
  end
end
