CREATE PROCEDURE DEPOSAYIMDANRAFTRANSFERIYAP(
    ADEPOSAYIMID INTEGER,
    AORMERAFID INTEGER,
    AKULLANICIID INTEGER,
    AMIKTAR DOUBLE PRECISION,
    ABRUTMIKTAR DOUBLE PRECISION,
    AKAPADEDI INTEGER)
AS
declare variable HedefRaf varchar(20);
declare variable FDepoSayimFisNo varchar(40);
declare variable FOrmeRafTransferFisiId integer;
declare variable FOrmeRafTransferFisiDetayId integer;
declare variable FFisYil integer;
declare variable FFisNo integer;
declare variable shtAlis        SmallInt;
declare variable shtFasonUretim SmallInt;
declare variable shtUretim      SmallInt;
declare variable FBarkod Varchar(40);
declare variable FStokHareketDetayID integer;

declare variable HamID           integer;
declare variable RenkID          Integer;
declare variable Lot             varchar(150);
declare variable MelanjKodu      varchar(20);
declare variable FlamKodu        varchar(20);
declare variable KaliteNo        Integer;
declare variable DesenVaryantID  Integer;
declare variable DesenID         DesenKoduSize;

begin
  Select shtAlis, shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir into :shtAlis, :shtFasonUretim, :shtUretim;

  Select DepoSayim.RafID, DepoSayim.SeriNo || '-' || DepoSayim.FisNo from DepoSayim
    where DepoSayim.DepoSayimID = :ADepoSayimID
  into :HedefRaf, :FDepoSayimFisNo;

  FFisYil = f_Year(Current_Date);
  select id from OrmeRafTransferFisiNumaratorAl(:FFisYil) into :FFisNo;
  FOrmeRafTransferFisiId = gen_ID(OrmeRafTransferFisi_Id_gen, 1);

  Insert Into OrmeRafTransferFisi(OrmeRafTransferFisiId, FisYil, FisNo, FisTarihi, Notu, RafId, InsertKullaniciId, KullaniciId)
                         Values(:FOrmeRafTransferFisiId, :FFisYil, :FFisNo, Current_Date, 'Depo Sayım Fiş No: ' || :FDepoSayimFisNo , :HedefRaf, :AKullaniciID, :AKullaniciID);


  Select OrmeRaf.HamID, OrmeRaf.RenkId, OrmeRaf.Lot, OrmeRaf.MelanjKodu, OrmeRaf.FlamKodu, OrmeRaf.KaliteNo, OrmeRaf.DesenId, OrmeRaf.DesenVaryantId from OrmeRaf where OrmeRafID = :AOrmeRafID
  into :HamID, :RenkId, :Lot, :MelanjKodu, :FlamKodu, :KaliteNo, :DesenId, :DesenVaryantId;

  FOrmeRafTransferFisiDetayId = gen_ID(OrmeRafTransferFisiDetay_Id_gen, 1);

  Insert Into OrmeRafTransferFisiDetay(OrmeRafTransferFisiDetayID, OrmeRafTransferFisiId, OrmeRafID, Kilo, BrutKilo, KapAdedi, RafID, KaliteNo, INSERTKULLANICIID, KULLANICIID)
                   Values(:FOrmeRafTransferFisiDetayId, :FOrmeRafTransferFisiId, :AOrmeRafID, :AMiktar, :ABrutMiktar, :AKapAdedi, :HedefRaf, :KaliteNo, :AKullaniciID, :AKullaniciID);


  for select StokHareketDetay.StokHareketDetayID from DepoSayimDetay
      left Join StokHareketDetay on StokHareketDetay.Barkod = DepoSayimDetay.Barkod  and (StokHareketDetay.HareketTipi = :shtAlis or StokHareketDetay.HareketTipi = :shtFasonUretim or StokHareketDetay.HareketTipi = :shtUretim)
      where DepoSayimDetay.DepoSayimID = :ADepoSayimID and DepoSayimDetay.HamID = :HamId and
            IntKarsilastir(DepoSayimDetay.RenkID, :RenkId) = 1 and
            StrKarsilastir(DepoSayimDetay.Lot, :Lot) = 1 and
            StrKarsilastir(DepoSayimDetay.MelanjKodu, :MelanjKodu) = 1 and
            StrKarsilastir(DepoSayimDetay.FlamKodu, :FlamKodu) = 1 and
            StrKarsilastir(DepoSayimDetay.DesenId, :DesenId) = 1 and
            IntKarsilastir(DepoSayimDetay.DesenVaryantId, :DesenVaryantId) = 1 and
            IntKarsilastir(DepoSayimDetay.KaliteNo, :KaliteNo) = 1
  into :FStokHareketDetayID do
  begin
    Update StokHareketDetay Set OrmeRafTransferFisiDetayID = :FOrmeRafTransferFisiDetayId where StokHareketDetayID = :FStokHareketDetayID;

 end
end
