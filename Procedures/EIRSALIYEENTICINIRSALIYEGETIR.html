CREATE PROCEDURE EIRSALIYEENTICINIRSALIYEGETIR(
    BASLANGICTARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    GONDERILMEDURUMU INTEGER,
    AMODULID VARCHAR(10),
    EARSIVAKTIF SMALLINT = 0)
RETURNS (
    IRSALIYEID INTEGER,
    MODULID INTEGER,
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYENOSTR VARCHAR(23),
    HAREKETTIPIID INTEGER,
    HAREKETTIPISTR VARCHAR(32),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(112),
    SEVKTARIHI TIMESTAMP,
    FIRMASEVKADRESIID INTEGER,
    FIRMASEMT VARCHAR(20),
    FIRMABINANO VARCHAR(60),
    FIRMASOKAK VARCHAR(255),
    FIRMAKAPINO VARCHAR(30),
    FIRMASEHIR VARCHAR(20),
    FIRMAULKE VARCHAR(20),
    FIRMAPOSTAKODU VARCHAR(20),
    FIRMAVERGIDAIRESI VARCHAR(20),
    FIRMAVERGIHESAPNO VARCHAR(20),
    FIRMATELEFON VARCHAR(15),
    FIRMAFAX VARCHAR(15),
    FIRMAEMAIL VARCHAR(80),
    FIRMAWEB VARCHAR(80),
    FIRMAYETKILI VARCHAR(50),
    EIRSALIYE SMALLINT,
    KAYITTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 1 SEGMENT SIZE 80,
    SEVKADRESADI ADI50,
    SEVKADRES VARCHAR(80),
    SEVKADRES2 VARCHAR(120),
    SEVKSEMT VARCHAR(20),
    SEVKSEHIR VARCHAR(20),
    SEVKULKE VARCHAR(20),
    SEVKPOSTAKODU VARCHAR(20),
    SEVKVERGIDAIRESI VARCHAR(20),
    SEVKVERGIHESAPNO VARCHAR(20),
    ARACPLAKA ARACKODUSIZE,
    DORSEPLAKA ARACKODUSIZE,
    SOFORADI ADI40,
    SOFORKIMLIKNO ADI40,
    TASIYICIFIRMAADI TYPE OF COLUMN SEVKFIRMA.ADI,
    TASIYICISEVKADRES TYPE OF COLUMN SEVKFIRMA.ADRESI1,
    TASIYICISEVKADRES2 VARCHAR(80),
    TASIYICISEMT TYPE OF COLUMN SEVKFIRMA.SEMT,
    TASIYICISEHIR TYPE OF COLUMN SEVKFIRMA.SEHIR,
    TASIYICIULKE TYPE OF COLUMN SEVKFIRMA.ULKE,
    TASIYICIVKN TYPE OF COLUMN SEVKFIRMA.VERGIHESAPNO,
    TASIYICIVERGIDAIRESI TYPE OF COLUMN SEVKFIRMA.VERGIDAIRESI,
    TOPLAMKAPADEDI INTEGER,
    TOPLAMTARTIBRUTMIKTAR DOUBLE PRECISION,
    TOPLAMBRUTMIKTAR DOUBLE PRECISION,
    TOPLAMNETMIKTAR DOUBLE PRECISION,
    TOPLAMBRUTKILO DOUBLE PRECISION,
    TOPLAMNETKILO DOUBLE PRECISION,
    TOPLAMMETRE DOUBLE PRECISION,
    TOPLAMADET DOUBLE PRECISION,
    DOVIZID VARCHAR(10),
    TABANDOVIZKURU2 DOUBLE PRECISION,
    DEPOID VARCHAR(10),
    TRANSFERDEPOID VARCHAR(10),
    EKNOTU VARCHAR(200),
    GIBENTEGRASYONID VARCHAR(50),
    EGEMENFIRMAID TYPE OF COLUMN SISTEM.EGEMENFIRMAID,
    FIRMATRANSFERKODU TYPE OF COLUMN FIRMA.TRANSFERKODU)
AS
declare variable FCDataUygula smallint;
declare variable FDefaultFirmaSevkAdresiID varchar(20);
declare variable FSevkFirmaID varchar(20);
declare variable FAdres varchar(255);
declare variable FAdresi1 type of column Firma.Adresi1;
declare variable FAdresi2 type of column Firma.Adresi2;
declare variable FAdresi3 type of column Firma.Adresi3;
declare variable FFirmaKodumuz type of column Sistem.FirmaKodumuz;
declare variable FOrmeFirmaKodumuz type of column Sistem.OrmeFirmaKodumuz ;
declare variable FIplikFirmaKodumuz type of column Sistem.IplikFirmaKodumuz ;
declare variable FFabrikalarArasiSevkNotu memo;
declare variable FUretimIslemTipi type of column UretimIslemTipi.UretimIslemTipiId;
declare variable FUretimIslemTipiAdi type of column UretimIslemTipi.Adi ;
declare variable FToplamTartiBrutMiktar double precision;
declare variable FFirmaEirsaliye type of column Firma.EIrsaliye ;
declare variable FFirmaAdi2Kullan smallint;
begin
  /* Kayit Tarihini Sadece Irsaliye Saatini Tespit Etmek Için Kullanıyoruz. Kayıt Tarihi Fiili Sevk Tarihinden Büyük Olma Durumu Olduğu Için Sevk Tarihini Atadım. */
  Select Sistem.EgemenFirmaId, Sistem.FirmaKodumuz, Sistem.OrmeFirmaKodumuz, Sistem.IplikFirmaKodumuz From Sistem Where Sistem.SistemId = 1 Into :EgemenFirmaId, :FFirmaKodumuz, :FOrmeFirmaKodumuz, :FIplikFirmaKodumuz ;
  FFirmaAdi2Kullan = 0 ;

  If ( EgemenFirmaId in (52)) then   /* Naz Örme */
    FFirmaAdi2Kullan = 1 ;

  FCDataUygula = 1 ;
  if ((EgemenFirmaId = 12) or (EgemenFirmaId = 2)) then /* Şevkat,Batı Basma*/
    FCDataUygula = 0 ;

  FFabrikalarArasiSevkNotu = Cast('Fabrikalar arası sevk.' as Blob Character set UTF8) ;

  if (AModulId like '%0%') then
  begin
    ModulId = 0 ;

    For Select BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo,
       Coalesce(BoyaIrsaliye.IrsaliyeSeriNo,'') || Format_Float('', BoyaIrsaliye.IrsaliyeNo) As IrsaliyeNoStr, BoyaIrsaliye.HareketTipiId,
       CASE BoyaIrsaliye.HareketTipiId WHEN 0 THEN 'Satış' WHEN 1 THEN 'Alış İade' ELSE '' END AS HareketTipiStr,
       BoyaIrsaliye.FirmaID, IIF(:FFirmaAdi2Kullan=0, Firma.Adi, Firma.Adi2), BoyaIrsaliye.FirmaSevkAdresiId, Firma.Semt, Firma.BinaNo, Coalesce(trim(Firma.Sokak),''), Firma.KapiNo,
       Firma.Sehir, Firma.Ulke, Firma.VergiDairesi, Firma.VergiHesapNo, Firma.Telefon1, Firma.Telefon2, Firma.EMail, Firma.Web,
       BoyaIrsaliye.EIrsaliye, BoyaIrsaliye.Sevktarihi, BoyaIrsaliye.Aracid, Arac.DorsePlaka, Sofor.Adi, Sofor.Kimlikno, BoyaIrsaliye.SevkFirmaID, Coalesce(trim(Firma.Adresi1),''), Coalesce(trim(Firma.Adresi2),''), Coalesce(trim(Firma.Adresi3),''), BoyaIrsaliye.Notu, Firma.EIrsaliye, DefaultFirmaSevkAdresi.FirmaSevkAdresiId,
       Boyairsaliye.Sevktarihi, Firma.PostaKodu,Firma.TransferKodu From BoyaIrsaliye
    Left Join Firma ON Firma.FirmaID = BoyaIrsaliye.FirmaID
    Left Join Sofor on Sofor.SoforID = BoyaIrsaliye.SoforId
    Left Join Arac on  Arac.AracId = BoyaIrsaliye.AracId
    Left Join FirmaSevkAdresi DefaultFirmaSevkAdresi on DefaultFirmaSevkAdresi.FirmaID = BoyaIrsaliye.FirmaId And DefaultFirmaSevkAdresi.DefaultAdres = 1
    where BoyaIrsaliye.EIrsaliye=1 and
          BoyaIrsaliye.Sinif='B' and
          (BoyaIrsaliye.HareketTipiId in (0,1)) and
          (BoyaIrsaliye.IrsaliyeTarihi >= :BaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :BitisTarihi) and
          ((BoyaIrsaliye.GIBEntegrasyonKullaniciID is null and :GonderilmeDurumu = 0) or
           (BoyaIrsaliye.GIBEntegrasyonKullaniciID is not null and :GonderilmeDurumu = 1) or
           (:GonderilmeDurumu = 2))
    into :IrsaliyeId, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo,
         :IrsaliyeNoStr, :HareketTipiId,
         :HareketTipiStr,
         :FirmaID, :FirmaAdi, :FirmaSevkAdresiId, :FirmaSemt, :FirmaBinano, :FirmaSokak, :FirmaKapino,
         :FirmaSehir, :FirmaUlke, :FirmaVergiDairesi, :FirmaVergiHesapno, :FirmaTelefon, :FirmaFax, :FirmaEmail, :FirmaWeb,
         :EIrsaliye, :KayitTarihi, :AracPlaka, :DorsePlaka, :SoforAdi, :SoforKimlikNo, :FSevkFirmaID, :FAdresi1, :FAdresi2, :FAdresi3, :Notu, :FFirmaEirsaliye, :FDefaultFirmaSevkAdresiID,
         :SevkTarihi, :FirmaPostaKodu, :FirmaTransferKodu  do
    begin
      FAdres = '' ;
      If (StrLength(FAdresi1) > 0 ) Then
        FAdres = FAdres || FAdresi1 ;

      If (StrLength(FAdresi2) > 0 ) Then
        FAdres = FAdres || ' ' || FAdresi2 ;

      If (StrLength(FAdresi3) > 0 ) Then
        FAdres = FAdres || ' ' || FAdresi3 ;

      If (StrLength(FirmaSokak) > 0 ) Then
        FAdres = FAdres || ' ' || FirmaSokak ;

      FirmaSokak = FAdres ;
      FirmaYetkili         = '';
      TasiyiciFirmaAdi     = '';
      TasiyiciSevkAdres    = '';
      TasiyiciSevkAdres2   = '';
      TasiyiciSemt         = '';
      TasiyiciSehir        = '';
      TasiyiciUlke         = '';
      TasiyiciVkn          = '';
      TasiyiciVergiDairesi = '';

      select AdresAdi, trim(SevkAdresi1), trim(Coalesce(SevkAdresi2,'')) || ' ' || trim(Coalesce(SevkAdresi3,'')), SevkSemt, SevkSehir, SevkUlke, PostaKodu From FirmaSevkAdresi
      where FirmaSevkAdresi.FirmaId = :FirmaId and FirmaSevkAdresi.FirmaSevkAdresiId = :FirmaSevkAdresiId
      into :SevkAdresAdi, :SevkAdres, :SevkAdres2, :SevkSemt, :SevkSehir, :SevkUlke, :SevkPostaKodu;

      select SevkFirma.Adi, trim(SevkFirma.Adresi1), trim(Coalesce(SevkFirma.Adresi2,'')) || ' ' || trim(Coalesce(SevkFirma.Adresi3,'')), SevkFirma.Semt, SevkFirma.Sehir, SevkFirma.Ulke, SevkFirma.VergiHesapNo, SevkFirma.VergiDairesi From SevkFirma
      where SevkFirma.SevkFirmaId = :FSevkFirmaId
      into :TasiyiciFirmaAdi, :TasiyiciSevkAdres, :TasiyiciSevkAdres2, :TasiyiciSemt, :TasiyiciSehir, :TasiyiciUlke, :TasiyiciVkn, :TasiyiciVergiDairesi ;

      Select First 1 FirmaYetkili.Adi from FirmaYetkili where FirmaYetkili.FirmaID = :FirmaID and Firmayetkili.Defaultyetkili = 1 order by DefaultYetkili Desc into :FirmaYetkili;

      If (FirmaID in (FFirmaKodumuz, FOrmeFirmaKodumuz, FIplikFirmaKodumuz) ) then
        Notu = FFabrikalarArasiSevkNotu || ASCII_CHAR(13) || coalesce(Notu, '') ;
      Else
      Begin
        /* Faturalandırılabilecek Hareket Tipilerinde Ise Earşiv Yada Efatura Durumunu Nota Yazıyoruz.
        Fakat Irsaliyedeki Bazı Satırların Ücretli Olma Durumlarında Daha Farklı Bir Ifade Bulamadık Gülledeki Ekiple Konuşup Kaldırmaya Karar Verdik. Zaten Isteyende Gülleydi. */

        If (HareketTipiId = 1) Then
          Notu = 'İade' || ASCII_CHAR(13) || coalesce(Notu, '') ;

      End

      If ( FirmaSevkAdresiId <> FDefaultFirmaSevkAdresiID ) Then
        Notu = Coalesce(Notu,'') || ASCII_CHAR(13) || ' Namına Teslim:' || Coalesce(SevkAdresAdi,'') || ' ' || Coalesce(SevkAdres,'') || ' '  || Coalesce(SevkAdres2, '') || ' ' || Coalesce(SevkSemt, '') || ' ' || Coalesce(SevkSehir,'') || ' ' || Coalesce(SevkUlke, '') ;

      /*Xmle Özel Karakterleri Atlayabilmek Için Cdata Içine Alıyorum.*/
      if ( FCDataUygula = 1 ) then
        FirmaAdi  = '<![CDATA[' || FirmaAdi || ']]>' ;
      select sum(EIrsaliyeENTIcinIrsDetGetir.KapAdedi),   sum(EIrsaliyeENTIcinIrsDetGetir.TartiBrutMiktar),
             sum(EIrsaliyeENTIcinIrsDetGetir.BrutMiktar), sum(EIrsaliyeENTIcinIrsDetGetir.NetMiktar),
       sum(EIrsaliyeENTIcinIrsDetGetir.BrutKilo), sum(EIrsaliyeENTIcinIrsDetGetir.NetKilo),
             sum(EIrsaliyeENTIcinIrsDetGetir.Metre), sum(EIrsaliyeENTIcinIrsDetGetir.Adet) from EIrsaliyeENTIcinIrsDetGetir(0,:IrsaliyeId)
      into :ToplamKapAdedi, :ToplamTartiBrutMiktar, :ToplamBrutMiktar, :ToplamNetMiktar, :ToplamBrutKilo, :ToplamNetKilo, :ToplamMetre, :ToplamAdet;

      suspend;
    end
  end

  if (AModulId like '%1%') then
  begin
    ModulId = 1;
    IrsaliyeId      = Null ;
    IrsaliyeTarihi  = Null ;
    IrsaliyeSerino  = Null ;
    IrsaliyeNo      = Null ;
    IrsaliyeNoStr   = Null ;
    HareketTipiId   = Null ;
    HareketTipiStr  = Null ;
    FirmaId         = Null ;
    FirmaAdi        = Null ;
    SevkTarihi      = Null ;
    FirmaSevkAdresiId = Null ;
    FirmaSemt         = Null ;
    FirmaBinaNo       = Null ;
    FirmaSokak        = Null ;
    FirmaKapiNo       = Null ;
    FirmaSehir        = Null ;
    FirmaUlke         = Null ;
    FirmaVergiDairesi = Null ;
    FirmaVergiHesapNo = Null ;
    FirmaTelefon      = Null ;
    FirmaFax          = Null ;
    FirmaEmail        = Null ;
    FirmaWeb          = Null ;
    FirmaYetkili      = Null ;
    EIrsaliye         = Null ;
    KayitTarihi       = Null ;
    Notu              = Null ;
    SevkAdresAdi      = Null ;
    SevkAdres         = Null ;
    SevkAdres2        = Null ;
    SevkSemt          = Null ;
    SevkSehir         = Null ;
    SevkUlke          = Null ;
    SevkVergiDairesi  = Null ;
    SevkVergiHesapNo  = Null ;
    AracPlaka         = Null ;
    DorsePlaka        = Null ;
    SoforAdi          = Null ;
    SoforKimlikNo     = Null ;
    TasiyiciFirmaAdi  = Null ;
    TasiyiciSevkAdres   = Null ;
    TasiyiciSevkAdres2  = Null ;
    TasiyiciSemt        = Null ;
    TasiyiciSehir       = Null ;
    TasiyiciUlke        = Null ;
    TasiyiciVkn         = Null ;
    TasiyiciVergiDairesi = Null ;
    SevkPostaKodu    = Null;
    FirmaPostaKodu   = Null;
    FSevkFirmaID  = Null ;
    FAdres        = Null ;
    FAdresi1      = Null ;
    FAdresi2      = Null ;
    FAdresi3      = Null ;
    ToplamKapAdedi        = Null;
    ToplamTartiBrutMiktar = Null;
    ToplamBrutMiktar      = Null;
    ToplamNetMiktar       = Null;
    ToplamBrutKilo        = Null;
    ToplamNetKilo         = Null;
    ToplamMetre           = Null;
    ToplamAdet            = Null;
    FirmaTransferKodu     = Null;


    For Select OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo,
       Coalesce(OrmeIrsaliye.SeriNo, '') || Format_Float('', OrmeIrsaliye.IrsaliyeNo) As IrsaliyeNoStr, OrmeIrsaliye.HareketTipi, OrmeStokHareketTipi.Adi,
       OrmeIrsaliye.FirmaID, IIF(:FFirmaAdi2Kullan=0, Firma.Adi, Firma.Adi2), OrmeIrsaliye.FirmaSevkAdresiId, Firma.Semt, Firma.BinaNo, Coalesce(trim(Firma.Sokak),''), Firma.KapiNo,
       Firma.Sehir, Firma.Ulke, Firma.VergiDairesi, Firma.VergiHesapNo, Firma.Telefon1, Firma.Telefon2, Firma.EMail, Firma.Web,
       OrmeIrsaliye.EIrsaliye, OrmeIrsaliye.Fiilisevktarihi, OrmeIrsaliye.Aracid, Arac.DorsePlaka, Sofor.Adi, sofor.Kimlikno, OrmeIrsaliye.SevkFirmaID, Coalesce(trim(Firma.Adresi1),''), Coalesce(trim(Firma.Adresi2),''), Coalesce(trim(Firma.Adresi3),''), OrmeIrsaliye.Notu,
       OrmeIrsaliye.UretimIslemTipiId, UretimIslemTipi.Adi, Firma.Eirsaliye, DefaultFirmaSevkAdresi.FirmaSevkAdresiId, Ormeirsaliye.Fiilisevktarihi, Firma.PostaKodu,Firma.TransferKodu From OrmeIrsaliye
    Left Join Firma ON Firma.FirmaID = OrmeIrsaliye.FirmaID
    Left Join Sofor on Sofor.SoforID = OrmeIrsaliye.SoforId
    Left Join Arac on  Arac.AracId = OrmeIrsaliye.AracId
    Left Join OrmeStokHareketTipi On OrmeStokHareketTipi.OrmeStokHareketTipiId = OrmeIrsaliye.Harekettipi
    Left Join UretimIslemTipi On UretimIslemTipi.UretimIslemTipiId = OrmeIrsaliye.UretimIslemTipiId
    Left Join FirmaSevkAdresi DefaultFirmaSevkAdresi on DefaultFirmaSevkAdresi.FirmaID = OrmeIrsaliye.FirmaId And DefaultFirmaSevkAdresi.DefaultAdres = 1
    where OrmeIrsaliye.EIrsaliye=1 and
          OrmeIrsaliye.Sinif='B' and
          (OrmeIrsaliye.HareketTipi in (2,3,6,14)) and
          (OrmeIrsaliye.IrsaliyeTarihi >= :BaslangicTarihi and OrmeIrsaliye.IrsaliyeTarihi <= :BitisTarihi) and
          ((OrmeIrsaliye.GIBEntegrasyonKullaniciID is null and :GonderilmeDurumu = 0) or
           (OrmeIrsaliye.GIBEntegrasyonKullaniciID is not null and :GonderilmeDurumu = 1) or
           (:GonderilmeDurumu = 2))
    into :IrsaliyeId, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo,
         :IrsaliyeNoStr, :HareketTipiId, :HareketTipiStr,
         :FirmaID, :FirmaAdi, :FirmaSevkAdresiId, :FirmaSemt, :FirmaBinano, :FirmaSokak, :FirmaKapino,
         :FirmaSehir, :FirmaUlke, :FirmaVergiDairesi, :FirmaVergiHesapno, :FirmaTelefon, :FirmaFax, :FirmaEmail, :FirmaWeb,
         :EIrsaliye, :KayitTarihi, :AracPlaka, :DorsePlaka, :SoforAdi, :SoforKimlikNo, :FSevkFirmaID, :FAdresi1, :FAdresi2, :FAdresi3, :Notu,
         :FUretimIslemTipi, :FUretimIslemTipiAdi, :FFirmaEirsaliye, :FDefaultFirmaSevkAdresiID, :SevkTarihi, :FirmaPostaKodu, :FirmaTransferKodu do
