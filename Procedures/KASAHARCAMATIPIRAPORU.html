CREATE PROCEDURE KASAHARCAMATIPIRAPORU(
    AKASAID VARCHAR(10),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICHARCAMATIPIID VARCHAR(10),
    ABITISHARCAMATIPIID VARCHAR(10),
    AHARCAMATIPIARALIKLI INTEGER,
    ASINIF VARCHAR(1),
    AHARCAMATIPIGRUPID HARCAMATIPIGRUPKODUSIZE,
    ADOVIZID DOVIZSIZE,
    ARAPORDOVIZID DOVIZSIZE)
RETURNS (
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    HARCAMATIPIID VARCHAR(10),
    HARCAMATIPIADI VARCHAR(30),
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    FIRMAID VARCHAR(10),
    FIRMAADI VARCHAR(100),
    BANKAHESAPID VARCHAR(20),
    BANKAHESAPADI VARCHAR(50),
    HARCAMATIPIGRUPID HARCAMATIPIGRUPKODUSIZE,
    HARCAMATIPIGRUPADI ADI30,
    KASAID KASAKODUSIZE,
    ADI ADI40,
    RAPORDOVIZID DOVIZSIZE)
AS
declare variable BorcAlacak     varchar(1);
declare variable Meblag         Double Precision;
declare variable Okey           integer;
declare variable DovizKuru      Double Precision;
begin
  RaporDovizId = :ARaporDovizId;
  for Select KasaHareket.Tarih, KasaHareket.Aciklama, KasaHareket.KasaHarcamaTipiId, KasaHarcamaTipi.Adi,
             KasaHareket.Meblag, KasaHareket.BorcAlacak, KasaHareket.FirmaId, Firma.Adi, KasaHareket.BankaHesapId, BankaHesap.Adi,
             HarcamaTipiGrup.HarcamaTipiGrupId, HarcamaTipiGrup.Adi, Kasa.KasaId, Kasa.DovizId, Kasa.Adi from KasaHareket
  left join Kasa on Kasa.KasaId = KasaHareket.KasaId
  left join Firma on Firma.FirmaId = KasaHareket.FirmaId
  left join BankaHesap on BankaHesap.BankaHesapId = KasaHareket.BankaHesapId
  left join KasaHarcamaTipi on KasaHarcamaTipi.KasaHarcamaTipiId = KasaHareket.KasaHarcamaTipiId
  left join HarcamaTipiGrup on HarcamaTipiGrup.HarcamaTipiGrupId = KasaHarcamaTipi.HarcamaTipiGrupId
  where (KasaHareket.KasaId = :AKasaId or :AKasaId = '') and
        (Kasa.DovizId = :ADovizId or :ADovizId is null) and
        KasaHareket.Tarih >= :ABaslangicTarihi and KasaHareket.Tarih <= :ABitisTarihi and
        (KasaHareket.Sinif = :ASinif or :ASinif is null) and
        ((:AHarcamaTipiAralikli = 1 and KasaHareket.KasaHarcamaTipiId >= :ABaslangicHarcamaTipiId and (KasaHareket.KasaHarcamaTipiId <= :ABitisHarcamaTipiId or :ABitisHarcamaTipiId = '')) or (:AHarcamaTipiAralikli = 0)) and
        (KasaHarcamaTipi.HarcamaTipiGrupId = :AHarcamaTipiGrupId or :AHarcamaTipiGrupId is null)
  Order By KasaHareket.Tarih
  into :Tarih, :Aciklama, :HarcamaTipiId, :HarcamaTipiAdi, :Meblag, :BorcAlacak,
       :FirmaId, :FirmaAdi, :BankaHesapId, :BankaHesapAdi, :HarcamaTipiGrupId, :HarcamaTipiGrupAdi, :KasaId, :DovizId, :Adi do
  begin
    Okey = 1;
    if (AHarcamaTipiAralikli = 0) then
    if (KutuKarsilastir(ABaslangicHarcamaTipiId, HarcamaTipiId) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      Borc = 0;
      Alacak = 0;

      if (DovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :Tarih) into :DovizKuru;
      else DovizKuru = 1;

      Meblag = Meblag * DovizKuru;

      if (BorcAlacak = 'A') then Alacak = Meblag;
      else Borc = Meblag;
      suspend;
    end
  end
end
