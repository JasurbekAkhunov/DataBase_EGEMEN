CREATE PROCEDURE ORMESTOKHAREKETKARTIGETIR(
    AHAMID INTEGER,
    ARENKID INTEGER,
    ADESENID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AALIS INTEGER,
    ASATIS INTEGER,
    AFASONALIS INTEGER,
    AFASONURETIMECIKIS INTEGER,
    AURETIMECIKIS INTEGER,
    AFASONURETIM INTEGER,
    AURETIM INTEGER,
    AKAYDIRMA INTEGER)
RETURNS (
    ORMEFATURAID INTEGER,
    ORMEIRSALIYEID INTEGER,
    ORMEFISID INTEGER,
    STOKHAREKETID INTEGER,
    STOKGIRISNO INTEGER,
    LOT VARCHAR(150),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    TARIHI TIMESTAMP,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(40),
    HAMIADE INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    URETIMISLEMTIPIID INTEGER,
    URETIMISLEMTIPIIDSTR VARCHAR(30),
    BOYAISLEMTIPIID INTEGER,
    BOYAISLEMTIPIADI VARCHAR(20),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ACIKLAMA VARCHAR(50),
    KAPADEDI INTEGER,
    MIKTAR DOUBLE PRECISION,
    BRUTMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    NETBIRIMFIYAT DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    TOPLAMTUTAR DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    KDV DOUBLE PRECISION,
    TEVKIFAT INTEGER,
    TEVKIFATSTR VARCHAR(3),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    AYARTOPU INTEGER,
    AYARTOPUSTR VARCHAR(10),
    URETIMPARTINO VARCHAR(30),
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    GIRISCIKIS INTEGER,
    RAFID VARCHAR(50),
    STOKKAPADEDI NUMERIC(12,3),
    STOKMIKTARI NUMERIC(12,3),
    ORMESTOKKAYDIRMAID INTEGER,
    MAKINALARARASITRANSFER INTEGER,
    OMIPLIKTRANSFERFISIFISYIL INTEGER,
    OMIPLIKTRANSFERFISIFISNO INTEGER,
    ALISFIYATI DOUBLE PRECISION,
    ALISFIYATIDOVIZID DOVIZSIZE,
    OLCUID INTEGER,
    OLCUADI VARCHAR(40))
AS
declare variable FHamIade           integer;
declare variable str                varchar(30);
declare variable FisDovizId         varchar(5);
declare variable IrsaliyeDovizId    varchar(5);
declare variable FaturaDovizId      varchar(5);
declare variable shtDevir       integer;
declare variable shtAlis        integer;
declare variable shtSatis       integer;
declare variable shtAlisIade    integer;
declare variable shtSatisIade   integer;
declare variable shtFasonAlis   integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis      integer;
declare variable shtFasonUretim       integer;
declare variable shtUretim            integer;
declare variable ShtKaydirmaCikis     integer;
declare variable ShtKaydirmaGiris     integer;
declare variable ShtSarf              integer;
declare variable Okey                 integer;
declare variable FUretimPartiNo       varchar(30);
declare variable FisTevkifatOrani1 Double Precision;
declare variable FisTevkifatOrani2 Double Precision;
declare variable IrsaliyeTevkifatOrani1 Double Precision;
declare variable IrsaliyeTevkifatOrani2 Double Precision;
declare variable TevkifatOrani1 Double Precision;
declare variable TevkifatOrani2 Double Precision;

declare variable FAltIskontoCarpani     Double Precision;
declare variable FStokGirisNo integer;
declare variable FRafId       varchar(10);
declare variable FSakat       integer;
declare variable FAyarTopu    integer;
begin
  if (ABaslangicTarihi is null) then ABaslangicTarihi = current_TimeStamp(0);
  if (ABitisTarihi is null) then ABitisTarihi = current_TimeStamp(0);

  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, ShtKaydirmaCikis, ShtKaydirmaGiris, ShtSarf from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim, :ShtKaydirmaCikis, :ShtKaydirmaGiris, :ShtSarf;

  if (ABaslangicTarihi is null) then ABaslangicTarihi = DoubleToTimeStamp(0);

  if (ABitisTarihi is null) then ABitisTarihi = current_TimeStamp(0);

  Select sum(StokHareket.Miktar * StokHareket.GirisCikis), sum(StokHareket.KapAdedi * StokHareket.GirisCikis) from StokHareket
  where StokHareket.HamId = :AHamId and StokHareket.Tarih < :ABaslangicTarihi and
        (StokHareket.RenkID = :ARenkId or (StokHareket.RenkID is null and :ARenkId is null) or :ARenkId = -1) and
        (StokHareket.DesenId = :ADesenId or :ADesenId is null)
  into :StokMiktari, :StokKapAdedi;

  if (StokMiktari is null) then StokMiktari = 0;
  if (StokKapAdedi is null) then StokKapAdedi = 0;

  For select StokHareket.OrmeFaturaId, StokHareket.OrmeIrsaliyeId, StokHareket.OrmeFisId, StokHareket.StokHareketId, StokHareket.StokGirisNo, StokHareket.Tarih,
             StokHareket.HareketTipi, StokHareket.HamIade, StokHareket.GirisCikis, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             StokHareket.UretimIslemTipiId, UretimIslemTipi.Adi, BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi, StokHareket.FirmaId, Firma.Adi,
             StokHareket.Lot, StokHareket.MelanjKodu, StokHareket.KapAdedi, StokHareket.Miktar, StokHareket.BrutMiktar, StokHareket.BirimFiyat,
             StokHareket.NetBirimFiyat, StokHareket.Kdv, StokHareket.Tevkifat, StokHareket.Aciklama, OrmeFatura.FaturaNo, OrmeFatura.FaturaTarihi,
             OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeFis.FisNo, OrmeFis.FisTarihi, OrmeFis.DovizId, OrmeIrsaliye.DovizId,
             Ormefatura.DovizId, OrmeMakina.OrmeMakinaId, OrmeMakina.Adi, StokHareket.UretimPartiNo, OrmeFatura.AltIskontoCarpani,
             OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, OrmeRaf.RafId, OrmeRaf.StokGirisNo, StokHareket.RenkId, Renk.Adi,
             StokHareket.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, OrmeFis.OrmeStokKaydirmaId, OrmeFatura.TevkifatOrani1, OrmeFatura.TevkifatOrani2,
             OrmeIrsaliye.TevkifatOrani1, OrmeIrsaliye.TevkifatOrani2, OrmeFis.TevkifatOrani1, OrmeFis.TevkifatOrani2, StokHareket.Sakat, StokHareket.AyarTopu, OrmeRaf.Sakat, OrmeRaf.AyarTopu,
             OrmeFis.MakinalarArasiTransfer, Ham.AlisFiyati, Ham.AlisFiyatiDovizID, StokHareket.OlcuId, Olcu.OlcuAdi from StokHareket
    left join Ham on Ham.HamId = StokHareket.HamId
    left join UretimIslemTipi on UretimIslemTipi.UretimIslemTipiId = StokHareket.UretimIslemTipiId
    left join Renk on Renk.RenkId = StokHareket.RenkId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = StokHareket.OrmeMakinaId
    left join OrmeFatura on OrmeFatura.OrmeFaturaId = StokHareket.OrmeFaturaId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
    left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
    left join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
    left join Firma on Firma.FirmaId = StokHareket.FirmaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayID
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = StokHareket.BoyaSiparisDetayID
    left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
    left join Desen on Desen.DesenId = StokHareket.DesenId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = StokHareket.DesenVaryantId
    left join Olcu on Olcu.OlcuId = StokHareket.OlcuId
    where StokHareket.HamId = :AHamId and StokHareket.Tarih >= :ABaslangicTarihi  and StokHareket.Tarih <= :ABitisTarihi and
        (StokHareket.RenkID = :ARenkId or (StokHareket.RenkID is null and :ARenkId is null) or :ARenkId = -1) and
        (StokHareket.DesenId = :ADesenId or :ADesenId is null) and
        (StokHareket.HareketTipi <> :shtAlis or :AAlis = 1) and
        (StokHareket.HareketTipi <> :shtAlisIade or :AAlis = 1) and
        (StokHareket.HareketTipi <> :shtSatis or
          (StokHareket.HareketTipi = :shtSatis and :ASatis = 1 and StokHareket.HamIade = 0) or
          (StokHareket.HareketTipi = :shtSatis and StokHareket.HamIade = 1 and :AFasonAlis = 1 ) ) and
        (StokHareket.HareketTipi <> :shtSatisIade or :ASatis = 1) and
        (StokHareket.HareketTipi <> :shtFasonAlis or :AFasonAlis = 1) and
        (StokHareket.HareketTipi <> :shtFasonUretimeCikis or :AFasonUretimeCikis = 1) and
        (StokHareket.HareketTipi <> :shtUretimeCikis or :AUretimeCikis = 1) and
        (StokHareket.HareketTipi <> :shtFasonUretim or :AFasonUretim = 1) and
        (StokHareket.HareketTipi <> :shtUretim or :AUretim = 1) and
        ( (StokHareket.HareketTipi <> :ShtKaydirmaGiris and StokHareket.HareketTipi <> :ShtKaydirmaCikis) or :AKaydirma = 1)
    order By StokHareket.Tarih, StokHareket.StokHareketId, StokHareket.HamId, StokHareket.RenkId
    into :OrmeFaturaId, :OrmeIrsaliyeId, :OrmeFisId, :StokHareketId, :StokGirisNo, :Tarihi, :HareketTipi, :HamIade, :GirisCikis, :BoyaSiparisYil, :BoyaSiparisId,
         :UretimIslemTipiId, :UretimIslemTipiIdStr, :BoyaIslemTipiId, :BoyaIslemTipiAdi, :FirmaId, :FirmaAdi, :Lot, :MelanjKodu, :KapAdedi, :Miktar, :BrutMiktar, :BirimFiyat, :NetBirimFiyat,
         :Kdv, :Tevkifat, :Aciklama, :FaturaNo, :FaturaTarihi,
         :IrsaliyeNo, :IrsaliyeTarihi, :FisNo, :FisTarihi, :FisDovizId, :IrsaliyeDovizId, :FaturaDovizId,
         :OrmeMakinaId, :OrmeMakinaAdi, :UretimPartiNo, :FAltIskontoCarpani,
         :SiparisSiraNo, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :RafId, :FStokGirisNo, :RenkId, :RenkAdi, :DesenId, :DesenAdi, :VaryantAdi,
         :OrmeStokKaydirmaId, :TevkifatOrani1, :TevkifatOrani2, :IrsaliyeTevkifatOrani1, :IrsaliyeTevkifatOrani2, :FisTevkifatOrani1, :FisTevkifatOrani2, :Sakat, :AyarTopu, :FSakat, :FayarTopu,
         :MakinalarArasiTransfer, :AlisFiyati, :AlisFiyatiDovizID, :OlcuId, :OlcuAdi do
  begin
    Okey = 1;
    if (Okey = 1) then
    begin
      if (FAltIskontoCarpani is Null) then FAltIskontoCarpani = 0;

      OMIplikTransferFisiFisYil = Null;
      OMIplikTransferFisiFisNo  = Null;

      if (MakinalarArasiTransfer = 1) then
      begin
        if (GirisCikis = 1) then
          Select FisYil, FisNo from OMIplikTransferFisi where OMIplikTransferFisi.IadeStokHareketID = :StokHareketID into :OMIplikTransferFisiFisYil, :OMIplikTransferFisiFisNo;
        else
          Select FisYil, FisNo from OMIplikTransferFisi where OMIplikTransferFisi.CikisStokHareketID = :StokHareketID into :OMIplikTransferFisiFisYil, :OMIplikTransferFisiFisNo;
      end

      if (Tevkifat = 1) then TevkifatStr = 'Var';
      else TevkifatStr = '';

      if (FaturaDovizId is not null) then
      begin
        DovizId = FaturaDovizId;
      end
      else if (FisDovizId is not null) then
      begin
        DovizId = FisDovizId;
        TevkifatOrani1 = FisTevkifatOrani1;
        TevkifatOrani2 = FisTevkifatOrani2;
      end
      else if (IrsaliyeDovizId is not null) then
      begin
        TevkifatOrani1 = IrsaliyeTevkifatOrani1;
        TevkifatOrani2 = IrsaliyeTevkifatOrani2;
        DovizId = IrsaliyeDovizId;
      end

      ToplamTutar = NetBirimFiyat * Miktar;
      ToplamTutar = ToplamTutar - ToplamTutar * FAltIskontoCarpani;

      if (Tevkifat = 1) then TahsilEdilecekKdv = (ToplamTutar * Kdv /100) * (TevkifatOrani2 - TevkifatOrani1) / TevkifatOrani2;
      else TahsilEdilecekKdv = ToplamTutar * Kdv /100;

      ToplamTutar = ToplamTutar + TahsilEdilecekKdv;

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);


      if (HareketTipi = shtDevir) then HareketTipiStr = 'Devir..';
      else if (HareketTipi = shtAlis) then HareketTipiStr = 'Alış';
      else if (HareketTipi = shtSatis and HamIade = 0) then HareketTipiStr = 'Satış';
      else if (HareketTipi = shtSatis and HamIade = 1) then HareketTipiStr = 'Fason Hammadde İade';
      else if (HareketTipi = shtAlisIade) then HareketTipiStr = 'Alış İade';
      else if (HareketTipi = shtSatisIade) then HareketTipiStr = 'Satış İade';
      else if (HareketTipi = shtFasonAlis) then HareketTipiStr = 'Fason İşlenmek Üzere Giriş';
      else if (HareketTipi = shtFasonUretimeCikis) then HareketTipiStr = 'Fason Üretime Çıkış';
      else if (HareketTipi = shtUretimeCikis) then HareketTipiStr = 'Üretime Çıkış';
      else if (HareketTipi = shtFasonUretim and HamIade = 0) then HareketTipiStr = 'Fason Üretim';
      else if (HareketTipi = shtFasonUretim and HamIade = 1) then HareketTipiStr = 'Fason Üretim İade';
      else if (HareketTipi = shtUretim and HamIade = 0) then HareketTipiStr = 'Üretim';
      else if (HareketTipi = shtUretim and HamIade = 1) then HareketTipiStr = 'Üretim İade';
      else if (HareketTipi = shtKaydirmaCikis) then HareketTipiStr = 'Stok Kaydırma (Çıkış)';
      else if (HareketTipi = shtKaydirmaGiris) then HareketTipiStr = 'Stok Kaydırma (Giriş)';
      else if (HareketTipi = shtSarf) then HareketTipiStr = 'Sarf';
      else HareketTipiStr = 'hatalı..';

      if (GirisCikis = 1) then
      begin
        StokGirisNo = FStokGirisNo;
        RafID = '';
        For Select OrmeRafDetay.RafID from OrmeRafDetay
          where OrmeRafDetay.StokHareketID = :StokHareketID
          group by OrmeRafDetay.RafID
          into :FRafId do
        begin
          if (Char_Length(RafId) + Char_Length(FRafId) < 49) then
          begin
            if (RafId <> '') then RafId = RafId || ',';
            RafId = RafId || FRafId;
          end
        end
      end
      else
      begin
        Sakat = FSakat;
        AyarTopu = FAyarTopu;
      end

      if (Sakat = 1) then SakatStr = 'Sakat';
      else SakatStr = '';

      if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
      else AyarTopuStr = '';

      StokMiktari = StokMiktari + Miktar * GirisCikis;
      StokKapAdedi = StokKapAdedi + KapAdedi * GirisCikis;
      suspend;
    end
  end
end
