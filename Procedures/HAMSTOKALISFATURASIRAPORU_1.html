CREATE PROCEDURE HAMSTOKALISFATURASIRAPORU_1(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    HAMSTOKFATURAID INTEGER,
    FATURATARIHI TIMESTAMP,
    FATURASERINO SERINOSIZE5,
    ODEMETARIHI TIMESTAMP,
    FATURANO FATURANOSIZE,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BRUTTOPLAM DOUBLE PRECISION,
    PARASALISKONTO DOUBLE PRECISION,
    ISKONTO1ORANI DOUBLE PRECISION,
    ISKONTO1ORANTUTARI DOUBLE PRECISION,
    ISKONTO2ORANI DOUBLE PRECISION,
    ISKONTO2ORANTUTARI DOUBLE PRECISION,
    KDVTOPLAMI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    ENTEGRASYON INTEGER,
    ENTEGRASYONSTR VARCHAR(4),
    ALTISKONTOCARPANI DOUBLE PRECISION,
    KDVYUVARLAMACARPANI DOUBLE PRECISION,
    NETFIYATYUVARLAMACARPANI DOUBLE PRECISION,
    ODEMESEKLIADI ADI40)
AS
begin
  for select HamStokFatura.HamStokFaturaId, HamStokFatura.FaturaTarihi, HamStokFatura.FaturaSeriNo, HamStokFatura.FaturaNo,
             HamStokFatura.OdemeTarihi, HamStokFatura.FirmaId, Firma.Adi, HamStokFatura.DovizId, HamStokFatura.BrutToplam, HamStokFatura.ParasalIskonto,
             HamStokFatura.Iskonto1Orani, HamStokFatura.Iskonto1OranTutari, HamStokFatura.Iskonto2Orani, HamStokFatura.Iskonto2OranTutari,
             HamStokFatura.KdvToplami, HamStokFatura.AltIskontoCarpani, HamStokFatura.KdvYuvarlamaCarpani, HamStokFatura.NetFiyatYuvarlamaCarpani, OdemeSekli.Adi
             from HamStokFatura
  left join Firma on Firma.FirmaId = HamStokFatura.FirmaId
  left join OdemeSekli on OdemeSekli.OdemeSekliId = HamStokFatura.OdemeSekliId
  where HamStokFatura.FaturaTarihi >= :ABaslangicTarihi and HamStokFatura.FaturaTarihi <= :ABitisTarihi and
        (HamStokFatura.FirmaID = :AFirmaID or :AFirmaID is null) and
        HamStokFatura.HamStokHareketTipiId = 0
  into :HamStokFaturaId, :FaturaTarihi, :FaturaSeriNo, :FaturaNo, :OdemeTarihi,  :FirmaId, :FirmaAdi, :DovizId,
       :BrutToplam, :ParasalIskonto, :Iskonto1Orani, :Iskonto1OranTutari, :Iskonto2Orani, :Iskonto2OranTutari,
       :KdvToplami, :AltIskontoCarpani, :KdvYuvarlamaCarpani, :NetFiyatYuvarlamaCarpani, :OdemeSekliAdi do
  begin
    if (Entegrasyon = 1) then EntegrasyonStr = 'Evet';
    else EntegrasyonStr = '';

    suspend;
  end
end
