    KARTELAFISIDETAYID INTEGER,
    KARTELAFISIID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAGRUP1 VARCHAR(4),
    FIRMAGRUP2 VARCHAR(4),
    FIRMAGRUP3 VARCHAR(4),
    KARTELATALEPDETAYID INTEGER,
    MIKTAR INTEGER,
    ACIKLAMA VARCHAR(255),
    GIRIS INTEGER,
    GIRISSTR VARCHAR(5),
    KARTELATALEPKAYITID INTEGER,
    KARTELATALEPYIL INTEGER,
    KARTELATALEPID INTEGER,
    TALEPTARIHI TIMESTAMP,
    KALITEID VARCHAR(20),
    KALITEADI ADI255,
    PARTISINIFID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    HAMADI VARCHAR(255),
    RECETEKODU VARCHAR(20),
    RECETEADI VARCHAR(40),
    DESENKODU VARCHAR(20),
    DESENADI VARCHAR(50),
    DESENVARYANTADI VARCHAR(30),
    DENYE INTEGER,
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    CEKMEZLIKENISTENEN DOUBLE PRECISION,
    CEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    DONMEISTENEN VARCHAR(10),
    SANSISTENEN VARCHAR(10),
    PUS INTEGER,
    FEINE INTEGER,
    DINLENMISGRAMAJ INTEGER,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),

    IPLIKBOYU4 VARCHAR(20),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    ENVERILEN VARCHAR(10),

    DONMEVERILEN VARCHAR(10),
    CEKMEZLIKENVERILEN DOUBLE PRECISION,
    CEKMEZLIKBOYVERILEN DOUBLE PRECISION,
    CEKMEZLIKENVERILEN2 DOUBLE PRECISION,
    CEKMEZLIKBOYVERILEN2 DOUBLE PRECISION,
    METRAJNUMUNESIVAR INTEGER,
    METRAJNUMUNESIVARSTR VARCHAR(5),
    KARTELASTOKMEVCUDU INTEGER,
    SIPARISTOPLAMKILO DOUBLE PRECISION,
    FIRMAYETKILIADI VARCHAR(50),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    COMPOSITION VARCHAR(255),
    TALEPNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    GONDERILDI SMALLINT)
AS
begin
  For Select KartelaFisi.KartelaFisiID, KartelaFisi.FisYil, KartelaFisi.FisNo, KartelaFisi.FisTarihi, KartelaFisi.FirmaId, Firma.Adi,
             KartelaFisiDetay.KartelaFisiDetayId, KartelaFisiDetay.KartelaTalepDetayId, KartelaFisiDetay.Miktar, KartelaFisiDetay.Aciklama, KartelaFisiDetay.Giris,
             KartelaTalep.KartelaTalepKayitId, KartelaTalep.KartelaTalepYil, KartelaTalep.KartelaTalepId, KartelaTalep.TalepTarihi,
             Kalite.KaliteId, Kalite.KaliteAdi, Kalite.PartiSinifId, Kalite.PartiYil, Kalite.PartiId, Kalite.HamAdi, Kalite.ReceteKodu,
             Kalite.ReceteAdi, Kalite.Denye, Kalite.SiparisProsesId, SiparisProses.Adi, Kalite.EnIstenen, Kalite.GramajIstenen, Kalite.CekmezlikEnIstenen,
             Kalite.CekmezlikBoyIstenen, Kalite.DonmeIstenen, Kalite.SansIstenen, Kalite.Pus, Kalite.Feine, Kalite.DinlenmisGramaj, Kalite.IplikBoyu1,
             Kalite.IplikBoyu2, Kalite.IplikBoyu3, Kalite.IplikBoyu4, Kalite.TupAcikEnMayli,
             PartiStokFizikKontrol.EnVerilen, PartiStokFizikKontrol.GramajVerilen, PartiStokFizikKontrol.DonmeVerilen,
             PartiStokFizikKontrol.CekmezlikEnVerilen, PartiStokFizikKontrol.CekmezlikBoyVerilen, PartiStokFizikKontrol.CekmezlikEnVerilen2,
             PartiStokFizikKontrol.CekmezlikBoyVerilen2,
             Kalite.DesenKodu, Kalite.DesenAdi, Kalite.DesenVaryantAdi,
             Kalite.MetrajNumunesiVar, Kalite.KartelaStokMevcudu,
             Kalite.Notu, IK.Adi, KartelaFisiDetay.InsertTarihi, Sk.Adi, KartelaFisiDetay.DegisimTarihi,
             FirmaYetkili.Adi, Firma.FirmaGrup1, Firma.FirmaGrup2, Firma.FirmaGrup3, Kalite.Composition,
             KartelaTalep.Notu, KartelaFisiDetay.BirimFiyat, KartelaFisiDetay.DovizId, KartelaFisi.Gonderildi from KartelaFisiDetay
    left join Kalite on Kalite.KaliteId = KartelaFisiDetay.KaliteId
    left join KartelaFisi on KartelaFisi.KartelaFisiId = KartelaFisiDetay.KartelaFisiId
    left join KartelaTalepDetay on KartelaTalepDetay.KartelaTalepDetayId = KartelaFisiDetay.KartelaTalepDetayId
    left join KartelaTalep on KartelaTalep.KartelaTalepKayitId = KartelaTalepDetay.KartelaTalepKayitId
    left join Firma on Firma.FirmaId = KartelaFisi.FirmaId
    left join PartiStok on PartiStok.PartiSinifId = Kalite.PartiSinifId
    left join SiparisProses on SiparisProses.SiparisProsesId = Kalite.SiparisProsesId
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
    left join Parti on Parti.PartikayitId = PartiStok.PartikayitId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    left join Ham on Ham.HamId = Kalite.HamId
    left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = KartelaFisiDetay.FirmaYetkiliId
    Left Join Kullanici IK on IK.KullaniciId = KartelaFisiDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = KartelaFisiDetay.KullaniciId
    where KartelaFisi.FisTarihi >= :ABaslangicTarihi and KartelaFisi.FisTarihi <= :ABitisTarihi and
          (Kalite.KaliteAdi like '%' || :AKaliteAdi || '%' or :AKaliteAdi is Null) and
          (Kalite.ReceteKodu = :AReceteKodu or :AReceteKodu is Null) and
          (Kalite.HamId = :AHamId or :AHamId is Null) and
          ((:AHamGrup1Aralikli = 1 and Ham.HamGrup1 >= :ABaslangicGrup1Id  and (Ham.HamGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AHamGrup1Aralikli = 0) or (:AHamGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
          ((:AHamGrup2Aralikli = 1 and Ham.HamGrup2 >= :ABaslangicGrup2Id and (Ham.HamGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AHamGrup2Aralikli = 0) or (:AHamGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
          ((:AHamGrup3Aralikli = 1 and Ham.HamGrup3 >= :ABaslangicGrup3Id and (Ham.HamGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AHamGrup3Aralikli = 0) or (:AHamGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
          ((:AHamGrup4Aralikli = 1 and Ham.HamGrup4 >= :ABaslangicGrup4Id and (Ham.HamGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AHamGrup4Aralikli = 0) or (:AHamGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = '')) and
          (Kalite.DesenKodu = :ADesenKodu or :ADesenKodu is null) and
          (Kalite.SiparisProsesId = :ASiparisProsesId or :ASiparisProsesId is Null) and
          (Kalite.OzelTestler like '%' || :AOzelTestler || '%' or :AOzelTestler is Null) and
          (Kalite.OzelApreler like '%' || :AOzelApreler || '%' or :AOzelApreler is Null) and
          (Kalite.IlaveIslemler like '%' || :AIlaveIslemler || '%' or :AIlaveIslemler is Null) and
          ((:AEnAralikli = 1 and Kalite.EnIstenen >= :AEnIstenenBaslangic  and (Kalite.EnIstenen <= :AEnIstenenBitis or :AEnIstenenBitis = '')) or (:AEnAralikli = 0 and :AEnIstenenBaslangic = '') or (:AEnAralikli = 0 and :AEnIstenenBaslangic = Kalite.EnIstenen) or (:AEnAralikli = 1 and :AEnIstenenBaslangic = '' and :AEnIstenenBitis = '')) and
          ((:AGramajAralikli = 1 and Kalite.GramajIstenen >= :AGramajIstenenBaslangic and (Kalite.GramajIstenen <= :AGramajIstenenBitis or :AGramajIstenenBitis = '')) or (:AGramajAralikli = 0 and :AGramajIstenenBaslangic = '') or (:AGramajAralikli = 0 and :AGramajIstenenBaslangic = Kalite.GramajIstenen) or (:AGramajAralikli = 1 and :AGramajIstenenBaslangic = '' and :AGramajIstenenBitis = '')) and
          (KartelaFisi.FirmaId = :AFirmaId or :AFirmaId is Null) and
          (KartelaFisi.Giris = :AGirisCikis or :AGirisCikis is Null) and
          (KartelaFisiDetay.KaliteID = :AKaliteID or :AKaliteID is null) and
          ((:AFirmaGrup1Aralikli = 1 and (Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id or :ABaslangicFirmaGrup1Id = '') and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
          ((:AFirmaGrup2Aralikli = 1 and (Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id or :ABaslangicFirmaGrup2Id = '') and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
          ((:AFirmaGrup3Aralikli = 1 and (Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id or :ABaslangicFirmaGrup3Id = '') and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = ''))
    into :KartelaFisiID, :FisYil, :FisNo, :FisTarihi, :FirmaId, :FirmaAdi,
         :KartelaFisiDetayId, :KartelaTalepDetayId, :Miktar, :Aciklama, :Giris,
         :KartelaTalepKayitId, :KartelaTalepYil, :KartelaTalepId, :TalepTarihi,
         :KaliteId, :KaliteAdi, :PartiSinifId, :PartiYil, :PartiId, :HamAdi, :ReceteKodu, :ReceteAdi,
         :Denye, :SiparisProsesId, :SiparisProsesAdi, :EnIstenen, :GramajIstenen, :CekmezlikEnIstenen,
         :CekmezlikBoyIstenen, :DonmeIstenen, :SansIstenen, :Pus, :Feine, :DinlenmisGramaj, :IplikBoyu1,
         :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :TupAcikEnMayli, :EnVerilen, :GramajVerilen, :DonmeVerilen,
         :CekmezlikEnVerilen, :CekmezlikBoyVerilen, :CekmezlikEnVerilen2, :CekmezlikBoyVerilen2,
         :DesenKodu, :DesenAdi, :DesenVaryantAdi,
         :MetrajNumunesiVar, :KartelaStokMevcudu,
         :Notu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :FirmaYetkiliAdi, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :Composition,
         :TalepNotu, :BirimFiyat, :DovizId, :Gonderildi do
  begin
    if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    else PartiIdStr = '';

    if (Giris = 1) then GirisStr = 'Giris';
    else GirisStr = 'Cikis';

    if (MetrajNumunesiVar = 1) then MetrajNumunesiVarStr = 'Var';
    else MetrajNumunesiVarStr = 'Yok';

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    Select Sum(BoyaSiparisDetay.SiparisKilo) from BoyaSiparis
      Left Join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitID = BoyaSiparis.BoyaSiparisKayitID
      Where BoyaSiparis.SiparisTarihi >= :ABaslangicTarihi and BoyaSiparis.FirmaID = :FirmaId and BoyaSiparisDetay.KaliteID = :KaliteId
      into :SiparisToplamKilo;
    suspend;
  end
end
