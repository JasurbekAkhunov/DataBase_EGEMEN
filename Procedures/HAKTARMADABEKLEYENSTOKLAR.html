CREATE PROCEDURE HAKTARMADABEKLEYENSTOKLAR
RETURNS (
    HAKTARMAFISIID INTEGER,
    HAKTARMAFISIDETAYID INTEGER,
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(20),
    FISTARIHI TIMESTAMP,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    LOT VARCHAR(150),
    TEORIKKILO NUMERIC(12,3),
    KILO NUMERIC(12,3),
    ISEMRIHAZIRLANANKILO NUMERIC(12,3),
    IADEEDILENKILO NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8))
AS
begin
  for select HAktarmaFisiDetay.HAktarmaFisiId, HAktarmaFisiDetay.HAktarmaFisiDetayId, HAktarmaFisi.FisTarihi, HAktarmaFisi.FisYil, HAktarmaFisi.FisNo,
             HamStokDetay.HamId, Ham.Adi, HamStokMaster.Lot, HAktarmaFisiDetay.Kilo, HAktarmaFisiDetay.TeorikKilo, HAktarmaFisiDetay.IsEmriHazirlananKilo,
             HAktarmaFisiDetay.IadeEdilenKilo, HamStokIrsaliye.FirmaId, Firma.Adi,
             HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi from HAktarmaFisiDetay
  left join HAktarmaFisi on HAktarmaFisi.HAktarmaFisiid = HAktarmaFisiDetay.HAktarmaFisiId
  left join HamStokDetay on HamStokDetay.HamStokDetayid = HAktarmaFisiDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
  left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
  left join Ham on Ham.HamId = HamStokDetay.HamId
  where HAktarmaFisiDetay.StokVar = 1 and HAktarmaFisi.TeslimAlindiDurumu = 1
  into :HAktarmaFisiId, :HAktarmaFisiDetayId, :FisTarihi, :FisYil, :FisNo, :HamId, :HamAdi, :Lot, :Kilo, :TeorikKilo,
       :IsEmriHazirlananKilo, :IadeEdilenKilo,
       :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi do
  begin
     KalanKilo = Kilo - IsEmriHazirlananKilo - IadeEdilenKilo;
     FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';

     if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
     else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
     else EvrakTipiStr = 'Hatalı';
     suspend;
  end
end
