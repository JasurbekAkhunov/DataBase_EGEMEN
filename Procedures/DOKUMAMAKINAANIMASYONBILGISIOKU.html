CREATE PROCEDURE DOKUMAMAKINAANIMASYONBILGISIOKU(
    ADOKUMAMAKINAID VARCHAR(10))
RETURNS (
    DOKUMAMAKINAID VARCHAR(10),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ADI VARCHAR(30),
    DOKUMAPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    HAMADI VARCHAR(255),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    TOPLAMSAKATDOKUNAN DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    MIKTARGUN DOUBLE PRECISION,
    DOKUMASIPARISTIPIID INTEGER,
    GERCEKLESENTURSAYISI INTEGER,
    PLANLANANTURSAYISI INTEGER,
    TUR INTEGER,
    CALISIYOR SMALLINT,
    ARIZA INTEGER,
    SERVISCI INTEGER,
    AYAR INTEGER,
    ATKIKOPTU SMALLINT,
    COZGUKOPTU SMALLINT,
    LEVENTSONU SMALLINT,
    STARTSTOP SMALLINT,
    MARKASI VARCHAR(30),
    MAKINAGRUPADI ADI30,
    MAKINADURUMSTR VARCHAR(10),
    DOKUMACIPERSONELID INTEGER,
    DOKUMACIPERSONELADI ADI40,
    DOKUMACIPERSONELSURESISTR VARCHAR(30),
    DOKUMACIPERSONELSURESI DOUBLE PRECISION,
    MAKINAHIZI INTEGER,
    PLANLAMAMAKINAHIZI INTEGER,
    TESLIMTARIHI TIMESTAMP,
    SIPARISTARIHI TIMESTAMP,
    SIPARISMETRE BOYAMIKTARSIZE,
    ORMESIPARISSEKLIADI ADI15,
    TARAKNUMARASI VARCHAR(10),
    TARAKENI INTEGER,
    CERCEVESAYISI VARCHAR(20),
    TEZGAHTAATILACAKATKI INTEGER,
    ATKIIPI MEMO,
    COZGUIPI MEMO,
    BASLAMATARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    COZGUISEMRIID VARCHAR(15),
    COZGUPLANLANANMIKTAR BOYAMIKTARSIZE,
    COZGUURETIMMIKTAR BOYAMIKTARSIZE,
    COZGUKALANMIKTAR BOYAMIKTARSIZE)
AS
declare variable MinSira integer;
declare variable FOrmeMakinaDurumuID integer;
declare variable EskiKayit SmallInt;
declare variable HazirlikSuresi Timestamp;
declare variable Sure Double Precision;
declare variable SonDurumTarihi TimeStamp;
begin
  for Select DokumaMakina.DokumaMakinaId, min(sira), DokumaMakina.Markasi, MakinaGrup.Adi, DokumaMakina.MakinaHizi
  from DokumaMakina
    left join DokumaPlanlama on DokumaPlanlama.DokumaMakinaId = DokumaMakina.DokumaMakinaId and Bitti = 0
    left join MakinaGrup on MakinaGrup.MakinaGrupId = DokumaMakina.MakinaGrupId
    where DokumaMakina.OtomasyonaDahil = 1 and (DokumaMakina.DokumaMakinaId = :ADokumaMakinaId or :ADokumaMakinaId is null)
    Group By DokumaMakina.DokumaMakinaId, DokumaMakina.Markasi, MakinaGrup.Adi, DokumaMakina.MakinaHizi into
    :DokumaMakinaId, :MinSira, :Markasi, :MakinaGrupAdi, :MakinaHizi do
  begin
    DokumaciPersonelId = Null;
    DokumaciPersonelAdi = Null;
    DokumaciPersonelSuresiStr = '';
    DokumaciPersonelSuresi = Null;

    select OrmeMakinaDurumuId from MakinaDurumuOzet where MakinaDurumuOzet.DokumaMakinaId = :DokumaMakinaId into :FOrmeMakinaDurumuID;

    Select OrmeVardiyaPersonel.PersonelId, Personel.Adi from OrmeSonVardiyaIdGetir(Null, :DokumaMakinaId)
      left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = OrmeSonVardiyaIdGetir.OrmeVardiyaId
      left join Personel on Personel.PersonelId = OrmeVardiyaPersonel.PersonelId
      where OrmeVardiyaPersonel.DokumaMakinaId = :DokumaMakinaId
      into :PersonelId, :PersonelAdi;

    if (MinSira is null) then
    begin
      BoyaSiparisYil      = Null;
      BoyaSiparisId       = Null;
      BoyaSiparisIdStr    = Null;
      SiparisFirmaId        = Null;
      SiparisFirmaAdi       = Null;
      DokumaPlanlamaId        = Null;
      UretimPartiNo         = Null;
      HamAdi                = Null;
      PlanlananMiktar       = Null;
      GerceklesenMiktar     = Null;
      ToplamSakatDokunan    = Null;
      KalanMiktar           = Null;
      MiktarGun             = Null;
      DokumaSiparisTipiId   = Null;
      SiparisGrupAdi      = Null;
      SiparisGrup2Adi      = Null;
      PlanlananTurSayisi      = Null;
      GerceklesenTurSayisi    = Null;
      PlanlamaMakinaHizi = Null;
      TeslimTarihi = Null;
      SiparisTarihi = Null;
      SiparisMetre = Null;
      OrmeSiparisSekliAdi = Null;
      TarakNumarasi = Null;
      TarakEni = Null;
      CerceveSayisi = Null;
      TezgahtaAtilacakAtki = Null;
      AtkiIpi = Null;
      CozguIpi = Null;
      BaslamaTarihi = Null;
      BitisTarihi = Null;
      CozguIsEmriId = Null;
      CozguPlanlananMiktar = Null;
      CozguUretimMiktar = Null;
      CozguKalanMiktar = Null;

      if (FOrmeMakinaDurumuID is not Null) then
      begin
        Select Tur, ArizaVar, ServisciVar, AyarYapiliyor, AtkiKoptu, CozguKoptu, LeventSonu, StartStop, iif(Current_TimeStamp(2) - Tarih > 0.0014, 1, 0), Personel.PersonelId, Personel.Adi, Tarih from OrmeMakinaDurumu
        left join Personel on Personel.PersonelId = OrmeMakinaDurumu.PersonelId
        where OrmeMakinaDurumu.OrmeMakinaDurumuID = :FOrmeMakinaDurumuID
          into :Tur, :Ariza, :Servisci, :Ayar, :AtkiKoptu, :CozguKoptu, :LeventSonu, :StartStop, :EskiKayit, :DokumaciPersonelId, :DokumaciPersonelAdi, :SonDurumTarihi;

        if (Ariza = 0 and Servisci = 0 and Ayar = 0 and AtkiKoptu = 0 and CozguKoptu = 0 and LeventSonu = 0 and StartStop = 0 and EskiKayit = 0) then Calisiyor = 1;
        else Calisiyor = 0;

        if (DokumaciPersonelId is not null) then
        begin
          select :SonDurumTarihi - Min(Tarih) from OrmeMakinaDurumu OMD where OMD.Tarih > (
            select Coalesce(Max(OrmeMakinaDurumu.Tarih),current_timestamp-365) from OrmeMakinaDurumu where OrmeMakinaDurumu.DokumaMakinaId = :DokumaMakinaId and OrmeMakinaDurumu.PersonelId <> :DokumaciPersonelId
          ) and OMD.DokumaMakinaId = :DokumaMakinaId and OMD.PersonelId = :DokumaciPersonelId into :DokumaciPersonelSuresi;

          if (DokumaciPersonelSuresi is not null) then
            DokumaciPersonelSuresiStr = TarihToStr(DokumaciPersonelSuresi);
        end
       end
       else
       begin
         Tur        = 0;
         Calisiyor  = 0;
         Ariza      = Null;
         Servisci   = Null;
         Ayar       = Null;
         AtkiKoptu  = Null;
         CozguKoptu = Null;
         LeventSonu = Null;
         StartStop  = Null;
       end

    end
    else
    begin
      Select first 1 BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, SiparisGrup.Adi, SiparisGrup2.Adi, DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlama.UretimPartiNo,
             Ham.Adi, DokumaPlanlama.PlanlananMiktar,
             DokumaPlanlama.GerceklesenMiktar, DokumaPlanlama.ToplamSakatDokunan, DokumaPlanlama.MiktarGun, BoyaSiparis.DokumaSiparisTipiId,
             DokumaPlanlama.PlanlananTurSayisi, DokumaPlanlama.MakinaHizi, OrmeSiparisDetay.TeslimTarihi, BoyaSiparis.SiparisTarihi,
             OrmeSiparisDetay.Hammiktar, OrmeSiparisSekli.Adi, Ham.TarakNumarasi, DokumaPlanlama.TarakEni, DokumaPlanlama.CerceveSayisi,
             DokumaPlanlama.TezgahtaAtilacakAtki, DokumaPlanlama.BaslamaTarihi, DokumaPlanlama.HazirlikSuresi from DokumaPlanlama
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
        left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
        left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
        left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
        left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
        left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
        left join OrmeSiparisSekli on OrmeSiparisSekli.OrmeSiparisSekliId = BoyaSiparis.OrmeSiparisSekliId
        where DokumaPlanlama.DokumaMakinaId = :DokumaMakinaId and DokumaPlanlama.Sira = :MinSira and DokumaPlanlama.Bitti = 0
        into :BoyaSiparisYil, :BoyaSiparisId, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisGrupAdi, :SiparisGrup2Adi, :DokumaPlanlamaId, :UretimPartiNo,
        :HamAdi, :PlanlananMiktar, :GerceklesenMiktar, :ToplamSakatDokunan, :MiktarGun, :DokumaSiparisTipiId, :PlanlananTurSayisi, :PlanlamaMakinaHizi,
        :TeslimTarihi, :SiparisTarihi, :SiparisMetre, :OrmeSiparisSekliAdi, :TarakNumarasi, :TarakEni, :CerceveSayisi, :TezgahtaAtilacakAtki, :BaslamaTarihi, :HazirlikSuresi;

       select First 1 CozguIsEmri.IsEmriId, DokumaStokPlanlama.Miktar, Sum(DokumaKaliteKontrol.Miktar) From DokumaStokPlanlama
       left join OrmeRaf on OrmeRaf.OrmeRafId = DokumaStokPlanlama.OrmeRafId
       left join CozguIsEmri on CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId
       left join DokumaKaliteKontrol on DokumaKaliteKontrol.OrmeRafId = OrmeRaf.OrmeRafId and DokumaKaliteKontrol.DokumaPlanlamaId = :DokumaPlanlamaId
       where DokumaStokPlanlama.DokumaPlanlamaId = :DokumaPlanlamaId
       group by CozguIsEmri.IsEmriId, DokumaStokPlanlama.Miktar
       into :CozguIsEmriId, :CozguPlanlananMiktar, :CozguUretimMiktar;

       CozguKalanMiktar = Coalesce(CozguPlanlananMiktar,0) - Coalesce(CozguUretimMiktar,0);

       select AtkiIpi, CozguIpi from DokumaPlanlamaAtkiCozguGetir(:DokumaPlanlamaId) into :AtkiIpi, :CozguIpi;

       KalanMiktar = PlanlananMiktar - GerceklesenMiktar + ToplamSakatDokunan;

       if (MiktarGun > 0 and KalanMiktar > 0) then Sure = KalanMiktar / MiktarGun;
       else Sure = 0;

       BitisTarihi = DoubleToTimeStamp(TimeStampToDouble(Current_TimeStamp(0)) + Sure);

       BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

       if (FOrmeMakinaDurumuID is not Null) then
       begin
          Select Tur, ArizaVar, ServisciVar, AyarYapiliyor, AtkiKoptu, CozguKoptu, LeventSonu, StartStop, iif(Current_TimeStamp(2) - Tarih > 0.0014, 1, 0), Personel.PersonelId, Personel.Adi, Tarih from OrmeMakinaDurumu
          left join Personel on Personel.PersonelId = OrmeMakinaDurumu.PersonelId
          where OrmeMakinaDurumu.OrmeMakinaDurumuID = :FOrmeMakinaDurumuID
            into :Tur, :Ariza, :Servisci, :Ayar, :AtkiKoptu, :CozguKoptu, :LeventSonu, :StartStop, :EskiKayit, :DokumaciPersonelId, :DokumaciPersonelAdi, :SonDurumTarihi;

         if (Ariza = 0 and Servisci = 0 and Ayar = 0 and AtkiKoptu = 0 and CozguKoptu = 0 and LeventSonu = 0 and StartStop = 0 and EskiKayit = 0) then Calisiyor = 1;
         else Calisiyor = 0;

        if (DokumaciPersonelId is not null) then
        begin
          select :SonDurumTarihi - Min(Tarih) from OrmeMakinaDurumu OMD where OMD.Tarih > (
            select Coalesce(Max(OrmeMakinaDurumu.Tarih),current_timestamp-365) from OrmeMakinaDurumu where OrmeMakinaDurumu.DokumaMakinaId = :DokumaMakinaId and OrmeMakinaDurumu.PersonelId <> :DokumaciPersonelId
          ) and OMD.DokumaMakinaId = :DokumaMakinaId and OMD.PersonelId = :DokumaciPersonelId into :DokumaciPersonelSuresi;

          if (DokumaciPersonelSuresi is not null) then
            DokumaciPersonelSuresiStr = TarihToStr(DokumaciPersonelSuresi);
        end
       end
       else
       begin
         Tur       = 0;
         Calisiyor = 0;
         Ariza            = Null;
         Servisci         = Null;
         Ayar             = Null;
         AtkiKoptu    = Null;
         CozguKoptu        = Null;
         LeventSonu      = Null;
         StartStop        = Null;
       end
    end

    if (Calisiyor = 0) then
    begin
      if (EskiKayit = 1) then
        MakinaDurumStr = 'Bilgi Yok';
      else if (StartStop = 1 and AtkiKoptu = 0 and CozguKoptu = 0 and LeventSonu = 0) then
        MakinaDurumStr = 'Duruyor';
      else
        MakinaDurumStr = 'Arızalı';
    end
    else
      MakinaDurumStr = 'Çalışıyor';
    suspend;
  end
end
