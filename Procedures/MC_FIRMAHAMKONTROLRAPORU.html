CREATE PROCEDURE MC_FIRMAHAMKONTROLRAPORU(
    ARAPORHAMTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AFIRMAID VARCHAR(20),
    AKULLANICIID INTEGER)
RETURNS (
    HAMSTOKIRSALIYEID INTEGER,
    HAMSTOKMASTERID INTEGER,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30),
    HAMADI VARCHAR(255),
    PUS INTEGER,
    FEINE INTEGER,
    ORMECIMAKINANO VARCHAR(10),
    EN INTEGER,
    GRM2 INTEGER,
    RENK VARCHAR(20),
    TUPACIKENMAYLISTR VARCHAR(10),
    LOT VARCHAR(150),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    ACIKLAMA BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TAMIR INTEGER,
    TAMIRSTR VARCHAR(5),
    TAMIRTIPIADI VARCHAR(20),
    ESKIPARTINO INTEGER,
    MUSTERIPARTINO VARCHAR(30))
AS
declare variable HstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable TupAcikEnMayli integer;
declare variable Okey     integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  for select HamStokMaster.HamStokMasterId, HamStokMaster.HamStokIrsaliyeId, HamStokIrsaliye.IrsaliyeSeriNo,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.OrmeciMakinaNo, HamStokMaster.En, HamStokMaster.Grm2,  HamStokMaster.Renk, HamStokMaster.Lot, HamStokMaster.TupAcikEnMayli,
             HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Adet, HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartino, HamStokMaster.MusteriPartiNo from HamStokIrsaliye
  left join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
  left join Ham on Ham.HamId = HamStokMaster.HamId
  left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
  left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  where ( (HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and :AIrsaliyeNo is null) or
          (HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo and :AIrsaliyeNo is not null) ) and
        (HamStokIrsaliye.FirmaId = :AFirmaId) and HamStokMaster.Fason  = 1 and
        HamStokMaster.KontrolKullaniciId is not null and
         ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporHamTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporHamTipi = 1) or :ARaporHamTipi is null)
  group By HamStokMaster.HamStokMasterId, HamStokMaster.HamStokIrsaliyeId, HamStokIrsaliye.IrsaliyeSeriNo,
           HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
           Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.OrmeciMakinaNo, HamStokMaster.En, HamStokMaster.Grm2,  HamStokMaster.Renk, HamStokMaster.Lot, HamStokMaster.TupAcikEnMayli,
           HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Adet, HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartino, HamStokMaster.MusteriPartiNo
  into :HamStokMasterId, :HamStokIrsaliyeId, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi,
       :HamAdi, :Pus, :Feine, :OrmeciMakinaNo, :En, :Grm2, :Renk, :Lot, :TupAcikEnMayli, :KapAdedi,  :Kilo, :Metre,
       :Adet, :Tamir, :TamirTipiAdi, :EskiPartino, :MusteriPartiNo do
  begin
    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    if (Tamir = 1) then TamirStr = 'Tamir';
    else TamirStr = '';

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';
    suspend;
  end
end
