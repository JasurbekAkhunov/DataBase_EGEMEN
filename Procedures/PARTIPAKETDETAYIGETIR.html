CREATE PROCEDURE PARTIPAKETDETAYIGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIPAKETDETAYID INTEGER,
    PARTIPAKETID INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTIILAVEISLEMLERID INTEGER,
    YAPILANISLEM VARCHAR(60),
    ASAMAID INTEGER,
    SIPARISILAVEISLEMLERID INTEGER)
AS
Declare variable AsamaAdi varchar(40);
Declare variable ReceteId varchar(20);
Declare variable IlaveIslemAdi varchar(30);
begin
  For select PartiPaketDetay.PartiPaketDetayId, PartiPaket.PartiPaketId, PartiPaketDetay.PartiAsamalariId, PartiPaketDetay.PartiIlaveIslemlerId, PartiPaketDetay.AsamaId,
        PartiPaketDetay.SiparisIlaveIslemlerId, Asama.Adi, partiAsamalari.ReceteId, IlaveIslemler.Adi from PartiPaket
  left join PartiPaketDetay on PartiPaketDetay.PartiPaketId = PartiPaket.PartiPaketId
  left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiPaketDetay.PartiAsamalariId
  left join PartiIlaveIslemler on PartiIlaveIslemler.PartiIlaveIslemlerId = PartiPaketDetay.PartiIlaveIslemlerId
  left join SiparisIlaveIslemler on SiparisIlaveIslemler.SiparisIlaveIslemlerId = PartiPaketDetay.SiparisIlaveIslemlerId
  left join Recete on Recete.ReceteId = partiAsamalari.ReceteId
  left join asama on asama.AsamaId = partiAsamalari.AsamaId or asama.AsamaId = recete.UstAsamaId or asama.AsamaId = PartiPaketDetay.AsamaId
  left join ilaveislemler on ilaveislemler.ilaveislemlerid = PartiIlaveIslemler.IlaveIslemlerId or ilaveislemler.ilaveislemlerid = SiparisIlaveIslemler.IlaveIslemlerId
  where PartiPaket.PartiKayitId = :APartiKayitId and PartiPaketDetay.PartiPaketDetayId is not null
  order by PartiPaket.PartiPaketId, PartiPaketDetay.PartiPaketDetayId
  into :PartiPaketDetayId, :PartiPaketId, :PartiAsamalariId, :PartiIlaveIslemlerId, :AsamaId, :SiparisIlaveIslemlerId, :AsamaAdi, :ReceteId, :IlaveIslemAdi do
  begin
    YapilanIslem = AsamaAdi;
    if (ReceteId is not null) then YapilanIslem = ReceteId;


    if (IlaveIslemAdi is not null) then
    begin
      if (YapilanIslem <> '') then YapilanIslem = YapilanIslem || '-';
      else YapilanIslem = '';

      YapilanIslem = YapilanIslem || IlaveIslemAdi;
    end

    suspend;
  end
end
