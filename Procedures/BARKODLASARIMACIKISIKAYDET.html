CREATE PROCEDURE BARKODLASARIMACIKISIKAYDET(
    AIPLIKLOTID INTEGER,
    AIPLIKSTOKID INTEGER,
    AKULLANICIID INTEGER,
    ACIKISISIL INTEGER,
    ACIKISIPLIKIRSALIYEDETAYID INTEGER)
AS
declare variable GirisIplikIrsaliyeDetayId integer;
declare variable CikisIplikIrsaliyeDetayId integer;
declare variable HamId integer;
declare variable IplikDepoId    varchar(10);
declare variable IplikIrsaliyeId integer;
declare variable IplikIrsaliyeDetayId integer;
declare variable Lot                  varchar(30);
declare variable MelanjKodu           varchar(30);
declare variable FlamKodu             varchar(30);
declare variable RenkId               integer;
declare variable Miktar               numeric(12,3);
declare variable NetMiktar            numeric(12,3);
declare variable AmbalajId            integer;
declare variable KS integer;
declare variable SarimSekliId integer;
declare variable shtSarimaCikis integer;
declare variable MaxNumarator integer;
declare variable hstIplik integer;
declare variable HamTipiId integer;
declare variable SeriNo                SeriNoSize5;
declare variable IrsaliyeNo            IrsaliyeNoSize;
declare variable SistemDovizId  varchar(5);
declare variable KayitSayisi   integer;
begin
  Select SistemDovizId from Sistem where sistemId = 1 into :SistemDovizId;

  select shtSarimaCikis from IplikStokHareketTipiGetir into :shtSarimaCikis;

  Select hstIplik from HamTipiGetir into :hstIplik;

  if (ACikisiSil = 0) then
    Select Max(Numarator) from IplikStok where IplikStok.IplikStokID = :AIplikStokId into :MaxNumarator;
  else
    Select Max(Numarator) from IplikStok where IplikStok.IplikStokID = :AIplikStokId and CikisIplikIrsaliyeDetayID = :ACikisIplikIrsaliyeDetayId into :MaxNumarator;

  if (MaxNumarator is null) then exception KayitHatasi ' Stok Bulunamadı, Status-1';

  Select IplikStok.CikisIplikIrsaliyeDetayId, IplikStok.IplikIrsaliyeDetayId, IplikStok.HamId, IplikStok.IplikDepoId,
         IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu, IplikIrsaliyeDetay.RenkId, IplikIrsaliyeDetay.AmbalajId, IplikStok.Miktar, IplikStok.NetMiktar, Ham.HamTipiId,
         IplikIrsaliyeDetay.SarimSekliId from IplikStok
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.IplikIrsaliyeDetayId
    left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
    where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator
    into :CikisIplikIrsaliyeDetayId, :GirisIplikIrsaliyeDetayId, :HamId, :IplikDepoId, :Lot, :MelanjKodu, :FlamKodu,  :RenkId, :AmbalajId, :Miktar, :NetMiktar, :HamTipiId,
         :SarimSekliId;

  if (GirisIplikIrsaliyeDetayId is null) then exception KayitHatasi ' Stok Bulunamadı, Status-2';

  if (ACikisiSil = 0) then
  if (CikisIplikIrsaliyeDetayId is not null) then exception KayitHatasi ' Stok Daha Önce Cıkılmıs, Status-3';

  if (HamTipiId <> hstIplik) then exception KayitHatasi ' Stok İplik Değil. Status-4';

  if (ACikisiSil = 0) then
  begin
    Select Max(IplikIrsaliyeId) from IplikIrsaliye
      where IplikIrsaliye.IrsaliyeTarihi = Current_date and IplikIrsaliye.HareketTipi = :shtSarimaCikis
      into :IplikIrsaliyeId;

    if (IplikIrsaliyeId is null) then
    begin
      Select IplikUretimeCikisSeriNo from Kullanici where KullaniciId = :AKullaniciId into :SeriNo;

      if (SeriNo is null) then SeriNo = '';

      Select IrsaliyeNo from IrsaliyeNumaratorAl(:SeriNo) into :IrsaliyeNo;

      IplikIrsaliyeId = gen_id(IplikIrsaliye_Id_Gen, 1);

      insert into IplikIrsaliye(IplikIrsaliyeId, HareketTipi, SeriNo, IrsaliyeNo, IrsaliyeTarihi, DovizId, InsertKullaniciId, KullaniciId)
               values(:IplikIrsaliyeId, :shtSarimaCikis, :SeriNo, :IrsaliyeNo, Current_date, :SistemDovizId, :AKullaniciId, :AKullaniciId);
    end


    Select Min(IplikIrsaliyeDetay.IplikIrsaliyeDetayId) from IplikIrsaliyeDetay
      left join IplikStok on IplikStok.CikisIplikIrsaliyeDetayId = IplikIrsaliyeDetay.IplikIrsaliyeDetayId
      where IplikIrsaliyeDetay.IplikIrsaliyeId = :IplikIrsaliyeId and
            IplikIrsaliyeDetay.HamID = :HamId and
            IplikIrsaliyeDetay.Lot = :Lot and
            IplikIrsaliyeDetay.MelanjKodu = :MelanjKodu and
            IplikIrsaliyeDetay.FlamKodu = :FlamKodu and
            IntKarsilastir(IplikIrsaliyeDetay.AmbalajId, :AmbalajId) = 1 and
            IplikIrsaliyeDetay.SarimSekliId = :SarimSekliId and
            IplikIrsaliyeDetay.IplikDepoId = :IplikDepoId and
            IplikIrsaliyeDetay.IplikLotId = :AIplikLotId
      into :IplikIrsaliyeDetayId;

    if (IplikIrsaliyeDetayId is null) then
    begin
      IplikIrsaliyeDetayId = gen_Id(IplikIrsaliyeDetay_Id_gen, 1);
      insert into IplikIrsaliyeDetay(IplikIrsaliyeId, IplikIrsaliyeDetayId, HareketTipi, IplikDepoId, HamId, AmbalajId, RenkId, SarimSekliId, Lot, MelanjKodu, FlamKodu, IplikLotId, KapAdedi, Miktar, NetMiktar, InsertKullaniciId, KullaniciId)
                  values(:IplikIrsaliyeId, :IplikIrsaliyeDetayId, :shtSarimaCikis, :IplikDepoId, :HamId, :AmbalajId, :RenkId, :SarimSekliId, :Lot, :MelanjKodu, :FlamKodu, :AIplikLotId, 1, :Miktar, :NetMiktar, :AKullaniciID, :AKullaniciID);
    end
    else
    begin
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi + 1, Miktar = Miktar + :Miktar, NetMiktar = NetMiktar + :NetMiktar, NeredenUpdate = 1, KullaniciID = :AKullaniciID where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :IplikIrsaliyeDetayId;
    end

    Update IplikStok set CikisIplikIrsaliyeDetayId = :IplikIrsaliyeDetayId where IplikStok.IplikStokID = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;
  end
  else
  begin
    if (CikisIplikIrsaliyeDetayId is Null) then exception BuStokSatisiZatenYapilmamis;

    Select count(*) from IplikStok where IplikStok.CikisIplikIrsaliyeDetayId = :CikisIplikIrsaliyeDetayId into :KayitSayisi;

    Update IplikStok Set CikisIplikIrsaliyeDetayId = Null where IplikStok.IplikStokId = :AIplikStokId;

    if (KayitSayisi = 1) then delete from IplikIrsaliyeDetay where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :CikisIplikIrsaliyeDetayId;
    else
    begin
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi - 1, Miktar = Miktar - :Miktar, NetMiktar = NetMiktar - :NetMiktar, KullaniciID = :AKullaniciId, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :CikisIplikIrsaliyeDetayId;
    end
  end
end
