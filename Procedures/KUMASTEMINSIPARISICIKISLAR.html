CREATE PROCEDURE KUMASTEMINSIPARISICIKISLAR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    STOKHAREKETID INTEGER,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    URETIMPARTINO VARCHAR(30),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    MIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ORMESTOKKAYDIRMAID INTEGER,
    STOKGIRISNO INTEGER)
AS
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable shtKaydirmaCikis integer;
declare variable shtSarf integer;
declare variable AHamId           integer;
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, shtKaydirmaCikis, shtSarf from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim, :shtKaydirmaCikis, :shtSarf;

  Select Min(HamID) from OrmeSiparisDetay where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId into :AHamID;

  For Select OrmeRaf.UretimPartiNo, StokHareket.StokHareketId, StokHareket.HareketTipi, StokHareket.FirmaId, SHFirma.Adi,
             OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi,
             StokHareket.Miktar, StokHareket.BrutMiktar, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, StokHareket.BoyaSiparisDetayId,
             StokHareket.OrmeSiparisDetayId, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, OrmeFis.OrmeStokKaydirmaId, OrmeRaf.StokGirisNo from OrmeRaf
  left join StokHareket on StokHareket.OrmeRafId = OrmeRaf.OrmeRafID
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
  left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
  left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
  left join Firma SHFirma on SHFirma.Firmaid = StokHareket.Firmaid
  where OrmeRaf.BoyaSiparisKayitID = :ABoyaSiparisKayitID and OrmeRaf.HamID = :AHamID and StokHareket.StokHareketId is not null and
        (StokHareket.BoyaSiparisKayitID <> :ABoyaSiparisKayitID or StokHareket.BoyaSiparisKayitId is null)
  into :UretimPartiNo, :StokHareketId, :HareketTipi, :FirmaId, :FirmaAdi, :Fisno, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi,
       :Miktar, :BrutMiktar,  :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
       :OrmeSiparisDetayId, :MusteriSiparisNo, :ReferansSiparisNo, :OrmeStokKaydirmaId, :StokGirisNo do
  begin
    if (HareketTipi = shtSatis) then HareketTipiStr = 'Satış';
    else if (HareketTipi = shtFasonUretimeCikis) then HareketTipiStr = 'Fason Üretime Çıkış';
    else if (HareketTipi = shtUretimeCikis) then HareketTipiStr = 'Üretime Çıkış';
    else if (HareketTipi = shtKaydirmaCikis) then HareketTipiStr = 'Stok Kaydırma (Çıkış)';
    else if (HareketTipi = shtSarf) then HareketTipiStr = 'Sarf';
    else HareketTipiStr = 'Hareket Tipi Hatalı...!!!';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    suspend;
  end
end
