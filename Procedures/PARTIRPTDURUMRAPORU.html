CREATE PROCEDURE PARTIRPTDURUMRAPORU(
    ALABRECETEGRUPID INTEGER,
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    PARTIRPTID INTEGER,
    PARTIKAYITID INTEGER,
    TARIH TIMESTAMP,
    DEGERLENDIRME SMALLINT,
    DEGERLENDIRMETARIHI TIMESTAMP,
    DEGERLENDIRENKULLANICIID INTEGER,
    BASLAMATARIHI TIMESTAMP,
    BASLAMAKULLANICIID INTEGER,
    BITISTARIHI TIMESTAMP,
    BITISKULLANICIID INTEGER,
    NOTU VARCHAR(255),
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    GECENSURE VARCHAR(40),
    SISTEMTARIHI TIMESTAMP,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    FIRMAID VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ACILISTARIHI TIMESTAMP,
    BOYASIPARISYIL INTEGER,
    PARTINOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    RECETEYAZILDI SMALLINT,
    BOYASIPARISID INTEGER,
    LABRECETEGRUPID INTEGER,
    LABRECETEGRUPADI VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DEGERLENDIRENKULLANICIADI VARCHAR(40),
    BASLAMAKULLANICIADI VARCHAR(40),
    BITISKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40),
    PARTIKILO NUMERIC(12,3),
    HAMADI VARCHAR(255),
    LOT VARCHAR(150),
    MUSTERIPARTINO VARCHAR(100),
    ORMECIMAKINANO VARCHAR(100),
    BUYUKSINIFSIPARISNOTU VARCHAR(1024),
    BUYUKSINIFSIPARISPLANLAMANOTU VARCHAR(1024),
    BUYUKSINIFMUSTERIRENKKODUVEADI VARCHAR(30),
    SONBOYANANPARTIYIL INTEGER,
    SONBOYANANPARTIID INTEGER,
    SONBOYANANPARTIIDSTR VARCHAR(30),
    SONBOYANANPARTIHAMADI VARCHAR(255),
    SONBOYANANPARTILOT VARCHAR(255),
    LABRECETENOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LABRECETEATISNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    RECETETURUID INTEGER,
    RECETETURUADI VARCHAR(15),
    LABRENKGRUPID INTEGER,
    LABRENKGRUPADI VARCHAR(30),
    ESKILABRECETEKODU VARCHAR(20),
    HAMSTOKMASTERID VARCHAR(255),
    REFERANSSIPARISNO VARCHAR(255),
    SIPARISPROSESADI VARCHAR(1024),
    DEGERLENDIRMESTR VARCHAR(100),
    BEKLEMENOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SONYAPILANASAMAID INTEGER,
    SONYAPILANASAMAADI VARCHAR(40),
    SONRAKIASAMAADI VARCHAR(40),
    SIPARISILAVEISLEMLER VARCHAR(1024),
    ONCELIKNO INTEGER,
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(120))
AS
declare variable PartiSinifId integer;
declare variable PartiAsamalariId integer;
declare variable SonrakiSiraId Integer;
declare variable SonSira Integer;
declare variable Sorgu varchar(8000);
declare variable Sorgu2 varchar(4000);
begin
  /*Emre: Rapor tipine göre farklılasan kisim yavasliga neden oluyordu. Bu yüzden sorguyu önce oluşturup sonra çalıştırttım.*/
  Sorgu = 'Select PartiRpt.PartiRptId, PartiRpt.PartiKayitId, PartiRpt.Tarih, PartiRpt.Degerlendirme, PartiRpt.DegerlendirmeTarihi, PartiRpt.DegerlendirenKullaniciId, '||
      'PartiRpt.BaslamaTarihi, PartiRpt.BaslamaKullaniciId, PartiRpt.BitisTarihi, PartiRpt.BitisKullaniciId, PartiRpt.Notu, PartiRpt.KullaniciId, PartiRpt.DegisimTarihi, '||
      '(PartiRpt.BitisTarihi-PartiRpt.BaslamaTarihi) GecenSure, (Current_Date + Current_Time) SistemTarihi, Parti.PartiYil, Parti.PartiId, Parti.FirmaId, Parti.ReceteId, '||
      'Recete.Adi ReceteAdi, Parti.AcilisTarihi, BoyaSiparis.BoyaSiparisYil, Parti.Notu PartiNotu, Parti.ReceteYazildi, BoyaSiparis.BoyaSiparisId, Recete.LabReceteGrupId, '||
      'LabReceteGrup.Adi LabReceteGrupAdi, Firma.Adi FirmaAdi, DEGK.Adi DegerlendirenKullaniciAdi, BASK.Adi BaslamaKullaniciAdi, BITK.Adi BitisKullaniciAdi, DK.Adi KullaniciAdi, '||
      'Sum(PartiStok.Kilo) PartiKilo, PartiLotBilgisi.Lot, PartiLotBilgisi.MusteriPartiNo, PartiLotBilgisi.OrmeciMakinaNo,PartiLotBilgisi.BuyukSinifSiparisNotu, '||
      'PartiLotBilgisi.BuyukSinifSiparisPlanlamaNotu, PartiLotBilgisi.BuyukSinifMusteriRenkKoduVeAdi, ReceteSonBoyananPartiBilgisi.SonBoyananPartiYil, ReceteSonBoyananPartiBilgisi.SonBoyananPartiId, '||
      'ReceteSonBoyananPartiBilgisi.SonBoyananPartiIdStr, ReceteSonBoyananPartiBilgisi.SonBoyananPartiHamAdi, ReceteSonBoyananPartiBilgisi.SonBoyananPartiLot, LabRecete.Notu LabReceteNotu, '||
      'LabReceteAtis.Notu LabReceteAtisNotu, Recete.ReceteTuruId, ReceteTuru.Adi ReceteTuruAdi, Recete.LabRenkGrupId, LabRenkGrup.Adi LabRenkGrupAdi, LabRecete.EskiLabReceteKodu, '||
      'PartiLotBilgisi.HamStokMasterId, PartiLotBilgisi.ReferansSiparisNo, PartiLotBilgisi.SiparisProsesAdi, PartiLotBilgisi.SiparisIlaveIslemler, '||
      'case when Degerlendirme = 0 then ''Yapılmadı'' '||
      '                        when Degerlendirme = 1 then ''Rpt Gerekmiyor'' '||
      '                        when Degerlendirme = 2 then ''Rpt Çalışılacak'' '||
      '                        when Degerlendirme = 3 then ''Rpt Çalışılacak (Ham Madde Bekliyor)'' end DegerlendirmeStr, '||
      'Max(PartiStokBekle.Notu ) BeklemeNotu, PartiRpt.OncelikNo, IslemGrubu.IslemGrubuId, IslemGrubu.Adi '||
    'From PartiRpt '||
    'Left Join Parti On Parti.PartiKayitId = PartiRpt.PartiKayitId '||
    'Left Join PartiStok On PartiStok.PartiKayitId = Parti.PartiKayitId '||
    'left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiStok.PartiAsamalariId '||
    'Left Join PartiStokBekle On PartiStokBekle.PartiSinifId = PartiStok.PartiSinifId and PartiStokBekle.BITTI = 0 '||
    'Left Join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId '||
    'Left Join Firma On Firma.FirmaId = Parti.FirmaId '||
    'Left Join Recete On Recete.ReceteId = Parti.ReceteId '||
    'Left Join ReceteTuru On ReceteTuru.ReceteTuruId = Recete.ReceteTuruId '||
    'Left Join LabRenkGrup On LabRenkGrup.LabRenkGrupId = Recete.LabRenkGrupId '||
    'Left Join LabRecete On LabRecete.LabReceteId = Recete.LabReceteId '||
    'Left Join LabReceteAtis On LabReceteAtis.LabReceteAtisId = Recete.LabReceteAtisId '||
    'Left Join Kullanici DEGK On DEGK.KullaniciId = PartiRpt.DegerlendirenKullaniciId '||
    'Left Join Kullanici BASK On BASK.KullaniciId = PartiRpt.BaslamaKullaniciId '||
    'Left Join Kullanici BITK On BITK.KullaniciId = PartiRpt.BitisKullaniciId '||
    'Left Join Kullanici DK On DK.KullaniciId = PartiRpt.KullaniciId '||
    'Left Join LabReceteGrup On LabReceteGrup.LabReceteGrupId = Recete.LabReceteGrupId '||
    'Left Join PartiLotBilgisi(PartiRpt.PartiKayitId) On PartiLotBilgisi.Lot is not null '||
    'Left Join ReceteSonBoyananPartiBilgisi(Parti.ReceteId) On ReceteSonBoyananPartiBilgisi.SonBoyananPartiYil IS NOT NULL '||
    'Left Join IslemGrubu On IslemGrubu.IslemGrubuId = Parti.IslemGrubuId '||
    'Where ';
    if (ARaporTipi = 0) then
      Sorgu = Sorgu || 'PartiRpt.Degerlendirme = 0';
    else if (ARaporTipi = 1) then
      Sorgu = Sorgu || '(PartiRpt.Degerlendirme = 1 and PartiRpt.Tarih >= '''||:ABaslangicTarihi||''' and PartiRpt.Tarih <=  '''||:ABitisTarihi||''')';
    else if (ARaporTipi = 2) then
      Sorgu = Sorgu || '((PartiRpt.Degerlendirme = 2 or PartiRpt.Degerlendirme = 3) and PartiRpt.BitisTarihi is Null)';
    else if (ARaporTipi = 3) then
      Sorgu = Sorgu || '((PartiRpt.Degerlendirme = 2 or PartiRpt.Degerlendirme = 3) and PartiRpt.BitisTarihi >= '''||:ABaslangicTarihi||''' and PartiRpt.BitisTarihi <= '''||:ABitisTarihi||''')';
    else if (ARaporTipi = 4) then
      Sorgu = Sorgu || '(PartiRpt.BitisTarihi is null and PartiStokBekle.PartiStokBekleId is not null)';
    else if (ARaporTipi = 5) then
      Sorgu = Sorgu || '(PartiRpt.BitisTarihi is not null and PartiStokBekle.PartiStokBekleId is not null)';
    else if (ARaporTipi = 6) then
      Sorgu = Sorgu || '(PartiRpt.Tarih >= ''' ||:ABaslangicTarihi||''' and PartiRpt.Tarih <= '''||:ABitisTarihi||''')';

   Sorgu = Sorgu || ' and (('||Coalesce(:ALabReceteGrupId, 'null')||' is not null and Recete.LabReceteGrupId = '||coalesce(:ALabReceteGrupId, 'null')||') or '||coalesce(:ALabReceteGrupId, 'null')||' is null) '||
     'Group By PartiRpt.PartiRptId, PartiRpt.PartiKayitId, PartiRpt.Tarih, PartiRpt.Degerlendirme, PartiRpt.DegerlendirmeTarihi, PartiRpt.DegerlendirenKullaniciId, '||
        'PartiRpt.BaslamaTarihi, PartiRpt.BaslamaKullaniciId, PartiRpt.BitisTarihi, PartiRpt.BitisKullaniciId, PartiRpt.Notu, PartiRpt.KullaniciId, PartiRpt.DegisimTarihi, '||
        'Parti.PartiYil, Parti.PartiId, Parti.FirmaId, Parti.ReceteId, Recete.Adi, Parti.AcilisTarihi, BoyaSiparis.BoyaSiparisYil, Parti.Notu, Parti.ReceteYazildi, '||
        'BoyaSiparis.BoyaSiparisId, Recete.LabReceteGrupId, LabReceteGrup.Adi, Firma.Adi, DEGK.Adi, BASK.Adi, BITK.Adi, DK.Adi, PartiLotBilgisi.Lot, PartiLotBilgisi.MusteriPartiNo, '||
        'PartiLotBilgisi.OrmeciMakinaNo, PartiLotBilgisi.BuyukSinifSiparisNotu, PartiLotBilgisi.BuyukSinifSiparisPlanlamaNotu, PartiLotBilgisi.BuyukSinifMusteriRenkKoduVeAdi, '||
        'ReceteSonBoyananPartiBilgisi.SonBoyananPartiYil, ReceteSonBoyananPartiBilgisi.SonBoyananPartiId, ReceteSonBoyananPartiBilgisi.SonBoyananPartiIdStr, '||
        'ReceteSonBoyananPartiBilgisi.SonBoyananPartiHamAdi, ReceteSonBoyananPartiBilgisi.SonBoyananPartiLot, LabRecete.Notu, LabReceteAtis.Notu, Recete.ReceteTuruId, ReceteTuru.Adi, '||
        'Recete.LabRenkGrupId, LabRenkGrup.Adi, LabRecete.EskiLabReceteKodu, PartiLotBilgisi.HamStokMasterId, PartiLotBilgisi.ReferansSiparisNo, PartiLotBilgisi.SiparisProsesAdi, '||
        'PartiLotBilgisi.SiparisIlaveIslemler, PartiRpt.OncelikNo, IslemGrubu.IslemGrubuId, IslemGrubu.Adi ';

   for execute statement (Sorgu)
     into :PartiRptId, :PartiKayitId, :Tarih, :Degerlendirme, :DegerlendirmeTarihi, :DegerlendirenKullaniciId,
     :BaslamaTarihi, :BaslamaKullaniciId, :BitisTarihi, :BitisKullaniciId, :Notu, :KullaniciId, :DegisimTarihi,
     :GecenSure, :SistemTarihi, :PartiYil, :PartiId, :FirmaId, :ReceteId,
     :ReceteAdi, :AcilisTarihi, :BoyaSiparisYil, :PartiNotu, :ReceteYazildi, :BoyaSiparisId, :LabReceteGrupId,
     :LabReceteGrupAdi, :FirmaAdi, :DegerlendirenKullaniciAdi, :BaslamaKullaniciAdi, :BitisKullaniciAdi, :KullaniciAdi,
     :PartiKilo, :Lot, :MusteriPartiNo, :OrmeciMakinaNo, :BuyukSinifSiparisNotu,
     :BuyukSinifSiparisPlanlamaNotu, :BuyukSinifMusteriRenkKoduVeAdi,:SonBoyananPartiYil, :SonBoyananPartiId,
     :SonBoyananPartiIdStr, :SonBoyananPartiHamAdi, :SonBoyananPartiLot, :LabReceteNotu,
     :LabReceteAtisNotu, :ReceteTuruId, :ReceteTuruAdi, :LabRenkGrupId, :LabRenkGrupAdi, :EskiLabReceteKodu,
     :HamStokMasterId, :ReferansSiparisNo, :SiparisProsesAdi, :SiparisIlaveIslemler, :DegerlendirmeStr,
     :BeklemeNotu, :OncelikNo, :IslemGrubuId, :IslemGrubuAdi do
   begin

    select first 1 PartiStok.PartiSinifId, PartiStok.PartiAsamalariId, PartiAsamalari.Sira,  PartiStok.HamAdiToplami from PartiStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiStok.PartiAsamalariId
    where PartiStok.PartiKayitId = :PartiKayitId
    order by Miktar desc
    into :PartiSinifId, :PartiAsamalariId, :SonSira, :HamAdi;

    SonrakiSiraId = null;
    if (SonSira is null) then SonSira = -1;

    SonYapilanAsamaId  = null;
    SonYapilanAsamaAdi = null;

    if (PartiAsamalariId is not null) then
    begin
      Select Asama.AsamaId, Asama.Adi from PartiAsamalari
          left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
          left join Asama on (Asama.AsamaId = PartiAsamalari.AsamaId or Asama.AsamaId = Recete.UstAsamaId)
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariID
      where PartiAsamalari.PartiAsamalariID = :PartiAsamalariId and PartiAsamaStok.PartiSinifID = :PartiSinifId
      group By Asama.AsamaId, Asama.Adi
      into :SonYapilanAsamaId, :SonYapilanAsamaAdi;
    end

    select min(Sira) from PartiAsamalari
       left join PartiAsamaStok on PartiAsamaStok.PartiKayitId = PartiAsamalari.PartiKayitId and PartiAsamaStok.PartiSinifId = :PartiSinifId
       where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.Sira > :SonSira
            and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1
       into :SonrakiSiraId;

    SonrakiAsamaAdi = null;

    if (SonrakiSiraId is not null) then
      Select Asama.Adi from PartiAsamalari
        left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId and PartiAsamalari.ReceteId is not null
        left join Asama on (PartiAsamalari.AsamaId is not null and Asama.AsamaId = PartiAsamalari.AsamaId) or (Recete.UstAsamaId is not null and Asama.AsamaId = Recete.UstAsamaId)
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.Sira = :SonrakiSiraId
        into :SonrakiAsamaAdi;
     suspend;
   end
end
