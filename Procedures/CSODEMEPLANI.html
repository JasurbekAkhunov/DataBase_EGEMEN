CREATE PROCEDURE CSODEMEPLANI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APROTESTO INTEGER,
    ABANKACIKIS INTEGER,
    ADOVIZID VARCHAR(5),
    ARAPORDOVIZID VARCHAR(5))
RETURNS (
    AV VARCHAR(1),
    CS VARCHAR(1),
    CEKSENETID INTEGER,
    GENELKAYITID INTEGER,
    BORDROID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    ACIKLAMA VARCHAR(40),
    VADETARIHI TIMESTAMP,
    MEBLAG DOUBLE PRECISION,
    TAHSILEDILENMEBLAG DOUBLE PRECISION,
    KALANMEBLAG DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    ODEME DOUBLE PRECISION,
    TAHSILAT DOUBLE PRECISION,
    BAKIYE DOUBLE PRECISION,
    BAKIYEBA VARCHAR(1),
    CEKSENETNO VARCHAR(30),
    HESAPNO VARCHAR(40),
    AITOLDUGUBANKAADI VARCHAR(40),
    ILKBORCLU VARCHAR(80),
    KESIDEYERI VARCHAR(20),
    BANKAHESAPID VARCHAR(20),
    BANKAHESAPADI VARCHAR(40),
    KARSILIKSIZ INTEGER,
    VADETARIHIAY TIMESTAMP)
AS
declare variable Okey           integer;
declare variable FRaportarihi   Date;
declare variable DovizKuru      Double Precision;
declare variable DovizMeblag    Double precision;
declare variable FBakiye        Double Precision;
declare variable SonBordroTipi  integer;
declare variable SonHareketId  integer;
declare variable BankaHesapNo  varchar(20);
begin
  FRaporTarihi = current_TimeStamp(0);
  Bakiye = 0;
  FBakiye = 0;
  BakiyeBa = '';
  for Select CekSenetDetay.Av, CekSenetDetay.Cs, CekSenet.BordroId, CekSenet.FirmaId, Firma1.Adi, Firma1.Adi2,
             CekSenetDetay.CekSenetId, CekSenetDetay.genelKayitId, CekSenetDetay.Aciklama, CekSenetDetay.VadeTarihi,
             CekSenetDetay.Meblag, CekSenetDetay.TahsilEdilenMeblag, CekSenetDetay.DovizId, CekSenetDetay.CekSenetNo,  CekSenetDetay.IlkBorclu,
             CekSenetDetay.KesideYeri, CekSenetDetay.Karsiliksiz, CekSenetDetay.SonBordroTipi, CekSenetDetay.SonHareketId,
             banka.Adi, CekSenetDetay.HesapNo, CekSenetDetay.BankaHesapId from CekSenetDetay
    left join CekSenet on CekSenet.csBordroKayitId = CekSenetDetay.CsBordroKayitId
    left join Firma Firma1 on Firma1.FirmaId = CekSenet.FirmaId
    left join BankaHesap on BankaHesap.BankaHesapId = CekSenetDetay.BankaHesapId
    left join banka on (banka.bankaId = CekSenetDetay.AitOlduguBankaId or banka.bankaId = BankaHesap.BankaId)
    where (CekSenetDetay.VadeTarihi >= :ABaslangicTarihi and CekSenetDetay.VadeTarihi <= :ABitistarihi) and
          (CekSenetDetay.DovizId = :ADovizId or :ADovizId is null) and
          (CekSenetDetay.Karsiliksiz = :AProtesto or :AProtesto is null) and
          (CekSenetDetay.SonBordroTipi = 0 or CekSenetDetay.SonBordroTipi = 2 or CekSenetDetay.SonBordroTipi = 3 or CekSenetDetay.SonBordroTipi = 4 or CekSenetDetay.SonBordroTipi = 6) and
          (CekSenetDetay.Meblag > CekSenetDetay.TahsilEdilenMeblag)
          order by CekSenetDetay.VadeTarihi
    into :Av, :Cs, :BordroId, :FirmaId, :FirmaAdi, :FirmaAdi2, :CekSenetId, :GenelKayitId, :Aciklama, :VadeTarihi,
         :Meblag, :TahsilEdilenMeblag,  :DovizId,  :CekSenetNo, :IlkBorclu, :KesideYeri, :Karsiliksiz, :SonBordroTipi,
         :SonHareketId, :AitOlduguBankaAdi, :HesapNo, :BankaHesapNo do
  begin
    Okey = 1;

    if (Okey = 1) then
    if (ABankaCikis is not null) then
    if (ABankaCikis = 0 and SonBordroTipi = 3) then Okey = 0;
    else if (ABankaCikis = 1 and SonBordroTipi <> 3) then Okey = 0;
    if (Av = 'V') then HesapNo = BankaHesapNo;

    if (Okey = 1) then
    begin
      BankaHesapId  = Null;
      BankaHesapAdi = Null;
      if (SonBordroTipi = 3) then
      begin
        select CekSenetHareket.BankaHesapId, BankaHesap.Adi from ceksenethareketdetay
          left join CekSenethareket on CekSenetHareket.CshBordroKayitId = CekSenetHareketDetay.CshBordroKayitId
          left join BankaHesap on BankaHesap.BankaHesapId = CekSenetHareket.BankaHesapId
          where ceksenethareketdetay.cshareketdetayid = :SonHareketId
          into :BankaHesapId, :BankaHesapAdi;
      end

      if (DovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, :FRaporTarihi) into :DovizKuru;
      else DovizKuru = 1;

      KalanMeblag = Meblag - TahsilEdilenMeblag;

      DovizMeblag = KalanMeblag * DovizKuru;
      DovizMeblag = Yuvarla(DovizMeblag, 1000);

      if (Av = 'A') then
      begin
        FBakiye = FBakiye + DovizMeblag;
        Tahsilat = DovizMeblag;
        Odeme = Null;
      end
      else
      begin
        FBakiye = FBakiye - DovizMeblag;
        Odeme = DovizMeblag;
        Tahsilat = null;
      end

      if (FBakiye < 0) then
      begin
        Bakiye = FBakiye * -1;
        BakiyeBa = 'O';
      end
      else
      begin
        Bakiye = FBakiye;
        BakiyeBa = 'T';
      end
      VadeTarihiAy = TimeStampToAyIlkGun(VadeTarihi);
      suspend;
    end
  end
end
