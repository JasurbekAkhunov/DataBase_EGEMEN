CREATE PROCEDURE TRANSFERICINSARFFISILISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATRANSFERDURUMU INTEGER)
RETURNS (
    STOKID INTEGER,
    FISSERINO VARCHAR(2),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    TRANSFEREDILDI INTEGER,
    TRANSFEREDILDISTR VARCHAR(7),
    TRANSFERTARIHI TIMESTAMP,
    TRANSFERKULLANICIID INTEGER,
    TRANSFERKULLANICIADI VARCHAR(40))
AS
declare variable shtUretimeCikis integer;
begin
  select shtUretimeCikis from StokHareketTipiGetir into :shtUretimeCikis;

  for Select Stok.StokId, Stok.SeriNo, Stok.FisNo, Stok.FisTarihi,
             Stok.TransferEdildi, Stok.TransferTarihi, Stok.TransferKullaniciId, TK.Adi from Stok
    left join Kullanici TK On TK.KullaniciId = Stok.TransferKullaniciId
    where Stok.FisTarihi >= :ABaslangicTarihi and Stok.FisTarihi <= :ABitisTarihi and Stok.HareketTipi = :shtUretimeCikis and
         (Stok.TransferEdildi = :ATransferDurumu or :ATransferDurumu is null)
  into :StokId, :FisSeriNo, :FisNo, :FisTarihi, :TransferEdildi, :TransferTarihi, :TransferKullaniciId, :TransferKullaniciAdi do
  begin
    if (TransferEdildi = 1) then TransferEdildiStr = 'Yapıldı';
    else TransferEdildiStr = '';
    suspend;
  end
end
