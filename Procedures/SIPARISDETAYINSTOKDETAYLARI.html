CREATE PROCEDURE SIPARISDETAYINSTOKDETAYLARI(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    HAMADI VARCHAR(255),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    AKTARMA INTEGER,
    AKTARMASTR VARCHAR(12),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8))
AS
begin
  For Select SiparisDetayinPartiSiniflari.PartiKayitId, PartiStokDetay.PartiSinifId, Ham.Adi, PartiStokDetay.KapAdedi, PartiStokDetay.Kilo,
             PartiStokDetay.Metre, PartiStokDetay.Adet, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo,
             HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, RafDetay.Aktarma from SiparisDetayinPartiSiniflari(:ABoyaSiparisKayitId)
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = SiparisDetayinPartiSiniflari.PartiSinifId
    left join RafDetay on RafDetay.RafDetayid = PartiStokDetay.RafDetayId
    left join Ham on Ham.HamId = RafDetay.HamId
    left join HamStokDetay on HamStokDetay.HamStokDetayid = RafDetay.HamStokDetayId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
    where SiparisDetayinPartiSiniflari.PartiSinifId is not null
    group By SiparisDetayinPartiSiniflari.PartiKayitId, PartiStokDetay.PartiSinifId, Ham.Adi, PartiStokDetay.KapAdedi, PartiStokDetay.Kilo,
             PartiStokDetay.Metre, PartiStokDetay.Adet, HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo,
             HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, RafDetay.Aktarma
    into :PartiKayitId, :PartiSinifId, :HamAdi, :KapAdedi, :Kilo, :Metre, :Adet, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi,
         :Aktarma do
  begin
    if (Aktarma = 1) then AktarmaStr = 'HamAktarma';
    else AktarmaStr = '';
    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';
    suspend;
  end
end
