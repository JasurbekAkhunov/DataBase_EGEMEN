CREATE PROCEDURE DOKUMAMAKINACALISMAANIMASYON(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ADOKUMAMAKINAID MAKINAKODUSIZE,
    AMAKINAGRUPID VARCHAR(4))
RETURNS (
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    SURE DOUBLE PRECISION,
    LEVENTSONUSURESI DOUBLE PRECISION,
    LEVENTSONUSURESISTR VARCHAR(30),
    LEVENTSONUSAYISI INTEGER,
    ATKIKOPTUSURESI DOUBLE PRECISION,
    ATKIKOPTUSURESISTR VARCHAR(30),
    ATKIKOPTUSAYISI INTEGER,
    COZGUKOPTUSURESI DOUBLE PRECISION,
    COZGUKOPTUSURESISTR VARCHAR(30),
    COZGUKOPTUSAYISI INTEGER,
    STARTSTOPSURESI DOUBLE PRECISION,
    STARTSTOPSURESISTR VARCHAR(30),
    STARTSTOPSAYISI INTEGER,
    MINLEVENTSONUSURESI DOUBLE PRECISION,
    MAXLEVENTSONUSURESI DOUBLE PRECISION,
    MINLEVENTSONUSURESISTR VARCHAR(30),
    MAXLEVENTSONUSURESISTR VARCHAR(30),
    MINATKIKOPTUSURESI DOUBLE PRECISION,
    MAXATKIKOPTUSURESI DOUBLE PRECISION,
    MINATKIKOPTUSURESISTR VARCHAR(30),
    MAXATKIKOPTUSURESISTR VARCHAR(30),
    MINCOZGUKOPTUSURESI DOUBLE PRECISION,
    MAXCOZGUKOPTUSURESI DOUBLE PRECISION,
    MINCOZGUKOPTUSURESISTR VARCHAR(30),
    MAXCOZGUKOPTUSURESISTR VARCHAR(30),
    MINSTARTSTOPSURESI DOUBLE PRECISION,
    MAXSTARTSTOPSURESI DOUBLE PRECISION,
    MINSTARTSTOPSURESISTR VARCHAR(30),
    MAXSTARTSTOPSURESISTR VARCHAR(30),
    TOPLAMTUR INTEGER,
    KAPALI INTEGER,
    VERIM DOUBLE PRECISION,
    MAKINAHIZI INTEGER)
AS
declare variable Tarih  TimeStamp;
declare variable Tur integer;
declare variable DokumaPlanlamaId integer;

declare variable LastTarih            TimeStamp;
declare variable LastTur              integer;
declare variable FarkTarih            Double Precision;

declare variable LastLeventSonu  SmallInt;
declare variable LastAtkiKoptu   SmallInt;
declare variable LastCozguKoptu  SmallInt;
declare variable LastStartStop   SmallInt;
declare variable LeventSonu      SmallInt;
declare variable AtkiKoptu       SmallInt;
declare variable CozguKoptu      SmallInt;
declare variable StartStop       SmallInt;

declare variable AtkiKoptuSuresiAraToplam Double Precision;
declare variable CozguKoptuSuresiAraToplam Double Precision;
declare variable LeventSonuSuresiAraToplam Double Precision;
declare variable StartStopSuresiAraToplam Double Precision;

begin
  if (ABitisTarihi > Current_TimeStamp(0)) then ABitisTarihi = Current_TimeStamp(0);

  Sure = ABitistarihi - ABaslangicTarihi;

  For Select DokumaMakinaId, Adi, DokumaMakina.Kapali, MakinaHizi from DokumaMakina
    where (DokumaMakina.DokumaMakinaId = :ADokumaMakinaId or :ADokumaMakinaId is null) and
          (DokumaMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and DokumaMakina.OtomasyonaDahil = 1
    into :DokumaMakinaId, :DokumaMakinaAdi, :Kapali, :MakinaHizi do
  begin
    LeventSonuSuresi  = 0;
    AtkiKoptuSuresi   = 0;
    CozguKoptuSuresi  = 0;
    StartStopSuresi   = 0;

    MinLeventSonuSuresi  = 0;
    MinAtkiKoptuSuresi   = 0;
    MinCozguKoptuSuresi  = 0;
    MinStartStopSuresi   = 0;

    MaxLeventSonuSuresi  = 0;
    MaxAtkiKoptuSuresi   = 0;
    MaxCozguKoptuSuresi  = 0;
    MaxStartStopSuresi   = 0;

    AtkiKoptuSuresiAraToplam = 0;
    CozguKoptuSuresiAraToplam = 0;
    LeventSonuSuresiAraToplam = 0;
    StartStopSuresiAraToplam = 0;

    LastTur = -1;
    LastTarih = null;
    LastLeventSonu = 0;
    LastAtkiKoptu     = 0;
    LastCozguKoptu   = 0;
    LastStartStop     = 0;

    ToplamTur = 0;
    Verim = 0;
    AtkiKoptuSayisi = 0;
    CozguKoptuSayisi = 0;
    LeventSonuSayisi = 0;
    StartStopSayisi = 0;


    For select MakinaCalismaBilgisiAnimasyon.Tarih, Tur, MakinaCalismaBilgisiAnimasyon.DokumaPlanlamaId, LeventSonu, AtkiKoptu, CozguKoptu, StartStop from MakinaCalismaBilgisiAnimasyon(Null, :ABaslangicTarihi, :ABitisTarihi, :DokumaMakinaId)
    order by Tarih
         into :Tarih, :Tur, :DokumaPlanlamaId, :LeventSonu, :AtkiKoptu, :CozguKoptu, :StartStop do

    begin

      ToplamTur = ToplamTur + Tur;

      if (LastTarih is null) then LastTarih = Tarih;
      if (LastTur = -1) then LastTur = Tur;
      FarkTarih = Tarih - LastTarih;

      if (Tur > 0 and LeventSonu = 0 and CozguKoptu = 0 and AtkiKoptu = 0 and StartStop = 0) then Tur = 1;
      else Tur = 0;

      if (LastTur = -1) then LastTur = Tur;

      if (LastLeventSonu = 1 and LeventSonu = 0) then
      begin
        if (MinLeventSonuSuresi = 0 and MaxLeventSonuSuresi > 0) then MinLeventSonuSuresi = MaxLeventSonuSuresi;
        if (LeventSonuSuresiAraToplam > 0 and LeventSonuSuresiAraToplam < MinLeventSonuSuresi) then MinLeventSonuSuresi = LeventSonuSuresiAraToplam;
      end
      if (LastAtkiKoptu = 1 and AtkiKoptu = 0) then
      begin
        if (MinAtkiKoptuSuresi = 0 and MaxAtkiKoptuSuresi > 0) then MinAtkiKoptuSuresi = MaxAtkiKoptuSuresi;
        if (AtkiKoptuSuresiAraToplam > 0 and AtkiKoptuSuresiAraToplam < MinAtkiKoptuSuresi) then MinAtkiKoptuSuresi = AtkiKoptuSuresiAraToplam;
      end
      if (LastCozguKoptu = 1 and CozguKoptu = 0) then
      begin
        if (MinCozguKoptuSuresi = 0 and MaxCozguKoptuSuresi > 0) then MinCozguKoptuSuresi = MaxCozguKoptuSuresi;
        if (CozguKoptuSuresiAraToplam > 0 and CozguKoptuSuresiAraToplam < MinCozguKoptuSuresi) then MinCozguKoptuSuresi = CozguKoptuSuresiAraToplam;
      end
      if (LastStartStop = 1 and LastAtkiKoptu = 0 and LastCozguKoptu = 0 and LastLeventSonu = 0 and StartStop = 0) then
      begin
        if (MinStartStopSuresi = 0 and MaxStartStopSuresi > 0) then MinStartStopSuresi = MaxStartStopSuresi;
        if (StartStopSuresiAraToplam > 0 and StartStopSuresiAraToplam < MinStartStopSuresi) then MinStartStopSuresi = StartStopSuresiAraToplam;
      end

      if (LeventSonu = 1) then
      begin
        if (LastLeventSonu = 0) then
        begin
          LeventSonuSuresiAraToplam = FarkTarih;
          LeventSonuSayisi = LeventSonuSayisi + 1;
        end
        else LeventSonuSuresiAraToplam = LeventSonuSuresiAraToplam + FarkTarih;

        LeventSonuSuresi = LeventSonuSuresi + FarkTarih;

        if (LeventSonuSuresiAraToplam > MaxLeventSonuSuresi) then MaxLeventSonuSuresi = LeventSonuSuresiAraToplam;
      end

      if (AtkiKoptu = 1) then
      begin
        if (LastAtkiKoptu = 0) then
        begin
          AtkiKoptuSuresiAraToplam = FarkTarih;
          AtkiKoptuSayisi = AtkiKoptuSayisi + 1;
        end
        else AtkiKoptuSuresiAraToplam = AtkiKoptuSuresiAraToplam + FarkTarih;

        AtkiKoptuSuresi = AtkiKoptuSuresi + FarkTarih;

        if (AtkiKoptuSuresiAraToplam > MaxAtkiKoptuSuresi) then MaxAtkiKoptuSuresi = AtkiKoptuSuresiAraToplam;
      end

      if (CozguKoptu = 1) then
      begin
        if (LastCozguKoptu = 0) then
        begin
          CozguKoptuSuresiAraToplam = FarkTarih;
          CozguKoptuSayisi = CozguKoptuSayisi + 1;
        end
        else CozguKoptuSuresiAraToplam = CozguKoptuSuresiAraToplam + FarkTarih;

        CozguKoptuSuresi = CozguKoptuSuresi + FarkTarih;

        if (CozguKoptuSuresiAraToplam > MaxCozguKoptuSuresi) then MaxCozguKoptuSuresi = CozguKoptuSuresiAraToplam;
      end

      if (StartStop = 1 and LeventSonu = 0 and AtkiKoptu = 0 and CozguKoptu = 0) then
      begin
        if (LastStartStop = 0) then
        begin
          StartStopSuresiAraToplam = FarkTarih;
          StartStopSayisi = StartStopSayisi + 1;
        end
        else StartStopSuresiAraToplam = StartStopSuresiAraToplam + FarkTarih;

        StartStopSuresi = StartStopSuresi + FarkTarih;

        if (StartStopSuresiAraToplam > MaxStartStopSuresi) then MaxStartStopSuresi = StartStopSuresiAraToplam;
      end

      LastTarih = Tarih;
      LastTur = Tur;
      LastLeventSonu = LeventSonu;
      LastAtkiKoptu = AtkiKoptu;
      LastCozguKoptu = CozguKoptu;
      LastStartStop = StartStop;

    end

    if (MinLeventSonuSuresi = 0 and MaxLeventSonuSuresi > 0) then MinLeventSonuSuresi = MaxLeventSonuSuresi;
    if (MinAtkiKoptuSuresi = 0 and MaxAtkiKoptuSuresi > 0) then MinAtkiKoptuSuresi = MaxAtkiKoptuSuresi;
    if (MinCozguKoptuSuresi = 0 and MaxCozguKoptuSuresi > 0) then MinCozguKoptuSuresi = MaxCozguKoptuSuresi;
    if (MinStartStopSuresi = 0 and MaxStartStopSuresi > 0) then MinStartStopSuresi = MaxStartStopSuresi;

    if (LeventSonuSuresiAraToplam > 0 and LeventSonuSuresiAraToplam < MinLeventSonuSuresi) then MinLeventSonuSuresi = LeventSonuSuresiAraToplam;
    if (AtkiKoptuSuresiAraToplam > 0 and AtkiKoptuSuresiAraToplam < MinAtkiKoptuSuresi) then MinAtkiKoptuSuresi = AtkiKoptuSuresiAraToplam;
    if (CozguKoptuSuresiAraToplam > 0 and CozguKoptuSuresiAraToplam < MinCozguKoptuSuresi) then MinCozguKoptuSuresi = CozguKoptuSuresiAraToplam;
    if (StartStopSuresiAraToplam > 0 and StartStopSuresiAraToplam < MinStartStopSuresi) then MinStartStopSuresi = StartStopSuresiAraToplam;

    if (Sure > 0) then
    begin
      Verim = (ToplamTur / (Sure*24*60)) * 100 / MakinaHizi;
    end

    if (LeventSonuSuresi = 0) then LeventSonuSuresiStr = '';
    else LeventSonuSuresiStr = TarihToStr(LeventSonuSuresi);

    if (AtkiKoptuSuresi = 0) then AtkiKoptuSuresiStr = '';
    else AtkiKoptuSuresiStr = TarihToStr(AtkiKoptuSuresi);

    if (CozguKoptuSuresi = 0) then CozguKoptuSuresiStr = '';
    else CozguKoptuSuresiStr = TarihToStr(CozguKoptuSuresi);

    if (StartStopSuresi = 0) then StartStopSuresiStr = '';
    else StartStopSuresiStr = TarihToStr(StartStopSuresi);


    if (MinLeventSonuSuresi = 0) then MinLeventSonuSuresiStr = '';
    else MinLeventSonuSuresiStr = TarihToStr(MinLeventSonuSuresi);
    if (MaxLeventSonuSuresi = 0) then MaxLeventSonuSuresiStr = '';
    else MaxLeventSonuSuresiStr = TarihToStr(MaxLeventSonuSuresi);

    if (MinAtkiKoptuSuresi = 0) then MinAtkiKoptuSuresiStr = '';
    else MinAtkiKoptuSuresiStr = TarihToStr(MinAtkiKoptuSuresi);
    if (MaxAtkiKoptuSuresi = 0) then MaxAtkiKoptuSuresiStr = '';
    else MaxAtkiKoptuSuresiStr = TarihToStr(MaxAtkiKoptuSuresi);

    if (MinCozguKoptuSuresi = 0) then MinCozguKoptuSuresiStr = '';
    else MinCozguKoptuSuresiStr = TarihToStr(MinCozguKoptuSuresi);
    if (MaxCozguKoptuSuresi = 0) then MaxCozguKoptuSuresiStr = '';
    else MaxCozguKoptuSuresiStr = TarihToStr(MaxCozguKoptuSuresi);

    if (MinStartStopSuresi = 0) then MinStartStopSuresiStr = '';
    else MinStartStopSuresiStr = TarihToStr(MinStartStopSuresi);
    if (MaxStartStopSuresi = 0) then MaxStartStopSuresiStr = '';
    else MaxStartStopSuresiStr = TarihToStr(MaxStartStopSuresi);

    suspend;
  end
end
