CREATE PROCEDURE UCPASTALHAMLISTESI(
    AORMEMAKINAID VARCHAR(10),
    AFIRMAID VARCHAR(20),
    AKESIMTALIMATI INTEGER,
    ATASNIFMETO INTEGER)
RETURNS (
    HGISEMRIKAYITID INTEGER,
    HGISEMRIPASTALID INTEGER,
    PASTALEN BOYAMIKTARSIZE,
    PASTALBOY BOYAMIKTARSIZE,
    KATSAYISI INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI ADI50,
    LABRECETEID INTEGER,
    LABRECETEADI VARCHAR(40),
    LABRECETEKODU RECETEKODUSIZE,
    LABRECETEATISID INTEGER,
    LABRECETEATISNO VARCHAR(20),
    PANTONEKODU VARCHAR(30),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(20),
    BOYALIGRAMAJ INTEGER,
    BOYALIEN INTEGER,
    PLANLANANMIKTAR INTEGER,
    GERCEKLESENMIKTAR INTEGER,
    METOMIKTAR INTEGER,
    TASNIFMIKTAR INTEGER)
AS
Declare Variable FDurum EgeBoolean;
begin
  for Select HGIsEmri.HGIsEmriKayitId, HGIsEmriPastal.HGIsEmriPastalID, HGIsEmriPastal.En, HGIsEmriPastal.Boy, HGIsEmriPastalKumas.KatSayisi,
             HGSiparisDetayKumas.HamId, Ham.Adi, HGSiparisDetayKumas.RenkId, HGSiparisDetayKumas.LabReceteId, LabRecete.Adi, HGSiparisDetayKumas.LabReceteKodu, HGSiparisDetayKumas.LabReceteAtisID,
             LabReceteAtis.AtisNoStr, HGSiparisDetayKumas.RenkAdi, HGSiparisDetayKumas.PantoneKodu, HGSiparisDetayKumas.TupAcikEnMayli,
             HGSiparisDetayKumas.BoyaliGramaj, HGSiparisDetayKumas.BoyaliEn from HGIsEmri
    left join HGIsEmriPastal on HGIsEmriPastal.HGIsEmriKayitID = HGIsEmri.HGIsEmriKayitID
    left join HGIsEmriPastalKumas on HGIsEmriPastalKumas.HGIsEmriPastalID = HGIsEmriPastal.HGIsEmriPastalID
    left join HGSiparisDetayKumas on HGSiparisDetayKumas.HGSiparisDetayKumasID = HGIsEmriPastalKumas.HGSiparisDetayKumasID
    left join Ham on Ham.HamId = HGSiparisDetayKumas.HamId
    left join LabRecete on LabRecete.LabReceteId = HGSiparisDetayKumas.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = HGSiparisDetayKumas.LabReceteAtisId
    where ((HGIsEmri.OrmeMakinaId = :AOrmeMakinaId and :AOrmeMakinaId is not Null and HgIsEmri.Planlandi = 1) or (:AOrmeMakinaId is Null) or
           (HGIsEmri.FirmaID = :AFirmaId and :AFirmaId is not Null) ) and
           (HGIsEmri.Bitti = 0 or :ATasnifMeto > -1) and HGIsEmri.OnayDurumu = 1 and
           (HGIsEmri.KesimTalimati = :AKesimTalimati or :AKesimTalimati is Null)
    into :HGIsEmriKayitId, :HGIsEmriPastalID, :PastalEn, :PastalBoy, :KatSayisi, :HamId, :HamAdi, :RenkId,
         :LabReceteId, :LabReceteAdi, :LabReceteKodu, :LabReceteAtisID, :LabReceteAtisNo, :RenkAdi, :PantoneKodu,
         :TupAcikEnMayli, :BoyaliGramaj, :BoyaliEn do
  begin
    FDurum = 1;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikEnMayli) into :TupAcikEnMayliStr;

    Select Sum(HGIsEmriPastalUrun.PlanlananMiktar), Sum(HGIsEmriPastalUrun.GerceklesenMiktar), Sum(HGIsEmriPastalUrun.MetoMiktar), Sum(HGIsEmriPastalUrun.TasnifMiktar) From HGIsEmriPastalUrun
    Where HGIsEmriPastalUrun.HGIsEmriPastalID = :HGIsEmriPastalID
    Into :PlanlananMiktar, :GerceklesenMiktar, :MetoMiktar, :TasnifMiktar;

    if (ATasnifMeto in (0, 1, 2)) then
    begin
      if (ATasnifMeto = 0) then
        FDurum = Iif(GerceklesenMiktar > MetoMiktar, 1, 0);
      else if (ATasnifMeto = 1) then
        FDurum = Iif(GerceklesenMiktar > TasnifMiktar, 1, 0);
      else
        FDurum = Iif((GerceklesenMiktar > TasnifMiktar) and (GerceklesenMiktar > MetoMiktar), 1, 0);
    end

    if (FDurum = 1) then suspend;
  end
end
