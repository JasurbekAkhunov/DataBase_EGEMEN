        FHarmanCinsiID <> FLastHarmanCinsiID or FHarmanAdi <> FLastHarmanAdi  or FSarimSekliId <> FLastSarimSekliId or FAmbalajId <> FLastAmbalajId) then
    begin
      if (FLastHamId <> -1) then
      begin
        if (IplikIsletmeId < 1) then IplikIsletmeId = Null;
        if (HarmanCinsiId < 1) then HarmanCinsiId = Null;
        if (HarmanIplikIsletmeId < 1) then HarmanIplikIsletmeId= Null;
        if (AmbalajSekliId < 1) then AmbalajSekliId = Null;
        if (HarmanAdi = '') then HarmanAdi = Null;
        if (AmbalajId < 1) then AmbalajId = Null;
        if (AmbalajAdi = '') then AmbalajAdi = Null;
        suspend;
      end
      SevkiyatKapAdedi      = 0;
      SevkiyatMiktar        = 0;
      SevkiyatNetMiktar     = 0;
      SevkiyatIrsaliyeNo    = '';
      SevkiyatIrsaliyeTarihi = '';
      SevkiyatAracId = '';
      AmbalajAdi = '';

      FLastIplikDepoId = FIplikDepoId;
      FLastHamId = FHamId;
      FLastLot = FLot;
      FLastAnaLotBolum = FAnaLotBolum;
      FLastMelanjKodu = FMelanjKodu;
      FLastFlamKodu = FFlamKodu;
      FLastTamir = FTamir;
      FLastAmbalajSekliId = FAmbalajSekliId;
      FLastSiparisFirmaId = FSiparisFirmaId;
      FLastHarmanCinsiId = FHarmanCinsiId;
      FLastHarmanIplikIsletmeId = FHarmanIplikIsletmeId;
      FLastHarmanAdi  = FHarmanAdi;
      FLastIplikIsletmeID = FIplikIsletmeId;
      FLastSarimSekliId = FSarimSekliId;
      FLastAmbalajId = FAmbalajId;

      IplikDepoId = FIplikDepoId;
      HamID = FHamId;
      Lot = FLot;
      AnaLotBolum = FAnaLotBolum;
      MelanjKodu = FMelanjKodu;
      FlamKodu = FFlamKodu;
      Tamir = FTamir;
      AmbalajSekliID = FAmbalajSekliID;
      SiparisFirmaId = FSiparisFirmaId;
      HarmanCinsiID = FHarmanCinsiID;
      HarmanIplikIsletmeId = FHarmanIplikIsletmeId;
      HarmanCinsiAdi = FHarmanCinsiAdi;
      HarmanAdi = FHarmanAdi;
      IplikIsletmeID = FIplikIsletmeId;
      HarmanIplikIsletmeAdi = FHarmanIplikIsletmeAdi;
      SarimSekliId = FSarimSekliId;
      AmbalajId = FAmbalajId;
    end

    SevkiyatKapAdedi      = SevkiyatKapAdedi + FKapAdedi;
    SevkiyatMiktar        = SevkiyatMiktar + FMiktar;
    SevkiyatNetMiktar     = SevkiyatNetMiktar + FNetMiktar;

    if ((Char_Length(SevkiyatIrsaliyeNo) + Char_Length(FIrsaliyeNo)) < 90) then --> Char_Length(FIrsaliyeNo) bu saysal bir parametreyi kendi iinde stringe dntrebiliyor. UP (FB 2.0 dan beri... Fahri ile birlikte test edildi. )
    begin
     if (SevkiyatIrsaliyeNo <> '' ) then SevkiyatIrsaliyeNo = SevkiyatIrsaliyeNo || cast(', ' as varchar(2) character set UTF8);
     SevkiyatIrsaliyeNo = SevkiyatIrsaliyeNo || format_float('',FIrsaliyeNo);
    end

    if (Char_Length(SevkiyatIrsaliyeTarihi) < 70) then
    begin
      if (SevkiyatIrsaliyeTarihi <> '') then SevkiyatIrsaliyeTarihi = SevkiyatIrsaliyeTarihi || cast(', ' as varchar(2) character set UTF8);
      SevkiyatIrsaliyeTarihi = SevkiyatIrsaliyeTarihi || TimeStampToDateStr(FIrsaliyeTarihi);
    end

    if (Char_Length(SevkiyatAracId) < 85) then
    begin
      if (SevkiyatAracId <> '') then SevkiyatAracId = SevkiyatAracId || cast(', ' as varchar(2) character set UTF8);
      SevkiyatAracId = SevkiyatAracId || FAracId;
    end

    if (Char_Length(AmbalajAdi) < 200) then
    begin
      if (AmbalajAdi <> '') then AmbalajAdi = AmbalajAdi || cast(', ' as varchar(2) character set UTF8);
      AmbalajAdi = AmbalajAdi || FAmbalaj;
    end
  end

  if (FLastHamId <> -1) then
  begin
    if (IplikIsletmeId < 1) then IplikIsletmeId = Null;
    if (HarmanCinsiId < 1) then HarmanCinsiId = Null;
    if (HarmanIplikIsletmeId < 1) then  HarmanIplikIsletmeId = Null;
    if (AmbalajSekliId < 1) then AmbalajSekliId = Null;
    if (HarmanAdi = '') then HarmanAdi = Null;
    if (AmbalajId < 1) then AmbalajId = Null;
    if (AmbalajAdi = '') then AmbalajAdi = Null;
    suspend;
  end

  SevkiyatKapAdedi      = 0;
  SevkiyatMiktar        = 0;
  SevkiyatNetMiktar     = 0;
  SevkiyatIrsaliyeNo    = '';
  SevkiyatIrsaliyeTarihi = '';
  SevkiyatAracId = '';
  AmbalajId = Null;
  AmbalajAdi = '';
  BukumIsEmriMi = 0;

  For Select IplikIrsaliyeDetay.IplikIsletmeId, Harman.IplikIsletmeId, IplikIsletme.Adi, IplikIrsaliyeDetay.IplikDepoId, IplikIrsaliyeDetay.HamId, Harman.Adi, Harman.HarmanCinsiId,
             HarmanCinsi.Adi, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu, IplikLot.Tamir,
             Ambalaj.AmbalajSekliId, Boyasiparis.FirmaId, IplikIrsaliyeDetay.SarimSekliId,
             IplikIrsaliyeDetay.AmbalajId, Ambalaj.Adi,
             Sum(IplikIrsaliyeDetay.KapAdedi), Sum(IplikIrsaliyeDetay.Miktar), Sum(IplikIrsaliyeDetay.NetMiktar), iif(IplikLot.BukumIsemriKayitId > 0, 1, 0), Renk.Adi from IplikIrsaliye
  left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeId = IplikIrsaliye.IplikIrsaliyeId
  left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
  left join Ambalaj on Ambalaj.AmbalajId = IplikIrsaliyeDetay.AmbalajId
  left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = Ambalaj.AmbalajSekliId
  left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikIrsaliyeDetay.IplikTalepDetayId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
  left join Firma  SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
  left join Harman on Harman.HarmanKayitId = IplikIrsaliyeDetay.HarmanKayitId
  left join Renk on Renk.RenkId = Harman.RenkId
  left join IplikIsletme on IplikIsletme.IplikIsletmeId = Harman.IplikIsletmeId
  left join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Harman.HarmanCinsiId
  left join IplikLot on IplikLot.Lot = IplikIrsaliyeDetay.Lot
  where IplikIrsaliye.FiiliSevkTarihi >= :ABaslangicTarihi and IplikIrsaliye.FiiliSevkTarihi < :ABitisTarihi and IplikIrsaliye.HareketTipi = :shtSatis and IplikIrsaliyeDetay.IplikIrsaliyeDetayID is not null and
        Ham.HamTipiID = :hstIplik and IplikIrsaliye.FirmaID = :FirmaKodumuz
  group by IplikIrsaliyeDetay.IplikIsletmeId, Harman.IplikIsletmeId, IplikIsletme.Adi, IplikIrsaliyeDetay.IplikDepoId, IplikIrsaliyeDetay.HamId, Harman.Adi, Harman.HarmanCinsiId,
           HarmanCinsi.Adi, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu, IplikLot.Tamir,
           Ambalaj.AmbalajSekliId, Boyasiparis.FirmaId, IplikIrsaliyeDetay.SarimSekliId,
           IplikIrsaliyeDetay.AmbalajId, Ambalaj.Adi, IplikLot.BukumIsemriKayitId, Renk.Adi
  into :IplikIsletmeId, :HarmanIplikIsletmeId, :HarmanIplikIsletmeAdi, :IplikDepoId, :HamId, :HarmanAdi, :HarmanCinsiId, :HarmanCinsiAdi, :Lot, :MelanjKodu, :FlamKodu, :Tamir, :AmbalajSekliId, :SiparisFirmaId, :SarimSekliId,
       :AmbalajId, :AmbalajAdi,
       :BySevkiyatKapAdedi, :BYSevkiyatMiktar, :BySevkiyatNetMiktar, :BukumIsEmriMi, :HarmanRenkAdi do
  begin

    AnaLotBolum = Null ;
    if (Char_Length(Lot) > 0 and Position('.' in Lot) > 0)  then
      AnaLotBolum = substring(Lot from 1 for (Position('.' in Lot)-1) );

    suspend;
  end
