CREATE PROCEDURE HGSIPARISRAPORU_5(
    ABASLANGICSIPARISYIL INTEGER,
    ABASLANGICSIPARISID INTEGER,
    ABITISSIPARISYIL INTEGER,
    ABITISSIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASINIF VARCHAR(1),
    AMODELID MODELKODUSIZE,
    ASIPONAYDURUMU SMALLINT,
    AMALIONAYDURUMU SMALLINT)
RETURNS (
    HGSIPARISKAYITID INTEGER,
    HGSIPARISDETAYID INTEGER,
    HGISEMRIKAYITID INTEGER,
    MODELKUMASID INTEGER,
    MODELID MODELKODUSIZE,
    URUNID URUNKODUSIZE,
    MODELBEDENID INTEGER,
    RENKADI ADI40,
    KISAKODU TYPE OF COLUMN BEDEN.KISAKODU,
    RENKID INTEGER,
    BEDENID INTEGER,
    BEDENADI TYPE OF COLUMN BEDEN.ADI,
    KESIMTALIMATI INTEGER,
    MIKTAR BOYAMIKTARSIZE,
    PLANLANANMIKTAR INTEGER,
    GERCEKLESENMIKTAR INTEGER,
    GERCEKLESENKESIM INTEGER,
    METOMIKTAR INTEGER,
    TASNIFMIKTAR INTEGER,
    KALANMIKTAR INTEGER)
AS
Declare Variable ModelKayitId Integer;
begin
  Select ModelKayitId From Model Where Model.ModelID = :AModelID into :ModelKayitId;

  For Select HGSiparis.HgSiparisKayitId, HGSiparisDetay.HgSiparisDetayId, Model.ModelID, Urun.UrunID, Urun.ModelKumasID, Urun.ModelBedenID, ModelKumas.RenkADI, Beden.KisaKodu,
             ModelKumas.RenkID, Beden.BedenID, Beden.KisaKodu, HGSiparisDetayUrun.Miktar, HGIsEmri.HGIsEmriKayitId, HGIsEmri.KesimTalimati from HgSiparis
      Left Join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
      Left Join Firma On Firma.FirmaId = HgSiparis.FirmaId
      Left Join Model on Model.ModelKayitID = HgSiparisDetay.ModelKayitID
      Left Join Urun ON Urun.ModelKayitId = Model.ModelKayitId and Urun.ModelParcaId is Null and Urun.ModelBedenID IS NOT NULL
      Left Join ModelKumas ON ModelKumas.ModelKumasID = Urun.ModelKumasID
      Left Join ModelBeden ON ModelBeden.ModelBedenID = Urun.ModelBedenID
      Left Join Beden ON Beden.BedenID = ModelBeden.BedenID
      Left Join HGSiparisDetayUrun ON HGSiparisDetayUrun.ModelBedenid = ModelBeden.ModelBedenID and HGSiparisDetayUrun.ModelKumasid = ModelKumas.ModelKumasid and HGSiparisDetayUrun.HGSiparisDetayId = HGSiparisDetay.HGSiparisDetayId and HGSiparisDetayUrun.HGSiparisDetayAsortiId IS NULL
      Left Join HGIsEmri ON HGIsEmri.HgSiparisDetayId = HGSiparisDetay.HgSiparisDetayId
      Where (:ARaporTipi = 1 and HgSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and HgSiparis.SiparisTarihi <= :ABitisSiparisTarihi or
             :ARaporTipi = 2 and HgSiparisDetay.TeslimTarihi >= :ABaslangicSiparisTarihi and HgSiparisDetay.TeslimTarihi <= :ABitisSiparisTarihi or
             :ARaporTipi = 0 and
                (HgSiparis.HgSiparisYil > :ABaslangicSiparisYil or (HgSiparis.HgSiparisYil = :ABaslangicSiparisYil and HgSiparis.HgSiparisId >= :ABaslangicSiparisId)) and
                (HgSiparis.HgSiparisYil < :ABitisSiparisYil or (HgSiparis.HgSiparisYil = :ABitisSiparisYil  and HgSiparis.HgSiparisId <= :ABitisSiparisId))
             ) and
            (Firma.FirmaId = :AFirmaId or :AFirmaId is null) and
            (HgSiparisDetay.Kapali = :AKapali or :AKapali is null) and
            (HgSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
            (HgSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
            (HgSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
            (HgSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null) and
            (HgSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
            (HgSiparisDetay.Sinif = :ASinif or :ASinif is null) and
            (HgSiparisDetay.ModelKayitId = :ModelKayitId or :ModelKayitId is Null) and
            (HgSiparisDetay.OnayDurumu = :ASipOnayDurumu or :ASipOnayDurumu is Null) and
            (HgSiparisDetay.MaliOnayDurumu = :AMaliOnayDurumu or :AMaliOnayDurumu is Null) and
            (HgSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null) and
            HGIsEmri.HGIsEmriKayitId is Not Null and
            ((Select Sum(Miktar) From HGSiparisDetayUrun Where HGSiparisDetayUrun.ModelBedenID = ModelBeden.ModelBedenID and HGSiparisDetayUrun.HGSiparisDetayId = HGSiparisDetay.HGSiparisDetayId AND HGSiparisDetayAsortiId IS NULL) > 0
              and
            (Select Sum(Miktar) From HGSiparisDetayUrun Where HGSiparisDetayUrun.ModelKumasid = ModelKumas.ModelKumasid and HGSiparisDetayUrun.HGSiparisDetayId = HGSiparisDetay.HGSiparisDetayId and HGSiparisDetayAsortiId is null) > 0)
      Order By HGSiparisDetay.HgSiparisDetayId, HGIsEmri.HGIsEmriKayitId, MODELKUMAS.MODELKUMASID, MODELBEDEN.MODELBEDENID
  Into :HgSiparisKayitId, :HgSiparisDetayId, :ModelID, :UrunID, :ModelKumasID, :ModelBedenID, :RenkAdi, :KisaKodu,  :RenkID,
       :BedenID, :BedenAdi, :Miktar, :HGIsEmriKayitId, :KesimTalimati do
  begin
    PlanlananMiktar        = Null;
    GerceklesenMiktar      = Null;
    GerceklesenKesim       = Null;
    KalanMiktar            = Null;
    MetoMiktar             = Null;
    TasnifMiktar           = Null;

    if (KesimTalimati = 0) then
    begin
      Select Sum(HgIsEmriUrun.PlanlananMiktar), Sum(HgIsEmriUrun.GerceklesenMiktar), Sum(HgIsEmriUrun.PlanlananMiktar - HgIsEmriUrun.GerceklesenMiktar) from HgIsEmri
        Left Join HgIsEmriUrun on HgIsEmriUrun.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
        where HgIsEmri.HGIsEmriKayitId = :HGIsEmriKayitId and HgIsEmriUrun.UrunId = :UrunId
      Into :PlanlananMiktar, :GerceklesenMiktar, :KalanMiktar;
    end
    else
    begin
      Select Sum(HGIsemriPastalUrun.PlanlananMiktar), sum(HGIsemriPastalUrun.GerceklesenMiktar),
           Sum(HGIsemriPastalUrun.MetoMiktar), Sum(HGIsemriPastalUrun.TasnifMiktar) from HgIsEmri
        Left Join HGISEmriPastal on HGISEmriPastal.HGIsEmriKayitId = HgIsEmri.HGIsEmriKayitId
        Left Join HGIsemriPastalUrun on HGIsemriPastalUrun.HGIsEmriPastalID = HGISEmriPastal.HGIsEmriPastalID
        Where HgIsEmri.HGIsEmriKayitId = :HGIsEmriKayitId and HgIsEmri.KesimTalimati = 1 and HGIsemriPastalUrun.UrunId = :UrunId
      into :PlanlananMiktar, :GerceklesenKesim, :MetoMiktar, :TasnifMiktar;
    end
    suspend;
  end
end
