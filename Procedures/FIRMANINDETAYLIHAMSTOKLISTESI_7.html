CREATE PROCEDURE FIRMANINDETAYLIHAMSTOKLISTESI_7(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    AORMEGIRISIRSALIYENO IRSALIYENOSIZE,
    AEVRAKNO VARCHAR(10),
    AHAMSTOKMASTERID INTEGER,
    ATAMIR INTEGER,
    AFASON INTEGER,
    ARAPORHAMTIPI INTEGER,
    ATAMIRTIPIID INTEGER,
    ASIFIROLANLARIDOK INTEGER,
    AHAMID INTEGER,
    ALOT VARCHAR(150),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AREFERANSFIRMAID VARCHAR(20),
    AORMECIID INTEGER,
    ASINIF VARCHAR(1),
    AHAMSTOKHAREKETTIPIID INTEGER,
    AIRSALIYEGRUPID VARCHAR(4),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ASIPARISFIRMAID VARCHAR(20),
    AIPLIKTALEPYIL INTEGER,
    AIPLIKTALEPID INTEGER,
    AURETIMPARTINO VARCHAR(30),
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AMUSTERIPARTINO VARCHAR(30),
    AVERILENSIPARISKAYITID INTEGER,
    AREFERANSKODU VARCHAR(30),
    ATARIHFILTRETIPI INTEGER,
    AKALITEID VARCHAR(20))
RETURNS (
    FISYIL INTEGER,
    FISNO INTEGER,
    FISNOSTR VARCHAR(30),
    FISTARIHI TIMESTAMP,
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    SARFTIPIID INTEGER,
    SARFTIPIADI VARCHAR(30),
    HAMADI VARCHAR(255),
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    HAMTIPIID INTEGER,
    REFERANSKODU VARCHAR(30),
    RAFID VARCHAR(10),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    IRSALIYENO VARCHAR(255),
    IRSALIYEGRUPID VARCHAR(4),
    LOT VARCHAR(150),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable PartiSinifId integer;
declare variable HamStokFason integer;
declare variable PartiFason integer;
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;

  For Select HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo, HamStokSarfFisi.FisTarihi, HamStokSarfFisi.SarfTipiId, SarfTipi.Adi, Ham.Adi,
             Ham.ReferansKodu, HamStokSarfFisiDetay.Kilo, HamStokSarfFisiDetay.Metre,
             HamStokSarfFisiDetay.PartiSinifId, Parti.PartiYil, Parti.PartiId, HamStokIrsaliye.IrsaliyeGrupId, HamStokIrsaliye.IrsaliyeNo,
             HamStokIrsaliye.Fason, Parti.Fason, Firma.FirmaId, Firma.Adi, RafDetay.RafId, HamStokMaster.Lot, HamStokMaster.HamStokMasterID, HamStokSarfFisi.Notu,
             HamStokDetay.HamStokDetayID from
  FirmaninDetayliHamStokListesi_1(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :AIrsaliyeNo, :AOrmeGirisIrsaliyeNo, :AEvrakNo, :AHamStokMasterId, :ATamir, :AFason, :ARaporHamTipi,
                                     :ATamirTipiId, :ASifirOlanlariDok, :AHamId, :ALot, :AMusteriSiparisNo, :AReferansSiparisNo, :AReferansFirmaId, :AOrmeciId, :ASinif, :AHamStokHareketTipiId, :AIrsaliyeGrupId,
                                     :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id, :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id,
                                     :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ABoyaSiparisYil, :ABoyaSiparisId, :ASiparisFirmaId, :AIplikTalepYil, :AIplikTalepId, :AUretimPartiNo, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AMusteriPartiNo, :AVerilenSiparisKayitId, :AReferansKodu, null, null, :ATarihFiltreTipi, :AKaliteId) HSD
  left join HamStokDetay on HamStokDetay.HamStokDetayId = HSD.HamStokDetayId
  left join RafDetay on RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId
  left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
  left join HamStokSarfFisiDetay on HamStokSarfFisiDetay.RafDetayId = RafDetay.RafDetayId
  left join HamStokSarfFisi on HamStokSarfFisi.HamStokSarfFisiId = HamStokSarfFisiDetay.HamStokSarfFisiId
  left join SarfTipi on SarfTipi.SarfTipiId = HamStokSarfFisi.SarfTipiId
  left join Ham on Ham.HamId = RafDetay.HamId
  left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamaStokDetayId = HamStokSarfFisiDetay.PartiAsamaStokDetayId
  left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
  left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = RafDetay.HamStokIrsaliyeId
  left join Firma on Firma.FirmaId = Parti.FirmaId or Firma.FirmaId = HamStokIrsaliye.FirmaId
  where HamStokSarfFisiDetay.PartiAsamaStokDetayId is not null
  order by HamStokSarfFisi.FisTarihi, HamStokSarfFisi.FisYil, HamStokSarfFisi.FisNo
  into :FisYil, :FisNo, :FisTarihi, :SarfTipiId, :SarfTipiAdi, :HamAdi, :ReferansKodu, :Kilo, :Metre,
       :PartiSinifId, :PartiYil, :PartiId, :IrsaliyeGrupId, :IrsaliyeNo, :HamStokFason, :PartiFason,
       :FirmaId, :FirmaAdi, :RafId, :Lot, :HamStokMasterID, :Notu, :HamStokDetayID do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') '  || EgemenIntegerToStr(PartiId);
    Fason = PartiFason;

    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    FisNoStr = '(' || EgemenIntegerToStr(FisYil) || ') ' || EgemenIntegerToStr(FisNo) || ' ';

    suspend;
  end
end
