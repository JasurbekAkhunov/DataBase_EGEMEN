CREATE PROCEDURE HAKTARMAISEMRIDETAYKAYDET(
    AHAKTARMAISEMRIKAYITID INTEGER,
    AHAKTARMAFISIDETAYID INTEGER,
    AKILO NUMERIC(12,3),
    AKULLANICIID INTEGER)
AS
declare variable FHAktarmaIsEmriDetayId integer;
begin
  Select HAktarmaIsEmriDetayId from HAktarmaIsEmriDetay
  where HAktarmaIsEmriDetay.HAktarmaIsEmrikayitId = :AHAktarmaIsEmrikayitId and HAktarmaIsEmriDetay.HAktarmaFisiDetayId = :AHAktarmaFisiDetayId
  into :FHAktarmaIsEmriDetayId;

  if (FHAktarmaIsEmriDetayId is null) then insert into HAktarmaIsEmriDetay(HAktarmaIsEmrikayitId, HAktarmaFisiDetayId, Kilo, InsertKullaniciid, Kullaniciid) values(:AHAktarmaIsEmrikayitId, :AHAktarmaFisiDetayId, :AKilo, :AKullaniciid, :AKullaniciId);
  else Update HAktarmaIsEmriDetay set Kilo = Kilo + :AKilo, KullaniciId = :AKullaniciId where HAktarmaIsEmriDetay.HAktarmaIsEmriDetayID = :FHAktarmaIsEmriDetayId;
end
