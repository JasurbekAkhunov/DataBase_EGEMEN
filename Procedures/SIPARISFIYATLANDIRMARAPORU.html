    BOYASIPARISDETAYID INTEGER,

    FASON INTEGER,
    FASONSTR VARCHAR(5),
    ODEMESEKLIADI VARCHAR(40),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),

    LABRECETEATISNO VARCHAR(20),

    SIPARISRENKDERINLIGIADI VARCHAR(20),
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENKILOBASKI NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    PARTILENENMETREBASKI NUMERIC(12,3),
    IPTALKILO NUMERIC(12,3),
    IPTALKILOBASKI NUMERIC(12,3),
    IPTALMETRE NUMERIC(12,3),
    IPTALMETREBASKI NUMERIC(12,3),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    SIPARISRENKADI VARCHAR(40),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    FASONBIRIMFIYATI DOUBLE PRECISION,

    LISTEFIYATBIRIMFIYAT DOUBLE PRECISION,

    MUSTERIKALITEADI VARCHAR(50),

    NETSEVKEDILENMETRE NUMERIC(12,3),

    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKADI VARCHAR(40),
    FIYATKOMISYON INTEGER,
    FIYATKOMISYONSTR VARCHAR(20),
    FIYATKOMISYONORANI DOUBLE PRECISION,
    DOKUMASIPARISTIPIID INTEGER,
    DOKUMASIPARISI INTEGER,
    SEVKFIREORANI DOUBLE PRECISION,
    SEVKFIREKILO DOUBLE PRECISION,
    KAPALI SMALLINT,
    KAPALISTR VARCHAR(6))
AS
declare variable ABoyaSiparisKayitId integer;
declare variable astBoyaliIplik      integer;
declare variable astBoyaliKumas      integer;
declare variable astIplikBoyaliKumas integer;
declare variable ElyafCinsiID        varchar(5);
declare variable HamTipiId           integer;
declare variable PartilenenMiktardanIptaliDus Boolean;
begin
  select astBoyaliIplik, AstBoyaliKumas, AstIplikBoyaliKumas from OrmeSiparisTipiGetir
    into :astBoyaliIplik, :AstBoyaliKumas, :AstIplikBoyaliKumas;

  select PartilenenMiktardanIptaliDus from Sistem where Sistem.SistemId = 1 into :PartilenenMiktardanIptaliDus;

  if (ABoyaSiparisYil is not null) then
  begin
    select BoyaSiparis.BoyaSiparisKayitId from BoyaSiparis
      where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId
      into :ABoyaSiparisKayitId;
    if (ABoyaSiparisKayitId is null) then ABoyaSiparisKayitId = -1;
  end

  if (ABoyaSiparisKayitId is null) then
  begin
    for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.SiparisTipiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId,
               Firma.Adi, BoyaSiparis.SiparisGrupId, SG1.Adi, BoyaSiparis.SiparisGrup2Id, SG2.Adi,
               BoyaSiparis.SiparisGrup3Id, SG3.Adi, BoyaSiparis.SiparisGrup4Id, SG4.Adi, BoyaSiparis.Fason, OdemeSekli.Adi, BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi,
               BoyaSiparisDetay.IslemGrubuId, IslemGrubu.Adi,
               BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.OrmeSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.ReferansSiparisNo, BoyaSiparisDetay.LabReceteId,
               LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.LabReceteAtisId, LabReceteAtis.AtisNoStr, BoyaSiparisDetay.SiparisKilo,
               BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.PartilenenKilo, BoyaSiparisDetay.IptalKilo, BoyaSiparisDetay.PartilenenMetre, BoyaSiparisDetay.IptalMetre,
               BoyaSiparisDetay.HamId, Ham.Adi, Ham.HamTipiId, IplikCinsi.ElyafCinsiID, BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi,
               BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, BoyaSiparisDetay.VadeGun,
               BoyaSiparisDetay.FasonBirimFiyati, BoyaSiparisDetay.FasonBirimFiyatDovizId,
               BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
               BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.RenkDerinligiId, RenkDerinligi.Adi, OK.Adi, BoyaSiparis.OrmeSiparisi,
               BoyaSiparis.OrmeSiparisSekliId, BoyaSiparis.OrmeSiparisTipiId, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.TerminTarihi,
               BoyaSiparisDetay.OnayTarihi, BoyaSiparisDetay.MaximumFireOrani, BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi, BoyaSiparisDetay.SiparisBrutKilo,
               BoyaSiparisDetay.FiyatKomisyon, BoyaSiparisDetay.FiyatKomisyonOrani, BoyaSiparis.DokumaSiparisTipiId, BoyaSiparis.DokumaSiparisi,
               BoyaSiparisDetay.PartilenenKiloBaski, BoyaSiparisDetay.IptalKiloBaski, BoyaSiparisDetay.PartilenenMetreBaski, BoyaSiparisDetay.IptalMetreBaski, BoyaSiparis.Kapali from BoyaSiparis
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisKayitId = BoyaSiparis.BoyaSiparisKayitId
    left join OdemeSekli on OdemeSekli.OdemeSekliId = BoyaSiparisDetay.OdemeSekliId
    left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = BoyaSiparisDetay.IslemGrubuId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
    left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
    left join Kullanici OK on OK.KullaniciId = BoyaSiparisDetay.OnaylayanKullaniciId
    left join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    left join SiparisGrup SG1 on SG1.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SG2 on SG2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join SiparisGrup SG3 on SG3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
    left join SiparisGrup SG4 on SG4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
    where  (BoyaSiparis.SiparisTarihi >= :ABaslangicTarihi and BoyaSiparis.SiparisTarihi <= :ABitisTarihi) and
             (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
             (BoyaSiparis.Fason = :ASatisTipi or :ASatisTipi is null) and
             (BoyaSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
             (BoyaSiparisDetay.IslemTipiId = :AIslemTipiId or :AIslemTipiId is null) and
             (BoyaSiparisDetay.SiparisTipiId = :ABoyaSiparisTipiId or :ABoyaSiparisTipiId is null) and
             (BoyaSiparisDetay.OnayDurumu = :AOnayDurumu or :AOnayDurumu is null) and
             (BoyaSiparisDetay.BirimFiyat = 0 or :ASadeceFiyatiSifirOlanlar = 0) and
             (BoyaSiparis.IslemTipiID <> 3 or BoyaSiparisDetay.UcretliTamir = 0) and
              BoyaSiparisDetay.BoyaSiparisDetayID is not null and
             (BoyaSiparisDetay.FiyatKomisyon = :AFiyatKomisyon or :AFiyatKomisyon is null) and
             (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
             (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
             (BoyaSiparis.SiparisGrup3Id = :ASiparisGrup3Id or :ASiparisGrup3Id is null) and
             (BoyaSiparis.SiparisGrup4Id = :ASiparisGrup4Id or :ASiparisGrup4Id is null)
    order by BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId
    into :BoyaSiparisKayitId, :SiparisTipiId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
         :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi, :Fason, :OdemeSekliAdi, :IslemTipiId, :IslemTipiAdi, :IslemGrubuId, :IslemGrubuAdi,
         :BoyaSiparisDetayId, :OrmeSiparisDetayId,  :MusteriSiparisNo, :ReferansSiparisNo, :LabReceteId, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisId, :LabReceteAtisNo,
         :SiparisKilo, :SiparisMetre, :PartilenenKilo, :IptalKilo, :PartilenenMetre, :IptalMetre,
         :HamId, :HamAdi, :HamTipiId, :ElyafCinsiID, :SiparisProsesId, :SiparisProsesAdi, :BirimFiyat, :DovizId, :VadeGun, :FasonBirimFiyati, :FasonBirimFiyatDovizId,
         :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
         :IlaveIslemler, :SiparisRenkDerinligiId, :SiparisRenkDerinligiAdi, :OnaylayanKullaniciAdi, :OrmeSiparisi, :OrmeSiparisSekliId, :OrmeSiparisTipiId,
         :SiparisRenkAdi, :TerminTarihi, :OnayTarihi, :MaximumFireOrani, :SiparisiAlanKullaniciId, :SiparisiAlanKAdi, :SiparisBrutKilo,
         :FiyatKomisyon, :FiyatKomisyonOrani, :DokumaSiparisTipiId, :DokumaSiparisi, :PartilenenKiloBaski, :IptalMetreBaski, :PartilenenKiloBaski, :IptalMetreBaski, :Kapali do
    begin
      ListeFiyatBirimFiyat = null;
      ListeFiyatDovizId = null;
      FasonStr = '';
      KapaliStr = '';

      if (Kapali = 1) then
        KapaliStr = 'KapalÄ±';

      if (Fason = 1) then
      begin
        FasonStr = 'Fason';

        if (HamTipiID = 0) then
        begin
          select BirimFiyat, DovizId from MusteriFiyatListesi
            where MusteriFiyatListesi.FirmaID = :FirmaID and MusteriFiyatListesi.Fason = :Fason and
                  MusteriFiyatListesi.RenkDerinligiId = :SiparisRenkDerinligiId and
                  MusteriFiyatListesi.IslemGrubuId = :IslemGrubuId and
                  IntKarsilastir(MusteriFiyatListesi.ElyafCinsiId, :ElyafCinsiId) = 1 and MusteriFiyatListesi.HamID is null
            into :ListeFiyatBirimFiyat, :ListeFiyatDovizId;
        end
      end
      else
      begin

        select BirimFiyat, DovizId from MusteriFiyatListesi
          where MusteriFiyatListesi.FirmaID = :FirmaID and MusteriFiyatListesi.HamID = :HamId and MusteriFiyatListesi.Fason = :Fason and
                MusteriFiyatListesi.RenkDerinligiId = :SiparisRenkDerinligiId and
                MusteriFiyatListesi.IslemGrubuId = :IslemGrubuId
          into :ListeFiyatBirimFiyat, :ListeFiyatDovizId;

      end

      BoyaSiparisNoStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      BrutSevkEdilenKilo = null;
      BrutSevkEdilenMetre = null;
      NetSevkEdilenKilo = null;
      NetSevkEdilenMetre = null;

      select sum(PartiStok.BrutSevkKilo), sum(PartiStok.BrutSevkMetre), sum(PartiStok.NetSevkKilo), sum(PartiStok.NetSevkMetre) from BoyaSiparisDetay
        left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
        where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
          into :BrutSevkEdilenKilo, :BrutSevkEdilenMetre, :NetSevkEdilenKilo, :NetSevkEdilenMetre;

      SevkFireKilo = Coalesce(BrutSevkEdilenKilo, 0) - Coalesce(NetSevkEdilenKilo, 0);
      if (BrutSevkEdilenKilo > 0) then
      begin
        SevkFireOrani = (SevkFireKilo / BrutSevkEdilenKilo) * 100;
        SevkFireOrani = Yuvarla(SevkFireOrani, 100);
      end

      if (FiyatKomisyon = 1) then FiyatKomisyonStr = 'Komisyon var';
      else FiyatKomisyonStr = 'Komisyon yok';

      if (PartilenenMiktardanIptaliDus = 1) then
      begin
        PartilenenKilo = PartilenenKilo - IptalKilo;
        PartilenenMetre = PartilenenMetre - IptalMetre;
        PartilenenKiloBaski = PartilenenKiloBaski - coalesce(IptalKiloBaski,0);
        PartilenenMetreBaski = PartilenenMetreBaski - IptalMetreBaski;
      end

      suspend;
    end
  end
  else
  begin
    for select BoyaSiparis.BoyaSiparisKayitId, BoyaSiparisDetay.SiparisTipiId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId,
               Firma.Adi, BoyaSiparis.SiparisGrupId, SG1.Adi, BoyaSiparis.SiparisGrup2Id, SG2.Adi,
               BoyaSiparis.SiparisGrup3Id, SG3.Adi, BoyaSiparis.SiparisGrup4Id, SG4.Adi, BoyaSiparis.Fason, OdemeSekli.Adi, BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi, BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi,
               BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.OrmeSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.ReferansSiparisNo, BoyaSiparisDetay.LabReceteId,
