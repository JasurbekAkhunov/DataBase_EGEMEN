CREATE PROCEDURE DOKUMAPLANLANACAKSIPARISLISTESI(
    ADOKUMAMAKINABOLUMID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISSIRANO INTEGER,
    MAKINAGRUPID VARCHAR(4),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    SIPARISMIKTARI DOUBLE PRECISION,
    PLANLANANMIKTAR DOUBLE PRECISION,
    STOKTANKULLANILACAKMIKTAR DOUBLE PRECISION,
    SATINALINACAKMIKTAR DOUBLE PRECISION,
    PLANLANACAKMIKTAR DOUBLE PRECISION,
    TOPLAMDOKUNAN DOUBLE PRECISION,
    TOPLAMSAKATDOKUNAN DOUBLE PRECISION,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    PLANLANDIGIMAKINALAR VARCHAR(255),
    PLANLANDIGIFASONCULAR VARCHAR(255),
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    KUMASTEMIN INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    TESLIMTARIHI TIMESTAMP,
    MINBOYALITERMINTARIHI TIMESTAMP,
    MAXBOYALITERMINTARIHI TIMESTAMP,
    DOKUNACAKMIKTAR BOYAMIKTARSIZE,
    KALITEADI ADI255)
AS
declare variable dstIplik               integer;
declare variable dstDokuma              integer;
declare variable dstBoyaliDokuma        integer;
declare variable dstIplikBoyaliDokuma   integer;
declare variable dstBoyaliIplik         integer;
declare variable PlanliGerceklesen      Double Precision;
declare variable Uretim                 Double Precision;
declare variable FDokumaMakinaId        varchar(10);
declare variable FFirmaKisaAdi          varchar(10);
declare variable FBitti                 integer;
declare variable RevizeTeslimTarihi     TimeStamp;
declare variable OSDPlanlamaIzineBagli  integer;
begin
  Select dstDokuma, dstBoyaliDokuma, dstIplik, dstBoyaliIplik, dstIplikBoyaliDokuma from DokumaSiparisTipiGetir
    into :dstDokuma, :dstBoyaliDokuma, :dstIplik, :dstBoyaliIplik, :dstIplikBoyaliDokuma;

  Select OSDPlanlamaIzineBagli from Sistem where SistemID = 1 into :OSDPlanlamaIzineBagli;

  For Select BoyaSiparis.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.BoyaSiparisYil,
             BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo,  OrmeSiparisDetay.MakinaGrupId,
             OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.RenkId, Renk.Adi, OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.ToplamDokunan, OrmeSiparisDetay.ToplamSakatDokunan,
             OrmeSiparisDetay.StoktanKullanilacakMiktar, OrmeSiparisDetay.SatinAlinacakMiktar, OrmeSiparisDetay.PlanlananMiktar,
             BoyaSiparis.FirmaId, Firma.Adi, BoyaSiparis.KumasTemin,
             BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi, BoyaSiparis.SiparisTarihi, OrmeSiparisDetay.TeslimTarihi,
             OrmeSiparisDetay.RevizeTeslimTarihi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, Kalite.KaliteAdi from OrmeSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
    left join Kalite on Kalite.KaliteId = OrmeSiparisDetay.KaliteId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    where OrmeSiparisDetay.Kapali = 0 and OrmeSiparisDetay.KTBoyaSiparisKayitId is null and BoyaSiparis.DokumaSiparisTipiId >= :dstDokuma and OrmeSiparisDetay.HamMiktar > (OrmeSiparisDetay.PlanlananMiktar + OrmeSiparisDetay.StoktanKullanilacakMiktar + OrmeSiparisDetay.SatinAlinacakMiktar) and
          (:OSDPlanlamaIzineBagli = 0 or OrmeSiparisDetay.Planlanabilir = 1) and
          (:ADokumaMakinaBolumId is Null or OrmeSiparisDetay.DokumaMakinaBolumId = :ADokumaMakinaBolumId)
    into :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo,  :BoyaSiparisYil, :BoyaSiparisId, :MusteriSiparisNo, :ReferansSiparisNo, :MakinaGrupId,
         :ArtikelNo, :HamId, :HamAdi, :RenkId, :RenkAdi, :SiparisMiktari, :ToplamDokunan, :ToplamSakatDokunan, :StoktanKullanilacakMiktar,
         :SatinAlinacakMiktar, :PlanlananMiktar, :SiparisFirmaId, SiparisFirmaAdi, :KumasTemin,
         :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi, :SiparisTarihi, :TeslimTarihi, :RevizeTeslimTarihi,
         :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :KaliteAdi do
  begin
    if (RevizeTeslimTarihi is not null) then TeslimTarihi = RevizeTeslimTarihi;

    Select Sum(GerceklesenMiktar) from DokumaPlanlama
      where DokumaPlanlama.OrmeSiparisDetayID = :OrmeSiparisDetayID into :PlanliGerceklesen;

    if (PlanliGerceklesen is null) then PlanliGerceklesen = 0;

    Uretim = ToplamDokunan - PlanliGerceklesen;

    BoyaSiparisIdStr = '';
    if (BoyaSiparisYil is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    PlanlanacakMiktar = SiparisMiktari - PlanlananMiktar - Uretim - StoktanKullanilacakMiktar - SatinAlinacakMiktar;


    PlanlandigiMakinalar  = '';
    PlanlandigiFasoncular = '';

    if (PlanlanacakMiktar > 0) then
    begin
      if (PlanlananMiktar > 0) then /* planlandıgı yerleri göstermek icin */
      begin

        for Select DokumaPlanlama.DokumaMakinaId, Firma.KisaAdi, min(Bitti) from DokumaPlanlama
        left join Firma on Firma.FirmaID = DokumaPlanlama.FirmaId
        where DokumaPlanlama.OrmeSiparisDetayID = :OrmeSiparisDetayID
        group by DokumaPlanlama.DokumaMakinaId, Firma.KisaAdi
        order by DokumaPlanlama.DokumaMakinaId, Firma.KisaAdi
        into :FDokumaMakinaId, :FFirmaKisaAdi, :FBitti do
        begin
          if (FDokumaMakinaID is not null) then
          begin
            if ( Char_Length(PlanlandigiMakinalar) + Char_Length(FDokumaMakinaID) < 230) then
            begin
              if (PlanlandigiMakinalar <> '') then PlanlandigiMakinalar = PlanlandigiMakinalar || cast(', ' as varchar(2) character set UTF8);
              PlanlandigiMakinalar = PlanlandigiMakinalar || FDokumaMakinaID;

              if (FBitti = 0) then PlanlandigiMakinalar = PlanlandigiMakinalar || '(-)';
              else PlanlandigiMakinalar = PlanlandigiMakinalar || '(+)';
            end
          end
          else if (FFirmaKisaAdi is not null) then
          begin
            if (Char_Length(PlanlandigiFasoncular) + Char_Length(FFirmaKisaAdi) < 230) then
            begin
              if (PlanlandigiFasoncular <> '') then PlanlandigiFasoncular = PlanlandigiFasoncular || cast(', ' as varchar(2) character set UTF8);
              PlanlandigiFasoncular = PlanlandigiFasoncular || FFirmaKisaAdi;

              if (FBitti = 0) then PlanlandigiFasoncular = PlanlandigiFasoncular || '(-)';
              else PlanlandigiFasoncular = PlanlandigiFasoncular || '(+)';
            end
          end
        end
      end

      DokunacakMiktar = SiparisMiktari - ToplamDokunan;

      MinBoyaliTerminTarihi = Null;
      MaxBoyaliTerminTarihi = Null;

      Select Min(SonTerminTarihi), max(SonTerminTarihi) from BoyaSiparisDetay where BoyaSiparisDetay.OrmeSiparisDetayID = :OrmeSiparisDetayID into :MinBoyaliTerminTarihi, :MaxBoyaliTerminTarihi;

      suspend;
    end
  end
end
