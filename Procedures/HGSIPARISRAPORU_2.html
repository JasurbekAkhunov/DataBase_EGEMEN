CREATE PROCEDURE HGSIPARISRAPORU_2(
    ABASLANGICSIPARISYIL INTEGER,
    ABASLANGICSIPARISID INTEGER,
    ABITISSIPARISYIL INTEGER,
    ABITISSIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASINIF VARCHAR(1),
    AMODELID MODELKODUSIZE,
    ASIPONAYDURUMU SMALLINT,
    AMALIONAYDURUMU SMALLINT)
RETURNS (
    HGISEMRIKAYITID INTEGER,
    HGSIPARISKAYITID INTEGER,
    HGSIPARISDETAYID INTEGER,
    TARIH TIMESTAMP,
    FIRMAID FIRMAKODUSIZE,
    FIRMAADI ADI100,
    ORMEMAKINAID MAKINAKODUSIZE,
    PARCAISCILIK EGEBOOLEAN,
    PARCAISCILIKSTR VARCHAR(5),
    KESIMTALIMATI EGEBOOLEAN,
    KESIMTALIMATISTR VARCHAR(5),
    URETIMPARTINO VARCHAR(30),
    NUMARATOR INTEGER,
    TERMINTARIHI TIMESTAMP,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    VADEGUN INTEGER,
    NOTU MEMO,
    BITTI EGEBOOLEAN,
    BITTISTR VARCHAR(5),
    ONAYDURUMU SMALLINT,
    ONAYDURUMUSTR VARCHAR(20),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI ADI40,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
Declare Variable ModelKayitId Integer;
begin
  Select ModelKayitId From Model Where Model.ModelID = :AModelID into :ModelKayitId;

  for select HgIsEmri.HGIsEmriKayitID, HgIsEmri.HGSiparisKayitID, HgIsEmri.HGSiparisDetayID, HgIsEmri.Tarih, HgIsEmri.FirmaId, Firma.Adi, HgIsEmri.OrmeMakinaId,
             HgIsEmri.ParcaIscilik, HgIsEmri.KesimTalimati, HgIsEmri.UretimPartiNo, HgIsEmri.Numarator, HgIsEmri.TerminTarihi, HgIsEmri.BirimFiyat, HgIsEmri.DovizId,
             HgIsEmri.VadeGun, HgIsEmri.Notu, HgIsEmri.Bitti, HgIsEmri.OnayDurumu, HgIsEmri.OnayTarihi, HgIsEmri.OnaylayanKullaniciId, OK.Adi,
             IK.Adi, HgIsEmri.InsertTarihi, SK.Adi, HgIsEmri.DegisimTarihi
      from HgSiparis
      Left Join HgSiparisDetay on HgSiparisDetay.HgSiparisKayitId = HgSiparis.HgSiparisKayitId
      Left Join HgIsEmri on HgIsEmri.HgSiparisDetayId = HgSiparisDetay.HgSiparisDetayId
      Left Join Firma on HgIsEmri.FirmaId = Firma.FirmaId
      Left Join Kullanici OK on OK.KullaniciId = HgIsEmri.OnaylayanKullaniciId
      Left Join Kullanici IK ON IK.KullaniciId = HgIsEmri.InsertKullaniciID
      Left Join Kullanici SK ON SK.KullaniciId = HgIsEmri.KullaniciID
      where (:ARaporTipi = 1 and HgSiparis.SiparisTarihi >= :ABaslangicSiparisTarihi and HgSiparis.SiparisTarihi <= :ABitisSiparisTarihi or
             :ARaporTipi = 2 and HgSiparisDetay.TeslimTarihi >= :ABaslangicSiparisTarihi and HgSiparisDetay.TeslimTarihi <= :ABitisSiparisTarihi or
             :ARaporTipi = 0 and
                (HgSiparis.HgSiparisYil > :ABaslangicSiparisYil or (HgSiparis.HgSiparisYil = :ABaslangicSiparisYil and HgSiparis.HgSiparisId >= :ABaslangicSiparisId)) and
                (HgSiparis.HgSiparisYil < :ABitisSiparisYil or (HgSiparis.HgSiparisYil = :ABitisSiparisYil  and HgSiparis.HgSiparisId <= :ABitisSiparisId))
             ) and
            (HgSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
            (HgSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
            (HgSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
            (HgSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null) and
            (HgSiparis.FirmaId = :AFirmaId or :AFirmaId is null) and
            (HgSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
            (HgSiparisDetay.Sinif = :ASinif or :ASinif is null) and
            (HgSiparisDetay.ModelKayitId = :ModelKayitId or :ModelKayitId is Null) and
            (HgSiparisDetay.OnayDurumu = :ASipOnayDurumu or :ASipOnayDurumu is Null) and
            (HgSiparisDetay.MaliOnayDurumu = :AMaliOnayDurumu or :AMaliOnayDurumu is Null) and
            (HgSiparisDetay.Kapali = :AKapali or :AKapali is null) and
            (HgSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null)
            into :HGIsEmriKayitID, :HGSiparisKayitID, :HGSiparisDetayID, :Tarih, :FirmaId, :FirmaAdi, :OrmeMakinaId, :ParcaIscilik, :KesimTalimati, :UretimPartiNo,
             :Numarator, :TerminTarihi, :BirimFiyat, :DovizId, :VadeGun, :Notu, :Bitti, :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi,
             :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    ParcaIscilikStr = 'Hay覺r';
    if (ParcaIscilik = 1) then ParcaIscilikStr = 'Evet';

    BittiStr = 'Hay覺r';
    if (Bitti = 1) then BittiStr = 'Evet';

    KesimTalimatiStr = 'Hay覺r';
    if (KesimTalimati = 1) then KesimTalimatiStr = 'Evet';

    OnayDurumuStr = '';
    if (OnayDurumu = 1) then
      OnayDurumuStr = 'Onayland覺';
    else if (OnayDurumu = 2) then
      OnayDurumuStr = 'Reddedildi';

    suspend;
  end
end
