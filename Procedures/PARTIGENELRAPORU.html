CREATE PROCEDURE PARTIGENELRAPORU(
    AIMALATTIPIID INTEGER,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    AFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    ARECETEID VARCHAR(20),
    AMODELNO VARCHAR(20),
    ABASLANGICACILISTARIHI TIMESTAMP,
    ABITISACILISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ASINIF VARCHAR(1),
    AISLEMGRUBU INTEGER,
    AISLEMTIPI INTEGER,
    AFATURAKESILECEK INTEGER,
    AACILIYETDERECESI INTEGER,
    AFASON INTEGER,
    ARECETEOKEY INTEGER,
    ATALIMAT INTEGER,
    ATALIMATOKEY INTEGER,
    ARAPORTIPI INTEGER,
    ARECETEYAZILDI INTEGER,
    APARTIGRUPID VARCHAR(4))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    MODELNO VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    ISLEMGRUBUADI VARCHAR(20),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ACILIYETDERECESIADI VARCHAR(20),
    PLANLANANIMALATTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    RECETEOKEY VARCHAR(3),
    TALIMAT VARCHAR(3),
    TALIMATOKEY VARCHAR(3),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SEVK VARCHAR(7),
    FATURAKESILECEK VARCHAR(4),
    FASON VARCHAR(5),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FATURAKESILDI VARCHAR(4),
    SINIF VARCHAR(1),
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    SEVKKILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    NETMETRE NUMERIC(12,3),
    SEVKMETRE NUMERIC(12,3),
    ADET INTEGER,
    SEVKADET INTEGER,
    KAPADEDI INTEGER,
    SEVKKAPADEDI INTEGER,
    RECETEYAZILDI VARCHAR(4),
    RECETEDETAYVAR INTEGER,
    RECETEDETAYVARSTR VARCHAR(3),
    REFAKATKARTIYAZILDI VARCHAR(4),
    BOYAKIMYASALMALIYETI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    TAMIRTIPIADI VARCHAR(100),
    TAMIRDUZELTMETIPIADI VARCHAR(100),
    TEKRARURETIM INTEGER,
    TEKRARURETIMSTR VARCHAR(20))
AS
declare variable FReceteOkey            integer;
declare variable FTalimat               integer;
declare variable FTalimatOkey           integer;
declare variable FParcaliSevk           integer;
declare variable FFaturaKesilecek       integer;
declare variable FFason                 integer;
declare variable FFaturaKesildi         integer;
declare variable FReceteyazildi         integer;
declare variable FRefakatKartiYazildi   integer;
declare variable FTamirTipiAdi          varchar(20);
declare variable FTamirDuzeltmeTipiAdi  varchar(30);
begin
  Select StokTabanDovizId from sistem where sistemid = 1 into :DovizId;

  if (ARaporTipi = 0) then
  begin
    for select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi, Parti.ModelNo,
           Parti.receteId, Recete.Adi, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
           IslemGrubu.Adi, Parti.IslemTipiId, IslemTipi.Adi, AciliyetDerecesi.Adi, Parti.PlanlananImalatTarihi,
           Parti.TerminTarihi, Parti.ReceteOkey, Parti.Talimat, Parti.TalimatOkey, Parti.Notu, Parti.ParcaliSevk,
           Parti.FaturaKesilecek, Parti.Fason, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
           Parti.FaturaKesildi, Parti.Sinif, Parti.Receteyazildi, Recete.ReceteDetayVar, Parti.RefakatKartiYazildi from Parti
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    where (Parti.PartiYil > :ABaslangicPartiYil or (Parti.PartiYil = :ABaslangicPartiYil and Parti.PartiId >= :ABaslangicPartiId)) and
          (Parti.PartiYil < :ABitisPartiYil or (Parti.PartiYil = :ABitisPartiYil  and Parti.PartiId <= :ABitisPartiId)) and
          (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
          (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and
          (:AReceteId is null or Parti.ReceteId = :AReceteId) and
          (:AModelNo is null or Parti.ModelNo = :AModelNo) and
          (BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil or :ABoyaSiparisYil is null) and
          (BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId or :ABoyaSiparisId is null) and
          (Parti.Sinif = :ASinif or :ASinif is null) and
          (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null) and
          (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
          (:AIslemGrubu is null or Parti.IslemGrubuId = :AIslemGrubu) and
          (:AIslemTipi is null or Parti.IslemTipiId = :AIslemTipi) and
          (:AAciliyetDerecesi is null or Parti.AciliyetDerecesiId = :AAciliyetDerecesi) and
          (Parti.Fason = :AFason or :AFason is null) and
          (Parti.ReceteOkey = :AReceteOkey or :AReceteOkey is null) and
          (Parti.FaturaKesilecek = :AFaturaKesilecek or :AFaturaKesilecek is null) and
          (Parti.Talimat = :ATalimat or :ATalimat is null) and
          (Parti.TalimatOkey = :ATalimatOkey or :ATalimatOkey is null) and
          (:AReceteYazildi is null or (Parti.ReceteId is not null and Parti.ReceteYazildi = :AReceteYazildi))
    into :PartiKayitId, :PartiYil, :PartiId, :AcilisTarihi, :PartigrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi, :ModelNo, :ReceteId, :ReceteAdi,
         :DesenId, :DesenAdi, :VaryantAdi, :IslemGrubuAdi, :IslemTipiId, :IslemTipiAdi, :AciliyetDerecesiAdi, :PlanlananImalatTarihi, :TerminTarihi,
         :FReceteOkey, :FTalimat, :FTalimatOkey, :Notu, :FParcaliSevk, :FFaturaKesilecek, :FFason, :BoyaSiparisYil,
         :BoyaSiparisId, :FFaturaKesildi, :Sinif, :FReceteyazildi, :ReceteDetayVar, :FRefakatKartiYazildi do
    begin
      if (FReceteOkey = 1) then ReceteOkey = 'Var';  else ReceteOkey = null;
      if (FTalimat = 1) then Talimat = 'Var';  else Talimat = null;
      if (FTalimatOkey = 1) then TalimatOkey = 'Var';  else TalimatOkey = null;
      if (FParcaliSevk = 1) then Sevk = 'Parcali';  else Sevk = null;
      if (FFaturaKesilecek = 1) then FaturaKesilecek = 'Evet';  else FaturaKesilecek = null;
      if (FFason = 1) then Fason = 'Evet';  else Fason = null;
      if (FFaturaKesildi = 1) then FaturaKesildi = 'Evet';  else FaturaKesildi = null;

      Select sum(PartiStok.Kilo), Sum(PartiStok.NetKilo), Sum(PartiStok.NetSevkKilo),
             sum(PartiStok.Metre), Sum(PartiStok.NetMetre), Sum(PartiStok.NetSevkMetre),
             Sum(PartiStok.Adet), Sum(PartiStok.NetSevkAdet), Sum(PartiStok.KapAdedi), Sum(PartiStok.NetSevkKapAdedi)
             from PartiStok where PartiStok.PartiKayitId = :PartiKayitId
       into :BrutKilo, :NetKilo, :SevkKilo, :BrutMetre, :NetMetre, :SevkMetre,
            :Adet, :SevkAdet, :KapAdedi, :SevkKapAdedi;

      if (PartiYil is null) then PartiIdStr = '';
      else PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (BoyaSiparisYil is null) then BoyaSiparisIdStr = '';
      else BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      if (FReceteYazildi = 1) then ReceteYazildi = 'Evet';
      else ReceteYazildi = '';

      if (ReceteDetayVar = 0) then ReceteDetayVarStr = 'Bo≈ü';
      else ReceteDetayVarStr = '';

      if (FRefakatKartiYazildi = 1) then RefakatKartiYazildi = 'Evet';
      else RefakatKartiYazildi = '';

      BoyaKimyasalMaliyeti = 0;

      Select sum(StokDetay.Miktar * StokDetay.BirimFiyat) from StokDetay
        where StokDetay.PartiKayitId = :PartiKayitId and StokDetay.HareketTipi = 6
        into :BoyaKimyasalMaliyeti;

      TamirTipiAdi          = '';
      TamirDuzeltmeTipiAdi  = '';

      if (IslemTipiId = 3) then
      begin
        For Select TamirTipi.Adi from PartiStok
          left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
          where PartiStok.PartiKayitID = :PartiKayitID and PartiStok.TamirTipiId is not null
          group by TamirTipi.Adi
          into :FTamirTipiAdi do
        begin
          if (Char_Length(TamirTipiAdi) + Char_Length(FTamirTipiAdi) < 97) then
          begin
            if (TamirTipiAdi <> '' ) then TamirTipiAdi = TamirTipiAdi || cast(', ' as varchar(2) character set UTF8);
            TamirTipiAdi = TamirTipiAdi || FTamirTipiAdi;
          end
        end

        For Select TamirDuzeltmeTipi.Adi from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayId
          left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
          left join DisTamir on DisTamir.HamStokMasterID = HamStokMaster.HamStokMasterId
          Left Join TamirDuzeltmeTipi on TamirDuzeltmeTipi.TamirDuzeltmeTipiId = DisTamir.TamirDuzeltmeTipiId
          where PartiStokDetay.PartiKayitID = :PartiKayitID and DisTamir.TamirDuzeltmeTipiId is not null
          group by TamirDuzeltmeTipi.Adi
          into :FTamirDuzeltmeTipiAdi do
        begin
          if (Char_Length(TamirDuzeltmeTipiAdi) + Char_Length(FTamirDuzeltmeTipiAdi) < 97) then
          begin
            if (TamirDuzeltmeTipiAdi <> '' ) then TamirDuzeltmeTipiAdi = TamirDuzeltmeTipiAdi || cast(', ' as varchar(2) character set UTF8);
            TamirDuzeltmeTipiAdi = TamirDuzeltmeTipiAdi || FTamirDuzeltmeTipiAdi;
          end
        end
      end

      suspend;
    end
  end
  else
  begin
   for select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi, Parti.ModelNo,
           Parti.receteId, Recete.Adi, DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
           IslemGrubu.Adi, Parti.IslemTipiId, IslemTipi.Adi, AciliyetDerecesi.Adi, Parti.PlanlananImalatTarihi,
           Parti.TerminTarihi, Parti.ReceteOkey, Parti.Talimat, Parti.TalimatOkey, Parti.Notu, Parti.ParcaliSevk,
           Parti.FaturaKesilecek, Parti.Fason, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
           Parti.FaturaKesildi, Parti.Sinif, Parti.Receteyazildi, Recete.ReceteDetayVar, Parti.RefakatKartiYazildi from Parti
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    where (Parti.AcilisTarihi >= :ABaslangicAcilisTarihi and Parti.AcilisTarihi <= :ABitisAcilisTarihi ) and
          (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
          (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and
          (:AReceteId is null or Parti.ReceteId = :AReceteId) and
          (:AModelNo is null or Parti.ModelNo = :AModelNo) and
          (BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil or :ABoyaSiparisYil is null) and
          (BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId or :ABoyaSiparisId is null) and
          (Parti.Sinif = :ASinif or :ASinif is null) and
          (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null) and
          (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
          (:AIslemGrubu is null or Parti.IslemGrubuId = :AIslemGrubu) and
          (:AIslemTipi is null or Parti.IslemTipiId = :AIslemTipi) and
          (:AAciliyetDerecesi is null or Parti.AciliyetDerecesiId = :AAciliyetDerecesi) and
          (Parti.Fason = :AFason or :AFason is null) and
          (Parti.ReceteOkey = :AReceteOkey or :AReceteOkey is null) and
          (Parti.FaturaKesilecek = :AFaturaKesilecek or :AFaturaKesilecek is null) and
          (Parti.Talimat = :ATalimat or :ATalimat is null) and
          (Parti.TalimatOkey = :ATalimatOkey or :ATalimatOkey is null) and
          (:AReceteYazildi is null or (Parti.ReceteId is not null and Parti.ReceteYazildi = :AReceteYazildi))
    into :PartiKayitId, :PartiYil, :PartiId, :AcilisTarihi, :PartigrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi, :ModelNo, :ReceteId, :ReceteAdi,
         :DesenId, :DesenAdi, :VaryantAdi, :IslemGrubuAdi, :IslemTipiId, :IslemTipiAdi, :AciliyetDerecesiAdi, :PlanlananImalatTarihi, :TerminTarihi,
         :FReceteOkey, :FTalimat, :FTalimatOkey, :Notu, :FParcaliSevk, :FFaturaKesilecek, :FFason, :BoyaSiparisYil,
         :BoyaSiparisId, :FFaturaKesildi, :Sinif, :FReceteyazildi, :ReceteDetayVar, :FRefakatKartiYazildi do
    begin
      if (FReceteOkey = 1) then ReceteOkey = 'Var';  else ReceteOkey = null;
      if (FTalimat = 1) then Talimat = 'Var';  else Talimat = null;
      if (FTalimatOkey = 1) then TalimatOkey = 'Var';  else TalimatOkey = null;
      if (FParcaliSevk = 1) then Sevk = 'Parcali';  else Sevk = null;
      if (FFaturaKesilecek = 1) then FaturaKesilecek = 'Evet';  else FaturaKesilecek = null;
      if (FFason = 1) then Fason = 'Evet';  else Fason = null;
      if (FFaturaKesildi = 1) then FaturaKesildi = 'Evet';  else FaturaKesildi = null;

      Select sum(PartiStok.Kilo), Sum(PartiStok.NetKilo), Sum(PartiStok.NetSevkKilo),
             sum(PartiStok.Metre), Sum(PartiStok.NetMetre), Sum(PartiStok.NetSevkMetre),
             Sum(PartiStok.Adet), Sum(PartiStok.NetSevkAdet), Sum(PartiStok.KapAdedi), Sum(PartiStok.NetSevkKapAdedi)
             from PartiStok where PartiStok.PartiKayitId = :PartiKayitId
         into :BrutKilo, :NetKilo, :SevkKilo, :BrutMetre, :NetMetre, :SevkMetre,
              :Adet, :SevkAdet, :KapAdedi, :SevkKapAdedi;

      if (PartiYil is null) then PartiIdStr = '';
      else PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId)  || ' ';

      if (BoyaSiparisYil is null) then BoyaSiparisIdStr = '';
      else BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      if (FReceteYazildi = 1) then ReceteYazildi = 'Evet';
      else ReceteYazildi = '';

      if (ReceteDetayVar = 0) then ReceteDetayVarStr = 'Bo≈ü';
      else ReceteDetayVarStr = '';

      if (FRefakatKartiYazildi = 1) then RefakatKartiYazildi = 'Evet';
      else RefakatKartiYazildi = '';

      BoyaKimyasalMaliyeti = 0;

      Select sum(StokDetay.Miktar * StokDetay.BirimFiyat) from StokDetay
        where StokDetay.PartiKayitId = :PartiKayitId and StokDetay.HareketTipi = 6
        into :BoyaKimyasalMaliyeti;


      TamirTipiAdi          = '';
      TamirDuzeltmeTipiAdi  = '';

      if (IslemTipiId = 3) then
      begin
        For Select TamirTipi.Adi from PartiStok
          left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
          where PartiStok.PartiKayitID = :PartiKayitID and PartiStok.TamirTipiId is not null
          group by TamirTipi.Adi
          into :FTamirTipiAdi do
        begin
          if (Char_Length(TamirTipiAdi) + Char_Length(FTamirTipiAdi) < 97) then
          begin
            if (TamirTipiAdi <> '' ) then TamirTipiAdi = TamirTipiAdi || cast(', ' as varchar(2) character set UTF8);
            TamirTipiAdi = TamirTipiAdi || FTamirTipiAdi;
          end
        end

        For Select TamirDuzeltmeTipi.Adi from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayId
          left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
          left join DisTamir on DisTamir.HamStokMasterID = HamStokMaster.HamStokMasterId
          Left Join TamirDuzeltmeTipi on TamirDuzeltmeTipi.TamirDuzeltmeTipiId = DisTamir.TamirDuzeltmeTipiId
          where PartiStokDetay.PartiKayitID = :PartiKayitID and DisTamir.TamirDuzeltmeTipiId is not null
          group by TamirDuzeltmeTipi.Adi
          into :FTamirDuzeltmeTipiAdi do
        begin
          if (Char_Length(TamirDuzeltmeTipiAdi) + Char_Length(FTamirDuzeltmeTipiAdi) < 97) then
          begin
            if (TamirDuzeltmeTipiAdi <> '' ) then TamirDuzeltmeTipiAdi = TamirDuzeltmeTipiAdi || cast(', ' as varchar(2) character set UTF8);
            TamirDuzeltmeTipiAdi = TamirDuzeltmeTipiAdi || FTamirDuzeltmeTipiAdi;
          end
        end
      end

      suspend;
    end
  end
