CREATE PROCEDURE KARTELATALEPRAPORU(
    ARAPORTIPI INTEGER,
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKALITEID VARCHAR(20))
RETURNS (
    KARTELATALEPKAYITID INTEGER,
    KARTELATALEPYIL INTEGER,
    KARTELATALEPID INTEGER,
    KARTELATALEPIDSTR VARCHAR(30),
    KARTELATALEPDETAYID INTEGER,
    TALEPTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KALITEID VARCHAR(20),
    KALITEADI ADI255,
    ACIKLAMA VARCHAR(255),
    KARTELAFISIDETAYID INTEGER,
    MUSTERI VARCHAR(60),
    MIKTAR BOYAMIKTARSIZE,
    BIRIMID BIRIMKODUSIZE,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID DOVIZSIZE)
AS
begin
  if (ARaporTipi = 0) then
  begin
    for select KartelaTalep.KartelaTalepKayitId, KartelaTalep.KartelaTalepYil, KartelaTalep.KartelaTalepId, KartelaTalep.FirmaId,
               Firma.Adi, KartelaTalep.TalepTarihi, KartelaTalep.Notu, KartelaTalepDetay.KartelaTalepDetayId, KartelaTalepDetay.KaliteId, Kalite.KaliteAdi, KartelaTalepDetay.Aciklama,
               KartelaTalepDetay.KartelaFisiDetayID, KartelaTalepDetay.Musteri, KartelaTalepDetay.Miktar, KartelaTalepDetay.BirimID, KartelaTalepDetay.BirimFiyat,
               KartelaTalepDetay.DovizId from KartelaTalepDetay
    left join KartelaTalep on KartelaTalep.KartelaTalepKayitid = KartelaTalepDetay.KartelaTalepKayitid
    left join Kalite on Kalite.Kaliteid = KartelaTalepDetay.Kaliteid
    left join Firma on Firma.Firmaid = KartelaTalep.Firmaid
    where KartelaTalepDetay.KartelaFisiDetayId is null and (KartelaTalep.FirmaId = :AFirmaId or :AFirmaId is null) and
         (KartelaTalepDetay.KaliteID = :AKaliteID or :AKaliteID is null)
    order by KartelaTalep.TalepTarihi, KartelaTalep.KartelaTalepYil, KartelaTalep.KartelaTalepId, KartelaTalepDetay.KartelaTalepDetayId
    into :KartelaTalepKayitId, :KartelaTalepYil, :KartelaTalepId, :FirmaId, :FirmaAdi, :TalepTarihi, :Notu, :KartelaTalepDetayId,  :KaliteId, :KaliteAdi,
         :Aciklama, :KartelaFisiDetayID, :Musteri, :Miktar, :BirimID, :BirimFiyat, :DovizId do
    begin
      KartelaTalepIdStr = '(' || EgemenIntegerToStr(KartelaTalepYil)  || ') ' || EgemenIntegerToStr(KartelaTalepId) || ' ';

      suspend;
    end
  end
  else if (ARaporTipi = 1) then
  begin
    for select KartelaTalep.KartelaTalepKayitId, KartelaTalep.KartelaTalepYil, KartelaTalep.KartelaTalepId, KartelaTalep.FirmaId,
               Firma.Adi, KartelaTalep.TalepTarihi, KartelaTalep.Notu, KartelaTalepDetay.KartelaTalepDetayId, KartelaTalepDetay.KaliteId, Kalite.KaliteAdi, KartelaTalepDetay.Aciklama,
               KartelaTalepDetay.KartelaFisiDetayID, KartelaTalepDetay.Musteri, KartelaTalepDetay.Miktar, KartelaTalepDetay.BirimID, KartelaTalepDetay.BirimFiyat,
               KartelaTalepDetay.DovizId  from KartelaTalep
    left join KartelaTalepDetay on KartelaTalepDetay.KartelaTalepKayitid = KartelaTalep.KartelaTalepKayitid
    left join Kalite on Kalite.Kaliteid = KartelaTalepDetay.Kaliteid
    left join Firma on Firma.Firmaid = KartelaTalep.Firmaid
    where KartelaTalep.TalepTarihi >= :ABaslangicTarihi and KartelaTalep.TalepTarihi <= :ABitisTarihi and
          (KartelaTalep.FirmaId = :AFirmaId or :AFirmaId is null) and
           KartelaTalepDetay.KartelaFisiDetayID is not null and
          (KartelaTalepDetay.KaliteID = :AKaliteID or :AKaliteID is null)
    order by KartelaTalep.TalepTarihi, KartelaTalep.KartelaTalepYil, KartelaTalep.KartelaTalepId, KartelaTalepDetay.KartelaTalepDetayId
    into :KartelaTalepKayitId, :KartelaTalepYil, :KartelaTalepId, :FirmaId, :FirmaAdi, :TalepTarihi, :Notu, :KartelaTalepDetayId, :KaliteId, :KaliteAdi,
         :Aciklama, :KartelaFisiDetayID, :Musteri, :Miktar, :BirimID, :BirimFiyat, :DovizId do
    begin
      KartelaTalepIdStr = '(' || EgemenIntegerToStr(KartelaTalepYil)  || ') ' || EgemenIntegerToStr(KartelaTalepId) || ' ';

      suspend;
    end
  end
  else if (ARaporTipi is null) then
  begin
    for select KartelaTalep.KartelaTalepKayitId, KartelaTalep.KartelaTalepYil, KartelaTalep.KartelaTalepId, KartelaTalep.FirmaId,
               Firma.Adi, KartelaTalep.TalepTarihi, KartelaTalep.Notu, KartelaTalepDetay.KartelaTalepDetayId, KartelaTalepDetay.KaliteId, Kalite.KaliteAdi, KartelaTalepDetay.Aciklama,
               KartelaTalepDetay.KartelaFisiDetayID, KartelaTalepDetay.Musteri, KartelaTalepDetay.Miktar, KartelaTalepDetay.BirimID, KartelaTalepDetay.BirimFiyat,
               KartelaTalepDetay.DovizId  from KartelaTalep
    left join KartelaTalepDetay on KartelaTalepDetay.KartelaTalepKayitid = KartelaTalep.KartelaTalepKayitid
    left join Kalite on Kalite.Kaliteid = KartelaTalepDetay.Kaliteid
    left join Firma on Firma.Firmaid = KartelaTalep.Firmaid
    where KartelaTalep.TalepTarihi >= :ABaslangicTarihi and KartelaTalep.TalepTarihi <= :ABitisTarihi and
         (KartelaTalep.FirmaId = :AFirmaId or :AFirmaId is null) and
         (KartelaTalepDetay.KaliteID = :AKaliteID or :AKaliteID is null)
    order by KartelaTalep.TalepTarihi, KartelaTalep.KartelaTalepYil, KartelaTalep.KartelaTalepId, KartelaTalepDetay.KartelaTalepDetayId
    into :KartelaTalepKayitId, :KartelaTalepYil, :KartelaTalepId, :FirmaId, :FirmaAdi, :TalepTarihi, :Notu, :KartelaTalepDetayId,  :KaliteId, :KaliteAdi,
         :Aciklama, :KartelaFisiDetayID, :Musteri, :Miktar, :BirimID, :BirimFiyat, :DovizId do
    begin
      KartelaTalepIdStr = '(' || EgemenIntegerToStr(KartelaTalepYil)  || ') ' || EgemenIntegerToStr(KartelaTalepId) || ' ';

      suspend;
    end
  end
end
