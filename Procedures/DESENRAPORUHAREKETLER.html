declare variable shtAlis        integer;
declare variable shtSatis       integer;
declare variable shtAlisIade    integer;
declare variable shtSatisIade   integer;
declare variable shtDepoTransfer integer;
declare variable shtUretimeCikis integer;
declare variable shtUretim integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable Okey            integer;
declare variable SDovizId varchar(5);
declare variable IDovizId varchar(5);
declare variable FDovizId varchar(5);
declare variable DovizKuru Double Precision;
declare variable StokTabanDovizId varchar(5);
declare Variable FBirimCarpan Double Precision;
begin
  Select StokTabanDovizId from Sistem where SistemId = 1 into :StokTabanDovizId;

  select shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtDepoTransfer, shtUretimeCikis, shtUretim, shtFasonUretimeCikis, shtFasonUretim from StokHareketTipiGetir
         into :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtDepoTransfer, :shtUretimeCikis, :shtUretim, :shtFasonUretimeCikis, :shtFasonUretim;

  For Select Desen.DesenKayitId, Desen.DesenId, Urun.UrunId, Urun.Adi, Urun.Raf, Urun.UrunKategoriId, Urun.KimyasalTipiId, KimyasalTipi.Adi,
             UrunFiyat.AlisFiyati, urunFiyat.AlisFiyatiDovizId,
             StokDetay.StokDetayId, StokDetay.StokId, StokDetay.IrsaliyeId, StokDetay.FaturaId, StokDetay.HareketTipi, StokDetay.FirmaId,
             Firma.Adi, Stok.FisNo, Stok.FisTarihi, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi, Irsaliye.FiiliSevkTarihi, Fatura.FaturaNo, Fatura.FaturaTarihi,
             StokDetay.DepoId, Dp.Adi, StokDetay.TransferDepoId, TDp.Adi, StokDetay.StokGrupId, Irsaliye.IrsaliyeGrupId, Fatura.FaturaGrupId,
             SGrup.Adi, IGrup.Adi, FGrup.Adi,
             StokDetay.Miktar, StokDetay.NetBirimFiyat,
             StokDetay.UrunBirimId, UrunBirim.BirimId, StokDetay.Notu, StokDetay.Yatirim,
             Ik.Adi, StokDetay.InsertTarihi, Sk.Adi, StokDetay.DegisimTarihi, StokDetay.Tarih,
             Stok.DovizId, Irsaliye.DovizId, Fatura.DovizId, VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId,
             StokDetay.UniteID, Unite.Adi, coalesce(Stok.MasrafMerkeziId, StokDetay.MasrafMerkeziId), coalesce(MasrafMerkezi.Adi, DetayMasrafMerkezi.Adi), coalesce(Stok.DepartmanID, StokDetay.DepartmanID), coalesce(Departman.Adi, DetayDepartman.Adi),
             StokDetay.TabanDovizKuru1, StokDetay.TabanDovizKuru2, Urun.UrunTipiId, UrunTipi.Adi,
             Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi, StokDetay.MakinaId, Makina.Adi, UrunBirim.Carpan,
             coalesce(Stok.Sinif, coalesce(Irsaliye.Sinif, Fatura.Sinif)), StokDetay.FiyatAciklama, Stok.DengelemeEvraki,
             StokDetay.OrmeMakinaId, OrmeMakina.Adi, StokDetay.IplikMakinaId, IplikMakina.Adi, Urun.TransferKodu, Urun.AlisTransferKodu, StokDetay.Lot,
             Parti.PartiYil, Parti.PartiID, Parti.FirmaID, PartiFirma.Adi, Parti.ReceteID, Recete.Adi, Stok.IlgiliKullaniciID, IlgiliKullanici.Adi,
             MakinaGrup.MakinaGrupID, MakinaGrup.Adi, OrmeMakinaGrup.MakinaGrupID, OrmeMakinaGrup.Adi, IplikMakinaGrup.MakinaGrupID, IplikMakinaGrup.Adi, StokDetay.DokumaMakinaId, DokumaMakina.Adi, DokumaMakinaGrup.MakinaGrupID, DokumaMakinaGrup.Adi,
             iif(StokDetay.MakinaID is not Null, StokDetay.MakinaID, '') || iif(StokDetay.OrmeMakinaID is not Null, StokDetay.OrmeMakinaID, '') || iif(StokDetay.IplikMakinaID is not Null,StokDetay.IplikMakinaID,'') || iif(StokDetay.DokumaMakinaID is not Null,StokDetay.DokumaMakinaID,'') as RaporMakinaID,
             iif(Makina.Adi is not Null, Makina.Adi, '') || iif(OrmeMakina.Adi is not Null, OrmeMakina.Adi, '') || iif(IplikMakina.Adi is not Null,IplikMakina.Adi,'') || iif(DokumaMakina.Adi is not Null,DokumaMakina.Adi,'') as RaporMakinaAdi,
             iif(MakinaGrup.MakinaGrupID is not Null, MakinaGrup.MakinaGrupID, '') || iif(OrmeMakinaGrup.MakinaGrupID is not Null, OrmeMakinaGrup.MakinaGrupID, '') || iif(IplikMakinaGrup.MakinaGrupID is not Null,IplikMakinaGrup.MakinaGrupID,'') || iif(DokumaMakinaGrup.MakinaGrupID is not Null,DokumaMakinaGrup.MakinaGrupID,'') as RaporMakinaGrupID,
             iif(MakinaGrup.Adi is not Null, MakinaGrup.Adi, '') || iif(OrmeMakinaGrup.Adi is not Null, OrmeMakinaGrup.Adi, '') || iif(IplikMakinaGrup.Adi is not Null,IplikMakinaGrup.Adi,'') || iif(DokumaMakinaGrup.Adi is not Null,DokumaMakinaGrup.Adi,'') as RaporMakinaGrupAdi,
             Stok.IlgiliPersonelID, IlgiliPersonel.Adi, StokDetay.ReferansStokDetayID, Urun.StokUreticiId, StokUretici.Adi  from Desen
    left join StokDetay on StokDetay.DesenId = Desen.DesenId
    left join Urun on urun.UrunId = StokDetay.UrunId
    left join UrunFiyat on UrunFiyat.UrunId = urun.Urunid
    left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = urun.KimyasalTipiid
    left join Stok on Stok.StokId = StokDetay.StokId
    left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
    left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join Firma on Firma.FirmaId = StokDetay.FirmaId
    left join Depo Dp on Dp.DepoId = StokDetay.DepoId
    left join Depo TDp on TDp.DepoId = StokDetay.TransferDepoId
    left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = StokDetay.VerilenSiparisDetayId
    left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisDetay.VerilenSiparisKayitId
    left join StokHareketGrup SGrup on SGrup.StokHareketGrupId = StokDetay.StokGrupId
    left join StokHareketGrup IGrup on IGrup.StokHareketGrupId = Irsaliye.IrsaliyeGrupId
    left join StokHareketGrup FGrup on FGrup.StokHareketGrupId = Fatura.FaturaGrupId
    Left Join Kullanici IK on IK.KullaniciId = StokDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = StokDetay.KullaniciId
    left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
    left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
    left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
    left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
    left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
    left join Makina on Makina.MakinaId = StokDetay.MakinaId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = StokDetay.OrmeMakinaId
    left join IplikMakina on IplikMakina.IplikMakinaId = StokDetay.IplikMakinaId
    left join MasrafMerkezi on MasrafMerkezi.MasrafMerkeziId = Stok.MasrafMerkeziId
    left Join Departman on Departman.DepartmanID = Stok.DepartmanID
    left join MasrafMerkezi DetayMasrafMerkezi on DetayMasrafMerkezi.MasrafMerkeziId = StokDetay.MasrafMerkeziId
    left Join Departman DetayDepartman on DetayDepartman.DepartmanID = StokDetay.DepartmanID
    left Join Unite on Unite.UniteID = StokDetay.UniteID
    left Join Parti on Parti.PartiKayitID = StokDetay.PartiKayitID
    left join Firma PartiFirma on PartiFirma.FirmaId = Parti.FirmaId
    left Join Recete on Recete.ReceteID = Parti.ReceteID
    left Join Kullanici IlgiliKullanici on IlgiliKullanici.KullaniciID = Stok.IlgiliKullaniciID
    left Join Personel IlgiliPersonel on IlgiliPersonel.PersonelID = Stok.IlgiliPersonelID
    left join MakinaGrup on MakinaGrup.MakinaGrupID = Makina.MakinaGrubu
    left join MakinaGrup OrmeMakinaGrup on OrmeMakinaGrup.MakinaGrupID = OrmeMakina.MakinaGrupID
    left join MakinaGrup IplikMakinaGrup on IplikMakinaGrup.MakinaGrupID = IplikMakina.MakinaGrupID
    left join DokumaMakina on DokumaMakina.DokumaMakinaID = StokDetay.DokumaMakinaID
    left join MakinaGrup DokumaMakinaGrup on DokumaMakinaGrup.MakinaGrupID = DokumaMakina.MakinaGrupID
    left join StokUretici on StokUretici.StokUreticiID = Urun.StokUreticiID
    where StokDetay.StokDetayID is not Null and
          Desen.GelisTarihi >= :ABaslangicTarihi and Desen.GelisTarihi <= :ABitisTarihi and
          (Desen.FirmaID = :AFirmaID or :AFirmaId is null) and
          (Desen.TasarimFirmaID = :ATasarimFirmaID or :ATasarimFirmaId is null) and
          (Desen.SablonSayisi = :ASablonSayisi or :ASablonSayisi is null) and
          (Desen.DesenKatalogId = :ADesenKatalogId or :ADesenKatalogId is null) and
          (Desen.DesenGrupId = :ADesenGrupId or :ADesenGrupId is null) and
          (Desen.DesenIcerikId = :ADesenIcerikId or :ADesenIcerikId is null) and
          (Desen.CalismaNo = :ACalismaNo or :ACalismaNo is null) and
          (Desen.TasarimNo = :ATasarimNo or :ATasarimNo is null)
    into :DesenKayitId, :DesenId, :UrunId, :UrunAdi, :Raf, :UrunKategoriId, :KimyasalTipiId, :KimyasalTipiAdi,
         :UrunAlisFiyati, :UrunAlisFiyatiDovizId,
         :StokDetayId, :StokId, :IrsaliyeId, :FaturaId, :HareketTipi, :FirmaId, :FirmaAdi, :FisNo, :FisTarihi, :IrsaliyeNo,
         :IrsaliyeTarihi, :FiiliSevkTarihi, :FaturaNo, :FaturaTarihi, :DepoId, :DepoAdi, :TransferDepoId, :TransferDepoAdi, :FisGrupId, :IrsaliyeGrupId, :FaturaGrupId,
         :FisGrupAdi, :IrsaliyeGrupAdi, :FaturaGrupAdi, :Miktar, :NetBirimFiyat, :UrunBirimId, :BirimId, :Notu, :Yatirim,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :Tarih,
         :SDovizId, :IDovizId, :FDovizId, :VerilenSiparisYil, :VerilenSiparisId, :UniteID, :UniteAdi, :MasrafMerkeziId, :MasrafMerkeziAdi, :DepartmanID, :DepartmanAdi, :TabanDovizKuru1, :TabanDovizKuru2,
         :UrunTipiId, :UrunTipiAdi, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi, :MakinaId, :MakinaAdi, :FBirimCarpan, :Sinif,
         :FiyatAciklama, :DengelemeEvraki, :OrmeMakinaId, :OrmeMakinaAdi, :IplikMakinaId, :IplikMakinaAdi, :TransferKodu , :AlisTransferKodu, :Lot,
         :PartiYil, :PartiID, :PartiFirmaID, :PartiFirmaAdi, :PartiReceteID, :PartiReceteAdi, :IlgiliKullaniciID, :IlgiliKullaniciAdi, :MakinaGrupID, :MakinaGrupAdi, :OrmeMakinaGrupID, :OrmeMakinaGrupAdi,
         :IplikMakinaGrupID, :IplikMakinaGrupAdi, :DokumaMakinaID, :DokumaMakinaAdi, :DokumaMakinaGrupID, :DokumaMakinaGrupAdi, :RaporMakinaID, :RaporMakinaAdi, :RaporMakinaGrupID, :RaporMakinaGrupAdi,
         :IlgiliPersonelID, :IlgiliPersonelAdi, :ReferansStokDetayID, :StokUreticiId, :StokUreticiAdi do
  begin
    select UrunBirim.Carpan, UrunBirim.BirimId from UrunBirim where UrunBirim.UrunId = :UrunId and UrunBirim.DefaultBirim = 1 into :DefaultBirimCarpan, :DefaultBirim;
    if (DefaultBirimCarpan is null or DefaultBirimCarpan = 0) then DefaultBirimCarpan = 1.0;

    ReferansIrsFirmaID = Null;
    ReferansIrsTarihi = Null;
    ReferansIrsNo = Null;
    if (ReferansStokDetayID is not Null) then
    begin
      Select First 1 Irsaliye.FirmaID, Irsaliye.IrsaliyeTarihi, Irsaliye.SeriNo || Irsaliye.IrsaliyeNo from StokDetay
        left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeID
        where StokDetay.StokDetayID = :ReferansStokDetayID
      into :ReferansIrsFirmaID, :ReferansIrsTarihi, :ReferansIrsNo;
    end

    if (DengelemeEvraki = 1) then DengelemeEvrakiStr = 'Evet';
    else DengelemeEvrakiStr = '';

    VerilenSiparisIdStr = '';
    if (VerilenSiparisYil is not null) then VerilenSiparisIdStr = '(' || Format_Float('#0', VerilenSiparisYil) || ') '  || Format_Float('#0', VerilenSiparisId)  || ' ';

    if (FDovizId is not null) then DovizId = FDovizId;
    else if (IDovizId is not null) then DovizId = IDovizId;
    else if (SDovizId is not null) then DovizId = SDovizId;

    if (HareketTipi = shtAlis) then HareketTipiStr = 'Alış';
    else if (HareketTipi = shtSatis) then HareketTipiStr = 'Satış';
    else if (HareketTipi = shtAlisIade) then HareketTipiStr = 'Alış İade';
    else if (HareketTipi = shtSatisIade) then HareketTipiStr = 'Satış İade';
    else if (HareketTipi = shtUretimeCikis) then HareketTipiStr = 'Üretime Çıkış';
    else if (HareketTipi = ShtDepoTransfer) then HareketTipiStr = 'Depo Transfer';
    else if (HareketTipi = ShtUretim) then HareketTipiStr = 'Üretim';
    else if (HareketTipi = shtFasonUretimeCikis) then HareketTipiStr = 'Fason Üretime Çıkış';
    else if (HareketTipi = ShtFasonUretim) then HareketTipiStr = 'Fason Üretim';
    else HareketTipiStr = 'Hatalı!';

    if (Yatirim = 1) then YatirimStr = 'Yatırım';
    else YatirimStr = '';

    Tutar = NetBirimFiyat * Miktar;

    if (DovizId <> StokTabanDovizId) then
    begin
      select DovizKuru, Tarih from EnYakinDovizKuruGetir(:DovizId, :StokTabanDovizId, :Tarih) into :DovizKuru, :DovizKuruTarihi;
      if (Tarih <> DovizKuruTarihi) then DovizKuruMesaj = 'Fatura Tarihine Ait Kur Girisi Bulunamadigi icin ' || TimeStampToDateTimeStr(DovizKuruTarihi) || ' Tarihli Kur ile Hesaplandi';
    end
    else DovizKuru = 1;

    DefaultBirimMiktar = Miktar;
    DefaultBirimFiyat = NetBirimFiyat;

    if (FBirimCarpan is Null) then FBirimCarpan = 0;
    if (DefaultBirimCarpan is Null) then DefaultBirimCarpan = 0;
    if (FBirimCarpan <> 0 and DefaultBirimCarpan <> 0) then
    begin
      if (FBirimCarpan <> DefaultBirimCarpan) then
      begin
         DefaultBirimMiktar = DefaultBirimMiktar / FBirimCarpan * DefaultBirimCarpan;
         DefaultBirimFiyat = DefaultBirimFiyat / FBirimCarpan * DefaultBirimCarpan;
      end
    end
    DefaultBirimFiyatTL = DefaultBirimFiyat * DovizKuru;

    suspend;
  end
end
