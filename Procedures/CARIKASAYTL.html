CREATE PROCEDURE CARIKASAYTL
RETURNS (
    OKEY INTEGER)
AS
declare variable KayitSayisi   integer;
declare variable FirmaId       varchar(20);
declare variable CariId        integer;
declare variable KasaHareketId integer;
declare variable Tarih          TimeStamp;
begin
  Select Count(*) from Doviz where Doviz.DovizId = 'YTL' into :KayitSayisi;

  okey = 0;

  if (KayitSayisi = 1) then
  begin
    suspend;
    exit;
  end

  insert into Doviz(DovizId, Cinsi, Ulke, KdvsizFiyatYuvarlamaCarpani, KdvsizFiyatYuvarlamaTipi, BrIskYuvarlamaCarpani, BrIskYuvarlamaTipi,
                    AltIskYuvarlamaCarpani, AltIskYuvarlamaTipi, KdvYuvarlamaCarpani, KdvYuvarlamaTipi, NetFiyatYuvarlamaCarpani, NetFiyatYuvarlamaTipi)
            Select 'YTL', 'Yeni Türk Lirası', 'TR', KdvsizFiyatYuvarlamaCarpani, KdvsizFiyatYuvarlamaTipi, BrIskYuvarlamaCarpani, BrIskYuvarlamaTipi,
                    AltIskYuvarlamaCarpani, AltIskYuvarlamaTipi, KdvYuvarlamaCarpani, KdvYuvarlamaTipi, NetFiyatYuvarlamaCarpani, NetFiyatYuvarlamaTipi from Doviz where Doviz.DovizId = 'USD';


  For Select Firma .FirmaId from Firma
    left join BankaHesap on BankaHesap.FirmaId = Firma.FirmaId
    where Firma.DovizID = 'TL' and BankaHesap.FirmaId is null
    into :FirmaId do
  begin
    Update Firma Set DovizId = 'YTL' where Firma.FirmaID = :FirmaId;
  end

  Update Sistem Set SistemDovizId = 'YTL';

  For Select CariId from Cari
    left join Firma on Firma.FirmaId = Cari.FirmaId
    where Firma.DovizId = 'YTL' and CSGReferansId is null and CSHReferansId is null and
                                    CSGBReferansId is null and CSHBReferansId is null and
                                    STReferansId is null and KSReferansId is null and  BCReferansId is null and BFReferansId is null and UniteReferansId is null
    into :CariId do
  begin
    update Cari Set DovizKuru2 = DovizKuru2 / 1000000 where CariID = :CariId;
  end

  For Select KasaHareket.KasaHareketId from KasaHareket
    left join Firma on Firma.FirmaId = KasaHareket.FirmaId
    where KasaHareket.FirmaId is not null and Firma.DovizId = 'YTL'
    into :KasaHareketId do
  begin
    Update KasaHareket set DovizKuru2 = DovizKuru2 / 1000000 where KasaHareket.KasaHareketID = :KasaHareketId;
  end

  Update Sistem Set StokTabanDovizId = 'YTL';

  Update Stok Set TabanDovizKuru2 = TabanDovizKuru2 / 1000000;
  Update Irsaliye Set TabanDovizKuru2 = TabanDovizKuru2 / 1000000;
  Update Fatura Set TabanDovizKuru2 = TabanDovizKuru2 / 1000000;

  Update DovizKuru Set AlisFiati = AlisFiati / 1000000,
                       SatisFiati = SatisFiati / 1000000,
                       SerbestAlisFiati = SerbestAlisFiati / 1000000,
                       SerbestSatisFiati = SerbestSatisFiati / 1000000
    where DovizKuru.DovizId <> 'YTL' and Tarih < '01.01.2005';

  For Select Tarih from DovizKuru group By Tarih
    into :Tarih do
  begin
    insert into DovizKuru(Tarih, DovizId, AlisFiati, SatisFiati, SerbestAlisFiati, SerbestSatisFiati)
                   values(:Tarih, 'YTL', 1, 1,1,1);
  end

  Okey = 1;
  suspend;
end
