CREATE PROCEDURE PSKALITEKONTROLBITIR(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    APERSONELID INTEGER,
    ABITIR INTEGER,
    ATARTIDANGELDI INTEGER,
    APSKALITEKONTROLASAMAID INTEGER)
AS
declare variable TamamiTartildi integer;
declare variable KayitSayisi    integer;
declare variable NetKilo        Numeric(12,3);
declare variable Dara           Numeric(12,3);
declare variable NetMetre       Numeric(12,3);
declare variable KapAdedi       integer;
declare variable PartiStokTartiId integer;
declare variable PsKaliteKontrolId integer;
declare variable KS                integer;
declare variable KaliteNo          integer;
declare variable KKBitinceTartiyaAktar integer;
declare variable KKMetreleriHataPuanaGoreEksilt integer;
declare variable KullaniciId integer;
declare variable FSonKayit integer;
declare variable KKTartiTamamlanmamisAsOnemseme integer;
declare variable hstDokumaKumas integer;
declare variable hstOrmeKumas integer;
declare variable TopAktarimiVar EgeBoolean;
declare variable OlcuId integer;
declare variable Adet integer;
declare variable FHataPuaniCarpani Double Precision;
declare variable EgemenFirmaID     integer;
declare variable DKumasdaKGveMtSifirOlamaz integer;
declare variable KKOrmeKumasKiloKontrolu    EgeBoolean ;
declare variable PlanlananKilo  double precision;
declare variable NetKiloToplam  double precision;
declare variable MaximumFireOrani double precision;
begin
  Select EgemenFirmaID, DKumasdaKGveMtSifirOlamaz, KKOrmeKumasKiloKontrolu from Sistem where sistemId = 1 into :EgemenFirmaID, :DKumasdaKGveMtSifirOlamaz, :KKOrmeKumasKiloKontrolu;

/* Kalite Kontrol bitti degiginde bu procedure calısacak, 1.tartıyı olusturacak, default rafa koyacak */

  select HamTipiGetir.hstDokumaKumas, HamTipiGetir.hstOrmeKumas from HamTipiGetir into :hstDokumaKumas, :hstOrmeKumas;
  Select KKBitinceTartiyaAktar, KKMetreleriHataPuanaGoreEksilt, KKTartiTamamlanmamisAsOnemseme from Sistem where Sistemid = 1 into :KKBitinceTartiyaAktar, :KKMetreleriHataPuanaGoreEksilt, :KKTartiTamamlanmamisAsOnemseme;

  Select Max(KullaniciId) from Kullanici where Kullanici.PersonelId = :APersonelId into :KullaniciId;

  FSonKayit = 1;

  if (APSKaliteKontrolAsamaID is not null) then
    Select SonKayit from PSKaliteKontrolAsama where PSKaliteKontrolAsama.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID into :FSonKayit;

  select count(*) from PartiStok
  left join Ham on Ham.HamId = PartiStok.HamId
  where PartiStok.PartiSinifId = :APartiSinifId and Ham.HamTipiId = :hstDokumaKumas into :KS;

  if (KS = 0) then
  begin
    Select count(*) from PSKaliteKontrol
      where (:APSKaliteKontrolAsamaID is Null or PSKaliteKontrol.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID) and
            PSKaliteKontrol.PartiSinifId = :APartiSinifId and PSKaliteKontrol.PartiStokTartiId is null and
           (PSKaliteKontrol.Kilo - PSKaliteKontrol.Dara) <= 0
      into :KS;


    if (KS > 0) then exception KayitHatasi ' Top net kilo 0 olamaz...';
  end

  select count(*) from PartiStok
  left join Ham on Ham.HamId = PartiStok.HamId
  where PartiStok.PartiSinifId = :APartiSinifId and Ham.HamTipiId = :hstDokumaKumas into :KS;

  if (DKumasdaKGveMtSifirOlamaz = 1 and KS > 0) then
  begin
    Select count(*) from PSKaliteKontrol
      where (:APSKaliteKontrolAsamaID is Null or PSKaliteKontrol.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID) and
            PSKaliteKontrol.PartiSinifId = :APartiSinifId and PSKaliteKontrol.PartiStokTartiId is null and
           ((PSKaliteKontrol.Kilo - PSKaliteKontrol.Dara) <= 0 or
           (PSKaliteKontrol.metre) <= 0 )
      into :KS;

    if (KS > 0) then exception KayitHatasi ' Top net metre ve kilo 0 olamaz...';
  end

  select PartiStok.Kilo from PartiStok
    left join Ham on Ham.HamId = PartiStok.HamId
    where PartiStok.PartiSinifId = :APartiSinifId and Ham.HamTipiId = :hstOrmeKumas into :PlanlananKilo;

  if (KKOrmeKumasKiloKontrolu = 1 and :PlanlananKilo > 0) then
  begin
    Select sum(PSKaliteKontrol.kilo - PSKaliteKontrol.dara) from PSKaliteKontrol
      where (:APSKaliteKontrolAsamaID is Null or PSKaliteKontrol.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID) and
            PSKaliteKontrol.PartiSinifId = :APartiSinifId and PSKaliteKontrol.PartiStokTartiId is null and PSKaliteKontrol.OnayDurumu = 1
      into :NetKiloToplam;

   Select BoyaSiparisDetay.MaximumFireOrani from PartiStok
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      where PartiStok.PartiSinifId = :APartiSinifId into :MaximumFireOrani;

    if (NetKiloToplam > PlanlananKilo + (PlanlananKilo *(MaximumFireOrani / 100))) then exception KayitHatasi 'Parti Açılış Kilosunu Geçti ( Parti Açılış Kilosu : ' || PlanlananKilo ||' Bitirilen Sinifin Kilosu : ' || NetKiloToplam;
  end

  if (KKMetreleriHataPuanaGoreEksilt = 1 and ATartidanGeldi = 0) then
  begin

   FHataPuaniCarpani = null;
    Select BoyaSiparisDetay.HataPuaniCarpani from PartiStok
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      where PartiStok.PartiSinifId = :APartiSinifId
      into  :FHataPuaniCarpani;

    Update PSKaliteKontrol set Metre = Metre - (PaketlemeHataPuani * Coalesce(:FHataPuaniCarpani , 0.2)), KullaniciId = :KullaniciId where (:APSKaliteKontrolAsamaID is Null or PSKaliteKontrol.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID) and PSKaliteKontrol.PartiSinifId = :APartiSinifId and PSKaliteKontrol.PuanliSistem = 1 and PaketlemeHataPuani is not Null and Metre is not Null;
  end

  Select Count(*) from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and
          PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamaStok.CikisTarihi is null
    into :KayitSayisi;
  if (KayitSayisi > 0 and ATartidanGeldi = 1 and KKTartiTamamlanmamisAsOnemseme = 0) then exception YapilmamisAsamaVarTartiGiremez; /*Sonraki aşama varsa ve tartıdan girilmişse işlem yapılmasın*/

  if (KKBitinceTartiyaAktar = 1  and FSonKayit = 1 and (KKTartiTamamlanmamisAsOnemseme = 1 or (ATartidanGeldi = 0 and KayitSayisi = 0) or ATartidanGeldi = 1) ) then  /*Sonraki aşama yoksa tartıya otomatik Atalım*/
  begin

    Select TamamiTartildi from PartiStok where PartiStok.PartiSinifId = :APartiSinifId into :TamamiTartildi;
    if (TamamiTartildi = 1) then exception TamamiTartilmisKiloGiremezsiniz;

    PartiStokTartiId = null;
    TopAktarimiVar = 0;

    For Select PsKaliteKontrolId, KaliteNo, Dara, (Kilo - Dara), Metre, OlcuId, Adet from PSKaliteKontrol
      where (:APSKaliteKontrolAsamaID is Null or PSKaliteKontrol.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID) and
            PSKaliteKontrol.PartiSinifId = :APartiSinifId and PSKaliteKontrol.PartiStokTartiId is null and PSKaliteKontrol.OnayDurumu = 1
      into :PsKaliteKontrolId, :KaliteNo, :Dara, :NetKilo, :NetMetre, :OlcuId, :Adet do
    begin
      TopAktarimiVar = 1;
      KapAdedi = 1;
      PartiStokTartiId = gen_id(PartiStokTarti_Id_Gen, 1);

      Insert Into PartiStokTarti(PartiStokTartiId, PartiKayitId, PartiSinifId, Kilo, Dara, Metre, KapAdedi, KaliteNo, KaliteKontrol, InsertKullaniciId, OlcuId, Adet)
                          values(:PartiStokTartiId, :APartiKayitId, :APartiSinifId, :NetKilo, :Dara, :NetMetre, :KapAdedi, :KaliteNo, 1, :KullaniciId, :OlcuId, :Adet);

      Update PsKaliteKontrol set PartiStokTartiId = :PartiStokTartiId, LinkIslemi = 1, KullaniciId = :KullaniciId where PsKaliteKontrol.PsKaliteKontrolId = :PsKaliteKontrolId;
    end

    if (ABitir = 1) then
    begin
      if ((TopAktarimiVar = 0) and (EgemenFirmaID <> 37)) then Exception KayitHatasi 'Aktarılacak Kalite Kontrol Kaytdı Bulunamadı.';
      if (APSKaliteKontrolAsamaID is Not Null) then
        Update PSKaliteKontrolAsama set TartiyaAktarildi = 1, KKBITIRENPERSONELID = :APersonelId  where PSKaliteKontrolAsama.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID;

      Update PartiStok Set KKBitirenPersonelId = :APersonelId, KKBitisTarihi = current_TimeStamp(0), TamamiTartildi = 1 where PartiStok.PartiSinifId = :APartiSinifId;

      Execute Procedure PartiStokTartiBrutHesapla(:APartiSinifId, 0);
    end
  end
  else
  begin
    if (ABitir = 1) then
    begin
      if (APSKaliteKontrolAsamaID is Not Null) then
        Update PSKaliteKontrolAsama set KKBITIRENPERSONELID = :APersonelId  where PSKaliteKontrolAsama.PSKaliteKontrolAsamaID = :APSKaliteKontrolAsamaID;

      if (FSonKayit = 1) then
        Update PartiStok Set KKBitirenPersonelId = :APersonelId, KKBitisTarihi = current_TimeStamp(0) where PartiStok.PartiSinifId = :APartiSinifId;
    end
  end
end
