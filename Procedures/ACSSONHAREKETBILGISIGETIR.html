CREATE PROCEDURE ACSSONHAREKETBILGISIGETIR(
    CS VARCHAR(1),
    ACEKSENETID INTEGER,
    AGENELKAYITID INTEGER,
    ACSHAREKETDETAYID INTEGER)
RETURNS (
    BORDROID INTEGER,
    BORDROTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    CEKSENETID INTEGER,
    GENELKAYITID INTEGER,
    ACIKLAMA VARCHAR(40),
    VADETARIHI TIMESTAMP,
    MEBLAG DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    DOVIZKURU1 DOUBLE PRECISION,
    DOVIZKURU2 DOUBLE PRECISION,
    DOVIZMEBLAG DOUBLE PRECISION,
    CEKSENETNO VARCHAR(30),
    HESAPNO VARCHAR(40),
    ILKBORCLU VARCHAR(80),
    KESIDEYERI VARCHAR(20),
    AITOLDUGUBANKAADI VARCHAR(40),
    CEKSENETGRUP1 VARCHAR(5),
    CEKSENETGRUP2 VARCHAR(5),
    CEKSENETGRUP3 VARCHAR(5),
    HAREKETTARIHI TIMESTAMP,
    HAREKETBORDRONO INTEGER,
    HAREKETFIRMAID VARCHAR(20),
    HAREKETBANKAHESAPID VARCHAR(20),
    HAREKETAVUKATID VARCHAR(10),
    HAREKETACIKLAMA VARCHAR(40),
    HAREKETKUR1 DOUBLE PRECISION,
    HAREKETKUR2 DOUBLE PRECISION,
    KARSILIKSIZ INTEGER,
    TAHSILEDILENMEBLAG DOUBLE PRECISION,
    TAHSILEDILMIS INTEGER,
    SONBORDROTIPI INTEGER,
    HARISLEMDOVIZID VARCHAR(5),
    HARISLEMDOVIZKURU1 DOUBLE PRECISION,
    HARISLEMDOVIZKURU2 DOUBLE PRECISION)
AS
begin
  if (:ACekSenetId = -1 and :AGenelKayitId = -1) then exception CSParametreHatasi1;
  if (:ACekSenetId > -1 and :AGenelKayitId > -1) then exception CSParametreHatasi1;
  if (:Cs <> 'C' and :CS <> 'S') then exception CSParametreHatasi2;

  if (ACekSenetId = -1) then
  begin
    select CekSenet.BordroId, CekSenet.Tarih, CekSenet.FirmaId, CekSenetDetay.GenelKayitId, CekSenetId, CekSenetDetay.Aciklama,
          Vadetarihi, Meblag, TahsilEdilenMeblag, CekSenetDetay.DovizId, DovizKuru1, DovizKuru2, Banka.Adi, CekSenetno, HesapNo,
          IlkBorclu, KesideYeri, CekSenetGrup1, CekSenetGrup2, CekSenetGrup3, Karsiliksiz from CekSenetDetay
    left join CekSenet on CekSenet.CsBordroKayitId = CekSenetDetay.CsBordroKayitId
    left join banka on Banka.bankaId = CekSenetDetay.AitOlduguBankaId
    where CekSenetDetay.GenelKayitId = :AGenelKayitId and CekSenetDetay.AV = 'A' and CekSenetDetay.CS = :Cs
    into :BordroId, :BordroTarihi, :FirmaId, :GenelKayitId, :CekSenetId, :Aciklama, :Vadetarihi, :Meblag, :TahsilEdilenMeblag, :DovizId,
         :DovizKuru1, :DovizKuru2, :AitOlduguBankaAdi, :CekSenetno, :HesapNo,
         :IlkBorclu, :KesideYeri, :CekSenetGrup1, :CekSenetGrup2, :CekSenetGrup3, :Karsiliksiz;
  end
  else
  begin
    select CekSenet.BordroId, CekSenet.Tarih, CekSenet.FirmaId, CekSenetDetay.GenelKayitId, CekSenetId, CekSenetDetay.Aciklama,
          Vadetarihi, Meblag, TahsilEdilenMeblag, CekSenetDetay.DovizId, DovizKuru1, DovizKuru2, Banka.Adi, CekSenetno, HesapNo,
          IlkBorclu, KesideYeri, CekSenetGrup1, CekSenetGrup2, CekSenetGrup3, Karsiliksiz from CekSenetDetay
    left join CekSenet on CekSenet.CsBordroKayitId = CekSenetDetay.CsBordroKayitId
    left join banka on Banka.bankaId = CekSenetDetay.AitOlduguBankaId
    where CekSenetDetay.CekSenetId = :ACekSenetId and CekSenetDetay.AV = 'A' and CekSenetDetay.CS = :Cs
    into :BordroId, :BordroTarihi, :FirmaId, :GenelKayitId, :CekSenetId, :Aciklama, :Vadetarihi, :Meblag, :TahsilEdilenMeblag, :DovizId,
         :DovizKuru1, :DovizKuru2, :AitOlduguBankaAdi, :CekSenetno, :HesapNo,
         :IlkBorclu, :KesideYeri, :CekSenetGrup1, :CekSenetGrup2, :CekSenetGrup3, :Karsiliksiz;
  end
  DovizMeblag = Meblag / DovizKuru1 * DovizKuru2;


  select CekSenetHareket.tarih, CekSenetHareket.BordroId,CekSenetHareket.FirmaId, CekSenetHareket.BankaHesapId, CekSenetHareket.AvukatId, CekSenetHareketDetay.BordroTipi,
         CekSenetHareketDetay.Aciklama, CekSenetHareketDetay.DovizKuru1, CekSenetHareketDetay.DovizKuru2, CekSenetHareketDetay.IslemDovizID,CekSenetHareketDetay.IslemDovizKuru1, CekSenetHareketDetay.IslemDovizKuru2  from CekSenetHareketDetay
    left join CekSenethareket on CekSenetHareket.CshBordroKayitId = CekSenetHareketDetay.CshBordroKayitId
    where CekSenetHareketDetay.cshareketdetayid = (select max(cshareketdetayid) from CekSenetHareketDetay where CekSenetHareketDetay.genelkayitid = :genelkayitid and ((CekSenetHareketDetay.CSHareketDetayId < :ACSHareketDetayId and :AcsHareketDetayId > -1) or  :AcsHareketDetayId = -1 ))
    into :Harekettarihi, :HareketBordroNo, :HareketFirmaId, :HareketBankaHesapId, :HareketAvukatId,
         :SonBordroTipi, :HareketAciklama, :HareketKur1, :HareketKur2, :HarIslemDovizID, :HarIslemDovizKuru1, :HarIslemDovizKuru2;

  if (TahsilEdilenMeblag = 0) then TahsilEdilmis = 0;
  else if (TahsilEdilenMeblag < Meblag) then TahsilEdilmis = 1;
  else TahsilEdilmis = 2;

  if (SonBordroTipi is null) then SonBordroTipi = 0;

  if (CekSenetId is not null) then suspend;
end
