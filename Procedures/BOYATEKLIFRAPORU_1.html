CREATE PROCEDURE BOYATEKLIFRAPORU_1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ASATISTIPI INTEGER,
    AGLOBAL INTEGER,
    AMUSTERIREFERANSI_1 VARCHAR(30),
    AMUSTERIREFERANSI_2 VARCHAR(30),
    AKONTROLDURUMU INTEGER,
    AONAYDURUMU INTEGER,
    ASIPARISPROSESID INTEGER,
    AHAMID INTEGER,
    AKALITEID VARCHAR(20),
    AISLEMKAYNAGI SMALLINT)
RETURNS (
    BOYATEKLIFKAYITID INTEGER,
    BOYATEKLIFDETAYID INTEGER,
    BOYATEKLIFYIL INTEGER,
    BOYATEKLIFID INTEGER,
    BOYATEKLIFIDSTR VARCHAR(30),
    ISLEMKAYNAGI SMALLINT,
    ISLEMKAYNAGISTR VARCHAR(15),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TEKLIFTARIHI TIMESTAMP,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    LABRECETEKODU VARCHAR(20),
    GLOBALTEKLIF INTEGER,
    GLOBALTEKLIFSTR VARCHAR(6),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI VARCHAR(20),
    HESAPLANANFIYAT DOUBLE PRECISION,
    TEKLIFEDILENFIYAT DOUBLE PRECISION,
    RECETEMALIYETI DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    MUSTERIREFERANSI_1 VARCHAR(30),
    MUSTERIREFERANSI_2 VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    KONTROLDURUMU INTEGER,
    KONTROLDURUMUSTR VARCHAR(5),
    KONTROLKULLANICIID INTEGER,
    KONTROLKULLANICIADI VARCHAR(40),
    KONTROLTARIHI TIMESTAMP,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP,
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    VADEGUN INTEGER,
    ODEMESEKLIID INTEGER,
    ODEMESEKLIADI ADI40,
    TEKLIFDETAYNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    HAMSTOKBEDELI DOUBLE PRECISION,
    URETIMKARORANI DOUBLE PRECISION,
    SATISKARORANI DOUBLE PRECISION,
    TEKLIFFIREORANI DOUBLE PRECISION,
    VADEFARKIORANI DOUBLE PRECISION,
    ILAVEISLEMLERADI VARCHAR(1024),
    OZELTESTLERADI VARCHAR(1024),
    KALITEID VARCHAR(20),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable FAdi varchar(40);
begin
  for Select BoyaTeklif.BoyaTeklifKayitId, BoyaTeklifDetay.BoyaTeklifDetayId, BoyaTeklif.BoyaTeklifYil, BoyaTeklif.BoyaTeklifId, BoyaTeklif.FirmaId, Firma.Adi, BoyaTeklif.TeklifTarihi,
             BoyaTeklif.Fason, LabRecete.LabReceteKodu, BoyaTeklifDetay.GlobalTeklif, BoyaTeklifDetay.RenkDerinligiId, RenkDerinligi.Adi, BoyaTeklifDetay.HesaplananFiyat,
             BoyaTeklifDetay.TeklifEdilenFiyat, BoyaTeklifDetay.ReceteMaliyeti, BoyaTeklif.DovizId, BoyaTeklifDetay.Kilo, BoyaTeklifDetay.Metre, BoyaTeklifDetay.Adet,
             BoyaTeklif.MusteriReferansi_1, BoyaTeklif.MusteriReferansi_2, BoyaTeklifDetay.HamId, Ham.Adi, BoyaTeklif.KontrolDurumu, BoyaTeklif.KontrolKullaniciId,
             KK.Adi, BoyaTeklif.KontrolTarihi, BoyaTeklif.OnayDurumu, BoyaTeklif.OnaylayanKullaniciId, OK.Adi, BoyaTeklif.OnayTarihi,
             BoyaTeklifDetay.SiparisProsesId, SiparisProses.Adi, BoyaTeklifDetay.VadeGun, BoyaTeklifDetay.HamStokBedeli, BoyaTeklifDetay.UretimKarOrani, BoyaTeklifDetay.SatisKarOrani, BoyaTeklifDetay.KaliteID, Kalite.EnIstenen,  Kalite.GramajIstenen,
             BoyaTeklifDetay.TeklifFireOrani, BoyaTeklif.VadeFarkiOrani, BoyaTeklifDetay.Notu,
             Ik.Adi, BoyaTeklif.Inserttarihi, Sk.Adi, BoyaTeklif.DegisimTarihi, Ham.HamTipiID, BoyaTeklif.IslemKaynagi, iif(BoyaTeklif.IslemKaynagi = 0, 'Boya', 'HazÄ±r Giyim'), BoyaTeklif.OdemeSekliID, OdemeSekli.Adi from BoyaTeklif
    left join Firma On Firma.FirmaId = BoyaTeklif.FirmaId
    left join BoyaTeklifDetay On BoyaTeklifDetay.BoyaTeklifKayitId = BoyaTeklif.BoyaTeklifKayitId
    left join LabReceteAtis On LabReceteAtis.LabReceteAtisId = BoyaTeklifDetay.LabReceteAtisId
    left join LabRecete On LabRecete.LabReceteId = LabReceteAtis.LabReceteId
    left join RenkDerinligi On RenkDerinligi.RenkDerinligiId = BoyaTeklifDetay.RenkDerinligiId
    left join Ham On Ham.HamId = BoyaTeklifDetay.HamId
    left join SiparisProses On SiparisProses.SiparisProsesId = BoyaTeklifDetay.SiparisProsesId
    Left Join Kullanici OK on OK.KullaniciId = BoyaTeklif.OnaylayanKullaniciId
    Left Join Kullanici KK on KK.KullaniciId = BoyaTeklif.KontrolKullaniciId
    Left Join Kullanici IK on IK.KullaniciId = BoyaTeklif.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = BoyaTeklif.KullaniciId
    Left Join OdemeSekli on OdemeSekli.OdemeSekliID = BoyaTeklif.OdemeSekliID
    left Join Kalite on Kalite.KaliteId = BoyaTeklifDetay.KaliteId
    where BoyaTeklif.TeklifTarihi >= :ABaslangicTarihi and BoyaTeklif.TeklifTarihi <= :ABitisTarihi and
          (BoyaTeklif.FirmaId = :AFirmaId or :AFirmaId is null) and
          (BoyaTeklif.Fason = :ASatisTipi or :ASatisTipi is null) and
          (BoyaTeklifDetay.GlobalTeklif = :AGlobal or :AGlobal is null) and
          (BoyaTeklifDetay.KaliteID = :AKaliteID or :AKaliteID is null) and
          (BoyaTeklif.MusteriReferansi_1 = :AMusteriReferansi_1 or :AMusteriReferansi_1 is null) and
          (BoyaTeklif.MusteriReferansi_2 = :AMusteriReferansi_2 or :AMusteriReferansi_2 is null) and
          (BoyaTeklif.KontrolDurumu = :AKontrolDurumu or :AKontrolDurumu is null) and
          (BoyaTeklif.OnayDurumu = :AOnayDurumu or :AOnayDurumu is null) and
          (BoyaTeklifDetay.SiparisProsesId = :ASiparisProsesId or :ASiparisProsesId is null) and
          (BoyaTeklifDetay.HamId = :AHamId or :AHamId is null) and
          (BoyaTeklif.IslemKaynagi = :AIslemKaynagi or :AIslemKaynagi is null)
   into :BoyaTeklifKayitId, :BoyaTeklifDetayId, :BoyaTeklifYil, :BoyaTeklifId, :FirmaId, :FirmaAdi, :TeklifTarihi, :Fason, :LabReceteKodu, :GlobalTeklif,
        :RenkDerinligiId, :RenkDerinligiAdi, :HesaplananFiyat, :TeklifEdilenFiyat, :ReceteMaliyeti, :DovizId, :Kilo, :Metre,
        :Adet, :MusteriReferansi_1, :MusteriReferansi_2, :HamId, :HamAdi, :KontrolDurumu, :KontrolKullaniciId, :KontrolKullaniciAdi,
        :KontrolTarihi, :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi, :SiparisProsesId, :SiparisProsesAdi,
        :VadeGun, :HamStokBedeli, :UretimKarOrani, :SatisKarOrani, :KaliteId, :enistenen, :gramajistenen,  :TeklifFireOrani, :VadeFarkiOrani, :TeklifDetayNotu,
        :InsertKullaniciAdi, :Inserttarihi, :KullaniciAdi, :DegisimTarihi, :HamTipiID, :IslemKaynagi, :IslemKaynagiStr, :OdemeSekliID, :OdemeSekliAdi do
  begin
    if (Fason = 1) then FasonStr = 'Fason'; else FasonStr = '';
    if (GlobalTeklif = 1) then GlobalTeklifStr = 'Global'; else GlobalTeklifStr = '';

    BoyaTeklifIdStr = '(' || EgemenIntegerToStr(BoyaTeklifYil) || ') ' || EgemenIntegerToStr(BoyaTeklifId) || ' ';

    if (KontrolDurumu = 1) then KontrolDurumuStr = 'Kabul';
    else if (KontrolDurumu = 2) then KontrolDurumuStr = 'Ret';
    else KontrolDurumuStr = '';

    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    SiparisKilo = Null;
    SiparisMetre = Null;
    Select Sum(BoyaSiparisDetay.SiparisKilo), Sum(BoyaSiparisDetay.SiparisMetre) from BoyaSiparisDetay
    where BoyaSiparisDetay.BoyaTeklifDetayId = :BoyaTeklifDetayId
    into :SiparisKilo, :SiparisMetre;

    IlaveIslemlerAdi = '';
    OzelTestlerAdi   = '';


    For Select IlaveIslemler.Adi from BoyaTeklifIlaveIslem
      left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = BoyaTeklifIlaveIslem.IlaveIslemlerId
      where BoyaTeklifIlaveIslem.BoyaTeklifDetayId = :BoyaTeklifDetayId
      order By IlaveIslemler.Adi
      into :FAdi do
    begin
      if (Char_Length(IlaveIslemlerAdi) + Char_Length(FAdi) < 1020) then
      begin
        if (IlaveIslemlerAdi <> '') then IlaveIslemlerAdi = IlaveIslemlerAdi || ',';

        IlaveIslemlerAdi = IlaveIslemlerAdi || FAdi;
      end
    end

    For Select OzelTest.Adi from BoyaTeklifOzelTest
      left join OzelTest on OzelTest.OzelTestId = BoyaTeklifOzelTest.OzelTestId
      where BoyaTeklifOzelTest.BoyaTeklifDetayId = :BoyaTeklifDetayId
      order By OzelTest.Adi
      into :FAdi do
    begin
      if (Char_Length(OzelTestlerAdi) + Char_Length(FAdi) < 1020) then
      begin
        if (OzelTestlerAdi <> '') then OzelTestlerAdi = OzelTestlerAdi || ',';

        OzelTestlerAdi = OzelTestlerAdi || FAdi;
      end
    end

    suspend;
  end
end
