CREATE PROCEDURE RECETEISEMRIPLANLAMAGRUPKAYDET(
    ARECETEISEMRIYIL INTEGER,
    ARECETEISEMRIID INTEGER,
    AMAKINAPLANLAMAGRUPID VARCHAR(10))
AS
declare variable PartiAsamalariID integer;
begin
  For Select PartiAsamalariId from ReceteIsEmriDetay
     where ReceteIsEmriDetay.ReceteIsEmriYil = :AReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = :AReceteIsEmriId
     into :PartiAsamalariId do
  begin
    Update PartiAsamaStok Set MakinaPlanlamaGrupId = :AMakinaPlanlamaGrupId where PartiAsamaStok.PartiAsamalariID = :PartiAsamalariId;
  end
end
