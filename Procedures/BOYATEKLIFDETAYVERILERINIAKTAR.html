CREATE PROCEDURE BOYATEKLIFDETAYVERILERINIAKTAR(
    ABOYATEKLIFDETAYID INTEGER,
    AKULLANICIID INTEGER)
AS
Declare Variable  FBoyaTeklifKayitId  Integer;
Declare Variable  FNavlun             Double Precision;
Declare Variable  FUretimKarOrani     Double Precision;
Declare Variable  FSatisKarOrani      Double Precision;
Declare Variable  FKomisyon           Double Precision;
begin
  Select BoyaTeklifKayitId, Navlun, UretimKarOrani, SatisKarOrani, Komisyon From BoyaTeklifDetay Where BoyaTeklifDetayId = :ABoyaTeklifDetayId
  Into :FBoyaTeklifKayitId, :FNavlun, :FUretimKarOrani, :FSatisKarOrani, :FKomisyon;

  Update BoyaTeklifDetay Set Navlun = :FNavlun, UretimKarOrani = :FUretimKarOrani, SatisKarOrani = :FSatisKarOrani, Komisyon = :FKomisyon, KullaniciId = :AKullaniciId, DegisimTarihi = Current_TimeStamp(0)
  Where BoyaTeklifKayitId = :FBoyaTeklifKayitId And BoyaTeklifDetayId <> :ABoyaTeklifDetayId;
end
