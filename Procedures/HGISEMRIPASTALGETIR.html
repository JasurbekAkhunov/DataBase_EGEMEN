CREATE PROCEDURE HGISEMRIPASTALGETIR(
    AHGSIPARISKAYITID INTEGER,
    AHGISEMRIKAYITID INTEGER)
RETURNS (
    HGISEMRIKAYITID INTEGER,
    HGISEMRIPASTALID INTEGER,
    HGSIPARISDETAYID INTEGER,
    EN BOYAMIKTARSIZE,
    BOY BOYAMIKTARSIZE,
    ANAKUMAS EGEBOOLEAN,
    ANAKUMASSTR VARCHAR(20),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For select HGIsEmri.HGSiparisDetayId, HGIsEmriPastal.HGIsEmriPastalID, HGIsEmriPastal.HGIsEmriKayitID, HGIsEmriPastal.En, HGIsEmriPastal.Boy, HGIsEmriPastal.AnaKumas,
             HGIsEmriPastal.Notu, Ik.Adi, HGIsEmriPastal.InsertTarihi, Sk.Adi, HGIsEmriPastal.DegisimTarihi From HGIsEmri
    Left Join HGIsEmriPastal on HGIsEmriPastal.HGIsEmriKayitID = HGIsEmri.HGIsEmriKayitID
    Left Join Kullanici IK on IK.KullaniciId = HGIsEmriPastal.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = HGIsEmriPastal.KullaniciId
    where (HGIsEmri.HGSiparisKayitId = :AHGSiparisKayitId or :AHGSiparisKayitId is Null) and
          (HGIsEmriPastal.HGIsEmriKayitID = :AHGIsEmriKayitID or :AHGIsEmriKayitID is Null) and
          HGIsEmriPastal.HGIsEmriPastalID is not Null
    into :HGSiparisDetayId, :HGIsEmriPastalID, :HGIsEmriKayitID, :En, :Boy, :AnaKumas, :Notu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    if (AnaKumas = 1) then AnaKumasStr = 'Ana Kuma≈ü';
    else AnaKumasStr = 'Garni';

    suspend;
  end
end
