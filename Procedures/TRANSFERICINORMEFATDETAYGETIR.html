CREATE PROCEDURE TRANSFERICINORMEFATDETAYGETIR(
    AORMEFATURAID INTEGER)
RETURNS (
    BOYAFATURAID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMTIPIID INTEGER,
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    FATURABIRIM INTEGER,
    DOVIZID VARCHAR(5),
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZKURU DOUBLE PRECISION,
    KDVORANI DOUBLE PRECISION,
    TRANSFERKODU VARCHAR(55),
    ALISTRANSFERKODU VARCHAR(55),
    RENKLIALISTRANSFERKODU VARCHAR(55),
    SATISIADETRANSFERKODU VARCHAR(55),
    ALISIADETRANSFERKODU VARCHAR(55),
    BIRIMTRANSFERKODU VARCHAR(10),
    ODEMESEKLITRANSFERKODU VARCHAR(10),
    BIRIMFIYATTL DOUBLE PRECISION,
    TUTARTL DOUBLE PRECISION,
    TUTAR DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    VADEGUN INTEGER,
    TEVKIFAT INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    MUHGELIRHESAPKODU VARCHAR(30),
    KDVMUHASEBEHESAPKODU VARCHAR(30),
    RENKID INTEGER,
    ACIKLAMA VARCHAR(255),
    HAREKETTIPIID INTEGER,
    DOVIZKURU DOUBLE PRECISION,
    HAREKETSEKLI VARCHAR(25),
    BOYASIPARISNOSTR VARCHAR(30),
    PARTINOSTR VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    STOKHAREKETID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    SIPILAVEISLEMLER VARCHAR(255),
    FASON INTEGER,
    RENKADI VARCHAR(60),
    SIPARISIALANKULLANICIID INTEGER,
    ODEMETARIHI TIMESTAMP,
    KILO BOYAMIKTARSIZE)
AS
declare variable FaturaTarihi       TimeStamp;
declare variable SistemDovizId      varchar(5);
declare variable FHamSatisTransferKodu varchar(55);
declare variable FSatisTransferKodu varchar(55);
declare variable FFasonTransferKodu varchar(55);
declare variable FFasonOrmeTransferKodu  varchar(55);
declare variable FFasonArtiSatOrmeTransferKodu  varchar(55);
declare variable HamStokMuhSatisHesapKodu varchar(30);
declare variable HamStokMuhFasonSatisHesapKodu varchar(30);
declare variable HamStokMuhHamSatisHesapKodu varchar(30);
declare variable HamStokMuhFasonOrmeHesapKodu varchar(30);
declare variable FKdvYuvarlamaCarpani Double PRecision;
declare variable OrmeSiparisSekliID integer;

declare variable assSatis               integer;
declare variable assFasonArtiSatis      integer;
declare variable assFason               integer;
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable SiparisIslemGrubuId    integer;
declare variable FOrmeOdemeSekliTransferKodu  varchar(10);
declare variable FBoyaOdemeSekliTransferKodu  varchar(10);
declare variable TamirBoyaSiparisDetayId   integer;
declare variable TevkifatOrani1     integer;
declare variable TevkifatOrani2     integer;
declare variable OrmeSiparisTipiId  integer;
declare variable FirmaId varchar(20);
declare variable DesenId varchar(20);
declare variable Sinif  varchar(1);
declare variable ToplamKdv Double Precision;
declare variable EgemenFirmaID         integer;

declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable HamKumasMSatisTipi      varchar(20);
declare variable SekerKasarTransferKodu    varchar(55);
declare variable FSistemFasonSatisOrmeTransferKodu varchar(55);
declare variable FSistemFasonOrmeTransferKodu varchar(55);
declare variable FIslemDovizKur Double Precision;
declare variable FBoyaSiparisYil Integer;
declare variable FBoyaSiparisID Integer;
begin
/* gülle icin yazıldı */
  BoyaFaturaId = null;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  Select assSatis, assFasonArtiSatis, assFason from OrmeSiparisSekliGetir into :assSatis, :assFasonArtiSatis, :assFason;

  Select SistemDovizId, EgemenFirmaId, HamKumasMSatisTipi, FasonOrmeTransferKodu, FasonSatisOrmeTransferKodu  from Sistem where sistemId = 1
         into :SistemDovizId, :EgemenFirmaId, :HamKumasMSatisTipi, :FSistemFasonOrmeTransferKodu, FSistemFasonSatisOrmeTransferKodu;

  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim;

  for select StokHareket.HamId, Ham.Adi, StokHareket.RenkId, Ham.HamTipiId, StokHareket.FaturaMiktar, StokHareket.KdvHaricFiyat,
             OrmeFatura.FirmaId, OrmeFatura.DovizId, OrmeFatura.IslemDovizID, StokHareket.Kdv, Ham.HamSatisTransferKodu, Ham.SatisTransferKodu, Ham.FasonOrmeTransferKodu, Ham.FasonArtiSatOrmeTransferKodu,
             Ham.FasonTransferKodu, OrmeFatura.FaturaTarihi, StokHareket.VadeGun, Ham.MuhSatisHesapKodu,
             Ham.MuhHamSatisHesapKodu, Ham.MuhFasonSatisHesapKodu, Ham.MuhFasonOrmeHesapKodu, StokHareket.Tevkifat, OrmeFatura.KdvYuvarlamaCarpani,
             BoyaSiparis.OrmeSiparisSekliID, BoyaSiparis.Fason, StokHareket.OrmeSiparisDetayId, StokHareket.BoyaSiparisDetayId,
             StokHareket.BoyaSiparisBaskiId,
             BoyaSiparisDetay.IslemGrubuId, BoyaSiparisDetay.TamirBoyaSiparisDetayId, OrmeOdemeSekli.TransferKodu, BoyaOdemeSekli.TransferKodu,
             OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.FiiliSevkTarihi,
             OrmeFatura.TevkifatOrani1, OrmeFatura.TevkifatOrani2, BoyaSiparis.OrmeSiparisTipiId, StokHareket.DesenId,
             OrmeFatura.Sinif, Ham.AlisTransferKodu, Ham.RenkliAlisTransferKodu, Ham.SatisIadeTransferKodu, OrmeFatura.HareketTipi,
             Ham.AlisIadeTransferKodu, Ham.SekerKasarTransferKodu, StokHareket.Aciklama,
             iif(OrmeFatura.IslemDovizKuru1>OrmeFatura.IslemDovizKuru2,OrmeFatura.IslemDovizKuru1, OrmeFatura.IslemDovizKuru2), (OrmeFatura.IslemDovizKuru2 / OrmeFatura.IslemDovizKuru1),
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, BoyaSiparis.SiparisTarihi, StokHareket.StokHareketId, BoyaSiparisDetay.IlaveIslemler, SiparisProses.Adi, Renk.Adi, BoyaSiparis.SiparisiAlanKullaniciId, Ham.Adi2, OrmeFatura.OdemeTarihi,
             StokHareket.FatEdilecekHamStokBirimId, iif(OrmeRaf.FasonStok = 1, StokHareket.BrutMiktar, StokHareket.Miktar) from OrmeFatura
    left join StokHareket on StokHareket.OrmeFaturaId = OrmeFatura.OrmeFaturaId
    left join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = StokHareket.BoyaSiparisDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    Left Join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    Left Join Ham on Ham.HamId = StokHareket.HamId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    Left Join OdemeSekli OrmeOdemeSekli on OrmeOdemeSekli.OdemeSekliId = OrmeSiparisDetay.OdemeSekliId
    Left Join OdemeSekli BoyaOdemeSekli on BoyaOdemeSekli.OdemeSekliId = BoyaSiparisDetay.OdemeSekliId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
    Left Join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join Renk on Renk.RenkID = StokHareket.RenkId
    where OrmeFatura.OrmeFaturaId = :AOrmeFaturaId and StokHareket.HamIade = 0
    order by StokHareketId
    into :HamId, :HamAdi, :RenkId, :HamTipiId, :Miktar, :BirimFiyat, :FirmaId, :DovizId, :IslemDovizID, :KdvOrani,
         :FHamSatisTransferKodu, :FSatisTransferKodu, :FFasonOrmeTransferKodu, :FFasonArtiSatOrmeTransferKodu, :FFasonTransferKodu,
         :FaturaTarihi, :VadeGun, :HamStokMuhSatisHesapKodu, :HamStokMuhHamSatisHesapKodu, :HamStokMuhFasonSatisHesapKodu, :HamStokMuhFasonOrmeHesapKodu,
         :Tevkifat, :FKdvYuvarlamaCarpani, :OrmeSiparisSekliID, :Fason, :OrmeSiparisDetayId, :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :SiparisIslemGrubuId, :TamirBoyaSiparisDetayId,
         :FOrmeOdemeSekliTransferKodu, :FBoyaOdemeSekliTransferKodu, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi,
         :TevkifatOrani1, :TevkifatOrani2, :OrmeSiparisTipiId, :DesenId, :Sinif,
         :AlisTransferKodu, :RenkliAlisTransferKodu, :SatisIadeTransferKodu, :HareketTipiId,
         :AlisIadeTransferKodu, :SekerKasarTransferKodu, :Aciklama, :IslemDovizKuru, :FIslemDovizKur, :FBoyaSiparisYil, :FBoyaSiparisID, :SiparisTarihi, :StokHareketId,
         :SipIlaveIslemler, :SiparisProsesAdi, :RenkAdi, :SiparisiAlanKullaniciId, :HamAdi2, :OdemeTarihi, :FaturaBirim, :Kilo do
  begin
    if (FBoyaSiparisYil is not null) then BoyaSiparisNoStr = EgemenCopyStr(EgemenIntegerToStr(FBoyaSiparisYil),3,2) || '-' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
    else BoyaSiparisNoStr = '';


    if (DovizID <> SistemDovizId) then
    begin
      select DovizKuru from FirmaDovizKurlariniGetir(:FirmaId, :DovizId, :SistemDovizId, :FaturaTarihi) into :DovizKuru;
    end
    else DovizKuru = 1;

    if (Fason is Null) then Fason = 0;
/* acıkları kendi döviz cinsinden transfer ediyoruz, gülle icin böyle yapıldı, diger firmalarda acıklar transfer edilmiyor */
    if (Sinif = 'B') then
    begin
      BirimFiyatTL = BirimFiyat * DovizKuru;
    end
    else
      BirimFiyatTL = BirimFiyat;

    Tutar = BirimFiyat * Miktar;
    TutarTL = BirimFiyatTL * Miktar;
    IslemDovizToplam =  (BirimFiyat * FIslemDovizKur) * Miktar;
    TransferKodu = Null;

    if (OrmeSiparisTipiId = 5 and EgemenFirmaID <> 34) then RenkId = Null;

    if (BoyaSiparisDetayId is null and RenkId is not null) then SiparisIslemGrubuId = 1; /* depodan renk satıs yapılıyor ise renkli islem grubuna dahil olsun */

    if (BoyaSiparisBaskiId is not null or DesenId is not null) then
    begin
      if (TamirBoyaSiparisDetayId is not null) then
         Select IslemGrubuId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = :TamirBoyaSiparisDetayId into :SiparisIslemGrubuId;

      if (Fason = 0 or OrmeSiparisSekliId = assSatis) then
      begin
        TransferKodu = Null;
        MuhGelirHesapKodu = Null;

        Select iif(:FaturaBirim = 0, BaskiSatisKumasTransferKodu, BaskiSatisKumasMtTransferKodu), BaskiSKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
      end
      else
      begin
        TransferKodu = Null;
        MuhGelirHesapKodu = Null;

        Select iif(:FaturaBirim = 0, BaskiFasonKumasTransferKodu, BaskiFasonKumasMtTransferKodu), BaskiFKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
      end
      OdemeSekliTransferKodu = FBoyaOdemeSekliTransferKodu;
    end
    else if (BoyaSiparisDetayId is not null or RenkId is not null) then
    begin
      if (TamirBoyaSiparisDetayId is not null) then
         Select IslemGrubuId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = :TamirBoyaSiparisDetayId into :SiparisIslemGrubuId;

      if (Fason = 0 or OrmeSiparisSekliId = assSatis) then
      begin
        TransferKodu = Null;
        MuhGelirHesapKodu = Null;

        if (HamTipiId = hstIplik OR HamTipiId = hstElyaf or HamTipiId = hstTelef) then
          Select iif(:FaturaBirim = 0, SatisIplikTransferKodu, SatisIplikMtTransferKodu), SatisIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
        else
        begin
          Select iif(:FaturaBirim = 0, SatisKumasTransferKodu, SatisKumasMtTransferKodu), SatisKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
        end
      end
      else
      begin
        TransferKodu = Null;
        MuhGelirHesapKodu = Null;

        if (HamTipiId = hstIplik OR HamTipiId = hstElyaf or HamTipiId = hstTelef) then
          Select iif(:FaturaBirim = 0, FasonIplikTransferKodu, FasonIplikMtTransferKodu), FasonIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
        else
          Select iif(:FaturaBirim = 0, FasonKumasTransferKodu, FasonKumasMtTransferKodu), FasonKumasMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;
      end
      OdemeSekliTransferKodu = FBoyaOdemeSekliTransferKodu;
    end
    else
    begin
      OdemeSekliTransferKodu = FOrmeOdemeSekliTransferKodu;

      if (OrmeSiparisDetayID is not null and OrmeSiparisSekliID = 2) then
      begin
        TransferKodu = FFasonOrmeTransferKodu;
        MuhGelirHesapKodu = HamStokMuhFasonOrmeHesapKodu;
      end
      else
      begin
        TransferKodu = FHamSatisTransferKodu;
        MuhGelirHesapKodu = HamStokMuhHamSatisHesapKodu;
      end
    end

    BirimTransferKodu = '';

    Select TransferKodu from Birim
    where Birim.BirimID = (Select HamStokBirim.BirimId From HamStokBirim
          Where HamStokBirim.HamStokBirimId = :FaturaBirim )
    into :BirimTransferKodu;

    KdvMuhasebeHesapKodu = Null;

    Select MuhSatisHesapKodu from KdvHesapTablosu where KdvHesapTablosu.KdvOrani = :KdvOrani into :KdvMuhasebeHesapKodu;

    IndirilecekKdv = 0;
    TahsilEdilecekKdv = 0;

    if (Tevkifat = 0) then
    begin
      TahsilEdilecekKdv = TutarTL * KdvOrani / 100;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, FKdvYuvarlamaCarpani);
    end
    else
    begin
      ToplamKdv = (TutarTL * KdvOrani / 100);

      IndirilecekKdv = ToplamKdv * TevkifatOrani1 / TevkifatOrani2;
      IndirilecekKdv = Yuvarla(IndirilecekKdv, FKdvYuvarlamaCarpani);

      TahsilEdilecekKdv = ToplamKdv - IndirilecekKdv;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, FKdvYuvarlamaCarpani);
    end


    if (HareketTipiId = shtAlis) then
    begin
      if (RenkId is Null) then
        TransferKodu = AlisTransferKodu;
      else
        TransferKodu = RenkliAlisTransferKodu;
    end
    else if (HareketTipiId = shtSatisIade) then
      TransferKodu = SatisIadeTransferKodu;
    else if (HareketTipiId = shtAlisIade) then
      TransferKodu = AlisIadeTransferKodu;
    else if (HareketTipiId = shtFasonUretim) then
      TransferKodu = FFasonOrmeTransferKodu;
    else if (EgemenFirmaID = 34 and HareketTipiId = shtSatis and RenkID is not Null) then
      TransferKodu = RenkliAlisTransferKodu;

    HareketSekli = Null;

    if (EgemenFirmaID = 34) then  /* örkum */
    begin
      if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
      begin
        if (HareketTipiId = shtAlis) then
        begin
          if (RenkId is not Null) then
            HareketSekli = 'RENKLIIPLIKALISDEPOKOD';
          else
            HareketSekli = 'HAMIPLIKALISDEPOKOD';
        end
        else
        begin
          if (RenkId is not Null and (Fason = 0 or Fason is Null) ) then
            HareketSekli = 'MBOYALIIPLIK';
          else if (Fason = 0 or Fason is Null) then
            HareketSekli = 'HAMIPLIK';
          else if (Fason = 1) then
            HareketSekli = 'MFBOYALIIPLIK';
        end
      end
      else
      begin
        if (HareketTipiId = shtFasonUretim) then
        begin
          TransferKodu = FFasonOrmeTransferKodu;
          HareketSekli = 'FASONORME';
        end
        else if (Fason = 0 or OrmeSiparisSekliId = assSatis) then
        begin
          if (HareketTipiId = shtSatis or HareketTipiId = shtSatisIade) then
          begin
            HareketSekli = 'MHAMKUMAS';
            TransferKodu = HamKumasMSatisTipi || ' ' || SekerKasarTransferKodu;
          end
          else
          begin
            if (RenkId is not Null) then
              HareketSekli = 'RENKLIKUMASALISDEPOKOD';
            else
              HareketSekli = 'HAMKUMASALISDEPOKOD';

            TransferKodu = AlisTransferKodu;
          end
        end
        else if (OrmeSiparisSekliId = assFason) then
        begin
          HareketSekli = 'MFHAMKUMAS';
          TransferKodu = FSistemFasonOrmeTransferKodu;
        end
        else if (OrmeSiparisSekliId = assFasonArtiSatis) then
        begin
