CREATE PROCEDURE IPIMALATAARAVERMERAPORUGECMIS(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIPLIKARAVERMESEBEBIID INTEGER,
    AMAKINAGRUPID VARCHAR(4),
    AIPLIKMAKINAID VARCHAR(10))
RETURNS (
    IPLIKIMALATAARAVERMEID INTEGER,
    IPLIKMAKINAID VARCHAR(10),
    IPLIKARAVERMESEBEBIID INTEGER,
    IPLIKARAVERMESEBEBIADI VARCHAR(50),
    BASLADIIPLIKMAKINADURUMUID INTEGER,
    ARAVERDIRENPERSONELID INTEGER,
    ARAVERDIRENPERSONELADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITTIIPLIKMAKINADURUMUID INTEGER,
    ARAYIBITIRENPERSONELID INTEGER,
    ARAYIBITIRENPERSONELADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30))
AS
begin
  for select IplikImalataAraVerme.IplikImalataAraVermeId, MK.IplikMakinaId, IplikImalataAraVerme.IplikAraVermeSebebiId, IplikAraVermeSebebi.Adi IplikAraVermeSebebiAdi,
             IplikImalataaraverme.BasladiIplikMakinaDurumuId, IplikImalataAraVerme.AraVerdirenPersonelId, Bs.Adi AraVerdirenPersonelAdi, MK.Tarih BaslamaTarihi,
             IplikImalataAraVerme.BittiIplikMakinaDurumuId, IplikImalataAraVerme.ArayiBitirenPersonelId, bt.adi ArayiBitirenPersonelAdi, Bmk.tarih Bitistarihi from IplikMakinaDurumu MK
  left join IplikImalataaraverme on IplikImalataaraverme.BasladiIplikMakinaDurumuId = MK.IplikMakinaDurumuId
  left join IplikAraVermeSebebi on IplikAraVermeSebebi.IplikAraVermeSebebiId = IplikImalataAraVerme.IplikAraVermeSebebiId
  left join IplikMakina on IplikMakina.IplikMakinaId = MK.IplikMakinaId
  left join Personel bs on bs.PersonelId = IplikImalataAraVerme.AraVerdirenPersonelId
  left join Personel bt on bt.PersonelId = IplikImalataAraVerme.ArayiBitirenPersonelId
  left join IplikMakinaDurumu BMK on BMK.IplikMakinaDurumuId = IplikImalataAraVerme.bittiIplikMakinaDurumuId
  where MK.tarih >= :ABaslangicTarihi and MK.tarih <= :ABitisTarihi and MK.ImalataAraVerildi = 1 and
       (IplikImalataaraverme.IplikAraVermeSebebiId = :AIplikAraVermeSebebiId or :AIplikAraVermeSebebiId is null) and
       (Mk.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null) and
       (IplikMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null)
  into :IplikImalataAraVermeId, :IplikMakinaId, :IplikAraVermeSebebiId, :IplikAraVermeSebebiAdi,
       :BasladiIplikMakinaDurumuId, :AraVerdirenPersonelId, :AraVerdirenPersonelAdi, :BaslamaTarihi,
       :BittiIplikMakinaDurumuId, :ArayiBitirenPersonelId, :ArayiBitirenPersonelAdi, :BitisTarihi do
  begin
    if (BitisTarihi is not null) then GecenSure = BitisTarihi - BaslamaTarihi;
    else GecenSure = (CURRENT_DATE + CURRENT_TIME) - BaslamaTarihi;

    GecenSureStr = TarihToStr(GecenSure);
    suspend;
  end
end
