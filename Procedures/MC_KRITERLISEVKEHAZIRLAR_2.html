CREATE PROCEDURE MC_KRITERLISEVKEHAZIRLAR_2(
    AFIRMAID VARCHAR(20) CHARACTER SET WIN1254,
    AKULLANICIID INTEGER,
    AKUMASCINSIID INTEGER,
    ARECETEID VARCHAR(20) CHARACTER SET WIN1254,
    AMUSTERISIPARISNO VARCHAR(50) CHARACTER SET WIN1254)
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTITIPIID INTEGER,
    PARTIIDSTR VARCHAR(30) CHARACTER SET WIN1254,
    PARTIKAYITID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30) CHARACTER SET WIN1254,
    ISLEMTIPIADI VARCHAR(20) CHARACTER SET WIN1254,
    FASON INTEGER,
    FASONSTR VARCHAR(5) CHARACTER SET WIN1254,
    RECETEID VARCHAR(20) CHARACTER SET WIN1254,
    RECETEADI VARCHAR(40) CHARACTER SET WIN1254,
    PARTIKILO NUMERIC(12,3),
    PARTIMETRE NUMERIC(12,3),
    SEVKEDILENKILO NUMERIC(12,3),
    SEVKEDILENMETRE NUMERIC(12,3),
    SEVKEHAZIRKILO NUMERIC(12,3),
    SEVKEHAZIRMETRE NUMERIC(12,3),
    MODELNO VARCHAR(20) CHARACTER SET WIN1254)
AS
declare variable Okey     integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;
  for select Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, PS1.PartiKayitId, IslemTipi.Adi, Parti.ReceteId, Recete.Adi, Parti.ModelNo,
             Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Parti.Fason,
    (select sum(PS2.Kilo) from PartiStok PS2 where Ps2.PartiKayitId = PS1.PartiKayitId),
    (select sum(PS3.Metre) from PartiStok PS3 where Ps3.PartiKayitId = PS1.PartiKayitId),
    (sum(PS1.Kilo) - sum(PS1.BrutSevkKilo)), sum(PS1.BrutSevkKilo),
    (sum(PS1.Metre) - sum(PS1.BrutSevkMetre)), sum(PS1.BrutSevkMetre) from PartiStok PS1
    left join Parti on Parti.PartiKayitId = PS1.PartiKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PS1.BoyaSiparisDetayId
    left join Ham on Ham.HamId = PS1.HamId
    where PS1.durumu = 7 and (Parti.FirmaId = :AFirmaId) and
    (Ham.KumasCinsiId = :AKumasCinsiId or :AKumasCinsiId is null) and
    (Parti.ReceteId=:ARECETEID or :ARECETEID is null) and
    (BoyaSiparisDetay.MusteriSiparisNo=:AMUSTERISIPARISNO or :AMUSTERISIPARISNO is null)
    group by Parti.FirmaId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, PS1.PartiKayitId, IslemTipi.Adi, Parti.ReceteId, Recete.Adi, Parti.ModelNo,
             Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Parti.Fason
    order by Parti.FirmaId
    into :PartiYil, :PartiId, :PartiTipiId, :PartiKayitId, :IslemTipiAdi, :ReceteId, :ReceteAdi, :ModelNo,
         :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :Fason, :PartiKilo, :PartiMetre, :SevkeHazirKilo, :SevkEdilenKilo,
         :SevkeHazirMetre, :SevkEdilenMetre do
    begin
      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') '  || EgemenIntegerToStr(PartiId);
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';
      suspend;
    end
end
