   left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
   left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
   left join FirmaSevkAdresi on FirmaSevkAdresi.FirmaSevkAdresiId = PartiStok.FirmaSevkAdresiId
   left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
   left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
   left join TupAcikEnMayliStrGetir(BoyaSiparisDetay.TupAcikEnMayli) on 1=1
   left join Ham OrmeSiparisHam on OrmeSiparisHam.HamId = OrmeSiparisDetay.HamId
   left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
   left join BoyaSiparisKullanimAmaci on BoyaSiparisKullanimAmaci.BoyaSiparisKullanimAmaciId = BoyaSiparisDetay.BoyaSiparisKullanimAmaciId
   left join BaskiTipi on BaskiTipi.BaskiTipiID = BoyaSiparisDetay.BaskiTipiID
   left join HamStokBirim on HamStokBirim.HamStokBirimId = BoyaSiparisDetay.HamStokBirimId
   left join KumasCinsi on KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
   left join PartiStok PSBolunen on PSBolunen.PartiSinifId = PartiStok.BolunduguPartiSinifId
   left join Parti PartiBolunen on PartiBolunen.PartiKayitId = PSBolunen.PartiKayitId
   left join IslemGrubu on IslemGrubu.IslemGrubuId = BoyaSiparisDetay.IslemGrubuId
   left join DesenKenarYazisi on DesenKenarYazisi.DesenKenarYazisiId = BoyaSiparisDetay.DesenKenarYazisiId
   left join renkderinligi on RenkDerinligi.renkderinligiid = BoyaSiparisDetay.renkderinligiid
   left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
   left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
   where PartiStok.PartiKayitId = :APartiKayitId and PartiStok.PartiStokIptalKayitid is null
   order by PartiStok.PartiSinifId
   into :PartiSinifId, :BoyaSiparisDetayId, :HamAdiToplami, :HamAdi2, :HamAdi3, :HamAdi4,  :PartiSiraNo, :BobinSayisi, :Denye, :SiparisBobinSayisi, :SiparisSarjGramaji, :Kilo,
         :Metre, :KapAdedi, :Adet, :IstenenRaporBoyu, :IstenenOzelTuse, :EnIstenen,
         :GramajIstenen, :DonmeIstenen, :CekmezlikEnIstenen, :CekmezlikBoyIstenen, :YikamaHasligiIstenen,
         :SiparisPlanlamaNotu, :SiparisSiparisNotu, :MusteriSiparisNo, :ReferansSiparisNo, :SiparisKilo, :SiparisMetre, :ProsesId, :ProsesAdi, :SiparisProsesId, :SiparisProsesAdi,
         :KurutmaSekliAdi, :YikamaDerece, :YikamaHasligiTestiAdi,
         :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
         :ToleransEn, :ToleransGramaj, :ToleransEndenCekmezlik, :ToleransBoydanCekmezlik, :ToleransDonme,
         :KaliteId, :KaliteAdi, :ReferansPartiSinifId, :OrnekPartiSinifId, :SiparisKaliteId, :SiparisKaliteAdi, :SiparisKalitePartiYil, :SiparisKalitePartiId,
         :SiparisRenkAdi, :PaketlemeSekliId, :PaketlemeSekliAdi,
         :SiparisPus, :SiparisFeine, :SiparisLot, :SiparisDenye, :SipIlaveIslemler, :SiparisLabReceteKodu, :SiparisLabReceteAdi, :FAtisNoStr,
         :OzelTestler, :OzelApreler, :FirmaSevkAdresiId, :FirmaSevkAdresiAdi, :SiparisTerminTarihi, :SiparisRevizeTerminTarihi, :SiparisTerminNotu,
         :SiparisDesenId, :SiparisDesenAdi, :SiparisMelanjKodu, :SiparisHamRenkAdi, :Durumu, :OrmeSiparisSiraNo, :GecikmeNotu,
         :FasonBirimFiyati, :FasonBirimFiyatDovizId, :MaximumFireOrani, :BSDetayFireOrani, :HamId, :KullanimAmaciId, :KullanimAmaci,
         :ReferansIsEmriNo, :SipDetayTransferKodu, :BaskiTipiID, :BaskiTipiAdi,
         :OrmeSiparisPus, :OrmeSiparisFeine, :OrmeSiparisNotu, :OrmeSiparisUretimNotu, :OrmeSiparisDetayId,
         :SipDetayBobinSayisi, :BelgeHamAdi, :SiparisDovizID, :PartiDovizID, :MusteriTalimati, :HamStokBirimId, :BirimId, :OrmeSiparisHamID, :OrmeSiparisHamAdi,
         :HamKontrolNotu, :HamEnCm, :EkBelgeDosyaAdi, :SansIstenen, :ElastikiyetEnIstenen, :ElastikiyetBoyIstenen, :KumasCinsiId, :KumasCinsiAdi, :ArtikelNo, :BatchNo, :BolunenPartiIdStr,
         :SiparisIslemGrubuId, :SiparisIslemGrubuAdi, :ReferansKodu,
         :DesenIlkBaski, :VaryantIlkBaski, :OzelYerlesim, :OzelYerlesimDosyaAdi, :YanRaportVarmi,
         :KenarBeyazligi, :Konfeksiyon, :Tasma, :SiparisDesenKenarYazisi, :DesenKenarYazisiAdi,
         :TupAcikEnMayliStr, :KaliteNotu, :TanitimNotu, :TalepNotu, :ProsesNotu, :RenkDerinligiAdi, :Proxyid, :SiparisDetayGrupId,
         :SasNo, :PoNo, :SatisBirimFiyati, :SatisDovizId, :KomisyonOraniNotu, :SiparisSarjMakina, :SiparisBrutKilo, :KalinlikAdi, :IplikCinsiAdi, :Composition do
  begin
    ReferansPartiYil = null;
    ReferansPartiId = null;
    ReferansPartiIdStr = null;
    ReferansPartiHamAdiToplami = null;
    ReferansPartiSinifKilo = null;
    EskiPartiNoStr = '';

    if (FAtisNoStr is not null) then SiparisLabReceteKodu = SiparisLabReceteKodu || '.' || FAtisNoStr;

    if (ASiparisFasonBrFiyatGorebilir = 0) then FasonBirimFiyati = null;

    if (ReferansPartiSinifId is not null) then
    begin
      select Parti.PartiYil, Parti.PartiId, PartiStok.HamAdiToplami, PartiStok.Kilo from PartiStok
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
        where PartiStok.PartiSinifId = :ReferansPartiSinifId
        into :ReferansPartiYil, :ReferansPartiId, :ReferansPartiHamAdiToplami, :ReferansPartiSinifKilo;

      ReferansPartiIdStr =  '(' || EgemenIntegerToStr(ReferansPartiYil)  || ') ' || EgemenIntegerToStr(ReferansPartiId) || ' ';
    end

    OrnekPartiYil = null;
    OrnekPartiId = null;
    OrnekPartiIdStr = null;

    if (OrnekPartiSinifId is not null) then
    begin
      select Parti.PartiYil, Parti.PartiId from PartiStok
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
        where PartiStok.PartiSinifId = :OrnekPartiSinifId
        into :OrnekPartiYil, :OrnekPartiId;
      OrnekPartiIdStr =  '(' || EgemenIntegerToStr(OrnekPartiYil)  || ') ' || EgemenIntegerToStr(OrnekPartiId) || ' ';
    end

    PartiYilStr = EgemenIntegerToStr(PartiYil);
    IplikPaketlemeReferansId = EgemenCopyStr(PartiYilStr, 3, 2) || '-' || EgemenIntegerToStr(PartiId) || '-' || EgemenIntegerToStr(PartiSinifId);

    Yapilacaklar = '';
    for select PartiAsamalari.ReceteId, Asama.Adi from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join Asama on Asama.Asamaid = PartiAsamalari.Asamaid
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1
      order by PartiAsamalari.Sira
      into :FReceteId, :FAsamaAdi do
    begin
      if (FReceteID <> '') then
      begin
       if (char_length(Yapilacaklar) + char_length(FReceteId) < 1000) then
       begin
         if (Yapilacaklar <> '' ) then Yapilacaklar = Yapilacaklar || cast(', ' as varchar(2) character set UTF8);
         Yapilacaklar = cast(Yapilacaklar as blob character set utf8) || FReceteId;
       end
      end
      else
      begin
       if (char_length(Yapilacaklar) + char_length(FAsamaAdi) < 1000) then
       begin
         if (Yapilacaklar <> '' ) then Yapilacaklar = Yapilacaklar || cast(', ' as varchar(2) character set UTF8);
         Yapilacaklar = cast(Yapilacaklar as blob character set utf8) || FAsamaAdi;
       end
      end
    end

    IlaveIslemlerAdi = '';

    for select IlaveIslemler.Adi from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and Asamayadahil = 1 and IlaveIslemler.Adi is not null
      into :FIlaveIslemlerAdi do
    begin
       if (char_length(IlaveIslemlerAdi) + char_length(FAsamaAdi) < 1000) then
       begin
         if (IlaveIslemlerAdi <> '' ) then IlaveIslemlerAdi = IlaveIslemlerAdi || cast(', ' as varchar(2) character set UTF8);
         IlaveIslemlerAdi = cast(IlaveIslemlerAdi as blob character set utf8) || FIlaveIslemlerAdi;
       end
    end
    GirisIrsaliyeNotu = '';
    YuzeyKontrolKontrolNotu = '';


    for select HamStokMaster.HamStokMasterId from Partistok
      left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
      left join RafDetay on RafDetay.RafDetayid = PartiStokDetay.RafDetayId
      left join HamStokDetay on HamStokDetay.HamStokDetayid = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterid = HamStokDetay.HamStokMasterId
      where Partistok.PartiSinifID = :PartiSinifID
      group by HamStokMaster.HamStokMasterId
      into :HamStokMasterId do
    begin
      select HamStokMaster.Aciklama, YuzeyKontrolTalep.KontrolNotu from HamStokMaster
        left join YuzeyKontrolTalep on YuzeyKontrolTalep.HamStokMasterID = HamStokMaster.HamStokMasterId
        where HamStokMaster.HamStokMasterID = :HamStokMasterID
        into :FIrsaliyeNotu, :FYuzeyKontrolKontrolNotu;

      BlobSize = char_length(FIrsaliyeNotu);

      if (BlobSize > 0) then
      begin
        FSize = 1000 - char_length(GirisIrsaliyeNotu);
        if (FSize > BlobSize) then FSize = BlobSize;

        if (FSize > 0) then
        begin
          if (GirisIrsaliyeNotu <> '') then GirisIrsaliyeNotu = GirisIrsaliyeNotu || cast(', ' as varchar(2) character set UTF8);

          GirisIrsaliyeNotu = cast(GirisIrsaliyeNotu as blob character set utf8) ||left(FIrsaliyeNotu, FSize);
        end
      end

      BlobSize = char_length(FYuzeyKontrolKontrolNotu);

      if (BlobSize > 0) then
      begin
        FSize = 1000 - char_length(YuzeyKontrolKontrolNotu);
        if (FSize > BlobSize) then FSize = BlobSize;

        if (FSize > 0) then
        begin
          if (YuzeyKontrolKontrolNotu <> '') then YuzeyKontrolKontrolNotu = YuzeyKontrolKontrolNotu || cast(', ' as varchar(2) character set UTF8);

          YuzeyKontrolKontrolNotu = cast(YuzeyKontrolKontrolNotu as blob character set utf8) ||left(FYuzeyKontrolKontrolNotu, FSize);
        end
      end

    end

    if (SiparisKalitePartiId is not null) then
    begin
      SiparisKalitePartiIDStr =  '(' || EgemenIntegerToStr(SiparisKalitePartiYil)  || ') ' || EgemenIntegerToStr(SiparisKalitePartiId) || ' ';
    end
    else SiparisKalitePartiIDStr = '';


    if (IslemTipiId = 3) then
    begin
      for select HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo from PartiStok
        left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
        left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where PartiStok.PartiSinifId = :PartiSinifId
        group by HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo
        into :FEskiPartiYil, :FEskiPartiNo do
      begin
        if (char_length(EskiPartiNoStr) < 80) then
        begin
          if (EskiPartiNoStr <> '' ) then EskiPartiNoStr = EskiPartiNoStr || cast(', ' as varchar(2) character set UTF8);

          EskiPartiNoStr = EskiPartiNoStr || '(' || FEskiPartiYil || ') ' || FEskiPartiNo;
        end
      end
    end

    select KumasPaketlemeNo from PartiKumasPaketlemeNoGetir(:APartiKayitId, :PartiSinifId) into :KumasPaketlemeNo;

    NetSevkKilo  = 0;
    BrutSevkKilo = 0;
    SevkKapAdedi = 0;

    select sum(BrutKilo), sum(Kilo), sum(KapAdedi) from BoyaIrsaliyeDetay
      where BoyaIrsaliyeDetay.PartiSinifID = :PartiSinifID
      into :BrutSevkKilo, :NetSevkKilo, :SevkKapAdedi;

    NetSarfKilo  = 0;
    BrutSarfKilo = 0;

    select sum(BrutKilo), sum(Kilo) from HamStokSarfFisiDetay
      where HamStokSarfFisiDetay.PartiSinifID = :PartiSinifID
      into :BrutSarfKilo, :NetSarfKilo;

    suspend;
  end
