CREATE PROCEDURE TARTIIPLIKURETIMISEMRILISTESI(
    AIPLIKMAKINAID VARCHAR(10),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    IPLIKURETIMISEMRIKAYITID INTEGER,
    IPLIKURETIMISEMRIYIL INTEGER,
    IPLIKURETIMISEMRIID INTEGER,
    IPLIKURETIMISEMRIIDSTR VARCHAR(30),
    PLANLAMATARIHI TIMESTAMP,
    IPLIKMAKINAID VARCHAR(10),
    IPLIKMAKINAADI VARCHAR(40),
    IPLIKLOTID INTEGER,
    LOT VARCHAR(30),
    BUKUMTIPIID INTEGER,
    BUKUMTIPIADI VARCHAR(10),
    KALINLIKNE INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    HAMADI VARCHAR(255),
    PLANLANANMIKTAR NUMERIC(12,3),
    GERCEKLESENMIKTAR NUMERIC(12,3),
    BASLAMATARIHI TIMESTAMP)
AS
begin
  For Select IplikUretimIsEmri.IplikUretimIsEmriKayitId, IplikUretimIsEmri.IplikUretimIsEmriYil, IplikUretimIsEmri.IplikUretimIsEmriId,
             IplikUretimIsEmri.PlanlamaTarihi, IplikUretimIsEmri.IplikMakinaId, IplikMakina.Adi, IplikUretimIsEmri.IplikLotId,
             IplikLot.Lot, IplikLot.BukumTipiId, BukumTipi.Adi, IplikLot.KalinlikNe, IplikCinsi.Adi,
             Ham.Adi, IplikUretimIsEmri.PlanlananMiktar, IplikUretimIsEmri.GerceklesenMiktar, IplikUretimIsEmri.BaslamaTarihi from IplikUretimIsEmri
    left join IplikMakina on IplikMakina.IplikMakinaID = IplikUretimIsEmri.IplikMakinaId
    left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotId
    left join Ham on Ham.HamID = IplikLot.HamId
    left join IplikCinsi on IplikCinsi.IplikCinsiID = IplikLot.IplikCinsiId
    left join BukumTipi on BukumTipi.BukumTipiID = IplikLot.BukumTipiId
    where IplikUretimIsEmri.PlanlamaTarihi >= :ABaslangicTarihi and IplikUretimIsEmri.PlanlamaTarihi <= :ABitisTarihi and
          (IplikUretimIsEmri.IplikMakinaID = :AIplikMakinaID or :AIplikMakinaID is null)
    into :IplikUretimIsEmriKayitId, :IplikUretimIsEmriYil, :IplikUretimIsEmriId, :PlanlamaTarihi, :IplikMakinaId, :IplikMakinaAdi,
         :IplikLotId, :Lot, :BukumTipiId, :BukumTipiAdi, :KalinlikNe, :IplikCinsiAdi, :HamAdi,  :PlanlananMiktar, :GerceklesenMiktar,  :BaslamaTarihi do
  begin
    IplikUretimIsEmriIdStr = '(' || EgemenIntegerToStr(IplikUretimIsEmriYil) || ') ' || EgemenIntegerToStr(IplikUretimIsEmriId) || ' ';
    suspend;
  end
end
