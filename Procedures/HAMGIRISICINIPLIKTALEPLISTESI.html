CREATE PROCEDURE HAMGIRISICINIPLIKTALEPLISTESI(
    AHAMID INTEGER,
    AIPLIKTALEPYIL INTEGER,
    AIPLIKTALEPID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    IPLIKTALEPKAYITID INTEGER,
    IPLIKTALEPDETAYID INTEGER,
    IPLIKTALEPYIL INTEGER,
    IPLIKTALEPID INTEGER,
    IPLIKTALEPIDSTR VARCHAR(30),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    TALEPMIKTARI DOUBLE PRECISION,
    TESLIMEDILENMIKTAR DOUBLE PRECISION,
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    LOT VARCHAR(30))
AS
begin

  if (AIplikTalepID is null) then
  begin
    for Select IplikTalep.IplikTalepKayitId, IplikTalepDetay.IplikTalepDetayId, IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, IplikTalepDetay.BoyaSiparisDetayId,
               IplikTalepDetay.HamId, Ham.Adi, IplikTalepDetay.Miktar, IplikTalepDetay.MelanjKodu, IplikTalepDetay.FlamKodu, IplikTalepDetay.Lot,
               IplikTalepDetay.TeslimEdilenMiktar FROM IplikTalep
      left join IplikTalepDetay on IplikTalepDetay.IplikTalepKayitId = IplikTalep.IplikTalepKayitId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
      LEFT JOIN Ham ON Ham.HamId = IplikTalepDetay.HamId
      WHERE IplikTalep.TalepTarihi >= :ABaslangicTarihi  and IplikTalep.TalepTarihi <= :ABitisTarihi and
            IplikTalepDetay.IplikTalepDetayId is not null and
            IplikTalepDetay.HamId = :AHamId and IplikTalepDetay.Kapali = 0 and IplikTalep.OnayDurumu = 1
     into :IplikTalepKayitId, :IplikTalepDetayId, :IplikTalepYil, :IplikTalepId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
          :HamId, :HamAdi, :TalepMiktari, :MelanjKodu, :FlamKodu, :Lot, :TeslimEdilenMiktar do
    begin

      IplikTalepIdStr = '';
      if (IplikTalepYil is not null) then IplikTalepIdStr = '(' || Format_Float('#0', IplikTalepYil) || ') '  || Format_Float('#0', IplikTalepId)  || ' ';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
      suspend;
    end
  end
  else
  begin
    for Select IplikTalep.IplikTalepKayitId, IplikTalepDetay.IplikTalepDetayId, IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, IplikTalepDetay.BoyaSiparisDetayId,
               IplikTalepDetay.HamId, Ham.Adi, IplikTalepDetay.Miktar, IplikTalepDetay.MelanjKodu, IplikTalepDetay.FlamKodu, IplikTalepDetay.Lot,
               IplikTalepDetay.TeslimEdilenMiktar FROM IplikTalep
      left join IplikTalepDetay on IplikTalepDetay.IplikTalepKayitId = IplikTalep.IplikTalepKayitId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
      LEFT JOIN Ham ON Ham.HamId = IplikTalepDetay.HamId
      WHERE IplikTalep.IpliKTalepYil = :AIpliKTalepYil and IplikTalep.IpliKTalepId = :AIpliKTalepId and
            IplikTalepDetay.IplikTalepDetayId is not null and
            IplikTalepDetay.HamId = :AHamId and IplikTalepDetay.Kapali = 0 and IplikTalep.OnayDurumu = 1
     into :IplikTalepKayitId, :IplikTalepDetayId, :IplikTalepYil, :IplikTalepId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
          :HamId, :HamAdi, :TalepMiktari, :MelanjKodu, :FlamKodu, :Lot, :TeslimEdilenMiktar do
    begin

      IplikTalepIdStr = '';
      if (IplikTalepYil is not null) then IplikTalepIdStr = '(' || Format_Float('#0', IplikTalepYil) || ') '  || Format_Float('#0', IplikTalepId)  || ' ';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
      suspend;
    end
  end
end
