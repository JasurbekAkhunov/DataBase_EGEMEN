CREATE PROCEDURE LOGGETIR(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATIP CHAR(1),
    ATABLOADI VARCHAR(32))
RETURNS (
    LOGMASTERID INTEGER,
    TIP CHAR(1),
    TABLOADI VARCHAR(32),
    KEYVALUE VARCHAR(20),
    ACIKLAMA VARCHAR(1000),
    ANAHTAR VARCHAR(1000),
    TARIH TIMESTAMP,
    KULLANICIID INTEGER,
    KULLANICIADI ADI40)
AS
begin
  for select LogMasterId, Tip, TabloAdi, KeyValue, Left(Aciklama, 1000), Left(Anahtar, 1000), Tarih, Kullanici.KullaniciId, Kullanici.Adi
    from LogMaster
    left join Kullanici on Kullanici.KullaniciID = LogMaster.KullaniciId
    where LogMaster.Tarih >= :ABaslangicTarihi and LogMaster.Tarih <= :ABitisTarihi and
    (LogMaster.Tip = :ATip or :ATip is null) and
    (LogMaster.TabloAdi = :ATabloAdi or :ATabloAdi is null)
    order by Tarih desc
    into :LogMasterId, :Tip, :TabloAdi, :KeyValue, :Aciklama, :Anahtar, :Tarih, :KullaniciId, :KullaniciAdi do
  begin
    suspend;
  end
end
