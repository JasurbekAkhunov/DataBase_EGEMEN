CREATE PROCEDURE ORMEFISSIL(
    AORMEFISID INTEGER)
AS
declare variable FOrmeIrsaliyeId integer;
declare variable FOrmeFaturaId   integer;
declare variable FSarfFisiId     integer;
begin
  Select OrmeIrsaliyeId, OrmeFaturaId from OrmeFis Where OrmeFis.OrmeFisId = :AOrmeFisID into :FOrmeIrsaliyeId, :FOrmeFaturaId;
  if (FOrmeIrsaliyeId is not null) then exception FisIrsaliyelestirilmisSilme;
  if (FOrmeIrsaliyeId is not null) then exception FisFaturalastirilmisSilme;

  Select OrmeFis.SarfFisiId from OrmeFis where OrmeFis.OrmeFisId = :AOrmeFisId into :FSarfFisiId;

  Delete from StokHareket where StokHareket.OrmeFisId = :AOrmeFisId;
  Delete from OrmeFis where OrmeFis.OrmeFisId = :AOrmeFisId;

  Delete from StokHareket where StokHareket.OrmeFisId = :FSarfFisiId;
  Delete from OrmeFis where OrmeFis.OrmeFisId = :FSarfFisiId;
end
