CREATE PROCEDURE BOYAIRSALIYELISTESI_PARTIDETAY2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASATISTIPI INTEGER,
    ARAPORTIPI INTEGER,
    AHAMID INTEGER,
    AFIRMAID VARCHAR(20),
    AFIRMAYETKILIID INTEGER,
    ASEVKFIRMAID VARCHAR(20),
    ASIPARISFIRMAID VARCHAR(20),
    ARECETEID VARCHAR(20),
    AIRSALIYENO IRSALIYENOSIZE,
    AFIYATONAY INTEGER,
    AUNITEID INTEGER,
    APARTIGRUPID VARCHAR(4),
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AISLEMTIPIID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AKALINLIKID INTEGER,
    AIPLIKCINSIID INTEGER,
    AOZELLIKID INTEGER,
    AKUMASCINSIID INTEGER,
    ALYCRATIPIID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    AYURTDISI INTEGER,
    AMAMULSEVK INTEGER,
    AHAMSATIS INTEGER,
    AHAMIADE INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ASINIF VARCHAR(1),
    ABASLANGICPARTIMIKTAR NUMERIC(12,3),
    ABITISPARTIMIKTAR NUMERIC(12,3),
    AORMEOLMAYANLAR INTEGER,
    AIPLIKSIPARISI INTEGER,
    ABOYALIIPLIKSIPARISI INTEGER,
    ALYCRASIPARISI INTEGER,
    AHAMKUMASSIPARISI INTEGER,
    ABOYALIKUMASSIPARISI INTEGER,
    AIPLIKBOYALIKUMASSIPARISI INTEGER,
    AIMALATTIPIID INTEGER,
    AKALITEID VARCHAR(20),
    AARACID VARCHAR(11),
    ADESENID VARCHAR(20),
    ADESENVARYANTID INTEGER,
    AHAREKETTIPIID INTEGER,
    ARAPORALANKULLANICIID INTEGER,
    APARTIKAYITID INTEGER,
    ATEMSILCIID INTEGER,
    ARECETETURUID INTEGER,
    AMUSTERIKALITENO VARCHAR(20),
    ABASKITIPIID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER)
AS
declare ABoyaSiparisKayitID integer;
declare itTamir integer;
declare  AstIplik               integer;
declare  astBoyaliIplik         integer;
declare  astLycra               integer;
declare  astHamKumas            integer;
declare  astBoyaliKumas         integer;
declare  astIplikBoyaliKumas    integer;
declare  hstIplik        integer;
declare  hstOrmeKumas    integer;
declare  hstDokumaKumas  integer;
declare  hstRibana       integer;
declare  hstAksesuar     integer;
declare  hstElyaf        integer;
declare  hstLycra        integer;
declare  hstTelef        integer;
begin
  Select astIplik, astBoyaliIplik, AstLycra, AstHamKumas, AstBoyaliKumas, AstIplikBoyaliKumas from OrmeSiparisTipiGetir into :astIplik, :astBoyaliIplik, :AstLycra, :AstHamKumas, :AstBoyaliKumas, :AstIplikBoyaliKumas;
  Select itTamir from IslemTipiGetir into :itTamir;
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstLycra, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstLycra, :hstTelef;

  if (ABoyaSiparisYil is not Null) then
  begin
    Select BoyaSiparis.BoyaSiparisKayitId from BoyaSiparis
      where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId into :ABoyaSiparisKayitId;
    if (ABoyaSiparisKayitId is null) then ABoyaSiparisKayitId = -1;
  end

  for select PartiStok.BoyaSiparisDetayId from BoyaIrsaliye
    left join Sofor On Sofor.SoforId = BoyaIrsaliye.SoforId
    left join Arac On Arac.AracId = BoyaIrsaliye.AracId
    left join Unite GU on GU.UniteId = BoyaIrsaliye.UniteId
    left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = BoyaIrsaliye.IrsaliyeGrupId
    left join SevkFirma On SevkFirma.SevkFirmaId = BoyaIrsaliye.SevkFirmaId
    left join BoyaIrsaliyeDetay On BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
    left join Firma SHFirma On SHFirma.FirmaId = BoyaIrsaliye.FirmaId
    left join Parti On Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
    left join Recete On Recete.ReceteId = Parti.ReceteId
    left join LabRecete RLabRecete On RLabRecete.LabReceteId = Recete.LabReceteId
    left join ReceteTuru On ReceteTuru.ReceteTuruId = Recete.ReceteTuruId
    left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = Recete.LabRenkGrupId
    left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu On IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
    left join Firma SipFirma On SipFirma.FirmaId = BoyaSiparis.FirmaId
    left join PartiStok On PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
    left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join GecikmeTipi On GecikmeTipi.GecikmeTipiId = PartiStok.GecikmeTipiId
    left join LabRecete SipLabRecete On SipLabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join RenkDerinligi RRenkDerinligi On RRenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
    left join RenkDerinligi SRenkDerinligi On SRenkDerinligi.RenkDerinligiId = BoyasiparisDetay.RenkDerinligiId
    left join Ham On Ham.HamId = BoyaIrsaliyeDetay.HamId
    left join Kalinlik On Kalinlik.KalinlikId = Ham.KalinlikId
    left join IplikCinsi On IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join HamRenk On HamRenk.HamRenkId = Ham.HamRenkId
    left join Ozellik On Ozellik.OzellikId = Ham.OzellikId
    left join LycraTipi On LycraTipi.LycraTipiId = Ham.LycraTipiId
    left join KumasCinsi On KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
    left join Kullanici FOK On FOK.KullaniciId = PartiStok.FiyatOnaylayanKullaniciId
    left join BoyaFatura On BoyaFatura.BoyaFaturaId = BoyaIrsaliyeDetay.BoyaFaturaId
    left join OrmeSiparisDetay On OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
    left join KaliteAciklamaTipi On KaliteAciklamaTipi.KaliteAciklamaTipiId = BoyaIrsaliyeDetay.KaliteAciklamaTipiId
    left join Renk OrmeRenk On OrmeRenk.RenkId = OrmeSiparisDetay.RenkId
    left join PaketlemeSekli On PaketlemeSekli.PaketlemeSekliId = BoyaSiparisDetay.PaketlemeSekliId
    left join SiparisProses On SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join SiparisGrup SiparisGrup3 on SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
    left join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
    left join HamIadeRezerv on HamIadeRezerv.HamIadeRezervid = BoyaIrsaliyeDetay.HamIadeRezervId
    left join HamStokIadeSebebi on HamStokIadeSebebi.HamStokIadeSebebiid = HamIadeRezerv.HamStokIadeSebebiid
    Left Join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join Desen on Desen.DesenId = DesenVaryant.DesenId
    left join DesenGrup on DesenGrup.DesenGrupId = Desen.DesenGrupId
    left join Kalite On Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
    left join HamGrup HG1 on HG1.HamGrupId = Ham.HamGrup1
    left join HamGrup HG2 on HG2.HamGrupId = Ham.HamGrup2
    left join HamGrup HG3 on HG3.HamGrupId = Ham.HamGrup3
    left join HamGrup HG4 on HG4.HamGrupId = Ham.HamGrup4
    left join BaskiTipi on BaskiTipi.BaskiTipiId = BoyaSiparisDetay.BaskiTipiId

    Left Join BoyaSiparisKullanimAmaci on BoyaSiparisKullanimAmaci.BoyaSiparisKullanimAmaciId = BoyaSiparisDetay.BoyaSiparisKullanimAmaciId
    left join Proses on Proses.ProsesId = PartiStok.ProsesId
    left join FirmaGrup FG1 on Fg1.FirmaGrupId = SHFirma.FirmaGrup1
    left join FirmaGrup FG2 on Fg2.FirmaGrupId = SHFirma.FirmaGrup2
    left join FirmaGrup FG3 on Fg3.FirmaGrupId = SHFirma.FirmaGrup3
    left join BoyaRezervFisiDetay on BoyaRezervFisiDetay.BoyaRezervFisiDetayId = BoyaIrsaliyeDetay.BoyaRezervFisiDetayId
    left join BoyaSiparisDetay RS on RS.BoyaSiparisDetayId = BoyaRezervFisiDetay.BoyaSiparisDetayId
    left join MiktarTipi On MiktarTipi.MiktarTipiId = Coalesce(RS.FaturaEdilecekMiktarTipi, BoyaSiparisDetay.FaturaEdilecekMiktarTipi)
    left join HamStokBirim on HamStokBirim.HamStokBirimID = BoyaSiparisDetay.HamStokBirimID
    where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
          (BoyaIrsaliye.Sinif = :ASinif or :ASinif is null) and
          (SHFirma.AnaHesapFirmaId = :AFirmaId or :AFirmaID is null) and
          (BoyaIrsaliye.UniteId = :AUniteId or :AUniteId is null) and
          (BoyaIrsaliye.SevkFirmaId = :ASevkFirmaId or :ASevkFirmaId is null) and
          BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId is not null and
          ( :ASatisTipi is null or
            (BoyaIrsaliyeDetay.PartiSinifID is not null and Parti.Fason = :ASatisTipi) or
            (BoyaIrsaliyeDetay.PartiSinifID is null and BoyaIrsaliyeDetay.FasonHamIade = :ASatisTipi)) and
          (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is null) and
          (Parti.ReceteId like :AReceteId or :AReceteId is null) and
          (Parti.BoyaSiparisKayitID = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          (:AIslemTipiId is null or (Parti.IslemTipiId = :AIslemTipiId) or (:AIslemTipiId = -1 and Parti.IslemTipiId <> :itTamir)) and
          (:AIslemGrubuId is null or Parti.IslemGrubuId = :AIslemGrubuId) and
          (SipFirma.AnaHesapFirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
          (BoyaSiparis.SiparisiAlanKullaniciId = :ASiparisiAlanKullaniciId or :ASiparisiAlanKullaniciId is null) and
          (BoyaSiparis.YurtDisi = :AYurtDisi or :AYurtDisi is null) and
          (BoyaSiparis.FirmaYetkiliId = :AFirmaYetkiliId or :AFirmaYetkiliId is null) and
          (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
          (BoyaSiparis.SiparisGrup3Id = :ASiparisGrup3Id or :ASiparisGrup3Id is null) and
          (BoyaSiparis.SiparisGrup4Id = :ASiparisGrup4Id or :ASiparisGrup4Id is null) and
          (Coalesce(RS.MusteriSiparisNo, BoyaSiparisDetay.MusteriSiparisNo) = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
          (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is null) and
          (BoyaSiparisDetay.MusteriKaliteNo = :AMusteriKaliteNo or :AMusteriKaliteNo is null) and
          ( (:AFiyatOnay = 0 and PartiStok.FiyatOnayiIstensin = 1 and PartiStok.FiyatOnaylayanKullaniciId is null) or
            (:AFiyatOnay = 1 and (PartiStok.FiyatOnayiIstensin = 0 or PartiStok.FiyatOnaylayanKullaniciId is not null) or :AFiyatOnay is null)) and
          ( Ham.HamId = :AHamId or :AHamId is null) and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporTipi = 0) or
            (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporTipi = 1) or
            ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or PartiStok.HamTipiId = :hstAksesuar) and :ARaporTipi = 2) or
            (Ham.HamTipiId = :hstDokumaKumas and :ARaporTipi = 3) or :ARaporTipi is null) and
          ( (BoyaIrsaliyeDetay.PartiSinifID is not null and :AMamulSevk = 1) or
            (BoyaIrsaliyeDetay.HamIadeRezervId is not null and :AHamIade = 1) or
            (BoyaIrsaliyeDetay.PartiSinifID is null and BoyaIrsaliyeDetay.HamIadeRezervId is null and :AHamSatis = 1)) and
          (Ham.KalinlikId = :AKalinlikId or :AKalinlikId is null) and
          (Ham.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is null) and
          (Ham.KumasCinsiId = :AKumasCinsiId or :AKumasCinsiId is null) and
          (Ham.OzellikId = :AOzellikId or :AOzellikId is null) and
          (Ham.LycraTipiId = :ALycraTipiId or :ALycraTipiId is null) and
          (((BoyaIrsaliyeDetay.PartiSinifId is not null) and (PartiStok.Miktar >= :ABaslangicPartiMiktar and PartiStok.Miktar <= :ABitisPartiMiktar)) or
          ((BoyaIrsaliyeDetay.PartiSinifId is null) and (BoyaIrsaliyeDetay.Kilo >= :ABaslangicPartiMiktar and BoyaIrsaliyeDetay.Kilo <= :ABitisPartiMiktar))) and
          ((:AHamGrup1Aralikli = 1 and HamGrup1 >= :ABaslangicGrup1Id  and (HamGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AHamGrup1Aralikli = 0) or (:AHamGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
          ((:AHamGrup2Aralikli = 1 and HamGrup2 >= :ABaslangicGrup2Id and (HamGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AHamGrup2Aralikli = 0) or (:AHamGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
          ((:AHamGrup3Aralikli = 1 and HamGrup3 >= :ABaslangicGrup3Id and (HamGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AHamGrup3Aralikli = 0) or (:AHamGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
          ((:AHamGrup4Aralikli = 1 and HamGrup4 >= :ABaslangicGrup4Id and (HamGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AHamGrup4Aralikli = 0) or (:AHamGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = '')) and
            (BoyaSiparis.OrmeSiparisTipiId is not Null or :AOrmeOlmayanlar = 1) and
            (BoyaSiparis.OrmeSiparisTipiId is Null or BoyaSiparis.OrmeSiparisTipiId <> :astIplik or :AIplikSiparisi = 1) and
            (BoyaSiparis.OrmeSiparisTipiId is Null or BoyaSiparis.OrmeSiparisTipiId <> :astBoyaliIplik or :ABoyaliIplikSiparisi = 1) and
            (BoyaSiparis.OrmeSiparisTipiId is Null or BoyaSiparis.OrmeSiparisTipiId <> :AstLycra or :ALycraSiparisi = 1) and
            (BoyaSiparis.OrmeSiparisTipiId is Null or BoyaSiparis.OrmeSiparisTipiId <> :AstHamKumas or :AHamKumasSiparisi = 1) and
            (BoyaSiparis.OrmeSiparisTipiId is Null or BoyaSiparis.OrmeSiparisTipiId <> :AstBoyaliKumas or :ABoyaliKumasSiparisi = 1) and
            (BoyaSiparis.OrmeSiparisTipiId is Null or BoyaSiparis.OrmeSiparisTipiId <> :AstIplikBoyaliKumas or :AIplikBoyaliKumasSiparisi = 1) and
          (BoyaSiparisDetay.KaliteID = :AKaliteId or :AKaliteId is Null) and
          (BoyaSiparisDetay.BaskiTipiId = :ABaskiTipiId or :ABaskiTipiId is Null) and
          (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is null) and
        (BoyaIrsaliye.AracID = :AAracID or :AAracID is null) and
        (Parti.DesenVaryantID = :ADesenVaryantID or :ADesenVaryantID is null) and
        (DesenVaryant.DesenID = :ADesenID or :ADesenID is null) and
        (BoyaIrsaliye.HareketTipiID = :AHareketTipiID or :AHareketTipiID is null) and
        (Parti.PartiKayitId = :APartiKayitId or :APartiKayitId is null) and

        (SHFirma.TemsilciId = :ATemsilciId or :ATemsilciId is null) and
        (ReceteTuru.ReceteTuruId = :AReceteTuruId or :AReceteTuruId is null)
    group by PartiStok.BoyaSiparisDetayId into :BoyaSiparisDetayId do suspend;
end
