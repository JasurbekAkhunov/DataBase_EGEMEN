CREATE PROCEDURE ONAYLANMISHAMSTOKLISTESI(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AONAYLAYANKULLANICIID INTEGER)
RETURNS (
    HAMSTOKIRSALIYEID INTEGER,
    HAMSTOKMASTERID INTEGER,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SINIF VARCHAR(1),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30),
    HAMADI VARCHAR(255),
    PUS INTEGER,
    FEINE INTEGER,
    ORMECIMAKINANO VARCHAR(10),
    EN INTEGER,
    GRM2 INTEGER,
    RENK VARCHAR(20),
    TUPACIKENMAYLISTR VARCHAR(10),
    LOT VARCHAR(150),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ADET INTEGER,
    ACIKLAMA BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TAMIR INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    ESKIPARTINO INTEGER,
    MUSTERIPARTINO VARCHAR(30),
    NEMORANI DOUBLE PRECISION,
    YIKAMAONCESIEN DOUBLE PRECISION,
    YIKAMASONRASIEN DOUBLE PRECISION,
    YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    YIKAMASONRASIGRAMAJ DOUBLE PRECISION,
    ENDENCEKMEZLIK DOUBLE PRECISION,
    BOYDANCEKMEZLIK DOUBLE PRECISION,
    HAMDONME DOUBLE PRECISION,
    YIKAMAONCESIRAPORBOYU VARCHAR(100),
    YIKAMASONRASIRAPORBOYU VARCHAR(100),
    YIKAMAONCESICPI VARCHAR(10),
    YIKAMASONRASICPI VARCHAR(10),
    YIKAMAONCESIWPI VARCHAR(10),
    YIKAMASONRASIWPI VARCHAR(10),
    HAMKONTROLIPLIKUZUNLUGU1 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU2 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU3 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU4 VARCHAR(40),
    BOYAMAKABILIYETI VARCHAR(10),
    YIKAMAHASLIGI VARCHAR(10),
    NUMARATIPIADI VARCHAR(10),
    NUMARA VARCHAR(10),
    BUKUM VARCHAR(10),
    MUKAVEMET VARCHAR(10),
    UZAMA VARCHAR(10),
    UVKONTROL INTEGER,
    UVKONTROLSTR VARCHAR(5),
    KONTROLKULLANICIADI VARCHAR(40),
    KONTROLTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable TupAcikEnMayli   integer;
declare variable hstIplik        integer;
declare variable hstElyaf       integer;
declare variable hstTelef       integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  for select HamStokMaster.HamStokIrsaliyeId, HamStokMaster.Fason, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, Ormeci.Adi,
             Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.OrmeciMakinaNo, HamStokMaster.En, HamStokMaster.Grm2,  HamStokMaster.Renk,
             HamStokMaster.Lot, HamStokMaster.TupAcikEnMayli, HamStokMaster.KapAdedi, HamStokMaster.Kilo, HamStokMaster.Metre, HamStokMaster.Adet, HamStokMaster.Aciklama,
             HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartino, HamStokMaster.MusteriPartiNo,
             HamStokMaster.OnaylayanKullaniciId, OK.Adi, HamStokMaster.OnayTarihi,
             HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiGramaj, HamStokMaster.EndenCekmezlik,
             HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
             HamStokMaster.YikamaOncesiCPI, HamStokMaster.YikamaSonrasiCPI, HamStokMaster.YikamaOncesiWPI, HamStokMaster.YikamaSonrasiWPI,
             HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2, HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4,
             HamStokMaster.BoyamaKabiliyeti, HamStokMaster.YikamaHasligi, NumaraTipi.Adi, HamStokMaster.Numara, HamStokMaster.Bukum, HamStokMaster.Mukavemet, HamStokMaster.NemOrani, HamStokMaster.Uzama, HamStokMaster.UVKontrol,
             KK.Adi, HamStokMaster.Kontroltarihi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, VerilenSiparisDetay.BoyaSiparisDetayId,
             IK.Adi, HamStokMaster.InsertTarihi, Sk.Adi, HamStokMaster.DegisimTarihi from HamStokMaster
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
  left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = HamStokMaster.VerilenSiparisDetayId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = VerilenSiparisDetay.BoyaSiparisKayitId
  left join Ham on Ham.HamId = HamStokMaster.HamId
  left join NumaraTipi on NumaraTipi.NumaraTipiId = HamStokMaster.NumaraTipiId
  left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
  left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  left join Firma on Firma.Firmaid = HamstokIrsaliye.Firmaid
  left join Ormeci on Ormeci.Ormeciid = HamstokIrsaliye.Ormeciid
  Left Join Kullanici OK on OK.KullaniciId = HamStokMaster.OnaylayanKullaniciId
  Left Join Kullanici KK on KK.KullaniciId = HamStokMaster.KontrolKullaniciId
  Left Join Kullanici IK on IK.KullaniciId = HamStokMaster.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HamStokMaster.KullaniciId
  where HamStokMaster.OnayTarihi >= :ABaslangicTarihi and HamStokMaster.OnayTarihi <= :ABitisTarihi and HamStokMaster.OnayDurumu = 1 and
       ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporTipi = 0) or ( Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporTipi = 1) ) and
       (HamStokMaster.OnaylayanKullaniciId = :AOnaylayanKullaniciId or :AOnaylayanKullaniciId is null)
  into :HamStokIrsaliyeId, :Fason, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :OrmeciAdi,
       :HamAdi, :Pus, :Feine, :OrmeciMakinaNo, :En, :Grm2, :Renk, :Lot, :TupAcikEnMayli, :KapAdedi,  :Kilo, :Metre,
       :Adet, :Aciklama, :Tamir, :TamirTipiAdi, :EskiPartino, :MusteriPartiNo,
       :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi,
       :YikamaOncesiEn, :YikamaSonrasiEn, :YikamaOncesiGramaj, :YikamaSonrasiGramaj, :EndenCekmezlik, :BoydanCekmezlik, :HamDonme,
       :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu,
       :YikamaOncesiCPI, :YikamaSonrasiCPI, :YikamaOncesiWPI, :YikamaSonrasiWPI, :HamKontrolIplikUzunlugu1, :HamKontrolIplikUzunlugu2,
       :HamKontrolIplikUzunlugu3, :HamKontrolIplikUzunlugu4,
       :BoyamaKabiliyeti, :YikamaHasligi, :NumaraTipiAdi, :Numara, :Bukum, :Mukavemet, :NemOrani, :Uzama, :UVKontrol,
       :KontrolKullaniciAdi, :Kontroltarihi, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    Select TupAcikEnMayliStrGetir.tupacikenmaylistr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;


    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (UVKontrol = 1) then UVKontrolStr = 'Uygun';
    else UVKontrolStr = '';

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';

    suspend;
  end
end
