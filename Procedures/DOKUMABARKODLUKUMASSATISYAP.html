CREATE PROCEDURE DOKUMABARKODLUKUMASSATISYAP(
    AKALITEKONTROLID INTEGER,
    AIRSALIYEID INTEGER,
    ADOKUMAPLANLAMAID INTEGER,
    AORMESIPARISDETAYID INTEGER)
AS
declare variable FIrsaliyeId           integer;
declare variable FUIrsaliyeId           integer;
declare variable DokumaPlanlamaId        integer;
declare variable KKID                  integer;
declare variable BoyaSiparisKayitId    integer;
declare variable OrmeSiparisDetayId    integer;
declare variable HamId                 integer;
declare variable StokHareketId         integer;
declare variable OrmeRafDetayId        integer;
declare variable RafId                 varchar(10);
declare variable OrmeRafId             integer;
declare variable Miktar                Numeric(12,3);
declare variable KapAdedi              Integer;
declare variable Metre                 Numeric(12,3);
declare variable Kilo                  BoyaMiktarSize;
declare variable SatisFiyati           Double Precision;
declare variable UretimPartiNo         varchar(30);
declare variable KayitSayisi           integer;
declare variable KayitStokHareketId    integer;
declare variable SiparisDovizId        varchar(5);
declare variable IrsaliyeDovizId       varchar(5);
declare variable IrsaliyeTarihi        TimeStamp;
declare variable DovizKuru             Double Precision;
declare variable SatisStokHareketId    integer;
declare variable UretimeCikisStokHareketId    integer;
declare variable Sakat                 integer;
declare variable FHamId                integer; /* baska siparise satiyor ise kullaniyoruz */
declare variable FBaskaSipariseSatilabilir integer; /* baska siparise satiyor ise kullaniyoruz */
declare variable FIptal                integer; /* baska siparise satiyor ise kullaniyoruz */
declare variable FLot                  varchar(150);
declare variable FCozguLotu            varchar(30);
declare variable FAtkiLotu             varchar(30);
declare variable FDepoId               varchar(10);
declare variable FAyarTopu             integer;
declare variable FSarfOrmeFisId        integer;
declare variable Adet                  integer;
begin
  /*Boya'da boyle bir secenek yok */
  FBaskaSipariseSatilabilir = 1;
  if (ADokumaPlanlamaId = 0) then ADokumaPlanlamaId = Null;

  Select DokumaKaliteKontrol.DokumaKaliteKontrolId, DokumaKaliteKontrol.OrmeIrsaliyeId, DokumaKaliteKontrol.DokumaPlanlamaId, DokumaPlanlama.BoyaSiparisKayitId, DokumaPlanlama.OrmeSiparisDetayId,
         OrmeSiparisDetay.HamId, DokumaKaliteKontrol.RafId, DokumaKaliteKontrol.Miktar, DokumaKaliteKontrol.KapAdedi, DokumaKaliteKontrol.StokHareketId,
         DokumaKaliteKontrol.OrmeRafDetayId, OrmeSiparisDetay.DovizId, DokumaKaliteKontrol.SatisStokHareketId, DokumaKaliteKontrol.Sakat,
         OrmeSiparisDetay.Iptal, StokHareket.Lot, StokHareket.CozguLotu, StokHareket.AtkiLotu, DokumaKaliteKontrol.UretimeCikisStokHareketId, DokumaKaliteKontrol.OrmeUCikisIrsaliyeId,
         DokumaKaliteKontrol.AyarTopu, DokumaKaliteKontrol.Kilo, DokumaKaliteKontrol.SarfOrmeFisId, DokumaKaliteKontrol.OrmeRafId, DokumaKaliteKontrol.Adet from DokumaKaliteKontrol
  left join StokHareket on StokHareket.StokHareketID = DokumaKaliteKontrol.StokHareketId
  left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaKaliteKontrol.DokumaPlanlamaId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
  left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
  where DokumaKaliteKontrol.DokumaKaliteKontrolId = :AKaliteKontrolId
  into :KKId, :FIrsaliyeId, :DokumaPlanlamaId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :HamId, :RafId,
       :Miktar, :KapAdedi, :StokHareketId, :OrmeRafDetayId, :SiparisDovizId, :SatisStokHareketId, :Sakat,
       :FIptal, :FLot, :FCozguLotu, :FAtkiLotu, :UretimeCikisStokHareketId, :FUIrsaliyeId, :FAyarTopu, :Kilo, :FSarfOrmeFisId, :OrmeRafId, :Adet;

  KapAdedi = 1;
  if (ADokumaPlanlamaId is not null) then
  if (ADokumaPlanlamaId <> DokumaPlanlamaId) then exception OkutulanUretimPartiNoyaAitDegil;

  if (KKId is not null) then
  begin
     OrmeRafId=0;
    Select ormerafdetay.ormerafid from ormerafdetay  where   ormerafdetayid = :OrmeRafDetayId into :OrmeRafId;
  end

  if (KKId is null) then exception KaliteKontrolIdKaydiBulunamadi;

  if (AIrsaliyeId is Not null and FIrsaliyeId is not null) then exception BuUretiminCikisiZatenYapilmis  '(Satis)';
  if (AIrsaliyeId is Not null and FUIrsaliyeId is not null) then exception BuUretiminUCikisiZatenYapilmis  '(Uretime Cikis)';
  if (AIrsaliyeId is Not null and FSarfOrmeFisId is Not null) then exception BuUretiminUCikisiZatenYapilmis  '(Sarf)';

  if (OrmeRafId is null) then
  begin
    if (OrmeRafDetayId is not null) then
    begin
      if (Sakat = 0) then Select UrunRafId from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, Null, null, :StokHareketId, Null) where OrmeRafDetayId = :OrmeRafDetayId and AyarTopu = :FAyarTopu into :OrmeRafId;
      else Select UrunRafId from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, Null, Null, Null, Null) where Sakat = 1 and RafID = :RafId and AyarTopu = :FAyarTopu into :OrmeRafId;
    end
    else
    begin
      if (Sakat = 0) then Select Max(UrunRafId) from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, Null, Null, :StokHareketId, Null) where AyarTopu = :FAyarTopu into :OrmeRafId;
      else Select max(UrunRafId) from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, Null, Null, Null, Null) where Sakat = 1  and AyarTopu = :FAyarTopu into :OrmeRafId;
    end
  end

  if (AOrmeSiparisDetayId is not null) then
  begin
    if (FBaskaSipariseSatilabilir = 0 and FIptal = 0) then exception KKUretimBaskaSipariseSatilamaz;

    Select BoyaSiparisKayitId, HamId from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId into :BoyaSiparisKayitId, :FHamId;

    if (FHamId is null) then exception SiparisKayitliDegil;
    if (HamId <> FHamId) then exception OkutulanUretimSipariseUymuyor;
    OrmeSiparisDetayId = AOrmeSiparisDetayId;
  end

  if (AIrsaliyeId is not null)  then
  begin
    Select StokHareketId from StokHareket
      where StokHareket.OrmeIrsaliyeId = :AIrsaliyeId and StokHareket.OrmeRafId = :OrmeRafId and StokHareket.OrmeFisId is null and
           (StrKarsilastir(StokHareket.Lot, :FLot) = 1) and
           (IntKarsilastir(StokHareket.OrmeSiparisDetayId, :OrmeSiparisDetayId) = 1) and
           DokumaHareketi = 1
      into :KayitStokHareketId;

    Select SatisFiyati from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayID = :OrmeSiparisDetayId into :SatisFiyati;

    if (KayitStokHareketId is not Null) then
    begin
      Update StokHareket Set Adet = Adet + :Adet, Metre = Metre + :Miktar, BrutMiktar = BrutMiktar + :Kilo, Kilo = Kilo + :Kilo, KapAdedi = KapAdedi + :KapAdedi, NeredenUpdate = 1 where StokHareket.StokHareketId = :KayitStokHareketId;
    end
    else
    begin
      Select DovizId, IrsaliyeTarihi from OrmeIrsaliye where OrmeIrsaliyeId = :AIrsaliyeId into :IrsaliyeDovizId, :IrsaliyeTarihi;

      if (SiparisDovizId <> IrsaliyeDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:SiparisDovizId, :IrsaliyeDovizId, :IrsaliyeTarihi) into :DovizKuru;
      else DovizKuru = 1;

      if (DovizKuru is null or DovizKuru = 0) then exception DovizKuruGirisiYapilmamis;
      SatisFiyati = SatisFiyati * DovizKuru;

      KayitStokHareketId = gen_id(StokHareket_Id_Gen, 1);

      insert into StokHareket(StokHareketId, HareketTipi, OrmeIrsaliyeId, NeredenUpdate, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, OrmeRafId, Metre, Kilo, BrutMiktar, Adet, KapAdedi, BirimFiyat, Lot, CozguLotu, AtkiLotu, DokumaHareketi)
                    values(:KayitStokHareketId, 2, :AIrsaliyeId, 1, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :HamId, :OrmeRafId, :Miktar, :Kilo, :Kilo, :Adet, :KapAdedi, :SatisFiyati, :FLot, :FCozguLotu, :FAtkiLotu, 1);

    end
    Update DokumaKaliteKontrol Set OrmeIrsaliyeID = :AIrsaliyeId, SatisStokHareketId = :KayitStokHareketId, LinkIslemi = 1 where DokumaKaliteKontrol.DokumaKaliteKontrolId = :AKaliteKontrolId;
  end
  else
  begin
    if (SatisStokHareketId is Null) then exception BuUretimSatisiZatenYapilmamis;

    Select count(*) from DokumaKaliteKontrol where DokumaKaliteKontrol.SatisStokHareketId = :SatisStokHareketId into :KayitSayisi;
    if (KayitSayisi = 1) then delete from StokHareket where StokHareket.StokHareketId = :SatisStokHareketId;
    else
    begin
      Update StokHareket Set Adet = Adet - :Adet, Metre = Metre - :Miktar, Kilo = Kilo - :Kilo, BrutMiktar = BrutMiktar - :Kilo, KapAdedi = KapAdedi - :KapAdedi, NeredenUpdate = 1 where StokHareket.StokHareketId = :SatisStokHareketId;
      Update DokumaKaliteKontrol Set OrmeIrsaliyeID = Null, SatisStokHareketId = Null, LinkIslemi = 1 where DokumaKaliteKontrol.DokumaKaliteKontrolId = :AKaliteKontrolId;
    end
  end
end
