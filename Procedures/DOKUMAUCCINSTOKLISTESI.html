CREATE PROCEDURE DOKUMAUCCINSTOKLISTESI(
    ADOKUMAPLANLAMAID INTEGER,
    ADEPOID VARCHAR(10),
    ARAFID VARCHAR(10))
RETURNS (
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    TALIMATKULLANILACAKMIKTAR DOUBLE PRECISION,
    TALIMATCIKISMIKTARI DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    RAFID VARCHAR(10),
    KALANMIKTAR NUMERIC(12,3),
    TALIMATLOT VARCHAR(150),
    TALIMATURETICIADI VARCHAR(40),
    KALITENO INTEGER,
    LOT VARCHAR(150),
    ATKILOTU VARCHAR(30),
    URETICIADI VARCHAR(40),
    URETIMPARTINO VARCHAR(30),
    LEVENTNO INTEGER,
    DEPOPARTINO VARCHAR(30),
    TELADEDI INTEGER,
    COZGUISEMRIKAYITID INTEGER,
    COZGUISEMRIID VARCHAR(15),
    HASILISEMRIKAYITID INTEGER,
    HASILISEMRIID VARCHAR(15),
    LEVENTTIPIID INTEGER,
    BARKODID VARCHAR(30))
AS
declare variable SiparisFirmaId varchar(20);
declare variable StokFirmaId varchar(20);
declare variable DokumaFirmaKodumuz varchar(20);
declare variable TahminiFireOrani Double Precision;
declare variable PlanlananMiktar Double Precision;
declare variable DokumaPlansizStokCikisiEngelle EgeBoolean;
declare variable KayitSayisi Integer;

begin
  Select DokumaPlanlama.PlanlananMiktar, BoyaSiparis.FirmaId from DokumaPlanlama
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
  where DokumaPlanlama.DokumaPlanlamaId = :ADokumaPlanlamaId
  into :PlanlananMiktar, :SiparisFirmaId;

  Select DokumaFirmaKodumuz, DokumaPlansizStokCikisiEngelle from Sistem where SistemId = 1 into :DokumaFirmaKodumuz, :DokumaPlansizStokCikisiEngelle;

  For Select DokumaPlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi, Ham.Adi, DokumaPlanlamaRecete.RenkId, Renk.Adi,
             DokumaPlanlamaRecete.Fason, DokumaPlanlamaRecete.Lot, Uretici.Adi,
             Sum(:PlanlananMiktar * DokumaPlanlamaRecete.KullanimOrani / 100), Sum(OrmeSiparisDetay.TahminiFireOrani),
             Sum(DokumaPlanlamaRecete.CikisMiktari - DokumaPlanlamaRecete.IadeMiktari) from DokumaPlanlamaRecete
    Left Join Ham On Ham.HamId = DokumaPlanlamaRecete.HamId
    Left Join HamTipi On HamTipi.HamTipiId = Ham.HamTipiId
    left join Renk on Renk.RenkId = DokumaPlanlamaRecete.RenkId
    left join Uretici on Uretici.UreticiId = DokumaPlanlamaRecete.UreticiId
    left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaPlanlamaRecete.DokumaPlanlamaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    where DokumaPlanlamaRecete.DokumaPlanlamaId = :ADokumaPlanlamaId
    Group By DokumaPlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi, Ham.Adi, DokumaPlanlamaRecete.RenkId, Renk.Adi,
             DokumaPlanlamaRecete.Fason, DokumaPlanlamaRecete.Lot, Uretici.Adi
  into  :HamId, :HamTipiId, :HamTipiAdi, :HamAdi, :RenkId, :RenkAdi, :Fason, :TalimatLot, :TalimatUreticiAdi,
        :TalimatKullanilacakMiktar, :TahminiFireOrani, :TalimatCikisMiktari do
  begin
    if (HamTipiID = 0) then TalimatKullanilacakMiktar = TalimatKullanilacakMiktar / ((100 - TahminiFireOrani) / 100);

    TalimatKullanilacakMiktar = Yuvarla(TalimatKullanilacakMiktar, 100);

    if (Fason = 1) then
    begin
      FasonStr = 'Fason';
      StokFirmaID = SiparisFirmaId;
    end
    else
    begin
      FasonStr = '';
      StokFirmaID = DokumaFirmaKodumuz;
    end

    if (TalimatLot = '') then TalimatLot = Null;

    For Select OrmeRaf.OrmeRafId, OrmeRaf.RafId, OrmeRaf.StokGirisNo, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
               OrmeRaf.Lot, OrmeRaf.KaliteNo, Uretici.Adi, UretimPartiNo, OrmeRaf.LeventNo, OrmeRaf.CozguIsemriKayitID,
               CozguIsemri.IsEmriID, OrmeRaf.HasilIsemriKayitID, HasilIsemri.IsEmriID, OrmeRaf.DepoPartiNo, OrmeRaf.TelAdedi,
               OrmeRaf.AtkiLotu, OrmeRaf.LeventTipiId, OrmeRaf.BarkodId from OrmeRaf
      left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      left join CozguIsemri on CozguIsemri.CozguIsemriKayitID = OrmeRaf.CozguIsemriKayitID
      left join HasilIsemri on HasilIsemri.HasilIsemriKayitID = OrmeRaf.HasilIsemriKayitID
      where OrmeRaf.HamID = :HamId and OrmeRaf.StokFirmaId = :StokFirmaID and OrmeRaf.StokVar = 1 and
           IntKarsilastir(OrmeRaf.RenkID, :RenkId) = 1 and OrmeRaf.FasonStok = :Fason and
           (OrmeRaf.Lot = :TalimatLot or :TalimatLot is null) and
           (DepoRaf.DepoId = :ADepoID or :ADepoID is null) and
           (OrmeRaf.RafID = :ARafID or :ARafID is null)
      into :OrmeRafId, :RafId, :StokGirisNo, :KalanMiktar, :Lot, :KaliteNo, :UreticiAdi, :UretimPartiNo, :LeventNo,
           :CozguIsEmriKayitID, :CozguIsemriID, :HasilIsemriKayitID, :HasilIsEmriID, :DepoPartiNo, :TelAdedi, :AtkiLotu,
           :LeventTipiId, :BarkodId do
    begin
      if (DokumaPlansizStokCikisiEngelle = 0) then
      begin
        suspend;
      end
      else
      begin
        if (CozguIsEmriKayitID is not null) then
        begin
          select count(*) from DokumaStokPlanlama where DokumaStokPlanlama.OrmeRafId = :OrmeRafId into :KayitSayisi;
          if (KayitSayisi > 0) then
          begin
            suspend;
          end
        end
      end
    end
  end
end
