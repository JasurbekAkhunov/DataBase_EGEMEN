CREATE PROCEDURE IPLIKARIZABAKIMKAYDET(
    AIPLIKMAKINAID VARCHAR(10),
    APERSONELID INTEGER,
    AGELDI INTEGER)
RETURNS (
    STATUS INTEGER,
    MESAJ VARCHAR(255))
AS
declare variable GeldiIplikMakinaDurumuId   integer;
declare variable GittiIplikMakinaDurumuId   integer;
declare variable IplikArizaIhbarId          integer;
declare variable IplikArizaBakimId          integer;
declare variable KayitSayisi           integer;

begin
  if (AGeldi = 1) then
  begin
    Select IplikArizaIhbarId from IplikArizaIhbar
      where IplikArizaIhbar.IplikMakinaId = :AIplikMakinaId and IplikArizaIhbar.BittiIplikMakinaDurumuId is null into :IplikArizaIhbarId;

    if (IplikArizaIhbarId is not null) then
    begin
      Select Count(*) from IplikArizaBakim where IplikArizaBakim.IplikMakinaId = :AIplikMakinaId and IplikArizaBakim.PersonelId = :APersonelId and IplikArizaBakim.GittiIplikMakinaDurumuId is null into :KayitSayisi;

      if (kayitSayisi = 0) then
      begin
        GeldiIplikMakinaDurumuId = gen_Id(IplikMakinaDurumu_id_gen, 1);

        insert into IplikMakinaDurumu(IplikMakinaDurumuId, IplikMakinaId, Tarih, ServisciGeldi, ServisciVar) values(:GeldiIplikMakinaDurumuId, :AIplikMakinaId, Current_TimeStamp(0), 1, 1); /* bakımcı geldi  */

        insert into IplikArizaBakim(IplikMakinaId, GeldiIplikMakinaDurumuId, IplikArizaIhbarId, PersonelId) values(:AIplikMakinaId, :GeldiIplikMakinaDurumuId, :IplikArizaIhbarId, :APersonelId);
        Status = 1;
        Mesaj = 'Kayıt yapıldı';
      end
      else
      begin
        Status = 2;
        Mesaj = 'Bakımcı Zaten Var';
      end
    end
    else
    begin
      Status = 3;
      Mesaj = 'Ariza Yok';
    end
  end
  else if (AGeldi = 2) then
  begin
    Select IplikArizaBakimId from IplikArizaBakim where IplikArizaBakim.GittiIplikMakinaDurumuId is null and IplikArizaBakim.PersonelId = :APersonelId and IplikArizaBakim.IplikMakinaId = :AIplikMakinaId into :IplikArizaBakimId;

    if (IplikArizaBakimId is not null) then
    begin
      GittiIplikMakinaDurumuId = gen_Id(IplikMakinaDurumu_id_gen, 1);
/* bu bakımcıdan baska bakımcı varmı bak, servisci(bakımcı)var degiskenine ne koyacagına karar ver.. */
      Select Count(*) from IplikArizaBakim where IplikArizaBakim.GittiIplikMakinaDurumuId is null and IplikArizaBakim.IplikMakinaId = :AIplikMakinaId and IplikArizaBakim.PersonelId <> :APersonelId into :KayitSayisi;

      if (KayitSayisi = 0) then  insert into IplikMakinaDurumu(IplikMakinaDurumuId, IplikMakinaId, Tarih, ServisciGeldi, ServisciVar) values(:GittiIplikMakinaDurumuId, :AIplikMakinaId, Current_TimeStamp(0), 2, 0); /* bakımcı gitti  */
      else insert into IplikMakinaDurumu(IplikMakinaDurumuId, IplikMakinaId, Tarih, ServisciGeldi, ServisciVar) values(:GittiIplikMakinaDurumuId, :AIplikMakinaId, Current_TimeStamp(0), 2, 1); /* bakımcı gitti  */

      Update IplikArizaBakim Set GittiIplikMakinaDurumuId = :GittiIplikMakinaDurumuId where IplikArizaBakim.IplikArizaBakimID = :IplikArizaBakimId;
      Status = 1;
      Mesaj = 'Kayıt Yapıldı..';
    end
    else
    begin
      Status = 4;
      Mesaj = 'Bakımcı Yok';
    end
  end
  else
  begin
    Status = 5;
    mesaj = 'Parametre Hatalı..Egemen Yazılıma Haber ver.. ';
  end
  suspend;
end
