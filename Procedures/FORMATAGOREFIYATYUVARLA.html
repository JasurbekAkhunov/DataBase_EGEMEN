CREATE PROCEDURE FORMATAGOREFIYATYUVARLA(
    AFIYAT DOUBLE PRECISION,
    ATABLEADI VARCHAR(40),
    AFIELDNAME VARCHAR(40))
RETURNS (
    FIYAT DOUBLE PRECISION)
AS
declare variable Ondalik integer;
declare variable YuvarlamaCarpani Double Precision;
declare variable I integer;
begin
  Select Ondalik from FieldFormat
    left join ModulFormat on ModulFormat.ModulFormatId = FieldFormat.ModulFormatId
    where FieldFormat.TableAdi = :ATableAdi and FieldFormat.FieldAdi = :AFieldName
    into :Ondalik;

  I = 1;
  YuvarlamaCarpani = 1.0;

  if (Ondalik is null) then Ondalik = 6;

  if (Ondalik is not null) then
  begin
    While (I <= Ondalik) do
    begin
      YuvarlamaCarpani = YuvarlamaCarpani * 10.0;
      I = I + 1;
    end
    Fiyat = Yuvarla(AFiyat, YuvarlamaCarpani);
  end
  else Fiyat = AFiyat;
  suspend;
end
