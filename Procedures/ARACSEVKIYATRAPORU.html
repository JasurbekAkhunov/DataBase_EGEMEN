CREATE PROCEDURE ARACSEVKIYATRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AARACID VARCHAR(11))
RETURNS (
    HAREKETTIPISTR VARCHAR(30),
    ARACID VARCHAR(11),
    IRSALIYETARIHI TIMESTAMP,
    SEVKSEFERNO INTEGER,
    SOFORADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ALICIUNITEADI VARCHAR(30),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    TOPLAMKILO NUMERIC(12,3),
    TOPLAMBRUTKILO NUMERIC(12,3))
AS
declare variable HareketTipiId integer;
declare variable FasonHamIade  integer;
declare variable HamIade  integer;
declare variable Fason  integer;

declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable ShtKaydirmaCikis integer;
declare variable ShtKaydirmaGiris integer;
declare variable ShtSarf integer;
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, ShtKaydirmaCikis, ShtKaydirmaGiris, ShtSarf from OrmeStokHareketTipiGetir
   into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
        :shtFasonUretim, :ShtUretim, :ShtKaydirmaCikis, :ShtKaydirmaGiris, :ShtSarf;

  For Select BoyaIrsaliye.HareketTipiId, BoyaIrsaliye.AracId, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.SevkSeferNo, Sofor.Adi, BoyaIrsaliye.FirmaId, Firma.Adi,
             BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliyeDetay.FasonHamIade, Sum(BoyaIrsaliyeDetay.Kilo), Sum(BoyaIrsaliyeDetay.BrutKilo) from BoyaIrsaliye
  left join Firma on Firma.FirmaId = BoyaIrsaliye.FirmaId
  Left Join Sofor On Sofor.SoforId = BoyaIrsaliye.SoforId
  left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
  where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
        (BoyaIrsaliye.AracId = :AAracId or :AAracId is null) and BoyaIrsaliyeDetay.BoyaIrsaliyeID is not null
  group by BoyaIrsaliye.HareketTipiId, BoyaIrsaliye.AracId, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.SevkSeferNo, Sofor.Adi, BoyaIrsaliye.FirmaId, Firma.Adi,
           BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliyeDetay.FasonHamIade
  into :HareketTipiId, :AracId, :IrsaliyeTarihi, :SevkSeferNo, :SoforAdi, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :FasonHamIade,
       :ToplamKilo, :ToplamBrutKilo do
  begin
    if (HareketTipiId = 0) then
    begin
      if (FasonHamIade = 0) then HareketTipiStr = 'Satış';
      else HareketTipiStr = 'Fason Ham İade';
    end
    else if (HareketTipiId = 1) then HareketTipiStr = 'Alış İade';
    else HareketTipiStr = 'Hatalı...!!!';
    suspend;
  end

  For Select HamStokIrsaliye.HamStokHareketTipiId, HamStokIrsaliye.AracId, HamStokIrsaliye.IrsaliyeTarihi, Sofor.Adi, HamStokIrsaliye.FirmaId, Firma.Adi,
             HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.Fason, Sum(HamStokMaster.Kilo) from HamStokIrsaliye
  left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
  Left Join Sofor On Sofor.SoforId = HamStokIrsaliye.SoforId
  left join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
  where HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
        (HamStokIrsaliye.AracId = :AAracId or :AAracId is null) and HamStokMaster.HamStokIrsaliyeId is not null
  group by HamStokIrsaliye.HamStokHareketTipiId, HamStokIrsaliye.AracId, HamStokIrsaliye.IrsaliyeTarihi, Sofor.Adi, HamStokIrsaliye.FirmaId, Firma.Adi,
           HamStokIrsaliye.IrsaliyeSeriNo, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.Fason
  into :HareketTipiId, :AracId, :IrsaliyeTarihi, :SoforAdi, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :Fason,
       :ToplamKilo do
  begin
    SevkSeferNo = Null;
    ToplamBrutKilo = ToplamKilo;

    if (Fason = 0) then
    begin
      if (HareketTipiId = 0) then HareketTipiStr = 'Alış';
      else if (HareketTipiId = 1) then HareketTipiStr = 'Satış İade';
      else HareketTipiStr = 'Hatalı...!!!';
    end
    else HareketTipiStr = 'Fason Giriş';

    suspend;
  end

  For Select OrmeIrsaliye.HareketTipi, OrmeIrsaliye.AracId, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.SevkSeferNo, Sofor.Adi, OrmeIrsaliye.FirmaId, Firma.Adi,
             OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, StokHareket.HamIade, Sum(StokHareket.Miktar), Sum(StokHareket.BrutMiktar) from OrmeIrsaliye
  left join Firma on Firma.FirmaId = OrmeIrsaliye.FirmaId
  Left Join Sofor On Sofor.SoforId = OrmeIrsaliye.SoforId
  left join StokHareket on StokHareket.OrmeIrsaliyeId = OrmeIrsaliye.OrmeIrsaliyeId
  where OrmeIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and OrmeIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
        (OrmeIrsaliye.AracId = :AAracId or :AAracId is null) and /*OrmeIrsaliye.AracID is not null and M.Erdem Kaldirtti*/ StokHareket.OrmeIrsaliyeId is not null
  group by OrmeIrsaliye.HareketTipi, OrmeIrsaliye.AracId, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.SevkSeferNo, Sofor.Adi, OrmeIrsaliye.FirmaId, Firma.Adi,
           OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, StokHareket.HamIade
  into :HareketTipiId, :AracId, :IrsaliyeTarihi, :SevkSeferNo, :SoforAdi, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :HamIade,
       :ToplamKilo, :ToplamBrutKilo do
  begin
    if (HareketTipiId = shtAlis) then HareketTipiStr = 'Alış';
    else if (HareketTipiId = shtSatis and HamIade = 0) then HareketTipiStr = 'Satış';
    else if (HareketTipiId = shtSatis and HamIade = 1) then HareketTipiStr = 'Fason Hammadde İade';
    else if (HareketTipiId = shtAlisIade) then HareketTipiStr = 'Alış İade';
    else if (HareketTipiId = shtSatisIade) then HareketTipiStr = 'Satış İade';
    else if (HareketTipiId = shtFasonAlis) then HareketTipiStr = 'Fason İşlenmek Üzere Giriş';
    else if (HareketTipiId = shtFasonUretimeCikis) then HareketTipiStr = 'Fason Üretime Çıkış';
    else if (HareketTipiId = shtFasonUretim and HamIade = 0) then HareketTipiStr = 'Fason Üretim';
    else if (HareketTipiId = shtFasonUretim and HamIade = 1) then HareketTipiStr = 'Fason Üretim İade';
    else HareketTipiStr = 'HATALI !!!';
    suspend;
  end


  for Select UniteIrsaliye.AracId, UniteIrsaliye.IrsaliyeTarihi, UniteIrsaliye.SevkSeferNo, Sofor.Adi, AU.Adi,
             UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo, Sum(UniteIrsaliyeDetay.NetKilo), Sum(UniteIrsaliyeDetay.Kilo),
             AU.FaturaFirmaID, Firma.Adi from UniteIrsaliye
    left join Unite AU on AU.UniteId = UniteIrsaliye.AliciUniteId
    Left Join Sofor On Sofor.SoforId = UniteIrsaliye.SoforId
    left join UniteIrsaliyeDetay on UniteIrsaliyeDetay.UniteIrsaliyeId = UniteIrsaliye.UniteIrsaliyeId
  Left Join Firma ON Firma.FirmaID = AU.FaturaFirmaID
    where UniteIrsaliye.IrsaliyeTarihi >= :ABaslangictarihi and UniteIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
          (UniteIrsaliye.AracId = :AAracId or :AAracId is null) and
          UniteIrsaliyeDetay.UniteIrsaliyeId is not null
    group by UniteIrsaliye.AracId, UniteIrsaliye.IrsaliyeTarihi, UniteIrsaliye.SevkSeferNo, Sofor.Adi, AU.Adi,
             UniteIrsaliye.IrsaliyeSeriNo, UniteIrsaliye.IrsaliyeNo, AU.FaturaFirmaID, Firma.Adi
    into :AracId, :IrsaliyeTarihi, :SevkSeferNo, :SoforAdi, :AliciUniteAdi, :IrsaliyeSeriNo, :IrsaliyeNo,
         :ToplamKilo, :ToplamBrutKilo, :FirmaId, :FirmaAdi do
  begin

    if (FirmaId is null) then
    begin
      FirmaID = Null;
      FirmaAdi = Null;
    end

    HareketTipiStr = 'Üniteler Arası Sevk';
    suspend;
  end

  for Select MamulDepoTransferFisi.AracId, MamulDepoTransferFisi.IrsaliyeTarihi, MamulDepoTransferFisi.SevkSeferNo, Sofor.Adi, AU.Adi,
             MamulDepoTransferFisi.IrsaliyeSeriNo, MamulDepoTransferFisi.IrsaliyeNo, Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.BrutKilo) from MamulDepoTransferFisi
    left join Depo AD on AD.DepoId = MamulDepoTransferFisi.TransferDepoId
    left join Unite AU on AU.UniteId = AD.UniteId
    Left Join Sofor On Sofor.SoforId = MamulDepoTransferFisi.SoforId
    left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = MamulDepoTransferFisi.MamulDepoTransferFisiId
    Left Join PartiStokTarti on PartiStokTarti.PartiStokTartiId = MamulDepoTransferFisiDetay.PartiStokTartiId
    where MamulDepoTransferFisi.IrsaliyeTarihi >= :ABaslangictarihi and MamulDepoTransferFisi.IrsaliyeTarihi <= :ABitisTarihi and MamulDepoTransferFisi.IrsaliyeKaydi = 1 and
         (MamulDepoTransferFisi.AracID = :AAracID or :AAracID is null) and
          MamulDepoTransferFisiDetay.MamulDepoTransferFisiId is not null
    group by MamulDepoTransferFisi.AracId, MamulDepoTransferFisi.IrsaliyeTarihi, MamulDepoTransferFisi.SevkSeferNo, Sofor.Adi, AU.Adi,
             MamulDepoTransferFisi.IrsaliyeSeriNo, MamulDepoTransferFisi.IrsaliyeNo
    into :AracId, :IrsaliyeTarihi, :SevkSeferNo, :SoforAdi, :AliciUniteAdi, :IrsaliyeSeriNo, :IrsaliyeNo,
         :ToplamKilo, :ToplamBrutKilo do
  begin
    HareketTipiStr = 'Mamul Depo Transferi';
    suspend;
  end
end
