CREATE PROCEDURE MC_ISLETMEDEKIPARTILERINLISTESI(
    AFIRMAID VARCHAR(20),
    APARTITIPIID INTEGER,
    AKUMASCINSIID INTEGER,
    ATAMIR INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ALABRECETEKODU VARCHAR(20),
    AMUSTERISIPARISNO VARCHAR(50),
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    TERMINTARIHI TIMESTAMP,
    PARTISINIFID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    HAMADITOPLAMI VARCHAR(255),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    RECETEID VARCHAR(20),
    RENKADI VARCHAR(40),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISNO VARCHAR(20),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    NETSEVKKILO NUMERIC(12,3),
    NETSEVKMETRE NUMERIC(12,3),
    DURUMU INTEGER,
    DURUMUSTR VARCHAR(30),
    SONYAPILANASAMAADI VARCHAR(40),
    SONRAKIASAMAADI VARCHAR(40),
    HAMSTOKGIRISTARIHI TIMESTAMP,
    ACILISTARIHI TIMESTAMP,
    RENKOKTARIHI TIMESTAMP,
    HAMLOT VARCHAR(255),
    SIPARISGRUP2ID SIPARISGRUPKODUSIZE,
    SIPARISIVERENKULLANICI ADI50,
    GIRISHAMSTOKMASTERIDSTR VARCHAR(64),
    KALANMETRE NUMERIC(12,3),
    ISLEMGRUBUADI VARCHAR(20),
    BRUTSARFMETRE NUMERIC(12,3))
AS
declare variable ABoyaSiparisKayitId  integer;
declare variable APartiKayitId        integer;
declare variable hstIplik             integer;
declare variable hstElyaf             integer;
declare variable hstTelef             integer;
declare variable FPartiAsamalariId    integer;
declare variable SonSira              integer;
declare variable FLot                 varchar(150);
declare variable itTamir              integer;
declare variable Okey                 integer;
declare variable BrutSevkMetre        numeric(12,3);
declare variable BrutIadeMetre        numeric(12,3);
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  Select itTamir from IslemTipiGetir into :itTamir;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  ABoyaSiparisKayitId = NUll;
  APartiKayitId = null;

  if (ABoyaSiparisId is not null) then
  begin
    Select BoyaSiparisKayitId from BoyaSiparis
       where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId into :ABoyaSiparisKayitId;
  end

  if (APartiId is not null) then
  begin
    Select PartiKayitId from Parti
       where Parti.PartiYil = :APartiYil and Parti.PartiId = :APartiId into :APartiKayitId;
  end

  for select PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil,
            BoyaSiparis.BoyaSiparisId, Parti.IslemTipiId, IslemTipi.Adi, Parti.TerminTarihi,
            PartiStok.PartiSinifId, PartiStok.BoyaSiparisDetayId, PartiStok.HamAdiToplami, Parti.ReceteId, Recete.Adi,
            Parti.Fason, PartiStok.Kilo, PartiStok.Metre, PartiStok.NetSevkKilo, PartiStok.NetSevkMetre, PartiStok.Durumu,
            BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.MusteriRenkKoduveAdi,
            PartiStok.PartiAsamalariId, LabRecete.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr, BoyaSiparis.SiparisGrup2Id, FirmaYetkili.Adi,
            PartiStok.BrutSevkMetre, Partistok.BrutIadeMetre, IslemGrubu.Adi, partistok.brutsarfmetre from PartiStok
      left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      left join Firma On Firma.FirmaId = Parti.FirmaId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join Ham on Ham.HamId = PartiStok.HamId
      left join PartiTamirSinif on PartiTamirSinif.PartiSinifId = PartiStok.PartiSinifId and PartiTamirSinif.Bitti = 0
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
      left join Recete on Recete.ReceteId = Parti.ReceteId
      left join PartiDurum on PartiDurum.PartiDurumId = PartiStok.Durumu
      left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
      left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
      left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
      left join FirmaYetkili on FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
      left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
      where Parti.FirmaId = :AFirmaId and PartiStok.Durumu < 8 and PartiStok.PartiStokIptalKayitId is null and
             ( :APartiKayitId is null and
             (Parti.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
             (BoyaSiparisDetay.LabReceteKodu = :ALabReceteKodu or :ALabReceteKodu is null) and
             (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is null) and
             ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :APartiTipiId = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :APartiTipiId = 1) or :APartiTipiId is null) and
             (Ham.KumasCinsiId = :AKumasCinsiId or :AKumasCinsiId is null) and
             ( (Parti.IslemTipiId <> :itTamir and :ATamir = 0) or (Parti.IslemTipiId = :itTamir and :ATamir = 1) or :ATamir is null) or
             (Parti.PartiKayitId = :APartiKayitId and :APartiKayitId is not null ))
      order by PartiStok.PartiKayitId, PartiStok.PartiSinifId
      into :PartiKayitId, :PartiYil, :PartiId, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :IslemTipiId, :IslemTipiAdi,
           :TerminTarihi, :PartiSinifId, :BoyaSiparisDetayId, :HamAdiToplami, :ReceteId, :RenkAdi,
           :Fason, :Kilo, :Metre, :NetSevkKilo, :NetSevkMetre, :Durumu, :MusteriSiparisNo, :MusteriRenkKoduveAdi,
           :FPartiAsamalariId, :LabReceteKodu, :LabReceteAdi, :LabReceteAtisNo, :SiparisGrup2Id, :SiparisiVerenKullanici,
           :BrutSevkMetre, :BrutIadeMetre, :IslemGrubuAdi, :BrutSarfMetre do
  begin
    KalanMetre = Metre - BrutSevkMetre - BrutSarfMetre + Coalesce(BrutIadeMetre,0);
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    else BoyaSiparisIdStr = '';

    if (Fason = 1) then
    begin
      FasonStr = 'Fason';

      Select Max(HamStokIrsaliye.IrsaliyeTarihi) from PartiStokDetay
        left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
        left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left Join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
        where PartiStokDetay.PartiSinifId = :PartiSinifId
        into :HamStokGirisTarihi;

    end
    else
    begin
      FasonStr = '';
      HamStokGirisTarihi = Null;
    end

    if (FPartiAsamalariId is not null) then
    begin
      select Asama.Adi, PartiAsamalari.Sira from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariID
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        where PartiAsamaStok.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId
        into :SonYapilanAsamaAdi, :SonSira;
    end
    else
    begin
      SonYapilanAsamaAdi = Null;
      SonSira = -1;
    end

    SonrakiAsamaAdi = Null;

    select First 1 Asama.Adi from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira > :SonSira and PartiAsamaStok.AsamayaDahil = 1
      order by PartiAsamalari.Sira
      into :SonrakiAsamaAdi;


    Select DurumuStr from MC_PartiDurumHesapla(:PartiSinifId, :Durumu) into :DurumuStr;

    HamLot = '';
    Select Substring(List(Distinct(HamStokMaster.Lot)) From 1 For 255) from PartiStokDetay
            Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
            Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
            Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
            where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.Lot is not null and HamStokMaster.Lot <> ''
            group by PartiStokDetay.PartiSinifId
            into :HamLot;

    GirisHamStokMasterIdStr = Null ;
    Select HamStokMasterIdStr from PartiStokGirisBilgisiGetir(:PartiSinifID) into :GirisHamStokMasterIdStr ;

    suspend;
  end
end
