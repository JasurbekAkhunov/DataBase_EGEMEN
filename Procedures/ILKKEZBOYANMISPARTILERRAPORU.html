CREATE PROCEDURE ILKKEZBOYANMISPARTILERRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ADIGERPARTILERIDEDOK EGEBOOLEAN)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    ACILISTARIHI TIMESTAMP,
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    MODELNO VARCHAR(20),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ISLEMGRUBUADI VARCHAR(20),
    ISLEMTIPIADI VARCHAR(20),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUP2ID VARCHAR(4),
    HAMADITOPLAMI VARCHAR(255),
    DENYE VARCHAR(40),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    MAKINAID VARCHAR(10),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    ICTAMIRVAR INTEGER,
    ICTAMIRVARSTR VARCHAR(4),
    ICTAMIRTIPIADI VARCHAR(100),
    ILAVESARFVAR INTEGER,
    ILAVESARFVARSTR VARCHAR(4),
    ILKBOYANANPARTI EGEBOOLEAN,
    HAMTIPIID INTEGER,
    HAMTIPIADI ADI20,
    HAMSTOKMALIYETI DOUBLE PRECISION,
    ASAMAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    MAKINAUMUZERINDENASAMAMALIYETI DOUBLE PRECISION,
    BOYAKIMYASALMALIYETI DOUBLE PRECISION,
    TEORIKBOYAKIMYASALMALIYETI DOUBLE PRECISION,
    UNITELERASEVKMALIYETI DOUBLE PRECISION)
AS
declare variable FHamAdiToplami varchar(255);
declare variable KS integer;
declare variable FTamirTipiAdi varchar(20);
declare variable FDenye varchar(30);
begin
  for select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi,
         Parti.ModelNo, PartiAsamalari.ReceteId, Recete.Adi, IslemGrubu.Adi, IslemTipi.Adi,
         BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id, Parti.Fason, PartiAsamaStokDetay.MakinaId,
         Min(PartiAsamaStokDetay.GirisTarihi),  max(PartiAsamaStokDetay.CikisTarihi),  iif(Recete.IlkBoyananPartiKayitId = Parti.PartiKayitId, 1, 0)  from PartiAsamaStokDetay
  left join PartiAsamalari On PartiAsamalari.PartiAsamalariID = PartiAsamaStokDetay.PartiAsamalariId
  left join Parti On Parti.PartiKayitId = PartiAsamalari.PartiKayitId
  left join Firma on Firma.FirmaId = Parti.FirmaId
  left join Recete On Recete.ReceteID = PartiAsamalari.ReceteId
  left join LabRecete On LabRecete.LabReceteID = Recete.LabReceteId
  left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
  left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  where PartiAsamaStokDetay.CikisTarihi >= :ABaslangicTarihi and
        PartiAsamaStokDetay.CikisTarihi < :ABitisTarihi and
        PartiAsamalari.ReceteID is not null and
        PartiAsamalari.PartiTamirID is null and
        (:ADigerPartilerideDok = 1 or :ADigerPartilerideDok = 0 and Recete.IlkBoyananPartiKayitId = PartiAsamaStokDetay.PartiKayitId) and
        PartiAsamalari.IlaveProses = 0 and
        (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
        Recete.SokumUzeriBoya = 0 and
        Recete.Ilave = 0
  group by Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi,
           Parti.ModelNo, PartiAsamalari.ReceteId, Recete.Adi, Recete.IlkBoyananPartiKayitId, IslemGrubu.Adi, IslemTipi.Adi,
           BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,  BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id, Parti.Fason, PartiAsamaStokDetay.MakinaId
  into :PartiKayitId, :PartiYil, :PartiId, :AcilisTarihi, :PartigrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi,
       :ModelNo, :ReceteId, :ReceteAdi, :IslemGrubuAdi, :IslemTipiAdi,
       :BoyaSiparisYil, :BoyaSiparisId, :SiparisGrupId, :SiparisGrup2Id, :Fason, :MakinaId, :GirisTarihi, :CikisTarihi, :IlkBoyananParti do
  begin
    if (Fason = 1) then FasonStr = 'Evet';  else FasonStr = null;

    Select sum(PartiStok.Kilo), sum(PartiStok.Metre) from PartiStok where PartiStok.PartiKayitId = :PartiKayitId into :Kilo, :Metre;

    if (PartiYil is null) then PartiIdStr = '';
    else PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId)  || ' ';

    if (BoyaSiparisYil is null) then BoyaSiparisIdStr = '';
    else BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    HamAdiToplami = '';

    For Select PartiStok.HamAdiToplami from PartiStok
      where PartiStok.PartiKayitId = :PartiKayitId
      group By PartiStok.HamAdiToplami
      into :FHamAdiToplami do
    begin
      if (Char_Length(HamAdiToplami) + Char_Length(FHamAdiToplami) < 250) then
      begin
        if (HamAdiToplami <> '' ) then HamAdiToplami = HamAdiToplami || cast(', ' as varchar(2) character set UTF8);
        HamAdiToplami = HamAdiToplami || FHamAdiToplami;
      end
    end

    Denye = '';

    for Select HamStokMaster.Denye from PartiStokDetay
     left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
     left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
     left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
     where PartiStokDetay.PartiKayitId = :PartiKayitId and HamStokMaster.Denye is not null and HamStokMaster.Denye <> 0
     group by HamStokMaster.Denye
     into :FDenye do
    begin
      if (Char_Length(Denye) + Char_Length(FDenye) < 38) then
      begin
        if (Denye <> '') then Denye = Denye || cast(', ' as varchar(2) character set UTF8);
        Denye = Denye || FDenye;
      end
    end

    Select count(*) From PartiAsamaIlaveSarf
      Left Join PartiAsamaIlaveSarfDetay on PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfID = PartiAsamaIlaveSarf.PartiAsamaIlaveSarfID
      Left Join Urun on Urun.UrunID = PartiAsamaIlaveSarfDetay.UrunID
      where PartiAsamaIlaveSarf.PartiKayitId = :PartiKayitId and Urun.UrunTipiID = 0 and PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfDetayID is Not Null
    into :KS;


    if (KS = 0) then
    begin
      IlaveSarfvar = 0;
      IlaveSarfvarStr = '';
    end
    else
    begin
      IlaveSarfVar = 1;
      IlaveSarfvarStr = 'Evet';
    end

    Select count(*) From PartiTamir
      left join TamirTipi on TamirTipi.TamirTipiID = PartiTamir.TamirTipiID
      where PartiTamir.PartiKayitId = :PartiKayitId and TamirTipi.RenkTamirimi = 1 into :KS;

    if (KS = 0) then
    begin
      IcTamirVar = 0;
      IcTamirVarStr = '';
      IcTamirTipiAdi = '';
    end
    else
    begin
      IcTamirVar = 1;
      IcTamirVarStr = 'Evet';
      IcTamirTipiAdi = '';

      For Select TamirTipi.Adi, Count(*) from PartiTamir
        left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
        where PartiTamir.PartiKayitId = :PartiKayitId and TamirTipi.RenkTamirimi = 1
        group by TamirTipi.Adi
        into :FTamirTipiAdi, :KS do
      begin
        if (Char_Length(IcTamirTipiAdi) + Char_Length(FTamirTipiAdi) < 90) then
        begin
          if (IcTamirTipiAdi <> '' ) then IcTamirTipiAdi = IcTamirTipiAdi || cast(', ' as varchar(2) character set UTF8);

          IcTamirTipiAdi = IcTamirTipiAdi || FTamirTipiAdi;

          if (KS > 1) then
            IcTamirTipiAdi = IcTamirTipiAdi || '(' || EgemenIntegerToStr(KS) || ')';
        end
      end
    end

    /*Partinin ham tipini belirleyebilmek için PartiStok larındaki bir hamın tipine bakıyoruz*/
    select First 1 HamTipi.HamTipiId, HamTipi.Adi from PartiStok
    left join HamTipi on HamTipi.HamTipiId = PartiStok.HamTipiId
    where PartiStok.PartiKayitId = :PartiKayitId into :HamTipiId, :HamTipiAdi;

    suspend;

    select first 1 HamStokMaliyeti, AsamaUMUzerindenAsamaMaliyeti, MakinaUMUzerindenAsamaMaliyeti, BoyaKimyasalMaliyeti, TeorikBoyaKimyasalMaliyeti, UnitelerASevkMaliyeti from PartiMaliyetKumulatif
       where PartiKayitID = :PartiKayitID
    into :HamStokMaliyeti, :AsamaUMUzerindenAsamaMaliyeti, :MakinaUMUzerindenAsamaMaliyeti, :BoyaKimyasalMaliyeti, :TeorikBoyaKimyasalMaliyeti, :UnitelerASevkMaliyeti;

  end
end
