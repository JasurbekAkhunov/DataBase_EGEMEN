CREATE PROCEDURE HAMSTOKANALIZIFASON(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    ATEMSILCIID INTEGER,
    AHAREKETIOLMAYANLARIDOKSUNMU INTEGER,
    AONAYVERILMEMISLERDAHIL INTEGER,
    AHAMSTOKTIPI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAGRUP1 VARCHAR(4),
    FIRMAGRUP2 VARCHAR(4),
    FIRMAGRUP3 VARCHAR(4),
    TEMSILCIADI VARCHAR(40),
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    DEVREDENMIKTAR NUMERIC(12,3),
    GIRISMIKTAR NUMERIC(12,3),
    PARTILENENMIKTAR NUMERIC(12,3),
    IADEEDILENMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3))
AS
declare variable Okey integer;
declare variable FirmaKodumuz varchar(20);
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef       integer;
declare variable Miktar         Numeric(12,3);
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  Select FirmaKodumuz from Sistem into :FirmaKodumuz;

  for Select Firma.FirmaId, Firma.Adi, FirmaGrup1, FirmaGrup2, FirmaGrup3, Kullanici.Adi from Firma
    left join Kullanici on Kullanici.KullaniciId = Firma.TemsilciId
  where ((:AFirmaAralikli = 1 and Firma.FirmaId >= :ABaslangicFirmaId and (Firma.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0))  and
        ((:AFirmaGrup1Aralikli = 1 and FirmaGrup1 >= :ABaslangicGrup1Id  and (FirmaGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and FirmaGrup2 >= :ABaslangicGrup2Id and (FirmaGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and FirmaGrup3 >= :ABaslangicGrup3Id and (FirmaGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
        (Firma.TemsilciId = :ATemsilciId or :ATemsilciId is null)
  into :FirmaId, :FirmaAdi, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :TemsilciAdi  do
  begin
    Okey = 1;

    if (FirmaId = FirmaKodumuz) then Okey = 0;

    if (okey = 1) then
    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (okey = 1) then
    begin
      For select HamTipi.HamTipiId, HamTipi.Adi, sum(GirisMiktar), sum(PartilenenMiktar), sum(IadeEdilenMiktar) from HamTipi
        left join Ham on Ham.HamTipiId = HamTipi.HamTipiId
        left join HamStokKumilatif on HamStokKumilatif.HamId = Ham.HamId and HamStokKumilatif.Firmaid = :FirmaId and HamStokKumilatif.Tarih < :ABaslangicTarihi
        where ( ( (HamTipi.HamTipiId = :hstIplik or HamTipi.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamStokTipi = 0) or (HamTipi.HamTipiId <> :hstIplik and HamTipi.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :AHamStokTipi = 1))
        group by HamTipi.HamTipiId, HamTipi.Adi
        into :HamTipiId, :HamTipiAdi, :GirisMiktar, :PartilenenMiktar, :IadeEdilenMiktar do
      begin
        Okey = 1;
        if (GirisMiktar is null) then GirisMiktar = 0.0;
        if (PartilenenMiktar is null) then PartilenenMiktar = 0.0;
        if (IadeEdilenMiktar is null) then IadeEdilenMiktar = 0.0;

        DevredenMiktar = GirisMiktar - PartilenenMiktar  - IadeEdilenMiktar;

        if (DevredenMiktar > -1 and DevredenMiktar < 0 ) then DevredenMiktar = 0;

        if (AOnayVerilmemislerDahil = 1) then
        begin
          Miktar = 0;

          Select Sum(HamStokMaster.Miktar) from HamStokMaster
            left join Ham On Ham.HamID = HamStokMaster.HamId
            where HamStokMaster.PartiSinifId is null and HamStokMaster.StokFirmaId = :FirmaId and HamStokMaster.Tarih < :ABaslangicTarihi and
                  HamStokMaster.OnayDurumu = 0 and ham.HamTipiID = :HamTipiId
            into :Miktar;

          if (Miktar is not null) then DevredenMiktar = DevredenMiktar + Miktar;
        end


        select sum(girisMiktar), Sum(PartilenenMiktar), Sum(iadeedilenMiktar) from HamStokKumilatif
          left join Ham on Ham.HamId = HamStokKumilatif.HamId
          where HamStokKumilatif.FirmaId = :FirmaId and Ham.HamTipiId = :HamTipiId and
               HamStokKumilatif.Tarih >= :ABaslangicTarihi and HamStokKumilatif.Tarih <= :ABitisTarihi
          into :GirisMiktar, :PartilenenMiktar, :IadeEdilenMiktar;

        if (GirisMiktar is null) then GirisMiktar = 0.0;
        if (PartilenenMiktar is null) then PartilenenMiktar = 0.0;
        if (IadeEdilenMiktar is null) then IadeEdilenMiktar = 0.0;

        if (AOnayVerilmemislerDahil = 1) then
        begin
          Miktar = 0;

          Select Sum(HamStokMaster.Miktar) from HamStokMaster
            left join Ham On Ham.HamID = HamStokMaster.HamId
            where HamStokMaster.PartiSinifId is null and HamStokMaster.StokFirmaId = :FirmaId and HamStokMaster.Tarih >= :ABaslangicTarihi and HamStokMaster.Tarih <= :ABitisTarihi and
                  HamStokMaster.OnayDurumu = 0 and ham.HamTipiID = :HamTipiId
            into :Miktar;

          if (Miktar is not null) then GirisMiktar = GirisMiktar + Miktar;
        end

        KalanMiktar = DevredenMiktar + GirisMiktar - (PartilenenMiktar + IadeEdilenMiktar);

        if (DevredenMiktar = 0 and GirisMiktar = 0 and PartilenenMiktar = 0 and IadeEdilenMiktar = 0) then Okey = 0;

        if (AHareketiOlmayanlariDoksunmu = 0) then
        if (GirisMiktar = 0 and PartilenenMiktar = 0 and IadeEdilenMiktar = 0) then Okey = 0;

       if (Okey = 1) then suspend;
      end
    end
  end
end
