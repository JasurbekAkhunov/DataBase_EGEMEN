CREATE PROCEDURE MODELLISTESI(
    AMODELID MODELKODUSIZE,
    ASEZONID INTEGER,
    AMODELGRUPID MODELGRUPKODUSIZE,
    AMODELGRUPID2 MODELGRUPKODUSIZE,
    AMUSTERIMODELNO VARCHAR(30),
    AONAYDURUMU INTEGER)
RETURNS (
    MODELID MODELKODUSIZE,
    MODELKAYITID INTEGER,
    ADI ADI40,
    KISAADI ADI20,
    SEZONID INTEGER,
    SEZONADI ADI30,
    KOLEKSIYONID INTEGER,
    KOLEKSIYONADI ADI30,
    MODELGRUPID MODELGRUPKODUSIZE,
    MODELGRUPADI ADI20,
    MODELGRUP2ID MODELGRUPKODUSIZE,
    MODELGRUP2ADI ADI20,
    YIKAMATALIMATIID INTEGER,
    MUSTERIMODELNO VARCHAR(30),
    FIRMAID FIRMAKODUSIZE,
    FIRMAADI ADI100,
    NOTU MEMO,
    ORMENOTU MEMO,
    DOKUMANOTU MEMO,
    BOYAHANENOTU MEMO,
    AKSESUARNOTU MEMO,
    NUMUNENOTU MEMO,
    DIKISTALIMATI MEMO,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYTARIHI TIMESTAMP,
    ONAYLANAKULLANICIADI ADI40,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(20),
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    INSERTKULLANICIADI ADI40,
    YIKAMATALIMATIADI ADI60,
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for select Model.ModelId, Model.ModelKayitId, Model.Adi, Model.KisaAdi, Model.SezonId, Sezon.Adi SezonAdi, Model.KoleksiyonId, Koleksiyon.Adi KoleksiyonAdi, Model.ModelGrupId,
             ModelGrup.Adi ModelGrupAdi, Model.YikamaTalimatiId, Model.MusteriModelNo, Model.FirmaId,
             Firma.Adi FirmaAdi, Model.Notu, Model.OrmeNotu, Model.DokumaNotu, Model.BoyahaneNotu, Model.AksesuarNotu,
             Model.NumuneNotu, Model.DikisTalimati, Model.OnaylayanKullaniciId, Model.OnayTarihi,
             Kullanici.Adi OnaylanaKullaniciAdi, Model.OnayDurumu, decode(Model.OnayDurumu, 0, 'Onay Bekleniyor', 1, 'Onaylandı', 2, 'Reddedildi', ''),
             Model.InsertKullaniciId, Model.InsertTarihi, EkleyenKullanici.Adi InsertKullaniciAdi, YikamaTalimati.Adi YikamaTalimatiAdi, Model.DegisimTarihi,
             Model.ModelGrup2Id, ModelGrup2.Adi ModelGrup2Adi from Model
      left join Sezon on Sezon.SezonId = Model.SezonId
      left join Koleksiyon on Koleksiyon.KoleksiyonId = Model.KoleksiyonId
      left join Firma on Firma.FirmaId = Model.FirmaId
      left join ModelGrup on ModelGrup.ModelGrupId = Model.ModelGrupId
      left join ModelGrup ModelGrup2 on ModelGrup2.ModelGrupId = Model.ModelGrup2Id
      left join Kullanici on Kullanici.KullaniciId = Model.OnaylayanKullaniciId
      left join Kullanici EkleyenKullanici on EkleyenKullanici.KullaniciId = Model.InsertKullaniciId
      left join YikamaTalimati on YikamaTalimati.YikamaTalimatiId = Model.YikamaTalimatiId
      where (Model.ModelId like :AModelId || '%' or :AModelId is Null) and
            (Model.SezonId = :ASezonId or :ASezonId is null) and
            (Model.MusteriModelNo = :AMusteriModelNo or :AMusteriModelNo is null) and
            (Model.ModelGrupId = :AModelGrupID or :AModelGrupID is null) and
            (Model.ModelGrup2Id = :AModelGrupId2 or :AModelGrupId2 is null) and
            (Model.OnayDurumu = :AOnayDurumu or :AOnayDurumu is null)
      into :ModelId, :ModelKayitId, :Adi, :KisaAdi, :SezonId, :SezonAdi, :KoleksiyonId, :KoleksiyonAdi, :ModelGrupId, :ModelGrupAdi, :YikamaTalimatiId,
           :MusteriModelNo, :FirmaId, :FirmaAdi, :Notu, :OrmeNotu, :DokumaNotu, :BoyahaneNotu, :AksesuarNotu,
           :NumuneNotu, :DikisTalimati, :OnaylayanKullaniciId, :OnayTarihi, :OnaylanaKullaniciAdi, :OnayDurumu, :OnayDurumuStr,
           :InsertKullaniciId, :InsertTarihi, :InsertKullaniciAdi, :YikamaTalimatiAdi, :DegisimTarihi, :ModelGrup2Id, :ModelGrup2Adi
  do
  begin
    suspend;
  end
end
