CREATE PROCEDURE COZGUISEMRIRECETEGETIR(
    ACOZGUISEMRIKAYITID INTEGER)
RETURNS (
    COZGUISEMRIKAYITID INTEGER,
    COZGUISEMRIRECETEID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    LOT VARCHAR(30),
    KULLANIMORANI DOUBLE PRECISION,
    FIREORANI DOUBLE PRECISION,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    TELSAYISI INTEGER,
    SIRANO SMALLINT,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATDOVIZID DOVIZSIZE,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    NETSTOKMIKTARI BOYAMIKTARSIZE,
    BRUTSTOKMIKTARI BOYAMIKTARSIZE,
    IHTIYACAGIRLIKKILO BOYAMIKTARSIZE,
    IHTIYACUZUNLUKMETRE BOYAMIKTARSIZE,
    IHTIYACHATASTR VARCHAR(500),
    URETICIADI VARCHAR(40),
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    FIRMAADI VARCHAR(100),
    IPLIKUZAMADEGERI DOUBLE PRECISION,
    RESULTID VARCHAR(255))
AS
begin
  For select CozguIsEmriRecete.CozguIsEmriKayitId, CozguIsEmriRecete.CozguIsEmriReceteId, CozguIsEmriRecete.HamID, Ham.Adi, CozguIsEmriRecete.Lot,
             CozguIsEmriRecete.KullanimOrani, CozguIsEmriRecete.FireOrani, CozguIsEmriRecete.RenkID, Renk.Adi, CozguIsEmriRecete.TelSayisi, CozguIsEmriRecete.SiraNo,
             CozguIsEmriRecete.BirimFiyat, CozguIsEmriRecete.BirimFiyatDovizId, Ik.Adi, CozguIsEmriRecete.InsertTarihi, Sk.Adi, CozguIsEmriRecete.DegisimTarihi, Uretici.Adi,
             OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, Firma.Adi ,StokHareket.IplikUzamaDegeri from CozguIsEmriRecete
  Left Join Kullanici IK on IK.KullaniciId = CozguIsEmriRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = CozguIsEmriRecete.KullaniciId
  left join Renk on Renk.RenkId = CozguIsEmriRecete.RenkId
  left join Ham on Ham.HamId = CozguIsEmriRecete.HamId
  left join Uretici on Uretici.UReticiId = CozguIsEmriRecete.UreticiId
  left join OrmeRaf on OrmeRaf.OrmeRafId = CozguIsEmriRecete.OrmeRafId
  left join StokHareket on StokHareket.StokGirisNo = OrmeRaf.StokGirisNo
  left join Firma on Firma.FirmaId = StokHareket.FirmaId
  where CozguIsEmriRecete.CozguIsEmriKayitId = :ACozguIsEmriKayitId
  Order By SiraNo
  into :CozguIsEmriKayitId, :CozguIsEmriReceteId, :HamID, :HamAdi, :Lot, :KullanimOrani, :FireOrani, :RenkID, :RenkAdi, :TelSayisi, :SiraNo,
       :BirimFiyat, :BirimFiyatDovizId, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :UreticiAdi,
       :OrmeRafId, :StokGirisNo, :FirmaAdi, :IplikUzamaDegeri do
  begin
    select Sum(OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari), Sum(OrmeRaf.BrutGirisMiktari - OrmeRaf.BrutCikisMiktari)
    from OrmeRaf
    left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
    left join Depo on Depo.DepoId = DepoRaf.DepoId
    where OrmeRaf.HamId = :HamId and
    (:Lot is null and OrmeRaf.Lot is null or :Lot is not null and OrmeRaf.Lot = :Lot) and
    (:RenkId is null and OrmeRaf.RenkId is null or :RenkId is not null and OrmeRaf.RenkId = :RenkId) and
    OrmeRaf.StokVar = 1 and
    Depo.CozguDeposu = 0 and
    Depo.HasilDeposu = 0 and
    Depo.IndigoDeposu = 0
    into :NetStokMiktari, :BrutStokMiktari;

    select DokumaIplikMiktariHesaplama.AgirlikKilo, DokumaIplikMiktariHesaplama.UzunlukMetre, DokumaIplikMiktariHesaplama.HataStr
    from DokumaIplikMiktariHesaplama(Null, Null, :CozguIsEmriReceteId, 0, Null, Null) into :IhtiyacAgirlikKilo, :IhtiyacUzunlukMetre, :IhtiyacHataStr;

    ResultId = '"' || EgemenIntegerToStr(HamId) || '",' ;

    if (RenkId is null) then
      ResultId = ResultId || '"",' ;
    else
      ResultId = ResultId || '"' || RenkId || '",' ;

    if (Lot is null or Lot = '') then
      ResultId = ResultId || '""' ;
    else
      ResultId = ResultId || '"' || Lot || '"' ;

    suspend;
  end
end
