CREATE PROCEDURE SINIFIYENIPARTIACTASI(
    AAKTARILACAKPARTISINIFID INTEGER,
    AYENIBOYASIPARISDETAYID INTEGER,
    AYENIBOYASIPARISSARJID INTEGER,
    AYENIBOYASIPARISBASKIID INTEGER)
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIKAYITID INTEGER)
AS
Declare variable FirmaId      varchar(20);
Declare variable partiGrupId  varchar(4);
Declare variable Fason        integer;
Declare variable IslemTipiId  integer;
Declare variable IslemGrubuId  integer;
Declare variable AciliyetDerecesiId  integer;
Declare variable TerminTarihi     TimeStamp;
Declare variable AcilisTarihi     TimeStamp;
Declare variable BoyaSiparisKayitId integer;
Declare variable Sinif              varchar(1);
Declare variable PartiTipiId        integer;
Declare variable ParcaliSevk        integer;
Declare variable OnayDurumu         integer;
Declare variable OnayTarihi         TimeStamp;
Declare variable OnaylayanKullaniciId integer;
Declare variable ImalatTipiId         integer;
begin
  PartiYil = TimeStampToYil(Current_Date);

  AciliyetDerecesiId = 0;
  Select ID FROM PartiNumaratorAL(:PartiYil) into :PartiId;

  Select BoyaSiparis.FirmaId, BoyaSiparisDetay.IslemTipiId, BoyaSiparisDetay.SonTerminTarihi, BoyaSiparis.BoyaSiparisKayitID, BoyaSiparis.Fason from BoyaSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitID
    where BoyaSiparisDetay.BoyaSiparisDetayID = :AYeniBoyaSiparisDetayId
    into :FirmaId, :IslemTipiId, :TerminTarihi, :BoyaSiparisKayitID, :Fason;

  Select Parti.IslemGrubuId, Parti.PartiGrupId, Parti.Sinif, Parti.PartiTipiId, Parti.AcilisTarihi, Parti.OnayDurumu,
         Parti.OnayTarihi, Parti.OnaylayanKullaniciId from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    where PartiStok.PartiSinifId = :AAktarilacakPartiSinifId
    into :IslemGrubuId, :PartiGrupId, :Sinif, :PartiTipiId, :AcilisTarihi, :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId;

  if (PartiTipiId = 0) then
    Select IplikDefaultParcaliSevkDurumu from Sistem where SistemID = 1 into :ParcaliSevk;
  else
    Select DefaultParcaliSevkDurumu from Sistem where SistemID = 1 into :ParcaliSevk;

  if (AYeniBoyaSiparisBaskiId is null) then ImalatTipiId = 0;
  else ImalatTipiId = 1;

  insert into Parti(PartiYil, PartiId, AcilisTarihi, FirmaId, PartiGrupId, Fason, IslemGrubuId, IslemTipiId, ImalatTipiId, AciliyetDerecesiId, TerminTarihi,
                    BoyaSiparisKayitId, Sinif, PartiTipiId, ParcaliSevk, OnayDurumu, OnayTarihi, OnaylayanKullaniciId)
              values(:PartiYil, :PartiId, :AcilisTarihi, :FirmaId, :PartiGrupId, :Fason, :IslemGrubuId, :IslemTipiId, :ImalatTipiId, :AciliyetDerecesiId, :TerminTarihi,
                     :BoyaSiparisKayitId, :Sinif, :PartiTipiId, :ParcaliSevk, :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId);

  Select Parti.PartiKayitID from Parti where Parti.PartiYil = :PartiYil and Parti.PartiID = :PartiId into :PartiKayitId;

  Execute Procedure SinifiBaskaPartiyeTasi(:AAktarilacakPartiSinifId, :PartiKayitId, :AYeniBoyaSiparisDetayId, :AYeniBoyaSiparisSarjId, :AYeniBoyaSiparisBaskiId, 0, 0);
  suspend;
end
