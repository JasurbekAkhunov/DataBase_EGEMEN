CREATE PROCEDURE BOYAYUKDAGILIMIRAPORUDETAYI(
    ABOYAHANEBOLUMID INTEGER,
    AFIRMAID VARCHAR(20),
    ATEMSILCIID INTEGER,
    ASATISTIPI INTEGER)
RETURNS (
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    RAPORSIRANO INTEGER,
    FIRMAID VARCHAR(20),
    YIKAMAADET INTEGER,
    BEYAZADET INTEGER,
    RENKLIADET INTEGER,
    ONAYSIZADET INTEGER,
    UCRETSIZDISTAMIRADET INTEGER,
    YIKAMAKILO NUMERIC(12,3),
    BEYAZKILO NUMERIC(12,3),
    RENKLIKILO NUMERIC(12,3),
    ONAYSIZKILO NUMERIC(12,3),
    UCRETSIZDISTAMIRKILO NUMERIC(12,3),
    TEMSILCIID INTEGER,
    TEMSILCIADI VARCHAR(40),
    FASON INTEGER)
AS
declare variable FMakinaPlanlamaGrupId    varchar(10);
declare variable FLastMakinaPlanlamaGrupId varchar(10);
declare variable FFirmaId                  varchar(20);
declare variable FLastFirmaId              varchar(20);
declare variable FUstIslemGrubu            integer;
declare variable FUcretsizDisTamir         integer;
declare variable FOnayDurumu               integer;
declare variable FKilo                     Numeric(12,3);
declare variable FSarjSayisi               integer;
begin
  FLastMakinaPlanlamaGrupId = '@@@';
  FLastFirmaId = '';

  For Select MakinaPlanlamaGrupId, FirmaId, UstIslemGrubu, UcretsizDisTamir, OnayDurumu, TemsilciId, TemsilciAdi, Fason, Sum(SarjKilo), Count(*) from BoyaYukDagilimiRaporuHazirla(:ABoyahaneBolumId, :AFirmaID, :ATemsilciId, :ASatisTipi)
  group by MakinaPlanlamaGrupId, FirmaId, UstIslemGrubu, UcretsizDisTamir, OnayDurumu, FirmaSarj, TemsilciId, TemsilciAdi, Fason
  into :FMakinaPlanlamaGrupId, :FFirmaId, :FUstIslemGrubu, :FUcretsizDisTamir, :FOnayDurumu, :TemsilciId, :TemsilciAdi, :Fason, :FKilo, :FSarjSayisi do
  begin
    if (FMakinaPlanlamaGrupId <> FLastMakinaPlanlamaGrupId or FFirmaID <> FLastFirmaId) then
    begin
      if (FLastMakinaPlanlamaGrupId <> '@@@') then
      begin
        suspend;
      end

      MakinaPlanlamaGrupId = FMakinaPlanlamaGrupId;
      FLastMakinaPlanlamaGrupId = FMakinaPlanlamaGrupId;
      FirmaId = FFirmaId;
      FLastFirmaId = FFirmaId;

      YikamaAdet                 = 0;
      BeyazAdet                  = 0;
      RenkliAdet                  = 0;
      OnaysizAdet                = 0;
      UcretsizDisTamirAdet       = 0;
      YikamaKilo                 = 0;
      BeyazKilo                  = 0;
      RenkliKilo                  = 0;
      OnaysizKilo                = 0;
      UcretsizDisTamirKilo       = 0;
    end

    if (FOnayDurumu = 0) then
    begin
      OnaysizAdet  = OnaysizAdet + FSarjSayisi;
      OnaysizKilo  = OnaysizKilo + FKilo;
    end
    else if (FUcretsizDisTamir = 1) then
    begin
      UcretsizDisTamirAdet  = UcretsizDisTamirAdet + FSarjSayisi;
      UcretsizDisTamirKilo  = UcretsizDisTamirKilo + Fkilo;
    end
    else if (FUstIslemGrubu = 1) then
    begin
      YikamaAdet   = YikamaAdet + FSarjSayisi;
      YikamaKilo   = YikamaKilo + FKilo;
    end
    else if (FUstIslemGrubu = 2) then
    begin
      BeyazAdet   = BeyazAdet + FSarjSayisi;
      BeyazKilo   = BeyazKilo + FKilo;
    end
    else if (FUstIslemGrubu = 3) then
    begin
      RenkliAdet   = RenkliAdet + FSarjSayisi;
      RenkliKilo   = RenkliKilo + FKilo;
    end
  end

  if (FLastMakinaPlanlamaGrupId <> '@@@') then
  begin
    suspend;
  end
end
