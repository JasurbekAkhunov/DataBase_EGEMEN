CREATE PROCEDURE EFICINFISDETAYIGETIR(
    BASLANGICTARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP,
    GONDERILMEDURUMU INTEGER,
    EARSIVAKTIF SMALLINT = 0)
RETURNS (
    FATURAID INTEGER,
    STOKHAREKETID INTEGER,
    MODULID INTEGER,
    URUNADI VARCHAR(255),
    MUSTERISIPARISNO VARCHAR(50),
    BIRIMID VARCHAR(2),
    KDV DOUBLE PRECISION,
    TEVKIFAT INTEGER,
    KAPADEDI DOUBLE PRECISION,
    FATURAMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMISKONTOPARASAL DOUBLE PRECISION,
    BIRIMISKONTOORANI1 DOUBLE PRECISION,
    BIRIMISKONTOORANI2 DOUBLE PRECISION,
    BIRIMISKONTOTOPLAMI DOUBLE PRECISION,
    ARATOPLAM NUMERIC(18,2),
    KDVTUTARI NUMERIC(18,2),
    GENELTOPLAM NUMERIC(18,2),
    ISKONTOORANI DOUBLE PRECISION,
    TEVKIFATORANI DOUBLE PRECISION,
    TEVKIFATTUTARI NUMERIC(18,2),
    SIPARISPROSESADI VARCHAR(255),
    ILAVEISLEMLER VARCHAR(1024),
    ESKIPARTINO VARCHAR(100),
    HAREKETTIPI INTEGER)
AS
declare variable FBelgeHamAdi varchar(60);
declare variable EgemenFirmaID Integer;
declare variable FDokumaHareketi smallint;
declare variable FHamAdi2   type of column Ham.Adi2;
declare variable FHamTipiId type of column Ham.HamTipiId;
begin
  Select EgemenFirmaID from Sistem where Sistem.SistemId = 1 into :EgemenFirmaID;

  BirimIskontoParasal = 0;
  BirimIskontoOrani1  = 0;
  BirimIskontoOrani2  = 0;
  BirimIskontoToplami = 0;
  IskontoOrani = 0;
  ModulID = 0; /*Boya Fatura*/

  for Select BoyaFatura.BoyaFaturaId, BoyaFaturaDetay.BoyaFaturaDetayID, case when coalesce(BoyaSiparisDetay.BelgeHamAdi,'')<>'' then BoyaSiparisDetay.BelgeHamAdi else left(Ham.Adi, 255) end,
             HamStokBirim.BirimID, BoyaFaturaDetay.KdvOrani, BoyaFatura.Tevkifat, BoyaFaturaDetay.Miktar,
             BoyaFaturaDetay.BirimFiyat,(BoyaFaturaDetay.Miktar * BoyaFaturaDetay.BirimFiyat),
             (cast(BoyaFatura.TevkifatOrani1 as Double Precision) / BoyaFatura.TevkifatOrani2) TevkifatOrani, BoyaSiparisDetay.MusteriSiparisNo,
             BoyaSiparisDetay.BelgeHamAdi, BoyaSiparisDetay.IlaveIslemler, SiparisProses.Adi, (Select EskiPartiNoGetir.EskiPartiNoStr from EskiPartiNoGetir(PartiStok.PartiSinifID))
             from BoyaFatura
             left join BoyaFaturaDetay on BoyaFaturaDetay.BoyaFaturaId = BoyaFatura.BoyaFaturaid
             left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = BoyaFaturaDetay.BoyaIrsaliyeDetayid
             left join PartiStok on PartiStok.PartiSinifID = BoyaIrsaliyeDetay.PartiSinifID
             left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
             left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamID
             left join HamStokBirim on HamStokBirim.HamStokBirimId = BoyaFaturaDetay.MiktarBirim
             left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  where (BoyaFatura.EFatura = 1 or :EArsivAktif = 1) and (BoyaFatura.Entegrasyon = 1)and (BoyaFatura.HareketTipiID = 0) and
        (BoyaFatura.FaturaTarihi >= :BaslangicTarihi and BoyaFatura.FaturaTarihi <= :BitisTarihi) and
        ((BoyaFatura.GIBEntegrasyonKullaniciID is null and :GonderilmeDurumu = 0) or
         (BoyaFatura.GIBEntegrasyonKullaniciID is not null and :GonderilmeDurumu = 1) or
         (:GonderilmeDurumu = 2))
  into :Faturaid, :StokHareketID, UrunAdi,
        :BirimId, :Kdv,
        :Tevkifat, :FaturaMiktar,
        :BirimFiyat, :AraToplam, :TevkifatOrani, :MusteriSiparisNo,
        :FBelgeHamAdi, :IlaveIslemler, :SiparisProsesAdi, :EskiPartiNo do
  begin
    KDVTutari = AraToplam * KDV / 100;

    if (Tevkifat = 1) then
    begin
      TevkifatTutari = Yuvarla(KDVTutari * TevkifatOrani, 1000);
    end
    else
    begin
      TevkifatTutari = 0;
      TevkifatOrani = 0;
    end

    GenelToplam = Yuvarla(AraToplam + KDVTutari, 100);
    KDVTutari = Yuvarla(KDVTutari, 100);

    if (EgemenFirmaID = 56 and FBelgeHamAdi is Not Null) then UrunAdi = FBelgeHamAdi; /*RemÖrme*/
    suspend;
  end
  KapAdedi = Null;

  for Select OrmeFatura.OrmeFaturaId, SH.StokHareketId, case when coalesce(OrmeSiparisDetay.BelgeHamAdi,'')<>'' then OrmeSiparisDetay.BelgeHamAdi else left(Ham.Adi, 255) end,
             HamStokBirim.BirimID, SH.Kdv, SH.Tevkifat, SH.KapAdedi, SH.FaturaMiktar,
             SH.NetBirimFiyat, SH.BirimIskontoParasal, SH.BirimIskontoOrani1, SH.BirimIskontoOrani2, SH.BirimIskontoToplami, (SH.FaturaMiktar * SH.NetBirimFiyat),
             (cast(OrmeFatura.TevkifatOrani1 as Double Precision) / OrmeFatura.TevkifatOrani2) TevkifatOrani, OrmeSiparisDetay.MusteriSiparisNo,
             OrmeSiparisDetay.BelgeHamAdi, OrmeFatura.DokumaHareketi, OrmeFatura.HareketTipi,Ham.Adi2,Ham.HamTipiId
             from OrmeFatura
             left join StokHareket SH on SH.OrmefaturaId = OrmeFatura.OrmeFaturaid
             left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = SH.OrmeSiparisDetayID
             left join Ham on Ham.HamID = SH.HamID
             left join HamStokBirim on HamStokBirim.HamStokBirimId = SH.FatEdilecekHamStokBirimID
  where (OrmeFatura.EFatura = 1 or :EArsivAktif = 1) and (OrmeFatura.Entegrasyon = 1)and (OrmeFatura.HareketTipi = 2 or OrmeFatura.HareketTipi = 3 or OrmeFatura.HareketTipi = 14) and
        (OrmeFatura.FaturaTarihi >= :BaslangicTarihi and OrmeFatura.FaturaTarihi <= :BitisTarihi) and SH.HamIade = 0 and
        ((OrmeFatura.GIBEntegrasyonKullaniciID is null and :GonderilmeDurumu = 0) or
         (OrmeFatura.GIBEntegrasyonKullaniciID is not null and :GonderilmeDurumu = 1) or
         (:GonderilmeDurumu = 2)
         )
  into :Faturaid, :StokHareketId, UrunAdi,
        :BirimId, :Kdv, :Tevkifat, :KapAdedi, :FaturaMiktar, :BirimFiyat,
        :BirimIskontoParasal, :BirimIskontoOrani1, :BirimIskontoOrani2, :BirimIskontoToplami, :AraToplam, :TevkifatOrani, :MusteriSiparisNo, :FBelgeHamAdi, :FDokumaHareketi,
        :HareketTipi, :FHamAdi2, :FHamTipiId do
  begin
    ModulID = FDokumaHareketi + 1; /*Örme veya Dokuma Fatura*/

    KDVTutari = AraToplam * KDV / 100;
    if (AraToplam > 0) then IskontoOrani = BirimIskontoToplami / AraToplam * 100;
    else IskontoOrani = 0;

    if (Tevkifat = 1) then
    begin
      TevkifatTutari = Yuvarla(KDVTutari * TevkifatOrani, 1000);
    end
    else
    begin
      TevkifatTutari = 0;
      TevkifatOrani = 0;
    end

    GenelToplam = Yuvarla(AraToplam + KDVTutari, 100);
    KDVTutari = Yuvarla(KDVTutari, 100);

    if (EgemenFirmaID = 56 and FHamTipiId = 0  and HareketTipi=2) then   /*mustafa NS örme*/
    UrunAdi = FHamAdi2  ;

    suspend;
  end

  /* ------------------- */
  BirimIskontoParasal = 0;
  BirimIskontoOrani1  = 0;
  BirimIskontoOrani2  = 0;
  BirimIskontoToplami = 0;
  IskontoOrani = 0;
  ModulID      = 0;
  KapAdedi     = Null;

    for Select Fatura.FaturaId, StokDetay.StokDetayId, Urun.Adi, UrunBirim.BirimID, StokDetay.Kdv, Fatura.Tevkifat/*?BoyaFatura.Tevkifat*/, StokDetay.KapAdedi, StokDetay.Miktar,
             StokDetay.NetBirimFiyat, StokDetay.BirimIskontoToplami/*?BirimIskontoParasal*/, StokDetay.BirimIskontoOrani1, StokDetay.BirimIskontoOrani2, StokDetay.BirimIskontoToplami,
             (StokDetay.Miktar * StokDetay.NetBirimFiyat), (cast(Fatura.TevkifatOrani1 as Double Precision) / Fatura.TevkifatOrani2) TevkifatOrani,
             Null, Null from Fatura
             left join Firma On Firma.FirmaID = Fatura.firmaid
             left join StokDetay on StokDetay.FaturaId =     Fatura.Faturaid
             left join Urun on Urun.UrunId = StokDetay.UrunId
             left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
/* Hareket tipi 1=alis f, 2,Satis , 3 Alis I, 4 Satis İ */
  where (Fatura.EFatura = 1 or :EArsivAktif = 1) and (Fatura.Entegrasyon = 1)and (Fatura.HareketTipi = 1 or  Fatura.HareketTipi = 2 or Fatura.HareketTipi = 3 or Fatura.HareketTipi = 4) and
        (Fatura.FaturaTarihi >= :BaslangicTarihi and Fatura.FaturaTarihi <= :BitisTarihi) and /* StokDetay.HamIade = 0 and*/
        ((Fatura.GIBEntegrasyonKullaniciID is null and :GonderilmeDurumu = 0) or
         (Fatura.GIBEntegrasyonKullaniciID is not null and :GonderilmeDurumu = 1) or
         (:GonderilmeDurumu = 2)
         )
  into :Faturaid, :StokHareketId, :UrunAdi, :BirimId, :Kdv, :Tevkifat, :KapAdedi, :FaturaMiktar, :BirimFiyat,
       :BirimIskontoParasal, :BirimIskontoOrani1, :BirimIskontoOrani2, :BirimIskontoToplami, :AraToplam, :TevkifatOrani, :MusteriSiparisNo, :FBelgeHamAdi do
  begin
    ModulID = 3; /*Stok Fatura*/

    KDVTutari = AraToplam * KDV / 100;

    if (AraToplam > 0 and BirimIskontoToplami > 0) then IskontoOrani = 100 - ((AraToplam / (AraToplam + BirimIskontoToplami)) *  100);
    else IskontoOrani = 0;

    if (Tevkifat = 1) then
    begin
      TevkifatTutari = Yuvarla(KDVTutari * TevkifatOrani, 1000);
    end
    else
    begin
      TevkifatTutari = 0;
      TevkifatOrani = 0;
    end

    GenelToplam = Yuvarla(AraToplam + KDVTutari, 100);
    KDVTutari = Yuvarla(KDVTutari, 100);

    suspend;
  end
end
