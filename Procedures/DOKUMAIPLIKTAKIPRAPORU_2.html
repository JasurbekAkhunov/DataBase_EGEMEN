CREATE PROCEDURE DOKUMAIPLIKTAKIPRAPORU_2(
    ABASLAMATARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMID INTEGER)
RETURNS (
    HAMID INTEGER,
    LOT VARCHAR(150),
    FIILIDENYE DOUBLE PRECISION,
    FIRMAID FIRMAKODUSIZE,
    URETIMPARTINO VARCHAR(30),
    DOKUMAPLANLAMAID INTEGER,
    KUMASHAMID INTEGER,
    KUMASHAMADI ADI255,
    TEZGAHTAATILACAKATKI DOUBLE PRECISION,
    TARAKENI INTEGER,
    ATKICIKISMIKTARI BOYAMIKTARSIZE,
    HESAPLANANMETRE BOYAMIKTARSIZE,
    DOKUNANMIKTAR BOYAMIKTARSIZE,
    TELADEDI INTEGER,
    COZGUCIKISMETRE BOYAMIKTARSIZE,
    COZGUCIKISKILO BOYAMIKTARSIZE)
AS
declare variable shtAlis Integer;
declare variable shtUretimeCikis Integer;
declare variable shtUretim Integer;
declare variable hstIplik Integer;
begin
  /*iki tarih arasi alinan ipliklerin kullanÄ±mlari*/
  select shtAlis, shtUretimeCikis, shtUretim from OrmeStokHareketTipiGetir into :shtAlis, :shtUretimeCikis, :shtUretim;
  select hstIplik from HamTipiGetir into :hstIplik;

  for select Cikislar.HamId, Coalesce(Cikislar.Lot, ''), Coalesce(Cikislar.FiiliDenye, 0), Cikislar.FirmaId, DokumaPlanlama.DokumaPlanlamaId,
             DokumaPlanlama.UretimPartiNo, Ham.HamId, Ham.Adi, DokumaPlanlama.TezgahtaAtilacakAtki,
             DokumaPlanlama.TarakEni, Cikislar.AtkiCikisMiktar,
             (select Sum(DokumaKaliteKontrol.HesaplananMetre)
              from DokumaKaliteKontrol
              where DokumaKaliteKontrol.DokumaPlanlamaId = DokumaPlanlama.DokumaPlanlamaId),
             DokumaPlanlama.GerceklesenMiktar, Ham.TelSayisi, Cikislar.CozguCikisMetre, Cikislar.CozguCikisKilo
      from (select HamId, Lot, FiiliDenye, FirmaId, DokumaPlanlamaId, Sum(AtkiCikisMiktar) AtkiCikisMiktar,
                   Sum(CozguCikisMetre) CozguCikisMetre, Sum(CozguCikisKilo) CozguCikisKilo
            from (select SHAtkiCikis.StokHareketId, StokHareket.HamId, StokHareket.Lot, StokHareket.FiiliDenye,
                         StokHareket.FirmaId, SHAtkiCikis.DokumaPlanlamaId, SHAtkiCikis.Miktar AtkiCikisMiktar,
                         0 CozguCikisMetre, 0 CozguCikisKilo
                  from StokHareket
                  left join Ham on Ham.HamId = StokHareket.HamId
                  left join Depo on Depo.DepoId = StokHareket.DepoId
                  left join OrmeRafDetay on OrmeRafDetay.StokHareketId = StokHAreket.StokHareketId
                  left join OrmeRaf on OrmeRaf.OrmeRafId = OrmeRafDetay.OrmeRafId
                  left join StokHareket SHAtkiCikis on SHAtkiCikis.OrmeRafId = OrmeRaf.OrmeRafId
                  where StokHareket.Tarih >= :ABaslamaTarihi and
                        StokHareket.Tarih <= :ABitisTarihi and
                        StokHareket.HareketTipi = :shtAlis and
                        StokHareket.DokumaHareketi = 1 and
                        StokHareket.HamIade = 0 and
                        (StokHareket.HamId = :AHamId or :AHamId is null) and
                        Ham.HamTipiId = :hstIplik and
                        Depo.CozguDeposu = 0 and
                        Depo.HasilDeposu = 0 and
                        Depo.IndigoDeposu = 0 and
                        SHAtkiCikis.DokumaPlanlamaId is not null and
                        SHAtkiCikis.HareketTipi = :shtUretimeCikis and
                        SHAtkiCikis.HamIade = 0 and
                        SHAtkiCikis.DokumaHareketi = 1
                  union all
                  select distinct SHCozguCikis.StokHareketId, StokHareket.HamId, StokHareket.Lot,
                                  StokHareket.FiiliDenye, StokHareket.FirmaId, SHCozguCikis.DokumaPlanlamaId, 0,
                                  SHCozguCikis.Miktar, SHCozguCikis.Kilo
                  from StokHareket
                  left join Ham on Ham.HamId = StokHareket.HamId
                  left join Depo on Depo.DepoId = StokHareket.DepoId
                  left join OrmeRafDetay on OrmeRafDetay.StokHareketId = StokHareket.StokHareketId
                  left join OrmeRaf on OrmeRaf.OrmeRafId = OrmeRafDetay.OrmeRafId
                  left join StokHareket SHCozguyeCikis on SHCozguyeCikis.OrmeRafId = OrmeRaf.OrmeRafId
                  left join OrmeRaf ORCozgu on OrCozgu.CozguIsEmriKayitId = SHCozguyeCikis.CozguIsEmriKayitId
                  left join StokHareket SHCozguCikis on SHCozguCikis.OrmeRafId = ORCozgu.ormeRafId
                  where StokHareket.Tarih >= :ABaslamaTarihi and
                        StokHareket.Tarih <= :ABitisTarihi and
                        StokHareket.HareketTipi = :shtAlis and
                        StokHareket.DokumaHareketi = 1 and
                        StokHareket.HamIade = 0 and
                        (StokHareket.HamId = :AHamId or :AHamId is null) and
                        Ham.HamTipiId = :hstIplik and
                        Depo.CozguDeposu = 0 and
                        Depo.HasilDeposu = 0 and
                        Depo.IndigoDeposu = 0 and
                        SHCozguyeCikis.CozguIsEmriKayitId is not null and
                        SHCozguyeCikis.HareketTipi = :shtUretimeCikis and
                        SHCozguyeCikis.HamIade = 0 and
                        SHCozguyeCikis.DokumaHareketi = 1 and
                        SHCozguCikis.OrmeRafId is not null and
                        SHCozguCikis.HareketTipi = :shtUretimeCikis and
                        SHCozguCikis.HamIade = 0 and
                        SHCozguCikis.DokumaHareketi = 1)
            group by HamId, Lot, FiiliDenye, FirmaId, DokumaPlanlamaId) Cikislar
      left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = Cikislar.DokumaPlanlamaId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
      left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
      into :HamId, :Lot, :FiiliDenye, :FirmaId, :DokumaPlanlamaId, :UretimPartiNo, :KumasHamId, :KumasHamAdi,
           :TezgahtaAtilacakAtki, :TarakEni, :AtkiCikisMiktari, :HesaplananMetre, :DokunanMiktar, :TelAdedi,
           :CozguCikisMetre, :CozguCikisKilo
  do
  begin
    suspend;
  end
end
