CREATE PROCEDURE ORMEPLANLAMARECETEYAZ(
    AORMEPLANLAMAGRUPID VARCHAR(10),
    AORMEMAKINAID VARCHAR(20),
    AFIRMAID VARCHAR(20))
RETURNS (
    ORMEPLANLAMAID INTEGER,
    ORMEPLANLAMARECETEID INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    KULLANIMORANI DOUBLE PRECISION,
    MELANJKODU VARCHAR(20),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    TAHMINIFIREORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION)
AS
declare variable hstIplik       integer;
begin
  Select hstIplik from HamTipiGetir into :hstIplik;

  For select OrmePlanlamaYaz.OrmePlanlamaId, OrmePlanlamaRecete.OrmePlanlamaReceteId, OrmePlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, OrmePlanlamaRecete.RenkId, Renk.Adi, OrmePlanlamaRecete.KullanimOrani, OrmePlanlamaRecete.CikisMiktari, OrmePlanlamaRecete.IadeMiktari,
             OrmePlanlamaYaz.PlanlananMiktar * OrmePlanlamaRecete.KullanimOrani / 100, OrmePlanlamaYaz.TahminiFireOrani,
             OrmePlanlamaRecete.MelanjKodu, OrmePlanlamaRecete.UreticiID, Uretici.Adi, OrmePlanlamaRecete.Lot from OrmePlanlamaYaz(:AOrmePlanlamaGrupId, :AOrmeMakinaId, :AFirmaId)
  left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = OrmePlanlamaYaz.OrmePlanlamaId
  left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
  left join Renk on Renk.RenkId = OrmePlanlamaRecete.RenkId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  left join Uretici on Uretici.UreticiID = OrmePlanlamaRecete.UreticiID
  where OrmePlanlamaRecete.OrmePlanlamaId is not null
  into :OrmePlanlamaId, :OrmePlanlamaReceteId, :HamId, :HamTipiId, :HamTipiAdi,
       :HamAdi, :RenkId, :RenkAdi, :KullanimOrani, :CikisMiktari, :IadeMiktari, :KullanilacakMiktar,
       :TahminiFireOrani, :MelanjKodu, :UreticiID, :UreticiAdi, :Lot do
  begin
    if (TahminiFireOrani is null) then TahminiFireOrani = 0;

    if (HamTipiId = hstIplik) then KullanilacakMiktar = KullanilacakMiktar / ( (100 - TahminiFireOrani) / 100);
    suspend;
  end
end
