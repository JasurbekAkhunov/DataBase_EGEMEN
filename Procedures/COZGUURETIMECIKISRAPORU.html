CREATE PROCEDURE COZGUURETIMECIKISRAPORU(
    ABASLAMATARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHESAPLAMAYAP EGEBOOLEAN)
RETURNS (
    STOKHAREKETID INTEGER,
    DOKUMAPLANLAMAID INTEGER,
    REFERANSKODU HAMKODUSIZE,
    URETIMECIKISTARIHI TIMESTAMP,
    HAMID INTEGER,
    HAMADI ADI255,
    KUMASHAMID INTEGER,
    KUMASHAMADI ADI255,
    MAKINANO INTEGER,
    DOKUMAMAKINAID MAKINAKODUSIZE,
    DOKUMAMAKINAADI ADI40,
    ATKILOTU VARCHAR(4000),
    LOT VARCHAR(150),
    TELADEDI INTEGER,
    TEZGAHTAATILACAKATKI DOUBLE PRECISION,
    TARAKENI INTEGER,
    TARAKNUMARASI VARCHAR(10),
    BARKODID VARCHAR(30),
    MIKTAR BOYAMIKTARSIZE,
    TOPLAMKKHESAPLANANMETRE BOYAMIKTARSIZE,
    TOPLAMKKMETRE BOYAMIKTARSIZE,
    CEKMEORANI DOUBLE PRECISION,
    ICCEVREMM INTEGER,
    DISCEVREMM INTEGER,
    TURSAYISI INTEGER)
AS
declare variable ShtUretimeCikis integer;
declare variable OrmeRafId       integer;
begin
  Select ShtUretimeCikis from OrmeStokHareketTipiGetir into :shtUretimeCikis;

  for select StokHareket.StokHareketId, HamKumas.ReferansKodu, StokHareket.UretimeCikisTarihi, Ham.Adi, DokumaMakina.MakinaNo, DokumaMakina.DokumaMakinaId, DokumaMakina.Adi,
      StokHareket.Lot, StokHareket.TelAdedi, DokumaPlanlama.TezgahtaAtilacakAtki, DokumaPlanlama.TarakEni, HamKumas.TarakNumarasi, OrmeRaf.BarkodId,
      StokHareket.Miktar, StokHareket.OrmeRafId, StokHareket.HamId, HamKumas.HamId, HamKumas.Adi, StokHareket.DokumaPlanlamaId, StokHareket.IcCevreMM, StokHareket.DisCevreMM, StokHareket.TurSayisi
      from StokHareket
      left join Depo on Depo.DepoId = StokHareket.DepoId
      left join Ham on Ham.HamId = StokHareket.HamId
      left join DokumaMakina on DokumaMakina.DokumaMakinaId = StokHareket.DokumaMakinaId
      left join OrmeRaf on OrmeRaf.OrmeRafId = StokHareket.OrmeRafId
      left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = StokHareket.DokumaPlanlamaId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
      left join Ham HamKumas on HamKumas.HamId = OrmeSiparisDetay.HamId
      where StokHareket.DokumaHareketi = 1 and
      StokHareket.HareketTipi = :ShtUretimeCikis and
      (Depo.CozguDeposu = 1 or Depo.HasilDeposu = 1 or Depo.IndigoDeposu = 1) and
      StokHareket.HamIade = 0 and
      StokHareket.UretimeCikisTarihi <= :ABitisTarihi and StokHareket.UretimeCikisTarihi >= :ABaslamaTarihi
      into :StokHareketId, :ReferansKodu, :UretimeCikisTarihi, :HamAdi, :MakinaNo, :DokumaMakinaId, :DokumaMakinaAdi,
      :Lot, :TelAdedi, :TezgahtaAtilacakAtki, :TarakEni, :TarakNumarasi, :BarkodId, :Miktar,
      :OrmeRafId, :HamId, :KumasHamId, :KumasHamAdi, :DokumaPlanlamaId, :IcCevreMM, :DisCevreMM, :TurSayisi do
  begin
    if (AHesaplamaYap = 1) then
    begin
      select Sum(DokumaKaliteKontrol.Miktar), Sum(HesaplananMetre) from DokumaKaliteKontrol where DokumaKaliteKontrol.OrmeRafId = :OrmeRafId into :ToplamKKMetre, :ToplamKKHesaplananMetre;

      select DokumaPlanlamaLotAtkiLotuGetir.AtkiLotu from DokumaPlanlamaLotAtkiLotuGetir(:DokumaPlanlamaId) into :AtkiLotu;

      CekmeOrani = Null;
      if (ToplamKKHesaplananMetre > 0) then
        CekmeOrani = Yuvarla(100 * (ToplamKKHesaplananMetre-ToplamKKMetre) / ToplamKKHesaplananMetre, 100);
    end

    suspend;
  end
end
