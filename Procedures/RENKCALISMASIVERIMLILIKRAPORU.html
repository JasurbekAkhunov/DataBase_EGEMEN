CREATE PROCEDURE RENKCALISMASIVERIMLILIKRAPORU(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IPLIKATISSAYISI INTEGER,
    CEKTIRMEBOYAATISSAYISI INTEGER,
    HTMERSERIZEBOYAATISSAYISI INTEGER,
    PADPATCHATISSAYISI INTEGER,
    PADPATCHMERSERIZEATISSAYISI INTEGER,
    ELYAFATISSAYISI INTEGER,
    RPTATISSAYISI INTEGER,
    ILAVEATISSAYISI INTEGER,
    SOKUMUZERIBOYAATISSAYISI INTEGER,
    KRITIKCALISMASIATISSAYISI INTEGER,
    TOPLAMATISSAYISI DOUBLE PRECISION,
    IPLIKRENKSAYISI INTEGER,
    CEKTIRMEBOYARENKSAYISI INTEGER,
    HTMERSERIZEBOYARENKSAYISI INTEGER,
    PADPATCHRENKSAYISI INTEGER,
    PADPATCHMERSERIZERENKSAYISI INTEGER,
    ELYAFRENKSAYISI INTEGER,
    TOPLAMRENKSAYISI DOUBLE PRECISION,
    OKEYLIRENKSAYISI INTEGER,
    GIDENATISSAYISI INTEGER,
    OKEYATISSAYISI INTEGER,
    KRITIKATISSAYISI INTEGER,
    OKEYORANI DOUBLE PRECISION,
    TERMINPERFORMANSORANI DOUBLE PRECISION,
    TERMINITUTAN INTEGER)
AS
declare variable LastFirmaId      varchar(20);
declare variable FFirmaId         varchar(20);
declare variable FFirmaAdi        varchar(100);
declare variable FLabReceteId     integer;
declare variable FLastLabReceteId integer;
declare variable FLabReceteGrupId integer;
declare variable FGitti    integer;
declare variable FOkey     integer;
declare variable FRptmi    integer;
declare variable FIlave    integer;
declare variable FSokumUzeriBoya    integer;
declare variable FKritikCalismasi   integer;
declare variable FKritik   integer;
declare variable FTermin   integer;
declare variable AtisSayisi    integer;
declare variable OkeyVar       integer;

begin
  LastFirmaID = '';
  FLastLabReceteId = -1;

  if (ARaporTipi = 0) then
  begin
    For Select LabRecete.FirmaId, Firma.Adi, LabRecete.LabReceteId, LabRecete.LabReceteGrupId, LabReceteAtis.Gitti,
               LabReceteAtis.Okey, LabReceteAtis.Kritik, LabReceteAtis.Termin, LabReceteCalisma.Rptmi,
               LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, Count(*) from LabReceteCalisma
      left join LabRecete on LabRecete.LabReceteID = LabReceteCalisma.LabReceteId
      left join Firma on Firma.FirmaID = LabRecete.FirmaId
      left join LabReceteAtis on LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
      where LabReceteCalisma.CalismaTarihi >= :ABaslangicTarihi and LabReceteCalisma.CalismaTarihi <= :ABitisTarihi and
            LabReceteAtis.LabReceteAtisID is not null
      group by LabRecete.FirmaId, Firma.Adi, LabRecete.LabReceteId, LabRecete.LabRenkGrupId, LabRecete.LabReceteGrupId, LabReceteAtis.Gitti,
               LabReceteAtis.Okey, LabReceteAtis.Kritik, LabReceteAtis.Termin,
               LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi
      order by LabRecete.FirmaId
      into :FFirmaId, :FFirmaAdi, :FLabReceteId, :FLabReceteGrupId, :FGitti, :FOkey, :FKritik, :FTermin, :FRptmi, :FIlave, :FSokumUzeriBoya, :FKritikCalismasi, :AtisSayisi do
    begin
      if (LastFirmaId <> FFirmaId) then
      begin
        if (LastFirmaId <> '') then
        begin
          OkeyOrani = OkeyAtisSayisi / ToplamAtisSayisi * 100;
          TerminPerformansOrani = TerminiTutan / ToplamAtisSayisi * 100;
          ToplamRenkSayisi = IplikRenkSayisi + CektirmeBoyaRenkSayisi + HTMerserizeBoyaRenkSayisi + PadPatchRenkSayisi + PadPatchMerserizeRenkSayisi + ElyafRenkSayisi;
          suspend;
        end

        LastFirmaId = FFirmaId;
        FirmaId = FFirmaId;
        FirmaAdi = FFirmaAdi;
        FLastLabReceteId = -1;

        IplikRenkSayisi        = 0;
        CektirmeBoyaRenkSayisi = 0;
        PadPatchRenkSayisi     = 0;
        HTMerserizeBoyaRenkSayisi     = 0;
        PadPatchMerserizeRenkSayisi     = 0;
        ElyafRenkSayisi        = 0;
        ToplamRenkSayisi       = 0;
        OkeyliRenkSayisi       = 0;



        IplikAtisSayisi        = 0;
        CektirmeBoyaAtisSayisi = 0;
        PadPatchAtisSayisi     = 0;
        HTMerserizeBoyaAtisSayisi = 0;
        PadPatchMerserizeAtisSayisi = 0;
        ElyafAtisSayisi        = 0;
        RptAtisSayisi          = 0;
        IlaveAtisSayisi        = 0;
        SokumUzeriBoyaAtisSayisi = 0;
        KritikCalismasiAtisSayisi = 0;
        ToplamAtisSayisi       = 0;

        GidenAtisSayisi        = 0;
        OkeyAtisSayisi         = 0;
        KritikAtisSayisi       = 0;
        OkeyOrani              = 0.0;
        TerminPerformansOrani  = 0.0;
        TerminiTutan = 0;
      end

      if (FLastLabReceteId <> FLabReceteId) then
      begin
        if (FLabReceteGrupId = 1) then IplikRenkSayisi = IplikRenkSayisi + 1;
        else if (FLabReceteGrupId = 2) then CektirmeBoyaRenkSayisi = CektirmeBoyaRenkSayisi + 1;
        else if (FLabReceteGrupId = 3) then PadPatchRenkSayisi = PadPatchRenkSayisi + 1;
        else if (FLabReceteGrupId = 4) then ElyafRenkSayisi = ElyafRenkSayisi + 1;
        else if (FLabReceteGrupId = 5) then HTMerserizeBoyaRenkSayisi = HTMerserizeBoyaRenkSayisi + AtisSayisi;
        else if (FLabReceteGrupId = 6) then PadPatchMerserizeRenkSayisi = PadPatchMerserizeRenkSayisi + AtisSayisi;

        FLastLabReceteId = FLabReceteId;
        OkeyVar = 0;
      end

      if (FLabReceteGrupId = 1) then IplikAtisSayisi = IplikAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 2) then CektirmeBoyaAtisSayisi = CektirmeBoyaAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 3) then PadPatchAtisSayisi = PadPatchAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 4) then ElyafAtisSayisi = ElyafAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 5) then HTMerserizeBoyaAtisSayisi = HTMerserizeBoyaAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 6) then PadPatchMerserizeAtisSayisi = PadPatchMerserizeAtisSayisi + AtisSayisi;

      if (FRptmi = 1) then  RptAtisSayisi = RptAtisSayisi + AtisSayisi;
      else if (FIlave = 1) then IlaveAtisSayisi = IlaveAtisSayisi + AtisSayisi;
      else if (FSokumUzeriBoya = 1) then SokumUzeriBoyaAtisSayisi = SokumUzeriBoyaAtisSayisi + AtisSayisi;
      else if (FKritikCalismasi = 1) then KritikCalismasiAtisSayisi = KritikCalismasiAtisSayisi + AtisSayisi;


      if (FGitti = 1) then
      begin
        if (FTermin = 1) then TerminiTutan = TerminiTutan + AtisSayisi;
        GidenAtisSayisi = GidenAtisSayisi + AtisSayisi;
      end
      if (FOkey = 1) then
      begin
        OkeyAtisSayisi = OkeyAtisSayisi + AtisSayisi;
        if (OkeyVar = 0) then
        begin
          OkeyVar = 1;
          OkeyliRenkSayisi = OkeyliRenkSayisi + 1;
        end
      end
      if (FKritik = 1) then KritikAtisSayisi = KritikAtisSayisi + AtisSayisi;

      ToplamAtisSayisi = ToplamAtisSayisi  + AtisSayisi;
    end

    if (LastFirmaId <> '') then
    begin
      OkeyOrani = OkeyAtisSayisi / ToplamAtisSayisi * 100;
      TerminPerformansOrani = TerminiTutan / ToplamAtisSayisi * 100;
      ToplamRenkSayisi = IplikRenkSayisi + CektirmeBoyaRenkSayisi + PadPatchRenkSayisi + ElyafRenkSayisi;
      suspend;
    end
  end
  else
  begin
    For Select LabRecete.FirmaId, Firma.Adi, LabRecete.LabReceteId, LabRecete.LabReceteGrupId, LabReceteAtis.Gitti,
               LabReceteAtis.Okey, LabReceteAtis.Kritik, LabReceteAtis.Termin, LabReceteCalisma.Rptmi,
               LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi, Count(*) from LabRecete
      left join Firma on Firma.FirmaID = LabRecete.FirmaId
      left join LabReceteCalisma on LabReceteCalisma.LabReceteID = LabRecete.LabReceteId
      left join LabReceteAtis on LabReceteAtis.LabReceteCalismaId = LabReceteCalisma.LabReceteCalismaId
      where LabRecete.TalepTarihi >= :ABaslangicTarihi and LabRecete.TalepTarihi <= :ABitisTarihi and
            LabReceteAtis.LabReceteAtisID is not null
      group by LabRecete.FirmaId, Firma.Adi, LabRecete.LabReceteId, LabRecete.LabRenkGrupId, LabRecete.LabReceteGrupId, LabReceteAtis.Gitti,
               LabReceteAtis.Okey, LabReceteAtis.Kritik, LabReceteAtis.Termin,
               LabReceteCalisma.Rptmi, LabReceteCalisma.Ilave, LabReceteCalisma.SokumUzeriBoya, LabReceteCalisma.KritikCalismasi
      order by LabRecete.FirmaId
      into :FFirmaId, :FFirmaAdi, :FLabReceteId, :FLabReceteGrupId, :FGitti, :FOkey, :FKritik, :FTermin, :FRptmi, :FIlave, :FSokumUzeriBoya, :FKritikCalismasi, :AtisSayisi do
    begin
      if (LastFirmaId <> FFirmaId) then
      begin
        if (LastFirmaId <> '') then
        begin
          OkeyOrani = OkeyAtisSayisi / ToplamAtisSayisi * 100;
          TerminPerformansOrani = TerminiTutan / ToplamAtisSayisi * 100;
          ToplamRenkSayisi = IplikRenkSayisi + CektirmeBoyaRenkSayisi + HTMerserizeBoyaRenkSayisi + PadPatchRenkSayisi + PadPatchMerserizeRenkSayisi + ElyafRenkSayisi;
          suspend;
        end

        LastFirmaId = FFirmaId;
        FirmaId = FFirmaId;
        FirmaAdi = FFirmaAdi;
        FLastLabReceteId = -1;

        IplikRenkSayisi        = 0;
        CektirmeBoyaRenkSayisi = 0;
        PadPatchRenkSayisi     = 0;
        HTMerserizeBoyaRenkSayisi     = 0;
        PadPatchMerserizeRenkSayisi     = 0;
        ElyafRenkSayisi        = 0;
        ToplamRenkSayisi       = 0;
        OkeyliRenkSayisi       = 0;

        IplikAtisSayisi        = 0;
        CektirmeBoyaAtisSayisi = 0;
        PadPatchAtisSayisi     = 0;
        HTMerserizeBoyaAtisSayisi = 0;
        PadPatchMerserizeAtisSayisi = 0;
        ElyafAtisSayisi        = 0;
        RptAtisSayisi          = 0;
        IlaveAtisSayisi        = 0;
        SokumUzeriBoyaAtisSayisi = 0;
        KritikCalismasiAtisSayisi = 0;
        ToplamAtisSayisi       = 0;

        GidenAtisSayisi        = 0;
        OkeyAtisSayisi         = 0;
        KritikAtisSayisi       = 0;
        OkeyOrani              = 0.0;
        TerminPerformansOrani  = 0.0;
        TerminiTutan = 0;
      end

      if (FLastLabReceteId <> FLabReceteId) then
      begin
        if (FLabReceteGrupId = 1) then IplikRenkSayisi = IplikRenkSayisi + 1;
        else if (FLabReceteGrupId = 2) then CektirmeBoyaRenkSayisi = CektirmeBoyaRenkSayisi + 1;
        else if (FLabReceteGrupId = 3) then PadPatchRenkSayisi = PadPatchRenkSayisi + 1;
        else if (FLabReceteGrupId = 4) then ElyafRenkSayisi = ElyafRenkSayisi + 1;
        else if (FLabReceteGrupId = 5) then HTMerserizeBoyaRenkSayisi = HTMerserizeBoyaRenkSayisi + AtisSayisi;
        else if (FLabReceteGrupId = 6) then PadPatchMerserizeRenkSayisi = PadPatchMerserizeRenkSayisi + AtisSayisi;

        FLastLabReceteId = FLabReceteId;
        OkeyVar = 0;
      end

      if (FLabReceteGrupId = 1) then IplikAtisSayisi = IplikAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 2) then CektirmeBoyaAtisSayisi = CektirmeBoyaAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 3) then PadPatchAtisSayisi = PadPatchAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 4) then ElyafAtisSayisi = ElyafAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 5) then HTMerserizeBoyaAtisSayisi = HTMerserizeBoyaAtisSayisi + AtisSayisi;
      else if (FLabReceteGrupId = 6) then PadPatchMerserizeAtisSayisi = PadPatchMerserizeAtisSayisi + AtisSayisi;

      if (FRptmi = 1) then  RptAtisSayisi = RptAtisSayisi + AtisSayisi;
      else if (FIlave = 1) then IlaveAtisSayisi = IlaveAtisSayisi + AtisSayisi;
      else if (FSokumUzeriBoya = 1) then SokumUzeriBoyaAtisSayisi = SokumUzeriBoyaAtisSayisi + AtisSayisi;
      else if (FKritikCalismasi = 1) then KritikCalismasiAtisSayisi = KritikCalismasiAtisSayisi + AtisSayisi;



      if (FGitti = 1) then
      begin
        if (FTermin = 1) then TerminiTutan = TerminiTutan + AtisSayisi;
        GidenAtisSayisi = GidenAtisSayisi + AtisSayisi;
      end
      if (FOkey = 1) then
      begin
        OkeyAtisSayisi = OkeyAtisSayisi + AtisSayisi;
        if (OkeyVar = 0) then
        begin
          OkeyVar = 1;
          OkeyliRenkSayisi = OkeyliRenkSayisi + 1;
        end
      end
      if (FKritik = 1) then KritikAtisSayisi = KritikAtisSayisi + AtisSayisi;

      ToplamAtisSayisi = ToplamAtisSayisi  + AtisSayisi;
    end

    if (LastFirmaId <> '') then
    begin
      OkeyOrani = OkeyAtisSayisi / ToplamAtisSayisi * 100;
      TerminPerformansOrani = TerminiTutan / ToplamAtisSayisi * 100;
      ToplamRenkSayisi = IplikRenkSayisi + CektirmeBoyaRenkSayisi + HTMerserizeBoyaRenkSayisi + PadPatchRenkSayisi + PadPatchMerserizeRenkSayisi + ElyafRenkSayisi;
      suspend;
    end
  end
