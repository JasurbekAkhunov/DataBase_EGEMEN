CREATE PROCEDURE IPLIKPAKETLEMEKAYDET(
    APARTIKAYITID INTEGER,
    APARTISINIFID INTEGER,
    APAKETLEMEID INTEGER,
    AMAKINAID VARCHAR(10),
    APAKETLEMETARIHI TIMESTAMP,
    AKOLINO INTEGER,
    ADARA NUMERIC(12,3),
    ABRUTKILO NUMERIC(12,3),
    AKULLANICIID INTEGER)
RETURNS (
    PAKETLEMEID INTEGER)
AS
declare variable IplikPaketlemeId  integer;
declare variable PartiStokTartiId  integer;
begin
  Select IplikPaketlemeId, PartiStokTartiId from IplikPaketleme
    where IplikPaketleme.PartiSinifId = :APartiSinifId and IplikPaketleme.PaketlemeID = :APaketlemeId
    into :IplikPaketlemeId, :PartiStokTartiId;

  if (IplikPaketlemeId is not null) then
  begin
    if (PartiStokTartiId is not null) then
    begin
      PaketlemeID = :APaketlemeId;
      suspend;
      exit;
    end
    else
    begin
      Update IplikPaketleme Set KoliNo = :AKoliNo, Dara = :ADara, BrutKilo = :ABrutKilo, KullaniciId = :AKullaniciId where IplikPaketleme.IplikPaketlemeId  = :IplikPaketlemeId;
    end
  end
  else
  begin
    Insert into IplikPaketleme(PartiKayitId, PartiSinifId, MakinaId, PaketlemeId, PaketlemeTarihi, KoliNo, Dara, BrutKilo, InsertKullaniciId, KullaniciId)
                values(:APartiKayitId, :APartiSinifId, :AMakinaId, :APaketlemeId, :APaketlemeTarihi, :AKoliNo, :ADara, :ABrutKilo, :AKullaniciId, :AKullaniciId);
  end
end
