CREATE PROCEDURE IADEREZERVICINHAMSTOKLISTESI(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    AORMECIID INTEGER,
    AIRSALIYENO IRSALIYENOSIZE,
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    RAFDETAYID INTEGER,
    HAMSTOKMASTERID INTEGER,
    SINIF VARCHAR(1),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    PARTISTOKIPTALKAYITID INTEGER,
    ACIKLAMA VARCHAR(30),
    ORMECIADI VARCHAR(30),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    LOT VARCHAR(150),
    TAMIR VARCHAR(5),
    TAMIRTIPIADI VARCHAR(20),
    ESKIPARTINO INTEGER,
    RAFID VARCHAR(10),
    MIKTAR NUMERIC(12,3),
    KALITENO INTEGER,
    KULLANILANMIKTAR NUMERIC(12,3),
    RAFTRANSFERMIKTAR NUMERIC(12,3),
    AKTARMAYACIKANKILO NUMERIC(12,3),
    AKTARMADANIADEEDIENKILO NUMERIC(12,3),
    IADEEDILECEKMIKTAR NUMERIC(12,3),
    IADEEDILENMIKTAR NUMERIC(12,3),
    SATISMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    OLCUADI ADI40,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20))
AS
declare variable TamirX integer;
declare variable HamStokDetayId integer;
declare variable HamStokDetayKalanMiktar   numeric(12,3);
declare variable LastHamStokDetayId integer;
begin
  LastHamStokDetayId = -1;
  for select RafDetay.HamStokDetayId, RafDetay.RafDetayId, HamStokDetay.HamStokMasterId, RafDetay.PartiStokIptalKayitId, HamStokIrsaliye.Sinif, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, Ormeci.Adi,
         hamstokdetay.HamId, Ham.Adi, HamStokMaster.Lot, Ham.HamTipiId, HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartiNo, RafDetay.RafId,
         RafDetay.Miktar, RafDetay.IadeEdilecekMiktar, RafDetay.IadeEdilenMiktar, RafDetay.PartilenenMiktar, RafDetay.RafTransferMiktar,
         RafDetay.SatisMiktar, RafDetay.AktarmayaCikanKilo, RafDetay.AktarmadanIadeEdienKilo, RafDetay.KaliteNo, Olcu.OlcuAdi,
         OrmeSiparisDetay.IplikBoyu1, OrmeSiparisDetay.IplikBoyu2, OrmeSiparisDetay.IplikBoyu3 from HamStokIrsaliye
  left join Ormeci on Ormeci.OrmeciId = HamstokIrsaliye.OrmeciId
  left join HamStokDetay on HamStokDetay.HamStokIrsaliyeid = HamstokIrsaliye.HamStokIrsaliyeid
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join RafDetay on RafDetay.HamStokDetayid = HamStokDetay.HamStokDetayid
  left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
  left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  left join Ham on Ham.HamId = hamstokdetay.hamid
  left join StokHareket on StokHareket.StokHareketId = HamStokMaster.StokHareketId
  left join Olcu on Olcu.OlcuId = StokHareket.OlcuId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = HamStokMaster.OrmeSiparisDetayId
  where (HamstokIrsaliye.FirmaId = :AFirmaId) and
        (HamStokDetay.HamStokMasterID = :AHamStokMasterID or :AHamStokMasterID is null) and
       ((:AOrmeciId is not null and HamstokIrsaliye.OrmeciId = :AOrmeciId) or :AOrmeciId is null) and
       ((:AIrsaliyeNo is not null and HamstokIrsaliye.IrsaliyeNo = :AIrsaliyeNo) or :AIrsaliyeNo is null) and
       ((:AIrsaliyeNo is null and HamstokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi) or :AIrsaliyeNo is not null)
       order by RafDetay.HamStokDetayId, RafDetay.RafDetayId
    into :HamStokDetayId, :RafDetayId, :HamStokMasterId, :PartiStokIptalKayitId, :Sinif, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi, :OrmeciAdi, :HamId, :HamAdi,
         :Lot, :HamTipiId, :TamirX, :TamirTipiAdi, :EskiPartiNo, :RafId,
         :Miktar, :IadeEdilecekMiktar, :IadeEdilenMiktar, :KullanilanMiktar, :RafTransferMiktar, :SatisMiktar,
         :AktarmayaCikanKilo, :AktarmadanIadeEdienKilo, :KaliteNo, :OlcuAdi, :IplikBoyu1, :IplikBoyu2, :IplikBoyu3 do
  begin
    if (HamStokDetayID <> LastHamStokDetayID) then
    begin
      select KalanMiktar from HamStokDetayKalaniGetir(:HamStokDetayId) into :HamStokDetayKalanMiktar;
      LastHamStokDetayID = HamStokDetayId;
    end

    if (TamirX = 1) then Tamir = 'Tamir';

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';

    if (KullanilanMiktar is null) then KullanilanMiktar = 0;

    KalanMiktar = Miktar - IadeEdilecekMiktar - IadeEdilenMiktar - KullanilanMiktar - RafTransferMiktar - SatisMiktar - AktarmayaCikanKilo + AktarmadanIadeEdienKilo;

    if (PartiStokIptalKayitId is not null) then Aciklama = 'Parti İptal';
    else Aciklama = '';

    if (KalanMiktar > 0 and HamStokDetayKalanMiktar > 0) then
    begin
      if (KalanMiktar > HamStokDetayKalanMiktar) then
      begin
        KalanMiktar = HamStokDetayKalanMiktar;
        HamStokDetayKalanMiktar = 0;
      end
      else HamStokDetayKalanMiktar = HamStokDetayKalanMiktar - KalanMiktar;

      suspend;
    end
  end
end
