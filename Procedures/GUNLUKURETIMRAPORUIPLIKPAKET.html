CREATE PROCEDURE GUNLUKURETIMRAPORUIPLIKPAKET(
    AMAKINAGRUBU VARCHAR(4),
    AMAKINARAPORGRUPID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP)
RETURNS (
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    MAKINAGRUBU VARCHAR(10),
    MAKINAGRUPADI VARCHAR(30),
    MAKINARAPORGRUPID INTEGER,
    URETIMTOPLAMI NUMERIC(12,3),
    VARDIYA1KILO DOUBLE PRECISION,
    VARDIYA2KILO DOUBLE PRECISION,
    VARDIYA3KILO DOUBLE PRECISION,
    AYLIKURETIMTOPLAMI DOUBLE PRECISION,
    GUNLUKKAPASITE DOUBLE PRECISION)
AS
declare variable FMiktar        Double Precision;
declare variable FCikisTarihi TimeStamp;
declare variable VardiyaSayisi      integer;
declare variable Vardiya           integer;
declare variable T1                Double Precision;
begin
  Select VardiyaSayisi from sistem where sistemid = 1 into :VardiyaSayisi;
  if (VardiyaSayisi = 0 or VardiyaSayisi is null) then VardiyaSayisi = 3;

  For select IplikPaketleme.MakinaId, Makina.Adi, Makina.GunlukKapasite, Makina.MakinaGrubu, MakinaGrup.Adi,Makina.RaporGrupId,
             Sum(IplikPaketleme.NetKilo) from IplikPaketleme
    left join Makina on Makina.MakinaId = IplikPaketleme.MakinaId
    left join MakinaGrup on MakinaGrup.MakinaGrupId = Makina.MakinaGrubu
    left join Parti on Parti.PartiKayitId = IplikPaketleme.PartiKayitId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join PartiStok on PartiStok.PartiSinifID = IplikPaketleme.PartiSinifId
    where IplikPaketleme.PaketlemeTarihi >= :AAylikBaslangic and IplikPaketleme.PaketlemeTarihi <= :AAylikBitis and
          (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
          (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null)
    group by IplikPaketleme.MakinaId, Makina.Adi, Makina.GunlukKapasite, Makina.MakinaGrubu, MakinaGrup.Adi,Makina.RaporGrupId
  into :MakinaId, :MakinaAdi, :GunlukKapasite, :MakinaGrubu, :MakinaGrupAdi, :MakinaRaporGrupId, :AylikUretimToplami do
  begin

    UretimToplami    = 0;
    Vardiya1Kilo     = 0;
    Vardiya2Kilo     = 0;
    Vardiya3Kilo     = 0;

    for select IplikPaketleme.NetKilo, IplikPaketleme.PaketlemeTarihi from IplikPaketleme
      where IplikPaketleme.MakinaID = :MakinaId and
      IplikPaketleme.PaketlemeTarihi >= :ABaslangicTarihi and IplikPaketleme.PaketlemeTarihi <= :ABitisTarihi
      order by IplikPaketleme.MakinaId
    into :FMiktar, :FCikistarihi do
    begin
      UretimToplami = UretimToplami + FMiktar;

      T1 = FCikisTarihi - ABaslangicTarihi;
      vardiya = VardiyaHesapla(T1, VardiyaSayisi);

      if (Vardiya = 1) then Vardiya1Kilo       = Vardiya1Kilo + FMiktar;
      else if (Vardiya = 2) then Vardiya2Kilo  = Vardiya2Kilo + FMiktar;
      else if (Vardiya >= 3) then Vardiya3Kilo = Vardiya3Kilo + FMiktar;
    end

    suspend;
  end
end
