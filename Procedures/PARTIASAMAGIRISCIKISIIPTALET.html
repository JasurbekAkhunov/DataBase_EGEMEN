CREATE PROCEDURE PARTIASAMAGIRISCIKISIIPTALET(
    APARTIASAMASTOKDETAYID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    PARTIHAREKETLOGID INTEGER)
AS
declare variable RV integer;
declare variable FHamSTokMasterId      integer;
declare variable HamSTokMasterIdStr    varchar(100);
declare variable PartiSinifId          integer;
begin

  Select PartiAsamaStokDetay.PartiSinifId from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    where PartiAsamaStokDetay.PartiAsamaStokDetayId = :APartiAsamaStokDetayId
    into :PartiSinifId;

  HamSTokMasterIdStr = '';

  for select HamStokDetay.HamStokMasterId from PartiStokDetay
    left join BoyaSiparisRezerv on BoyaSiparisRezerv.BoyaSiparisRezervId = PartiStokDetay.BoyaSiparisRezervId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
    where PartiStokDetay.PartiSinifId = :PartiSinifId
    group by HamStokDetay.HamStokMasterId
    into :FHamStokMasterId do
  begin
    if (Char_Length(HamSTokMasterIdStr) + Char_Length(FHamStokMasterId) < 100) then
    begin
     if (HamSTokMasterIdStr <> '' ) then HamSTokMasterIdStr = HamSTokMasterIdStr || cast(', ' as varchar(2) character set UTF8);
     HamSTokMasterIdStr = HamSTokMasterIdStr || FHamStokMasterId;
    end
  end

  PartiHareketLogId = Gen_ID(PartiHareketLog_Id_Gen, 1);

  insert into PartiHareketLog(PartiHareketLogId, PartiYil, PartiId, HamAdiToplami, AsamaId, ReceteId, BoyaSiparisYil, BoyaSiparisId,
                              BoyaSiparisDetayId, MakinaId, Kilo, GirisPersonelId, GirisTarihi, CikisPersonelId, CikisTarihi,
                              GirisCikis, KullaniciId, HamSTokMasterIdStr)
    Select :PartiHareketLogId, Parti.PartiYil, Parti.PartiId, PartiStok.HamAdiToplami, PartiAsamalari.AsamaId, PartiAsamalari.ReceteId,
            BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId, PartiAsamaStokDetay.MakinaId, PartiAsamaStokDetay.Kilo,
            PartiAsamaStokDetay.GirisPersonelId, PartiAsamaStokDetay.GirisTarihi,
            PartiAsamaStokDetay.CikisPersonelId, PartiAsamaStokDetay.CikisTarihi, 2, :AKullaniciId, :HamSTokMasterIdStr from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    Left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where PartiAsamaStokDetay.PartiAsamaStokDetayID = :APartiAsamaStokDetayID;

  Select Count(*) from PartiAsamaCikisiIptalEt(:APartiAsamaStokDetayId, :AKullaniciId, 1) into :RV;
  Select Count(*) from PartiAsamaGirisiIptalEt(:APartiAsamaStokDetayId, :AKullaniciId, 1) into :RV;
  suspend;
end
