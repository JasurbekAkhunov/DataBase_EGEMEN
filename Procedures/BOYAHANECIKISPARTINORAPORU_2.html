CREATE PROCEDURE BOYAHANECIKISPARTINORAPORU_2(
    ACIKISPARTINO VARCHAR(30),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    ARAPORTIPI INTEGER,
    ASIPARISFIRMAID VARCHAR(20))
RETURNS (
    CIKISPARTINO VARCHAR(30),
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(50),
    TARIH TIMESTAMP,
    IRSALIYENO IRSALIYENOSIZE,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    URETIMPARTINO VARCHAR(30),
    BOYATALIMATNO VARCHAR(30),
    LOT VARCHAR(150),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    MIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    HAMIADE INTEGER)
AS
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
begin
  select shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim, ShtUretim from OrmeStokHareketTipiGetir
       into :shtFasonUretimeCikis,  :shtUretimeCikis, :shtFasonUretim, :ShtUretim;

  For Select BoyahaneCikisPartiNoRaporu_1.CikisPartiNo, StokHareket.Tarih, OrmeIrsaliye.IrsaliyeNo, BoyaSiparis.BoyaSiparisYil,
             BoyaSiparis.BoyaSiparisId, StokHareket.UretimPartiNo, BoyaTalimat.BoyaTalimatNo, BoyaSiparis.FirmaId, SipFirma.Adi, Stokhareket.HamId, Ham.Adi, Renk.Adi, Stokhareket.DesenId, Desen.Adi,
             DesenVaryant.VaryantAdi, Stokhareket.Lot, Stokhareket.Miktar, Stokhareket.BrutMiktar, Stokhareket.HamIade from BoyahaneCikisPartiNoRaporu_1(:ACikisPartiNo, :ABaslangicTarihi, :ABitisTarihi, :AFirmaId, :AHamId, :ARaporTipi, :ASiparisFirmaId)
  left join StokHareket on StokHareket.CikisPartiNo = BoyahaneCikisPartiNoRaporu_1.CikisPartiNo
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
  left join Firma SipFirma on SipFirma.FirmaId = BoyaSiparis.FirmaId
  left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
  left join Ham on Ham.HamId = StokHareket.HamId
  left join Renk on Renk.RenkId = StokHareket.RenkId
  left join Desen on Desen.DesenId = StokHareket.DesenId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = StokHareket.DesenVaryantId
  Left join BoyaTalimat on BoyaTalimat.BoyaTalimatKayitID = StokHareket.BoyaTalimatKayitId
  where StokHareket.StokHareketID is not null
  into :CikisPartiNo, :Tarih, :IrsaliyeNo, :BoyaSiparisYil, :BoyaSiparisId, :UretimPartiNo, :BoyaTalimatNo, :SiparisFirmaId, :SiparisFirmaAdi, :HamId, :HamAdi,
       :RenkAdi, :DesenId, :DesenAdi, :VaryantAdi, :Lot, :Miktar, :BrutMiktar, :HamIade do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    else if (HareketTipi = shtFasonUretim and HamIade = 0) then HareketTipiStr = 'Fason Üretim';
    else if (HareketTipi = shtFasonUretim and HamIade = 1) then HareketTipiStr = 'Fason Üretim İade';
    else if (HareketTipi = shtUretim and HamIade = 0) then HareketTipiStr = 'Üretim';
    else if (HareketTipi = shtUretim and HamIade = 1) then HareketTipiStr = 'Üretim İade';
    else HareketTipiStr = 'HATALI !!!';

    suspend;
  end
end
