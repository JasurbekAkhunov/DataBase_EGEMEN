         (PartiStokTarti.KaliteNo = :AKaliteNo or :AKaliteNo is null)
    group by Partistok.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, PartiStok.PartiSinifId, Parti.ReceteId, Parti.ModelNo, PartiStok.HamAdiToplami, PartiStok.TamamiTartildi,
             Parti.ParcaliSevk, PartiStok.Kilo, PartiStok.NetKilo, PartiStok.BrutSevkKilo, PartiStok.NetSevkKilo,

             PartiStok.FirmaSevkAdresiId, FirmaSevkAdresi.AdresAdi,
             PartiStok.FireOnayiIstensin, PartiStok.FireOrani, PartiStok.FireOnayiKullaniciId, FOK.Adi, PartiStok.FireOnayTarihi, Left(PartiStok.FireOnayNotu, char_length(PartiStok.FireOnayNotu)),
             PartiStok.FireSebepTipiId, FireSebepTipi.Adi, BoyaSiparis.BaskaFirmayaSatilabilir, PartistokTarti.KaliteNo, PartiStok.PartiSiraNo,
             BoyaSiparisDetay.DesenId, BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.LabReceteAtisId, Kalite.KaliteAdi, DesenVaryant.VaryantAdi
    order by Parti.PartiYil, Parti.PartiId

         :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :MusteriSiparisNo,
         :FirmaSevkAdresiId, :FirmaSevkAdresiAdi,
         :FireOnayiIstensin, :FireOrani, :FireOnayiKullaniciId, :FireOnayiKullaniciAdi, :FireOnayTarihi, :FireOnayNotuStr,
         :FireSebepTipiId, :FireSebepTipiAdi, :PaketlemeVar,
         :SevkeHazirTahminiBrutKilo, :SevkeHazirTahminiBrutMetre, :SevkeHazirKilo, :SevkeHazirMetre, :SevkeHazirKapAdedi, :SevkeHazirAdet, :BaskaFirmayaSatilabilir, :KaliteNo, :PartiSiraNo,
