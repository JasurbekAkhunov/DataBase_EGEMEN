CREATE PROCEDURE GUNLUKURETIMRAPORUTAMIRDETAY(
    AMAKINAGRUBU VARCHAR(4),
    AMAKINARAPORGRUPID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    MAKINAID VARCHAR(10),
    TAMIRDUZELTMETIPIID VARCHAR(10),
    TAMIRDUZELTMETIPIADI VARCHAR(30),
    UCRETLI INTEGER,
    ICTAMIR INTEGER,
    MIKTAR DOUBLE PRECISION)
AS
begin
  for select PartiAsamaStokDetay.MakinaId, PartiTamir.TamirDuzeltmeTipiId, TamirDuzeltmeTipi.Adi, TamirCozumSekli.Ucretli, Sum(PartiAsamaStokDetay.Kilo) from PartiAsamaStokDetay
        left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
        left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
        left join TamirDuzeltmeTipi on TamirDuzeltmeTipi.TamirDuzeltmeTipiId = PartiTamir.TamirDuzeltmeTipiId
        left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
        where PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi <= :ABitisTarihi and
        (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
        (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null) and
        PartiAsamalari.PartiTamirId is not Null
        Group By PartiAsamaStokDetay.MakinaId, PartiTamir.TamirDuzeltmeTipiId,  TamirDuzeltmeTipi.Adi, TamirCozumSekli.Ucretli
      into :MakinaID, :TamirDuzeltmeTipiId, :TamirDuzeltmeTipiAdi, :Ucretli, :Miktar do
  begin
    IcTamir = 1;
    Suspend;
  end
  for select PartiAsamaStokDetay.MakinaId, DisTamir.TamirDuzeltmeTipiId, TamirDuzeltmeTipi.Adi, DisTamirCozumSekli.Ucretli, Sum(PartiAsamaStokDetay.kilo) from PartiAsamaStokDetay
        left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
        left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
        left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
        left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        left join DisTamir on DisTamir.HamStokMasterId = HamStokDetay.HamStokMasterId
        left join TamirDuzeltmeTipi on TamirDuzeltmeTipi.TamirDuzeltmeTipiId = DisTamir.TamirDuzeltmeTipiId
        left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliid = HamStokMAster.DisTamirCozumSekliId
        where  PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.Cikistarihi <= :ABitisTarihi and
        (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
        (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null) and
        DisTamir.HamStokMasterId is not Null
        Group By PartiAsamaStokDetay.MakinaId, DisTamir.TamirDuzeltmeTipiId, TamirDuzeltmeTipi.Adi, DisTamirCozumSekli.Ucretli
      into :MakinaID, :TamirDuzeltmeTipiId, :TamirDuzeltmeTipiAdi, :Ucretli, :Miktar do
  begin
    IcTamir = 0;
    Suspend;
  end

end
