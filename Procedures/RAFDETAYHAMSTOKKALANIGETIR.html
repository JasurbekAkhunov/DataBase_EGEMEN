CREATE PROCEDURE RAFDETAYHAMSTOKKALANIGETIR(
    ARAFDETAYID INTEGER)
RETURNS (
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    KALANADET INTEGER,
    IADEEDILECEKKILO NUMERIC(12,3),
    IADEEDILENKILO NUMERIC(12,3),
    IADEEDILECEKMETRE NUMERIC(12,3),
    IADEEDILENMETRE NUMERIC(12,3),
    IADEEDILECEKADET INTEGER,
    IADEEDILENADET INTEGER,
    AKTARMAYACIKANKILO NUMERIC(12,3),
    AKTARMADANIADEEDIENKILO NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    PARTILENENADET INTEGER,
    RAFTRANSFERKILO NUMERIC(12,3),
    RAFTRANSFERMETRE NUMERIC(12,3),
    SATISKILO NUMERIC(12,3),
    SATISMETRE NUMERIC(12,3))
AS
declare variable Kilo                   Numeric(12,3);
declare variable Metre                  Numeric(12,3);
declare variable Adet                   integer;
begin
  select RafDetay.Kilo, RafDetay.Metre, RafDetay.Adet, RafDetay.IadeEdilecekKilo, RafDetay.IadeEdilenKilo,
         RafDetay.IadeEdilecekMetre, rafDetay.IadeEdilenMetre, RafDetay.IadeEdilecekAdet, rafDetay.IadeEdilenAdet,
         RafDetay.PartilenenKilo, RafDetay.PartilenenMetre, RafDetay.PartilenenAdet,
         RafDetay.RafTransferKilo, RafDetay.RafTransferMetre,
         RafDetay.AktarmayaCikanKilo, RafDetay.AktarmadanIadeEdienKilo, RafDetay.SatisKilo, RafDetay.SatisMetre from RafDetay
  where RafDetay.RafDetayId = :ARafDetayId
  into :Kilo, :Metre, :Adet, :IadeEdilecekKilo, :IadeEdilenKilo, :IadeEdilecekMetre, :IadeEdilenMetre, :IadeEdilecekAdet, :IadeEdilenAdet,
       :PartilenenKilo, :PartilenenMetre, :PartilenenAdet, :RafTransferKilo, :RafTransferMetre,
       :AktarmayaCikanKilo, :AktarmadanIadeEdienKilo,
       :SatisKilo, :SatisMetre;

  KalanKilo = Kilo + AktarmadanIadeEdienKilo - AktarmayaCikanKilo - PartilenenKilo - RafTransferKilo - IadeEdilecekKilo - IadeEdilenKilo - SatisKilo;
  KalanMetre = Metre - PartilenenMetre - RafTransferMetre - IadeEdilecekMetre - IadeEdilenMetre - SatisMetre;
  KalanAdet = Adet - PartilenenAdet - IadeEdilecekAdet - IadeEdilenAdet;
  suspend;
end
