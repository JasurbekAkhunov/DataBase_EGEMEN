CREATE PROCEDURE HAMSTOGUNAKIBETIKALAN(
    AHAMSTOKMASTERID INTEGER,
    AHAMSTOKDETAYID INTEGER)
RETURNS (
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    KULLANILANKILO NUMERIC(12,3),
    KULLANILANMETRE NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    REZERVEDILENKILO NUMERIC(12,3),
    REZERVEDILENMETRE NUMERIC(12,3),
    RAFSTR VARCHAR(50))
AS
declare variable HamStokDetayId integer;
declare variable FKullanilanKilo       Numeric(12,3);
declare variable FKullanilanMetre     Numeric(12,3);
declare variable PartiIptalKilo        Numeric(12,3);
declare variable PartiIptalMetre       Numeric(12,3);
declare variable FRafId                varchar(10);
declare variable FRezervEdilenKilo     Numeric(12,3);
declare variable FRezervEdilenMetre    Numeric(12,3);
begin
  if (AHamStokDetayId is not null) then
    Select Kilo, Metre, HamStokMasterId from HamStokDetay
      where HamStokDetay.HamStokDetayID = :AHamStokDetayID into :Kilo, :Metre, :AHamStokMasterId;
  else
    Select Kilo, Metre from HamStokMaster where HamStokMaster.HamStokMasterID = :AHamStokMasterID into :Kilo, :Metre;

  KalanKilo = Kilo;
  KalanMetre = Metre;
  RafStr = '';
  FRezervEdilenKilo = 0;
  FRezervEdilenMetre = 0;
  RezervEdilenKilo = 0;
  RezervEdilenMetre = 0;
  KullanilanKilo = 0;
  KullanilanMetre = 0;

  For Select HamStokDetay.HamStokDetayId, HamStokDetay.RezervEdilenKilo, HamStokDetay.RezervEdilenMetre from HamStokDetay where HamStokDetay.HamStokMasterId = :AHamStokMasterId and (HamStokDetay.HamStokDetayId = :AHamStokDetayId or :AHamStokDetayId is null)
  into :HamStokDetayId, :FRezervEdilenKilo, :FRezervEdilenMetre do
  begin
    Select sum(RafDetay.PartilenenKilo + RafDetay.IadeEdilenKilo+ RafDetay.SatisKilo),
           sum(RafDetay.PartilenenMetre+ RafDetay.IadeEdilenMetre + RafDetay.SatisMetre) from RafDetay
     where RafDetay.HamStokDetayId = :HamStokDetayId
     into :FKullanilanKilo, :FKullanilanMetre;

    Select sum(RafDetay.Kilo), sum(RafDetay.Metre) from RafDetay
    where RafDetay.HamStokDetayId = :HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
    into :PartiIptalKilo, :PartiIptalMetre;

    if (FKullanilanKilo is null) then FKullanilanKilo = 0;
    if (FKullanilanMetre is null) then FKullanilanMetre = 0;
    if (PartiIptalKilo is null) then PartiIptalKilo = 0;
    if (PartiIptalMetre is null) then PartiIptalMetre = 0;

    KalanKilo = KalanKilo - FKullanilanKilo + PartiIptalKilo;
    KalanMetre = KalanMetre - FKullanilanMetre + PartiIptalMetre;

    RezervEdilenKilo = RezervEdilenKilo + FRezervEdilenKilo;
    RezervEdilenMetre = RezervEdilenMetre + FRezervEdilenMetre;

    KullanilanKilo = KullanilanKilo + FKullanilanKilo;
    KullanilanMetre = KullanilanMetre + FKullanilanMetre;

    for select RafDetay.RafId from HamStokDetay
      left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId
      where HamStokDetay.HamStokDetayId = :HamStokDetayId and RafDetay.RafDetayID is not null
      Group By RafDetay.RafId
      into :FRafId do
    begin
      if (Char_Length(RafStr) + Char_Length(FRafId) < 48) then
      begin
       if (RafStr <> '' ) then RafStr = RafStr || cast(', ' as varchar(2) character set UTF8);
       RafStr = RafStr || FRafId;
      end
    end

  end

  suspend;
end
