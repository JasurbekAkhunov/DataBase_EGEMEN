CREATE PROCEDURE FASONALISSTOKLISTESI(
    AFIRMAID VARCHAR(20))
RETURNS (
    LOT VARCHAR(30),
    FLAMKODU VARCHAR(20),
    MIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    IPLIKDEPOID VARCHAR(10),
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20))
AS
declare variable shtFasonAlis     Smallint;
begin
  Select shtFasonAlis from IplikStokHareketTipiGetir into :shtFasonAlis;

  for Select  IplikIrsaliyeDetay.HamId, Ham.Adi, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.FlamKodu, IplikStok.IplikDepoID,
       Sum(IplikStok.Miktar), Sum(IplikStok.NetMiktar) from IplikIrsaliye
    left Join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeID = IplikIrsaliye.IplikIrsaliyeID
    left Join IplikStok on IplikStok.IplikIrsaliyeDetayId = IplikIrsaliyeDetay.IplikIrsaliyeDetayID
    left join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
    where IplikIrsaliye.HareketTipi = :shtFasonAlis and
          IplikStok.CikisIplikIrsaliyeDetayId is Null and
          (IplikIrsaliye.FirmaID = :AFirmaId)
    Group By IplikIrsaliyeDetay.HamId, Ham.Adi, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.FlamKodu, IplikStok.IplikDepoID
    into :HamId, :HamAdi, :Lot, :FlamKodu, :IplikDepoID, :Miktar, :NetMiktar do
  begin
    suspend;
  end
end
