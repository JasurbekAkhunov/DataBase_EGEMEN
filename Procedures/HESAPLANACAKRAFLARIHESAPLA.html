CREATE PROCEDURE HESAPLANACAKRAFLARIHESAPLA
AS
declare variable OrmeRafId integer;
begin
  for select OrmeRafId from HesaplanacakRaflar into :OrmeRafId do
  begin
    update OrmeRaf set IslemDevamEdiyor = 0 where OrmeRafId = :OrmeRafId;
    execute procedure RaflamayiSumIleOrganizeEt(:OrmeRafId);
    delete from HesaplanacakRaflar where OrmeRafId = :OrmeRafId;
  end
end
