CREATE PROCEDURE ORMEDOKUMAIMALATAARAVERBASLAT(
    AORMEMAKINAID MAKINAKODUSIZE,
    ADOKUMAMAKINAID MAKINAKODUSIZE,
    APERSONELID INTEGER,
    AARAVERMESEBEBIID INTEGER,
    ABASLADIBITTI INTEGER)
RETURNS (
    STATUS INTEGER,
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    ARAVERMESEBEBIID INTEGER,
    ARAVERMESEBEBIADI VARCHAR(50),
    TARIH TIMESTAMP,
    MESAJ VARCHAR(255))
AS
declare variable BasladiOrmeMakinaDurumuId  integer;
declare variable BittiOrmeMakinaDurumuId    integer;
declare variable ImalataAraVermeId      integer;
declare variable KayitSayisi            integer;
declare variable BaslamaTarihi    TimeStamp;
declare variable GecenSure        Double Precision;
begin
  if (AOrmeMakinaId is Null and ADokumaMakinaId is Null) then
  begin
    Status = 6;
    mesaj = 'Makina Kodu Boş Olamaz...';
    suspend;
    Exit;
  end
  else if (AOrmeMakinaId is not Null and ADokumaMakinaId is not Null) then
  begin
    Status = 7;
    mesaj = 'Parametre Hatası, İki Makina Kodu Birden Gönderilmiş...';
    suspend;
    Exit;
  end

  if (ABasladiBitti = 1) then
  begin
    Select max(ImalataAraVermeId) from ImalataAraVerme
      where (ImalataAraVerme.OrmeMakinaId = :AOrmeMakinaId or ImalataAraVerme.DokumaMakinaId = :ADokumaMakinaId) and
             BittiOrmeMakinaDurumuId is null
    into :ImalataAraVermeId;

    if (ImalataAraVermeId is null) then
    begin
      Select AraVermeSebebi.Adi from AraVermeSebebi where AraVermeSebebi.AraVermeSebebiId = :AAraVermeSebebiId into :AraVermeSebebiAdi;
      if (AraVermeSebebiAdi is null) then
      begin
        Status = 5;
        mesaj = 'Ara Verme Sebebi Kayitli Degil..';
      end
      else
      begin
        BasladiOrmeMakinaDurumuId = gen_Id(OrmeMakinaDurumu_id_gen, 1);

        Tarih = Current_TimeStamp(0);

        insert into OrmeMakinaDurumu(OrmeMakinaDurumuID, OrmeMakinaId, DokumaMakinaId, Tarih, ImalataAraVerildi, Tur) values(:BasladiOrmeMakinaDurumuId, :AOrmeMakinaId, :ADokumaMakinaId, :Tarih, 1, 0);

        insert into ImalataAraVerme(OrmeMakinaId, DokumaMakinaId, BasladiOrmeMakinaDurumuId, AraVerdirenPersonelId, AraVermeSebebiId)
                          values(:AOrmeMakinaId, :ADokumaMakinaId, :BasladiOrmeMakinaDurumuId, :APersonelId, :AAraVermeSebebiId);
        Status = 1;
        mesaj = 'Kayit Yapıldı';
      end
    end
    else
    begin
      Status = 2;
      mesaj = 'imalata daha once ara verilmis';

      Select ImalataAraVerme.AraVerdirenPersonelId, Personel.Adi, ImalataAraVerme.AraVermeSebebiId, AraVermeSebebi.Adi, ImalataAraVerme.BaslamaTarihi from ImalataAraVerme
        left join Personel on Personel.PersonelId = ImalataAraVerme.AraVerdirenPersonelId
        left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
        where ImalataAraVerme.ImalataAraVermeId = :ImalataAraVermeId
        into :PersonelId, :PersonelAdi, :AraVermeSebebiId, :AraVermeSebebiAdi, :Tarih;
    end
  end
  else if (ABasladiBitti = 2) then
  begin
    Select ImalataAraVermeId, ImalataAraVerme.BaslamaTarihi from ImalataAraVerme
        where (ImalataAraVerme.OrmeMakinaId = :AOrmeMakinaId or ImalataAraVerme.DokumaMakinaId = :ADokumaMakinaId) and ImalataAraVerme.BittiOrmeMakinaDurumuId is null
        into :ImalataAraVermeId, :BaslamaTarihi;

    if (ImalataAraVermeId is not null) then
    begin
      Tarih = Current_TimeStamp(0);

      BittiOrmeMakinaDurumuId = gen_Id(OrmeMakinaDurumu_id_gen, 1);
      insert into OrmeMakinaDurumu(OrmeMakinaDurumuID, OrmeMakinaId, DokumaMakinaId, Tarih, ImalataAraVerildi, Tur) values(:BittiOrmeMakinaDurumuId, :AOrmeMakinaId, :ADokumaMakinaId, :Tarih, 2, 0); /* ara bitirildi */

      Update ImalataAraVerme Set BittiOrmeMakinaDurumuId = :BittiOrmeMakinaDurumuId, ArayiBitirenPersonelId = :APersonelId, BitisTarihi = :Tarih where ImalataAraVerme.ImalataAraVermeID = :ImalataAraVermeId;

      Status = 1;
      mesaj = 'Kayit Yapıldı';

/*      GecenSure = Tarih - BaslamaTarihi;

      Update PartiAsamaStokDetay set AraVermeSuresi = AraVermeSuresi + :GecenSure
        where PartiAsamaStokDetay.MakinaId = :AMakinaId and PartiAsamaStokDetay.CikisTarihi is null and PartiAsamaStokDetay.GirisTarihi is not null;*/
    end
    else
    begin
      Status = 3;
      mesaj = 'imalata ara verilmemiş';
    end
  end
  else
  begin
    Status = 4;
    mesaj = 'Parametre Hatalı..Egemen Yazılıma Haber ver.. ';
  end
  suspend;
end
