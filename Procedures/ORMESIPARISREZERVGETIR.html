CREATE PROCEDURE ORMESIPARISREZERVGETIR(
    AORMESIPARISDETAYID INTEGER)
RETURNS (
    ORMESIPARISREZERVID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    ORMESIPARISDETAYRECETEID INTEGER,
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    URETIMPARTINO VARCHAR(30),
    RAFID VARCHAR(10),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    LOT VARCHAR(150),
    FASONSTOK INTEGER,
    FASONSTOKSTR VARCHAR(5),
    MIKTAR NUMERIC(12,3),
    KULLANILANMIKTAR NUMERIC(12,3),
    KALITENO INTEGER,
    MELANJKODU VARCHAR(20))
AS
declare variable CikisMiktari  Numeric(12,3);
declare variable CikisKapAdedi integer;
declare variable hstIplik       integer;
declare variable hstLycra       integer;
begin
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  for select OrmeSiparisRezerv.OrmeSiparisRezervID, OrmeSiparisRezerv.OrmeSiparisDetayID, OrmeSiparisRezerv.OrmeSiparisDetayReceteID, OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.HamId, Ham.Adi, OrmeRaf.UretimPartiNo,
             OrmeRaf.RafId, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.Lot, OrmeRaf.FasonStok,
             OrmeRaf.KaliteNo, OrmeRaf.MelanjKodu, OrmeSiparisRezerv.Miktar, OrmeSiparisRezerv.KullanilanMiktar from OrmeSiparisRezerv
  left join OrmeRaf on OrmeRaf.OrmeRafID = OrmeSiparisRezerv.OrmeRafID
  left join Ham on Ham.HamId = OrmeSiparisRezerv.hamid
  left join Firma UreticiFirma on UreticiFirma.Firmaid = OrmeRaf.UreticiFirmaId
  left join Renk on Renk.RenkId = OrmeRaf.Renkid
  Left Join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = OrmeRaf.KaliteAciklamaTipiId
  Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
  Left join DepoRaf On DepoRaf.RafID = OrmeRaf.RafId
  where OrmeSiparisRezerv.OrmeSiparisDetayID = :AOrmeSiparisDetayID
  into :OrmeSiparisRezervID, :OrmeSiparisDetayID, :OrmeSiparisDetayReceteID, :OrmeRafId, :StokGirisNo, :HamId, :HamAdi, :UretimPartiNo, :RafId, :RenkId, :RenkAdi, :Lot, :FasonStok,
       :KaliteNo, :MelanjKodu, :Miktar, :KullanilanMiktar  do
  begin
    if (FasonStok = 1) then FasonStokStr = 'Fason';
    else FasonStokStr = '';

    suspend;
  end
end
