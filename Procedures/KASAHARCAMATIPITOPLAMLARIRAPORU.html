CREATE PROCEDURE KASAHARCAMATIPITOPLAMLARIRAPORU(
    AKASAID VARCHAR(10),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICHARCAMATIPIID VARCHAR(10),
    ABITISHARCAMATIPIID VARCHAR(10),
    AHARCAMATIPIARALIKLI INTEGER,
    ASINIF VARCHAR(1),
    AHARCAMATIPIGRUPID HARCAMATIPIGRUPKODUSIZE,
    ADOVIZID DOVIZSIZE,
    ARAPORDOVIZID DOVIZSIZE)
RETURNS (
    HARCAMATIPIID VARCHAR(10),
    HARCAMATIPIADI VARCHAR(30),
    HARCAMATIPIGRUPID HARCAMATIPIGRUPKODUSIZE,
    HARCAMATIPIGRUPADI ADI30,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    KASAID KASAKODUSIZE,
    DOVIZID DOVIZSIZE,
    ADI ADI40,
    RAPORDOVIZID DOVIZSIZE)
AS
declare variable BorcAlacak     varchar(1);
declare variable Meblag         double Precision;
declare variable Okey           integer;
begin
  RaporDovizId = :ARaporDovizId;
  For select KasaId, DovizId, Adi, HarcamaTipiId, HarcamaTipiAdi, HarcamaTipiGrupId, HarcamaTipiGrupAdi, Sum(borc), Sum(Alacak)
  from KasaHarcamaTipiRaporu(:AKasaId, :ABaslangicTarihi, :ABitisTarihi, :ABaslangicHarcamaTipiId, :ABitisHarcamaTipiId, :AHarcamaTipiAralikli, :ASinif, :AHarcamaTipiGrupId, :ADovizId, :ARaporDovizId)
  group by KasaId, DovizId, Adi, HarcamaTipiId, HarcamaTipiAdi, HarcamaTipiGrupId, HarcamaTipiGrupAdi
  into :KasaId, :DovizId, :Adi, :HarcamaTipiId, :HarcamaTipiAdi, :HarcamaTipiGrupId, :HarcamaTipiGrupAdi, :Borc, :Alacak do suspend;
end
