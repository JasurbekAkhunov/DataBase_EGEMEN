CREATE PROCEDURE PARTININRECETESINIARSIVLE(
    APARTIKAYITID INTEGER,
    APARTIASAMALARIID INTEGER,
    ARECETEID VARCHAR(20))
AS
Declare variable ReceteAsamaId         integer;
Declare variable ReceteAltAsamaId      integer;
Declare variable ReceteGrafikId        varchar(10);
declare variable UrunId                varchar(20);
declare variable UrunBirimId           integer;
declare Variable Miktar                Double Precision;
declare Variable Banyo                 Double Precision;
declare variable FPartiKilo            Double Precision;
declare Variable FMiktar               Double Precision;
declare Variable FBanyo                Double Precision;
declare variable FReceteMiktarOrani    Double Precision;
declare variable FKimyasalTipiReceteOrani Double Precision;
Declare variable FBoyaAsamasimi           integer;
Declare variable FFBoyaAsamasimi          integer;
declare variable FToplamKilo            Double Precision;
declare variable FKilo                  Double Precision;
declare variable FPartiKayitId          integer;
declare variable FPartiAsamalariId      integer;
declare variable OrjBanyoCarpani          Double Precision;
declare variable BanyoCarpani          Double Precision;
declare variable BanyoCarpani2         Double Precision;
declare variable BanyoCarpani3         Double Precision;
declare variable BanyoCarpani4         Double Precision;
declare variable Koyultma              Double precision;
declare variable BirimFiyat            Double precision;
declare variable DovizId               varchar(5);
declare variable StokTabanDovizId      varchar(5);
declare variable DovizKuru             Double precision;
Declare variable FBanyoCarpanNo        integer;
Declare variable FFBanyoCarpanNo       integer;
Declare variable FLabReceteGrupId       integer;

begin
  Select StokTabanDovizId from Sistem where SistemID = 1 into :StokTabanDovizId;

  FToplamKilo = 0;

  for select PartiKayitId, PartiAsamalariId, BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, Koyultma from PartiRecetesiDokumu_1(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    into :FPartiKayitId, :FPartiAsamalariId, :BanyoCarpani, :BanyoCarpani2, :BanyoCarpani3, :BanyoCarpani4, :Koyultma do
  begin
    select sum(partiasamastok.GirisKilo) from PartiAsamalari
      left join partiasamastok on partiasamastok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      where PartiAsamalari.PartiKayitId = :FPartiKayitId and PartiAsamalari.PartiAsamalariId = :FPartiAsamalariId and PartiAsamalari.ReceteId = :AReceteId and PartiAsamaStok.AsamayaDahil = 1
      into :FKilo;

    if (APartiKayitId = FPartiKayitId) then FPartiKilo = FKilo;
    FToplamKilo = FToplamKilo + FKilo;
  end

  if (FPartiKilo is null or FPartiKilo = 0 or FToplamKilo is null or FToplamKilo = 0) then exit;

  if (BanyoCarpani is null) then BanyoCarpani = 0;
  if (BanyoCarpani2 is null) then BanyoCarpani2 = 0;
  if (BanyoCarpani3 is null) then BanyoCarpani3 = 0;
  if (BanyoCarpani4 is null) then BanyoCarpani4 = 0;

  if (Koyultma is null or Koyultma = 0) then Koyultma = 1;

  BanyoCarpani = BanyoCarpani / FToplamKilo * FPartiKilo;
  BanyoCarpani2 = BanyoCarpani2 / FToplamKilo * FPartiKilo;
  BanyoCarpani3 = BanyoCarpani3 / FToplamKilo * FPartiKilo;
  BanyoCarpani4 = BanyoCarpani4 / FToplamKilo * FPartiKilo;

  BanyoCarpani = Yuvarla(BanyoCarpani, 1);
  BanyoCarpani2 = Yuvarla(BanyoCarpani2, 1);
  BanyoCarpani3 = Yuvarla(BanyoCarpani3, 1);
  BanyoCarpani4 = Yuvarla(BanyoCarpani4, 1);

  select RECETEMIKTARORANI from SISTEM where SISTEMID = 1 into :FReceteMiktarOrani;

  if (FReceteMiktarOrani is null) then FReceteMiktarOrani = 1;

  FPartiKilo = FPartiKilo / FReceteMiktarOrani;

  FFBanyoCarpanNo = 1;

  for select ReceteDetay.ReceteAsamaId, ReceteDetay.ReceteAltAsamaId, ReceteDetay.ReceteGrafikId, ReceteDetay.UrunId, ReceteDetay.UrunBirimId,
             ReceteDetay.Miktar , ReceteDetay.Banyo, KimyasalTipi.ReceteOrani, ReceteAsama.BoyaAsamasimi, ReceteAsama.BanyoCarpanNo,
             UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / UBSD.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId, Recete.LabReceteGrupId from recetedetay
    left join Recete on Recete.ReceteId = ReceteDetay.ReceteId
    left join Urun on Urun.UrunId = ReceteDetay.UrunId
    left join KimyasalTipi on KimyasalTipi.KimyasalTipiId = Urun.KimyasalTipiId
    left join ReceteAsama on ReceteAsama.ReceteAsamaId = ReceteDetay.ReceteAsamaId
    left join UrunFiyat on UrunFiyat.UrunId = ReceteDetay.UrunId
    left join UrunBirim UBSD on UBSD.UrunBirimId = ReceteDetay.UrunBirimId
    left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
    where ReceteDetay.Receteid = :AReceteId
    order by ReceteDetay.Sira
    into :ReceteAsamaId, :ReceteAltAsamaId, :ReceteGrafikId, :UrunId, :UrunBirimId, :Miktar, :Banyo, :FKimyasalTipiReceteOrani, :FBoyaAsamasimi, :FBanyoCarpanNo,
         :BirimFiyat, :DovizId, :FLabReceteGrupId do
  begin
    if (ReceteAsamaId is not null) then
    begin
      FFBoyaAsamasimi = FBoyaAsamasimi;     /* Boya Asamasimi diye not alıyoruz.. eger boya asamasi ise kimyasal tipi carpani ile  islem yapıyoruz */
      FFBanyoCarpanNo = FBanyoCarpanNo;
    end

    FMiktar = 0;
    FBanyo = 0;


    if (FFBanyoCarpanNo = 2) then ORJBanyoCarpani = BanyoCarpani2;
    else if (FFBanyoCarpanNo = 3) then ORJBanyoCarpani = BanyoCarpani3;
    else if (FFBanyoCarpanNo = 4) then ORJBanyoCarpani = BanyoCarpani4;
    else ORJBanyoCarpani = BanyoCarpani;

    if (UrunId is not null) then
    if ((Banyo is not null and Banyo > 0) or (Miktar is not null and Miktar > 0)) then
    begin
      FMiktar = Miktar * FPartiKilo * Koyultma;

      if (FLabReceteGrupId = 0) then FBanyo = Banyo * ORJBanyoCarpani * Koyultma;
      else FBanyo = Banyo * ORJBanyoCarpani;

      if (FFBoyaAsamasimi = 1) then      /* boya asamasi ise kimyasal tipirecete orani devreye giriyor */
      if (FKimyasalTipiReceteOrani is not null) then FBanyo = FBanyo / FKimyasalTipiReceteOrani;

      if (FMiktar is null) then FMiktar = 0;
      if (FBanyo is null) then FBanyo = 0;
    end

    if (UrunId is not null) then
    begin
      if (StokTabanDovizId <> DovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :StokTabanDovizId, Current_date) into :DovizKuru;
      else DovizKuru = 1;
      BirimFiyat = BirimFiyat * DovizKuru;
    end
    else BirimFiyat = 0;

    if (BirimFiyat is null) then BirimFiyat = 0;

    insert into PartiReceteDetay(PartiAsamalariId, Sira, ReceteAsamaId, ReceteAltAsamaId, UrunId, UrunBirimId, ReceteGrafikId, Miktar, Banyo, HesaplananMiktar, BirimFiyat,DovizId)
         Values(:APartiAsamalariId, -1, :ReceteAsamaId, :ReceteAltAsamaId, :UrunId, :UrunBirimId, :ReceteGrafikId, :Miktar, :Banyo, (:Fmiktar + :Fbanyo), :BirimFiyat, :StokTabanDovizId);
  end
end
