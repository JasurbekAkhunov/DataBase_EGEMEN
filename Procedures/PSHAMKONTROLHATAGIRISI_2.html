CREATE PROCEDURE PSHAMKONTROLHATAGIRISI_2(
    APSHAMKONTROLHATAMETREID INTEGER,
    APSHAMKONTROLID INTEGER,
    AHAMHATATIPIID INTEGER,
    AMETRE DOUBLE PRECISION,
    ABITIS INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    PSHAMKONTROLHATAMETREID INTEGER)
AS
begin
  if (ABitis = 0) then
  begin
    if (APSHamKontrolHataMetreId is not null) then Update PSHamKontrolHataMetre Set BaslangicMetre = :AMetre where PSHamKontrolHataMetre.PSHamKontrolHataMetreID = :APSHamKontrolHataMetreID;
    else
    begin
      PSHamKontrolHataMetreID = Gen_Id(PSHamKontrolHataMetre_Id_Gen,1);
      insert into PSHamkontrolHataMetre(PSHamkontrolid, PSHamKontrolHataMetreID, HamHataTipiId, BaslangicMetre, InsertKullaniciId, KullaniciId)
                                 values(:APSHamKontrolId, :PSHamKontrolHataMetreID, :AHamHataTipiId, :AMetre, :AKullaniciId, :AKullaniciId);
    end
  end
  else Update PSHamKontrolHataMetre Set BitisMetre = :AMetre where PSHamKontrolHataMetre.PSHamKontrolHataMetreID = :APSHamKontrolHataMetreID;
  suspend;
end
