CREATE PROCEDURE KIMYASALSARFIYATYAP2(
    APARTISINIFID INTEGER,
    AURUNBIRIM VARCHAR(10),
    AURUNKODU1 VARCHAR(20),
    AURUNDEGER1 DOUBLE PRECISION,
    AURUNKODU2 VARCHAR(20),
    AURUNDEGER2 DOUBLE PRECISION,
    AURUNKODU3 VARCHAR(20),
    AURUNDEGER3 DOUBLE PRECISION,
    AURUNKODU4 VARCHAR(20),
    AURUNDEGER4 DOUBLE PRECISION,
    ARECETEID RECETEKODUSIZE)
AS
declare variable StoktanDusuldu integer;
declare variable FUrunId                varchar(20);
declare variable FUrunBirimId           integer;
declare variable FUrunMutfakDepoNo      varchar(10);
declare variable FBirimFiyat            double precision;
declare variable FUrunDovizId           varchar(5);
declare variable FStokTabanDovizId      varchar(5);
declare variable FHareketTipi           integer;
declare variable FFisSeriNo             varchar(2);
declare variable FFisNo                 integer;
declare variable FDovizKuru             double precision;
declare variable FPartiYil integer;
declare variable FPartiID integer;
declare variable PartiKayitID integer;
declare variable PartiSinif varchar(2);
declare variable FUrunBirim varchar(2);
declare variable PartiTopMiktar double precision;
declare variable PartiMiktar double precision;
declare variable FMiktar double precision;
declare variable AStokId integer;
declare variable StokId integer;
declare variable ATarih timestamp;
declare variable FUrunSarfiyati double precision;
declare variable FMakinaId MakinaKoduSize;
begin
  select STOKTABANDOVIZID from SISTEM where SISTEMID = 1 into :FStokTabanDovizId;
  FHareketTipi = 6;  /* Uretime Cikis */
  FUrunBirim = AUrunBirim;
  ATarih = current_timestamp;


  select FisSeriNo from Kullanici where KullaniciId = 1 into :FFisSeriNo;
  if (FFisSeriNo is null) then FFisSeriNo = '';

  select first 1 PartiStok.PartiKayitID, Parti.Sinif from PartiStok
    left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
  where PartiStok.PartiSinifid = :APartiSinifId into :PartiKayitID, :PartiSinif;

  if (PartiKayitID is null) then exception KayitHatasi ' Parti bilgisi bulunamadÄ±';

  if (AStokId is null or AStokId = 0) then
  begin
    select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;
    StokId = gen_id(Stok_Id_Gen, 1);

    insert into Stok(StokId, HareketTipi, SeriNo, FisNo, FisTarihi, DovizId, TabanDovizKuru1, TabanDovizKuru2, Sinif)
             values(:StokId, :FHareketTipi, :FFisSeriNo, :FFisNo, :ATarih, :FStokTabanDovizId, 1, 1, :PartiSinif);

  end
  else StokId = AStokId;

  Select First 1 MakinaId from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiSinifId = :APartiSinifId and CikisTarihi is not Null Order By CikisTarihi Desc into :FMakinaId;

  for select Urun.UrunID, RUB.UrunBirimID, Urun.MutfakDepoNo, UrunFiyat.UretimeCikisFiyati * UBUCF.Carpan / RUB.Carpan, UrunFiyat.UretimeCikisFiyatiDovizId,
    (case Urun.Transferkodu
  when :AUrunKodu1 then :AUrunDeger1
  when :AUrunKodu2 then :AUrunDeger2
  when :AUrunKodu3 then :AUrunDeger3
  when :AUrunKodu4 then :AUrunDeger4
  else 0 end) as UrunSarfiyati from ReceteDetay
    Left Join Urun on Urun.UrunId = ReceteDetay.UrunId
    left join UrunBirim RUB on RUB.BirimID = :FUrunBirim and RUB.UrunId = Urun.UrunID
    left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
    left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
    where ReceteDetay.ReceteId = :AReceteId and (Urun.Transferkodu = :AUrunKodu1 or Urun.Transferkodu = :AUrunKodu2 or Urun.Transferkodu = :AUrunKodu3 or Urun.Transferkodu = :AUrunKodu4)
  into :FUrunId, :FUrunBirimID, :FUrunMutfakDepoNo, :FBirimFiyat, :FUrunDovizId, :FUrunSarfiyati do
  begin
    if (FUrunId is null) then exception KayitHatasi ' ÃrÃ¼n Kodu TanÄ±mlanmamÄ±Å!';

    if (FBirimFiyat is null) then FBirimFiyat = 0;

    if (FUrunDovizId is not null and FBirimFiyat > 0) then
    if (FStokTabanDovizId <> FUrunDovizId) then
    begin
      select DovizKuru from EnYakinDovizKuruGetir(:FUrunDovizId, :FStokTabanDovizId, current_date) into :FDovizKuru;
      FBirimFiyat = FBirimFiyat * FDovizKuru;
    end

    insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, UrunId, UrunBirimId, Miktar, BirimFiyat, PartiKayitId, Otomasyon, UretimeCikisTarihi, MakinaId)
                   values(:ATarih, :FHareketTipi, :StokId, :FUrunMutfakDepoNo, :FUrunId, :FUrunBirimID, :FUrunSarfiyati, :FBirimFiyat, :PartiKayitID, 1, :ATarih, :FMakinaId);

  end
  --suspend;
end
