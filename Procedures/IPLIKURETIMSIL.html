CREATE PROCEDURE IPLIKURETIMSIL(
    AIPLIKURETIMID INTEGER,
    AKULLANICIID INTEGER)
AS
begin
  insert into IplikUretimLog(IptalKullaniciId, IplikLotId, IplikMakinaId, IplikDepoId, GirisTipi, UretimTarihi, SarimSekliId, IplikStokId,
                             AmbalajId, MelanjKodu, Lot, KonikRengiId, KonikCinsiId, BobinSayisi, AmbalajAgirligi, KapNo, PaletAgirligi, Miktar, NetMiktar)
              Select :AKullaniciId, IplikUretim.IplikLotId, IplikUretim.IplikMakinaId, IplikUretim.IplikDepoId, IplikUretim.GirisTipi, IplikUretim.UretimTarihi,
                     IplikUretim.SarimSekliId, IplikUretim.IplikStokId, IplikUretim.AmbalajId, IplikUretim.MelanjKodu, IplikUretim.Lot,
                     IplikUretim.KonikRengiId, IplikUretim.KonikCinsiId, IplikUretim.BobinSayisi, IplikUretim.AmbalajAgirligi, IplikUretim.KapNo, IplikUretim.PaletAgirligi, IplikUretim.Miktar, IplikUretim.NetMiktar from IplikUretim
                  where IplikUretim.IplikUretimID = :AIplikUretimID;

  delete from IplikUretim where IplikUretim.IplikUretimId = :AIplikUretimId;
end
