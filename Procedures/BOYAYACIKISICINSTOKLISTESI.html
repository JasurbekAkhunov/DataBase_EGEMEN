CREATE PROCEDURE BOYAYACIKISICINSTOKLISTESI(
    AORMESIPARISDETAYID INTEGER,
    ABOYASIPARISDETAYID INTEGER,
    AHAMID INTEGER,
    ARENKID INTEGER,
    ADEPOID VARCHAR(10),
    ADOKUMAHAREKETI EGEBOOLEAN)
RETURNS (
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    RAFID VARCHAR(10),
    KALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI INTEGER,
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    KALITENO INTEGER,
    ORMEMAKINAID VARCHAR(10),
    URETICIADI VARCHAR(40),
    URETIMPARTINO VARCHAR(30),
    ESKIPARTIYIL INTEGER,
    ESKIPARTINO INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    OLCUADI ADI40,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    KALANADET INTEGER,
    DESENID DESENKODUSIZE,
    ORMEMAKINAADI VARCHAR(40))
AS
declare variable SiparisFirmaId       varchar(20);
declare variable OrmeFirmaKodumuz     varchar(20);
declare variable DokumaFirmaKodumuz   varchar(20);
declare variable Fason                integer;
declare variable IslemTipiId          integer;
declare variable OrmeSiparisDetayId   integer;
declare variable itTamir              integer;
declare variable KTBoyaSiparisKayitID integer;
declare variable TupAcikEnMayli integer;
declare variable KS             integer;
begin
  Select BoyaSiparis.FirmaId, BoyaSiparis.Fason, OrmeSiparisDetay.KTBoyaSiparisKayitID from OrmeSiparisDetay
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
  where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId
  into :SiparisFirmaId, :Fason, :KTBoyaSiparisKayitID;

  Select BoyaSiparisDetay.IslemTipiId from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId into :IslemTipiId;

  Select itTamir from islemTipiGetir into :itTamir;

  if (Fason = 0) then SiparisFirmaId = Null;

  Select OrmeFirmaKodumuz, DokumaFirmaKodumuz  from sistem where sistemid = 1 into :OrmeFirmaKodumuz, :DokumaFirmaKodumuz;

  if (ADokumaHareketi = 1) then
    OrmeFirmaKodumuz = DokumaFirmaKodumuz;

  For Select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.RafId, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
             OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi, OrmeRaf.KaliteNo, OrmeRaf.Lot, OrmeRaf.MelanjKodu, Uretici.Adi,
             coalesce(OrmeRaf.OrmeMakinaId, OrmeRaf.OrmeciMakinaNo), OrmeRaf.UretimPartiNo, OrmeRaf.EskiPartiYil, OrmeRaf.EskiPartiNo, OrmeSiparisDetay.TupAcikEnMayli,
             Olcu.OlcuAdi, OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmeRaf.GirisAdet - OrmeRaf.CikisAdet, OrmeRaf.DesenId, coalesce(OrmeMakina.Adi, '') from OrmeRaf
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
    left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = OrmeRaf.OrmeSiparisDetayID
    left join Olcu on Olcu.OlcuId = OrmeRaf.OlcuId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmeRaf.OrmeMakinaId
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeRaf.OrmePlanlamaId
    where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and
         (OrmeRaf.FasonStok = :Fason and (OrmeRaf.FasonStok = 0 or OrmeRaf.StokFirmaID = :SiparisFirmaID)) and
         /*(IntKarsilastir(OrmeRaf.RenkId, :ARenkId) = 1 or (:IslemTipiId = :itTamir and OrmeRaf.RenkId is not null)) and*/
         /*(OrmeRaf.DesenId is null or :IslemTipiId = :itTamir) and Telep: EB-377*/
         ( OrmeRaf.OrmeSiparisDetayId = :AOrmeSiparisDetayId or
           OrmeRaf.OrmeSiparisDetayId is null or
          (OrmeRaf.OrmeSiparisDetayId is not null and BoyaSiparis.FirmaID = :OrmeFirmaKodumuz) or
          (OrmeRaf.BoyaSiparisKayitID = :KTBoyaSiparisKayitID) ) and
         (DepoRaf.DepoID = :ADepoID or :ADepoId is null)
    into :OrmeRafId, :StokGirisNo, :RafId, :RenkId, :RenkAdi, :KalanMiktar, :KalanKapAdedi, :KaliteNo,  :Lot, :MelanjKodu,  :UreticiAdi, :OrmeMakinaId, :UretimPartiNo,
         :EskiPartiYil, :EskiPartiNo, :TupAcikEnMayli, :OlcuAdi, :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :KalanAdet, :DesenId, :OrmeMakinaAdi do
  begin
    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    /*top top kayit varsa raf sectirmiyoruz, top top isleme izin verecegiz*/
    select count(*) from OrmeRafDetay
    left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketId
    left join GirisKaliteKontrol on GirisKaliteKontrol.StokHareketId = StokHareket.StokHareketId
    left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketId = StokHareket.StokHareketId
    where OrmeRafDetay.OrmeRafId = :OrmeRafId and
    StokHareket.DokumaHareketi = 1 and
    (GirisKaliteKontrol.GirisKaliteKontrolId is not null or DokumaKaliteKontrol.DokumaKaliteKontrolId is not null) into :KS;

    if (KS = 0) then
      suspend;
  end
end
