CREATE PROCEDURE DEPOSTOKGUNCELLE(
    AHAREKETTIPI INTEGER,
    ADEPOID VARCHAR(10),
    ATRANSFERDEPOID VARCHAR(10),
    AURUNID VARCHAR(20),
    AMIKTAR DOUBLE PRECISION,
    ADURUM INTEGER,
    AIADE EGEBOOLEAN)
AS
declare variable KayitSayisi integer;
declare variable shtDevir       integer;
declare variable shtAlis        integer;
declare variable shtSatis       integer;
declare variable shtAlisIade    integer;
declare variable shtSatisIade   integer;
declare variable shtDepoTransfer integer;
declare variable shtUretimeCikis integer;
declare variable shtUretim      integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtFasonUretim      integer;
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtDepoTransfer, shtUretimeCikis, ShtUretim, shtFasonUretimeCikis, shtFasonUretim from StokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtDepoTransfer, :shtUretimeCikis, :ShtUretim, :shtFasonUretimeCikis, :shtFasonUretim;
  if (ADurum = 0) then AMiktar = AMiktar * -1;

  select count(*) from DepoStok where DepoId = :ADepoId and UrunId = :AUrunId into :KayitSayisi;
  if (KayitSayisi = 0) then insert into DepoStok(DepoId, UrunId) values(:ADepoId, :AUrunId);

  if (AHareketTipi = shtDevir) then Update DepoStok set Devir = Devir + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Devir */
  else
  if (AHareketTipi = shtAlis) then Update DepoStok set Alis = Alis + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Alis */
  else
  if (AHareketTipi = shtSatis) then Update DepoStok set Satis = Satis + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Satis */
  else
  if (AHareketTipi = shtAlisIade) then Update DepoStok set AlisIadesi = AlisIadesi + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId;  /* Alis Iade */
  else
  if (AHareketTipi = shtSatisIade) then Update DepoStok set SatisIadesi = SatisIadesi + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Satis Iade */
  else
  if (AHareketTipi = shtDepoTransfer) then
  begin
    Update DepoStok set DepoyaCikan = DepoyaCikan + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Depo tarnsfer */
    select count(*) from DepoStok where DepoId = :ATransferDepoId and UrunId = :AUrunId into :KayitSayisi;

    if (KayitSayisi = 0) then insert into DepoStok(DepoId, UrunId) values(:ATransferDepoId, :AUrunId);
    Update DepoStok set DepodanGelen = DepodanGelen + :AMiktar where DepoId = :ATransferDepoId and UrunId = :AUrunId; /* Depo tarnsfer */
  end
  else
  if (AHareketTipi = shtUretimeCikis) then Update DepoStok set UretimeCikis = UretimeCikis + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Uretime Cikis */
  else
  if (AHareketTipi = shtUretim) then Update DepoStok set Uretim = Uretim + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Uretim */
  else
  if (AHareketTipi = shtFasonUretimeCikis) then Update DepoStok set FasonUretimeCikis = FasonUretimeCikis + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Fason Uretime Cikis */
  else
  if (AHareketTipi = shtFasonUretim) then
  begin
    if (AIade = 1) then
      Update DepoStok set FasonUretimIade = FasonUretimIade + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Fason Uretim */
    else
      Update DepoStok set FasonUretim = FasonUretim + :AMiktar where DepoId = :ADepoId and UrunId = :AUrunId; /* Fason Uretim */
  end
end
