CREATE PROCEDURE BOYAFATURADETAYCEKILISTESI(
    ABOYAFATURAID INTEGER)
RETURNS (
    BOYAIRSALIYEDETAYID INTEGER,
    IPLIKPAKETLEMEID INTEGER,
    PSKALITEKONTROLID INTEGER,
    BARKODID VARCHAR(20),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    KONIKCINSIID INTEGER,
    KONIKCINSIADI VARCHAR(20),
    KOLINO INTEGER,
    KAPNO INTEGER,
    KALITENO INTEGER,
    EN VARCHAR(10),
    GRAMAJ VARCHAR(10),
    DARA NUMERIC(12,3),
    AMBALAJAGIRLIGI NUMERIC(12,3),
    BOBINSAYISI INTEGER,
    BOBINDARASI NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    HACIMEN INTEGER,
    HACIMBOY INTEGER,
    HACIMYUKSEKLIK INTEGER,
    HACIM DOUBLE PRECISION,
    PALETNO VARCHAR(5))
AS
declare variable FBoyaIrsaliyeDetayId integer;
begin
  En       = null;
  Gramaj   = Null;
  HacimEn  = 0;
  HacimBoy = 0;
  HacimYukseklik = 0;

  for select BoyaIrsaliyeDetayId from BoyaFaturaDetay
    where BoyaFaturaId = :ABoyaFaturaId
    group by BoyaIrsaliyeDetayId
    into :FBoyaIrsaliyeDetayId do
  begin
    For select IplikPaketleme.IplikPaketlemeId, IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, PartiStokTarti.PartiStokTartiId,
               PartiStokTarti.BoyaIrsaliyeDetayId, PartiStokTarti.KaliteNo, IplikPaketleme.KoliNo,
               IplikPaketleme.Dara, IplikPaketleme.BobinSayisi, IplikPaketleme.BobinDarasi,
               IplikPaketleme.AmbalajAgirligi, IplikPaketleme.BrutKilo,
               IplikPaketleme.NetKilo, PartiStokTarti.PaletNo,
               IplikPaketleme.AmbalajId, Ambalaj.Adi, IplikPaketleme.KonikCinsiId, KonikCinsi.Adi from PartiStokTarti
    left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    Left Join Ambalaj on Ambalaj.AmbalajId = IplikPaketleme.AmbalajId
    Left Join KonikCinsi on KonikCinsi.KonikCinsiId = IplikPaketleme.KonikCinsiId
    where PartiStokTarti.BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId and IplikPaketleme.IplikPaketlemeId is not null
    into :IplikPaketlemeId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :KaliteNo, :KoliNo,
         :Dara, :BobinSayisi, :BobinDarasi, :AmbalajAgirligi,
         :BrutKilo, :NetKilo, :PaletNo, :AmbalajId, :AmbalajAdi, :KonikCinsiId, :KonikCinsiAdi do
    begin
      BarkodID = 'I-' || EgemenIntegerToStr(IplikPaketlemeId);
      Hacim = 0;
      suspend;
    end
  end

  KoliNo = null;
  AmbalajId = Null;
  AmbalajAdi= Null;
  KonikCinsiId= Null;
  KonikCinsiAdi= Null;
  AmbalajAgirligi = Null;
  BobinSayisi     = Null;
  BobinDarasi     = Null;


  for select BoyaIrsaliyeDetayId from BoyaFaturaDetay
    where BoyaFaturaId = :ABoyaFaturaId
    group by BoyaIrsaliyeDetayId
    into :FBoyaIrsaliyeDetayId do
  begin
    For select PsKaliteKontrol.PsKaliteKontrolId, PsKaliteKontrol.PartiKayitId, PsKaliteKontrol.PartiSinifId, PartiStokTarti.PartiStokTartiId,
               PartiStokTarti.BoyaIrsaliyeDetayId, PsKaliteKontrol.TopNo, PsKaliteKontrol.Kilo, PsKaliteKontrol.Dara, PsKaliteKontrol.Metre,
               PartiStokTarti.KaliteNo, PsKaliteKontrol.En, PsKaliteKontrol.Gramaj,
               PartiStok.HacimEn, PartiStok.HacimBoy, PartiStok.HacimYukseklik, PartiStokTarti.PaletNo from PartiStokTarti
      left join PsKaliteKontrol on PsKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
      left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
    where PartiStokTarti.BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId and PsKaliteKontrol.PsKaliteKontrolId is not null
    into :PsKaliteKontrolId, :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :BoyaIrsaliyeDetayId, :KapNo, :BrutKilo, :Dara, :Metre,
         :KaliteNo, :En, :Gramaj, :HacimEn, :HacimBoy, :HacimYukseklik, :PaletNo do
    begin
      NetKilo = BrutKilo - Dara;
      BarkodID = 'K-' || EgemenIntegerToStr(PSKaliteKontrolId);
      Hacim = HacimEn * HacimBoy * HacimYukseklik;
      Hacim = Hacim / 1000000;
      suspend;
    end
  end
end
