CREATE PROCEDURE MAKINASARJORTALAMAMBILGIGETIR(
    AMAKINASARJID INTEGER)
RETURNS (
    NEM DOUBLE PRECISION,
    ISI DOUBLE PRECISION,
    HIZ DOUBLE PRECISION,
    BACANEM DOUBLE PRECISION,
    KS INTEGER)
AS
declare variable BaslamaTarihi TimeStamp;
declare variable BitisTarihi   TimeStamp;
declare variable MakinaID      varchar(10);
begin
  Select MakinaId, BaslamaTarihi, BitisTarihi from MakinaSarj
    where MakinaSarj.MakinaSarjID = :AMakinaSarjID
    into :MakinaId, :BaslamaTarihi, :BitisTarihi;

  if (BitisTarihi is not null) then
  begin
    Select Sum(Nem) / Count(*), Sum(Isi) / Count(*), Sum(Hiz) / Count(*), Sum(BacaNem) / Count(*), Count(*) from MakinaBilgi
       where MakinaBilgi.Tarih >= :BaslamaTarihi and MakinaBilgi.Tarih <= :BitisTarihi and MakinaBilgi.MakinaID = :MakinaID
       into :Nem, :Isi, :Hiz, :BacaNem, :KS;
    suspend;
  end
end
