CREATE PROCEDURE URETIMRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    SIRANO INTEGER,
    METRETAKIBI INTEGER,
    MAKINAID VARCHAR(10),
    URETIMTOPLAMI NUMERIC(12,3),
    ILAVEPROSESTOPLAMI NUMERIC(12,3),
    ICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    UCRETLIICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    UCRETSIZICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    DISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    UCRETLIDISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    UCRETSIZDISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    FASONURETIMTOPLAMI NUMERIC(12,3),
    FASONILAVEPROSESTOPLAMI NUMERIC(12,3),
    FASONICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    FASUCRETLIICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    FASUCRETSIZICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    FASONDISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    FASUCRETLIDISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    FASUCRETSIZDISTAMIRURETIMTOPLAM NUMERIC(12,3),
    SATISURETIMTOPLAMI NUMERIC(12,3),
    SATISILAVEPROSESTOPLAMI NUMERIC(12,3),
    SATISICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    SATUCRETLIICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    SATUCRETSIZICTAMIRURETIMTOPLAMI NUMERIC(12,3),
    SATISDISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    SATUCRETLIDISTAMIRURETIMTOPLAMI NUMERIC(12,3),
    SATUCRETSIZDISTAMIRURETIMTOPLAM NUMERIC(12,3),
    URETIMTOPLAMIMT NUMERIC(12,3),
    ILAVEPROSESTOPLAMIMT NUMERIC(12,3),
    ICTAMIRURETIMTOPLAMIMT NUMERIC(12,3),
    DISTAMIRURETIMTOPLAMIMT NUMERIC(12,3),
    CIKISV1MIKTAR NUMERIC(12,3),
    CIKISV2MIKTAR NUMERIC(12,3),
    CIKISV3MIKTAR NUMERIC(12,3))
AS
declare variable FKilo                        Double Precision;
declare variable FPartiTamirId                integer;
declare variable FIlaveProses                 integer;
declare variable FCikisTarihi                 TimeStamp;
declare variable FIslemTipiId                 integer;
declare variable FFason                       integer;
declare variable FLastAsamaId                 integer;
declare variable FLastMakinaId                varchar(10);
declare variable FEn                          integer;
declare variable FGramaj                      integer;
declare variable FMetreTul                    Double Precision;
declare variable FMetre                       Double Precision;


declare variable FAsamaId                     integer;
declare variable FAsamaAdi                    varchar(40);
declare variable FMakinaId                    varchar(10);
declare variable FMetreTakibi                 integer;
declare variable FSiraNo                      integer;

declare variable FBitisTarihi   TimeStamp;
declare variable FAktarmaKilo   Numeric(12,3);
declare variable FIcTamirUcretli integer;
declare variable FFaturaKesilecek integer;
begin
  UretimToplami = 0;

  IcTamirUretimToplami = 0;
  UcretliIcTamirUretimToplami = 0;
  UcretsizIcTamirUretimToplami = 0;
  DisTamirUretimToplami = 0;
  UcretliDisTamirUretimToplami = 0;
  UcretsizDisTamirUretimToplami = 0;

  FasonUretimToplami = 0;
  FasonIcTamirUretimToplami = 0;
  FasUcretliIcTamirUretimToplami = 0;
  FasUcretsizIcTamirUretimToplami = 0;

  FasonDisTamirUretimToplami = 0;
  FasUcretliDisTamirUretimToplami = 0;
  FasUcretsizDisTamirUretimToplam = 0;

  SatisUretimToplami = 0;
  SatisIcTamirUretimToplami = 0;
  SatUcretliIcTamirUretimToplami = 0;
  SatUcretsizIcTamirUretimToplami = 0;
  SatisDisTamirUretimToplami = 0;
  SatUcretliDisTamirUretimToplami = 0;
  SatUcretsizDisTamirUretimToplam = 0;

  UretimToplamiMt = 0;
  IcTamirUretimToplamiMt = 0;
  DisTamirUretimToplamiMt = 0;

  IlaveProsesToplami = 0;
  FasonIlaveProsesToplami = 0;
  SatisIlaveProsesToplami = 0;
  IlaveProsesToplamiMt = 0;

  FLastAsamaId = -1;
  FLastMakinaId = '';

  for select Asama.AsamaId, Asama.Adi, Asama.GURaporuSiraNo, Asama.MetreTakibi, PartiAsamaStokDetay.MakinaId, Parti.IslemTipiId, Parti.Fason,
             PartiAsamalari.PartiTamirId, PartiAsamaStokDetay.CikisTarihi, PartiStok.OngorulenEn, PartiStok.OngorulenGramaj,
             PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.Metre, PartiAsamalari.IlaveProses, TamirCozumSekli.Ucretli,
             Parti.FaturaKesilecek, PartiAsamaStokDetay.CikisV1Miktar, PartiAsamaStokDetay.CikisV2Miktar, PartiAsamaStokDetay.CikisV3Miktar from PartiAsamaStokDetay
    left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join PartiAsamaStok On PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join PartiAsamalari On PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliId = PartiTamir.TamirCozumSekliId
    where PartiAsamaStokDetay.Cikistarihi >= :ABaslangicTarihi and PartiAsamaStokDetay.CikisTarihi < :ABitisTarihi and
          Asama.GunlukUretimRaporunaDahil = 1
    order by Asama.Adi, PartiAsamaStokDetay.MakinaId
    into :FAsamaId, :FAsamaAdi, :FSiraNo, :FMetreTakibi, :FMakinaId, :FIslemTipiId, :FFason, :FPartiTamirId, :FCikisTarihi, :FEn, :FGramaj,
         :FKilo, :FMetre, :FIlaveProses, :FIcTamirUcretli, :FFaturaKesilecek, :CikisV1Miktar, :CikisV2Miktar, :CikisV3Miktar do
  begin
    if (FEn is null) then FEn = 0;
    if (FGramaj is null) then FGramaj = 0;

    if (FAsamaId <> FLastAsamaId or FMakinaId <> FLastMakinaId) then
    begin
      if (FLastAsamaId <> -1) then
      begin
        suspend;
      end
      FLastAsamaId = FAsamaId;
      FLastMakinaId = FMakinaId;

      AsamaId = FAsamaId;
      AsamaAdi = FAsamaAdi;
      MakinaId = FMakinaId;
      MetreTakibi = FMetreTakibi;
      SiraNo = FSiraNo;

      UretimToplami = 0;
      IcTamirUretimToplami = 0;
      UcretliIcTamirUretimToplami = 0;
      UcretsizIcTamirUretimToplami = 0;
      DisTamirUretimToplami = 0;
      UcretliDisTamirUretimToplami = 0;
      UcretsizDisTamirUretimToplami = 0;

      FasonUretimToplami = 0;
      FasonIcTamirUretimToplami = 0;
      FasUcretliIcTamirUretimToplami = 0;
      FasUcretsizIcTamirUretimToplami = 0;
      FasonDisTamirUretimToplami = 0;
      FasUcretliDisTamirUretimToplami = 0;
      FasUcretsizDisTamirUretimToplam = 0;

      SatisUretimToplami = 0;
      SatisIcTamirUretimToplami = 0;
      SatUcretliIcTamirUretimToplami = 0;
      SatUcretsizIcTamirUretimToplami = 0;
      SatisDisTamirUretimToplami = 0;
      SatUcretliDisTamirUretimToplami = 0;
      SatUcretsizDisTamirUretimToplam = 0;

      UretimToplamiMt = 0;
      IcTamirUretimToplamiMt = 0;
      DisTamirUretimToplamiMt = 0;

      IlaveProsesToplami = 0;
      FasonIlaveProsesToplami = 0;
      SatisIlaveProsesToplami = 0;
      IlaveProsesToplamiMt = 0;
    end

    if (FMetre = 0) then
    begin
/*      FMetreTul = (FEn / 100.0)  * (FGramaj / 1000.0); */
      FMetreTul = FEn;
      FMetreTul = FMetreTul / 100;

      FMetreTul = FMetreTul  * FGramaj / 1000;

      if (FMetreTul is null) then FMetreTul = 0;

      if (FMetreTul <> 0) then FMetre = FKilo / FMetreTul;
      else FMetre = 0;
    end

    UretimToplami = UretimToplami + FKilo;
    UretimToplamiMt = UretimToplamiMt + FMetre;

    if (FFason = 1) then FasonUretimToplami = FasonUretimToplami + FKilo;
    else SatisUretimToplami = SatisUretimToplami + FKilo;

    if (FIlaveProses = 1) then
    begin
      IlaveProsesToplami = IlaveProsesToplami + FKilo;
      IlaveProsesToplamiMt = IlaveProsesToplamiMt + FMetre;

      if (FFason = 1) then FasonIlaveProsesToplami = FasonIlaveProsesToplami + FKilo;
      else SatisIlaveProsesToplami = SatisIlaveProsesToplami + FKilo;

    end
    else if (FPartiTamirId is not null) then
    begin
      IcTamirUretimToplami =  IcTamirUretimToplami + FKilo;
      IcTamirUretimToplamiMt =  IcTamirUretimToplamiMt + FMetre;

      if (FIcTamirUcretli = 1) then UcretliIcTamirUretimToplami = UcretliIcTamirUretimToplami + FKilo;
      else UcretsizIcTamirUretimToplami = UcretsizIcTamirUretimToplami + FKilo;

      if (FFason = 1) then
      begin
        FasonIcTamirUretimToplami = FasonIcTamirUretimToplami + FKilo;

        if (FIcTamirUcretli = 1) then FasUcretliIcTamirUretimToplami = FasUcretliIcTamirUretimToplami + FKilo;
        else FasUcretsizIcTamirUretimToplami = FasUcretsizIcTamirUretimToplami + FKilo;
      end
      else
      begin
        SatisIcTamirUretimToplami = SatisIcTamirUretimToplami + FKilo;

        if (FIcTamirUcretli = 1) then SatUcretliIcTamirUretimToplami = SatUcretliIcTamirUretimToplami + FKilo;
        else SatUcretsizIcTamirUretimToplami = SatUcretsizIcTamirUretimToplami + FKilo;
      end
    end
    else if (FIslemTipiId = 3) then
    begin
      DisTamirUretimToplami =  DisTamirUretimToplami + FKilo;
      DisTamirUretimToplamiMt =  DisTamirUretimToplamiMt + FMetre;

      if (FFaturaKesilecek = 1) then UcretliDisTamirUretimToplami = UcretliDisTamirUretimToplami + FKilo;
      else UcretsizDisTamirUretimToplami = UcretsizDisTamirUretimToplami + FKilo;

      if (FFason = 1) then
      begin
        FasonDisTamirUretimToplami = FasonDisTamirUretimToplami + FKilo;

        if (FFaturaKesilecek = 1) then FasUcretliDisTamirUretimToplami = FasUcretliDisTamirUretimToplami + FKilo;
        else FasUcretsizDisTamirUretimToplam = FasUcretsizDisTamirUretimToplam + FKilo;

      end
      else
      begin
        SatisDisTamirUretimToplami = SatisDisTamirUretimToplami + FKilo;

        if (FFaturaKesilecek = 1) then SatUcretliDisTamirUretimToplami = SatUcretliDisTamirUretimToplami + FKilo;
        else SatUcretsizDisTamirUretimToplam = SatUcretsizDisTamirUretimToplam + FKilo;
      end
    end
  end

  if (FLastAsamaId <> -1) then
  begin
    suspend;
  end


  AsamaId = Null;

  Select Asama.AsamaId, Asama.Adi, Asama.GURaporuSiraNo from Sistem
    left join Asama on Asama.AsamaId = Sistem.HamAktarmaAsamaId
    where sistemid = 1
    into AsamaId, :AsamaAdi, :SiraNo;

  if (AsamaId is Null) then exit;

  UretimToplami = 0;
  IcTamirUretimToplami = 0;
  UcretliIcTamirUretimToplami = 0;
  UcretsizIcTamirUretimToplami = 0;
  DisTamirUretimToplami = 0;
  UcretliDisTamirUretimToplami = 0;
  UcretsizDisTamirUretimToplami = 0;

  FasonUretimToplami = 0;
  FasonIcTamirUretimToplami = 0;
  FasUcretliIcTamirUretimToplami = 0;
  FasUcretsizIcTamirUretimToplami = 0;
  FasonDisTamirUretimToplami = 0;
  FasUcretliDisTamirUretimToplami = 0;
  FasUcretsizDisTamirUretimToplam = 0;

  SatisUretimToplami = 0;
  SatisIcTamirUretimToplami = 0;
  SatUcretliIcTamirUretimToplami = 0;
  SatUcretsizIcTamirUretimToplami = 0;
  SatisDisTamirUretimToplami = 0;
  SatUcretliDisTamirUretimToplami = 0;
  SatUcretsizDisTamirUretimToplam = 0;

  UretimToplamiMt = 0;
  IcTamirUretimToplamiMt = 0;
  DisTamirUretimToplamiMt = 0;

  IlaveProsesToplami = 0;
  FasonIlaveProsesToplami = 0;
  SatisIlaveProsesToplami = 0;
  IlaveProsesToplamiMt = 0;

  FLastMakinaID = '';
  CikisV1Miktar = 0;
  CikisV2Miktar = 0;
  CikisV3Miktar = 0;

  For Select HAktarmaIsEmri.MakinaId, HamStokDetay.Fason, HAktarmaIsEmri.BitisTarihi, Sum(HAktarmaIsEmriDetay.Kilo) from HAktarmaIsEmri
    left join HAktarmaIsEmriDetay on HAktarmaIsEmriDetay.HAktarmaIsEmriKayitId = HAktarmaIsEmri.HAktarmaIsEmriKayitId
    left join HAktarmaFisiDetay on HAktarmaFisiDetay.HAktarmaFisiDetayId = HAktarmaIsEmriDetay.HAktarmaFisiDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = HAktarmaFisiDetay.HamStokDetayId
    where HAktarmaIsEmri.BitisTarihi >= :ABaslangicTarihi and HAktarmaIsEmri.BitisTarihi < :ABitisTarihi and
          HAktarmaIsEmriDetay.HAktarmaIsEmriKayitId is not null
    group By HAktarmaIsEmri.MakinaId, HAktarmaIsEmri.HAktarmaIsEmriKayitId, HamStokDetay.Fason, HAktarmaIsEmri.BitisTarihi
    into :FMakinaId, :FFason, :FBitisTarihi, :FAktarmaKilo do
  begin
    if (FMakinaId <> FLastMakinaId) then
    begin
      if (FLastMakinaId <> '') then
      begin
        suspend;
      end
      FLastMakinaId = FMakinaId;
      MakinaId = FMakinaId;

      UretimToplami = 0;
      FasonUretimToplami = 0;
      SatisUretimToplami = 0;
    end

    UretimToplami = UretimToplami + FAktarmaKilo;
    if (FFason = 1) then FasonUretimToplami = FasonUretimToplami + FAktarmaKilo;
    else SatisUretimToplami = SatisUretimToplami + FAktarmaKilo;
  end

  if (FLastMakinaId <> '') then
  begin
    suspend;
  end
end
