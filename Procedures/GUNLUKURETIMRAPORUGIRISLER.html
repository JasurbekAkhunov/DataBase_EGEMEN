Definition

CREATE PROCEDURE GUNLUKURETIMRAPORUGIRISLER(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP)
RETURNS (
    CALISMAGUNSAYISI DOUBLE PRECISION,
    FASONIPLIKGIRISI NUMERIC(12,3),
    SATISIPLIKGIRISI NUMERIC(12,3),
    FASONKUMASGIRISI NUMERIC(12,3),
    SATISKUMASGIRISI NUMERIC(12,3),
    FASONORMEKUMASGIRISI NUMERIC(12,3),
    SATISORMEKUMASGIRISI NUMERIC(12,3),
    FASONDOKUMAKUMASGIRISI NUMERIC(12,3),
    SATISDOKUMAKUMASGIRISI NUMERIC(12,3),
    IPLIKUCRETLITAMIRGIRISI NUMERIC(12,3),
    IPLIKUCRETSIZTAMIRGIRISI NUMERIC(12,3),
    ORMEKUMASUCRETLITAMIRGIRISI NUMERIC(12,3),
    ORMEKUMASUCRETSIZTAMIRGIRISI NUMERIC(12,3),
    DOKUMAKUMASUCRETLITAMIRGIRISI NUMERIC(12,3),
    DOKUMAKUMASUCRETSIZTAMIRGIRISI NUMERIC(12,3),
    TAMIRGIRISI NUMERIC(12,3),
    AYLIKFASONIPLIKGIRISI NUMERIC(12,3),
    AYLIKSATISIPLIKGIRISI NUMERIC(12,3),
    AYLIKFASONKUMASGIRISI NUMERIC(12,3),
    AYLIKSATISKUMASGIRISI NUMERIC(12,3),
    AYLIKFASONORMEKUMASGIRISI NUMERIC(12,3),
    AYLIKSATISORMEKUMASGIRISI NUMERIC(12,3),
    AYLIKFASONDOKUMAKUMASGIRISI NUMERIC(12,3),
    AYLIKSATISDOKUMAKUMASGIRISI NUMERIC(12,3),
    AYLIKTAMIRGIRISI NUMERIC(12,3),
    AYLIKIPLIKUCRETLITAMIRGIRISI NUMERIC(12,3),
    AYLIKIPLIKUCRETSIZTAMIRGIRISI NUMERIC(12,3),
    AYLIKOKUMASUCRETLITAMIRGIRISI NUMERIC(12,3),
    AYLIKOKUMASUCRETSIZTAMIRGIRISI NUMERIC(12,3),
    AYLIKDKUMASUCRETLITAMIRGIRISI NUMERIC(12,3),
    AYLIKDKUMASUCRETSIZTAMIRGIRISI NUMERIC(12,3),
    HAMDEPOSTOKMIKTARI NUMERIC(12,3))
AS
declare variable FIrsaliyeTarihi        TimeStamp;
declare variable FHamTipiId             integer;
declare variable FFason                 integer;
declare variable FTamir                 integer;
declare variable FUcretli               integer;
declare variable FKilo                  Numeric(12,3);
declare variable FirmaId                varchar(20);
begin
  Select FirmaKodumuz from Sistem into :FirmaId;

  select sum(girisMiktar) - sum(PartilenenMiktar) - sum(iadeedilenMiktar) - sum(satisMiktar) from HamStokKumilatif
    where HamStokKumilatif.Firmaid = :FirmaId  into :HamDepoStokMiktari;



  Select Sum(coalesce(KumasGirisiCalismaSaati, 0)) from GunlukCalismaSaati where GunlukCalismaSaati.Tarih >= :AAylikBaslangic and GunlukCalismaSaati.Tarih <= :AAylikBitis into :CalismaGunSayisi;

  if (CalismaGunSayisi is not null) then CalismaGunSayisi = CalismaGunSayisi / 24;
  else CalismaGunSayisi = AAylikBitis - AAylikBaslangic;

  CalismaGunSayisi = Yuvarla(CalismaGunSayisi, 10);

  FasonIplikGirisi       = 0;
  SatisIplikGirisi       = 0;
  FasonKumasGirisi       = 0;
  SatisKumasGirisi       = 0;
  FasonOrmeKumasGirisi   = 0;
  SatisOrmeKumasGirisi   = 0;
  FasonDokumaKumasGirisi = 0;
  SatisDokumaKumasGirisi = 0;


  IplikUcretliTamirGirisi = 0;
  IplikUcretsizTamirGirisi = 0;
  OrmeKumasUcretliTamirGirisi   = 0;
  OrmeKumasUcretsizTamirGirisi   = 0;
  DokumaKumasUcretliTamirGirisi = 0;
  DokumaKumasUcretsizTamirGirisi = 0;
  TamirGirisi            = 0;

  AylikFasonIplikGirisi  = 0;
  AylikSatisIplikGirisi  = 0;
  AylikFasonKumasGirisi  = 0;
  AylikSatisKumasGirisi  = 0;
  AylikFasonOrmeKumasGirisi = 0;
  AylikSatisOrmeKumasGirisi = 0;
  AylikFasonDokumaKumasGirisi = 0;
  AylikSatisDokumaKumasGirisi = 0;

  AylikOKumasUcretliTamirGirisi = 0;
  AylikOKumasUcretsizTamirGirisi = 0;
  AylikDKumasUcretliTamirGirisi = 0;
  AylikDKumasUcretsizTamirGirisi = 0;
  AylikIplikUcretliTamirGirisi = 0;
  AylikIplikUcretsizTamirGirisi = 0;
  AylikTamirGirisi       = 0;

  For Select HamStokMaster.Tarih, Ham.HamTipiId, HamStokMaster.Fason, HamStokMaster.Tamir, HamStokMaster.Kilo, Max(DisTamirCozumSekli.Ucretli) from HamStokMaster
    left join Ham on Ham.HamID = HamStokMaster.HamId
    left join DisTamirCozum on DisTamirCozum.HamStokMasterID = HamStokMaster.HamStokMasterId
    left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliID = DisTamirCozum.DisTamirCozumSekliId
    where HamStokMaster.Tarih >= :AAylikBaslangic and HamStokMaster.Tarih <= :AAylikBitis
    group by HamStokMaster.HamStokMasterId,  HamStokMaster.Tarih, Ham.HamTipiId, HamStokMaster.Fason, HamStokMaster.Tamir, HamStokMaster.Kilo
    into :FIrsaliyeTarihi, :FHamTipiId, :FFason, :FTamir, :FKilo, :FUcretli do
  begin
    if (FTamir = 0) then
    begin
      if (FHamTipiId = 0) then
      begin
        if (FFason = 1) then AylikFasonIplikGirisi = AylikFasonIplikGirisi + FKilo;
        else AylikSatisIplikGirisi = AylikSatisIplikGirisi + FKilo;
      end
      else
      begin
        if (FFason = 1) then AylikFasonKumasGirisi = AylikFasonKumasGirisi + FKilo;
        else AylikSatisKumasGirisi = AylikSatisKumasGirisi + FKilo;

        if (FHamTipiID <> 2) then
        begin
          if (FFason = 1) then AylikFasonOrmeKumasGirisi = AylikFasonOrmeKumasGirisi + FKilo;
          else AylikSatisOrmeKumasGirisi = AylikSatisOrmeKumasGirisi + FKilo;
        end
        else
        begin
          if (FFason = 1) then AylikFasonDokumaKumasGirisi = AylikFasonDokumaKumasGirisi + FKilo;
          else AylikSatisDokumaKumasGirisi = AylikSatisDokumaKumasGirisi + FKilo;
        end

      end
    end
    else
    begin
      AylikTamirGirisi = AylikTamirGirisi + FKilo;

      if (FHamTipiId <> 0) then
      begin
        if (FHamTipiID <> 2) then
        begin
          if (FUcretli = 0) then
            AylikOKumasUcretsizTamirGirisi = AylikOKumasUcretsizTamirGirisi + FKilo;
          else
            AylikOKumasUcretliTamirGirisi = AylikOKumasUcretliTamirGirisi + FKilo;
        end
        else
        begin
          if (FUcretli = 0) then
            AylikDKumasUcretsizTamirGirisi = AylikDKumasUcretsizTamirGirisi + FKilo;
          else
            AylikDKumasUcretliTamirGirisi = AylikDKumasUcretliTamirGirisi + FKilo;
        end
      end
      else
      begin
        if (FUcretli = 0) then
          AylikIplikUcretsizTamirGirisi = AylikIplikUcretsizTamirGirisi + FKilo;
        else
          AylikIplikUcretliTamirGirisi = AylikIplikUcretliTamirGirisi + FKilo;
      end
    end

    if (FIrsaliyeTarihi >= ABaslangicTarihi and FIrsaliyeTarihi <= ABitisTarihi) then
    begin
      if (FTamir = 0) then
      begin
        if (FHamTipiId = 0) then
        begin
          if (FFason = 1) then FasonIplikGirisi = FasonIplikGirisi + FKilo;
          else SatisIplikGirisi = SatisIplikGirisi + FKilo;
        end
        else
        begin
          if (FFason = 1) then FasonKumasGirisi = FasonKumasGirisi + FKilo;
          else SatisKumasGirisi = SatisKumasGirisi + FKilo;

          if (FHamTipiID <> 2) then
          begin
            if (FFason = 1) then FasonOrmeKumasGirisi = FasonOrmeKumasGirisi + FKilo;
            else SatisOrmeKumasGirisi = SatisOrmeKumasGirisi + FKilo;
          end
          else
          begin
            if (FFason = 1) then FasonDokumaKumasGirisi = FasonDokumaKumasGirisi + FKilo;
            else SatisDokumaKumasGirisi = SatisDokumaKumasGirisi + FKilo;
          end
        end
      end
      else
      begin
        TamirGirisi = TamirGirisi + FKilo;

        if (FHamTipiId <> 0) then
        begin
          if (FHamTipiID <> 2) then
          begin
            if (FUcretli = 0) then
              OrmeKumasUcretsizTamirGirisi = OrmeKumasUcretsizTamirGirisi + FKilo;
            else
              OrmeKumasUcretliTamirGirisi = OrmeKumasUcretliTamirGirisi + FKilo;
          end
          else
          begin
            if (FUcretli = 0) then
              DokumaKumasUcretsizTamirGirisi = DokumaKumasUcretsizTamirGirisi + FKilo;
            else
              DokumaKumasUcretliTamirGirisi = DokumaKumasUcretliTamirGirisi + FKilo;
          end
        end
        else
        begin
          if (FUcretli = 0) then
            IplikUcretsizTamirGirisi = IplikUcretsizTamirGirisi + FKilo;
          else
            IplikUcretliTamirGirisi = IplikUcretliTamirGirisi + FKilo;
        end
      end
    end
  end


  suspend;
end
