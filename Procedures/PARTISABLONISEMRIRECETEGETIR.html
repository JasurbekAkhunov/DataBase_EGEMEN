CREATE PROCEDURE PARTISABLONISEMRIRECETEGETIR(
    APARTISABLONISEMRIID INTEGER)
RETURNS (
    SABLONNO INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BARKOD VARCHAR(255),
    KULLANILANMIKTAR DOUBLE PRECISION)
AS
begin
  for select Urun.UrunId, Urun.Adi, Sum(PartiSablonRecete.KullanilanMiktar), PartiSablon.SablonNo, Urun.Barkod from PartiSablon
    left join PartiSablonRecete on PartiSablonRecete.PartiSablonId = PartiSablon.PArtiSablonId
    left join Urun on Urun.UrunId = PartiSablonRecete.UrunId
    left join UrunBirim PSRUB on PSRUB.UrunBirimId = PartiSablonRecete.UrunBirimId and PSRUB.UrunId = PartiSablonRecete.UrunId
    left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
    left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
    left join Kullanici IK on IK.KullaniciId = PartiSablonRecete.KullaniciId
    left join Kullanici SK on SK.KullaniciId = PartiSablonRecete.KullaniciId
    where PartiSablon.PartiSablonIsEmriId = :APartiSablonIsEmriId
    group by PartiSablon.SablonNo, Urun.UrunId, Urun.Adi, Urun.UrunTipiId, Urun.Barkod
    order by PartiSablon.SablonNo, Urun.UrunTipiId
  into :UrunId, :UrunAdi, :KullanilanMiktar, :SablonNo, :Barkod do
  begin

    suspend;
  end
end
