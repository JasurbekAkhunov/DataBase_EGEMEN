CREATE PROCEDURE GUNLUKURETIMRAPORUDETAYLITIP1(
    AMAKINAGRUBU VARCHAR(4),
    AMAKINARAPORGRUPID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP)
RETURNS (
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    RECETETURUID INTEGER,
    RECETETURUADI VARCHAR(15),
    MAKINAID VARCHAR(10),
    MAKINARAPORGRUPID INTEGER,
    MAKINAGRUBU VARCHAR(4),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    CALISMAGUNSAYISI DOUBLE PRECISION,
    KURUBOLUMCALISMAGUNSAYISI DOUBLE PRECISION,
    GUNLUKKAPASITE DOUBLE PRECISION,
    GURAPORGRUPNO INTEGER,
    ORMEKUMASKILO DOUBLE PRECISION,
    ILAVEPROSESORMEKUMASKILO DOUBLE PRECISION,
    UCRETLIICTAMIRORMEKUMASKILO DOUBLE PRECISION,
    UCRETSIZICTAMIRORMEKUMASKILO DOUBLE PRECISION,
    UCRETLIDISTAMIRORMEKUMASKILO DOUBLE PRECISION,
    UCRETSIZDISTAMIRORMEKUMASKILO DOUBLE PRECISION,
    ORMEKUMASAYKILO DOUBLE PRECISION,
    ILAVEPROSESORMEKUMASAYKILO DOUBLE PRECISION,
    UCRETLIICTAMIRORMEKUMASAYKILO DOUBLE PRECISION,
    UCRETSIZICTAMIRORMEKUMASAYKILO DOUBLE PRECISION,
    UCRETLIDISTAMIRORMEKUMASAYKILO DOUBLE PRECISION,
    UCRETSIZDISTAMIRORMEKUMASAYKILO DOUBLE PRECISION,
    PAKETLEMESEKLITUPTOPLAMKILO DOUBLE PRECISION,
    PAKETLEMESEKLITUPAYKILO DOUBLE PRECISION,
    DOKUMAKUMASKILO DOUBLE PRECISION,
    ILAVEPROSESDOKUMAKUMASKILO DOUBLE PRECISION,
    UCRETLIICTAMIRDOKUMAKUMASKILO DOUBLE PRECISION,
    UCRETSIZICTAMIRDOKUMAKUMASKILO DOUBLE PRECISION,
    UCRETLIDISTAMIRDOKUMAKUMASKILO DOUBLE PRECISION,
    UCRETSIZDISTAMIRDOKUMAKUMASKILO DOUBLE PRECISION,
    DOKUMAKUMASAYKILO DOUBLE PRECISION,
    ILAVEPROSESDOKUMAKUMASAYKILO DOUBLE PRECISION,
    UCRETLIICTAMIRDOKUMAAYKILO DOUBLE PRECISION,
    UCRETSIZICTAMIRDOKUMAAYKILO DOUBLE PRECISION,
    UCRETLIDISTAMIRDOKUMAAYKILO DOUBLE PRECISION,
    UCRETSIZDISTAMIRDOKUMAAYKILO DOUBLE PRECISION)
AS
declare variable FIslemTipiId integer;
declare variable FPartiTamirId integer;
declare variable FIlaveProses integer;
declare variable FHamTipiId    integer;
declare variable FCikisTarihi TimeStamp;
declare variable FKilo Double Precision;

declare variable FIslemGrubuId    integer;
declare variable FIslemGrubuAdi   varchar(20);
declare variable FReceteTuruId    integer;
declare variable FReceteTuruAdi   varchar(15);
declare variable FMakinaId        varchar(10);
declare variable FLastMakinaId    varchar(10);
declare variable FGunlukKapasite  Double Precision;
declare variable FAsamaId         integer;
declare variable FAsamaAdi        varchar(40);
declare variable FMakinaRaporGrupId  integer;
declare variable FMakinaGrubu      varchar(4);

declare variable FLastIslemGrubuId    integer;
declare variable FLastReceteTuruId    integer;
declare variable FLastAsamaId         integer;
declare variable FLastMakinaRaporGrupId  integer;
declare variable FAsamaGURaporGrupNo      integer;
declare variable FIslemGrubuGURaporGrupNo integer;
declare variable FReceteTuruGURaporGrupNo integer;
declare variable FPaketlemeSekliTup integer;
declare variable FFaturaKesilecek integer;
declare variable FUcretliTamir integer;
begin
  Select Sum(coalesce(BoyahaneCalismaSaati, 0)), Sum(coalesce(KuruBolumCalismaSaati, 0)) from GunlukCalismaSaati
    where GunlukCalismaSaati.Tarih >= :AAylikBaslangic and GunlukCalismaSaati.Tarih <= :AAylikBitis
    into :CalismaGunSayisi, :KuruBolumCalismaGunSayisi;


  if (CalismaGunSayisi is not null) then CalismaGunSayisi = CalismaGunSayisi / 24;
  else CalismaGunSayisi = AAylikBitis - AAylikBaslangic;
  CalismaGunSayisi = Yuvarla(CalismaGunSayisi, 10);


  if (KuruBolumCalismaGunSayisi is not null) then KuruBolumCalismaGunSayisi = KuruBolumCalismaGunSayisi / 24;
  else KuruBolumCalismaGunSayisi = AAylikBaslangic - AAylikBaslangic;
  KuruBolumCalismaGunSayisi = Yuvarla(KuruBolumCalismaGunSayisi, 10);

  FLastIslemGrubuId = -1;
  FLastReceteTuruId = -1;
  FLastAsamaId      = -1;
  FLastMakinaId     = '';
  FLastMakinaRaporGrupId = -1;

  for select Parti.IslemGrubuId, IslemGrubu.Adi, IslemGrubu.GURaporGrupNo, Recete.ReceteTuruId, ReceteTuru.Adi, ReceteTuru.GURaporGrupNo,
             PartiAsamaStok.AsamaId, Asama.Adi, Asama.GURaporGrupNo, Parti.IslemTipiId, PartiStok.HamTipiId, PartiAsamalari.PartiTamirId,
             PartiAsamaStokDetay.MakinaId, Makina.MakinaGrubu, Makina.RaporGrupId, Makina.GunlukKapasite, PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.CikisTarihi,
             PaketlemeSekli.Tup, PartiAsamalari.IlaveProses, Parti.FaturaKesilecek, TamirCozumSekli.Ucretli from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    left join Unite on Unite.UniteId = Makina.UniteId
    left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join ReceteTuru on ReceteTuru.ReceteTuruId = Recete.ReceteTuruId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliId = BoyaSiparisDetay.PaketlemeSekliId
    left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
    where PartiAsamaStokDetay.CikisTarihi >= :AAylikBaslangic and PartiAsamaStokDetay.CikisTarihi < :AAylikBitis and
          Makina.UretimRaporlarinaDahil = 1 and Asama.GunlukUretimRaporunaDahil = 1 and
         (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
         (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null)
    order by Parti.IslemGrubuId, Recete.ReceteTuruId, PartiAsamaStok.AsamaId /* orderby değiştirme rapor bozulur */
    into :FIslemGrubuId, :FIslemGrubuAdi, :FIslemGrubuGURaporGrupNo, :FReceteTuruId, :FReceteTuruAdi, :FReceteTuruGURaporGrupNo,
         :FAsamaId, :FAsamaAdi, :FAsamaGURaporGrupNo, :FIslemTipiId, :FHamTipiId, :FPartiTamirId, :FMakinaId, :FMakinaGrubu, :FMakinaRaporGrupId,
         :FGunlukKapasite, :FKilo, :FCikisTarihi, :FPaketlemeSekliTup, :FIlaveProses, :FFaturaKesilecek, :FUcretliTamir do
  begin
    if (FMakinaId <> FLastMakinaId or FLastIslemGrubuId <> FIslemgrubuId or FLastMakinaRaporGrupId <> FMakinaRaporGrupId or IntKarsilastir(FLastReceteTuruId, FReceteTuruId) = 0 or FLastAsamaId <> FAsamaId) then
    begin
      if (FLastIslemGrubuId <> -1) then suspend;

      IslemGrubuId = FIslemGrubuId;
      IslemGrubuAdi = FIslemGrubuAdi;
      ReceteTuruId = FReceteTuruId;
      ReceteTuruAdi = FReceteTuruAdi;
      AsamaId = FAsamaId;
      AsamaAdi = FAsamaAdi;
      MakinaRaporGrupId = FMakinaRaporGrupId;
      MakinaGrubu = FMakinaGrubu;
      MakinaID = FMakinaId;
      GunlukKapasite = FGunlukKapasite;
      if (MakinaGrubu is null) then MakinaGrubu = '';

      if (FAsamaGURaporGrupNo is null) then FAsamaGURaporGrupNo = 0;
      if (FIslemGrubuGURaporGrupNo is null) then FIslemGrubuGURaporGrupNo = 0;
      if (FReceteTuruGURaporGrupNo is null) then FReceteTuruGURaporGrupNo = 0;

      if (FAsamaGURaporGrupNo > 0) then GURaporGrupNo = FAsamaGURaporGrupNo;
      else if (FIslemGrubuGURaporGrupNo > 0) then GURaporGrupNo = FIslemGrubuGURaporGrupNo;
      else if (FReceteTuruGURaporGrupNo > 0) then GURaporGrupNo = FReceteTuruGURaporGrupNo;
      else GURaporGrupNo = -1;

      FLastIslemGrubuId = FIslemGrubuId;
      FLastReceteTuruId = FReceteTuruId;
      FLastAsamaId      = FAsamaId;
      FLastMakinaRaporGrupId = FMakinaRaporGrupId;
      FLastMakinaID = FMakinaId;

      OrmeKumasKilo                       = 0;
      UcretliIcTamirOrmeKumasKilo         = 0;
      UcretsizIcTamirOrmeKumasKilo        = 0;
      UcretliDisTamirOrmeKumasKilo        = 0;
      UcretsizDisTamirOrmeKumasKilo       = 0;

      OrmeKumasAyKilo                     = 0;
      UcretliIcTamirOrmeKumasAyKilo       = 0;
      UcretsizIcTamirOrmeKumasAyKilo      = 0;
      UcretliDisTamirOrmeKumasAyKilo      = 0;
      UcretsizDisTamirOrmeKumasAyKilo     = 0;

      DokumaKumasKilo              = 0;
      DokumaKumasAyKilo            = 0;

      UcretliIcTamirDokumaKumasKilo    = 0;
      UcretsizIcTamirDokumaKumasKilo   = 0;
      UcretliDisTamirDokumaKumasKilo   = 0;
      UcretsizDisTamirDokumaKumasKilo  = 0;

      UcretliIcTamirDokumaAyKilo    = 0;
      UcretsizIcTamirDokumaAyKilo   = 0;
      UcretliDisTamirDokumaAyKilo   = 0;
      UcretsizDisTamirDokumaAyKilo  = 0;

      PaketlemeSekliTupToplamKilo  = 0;
      PaketlemeSekliTupAyKilo      = 0;

      IlaveProsesOrmeKumasKilo     = 0;
      IlaveProsesOrmeKumasAyKilo   = 0;
      IlaveProsesDokumaKumasKilo   = 0;
      IlaveProsesDokumaKumasAyKilo = 0;

    end

    if (FCikisTarihi >= ABaslangicTarihi and FCikisTarihi < ABitisTarihi) then
    begin
      if (FHamTipiID <> 2) then
      begin
        if (FIlaveProses = 1) then IlaveProsesOrmeKumasKilo = IlaveProsesOrmeKumasKilo + FKilo;
        else if (FPartiTamirId is not null) then
        begin
          if (FUcretliTamir = 1) then
            UcretliIcTamirOrmeKumasKilo = UcretliIcTamirOrmeKumasKilo + FKilo;
          else
            UcretsizIcTamirOrmeKumasKilo = UcretsizIcTamirOrmeKumasKilo + FKilo;
        end
        else if (FIslemTipiId = 3) then
        begin
          if (FFaturaKesilecek = 1) then
            UcretliDisTamirOrmeKumasKilo = UcretliDisTamirOrmeKumasKilo + FKilo;
          else
            UcretsizDisTamirOrmeKumasKilo = UcretsizDisTamirOrmeKumasKilo + FKilo;
        end
        else
        begin
          if (FPaketlemeSekliTup is not null and FPaketlemeSekliTup = 1) then
            PaketlemeSekliTupToplamKilo = PaketlemeSekliTupToplamKilo + FKilo;

          OrmeKumasKilo = OrmeKumasKilo + FKilo;
        end
      end
      else
      begin
        if (FIlaveProses = 1) then IlaveProsesDokumaKumasKilo = IlaveProsesDokumaKumasKilo + FKilo;
        else if (FPartiTamirId is not null) then
        begin
          if (FUcretliTamir = 1) then
            UcretliIcTamirDokumaKumasKilo = UcretliIcTamirDokumaKumasKilo + FKilo;
          else
            UcretsizIcTamirDokumaKumasKilo = UcretsizIcTamirDokumaKumasKilo + FKilo;
        end
        else if (FIslemTipiId = 3) then
        begin
          if (FFaturaKesilecek = 1) then
            UcretliDisTamirDokumaKumasKilo = UcretliDisTamirDokumaKumasKilo + FKilo;
          else
            UcretsizDisTamirDokumaKumasKilo = UcretsizDisTamirDokumaKumasKilo + FKilo;

        end
        else DokumaKumasKilo = DokumaKumasKilo + FKilo;
      end
    end

    if (FHamTipiID <> 2) then
    begin
      if (FIlaveProses = 1) then IlaveProsesOrmeKumasAyKilo = IlaveProsesOrmeKumasAyKilo + FKilo;
      else if (FPartiTamirId is not null) then
      begin
        if (FUcretliTamir = 1) then
          UcretliIcTamirOrmeKumasAyKilo = UcretliIcTamirOrmeKumasAyKilo + FKilo;
        else
          UcretsizIcTamirOrmeKumasAyKilo = UcretsizIcTamirOrmeKumasAyKilo + FKilo;

      end
      else if (FIslemTipiId = 3) then
      begin
        if (FFaturaKesilecek = 1) then
          UcretliDisTamirOrmeKumasAyKilo = UcretliDisTamirOrmeKumasAyKilo + FKilo;
        else
          UcretsizDisTamirOrmeKumasAyKilo = UcretsizDisTamirOrmeKumasAyKilo + FKilo;
      end
      else
      begin
        if (FPaketlemeSekliTup is not null and FPaketlemeSekliTup = 1) then
          PaketlemeSekliTupAyKilo = PaketlemeSekliTupAyKilo + FKilo;
        OrmeKumasAyKilo = OrmeKumasAyKilo + FKilo;
      end
    end
    else
    begin
      if (FIlaveProses = 1) then IlaveProsesDokumaKumasAyKilo = IlaveProsesDokumaKumasAyKilo + FKilo;
      else if (FPartiTamirId is not null) then
      begin
        if (FUcretliTamir = 1) then
          UcretliIcTamirDokumaAyKilo = UcretliIcTamirDokumaAyKilo + FKilo;
        else
          UcretsizIcTamirDokumaAyKilo = UcretsizIcTamirDokumaAyKilo + FKilo;
      end
      else if (FIslemTipiId = 3) then
      begin
        if (FFaturaKesilecek = 1) then
          UcretliDisTamirDokumaAyKilo = UcretliDisTamirDokumaAyKilo + FKilo;
        else
          UcretsizDisTamirDokumaAyKilo = UcretsizDisTamirDokumaAyKilo + FKilo;
      end
      else DokumaKumasAyKilo = DokumaKumasAyKilo + FKilo;
    end
  end

  if (FLastIslemGrubuId <> -1) then suspend;
end
