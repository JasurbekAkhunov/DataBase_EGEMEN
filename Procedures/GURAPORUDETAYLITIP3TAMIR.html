CREATE PROCEDURE GURAPORUDETAYLITIP3TAMIR(
    AMAKINAGRUBU VARCHAR(4),
    AMAKINARAPORGRUPID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP,
    AIMALATTIPIID INTEGER,
    APARTITIPIID INTEGER)
RETURNS (
    PARTITIPIID INTEGER,
    IMALATTIPIID INTEGER,
    TAMIRTIPIID INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    ICTAMIR INTEGER,
    ORMEKUMASKILO DOUBLE PRECISION,
    ILAVEPROSESORMEKUMASKILO DOUBLE PRECISION,
    UCRETLITAMIRORMEKUMASKILO DOUBLE PRECISION,
    UCRETSIZTAMIRORMEKUMASKILO DOUBLE PRECISION,
    FUCRETLITAMIRORMEKUMASKILO DOUBLE PRECISION,
    FUCRETSIZTAMIRORMEKUMASKILO DOUBLE PRECISION,
    ORMEKUMASAYKILO DOUBLE PRECISION,
    ILAVEPROSESORMEKUMASAYKILO DOUBLE PRECISION,
    UCRETLITAMIRORMEKUMASAYKILO DOUBLE PRECISION,
    UCRETSIZTAMIRORMEKUMASAYKILO DOUBLE PRECISION,
    FUCRETLITAMIRORMEKUMASAYKILO DOUBLE PRECISION,
    FUCRETSIZTAMIRORMEKUMASAYKILO DOUBLE PRECISION,
    DOKUMAKUMASKILO DOUBLE PRECISION,
    ILAVEPROSESDOKUMAKUMASKILO DOUBLE PRECISION,
    UCRETLITAMIRDOKUMAKUMASKILO DOUBLE PRECISION,
    UCRETSIZTAMIRDOKUMAKUMASKILO DOUBLE PRECISION,
    FUCRETLITAMIRDOKUMAKUMASKILO DOUBLE PRECISION,
    FUCRETSIZTAMIRDOKUMAKUMASKILO DOUBLE PRECISION,
    DOKUMAKUMASAYKILO DOUBLE PRECISION,
    ILAVEPROSESDOKUMAKUMASAYKILO DOUBLE PRECISION,
    UCRETLITAMIRDOKUMAAYKILO DOUBLE PRECISION,
    UCRETSIZTAMIRDOKUMAAYKILO DOUBLE PRECISION,
    FUCRETLITAMIRDOKUMAAYKILO DOUBLE PRECISION,
    FUCRETSIZTAMIRDOKUMAAYKILO DOUBLE PRECISION,
    SATISMETRE DOUBLE PRECISION,
    FASONMETRE DOUBLE PRECISION,
    SATISAYMETRE DOUBLE PRECISION,
    FASONAYMETRE DOUBLE PRECISION,
    SATISKILO DOUBLE PRECISION,
    FASONKILO DOUBLE PRECISION,
    SATISAYKILO DOUBLE PRECISION,
    FASONAYKILO DOUBLE PRECISION)
AS
declare variable FPartiAsamalariId integer;
declare variable FPartiKayitId integer;
declare variable FPartiTipiId integer;
declare variable FImalatTipiId integer;
declare variable FTamirTipiId integer;
declare variable FTamirTipiAdi varchar(20);
declare variable FPStokTamirTipiId integer;
declare variable FPStokTamirTipiAdi varchar(20);
declare variable FIslemTipiId integer;
declare variable FHamTipiId    integer;
declare variable FCikisTarihi TimeStamp;
declare variable FKilo Double Precision;
declare variable FMetre Double Precision;
declare variable FLastPartiTipiId integer;
declare variable FLastImalatTipiId integer;
declare variable FLastTamirTipiId integer;
declare variable FLastPStokTamirTipiId integer;
declare variable FLastIslemTipiId integer;
declare variable FFaturaKesilecek integer;
declare variable FUcretliTamir integer;
declare variable FFason integer;
begin


  FLastPartiTipiId = -1;
  FLastImalatTipiId = -1;
  FLastTamirTipiId = -1;
  FLastPStokTamirTipiId = -1;
  FLastIslemTipiId = -1;

  for select PartiAsamalari.PartiAsamalariId, Parti.PartiKayitId, Parti.PartiTipiId,
             Parti.ImalatTipiId, PartiStok.HamTipiId,
             PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.CikisTarihi,
             Parti.FaturaKesilecek, TamirCozumSekli.Ucretli, PartiAsamaStokDetay.Metre, Parti.Fason,
             PartiTamir.TamirTipiId, TamirTipi.Adi, Parti.IslemTipiId, PartiStok.TamirTipiId, PStokTamirTipi.Adi from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join PartiTamir on PartiTamir.PartiTamirid = PartiAsamalari.PartiTamirId
    left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
    left join TamirTipi PStokTamirTipi on PStokTamirTipi.TamirTipiId = PartiStok.TamirTipiId
    left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
    where PartiAsamaStokDetay.Cikistarihi >= :AAylikBaslangic and PartiAsamaStokDetay.Cikistarihi < :AAylikBitis and
          Makina.UretimRaporlarinaDahil = 1 and Asama.GunlukUretimRaporunaDahil = 1 and
         (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
         (Makina.RaporGrupId = :AMakinaRaporGrupId or :AMakinaRaporGrupId is null) and
         (Parti.ImalatTipiId = :AImalatTipiId or :AImalatTipiId is Null) and
         (Parti.PartiTipiId = :APartiTipiId or :APartiTipiId is null) and
          (PartiTamir.TamirTipiId is not Null or Parti.IslemTipiId = 3)
    order by Parti.PartiTipiId, Parti.ImalatTipiId, Parti.IslemTipiId, PartiTamir.TamirTipiId, PartiStok.TamirTipiId /* orderby değiştirme rapor bozulur */
    into :FPartiAsamalariId, :FPartiKayitId, :FPartiTipiId,
         :FImalatTipiId, :FHamTipiId,
         :FKilo, :FCikisTarihi, :FFaturaKesilecek, :FUcretliTamir, :FMetre, :FFason,
         :FTamirTipiId, :FTamirTipiAdi, :FIslemTipiId, :FPStokTamirTipiId, :FPStokTamirTipiAdi do
  begin

    if (FLastPartiTipiId <> FPartiTipiId or FLastImalatTipiId <> FImalatTipiId or IntKarsilastir(FLastTamirTipiId, FTamirTipiId) <> 1 or IntKarsilastir(FLastPStokTamirTipiId, FPStokTamirTipiId) <> 1) then
    begin
      if (FLastPartiTipiId <> -1) then suspend;


      PartiTipiId = FPartiTipiId;
      ImalatTipiId = FImalatTipiId;
      TamirTipiId = FTamirTipiId;
      TamirTipiAdi = FTamirTipiAdi;

      FLastPartiTipiId = FPartiTipiId;
      FLastImalatTipiId = FImalatTipiId;
      FLastIslemTipiId = FIslemTipiId;
      FLastTamirTipiId = FTamirTipiId;
      FLastPStokTamirTipiId = FPStokTamirTipiId;

      IcTamir = 1;
      if (FIslemTipiId = 3) then
      begin
        IcTamir = 0;
        TamirTipiAdi = FPStokTamirTipiAdi;
        TamirTipiId = FPStokTamirTipiId;
      end

      OrmeKumasKilo                       = 0;
      UcretliTamirOrmeKumasKilo         = 0;
      UcretsizTamirOrmeKumasKilo        = 0;
      FUcretliTamirOrmeKumasKilo         = 0;
      FUcretsizTamirOrmeKumasKilo        = 0;

      OrmeKumasAyKilo                     = 0;
      UcretliTamirOrmeKumasAyKilo       = 0;
      UcretsizTamirOrmeKumasAyKilo      = 0;
      FUcretliTamirOrmeKumasAyKilo       = 0;
      FUcretsizTamirOrmeKumasAyKilo      = 0;

      DokumaKumasKilo              = 0;
      DokumaKumasAyKilo            = 0;

      UcretliTamirDokumaKumasKilo    = 0;
      UcretsizTamirDokumaKumasKilo   = 0;

      FUcretliTamirDokumaKumasKilo    = 0;
      FUcretsizTamirDokumaKumasKilo   = 0;

      UcretliTamirDokumaAyKilo    = 0;
      UcretsizTamirDokumaAyKilo   = 0;

      FUcretliTamirDokumaAyKilo    = 0;
      FUcretsizTamirDokumaAyKilo   = 0;

      SatisMetre = 0;
      FasonMetre = 0;
      SatisAyMetre = 0;
      FasonAyMetre = 0;

      SatisKilo = 0;
      FasonKilo = 0;
      SatisAyKilo = 0;
      FasonAyKilo = 0;

    end

    if (FCikisTarihi >= ABaslangicTarihi and FCikisTarihi < ABitisTarihi) then
    begin
      if (FFason = 0) then SatisMetre = SatisMetre + FMetre;
      else FasonMetre = FasonMetre + FMetre;

      if (FFason = 0) then SatisKilo = SatisKilo + FKilo;
      else FasonKilo = FasonKilo + FKilo;

      if (FHamTipiID <> 2) then
      begin
        if (FTamirTipiId is not null) then
        begin
          if (FFason = 0) then
          begin
            if (FUcretliTamir = 1) then
              UcretliTamirOrmeKumasKilo = UcretliTamirOrmeKumasKilo + FKilo;
            else
              UcretsizTamirOrmeKumasKilo = UcretsizTamirOrmeKumasKilo + FKilo;
          end
          else
          begin
            if (FUcretliTamir = 1) then
              FUcretliTamirOrmeKumasKilo = FUcretliTamirOrmeKumasKilo + FKilo;
            else
              FUcretsizTamirOrmeKumasKilo = FUcretsizTamirOrmeKumasKilo + FKilo;
          end
        end
        else if (FIslemTipiId = 3) then
        begin
          if (FFason = 0) then
          begin
            if (FFaturaKesilecek = 1) then
              UcretliTamirOrmeKumasKilo = UcretliTamirOrmeKumasKilo + FKilo;
            else
              UcretsizTamirOrmeKumasKilo = UcretsizTamirOrmeKumasKilo + FKilo;
          end
          else
          begin
            if (FFaturaKesilecek = 1) then
              FUcretliTamirOrmeKumasKilo = FUcretliTamirOrmeKumasKilo + FKilo;
            else
              FUcretsizTamirOrmeKumasKilo = FUcretsizTamirOrmeKumasKilo + FKilo;
          end
        end
        OrmeKumasKilo = OrmeKumasKilo + FKilo;
      end
      else
      begin
        if (FTamirTipiId is not null) then
        begin
          if (FFason = 0) then
          begin
            if (FUcretliTamir = 1) then
              UcretliTamirDokumaKumasKilo = UcretliTamirDokumaKumasKilo + FKilo;
            else
              UcretsizTamirDokumaKumasKilo = UcretsizTamirDokumaKumasKilo + FKilo;
          end
          else
          begin
            if (FUcretliTamir = 1) then
              FUcretliTamirDokumaKumasKilo = FUcretliTamirDokumaKumasKilo + FKilo;
            else
              FUcretsizTamirDokumaKumasKilo = FUcretsizTamirDokumaKumasKilo + FKilo;
          end
        end
        else if (FIslemTipiId = 3) then
        begin
          if (FFason = 0) then
          begin
            if (FFaturaKesilecek = 1) then
              UcretliTamirDokumaKumasKilo = UcretliTamirDokumaKumasKilo + FKilo;
            else
              UcretsizTamirDokumaKumasKilo = UcretsizTamirDokumaKumasKilo + FKilo;
          end
          else
          begin
            if (FFaturaKesilecek = 1) then
              FUcretliTamirDokumaKumasKilo = FUcretliTamirDokumaKumasKilo + FKilo;
            else
              FUcretsizTamirDokumaKumasKilo = FUcretsizTamirDokumaKumasKilo + FKilo;
          end
        end
        DokumaKumasKilo = DokumaKumasKilo + FKilo;
      end
    end

    if (FFason = 0) then SatisAyMetre = SatisAyMetre + FMetre;
    else FasonAyMetre = FasonAyMetre + FMetre;

    if (FFason = 0) then SatisAyKilo = SatisAyKilo + FKilo;
    else FasonAyKilo = FasonAyKilo + FKilo;
    if (FHamTipiID <> 2) then
    begin
      if (FTamirTipiId is not null) then
      begin
        if (FFason = 0) then
        begin
          if (FUcretliTamir = 1) then
            UcretliTamirOrmeKumasAyKilo = UcretliTamirOrmeKumasAyKilo + FKilo;
          else
            UcretsizTamirOrmeKumasAyKilo = UcretsizTamirOrmeKumasAyKilo + FKilo;
        end
        else
        begin
          if (FUcretliTamir = 1) then
            FUcretliTamirOrmeKumasAyKilo = FUcretliTamirOrmeKumasAyKilo + FKilo;
          else
            FUcretsizTamirOrmeKumasAyKilo = FUcretsizTamirOrmeKumasAyKilo + FKilo;
        end
      end
      else if (FIslemTipiId = 3) then
      begin
        if (FFason = 0) then
        begin
          if (FFaturaKesilecek = 1) then
            UcretliTamirOrmeKumasAyKilo = UcretliTamirOrmeKumasAyKilo + FKilo;
          else
            UcretsizTamirOrmeKumasAyKilo = UcretsizTamirOrmeKumasAyKilo + FKilo;
        end
        else
        begin
          if (FFaturaKesilecek = 1) then
            FUcretliTamirOrmeKumasAyKilo = FUcretliTamirOrmeKumasAyKilo + FKilo;
          else
            FUcretsizTamirOrmeKumasAyKilo = FUcretsizTamirOrmeKumasAyKilo + FKilo;
        end
      end
      OrmeKumasAyKilo = OrmeKumasAyKilo + FKilo;
    end
    else
    begin
      if (FTamirTipiId is not null) then
      begin
        if (FFason = 0) then
        begin
          if (FUcretliTamir = 1) then
            UcretliTamirDokumaAyKilo = UcretliTamirDokumaAyKilo + FKilo;
          else
            UcretsizTamirDokumaAyKilo = UcretsizTamirDokumaAyKilo + FKilo;
        end
        else
        begin
          if (FUcretliTamir = 1) then
            FUcretliTamirDokumaAyKilo = FUcretliTamirDokumaAyKilo + FKilo;
          else
            FUcretsizTamirDokumaAyKilo = FUcretsizTamirDokumaAyKilo + FKilo;
        end
      end
      else if (FIslemTipiId = 3) then
      begin
        if (FFason = 0) then
        begin
          if (FFaturaKesilecek = 1) then
            UcretliTamirDokumaAyKilo = UcretliTamirDokumaAyKilo + FKilo;
          else
            UcretsizTamirDokumaAyKilo = UcretsizTamirDokumaAyKilo + FKilo;
        end
        else
        begin
          if (FFaturaKesilecek = 1) then
            FUcretliTamirDokumaAyKilo = FUcretliTamirDokumaAyKilo + FKilo;
          else
            FUcretsizTamirDokumaAyKilo = FUcretsizTamirDokumaAyKilo + FKilo;
        end
      end
      DokumaKumasAyKilo = DokumaKumasAyKilo + FKilo;
    end
  end

  if (FLastPartiTipiId <> -1) then suspend;
end
