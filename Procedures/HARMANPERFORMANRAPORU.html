CREATE PROCEDURE HARMANPERFORMANRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHARMANKAYITID INTEGER)
RETURNS (
    HARMANKAYITID INTEGER,
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    ADI VARCHAR(50),
    IPLIKISLETMEID INTEGER,
    IPLIKISLETMEADI VARCHAR(30),
    TARIHI TIMESTAMP,
    HARMANMIKTARI DOUBLE PRECISION,
    HARMANCINSIID INTEGER,
    HARMANCINSIADI VARCHAR(50),
    HARMANKARISIMID INTEGER,
    HARMANKARISIMADI VARCHAR(20),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ANALOT VARCHAR(50),
    KAPALI INTEGER,
    KARISIKHARMAN INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    MAXURETIMTARIHI TIMESTAMP,
    MINURETIMTARIHI TIMESTAMP,
    HARMANBOLGESIID ADI10,
    HARMANBOLGESIADI ADI50,
    KARISIKHARMANORANIID ADI20,
    KARISIKHARMANORANIADI ADI50,
    RENKADI ADI50,
    RENKDIGERADI ADI50)
AS
begin
  For Select Harman.HarmanKayitId, Harman.HarmanYil, Harman.HarmanId, Harman.Adi, Harman.Tarihi, Harman.HarmanCinsiId, HarmanCinsi.Adi,
             Harman.HarmanMiktari, Harman.IplikIsletmeId, IplikIsletme.Adi, Harman.Notu, Harman.Kapali, Ik.Adi, Harman.InsertTarihi, SK.Adi, Harman.DegisimTarihi,
             Harman.AnaLot, Harman.KarisikHarman, Harman.HarmanKarisimId, HarmanKarisim.Adi, HarmanBolgesi.HarmanBolgesiId, HarmanBolgesi.Adi,
             KarisikHarmanOrani.KarisikHarmanOraniId, KarisikHarmanOrani.Adi, Renk.Adi, Renk.DigerAdi
  from Harman
  Left Join HarmanCinsi on HarmanCinsi.HarmanCinsiId = Harman.HarmanCinsiId
  Left Join HarmanKarisim on HarmanKarisim.HarmanKarisimID = Harman.HarmanKarisimID
  Left Join IplikIsletme on IplikIsletme.IplikIsletmeId = Harman.IplikIsletmeId
  Left Join Kullanici IK on IK.KullaniciId = Harman.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = Harman.KullaniciId
  left join HarmanBolgesi on HarmanBolgesi.HarmanBolgesiId = Harman.HarmanBolgesiId
  left join Renk on Renk.RenkId = Harman.RenkId
  left join KarisikHarmanOrani on KarisikHarmanOrani.KarisikHarmanOraniId = Harman.KarisikHarmanOraniId
  where Harman.Tarihi >= :ABaslangicTarihi and Harman.Tarihi <= :ABitisTarihi and
        (Harman.HarmanKayitId = :AHarmanKayitId or :AHarmanKayitId is null)
  into :HarmanKayitId, :HarmanYil, :HarmanId, :Adi, :Tarihi, :HarmanCinsiId, :HarmanCinsiAdi, :HarmanMiktari,
       :IplikIsletmeId, :IplikIsletmeAdi,  :Notu, :Kapali,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :AnaLot, :KarisikHarman, :HarmanKarisimId, :HarmanKarisimAdi,
       :HarmanBolgesiId, :HarmanBolgesiAdi, :KarisikHarmanOraniId, :KarisikHarmanOraniAdi, :RenkAdi, :RenkDigerAdi do
  begin
    HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') ' || EgemenIntegerToStr(HarmanId) || ' ';

    select Min(IplikUretim.UretimTarihi), Max(IplikUretim.UretimTarihi) from IplikLot
    left join IplikUretim on IplikUretim.IplikLotId = IplikLot.IplikLotId
    where IplikLot.HarmanKayitId = :HarmanKayitId into :MinUretimTarihi, :MaxUretimTarihi;

    suspend;
  end
end
