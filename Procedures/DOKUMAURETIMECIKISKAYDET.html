CREATE PROCEDURE DOKUMAURETIMECIKISKAYDET(
    AEVRAKTIPI INTEGER,
    AORMEFISIRSALIYEID INTEGER,
    ADOKUMAPLANLAMAID INTEGER,
    AORMERAFID INTEGER,
    AMIKTAR NUMERIC(12,3),
    AMAKINAID VARCHAR(10),
    AKULLANICIID INTEGER)
AS
declare variable HareketTipi integer;
declare variable RenkId      integer;
declare variable Lot         varchar(150);
declare variable MelanjKodu  varchar(20);
declare variable UreticiId   integer;
declare variable OrmeFisId   integer;
declare variable OrmeIrsaliyeId   integer;
declare variable DepoId      varchar(10);
declare variable LeventNo    integer;
declare variable TelAdedi    integer;
declare variable DepoPartiNo varchar(30);
declare variable AtkiLotu    varchar(30);
begin
  if (AEvrakTipi = 0) then
  begin
    OrmeFisID = AOrmeFisIrsaliyeId;
    Select OrmeFis.HareketTipi from OrmeFis where OrmeFis.OrmeFisID = :OrmeFisId into :HareketTipi;

    Select OrmeRaf.RenkId, OrmeRaf.Lot, OrmeRaf.UreticiId, OrmeRaf.MelanjKodu, DepoRaf.DepoId, OrmeRaf.LeventNo, OrmeRaf.DepoPartiNo, OrmeRaf.TelAdedi, OrmeRaf.AtkiLotu from OrmeRaf
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      where OrmeRaf.OrmeRafID = :AOrmeRafId
      into :RenkId, :Lot, :UreticiId, :MelanjKodu, :DepoId, :LeventNo, :DepoPartiNo, :TelAdedi, :AtkiLotu;

    insert into Stokhareket(OrmeFisId, OrmeIrsaliyeId, HareketTipi, DokumaHareketi, Iade, HamIade, OrmeSiparisDetayId, BoyaSiparisKayitId, DokumaPlanlamaId, HamId, OrmeRafId,
                            Miktar, DokumaMakinaId, RenkId, Lot, UreticiId, MelanjKodu, DepoId, LeventNo, DepoPartiNo, TelAdedi, AtkiLotu, InsertKullaniciId, KullaniciId)
      Select :OrmeFisId, :OrmeIrsaliyeId, :HareketTipi, 1, 0, 0, DokumaPlanlama.OrmeSiparisDetayId, DokumaPlanlama.BoyaSiparisKayitId, :ADokumaPlanlamaId, OrmeRaf.HamId, :AOrmeRafId,
                            :AMiktar, DokumaPlanlama.DokumaMakinaId, :RenkId, :Lot, :UreticiId, :MelanjKodu, :DepoId, :LeventNo, :DepoPartiNo, :TelAdedi, :AtkiLotu, :AKullaniciId, :AKullaniciId from DokumaPlanlama
        left Join OrmeRaf on OrmeRaf.OrmeRafID = :AOrmeRafId
        where DokumaPlanlama.DokumaPlanlamaId = :ADokumaPlanlamaId;
  end
  else
  begin
    OrmeIrsaliyeID = AOrmeFisIrsaliyeId;
    Select OrmeIrsaliye.HareketTipi from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = :OrmeIrsaliyeId into :HareketTipi;

    Select OrmeRaf.RenkId, OrmeRaf.Lot, OrmeRaf.UreticiId, OrmeRaf.MelanjKodu, DepoRaf.DepoId, OrmeRaf.LeventNo, OrmeRaf.DepoPartiNo, OrmeRaf.TelAdedi, OrmeRaf.AtkiLotu from OrmeRaf
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      where OrmeRaf.OrmeRafID = :AOrmeRafId
      into :RenkId, :Lot, :UreticiId, :MelanjKodu, :DepoId, :LeventNo, :DepoPartiNo, :TelAdedi, :AtkiLotu;

    insert into Stokhareket(OrmeFisId, OrmeIrsaliyeId, HareketTipi, DokumaHareketi, Iade, HamIade, OrmeSiparisDetayId, BoyaSiparisKayitId, DokumaPlanlamaId, HamId, OrmeRafId,
                            Miktar, DokumaMakinaId, RenkId, Lot, UreticiId, MelanjKodu, DepoId, LeventNo, DepoPartiNo, TelAdedi, AtkiLotu, InsertKullaniciId, KullaniciId)
      Select :OrmeFisId, :OrmeIrsaliyeId, :HareketTipi, 1, 0, 0, DokumaPlanlama.OrmeSiparisDetayId, DokumaPlanlama.BoyaSiparisKayitId, :ADokumaPlanlamaId, OrmeRaf.HamId, :AOrmeRafId,
                            :AMiktar, :AMakinaId, :RenkId, :Lot, :UreticiId, :MelanjKodu, :DepoId, :LeventNo, :DepoPartiNo, :TelAdedi, :AtkiLotu, :AKullaniciId, :AKullaniciId from DokumaPlanlama
        left Join OrmeRaf on OrmeRaf.OrmeRafID = :AOrmeRafId
        where DokumaPlanlama.DokumaPlanlamaId = :ADokumaPlanlamaId;
  end
end
