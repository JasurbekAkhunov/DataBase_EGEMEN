CREATE PROCEDURE DOKUMAIPLIKTAKIPRAPORU(
    ABASLAMATARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AHAMID INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMADI ADI255,
    LOT VARCHAR(150),
    FIILIDENYE DOUBLE PRECISION,
    FIRMAID FIRMAKODUSIZE,
    FIRMAADI ADI100,
    MIKTAR BOYAMIKTARSIZE,
    KALANMIKTAR BOYAMIKTARSIZE)
AS
declare variable shtAlis Integer;
declare variable hstIplik Integer;
begin
  /*iki tarih arasi alinan iplikler*/
  select shtAlis  from OrmeStokHareketTipiGetir into :shtAlis;
  select hstIplik from HamTipiGetir into :hstIplik;

  /*Kalan miktar stok giris no devrede olan mantiga göre yapıldı. Aksi halde hata versin diye ic sorgu yapildi*/
  for select Ham.HamId, Ham.Adi, Coalesce(StokHareket.Lot, ''), Coalesce(StokHareket.FiiliDenye, 0), Firma.FirmaId, Firma.Adi,
             Sum(StokHareket.Miktar), Sum((Select OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari
                                           from OrmeRafDetay
                                           left join OrmeRaf on OrmeRaf.OrmeRafId = OrmeRafDetay.OrmeRafId
                                           where OrmeRafDetay.StokHareketId = StokHareket.StokHareketId))
      from StokHareket
      left join Ham on Ham.HamId = StokHareket.HamId
      left join Depo on Depo.DepoId = StokHareket.DepoId
      left join Firma on Firma.FirmaId = StokHareket.FirmaId
      where StokHareket.Tarih >= :ABaslamaTarihi and
            StokHareket.Tarih <= :ABitisTarihi and
            StokHareket.HareketTipi = :shtAlis and
            StokHareket.DokumaHareketi = 1 and
            StokHareket.HamIade = 0 and
            (StokHareket.HamId = :AHamId or :AHamId is null) and
            Ham.HamTipiId = :hstIplik and
            Depo.CozguDeposu = 0 and
            Depo.HasilDeposu = 0 and
            Depo.IndigoDeposu = 0
      group by Ham.HamId, Ham.Adi, StokHareket.Lot, StokHareket.FiiliDenye, Firma.FirmaId, Firma.Adi
      into :HamId, :HamAdi, :Lot, :FiiliDenye, :FirmaId, :FirmaAdi, :Miktar, :KalanMiktar
  do
  begin
    suspend;
  end
end
