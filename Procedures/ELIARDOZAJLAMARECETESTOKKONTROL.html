CREATE PROCEDURE ELIARDOZAJLAMARECETESTOKKONTROL(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER)
RETURNS (
    SIRANO INTEGER,
    URUNID VARCHAR(20),
    DOZAJLAMAURUNID VARCHAR(15),
    DOZAJLAMAPROGRAMNO INTEGER,
    URUNTIPIID INTEGER,
    MIKTAR DOUBLE PRECISION,
    BANYO DOUBLE PRECISION,
    OTOMASYONMIKTAR DOUBLE PRECISION,
    DOZAJLAMABIRIMKODU VARCHAR(10),
    URUNADI VARCHAR(80),
    OTOMASYONRECETEMIKTAR DOUBLE PRECISION,
    STOKYETERSIZ SMALLINT,
    STOKMIKTARI DOUBLE PRECISION)
AS
declare variable ReceteAsamaId         integer;
declare variable ReceteAltAsamaId      integer;
declare variable ReceteMiktar double precision;
declare variable ReceteBanyo double precision;

declare variable FUrunId               varchar(20);
declare variable DevirMiktari          double precision;
declare variable AlisMiktari           double precision;
declare variable AlisIadesiMiktari     double precision;
declare variable UretimMiktari         double precision;
declare variable SatisMiktari          double precision;
declare variable SatisIadesiMiktari    double precision;
declare variable UretimeCikisMiktari   double precision;
declare variable DepolaraCikanMiktar   double precision;
declare variable DepolardanGelenMiktar double precision;

declare variable FasonUretimMiktari    double precision;
declare variable FasonUretimIadeMiktari double precision;
declare variable FasonUretimeCikisMiktari double precision;
begin
  SiraNo = 0;
  StokYetersiz = 0;

  for select DozajlamaIcinReceteHazirla.SiraNo, DozajlamaIcinReceteHazirla.ReceteAsamaId, DozajlamaIcinReceteHazirla.ReceteAltAsamaId, DozajlamaIcinReceteHazirla.UrunId, DozajlamaIcinReceteHazirla.DozajlamaUrunId,
             DozajlamaIcinReceteHazirla.UrunTipiId, DozajlamaIcinReceteHazirla.Miktar, DozajlamaIcinReceteHazirla.Banyo,
             DozajlamaIcinReceteHazirla.DozajlamaProgramNo,
             DozajlamaIcinReceteHazirla.DozajlamaBirimKodu,  DozajlamaIcinReceteHazirla.UrunAdi, DozajlamaIcinReceteHazirla.ReceteMiktar, DozajlamaIcinReceteHazirla.ReceteBanyo from DozajlamaIcinReceteHazirla(:APartiKayitId, :AReceteId, :APartiAsamalariId, 2)
    where UrunID is not null
    into :SiraNo, :ReceteAsamaId, :ReceteAltAsamaId, :UrunId, :DozajlamaUrunId, :UrunTipiId, :Miktar, :Banyo, :DozajlamaProgramNo,
    :DozajlamaBirimKodu, :UrunAdi, :ReceteMiktar, :ReceteBanyo do
  begin
   SiraNo = SiraNo + 1;
   StokMiktari              = 0;
   DevirMiktari             = 0;
   AlisMiktari              = 0;
   AlisIadesiMiktari        = 0;
   SatisMiktari             = 0;
   SatisIadesiMiktari       = 0;
   UretimeCikisMiktari      = 0;
   DepolaraCikanMiktar      = 0;
   DepolardanGelenMiktar    = 0;
   UretimMiktari            = 0;
   FasonUretimeCikisMiktari = 0;
   FasonUretimMiktari       = 0;
   FasonUretimIadeMiktari   = 0;
   FUrunId                  = UrunId;
   StokYetersiz              = 0;

   Select Urun.UrunId,
             Sum(DepoStok.Devir * UrunBirim.Carpan), Sum(DepoStok.Alis * UrunBirim.Carpan), Sum(DepoStok.AlisIadesi * UrunBirim.Carpan),
             Sum(DepoStok.Satis * UrunBirim.Carpan), Sum(DepoStok.SatisIadesi * UrunBirim.Carpan),
             Sum(DepoStok.UretimeCikis * UrunBirim.Carpan), Sum(DepoStok.DepoyaCikan * UrunBirim.Carpan),
             Sum(DepoStok.DepodanGelen * UrunBirim.Carpan), Sum(DepoStok.Uretim * UrunBirim.Carpan),
             Sum(DepoStok.FasonUretimeCikis * UrunBirim.Carpan), Sum(DepoStok.FasonUretim * UrunBirim.Carpan), Sum(DepoStok.FasonUretimIade * UrunBirim.Carpan) from DepoStok
    left join Urun on Urun.UrunId = DepoStok.UrunId
    left join UrunBirim on UrunBirim.urunId = DepoStok.UrunId and UrunBirim.DefaultBirim = 1
    where urun.urunId = :FUrunId and (Urun.Kapali = 0)
     group by Urun.UrunId
     into :UrunId,
          :DevirMiktari, :AlisMiktari, :AlisIadesiMiktari, :SatisMiktari, :SatisIadesiMiktari, :UretimeCikisMiktari,
          :DepolaraCikanMiktar, :DepolardanGelenMiktar, :UretimMiktari,
          :FasonUretimeCikisMiktari, :FasonUretimMiktari, :FasonUretimIadeMiktari;

    StokMiktari = DevirMiktari + AlisMiktari + UretimMiktari + SatisIadesiMiktari + DepolardanGelenMiktar + FasonUretimMiktari + FasonUretimIadeMiktari -
                   (SatisMiktari + AlisIadesiMiktari + UretimeCikisMiktari + DepolaraCikanMiktar + FasonUretimeCikisMiktari);
	StokMiktari = StokMiktari * 1000 ;

    if (Miktar > 0) then OtomasyonMiktar = Miktar;
    else OtomasyonMiktar = Banyo;

    --if (ReceteMiktar > 0) then OtomasyonReceteMiktar = ReceteMiktar;
    --else OtomasyonReceteMiktar = ReceteBanyo;

    if (OtomasyonMiktar > StokMiktari) then StokYetersiz = 1;

    suspend;
  end
end
