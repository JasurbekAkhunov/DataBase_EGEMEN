CREATE PROCEDURE ENVANTERRAPORUIPLIKHAZIRLA(
    AIPLIKDEPOID VARCHAR(10),
    ABASLANGICTARIH TIMESTAMP,
    ABITISTARIH TIMESTAMP,
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    AKALINLIKID INTEGER,
    AELYAFSINIFID INTEGER,
    ACIRCIRID INTEGER,
    AMENSEIID INTEGER,
    AIPLIKCINSIID INTEGER,
    ALOT VARCHAR(30),
    AGIRISLERFIRMADETAYLI INTEGER,
    ARENKID INTEGER)
RETURNS (
    IPLIKDEPOID VARCHAR(10),
    HAMID INTEGER,
    LOT VARCHAR(30),
    IPLIKLOTGRUBUID INTEGER,
    IPLIKLOTGRUBUADI ADI20,
    ELYAFDEGERLENDIRME INTEGER,
    RENKID INTEGER,
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    SARIMSEKLIID INTEGER,
    KONIKCINSIID INTEGER,
    KONIKRENGIID INTEGER,
    DEVREDENKAPADEDI INTEGER,
    DEVREDENMIKTAR DOUBLE PRECISION,
    DEVREDENNETMIKTAR DOUBLE PRECISION,
    GIRISFIRMAID VARCHAR(20),
    GIRISFIRMAADI VARCHAR(100),
    GIRISKAPADEDI INTEGER,
    GIRISMIKTAR DOUBLE PRECISION,
    GIRISNETMIKTAR DOUBLE PRECISION,
    CIKISKAPADEDI INTEGER,
    CIKISMIKTAR DOUBLE PRECISION,
    CIKISNETMIKTAR DOUBLE PRECISION,
    KALANKAPADEDI INTEGER,
    KALANMIKTAR DOUBLE PRECISION,
    KALANNETMIKTAR DOUBLE PRECISION,
    HARMANELYAFBIRIMMALIYET DOUBLE PRECISION,
    HARMANISCILIKBIRIMMALIYETI DOUBLE PRECISION,
    HARMANDIGERBIRIMMALIYET DOUBLE PRECISION,
    AMBALAJID INTEGER,
    AMBALAJADI ADI20,
    AMBALAJSEKLIID INTEGER,
    AMBALAJSEKLIADI ADI10,
    HARMANIPLIKISLETMEADI ADI50,
    HARMANIPLIKISLETMEID INTEGER,
    IPLIKTALEPNOSTR VARCHAR(30),
    BOYASIPARISNOSTR VARCHAR(30),
    BOYASIPARISFIRMAKODU FIRMAKODUSIZE,
    BOYASIPARISFIRMAADI ADI100)
AS
declare variable FGirisCikis        integer;
declare variable FMiktar            Double Precision;
declare variable FKapAdedi          integer;
declare variable FIplikTalepYil     integer;
declare variable FIplikTalepId      integer;
declare variable FBoyaSiparisYil    integer;
declare variable FBoyaSiparisId     integer;
begin

  if (AHamTipiId = 0) then Select hstIplik from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 1) then Select hstElyaf from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 2) then Select hstTelef from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 3) then Select hstHamPamuk from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 4) then Select hstCigit from HamTipiGetir into :AHamTipiId;
  else if (AHamTipiId = 5) then Select hstKirpinti from HamTipiGetir into :AHamTipiId;
  else AHamTipiId = null;

  DevredenKapAdedi      = 0;
  DevredenMiktar        = 0;
  GirisKapAdedi         = 0;
  GirisMiktar           = 0;
  CikisKapAdedi         = 0;
  CikisMiktar           = 0;
  KalanKapAdedi         = 0;
  KalanMiktar           = 0;

  KalanNetMiktar        = 0;
  CikisNetMiktar        = 0;
  GirisNetMiktar        = 0;
  DevredenNetMiktar     = 0;
  GirisFirmaID  = Null;
  GirisFirmaAdi = Null;

  if (AGirislerFirmaDetayli = 1) then
  begin
    For Select IplikIrsaliye.FirmaID, Firma.Adi, IplikStok.IplikDepoId, IplikStok.HamId, IplikIrsaliyeDetay.RenkId, IplikIrsaliyeDetay.Lot, IplikLot.ElyafDegerlendirme, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu,
               IplikIrsaliyeDetay.SarimSekliId,
               Count(iif(IplikStok.GirisTarihi <= :ABitisTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABitisTarih), 1, null)),
               sum(iif(IplikStok.GirisTarihi <= :ABitisTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABitisTarih), IplikStok.Miktar, 0)),
               sum(iif(IplikStok.GirisTarihi <= :ABitisTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABitisTarih), IplikStok.NetMiktar, 0)),
               Count(iif(IplikIrsaliye.IrsaliyeTarihi < :ABaslangicTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABaslangicTarih), 1, null)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi < :ABaslangicTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABaslangicTarih), IplikStok.Miktar, 0)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi < :ABaslangicTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABaslangicTarih), IplikStok.NetMiktar, 0)),
               Count(iif(IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih, 1, null)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih, IplikStok.Miktar, 0)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih, IplikStok.NetMiktar, 0)),
               Count(iif(IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and CikisII.IrsaliyeTarihi <= :ABitisTarih, 1, null)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and CikisII.IrsaliyeTarihi <= :ABitisTarih, IplikStok.Miktar, 0)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and CikisII.IrsaliyeTarihi <= :ABitisTarih, IplikStok.NetMiktar, 0)),
               Harman.ElyafBirimMaliyet, Harman.IscilikBirimMaliyeti, Harman.DigerBirimMaliyet, IplikIrsaliyeDetay.KonikCinsiId, IplikIrsaliyeDetay.KonikRengiId,
               IplikLotGrubu.IplikLotGrubuId, IplikLotGrubu.Adi, Ambalaj.AmbalajId, Ambalaj.Adi, AmbalajSekli.AmbalajSekliId, AmbalajSekli.Adi, Harman.IplikIsletmeId, IplikIsletme.Adi,
               IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, SiparisFirma.FirmaId, SiparisFirma.Adi
               from IplikStok
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.IplikIrsaliyeDetayId
      left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikStok.IplikIrsaliyeID
      left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
      left join IplikLot on IplikLot.Lot = IplikIrsaliyeDetay.Lot and IplikLot.HamID = IplikIrsaliyeDetay.HamID
      left join Firma On Firma.FirmaId = IplikIrsaliye.FirmaId
      left join Harman on Harman.HarmanKayitId = IplikLot.HarmanKayitId
      left join IplikIsletme on IplikIsletme.IplikIsletmeId = Harman.IplikIsletmeId
      left join IplikLotGrubu on IplikLotGrubu.IplikLotGrubuId = IplikLot.IplikLotGrubuId
      left join Ambalaj on Ambalaj.AmbalajId = IplikIrsaliyeDetay.AmbalajId
      left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = Ambalaj.AmbalajSekliId
      left join IplikIrsaliyeDetay CikisIID on CikisIID.IplikIrsaliyeDetayId = IplikStok.CikisIplikIrsaliyeDetayId
      left join IplikIrsaliye CikisII on CikisII.IplikIrsaliyeId = CikisIID.IplikIrsaliyeId
      left join IplikTalep On IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
      left join IplikTalepDetay On IplikTalepDetay.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepDetayId
      left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
      left join Firma SiparisFirma On Firma.FirmaId = BoyaSiparis.FirmaId
      where ((IplikStok.GirisTarihi <= :ABitisTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABitisTarih)) or
            (IplikIrsaliye.IrsaliyeTarihi < :ABaslangicTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABaslangicTarih)) or
            (IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih) or
            (IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and CikisII.IrsaliyeTarihi <= :ABitisTarih)) and
            (IplikStok.IplikDepoId = :AIplikDepoId or :AIplikDepoId is null) and
            (IplikStok.HamId = :AHamId or :AHamId is null) and
            (IplikIrsaliyeDetay.Lot = :ALot or :ALot is null) and
            (Ham.HamTipiID = :AHamTipiID or :AHamTipiID is null) and
            (Ham.KalinlikId = :AKalinlikId or :AKalinlikId is null) and
            (Ham.ElyafSinifId = :AElyafSinifId or :AElyafSinifId is null) and
            (Ham.CircirId = :ACircirId or :ACircirId is null) and
            (Ham.MenseiId = :AMenseiId or :AMenseiId is null) and
            (Ham.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is null) and
            (Harman.RenkId = :ARenkId or :ARenkId is null)
      group by IplikIrsaliye.FirmaID, Firma.Adi, IplikStok.IplikDepoId, IplikStok.HamId, IplikIrsaliyeDetay.RenkId, IplikIrsaliyeDetay.Lot, IplikLot.ElyafDegerlendirme,
               IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu, IplikIrsaliyeDetay.SarimSekliId, Harman.ElyafBirimMaliyet, Harman.IscilikBirimMaliyeti, Harman.DigerBirimMaliyet,
               IplikIrsaliyeDetay.KonikCinsiId, IplikIrsaliyeDetay.KonikRengiId, IplikLotGrubu.IplikLotGrubuId, IplikLotGrubu.Adi,
               Ambalaj.AmbalajId, Ambalaj.Adi, AmbalajSekli.AmbalajSekliId, AmbalajSekli.Adi, Harman.IplikIsletmeId, IplikIsletme.Adi,
               IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, SiparisFirma.FirmaId, SiparisFirma.Adi
      into :GirisFirmaID, :GirisFirmaAdi, :IplikDepoId, :HamId, :RenkId, :Lot, :ElyafDegerlendirme, :MelanjKodu, :FlamKodu,
           :SarimSekliId, :KalanKapAdedi, :KalanMiktar, :KalanNetMiktar,
           :DevredenKapAdedi, :DevredenMiktar, :DevredenNetMiktar,
           :GirisKapAdedi, :GirisMiktar, :GirisNetMiktar,
           :CikisKapAdedi, :CikisMiktar, :CikisNetMiktar,
           :HarmanElyafBirimMaliyet, :HarmanIscilikBirimMaliyeti, :HarmanDigerBirimMaliyet, :KonikCinsiId, :KonikRengiId,
           :IplikLotGrubuId, :IplikLotGrubuAdi, :AmbalajId, :AmbalajAdi, :AmbalajSekliId, :AmbalajSekliAdi, :HarmanIplikIsletmeId, :HarmanIplikIsletmeAdi,
           :FIplikTalepYil, :FIplikTalepId, :FBoyaSiparisYil, :FBoyaSiparisId, :BoyaSiparisFirmaKodu, :BoyaSiparisFirmaAdi
      do
      begin
        if (FIplikTalepYil is not null) then IplikTalepNoStr = '(' || EgemenIntegerToStr(FIplikTalepYil) || ') ' || EgemenIntegerToStr(FIplikTalepId) || ' ';
          else IplikTalepNoStr = '';

        if (FBoyaSiparisYil is not null) then BoyaSiparisNoStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') ' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
          else BoyaSiparisNoStr = '';

        suspend;
      end
  end
  else
  begin
    For Select IplikStok.IplikDepoId, IplikStok.HamId, IplikIrsaliyeDetay.RenkId, IplikIrsaliyeDetay.Lot, IplikLot.ElyafDegerlendirme, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu,
               IplikIrsaliyeDetay.SarimSekliId,
               Count(iif((IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABitisTarih)), 1, null)),
               sum(iif((IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABitisTarih)), IplikStok.Miktar, 0)),
               sum(iif((IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABitisTarih)), IplikStok.NetMiktar, 0)),
               Count(iif(IplikIrsaliye.IrsaliyeTarihi < :ABaslangicTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABaslangicTarih), 1, null)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi < :ABaslangicTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABaslangicTarih), IplikStok.Miktar, 0)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi < :ABaslangicTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABaslangicTarih), IplikStok.NetMiktar, 0)),
               Count(iif(IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih, 1, null)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih, IplikStok.Miktar, 0)),
               sum(iif(IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih, IplikStok.NetMiktar, 0)),
               Count(iif(CikisII.IrsaliyeTarihi >= :ABaslangicTarih and CikisII.IrsaliyeTarihi <= :ABitisTarih, 1, null)),
               sum(iif(CikisII.IrsaliyeTarihi >= :ABaslangicTarih and CikisII.IrsaliyeTarihi <= :ABitisTarih, IplikStok.Miktar, 0)),
               sum(iif(CikisII.IrsaliyeTarihi >= :ABaslangicTarih and CikisII.IrsaliyeTarihi <= :ABitisTarih, IplikStok.NetMiktar, 0)),
               Harman.ElyafBirimMaliyet, Harman.IscilikBirimMaliyeti, Harman.DigerBirimMaliyet, IplikIrsaliyeDetay.KonikCinsiId, IplikIrsaliyeDetay.KonikRengiId,
               IplikLotGrubu.IplikLotGrubuId, IplikLotGrubu.Adi, Ambalaj.AmbalajId, Ambalaj.Adi, AmbalajSekli.AmbalajSekliId, AmbalajSekli.Adi, Harman.IplikIsletmeId, IplikIsletme.Adi,
               IplikTalep.IplikTalepYil, IplikTalep.IplikTalepId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Firma.FirmaId, Firma.Adi from IplikStok
      left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.IplikIrsaliyeDetayId
      left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeID = IplikStok.IplikIrsaliyeID
      left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
      left join IplikLot on IplikLot.Lot = IplikIrsaliyeDetay.Lot and IplikLot.HamID = IplikIrsaliyeDetay.HamID
      left join Harman on Harman.HarmanKayitId = IplikLot.HarmanKayitId
      left join IplikIsletme on IplikIsletme.IplikIsletmeId = Harman.IplikIsletmeId
      left join IplikLotGrubu on IplikLotGrubu.IplikLotGrubuId = IplikLot.IplikLotGrubuId
      left join Ambalaj on Ambalaj.AmbalajId = IplikIrsaliyeDetay.AmbalajId
      left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = Ambalaj.AmbalajSekliId
      left join IplikIrsaliyeDetay CikisIID on CikisIID.IplikIrsaliyeDetayId = IplikStok.CikisIplikIrsaliyeDetayId
      left join IplikIrsaliye CikisII on CikisII.IplikIrsaliyeId = CikisIID.IplikIrsaliyeId
      left join IplikTalep On IplikTalep.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepKayitId
      left join IplikTalepDetay On IplikTalepDetay.IplikTalepKayitId = IplikIrsaliyeDetay.IplikTalepDetayId
      left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
      left join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
      where ((IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABitisTarih)) or
             (IplikIrsaliye.IrsaliyeTarihi < :ABaslangicTarih and (IplikStok.CikisTarihi is null or CikisII.IrsaliyeTarihi > :ABaslangicTarih)) or
             (IplikIrsaliye.IrsaliyeTarihi >= :ABaslangicTarih and IplikIrsaliye.IrsaliyeTarihi <= :ABitisTarih) or
             (CikisII.IrsaliyeTarihi >= :ABaslangicTarih and CikisII.IrsaliyeTarihi <= :ABitisTarih)) and
            (IplikStok.IplikDepoId = :AIplikDepoId or :AIplikDepoId is null) and
            (IplikStok.HamId = :AHamId or :AHamId is null) and
            (IplikIrsaliyeDetay.Lot = :ALot or :ALot is null) and
            (Ham.HamTipiID = :AHamTipiID or :AHamTipiID is null) and
            (Ham.KalinlikId = :AKalinlikId or :AKalinlikId is null) and
            (Ham.ElyafSinifId = :AElyafSinifId or :AElyafSinifId is null) and
            (Ham.CircirId = :ACircirId or :ACircirId is null) and
            (Ham.MenseiId = :AMenseiId or :AMenseiId is null) and
            (Ham.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is null) and
            (Harman.RenkId = :ARenkId or :ARenkId is null)
      group by IplikStok.IplikDepoId, IplikStok.HamId, IplikIrsaliyeDetay.RenkId, IplikIrsaliyeDetay.Lot, IplikLot.ElyafDegerlendirme,
               IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu, IplikIrsaliyeDetay.SarimSekliId, Harman.ElyafBirimMaliyet, Harman.IscilikBirimMaliyeti, Harman.DigerBirimMaliyet,
