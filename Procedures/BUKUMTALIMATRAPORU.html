CREATE PROCEDURE BUKUMTALIMATRAPORU(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASIPARISFIRMAID VARCHAR(10),
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    AORMESIPARISSEKLIID INTEGER,
    APLANLAMAFIRMAID VARCHAR(10),
    APLANLAMAMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4),
    ABITTI INTEGER,
    ASIPKAPALIDURUMU SMALLINT,
    ADOKUMATALIMATI SMALLINT)
RETURNS (
    BUKUMTALIMATID INTEGER,
    TARIH TIMESTAMP,
    BOYASIPARISKAYITID INTEGER,
    URETIMPARTINO VARCHAR(15),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(15),
    SIPARISFIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    TALIMATMIKTARI DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    TAMIRECIKILANMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    BITTI INTEGER,
    BITTISTR VARCHAR(4),
    PUNTOTALIMATI EGEBOOLEAN,
    PUNTOTALIMATISTR VARCHAR(15),
    PUNTOMAKINADEVIR DOUBLE PRECISION,
    HAVABASINCI DOUBLE PRECISION,
    METREDEKIPUNTO DOUBLE PRECISION,
    BUKUMTURU VARCHAR(15),
    SIPKAPALI SMALLINT,
    SIPKAPALISTR VARCHAR(7),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin

  for Select BukumTalimat.BukumTalimatId, BukumTalimat.Tarih, BukumTalimat.UretimPartiNo, BukumTalimat.FirmaId, OPF.Adi,
         iif(BukumTalimat.DokumaTalimati = 1, BukumTalimat.DokumaMakinaId, BukumTalimat.OrmeMakinaId),
         iif(BukumTalimat.DokumaTalimati = 1, DokumaMakina.Adi, OrmeMakina.Adi),  BukumTalimat.BoyaSiparisKayitId,
         BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Ham.HamID, Ham.Adi, BukumTalimat.TalimatMiktari,
         BukumTalimat.GerceklesenMiktar, BoyaSiparis.FirmaId, ASF.Adi,
         BukumTalimat.BirimFiyat, BukumTalimat.DovizId, BukumTalimat.vadeGun, BukumTalimat.Notu,
         BukumTalimat.Bitti, BukumTalimat.InsertTarihi, BukumTalimat.DegisimTarihi, Ik.Adi, Sk.Adi,
         BukumTalimat.BukumTuru, OrmeSiparisDetay.Kapali, BukumTalimat.PuntoTalimati, BukumTalimat.PuntoMakinaDevir, BukumTalimat.HavaBasinci, BukumTalimat.MetredekiPunto from BukumTalimat
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BukumTalimat.BoyaSiparisKayitId
    left join Ham on Ham.HamId = BukumTalimat.HamId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = BukumTalimat.OrmeMakinaID
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = BukumTalimat.DokumaMakinaID
    left join Firma OPF on OPF.FirmaId = BukumTalimat.FirmaId
    left join Firma ASF on ASF.FirmaId = BoyaSiparis.FirmaId
    Left Join Kullanici IK on IK.KullaniciId = BukumTalimat.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = BukumTalimat.KullaniciId
    Left Join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BukumTalimat.OrmeSiparisDetayId
    where BukumTalimat.DokumaTalimati = :ADokumaTalimati and
          (BukumTalimat.Tarih >= :ABaslangicTarihi and BukumTalimat.Tarih <= :ABitisTarihi) and
          (BukumTalimat.FirmaId = :APlanlamaFirmaId or :APlanlamaFirmaId is null) and
          (BukumTalimat.OrmeMakinaID = :APlanlamaMakinaId or BukumTalimat.DokumaMakinaID = :APlanlamaMakinaId or :APlanlamaMakinaId is null) and
          (BoyaSiparis.FirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
          (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (BoyaSiparis.OrmeSiparisSekliId = :AOrmeSiparisSekliId or :AOrmeSiparisSekliId is null) and
          (OrmeMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
          (BukumTalimat.Bitti = :ABitti or :ABitti is null) and
          (:ARaporTipi is null or
           ((BukumTalimat.OrmeMakinaId is not null or BukumTalimat.DokumaMakinaId is not null)and :ARaporTipi = 0) or
           (BukumTalimat.FirmaId is not null and :ARaporTipi = 1) ) and
          (BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
          (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil  and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId)) and
          (OrmeSiparisDetay.Kapali = :ASipKapaliDurumu or (:ASipKapaliDurumu is Null or BukumTalimat.OrmeSiparisDetayId is Null))
          order by BukumTalimat.OrmeMakinaId, BukumTalimat.FirmaId, BukumTalimat.TerminTarihi
    into :BukumTalimatId, :Tarih, :UretimPartiNo, :FirmaId, :FirmaAdi, :MakinaId, :MakinaAdi, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId,
         :HamId, :HamAdi, :TalimatMiktari, :GerceklesenMiktar, :SiparisFirmaId, :SiparisFirmaAdi, :BirimFiyat, :DovizId, :vadeGun, :Notu, :Bitti,
         :InsertTarihi, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi, :BukumTuru, :SipKapali, :PuntoTalimati, :PuntoMakinaDevir, :HavaBasinci, :MetredekiPunto do
  begin
    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    PuntoTalimatiStr = '';
    if (PuntoTalimati = 1) then PuntoTalimatiStr = 'Punto Talimatı';

    if (Bitti = 0) then BittiStr = '';
    else BittiStr = 'Evet';

    if (SipKapali = 0) then SipKapaliStr = 'Açık';
    else if (SipKapali = 1) then SipKapaliStr = 'Kapalı';
    else SipKapaliStr = '';

    KalanMiktar = TalimatMiktari - GerceklesenMiktar;

    Select Sum(Miktar) from StokHareket where StokHareket.BukumTalimatID = :BukumTalimatID and StokHareket.Tamir = 1 into :TamireCikilanMiktar;
    if (TamireCikilanMiktar is Null) then TamireCikilanMiktar = 0;

    suspend;
  end
end
