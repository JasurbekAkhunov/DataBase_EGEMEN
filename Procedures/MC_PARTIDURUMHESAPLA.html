CREATE PROCEDURE MC_PARTIDURUMHESAPLA(
    APARTISINIFID INTEGER,
    ADURUMU INTEGER)
RETURNS (
    DURUMUSTR VARCHAR(30))
AS
declare variable KaliteKontroleGelmis integer;
begin
  Select Count(*) from PartiAsamaStok
    left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
    where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamaStok.GirisTarihi is not null and Asama.KaliteKontrolAsamasimi = 1
    into :KaliteKontroleGelmis;

  if (ADurumu = 7) then DurumuStr = 'Sevke Hazır';
  else if (ADurumu = 8) then DurumuStr = 'Sevk Edildi';
  else if (KaliteKontroleGelmis = 1) then DurumuStr = 'Kalite Kontrol';
  else if (ADurumu < 4) then DurumuStr = 'Boyamaya Hazırlık';
  else if (ADurumu = 4) then DurumuStr = 'Boyamada';
  else if (ADurumu = 5 or ADurumu = 6) then DurumuStr = 'Boya Sonrası İşlemler';
  else DurumuStr = '';
  suspend;
end
