CREATE PROCEDURE DISTAMIRFIRMAANALIZI(
    ABASLANGICBEYANTARIHI TIMESTAMP,
    ABITISBEYANTARIHI TIMESTAMP,
    ARAPORTIPI INTEGER,
    AFIRMAID VARCHAR(20),
    AHAMTIPI INTEGER,
    AHAMID INTEGER,
    APARTIGRUPID VARCHAR(4),
    ATAMIRDUZELTMETIPIID VARCHAR(10),
    ATAMIRTIPIID INTEGER,
    ATAMIRTIPIGRUPID VARCHAR(4),
    ADISTAMIRCOZUMSEKLIID VARCHAR(10),
    AISLEMEALINACAK INTEGER,
    AHAMSTOKMASTERID INTEGER,
    ASATISTIPI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TOPLAMKILO NUMERIC(12,3),
    TOPLAMMETRE NUMERIC(12,3),
    ORAN DOUBLE PRECISION)
AS
declare variable FKilo   Numeric(12,3);
begin
  Select Sum(kilo) from DisTamirListesi(:ABaslangicBeyanTarihi, :ABitisBeyanTarihi, :ARaporTipi, :AFirmaId, :AHamTipi, :AHamId, :APartiGrupId, :ATamirDuzeltmeTipiId, :ATamirTipiId, :ATamirTipiGrupId, :ADisTamirCozumSekliID, :AIslemeAlinacak, :AHamStokMasterId, :ASatisTipi) into :FKilo;

  for Select OzetFirmaId, OzetFirmaAdi, Sum(Kilo), Sum(Metre) from DisTamirListesi(:ABaslangicBeyanTarihi, :ABitisBeyanTarihi, :ARaporTipi, :AFirmaId, :AHamTipi, :AHamId, :APartiGrupId, :ATamirDuzeltmeTipiId, :ATamirTipiId, :ATamirTipiGrupId, :ADisTamirCozumSekliID, :AIslemeAlinacak, :AHamStokMasterId, :ASatisTipi)
  group by OzetFirmaId, OzetFirmaAdi
  into :FirmaId, :FirmaAdi, :ToplamKilo, :ToplamMetre do
  begin
    if (FKilo > 0) then Oran = ToplamKilo * 100 / FKilo;
    else Oran = Null;
    suspend;
  end
end
