CREATE PROCEDURE FIYATLANDIRILMISPARTILER(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    MODELNO VARCHAR(30),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    KILO DOUBLE PRECISION,
    FASONSTR VARCHAR(10),
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    FIYATONAYTARIHI TIMESTAMP,
    FIYATONAYLAYANKULLANICIADI VARCHAR(40))
AS
declare variable FFason integer;
begin
  for select Parti.FirmaId, Firma.Adi, parti.partikayitid, partiStok.PartiSinifId, parti.partiYil, Parti.PartiId, Parti.ModelNo,
             Parti.ReceteId, Recete.Adi, Parti.fason, PartiStok.Kilo, IslemTipi.Adi, IslemGrubu.Adi, Kullanici.Adi,
             PartiStok.FiyatOnayTarihi from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitID
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join Kullanici on Kullanici.KullaniciId = PartiStok.FiyatOnaylayanKullaniciId
    where PartiStok.FiyatOnayTarihi >= :ABaslangicTarihi and PartiStok.FiyatOnayTarihi <= :ABitisTarihi and PartiStok.PartiStokIptalKayitId is null and
          PartiStok.PartiPakettenFiyatlandirildi = 1 and (Parti.Firmaid = :AFirmaId or :AFirmaId is null)
    order by Parti.FirmaId, Parti.PartiKayitId, PartiStok.PartiSinifId
    into :FirmaId, :FirmaAdi, :Partikayitid, :PartiSinifId, :PartiYil, :PartiId, :ModelNo, :ReceteId, :ReceteAdi, :FFason, :Kilo,
         :IslemTipiAdi, :IslemGrubuAdi, :FiyatOnaylayanKullaniciAdi, :FiyatOnayTarihi do
  begin
    if (FFason = 1) then FasonStr = 'Evet';  else FasonStr = null;

    PartiIdStr = '(';
    PartiIdStr = PartiIdStr || Format_Float('#0', PartiYil);
    PartiIdStr = PartiIdStr || ') ';
    PartiIdStr = PartiIdStr || Format_Float('#0', PartiId);
    PartiIdStr = PartiIdStr || ' ';
    suspend;
  end
end
