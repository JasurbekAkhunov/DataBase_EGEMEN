CREATE PROCEDURE ORMEMPBAZINDACALISMARAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AORMEMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4))
RETURNS (
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    TOPLAMSURE DOUBLE PRECISION,
    PLANLAMALICALISMASURESI DOUBLE PRECISION,
    PLANLAMASIZCALISMASURESI DOUBLE PRECISION,
    TOPLAMCALISMASURESI DOUBLE PRECISION,
    DURUSSURESI DOUBLE PRECISION,
    ENYUKSEKTUR INTEGER,
    ARIZASURESI DOUBLE PRECISION,
    SERVISSURESI DOUBLE PRECISION,
    AYARSURESI DOUBLE PRECISION,
    MEMINGERKOPTUSURESI DOUBLE PRECISION,
    ALTMEMINGERKOPTUSURESI DOUBLE PRECISION,
    LYCRAKOPTUSURESI DOUBLE PRECISION,
    IGNEKOPTUSURESI DOUBLE PRECISION,
    KAPAKACILDISURESI DOUBLE PRECISION,
    STARTSTOPSURESI DOUBLE PRECISION,
    TOPLAMTOPKESME INTEGER,
    TOPLAMTEMIZLIKYAPILDI INTEGER,
    TOPLAMTUR INTEGER,
    PLANLANANTURSAYISI DOUBLE PRECISION,
    VARDIYANO INTEGER,
    TARIH TIMESTAMP,
    TOPLAMISTENENMIKTAR DOUBLE PRECISION,
    TOPLAMGERCEKLESENMIKTAR DOUBLE PRECISION)
AS
declare variable Tur integer;
declare variable OrmePlanlamaId integer;

declare variable LastTarih     TimeStamp;
declare variable LastTur integer;
declare variable FarkTarih              Double Precision;
declare variable FLastPersonelId        integer;
declare variable FPersonelId            integer;
declare variable FPersonelAdi           varchar(40);
declare variable LastAriza integer;
declare variable LastServis integer;
declare variable LastAyar integer;
declare variable Ariza integer;
declare variable Servisci integer;
declare variable Ayar integer;
declare variable FToplamIstenenMiktar       Double Precision;
declare variable FToplamGerceklesenMiktar   Double Precision;

declare variable LastMemingerKoptu      SmallInt;
declare variable LastAltMemingerKoptu   SmallInt;
declare variable LastLycraKoptu         SmallInt;
declare variable LastIgneKoptu          SmallInt;
declare variable LastKapakAcildi        SmallInt;
declare variable LastStartStop          SmallInt;
declare variable MemingerKoptu      SmallInt;
declare variable AltMemingerKoptu   SmallInt;
declare variable LycraKoptu         SmallInt;
declare variable IgneKoptu          SmallInt;
declare variable KapakAcildi        SmallInt;
declare variable StartStop          SmallInt;
declare variable TopKesme           SmallInt;
declare variable TemizlikYapildi    SmallInt;
declare variable LastVardiyaNo      Integer;
declare variable PlanlananTurToplamSure   Double Precision;
begin

  For Select OrmeMakinaId, Adi from OrmeMakina
    where (OrmeMakina.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
          (OrmeMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and OrmeMakina.OtomasyonaDahil = 1
    into :OrmeMakinaId, :OrmeMakinaAdi do
  begin
    PlanlamaliCalismaSuresi       = 0;
    PlanlamasizCalismaSuresi      = 0;
    ToplamCalismaSuresi           = 0;
    DurusSuresi                   = 0;
    EnYuksekTur                   = 0;
    FLastPersonelId               = Null;
    PersonelId                    = Null;
    PersonelAdi                   = Null;
    ArizaSuresi                   = 0;
    ServisSuresi                  = 0;
    AyarSuresi                    = 0;
    ToplamIstenenMiktar           = 0;
    ToplamGerceklesenMiktar       = 0;
    MemingerKoptuSuresi     = 0;
    AltMemingerKoptuSuresi  = 0;
    LycraKoptuSuresi     = 0;
    IgneKoptuSuresi      = 0;
    KapakAcildiSuresi    = 0;
    StartStopSuresi      = 0;
    ToplamTopKesme       = 0;
    ToplamTur              = 0;
    ToplamTemizlikYapildi  = 0;
    PlanlananTurSayisi     = 0;
    VardiyaNo              = 0;
    PlanlananTurToplamSure = 0;

    LastTur = -1;
    LastTarih = null;
    ToplamSure = 0;
    LastAriza = -1;
    LastServis = -1;
    LastAyar = -1;
    LastMemingerKoptu = -1;
    LastAltMemingerKoptu = -1;
    LastLycraKoptu    = -1;
    LastIgneKoptu     = -1;
    LastKapakAcildi   = -1;
    LastStartStop     = -1;

    For select Tarih, Tur, OrmePlanlamaId, PersonelId, PersonelAdi, Ariza, Servisci, Ayar, ToplamIstenenMiktar, ToplamGerceklesenMiktar, MemingerKoptu, IgneKoptu, KapakAcildi, StartStop, TopKesme, TemizlikYapildi, PlanlananTurSayisi from OrmeMakinaCalismaHareketiRaporu(:ABaslangicTarihi, :ABitisTarihi, :OrmeMakinaId, :AMakinaGrupId, 0)
       order by OrmeMakinaCalismaHareketiRaporu.Tarih
       into :Tarih, :Tur, :OrmePlanlamaId, :FPersonelID, :FPersonelAdi, :Ariza, :Servisci, :Ayar, :FToplamIstenenMiktar, :FToplamGerceklesenMiktar, :MemingerKoptu, :IgneKoptu, :KapakAcildi, :StartStop, :TopKesme, :TemizlikYapildi, :PlanlananTurSayisi do
    begin

      if (FPersonelID is null) then FPersonelID = -1;

      if (FLastPersonelId is null) then
      begin
        Select VardiyaNo from OrmeVardiyaHesapla(:Tarih) into :VardiyaNo;
        FLastPersonelId = FPersonelId;
        PersonelId      = FPersonelId;
        PersonelAdi     = FPersonelAdi;
      end

      if (LastTarih is null) then LastTarih = Tarih;

      if (Tur > EnYuksekTur) then EnYuksekTur = Tur;
      ToplamTur = ToplamTur + Tur;

      if (Tur > 0 and MemingerKoptu = 0 and KapakAcildi = 0 and IgneKoptu = 0 and StartStop = 0) then Tur = 1;
      else Tur = 0;

      if (LastTur = -1) then LastTur = Tur;
      if (LastAriza = -1) then LastAriza = Ariza;
      if (LastServis = -1) then LastServis = Servisci;
      if (LastAyar = -1) then LastAyar = Ayar;
      if (LastMemingerKoptu = -1) then LastMemingerKoptu = MemingerKoptu;
      if (LastIgneKoptu = -1) then LastIgneKoptu = IgneKoptu;
      if (LastKapakAcildi = -1) then LastKapakAcildi = KapakAcildi;
      if (LastStartStop = -1) then LastStartStop = StartStop;

      FarkTarih = Tarih - LastTarih;
      ToplamSure = ToplamSure + FarkTarih;
      PlanlananTurToplamSure = PlanlananTurToplamSure + (FarkTarih * PlanlananTurSayisi);
      if (ToplamSure > 0) then  PlanlananTurSayisi =  Round(PlanlananTurToplamSure / ToplamSure, 2);


      if (LastTur = 0) then DurusSuresi = DurusSuresi + FarkTarih;
      else if (LastTur = 1) then
      begin
        if (OrmePlanlamaId is null) then PlanlamasizCalismaSuresi = PlanlamasizCalismaSuresi + FarkTarih;
        else PlanlamaliCalismaSuresi = PlanlamaliCalismaSuresi + FarkTarih;

        ToplamCalismaSuresi = ToplamCalismaSuresi + FarkTarih;
      end

      if (FToplamIstenenMiktar is Null) then FToplamIstenenMiktar = 0;
      if (FToplamGerceklesenMiktar is Null) then FToplamGerceklesenMiktar = 0;

      ToplamIstenenMiktar        = ToplamIstenenMiktar + FToplamIstenenMiktar;
      ToplamGerceklesenMiktar    = ToplamGerceklesenMiktar + FToplamGerceklesenMiktar;

      if (TopKesme is Null) then TopKesme = 0;
      ToplamTopKesme        = ToplamTopKesme + TopKesme;

      if (TemizlikYapildi is Null) then TemizlikYapildi = 0;
      ToplamTemizlikYapildi        = ToplamTemizlikYapildi + TemizlikYapildi;

      if (LastAriza = 1) then ArizaSuresi = ArizaSuresi + FarkTarih;
      if (LastServis = 1) then ServisSuresi = ServisSuresi + FarkTarih;
      if (LastAyar = 1) then AyarSuresi = AyarSuresi + FarkTarih;
      if (LastMemingerKoptu = 1) then MemingerKoptuSuresi = MemingerKoptuSuresi + FarkTarih;
      if (LastIgneKoptu = 1) then IgneKoptuSuresi = IgneKoptuSuresi + FarkTarih;
      if (LastKapakAcildi = 1) then KapakAcildiSuresi = KapakAcildiSuresi + FarkTarih;
      if (LastStartStop = 1 and LastMemingerKoptu = 0 and LastIgneKoptu = 0 and LastKapakAcildi = 0) then StartStopSuresi = StartStopSuresi + FarkTarih;

      LastTarih = Tarih;
      LastTur = Tur;
      LastAriza = Ariza;
      LastServis = Servisci;
      LastAyar = Ayar;
      LastMemingerKoptu = MemingerKoptu;
      LastIgneKoptu = IgneKoptu;
      LastKapakAcildi = KapakAcildi;
      LastStartStop = StartStop;

      if (Intkarsilastir(FLastPersonelId, FPersonelId) = 0) then
      begin
        suspend;

        PersonelId       = FPersonelId;
        PersonelAdi      = FPersonelAdi;
        FLastPersonelId  = FPersonelId;
        PlanlamaliCalismaSuresi       = 0;
        PlanlamasizCalismaSuresi      = 0;
        ToplamCalismaSuresi           = 0;
        DurusSuresi      = 0;
        EnYuksekTur      = 0;
        ArizaSuresi      = 0;
        ServisSuresi      = 0;
        AyarSuresi      = 0;
        ToplamIstenenMiktar        = 0;
        ToplamGerceklesenMiktar    = 0;
        MemingerKoptuSuresi      = 0;
        IgneKoptuSuresi          = 0;
        KapakAcildiSuresi        = 0;
        StartStopSuresi          = 0;
        ToplamTopKesme           = 0;
        ToplamTemizlikYapildi    = 0;
        ToplamTur                = 0;
        PlanlananTurSayisi       = 0;
        PlanlananTurToplamSure   = 0;
        Select VardiyaNo from OrmeVardiyaHesapla(:Tarih) into :VardiyaNo;

        ToplamSure = 0;
      end

    end

    if (FLastPersonelId is not null) then suspend;
  end
end
