CREATE PROCEDURE UNITESEVKIRSALIYESIBARKODOKUT(
    AUNITEIRSALIYEID INTEGER,
    ABARKODID VARCHAR(20),
    ASISTEMDOVIZID VARCHAR(5),
    AKULLANICIID INTEGER)
AS
declare variable TF                    Integer;
declare variable IK                    Varchar(10);
declare variable KS                    Integer;
declare variable PPartiKayitId         Integer;
declare variable PPartiSinifId         Integer;
declare variable PKilo                 BoyaMiktarSize;
declare variable PNetKilo              BoyaMiktarSize;
declare variable PMetre                BoyaMiktarSize;
declare variable PIrsaliyeKayitSayisi  Integer;
declare variable PUniteIrsaliyeDetayId Integer;
declare variable PaketId               Integer;
declare variable FBarkodId             Varchar(30);
begin
  if (Char_Length(ABarkodId) < 3) then exception BarkodHatali1;

  ABarkodId = Replace(ABarkodId, '*', '-');

  FBarkodID = ABarkodId;
  IK = EgemenCopyStr(ABarkodId, 1, 2);

  ABarkodId = EgemenCopyStr(ABarkodId, 3, Char_Length(ABarkodId) - 2);
  TF = EgemenStrToIntOlurmu(ABarkodId);

  if (TF = 0) then exception BarkodHatali1;

  select count(*), Max(PartiStokTartiId) from PartiStokTarti where BarkodId = :FBarkodId into :KS, :PaketId;
  if (KS > 0) then /*iade kaydidir*/
    IK = 'R-';
  else
    PaketId = EgemenStrToInt(ABarkodId);

/* NetKilo PartiStok'tan diğer alanlar PSKaliteKontrol'dan alındı*/
  Select
       PSKaliteKontrol.PartiKayitId, PSKaliteKontrol.PartiSinifId, PSKaliteKontrol.kilo,PSKaliteKontrol.Dara,PSKaliteKontrol.metre
  from PSKaliteKontrol
    where PSKaliteKontrol.PSKaliteKontrolId = :PaketId
    into :pPartiKayitId,:pPartiSinifId,:pKilo,:pNetKilo,:pMetre;

  select count(*) from UniteIrsaliyeDetay
    where UniteIrsaliyeDetay.UniteIrsaliyeId= :aUniteIrsaliyeId and
          UniteIrsaliyeDetay.PartiSinifId= :pPartiSinifId
  into :pIrsaliyeKayitSayisi;

  if (:pIrsaliyeKayitSayisi=0) then
  begin
    insert into UniteIrsaliyeDetay(UniteIrsaliyeId, PartiKayitId, PartiSinifId, Kilo, NetKilo, Metre, DovizId, KullaniciId, KapAdedi)
                            values(:aUniteIrsaliyeId,:pPartiKayitId,:pPartiSinifId,:pKilo,:pNetKilo,:pMetre,:aSistemDovizId,:aKullaniciId,1);
  end

  if (:pIrsaliyeKayitSayisi>0) then
  begin
    select UniteIrsaliyeDetay.UniteIrsaliyeDetayId from  UniteIrsaliyeDetay
    where
      UniteIrsaliyeDetay.UniteIrsaliyeId = :aUniteIrsaliyeId and
      UniteIrsaliyeDetay.PartiKayitId    = :pPartiKayitId and
      UniteIrsaliyeDetay.PartiSinifId    = :pPartiSinifId
    into :pUniteIrsaliyeDetayId;

    if (:pUniteIrsaliyeDetayId>0) then
    begin
      update
        UniteIrsaliyeDetay
      set
        UniteIrsaliyeDetay.Kilo     = UniteIrsaliyeDetay.Kilo + :pKilo,
        UniteIrsaliyeDetay.NetKilo  = UniteIrsaliyeDetay.NetKilo + :pNetKilo,
        UniteIrsaliyeDetay.Metre    = UniteIrsaliyeDetay.Metre + :pMetre,
        UniteIrsaliyeDetay.KapAdedi = Coalesce(UniteIrsaliyeDetay.KapAdedi,0)+1
      where
        UniteIrsaliyeDetay.UniteIrsaliyeDetayId = :pUniteIrsaliyeDetayId;
   end
  end

end
