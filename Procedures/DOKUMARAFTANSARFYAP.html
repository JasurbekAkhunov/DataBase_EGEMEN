CREATE PROCEDURE DOKUMARAFTANSARFYAP(
    AORMERAFID INTEGER,
    AMIKTAR BOYAMIKTARSIZE,
    AKULLANICIID INTEGER)
RETURNS (
    ORMEFISID INTEGER)
AS
declare variable FisNo              integer;
declare variable SeriNo             varchar(2);
declare variable shtSarf            integer;
declare variable KayitStokHareketId integer;
declare variable SistemDovizId      DovizSize;
begin
  select shtSarf from OrmeStokHareketTipiGetir into :shtSarf;
  select Sistem.SistemDovizId from Sistem where SistemID = 1 into :SistemDovizID;

  Select DokumaUretimeCikisSeriNo from Kullanici where Kullanici.KullaniciID = :AKullaniciID into :SeriNo;
  Select FisNo from FisNumaratorAL(:SeriNo) into :FisNO;

  OrmeFisId = gen_id(OrmeFis_Id_Gen, 1);
  KayitStokHareketId = gen_id(StokHareket_Id_Gen, 1);

  insert into OrmeFis(OrmeFisId, HareketTipi, UretimIslemTipiId, SeriNo, FisNo, FisTarihi, OtomatikOlusturuldu, InsertKullaniciId, KullaniciId, DokumaHareketi, DovizId)
           values(:OrmeFisId, :shtSarf, null, :SeriNo, :FisNo, current_date, 1, :AKullaniciId, :AKullaniciId, 1, :SistemDovizID);

  insert into StokHareket (StokHareketId, HareketTipi, OrmeFisId, NeredenGirildi, NeredenUpdate, HamId, OrmeRafId, Miktar, Lot, DokumaHareketi, DepoId, RenkId, MelanjKodu, FlamKodu)
  Select :KayitStokHareketId, :shtSarf, :OrmeFisId, 0, 0, OrmeRaf.HamId, OrmeRaf.OrmeRafId, :AMiktar, OrmeRaf.Lot, 1, DepoRaf.DepoId, OrmeRaf.RenkId, OrmeRaf.MelanjKodu, OrmeRaf.FlamKodu
  from OrmeRaf
  left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
  where OrmeRaf.OrmeRafId = :AOrmeRafId;

  suspend;
end
