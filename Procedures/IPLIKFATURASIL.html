CREATE PROCEDURE IPLIKFATURASIL(
    AIPLIKFATURAID INTEGER,
    AKULLANICIID INTEGER)
AS
begin
  insert into IplikFaturaIptal(IplikFaturaId, FaturaSeriNo, FaturaNo, FaturaTarihi, FirmaId, HareketTipi, IptalKullaniciId)
      Select IplikFaturaId, SeriNo, FaturaNo, FaturaTarihi, FirmaId, HareketTipi, :AKullaniciId from IplikFatura
        where IplikFaturaId = :AIplikFaturaId;

  update IplikFatura set Entegrasyon = 0 where IplikFaturaId = :AIplikFaturaId;
  update IplikIrsaliye set IplikFaturaId = null where IplikIrsaliye.IplikFaturaId = :AIplikFaturaId;
  delete from IplikIrsaliyeDetay where IplikIrsaliyeDetay.IplikFaturaId = :AIplikFaturaId and NeredenGirildi = 2;
  delete from IplikFatura where IplikFaturaId = :AIplikFaturaId;
end
