    AORMERAFID INTEGER)
AS
declare variable OrmeRafId                     integer;
declare variable DigerRafId                    integer;
declare variable RafId                         varchar(10);
declare variable StokGirisNo                   integer;
declare variable HamId                         integer;
declare variable RenkId                        integer;
declare variable DesenId                       varchar(20);
declare variable DesenVaryantId                integer;
declare variable BoyaSiparisKayitId            integer;
declare variable OrmeSiparisDetayId            integer;
declare variable BoyaSiparisDetayId            integer;
declare variable BoyaSiparisBaskiId            integer;
declare variable OrmePlanlamaId                integer;
declare variable DokumaPlanlamaId              integer;
declare variable Sakat                         integer;
declare variable AyarTopu                      integer;
declare variable BoyaTalimatKayitId            integer;
declare variable BukumTalimatId                integer;
declare variable UretimPartiNo                 varchar(30);
declare variable CikisPartiNo                  varchar(30);
declare variable Lot                           varchar(150);
declare variable MelanjKodu                    varchar(20);
declare variable FlamKodu                      varchar(20);
declare variable UreticiFirmaId                varchar(20);
declare variable OrmeMakinaId                  varchar(10);
declare variable DokumaMakinaId                varchar(10);
declare variable OrmeciMakinaNo                varchar(10);
declare variable UreticiId                     integer;
declare variable Tamir                         integer;
declare variable EskiPartiYil                  integer;
declare variable EskiPartiNo                   integer;
declare variable KaliteNo                      integer;
declare variable FasonStok                     integer;
declare variable StokFirmaId                   varchar(20);
declare variable KaliteAciklamaTipiId          integer;
declare variable MamulProsesId                 integer;
declare variable BoyahaneFasonIade             integer;
declare variable Brut                          integer;
declare variable MiktarBirim                   integer;
declare variable BirimMaliyet                  Double Precision;
declare variable TamirTipiId                   integer;
declare variable DisTamirCozumSekliId          varchar(10);
declare variable CozguIsEmriKayitID            integer;
declare variable HasilIsEmriKayitID            integer;
declare variable IndigoIsEmriKayitID           integer;
declare variable LeventNo                      integer;
declare variable DepoPartiNo                   varchar(30);
declare variable TelAdedi                      Integer;
declare variable AtkiLotu                      varchar(30);
declare variable OlcuId                        Integer;
begin
  for Select OrmeRaf.OrmeRafId, OrmeRaf.HamId, OrmeRaf.RafId, OrmeRaf.Sakat, OrmeRaf.AyarTopu,
          OrmeRaf.StokGirisNo, OrmeRaf.RenkId, OrmeRaf.DesenId, OrmeRaf.DesenVaryantId, OrmeRaf.BoyaSiparisKayitId, OrmeRaf.OrmeSiparisDetayId, OrmeRaf.BoyaSiparisDetayId,
          OrmeRaf.BoyaSiparisBaskiId, OrmeRaf.OrmePlanlamaId, OrmeRaf.DokumaPlanlamaId, OrmeRaf.BoyaTalimatKayitId, OrmeRaf.UretimPartiNo, OrmeRaf.CikisPartiNo, OrmeRaf.Lot, OrmeRaf.MelanjKodu,
          OrmeRaf.FlamKodu, OrmeRaf.UreticiFirmaId, OrmeRaf.OrmeMakinaId, OrmeRaf.DokumaMakinaId, OrmeRaf.OrmeciMakinaNo, OrmeRaf.UreticiId, OrmeRaf.EskiPartiYil, OrmeRaf.DisTamirCozumSekliId,
          OrmeRaf.TamirTipiId, OrmeRaf.KaliteAciklamaTipiId, OrmeRaf.MamulProsesId, OrmeRaf.Brut, OrmeRaf.MiktarBirim, OrmeRaf.BoyahaneFasonIade, OrmeRaf.Tamir, OrmeRaf.EskiPartiNo,
          OrmeRaf.KaliteNo, OrmeRaf.FasonStok, OrmeRaf.StokFirmaId, OrmeRaf.BirimMaliyet, OrmeRaf.CozguIsEmriKayitID, OrmeRaf.HasilIsEmriKayitID, OrmeRaf.LeventNo, OrmeRaf.DepoPartiNo, OrmeRaf.TelAdedi, OrmeRaf.BukumTalimatId, OrmeRaf.AtkiLotu,
          OrmeRaf.OlcuId, OrmeRaf.IndigoIsEmriKayitID from OrmeRaf
    where (OrmeRafId = :AOrmeRafId or :AOrmeRafId is Null)
    into :OrmeRafId, :HamId, :RafId, :Sakat, :AyarTopu,
          :StokGirisNo, :RenkId, :DesenId, :DesenVaryantId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisDetayId,
          :BoyaSiparisBaskiId, :OrmePlanlamaId, :DokumaPlanlamaId, :BoyaTalimatKayitId, :UretimPartiNo, :CikisPartiNo, :Lot, :MelanjKodu,
          :FlamKodu, :UreticiFirmaId, :OrmeMakinaId, :DokumaMakinaId, :OrmeciMakinaNo, :UreticiId, :EskiPartiYil, :DisTamirCozumSekliId,
          :TamirTipiId, :KaliteAciklamaTipiId, :MamulProsesId, :Brut, :MiktarBirim, :BoyahaneFasonIade, :Tamir, :EskiPartiNo,
          :KaliteNo, :FasonStok, :StokFirmaId, :BirimMaliyet, :CozguIsEmriKayitID, :HasilIsEmriKayitID, :LeventNo, :DepoPartiNo, :TelAdedi, :BukumTalimatId, :AtkiLotu,
          :OlcuId, :IndigoIsEmriKayitID do
    begin
      for Select OrmeRafId from OrmeRaf
      where OrmeRaf.HamId = :HamId and OrmeRaf.RafId = :RafId and OrmeRaf.Sakat = :sakat and OrmeRaf.AyarTopu = :AyarTopu and
            IntKarsilastir(OrmeRaf.StokGirisNo, :StokGirisNo) = 1 and
            IntKarsilastir(OrmeRaf.RenkId, :RenkID) = 1 and
            StrKarsilastir(OrmeRaf.DesenId, :DesenId) = 1 and
            IntKarsilastir(OrmeRaf.DesenVaryantId, :DesenVaryantId) = 1 and
            IntKarsilastir(OrmeRaf.BoyaSiparisKayitId, :BoyaSiparisKayitID) = 1 and
            IntKarsilastir(OrmeRaf.OrmeSiparisDetayId, :OrmeSiparisDetayID) = 1 and
            IntKarsilastir(OrmeRaf.BoyaSiparisDetayId, :BoyaSiparisDetayID) = 1 and
            IntKarsilastir(OrmeRaf.BoyaSiparisBaskiId, :BoyaSiparisBaskiID) = 1 and
            IntKarsilastir(OrmeRaf.OrmePlanlamaId, :OrmePlanlamaID) = 1 and
            IntKarsilastir(OrmeRaf.DokumaPlanlamaId, :DokumaPlanlamaID) = 1 and
            IntKarsilastir(OrmeRaf.BoyaTalimatKayitId, :BoyaTalimatKayitId) = 1 and
            IntKarsilastir(OrmeRaf.BukumTalimatId, :BukumTalimatId) = 1 and
            StrKarsilastir(OrmeRaf.UretimPartiNo, :UretimPartiNo) = 1 and
            StrKarsilastir(OrmeRaf.CikisPartiNo, :CikisPartiNo) = 1 and
            StrKarsilastir(OrmeRaf.Lot, :Lot) = 1 and
            StrKarsilastir(OrmeRaf.MelanjKodu, :MelanjKodu) = 1 and
            StrKarsilastir(OrmeRaf.FlamKodu, :FlamKodu) = 1 and
            StrKarsilastir(OrmeRaf.UreticiFirmaId, :UreticiFirmaId) = 1 and
            StrKarsilastir(OrmeRaf.OrmeMakinaId, :OrmeMakinaId) = 1 and
            StrKarsilastir(OrmeRaf.DokumaMakinaId, :DokumaMakinaId) = 1 and
            StrKarsilastir(OrmeRaf.OrmeciMakinaNo, :OrmeciMakinaNo) = 1 and
            IntKarsilastir(OrmeRaf.UreticiId, :UreticiId) = 1 and
            IntKarsilastir(OrmeRaf.EskiPartiYil, :EskiPartiYil) = 1 and
            StrKarsilastir(OrmeRaf.DisTamirCozumSekliId, :DisTamirCozumSekliId) = 1 and
            IntKarsilastir(OrmeRaf.TamirTipiId, :TamirTipiId) = 1 and
            IntKarsilastir(OrmeRaf.KaliteAciklamaTipiId, :KaliteAciklamaTipiId) = 1 and
            IntKarsilastir(OrmeRaf.MamulProsesId, :MamulProsesId) = 1 and
            IntKarsilastir(OrmeRaf.Brut, :Brut) = 1 and
            IntKarsilastir(OrmeRaf.MiktarBirim, :MiktarBirim) = 1 and
            OrmeRaf.BoyahaneFasonIade = :BoyahaneFasonIade and
            OrmeRaf.Tamir = :Tamir and
            IntKarsilastir(OrmeRaf.EskiPartiNo, :EskiPartiNo) = 1 and
            OrmeRaf.KaliteNo = :KaliteNo and
            OrmeRaf.FasonStok = :FasonStok and
            OrmeRaf.StokFirmaId = :StokFirmaId and
            IntKarsilastir(OrmeRaf.CozguIsEmriKayitID, :CozguIsEmriKayitID) = 1 and
            IntKarsilastir(OrmeRaf.HasilIsEmriKayitID, :HasilIsEmriKayitID) = 1 and
            IntKarsilastir(OrmeRaf.IndigoIsEmriKayitID, :IndigoIsEmriKayitID) = 1 and
            IntKarsilastir(OrmeRaf.LeventNo, :LeventNo) = 1 and
            StrKarsilastir(OrmeRaf.DepoPartiNo, :DepoPartiNo) = 1 and
            IntKarsilastir(OrmeRaf.TelAdedi, :TelAdedi) = 1 and
            StrKarsilastir(OrmeRaf.AtkiLotu, :AtkiLotu) = 1 and
            IntKarsilastir(OrmeRaf.OlcuId, :OlcuId) = 1 and
            /*OrmeRaf.BirimMaliyet = :BirimMaliyet and raflama iptal edilmis*/
            OrmeRaf.OrmeRafId <> :OrmeRafId
      into :DigerRafId do
    begin
      update OrmeRaf set IslemDevamEdiyor = 1 where OrmeRafId = :DigerRafId;
      update OrmeStokKaydirma set OrmeRafId = :OrmeRafId where OrmeRafId = :DigerRafId;
      update OrmeRafDetay set OrmeRafId = :OrmeRafId where OrmeRafId = :DigerRafId;
      update OrmeRafTransferFisiDetay set YeniOrmeRafId = :OrmeRafId where YeniOrmeRafId = :DigerRafId;
      update StokHareket set OrmeRafId = :OrmeRafId, NeredenUpdate = 3 where OrmeRafId = :DigerRafId;
      update OrmeRafTransferFisiDetay set OrmeRafId = :OrmeRafId where OrmeRafId = :DigerRafId;
      update OrmeRaf set IslemDevamEdiyor = 0 where OrmeRafId = :DigerRafId;
    end
  end
end
