CREATE PROCEDURE SONRAKIASAMAYIBEKLEMEYEAL(
    APARTISINIFID INTEGER,
    APARTIASAMALARIID INTEGER,
    AARAKONTROLTIPIID INTEGER,
    AFIZIKKONTROL INTEGER,
    AHAMKONTROL INTEGER)
AS
declare variable PartiKayitId     integer;
declare variable PartiYil         integer;
declare variable PartiId         integer;
declare variable Sira             integer;
declare variable SonrakiSira      integer;
declare variable PartiStokBekleId  integer;
declare variable PartiStokBekleBitti integer;
declare variable FPartiAsamalariId integer;
declare variable FBlob             Blob;
declare variable Aciklama          varchar(1024);
declare variable F1                varchar(1024);
declare variable AsamaAdi          varchar(40);
declare variable Konu              varchar(50);
declare variable AraKontrolTipiAdi varchar(30);
declare variable KayitSayisi integer;
declare variable FirmaAdi    varchar(100);
declare variable ReceteId    varchar(20);
declare variable ReceteAdi   varchar(40);
declare variable BoyaSiparisYil        integer;
declare variable BoyaSiparisId         integer;
declare variable FHamAdi varchar(255);
begin
  if (AAraKontrolTipiId is not null) then
  begin
    Select Count(*) from PartiAsamaAraKontrol
      where PartiAsamaAraKontrol.PartiAsamalariId = :APartiAsamalariId and PartiAsamaAraKontrol.PartiSinifId = :APartiSinifId and PartiAsamaAraKontrol.OnayDurumu = 2
      into :KayitSayisi;

    if (KayitSayisi > 0) then exit;

    Select Ham.Adi from PartiStok
      Left Join Ham on Ham.HamID = PartiStok.HamID
      where PartiStok.PartiSinifId = :APartiSinifId
    into :FHamAdi;
  end

  if (APartiAsamalariId is not null) then
  begin
    Select PartiAsamalari.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiAsamalari.Sira, Asama.Adi, Firma.Adi,
           Parti.ReceteId, Recete.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId from PartiAsamalari
      Left Join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
      Left Join Parti on Parti.PartiKayitId = PartiAsamalari.PartiKayitId
      Left Join Recete on Recete.ReceteId = Parti.ReceteId
      Left Join Firma on Firma.FirmaId = Parti.FirmaId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      where PartiAsamalari.PartiAsamalariID = :APartiAsamalariID
      into :PartiKayitId, :PartiYil, :PartiId, :Sira, :AsamaAdi, :FirmaAdi, :ReceteId, :ReceteAdi, :BoyaSiparisYil, :BoyaSiparisId;
  end
  else
  begin
    AsamaAdi = '';
    Sira = -1;

    Select PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, Firma.Adi,
           Parti.ReceteId, Recete.Adi, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId from PartiStok
      Left Join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
      Left Join Recete on Recete.ReceteId = Parti.ReceteId
      Left Join Firma on Firma.FirmaId = Parti.FirmaId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
      where PartiStok.PartiSinifID = :APartiSinifId
      into :PartiKayitId, :PartiYil, :PartiId, :FirmaAdi, :ReceteId, :ReceteAdi, :BoyaSiparisYil, :BoyaSiparisId;
  end

  if (AAraKontrolTipiID is not null) then
  begin
    Select AraKontrolTipi.Adi from AraKontrolTipi where AraKontrolTipiId = :AAraKontrolTipiID into :AraKontrolTipiAdi;

    Aciklama = FHamAdi ||CAst(' Tipinin ' || AsamaAdi || ' Aşamasının ' || AraKontrolTipiAdi || ' Ara Kontrolü Ret Edildiği için, Parti''nin sonraki Aşaması Sistem Tarafından Beklemeye Alındı.' as Blob character set UTF8);
  end
  else if (AFizikKontrol = 1) then
  begin
    Aciklama = Cast(' Fizik Kontrol Onayı Ret Edildiği için, Parti''nin sonraki Aşaması Sistem Tarafından Beklemeye Alındı.' as Blob character set UTF8);
  end
  else if (AHamKontrol = 1) then
  begin
    Aciklama = Cast(' Ham Kontrol Onayı Ret Edildiği için, Parti''nin sonraki Aşaması Sistem Tarafından Beklemeye Alındı.' as Blob character set UTF8);
  end
  else
    Aciklama = Cast('Hatalı İşlem Var, Sistem Yöneticisine Haber Veriniz..' as Blob character set UTF8);

  Select min(Sira) from PartiAsamalari
     left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
     where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.Sira > :Sira and
           PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamaStok.GirisTarihi is null
     into :SonrakiSira;

  if (SonrakiSira is not null) then
  begin
    select PartiAsamalari.PartiAsamalariId from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamalari.Sira = :SonrakiSira
      into :FPartiAsamalariId;


    if (FPartiAsamalariId is not null) then
    begin
      Select PartiStokBekle.PartiStokBekleId, PartiStokBekle.Bitti from PartiStokBekle
        where PartiStokBekle.PartiSinifId = :APartiSinifId and PartiStokBekle.PartiAsamalariId = :FPartiAsamalariId
        into :PartiStokBekleId, :PartiStokBekleBitti;

      if (PartiStokBekleId is null) then
      begin
        FBlob = Cast(Aciklama as Blob character set UTF8);
        Insert into PartiStokBekle(PartiKayitId, PartiSinifId, PartiAsamalariId, Notu, OtomatikBekleme)
                     Values(:PartiKayitId, :APartiSinifId, :FPartiAsamalariId, :FBlob, 1);

      end
      else if (PartiStokBekleBitti = 0) then
      begin
        if (AAraKontrolTipiID is not null) then
          Aciklama = FHamAdi || Cast(' Tipinin ' || AsamaAdi || ' Aşamasının ' || AraKontrolTipiAdi || ' Ara Kontrolü Ret Edilmiştir.Sonraki Aşaması Daha Önce Beklemeye Alındığı için işlem Yapılmamıştır.' as Blob character set UTF8);
        else if (AFizikKontrol = 1 ) then
          Aciklama = Cast('Fizik Kontrol Onayı Ret Edilmiştir.Sonraki Aşaması Daha Önce Beklemeye Alındığı için işlem Yapılmamıştır.' as Blob character set UTF8);
        else if (AHamKontrol = 1 ) then
          Aciklama = Cast('Ham Kontrol Onayı Ret Edilmiştir.Sonraki Aşaması Daha Önce Beklemeye Alındığı için işlem Yapılmamıştır.' as Blob character set UTF8);
      end
      else Update PartiStokBekle Set BitirenKullaniciId = Null where PartiStokBekle.PartiStokBekleId = :PartiStokBekleId;
    end
  end
  else
  begin
    if (AAraKontrolTipiID is not null) then
      Aciklama = FHamAdi ||Cast( ' Tipinin ' || AsamaAdi || ' Aşamasının ' || AraKontrolTipiAdi || ' Ara Kontrolü Ret Edilmiştir.Sonra Yapılacak Aşaması Olmadığı için beklemeye alınamamıştır.' as Blob character set UTF8);
    else if (AFizikKontrol = 1) then
      Aciklama = Cast('Fizik Kontrol Onayı Ret Edilmiştir.Sonra Yapılacak Aşaması Olmadığı için beklemeye alınamamıştır.' as Blob character set UTF8);
    else if (AHamKontrol = 1) then
      Aciklama = Cast('Ham Kontrol Onayı Ret Edilmiştir.Sonra Yapılacak Aşaması Olmadığı için beklemeye alınamamıştır.' as Blob character set UTF8);
  end


  if (BoyaSiparisYil is not null) then
    F1 = Cast('Sipariş No : (' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisID) as Blob character set UTF8);
  else
    F1 = Cast('' as Blob character set UTF8);

  if (ReceteID is not null) then F1 = F1 ||Cast( chr(13) || 'Reçete : (' || ReceteId || ') ' || ReceteAdi as Blob character set UTF8);

  Aciklama = Aciklama || Cast(chr(13) as Blob character set UTF8) || F1;
  FBlob = Cast(Aciklama as Blob character set UTF8);

  Konu = 'Parti : (' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

  Konu = Konu || SubString(FirmaAdi from 1 for (49 - Char_Length(Konu)));

  if (AAraKontrolTipiID is not null) then
    insert into GelenMail(KullaniciId, KimdenKullaniciId, KimdenKullaniciAdi, Konu, Mesaj)
      Select Kullanici.KullaniciId, 1, 'Ara Kontrol Ret', :Konu, :FBlob from Kullanici where Kullanici.AraKontrolRetBildirilsin = 1;
  else if (AFizikKontrol = 1) then
    insert into GelenMail(KullaniciId, KimdenKullaniciId, KimdenKullaniciAdi, Konu, Mesaj)
      Select Kullanici.KullaniciId, 1, 'Fizik Kontrol Ret', :Konu, :FBlob from Kullanici where Kullanici.AraKontrolRetBildirilsin = 1;
  else if (AHamKontrol = 1) then
    insert into GelenMail(KullaniciId, KimdenKullaniciId, KimdenKullaniciAdi, Konu, Mesaj)
      Select Kullanici.KullaniciId, 1, 'Ham Kontrol Ret', :Konu, :FBlob from Kullanici where Kullanici.AraKontrolRetBildirilsin = 1;
end
