CREATE PROCEDURE PARTITARTISIPARISMIKTARGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    SIPARISKILO NUMERIC(12,3),
    SIPARISBRUTKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    IPTALKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    IPTALMETRE NUMERIC(12,3),
    BRUTSEVKEDILENKILO NUMERIC(12,3),
    BRUTSEVKEDILENMETRE NUMERIC(12,3),
    NETSEVKEDILENKILO NUMERIC(12,3),
    NETSEVKEDILENMETRE NUMERIC(12,3),
    SEVKKILOFIRE NUMERIC(12,3),
    SEVKMETREFIRE NUMERIC(12,3),
    GENELFIREORANI NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    SEVKEHAZIRKILO NUMERIC(12,3),
    SEVKEHAZIRNETKILO NUMERIC(12,3),
    SEVKEHAZIRMETRE NUMERIC(12,3))
AS
declare variable SiparisMaximumFireOrani       Double Precision;
declare variable FFason      integer;
declare variable ToplamBrutKilo  Numeric(12,3);
declare variable ToplamNetKilo   Numeric(12,3);
declare variable PartilenenMiktardanIptaliDus EgeBoolean;
begin
  select PartilenenMiktardanIptaliDus from Sistem where Sistem.SistemId = 1 into :PartilenenMiktardanIptaliDus;

  For Select PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisMetre,
             BoyaSiparisDetay.PartilenenKilo, BoyaSiparisDetay.IptalKilo, BoyaSiparisDetay.PartilenenMetre, BoyaSiparisDetay.IptalMetre,
             BoyaSiparisDetay.MaximumFireOrani, BoyaSiparis.Fason, BoyaSiparisDetay.SiparisBrutKilo from PartiStok
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitID = BoyaSiparisDetay.BoyaSiparisKayitId
  where PartiStok.PartiKayitID = :APartiKayitID
  group by PartiStok.BoyaSiparisDetayId, BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisMetre,
             BoyaSiparisDetay.PartilenenKilo, BoyaSiparisDetay.IptalKilo, BoyaSiparisDetay.PartilenenMetre, BoyaSiparisDetay.IptalMetre,
             BoyaSiparisDetay.MaximumFireOrani, BoyaSiparis.Fason, BoyaSiparisDetay.SiparisBrutKilo
  into :BoyaSiparisDetayId, :SiparisKilo, :SiparisMetre, :PartilenenKilo, :IptalKilo, :PartilenenMetre, :IptalMetre, :SiparisMaximumFireOrani, :FFason, :SiparisBrutKilo do
  begin
    BrutSevkEdilenKilo = Null;
    BrutSevkEdilenMetre = Null;
    NetSevkEdilenKilo = Null;
    NetSevkEdilenMetre = Null;
    SevkeHazirKilo    = Null;
    SevkeHazirNetKilo = Null;
    SevkeHazirMetre   = Null;

    Select Sum(PartiStok.BrutSevkKilo), Sum(PartiStok.BrutSevkMetre), Sum(PartiStok.NetSevkKilo), Sum(PartiStok.NetSevkMetre),
           Sum(PartiStok.Kilo - PartiStok.BrutSevkKilo), Sum(PartiStok.NetKilo - PartiStok.NetSevkKilo),
           Sum(PartiStok.Metre - PartiStok.BrutSevkMetre) from BoyaSiparisDetay
      left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
      where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and PartiStok.PartiStokIptalKayitID is null
        into :BrutSevkEdilenKilo, :BrutSevkEdilenMetre, :NetSevkEdilenKilo, :NetSevkEdilenMetre, :SevkeHazirKilo, :SevkeHazirNetKilo, :SevkeHazirMetre;

    if (NetSevkEdilenKilo <> 0 and NetSevkEdilenKilo is not null and BrutSevkEdilenKilo <> 0 and BrutSevkEdilenKilo is not null) then SevkKiloFire = (BrutSevkEdilenKilo - NetSevkEdilenKilo) / BrutSevkEdilenKilo *  100;
      else SevkKiloFire = 0;

    if (NetSevkEdilenMetre <> 0 and NetSevkEdilenMetre is not null and BrutSevkEdilenMetre <> 0 and BrutSevkEdilenMetre is not null) then SevkMetreFire = (BrutSevkEdilenMetre - NetSevkEdilenMetre) / BrutSevkEdilenMetre *  100;
      else SevkMetreFire = 0;

    if (FFason = 0) then
      KalanKilo = SiparisKilo - NetSevkEdilenKilo;
    else
      KalanKilo = SiparisBrutKilo - BrutSevkEdilenKilo;

    Select Sum(PartiStok.Kilo), Sum(PartiStok.NetKilo) from BoyaSiparisDetay
      left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
      where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId and PartiStok.PartiStokIptalKayitID is null and
            PartiStok.TamamiTartildi = 1
       into :ToplamBrutKilo, :ToplamNetKilo;

    if (ToplamBrutKilo <> 0) then
      GenelFireOrani = (ToplamBrutKilo - ToplamNetKilo) / ToplamBrutKilo * 100;
    else
      GenelFireOrani = 0;

    if (PartilenenMiktardanIptaliDus = 1) then
    begin
      PartilenenKilo = PartilenenKilo - IptalKilo;
      PartilenenMetre = PartilenenMetre - IptalMetre;
    end

    suspend;
  end
end
