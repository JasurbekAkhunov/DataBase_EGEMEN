CREATE PROCEDURE DEPOMAMULSTOKLISTESI(
    ADEPOID VARCHAR(10),
    APARTIYIL INTEGER,
    APARTIID INTEGER,
    APARTITIPIID INTEGER,
    ASIPARISYIL INTEGER,
    ASIPARISID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    PSKALITEKONTROLID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    TOPNO INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    RAFID VARCHAR(10),
    KALITENO INTEGER,
    REFERANSKODU VARCHAR(30),
    KALITEADI VARCHAR(250),
    BOYASIPARISIDSTR VARCHAR(30),
    PARTISIRANO INTEGER)
AS
declare variable APartiKayitId integer;
declare Variable FBoyaSiparisId  integer;
begin
  if (APartiId is not null) then
  begin
    Select PartiKayitId from Parti
       where Parti.PartiYil = :APartiYil and Parti.PartiId = :APartiId into :APartiKayitId;
    if (APartiKayitId is null) then APartiKayitId = -1;
  end
  FBoyaSiparisId = null;
  if (ASiparisYil is not null and ASiparisId is not null) then
  begin
    Select BoyaSiparisKayitId From BoyaSiparis Where BoyaSiparisYil = :ASiparisYil and BoyaSiparisId  = :ASiparisId into :FBoyaSiparisId;
  end

  for Select PartiStokTarti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.FirmaId, Firma.Adi, MamulDepoTransferFisiDetay.PartiSinifId, MamulDepoTransferFisiDetay.PartiStokTartiId,
             PartiStokTarti.PartiStokTartiId, PartiStok.HamAdiToplami, PsKaliteKontrol.TopNo, PartiStokTarti.Kilo, PartiStokTarti.Metre,
             Parti.ReceteId, Recete.Adi, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisBaski.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
             MamulDepoTransferFisiDetay.RafId, PartiStokTarti.KaliteNo, Ham.ReferansKodu, Kalite.KaliteAdi, PartiStok.PartiSiraNo from PartiStokTarti
    left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId and
                                            MamulDepoTransferFisiDetay.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    left join PsKaliteKontrol on PsKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    left join PartiStok on PartiStok.PartiSinifId = MamulDepoTransferFisiDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisBaskiId = PartiStok.BoyaSiparisBaskiId
    left join Desen on Desen.DesenId = BoyaSiparisBaski.DesenId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
    left join MamulDepoTransferFisi on MamulDepoTransferFisi.MamulDepoTransferFisiId = PartiStokTarti.MamulDepoTransferFisiId
    left join Ham on Ham.HamId = PartiStok.HamId
    Left join Kalite on  Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    where PartiStokTarti.BoyaIrsaliyeDetayId is null and PartiStokTarti.MamulDepoTransferFisiId is not null and
          PartistokTarti.HamStokSarfFisiDetayId is null and PartiStok.PartiStokIptalKayitId is null and
          (PartiStokTarti.PartiKayitId = :APartiKayitId or :APartiKayitId is null) and MamulDepoTransferFisi.TransferDepoId = :ADepoId and
          (BoyaSiparisDetay.BoyaSiparisKayitId = :FBoyaSiparisId or :FBoyaSiparisId is null)
  into :PartiKayitId, :PartiYil, :PartiId, :FirmaId, :FirmaAdi, :PartiSinifId, :PsKaliteKontrolId, :PartiStokTartiId,
       :HamAdiToplami, :TopNo, :Kilo, :Metre, :ReceteId, :ReceteAdi, :MusteriRenkKoduveAdi, :DesenId, :DesenAdi, :VaryantAdi,
       :RafId, :KaliteNo, :ReferansKodu, :KaliteAdi, :PartiSiraNo do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    if (FBoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(ASiparisYil)  || ') ' || EgemenIntegerToStr(ASiparisId) || ' ';
    else BoyaSiparisIdStr = '';
    suspend;
  end
end
