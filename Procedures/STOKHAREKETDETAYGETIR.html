CREATE PROCEDURE STOKHAREKETDETAYGETIR(
    AEVRAKID INTEGER,
    AEVRAKTIPI INTEGER)
RETURNS (
    STOKHAREKETDETAYID INTEGER,
    STOKHAREKETID INTEGER,
    HAREKETTIPI INTEGER,
    BARKOD VARCHAR(30),
    IPLIKSTOKID INTEGER,
    IPLIKSTOKNUMARATOR INTEGER,
    KILO NUMERIC(18,3),
    BRUTKILO NUMERIC(18,3),
    METRE NUMERIC(18,3),
    AMBALAJNO VARCHAR(10),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40),
    KAPADEDI DOUBLE PRECISION,
    BOBINGRAMAJI DOUBLE PRECISION,
    KONIKRENGIADI ADI20)
AS
declare variable FKilo numeric(18,3);
declare variable FBrutKilo numeric(18,3);
declare variable FMetre numeric(18,3);
begin
  for Select StokHareketDetay.StokHareketDetayID, StokHareketDetay.StokHareketID, StokHareketDetay.HareketTipi, StokHareketDetay.IplikStokID, StokHareketDetay.IplikStokNumarator,
             StokHareketDetay.Kilo, StokHareketDetay.BrutKilo, StokHareketDetay.Metre, StokHareketDetay.AmbalajNo, StokHareketDetay.Notu, StokHareketDetay.InsertKullaniciID,
             StokHareketDetay.InsertTarihi, StokHareketDetay.KullaniciID, StokHareketDetay.DegisimTarihi, IK.Adi, SK.Adi, StokHareketDetay.Barkod,
             StokHareketDetay.KapAdedi, StokHareketDetay.BobinGramaji, KonikRengi.Adi from StokHareket
  Left Join StokHareketDetay on StokHareketDetay.StokHareketID = StokHareket.StokHareketID
  left join KonikRengi on StokHareketDetay.KonikRengiId = KonikRengi.KonikRengiId
  left join Kullanici IK on IK.KullaniciId  = StokHareket.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = StokHareket.KullaniciId
  where ((StokHareket.OrmeFisId = :AEvrakId and :AEvrakTipi = 0) or
        (StokHareket.OrmeIrsaliyeId = :AEvrakId and :AEvrakTipi = 1) or
        (StokHareket.OrmeFaturaId = :AEvrakId and StokHareket.FaturaKesilecek = 1 and :AEvrakTipi = 2)) and StokHareketDetay.StokHareketDetayId is not Null and
        StokHareketDetay.BolunduguStokHareketDetayId is Null
  into :StokHareketDetayID, :StokHareketID, :HareketTipi, :IplikStokID, :IplikStokNumarator, :Kilo, :BrutKilo, :Metre,
       :AmbalajNo, :Notu, :InsertKullaniciID, :InsertTarihi, :KullaniciID, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi, :Barkod, :KapAdedi, :BobinGramaji, :KonikRengiAdi do
  begin
    --Select sum(Kilo), sum(BrutKilo), sum(Metre) from StokHareketDetay where StokHareketDetay.BolunduguStokHareketDetayId = :StokHareketDetayID into :FKilo, :FBrutKilo, :FMetre;

    Select sum(StokHareketDetay.Kilo), sum(StokHareketDetay.BrutKilo), sum(StokHareketDetay.Metre) from BolunenStokDetaylar(:StokHareketDetayID)
      left join StokHareketDetay on StokHareketDetay.StokHareketDetayId = BolunenStokDetaylar.StokHareketDetayId into :FKilo, :FBrutKilo, :FMetre;

    if (FKilo is not null) then Kilo = Kilo + FKilo;
    if (FBrutKilo is not null) then BrutKilo = BrutKilo + FBrutKilo;
    if (FMetre is not null) then Metre = Metre + FMetre;

    suspend;
  end
end
