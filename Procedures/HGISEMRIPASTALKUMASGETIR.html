CREATE PROCEDURE HGISEMRIPASTALKUMASGETIR(
    AHGISEMRIPASTALID INTEGER)
RETURNS (
    HGISEMRIPASTALKUMASID INTEGER,
    HGISEMRIPASTALID INTEGER,
    HGSIPARISDETAYKUMASID INTEGER,
    KATSAYISI INTEGER,
    MODELKUMASID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI ADI50,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATDOVIZID DOVIZSIZE,
    LABRECETEID INTEGER,
    LABRECETEADI VARCHAR(40),
    LABRECETEKODU RECETEKODUSIZE,
    LABRECETEATISID INTEGER,
    LABRECETEATISNO VARCHAR(20),
    PANTONEKODU VARCHAR(30),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(20),
    BOYALIGRAMAJ INTEGER,
    BOYALIEN INTEGER,
    ANAKUMAS EGEBOOLEAN,
    ANAKUMASSTR VARCHAR(10),
    ANAKUMASMODELKUMASID INTEGER,
    INSERTKULLANICIID INTEGER,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    DEGISIMTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    KULLANICIADI VARCHAR(40),
    TOPLAMMIKTAR INTEGER)
AS
declare variable FToplamMiktar Integer;
begin
  Select Sum(HGIsEmriPastalBeden.Miktar) From HGIsEmriPastalBeden Where HGIsEmriPastalBeden.HGIsEmriPastalID = :AHGIsEmriPastalID Into :FToplamMiktar;

  For Select HGIsEmriPastalKumas.HGIsEmriPastalKumasID, HGIsEmriPastalKumas.HGIsEmriPastalID, HGIsEmriPastalKumas.HGSiparisDetayKumasID, HGIsEmriPastalKumas.KatSayisi,
             HGSiparisDetayKumas.ModelKumasID, HGSiparisDetayKumas.HamId, Ham.Adi, HGSiparisDetayKumas.RenkId, HGSiparisDetayKumas.BirimFiyat,
             HGSiparisDetayKumas.BirimFiyatDovizId, HGIsEmriPastalKumas.InsertKullaniciId, HGIsEmriPastalKumas.InsertTarihi, HGIsEmriPastalKumas.KullaniciId, HGIsEmriPastalKumas.DegisimTarihi,
             IK.Adi, SK.Adi, HGSiparisDetayKumas.LabReceteId, LabRecete.Adi, HGSiparisDetayKumas.LabReceteKodu, HGSiparisDetayKumas.LabReceteAtisID,
             LabReceteAtis.AtisNoStr, HGSiparisDetayKumas.RenkAdi, HGSiparisDetayKumas.PantoneKodu, HGSiparisDetayKumas.TupAcikEnMayli,
             HGSiparisDetayKumas.BoyaliGramaj, HGSiparisDetayKumas.BoyaliEn, HGSiparisDetayKumas.AnaKumas, iif(HGSiparisDetayKumas.AnaKumas = 1, 'Ana Kuma≈ü', 'Garni'), HGSiparisDetayKumas.AnaKumasModelKumasID, :FToplamMiktar * HGIsEmriPastalKumas.KatSayisi  From HGIsEmriPastalKumas
    Left Join HGSiparisDetayKumas on HGSiparisDetayKumas.HGSiparisDetayKumasID = HGIsEmriPastalKumas.HGSiparisDetayKumasID
    left join Ham on Ham.HamId = HGSiparisDetayKumas.HamId
    left join LabRecete on LabRecete.LabReceteId = HGSiparisDetayKumas.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = HGSiparisDetayKumas.LabReceteAtisId
    left join Kullanici IK on IK.KullaniciId  = HGIsEmriPastalKumas.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = HGIsEmriPastalKumas.KullaniciId
    Where HGIsEmriPastalKumas.HGIsEmriPastalID = :AHGIsEmriPastalID
    into :HGIsEmriPastalKumasID, :HGIsEmriPastalID, :HGSiparisDetayKumasID, :KatSayisi, :ModelKumasID, :HamId, :HamAdi, :RenkId,
         :BirimFiyat, :BirimFiyatDovizId, :InsertKullaniciId, :InsertTarihi, :KullaniciId, :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi,
         :LabReceteId, :LabReceteAdi, :LabReceteKodu, :LabReceteAtisID, :LabReceteAtisNo, :RenkAdi, :PantoneKodu,
         :TupAcikEnMayli, :BoyaliGramaj, :BoyaliEn, :AnaKumas, :AnaKumasStr, :AnaKumasModelKumasID, :ToplamMiktar do
  begin
    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikEnMayli) into :TupAcikEnMayliStr;

    Suspend;
  end
end
