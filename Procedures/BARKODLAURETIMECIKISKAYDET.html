CREATE PROCEDURE BARKODLAURETIMECIKISKAYDET(
    AIPLIKSTOKID INTEGER,
    ANUMARATOR INTEGER,
    AIPLIKIRSALIYEID INTEGER,
    AHARMANKAYITID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    STATUS INTEGER)
AS
declare variable GirisIplikIrsaliyeDetayId integer;
declare variable CikisIplikIrsaliyeDetayId integer;
declare variable HamId integer;
declare variable IplikDepoId    varchar(10);
declare variable IplikIrsaliyeDetayId integer;
declare variable Lot                  varchar(30);
declare variable HLot                 varchar(30);
declare variable RenkId               integer;
declare variable HRenkId              integer;
declare variable Miktar               numeric(12,3);
declare variable NetMiktar            numeric(12,3);
declare variable KS integer;
declare variable HareketTipi integer;
declare variable MaxNumarator integer;
declare variable hstElyaf integer;
declare variable hstTelef integer;
declare variable HamTipiId integer;
declare variable BirimFiyat Double Precision;
declare variable GirisFirmaID     varchar(20);
declare variable HarmanSipFirmaID varchar(20);
declare variable GirisHareketTipi SmallInt;
declare variable shtFasonAlis SmallInt;
declare variable shtDepoTransfer SmallInt;
declare variable HRDisinaCikilmasin Smallint;
begin
  select HRDisinaCikilmasin from Sistem into :HRDisinaCikilmasin;
  select shtUretimeCikis, shtFasonAlis, shtDepoTransfer from IplikStokHareketTipiGetir into :HareketTipi, :shtFasonAlis, shtDepoTransfer;
  Select hstElyaf, hstTelef from HamTipiGetir into :hstElyaf, :hstTelef;

  if (AIplikIrsaliyeId is not null) then
    Select Max(Numarator) from IplikStok where IplikStok.IplikStokID = :AIplikStokId into :MaxNumarator;
  else
    MaxNumarator = ANumarator;

  if (MaxNumarator is null) then
  begin
    Status = -1;
    suspend;
    exit;
  end


  Select First 1 Distinct IplikStok.CikisIplikIrsaliyeDetayId, IplikStok.IplikIrsaliyeDetayId, IplikStok.HamId, IplikStok.IplikDepoId,
         IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.RenkId, IplikStok.Miktar, IplikStok.NetMiktar, Ham.HamTipiId, IplikIrsaliyeDetay.BirimFiyat,
         Coalesce(TransIplikIrsaliyeDetay.FirmaID, IplikIrsaliyeDetay.FirmaID), Coalesce(TransIplikIrsaliyeDetay.HareketTipi, IplikIrsaliyeDetay.HareketTipi) from IplikStok
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = IplikStok.IplikIrsaliyeDetayId
    left join Ham on Ham.HamID = IplikIrsaliyeDetay.HamId
    left join IplikStok TransferIplikStok on TransferIplikStok.CikisIplikIrsaliyeDetayID = IplikIrsaliyeDetay.IplikIrsaliyeDetayId and IplikIrsaliyeDetay.HareketTipi = :shtDepoTransfer
    left join IplikIrsaliyeDetay TransIplikIrsaliyeDetay on TransIplikIrsaliyeDetay.IplikIrsaliyeDetayID = TransferIplikStok.IplikIrsaliyeDetayId
    where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator
    into :CikisIplikIrsaliyeDetayId, :GirisIplikIrsaliyeDetayId, :HamId, :IplikDepoId, :Lot, :RenkId, :Miktar, :NetMiktar, :HamTipiId, :BirimFiyat, :GirisFirmaID, :GirisHareketTipi;

  if (GirisIplikIrsaliyeDetayId is null) then
  begin
    Status = -1;
    suspend;
    exit;
  end

  if (AIplikIrsaliyeId is not null and GirisHareketTipi = shtFasonAlis) then
  begin
    Select BoyaSiparis.FirmaId from Harman
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Harman.BoyaSiparisKayitID
    where Harman.HarmanKayitID = :AHarmanKayitId and BoyaSiparis.Fason = 1
    into :HarmanSipFirmaID;

    if (StrKarsilastir(HarmanSipFirmaID, GirisFirmaID) = 0) then
    begin
      Status = -4;
      suspend;
      exit;
    end
  end
  else if (AIplikIrsaliyeId is not null) then
  begin
    Select BoyaSiparis.FirmaId from Harman
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Harman.BoyaSiparisKayitID
      where Harman.HarmanKayitID = :AHarmanKayitId and BoyaSiparis.Fason = 1
      into :HarmanSipFirmaID;

    if (HarmanSipFirmaID is not Null) then
    begin
      Status = -5;
      suspend;
      exit;
    end
  end

  if (AIplikIrsaliyeId is not null) then
  if (CikisIplikIrsaliyeDetayId is not null) then
  begin
    Status = -2;
    suspend;
    exit;
  end

  if (AIplikIrsaliyeId is not null) then
  if (HamTipiId <> hstElyaf and hamTipiId <> hstTelef) then
  begin
    Status = -3;
    suspend;
    exit;
  end

 /*-----------------------*/
  if (HRDisinaCikilmasin = 1) then
  if (AIplikIrsaliyeId is not null) then                   /*HRDisinaCikilmasin icin eklendi*/
  begin
    Select count(*) from HarmanRecete
      where HarmanRecete.HarmanKayitId = :AHarmanKayitId and HarmanRecete.IplikIrsaliyeDetayId = :GirisIplikIrsaliyeDetayId
      into :KS;
    if (KS = 0) then
    begin
      Select IplikIrsaliyeDetay.Renkid, IplikIrsaliyeDetay.Lot  from Harman
        left join HarmanRecete on HarmanRecete.HarmanKayitId = Harman.HarmanKayitID
        left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = HarmanRecete.IplikIrsaliyeDetayId
        where Harman.HarmanKayitID = :AHarmanKayitId and (IplikIrsaliyeDetay.Lot = :Lot or :LOT is null)
        into :HRenkId, :HLot;
     /* Select Harman.Renkid, IplikIrsaliyeDetay.Lot  from Harman 14.02 Tolga ile beraber
        left join HarmanRecete on HarmanRecete.HarmanKayitId = Harman.HarmanKayitID
        left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = HarmanRecete.IplikIrsaliyeDetayId
        where Harman.HarmanKayitID = :AHarmanKayitId into :HRenkId, :HLot;*/

      if (StrKarsilastir(HLot, Lot) = 0) then
      begin
        Status = -7;
        suspend;
        exit;
      end

      if (IntKarsilastir(HRenkId, Renkid) = 0) then
      begin
        Status = -6;
        suspend;
        exit;
      end

    end
  end
 /*-----------------------*/

  if (AIplikIrsaliyeId is not null) then
  begin
    Select count(*) from HarmanRecete
      where HarmanRecete.HarmanKayitId = :AHarmanKayitId and HarmanRecete.IplikIrsaliyeDetayId = :GirisIplikIrsaliyeDetayId
      into :KS;

    if (KS > 0) then Status = 1;
    else Status = 2;



    Select Min(IplikIrsaliyeDetay.IplikIrsaliyeDetayId) from IplikIrsaliyeDetay
      left join IplikStok on IplikStok.CikisIplikIrsaliyeDetayId = IplikIrsaliyeDetay.IplikIrsaliyeDetayId
      where IplikIrsaliyeDetay.IplikIrsaliyeId = :AIplikIrsaliyeId and IplikIrsaliyeDetay.HarmanKayitId = :AHarmanKayitId and
            IplikIrsaliyeDetay.HamID = :HamId and IplikStok.IplikIrsaliyeDetayID = :GirisIplikIrsaliyeDetayID and
            IplikIrsaliyeDetay.IplikDepoId = :IplikDepoId
      into :IplikIrsaliyeDetayId;

    if (IplikIrsaliyeDetayId is null) then
    begin

      IplikIrsaliyeDetayId = gen_Id(IplikIrsaliyeDetay_Id_gen, 1);
      insert into IplikIrsaliyeDetay(IplikIrsaliyeId, IplikIrsaliyeDetayId, HareketTipi, IplikDepoId, HarmanKayitId, HamId, RenkId, Lot, KapAdedi, Miktar, NetMiktar, InsertKullaniciId, KullaniciId, BirimFiyat)
                  values(:AIplikIrsaliyeId, :IplikIrsaliyeDetayId, :HareketTipi, :IplikDepoId, :AHarmanKayitId, :HamId, :RenkId, :Lot, 1, :Miktar, :NetMiktar, :AKullaniciID, :AKullaniciID, :BirimFiyat);
    end
    else
    begin
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi + 1, Miktar = Miktar + :Miktar, NetMiktar = NetMiktar + :NetMiktar, NeredenUpdate = 1, KullaniciID = :AKullaniciID where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :IplikIrsaliyeDetayId;
    end

    Update IplikStok set CikisIplikIrsaliyeDetayId = :IplikIrsaliyeDetayId where IplikStok.IplikStokID = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;
    suspend;
  end
  else
  begin
    if (CikisIplikIrsaliyeDetayId is Null) then
    begin
      Status = -1;
      Suspend;
      exit;
    end

    Update IplikStok set CikisIplikIrsaliyeDetayId = Null where IplikStok.IplikStokID = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;

    Select count(*) from IplikStok where IplikStok.CikisIplikIrsaliyeDetayId = :CikisIplikIrsaliyeDetayId into :KS;

    if (KS = 0) then delete from IplikIrsaliyeDetay where IplikIrsaliyeDetay.IplikIrsaliyeDetayID = :CikisIplikIrsaliyeDetayId;
    else
    begin
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi - 1, Miktar = Miktar - :Miktar, NetMiktar = NetMiktar - :NetMiktar, KullaniciID = :AKullaniciId, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :CikisIplikIrsaliyeDetayId;
    end

    Status = 1;
    suspend;
  end
end
