CREATE PROCEDURE ONAYBEKLEYENVERILENSIPARISLER(
    AKULLANICIID INTEGER,
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    VERILENSIPARISKAYITID INTEGER,
    VERILENSIPARISYIL INTEGER,
    VERILENSIPARISID INTEGER,
    VERILENSIPARISIDSTR VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    VERILENSIPARISTIPIID INTEGER,
    VERILENSIPARISTIPIIDSTR VARCHAR(15),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP)
AS
declare variable VerilenSipteDetayOnayiKullan integer;
begin
  Select VerilenSipteDetayOnayiKullan from Sistem into :VerilenSipteDetayOnayiKullan;

  if (VerilenSipteDetayOnayiKullan = 0) then
  begin
    For Select VerilenSiparis.VerilenSiparisKayitId, VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId,
               VerilenSiparis.FirmaId, Firma.Adi,
               VerilenSiparis.SiparisTarihi,  VerilenSiparis.Notu, VerilenSiparis.VerilenSiparisTipiId,
               VerilenSiparisTipi.Adi, VerilenSiparisOnay.OnayDurumu, VerilenSiparisOnay.OnayTarihi from VerilenSiparisOnay
    left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisOnay.VerilenSiparisKayitId
    left join Firma On Firma.FirmaId = VerilenSiparis.FirmaId
    left join VerilenSiparisTipi on VerilenSiparisTipi.VerilenSiparisTipiId = VerilenSiparis.VerilenSiparisTipiId
    where VerilenSiparisOnay.OnaylayacakKullaniciID = :AKullaniciId and
          (VerilenSiparisOnay.OnayDurumu = :ARaporTipi) and (VerilenSiparis.OnayDurumu = :ARaporTipi) and
          (:ARaporTipi = 0 or (VerilenSiparisOnay.OnayTarihi >= :ABaslangicTarihi and VerilenSiparisOnay.OnayTarihi <= :ABitisTarihi))
    into :VerilenSiparisKayitId, :VerilenSiparisYil, :VerilenSiparisId, :FirmaId, :FirmaAdi, :SiparisTarihi,
         :Notu, :VerilenSiparisTipiId, :VerilenSiparisTipiIdStr, :OnayDurumu, :OnayTarihi do
    begin
      if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';

      VerilenSiparisIdStr = '';
      if (VerilenSiparisId is not null) then VerilenSiparisIdStr = '(' || EgemenIntegerToStr(VerilenSiparisYil) || ') '  || EgemenIntegerToStr(VerilenSiparisId);
      suspend;
    end
  end
  else
  begin
    For Select Distinct VerilenSiparis.VerilenSiparisKayitId, VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisId,
               VerilenSiparis.FirmaId, Firma.Adi,
               VerilenSiparis.SiparisTarihi,  VerilenSiparis.Notu, VerilenSiparis.VerilenSiparisTipiId,
               VerilenSiparisTipi.Adi, VerilenSiparisOnay.OnayDurumu, VerilenSiparisOnay.OnayTarihi from VerilenSiparisOnay
    left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitId = VerilenSiparisOnay.VerilenSiparisKayitId
    left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisKayitId = VerilenSiparis.VerilenSiparisKayitId
    left join Firma On Firma.FirmaId = VerilenSiparis.FirmaId
    left join VerilenSiparisTipi on VerilenSiparisTipi.VerilenSiparisTipiId = VerilenSiparis.VerilenSiparisTipiId
    where (VerilenSiparisDetay.OnayDurumu = :ARaporTipi) and
          (:ARaporTipi = 0 or (VerilenSiparisDetay.OnayTarihi >= :ABaslangicTarihi and VerilenSiparisDetay.OnayTarihi <= :ABitisTarihi))
    into :VerilenSiparisKayitId, :VerilenSiparisYil, :VerilenSiparisId, :FirmaId, :FirmaAdi, :SiparisTarihi,
         :Notu, :VerilenSiparisTipiId, :VerilenSiparisTipiIdStr, :OnayDurumu, :OnayTarihi do
    begin
      if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';

      VerilenSiparisIdStr = '';
      if (VerilenSiparisId is not null) then VerilenSiparisIdStr = '(' || EgemenIntegerToStr(VerilenSiparisYil) || ') '  || EgemenIntegerToStr(VerilenSiparisId);
      suspend;
    end
  end
end
