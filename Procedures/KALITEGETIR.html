CREATE PROCEDURE KALITEGETIR(
    AKALITEID VARCHAR(20),
    AKULLANICIID INTEGER = Null)
RETURNS (
    KALITEADI ADI255,
    COMPOSITION VARCHAR(255),
    RESIMDOSYAADI1 VARCHAR(255),
    RESIMDOSYAADI2 VARCHAR(255),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID DOVIZSIZE,
    MINSIPARISMIKTARI DOUBLE PRECISION,
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    NOTU BLOB SUB_TYPE 1 SEGMENT SIZE 80,
    KAPALI EGEBOOLEAN)
AS
declare variable FBoyaTeklifKayitID  integer;
declare variable SirketDisiKullanici EgeBoolean;
declare variable EgemenFirmaID       Integer;
begin
  Select EgemenFirmaID from Sistem where SistemID = 1 into :EgemenFirmaId;
  if (EgemenFirmaId = 0) then /*NY*/
    Select first 1 BoyaTeklifKayitID, DovizID from BoyaTeklif where FirmaID = (Select Sistem.FirmaKodumuz from Sistem where Sistem.SistemId = 1) Order By TeklifTarihi Desc, BoyaTeklifKayitID desc into :FBoyaTeklifKayitID, :DovizID;

  if (AKullaniciID is not Null) then
    Select SirketDisiKullanici from Kullanici where KullaniciId = :AKullaniciID into :SirketDisiKullanici;
  else SirketDisiKullanici = 0;


  if (FBoyaTeklifKayitID is not Null) then /*NY*/
  begin
    Select First 1 Kalite.KaliteAdi, Kalite.ResimDosyaAdi1, Max(TeklifEdilenFiyat) BirimFiyat, Kalite.ResimDosyaAdi2, Kalite.Composition,
           Kalite.MinSiparisMiktari, Kalite.EnIstenen, Kalite.GramajIstenen, Kalite.Notu, Kalite.Kapali
      FROM Kalite
      Left Join BoyaTeklifDetay ON BoyaTeklifDetay.BoyaTeklifKayitID = :FBoyaTeklifKayitID and BoyaTeklifDetay.KaliteID = Kalite.KaliteID
      Where Kalite.KaliteID = :AKaliteID
      and Kalite.Kapali = 0
      Group By Kalite.KaliteAdi, Kalite.ResimDosyaAdi1, Kalite.ResimDosyaAdi2, Kalite.Composition, Kalite.MinSiparisMiktari, Kalite.EnIstenen, Kalite.GramajIstenen, Kalite.Notu, Kalite.Kapali
    into :KaliteAdi, :ResimDosyaAdi1, :BirimFiyat, :ResimDosyaAdi2, :Composition, :MinSiparisMiktari, :EnIstenen, :GramajIstenen, :Notu, :Kapali;
  end
  else
  begin
    Select Kalite.KaliteAdi, Kalite.ResimDosyaAdi1, Kalite.BirimFiyatMt, Kalite.DovizId, Kalite.ResimDosyaAdi2, Kalite.Composition,
           Kalite.MinSiparisMiktari, Kalite.EnIstenen, Kalite.GramajIstenen, Kalite.Notu, Kalite.Kapali
    FROM   Kalite
    Where  Kalite.KaliteID = :AKaliteID
      and Kalite.Kapali = 0
    into :KaliteAdi, :ResimDosyaAdi1, :BirimFiyat, :DovizId, :ResimDosyaAdi2, :Composition, :MinSiparisMiktari, :EnIstenen, :GramajIstenen, :Notu, :Kapali;
  end

  if (DovizID is Null) then Select SistemDovizID from Sistem where Sistem.SistemId = 1 into :DovizID;

  if (SirketDisiKullanici = 1) then Notu = Null;
  if (KaliteAdi is not Null) then Suspend;
end
