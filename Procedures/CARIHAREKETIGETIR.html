CREATE PROCEDURE CARIHAREKETIGETIR(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1),
    ADOVIZID VARCHAR(5))
RETURNS (
    CARIID INTEGER,
    FIRMAID VARCHAR(20),
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    CARITIPIADI VARCHAR(20),
    BORCALACAK VARCHAR(1),
    SINIF VARCHAR(1),
    MEBLAG DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    DOVIZKURUSTR VARCHAR(40),
    VALORTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    CEKSENETNO VARCHAR(30),
    HESAPNO VARCHAR(40),
    ILKBORCLU VARCHAR(80),
    KESIDEYERI VARCHAR(20),
    AITOLDUGUBANKAID INTEGER,
    BANKAADI VARCHAR(40),
    BANKAHESAPID VARCHAR(20),
    KARSILIKSIZ EGEBOOLEAN,
    KARSILIKSIZTARIHI TIMESTAMP,
    TAHSILEDILENMEBLAG DOUBLE PRECISION,
    TAHSILTARIHI TIMESTAMP,
    CEKSENETID INTEGER,
    VADETARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSYERI INTEGER,
    REFERANSID INTEGER,
    BORC DOUBLE PRECISION,
    ALACAK DOUBLE PRECISION,
    BORCBAKIYE DOUBLE PRECISION,
    ALACAKBAKIYE DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    KDVHARICBIRIMFIYAT DOUBLE PRECISION,
    KAPADEDI INTEGER,
    RENKADI VARCHAR(100),
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    FATURANO FATURANOSIZE,
    FASONSTR VARCHAR(6),
    FASON INTEGER,
    RENKDERINLIGI VARCHAR(20),
    PARTIIDSTR VARCHAR(30),
    HAREKETDOVIZID VARCHAR(5),
    HAREKETMEBLAG DOUBLE PRECISION,
    URETIMPARTINO VARCHAR(30),
    VERILENSIPBIRIMFIYAT DOUBLE PRECISION,
    VERILENSIPDOVIZID VARCHAR(5),
    KASAHARCAMATIPIID KASAHARCAMATIPISIZE,
    KASAHARCAMATIPIADI ADI30,
    CARIHAREKETTIPIID BANKACARIHAREKETTIPISIZE,
    CARIHAREKETTIPIADI ADI30,
    TAHSILEDILECEKKDV DOUBLE PRECISION)
AS
declare Variable ToplamBorc double precision;
declare variable ToplamAlacak double precision;
declare variable CSGReferansId integer;
declare variable CSHReferansId integer;
declare variable CSGBReferansId integer;
declare variable CSHBReferansId integer;
declare variable StReferansId integer;
declare variable UniteReferansId integer;
declare variable OrmeStReferansId integer;
declare variable IplikStReferansId integer;
declare variable BcReferansId integer;
declare variable KsReferansId integer;
declare variable BfReferansId integer;
declare variable HSReferansId integer;
declare variable CariVirmanId integer;

declare variable DovizKuru1 Double Precision;
declare variable DovizKuru2 Double Precision;
declare variable FirmaDovizId varchar(5);
declare variable FBakiye Double Precision;
declare variable FBorcAlacak varchar(1);
declare variable BakiyeBA varchar(1);
begin
  ToplamBorc = 0;
  ToplamAlacak = 0;

  For select BorcAlacak, Sum(iif(:ADovizId is Null, Meblag / dovizkuru1 * DovizKuru2, Meblag))  from cari
    where FirmaId = :AFirmaId and tarih < :ABaslangicTarihi and  (Cari.Sinif = :ASinif or :ASinif is null) and (Cari.DovizID = :ADovizId or :ADovizId is Null)
    group by BorcAlacak
    into :FBorcAlacak, :FBakiye do
  begin
    if (FBorcAlacak = 'B') then ToplamBorc = FBakiye;
    if (FBorcAlacak = 'A') then ToplamAlacak = FBakiye;
  end

  CariId = -1;
  ReferansYeri = -1;
  ReferansId = -1;
  FirmaId = AFirmaId;
  Tarih  = null;
  Aciklama = 'Devir';
  CaritipiAdi = null;
  BorcAlacak = null;
  Sinif  = Null;
  Meblag = null;
  DovizId = null;
  DovizKuruStr = null;
  ValorTarihi = null;
  Notu = null;
  KullaniciAdi = null;
  DegisimTarihi = null;
  Borc = null;
  Alacak = Null;
  BorcBakiye = null;
  AlacakBakiye = Null;
  FaturaNo = Null;
  HareketDovizID  = Null;
  HareketMeblag  = Null;

  if (ToplamBorc > 0) then Borc = ToplamBorc;
  if (ToplamAlacak > 0) then Alacak = ToplamAlacak;

  if (toplamBorc is Null) then ToplamBorc = 0;
  if (ToplamAlacak is Null) then ToplamAlacak = 0;

  if (ToplamBorc > ToplamAlacak) then
  begin
    BorcBakiye = ToplamBorc - ToplamAlacak;
    BakiyeBA = 'B';
  end
  else
  begin
    AlacakBakiye = ToplamAlacak - ToplamBorc;
    BakiyeBA = 'A';
  end

  if (BorcBakiye = 0) then BorcBakiye = null;
  if (AlacakBakiye = 0) then AlacakBakiye = null;

  suspend;


  Select DovizId from Firma where Firma.FirmaId = :AFirmaId into :FirmaDovizId;

  for select CariId,  Cari.FirmaId, Cari.Tarih, Cari.Aciklama,BorcAlacak, Cari.Sinif, Meblag,DovizId, DovizKuru1, DovizKuru2,ValorTarihi, Notu, DegisimTarihi, CariTipi.Adi, Kullanici.Adi,CSGReferansId,CSHReferansId, CSGBReferansId, CSHBReferansId, STReferansId, OrmeStReferansId, IplikStReferansId, BcReferansId, KSReferansId, BfReferansId, HsReferansId, CariVirmanId, UniteReferansId from Cari
  left join CariTipi on CariTipi.CariTipiId = Cari.CariTipiId
  left join Kullanici on Kullanici.KullaniciId = Cari.KullaniciId
  where Cari.FirmaId = :AFirmaId and Cari.Tarih >= :ABaslangicTarihi and Cari.Tarih <= :ABitisTarihi and (Cari.Sinif = :ASinif or :ASinif is null) and (Cari.DovizID = :ADovizId or :ADovizId is Null)
  order by Cari.Tarih, Cari.CariId
  into :CariId, :FirmaId, :Tarih, :Aciklama, :BorcAlacak, :Sinif, :Meblag, :DovizId, :DovizKuru1, :DovizKuru2, :ValorTarihi, :Notu, :DegisimTarihi, :CariTipiAdi, :KullaniciAdi, :CSGReferansId, :CSHReferansId, :CSGBReferansId, :CSHBReferansId, :StReferansId, :OrmeStReferansId, :IplikStReferansId, :BcReferansId, :KsReferansId, :BfReferansId, :HsReferansId, :CariVirmanId, :UniteReferansId do
  begin
    Borc = Null;
    Alacak = Null;
    ReferansYeri = Null;
    HareketDovizID  = Null;
    HareketMeblag  = Null;
    KasaHarcamaTipiId  = null;
    KasaHarcamaTipiAdi = null;
    CariHareketTipiId  = null;
    CariHareketTipiAdi = null;
    TahsilEdilecekKdv  = null;

    if (CSGReferansId is not null) then
    begin
      ReferansYeri = 1;
      ReferansId = CSGReferansId;
      Select DovizID, Meblag from CekSenetDetay where GenelKayitID = :ReferansId into :HareketDovizID, :HareketMeblag;
    end
    else if (CSHReferansId is not null) then
    begin
      ReferansYeri = 2;
      ReferansId = CSHReferansId;
      Select CekSenetDetay.DovizID, CekSenetDetay.Meblag from CekSenetHareketDetay
        Left Join CekSenetDetay on CekSenetDetay.GenelKayitID = CekSenetHareketDetay.GenelKayitID
      where CSHareketDetayID = :ReferansId into :HareketDovizID, :HareketMeblag;
    end
    else if (StReferansId is not null) then
    begin
      ReferansYeri = 3;
      ReferansId = StReferansId;
    end
    else if (KsReferansId is not null) then
    begin
      ReferansYeri = 4;
      ReferansId = KsReferansId;
      Select Kasa.DovizID, KasaHareket.Meblag, KasaHareket.KasaHarcamaTipiId, KasaHarcamaTipi.Adi from KasaHareket
        Left Join Kasa on Kasa.KasaID = KasaHareket.KasaID
        Left Join KasaHarcamaTipi on KasaHareket.KasaHarcamaTipiId = KasaHarcamaTipi.KasaHarcamaTipiId
      where KasaHareketID = :ReferansId into :HareketDovizID, :HareketMeblag, :KasaHarcamaTipiId, :KasaHarcamaTipiAdi;
    end
    else if (BfReferansId is not null) then
    begin
      ReferansYeri = 5;
      ReferansId = BfReferansId;
    end
    else if (BcReferansId is not null) then
    begin
      ReferansYeri = 6;
      ReferansId = BcReferansId;
      Select BankaHesap.DovizID, BankaHesapCari.Meblag, BankaHesapCari.CariHareketTipiId, BankaCariHareketTipi.Adi from BankaHesapCari
        Left Join BankaHesap on BankaHesap.BankaHesapID = BankaHesapCari.BankaHesapID
        Left Join BankaCariHareketTipi On BankaCariHareketTipi.BankaCariHareketTipiId = BankaHesapCari.CariHareketTipiId
      where BankaHesapCariID = :ReferansId into :HareketDovizID, :HareketMeblag, :CariHareketTipiId, :CariHareketTipiAdi;
    end
    else if (CSGBReferansId is not null) then
    begin
      ReferansYeri = 7;
      ReferansId = CSGBReferansId;
    end
    else if (CSHBReferansId is not null) then
    begin
      ReferansYeri = 8;
      ReferansId = CSHBReferansId;
    end
    else if (HSReferansId is not null) then
    begin
      ReferansYeri = 9;
      ReferansId = HSReferansId;
    end
    else if (OrmeStReferansId is not null) then
    begin
      ReferansYeri = 10;
      ReferansId = OrmeStReferansId;
    end
    else if (CariVirmanId is not null) then
    begin
      ReferansYeri = 11;
      ReferansId = CariVirmanId;
    end
    else if (IplikStReferansId is not null) then
    begin
      ReferansYeri = 12;
      ReferansId = IplikStReferansId;
    end
    else if (UniteReferansId is not null) then
    begin
      ReferansYeri = 13;
      ReferansId = UniteReferansId;
    end

    if (BorcAlacak = 'B') then
    begin
      if (ADovizID is Null) then Borc = Meblag / DovizKuru1 * DovizKuru2;
      else Borc = Meblag;
      ToplamBorc = ToplamBorc + Borc;
    end
    else
    begin
      if (ADovizID is Null) then Alacak = Meblag / DovizKuru1 * DovizKuru2;
      else Alacak = Meblag;
      ToplamAlacak = ToplamAlacak + Alacak;
    end

    if (toplamBorc is Null) then ToplamBorc = 0;
    if (ToplamAlacak is Null) then ToplamAlacak = 0;

    if (ToplamBorc > ToplamAlacak) then
    begin
      BorcBakiye = ToplamBorc - ToplamAlacak;
      AlacakBakiye = Null;
      BakiyeBA = 'B';
    end
    else
    begin
      AlacakBakiye = ToplamAlacak - ToplamBorc;
      BorcBakiye = Null;
      BakiyeBA = 'A';
    end

    if (BorcBakiye = 0) then BorcBakiye = null;
    if (AlacakBakiye = 0) then AlacakBakiye = null;

    if (FirmaDovizId <> DovizId) then
    begin
      DovizKuruStr = Format_Float('', DovizKuru1) || ' ' || DovizId || '-' || Format_Float('', DovizKuru2) || ' ' || FirmaDovizId;
    end
    else
      DovizKuruStr = '';

    suspend;
  end
end
