    SKUCRETLITAMIRMETREORAN DOUBLE PRECISION,
    SKUCRETSIZTAMIRMETREORAN DOUBLE PRECISION,
    SKISLETMEICITAMIRMETREORAN DOUBLE PRECISION,
    SKGENELMETRETOPLAM DOUBLE PRECISION,
    SKSEVKEHAZIRLARINMETRETOPLAMI DOUBLE PRECISION)
AS
declare variable Fason          integer;
declare variable PartiTipiId    integer;
declare variable IslemTipi      integer;
declare variable Bitti          integer;
declare variable FaturaKesilecek integer;
declare variable Kilo           Numeric(12,3);
declare variable Metre          Numeric(12,3);
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable itDesenTasarim integer;
begin
  Select itNormal, itRpt, itKontrol, itTamir, itNumune, itDesenTasarim from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune, :itDesenTasarim;
  FINormalKiloToplam = 0;
  FIUcretliTamirKiloToplam = 0;
  FIUcretsizTamirKiloToplam = 0;
  FIIsletmeIciTamirKiloToplam = 0;
  FIGenelKiloToplam = 0;
  FISevkeHazirlarinKiloToplami = 0;

  FINormalMetreToplam           = 0;
  FIUcretliTamirMetreToplam     = 0;
  FIUcretsizTamirMetreToplam    = 0;
  FIIsletmeIciTamirMetreToplam  = 0;
  FINormalMetreOran             = 0;
  FIUcretliTamirMetreOran       = 0;
  FIUcretsizTamirMetreOran      = 0;
  FIIsletmeIciTamirMetreOran    = 0;
  FIGenelMetreToplam            = 0;
  FISevkeHazirlarinMetreToplami = 0;


  FKNormalKiloToplam = 0;
  FKUcretliTamirKiloToplam  = 0;
  FKUcretsizTamirKiloToplam = 0;
  FKIsletmeIciTamirKiloToplam = 0;
  FKGenelKiloToplam = 0;
  FKSevkeHazirlarinKiloToplami = 0;

  FKNormalMetreToplam          = 0;
  FKUcretliTamirMetreToplam    = 0;
  FKUcretsizTamirMetreToplam   = 0;
  FKIsletmeIciTamirMetreToplam = 0;
  FKNormalMetreOran            = 0;
  FKUcretliTamirMetreOran      = 0;
  FKUcretsizTamirMetreOran     = 0;
  FKIsletmeIciTamirMetreOran   = 0;
  FKGenelMetreToplam           = 0;
  FKSevkeHazirlarinMetreToplami  = 0;


  SINormalKiloToplam = 0;
  SIUcretliTamirKiloToplam = 0;
  SIUcretsizTamirKiloToplam = 0;
  SIIsletmeIciTamirKiloToplam = 0;
  SIGenelKiloToplam = 0;
  SISevkeHazirlarinKiloToplami = 0;

  SINormalMetreToplam          = 0;
  SIUcretliTamirMetreToplam    = 0;
  SIUcretsizTamirMetreToplam   = 0;
  SIIsletmeIciTamirMetreToplam = 0;
  SINormalMetreOran            = 0;
  SIUcretliTamirMetreOran      = 0;
  SIUcretsizTamirMetreOran     = 0;
  SIIsletmeIciTamirMetreOran   = 0;
  SIGenelMetreToplam              = 0;
  SISevkeHazirlarinMetreToplami   = 0;


  SKNormalKiloToplam = 0;
  SKUcretliTamirKiloToplam = 0;
  SKUcretsizTamirKiloToplam = 0;
  SKIsletmeIciTamirKiloToplam = 0;
  SKGenelKiloToplam = 0;
  SKSevkeHazirlarinKiloToplami = 0;

  SKNormalMetreToplam           = 0;
  SKUcretliTamirMetreToplam     = 0;
  SKUcretsizTamirMetreToplam    = 0;
  SKIsletmeIciTamirMetreToplam  = 0;
  SKNormalMetreOran             = 0;
  SKUcretliTamirMetreOran       = 0;
  SKUcretsizTamirMetreOran      = 0;
  SKIsletmeIciTamirMetreOran    = 0;
  SKGenelMetreToplam            = 0;
  SkSevkeHazirlarinMetreToplami = 0;

  for select Parti.Fason, Parti.PartiTipiId, Parti.IslemTipiId, Parti.FaturaKesilecek, PartiTamirSinif.Bitti, sum(PartiStok.Kilo), sum(PartiStok.Metre) from PartiStok
    left join PartiTamirSinif on PartiTamirSinif.PartiSinifId = PartiStok.PartiSinifId and PartiTamirSinif.Bitti = 0
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    where durumu < 7 and PartiStok.PartiStokIptalKayitId is null and
          Parti.IslemTipiID <> :itDesenTasarim
    group by Parti.Fason, Parti.PartiTipiId, Parti.IslemTipiId, Parti.FaturaKesilecek, PartiTamirSinif.Bitti
    into :Fason, :PartiTipiId, :IslemTipi, :FaturaKesilecek, :Bitti, :Kilo, :Metre do
  begin
    if (PartiTipiID = 0) then
    begin
      if (Fason = 1) then
      begin
        if (IslemTipi = itTamir) then
        begin
          if (FaturaKesilecek = 1) then FIUcretliTamirKiloToplam = FIUcretliTamirKiloToplam + Kilo;
          else FIUcretsizTamirKiloToplam = FIUcretsizTamirKiloToplam + Kilo;

          if (FaturaKesilecek = 1) then FIUcretliTamirMetreToplam = FIUcretliTamirMetreToplam + Metre;
          else FIUcretsizTamirMetreToplam = FIUcretsizTamirMetreToplam + Metre;
        end
        else if (Bitti is null) then
        begin
          FINormalKiloToplam = FINormalKiloToplam + Kilo;
          FINormalMetreToplam = FINormalMetreToplam + Metre;
        end
        else
        begin
          FIIsletmeIciTamirKiloToplam = FIIsletmeIciTamirKiloToplam + Kilo;
          FIIsletmeIciTamirMetreToplam = FIIsletmeIciTamirMetreToplam + Metre;
        end

        FIGenelKiloToplam = FIGenelKiloToplam + Kilo;
        FIGenelMetreToplam = FIGenelMetreToplam + Metre;
      end
      else
      begin
        if (IslemTipi = itTamir) then
        begin
          if (FaturaKesilecek = 1) then SIUcretliTamirKiloToplam = SIUcretliTamirKiloToplam + Kilo;
          else SIUcretsizTamirKiloToplam = SIUcretsizTamirKiloToplam + Kilo;

          if (FaturaKesilecek = 1) then SIUcretliTamirMetreToplam = SIUcretliTamirMetreToplam + Metre;
          else SIUcretsizTamirMetreToplam = SIUcretsizTamirMetreToplam + Metre;
        end
        else if (Bitti is null) then
        begin
          SINormalKiloToplam = SINormalKiloToplam + Kilo;
          SINormalMetreToplam = SINormalMetreToplam + Metre;
        end
        else
        begin
          SIIsletmeIciTamirKiloToplam = SIIsletmeIciTamirKiloToplam + Kilo;
          SIIsletmeIciTamirMetreToplam = SIIsletmeIciTamirMetreToplam + Metre;
        end

        SIGenelKiloToplam = SIGenelKiloToplam + Kilo;
        SIGenelMetreToplam = SIGenelMetreToplam + Metre;
      end
    end
    else
    begin
      if (Fason = 1) then
      begin
        if (IslemTipi = itTamir) then
        begin
          if (FaturaKesilecek = 1) then FKUcretliTamirKiloToplam = FKUcretliTamirKiloToplam + Kilo;
          else FKUcretsizTamirKiloToplam = FKUcretsizTamirKiloToplam + Kilo;

          if (FaturaKesilecek = 1) then FKUcretliTamirMetreToplam = FKUcretliTamirMetreToplam + Metre;
          else FKUcretsizTamirMetreToplam = FKUcretsizTamirMetreToplam + Metre;
        end
        else if (Bitti is null) then
        begin
          FKNormalKiloToplam = FKNormalKiloToplam + Kilo;
          FKNormalMetreToplam = FKNormalMetreToplam + Metre;
        end
        else
        begin
          FKIsletmeIciTamirKiloToplam = FKIsletmeIciTamirKiloToplam + Kilo;
          FKIsletmeIciTamirMetreToplam = FKIsletmeIciTamirMetreToplam + Metre;
        end

        FKGenelKiloToplam = FKGenelKiloToplam + Kilo;
        FKGenelMetreToplam = FKGenelMetreToplam + Metre;
      end
      else
      begin
        if (IslemTipi = itTamir) then
        begin
          if (FaturaKesilecek = 1) then SKUcretliTamirKiloToplam = SKUcretliTamirKiloToplam + Kilo;
          else SKUcretsizTamirKiloToplam = SKUcretsizTamirKiloToplam + Kilo;

          if (FaturaKesilecek = 1) then SKUcretliTamirMetreToplam = SKUcretliTamirMetreToplam + Metre;
          else SKUcretsizTamirMetreToplam = SKUcretsizTamirMetreToplam + Metre;
        end
        else if (Bitti is null) then
        begin
          SKNormalKiloToplam = SKNormalKiloToplam + Kilo;
          SKNormalMetreToplam = SKNormalMetreToplam + Metre;
        end
        else
        begin
          SKIsletmeIciTamirKiloToplam = SKIsletmeIciTamirKiloToplam + Kilo;
          SKIsletmeIciTamirMetreToplam = SKIsletmeIciTamirMetreToplam + Metre;
        end

        SKGenelKiloToplam = SKGenelKiloToplam + Kilo;
        SKGenelMetreToplam = SKGenelMetreToplam + Metre;
      end
    end
  end
  Kilo = 0;

  for select Parti.Fason, Parti.PartiTipiId, sum(PartiStok.Kilo-PartiStok.BrutSevkKilo), sum(PartiStok.Metre - PartiStok.BrutSevkMetre) from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    where PartiStok.Durumu = 7 and PartiStok.PartiStokIptalKayitId is null and
          Parti.IslemTipiID <> :itDesenTasarim
    group By Parti.Fason, Parti.PartiTipiId
    into :Fason, :PartiTipiId, :Kilo, :Metre do
  begin
    if (PartiTipiID = 0) then
    begin
      if (Fason = 1) then FISevkeHazirlarinKiloToplami = Kilo;
      else SISevkeHazirlarinKiloToplami = Kilo;

      if (Fason = 1) then FISevkeHazirlarinMetreToplami = Metre;
      else SISevkeHazirlarinMetreToplami = Metre;
    end
    else
    begin
      if (Fason = 1) then FKSevkeHazirlarinKiloToplami = Kilo;
      else SKSevkeHazirlarinKiloToplami = Kilo;

      if (Fason = 1) then FKSevkeHazirlarinMetreToplami = Metre;
      else SKSevkeHazirlarinMetreToplami = Metre;
    end
  end

  if (FIGenelKiloToplam <> 0) then
  begin
    FINormalOran  = FINormalKiloToplam * 100 / FIGenelKiloToplam;
    FIUcretliTamirOran = FIUcretliTamirKiloToplam * 100 / FIGenelKiloToplam;
    FIUcretsizTamirOran = FIUcretsizTamirKiloToplam * 100 / FIGenelKiloToplam;
    FIIsletmeIciTamirOran = FIIsletmeIciTamirKiloToplam * 100 / FIGenelKiloToplam;
  end

  if (FIGenelMetreToplam <> 0) then
  begin
    FINormalMetreOran  = FINormalMetreToplam * 100 / FIGenelMetreToplam;
    FIUcretliTamirMetreOran = FIUcretliTamirMetreToplam * 100 / FIGenelMetreToplam;
    FIUcretsizTamirMetreOran = FIUcretsizTamirMetreToplam * 100 / FIGenelMetreToplam;
    FIIsletmeIciTamirMetreOran = FIIsletmeIciTamirMetreToplam * 100 / FIGenelMetreToplam;
  end

  if (FKGenelKiloToplam <> 0) then
  begin
    FKNormalOran  = FKNormalKiloToplam * 100 / FKGenelKiloToplam;
    FKUcretliTamirOran = FKUcretliTamirKiloToplam * 100 / FKGenelKiloToplam;
    FKUcretsizTamirOran = FKUcretsizTamirKiloToplam * 100 / FKGenelKiloToplam;
    FKIsletmeIciTamirOran = FKIsletmeIciTamirKiloToplam * 100 / FKGenelKiloToplam;
  end

  if (FKGenelMetreToplam <> 0) then
  begin
    FKNormalMetreOran  = FKNormalMetreToplam * 100 / FKGenelMetreToplam;
    FKUcretliTamirMetreOran = FKUcretliTamirMetreToplam * 100 / FKGenelMetreToplam;
    FKUcretsizTamirMetreOran = FKUcretsizTamirMetreToplam * 100 / FKGenelMetreToplam;
    FKIsletmeIciTamirMetreOran = FKIsletmeIciTamirMetreToplam * 100 / FKGenelMetreToplam;
  end

  if (SIGenelKiloToplam <> 0) then
  begin
    SINormalOran  = SINormalKiloToplam * 100 / SIGenelKiloToplam;
    SIUcretliTamirOran = SIUcretliTamirKiloToplam * 100 / SIGenelKiloToplam;
    SIUcretsizTamirOran = SIUcretsizTamirKiloToplam * 100 / SIGenelKiloToplam;
    SIIsletmeIciTamirOran = SIIsletmeIciTamirKiloToplam * 100 / SIGenelKiloToplam;
  end

  if (SIGenelMetreToplam <> 0) then
  begin
    SINormalMetreOran  = SINormalMetreToplam * 100 / SIGenelMetreToplam;
    SIUcretliTamirMetreOran = SIUcretliTamirMetreToplam * 100 / SIGenelMetreToplam;
    SIUcretsizTamirMetreOran = SIUcretsizTamirMetreToplam * 100 / SIGenelMetreToplam;
    SIIsletmeIciTamirMetreOran = SIIsletmeIciTamirMetreToplam * 100 / SIGenelMetreToplam;
  end

  if (SKGenelKiloToplam <> 0) then
  begin
    SKNormalOran  = SKNormalKiloToplam * 100 / SKGenelKiloToplam;
    SKUcretliTamirOran = SKUcretliTamirKiloToplam * 100 / SKGenelKiloToplam;
    SKUcretsizTamirOran = SKUcretsizTamirKiloToplam * 100 / SKGenelKiloToplam;
    SKIsletmeIciTamirOran = SKIsletmeIciTamirKiloToplam * 100 / SKGenelKiloToplam;
  end

  if (SKGenelMetreToplam <> 0) then
  begin
    SKNormalMetreOran  = SKNormalMetreToplam * 100 / SKGenelMetreToplam;
    SKUcretliTamirMetreOran = SKUcretliTamirMetreToplam * 100 / SKGenelMetreToplam;
    SKUcretsizTamirMetreOran = SKUcretsizTamirMetreToplam * 100 / SKGenelMetreToplam;
    SKIsletmeIciTamirMetreOran = SKIsletmeIciTamirMetreToplam * 100 / SKGenelMetreToplam;
  end

  suspend;
