CREATE PROCEDURE URUNFIFOMALIYETHAZIRLA(
    AURUNID VARCHAR(20),
    ADUBCARPAN DOUBLE PRECISION,
    ABITISTARIHI TIMESTAMP,
    ARAPORDOVIZID VARCHAR(5))
RETURNS (
    TARIH TIMESTAMP,
    NETBIRIMFIYAT DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    MALFAZLASI INTEGER)
AS
declare variable shtAlis        integer;
declare variable shtSatisIade   integer;
declare variable shtUretim   integer;
declare variable FNetBirimFiyat Double Precision;
declare variable FAltIskontoCarpani Double Precision;
declare variable HUBCarpan          Double Precision;
declare variable FStokTabanDovizId  varchar(5);
declare variable ARaporTarihi       Date;
declare variable DovizKuru          Double Precision;
declare variable FTabanDovizKuru    Double Precision;
declare variable StokFifoMaliyetBaslangicTarihi TimeStamp;
begin
  select shtAlis, ShtSatisIade, shtUretim from StokHareketTipiGetir into :shtAlis, :ShtSatisIade, :shtUretim;
  select Sistem.StokFifoMaliyetBaslangicTarihi from sistem where sistem.sistemid = 1 into :StokFifoMaliyetBaslangicTarihi ;

  ARaporTarihi = current_TimeStamp(0);
  select STOKTABANDOVIZID from SISTEM where SISTEMID = 1 into :FStokTabanDovizId;

  if (FStokTabanDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FStokTabanDovizId, :ARaporDovizId, :ARaporTarihi) into :DovizKuru;
  else DovizKuru = 1;

  For Select StokDetay.Tarih, StokDetay.Miktar, StokDetay.NetBirimFiyat, StokDetay.TabanDovizKuru2 / StokDetay.TabanDovizKuru1,
             Fatura.AltIskontoCarpani, UrunBirim.Carpan from stokdetay
    left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    where Stokdetay.Urunid = :AUrunId and ( StokDetay.Tarih >= :StokFifoMaliyetBaslangicTarihi or :StokFifoMaliyetBaslangicTarihi is null ) and StokDetay.Tarih <= :ABitisTarihi and
         (Stokdetay.HareketTipi = :shtAlis or Stokdetay.HareketTipi = :ShtSatisIade or Stokdetay.HareketTipi = :shtUretim)
    into :Tarih, :Miktar, :FNetBirimFiyat, :FTabanDovizKuru, :FAltIskontoCarpani,  :HUBCarpan do
  begin
    if (FTabanDovizKuru is null) then FTabanDovizKuru = 1;
    if (FAltIskontoCarpani is null) then FAltIskontoCarpani = 0;

    FNetBirimFiyat =  FNetBirimFiyat * FTabanDovizKuru;

    if (HUBCarpan <> 0) then Miktar = Miktar / HUBCarpan * ADUBCarpan;
    else Miktar = 0;

    if (ADUBCarpan <> 0) then
      FNetBirimFiyat = FNetBirimFiyat * HUBCarpan / ADUBCarpan;
    else
      FNetBirimFiyat = 0;

    NetBirimFiyat = FNetBirimFiyat - FNetBirimFiyat * FAltIskontoCarpani;

    NetBirimFiyat = NetBirimFiyat * DovizKuru;
    suspend;
  end
end
