CREATE PROCEDURE IPLIKMAKINASAYACKAYDET2(
    AIPLIKMAKINAID VARCHAR(10),
    AIPLIKURETIMISEMRIKAYITID INTEGER,
    ASAYAC DOUBLE PRECISION,
    AKAYIT INTEGER)
RETURNS (
    IPLIKURETIMISEMRIKAYITID INTEGER,
    IPLIKURETIMISEMRIYIL INTEGER,
    IPLIKURETIMISEMRIID INTEGER,
    LOT VARCHAR(30),
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    BUKUMTIPIADI VARCHAR(10),
    HARMANADI VARCHAR(50),
    HARMANCINSIADI VARCHAR(50),
    KALINLIKNE INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    IPLIKCINSIURETIMPARTINOLABEL VARCHAR(10),
    BUKUMSAYISI INTEGER,
    COPSTIPIADI VARCHAR(20),
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    BASLAMATARIHI TIMESTAMP,
    TAHMINIBITISTARIHI TIMESTAMP,
    SAYACISIFIRLA INTEGER,
    ONCEKISAYAC DOUBLE PRECISION)
AS
declare variable IplikV1BaslangicSaati Double Precision;
declare variable IplikV2BaslangicSaati Double Precision;
declare variable IplikV3BaslangicSaati Double Precision;
declare variable SuAnkiSaat            Double Precision;
declare variable VardiyaTarihi             Date;
declare variable VardiyaNo                 integer;

declare variable FIplikMakinaSayacId        integer;
declare variable FVardiyaTarihi             Date;
declare variable FVardiyaNo                 integer;
declare variable FIplikUretimIsEmriKayitId  integer;
declare variable EksikIgSayisi  integer;
declare variable FEksikIgSayisi  integer;
begin
  Select VardiyaTarihi, VardiyaNo from IplikVardiyaHesapla(Current_TimeStamp(0)) into :VardiyaTarihi, :VardiyaNo;
  Select EksikIgSayisi from IplikMakina where IplikMakina.IplikMakinaID = :AIplikMakinaID into :EksikIgSayisi;

  if (AIplikUretimIsEmriKayitID is not null) then
  begin
    Select IplikUretimIsEmri.IplikUretimIsEmriKayitId, IplikUretimIsEmri.IplikUretimIsEmriYil, IplikUretimIsEmri.IplikUretimIsEmriId,
          IplikLot.Lot, IplikLot.KalinlikNe, IplikCinsi.Adi, IplikCinsi.UretimPartiNoLabel, IplikUretimIsEmri.PlanlananMiktar, IplikUretimIsEmri.GerceklesenMiktar,
          Harman.HarmanYil, Harman.HarmanId, Harman.Adi, HarmanCinsi.Adi, IplikUretimIsEmri.BaslamaTarihi, BukumTipi.Adi, IplikLot.BukumSayisi,
          CopsTipi.Adi from IplikUretimIsEmri
      left join IplikLot on IplikLot.IplikLotID = IplikUretimIsEmri.IplikLotID
      left join IplikCinsi on IplikCinsi.IplikCinsiID = IplikLot.IplikCinsiID
      left join BukumTipi on BukumTipi.BukumTipiID = IplikLot.BukumTipiID
      left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitID
      left join HarmanCinsi on HarmanCinsi.HarmanCinsiID = Harman.HarmanCinsiID
      left join CopsTipi on CopsTipi.CopsTipiID = IplikUretimIsEmri.CopsTipiID
      where IplikUretimIsEmri.IplikUretimIsEmriKayitID = :AIplikUretimIsEmriKayitID
      order by IplikUretimIsEmri.Sira
      into :IplikUretimIsEmriKayitId, :IplikUretimIsEmriYil, :IplikUretimIsEmriId, :Lot, :KalinlikNe, :IplikCinsiAdi, :IplikCinsiUretimPartiNoLabel, :PlanlananMiktar, :GerceklesenMiktar,
           :HarmanYil, :HarmanId, :HarmanAdi, :HarmanCinsiAdi, :BaslamaTarihi, :BukumTipiAdi, :BukumSayisi, :CopsTipiAdi;
  end

  Select First 1 IplikMakinaSayac.IplikMakinaSayacId, IplikMakinaSayac.VardiyaTarihi, IplikMakinaSayac.VardiyaNo, IplikMakinaSayac.IplikUretimIsEmriKayitId,
         IplikMakinaSayac.EksikIgSayisi, IplikMakinaSayac.Sayac from IplikMakinaSayac
    where IplikMakinaSayac.IplikMakinaID = :AIplikMakinaID
    order by IplikMakinaSayac.IplikMakinaSayacId desc
    into :FIplikMakinaSayacId, :FVardiyaTarihi, :FVardiyaNo, :FIplikUretimIsEmriKayitId, :FEksikIgSayisi, :OncekiSayac;

  if (FIplikMakinaSayacId is not null and
      FVardiyaTarihi = VardiyaTarihi and
      FVardiyaNo = VardiyaNo and
      FEksikIgSayisi = EksikIgSayisi and
      IntKarsilastir(FIplikUretimIsEmriKayitId, AIplikUretimIsEmriKayitId) = 1) then
  begin
    ASayac = ASayac + OncekiSayac;

    if (AKayit = 1) then
    begin
      Update IplikMakinaSayac set SayacMakina = :ASayac, Sayac = :ASayac where IplikMakinaSayac.IplikMakinaSayacID = :FIplikMakinaSayacID;

/*      insert into DeneIplikMakinaSayac(IPLIKMAKINASAYACID, IPLIKMAKINAID, VARDIYANO, SAYAC, YENISATIR)
                          values(:FIplikMakinaSayacId, :AIplikMakinaId, :VardiyaNo, :ASayac, 0);  */
    end
  end
  else
  begin
    OncekiSayac = 0;
    if (AKayit = 1) then
    begin
      insert into IplikMakinaSayac(IplikMakinaId, VardiyaTarihi, VardiyaNo, IplikUretimIsEmriKayitId, SayacMakina, Sayac, EksikIgSayisi, ManuelGiris)
                          values(:AIplikMakinaId, :VardiyaTarihi, :VardiyaNo, :AIplikUretimIsEmriKayitId, :ASayac, :ASayac, :EksikIgSayisi, 0);
/*      insert into DeneIplikMakinaSayac(IPLIKMAKINASAYACID, IPLIKMAKINAID, VARDIYANO, SAYAC, YENISATIR)
                          values(:FIplikMakinaSayacId, :AIplikMakinaId, :VardiyaNo, :ASayac, 1);*/
    end
  end

  if (BaslamaTarihi is not null) then TahminiBitisTarihi = BaslamaTarihi + 1;

  suspend;
end
