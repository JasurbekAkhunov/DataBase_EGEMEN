AS
declare variable Okey                   integer;
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable ShtKaydirmaCikis integer;
declare variable ShtKaydirmaGiris integer;
declare variable ShtSarf integer;

declare variable FHareketTipi   integer;
declare variable FMiktar        Numeric(12,3);
declare variable FMetre         Numeric(12,3);
declare variable FStokMetre     Numeric(12,3);
declare variable FBrutMiktar    Numeric(12,3);
declare variable FKapAdedi      integer;
declare variable FGirisCikis    integer;
declare variable FHamIade       integer;
declare variable FDovizId       varchar(5);
declare variable FDovizKuru Numeric(12,3);
declare variable FirmaKodumuz    varchar(20);
declare variable uitBoya integer;
declare variable uitOrme integer;
declare variable uitBukum integer;
declare variable FFirmamizaSevk Numeric(12,3);
declare variable FUretimIslemTipiId   integer;
declare variable FMaliyet                Double Precision;
declare variable FBrutMaliyet                Double Precision;
begin
  Select FirmaKodumuz from Sistem where sistemid = 1 into :FirmaKodumuz;

  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, ShtKaydirmaCikis, ShtKaydirmaGiris, ShtSarf from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim, :ShtKaydirmaCikis, :ShtKaydirmaGiris, :ShtSarf;

  Select uitBukum, uitOrme, uitBoya from UretimIslemTipiGetir into :uitBukum, :uitOrme, :uitBoya;

  Devir                  = 0;
  AlisToplami            = 0;
  AlisIadeToplami        = 0;
  FasonGirisToplami      = 0;
  FasonGirisIadeToplami  = 0;
  SatisToplami           = 0;
  BrutSatisToplami       = 0;
  SatisIadeToplami       = 0;
  UretimeCikisToplami    = 0;

  UretimdenIadeToplami   = 0;

  SakatUretimToplami     = 0;
  UretimToplami          = 0;

  FasonUretimeCikisToplami   = 0;
  FasonUretimdenIadeToplami  = 0;
  FasonUretimToplami         = 0;

  IcUretimeCikisToplami      = 0;
  IcUretimdenIadeToplami     = 0;
  IcUretimToplami            = 0;

  StokKaydirmaGiris  = 0;
  StokKaydirmaCikis  = 0;
  SarfToplami        = 0;

  FStokKapAdedi          = 0;
  FStokMetre             = 0;


  FasonUretimeCikisToplamiBoya   = 0;
  FasonUretimdenIadeToplamiBoya  = 0;
  FasonUretimToplamiBoya         = 0;

  IcUretimeCikisToplamiBoya      = 0;
  IcUretimdenIadeToplamiBoya     = 0;
  IcUretimToplamiBoya            = 0;

  KendiBoyamizaCikisToplamiBoya     = 0;
  KendiBoyamizdanIadeToplamiBoya    = 0;
  KendiBoyamizUretimToplamiBoya     = 0;

  FasonUretimeCikisToplamiBukum   = 0;
  FasonUretimdenIadeToplamiBukum  = 0;
  FasonUretimToplamiBukum         = 0;

  IcUretimeCikisToplamiBukum      = 0;
  IcUretimdenIadeToplamiBukum     = 0;
  IcUretimToplamiBukum            = 0;

  DevirTL                = 0;
  AlisToplamiTL          = 0;
  AlisIadeToplamiTL      = 0;
  FasonGirisToplamiTL    = 0;
  FasonGirisIadeToplamiTL= 0;
  SatisToplamiTL         = 0;
  BrutSatisToplamiTL     = 0;
  SatisIadeToplamiTL     = 0;
  UretimeCikisToplamiTL  = 0;
  UretimdenIadeToplamiTL = 0;
  NormalUretimToplamiTL  = 0;
  SakatUretimToplamiTL   = 0;
  UretimToplamiTL        = 0;
  FasonUretimeCikisToplamiTL = 0;
  FasonUretimdenIadeToplamiTL= 0;
  FasonUretimToplamiTL       = 0;
  IcUretimeCikisToplamiTL    = 0;
  IcUretimdenIadeToplamiTL   = 0;
  IcUretimToplamiTL          = 0;
  StokKaydirmaGirisTL= 0;
  StokKaydirmaCikisTL= 0;
  SarfToplamiTL      = 0;
  FasonUretimeCikisTBoyaTL = 0;
  FasonUretimdenIadeTBoyaTL= 0;
  FasonUretimTBoyaTL       = 0;
  IcUretimeCikisTBoyaTL    = 0;
  IcUretimdenIadeTBoyaTL   = 0;
  IcUretimTBoyaTL          = 0;
  KendiBoyamizaCikisTBoyaTL   = 0;
  KendiBoyamizdanIadeTBoyaTL  = 0;
  KendiBoyamizUretimTBoyaTL   = 0;
  FasonUretimeCikisTBukumTL = 0;
  FasonUretimdenIadeTBukumTL= 0;
  FasonUretimTBukumTL       = 0;
  IcUretimeCikisTBukumTL    = 0;
  IcUretimdenIadeTBukumTL   = 0;
  IcUretimTBukumTL          = 0;


  select sum(StokHareket.GirisCikis * StokHareket.Miktar),
           sum((StokHareket.GirisCikis * StokHareket.Miktar) * (StokHareket.BirimMaliyet) / StokHareket.TabanDovizKuru1 * StokHareket.TabanDovizKuru2) from stokhareket
    where StokHareket.HamId = :AHamId and (StokHareket.Tarih < :ARaporBaslangicTarihi) and
    Stokhareket.GirisCikis <> 0 and ((RenkId is Null and :ARenkDurumuId = 0) or (RenkId is not Null and :ARenkDurumuId >= 1)) into :Devir, :DevirTL;

  if (Devir is null) then Devir = 0;
  if (DevirTL is null) then DevirTL = 0;

  For select StokHareket.harekettipi, StokHareket.GirisCikis, StokHareket.HamIade, StokHareket.UretimIslemTipiId, StrKarsilastir(StokHareket.FirmaId, :FirmaKodumuz),
            sum(StokHareket.Miktar), sum(StokHareket.BrutMiktar), Sum(StokHareket.KapAdedi),
           sum(StokHareket.Miktar * (StokHareket.BirimMaliyet) / StokHareket.TabanDovizKuru1 * StokHareket.TabanDovizKuru2),
           sum(StokHareket.BrutMiktar * (StokHareket.BirimMaliyet) / StokHareket.TabanDovizKuru1 * StokHareket.TabanDovizKuru2) from stokhareket
    where StokHareket.HamId = :AHamId and (StokHareket.Tarih <= :ARaporBitisTarihi and StokHareket.Tarih >= :ARaporBaslangicTarihi) and
    Stokhareket.GirisCikis <> 0 and ((RenkId is Null and :ARenkDurumuId = 0) or (RenkId is not Null and :ARenkDurumuId >= 1))
    group by StokHareket.harekettipi, StokHareket.GirisCikis, StokHareket.HamIade, StokHareket.UretimIslemTipiId, StrKarsilastir(StokHareket.FirmaId, :FirmaKodumuz)
    into :FHareketTipi, :FGirisCikis, :FHamIade, :FUretimIslemTipiId, :FFirmamizaSevk,
            :FMiktar, :FBrutMiktar, :FKapAdedi, :FMaliyet, :FBrutMaliyet do

  begin
    if (FHareketTipi = shtDevir) then Devir = Devir + FMiktar;
    else if (FHareketTipi = shtAlis) then AlisToplami = AlisToplami + FMiktar;
    else if (FHareketTipi = shtSatis and FHamIade = 0) then
    begin
      SatisToplami = SatisToplami + FMiktar;
      BrutSatisToplami = BrutSatisToplami + FBrutMiktar;
    end
    else if (FHareketTipi = shtSatis and FHamIade = 1) then FasonGirisIadeToplami = FasonGirisIadeToplami + FMiktar;
    else if (FHareketTipi = shtAlisIade) then AlisIadeToplami = AlisIadeToplami + FMiktar;
    else if (FHareketTipi = shtSatisIade) then SatisIadeToplami = SatisIadeToplami + FMiktar;
    else if (FHareketTipi = shtFasonAlis) then FasonGirisToplami = FasonGirisToplami + FMiktar;
    else if (FHareketTipi = shtFasonUretimeCikis or FHareketTipi = shtUretimeCikis) then UretimeCikisToplami = UretimeCikisToplami + FMiktar;
    else if ( (FHareketTipi = shtFasonUretim or FHareketTipi = shtUretim) and FHamIade = 0) then UretimToplami = UretimToplami + FMiktar;
    else if ( (FHareketTipi = shtFasonUretim or FHareketTipi = shtUretim ) and FHamIade = 1) then UretimdenIadeToplami = UretimdenIadeToplami + FMiktar;
    else if ( FHareketTipi = shtKaydirmaCikis) then StokKaydirmaCikis = StokKaydirmaCikis + FMiktar;
    else if ( FHareketTipi = shtKaydirmaGiris) then StokKaydirmaGiris = StokKaydirmaGiris + FMiktar;
    else if ( FHareketTipi = shtSarf) then SarfToplami = SarfToplami + FMiktar;

    if (FHareketTipi = shtDevir) then DevirTL = DevirTL + FMaliyet;
    else if (FHareketTipi = shtAlis) then AlisToplamiTL = AlisToplamiTL + FMaliyet;
    else if (FHareketTipi = shtSatis and FHamIade = 0) then
    begin
      SatisToplamiTL = SatisToplamiTL + FMaliyet;
      BrutSatisToplamiTL = BrutSatisToplamiTL + FBrutMaliyet;
    end
    else if (FHareketTipi = shtSatis and FHamIade = 1) then FasonGirisIadeToplamiTL = FasonGirisIadeToplamiTL + FMaliyet;
    else if (FHareketTipi = shtAlisIade) then AlisIadeToplamiTL = AlisIadeToplamiTL + FMaliyet;
    else if (FHareketTipi = shtSatisIade) then SatisIadeToplamiTL = SatisIadeToplamiTL + FMaliyet;
    else if (FHareketTipi = shtFasonAlis) then FasonGirisToplamiTL = FasonGirisToplamiTL + FMaliyet;
    else if (FHareketTipi = shtFasonUretimeCikis or FHareketTipi = shtUretimeCikis) then UretimeCikisToplamiTL = UretimeCikisToplamiTL + FMaliyet;
    else if ( (FHareketTipi = shtFasonUretim or FHareketTipi = shtUretim) and FHamIade = 0) then UretimToplamiTL = UretimToplamiTL + FMaliyet;
    else if ( (FHareketTipi = shtFasonUretim or FHareketTipi = shtUretim ) and FHamIade = 1) then UretimdenIadeToplamiTL = UretimdenIadeToplamiTL + FMaliyet;
    else if ( FHareketTipi = shtKaydirmaCikis) then StokKaydirmaCikisTL = StokKaydirmaCikisTL + FMaliyet;
    else if ( FHareketTipi = shtKaydirmaGiris) then StokKaydirmaGirisTL = StokKaydirmaGirisTL + FMaliyet;
    else if ( FHareketTipi = shtSarf) then SarfToplamiTL = SarfToplamiTL + FMaliyet;


    if (FHareketTipi = shtFasonUretimeCikis and FUretimIslemTipiId = uitOrme) then FasonUretimeCikisToplami = FasonUretimeCikisToplami + FMiktar;
    else if (FHareketTipi = shtUretimeCikis and FUretimIslemTipiId = uitOrme) then IcUretimeCikisToplami = IcUretimeCikisToplami + FMiktar;
    else if ( FHareketTipi = shtFasonUretim and FHamIade = 0 and FUretimIslemTipiId = uitOrme) then FasonUretimToplami = FasonUretimToplami + FMiktar;
    else if ( FHareketTipi = shtUretim and FHamIade = 0 and FUretimIslemTipiId = uitOrme) then IcUretimToplami = IcUretimToplami + FMiktar;
    else if ( FHareketTipi = shtFasonUretim and FHamIade = 1 and FUretimIslemTipiId = uitOrme) then FasonUretimdenIadeToplami = FasonUretimdenIadeToplami + FMiktar;
