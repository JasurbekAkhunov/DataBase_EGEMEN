CREATE PROCEDURE MAMULKODUHESAPLA(
    APARTISINIFID INTEGER)
RETURNS (
    MAMULKODU VARCHAR(20))
AS
declare variable HamId          integer;
declare variable HamTipiId      integer;
declare variable ReceteId       varchar(20);
declare variable YapilanAsamalar       varchar(255);
declare variable AsamaIlaveIslemler    varchar(255);
declare variable IlaveIslemlerId       integer;
declare variable AsamaId               integer;
declare variable MamulKoduIcinId       varchar(1);
declare variable MamulKoduNumarator    integer;
declare variable GSira                 integer;
begin
  Select HamId from PartiStok where PartiStok.PartiSinifId = :APartiSinifId into :HamId;

  Select Parti.ReceteId from PartiStok
    left join Parti On Parti.PartiKayitId = PartiStok.PartiKayitId
    where PartiStok.PartiSinifId = :APartiSinifId into :ReceteId;

  YapilanAsamalar = '';

  for Select Asama.AsamaId, PartiAsamalari.Sira from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
    left join asama on asama.AsamaId = PartiAsamalari.AsamaId or asama.AsamaId = recete.UstAsamaId
    where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamalari.SerbestAsama = 0 and
          PartiAsamalari.KontrolAsama = 0 and PartiAsamalari.IlaveProses = 0 and PartiAsamalari.PartiTamirID is null and Asama.MamulKodunaDahil = 1
    order by PartiAsamalari.Sira
    into :AsamaId, :GSira do
  begin
    if (YapilanAsamalar <> '') then YapilanAsamalar = YapilanAsamalar || ',';
    YapilanAsamalar = YapilanAsamalar || EgemenIntegertoStr(AsamaId);
  end

  AsamaIlaveIslemler = '';

  for Select Asama.AsamaId, PartiIlaveIslemler.IlaveIslemlerId, PartiAsamalari.Sira from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join PartiIlaveIslemler on PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
    left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
    left join asama on asama.AsamaId = PartiAsamalari.AsamaId or asama.AsamaId = recete.UstAsamaId
    where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamalari.SerbestAsama = 0 and
          PartiAsamalari.KontrolAsama = 0 and PartiAsamalari.IlaveProses = 0 and PartiAsamalari.PartiTamirID is null and Asama.MamulKodunaDahil = 1 and
          PartiIlaveIslemler.IlaveIslemlerId is not null
    order by PartiAsamalari.Sira
    into :AsamaId, :IlaveIslemlerId, :GSira do
  begin
    if (AsamaIlaveIslemler <> '') then AsamaIlaveIslemler = AsamaIlaveIslemler || ',';
    AsamaIlaveIslemler = AsamaIlaveIslemler || '(' || EgemenIntegertoStr(AsamaId) || ',' || EgemenIntegertoStr(IlaveIslemlerId) || ')';
  end

  MamulKodu = Null;
  if (ReceteId is not null) then
  begin
    Select MamulKodu from MamulKodu Where MamulKodu.HamId = :HamId and MamulKodu.ReceteID = :ReceteId and
                                          MamulKodu.YapilanAsamalar = :YapilanAsamalar and MamulKodu.AsamaIlaveIslemler = :AsamaIlaveIslemler
                                          into :MamulKodu;

  end
  else
  begin
    Select MamulKodu from MamulKodu Where MamulKodu.HamId = :HamId and MamulKodu.ReceteID is null and
                                          MamulKodu.YapilanAsamalar = :YapilanAsamalar and MamulKodu.AsamaIlaveIslemler = :AsamaIlaveIslemler into :MamulKodu;
  end

  if (MamulKodu is null) then
  begin
    Select HamTipi.HamTipiId, HamTipi.MamulKoduIcinId, HamTipi.MamulKoduNumarator from Ham
      left join HamTipi on HamTipi.HamTipiID = Ham.HamTipiId
      where Ham.HamID = :HamId
      into :HamTipiId, :MamulKoduIcinId, :MamulKoduNumarator;
    if (MamulKoduIcinId is null) then exception MamulKoduIcinIdTanimlanmamis;

    MamulKoduNumarator = MamulKoduNumarator + 1;

    MamulKodu = MamulKoduIcinId || EgemenIntegerToStr(MamulKoduNumarator);

    insert into MamulKodu(MamulKodu, HamId, ReceteId, YapilanAsamalar, AsamaIlaveIslemler)
                   values(:MamulKodu, :HamId, :ReceteId, :YapilanAsamalar, :AsamaIlaveIslemler);

    Update HamTipi Set MamulKoduNumarator = :MamulKoduNumarator where HamTipi.HamTipiId = :HamTipiId;

  end

  suspend;
end
