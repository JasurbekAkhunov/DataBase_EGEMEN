CREATE PROCEDURE BOYASATISAUYGUNSIPARISLISTESI(
    AHAMID INTEGER,
    AKALITEID KALITEKODUSIZE)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    FIRMAID FIRMAKODUSIZE,
    FIRMAADI ADI100,
    HAMID INTEGER,
    HAMADI ADI255,
    KALITEID KALITEKODUSIZE,
    KALITEADI ADI255,
    GEREKENKILO BOYAMIKTARSIZE,
    TALIMATKODU VARCHAR(20))
AS
declare variable SIPARISKILO BOYAMIKTARSIZE;
declare variable REZERVKILO BOYAMIKTARSIZE;
declare variable REZERVSIZSEVKKILO BOYAMIKTARSIZE;
declare variable FIRMAKODUMUZ FIRMAKODUSIZE;
begin
  select FirmaKodumuz from Sistem where SistemId = 1 into :FirmaKodumuz;

  for select BoyaSiparisDetay.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Firma.FirmaId , Firma.Adi,
    Ham.HamId, Ham.Adi, Kalite.KaliteId, Kalite.KaliteAdi, BoyaSiparisDetay.SiparisKilo,
    (select Coalesce(Sum(BoyaRezervFisiDetay.EdilecekKilo),0) from BoyaRezervFisiDetay where BoyaRezervFisiDetay.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId),
    (select Coalesce(Sum(PartiStokTarti.Kilo),0) from BoyaIrsaliyeDetay
    left join PartiStokTarti on PartiStokTarti.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId and PartiStokTarti.BoyaRezervFisiDetayId is null
    where BoyaIrsaliyeDetay.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId), BoyaSiparisDetay.TalimatKodu
    from BoyaSiparisDetay
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    where BoyaSiparisDetay.Kapali = 0 and
    BoyaSiparisDetay.HamId = :AHamId and
    (:AKaliteId is null or BoyaSiparisDetay.KaliteId = :AKaliteId) and
    BoyaSiparis.FirmaId <> :FirmaKodumuz
  into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :HamId, :HamAdi, :KaliteId, :KaliteAdi,
       :SiparisKilo, :RezervKilo, :RezervsizSevkKilo, :TalimatKodu do
  begin
    GerekenKilo = SiparisKilo - RezervKilo - RezervsizSevkKilo;
    if (GerekenKilo > 0) then
      suspend;
  end
end
