CREATE PROCEDURE PARTITAMIRASAMALARIEKLE(
    APARTIKAYITID INTEGER,
    APARTITAMIRID INTEGER,
    AASAMAID INTEGER,
    ASIRA INTEGER)
AS
declare variable AsamaId integer;
declare variable MinSira integer;
declare variable FPartiAsamalariId integer;
declare variable FAsamaId   integer;
declare variable FReceteId  Varchar(20);
declare variable FOtomasyonaDahilmi integer;
declare variable FKullaniciId integer;
begin
  Select max(Sira) from PartiTamirSinif
    left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiTamirSinif.PartiSinifId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join Recete on Recete.ReceteID = PartiAsamalari.receteId
    left join Asama on Asama.AsamaID = PartiAsamalari.AsamaId or Asama.AsamaID = Recete.UstAsamaId
    where PartiTamirSinif.PartiTamirid = :APartiTamirId and PartiAsamaStok.AsamayaDahil = 1 and
          PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamalari.Sira < :ASira and Asama.AsamaID = :AAsamaId
    into :MinSira;

/*
  Insert into PartiAsamalari(PartiAsamalariId, PartiKayitId, PartiTamirId, AsamaId, ReceteId, OtomasyonaDahilmi, InsertKullaniciId, KullaniciId)
    Select distinct gen_Id(PartiAsamalari_Id_gen,1), :APartiKayitId, :APartiTamirId, PartiAsamalari.AsamaId, PartiAsamalari.ReceteId, PartiAsamalari.OtomasyonaDahilmi, PartiAsamalari.KullaniciId, PartiAsamalari.KullaniciId from PartiTamirSinif
    left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiTamirSinif.PartiSinifId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    where PartiTamirSinif.PartiTamirid = :APartiTamirId and PartiAsamaStok.AsamayaDahil = 1 and
          PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamalari.Sira > :MinSira and
          PartiAsamalari.Sira < :ASira Order By Sira;
*/

    For Select distinct PartiAsamalari.AsamaId, PartiAsamalari.ReceteId, PartiAsamalari.OtomasyonaDahilmi, PartiAsamalari.KullaniciId from PartiTamirSinif
          left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiTamirSinif.PartiSinifId
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          where PartiTamirSinif.PartiTamirid = :APartiTamirId and PartiAsamaStok.AsamayaDahil = 1 and
            PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamalari.Sira > :MinSira and
            PartiAsamalari.Sira < :ASira Order By Sira
          into :FAsamaId, :FReceteId, :FOtomasyonaDahilmi, :FKullaniciId do
    begin
      FPartiAsamalariId = gen_Id(PartiAsamalari_Id_gen,1);

      Insert into PartiAsamalari(PartiAsamalariId, PartiKayitId, PartiTamirId, AsamaId, ReceteId, OtomasyonaDahilmi, InsertKullaniciId, KullaniciId)
                          Values(:FPartiAsamalariId, :APartiKayitId, :APartiTamirId, :FAsamaId, :FReceteId, :FOtomasyonaDahilmi, :FKullaniciId, :FKullaniciId);
    end

end
