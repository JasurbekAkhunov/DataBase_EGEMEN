CREATE PROCEDURE RECETESANALPARTIGETIR(
    ARECETEID VARCHAR(20))
RETURNS (
    RECETESANALPARTIKAYITID INTEGER,
    RECETESANALPARTIYIL INTEGER,
    RECETESANALPARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    TARIH TIMESTAMP,
    ACIKLAMA VARCHAR(50),
    PARTIKILO NUMERIC(12,3),
    BANYOCARPANI DOUBLE PRECISION,
    FULARBASINCI DOUBLE PRECISION,
    KOYULTMA DOUBLE PRECISION,
    DURULAMASUYU INTEGER,
    BOYASUYU INTEGER,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    SARFFISNO VARCHAR(15))
AS
begin
  For Select ReceteSanalParti.ReceteSanalPartiKayitId, ReceteSanalParti.Tarih, ReceteSanalParti.ReceteSanalPartiYil, ReceteSanalParti.ReceteSanalPartiId, ReceteSanalParti.Aciklama,
             ReceteSanalParti.PartiKilo, ReceteSanalParti.BanyoCarpani, ReceteSanalParti.FularBasinci, ReceteSanalParti.Koyultma, ReceteSanalParti.DurulamaSuyu,
             ReceteSanalParti.BoyaSuyu, ReceteSanalParti.Notu, Ik.Adi, ReceteSanalParti.InsertTarihi, Sk.Adi, ReceteSanalParti.DegisimTarihi from ReceteSanalParti
    Left Join Kullanici IK on IK.KullaniciId = ReceteSanalParti.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = ReceteSanalParti.KullaniciId
    where ReceteSanalParti.ReceteID = :AReceteID
    order by ReceteSanalParti.ReceteID, ReceteSanalParti.Tarih
    into :ReceteSanalPartiKayitId, :Tarih, :ReceteSanalPartiYil, :ReceteSanalPartiId, :Aciklama, :PartiKilo, :BanyoCarpani, :FularBasinci, :Koyultma, :DurulamaSuyu,
         :BoyaSuyu, :Notu, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(ReceteSanalPartiYil)  || ') ' || EgemenIntegerToStr(ReceteSanalPartiId) || ' ';

     Select First 1 Stok.SeriNo||Stok.FisNo  from StokDetay
       Left Join Stok on Stok.StokID = StokDetay.StokID
       where StokDetay.ReceteSanalPartiKayitId = :ReceteSanalPartiKayitId into :SarfFisNo;

    suspend;
  end
end
