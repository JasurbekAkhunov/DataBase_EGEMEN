CREATE PROCEDURE ONAYBEKLEYENDESENISTEKRAPORU(
    ARAPORTIPI INTEGER,
    ADESENISTEKTIPIID VARCHAR(5),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ADESENISTEKGRUPID VARCHAR(4))
RETURNS (
    DESENISTEKLERIID INTEGER,
    DESENID VARCHAR(10),
    DESENADI VARCHAR(80),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DESENISTEKTIPIID VARCHAR(5),
    DESENISTEKTIPIADI VARCHAR(40),
    ISTEKTARIHI TIMESTAMP,
    ISTEKYAPANKULLANICIID INTEGER,
    ISTEKYAPANKULLANICIADI VARCHAR(40),
    BASLATANKULLANICIADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENKULLANICIADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    DEPARTMANID INTEGER,
    DEPARTMANADI VARCHAR(30),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTIPI INTEGER,
    ONAYTIPISTR VARCHAR(30),
    ONAYAHAZIR INTEGER)
AS
begin
  if (ARaporTipi = 0) then
  begin
    For Select DesenIstekleri.DesenIstekleriId, Desen.DesenId, Desen.Adi, Desen.FirmaId, Firma.Adi,
               DesenIstekleri.DesenIstekTipiId, DesenIstekTipi.Adi,
               DesenIstekleri.IstekTarihi, DesenIstekleri.IstekYapanKullaniciId, IYK.Adi, BasK.Adi, DesenIstekleri.BaslamaTarihi, BitK.Adi, DesenIstekleri.BitisTarihi,
               DesenIstekTipi.YapacakDepartmanId, Departman.Adi, DesenIstekleri.OnayDurumu, DesenIstekleri.OnayTarihi, OK.Adi, DesenIstekleri.OnayTipi from DesenIstekleri
    Left Join Desen on Desen.DesenId = DesenIstekleri.DesenId
    Left Join Firma on Firma.FirmaId = Desen.FirmaId
    Left Join Kullanici BasK on BasK.KullaniciId = DesenIstekleri.BaslatanKullaniciId
    Left Join Kullanici BitK on BitK.KullaniciId = DesenIstekleri.BitirenKullaniciId
    Left Join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiId
    Left Join Departman on Departman.DepartmanId = DesenIstekTipi.YapacakDepartmanId
    left join Kullanici IYK on IYK.KullaniciId = DesenIstekleri.IstekYapanKullaniciId
    left join Kullanici OK on OK.KullaniciId = DesenIstekleri.OnaylayanKullaniciId
    where DesenIstekleri.OnayTipi <> 0 and OnayDurumu = 0 and
          (DesenIstekTipi.DesenIstekGrupId = :ADesenIstekGrupId or :ADesenIstekGrupId is null) and
          (DesenIstekleri.DesenIstekTipiId = :ADesenIstekTipiId or :ADesenIstekTipiId is null)
    into :DesenIstekleriId, :DesenId, :DesenAdi, :FirmaId, :FirmaAdi,
         :DesenIstekTipiId, :DesenIstekTipiAdi, :IstekTarihi, :IstekYapanKullaniciId, :IstekYapanKullaniciAdi,
         :BaslatanKullaniciAdi, :BaslamaTarihi, :BitirenKullaniciAdi, :BitisTarihi, :DepartmanId, :DepartmanAdi, :OnayDurumu,
         :OnayTarihi, :OnaylayanKullaniciAdi, :OnayTipi do
    begin
      if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';

      OnayaHazir = 1;

      if (OnayTipi = 1) then OnayTipiStr = 'Başlamadan Onay Ver';
      else
      begin
        OnayTipiStr = 'Bitince Onay Ver';
        if (BitisTarihi is null) then OnayaHazir = 0;
      end


      suspend;
    end
  end
  else
  begin
    For Select DesenIstekleri.DesenIstekleriId, Desen.DesenId, Desen.Adi, Desen.FirmaId, Firma.Adi,
               DesenIstekleri.DesenIstekTipiId, DesenIstekTipi.Adi,
               DesenIstekleri.IstekTarihi, DesenIstekleri.IstekYapanKullaniciId, IYK.Adi, BasK.Adi, DesenIstekleri.BaslamaTarihi, BitK.Adi, DesenIstekleri.BitisTarihi,
               DesenIstekTipi.YapacakDepartmanId, Departman.Adi, DesenIstekleri.OnayDurumu, DesenIstekleri.OnayTarihi, OK.Adi, DesenIstekleri.OnayTipi from DesenIstekleri
    Left Join Desen on Desen.DesenId = DesenIstekleri.DesenId
    Left Join Firma on Firma.FirmaId = Desen.FirmaId
    Left Join Kullanici BasK on BasK.KullaniciId = DesenIstekleri.BaslatanKullaniciId
    Left Join Kullanici BitK on BitK.KullaniciId = DesenIstekleri.BitirenKullaniciId
    Left Join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiId
    Left Join Departman on Departman.DepartmanId = DesenIstekTipi.YapacakDepartmanId
    left join Kullanici IYK on IYK.KullaniciId = DesenIstekleri.IstekYapanKullaniciId
    left join Kullanici OK on OK.KullaniciId = DesenIstekleri.OnaylayanKullaniciId
    where (DesenIstekleri.OnayTarihi >= :ABaslangicTarihi and DesenIstekleri.OnayTarihi <= :ABitisTarihi) and
          (DesenIstekleri.OnayDurumu = :ARaporTipi) and
          (DesenIstekTipi.DesenIstekGrupId = :ADesenIstekGrupId or :ADesenIstekGrupId is null) and
          (DesenIstekleri.DesenIstekTipiId = :ADesenIstekTipiId or :ADesenIstekTipiId is null)
    into :DesenIstekleriId, :DesenId, :DesenAdi, :FirmaId, :FirmaAdi,
         :DesenIstekTipiId, :DesenIstekTipiAdi, :IstekTarihi, :IstekYapanKullaniciId, :IstekYapanKullaniciAdi,
         :BaslatanKullaniciAdi, :BaslamaTarihi, :BitirenKullaniciAdi, :BitisTarihi, :DepartmanId, :DepartmanAdi, :OnayDurumu,
         :OnayTarihi, :OnaylayanKullaniciAdi, :OnayTipi do
    begin
      if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
      else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
      else OnayDurumuStr = '';

      OnayaHazir = 1;

      if (OnayTipi = 1) then OnayTipiStr = 'Başlamadan Onay Ver';
      else
      begin
        OnayTipiStr = 'Bitince Onay Ver';
        if (BitisTarihi is null) then OnayaHazir = 0;
      end

      suspend;
    end
  end
end
