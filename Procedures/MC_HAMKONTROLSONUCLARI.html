CREATE PROCEDURE MC_HAMKONTROLSONUCLARI(
    AHAMSTOKMASTERID INTEGER,
    AFIRMAID VARCHAR(20),
    AKULLANICIID INTEGER)
RETURNS (
    IRSALIYENO IRSALIYENOSIZE,
    HAMADI VARCHAR(255),
    NEMORANI DOUBLE PRECISION,
    YIKAMAONCESIEN DOUBLE PRECISION,
    YIKAMASONRASIEN DOUBLE PRECISION,
    YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    YIKAMASONRASIGRAMAJ DOUBLE PRECISION,
    ENDENCEKMEZLIK DOUBLE PRECISION,
    BOYDANCEKMEZLIK DOUBLE PRECISION,
    HAMDONME DOUBLE PRECISION,
    YIKAMAONCESIRAPORBOYU VARCHAR(100),
    YIKAMASONRASIRAPORBOYU VARCHAR(100),
    YIKAMAONCESICPI VARCHAR(10),
    YIKAMASONRASICPI VARCHAR(10),
    YIKAMAONCESIWPI VARCHAR(10),
    YIKAMASONRASIWPI VARCHAR(10),
    HAMKONTROLIPLIKUZUNLUGU1 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU2 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU3 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU4 VARCHAR(40),
    CEKMEZLIKSTANDARDIID INTEGER,
    CEKMEZLIKSTANDARDIADI VARCHAR(50),
    BOYAMAKABILIYETI VARCHAR(10),
    YIKAMAHASLIGI VARCHAR(10),
    NUMARATIPIADI VARCHAR(10),
    NUMARA VARCHAR(10),
    BUKUM VARCHAR(10),
    MUKAVEMET VARCHAR(10),
    UZAMA VARCHAR(10),
    UVKONTROL INTEGER,
    KONTROLDURUMU INTEGER,
    KONTROLKULLANICIADI VARCHAR(40),
    KONTROLTARIHI TIMESTAMP)
AS
declare variable Okey     integer;
begin
  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  select HamStokIrsaliye.IrsaliyeNo, Ham.Adi, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiGramaj, HamStokMaster.EndenCekmezlik,
             HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
             HamStokMaster.YikamaOncesiCPI, HamStokMaster.YikamaSonrasiCPI, HamStokMaster.YikamaOncesiWPI, HamStokMaster.YikamaSonrasiWPI,
             HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2, HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4,
             HamStokMaster.BoyamaKabiliyeti, HamStokMaster.YikamaHasligi, NumaraTipi.Adi, HamStokMaster.Numara, HamStokMaster.Bukum, HamStokMaster.Mukavemet, HamStokMaster.NemOrani, HamStokMaster.Uzama, HamStokMaster.UVKontrol,
             HamStokMaster.KontrolDurumu, KK.Adi, HamStokMaster.Kontroltarihi, HamStokMaster.CekmezlikStandardiId, CekmezlikStandardi.Adi from HamStokMaster
  left join Ham on Ham.HamId = HamStokMaster.HamId
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
  left join NumaraTipi on NumaraTipi.NumaraTipiId = HamStokMaster.NumaraTipiId
  Left Join Kullanici KK on KK.KullaniciId = HamStokMaster.KontrolKullaniciId
  left join CekmezlikStandardi on CekmezlikStandardi.CekmezlikStandardiId = HamStokMaster.CekmezlikStandardiId
  where HamStokMaster.HamStokMasterId = :AHamStokMasterId and HamStokMaster.FirmaId = :AFirmaId and
        HamStokMaster.Fason = 1
  into :IrsaliyeNo, :HamAdi, :YikamaOncesiEn, :YikamaSonrasiEn, :YikamaOncesiGramaj, :YikamaSonrasiGramaj, :EndenCekmezlik,
       :BoydanCekmezlik, :HamDonme, :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu,
       :YikamaOncesiCPI, :YikamaSonrasiCPI, :YikamaOncesiWPI, :YikamaSonrasiWPI, :HamKontrolIplikUzunlugu1, :HamKontrolIplikUzunlugu2,
       :HamKontrolIplikUzunlugu3, :HamKontrolIplikUzunlugu4,
       :BoyamaKabiliyeti, :YikamaHasligi, :NumaraTipiAdi,
       :Numara, :Bukum, :Mukavemet, :NemOrani, :Uzama, :UVKontrol, :KontrolDurumu, :KontrolKullaniciAdi, :Kontroltarihi,
       :CekmezlikStandardiId, :CekmezlikStandardiAdi;
  suspend;
end
