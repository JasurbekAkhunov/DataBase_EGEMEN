CREATE PROCEDURE DETAYLIYARIMAMULRAPORU_2(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIH TIMESTAMP,
    ABITISTARIH TIMESTAMP,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    AFASON INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    RAFDETAYID INTEGER,
    KALANMIKTAR NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    TARIH TIMESTAMP,
    MIKTAR NUMERIC(12,3),
    IADEEDILENMIKTAR NUMERIC(12,3),
    PARTILENENMIKTAR NUMERIC(12,3),
    SATISMIKTAR NUMERIC(12,3),
    RAFTRANSFERMIKTAR NUMERIC(12,3),
    KILO NUMERIC(12,3),
    IADEEDILENKILO NUMERIC(12,3),
    PARTILENENKILO NUMERIC(12,3),
    AKTARMAYACIKANKILO NUMERIC(12,3),
    SATISKILO NUMERIC(12,3),
    RAFTRANSFERKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    IADEEDILENMETRE NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    SATISMETRE NUMERIC(12,3),
    RAFTRANSFERMETRE NUMERIC(12,3))
AS
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
begin
  select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf;

  for select PartiStok.PartiKayitId, PartiStok.PartiSinifId, RafDetay.RafDetayId, HamStokMaster.InsertTarihi,
             RafDetay.Miktar, RafDetay.IadeEdilenMiktar, RafDetay.PartilenenMiktar, RafDetay.SatisMiktar, RafDetay.RafTransferMiktar,
             RafDetay.Kilo, RafDetay.IadeEdilenKilo, RafDetay.PartilenenKilo, RafDetay.AktarmayaCikanKilo, RafDetay.SatisKilo, RafDetay.RafTransferKilo,
             RafDetay.Metre, RafDetay.IadeEdilenMetre, RafDetay.PartilenenMetre, RafDetay.SatisMetre, RafDetay.RafTransferMetre from RafDetay
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      left join Ham on Ham.HamID = RafDetay.HamId
      left join PartiStok on PartiStok.partiSinifID = HamStokMaster.PartiSinifId
      left join Parti on Parti.PartiKayitId =PartiStok.PartiKayitId
      where  (HamStokMaster.PartiSinifId is not null) and (:ARaporTipi = 0 and HamStokMaster.Tarih >= :ABaslangicTarih and HamStokMaster.Tarih <= :ABitisTarih or
              :ABaslangicPartiYil = :ABitisPartiYil and :ABaslangicPartiID = :ABitisPartiId and Parti.PartiYil = :ABaslangicPartiYil and Parti.PartiId = :ABaslangicPartiId or
              :ARaporTipi <> 0 and (:ABaslangicPartiYil <> :ABitisPartiYil or :ABaslangicPartiID <> :ABitisPartiId) and
              ((Parti.PartiYil > :ABaslangicPartiYil or (Parti.PartiYil = :ABaslangicPartiYil and Parti.PartiId >= :ABaslangicPartiId)) and
               (Parti.PartiYil < :ABitisPartiYil or (Parti.PartiYil = :ABitisPartiYil  and Parti.PartiId <= :ABitisPartiId)))) and
            (RafDetay.HamId = :AHamId or :AHamId is null) and
            (( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 0) or
               (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 1) or :AHamTipiId is null) and
            (HamStokMaster.Fason = :AFason or :AFason is null)
      into :PartiKayitId, :PartiSinifId, :RafDetayId, :Tarih,
           :Miktar, :IadeEdilenMiktar, :PartilenenMiktar, :SatisMiktar, :RafTransferMiktar,
           :Kilo, :IadeEdilenKilo, :PartilenenKilo, :AktarmayaCikanKilo, :SatisKilo, :RafTransferKilo,
           :Metre, :IadeEdilenMetre, :PartilenenMetre, :SatisMetre, :RafTransferMetre   do
  begin
    KalanMiktar = Miktar - IadeEdilenMiktar  - PartilenenMiktar -  SatisMiktar - RafTransferMiktar;
    KalanKilo   = Kilo - IadeEdilenKilo - PartilenenKilo - AktarmayaCikanKilo  - RafTransferKilo;
    KalanMetre  = Metre - IadeEdilenMetre - PartilenenMetre -  SatisMetre - RafTransferMetre;

    suspend;
  end
end
