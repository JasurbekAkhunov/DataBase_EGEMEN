CREATE PROCEDURE REFPARTIARAKONTROLDEGERLERGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    REFPARTISINIFID INTEGER,
    ARAKONTROLTIPIID INTEGER,
    ARAKONTROLTIPIADI VARCHAR(30),
    SONUC VARCHAR(50),
    ONAYLAYANPERSONELID INTEGER,
    ONAYLAYANPERSONELADI VARCHAR(40),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    PERSONELADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40))
AS
begin
  For select PS.PartiSinifId, PS.ReferansPartiSinifId,PartiAsamaAraKontrol.AraKontrolTipiId,
    AraKontrolTipi.Adi,PartiAsamaAraKontrol.Sonuc,PartiAsamaAraKontrol.OnaylayanPersonelId,
    OP.Adi,PartiAsamaAraKontrol.OnayDurumu,PartiAsamaAraKontrol.OnayTarihi,SP.Adi,
    PartiAsamaAraKontrol.DegisimTarihi,PartiAsamalari.AsamaId,Asama.Adi from PartiStok PS
  left join PartiStok RefPS on RefPS.PartiSinifID = PS.ReferansPartiSinifId
  left join PartiAsamaAraKontrol on PartiAsamaAraKontrol.PartiSinifID = RefPs.PartiSinifId
  left join Personel OP on OP.PersonelId = PartiAsamaAraKontrol.OnaylayanPersonelId
  left join Personel SP on SP.PersonelId = PartiAsamaAraKontrol.PersonelId
  left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaAraKontrol.PartiAsamalariId
  left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
  left join AraKontrolTipi on AraKontrolTipi.AraKontrolTipiId = PartiAsamaAraKontrol.AraKontrolTipiId
  where PS.PartiKayitID = :APartiKayitId and PS.ReferansPartiSinifId is not null
  into :PartiSinifId, :RefPartiSinifId, :AraKontrolTipiId, :AraKontrolTipiAdi, :Sonuc, :OnaylayanPersonelId, :OnaylayanPersonelAdi,
       :OnayDurumu, :OnayTarihi, :PersonelAdi, :DegisimTarihi, :AsamaId, :AsamaAdi do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';
    suspend;
  end
end
