CREATE PROCEDURE HGURETIMECIKISURUNLISTESI(
    AHGISEMRIKAYITID INTEGER,
    ADEPOID VARCHAR(10))
RETURNS (
    HGISEMRIKAYITID INTEGER,
    HGISEMRIURUNID INTEGER,
    URUNID URUNKODUSIZE,
    ISEMRITARIH TIMESTAMP,
    URUNADI ADI80,
    ISEMRIURUNID URUNKODUSIZE,
    ISEMRIURUNADI ADI80,
    PASTALNO TYPE OF COLUMN HGISEMRI.URETIMPARTINO,
    PLANLANANMIKTAR BOYAMIKTARSIZE,
    CIKISMIKTAR BOYAMIKTARSIZE,
    DEPOID DEPOKODUSIZE,
    DEPOADI ADI30,
    STOKMEVCUDU DOUBLE PRECISION,
    GERCEKLESENMIKTAR INTEGER,
    KALANMIKTAR DOUBLE PRECISION)
AS
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable UrunTipiId integer;
declare variable utAksesuar integer;
declare variable HGSiparisDetayId integer;
declare variable HGPastalDetayliTakipEt EgeBoolean;
declare variable DikimIslemi Integer;
begin
  Select shtFasonUretimeCikis, shtUretimeCikis from StokHareketTipiGetir into :shtFasonUretimeCikis, shtUretimeCikis;
  select utAksesuar from UrunTipiGetir into :utAksesuar;
  Select HGPastalDetayliTakipEt from Sistem Where SistemId = 1 into :HGPastalDetayliTakipEt;

  Select Sum(ModelUretimIslemTipi.DikimIslemi) from HGIsEmriUretimIslemleri
    Left Join HGSipDetayUretimIslemleri on HGSipDetayUretimIslemleri.HGSipDetayUretimIslemleriID = HGIsEmriUretimIslemleri.HGSipDetayUretimIslemleriID
    Left Join ModelUretimIslemTipi on ModelUretimIslemTipi.ModelUretimIslemTipiID = HGSipDetayUretimIslemleri.ModelUretimIslemTipiID
    where HGIsEmriUretimIslemleri.HGIsEmriKayitID = :AHGIsEmriKayitID
  into :DikimIslemi;

  if (DikimIslemi = Null) then DikimIslemi = 0;

  For Select HgIsEmri.HGSiparisDetayId, HgIsEmri.Tarih, HGIsEmriUrun.HGIsEmriKayitID, HGIsEmriUrun.HGIsEmriUrunID, HGIsEmriUrun.UrunID, HGUrun.Adi, HGIsEmriUrun.UrunID, HGUrun.Adi,
             PlanlananMiktar, HGUrun.UrunTipiId, HgIsEmriUrun.GerceklesenMiktar From HgIsEmri
        Left Join HGIsEmriUrun on HgIsEmriUrun.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
             Join Urun HGUrun ON HGUrun.UrunID = HGIsEmriUrun.UrunID and (HgIsEmri.ParcaIscilik = 0 or (HgIsEmri.ParcaIscilik = 1 and HGUrun.ModelParcaId is not null))
        Where HgIsEmri.HGIsEmriKayitID = :AHGIsEmriKayitID
      Union All
      Select HgIsEmri.HGSiparisDetayId, HgIsEmri.Tarih, HGIsEmriDetay.HGIsEmriKayitID, HGIsEmriDetay.HGIsEmriUrunID, HGIsEmriUrun.UrunID, HGUrun.Adi, HGIsEmriDetay.UrunID, UrunDetay.Adi,
            (HGIsEmriUrun.PlanlananMiktar * HGIsEmriDetay.Miktar) PlanlananMiktar, UrunDetay.UrunTipiId, HgIsEmriUrun.GerceklesenMiktar From HgIsEmri
        Left Join HGIsEmriDetay on HgIsEmriDetay.HgIsEmriKayitId = HgIsEmri.HgIsEmriKayitId
        Left join Urun UrunDetay ON UrunDetay.UrunID = HGIsEmriDetay.UrunID
        Left Join HGIsEmriUrun ON HGIsEmriUrun.HGIsEmriUrunID = HGIsEmriDetay.HGIsEmriUrunID
             Join Urun HGUrun ON HGUrun.UrunID = HGIsEmriUrun.UrunID and (HgIsEmri.ParcaIscilik = 0 or (HgIsEmri.ParcaIscilik = 1 and UrunDetay.ModelParcaId is not null))
        Where HgIsEmri.HGIsEmriKayitId = :AHGIsEmriKayitID
      Order By 2
  into :HGSiparisDetayId, :IsEmriTarih, :HGIsEmriKayitID, :HGIsEmriUrunID, :IsEmriUrunID, :IsEmriUrunAdi, :UrunID, :UrunAdi, :PlanlananMiktar, :UrunTipiId, :GerceklesenMiktar do
  begin
    if (UrunTipiId = utAksesuar or  HGPastalDetayliTakipEt = 0 or DikimIslemi = 0) then
    begin
      Select Abs(Sum(Miktar * GirisCikis)) from StokDetay
        where StokDetay.HGIsEmriKayitID = :HGIsEmriKayitID and
              StokDetay.HGIsEmriUrunID = :HGIsEmriUrunID and
              StokDetay.UrunID = :UrunID and
              (StokDetay.HareketTipi = :shtFasonUretimeCikis or StokDetay.HareketTipi = :shtUretimeCikis)
      into :CikisMiktar;

      if (CikisMiktar is Null) then CikisMiktar = 0;
      if (PlanlananMiktar is Null) then PlanlananMiktar = 0;

      KalanMiktar = PlanlananMiktar - GerceklesenMiktar;

      for Select DepoId, DepoAdi, DepoStokMevcudu from UrunDepoStokGetir(:UrunID, :ADepoId) where (HGYariMamul = 1 or :UrunTipiId = :utAksesuar) into :DepoID, :DepoAdi, :StokMevcudu do
      begin
        suspend;
      end
    end
    else
    begin
      For Select Distinct HGIsEmri.UretimPartiNo from HGIsEmri
            Left Join HGIsEmriPastal on HGIsEmriPastal.HgIsEmriKayitId = HGIsEmri.HgIsEmriKayitId
            Left Join HGIsEmriPastalUrun on HGIsEmriPastalUrun.HGIsEmriPastalId = HGIsEmriPastal.HGIsEmriPastalId
          where HGIsEmri.HGSiparisDetayId = :HGSiparisDetayId and HGIsEmri.KesimTalimati = 1 and
                HGIsEmriPastalUrun.UrunID = :UrunId and HGIsEmriPastalUrun.Gerceklesenmiktar > 0
      into :PastalNo do
      begin
        Select Abs(Sum(Miktar * GirisCikis)) from StokDetay
          where StokDetay.HGIsEmriKayitID = :HGIsEmriKayitID and
                StokDetay.HGIsEmriUrunID = :HGIsEmriUrunID and
                StokDetay.UrunID = :UrunID and
                (StokDetay.HareketTipi = :shtFasonUretimeCikis or StokDetay.HareketTipi = :shtUretimeCikis) and
                StokDetay.PastalNo = :PastalNo
        into :CikisMiktar;

        if (CikisMiktar is Null) then CikisMiktar = 0;
        if (PlanlananMiktar is Null) then PlanlananMiktar = 0;

        KalanMiktar = PlanlananMiktar - GerceklesenMiktar;
        For Select Depo.DepoId, Depo.Adi, Sum(Miktar * GirisCikis) from StokDetay
          Left Join Depo on Depo.DepoID = StokDetay.DepoID
        where StokDetay.DepoId = :ADepoId and StokDetay.UrunID = :UrunID and StokDetay.PastalNo = :PastalNo and HGYariMamul = 1
        Group By Depo.DepoId, Depo.Adi
        into :DepoID, :DepoAdi, :StokMevcudu do
        begin
          suspend;
        end
      end
    end
  end
end
