CREATE PROCEDURE HGISEMRIURETIMISLEMIGETIR(
    AHGISEMRIKAYITID INTEGER,
    AKESIMTALIMATI INTEGER)
RETURNS (
    URETIMISLEMLERIID VARCHAR(100),
    URETIMISLEMLERIADI VARCHAR(400))
AS
declare variable FModelUretimIslemTipiID ModelGrupKoduSize;
declare variable FModelUretimIslemTipiAdi Adi20;
begin
    UretimIslemleriId = '';
    UretimIslemleriAdi = '';
    if (AKesimTalimati = 0) then
    begin
      For Select HGSipDetayUretimIslemleri.ModelUretimIslemTipiID, ModelUretimIslemTipi.Adi from HGIsEmriUretimIslemleri
            Left Join HGSipDetayUretimIslemleri on HGSipDetayUretimIslemleri.HGSipDetayUretimIslemleriID = HGIsEmriUretimIslemleri.HGSipDetayUretimIslemleriID
            Left Join ModelUretimIslemTipi on ModelUretimIslemTipi.ModelUretimIslemTipiID = HGSipDetayUretimIslemleri.ModelUretimIslemTipiID
          where HGIsEmriUretimIslemleri.HGIsEmriKayitID = :AHGIsEmriKayitID
          into :FModelUretimIslemTipiID, :FModelUretimIslemTipiAdi do
      begin
        if (Char_Length(FModelUretimIslemTipiID) + Char_Length(UretimIslemleriId) < 100) then
        begin
          if (UretimIslemleriId <> '' ) then UretimIslemleriId = UretimIslemleriId || cast(', ' as varchar(2) character set UTF8);
          UretimIslemleriId = UretimIslemleriId || FModelUretimIslemTipiID;
        end

        if (Char_Length(FModelUretimIslemTipiAdi) + Char_Length(UretimIslemleriAdi) < 400) then
        begin
          if (UretimIslemleriAdi <> '' ) then UretimIslemleriAdi = UretimIslemleriAdi || cast(', ' as varchar(2) character set UTF8);
          UretimIslemleriAdi = UretimIslemleriAdi || FModelUretimIslemTipiAdi;
        end
      end
    end
    else
    begin
      UretimIslemleriId = 'KSM';
      UretimIslemleriAdi = 'Kesim';
    end

    suspend;
end
