CREATE PROCEDURE DOKUMAURETIMRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AURETIMISLEMTIPIID INTEGER,
    ABASLANGICMAKINAID VARCHAR(10),
    ABITISMAKINAID VARCHAR(10),
    AMAKINAKODUARALIKLI INTEGER,
    AMAKINAGRUPID VARCHAR(4),
    AHAMID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ORMEFISID INTEGER,
    ORMEIRSALIYEID INTEGER,
    ORMEFATURAID INTEGER,
    HAMID INTEGER,
    TEVKIFAT INTEGER,
    STOKHAREKETID INTEGER,
    HAREKETTIPI INTEGER,
    SINIF VARCHAR(1),
    TARIH TIMESTAMP,
    URETIMISLEMTIPISTR VARCHAR(25),
    URETIMISLEMTIPIID INTEGER,
    FISSERINO VARCHAR(2),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(50),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(50),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    SIPARISSIRANO INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    MIKTAR DOUBLE PRECISION,
    HAMMIKTAR DOUBLE PRECISION,
    TOPLAMBOYAYACIKIS DOUBLE PRECISION,
    DOKUMAMAKINAID VARCHAR(10),
    DOKUMAMAKINAADI VARCHAR(40),
    MAKINAGRUPID VARCHAR(4),
    MAKINAGRUPADI VARCHAR(30),
    MAKINAMARKASI VARCHAR(30),
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    AYARTOPU INTEGER,
    AYARTOPUSTR VARCHAR(10),
    URETIMPARTINO VARCHAR(30),
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIADI VARCHAR(80),
    KUMASCINSIADI VARCHAR(40),
    KULLANILANLOTLAR VARCHAR(255),
    LOT VARCHAR(150),
    ATKILOTU VARCHAR(30),
    COZGULOTU VARCHAR(30),
    ORMEBIRIMMALIYET DOUBLE PRECISION,
    ORMEBIRIMMALIYETDOVIZID VARCHAR(5),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    DOKUMASIPARISTIPIID INTEGER,
    DOKUMASIPARISTIPISTR VARCHAR(30),
    DOKUMAPLANLAMAID INTEGER,
    COZGUISEMRIID VARCHAR(30),
    HASILISEMRIID VARCHAR(30),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    HAMSATISTRANSFERKODU VARCHAR(55),
    FASONTRANSFERKODU VARCHAR(55),
    FASONORMETRANSFERKODU VARCHAR(55),
    FASONARTISATORMETRANSFERKODU VARCHAR(55),
    ALISTRANSFERKODU VARCHAR(55),
    RENKLIALISTRANSFERKODU VARCHAR(55),
    PLANLAMAMIKTARGUN DOUBLE PRECISION,
    VARDIYA INTEGER,
    LEVENTNO INTEGER,
    DEPOPARTINO VARCHAR(30),
    TELADEDI INTEGER,
    BITENBOBIN INTEGER,
    SARIMHATALIBOBINSAYISI INTEGER,
    URETIMTARIHI TIMESTAMP,
    VISCOSITY DOUBLE PRECISION,
    REFACTO INTEGER,
    BOLGESELSIKLIK EGEBOOLEAN,
    ATKIIPLIKCM DOUBLE PRECISION,
    INDIGOISEMRIID VARCHAR(30))
AS
declare variable uitDokuma integer;
declare variable uitHasil integer;
declare variable uitCozgu integer;

declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable Okey integer;
declare variable FisSinif varchar(1);
declare variable IrsaliyeSinif varchar(1);
declare variable FaturaSinif varchar(1);
declare variable FLot varchar(150);
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim from OrmeStokHareketTipiGetir
         into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
              :shtFasonUretim, :ShtUretim;

  select uitDokuma, uitCozgu, uitHasil from UretimIslemTipiGetir into :uitDokuma, :uitCozgu, :uitHasil;

  For Select StokHareket.StokHareketId, StokHareket.OrmeFisId, StokHareket.OrmeIrsaliyeId, StokHareket.OrmeFaturaId,
             StokHareket.HareketTipi, StokHareket.Tarih, UretimIslemTipi.Adi, StokHareket.UretimIslemTipiID, OrmeFis.SeriNo, OrmeFis.FisNo, OrmeFis.FisTarihi,
             OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeFatura.SeriNo, OrmeFatura.FaturaNo, OrmeFatura.FaturaTarihi,
             StokHareket.FirmaId, SHFirma.Adi, StokHareket.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             StokHareket.HamId, Ham.Adi, Renk.Adi, StokHareket.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
             StokHareket.Miktar, StokHareket.DokumaMakinaId,
             DokumaMakina.Adi, DokumaMakina.MakinaGrupId, MakinaGrup.Adi, DokumaMakina.Markasi,
             StokHareket.UretimPartiNo, StokHareket.InsertTarihi,
             StokHareket.DegisimTarihi, IK.Adi, SK.Adi, OrmeFis.Sinif, OrmeIrsaliye.Sinif, OrmeFatura.Sinif,
             BoyaSiparis.FirmaId, SiparisFirma.Adi, StokHareket.BoyaSiparisDetayId, StokHareket.OrmeSiparisDetayId,
             BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, OrmeSiparisDetay.SiparisSiraNo,
             IplikCinsi.Adi, Kalinlik.Adi, KumasCinsi.Adi, StokHareket.Sakat, StokHareket.AyarTopu, OrmeSiparisDetay.OrmeBirimMaliyet, OrmeSiparisDetay.OrmeBirimMaliyetDovizId,
             StokHareket.DokumaPlanlamaId, BoyaSiparis.Fason, StokHareket.Lot, OrmeSiparisDetay.HamMiktar, BoyaSiparis.DokumaSiparisTipiId, DokumaSiparisTipi.Adi,
             Ham.HamSatisTransferKodu, Ham.FasonTransferKodu, Ham.FasonOrmeTransferKodu, Ham.FasonArtiSatOrmeTransferKodu, Ham.AlisTransferKodu, Ham.RenkliAlisTransferKodu, CozguIsemri.IsemriID, HasilIsemri.IsemriID,
             DokumaPlanlama.MiktarGun, StokHareket.Vardiya, StokHareket.LeventNo, StokHareket.DepoPartiNo, StokHareket.TelAdedi, StokHareket.AtkiLotu, StokHareket.CozguLotu, StokHareket.BitenBobin, StokHareket.SarimHataliBobinSayisi, StokHareket.UretimTarihi,
             StokHareket.Viscosity, StokHareket.Refacto, Ham.BolgeselSiklik, Ham.AtkiIplikCm, IndigoIsEmri.IsEmriId from StokHareket
    left join Ham on Ham.HamId = StokHareket.HamId
    left join Renk on Renk.RenkId = StokHareket.RenkId
    left join Desen on Desen.DesenId = StokHareket.DesenId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = StokHareket.DesenVaryantId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = StokHareket.BoyaSiparisDetayId
    left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
    left join OrmeFatura on OrmeFatura.OrmeFaturaId = StokHareket.OrmeFaturaId
    left join Firma SHFirma on SHFirma.FirmaId = StokHareket.FirmaId
    left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = StokHareket.DokumaMakinaId
    left join MakinaGrup on MakinaGrup.MakinaGrupId = DokumaMakina.MakinaGrupId
    left join UretimIslemTipi on UretimIslemTipi.UretimIslemTipiId  = StokHareket.UretimIslemTipiId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    Left Join KumasCinsi ON KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
    left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
    left join Kullanici IK on IK.KullaniciId  = StokHareket.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = StokHareket.KullaniciId
    left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 On SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join DokumaSiparisTipi On DokumaSiparisTipi.DokumaSiparisTipiId = BoyaSiparis.DokumaSiparisTipiId
    Left join CozguIsemri on CozguIsemri.CozguIsemriKayitId = StokHareket.CozguIsemriKayitId
    Left join HasilIsemri on HasilIsemri.HasilIsemriKayitId = StokHareket.HasilIsemriKayitId
    Left join IndigoIsemri on IndigoIsemri.IndigoIsemriKayitId = StokHareket.IndigoIsemriKayitId
    Left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = StokHareket.DokumaPlanlamaId
    where StokHareket.Tarih >= :ABaslangicTarihi and StokHareket.Tarih <= :ABitistarihi and
         ((StokHareket.HareketTipi = :ShtUretim or StokHareket.HareketTipi = :ShtFasonUretim) and StokHareket.HamIade = 0) and
         (StokHareket.UretimIslemTipiId = :AUretimIslemTipiId or (:AUretimIslemTipiId is null and StokHareket.UretimIslemTipiId >= :uitDokuma)) and
         ((:AMakinaKoduAralikli = 1 and StokHareket.DokumaMakinaId >= :ABaslangicMakinaId and (StokHareket.DokumaMakinaId <= :ABitisMakinaId or :ABitisMakinaId = '')) or
          (:AMakinaKoduAralikli = 0) or (:AMakinaKoduAralikli = 1 and :ABaslangicMakinaId = '' and :ABitisMakinaId = '')) and
         (DokumaMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
         (StokHareket.HamId = :AHamId or :AHamId is null)
  into :StokHareketId, :OrmeFisId, :OrmeIrsaliyeId, :OrmeFaturaId, :HareketTipi, :Tarih, :UretimIslemTipiStr, :UretimIslemTipiID, :FisSeriNo, :FisNo, :FisTarihi, :IrsaliyeSeriNo,
       :IrsaliyeNo, :IrsaliyeTarihi, :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :FirmaId, :FirmaAdi, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :HamId,
       :HamAdi, :RenkAdi, :DesenId, :DesenAdi, :VaryantAdi, :Miktar, :DokumaMakinaId, :DokumaMakinaAdi, :MakinaGrupId, :MakinaGrupAdi,
       :MakinaMarkasi,
       :UretimPartiNo, :InsertTarihi, :DegisimTarihi,
       :InsertKullaniciAdi, :KullaniciAdi, :FisSinif, :IrsaliyeSinif, :FaturaSinif,
       :SiparisFirmaId, :SiparisFirmaAdi, :BoyaSiparisDetayId, :OrmeSiparisDetayId, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id,
       :SiparisGrup2Adi, :SiparisSiraNo, :IplikCinsiAdi, :KalinlikAdi, :KumasCinsiAdi, :Sakat, :AyarTopu,
       :OrmeBirimMaliyet, :OrmeBirimMaliyetDovizId, :DokumaPlanlamaId, :Fason, :Lot, :HamMiktar, :DokumaSiparisTipiId, :DokumaSiparisTipiStr,
       :HamSatisTransferKodu, :FasonTransferKodu, :FasonOrmeTransferKodu, :FasonArtiSatOrmeTransferKodu, :AlisTransferKodu, :RenkliAlisTransferKodu,
       :CozguIsemriID, :HasilIsemriID, :PlanlamaMiktarGun, :Vardiya, :LeventNo, :DepoPartiNo, :TelAdedi, :AtkiLotu, :CozguLotu, :BitenBobin, :SarimHataliBobinSayisi,
       :UretimTarihi, :Viscosity, :Refacto, :BolgeselSiklik, :AtkiIplikCm, :IndigoIsEmriID do
  begin
    Okey = 1;

    if (FaturaSinif is not null) then Sinif = FaturaSinif;
    else if (IrsaliyeSinif is not null) then Sinif = IrsaliyeSinif;
    else if (FisSinif is not null) then Sinif = FisSinif;

    if (okey = 1) then
    if (AMakinaKoduAralikli = 0) then
    if (KutuKarsilastir(ABaslangicMakinaId, DokumaMakinaId) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      KullanilanLotlar = '';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      if (Sakat = 1) then SakatStr = 'Sakat';
      else SakatStr = '';

      if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
      else AyarTopuStr = '';

      if (DokumaPlanlamaId is not null) then
      begin
        For Select OrmeRaf.Lot from StokHareket
          left join OrmeRaf on OrmeRaf.OrmeRafID = StokHareket.OrmeRafID
          where StokHareket.DokumaPlanlamaId = :DokumaPlanlamaId and
               (StokHareket.HareketTipi = :ShtUretimeCikis or StokHareket.HareketTipi = :ShtFasonUretimeCikis)
          group by OrmeRaf.Lot
          into :FLot do
        begin
          if (FLot is not Null) then
          begin
            if (StrLength(KullanilanLotlar) + StrLength(FLot) < 240) then
            begin
              if (KullanilanLotlar <> '') then KullanilanLotlar = KullanilanLotlar || ',';
              KullanilanLotlar = KullanilanLotlar || FLot;
            end
          end
        end
      end

      if (Fason = 1) then
        FasonStr = 'Fason';
      else
        FasonStr = 'Satış';

      suspend;
    end
  end
end
