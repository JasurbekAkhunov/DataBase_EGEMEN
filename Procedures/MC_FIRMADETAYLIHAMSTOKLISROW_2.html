CREATE PROCEDURE MC_FIRMADETAYLIHAMSTOKLISROW_2(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE,
    ATAMIR INTEGER,
    ARAPORHAMTIPI INTEGER,
    ATAMIRTIPIID INTEGER,
    AHAMID INTEGER,
    ALOT VARCHAR(150),
    AKULLANICIID INTEGER,
    AHAMSTOKDETAYID INTEGER)
RETURNS (
    HAMSTOKDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ISLEMGRUBUADI VARCHAR(20),
    ISLEMTIPIADI VARCHAR(20),
    MODELNO VARCHAR(20),
    MIKTAR NUMERIC(12,3),
    IRSALIYEMIKTAR NUMERIC(12,3),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISTIPIID INTEGER,
    LABRECETEKODU VARCHAR(20),
    RENKADI VARCHAR(40))
AS
declare variable FIslemTipi       integer;
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable Okey     integer;
declare variable BoyaSiparisDetayId  integer;
declare variable hstIplik               integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  Select Okey from MC_FirmaKullaniciKontrolu(:AFirmaId, :AKullaniciId) into :Okey;
  if (Okey <> 1) then exit;

  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  for select HSD.HamStokDetayId, Parti.PartiYil, Parti.PartiId, Parti.PartiKayitId, Parti.ReceteId, Recete.Adi, IslemGrubu.Adi,
        IslemTipi.Adi, Parti.ModelNo, Parti.IslemTipiId,
        BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.SiparisTipiId,
        (select sum(Miktar) from Partistok where Partistok.PartiKayitId = Parti.PartiKayitId),
        (select sum(PSD.Miktar) from RafDetay RD3
           left join PartiStokDetay PSD on PSD.rafDetayId = Rd3.RafDetayId and PSD.PartiKayitId = Parti.PartiKayitId
           where RD3.HamStokDetayId = HSD.HamStokDetayId
        ) from HamStokIrsaliye
  left join HamStokDetay HSD on HSD.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HSD.HamStokMasterId
  left join DisTamir on DisTamir.HamStokMasterId = HSD.HamStokMasterId
  left join Ham on Ham.HamId = HSD.HamId
  left join RafDetay on RafDetay.HamStokDetayId = HSD.HamStokDetayId
  left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
  left join Parti on Parti.PartiKayitId = PartiStokDetay.PartiKayitId
  left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  where (HSD.HamStokDetayId = :AHamStokDetayId or :AHamStokDetayId is Null) and
         HamStokIrsaliye.FirmaId = :AFirmaId and HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and
        (HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
         Parti.PartiKayitId is not null and   (HamStokMaster.Tamir = :ATamir or :ATamir is null) and
        (DisTamir.TamirTipiID = :ATamirTipiId or :ATamirTipiId is null) and
        (HSD.HamId = :AHamId or :AHamId is null) and (HSD.Fason = 1) and
        ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporHamTipi = 0) or (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporHamTipi = 1) or :ARaporHamTipi is null) and
        (HamStokMaster.Lot = :ALot or :ALot is null)
  group By HSD.HamStokDetayId, Parti.PartiYil, Parti.PartiId, Parti.PartiKayitId, Parti.ReceteId, Recete.Adi, IslemGrubu.Adi,
       IslemTipi.Adi, Parti.ModelNo, Parti.IslemTipiId,
       BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.SiparisTipiId
  order by Parti.PartiKayitId
   into :HamStokDetayId, :PartiYil, :PartiId, :PartiKayitId, :ReceteId, :ReceteAdi, :IslemGrubuAdi,
        :IslemTipiAdi, :ModelNo, :FIslemTipi,
        :BoyaSiparisYil, :BoyaSiparisId, :SiparisTipiId, :Miktar, :IrsaliyeMiktar do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';


    BoyaSiparisDetayId = Null;
    Select Min(BoyaSiparisDetayId) from PartiStok where PartiStok.PartiKayitId = :PartiKayitId into :BoyaSiparisDetayId;

    Select LabReceteKodu, RenkAdi from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = :BoyaSiparisDetayId into :LabReceteKodu, :RenkAdi;
    suspend;
  end
end
