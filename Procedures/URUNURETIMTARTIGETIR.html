CREATE PROCEDURE URUNURETIMTARTIGETIR(
    AURUNURETIMID INTEGER)
RETURNS (
    URUNURETIMTARTIID INTEGER,
    URUNID VARCHAR(20),
    MIKTAR DOUBLE PRECISION,
    RECETEMIKTAR DOUBLE PRECISION,
    PLANLANANMIKTAR DOUBLE PRECISION,
    URUNBIRIMID INTEGER,
    URUNADI ADI80,
    BARKOD VARCHAR(255),
    URUNRAF VARCHAR(15))
AS
declare variable FUrunMiktar double precision;
declare variable FMiktar double precision;
begin

  for select UrunUretimTarti.UrunUretimTartiId, UrunUretimTarti.UrunId, UrunDefaultBirim.UrunBirimId, Urun.Adi,
             UrunUretimTarti.Miktar, UrunUretimTarti.ReceteMiktar, UrunUretimTarti.PlanlananMiktar, Urun.Barkod, Urun.Raf from UrunUretimTarti
    left join UrunBirim RUB on RUB.urunBirimId = UrunUretimTarti.urunBirimId
    left join UrunBirim UrunDefaultBirim on UrunDefaultBirim.urunId = UrunUretimTarti.UrunId and UrunDefaultBirim.DefaultBirim = 1
    left join Urun on Urun.UrunId = UrunUretimTarti.UrunId
    left join UrunFiyat on UrunFiyat.UrunId = UrunUretimTarti.UrunId
    left join UrunBirim UBUCF on UBUCF.urunBirimId = UrunFiyat.UretimeCikisFiyatiUrunBirimId
    where UrunUretimTarti.UrunUretimId = :AUrunUretimId
    into :UrunUretimTartiId,:UrunId, :UrunBirimId, :UrunAdi, :Miktar, :ReceteMiktar, :PlanlananMiktar, :Barkod, :UrunRaf do
  begin
      Suspend;
  end
end
