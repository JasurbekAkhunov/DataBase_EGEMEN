CREATE PROCEDURE PARTIASAMAGIRISIIPTALET(
    APARTIASAMASTOKDETAYID INTEGER,
    AKULLANICIID INTEGER,
    AGIRISCIKIS INTEGER)
RETURNS (
    PARTIHAREKETLOGID INTEGER)
AS
declare variable CikisTarihi    timestamp;
declare variable Sira           integer;
declare variable KayitSayisi    integer;
declare variable PartiAsamalariId integer;
declare variable PartiSinifId     integer;

declare variable LogPartiYil              integer;
declare variable LogPartiId               integer;
declare variable LogHamAdiToplami         varchar(255);
declare variable LogAsamaId               integer;
declare variable LogReceteId              varchar(20);
declare variable LogBoyaSiparisYil        integer;
declare variable LogBoyaSiparisId         integer;
declare variable LogBoyaSiparisDetayId    integer;
declare variable LogMakinaId              varchar(10);
declare variable LogKilo                  Numeric(12,3);
declare variable LogGirisPersonelId  integer;
declare variable LogGirisTarihi           timestamp;
declare variable FHamSTokMasterId    integer;
declare variable HamSTokMasterIdStr     varchar(100);
begin
  Select PartiAsamaStokDetay.PartiSinifId, PartiAsamalari.PartiAsamalariId, PartiAsamalari.Sira, PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.CikisTarihi,
         PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.MakinaId, PartiAsamaStokDetay.GirisPersonelId, PartiAsamalari.AsamaId, PartiAsamalari.ReceteId  from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    where PartiAsamaStokDetay.PartiAsamaStokDetayId = :APartiAsamaStokDetayId
    into :PartiSinifId, :PartiAsamalariId, :Sira, :LogGirisTarihi, :CikisTarihi, :LogKilo, :LogMakinaId, :LogGirisPersonelId, :LogAsamaId, :LogReceteId;

  if (PartiSinifId is null) then exception AsamaStokDetayKaydiBulunamadi;

  if (CikisTarihi is not null) then exception CikisVarGirisSilinemez;

  Select Count(*) from PartiStokSonKontrol where PartiStokSonKontrol.PartiAsamalariId = :PartiAsamalariId into :KayitSayisi;
  if (KayitSayisi > 0) then exception BuradaGirisCikisIptalEdilemez;

  Select Count(*) from PartiStokFizikKontrol where PartiStokFizikKontrol.PartiAsamalariId = :PartiAsamalariId into :KayitSayisi;
  if (KayitSayisi > 0) then exception BuradaGirisCikisIptalEdilemez;

  Select Count(*) from PSKaliteKontrolAsama where PSKaliteKontrolAsama.PartiAsamalariId = :PartiAsamalariId into :KayitSayisi;
  if (KayitSayisi > 0) then exception BuradaGirisCikisIptalEdilemez;

  Select Parti.PartiYil, Parti.PartiId, PartiStok.HamAdiToplami, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
         PartiStok.BoyaSiparisDetayId from PartiStok
    Left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    Left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where PartiStok.PartiSinifId = :PartiSinifId
    into :LogPartiYil, :LogPartiId, :LogHamAdiToplami, :LogBoyaSiparisYil, :LogBoyaSiparisId, :LogBoyaSiparisDetayId;


/* sonra yapmaya basladıgı asama varmı */
  Select Count(*) from PartiAsamaStokDetay
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    where PartiAsamaStokDetay.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira > :Sira and PartiAsamaStokDetay.GirisTarihi is not null
    into :KayitSayisi;

/* sonra yapmaya basladıgı asama var ise  */
  if (KayitSayisi > 0) then
  begin
    /* Bolunerekmi yapılmıs bakıyorum, eger oyle ise girisi silebilir, aksi halde sonra yapılmıs asamasi var  expception */
    Select count(*) from PartiAsamaStokDetay
      where PartiAsamaStokDetay.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId and
            PartiAsamaStokDetay.PartiAsamaStokDetayId <> :APartiAsamaStokDetayId and PartiAsamaStokDetay.GirisTarihi is not null
      into :KayitSayisi;

    if (KayitSayisi = 0) then exception SonraYapilmisAsamasiVarGiris;
    else Delete from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiAsamaStokDetayId = :APartiAsamaStokDetayId; /* bolenerek basladıgı icin kaydı siliyoruz */
  end
  else
  begin
/* sonra yapmaya basladıgı asama yok ise  */
    Select count(*) from PartiAsamaStokDetay
      where PartiAsamaStokDetay.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId and
            PartiAsamaStokDetay.PartiAsamaStokDetayId <> :APartiAsamaStokDetayId and PartiAsamaStokDetay.GirisTarihi is not null
      into :KayitSayisi;   /* Bolunerekmi yapılmıs bakıyorum, eger oyle ise giris kaydı tamamen silinecek, aksi halde sonra update  */

    if (KayitSayisi = 0) then Update PartiAsamaStokDetay Set GirisTarihi = Null, GirisPersonelId = Null, GirisOtomatik = null where PartiAsamaStokDetay.PartiAsamaStokDetayID = :APartiAsamaStokDetayID;
    else Delete from PartiAsamaStokDetay where PartiAsamaStokDetay.PartiAsamaStokDetayId = :APartiAsamaStokDetayId; /* bolenerek basladıgı icin kaydı siliyoruz */
  end

  HamSTokMasterIdStr = '';

  for select HamStokDetay.HamStokMasterId from PartiStokDetay
    left join BoyaSiparisRezerv on BoyaSiparisRezerv.BoyaSiparisRezervId = PartiStokDetay.BoyaSiparisRezervId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
    where PartiStokDetay.PartiSinifId = :PartiSinifId
    group by HamStokDetay.HamStokMasterId
    into :FHamStokMasterId do
  begin
    if (Char_Length(HamSTokMasterIdStr) + Char_Length(FHamStokMasterId) < 100) then
    begin
     if (HamSTokMasterIdStr <> '' ) then HamSTokMasterIdStr = HamSTokMasterIdStr || cast(', ' as varchar(2) character set UTF8);
     HamSTokMasterIdStr = HamSTokMasterIdStr || FHamStokMasterId;
    end
  end

  if (AGirisCikis is null or AGirisCikis <> 1) then
  begin

    PartiHareketLogId = Gen_ID(PartiHareketLog_Id_Gen, 1);

    insert into PartiHareketLog(PartiHareketLogId, PartiYil, PartiId, HamAdiToplami, AsamaId, ReceteId, BoyaSiparisYil, BoyaSiparisId, BoyaSiparisDetayId, MakinaId,
                                 Kilo, GirisPersonelId, GirisTarihi, GirisCikis, KullaniciId, HamSTokMasterIdStr)
                         Values(:PartiHareketLogId, :LogPartiYil, :LogPartiId, :LogHamAdiToplami, :LogAsamaId, :LogReceteId, :LogBoyaSiparisYil, :LogBoyaSiparisId, :LogBoyaSiparisDetayId, :LogMakinaId,
                                 :LogKilo, :LogGirisPersonelId, :LogGirisTarihi, 1, :AKullaniciId, :HamSTokMasterIdStr);
  end
  suspend;
end
