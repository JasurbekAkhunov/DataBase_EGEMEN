CREATE PROCEDURE LABRECETEKOPYALA(
    ALABRECETEID INTEGER,
    AYENILABRECETEID INTEGER,
    AYENILABRECETEKODU VARCHAR(20),
    AFIRMAID VARCHAR(20),
    ALABRECETEATISIDLER VARCHAR(255),
    AKULLANICIID INTEGER)
AS
declare variable FLabReceteCalismaId integer;
declare variable LabReceteCalismaId integer;
declare variable FLabReceteAtisId integer;
declare variable LabReceteAtisId integer;
begin
  Insert into LabRecete(LabReceteId, LabReceteKodu, Adi, FirmaId, TalepTarihi, LabReceteGrupId, LabRenkGrupId, UstAsamaId, Fason, MusteriRenkKodu,
                        HamId, DovizId, Aciklama, NotuStr, EskiLabReceteKodu, PantoneKodu, ReceteTuruId, RenkDerinligiId, Notu, FikseTipiId, InsertKullaniciId, KullaniciId, Aciklama2)
      Select :AYeniLabReceteId, :AYeniLabReceteKodu, Adi, :AFirmaId, TalepTarihi, LabReceteGrupId, LabRenkGrupId, UstAsamaId, Fason, MusteriRenkKodu,
              HamId, DovizId, Aciklama, NotuStr, EskiLabReceteKodu, PantoneKodu, ReceteTuruId, RenkDerinligiId, Notu, FikseTipiId, :AKullaniciId, :AKullaniciId, Aciklama2 from LabRecete
             where LabRecete.LabReceteId = :ALabReceteId;

    For Select distinct LabReceteCalisma.LabReceteCalismaId from LabReceteCalisma, LabReceteAtis
          where LabReceteCalisma.LabReceteID = :ALabReceteId and ReferansLabReceteAtisId is null and
                LabReceteAtis.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId and
                (:ALabReceteAtisIdler like '%*'||LabReceteAtis.LabReceteAtisId||'*%' or  :ALabReceteAtisIdler is null)
      into :FLabReceteCalismaId do
    begin
      LabReceteCalismaId = gen_ID(LabReceteCalisma_Id_gen, 1);

      Insert into LabReceteCalisma(LabReceteCalismaId, LabReceteId, FirmaId, HamId, CalismaTarihi, Termintarihi, Ilave, SokumUzeriBoya, KritikCalismasi, Notu, Bitti, InsertKullaniciId, KullaniciId, BirimFiyat, DovizId)
        Select :LabReceteCalismaId, :AYeniLabReceteId, FirmaId, HamId, CalismaTarihi, Termintarihi, Ilave, SokumUzeriBoya, KritikCalismasi, Notu, Bitti, :AKullaniciId, :AKullaniciId, BirimFiyat, DovizId from LabReceteCalisma
        where LabReceteCalisma.LabReceteCalismaID = :FLabReceteCalismaID;

      For Select LabReceteAtisId from LabReceteAtis
              where LabReceteAtis.LabReceteCalismaID = :FLabReceteCalismaId and
                   (:ALabReceteAtisIdler like '%*'||LabReceteAtis.LabReceteAtisId||'*%' or  :ALabReceteAtisIdler is null)
        into :FLabReceteAtisId do
      begin
        LabReceteAtisId = gen_ID(LabReceteAtis_Id_gen, 1);

        Insert into LabReceteAtis(LabReceteAtisId, LabReceteCalismaId, LabReceteId, AtisNo, GidisTarihi, Okey, Kritik, Migrasyon, OnayKullaniciId, DurumuStr, Tarihi, BoyaMaliyeti,
                                BoyaYuzdesi, Notu, NotuStr, RenkDerinligiId, Gitti, Termin, Color, ReceteAsamaId, InsertKullaniciId, KullaniciId)
         Select :LabReceteAtisId, :LabReceteCalismaId, :AYeniLabReceteId, AtisNo, current_date, Okey, Kritik, Migrasyon, OnayKullaniciId, DurumuStr, Tarihi, BoyaMaliyeti,
                BoyaYuzdesi, Notu, NotuStr, RenkDerinligiId, Gitti, Termin, Color, ReceteAsamaId, :AKullaniciId, :AKullaniciId from LabReceteAtis where LabReceteAtis.LabReceteAtisID = :FLabReceteAtisID;

        Insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, InsertKullaniciId, KullaniciId)
        Select :LabReceteAtisId, :AYeniLabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, :AKullaniciId, :AKullaniciId from LabReceteUrun
           where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId
           order by LabReceteUrun.LabReceteUrunId, LabReceteUrun.Sira;
      end
    end
end
