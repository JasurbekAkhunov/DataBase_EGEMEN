CREATE PROCEDURE DOKUMAPLANLAMACOZGUSTOKLISTESI(
    ADOKUMAPLANLAMAID INTEGER,
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    RAFID RAFKODUSIZE,
    URETICIADI ADI40,
    STOKGIRISNO INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    RENKID INTEGER,
    RENKADI ADI50,
    FASONSTOK EGEBOOLEAN,
    DEPOID DEPOKODUSIZE,
    DEPOADI ADI30,
    STOKFIRMAID FIRMAKODUSIZE,
    STOKFIRMAADI ADI100,
    STOKFIRMAADI2 ADI100,
    LEVENTNO INTEGER,
    TELADEDI INTEGER,
    ORMERAFID INTEGER,
    ISEMRIID VARCHAR(15),
    PLANLANANMIKTAR BOYAMIKTARSIZE,
    PLANLANABILIRMIKTAR BOYAMIKTARSIZE,
    DOKUMASTOKPLANLAMAID INTEGER,
    ATKILOTU VARCHAR(30),
    LOT VARCHAR(150),
    LEVENTTIPIID INTEGER,
    LEVENTTIPIADI ADI40,
    HESAPLANANURETIM BOYAMIKTARSIZE,
    CIKISYAPANAKULLANICIID INTEGER,
    CIKISYAPANKULLANICIADI ADI40,
    CIKISTARIHI TIMESTAMP,
    BARKODID VARCHAR(30),
    DOKUMAPLANLAMAID INTEGER,
    STOKGIRISTARIHI TIMESTAMP,
    STOKGIRISIYAPANKULLANICIID INTEGER,
    STOKGIRISIYAPANKULLANICIADI ADI40,
    STOKTAKALANMIKTAR BOYAMIKTARSIZE,
    DOKUMAPLANLAMARECETESIRANO INTEGER)
AS
declare variable FPlanlananMiktar BoyaMiktarSize;
declare variable FPlanlamaIcinCikisMiktari BoyaMiktarSize;
declare variable FNetCikisMiktari BoyaMiktarSize;
declare variable FNetGirisMiktari BoyaMiktarSize;
declare variable shtFasonUretimeCikis smallint;
declare variable shtUretimeCikis smallint;
declare variable FMiktar Boyamiktarsize;
declare variable FBarkodId varchar(30);
declare variable DokumaBarkodluGirisCikisTakibi integer;
begin
  select shtFasonUretimeCikis,  shtUretimeCikis from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis, :shtUretimeCikis;

  for select OrmeRaf.RafId, Uretici.Adi UreticiAdi, OrmeRaf.StokGirisNo, OrmeRaf.HamId, Ham.Adi HamAdi, OrmeRaf.RenkId, Renk.Adi RenkAdi, OrmeRaf.FasonStok,
       DepoRaf.DepoId, Depo.Adi DepoAdi, OrmeRaf.StokFirmaId, StokFirma.Adi StokFirmaAdi, StokFirma.Adi2 StokFirmaAdi2,
       OrmeRaf.LeventNo, OrmeRaf.TelAdedi, OrmeRaf.AtkiLotu, OrmeRaf.Lot, OrmeRaf.OrmeRafId, CozguIsEmri.IsEmriId, DokumaStokPlanlama.Miktar, DokumaStokPlanlama.DokumaStokPlanlamaId,
       OrmeRaf.NetGirisMiktari, OrmeRaf.NetCikisMiktari, LeventTipi.LeventTipiId, LeventTipi.Adi, DokumaStokPlanlama.DokumaPlanlamaId, OrmeRaf.BarkodId,
       OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari, DokumaStokPlanlama.DokumaPlanlamaReceteSiraNo
    from DokumaStokPlanlama
    join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaStokPlanlama.DokumaPlanlamaId
    join OrmeRaf on DokumaStokPlanlama.OrmeRafId = OrmeRaf.OrmeRafId
    left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
    left join Depo on Depo.DepoId = DepoRaf.DepoId
    left join Ham on Ham.HamId = OrmeRaf.HamId
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join Firma StokFirma on StokFirma.Firmaid = OrmeRaf.StokFirmaid
    left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiID
    left join CozguIsEmri on CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId
    left join LeventTipi on LeventTipi.LeventTipiId = OrmeRaf.LeventTipiId
    where (DokumaStokPlanlama.DokumaPlanlamaId = :ADokumaPlanlamaId or :ADokumaPlanlamaId is null) and
          (DokumaPlanlama.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null)
    into :RafId, :UreticiAdi, :StokGirisNo, :HamId, :HamAdi, :RenkId, :RenkAdi,
      :FasonStok, :DepoId, :DepoAdi, :StokFirmaId, :StokFirmaAdi, :StokFirmaAdi2, :LeventNo, :TelAdedi, :AtkiLotu, :Lot, :OrmeRafId, :IsEmriId, :PlanlananMiktar, :DokumaStokPlanlamaId,
      :FNetGirisMiktari, :FNetCikisMiktari, :LeventTipiId, :LeventTipiAdi, :DokumaPlanlamaId, :BarkodId, :StoktaKalanMiktar, :DokumaPlanlamaReceteSiraNo do
    begin
       FPlanlamaIcinCikisMiktari = 0;

       for select StokHareket.Miktar, Kullanici.KullaniciId, Kullanici.Adi, StokHareket.InsertTarihi from StokHareket
         left join Kullanici on Kullanici.KullaniciId = StokHareket.InsertKullaniciId
         where StokHareket.DokumaHareketi = 1 and
         StokHareket.HareketTipi = :shtUretimeCikis and
         StokHAreket.OrmeRafId = :OrmeRafId into :FMiktar, :CikisYapanaKullaniciId, :CikisYapanKullaniciAdi, :CikisTarihi do
       begin
         FPlanlamaIcinCikisMiktari = FPlanlamaIcinCikisMiktari + FMiktar;
       end

       select StokHareket.InsertTarihi, Kullanici.KullaniciId, Kullanici.Adi from StokHareket
       left join Kullanici on Kullanici.KullaniciId = StokHareket.InsertKullaniciId
       where StokHareket.StokGirisNo = :StokGirisNo into :StokGirisTarihi, :StokGirisiYapanKullaniciId, :StokGirisiYapanKullaniciAdi;

       FBarkodId = Null;
       Select Substring(List(StokHareketDetay.Barkod, ',') From 1 for 30) from StokHareket
        Left Join StokHareketDetay on StokHareketDetay.Stokhareketid = StokHareket.Stokhareketid
        where StokGirisNo = :StokGirisNo into :FBarkodId;

      if ((FBarkodId is not Null) and (DokumaBarkodluGirisCikisTakibi =1)) then BarkodId = FBarkodId;

       select coalesce(sum(DokumaStokPlanlama.Miktar),0) from DokumaStokPlanlama where DokumaStokPlanlama.OrmeRafId = :OrmeRafId into :FPlanlananMiktar;

       PlanlanabilirMiktar = FNetGirisMiktari - FNetCikisMiktari + FPlanlamaIcinCikisMiktari - FPlanlananMiktar;

       select HesaplananUretim from DokumaHesaplananUretim(:ADokumaPlanlamaId, :OrmeRafId) into :HesaplananUretim;

       suspend;
    end
end
