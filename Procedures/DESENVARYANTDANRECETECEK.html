CREATE PROCEDURE DESENVARYANTDANRECETECEK(
    APARTIKAYITID INTEGER)
AS
declare variable DesenVaryantID integer;
declare variable PartiSablonId Integer;
declare variable DesenVaryantSablonID Integer;
declare variable SablonNo Integer;
declare variable Acilim1 Double Precision;
declare variable Acilim2 Double Precision;
declare variable DepoId varchar(10);
declare variable EmulsiyonUrunId varchar(20);
begin
  delete from PartiSablonRecete where PartiSablonRecete.PartiKayitId = :APartiKayitId;
  delete from PartiSablon where PartiSablon.PartiKayitId = :APartiKayitId;

  Select DesenVaryantID from Parti where Parti.PartiKayitId = :APartiKayitId into :DesenVaryantID;


  for Select SablonNo, Acilim1, Acilim2, DepoId,  EmulsiyonUrunId, DesenVaryantSablonID from DesenVaryantSablon
     where DesenVaryantSablon.DesenVaryantID = :DesenvaryantId
     into :SablonNo, :Acilim1, :Acilim2, :DepoId,  :EmulsiyonUrunId, :DesenVaryantSablonID do
  begin
    PartiSablonId = gen_id(PartiSablon_Id_Gen, 1);

    insert into PartiSablon(PartiSablonId, PartiKayitId, SablonNo, Acilim1, Acilim2, DepoId, Sira, EmulsiyonUrunId)
                    Values (:PartiSablonId, :APartiKayitId, :SablonNo, :Acilim1, :Acilim2, :DepoId,  -1, :EmulsiyonUrunId);

    Insert into PartiSablonRecete(PartiSablonId, PartiKayitId, UrunId, UrunBirimId, Miktar)
        Select :PartiSablonId, :APartiKayitId, UrunId, UrunBirimId, Miktar from DesenVaryantSablonRecete
          where DesenVaryantSablonRecete.DesenVaryantSablonID = :DesenVaryantSablonID and
                DesenVaryantSablonRecete.UrunID is not null;


  end
end
