CREATE PROCEDURE ORMETALIMATHAREKETRAPORU(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASIPARISFIRMAID VARCHAR(20),
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AORMESIPARISSEKLIID INTEGER,
    APLANLAMAFIRMAID VARCHAR(20),
    APLANLAMAMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4),
    AMAKINAPUS INTEGER,
    AMAKINAFEINE INTEGER,
    ABITTI INTEGER,
    AKDVDAHIL INTEGER,
    AONAYDURUMU INTEGER,
    AHAMID INTEGER)
RETURNS (
    STOKHAREKETID INTEGER,
    ORMEPLANLAMAID INTEGER,
    TARIH TIMESTAMP,
    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(50),
    URETIMISLEMTIPISTR VARCHAR(25),
    URETIMISLEMTIPIID INTEGER,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FISSERINO VARCHAR(2),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    LOT VARCHAR(150),
    HAMIADE INTEGER,
    TAMIR INTEGER,
    MIKTARI DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    STOKGIRISNO INTEGER,
    STOKGIRISFIYATI DOUBLE PRECISION,
    STOKGIRISDOVIZID DOVIZSIZE)
AS
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable ShtKaydirmaCikis integer;
declare variable ShtKaydirmaGiris integer;
declare variable ShtSarf integer;
declare variable GirisCikis integer;
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, ShtKaydirmaCikis, ShtKaydirmaGiris, ShtSarf from OrmeStokHareketTipiGetir
   into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
        :shtFasonUretim, :ShtUretim, :ShtKaydirmaCikis, :ShtKaydirmaGiris, :ShtSarf;



  For select StokHareket.StokHareketID, StokHareket.HareketTipi, OrmeTalimatRaporu.OrmePlanlamaId, StokHareket.HamId, Ham.Adi, StokHareket.RenkId, Renk.Adi, StokHareket.Lot, StokHareket.MelanjKodu, StokHareket.UreticiId,
             Uretici.Adi, IK.Adi, StokHareket.InsertTarihi, SK.Adi, StokHareket.DegisimTarihi, StokHareket.Tarih, StokHareket.HamIade, StokHareket.Tamir, StokHareket.Miktar, StokHareket.UretimIslemTipiId, UretimIslemTipi.Adi,
             OrmeIrsaliye.SeriNo, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeFis.SeriNo, OrmeFis.FisNo, OrmeFis.FisTarihi, Ham.HamTipiID, StokHareket.GirisCikis, StokHareket.StokGirisNo
             from OrmeTalimatRaporu(:ARaporTipi, :ABaslangicTarihi, :ABitisTarihi, :ASiparisFirmaId, :ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ASiparisGrupId, :ASiparisGrup2Id, :AOrmeSiparisSekliId, :APlanlamaFirmaId, :APlanlamaMakinaId, :AMakinaGrupId,  :AMakinaPus, :AMakinaFeine, :ABitti, :AKdvDahil, :AOnayDurumu, :AHamId)
  left join StokHareket on StokHareket.OrmePlanlamaId = OrmeTalimatRaporu.OrmePlanlamaId
  left join Ham on Ham.HamId = StokHareket.HamId
  left join Renk on Renk.RenkId = StokHareket.RenkId
  Left Join Kullanici IK on IK.KullaniciId = StokHareket.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = StokHareket.KullaniciId
  left join Uretici on Uretici.UreticiId = StokHareket.UreticiId
  left join UretimIslemTipi on UretimIslemTipi.UretimIslemTipiId  = StokHareket.UretimIslemTipiId
  left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
  left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeIrsaliyeId
  where StokHareket.OrmePlanlamaId is not null
  order by OrmeTalimatRaporu.OrmePlanlamaId
  into :StokHareketID, :HareketTipi, :OrmePlanlamaId, :HamId, :HamAdi, :RenkId, :RenkAdi, :Lot, :MelanjKodu, UreticiID, :UreticiAdi, :InsertKullaniciAdi, :InsertTarihi,
       :KullaniciAdi, :DegisimTarihi, :Tarih, :HamIade, :Tamir, :Miktari, :UretimIslemTipiID, :UretimIslemTipiStr,
       :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FisSeriNo, :FisNo, :FisTarihi, :HamTipiID, :GirisCikis, :StokGirisNo  do
  begin

    if (HareketTipi = shtDevir) then HareketTipiStr = 'Devir..';
    else if (HareketTipi = shtAlis) then HareketTipiStr = 'Alış';
    else if (HareketTipi = shtSatis and HamIade = 0) then HareketTipiStr = 'Satış';
    else if (HareketTipi = shtSatis and HamIade = 1) then HareketTipiStr = 'Fason Hammadde İade';
    else if (HareketTipi = shtAlisIade) then HareketTipiStr = 'Alış İade';
    else if (HareketTipi = shtSatisIade) then HareketTipiStr = 'Satış İade';
    else if (HareketTipi = shtFasonAlis) then HareketTipiStr = 'Fason İşlenmek Üzere Giriş';
    else if (HareketTipi = shtFasonUretimeCikis) then HareketTipiStr = 'Fason Üretime Çıkış';
    else if (HareketTipi = shtUretimeCikis) then HareketTipiStr = 'Üretime Çıkış';
    else if (HareketTipi = shtFasonUretim and HamIade = 0) then HareketTipiStr = 'Fason Üretim';
    else if (HareketTipi = shtFasonUretim and HamIade = 1) then HareketTipiStr = 'Fason Üretim İade';
    else if (HareketTipi = shtUretim and HamIade = 0) then HareketTipiStr = 'Üretim';
    else if (HareketTipi = shtUretim and HamIade = 1) then HareketTipiStr = 'Üretim İade';
    else if (HareketTipi = shtKaydirmaCikis) then HareketTipiStr = 'Stok Kaydırma (Çıkış)';
    else if (HareketTipi = shtKaydirmaGiris) then HareketTipiStr = 'Stok Kaydırma (Giriş)';
    else if (HareketTipi = shtSarf) then HareketTipiStr = 'Sarf';
    else HareketTipiStr = 'HATALI !!!';

    if (Tamir  = 1) then HareketTipiStr = HareketTipiStr || ' ( Tamir )';

    StokGirisFiyati = Null;
    StokGirisDovizId = Null;
    if (GirisCikis = -1 and StokGirisNo is not null) then
    begin
      select StokHareket.BirimFiyat, StokHareket.DovizId from StokHareket where StokHareket.StokGirisNo = :StokGirisNo and StokHareket.HamIade = 0 and StokHareket.GirisCikis = 1
      into :StokGirisFiyati, :StokGirisDovizId;
    end

    suspend;
  end
end
