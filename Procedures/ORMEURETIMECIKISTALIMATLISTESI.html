CREATE PROCEDURE ORMEURETIMECIKISTALIMATLISTESI(
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    ARENKID INTEGER,
    AORMERAFID INTEGER,
    ALOT VARCHAR(150),
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    RESULTID VARCHAR(255),
    ORMEPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    TARIH TIMESTAMP,
    BOYASIPARISIDSTR VARCHAR(30),
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    PLANLANANMIKTAR DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    TOPLAMSAKATORULEN DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    SIPTUPACIKENMAYLI INTEGER,
    SIPTUPACIKENMAYLISTR VARCHAR(10),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LOT VARCHAR(150),
    IPLIKFIREORANI DOUBLE PRECISION,
    LYCRAKULLANIMORANI DOUBLE PRECISION,
    BASLAMATARIHI TIMESTAMP,
    BEKLEMEDE INTEGER,
    BEKLEMEDESTR VARCHAR(9))
AS
declare variable Iplik        Double Precision;
declare variable Lycra        Double Precision;
declare variable FHamTipiId           integer;
declare variable FCikisMiktari          Double Precision;
declare variable YuvarlamaCarpani       Double Precision;
declare variable FasonStok              integer;
declare variable StokFirmaId            varchar(20);
declare variable FBoyaSiparisYil   integer;
declare variable FBoyaSiparisId    integer;
declare variable BoyaSiparisKayitId integer;
begin
  Select FasonStok, StokFirmaID from OrmeRaf where OrmeRaf.OrmeRafID = :AOrmeRafId into :FasonStok, :StokFirmaID;


  For select OrmePlanlama.OrmePlanlamaId, OrmePlanlama.Tarih, OrmePlanlama.OrmeSiparisDetayId,
      OrmePlanlama.PlanlananMiktar, OrmePlanlama.GerceklesenMiktar, OrmePlanlama.ToplamSakatOrulen,
      OrmePlanlama.BaslamaTarihi, OrmePlanlama.UretimPartiNo, OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4,
      OrmePlanlama.TupAcikEnMayli, OrmePlanlama.Lot, OrmePlanlama.Notu, OrmePlanlama.Beklemede,
      OrmeSiparisDetay.TupAcikEnMayli, OrmeSiparisDetay.SiparisSiraNo,
      BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Ham.Adi, Renk.Adi from OrmePlanlama
    left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = OrmePlanlama.OrmePlanlamaId
    left join OrmesiparisDetay on OrmesiparisDetay.OrmesiparisDetayId = OrmePlanlama.OrmesiparisDetayId
    left join Ham on Ham.HamId = OrmesiparisDetay.HamId
    left join Renk on Renk.RenkId = OrmesiparisDetay.RenkId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmesiparisDetay.BoyaSiparisKayitId
    where OrmePlanlama.FirmaID = :AFirmaId and OrmePlanlama.Bitti = 0 and
         (OrmePlanlama.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
          OrmePlanlamaRecete.HamID = :AHamId and StrKarsilastir(OrmePlanlamaRecete.RenkID, :ARenkId) = 1 and
         (OrmePlanlamaRecete.Lot = :ALot or OrmePlanlamaRecete.Lot is null) and
         (OrmePlanlamaRecete.Fason = :FasonStok) and
         (BoyaSiparis.FirmaId = :StokFirmaId or :FasonStok = 0)  and OrmePlanlama.OnayDurumu = 1
    into :OrmePlanlamaId, :Tarih, :OrmeSiparisDetayId, :PlanlananMiktar, :GerceklesenMiktar, :ToplamSakatOrulen,
         :BaslamaTarihi, :UretimPartiNo, :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :TupAcikEnMayli,
         :Lot, :Notu, :Beklemede, :SipTupAcikEnMayli, :SiparisSiraNo, :BoyaSiparisKayitId, :FBoyaSiparisYil, :FBoyaSiparisId, :HamAdi, :RenkAdi do
  begin
    ResultId = '"' || EgemenIntegerToStr(OrmePlanlamaId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisKayitId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(OrmeSiparisDetayId) || '"' ;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:SipTupAcikenMayli) into :SipTupAcikEnMayliStr;

    if (FBoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil) || ') '  || EgemenIntegerToStr(FBoyaSiparisId);
    else
      BoyaSiparisIdStr = '';

    BeklemedeStr = '';
    if (Beklemede = 1) then BeklemedeStr = 'Beklemede';

    KalanMiktar = PlanlananMiktar - GerceklesenMiktar + ToplamSakatOrulen;

    Iplik = 0;
    Lycra = 0;
    IplikFireOrani = 0;
    LycraKullanimOrani = 0;

    For Select Ham.HamTipiId, Sum(CikisMiktari - IadeMiktari) from OrmePlanlamaRecete
       left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
       where OrmePlanlamaRecete.OrmePlanlamaId = :OrmePlanlamaId
      group by Ham.HamTipiId
      into :FHamTipiId, :FCikisMiktari do
    begin
      if (FHamTipiId = 0) then Iplik = FCikisMiktari;
      else if (FHamTipiId = 6) then Lycra = FCikisMiktari;
    end

    if (GerceklesenMiktar > 0) then
    begin

      IplikFireOrani = ((Iplik + Lycra - GerceklesenMiktar) / GerceklesenMiktar) * 100;
      LycraKullanimOrani = Lycra / GerceklesenMiktar * 100;

      YuvarlamaCarpani = 100;
      IplikFireOrani = Yuvarla(IplikFireOrani, YuvarlamaCarpani);
      LycraKullanimOrani = Yuvarla(LycraKullanimOrani, YuvarlamaCarpani);

    end

    suspend;
  end
end
