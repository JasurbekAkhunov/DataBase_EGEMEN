CREATE PROCEDURE HAMSTOKGIRISGUNLUKRAPORUFVES(
    ABASLANGICTARIHI TIMESTAMP,
    AAYBASLANGICTARIHI TIMESTAMP,
    ARAPORHAMTIPI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    RAPORHAMTIPIID INTEGER,
    HAMKILO NUMERIC(12,3),
    TAMIRKILO NUMERIC(12,3),
    UCRETLITAMIRKILO NUMERIC(12,3),
    UCRETSIZTAMIRKILO NUMERIC(12,3),
    DEGERLENDIRILMEMISTAMIRKILO DOUBLE PRECISION,
    TOPLAMKILO NUMERIC(12,3),
    HAMORAN DOUBLE PRECISION,
    TAMIRORAN DOUBLE PRECISION,
    AYHAMKILO NUMERIC(12,3),
    AYTAMIRKILO NUMERIC(12,3),
    AYUCRETLITAMIRKILO NUMERIC(12,3),
    AYUCRETSIZTAMIRKILO NUMERIC(12,3),
    AYDEGERLENDIRILMEMISTAMIRKILO DOUBLE PRECISION,
    AYTOPLAMKILO NUMERIC(12,3),
    AYHAMORAN DOUBLE PRECISION,
    AYTAMIRORAN DOUBLE PRECISION,
    FASONHAMKILO NUMERIC(12,3),
    FASONTAMIRKILO NUMERIC(12,3),
    FASONUCRETLITAMIRKILO NUMERIC(12,3),
    FASONUCRETSIZTAMIRKILO NUMERIC(12,3),
    FDEGERLENDIRILMEMISTAMIRKILO DOUBLE PRECISION,
    FASONTOPLAMKILO NUMERIC(12,3),
    FASONHAMORAN DOUBLE PRECISION,
    FASONTAMIRORAN DOUBLE PRECISION,
    AYFASONHAMKILO NUMERIC(12,3),
    AYFASONTAMIRKILO NUMERIC(12,3),
    AYFASONUCRETLITAMIRKILO NUMERIC(12,3),
    AYFASONUCRETSIZTAMIRKILO NUMERIC(12,3),
    AYFDEGERLENDIRILMEMISTAMIRKILO DOUBLE PRECISION,
    AYFASONTOPLAMKILO NUMERIC(12,3),
    AYFASONHAMORAN DOUBLE PRECISION,
    AYFASONTAMIRORAN DOUBLE PRECISION)
AS
declare variable FFirmaId       varchar(20);
declare variable FFirmaAdi      varchar(100);
declare variable FHamTipiId     integer;
declare variable FHamTipiAdi    varchar(20);
declare variable Tamir          integer;
declare variable Fason         integer;
declare variable Kilo           Numeric(12,3);
declare variable SonFirmaId     varchar(20);
declare variable SonHamTipiId   integer;

declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
declare variable IrsaliyeTarihi  TimeStamp;
declare variable MinUcretli integer;
declare variable MaxUcretli integer;
declare variable FHamStokMasterId integer;
begin
/* HamStokGirisGunlukRaporu Fason ve Satıs Karışık */
  SonFirmaId = '';
  SonHamTipiId = -1;
  ToplamKilo = 0;

  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;

  for select HamStokIrsaliye.FirmaId, Firma.Adi, Ham.HamTipiId, HamTipi.Adi, HamStokIrsaliye.Fason,
             HamStokMaster.Tamir, HamStokIrsaliye.IrsaliyeTarihi,
             HamStokMaster.Kilo, HamStokMaster.HamStokMasterId from HamStokIrsaliye
    left join firma on firma.firmaid = HamStokIrsaliye.firmaid
    left join HamStokMaster on HamStokMaster.HamStokIrsaliyeId = HamStokIrsaliye.HamStokIrsaliyeId
    left join Ham on Ham.HamId = HamStokMaster.HamId
    left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
    where HamStokMaster.PartiSinifId is null and HamStokIrsaliye.IrsaliyeTarihi >= :AAyBaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABaslangicTarihi and HamStokMaster.Kilo > 0 and
          ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :ARaporHamTipi = 0) or
            (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf and Ham.HamTipiId <> :hstTelef and :ARaporHamTipi = 1) or
            ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :ARaporHamTipi = 2) or
            (Ham.HamTipiId = :hstDokumaKumas and :ARaporHamTipi = 3) or :ARaporHamTipi is null)
    order by HamStokIrsaliye.FirmaId, Ham.HamTipiId
    into :FFirmaId, :FFirmaAdi, :FHamTipiId, :FHamTipiAdi, :Fason, :Tamir, :IrsaliyeTarihi, :Kilo, :FHamStokMasterId do
  begin
    if (SonFirmaId <> FFirmaId or SonHamTipiID <> FHamTipiId) then
    begin
      if (SonFirmaId <> '') then
      begin
        ToplamKilo = HamKilo + TamirKilo;
        FasonToplamKilo = FasonHamKilo + FasonTamirKilo;
        AyToplamKilo = AyHamKilo + AyTamirKilo;
        AyFasonToplamKilo = AyFasonHamKilo + AyFasonTamirKilo;

        HamOran = Null;
        TamirOran = Null;

        FasonHamOran = Null;
        FasonTamirOran = Null;

        AyHamOran = Null;
        AyTamirOran = Null;

        AyFasonHamOran = Null;
        AyFasonTamirOran = Null;

        if (ToplamKilo <> 0) then
        begin
          HamOran = HamKilo * 100 / ToplamKilo;
          TamirOran = TamirKilo * 100 / ToplamKilo;
        end

        if (AyToplamKilo <> 0) then
        begin
          AyHamOran = AyHamKilo * 100 / AyToplamKilo;
          AyTamirOran = AyTamirKilo * 100 / AyToplamKilo;
        end

        if (FasonToplamKilo <> 0) then
        begin
          FasonHamOran = FasonHamKilo * 100 / FasonToplamKilo;
          FasonTamirOran = FasonTamirKilo * 100 / FasonToplamKilo;
        end

        if (AyFasonToplamKilo <> 0) then
        begin
          AyFasonHamOran = AyFasonHamKilo * 100 / AyFasonToplamKilo;
          AyFasonTamirOran = AyFasonTamirKilo * 100 / AyFasonToplamKilo;
        end

        if (HamTipiID = hstIplik or HamTipiID = hstElyaf or HamTipiID = hstTelef) then RaporHamTipiId = 0;
        else if (HamTipiID = hstOrmeKumas or HamTipiID = hstRibana or HamTipiID = hstAksesuar) then RaporHamTipiId = 1;
        else if (HamTipiID = hstDokumaKumas) then RaporHamTipiId = 2;
        else RaporHamTipiId = -1;

        suspend;
      end

      SonFirmaId = FFirmaId;
      SonHamTipiId = FHamTipiId;
      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      HamTipiId = FHamTipiId;
      HamTipiAdi = FHamTipiAdi;
      HamKilo = 0;
      TamirKilo = 0;
      ToplamKilo = 0;
      HamOran = 0;
      TamirOran = 0;

      UcretliTamirKilo   = 0;
      UcretsizTamirKilo  = 0;
      DegerlendirilmemisTamirKilo = 0;

      AyHamKilo = 0;
      AyTamirKilo = 0;
      AyToplamKilo = 0;
      AyHamOran = 0;
      AyTamirOran = 0;

      AyUcretliTamirKilo            = 0;
      AyUcretsizTamirKilo           = 0;
      AyDegerlendirilmemisTamirKilo = 0;

      FasonHamKilo = 0;
      FasonTamirKilo = 0;
      FasonToplamKilo = 0;
      FasonHamOran = 0;
      FasonTamirOran = 0;

      FasonUcretliTamirKilo       = 0;
      FasonUcretsizTamirKilo       = 0;
      FDegerlendirilmemisTamirKilo = 0;

      AyFasonHamKilo = 0;
      AyFasonTamirKilo = 0;
      AyFasonToplamKilo = 0;
      AyFasonHamOran = 0;
      AyFasonTamirOran = 0;

      AyFasonUcretliTamirKilo         = 0;
      AyFasonUcretsizTamirKilo        = 0;
      AyFDegerlendirilmemisTamirKilo  = 0;
    end

    if (Tamir <> 0) then
    begin
      MinUcretli = -1;
      MaxUcretli = -1;

      Select Min(Ucretli), Max(Ucretli) from DisTamirCozum
        left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliID = DisTamirCozum.DisTamirCozumSekliID
        where DisTamirCozum.HamStokMasterId = :FHamStokMasterId
        into :MinUcretli, :MaxUcretli;
    end

    if (IrsaliyeTarihi = ABaslangicTarihi) then
    begin
      if (Fason = 0) then
      begin
        if (Tamir = 0) then HamKilo = HamKilo + Kilo;
        else
        begin
          TamirKilo = TamirKilo + Kilo;

          if (MinUcretli is not null) then
          begin
            if (MaxUcretli = 0) then
              UcretsizTamirKilo = UcretsizTamirKilo + Kilo;
            else
              UcretliTamirKilo = UcretliTamirKilo + Kilo;

          end
          else
            DegerlendirilmemisTamirKilo = DegerlendirilmemisTamirKilo + Kilo;
        end
      end
      else
      begin
        if (Tamir = 0) then FasonHamKilo = FasonHamKilo + Kilo;
        else
        begin
          FasonTamirKilo = FasonTamirKilo + Kilo;

          if (MinUcretli is not null) then
          begin
            if (MaxUcretli = 0) then
              FasonUcretsizTamirKilo = FasonUcretsizTamirKilo + Kilo;
            else
              FasonUcretliTamirKilo = FasonUcretliTamirKilo + Kilo;

          end
          else
            FDegerlendirilmemisTamirKilo = FDegerlendirilmemisTamirKilo + Kilo;
        end
      end
    end

    if (Fason = 0) then
    begin
      if (Tamir = 0) then AyHamKilo = AyHamKilo + Kilo;
      else
      begin
        AyTamirKilo = AyTamirKilo + Kilo;

        if (MinUcretli is not null) then
        begin
          if (MaxUcretli = 0) then
            AyUcretsizTamirKilo = AyUcretsizTamirKilo + Kilo;
          else
            AyUcretliTamirKilo = AyUcretliTamirKilo + Kilo;

        end
        else
          AyDegerlendirilmemisTamirKilo = AyDegerlendirilmemisTamirKilo + Kilo;
      end
    end
    else
    begin
      if (Tamir = 0) then AyFasonHamKilo = AyFasonHamKilo + Kilo;
      else
      begin
        AyFasonTamirKilo = AyFasonTamirKilo + Kilo;


        if (MinUcretli is not null) then
        begin
          if (MaxUcretli = 0) then
            AyFasonUcretsizTamirKilo = AyFasonUcretsizTamirKilo + Kilo;
          else
            AyFasonUcretliTamirKilo = AyFasonUcretliTamirKilo + Kilo;

        end
        else
          AyFDegerlendirilmemisTamirKilo = AyFDegerlendirilmemisTamirKilo + Kilo;
      end
    end

  end

  if (SonFirmaId <> '') then
  begin
    ToplamKilo = HamKilo + TamirKilo;
    FasonToplamKilo = FasonHamKilo + FasonTamirKilo;
    AyToplamKilo = AyHamKilo + AyTamirKilo;
    AyFasonToplamKilo = AyFasonHamKilo + AyFasonTamirKilo;

    HamOran = Null;
    TamirOran = Null;

    FasonHamOran = Null;
    FasonTamirOran = Null;

    AyHamOran = Null;
    AyTamirOran = Null;

    AyFasonHamOran = Null;
    AyFasonTamirOran = Null;

    if (ToplamKilo <> 0) then
    begin
      HamOran = HamKilo * 100 / ToplamKilo;
      TamirOran = TamirKilo * 100 / ToplamKilo;
    end

    if (AyToplamKilo <> 0) then
    begin
      AyHamOran = AyHamKilo * 100 / AyToplamKilo;
      AyTamirOran = AyTamirKilo * 100 / AyToplamKilo;
    end

    if (FasonToplamKilo <> 0) then
    begin
      FasonHamOran = FasonHamKilo * 100 / FasonToplamKilo;
      FasonTamirOran = FasonTamirKilo * 100 / FasonToplamKilo;
    end

    if (AyFasonToplamKilo <> 0) then
    begin
      AyFasonHamOran = AyFasonHamKilo * 100 / AyFasonToplamKilo;
      AyFasonTamirOran = AyFasonTamirKilo * 100 / AyFasonToplamKilo;
    end

    if (HamTipiID = hstIplik or HamTipiID = hstElyaf or HamTipiID = hstTelef) then RaporHamTipiId = 0;
    else if (HamTipiID = hstOrmeKumas or HamTipiID = hstRibana or HamTipiID = hstAksesuar) then RaporHamTipiId = 1;
    else if (HamTipiID = hstDokumaKumas) then RaporHamTipiId = 2;
    else RaporHamTipiId = -1;

    suspend;
  end
end
