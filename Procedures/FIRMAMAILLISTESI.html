CREATE PROCEDURE FIRMAMAILLISTESI
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    YETKILIADI VARCHAR(50),
    EMAIL VARCHAR(80))
AS
begin
  For Select FirmaId, Adi, Email from Firma
  into :FirmaId, :FirmaAdi, :Email do
  begin
    YetkiliAdi = '';
    if (Email is not null and Email <> '') then suspend;

    For Select Adi, Email from FirmaYetkili where FirmaYetkili.FirmaID = :FirmaId and FirmaYetkili.Email is not null and FirmaYetkili.Email <> '' into :YetkiliAdi, :EMail do suspend;
  end
end
