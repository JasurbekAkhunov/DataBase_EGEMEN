CREATE PROCEDURE BOYASIPARISRAPORU_6(
    AIMALATTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AFIRMAYETKILIID INTEGER,
    AISLEMTIPIID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AMUSTERIKALITENO VARCHAR(20),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AHAMID INTEGER,
    AKULLANICIID INTEGER,
    AKALITEID VARCHAR(20),
    ATALIMATKODU VARCHAR(20),
    AMAKINAID VARCHAR(10),
    ATEKRARURETIM INTEGER,
    ADESENID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AMARKALI INTEGER,
    AKOMISYONLU INTEGER,
    ALAMINASYON INTEGER,
    ARECETETURUID INTEGER,
    ABATCHNO VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ABASKITIPIID INTEGER,
    AHAMGRUP1ID VARCHAR(4),
    AHAMGRUP2ID VARCHAR(4),
    AHAMGRUP3ID VARCHAR(4),
    AHAMGRUP4ID VARCHAR(4),
    ASIPARISRECETEDURUMU INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISSARJID INTEGER,
    HAMSTOKDETAYID INTEGER,
    HAMSTOKIRSALIYEID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30),
    ORMECIMAKINANO VARCHAR(10),
    HAMADI VARCHAR(255),
    LOT VARCHAR(150),
    PUS INTEGER,
    FEINE INTEGER,
    REZERVMIKTAR NUMERIC(12,3),
    PARTILENENMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    IRSALIYEGIRISMIKTAR NUMERIC(12,3),
    ALISFIYATI DOUBLE PRECISION,
    ALISFIYATIDOVIZID VARCHAR(5),
    IPLIKBOBINSAYISI INTEGER,
    BOYASIPARISREZERVID INTEGER)
AS
begin
  For Select BoyaSiparisRaporu_2.BoyaSiparisDetayId, BoyaSiparisRezerv.BoyaSiparisSarjId, BoyaSiparisRezerv.HamStokDetayId, HamStokDetay.HamStokIrsaliyeId,
             HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             Ormeci.Adi, HamStokMaster.OrmeciMakinaNo, Ham.Adi, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Lot,
             BoyaSiparisRezerv.Miktar, BoyaSiparisRezerv.KullanilanMiktar, HamStokMaster.Miktar,
             HamStokMaster.NetBirimFiyat, coalesce(HamStokFatura.DovizId,HamStokIrsaliye.DovizId), IplikIrsaliyeDetay.BobinSayisi, BoyaSiparisRezerv.BoyaSiparisRezervId
             from BoyaSiparisRaporu_2(:AImalatTipiId, :ABaslangicTarihi, :ABitisTarihi, :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AFirmaYetkiliId, :AIslemTipiId, :AIslemGrubuId, :ASatisTipi, :ABoyaSiparisTipiId, :ALabReceteId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMusteriKaliteNo, :AOnayDurumu, :APartilenen, :AKapaliDurumu, 0, 0, :ASiparisiAlanKullaniciId, :AYurtDisi, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AHamId, :AKullaniciId, :AKaliteId, :ATalimatKodu, :AMakinaId, :ATekrarUretim, :ADesenID, :ATemsilciId, :AMarkali, :AKomisyonlu, :ALaminasyon, :AReceteTuruId, :ABatchNo, 1, :ASertifikaNo, :ABaskiTipiId, :AHamGrup1Id, :AHamGrup2Id, :AHamGrup3Id, :AHamGrup4Id, :ASiparisReceteDurumu)
    left join BoyaSiparisRezerv on BoyaSiparisRezerv.BoyaSiparisDetayId = BoyaSiparisRaporu_2.BoyaSiparisDetayId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
    left join HamStokFatura on HamStokFatura.HamStokFaturaId = HamStokMaster.HamStokFaturaId
    left join Firma on Firma.Firmaid = hamstokIrsaliye.Firmaid
    left join Ormeci on Ormeci.Ormeciid = HamstokIrsaliye.Ormeciid
    left join Ham on Ham.Hamid = HamStokDetay.Hamid
    left Join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayID = HamStokMaster.IplikIrsaliyeDetayID
    where BoyaSiparisRezerv.BoyaSiparisDetayId is not null
    into :BoyaSiparisDetayId, :BoyaSiparisSarjId, :HamStokDetayId, :HamStokIrsaliyeId, :FirmaId, :FirmaAdi, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi,
         :OrmeciAdi, :OrmeciMakinaNo, :HamAdi, :Pus, :Feine, :Lot,:RezervMiktar, :PartilenenMiktar, :IrsaliyeGirisMiktar,
         :AlisFiyati, :AlisFiyatiDovizId, :IplikBobinSayisi, :BoyaSiparisRezervId do
  begin
    KalanMiktar = RezervMiktar - PartilenenMiktar;

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';
    suspend;
  end
end
