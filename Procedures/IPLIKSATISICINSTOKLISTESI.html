CREATE PROCEDURE IPLIKSATISICINSTOKLISTESI(
    AMINSTOKID INTEGER,
    AMAXSTOKID INTEGER,
    AMINMIKTAR NUMERIC(12,3),
    AMAXMIKTAR NUMERIC(12,3),
    ALOT VARCHAR(30),
    AHAMID INTEGER,
    AFIRMAID VARCHAR(20),
    AMINKAPNO INTEGER,
    AMAXKAPNO INTEGER,
    AIPLIKDEPOID VARCHAR(10))
RETURNS (
    IPLIKSTOKID INTEGER,
    NUMARATOR INTEGER,
    BOBINSAYISI INTEGER,
    KAPNO INTEGER,
    MIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMTIPIID INTEGER,
    IPLIKDEPOID VARCHAR(10),
    LOT VARCHAR(30),
    FLAMKODU VARCHAR(20),
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20))
AS
declare variable shtDepoTransfer  Smallint;
declare variable shtFasonAlis     Smallint;
begin
  Select shtDepoTransfer, shtFasonAlis from IplikStokHareketTipiGetir into :shtDepoTransfer, :shtFasonAlis;

  for Select distinct IplikStok.IplikStokId, IplikStok.Numarator, IplikStok.Miktar, IplikStok.NetMiktar, IplikStok.HamId, Ham.Adi,
         coalesce(case when IplikIrsaliyeDetay.Lot = '' then null else IplikIrsaliyeDetay.Lot  end , IplikUretim.Lot, KirpintiIsemri.Lot, ElyafUretim.Lot), coalesce(IplikIrsaliyeDetay.FlamKodu, IplikUretim.FlamKodu), IplikStok.KapNo, IplikStok.BobinSayisi,
         IplikStok.IplikDepoID, coalesce(IplikIrsaliyeDetay.AmbalajID, IplikUretim.AmbalajID), coalesce(IrsaliyeAmbalaj.Adi, UretimAmbalaj.Adi), Ham.HamTipiId from IplikStok
    left join Ham on Ham.HamId = IplikStok.HamId
    left join IplikUretim On IplikUretim.IplikStokID = IplikStok.IplikStokId
    left join IplikLot On IplikLot.IplikLotID = IplikUretim.IplikLotId
    left join Harman On Harman.HarmanKayitID = IplikLot.HarmanKayitID
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = Harman.BoyaSiparisKayitID
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = IplikStok.IplikIrsaliyeDetayId
    left join Ambalaj IrsaliyeAmbalaj on IrsaliyeAmbalaj.AmbalajId = IplikIrsaliyeDetay.AmbalajId
    left join Ambalaj UretimAmbalaj on UretimAmbalaj.AmbalajId = IplikUretim.AmbalajId
    left join IplikStok TransferIplikStok on TransferIplikStok.CikisIplikIrsaliyeDetayID = IplikIrsaliyeDetay.IplikIrsaliyeDetayId and IplikIrsaliyeDetay.HareketTipi = :shtDepoTransfer
    left join IplikIrsaliyeDetay TransIplikIrsaliyeDetay on TransIplikIrsaliyeDetay.IplikIrsaliyeDetayID = TransferIplikStok.IplikIrsaliyeDetayId
    left join KirpintiIsEmriRecete on KirpintiIsEmriRecete.IplikIrsaliyeDetayId = iplikstok.IplikIrsaliyeDetayId
    left join KirpintiIsemri on KirpintiIsemri.KirpintiIsemriKayitId = KirpintiIsEmriRecete.KirpintiIsemriKayitId
    left join ElyafUretim  on IplikStok.IplikStokId = ElyafUretim.IplikStokId and IplikStok.Numarator = ElyafUretim.Numarator
    where (:AMinStokId is Null or IplikStok.IplikStokId >= :AMinStokId) and
          (:AMaxStokId is Null or IplikStok.IplikStokId <= :AMaxStokId) and
          (:AMinKapNo is Null or IplikStok.KapNo >= :AMinKapNo) and
          (:AMaxKapNo is Null or IplikStok.KapNo <= :AMaxKapNo) and
          (:AMinMiktar is Null or IplikStok.Miktar >= :AMinMiktar) and
          (:AMaxMiktar is Null or IplikStok.Miktar <= :AMaxMiktar) and
          (:ALot is Null or IplikUretim.Lot = :ALot or IplikIrsaliyeDetay.Lot = :ALot or KirpintiIsemri.Lot = :Alot) and
          (:AHamId is Null or IplikStok.HamId = :AHamId) and
          IplikStok.CikisIplikIrsaliyeDetayId is Null and
          (:AIplikDepoId is Null or IplikStok.IplikDepoId = :AIplikDepoId) and
          (Harman.BoyaSiparisKayitID is Null or BoyaSiparis.FirmaID = :AFirmaId) and /*Fason olanları ilgili firmadan başkasına satmamak için*/
          ((IplikIrsaliyeDetay.HareketTipi = :shtFasonAlis and IplikIrsaliyeDetay.FirmaID = :AFirmaID) or IplikIrsaliyeDetay.HareketTipi <> :shtFasonAlis) and /*Fason olanları ilgili firmadan başkasına satmamak için*/
          ((IplikIrsaliyeDetay.HareketTipi = :shtDepoTransfer and ((TransIplikIrsaliyeDetay.HareketTipi = :shtFasonAlis and TransIplikIrsaliyeDetay.FirmaID = :AFirmaID) or TransIplikIrsaliyeDetay.HareketTipi <> :shtFasonAlis)) or IplikIrsaliyeDetay.HareketTipi <> :shtDepoTransfer) /*Fason olanları ilgili firmadan başkasına satmamak için*/
    into :IplikStokId, :Numarator, :Miktar, :NetMiktar, :HamId, :HamAdi, :Lot, :FlamKodu, :KapNo, :BobinSayisi, :IplikDepoID, :AmbalajID, :AmbalajAdi, :HamTipiId do
  begin
    suspend;
  end
end
