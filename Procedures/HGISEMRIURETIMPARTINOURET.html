CREATE PROCEDURE HGISEMRIURETIMPARTINOURET(
    AHGSIPARISKAYITID INTEGER,
    AFIRMAID VARCHAR(20),
    AMAKINAID MAKINAKODUSIZE,
    AKESIMTALIMATI EGEBOOLEAN)
RETURNS (
    URETIMPARTINO VARCHAR(30),
    NUMARATOR INTEGER)
AS
declare variable HGSiparisYil         integer;
declare variable HGSiparisId          integer;
Declare variable FirmaKisaAdi           varchar(10);
begin
  Select HGSiparis.HGSiparisYil, HGSiparis.HGSiparisId from HGSiparis
    where HGSiparis.HGSiparisKayitId = :AHGSiparisKayitId
    into :HGSiparisYil, :HGSiparisId;

  Select Max(Numarator) from HGIsEmri
    where HGIsEmri.HGSiparisKayitId = :AHGSiparisKayitId and
          StrKarsilastir(HGIsEmri.FirmaId, :AFirmaId) = 1 and
          StrKarsilastir(HGIsEmri.OrmeMakinaID, :AMakinaID) = 1
    into :Numarator;

  if (Numarator is null) then Numarator = 0;
  Numarator = Numarator + 1;

  if (AFirmaId is not Null) then
  begin
    Select Firma.KisaAdi from Firma where Firma.FirmaID = :AFirmaID into :FirmaKisaAdi;
    UretimPartiNo = EgemenIntegerToStr(HGSiparisYil) || '-'  || EgemenIntegerToStr(HGSiparisId) || '-'  || iif(AKesimTalimati = 0, FirmaKisaAdi || '-', '') || EgemenIntegerToStr(Numarator);
  end
  else if (AMakinaID is not Null) then
  begin
    UretimPartiNo = EgemenIntegerToStr(HGSiparisYil) || '-'  || EgemenIntegerToStr(HGSiparisId) || '-'  || iif(AKesimTalimati = 0, AMakinaID || '-', '')    || EgemenIntegerToStr(Numarator);
  end

  suspend;
end
