  select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.ImalatTipiId, Parti.AcilisTarihi, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Parti.ModelNo, Parti.IslemGrubuId, Parti.IslemTipiId,

    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId

         :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :YurtDisi, :PartiSevkiyatNotu, :LabReceteUretimNotu, :FirmaAdi2, :SiparisiVeren, :SiparisiAlanKullaniciId,
         :SiparisiAlanKullaniciAdi, :FirmaKisaAdi, :VaryantMusteriBarkodNo, :VaryantAciklama, :DesenTasarimNo, :SiparisTarihi,
         :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi, :DesenVaryantResimDosyaAdi, :PantoneKodu;

  BoyaSiparisIdStr = '';
  if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

  DesenKoduAdiVaryant = cast( '(' || Coalesce(DesenId, '') || ') ' || Coalesce(DesenTasarimNo, '') || iif(coalesce(DesenTasarimNo, '') <> '', ' - ', '') || DesenAdi || ' - ' || VaryantAdi as blob character set UTF8);

  if (PartiTipiID = 0) then
  if (PsKKEtiketTipiId is null) then
  if (DefaultIplikPsKKEtiketTipiId is not null and DefaultIplikPsKKEtiketTipiId > 0) then
  begin
    PsKKEtiketTipiId = DefaultIplikPsKKEtiketTipiId;

    Select Adi, PaketlemeEtiketNo from PSKKEtiketTipi where PsKKEtiketTipi.PsKKEtiketTipiID = :PsKKEtiketTipiId into :PsKKEtiketTipiAdi, :PaketlemeEtiketNo;

  begin

  end

  else OrnekPartimi = 0;

    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiKayitId = :PartiKayitId and BoyaSiparisDetay.OrnekPartiSinifId is not null and
          PartiStok.PartiSinifId <> BoyaSiparisDetay.OrnekPartiSinifId and PartiStok.PartiStokIptalKayitId is null
    into :KayitSayisi;

  if (KayitSayisi > 0) then OrnekPartisiVarmi = 1;
  else OrnekPartisiVarmi = 0;

  Select Count(*) from PartiStok where PartiStok.PartiKayitId = :PartiKayitId and PartiStok.ReferansPartiSinifId is not null into :KayitSayisi;

  if (KayitSayisi > 0) then ReferansPartisiVar = 1;
  else ReferansPartisiVar = 0;


  PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

  HamAdi = '';

  For Select Ham.Adi from PartiStok
    left join Ham on Ham.HamId = PartiStok.HamId
    where PartiStok.PartiKayitId = :PartiKayitId
    group by Ham.Adi
    into :FHamAdi do
  begin
    if (Char_Length(HamAdi) + Char_Length(FHamAdi) < 1000) then
    begin
      if (HamAdi <> '' ) then HamAdi = HamAdi || cast(', ' as blob character set UTF8);
      HamAdi = cast(HamAdi as blob character set UTF8) || FHamAdi;
    end
  end

  MusteriRenkKoduveAdi = '';

  For Select BoyaSiparisDetay.MusteriRenkKoduveAdi from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiKayitId = :PartiKayitId
    group by BoyaSiparisDetay.MusteriRenkKoduveAdi
    into :FMusteriRenkKoduveAdi do
  begin
    if (Char_Length(cast(MusteriRenkKoduveAdi as blob character set UTF8)) + Char_Length(cast(FMusteriRenkKoduveAdi as blob character set UTF8)) < 1000) then
    begin
      if (MusteriRenkKoduveAdi <> '' ) then MusteriRenkKoduveAdi = MusteriRenkKoduveAdi || cast(', ' as blob character set UTF8);
      MusteriRenkKoduveAdi = cast(MusteriRenkKoduveAdi as blob character set UTF8) || FMusteriRenkKoduveAdi;
    end
  end

  MusteriSiparisNo = '';
  For Select BoyaSiparisDetay.MusteriSiparisNo from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiKayitId = :PartiKayitId
    group by BoyaSiparisDetay.MusteriSiparisNo
    into :FMusteriSiparisNo do
  begin
    if (Char_Length(cast(MusteriSiparisNo as blob character set UTF8)) + Char_Length(cast(FMusteriSiparisNo as blob character set UTF8)) < 1500) then
    begin
      if (MusteriSiparisNo <> '' ) then MusteriSiparisNo = MusteriSiparisNo || cast(', ' as blob character set UTF8);
      MusteriSiparisNo = cast(MusteriSiparisNo as blob character set UTF8) || FMusteriSiparisNo;
    end
  end

  KaliteAdi = '';
  KaliteId = '';

  For Select Kalite.KaliteAdi, Kalite.KaliteId from PartiStok
    left join Kalite on Kalite.KaliteId = PartiStok.KaliteId
    where PartiStok.PartiKayitId = :PartiKayitId
    group by Kalite.KaliteAdi, Kalite.KaliteId
    into :FKaliteAdi, :FKaliteId do
  begin
    if (Char_Length(cast(KaliteAdi as blob character set UTF8)) + Char_Length(cast(FKaliteAdi as blob character set UTF8)) < 1540) then
    begin
      if (KaliteAdi <> '' ) then KaliteAdi = KaliteAdi || cast(', ' as blob character set UTF8);
      KaliteAdi = cast(KaliteAdi as blob character set UTF8) || FKaliteAdi;
    end

    if (Char_Length(cast(KaliteId as blob character set UTF8)) + Char_Length(cast(FKaliteId as blob character set UTF8)) < 129) then
    begin
      if (KaliteId <> '' ) then KaliteId = KaliteId || cast(', ' as blob character set UTF8);
      KaliteId = cast(KaliteId as blob character set UTF8) || FKaliteId;
    end

  end

  PaketlemeSekliAdi = '';
  For Select PaketlemeSekli.Adi from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliId = BoyaSiparisDetay.PaketlemeSekliId
    where PartiStok.PartiKayitId = :PartiKayitId
    group by PaketlemeSekli.Adi
    into :FPaketlemeSekliAdi do
  begin
    if (Char_Length(cast(PaketlemeSekliAdi as blob character set UTF8)) + Char_Length(cast(FPaketlemeSekliAdi as blob character set UTF8)) < 260) then
    begin
      if (PaketlemeSekliAdi <> '' ) then PaketlemeSekliAdi = PaketlemeSekliAdi || cast(', ' as blob character set UTF8);
      PaketlemeSekliAdi = cast(PaketlemeSekliAdi as blob character set UTF8) || FPaketlemeSekliAdi;
    end
  end

  if (PartiKayitId is not null) then
  begin
    Select first 1 PartiStok.PartiSinifId from PartiStok
      where PartiStok.PartiKayitId = :PartiKayitId
      order by PartiStok.Kilo desc
      into :BuyukPartiSinifId;

    suspend;
  end
end
