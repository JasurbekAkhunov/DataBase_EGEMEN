CREATE PROCEDURE AYLIKTAMIRSEVKRAPORUTIP2(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFASON INTEGER,
    APARTITIPI INTEGER,
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICFIRMAGRUP1ID VARCHAR(4),
    ABITISFIRMAGRUP1ID VARCHAR(4),
    ABASLANGICFIRMAGRUP2ID VARCHAR(4),
    ABITISFIRMAGRUP2ID VARCHAR(4),
    ABASLANGICFIRMAGRUP3ID VARCHAR(4),
    ABITISFIRMAGRUP3ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TEMSILCIADI VARCHAR(40),
    UCRETLITAMIRKILO NUMERIC(12,3),
    UCRETSIZTAMIRKILO NUMERIC(12,3),
    TAMIRKILOTOPLAMI NUMERIC(12,3),
    BRUTUCRETLITAMIRKILO NUMERIC(12,3),
    BRUTUCRETSIZTAMIRKILO NUMERIC(12,3),
    BRUTTAMIRKILOTOPLAMI NUMERIC(12,3))
AS
declare variable FFaturaKesilecek integer;
declare variable FKilo Numeric(12,3);
declare variable FBrutKilo Numeric(12,3);
declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;
declare variable FirmaGrup1            varchar(4);
declare variable FirmaGrup2            varchar(4);
declare variable FirmaGrup3            varchar(4);
declare variable Okey integer;
begin
/* has orme icin yapıldı */
  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;


  For Select Firma.FirmaId, Firma.Adi, Firma.FirmaGrup1,Firma.FirmaGrup2,Firma.FirmaGrup3, Kullanici.Adi from Firma
  left join Kullanici On Kullanici.KullaniciID = Firma.TemsilciId
  where Firma.Musteri = 1 and ((:AFirmaAralikli = 1 and (Firma.FirmaId >= :ABaslangicFirmaId) and (Firma.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0)) and
        ((:AFirmaGrup1Aralikli = 1 and (Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id or :ABaslangicFirmaGrup1Id = '') and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and (Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id or :ABaslangicFirmaGrup2Id = '') and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and (Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id or :ABaslangicFirmaGrup3Id = '') and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = ''))
  into :FirmaId, :FirmaAdi, :FirmaGrup1, :FirmaGrup2, :FirmaGrup3, :TemsilciAdi do
  begin
    Okey = 1;
    if (AFirmaAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaId, FirmaId) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaGrup1Id, FirmaGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaGrup2Id, FirmaGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AFirmaGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicFirmaGrup3Id, FirmaGrup3) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      UcretliTamirKilo  = 0;
      UcretsizTamirKilo = 0;

      BrutUcretliTamirKilo  = 0;
      BrutUcretsizTamirKilo = 0;
      TamirKiloToplami      = 0;
      BrutTamirKiloToplami  = 0;

      For Select Parti.FaturaKesilecek, Sum(BoyaIrsaliyeDetay.Kilo), Sum(BoyaIrsaliyeDetay.BrutKilo) from BoyaIrsaliye
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeId = BoyaIrsaliye.BoyaIrsaliyeId
      left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
      where BoyaIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and BoyaIrsaliye.IrsaliyeTarihi <= :ABitisTarihi and BoyaIrsaliye.FirmaID = :FirmaId and
           (Parti.PartiTipiID = :APartiTipi or :APartiTipi is null) and
           (Parti.Fason = :AFason or :AFason is null) and BoyaIrsaliyeDetay.PartiSinifID is not null and Parti.IslemTipiID = 3
      Group By Parti.FaturaKesilecek
      into :FFaturaKesilecek, :FKilo, :FBrutKilo do
      begin
        if (FFaturaKesilecek = 1) then
        begin
          UcretliTamirKilo  = UcretliTamirKilo + FKilo;
          BrutUcretliTamirKilo  = BrutUcretliTamirKilo + FBrutKilo;
        end
        else
        begin
          UcretsizTamirKilo  = UcretsizTamirKilo + FKilo;
          BrutUcretsizTamirKilo  = BrutUcretsizTamirKilo + FBrutKilo;
        end
        TamirKiloToplami = TamirKiloToplami + FKilo;
        BrutTamirKiloToplami = BrutTamirKiloToplami + FKilo;
      end

      suspend;
    end
  end
end
