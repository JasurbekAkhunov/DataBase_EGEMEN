CREATE PROCEDURE OKEYIGELMEYENRENKCALISMARAPORU(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ALABRENKGRUPID INTEGER,
    ALABRECETEGRUPID INTEGER,
    ARECETETURUID INTEGER,
    ARENKDERINLIGIID INTEGER,
    ACALISMADURUMU INTEGER)
RETURNS (
    LABRECETECALISMAID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TEMSILCIID INTEGER,
    TEMSILCIADI VARCHAR(40),
    FASON INTEGER,
    FASONSATISSTR VARCHAR(5),
    HAMADI VARCHAR(255),
    CALISMATARIHI TIMESTAMP,
    GIDISTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    ATISSAYISI INTEGER,
    LABRECETEGRUPID INTEGER,
    LABRECETEGRUPADI VARCHAR(20),
    LABRENKGRUPID INTEGER,
    LABRENKGRUPADI VARCHAR(30),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI VARCHAR(20),
    RECETETURUID INTEGER,
    RECETETURUADI VARCHAR(15))
AS
begin
  For Select LabReceteCalisma.LabReceteCalismaID, LabRecete.LabReceteKodu, LabRecete.FirmaId, Firma.Adi, LabRecete.Adi, LabRecete.Fason, Ham.Adi, LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi,
             LabReceteAtis.GidisTarihi, Firma.TemsilciId, Kullanici.Adi, Count(*), LabRecete.LabReceteGrupId, LabReceteGrup.Adi, LabRecete.LabRenkGrupId, LabRenkGrup.Adi,  LabRecete.RenkDerinligiId, RenkDerinligi.Adi,
             LabRecete.ReceteTuruId, ReceteTuru.Adi from LabReceteAtis
    left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaID = LabReceteAtis.LabReceteCalismaID
    left join LabRecete on LabRecete.LabReceteID = LabReceteCalisma.LabReceteId
    left join LabReceteGrup On LabReceteGrup.LabReceteGrupId = LabRecete.LabReceteGrupId
    left join LabRenkGrup On LabRenkGrup.LabRenkGrupId = LabRecete.LabRenkGrupId
    left join ReceteTuru On ReceteTuru.ReceteTuruId = LabRecete.ReceteTuruId
    left join RenkDerinligi On RenkDerinligi.RenkDerinligiId = LabRecete.RenkDerinligiId
    left join Firma on Firma.FirmaID = LabRecete.FirmaId
    left join Ham on Ham.HamId = LabReceteCalisma.HamId
    left join Kullanici on Kullanici.KullaniciId = Firma.TemsilciId
    where LabReceteAtis.GidisTarihi >= :ABaslangicTarihi and LabReceteAtis.GidisTarihi <= :ABitisTarihi and
         (LabReceteCalisma.FirmaId = :AFirmaId or :AFirmaId is null) and
         (LabRecete.LabRenkGrupId = :ALabRenkGrupId or :ALabRenkGrupId is null) and
         (LabRecete.LabReceteGrupId = :ALabReceteGrupId or :ALabReceteGrupId is null) and
         (LabRecete.ReceteTuruId = :AReceteTuruId or :AReceteTuruId is null) and
         (LabRecete.RenkDerinligiId = :ARenkDerinligiId or :ARenkDerinligiId is null) and
         ( (:ACalismaDurumu = 1 and LabReceteCalisma.Rptmi = 1) or
           (:ACalismaDurumu = 2 and LabReceteCalisma.Ilave = 1) or
           (:ACalismaDurumu = 3 and LabReceteCalisma.SokumUzeriBoya = 1) or
           (:ACalismaDurumu = 0 and (LabReceteCalisma.Rptmi = 0 and LabReceteCalisma.Ilave = 0 and LabReceteCalisma.SokumUzeriBoya = 0) ) or
           :ACalismaDurumu is null)
    group by LabReceteCalisma.LabReceteCalismaID, LabRecete.LabReceteKodu, LabRecete.FirmaId, Firma.Adi, LabRecete.Adi, LabRecete.Fason, Ham.Adi, LabReceteCalisma.CalismaTarihi, LabReceteCalisma.TerminTarihi,
             LabReceteAtis.GidisTarihi, Firma.TemsilciId, Kullanici.Adi, LabRecete.LabReceteGrupId, LabReceteGrup.Adi, LabRecete.LabRenkGrupId, LabRenkGrup.Adi,  LabRecete.RenkDerinligiId, RenkDerinligi.Adi,
               LabRecete.ReceteTuruId, ReceteTuru.Adi
    having max(LabReceteAtis.Okey) = 0
    into :LabReceteCalismaID, :LabReceteKodu, :FirmaId, :FirmaAdi, :LabReceteAdi, :Fason, :HamAdi, :CalismaTarihi, :TerminTarihi, :GidisTarihi,
         :TemsilciId, :TemsilciAdi, :AtisSayisi, :LabReceteGrupId, :LabReceteGrupAdi, :LabRenkGrupId, :LabRenkGrupAdi,
         :RenkDerinligiId, :RenkDerinligiAdi, :ReceteTuruId, :ReceteTuruAdi do
  begin
    if (Fason = 0) then FasonSatisStr = 'Satış';
    else FasonSatisStr = 'Fason';
    suspend;
  end
end
