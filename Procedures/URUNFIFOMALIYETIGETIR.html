CREATE PROCEDURE URUNFIFOMALIYETIGETIR(
    AURUNID VARCHAR(20),
    ADEFAULTBIRIMCARPAN DOUBLE PRECISION,
    ABITISTARIHI TIMESTAMP,
    ARAPORDOVIZID VARCHAR(5))
RETURNS (
    NETBIRIMFIYAT DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    KALANMALIYET DOUBLE PRECISION)
AS
declare variable CikisMiktari   Double Precision;
declare variable FMiktar        Double Precision;
declare variable StokFifoMaliyetBaslangicTarihi TimeStamp;
begin
  select Sistem.StokFifoMaliyetBaslangicTarihi from sistem where sistem.sistemid = 1 into :StokFifoMaliyetBaslangicTarihi ;

  CikisMiktari = 0;

  Select sum(StokDetay.Miktar / UrunBirim.Carpan * :ADefaultBirimCarpan) from StokDetay
   left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
   where stokdetay.Urunid = :AUrunId and ( StokDetay.Tarih >= :StokFifoMaliyetBaslangicTarihi or :StokFifoMaliyetBaslangicTarihi is null ) and StokDetay.Tarih <= :ABitisTarihi and StokDetay.GirisCikis = -1 and StokDetay.TransferDepoId is null
  into :CikisMiktari;


  if (CikisMiktari is null) then CikisMiktari = 0;

  for Select NetBirimFiyat, Miktar from UrunFifoMaliyetHazirla(:AUrunId, :ADefaultBirimCarpan, :ABitisTarihi, :ARaporDovizId)
    order by UrunFifoMaliyetHazirla.Tarih asc
    into :NetBirimFiyat, :Miktar do
  begin
    if (CikisMiktari > Miktar) then CikisMiktari = CikisMiktari - Miktar;
    else if (CikisMiktari = Miktar) then CikisMiktari = 0;
    else
    begin
      KalanMiktar = Miktar - CikisMiktari;
      CikisMiktari = 0;
      KalanMaliyet = KalanMiktar * NetBirimFiyat;
      suspend;
    end
  end
end
