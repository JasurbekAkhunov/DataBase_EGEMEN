CREATE PROCEDURE URUNKATEGORIFIRMALISTESI(
    AURUNKATEGORIID VARCHAR(4))
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100))
AS
begin
  For Select FirmaUrunKategori.FirmaId, Firma.Adi from FirmaUrunKategori
    left join Firma on Firma.FirmaId = FirmaUrunKategori.FirmaId
    where FirmaUrunKategori.UrunKategoriId = :AUrunKategoriId and Firma.Kapali = 0
    into :FirmaId, :FirmaAdi do suspend;
end
