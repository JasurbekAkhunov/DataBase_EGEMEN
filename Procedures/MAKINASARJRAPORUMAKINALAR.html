CREATE PROCEDURE MAKINASARJRAPORUMAKINALAR(
    AUNITEID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AMAKINAGRUBU VARCHAR(4),
    AMAKINAID VARCHAR(10))
RETURNS (
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    KAPASITE DOUBLE PRECISION,
    TOPLAMKILO NUMERIC(12,3),
    TOPLAMMETRE NUMERIC(12,3),
    ARAVERMESURESI DOUBLE PRECISION,
    ARAVERMESURESISTR VARCHAR(20),
    IMALATSURESI DOUBLE PRECISION,
    IMALATSURESISTR VARCHAR(20),
    BOSGECENSURE DOUBLE PRECISION,
    BOSGECENSURESTR VARCHAR(20),
    SARJSAYISI INTEGER)
AS
declare variable MinTarih TimeStamp;
declare variable  MaxTarih TimeStamp;
begin
  For Select Makina.MakinaId, Makina.Adi, Makina.MakinaPlanlamaGrupId, Makina.Kapasite, Sum(MakinaSarjRaporuSarjlar.ToplamKilo), Sum(MakinaSarjRaporuSarjlar.ToplamMetre),
             Sum(MakinaSarjRaporuSarjlar.AraVermeSuresi), Sum(MakinaSarjRaporuSarjlar.ImalatSuresi),
             Sum(MakinaSarjRaporuSarjlar.BosGecenSure),
             Min(MakinaSarjRaporuSarjlar.IlkGirisTarihi),
			 Max(MakinaSarjRaporuSarjlar.SonCikisTarihi) from Makina
	Left Join MakinaSarjRaporuSarjlar(Makina.UniteId, :ABaslangicTarihi, :ABitisTarihi, Makina.MakinaGrubu, Makina.MakinaID) on 1=1
    Where Makina.OtomasyonaDahil = 1 and (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null) and
          (Makina.UniteID = :AUniteID or :AUniteID is null)
    group by Makina.MakinaId, Makina.Adi, Makina.MakinaPlanlamaGrupId, Makina.Kapasite
    Order by Makina.MakinaPlanlamaGrupId, Makina.Kapasite, Makina.MakinaId
    into :MakinaId, :MakinaAdi, :MakinaPlanlamaGrupId, :Kapasite, :ToplamKilo, :ToplamMetre, :AraVermeSuresi, :ImalatSuresi, :BosGecenSure, :MinTarih, :MaxTarih do
  begin
    ImalatSuresiStr = TarihtoStr(ImalatSuresi);

    BosGecenSureStr = TarihtoStr(BosGecenSure);

    if (AraVermeSuresi <> 0) then AraVermeSuresiStr = TarihtoStr(AraVermeSuresi);
    else AraVermeSuresiStr = '';

    Select Count(*) from MakinaSarjRaporuSarjlar(:AUniteId, :ABaslangicTarihi, :ABitisTarihi, :AMakinaGrubu, :MakinaId) where ID = 1 into :SarjSayisi;
    suspend;
  end
end
