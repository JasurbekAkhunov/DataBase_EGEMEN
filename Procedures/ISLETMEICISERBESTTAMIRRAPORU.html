CREATE PROCEDURE ISLETMEICISERBESTTAMIRRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ATAMIRTIPIID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    MODELNO VARCHAR(20),
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    MAKINAID VARCHAR(10),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    GIRISPERSONELADI VARCHAR(40),
    GIRISTARIHI TIMESTAMP,
    CIKISPERSONELADI VARCHAR(40),
    CIKISTARIHI TIMESTAMP,
    TAMIRTARIHI TIMESTAMP,
    TAMIRTIPIADI VARCHAR(20),
    TAMIRNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30))
AS
begin
  for select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.FirmaId, Firma.Adi, Parti.ReceteId, Recete.Adi,
             Parti.Modelno, Partistok.HamAdiToplami, PartiAsamaStokDetay.Kilo, IslemTipi.Adi, IslemGrubu.Adi, PartiAsamaStokDetay.MakinaId, Asama.AsamaId,
             Asama.Adi, Gp.Adi, PartiAsamaStokDetay.GirisTarihi, Cp.Adi, PartiAsamaStokDetay.CikisTarihi,
             PartiTamir.tarih, PartiTamir.Notu, TamirTipi.Adi, Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId from PartiAsamaStokDetay
    left join Partitamirsinif on Partitamirsinif.Partitamirsinifid = PartiAsamaStokDetay.Partitamirsinifid and Partitamirsinif.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiAsamaStokDetay.PartiKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
    left join Partitamir on Partitamir.Partitamirid = Partitamirsinif.Partitamirid
    left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join Asama on (Asama.AsamaId = PartiAsamalari.AsamaId or (PartiAsamalari.AsamaId is null and Asama.AsamaId = Recete.UstAsamaId))
    left join Personel Gp on Gp.PersonelId = PartiAsamaStokDetay.GirisPersonelId
    left join Personel Cp on Cp.PersonelId = PartiAsamaStokDetay.CikisPersonelId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    where PartiAsamaStokDetay.CikisTarihi >= :ABaslangicTarihi and
          PartiAsamaStokDetay.CikisTarihi <= :ABitisTarihi and
          (PartiTamir.TamirTipiId = :ATamirTipiId or :ATamirTipiId is null) and
          PartiAsamaStokDetay.PartiTamirSinifId is not null
    into :PartiKayitId, :PartiYil, :PartiId, :FirmaId, :FirmaAdi, :ReceteId, :ReceteAdi,
         :Modelno, :HamAdiToplami, :Kilo, :IslemTipiAdi, :IslemGrubuAdi, :MakinaId, :AsamaId, :AsamaAdi,
         :GirisPersonelAdi, :GirisTarihi, :CikisPersonelAdi, :CikisTarihi, :TamirTarihi, :TamirNotu, :TamirTipiAdi,
         :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId do
  begin
    if (PartiYil is null) then PartiIdStr = '';
    else PartiIdStr = '('  || EgemenIntegertoStr(PartiYil)  || ') '  || EgemenIntegertoStr(PartiId) || ' ';

    if (BoyaSiparisYil is null) then BoyaSiparisIdStr = '';
    else BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

    suspend;
  end
end
