CREATE PROCEDURE MAKINANINIMALATTAKIPARTILERI(
    AMAKINAID VARCHAR(10))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    PARTIASAMALARIID INTEGER,
    GIRISPERSONELID INTEGER,
    GIRISTARIHI TIMESTAMP,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    GIRISPERSONELADI VARCHAR(40),
    ISLEMTIPIID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    HAMTIPIID INTEGER,
    KILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION,
    CIKISKILO DOUBLE PRECISION,
    CIKISMETRE DOUBLE PRECISION,
    CIKISRAFID RAFKODUSIZE,
    BOBINSAYISI INTEGER,
    CIKISV1MIKTAR DOUBLE PRECISION,
    CIKISV2MIKTAR DOUBLE PRECISION,
    CIKISV3MIKTAR DOUBLE PRECISION,
    ISLETMEICITAMIR INTEGER,
    ISLETMEDISITAMIR INTEGER,
    NUMUNE INTEGER,
    KONTROLASAMA INTEGER,
    ILAVEISLEMLERADITOPLAMI VARCHAR(255),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ADI VARCHAR(30),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    FASON INTEGER,
    ORNEKPARTISINIFMI INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    RECETEISEMRIIDSTR VARCHAR(30),
    TEORIKIMALATSURESI INTEGER,
    RECETEASAMATOPLAMSURE INTEGER,
    PARTIASAMASTOKDETAYID INTEGER,
    CIKISTAKILOZORUNLU INTEGER,
    CIKISTAMETREZORUNLU INTEGER,
    CIKISTAARABAZORUNLU INTEGER,
    TOLERANSEN VARCHAR(10),
    ASAMAPARTILEMEMI INTEGER)
AS
declare variable FPartiKayitId          integer;
declare variable FPartiYil              integer;
declare variable FPartiId               integer;
declare variable FPartiSinifId          integer;
declare variable FGirisPersonelId       integer;
declare variable FGirisTarihi           Timestamp;
declare variable FAsamaId               integer;
declare variable FAsamaAdi              varchar(40);
declare variable FReceteId              varchar(20);
declare variable FReceteAdi             varchar(40);
declare variable FGirisPersonelAdi      varchar(40);
declare variable FIslemTipiId           integer;
declare variable FHamAdiToplami         varchar(255);
declare variable FHamTipiId             integer;
declare variable FKilo                  Numeric(12,3);
declare variable FCikisKilo             Numeric(12,3);
declare variable FCikisV1Miktar         Numeric(12,3);
declare variable FCikisV2Miktar         Numeric(12,3);
declare variable FCikisV3Miktar         Numeric(12,3);
declare variable FMetre                 Numeric(12,3);
declare variable FCikisMetre            Numeric(12,3);
declare variable FBobinSayisi           integer;
declare variable FCikisRafId            RafKoduSize;
declare variable FIsletmeIciTamir       integer;
declare variable FKontrolAsama          integer;
declare variable FIlaveIslemlerAdi      varchar(30);
declare variable FLastPartiSinifId      integer;
declare variable FPartiTamirId     integer;
declare variable FBoyaSiparisYil        integer;
declare variable FBoyaSiparisId         integer;
declare variable FFason                 integer;
declare variable FOrnekPartiSinifId     integer;
declare variable FFirmaId               varchar(20);
declare variable FFirmaAdi              varchar(100);
declare variable FManuelSure            integer;
declare variable FTeorikImalatSuresi    integer;
declare variable FReceteAsamaToplamSure integer;
declare variable FPartiAsamalariID      integer;
declare variable FPartiAsamaStokDetayId integer;
declare variable FReceteIsEmriIdStr     varchar(30);
declare variable FCikistaKiloZorunlu    integer;
declare variable FCikistaMetreZorunlu   integer;
declare variable FCikistaArabaZorunlu   integer;
begin
  FLastPartiSinifId = -1;
  OrnekPartiSinifmi = 0;

  For select Parti.PartiYil, Parti.PartiId, PartiAsamaStok.PartiKayitId, PartiAsamaStok.PartiSinifId, PartiAsamaStokDetay.GirisPersonelId,
             PartiAsamaStokDetay.GirisTarihi, asama.asamaid, asama.adi, PartiAsamalari.receteid, recete.adi, Personel.adi, Parti.IslemTipiId,
             (PartiStok.HamAdiToplami ||  iif(coalesce(Kalite.KaliteAdi, '') <> '', ' - ' || Kalite.KaliteAdi,'')), PartiAsamaStokDetay.Kilo, PartiAsamaStokDetay.CikisKilo, PartiAsamaStokDetay.Metre, PartiAsamaStokDetay.CikisMetre, PartiAsamalari.PartiTamirId, PartiAsamalari.KontrolAsama, IlaveIslemler.Adi,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Parti.Fason,
             BoyaSiparisDetay.OrnekPartiSinifId, Parti.FirmaId, Firma.Adi, Ham.HamTipiId, UretimPlanlamaDetay.Sure, UretimPlanlama.ManuelSure,
             DesenVaryant.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, SiparisGrup.Adi, SiparisGrup2.Adi, PartiAsamaStok.PartiAsamalariID, PartiAsamaStokDetay.CikisRafId, PartiAsamaStokDetay.PartiAsamaStokDetayId, BoyaSiparisDetay.EnIstenen,
             BoyaSiparisDetay.GramajIstenen, (ReceteIsEmriDetay.ReceteIsEmriYil || '-' || ReceteIsEmriDetay.ReceteIsEmriId),
             PartiAsamalari.CikistaKiloZorunlu, PartiAsamalari.CikistaMetreZorunlu, PartiAsamalari.CikistaArabaZorunlu, PartiAsamaStokDetay.BobinSayisi,
             PartiAsamaStokDetay.CikisV1Miktar, PartiAsamaStokDetay.CikisV2Miktar, PartiAsamaStokDetay.CikisV3Miktar, BoyaSiparisDetay.ToleransEn, Asama.PartilemeMi  from PartiAsamaStokDetay
  left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaStokDetay.PartiSinifId
  left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
  left join Firma on Firma.FirmaId = Parti.FirmaId
  left join BoyaSiparis on BoyaSiparis.BoyaSipariskayitId = Parti.BoyaSiparisKayitId
  left join PartiStok on  PartiStok.PartiSinifId= PartiAsamaStok.PartiSinifId
  left join Ham on Ham.HamId= PartiStok.HamId
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
  left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
  left join PartiIlaveIslemler on PartiIlaveIslemler.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
  left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId
  left join recete on recete.receteid = PartiAsamalari.receteid
  left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId  or asama.asamaid = recete.ustasamaid
  left join personel on personel.personelid = PartiAsamaStokDetay.girispersonelid
  left join BolumPlanlama on BolumPlanlama.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and BolumPlanlama.PartiSinifId = PartiAsamaStok.PartiSinifId
  left join UretimPlanlamaDetay on UretimPlanlamaDetay.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
  left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
  left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  left join Desen on Desen.DesenId = DesenVaryant.DesenId
  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  Left Join ReceteIsEmriDetay On ReceteIsEmriDetay.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
  Left join Kalite on Kalite.KaliteId = PartiStok.KaliteId
  where PartiAsamaStokDetay.MakinaId = :AMakinaId and PartiAsamaStokDetay.GirisTarihi is not null and PartiAsamaStokDetay.cikistarihi is null and PartiAsamaStok.AsamayaDahil = 1
  into :FPartiYil, :FPartiId, :FPartiKayitId, :FPartiSinifId, :FGirisPersonelId, :FGirisTarihi, :Fasamaid, :FAsamaAdi,
       :FReceteid, :Freceteadi, :FGirisPersonelAdi, :FIslemTipiId, :FHamAdiToplami, :FKilo, :FCikisKilo, :FMetre, :FCikisMetre, :FPartiTamirId, :FKontrolAsama, :FIlaveIslemlerAdi,
       :FBoyaSiparisYil, :FBoyaSiparisId, :FFason, :FOrnekPartiSinifId, :FFirmaId, :FFirmaAdi, :FHamTipiId, :FTeorikImalatSuresi, :FManuelSure,
       :DesenId, :DesenAdi, :VaryantAdi, :SiparisGrupAdi, :SiparisGrup2Adi, :FPartiAsamalariID, :FCikisRafId, :FPartiAsamaStokDetayId, :EnIstenen, :GramajIstenen, :FReceteIsEmriIdStr,
       :FCikistaKiloZorunlu, :FCikistaMetreZorunlu, :FCikistaArabaZorunlu, :FBobinSayisi, :FCikisV1Miktar, :FCikisV2Miktar, :FCikisV3Miktar, :ToleransEn ,:AsamaPartilemeMi do
  begin
    if (FLastPartiSinifId <> FPartiSinifId) then
    begin
      if (FLastPartiSinifId <> -1) then
      begin
        if (ReceteId is not null) then
        begin
          Select Sum(coalesce(MakinaReceteAsamalari.Sure, 0)) from ReceteDetay
            left join MakinaReceteAsamalari on MakinaReceteAsamalari.MakinaID = :AMakinaId and MakinaReceteAsamalari.ReceteAsamaId = ReceteDetay.ReceteAsamaId
            where ReceteDetay.ReceteID = :ReceteId and ReceteDetay.ReceteAsamaId is not null into :ReceteAsamaToplamSure;
        end
        else ReceteAsamaToplamSure = 0;

        suspend;
      end

      PartiKayitId      = FPartiKayitId;
      PartiYil          = FPartiYil;
      PartiId           = FPartiId;
      PartiSinifId      = FPartiSinifId;
      FLastPartiSinifId = FPartiSinifId;
      GirisPersonelId   = FGirisPersonelId;
      GirisTarihi       = FGirisTarihi;
      AsamaId           = FAsamaId;
      AsamaAdi          = FAsamaAdi;
      ReceteId          = FReceteId;
      ReceteAdi         = FReceteAdi;
      GirisPersonelAdi  = FGirisPersonelAdi;
      IslemTipiId       = FIslemTipiId;
      HamAdiToplami     = FHamAdiToplami;
      Kilo              = FKilo;
      CikisV1Miktar     = FCikisV1Miktar;
      CikisV2Miktar     = FCikisV2Miktar;
      CikisV3Miktar     = FCikisV3Miktar;
      CikisKilo         = FCikisKilo;
      Metre             = FMetre;
      CikisMetre        = FCikisMetre;
      BobinSayisi       = FBobinSayisi;
      IsletmeIciTamir   = FIsletmeIciTamir;
      KontrolAsama      = FKontrolAsama;
      BoyaSiparisYil    = FBoyaSiparisYil;
      BoyaSiparisid     = FBoyaSiparisId;
      Fason             = FFason;
      FirmaId           = FFirmaId;
      FirmaAdi          = FFirmaAdi;
      HamTipiId         = FHamTipiId;
      PartiAsamalariID  = FPartiAsamalariID;
      CikisRafId        = FCikisRafId;
      PartiAsamaStokDetayId = FPartiAsamaStokDetayId;
      ReceteIsEmriIdStr = FReceteIsEmriIdStr;
      CikistaKiloZorunlu   = FCikistaKiloZorunlu;
      CikistaMetreZorunlu  = FCikistaMetreZorunlu;
      CikistaArabaZorunlu  = FCikistaArabaZorunlu;

      if (FManuelSure <> 0) then FTeorikImalatSuresi = FManuelSure;
      TeorikImalatSuresi = FTeorikImalatSuresi;

      if (FOrnekPartiSinifId is not null and FOrnekPartiSinifId = PartiSinifId) then OrnekPartiSinifmi = 1;
      else OrnekPartiSinifmi = 0;

      IlaveIslemlerAdiToplami = '';
      if (FPartiTamirId is not null) then IsletmeIciTamir = 1;
      else IsletmeIciTamir = 0;

      if (IslemTipiId = 3) then IsletmeDisiTamir = 1;
      else IsletmeDisiTamir = 0;
      if (IslemTipiId = 4) then Numune = 1;
      else Numune = 0;

    end

    if (Char_Length(IlaveIslemlerAdiToplami) + Char_Length(FIlaveIslemlerAdi) < 250) then
    begin
     if (IlaveIslemlerAdiToplami <> '' ) then IlaveIslemlerAdiToplami = IlaveIslemlerAdiToplami || cast(', ' as varchar(2) character set UTF8);
     IlaveIslemlerAdiToplami = IlaveIslemlerAdiToplami || FIlaveIslemlerAdi;
    end
  end

  if (FLastPartiSinifId <> -1) then
  begin
    if (ReceteId is not null) then
    begin
      Select Sum(coalesce(MakinaReceteAsamalari.Sure, 0)) from ReceteDetay
        left join MakinaReceteAsamalari on MakinaReceteAsamalari.MakinaID = :AMakinaId and MakinaReceteAsamalari.ReceteAsamaId = ReceteDetay.ReceteAsamaId
        where ReceteDetay.ReceteID = :ReceteId and ReceteDetay.ReceteAsamaId is not null into :ReceteAsamaToplamSure;
    end
    else ReceteAsamaToplamSure = 0;

    suspend;
  end
end
