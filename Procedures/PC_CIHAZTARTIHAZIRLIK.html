CREATE PROCEDURE PC_CIHAZTARTIHAZIRLIK(
    APARTISINIFID INTEGER,
    APERSONELID INTEGER)
RETURNS (
    IPLIKTARTIDANCIKISKILOKAYDET INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTISINIFID INTEGER,
    PARTITIPIID INTEGER,
    HAMTIPIID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BOBINSAYISI INTEGER,
    TARTIBASLADI INTEGER,
    TARTIBITTI INTEGER,
    PARTIASAMASTOKDETAYID INTEGER,
    KULLANICIID INTEGER)
AS
declare variable IplikTartiAsamaId             integer;
declare variable FSira                         integer;
declare variable FGirisTarihi                  TimeStamp;
declare variable FCikisTarihi                  TimeStamp;
begin
  Select IplikTartidanCikisKiloKaydet, IplikTartiAsamaId from Sistem where SistemID = 1 into :IplikTartidanCikisKiloKaydet, :IplikTartiAsamaId;

  Select Parti.PartiYil, Parti.PartiId, PartiStok.PartiSinifId, PartiStok.HamTipiId, Parti.PartiTipiId, PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.Metre, PartiStok.BobinSayisi from PartiStok
   left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
   where partiStok.PartiSinifID = :APartiSinifID
   into :PartiYil, :PartiId, :PartiSinifId, :HamTipiId, PartiTipiId, :HamAdiToplami, :Kilo, :Metre, :BobinSayisi;

  if (PartiSinifID is not null and PartiTipiId = 0) then
  begin
    Select Min(Sira) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.OtomasyonaDahilmi = 1 and PartiAsamaStok.AsamayaDahil = 1 and
            PartiAsamaStok.AsamaId = :IplikTartiAsamaId
      into :FSira;

    Select PartiAsamaStokDetay.PartiAsamaStokDetayId, PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :FSira and PartiAsamaStok.AsamayaDahil = 1 and
            PartiAsamaStok.AsamaId = :IplikTartiAsamaId
      into :PartiAsamaStokDetayId, :FGirisTarihi, :FCikisTarihi;

    if (FGirisTarihi is not null) then TartiBasladi = 1;
    else TartiBasladi = 0;

    if (FCikisTarihi is not null) then TartiBitti = 1;
    else TartiBitti = 0;
  end

  if (HamTipiId = 5) then HamTipiId = 0;

  Select Max(KullaniciId) from Kullanici where Kullanici.PersonelId = :APersonelId into :KullaniciId;
  suspend;
end
