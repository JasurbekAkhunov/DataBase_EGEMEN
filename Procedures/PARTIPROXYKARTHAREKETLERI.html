CREATE PROCEDURE PARTIPROXYKARTHAREKETLERI(
    APROXYID VARCHAR(8))
RETURNS (
    TARIH TIMESTAMP,
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100))
AS
begin
  For select PartiProxyLog.Tarih, PartiProxyLog.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiProxyLog.PartiSinifId, PartiStok.HamAdiToplami,
             Parti.FirmaId, Firma.Adi from Partiproxylog
   left join Parti on Parti.PartiKayitId = Partiproxylog.PartiKayitId
   left join PartiStok on PartiStok.PartiSinifId = PartiProxyLog.PartiSinifId
   left join Firma on Firma.FirmaId = Parti.FirmaId
   where PartiProxyLog.ProxyId = :AProxyId order by Tarih
   into :Tarih, :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :HamAdiToplami, :FirmaId, :FirmaAdi do
  begin
    if (PartiYil is not null) then
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
      if (HamAdiToplami is null) then HamAdiToplami = 'Parti Stok Silinmis';
    end
    else PartiIdStr = 'Parti Silinmis';
    suspend;
  end
end
