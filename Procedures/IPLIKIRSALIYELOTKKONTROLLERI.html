CREATE PROCEDURE IPLIKIRSALIYELOTKKONTROLLERI(
    AIRSALIYEID INTEGER)
RETURNS (
    IPLIKIRSALIYEDETAYID INTEGER,
    IPLIKLOTKALITEKONTROLID INTEGER,
    HAMADI VARCHAR(255),
    LOT VARCHAR(30),
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    OLCULENNE DOUBLE PRECISION,
    TEORIKKALINLIKNE INTEGER,
    NECV DOUBLE PRECISION,
    TEORIKBUKUMSAYISI INTEGER,
    OLCULENBUKUM INTEGER,
    ALFA DOUBLE PRECISION,
    UYUZDE DOUBLE PRECISION,
    CVM DOUBLE PRECISION,
    THINEKSI50 DOUBLE PRECISION,
    THICKARTI50 DOUBLE PRECISION,
    NEPSARTI200 DOUBLE PRECISION,
    HEKSI DOUBLE PRECISION,
    GRAM DOUBLE PRECISION,
    MYUZDECV DOUBLE PRECISION,
    RKM DOUBLE PRECISION,
    EYUZDE DOUBLE PRECISION,
    ECVYUZDE DOUBLE PRECISION,
    S3 DOUBLE PRECISION,
    REF VARCHAR(10),
    RENK VARCHAR(10),
    SHEKSI DOUBLE PRECISION,
    THICKARTI35 DOUBLE PRECISION,
    THINEKSI30 DOUBLE PRECISION,
    HAIR VARCHAR(10),
    TINCH VARCHAR(10),
    CVYUZDE VARCHAR(10),
    A1B1C1D1 VARCHAR(10),
    EFG VARCHAR(10),
    H1I1 VARCHAR(10),
    RUTUBET VARCHAR(10),
    A3B3C3D3 VARCHAR(10),
    NOTU VARCHAR(255),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable FLOT varchar(30);
declare variable IrsaliyeTarihi TimeStamp;
begin
  for select IplikIrsaliyeDetayId, Lot, IrsaliyeTarihi from IplikIrsaliye
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeId = IplikIrsaliye.IplikIrsaliyeId
    where IplikIrsaliye.IplikIrsaliyeId = :AIrsaliyeId into :IplikIrsaliyeDetayId, :FLot, :IrsaliyeTarihi do
  begin
    For Select first 1 IplikLotKaliteKontrol.IplikLotKaliteKontrolId, Ham.Adi, IplikLot.Lot, Harman.HarmanYil, Harman.HarmanId,
               IplikLotKaliteKontrol.OlculenNe, IplikLot.KalinlikNe,
               IplikLotKaliteKontrol.NeCv, IplikLot.BukumSayisi, IplikLotKaliteKontrol.OlculenBukum, IplikLotKaliteKontrol.Alfa, IplikLotKaliteKontrol.UYuzde,
               IplikLotKaliteKontrol.CVm, IplikLotKaliteKontrol.ThinEksi50, IplikLotKaliteKontrol.ThickArti50, IplikLotKaliteKontrol.NepsArti200, IplikLotKaliteKontrol.HEksi,
               IplikLotKaliteKontrol.Gram, IplikLotKaliteKontrol.MYuzdeCv, IplikLotKaliteKontrol.RKm, IplikLotKaliteKontrol.EYuzde, IplikLotKaliteKontrol.ECvYuzde, IplikLotKaliteKontrol.S3,
               IplikLotKaliteKontrol.Ref, IplikLotKaliteKontrol.Renk, IplikLotKaliteKontrol.ShEksi, IplikLotKaliteKontrol.ThickArti35, IplikLotKaliteKontrol.ThinEksi30, IplikLotKaliteKontrol.Notu,
               IplikLotKaliteKontrol.Hair, IplikLotKaliteKontrol.Tinch, IplikLotKaliteKontrol.CvYuzde, IplikLotKaliteKontrol.A1B1C1D1,
               IplikLotKaliteKontrol.EFG, IplikLotKaliteKontrol.H1I1, IplikLotKaliteKontrol.Rutubet, IplikLotKaliteKontrol.A3B3C3D3,
               Ik.Adi, IplikLotKaliteKontrol.InsertTarihi, Sk.Adi, IplikLotKaliteKontrol.DegisimTarihi from IplikLot
      left join IplikLotKaliteKontrol on IplikLotKaliteKontrol.IplikLotID = IplikLot.IplikLotID
      left join Harman on Harman.HarmanKayitID = IplikLot.HarmanKayitID
      left join Ham on Ham.HamID = IplikLot.HamID
      Left Join Kullanici IK on IK.KullaniciId = IplikLotKaliteKontrol.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = IplikLotKaliteKontrol.KullaniciId
      where IplikLot.Lot = :FLot and IplikLotKaliteKontrol.Tarih <= :IrsaliyeTarihi
      into :IplikLotKaliteKontrolId, :HamAdi, :Lot, :HarmanYil, :HarmanId, :OlculenNe, :TeorikKalinlikNe,
           :NeCv, :TeorikBukumSayisi, :OlculenBukum, :Alfa, :UYuzde, :CVm, :ThinEksi50, :ThickArti50, :NepsArti200, :HEksi, :Gram, :MYuzdeCv, :RKm,
           :EYuzde, :ECvYuzde, :S3, :Ref, :Renk, :ShEksi, :ThickArti35, :ThinEksi30, :Notu,
           :Hair, :Tinch, :CvYuzde, :A1B1C1D1, :EFG, :H1I1, :Rutubet, :A3B3C3D3,
           :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi  do
    begin
      HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') ' || EgemenIntegerToStr(HarmanId) || ' ';
      suspend;
    end
  end
end
