CREATE PROCEDURE PARTIPAKETGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    PARTIPAKETID INTEGER,
    BRUT INTEGER,
    BRUTSTR VARCHAR(4),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    ACIKLAMA1 VARCHAR(255),
    ACIKLAMA2 VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable FReceteId varchar(20);
declare variable FReceteAdi varchar(40);
begin
  For select PartiPaket.PartiSinifId, PartiPaket.PartiPaketId, PartiPaket.Aciklama1, PartiPaket.Aciklama2, PartiPaket.DovizId, PartiPaket.Brut, PartiPaket.BirimFiyat,
             IK.Adi, PartiPaket.InsertTarihi, SK.Adi, PartiPaket.DegisimTarihi from PartiPaket
    Left Join Kullanici IK on IK.KullaniciId = PartiPaket.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = PartiPaket.KullaniciId
    where PartiPaket.PartiKayitId = :APartiKayitId
    into :PartiSinifId, :PartiPaketId, :Aciklama1, :Aciklama2, :DovizId, :Brut, :BirimFiyat, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    ReceteAdi = null;
    ReceteId = null;
    if (Brut = 1) then BrutStr = 'Br√ºt';
    else BrutStr = 'Net';

    for select PartiAsamalari.ReceteId, Recete.Adi from PartiPaketDetay
    left join partiasamalari on partiasamalari.partiasamalariid = partipaketdetay.partiasamalariid
    left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
    where PartiPaketDetay.PartiPaketId = :PartiPaketId and PartiPaketDetay.PartiAsamalariId is not null and PartiPaketDetay.PartiilaveIslemlerId is null and PartiAsamalari.ReceteID is not null
    into :FReceteId, :FReceteAdi do
    begin
      if (ReceteId is null) then
      begin
        ReceteID = :FReceteID;
        ReceteAdi = :FReceteAdi;
      end
    end
    suspend;
  end
end
