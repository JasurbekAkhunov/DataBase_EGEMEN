declare variable LabReceteId integer;

begin
  select Parti.PartiYil, Parti.PartiId, Parti.AcilisTarihi, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi, Parti.ModelNo, Parti.ReceteId, Recete.Adi,
          LabRecete.EskiLabReceteKodu, LabRecete.Aciklama, Recete.LabrenkGrupID, LabrenkGrup.Adi,
          IslemGrubu.Adi, IslemTipi.Adi, AciliyetDerecesi.Adi,

          BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparis.SiparisGrup3Id, SiparisGrup3.Adi, BoyaSiparis.SiparisGrup4Id, SiparisGrup4.Adi, BoyaSiparis.SiparisTarihi, BoyaSiparis.OrmeSiparisi, Parti.Fason,
          LabreceteAtis.BoyaYuzdesi, DesenVaryant.DesenId, Desen.Adi, RaportB.Olcu, DesenVaryant.VaryantAdi, BoyaSiparis.Notu, Parti.FaturaKesilecek,
          Recete.IlkBoyananPartiKayitId, LabRecete.IlkBoyananPartiKayitId, Recete.SokumUzeriBoya, Recete.Ilave, FirmaYetkili.Adi, BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi,
          ReceteRenkDerinligi.Adi, LabRecete.MusteriRenkKodu, LabRecete.PantoneKodu, BoyaSiparis.OrmeSiparisTipiId, OrmeSiparisTipi.Adi,
          Parti.ReceteYazildi, Parti.RefakatKartiYazmaSayisi, BoyaSiparis.YurtDisi, Desen.DesenGrupID, DesenGrup.Adi, BoyaSiparis.PSKKEtiketTipiId, PSKKEtiketTipi.Adi, LabreceteAtis.Notu, Parti.ArgeNo,
          DesenGravurBoyu.Olcu, DesenSablonEni.Olcu, Desen.TasarimNo, OdemeSekli.Adi, iif(LabReceteAtis.Color is null, -1,LabReceteAtis.Color), Desen.OzelYerlesimDosyaAdi from Parti
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
    left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join Labrecete on Labrecete.LabreceteId = Recete.LabreceteId
    left join LabreceteAtis on LabreceteAtis.LabreceteAtisId = Recete.LabreceteAtisId
    left join LabrenkGrup on LabrenkGrup.LabrenkGrupId = Recete.LabrenkGrupId
    LEFT JOIN DesenVaryant ON DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    LEFT JOIN Desen ON Desen.DesenId = DesenVaryant.DesenId
    Left Join DesenOlcu RaportB on RaportB.DesenOlcuID = Desen.DesenRaportBoyuDesenOlcuId
    LEFT JOIN DesenGrup ON DesenGrup.DesenGrupId = Desen.DesenGrupId
    LEFT JOIN SiparisGrup ON SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    LEFT JOIN SiparisGrup SiparisGrup2 ON SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    LEFT JOIN SiparisGrup SiparisGrup3 ON SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
    LEFT JOIN SiparisGrup SiparisGrup4 ON SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
    LEFT JOIN Kullanici SAK ON SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    LEFT JOIN OrmeSiparisTipi ON OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
    left join RenkDerinligi ReceteRenkDerinligi on ReceteRenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
    left join PSKKEtiketTipi on BoyaSiparis.PSKKEtiketTipiID = PSKKEtiketTipi.PSKKEtiketTipiID
    left join DesenOlcu DesenGravurBoyu on DesenGravurBoyu.DesenolcuId = Desen.GravurBoyuDesenOlcuId
    left join DesenOlcu DesenSablonEni on DesenSablonEni.DesenolcuId = Desen.SablonEniDesenOlcuId
    left join OdemeSekli on OdemeSekli.OdemeSekliId = BoyaSiparis.OdemeSekliID
    where Parti.PartiKayitId = :APartiKayitId
    into :PartiYil, :PartiId, :AcilisTarihi, :PartigrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi, :ModelNo, :ReceteId, :ReceteAdi,
         :EskiLabReceteKodu, :LabReceteAciklama, :LabrenkGrupID, :LabrenkGrupAdi,:IslemGrubuAdi, :IslemTipiAdi,
         :AciliyetDerecesiAdi, :PlanlananImalatTarihi, :TerminTarihi, :Notu, :Sinif, :RafId,
         :BoyaSiparisYil, :BoyaSiparisId, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi, :SiparisTarihi, :OrmeSiparisi, :Fason, :LabReceteBoyaYuzdesi,

         :SiparisiVeren, :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi, :ReceteRenkDerinligiAdi, :LabReceteMusteriRenkKodu, :LabRecetePantoneKodu,
         :OrmeSiparisTipiId, :OrmeSiparisTipiAdi, :ReceteYazildi, :RefakatKartiYazmaSayisi, :SiparisYurtDisi, :DesenGrupID, :DesenGrupAdi, :PSKKEtiketTipiId,
         :PSKKEtiketTipiAdi, :LabreceteAtisNotu, :ArgeNo, :DesenSablonEni, :DesenGravurBoyu, :DesenTasarimNo, :OdemeSekliAdi, :LRAColor, :OYDosyaAdi;

  IlkKezBoyanacak = 0;
