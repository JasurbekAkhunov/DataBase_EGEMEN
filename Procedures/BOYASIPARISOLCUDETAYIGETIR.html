CREATE PROCEDURE BOYASIPARISOLCUDETAYIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISOLCUID INTEGER,
    OLCUID INTEGER,
    OLCUADI VARCHAR(40),
    ADET DOUBLE PRECISION,
    KAPADEDI INTEGER,
    KILO DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    PARTILENENADET NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    KALANADET NUMERIC(12,3))
AS
  declare PartilenenKilo_Rezerv numeric(12,3);
  declare PartilenenAdet_Rezerv numeric(12,3);
  declare PartilenenMetre_Rezerv numeric(12,3);
begin
  for select BoyaSiparisOlcu.BoyaSiparisDetayId, BoyaSiparisOlcu.OlcuId, Olcu.OlcuAdi, BoyaSiparisOlcu.Adet, BoyaSiparisOlcu.Kilo, BoyaSiparisOlcu.Boyasiparisolcuid,
             Ik.Adi, BoyaSiparisOlcu.InsertTarihi, Sk.Adi, BoyaSiparisOlcu.DegisimTarihi, BoyaSiparisOlcu.KapAdedi /*, BoyaSiparisDetay.PartilenenKilo, BoyaSiparisDetay.PartilenenMetre*/ from BoyaSiparisOlcu
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisOlcu.BoyaSiparisDetayId
      left join Olcu on Olcu.OlcuId = BoyaSiparisOlcu.OlcuId
      Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisOlcu.InsertKullaniciId
      Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisOlcu.KullaniciId
      where BoyaSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId
      into :BoyaSiparisDetayId, :OlcuId, :OlcuAdi, :Adet, :Kilo, :BoyaSiparisOlcuId, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :KapAdedi do --, :PartilenenKilo, :PartilenenMetre
      begin
        select coalesce(Sum(BoyaSiparisRezervDetay.Kilo), 0), coalesce(Sum(BoyaSiparisRezervDetay.Adet), 0), coalesce(Sum(BoyaSiparisRezervDetay.Metre), 0) from BoyaSiparisRezervDetay
        left join HamStokDetay  on HamStokDetay .HamStokDetayId  = BoyaSiparisRezervDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where (BoyaSiparisRezervDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId) and (HamStokMaster.OlcuId = :OlcuId)
        into  :PartilenenKilo_Rezerv, :PartilenenAdet_Rezerv, :PartilenenMetre_Rezerv;

        -->  Boya sipariş için; Toplam Partilenen kilo kalan bilgisi ve adet. Kalan Bilgisi BoyaSiparisOlcu adetinden cıkarılarak bulunur. -mahmut-
        KalanKilo = :Kilo - :PartilenenKilo_Rezerv;
        KalanAdet = :Adet - :PartilenenAdet_Rezerv;
        PartilenenKilo  = :PartilenenKilo_Rezerv;
        PartilenenAdet  = :PartilenenAdet_Rezerv;
        PartilenenMetre = :PartilenenMetre_Rezerv;
        suspend;
      end
end
