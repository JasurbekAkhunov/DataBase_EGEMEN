CREATE PROCEDURE SATINALMATALEPRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AONAYDURUMU INTEGER,
    ASIPARISDURUMU INTEGER,
    ATEKLIFDURUMU INTEGER,
    AKAPALIDURUMU INTEGER,
    ATALEPKULLANICIID INTEGER,
    ASATINALMATALEPYERIID INTEGER,
    ADEPARTMANID INTEGER,
    AVERILENSIPARISTIPIID INTEGER,
    ASATINALMABIRIMID INTEGER,
    AURUNID VARCHAR(20),
    AHAMID INTEGER,
    ARAPORURUNTIPI INTEGER,
    AISTEKYERI INTEGER,
    AKULLANICIID INTEGER,
    AMODELKODU MODELKODUSIZE,
    ABASLANGICHGSIPARISYIL INTEGER,
    ABASLANGICHGSIPARISID INTEGER,
    ABITISHGSIPARISYIL INTEGER,
    ABITISHGSIPARISID INTEGER)
RETURNS (
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30),
    TALEPEDENKULLANICIADI VARCHAR(40),
    SATINALMATALEPDETAYID INTEGER,
    SATINALMATALEPTARIHI TIMESTAMP,
    VERILENSIPARISTARIHI VARCHAR(150),
    TEKLIFTALEPTARIHI VARCHAR(150),
    URUNID VARCHAR(20),
    HAMID INTEGER,
    URUNADI VARCHAR(80),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    RENKDIGERADI VARCHAR(50),
    MIKTAR DOUBLE PRECISION,
    SIPARISIVERILENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    BIRIMID VARCHAR(2),
    DOVIZID VARCHAR(5),
    TEDARIKCIFIRMA VARCHAR(1024),
    VERILENSIPARISTIPIID INTEGER,
    VERILENSIPARISTIPIIDSTR VARCHAR(15),
    SATINALMABIRIMID INTEGER,
    SATINALMABIRIMADI VARCHAR(20),
    DEPARTMANID INTEGER,
    DEPARTMANADI VARCHAR(30),
    KAPALI INTEGER,
    KAPALISTR VARCHAR(6),
    SATINALMATALEPYERIID INTEGER,
    SATINALMATALEPYERIADI VARCHAR(30),
    SATINALMAKULLANIMYERIID INTEGER,
    SATINALMAKULLANIMYERIADI VARCHAR(50),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    ONAYTARIHI TIMESTAMP,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(4),
    VERILENSIPARISIDSTR VARCHAR(100),
    VERILENSIPARISFIRMAADI VARCHAR(100),
    VERILENSIPARISONAYDURUMU INTEGER,
    VERILENSIPARISONAYDURUMUSTR VARCHAR(10),
    VERILENSIPGELENMIKTAR DOUBLE PRECISION,
    TEKLIFTALEPIDSTR VARCHAR(100),
    URUNKATEGORIID VARCHAR(4),
    URUNKATEGORIADI VARCHAR(30),
    YURTDISI INTEGER,
    YURTDISISTR VARCHAR(10),
    SONSIPARISIVERILENMIKTAR DOUBLE PRECISION,
    SONSIPARISIVERILENBIRIMFIYAT DOUBLE PRECISION,
    SONSIPARISIVERILENTUTAR DOUBLE PRECISION,
    SONSIPARISIVERILENDOVIZID VARCHAR(5),
    VERILENSIPTOPLAMMIKTAR DOUBLE PRECISION,
    VERILENSIPTOPLAMGELENMIKTAR DOUBLE PRECISION,
    TALEPNOTU VARCHAR(1024),
    NOTU VARCHAR(1024),
    URUNADI2 VARCHAR(80),
    MODELKODU MODELKODUSIZE,
    MODELADI ADI40,
    HGSIPARISIDSTR VARCHAR(30),
    HGSIPARISYIL INTEGER,
    HGSIPARISID INTEGER)
AS
declare variable FUrunAdi        varchar(80);
declare variable FHamAdi         varchar(255);
declare variable VerilenSiparisYil  integer;
declare variable VerilenSiparisId   integer;
declare variable TeklifTalepYil     integer;
declare variable TeklifTalepId      integer;
declare variable FFirmaAdi          varchar(100);
declare variable FOnayDurumu        integer;
declare variable Okey               integer;
declare variable KS                 integer;
declare variable FTarih             TimeStamp;
declare variable FBoyaSiparisKayitId integer;
begin
  For Select Distinct SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId, TK.Adi, SatinAlmaTalep.DepartmanId, Departman.Adi,
             SatinAlmaTalep.SatinAlmaTalepYeriId, SatinAlmaTalepYeri.Adi, SatinAlmaTalepDetay.SatinAlmaTalepDetayId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, SatinAlmaTalepDetay.BoyaSiparisDetayId,
             SatinAlmaTalepDetay.UrunId, SatinAlmaTalepDetay.HamId, Urun.Adi, Ham.Adi, SatinAlmaTalepDetay.Adi,
             SatinAlmaTalepDetay.RenkId, Renk.Adi, Renk.DigerAdi, SatinAlmaTalepDetay.Miktar, SatinAlmaTalepDetay.SiparisiVerilenMiktar, SatinAlmaTalepDetay.BirimFiyat, SatinAlmaTalepDetay.DovizId,
             UrunBirim.BirimId,
             Cast(SubString(Cast(SatinAlmaTalepDetay.TedarikciFirma as Blob Sub_Type Text Character set UTF8) from 1 for 1024) as varchar(1024)),
             SatinAlmaTalepDetay.SatinAlmaKullanimYeriId,
             SatinAlmaKullanimYeri.Adi, SatinAlmaTalep.VerilenSiparisTipiId, VerilenSiparisTipi.Adi,
             SatinAlmaTalep.SatinAlmaBirimId, SatinAlmaBirim.Adi, SatinAlmaTalepDetay.Kapali,
             SatinAlmaTalep.OnayTarihi, SatinAlmaTalep.OnayDurumu, coalesce(Urun.UrunKategoriId, HamTipi.UrunKategoriID),
             UrunKategori.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparis.YurtDisi,
             SatinAlmaTalep.Tarih, BoyaSiparis.BoyaSiparisKayitId,
             cast(Substring(cast(SatinAlmaTalepDetay.TalepNotu as blob sub_type text character set utf8) from 1 for 1024) as varchar(1024)),
             cast(Substring(cast(SatinAlmaTalepDetay.Notu as blob sub_type text character set utf8) from 1 for 1024) as varchar(1024)), Urun.Adi2, Model.ModelId, Model.Adi
             FROM SatinAlmaTalep
    left join SatinAlmaBirim on SatinAlmaBirim.SatinAlmaBirimId = SatinAlmaTalep.SatinAlmaBirimId
    LEFT JOIN SatinAlmaTalepDetay ON SatinAlmaTalepDetay.SatinAlmaTalepKayitId = SatinAlmaTalep.SatinAlmaTalepKayitId
    LEFT JOIN Urun ON Urun.UrunId = SatinAlmaTalepDetay.UrunId
    LEFT JOIN Ham ON Ham.HamId = SatinAlmaTalepDetay.HamId
    LEFT JOIN HamTipi ON HamTipi.HamTipiId = Ham.HamTipiId
    left join UrunBirim on UrunBirim.UrunBirimId = SatinAlmaTalepDetay.UrunBirimId and UrunBirim.UrunId = SatinAlmaTalepDetay.UrunId
    LEFT JOIN BoyaSiparis ON BoyaSiparis.BoyaSiparisKayitId = SatinAlmaTalepDetay.BoyaSiparisKayitId
    left join SatinAlmaKullanimYeri on SatinAlmaKullanimYeri.SatinAlmaKullanimYeriId = SatinAlmaTalepDetay.SatinAlmaKullanimYeriId
    LEFT JOIN Renk ON Renk.RenkId = SatinAlmaTalepDetay.RenkId
    left join Departman on Departman.DepartmanId = SatinAlmaTalep.DepartmanId
    left join SatinAlmaTalepYeri on SatinAlmaTalepYeri.SatinAlmaTalepYeriId = SatinAlmaTalep.SatinAlmaTalepYeriId
    left join VerilenSiparisTipi on VerilenSiparisTipi.VerilenSiparisTipiId = SatinAlmaTalep.VerilenSiparisTipiId
    Left Join Kullanici TK on TK.KullaniciId = SatinAlmaTalep.TalepKullaniciId
    Left Join UrunKategori on UrunKategori.UrunKategoriId = coalesce(Urun.UrunKategoriId, HamTipi.UrunKategoriID)
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    Left Join SatinAlmaTalepOnay on SatinAlmaTalepOnay.SatinAlmaTalepKayitId = SatinAlmaTalep.SatinAlmaTalepKayitId
    Left Join HGSiparisDetay On HGSiparisDetay.HGSiparisDetayId = SatinAlmaTalepDetay.HGSiparisDetayId
    Left Join Model On Model.ModelKayitID = HGSiparisDetay.ModelKayitID
    Left Join HGSiparis On HGSiparis.HGSiparisKayitId = HGSiparisDetay.HGSiparisKayitId
    WHERE SatinAlmaTalep.Tarih >= :ABaslangicTarihi and SatinAlmaTalep.Tarih <= :ABitisTarihi and
         (SatinAlmaTalep.OnayDurumu = :AOnayDurumu or :AOnayDurumu is null) and
         (SatinAlmaTalep.SatinAlmaBirimId = :ASatinAlmaBirimId or :ASatinAlmaBirimId is null) and
         (:ASiparisDurumu is null or (:ASiparisDurumu = 0 and SatinAlmaTalepDetay.SiparisiVerilenMiktar = 0) or
                                     (:ASiparisDurumu = 1 and SatinAlmaTalepDetay.Miktar > SatinAlmaTalepDetay.SiparisiVerilenMiktar) or
                                     (:ASiparisDurumu = 2 and SatinAlmaTalepDetay.Miktar <= SatinAlmaTalepDetay.SiparisiVerilenMiktar) ) and
         (SatinAlmaTalep.TalepKullaniciId = :ATalepKullaniciId or :ATalepKullaniciId is null) and
         (SatinAlmaTalep.SatinAlmaTalepYeriId = :ASatinAlmaTalepYeriId or :ASatinAlmaTalepYeriId is null) and
         (SatinAlmaTalep.DepartmanId = :ADepartmanId or :ADepartmanId is null) and
         (SatinAlmaTalepDetay.Kapali = :AKapaliDurumu or :AKapaliDurumu is null) and
         (SatinAlmaTalepDetay.UrunId = :AUrunId or :AUrunId is null) and
         (SatinAlmaTalepDetay.HamId = :AHamId or :AHamId is null) and
         (SatinAlmaTalepDetay.VerilenSiparisTipiId = :AVerilenSiparisTipiId or :AVerilenSiparisTipiId is null) and
         (:ARaporUrunTipi is null or
          (Urun.UrunTipiID = 0 and (:ARaporUrunTipi = 0 or :ARaporUrunTipi = 3)) or
          (Urun.UrunTipiID = 1 and (:ARaporUrunTipi = 1 or :ARaporUrunTipi = 3)) or
          (Urun.UrunTipiID = 2 and :ARaporUrunTipi = 2) ) and
         (SatinAlmaTalep.IstekYeri = :AIstekYeri or :AIstekYeri is null ) and
         (Model.ModelId = :AModelKodu or :AModelKodu is null) and
         (:ABaslangicHGSiparisYil = 0 or
         ((HGSiparis.HGSiparisYil > :ABaslangicHGSiparisYil or (HGSiparis.HGSiparisYil = :ABaslangicHGSiparisYil and HGSiparis.HGSiparisId >= :ABaslangicHGSiparisId)) and
         (HGSiparis.HGSiparisYil < :ABitisHGSiparisYil or (HGSiparis.HGSiparisYil = :ABitisHGSiparisYil and HGSiparis.HGSiparisId <= :ABitisHGSiparisId)))) and
         (SatinAlmaTalepOnay.OnaylayacakKullaniciId = :AKullaniciId or :AKullaniciId is null  )
  into :SatinAlmaTalepYil, :SatinAlmaTalepId, :TalepEdenKullaniciAdi, :DepartmanId, :DepartmanAdi, :SatinAlmaTalepYeriId,
       :SatinAlmaTalepYeriAdi, :SatinAlmaTalepDetayId, :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId, :UrunId, :HamId, :UrunAdi, :FHamAdi,
       :FUrunAdi, :RenkId, :RenkAdi, :RenkDigerAdi, :Miktar, :SiparisiVerilenMiktar, :BirimFiyat, :DovizId, :BirimId, :TedarikciFirma, :SatinAlmaKullanimYeriId, :SatinAlmaKullanimYeriAdi,
       :VerilenSiparisTipiId, :VerilenSiparisTipiIdStr, :SatinAlmaBirimId, :SatinAlmaBirimAdi, :Kapali, :OnayTarihi, :OnayDurumu, :UrunKategoriId, :UrunKategoriAdi,
       :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :YurtDisi, :SatinAlmaTalepTarihi, :FBoyaSiparisKayitId, :TalepNotu, :Notu, :UrunAdi2, :ModelKodu, :ModelAdi do
  begin
    if (UrunId is null and HamId is null) then UrunAdi = FUrunAdi;
    else if (HamId is not null) then UrunAdi = FHamAdi;

    SatinAlmaTalepIdStr = '';
    if (SatinAlmaTalepId is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId);

    BoyaSiparisIdStr = '';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    HGSiparisIdStr = '';
    if (HGSiparisYil is not null) then HGSiparisIdStr = '(' || EgemenIntegerToStr(HGSiparisYil) || ') '  || EgemenIntegerToStr(HGSiparisId)  || ' ';

    KalanMiktar = Miktar - SiparisiVerilenMiktar;

    if (OnayDurumu = 1) then OnayDurumuStr = 'Okey';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    if (Kapali = 1) then KapaliStr = 'Kapalı';
    else KapaliStr = '';

    VerilenSiparisIdStr = '';
    VerilenSiparisFirmaAdi = '';
    VerilenSiparisOnayDurumu = Null;
    VerilenSiparisTarihi= '';

    For Select VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisID, Firma.Adi, VerilenSiparis.OnayDurumu, VerilenSiparis.SiparisTarihi, Sum(VerilenSiparisDetay.GelenMiktar) from VerilenSiparisDetay
      left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitID = VerilenSiparisDetay.VerilenSiparisKayitID
      left join Firma on Firma.FirmaID = VerilenSiparis.FirmaID
      where VerilenSiparisDetay.SatinAlmaTalepDetayId = :SatinAlmaTalepDetayId
      group by VerilenSiparis.VerilenSiparisYil, VerilenSiparis.VerilenSiparisID, Firma.Adi, VerilenSiparis.OnayDurumu, VerilenSiparis.SiparisTarihi
      into :VerilenSiparisYil, :VerilenSiparisID, :FFirmaAdi, :FOnayDurumu, :FTarih, :VerilenSipGelenMiktar do
    begin
      if ( Char_Length(VerilenSiparisIdStr) < 70 ) then
      begin
        if (VerilenSiparisIdStr <> '') then VerilenSiparisIdStr = VerilenSiparisIdStr || cast(', ' as varchar(2) character set UTF8);
        VerilenSiparisIdStr = VerilenSiparisIdStr || '(' || EgemenIntegerToStr(VerilenSiparisYil)  || ') ' || EgemenIntegerToStr(VerilenSiparisId);
      end

      if ( Char_Length(VerilenSiparisFirmaAdi) + Char_Length(FFirmaAdi) < 97 ) then
      begin
        if (VerilenSiparisFirmaAdi <> '') then VerilenSiparisFirmaAdi = VerilenSiparisFirmaAdi || cast(', ' as varchar(2) character set UTF8);
        VerilenSiparisFirmaAdi = VerilenSiparisFirmaAdi || FFirmaAdi;
      end

      if ( Char_Length(VerilenSiparisTarihi) + Char_Length(FTarih) < 130 ) then
      begin
        if (VerilenSiparisTarihi <> '') then VerilenSiparisTarihi = VerilenSiparisTarihi || cast(', ' as varchar(2) character set UTF8);
        VerilenSiparisTarihi = VerilenSiparisTarihi || TimeStampToDateTimeStr(FTarih);
      end

      VerilenSiparisOnayDurumu = FOnayDurumu;
    end

    if (VerilenSiparisOnayDurumu = 1) then VerilenSiparisOnayDurumuStr = 'Onaylı';
    else if (VerilenSiparisOnayDurumu = 2) then VerilenSiparisOnayDurumuStr = 'Ret';
    else VerilenSiparisOnayDurumuStr = '';


    Okey = 1;

    if (ATeklifDurumu is not null) then
    begin
      Select Count(*) from TeklifTalepDetay where TeklifTalepDetay.SatinAlmaTalepDetayID = :SatinAlmaTalepDetayID into :KS;

      if (ATeklifDurumu = 0 and KS > 0) then Okey = 0;
      else if (ATeklifDurumu = 1 and KS = 0) then Okey = 0;
    end

    if (Okey = 1) theN
    begin
      TeklifTalepIdStr = '';
      TeklifTalepTarihi = '';

      /*Urun icin son alinan miktar*/
      if (UrunId is Not Null and (AUrunId is Not Null or  (ABitisTarihi - ABaslangicTarihi) <= 92 )) then
      begin
        select first 1 StokDetay.Miktar, StokDetay.BirimFiyat, StokDetay.Miktar * StokDetay.BirimFiyat, Irsaliye.DovizId from VerilenSiparisDetay
          left join StokDetay on StokDetay.VerilenSiparisDetayId = VerilenSiparisDetay.VerilenSiparisDetayId
          left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
          where VerilenSiparisDetay.UrunId = :UrunId
          order by StokDetay.Tarih desc
        into :SonSiparisiVerilenMiktar, :SonSiparisiVerilenBirimFiyat, :SonSiparisiVerilenTutar, :SonSiparisiVerilenDovizId;
      end

      VerilenSipToplamMiktar  = Null;
      VerilenSipToplamGelenMiktar = Null;
      if (FBoyaSiparisKayitId is not Null) then
      begin
        select Sum(VerilenSiparisDetay.Miktar), Sum(VerilenSiparisDetay.GelenMiktar)
        from VerilenSiparisDetay
        where (VerilenSiparisDetay.UrunId = :UrunId or VerilenSiparisDetay.HamId = :HamId)
        and Coalesce(VerilenSiparisDetay.RenkId, 0) = Coalesce(:RenkId, 0)
        and VerilenSiparisDetay.BoyaSiparisKayitId = :FBoyaSiparisKayitId into :VerilenSipToplamMiktar, :VerilenSipToplamGelenMiktar;
      end


      For Select TeklifTalep.TeklifTalepYil, TeklifTalep.TeklifTalepID, TeklifTalep.TalepTarihi from TeklifTalepDetay
        left join TeklifTalep on TeklifTalep.TeklifTalepKayitID = TeklifTalepDetay.TeklifTalepKayitID
        where TeklifTalepDetay.SatinAlmaTalepDetayId = :SatinAlmaTalepDetayId
        group by TeklifTalep.TeklifTalepYil, TeklifTalep.TeklifTalepID, TeklifTalep.TalepTarihi
        into :TeklifTalepYil, :TeklifTalepID, :FTarih do
      begin
        if ( Char_Length(TeklifTalepIdStr) < 70 ) then
        begin
          if (TeklifTalepIdStr <> '') then TeklifTalepIdStr = TeklifTalepIdStr || cast(', ' as varchar(2) character set UTF8);
          TeklifTalepIdStr = TeklifTalepIdStr || '(' || EgemenIntegerToStr(TeklifTalepYil)  || ') ' || EgemenIntegerToStr(TeklifTalepId);
        end

        if ( Char_Length(TeklifTalepTarihi) + Char_Length(FTarih) < 130 ) then
        begin
          if (TeklifTalepTarihi <> '') then TeklifTalepTarihi = TeklifTalepTarihi || cast(', ' as varchar(2) character set UTF8);
          TeklifTalepTarihi = TeklifTalepTarihi || TimeStampToDateTimeStr(FTarih);
        end

      end

      if (YurtDisi = 1) then YurtDisiStr = 'Yurtdışı';
      else YurtDisiStr = '';

      suspend;
    end
  end
