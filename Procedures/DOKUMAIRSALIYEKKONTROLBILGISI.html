CREATE PROCEDURE DOKUMAIRSALIYEKKONTROLBILGISI(
    AORMEIRSALIYEID INTEGER)
RETURNS (
    IDSTR VARCHAR(30),
    DOKUMAKALITEKONTROLID INTEGER,
    STOKHAREKETDETAYID INTEGER,
    GIRISKALITEKONTROLID INTEGER,
    DOKUMAPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    VARDIYAID INTEGER,
    SATISSTOKHAREKETID INTEGER,
    VARDIYATARIHI TIMESTAMP,
    VARDIYA INTEGER,
    VARDIYAAMIRIPERSONELADI VARCHAR(40),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    TARIHI TIMESTAMP,
    RAFID VARCHAR(10),
    FISSERINO VARCHAR(2),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    TOPNO INTEGER,
    BRUTMIKTAR DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    LOT VARCHAR(255),
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    TOPFIRE DOUBLE PRECISION,
    KILO BOYAMIKTARSIZE,
    DKKINSERTTARIHI TIMESTAMP,
    HAMADI ADI255,
    KALITENO INTEGER,
    DOKUMATALIMATLOT VARCHAR(150),
    ADET INTEGER,
    METRE DOUBLE PRECISION,
    BRUTKILO BOYAMIKTARSIZE)
AS
declare variable FLot varchar(20);
Declare Variable DokumaKumasiKilodanTakipEt SmallInt;
Declare Variable hstDokumaKumas SmallInt;
Declare Variable HamTipiId Integer;
begin
  Select DokumaKumasiKilodanTakipEt from Sistem into :DokumaKumasiKilodanTakipEt;
  Select hstDokumaKumas from HamTipiGetir into :hstDokumaKumas;

  For Select DokumaKaliteKontrol.DokumaKaliteKontrolId, DokumaKaliteKontrol.DokumaPlanlamaId, DokumaKaliteKontrol.OrmeVardiyaId, DokumaKaliteKontrol.SatisStokHareketId,
             DokumaKaliteKontrol.VardiyaTarihi, DokumaKaliteKontrol.Vardiya, VAP.Adi, DokumaKaliteKontrol.PersonelId, P.Adi,
             DokumaKaliteKontrol.RafId,  DokumaKaliteKontrol.TopNo, DokumaKaliteKontrol.Miktar, DokumaKaliteKontrol.Sakat,
             OrmeFis.Serino, OrmeFis.FisNo, OrmeFis.Fistarihi,  DokumaPlanlama.UretimPartiNo, DokumaPlanlama.DokumaMakinaId, DokumaMakina.Adi,
             DokumaKaliteKontrol.Kilo, DokumaKaliteKontrol.InsertTarihi, Ham.Adi, DokumaKaliteKontrol.KaliteNo, DokumaPlanlama.Lot, DokumaKaliteKontrol.Adet, Ham.HamTipiId from DokumaKaliteKontrol
    left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaKaliteKontrol.DokumaPlanlamaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = DokumaPlanlama.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = DokumaPlanlama.DokumaMakinaId
    left join StokHareket on StokHareket.StokHareketId = DokumaKaliteKontrol.StokHareketId
    left join OrmeFis on OrmeFis.OrmeFisId  = StokHareket.OrmeFisId
    left join Personel VAP on VAP.PersonelId  = DokumaKaliteKontrol.VardiyaAmiriPersonelId
    left join Personel P on P.PersonelId = DokumaKaliteKontrol.PersonelId
    left join Kullanici IK on IK.KullaniciId = DokumaKaliteKontrol.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId = DokumaKaliteKontrol.KullaniciId
    left join Ham on Ham.Hamid = OrmeSiparisDetay.HamId
    where DokumaKaliteKontrol.OrmeIrsaliyeId = :AOrmeIrsaliyeId
    order by DokumaKaliteKontrol.DokumaPlanlamaId, DokumaKaliteKontrol.DokumaKaliteKontrolId
  into :DokumaKaliteKontrolId, :DokumaPlanlamaId, :VardiyaId, :SatisStokHareketId, :VardiyaTarihi, :Vardiya, :VardiyaAmiriPersonelAdi,
       :PersonelId, :PersonelAdi, :RafId, :TopNo, :Metre, :Sakat,
       :FisSerino, :FisNo, :Fistarihi, :UretimPartiNo, :MakinaId, :MakinaAdi, :Kilo, :DKKInsertTarihi, :HamAdi, :KaliteNo, :DokumaTalimatLot, :Adet, :HamTipiId do
  begin
    IdStr = DokumaKaliteKontrolId;

    SakatStr = '';
    if (Sakat = 1) then SakatStr = 'Sakat';

    if (HamTipiID <> hstDokumaKumas or DokumaKumasiKilodanTakipEt = 1) then
    begin
      BrutMiktar = Kilo;
      Miktar = Kilo;
    end
    else
    begin
      BrutMiktar = Metre;
      Miktar = Metre;
    end


    Lot = '';

    For Select Distinct DokumaPlanlamaRecete.Lot from DokumaPlanlamaRecete
      where DokumaPlanlamaRecete.DokumaPlanlamaId = :DokumaPlanlamaId
      into :FLot do
    begin
      if (FLot is not null and FLot <> '') then
      begin
        if (Char_Length(Lot) + Char_Length(FLot) < 250) then
        begin
          if (Lot <> '') then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
          Lot = Lot || FLot;
        end
      end
    end

    suspend;
  end
  Adet = null;
  For Select StokHareketDetay.Barkod, Null, GirisKaliteKontrol.GirisKaliteKontrolId, StokHareketDetay.StokHareketDetayId, Null, Null, StokHareket.StokHareketId,
             Null, Null, Null, Null, Null, OrmeRafDetay.RafId,
             GirisKaliteKontrol.TopNo, StokHareketDetay.Metre, StokHareket.Sakat,
             OrmeFis.Serino, OrmeFis.FisNo, OrmeFis.Fistarihi,
             Null, DokumaMakina.DokumaMakinaId, DokumaMakina.Adi, StokHareketDetay.Kilo, StokHareketDetay.BrutKilo,
             Ham.Adi, GirisKaliteKontrol.KaliteNo, Ham.HamTipiId from StokHareket
    left join StokHareketDetay on StokHareketDetay.StokHareketId = StokHareket.StokHareketId
    left join GirisKaliteKontrol  on 'G-'||GirisKaliteKontrol.GirisKaliteKontrolId = StokHareketDetay.Barkod
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join DokumaMakina on DokumaMakina.DokumaMakinaId = StokHareket.DokumaMakinaId
    left join OrmeFis on OrmeFis.OrmeFisId  = StokHareket.OrmeFisId
    left join Kullanici IK on IK.KullaniciId  = GirisKaliteKontrol.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = GirisKaliteKontrol.KullaniciId
    left join Ham on Ham.HamId = StokHareket.HamId
    left join OrmeRafDetay on OrmeRafDetay.StokHareketId = StokHareket.StokHareketId
    where StokHareket.OrmeIrsaliyeId = :AOrmeIrsaliyeId and StokHareketDetay.StokHareketDetayId is not null
    order by GirisKaliteKontrol.Tarih, GirisKaliteKontrol.InsertTarihi
  into :IdStr, :DokumaKaliteKontrolId, :GirisKaliteKontrolId, :StokHareketDetayId, :DokumaPlanlamaId, :VardiyaId, :SatisStokHareketId, :VardiyaTarihi, :Vardiya, :VardiyaAmiriPersonelAdi,
       :PersonelId, :PersonelAdi, :RafId, :TopNo, :Metre, :Sakat,
       :FisSerino, :FisNo, :Fistarihi, :UretimPartiNo, :MakinaId, :MakinaAdi, :Kilo, :BrutKilo, :HamAdi, :KaliteNo, :HamTipiId do
  begin
    SakatStr = '';
    if (Sakat = 1) then SakatStr = 'Sakat';

    if (HamTipiID <> hstDokumaKumas or DokumaKumasiKilodanTakipEt = 1) then
    begin
      BrutMiktar = BrutKilo;
      Miktar = Kilo;
    end
    else
    begin
      BrutMiktar = Metre;
      Miktar = Metre;
    end

    Lot = '';

    suspend;
  end
end
