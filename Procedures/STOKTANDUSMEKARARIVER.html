CREATE PROCEDURE STOKTANDUSMEKARARIVER(
    APARTIKAYITID INTEGER,
    APARTIASAMALARIID INTEGER,
    ANEREDENGELDI INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable FReceteIsEmriYil       integer;
declare variable FReceteIsEmriId        integer;
declare variable FReceteStoktanDusmeSekli integer;
declare variable FPartiKayitId           integer;
declare variable FPartiAsamalariId       integer;
declare variable FReceteId               varchar(20);
declare variable FBanyoCarpani           Double Precision;
declare variable FBanyoCarpani2          Double Precision;
declare variable FBanyoCarpani3          Double Precision;
declare variable FBanyoCarpani4          Double Precision;
declare variable FKoyultma               Double Precision;
declare variable FStoktanDusuldu         integer;
declare variable FKayitSayisi            integer;
declare variable KS                      integer;
declare variable FDozajlamaSistemiId     integer;
declare variable TeoriktenDus            integer;
declare variable SarfiyatlariTeoriktenDus integer;
begin
  if (ANeredenGeldi < 0 or ANeredenGeldi > 3) then exception StoktanDusmeParametreHatasi1;

  select Count(*) from PartiAsamalari where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId and PartiAsamalari.PartiKayitId = :APartiKayitId into :FKayitSayisi;

  if (FKayitSayisi <> 1) then exception StoktanDusmeParametreHatasi2;

  select ReceteStoktanDusmeSekli from Sistem Where SistemId = 1 into :FReceteStoktanDusmeSekli;
  if (FReceteStoktanDusmeSekli <> ANeredenGeldi) then exit;

  Select Max(Makina.DozajlamaSistemiId) from PartiAsamaStokDetay
    left join Makina on Makina.MakinaId = PartiAsamaStokDetay.MakinaId
    where PartiAsamaStokDetay.PartiAsamalariId = :APartiAsamalariId
    into :FDozajlamaSistemiId;

  Select ReceteIsEmriYil, ReceteIsEmriId from ReceteIsEmriDetay
    where ReceteIsEmriDetay.PartiAsamalariId = :APartiAsamalariId
    into :FReceteIsEmriYil, :FReceteIsEmriId;

  Select ReceteId from PartiAsamalari where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId into :FReceteId;

  if (FDozajlamaSistemiID is not null) then
  begin
    Select SarfiyatlariTeoriktenDus from DozajlamaSistemi where DozajlamaSistemi.DozajlamaSistemiID = :FDozajlamaSistemiID into :SarfiyatlariTeoriktenDus;

    if (SarfiyatlariTeoriktenDus = 0) then
    begin
      Select Count(*) from ReceteDetay
      left join DozajlamaSistemiUrun on DozajlamaSistemiUrun.UrunID = ReceteDetay.UrunId
      where ReceteDetay.ReceteID = :FReceteID and ReceteDetay.UrunID is not null and DozajlamaSistemiUrun.DozajlamaSistemiID = :FDozajlamaSistemiID
      into :TeoriktenDus;
    end
    else TeoriktenDus = 1;
  end
  else TeoriktenDus = 1;

  if (TeoriktenDus > 0) then
  if (FReceteStoktanDusmeSekli = 0) then  /* manuel */
  begin
/* eger partinin o asamasi daha once stoktan düsülmedi ise manuel stoktan düş */
    Select ReceteId, BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, Koyultma, StoktanDusuldu from PartiAsamalari where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId
      into :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4, :FKoyultma, :FStoktanDusuldu;

    if (FStoktanDusuldu = 0) then  Execute procedure PartininRecetesiniStoktanDus(:APartiKayitId, :APartiAsamalariId, :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4, :FKoyultma, :AKullaniciId, :FDozajlamaSistemiID);
  end
  else if (FReceteStoktanDusmeSekli = 1) then /* recete yazdirilinca */
  begin
    if (FReceteIsEmriYil is not null) then
    begin
/*    iş emrinin  bütün partiler için stoktan düsülmemis olanlari düş */
      For Select ReceteIsEmriDetay.PartiKayitId, ReceteIsEmriDetay.PartiAsamalariId, ReceteIsEmri.ReceteId, ReceteIsEmri.BanyoCarpani,
                 ReceteIsEmri.BanyoCarpani2, ReceteIsEmri.BanyoCarpani3, ReceteIsEmri.BanyoCarpani4, ReceteIsEmri.Koyultma from ReceteIsEmri
         left Join ReceteIsEmriDetay on ReceteIsEmriDetay.ReceteIsEmriYil = ReceteIsEmri.ReceteIsEmriYil and ReceteIsEmriDetay.ReceteIsEmriId = ReceteIsEmri.ReceteIsEmriId
         left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = ReceteIsEmriDetay.PartiAsamalariId
         where ReceteIsEmri.ReceteIsEmriYil = :FReceteIsEmriYil and ReceteIsEmri.ReceteIsEmriId = :FReceteIsEmriId and PartiAsamalari.StoktanDusuldu = 0
         into :FPartiKayitId, :FPartiAsamalariId, :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4, :FKoyultma do
      begin
        Execute procedure PartininRecetesiniStoktanDus(:FPartiKayitId, :FPartiAsamalariId, :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4, :FKoyultma, :AKullaniciId, :FDozajlamaSistemiID);
      end
    end
    else
    begin
/* eger partinin o asamasi daha once stoktan düsülmedi ise stoktan düş */
      Select ReceteId, BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, Koyultma, StoktanDusuldu from PartiAsamalari where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId
        into :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4, :FKoyultma, :FStoktanDusuldu;

      if (FStoktanDusuldu = 0) then Execute procedure PartininRecetesiniStoktanDus(:APartiKayitId, :APartiAsamalariId, :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4, :FKoyultma, :AKullaniciId, :FDozajlamaSistemiID);
    end
  end
  else if (FReceteStoktanDusmeSekli = 2) then /* imalat baslayinca */
  begin
    select Count(*) from PartiAsamaStok
      where PartiAsamaStok.PartiAsamalariId = :APartiAsamalariId and Giristarihi is not null into :FKayitSayisi;

    if (FKayitSayisi > 0) then
    begin
      Select ReceteId, BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, Koyultma, StoktanDusuldu from PartiAsamalari
       where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId
       into :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4, :FKoyultma, :FStoktanDusuldu;

      if (FStoktanDusuldu = 0) then
      begin
        Execute procedure PartininRecetesiniStoktanDus(:APartiKayitId, :APartiAsamalariId, :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4, :FKoyultma, :AKullaniciId, :FDozajlamaSistemiID);
      end
    end
  end
  else if (FReceteStoktanDusmeSekli = 3) then /* imalat bitince */
  begin
    select Count(*) from PartiAsamaStok
      where PartiAsamaStok.PartiAsamalariId = :APartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1 and CikisTarihi is not null into :FKayitSayisi;

    if (FKayitSayisi > 0) then
    begin
      Select ReceteId, BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, Koyultma, StoktanDusuldu from PartiAsamalari
       where PartiAsamalari.PartiAsamalariId = :APartiAsamalariId
       into :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4, :FKoyultma, :FStoktanDusuldu;
      if (FStoktanDusuldu = 0) then Execute procedure PartininRecetesiniStoktanDus(:APartiKayitId, :APartiAsamalariId, :FReceteId, :FBanyoCarpani, :FBanyoCarpani2, :FBanyoCarpani3, :FBanyoCarpani4,  :FKoyultma, :AKullaniciId, :FDozajlamaSistemiID);
    end
  end
end
