CREATE PROCEDURE ORMEKARMASTOKHAREKETFIREHESABI(
    ATARIHTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    ASIPARISFIRMAID VARCHAR(20),
    AORMEMAKINAID VARCHAR(10),
    AALIS INTEGER,
    ASATIS INTEGER,
    AALISIADE INTEGER,
    ASATISIADE INTEGER,
    AFASONALIS INTEGER,
    AFASONURETIMECIKIS INTEGER,
    AFASONURETIMECIKISIADE INTEGER,
    AURETIMECIKIS INTEGER,
    AURETIMECIKISIADE INTEGER,
    AFASONURETIM INTEGER,
    AURETIM INTEGER,
    AKAYDIRMA INTEGER,
    ASARF INTEGER,
    AREKLAMASYONGIRIS INTEGER,
    AREKLAMASYONCIKIS INTEGER,
    AHSTIPLIK INTEGER,
    AHSTLYCRA INTEGER,
    AHSTKUMAS INTEGER,
    ABUKUM INTEGER,
    AORME INTEGER,
    ABOYA INTEGER,
    AHAMID INTEGER,
    ARENKID INTEGER,
    AURETIMPARTINO VARCHAR(30),
    ACIKISPARTINO VARCHAR(30),
    ABOYATALIMATKAYITID INTEGER,
    AKUMASCINSIID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    AARACID VARCHAR(11),
    ASEVKSEFERNO INTEGER,
    ASTOKISLEMTIPIID INTEGER,
    AKALITENO INTEGER,
    ALOT VARCHAR(150),
    AFIRMAADI2YIGOREBILIR INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AMUSTERISIPARISNO VARCHAR(50),
    ADEPOID VARCHAR(10),
    ARAFTRANSFERFISI INTEGER,
    AREFERANSNO VARCHAR(30),
    ABUKUMTALIMATID INTEGER,
    AKAPALIDURUMU INTEGER,
    ALOTFILTRE INTEGER,
    AMODELKODU MODELKODUSIZE,
    ARAFID RAFKODUSIZE)
RETURNS (
    ORMEFIRE DOUBLE PRECISION,
    ORMEFIREORANI DOUBLE PRECISION)
AS
declare variable uitOrme        integer;
declare variable HareketTipi    integer;
declare variable HamIade        integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable FMiktar        Double Precision;
declare variable ToplamOrmeyeCikis   Double Precision;
declare variable ToplamOrulen        Double Precision;
declare variable FireHesapTipi       integer;
begin
  select uitOrme from UretimIslemTipiGetir into :uitOrme;

  select shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim, ShtUretim from OrmeStokHareketTipiGetir
         into :shtFasonUretimeCikis,  :shtUretimeCikis, :shtFasonUretim, :ShtUretim;

  ToplamOrmeyeCikis = 0;
  ToplamOrulen = 0;

  Select FireHesapTipi from Sistem where SistemId = 1 into :FireHesapTipi;

  For Select HareketTipi, HamIade, ToplamMiktar From OrmeKarmaStokHareketToplami(:ATarihTipi, :ABaslangicTarihi, :ABitisTarihi, :ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :AFirmaId, :ASiparisFirmaId, :AOrmeMakinaId,
                               :AAlis, :ASatis, :AAlisIade, :ASatisIade, :AFasonAlis, :AFasonUretimeCikis, :AFasonUretimeCikisIade, :AUretimeCikis, :AUretimeCikisIade, :AFasonUretim, :AUretim, :AKaydirma, :ASarf, :AReklamasyonGiris,
                               :AReklamasyonCikis, :AhstIplik, :AhstLycra, :AhstKumas, :ABukum, :AOrme, :ABoya, :AHamId, :ARenkId, :AUretimPartiNo, :ACikisPartiNo, :ABoyaTalimatKayitId, :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId,
                               :AFasonSiparisler, :AFasonSatisSiparisler, :ASatisSiparisler, :AAracId, :ASevkSeferNo, :AStokIslemTipiId, :AKaliteNo, :ALot, :AFirmaAdi2yiGorebilir, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id,
                               :ASiparisGrup4Id, :AMusteriSiparisNo, :ADepoId, :ARafTransferFisi, :AReferansNo, :ABukumTalimatId, :AKapaliDurumu, :ALotFiltre, :AModelKodu, :ARafId)
    where UretimIslemTipiId = :uitOrme
  into :HareketTipi, :HamIade, :FMiktar do
  begin
    if (HareketTipi = shtFasonUretimeCikis or HareketTipi = shtUretimeCikis) then ToplamOrmeyeCikis = ToplamOrmeyeCikis + FMiktar;
    else  if (HareketTipi = shtFasonUretim or HareketTipi = shtUretim) then
    begin
      if (HamIade = 0) then ToplamOrulen = ToplamOrulen + FMiktar;
      else ToplamOrmeyeCikis = ToplamOrmeyeCikis - FMiktar;
     end
  end

  OrmeFire = ToplamOrmeyeCikis - ToplamOrulen;
  OrmeFireOrani = 0;

  if (FireHesapTipi = 0) then
  begin
    if (ToplamOrulen > 0) then OrmeFireOrani = ((ToplamOrmeyeCikis - ToplamOrulen) / ToplamOrulen) * 100;
  end
  else if (ToplamOrmeyeCikis > 0) then OrmeFireOrani = ((ToplamOrmeyeCikis - ToplamOrulen) / ToplamOrmeyeCikis) * 100;
  suspend;
end
