CREATE PROCEDURE HGKOLIDETAYGETIR(
    AHGKOLIFISIID INTEGER)
RETURNS (
    HGKOLIID INTEGER,
    HGKOLIDETAYID INTEGER,
    URUNID URUNKODUSIZE,
    URUNADI ADI80,
    HGSIPARISDETAYASORTIID INTEGER,
    HGSIPARISDETAYASORTIADI ADI100,
    ASORTISTR VARCHAR(10),
    MIKTAR INTEGER,
    RENKID INTEGER,
    RENKADI ADI50,
    BEDENID INTEGER,
    BEDENADI ADI30)
AS
begin
  for select HgKoli.HgKoliId, HgKoliDetay.HgKoliDetayId, HgKoliDetay.UrunId, Urun.Adi, HgKoliDetay.HgSiparisDetayAsortiId, HgSiparisDetayAsorti.Adi,
             iif(HgSiparisDetayAsorti.HgSiparisDetayAsortiId is null, '', 'Evet'), HgKoliDetay.Miktar, Renk.RenkId, Renk.Adi, Beden.BedenId, Beden.Adi from HgKoli
      left join HgKoliDetay on HgKoliDetay.HgKoliId = HgKoli.HgKoliId
      left join Urun on Urun.UrunId = HgKoliDetay.UrunId
      left join HgSiparisDetayAsorti on HgSiparisDetayAsorti.HgSiparisDetayAsortiId = HgKoliDetay.HgsiparisDetayAsortiId
      left join ModelKumas on ModelKumas.ModelKumasId = Urun.ModelKumasId
      left join Renk on Renk.RenkId = ModelKumas.RenkId
      left join ModelBeden on ModelBeden.ModelBedenId = Urun.ModelBedenId
      left join Beden on Beden.BedenId = ModelBeden.BedenId
      where HgKoli.HgKoliFisiId = :AHgKoliFisiId and
      HgKoliDetay.HgKoliDetayId is not null
      order by HgKoliDetay.HgKoliDetayId
      into :HgKoliId, :HgKoliDetayId, :UrunId, :UrunAdi, :HgSiparisDetayAsortiId, :HgSiparisDetayAsortiAdi,
             :AsortiStr, :Miktar, :RenkId, :RenkAdi, :BedenId, :BedenAdi do
  begin
    suspend;
  end
end
