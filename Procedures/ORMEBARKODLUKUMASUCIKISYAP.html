CREATE PROCEDURE ORMEBARKODLUKUMASUCIKISYAP(
    AKALITEKONTROLID INTEGER,
    AIRSALIYEID INTEGER,
    ABOYATALIMATKAYITID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    ABOYASIPARISDETAYID INTEGER,
    AORMESIPARISDETAYID INTEGER,
    ABARKODNO VARCHAR(30),
    AHAREKETTIPI INTEGER,
    AKILO DOUBLE PRECISION,
    ABRUTKILO DOUBLE PRECISION,
    AMETRE DOUBLE PRECISION,
    AKULLANICIID INTEGER)
AS
declare variable FIrsaliyeId            integer;
declare variable FUIrsaliyeId           integer;
declare variable OrmePlanlamaId         integer;
declare variable KKID                   integer;
declare variable TalBoyaSiparisKayitId  integer;
declare variable BoyaSiparisDetayId     integer;
declare variable TalOrmeSiparisDetayId  integer;
declare variable TalHamId               integer;
declare variable BoyaSiparisKayitId     integer;
declare variable OrmeSiparisDetayId     integer;
declare variable HamId                  integer;
declare variable StokHareketId          integer;
declare variable OrmeRafDetayId         integer;
declare variable RafId                  varchar(10);
declare variable OrmeRafId              integer;
declare variable Miktar                 numeric(12,3);
declare variable Metre                  numeric(12,3);
declare variable SatisFiyati            double precision;
declare variable KayitSayisi            integer;
declare variable KayitStokHareketId     integer;
declare variable SiparisDovizId         varchar(5);
declare variable IrsaliyeDovizId        varchar(5);
declare variable IrsaliyeTarihi         timestamp;
declare variable DovizKuru              double precision;
declare variable SatisStokHareketId     integer;
declare variable UretimeCikisStokHareketId      integer;
declare variable Sakat                  integer;
declare variable FHamId                 integer; /* baska siparise satiyor ise kullaniyoruz */
declare variable FBaskaSipariseSatilabilir      integer; /* baska siparise satiyor ise kullaniyoruz */
declare variable FIptal                         integer; /* baska siparise satiyor ise kullaniyoruz */
declare variable FLot                   varchar(150);
declare variable FDepoId                varchar(10);
declare variable FKTBoyaSiparisKayitId  integer;
declare variable FKumasTemin            smallint;
declare variable FKontrol               integer;
declare variable IK                     varchar(5);
declare variable PaketId                integer;
declare variable shtAlis                integer;
declare variable shtFasonUretim         integer;
declare variable shtUretim              integer;
declare variable FBoyaSiparisKayitID    integer;
declare variable FBoyaSiparisDetayID    integer;
declare variable FBoyaSiparisBaskiId    integer;
declare variable FRenkId                integer;
declare variable FUreticiId             integer;
declare variable FMelanjKodu            varchar(20);
declare variable FOrmeSiparisDetayID    integer;
declare variable DokumaHareketi         integer;
declare variable MiktarBirim            integer;
declare variable DovizId                DovizSize;
declare variable KaynakDovizId          DovizSize;
declare variable cBarkod                integer;
declare variable GELENORMESIPDETAYID    integer;
declare variable FHamStokIrsaliyeId     integer;
declare variable FEgemenFirmaId         integer;
declare variable FResimDosyaAdi1        varchar(255);
declare variable FResimDosyaAdi2        varchar(255);
begin
  /*Boya'da boyle bir secenek yok */
  FBaskaSipariseSatilabilir = 0;

  select Sistem.EgemenFirmaId from Sistem into :FEgemenFirmaId;
  FResimDosyaAdi1 = null;
  FResimDosyaAdi2 = null;

  select HamStokIrsaliyeId from OrmeIrsaliye where OrmeIrsaliye.OrmeIrsaliyeId = :AIrsaliyeId into :FHamStokIrsaliyeId;
  if (FHamStokIrsaliyeId is not null) then exception kayithatasi + ' irsaliye boyahaneye transfer edilmiş, işlem yapılamaz!';
  IK = EgemenCopyStr(ABarkodNo, 1, 2);
  if ((ABoyaSiparisDetayId is not null ) and (IK<>'G-') ) then
  begin
    select OrmeSiparisDetay.BoyaSiparisKayitId,BoyaSiparis.KumasTemin from OrmeKaliteKontrol
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeKaliteKontrol.OrmePlanlamaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
    where OrmeKaliteKontrol.OrmeKaliteKontrolId = :AKaliteKontrolId
    into  :FKTBoyaSiparisKayitId,  :FKumasTemin ;

    Select count(*) from BoyaSiparisDetay where BoyaSiparisDetay.boyasiparisdetayid = :ABoyaSiparisDetayId and
      BoyaSiparisDetay.toplamboyayacikis>=BoyaSiparisDetay.siparisbrutkilo
      into :KayitSayisi;
    if (KayitSayisi>0) then exception kayithatasi +' Siparişin rengine ait Kiloyu aştınız' ;

    if (coalesce(FKumasTemin,0)=1) then
    begin
      select count(BoyaSiparis.BoyaSiparisId ) from OrmeSiparisDetay
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
      where OrmeSiparisDetay.KTBoyaSiparisKayitId = :FKTBoyaSiparisKayitId and BoyaSiparis.boyasipariskayitid = :ABoyaSiparisKayitId
      into  :FKontrol;
      if (coalesce(FKontrol,0) =0) then exception OrmeBarkoduTeminSiparisiDegil; /*Kumaş temin siparişinin içinde olmayanlar için */
    end
  end

  if (ABoyaTalimatKayitId is not null) then
  begin
    select BoyaTalimat.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayID,
           OrmeSiparisDetay.HamId, BoyaTalimat.BoyaSiparisDetayId from BoyaTalimat
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaTalimat.BoyaSiparisDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
    left join Renk TalimatRenk on TalimatRenk.RenkId = BoyaTalimat.RenkId
    left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
    left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
    left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisBaskiId = BoyaTalimat.BoyaSiparisBaskiId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    where BoyaTalimat.BoyaTalimatKayitId = :ABoyaTalimatKayitId
    into :TalBoyaSiparisKayitId, :TalOrmeSiparisDetayID, :TalHamId, :BoyaSiparisDetayId;
  end

  select OrmeKaliteKontrol.OrmeKaliteKontrolId, OrmeKaliteKontrol.OrmeIrsaliyeId, OrmeKaliteKontrol.OrmePlanlamaId, OrmePlanlama.BoyaSiparisKayitId, OrmePlanlama.OrmeSiparisDetayId,
         OrmeSiparisDetay.HamId, OrmeKaliteKontrol.RafId, OrmeKaliteKontrol.Miktar, OrmeKaliteKontrol.StokHareketId,
         OrmeKaliteKontrol.OrmeRafDetayId, OrmeSiparisDetay.DovizId, OrmeKaliteKontrol.SatisStokHareketId, OrmeKaliteKontrol.Sakat,
         OrmeSiparisDetay.Iptal, StokHareket.Lot, OrmeKaliteKontrol.UretimeCikisStokHareketId, OrmeKaliteKontrol.OrmeUCikisIrsaliyeId,
         OrmeSiparisDetay.ResimDosyaAdi1, OrmeSiparisDetay.ResimDosyaAdi2 from OrmeKaliteKontrol
  left join StokHareket on StokHareket.StokHareketID = OrmeKaliteKontrol.StokHareketId
  left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeKaliteKontrol.OrmePlanlamaId
  left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
  left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
  where OrmeKaliteKontrol.OrmeKaliteKontrolId = :AKaliteKontrolId
  into :KKId, :FIrsaliyeId, :OrmePlanlamaId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :HamId, :RafId,
       :Miktar, :StokHareketId, :OrmeRafDetayId, :SiparisDovizId, :SatisStokHareketId, :Sakat,
       :FIptal, :FLot, :UretimeCikisStokHareketId, :FUIrsaliyeId, :FResimDosyaAdi1, :FResimDosyaAdi2;

  if ((FEgemenFirmaId = 4) and (FKumasTemin=0)) then
  begin
    if ((FResimDosyaAdi1 is null) or (FResimDosyaAdi2 is null)) then exception kayithatasi + 'Örme sipariş detayında eksik dosyalar var.';
  end

  if (coalesce(FKumasTemin,0)=1) then
  begin
    BoyaSiparisDetayId = ABoyaSiparisDetayId;
    OrmeSiparisDetayId = AOrmeSiparisDetayId;
    BoyaSiparisKayitId = ABoyaSiparisKayitId;
    select Hamid, ResimDosyaAdi1, ResimDosyaAdi2 from OrmeSiparisDetay where OrmeSiparisDetayid = :aormesiparisdetayid into :TalHamid, :FResimDosyaAdi1, :FResimDosyaAdi2;
    if (TalHamid<>Hamid) then  exception kayithatasi + FKumasTemin || ' seçtiğiniz kumaş ile okuttuğunuz kumaş farklı';
    if ((FEgemenFirmaId = 4) and (coalesce(FKumasTemin,0)=0)) then
    begin
      if ((FResimDosyaAdi1 is null) or (FResimDosyaAdi2 is null)) then exception kayithatasi + 'Örme sipariş detayında eksik dosyalar var.';
    end
  end
  else
  begin
    if (IK<>'G-')  then
    begin
      select BoyaSiparisDetay.OrmeSiparisDetayid from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId into :GelenOrmeSipDetayid;
      if (GelenOrmeSipDetayid<>OrmeSiparisDetayId) then exception kayithatasi + 'Okuttuğunuz sipariş ile mevcut sipariş farklı :G' || GelenOrmeSipDetayid || 'StokOSD' || OrmeSiparisDetayId;
      BoyaSiparisDetayId = :ABoyaSiparisDetayId;
    end
  end



  if (ABoyaTalimatKayitId is not null) then
  begin
    if (BoyaSiparisKayitId <> TalBoyaSiparisKayitId or OrmeSiparisDetayId <> TalOrmeSiparisDetayId or HamId <> TalHamId) then exception CikislaKaliteBilgileriUyusmuyor;
  end

  if (IK = 'G-') then
  begin
    select BoyaSiparis.BoyaSiparisKayitId,BoyaSiparis.KumasTemin from GirisKaliteKontrol
    left join Stokhareket on StokHareket.stokhareketid = GirisKaliteKontrol.stokhareketid
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = Stokhareket.ormeplanlamaid
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
    where GirisKaliteKontrol.giriskalitekontrolid = :AKaliteKontrolId
    into  :FKTBoyaSiparisKayitId,  :FKumasTemin;

    Select count(*) from BoyaSiparisDetay where BoyaSiparisDetay.boyasiparisdetayid = :ABoyaSiparisDetayId and
      BoyaSiparisDetay.toplamboyayacikis>=BoyaSiparisDetay.siparisbrutkilo
      into :KayitSayisi;

    if (KayitSayisi>0) then exception kayithatasi +' Siparişin rengine ait Kiloyu aştınız' ;

    select shtAlis, shtFasonUretim, shtUretim from OrmeStokHareketTipiGetir into :shtAlis, :shtFasonUretim, :shtUretim;
    select first 1 StokHareket.HamID, StokHareket.BoyaSiparisKayitID, StokHareket.BoyaSiparisDetayID, StokHareket.BoyaSiparisBaskiId, StokHareket.OrmeSiparisDetayID, StokHareket.Lot, StokHareket.RenkID,
           StokHareket.UreticiId, StokHareket.MelanjKodu, StokHareket.DepoId, StokHareket.Sakat, StokHareket.DokumaHareketi, StokHareket.MiktarBirim, StokHAreket.DovizId from StokHareketDetay
    left join StokHareket on StokHareket.StokHareketID = StokHareketDetay.StokHareketID
    where StokHareketDetay.Barkod = :ABarkodNo and (StokHareket.HareketTipi = :shtAlis or StokHareket.HareketTipi = :shtFasonUretim or StokHareket.HareketTipi = :shtUretim)
    into :FHamID, :FBoyaSiparisKayitID, :FBoyaSiparisDetayID, :FBoyaSiparisBaskiId, :FOrmeSiparisDetayID, :FLot, :FRenkID, :FUreticiId, :FMelanjKodu, :FDepoId, :Sakat,
    :DokumaHareketi, :MiktarBirim, :KaynakDovizId;

    if (coalesce(FKumasTemin,0)=1) then
    begin
      FBoyaSiparisDetayID = ABoyaSiparisDetayId;
      FOrmeSiparisDetayID = AOrmeSiparisDetayId;
      FBoyaSiparisKayitID = ABoyaSiparisKayitId;
      select Hamid, ResimDosyaAdi1, ResimDosyaAdi2 from OrmeSiparisDetay where OrmeSiparisDetayid = :aormesiparisdetayid into :TalHamid, :FResimDosyaAdi1, :FResimDosyaAdi2;
      if (TalHamid<>Fhamid) then  exception kayithatasi + 'seçtiğiniz kumaş ile okuttuğunuz kumaş farklı';
    end
    else
    begin
      select BoyaSiparisDetay.OrmeSiparisDetayid from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayId = :ABoyaSiparisDetayId into :GelenOrmeSipDetayid;
      if (GelenOrmeSipDetayid<>FOrmeSiparisDetayID) then exception kayithatasi + 'Okuttuğunuz sipariş ile mevcut sipariş farklı :G' || GelenOrmeSipDetayid || 'VarOLanOSD' || FOrmeSiparisDetayID;
      BoyaSiparisDetayId = :ABoyaSiparisDetayId;
    end

    if (FEgemenFirmaId = 4) then
    begin
      if ((FResimDosyaAdi1 is null ) or (FResimDosyaAdi2 is null )) then exception kayithatasi + 'Örme sipariş detayında eksik dosyalar var.';
    end

    PaketId = EgemenStrToInt(EgemenCopyStr(ABarkodNo, 3, StrLength(ABarkodNo) - 2));
    select StokHareketDetay.OrmeRafId from StokHareketDetay where StokHareketDetay.GirisKaliteKontrolId = :PaketId into :OrmeRafId;

    select count(*) from stokhareketdetay where stokhareketdetay.Barkod = :ABarkodNo and StokHareketDetay.HareketTipi = :AHareketTipi into :cBarkod;
    if (paketid is null) then exception KaliteKontrolIdKaydiBulunamadi;
    if (AIrsaliyeId is not null and cBarkod > 0 ) then exception BuUretiminCikisiZatenYapilmis;


    select StokHareketId from StokHareket
    where (StokHareket.OrmeIrsaliyeId = :AIrsaliyeId or :AIrsaliyeId is null) and
           StokHareket.OrmeRafId = :OrmeRafId and (StokHareket.BoyaTalimatKayitID = :ABoyaTalimatKayitId or :ABoyaTalimatKayitId is null) and
           (StrKarsilastir(StokHareket.Lot, :FLot) = 1) and
           IntKarsilastir(Stokhareket.BoyaSiparisKayitId, :FBoyaSiparisKayitId) = 1 and
           IntKarsilastir(Stokhareket.OrmeSiparisDetayId, :FOrmeSiparisDetayId) = 1 and
           IntKarsilastir(Stokhareket.BoyaSiparisDetayId, :FBoyaSiparisDetayId) = 1
    into :KayitStokHareketId;

    if (KayitStokHareketId is not null) then
    begin
      if (MiktarBirim = 0) then
        update StokHareket set KapAdedi = KapAdedi + 1, Miktar = Miktar + :AKilo, BrutMiktar = BrutMiktar + :ABrutKilo, Metre = Metre + :AMetre, NeredenUpdate = 3 where StokHareket.StokHareketId = :KayitStokHareketId;
      else
        update StokHareket set KapAdedi = KapAdedi + 1, Metre = Metre + :AMetre, Kilo = Kilo + :AKilo, NeredenUpdate = 3 where StokHareket.StokHareketId = :KayitStokHareketId;
    end
    else
    begin
      KayitStokHareketId = gen_id(StokHareket_Id_Gen, 1);
      if (MiktarBirim = 0) then
        insert into Stokhareket(StokHareketId, DovizId, Tarih, FiiliSevkTarihi,  OrmeIrsaliyeId, HareketTipi, Iade, HamIade, OrmeSiparisDetayId, BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisBaskiId, BoyaTalimatKayitID, HamId, OrmeRafId,
                              KapAdedi, Miktar, Kilo, BrutMiktar, Metre, RenkId, Lot, UreticiId, MelanjKodu, DepoId, InsertKullaniciId, KullaniciId, DokumaHareketi) values
                             (:KayitStokHareketId, :DovizId, current_timestamp(0), current_timestamp(0),  :AIrsaliyeId, :AHareketTipi, 0, 0, :FOrmeSiparisDetayId, :FBoyaSiparisKayitId, :FBoyaSiparisDetayId, :FBoyaSiparisBaskiId, :aboyatalimatkayitid, :FHamId, :OrmeRafId,
                              1, :AKilo, :AKilo, :ABrutKilo, :AMetre, :FRenkId, :FLot, :FUreticiId, :FMelanjKodu, :FDepoId, :AKullaniciId, :AKullaniciId, :DokumaHareketi);
      else
      insert into Stokhareket(StokHareketId, DovizId, Tarih, FiiliSevkTarihi,  OrmeIrsaliyeId, HareketTipi, Iade, HamIade, OrmeSiparisDetayId, BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisBaskiId, BoyaTalimatKayitID, HamId, OrmeRafId,
                             KapAdedi, Metre, Kilo, RenkId, Lot, UreticiId, MelanjKodu, DepoId, InsertKullaniciId, KullaniciId, DokumaHareketi) values
                             (:KayitStokHareketId, :DovizId, current_timestamp(0), current_timestamp(0),  :AIrsaliyeId, :AHareketTipi, 0, 0, :FOrmeSiparisDetayId, :FBoyaSiparisKayitId, :FBoyaSiparisDetayId, :FBoyaSiparisBaskiId, :aboyatalimatkayitid, :FHamId, :OrmeRafId,
                            1, :AMetre, :AKilo, :FRenkId, :FLot, :FUreticiId, :FMelanjKodu, :FDepoId, :AKullaniciId, :AKullaniciId, :DokumaHareketi);
    end

    insert into StokHareketDetay(StokHareketID, HareketTipi, Barkod, Kilo, BrutKilo, Metre, InsertKullaniciId, KullaniciId) values
                                (:KayitStokHareketId, :AHareketTipi, :ABarkodNo, :AKilo, :ABrutKilo, :AMetre, :AKullaniciId, :AKullaniciId);
  end
  else
  begin
    FKumasTemin = 0;
    if (KKId is null) then exception KaliteKontrolIdKaydiBulunamadi;

    if (AIrsaliyeId is not null and FIrsaliyeId is not null) then exception BuUretiminCikisiZatenYapilmis;
    if (AIrsaliyeId is not null and FUIrsaliyeId is not null) then exception BuUretiminUCikisiZatenYapilmis;

    if (OrmeRafDetayId is not null) then
    begin
      if (Sakat = 0) then select UrunRafId from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, null, null, :StokHareketId, null) where OrmeRafDetayId = :OrmeRafDetayId into :OrmeRafId;
      else select UrunRafId from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, null, null, null, null) where Sakat = 1 and RafID = :RafId into :OrmeRafId;
    end
    else
    begin
      if (Sakat = 0) then select max(UrunRafId) from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, null, null, :StokHareketId, null) into :OrmeRafId;
      else select max(UrunRafId) from OrmeHamRafListesiGetir(:HamId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, null, null, null, null) where Sakat = 1 into :OrmeRafId;
    end

    if (AIrsaliyeId is not null)  then
    begin
      select StokHareketId from StokHareket where StokHareket.OrmeIrsaliyeId = :AIrsaliyeId and
            StokHareket.OrmeRafId = :OrmeRafId and StokHareket.OrmeFisId is null and
            (StrKarsilastir(StokHareket.Lot, :FLot) = 1) and
            IntKarsilastir(Stokhareket.BoyaSiparisKayitId, :BoyaSiparisKayitId) = 1 and
            IntKarsilastir(Stokhareket.OrmeSiparisDetayId, :OrmeSiparisDetayId) = 1 and
            IntKarsilastir(Stokhareket.BoyaSiparisDetayId, :BoyaSiparisDetayId) = 1
      into :KayitStokHareketId;

      select SatisFiyati from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayID = :OrmeSiparisDetayId into :SatisFiyati;

      if (KayitStokHareketId is not null) then
      begin
        update StokHareket set KapAdedi = KapAdedi + 1, Miktar = Miktar + :Miktar, BrutMiktar = BrutMiktar + :Miktar, NeredenUpdate = 1 where StokHareket.StokHareketId = :KayitStokHareketId;
      end
      else
      begin
        select DovizId, IrsaliyeTarihi from OrmeIrsaliye where OrmeIrsaliyeId = :AIrsaliyeId into :IrsaliyeDovizId, :IrsaliyeTarihi;

        if (SiparisDovizId <> IrsaliyeDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:SiparisDovizId, :IrsaliyeDovizId, :IrsaliyeTarihi) into :DovizKuru;
        else DovizKuru = 1;

        if (DovizKuru is null or DovizKuru = 0) then exception DovizKuruGirisiYapilmamis;
        SatisFiyati = SatisFiyati * DovizKuru;

        KayitStokHareketId = gen_id(StokHareket_Id_Gen, 1);

        insert into StokHareket(StokHareketId, HareketTipi, OrmeIrsaliyeId, NeredenUpdate, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, OrmeRafId, KapAdedi, Miktar, BrutMiktar, BirimFiyat, Lot, BoyaTalimatKayitId, BoyaSiparisDetayId)
                      values(:KayitStokHareketId, 6, :AIrsaliyeId, 1, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :HamId, :OrmeRafId, 1, :Miktar, :Miktar, :SatisFiyati, :FLot, :ABoyaTalimatKayitId, :BoyaSiparisDetayId);

      end
      update OrmeKaliteKontrol set OrmeUCikisIrsaliyeId = :AIrsaliyeId, UretimeCikisStokHareketId = :KayitStokHareketId, LinkIslemi = 1 where OrmeKaliteKontrol.OrmeKaliteKontrolId = :AKaliteKontrolId;
    end
    else
    begin
      if (UretimeCikisStokHareketId is null) then exception BuUretimUCikisiZatenYapilmamis;

      select count(*) from OrmeKaliteKontrol where OrmeKaliteKontrol.SatisStokHareketId = :SatisStokHareketId into :KayitSayisi;
      if (KayitSayisi = 1) then delete from StokHareket where StokHareket.StokHareketId = :SatisStokHareketId;
      else
      begin
        update StokHareket set KapAdedi = KapAdedi - 1, Miktar = Miktar - :Miktar, BrutMiktar = BrutMiktar - :Miktar, NeredenUpdate = 1 where StokHareket.StokHareketId = :UretimeCikisStokHareketId;
        update OrmeKaliteKontrol set OrmeUCikisIrsaliyeId = null, UretimeCikisStokHareketId = null, LinkIslemi = 1 where OrmeKaliteKontrol.OrmeKaliteKontrolId = :AKaliteKontrolId;
      end
    end
  end
