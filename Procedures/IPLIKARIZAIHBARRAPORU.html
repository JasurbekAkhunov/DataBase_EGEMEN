CREATE PROCEDURE IPLIKARIZAIHBARRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIPLIKMAKINAID VARCHAR(10),
    AIPLIKARIZATIPIID INTEGER,
    ABITTI INTEGER,
    APARCABEKLIYOR INTEGER)
RETURNS (
    IPLIKARIZAIHBARID INTEGER,
    IPLIKMAKINAID VARCHAR(10),
    BASLATANPERSONELID INTEGER,
    BASLATANPERSONELADI VARCHAR(40),
    BASLAMATARIHI TIMESTAMP,
    BITIRENPERSONELID INTEGER,
    BITIRENPERSONELADI VARCHAR(40),
    BITISTARIHI TIMESTAMP,
    BITTI INTEGER,
    BITTISTR VARCHAR(5),
    IPLIKARIZATIPIID INTEGER,
    IPLIKARIZATIPIADI VARCHAR(20),
    ARIZAYERIID INTEGER,
    ARIZAYERIADI VARCHAR(20),
    PLANLIBAKIMNO VARCHAR(20),
    NOTU VARCHAR(1023),
    SERVISNOTU VARCHAR(1023),
    BAKIMYAPANPERSONELADI VARCHAR(550),
    PARCABEKLIYOR INTEGER,
    PARCABEKLIYORSTR VARCHAR(4),
    GECENSURE DOUBLE PRECISION,
    GECENSURESTR VARCHAR(30),
    BAKIMCIILKGELISTARIHI TIMESTAMP,
    BAKIMCIGECENSURE DOUBLE PRECISION,
    BAKIMCIGECENSURESTR VARCHAR(30))
AS
declare variable FNotu Blob;
declare variable FServisNotu Blob;
declare variable BlobSize integer;
declare variable FPersonelAdi varchar(40);
begin
  For Select IplikArizaIhbar.IplikArizaIhbarId, IplikArizaIhbar.IplikMakinaId, IplikArizaIhbar.BaslatanPersonelId, BasP.Adi, MD1.Tarih,
             IplikArizaIhbar.BitirenPersonelId, BitP.Adi, MD2.Tarih, IplikArizaIhbar.IplikArizaTipiId, IplikArizaTipi.Adi, IplikArizaIhbar.ArizaYeriId,
             IplikArizaIhbar.PlanliBakimNo, IplikArizaIhbar.Notu, IplikArizaIhbar.ServisNotu, IplikArizaIhbar.ParcaBekliyor from IplikArizaIhbar
    left join Personel BasP on BasP.PersonelId = IplikArizaIhbar.BaslatanPersonelId
    left join Personel BitP on BitP.PersonelId = IplikArizaIhbar.BitirenPersonelId
    left join IplikMakinaDurumu MD1 on MD1.IplikMakinaDurumuId = IplikArizaIhbar.BasladiIplikMakinaDurumuId
    left join IplikMakinaDurumu MD2 on MD2.IplikMakinaDurumuId = IplikArizaIhbar.BittiIplikMakinaDurumuId
    left join IplikArizaTipi on IplikArizaTipi.IplikArizaTipiId = IplikArizaIhbar.IplikArizaTipiId
    Where (IplikArizaIhbar.IplikMakinaId = :AIplikMakinaId or :AIplikMakinaId is null) and
          (IplikArizaIhbar.IplikArizaTipiId = :AIplikArizaTipiId or :AIplikArizaTipiId is null) and
          (MD1.Tarih >= :ABaslangicTarihi and MD1.Tarih <= :ABitisTarihi) and
          (:ABitti is null or (IplikArizaIhbar.BitirenPersonelId is not null and :ABitti = 1) or
            (IplikArizaIhbar.BitirenPersonelId is null and :ABitti = 0) ) and
          (IplikArizaIhbar.ParcaBekliyor = :AParcaBekliyor or :AParcaBekliyor is null)
    into :IplikArizaIhbarId, :IplikMakinaId, :BaslatanPersonelId, :BaslatanPersonelAdi, :BaslamaTarihi, :BitirenPersonelId, :BitirenPersonelAdi,
         :BitisTarihi, :IplikArizaTipiId, :IplikArizaTipiAdi, :ArizaYeriId, :PlanliBakimNo, :FNotu, :FServisNotu, :ParcaBekliyor do
  begin
    if (BitirenPersonelId is not null) then Bitti = 1;
    else Bitti = 0;

    if (Bitti = 1) then BittiStr = 'Bitti';
    else BittiStr = '';

    if (ParcaBekliyor = 1) then ParcaBekliyorStr = 'Evet';
    else ParcaBekliyorStr = '';

    if (ArizaYeriId = 0) then ArizaYeriAdi = 'Mekanik';
    else if (ArizaYeriId = 1) then ArizaYeriAdi = 'Elektrik';
    else if (ArizaYeriId is not null) then ArizaYeriAdi = 'HatalÄ±';
    else ArizaYeriAdi = '';

    BlobSize = char_length(FNotu);

    if (BlobSize > 1020) then Notu = Left(FNotu, 1020);
    else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
    else Notu = '';

    BlobSize = char_length(FServisNotu);

    if (BlobSize > 1020) then ServisNotu = Left(FServisNotu, 1020);
    else if (BlobSize > 0) then ServisNotu = Left(FServisNotu, BlobSize);
    else ServisNotu = '';

    if (BitisTarihi is not null) then GecenSure = BitisTarihi - BaslamaTarihi;
    else GecenSure = (CURRENT_DATE + CURRENT_TIME) - BaslamaTarihi;

    GecenSureStr = TarihToStr(GecenSure);

    BakimYapanPersonelAdi = '';

    For Select Personel.Adi from IplikArizaBakim
      left join Personel on Personel.PersonelId = IplikArizaBakim.PersonelId
      where IplikArizaBakim.IplikArizaIhbarId = :IplikArizaIhbarId
      group by Personel.Adi
      into :FPersonelAdi do
    begin
      if (Char_Length(BakimYapanPersonelAdi) < 500) then
      begin
        if (BakimYapanPersonelAdi <> '') then BakimYapanPersonelAdi = BakimYapanPersonelAdi || ',';
        BakimYapanPersonelAdi = BakimYapanPersonelAdi || FPersonelAdi;
      end
    end

    BakimciIlkGelisTarihi = Null;

    Select Min(IplikMakinaDurumu.Tarih) from IplikArizaBakim
      left join IplikMakinaDurumu on IplikMakinaDurumu.IplikMakinaDurumuId = IplikArizaBakim.GeldiIplikMakinaDurumuId
      where IplikArizaBakim.IplikArizaIhbarId = :IplikArizaIhbarId
      into :BakimciIlkGelisTarihi;


    if (BakimciIlkGelisTarihi is not null) then
    begin
      BakimciGecenSure = BakimciIlkGelisTarihi - BaslamaTarihi;
      BakimciGecenSureStr = TarihToStr(BakimciGecenSure);
    end
    else
    begin
      BakimciGecenSure = 0;
      BakimciGecenSureStr = '';
    end

    suspend;
  end
end
