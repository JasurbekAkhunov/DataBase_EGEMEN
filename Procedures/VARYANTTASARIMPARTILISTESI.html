CREATE PROCEDURE VARYANTTASARIMPARTILISTESI(
    ABASLANGICACILISTARIHI TIMESTAMP,
    ABITISACILISTARIHI TIMESTAMP,
    ADESENID VARCHAR(20))
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIKAYITID INTEGER,
    PARTITIPIID INTEGER,
    PARTITIPIIDSTR VARCHAR(10),
    IMALATTIPIID INTEGER,
    IMALATTIPIIDSTR VARCHAR(5),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FASON INTEGER,
    FASONSATISSTR VARCHAR(5),
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ISLEMTIPIID INTEGER,
    ISLEMGRUBUID INTEGER,
    ACILIYETDERECESIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUADI VARCHAR(20),
    ACILIYETDERECESIADI VARCHAR(20),
    ASAMAVAR VARCHAR(3),
    PLANLANANIMALATTARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    DESENID VARCHAR(20),
    DESENADI VARCHAR(80),
    VARYANTADI VARCHAR(30),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    PARTIKAPADEDI INTEGER,
    PARTIKILO NUMERIC(12,3),
    PARTIMETRE NUMERIC(12,3),
    BRUTSEVKKILO NUMERIC(12,3),
    BRUTSEVKMETRE NUMERIC(12,3),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    MODELNO VARCHAR(20),
    ACILISTARIHI TIMESTAMP)
AS
declare variable itDesenTasarim       integer;
begin
  Select itDesenTasarim from IslemTipiGetir into :itDesenTasarim;

  for select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.ImalatTipiId, Parti.PartiTipiId, Parti.PartigrupId, PartiGrup.Adi, Parti.FirmaId, Firma.Adi,Parti.IslemTipiId, IslemTipi.Adi, Parti.IslemGrubuId, IslemGrubu.Adi, Parti.AciliyetDerecesiId,
        AciliyetDerecesi.Adi, Parti.PlanlananImalatTarihi, Parti.TerminTarihi, Parti.ReceteId, Recete.Adi, Parti.Modelno, Parti.Acilistarihi,
         BoyaSiparis.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.Fason,
         Desen.DesenId, Desen.Adi, DesenVaryant.VaryantAdi from Parti
  left join DesenVaryant on DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
  left join Desen on Desen.DesenId = DesenVaryant.DesenId
  left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
  left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join AciliyetDerecesi on AciliyetDerecesi.AciliyetDerecesiId = Parti.AciliyetDerecesiId
  left join Firma on Firma.FirmaId = Parti.FirmaId
  left join Recete on Recete.ReceteId = Parti.ReceteId
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
  where Parti.IslemTipiID = :itDesenTasarim and
       (Parti.AcilisTarihi >= :ABaslangicAcilisTarihi and Parti.AcilisTarihi <= :ABitisAcilisTarihi ) and
       (Desen.DesenID = :ADesenId or :ADesenId is null)
    into :PartiKayitId, :PartiYil, :PartiId, :ImalatTipiId, :PartiTipiId, :PartigrupId, :PartiGrupAdi, :FirmaId, :FirmaAdi, :IslemTipiId, :IslemTipiAdi, :IslemGrubuId, :IslemGrubuAdi, :AciliyetDerecesiId,:AciliyetDerecesiAdi,
         :PlanlananImalatTarihi, :TerminTarihi, :ReceteId, :ReceteAdi, :Modelno, :Acilistarihi, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :Fason,
         :DesenId, :DesenAdi, :VaryantAdi do
  begin
    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    else BoyaSiparisIdStr = '';


    if (Fason = 0) then FasonSatisStr = 'Satış';
    else FasonSatisStr = 'Fason';

    if (PartiTipiId = 0) then PartiTipiIdStr = 'İplik';
    else PartiTipiIdStr = 'Kumaş';

    if (ImalatTipiId = 0) then ImalatTipiIdStr = 'Boya';
    else ImalatTipiIdStr = 'Baskı';

    select sum(kilo), sum(Metre), Sum(KapAdedi), sum(BrutSevkKilo), sum(BrutSevkMetre) from Partistok where Partistok.PartiKayitId = :PartiKayitId
      into :PartiKilo, :PartiMetre, :PartiKapAdedi, :BrutSevkKilo, :BrutSevkMetre;

    if (PartiKilo > 0) then KalanKilo = PartiKilo - BrutSevkKilo;
    else KalanKilo = 0;

    if (PartiMetre > 0) then KalanMetre = PartiMetre - BrutSevkMetre;
    else KalanMetre = 0;

    suspend;
  end
end
