CREATE PROCEDURE MAKINACALISMABILGISIKAYDET(
    AMAKINAID VARCHAR(10),
    ADURUMU INTEGER)
AS
declare variable FNow TimeStamp;
declare variable F    TimeStamp;
declare variable VardiyaTarihi TimeStamp;
declare variable VardiyaBaslangicSaati Double Precision;
declare variable SuankiSaat Double Precision;
declare variable DKTarihi          TimeStamp;
declare variable DKVardiyaTarihi   TimeStamp;
declare variable DKDurumu          integer;
declare variable Sure              Double Precision;
declare variable KayitSayisi       integer;
declare variable AsilMakinaID      MakinaKoduSize;
begin
  Select AsilMakinaID from Makina where Makina.MakinaId = :AMakinaId into :AsilMakinaID;
  if (AsilMakinaID is Not Null) then AMakinaID = :AsilMakinaID;

  FNow = current_TimeStamp(0);
  SuankiSaat = TimeStampToDouble(TimeStampToTime(FNow));


  Select  DKTarihi, DKVardiyaTarihi, DKDurumu from  MakinaCalismaSuresi where  MakinaCalismaSuresi.MakinaId = :AMakinaId and  MakinaCalismaSuresi.VardiyaTarihi= (Select  MAX(VardiyaTarihi) from  MakinaCalismaSuresi where  MakinaCalismaSuresi.MakinaId =:AMakinaId)  into :DKTarihi, :DKVardiyaTarihi, :DKDurumu;
  if (DKDurumu is null and DKVardiyaTarihi is null  and DKTarihi is null ) then
  begin
     select DKTarihi, DKVardiyaTarihi, DKDurumu from Makina where MakinaId = :AMakinaID   into :DKTarihi, :DKVardiyaTarihi, :DKDurumu;
  end

 if (DKDurumu is null) then DKDurumu = -1;

  if (DkDurumu = ADurumu) then exit;

  Select TimeStampToDouble(TimeStampToTime(VardiyaBaslangicSaati)) from sistem into :VardiyaBaslangicSaati;

  if (SuankiSaat < VardiyaBaslangicSaati) then VardiyaTarihi = cast(FNow as Date) - 1;
  else VardiyaTarihi = cast(FNow as Date);

  if (DKTarihi is null) then
  begin
    DKTarihi = FNow;
    DkVardiyaTarihi = VardiyaTarihi;
  end

  if (FNow < DKTarihi) then exit;

  if (DkVardiyaTarihi is null or (DKDurumu=-1)  ) then
  begin
    Insert into MakinaCalismaSuresi(MakinaId, VardiyaTarihi, CalismaSuresi,DKTarihi,DKVardiyaTarihi,DKDurumu) values(:AMakinaId, :VardiyaTarihi, 0,:Fnow,:VardiyaTarihi,:ADurumu);


    exit;
  end

  if (ADurumu = 0 and DKDurumu = 1) then
  if (DkVardiyaTarihi <= VardiyaTarihi) then
  begin
    While (DkVardiyaTarihi <= VardiyaTarihi) do
    begin
      F = (DKVardiyaTarihi + 1) + VardiyaBaslangicSaati;
      if (F > FNow) then F = FNow;

      Sure = F - DKTarihi;

      Select count(*) from MakinaCalismaSuresi where MakinaCalismaSuresi.MakinaId = :AMakinaId and MakinaCalismaSuresi.VardiyaTarihi = :DkVardiyaTarihi into :KayitSayisi;

      if (KayitSayisi = 0) then
        Insert into MakinaCalismaSuresi(MakinaId, VardiyaTarihi, CalismaSuresi) values(:AMakinaId, :DKVardiyaTarihi, :Sure);
      else
       Update MakinaCalismaSuresi set CalismaSuresi = CalismaSuresi + :Sure where MakinaCalismaSuresi.MakinaId = :AMakinaId and MakinaCalismaSuresi.VardiyaTarihi = :DkVardiyaTarihi;

      DKTarihi = (DKVardiyaTarihi + 1) + VardiyaBaslangicSaati;
      if (DKTarihi > FNow) then DKTarihi = FNow;
      DkVardiyaTarihi = DkVardiyaTarihi + 1;
    end
  end
  else
  begin
    Update MakinaCalismaSuresi set CalismaSuresi = CalismaSuresi + (:FNow - :DKTarihi) where MakinaCalismaSuresi.MakinaId = :AMakinaId and MakinaCalismaSuresi.VardiyaTarihi = :VardiyaTarihi;
  end

  Update MakinaCalismaSuresi Set DKTarihi =:Fnow , DKVardiyaTarihi =:VardiyaTarihi, DKDurumu =:ADurumu where MakinaCalismaSuresi.MakinaId =:AMakinaId and  MakinaCalismaSuresi.VardiyaTarihi=(Select MAX(VardiyaTarihi) from  MakinaCalismaSuresi where  MakinaCalismaSuresi.MakinaId = :AMakinaId);
end
