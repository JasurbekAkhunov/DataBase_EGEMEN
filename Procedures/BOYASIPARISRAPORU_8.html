CREATE PROCEDURE BOYASIPARISRAPORU_8(
    AIMALATTIPIID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AFIRMAID VARCHAR(20),
    AFIRMAYETKILIID INTEGER,
    AISLEMTIPIID INTEGER,
    AISLEMGRUBUID INTEGER,
    ASATISTIPI INTEGER,
    ABOYASIPARISTIPIID INTEGER,
    ALABRECETEID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    AMUSTERIKALITENO VARCHAR(20),
    AONAYDURUMU INTEGER,
    APARTILENEN INTEGER,
    AKAPALIDURUMU INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ASIPARISIALANKULLANICIID INTEGER,
    AYURTDISI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    AHAMID INTEGER,
    AKULLANICIID INTEGER,
    AKALITEID VARCHAR(20),
    ATALIMATKODU VARCHAR(20),
    AMAKINAID VARCHAR(10),
    ATEKRARURETIM INTEGER,
    ADESENID VARCHAR(20),
    ATEMSILCIID INTEGER,
    AMARKALI INTEGER,
    AKOMISYONLU INTEGER,
    ALAMINASYON INTEGER,
    ARECETETURUID INTEGER,
    ABATCHNO VARCHAR(30),
    ASERTIFIKANO VARCHAR(30),
    ABASKITIPIID INTEGER,
    AHAMGRUP1ID VARCHAR(4),
    AHAMGRUP2ID VARCHAR(4),
    AHAMGRUP3ID VARCHAR(4),
    AHAMGRUP4ID VARCHAR(4),
    ASIPARISRECETEDURUMU INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISSARJID INTEGER,
    SARJNO INTEGER,
    KAZANKAPASITE INTEGER,
    BOBINSAYISI INTEGER,
    GRAMAJ NUMERIC(12,3),
    MIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    FIREORANI DOUBLE PRECISION,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(255),
    MAKINAID VARCHAR(10),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable LastBoyaSiparisDetayId integer;
declare variable FPartiIdStr varchar(30);
begin
  For Select DISTINCT BoyaSiparisRaporu_2.BoyaSiparisDetayId from BoyaSiparisRaporu_2(:AImalatTipiId,:ABaslangicTarihi,:ABitisTarihi,
    :ABoyaSiparisYil, :ABoyaSiparisId, :AFirmaId, :AFirmaYetkiliId, :AIslemTipiId, :AIslemGrubuId, :ASatisTipi,
    :ABoyaSiparisTipiId, :ALabReceteId, :AMusteriSiparisNo, :AReferansSiparisNo, :AMusteriKaliteNo, :AOnayDurumu, :APartilenen,
    :AKapaliDurumu, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASiparisiAlanKullaniciId,
    :AYurtDisi, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :AHamId, :AKullaniciId, :AKaliteId, :ATalimatKodu, :AMakinaId, :ATekrarUretim, :ADesenID, :ATemsilciId, :AMarkali, :AKomisyonlu, :ALaminasyon, :AReceteTuruId, :ABatchNo, 1, :ASertifikaNo, :ABaskiTipiId, :AHamGrup1Id, :AHamGrup2Id, :AHamGrup3Id, :AHamGrup4Id, :ASiparisReceteDurumu)
    into :BoyaSiparisDetayId do
  begin

    SarjNo = 0;
    LastBoyaSiparisDetayId = -1;
    for select Distinct BoyaSiparisSarj.BoyaSiparisDetayId, BoyaSiparisSarj.BoyaSiparisSarjId, BoyaSiparisSarj.KazanKapasite, BoyaSiparisSarj.BobinSayisi,
               BoyaSiparisSarj.Gramaj, BoyaSiparisSarj.FireOrani, BoyaSiparisSarj.SarjKilo, BoyaSiparisSarj.BrutSarjKilo,
               BoyaSiparisSarj.MakinaId,
               IK.Adi, BoyaSiparisSarj.InsertTarihi, Sk.Adi, BoyaSiparisSarj.DegisimTarihi from BoyaSiparisSarj
    Left Join Kullanici IK on IK.KullaniciId = BoyaSiparisSarj.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = BoyaSiparisSarj.KullaniciId
    where BoyaSiparisSarj.BoyaSiparisDetayId = :BoyaSiparisDetayId and
      (BoyaSiparisSarj.MakinaId = :AMakinaId or :AMakinaId is Null)
    order by BoyaSiparisSarj.BoyaSiparisDetayId, BoyaSiparisSarj.BoyaSiparisSarjId
    into :BoyaSiparisDetayId, :BoyaSiparisSarjId, :KazanKapasite, :BobinSayisi, :Gramaj, :FireOrani, :Miktar, :BrutMiktar,
         :MakinaId,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
    begin
      if (LastBoyaSiparisDetayId <> BoyaSiparisDetayId) then
      begin
        LastBoyaSiparisDetayId = BoyaSiparisDetayId;
        SarjNo = 0;
      end
      SarjNo = SarjNo + 1;
      PartiIdStr = '';

      For Select Distinct Parti.PartiYil, Parti.PartiId from PartiStok
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
        where PartiStok.BoyaSiparisSarjId = :BoyaSiparisSarjId and PartiStok.PartiStokIptalKayitId is null
        into :PartiYil, :PartiId do
      begin
        FPartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

        if (Char_Length(PartiIdStr) + Char_Length(FPartiIdStr) < 253) then
        begin
          if (PartiIdStr <> '' ) then PartiIdStr = PartiIdStr || cast(', ' as varchar(2) character set UTF8);
          PartiIdStr = PartiIdStr || FPartiIdStr;
        end
      end
      PartiYil = Null;
      PartiId = Null;
      suspend;
    end

  end
end
