CREATE PROCEDURE PSDOKUMAISEMRIBILGISIGETIR(
    APARTISINIFID INTEGER,
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    HAMSTOKMASTERID INTEGER,
    COZGUINDIGOISEMRIIDSTR VARCHAR(50),
    TOPNO INTEGER,
    INDIGORECETEID VARCHAR(20))
AS
Declare variable FReceteIsEmriYil integer;
Declare variable FReceteIsEmriId  integer;
Declare variable FPartiYil integer;
Declare variable FPartiId  integer;
declare variable FPartiIdStr varchar(30);
declare variable DokumaTopTakibiDevrede integer;
begin
  select DokumaTopTakibiDevrede from sistem where sistem.sistemid = 1 into :DokumaTopTakibiDevrede;

  if (DokumaTopTakibiDevrede = 1) then
  begin
    if (APartiSinifId is Not null ) then
     begin
       for select distinct HamStokMaster.HamStokMasterId,
     coalesce(IndigoIsEmri.IsEmriId, CozguIsEmri.IsEmriId) || lpad(OrmeRaf.LeventSirano,2,'0') || lpad(DokumaKaliteKontrol.TopNo, 2, '0'), DokumaKaliteKontrol.TopNo, IndigoIsEmri.ReceteId from PartiStokDetay
              left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
              left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
              left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
              left join DokumaStokPlanlama on DokumaStokPlanlama.Dokumaplanlamaid = HAmStokMAster.Dokumaplanlamaid
              left join OrmeRaf on OrmeRaf.OrmeRafId = DokumaStokPlanlama.Ormerafid
              left join IndigoIsEmri on IndigoIsEmri.IndigoIsEmriKayitId = OrmeRaf.IndigoIsEmriKayitId
              left join CozguIsEmri on CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId
              left join StokHareket on StokHareket.StokHareketId = HAmStokMAster.StokHareketId and  :DokumaTopTakibiDevrede = 1
              left join OrmeRafDetay on OrmeRafDetay.Ormerafid = StokHareket.Ormerafid
              left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketID = OrmeRafDetay.StokHareketId
           where PartiStokDetay.PartiSinifId = :APartiSinifId
       into :HamStokMasterID, :CozguIndigoIsEmriIdStr, :TopNo, :IndigoReceteId  do
       begin
         suspend;
       end
    end
    else
    begin
       For  select distinct HamStokMaster.HamStokMasterId,
     coalesce(IndigoIsEmri.IsEmriId, CozguIsEmri.IsEmriId) || lpad(OrmeRaf.LeventSirano,2,'0') || lpad(DokumaKaliteKontrol.TopNo, 2, '0'), DokumaKaliteKontrol.TopNo, IndigoIsEmri.ReceteId from Hamstokmaster
                left join DokumaStokPlanlama on DokumaStokPlanlama.Dokumaplanlamaid = HAmStokMAster.Dokumaplanlamaid
                left join OrmeRaf on OrmeRaf.OrmeRafId = DokumaStokPlanlama.Ormerafid
                left join IndigoIsEmri on IndigoIsEmri.IndigoIsEmriKayitId = OrmeRaf.IndigoIsEmriKayitId
                left join CozguIsEmri on CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId
                left join StokHareket on StokHareket.StokHareketId = HAmStokMAster.StokHareketId
                left join OrmeRafDetay on OrmeRafDetay.Ormerafid = StokHareket.Ormerafid
                left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketID = OrmeRafDetay.StokHareketId
            where HamStokMaster.hamstokmasterid = :AHamStokMasterId
      into :HamStokMasterID, :CozguIndigoIsEmriIdStr, :TopNo, :IndigoReceteId  do
       begin
          suspend;
       end
    end
  end
  else   if (:DokumaTopTakibiDevrede=2) then
  begin
  if (AHamStokMasterId is not null) then
  begin

    for select distinct HamStokMaster.HamStokMasterId, DokumaKaliteKontrol.TopNo from HamStokMaster

          left join StokHareket on StokHareket.StokHareketId = HAmStokMAster.StokHareketId
          left join OrmeRaf on OrmeRaf.OrmeRafId = Stokhareket.Ormerafid
          left join OrmeRafDetay on OrmeRafDetay.Ormerafid = StokHareket.Ormerafid
          left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketID = OrmeRafDetay.StokHareketId
        where Hamstokmaster.hamstokmasterid = :AHamStokMasterId into :HamStokMasterID, :TopNo   do
    begin
      suspend;
    end

  end
  else
  begin
    for select distinct HamStokMaster.HamStokMasterId, DokumaKaliteKontrol.TopNo from PartiStokDetay
            left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
            left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
            left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
            left join StokHareket on StokHareket.StokHareketId = HAmStokMAster.StokHareketId and  :DokumaTopTakibiDevrede = 2
            left join OrmeRaf on OrmeRaf.OrmeRafId = Stokhareket.Ormerafid
            left join OrmeRafDetay on OrmeRafDetay.Ormerafid = StokHareket.Ormerafid
            left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketID = OrmeRafDetay.StokHareketId
        where PartiStokDetay.PartiSinifId = :APartiSinifId into :HamStokMasterID, :TopNo   do
        begin
          suspend;
        end
    end
  end
end
