CREATE PROCEDURE HAMSTOKMASTERORMEGIRISIGETIR(
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    ORMEGIRISIRSALIYENO VARCHAR(400),
    ORMEGIRISIRSALIYETARIHI VARCHAR(400),
    GIRISORMEFIRMAID VARCHAR(400),
    GIRISORMEFIRMAADI VARCHAR(400))
AS
declare variable FIrsaliyeNo IrsaliyeNoSize;
declare variable FIrsaliyeTarihi TimeStamp;
declare variable Str        varchar(30);
declare variable FFirmaID   varchar(20);
declare variable FFirmaAdi  varchar(100);
begin
  OrmeGirisIrsaliyeNo         = '';
  OrmeGirisIrsaliyeTarihi     = '';
  GirisOrmeFirmaId            = '';
  GirisOrmeFirmaAdi           = '';

  For select OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.FirmaId, Firma.Adi from HamStokMaster
    left join StokHareket CikisStokHareket on CikisStokHareket.StokHareketId = HamStokMaster.StokHareketId
    left join OrmeRaf on OrmeRaf.OrmeRafId = CikisStokHareket.OrmeRafId
    left join OrmeRafDetay on OrmeRafDetay.OrmeRafId = OrmeRaf.OrmeRafId
    left join StokHareket GirisStokHareket on GirisStokHareket.StokHareketId = OrmeRafDetay.StokHareketId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = GirisStokHareket.OrmeIrsaliyeId
    left join Firma on Firma.FirmaId = OrmeIrsaliye.FirmaId
    where HamStokMaster.HamStokMasterId = :AHamStokMasterId and HamStokMaster.StokHareketId is not null
    group by OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.FirmaId, Firma.Adi
    into :FIrsaliyeNo, :FIrsaliyeTarihi, :FFirmaId, :FFirmaAdi do
  begin

    if (FIrsaliyeNo is not null) THen Str = Format_Float('#0', FIrsaliyeNo);
    else Str = '';

    if (Str <> '') then
    if (PosBul(OrmeGirisIrsaliyeTarihi, TimeStampToDateStr(FIrsaliyeTarihi)) = 0) then
    if (Char_Length(OrmeGirisIrsaliyeTarihi) + Char_Length(TimeStampToDateStr(FIrsaliyeTarihi)) < 400) then
    if (Char_Length(OrmeGirisIrsaliyeNo) + Char_Length(Str) < 400) then
    begin
      if (OrmeGirisIrsaliyeNo <> '') then OrmeGirisIrsaliyeNo = OrmeGirisIrsaliyeNo ||  cast(', ' as varchar(2) character set UTF8);
      OrmeGirisIrsaliyeNo = OrmeGirisIrsaliyeNo || Str;

      Str = TimeStampToDateStr(FIrsaliyeTarihi);

      if (OrmeGirisIrsaliyeTarihi <> '') then OrmeGirisIrsaliyeTarihi = OrmeGirisIrsaliyeTarihi ||  cast(', ' as varchar(2) character set UTF8);
      OrmeGirisIrsaliyeTarihi = OrmeGirisIrsaliyeTarihi || Str;
    end

    if (PosBul(GirisOrmeFirmaAdi, FFirmaAdi) = 0) then
    if (Char_Length(GirisOrmeFirmaAdi) + Char_Length(FFirmaAdi) < 400) then
    if (Char_Length(GirisOrmeFirmaId) + Char_Length(FFirmaId) < 400) then
    begin

      if (GirisOrmeFirmaAdi <> '') then GirisOrmeFirmaAdi = GirisOrmeFirmaAdi ||  cast(', ' as varchar(2) character set UTF8);
      GirisOrmeFirmaAdi = GirisOrmeFirmaAdi || FFirmaAdi;

      if (GirisOrmeFirmaId <> '') then GirisOrmeFirmaId = GirisOrmeFirmaId ||  cast(', ' as varchar(2) character set UTF8);
      GirisOrmeFirmaId = GirisOrmeFirmaId || FFirmaId;
    end

  end

  suspend;
end
