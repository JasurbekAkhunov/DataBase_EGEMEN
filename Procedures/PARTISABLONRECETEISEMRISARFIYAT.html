Description

İş Emrine bağlı tüm partilerin "İş Emri ile alakalı" sarfiyatlarını listeler. Bu liste üzerinden gruplayarak toplam sarfiyatlar da hesaplanabilir. Parametre olarak İş Emrine bağlı herhangi bir PartiKayıtID verilmesi yeterlidir.

 

Definition

CREATE PROCEDURE PARTISABLONRECETEISEMRISARFIYAT(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTISABLONID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BIRIMID VARCHAR(2),
    DESENVARYANTID INTEGER,
    URETIMECIKISFIYATI DOUBLE PRECISION,
    URETIMECIKISFIYATIBUGUN DOUBLE PRECISION,
    URETIMECIKISFIYATIBIRIMID VARCHAR(2),
    BIRIMKULLANILANMIKTAR DOUBLE PRECISION,
    KULLANILANMIKTAR DOUBLE PRECISION,
    ILAVEMIKTARI BOYAMIKTARSIZE,
    DUZELTMEMIKTARI BOYAMIKTARSIZE,
    ILAVEKULLANILANMIKTAR BOYAMIKTARSIZE,
    DUZELTMEKULLANILANMIKTAR BOYAMIKTARSIZE)
AS
declare FSablonHazirlananMiktar    double precision;
declare FAcilisTarihi              timestamp;
declare FDovizKuru                 double precision;
declare FUretimeCikisFiyati        double precision;
declare FUretimeCikisFiyatiDovizId varchar(5);
declare FStokTabanDovizId          varchar(5);
begin
  select first 1 StokTabanDovizId from SISTEM where SISTEMID = 1 into :FStokTabanDovizId;
  for select PartiSablonRecete.PartiKayitID, PartiSablon.PartiSablonID, PartiSablonRecete.UrunId, Urun.Adi, PSRUB.BirimId, Parti.DesenVaryantID, PartiSablonRecete.KullanilanMiktar,
    UrunFiyat.UretimeCikisFiyati, UBUCF.BirimId, Parti.AcilisTarihi, PartiSablon.HazirlananMiktar from PartiSablonRecete
    left join PartiSablon on PartiSablon.PartiSablonID = PartiSablonRecete.PartiSablonID
    left join Parti on Parti.PartiKayitId = PartiSablonRecete.PartiKayitID
    left join Urun on Urun.UrunId = PartiSablonRecete.UrunId
    left join UrunBirim PSRUB on PSRUB.UrunBirimId = PartiSablonRecete.UrunBirimId and PSRUB.UrunId = PartiSablonRecete.UrunId
    left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
    left join UrunBirim UBUCF on UBUCF.UrunBirimId = urunFiyat.UretimeCikisFiyatiUrunBirimId
    where  PartiSablonRecete.PartiKayitID = :APartiKayitId and PartiSablon.PartiSablonIsEmriId in ( SELECT DISTINCT PartiSablonIsEmriId FROM PartiSablon WHERE PartiSablon.PartiKayitID = :APartiKayitId )
    order by  PartiSablonRecete.PartiKayitId, PartiSablon.SablonNo, Urun.UrunTipiId
    into :PartiKayitID, :PartiSablonId, :UrunId, :UrunAdi, :BirimId, :DesenVaryantID, :KullanilanMiktar,
    :FUretimeCikisFiyati, :UretimeCikisFiyatiBirimId, :FAcilisTarihi, :FSablonHazirlananMiktar do
  begin
    BirimKullanilanMiktar = iif(FSablonHazirlananMiktar = 0, 0, KullanilanMiktar / FSablonHazirlananMiktar);

    if (FUretimeCikisFiyatiDovizId = FStokTabanDovizId) then FDovizKuru = 1;
    else select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :FStokTabanDovizId, Current_Date) into :FDovizKuru;

    UretimeCikisFiyatiBugun = FUretimeCikisFiyati * IIF(FDovizKuru is null, 0, FDovizKuru);

    if (FUretimeCikisFiyatiDovizId = FStokTabanDovizId) then FDovizKuru = 1;
    else select DovizKuru from EnYakinDovizKuruGetir(:FUretimeCikisFiyatiDovizId, :FStokTabanDovizId, :FAcilisTarihi) into :FDovizKuru;

    UretimeCikisFiyati = FUretimeCikisFiyati * IIF(FDovizKuru is null, 0, FDovizKuru);

    select Sum(iif(PartiSablonIlave.IlaveTipi = 1, PartiSablonReceteIlave.Miktar, 0)),
      Sum(iif(PartiSablonIlave.IlaveTipi = 2, PartiSablonReceteIlave.Miktar, 0)),
      Sum(iif(PartiSablonIlave.IlaveTipi = 1, PartiSablonReceteIlave.KullanilanMiktar, 0)),
      Sum(iif(PartiSablonIlave.IlaveTipi = 2, PartiSablonReceteIlave.KullanilanMiktar, 0)) from PartiSablonIlave
    left join PartiSablonReceteIlave on PartiSablonReceteIlave.PartiSablonIlaveId = PartiSablonIlave.PartiSablonIlaveId
    where PartiSablonIlave.PartiSablonId = :PartiSablonId and PartiSablonReceteIlave.UrunId = :UrunId
    into :IlaveMiktari, :DuzeltmeMiktari, :IlaveKullanilanMiktar, :DuzeltmeKullanilanMiktar;
    suspend;
  end
end
