    ABASLANGICTARIHI TIMESTAMP,

    MIKTARGUN DOUBLE PRECISION,

    DOKUMASIPARISTIPIIDSTR VARCHAR(30),

    ONMALIYETYAPILDISTR VARCHAR(5),

    CERCEVESAYISI VARCHAR(20),

AS
declare variable shtSatis integer;
declare variable shtSatisIade integer;
begin

  select shtSatis, ShtSatisIade from OrmeStokHareketTipiGetir into :shtSatis, :ShtSatisIade;

  if (ABitti = 1) then
  begin
    for Select DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlama.Tarih, DokumaPlanlama.UretimPartiNo, DokumaPlanlama.FirmaId, OPF.Adi, DokumaPlanlama.DokumaMakinaId, DokumaMakina.Adi,

      Left Join Kullanici SK on SK.KullaniciId = DokumaPlanlama.KullaniciId
      Left Join Kullanici BK on BK.KullaniciId = DokumaPlanlama.BitirenKullaniciId
      left join FirmaYetkili on FirmaYetkili.FirmaId = OPF.FirmaId and FirmaYetkili.DefaultYetkili = 1

      else BittiStr = 'Evet';

      if (KdvDahil = 1) then KdvDahilStr = 'Evet';
      else KdvDahilStr = '';

      if (Beyazlik = 1) then BeyazlikStr = 'Beyazlık';
      else BeyazlikStr = '';

      KalanMiktar = TalimatMiktari - GerceklesenMiktar + SakatUretimToplami;

      suspend;
    end
  end
  else
  begin
    for Select DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlama.Tarih, DokumaPlanlama.UretimPartiNo, DokumaPlanlama.FirmaId, OPF.Adi, DokumaPlanlama.DokumaMakinaId, DokumaMakina.Adi,
           DokumaPlanlama.Sira, DokumaPlanlama.BoyaSiparisKayitId,
           DokumaPlanlama.OrmeSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.HamId, Ham.Adi, Renk.Adi, DokumaPlanlama.PlanlananMiktar,

           OrmeSiparisDetay.TahminiFireOrani, DokumaPlanlama.Bitti, DokumaPlanlama.BitirenKullaniciId, BK.Adi, DokumaPlanlama.BitirmeTarihi,
           DokumaPlanlama.InsertTarihi, DokumaPlanlama.DegisimTarihi, Ik.Adi, Sk.Adi,
           BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4,
           FirmaYetkili.Adi, FirmaYetkili.Telefon, FirmaYetkili.DefaultYetkili,
           OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.RevizeTeslimTarihi, DokumaPlanlama.KdvDahil, DokumaPlanlama.MakinaSayisi, DokumaPlanlama.BaslamaTarihi,

      where (DokumaPlanlama.Tarih >= :ABaslangicTarihi and DokumaPlanlama.Tarih <= :ABitisTarihi) and
            (OPF.AnaHesapFirmaId = :APlanlamaFirmaId or :APlanlamaFirmaId is null) and
            (DokumaPlanlama.DokumaMakinaId = :APlanlamaMakinaId or :APlanlamaMakinaId is null) and
            (DokumaPlanlama.MakinaAyar = 0) and
            (ASF.AnaHesapFirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
            (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
            (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
            (BoyaSiparis.OrmeSiparisSekliId = :AOrmeSiparisSekliId or :AOrmeSiparisSekliId is null) and
            (DokumaMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
            (DokumaPlanlama.Bitti = :ABitti or :ABitti is null) and
            (:ARaporTipi is null or
             (DokumaPlanlama.DokumaMakinaId is not null and :ARaporTipi = 0) or
             (DokumaPlanlama.FirmaId is not null and :ARaporTipi = 1) ) and
            (BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
            (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil  and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId)) and
            (DokumaPlanlama.KdvDahil = :AKdvDahil or :AKdvDahil is Null)
            order by DokumaPlanlama.DokumaMakinaId, DokumaPlanlama.FirmaId, Sira
      into :DokumaPlanlamaId, :Tarih, :UretimPartiNo, :FirmaId, :FirmaAdi, :DokumaMakinaId, :DokumaMakinaAdi,
           :Sira, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId,
           :HamId, :HamAdi, :RenkAdi, :TalimatMiktari, :GerceklesenMiktar, :SakatUretimToplami, :MiktarGun, :MusteriSiparisNo, :ReferansSiparisNo,
           :SiparisFirmaId, :SiparisFirmaAdi, :SiparisSiraNo, :HamMiktar,
           :ArtikelNo, :BirimFiyat, :DovizId, :vadeGun, :Notu,
