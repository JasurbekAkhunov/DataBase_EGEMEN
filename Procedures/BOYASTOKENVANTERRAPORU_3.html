    sum(distinct PartiSinifMaliyetKumulatif.BoyaKimyasalMaliyeti) + sum(distinct PartiSinifMaliyetKumulatif.UnitelerASevkMaliyeti)) from PartiStokDetay

    left join PartiSinifMaliyetKumulatif on PartiSinifMaliyetKumulatif.PartiSinifId = IptalPartiStok.PartiSinifId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId

    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.InsertTarihi >= :ARaporBaslangicTarihi and PartiStok.InsertTarihi <= :ARaporBitisTarihi and

          (Parti.Fason = :AFason or :AFason is Null) and RafDetay.PartiStokIptalKayitId is Not Null and
          HamStokMaster.Renk is not Null and
          (BoyaSiparisDetay.RenkDerinligiId = :RenkDerinligiId or :RenkDerinligiId is Null)
    group by PartiStokDetay.HamId
    into :BoyaIadePartilenenRnk, :BoyaIadePartilenenRnkMt, :BoyaIadePartilenenRnkTL;

  if (BoyaIadePartilenenRnkTL is Null) then BoyaIadePartilenenRnkTL = 0;
  if (AMaliyetleriHesapla <> 1) then BoyaIadePartilenenRnkTL = 0;

  select sum(PartiStokDetay.Kilo), sum(PartiStokDetay.Metre),
    (sum(distinct PartiSinifMaliyetKumulatif.HamStokMaliyeti) + sum(distinct PartiSinifMaliyetKumulatif.AsamaUMUzerindenAsamaMaliyeti) +
    sum(distinct PartiSinifMaliyetKumulatif.BoyaKimyasalMaliyeti) + sum(distinct PartiSinifMaliyetKumulatif.UnitelerASevkMaliyeti)) from PartiStokDetay
    left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
    left join PartiStokIptal on PartiStokIptal.PartiStokIptalKayitId = RafDetay.PartiStokIptalKayitId
    left join PartiStok IptalPartiStok on IptalPartiStok.PartiSinifId = PartiStokIptal.PartiSinifId
    left join PartiSinifMaliyetKumulatif on PartiSinifMaliyetKumulatif.PartiSinifId = IptalPartiStok.PartiSinifId
    left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
    left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.InsertTarihi >= :ARaporBaslangicTarihi and PartiStok.InsertTarihi <= :ARaporBitisTarihi and
          Parti.ImalatTipiId = 1 and PartiStokDetay.HamId = :HamId and
          /*(PartiStok.PartiStokIptalKayitId is null or PartiStokIptal.IptalTarihi <= :ARaporBaslangicTarihi or PartiStokIptal.IptalTarihi >= :ARaporBitisTarihi) and*/
          (Parti.Fason = :AFason or :AFason is Null) and RafDetay.PartiStokIptalKayitId is Not Null and
          HamStokMaster.Renk is not Null and
          (BoyaSiparisDetay.RenkDerinligiId = :RenkDerinligiId or :RenkDerinligiId is Null)
    group by PartiStokDetay.HamId
    into :BaskiIadePartilenenRnk, :BaskiIadePartilenenRnkMt, :BaskiIadePartilenenRnkTL;
  if (BaskiIadePartilenenRnk is Null) then BaskiIadePartilenenRnk = 0;
  if (BaskiIadePartilenenRnkMt is Null) then BaskiIadePartilenenRnkMt = 0;
  if (BaskiIadePartilenenRnkTL is Null) then BaskiIadePartilenenRnkTL = 0;
  if (AMaliyetleriHesapla <> 1) then BaskiIadePartilenenRnkTL = 0;
