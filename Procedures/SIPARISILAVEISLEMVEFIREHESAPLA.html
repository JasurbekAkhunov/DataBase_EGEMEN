CREATE PROCEDURE SIPARISILAVEISLEMVEFIREHESAPLA(
    ABOYASIPARISDETAYID INTEGER,
    AHAMID INTEGER,
    ASIPARISPROSESID INTEGER,
    ARENKDERINLIGIID INTEGER)
RETURNS (
    ILAVEISLEMLER VARCHAR(255),
    ILAVEISLEMLERID VARCHAR(100),
    MAXIMUMFIREORANI DOUBLE PRECISION)
AS
declare variable FAdi varchar(40);
declare variable FIlaveIslemlerId varchar(40);
declare variable FFireOrani Double Precision;
declare variable ProsesFireOrani Double Precision;
begin
  IlaveIslemler = '';
  ilaveIslemlerId = '';
  MaximumFireOrani = Null;

  if (AHamID is not null) then
  begin
    Select MaximumFireOrani from Ham where HamID = :AHamID into :MaximumFireOrani;

    if (ASiparisProsesId is not null) then
    begin
      Select Sum(Asama.FireOrani) from SiparisProsesAsamalari
        left join Asama On Asama.AsamaID = SiparisProsesAsamalari.AsamaId
        where SiparisProsesAsamalari.SiparisProsesID = :ASiparisProsesID and Asama.FireOrani is not null
        into :ProsesFireOrani;

      if (ProsesFireOrani is not null) then MaximumFireOrani = MaximumFireOrani + ProsesFireOrani;
    end

    if (ARenkDerinligiId is not null) then
    begin
      Select FireOrani from RenkDerinligi where RenkDerinligiId = :ARenkDerinligiId into :FFireOrani;

      if (FFireOrani is not null) then MaximumFireOrani = MaximumFireOrani + FFireOrani;
    end
  end

  if (MaximumFireOrani is null) then MaximumFireOrani = 0;

  For Select IlaveIslemler.Adi, IlaveIslemler.FireOrani from SiparisIlaveIslemler
    left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = SiparisIlaveIslemler.IlaveIslemlerId
    where SiparisIlaveIslemler.BoyaSiparisDetayId = :ABoyaSiparisDetayId
    order By IlaveIslemler.Adi
    into :FAdi, :FFireOrani do
  begin
    if (Char_Length(IlaveIslemler) + Char_Length(FAdi) < 250) then
    begin
      if (IlaveIslemler <> '') then IlaveIslemler = IlaveIslemler || cast(', ' as varchar(2) character set UTF8);
      IlaveIslemler = IlaveIslemler || FAdi;
    end

    if (FFireOrani is not null) then MaximumFireOrani = MaximumFireOrani + FFireOrani;
  end

  /*İnteks İlave İşlemler Fiyatlandırma İçin İlave İşlem Id Lerini Tek Tek Gedirdim FASONSIPARISFIYATIHESAPLA Procedure Uygun Olması İçin*/ --Hürdeniz
    For Select IlaveIslemler.ilaveislemlerid from SiparisIlaveIslemler
    left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = SiparisIlaveIslemler.IlaveIslemlerId
    where SiparisIlaveIslemler.BoyaSiparisDetayId = :ABoyaSiparisDetayId
    order By IlaveIslemler.ilaveislemlerid
    into :FIlaveIslemlerId do
  begin
    if (Char_Length(IlaveIslemlerId) + Char_Length(FIlaveIslemlerId) < 250) then
    begin
      if (IlaveIslemlerId <> '') then
       IlaveIslemlerId = IlaveIslemlerId ;
      IlaveIslemlerId = IlaveIslemlerId || FIlaveIslemlerId || cast(',' as varchar(2) character set UTF8);
    end


  end

  suspend;
end
