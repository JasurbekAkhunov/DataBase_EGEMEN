CREATE PROCEDURE TRANSFERICINSTOKFATDETAYGETIR(
    AFATURAID INTEGER)
RETURNS (
    STOKDETAYID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    TRANSFERKODU VARCHAR(55),
    DEPOTRANSFERKODU VARCHAR(50),
    HAREKETDEPOTRANSFERKODU VARCHAR(55),
    ALISTRANSFERKODU VARCHAR(55),
    MUHASEBEHESAPKODU VARCHAR(30),
    MIKTAR DOUBLE PRECISION,
    BIRIMID VARCHAR(2),
    BIRIMTRANSFERKODU VARCHAR(10),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    BIRIMFIYATTL DOUBLE PRECISION,
    BIRIMISKONTOTUTAR DOUBLE PRECISION,
    BIRIMISKONTOORANI1 DOUBLE PRECISION,
    BIRIMISKONTOORANI2 DOUBLE PRECISION,
    BIRIMISKONTOORANI3 DOUBLE PRECISION,
    BIRIMISKONTOORANI4 DOUBLE PRECISION,
    KDVORANI DOUBLE PRECISION,
    KDVTOPLAMI DOUBLE PRECISION,
    KDVMUHASEBEHESAPKODU VARCHAR(30),
    VADEGUN INTEGER,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZKURU DOUBLE PRECISION,
    DOVIZKURU DOUBLE PRECISION,
    ODEMESEKLITRANSFERKODU VARCHAR(10))
AS
declare variable FaturaTarihi       TimeStamp;
declare variable SistemDovizId      varchar(5);
declare variable FKdvYuvarlamaCarpani Double Precision;
declare variable FTevkifat                      integer;
declare variable FTevkifatOrani1                integer;
declare variable FTevkifatOrani2                integer;
declare variable FIslemDovizKuru  Double Precision;
declare variable HareketTipi      Integer;
declare variable shtAlis integer;
begin
  select shtAlis from StokHareketTipiGetir into :shtAlis;
  Select SistemDovizId from Sistem where sistemId = 1 into :SistemDovizId; /* YTL*/

  for select StokDetay.StokDetayId, Urun.UrunId, Urun.TransferKodu, Urun.MuhasebeHesapKodu, StokDetay.Miktar, UrunBirim.BirimId, Birim.TransferKodu, StokDetay.BirimFiyat,
             StokDetay.Kdv, StokDetay.BirimIskontoTutar, StokDetay.BirimIskontoOrani1, StokDetay.BirimIskontoOrani2, StokDetay.BirimIskontoOrani3, StokDetay.BirimIskontoOrani4,
             Fatura.FaturaTarihi, Fatura.DovizId, StokDetay.Vadesi, Irsaliye.SeriNo, Irsaliye.IrsaliyeNo, Irsaliye.IrsaliyeTarihi, Irsaliye.FiiliSevkTarihi,
             Fatura.KdvYuvarlamaCarpani, Fatura.Tevkifat, Fatura.TevkifatOrani1, Fatura.TevkifatOrani2, Urun.AlisTransferKodu, UrunKategori.OzelKod, Fatura.IslemDovizId,
             iif(Fatura.IslemDovizKuru1 > Fatura.IslemDovizKuru2, Fatura.IslemDovizKuru1, Fatura.IslemDovizKuru2), (Fatura.IslemDovizKuru2 / Fatura.IslemDovizKuru1), Urun.Adi,
             Depo.TransferKodu, OdemeSekli.TransferKodu, StokDetay.HareketTipi from StokDetay
    left join Fatura on Fatura.FaturaId = StokDetay.FaturaId
    left join Urun on Urun.UrunId = StokDetay.UrunId
    left join UrunKategori on UrunKategori.UrunKategoriId = Urun.UrunKategoriId
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join Birim on Birim.BirimId = UrunBirim.BirimId
    left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
    left join Depo on Depo.DepoId = StokDetay.DepoId
    left join VerilenSiparis on VerilenSiparis.VerilenSiparisKayitID = StokDetay.VerilenSiparisKayitID
    Left Join OdemeSekli on OdemeSekli.OdemeSekliId = VerilenSiparis.OdemeSekliId
    where StokDetay.FaturaId = :AFaturaId order by StokDetayId
    into :StokDetayId, :UrunId, :TransferKodu, :MuhasebeHesapKodu,  :Miktar, :BirimId, :BirimTransferKodu, :BirimFiyat, :KdvOrani, :BirimIskontoTutar,
         :BirimIskontoOrani1, BirimIskontoOrani2, :BirimIskontoOrani3, :BirimIskontoOrani4, :FaturaTarihi, :DovizId, :VadeGun,
         :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi, :FKdvYuvarlamaCarpani, :FTevkifat, :FTevkifatOrani1,
         :FTevkifatOrani2, :AlisTransferKodu, :DepoTransferKodu, :IslemDovizId, :IslemDovizKuru , :FIslemDovizKuru, :UrunAdi,
         :HareketDepoTransferKodu, :OdemeSekliTransferKodu, :HareketTipi do
  begin
    if (DovizID <> SistemDovizId) then
    begin
      select DovizKuru from DovizKuruGetir(:DovizId, :SistemDovizId, :FaturaTarihi) into :DovizKuru;
    end
    else DovizKuru = 1;

    BirimFiyatTL = BirimFiyat * DovizKuru;

    IslemDovizToplam = BirimFiyat * FIslemDovizKuru * Miktar;

    KdvMuhasebeHesapKodu = Null;

    Select iif(:HareketTipi = :shtAlis, MuhAlisHesapKodu, MuhSatisHesapKodu) from KdvHesapTablosu where KdvHesapTablosu.KdvOrani = :KdvOrani into :KdvMuhasebeHesapKodu;


    KdvToplami = (BirimFiyatTL * Miktar) * KdvOrani / 100;
    KdvToplami = Yuvarla(KdvToplami, FKdvYuvarlamaCarpani);

    if (FTevkifat = 1) then
      KdvToplami = KdvToplami - KdvToplami * FTevkifatOrani1 / FTevkifatOrani2;

    suspend;
  end
end
