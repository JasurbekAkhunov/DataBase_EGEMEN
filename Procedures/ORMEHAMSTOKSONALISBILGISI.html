CREATE PROCEDURE ORMEHAMSTOKSONALISBILGISI(
    AHAMID INTEGER,
    ORTALAMAHESAPLA INTEGER)
RETURNS (
    SONALISFIYATI DOUBLE PRECISION,
    SONALISTARIHI TIMESTAMP,
    DOVIZID VARCHAR(5),
    ORTALAMAALISFIYATI DOUBLE PRECISION,
    SONALISFATURANO INTEGER,
    SONALISIRSALIYENO IRSALIYENOSIZE,
    SONALISFIRMAID VARCHAR(20),
    SONALISFIRMAADI VARCHAR(100))
AS
declare variable FMaxStokHareketId integer;
declare variable UretimeCikisFiyatiHesapOran  Double Precision;
declare variable SistemDovizKuru double precision;
declare variable SistemDovizId  varchar(5);
declare variable OrtalamaDovizId  varchar(5);
begin
  select max(Tarih) from StokHareket  where StokHareket.HamId = :AHamId and StokHareket.HareketTipi = 1 into :SonAlisTarihi;
  select max(StokHareketId) from StokHareket where StokHareket.HamId = :AHamId and StokHareket.HareketTipi = 1 and StokHareket.Tarih = :SonAlisTarihi into :FMaxStokHareketId;

  if (FMaxStokHareketId is not null) then
  begin
    Select StokHareket.BirimFiyat, StokHareket.DovizId, HamTipi.UretimeCikisFiyatiHesapOran, OrmeFatura.FaturaNo,
        OrmeIrsaliye.IrsaliyeNo, Coalesce(OrmeFatura.FirmaID, OrmeIrsaliye.FirmaID), Coalesce(FatFirma.Adi, IrsFirma.Adi) from StokHareket
      left join Ham on Ham.HamId = StokHareket.HamId
      left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeID
      left join OrmeFatura on OrmeFatura.OrmeFaturaId = StokHareket.OrmeFaturaID
      left join Firma IrsFirma on IrsFirma.FirmaId = OrmeIrsaliye.FirmaID
      left join Firma FatFirma on FatFirma.FirmaId = OrmeFatura.FirmaID
      left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
      where StokHareket.StokHareketId = :FMaxStokHareketId
      into :SonAlisFiyati, :DovizId, :UretimeCikisFiyatiHesapOran, :SonAlisFaturaNo, :SonAlisIrsaliyeNo, :SonAlisFirmaID, SonAlisFirmaAdi;

    SonAlisFiyati = SonAlisFiyati + SonAlisFiyati * UretimeCikisFiyatiHesapOran / 100;

    OrtalamaAlisFiyati = Null;
    if (OrtalamaHesapla = 1) then
    begin
      Select SistemDovizId from Sistem where sistemId = 1 into :SistemDovizId; /* YTL*/

      for Select StokHareket.DovizId, HamTipi.UretimeCikisFiyatiHesapOran, avg(StokHareket.BirimFiyat) from StokHareket
        left join Ham on Ham.HamId = StokHareket.HamId
        left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
        where StokHareket.HamId = :AHamId and StokHareket.HareketTipi = 1 and StokHareket.BirimFiyat > 0
        group by StokHareket.DovizId, HamTipi.UretimeCikisFiyatiHesapOran
        into :OrtalamaDovizId, :UretimeCikisFiyatiHesapOran, :OrtalamaAlisFiyati do
      begin
        if (OrtalamaAlisFiyati is not Null) then
        begin
          if (OrtalamaDovizId is not null) then
          begin
            if (OrtalamaDovizId <> SistemDovizId) then
            begin
               select DovizKuru from EnYakinDovizKuruGetir(:OrtalamaDovizId, :SistemDovizId, current_date) into :SistemDovizKuru;
               OrtalamaAlisFiyati = OrtalamaAlisFiyati * SistemDovizKuru;

               Select Fiyat from FormataGoreFiyatYuvarla(:OrtalamaAlisFiyati, 'BOYA', 'BIRIMFIYAT') into :OrtalamaAlisFiyati;
            end
          end
          OrtalamaAlisFiyati = OrtalamaAlisFiyati + OrtalamaAlisFiyati * UretimeCikisFiyatiHesapOran / 100;
        end
      end
    end

    Suspend;
  end
end
