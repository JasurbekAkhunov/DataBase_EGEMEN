CREATE PROCEDURE ORMEHAMSTOKRAFENVANTERRAPORU_3(
    AFASON INTEGER,
    AFIRMAID VARCHAR(20),
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AHAMTIPIID INTEGER,

    ABASLANGICGRUP1ID VARCHAR(4),

    ABASLANGICGRUP2ID VARCHAR(4),

    ABASLANGICGRUP3ID VARCHAR(4),

    ADEPOID VARCHAR(10),
    ABOYALI INTEGER,
    AKUMASCINSIID INTEGER,

    HAREKETTIPI INTEGER,
    HAREKETTIPISTR VARCHAR(30),
    COZGUISEMRIKAYITID INTEGER,
    HASILISEMRIKAYITID INTEGER,
    COZGUISEMRIID VARCHAR(30),
    HASILISEMRIID VARCHAR(30),
    URETIMPARTINO VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAADI2 VARCHAR(100),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    FISNO VARCHAR(30),
    FISTARIHI TIMESTAMP,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    MIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    BIRIMFIYATRAPORD DOUBLE PRECISION,
    TABANDOVIZKURU1 DOUBLE PRECISION,
    HAMIADE SMALLINT,
    HAMIADESTR VARCHAR(5),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISSIRANO INTEGER,
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    SIPARISFIRMAADI2 VARCHAR(100),
    RAFTRANSFERREFERANSNO VARCHAR(30),
    MUSTERIPARTINO VARCHAR(30),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    ESKIRAFID VARCHAR(10),
    ESKIORMERAFID INTEGER,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    HAREKETNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    HAREKETACIKLAMA VARCHAR(50),
    ADET INTEGER,
    PUS INTEGER,
    FEINE INTEGER,
    IPLIKUZUNLUGU1 VARCHAR(40),
    IPLIKUZUNLUGU2 VARCHAR(40),
    IPLIKUZUNLUGU3 VARCHAR(40),
    INDIGOISEMRIKAYITID INTEGER,
    INDIGOISEMRIID VARCHAR(30),
    LEVENTNO INTEGER,
    LEVENTSIRANO INTEGER,
    TELADEDI INTEGER,
    ICCEVREMM INTEGER,
    DISCEVREMM INTEGER,
    TURSAYISI INTEGER,
    FATURAMIKTARTIPI INTEGER,
    FATURAMIKTARTIPISTR VARCHAR(10))
AS
declare variable FDovizKuru     Double precision;
declare variable FFisYil integer;
declare variable FFisNo integer;
begin
/* girisleri */

  For Select OrmeHamStokRafEnvanterRaporu_1.OrmeRafId from OrmeHamStokRafEnvanterRaporu_1(:AFason, :AFirmaId, :ABoyaSiparisYil, :ABoyaSiparisID, :AHamTipiId,
                                 :AHamId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id, :ABaslangicGrup3Id,
                                 :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id, :AHamGrup1Aralikli, :AHamGrup2Aralikli, :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ARafId, :ADepoId, :ABoyali, :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId, :AKaliteNo, :AStokGirisNo, :ALot, :APus, :AFeine, :AUreticiFirmaId, :ASifirOlanlaridaDok, 0, 0, :AFirmaAdi2yiGorebilir, :ARaporDovizId, :AAlisFiyat,
                                 :AOlcuId)
      into :OrmeRafID do
  begin
    EskiRafID     = Null;
    EskiOrmeRafID = Null;
    RafTransferReferansNo = Null;

    For Select StokHareket.StokHareketId, StokHareket.HareketTipi, StokHareket.FirmaId, SHFirma.Adi, SHFirma.Adi2,
               OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeFatura.Faturano, OrmeFatura.FaturaTarihi,
               OrmeIrsaliye.FiiliSevkTarihi, OrmeStokHareketTipi.Adi,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.FirmaId, SiparisFirma.Adi, SiparisFirma.Adi2,
               StokHareket.BirimFiyat, StokHareket.DovizId, StokHareket.TabanDovizKuru1, Ik.Adi, StokHareket.InsertTarihi, Sk.Adi, StokHareket.DegisimTarihi,
               Sum(StokHareket.Miktar), Sum(StokHareket.BrutMiktar), StokHareket.MusteriPartiNo, CozguIsEmri.CozguIsEmriKayitID, HasilIsEmri.HasilIsEmriKayitID,
               CozguIsemri.IsemriID, HasilIsemri.IsemriID, StokHareket.UretimPartiNo, StokHareket.HamIade, StokHareket.Notu, StokHareket.Aciklama, Sum(StokHareket.Adet),
               VerilenSiparisDetay.Pus, VerilenSiparisDetay.Feine, VerilenSiparisDetay.IplikUzunlugu1, VerilenSiparisDetay.IplikUzunlugu2, VerilenSiparisDetay.IplikUzunlugu3, IndigoIsemri.IndigoIsemriKayitId, IndigoIsemri.IsemriId,
               StokHareket.LeventNo, StokHareket.LeventSiraNo, StokHareket.TelAdedi, StokHareket.IcCevreMM, StokHareket.DisCevreMM, StokHareket.TurSayisi, StokHareket.FaturaMiktarTipi from OrmeRafDetay
    left join StokHareket on StokHareket.StokHareketId = OrmeRafDetay.StokHareketID
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitID
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayID
    left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaID
    left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId
    left join OrmeIrsaliye on OrmeIrsaliye.OrmeIrsaliyeId = StokHareket.OrmeIrsaliyeId
    left join OrmeFatura on OrmeFatura.OrmeFaturaId = StokHareket.OrmeFaturaId
    left join Firma SHFirma on SHFirma.Firmaid = StokHareket.Firmaid
    left join OrmeStokHareketTipi on OrmeStokHareketTipi.OrmeStokHareketTipiId = StokHareket.HareketTipi
    left join Kullanici IK on IK.KullaniciId = StokHareket.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId = StokHareket.KullaniciId
    Left join CozguIsemri on CozguIsemri.CozguIsemriKayitId = StokHareket.CozguIsemriKayitId
    Left join HasilIsemri on HasilIsemri.HasilIsemriKayitId = StokHareket.HasilIsemriKayitId
    Left join IndigoIsemri on IndigoIsemri.IndigoIsemriKayitId = StokHareket.IndigoIsemriKayitId
    left join VerilenSiparisDetay on  VerilenSiparisDetay.VerilenSiparisDetayId = StokHareket.VerilenSiparisDetayId
    where OrmeRafDetay.OrmeRafId = :OrmeRafID and StokHareket.StokHareketId is not null
    group by StokHareket.StokHareketId, StokHareket.Notu, StokHareket.Aciklama, StokHareket.HareketTipi, StokHareket.FirmaId, SHFirma.Adi, SHFirma.Adi2,
             OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeFatura.Faturano, OrmeFatura.FaturaTarihi,
             OrmeIrsaliye.FiiliSevkTarihi, OrmeStokHareketTipi.Adi,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.FirmaId, SiparisFirma.Adi, SiparisFirma.Adi2,
             StokHareket.BirimFiyat, StokHareket.DovizId, StokHareket.TabanDovizKuru1, Ik.Adi, StokHareket.InsertTarihi, Sk.Adi, StokHareket.DegisimTarihi, StokHareket.MusteriPartiNo,
             CozguIsEmri.CozguIsEmriKayitID, HasilIsEmri.HasilIsEmriKayitID, CozguIsemri.IsemriID, HasilIsemri.IsemriID, StokHareket.UretimPartiNo, StokHareket.HamIade,
             VerilenSiparisDetay.Pus, VerilenSiparisDetay.Feine, VerilenSiparisDetay.IplikUzunlugu1, VerilenSiparisDetay.IplikUzunlugu2, VerilenSiparisDetay.IplikUzunlugu3, IndigoIsemri.IndigoIsemriKayitId, IndigoIsemri.IsemriId,
             StokHareket.LeventNo, StokHareket.LeventSiraNo, StokHareket.TelAdedi, StokHareket.IcCevreMM, StokHareket.DisCevreMM, StokHareket.TurSayisi, StokHareket.FaturaMiktarTipi
    into :StokHareketId, :HareketTipi, :FirmaId, :FirmaAdi, :FirmaAdi2, :Fisno, :FisTarihi, :IrsaliyeNo, :IrsaliyeTarihi,
         :Faturano, :FaturaTarihi, :FiiliSevkTarihi, :HareketTipiStr,
         :BoyaSiparisYil, :BoyaSiparisID, :SiparisSiraNo, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisFirmaAdi2, :BirimFiyat, :DovizId, :TabanDovizKuru1,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
         :Miktar, :BrutMiktar, :MusteriPartiNo, :CozguIsEmriKayitID, :HasilIsEmriKayitID, :CozguIsemriID, :HasilIsemriID, :UretimPartiNo, :HamIade, :HareketNotu, :HareketAciklama, :Adet, :Pus, :Feine, :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3,
         :IndigoIsemriKayitID, :IndigoIsemriID, :LeventNo, :LeventSiraNo, :TelAdedi, :IcCevreMM, :DisCevreMM, :TurSayisi, :FaturaMiktarTipi do
    begin
      FaturaMiktarTipiStr = '';
      Select MiktarTipi.Adi FROM MiktarTipi where MiktarTipi.MiktarTipiId = :FaturaMiktarTipi into :FaturaMiktarTipiStr;

      HamIadeStr = '';
      if (HamIade = 1) then HamIadeStr = 'Iade';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisYil is not null) then
        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      if (AFirmaAdi2yiGorebilir = 0) then
      begin
        FirmaAdi2 = NUll;
        SiparisFirmaAdi2 = Null;
      end

      if (BirimFiyat is not Null) then
      begin
        if (ARaporDovizId <> DovizId) then
        begin
          select DovizKuru from EnYakinDovizKuruGetir(:DovizId, :ARaporDovizId, Current_date) into :FDovizKuru;
          if (FDovizKuru is not Null) then BirimFiyatRaporD = BirimFiyat * FDovizKuru;
        end
        else
          BirimFiyatRaporD = BirimFiyat;
      end

      suspend;
    end

    StokHareketID         = Null;
    HareketTipi           = Null;
    HareketTipiStr        = Null;
    FirmaId               = Null;
    FirmaAdi              = Null;
    FirmaAdi2             = Null;
    IrsaliyeNo            = Null;
    IrsaliyeTarihi        = Null;
    FiiliSevkTarihi       = Null;
    FisNo                 = Null;
    FisTarihi             = Null;
    FaturaNo              = Null;
    FaturaTarihi          = Null;
    Miktar                = Null;
    BrutMiktar            = Null;
    BirimFiyat            = Null;
    DovizId               = Null;
    BirimFiyatRaporD      = Null;
    TabanDovizKuru1       = Null;
    BoyaSiparisYil        = Null;
    BoyaSiparisId         = Null;
    BoyaSiparisIdStr      = Null;
    SiparisSiraNo         = Null;
    SiparisFirmaId        = Null;
    SiparisFirmaAdi       = Null;
    SiparisFirmaAdi2      = Null;
    InsertKullaniciAdi    = Null;
    InsertTarihi          = Null;
    KullaniciAdi          = Null;
    DegisimTarihi         = Null;
    RafTransferReferansNo = Null;
    MusteriPartiNo        = Null;
    CozguIsEmriKayitID    = Null;
    HasilIsEmriKayitID    = Null;
    IndigoIsEmriKayitID   = Null;
    CozguIsemriID         = Null;
    HasilIsemriID         = Null;
    IndigoIsemriID         = Null;
    UretimPartiNo         = Null;
    HamIade               = Null;
    HamIadeStr            = Null;
    Adet                  = Null;
    Pus                   = Null;
    Feine                 = Null;
    IplikUzunlugu1        = Null;
    IplikUzunlugu2        = Null;
    IplikUzunlugu3        = Null;
    EskiOrmeRafID         = Null;

    For Select OrmeRafTransferFisi.FisYil, OrmeRafTransferFisi.FisNo, OrmeRafTransferFisi.FisTarihi,
               OrmeRafTransferFisiDetay.Kilo, OrmeRafTransferFisiDetay.BrutKilo, OrmeRafTransferFisiDetay.Adet,
               Ik.Adi, OrmeRafTransferFisiDetay.InsertTarihi, Sk.Adi, OrmeRafTransferFisiDetay.DegisimTarihi, EskiRaf.RafID,
               OrmeRafTransferFisi.ReferansNo, OrmeRafTransferFisiDetay.OrmeRafID from OrmeRafTransferFisiDetay
    left join OrmeRafTransferFisi on OrmeRafTransferFisi.OrmeRafTransferFisiid = OrmeRafTransferFisiDetay.OrmeRafTransferFisiid
    left join Kullanici IK on IK.KullaniciId = OrmeRafTransferFisiDetay.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId = OrmeRafTransferFisiDetay.KullaniciId
    left join OrmeRaf EskiRaf on EskiRaf.OrmeRafID = OrmeRafTransferFisiDetay.OrmeRafID
    where OrmeRafTransferFisiDetay.YeniOrmeRafId = :OrmeRafID
    into :FFisYil, :FFisNo, :FisTarihi, :Miktar, :BrutMiktar, :Adet,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :EskiRafID, :RafTransferReferansNo, :EskiOrmeRafID do
    begin
      HareketTipi = -1;

      HareketTipiStr = 'Raf Transfer';

      FisNo = '(' || EgemenIntegerToStr(FFisYil) || ') ' || EgemenIntegerToStr(FFisNo) || ' ';

      suspend;
    end
  end
end
