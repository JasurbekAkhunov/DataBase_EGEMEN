        FLastPartiGrupId = FPartiGrupId;
        FLastMakinaPlanlamaGrupId = FMakinaPlanlamaGrupId;
      end

      if (FIslemGrubuId = 0) then /* beyaz */
      begin
        BeyazSayisi = BeyazSayisi + 1;
        Beyaz       = Beyaz + FKilo;
      end
      else if (FIslemGrubuId = 1) then /* Renkli */
      begin
        RenkliSayisi = RenkliSayisi + 1;
        Renkli       = Renkli + FKilo;
      end
      else if (FIslemGrubuId = 2) then /* Yikama */
      begin
        YikamaSayisi = YikamaSayisi + 1;
        Yikama       = Yikama + FKilo;
      end
      else if (FIslemGrubuId = 3) then /* SekerKasar */
      begin
        SekerKasarSayisi = SekerKasarSayisi + 1;
        SekerKasar       = SekerKasar + FKilo;
      end
      else if (FIslemGrubuId = 4) then /* KuruBolum */
      begin
        KuruBolumSayisi = KuruBolumSayisi + 1;
        KuruBolum       = KuruBolum + FKilo;
      end
      else if (FIslemGrubuId = 5) then /* CiftBoyama */
      begin
        CiftBoyamaSayisi = CiftBoyamaSayisi + 1;
        CiftBoyama       = CiftBoyama + FKilo;
      end
      else if (FIslemGrubuId = 6) then /* Siyah */
      begin
        SiyahSayisi  = SiyahSayisi + 1;
        Siyah        = Siyah + FKilo;
      end
      else if (FIslemGrubuId = 9) then /* AntikBoyama */
      begin
        AntikBoyamaSayisi = AntikBoyamaSayisi + 1;
        AntikBoyama       = AntikBoyama + FKilo;
      end
      else if (FIslemGrubuId = 10) then /* BatikBoyama */
      begin
        BatikBoyamaSayisi = BatikBoyamaSayisi + 1;
        BatikBoyama       = BatikBoyama + FKilo;
      end
      else if (FIslemGrubuId = 11) then /* KukurtBoyama */
      begin
        KukurtBoyamaSayisi = KukurtBoyamaSayisi + 1;
        KukurtBoyama       = KukurtBoyama + FKilo;
      end
      else
      begin
        DigerSayisi = DigerSayisi + 1;
        Diger       = Diger + FKilo;
      end
      ToplamPartiSayisi     = ToplamPartiSayisi + 1;
      ToplamKilo            = ToplamKilo + FKilo;
    end
  end
  if (FLastPartiGrupId <> '-1') then suspend;
