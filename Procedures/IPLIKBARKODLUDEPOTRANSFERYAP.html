CREATE PROCEDURE IPLIKBARKODLUDEPOTRANSFERYAP(
    AIPLIKSTOKID INTEGER,
    AIPLIKIRSALIYEID INTEGER,
    ACIKISIPLIKIRSALIYEDETAYID INTEGER,
    AKULLANICIID INTEGER,
    AKAYNAKDEPOID VARCHAR(10),
    AHEDEFDEPOID VARCHAR(10))
AS
declare variable KayitSayisi    integer;
declare variable FIplikStokId   integer;
declare variable FIplikIrsaliyeId integer;
declare variable FHamId           integer;
declare variable FKayitIplikIrsaliyeDetayId integer;
declare variable FMiktar                Numeric(12,3);
declare variable FNetMiktar             Numeric(12,3);
declare variable FCikisIplikIrsaliyeDetayId integer;
declare variable FRenkID                integer;
declare variable FLot                  varchar(30);
declare variable FMelanjKodu           varchar(20);
declare variable FFlamKodu             varchar(20);
declare variable FKonikRengiId          integer;
declare variable FKonikCinsiId          integer;
declare variable FSarimSekliId          integer;
declare variable FBobinGramaji          Numeric(12,3);
declare variable FAmbalajId             integer;
declare variable FHarmanKayitId         integer;
declare variable FTamirGirisKayitId     integer;
declare variable HareketTipi integer;
declare variable FIplikDepoID           varchar(10);
declare variable FIplikIsletmeId        integer;
declare variable MaxNumarator              integer;
declare variable HamTipiId                 integer;
declare variable hstIplik       integer;
declare variable KS integer;
declare variable SIplikStokId integer;
declare variable FKapno integer;
begin
  Select hstIplik from HamTipiGetir into :hstIplik;

  Select shtDepoTransfer from IplikStokHareketTipiGetir into :HareketTipi;

  if (AIplikIrsaliyeId is not null) then
    Select Max(Numarator) from IplikStok where IplikStok.IplikStokID = :AIplikStokId into :MaxNumarator;
  else
    Select Max(Numarator) from IplikStok where IplikStok.IplikStokID = :AIplikStokId and CikisIplikIrsaliyeDetayID = :ACikisIplikIrsaliyeDetayId into :MaxNumarator;

  if (MaxNumarator is null) then MaxNumarator = -1;

  Select IplikStok.IplikStokId, IplikStok.Miktar, IplikStok.NetMiktar, IplikStok.CikisIplikIrsaliyeDetayId, IplikStok.HamId,
         IplikIrsaliyeDetay.RenkId, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu,
         IplikIrsaliyeDetay.KonikRengiId, IplikIrsaliyeDetay.KonikCinsiId, IplikIrsaliyeDetay.SarimSekliId,
         IplikIrsaliyeDetay.BobinGramaji, IplikIrsaliyeDetay.AmbalajId, IplikStok.IplikDepoId, IplikStok.IplikIsletmeId,
         IplikLot.HarmanKayitId, IplikIrsaliyeDetay.TamirGirisKayitId from IplikStok
    left join IplikUretim On IplikUretim.IplikStokID = IplikStok.IplikStokId
    left join IplikLot On IplikLot.IplikLotID = IplikUretim.IplikLotId
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = IplikStok.IplikIrsaliyeDetayId
    where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator
    into :FIplikStokId, :FMiktar, :FNetMiktar, :FCikisIplikIrsaliyeDetayId, :FHamId, :FRenkId, :FLot, :FMelanjKodu, :FFlamKodu,
         :FKonikRengiId, :FKonikCinsiId, :FSarimSekliId,
         :FBobinGramaji, :FAmbalajId, :FIplikDepoID, :FIplikIsletmeId, :FHarmanKayitId, :FTamirGirisKayitId;

  if (FIplikStokId is null) then exception IplikStokKaydiBulunamadi;

  if (AIplikIrsaliyeId is Not null and FCikisIplikIrsaliyeDetayId is not null) then
  begin

    Select First 1 IplikStok.IplikStokId, IplikStok.Miktar, IplikStok.NetMiktar, IplikStok.CikisIplikIrsaliyeDetayId, IplikStok.HamId,
         IplikIrsaliyeDetay.RenkId, IplikIrsaliyeDetay.Lot, IplikIrsaliyeDetay.MelanjKodu, IplikIrsaliyeDetay.FlamKodu,
         IplikIrsaliyeDetay.KonikRengiId, IplikIrsaliyeDetay.KonikCinsiId, IplikIrsaliyeDetay.SarimSekliId,
         IplikIrsaliyeDetay.BobinGramaji, IplikIrsaliyeDetay.AmbalajId, IplikStok.IplikDepoId, IplikStok.IplikIsletmeId,
         IplikLot.HarmanKayitId, IplikIrsaliyeDetay.TamirGirisKayitId from IplikStok
    left join IplikUretim On IplikUretim.IplikStokID = IplikStok.IplikStokId
    left join IplikLot On IplikLot.IplikLotID = IplikUretim.IplikLotId
    left join IplikIrsaliyeDetay on IplikIrsaliyeDetay.IplikIrsaliyeDetayId = IplikStok.IplikIrsaliyeDetayId
    where IplikStok.IplikIrsaliyeDetayId = :FCikisIplikIrsaliyeDetayId and CikisIplikIrsaliyeDetayId is Null
    into :FIplikStokId, :FMiktar, :FNetMiktar, :FCikisIplikIrsaliyeDetayId, :FHamId, :FRenkId, :FLot, :FMelanjKodu, :FFlamKodu,
         :FKonikRengiId, :FKonikCinsiId, :FSarimSekliId,
         :FBobinGramaji, :FAmbalajId, :FIplikDepoID, :FIplikIsletmeId, :FHarmanKayitId, :FTamirGirisKayitId;

    if (FIplikStokId is null) then exception BuStokDahaOnceStoktanCikmis;

    AIplikStokId = FIplikStokId;

  end

  if (FIplikDepoID <> AKaynakDepoID) then exception IplikTransferDepoUyusmuyor;

  if (AIplikIrsaliyeId is not null) then
  begin
    Select max(IplikIrsaliyeDetayId) from IplikIrsaliyeDetay
      where IplikIrsaliyeDetay.IplikIrsaliyeId = :AIplikIrsaliyeId and
            IplikIrsaliyeDetay.HamId = :FHamId and
            IplikIrsaliyeDetay.IplikDepoId = :AHedefDepoID and
            IntKarsilastir(IplikIrsaliyeDetay.IplikIsletmeId, :FIplikIsletmeId) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.Lot, :FLot) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.MelanjKodu, :FMelanjKodu) = 1 and
            StrKarsilastir(IplikIrsaliyeDetay.FlamKodu, :FFlamKodu) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.AmbalajId, :FAmbalajID) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.HarmanKayitId, :FHarmanKayitId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.RenkId, :FRenkId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.KonikRengiId, :FKonikRengiId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.KonikCinsiId, :FKonikCinsiId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.SarimSekliId, :FSarimSekliId) = 1 and
            IntKarsilastir(IplikIrsaliyeDetay.TamirGirisKayitId, :FTamirGirisKayitId) = 1 and
            IplikIrsaliyeDetay.BobinGramaji = :FBobinGramaji
      into :FKayitIplikIrsaliyeDetayId;

    if (FKayitIplikIrsaliyeDetayId is not Null) then
    begin
      Update IplikStok Set CikisIplikIrsaliyeDetayId = :FKayitIplikIrsaliyeDetayId, Transfer = 0 where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;-- xx
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi + 1, Miktar = Miktar + :FMiktar, NetMiktar = NetMiktar + :FNetMiktar, KullaniciID = :AKullaniciId, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :FKayitIplikIrsaliyeDetayId;

      FIplikStokID = null;
      Select Max(KapNo) from IplikStok where IplikStok.IplikStokId = :AIplikStokId into :FKapNo;

      if (FKapno is null) then FKapno =1; else FKapno = FKapno +1;

      insert into IplikStok(IplikIrsaliyeId, IplikIrsaliyeDetayId, KapNo, Miktar, NetMiktar, IplikStokID)
                   Values(:AIPLIKIRSALIYEID, :FKayitIplikIrsaliyeDetayId, :FKapno ,:FMiktar, :FNetMiktar, :AIplikStokId);
    end
    else
    begin
      FKayitIplikIrsaliyeDetayId = gen_id(IplikIrsaliyeDetay_Id_Gen, 1);
      --Update IplikStok Set CikisIplikIrsaliyeDetayId = :FKayitIplikIrsaliyeDetayId, Transfer=1 where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;--xx
      insert into IplikIrsaliyeDetay(IplikIrsaliyeDetayId, HareketTipi, IplikIsletmeId, IplikDepoId, IplikIrsaliyeId, NeredenUpdate,
                                     TamirGirisKayitId, HamId, RenkId, HarmanKayitId, Lot, MelanjKodu, FlamKodu, SarimSekliId, KonikCinsiId, KonikRengiId, AmbalajId, BobinGramaji, KapAdedi, Miktar, NetMiktar,
                                     InsertKullaniciId, KullaniciId)
                    values(:FKayitIplikIrsaliyeDetayId, :HareketTipi, :FIplikIsletmeId,  :AHedefDepoID, :AIplikIrsaliyeId, 1,
                            :FTamirGirisKayitId, :FHamId, :FRenkId, :FHarmanKayitId, :FLot, :FMelanjKodu, :FFlamKodu, :FSarimSekliId, :FKonikCinsiId, :FKonikRengiId, :FAmbalajId, :FBobinGramaji, 1, :FMiktar, :FNetMiktar,
                            :AKullaniciId, :AKullaniciId);
      Update IplikStok Set CikisIplikIrsaliyeDetayId = :FKayitIplikIrsaliyeDetayId, Transfer=0 where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;
      --
        FIplikStokID = null;
        Select Max(KapNo) from IplikStok where IplikStok.IplikStokId = :AIplikStokId into :FKapNo;

        if (FKapno is null) then FKapno =1; else FKapno = FKapno +1;

        insert into IplikStok(IplikIrsaliyeId, IplikIrsaliyeDetayId, KapNo, Miktar, NetMiktar, IplikStokID)
                     Values(:AIPLIKIRSALIYEID, :FKayitIplikIrsaliyeDetayId, :FKapno ,:FMiktar, :FNetMiktar, :AIplikStokId);
      --
    end
    --Update IplikStok Set CikisIplikIrsaliyeDetayId = :FKayitIplikIrsaliyeDetayId where IplikStok.IplikStokId = :AIplikStokId and IplikStok.Numarator = :MaxNumarator;
  end
  else
  begin
    if (FCikisIplikIrsaliyeDetayId is Null) then exception BuStokSatisiZatenYapilmamis;

    Select count(*) from IplikStok where IplikStok.CikisIplikIrsaliyeDetayId = :FCikisIplikIrsaliyeDetayId into :KayitSayisi;

    Delete from IplikStok where IplikStok.IplikIrsaliyeDetayId = :FCikisIplikIrsaliyeDetayId and IplikStok.IplikStokId = :AIplikStokId;
    Update IplikStok Set CikisIplikIrsaliyeDetayId = Null where IplikStok.IplikStokId = :AIplikStokId  and IplikStok.Numarator = :MaxNumarator;

    if (KayitSayisi = 1) then
    Begin
      delete from IplikIrsaliyeDetay where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :FCikisIplikIrsaliyeDetayId;
    end
    else
    begin
      Update IplikIrsaliyeDetay Set KapAdedi = KapAdedi - 1, Miktar = Miktar - :FMiktar, NetMiktar = NetMiktar - :FNetMiktar, KullaniciID = :AKullaniciId, NeredenUpdate = 1 where IplikIrsaliyeDetay.IplikIrsaliyeDetayId = :FCikisIplikIrsaliyeDetayId;
    end
  end
end
