CREATE PROCEDURE PARTIARAKONTROLRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AASAMAID INTEGER,
    AHAMID INTEGER,
    AFIRMAID VARCHAR(20),
    AONAYDURUMU INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIASAMALARIID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    MAKINASARJID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    DENYE VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ESKIPARTINOSTR VARCHAR(100),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    PARTITAMIRID INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    GIRISTARIHI TIMESTAMP,
    CIKISTARIHI TIMESTAMP,
    ONCEKIASAMACIKISTARIHI TIMESTAMP,
    PROSESNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    BOYAGIRISTARIHI TIMESTAMP,
    BOYACIKISTARIHI TIMESTAMP,
    BOYAGIRISPERSONELID INTEGER,
    BOYAGIRISPERSONELADI VARCHAR(40),
    BOYACIKISPERSONELID INTEGER,
    BOYACIKISPERSONELADI VARCHAR(40),
    BOYAMAKINAID VARCHAR(10))
AS
declare variable itTamir        integer;
declare variable FEskiPartiYil          integer;
declare variable FEskiPartiNo           integer;
begin
  Select itTamir from IslemTipiGetir into :itTamir;

  For Select PartiAsamaStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiAsamaStok.PartiAsamalariId, Parti.FirmaId, Firma.Adi, PartiAsamaStok.PartiSinifId,
             PartiStok.HamAdiToplami, PartiStok.Denye, Parti.Fason, Parti.ReceteId, Recete.Adi, Parti.IslemTipiId, IslemTipi.Adi, PartiAsamaStok.AsamaId, Asama.Adi,
             PartiAsamalari.PartiTamirId, TamirTipi.Adi, PartiStok.Kilo, PartiStok.Metre, PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi, max(PartiAsamaStokDetay.MakinaSarjId) from PartiAsamaStok
    left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join asama on asama.AsamaId = PartiAsamaStok.AsamaId
    left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join PartiAsamaAraKontrol on PartiAsamaAraKontrol.PartiSinifId = PartiAsamaStok.PartiSinifId and PartiAsamaAraKontrol.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
    where PartiAsamaStok.CikisTarihi >= :ABaslangicTarihi and PartiAsamaStok.CikisTarihi <= :ABitisTarihi and
          (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
          (BoyaSiparisDetay.HamID = :AHamId or :AHamId is null) and
          PartiAsamaStok.AsamaId = :AAsamaId and PartiAsamaAraKontrol.PartiKayitId is not null and
          (PartiAsamaAraKontrol.OnayDurumu = :AOnayDurumu + 1 or :AOnayDurumu is null)
    group by PartiAsamaStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiAsamaStok.PartiAsamalariId, Parti.FirmaId, Firma.Adi, PartiAsamaStok.PartiSinifId,
             PartiStok.HamAdiToplami, PartiStok.Denye, Parti.Fason, Parti.ReceteId, Recete.Adi, Parti.IslemTipiId, IslemTipi.Adi, PartiAsamaStok.AsamaId, Asama.Adi,
             PartiAsamalari.PartiTamirId, TamirTipi.Adi, PartiStok.Kilo, PartiStok.Metre, PartiAsamaStok.GirisTarihi, PartiAsamaStok.CikisTarihi
  into :PartiKayitId, :PartiYil, :PartiId, :PartiAsamalariId, :FirmaId, :FirmaAdi, :PartiSinifId, :HamAdiToplami, :Denye, :Fason, :ReceteId, :ReceteAdi, :IslemTipiId,
       :IslemTipiAdi, :AsamaId, :AsamaAdi, :PartiTamirId, :TamirTipiAdi, :Kilo, :Metre, :GirisTarihi, :CikisTarihi, :MakinaSarjId do
  begin
    Select ProsesNotu from PartiAsamalari where PartiAsamalari.PartiAsamalariID = :PartiAsamalariId into :ProsesNotu;

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId)  || ' ';
    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    EskiPartiNoStr = '';

    if (IslemTipiID = itTamir) then
    begin
      for Select HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo from PartiStokDetay
        left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where PartiStokDetay.PartiSinifId = :PartiSinifId
        group by HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo
        into :FEskiPartiYil, :FEskiPartiNo do
      begin
        if (Char_Length(EskiPartiNoStr) < 80) then
        begin
          if (EskiPartiNoStr <> '' ) then EskiPartiNoStr = EskiPartiNoStr || cast(', ' as varchar(2) character set UTF8);

          EskiPartiNoStr = EskiPartiNoStr || '(' || FEskiPartiYil || ') ' || FEskiPartiNo;
        end
      end
    end

    Select max(CikisTarihi) from PartiAsamaStok
      where PartiAsamaStok.PartiSinifID = :PartiSinifID and PartiAsamaStok.CikisTarihi < :GirisTarihi and PartiAsamaStok.PartiAsamalariId <> :PartiAsamalariId
      into :OncekiAsamaCikisTarihi;

    Select first 1 PartiAsamaStokDetay.GirisTarihi, PartiAsamaStokDetay.CikisTarihi, PartiAsamaStokDetay.GirisPersonelId, GirisPersonel.Adi,
    PartiAsamaStokDetay.CikisPersonelId, CikisPersonel.Adi, PartiAsamaStokDetay.MakinaId from PartiAsamaStokDetay
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStokDetay.PartiAsamalariId
      left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
      left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
      left join Asama ReceteAsama on ReceteAsama.AsamaId = Recete.UstAsamaId
      left join Personel GirisPersonel on GirisPersonel.PersonelId = PartiAsamaStokDetay.GirisPersonelId
      left join Personel CikisPersonel on CikisPersonel.PersonelId = PartiAsamaStokDetay.CikisPersonelId
      where PartiAsamaStokDetay.PartiSinifID = :PartiSinifID and PartiAsamaStokDetay.PartiAsamalariId <> :PartiAsamalariId and
      (Asama.BoyaAsamasimi = 1 or ReceteAsama.BoyaAsamasimi = 1)
      order by GirisTarihi
      into :BoyaGirisTarihi, :BoyaCikisTarihi, :BoyaGirisPersonelId, :BoyaGirisPersonelAdi,
      :BoyaCikisPersonelId, :BoyaCikisPersonelAdi, :BoyaMakinaId ;

    suspend;
  end
end
