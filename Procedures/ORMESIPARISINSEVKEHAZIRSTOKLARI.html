CREATE PROCEDURE ORMESIPARISINSEVKEHAZIRSTOKLARI(
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    AHAMTIPIID INTEGER,
    AHAMID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ARAFID VARCHAR(10),
    ABOYALI INTEGER,
    AKALITENO INTEGER,
    ASEVKONCELIKNO INTEGER)
RETURNS (
    ORMERAFID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    STOKGIRISNO INTEGER,
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    RAFID VARCHAR(10),
    URETIMPARTINO VARCHAR(30),
    CIKISPARTINO VARCHAR(30),
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    URETICIADI VARCHAR(40),
    URETICIFIRMAID VARCHAR(20),
    URETICIFIRMAADI VARCHAR(100),
    ORMEMAKINAID VARCHAR(10),
    KALITENO INTEGER,
    STOKFIRMAID VARCHAR(20),
    STOKFIRMAADI VARCHAR(100),
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER,
    SIPARISSIRANO INTEGER,
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    KALITEACIKLAMATIPIID INTEGER,
    KALITEACIKLAMATIPIADI VARCHAR(20),
    MAMULPROSESID INTEGER,
    MAMULPROSESADI VARCHAR(255),
    KAPADEDI NUMERIC(12,3),
    MIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    BRUTMIKTAR NUMERIC(12,3),
    BRUTKALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI INTEGER,
    TAMIR INTEGER,
    TAMIRSTR VARCHAR(4),
    ESKIPARTINOSTR VARCHAR(30),
    SIPARISIALANKULLANICIID INTEGER,
    SIPARISIALANKULLANICIADI VARCHAR(40),
    PUS INTEGER,
    FEINE INTEGER,
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    SEVKONCELIKNO INTEGER,
    OLCUID INTEGER,
    OLCUADI VARCHAR(40))
AS
declare variable BoyaSiparisYil integer;
declare variable BoyaSiparisId  integer;
declare variable Okey           integer;
declare variable HamGrup1       varchar(4);
declare variable HamGrup2       varchar(4);
declare variable HamGrup3       varchar(4);
declare variable HamGrup4       varchar(4);
declare variable EskiPartiYil   integer;
declare variable EskiPartiNo    integer;
declare variable ABoyaSiparisKayitId   integer;
declare variable HamTipiId integer;
declare variable OrmeSiparisTipiID integer;
declare variable astIplik               integer;
declare variable astBoyaliIplik         integer;
declare variable astLycra               integer;
declare variable astHamKumas            integer;
declare variable astBoyaliKumas         integer;
declare variable astIplikBoyaliKumas    integer;

declare variable hstIplik        integer;
declare variable hstLycra        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
declare variable MinBoyaSiparisBaskiId integer;
begin
  Select hstIplik, hstLycra, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstLycra, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;

  Select astIplik, astBoyaliIplik, astLycra, astHamKumas, AstBoyaliKumas, AstIplikBoyaliKumas from OrmeSiparisTipiGetir
    into :astIplik, :astBoyaliIplik, :astLycra, :astHamKumas, :AstBoyaliKumas, :AstIplikBoyaliKumas;

  if (ABoyaSiparisYil is not null) then
  begin
    Select BoyaSiparis.BoyaSiparisKayitID from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisID into :ABoyaSiparisKayitId;
    if (ABoyaSiparisKayitId is null) then ABoyaSiparisKayitId = -1;
  end
  else
    ABoyaSiparisKayitId = Null;

  For Select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.HamId, Ham.Adi, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.FasonStok, OrmeRaf.RafId,
             OrmeRaf.DesenId, Desen.Adi, DesenVaryant.VaryantAdi,
             OrmeRaf.UretimPartiNo, OrmeRaf.CikisPartiNo, OrmeRaf.StokFirmaId, StokFirma.Adi, OrmeRaf.GirisKapAdedi, OrmeRaf.NetGirisMiktari, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
             OrmeRaf.BrutGirisMiktari, OrmeRaf.BrutGirisMiktari - OrmeRaf.BrutCikisMiktari,
             OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi, OrmeRaf.KaliteNo, OrmeRaf.Lot, OrmeRaf.MelanjKodu, Uretici.Adi,
             OrmeRaf.UreticiFirmaId, UreticiFirma.Adi, coalesce(OrmeRaf.OrmeMakinaId, OrmeRaf.OrmeciMakinaNo),
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, OrmeRaf.BoyaSiparisDetayId, OrmeRaf.BoyaSiparisBaskiId, OrmeRaf.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo,
             BoyaSiparis.FirmaId, SiparisFirma.Adi, BoyaSiparis.SiparisiAlanKullaniciId, SAK.Adi,  OrmeRaf.EskiPartiYil, OrmeRaf.EskiPartiNo, OrmeRaf.Tamir,
             Ham.HamGrup1, Ham.HamGrup2, Ham.HamGrup3, Ham.HamGrup4,
             OrmeRaf.KaliteAciklamaTipiId, KaliteAciklamaTipi.Adi, OrmeRaf.MamulProsesId, MamulProses.Adi,
             BoyaSiparisDetay.Pus, BoyaSiparisDetay.Feine, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen,
             OrmeRaf.SevkOncelikNo, Ham.HamTipiId, BoyaSiparis.OrmeSiparisTipiID, OrmeRaf.OlcuId, Olcu.OlcuAdi from OrmeRaf
    left join Ham on Ham.HamId = OrmeRaf.HamId
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join uretici on Uretici.UreticiId = OrmeRaf.UreticiID
    left join Firma StokFirma on StokFirma.Firmaid = OrmeRaf.StokFirmaid
    left join Firma UreticiFirma on UreticiFirma.Firmaid = OrmeRaf.UreticiFirmaid
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
    left join Firma SiparisFirma on SiparisFirma.Firmaid = BoyaSiparis.FirmaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = OrmeRaf.BoyaSiparisDetayId
    left join Desen on Desen.DesenId = OrmeRaf.DesenId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = OrmeRaf.DesenVaryantId
    left join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = OrmeRaf.KaliteAciklamaTipiId
    left join MamulProses on MamulProses.MamulProsesId = OrmeRaf.MamulProsesId
    left join Kullanici SAK on SAK.KullaniciId = BoyaSiparis.SiparisiAlanKullaniciId
    left join Olcu on Olcu.OlcuId = OrmeRaf.OlcuId
    where OrmeRaf.StokVar = 1 and OrmeRaf.BoyaSiparisKayitId is not null and
         (OrmeRaf.HamId = :AHamId or :AHamId is null) and
         (OrmeRaf.BoyaSiparisKayitId = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
         (OrmeRaf.SevkOncelikNo <= :ASevkOncelikNo or :ASevkOncelikNo is null) and
         (OrmeRaf.RafId = :ARafId or :ARafId is null) and
         (OrmeRaf.KaliteNo = :AKaliteNo or :AKaliteNo is null) and
         ( (OrmeRaf.RenkId is null and :ABoyali = 0) or
           (OrmeRaf.RenkId is not null and :ABoyali = 1) or (:ABoyali is null)) and
         ( (:AHamTipiID = 0 and Ham.HamTipiId = :hstIplik) or
           (:AHamTipiID = 1 and Ham.HamTipiId = :hstLycra) or
           (:AHamTipiID = 2 and (Ham.HamTipiId <> :hstLycra and Ham.HamTipiId <> :hstIplik) ) or
            :AHamTipiID is null ) and
         ((:AHamGrup1Aralikli = 1 and HamGrup1 >= :ABaslangicGrup1Id  and (HamGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AHamGrup1Aralikli = 0) or (:AHamGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
         ((:AHamGrup2Aralikli = 1 and HamGrup2 >= :ABaslangicGrup2Id and (HamGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AHamGrup2Aralikli = 0) or (:AHamGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
         ((:AHamGrup3Aralikli = 1 and HamGrup3 >= :ABaslangicGrup3Id and (HamGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AHamGrup3Aralikli = 0) or (:AHamGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
         ((:AHamGrup4Aralikli = 1 and HamGrup4 >= :ABaslangicGrup4Id and (HamGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AHamGrup4Aralikli = 0) or (:AHamGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = ''))
  order by OrmeRaf.SevkOncelikNo, OrmeRaf.BoyaSiparisKayitId, OrmeRaf.OrmeSiparisDetayId, OrmeRaf.BoyaSiparisDetayId
  into :OrmeRafId, :StokGirisNo, :HamId, :HamAdi, :RenkId, :RenkAdi, :Fason, :RafId, :DesenId, :DesenAdi, :VaryantAdi, :UretimPartiNo, :CikisPartiNo,  :StokFirmaId, :StokFirmaAdi,
       :KapAdedi, :Miktar, :KalanMiktar, :BrutMiktar, :BrutKalanMiktar, :KalanKapAdedi, :KaliteNo, :Lot, :MelanjKodu,  :UreticiAdi, :UreticiFirmaId, :UreticiFirmaAdi, :OrmeMakinaId,
       :BoyaSiparisYil, :BoyaSiparisID, :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :OrmeSiparisDetayId, :SiparisSiraNo, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisiAlanKullaniciId, :SiparisiAlanKullaniciAdi,
       :EskiPartiYil, :EskiPartiNo, :Tamir, :HamGrup1, :HamGrup2, :HamGrup3, :HamGrup4,
       :KaliteAciklamaTipiId, :KaliteAciklamaTipiAdi, :MamulProsesId, :MamulProsesAdi,
       :Pus, :Feine, :EnIstenen, :GramajIstenen, :SevkOncelikNo, :HamTipiId, :OrmeSiparisTipiID, :OlcuId, :OlcuAdi do
  begin
    okey = 1;

    if (AHamGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, HamGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, HamGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, HamGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, HamGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      if (HamTipiId = hstIplik) then
      begin
        if (OrmeSiparisTipiID = astBoyaliIplik and BoyaSiparisDetayId is null) then Okey = 0;
        else if (OrmeSiparisTipiID <> astIplik) then Okey = 0;
      end
      else if (HamTipiId = hstLycra) then
      begin
        if (OrmeSiparisTipiID <> astLycra) then Okey = 0;
      end
      else if (HamTipiId = hstElyaf or HamTipiId = hstTelef) then Okey = 0;
      else if (HamTipiId = hstOrmeKumas) then
      begin
        if (OrmeSiparisTipiID = AstBoyaliKumas and BoyaSiparisDetayId is null) then Okey = 0;
        else if (OrmeSiparisTipiID = AstIplikBoyaliKumas and BoyaSiparisDetayId is null) then Okey = 0;
        else if ( (OrmeSiparisTipiID = AstIplikBoyaliKumas or OrmeSiparisTipiID = AstBoyaliKumas) and BoyaSiparisDetayId is not null) then
        begin
          MinBoyaSiparisBaskiId = Null;
          Select Min(BoyaSiparisBaskiId) from BoyaSiparisBaski where BoyaSiparisBaski.BoyaSiparisDetayID = :BoyaSiparisDetayId into :MinBoyaSiparisBaskiId;
          if (MinBoyaSiparisBaskiId is null) then Okey = 0;
        end
      end
    end


    if (Okey = 1) then
    begin
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      if (EskiPartiNo is not Null) then
        EskiPartiNoStr = '(' || EgemenIntegerToStr(EskiPartiYil) || ') ' || EgemenIntegerToStr(EskiPartiNo);
      else
        EskiPartiNoStr = '';

      if (Tamir = 1) then TamirStr = 'Evet';
      else TamirStr = '';

      Suspend;
    end
  end
end
