CREATE PROCEDURE TEDARIKCIDEGERLENDIRMERAPORU(
    AVERILENSIPARISTIPIID INTEGER,
    ASATINALMABIRIMID INTEGER,
    ABASLANGICVERILENSIPARISYIL INTEGER,
    ABASLANGICVERILENSIPARISID INTEGER,
    ABITISVERILENSIPARISYIL INTEGER,
    ABITISVERILENSIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    ASIPARISGRUPID VARCHAR(4),
    ATEDARIKCIFIRMAID VARCHAR(20),
    AURUNID VARCHAR(20),
    AHAMID INTEGER,
    ARAPORTIPI INTEGER,
    ARAPORURUNTIPI INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AGRUP1ARALIKLI INTEGER,
    AGRUP2ARALIKLI INTEGER,
    AGRUP3ARALIKLI INTEGER,
    AGRUP4ARALIKLI INTEGER,
    ABASLANGICURUNKATEGORIID VARCHAR(4),
    ABITISURUNKATEGORIID VARCHAR(4),
    AURUNKATEGORIARALIKLI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    TOPLAMSIPARISMIKTARI DOUBLE PRECISION,
    TOPLAMGELENMIKTAR DOUBLE PRECISION,
    TOPLAMIADEEDILENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    MIKTARPERFORMANSI DOUBLE PRECISION,
    KALITEPERFORMANSI DOUBLE PRECISION,
    TERMINPERFORMANSI DOUBLE PRECISION)
AS
declare variable FFirmaId       varchar(20);
declare variable FLastFirmaId   varchar(20);
declare variable FSiparisMiktari Double precision;
declare variable FGelenMiktar    Double precision;
declare variable FIadeEdilenMiktar  Double precision;
declare variable FTermin         Double precision;
declare variable SipSayisi              Double precision;
declare variable GecikenSipSayisi       Double precision;
begin
  FLastFirmaID = '';

  For SELECT VerilenSiparisRaporu.FirmaId, VerilenSiparisRaporu.Miktar,  VerilenSiparisRaporu.GelenMiktar, VerilenSiparisRaporu.IadeEdilenMiktar, VerilenSiparisRaporu.Termin FROM VerilenSiparisRaporu(:AVerilenSiparisTipiId, :ASatinAlmaBirimId, :ABaslangicVerilenSiparisYil, :ABaslangicVerilenSiparisId, :ABitisVerilenSiparisYil, :ABitisVerilenSiparisId,
                                 :ABaslangicSiparisTarihi, :ABitisSiparisTarihi, :ASiparisGrupId, :ATedarikciFirmaId, :AUrunId, :AHamId, :ARaporTipi, Null, Null, Null, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id, :ABitisGrup2Id,
                                     :ABaslangicGrup3Id, :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id, :AGrup1Aralikli, :AGrup2Aralikli, :AGrup3Aralikli, :AGrup4Aralikli, 0, :ARaporUrunTipi, null, 0, 0, 0, 0, Null, :ABaslangicUrunKategoriId, :ABitisUrunKategoriId, :AUrunKategoriAralikli, Null, Null, Null, Null, Null, Null)
     order by VerilenSiparisRaporu.FirmaId /* degistirme düzgün calısmaz */
    into :FFirmaId, :FSiparisMiktari, :FGelenMiktar, :FIadeEdilenMiktar, :FTermin do
  begin
    if (FLastFirmaId <> FFirmaId) then
    begin
      if (FLastFirmaId <> '') then
      begin
        if (SipSayisi > 0) then
        begin
          TerminPerformansi = (SipSayisi - GecikenSipSayisi) / SipSayisi * 100;
          TerminPerformansi = Yuvarla(TerminPerformansi, 100);
        end

        if (ToplamSiparisMiktari > 0) then
        begin
          MiktarPerformansi = (ToplamSiparisMiktari - KalanMiktar) / ToplamSiparisMiktari * 100;
          MiktarPerformansi = Yuvarla(MiktarPerformansi, 100);

          if (ToplamGelenMiktar <> 0) then
          begin
            KalitePerformansi = (ToplamGelenMiktar - ToplamIadeEdilenMiktar) / ToplamGelenMiktar * 100;
            KalitePerformansi = Yuvarla(KalitePerformansi, 100);
          end

        end
        suspend;
      end
      FLastFirmaId = FFirmaId;
      FirmaId = FFirmaId;
      ToplamSiparisMiktari = 0;
      KalanMiktar          = 0;
      ToplamGelenMiktar    = 0;
      ToplamIadeEdilenMiktar = 0;
      MiktarPerformansi    = 0;
      KalitePerformansi    = 0;
      TerminPerformansi    = 0;
      SipSayisi            = 0;
      GecikenSipSayisi     = 0;
    end
    ToplamSiparisMiktari = ToplamSiparisMiktari + FSiparisMiktari;
    if (FSiparisMiktari - FGelenMiktar  > 0) then KalanMiktar = KalanMiktar + FSiparisMiktari - FGelenMiktar;
    ToplamGelenMiktar = ToplamGelenMiktar + FGelenMiktar;

    ToplamIadeEdilenMiktar = ToplamIadeEdilenMiktar + FIadeEdilenMiktar;

    SipSayisi = SipSayisi + 1;
    if (FTermin = -1) then GecikenSipSayisi = GecikenSipSayisi + 1;
  end

  if (FLastFirmaId <> '') then
  begin
    if (SipSayisi > 0) then
    begin
      TerminPerformansi = (SipSayisi - GecikenSipSayisi) / SipSayisi * 100;
      TerminPerformansi = Yuvarla(TerminPerformansi, 100);
    end

    if (ToplamSiparisMiktari > 0) then
    begin
      MiktarPerformansi = (ToplamSiparisMiktari - KalanMiktar) / ToplamSiparisMiktari * 100;
      MiktarPerformansi = Yuvarla(MiktarPerformansi, 100);

      if (ToplamGelenMiktar <> 0) then
      begin
        KalitePerformansi = (ToplamGelenMiktar - ToplamIadeEdilenMiktar) / ToplamGelenMiktar * 100;
        KalitePerformansi = Yuvarla(KalitePerformansi, 100);
      end
    end
    suspend;
  end
end
