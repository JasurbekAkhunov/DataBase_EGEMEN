CREATE PROCEDURE SATINALMABOYASIPDETAYLISTESI(
    ABOYASIPARISKAYITID INTEGER,
    AVERILENSIPARISTIPIID INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    LABRECETEKODU VARCHAR(20),
    SIPARISKILO NUMERIC(12,3),
    SIPARISMETRE NUMERIC(12,3),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    TERMINNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    RESULTID VARCHAR(255),
    SIPARISRENKADI VARCHAR(100),
    SIPARISDESENNO VARCHAR(20),
    ORMESIPARISDETAYID INTEGER)
AS
declare variable HamTipiId integer;
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstLycra        integer;
declare variable Kayitvar        egeboolean;
declare variable Renkid          integer;
begin
  RenkId = 0;
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstLycra from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstLycra;

  for select BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.MusteriSiparisNo, LabRecete.LabReceteKodu,
             BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.SiparisMetre, BoyaSiparisDetay.HamId, Ham.Adi,
             BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi, BoyaSiparisDetay.TerminNotu, Ham.HamTipiId, BoyaSiparisDetay.MelanjKodu, BoyaSiparisDetay.FlamKodu,
             BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.DesenId    from BoyaSiparisDetay
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  where BoyaSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisDetay.OnayDurumu <> 2 order by BoyaSiparisDetay.BoyaSiparisDetayId
  into :BoyaSiparisDetayId, :MusteriSiparisNo, :LabReceteKodu, :SiparisKilo, :SiparisMetre, :HamId, :HamAdi,
       :TerminTarihi, :RevizeTerminTarihi, :TerminNotu, :HamTipiId, :MelanjKodu, :FlamKodu, :SiparisRenkAdi, :SiparisDesenNo do
  begin
     if ( (AVerilenSiparisTipiId = 0 and (HamTipiId = hstOrmeKumas or HamTipiId = hstRibana or HamTipiId = hstAksesuar)) or
          (AVerilenSiparisTipiId = 1 and HamTipiId = hstDokumaKumas) or
          (AVerilenSiparisTipiId = 2 and (HamTipiId = hstIplik or HamTipiID = hstLycra or HamTipiID = hstDokumaKumas) ) or
          (AVerilenSiparisTipiId is null and HamTipiId = hstIplik) ) then suspend;

      for select HamStokRecete.ReceteHamId, Ham.Adi, Ham.HamTipiId From HamStokRecete
      left join Ham on Ham.HamId = HamStokRecete.ReceteHamId
      where HamStokRecete.HamId = :HamId order by Ham.Adi
      into :HamId, :HamAdi, :HamTipiId do
      begin
         if ( (AVerilenSiparisTipiId = 0 and (HamTipiId = hstOrmeKumas or HamTipiId = hstRibana or HamTipiId = hstAksesuar)) or
              (AVerilenSiparisTipiId = 1 and HamTipiId = hstDokumaKumas) or
              (AVerilenSiparisTipiId = 2 and (HamTipiId = hstIplik or HamTipiID = hstLycra) ) or
              (AVerilenSiparisTipiId is null and HamTipiId = hstIplik) ) then
              begin
                ResultId = '"' || EgemenIntegerToStr(BoyaSiparisDetayId) || '",' ;
                ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",';
                ResultId = ResultId || '"' || EgemenIntegerToStr(OrmeSiparisDetayId) || '",';
                ResultId = ResultId || '"' || EgemenIntegerToStr(RenkId) || '"';
                suspend;
              end
      end
  end

  BoyaSiparisDetayId = null;

  for select OrmeSiparisDetay.OrmeSiparisDetayId, OrmeSiparisDetay.MusteriSiparisNo, HamStokRecete.hamstokreceteid,
             OrmeSiparisDetay.Kilo, OrmeSiparisDetay.Metre, OrmeSiparisDetay.HamId, Ham.Adi,
             OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.RevizeTeslimTarihi, OrmeSiparisDetay.Notu, Ham.HamTipiId, OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.FlamKodu,
             OrmeSiparisDetay.RenkId, null   from OrmeSiparisDetay
  left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
  left join HamStokRecete on  HamStokRecete.ReceteHamId = Ham.HamId
  where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and OrmeSiparisDetay.OnayDurumu <> 2 order by OrmeSiparisDetay.OrmeSiparisDetayId
  into :OrmeSiparisDetayId, :MusteriSiparisNo, :LabReceteKodu, :SiparisKilo, :SiparisMetre, :HamId, :HamAdi,
       :TerminTarihi, :RevizeTerminTarihi, :TerminNotu, :HamTipiId, :MelanjKodu, :FlamKodu, :SiparisRenkAdi, :SiparisDesenNo do
  begin
     KayitVar = 0;
     RenkId = 0;

     if ( (AVerilenSiparisTipiId = 0 and (HamTipiId = hstOrmeKumas or HamTipiId = hstRibana or HamTipiId = hstAksesuar)) or
          (AVerilenSiparisTipiId = 1 and HamTipiId = hstDokumaKumas) or
          (AVerilenSiparisTipiId = 2 and (HamTipiId = hstIplik or HamTipiID = hstLycra or HamTipiID = hstDokumaKumas) ) or
          (AVerilenSiparisTipiId is null and HamTipiId = hstIplik) ) then suspend;

      for select OrmeSiparisDetayRecete.HamId, Ham.Adi, Ham.HamTipiId, Renk.RenkId From OrmeSiparisDetayRecete
      left join Ham on Ham.HamId = OrmeSiparisDetayRecete.HamId
      left join Renk on Renk.RenkId = OrmeSiparisDetayRecete.RenkId
      where OrmeSiparisDetayRecete.OrmeSiparisDetayId = :OrmeSiparisDetayId order by Ham.Adi
      into :HamId, :HamAdi, :HamTipiId, :RenkId do
      begin
         if ( (AVerilenSiparisTipiId = 0 and (HamTipiId = hstOrmeKumas or HamTipiId = hstRibana or HamTipiId = hstAksesuar)) or
              (AVerilenSiparisTipiId = 1 and HamTipiId = hstDokumaKumas) or
              (AVerilenSiparisTipiId = 2 and (HamTipiId = hstIplik or HamTipiID = hstLycra) ) or
              (AVerilenSiparisTipiId is null and HamTipiId = hstIplik) ) then
              begin
                KayitVar = 1;
                ResultId = '"' || EgemenIntegerToStr(BoyaSiparisDetayId) || '",' ;
                ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",';
                ResultId = ResultId || '"' || EgemenIntegerToStr(OrmeSiparisDetayId) || '",';
                ResultId = ResultId || '"' || EgemenIntegerToStr(RenkId) || '"';
                suspend;
              end
      end

      if (KayitVar = 0) then
      begin
          for select HamStokRecete.ReceteHamId, HAM.Adi, Ham.HamTipiId From HamStokRecete
          left join Ham on Ham.HamId = HamStokRecete.ReceteHamId
          where HamStokRecete.HamId = :HamId order by Ham.Adi
          into :HamId, :HamAdi, :HamTipiId do
          begin
             if ( (AVerilenSiparisTipiId = 0 and (HamTipiId = hstOrmeKumas or HamTipiId = hstRibana or HamTipiId = hstAksesuar)) or
                  (AVerilenSiparisTipiId = 1 and HamTipiId = hstDokumaKumas) or
                  (AVerilenSiparisTipiId = 2 and (HamTipiId = hstIplik or HamTipiID = hstLycra) ) or
                  (AVerilenSiparisTipiId is null and HamTipiId = hstIplik) ) then
                  begin
                    ResultId = '"' || EgemenIntegerToStr(BoyaSiparisDetayId) || '",' ;
                    ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",';
                    ResultId = ResultId || '"' || EgemenIntegerToStr(OrmeSiparisDetayId) || '",';
                    ResultId = ResultId || '"' || EgemenIntegerToStr(RenkId) || '"';
                    suspend;
                  end
          end
      end
  end
end
