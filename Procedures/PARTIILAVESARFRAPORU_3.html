CREATE PROCEDURE PARTIILAVESARFRAPORU_3(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    APARTIKAYITID INTEGER,
    ASTOKTANDUSULDU INTEGER,
    APARTIGRUPID VARCHAR(4),
    ASIPARISGRUPID VARCHAR(4),
    ARECETEID VARCHAR(20))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIASAMAILAVESARFID INTEGER,
    PARTIASAMAILAVESARFDETAYID INTEGER,
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BIRIMID VARCHAR(2),
    MIKTARTIPI INTEGER,
    MIKTARTIPISTR VARCHAR(10),
    MIKTAR DOUBLE PRECISION,
    HESAPLANANMIKTAR DOUBLE PRECISION)
AS
declare variable FBanyoCarpani Double Precision;
declare variable FPartiKilo Double Precision;
declare variable FPartiAsamalariID integer;
declare variable FReceteMiktarOrani     Double Precision;
begin
  select RECETEMIKTARORANI from SISTEM where SISTEMID = 1 into :FReceteMiktarOrani;

  if (APartiKayitId is null) then
  begin
    For select PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfDetayId, PartiAsamaIlaveSarf.PartiKayitId, PartiAsamaIlaveSarf.PartiAsamaIlaveSarfId, PartiAsamaIlaveSarfDetay.UrunId,
               Urun.Adi, UrunBirim.BirimId, PartiAsamaIlaveSarfDetay.MiktarTipi, PartiAsamaIlaveSarfDetay.Miktar,
               PartiAsamalari.BanyoCarpani, PartiAsamaIlaveSarf.PartiAsamalariID from PartiAsamaIlaveSarf
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaIlaveSarf.PartiAsamalariID
    left join Recete On Recete.ReceteID = PartiAsamalari.ReceteId
    left join PartiAsamaIlaveSarfDetay on PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfId = PartiAsamaIlaveSarf.PartiAsamaIlaveSarfID
    left join Urun on Urun.UrunId = PartiAsamaIlaveSarfDetay.UrunID
    left join UrunBirim on UrunBirim.UrunBirimId = PartiAsamaIlaveSarfDetay.UrunBirimID
    left join Parti On Parti.PartiKayitId = PartiAsamalari.PartiKayitId
    left join BoyaSiparis On BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where PartiAsamaIlaveSarf.Tarih >= :ABaslangictarihi and PartiAsamaIlaveSarf.Tarih <= :ABitisTarihi and
          PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfID is not null and
          (PartiAsamaIlaveSarf.StoktanDusuldu = :AStoktanDusuldu or :AStoktanDusuldu is null) and
          (Parti.PartiGrupId = :APartiGrupId or :APartiGrupId is Null) and
          (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (Recete.ReceteId = :AReceteId or :AReceteId is null)
    into :PartiAsamaIlaveSarfDetayId, :PartiKayitId, :PartiAsamaIlaveSarfId, :UrunId, :UrunAdi, :BirimId, :MiktarTipi, :Miktar, :FBanyoCarpani, :FPartiAsamalariID do
    begin
      if (BirimID is not null) then
      begin
        if (MiktarTipi = 0) then MiktarTipiStr = BirimId;
        else if (MiktarTipi = 1) then MiktarTipiStr = BirimId || '/' || 'Kg ' ;
        else if (MiktarTipi = 2) then MiktarTipiStr = BirimId || '/' || 'Lt ';
      end
      else MiktarTipiStr = '';

      FPartiKilo = Null;

      Select sum(partiasamastok.GirisKilo)  from PartiAsamalari
        left join partiasamastok on partiasamastok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
        into :FPartiKilo;


      if (FPartiKilo is null) then FPartiKilo = 0;
      FPartiKilo = FPartiKilo / FReceteMiktarOrani;


      if (MiktarTipi = 0) then HesaplananMiktar = Miktar;
      else if (MiktarTipi = 1) then HesaplananMiktar = Miktar * FPartiKilo;
      else if (MiktarTipi = 2) then HesaplananMiktar = Miktar * FBanyoCarpani;
      else HesaplananMiktar = 0;

      suspend;
    end
  end
  else
  begin
    For select PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfDetayId, PartiAsamaIlaveSarf.PartiKayitId, PartiAsamaIlaveSarf.PartiAsamaIlaveSarfId, PartiAsamaIlaveSarfDetay.UrunId,
               Urun.Adi, UrunBirim.BirimId, PartiAsamaIlaveSarfDetay.MiktarTipi, PartiAsamaIlaveSarfDetay.Miktar,
               PartiAsamalari.BanyoCarpani, PartiAsamaIlaveSarf.PartiAsamalariID from PartiAsamaIlaveSarf
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaIlaveSarf.PartiAsamalariID
    left join PartiAsamaIlaveSarfDetay on PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfId = PartiAsamaIlaveSarf.PartiAsamaIlaveSarfID
    left join Urun on Urun.UrunId = PartiAsamaIlaveSarfDetay.UrunID
    left join UrunBirim on UrunBirim.UrunBirimId = PartiAsamaIlaveSarfDetay.UrunBirimID
    where PartiAsamaIlaveSarf.PartiKayitId = :APartiKayitId and
          PartiAsamaIlaveSarfDetay.PartiAsamaIlaveSarfID is not null
    into :PartiAsamaIlaveSarfDetayId, :PartiKayitId, :PartiAsamaIlaveSarfId, :UrunId, :UrunAdi, :BirimId, :MiktarTipi, :Miktar, :FBanyoCarpani, :FPartiAsamalariID do
    begin
      if (BirimID is not null) then
      begin
        if (MiktarTipi = 0) then MiktarTipiStr = BirimId;
        else if (MiktarTipi = 1) then MiktarTipiStr = BirimId || '/' || 'Kg ' ;
        else if (MiktarTipi = 2) then MiktarTipiStr = BirimId || '/' || 'Lt ';
      end
      else MiktarTipiStr = '';

      FPartiKilo = Null;

      Select sum(partiasamastok.GirisKilo)  from PartiAsamalari
        left join partiasamastok on partiasamastok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1
        into :FPartiKilo;


      if (FPartiKilo is null) then FPartiKilo = 0;

      FPartiKilo = FPartiKilo / FReceteMiktarOrani;

      if (MiktarTipi = 0) then HesaplananMiktar = Miktar;
      else if (MiktarTipi = 1) then HesaplananMiktar = Miktar * FPartiKilo;
      else if (MiktarTipi = 2) then HesaplananMiktar = Miktar * FBanyoCarpani;
      else HesaplananMiktar = 0;

      suspend;
    end
  end
end
