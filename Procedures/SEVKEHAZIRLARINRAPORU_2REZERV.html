CREATE PROCEDURE SEVKEHAZIRLARINRAPORU_2REZERV(
    AFIRMAID VARCHAR(20),
    AHAMTIPIID INTEGER,
    ATEMSILCIID INTEGER,
    AFASON INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AHAMID INTEGER,
    AKALITEID VARCHAR(20),
    ALABRECETEID INTEGER,
    AIPLIKCINSIID INTEGER,
    APARTIGRUPID VARCHAR(4),
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ADISUNITEDURUMU INTEGER,
    ABOYASIPARISYIL INTEGER,
    ABOYASIPARISID INTEGER,
    ADESENID VARCHAR(20),
    AUNITEID INTEGER,
    AYURTICIYURTDISI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAGRUP1ID FIRMAGRUPKODUSIZE,
    FIRMAGRUP2ID FIRMAGRUPKODUSIZE,
    FIRMAGRUP1ADI ADI30,
    FIRMAGRUP2ADI ADI30,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTITIPIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTIKAYITID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    ISLEMTIPIADI VARCHAR(20),
    TERMINTARIHI TIMESTAMP,
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI VARCHAR(20),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    PARTIKILO NUMERIC(12,3),
    PARTIMETRE NUMERIC(12,3),
    SEVKEDILENKILO NUMERIC(12,3),
    SEVKEDILENMETRE NUMERIC(12,3),
    SEVKEHAZIRKILO NUMERIC(12,3),
    SEVKEHAZIRMETRE NUMERIC(12,3),
    REZERVKILO BOYAMIKTARSIZE,
    REZERVMETRE BOYAMIKTARSIZE,
    REZERVMIKTAR BOYAMIKTARSIZE,
    REZSEVKEDILENKILO BOYAMIKTARSIZE,
    REZSEVKEDILENBRUTKILO BOYAMIKTARSIZE,
    MODELNO VARCHAR(20),
    TARTIBRUT NUMERIC(12,3),
    PARTIGRUPID VARCHAR(4),
    PARTIGRUPADI VARCHAR(30),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    MINSIPARISMIKTARI BOYAMIKTARSIZE,
    TASARIMNO VARCHAR(20),
    ACIKLAMA VARCHAR(255),
    TARTINET NUMERIC(12,3),
    YURTICIYURTDISI INTEGER,
    LRACOLOR INTEGER,
    OYDOSYAADI VARCHAR(255),
    SIPARISGRUP3ID VARCHAR(4),
    SIPARISGRUP3ADI VARCHAR(30),
    SIPARISGRUP4ID VARCHAR(4),
    SIPARISGRUP4ADI VARCHAR(30))
AS
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef       integer;
declare variable ABoyaSiparisKayitId integer;
declare variable hstOrmeKumas   integer;
declare variable hstRibana      integer;
declare variable hstAksesuar    integer;
declare variable hstDokumaKumas integer;
begin
  Select hstIplik, hstElyaf, hstTelef, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar;

  if (ABoyaSiparisYil is not Null) then
  begin
    Select BoyaSiparis.BoyaSiparisKayitId from BoyaSiparis where BoyaSiparis.BoyaSiparisYil = :ABoyaSiparisYil and BoyaSiparis.BoyaSiparisId = :ABoyaSiparisId into :ABoyaSiparisKayitId;
    if (ABoyaSiparisKayitId is null) then ABoyaSiparisKayitId = -1;
  end

  for select Firma.FirmaId, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, PS1.PartiKayitId, IslemTipi.Adi, Parti.ReceteId, Recete.Adi, Parti.ModelNo,
             Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Parti.Fason, Parti.TerminTarihi,
             Recete.RenkDerinligiId, RenkDerinligi.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, Parti.PartiGrupId, PartiGrup.Adi,
    (select sum(PS2.Kilo) from PartiStok PS2 where Ps2.PartiKayitId = PS1.PartiKayitId),
    (select sum(PS3.Metre) from PartiStok PS3 where Ps3.PartiKayitId = PS1.PartiKayitId),
    (sum(PS1.Kilo) - (sum(PS1.BrutSevkKilo) + sum(PS1.BrutSarfKilo))), sum(PS1.BrutSevkKilo),
    (sum(PS1.Metre) - (sum(PS1.BrutSevkMetre) + sum(PS1.BrutSarfMetre))), sum(PS1.BrutSevkMetre),
    Sum(PartiStokTarti.Dara + PartiStokTarti.Kilo), Sum(PartiStokTarti.Metre), Sum(PartiStokTarti.Kilo), DesenVaryant.DesenID, Desen.Adi, DesenVaryant.VaryantAdi, Min(Kalite.MinSiparisMiktari),
    FG1.FirmaGrupId, FG1.Adi, FG2.FirmaGrupId, FG2.Adi, Desen.TasarimNo, DesenVaryant.Aciklama, BoyaSiparis.Yurtdisi, LabReceteAtis.Color, Desen.OzelYerlesimDosyaAdi,
    BoyaSiparis.SiparisGrup3Id, SiparisGrup3.Adi, BoyaSiparis.SiparisGrup4Id, SiparisGrup4.Adi from BoyaRezervFisiDetay
    left join PartiStok PS1 on PS1.PartiSinifId = BoyaRezervFisiDetay.PartiSinifId
    left join Ham on Ham.HamId = PS1.HamId
    left join BoyaSiparisDetay On BoyaSiparisDetay.BoyaSiparisDetayId = BoyaRezervFisiDetay.BoyaSiparisDetayId
    left join Parti on Parti.PartiKayitId = PS1.PartiKayitId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = Recete.RenkDerinligiId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join FirmaGrup FG1 on FG1.FirmaGrupId = Firma.FirmaGrup1
    left join FirmaGrup FG2 on FG2.FirmaGrupId = Firma.FirmaGrup2
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join SiparisGrup SiparisGrup3 on SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
    left join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
    left join IplikCinsi On IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    left join PartiGrup on PartiGrup.PartiGrupId = Parti.PartiGrupId
    left join DesenVaryant on DesenVaryant.DesenVaryantID = Parti.DesenVaryantID
    left join Desen on Desen.DesenID = DesenVaryant.DesenID
    left join Kalite on Kalite.KaliteId = PS1.KaliteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId =  BoyaSiparisdetay.LabReceteAtisId
    join PartiStokTarti on PartiStokTarti.BoyaRezervFisiDetayId = BoyaRezervFisiDetay.BoyaRezervFisiDetayId
    where PS1.durumu = 7 and PS1.PartiStokIptalKayitId is null and (Parti.FirmaId = :AFirmaId or :AFirmaId is null) and
        (Parti.Fason = :AFason or :AFason is null) and
        (PS1.HamID = :AHamID or :AHamID is null) and
       ( ( (Ham.HamTipiId = :hstIplik or Ham.HamTipiId = :hstElyaf or Ham.HamTipiId = :hstTelef) and :AHamTipiId = 0) or
         ( (Ham.HamTipiId = :hstOrmeKumas or Ham.HamTipiId = :hstRibana or Ham.HamTipiId = :hstAksesuar) and :AHamTipiId = 1) or
         (Ham.HamTipiId = :hstDokumaKumas and :AHamTipiId = 2) or
         (Ham.HamTipiId <> :hstIplik and Ham.HamTipiId <> :hstElyaf  and Ham.HamTipiId <> :hstTelef and :AHamTipiId = 3) or :AHamTipiId is null) and
       (Firma.TemsilciID = :ATemsilciId or :ATemsilciId is null) and
       (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
       (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
       (BoyaSiparisDetay.KaliteID = :AKaliteId or :AKaliteId is Null) and
       (BoyaSiparisDetay.LabReceteId = :ALabReceteId or :ALabReceteId is Null) and
       (BoyaSiparisDetay.MusteriSiparisNo = :AMusteriSiparisNo or :AMusteriSiparisNo is Null) and
       (BoyaSiparisDetay.ReferansSiparisNo = :AReferansSiparisNo or :AReferansSiparisNo is Null) and
       (IplikCinsi.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is Null) and
       ( (PS1.UniteID is null and :ADisUniteDurumu = 0) or
         (PS1.UniteID is not null and :ADisUniteDurumu = 1) or :ADisUniteDurumu is null) and
       (parti.PartiGrupId = :APartiGrupId or :APartiGrupId is Null) and
       (BoyaSiparis.BoyaSiparisKayitID = :ABoyaSiparisKayitId or :ABoyaSiparisKayitId is null) and
       (PartiStokTarti.BoyaIrsaliyeId is null and PartiStokTarti.HamStokSarfFisiId is null) and
       (DesenVaryant.DesenId =: ADesenId or :ADesenId is Null) and
       (PS1.UniteId = :AUniteId or :AUniteId is Null ) and
       (BoyaSiparis.Yurtdisi = :AYurticiYurtdisi or :AYurticiYurtdisi is null)
    group by Firma.FirmaId, FG1.FirmaGrupId, FG1.Adi, FG2.FirmaGrupId, FG2.Adi, Parti.PartiYil, Parti.PartiId, Parti.PartiTipiId, PS1.PartiKayitId, IslemTipi.Adi, Parti.ReceteId, Recete.Adi, Parti.ModelNo,
             Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, Parti.Fason, Parti.TerminTarihi,
             Recete.RenkDerinligiId, RenkDerinligi.Adi, BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, Parti.PartiGrupId, PartiGrup.Adi,
             DesenVaryant.DesenID, Desen.Adi, DesenVaryant.VaryantAdi, Desen.TasarimNo, DesenVaryant.Aciklama, BoyaSiparis.Yurtdisi, LabReceteAtis.Color, Desen.OzelYerlesimDosyaAdi,
             BoyaSiparis.SiparisGrup3Id, SiparisGrup3.Adi, BoyaSiparis.SiparisGrup4Id, SiparisGrup4.Adi
    into :FirmaId, :PartiYil, :PartiId, :PartiTipiId, :PartiKayitId, :IslemTipiAdi, :ReceteId, :ReceteAdi, :ModelNo,
         :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :Fason, :TerminTarihi,
         :RenkDerinligiId, :RenkDerinligiAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :PartiGrupId, PartiGrupAdi,
         :PartiKilo, :PartiMetre, :SevkeHazirKilo, :SevkEdilenKilo, :SevkeHazirMetre, :SevkEdilenMetre, :RezervKilo, :RezervMetre, :RezervMiktar, :DesenID, :DesenAdi,
         :VaryantAdi, :MinSiparisMiktari, :FirmaGrup1Id, :FirmaGrup1Adi, :FirmaGrup2Id, :FirmaGrup2Adi, :TasarimNo, :Aciklama, :YurticiYurtdisi, :LRAColor, :OYDosyaAdi,
         :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi do
    begin
      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') '  || EgemenIntegerToStr(PartiId);

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      Select Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara), Sum(PartiStokTarti.Kilo - PartiStokTarti.Dara) from PartiStokTarti
        where PartiStokTarti.PartiKayitID = :PARTIKAYITID
         into :TartiBrut, :TartiNet;

      Select Sum(PartiStokTarti.Kilo), Sum(PartiStokTarti.Kilo + PartiStokTarti.Dara) from PartiStokTarti
        where PartiStokTarti.PartiKayitID = :PARTIKAYITID
         and PartiStokTarti.BoyaIrsaliyeDetayId is not null and PartiStokTarti.BoyaRezervFisiDetayId is not null
         into :RezSevkEdilenKilo, :RezSevkEdilenBrutKilo;

      suspend;
    end
end
