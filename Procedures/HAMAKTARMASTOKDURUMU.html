CREATE PROCEDURE HAMAKTARMASTOKDURUMU(
    AFASON INTEGER)
RETURNS (
    HAMSTOKDETAYID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMAKTARMAYAGELENKILO NUMERIC(12,3),
    AKTARILMISKILO NUMERIC(12,3),
    AKTARILMAMISKILO NUMERIC(12,3),
    KULLANILANKILO NUMERIC(12,3),
    DEPOYAIADEEDILENKILO NUMERIC(12,3),
    KALANKILO NUMERIC(12,3))
AS
declare variable FLastHamStokDetayId            integer;
declare variable FLastHAktarmaFisiDetayId       integer;
declare variable FHAktarmaFisiDetayId           integer;
declare variable FHamStokDetayId                integer;
declare variable FHamId                         integer;
declare variable FHamAdi                        varchar(255);
declare variable FBitisTarihi                   TimeStamp;
declare variable FGelenKilo                     Numeric(12,3);
declare variable FIadeEdilenKilo                Numeric(12,3);
declare variable FIsEmriHazirlananKilo          Numeric(12,3);
declare variable FIsEmriKilo                    Numeric(12,3);

declare variable FFason                         integer;
declare variable FIrsaliyeNo                    IrsaliyeNoSize;
declare variable FIrsaliyeTarihi                TimeStamp;
declare variable FFirmaId                       varchar(20);
declare variable FFirmaAdi                      varchar(100);
begin
  FLastHamStokDetayId = -1;
  FLastHAktarmaFisiDetayID = -1;

  For Select HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.Fason, HamStokIrsaliye.FirmaId, Firma.Adi, HAktarmaFisiDetay.HAktarmaFisiDetayId, HAktarmaFisiDetay.HamStokDetayId, HamStokDetay.HamId, Ham.Adi, HAktarmaFisiDetay.Kilo,
             HAktarmaFisiDetay.IsEmriHazirlananKilo, HAktarmaFisiDetay.IadeEdilenKilo,
             HAktarmaIsEmri.BitisTarihi, HAktarmaIsEmriDetay.Kilo from HAktarmaFisiDetay
    left join HamStokDetay on HamStokDetay.HamStokDetayId = HAktarmaFisiDetay.HamStokDetayId
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
    left join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
    left join Ham on Ham.HamId = HamStokDetay.HamId
    left join HAktarmaIsEmriDetay on HAktarmaIsEmriDetay.HAktarmaFisiDetayId = HAktarmaFisiDetay.HAktarmaFisiDetayId
    left join HAktarmaIsEmri on HAktarmaIsEmri.HAktarmaIsEmriKayitId = HAktarmaIsEmriDetay.HAktarmaIsEmriKayitId
    where HamStokIrsaliye.Fason = :AFason or :AFason is null
    order by HAktarmaFisiDetay.HamStokDetayId, HAktarmaFisiDetay.HAktarmaFisiDetayId
  into :FIrsaliyeNo, :FIrsaliyeTarihi, :FFason, :FFirmaId, :FFirmaAdi, :FHAktarmaFisiDetayId, :FHamStokDetayId, :FHamId, :FHamAdi, :FGelenKilo, :FIsEmriHazirlananKilo, :FIadeEdilenKilo, :FBitisTarihi, :FIsEmriKilo do
  begin
    if (FHamStokDetayId <> FLastHamStokDetayId) then
    begin
      if (FLastHamStokDetayId <> -1) then
      begin
        Select Sum(RafDetay.IadeEdilenKilo + RafDetay.PartilenenKilo + RafDetay.SatisKilo) from RafDetay where RafDetay.HamStokDetayID = :HamStokDetayId and RafDetay.Aktarma = 1 into :KullanilanKilo;

        if (KullanilanKilo is null) then KullanilanKilo = 0;
        KalanKilo = HamAktarmayaGelenKilo - DepoyaIadeEdilenKilo - KullanilanKilo;

        Suspend;
      end

      HamAktarmayaGelenKilo  = 0;
      AktarilmisKilo         = 0;
      AktarilmamisKilo       = 0;
      DepoyaIadeEdilenKilo   = 0;
      KullanilanKilo         = 0;
      KalanKilo              = 0;
      FLastHamStokDetayId = FHamStokDetayId;

      HamId = FHamId;
      HamAdi = FHamAdi;
      HamStokDetayId = FHamStokDetayId;
      IrsaliyeNo = FIrsaliyeNo;
      IrsaliyeTarihi = FIrsaliyeTarihi;
      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      Fason = FFason;

      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';
    end

    if (FHAktarmaFisiDetayId <> FLastHAktarmaFisiDetayId) then
    begin
      HamAktarmayaGelenKilo  = HamAktarmayaGelenKilo + FGelenKilo;
      DepoyaIadeEdilenKilo   = DepoyaIadeEdilenKilo + FIadeEdilenKilo;
      AktarilmamisKilo = AktarilmamisKilo + FGelenKilo - FIadeEdilenKilo - FIsEmriHazirlananKilo;

      FLastHAktarmaFisiDetayId = FHAktarmaFisiDetayId;
    end


    if (FIsEmriKilo is not null) then
    begin
      if (FBitisTarihi is not null) then AktarilmisKilo = AktarilmisKilo + FIsEmriKilo;
      else AktarilmamisKilo = AktarilmamisKilo + FIsEmriKilo;
    end

  end

  if (FLastHamStokDetayId <> -1) then
  begin
    Select Sum(RafDetay.IadeEdilenKilo + RafDetay.PartilenenKilo + RafDetay.SatisKilo) from RafDetay where RafDetay.HamStokDetayID = :HamStokDetayId and RafDetay.Aktarma = 1 into :KullanilanKilo;

    if (KullanilanKilo is null) then KullanilanKilo = 0;
    KalanKilo = HamAktarmayaGelenKilo - DepoyaIadeEdilenKilo - KullanilanKilo;
    Suspend;
  end
end
