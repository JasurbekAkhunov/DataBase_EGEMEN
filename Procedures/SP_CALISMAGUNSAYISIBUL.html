CREATE PROCEDURE SP_CALISMAGUNSAYISIBUL(
    ABASLAMATARIHI TIMESTAMP,
    AMIKTARGUN INTEGER,
    ASIPARISMIKTARI DOUBLE PRECISION)
RETURNS (
    BITISTARIHI TIMESTAMP)
AS
declare variable CalismaGunu Numeric(12,3);
declare variable FDay integer;
declare variable FBaslamaTarihi TimeStamp;
--
declare variable FHGHaftaIciVardiyaBaslamaSaati Time;
declare variable FHGHaftaIciVardiyaBitisSaati time;
declare variable FHGCumartesiVardiyaBaslamaSaati time;
declare variable FHGCumartesiVardiyaBitisSaati time;
declare variable FHGPazarVardiyaBaslamaSaati time;
declare variable FHGPazarVardiyaBitisSaati time;

declare variable FHaftaIcCalismaSn integer;
declare variable FCumartesiCalismaSn integer;
declare variable FPazarCalismaSn integer;
declare variable FKalanSureSn integer;
declare variable TmpSn integer;
declare variable HaftaIciVar integer;
declare variable CumartesiVar integer;
declare variable PazarVar integer;

begin
  FHaftaIcCalismaSn = null;
  FCumartesiCalismaSn = null;
  FPazarCalismaSn = null;
  FKalanSureSn = null;
  CumartesiVar = 0;
  PazarVar = 0;

  select HGHaftaIciVardiyaBaslamaSaati, HGHaftaIciVardiyaBitisSaati, HGCumartesiVardiyaBaslamaSaati, HGCumartesiVardiyaBitisSaati, HGPazarVardiyaBaslamaSaati, HGPazarVardiyaBitisSaati from sistem
  into :FHGHaftaIciVardiyaBaslamaSaati, :FHGHaftaIciVardiyaBitisSaati, :FHGCumartesiVardiyaBaslamaSaati, :FHGCumartesiVardiyaBitisSaati, :FHGPazarVardiyaBaslamaSaati, :FHGPazarVardiyaBitisSaati;


  if (FHGHaftaIciVardiyaBaslamaSaati <> FHGHaftaIciVardiyaBitisSaati) then
  begin
    FHaftaIcCalismaSn = (FHGHaftaIciVardiyaBitisSaati - FHGHaftaIciVardiyaBaslamaSaati);
    HaftaIciVar = 1;
  end

  if (FHGCumartesiVardiyaBaslamaSaati <> FHGCumartesiVardiyaBitisSaati) then
  begin
    FCumartesiCalismaSn = (FHGCumartesiVardiyaBitisSaati - FHGCumartesiVardiyaBaslamaSaati);
    CumartesiVar = 1;
  end

  if (FHGPazarVardiyaBaslamaSaati <> FHGPazarVardiyaBitisSaati) then
  begin
    FPazarCalismaSn = (FHGPazarVardiyaBitisSaati - FHGPazarVardiyaBaslamaSaati);
    PazarVar = 1;
  end


  CalismaGunu = ASiparisMiktari / AMiktarGun; -- Kaç gün sürecek
  FKalanSureSn = FHaftaIcCalismaSn * CalismaGunu; -- Kaç Gün Sürecek (Saniye Cinsinden)


  if (HaftaIciVar = 0 and CumartesiVar = 0 and PazarVar = 0) then
    exception kayithatasi 'Calisma Süreleri Girilmemiş'; -- calisma sureleri girilmedi ise çık

  if (cast( ABaslamaTarihi as time) < FHGHaftaIciVardiyaBaslamaSaati) then
    exception kayithatasi 'Başlangıc saati Vardiya başlama saatinden küçük olamaz';


  FBaslamaTarihi = ABaslamaTarihi;


  while (0 < FKalanSureSn) do
  begin
    FDay = (extract(weekday from FBaslamaTarihi)); -- hangi gündeyiz
    tmpSn = 0;

    if ((FDay = 0) and (FPazarCalismaSn is not null)) then -- pazar
    TmpSn = FHGPazarVardiyaBitisSaati - cast(FBaslamaTarihi as time); -- Gün sonuna kalan süre sn

    if ((FDay = 6) and (FCumartesiCalismaSn is not null)) then --cumartesi
      TmpSn = FHGCumartesiVardiyaBitisSaati - cast(FBaslamaTarihi as time)  ; -- Gün sonuna kalan süre sn

    if ((FDay <> 6) and (FDay <> 0)) then -- haftaici
      TmpSn = FHGHaftaIciVardiyaBitisSaati - cast(FBaslamaTarihi as time)  ; -- Gün sonuna kalan süre sn

    if (FKalanSureSn > tmpSn ) then
    begin
      FBaslamaTarihi = DateAdd(second, TMpSn,FBaslamaTarihi);
      FKalanSureSn = FKalanSureSn - tmpSn;

      FBaslamaTarihi = DateAdd(Day, 1,FBaslamaTarihi);
      FDay = (extract(weekday from FBaslamaTarihi));

      if ((FDay <> 6) and (FDay <> 0) and HaftaIciVar = 1) then
      begin
        FBaslamaTarihi = cast(FBaslamaTarihi as date)||' '||FHGHaftaIciVardiyaBaslamaSaati;
      end
      else if ((FDay <> 6) and (FDay <> 0) and HaftaIciVar = 0) then
      begin
        FBaslamaTarihi = DateAdd(Day, 1,FBaslamaTarihi);
        FBaslamaTarihi = cast(FBaslamaTarihi as date)||' '||'00:00:01';
      end
      else if ((FDay = 6)  and CumartesiVar = 1) then
      begin
        FBaslamaTarihi = cast(FBaslamaTarihi as date)||' '||FHGCumartesiVardiyaBaslamaSaati;
      end
      else if ((FDay = 6)  and CumartesiVar = 0) then
      begin

        FBaslamaTarihi = DateAdd(Day, 1,FBaslamaTarihi);
        FBaslamaTarihi = cast(FBaslamaTarihi as date)||' '||FHGHaftaIciVardiyaBaslamaSaati;
      end
      else if ((FDay = 0)  and PazarVar = 1) then
      begin
        FBaslamaTarihi = cast(FBaslamaTarihi as date)||' '||FHGPazarVardiyaBaslamaSaati;
      end
      else if ((FDay = 0)  and PazarVar = 0) then
      begin
        FBaslamaTarihi = DateAdd(Day, 1,FBaslamaTarihi);
        FBaslamaTarihi = cast(FBaslamaTarihi as date)||' '||FHGHaftaIciVardiyaBaslamaSaati;
      end
    end
    else
    begin
      FBaslamaTarihi = DateAdd(second, FKalanSureSn, FBaslamaTarihi);
      FKalanSureSn = 0;
    end
  end

  BitisTarihi = FBaslamaTarihi;

  suspend;
end
