CREATE PROCEDURE BOYAFURETIMFISIICINCIKISLISTESI(
    AIRSALIYEID INTEGER)
RETURNS (
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISBASKIID INTEGER,
    HAMID INTEGER,
    HAMADI ADI255,
    ORMERAFID INTEGER,
    RAFADI ADI30,
    KAPADEDI INTEGER,
    RAFMIKTAR BOYAMIKTARSIZE,
    SARFMIKTARI BOYAMIKTARSIZE,
    TALIMATRECETEMIKTAR BOYAMIKTARSIZE,
    BOYATALIMATNO VARCHAR(30),
    LOT VARCHAR(150),
    FIRMAID FIRMAKODUSIZE,
    FIRMAADI VARCHAR(100),
    BOYATALIMATKAYITID INTEGER,
    BIRIMFIYAT DOUBLE PRECISION,
    RENKID INTEGER,
    RENKADI ADI50,
    MELANJKODU VARCHAR(20),
    RAFLOT VARCHAR(150),
    RAFMELANJKODU VARCHAR(20),
    STOKHAREKETID INTEGER)
AS
declare variable IrsaliyeNo            IrsaliyeNoSize;
declare variable SeriNo                varchar(2);
declare variable shtFasonUretimeCikis  integer;
declare variable shtFasonUretim        integer;
declare variable uitBoya               integer;

declare variable FUretimIslemTipiID    integer;
declare variable FHareketTipi          integer;
declare variable TahminiFireOrani      Double Precision;
declare variable Miktar                Numeric(12,3);

declare variable IrsaliyeTarihi        TimeStamp;
declare variable BirimId               varchar(2);
declare variable UrunRafId             integer;
declare variable FKayitSayisi          integer;
declare variable ToplamSarfMiktari     Numeric(12,3);
declare variable FRenkId               integer;
declare variable SarfIrsaliyeId        Integer;
declare variable ReceteVar smallint;
begin
  Select ShtFasonUretimeCikis, shtFasonUretim from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis, :shtFasonUretim;

  Select uitBoya from UretimIslemTipiGetir into :uitBoya;

  Select HareketTipi, UretimIslemTipiID, SarfIrsaliyeId, IrsaliyeTarihi from OrmeIrsaliye
    where OrmeIrsaliye.OrmeIrsaliyeId = :AIrsaliyeId
    into :FHareketTipi, :FUretimIslemTipiID, :SarfIrsaliyeId, :IrsaliyeTarihi;

  if (FHareketTipi <> shtFasonUretim or FUretimIslemTipiID <> uitBoya) then Exception KayitHatasi 'Boya Fason Üretimi Dışında Bu işlem Yapılamaz..';

  if (SarfIrsaliyeId is not null) then
  begin
    suspend;
    exit;
  end

  Select Count(*) from StokHareket
    left Join BoyaTalimatRecete on BoyaTalimatRecete.BoyaTalimatKayitID = StokHareket.BoyaTalimatKayitID
    where StokHareket.OrmeIrsaliyeId = :AIrsaliyeId and HamIade = 0 and BoyaTalimatRecete.BoyaTalimatReceteID is Null
    into :FKayitSayisi;

  if (FKayitSayisi > 0) then exception KayitHatasi ' Reçete Bilgileri Girilmemiş Satırlar Var. Üretime Çıkış İşlemi Yapılamaz...';


  For Select StokHareket.BoyaTalimatKayitID, StokHareket.BrutMiktar, BoyaTalimat.BoyaSiparisKayitId,
             BoyaTalimat.BoyaTalimatNo, BoyaTalimat.FirmaID, Firma.Adi, StokHareket.StokHareketId,
             BoyaTalimat.BoyaSiparisDetayId, BoyaTalimat.BoyaSiparisBaskiId, OrmeSiparisDetay.OrmeSiparisDetayId from StokHareket
    left join BoyaTalimat on BoyaTalimat.BoyaTalimatKayitId = StokHareket.BoyaTalimatKayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaTalimat.BoyaSiparisDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
    left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisBaskiId = BoyaTalimat.BoyaSiparisBaskiId
    left join Firma on Firma.FirmaId = BoyaTalimat.FirmaId
    where StokHareket.OrmeIrsaliyeId = :AIrsaliyeId and HamIade = 0
    into :BoyaTalimatKayitId, :Miktar, :BoyaSiparisKayitId, :BoyaTalimatNo, :FirmaID, :FirmaAdi, :StokHareketId,
         :BoyaSiparisDetayId, :BoyaSiparisBaskiId, :OrmeSiparisDetayId do
  begin
    ReceteVar = 0;
    ToplamSarfMiktari = 0;
    For Select OrmeRaf.OrmeRafID, OrmeRaf.HamId, OrmeRaf.Lot, OrmeRaf.MelanjKodu, Ham.Adi,
               OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari, BoyaTalimatRecete.Miktar - BoyaTalimatRecete.KullanilanMiktar from BoyaTalimatRecete
      left join OrmeRaf on OrmeRaf.OrmeRafID = BoyaTalimatRecete.OrmeRafID
      left join Ham on Ham.HamId = OrmeRaf.HamId
      where BoyaTalimatRecete.BoyaTalimatKayitID = :BoyaTalimatKayitID
      into :OrmeRafID, :HamId, :Lot, :MelanjKodu,  :HamAdi, :RafMiktar, :TalimatReceteMiktar do
    begin
      SarfMiktari = 0;
      if (TalimatReceteMiktar > RafMiktar) then TalimatReceteMiktar = RafMiktar;
      if (Miktar > TalimatReceteMiktar and Miktar > ToplamSarfMiktari) then SarfMiktari = TalimatReceteMiktar;
      else if (Miktar > ToplamSarfMiktari) then SarfMiktari = Miktar;

      ToplamSarfMiktari = ToplamSarfMiktari + SarfMiktari;

      suspend;

    end

  end
end
