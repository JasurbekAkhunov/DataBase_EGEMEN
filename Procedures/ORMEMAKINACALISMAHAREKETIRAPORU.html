CREATE PROCEDURE ORMEMAKINACALISMAHAREKETIRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AORMEMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4),
    ADURUSSURESI INTEGER)
RETURNS (
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINADURUMUID INTEGER,
    ORMEPLANLAMAID INTEGER,
    TARIH TIMESTAMP,
    TUR INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(10),
    SIPARISFIRMAADI VARCHAR(100),
    HAMADI VARCHAR(255),
    SONRAKITARIH TIMESTAMP,
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    GECENSURE DOUBLE PRECISION,
    GECENSUREDAKIKA INTEGER,
    GECENSURESTR VARCHAR(30),
    PLANLANANTURSAYISI INTEGER,
    BINTURDAORULECEKKG DOUBLE PRECISION,
    ARIZA INTEGER,
    SERVISCI INTEGER,
    AYAR INTEGER,
    MEMINGERKOPTU SMALLINT,
    ALTMEMINGERKOPTU SMALLINT,
    IGNEKOPTU SMALLINT,
    LYCRAKOPTU SMALLINT,
    KAPAKACILDI SMALLINT,
    STARTSTOP SMALLINT,
    TOPKESME SMALLINT,
    TEMIZLIKYAPILDI SMALLINT,
    TOPLAMISTENENMIKTAR DOUBLE PRECISION,
    TOPLAMGERCEKLESENMIKTAR DOUBLE PRECISION)
AS
declare variable MaxOrmeMakinaDurumuId integer;
/*declare variable LastTarih TimeStamp; */
declare variable FOrmeMakinaDurumuId        integer;
declare variable FOrmePlanlamaId        integer;
declare variable FTur                   integer;
begin
  if (ABaslangicTarihi > Current_date + Current_time) then exit;

  For Select OrmeMakina.OrmeMakinaId from OrmeMakina
     where (OrmeMakina.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null) and
           (OrmeMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null)
     into :OrmeMakinaID do
  begin
    SonrakiTarih = Null;

    For Select OrmeMakinaDurumu.OrmeMakinaDurumuId, OrmeMakinaDurumu.OrmePlanlamaId, OrmeMakinaDurumu.Tarih, OrmeMakinaDurumu.Tur, OrmeMakinaDurumu.PersonelId, Personel.Adi,
               BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, Ham.Adi,
               OrmePlanlama.PlanlananTurSayisi, OrmeMakinaDurumu.ArizaVar, OrmeMakinaDurumu.ServisciVar, OrmeMakinaDurumu.AyarYapiliyor, OrmeMakinaDurumu.MemingerKoptu, OrmeMakinaDurumu.AltMemingerKoptu,
               OrmeMakinaDurumu.IgneKoptu, OrmeMakinaDurumu.KapakAcildi, OrmeMakinaDurumu.StartStop, OrmeMakinaDurumu.TopKesme, OrmeMakinaDurumu.TemizlikYapildi, OrmePlanlama.BinTurdaOrulecekKg, OrmeMakinaDurumu.LycraKoptu from OrmeMakinaDurumu
       left join Personel on Personel.PersonelID = OrmeMakinaDurumu.PersonelId
      left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeMakinaDurumu.OrmePlanlamaId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
      left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
      left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
       where OrmeMakinaDurumu.OrmeMakinaId = :OrmeMakinaId and OrmeMakinaDurumu.Tarih >= :ABaslangicTarihi and OrmeMakinaDurumu.Tarih <= :ABitisTarihi
       order by tarih desc
    into :OrmeMakinaDurumuId, :OrmePlanlamaId, :Tarih, :Tur, :PersonelId, :PersonelAdi, :BoyaSiparisYil, :BoyaSiparisId,
         :SiparisFirmaId, :SiparisFirmaAdi, :HamAdi, :PlanlananTurSayisi, :Ariza, :Servisci, :Ayar, :MemingerKoptu, :AltMemingerKoptu,
         :IgneKoptu, :KapakAcildi, :StartStop, :TopKesme, :TemizlikYapildi, :BinTurdaOrulecekKg, :LycraKoptu do
    begin
      if (SonrakiTarih is null) then
      begin
        Select Min(Tarih) from OrmeMakinaDurumu
           where OrmeMakinaDurumu.OrmeMakinaId = :AOrmeMakinaId and OrmeMakinaDurumu.Tarih > :Tarih and OrmeMakinaDurumu.Tur <> :Tur
          into :SonrakiTarih;

        if (SonrakiTarih is null) then SonrakiTarih = Tarih;
      end

      GecenSure = SonrakiTarih - Tarih;

      if (GecenSure = 0) then GecenSureStr = '';
      else GecenSureStr = TarihToStr(GecenSure);

      GecenSureDakika = GecenSure * 1440;

      ToplamGerceklesenMiktar = 0;
      ToplamIstenenMiktar     = 0;
      if (GecenSure > 0) then
      begin
        ToplamGerceklesenMiktar = ((cast(Tur as Double Precision)/ 1000) * BinTurdaOrulecekKg);
        ToplamIstenenMiktar     = (cast(GecenSure * 1440 * PlanlananTurSayisi as Double Precision) / 1000) * BinTurdaOrulecekKg;
      end

      if (BoyaSiparisYil is not null) then
        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      else
        BoyaSiparisIdStr = '';

      if (ADurusSuresi = 0 or (Tur = 0 and GecenSureDakika >= ADurusSuresi)) then suspend;

      SonrakiTarih = Tarih;
    end
  end
end
