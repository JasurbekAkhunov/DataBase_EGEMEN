  left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
  Left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
  Left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId

  left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
  left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
  left join SiparisGrup SiparisGrup3 on SiparisGrup3.SiparisGrupId = BoyaSiparis.SiparisGrup3Id
  left join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
  left join IlaveProsesSebebi on IlaveProsesSebebi.IlaveProsesSebebiId = PartiAsamalari.IlaveProsesSebebiId
  left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
  left join BaskiTipi on BaskiTipi.BaskiTipiID = BoyaSiparisDetay.BaskiTipiID
  left join BoyaSiparisSarj on BoyaSiparisSarj.BoyaSiparisSarjId = PartiStok.BoyaSiparisSarjId
  where MakinaPlanlamaGrup.BoyahaneBolumId = :ABoyahaneBolumId and (Makina.MakinaId = :AMakinaId or :AMakinaId is null) and
        UretimPlanlama.UretimPlanlamaID is not null
  into :UretimPlanlamaId, :UretimPlanlamaDetayId, :BolumPlanlamaId, :GrupPlanlamaId, :PartiSinifId,
       :MakinaId, :SepetNo, :ManuelSure, :Sure, :PartiKayitId, :PartiAsamalariId, :PlanlamaTipi, :PartiYil, :PartiId, :AciliyetDerecesiId,
       :FirmaId, :FirmaAdi,  :FirmaKisaAdi, :ReceteId, :ReceteAdi, :EskiLabReceteKodu, :Sira, :DesenId, :TasarimNo, :DesenAdi, :VaryantAdi, :Denye,
       :AsamaId, :AsamaAdi, :BoyaAsamasimi, :TerminTarihi, :PartiAcilisTarihi, :IslemTipiId, :IslemTipiAdi, :IslemGrubuId, :IslemGrubuAdi,
       :BaslamaTarihi, :HamAdiToplami, :BrutKilo, :BrutMetre, :Kilo, :Metre,
       :BoyaSiparisYil, :BoyaSiparisId, :BoyaSiparisDetayId, :MusteriSiparisNo,
       :SiparisRenkAdi, :MusteriRenkKoduveAdi, :SiparisLot, :MusteriKaliteNo, :MusteriKaliteAdi,
       :OrnekPartiSinifId, :KaliteId, :KaliteAdi, :RezervNotu, :PlanlayanKullaniciAdi, :FPartiAsamalariId, :FPartiTamirSinifId,
       :PartiReceteId, :PartiReceteAdi, :PartiEskiLabReceteKodu, :PartiStokBekleID, :PartiStokBekleBitti, :TamirTipiadi,
       :SiparisProsesId, :SiparisProsesAdi, :PartiRptDegerlendirme, :PartiRptBaslamaTarihi, :PartiRptBitisTarihi, :PartiRecetesiYazdirildi,
       :PartiNotu, :SiparisNotu, :SiparisTerminNotu, :SiparisPlanlamaNotu, :RenkDerinligiStr, :LabReceteAtisColor, :InsertTarihi,
       :SiparisHamRenkAdi, :Fason, :SiparisMaximumFireOrani, :SiparisKilo,
       :EnIstenen, :GramajIstenen, :SiparisIlaveIslemler, :RefakatKartiYazildi, :SiparisKaliteId, :SiparisKaliteAdi,
       :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisBrutKilo, :IlaveProses, :IlaveProsesSebebiId, :IlaveProsesSebebiAdi, :ProxyID,
       :PartiSiraNo, :BelgeHamAdi, :GenelKontrolNotu, :BaskiNumuneGidisTarihi, :BaskiOkeyGelisTarihi, :Kazan, :BaskiTipiID, :BaskiTipiAdi, :HamAdi2, :HamAdi3, :HamAdi4,
       :FPlanlamaDurumu, :SablonDurumuId, :SablonDurumuAdi, :OzelYerlesimDosyaAdi, :TupAcikenMayli, :SiparisGrup3Id, :SiparisGrup3Adi, :SiparisGrup4Id, :SiparisGrup4Adi, :HamTipiId, :DesenVaryantResimDosyaAdi,
       :SablonSayisi, :DesenOlcuGravurOlcu, :DesenOlcuSablonEnOlcu, :DesenOlcuRaporBoyuOlcu, :SiparisSarjBobinSayisi,:siparissarjgramaj, :KapAdedi, FirmaAdi2, :YikamaReceteId do
  begin

    Select first 1 DozajlamaIsEmri.DozajlamaIsEmriId from PartiSablon
      left join DozajlamaIsEmri ON DozajlamaIsEmri.PARTISABLONISEMRIID = PartiSablon.PARTISABLONISEMRIID
      where PartiSablon.PartiKayitId = :PartiKayitId into :SablonDozajlamaIsEmriId;

    ReceteIsEmriYil = null;
    ReceteIsEmriId = null;

    select ReceteIsEmriYil, ReceteIsEmriId from ReceteIsEmriDetay where ReceteIsEmriDetay.PartiAsamalariId = :PartiAsamalariId
      into :ReceteIsEmriYil, :ReceteIsEmriId;

    if (ReceteIsEmriYil is null) then
      select DozajlamaIsEmriId, Gonderildi from DozajlamaIsEmri where DozajlamaIsEmri.PartiAsamalariId = :PartiAsamalariId into :DozajlamaIsEmriId, :DozajlamayaGonderildi;
    else
      select DozajlamaIsEmriId, Gonderildi from DozajlamaIsEmri where DozajlamaIsEmri.ReceteIsEmriYil = :ReceteIsEmriYil and DozajlamaIsEmri.ReceteIsEmriId = :ReceteIsEmriId into :DozajlamaIsEmriId, :DozajlamayaGonderildi;

    if (DozajlamayaGonderildi = 1) then
       DozajlamayaGonderildiStr = 'Gnderildi';
    else
       DozajlamayaGonderildiStr = '';

    if (FPlanlamaDurumu = 1) then
      PlanlamaDurumuStr = 'Rpt';
    else if (FPlanlamaDurumu = 2) then
      PlanlamaDurumuStr = 'Kuma Yok';
    else if (FPlanlamaDurumu = 3) then
      PlanlamaDurumuStr = 'Okey Bekliyor';
    else if (FPlanlamaDurumu = 4) then
      PlanlamaDurumuStr = 'Dier';
    else
      PlanlamaDurumuStr = '';

    if (:TupAcikenMayli is not null) then
      Select TupAcikEnMayliStrGetir.tupacikenmaylistr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;


    if (UretimPlanlamadaSureReceteden = 1 and Kazan = 1 and ReceteId is not Null) then
    begin
      Select Sum(coalesce(MakinaReceteAsamalari.Sure, 0)) from ReceteDetay
          left join MakinaReceteAsamalari on MakinaReceteAsamalari.MakinaID = :MakinaId and MakinaReceteAsamalari.ReceteAsamaId = ReceteDetay.ReceteAsamaId
          where ReceteDetay.ReceteID = :ReceteId and ReceteDetay.ReceteAsamaId is not null into :ReceteAsamaToplamSure;

      Sure = ReceteAsamaToplamSure;
    end
    else
      ReceteAsamaToplamSure = 0;

    HamStokMasterIDStr = '';
    GirisRafIDler = '';

    For Select HamStokMaster.HamStokMasterID, RafDetay.RafID from PartiStok
      left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiStok.PartiSinifID
      left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left Join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
      where PartiStok.PartiSinifId = :PartiSinifId
      Group By HamStokMaster.HamStokMasterID, RafDetay.RafID
      into :FHamStokMasterID, :FRafID do
    begin

      Str = EgemenIntegerToStr(FHamStokMasterID);
      if (Str is not null) then
      if (PosBul(HamStokMasterIDStr, Str) = 0) then
      if (Char_Length(HamStokMasterIDStr) + Char_Length(Str) < 97) then
      begin
        if (HamStokMasterIDStr <> '') then HamStokMasterIDStr = HamStokMasterIDStr || cast(', ' as varchar(2) character set UTF8);
        HamStokMasterIDStr = HamStokMasterIDStr || Str;
      end

      Str = FRafID;
      if (Str is not null) then
      if (PosBul(GirisRafIDler, Str) = 0) then
      if (Char_Length(GirisRafIDler) + Char_Length(Str) < 97) then
      begin
        if (GirisRafIDler <> '') then GirisRafIDler = GirisRafIDler || cast(', ' as varchar(2) character set UTF8);
        GirisRafIDler = GirisRafIDler || Str;
      end
    end

    UretimPlanlamaGostergesiStr = '';
    For Select Coalesce(Asama.UretimPlanlamaGostergesi, '') from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.Giristarihi is null and
            PartiAsamaStok.AsamayaDahil = 1 And PartiAsamalari.OtomasyonaDahilmi = 1  and Coalesce(Asama.UretimPlanlamaGostergesi, '') <> ''
      order by PartiAsamalari.Sira
      into :UretimPlanlamaGostergesi do
    begin
      if ((Char_Length(UretimPlanlamaGostergesiStr) + Char_Length(UretimPlanlamaGostergesi) < 250)  and (UretimPlanlamaGostergesi <> '')) then
      begin
        if (UretimPlanlamaGostergesiStr <> '' ) then UretimPlanlamaGostergesiStr = UretimPlanlamaGostergesiStr || cast('+' as varchar(2) character set UTF8);
        UretimPlanlamaGostergesiStr = UretimPlanlamaGostergesiStr || UretimPlanlamaGostergesi;
      end
    end

    if (BaslamaTarihi is not null) then Imalatta = 1;
    else Imalatta = 0;

    if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    else PartiIdStr = '';

    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
    else BoyaSiparisIdStr = '';

    if (OrnekPartiSinifId is not null and OrnekPartiSinifId = PartiSinifId ) then OrnekPartimi = 1;
    else OrnekPartimi = 0;

    if (FPartiTamirSinifId is not null) then IsletmeIciTamir = 1;
    else IsletmeIciTamir = 0;


    if (PartiRptDegerlendirme = 1) then PartiRptDegerlendirmeStr = 'allmayacak';
    else if (PartiRptDegerlendirme = 2) then PartiRptDegerlendirmeStr = 'allacak';
    else if (PartiRptDegerlendirme = 3) then PartiRptDegerlendirmeStr = 'allacak (Hammadde Bekleniyor';
    else PartiRptDegerlendirmeStr = '';

    if (PartiRptBitisTarihi is not null) then
    if (PartiRptDegerlendirme = 2 or PartiRptDegerlendirme = 3) then PartiRptDegerlendirmeStr = 'alld';

    if (FPartiAsamalariId is not null) then
    begin
      select Asama.Adi, PartiAsamalari.Sira from PartiAsamaStok
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId

        into :SonYapilanAsamaAdi, :SonYapilanSira;
    end
    else
    begin
      SonYapilanAsamaAdi = Null;
      SonYapilanSira = -1;
    end

    ReceteIsEmriIdStr = Null;
    BirlestirilenPartiler = '';

    if (ReceteId is not null) then
      Select ReceteIsEmriIdStr, BirlestirilenPartiler from PAReceteIsEmriBilgisiGetir(:PartikayitId, :ReceteId, :PartiAsamalariId) into :ReceteIsEmriIdStr, :BirlestirilenPartiler;

    select Min(PartiAsamalari.Sira) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.Sira > :SonYapilanSira
      into :SonrakiSira;

    begin
      select Asama.Adi, PartiAsamalari.PartiAsamalariId from PartiAsamaStok
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :SonrakiSira
        into :SonrakiAsamaAdi, :SiradakiPartiAsamalariId;
    end
    else
    begin
      SonrakiAsamaAdi = Null;
      SiradakiPartiAsamalariId = Null;
    end

    if (SiradakiPartiAsamalariId is null) then ImalataHazir = 0;
    else if (SiradakiPartiAsamalariId = PartiAsamalariId) then ImalataHazir = 1;

      Select Min(PartiStokBekleId) from PartiStokBekle

      else if (BlobSize > 0) then PartiStokBekleNotu = Left(FBlob, BlobSize);
      else PartiStokBekleNotu = '';

        Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId

        into :YikamaSonrasiEn, :YikamaSonrasiGramaj;

        Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId

       if (Lot <> '' ) then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
       Lot = cast(Lot as blob character set utf8) || FLot;
      end
    end

    AsamaIlaveIslemler = '';
    for Select IlaveIslemler.Adi from PartiAsamaStok
     left join PartiIlaveIslemler on PartiIlaveIslemler.PartiKayitId = PartiAsamaStok.PartiKayitId and PartiIlaveIslemler.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
     left join IlaveIslemler on IlaveIslemler.IlaveIslemlerId = PartiIlaveIslemler.IlaveIslemlerId

     into :FIlaveIslemlerAdi do

      begin
       if (AsamaIlaveIslemler <> '' ) then AsamaIlaveIslemler = AsamaIlaveIslemler || cast(', ' as varchar(2) character set UTF8);
       AsamaIlaveIslemler = cast(AsamaIlaveIslemler as blob character set utf8) || FIlaveIslemlerAdi;

    For Select Coalesce(AsamaKisaAdi, ''), GirisTarihi from PartiStok
      left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = partistok.PartiSinifId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiStok.PartiSinifId = :PartiSinifId and
            PartiAsamaStok.AsamayaDahil = 1 And PartiAsamalari.OtomasyonaDahilmi = 1
      order by PartiAsamalari.Sira
      into :AsamaKisaAdi, :FGirisTarihi do
    begin
      if (AsamaKisaAdi is not null) then
      begin
        if (FGirisTarihi is Null) then
        begin
          if (Char_Length(PartiStokAsamalari) + Char_Length(AsamaKisaAdi) < 250) then
          begin
            if (PartiStokAsamalari <> '' ) then PartiStokAsamalari = PartiStokAsamalari || cast(', ' as varchar(2) character set UTF8);
            PartiStokAsamalari = cast(PartiStokAsamalari as blob character set utf8) || AsamaKisaAdi;
          end
        end
        if (Char_Length(PartiStokTumAsamalari) + Char_Length(AsamaKisaAdi) < 250) then
        begin
          if (PartiStokTumAsamalari <> '' ) then PartiStokTumAsamalari = PartiStokTumAsamalari || cast(', ' as varchar(2) character set UTF8);
          PartiStokTumAsamalari = cast(PartiStokTumAsamalari as blob character set utf8) || AsamaKisaAdi;
        end
      end
    end

    CikisRafID = Null;
    Select CikisRafID from PartiSinifSonCikisRafGetir(:PartiKayitId, :PartiSinifId) into :CikisRafId;

    VaryantTasarimFullProses = 0;

    if (IslemTipiID = itDesenTasarim) then
    begin
      IlkSira = Null;

      select Min(PartiAsamalari.Sira) from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1
        into :IlkSira;

      if (IlkSira is not null) then
      begin
        select Asama.Partilememi from PartiAsamaStok
          left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
          left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
          where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :IlkSira
          into :FPartilememi;
        if (FPartilememi = 1) then VaryantTasarimFullProses = 1;
      end
    end

    Select GecenSureStr from SonAsamadanGecenSureyiHesapla(:PartiSinifId) into :GecenSure;

    suspend;
  end
