  begin
    if (FAltIskontoCarpani is null) then FAltIskontoCarpani = 0;
    if (FFKdvYuvarlamaCarpani is not null) then FKdvYuvarlamaCarpani = FFKdvYuvarlamaCarpani; /* Faturanin carpanini kullansin */
    if (FFNetFiyatYuvarlamaCarpani is not null) then FNetFiyatYuvarlamaCarpani = FFNetFiyatYuvarlamaCarpani; /* Faturanin carpanini kullansin */
    if (FFBrutToplamYuvarlamaCarpani is not null) then FBrutToplamYuvarlamaCarpani = FFBrutToplamYuvarlamaCarpani; /* Faturanin carpanini kullansin */

    if (OrmeSiparisDetayID is null and BoyaSiparisDetayId is not null) then
    begin
      MusteriSiparisNo = FMusteriSiparisNo;
      ReferansSiparisNo = FReferansSiparisNo;
    end

    KirpintiTasnifStr = null;
    if (KirpintiTasnifYil is not null) then
      KirpintiTasnifStr = '(' || EgemenIntegerToStr(KirpintiTasnifYil) || ') '  || EgemenIntegerToStr(KirpintiTasnifId)  || ' ';

    if (Tevkifat = 1) then TevkifatStr = 'Var';
    else TevkifatStr = '';

    if (LotKapali = 1) then LotKapaliStr = 'Evet';
    else LotKapaliStr = '';

    if (BoyaSiparisYil is not null) then
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId)  || ' ';
    else
      BoyaSiparisIdStr = '';

    if (ElyafUretimIsEmriYil is not null) then
      ElyafUretimIsEmriIdStr = '(' || EgemenIntegerToStr(ElyafUretimIsEmriYil) || ') '  || EgemenIntegerToStr(ElyafUretimIsEmriId)  || ' ';
    else
      ElyafUretimIsEmriIdStr = '';

    if (IplikTalepYil is not null) then
      IplikTalepIdStr = '(' || EgemenIntegerToStr(IplikTalepYil) || ') '  || EgemenIntegerToStr(IplikTalepId)  || ' ';
    else
      IplikTalepIdStr = '';

    if (HarmanYil is not null) then
      HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') '  || EgemenIntegerToStr(HarmanId)  || ' ';
    else
      HarmanIdStr = '';

    if (TamirGirisYil is not null) then
      TamirGirisIdStr = '(' || EgemenIntegerToStr(TamirGirisYil) || ') '  || EgemenIntegerToStr(TamirGirisId)  || ' ';
    else
      TamirGirisIdStr = '';


    BirimIskontoToplami = BirimIskontoToplami * Miktar;
    BrutTutar = KdvHaricFiyat * Miktar;

    if (FAltIskontoCarpani is null) then FAltIskontoCarpani = 0; /* faturadan gelecek simdilik iplikfatura yok */

    BrutTutar = Yuvarla(BrutTutar, FBrutToplamYuvarlamaCarpani);

    T1 = NetBirimFiyat - NetBirimFiyat * FAltIskontoCarpani;
    T1 = Yuvarla(T1, FNetFiyatYuvarlamaCarpani);

    Tutar = T1 * Miktar;

    IndirilecekKdv = 0;
    TahsilEdilecekKdv = 0;


    if (Tevkifat = 1) then
    begin
      TahsilEdilecekKdv = (T1 * Miktar * Kdv / 100) * (TevkifatOrani2 - TevkifatOrani1) / TevkifatOrani2;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, :FKdvYuvarlamaCarpani);

      IndirilecekKdv = (T1 * Miktar * Kdv / 100) * TevkifatOrani1 / TevkifatOrani2;
      IndirilecekKdv = Yuvarla(IndirilecekKdv, :FKdvYuvarlamaCarpani);
    end
    else
    begin
      TahsilEdilecekKdv = T1 * Miktar * Kdv / 100;
      TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, :FKdvYuvarlamaCarpani);
    end

    if (TamirGirisKayitId is not null) then
    begin
      Tamir = 1;
      TamirStr = 'Evet';
    end
    else
    begin
      Tamir = 0;
      TamirStr = '';
    end

    SiparisOdemeSekliId  = Null;
    SiparisOdemeSekliAdi = Null;
    SiparisBirimFiyat    = Null;
    SiparisDovizId       = Null;

    if (BoyaSiparisDetayID is not null) then
    begin
      Select BoyaSiparisDetay.BirimFiyat, BoyaSiparisDetay.DovizId, BoyaSiparisDetay.OdemeSekliId, OdemeSekli.Adi from BoyaSiparisDetay
        left join OdemeSekli on OdemeSekli.OdemeSekliID = BoyaSiparisDetay.OdemeSekliId
        where BoyaSiparisDetay.BoyaSiparisDetayID = :BoyaSiparisDetayID
        into :SiparisBirimFiyat, :SiparisDovizId, :SiparisOdemeSekliId, :SiparisOdemeSekliAdi;
    end
    else if (OrmeSiparisDetayID is not null) then
    begin
      Select OrmeSiparisDetay.SatisFiyati, OrmeSiparisDetay.DovizId, OrmeSiparisDetay.OdemeSekliId, OdemeSekli.Adi from OrmeSiparisDetay
        left join OdemeSekli on OdemeSekli.OdemeSekliID = OrmeSiparisDetay.OdemeSekliId
        where OrmeSiparisDetay.OrmeSiparisDetayID = :OrmeSiparisDetayID
        into :SiparisBirimFiyat, :SiparisDovizId, :SiparisOdemeSekliId, :SiparisOdemeSekliAdi;
    end

    suspend;
  end
end
