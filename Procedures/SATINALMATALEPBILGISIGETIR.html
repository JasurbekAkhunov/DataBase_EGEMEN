CREATE PROCEDURE SATINALMATALEPBILGISIGETIR(
    ASATINALMATALEPYIL INTEGER,
    ASATINALMATALEPID INTEGER)
RETURNS (
    SATINALMATALEPKAYITID INTEGER,
    SATINALMATALEPYIL INTEGER,
    SATINALMATALEPID INTEGER,
    SATINALMATALEPIDSTR VARCHAR(30),
    TALEPKULLANICIID INTEGER,
    TALEPKULLANICIADI VARCHAR(40),
    TALEPTARIHI TIMESTAMP,
    DEPARTMANID INTEGER,
    ISTEKYERI INTEGER,
    DEPARTMANADI VARCHAR(30),
    SATINALMATALEPYERIID INTEGER,
    SATINALMATALEPYERIADI VARCHAR(30),
    VERILENSIPARISTIPIID INTEGER,
    VERILENSIPARISTIPIIDSTR VARCHAR(15),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
begin
  For Select SatinAlmaTalep.SatinAlmaTalepKayitId, SatinAlmaTalep.SatinAlmaTalepYil, SatinAlmaTalep.SatinAlmaTalepId,
         SatinAlmaTalep.TalepKullaniciId, TEK.Adi, SatinAlmaTalep.Tarih, SatinAlmaTalep.DepartmanId, Departman.Adi,
         SatinAlmaTalep.SatinAlmaTalepYeriId, SatinAlmaTalepYeri.Adi, SatinAlmaTalep.Notu, SatinAlmaTalep.VerilenSiparisTipiId,
         VerilenSiparisTipi.Adi, SatinAlmaTalep.OnayDurumu, SatinAlmaTalep.OnayTarihi, SatinAlmaTalep.IstekYeri from SatinAlmaTalep
    left join Kullanici TEK on TEK.KullaniciId = SatinAlmaTalep.TalepKullaniciId
    left join Departman on Departman.DepartmanId = SatinAlmaTalep.DepartmanId
    left join SatinAlmaTalepYeri on SatinAlmaTalepYeri.SatinAlmaTalepYeriId = SatinAlmaTalep.SatinAlmaTalepYeriId
    left join VerilenSiparisTipi on VerilenSiparisTipi.VerilenSiparisTipiId = SatinAlmaTalep.VerilenSiparisTipiId
    where SatinAlmaTalep.SatinAlmaTalepYil = :ASatinAlmaTalepYil and SatinAlmaTalep.SatinAlmaTalepId = :ASatinAlmaTalepId
    into :SatinAlmaTalepKayitId, :SatinAlmaTalepYil, :SatinAlmaTalepId, :TalepKullaniciId, :TalepKullaniciAdi, :TalepTarihi, :DepartmanId, :DepartmanAdi,
         :SatinAlmaTalepYeriId, :SatinAlmaTalepYeriAdi, :Notu, :VerilenSiparisTipiId, :VerilenSiparisTipiIdStr, :OnayDurumu,
         :OnayTarihi, :IstekYeri do
  begin
    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    SatinAlmaTalepIdStr = '';
    if (SatinAlmaTalepId is not null) then SatinAlmaTalepIdStr = '(' || EgemenIntegerToStr(SatinAlmaTalepYil) || ') '  || EgemenIntegerToStr(SatinAlmaTalepId);
    suspend;
  end
end
