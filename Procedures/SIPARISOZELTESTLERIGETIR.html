CREATE PROCEDURE SIPARISOZELTESTLERIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    SIPARISOZELTESTLERID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    OZELTESTID INTEGER,
    OZELTESTADI ADI80,
    OZELTESTSTANDARDIID INTEGER,
    OZELTESTSTANDARDIADI VARCHAR(40),
    ISTENEN VARCHAR(10),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select SiparisOzelTestler.SiparisOzelTestlerId, SiparisOzelTestler.BoyaSiparisDetayId, SiparisOzelTestler.OzelTestId, OzelTest.Adi,
             SiparisOzelTestler.OzelTestStandardiID, OzelTestStandardi.Adi, SiparisOzelTestler.Istenen,
             IK.ADI, SiparisOzelTestler.Inserttarihi, SK.ADI, SiparisOzelTestler.DegisimTarihi  FROM SiparisOzelTestler
  LEFT JOIN OzelTest ON OzelTest.OzelTestId = SiparisOzelTestler.OzelTestId
  LEFT JOIN OzelTestStandardi ON OzelTestStandardi.OzelTestStandardiId = SiparisOzelTestler.OzelTestStandardiId
  Left Join Kullanici IK on IK.KullaniciId = SiparisOzelTestler.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = SiparisOzelTestler.KullaniciId
  WHERE SiparisOzelTestler.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  into :SiparisOzelTestlerId, :BoyaSiparisDetayId, :OzelTestId, :OzelTestAdi, :OzelTestStandardiID, :OzelTestStandardiAdi, :Istenen,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do suspend;
end
