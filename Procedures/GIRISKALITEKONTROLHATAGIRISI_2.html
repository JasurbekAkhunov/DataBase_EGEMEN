CREATE PROCEDURE GIRISKALITEKONTROLHATAGIRISI_2(
    AGIRISKALITEKONTROLHATAMETREID INTEGER,
    AGIRISKALITEKONTROLID INTEGER,
    AHATATIPIID INTEGER,
    AMETRE DOUBLE PRECISION,
    ABITIS INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    GIRISKALITEKONTROLHATAMETREID INTEGER)
AS
begin
  if (ABitis = 0) then
  begin
    if (AGirisKaliteKontrolHataMetreId is not null) then Update GirisKaliteKontrolHataMetre Set BaslangicMetre = :AMetre where GirisKaliteKontrolHataMetre.GirisKaliteKontrolHataMetreID = :AGirisKaliteKontrolHataMetreID;
    else
    begin
      GirisKaliteKontrolHataMetreID = Gen_Id(GKaliteKontrolHataMetre_Id_Gen,1);
      insert into GirisKaliteKontrolHataMetre(GirisKaliteKontrolHataMetreID, GirisKaliteKontrolid, HataTipiId, BaslangicMetre, InsertKullaniciId, KullaniciId) values(:GirisKaliteKontrolHataMetreID, :AGirisKaliteKontrolId, :AHataTipiId, :AMetre, :AKullaniciId, :AKullaniciId);
    end
  end
  else Update GirisKaliteKontrolHataMetre Set BitisMetre = :AMetre where GirisKaliteKontrolHataMetre.GirisKaliteKontrolHataMetreID = :AGirisKaliteKontrolHataMetreID;
  suspend;
end
