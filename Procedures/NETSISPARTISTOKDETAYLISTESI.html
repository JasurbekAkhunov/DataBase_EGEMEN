CREATE PROCEDURE NETSISPARTISTOKDETAYLISTESI(
    APARTIYIL INTEGER,
    APARTIID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    ORMEPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    KILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION)
AS
begin
  For Select PartiStok.PartiSinifId, OrmeRaf.OrmePlanlamaId, OrmeRaf.UretimPartiNo, PartiStokDetay.Kilo, PartiStokDetay.Metre from Parti
    left join PartiStok on PartiStok.PartiKayitID = Parti.PartiKayitId
    left join PartiStokDetay on PartiStokDetay.PartiSinifID = PartiStok.PartiSinifID
    left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
    left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayID
    left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterID
    left join StokHareket on StokHareket.StokHareketID = HamStokMaster.StokHareketID
    left join OrmeRaf on OrmeRaf.OrmeRafID = StokHareket.OrmeRafID
    where Parti.PartiYil = :APartiYil and Parti.PartiId = :APartiId and PartiStok.PartiSinifID is not null and PartiStok.PartiStokIptalKayitID is null
    into :PartiSinifId, :OrmePlanlamaId, :UretimPartiNo, :Kilo, :Metre do
  begin
    suspend;
  end
end
