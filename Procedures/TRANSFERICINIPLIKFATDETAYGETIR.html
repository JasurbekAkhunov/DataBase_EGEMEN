CREATE PROCEDURE TRANSFERICINIPLIKFATDETAYGETIR(
    AIPLIKFATURAID INTEGER)
RETURNS (
    BOYAFATURAID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMTIPIID INTEGER,
    MIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    FATURABIRIM INTEGER,
    DOVIZID VARCHAR(5),
    KDVORANI DOUBLE PRECISION,
    TRANSFERKODU VARCHAR(55),
    BIRIMTRANSFERKODU VARCHAR(10),
    ODEMESEKLITRANSFERKODU VARCHAR(10),
    BIRIMFIYATTL DOUBLE PRECISION,
    TUTARTL DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    VADEGUN INTEGER,
    TEVKIFAT INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    MUHGELIRHESAPKODU VARCHAR(30),
    KDVMUHASEBEHESAPKODU VARCHAR(30),
    BOYASIPARISNOSTR VARCHAR(30),
    IPLIKTALEPNOSTR VARCHAR(30),
    PARTINOSTR VARCHAR(30),
    SIPARISTARIHI TIMESTAMP,
    SIPARISIALANKULLANICIID INTEGER,
    MUSTERISIPARISNO VARCHAR(50))
AS
declare variable FaturaTarihi       TimeStamp;
declare variable DovizKuru          Double Precision;
declare variable SistemDovizId      varchar(5);
declare variable FHamSatisTransferKodu varchar(55);
declare variable HamStokMuhHamSatisHesapKodu varchar(30);
declare variable FKdvYuvarlamaCarpani Double PRecision;

declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef        integer;
declare variable SiparisIslemGrubuId    integer;
declare variable FOrmeOdemeSekliTransferKodu  varchar(10);
declare variable FBoyaOdemeSekliTransferKodu  varchar(10);
declare variable RenkId integer;
declare variable FirmaId varchar(20);
declare variable EgemenFirmaId integer;
declare variable FBoyaSiparisYil Integer;
declare variable FBoyaSiparisID Integer;
declare variable FIplikTalepYil Integer;
declare variable FIplikTalepID Integer;
begin
/* gülle icin yazıldı */
  BoyaFaturaId = null;
  FaturaBirim = 0;

  Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

  Select SistemDovizId, EgemenFirmaId from Sistem where sistemId = 1 into :SistemDovizId, :EgemenFirmaId;

  for select IplikIrsaliyeDetay.HamId, Ham.Adi, IplikIrsaliyeDetay.RenkId, Ham.HamTipiId, IplikIrsaliyeDetay.FaturaMiktar,
             IplikIrsaliyeDetay.KdvHaricFiyat, IplikFatura.FirmaId, IplikFatura.DovizId, IplikIrsaliyeDetay.Kdv,
             iif(IplikIrsaliyeDetay.HareketTipi = 0, Ham.AlisTransferKodu, Ham.HamSatisTransferKodu),
             IplikFatura.FaturaTarihi, IplikIrsaliyeDetay.VadeGun,
             Ham.MuhHamSatisHesapKodu, IplikFatura.KdvYuvarlamaCarpani,
             IplikTalepDetay.OrmeSiparisDetayId, IplikTalepDetay.BoyaSiparisDetayId,
             OrmeOdemeSekli.TransferKodu, BoyaOdemeSekli.TransferKodu,
             IplikIrsaliye.SeriNo, IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.FiiliSevkTarihi,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, BoyaSiparis.SiparisTarihi, IplikTalep.IplikTalepYil, IplikTalep.IplikTalepID,
             BoyaSiparis.SiparisiAlanKullaniciId, IplikIrsaliyeDetay.Tevkifat, Ham.Adi2, BoyaSiparisDetay.MusteriSiparisNo from IplikIrsaliyeDetay
    left join IplikFatura on IplikFatura.IplikFaturaId = IplikIrsaliyeDetay.IplikFaturaId
    left join IplikTalepDetay on IplikTalepDetay.IplikTalepDetayId = IplikIrsaliyeDetay.IplikTalepDetayId
    left join IplikTalep on IplikTalep.IplikTalepKayitId = IplikTalepDetay.IplikTalepkayitId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = IplikTalepDetay.BoyaSiparisDetayId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = IplikTalepDetay.OrmeSiparisDetayId
    Left Join Ham on Ham.HamId = IplikIrsaliyeDetay.HamId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = IplikTalepDetay.BoyaSiparisKayitId
    Left Join OdemeSekli OrmeOdemeSekli on OrmeOdemeSekli.OdemeSekliId = OrmeSiparisDetay.OdemeSekliId
    Left Join OdemeSekli BoyaOdemeSekli on BoyaOdemeSekli.OdemeSekliId = BoyaSiparisDetay.OdemeSekliId
    left join IplikIrsaliye on IplikIrsaliye.IplikIrsaliyeId = IplikIrsaliyeDetay.IplikIrsaliyeId
    where IplikIrsaliyeDetay.IplikFaturaId = :AIplikFaturaId
    order by IplikIrsaliyeDetayId
    into :HamId, :HamAdi, :RenkId, :HamTipiId, :Miktar, :BirimFiyat, :FirmaId, :DovizId, :KdvOrani,
         :FHamSatisTransferKodu, :FaturaTarihi, :VadeGun, :HamStokMuhHamSatisHesapKodu,
         :FKdvYuvarlamaCarpani, :OrmeSiparisDetayId, :BoyaSiparisDetayId,
         :FOrmeOdemeSekliTransferKodu, :FBoyaOdemeSekliTransferKodu, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :FiiliSevkTarihi,
         :FBoyaSiparisYil, :FBoyaSiparisID, :SiparisTarihi, :FIplikTalepYil, :FIplikTalepID, :SiparisiAlanKullaniciId, :Tevkifat, :HamAdi2, :MusteriSiparisNo  do
  begin
    if (FBoyaSiparisYil is not null) then BoyaSiparisNoStr = EgemenCopyStr(EgemenIntegerToStr(FBoyaSiparisYil),3,2) || '-' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
    else BoyaSiparisNoStr = '';

    if (FIplikTalepID is not null) then IplikTalepNoStr = EgemenCopyStr(EgemenIntegerToStr(FIplikTalepYil),3,2) || '-' || EgemenIntegerToStr(FIplikTalepID) || ' ';
    else IplikTalepNoStr = '';

    if (DovizID <> SistemDovizId) then
    begin
      select DovizKuru from FirmaDovizKurlariniGetir(:FirmaId, :DovizId, :SistemDovizId, :FaturaTarihi) into :DovizKuru;
    end
    else DovizKuru = 1;

    BirimFiyatTL = BirimFiyat * DovizKuru;
    TutarTL = BirimFiyatTL * Miktar;

    TransferKodu = Null;
    OdemeSekliTransferKodu = Null;

    if (RenkId is not null) then SiparisIslemGrubuId = 1; /* depodan renk satıs yapılıyor ise renkli islem grubuna dahil olsun */

    if (RenkId is not null) then
    begin
      TransferKodu = Null;
      MuhGelirHesapKodu = Null;

      Select SatisIplikTransferKodu, SatisIplikMuhGelirHesapKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :SiparisIslemGrubuId into :TransferKodu, :MuhGelirHesapKodu;

      OdemeSekliTransferKodu = FBoyaOdemeSekliTransferKodu;
    end
    else
    begin
      OdemeSekliTransferKodu = FOrmeOdemeSekliTransferKodu;

      TransferKodu = FHamSatisTransferKodu;
      MuhGelirHesapKodu = HamStokMuhHamSatisHesapKodu;
    end

    BirimTransferKodu = '';

    Select TransferKodu from Birim where Birim.BirimID = 'Kg' into :BirimTransferKodu;

    KdvMuhasebeHesapKodu = Null;

    Select MuhSatisHesapKodu from KdvHesapTablosu where KdvHesapTablosu.KdvOrani = :KdvOrani into :KdvMuhasebeHesapKodu;

    IndirilecekKdv = 0;
    TahsilEdilecekKdv = 0;

    TahsilEdilecekKdv = TutarTL * KdvOrani / 100;
    TahsilEdilecekKdv = Yuvarla(TahsilEdilecekKdv, FKdvYuvarlamaCarpani);

    suspend;
  end
end
