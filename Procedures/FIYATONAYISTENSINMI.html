CREATE PROCEDURE FIYATONAYISTENSINMI(
    APARTISINIFID INTEGER,
    ABOYASIPARISDETAYID INTEGER,
    ABIRIMFIYAT DOUBLE PRECISION,
    ADOVIZID VARCHAR(5))
RETURNS (
    ONAYISTENSIN INTEGER)
AS
declare variable KayitSayisi         integer;
declare variable IlaveIslemlerId     integer;
declare variable IIS                 integer;
declare variable FiyatOnayKontrolTipi integer;
begin
  OnayIstensin = 0;

  Select FiyatOnayKontrolTipi from Sistem where sistemid= 1 into :FiyatOnayKontrolTipi;

  if (FiyatOnayKontrolTipi = 0) then
  begin
    OnayIstensin = 0;
    suspend;
    exit;
  end
  else if (FiyatOnayKontrolTipi = 2) then
  begin
    OnayIstensin = 1;
    suspend;
    exit;
  end
  else if (ABirimFiyat = 0) then
  begin
    OnayIstensin = 1;
    suspend;
    exit;
  end

  Select Count(*) from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiSinifId = :APartiSinifId and
          (BoyaSiparisDetay.BirimFiyat <> :ABirimFiyat or BoyaSiparisDetay.DovizId <> :ADovizId)
    into :KayitSayisi;

  if (KayitSayisi > 0) then
  begin
    OnayIstensin = 1;
    suspend;
    exit;
  end

  For Select PartiIlaveIslemler.IlaveIslemlerId, Count(*) from PartiAsamaStok
    left join PartiIlaveIslemler on PartiIlaveIslemler.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    where PartiAsamaStok.PartiSinifId = :APartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiIlaveIslemler.IlaveIslemlerId is not null
    group by PartiIlaveIslemler.IlaveIslemlerId
    into :IlaveIslemlerId, :IIS do
  begin

    Select count(*) from SiparisIlaveIslemler
      where SiparisIlaveIslemler.BoyaSiparisDetayId = :ABoyaSiparisDetayId and SiparisIlaveIslemler.IlaveIslemlerId = :IlaveIslemlerId
    into :KayitSayisi;

    if (KayitSayisi <> IIS) then
    begin
      OnayIstensin = 1;
      suspend;
      exit;
    end
  end

  suspend;
end
