CREATE PROCEDURE LABRECETEATISKOPYALA(
    ALABRECETEATISID INTEGER,
    AKULLANICIID INTEGER)
AS
declare variable LabReceteAtisId integer;
declare variable FLabReceteAtisRenkId integer;
declare variable LabReceteAtisRenkId integer;
declare variable KayitSayisi integer;
begin
  LabReceteAtisId = gen_id(LabReceteAtis_Id_Gen, 1);

  insert into LabReceteAtis(LabReceteCalismaId, LabReceteId, LabReceteAtisId, Migrasyon, Color, Notu, NotuStr, RenkDerinligiId, ReceteAsamaId, GidisTarihi, InsertKullaniciId, KullaniciId)
      Select LabReceteCalismaId, LabReceteId, :LabReceteAtisId, Migrasyon, Color, Notu, NotuStr, RenkDerinligiId, ReceteAsamaId, Current_date, :AKullaniciId, :AKullaniciId from LabReceteAtis
        where LabReceteAtis.LabReceteAtisID = :ALabReceteAtisId;

  Select Count(*) from LabReceteAtisRenk where LabReceteAtisRenk.LabReceteAtisID = :ALabReceteAtisId into :KayitSayisi;
  if (KayitSayisi = 0) then
  begin
    Insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, InsertKullaniciId, KullaniciId)
        Select :LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, :AKullaniciId, :AKullaniciId from LabReceteUrun
          where LabReceteUrun.LabReceteAtisID = :ALabReceteAtisId
          order by LabReceteUrun.LabReceteUrunId;
  end
  else
  begin
    for Select LabReceteAtisRenkID from LabReceteAtisRenk where LabReceteAtisRenk.LabReceteAtisID = :ALabReceteAtisId into :FLabReceteAtisRenkId do
    begin
      LabReceteAtisRenkID = gen_id(LabReceteAtisRenk_Id_Gen, 1);

      insert into LabReceteAtisRenk(LabReceteAtisRenkId, LabReceteId, LabReceteAtisId, RenkKodu, RenkAdi, Color, Sira, InsertKullaniciId, KullaniciId)
          Select :LabReceteAtisRenkId, LabReceteId, :LabReceteAtisId, RenkKodu, RenkAdi, Color, Sira, :AKullaniciId, :AKullaniciId from LabReceteAtisRenk
            where LabReceteAtisRenk.LabReceteAtisRenkID = :FLabReceteAtisRenkId
            order by LabReceteAtisRenk.Sira;

      Insert into LabReceteUrun(LabReceteAtisId, LabReceteId, LabReceteAtisRenkId, Sira, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, InsertKullaniciId, KullaniciId)
          Select :LabReceteAtisId, LabReceteId, :LabReceteAtisRenkId, Sira, UrunId, MiktarYuzde, UrunBirimId, BirimFiyat, FiyatUrunBirimId, FiyatCarpan, Notu, :AKullaniciId, :AKullaniciId from LabReceteUrun
            where LabReceteUrun.LabReceteAtisRenkID = :FLabReceteAtisRenkId
            order by LabReceteUrun.LabReceteUrunId;


    end
  end
end
