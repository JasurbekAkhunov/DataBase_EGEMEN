CREATE PROCEDURE BOYACIKISICINSIPHAMSTOKLISTESI(
    ABOYASIPARISKAYITID INTEGER,
    ADOKUMAHAREKETI EGEBOOLEAN)
RETURNS (
    RESULTID VARCHAR(2550),
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    BOYASIPARISDETAYRENKADI VARCHAR(40),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    SIPARISKILO NUMERIC(12,3),
    STOKTANKULLANILACAKKILO NUMERIC(12,3),
    TAHMINIBRUTSIPARISKILO NUMERIC(12,3),
    TOPLAMBOYAYACIKIS NUMERIC(12,3),
    BASKASIPARISEKAYDIRILAN NUMERIC(12,3),
    BASKASIPARISTENKAYDIRILAN NUMERIC(12,3),
    MUSTERISIPARISNO VARCHAR(50),
    ARTIKELNO VARCHAR(20),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    RAFIDSTR VARCHAR(40),
    TOPLAMORULEN DOUBLE PRECISION,
    TOPLAMSAKATORULEN DOUBLE PRECISION,
    TOPLAMDOKUNAN DOUBLE PRECISION,
    TOPLAMSAKATDOKUNAN DOUBLE PRECISION,
    TOPLAMSATIS DOUBLE PRECISION,
    TOPLAMBOYALISATIS DOUBLE PRECISION,
    TOPLAMSATISIADE DOUBLE PRECISION,
    ORGANIK SMALLINT,
    ORGANIKSTR VARCHAR(8),
    TERMINTARIHI TIMESTAMP,
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10))
AS
declare variable FHamId integer;
declare variable FRenkId integer;
declare variable MaximumFireOrani Double Precision;
declare variable FRafId           varchar(10);
declare variable BoyaUCikistaSipMiktarAsabilir SmallInt;
declare variable SiparisBrutKilo double precision;
declare variable EgemenFirmaId integer;


begin
  Select BoyaUCikistaSipMiktarAsabilir, Sistem.EgemenFirmaId from Sistem where Sistem.SistemID = 1 into :BoyaUCikistaSipMiktarAsabilir, :EgemenFirmaId;

  for Select OrmeSiparisDetay.OrmeSiparisDetayID, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.BoyaSiparisDetayId, OrmeSiparisDetay.HamId, BoyaSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.RenkId, Renk.Adi,
             BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.RenkAdi, BoyaSiparisDetay.MusteriRenkKoduveAdi,
             BoyaSiparisDetay.SiparisKilo, BoyaSiparisDetay.StoktanKullanilacakKilo, BoyaSiparisDetay.ToplamBoyayaCikis - BoyaSiparisDetay.ToplamBoyaIade,
             BoyaSiparisDetay.BaskaSipariseKaydirilan, BoyaSiparisDetay.BaskaSiparistenKaydirilan,
             OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MusteriSiparisNo,
             BoyaSiparisDetay.IslemTipiId, IslemTipi.Adi, BoyaSiparisDetay.MaximumFireOrani, BoyaSiparisDetay.MusteriKaliteNo,
             BoyaSiparisDetay.MusteriKaliteAdi, BoyaSiparisDetay.MelanjKodu, OrmeSiparisDetay.ToplamOrulen, OrmeSiparisDetay.ToplamSakatOrulen, OrmeSiparisDetay.ToplamDokunan, OrmeSiparisDetay.ToplamSakatDokunan,
             OrmeSiparisDetay.ToplamSatis, OrmeSiparisDetay.ToplamBoyaliSatis, OrmeSiparisDetay.ToplamSatisIade, OrmeSiparisDetay.Organik, BoyaSiparisDetay.SiparisBrutKilo, BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen from OrmeSiparisDetay
  left join BoyaSiparisDetay on BoyaSiparisDetay.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
  left join IslemTipi on IslemTipi.IslemTipiId = BoyaSiparisDetay.IslemTipiId
  where OrmeSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisDetay.Boyanacak = 1 and
        (BoyaSiparisDetay.SiparisKilo > BoyaSiparisDetay.StoktanKullanilacakKilo or BoyaSiparisDetay.SiparisMetre > BoyaSiparisDetay.StoktanKullanilacakMetre)
  order by OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.SiparisSiraNo, BoyaSiparisDetay.BoyaSiparisDetayId
  into :OrmeSiparisDetayID, :SiparisSiraNo, :BoyaSiparisDetayId, :FHamId, :HamId, :HamAdi, :FRenkId, :RenkAdi, :LabReceteKodu, :LabReceteAdi, :BoyaSiparisDetayRenkAdi,
       :MusteriRenkKoduveAdi, :SiparisKilo, :StoktanKullanilacakKilo, :ToplamBoyayaCikis, :BaskaSipariseKaydirilan, :BaskaSiparistenKaydirilan,
       :ArtikelNo, :MusteriSiparisNo, :IslemTipiId, :IslemTipiAdi, :MaximumFireOrani, :MusteriKaliteNo, :MusteriKaliteAdi, :MelanjKodu,
       :ToplamOrulen, :ToplamSakatOrulen, :ToplamDokunan, :ToplamSakatDokunan, :ToplamSatis, :ToplamBoyaliSatis, :ToplamSatisIade, :Organik, :SiparisBrutKilo, :TerminTarihi, :EnIstenen, :GramajIstenen  do
  begin
    if (HamId <> FHamID) then FRenkID = Null;


    OrganikStr = '';
    if (Organik = 1) then OrganikStr = 'Organik';

    ResultId = '"' || EgemenIntegerToStr(OrmeSiparisDetayID) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(SiparisSiraNo) || '",' ;
    ResultId = ResultId ||'"' || EgemenIntegerToStr(BoyaSiparisDetayId) || '",' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",' ;

    if (FRenkId is not null) then
      ResultId = ResultId || '"' || EgemenIntegerToStr(FRenkId) || '",' ;
    else
      ResultId = ResultId || '"",' ;

    if (BoyaSiparisDetayRenkAdi is null) then BoyaSiparisDetayRenkAdi = '';
    ResultId = ResultId || '"' || BoyaSiparisDetayRenkAdi || '", ' ;

    ResultId = ResultId || '"' || EgemenIntegerToStr(ISLEMTIPIID) || '", ' ;
    ResultId = ResultId || '"' || EgemenIntegerToStr(Organik) || '"' ;

    if (EgemenFirmaId = 9) then
      TahminiBrutSiparisKilo = SiparisBrutKilo + SiparisBrutKilo * MaximumFireOrani / 100;
    else
    TahminiBrutSiparisKilo = SiparisKilo + SiparisKilo * MaximumFireOrani / 100;
    TahminiBrutSiparisKilo = Yuvarla(TahminiBrutSiparisKilo, 100);


    RafIdStr = '';

    if (TahminiBrutSiparisKilo > (ToplamBoyayaCikis + BaskaSiparistenKaydirilan - BaskaSipariseKaydirilan) or BoyaUCikistaSipMiktarAsabilir = 1) then
    begin
      for Select RafId from BoyayaCikisIcinStokListesi(:OrmeSiparisDetayId, :BoyaSiparisDetayId, :HamId, :FRenkId, Null, :ADokumaHareketi)
        group by RafId
        into :FRafId do
      begin
        if (Char_Length(RafIdStr) + Char_Length(FRafId) < 38) then
        begin
          if (RafIdStr <> '' ) then RafIdStr = RafIdStr || cast(', ' as varchar(2) character set UTF8);
          RafIdStr = RafIdStr || FRafId;
        end
      end
      suspend;
    end
  end
end
