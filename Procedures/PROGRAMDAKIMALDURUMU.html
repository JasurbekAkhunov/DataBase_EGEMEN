CREATE PROCEDURE PROGRAMDAKIMALDURUMU(
    APARTITIPIID INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    TEMSILCIID INTEGER,
    TEMSILCIADI VARCHAR(40),
    BEYAZ NUMERIC(12,3),
    RENKLI NUMERIC(12,3),
    YIKAMA NUMERIC(12,3),
    SEKERKASAR NUMERIC(12,3),
    CIFTBOYAMA NUMERIC(12,3),
    SIYAH NUMERIC(12,3),
    NAYLONBOYAMA NUMERIC(12,3),
    DIGER NUMERIC(12,3),
    HAMKILOTOPLAMI NUMERIC(12,3),
    TAMIRKILOTOPLAMI NUMERIC(12,3))
AS
declare variable IslemTipi      integer;
declare variable IslemGrubu     integer;
declare variable Kilo           Numeric(12,3);
declare variable SonFirmaId     varchar(20);

declare variable FFirmaId     varchar(20);
declare variable FFirmaAdi    varchar(100);
declare variable FTemsilciId  integer;
declare variable FTemsilciAdi varchar(40);

declare variable itNormal       integer;
declare variable itRpt          integer;
declare variable itKontrol      integer;
declare variable itTamir        integer;
declare variable itNumune       integer;

begin
  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  SonFirmaId = '';

  for select Firma.FirmaId, Firma.Adi, Firma.TemsilciId, Kullanici.Adi, Parti.IslemTipiId, Parti.IslemGrubuId, sum(PartiStok.Kilo) from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId= Parti.FirmaId
    left join Kullanici on Kullanici.KullaniciId = Firma.TemsilciId
    where PartiStok.Durumu <= 2 and Parti.ReceteId <> '' and Parti.IslemgrubuId <> 4 and
         (Parti.PartiTipiId = :APartiTipiId or :APartiTipiId is null) and
         PartiStok.PartiStokIptalKayitID is Null
    group by Firma.FirmaId, Firma.Adi, Firma.TemsilciId, Kullanici.Adi, Parti.IslemTipiId, Parti.IslemGrubuId
    into :FFirmaId, :FFirmaAdi, :FTemsilciId, :FTemsilciAdi, :IslemTipi, :IslemGrubu, :Kilo do
  begin
    if (SonFirmaId <> FFirmaId) then
    begin
      if (SonFirmaId <> '') then suspend;
      Beyaz = 0;
      Renkli = 0;
      Yikama = 0;
      SekerKasar = 0;
      CiftBoyama  = 0;
      Siyah  = 0;
      NaylonBoyama  = 0;
      Diger = 0;

      HamKiloToplami = 0;
      TamirKiloToplami = 0;
      SonFirmaId = FFirmaId;

      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      TemsilciId = FTemsilciId;
      TemsilciAdi = FTemsilciAdi;

    end
    if (IslemGrubu = 0) then Beyaz = Beyaz + Kilo;
    else if (IslemGrubu = 1) then Renkli = Renkli + Kilo;
    else if (IslemGrubu = 2) then Yikama = Yikama + Kilo;
    else if (IslemGrubu = 3) then SekerKasar = SekerKasar + Kilo;
    else if (IslemGrubu = 5) then CiftBoyama = CiftBoyama + Kilo;
    else if (IslemGrubu = 6) then Siyah = Siyah + Kilo;
    else if (IslemGrubu = 8) then NaylonBoyama = NaylonBoyama + Kilo;
    else Diger = Diger + Kilo;

    if (IslemTipi = itTamir) then TamirKiloToplami = TamirKiloToplami + Kilo;
    else HamKiloToplami = HamKiloToplami + Kilo;
  end
  if (SonFirmaId <> '') then suspend;
end
