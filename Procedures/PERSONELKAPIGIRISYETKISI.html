CREATE PROCEDURE PERSONELKAPIGIRISYETKISI(
    APERSONELID INTEGER,
    ADEPARTMANID INTEGER)
RETURNS (
    KAPINO INTEGER,
    KAPIGIRIS EGEBOOLEAN)
AS
begin
  Select DepartmanKapi.KapiGiris, Departman.KapiNo from DepartmanKapi
  left join Departman on Departman.DepartmanID = DepartmanKapi.DepartmanID
  where DepartmanKapi.DepartmanId = :ADepartmanId and DepartmanKapi.PersonelId = :APersonelId
  into :KapiGiris, :KapiNo;
  suspend;
end
