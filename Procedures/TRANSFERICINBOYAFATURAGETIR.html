CREATE PROCEDURE TRANSFERICINBOYAFATURAGETIR(
    ABOYAFATURAID INTEGER)
RETURNS (
    FATURASERINO SERINOSIZE5,
    FATURANO FATURANOSIZE,
    FATURATARIHI TIMESTAMP,
    ODEMETARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    DOVIZID VARCHAR(5),
    FIRMADOVIZKURU1 DOUBLE PRECISION,
    FIRMADOVIZKURU2 DOUBLE PRECISION,
    TRANSFERKODU VARCHAR(55),
    ATRANSFERKODU VARCHAR(55),
    MUHASEBEKODU VARCHAR(20),
    BRUTTOPLAM DOUBLE PRECISION,
    ISKONTOTOPLAMI DOUBLE PRECISION,
    PARASALISKONTO DOUBLE PRECISION,
    ISKONTOORANI DOUBLE PRECISION,
    TAHSILEDILECEKKDV DOUBLE PRECISION,
    INDIRILECEKKDV DOUBLE PRECISION,
    ARATOPLAM DOUBLE PRECISION,
    GENELTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZKURU1 DOUBLE PRECISION,
    ISLEMDOVIZKURU2 DOUBLE PRECISION,
    ISLEMDOVIZTOPLAM DOUBLE PRECISION,
    ISLEMDOVIZTAHSILEDILECEKKDV DOUBLE PRECISION,
    FATURAGRUPTRANSFERKODU VARCHAR(10),
    TEVKIFAT INTEGER,
    TEVKIFATORANI DOUBLE PRECISION,
    SINIF VARCHAR(1),
    IRSALIYENO VARCHAR(255),
    EFATURA INTEGER,
    NOTU VARCHAR(1024),
    TESLIMSEKLITRANSFERKODU VARCHAR(55),
    INSERTTARIHI TIMESTAMP)
AS
declare variable MinDovizID varchar(5);
declare variable MaxDovizID varchar(5);
declare variable MinDovizKuru1 Double Precision;
declare variable MinDovizKuru2 Double Precision;
declare variable MaxDovizKuru1 Double Precision;
declare variable MaxDovizKuru2 Double Precision;
declare variable AltIskYuvarlamaCarpani Double Precision;
declare variable FIrsaliyeNo varchar(20);
declare variable FMusteriSiparisNo varchar(50);
declare variable FNotu Blob ;
declare variable BlobSize integer;
declare variable IlkSiparis SmallInt;

begin
  For Select BoyaFatura.FaturaTarihi, BoyaFatura.OdemeTarihi, BoyaFatura.FaturaSeriNo, BoyaFatura.FaturaNo, BoyaFatura.FirmaId,
         Coalesce(DovizliFirmaTransfer.TransferKodu,Firma.TransferKodu), Firma.ATransferKodu, Firma.MuhasebeKodu, BoyaFatura.DovizId, BoyaFatura.DovizKuru1, BoyaFatura.DovizKuru2,
         BoyaFatura.BrutToplam, BoyaFatura.ParasalIskonto, BoyaFatura.IskontoOrani, BoyaFatura.IskontoToplami,
         BoyaFatura.TahsilEdilecekKdv, BoyaFatura.IndirilecekKdv, AltIskYuvarlamaCarpani, BoyaFatura.IslemDovizID, BoyaFatura.IslemDovizKuru1, BoyaFatura.IslemDovizKuru2,
         IslemDovizToplam, IslemDovizTahsilEdilecekKdv, StokHareketGrup.TransferKodu, BoyaFatura.Tevkifat, BoyaFatura.Sinif,
         (BoyaFatura.BrutToplam + BoyaFatura.TahsilEdilecekKDV - BoyaFatura.IskontoToplami) GenelToplam,
         (BoyaFatura.BrutToplam + BoyaFatura.TahsilEdilecekKDV - BoyaFatura.IskontoToplami + BoyaFatura.IndirilecekKdv) AraToplam, (cast(BoyaFatura.TevkifatOrani1 as Double Precision) / BoyaFatura.TevkifatOrani2) TevkifatOrani,
         BoyaFatura.EFatura, BoyaFatura.Notu, TeslimSekli.TransferKodu, BoyaFatura.InsertTarihi from BoyaFatura
  left join Firma On Firma.FirmaID = BoyaFatura.FirmaId
  left join DovizliFirmaTransfer On DovizliFirmaTransfer.FirmaID = BoyaFatura.FirmaId and DovizliFirmaTransfer.DovizID = BoyaFatura.IslemDovizID
  left join StokHareketGrup On StokHareketGrup.StokHareketGrupID = BoyaFatura.FaturaGrupId
  left join TeslimSekli on TeslimSekli.TeslimSekliId = BoyaFatura.TeslimSekliId
  where BoyaFatura.BoyaFaturaID = :ABoyaFaturaId
  into :FaturaTarihi, :OdemeTarihi, :FaturaSeriNo, :FaturaNo, :FirmaId, TransferKodu, :ATransferKodu, :MuhasebeKodu, :DovizId,
       :FirmaDovizKuru1, :FirmaDovizKuru2, :BrutToplam,
       :ParasalIskonto, :IskontoOrani, :IskontoToplami, :TahsilEdilecekKdv, :IndirilecekKdv, :AltIskYuvarlamaCarpani, :IslemDovizID,
       :IslemDovizKuru1, :IslemDovizKuru2, :IslemDovizToplam, :IslemDovizTahsilEdilecekKdv, :FaturaGrupTransferKodu,
       :Tevkifat, :Sinif, :GenelToplam, :AraToplam, :TevkifatOrani, :EFatura, :FNotu, :TeslimSekliTransferKodu, :InsertTarihi do
  begin
    IrsaliyeNo = '';
    For Select BoyaIrsaliye.IrsaliyeSeriNo || BoyaIrsaliye.IrsaliyeNo from BoyaIrsaliyeDetay
      Left Join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
      where BoyaIrsaliyeDetay.BoyaFaturaID = :ABoyaFaturaId
      Group By BoyaIrsaliye.IrsaliyeSeriNo,BoyaIrsaliye.IrsaliyeNo
    into :FIrsaliyeNo do
    begin
      if (Char_Length(IrsaliyeNo) + Char_Length(FIrsaliyeNo) < 240) then
      begin
        if (IrsaliyeNo <> '' ) then IrsaliyeNo = IrsaliyeNo || cast(', ' as varchar(2) character set UTF8);
        IrsaliyeNo = IrsaliyeNo || FIrsaliyeNo;
      end
    end

    BlobSize = char_length(FNotu);

    if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
    else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
    else Notu = '';

    IlkSiparis = 1;

    For Select BoyaSiparisDetay.MusteriSiparisNo from BoyaIrsaliyeDetay
      left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
      Left Join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      where BoyaIrsaliyeDetay.BoyaFaturaID = :ABoyaFaturaId
      Group By BoyaSiparisDetay.MusteriSiparisNo
    into :FMusteriSiparisNo do
    begin
      if (Char_Length(Notu) + Char_Length(FMusteriSiparisNo) < 970) then
      begin
        if (IlkSiparis = 1) then
        begin
          Notu = Notu || ' Sipler: ';
          IlkSiparis = 0;
        end
        else if (Notu <> '' ) then Notu = Notu || cast(', ' as varchar(2) character set UTF8);
        Notu = Notu || FMusteriSiparisNo;
      end
    end

    suspend;
  end
end
