       :MakinaPlanlamaGrupId, :MakinaPlanlamaGrupRaporSiraNo, :MakinaId, :PaketlemeSekliTup, :FFaturaKesilecek, :IlaveProses, :IlaveProsesSebebiId, :IlaveProsesSebebiAdi,

       :KaliteId, :KaliteAdi, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi, :SiparisBrutKilo,
       :YikamaSonrasiEn, :YikamaSonrasiGramaj, :ProxyId, :PartiSiraNo, :UretimPlanlamaID, :GenelKontrolNotu, :HamAdi2, :HamAdi3, :HamAdi4, :SiparisProsesID, :SiparisProsesAdi, :FPartiStokBekleNotu,
       :BaskiTipiId, :BaskiTipiAdi, :SiparisIslemGrubuId, :SiparisIslemGrubuAdi do
  begin
    if (PartiStokBekleNotu is not null) then
    begin
      BlobSize = char_length(FPartiStokBekleNotu);

      if (BlobSize > 1023) then PartiStokBekleNotu = Left(FPartiStokBekleNotu, 1023);
      else if (BlobSize > 0) then PartiStokBekleNotu = Left(FPartiStokBekleNotu, BlobSize);
      else PartiStokBekleNotu = '';
    end

    if (MakinaPlanlamaGrupId is null) then
    begin
      MakinaPlanlamaGrupId = ''; /* dizayn icin yaptk */
      MakinaPlanlamaGrupRaporSiraNo = 9999;
    end

    CikisRafID = Null;
    Select CikisRafID from PartiSinifSonCikisRafGetir(:PartiKayitId, :PartiSinifId) into :CikisRafId;

    Select Notu, TerminNotu, PlanlamaNotu from BoyaSiparisDetay where BoyaSiparisDetay.BoyaSiparisDetayID = :BoyaSiparisDetayId into :SiparisNotu, :SiparisTerminNotu, :SiparisPlanlamaNotu;

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);
    else BoyaSiparisIdStr = '';

    PartiRptDegerlendirmeDurumu = 1;

    if (FDegerlendirme = 1) then
      PartiRptDegerlendirme = 'allmayacak';
    else if (FDegerlendirme = 2) then
    begin
      PartiRptDegerlendirme = 'allacak';
      PartiRptDegerlendirmeDurumu = 0;
    end
    else if (FDegerlendirme = 3) then
    begin
      PartiRptDegerlendirme = 'allacak (Hammadde Bekleniyor';
      PartiRptDegerlendirmeDurumu = 0;
    end
    else PartiRptDegerlendirme = '';

    if (PartiRptBitisTarihi is not null) then
    begin
      PartiRptDegerlendirmeDurumu = 1;
      if (FDegerlendirme = 2 or FDegerlendirme = 3) then PartiRptDegerlendirme = 'alld';
    end

    if (PartiOnayDurumu = 1) then PartiOnayDurumuStr = 'Evet';
    else PartiOnayDurumuStr = '';

    if (IlaveProses = 1) then
      IlaveProsesStr = 'Evet';
    else
      IlaveProsesStr = '';

    if (HamTipiID = hstIplik or HamTipiID = hstElyaf or HamTipiID = hstTelef) then RaporHamTipiId = 0;
    else if (HamTipiID = hstOrmeKumas or HamTipiID = hstRibana or HamTipiID = hstAksesuar) then RaporHamTipiId = 1;
    else if (HamTipiID = hstDokumaKumas) then RaporHamTipiId = 2;
    else RaporHamTipiId = -1;


    Select Sum(PartiStok.SonCikisKilo), Sum(PartiStok.SonCikisMetre) from PartiStok where PartiStok.PartiKayitID = :PartiKayitID and PartiStok.PartiStokIptalKayitId is null
      into :PartiKilo, :PartiMetre;

    if (FPartiAsamalariId is not null) then
    begin
      select Asama.Adi, PartiAsamalari.Sira from PartiAsamaStok
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiAsamalariId = :FPartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId
        into :SonYapilanAsamaAdi, :SonYapilanSira;
    end
    else
    begin
      SonYapilanSira = -1;
      SonYapilanAsamaAdi = Null;
    end

    select PlanlamaSepetNoGetir.Planlamasepetno from PlanlamaSepetNoGetir(:PartiKayitId,:PartiAsamalariID, :PartiSinifId) into :MakinaSepetNo;


    select Min(PartiAsamalari.Sira) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
      where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.Sira > :SonYapilanSira
      into :SonrakiSira;

    if (SonrakiSira is not null) then
    begin
      select Asama.Adi, PartiAsamalari.PartiAsamalariId from PartiAsamaStok
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.Sira = :SonrakiSira
        into :SonrakiAsamaAdi, :SiradakiPartiAsamalariId;
    end
    else
    begin
      SonrakiAsamaAdi = Null;
      SiradakiPartiAsamalariId = Null;
    end

    if (SiradakiPartiAsamalariId is null) then ImalataHazir = 0;
    else if (SiradakiPartiAsamalariId = PartiAsamalariId) then ImalataHazir = 1;
    else ImalataHazir = 0;

    if (FPartiTamirSinifId is not null) then IsletmeIciTamir = 1;
    else IsletmeIciTamir = 0;

    if (PartiStokBekleId is null and ImalataHazir = 0) then
    begin
      Select Min(PartiStokBekleId) from PartiStokBekle
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiStokBekle.PartiAsamalariId
        where PartiStokBekle.PartiSinifId = :PartiSinifId and PartiStokBekle.BitirenKullaniciId is null and
              PartiAsamalari.Sira < :Sira
        into :PartiStokBekleId;

      if (PartiStokBekleId is not null) then PartiStokBekleBitti = 0;
    end

    ReceteIsEmriIdStr = Null;
    BirlestirilenPartiler = '';

    if (ReceteId is not null) then
      Select ReceteIsEmriIdStr, BirlestirilenPartiler from PAReceteIsEmriBilgisiGetir(:PartikayitId, :ReceteId, :PartiAsamalariId) into :ReceteIsEmriIdStr, :BirlestirilenPartiler;


    OnFikseAsamasiVar = 0;
    OnFikseAsamasiYapildi = 0;

    Select count(*) from PartiAsamaStok
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      where PartiAsamaStok.PartiSinifID = :PartiSinifId and Asama.OnFikseAsamasimi = 1 and PartiAsamaStok.AsamayaDahil = 1
      into :KS;

    if (KS = 0) then OnFikseAsamasiVar = 0;
    else
    begin
      OnFikseAsamasiVar = 1;

      Select count(*) from PartiAsamaStok
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        where PartiAsamaStok.PartiSinifID = :PartiSinifId and Asama.OnFikseAsamasimi = 1 and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamaStok.GirisTarihi is not null
        into :KS;

      if (KS > 0) then OnFikseAsamasiYapildi = 1;
      else OnFikseAsamasiYapildi = 0;

    end

    UcretliTamir = 0;
    UcretliTamirStr = '';

    if (IslemTipiID = 3 and FFaturaKesilecek = 0) then
      UcretliTamirStr = 'cretsiz Tamir';
    else if (IslemTipiID = 3 and FFaturaKesilecek = 1) then
    begin
      UcretliTamirStr = 'cretli Tamir';
      UcretliTamir = 1;
    end

    Lot = '';
    for Select HamStokMaster.Lot from PartiStokDetay
        Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.Lot is not null and HamStokMaster.Lot <> ''
        group by HamStokMaster.Lot
        into :FLot do
    begin
      if (Char_Length(Lot) + Char_Length(FLot) < 253) then
      begin
       if (Lot <> '' ) then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
       Lot = Lot || FLot;
      end
    end

    Select GecenSureStr from SonAsamadanGecenSureyiHesapla(:PartiSinifId) into :GecenSure;

    suspend;
  end
end
