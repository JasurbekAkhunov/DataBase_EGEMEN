CREATE PROCEDURE BOYAYARIMAMULRAFLA(
    APARTIASAMASTOKDETAYID INTEGER)
AS
declare variable HamStokIrsaliyeId integer;
declare variable FirmaKodumuz       FirmaKoduSize;
declare variable SistemDovizId      DovizSize;
declare variable FKabulMiktar       BoyaMiktarSize;
declare variable FCikisRafId        RafKoduSize;
declare variable FPartiSinifId      integer;
declare variable HamStokMasterId    integer;
declare variable Aciklama           varchar(30);
declare variable ToplamKilo         BoyaMiktarSize;
declare variable ToplamMetre        BoyaMiktarSize;
declare variable FFisSeriNo         varchar(2);
declare variable FFisNo             integer;
declare variable FFason             EgeBoolean;
declare variable FFirmaID           FirmaKoduSize;
begin
  select CikisKilo, CikisMetre, CikisRafId, PartiAsamaStokDetay.PartiSinifId from PartiAsamaStokDetay
    where PartiAsamaStokDetay.PartiAsamaStokDetayId = :APartiAsamaStokDetayId
  into :ToplamKilo, :ToplamMetre, :FCikisRafId, :FPartiSinifId;

  if (FCikisRafId is not Null and (ToplamKilo > 0 or ToplamMetre > 0)) then
  begin
    Aciklama = 'Otomatik Yarı Mamul Girişi';

    Select FirmaKodumuz, SistemDovizId from sistem where sistemid = 1 into :FirmaKodumuz, :SistemDovizId;

    select Parti.Fason, Parti.FirmaId from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    where PartiStok.PartiSinifId = :FPartiSinifId into :FFason, :FFirmaId;

    if (FFason = 0) then
      FFirmaId = FirmaKodumuz;

    /*HamStokIrsaliye belirle*/
    select Max(HamStokMaster.HamStokIrsaliyeId) from HamStokMaster
    where HamStokMaster.Tarih = current_date and HamStokMaster.PartiSinifId is not null and
    HamStokMaster.FirmaId = :FFirmaID and HamStokMaster.Fason = :FFason into :HamStokIrsaliyeId;

    if (HamStokIrsaliyeId is null) then
    begin
      HamStokIrsaliyeId = Gen_ID(HamStokIrsaliye_Id_Gen, 1);

      Select FisSeriNo from Kullanici where KullaniciId = 1 into :FFisSeriNo;
      if (FFisSeriNo is null) then FFisSeriNo = '';

      select FisNo from FisNumaratorAl(:FFisSeriNo) into :FFisNo;
      update FisNumarator set FisNo = :FFisNo where SeriNo = :FFisSeriNo;

      insert into HamStokIrsaliye(HamStokIrsaliyeId, HamStokHareketTipiId, FirmaId, EvrakTipi, IrsaliyeSeriNo, IrsaliyeNo, IrsaliyeTarihi, FiiliSevkTarihi, DovizId, Sinif, Notu, Fason)
      values(:HamStokIrsaliyeId, 0, :FFirmaId, 0, :FFisSeriNo, :FFisNo, current_date, current_date, :SistemDovizId, 'B', :Aciklama, :FFason);
    end

    /*HamStokMaster ekle*/
    HamStokMasterId = gen_Id(HamStokMaster_Id_gen, 1);

    insert into HamStokMaster(HamStokIrsaliyeId, HamStokMasterId, StokFirmaId, HamId, Kilo, Metre, RenkId, Renk, Aciklama, KaliteId, PartiSinifId, PartiAsamaStokDetayId)
    Select :HamStokIrsaliyeId, :HamStokMasterId, Null, PartiStok.HamId, :ToplamKilo, :ToplamMetre, null, null, :Aciklama, BoyaSiparisDetay.KaliteId,
    :FPartiSinifId, :APartiAsamaStokDetayId from PartiStok
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    where PartiStok.PartiSinifId = :FPartiSinifId;

    /*Gecici kabul rafi olustur*/
    Insert into GeciciKabulRaf(HamStokMasterId, RafId, KapAdedi, Kilo, Metre) Values (:HamStokMasterId, :FCikisRafId, 0, :ToplamKilo, :ToplamMetre);

    /*Ham stok girisi onayla. Bu satir otomatik olarak hamstokdetay ve rafdetay kaydi olusturur*/
    select HamStokMaster.Miktar from HamStokMaster where HamStokMaster.HamStokMasterId = :HamStokMasterId into :FKabulMiktar;
    update HamStokMasterKabul set KabulMiktar = :FKabulMiktar, KontrolDurumu = 1,  KontrolKullaniciId = 1 where HamStokMasterId = :HamStokMasterId;
  end
end
