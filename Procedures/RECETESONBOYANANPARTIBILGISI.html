CREATE PROCEDURE RECETESONBOYANANPARTIBILGISI(
    ARECETEID VARCHAR(20))
RETURNS (
    SONBOYANANPARTIYIL INTEGER,
    SONBOYANANPARTIID INTEGER,
    SONBOYANANPARTIIDSTR VARCHAR(30),
    SONBOYANANPARTIHAMADI VARCHAR(255),
    SONBOYANANPARTILOT VARCHAR(255))
AS
declare variable PartiKayitID integer;
begin
  Select first 1 PartiAsamalari.PartiKayitId from PartiAsamalari
    left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariID = PartiAsamalari.PartiAsamalariId
    where PartiAsamalari.ReceteID = :AReceteId and PartiAsamaStok.CikisTarihi is not null and
          PartiAsamalari.PartininAsilRecetesimi = 1
    order by PartiAsamaStok.CikisTarihi desc
    into :PartiKayitID;

  if (PartiKayitID is not null) then
  begin
    Select PartiYil, PartiID from Parti where Parti.PartiKayitID = :PartiKayitId into :SonBoyananPartiYil, :SonBoyananPartiID;
    SonBoyananPartiIdStr = '(' || EgemenIntegerToStr(SonBoyananPartiYil)  || ') ' || EgemenIntegerToStr(SonBoyananPartiId) || ' ';

    Select Lot, BuyukSinifHamAdiToplami from PartiLotBilgisi(:PartiKayitId) into :SonBoyananPartiLot, :SonBoyananPartiHamAdi;

  end
  suspend;
end
