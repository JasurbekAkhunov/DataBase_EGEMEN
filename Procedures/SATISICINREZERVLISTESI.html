CREATE PROCEDURE SATISICINREZERVLISTESI(
    AFIRMAID FIRMAKODUSIZE)
RETURNS (
    BOYAREZERVFISIID INTEGER,
    FISTARIHI TIMESTAMP,
    FISNO INTEGER,
    NOTU MEMO,
    BOYAREZERVFISIDETAYID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTIIDSTR VARCHAR(30),
    HAMID INTEGER,
    HAMADI ADI255,
    RENKID INTEGER,
    RENKADI ADI50,
    EDILECEKKILO BOYAMIKTARSIZE,
    EDILECEKMIKTAR BOYAMIKTARSIZE,
    EDILENKILO BOYAMIKTARSIZE,
    EDILENMIKTAR BOYAMIKTARSIZE,
    EDILECEKBRUTKILO BOYAMIKTARSIZE,
    EDILECEKBRUTMIKTAR BOYAMIKTARSIZE,
    EDILENBRUTKILO BOYAMIKTARSIZE,
    EDILENBRUTMIKTAR BOYAMIKTARSIZE,
    INSERTKULLANICIID INTEGER,
    INSERTKULLANICIADI ADI40,
    INSERTTARIHI TIMESTAMP,
    KULLANICIID INTEGER,
    KULLANICIADI ADI40,
    DEGISIMTARIHI TIMESTAMP,
    KALITEID KALITEKODUSIZE,
    KALITEADI ADI255)
AS
declare variable PartiYil Integer;
declare variable PartiId  Integer;
begin
    for select BoyaRezervFisi.BoyaRezervFisiId, BoyaRezervFisi.FisTarihi, BoyaRezervFisi.FisNo, BoyaRezervFisi.Notu,
        BoyaRezervFisiDetay.BoyaRezervFisiDetayId, BoyaRezervFisiDetay.BoyaSiparisDetayId, BoyaRezervFisiDetay.PartiKayitId, BoyaRezervFisiDetay.PartiSinifId, BoyaRezervFisiDetay.HamId,
        Ham.Adi, BoyaRezervFisiDetay.EdilecekKilo, BoyaRezervFisiDetay.EdilecekMiktar, BoyaRezervFisiDetay.EdilenKilo, BoyaRezervFisiDetay.EdilenMiktar,
        BoyaRezervFisiDetay.EdilecekBrutKilo, BoyaRezervFisiDetay.EdilecekBrutMiktar, BoyaRezervFisiDetay.EdilenBrutKilo, BoyaRezervFisiDetay.EdilenBrutMiktar,
        BoyaRezervFisiDetay.InsertKullaniciId, BoyaRezervFisiDetay.InsertTarihi, BoyaRezervFisiDetay.KullaniciId, BoyaRezervFisiDetay.DegisimTarihi,
        IK.Adi InsertKullaniciAdi, Kullanici.Adi KullaniciAdi, Parti.PartiYil, Parti.PartiId, Renk.RenkId, Renk.Adi, Kalite.KaliteId, Kalite.KaliteAdi from BoyaRezervFisi
        join BoyaRezervFisiDetay on BoyaRezervFisiDetay.BoyaRezervFisiId = BoyaRezervFisi.BoyaRezervFisiId
        left join Parti on Parti.PartiKayitId = BoyaRezervFisiDetay.PartiKayitId
        left join PartiStok on PartiStok.PartiSinifId = BoyaRezervFisiDetay.PartiSinifId
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
        left join Renk on BoyaSiparisDetay.RenkId = Renk.RenkId
        left join Ham on Ham.HamId = BoyaRezervFisiDetay.HamId
        left join Kullanici IK on IK.KullaniciId = BoyaRezervFisiDetay.InsertKullaniciId
        left join Kullanici on Kullanici.KullaniciId = BoyaRezervFisiDetay.KullaniciId
        left join Kalite on BoyaSiparisDetay.KaliteId = Kalite.KaliteId
        join PartiStokTarti on PartiStokTarti.BoyaRezervFisiDetayId = BoyaRezervFisiDetay.BoyaRezervFisiDetayId
        where BoyaRezervFisi.FirmaId = :AFirmaId and PartiStokTarti.BoyaIrsaliyeDetayId is null
        group by BoyaRezervFisi.BoyaRezervFisiId, BoyaRezervFisi.FisTarihi, BoyaRezervFisi.FisNo, BoyaRezervFisi.Notu,
        BoyaRezervFisiDetay.BoyaRezervFisiDetayId, BoyaRezervFisiDetay.BoyaSiparisDetayId, BoyaRezervFisiDetay.PartiKayitId, Parti.PartiYil, Parti.PartiId, BoyaRezervFisiDetay.PartiSinifId, BoyaRezervFisiDetay.HamId,
        Ham.Adi, BoyaRezervFisiDetay.EdilecekKilo, BoyaRezervFisiDetay.EdilecekMiktar, BoyaRezervFisiDetay.EdilenKilo, BoyaRezervFisiDetay.EdilenMiktar,
        BoyaRezervFisiDetay.EdilecekBrutKilo, BoyaRezervFisiDetay.EdilecekBrutMiktar, BoyaRezervFisiDetay.EdilenBrutKilo, BoyaRezervFisiDetay.EdilenBrutMiktar,
        BoyaRezervFisiDetay.InsertKullaniciId, BoyaRezervFisiDetay.InsertTarihi, BoyaRezervFisiDetay.KullaniciId, BoyaRezervFisiDetay.DegisimTarihi,
        IK.Adi, Kullanici.Adi, Renk.RenkId, Renk.Adi, Kalite.KaliteId, Kalite.KaliteAdi
      into :BoyaRezervFisiId, :FisTarihi, :FisNo, :Notu, :BoyaRezervFisiDetayId, :BoyaSiparisDetayId, :PartiKayitId, :PartiSinifId, :HamId, :HamAdi,
           :EdilecekKilo, :EdilecekMiktar, :EdilenKilo, :EdilenMiktar, :EdilecekBrutKilo, :EdilecekBrutMiktar, :EdilenBrutKilo, :EdilenBrutMiktar, :InsertKullaniciId, :InsertTarihi, :KullaniciId,
           :DegisimTarihi, :InsertKullaniciAdi, :KullaniciAdi, :PartiYil, :PartiId, :RenkId, :RenkAdi, :KaliteId, :KaliteAdi do
    begin
      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      suspend;
    end
end
