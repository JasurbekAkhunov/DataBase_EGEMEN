CREATE PROCEDURE PARTIICKONTROLRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AASAMAID INTEGER,
    AHAMID INTEGER,
    AFIRMAID VARCHAR(20),
    AKALINLIKID INTEGER,
    AIPLIKCINSIID INTEGER,
    AOZELLIKID INTEGER,
    AKUMASCINSIID INTEGER,
    ALYCRATIPIID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4))
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    MAKINASARJID INTEGER,
    PARTIASAMALARIID INTEGER,
    PARTISINIFID INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUPADI VARCHAR(30),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISGRUP2ADI VARCHAR(30),
    MAKINAID VARCHAR(10),
    HAMADITOPLAMI VARCHAR(255),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    PARTITAMIRID INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    ESKIPARTINOSTR VARCHAR(100),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BOBINSAYISI INTEGER,
    PROSESNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    LOT VARCHAR(150),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    DONMEISTENEN VARCHAR(10),
    CEKMEZLIKENISTENEN DOUBLE PRECISION,
    CEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    YIKAMAHASLIGIISTENEN VARCHAR(10),
    KALINLIKID INTEGER,
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIID INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    OZELLIKID INTEGER,
    OZELLIKADI VARCHAR(40),
    KUMASCINSIID INTEGER,
    KUMASCINSIADI VARCHAR(40),
    LYCRATIPIID INTEGER,
    LYCRATIPIADI VARCHAR(40),
    HAMGRUP1 VARCHAR(4),
    HAMGRUP2 VARCHAR(4),
    HAMGRUP3 VARCHAR(4),
    HAMGRUP4 VARCHAR(4),
    KALITEID VARCHAR(20),
    KALITEADI VARCHAR(255))
AS
declare variable itTamir        integer;
declare variable FEskiPartiYil          integer;
declare variable FEskiPartiNo           integer;
declare variable Okey integer;
begin
  Select itTamir from IslemTipiGetir into :itTamir;

  For Select PartiAsamaStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiAsamaStok.PartiAsamalariId, Parti.FirmaId, Firma.Adi, PartiAsamaStok.PartiSinifId,
             PartiStok.HamAdiToplami, Parti.Fason, Parti.ReceteId, Recete.Adi, Parti.IslemTipiId, IslemTipi.Adi, PartiAsamaStok.AsamaId, Asama.Adi,
             PartiAsamalari.PartiTamirId, TamirTipi.Adi, PartiStok.Kilo, PartiStok.Metre, PartiStok.BobinSayisi,
             PartiStok.EnIstenen, PartiStok.GramajIstenen, PartiStok.DonmeIstenen, PartiStok.CekmezlikEnIstenen,
             PartiStok.CekmezlikBoyIstenen, PartiStok.YikamaHasligiIstenen,
             Ham.KalinlikId, Kalinlik.Adi, Ham.IplikCinsiId, IplikCinsi.Adi, Ham.OzellikId, Ozellik.Adi, Ham.KumasCinsiId,
             KumasCinsi.Adi, Ham.LycraTipiId, LycraTipi.Adi, Ham.HamGrup1, Ham.HamGrup2, Ham.HamGrup3, Ham.HamGrup4,
             BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi,
             Max(PartiAsamaStokDetay.MakinaId),
             max(MakinaSarjId), BoyaSiparisDetay.KaliteId, Kalite.KaliteAdi from PartiAsamaStok
    left join Parti on Parti.PartiKayitId = PartiAsamaStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    Left Join KumasCinsi ON KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
    left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
    left join Ozellik on Ozellik.OzellikId  = Ham.OzellikId
    left join LycraTipi on LycraTipi.LycraTipiId  = Ham.LycraTipiId
    left join asama on asama.AsamaId = PartiAsamaStok.AsamaId
    left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join PartiAsamaIcKontrol on PartiAsamaIcKontrol.PartiSinifId = PartiAsamaStok.PartiSinifId and PartiAsamaIcKontrol.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    left join PartiTamir on PartiTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = PartiAsamaStok.PartiSinifId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitID
    left join SiparisGrup on SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 on SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    where PartiAsamaStok.CikisTarihi >= :ABaslangicTarihi and PartiAsamaStok.CikisTarihi <= :ABitisTarihi and
          (Firma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
          (BoyaSiparisDetay.HamID = :AHamId or :AHamId is null) and
          PartiAsamaStok.AsamaId = :AAsamaId and PartiAsamaIcKontrol.PartiKayitId is not null and
          (Ham.KalinlikId = :AKalinlikId or :AKalinlikId is null) and
          (Ham.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is null) and
          (Ham.KumasCinsiId = :AKumasCinsiId or :AKumasCinsiId is null) and
          (Ham.OzellikId = :AOzellikId or :AOzellikId is null) and
          (Ham.LycraTipiId = :ALycraTipiId or :ALycraTipiId is null) and
          (BoyaSiparis.SiparisGrupId = :ASiparisGrupId or :ASiparisGrupId is null) and
          (BoyaSiparis.SiparisGrup2Id = :ASiparisGrup2Id or :ASiparisGrup2Id is null) and
          ((:AHamGrup1Aralikli = 1 and HamGrup1 >= :ABaslangicGrup1Id  and (HamGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AHamGrup1Aralikli = 0) or (:AHamGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
          ((:AHamGrup2Aralikli = 1 and HamGrup2 >= :ABaslangicGrup2Id and (HamGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AHamGrup2Aralikli = 0) or (:AHamGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
          ((:AHamGrup3Aralikli = 1 and HamGrup3 >= :ABaslangicGrup3Id and (HamGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AHamGrup3Aralikli = 0) or (:AHamGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
          ((:AHamGrup4Aralikli = 1 and HamGrup4 >= :ABaslangicGrup4Id and (HamGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AHamGrup4Aralikli = 0) or (:AHamGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = ''))
    group by PartiAsamaStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, PartiAsamaStok.PartiAsamalariId, Parti.FirmaId, Firma.Adi,  PartiAsamaStok.PartiSinifId,
             PartiStok.HamAdiToplami, Parti.Fason, Parti.ReceteId, Recete.Adi, Parti.IslemTipiId, IslemTipi.Adi, PartiAsamaStok.AsamaId, Asama.Adi,
             PartiAsamalari.PartiTamirId, TamirTipi.Adi, PartiStok.Kilo, PartiStok.Metre, PartiStok.BobinSayisi,
             PartiStok.EnIstenen, PartiStok.GramajIstenen, PartiStok.DonmeIstenen, PartiStok.CekmezlikEnIstenen,
             PartiStok.CekmezlikBoyIstenen, PartiStok.YikamaHasligiIstenen,
             Ham.KalinlikId, Kalinlik.Adi, Ham.IplikCinsiId, IplikCinsi.Adi, Ham.OzellikId, Ozellik.Adi, Ham.KumasCinsiId,
             KumasCinsi.Adi, Ham.LycraTipiId, LycraTipi.Adi, Ham.HamGrup1, Ham.HamGrup2, Ham.HamGrup3, Ham.HamGrup4,
             BoyaSiparis.SiparisGrupId, SiparisGrup.Adi, BoyaSiparis.SiparisGrup2Id, SiparisGrup2.Adi, BoyaSiparisDetay.KaliteId, Kalite.KaliteAdi
  into :PartiKayitId, :PartiYil, :PartiId, :PartiAsamalariId, :FirmaId, :FirmaAdi, :PartiSinifId, :HamAdiToplami, :Fason, :ReceteId, :ReceteAdi, :IslemTipiId,
       :IslemTipiAdi, :AsamaId, :AsamaAdi, :PartiTamirId, :TamirTipiAdi, :Kilo, :Metre, :BobinSayisi,
       :EnIstenen, :GramajIstenen, :DonmeIstenen, :CekmezlikEnIstenen, :CekmezlikBoyIstenen, :YikamaHasligiIstenen,
       :KalinlikId, :KalinlikAdi, :IplikCinsiId, :IplikCinsiAdi, :OzellikId, :OzellikAdi, :KumasCinsiId,
       :KumasCinsiAdi, :LycraTipiId, :LycraTipiAdi, :HamGrup1, :HamGrup2, :HamGrup3, :HamGrup4,
       :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id, :SiparisGrup2Adi,
       :MakinaId, :MakinaSarjId, :KaliteId, :KaliteAdi do
  begin
    okey = 1;

    if (AHamGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, HamGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, HamGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, HamGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AHamGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, HamGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      Select ProsesNotu from PartiAsamalari where PartiAsamalari.PartiAsamalariID = :PartiAsamalariId into :ProsesNotu;

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId)  || ' ';
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      EskiPartiNoStr = '';

      Select Lot from PartiStokLotBilgisi(:PartiSinifID) into :Lot;

      if (IslemTipiID = itTamir) then
      begin
        for Select HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
          where PartiStokDetay.PartiSinifId = :PartiSinifId
          group by HamStokMaster.EskiPartiYil, HamStokMaster.EskiPartiNo
          into :FEskiPartiYil, :FEskiPartiNo do
        begin
          if (Char_Length(EskiPartiNoStr) < 80) then
          begin
            if (EskiPartiNoStr <> '' ) then EskiPartiNoStr = EskiPartiNoStr || cast(', ' as varchar(2) character set UTF8);

            EskiPartiNoStr = EskiPartiNoStr || '(' || FEskiPartiYil || ') ' || FEskiPartiNo;
          end
        end
      end
      suspend;
    end
  end
end
