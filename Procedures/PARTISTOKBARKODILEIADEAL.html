CREATE PROCEDURE PARTISTOKBARKODILEIADEAL(
    ABOYAIRSALIYEID INTEGER,
    AFIRMAID VARCHAR(20),
    ABARKODID VARCHAR(20),
    AKULLANICIID INTEGER)
AS
declare variable PartiKayitId           integer;
declare variable PartiSinifId           integer;
declare variable DepoId                 varchar(10);
declare variable PartiStokTartiId       integer;
declare variable BoyaIrsaliyeDetayId    integer;
declare variable HamStokSarfFisiDetayId integer;
declare variable PaketId                integer;
declare variable IK                     varchar(2);
declare variable TF                     integer;
declare variable FirmaId                varchar(20);
declare variable ParcaliSevk            integer;
declare variable Durumu                 integer;
declare variable TamamiTartildi         integer;
declare variable Sevk                   integer;
declare variable OrmeFirmaKodumuz       varchar(20);
declare variable FirmaKodumuz           varchar(20);
declare variable EgemenFirmaId          varchar(20);
declare variable BoyaRezervFisiDetayId  integer;

declare variable KayitSayisi            integer;
declare variable FBoyaIrsaliyeId        integer;
declare variable FBoyaIrsaliyeDetayId   integer;
declare variable FHamStokSarfFisiDetayId integer;
declare variable FAdet                   integer;
declare variable FKapAdedi              integer;
declare variable FNetKilo               Numeric(12,3);
declare variable FNetMetre              Numeric(12,3);
declare variable FTahminiBrutKilo       numeric(12,3);
declare variable FTahminiBrutMetre       numeric(12,3);
declare variable BrutKilo               Numeric(12,3);
declare variable BrutMetre               Numeric(12,3);
declare variable FKaliteNo              integer;
declare variable FMsg                   varchar(255);
declare variable HamTipiId   integer;
declare variable FPartiStokTartiId       integer;
declare variable KumasSevkteTartiyiAyriGoster integer;
declare variable FKaliteAciklamaTipiId   integer;
declare variable FMamulProsesId   integer;
declare variable FBoyaRezervFisiDetayId integer;
declare variable FLinkIslemi integer;
declare variable YeniPartiStokTartiId integer;
declare variable FBarkodID          varchar(30);
declare variable KS integer;
declare variable FOlcuId integer;
begin
  if (Char_Length(ABarkodId) < 3) then exception BarkodHatali1;

  FBarkodID = ABarkodId;

  IK = EgemenCopyStr(ABarkodId, 1, 2);

  ABarkodId = EgemenCopyStr(ABarkodId, 3, Char_Length(ABarkodId) - 2);
  TF = EgemenStrToIntOlurmu(ABarkodId);

  if (TF = 0) then exception BarkodHatali1;

  select count(*), Max(PartiStokTartiId) from PartiStokTarti where BarkodId = :FBarkodId into :KS, :PaketId;
  if (KS > 0) then /*iade kaydidir*/
    IK = 'R-';
  else
    PaketId = EgemenStrToInt(ABarkodId);

  Select FirmaKodumuz, OrmeFirmaKodumuz, EgemenFirmaId, KumasSevkteTartiyiAyriGoster from Sistem where sistemid = 1 into :FirmaKodumuz, :OrmeFirmaKodumuz, :EgemenFirmaId, :KumasSevkteTartiyiAyriGoster;

  if (IK = 'K-') then
  begin
    Select PsKaliteKontrol.PartiKayitId, PsKaliteKontrol.PartiSinifId, PsKaliteKontrol.PartiStokTartiId, PartiStokTarti.DepoId,
           PartiStokTarti.BoyaIrsaliyeDetayId, BoyaIrsaliye.FirmaID, PartiStokTarti.HamStokSarfFisiDetayId, Depo.Sevk, PartiStokTarti.BoyaRezervFisiDetayId,
           PartiStokTarti.KaliteNo, PartiStokTarti.BrutKilo, PartiStokTarti.BrutMetre, PartiStokTarti.Kilo, PartiStokTarti.Metre, PartiStokTarti.KaliteAciklamaTipiId,
           PartiStokTarti.KapAdedi, PartiStokTarti.Adet, PsKaliteKontrol.OlcuId from PsKaliteKontrol
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = PsKaliteKontrol.PartiStokTartiId
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = PartiStokTarti.BoyaIrsaliyeId
      where PsKaliteKontrol.PsKaliteKontrolId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :DepoId, :BoyaIrsaliyeDetayId, :FirmaID, :HamStokSarfFisiDetayId, :Sevk, :BoyaRezervFisiDetayId,
           :FKaliteNo, :FTahminiBrutKilo, :FTahminiBrutMetre, :FNetKilo, :FNetMetre, :FKaliteAciklamaTipiId, :FKapAdedi, :FAdet, :FOlcuId;
  end
  else
  if (IK = 'R-') then /*iade kaydi*/
  begin
    Select PartiStokTarti.PartiKayitId, PartiStokTarti.PartiSinifId, PartiStokTarti.PartiStokTartiId, PartiStokTarti.DepoId,
           PartiStokTarti.BoyaIrsaliyeDetayId, BoyaIrsaliye.FirmaID, PartiStokTarti.HamStokSarfFisiDetayId, Depo.Sevk, PartiStokTarti.BoyaRezervFisiDetayId,
           PartiStokTarti.KaliteNo, PartiStokTarti.BrutKilo, PartiStokTarti.BrutMetre, PartiStokTarti.Kilo, PartiStokTarti.Metre, PartiStokTarti.KaliteAciklamaTipiId,
           PartiStokTarti.KapAdedi, PartiStokTarti.Adet, PartiStokTarti.OlcuID from PartiStokTarti
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = PartiStokTarti.BoyaIrsaliyeId
      where PartiStokTarti.PartiStokTartiId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :DepoId, :BoyaIrsaliyeDetayId, :FirmaID, :HamStokSarfFisiDetayId, :Sevk, :BoyaRezervFisiDetayId,
           :FKaliteNo, :FTahminiBrutKilo, :FTahminiBrutMetre, :FNetKilo, :FNetMetre, :FKaliteAciklamaTipiId, :FKapAdedi, :FAdet, :FOlcuId;
  end
  else
  begin
    Select IplikPaketleme.PartiKayitId, IplikPaketleme.PartiSinifId, IplikPaketleme.PartiStokTartiId, PartiStokTarti.DepoId,
           PartiStokTarti.BoyaIrsaliyeDetayId, BoyaIrsaliye.FirmaID, PartiStokTarti.HamStokSarfFisiDetayId, Depo.Sevk, PartiStokTarti.BoyaRezervFisiDetayId,
           PartiStokTarti.KaliteNo, PartiStokTarti.BrutKilo, PartiStokTarti.BrutMetre, PartiStokTarti.Kilo, PartiStokTarti.Metre, PartiStokTarti.KaliteAciklamaTipiId,
           PartiStokTarti.KapAdedi, PartiStokTarti.Adet, PartiStokTarti.OlcuId from IplikPaketleme
      left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = IplikPaketleme.PartiStokTartiId
      left join Depo on Depo.DepoId = PartiStokTarti.DepoId
      left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = PartiStokTarti.BoyaIrsaliyeId
      where IplikPaketleme.IplikPaketlemeId = :PaketId
      into :PartiKayitId, :PartiSinifId, :PartiStokTartiId, :DepoId, :BoyaIrsaliyeDetayId, :FirmaID, :HamStokSarfFisiDetayId, :Sevk, :BoyaRezervFisiDetayId,
           :FKaliteNo, :FTahminiBrutKilo, :FTahminiBrutMetre, :FNetKilo, :FNetMetre, :FKaliteAciklamaTipiId, :FKapAdedi, :FAdet, :FOlcuId;
  end


  if (PartiSinifId is null) then exception BarkodHatali2;
  if (PartiStokTartiId is null) then exception TartiKaydiYokSevkEdilemez;
  if (BoyaIrsaliyeDetayId is null) then exception BuPaketimIrsaliyesiKesilmemis;
  else
  begin
    Select Count(*) from PartiStokTarti
      where PartiStokTarti.BarkodId = :FBarkodID and
            PartiStokTarti.BoyaIrsaliyeDetayId is Null and
            PartiStokTarti.IadeBoyaIrsaliyeDetayId is not null
    into :KayitSayisi;

    if (KayitSayisi > 0) then Exception BuPaketimIrsaliyesiKesilmemis;
  end
  if (HamStokSarfFisiDetayId is not null) then exception BuPaketSarFisiIleCikmis;
  if (FirmaId <> AFirmaId) then exception SevkFirmaIadeFirmaFarkli;

  Select Parti.FirmaId, Parti.ParcaliSevk, PartiStok.HamTipiId, PartiStok.Durumu, PartiStok.TamamiTartildi from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    where PartiStok.PartiSinifId = :PartiSinifId
    into :FirmaId, :ParcaliSevk, :HamTipiId, :Durumu, :TamamiTartildi;

  FLinkIslemi = 0;

  FBoyaIrsaliyeDetayId = Null;

  if (HamTipiId = 0 or KumasSevkteTartiyiAyriGoster = 0) then
  begin
    Select BoyaIrsaliyeDetayId from BoyaIrsaliyeDetay
      where BoyaIrsaliyeDetay.BoyaIrsaliyeId = :ABoyaIrsaliyeId and BoyaIrsaliyeDetay.PartiSinifId = :PartiSinifId and
            BoyaIrsaliyeDetay.KaliteNo = :FKaliteNo and IntKarsilastir(BoyaIrsaliyeDetay.KaliteAciklamaTipiId, :FKaliteAciklamaTipiId) = 1 and
            IntKarsilastir(BoyaIrsaliyeDetay.OlcuId, :FOlcuId) = 1
      into :FBoyaIrsaliyeDetayId;
  end

  if (FBoyaIrsaliyeDetayId is not Null) then
  begin
      Update BoyaIrsaliyeDetay Set KapAdedi = KapAdedi + :FKapAdedi,
                                   BrutKilo = BrutKilo + :FTahminiBrutKilo,
                                   Kilo = Kilo + :FNetKilo,
                                   BrutMetre = BrutMetre + :FTahminiBrutMetre,
                                   Metre = Metre + :FNetMetre,
                                   BarkodLinkIslemi = 1,
                                   LinkIslemi = :FLinkIslemi
        where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = :FBoyaIrsaliyeDetayId;

  end
  else
  begin
    FBoyaIrsaliyeDetayId = gen_id(BoyaIrsaliyeDetay_Id_Gen, 1);

    insert into BoyaIrsaliyeDetay(BoyaIrsaliyeDetayId, BoyaIrsaliyeId, FasonHamIade, PartiSinifId, PartiKayitId, KapAdedi, BrutKilo, Kilo, BrutMetre, Metre, KaliteNo, KaliteAciklamaTipiId, InsertKullaniciId, KullaniciId, LinkIslemi, OlcuId)
                  values(:FBoyaIrsaliyeDetayId, :ABoyaIrsaliyeId, 0, :PartiSinifId, :PartiKayitId,  :FkapAdedi, :FTahminiBrutKilo, :FNetKilo, :FTahminiBrutMetre, :FNetMetre, :FKaliteNo, :FKaliteAciklamaTipiId, :AKullaniciId, :AKullaniciId, :FLinkIslemi, :FOlcuId);

  end

  Select DepoRaf.DepoId from BoyaIrsaliye
        left join Unite on Unite.UniteID = BoyaIrsaliye.UniteId
        left join DepoRaf on DepoRaf.RafId = Unite.BobinSevkeHazirAmbarRafId
        where BoyaIrsaliye.BoyaIrsaliyeID = :ABoyaIrsaliyeId into :DepoId;

  if (DepoId is null) then
    exception KayitHatasi 'İrsaliye Ünitesinin "Bobin Sevke Hazir Ambar RafId" si Belirtilmemiş.';

  YeniPartiStokTartiId = gen_id(PartiStokTarti_Id_Gen, 1);

  insert into PartiStokTarti (PartiStokTartiId, PartiKayitId, PartiSinifId, Tarih, KapAdedi, Dara, Kilo, BrutKilo, Metre, BrutMetre, Adet, BobinSayisi, KaliteNo, KaliteKontrol, DepoId, PaletNo, KaliteAciklamaTipiId, MamulProsesId, InsertKullaniciId, InsertTarihi, KullaniciId, LinkIslemi, BarkodId, IadeBoyaIrsaliyeID, IadeBoyaIrsaliyeDetayID, OlcuId)
  select :YeniPartiStokTartiId, PartiKayitId, PartiSinifId, current_timestamp(0), KapAdedi, Dara, Kilo, BrutKilo, Metre, BrutMetre, Adet, BobinSayisi, KaliteNo, 0, :DepoId, PaletNo, KaliteAciklamaTipiId, MamulProsesId, :AKullaniciId, current_timestamp(0), :AKullaniciId, 1, :FBarkodID, :ABoyaIrsaliyeID, :FBoyaIrsaliyeDetayId, OlcuId
  from PartiStokTarti where PartiStokTartiId = :PartiStokTartiId;

end
