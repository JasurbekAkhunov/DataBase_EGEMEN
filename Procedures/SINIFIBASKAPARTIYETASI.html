CREATE PROCEDURE SINIFIBASKAPARTIYETASI(
    AAKTARILACAKPARTISINIFID INTEGER,
    AYENIPARTIKAYITID INTEGER,
    AYENIBOYASIPARISDETAYID INTEGER,
    AYENIBOYASIPARISSARJID INTEGER,
    AYENIBOYASIPARISBASKIID INTEGER,
    ACOKLUSINIFAKTAR EGEBOOLEAN = 0,
    APARTIASAMALARIKAYDEDILDI EGEBOOLEAN = 0)
AS
declare variable KayitSayisi            integer;
declare variable BoyaSiparisKayitId     integer;
declare variable BoyaSiparisDetayId     integer;
declare variable EskiPartiKayitId       integer;
declare variable PartiSinifId           integer;
declare variable EskiPartiAsamalariId   integer;
declare variable PartiAsamalariId       integer;
declare variable HamId                  integer;

declare variable EHamStokDetayId        integer;
declare variable EBoyaSiparisRezervId   integer;
declare variable ERafDetayId            integer;

declare variable EKapAdedi              integer;
declare variable EKilo                  Numeric(12,3);
declare variable EMetre                 Numeric(12,3);
declare variable EAdet                  integer;

declare variable EBobinSayisi           integer;

declare variable YKilo                  Numeric(12,3);
declare variable YMetre                 Numeric(12,3);
declare variable YAdet                  integer;

declare variable RBoyaSiparisRezervDetayId      integer;
declare variable RKilo                  Numeric(12,3);
declare variable RMetre                 Numeric(12,3);
declare variable RAdet                  integer;
declare variable RKapAdedi              integer;

declare variable KapAdedi               integer;
declare variable Kilo                   Numeric(12,3);
declare variable Metre                  Numeric(12,3);
declare variable Adet                   integer;
declare variable BobinSayisi            integer;
declare variable BoyaSiparisRezervId    integer;
declare variable EskiPartiStokDurumu    integer;
declare variable EskiPartiFason         integer;
declare variable EskiPartiFirmaId       varchar(20);
declare variable YeniPartiFason         integer;
declare variable YeniPartiFirmaId       varchar(20);
declare variable FHamTipiId             integer;
declare variable EskiPartiTamirId       integer;
declare variable FPartiTamirId          integer;
declare variable FPartiTamirSinifId     integer;
declare variable ProxyId                varchar(8);
declare variable AsamaSilinecek         integer;
declare variable PartiIslemTipiId       integer;
declare variable SiparisIslemTipiId     integer;
declare variable SinifSayisi            integer;
declare variable itNormal               integer;
declare variable itRpt                  integer;
declare variable itKontrol              integer;
declare variable itTamir                integer;
declare variable itNumune               integer;
declare variable DokumaKumasiKilodanTakipEt   integer;
declare variable FKaliteKontrolAsamaId        integer;
declare variable PMiktarDegisinceRecBoyaDegis integer;
begin
  Select DokumaKumasiKilodanTakipEt, KaliteKontrolAsamaId from Sistem into :DokumaKumasiKilodanTakipEt, :FKaliteKontrolAsamaId;

  Select itNormal, itRpt, itKontrol, itTamir, itNumune from IslemTipiGetir into :itNormal, :itRpt, :itKontrol, :itTamir, :itNumune;

  Select PartiStok.HamTipiId, PartiStok.PartiKayitId, Parti.Fason, Parti.FirmaId, PartiStok.Durumu, PartiStok.ProxyId, Parti.IslemTipiId from PartiStok
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    where PartiStok.PartiSinifId = :AAktarilacakPartiSinifId
    into :FHamTipiId, :EskiPartiKayitId, :EskiPartiFason, :EskiPartiFirmaId, :EskiPartiStokDurumu, :ProxyId, :PartiIslemTipiId;


  Select Parti.Fason, Parti.FirmaId from Parti where Parti.PartiKayitId = :AYeniPartiKayitId into :YeniPartiFason, :YeniPartiFirmaId;

  if (EskiPartiFason <> YeniPartiFason) then exception SatisTipiUymuyorAktarilamaz;

  if (EskiPartiFason = 1) then
  if (EskiPartiFirmaId <> YeniPartiFirmaId) then exception FasonFirmaUymuyorAktarilamaz;

  Select count(*) from PartiAsamaStok
        where PartiAsamaStok.PartiSinifId = :AAktarilacakPartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and (PartiAsamaStok.AsamaID <> :FKaliteKontrolAsamaId or :FKaliteKontrolAsamaId is Null) and
              PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.CikisTarihi is null into :KayitSayisi;

  if (KayitSayisi > 0) then exception BitmemisAsamasiVarTasinamaz;

  if (APartiAsamalariKaydedildi <> 1) then
  begin
    Select Count(*) from PartiAsamalari where PartiAsamalari.PartiKayitId = :AYeniPartiKayitId into :KayitSayisi;
    if (KayitSayisi > 0) then exception AsamaGirilmisSinifTasinamaz;
  end

  if (AYeniBoyaSiparisBaskiId is not null) then
  begin
    Select BoyaSiparisDetay.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.HamId, BoyaSiparisDetay.IslemTipiId from BoyaSiparisBaski
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisBaski.BoyaSiparisDetayId
       where BoyaSiparisBaski.BoyaSiparisBaskiId = :AYeniBoyaSiparisBaskiId
       into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :HamId, :SiparisIslemTipiId;
  end
  else if (AYeniBoyaSiparisSarjId is not null) then
  begin
    Select BoyaSiparisSarj.BoyaSiparisKayitId, BoyaSiparisSarj.BoyaSiparisDetayId, BoyaSiparisDetay.HamId, BoyaSiparisDetay.IslemTipiId from BoyaSiparisSarj
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisSarj.BoyaSiparisDetayId
       where BoyaSiparisSarj.BoyaSiparisSarjId = :AYeniBoyaSiparisSarjId
       into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :HamId, :SiparisIslemTipiId;
  end
  else
  begin
    Select BoyaSiparisDetay.BoyaSiparisKayitId, BoyaSiparisDetay.HamId, BoyaSiparisDetay.IslemTipiId from BoyaSiparisDetay
       where BoyaSiparisDetay.BoyaSiparisDetayId = :AYeniBoyaSiparisDetayId
       into :BoyaSiparisKayitId, :HamId, :SiparisIslemTipiId;

    BoyaSiparisDetayId = AYeniBoyaSiparisDetayId;
  end

  if (PartiIslemTipiId <> SiparisIslemTipiId) then
  begin
    if (PartiIslemTipiId = itTamir or SiparisIslemTipiId = itTamir) then exception IslemTipiUymuyorAktarilamaz;
  end

  Select Count(*) from PartiStokDetay where PartiStokDetay.PartiSinifId = :AAktarilacakPartiSinifId and PartiStokDetay.HamId <> :HamId into :KayitSayisi;

  if (KayitSayisi > 0) then exception HamStokSipariseUygunDegil;

  Select Count(*) from PartiStok where PartiStok.PartiKayitId = :EskiPartiKayitId into :KayitSayisi;

  if (KayitSayisi = 1) then AsamaSilinecek = 1;
  else AsamaSilinecek = 0;

  PartiSinifId = gen_id(PartiSinif_Id_gen, 1);

  Update PartiStok Set ProxyId = Null where PartiStok.PartiSinifId = :AAktarilacakPartiSinifId;


/* Yeni Sinif aciyoruz */
  Insert Into PartiStok(PartiKayitId, PartiSinifId, BoyaSiparisDetayId, BoyaSiparisSarjId, BoyaSiparisBaskiId, MamulKodu, ProxyId,
             IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen, CekmezlikBoyIstenen,
             YikamaHasligiIstenen, MaximumFireOrani, MinimumFireOrani, Durumu, HamId, TasindigiPartiKayitId, BolunduguPartiSinifId, LinkIslemi)
        Select :AYeniPartiKayitId, :PartiSinifId, :BoyaSiparisDetayId, :AYeniBoyaSiparisSarjId, :AYeniBoyaSiparisBaskiId, MamulKodu, :ProxyId,
             IstenenRaporBoyu, IstenenOzelTuse, EnIstenen, GramajIstenen, DonmeIstenen, CekmezlikEnIstenen, CekmezlikBoyIstenen,
             YikamaHasligiIstenen, MaximumFireOrani, MinimumFireOrani,  :EskiPartiStokDurumu, HamId, :EskiPartiKayitId, BolunduguPartiSinifId, 1 from PartiStok where PartiStok.PartiSinifId = :AAktarilacakPartiSinifId;

  for Select PartiTamirSinif.PartiTamirId from PartiTamirSinif where PartiTamirSinif.PartiSinifId = :AAktarilacakPartiSinifId into :EskiPartiTamirId do
  begin
    FPartiTamirId = gen_id(PartiTamir_Id_gen, 1);
    FPartiTamirSinifId = Gen_Id(PartiTamirSinif_Id_Gen, 1);

    insert into PartiTamir(PartiTamirId, PartiKayitId, TamirTipiId, Notu, Tarih, TamirOnayKullaniciId, TamirOnayTarihi, TamirCozumSekliId,
                           SebepTespitEdildi, TamirSebebiId, TamirSebebiNotu, SebepOnayKullaniciId, SebepOnayTarihi, OnleyiciFaaliyeti,
                           OnleyiciFaaliyetTipiId,
                           OnleyiciOnayKullaniciId, OnleyiciOnayTarihi, TamirDuzeltmeTipiId, Bitti, HataPartiAsamalariId, EskiPartiTamirId,
                           InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
       select :FPartiTamirId, :AYeniPartiKayitId, PartiTamir.TamirTipiId, PartiTamir.Notu, PartiTamir.Tarih, PartiTamir.TamirOnayKullaniciId,
              PartiTamir.TamirOnayTarihi, PartiTamir.TamirCozumSekliId, PartiTamir.SebepTespitEdildi, PartiTamir.TamirSebebiId, PartiTamir.TamirSebebiNotu,
              PartiTamir.SebepOnayKullaniciId, PartiTamir.SebepOnayTarihi, PartiTamir.OnleyiciFaaliyeti,
              PartiTamir.OnleyiciFaaliyetTipiId, PartiTamir.OnleyiciOnayKullaniciId,
              PartiTamir.OnleyiciOnayTarihi, PartiTamir.TamirDuzeltmeTipiId, PartiTamir.Bitti, PartiTamir.HataPartiAsamalariId,
              :EskiPartiTamirId, PartiTamir.InsertKullaniciId, PartiTamir.InsertTarihi, PartiTamir.KullaniciId, PartiTamir.DegisimTarihi from PartiTamir
            where PartiTamir.PartiTamirId = :EskiPartiTamirId;


    insert into PartiTamirSinif(PartiTamirSinifId, PartiTamirId, PartiSinifId, Bitti, KullaniciId, DegisimTarihi)
        Select :FPartiTamirSinifId, :FPartiTamirId, :PartiSinifId, PartiTamirSinif.Bitti, PartiTamirSinif.KullaniciId, PartiTamirSinif.DegisimTarihi from PartiTamirSinif
          where PartiTamirSinif.PartiTamirId = :EskiPartiTamirId and PartiTamirSinif.PartiSinifId = :AAktarilacakPartiSinifId;


  end

  PartiAsamalariId = Null;
  For Select PartiAsamalari.PartiAsamalariId, PartiAsamalari.PartiTamirId from PartiAsamalari
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStok.PartiSinifId = :AAktarilacakPartiSinifId
       where PartiAsamalari.PartiKayitId = :EskiPartiKayitId
       order By PartiAsamalari.Sira
       into :EskiPartiAsamalariId, :EskiPartiTamirId do
  begin
    FPartiTamirId = Null;
    FPartiTamirSinifId = null;

/* işletme içi tamiri düzenle */
    if (EskiPartiTamirId is not null) then
    begin
      Select PartiTamirId from PartiTamir where PartiTamir.PartiKayitID = :AYeniPartiKayitId and PartiTamir.EskiPartiTamirId = :EskiPartiTamirId into :FPartiTamirId;
      Select PartiTamirSinifId from PartiTamirSinif where PartiTamirSinif.PartiSinifID = :PartiSinifId and PartiTamirSinif.PartiTamirId = :FPartiTamirId into :FPartiTamirSinifId;

    if (AsamaSilinecek = 0) then
      begin
        select count(*) from PartiTamirSinif where PartiTamirSinif.PartiTamirID = :EskiPartiTamirId and PartiTamirSinif.PartiSinifId <> :AAktarilacakPartiSinifId into :KayitSayisi;
        if (KayitSayisi = 0) then AsamaSilinecek = 1;
      end
    end

    if (APartiAsamalariKaydedildi <> 1) then
    begin
      PartiAsamalariId = Gen_Id(PartiAsamalari_Id_gen, 1);
      insert into PartiAsamalari(PartiKayitId, PartiAsamalariId, ReceteId, AsamaId, OtomasyonaDahilmi, KontrolAsama, PartiTamirId,
                                 BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, HamStokFlotteCarpani, Koyultma, DurulamaSuyu, BoyaSuyu, Tur, BoyamaSeviyesi, StoktanDusuldu, PartininAsilRecetesimi, MakinaPlanlamaGrupId,
                                 GrupProsesNo, ProsesNotu, IlaveProses, IlaveProsesSebebiId, SerbestAsama, Sira, EskiPartiAsamalariId, AsamaDegistirildi, AsamaDegisimTarihi)
            Select :AYeniPartiKayitId, :PartiAsamalariId, ReceteId, AsamaId, OtomasyonaDahilmi, KontrolAsama, :FPartiTamirId,
                    BanyoCarpani, BanyoCarpani2, BanyoCarpani3, BanyoCarpani4, HamStokFlotteCarpani, Koyultma, DurulamaSuyu, BoyaSuyu, Tur, BoyamaSeviyesi, StoktanDusuldu, PartininAsilRecetesimi, MakinaPlanlamaGrupId, GrupProsesNo, ProsesNotu,
                    IlaveProses, IlaveProsesSebebiId,  1, -1,
                    PartiAsamalari.PartiAsamalariId, PartiAsamalari.AsamaDegistirildi, PartiAsamalari.AsamaDegisimTarihi from PartiAsamalari
            where PartiAsamalari.PartiAsamalariId = :EskiPartiAsamalariId;
    end
    else
    begin
      Select PartiAsamalari.PartiAsamalariId from PArtiAsamalari where PartiAsamalari.EskiPartiAsamalariId = :EskiPartiAsamalariId into :PartiAsamalariId;
      --Exception KayitHatasi Egemenintegertostr(PartiAsamalariId);
    end

    insert into PartiReceteDetay(PartiAsamalariId, Sira, ReceteAsamaId, ReceteAltAsamaId, UrunId, UrunBirimId, ReceteGrafikId, Miktar, Banyo)
         Select :PartiAsamalariId, -1, ReceteAsamaId, ReceteAltAsamaId, UrunId, UrunBirimId, ReceteGrafikId, Miktar, Banyo from PartiReceteDetay
          where PartiReceteDetay.PartiAsamalariId = :EskiPartiAsamalariId order by PartiReceteDetay.Sira;

    if (APartiAsamalariKaydedildi <> 1) then
    begin
      delete from PartiAsamaStok where PartiAsamaStok.PartiAsamalariId = :PartiAsamalariId;
    end

    Update PartiAsamaStok Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId
       where PartiAsamaStok.PartiAsamalariId = :EskiPartiAsamalariId and PartiAsamaStok.PartiSinifId = :AAktarilacakPartiSinifId and PartiAsamaStok.PartiTamirSinifId is null;

    Update PartiAsamaStok Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId, PartiTamirSinifId = :FPartiTamirSinifId
       where PartiAsamaStok.PartiAsamalariId = :EskiPartiAsamalariId and PartiAsamaStok.PartiSinifId = :AAktarilacakPartiSinifId and PartiAsamaStok.PartiTamirSinifId is not null;

    Update PartiAsamaStokDetay Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId
       where PartiAsamaStokDetay.PartiAsamalariId = :EskiPartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = :AAktarilacakPartiSinifId and PartiAsamaStokDetay.PartiTamirSinifId is null;

    Update PartiAsamaStokDetay Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId, PartiTamirSinifId = :FPartiTamirSinifId
       where PartiAsamaStokDetay.PartiAsamalariId = :EskiPartiAsamalariId and PartiAsamaStokDetay.PartiSinifId = :AAktarilacakPartiSinifId and PartiAsamaStokDetay.PartiTamirSinifId is not null;

    Update BolumPlanlama Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId
       where BolumPlanlama.PartiAsamalariId = :EskiPartiAsamalariId and BolumPlanlama.PartiSinifId = :AAktarilacakPartiSinifId;

    Update PartiStokFizikKontrol Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId, LinkIslemi = 1
       where PartiStokFizikKontrol.PartiAsamalariId = :EskiPartiAsamalariId and PartiStokFizikKontrol.PartiSinifId = :AAktarilacakPartiSinifId;

    Update PartiStokSonKontrol Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId, LinkIslemi = 1
       where PartiStokSonKontrol.PartiAsamalariId = :EskiPartiAsamalariId and PartiStokSonKontrol.PartiSinifId = :AAktarilacakPartiSinifId;

    Update PartiAsamaIcKontrol Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId
       where PartiAsamaIcKontrol.PartiAsamalariId = :EskiPartiAsamalariId and PartiAsamaIcKontrol.PartiSinifId = :AAktarilacakPartiSinifId;

    Update PartiAsamaDeger Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId
       where PartiAsamaDeger.PartiAsamalariId = :EskiPartiAsamalariId and PartiAsamaDeger.PartiSinifId = :AAktarilacakPartiSinifId;

    Update PartiAsamaAraKontrol Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId
       where PartiAsamaAraKontrol.PartiAsamalariId = :EskiPartiAsamalariId and PartiAsamaAraKontrol.PartiSinifId = :AAktarilacakPartiSinifId;

    Update PSKaliteKontrolAsama set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, PartiAsamalariId = :PartiAsamalariId, LinkIslemi = 1
       where PSKaliteKontrolAsama.PartiAsamalariId = :EskiPartiAsamalariId and PSKaliteKontrolAsama.PartiSinifId = :AAktarilacakPartiSinifId;

    if (AsamaSilinecek = 1) then
    begin
      Update ReceteIsEmriDetay Set PartiKayitId = :AYeniPartiKayitId, PartiAsamalariId = :PartiAsamalariId where ReceteIsEmriDetay.PartiAsamalariId = :EskiPartiAsamalariId;
      Delete from PartiReceteDetay where PartiReceteDetay.PartiAsamalariId = :EskiPartiAsamalariId;
      Update PartiAsamaIlaveSarf Set PartiKayitId = :AYeniPartiKayitId, PartiAsamalariId = :PartiAsamalariId where PartiAsamaIlaveSarf.PartiAsamalariId = :EskiPartiAsamalariId;

      Update StokDetay Set PartiKayitID = :AYeniPartiKayitId, PartiAsamalariID = :PartiAsamalariId
             where StokDetay.PartiAsamalariId = :EskiPartiAsamalariId;

    end
  end

  Update PSKaliteKontrol set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId, LinkIslemi = 1
     where PSKaliteKontrol.PartiSinifId = :AAktarilacakPartiSinifId;

  Update PartiTamir Set EskiPartiTamirID = Null, HataPartiAsamalariId = (Select PartiAsamalari.PartiAsamalariId from PartiAsamalari where PartiAsamalari.PartiKayitId = :AYeniPartiKayitId and PartiAsamalari.EskiPartiAsamalariId = PartiTamir.HataPartiAsamalariId)
         where PartiTamir.PartiKayitId = :AYeniPartiKayitId;

  Update PartiKumasKBilgi Set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId
       where PartiKumasKBilgi.PartiSinifId = :AAktarilacakPartiSinifId;

  if (ACokluSinifAktar = 1) then Update PartiAsamalari Set SerbestAsama = 0 where PartiAsamalari.PartiKayitId = :AYeniPartiKayitId;
  else Update PartiAsamalari Set EskiPartiAsamalariId = null, SerbestAsama = 0 where PartiAsamalari.PartiKayitId = :AYeniPartiKayitId;

  Update PartiStokPartileme set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId where PartiStokPartileme.PartiSinifId = :AAktarilacakPartiSinifId ;

  Update UniteIrsaliyeDetay set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId where UniteIrsaliyeDetay.PartiSinifId = :AAktarilacakPartiSinifId ;

  Update MamulDepoTransferFisiDetay set LinkIslemi = 1, PartiSinifId = :PartiSinifId where MamulDepoTransferFisiDetay.PartiSinifId = :AAktarilacakPartiSinifId ;
  Update PartiStokTarti set PartiKayitId = :AYeniPartiKayitId, PartiSinifId = :PartiSinifId where PartiStokTarti.PartiSinifId = :AAktarilacakPartiSinifId ;

  delete from PartiTamirSinif where PartiTamirSinif.PartiSinifId = :AAktarilacakPartiSinifId;

/* PartiStokDetay ve rezerv aktarmasi yapiyoruz */
  For Select PartiStokDetay.RafDetayId, RafDetay.HamStokDetayId, PartiStokDetay.BoyaSiparisRezervId, PartiStokDetay.KapAdedi, PartiStokDetay.Kilo,
             PartiStokDetay.Metre, PartiStokDetay.Adet, PartiStokDetay.BobinSayisi from PartiStokDetay
     left join RafDetay on rafDetay.rafDetayId = PartiStokDetay.RafDetayId
     where PartiStokDetay.PartiSinifId = :AAktarilacakPartiSinifId
     into :ERafDetayId, :EHamStokDetayId, :EBoyaSiparisRezervId, :EKapAdedi, :EKilo, :EMetre, :EAdet, :EBobinSayisi do
  begin
    update PartiStokDetay set LinkIslemi = 2 where PartiStokDetay.PartiSinifId = :AAktarilacakPartiSinifId and
                                                   PartiStokDetay.RafDetayId = :ERafDetayId;

    Delete from PartiStokDetay where PartiStokDetay.PartiSinifId = :AAktarilacakPartiSinifId and
                                     PartiStokDetay.RafDetayId = :ERafDetayId;

    KapAdedi = EKapAdedi;
    Kilo = EKilo;
    Metre = EMetre;
    Adet = EAdet;
    BobinSayisi = EBobinSayisi;

    For Select BoyaSiparisRezervDetay.BoyaSiparisRezervDetayId, BoyaSiparisRezervDetay.Kilo, BoyaSiparisRezervDetay.Metre, BoyaSiparisRezervDetay.Adet, BoyaSiparisRezervDetay.KapAdedi  from BoyaSiparisRezerv
        left join BoyaSiparisRezervDetay on BoyaSiparisRezervDetay.BoyaSiparisDetayId = BoyaSiparisRezerv.BoyaSiparisDetayId and
                  IntKarsilastir(BoyaSiparisRezervDetay.BoyaSiparisSarjId, BoyaSiparisRezerv.BoyaSiparisSarjId) = 1 and
                  BoyaSiparisRezervDetay.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
        where BoyaSiparisRezerv.BoyaSiparisRezervId = :EBoyaSiparisRezervId and BoyaSiparisRezervDetay.BoyaSiparisRezervDetayId is not null
        into :RBoyaSiparisRezervDetayId, :RKilo, :RMetre, :RAdet, :RKapAdedi do
    begin
      if (FHamTipiId <> 2 or DokumaKumasiKilodanTakipEt = 1) then
      begin
        if (Ekilo > 0) then
        if (EKilo < RKilo) then
        begin
          Update BoyaSiparisRezervDetay Set Kilo = Kilo - :EKilo, Adet = Adet - :EAdet, KapAdedi = KapAdedi - :EKapAdedi where BoyaSiparisRezervDetayId = :RBoyaSiparisRezervDetayId;

          Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, HamStokDetayId, Kilo, Adet, KapAdedi)
                                 values(:BoyaSiparisKayitId, :BoyaSiparisDetayId, :AYeniBoyaSiparisSarjId, :EHamStokDetayId, :EKilo, :EAdet, :EKapAdedi);
          EKilo = 0;
          EMetre = 0;
          EAdet = 0;
          EKapAdedi = 0;
        end
        else
        if (EKilo >= RKilo) then
        begin
          Delete from BoyaSiparisRezervDetay where BoyaSiparisRezervDetayId = :RBoyaSiparisRezervDetayId;

          Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, HamStokDetayId, Kilo, Adet, KapAdedi)
                                      values(:BoyaSiparisKayitId, :BoyaSiparisDetayId, :AYeniBoyaSiparisSarjId, :EHamStokDetayId, :RKilo, :RAdet, :RKapAdedi);

          EKilo = EKilo - RKilo;
          EMetre = EMetre - RMetre;
          EAdet = EAdet - RAdet;
          EKapAdedi = EKapAdedi - RKapAdedi;
        end
      end
      else
      begin
        if (EMetre > 0) then
        if (EMetre < RMetre) then
        begin
          Update BoyaSiparisRezervDetay Set Metre = Metre - :EMetre, Adet = Adet - :EAdet, KapAdedi = KapAdedi - :EKapAdedi where BoyaSiparisRezervDetayId = :RBoyaSiparisRezervDetayId;

          Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, HamStokDetayId, Metre, Adet, KapAdedi)
                                 values(:BoyaSiparisKayitId, :BoyaSiparisDetayId, :AYeniBoyaSiparisSarjId, :EHamStokDetayId, :EMetre, :EAdet, :EKapAdedi);
          EKilo = 0;
          EMetre = 0;
          EAdet = 0;
          EKapAdedi = 0;
        end
        else
        if (EKilo >= RKilo) then
        begin
          Delete from BoyaSiparisRezervDetay where BoyaSiparisRezervDetayId = :RBoyaSiparisRezervDetayId;

          Insert into BoyaSiparisRezervDetay(BoyaSiparisKayitId, BoyaSiparisDetayId, BoyaSiparisSarjId, HamStokDetayId, Metre, Adet, KapAdedi)
                                 values(:BoyaSiparisKayitId, :BoyaSiparisDetayId, :AYeniBoyaSiparisSarjId, :EHamStokDetayId, :RMetre, :RAdet, :RKapAdedi);

          EKilo = EKilo - RKilo;
          EMetre = EMetre - RMetre;
          EAdet = EAdet - RAdet;
          EKapAdedi = EKapAdedi - RKapAdedi;

        end
      end
    end

    Select BoyaSiparisRezervId from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisDetayId = :BoyaSiparisDetayId and
                                BoyaSiparisRezerv.BoyaSiparisKayitId = :BoyaSiparisKayitId and
                                IntKarsilastir(BoyaSiparisRezerv.BoyaSiparisSarjId, :AYeniBoyaSiparisSarjId) = 1 and
                                BoyaSiparisRezerv.HamStokDetayId = :EHamStokDetayId into :BoyaSiparisRezervId;

    insert into PartiStokDetay(PartiKayitId, PartiSinifId, BoyaSiparisRezervId, RafDetayId, KapAdedi, Kilo, Metre, Adet, BobinSayisi, LinkIslemi)
                     values(:AYeniPartiKayitId, :PartiSinifId, :BoyaSiparisRezervId, :ERafDetayId, :KapAdedi, :Kilo, :Metre, :Adet, :BobinSayisi, 1);
  end

  For Select PartiTamir.PartiTamirId from PartiTamir
    left join PartiTamirSinif on PartiTamirSinif.PartiTamirId = PartiTamir.PartiTamirId
    where PartiTamir.PartiKayitId = :EskiPartiKayitId and PartiTamirSinif.PartiTamirSinifId is null
    group by PartiTamir.PartiTamirId into :EskiPartiTamirId do
  begin

  select count(*) from PartiTamirSinif where PartiTamirSinif.PartiTamirID = :EskiPartiTamirId into :KayitSayisi;
    if (KayitSayisi = 0) then delete from PartiTamir where PartiTamir.PartiTamirID = :EskiPartiTamirId;
  end

  Select Count(*) from PartiStok where PartiStok.PartiKayitId = :EskiPartiKayitId into :KayitSayisi;

  if (KayitSayisi = 0) then
  begin
    For Select PartiAsamalariId from PartiAsamalari
      where PartiAsamalari.PartiKayitId = :EskiPartiKayitId
      into :EskiPartiAsamalariId do
    begin
       Update StokDetay Set PartiKayitID = :AYeniPartiKayitId, PartiAsamalariID = :PartiAsamalariId
             where StokDetay.PartiAsamalariId = :EskiPartiAsamalariId;
    end

    delete from PartiAsamalari where PartiAsamalari.PartiKayitId = :EskiPartiKayitId;
  end
  else
  begin
    -- Bu noktada Partinin bir kısmı başka partiye taşındığında eski partideki aşamalarının reçete boya miktarlarının sıfırlanması istendi. ( UP )
    -- Buradaki amaç, kullanıcının reçeteyi yazdırmadan önce boya miktarlarını dikkatli bir şekilde düzenlemeye zorlamak... ( CAN )
    Select PMiktarDegisinceRecBoyaDegis from Sistem where SistemID = 1 into :PMiktarDegisinceRecBoyaDegis;
    if (PMiktarDegisinceRecBoyaDegis = 1) then
    begin
      Update PartiAsamalariReceteBoyaMiktar set Miktar = 0 where PartiAsamalariReceteBoyaMiktar.PartiKayitID = :EskiPartiKayitId;
    end
  end

  Execute Procedure PartiSinifDurumuHesapla(:AYeniPartiKayitId, :PartiSinifId);
end
