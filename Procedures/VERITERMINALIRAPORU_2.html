CREATE PROCEDURE VERITERMINALIRAPORU_2(
    AMAKINAID VARCHAR(10),
    AMAKINAGRUBU VARCHAR(4),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AKALINLIKID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKUMASCINSIID INTEGER,
    ALYCRATIPIID INTEGER,
    ARECETEID VARCHAR(20))
RETURNS (
    PARTIASAMASTOKDETAYID INTEGER,
    MAKINAID VARCHAR(10),
    MAKINABILGIID INTEGER,
    TARIH TIMESTAMP,
    NEM DOUBLE PRECISION,
    ISI DOUBLE PRECISION,
    HIZ DOUBLE PRECISION,
    BACANEM DOUBLE PRECISION,
    ELEKTIRIK INTEGER,
    DOGALGAZ INTEGER)
AS
declare variable GirisTarihi  TimeStamp;
declare variable CikisTarihi  TimeStamp;
begin
  for Select PartiAsamaStokDetayId, MakinaId, GirisTarihi, CikisTarihi from VeriTerminaliRaporu_1(:AMakinaId, :AMakinaGrubu, :ABaslangicTarihi, :ABitisTarihi,  :AKalinlikId, :AIplikCinsiId, :AKumasCinsiId, :ALycraTipiId, :AReceteId)
    order by GirisTarihi
    into :PartiAsamaStokDetayId, :MakinaId, :GirisTarihi, :CikisTarihi do
  begin
    for Select MakinaBilgi.MakinaBilgiId, MakinaBilgi.tarih, MakinaBilgi.nem, MakinaBilgi.Isi, MakinaBilgi.Hiz, MakinaBilgi.BacaNem,
               MakinaBilgi.Elektirik, MakinaBilgi.Dogalgaz  from MakinaBilgi
      where MakinaBilgi.Tarih >= :GirisTarihi and MakinaBilgi.Tarih <= :CikisTarihi and MakinaBilgi.MakinaId = :MakinaId
      into :MakinaBilgiId, :Tarih, :Nem, :Isi, :Hiz, :BacaNem, :Elektirik, :Dogalgaz do
    begin
      suspend;
    end
  end
end
