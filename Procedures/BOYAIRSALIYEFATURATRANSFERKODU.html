CREATE PROCEDURE BOYAIRSALIYEFATURATRANSFERKODU(
    ABOYAIRSALIYEDETAYID INTEGER,
    AHAMSTOKDETAYID INTEGER,
    AEGEMENFIRMAID FIRMAKODUSIZE)
RETURNS (
    TRANSFERKODU VARCHAR(55),
    ISLEMGRUBUID INTEGER,
    HAREKETSEKLI VARCHAR(20))
AS
declare variable hstIplik       integer;
declare variable hstElyaf       integer;
declare variable hstTelef       integer;
declare variable EgemenFirmaID                integer;
declare variable SiparisIslemGrubuId          integer;
declare variable SistemDovizId                varchar(5);
declare variable BoyaliIplikMSatisTipi        varchar(20);
declare variable BoyaliKumasMSatisTipi        varchar(20);
declare variable BaskiliKumasMSatisTipi       varchar(20);
declare variable DijitalBaskiliKumasMSatisTipi varchar(20);
declare variable FGirisFirmaID         FirmaKoduSize;
declare variable FFirmaKodumuz         FirmaKoduSize;
declare variable FPartiKayitID         integer;
declare variable KayitSayisi           integer;
declare variable FOrmeFirmaKodumuz     FirmaKoduSize;
declare variable PartiSinifId          integer;
declare variable Fason                 integer;
declare variable HamStokbirimId        integer;
declare variable HamTipiId             integer;
declare variable FDijitalBaski         SmallInt;
declare variable ImalatTipiId          integer;
declare variable RafDetayId            integer;
declare variable FHamStokHareketTipiID integer;
declare variable FEskiPartiSinifID     integer;
declare variable FDevirPartiYil        Integer;
declare variable FDevirPartiID         Integer;
declare variable FDevirIslemGrubuID    Integer;
declare variable FDevirDesenID         varchar(20);
declare variable GirisRenk             varchar(20);
declare variable HamSatisTransferKodu  varchar(80);


declare variable BoyaliTransferKodu        varchar(55);
declare variable SekerKasarTransferKodu    varchar(55);
declare variable OptikBeyazTransferKodu    varchar(55);
declare variable YikamaTransferKodu        varchar(55);
declare variable CiftBoyaTransferKodu      varchar(55);
declare variable PolyesterTransferKodu     varchar(55);
declare variable NaylonTransferKodu        varchar(55);

declare variable BoyaliTransferKoduMt        varchar(55);
declare variable SekerKasarTransferKoduMt    varchar(55);
declare variable OptikBeyazTransferKoduMt    varchar(55);
declare variable YikamaTransferKoduMt        varchar(55);
declare variable CiftBoyaTransferKoduMt      varchar(55);
declare variable PolyesterTransferKoduMt     varchar(55);
declare variable NaylonTransferKoduMt        varchar(55);

declare variable SABoyaliTransferKodu      varchar(55);
declare variable SASekerKasarTransferKodu  varchar(55);
declare variable SAOptikBeyazTransferKodu  varchar(55);
declare variable SAYikamaTransferKodu      varchar(55);
declare variable SACiftBoyaTransferKodu      varchar(55);
declare variable SAPolyesterTransferKodu     varchar(55);
declare variable SANaylonTransferKodu        varchar(55);

declare variable SABoyaliTransferKoduMt      varchar(55);
declare variable SASekerKasarTransferKoduMt  varchar(55);
declare variable SAOptikBeyazTransferKoduMt  varchar(55);
declare variable SAYikamaTransferKoduMt      varchar(55);
declare variable SACiftBoyaTransferKoduMt      varchar(55);
declare variable SAPolyesterTransferKoduMt     varchar(55);
declare variable SANaylonTransferKoduMt        varchar(55);

declare variable AlisBoyaliTransferKodu        varchar(55);
declare variable AlisSekerKasarTransferKodu    varchar(55);
declare variable AlisOptikBeyazTransferKodu    varchar(55);
declare variable AlisYikamaTransferKodu        varchar(55);
declare variable AlisTransferKoduMt            varchar(55);
declare variable HamKumasMSatisTipi            varchar(10);

declare variable FasonHamIade               EgeBoolean;
begin
  if (AEgemenFirmaID = 34) then /*örkum*/
  begin
    Select hstIplik, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstElyaf, :hstTelef;

    Select SistemDovizId, EgemenFirmaID, BoyaliIplikMSatisTipi, BoyaliKumasMSatisTipi, BaskiliKumasMSatisTipi, FirmaKodumuz, OrmeFirmaKodumuz, DijitalBaskiliKumasMSatisTipi, HamKumasMSatisTipi from Sistem where sistemId = 1
    into :SistemDovizId, :EgemenFirmaID, :BoyaliIplikMSatisTipi, :BoyaliKumasMSatisTipi, :BaskiliKumasMSatisTipi, :FFirmaKodumuz, :FOrmeFirmaKodumuz, :DijitalBaskiliKumasMSatisTipi, :HamKumasMSatisTipi;

    IslemGrubuID = Null;
    TransferKodu = Null;
    HareketSekli = Null;

    select BoyaIrsaliyeDetay.PartiSinifId, Parti.Fason, Coalesce(BoyaFaturaDetay.MiktarBirim, BoyaIrsaliyeDetay.HamStokBirimId), BoyaSiparisDetay.IslemGrubuId,
    Ham.BoyaliTransferKodu, Ham.SekerKasarTransferKodu,
    Ham.OptikBeyazTransferKodu, Ham.YikamaTransferKodu, Ham.SABoyaliTransferKodu, Ham.SASekerKasarTransferKodu,
    Ham.SAOptikBeyazTransferKodu, Ham.SAYikamaTransferKodu,
    Ham.AlisTransferKoduMt, Ham.BoyaliTransferKoduMt, Ham.SekerKasarTransferKoduMt,
    Ham.OptikBeyazTransferKoduMt, Ham.YikamaTransferKoduMt, Ham.SABoyaliTransferKoduMt, Ham.SASekerKasarTransferKoduMt,
    Ham.SAOptikBeyazTransferKoduMt, Ham.SAYikamaTransferKoduMt, Ham.AlisBoyaliTransferKodu,
    Ham.AlisSekerKasarTransferKodu, Ham.AlisOptikBeyazTransferKodu, Ham.AlisYikamaTransferKodu, Ham.HamTipiId,
    BaskiTipi.DijitalBaski, Parti.ImalatTipiId, Parti.PartiKayitId, BoyaIrsaliyeDetay.RafDetayID, Ham.HamSatisTransferKodu,
    BoyaIrsaliyeDetay.FasonHamIade,
    Ham.CiftBoyaTransferKodu, Ham.PolyesterTransferKodu, Ham.NaylonTransferKodu, Ham.CiftBoyaTransferKoduMt, Ham.PolyesterTransferKoduMt, Ham.NaylonTransferKoduMt,
    Ham.SACiftBoyaTransferKodu, Ham.SAPolyesterTransferKodu, Ham.SANaylonTransferKodu, Ham.SACiftBoyaTransferKoduMt, Ham.SAPolyesterTransferKoduMt, Ham.SANaylonTransferKoduMt from BoyaIrsaliyeDetay
    left join Parti on Parti.PartiKayitId = BoyaIrsaliyeDetay.PartiKayitId
    left join PartiStok on PartiStok.PartiSinifId = BoyaIrsaliyeDetay.PartiSinifId
    Left Join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    Left Join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
    Left Join BaskiTipi on BaskiTipi.BaskiTipiID = BoyaSiparisDetay.BaskiTipiID
    Left Join BoyaFaturaDetay on BoyaFaturaDetay.BoyaIrsaliyeDetayId = BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId
    where BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = :ABoyaIrsaliyeDetayId
    into :PartiSinifId, :Fason, :HamStokbirimId, :IslemGrubuId,
    :BoyaliTransferKodu, :SekerKasarTransferKodu,
    :OptikBeyazTransferKodu, :YikamaTransferKodu, :SABoyaliTransferKodu, :SASekerKasarTransferKodu,
    :SAOptikBeyazTransferKodu, :SAYikamaTransferKodu,
    :AlisTransferKoduMt, :BoyaliTransferKoduMt, :SekerKasarTransferKoduMt,
    :OptikBeyazTransferKoduMt, :YikamaTransferKoduMt, :SABoyaliTransferKoduMt, :SASekerKasarTransferKoduMt,
    :SAOptikBeyazTransferKoduMt, :SAYikamaTransferKoduMt, :AlisBoyaliTransferKodu,
    :AlisSekerKasarTransferKodu, :AlisOptikBeyazTransferKodu, :AlisYikamaTransferKodu,
    :HamTipiId, :FDijitalBaski, :ImalatTipiId, :FPartiKayitID, :RafDetayID, :HamSatisTransferKodu, :FasonHamIade,
    :CiftBoyaTransferKodu, :PolyesterTransferKodu, :NaylonTransferKodu, :CiftBoyaTransferKoduMt, :PolyesterTransferKoduMt, :NaylonTransferKoduMt,
    :SACiftBoyaTransferKodu, :SAPolyesterTransferKodu, :SANaylonTransferKodu, :SACiftBoyaTransferKoduMt, :SAPolyesterTransferKoduMt, :SANaylonTransferKoduMt;

    if (PartiSinifId is not null) then
    begin
      if (Fason = 0) then
      begin
        Select first 1 coalesce(HamStokFatura.FirmaID, HamStokIrsaliye.FirmaID) as GirisFirmaID, HamStokMaster.HamStokHareketTipiID, HamStokMaster.EskiPartiSinifID from PartiStokDetay
        left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
        left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
        left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
        left join HamStokFatura on HamStokFatura.HamStokFaturaId = HamStokMaster.HamStokFaturaId
        where PartiStokDetay.PartiSinifID = :PartiSinifID
        and (coalesce(HamStokFatura.FirmaID, HamStokIrsaliye.FirmaID) = :FFirmaKodumuz or
             coalesce(HamStokFatura.FirmaID, HamStokIrsaliye.FirmaID) = :FOrmeFirmaKodumuz)
        Group By GirisFirmaID, HamStokMaster.HamStokHareketTipiID, HamStokMaster.EskiPartiSinifID into :FGirisFirmaID, :FHamStokHareketTipiID, :FEskiPartiSinifID;

        if (FGirisFirmaID is Null) then
        begin
          Select first 1 coalesce(HamStokFatura.FirmaID, HamStokIrsaliye.FirmaID) as GirisFirmaID, HamStokMaster.HamStokHareketTipiID, HamStokMaster.EskiPartiSinifID from PartiStokDetay
            left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
            left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
            left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
            left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
            left join HamStokFatura on HamStokFatura.HamStokFaturaId = HamStokMaster.HamStokFaturaId
            where PartiStokDetay.PartiSinifID = :PartiSinifID
            Group By GirisFirmaID, HamStokMaster.HamStokHareketTipiID, HamStokMaster.EskiPartiSinifID into :FGirisFirmaID, :FHamStokHareketTipiID, :FEskiPartiSinifID;
        end


        if (FHamStokHareketTipiID = 1 and FEskiPartiSinifID is not Null) then
        begin
          Select first 1 coalesce(HamStokFatura.FirmaID, HamStokIrsaliye.FirmaID) as GirisFirmaID from PartiStokDetay
          left join RafDetay on RafDetay.RafDetayID = PartiStokDetay.RafDetayID
          left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
          left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
          left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
          left join HamStokFatura on HamStokFatura.HamStokFaturaId = HamStokMaster.HamStokFaturaId
          where PartiStokDetay.PartiSinifID = :FEskiPartiSinifID and HamStokMaster.HamStokHareketTipiID = 0
          and (coalesce(HamStokFatura.FirmaID, HamStokIrsaliye.FirmaID) = :FFirmaKodumuz or
               coalesce(HamStokFatura.FirmaID, HamStokIrsaliye.FirmaID) = :FOrmeFirmaKodumuz)
          Group By GirisFirmaID into :FGirisFirmaID;

          Select BoyaSiparisDetay.IslemGrubuID, Parti.ImalatTipiID, PartiStok.PartiKayitId, BaskiTipi.DijitalBaski  from PartiStok
          Left Join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
          Left Join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
          Left Join BaskiTipi on BaskiTipi.BaskiTipiID = BoyaSiparisDetay.BaskiTipiID
          where PartiStok.PartiSinifID = :FEskiPartiSinifID
          into :IslemGrubuID, :ImalatTipiID, :FPartiKayitId, :FDijitalBaski;
        end

        if (HamStokbirimId = 1) then /*Metre*/
        begin
          if (FGirisFirmaID is not Null) then
          begin
            if (IslemGrubuId = 0) then
              TransferKodu = OptikBeyazTransferKoduMt;
            else if (IslemGrubuId = 2) then
              TransferKodu = YikamaTransferKoduMt;
            else if (IslemGrubuId = 3) then
              TransferKodu = SekerKasarTransferKoduMt;
            else if (IslemGrubuId = 5) then
              TransferKodu = CiftBoyaTransferKoduMt;
            else if (IslemGrubuId = 7) then
              TransferKodu = PolyesterTransferKoduMt;
            else if (IslemGrubuId = 8) then
              TransferKodu = NaylonTransferKoduMt;
            else
              TransferKodu = BoyaliTransferKoduMt;
          end
          else
          begin
            if (IslemGrubuId = 0) then
              TransferKodu = SAOptikBeyazTransferKoduMt;
            else if (IslemGrubuId = 2) then
              TransferKodu = SAYikamaTransferKoduMt;
            else if (IslemGrubuId = 3) then
              TransferKodu = SASekerKasarTransferKoduMt;
            else if (IslemGrubuId = 5) then
              TransferKodu = SACiftBoyaTransferKoduMt;
            else if (IslemGrubuId = 7) then
              TransferKodu = SAPolyesterTransferKoduMt;
            else if (IslemGrubuId = 8) then
              TransferKodu = SANaylonTransferKoduMt;
            else
              TransferKodu = SABoyaliTransferKoduMt;
          end
        end
        else /*Metre degil*/
        begin
          if (FGirisFirmaID is not Null) then
          begin
            if (IslemGrubuId = 0) then
              TransferKodu = OptikBeyazTransferKodu;
            else if (IslemGrubuId = 2) then
              TransferKodu = YikamaTransferKodu;
            else if (IslemGrubuId = 3) then
              TransferKodu = SekerKasarTransferKodu;
            else if (IslemGrubuId = 5) then
              TransferKodu = CiftBoyaTransferKodu;
            else if (IslemGrubuId = 7) then
              TransferKodu = PolyesterTransferKodu;
            else if (IslemGrubuId = 8) then
              TransferKodu = NaylonTransferKodu;
            else
              TransferKodu = BoyaliTransferKodu;
          end
          else
          begin
            if (IslemGrubuId = 0) then
              TransferKodu = SAOptikBeyazTransferKodu;
            else if (IslemGrubuId = 2) then
              TransferKodu = SAYikamaTransferKodu;
            else if (IslemGrubuId = 3) then
              TransferKodu = SASekerKasarTransferKodu;
            else if (IslemGrubuId = 5) then
              TransferKodu = SACiftBoyaTransferKodu;
            else if (IslemGrubuId = 7) then
              TransferKodu = SAPolyesterTransferKodu;
            else if (IslemGrubuId = 8) then
              TransferKodu = SANaylonTransferKodu;
            else
              TransferKodu = SABoyaliTransferKodu;
          end
        end

        if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
        begin
          HareketSekli = 'MBOYALIIPLIK';
          TransferKodu = BoyaliIplikMSatisTipi || ' ' || TransferKodu;
        end
        else
        begin
          if (ImalatTipiId = 1) then /* baski ise */
          begin
            Select Count(*) from PartiAsamalari
              left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
              where PartiAsamalari.PartiKayitID = :FPartiKayitID and Asama.BaskiAsamasimi = 1
            into :KayitSayisi;

            if (KayitSayisi > 0) then
            begin
              if (FDijitalBaski = 1) then
              begin
                HareketSekli = 'MDIJBASKILIKUMAS';
                TransferKodu = DijitalBaskiliKumasMSatisTipi || ' ' || TransferKodu;
              end
              else
              begin
                HareketSekli = 'MBASKILIKUMAS';
                TransferKodu = BaskiliKumasMSatisTipi || ' ' || TransferKodu;
              end
            end
            else /* Baskının düz boyası için*/
            begin
              HareketSekli = 'MBOYALIKUMAS';
              TransferKodu = BoyaliKumasMSatisTipi || ' ' || TransferKodu;
            end
          end
          else /* boya ise */
          begin
            HareketSekli = 'MBOYALIKUMAS';
            TransferKodu = BoyaliKumasMSatisTipi || ' ' || TransferKodu;
          end
        end
      end /*fason*/
      else
      begin
        if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
        begin
          if (HamStokbirimId = 1) then Select FasonIplikMtTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuId into :TransferKodu; /*Metre ise*/
          else Select FasonIplikTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuId into :TransferKodu;
          HareketSekli = 'MFBOYALIIPLIK';
        end
        else if (ImalatTipiId = 1) then /* baski ise */
        begin
          Select Count(*) from PartiAsamalari
            left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId
            where PartiAsamalari.PartiKayitID = :FPartiKayitID and Asama.BaskiAsamasimi = 1
          into :KayitSayisi;
          if (KayitSayisi > 0) then
          begin
            if (HamStokbirimId = 1) then Select iif(:FDijitalBaski = 1, DBaskiFasonKumasMtTransferKodu, BaskiFasonKumasMtTransferKodu) from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuId into :TransferKodu; /*Metre ise*/
            else Select iif(:FDijitalBaski = 1, DijBaskiFasonKumasTransferKodu, BaskiFasonKumasTransferKodu) from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuId into :TransferKodu;

            if (FDijitalBaski = 1) then HareketSekli = 'MFDIJBASKILIKUMAS';
            else HareketSekli = 'MFBASKILIKUMAS';
          end
          else /* Baskının düz boyası için*/
          begin
            if (HamStokbirimId = 1) then Select FasonKumasMtTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuId into :TransferKodu; /*Metre ise*/
            else Select FasonKumasTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuId into :TransferKodu;
            HareketSekli = 'MFBOYALIKUMAS';
          end
        end
        else /* boya ise */
        begin
          if (HamStokbirimId = 1) then Select FasonKumasMtTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuId into :TransferKodu; /*Metre ise*/
          else Select FasonKumasTransferKodu from IslemGrubu where IslemGrubu.IslemGrubuID = :IslemGrubuId into :TransferKodu;
          HareketSekli = 'MFBOYALIKUMAS';
        end
      end /* fason değil */
    end /*partisinif var*/
    else if (FasonHamIade = 1) then
    begin
      TransferKodu = Null;
      HareketSekli = Null;
    end
    else if (RafDetayID is not null) then
    begin
      TransferKodu = Null;
      HareketSekli = Null;

      if (RafDetayId is not null) then
        select HamStokDetayId from RafDetay where RafDetayId = :RafDetayId into :AHamStokDetayId;

      Select HamStokMaster.DevirPartiYil, HamStokMaster.DevirPartiID, HamStokMaster.IslemGrubuID,
      HamStokMaster.DesenID from HamStokDetay
      left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
      where HamStokDetay.HamStokDetayId = :AHamStokDetayId
      into :FDevirPartiYil, :FDevirPartiID, :FDevirIslemGrubuID, :FDevirDesenID;

      if (FDevirPartiYil > 0 and FDevirPartiID > 0) then
      begin
        if (HamStokBirimId = 1) then /*Metre*/
        begin
          if (FDevirIslemGrubuID = 0) then
            TransferKodu = OptikBeyazTransferKoduMt;
          else if (FDevirIslemGrubuID = 2) then
            TransferKodu = YikamaTransferKoduMt;
          else if (FDevirIslemGrubuID = 3) then
            TransferKodu = SekerKasarTransferKoduMt;
          else if (FDevirIslemGrubuID = 5) then
            TransferKodu = CiftBoyaTransferKoduMt;
          else if (FDevirIslemGrubuID = 7) then
            TransferKodu = PolyesterTransferKoduMt;
          else if (FDevirIslemGrubuID = 8) then
            TransferKodu = NaylonTransferKoduMt;
          else
            TransferKodu = BoyaliTransferKoduMt;
        end
        else
        begin
          if (FDevirIslemGrubuID = 0) then
            TransferKodu = OptikBeyazTransferKodu;
          else if (FDevirIslemGrubuID = 2) then
            TransferKodu = YikamaTransferKodu;
          else if (FDevirIslemGrubuID = 3) then
            TransferKodu = SekerKasarTransferKodu;
          else if (FDevirIslemGrubuID = 5) then
            TransferKodu = CiftBoyaTransferKodu;
          else if (FDevirIslemGrubuID = 7) then
            TransferKodu = PolyesterTransferKodu;
          else if (FDevirIslemGrubuID = 8) then
            TransferKodu = NaylonTransferKodu;
          else
            TransferKodu = BoyaliTransferKodu;
        end

        if (HamTipiId = hstIplik or HamTipiId = hstElyaf or HamTipiId = hstTelef) then
        begin
          TransferKodu = BoyaliIplikMSatisTipi || ' ' || TransferKodu;
          HareketSekli = 'MBOYALIIPLIK';
        end
        else
        begin
          if ( FDevirDesenID is not null ) then /* baski ise */
          begin
            TransferKodu = BaskiliKumasMSatisTipi || ' ' || TransferKodu;
            HareketSekli = 'MBASKILIKUMAS';
          end
          else /* boya ise */
          begin
            TransferKodu = BoyaliKumasMSatisTipi || ' ' || TransferKodu;
            HareketSekli = 'MBOYALIKUMAS';
          end
        end
      end
      else
      begin
        Select HamStokMaster.Renk from HamStokDetay
          left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
          where HamStokDetay.HamStokDetayId = :AHamStokDetayId into :GirisRenk;

        if (HamStokBirimId = 1) then /*Metre*/
        begin
