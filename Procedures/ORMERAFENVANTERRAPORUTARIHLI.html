    HAMADI VARCHAR(255),

    RENKADI VARCHAR(50),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    DESENID VARCHAR(20),

    VARYANTADI VARCHAR(30),
    RAFID VARCHAR(10),
    URETIMPARTINO VARCHAR(30),
    CIKISPARTINO VARCHAR(30),
    LOT VARCHAR(150),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    URETICIADI VARCHAR(40),
    URETICIFIRMAID VARCHAR(20),
    URETICIFIRMAADI VARCHAR(100),
    ORMEMAKINAID VARCHAR(10),
    KALITENO INTEGER,
    STOKFIRMAID VARCHAR(20),
    STOKFIRMAADI VARCHAR(100),
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    KALITEACIKLAMATIPIID INTEGER,
    KALITEACIKLAMATIPIADI VARCHAR(20),
    MAMULPROSESID INTEGER,
    MAMULPROSESADI VARCHAR(255),
    TAMIR INTEGER,
    TAMIRSTR VARCHAR(4),
    SAKAT INTEGER,
    SAKATSTR VARCHAR(4),
    ESKIPARTINOSTR VARCHAR(30),
    KAPADEDI INTEGER,
    BRUTMIKTAR NUMERIC(12,3),
    NETMIKTAR NUMERIC(12,3),
    KALANKAPADEDI INTEGER,
    BRUTKALANMIKTAR NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    BRUTKALANMIKTARTL NUMERIC(12,3),
    KALANMIKTARTL NUMERIC(12,3),
    HARMANYIL INTEGER,
    HARMANID INTEGER,
    HARMANIDSTR VARCHAR(30),
    HARMANADI VARCHAR(50),
    HAMSATISTRANSFERKODU VARCHAR(55),
    SATISTRANSFERKODU VARCHAR(55),
    FASONTRANSFERKODU VARCHAR(55),
    FASONORMETRANSFERKODU VARCHAR(55),
    FASONBUKUMTRANSFERKODU VARCHAR(55),
    FASONARTISATORMETRANSFERKODU VARCHAR(55),
    YARIMAMULTRANSFERKODU VARCHAR(55),
    YARIMAMULSATISTRANSFERKODU VARCHAR(55),
    ALISTRANSFERKODU VARCHAR(55),
    RENKLIALISTRANSFERKODU VARCHAR(55),
    SATISIADETRANSFERKODU VARCHAR(55),
    DEKONTTRANSFERKODU VARCHAR(55),

    SEKERKASARTRANSFERKODU VARCHAR(55),

    YIKAMATRANSFERKODU VARCHAR(55),

    SAOPTIKBEYAZTRANSFERKODU VARCHAR(55),
    SAYIKAMATRANSFERKODU VARCHAR(55),
    ALISBOYALITRANSFERKODU VARCHAR(55),
    ALISSEKERKASARTRANSFERKODU VARCHAR(55),
    ALISOPTIKBEYAZTRANSFERKODU VARCHAR(55),
    ALISYIKAMATRANSFERKODU VARCHAR(55),
    ALISBASKILITRANSFERKODU VARCHAR(55),
    HAMALISFIYATI DOUBLE PRECISION,
    HAMALISFIYATIDOVIZID DOVIZSIZE)
AS
declare variable BoyaSiparisYil integer;
declare variable BoyaSiparisId  integer;
declare variable EskiPartiYil   integer;
declare variable EskiPartiNo    integer;
declare variable CikisKapAdedi                 integer;
declare variable hstIplik       integer;
declare variable hstLycra       integer;
declare variable BrutMiktarRaf  double precision;
declare variable NetMiktarRaf   double precision;
declare variable BrutCikisMiktar               Numeric(12,3);
declare variable NetCikisMiktar                Numeric(12,3);
declare variable BrutCikisMiktarRaf double precision;
declare variable NetCikisMiktarRaf  double precision;

declare variable BrutMiktarTL         double precision;
declare variable NetMiktarTL          double precision;
declare variable BrutMiktarRafTL      double precision;
declare variable NetMiktarRafTL       double precision;
declare variable BrutCikisMiktarTL    Double Precision;
declare variable NetCikisMiktarTL     Double Precision;
declare variable BrutCikisMiktarRafTL Double Precision;
declare variable NetCikisMiktarRafTL  double precision;

begin
  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  For Select OrmeRafDetay.OrmeRafId, OrmeRaf.HamId, Ham.Adi, OrmeRaf.StokGirisNo, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.FasonStok, OrmeRaf.RafId,
             OrmeRaf.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, OrmeRaf.UretimPartiNo, OrmeRaf.CikisPartiNo, OrmeRaf.StokFirmaId, StokFirma.Adi,
             OrmeRaf.KaliteNo, OrmeRaf.Lot, OrmeRaf.MelanjKodu, OrmeRaf.FlamKodu, Uretici.Adi,
             OrmeRaf.UreticiFirmaId, UreticiFirma.Adi, OrmeRaf.OrmeMakinaId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, OrmeRaf.BoyaSiparisDetayId, OrmeRaf.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo,
             BoyaSiparis.FirmaId, SiparisFirma.Adi, OrmeRaf.EskiPartiYil, OrmeRaf.EskiPartiNo, OrmeRaf.Tamir, OrmeRaf.Sakat,
             OrmeRaf.KaliteAciklamaTipiId, KaliteAciklamaTipi.Adi, OrmeRaf.MamulProsesId, MamulProses.Adi,
             Harman.HarmanYil, Harman.HarmanId, Harman.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4,
             Ham.HamSatisTransferKodu, Ham.SatisTransferKodu, Ham.FasonTransferKodu, Ham.FasonOrmeTransferKodu, Ham.FasonBukumTransferKodu, Ham.FasonArtiSatOrmeTransferKodu,
             Ham.YariMamulTransferKodu, Ham.YariMamulSatisTransferKodu, Ham.AlisTransferKodu, Ham.RenkliAlisTransferKodu, Ham.SatisIadeTransferKodu, Ham.DekontTransferKodu,
             Ham.AlisIadeTransferKodu, Ham.BoyaliTransferKodu, Ham.SekerKasarTransferKodu, Ham.OptikBeyazTransferKodu, Ham.YikamaTransferKodu, Ham.SABoyaliTransferKodu,
             Ham.SASekerKasarTransferKodu, Ham.SAOptikBeyazTransferKodu, Ham.SAYikamaTransferKodu, Ham.AlisBoyaliTransferKodu, Ham.AlisSekerKasarTransferKodu,
             Ham.AlisOptikBeyazTransferKodu, Ham.AlisYikamaTransferKodu, Ham.AlisBaskiliTransferKodu,
             Sum(StokHareket.BrutMiktar), Sum(StokHareket.Miktar), Sum(StokHareket.BrutMiktar * StokHareket.BirimMaliyet), Sum(StokHareket.Miktar * StokHareket.BirimMaliyet), Ham.HamTipiID,
             Ham.AlisFiyati,Ham.AlisFiyatiDovizId from StokHareket
    left join Ham On Ham.HamID = StokHareket.HamId
    left join OrmeRafDetay On OrmeRafDetay.StokHareketID = StokHareket.StokHareketId
    left join OrmeRaf On OrmeRaf.OrmeRafID = OrmeRafDetay.OrmeRafId
    left join Renk on Renk.RenkId = OrmeRaf.RenkId
    left join uretici on Uretici.UreticiId = OrmeRaf.UreticiID
    left join Firma StokFirma on StokFirma.Firmaid = OrmeRaf.StokFirmaid
    left join Firma UreticiFirma on UreticiFirma.Firmaid = OrmeRaf.UreticiFirmaid
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeRaf.BoyaSiparisKayitId
    left join Firma SiparisFirma on SiparisFirma.Firmaid = BoyaSiparis.FirmaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmeRaf.OrmeSiparisDetayId
    left join Desen on Desen.DesenId = OrmeRaf.DesenId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = OrmeRaf.DesenVaryantId
    left join KaliteAciklamaTipi on KaliteAciklamaTipi.KaliteAciklamaTipiId = OrmeRaf.KaliteAciklamaTipiId
    left join MamulProses on MamulProses.MamulProsesId = OrmeRaf.MamulProsesId
    Left join IplikLot on IplikLot.Lot = StokHareket.Lot and IplikLot.HamID = StokHareket.HamId
    Left join Harman on Harman.HarmanKayitId = IplikLot.HarmanKayitId
    where StokHareket.Tarih <= :ABitisTarihi and StokHareket.GirisCikis = 1 and OrmeRafDetay.StokHareketID is not null and
         (StokHareket.HamId = :AHamId or :AHamId is null) and
         ( (:AHamTipiID = 0 and Ham.HamTipiId = :hstIplik) or
           (:AHamTipiID = 1 and Ham.HamTipiId = :hstLycra) or
           (:AHamTipiID = 2 and (Ham.HamTipiId <> :hstLycra and Ham.HamTipiId <> :hstIplik) ) or
            :AHamTipiID is null ) and
         (Ham.KumasCinsiId = :AKumasCinsiId or :AKumasCinsiId is null) and
         (Ham.IplikCinsiId = :AIplikCinsiId or :AIplikCinsiId is null) and
         (Ham.KalinlikId = :AKalinlikId or :AKalinlikId is null)
    group by OrmeRafDetay.OrmeRafId, OrmeRaf.HamId, Ham.Adi, OrmeRaf.StokGirisNo, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.FasonStok, OrmeRaf.RafId,
             OrmeRaf.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, OrmeRaf.UretimPartiNo, OrmeRaf.CikisPartiNo, OrmeRaf.StokFirmaId, StokFirma.Adi,
             OrmeRaf.KaliteNo, OrmeRaf.Lot, OrmeRaf.MelanjKodu, OrmeRaf.FlamKodu, Uretici.Adi,
             OrmeRaf.UreticiFirmaId, UreticiFirma.Adi, OrmeRaf.OrmeMakinaId,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisID, OrmeRaf.BoyaSiparisDetayId, OrmeRaf.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo,
             BoyaSiparis.FirmaId, SiparisFirma.Adi, OrmeRaf.EskiPartiYil, OrmeRaf.EskiPartiNo, OrmeRaf.Tamir, OrmeRaf.Sakat,
             OrmeRaf.KaliteAciklamaTipiId, KaliteAciklamaTipi.Adi, OrmeRaf.MamulProsesId, MamulProses.Adi,
             Harman.HarmanYil, Harman.HarmanId, Harman.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4,
             Ham.HamSatisTransferKodu, Ham.SatisTransferKodu, Ham.FasonTransferKodu, Ham.FasonOrmeTransferKodu, Ham.FasonBukumTransferKodu, Ham.FasonArtiSatOrmeTransferKodu,
             Ham.YariMamulTransferKodu, Ham.YariMamulSatisTransferKodu, Ham.AlisTransferKodu, Ham.RenkliAlisTransferKodu, Ham.SatisIadeTransferKodu, Ham.DekontTransferKodu,
             Ham.AlisIadeTransferKodu, Ham.BoyaliTransferKodu, Ham.SekerKasarTransferKodu, Ham.OptikBeyazTransferKodu, Ham.YikamaTransferKodu, Ham.SABoyaliTransferKodu,
             Ham.SASekerKasarTransferKodu, Ham.SAOptikBeyazTransferKodu, Ham.SAYikamaTransferKodu, Ham.AlisBoyaliTransferKodu, Ham.AlisSekerKasarTransferKodu,
             Ham.AlisOptikBeyazTransferKodu, Ham.AlisYikamaTransferKodu, Ham.AlisBaskiliTransferKodu, Ham.HamTipiID, Ham.AlisFiyati,Ham.AlisFiyatiDovizId
    into :OrmeRafId, :HamId, :HamAdi, :StokGirisNo, :RenkId, :RenkAdi, :Fason, :RafId, :DesenId, :DesenAdi, :VaryantAdi, :UretimPartiNo, :CikisPartiNo,  :StokFirmaId, :StokFirmaAdi,
         :KaliteNo, :Lot, :MelanjKodu, :FlamKodu,  :UreticiAdi, :UreticiFirmaId, :UreticiFirmaAdi, :OrmeMakinaId,
         :BoyaSiparisYil, :BoyaSiparisID, :BoyaSiparisDetayId, :OrmeSiparisDetayId, :SiparisSiraNo, :SiparisFirmaId, :SiparisFirmaAdi,
         :EskiPartiYil, :EskiPartiNo, :Tamir, :Sakat,
         :KaliteAciklamaTipiId, :KaliteAciklamaTipiAdi, :MamulProsesId, :MamulProsesAdi, :HarmanYil, :HarmanId, :HarmanAdi, :HamAdi2, :HamAdi3, :HamAdi4,
         :HamSatisTransferKodu, :SatisTransferKodu, :FasonTransferKodu, :FasonOrmeTransferKodu, :FasonBukumTransferKodu, :FasonArtiSatOrmeTransferKodu,
         :YariMamulTransferKodu, :YariMamulSatisTransferKodu, :AlisTransferKodu, :RenkliAlisTransferKodu, :SatisIadeTransferKodu, :DekontTransferKodu,
         :AlisIadeTransferKodu, :BoyaliTransferKodu, :SekerKasarTransferKodu, :OptikBeyazTransferKodu, :YikamaTransferKodu, :SABoyaliTransferKodu,
         :SASekerKasarTransferKodu, :SAOptikBeyazTransferKodu, :SAYikamaTransferKodu, :AlisBoyaliTransferKodu, :AlisSekerKasarTransferKodu,
         :AlisOptikBeyazTransferKodu, :AlisYikamaTransferKodu, :AlisBaskiliTransferKodu,
         :BrutMiktar, :NetMiktar, :BrutMiktarTL, :NetMiktarTL, :HamTipiID, :HamAlisFiyati, :HamAlisFiyatiDovizId do
  begin
    /*StokHareket tablosunda OrmeRafTransferFişleri yer Almadığından...*/

    Select Sum(BrutKilo), Sum(Kilo), Sum(BrutKilo * BirimMaliyet), Sum(Kilo* BirimMaliyet) From OrmeRafTransferFisiDetay
      left join OrmeRafTransferFisi on OrmeRafTransferFisi.OrmeRafTransferFisiID = OrmeRafTransferFisiDetay.OrmeRafTransferFisiID
    where OrmeRafTransferFisi.FisTarihi <= :ABitisTarihi and OrmeRafTransferFisiDetay.YeniOrmeRafID = :OrmeRafID
    into :BrutMiktarRaf, :NetMiktarRaf, :BrutMiktarRafTL, :NetMiktarRafTL;

    BrutMiktar = BrutMiktar + coalesce(BrutMiktarRaf, 0);
    NetMiktar  = NetMiktar + coalesce(NetMiktarRaf, 0);

    BrutMiktarTL = coalesce(BrutMiktarTL, 0) + coalesce(BrutMiktarRafTL, 0);
    NetMiktarTL  = coalesce(NetMiktarTL, 0) + coalesce(NetMiktarRafTL, 0);

    BrutCikisMiktar = Null;
    NetCikisMiktar  = Null;
    CikisKapAdedi   = Null;

    BrutCikisMiktarTL = Null;
    NetCikisMiktarTL  = Null;

    Select Sum(KapAdedi), Sum(BrutMiktar), Sum(Miktar), Sum(BrutMiktar * BirimMaliyet ), Sum(Miktar * BirimMaliyet ) from StokHareket
      where StokHareket.OrmeRafID = :OrmeRafID and StokHareket.Tarih <= :ABitisTarihi and StokHareket.GirisCikis = -1
      into :CikisKapAdedi, :BrutCikisMiktar, :NetCikisMiktar, :BrutCikisMiktarTL, :NetCikisMiktarTL;

    /*StokHareket tablosunda OrmeRafTransferFişleri yer Almadığından...*/
    Select Sum(BrutKilo), Sum(Kilo), Sum(BrutKilo * BirimMaliyet), Sum(Kilo * BirimMaliyet) From OrmeRafTransferFisiDetay
      left join OrmeRafTransferFisi on OrmeRafTransferFisi.OrmeRafTransferFisiID = OrmeRafTransferFisiDetay.OrmeRafTransferFisiID
    where OrmeRafTransferFisi.FisTarihi <= :ABitisTarihi and OrmeRafTransferFisiDetay.OrmeRafID = :OrmeRafID
    into :BrutCikisMiktarRaf, :NetCikisMiktarRaf, :BrutCikisMiktarRafTL, :NetCikisMiktarRafTL;

    BrutCikisMiktar  = coalesce(BrutCikisMiktar, 0) + coalesce(BrutCikisMiktarRaf, 0);
    NetCikisMiktar   = coalesce(NetCikisMiktar, 0) + coalesce(NetCikisMiktarRaf, 0);

    BrutCikisMiktarTL  = coalesce(BrutCikisMiktarTL, 0) + coalesce(BrutCikisMiktarRafTL, 0);
    NetCikisMiktarTL   = coalesce(NetCikisMiktarTL, 0) + coalesce(NetCikisMiktarRafTL, 0);

    BrutKalanMiktar = BrutMiktar - coalesce(BrutCikisMiktar, 0);
    KalanMiktar     = NetMiktar - coalesce(NetCikisMiktar, 0);
    KalanKapAdedi   = KapAdedi - coalesce(CikisKapAdedi, 0);

    BrutKalanMiktarTL = coalesce(BrutMiktarTL, 0) - coalesce(BrutCikisMiktarTL, 0);
    KalanMiktarTL     = coalesce(NetMiktarTL, 0) - coalesce(NetCikisMiktarTL, 0);

    if (KalanMiktar <> 0) then
    begin
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      if (HarmanYil is not null) then
        HarmanIdStr = '(' || EgemenIntegerToStr(HarmanYil) || ') '  || EgemenIntegerToStr(HarmanId)  || ' ';
      else
        HarmanIdStr = '';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisYil is not null) then
        BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';


      if (EskiPartiNo is not Null) then
        EskiPartiNoStr = '(' || EgemenIntegerToStr(EskiPartiYil) || ') ' || EgemenIntegerToStr(EskiPartiNo);
      else
        EskiPartiNoStr = '';

      if (Tamir = 1) then TamirStr = 'Evet';
      else TamirStr = '';
      if (Sakat = 1) then SakatStr = 'Evet';
      else SakatStr = '';
      suspend;
    end
  end
