CREATE PROCEDURE PARTISTOKBOLEREKSEVKET(
    APARTISTOKTARTIID INTEGER,
    ABOBINSAYISI INTEGER,
    AKILO BOYAMIKTARSIZE,
    ABRUTKILO BOYAMIKTARSIZE,
    AMETRE BOYAMIKTARSIZE,
    ABRUTMETRE BOYAMIKTARSIZE,
    AKULLANICIID INTEGER,
    ABOYAIRSALIYEID INTEGER,
    AFIRMAID FIRMAKODUSIZE,
    ABARKODID VARCHAR(20))
AS
declare variable IK                   varchar(2);
declare variable YeniBarkodId         varchar(20);
declare variable PsKaliteKontrolID    integer;
begin
  IK = SubString(ABarkodId from 1 for 2);
  if (IK = 'K-') then
  begin
    select PartiStokTartiBol.YeniBarkodId, PartiStokTartiBol.PsKaliteKontrolID
    from PartiStokTartiBol(:APartiStokTartiId, :AKilo, :AKullaniciId, 0) into :YeniBarkodId, :PsKaliteKontrolID;

    if (PsKaliteKontrolID is not null) then /*iade degil, normal akis islemesi*/
      execute procedure PartiStokBarkodIleSevkEt(:ABoyaIrsaliyeId, :AFirmaId, 'K-'||:PsKaliteKontrolID, :AKullaniciId); /*eski hali yeni miktar oluyor*/
    else
      execute procedure PartiStokBarkodIleSevkEt(:ABoyaIrsaliyeId, :AFirmaId, :YeniBarkodId, :AKullaniciId);
  end
  else
  begin
    execute procedure PartiStokIPBolerekSevkEt(:APartiStokTartiId, :ABobinSayisi, :AKilo, :ABrutKilo, :AMetre, :ABrutMetre, :AKullaniciId, :ABoyaIrsaliyeId, :AFirmaId);
  end
end
