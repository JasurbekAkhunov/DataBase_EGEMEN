CREATE PROCEDURE YARIMAMULPARTIHAMSTOK(
    APARTISINIFID INTEGER)
RETURNS (
    HAMSTOKMASTERIDSTR VARCHAR(50),
    HAMSTOKDETAYIDSTR VARCHAR(50),
    LOTSTR VARCHAR(50),
    ENSTR VARCHAR(50),
    GRM2STR VARCHAR(50),
    PUSSTR VARCHAR(50),
    FEINESTR VARCHAR(50),
    IRSALIYENOSTR VARCHAR(50),
    IRSALIYEGRUPIDSTR VARCHAR(50),
    IRSALIYEGRUPADISTR VARCHAR(70),
    IRSALIYETARIHISTR VARCHAR(50),
    FIRMAIDSTR VARCHAR(70),
    FIRMAADISTR VARCHAR(100),
    ORMECIADISTR VARCHAR(100),
    ORMECIMAKINANOSTR VARCHAR(50),
    TUPACIKENMAYLI INTEGER,
    MUSTERIREFERANSNOSTR VARCHAR(70),
    MUSTERIPARTINOSTR VARCHAR(70),
    RENKSTR VARCHAR(70),
    TAMIR VARCHAR(5),
    TAMIRTIPIADI VARCHAR(20),
    ORTALAMAMETRETULAGIRLIGI INTEGER,
    KENARKONSTRUKSIYONTIPIADISTR VARCHAR(50))
AS
declare variable HamStokMasterId       integer;
declare variable HamStokDetayId        integer;
declare variable Lot                   varchar(15);
declare variable En                    integer;
declare variable Grm2                  integer;
declare variable Pus                   integer;
declare variable Feine                 integer;
declare variable IrsaliyeNo            IrsaliyeNoSize;
declare variable IrsaliyeGrupId        varchar(4);
declare variable IrsaliyeGrupAdi       varchar(30);
declare variable IrsaliyeTarihi        timestamp;
declare variable FirmaId               varchar(20);
declare variable FirmaAdi              varchar(100);
declare variable OrmeciAdi             varchar(30);
declare variable OrmeciMakinaNo        varchar(10);
declare variable MusteriReferansNo     varchar(30);
declare variable MusteriPartiNo        varchar(20);
declare variable Renk                  varchar(20);
declare variable KenarKonstruksiyonTipiAdi varchar(20);

declare variable FHamStokMasterIdStr            varchar(200);
declare variable FHamStokDetayIdStr             varchar(200);
declare variable FLotStr                        varchar(200);
declare variable FEnStr                         varchar(200);
declare variable FGrm2Str                       varchar(200);
declare variable FPusStr                        varchar(200);
declare variable FFeineStr                      varchar(200);
declare variable FIrsaliyeNoStr                 varchar(200);
declare variable FIrsaliyeGrupIdStr             varchar(200);
declare variable FIrsaliyeGrupAdiStr            varchar(200);
declare variable FIrsaliyeTarihiStr             varchar(200);
declare variable FFirmaIdStr                    varchar(200);
declare variable FFirmaAdiStr                   varchar(200);
declare variable FOrmeciAdiStr                  varchar(200);
declare variable FOrmeciMakinaNoStr             varchar(200);
declare variable FMusteriReferansNoStr          varchar(200);
declare variable FMusteriPartiNoStr             varchar(200);
declare variable FRenkStr                       varchar(200);
declare variable FKenarKonstruksiyonTipiAdi     varchar(200);

declare variable Str                            varchar(70);
declare variable TamirX                         integer;
declare variable HamStokSayisi                  integer;
begin
  HamStokMasterIdStr = '';
  HamStokDetayIdStr = '';
  LotStr = '';
  EnStr = '';
  Grm2Str = '';
  PusStr  = '';
  FeineStr = '';
  IrsaliyeNoStr = '';
  IrsaliyeTarihiStr = '';
  IrsaliyeGrupIdStr = '';
  IrsaliyeGrupAdiStr = '';
  OrmeciAdiStr = '';
  OrmeciMakinaNoStr = '';
  FirmaIdStr = '';
  FirmaAdiStr = '';
  MusteriReferansNoStr = '';
  MusteriPartiNoStr = '';
  KenarKonstruksiyonTipiAdiStr = '';
  RenkStr = '';


  FHamStokMasterIdStr = '';
  FHamStokDetayIdStr = '';
  FLotStr = '';
  FEnStr = '';
  FGrm2Str = '';
  FPusStr  = '';
  FFeineStr = '';
  FIrsaliyeNoStr = '';
  FIrsaliyeTarihiStr = '';
  FIrsaliyeGrupIdStr = '';
  IrsaliyeGrupAdiStr = '';
  FFirmaIdStr = '';
  FFirmaAdiStr = '';
  FOrmeciAdiStr = '';
  FOrmeciMakinaNoStr = '';
  FMusteriReferansNoStr = '';
  FMusteriPartiNoStr = '';
  FKenarKonstruksiyonTipiAdi = '';

  FRenkStr = '';
  OrtalamaMetreTulAgirligi = 0;

  Tamir = '';
  HamStokSayisi = 0;

  For Select PartiHamStok.HamStokDetayId, HamStokDetay.HamStokMasterId, HamStokMaster.Lot, HamStokMaster.En, HamStokMaster.Grm2,
             HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Tamir, TamirTipi.Adi,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.IrsaliyeGrupId, StokHareketGrup.Adi, Ormeci.Adi,
             HamStokMaster.OrmeciMakinaNo,
             HamStokIrsaliye.FirmaId, Firma.Adi, HamStokDetay.TupAcikEnMayli,
             Null, HamStokMaster.MusteriPartiNo, HamStokMaster.Renk, KenarKonstruksiyonTipi.Adi from PartiHamStok
    Left join HamStokDetay on HamStokDetay.HamStokDetayId = PartiHamStok.HamStokDetayId
    Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
    left Join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
    left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = HamStokIrsaliye.IrsaliyeGrupId
    left Join Firma on Firma.FirmaId = HamStokIrsaliye.FirmaId
    left join Ormeci on Ormeci.Ormeciid = HamstokIrsaliye.Ormeciid
    left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
    left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
    left join KenarKonstruksiyonTipi on KenarKonstruksiyonTipi.KenarKonstruksiyonTipiId = HamStokMaster.KenarKonstruksiyonTipiId
    where PartiHamStok.PartiSinifId = :APartiSinifId
    group by PartiHamStok.HamStokDetayId, HamStokDetay.HamStokMasterId, HamStokMaster.Lot, HamStokMaster.En, HamStokMaster.Grm2,
             HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Tamir, TamirTipi.Adi,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.IrsaliyeGrupId, StokHareketGrup.Adi, Ormeci.Adi,
             HamStokMaster.OrmeciMakinaNo, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokDetay.TupAcikEnMayli,
             HamStokMaster.MusteriPartiNo, HamStokMaster.Renk, KenarKonstruksiyonTipi.Adi
    into :HamStokDetayId, :HamStokMasterId, :Lot, :En, :Grm2, :Pus, :Feine, :TamirX, :TamirTipiAdi, :IrsaliyeNo, :IrsaliyeTarihi,
         :IrsaliyeGrupId, :IrsaliyeGrupAdi, :OrmeciAdi, :OrmeciMakinaNo, :FirmaId, :FirmaAdi, :TupAcikEnMayli,
         :MusteriReferansNo, :MusteriPartiNo, :Renk, :KenarKonstruksiyonTipiAdi do
  begin
    if (En is not null and En <> 0 and Grm2 is not null and Grm2 <> 0) then
    begin
      HamStokSayisi = HamStokSayisi + 1;
      OrtalamaMetreTulAgirligi = OrtalamaMetreTulAgirligi + En * Grm2 / 100;
    end

    if (TamirX = 1) then Tamir = 'Tamir';

    Str = '@' || EgemenIntegerToStr(HamStokDetayId) || '@';

    if (PosBul(FHamStokDetayIdStr, Str) = 0) then
    if (Char_Length(HamStokDetayIdStr) + Char_Length(EgemenIntegerToStr(HamStokDetayId)) < 48) then
    begin
      if (HamStokDetayIdStr <> '' ) then
      begin
        HamStokDetayIdStr = HamStokDetayIdStr || cast(', ' as varchar(2) character set UTF8);
        FHamStokDetayIdStr = FHamStokDetayIdStr || cast(', ' as varchar(2) character set UTF8);
      end
      HamStokDetayIdStr = HamStokDetayIdStr || EgemenIntegerToStr(HamStokDetayId);
      FHamStokDetayIdStr = FHamStokDetayIdStr || Str;
    end

    Str = '@' || EgemenIntegerToStr(HamStokMasterId) || '@';
    if (PosBul(FHamStokMasterIdStr, Str) = 0) then
    if (Char_Length(HamStokMasterIdStr) + Char_Length(EgemenIntegerToStr(HamStokMasterId)) < 48) then
    begin
      if (HamStokMasterIdStr <> '' ) then
      begin
        HamStokMasterIdStr = HamStokMasterIdStr || cast(', ' as varchar(2) character set UTF8);
        FHamStokMasterIdStr = FHamStokMasterIdStr || cast(', ' as varchar(2) character set UTF8);
      end
      HamStokMasterIdStr = HamStokMasterIdStr || EgemenIntegerToStr(HamStokMasterId);
      FHamStokMasterIdStr = FHamStokMasterIdStr || Str;
    end

    if (lot is not null and Lot <> '') then
    begin
      Str = '@' || Lot || '@';
      if (PosBul(FLotStr, Str) = 0) then
      if (Char_Length(LotStr) + Char_Length(Lot) < 48) then
      begin
        if (LotStr <> '' ) then
        begin
          LotStr = LotStr || cast(', ' as varchar(2) character set UTF8);
          FLotStr = FLotStr || ',';
        end
        LotStr = LotStr || Lot;
        FLotStr = FLotStr || Str;
      end
    end

    if (En is not null and En > 0) then
    begin
      Str = '@' || EgemenIntegerToStr(En) || '@';
      if (PosBul(FEnStr, Str) = 0) then
      if (Char_Length(EnStr) + Char_Length(EgemenIntegerToStr(En)) < 48) then
      begin
        if (EnStr <> '' ) then
        begin
          EnStr = EnStr || cast(', ' as varchar(2) character set UTF8);
          FEnStr = FEnStr || cast(', ' as varchar(2) character set UTF8);
        end
        EnStr = EnStr || EgemenIntegerToStr(En);
        FEnStr = FEnStr || Str;
      end
    end

    if (Grm2 is not null and Grm2 > 0) then
    begin
      Str = '@' || EgemenIntegerToStr(Grm2) || '@';
      if (PosBul(FGrm2Str, Str) = 0) then
      if (Char_Length(Grm2Str) + Char_Length(EgemenIntegerToStr(Grm2)) < 48) then
      begin
        if (Grm2Str <> '' ) then
        begin
          Grm2Str = Grm2Str || cast(', ' as varchar(2) character set UTF8);
          FGrm2Str = FGrm2Str || cast(', ' as varchar(2) character set UTF8);
        end
        Grm2Str = Grm2Str || EgemenIntegerToStr(Grm2);
        FGrm2Str = FGrm2Str || Str;
      end
    end

    if (Pus is not null and Pus > 0) then
    begin
      Str = '@' || EgemenIntegerToStr(Pus) || '@';
      if (PosBul(FPusStr, Str) = 0) then
      if (Char_Length(PusStr) + Char_Length(EgemenIntegerToStr(Pus)) < 48) then
      begin
        if (PusStr <> '' ) then
        begin
          PusStr = PusStr || cast(', ' as varchar(2) character set UTF8);
          FPusStr = FPusStr || cast(', ' as varchar(2) character set UTF8);
        end
        PusStr = PusStr || EgemenIntegerToStr(Pus);
        FPusStr = FPusStr || Str;
      end
    end

    if (Feine is not null and Feine > 0) then
    begin
      Str = '@' || EgemenIntegerToStr(Feine) || '@';
      if (PosBul(FFeineStr, Str) = 0) then
      if (Char_Length(FeineStr) + Char_Length(EgemenIntegerToStr(Feine)) < 48) then
      begin
        if (FeineStr <> '' ) then
        begin
          FeineStr = FeineStr || cast(', ' as varchar(2) character set UTF8);
          FFeineStr = FFeineStr || cast(', ' as varchar(2) character set UTF8);
        end
        FeineStr = FeineStr || EgemenIntegerToStr(Feine);
        FFeineStr = FFeineStr || Str;
      end
    end

    Str = '@' || EgemenIntegerToStr(IrsaliyeNo) || '@';
    if (PosBul(FIrsaliyeNoStr, Str) = 0) then
    if (Char_Length(IrsaliyeNoStr) + Char_Length(EgemenIntegerToStr(IrsaliyeNo)) < 48) then
    begin
      if (IrsaliyeNoStr <> '' ) then
      begin
        IrsaliyeNoStr = IrsaliyeNoStr || cast(', ' as varchar(2) character set UTF8);
        FIrsaliyeNoStr = FIrsaliyeNoStr || cast(', ' as varchar(2) character set UTF8);
      end
      IrsaliyeNoStr = IrsaliyeNoStr || EgemenIntegerToStr(IrsaliyeNo);
      FIrsaliyeNoStr = FIrsaliyeNoStr || Str;

      Str = '@' || TimeStampToDateStr(IrsaliyeTarihi) || '@';
      if (Char_Length(IrsaliyeTarihiStr) + Char_Length(TimeStampToDateStr(IrsaliyeTarihi)) < 48) then
      begin
        if (IrsaliyeTarihiStr <> '' ) then
        begin
          IrsaliyeTarihiStr = IrsaliyeTarihiStr || cast(', ' as varchar(2) character set UTF8);
          FIrsaliyeTarihiStr = FIrsaliyeTarihiStr || cast(', ' as varchar(2) character set UTF8);
        end
        IrsaliyeTarihiStr = IrsaliyeTarihiStr || TimeStampToDateStr(IrsaliyeTarihi);
        FIrsaliyeTarihiStr = FIrsaliyeTarihiStr || Str;
      end
    end


    if (IrsaliyeGrupId is not null) then
    begin
      Str = '@' || IrsaliyeGrupId || '@';
      if (PosBul(FIrsaliyeGrupIdStr, Str) = 0) then
      if (Char_Length(IrsaliyeGrupIdStr) + Char_Length(IrsaliyeGrupID) < 48) then
      begin
        if (IrsaliyeGrupIdStr <> '' ) then
        begin
          IrsaliyeGrupIdStr = IrsaliyeGrupIdStr || cast(', ' as varchar(2) character set UTF8);
          FIrsaliyeGrupIdStr = FIrsaliyeGrupIdStr || cast(', ' as varchar(2) character set UTF8);
        end
        IrsaliyeGrupIdStr = IrsaliyeGrupIdStr || IrsaliyeGrupId;
        FIrsaliyeGrupIdStr = FIrsaliyeGrupIdStr || Str;
      end
    end

    if (IrsaliyeGrupAdi is not null) then
    begin
      Str = '@' || IrsaliyeGrupAdi || '@';
      if (PosBul(FIrsaliyeGrupAdiStr, Str) = 0) then
      if (Char_Length(IrsaliyeGrupAdiStr) + Char_Length(IrsaliyeGrupAdi) < 65) then
      begin
        if (IrsaliyeGrupAdiStr <> '' ) then
        begin
          IrsaliyeGrupAdiStr = IrsaliyeGrupAdiStr || cast(', ' as varchar(2) character set UTF8);
          FIrsaliyeGrupAdiStr = FIrsaliyeGrupAdiStr || cast(', ' as varchar(2) character set UTF8);
        end
        IrsaliyeGrupAdiStr = IrsaliyeGrupAdiStr || IrsaliyeGrupAdi;
        FIrsaliyeGrupAdiStr = FIrsaliyeGrupAdiStr || Str;
      end
    end

    Str = '@' || FirmaId || '@';
    if (PosBul(FFirmaIdStr, Str) = 0) then
    if (Char_Length(FirmaIdStr) + Char_Length(FirmaId) < 60) then
    begin
      if (FirmaIdStr <> '' ) then
      begin
        FirmaIdStr = FirmaIdStr || cast(', ' as varchar(2) character set UTF8);
        FFirmaIdStr = FFirmaIdStr || cast(', ' as varchar(2) character set UTF8);
      end
      FirmaIdStr = FirmaIdStr || FirmaId;
      FFirmaIdStr = FFirmaIdStr || Str;
    end

    Str = '@' || FirmaAdi || '@';
    if (PosBul(FFirmaAdiStr, Str) = 0) then
    if (Char_Length(FirmaAdiStr) + Char_Length(FirmaAdi) < 90) then
    begin
      if (FirmaAdiStr <> '' ) then
      begin
        FirmaAdiStr = FirmaAdiStr || cast(', ' as varchar(2) character set UTF8);
        FFirmaAdiStr = FFirmaAdiStr || cast(', ' as varchar(2) character set UTF8);
      end
      FirmaAdiStr = FirmaAdiStr || FirmaAdi;
      FFirmaAdiStr = FFirmaAdiStr || Str;
    end

    if (OrmeciAdi is not null and OrmeciAdi <> '') then
    begin
      Str = '@' || OrmeciAdi || '@';
      if (PosBul(FOrmeciAdiStr, Str) = 0) then
      if (Char_Length(OrmeciAdiStr) + Char_Length(OrmeciAdi) < 97) then
      begin
        if (OrmeciAdiStr <> '' ) then
        begin
          OrmeciAdiStr = OrmeciAdiStr || cast(', ' as varchar(2) character set UTF8);
          FOrmeciAdiStr = FOrmeciAdiStr || ',';
        end
        OrmeciAdiStr = OrmeciAdiStr || OrmeciAdi;
        FOrmeciAdiStr = FOrmeciAdiStr || Str;
      end
    end

    if (MusteriReferansNo is not null and MusteriReferansNo <> '') then
    begin
      Str = '@' || MusteriReferansNo || '@';
      if (PosBul(FMusteriReferansNoStr, Str) = 0) then
      if (Char_Length(MusteriReferansNoStr) + Char_Length(MusteriReferansNo) < 48) then
      begin
        if (MusteriReferansNoStr <> '' ) then
        begin
          MusteriReferansNoStr = MusteriReferansNoStr || cast(', ' as varchar(2) character set UTF8);
          FMusteriReferansNoStr = FMusteriReferansNoStr || ',';
        end
        MusteriReferansNoStr = MusteriReferansNoStr || MusteriReferansNo;
        FMusteriReferansNoStr = FMusteriReferansNoStr || Str;
      end
    end

    if (MusteriPartiNo is not null and MusteriPartiNo <> '') then
    begin
      Str = '@' || MusteriPartiNo || '@';
      if (PosBul(FMusteriPartiNoStr, Str) = 0) then
      if (Char_Length(MusteriPartiNoStr) + Char_Length(MusteriPartiNo) < 48) then
      begin
        if (MusteriPartiNoStr <> '' ) then
        begin
          MusteriPartiNoStr = MusteriPartiNoStr || cast(', ' as varchar(2) character set UTF8);
          FMusteriPartiNoStr = FMusteriPartiNoStr || ',';
        end
        MusteriPartiNoStr = MusteriPartiNoStr || MusteriPartiNo;
        FMusteriPartiNoStr = FMusteriPartiNoStr || Str;
      end
    end

    if (Renk is not null and Renk <> '') then
    begin
      Str = '@' || Renk || '@';
      if (PosBul(FRenkStr, Str) = 0) then
      if (Char_Length(RenkStr) + Char_Length(Renk) < 48) then
      begin
        if (RenkStr <> '' ) then
        begin
          RenkStr = RenkStr || cast(', ' as varchar(2) character set UTF8);
          FRenkStr = FRenkStr || ',';
        end
        RenkStr = RenkStr || Renk;
        FRenkStr = FRenkStr || Str;
      end
    end
  end

  if (KenarKonstruksiyonTipiAdi is not null and KenarKonstruksiyonTipiAdi <> '') then
  begin
    Str = '@' || KenarKonstruksiyonTipiAdi || '@';
    if (PosBul(FKenarKonstruksiyonTipiAdi, Str) = 0) then
    if (Char_Length(KenarKonstruksiyonTipiAdiStr) + Char_Length(KenarKonstruksiyonTipiAdi) < 48) then
    begin
      if (KenarKonstruksiyonTipiAdiStr <> '' ) then
      begin
        KenarKonstruksiyonTipiAdiStr = KenarKonstruksiyonTipiAdiStr || cast(', ' as varchar(2) character set UTF8);
        FKenarKonstruksiyonTipiAdi = FKenarKonstruksiyonTipiAdi || ',';
      end
      KenarKonstruksiyonTipiAdiStr = KenarKonstruksiyonTipiAdiStr || KenarKonstruksiyonTipiAdi;
      FKenarKonstruksiyonTipiAdi = FKenarKonstruksiyonTipiAdi || Str;
    end
  end

  if (OrtalamaMetreTulAgirligi <> 0) then OrtalamaMetreTulAgirligi = OrtalamaMetreTulAgirligi / HamStokSayisi;
  else OrtalamaMetreTulAgirligi = Null;
  suspend;
end
