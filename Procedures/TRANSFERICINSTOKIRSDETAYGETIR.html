CREATE PROCEDURE TRANSFERICINSTOKIRSDETAYGETIR(
    AIRSALIYEID INTEGER)
RETURNS (
    STOKDETAYID INTEGER,
    URUNID VARCHAR(20),
    TRANSFERKODU VARCHAR(55),
    MIKTAR DOUBLE PRECISION,
    BIRIMID VARCHAR(2),
    BIRIMTRANSFERKODU VARCHAR(10),
    BIRIMFIYAT DOUBLE PRECISION,
    VERILENSIPBIRIMFIYAT DOUBLE PRECISION,
    BIRIMFIYATTL DOUBLE PRECISION,
    KDVORANI DOUBLE PRECISION,
    VADEGUN INTEGER,
    DOVIZID VARCHAR(5))
AS
declare variable IrsaliyeTarihi     TimeStamp;
declare variable VerilenSipDovizId  varchar(5);
declare variable DovizKuru          Double Precision;
declare variable SistemDovizId  varchar(5);
declare variable EgemenFirmaID  integer;
begin
  Select SistemDovizId, EgemenFirmaID from Sistem where sistemId = 1 into :SistemDovizId, :EgemenFirmaID; /* YTL*/

  for select StokDetay.StokDetayId, Urun.UrunId, Urun.TransferKodu, StokDetay.Miktar, UrunBirim.BirimId, Birim.TransferKodu, StokDetay.BirimFiyat,
             Urun.KdvOrani, Irsaliye.IrsaliyeTarihi, Irsaliye.DovizId, VerilenSiparisDetay.VadeGun, VerilenSiparisDetay.DovizID, VerilenSiparisDetay.BirimFiyat from StokDetay
    left join Irsaliye on Irsaliye.IrsaliyeId = StokDetay.IrsaliyeId
    left join Urun on Urun.UrunId = StokDetay.UrunId
    left join VerilenSiparisDetay on VerilenSiparisDetay.VerilenSiparisDetayId = StokDetay.VerilenSiparisDetayId
    left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
    left join Birim on Birim.BirimId = UrunBirim.BirimId
    where StokDetay.IrsaliyeId = :AIrsaliyeId order by StokDetayId
    into :StokDetayId, :UrunId, :TransferKodu, :Miktar, :BirimId, :BirimTransferKodu, :BirimFiyat, :KdvOrani, :IrsaliyeTarihi, :DovizId, :VadeGun, :VerilenSipDovizId, :VerilenSipBirimFiyat do
  begin
/*
    if (EgemenFirmaID = 0) then
    begin
      DovizId = VerilenSipDovizId;
      BirimFiyat = VerilenSipBirimFiyat;
    end
*/
    if (DovizID <> SistemDovizId) then
    begin
      select DovizKuru from DovizKuruGetir(:DovizId, :SistemDovizId, :IrsaliyeTarihi) into :DovizKuru;
    end
    else DovizKuru = 1;

    BirimFiyatTL = BirimFiyat * DovizKuru;
    suspend;
  end
end
