CREATE PROCEDURE ORMEIRSALIYEKKONTROLBILGISI(
    AORMEIRSALIYEID INTEGER)
RETURNS (
    IDSTR VARCHAR(30),
    ORMEKALITEKONTROLID INTEGER,
    ORMEPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    VARDIYAID INTEGER,
    SATISSTOKHAREKETID INTEGER,
    VARDIYATARIHI TIMESTAMP,
    VARDIYA INTEGER,
    VARDIYAAMIRIPERSONELADI VARCHAR(40),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    RAFID VARCHAR(10),
    FISSERINO VARCHAR(2),
    FISNO INTEGER,
    FISTARIHI TIMESTAMP,
    TOPNO INTEGER,
    BRUTMIKTAR DOUBLE PRECISION,
    MIKTAR DOUBLE PRECISION,
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    PUS INTEGER,
    FEINE INTEGER,
    MAKINAUSTUGRAMAJ INTEGER,
    DINLENMISGRAMAJ INTEGER,
    SIPTUPACIKENMAYLISTR VARCHAR(10),
    TALIMATTUPACIKENMAYLISTR VARCHAR(10),
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    LOT VARCHAR(255),
    MAKINAID VARCHAR(10),
    MAKINAADI VARCHAR(40),
    TOPFIRE DOUBLE PRECISION)
AS
declare variable SipTupAcikEnMayli integer;
declare variable TalimatTupAcikEnMayli integer;
declare variable FLot varchar(150);
begin
  For Select OrmeKaliteKontrol.OrmeKaliteKontrolId, OrmeKaliteKontrol.OrmePlanlamaId, OrmeKaliteKontrol.OrmeVardiyaId, OrmeKaliteKontrol.SatisStokHareketId,
             OrmeKaliteKontrol.vardiyaTarihi, OrmeKaliteKontrol.Vardiya, VAP.Adi, OrmeKaliteKontrol.PersonelId, P.Adi, Coalesce(OrmeRaf.Rafid, OrmeKaliteKontrol.RafId),
             OrmeKaliteKontrol.TopNo, OrmeKaliteKontrol.Miktar, OrmeKaliteKontrol.Sakat,
             OrmeFis.Serino, OrmeFis.FisNo, OrmeFis.Fistarihi, OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine,
             OrmeSiparisDetay.TupAcikEnMayli, OrmePlanlama.TupAcikEnMayli, OrmePlanlama.DinlenmisGramaj, OrmePlanlama.MakinaUstuGramaj,
             OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4,
             OrmePlanlama.UretimPartiNo, OrmePlanlama.OrmeMakinaId, OrmeMakina.Adi from OrmeKaliteKontrol
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeKaliteKontrol.OrmePlanlamaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
    left join StokHareket on StokHareket.StokHareketId = OrmeKaliteKontrol.StokHareketId
    left join OrmeFis on OrmeFis.OrmeFisId  = StokHareket.OrmeFisId
    left join Personel VAP on VAP.PersonelId  = OrmeKaliteKontrol.VardiyaAmiriPersonelId
    left join Personel P on P.PersonelId  = OrmeKaliteKontrol.PersonelId
    left join Kullanici IK on IK.KullaniciId  = OrmeKaliteKontrol.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = OrmeKaliteKontrol.KullaniciId
    left join StokHareket SatisSH on SatisSH.StokHareketId = OrmeKaliteKontrol.SatisStokHareketId
    left join OrmeRaf on OrmeRaf.OrmeRafId = SatisSH.OrmeRafId
    where OrmeKaliteKontrol.OrmeIrsaliyeId = :AOrmeIrsaliyeId
    order by OrmeKaliteKontrol.OrmePlanlamaId, OrmeKaliteKontrol.OrmeKaliteKontrolId
  into :OrmeKaliteKontrolId, :OrmePlanlamaId, :VardiyaId, :SatisStokHareketId, :VardiyaTarihi, :Vardiya, :VardiyaAmiriPersonelAdi,
       :PersonelId, :PersonelAdi, :RafId, :TopNo, :Miktar, :Sakat,
       :FisSerino, :FisNo, :Fistarihi, :Pus, :Feine, :SipTupAcikEnMayli, :TalimatTupAcikEnMayli, :DinlenmisGramaj, :MakinaUstuGramaj,
       :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :UretimPartiNo, :MakinaId, :MakinaAdi do
  begin
    IdStr = OrmeKaliteKontrolId;

    SakatStr = '';
    if (Sakat = 1) then SakatStr = 'Sakat';

    BrutMiktar = Miktar;
    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:SipTupAcikenMayli) into :SipTupAcikEnMayliStr;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TalimatTupAcikenMayli) into :TalimatTupAcikEnMayliStr;

    Lot = '';

    For Select Distinct OrmePlanlamaRecete.Lot from OrmePlanlamaRecete
      where OrmePlanlamaRecete.OrmePlanlamaId = :OrmePlanlamaId
      into :FLot do
    begin
      if (FLot is not null and FLot <> '') then
      begin
        if (Char_Length(Lot) + Char_Length(FLot) < 250) then
        begin
          if (Lot <> '') then Lot = Lot || cast(', ' as varchar(2) character set UTF8);
          Lot = Lot || FLot;
        end
      end
    end

    suspend;
  end

end
