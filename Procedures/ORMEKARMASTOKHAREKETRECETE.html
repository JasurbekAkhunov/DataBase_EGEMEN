    ATARIHTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,

    AORMEMAKINAID VARCHAR(10),
    AALIS INTEGER,

    AURETIMECIKISIADE INTEGER,

    AURETIM INTEGER,

    AFASONSIPARISLER INTEGER,

    ORMEPLANLAMARECETEID INTEGER,
    ORMEPLANLAMAID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    MELANJKODU VARCHAR(20),
    LOT VARCHAR(150),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    KULLANIMORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    KULLANILMISMIKTAR DOUBLE PRECISION,
    FASON INTEGER,
    FASONSTR VARCHAR(15),
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    RECETEFIREORANI DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
declare variable TahminiFireOrani Double Precision;
declare variable FTalimatReceteFireOrani Double Precision;
declare variable UrunHamId Integer;
declare variable OrmeTalimattaReceteFiresiYap Integer;
declare variable OtomatikUCikislardaReceteFire integer;
begin
  For Select OrmePlanlamaId From OrmeKarmaStokHareketRaporu(:ATarihTipi, :ABaslangicTarihi, :ABitisTarihi, :ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :AFirmaId, :ASiparisFirmaId, :AOrmeMakinaId,
                               :AAlis, :ASatis, :AAlisIade, :ASatisIade, :AFasonAlis, :AFasonUretimeCikis, :AFasonUretimeCikisIade, :AUretimeCikis, :AUretimeCikisIade, :AFasonUretim, :AUretim, :AKaydirma, :ASarf, :AReklamasyonGiris,
                               :AReklamasyonCikis, :AhstIplik, :AhstLycra, :AhstKumas, :ABukum, :AOrme, :ABoya, :AHamId, :ARenkId, :AUretimPartiNo, :ACikisPartiNo, :ABoyaTalimatKayitId, :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId,
                               :AFasonSiparisler, :AFasonSatisSiparisler, :ASatisSiparisler, :AAracId, :ASevkSeferNo, :AStokIslemTipiId, :AKaliteNo, :ALot, :AFirmaAdi2yiGorebilir, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id,
                               :ASiparisGrup4Id, :AMusteriSiparisNo, :ADepoId, :ARafTransferFisi, :AReferansNo, :ABukumTalimatId, :AKapaliDurumu, :ALotFiltre, :AModelKodu, :ARafId)
  where OrmePlanlamaId is not Null
  group by OrmePlanlamaId
  into :OrmePlanlamaId do
  begin
    For Select OrmePlanlamaRecete.OrmePlanlamaReceteId, OrmePlanlama.OrmeSiparisDetayId,
               OrmePlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
               Ham.Adi, Ham.Adi2, OrmePlanlamaRecete.RenkId, Renk.Adi, OrmePlanlamaRecete.MelanjKodu, OrmePlanlamaRecete.Lot, OrmePlanlamaRecete.UreticiId, Uretici.Adi, OrmePlanlamaRecete.Fason,
               OrmePlanlamaRecete.KullanimOrani, OrmePlanlamaRecete.CikisMiktari, OrmePlanlamaRecete.IadeMiktari,
               OrmePlanlama.PlanlananMiktar * OrmePlanlamaRecete.KullanimOrani / 100,
               OrmePlanlama.GerceklesenMiktar * OrmePlanlamaRecete.KullanimOrani / 100,
               Ik.Adi, OrmePlanlamaRecete.InsertTarihi, Sk.Adi, OrmePlanlamaRecete.DegisimTarihi,
               OrmeSiparisDetay.TahminiFireOrani, OrmeSiparisDetay.HamId, OrmePlanlamaRecete.TahminiFireOrani From OrmePlanlama
    left join OrmePlanlamaRecete on OrmePlanlamaRecete.OrmePlanlamaId = OrmePlanlama.OrmePlanlamaId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join Ham on Ham.HamId = OrmePlanlamaRecete.HamId
    left join Renk on Renk.RenkId = OrmePlanlamaRecete.RenkId
    left join Uretici on Uretici.UreticiId = OrmePlanlamaRecete.UreticiId
    left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
    Left Join Kullanici IK on IK.KullaniciId = OrmePlanlamaRecete.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = OrmePlanlamaRecete.KullaniciId
    where OrmePlanlama.OrmePlanlamaId = :OrmePlanlamaId
    into :OrmePlanlamaReceteId, :OrmeSiparisDetayId, :HamId, :HamTipiId, :HamTipiAdi,
         :HamAdi, :HamAdi2, :RenkID, :RenkAdi, :MelanjKodu,  :Lot, :UreticiId, :UreticiAdi, :Fason,
         :KullanimOrani, :CikisMiktari, :IadeMiktari, :KullanilacakMiktar, :KullanilmisMiktar,
         :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :TahminiFireOrani, :UrunHamId, :FTalimatReceteFireOrani do
    begin
      if (HamTipiID = 0 or HamTipiID = 6) then
      begin
        select first 1 OrmeTalimattaReceteFiresiYap, OtomatikUCikislardaReceteFire from Sistem into :OrmeTalimattaReceteFiresiYap, :OtomatikUCikislardaReceteFire;
        select first 1 FireOrani from HamStokRecete where HamId = :UrunHamId and ReceteHamId = :HamId into :ReceteFireOrani;
        if (OtomatikUCikislardaReceteFire = 1) then
        begin
          KullanilacakMiktar = KullanilacakMiktar / ((100 - FTalimatReceteFireOrani) / 100);
          ReceteFireOrani    = FTalimatReceteFireOrani;
        end
        else if (OrmeTalimattaReceteFiresiYap <> 1) then
        begin
          KullanilacakMiktar = KullanilacakMiktar / ((100 - TahminiFireOrani) / 100);
          ReceteFireOrani    = TahminiFireOrani;
        end
        else
          KullanilacakMiktar = KullanilacakMiktar / ((100 - ReceteFireOrani) / 100);
      end

      KullanilacakMiktar = Yuvarla(KullanilacakMiktar, 100);

      if (Fason = 1) then FasonStr = 'Fason';
      else if (Fason = 0) then FasonStr = '';
      else FasonStr = 'Bilinmiyor';
      suspend;
    end
  end
end
