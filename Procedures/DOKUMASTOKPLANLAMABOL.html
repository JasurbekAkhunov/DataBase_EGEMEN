CREATE PROCEDURE DOKUMASTOKPLANLAMABOL(
    ADOKUMASTOKPLANLAMAID INTEGER,
    ADOKUMAPLANLAMAID INTEGER,
    ABOLUNECEKMIKTAR BOYAMIKTARSIZE,
    AKULLANICIID INTEGER)
AS
  declare variable FOrmeRafId   Integer;
  declare variable FKalanMiktar BoyaMiktarSize;
begin
  if (ABolunecekMiktar > 0) then
  begin
    select DokumaStokPlanlama.OrmeRafId, DokumaStokPlanlama.Miktar - :ABolunecekMiktar from DokumaStokPlanlama where DokumaStokPlanlama.DokumaStokPlanlamaId = :ADokumaStokPlanlamaId into :FOrmeRafId, :FKalanMiktar;

    if (FKalanMiktar > 0) then
      update DokumaStokPlanlama set DokumaStokPlanlama.Miktar = DokumaStokPlanlama.Miktar - :ABolunecekMiktar where DokumaStokPlanlama.DokumaStokPlanlamaId = :ADokumaStokPlanlamaId;
    else
      delete from DokumaStokPlanlama where DokumaStokPlanlama.DokumaStokPlanlamaId = :ADokumaStokPlanlamaId;

    insert into DokumaStokPlanlama (DokumaStokPlanlamaId, DokumaPlanlamaId, OrmeRafId, Miktar, InsertKullaniciId) values (Gen_Id(DokumaStokPlanlama_Id_Gen, 1), :ADokumaPlanlamaId, :FOrmeRafId, :ABolunecekMiktar, :AKullaniciId);
   end
end
