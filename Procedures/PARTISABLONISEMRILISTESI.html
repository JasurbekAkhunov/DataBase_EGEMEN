CREATE PROCEDURE PARTISABLONISEMRILISTESI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ADESENID DESENKODUSIZE)
RETURNS (
    PARTISABLONISEMRIID INTEGER,
    RECETEISEMRIYIL INTEGER,
    RECETEISEMRIID INTEGER,
    RECETEISEMRIIDSTR VARCHAR(30),
    DESENID DESENKODUSIZE,
    DESENADI VARCHAR(50),
    TARIH TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select PartiSablonIsEmri.PartiSablonIsEmriID, PartiSablonIsEmri.ReceteIsEmriYil, PartiSablonIsEmri.ReceteIsEmriId, PartiSablonIsEmri.DesenId, Desen.Adi, PartiSablonIsEmri.Tarih,
             Ik.Adi, PartiSablonIsEmri.InsertTarihi, Sk.Adi, PartiSablonIsEmri.DegisimTarihi from PartiSablonIsEmri
    left join Desen on Desen.DesenId = PartiSablonIsEmri.DesenId
    left join Kullanici IK on IK.KullaniciId = PartiSablonIsEmri.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId = PartiSablonIsEmri.KullaniciId
    where PartiSablonIsEmri.Tarih >= :ABaslangicTarihi and PartiSablonIsEmri.Tarih <= :ABitisTarihi and
          ( PartiSablonIsEmri.DesenId = :ADesenId or :ADesenId is null)
    into :PartiSablonIsEmriID, :ReceteIsEmriYil, :ReceteIsEmriId, :DesenId, :DesenAdi, :Tarih,
         :InsertKullaniciAdi, InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin
    ReceteIsEmriIdStr = '(' || EgemenIntegerToStr(ReceteIsEmriYil) || ') ' || EgemenIntegerToStr(ReceteIsEmriId) || ' ';
    suspend;
  end
end
