CREATE PROCEDURE FIRMANINPARTIRAPORU_2(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AISLEMTIPIID INTEGER,
    ASATISTIPI INTEGER,
    APARTITIPIID INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    ADURUMU INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    BOYAIRSALIYEID INTEGER,
    BOYAIRSALIYEDETAYID INTEGER,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    BRUTKILO NUMERIC(12,3),
    NETKILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    NETMETRE NUMERIC(12,3))
AS
declare variable PartiKayitId               integer;
begin
  For Select DISTINCT BoyaIrsaliyeDetay.PartiKayitId, BoyaIrsaliyeDetay.PartiSinifId, BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId, BoyaIrsaliye.IrsaliyeSeriNo, BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi,
         BoyaIrsaliyeDetay.BrutKilo, BoyaIrsaliyeDetay.Kilo, BoyaIrsaliyeDetay.BrutMetre, BoyaIrsaliyeDetay.Metre from FirmaninPartiRaporu_1(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :AIslemTipiId, :ASatisTipi, :APartiTipiId, :AMusteriSiparisNo, :ADurumu)
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.PartiSinifId = FirmaninPartiRaporu_1.PartiSinifId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = BoyaIrsaliyeDetay.BoyaIrsaliyeId
    where BoyaIrsaliyeDetay.PartiSinifId is not null
    into :PartiKayitId, :PartiSinifId, :BoyaIrsaliyeId, :BoyaIrsaliyeDetayId, :IrsaliyeSeriNo, :IrsaliyeNo,
         :Irsaliyetarihi, :BrutKilo, :NetKilo, :BrutMetre, :NetMetre do
  begin
    suspend;
  end
end
