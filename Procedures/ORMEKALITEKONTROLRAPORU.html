CREATE PROCEDURE ORMEKALITEKONTROLRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AORMEMAKINAID VARCHAR(10),
    AORMEPLANLAMAID INTEGER,
    ASAKAT INTEGER,
    AOTOMATIKCIKISYAPILAN EGEBOOLEAN,
    AVARDIYATARIHINEGORE INTEGER)
RETURNS (
    ORMEKALITEKONTROLID INTEGER,
    ORMEVARDIYAID INTEGER,
    TARIH TIMESTAMP,
    VARDIYATARIHI TIMESTAMP,
    VARDIYA INTEGER,
    ORMEPLANLAMAID INTEGER,
    URETIMPARTINO VARCHAR(30),
    VARDIYAAMIRIPERSONELID INTEGER,
    VARDIYAAMIRIPERSONELADI VARCHAR(40),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMSTOKALISFIYATI DOUBLE PRECISION,
    SIPARISSATISFIYATI DOUBLE PRECISION,
    RAPORDOVIZID VARCHAR(5),
    KUMASCINSIID INTEGER,
    KUMASCINSIADI VARCHAR(40),
    TOPNO INTEGER,
    MIKTAR DOUBLE PRECISION,
    SAKAT INTEGER,
    SAKATSTR VARCHAR(5),
    AYARTOPU INTEGER,
    AYARTOPUSTR VARCHAR(10),
    ORMEMAKINAID VARCHAR(10),
    ORMEMAKINAADI VARCHAR(40),
    RAFID VARCHAR(10),
    SAKATHAMHATATIPIID INTEGER,
    SAKATHAMHATATIPIADI VARCHAR(20),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    LOT VARCHAR(100),
    MAKINAPUS TYPE OF COLUMN ORMEMAKINA.PUS,
    MAKINAFEINE TYPE OF COLUMN ORMEMAKINA.FEINE,
    URETIMNOTU MEMO,
    KALITEADI ADI255,
    KALITEID VARCHAR(20),
    MAKINAUSTUGRAMAJ INTEGER,
    REFERANSSIPARISNO VARCHAR(30),
    MIKTARGUN DOUBLE PRECISION)
AS
declare variable HamStokAlisFiyatiDovizId varchar(5);
declare variable SiparisDovizId varchar(5);
declare variable DovizKuru Double Precision;
declare variable Ok integer;
declare variable KS integer;
begin
  Select SistemDovizId from Sistem where sistemid = 1 into :RaporDovizId;

  For Select OrmeKaliteKontrol.OrmeKaliteKontrolId, OrmeKaliteKontrol.OrmeVardiyaId, OrmeKaliteKontrol.Tarih, OrmeKaliteKontrol.VardiyaTarihi,
             OrmeKaliteKontrol.Vardiya, OrmeKaliteKontrol.OrmePlanlamaId, OrmePlanlama.UretimPartiNo, OrmeKaliteKontrol.VardiyaAmiriPersonelId, VAmiri.Adi,
             OrmeKaliteKontrol.PersonelId, VPersonel.Adi, OrmePlanlama.BoyaSiparisKayitId, OrmePlanlama.OrmeSiparisDetayid,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi,
             OrmeSiparisDetay.HamId, Ham.Adi, OrmeKaliteKontrol.TopNo, OrmeKaliteKontrol.Miktar, OrmeKaliteKontrol.Sakat, OrmeKaliteKontrol.AyarTopu,
             OrmeKaliteKontrol.RafId, IK.Adi, OrmeKaliteKontrol.InsertTarihi, SK.Adi, OrmeKaliteKontrol.DegisimTarihi,
             Ham.KumasCinsiId, KumasCinsi.Adi, OrmePlanlama.OrmeMakinaId, OrmeMakina.Adi, Ham.AlisFiyati, Ham.AlisFiyatiDovizId,
             OrmeSiparisDetay.SatisFiyati, OrmeSiparisDetay.DovizId, OrmeKaliteKontrol.SakatHamHataTipiId, HamHataTipi.Adi,
             OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4, OrmePlanlama.LOT,
             OrmeMakina.Pus, OrmeMakina.Feine,  OrmeKaliteKontrol.UretimNotu, Kalite.KaliteAdi, Kalite.kaliteid,
             OrmeSiparisDetay.makinaustugramaj,OrmeSiparisDetay.ReferansSiparisNo, OrmePlanlama.MiktarGun from OrmeKaliteKontrol
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = OrmeKaliteKontrol.OrmePlanlamaId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
    left join KumasCinsi on KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
    Left Join Kullanici IK on IK.KullaniciId = OrmeKaliteKontrol.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = OrmeKaliteKontrol.KullaniciId
    left join Personel VAmiri on VAmiri.PersonelId  = OrmeKaliteKontrol.VardiyaAmiriPersonelId
    left join Personel VPersonel on VPersonel.PersonelId  = OrmeKaliteKontrol.PersonelId
    left join HamHataTipi on HamHataTipi.HamHataTipiId  = OrmeKaliteKontrol.SakatHamHataTipiId
    left join Kalite on Kalite.KaliteId = OrmeSiparisDetay.KaliteId
    where ((:AVardiyaTarihineGore = 0 and OrmeKaliteKontrol.Tarih >= :ABaslangicTarihi and OrmeKaliteKontrol.Tarih <= :ABitistarihi) or
           (:AVardiyaTarihineGore = 1 and OrmeKaliteKontrol.VardiyaTarihi >= :ABaslangicTarihi and OrmeKaliteKontrol.VardiyaTarihi <= :ABitistarihi)) and
          (OrmeKaliteKontrol.OrmePlanlamaId = :AOrmePlanlamaId or :AOrmePlanlamaId is null) and
          (OrmeKaliteKontrol.Sakat = :ASakat or :ASakat is null) and
          (OrmePlanlama.OrmeMakinaId = :AOrmeMakinaId or :AOrmeMakinaId is null)
    order by OrmePlanlama.UretimPartiNo, VardiyaTarihi, Vardiya
    into :OrmeKaliteKontrolId, :OrmeVardiyaId, :Tarih, :VardiyaTarihi, :Vardiya, :OrmePlanlamaId, :UretimPartiNo, :VardiyaAmiriPersonelId, :VardiyaAmiriPersonelAdi,
         :PersonelId, :PersonelAdi, :BoyaSiparisKayitId, :OrmeSiparisDetayid, :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :SiparisFirmaAdi,
         :HamId, :HamAdi, :TopNo, :Miktar, :Sakat, :AyarTopu, :RafId, :InsertKullaniciAdi, InsertTarihi, :KullaniciAdi, :DegisimTarihi,
         :KumasCinsiId, :KumasCinsiAdi, :OrmeMakinaId, :OrmeMakinaAdi, :HamStokAlisFiyati, :HamStokAlisFiyatiDovizId,
         :SiparisSatisFiyati, :SiparisDovizId, :SakatHamHataTipiId, :SakatHamHataTipiAdi, :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :Lot,
         :MakinaPus, :MakinaFeine, :UretimNotu, :KaliteAdi, :KaliteId ,:MakinaUstuGramaj, :ReferansSiparisNo, :MiktarGun do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    if (Sakat = 1) then SakatStr = 'Sakat';
    else SakatStr = '';

    if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
    else AyarTopuStr = '';

    if (HamStokAlisFiyatiDovizId <> RaporDovizId) then
      select DovizKuru from EnYakinDovizKuruGetir(:HamStokAlisFiyatiDovizId, :RaporDovizId, Current_Date) into :DovizKuru;
    else
      DovizKuru = 1;

    HamStokAlisFiyati = HamStokAlisFiyati * DovizKuru;

    if (SiparisDovizId <> RaporDovizId) then
      select DovizKuru from EnYakinDovizKuruGetir(:SiparisDovizId, :RaporDovizId, Current_Date) into :DovizKuru;
    else
      DovizKuru = 1;

    SiparisSatisFiyati = SiparisSatisFiyati * DovizKuru;

    Ok = 1;
    if (AOtomatikCikisYapilan is not null) then
    begin
      select sum(DokumaUretimeOtomatikCikis.TamamiCikildi) from OrmeFis
      left join DokumaUretimeOtomatikCikis on DokumaUretimeOtomatikCikis.OrmeFisId = OrmeFis.OrmeFisId
      where OrmeFis.OrmeKaliteKontrolId = :OrmeKaliteKontrolId into :KS;

      if (AOtomatikCikisYapilan = 1 and KS = 0) then Ok = 0;
      else if (AOtomatikCikisYapilan = 0 and KS > 0) then Ok = 0;
    end

    if (Ok = 1) then
      suspend;
  end
end
