CREATE PROCEDURE PARTININHAMSTOKLISTESI(
    APARTIKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI3 VARCHAR(255),
    PUS INTEGER,
    FEINE INTEGER,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    RESULTSET VARCHAR(30),
    RESULTSET2 VARCHAR(30))
AS
declare variable FPartiSinifID integer;
begin
  For SELECT PartiStok.PartiSinifID, BoyaSiparisDetay.BoyaSiparisDetayId, HAM.HAMID, HAM.ADI, HAM.ADI3, HAM.PUS, HAM.FEINE, RECETE.ADI, PARTI.RECETEID
      from PartiStok
      LEFT JOIN PARTI ON PARTI.PARTIKAYITID=PARTISTOK.PARTIKAYITID
      LEFT JOIN BOYASIPARISDETAY ON BOYASIPARISDETAY.BOYASIPARISDETAYID=PARTISTOK.BOYASIPARISDETAYID
      LEFT JOIN PAKETLEMESEKLI ON PAKETLEMESEKLI.PAKETLEMESEKLIID=BOYASIPARISDETAY.PAKETLEMESEKLIID
      LEFT JOIN HAM ON HAM.HAMID=PARTISTOK.HAMID
      LEFT JOIN RECETE ON RECETE.RECETEID = PARTI.RECETEID
      where PARTISTOK.PARTIKAYITID= :APartiKayitID
      into :FPartiSinifID, :BoyaSiparisDetayId, :HamId, :HamAdi, HamAdi3, :Pus, :Feine, :ReceteAdi, :ReceteId do
  begin
    ResultSet = '"' || EgemenIntegerToStr(BoyaSiparisDetayId) || '",' ;
    ResultSet = ResultSet || '"' || EgemenIntegerToStr(FPartiSinifID) || '"' ;

    ResultSet2 = '"' || EgemenIntegerToStr(HamId) || '",' ;
    ResultSet2 = ResultSet2 || '"' || EgemenIntegerToStr(FPartiSinifID) || '"' ;
    suspend;
  end
end
