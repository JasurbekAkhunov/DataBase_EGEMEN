CREATE PROCEDURE PARTITAMIRYENIGIRISHAMSTOKLARI(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADITOPLAMI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BITTI INTEGER)
AS
declare variable KayitSayisi integer;
begin
  for select PartiStok.PartiSinifId, PartiStok.HamAdiToplami, PartiStok.Kilo, PartiStok.Metre, PartiStok.HamTipiId, HamTipi.Adi from PartiStok
      left join HamTipi on HamTipi.HamTipiid = PartiStok.HamTipiId
      where PartiStok.PartiKayitId = :APartiKayitId and PartiStok.Durumu < 8 and PartiStok.NetSevkKilo = 0
     into :PartiSinifId, :HamAdiToplami, :Kilo, :Metre, :HamTipiId, :HamTipiAdi do
  begin
    select count(*) from PartiTamirSinif where PartiTamirSinif.PartiSinifId = :PartiSinifId and Bitti = 0 into :KayitSayisi;
    bitti = 0;
    if(KayitSayisi = 0) then suspend;
  end
end
