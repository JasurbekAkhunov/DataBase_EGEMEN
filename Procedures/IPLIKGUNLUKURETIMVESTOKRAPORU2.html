    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    SIRANO INTEGER,
    IPLIKDEPOID VARCHAR(10),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    KALINLIKID INTEGER,
    KALINLIKADI VARCHAR(50),
    IPLIKCINSIID INTEGER,
    IPLIKCINSIADI VARCHAR(80),
    HARMANCINSIID INTEGER,
    HARMANCINSIADI VARCHAR(50),
    HARMANADI VARCHAR(50),
    HAMGRUP1 VARCHAR(4),
    HAMGRUP2 VARCHAR(4),
    HAMGRUP3 VARCHAR(4),
    HAMGRUP4 VARCHAR(4),
    LOT VARCHAR(30),
    ANALOTBOLUM VARCHAR(30),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    TAMIR INTEGER,
    TAMIRSTR VARCHAR(5),
    SARIMSEKLIID INTEGER,
    SARIMSEKLIIDSTR VARCHAR(20),
    AMBALAJSEKLIID INTEGER,
    AMBALAJSEKLIADI VARCHAR(10),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    DEVIRSTOKKAPADEDI INTEGER,
    DEVIRSTOKMIKTAR DOUBLE PRECISION,
    DEVIRSTOKNETMIKTAR DOUBLE PRECISION,
    URETIMKAPADEDI INTEGER,
    URETIMMIKTAR DOUBLE PRECISION,
    URETIMNETMIKTAR DOUBLE PRECISION,
    SARIMURETIMKAPADEDI INTEGER,
    SARIMURETIMMIKTAR DOUBLE PRECISION,
    SARIMURETIMNETMIKTAR DOUBLE PRECISION,
    SARIMACIKANKAPADEDI INTEGER,
    SARIMACIKANMIKTAR DOUBLE PRECISION,
    SARIMACIKANNETMIKTAR DOUBLE PRECISION,
    BUKUMECIKANKAPADEDI INTEGER,
    BUKUMECIKANMIKTAR DOUBLE PRECISION,
    BUKUMECIKANNETMIKTAR DOUBLE PRECISION,
    BUKUMURETIMKAPADEDI INTEGER,
    BUKUMURETIMMIKTAR DOUBLE PRECISION,
    BUKUMURETIMNETMIKTAR DOUBLE PRECISION,
    SEVKIYATKAPADEDI INTEGER,
    SEVKIYATMIKTAR DOUBLE PRECISION,
    SEVKIYATNETMIKTAR DOUBLE PRECISION,
    GIRISKAPADEDI INTEGER,
    GIRISMIKTAR DOUBLE PRECISION,
    GIRISNETMIKTAR DOUBLE PRECISION,
    SEVKIYATIRSALIYENO VARCHAR(100),
    SEVKIYATIRSALIYETARIHI VARCHAR(100),
    SEVKIYATARACID VARCHAR(100),
    BYSEVKIYATKAPADEDI INTEGER,
    BYSEVKIYATMIKTAR DOUBLE PRECISION,
    BYSEVKIYATNETMIKTAR DOUBLE PRECISION,
    KALANKAPADEDI INTEGER,
    KALANMIKTAR DOUBLE PRECISION,
    KALANNETMIKTAR DOUBLE PRECISION,
    LOTKAPALI INTEGER,
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(255),
    HARMANIPLIKISLETMEADI ADI50,
    HARMANIPLIKISLETMEID INTEGER,
    HARMANRENKADI VARCHAR(50),
    HAMDIGERADI1 ADI255,
    HAMDIGERADI2 ADI255,
    HAMDIGERADI3 ADI255,
    BUKUMISEMRIMI EGEBOOLEAN)
AS
begin
  SiraNo = 0;

  For Select IplikGunlukUretimveStokRaporu1.IplikDepoId, IplikGunlukUretimveStokRaporu1.HamId, Ham.Adi, Ham.KalinlikId, Kalinlik.Adi, Ham.IplikCinsiId, IplikCinsi.Adi,
             IplikGunlukUretimveStokRaporu1.HarmanCinsiId, IplikGunlukUretimveStokRaporu1.HarmanCinsiAdi, IplikGunlukUretimveStokRaporu1.HarmanAdi,
             Ham.HamGrup1, Ham.HamGrup2, Ham.HamGrup3, Ham.HamGrup4, Ham.Adi2, Ham.Adi3, Ham.Adi4,
             IplikGunlukUretimveStokRaporu1.Lot, IplikGunlukUretimveStokRaporu1.AnaLotBolum, IplikGunlukUretimveStokRaporu1.MelanjKodu, IplikGunlukUretimveStokRaporu1.FlamKodu, IplikGunlukUretimveStokRaporu1.AmbalajSekliId,
             AmbalajSekli.Adi, IplikGunlukUretimveStokRaporu1.SiparisFirmaId, Firma.Adi, IplikGunlukUretimveStokRaporu1.SarimSekliId,
             IplikGunlukUretimveStokRaporu1.AmbalajId, IplikGunlukUretimveStokRaporu1.AmbalajAdi,
             Max(IplikGunlukUretimveStokRaporu1.Tamir), Max(IplikGunlukUretimveStokRaporu1.LotKapali),
             Sum(DevirStokKapAdedi), Sum(DevirStokMiktar), Sum(DevirStokNetMiktar),
             Sum(UretimKapAdedi), sum(UretimMiktar), sum(UretimNetMiktar),
             Sum(SarimUretimKapAdedi), sum(SarimUretimMiktar), sum(SarimUretimNetMiktar),
             sum(SarimaCikanKapAdedi), Sum(SarimaCikanMiktar), Sum(SarimaCikanNetMiktar),
             sum(BukumeCikanKapAdedi), sum(BukumeCikanMiktar), Sum(BukumeCikanNetMiktar),
             Sum(BukumUretimKapAdedi), sum(BukumUretimMiktar), sum(BukumUretimNetMiktar),
             Sum(SevkiyatKapAdedi), Sum(SevkiyatMiktar), Sum(SevkiyatNetMiktar), Max(SevkiyatIrsaliyeNo), Max(SevkiyatIrsaliyeTarihi), Max(SevkiyatAracId),
             Sum(BySevkiyatKapAdedi), Sum(BySevkiyatMiktar), Sum(BySevkiyatNetMiktar), sum(GirisKapAdedi), sum(GirisMiktar), sum(GirisNetMiktar),
             IplikGunlukUretimveStokRaporu1.HarmanIplikIsletmeId, IplikGunlukUretimveStokRaporu1.HarmanIplikIsletmeAdi, IplikGunlukUretimveStokRaporu1.HarmanRenkAdi, IplikGunlukUretimveStokRaporu1.BukumIsEmrimi from IplikGunlukUretimveStokRaporu1(:ABaslangicTarihi, :ABitisTarihi)
  left join Ham on Ham.HamId = IplikGunlukUretimveStokRaporu1.HamId
  left join Kalinlik on Kalinlik.KalinlikId = Ham.KalinlikId
  left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
  left join AmbalajSekli on AmbalajSekli.AmbalajSekliId = IplikGunlukUretimveStokRaporu1.AmbalajSekliId
  left join Firma on Firma.FirmaId = IplikGunlukUretimveStokRaporu1.SiparisFirmaId
  group by IplikGunlukUretimveStokRaporu1.IplikDepoId, IplikGunlukUretimveStokRaporu1.HamId, Ham.Adi, Ham.KalinlikId, Kalinlik.Adi, Ham.IplikCinsiId, IplikCinsi.Adi,
           IplikGunlukUretimveStokRaporu1.HarmanCinsiId, IplikGunlukUretimveStokRaporu1.HarmanCinsiAdi,
           IplikGunlukUretimveStokRaporu1.HarmanAdi, Ham.HamGrup1, Ham.HamGrup2, Ham.HamGrup3, Ham.HamGrup4, Ham.Adi2, Ham.Adi3, Ham.Adi4,
           IplikGunlukUretimveStokRaporu1.Lot, IplikGunlukUretimveStokRaporu1.AnaLotBolum, IplikGunlukUretimveStokRaporu1.MelanjKodu, IplikGunlukUretimveStokRaporu1.FlamKodu,
           IplikGunlukUretimveStokRaporu1.AmbalajSekliId,
           AmbalajSekli.Adi, IplikGunlukUretimveStokRaporu1.SiparisFirmaId, Firma.Adi,
           IplikGunlukUretimveStokRaporu1.SarimSekliId, IplikGunlukUretimveStokRaporu1.AmbalajId, IplikGunlukUretimveStokRaporu1.AmbalajAdi, IplikGunlukUretimveStokRaporu1.HarmanIplikIsletmeId,
           IplikGunlukUretimveStokRaporu1.HarmanIplikIsletmeAdi, IplikGunlukUretimveStokRaporu1.HarmanRenkAdi, IplikGunlukUretimveStokRaporu1.BukumIsEmrimi
  into :IplikDepoId, :HamId, :HamAdi, :KalinlikId, :KalinlikAdi, :IplikCinsiId, :IplikCinsiAdi,
       :HarmanCinsiId, :HarmanCinsiAdi, :HarmanAdi, :HamGrup1, :HamGrup2, :HamGrup3, :HamGrup4, :HamDigerAdi1, Hamdigeradi2, :HamDigerAdi3,
       :Lot, :AnaLotBolum, :MelanjKodu, :FlamKodu,  :AmbalajSekliId, :AmbalajSekliAdi, :SiparisFirmaId, :SiparisFirmaAdi, :SarimSekliId,
       :AmbalajId, :AmbalajAdi,
       :Tamir, :LotKapali, :DevirStokKapAdedi, :DevirStokMiktar, :DevirStokNetMiktar,
       :UretimKapAdedi, :UretimMiktar, :UretimNetMiktar, :SarimUretimKapAdedi, :SarimUretimMiktar, :SarimUretimNetMiktar,
       :SarimaCikanKapAdedi, :SarimaCikanMiktar, :SarimaCikanNetMiktar,
       :BukumeCikanKapAdedi, :BukumeCikanMiktar, :BukumeCikanNetMiktar,
       :BukumUretimKapAdedi, :BukumUretimMiktar, :BukumUretimNetMiktar,
       :SevkiyatKapAdedi, :SevkiyatMiktar, :SevkiyatNetMiktar,
       :SevkiyatIrsaliyeNo, :SevkiyatIrsaliyeTarihi, :SevkiyatAracId, :BySevkiyatKapAdedi, :BySevkiyatMiktar, :BySevkiyatNetMiktar,
       :GirisKapAdedi, :GirisMiktar, :GirisNetMiktar, :HarmanIplikIsletmeId, :HarmanIplikIsletmeAdi, :HarmanRenkAdi, :BukumIsEmrimi do
  begin
    SiraNo = SiraNo + 1;
    if (SarimUretimNetMiktar is null) then SarimUretimNetMiktar = 0;
    if (GirisKapAdedi is null) then GirisKapAdedi = 0;
    if (GirisMiktar is null) then GirisMiktar = 0;
    if (GirisNetMiktar is null) then GirisNetMiktar = 0;

    KalanKapAdedi = DevirStokKapAdedi + UretimKapAdedi + SarimUretimKapAdedi + BukumUretimKapAdedi + GirisKapAdedi - SevkiyatKapAdedi - BySevkiyatKapAdedi - SarimaCikanKapAdedi - BukumeCikanKapAdedi;
    KalanMiktar = DevirStokMiktar + UretimMiktar + SarimUretimMiktar + BukumUretimMiktar + GirisMiktar - SevkiyatMiktar - BySevkiyatMiktar - SarimaCikanMiktar - BukumeCikanMiktar;
    KalanNetMiktar = DevirStokNetMiktar + UretimNetMiktar + SarimUretimNetMiktar  + BukumUretimNetMiktar + GirisNetMiktar - SevkiyatNetMiktar - BySevkiyatNetMiktar - SarimaCikanNetMiktar - BukumeCikanNetMiktar;

    if (Tamir = 1) then TamirStr = 'Evet';
    else TamirStr = '';

    if (SarimSekliId = 0) then SarimSekliIdStr = 'Sert Sarım';
    else SarimSekliIdStr ='Yumuşak Sarım';

    if (DevirStokMiktar <> 0 or UretimMiktar <> 0 or SevkiyatMiktar <> 0 or KalanMiktar <> 0) then suspend;
  end
end
