CREATE PROCEDURE HAMSTOKMASTERTOPGETIR(
    AHAMSTOKIRSALIYEID INTEGER)
RETURNS (
    HAMSTOKMASTERTOPID INTEGER,
    HAMSTOKMASTERID INTEGER,
    HAMTIPIID INTEGER,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    PUS INTEGER,
    FEINE INTEGER,
    ORMECIMAKINANO VARCHAR(10),
    EN INTEGER,
    GRM2 INTEGER,
    RENK VARCHAR(20),
    TUPACIKENMAYLISTR VARCHAR(10),
    LOT VARCHAR(150),
    DENYE INTEGER,
    IPLIKUZUNLUGU1 VARCHAR(40),
    IPLIKUZUNLUGU2 VARCHAR(40),
    IPLIKUZUNLUGU3 VARCHAR(40),
    KAPADEDI INTEGER,
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    TAMIR INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    ESKIPARTINO INTEGER,
    MUSTERIPARTINO VARCHAR(30),
    NEMORANI DOUBLE PRECISION,
    YIKAMAONCESIEN DOUBLE PRECISION,
    YIKAMASONRASIEN DOUBLE PRECISION,
    YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    YIKAMASONRASIGRAMAJ DOUBLE PRECISION,
    ENDENCEKMEZLIK DOUBLE PRECISION,
    BOYDANCEKMEZLIK DOUBLE PRECISION,
    HAMDONME DOUBLE PRECISION,
    YIKAMAONCESIRAPORBOYU VARCHAR(100),
    YIKAMASONRASIRAPORBOYU VARCHAR(100),
    BOYAMAKABILIYETI VARCHAR(10),
    YIKAMAHASLIGI VARCHAR(10),
    NUMARATIPIADI VARCHAR(10),
    NUMARA VARCHAR(10),
    BUKUM VARCHAR(10),
    MUKAVEMET VARCHAR(10),
    UZAMA VARCHAR(10),
    UVKONTROL INTEGER,
    KONTROLKULLANICIADI VARCHAR(40),
    KONTROLTARIHI TIMESTAMP,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(4),
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    ONAYTARIHI TIMESTAMP,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    HAMSTOKKABULTIPIID INTEGER,
    HAMSTOKKABULTIPIADI VARCHAR(20),
    RAFID VARCHAR(10),
    AMBALAJID INTEGER,
    AMBALAJADI VARCHAR(20),
    AMBALAJAGIRLIGI DOUBLE PRECISION,
    KOLINO INTEGER,
    BOBINSAYISI INTEGER,
    BOBINDARASI DOUBLE PRECISION,
    DARA DOUBLE PRECISION,
    BRUTKILO DOUBLE PRECISION,
    BARKODID TYPE OF COLUMN HAMSTOKMASTERTOP.BARKODID,
    KONTROLEDILDI TYPE OF COLUMN HAMSTOKMASTERTOP.KONTROLEDILDI,
    KALITEID VARCHAR(20))
AS
declare variable TupAcikEnMayli  integer;
begin
  for select HamStokMasterTop.HamStokMasterTopId, HamStokMasterTop.HamStokMasterId, HamStokMaster.HamId, Ham.HamTipiId, Ham.Adi, Ham.Adi2, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.OrmeciMakinaNo, HamStokMaster.En,
             HamStokMaster.Grm2,  HamStokMaster.Renk, HamStokMaster.Lot, HamStokMaster.TupAcikEnMayli,
             HamStokMaster.Denye, HamStokMaster.IplikUzunlugu1, HamStokMaster.IplikUzunlugu2, HamStokMaster.IplikUzunlugu3,
             HamStokMasterTop.Kilo, HamStokMasterTop.Metre, HamStokMaster.Tamir, TamirTipi.Adi, HamStokMaster.EskiPartino, HamStokMaster.MusteriPartiNo,
             HamStokMaster.OnayDurumu, HamStokMaster.OnaylayanKullaniciId, OK.Adi, HamStokMaster.OnayTarihi,
             HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiGramaj, HamStokMaster.EndenCekmezlik,
             HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
             HamStokMaster.BoyamaKabiliyeti, HamStokMaster.YikamaHasligi, NumaraTipi.Adi, HamStokMaster.Numara, HamStokMaster.Bukum, HamStokMaster.Mukavemet, HamStokMaster.NemOrani, HamStokMaster.Uzama, HamStokMaster.UVKontrol,
             KK.Adi, HamStokMaster.Kontroltarihi, IK.Adi, HamStokMasterTop.InsertTarihi, Sk.Adi, HamStokMasterTop.DegisimTarihi,
             HamStokMasterTop.RafId, HamStokMasterTop.AmbalajId, Ambalaj.Adi, HamStokMasterTop.AmbalajAgirligi,
             HamStokMasterTop.KoliNo, HamStokMasterTop.BobinSayisi, HamStokMasterTop.BobinDarasi, HamStokMasterTop.Dara, HamStokMasterTop.BrutKilo,
             HamStokMasterTop.BarkodId, HamStokMasterTop.KontrolEdildi, HamStokMaster.KaliteId from HamStokMasterTop
  left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokMasterTop.HamStokMasterId
  left join Ham on Ham.HamId = HamStokMaster.HamId
  left join NumaraTipi on NumaraTipi.NumaraTipiId = HamStokMaster.NumaraTipiId
  left join DisTamir on DisTamir.HamStokMasterId = HamStokMaster.HamStokMasterId
  left join TamirTipi on TamirTipi.TamirTipiId = DisTamir.TamirTipiId
  left join Ambalaj on Ambalaj.AmbalajId = HamStokMasterTop.AmbalajId
  Left Join Kullanici OK on OK.KullaniciId = HamStokMaster.OnaylayanKullaniciId
  Left Join Kullanici KK on KK.KullaniciId = HamStokMaster.KontrolKullaniciId
  Left Join Kullanici IK on IK.KullaniciId = HamStokMasterTop.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = HamStokMasterTop.KullaniciId
  where HamStokMaster.HamStokIrsaliyeId = :AHamStokIrsaliyeId order by HamStokMasterTop.HamStokMasterId
  into :HamStokMasterTopId, :HamStokMasterId, :HamId, :HamTipiId, :HamAdi, :HamAdi2, :Pus, :Feine, :OrmeciMakinaNo, :En, :Grm2, :Renk, :Lot, :TupAcikEnMayli, :Denye,
       :IplikUzunlugu1, :IplikUzunlugu2, :IplikUzunlugu3,
       :Kilo, :Metre, :Tamir, :TamirTipiAdi, :EskiPartino, :MusteriPartiNo,
       :OnayDurumu, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi, :OnayTarihi,
       :YikamaOncesiEn, :YikamaSonrasiEn, :YikamaOncesiGramaj, :YikamaSonrasiGramaj, :EndenCekmezlik,
       :BoydanCekmezlik, :HamDonme, :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu,
       :BoyamaKabiliyeti, :YikamaHasligi, :NumaraTipiAdi, :Numara, :Bukum, :Mukavemet, :NemOrani, :Uzama, :UVKontrol,
       :KontrolKullaniciAdi, :Kontroltarihi, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi,
       :RafId, :AmbalajId, :AmbalajAdi, :AmbalajAgirligi,
       :KoliNo, :BobinSayisi, :BobinDarasi, :Dara, :BrutKilo, :BarkodId, :KontrolEdildi, :KaliteId  do
  begin
    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    if (OnayDurumu = 1) then OnayDurumuStr = 'Okey';
    else OnayDurumuStr = '';

    suspend;
  end
end
