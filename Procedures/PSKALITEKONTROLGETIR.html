CREATE PROCEDURE PSKALITEKONTROLGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PSKALITEKONTROLASAMAID INTEGER,
    PSKALITEKONTROLID INTEGER,
    TESTATOPID SMALLINT,
    BARKODID VARCHAR(20),
    PARTISINIFID INTEGER,
    MAKINAID VARCHAR(10),
    TARIH TIMESTAMP,
    KALITENO INTEGER,
    TOPNO INTEGER,
    BRUTKILO NUMERIC(12,3),
    DARA NUMERIC(12,3),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    EN VARCHAR(10),
    GRAMAJ VARCHAR(10),
    BASLANGICEN VARCHAR(10),
    ORTAEN VARCHAR(10),
    BITISEN VARCHAR(10),
    ORTALAMAEN INTEGER,
    ENSAPMASICM INTEGER,
    ENSAPMASIORANI INTEGER,
    DONME VARCHAR(10),
    EGRILIK VARCHAR(10),
    PAKETLEMEID VARCHAR(15),
    ACIKLAMA VARCHAR(100),
    MUSTERIACIKLAMA VARCHAR(50),
    HATAPUANI INTEGER,
    PAKETLEMEHATAPUANI DOUBLE PRECISION,
    HACIMEN INTEGER,
    HACIMBOY INTEGER,
    HACIMYUKSEKLIK INTEGER,
    PARTISTOKGRAMAJ VARCHAR(10),
    HACIM DOUBLE PRECISION,
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(4),
    ONAYTARIHI TIMESTAMP,
    ONAYKULLANICIADI VARCHAR(40),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    PARCASAYISI INTEGER,
    PARCAMETRE NUMERIC(12,3),
    RAPORBOYU ADI100,
    PARTIRENKLOTNO INTEGER,
    REDDENBOLUNENYENIPARTISINIFID INTEGER,
    GRAMAJDEGER INTEGER,
    ENDEGER INTEGER,
    PARTISTOKEN INTEGER,
    SIPARISEACILANPARTISIRASI INTEGER,
    REDKILO NUMERIC(12,3),
    REDMETRE NUMERIC(12,3),
    HESAPLAMACARPIMDEGERI INTEGER,
    ENCEKMEZLIK DOUBLE PRECISION,
    BOYCEKMEZLIK DOUBLE PRECISION,
    PACADONMESI DOUBLE PRECISION,
    OLCUID INTEGER,
    OLCUADI ADI40,
    ADET INTEGER,
    ESKILABRECETEKODU TYPE OF COLUMN LABRECETE.ESKILABRECETEKODU,
    ANAHATATIPIADI VARCHAR(20),
    ANAHATABOLUMADI VARCHAR(30),
    HESAPLANANMETRE BOYAMIKTARSIZE,
    SARTLIKABUL EGEBOOLEAN,
    COZGUINDIGOISEMRIIDSTR VARCHAR(50),
    DOKUMATOPNO INTEGER,
    INSERTKULLANICIID INTEGER)
AS
declare variable EnIstenen          varchar(10);
declare variable EnIstenenInt       integer;
declare variable BoyaSiparisDetayId integer;
declare variable PartiAcilisTarihi  timestamp;
declare variable DokumaTopTakibiDevrede integer;
begin
  select DokumaTopTakibiDevrede from sistem where sistem.sistemid = 1 into :DokumaTopTakibiDevrede;

  For Select PSKaliteKontrol.PSKaliteKontrolAsamaId, PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.PartiSinifId, PSKaliteKontrol.Tarih, PSKaliteKontrol.MakinaId,
             PSKaliteKontrol.KaliteNo, PSKaliteKontrol.TopNo, PSKaliteKontrol.Kilo, PSKaliteKontrol.Dara, PSKaliteKontrol.Metre, PSKaliteKontrol.BrutMetre, PSKaliteKontrol.En, PSKaliteKontrol.Gramaj,
             PSKaliteKontrol.BaslangicEn, PSKaliteKontrol.OrtaEn, PSKaliteKontrol.BitisEn, PSKaliteKontrol.OrtalamaEn, PSKaliteKontrol.Donme, PSKaliteKontrol.Egrilik,
             PSKaliteKontrol.PaketlemeId, PSKaliteKontrol.HataPuani, PSKaliteKontrol.PaketlemeHataPuani, IK.Adi, PSKaliteKontrol.InsertTarihi, SK.Adi, PSKaliteKontrol.DegisimTarihi,
             BoyaSiparisDetay.EnIstenen, PSKaliteKontrol.Aciklama, PartiStok.HacimEn, PartiStok.HacimBoy, PartiStok.HacimYukseklik,
             PSKaliteKontrol.OnayDurumu, PSKaliteKontrol.OnayTarihi, Ok.Adi, PSKaliteKontrol.ParcaSayisi, PSKaliteKontrol.ParcaMetre, PSKaliteKontrol.PartiRenkLotNo, PSKaliteKontrol.ReddenBolunenYeniPartiSinifID,
             PSKaliteKontrol.MusteriAciklama, PSKaliteKontrol.RaporBoyu, PartiStok.Gramaj,
             coalesce(iif(EgemenStrToIntOlurmu(PartiStokFizikKontrol.GramajVerilen) = 1, EgemenStrToInt(PartiStokFizikKontrol.GramajVerilen), 0), 0), PartiStok.En, BoyaSiparisDetay.BoyaSiparisDetayId, Partistok.InsertTarihi, PSKaliteKontrol.TestaTopId,
             BoyaSiparisDetay.HesaplamaCarpimDegeri, PSKaliteKontrol.EnCekmezlik, PSKaliteKontrol.BoyCekmezlik, PSKaliteKontrol.PacaDonmesi, Olcu.OlcuId, Olcu.OlcuAdi, PSKaliteKontrol.Adet,
             LabRecete.EskiLabReceteKodu, HataTipi.Adi, BoyahaneBolum.Adi, coalesce(iif(EgemenStrToIntOlurmu(PartiStokFizikKontrol.EnVerilen) = 1, EgemenStrToInt(PartiStokFizikKontrol.EnVerilen), 0), 0), PSKaliteKontrol.SartliKabul,
             PSDokumaIsEmriBilgisiGetir.CozguIndigoIsEmriIdStr, PSDokumaIsEmriBilgisiGetir.TopNo, IK.KullaniciId from PSKaliteKontrol
    left join PartiStok on PartiStok.PartiSinifID = PSKaliteKontrol.PartiSinifID
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayID = PartiStok.BoyaSiparisDetayId
    left join Kullanici IK on IK.KullaniciId  = PSKaliteKontrol.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = PSKaliteKontrol.KullaniciId
    left join Kullanici OK on OK.KullaniciId  = PSKaliteKontrol.OnayKullaniciId
    left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifID = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
    left join Olcu on Olcu.OlcuId = PsKaliteKontrol.OlcuId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join BoyahaneBolum on BoyahaneBolum.BoyahaneBolumId = PsKaliteKontrol.AnaHataBolumId
    left join HataTipi on HataTipi.HataTipiId = PsKaliteKontrol.AnaHataTipiId
    left join PSDokumaIsEmriBilgisiGetir(null,PsKaliteKontrol.HamStokMasterId) ON 1=1
    /*
    left join HamStokMaster on HamStokMaster.HamStokMasterID = PSKaliteKontrol.HamStokMasterId
    left join DokumaStokPlanlama On DokumaStokPlanlama.Dokumaplanlamaid = HAmStokMAster.Dokumaplanlamaid
    left join OrmeRaf On OrmeRaf.OrmeRafId = DokumaStokPlanlama.Ormerafid
    left join IndigoIsEmri On IndigoIsEmri.IndigoIsEmriKayitId = OrmeRaf.IndigoIsEmriKayitId
    left join CozguIsEmri On CozguIsEmri.CozguIsEmriKayitId = OrmeRaf.CozguIsEmriKayitId
    left join StokHareket On StokHareket.StokHareketId = HAmStokMAster.StokHareketId and :DokumaTopTakibiDevrede = 1
    left join OrmeRafDetay on OrmeRafDetay.Ormerafid = StokHareket.Ormerafid
    left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketID = OrmeRafDetay.StokHareketId
    */
    where PSKaliteKontrol.PartiKayitId = :APartiKayitId
    order by PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrol.TopNo
  into :PSKaliteKontrolAsamaId, :PSKaliteKontrolId, :PartiSinifId, :Tarih, :MakinaId, :KaliteNo, :TopNo, :BrutKilo, :Dara, :Metre, :BrutMetre, :En, :Gramaj,
       :BaslangicEn, :OrtaEn, :BitisEn, :OrtalamaEn, :Donme, :Egrilik, :PaketlemeId, :HataPuani, :PaketlemeHataPuani,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :EnIstenen, :Aciklama,
       :HacimEn, :HacimBoy, :HacimYukseklik, :OnayDurumu, :OnayTarihi, :OnayKullaniciAdi, :ParcaSayisi, :ParcaMetre, :PartiRenkLotNo, :ReddenBolunenYeniPartiSinifID,
       :MusteriAciklama, :RaporBoyu, :PartiStokGramaj, :GramajDeger, :PartiStokEn, :BoyaSiparisDetayId, :PartiAcilisTarihi, :TestaTopId, :HesaplamaCarpimDegeri, :EnCekmezlik, :BoyCekmezlik, :PacaDonmesi,
       :OlcuId, :OlcuAdi, :Adet, :EskiLabReceteKodu, :AnaHataTipiAdi, :AnaHataBolumAdi, :EnDeger, :SartliKabul, :CozguIndigoIsEmriIdStr, :DokumaTopno, :InsertKullaniciId do
  begin
    Select count(*) from PartiStok where PartiStok.BoyasiparisDetayId = :BoyaSiparisDetayId  and PartiStok.InsertTarihi <= :PartiAcilisTarihi into :SipariseAcilanPartiSirasi;

    Hacim = HacimEn * HacimBoy * HacimYukseklik;
    Hacim = Hacim / 1000000;

    BarkodID = 'K-' || EgemenIntegerToStr(PSKaliteKontrolId);


    if (OnayDurumu = 1) then OnayDurumuStr = 'Onay';
    else if (OnayDurumu = 2) then
      OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';


    Kilo = BrutKilo - Dara;
    EnSapmasiCm = null;
    EnSapmasiOrani = null;

    if (EgemenStrToIntOlurmu(EnIstenen) = 1) then
    begin
      EnIstenenInt = EgemenStrtoInt(EnIstenen);
      EnSapmasiCm = OrtalamaEn - EnIstenenInt;
      EnSapmasiOrani = EnSapmasiCm / EnIstenenInt * 100;
    end

    RedKilo = 0;
    RedMetre = 0;
    if (OnayDurumu = 2) then
    begin
      ReDKilo = Kilo;
      RedMetre = Metre;
    end

    if ((EnDeger * GramajDeger) <> 0) then HesaplananMetre = (1000 / Cast(EnDeger * GramajDeger as Double Precision) ) * 100 * Kilo;
    else HesaplananMetre = 0;

    suspend;
  end
end
