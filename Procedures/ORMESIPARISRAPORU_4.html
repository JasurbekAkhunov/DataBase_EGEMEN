CREATE PROCEDURE ORMESIPARISRAPORU_4(
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ABASLANGICSIPARISTARIHI TIMESTAMP,
    ABITISSIPARISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AFASONSIPARISLER INTEGER,
    AFASONSATISSIPARISLER INTEGER,
    ASATISSIPARISLER INTEGER,
    AIPLIKSIPARISI INTEGER,
    ABOYALIIPLIKSIPARISI INTEGER,
    ALYCRASIPARISI INTEGER,
    AHAMKUMASSIPARISI INTEGER,
    ABOYALIKUMASSIPARISI INTEGER,
    AIPLIKBOYALIKUMASSIPARISI INTEGER,
    AMUSTERISIPARISNO VARCHAR(50),
    AREFERANSSIPARISNO VARCHAR(30),
    ARAPORTIPI INTEGER,
    AKAPALI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    ASIPARISGRUP3ID VARCHAR(4),
    ASIPARISGRUP4ID VARCHAR(4),
    ASIPARISIALANKULLANICIID INTEGER,
    AKUMASCINSIID INTEGER,
    AIPLIKCINSIID INTEGER,
    AKALINLIKID INTEGER,
    AYURTDISI INTEGER,
    AKUMASTEMINSIPARISI INTEGER,
    AFASONBOYASIPARISFIYATGOREBILIR INTEGER,
    ASATISBOYASIPARISFIYATGOREBILIR INTEGER,
    ASINIF VARCHAR(1),
    AKALITEID VARCHAR(20),
    ASIPONAYDURUMU SMALLINT,
    AMALIONAYDURUMU SMALLINT,
    AKAPATMATARIHIBASLANGIC TIMESTAMP,
    AKAPATMATARIHIBITIS TIMESTAMP,
    AIMALATTIPIID INTEGER)
RETURNS (
    BOYATALIMATKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYATALIMATNO VARCHAR(30),
    TALIMATTIPIID INTEGER,
    TALIMATTIPIIDSTR VARCHAR(10),
    TARIH TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    RENKADI VARCHAR(50),
    RECETEKODU VARCHAR(20),
    VARYANTADI VARCHAR(30),
    TALIMATMIKTARI DOUBLE PRECISION,
    GIDENMIKTAR DOUBLE PRECISION,
    IADEALINANMIKTAR DOUBLE PRECISION,
    BRUTGELENMIKTAR DOUBLE PRECISION,
    NETGELENMIKTAR DOUBLE PRECISION,
    BOYAHANEDEKALANMIKTAR DOUBLE PRECISION,
    BIRIMFIYAT DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    VADEGUN INTEGER,
    TERMINTARIHI TIMESTAMP)
AS
begin
  for Select BoyaTalimat.BoyaTalimatKayitId, BoyaTalimat.BoyaSiparisDetayId, BoyaTalimat.BoyaTalimatNo, BoyaTalimat.TalimatTipiId,
         BoyaTalimat.Tarih, BoyaTalimat.FirmaId, BTFirma.Adi, TalimatRenk.Adi, BoyaTalimat.ReceteKodu,
         DesenVaryant.VaryantAdi, BoyaTalimat.Miktar, BoyaTalimat.GidenMiktar, BoyaTalimat.IadeAlinanMiktar, BoyaTalimat.BrutGelenMiktar, BoyaTalimat.NetGelenMiktar,
         BoyaTalimat.BirimFiyat, BoyaTalimat.DovizId, BoyaTalimat.VadeGun, BoyaTalimat.TerminTarihi from OrmeSiparisRaporu_2(:ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ABaslangicSiparisTarihi, :ABitisSiparisTarihi, :AFirmaId,
              :AFASONSIPARISLER, :AFASONSATISSIPARISLER, :ASATISSIPARISLER, :AIplikSiparisi, :ABoyaliIplikSiparisi, :ALycraSiparisi, :AHamKumasSiparisi, :ABoyaliKumasSiparisi,
              :AIplikBoyaliKumasSiparisi, :AMusteriSiparisNo, :AReferansSiparisNo, :ARaporTipi, :AKapali, :ASiparisGrupId, :ASiparisGrup2Id, :ASiparisGrup3Id, :ASiparisGrup4Id, :ASiparisiAlanKullaniciId,
              :AKumasCinsiId, :AIplikCinsiId, :AKalinlikId, :AYurtDisi, :AKumasTeminSiparisi, :AFasonBoyaSiparisFiyatGorebilir, :ASatisBoyaSiparisFiyatGorebilir, :ASinif, :AKaliteId, :ASipOnayDurumu, :AMaliOnayDurumu, :akapatmatarihibaslangic,  :AKapatmaTarihiBitis, :AImalatTipiId)
    left join BoyaTalimat on BoyaTalimat.BoyaSiparisDetayId = OrmeSiparisRaporu_2.BoyaSiparisDetayId
    left join Renk TalimatRenk on TalimatRenk.RenkId = Boyatalimat.RenkId
    left join BoyaSiparisBaski on BoyaSiparisBaski.BoyaSiparisBaskiId = BoyaTalimat.BoyaSiparisBaskiId
    left join DesenVaryant on DesenVaryant.DesenVaryantId = BoyaSiparisBaski.DesenVaryantId
    left join Firma BTFirma on BTFirma.FirmaId = BoyaTalimat.FirmaId
    where BoyaTalimat.BoyaTalimatKayitId is not null
    into :BoyaTalimatKayitId, :BoyaSiparisDetayId, :BoyaTalimatNo, :TalimatTipiId, :Tarih, :FirmaId, :FirmaAdi, :RenkAdi, :ReceteKodu,
         :VaryantAdi, :TalimatMiktari, :GidenMiktar, :IadeAlinanMiktar, :BrutGelenMiktar, :NetGelenMiktar,
         :BirimFiyat, :DovizId, :VadeGun, :TerminTarihi do
  begin
    BoyahanedeKalanMiktar = GidenMiktar - IadeAlinanMiktar - BrutGelenMiktar;

    if (TalimatTipiId = 0) then
      TalimatTipiIdStr = 'Boya';
    else if (TalimatTipiId = 1) then
      TalimatTipiIdStr = 'Boya+Baskı';
    else if (TalimatTipiId = 2) then
      TalimatTipiIdStr = 'Baskı';
    suspend;
  end
end
