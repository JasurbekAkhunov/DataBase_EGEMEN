CREATE PROCEDURE MDEPOTRANSFERFISIDETAYGETIR(
    AMAMULDEPOTRANSFERFISIID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    PSKALITEKONTROLID INTEGER,
    IPLIKPAKETLEMEID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KALITENO INTEGER,
    TOPNO INTEGER,
    KAPADEDI INTEGER,
    BRUTKILO NUMERIC(12,3),
    BRUTMETRE NUMERIC(12,3),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    BOBINSAYISI INTEGER,
    ONCEKIRAFID VARCHAR(10),
    RAFID VARCHAR(10),
    FASON INTEGER,
    UCRETLITAMIR INTEGER,
    TAMIRTIPIADI VARCHAR(20),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    ISLEMGRUBUID INTEGER,
    ISLEMGRUBUADI VARCHAR(20),
    PAKETLEMESEKLIID INTEGER,
    PAKETLEMESEKLIADI VARCHAR(40),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    FIRMAKISAADI VARCHAR(10),
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    MUSTERISIPARISNO VARCHAR(50),
    TERMINTARIHI TIMESTAMP,
    MUSTERIRENKKODUVEADI VARCHAR(30),
    SIPARISRENKADI VARCHAR(40),
    MUSTERIIRSALIYENO VARCHAR(255),
    RECETEISEMRIIDSTR VARCHAR(30),
    BIRLESTIRILENPARTILER VARCHAR(100),
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    REFERANSKODU HAMKODUSIZE,
    BARKODID VARCHAR(20),
    DESENID VARCHAR(20),
    DESENADI VARCHAR(50),
    VARYANTADI VARCHAR(30),
    PARTISIRANO INTEGER,
    SEVKDURUMU INTEGER,
    SEVKDURUMUSTR VARCHAR(25),
    KALITEADI VARCHAR(250),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIADI ADI20)
AS
declare Variable FBoyaSiparisYil integer;
declare Variable FBoyaSiparisId  integer;
declare variable FAtisNoStr      Varchar(20);
declare variable Str             varchar(20);
declare variable FIrsaliyeNo     IrsaliyeNoSize;
declare variable FKoliNo         integer;
declare variable AsilRecetePartiAsamalariId integer;
declare variable FBarkodId       varchar(20);
begin
  for Select Parti.PartiKayitId, Parti.PartiYil, Parti.PartiId, MamulDepoTransferFisiDetay.PartiSinifId, MamulDepoTransferFisiDetay.PartiStokTartiId,
             PsKaliteKontrol.PsKaliteKontrolId, PartiStok.HamAdiToplami, PartiStokTarti.KaliteNo,
             IplikPaketleme.KoliNo, PsKaliteKontrol.TopNo,  PartiStokTarti.KapAdedi,
             PartiStokTarti.Kilo, PartiStokTarti.Metre,
             PartiStokTarti.BrutKilo, PartiStokTarti.BrutMetre,
             MamulDepoTransferFisiDetay.OncekiRafId, MamulDepoTransferFisiDetay.RafId,
             TamirTipi.Adi, Parti.IslemTipiId, IslemTipi.Adi, Parti.IslemGrubuId, IslemGrubu.Adi,
             PartiStok.PaketlemeSekliId, PaketlemeSekli.Adi, Parti.FirmaId, Firma.Adi, Firma.KisaAdi,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, PartiStok.BoyaSiparisDetayId,
             SiparisProses.Adi, Parti.ReceteId, Recete.Adi, BoyaSiparisDetay.LabReceteKodu, LabRecete.Adi, LabReceteAtis.AtisNoStr,
             BoyaSiparisDetay.MusteriSiparisNo, Parti.TerminTarihi, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.RenkAdi, Parti.Fason,
             Ik.Adi, MamulDepoTransferFisiDetay.InsertTarihi, Sk.Adi, MamulDepoTransferFisiDetay.Degisimtarihi, BoyaSiparisDetay.UcretliTamir,
             Ham.ReferansKodu, IplikPaketleme.IplikPaketlemeId, PartiStokTarti.BobinSayisi, Coalesce(IplikPaketleme.IplikPaketlemeId, PartiStokTarti.BarkodId),
             BoyaSiparisDetay.DesenId, Desen.Adi, DesenVaryant.VaryantAdi, PartiStok.PartiSiraNo, case when PartiStokTarti.BoyaIrsaliyeId is null then 0 else 1 end,
             Kalite.KaliteAdi, RenkDerinligi.RenkDerinligiId, RenkDerinligi.Adi from MamulDepoTransferFisiDetay
    left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = MamulDepoTransferFisiDetay.PartiStokTartiId
    left join PsKaliteKontrol on PsKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    left join IplikPaketleme on IplikPaketleme.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    left join PartiStok on PartiStok.PartiSinifId = MamulDepoTransferFisiDetay.PartiSinifId
    left join Ham on Ham.HamId = PartiStok.HamId
    left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Recete on Recete.ReceteId = Parti.ReceteId
    left join IslemTipi on IslemTipi.IslemTipiId = Parti.IslemTipiId
    left join IslemGrubu on IslemGrubu.IslemGrubuId = Parti.IslemGrubuId
    left join PaketlemeSekli on PaketlemeSekli.PaketlemeSekliId = PartiStok.PaketlemeSekliId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
    left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
    left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
    left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
    left join Desen on Desen.DesenId = BoyaSiparisDetay.DesenId
    Left Join DesenVaryant ON DesenVaryant.DesenVaryantId = Parti.DesenVaryantId
    left join LabReceteCalisma on LabReceteCalisma.LabReceteCalismaId = LabReceteAtis.LabReceteCalismaId
    Left Join Kullanici IK on IK.KullaniciId = MamulDepoTransferFisiDetay.InsertKullaniciId
    Left Join Kullanici SK on SK.KullaniciId = MamulDepoTransferFisiDetay.KullaniciId
    Left Join Kalite on  Kalite.KaliteId = BoyaSiparisDetay.KaliteId
    Left Join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
    where MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = :AMamulDepoTransferFisiId
  into :PartiKayitId, :PartiYil, :PartiId, :PartiSinifId, :PartiStokTartiId, :PsKaliteKontrolId, :HamAdiToplami, :KaliteNo, :FKoliNo, :TopNo, :KapAdedi,
       :Kilo, :Metre, :BrutKilo, :BrutMetre, :OncekiRafId, :RafId, :TamirTipiAdi,
       :IslemTipiId, :IslemTipiAdi, :IslemGrubuId, :IslemGrubuAdi,
       :PaketlemeSekliId, :PaketlemeSekliAdi, :FirmaId, :FirmaAdi, :FirmaKisaAdi, :FBoyaSiparisYil, :FBoyaSiparisId,
       :BoyaSiparisDetayId, :SiparisProsesAdi, :ReceteId, :ReceteAdi, :LabReceteKodu, :LabReceteAdi, :FAtisNoStr,
       :MusteriSiparisNo, :TerminTarihi, :MusteriRenkKoduveAdi, :SiparisRenkAdi, :Fason,
       :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :Degisimtarihi, :UcretliTamir, :ReferansKodu, :IplikPaketlemeId, :BobinSayisi, :FBarkodId,
       :DesenId, :DesenAdi, :VaryantAdi, :PartiSiraNo, :SevkDurumu, :KaliteAdi, :RenkDerinligiId, :RenkDerinligiAdi do
  begin
    BarkodId = '';
    if (IplikPaketlemeId is not null) then
      BarkodId =  'I-'||IplikPaketlemeId;
    else if (FBarkodID is not Null) then
      BarkodId = FBarkodID;
    else if (PsKaliteKontrolID is not null) then
      BarkodId = 'K-'||PsKaliteKontrolID;

    PartiIdStr = '(' || EgemenIntegerToStr(PartiYil) || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    if (FBoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(FBoyaSiparisYil)  || ') ' || EgemenIntegerToStr(FBoyaSiparisId) || ' ';
    else BoyaSiparisIdStr = '';

    LabReceteKodu = LabReceteKodu || '.' || FAtisNoStr;

    MusteriIrsaliyeNo = '';
    if (FKoliNo is not null) then Topno = FKoliNo;

    ReceteIsEmriIdStr = Null;
    BirlestirilenPartiler = Null;

    if (PartiKayitId is not null and ReceteId is not null) then
    begin
      AsilRecetePartiAsamalariId = Null;
      Select PartiAsamalariId from PartiAsamalari where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.PartininAsilRecetesimi = 1 into :AsilRecetePartiAsamalariId;

      if (AsilRecetePartiAsamalariId is not null) then
        Select ReceteIsEmriIdStr, BirlestirilenPartiler from PAReceteIsEmriBilgisiGetir(:PartikayitId, :ReceteId, :AsilRecetePartiAsamalariId) into :ReceteIsEmriIdStr, :BirlestirilenPartiler;
    end

    if (SevkDurumu = 0) then SevkDurumuStr = 'Sevk Edilmemiş';
      else if (SevkDurumu = 1) then SevkDurumuStr = 'Sevk Edilmiş';
      else     SevkDurumuStr = Null;



    if (Fason = 1) then
    begin
      For Select HamStokIrsaliye.IrsaliyeNo from PartiStokDetay
        left join RafDetay  on rafDetay.RafDetayId = PartiStokDetay.RafDetayId
        left Join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
        left Join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokDetay.HamStokIrsaliyeId
        where PartiStokDetay.PartiSinifId = :PartiSinifId
        Group By HamStokIrsaliye.IrsaliyeNo
        into :FIrsaliyeNo do
      begin
        if (FIrsaliyeNo is not null) Then Str = Format_Float('#0', FIrsaliyeNo);
        else Str = '';

        if (Str <> '') then
        if (PosBul(MusteriIrsaliyeNo, Str) = 0) then
        if (Char_Length(MusteriIrsaliyeNo) + Char_Length(Str) < 250) then
        begin
          if (MusteriIrsaliyeNo <> '') then MusteriIrsaliyeNo = MusteriIrsaliyeNo || cast(', ' as varchar(2) character set UTF8);
          MusteriIrsaliyeNo = MusteriIrsaliyeNo || Str;
        end
      end
    end

    suspend;
  end
end
