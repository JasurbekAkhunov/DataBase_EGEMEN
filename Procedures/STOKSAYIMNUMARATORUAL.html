CREATE PROCEDURE STOKSAYIMNUMARATORUAL(
    AYIL INTEGER)
RETURNS (
    ID INTEGER)
AS
begin
  select Id from StokSayimNumarator where Yil = :AYil into :Id;
  if (Id is null) then
  begin
    Id = 1;
    insert into StokSayimNumarator(Yil, Id) Values(:Ayil, :Id);
  end
  else
  begin
    Id = Id + 1;
    update StokSayimNumarator set Id = :Id where Yil = :Ayil;
  end
  suspend;
end
