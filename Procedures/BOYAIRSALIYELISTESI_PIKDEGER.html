AS
begin
  for SELECT PartiKayitId FROM BoyaIrsaliyeListesi(:ABaslangicTarihi, :ABitisTarihi, :ASatisTipi, :ARaporTipi, :AHamId, :AFirmaId,
    :AFirmaYetkiliId, :ASevkFirmaId, :ASiparisFirmaId, :AReceteId, :AIrsaliyeNo, :AFiyatOnay,

    Group By PartiKayitId
    into :PartiKayitId do

  begin
    for Select PartiAsamalariId, Sira, PartiSinifId, AsamaId, AsamaAdi, IcKontrolTipiId, IcKontrolTipiAdi, OnerilenDeger, Istenen, Sonuc, PersonelId, PersonelAdi from PartiIcKontrolDegerleriGetir(:PartiKayitId)
     into :PartiAsamalariId, :Sira, :PartiSinifId, :AsamaId, :AsamaAdi, :IcKontrolTipiId, :IcKontrolTipiAdi, :OnerilenDeger, :Istenen, :Sonuc, :PersonelId, :PersonelAdi do
    begin
      suspend;
    end
  end
end
