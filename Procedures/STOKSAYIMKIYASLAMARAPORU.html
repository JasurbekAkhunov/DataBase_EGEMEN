CREATE PROCEDURE STOKSAYIMKIYASLAMARAPORU(
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    ADEPOID VARCHAR(10),
    ASAYIMTARIHI TIMESTAMP,
    ALOTDETAYLI INTEGER,
    ABOYAURUNTIPI INTEGER,
    AKIMYASALURUNTIPI INTEGER,
    ADIGERURUNTIPI INTEGER,
    AYEDEKPARCAURUNTIPI INTEGER,
    ASARFMALZEMESIURUNTIPI INTEGER,
    AHIZMETURUNTIPI INTEGER,
    ASABLONURUNTIPI INTEGER,
    AAKSESUARURUNTIPI INTEGER,
    ATICARIMALURUNTIPI INTEGER)
RETURNS (
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    BIRIMID VARCHAR(2),
    URUNBIRIMID INTEGER,
    MEVCUTSTOKMIKTARI DOUBLE PRECISION,
    SAYIMMIKTARI DOUBLE PRECISION,
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    SAYIMFARKI DOUBLE PRECISION,
    LOT VARCHAR(150),
    SAYIMFISISTR VARCHAR(150))
AS
declare variable Okey integer;
declare variable HareketTipi            integer;
declare variable Miktar                 Double Precision;
declare variable FDepoId                varchar(10);
declare variable FTransferDepoId        varchar(10);
declare variable FSayimMiktar           Double Precision;
declare variable FStokSayimId           integer;
declare variable FSayimTarihi           TimeStamp;
begin
  FSayimTarihi = f_stripTime(ASayimTarihi) || ' 23:59:59';
  for Select Urun.UrunId, Urun.Adi,  Urun.urunGrup1, Urun.urunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, UrunBirim.BirimId, UrunBirim.UrunBirimId from urun
    left join UrunBirim on UrunBirim.urunId = Urun.UrunId and UrunBirim.DefaultBirim = 1
    where (:AUrunAralikli = 0 or (:AUrunAralikli = 1 and Urun.UrunId >= :ABaslangicUrunId) and (Urun.UrunId <= :ABitisUrunId or :ABitisUrunId  = '')) and
           ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicGrup1Id  and (UrunGrup1 <= :ABitisGrup1Id or :ABitisGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicGrup1Id = '' and :ABitisGrup1Id = '')) and
           ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicGrup2Id and (UrunGrup2 <= :ABitisGrup2Id or :ABitisGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicGrup2Id = '' and :ABitisGrup2Id = '')) and
           ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicGrup3Id and (UrunGrup3 <= :ABitisGrup3Id or :ABitisGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicGrup3Id = '' and :ABitisGrup3Id = '')) and
           ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicGrup4Id and (UrunGrup4 <= :ABitisGrup4Id or :ABitisGrup3Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicGrup4Id = '' and :ABitisGrup4Id = '')) and
           ((:ABoyaUrunTipi = 1 or Urun.UrunTipiId <> 0) and
            (:AKimyasalUrunTipi = 1 or Urun.UrunTipiId <> 1) and
            (:ADigerUrunTipi = 1 or Urun.UrunTipiId <> 2) and
            (:AYedekParcaUrunTipi = 1 or Urun.UrunTipiId <> 3) and
            (:ASarfMalzemesiUrunTipi = 1 or Urun.UrunTipiId <> 4) and
            (:AHizmetUrunTipi = 1 or Urun.UrunTipiId <> 5) and
            (:ASablonUrunTipi = 1 or Urun.UrunTipiId <> 6) and
            (:AAksesuarUrunTipi = 1 or Urun.UrunTipiId <> 7) and
            (:ATicariMalUrunTipi = 1 or Urun.UrunTipiId <> 8))
     into :UrunId, :UrunAdi, :UrunGrup1, :UrunGrup2, :UrunGrup3, :UrunGrup4, :BirimId, :UrunBirimId do
  begin
    Okey = 1;
    if (AUrunAralikli = 0) then
    if (KutuKarsilastir(ABaslangicUrunId, UrunId) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup1Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup1Id, UrunGrup1) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup2Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup2Id, UrunGrup2) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup3Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup3Id, UrunGrup3) = 0) then okey = 0;

    if (okey = 1) then
    if (AUrunGrup4Aralikli = 0) then
    if (KutuKarsilastir(ABaslangicGrup4Id, UrunGrup4) = 0) then okey = 0;

    if (Okey = 1) then
    begin
      if (:ALotDetayli = 1) then
      begin
        For select  distinct X.lot from
            (Select StokDetay.Lot from stokdetay
             where Stokdetay.Urunid = :UrunId and StokDetay.Tarih <= :FSayimTarihi and
                         ( (:ADepoId is null and StokDetay.HareketTipi <> 5) or (:ADepoId is not null and stokdetay.DepoId = :ADepoId or StokDetay.TransferDepoId = :ADepoId))
            union all
            select StokSayimDetay.Lot from StokSayimDetay
            Left Join StokSayim on StokSayim.StokSayimKayitId =StokSayimDetay.StokSayimKayitId
            Where StokSayimDetay.Urunid = :UrunId and StokSayim.Tarih = :ASayimTarihi and
                         (StrKarsilastir(StokSayim.DepoID, :ADepoId) = 1)
            )X
        into :Lot do
        begin
          MevcutStokMiktari = 0;
          SayimMiktari = 0;
          For Select StokDetay.HareketTipi, StokDetay.DepoId, StokDetay.TransferDepoId, sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from stokdetay
            left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
            where Stokdetay.Urunid = :UrunId and StokDetay.Tarih <= :FSayimTarihi and
                  ((:ADepoId is null and StokDetay.HareketTipi <> 5) or (:ADepoId is not null and stokdetay.DepoId = :ADepoId or StokDetay.TransferDepoId = :ADepoId))and
                  (StrKarsilastir(StokDetay.Lot, :Lot) = 1)
            group by StokDetay.HareketTipi, StokDetay.DepoId, StokDetay.TransferDepoId
            into :HareketTipi, :FDepoId, :FTransferDepoId, :Miktar do
            begin
              if (HareketTipi = 5 and ADepoId = FTransferDepoId) then MevcutStokMiktari = MevcutStokMiktari - Miktar;
              else MevcutStokMiktari = MevcutStokMiktari + Miktar;
            end
            SayimFisiStr = '';
          For Select StokSayimDetay.Miktar / UrunBirim.Carpan, StokSayim.StokSayimId from StokSayimDetay
            left join StokSayim on StokSayim.StokSayimKayitID = StokSayimDetay.StokSayimKayitId
            left join UrunBirim on UrunBirim.UrunBirimId = StokSayimDetay.UrunBirimId
            where StokSayimDetay.UrunID = :UrunId and StokSayim.Tarih = :ASayimTarihi and
                  StrKarsilastir(StokSayim.DepoID, :ADepoId) = 1 and
                  (StrKarsilastir(StokSayimDetay.Lot, :Lot) = 1)
            into :FSayimMiktar, :FStokSayimId do
            begin
              SayimMiktari = SayimMiktari + FSayimMiktar;
              if (SayimFisiStr = '') then
              begin
                SayimFisiStr = EgemenIntegerToStr(FStokSayimId);
              end
              else
              begin
                if (POSITION(EgemenIntegerToStr(FStokSayimId) IN SayimFisiStr) = 0) then  SayimFisiStr = SayimFisiStr || ', ' || EgemenIntegerToStr(FStokSayimId);
              end
            end
            if (SayimMiktari is null) then SayimMiktari = 0;

            if (SayimMiktari <> MevcutStokMiktari) then SayimFarki = SayimMiktari - MevcutStokMiktari;
            else SayimFarki = 0;

            Select Miktar from FormataGoreKiloYuvarla(:SayimMiktari, 'STOKRAPOR', 'STOKMIKTARI') into :SayimMiktari;
            Select Miktar from FormataGoreKiloYuvarla(:MevcutStokMiktari, 'STOKRAPOR', 'STOKMIKTARI') into :MevcutStokMiktari;

            if (SayimMiktari <> 0 or MevcutStokMiktari <> 0) then suspend;
        end
      end
      else
      begin
        MevcutStokMiktari = 0;
        SayimMiktari = 0;

        For Select StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId, sum(StokDetay.Miktar * StokDetay.GirisCikis / UrunBirim.Carpan) from stokdetay
          left join UrunBirim on UrunBirim.UrunBirimId = StokDetay.UrunBirimId
          where Stokdetay.Urunid = :UrunId and StokDetay.Tarih <= :FSayimTarihi and
             ( (:ADepoId is null and StokDetay.HareketTipi <> 5) or (:ADepoId is not null and stokdetay.DepoId = :ADepoId or StokDetay.TransferDepoId = :ADepoId))
        group by StokDetay.harekettipi, StokDetay.DepoId, StokDetay.TransferDepoId
        into :HareketTipi, :FDepoId, :FTransferDepoId, :Miktar do
        begin
          if (HareketTipi = 5 and ADepoId = FTransferDepoId) then MevcutStokMiktari = MevcutStokMiktari - Miktar;
          else MevcutStokMiktari = MevcutStokMiktari + Miktar;
        end

        For Select StokSayimDetay.Miktar / UrunBirim.Carpan, StokSayim.StokSayimId  from StokSayimDetay
          left join StokSayim on StokSayim.StokSayimKayitID = StokSayimDetay.StokSayimKayitId
          left join UrunBirim on UrunBirim.UrunBirimId = StokSayimDetay.UrunBirimId
          where StokSayimDetay.UrunID = :UrunId and StokSayim.Tarih = :ASayimTarihi and
                StrKarsilastir(StokSayim.DepoID, :ADepoId) = 1
          into :FSayimMiktar, :FStokSayimId do
          begin
            SayimMiktari = SayimMiktari + FSayimMiktar;
            if (SayimFisiStr = '') then
            begin
              SayimFisiStr = EgemenIntegerToStr(FStokSayimId);
            end
            else
            begin
              if (POSITION(EgemenIntegerToStr(FStokSayimId) IN SayimFisiStr) = 0) then  SayimFisiStr = SayimFisiStr || ', ' || EgemenIntegerToStr(FStokSayimId);
            end
          end

        if (SayimMiktari is null) then SayimMiktari = 0;

        if (SayimMiktari <> MevcutStokMiktari) then SayimFarki = SayimMiktari - MevcutStokMiktari;
        else SayimFarki = 0;

        Select Miktar from FormataGoreKiloYuvarla(:SayimMiktari, 'STOKRAPOR', 'STOKMIKTARI') into :SayimMiktari;
        Select Miktar from FormataGoreKiloYuvarla(:MevcutStokMiktari, 'STOKRAPOR', 'STOKMIKTARI') into :MevcutStokMiktari;

        if (SayimMiktari <> 0 or MevcutStokMiktari <> 0) then suspend;

      end
    end
  end
end
