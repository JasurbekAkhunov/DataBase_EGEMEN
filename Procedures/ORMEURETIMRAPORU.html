declare variable shtDevir integer;               -- Kullanılmıyor
declare variable shtAlis integer;                -- Kullanılmıyor

declare variable shtAlisIade integer;            -- Kullanılmıyor

  select uitBoya from UretimIslemTipiGetir into :uitBoya;
  OldOrmeSiparisDetayID = -1;
  ToplamBoyayaCikis = null;
  For Select StokHareket.StokHareketId, StokHareket.OrmeFisId, StokHareket.OrmeIrsaliyeId, StokHareket.OrmeFaturaId,
             StokHareket.HareketTipi, StokHareket.Tarih, UretimIslemTipi.Adi, OrmeFis.SeriNo, OrmeFis.FisNo, OrmeFis.FisTarihi,

             StokHareket.Notu, StokHareket.Aciklama, OrmePlanlama.IplikBoyu1, OrmePlanlama.IplikBoyu2, OrmePlanlama.IplikBoyu3, OrmePlanlama.IplikBoyu4, OrmePlanlama.BirimFiyat, OrmePlanlama.DovizId, StokHareket.Adet,

    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = StokHareket.BoyaSiparisKayitId
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = StokHareket.OrmeSiparisDetayId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = StokHareket.BoyaSiparisDetayId
    left join OrmeFis on OrmeFis.OrmeFisId = StokHareket.OrmeFisId

    left join OrmeFatura on OrmeFatura.OrmeFaturaId = StokHareket.OrmeFaturaId

    left join Firma SiparisFirma on SiparisFirma.FirmaId = BoyaSiparis.FirmaId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = StokHareket.OrmeMakinaId
    left join MakinaTipi on MakinaTipi.MakinaTipiId = OrmeMakina.MakinaTipiId
    left join MakinaGrup on MakinaGrup.MakinaGrupId = OrmeMakina.MakinaGrupId
    left join UretimIslemTipi on UretimIslemTipi.UretimIslemTipiId  = StokHareket.UretimIslemTipiId
    left join Personel VAPersonel on VAPersonel.PersonelId  = OrmeFis.VardiyaAmiriPersonelId
    left join Personel P on P.PersonelId  = StokHareket.PersonelId
    left join IplikCinsi on IplikCinsi.IplikCinsiId = Ham.IplikCinsiId
    Left Join KumasCinsi ON KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
    left join Kalinlik on Kalinlik.KalinlikId  = Ham.KalinlikId
    left join Kullanici IK on IK.KullaniciId  = StokHareket.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = StokHareket.KullaniciId
    left join SiparisGrup On SiparisGrup.SiparisGrupId = BoyaSiparis.SiparisGrupId
    left join SiparisGrup SiparisGrup2 On SiparisGrup2.SiparisGrupId = BoyaSiparis.SiparisGrup2Id
    left join OrmeSiparisTipi On OrmeSiparisTipi.OrmeSiparisTipiId = BoyaSiparis.OrmeSiparisTipiId
    left join OrmePlanlama on OrmePlanlama.OrmePlanlamaId = StokHareket.OrmePlanlamaId
    left join HamGrup on HamGrup.HamGrupId = Ham.HamGrup1
    left join HamGrup HG2 on HG2.HamGrupId = ham.HamGrup2
    left join Kalite on Kalite.KaliteId = OrmesiparisDetay.KaliteId
    --left join OrmeRaf on OrmeRaf.OrmeRafID = StokHareket.OrmeRafID
    where StokHareket.Tarih >= :ABaslangicTarihi and StokHareket.Tarih <= :ABitistarihi and
         ((StokHareket.HareketTipi = :ShtUretim or StokHareket.HareketTipi = :ShtFasonUretim) and StokHareket.HamIade = 0) and
         ((:AImalatSekli = 0 and StokHareket.HareketTipi = :ShtUretim) or :AImalatSekli <> 0 or :AImalatSekli is null) and
         ((:AImalatSekli = 1 and StokHareket.HareketTipi = :ShtFasonUretim) or :AImalatSekli <> 1 or :AImalatSekli is null) and
         (StokHareket.UretimIslemTipiId = :AUretimIslemTipiId or :AUretimIslemTipiId is null) and
         (ShFirma.AnaHesapFirmaId = :AFirmaId or :AFirmaId is null) and
         (SiparisFirma.AnaHesapFirmaId = :ASiparisFirmaId or :ASiparisFirmaId is null) and
         (OrmeMakina.UniteId = :AUniteId or :AUniteId is null) and
         ((:AMakinaKoduAralikli = 1 and StokHareket.OrmeMakinaId >= :ABaslangicMakinaId and (StokHareket.OrmeMakinaId <= :ABitisMakinaId or :ABitisMakinaId = '')) or
          (:AMakinaKoduAralikli = 0) or (:AMakinaKoduAralikli = 1 and :ABaslangicMakinaId = '' and :ABitisMakinaId = '')) and
         (OrmeMakina.MakinaGrupId = :AMakinaGrupId or :AMakinaGrupId is null) and
         (StokHareket.RenkId = :ARenkID Or :ARenkId is null) and
         (:AKumasCinsiId is null or (Ham.KumasCinsiId is not null and Ham.KumasCinsiId = :AKumasCinsiId)) and
         (:AIplikCinsiId is null or (Ham.IplikCinsiId is not null and Ham.IplikCinsiId = :AIplikCinsiId)) and
         (:AKalinlikId is null or (Ham.KalinlikId is not null and Ham.KalinlikId = :AKalinlikId)) and
         (StokHareket.FasonStok = :AImalatTipi or :AImalatTipi is null) and
         (StokHareket.HamId = :AHamId or :AHamId is null)
    order by StokHareket.OrmeSiparisDetayId
  into :StokHareketId, :OrmeFisId, :OrmeIrsaliyeId, :OrmeFaturaId, :HareketTipi, :Tarih, :UretimIslemTipiStr, :FisSeriNo, :FisNo, :FisTarihi, :Vardiya,
       :VardiyaAmiriPersonelId, :VardiyaAmiriPersonelAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi,
       :FaturaSeriNo, :FaturaNo, :FaturaTarihi, :FirmaId, :FirmaAdi, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :HamId,
       :HamAdi, :RenkAdi, :DesenId, :DesenAdi, :VaryantAdi, :KapAdedi, :Miktar, :OrmeMakinaId, :OrmeMakinaAdi, :MakinaGrupId, :MakinaGrupAdi,
       :MakinaPus, :MakinaFeine, :MakinaMarkasi, :MakinaTipiAdi,
       :UretimPartiNo, :PersonelId, :PersonelAdi, :InsertTarihi, :DegisimTarihi,
       :InsertKullaniciAdi, :KullaniciAdi, :FisSinif, :IrsaliyeSinif, :FaturaSinif,
       :SiparisFirmaId, :SiparisFirmaAdi, :BoyaSiparisDetayId, :OrmeSiparisDetayId, :SiparisGrupId, :SiparisGrupAdi, :SiparisGrup2Id,
       :SiparisGrup2Adi, :SiparisSiraNo, :IplikCinsiAdi, :KalinlikAdi, :KumasCinsiAdi, :Pus, :Feine, :Sakat, :AyarTopu,
       :OrmeBirimMaliyet, :OrmeBirimMaliyetDovizId, :OrmePlanlamaId, :Fason, :Lot, :HamMiktar, :OrmeSiparisTipiId, :OrmeSiparisTipiStr,
       :HamSatisTransferKodu, :FasonTransferKodu, :FasonOrmeTransferKodu, :FasonArtiSatOrmeTransferKodu, :AlisTransferKodu, :RenkliAlisTransferKodu, :FNotu, :Aciklama,
       :IplikBoyu1, :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :BirimFiyat, :DovizId, :Adet, :HamGrupAdi, :HamGrupId, :HamGrup2Adi, :HamGrup2Id, :PlanlamaTupAcikEnMayli,
     :OrmeSiparisKaliteId, :KaliteAdi, :SBirimFiyat, :IgneSifirlamaTarihi, :MusteriSiparisNo, :ReferansSiparisNo do
  begin
    Okey = 1;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:PlanlamaTupAcikEnMayli) into :PlanlamaTupAcikEnMayliStr;

    BlobSize = char_length(FNotu);

    if (BlobSize > 1023) then Notu = Left(FNotu, 1023);
    else if (BlobSize > 0) then Notu = Left(FNotu, BlobSize);
    else Notu = '';


    if (FaturaSinif is not null) then Sinif = FaturaSinif;
    else if (IrsaliyeSinif is not null) then Sinif = IrsaliyeSinif;
    else if (FisSinif is not null) then Sinif = FisSinif;

    if (AVardiya is not null and (Vardiya <> AVardiya or Vardiya is null)) then Okey = 0;
    else if (AVardiyaAmiriPersonelId is not null and (AVardiyaAmiriPersonelId <> VardiyaAmiriPersonelId or VardiyaAmiriPersonelId is null)) then Okey = 0;
    else if (APersonelId is not null and (APersonelId <> PersonelId or PersonelId is null)) then Okey = 0;

    if (okey = 1) then
    if (AMakinaKoduAralikli = 0) then
    if (KutuKarsilastir(ABaslangicMakinaId, OrmeMakinaId) = 0) then okey = 0;
    if (OrmeSiparisDetayId <> oldOrmeSiparisDetayID) then
    begin
      ToplamBoyayaCikis = null;
      oldOrmeSiparisDetayID = OrmeSiparisDetayId;
      Select Sum(StokHareket.Miktar) from StokHareket
      where StokHareket.OrmeSiparisDetayId = :OrmeSiparisDetayId and UretimIslemTipiId = :uitBoya and
           (StokHareket.HareketTipi = :shtFasonUretimeCikis or StokHareket.HareketTipi = :shtUretimeCikis)
      Plan(StokHareket Index (FK_SHOSiparisDetayId, StokHareket_IDX_UretimIslemTipi, StokHareket_IDX_HareketTipi, StokHareket_IDX_HareketTipi))
      into :ToplamBoyayaCikis;
    end

    if (Okey = 1) then
    begin
      KullanilanLotlar = '';

      BoyaSiparisIdStr = '';
      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

      if (Sakat = 1) then SakatStr = 'Sakat';
      else SakatStr = '';

      if (AyarTopu = 1) then AyarTopuStr = 'Ayar Topu';
      else AyarTopuStr = '';

      if (OrmePlanlamaId is not null) then
      begin
        For Select Distinct OrmeRaf.Lot from StokHareket
          left join OrmeRaf on OrmeRaf.OrmeRafID = StokHareket.OrmeRafID
          where StokHareket.OrmePlanlamaID = :OrmePlanlamaID and
               (StokHareket.HareketTipi = :ShtUretimeCikis or StokHareket.HareketTipi = :ShtFasonUretimeCikis)
          into :FLot do
        begin
          if (FLot is not Null) then
          begin
            if (Char_Length(KullanilanLotlar) + Char_Length(FLot) < 240) then
            begin
              if (KullanilanLotlar <> '') then KullanilanLotlar = KullanilanLotlar || ',';
              KullanilanLotlar = KullanilanLotlar || FLot;
            end
          end
        end
      end

      if (Fason = 1) then
        FasonStr = 'Fason';
      else
        FasonStr = 'Satış';

      if (ARafDetayi = 1) then
      begin
        RafID = '';
        For Select OrmeRafDetay.RafID, OrmeRafDetay.Miktar, OrmeRafDetay.KapAdedi from OrmeRafDetay
          where OrmeRafDetay.StokHareketID = :StokHareketID
          into :RafId, :Miktar, :KapAdedi do
        begin
          if (Miktar <> 0) then suspend;
        end

      end
      else
      begin
        RafID = '';
        For Select Distinct OrmeRafDetay.RafID from OrmeRafDetay
          where OrmeRafDetay.StokHareketID = :StokHareketID

          into :FRafId do
        begin
          if (Char_Length(RafId) + Char_Length(FRafId) < 99) then
          begin
            if (RafId <> '') then RafId = RafId || ',';
            RafId = RafId || FRafId;
          end
        end
        suspend;
      end
    end
  end
