CREATE PROCEDURE DOKUMAURETIMRAPORU_5(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AURETIMISLEMTIPIID INTEGER,
    ABASLANGICMAKINAID VARCHAR(10),
    ABITISMAKINAID VARCHAR(10),
    AMAKINAKODUARALIKLI INTEGER,
    AMAKINAGRUPID VARCHAR(4),
    AHAMID INTEGER,
    AMOD SMALLINT)
RETURNS (
    HAMID INTEGER,
    HAMADI ADI255,
    RECETEHAMID INTEGER,
    RECETEHAMADI ADI255,
    ATKICOZGU INTEGER,
    HESAPLANANCIKISKILO BOYAMIKTARSIZE,
    HESAPLANANCIKISMETRE BOYAMIKTARSIZE,
    DOKUNANMIKTAR BOYAMIKTARSIZE,
    PLANLANANMIKTAR BOYAMIKTARSIZE,
    SONALISFIYATI DOUBLE PRECISION,
    SONALISDOVIZID DOVIZSIZE,
    HATASTR BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    TALIMATCIKISMIKTARI BOYAMIKTARSIZE,
    TALIMATIADEMIKTARI BOYAMIKTARSIZE)
AS
declare variable SonUretimTarihi  timestamp;
declare variable DokumaPlanlamaId integer;
declare variable FHataStr         varchar(500);
declare variable TmpHataStr       blob;
declare variable FDokunanMiktar   BoyaMiktarSize;
declare variable FPlanlananMiktar BoyaMiktarSize;
declare variable SonHamId         integer;
declare variable UretimPartiNo    varchar(30);
declare variable LOT varchar(30);
declare variable RENKID integer;
begin
  if (AMod = 1) then
  begin
    SonHamId = 0;
    for select DokumaUretimRaporu.HamId, Ham.Adi, DokumaUretimRaporu.DokumaPlanlamaId, Sum(DokumaUretimRaporu.Miktar), Max(DokumaPlanlama.PlanlananMiktar), Max(DokumaPlanlama.UretimPartiNo)
      from DokumaUretimRaporu(:ABaslangicTarihi, :ABitisTarihi, :AUretimIslemTipiId, :ABaslangicMakinaId, :ABitisMakinaId, :AMakinaKoduAralikli, :AMakinaGrupId, :AHamId)
      left join DokumaPlanlama on DokumaPlanlama.DokumaPlanlamaId = DokumaUretimRaporu.DokumaPlanlamaId
      left join Ham on Ham.HamId = DokumaUretimRaporu.HamId
      group by DokumaUretimRaporu.HamId, Ham.Adi, DokumaUretimRaporu.DokumaPlanlamaID
      order by DokumaUretimRaporu.HamId
      into :HamId, :HamAdi, :DokumaPlanlamaId, :FDokunanMiktar, :FPlanlananMiktar, :UretimPartiNo do
      begin
        if (SonHamId <> HamId) then
        begin
          TmpHataStr = '';
          select Max(DokumaUretimRaporu.Tarih) from
          DokumaUretimRaporu(:ABaslangicTarihi, :ABitisTarihi, :AUretimIslemTipiId, :ABaslangicMakinaId, :ABitisMakinaId, :AMakinaKoduAralikli, :AMakinaGrupId, :HamId) into :SonUretimTarihi;
        end

        for select DokumaPlanlamaRecete.HamId, Ham.Adi, DokumaPlanlamaRecete.AtkiCozgu, DokumaIplikMiktariHesaplama.AgirlikKilo, DokumaIplikMiktariHesaplama.UzunlukMetre,
          DokumaIplikMiktariHesaplama.HataStr, DokumaPlanlamaRecete.CikisMiktari, DokumaPlanlamaRecete.IadeMiktari, DokumaPlanlamaRecete.renkid, DokumaPlanlamaRecete.lot
          from DokumaPlanlamaRecete
          left join DokumaIplikMiktariHesaplama(null, DokumaPlanlamaRecete.DokumaPlanlamaReceteID, null, 0, :FDokunanMiktar, null) on 1=1
          left join Ham on DokumaPlanlamaRecete.HamId = Ham.HamId
          where DokumaPlanlamaRecete.DokumaPlanlamaId = :DokumaPlanlamaId
          into :ReceteHamId, :ReceteHamAdi, :AtkiCozgu, :HesaplananCikisKilo, :HesaplananCikisMetre, :FHataStr, :TalimatCikisMiktari, :TalimatIadeMiktari, :renkid, :lot do
          begin
            /*Select Sum(CikisMiktari), Sum(IadeMiktari) from
              (Select iif((HareketTipi = 6 or HareketTipi = 7) and (HamIade = 0), Miktar,0) CikisMiktari,
                      iif((HareketTipi = 8 or HareketTipi = 9) and (HamIade = 1), Miktar, 0) IadeMiktari, Lot from StokHareket
              where StokHareket.DokumaPlanlamaId = :DokumaPlanlamaId and StokHareket.HamId = :ReceteHamId and
              IntKarsilastir(StokHareket.RenkId, :RenkId) = 1
              --and (:Lot is Null or StokHareket.Lot = :Lot)
              ) into :TalimatCikisMiktari, :TalimatIadeMiktari;*/

              TalimatIadeMiktari = null;

              select sum(SHCikis.Miktar)
              from StokHareket
              left join DokumaKaliteKontrol on DokumaKaliteKontrol.StokHareketId = StokHareket.StokHareketId
              left join OrmeFis on OrmeFis.StokHareketId = StokHareket.StokHareketId
              left join StokHareket SHCikis on SHCikis.OrmeFisId = OrmeFis.OrmeFisId
              where (DokumaKaliteKontrol.StokHareketId is null and StokHareket.Tarih >= :ABaslangicTarihi and StokHareket.Tarih <= :ABitistarihi or
                    DokumaKaliteKontrol.StokHareketId is not null and DokumaKaliteKontrol.Tarih >= :ABaslangicTarihi and DokumaKaliteKontrol.Tarih <= :ABitistarihi + 1) and
                    StokHareket.DokumaPlanlamaId = :DokumaPlanlamaId and
                    SHCikis.HamId = :ReceteHamId and
                    StrKarsilastir(SHCikis.RenkId, :RenkId) = 1 and
                    OrmeFis.HareketTipi = 7 and
                    StokHareket.HamIade = 0 and
                    (StokHareket.HareketTipi = 8 or StokHareket.HareketTipi = 9) into :TalimatCikisMiktari;


            if (FHataStr <> '') then
            begin
              if (TmpHataStr = '') then
                TmpHataStr = '('||UretimPartiNo||'):';
              if (Position(FHataStr, TmpHataStr) = 0) then
                TmpHataStr = TmpHataStr ||' '||ReceteHamAdi||': '||FHataStr;
            end

            select OrmeHamStokSonAlisFiyati_v2.SonAlisFiyati, OrmeHamStokSonAlisFiyati_v2.SonAlisDovizId from
            OrmeHamStokSonAlisFiyati_v2(:ReceteHamId, :SonUretimTarihi) into :SonAlisFiyati, :SonAlisDovizId;

            suspend;

            TalimatCikisMiktari = null;
            TalimatIadeMiktari = null;
            if (AtkiCozgu = 2) then
            begin
              for select HamStokRecete.ReceteHamId, Ham.Adi, 3, DokumaIplikMiktariHesaplama.AgirlikKilo, null, DokumaIplikMiktariHesaplama.HataStr
                from HamStokRecete
                left join DokumaIplikMiktariHesaplama(null, null, null, 0, :FDokunanMiktar, HamStokRecete.HamStokReceteId) on 1=1
                left join Ham on HamStokRecete.ReceteHamId = Ham.HamId
                where HamStokRecete.HamId = :ReceteHamId
                into :ReceteHamId, :ReceteHamAdi, :AtkiCozgu, :HesaplananCikisKilo, :HesaplananCikisMetre, :FHataStr do
                begin
                  if (FHataStr <> '') then
                  begin
                    if (TmpHataStr = '') then
                      TmpHataStr = '('||UretimPartiNo||'):';
                    if (Position(FHataStr, TmpHataStr) = 0) then
                      TmpHataStr = TmpHataStr ||' '||ReceteHamAdi||': '||FHataStr;
                  end

                  select OrmeHamStokSonAlisFiyati_v2.SonAlisFiyati, OrmeHamStokSonAlisFiyati_v2.SonAlisDovizId from
                  OrmeHamStokSonAlisFiyati_v2(:ReceteHamId, :SonUretimTarihi) into :SonAlisFiyati, :SonAlisDovizId;

                  suspend;
                end
            end
          end
        SonHamId        = HamId;
        DokunanMiktar   = FDokunanMiktar;
        PlanlananMiktar = FPlanlananMiktar;
        ReceteHamId     = null;
        ReceteHamAdi    = null;
        AtkiCozgu = null;
        HesaplananCikisKilo = null;
        HesaplananCikisMetre = null;
        TalimatCikisMiktari = null;
        TalimatIadeMiktari = null;
        HataStr = TmpHataStr;
        suspend; /*en son baslik satiri eklenir*/
        HataStr = '';
        DokunanMiktar = null;
        PlanlananMiktar = null;
      end
  end
  else if (AMod = 2) then
  begin
    for select
    DOKUMAURETIMRAPORU_5.hamid, DOKUMAURETIMRAPORU_5.hamadi, DOKUMAURETIMRAPORU_5.recetehamid, DOKUMAURETIMRAPORU_5.recetehamadi,
    DOKUMAURETIMRAPORU_5.atkicozgu, sum(DOKUMAURETIMRAPORU_5.hesaplanancikiskilo) hesaplanancikiskilo, sum(DOKUMAURETIMRAPORU_5.hesaplanancikismetre) hesaplanancikismetre,
    sum(DOKUMAURETIMRAPORU_5.dokunanmiktar) dokunanmiktar, sum(DOKUMAURETIMRAPORU_5.planlananmiktar) planlananmiktar, max(DOKUMAURETIMRAPORU_5.sonalisfiyati) sonalisfiyati,
    max(DOKUMAURETIMRAPORU_5.SonAlisDovizId) sonalisdovizid, sum(DOKUMAURETIMRAPORU_5.talimatcikismiktari) talimatcikismiktari, sum(DOKUMAURETIMRAPORU_5.talimatiademiktari) talimatiademiktari,
    max(DOKUMAURETIMRAPORU_5.hatastr) hatastr
    from DokumaUretimRaporu_5(:ABaslangicTarihi, :ABitisTarihi, :AUretimIslemTipiId, :ABaslangicMakinaId, :ABitisMakinaId, :AMakinaKoduAralikli, :AMakinaGrupId, :AHamId, 1)
    group by DOKUMAURETIMRAPORU_5.hamid, DOKUMAURETIMRAPORU_5.hamadi, DOKUMAURETIMRAPORU_5.recetehamid, DOKUMAURETIMRAPORU_5.recetehamadi, DOKUMAURETIMRAPORU_5.atkicozgu
    order by HamId, AtkiCozgu nulls first
    into :HamId, :HamAdi, :ReceteHamId, :ReceteHamAdi, :AtkiCozgu, :HesaplananCikisKilo, :HesaplananCikisMetre, :DokunanMiktar, :PlanlananMiktar, :SonAlisFiyati, :SonAlisDovizId,
    :TalimatCikisMiktari, :TalimatIadeMiktari, :HataStr do
      suspend;
  end
end
