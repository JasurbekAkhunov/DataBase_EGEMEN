CREATE PROCEDURE KARTBILGISIGETIR(
    APROXYID VARCHAR(8))
RETURNS (
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(20),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    KILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION,
    TUMASAMALARIBITMIS INTEGER,
    YAPILANSONISLEMMAKINAID VARCHAR(255),
    YAPILANSONISLEMASAMAID INTEGER,
    YAPILANSONISLEMASAMAADI VARCHAR(40),
    SONISLEMDEVAMEDIYOR INTEGER,
    SONRAKIISLEMASAMAID INTEGER,
    SONRAKIISLEMASAMAADI VARCHAR(40),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    PARTINOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    BOYASIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50))
AS
declare variable SonIslemSiraNo integer;
declare variable SonrakiIslemSiraNo integer;
declare variable CikisTarihi        timestamp;
declare variable FMakinaId          varchar(10);
declare variable FYapilanSonIslemPAId    integer;
begin
  select PartiYil, PartiId, Parti.PartiKayitId, PartiStok.PartiSinifId, PartiStok.Kilo, PartiStok.Metre, PartiStok.HamAdiToplami,
         PartiStok.EnIstenen, PartiStok.GramajIstenen, Parti.Notu, PartiStok.BoyaSiparisDetayId, BoyaSiparis.BoyaSiparisYil,
         BoyaSiparis.BoyaSiparisId, BoyaSiparisDetay.MusteriSiparisNo from Partistok
  left join Parti on Parti.PartiKayitId = Partistok.PartiKayitId
  left join BoyaSiparis on BoyaSiparis.BoyaSipariskayitid = Parti.BoyaSipariskayitid
  left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayid = PartiStok.BoyaSiparisDetayid
  where Partistok.Proxyid = :AProxyId
  into :PartiYil, :PartiId, :PartiKayitId, :PartiSinifId, :Kilo, :Metre, :HamAdiToplami, :EnIstenen, :GramajIstenen, :PartiNotu,
       :BoyaSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :MusteriSiparisNo;

  PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
  BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

  if (PartiSinifId is not null) then
  begin
    select max(sira) from PartiAsamaStok
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
    where PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamalari.OtomasyonaDahilmi = 1 and
          PartiAsamaStok.GirisTarihi is not null and PartiAsamaStok.AsamayaDahil = 1
    into :SonIslemSiraNo;

    YapilanSonIslemMakinaId = '';

    if (SonIslemSiraNo is not null) then
    begin
      select PartiAsamalari.PartiAsamalariId, Asama.AsamaId, Asama.Adi, CikisTarihi from PartiAsamalari
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId
        left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId or Asama.AsamaId = Recete.UstAsamaId
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.sira = :SonIslemSiraNo
        into :FYapilanSonIslemPAId, :YapilanSonIslemAsamaId, :YapilanSonIslemAsamaAdi, :CikisTarihi;

      for select PartiAsamaStokDetay.MakinaId from PartiAsamaStokDetay
        where PartiAsamaStokDetay.PartiAsamalariId = :FYapilanSonIslemPAId and PartiAsamaStokDetay.PartiSinifId = :PartiSinifId
        group by PartiAsamaStokDetay.MakinaId
        into :FMakinaId do
      begin
        if (Char_Length(YapilanSonIslemMakinaId) + Char_Length(FMakinaId) < 250) then
        begin
          if (YapilanSonIslemMakinaId <> '') then YapilanSonIslemMakinaId = YapilanSonIslemMakinaId || ',';

          YapilanSonIslemMakinaId = YapilanSonIslemMakinaId || FMakinaId;
        end
      end

      if (CikisTarihi is not null) then SonIslemDevamEdiyor = 0;
      else SonIslemDevamEdiyor = 1;
    end
    else
    begin
      SonIslemSiraNo = -1;
      SonIslemDevamEdiyor = 0;
    end

    select min(sira) from PartiAsamalari
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiSinifId = :PartiSinifId
      where PartiAsamalari.PartiKayitId  = :PartiKayitId and PartiAsamalari.OtomasyonaDahilmi = 1 and
            PartiAsamalari.Sira > :SonIslemSiraNo and PartiAsamaStok.AsamayaDahil = 1
      into :SonrakiIslemSiraNo;

    if (SonrakiIslemSiraNo is Not null) then
    begin
      select Asama.AsamaId, Asama.Adi from PartiAsamalari
        left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStok.PartiSinifId = :PartiSinifId
        left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join Asama on Asama.AsamaId = PartiAsamalari.AsamaId or Asama.AsamaId = Recete.UstAsamaId
        where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.sira = :SonrakiIslemSiraNo and PartiAsamaStok.AsamayaDahil = 1
        into :SonrakiIslemAsamaId, :SonrakiIslemAsamaAdi;
        TumAsamalariBitmis = 0;
    end
    else if (SonIslemDevamEdiyor = 1) then TumAsamalariBitmis = 0;
    else TumAsamalariBitmis = 1;
    suspend;
  end
end
