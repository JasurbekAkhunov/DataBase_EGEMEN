CREATE PROCEDURE LRTALEPICINHAMSTOKDETAYLISTESI(
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AIRSALIYENO IRSALIYENOSIZE)
RETURNS (
    HAMSTOKMASTERID INTEGER,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    FASONSTR VARCHAR(5),
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    LOT VARCHAR(150),
    PUS INTEGER,
    FEINE INTEGER,
    GIRISMIKTAR NUMERIC(12,3))
AS
declare variable Fason  integer;
begin
  for select HamStokMaster.HamStokMasterId, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             HamStokIrsaliye.Fason, HamStokMaster.Lot, HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.Miktar from HamStokMaster
  left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeid = HamStokMaster.HamStokIrsaliyeid
  where ( HamStokMaster.HamID = :AHamId ) and
        ( ( HamStokIrsaliye.IrsaliyeTarihi >= :ABaslangicTarihi and HamStokIrsaliye.IrsaliyeTarihi <= :ABitisTarihi ) or :AIrsaliyeNo is not null) and
        ( HamStokIrsaliye.IrsaliyeNo = :AIrsaliyeNo or :AIrsaliyeNo is null) and
        ( HamStokMaster.StokFirmaId = :AFirmaId or HamStokIrsaliye.Fason = 0)
  into :HamStokMasterId, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi, :Fason, :Lot, :Pus, :Feine, :GirisMiktar do
  begin
    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';
    suspend;
  end
end
