CREATE PROCEDURE FIRMAALISSATISANALIZI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI VARCHAR(20),
    ABASLANGICFIRMAID VARCHAR(20),
    ABITISFIRMAID VARCHAR(20),
    ABASLANGICFIRMAGRUP1ID VARCHAR(4),
    ABITISFIRMAGRUP1ID VARCHAR(4),
    ABASLANGICFIRMAGRUP2ID VARCHAR(4),
    ABITISFIRMAGRUP2ID VARCHAR(4),
    ABASLANGICFIRMAGRUP3ID VARCHAR(4),
    ABITISFIRMAGRUP3ID VARCHAR(4),
    ABASLANGICURUNID VARCHAR(20),
    ABITISURUNID VARCHAR(20),
    ABASLANGICURUNGRUP1ID VARCHAR(4),
    ABITISURUNGRUP1ID VARCHAR(4),
    ABASLANGICURUNGRUP2ID VARCHAR(4),
    ABITISURUNGRUP2ID VARCHAR(4),
    ABASLANGICURUNGRUP3ID VARCHAR(4),
    ABITISURUNGRUP3ID VARCHAR(4),
    ABASLANGICURUNGRUP4ID VARCHAR(4),
    ABITISURUNGRUP4ID VARCHAR(4),
    AFIRMAARALIKLI INTEGER,
    AFIRMAGRUP1ARALIKLI INTEGER,
    AFIRMAGRUP2ARALIKLI INTEGER,
    AFIRMAGRUP3ARALIKLI INTEGER,
    AURUNARALIKLI INTEGER,
    AURUNGRUP1ARALIKLI INTEGER,
    AURUNGRUP2ARALIKLI INTEGER,
    AURUNGRUP3ARALIKLI INTEGER,
    AURUNGRUP4ARALIKLI INTEGER,
    ABASLANGICFATURAGRUPID VARCHAR(4),
    ABITISFATURAGRUPID VARCHAR(4),
    AFATURAGRUPARALIKLI INTEGER,
    ABASLANGICURUNKATEGORIID VARCHAR(4),
    ABITISURUNKATEGORIID VARCHAR(4),
    AURUNKATEGORIARALIKLI INTEGER,
    AALISSATIS INTEGER,
    ARAPORDOVIZID VARCHAR(5),
    AURUNTIPIID INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    URUNID VARCHAR(20),
    URUNADI VARCHAR(80),
    MIKTAR DOUBLE PRECISION,
    BIRIMID VARCHAR(2),
    BRUTTUTAR DOUBLE PRECISION,
    NETTUTAR DOUBLE PRECISION,
    KDVTUTARI DOUBLE PRECISION,
    KDVLITUTAR DOUBLE PRECISION,
    URUNKATEGORIID VARCHAR(4),
    URUNGRUP1 VARCHAR(4),
    URUNGRUP2 VARCHAR(4),
    URUNGRUP3 VARCHAR(4),
    URUNGRUP4 VARCHAR(4),
    URUNTIPIID INTEGER,
    URUNTIPIADI VARCHAR(20),
    URUNGRUP1ADI VARCHAR(30),
    URUNGRUP2ADI VARCHAR(30),
    URUNGRUP3ADI VARCHAR(30),
    URUNGRUP4ADI VARCHAR(30))
AS
declare variable FAltIskontoCarpani     Double Precision;
declare variable Okey                   integer;
declare variable FHareketTipi           integer;
declare variable FLastFirmaId                   varchar(20);
declare variable FLastUrunId                    Varchar(20);
declare variable FFirmaId                       varchar(20);
declare variable FFaturaGrupId                  varchar(20);
declare variable Kayitvar                       integer;
declare variable FFirmaAdi                      varchar(100);
declare variable FUrunId                        Varchar(20);
declare variable FUrunAdi                       Varchar(80);
declare variable FBirimId                       varchar(2);
declare variable FMiktar                        double precision;
declare variable FKdvOrani                      double precision;
declare variable FBrutTutar                     Double Precision;
declare variable FFirmaGrup1                    varchar(4);
declare variable FFirmaGrup2                    varchar(4);
declare variable FFirmaGrup3                    varchar(4);
declare variable FUrunGrup1                     varchar(4);
declare variable FUrunGrup2                     varchar(4);
declare variable FUrunGrup3                     varchar(4);
declare variable FUrunGrup4                     varchar(4);
declare variable FUrunKategoriId                varchar(4);
declare variable FNetTutar                      Double Precision;
declare variable FKdvTutari                     Double Precision;
declare variable HareketTipi1                   integer;
declare variable HareketTipi2                   integer;
declare variable FStokTabanDovizKuru            Double Precision;
declare variable FIrsaliyeTabanDovizKuru        Double Precision;
declare variable FFaturaTabanDovizKuru          Double Precision;

declare variable FTabanDovizKuru                Double Precision;
declare variable FDovizKuru                     Double Precision;

declare variable ARaportarihi                   date;
declare variable FStokTabanDovizId              varchar(5);
declare variable FNetFiyatYuvarlamaCarpani      double precision;
declare variable FaturaGrupOkey         integer;
declare variable FTarih                 TimeStamp;
declare variable FTevkifat                      integer;
declare variable FTevkifatOrani1                integer;
declare variable FTevkifatOrani2                integer;
begin
  FLastFirmaId = '';
  FLastUrunId = '';
  Okey = 0;
  if (AAlisSatis = 1) then
  begin
    HareketTipi1 = 1;
    HareketTipi2 = 3;
  end
  else
  if (AAlisSatis = 2) then
  begin
    HareketTipi1 = 2;
    HareketTipi2 = 4;
  end
  else exit;

  ARaportarihi = Current_TimeStamp(0);
  select STOKTABANDOVIZID from SISTEM where SISTEMID = 1 into :FStokTabanDovizId;
/*
  if (FStokTabanDovizId <> ARaporDovizId) then  select DovizKuru from EnYakinDovizKuruGetir(:FStokTabanDovizId, :ARaporDovizId, :ARaporTarihi) into :FDovizKuru;
  else FDovizKuru = 1;
*/
  select NetFiyatYuvarlamaCarpani from Doviz where DovizId = :ARaporDovizId  into :FNetFiyatYuvarlamaCarpani;

  For Select Firma.firmaid, Firma.Adi, Firma.FirmaGrup1, Firma.FirmaGrup2,Firma.FirmaGrup3, Stokdetay.Urunid, Urun.Adi,
             Urun.UrunKategoriId, Urun.UrunGrup1, Urun.UrunGrup2, Urun.UrunGrup3, Urun.UrunGrup4, DUB.Birimid, Miktar / HUB.Carpan * DUB.Carpan,
             Miktar * NetBirimfiyat, Fatura.AltIskontoCarpani, Stokdetay.Kdv, StokDetay.HareketTipi, Stok.TabanDovizKuru2 / Stok.TabanDovizKuru1,
             Irsaliye.TabanDovizKuru2 / Irsaliye.TabanDovizKuru1, Fatura.TabanDovizKuru2 /  Fatura.TabanDovizKuru1, Fatura.FaturaGrupId, StokDetay.Tarih,
             Fatura.Tevkifat, Fatura.TevkifatOrani1, Fatura.TevkifatOrani2, Urun.UrunTipiId, UrunTipi.Adi,
             UG1.Adi,UG2.Adi, UG3.Adi, UG4.Adi from stokdetay
  left join stok on stok.stokid = stokdetay.stokid
  left join irsaliye on irsaliye.irsaliyeid = stokdetay.irsaliyeid
  left join fatura on fatura.faturaid = stokdetay.faturaid
  left join firma on Firma.firmaid = StokDetay.firmaid
  left join urun on urun.urunid = StokDetay.urunId
  left join UrunBirim HUB on HUB.UrunId = stokdetay.Urunid and HUB.urunbirimid = stokdetay.urunbirimid
  left join UrunBirim DUB on DUB.UrunId = stokdetay.Urunid and DUB.DefaultBirim = 1
  left join UrunTipi on UrunTipi.UrunTipiId = Urun.UrunTipiId
  left join UrunGrup UG1 on UG1.UrunGrupId = urun.UrunGrup1
  left join UrunGrup UG2 on UG2.UrunGrupId = urun.UrunGrup2
  left join UrunGrup UG3 on UG3.UrunGrupId = urun.UrunGrup3
  left join UrunGrup UG4 on UG4.UrunGrupId = urun.UrunGrup4
  where (stokdetay.harekettipi = :HareketTipi1 or stokdetay.harekettipi = :HareketTipi2 ) and tarih >= :ABaslangicTarihi and tarih <= :ABitisTarihi and
        ((:AFirmaAralikli = 1 and Firma.FirmaId >= :ABaslangicFirmaId and (Firma.FirmaId <= :ABitisFirmaId or :ABitisFirmaId = '')) or (:AFirmaAralikli = 0))  and
        ((:AFirmaGrup1Aralikli = 1 and Firma.FirmaGrup1 >= :ABaslangicFirmaGrup1Id  and (Firma.FirmaGrup1 <= :ABitisFirmaGrup1Id or :ABitisFirmaGrup1Id = '')) or (:AFirmaGrup1Aralikli = 0) or (:AFirmaGrup1Aralikli = 1 and :ABaslangicFirmaGrup1Id = '' and :ABitisFirmaGrup1Id = '')) and
        ((:AFirmaGrup2Aralikli = 1 and Firma.FirmaGrup2 >= :ABaslangicFirmaGrup2Id and (Firma.FirmaGrup2 <= :ABitisFirmaGrup2Id or :ABitisFirmaGrup2Id = '')) or (:AFirmaGrup2Aralikli = 0) or (:AFirmaGrup2Aralikli = 1 and :ABaslangicFirmaGrup2Id = '' and :ABitisFirmaGrup2Id = '')) and
        ((:AFirmaGrup3Aralikli = 1 and Firma.FirmaGrup3 >= :ABaslangicFirmaGrup3Id and (Firma.FirmaGrup3 <= :ABitisFirmaGrup3Id or :ABitisFirmaGrup3Id = '')) or (:AFirmaGrup3Aralikli = 0) or (:AFirmaGrup3Aralikli = 1 and :ABaslangicFirmaGrup3Id = '' and :ABitisFirmaGrup3Id = '')) and
        ((:AFaturaGrupAralikli = 1 and (Fatura.FaturaGrupId >= :ABaslangicFaturaGrupId or :ABaslangicFaturaGrupId = '') and (Fatura.FaturaGrupId <= :ABitisFaturaGrupId or :ABitisFaturaGrupId = '')) or (:AFaturaGrupAralikli = 0)) and
        ((:AUrunAralikli = 1 and StokDetay.UrunId >= :ABaslangicUrunId and (StokDetay.UrunId <= :ABitisUrunId or :ABitisUrunId = '')) or (:AUrunAralikli = 0)) and
        ((:AUrunGrup1Aralikli = 1 and UrunGrup1 >= :ABaslangicUrunGrup1Id  and (UrunGrup1 <= :ABitisUrunGrup1Id or :ABitisUrunGrup1Id = '')) or (:AUrunGrup1Aralikli = 0) or (:AUrunGrup1Aralikli = 1 and :ABaslangicUrunGrup1Id = '' and :ABitisUrunGrup1Id = '')) and
        ((:AUrunGrup2Aralikli = 1 and UrunGrup2 >= :ABaslangicUrunGrup2Id and (UrunGrup2 <= :ABitisUrunGrup2Id or :ABitisUrunGrup2Id = '')) or (:AUrunGrup2Aralikli = 0) or (:AUrunGrup2Aralikli = 1 and :ABaslangicUrunGrup2Id = '' and :ABitisUrunGrup2Id = '')) and
        ((:AUrunGrup3Aralikli = 1 and UrunGrup3 >= :ABaslangicUrunGrup3Id and (UrunGrup3 <= :ABitisUrunGrup3Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup3Aralikli = 0) or (:AUrunGrup3Aralikli = 1 and :ABaslangicUrunGrup3Id = '' and :ABitisUrunGrup3Id = '')) and
        ((:AUrunGrup4Aralikli = 1 and UrunGrup4 >= :ABaslangicUrunGrup4Id and (UrunGrup4 <= :ABitisUrunGrup4Id or :ABitisUrunGrup3Id = '')) or (:AUrunGrup4Aralikli = 0) or (:AUrunGrup4Aralikli = 1 and :ABaslangicUrunGrup4Id = '' and :ABitisUrunGrup4Id = '')) and
        ((:AUrunKategoriAralikli = 1 and UrunKategoriId >= :ABaslangicUrunKategoriId and (UrunKategoriId <= :ABitisUrunKategoriId or :ABitisUrunKategoriId = '')) or (:AUrunKategoriAralikli = 0) or (:AUrunKategoriAralikli = 1 and :ABaslangicUrunKategoriId = '' and :ABitisUrunKategoriId = '')) and
        (Urun.UrunTipiId = :AUrunTipiId or :AUrunTipiId is null)
  order by firma.firmaid, stokdetay.urunid
  into :FFirmaid, :FFirmaAdi, :FFirmaGrup1, :FFirmaGrup2, :FFirmaGrup3, :FUrunid, :FUrunAdi, :FUrunKategoriId, :FUrunGrup1, :FUrunGrup2,
       :FUrunGrup3, :FUrunGrup4, :FBirimid, :FMiktar, :FBrutTutar, :FAltIskontoCarpani, :FKdvOrani,
       :FHareketTipi, :FStokTabanDovizKuru, :FIrsaliyeTabanDovizKuru, :FfaturaTabanDovizKuru, :FFaturaGrupId, :FTarih,
       :FTevkifat, :FTevkifatOrani1, :FTevkifatOrani2,
       :UrunTipiId, :UrunTipiAdi, :UrunGrup1Adi, :UrunGrup2Adi, :UrunGrup3Adi, :UrunGrup4Adi do
  begin
    if (FLastFirmaId <> FFirmaId or FLastUrunId <> FUrunId) then
    begin
      if (FLastFirmaId <> '' and FLastUrunId <> '') then
      begin
        if (Okey = 1 and KayitVar = 1) then suspend;
      end
      FLastFirmaId = FFirmaId;
      FLastUrunId = FUrunId;
      FirmaId = FFirmaId;
      UrunId = FUrunId;
      BirimId = FBirimId;
      UrunAdi = FUrunAdi;
      FirmaAdi = FFirmaAdi;
      UrunGrup1 = FUrunGrup1;
      UrunGrup2 = FUrunGrup2;
      UrunGrup3 = FUrunGrup3;
      UrunGrup4 = FUrunGrup4;
      UrunKategoriId = FUrunKategoriId;

      Kayitvar = 0;
      BrutTutar   = 0;
      NetTutar    = 0;
      KdvTutari   = 0;
      KdvliTutar  = 0;
      Miktar = 0;
      Okey = 1;
      if (AFirmaAralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaId, FFirmaId) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup1Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup1Id, FFirmaGrup1) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup2Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup2Id, FFirmaGrup2) = 0) then okey = 0;

      if (okey = 1) then
      if (AFirmaGrup3Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicFirmaGrup3Id, FFirmaGrup3) = 0) then okey = 0;

      if (AUrunAralikli = 0) then
      if (KutuKarsilastir(ABaslangicUrunId, FUrunId) = 0) then okey = 0;

      if (okey = 1) then
      if (AUrunGrup1Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicUrunGrup1Id, FUrunGrup1) = 0) then okey = 0;

      if (okey = 1) then
      if (AUrunGrup2Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicUrunGrup2Id, FUrunGrup2) = 0) then okey = 0;

      if (okey = 1) then
      if (AUrunGrup3Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicUrunGrup3Id, FUrunGrup3) = 0) then okey = 0;

      if (okey = 1) then
      if (AUrunGrup4Aralikli = 0) then
      if (KutuKarsilastir(ABaslangicUrunGrup4Id, FUrunGrup4) = 0) then okey = 0;

      if (okey = 1) then
      if (AUrunKategoriAralikli = 0) then
      if (KutuKarsilastir(ABaslangicUrunKategoriId, FUrunKategoriId) = 0) then okey = 0;
    end

    FaturaGrupOkey = 1;

    if (AFaturaGrupAralikli = 0) then
    if (KutuKarsilastir(ABaslangicFaturaGrupId, FFaturaGrupId) = 0) then FaturaGrupOkey = 0;

    if (Okey = 1 and FaturaGrupOkey = 1) then
    begin
      KayitVar = 1;
      if (FFaturaTabanDovizKuru is not null) then FTabanDovizKuru = FFaturaTabanDovizKuru;
      else if (FIrsaliyeTabanDovizKuru is not null) then FTabanDovizKuru = FIrsaliyeTabanDovizKuru;
      else if (FStokTabanDovizKuru is not null) then FTabanDovizKuru = FStokTabanDovizKuru;

      if (FTabanDovizKuru is null) then FTabanDovizKuru = 1;

      FNetTutar = FNetTutar * FTabanDovizKuru;
      FBrutTutar = FBrutTutar * FTabanDovizKuru;

      if (FStokTabanDovizId <> ARaporDovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FStokTabanDovizId, :ARaporDovizId, :FTarih) into :FDovizKuru;
      else FDovizKuru = 1;

      FNetTutar = FNetTutar * FDovizKuru;
      FBrutTutar = FBrutTutar * FDovizKuru;

      FNetTutar = Yuvarla(FNetTutar, FNetFiyatYuvarlamaCarpani);
      FBrutTutar = Yuvarla(FBrutTutar, FNetFiyatYuvarlamaCarpani);

      if (FAltIskontoCarpani is Null) then FAltIskontoCarpani = 0;
      FNetTutar = FBrutTutar - FBrutTutar * FAltIskontoCarpani;
      FKdvTutari = FNetTutar * FKdvOrani /100;

      if (FTevkifat = 1) then
        FKdvTutari = FKdvTutari - FKdvTutari * FTevkifatOrani1 / FTevkifatOrani2;

      if (FHareketTipi = 1 or FHareketTipi = 2) then
      begin
        BrutTutar = BrutTutar + FBrutTutar;
        NetTutar = NetTutar + FNetTutar;
        KdvTutari  = KdvTutari + FKdvTutari;
        KdvliTutar = KdvliTutar + FNetTutar + FKdvTutari;
        Miktar = Miktar + FMiktar;
      end
      else
      begin
        BrutTutar = BrutTutar - FBrutTutar;
        NetTutar = NetTutar - FNetTutar;
        KdvTutari  = KdvTutari - FKdvTutari;
        KdvliTutar = KdvliTutar - (FNetTutar + FKdvTutari);
        Miktar = Miktar - FMiktar;
      end
    end
  end

  if (Okey = 1) then
  if (FLastFirmaId <> '' and FLastUrunId <> '' and KayitVar = 1) then suspend;
end
