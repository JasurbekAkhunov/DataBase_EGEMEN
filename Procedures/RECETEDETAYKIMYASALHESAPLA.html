CREATE PROCEDURE RECETEDETAYKIMYASALHESAPLA(
    ARECETEID VARCHAR(20),
    ARECETEASAMAID INTEGER,
    ASIRA INTEGER)
AS
declare variable FBoyaMiktari   Double Precision;
declare variable FMiktar        Double Precision;
declare variable FBanyo         Double Precision;
declare variable FReceteAsamaId integer;
declare variable FBitisSira     integer;
declare variable FSira          integer;
declare variable FUrunTipiId    integer;
declare variable FBoyaTipiId    integer;
declare variable FKimyasalTipiId    integer;
declare variable AsamaDegisti   integer;
declare variable FReceteTuruId  integer;
declare variable FReceteDetayId integer;
declare variable KayitYapildi   integer;
declare variable BoyaGrubaGoreKimyasalHesapla integer;
declare variable BoyaGrupId                  varchar(4);
declare variable FMiktarYuzde                 Double Precision;
begin
  BoyaGrupId = Null;

  Select BoyaGrubaGoreKimyasalHesapla from sistem where sistemid = 1 into :BoyaGrubaGoreKimyasalHesapla;

  FBoyaMiktari = 0;
  AsamaDegisti = 0;

  for select ReceteDetay.Sira, ReceteDetay.ReceteAsamaId, Urun.UrunTipiId, Urun.BoyaTipiId, ReceteDetay.Miktar, ReceteDetay.Banyo from ReceteDetay
    left join Urun on Urun.UrunId = ReceteDetay.UrunId
    where ReceteDetay.ReceteId = :AReceteId and ReceteDetay.Sira > :ASira order by ReceteDetay.Sira
    into :FSira, :FReceteAsamaId, :FUrunTipiId, :FBoyaTipiId, :FMiktar, :FBanyo do
  begin
    if (AsamaDegisti = 0) then
    if (FReceteAsamaId is not null and FReceteAsamaId <> AReceteAsamaId) then AsamaDegisti = 1;

    if (AsamaDegisti = 0) then
    begin
      FBitisSira = Fsira;

      if (FUrunTipiID = 0 and FBoyaTipiId = 0) then
      begin
        if (FMiktar is not null) then FBoyaMiktari = FBoyaMiktari + FMiktar;
        if (FBanyo is not null) then FBoyaMiktari = FBoyaMiktari + FBanyo;
      end
    end
  end

  if (BoyaGrubaGoreKimyasalHesapla = 1) then
  begin
    select First 1 Urun.BoyaGrupId, Sum( coalesce(ReceteDetay.Miktar,0) + coalesce(ReceteDetay.Banyo, 0) ) from ReceteDetay
      left join Urun on Urun.UrunId = ReceteDetay.UrunId
      where ReceteDetay.ReceteId = :AReceteId and ReceteDetay.Sira > :ASira and ReceteDetay.Sira <= :FBitisSira and Urun.UrunTipiId = 0 and Urun.BoyaTipiId = 0
      group by Urun.BoyaGrupId
      order by 2 desc
      into :BoyaGrupId, :FMiktarYuzde;

      if (BoyaGrupId is null) then exception KayitHatasi 'Boya Grupları girilmemiş, kimyasal hesaplanamadı..';
  end

  select Recete.ReceteTuruId from Recete where ReceteId = :AReceteId into :FReceteTuruId;

  if (FReceteTuruId is null) then exception ReceteninTuruBelliDegil;

  for select ReceteDetay.ReceteDetayId, Urun.KimyasalTipiId from ReceteDetay
    left join Urun on Urun.UrunId = ReceteDetay.UrunId
    where ReceteDetay.ReceteId = :AReceteId and ReceteDetay.Sira > :ASira and ReceteDetay.Sira <= :FBitisSira and Urun.KimyasalTipiId is not null
    Order by ReceteDetay.ReceteId, ReceteDetay.Sira
    into :FReceteDetayId, :FKimyasalTipiId do
  begin
    KayitYapildi = 0;

    if (BoyaGrubaGoreKimyasalHesapla = 1) then
    begin
      for select Miktar from ReceteTuruBoyaGrupKimyasal
        where ReceteTuruBoyaGrupKimyasal.ReceteTuruId = :FReceteTuruId and ReceteTuruBoyaGrupKimyasal.ReceteAsamaId = :AReceteAsamaId and ReceteTuruBoyaGrupKimyasal.KimyasalTipiId = :FKimyasalTipiId and
              ReceteTuruBoyaGrupKimyasal.BoyaGrupID = :BoyaGrupId and :FBoyaMiktari > ReceteTuruBoyaGrupKimyasal.BaslangicMiktar and :FBoyaMiktari <= ReceteTuruBoyaGrupKimyasal.BitisMiktar
        into :FMiktar do
      begin
        if (KayitYapildi = 0) then
        if (FMiktar is not null) then
        begin
          update ReceteDetay set Banyo = :FMiktar where ReceteDetay.ReceteDetayId = :FReceteDetayId;
          KayitYapildi = 1;
        end
        else exception KayitHatasi 'Uygun Aralık Bulunamadığı İçin Kimyasallar Hesaplanamadı..1';
      end
    end
    else
    begin
      for select First 1 ReceteTuruKimyasal.Miktar from ReceteTuruKimyasal
        where ReceteTuruKimyasal.ReceteTuruId = :FReceteTuruId and (ReceteTuruKimyasal.ReceteAsamaId = :FReceteAsamaId or ReceteTuruKimyasal.ReceteAsamaId is null) and ReceteTuruKimyasal.KimyasalTipiId = :FKimyasalTipiId and
              :FBoyaMiktari > ReceteTuruKimyasal.BaslangicMiktar and :FBoyaMiktari <= ReceteTuruKimyasal.BitisMiktar
        order by ReceteTuruKimyasal.ReceteAsamaId nulls last
        into :FMiktar do
      begin
        if (KayitYapildi = 0) then
        if (FMiktar is not null) then
        begin
          update ReceteDetay set Banyo = :FMiktar where ReceteDetay.ReceteDetayId = :FReceteDetayId;
          KayitYapildi = 1;
        end
        else exception KayitHatasi 'Uygun Aralık Bulunamadığı İçin Kimyasallar Hesaplanamadı..1';
      end
    end
  end
end
