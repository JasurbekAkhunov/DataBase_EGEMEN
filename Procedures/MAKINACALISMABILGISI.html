CREATE PROCEDURE MAKINACALISMABILGISI(
    AORMEMAKINAID MAKINAKODUSIZE,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ADOKUMAMAKINAID MAKINAKODUSIZE default Null)
RETURNS (
    ORMEMAKINADURUMUID INTEGER,
    ORMEPLANLAMAID INTEGER,
    DOKUMAPLANLAMAID INTEGER,
    TARIH TIMESTAMP,
    TUR INTEGER,
    SANAL INTEGER,
    PERSONELID INTEGER,
    ARIZA INTEGER,
    SERVISCI INTEGER,
    AYAR INTEGER,
    MEMINGERKOPTU SMALLINT,
    ALTMEMINGERKOPTU SMALLINT,
    LYCRAKOPTU SMALLINT,
    IGNEKOPTU SMALLINT,
    KAPAKACILDI SMALLINT,
    STARTSTOP SMALLINT,
    ATKIKOPTU SMALLINT,
    COZGUKOPTU SMALLINT,
    LEVENTSONU SMALLINT)
AS
declare variable MaxOrmeMakinaDurumuId integer;
/*declare variable LastTarih TimeStamp; */
declare variable FOrmeMakinaDurumuId    integer;
declare variable FOrmePlanlamaId        integer;
declare variable FDokumaPlanlamaId      integer;
declare variable FTur                   integer;
begin
  if (ABaslangicTarihi > Current_date + Current_time) then exit;
  Sanal = 1;
  Tarih = ABaslangicTarihi;
  /*LastTarih = ABaslangicTarihi;*/
  Select Max(OrmeMakinaDurumuId) from OrmeMakinaDurumu where (OrmeMakinaDurumu.OrmeMakinaId = :AOrmeMakinaId or OrmeMakinaDurumu.DokumaMakinaId = :ADokumaMakinaId)
  and OrmeMakinaDurumu.Tarih < :ABaslangicTarihi into :MaxOrmeMakinaDurumuId;

  if (MaxOrmeMakinaDurumuId is not null) then
  begin
    Select OrmeMakinaDurumuId, OrmePlanlamaId, Tur, PersonelId, ArizaVar, ServisciVar, AyarYapiliyor, MemingerKoptu, AltMemingerKoptu, LycraKoptu, IgneKoptu, KapakAcildi, StartStop,
           DokumaPlanlamaId, AtkiKoptu, CozguKoptu, LeventSonu from OrmeMakinaDurumu
     where (OrmeMakinaDurumu.OrmeMakinaId = :AOrmeMakinaId or OrmeMakinaDurumu.DokumaMakinaId = :ADokumaMakinaId) and OrmeMakinaDurumu.OrmeMakinaDurumuId = :MaxOrmeMakinaDurumuId
     into :OrmeMakinaDurumuId, :OrmePlanlamaId, :Tur, :PersonelId, :Ariza, :Servisci, :Ayar, :MemingerKoptu, :AltMemingerKoptu, :LycraKoptu, :IgneKoptu, :KapakAcildi, :StartStop,
     :DokumaPlanlamaId, :AtkiKoptu, :CozguKoptu, :LeventSonu;
  end
  else
  begin
    OrmeMakinaDurumuId = null;
    OrmePlanlamaId = null;
    DokumaPlanlamaId = Null;
    PersonelId = Null;
    Tur = 0;

  end
  suspend;

  For Select OrmeMakinaDurumuId, OrmePlanlamaId, Tarih, Tur, PersonelId, ArizaVar, ServisciVar, AyarYapiliyor, MemingerKoptu, AltMemingerKoptu, LycraKoptu, IgneKoptu, KapakAcildi, StartStop,
             DokumaPlanlamaId, AtkiKoptu, CozguKoptu, LeventSonu from OrmeMakinaDurumu
     where (OrmeMakinaDurumu.OrmeMakinaId = :AOrmeMakinaId or OrmeMakinaDurumu.DokumaMakinaId = :ADokumaMakinaId) and OrmeMakinaDurumu.Tarih >= :ABaslangicTarihi and OrmeMakinaDurumu.Tarih <= :ABitisTarihi
     order by tarih
  into :OrmeMakinaDurumuId, :OrmePlanlamaId, :Tarih, :Tur, :PersonelId, :Ariza, :Servisci, :Ayar, :MemingerKoptu, :AltMemingerKoptu, :LycraKoptu, :IgneKoptu, :KapakAcildi, :StartStop,
  :DokumaPlanlamaId, :AtkiKoptu, :CozguKoptu, :LeventSonu do
  begin
    FOrmeMakinaDurumuId = OrmeMakinaDurumuId;
    FOrmePlanlamaId = OrmePlanlamaId;
    FTur = Tur;
    Sanal = 0;

    suspend;
    /*LastTarih = Tarih; */
  end

  /*if (LastTarih < ABitisTarihi) then*/
  if (Tarih < ABitisTarihi) then
  begin
    Sanal = 1;
    Tarih = ABitisTarihi;
    OrmeMakinaDurumuId = Null;
    Tur = FTur;
    suspend;
  end
end
