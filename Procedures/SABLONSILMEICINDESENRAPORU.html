CREATE PROCEDURE SABLONSILMEICINDESENRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    ASONSIPARISTARIHI TIMESTAMP,
    ADESENGRUPID VARCHAR(4),
    ADESENID VARCHAR(20))
RETURNS (
    DESENID VARCHAR(20),
    DESENADI VARCHAR(80),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    DESENGRUPID VARCHAR(4),
    SABLONSAYISI INTEGER,
    SABLONBOYUID VARCHAR(10),
    MUSTERIDESENNO1 VARCHAR(20),
    MUSTERIDESENADI1 VARCHAR(80),
    MUSTERIDESENNO2 VARCHAR(20),
    MUSTERIDESENADI2 VARCHAR(80),
    DESENTAKIMID INTEGER,
    TAKIMPARCASIMI INTEGER,
    TAKIMPARCASIMISTR VARCHAR(20),
    ANADESEN INTEGER,
    ANADESENSTR VARCHAR(10),
    SONSIPARISTARIHI TIMESTAMP,
    SABLONCEKIMTARIHI TIMESTAMP,
    SONPARTIYIL INTEGER,
    SONPARTIID INTEGER,
    SONPARTIACILISTARIHI TIMESTAMP,
    SONPARTIIDSTR VARCHAR(30),
    DESENTASARIMNO VARCHAR(20),
    DESENSABLONENI VARCHAR(20),
    DESENGRAVURBOYU INTEGER)
AS
declare variable KS integer;
begin
  if (ADesenID is not Null) then
  begin
    ABaslangicTarihi   = Null;
    ABitisTarihi       = Null;
    AFirmaId           = Null;
    ASonSiparisTarihi  = Null;
    ADesenGrupId       = Null;
  end

  For Select Desen.DesenId, Desen.Adi, Desen.FirmaId, Firma.Adi, Desen.DesenGrupId,
             Desen.SablonSayisi, Desen.MusteriDesenNo1, Desen.MusteriDesenAdi1, Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2,
             Desen.DesenTakimId, Desen.TakimParcasimi, Desen.AnaDesen, SablonB.Olcu,
             Max(BoyaSiparis.SiparisTarihi), Max(DesenIstekleri.BitisTarihi), DesenGravurBoyu.Olcu, DesenSablonEni.Olcu, Desen.TasarimNo from Desen
    Left Join Firma on Firma.FirmaId = Desen.FirmaId
    Left Join BoyaSiparisDetay on BoyaSiparisDetay.DesenId = Desen.DesenId
    left join DesenOlcu SablonB on SablonB.DesenOlcuID = Desen.SablonBoyuDesenOlcuId
    Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId and BoyaSiparis.IslemTipiId <> 5
    Left Join DesenIstekleri on DesenIstekleri.DesenId = Desen.DesenId
    Left Join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiId and DesenIstekTipi.SablonCekimmi= 1 and DesenIstekleri.BitisTarihi is not null
    left join DesenOlcu DesenGravurBoyu on DesenGravurBoyu.DesenolcuId = Desen.GravurBoyuDesenOlcuId
    left join DesenOlcu DesenSablonEni on DesenSablonEni.DesenolcuId = Desen.SablonEniDesenOlcuId
    where ((Desen.InsertTarihi >= :ABaslangicTarihi and Desen.InsertTarihi <= :ABitisTarihi) or :ABaslangicTarihi is Null) and Desen.SablonSayisi is not null and
          (Desen.SablonCekildi = 1 and Desen.SablonSilinmis = 0) and
          (Desen.FirmaID = :AFirmaID or :AFirmaID is null) and
          (Desen.DesenGrupId = :ADesenGrupId or :ADesenGrupId is null) and
          (Desen.DesenId = :ADesenId or :ADesenId is null)
  group by Desen.DesenId, Desen.Adi, Desen.FirmaId, Firma.Adi, Desen.DesenGrupId, DesenGravurBoyu.Olcu, DesenSablonEni.Olcu, Desen.TasarimNo,
           Desen.SablonSayisi, Desen.MusteriDesenNo1, Desen.MusteriDesenAdi1, Desen.MusteriDesenNo2, Desen.MusteriDesenAdi2,
           Desen.DesenTakimId, Desen.TakimParcasimi, Desen.AnaDesen,SablonB.Olcu
  into :DesenId, :DesenAdi, :FirmaId, :FirmaAdi, :DesenGrupId,
       :SablonSayisi, :MusteriDesenNo1, :MusteriDesenAdi1, :MusteriDesenNo2, :MusteriDesenAdi2, :DesenTakimId, :TakimParcasimi, :AnaDesen, :SablonBoyuId,
       :SonSiparisTarihi, :SablonCekimTarihi, :DesenSablonEni, :DesenGravurBoyu, :DesenTasarimNo do
  begin
    if (TakimParcasimi = 1) then TakimParcasimiStr = 'Takım Parçası';
    else TakimParcasimiStr = '';

    if (AnaDesen = 1) then AnaDesenStr = 'Ana Desen';
    else AnaDesenStr = '';

    SonPartiYil = Null;
    SonPartiID  = Null;
    SonPartiAcilisTarihi = Null;
    SonPartiIDStr = Null;

    Select First 1 Parti.PartiYil,  Parti.PartiID, Parti.AcilisTarihi from DesenVaryant
    left Join Parti on Parti.DesenVaryantID = DesenVaryant.DesenVaryantID
    where DesenVaryant.DesenID = :DesenId
    Order By Parti.AcilisTarihi Desc
    into :SonPartiYil, :SonPartiID, :SonPartiAcilisTarihi;

    if (SonPartiID is not Null) then SonPartiIDStr = '(' || EgemenIntegerToStr(SonPartiYil) || ') '  || EgemenIntegerToStr(SonPartiID);

    if (ASonSiparisTarihi is null or ASonSiparisTarihi > SonSiparisTarihi) then
    begin
      Select count(*) from SablonSilmeTalimatDetay
        left join SablonSilmeTalimati on SablonSilmeTalimati.SablonSilmeTalimatiID = SablonSilmeTalimatDetay.SablonSilmeTalimatiID
        where SablonSilmeTalimatDetay.DesenId = :DesenId and SablonSilmeTalimati.BitisTarihi is null
        into :KS;

      if (KS = 0) then suspend;
    end
  end
end
