CREATE PROCEDURE PARTIASAMADEGERLERIGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIASAMALARIID INTEGER,
    PARTISINIFID INTEGER,
    DEGERTIPIID INTEGER,
    DEGERTIPIADI VARCHAR(20),
    ISTENEN VARCHAR(10),
    GERCEKLESEN VARCHAR(10),
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select PartiAsamaDeger.PartiAsamalariId, PartiAsamaDeger.PartiSinifId, PartiAsamaDeger.DegerTipiId, DegerTipi.Adi, PartiAsamaDeger.Istenen, PartiAsamaDeger.Gerceklesen,
         Sk.Adi, PartiAsamaDeger.DegisimTarihi from PartiAsamaDeger
  Left Join DegerTipi on DegerTipi.DegerTipiId = PartiAsamaDeger.DegerTipiId
  Left Join Kullanici SK on SK.KullaniciId = PartiAsamaDeger.KullaniciId
  where PartiAsamaDeger.PartiKayitId = :APartiKayitId
  into :PartiAsamalariId, :PartiSinifId, :DegerTipiId, :DegerTipiAdi, :Istenen, :Gerceklesen, :KullaniciAdi, :DegisimTarihi do suspend;
end
