CREATE PROCEDURE DEPARTMANDANISTASYONASURUKLE(
    ADESENISTASYONID INTEGER,
    ADESENISTASYONPLANLAMAID INTEGER,
    ADESENPLANLAMAID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    DESENISTASYONPLANLAMAID INTEGER)
AS
declare variable FDesenIstasyonPlanlamaId integer;
declare variable FSepetNo               integer;
declare variable ImalatBaslamis         integer;
declare variable DesenIstekleriId       integer;
declare variable XDesenIstasyonId       integer;
begin
  if (ADesenPlanlamaId is null and ADesenIstasyonId is null) then exception UretimPlanlamaParemetreHatasi;

  Select DesenIstasyonId from DesenPlanlama where DesenPlanlama.DesenPlanlamaID = :ADesenPlanlamaId into :XDesenIstasyonId;

  if (XDesenIstasyonId is not null) then exception KayitHatasi 'Daha önce İstasyon''a Planlanmış..';

  if (ADesenIstasyonPlanlamaId is null) then FSepetNo = -1;
  else Select SepetNo from DesenIstasyonPlanlama where DesenIstasyonPlanlama.DesenIstasyonId = :ADesenIstasyonId and DesenIstasyonPlanlama.DesenIstasyonPlanlamaId = :ADesenIstasyonPlanlamaId into :FSepetNo;

  if (FSepetNo is null) then exception MakinaSepetBulunamadi;

  Select Count(*) from DesenPlanlama
  left join DesenIstekleri on DesenIstekleri.DesenIstekleriId = DesenPlanlama.DesenIstekleriId
  where DesenPlanlama.DesenPlanlamaID = :ADesenPlanlamaId and DesenIstekleri.BaslamaTarihi is not null into :ImalatBaslamis;

  if (ImalatBaslamis > 0) then exception IstekYapilmayaBaslamisTasinamaz;

  FDesenIstasyonPlanlamaId = gen_id(DesenIstasyonPlanlama_Id_Gen, 1);

  insert into DesenIstasyonPlanlama(DesenIstasyonPlanlamaId, DesenPlanlamaId, SepetNo, DesenIstasyonId, PlanlayanKullaniciId, KullaniciId)
                      values(:FDesenIstasyonPlanlamaId, :ADesenPlanlamaId, :FSepetNo, :ADesenIstasyonId, :AKullaniciId, :AKullaniciId);

  DesenIstasyonPlanlamaId = FDesenIstasyonPlanlamaId;
  suspend;
end
