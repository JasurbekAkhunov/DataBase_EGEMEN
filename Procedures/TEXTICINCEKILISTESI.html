CREATE PROCEDURE TEXTICINCEKILISTESI(
    ABOYAIRSALIYEID INTEGER)
RETURNS (
    SIRA INTEGER,
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYENO IRSALIYENOSIZE,
    HAMID INTEGER,
    MUSTERIRENKKODU VARCHAR(30),
    EN VARCHAR(10),
    HAMADI VARCHAR(255),
    KALITENO INTEGER,
    PSKALITEKONTROLID INTEGER,
    NETKILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    IPLIKPAKETLEMEID INTEGER,
    BARKODID VARCHAR(20),
    PARTIKAYITID INTEGER,
    PARTISINIFID INTEGER,
    PARTISTOKTARTIID INTEGER,
    DARA NUMERIC(12,3),
    BRUTKILO NUMERIC(12,3))
AS
declare Variable FBoyaIrsaliyeDetayId integer;
declare Variable SonBoyaIrsaliyeDetayId integer;
begin
  Sira = 0;
  IrsaliyeTarihi = Null;
  IrsaliyeNo = Null;
  HamId = Null;
  MusteriRenkKodu = Null;
  En = null;
  HamAdi = Null;
  KaliteNo = Null;
  PsKaliteKontrolId = Null;
  BrutKilo = Null;
  Dara = Null;
  NetKilo = Null;
  Metre = Null;
  PartiKayitId = Null;
  PartiSinifId = Null;
  PartiStokTartiId = Null;
  FBoyaIrsaliyeDetayId = Null;
  SonBoyaIrsaliyeDetayId = Null;

  For select PsKaliteKontrol.PsKaliteKontrolId, PsKaliteKontrol.En, PartiStokTarti.BoyaIrsaliyeDetayId,
             PsKaliteKontrol.Kilo, PsKaliteKontrol.Dara, PsKaliteKontrol.Metre,
             BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliyeDetay.HamId, Ham.Adi,
             PartiStokTarti.KaliteNo, PsKaliteKontrol.PartiSinifId, PartiStokTarti.PartiStokTartiId,
             PsKaliteKontrol.PartiKayitId, BoyaSiparisDetay.MusteriRenkKoduveAdi
             from PartiStokTarti
    left join PsKaliteKontrol on PsKaliteKontrol.PartiStokTartiId = PartiStokTarti.PartiStokTartiId
    left join PartiStok on PartiStok.PartiSinifId = PartiStokTarti.PartiSinifId
    left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
    left join BoyaIrsaliye on BoyaIrsaliye.BoyaIrsaliyeId = PartiStokTarti.BoyaIrsaliyeId
    left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.BoyaIrsaliyeDetayId = PartiStokTarti.BoyaIrsaliyeDetayId
    left join Ham on Ham.HamId = BoyaIrsaliyeDetay.HamId
  where PartiStokTarti.BoyaIrsaliyeId = :ABoyaIrsaliyeId and PsKaliteKontrol.PsKaliteKontrolId is not null
  order by PartiStokTarti.BoyaIrsaliyeDetayId
  into :PsKaliteKontrolId, :En, :FBoyaIrsaliyeDetayId,
       :BrutKilo, :Dara, :Metre,
       :IrsaliyeNo, :IrsaliyeTarihi, :HamId, :HamAdi,
       :KaliteNo, :PartiSinifId, :PartiStokTartiId,
       :PartiKayitId, :MusteriRenkKodu do
  begin
    NetKilo = BrutKilo - Dara;
    if (FBoyaIrsaliyeDetayId <> SonBoyaIrsaliyeDetayId) then Sira = 1;
    else Sira = Sira + 1;
    SonBoyaIrsaliyeDetayId = FBoyaIrsaliyeDetayId;
    suspend;
  end
end
