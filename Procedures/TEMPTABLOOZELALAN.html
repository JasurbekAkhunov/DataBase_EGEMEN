CREATE PROCEDURE TEMPTABLOOZELALAN(
    AOZELALANKAYNAGIID INTEGER NOT NULL,
    ATABLOKAYITIDSTR ADI50 NOT NULL)
RETURNS (
    RKOLONID INTEGER,
    RKOLON ADI100,
    RTABLOKAYITIDSTR ADI50,
    ROZELALANID INTEGER,
    RDEGER ADI255,
    ROZELALANKOLONTIPIID INTEGER)
AS
Declare Variable Sayi Integer;
declare variable oaktListe Integer;
Begin
  Select oaktListe From OzelAlanKolonTipiGetir Into :oaktListe;

  For Select OzelAlanTipi.OzelAlanTipiId as KolonId, OzelAlanTipi.Adi as Kolon, :ATabloKayitIdStr as TabloKayitIdStr,
  OzelAlan.OzelAlanId, OzelAlan.Deger, OzelAlanTipi.OzelAlanKolonTipiId From OzelAlanKaynagi
  Join OzelAlanTipi on OzelAlanKaynagi.OzelAlanKaynagiId = OzelAlanTipi.OzelAlanKaynagiId And OzelAlanTipi.Kapali = 0
  Left Join OzelAlan On OzelAlan.OzelAlanTipiId = OzelAlanTipi.OzelAlanTipiId And OzelAlan.TabloKayitIdStr = :ATabloKayitIdStr
  Where OzelAlanKaynagi.OzelAlanKaynagiId = :AOzelAlanKaynagiId
  Into :RKolonId, :RKolon, :RTabloKayitIdStr, :ROzelalanId, :RDeger, :ROzelAlanKolonTipiId Do
  Begin
    If(oaktListe = ROzelAlanKolonTipiId) then
    Begin
        if (EgemenStrToIntOlurMu(RDeger) = 1) then
            Sayi = CAST(RDeger As Integer);
        else
            Sayi = -1;
    End

    If(oaktListe = ROzelAlanKolonTipiId) then
        RDeger = CAST(Sayi as Adi255);

    Suspend;
  End
End
