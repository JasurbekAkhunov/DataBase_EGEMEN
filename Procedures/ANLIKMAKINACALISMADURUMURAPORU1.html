CREATE PROCEDURE ANLIKMAKINACALISMADURUMURAPORU1(
    AUNITEID INTEGER,
    AMAKINAGRUBU VARCHAR(4),
    AMAKINAID VARCHAR(10))
RETURNS (
    MAKINAID VARCHAR(20),
    MAKINAADI VARCHAR(40),
    MAKINAGRUBU VARCHAR(4),
    TARIH TIMESTAMP,
    CALISIYOR SMALLINT,
    ARIZA SMALLINT,
    TEMIZLIK SMALLINT,
    ARAVERMETARIHI TIMESTAMP,
    ARAVERMESEBEBIID INTEGER,
    ARAVERMESEBEBIADI VARCHAR(50),
    IMALATAARAVERENPERSONELID INTEGER,
    IMALATAARAVERENPERSONELADI VARCHAR(40),
    PARCABEKLIYOR INTEGER,
    PARCABEKLIYORSTR VARCHAR(4),
    ARIZABASLAMATARIHI TIMESTAMP,
    ARIZAIHBARPERSONELID INTEGER,
    ARIZAIHBARPERSONELADI VARCHAR(40),
    ARIZATIPIID INTEGER,
    ARIZAIHBARID INTEGER,
    ARIZATIPIADI VARCHAR(50),
    MAKINADURUMUID INTEGER,
    IMALATAARAVERILDI INTEGER)
AS
declare variable ImalataAraVermeId integer;
begin
  For Select Makina.MakinaId, Makina.Adi, Makina.MakinaGrubu, Makina.MakinaDurumuId, MakinaDurumu.Tarih, MakinaDurumu.Calisiyor, MakinaDurumu.Ariza,
             MakinaDurumu.Temizlik, MakinaDurumu.ImalataAraVerildi from Makina
    left join MakinaDurumu on MakinaDurumu.MakinaDurumuID = Makina.MakinaDurumuID
    where Makina.OtomasyonaDahil = 1 and Makina.AlarmIkaz = 0 and
         (Makina.MakinaId = :AMakinaId or :AMakinaId is null) and
         (Makina.uniteId = :AUniteId or :AUniteId is null) and
         (Makina.MakinaGrubu = :AMakinaGrubu or :AMakinaGrubu is null)
    into :MakinaId, :MakinaAdi, :MakinaGrubu, :MakinaDurumuId, :Tarih, :Calisiyor, :Ariza, :Temizlik, :ImalataAraVerildi do
  begin
    AraVermeTarihi        = null;
    AraVermeSebebiId      = Null;
    AraVermeSebebiAdi     = NUll;
    ImalataAraVerenPersonelId = NUll;
    ImalataAraVerenPersonelAdi = NUll;
    ParcaBekliyor         = 0;
    ArizaBaslamaTarihi    = Null;
    ArizaIhbarPersonelId  = Null;
    ArizaIhbarPersonelAdi = null;
    ArizaIhbarId          = Null;
    ArizaTipiID = Null;
    ArizaTipiAdi = Null;

    if (MakinaDurumuId is not null) then
    begin
      if (Ariza = 1 or Ariza = 3) then Ariza = 1;
      else Ariza = 0;

      if (ImalataAraVerildi = 1 or ImalataAraVerildi = 3) then ImalataAraVerildi = 1;
      else ImalataAraVerildi = 0;

      ParcaBekliyor = 0;
      ParcaBekliyorStr = '';

      if (Ariza = 1) then
      begin
        Select Max(ArizaIhbarId) from ArizaIhbar where ArizaIhbar.MakinaId = :MakinaId and ArizaIhbar.BittiMakinaDurumuId is null into :ArizaIhbarId;

        Select MakinaDurumu.Tarih, ArizaIhbar.BaslatanPersonelId, Personel.Adi, ArizaIhbar.ParcaBekliyor, ArizaIhbar.ArizaTipiId, ArizaTipi.Adi from ArizaIhbar
          left join Personel on Personel.PersonelId = ArizaIhbar.BaslatanPersonelId
          left join ArizaTipi on ArizaTipi.ArizaTipiId = ArizaIhbar.ArizaTipiId
          left join MakinaDurumu on MakinaDurumu.MakinaDurumuId = ArizaIhbar.BasladiMakinaDurumuId
          where ArizaIhbar.ArizaIhbarId = :ArizaIhbarId
          into :ArizaBaslamaTarihi, :ArizaIhbarPersonelId, :ArizaIhbarPersonelAdi, :ParcaBekliyor, :ArizaTipiId, :ArizaTipiAdi;

        if (ParcaBekliyor is null) then ParcaBekliyor = 0;
        if (ParcaBekliyor = 1) then ParcaBekliyorStr = 'Evet';
      end

      if (ImalataAraVerildi = 1) then
      begin
        Select Max(ImalataAraVermeId) from ImalataAraVerme
          where ImalataAraVerme.MakinaId = :MakinaId and ImalataAraVerme.BittiMakinaDurumuId is null into :ImalataAraVermeId;

        Select ImalataAraVerme.AraVermeSebebiId, AraVermeSebebi.Adi, MakinaDurumu.Tarih, ImalataAraVerme.AraVerdirenPersonelId,
               Personel.Adi  from ImalataAraVerme
          left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
          left join Personel on Personel.PersonelId = ImalataAraVerme.AraVerdirenPersonelId
          left join MakinaDurumu on MakinaDurumu.MakinaDurumuId = ImalataAraVerme.BasladiMakinaDurumuId
          where ImalataAraVerme.ImalataAraVermeId = :ImalataAraVermeId
          into :AraVermeSebebiId, :AraVermeSebebiAdi, :AraVermeTarihi, :ImalataAraVerenPersonelId, :ImalataAraVerenPersonelAdi;
      end

    end
    suspend;
  end
end
