    ASTOKVAR EGEBOOLEAN)

    STOKVAR EGEBOOLEAN)
AS
begin
  For Select Lot from OrmeRaf
    where OrmeRaf.HamId = :AHamId and IntKarsilastir(OrmeRaf.RenkId, :ARenkId) = 1 and OrmeRaf.Lot is not null and OrmeRaf.Lot <> ''

    suspend;
  end
end
