CREATE PROCEDURE PARTIICKONTROLRAPORUDETAYI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AASAMAID INTEGER,
    AHAMID INTEGER,
    AFIRMAID VARCHAR(20),
    AKALINLIKID INTEGER,
    AIPLIKCINSIID INTEGER,
    AOZELLIKID INTEGER,
    AKUMASCINSIID INTEGER,
    ALYCRATIPIID INTEGER,
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4))
RETURNS (
    PARTIASAMALARIID INTEGER,
    PARTISINIFID INTEGER,
    ICKONTROLTIPIID INTEGER,
    ICKONTROLTIPIADI VARCHAR(30),
    ONERILENDEGER VARCHAR(15),
    ISTENEN VARCHAR(50),
    SONUC VARCHAR(15),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP)
AS
begin
  For Select PartiIcKontrolRaporu.PartiAsamalariId, PartiIcKontrolRaporu.PartiSinifId, PartiAsamaIcKontrol.IcKontrolTipiId,
             IcKontrolTipi.Adi, IcKontrolTipi.OnerilenDeger, PartiAsamaIcKontrol.Istenen, PartiAsamaIcKontrol.Sonuc,
             PartiAsamaIcKontrol.PersonelId, Personel.Adi, PartiAsamaIcKontrol.InsertTarihi from PartiIcKontrolRaporu(:ABaslangicTarihi, :ABitisTarihi, :AAsamaId, :AHamId, :AFirmaId,
             :AKalinlikId, :AIplikCinsiId, :AOzellikId, :AKumasCinsiId, :ALycraTipiId, :ABaslangicGrup1Id, :ABitisGrup1Id, :ABaslangicGrup2Id,
             :ABitisGrup2Id, :ABaslangicGrup3Id, :ABitisGrup3Id, :ABaslangicGrup4Id, :ABitisGrup4Id, :AHamGrup1Aralikli, :AHamGrup2Aralikli,
             :AHamGrup3Aralikli, :AHamGrup4Aralikli, :ASiparisGrupId, :ASiparisGrup2Id)
    left join PartiAsamaIcKontrol on PartiAsamaIcKontrol.PartiSinifId = PartiIcKontrolRaporu.PartiSinifId and PartiAsamaIcKontrol.PartiAsamalariId = PartiIcKontrolRaporu.PartiAsamalariId
    Left Join IcKontrolTipi on IcKontrolTipi.IcKontrolTipiId = PartiAsamaIcKontrol.IcKontrolTipiId
    Left Join Personel on Personel.PersonelId = PartiAsamaIcKontrol.PersonelId
    into :PartiAsamalariId, :PartiSinifId, :IcKontrolTipiId, :IcKontrolTipiAdi, :OnerilenDeger, :Istenen, :Sonuc,
         :PersonelId, :PersonelAdi, :InsertTarihi do suspend;
end
