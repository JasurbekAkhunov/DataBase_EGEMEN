CREATE PROCEDURE ORMEANLIKIPLIKSTOKDURUMUCIKIS
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    MIKTAR NUMERIC(12,3),
    DEPOID VARCHAR(10),
    TARIH TIMESTAMP)
AS
declare variable hstIplik integer;
declare variable shtAlis integer;
begin
  Select hstIplik from HamTipiGetir into :hstIplik;

  For Select first 10 StokHareket.HamId, Ham.Adi, Ham.Adi2, Ham.Adi3, Ham.Adi4, StokHareket.Miktar, StokHareket.DepoId, StokHareket.Tarih from StokHareket
     left join Ham on Ham.HamID = StokHareket.HamID
     where StokHareket.GirisCikis = -1 and Ham.HamTipiID = :hstIplik
     order by StokHareket.tarih Desc
  into :HamId, :HamAdi, :HamAdi2, :HamAdi3, :HamAdi4, :Miktar, :DepoId, :Tarih do
  begin
    suspend;
  end
end
