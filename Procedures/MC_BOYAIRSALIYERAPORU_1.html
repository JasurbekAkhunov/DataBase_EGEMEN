CREATE PROCEDURE MC_BOYAIRSALIYERAPORU_1(
    AFIRMAID VARCHAR(20),
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASINIF VARCHAR(1),
    ASATISTIPI INTEGER,
    ARAPORTIPI INTEGER,
    AISLEMTIPI INTEGER,
    ARAPORTARIHTIPI INTEGER,
    AKULLANICIID INTEGER,
    ASIPARISGRUPID VARCHAR(4))
RETURNS (
    BOYAIRSALIYEID INTEGER,
    IRSALIYETARIHI TIMESTAMP,
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    SEVKTARIHI TIMESTAMP,
    IRSALIYEGRUPID VARCHAR(4),
    ARACID VARCHAR(11))
AS
begin
  for select BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeSeriNo,
             BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.IrsaliyeGrupId, BoyaIrsaliye.AracId from MC_BoyaIrsaliyeRaporu_2(:AFirmaId, :ABaslangicTarihi, :ABitisTarihi, :ASinif, :ASatisTipi, :ARaporTipi, :AIslemTipi, :ARaporTarihTipi, :AKullaniciId, :ASiparisGrupId)

    left join BoyaIrsaliye On BoyaIrsaliye.BoyaIrsaliyeId = MC_BoyaIrsaliyeRaporu_2.BoyaIrsaliyeId
    group By BoyaIrsaliye.BoyaIrsaliyeId, BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.IrsaliyeSeriNo,
             BoyaIrsaliye.IrsaliyeNo, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.IrsaliyeGrupId, BoyaIrsaliye.AracId
    order by BoyaIrsaliye.IrsaliyeTarihi, BoyaIrsaliye.SevkTarihi, BoyaIrsaliye.IrsaliyeNo
    into :BoyaIrsaliyeId, :IrsaliyeTarihi, :IrsaliyeSeriNo, :IrsaliyeNo, :SevkTarihi, :IrsaliyeGrupId,
         :AracId do suspend;
end
