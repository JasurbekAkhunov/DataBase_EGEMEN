CREATE PROCEDURE SIPARISICINIPLIKTALEPOLUSTUR(
    ABOYASIPARISKAYITID INTEGER,
    AORMESIPARISDETAYID INTEGER,
    AKULLANICIID INTEGER)
RETURNS (
    IPLIKTALEPYIL INTEGER,
    IPLIKTALEPID INTEGER)
AS
declare variable IplikTalepKayitID integer;
declare variable OrmeSiparisTipiId integer;
declare variable astBoyaliIplik         integer;
declare variable KS integer;
Declare variable IplikTalepYeriId integer;
declare variable OnaysizOrmeSipPlanlanmasin integer;
begin
  Select OnaysizOrmeSipPlanlanmasin from Sistem where sistemId = 1 into :OnaysizOrmeSipPlanlanmasin;

  Select astBoyaliIplik from OrmeSiparisTipiGetir into :astBoyaliIplik;

  if (OnaysizOrmeSipPlanlanmasin = 1) then
  begin
    Select OrmeSiparisDetay.OnayDurumu from OrmeSiparisDetay where OrmeSiparisDetay.OrmeSiparisDetayID = :AOrmeSiparisDetayID into :KS;
    if (KS <> 1) then exception SiparisOnaylanmamisPlanlanamaz;
  end

  Select Max(IplikTalepKayitID) from IplikTalepDetay where IplikTalepDetay.OrmeSiparisDetayID = :AOrmeSiparisDetayID into :IplikTalepKayitID;

  Select IplikTalepYeriId from Kullanici where Kullanici.KullaniciID = :AKullaniciID into :IplikTalepYeriId;

  if (IplikTalepKayitID is null) then
  begin
    Select OrmeSiparisTipiID from BoyaSiparis where BoyaSiparis.BoyaSiparisKayitID =:ABoyaSiparisKayitID into :OrmeSiparisTipiID;

    if (OrmeSiparisTipiID = astBoyaliIplik) then
    begin
      Select Count(*) from BoyaSiparisDetay where BoyaSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId into :KS;
      if (KS = 0) then exit;
    end

    IplikTalepYil = f_Year(Current_date);

    IplikTalepId = Null;

    Select Id from IplikTalepNumaratorAl(:IplikTalepYil) into :IplikTalepId;

    insert into IplikTalep(IplikTalepYil, IplikTalepId, TalepTarihi, TalepYeriId, InsertKullaniciId, KullaniciId)
                   values(:IplikTalepYil, :IplikTalepId, Current_date, :IplikTalepYeriId, :AKullaniciId, :AKullaniciId);

    Select IplikTalepKayitId from IplikTalep where IplikTalep.IplikTalepYil = :IplikTalepYil and IplikTalep.IplikTalepID = :IplikTalepID into :IplikTalepKayitId;

    if (OrmeSiparisTipiID <> astBoyaliIplik) then
    begin
      insert into IplikTalepDetay(IplikTalepkayitId, BoyaSiparisKayitId, OrmeSiparisDetayId, HamId, MelanjKodu, FlamKodu, TeslimTarihi, Miktar, InsertKullaniciId, KullaniciId)
                    Select :IplikTalepkayitId, :ABoyaSiparisKayitId, :AOrmeSiparisDetayId, OrmeSiparisDetay.HamId, OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.FlamKodu, OrmeSiparisDetay.TeslimTarihi, OrmeSiparisDetay.HamMiktar, :AKullaniciId, :AKullaniciId from OrmeSiparisDetay
                    where OrmeSiparisDetay.OrmeSiparisDetayID = :AOrmeSiparisDetayID;
    end
    else
    begin
      insert into IplikTalepDetay(IplikTalepkayitId, BoyaSiparisKayitId, OrmeSiparisDetayId, BoyaSiparisDetayId, HamId, MelanjKodu, FlamKodu, TeslimTarihi, Miktar, InsertKullaniciId, KullaniciId)
             Select :IplikTalepkayitId, :ABoyaSiparisKayitId, BoyaSiparisDetay.OrmeSiparisDetayId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisDetay.HamId, OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.FlamKodu, OrmeSiparisDetay.TeslimTarihi, BoyaSiparisDetay.SiparisKilo, :AKullaniciId, :AKullaniciId from BoyaSiparisDetay
             left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayID = BoyaSiparisDetay.OrmeSiparisDetayId
             where BoyaSiparisDetay.OrmeSiparisDetayID = :AOrmeSiparisDetayID;
    end
  end
  else
  begin
    Select IplikTalepYil, IplikTalepID from IplikTalep where IplikTalep.IplikTalepKayitID = :IplikTalepKayitID into :IplikTalepYil, :IplikTalepID;
  end

  suspend;
end
