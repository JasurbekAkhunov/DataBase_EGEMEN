CREATE PROCEDURE BOYASIPARISDETAYSARJLIGETIR(
    ABOYASIPARISKAYITID INTEGER)
RETURNS (
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISSARJID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    SARJNO INTEGER,
    KAZANKAPASITE INTEGER,
    BOBINSAYISI INTEGER,
    GRAMAJ NUMERIC(12,3),
    SARJKILO NUMERIC(12,3),
    BRUTSARJKILO NUMERIC(12,3),
    PARTISIACILANKILO NUMERIC(12,3),
    REZERVMIKTAR NUMERIC(12,3),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    PROSESID INTEGER,
    PROSESADI VARCHAR(255),
    LOT VARCHAR(30),
    MUSTERIRENKKODUVEADI VARCHAR(30),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    RENKDERINLIGIID INTEGER,
    RENKDERINLIGIIDSTR VARCHAR(20),
    MUSTERISIPARISNO VARCHAR(50),
    LABRECETEID INTEGER,
    LABRECETEKODU VARCHAR(20),
    LABRECETEADI VARCHAR(40),
    LABRECETEATISID INTEGER,
    LABRECETEATISNO VARCHAR(20),
    ONAYDURUMU INTEGER,
    ONAYDURUMUSTR VARCHAR(5),
    ONAYTARIHI TIMESTAMP,
    ONAYLAYANKULLANICIID INTEGER,
    ONAYLAYANKULLANICIADI VARCHAR(40),
    PAKETLEMESEKLIID INTEGER,
    TERMINTARIHI TIMESTAMP,
    REVIZETERMINTARIHI TIMESTAMP,
    TERMINNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    PLANLAMANOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
declare variable LastBoyaSiparisDetayId integer;
begin
  SarjNo = 0;
  LastBoyaSiparisDetayId = -1;
  for select BoyaSiparisDetay.BoyaSiparisKayitId, BoyaSiparisDetay.BoyaSiparisDetayId, BoyaSiparisSarj.BoyaSiparisSarjId, BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.LabReceteId,
             LabRecete.LabReceteKodu, LabRecete.Adi, BoyaSiparisDetay.LabReceteAtisId, LabReceteAtis.AtisNoStr, BoyaSiparisDetay.ProsesId, Proses.Adi,
             BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi,
             BoyaSiparisDetay.PaketlemeSekliId, BoyaSiparisDetay.HamId, Ham.Adi,
             BoyaSiparisDetay.TerminTarihi, BoyaSiparisDetay.RevizeTerminTarihi, BoyaSiparisDetay.TerminNotu, BoyaSiparisDetay.Notu, BoyaSiparisDetay.PlanlamaNotu,
             BoyaSiparisDetay.RenkDerinligiId, RenkDerinligi.Adi, BoyaSiparisDetay.Lot, BoyaSiparisDetay.MusteriRenkKoduveAdi, BoyaSiparisDetay.MusteriKaliteNo,
             BoyaSiparisDetay.MusteriKaliteAdi, BoyaSiparisDetay.OnayDurumu, BoyaSiparisDetay.OnayTarihi, BoyaSiparisDetay.OnaylayanKullaniciId, Ok.Adi,
             BoyaSiparisSarj.KazanKapasite, BoyaSiparisSarj.BobinSayisi, BoyaSiparisSarj.Gramaj, BoyaSiparisSarj.SarjKilo, BoyaSiparisSarj.BrutSarjKilo from BoyaSiparisDetay
  left join Ham on Ham.HamId = BoyaSiparisDetay.HamId
  left join Proses on Proses.ProsesId = BoyaSiparisDetay.ProsesId
  left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
  left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
  left join LabReceteAtis on LabReceteAtis.LabReceteAtisId = BoyaSiparisDetay.LabReceteAtisId
  left join RenkDerinligi on RenkDerinligi.RenkDerinligiId = BoyaSiparisDetay.RenkDerinligiId
  Left Join Kullanici OK on OK.KullaniciId = BoyaSiparisDetay.OnaylayanKullaniciId
  left join BoyaSiparisSarj on BoyaSiparisSarj.BoyaSiparisDetayId = BoyaSiparisDetay.BoyaSiparisDetayId
  where BoyaSiparisDetay.BoyaSiparisKayitId = :ABoyaSiparisKayitId and BoyaSiparisDetay.Boyanacak = 1 and BoyaSiparisSarj.BoyaSiparisSarjId is not null
  order by BoyaSiparisSarj.BoyaSiparisDetayId, BoyaSiparisSarj.BoyaSiparisSarjId
  into :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisSarjId, :MusteriSiparisNo, :LabReceteId, :LabReceteKodu,
       :LabReceteAdi, :LabReceteAtisId, :LabReceteAtisNo,
       :ProsesId, :ProsesAdi, :SiparisProsesId, :SiparisProsesAdi, :PaketlemeSekliId,
       :HamId, :HamAdi, :TerminTarihi, :RevizeTerminTarihi, :TerminNotu, :Notu, :PlanlamaNotu,
       :RenkDerinligiId, :RenkDerinligiIdStr, :Lot, :MusteriRenkKoduveAdi, :MusteriKaliteNo, :MusteriKaliteAdi,
       :OnayDurumu, :OnayTarihi, :OnaylayanKullaniciId, :OnaylayanKullaniciAdi,
       :KazanKapasite, :BobinSayisi, :Gramaj, :SarjKilo, :BrutSarjKilo do
  begin
    if (LastBoyaSiparisDetayId <> BoyaSiparisDetayId) then
    begin
      LastBoyaSiparisDetayId = BoyaSiparisDetayId;
      SarjNo = 0;
    end
    SarjNo = SarjNo + 1;

    if (OnayDurumu = 1) then OnayDurumuStr = 'Kabul';
    else if (OnayDurumu = 2) then OnayDurumuStr = 'Ret';
    else OnayDurumuStr = '';

    Select Sum(PartiStok.Kilo) from PartiStok where PartiStok.BoyaSiparisSarjId = :BoyaSiparisSarjId into :PartisiAcilanKilo;

    RezervMiktar = 0;
    Select Sum(Miktar - KullanilanMiktar) from BoyaSiparisRezerv where BoyaSiparisRezerv.BoyaSiparisSarjId = :BoyaSiparisSarjId into :RezervMiktar;

    suspend;
  end
end
