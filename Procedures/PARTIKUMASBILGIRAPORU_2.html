CREATE PROCEDURE PARTIKUMASBILGIRAPORU_2(
    ARAPORTIPI INTEGER,
    AASAMAID INTEGER,
    ABASLANGICINSERTTARIHI TIMESTAMP,
    ABITISINSERTTARIHI TIMESTAMP,
    ABASLANGICPARTIYIL INTEGER,
    ABASLANGICPARTIID INTEGER,
    ABITISPARTIYIL INTEGER,
    ABITISPARTIID INTEGER,
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    HAMSTOKMASTERID INTEGER,
    YIKAMAONCESIEN DOUBLE PRECISION,
    YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    YIKAMASONRASIEN DOUBLE PRECISION,
    YIKAMASONRASIGRAMAJ DOUBLE PRECISION,
    ENDENCEKMEZLIK DOUBLE PRECISION,
    BOYDANCEKMEZLIK DOUBLE PRECISION,
    HAMDONME DOUBLE PRECISION,
    YIKAMAONCESIRAPORBOYU ADI100,
    YIKAMASONRASIRAPORBOYU ADI100,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    ORMECIADI ADI30,
    ORMECIMAKINANO VARCHAR(10),
    PUS INTEGER,
    FEINE INTEGER,
    LYCRAURETICIADI ADI20,
    LOT VARCHAR(150),
    ENDENCEKMEZLIKSTR VARCHAR(30),
    BOYDANCEKMEZLIKSTR VARCHAR(30))
AS
begin
  if (ARaporTipi = 0) then
  begin
    for

    select PartiStokDetay.PartiSinifId, HamStokMaster.HamStokMasterID, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj,
           HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
           HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, Ormeci.Adi OrmeciAdi, HamStokMaster.OrmeciMakinaNo,
           HamStokMaster.Pus, HamStokMaster.Feine, LycraUretici.Adi LycraUreticiAdi, HamStokMaster.Lot,
          cast (case when (HamStokMaster.EndenCekmezlik is null) then ''
                 else format_float('0.##', HamStokMaster.EndenCekmezlik)
                 end as varchar (30)) EndenCekmezlikStr,
          cast (case when (HamStokMaster.BoydanCekmezlik is null) then ''
                 else format_float('0.##', HamStokMaster.BoydanCekmezlik)
                 end as varchar (30)) BoydanCekmezlikStr  from Parti
      left join PartiKumasKBilgi on  PartiKumasKBilgi.PartiKayitId = Parti.PartiKayitId
      left join Asama on Asama.AsamaId = PartiKumasKBilgi.AsamaId
      left join PartiStokDetay on PartiStokDetay.partikayitid = Parti.partikayitid
      left join RafDetay  on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterID
      left join Ormeci on Ormeci.OrmeciID = HamStokIrsaliye.OrmeciID
      left join LycraUretici on LycraUretici.LycraUreticiId = HamStokMaster.LycraUreticiId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where  PartiKumasKBilgi.PartiKumasKBilgiID is not null and
           (Parti.PartiYil > :ABaslangicPartiYil or (Parti.PartiYil = :ABaslangicPartiYil and Parti.PartiId >= :ABaslangicPartiId)) and
           (Parti.PartiYil < :ABitisPartiYil or (Parti.PartiYil = :ABitisPartiYil  and Parti.PartiId <= :ABitisPartiId)) and
           (Asama.AsamaId = :AAsamaId or :AAsamaId is null) and
           (:ABaslangicBoyaSiparisYil = 0 or
            ((BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
            (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId))))
    group by PartiStokDetay.PartiSinifId, HamStokMaster.HamStokMasterID, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj,
             HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, Ormeci.Adi, HamStokMaster.OrmeciMakinaNo,
             HamStokMaster.Pus, HamStokMaster.Feine, LycraUretici.Adi, HamStokMaster.Lot

     into :PartiSinifId,:HamStokMasterID, :YikamaOncesiEn, :YikamaOncesiGramaj, :YikamaSonrasiEn, :YikamaSonrasiGramaj, :EndenCekmezlik,
          :BoydanCekmezlik, :HamDonme, :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu, :IrsaliyeNo, :IrsaliyeTarihi, :OrmeciAdi, :OrmeciMakinaNo,
          :Pus, :Feine, :LycraUreticiAdi, :Lot, :EndenCekmezlikStr, :BoydanCekmezlikStr

    do begin
       suspend;
    end
    end
  else
   begin
     for

    select PartiStokDetay.PartiSinifId, HamStokMaster.HamStokMasterID, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj,
           HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
           HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, Ormeci.Adi OrmeciAdi, HamStokMaster.OrmeciMakinaNo,
           HamStokMaster.Pus, HamStokMaster.Feine, LycraUretici.Adi LycraUreticiAdi, HamStokMaster.Lot,
          cast (case when (HamStokMaster.EndenCekmezlik is null) then ''
                 else format_float('0.##', HamStokMaster.EndenCekmezlik)
                 end as varchar (30)) EndenCekmezlikStr,
          cast (case when (HamStokMaster.BoydanCekmezlik is null) then ''
                 else format_float('0.##', HamStokMaster.BoydanCekmezlik)
                 end as varchar (30)) BoydanCekmezlikStr  from Parti
      left join PartiKumasKBilgi on PartiKumasKBilgi.PartiKayitId = Parti.PartiKayitId
      left join Asama on Asama.AsamaId = PartiKumasKBilgi.AsamaId
      left join PartiStokDetay on PartiStokDetay.partikayitid = PARTI.partikayitid
      left join RafDetay  on RafDetay.RAFDETAYID = PartiStokDetay.RAFDETAYID
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = RafDetay.HamStokIrsaliyeId
      left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
      left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterID
      left join Ormeci on Ormeci.OrmeciID = HamStokIrsaliye.OrmeciID
      left join LycraUretici on LycraUretici.LycraUreticiId = HamStokMaster.LycraUreticiId
      left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = Parti.BoyaSiparisKayitId
    where  PartiKumasKBilgi.PartiKumasKBilgiID is not null and
           (PartiKumasKBilgi.inserttarihi >= :ABaslangicInsertTarihi and PartiKumasKBilgi.inserttarihi <= :ABitisInsertTarihi ) and
           (Asama.AsamaId = :AAsamaId or :AAsamaId is null) and
           (:ABaslangicBoyaSiparisYil = 0 or
           ((BoyaSiparis.BoyaSiparisYil > :ABaslangicBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABaslangicBoyaSiparisYil and BoyaSiparis.BoyaSiparisId >= :ABaslangicBoyaSiparisId)) and
           (BoyaSiparis.BoyaSiparisYil < :ABitisBoyaSiparisYil or (BoyaSiparis.BoyaSiparisYil = :ABitisBoyaSiparisYil and BoyaSiparis.BoyaSiparisId <= :ABitisBoyaSiparisId))))
    group by PartiStokDetay.PartiSinifId, HamStokMaster.HamStokMasterID, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj,
             HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, Ormeci.Adi, HamStokMaster.OrmeciMakinaNo,
             HamStokMaster.Pus, HamStokMaster.Feine, LycraUretici.Adi, HamStokMaster.Lot

     into :PartiSinifId,:HamStokMasterID, :YikamaOncesiEn, :YikamaOncesiGramaj, :YikamaSonrasiEn, :YikamaSonrasiGramaj, :EndenCekmezlik,
          :BoydanCekmezlik, :HamDonme, :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu, :IrsaliyeNo, :IrsaliyeTarihi, :OrmeciAdi, :OrmeciMakinaNo,
          :Pus, :Feine, :LycraUreticiAdi, :Lot, :EndenCekmezlikStr, :BoydanCekmezlikStr


    do begin
       suspend;
    end
   end
end
