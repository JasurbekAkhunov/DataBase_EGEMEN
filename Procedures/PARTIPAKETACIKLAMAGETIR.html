CREATE PROCEDURE PARTIPAKETACIKLAMAGETIR(
    APARTIPAKETID INTEGER)
RETURNS (
    ACIKLAMA VARCHAR(255))
AS
declare variable FAsamaKisaAdi varchar(10);
declare variable FReceteId varchar(20);
declare variable FReceteAdi varchar(40);
declare variable FIlaveIslemAdi varchar(30);
begin
  Aciklama = '';
  For select Coalesce(Asama.AsamaKisaAdi, ''), partiAsamalari.ReceteId, Recete.Adi, IlaveIslemler.Adi from PartiPaketDetay
  left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiPaketDetay.PartiAsamalariId
  left join PartiIlaveIslemler on PartiIlaveIslemler.PartiIlaveIslemlerId = PartiPaketDetay.PartiIlaveIslemlerId
  left join SiparisIlaveIslemler on SiparisIlaveIslemler.SiparisIlaveIslemlerId = PartiPaketDetay.SiparisIlaveIslemlerId
  left join Recete on Recete.ReceteId = partiAsamalari.ReceteId
  left join asama on asama.AsamaId = partiAsamalari.AsamaId or asama.AsamaId = recete.UstAsamaId or asama.AsamaId = PartiPaketDetay.AsamaId
  left join ilaveislemler on ilaveislemler.ilaveislemlerid = PartiIlaveIslemler.IlaveIslemlerId or ilaveislemler.ilaveislemlerid = SiparisIlaveIslemler.IlaveIslemlerId
  where PartiPaketDetay.PartiPaketId = :APartiPaketId
  order by PartiPaketDetay.PartiPaketDetayId
  into :FAsamaKisaAdi, :FReceteId, :FReceteAdi, :FIlaveIslemAdi do
  begin
    if (Char_Length(Aciklama) < 210) then
    begin
      if (Aciklama <> '') then Aciklama = Aciklama || '+';

      if (FIlaveIslemAdi is not null) then
        Aciklama = Aciklama || FIlaveIslemAdi;
      else
        if (FAsamaKisaAdi is not null) then Aciklama = Aciklama || FAsamaKisaAdi;
    end
  end
  suspend;
end
