CREATE PROCEDURE FASONIADEICINHAMSTOKRAFLISTESI(
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER,
    ADEPOID VARCHAR(10))
RETURNS (
    ORMERAFID INTEGER,
    RAFID VARCHAR(10),
    STOKGIRISNO INTEGER,
    KALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI INTEGER,
    LOT VARCHAR(150),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    URETICIADI VARCHAR(40))
AS
declare variable HamTipiID integer;
declare variable hstIplik integer;
declare variable hstLycra integer;
begin
  Select Ham.HamTipiID from Ham where Ham.HamID = :AHamId into :HamTipiID;

  Select hstIplik, hstLycra from HamTipiGetir into :hstIplik, :hstLycra;

  if (HamTipiID = hstLycra or HamTipiID = hstIplik) then
  begin
    For Select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.RafId, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
               OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi,  OrmeRaf.Lot, Uretici.Adi from OrmeRaf
      left join Renk on Renk.RenkId = OrmeRaf.RenkId
      left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
      left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
      where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and
           ( OrmeRaf.StokFirmaId = :AFirmaId and OrmeRaf.FasonStok = 1) and
           (DepoRaf.DepoId = :ADepoID or :ADepoID is null)
      into :OrmeRafId, :StokGirisNo, :RafId, :RenkId, :RenkAdi, :KalanMiktar, :KalanKapAdedi, :Lot, :UreticiAdi do
    begin
      suspend;
    end
  end
  else
  begin
    For Select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.RafId, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
               OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi,  OrmeRaf.Lot, Uretici.Adi from OrmeRaf
      left join Renk on Renk.RenkId = OrmeRaf.RenkId
      left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
      left join DepoRaf on DepoRaf.RafId = OrmeRaf.RafId
      where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and
           ( OrmeRaf.StokFirmaId = :AFirmaId and OrmeRaf.FasonStok = 1 and BoyahaneFasonIade = 1) and
           (DepoRaf.DepoId = :ADepoID or :ADepoID is null)
      into :OrmeRafId, :StokGirisNo, :RafId, :RenkId, :RenkAdi, :KalanMiktar, :KalanKapAdedi, :Lot, :UreticiAdi do
    begin
      suspend;
    end
  end
end
