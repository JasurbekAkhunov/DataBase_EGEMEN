CREATE PROCEDURE PARTIRECETESIDOKUMU_3(
    APARTIKAYITID INTEGER,
    ARECETEID VARCHAR(20),
    APARTIASAMALARIID INTEGER)
RETURNS (
    PARTIKAYITID INTEGER,
    BATCHNO VARCHAR(30),
    PARTISINIFID INTEGER,
    PARTISIRANO INTEGER,
    BOYASIPARISDETAYID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    MUSTERISIPARISNO VARCHAR(50),
    MUSTERIKALITENO VARCHAR(20),
    MUSTERIKALITEADI VARCHAR(50),
    MAKINAPLANLAMAGRUPID VARCHAR(10),
    PLANLANANMAKINAID VARCHAR(10),
    HAMADITOPLAMI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    ORMESIPDETAYHAMID INTEGER,
    ORMESIPDETAYHAMADI VARCHAR(255),
    KILO NUMERIC(12,3),
    METRE NUMERIC(12,3),
    ONGORULENEN INTEGER,
    ONGORULENGRAMAJ INTEGER,
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    BELGEHAMADI VARCHAR(60),
    PROSESNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SIPARISPLANLAMANOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SIPARISSIPARISNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    REFERANSPARTIYIL INTEGER,
    REFERANSPARTIID INTEGER,
    REFERANSPARTIIDSTR VARCHAR(30),
    REFERANSPARTISINIFID INTEGER,
    SIPARISMELANJKODU VARCHAR(20),
    SIPARISHAMRENKADI VARCHAR(50),
    TAMIRTIPIADI VARCHAR(20),
    PARTISTOKASAMALARI VARCHAR(1024),
    PARTISTOKTUMASAMALARI VARCHAR(1024),
    LABRECETEACIKLAMA VARCHAR(40),
    LABRECETEACIKLAMA2 VARCHAR(40),
    KALITEID VARCHAR(20),
    TOPLAMBOBINSAYISI INTEGER,
    REFERANSSIPARISNO VARCHAR(30),
    LABRECETEKODU VARCHAR(20),
    LABRECETEATISID INTEGER,
    LABRECETEATISNO INTEGER,
    ORMESIPARISNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ORMESIPARISURETIMNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    MUSTERITALIMATI BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SIPARISTERMINNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SIPILAVEISLEMLER VARCHAR(255),
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    ILAVEISLEMLER VARCHAR(255),
    OZELTESTLER VARCHAR(255),
    BOYANANKILO DOUBLE PRECISION,
    SIPARISKILO DOUBLE PRECISION,
    ICTAMIRSTR VARCHAR(255),
    PONO VARCHAR(50),
    SASNO VARCHAR(50),
    SIPARISSARJBOBINSAYISI NUMERIC(12,3),
    SIPARISSARJGRAMAJ INTEGER,
    KALITERECETEKODU VARCHAR(20),
    KALITERECETEADI VARCHAR(40),
    KALITECOMPOSITION VARCHAR(255))
AS
declare variable HamStokDetayId integer;
declare variable PartiAsamalariId integer;
declare variable AsamaKisaAdi varchar(10);
declare variable FTamirTipiAdi varchar(20);
begin
  for select PartiKayitId, PartiAsamalariId from PartiRecetesiDokumu_1(:APartiKayitId, :AReceteId, :APartiAsamalariId)
    into :PartiKayitId, :PartiAsamalariId do
  begin
    Select ProsesNotu from PartiAsamalari where PartiAsamalari.PartiAsamalariID = :PartiAsamalariID into :ProsesNotu;

    for select PartiStok.PartiSinifId, PartiStok.BoyaSiparisDetayId, PartiAsamaStok.GirisKilo, PartiAsamaStok.GirisMetre, PartiStok.HamAdiToplami, Ham.Adi2, GrupPlanlama.MakinaPlanlamaGrupId, UretimPlanlama.MakinaId,
               PartiStok.ReferansPartiSinifId, BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen,
               BoyaSiparisDetay.MelanjKodu, HamRenk.Adi, TamirTipi.Adi, LabRecete.Aciklama, LabRecete.Aciklama2, BoyaSiparisDetay.KaliteId, BoyaSiparisDetay.ReferansSiparisNo,
               Min(RafDetay.HamStokDetayId), sum(PartiStokDetay.BobinSayisi), BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.LabReceteAtisId, LabReceteAtis.AtisNo, BoyaSiparisDetay.BelgeHamAdi, BoyaSiparisDetay.OrmeSiparisDetayId,
               BoyaSiparisDetay.SiparisKilo, PartiStok.PartiSiraNo, icTamirTipi.Adi, BoyaSiparisSarj.BobinSayisi, BoyaSiparisSarj.Gramaj, Kalite.ReceteKodu, Kalite.ReceteAdi, Kalite.Composition from PartiAsamalari
      left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
      left join PartiStok on PartiStok.PartiSinifId = PartiAsamaStok.PartiSinifId
      left join Ham on Ham.HamId = PartiStok.HamId
      left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
      left join BolumPlanlama on BolumPlanlama.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId and BolumPlanlama.PartiSinifId = PartiAsamaStok.PartiSinifId
      left join GrupPlanlama on GrupPlanlama.BolumPlanlamaID = BolumPlanlama.BolumPlanlamaId
      left join UretimPlanlamaDetay on UretimPlanlamaDetay.GrupPlanlamaID = GrupPlanlama.GrupPlanlamaId
      left join UretimPlanlama on UretimPlanlama.UretimPlanlamaID = UretimPlanlamaDetay.UretimPlanlamaId
      left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BoyaSiparisDetay.OrmeSiparisDetayId
      left join Renk HamRenk on HamRenk.RenkId = OrmeSiparisDetay.RenkId
      left join TamirTipi on TamirTipi.TamirTipiId = PartiStok.TamirTipiId
      left join LabRecete on LabRecete.LabReceteId = BoyaSiparisDetay.LabReceteId
      left join LabReceteAtis on LabReceteAtis.LabReceteAtisID = BoyaSiparisDetay.LabReceteAtisID
      left join PartiTamir PartiIcTamir on partiIcTamir.PartiTamirId = PartiAsamalari.PartiTamirId
    left join Kalite on Kalite.KaliteId = BoyaSiparisDetay.KaliteId
      left join TamirTipi IcTamirTipi on icTamirTipi.TamirTipiId = partiIcTamir.TamirTipiId
      left join BoyaSiparisSarj on BoyaSiparisSarj.BoyaSiparisSarjId = PartiStok.BoyaSiparisSarjId
      where PartiAsamalari.PartiAsamalariId = :PartiAsamalariId and PartiAsamaStok.AsamayaDahil = 1 and PartiStok.Kilo is not null
      group By PartiStok.PartiSinifId, PartiStok.BoyaSiparisDetayId, PartiAsamaStok.GirisKilo, PartiAsamaStok.GirisMetre, PartiStok.HamAdiToplami, Ham.Adi2,
              GrupPlanlama.MakinaPlanlamaGrupId, UretimPlanlama.MakinaId, PartiStok.ReferansPartiSinifId,
              BoyaSiparisDetay.EnIstenen, BoyaSiparisDetay.GramajIstenen, BoyaSiparisDetay.MelanjKodu, HamRenk.Adi,
              TamirTipi.Adi, LabRecete.Aciklama, LabRecete.Aciklama2, BoyaSiparisDetay.KaliteId, BoyaSiparisDetay.ReferansSiparisNo,
              BoyaSiparisDetay.LabReceteKodu, BoyaSiparisDetay.LabReceteAtisId, LabReceteAtis.AtisNo, BoyaSiparisDetay.BelgeHamAdi, BoyaSiparisDetay.OrmeSiparisDetayId,
              BoyaSiparisDetay.SiparisKilo, PartiStok.PartiSiraNo, icTamirTipi.Adi, BoyaSiparisSarj.BobinSayisi, BoyaSiparisSarj.Gramaj, Kalite.ReceteKodu, Kalite.ReceteAdi, Kalite.Composition
      order by PartiStok.PartiSinifId
      into :PartiSinifId, :BoyaSiparisDetayId, :Kilo, :Metre, :HamAdiToplami, :HamAdi2,  :MakinaPlanlamaGrupId, :PlanlananMakinaId,
            :ReferansPartiSinifId, :EnIstenen, :GramajIstenen,
            :SiparisMelanjKodu, :SiparisHamRenkAdi, :TamirTipiAdi, :LabReceteAciklama, :LabReceteAciklama2, :KaliteId, :ReferansSiparisNo,
            :HamStokDetayId, :ToplamBobinSayisi, :LabReceteKodu, :LabReceteAtisId, :LabReceteAtisNo, :BelgeHamAdi, :OrmeSiparisDetayId,
            :SiparisKilo, :PartiSiraNo, :IcTamirStr, :SiparisSarjBobinSayisi, :SiparisSarjGramaj, :KaliteReceteKodu, :KaliteReceteAdi, :KaliteComposition do
    begin

      Select BoyaSiparisDetay.MusteriSiparisNo, BoyaSiparisDetay.BatchNo, BoyaSiparisDetay.MusteriKaliteNo, BoyaSiparisDetay.MusteriKaliteAdi,
             BoyaSiparisDetay.PlanlamaNotu, BoyaSiparisDetay.Notu, BoyaSiparisDetay.TerminNotu, BoyaSiparisDetay.IlaveIslemler,
             BoyaSiparisDetay.SiparisProsesId, SiparisProses.Adi, BoyaSiparisDetay.IlaveIslemler, BoyaSiparisDetay.OzelTestler, BoyaSiparisDetay.SasNo, BoyaSiparisDetay.PoNo from BoyaSiparisDetay
        left join SiparisProses on SiparisProses.SiparisProsesId = BoyaSiparisDetay.SiparisProsesId
        where BoyaSiparisDetay.BoyaSiparisDetayID = :BoyaSiparisDetayId
        into :MusteriSiparisNo, :BatchNo, :MusteriKaliteNo, :MusteriKaliteAdi,
             :SiparisPlanlamaNotu, :SiparisSiparisNotu, :SiparisTerminNotu, :SipIlaveIslemler,
             :SiparisProsesId, :SiparisProsesAdi, :IlaveIslemler, :OzelTestler, :SasNo, :PoNo;

      Select OrmeSiparisDetay.HamID, Ham.Adi, OrmeSiparisDetay.Notu, OrmeSiparisDetay.UretimNotu, OrmeSiparisDetay.MusteriTalimati from OrmeSiparisDetay
        left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
        where OrmeSiparisDetay.OrmeSiparisDetayId = :OrmeSiparisDetayId
        into :OrmeSipDetayHamID, :OrmeSipDetayHamAdi, :OrmeSiparisNotu, :OrmeSiparisUretimNotu, :MusteriTalimati;


      Select BoyaSiparisDetayOngoru.OngorulenEn, BoyaSiparisDetayOngoru.OngorulenGramaj from BoyaSiparisDetayOngoru
        where BoyaSiparisDetayOngoru.BoyaSiparisDetayId = :BoyaSiparisDetayId and BoyaSiparisDetayOngoru.HamStokDetayId = :HamStokDetayId
        into :OngorulenEn, :OngorulenGramaj;

      Select Sum(PartiStok.Kilo) From PartiStok where PartiStok.Durumu > 4 and PartiStok.BoyaSiparisDetayID = :BoyaSiparisDetayId into :BoyananKilo;

      ReferansPartiYil = Null;
      ReferansPartiId = Null;
      ReferansPartiIdStr = Null;

      if (ReferansPartiSinifId is not null) then
      begin
        Select Parti.PartiYil, Parti.PartiId  from PartiStok
          left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
          where PartiStok.PartiSinifId = :ReferansPartiSinifId
          into :ReferansPartiYil, :ReferansPartiId;

        ReferansPartiIdStr =  '(' || EgemenIntegerToStr(ReferansPartiYil)  || ') ' || EgemenIntegerToStr(ReferansPartiId) || ' ';
      end

      PartiStokAsamalari = '';
      For Select Asama.AsamaKisaAdi from PartiStok
        left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = partistok.PartiSinifId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        where PartiStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.Giristarihi is null and
              PartiAsamaStok.AsamayaDahil = 1 And PartiAsamalari.OtomasyonaDahilmi = 1
        order by PartiAsamalari.Sira
        into :AsamaKisaAdi do
      begin
        if (AsamaKisaAdi is not Null) then
        begin
          if (Char_Length(AsamaKisaAdi) > 0) then
          begin
            if (Char_Length(PartiStokAsamalari) + Char_Length(AsamaKisaAdi) < 1014) then
            begin
              if (PartiStokAsamalari <> '' ) then PartiStokAsamalari = PartiStokAsamalari || cast(', ' as varchar(2) character set UTF8);
              PartiStokAsamalari = PartiStokAsamalari || AsamaKisaAdi;
            end
          end
        end
      end

      PartiStokTumAsamalari = '';
      For Select Asama.AsamaKisaAdi from PartiStok
        left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = partistok.PartiSinifId
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariId
        left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
        where PartiStok.PartiSinifId = :PartiSinifId and
              PartiAsamaStok.AsamayaDahil = 1 And PartiAsamalari.OtomasyonaDahilmi = 1
        order by PartiAsamalari.Sira
        into :AsamaKisaAdi do
      begin
        if (AsamaKisaAdi is not Null) then
        begin
          if (Char_Length(AsamaKisaAdi) > 0) then
          begin
            if (Char_Length(PartiStokTumAsamalari) + Char_Length(AsamaKisaAdi) < 1014) then
            begin
              if (PartiStokTumAsamalari <> '' ) then PartiStokTumAsamalari = PartiStokTumAsamalari || cast(', ' as varchar(2) character set UTF8);
              PartiStokTumAsamalari = PartiStokTumAsamalari || AsamaKisaAdi;
            end
          end
        end
      end

   /*   IcTamirStr = '';
      for Select TamirTipi.Adi from PartiTamirSinif
        left join PartiTamir on PartiTamir.PartiTamirId = PartiTamirSinif.PartiTamirId
        left join TamirTipi on TamirTipi.TamirTipiId = PartiTamir.TamirTipiId
        where PartiTamirSinif.PartiSinifId = :PartiSinifId order by PartiTamir.Tarih into :FTamirTipiAdi do
      begin
        if (IcTamirStr = '') then
          IcTamirStr = FTamirTipiAdi;
        else
          IcTamirStr = IcTamirStr || ' - ' || FTamirTipiAdi;
      end    */

      suspend;
    end
  end
end
