CREATE PROCEDURE BUKUMECIKISICINSTOKLISTESI(
    ABOYASIPARISKAYITID INTEGER,
    AHAMID INTEGER,
    ADEPOID VARCHAR(10),
    ABUKUMTALIMATID INTEGER)
RETURNS (
    ORMERAFID INTEGER,
    STOKGIRISNO INTEGER,
    RAFID VARCHAR(10),
    KALANMIKTAR NUMERIC(12,3),
    KALANKAPADEDI INTEGER,
    LOT VARCHAR(150),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    URETICIADI VARCHAR(40),
    MELANJKODU VARCHAR(20),
    FASON INTEGER,
    FASONSTR VARCHAR(5))
AS
declare variable SiparisFirmaId       varchar(20);
declare variable OrmeFirmaKodumuz     varchar(20);
declare variable assSatis             integer;
declare variable assFasonArtiSatis    integer;
declare variable assFason             integer;
declare variable OrmeSiparisSekliId   integer;
declare variable IplikUretimCikistaSipTakibiVar SmallInt;
declare variable DokumaSiparisi SmallInt;
declare variable KS Integer;
declare variable StokFirmaId varchar(20);
begin
  Select assSatis, assFasonArtiSatis, assFason from OrmeSiparisSekliGetir into :assSatis, :assFasonArtiSatis, :assFason;

  Select BoyaSiparis.FirmaId, BoyaSiparis.OrmeSiparisSekliId, BoyaSiparis.DokumaSiparisi from BoyaSiparis
  where BoyaSiparis.BoyaSiparisKayitId = :ABoyaSiparisKayitId
  into :SiparisFirmaId, :OrmeSiparisSekliId, :DokumaSiparisi;
  Select iif(:DokumaSiparisi = 1, DokumaFirmaKodumuz, OrmeFirmaKodumuz), IplikUretimCikistaSipTakibiVar from Sistem where SistemId = 1 into :OrmeFirmaKodumuz, :IplikUretimCikistaSipTakibiVar;

  if (ABukumTalimatID is null) then
  begin

    if (OrmeSiparisSekliId is null or OrmeSiparisSekliId = assSatis) then SiparisFirmaId = Null;
    if (OrmeSiparisSekliId <> assSatis) then OrmeFirmaKodumuz = Null;

    For Select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.RafId, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
               OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi, OrmeRaf.Lot, Uretici.Adi, OrmeRaf.MelanjKodu from OrmeRaf
      left join Renk on Renk.RenkId = OrmeRaf.RenkId
      left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
      left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
      where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and OrmeRaf.DesenID is null and
           ( ((OrmeRaf.StokFirmaId = :SiparisFirmaId or OrmeRaf.StokFirmaId = :OrmeFirmaKodumuz) and OrmeRaf.FasonStok = 1) or
             (OrmeRaf.StokFirmaId = :OrmeFirmaKodumuz and OrmeRaf.FasonStok = 0) ) and
           (DepoRaf.DepoID = :ADepoID or :ADepoID is null) and
           (MiktarBirim = 0) and
           ((OrmeRaf.BoyaSiparisKayitId is Null and :IplikUretimCikistaSipTakibiVar = 1) or :IplikUretimCikistaSipTakibiVar = 0)
      into :OrmeRafId, :StokGirisNo, :RafId, :RenkId, :RenkAdi, :KalanMiktar, :KalanKapAdedi, :Lot, :UreticiAdi, :MelanjKodu do
    begin
      suspend;
    end
  end
  else
  begin
    For Select distinct BukumTalimatRecete.Fason, BukumTalimatRecete.RenkId from BukumTalimatRecete
      Left Join Ham On Ham.HamId = BukumTalimatRecete.HamId
      Left Join HamTipi On HamTipi.HamTipiId = Ham.HamTipiId
      left join Renk on Renk.RenkId = BukumTalimatRecete.RenkId
      left join BukumTalimat on BukumTalimat.BukumTalimatId = BukumTalimatRecete.BukumTalimatId
      left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = BukumTalimat.OrmeSiparisDetayId
      where BukumTalimatRecete.BukumTalimatId = :ABukumTalimatId  and BukumTalimatRecete.HamID = :AHamId
    into :Fason, :RenkId do
    begin
      if (Fason = 1) then
      begin
        FasonStr = 'Fason';
        StokFirmaID = SiparisFirmaId;
      end
      else
      begin
        FasonStr = '';
        StokFirmaID = OrmeFirmaKodumuz;
      end

      For Select OrmeRaf.OrmeRafId, OrmeRaf.StokGirisNo, OrmeRaf.RafId, OrmeRaf.RenkId, Renk.Adi, OrmeRaf.NetGirisMiktari - OrmeRaf.NetCikisMiktari,
                 OrmeRaf.GirisKapAdedi - OrmeRaf.CikisKapAdedi, OrmeRaf.Lot, Uretici.Adi, OrmeRaf.MelanjKodu from OrmeRaf
          left join Renk on Renk.RenkId = OrmeRaf.RenkId
          left join Uretici on Uretici.UreticiId = OrmeRaf.UreticiId
          left join DepoRaf on DepoRaf.RafID = OrmeRaf.RafId
          where OrmeRaf.HamID = :AHamId and OrmeRaf.StokVar = 1 and OrmeRaf.DesenID is null and
               (OrmeRaf.StokFirmaId = :StokFirmaID) and
               (OrmeRaf.FasonStok = :Fason) and
               (DepoRaf.DepoID = :ADepoID or :ADepoID is null) and
               (MiktarBirim = 0) and
               ((OrmeRaf.BoyaSiparisKayitId is Null and :IplikUretimCikistaSipTakibiVar = 1) or :IplikUretimCikistaSipTakibiVar = 0) and
               IntKarsilastir(OrmeRaf.RenkId, :RenkId) = 1
        into :OrmeRafId, :StokGirisNo, :RafId, :RenkId, :RenkAdi, :KalanMiktar, :KalanKapAdedi, :Lot, :UreticiAdi, :MelanjKodu do
      begin
        suspend;
      end
    end
  end
end
