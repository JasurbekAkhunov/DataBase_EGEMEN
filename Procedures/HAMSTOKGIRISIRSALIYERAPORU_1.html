CREATE PROCEDURE HAMSTOKGIRISIRSALIYERAPORU_1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20),
    AGIRISTIPI INTEGER,
    ATAMIR INTEGER,
    AIRSALIYENO IRSALIYENOSIZE,
    AHAMTIPI INTEGER,
    AHAMSTOKHAREKETTIPIID INTEGER,
    AIRSALIYEGRUPID VARCHAR(4))
RETURNS (
    HAMSTOKIRSALIYEID INTEGER,
    HAMSTOKHAREKETTIPIID INTEGER,
    HAMSTOKHAREKETTIPIIDSTR VARCHAR(10),
    KANTARKAYITID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SINIF VARCHAR(1),
    EVRAKNO VARCHAR(10),
    IRSALIYESERINO SERINOSIZE5,
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    IRSALIYEGRUPID VARCHAR(4),
    IRSALIYEGRUPADI VARCHAR(30),
    ARACID VARCHAR(11),
    ARACACIKLAMA VARCHAR(30),
    ARACKAPASITE DOUBLE PRECISION,
    SOFORADI VARCHAR(40),
    FIILISEVKTARIHI TIMESTAMP,
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    ORMECIADI VARCHAR(30),
    FASON INTEGER,
    FASONSTR VARCHAR(5))
AS
begin
  for select DISTINCT HamStokIrsaliye.HamStokIrsaliyeId, HamStokIrsaliye.HamStokHareketTipiId, HamStokIrsaliye.FirmaId, Firma.Adi, HamStokIrsaliye.IrsaliyeSeriNo,
             HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi, HamStokIrsaliye.IrsaliyeGrupId, StokHareketGrup.Adi,
             HamStokIrsaliye.AracId, Arac.Kapasite, Arac.Aciklama, Sofor.Adi, HamStokIrsaliye.FiiliSevkTarihi, HamStokIrsaliye.Notu, HamStokIrsaliye.KantarKayitId ,
             HamStokIrsaliye.Fason, Ormeci.Adi, HamStokIrsaliye.EvrakNo from HamStokGirisIrsaliyeRaporu_2(:ABaslangicTarihi, :ABitisTarihi, :AFirmaId, :AGirisTipi, :ATamir, :AIrsaliyeNo, :AHamTipi, :AHamStokHareketTipiId, :AIrsaliyeGrupID)
    Left Join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokGirisIrsaliyeRaporu_2.HamStokIrsaliyeId
    left join Firma on Firma.Firmaid = HamStokIrsaliye.Firmaid
    left join Arac on Arac.Aracid = HamstokIrsaliye.Aracid
    left join Sofor on Sofor.Soforid = HamstokIrsaliye.Soforid
    left join Ormeci on Ormeci.Ormeciid = HamstokIrsaliye.Ormeciid
    left join StokHareketGrup on StokHareketGrup.StokHareketGrupId = HamstokIrsaliye.IrsaliyeGrupId
    into :HamStokIrsaliyeId, :HamStokHareketTipiId, :FirmaId, :FirmaAdi, :IrsaliyeSeriNo, :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi,  :IrsaliyeGrupId, :IrsaliyeGrupAdi,
         :AracId, :AracKapasite,  :AracAciklama, :SoforAdi, :FiiliSevkTarihi, :Notu, :KantarKayitId, :Fason, :OrmeciAdi, :EvrakNo do
  begin
    if (Fason = 1) then FasonStr = 'Fason';
    else FasonStr = '';

    if (HamStokHareketTipiId = 0) then HamStokHareketTipiIdStr = 'Giriş';
    else if (HamStokHareketTipiId = 1) then HamStokHareketTipiIdStr = 'Satış İade';
    else HamStokHareketTipiIdStr = '';

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';

    suspend;
  end
end
