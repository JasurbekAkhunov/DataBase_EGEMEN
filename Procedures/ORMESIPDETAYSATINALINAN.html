CREATE PROCEDURE ORMESIPDETAYSATINALINAN(
    AORMESIPARISDETAYID INTEGER)
RETURNS (
    BYSATINALINANKUMAS DOUBLE PRECISION,
    ORMESATINALINANKUMAS DOUBLE PRECISION)
AS
declare variable hstElyaf               integer;
declare variable hstIplik               integer;
declare variable hstOrmeKumas           integer;
declare variable hstLycra               integer;
declare variable hstTelef        integer;
declare variable ShtAlis                integer;

begin
  Select hstIplik, hstOrmeKumas, hstLycra, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas,:hstLycra, :hstElyaf, :hstTelef;

  Select ShtAlis from OrmeStokHareketTipiGetir into :ShtAlis;

  Select Sum(HamStokMaster.Miktar) from OrmeSiparisDetay
    left join HamStokMaster on HamStokMaster.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
    left join Ham On Ham.HamID = HamStokMaster.HamId
    where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId and
          HamStokMaster.HamStokMasterID is not null and HamStokMaster.HamStokHareketTipiId = 0 and (Ham.HamTipiID <> :hstIplik and Ham.HamTipiID <> :hstElyaf and Ham.HamTipiID <> :hstTelef) and
          HamStokMaster.StokHareketId is null and HamStokMaster.Fason = 0
    into :BYSatinAlinanKumas;

  Select Sum(StokHareket.Miktar) from OrmeSiparisDetay
    left join StokHareket on StokHareket.OrmeSiparisDetayId = OrmeSiparisDetay.OrmeSiparisDetayId
    left join Ham On Ham.HamID = StokHareket.HamId
    where OrmeSiparisDetay.OrmeSiparisDetayId = :AOrmeSiparisDetayId and
          StokHareket.StokHareketID is not null and (Ham.HamTipiID <> :hstIplik and Ham.HamTipiID <> :hstElyaf and Ham.HamTipiID <> :hstTelef) and
          StokHareket.HareketTipi = :shtAlis
    into :OrmeSatinAlinanKumas;
  suspend;
end
