CREATE PROCEDURE DOKUMAKKHATAMETREKARTI(
    ADOKUMAKALITEKONTROLID INTEGER)
RETURNS (
    DOKUMAKALITEKONTROLHATAMETREID INTEGER,
    DOKUMAKALITEKONTROLID INTEGER,
    DOKUMAHATATIPIID INTEGER,
    DOKUMAHATATIPIADI VARCHAR(20),
    NOKTASALHATA INTEGER,
    NOKTASALHATASTR VARCHAR(4),
    BASLANGICMETRE NUMERIC(12,3),
    BITISMETRE NUMERIC(12,3),
    HATAPUANI DOUBLE PRECISION,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  for Select DokumaKaliteKontrolHataMetre.DokumaKaliteKontrolId, DokumaKaliteKontrolHataMetre.DokumaKaliteKontrolHataMetreId,
             DokumaKaliteKontrolHataMetre.DokumaHataTipiId, DokumaHataTipi.Adi, DokumaHataTipi.NoktasalHata,
             DokumaKaliteKontrolHataMetre.BaslangicMetre, DokumaKaliteKontrolHataMetre.BitisMetre, DokumaHataTipi.HataPuani,
             IK.Adi, DokumaKaliteKontrolHataMetre.InsertTarihi, SK.Adi, DokumaKaliteKontrolHataMetre.DegisimTarihi from DokumaKaliteKontrolHataMetre
  left join DokumaHataTipi on DokumaHataTipi.DokumaHataTipiId  = DokumaKaliteKontrolHataMetre.DokumaHataTipiId
  left join Kullanici IK on IK.KullaniciId  = DokumaKaliteKontrolHataMetre.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = DokumaKaliteKontrolHataMetre.KullaniciId
  where DokumaKaliteKontrolHataMetre.DokumaKaliteKontrolID = :ADokumaKaliteKontrolId
  into :DokumaKaliteKontrolId, :DokumaKaliteKontrolHataMetreId, :DokumaHataTipiId, :DokumaHataTipiAdi, :NoktasalHata,
       :BaslangicMetre, :BitisMetre, :HataPuani, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi do
  begin

    if (NoktasalHata = 1) then NoktasalHataStr = 'Evet';
    else NoktasalHataStr = '';

    suspend;
  end
end
