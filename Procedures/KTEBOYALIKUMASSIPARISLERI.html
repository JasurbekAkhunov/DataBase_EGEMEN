CREATE PROCEDURE KTEBOYALIKUMASSIPARISLERI(
    AFIRMAID VARCHAR(20),
    AHAMID INTEGER)
RETURNS (
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    SIPARISGRUPID VARCHAR(4),
    SIPARISGRUP2ID VARCHAR(4),
    SIPARISIVEREN VARCHAR(50),
    SIPARISTARIHI TIMESTAMP,
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    HAMMIKTAR DOUBLE PRECISION,
    SATINALINACAKMIKTAR DOUBLE PRECISION,
    STOKTANKULLANILACAKMIKTAR DOUBLE PRECISION,
    TESLIMTARIHI TIMESTAMP,
    REVIZETESLIMTARIHI TIMESTAMP,
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    MELANJKODU VARCHAR(20),
    FLAMKODU VARCHAR(20),
    PUS INTEGER,
    FEINE INTEGER,
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    MAKINAUSTUGRAMAJ INTEGER)
AS
declare variable astBoyaliKumas         integer;
declare variable dstDokuma              integer;
declare variable assSatis               integer;
declare variable OnaysizSiptenKTeminSipOlmaz integer;
begin /* sadece satış ve boyalı kumas siparişleri listelenir */
  Select AstBoyaliKumas from OrmeSiparisTipiGetir into :AstBoyaliKumas;
  Select dstDokuma from DokumaSiparisTipiGetir into :dstDokuma;

  Select assSatis from OrmeSiparisSekliGetir into :assSatis;
  Select OnaysizSiptenKTeminSipOlmaz from Sistem where SistemId = 1 into :OnaysizSiptenKTeminSipOlmaz;

  For Select BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, BoyaSiparis.FirmaId, Firma.Adi, BoyaSiparis.SiparisTarihi, FirmaYetkili.Adi, OrmeSiparisDetay.BoyaSiparisKayitId, OrmeSiparisDetay.OrmeSiparisDetayId,
     OrmeSiparisDetay.SiparisSiraNo, BoyaSiparis.SiparisGrupId, BoyaSiparis.SiparisGrup2Id, OrmeSiparisDetay.HamId, Ham.Adi, OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.SatinAlinacakMiktar, OrmeSiparisDetay.StoktanKullanilacakMiktar,
     OrmeSiparisDetay.Teslimtarihi, OrmeSiparisDetay.RevizeTeslimtarihi,
     OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, OrmeSiparisDetay.ArtikelNo, OrmeSiparisDetay.MelanjKodu, OrmeSiparisDetay.FlamKodu,
     OrmeSiparisDetay.Pus, OrmeSiparisDetay.Feine, OrmeSiparisDetay.TupAcikEnMayli, OrmeSiparisDetay.MakinaustuGramaj from OrmeSiparisDetay
  left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
  left join FirmaYetkili On FirmaYetkili.FirmaYetkiliId = BoyaSiparis.FirmaYetkiliId
  Left Join Ham ON Ham.HamId = OrmeSiparisDetay.HamId
  Left Join Firma On Firma.FirmaId = BoyaSiparis.FirmaId
  where OrmeSiparisDetay.Kapali = 0 and (BoyaSiparis.OrmeSiparisTipiId = :astBoyaliKumas or BoyaSiparis.DokumaSiparisTipiId = :dstDokuma) and OrmeSiparisDetay.KTBoyaSiparisKayitId is null and OrmeSiparisDetay.PlanlananMiktar = 0 and
        BoyaSiparis.OrmeSiparisSekliID = :assSatis and OrmeSiparisDetay.HamMiktar > (OrmeSiparisDetay.StoktanKullanilacakMiktar + OrmeSiparisDetay.SatinAlinacakMiktar) and
        (BoyaSiparis.FirmaId = :AFirmaId or :AFirmaID is null) and
        (OrmeSiparisDetay.HamId = :AHamId or :AHamId is null) and
        ((:OnaysizSiptenKTeminSipOlmaz = 1 and OrmeSiparisDetay.OnayDurumu = 1) or :OnaysizSiptenKTeminSipOlmaz = 0)
  into :BoyaSiparisYil, :BoyaSiparisId, :FirmaId, :FirmaAdi, :SiparisTarihi, :SiparisiVeren, :BoyaSiparisKayitId, :OrmeSiparisDetayId,
       :SiparisSiraNo, :SiparisGrupId, :SiparisGrup2Id, :HamId, :HamAdi, :HamMiktar, :SatinAlinacakMiktar, :StoktanKullanilacakMiktar,
       :Teslimtarihi, :RevizeTeslimtarihi, :MusteriSiparisNo, :ReferansSiparisNo, :ArtikelNo, :MelanjKodu, :FlamKodu,
       :Pus, :Feine, :TupAcikEnMayli, :MakinaustuGramaj do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    suspend;
  end
end
