CREATE PROCEDURE ELYAFCINSIMOGKONTROLU(
    AELYAFCINSIID VARCHAR(5),
    AKULLANICIID INTEGER)
AS
declare variable MakinaOtomasyonGrupId varchar(4);
declare variable KayitSayisi integer;
begin
  for select MakinaOtomasyonGrupId from MakinaOtomasyonGrup into :MakinaOtomasyonGrupId do
  begin
    select count(*) from ElyafCinsiMOGruplari where ElyafCinsiId = :AElyafCinsiId and MakinaOtomasyonGrupId = :MakinaOtomasyonGrupId into :KayitSayisi;
    if (KayitSayisi < 1) then
    begin
      insert into ElyafCinsiMOGruplari(ElyafCinsiId, MakinaOtomasyonGrupId, InsertKullaniciId, InsertTarihi, KullaniciId, DegisimTarihi)
                values(:AElyafCinsiId, :MakinaOtomasyonGrupId, :AKullaniciId, current_timestamp(0), :AKullaniciId, current_timestamp(0));
    end
  end
end
