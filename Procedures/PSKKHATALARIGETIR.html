CREATE PROCEDURE PSKKHATALARIGETIR(
    APSKALITEKONTROLID INTEGER)
RETURNS (
    PSKALITEKONTROLHATALARIID INTEGER,
    PSKALITEKONTROLID INTEGER,
    HATATIPIID INTEGER,
    HATATIPIADI VARCHAR(20),
    HATATIPIDIGERADI1 VARCHAR(20),
    HATATIPIDIGERADI2 VARCHAR(20),
    NOKTASALHATA INTEGER,
    NOKTASALHATASTR VARCHAR(4),
    ADET INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    PUANLAMAYADAHIL SMALLINT,
    HATAPUANI DOUBLE PRECISION,
    HATATIPIPUANI INTEGER)
AS
begin

for select PSKaliteKontrolHatalari.PSKaliteKontrolHatalariId, PSKaliteKontrolHatalari.PSKaliteKontrolId, HataTipi.HataTipiId, HataTipi.Adi, HataTipi.DigerAdi1, HataTipi.DigerAdi2,
      HataTipi.NoktasalHata, PSKaliteKontrolHatalari.Adet, IK.Adi, PSKaliteKontrolHatalari.InsertTarihi, SK.Adi, PSKaliteKontrolHatalari.DegisimTarihi,
      HataTipi.PuanlamayaDahil,HataTipi.HataTipiPuani,PsKaliteKontrolHataMetre.HataPuani from PSKaliteKontrolHatalari
  left join PSKaliteKontrolHataMetre on PSKaliteKontrolHataMetre.PSKaliteKontrolId = PSKaliteKontrolHatalari.PSKaliteKontrolId and PSKaliteKontrolHataMetre.HataTipiId = PSKaliteKontrolHatalari.HataTipiId
  left join HataTipi on HataTipi.HataTipiId = PSKaliteKontrolHatalari.HataTipiId
  left join Kullanici IK on IK.KullaniciId  = PSKaliteKontrolHatalari.InsertKullaniciId
  left join Kullanici SK on SK.KullaniciId  = PSKaliteKontrolHatalari.KullaniciId
  where PSKaliteKontrolHatalari.PSKaliteKontrolId = :APSKaliteKontrolId
  into :PSKaliteKontrolHatalariId, :PSKaliteKontrolId, :HataTipiId, :HataTipiAdi, :HataTipiDigerAdi1, :HataTipiDigerAdi2,
       :NoktasalHata, :Adet, :InsertKullaniciAdi, :InsertTarihi, :KullaniciAdi, :DegisimTarihi, :PuanlamayaDahil, :HataTipiPuani, :HataPuani do
  begin
    if (NoktasalHata = 1) then NoktasalHataStr = 'Evet';
    else NoktasalHataStr = '';

    suspend;
  end

end
