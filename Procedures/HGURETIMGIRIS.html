CREATE PROCEDURE HGURETIMGIRIS(
    ASTOKID INTEGER,
    AIRSALIYEID INTEGER,
    AFATURAID INTEGER,
    AHAREKETTIPI INTEGER,
    AHGISEMRIKAYITID INTEGER,
    AHGISEMRIPASTALID INTEGER,
    AHGISEMRIURUNID INTEGER,
    ADEPOID DEPOKODUSIZE,
    AORMEMAKINAID MAKINAKODUSIZE,
    AURUNID URUNKODUSIZE,
    AMIKTAR DOUBLE PRECISION,
    AIADE EGEBOOLEAN,
    APASTALNO TYPE OF COLUMN HGISEMRI.URETIMPARTINO,
    AKULLANICIID INTEGER)
AS
declare variable shtFasonUretim    integer;
declare variable sikStok           integer;
declare variable sikHazirGiyim     integer;
declare variable sikHazirGiyimKesim integer;
declare variable HGSiparisKayitID   integer;
declare variable HGSiparisDetayID   integer;
declare variable IslemKaynagi   EgeBoolean;
declare variable IsEmriFirmaID  FirmaKoduSize;
declare variable HarFirmaID     FirmaKoduSize;
declare variable HarDepoID      DepoKoduSize;
declare variable IsEmriMakinaID MakinaKoduSize;
declare variable IEBirimFiyat   Double Precision;
declare variable DovizKuru      Double Precision;
declare variable FKdvOrani      Double Precision;
declare variable FUrunBirimID   Integer;
declare variable IEDovizID      DovizSize;
declare variable HarDovizID     DovizSize;
declare variable IrsaliyeTarihi TimeStamp;
declare variable IEVade Integer;
declare variable IEUretimPartiNo Type of Column HGIsEmri.UretimPartiNo;
begin
  select shtFasonUretim from StokHareketTipiGetir into :shtFasonUretim;
  select sikStok, sikHazirGiyim, sikHazirGiyimKesim from StokIslemKaynagiGetir into :sikStok, :sikHazirGiyim, :sikHazirGiyimKesim;

  if (AIrsaliyeID is not Null) then
  begin
    Select IslemKaynagi, FirmaID, DovizID, IrsaliyeTarihi, DepoID from Irsaliye where Irsaliye.IrsaliyeID = :AIrsaliyeID into :IslemKaynagi, :HarFirmaID, :HarDovizID, :IrsaliyeTarihi, :HarDepoID;
    if (HarDepoID is not Null and StrKarsilastir(HarDepoID, ADepoID) = 0) then Exception KayitHatasi  ' Üst Belgenin Deposu uygun değil!';
  end
  else if (AFaturaID is not Null) then
  begin
    Select IslemKaynagi, FirmaID, DovizID, FaturaTarihi, DepoID from Fatura where Fatura.FaturaID = :AFaturaID into :IslemKaynagi, :HarFirmaID, :HarDovizID, :IrsaliyeTarihi, :HarDepoID;
    if (HarDepoID is not Null and StrKarsilastir(HarDepoID, ADepoID) = 0) then Exception KayitHatasi  ' Üst Belgenin Deposu uygun değil!';
  end
  else if (AStokId is not null) then
  begin
    Select IslemKaynagi, Null, DovizID, FisTarihi, DepoID from Stok where Stok.StokID = :AStokID into :IslemKaynagi, :HarFirmaID, :HarDovizID, :IrsaliyeTarihi, :HarDepoID;
    if (HarDepoID is not Null and StrKarsilastir(HarDepoID, ADepoID) = 0) then Exception KayitHatasi  ' Üst Belgenin Deposu uygun değil!';
  end

  if (IslemKaynagi = sikStok) then Exception KayitHatasi  ' Üst Belgenin Islem Kaynağı uygun değil!';

  Select HGSiparisKayitID, HGSiparisDetayID, FirmaID, OrmeMakinaID, BirimFiyat, DovizID, VadeGun, UretimPartiNo from HGIsEmri where HGIsEmri.HGIsEmriKayitID = :AHGIsEmriKayitID
    into :HGSiparisKayitID, :HGSiparisDetayID, :IsEmriFirmaID, :IsEmriMakinaID, :IEBirimFiyat, :IEDovizID, :IEVade, :IEUretimPartiNo;

  if (StrKarsilastir(HarFirmaID, IsEmriFirmaID) = 0) then Exception KayitHatasi  ' İş Emri Firması ile Hareket Firması Farklı!';
  if (StrKarsilastir(IsEmriMakinaID, AOrmeMakinaID) = 0) then Exception KayitHatasi  ' İş Emrindeki Makina ile Hareketin Makina Kodu Farklı!';

  if (HarDovizID <> IEDovizID) then  select DovizKuru from EnYakinDovizKuruGetir(:IEDovizID, :HarDovizID, :IrsaliyeTarihi) into :DovizKuru;
  else DovizKuru = 1;
  IEBirimFiyat = IEBirimFiyat * DovizKuru;

  Select Urun.KdvOrani, UrunBirimID from Urun
    left join UrunBirim on UrunBirim.UrunId = Urun.UrunId and UrunBirim.DefaultBirim = 1
  Where Urun.UrunID = :AUrunID into :FKdvOrani, :FUrunBirimID;

  Insert Into StokDetay(FaturaID, IrsaliyeID, StokID, HareketTipi, Iade, HGSiparisKayitID, HGSiparisDetayID, HGIsEmriKayitID, HGIsEmriPastalID, HGIsEmriUrunID, DepoId,  UrunID, Miktar, Kdv, Vadesi, UrunBirimID, BirimFiyat, InsertKullaniciID, OrmeMakinaId, PastalNo) values
                      (:AFaturaID, :AIrsaliyeID, :AStokID, :AHareketTipi, :AIade, :HGSiparisKayitID, :HGSiparisDetayID, :AHGIsEmriKayitID, :AHGIsEmriPastalID, :AHGIsEmriUrunID, :ADepoID, :AUrunID, :AMiktar, :FKdvOrani, :IEVade, :FUrunBirimID, :IEBirimFiyat, :AKullaniciID, :AOrmeMakinaID, iif(:AHGIsEmriPastalID is not Null, :IEUretimPartiNo, :APastalNo));

  suspend;
end
