CREATE PROCEDURE REFPARTIICKONTROLDEGERLERIGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTISINIFID INTEGER,
    REFERANSPARTISINIFID INTEGER,
    SIRA INTEGER,
    ASAMAID INTEGER,
    ASAMAADI VARCHAR(40),
    ICKONTROLTIPIID INTEGER,
    ICKONTROLTIPIADI VARCHAR(30),
    ONERILENDEGER VARCHAR(15),
    ISTENEN VARCHAR(50),
    SONUC VARCHAR(15))
AS
begin
/* partinin stoklarının referans partilerinin ic kontrol degerlerini getirir */
  For Select Ps.PartiSinifId, Ps.ReferansPartiSinifId, PartiAsamalari.Sira, PartiAsamaStok.AsamaId, Asama.Adi, PartiAsamaIcKontrol.IcKontrolTipiId,
             IcKontrolTipi.Adi, IcKontrolTipi.OnerilenDeger, PartiAsamaIcKontrol.Istenen, PartiAsamaIcKontrol.Sonuc from PartiStok PS
  left join PartiStok RefPS on RefPS.PartiSinifID = Ps.ReferansPartiSinifId
  left join PartiAsamaIcKontrol on PartiAsamaIcKontrol.PartiSinifID = RefPs.PartiSinifId
  Left Join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaIcKontrol.PartiAsamalariId
  Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamaIcKontrol.PartiAsamalariId and PartiAsamaStok.PartiSinifId = PartiAsamaIcKontrol.PartiSinifId
  Left Join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
  Left Join IcKontrolTipi on IcKontrolTipi.IcKontrolTipiId = PartiAsamaIcKontrol.IcKontrolTipiId
  where PS.PartiKayitID = :APartiKayitId and PS.ReferansPartiSinifId is not null
  order by PartiAsamaIcKontrol.PartiKayitId, PartiAsamalari.Sira
  into :PartiSinifId, :ReferansPartiSinifId, :Sira, :AsamaId, :AsamaAdi, :IcKontrolTipiId, :IcKontrolTipiAdi, :OnerilenDeger, :Istenen, :Sonuc  do
  begin
    suspend;
  end
end
