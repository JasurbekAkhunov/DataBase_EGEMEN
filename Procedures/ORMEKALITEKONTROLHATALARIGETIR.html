CREATE PROCEDURE ORMEKALITEKONTROLHATALARIGETIR(
    AORMEPLANLAMAID INTEGER)
RETURNS (
    ORMEKALITEKONTROLHATALARIID INTEGER,
    ORMEKALITEKONTROLID INTEGER,
    HAMHATATIPIID INTEGER,
    HAMHATATIPIADI VARCHAR(20),
    ADET INTEGER,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP)
AS
begin
  For Select OrmeKaliteKontrolHatalari.OrmeKaliteKontrolHatalariId, OrmeKaliteKontrol.OrmeKaliteKontrolId, OrmeKaliteKontrolHatalari.HamhataTipiId, HamHataTipi.Adi,
             OrmeKaliteKontrolHatalari.Adet,  IK.Adi, OrmeKaliteKontrolHatalari.InsertTarihi, SK.Adi, OrmeKaliteKontrolHatalari.DegisimTarihi from OrmeKaliteKontrol
    left join OrmekaliteKontrolHatalari on OrmekaliteKontrolHatalari.OrmeKaliteKontrolId  = OrmeKaliteKontrol.OrmeKaliteKontrolId
    left join HamHataTipi on HamHataTipi.HamHataTipiId  = OrmeKaliteKontrolHatalari.HamHataTipiId
    left join Kullanici IK on IK.KullaniciId  = OrmeKaliteKontrolHatalari.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = OrmeKaliteKontrolHatalari.KullaniciId
    where OrmeKaliteKontrol.OrmePlanlamaId = :AOrmePlanlamaId
  into :OrmeKaliteKontrolHatalariId, :OrmeKaliteKontrolId, :HamHataTipiId, :HamHataTipiAdi, :Adet, :InsertKullaniciAdi, :InsertTarihi,
       :KullaniciAdi, :DegisimTarihi do suspend;
end
