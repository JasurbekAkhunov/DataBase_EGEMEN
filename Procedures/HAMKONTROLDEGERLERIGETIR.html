CREATE PROCEDURE HAMKONTROLDEGERLERIGETIR(
    AHAMSTOKMASTERID INTEGER)
RETURNS (
    NEMORANI DOUBLE PRECISION,
    IGNEYOLU INTEGER,
    YIKAMAONCESIEN DOUBLE PRECISION,
    YIKAMAONCESIGRAMAJ DOUBLE PRECISION,
    YIKAMASONRASIEN DOUBLE PRECISION,
    YIKAMASONRASIGRAMAJ DOUBLE PRECISION,
    ENDENCEKMEZLIK DOUBLE PRECISION,
    BOYDANCEKMEZLIK DOUBLE PRECISION,
    HAMDONME DOUBLE PRECISION,
    YIKAMAONCESIRAPORBOYU VARCHAR(100),
    YIKAMASONRASIRAPORBOYU VARCHAR(100),
    YIKAMAONCESICPI VARCHAR(10),
    YIKAMASONRASICPI VARCHAR(10),
    YIKAMAONCESIWPI VARCHAR(10),
    YIKAMASONRASIWPI VARCHAR(10),
    YIKAMAHASLIGIYAPILDI INTEGER,
    HAMKONTROLIPLIKUZUNLUGU1 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU2 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU3 VARCHAR(40),
    HAMKONTROLIPLIKUZUNLUGU4 VARCHAR(40),
    MERSERIZEHASLIGIYAPILDI INTEGER,
    YIKAMAHASLIGIKABUL INTEGER,
    MERSERIZEHASLIGIKABUL INTEGER,
    CEKMEZLIKSTANDARDIID INTEGER,
    DEMIRTESTIYAPILDI INTEGER,
    DEMIRTESTIKABUL INTEGER,
    BOBINGRAMAJI NUMERIC(12,3),
    BOYAMAKABILIYETI VARCHAR(10),
    YIKAMAHASLIGI VARCHAR(10),
    NUMARATIPIID INTEGER,
    NUMARA VARCHAR(10),
    BUKUM VARCHAR(10),
    MUKAVEMET VARCHAR(10),
    UZAMA VARCHAR(10),
    UVKONTROL INTEGER,
    DUZGUNSUZLUKONAYI INTEGER,
    HAMKONTROLBUKUMBILDIRILEN INTEGER,
    HAMKONTROLBUKUMGERCEKLESEN INTEGER,
    HAMKONTROLKALINLIKBILDIRILEN VARCHAR(10),
    HAMKONTROLKALINLIKGERCEKLESEN VARCHAR(10),
    HAMKONTROLBIT INTEGER,
    HAMKONTROLNOPE INTEGER,
    HAMKONTROLJUT INTEGER,
    ICNEMORANI DOUBLE PRECISION,
    HAMKONTROLTOZUMA INTEGER,
    HAMKONTROLKOPUS INTEGER,
    HAMKONTROLBUKUMTURU INTEGER,
    HAMKONTROLYOGUNLUK DOUBLE PRECISION,
    HAMKONTROLKONIKRENKFARKLILIK INTEGER,
    HAMKONTROLKONTROLBOBINSAYISI INTEGER,
    HAMKONTROLKONTROLCUVALSAYISI INTEGER,
    KONTROLDURUMU INTEGER,
    KONTROLKULLANICIID INTEGER,
    KONTROLTARIHI TIMESTAMP,
    HAMKONTROLNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    CV VARCHAR(10),
    IGNESAYISI VARCHAR(20),
    HAMSTOKHAREKETTIPIID INTEGER,
    HAMSTOKHAREKETTIPIIDSTR VARCHAR(10),
    BRUTNET INTEGER,
    BRUTNETSTR VARCHAR(5),
    HAMKONTROLKULLANIMYERI INTEGER,
    HAMKONTROLKULLANIMYERISTR VARCHAR(20),
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    TAMIRMI VARCHAR(10),
    UCRETLITAMIR INTEGER,
    UCRETLITAMIRSTR VARCHAR(30))
AS
Declare variable MinUcretli integer;
Declare variable MaxUcretli integer;
Declare variable Tamir      integer;
begin
  select HamStokMaster.NemOrani, HamStokMaster.IgneYolu, HamStokMaster.YikamaOncesiEn, HamStokMaster.YikamaOncesiGramaj, HamStokMaster.YikamaSonrasiEn, HamStokMaster.YikamaSonrasiGramaj,
      HamStokMaster.EndenCekmezlik, HamStokMaster.BoydanCekmezlik, HamStokMaster.HamDonme, HamStokMaster.YikamaOncesiRaporBoyu, HamStokMaster.YikamaSonrasiRaporBoyu, HamStokMaster.YikamaOncesiCPI,
      HamStokMaster.YikamaSonrasiCPI, HamStokMaster.YikamaOncesiWPI, HamStokMaster.YikamaSonrasiWPI, HamStokMaster.YikamaHasligiYapildi, HamStokMaster.HamKontrolIplikUzunlugu1, HamStokMaster.HamKontrolIplikUzunlugu2,
      HamStokMaster.HamKontrolIplikUzunlugu3, HamStokMaster.HamKontrolIplikUzunlugu4, HamStokMaster.MerserizeHasligiYapildi, HamStokMaster.YikamaHasligiKabul, HamStokMaster.MerserizeHasligiKabul,
      HamStokMaster.CekmezlikStandardiId, HamStokMaster.DemirTestiYapildi, HamStokMaster.DemirTestiKabul, HamStokMaster.BobinGramaji, HamStokMaster.BoyamaKabiliyeti, HamStokMaster.YikamaHasligi, HamStokMaster.NumaraTipiId,
      HamStokMaster.Numara, HamStokMaster.Bukum, HamStokMaster.Mukavemet, HamStokMaster.Uzama, HamStokMaster.UVKontrol, HamStokMaster.DuzgunsuzlukOnayi, HamStokMaster.HamKontrolBukumBildirilen, HamStokMaster.HamKontrolBukumGerceklesen,
      HamStokMaster.HamKontrolKalinlikBildirilen, HamStokMaster.HamKontrolKalinlikGerceklesen, HamStokMaster.HamKontrolBit, HamStokMaster.HamKontrolNope, HamStokMaster.HamKontrolJut, HamStokMaster.IcNemOrani, HamStokMaster.HamKontrolTozuma,
      HamStokMaster.HamKontrolKopus, HamStokMaster.HamKontrolBukumTuru, HamStokMaster.HamKontrolKullanimYeri, HamStokMaster.HamKontrolYogunluk, HamStokMaster.HamKontrolKonikRenkFarklilik,
      HamStokMaster.HamKontrolKontrolBobinSayisi, HamStokMaster.HamKontrolKontrolCuvalSayisi, HamStokMaster.KontrolDurumu, HamStokMaster.KontrolKullaniciId, HamStokMaster.KontrolTarihi, HamStokMaster.HamKontrolNotu,
      HamStokMaster.CV, HamStokMaster.IgneSayisi, HamStokMaster.HamStokHareketTipiId, HamStokMaster.Brut, HamStokMaster.HamKontrolKullanimYeri, HamStokIrsaliye.EvrakTipi, HamStokMaster.Tamir from HamStokMaster
    left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
    where HamStokMaster.HamStokMasterId = :AHamStokMasterId
    into :NemOrani, :IgneYolu, :YikamaOncesiEn, :YikamaOncesiGramaj, :YikamaSonrasiEn, :YikamaSonrasiGramaj,
      :EndenCekmezlik, :BoydanCekmezlik, :HamDonme, :YikamaOncesiRaporBoyu, :YikamaSonrasiRaporBoyu, :YikamaOncesiCPI,
      :YikamaSonrasiCPI, :YikamaOncesiWPI, :YikamaSonrasiWPI, :YikamaHasligiYapildi, :HamKontrolIplikUzunlugu1, :HamKontrolIplikUzunlugu2,
      :HamKontrolIplikUzunlugu3, :HamKontrolIplikUzunlugu4, :MerserizeHasligiYapildi, :YikamaHasligiKabul, :MerserizeHasligiKabul,
      :CekmezlikStandardiId, :DemirTestiYapildi, :DemirTestiKabul, :BobinGramaji, :BoyamaKabiliyeti, :YikamaHasligi, :NumaraTipiId,
      :Numara, :Bukum, :Mukavemet, :Uzama, :UVKontrol, :DuzgunsuzlukOnayi, :HamKontrolBukumBildirilen, :HamKontrolBukumGerceklesen,
      :HamKontrolKalinlikBildirilen, :HamKontrolKalinlikGerceklesen, :HamKontrolBit, :HamKontrolNope, :HamKontrolJut, :IcNemOrani, :HamKontrolTozuma,
      :HamKontrolKopus, :HamKontrolBukumTuru, :HamKontrolKullanimYeri, :HamKontrolYogunluk, :HamKontrolKonikRenkFarklilik,
      :HamKontrolKontrolBobinSayisi, :HamKontrolKontrolCuvalSayisi, :KontrolDurumu, :KontrolKullaniciId, :KontrolTarihi, :HamKontrolNotu,
      :CV, :IgneSayisi, :HamStokHareketTipiId, :BrutNet, :HamKontrolKullanimYeri, :EvrakTipi, :Tamir;

    if (HamStokHareketTipiId = 0) then HamStokHareketTipiIdStr = 'Giriş';
    else if (HamStokHareketTipiId = 1) then HamStokHareketTipiIdStr = 'Satış İade';
    else HamStokHareketTipiIdStr = '';

    if (BrutNet = 0) then BrutNetStr = 'Net';
    else if (BrutNet = 1) then BrutNetStr = 'Brüt';
    else BrutNetStr = '';

    if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
    else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
    else EvrakTipiStr = 'Hatalı';

    if (HamKontrolKullanimYeri = 0) then HamKontrolKullanimYeriStr = 'Beyazlık';
    else if (HamKontrolKullanimYeri = 1) then HamKontrolKullanimYeriStr = 'Siyah';
    else if (HamKontrolKullanimYeri = 2) then HamKontrolKullanimYeriStr = 'Baskı Altı';
    else HamKontrolKullanimYeriStr = '';

    if (Tamir = 1) then
    begin
      Tamirmi = 'Tamir';
      MinUcretli = -1;
      MaxUcretli = -1;

      Select Min(Ucretli), Max(Ucretli) from DisTamirCozum
        left join DisTamirCozumSekli on DisTamirCozumSekli.DisTamirCozumSekliID = DisTamirCozum.DisTamirCozumSekliID
        where DisTamirCozum.HamStokMasterId = :AHamStokMasterId
        into :MinUcretli, :MaxUcretli;

      if (MinUcretli is not null) then
      begin
        if (MaxUcretli = 0) then
        begin
          UcretliTamir = 0;
          UcretliTamirStr = 'Ücretsiz';
        end
        else
        begin
          UcretliTamir = 1;
          UcretliTamirStr = '';
        end
      end
      else
      begin
        UcretliTamir = -1;
        UcretliTamirStr = 'Değerlendirme Yapılmadı';
      end
    end
    else
    begin
      Tamirmi = null;
      UcretliTamir = Null;
      UcretliTamirStr = Null;
    end

    Suspend;
end
