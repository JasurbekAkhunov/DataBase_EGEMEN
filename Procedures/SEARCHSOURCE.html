CREATE PROCEDURE SEARCHSOURCE(
    ATEXT VARCHAR(4000))
RETURNS (
    SOURCETYPE VARCHAR(20),
    SOURCENAME VARCHAR(50),
    SOURCE BLOB SUB_TYPE 0 SEGMENT SIZE 80)
AS
begin
  for select 'Procedure', RDB$procedure_name, RDB$procedure_source from RDB$PROCEDURES  where upper(RDB$procedure_source) like '%'||upper(:AText)||'%'
  union
  select 'Trigger', RDB$trigger_name, RDB$trigger_source from RDB$TRIGGERS  where upper(RDB$TRIGGER_source) like '%'||(:AText)||'%'
  into :SourceType, :SourceName, :Source do suspend;
end
