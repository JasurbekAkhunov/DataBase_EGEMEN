             :IadeStokReferansFirmaId, :IadeStokReferansFirmaAdi, :IadeStokMusteriSiparisNo, :IadeStokReferansSiparisNo, :IadeStokMusteriPartiNo,
             :HamStokMasterIdStr, :Pus, :Feine, :IadeStokOrmeciMakinaNo, :IadeStokLot, :En, :Gramaj, :TupAcikEnMayli;
    end
    else if (PartiSinifID is null) then
    begin
      Select HamStokMaster.MusteriSiparisNo, HamStokMaster.Lot, HamStokMaster.MusteriPartiNo, HamStokMaster.En, HamStokMaster.Grm2, HamStokMaster.TupAcikEnMayli,
      HamStokMaster.Pus, HamStokMaster.Feine, HamStokMaster.ReferansSiparisNo, HamStokMaster.MusteriPartiNo, HamStokMaster.Lot, SiparisProses.Adi
        from RafDetay
        left join HamStokDetay on HamStokDetay.HamStokDetayID = RafDetay.HamStokDetayId
        left join HamStokMaster on HamStokMaster.HamStokMasterID = HamStokDetay.HamStokMasterId
    left join SiparisProses on SiparisProses.SiparisProsesId = HamStokMaster.SiparisProsesId
        where RafDetay.RafDetayID = :RafDetayID
        into :HamStokMusteriSiparisNo, :GirisLot, :MusteriPartiNo, :En, :Gramaj, :TupAcikEnMayli, :Pus, :Feine, :ReferansSiparisNo, :HamStokMusteriPartiNo, :HamStokLot, :HamStokSiparisProsesAdi;
    end

    select TupAcikEnMayliStrGetir.TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikEnMayli) into :TupAcikEnMayliStr;

    Hacim = HacimEn * FHacimBoy * FHacimYukseklik;
    Hacim = (Hacim / 1000000) * KapAdedi;

    IcKontrolCikisEn = Null;
    IcKontrolCikisGr = Null;

    if (PartiSinifId is not null) then
    if (CikisEnIcKontrolTipiId is not null and CikisGrIcKontrolTipiId is not null) then
    begin
      MaxSira = Null;

      Select Max(PartiAsamalari.Sira) from PartiStok
        left join PartiAsamaStok on PartiAsamaStok.PartiSinifID = PartiStok.PartiSinifID
        left join PartiAsamalari on PartiAsamalari.PartiAsamalariID = PartiAsamaStok.PartiAsamalariID
        left join PartiAsamaIcKontrol on PartiAsamaIcKontrol.PartiAsamalariID = PartiAsamalari.PartiAsamalariId
        where PartiStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1 and PartiAsamalari.OtomasyonaDahilmi = 1 and
              PartiAsamaIcKontrol.IcKontrolTipiId = :CikisEnIcKontrolTipiId
        into :MaxSira;

      if (MaxSira is not null) then
      begin
        Select Max(Sonuc) from PartiAsamalari
          left join PartiAsamaIcKontrol on PartiAsamaIcKontrol.PartiAsamalariID = PartiAsamalari.PartiAsamalariId
          where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.Sira = :MaxSira and
                PartiAsamaIcKontrol.IcKontrolTipiId = :CikisEnIcKontrolTipiId and
                PartiAsamaIcKontrol.PartiSinifID = :PartiSinifID
          into :FSonuc;

        if (EgemenStrToIntOlurmu(FSonuc) = 1) then
        begin
          IcKontrolCikisEn = EgemenStrToInt(FSonuc);
          FSonuc = null;
          Select Max(Sonuc) from PartiAsamalari
            left join PartiAsamaIcKontrol on PartiAsamaIcKontrol.PartiAsamalariID = PartiAsamalari.PartiAsamalariId
            where PartiAsamalari.PartiKayitId = :PartiKayitId and PartiAsamalari.Sira = :MaxSira and
                  PartiAsamaIcKontrol.IcKontrolTipiId = :CikisGrIcKontrolTipiId and
                  PartiAsamaIcKontrol.PartiSinifID = :PartiSinifID
            into :FSonuc;

            if (EgemenStrToIntOlurmu(FSonuc) = 1) then IcKontrolCikisGr = EgemenStrToInt(FSonuc);
        end
      end
    end

    BrutSevkEdilenKilo = NUll;
    BrutSevkEdilenMetre = NUll;
    NetSevkEdilenKilo = NUll;
    NetSevkEdilenMetre = NUll;

    if (BoyaSiparisBaskiID is not null) then
      Select Sum(PartiStok.BrutSevkKilo), Sum(PartiStok.BrutSevkMetre), Sum(PartiStok.NetSevkKilo), Sum(PartiStok.NetSevkMetre) from BoyaSiparisBaski
        left join PartiStok on PartiStok.BoyaSiparisBaskiID = BoyaSiparisBaski.BoyaSiparisBaskiId
        where BoyaSiparisBaski.BoyaSiparisBaskiId = :BoyaSiparisBaskiId
          into :BrutSevkEdilenKilo, :BrutSevkEdilenMetre, :NetSevkEdilenKilo, :NetSevkEdilenMetre;
    else if (BoyaSiparisDetayID is not null) then
      Select Sum(PartiStok.BrutSevkKilo), Sum(PartiStok.BrutSevkMetre), Sum(PartiStok.NetSevkKilo), Sum(PartiStok.NetSevkMetre) from BoyaSiparisDetay
        left join PartiStok on PartiStok.BoyaSiparisDetayID = BoyaSiparisDetay.BoyaSiparisDetayId
        where BoyaSiparisDetay.BoyaSiparisDetayId = :BoyaSiparisDetayId
          into :BrutSevkEdilenKilo, :BrutSevkEdilenMetre, :NetSevkEdilenKilo, :NetSevkEdilenMetre;


    if (PartiSinifId is not Null) then
    begin
      HamGirisRenk = '';
      for Select HamStokMaster.Renk from PartiStokDetay
          Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
          Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
          where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.Renk is not null and HamStokMaster.Renk <> ''
          group by HamStokMaster.Renk
          into :FRenk do
      begin
        if (Char_Length(HamGirisRenk) + Char_Length(FRenk) < 255) then
        begin
          if (HamGirisRenk <> '' ) then HamGirisRenk = HamGirisRenk || cast(', ' as varchar(2) character set UTF8);
          HamGirisRenk = HamGirisRenk || FRenk;
        end
      end

      GirisEvrakTipiStr = '';
      for Select Case HamStokIrsaliye.EvrakTipi when 0 then 'Fi' when 1 then 'rsaliye' else '' end from PartiStokDetay
          Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
          Left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeID = HamStokDetay.HamStokIrsaliyeId
          where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokIrsaliye.EvrakTipi is not null
          group by HamStokIrsaliye.EvrakTipi
          into :FGirisEvrakTipiStr do
      begin
        if (GirisEvrakTipiStr <> '' ) then GirisEvrakTipiStr = GirisEvrakTipiStr || cast(', ' as varchar(2) character set UTF8);
        GirisEvrakTipiStr = GirisEvrakTipiStr || FGirisEvrakTipiStr;
      end

      HamStokMusteriSiparisNo = '';
      for Select HamStokMaster.MusteriSiparisNo from PartiStokDetay
          Left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
          Left join HamStokDetay on HamStokDetay.HamStokDetayId = RafDetay.HamStokDetayId
          Left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
          where PartiStokDetay.PartiSinifId = :PartiSinifId and HamStokMaster.MusteriSiparisNo is not null and HamStokMaster.MusteriSiparisNo <> ''
          group by HamStokMaster.MusteriSiparisNo
          into :FHamStokMusteriSiparisNo do
      begin
        if (Char_Length(HamStokMusteriSiparisNo) + Char_Length(FHamStokMusteriSiparisNo) < 200) then
        begin
          if (HamStokMusteriSiparisNo <> '' ) then HamStokMusteriSiparisNo = HamStokMusteriSiparisNo || cast(', ' as varchar(2) character set UTF8);
          HamStokMusteriSiparisNo = HamStokMusteriSiparisNo || FHamStokMusteriSiparisNo;
        end
      end
    end

    PartiTamirUAsamalar = '';

    for select PartiTamir.PartiTamirId from PartiTamirSinif
      left join PartiTamir on PartiTamir.PartiTamirID = PartiTamirSinif.PartiTamirID
      left join TamirCozumSekli on TamirCozumSekli.TamirCozumSekliid = Partitamir.TamirCozumSekliId
      where PartiTamirSinif.PartiSinifId = :PartiSinifId and TamirCozumSekli.Ucretli = 1
      into :FPartiTamirId do
    begin
      for select Asama.Adi from PartiAsamalari
        Left Join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
        left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
        left join Asama on Asama.AsamaId = coalesce(PartiAsamalari.AsamaId, Recete.UstAsamaID)
        where PartiAsamalari.PartiTamirId = :FPartiTamirId and Asama.ParasalDegeriVar = 1 and
              PartiAsamaStok.PartiSinifId = :PartiSinifId and PartiAsamaStok.AsamayaDahil = 1
        into :FAsamaAdi do
      begin
        if (Char_Length(PartiTamirUAsamalar) < 210) then
        begin
          if (PartiTamirUAsamalar <> '') then PartiTamirUAsamalar = PartiTamirUAsamalar || cast(', ' as varchar(2) character set UTF8);
          PartiTamirUAsamalar = PartiTamirUAsamalar || FAsamaAdi;
        end
      end
    end

    Makinalar = '';

    for select PartiAsamaStokDetay.MakinaID from PartiAsamaStokDetay
      left join partiasamastok on partiasamastok.partiasamalariid = partiasamastokdetay.partiasamalariid
      left join partiasamalari on partiasamalari.partiasamalariid = partiasamastokdetay.partiasamalariid
      where PartiAsamaStokDetay.PartiSinifId = :PartiSinifId and partiasamastok.Asamayadahil = 1  order by sira
      into :FMakina do
      begin
       if (Char_Length(Makinalar) < 240) then
        begin
          if (FMakina <> '') then Makinalar = Makinalar || cast(', ' as varchar(2) character set UTF8);
          Makinalar = Makinalar || FMakina;
        end
      end

    suspend;
  end
