CREATE PROCEDURE LABRECETEATISKIMYASALHESAPLA(
    ALABRECETEID INTEGER,
    ALABRECETECALISMAID INTEGER,
    ALABRECETEATISID INTEGER)
AS
declare variable FLabReceteAtisId integer;
declare variable FBoyaMiktari   Double Precision;
declare variable FMiktar        Double Precision;
declare variable FMiktar2 double precision;
declare variable FMiktar3 double precision;
declare variable FMiktar4 double precision;
declare variable FKimyasalTipiId    integer;

declare variable FReceteTuruId    integer;
declare variable FLabReceteUrunId integer;
declare variable KayitYapildi     integer;
declare variable FReceteAsamaId   integer;
declare variable BoyaGrubaGoreKimyasalHesapla integer;
declare variable BoyaGrupId       varchar(4);
declare variable FBoyaGrupId      varchar(4);
declare variable FMiktarYuzde     Double Precision;
declare variable AtisReceteAsamaKontroluYapilsin EgeBoolean;
declare variable FTuzSulfatBastan EgeBoolean;
declare variable FTuzKimyasalTipiId integer;
declare variable FSodaKimyasalTipiId integer;
declare variable FSulfatKimyasalTipiId integer;
declare variable FBoyaTipi integer;
declare variable FSiviKostikKimyasalTipiId integer;
declare variable FAlkaliKimyasalTipiId integer;

declare variable FLabReceteTuzVarmi integer;
declare variable FLabReceteSulfatVarmi integer;
declare variable FUrunBirimID integer;
declare variable FUrunId UrunKoduSize;
declare variable FReceteTanitimTuzVarmi integer;
declare variable FReceteTanitimSulfatVarmi integer;
declare variable FReceteTanitimSodaVarmi integer;
declare variable FLRDovizID varchar(5);
declare variable FBirimFiyat  Double Precision;

declare variable FSira integer;
begin
  BoyaGrupId = null;
  FBoyaGrupId = null;
  select DovizID from LabRecete where LabRecete.LabReceteID = :ALabReceteId into :FLRDovizID;

  select BoyaGrubaGoreKimyasalHesapla, AtisReceteAsamaKontroluYapilsin, TuzKimyasalTipiId, SodaKimyasalTipiId, SulfatKimyasalTipiId, SiviKostikKimyasalTipiId, AlkaliKimyasalTipiId  from sistem where sistemid = 1 into :BoyaGrubaGoreKimyasalHesapla, :AtisReceteAsamaKontroluYapilsin, :FTuzKimyasalTipiId, :FSodaKimyasalTipiId, :FSulfatKimyasalTipiId, :FSiviKostikKimyasalTipiId, :FAlkaliKimyasalTipiId;
   --exception KayitHatasi ALabReceteId||'-'||ALabReceteCalismaId||'-'||ALabReceteAtisId    ;

  select LabRecete.ReceteTuruId from LabRecete where LabReceteId = :ALabReceteId into :FReceteTuruId;

  if (FReceteTuruId is null) then exception ReceteninTuruBelliDegil;

  for select LabReceteAtisId, ReceteAsamaId from LabReceteAtis
    where LabReceteAtis.LabReceteId = :ALabReceteId and
          LabReceteAtis.LabReceteCalismaId = :ALabReceteCalismaId and
          (LabReceteAtis.LabReceteAtisId = :ALabReceteAtisId or :ALabReceteAtisId is null)
    into :FLabReceteAtisId, :FReceteAsamaId do
  begin
    FBoyaMiktari = null;

    select ReceteAsama.TuzSulfatBastan, ReceteAsama.BoyaTipiId  from ReceteAsama where ReceteAsama.ReceteAsamaId = :FReceteAsamaId into :FTuzSulfatBastan, :FBoyaTipi;

    select sum(LabReceteUrun.MiktarYuzde) from LabReceteUrun
    left join Urun on Urun.UrunID = LabReceteUrun.UrunId
    where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId and Urun.UrunTipiID = 0 and Urun.BoyaTipiID = 0
    into :FBoyaMiktari;

    if (FBoyaMiktari is null) then FBoyaMiktari = 0;

    if (AtisReceteAsamaKontroluYapilsin = 1 and FReceteAsamaId is null) then exception AtistaReceteAsamasiGirilmemis;

    if (BoyaGrubaGoreKimyasalHesapla = 1) then
    begin
      BoyaGrupId = null;
      FBoyaGrupId = null;

      select min(Urun.BoyaGrupId),  max(Urun.BoyaGrupId)  from LabReceteUrun
        left join Urun on Urun.UrunID = LabReceteUrun.UrunId
        where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId and Urun.UrunTipiId = 0 and Urun.BoyaTipiID = 0
        into :BoyaGrupId, :FBoyaGrupId;

      if (BoyaGrupId is null) then exception KayitHatasi 'Boya Grupları girilmemiş, kimyasal hesaplanamadı..';
      if (BoyaGrupId <> FBoyaGrupId) then
      begin
        select first 1 Urun.BoyaGrupId, sum(LabReceteUrun.MiktarYuzde) from LabReceteUrun
          left join Urun on Urun.UrunID = LabReceteUrun.UrunId
          where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId and Urun.UrunTipiId = 0 and Urun.BoyaTipiId = 0
          group by Urun.BoyaGrupId
          order by 2 desc
          into :BoyaGrupId, :FMiktarYuzde;
      end
    end


    for select LabReceteUrun.LabReceteUrunId, Urun.KimyasalTipiId from LabReceteUrun
      left join Urun on Urun.UrunId = LabReceteUrun.UrunId
      where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId and Urun.KimyasalTipiId is not null
      order by Sira
      into :FLabReceteUrunId, :FKimyasalTipiId do
    begin
      if (BoyaGrubaGoreKimyasalHesapla = 1 and FReceteAsamaId is null) then exception AtistaReceteAsamasiGirilmemis;

      update LabReceteUrun set MiktarYuzde = 0 where LabReceteUrun.LabReceteUrunId = :FLabReceteUrunId;
      KayitYapildi = 0;
      if (BoyaGrubaGoreKimyasalHesapla = 1) then
      begin
        for select Miktar from ReceteTuruBoyaGrupKimyasal
          where ReceteTuruBoyaGrupKimyasal.ReceteTuruId = :FReceteTuruId and ReceteTuruBoyaGrupKimyasal.ReceteAsamaId = :FReceteAsamaId and ReceteTuruBoyaGrupKimyasal.KimyasalTipiId = :FKimyasalTipiId and
                ReceteTuruBoyaGrupKimyasal.BoyaGrupID = :BoyaGrupID and :FBoyaMiktari > BaslangicMiktar and :FBoyaMiktari <= BitisMiktar
          into :FMiktar do
        begin
          if (KayitYapildi = 0) then
          if (FMiktar is not null) then
          begin
            update LabReceteUrun set MiktarYuzde = :FMiktar where LabReceteUrun.LabReceteUrunId = :FLabReceteUrunId;
            KayitYapildi = 1;
          end
          else exception KayitHatasi 'Uygun Aralık Bulunamadığı için Kimyasallar Hesaplanamadı..1';
        end
      end
      else
      begin
        for select first 1 ReceteTuruKimyasal.Miktar from ReceteTuruKimyasal
          where ReceteTuruKimyasal.ReceteTuruId = :FReceteTuruId and (ReceteTuruKimyasal.ReceteAsamaId = :FReceteAsamaId or ReceteTuruKimyasal.ReceteAsamaId is null) and ReceteTuruKimyasal.KimyasalTipiId = :FKimyasalTipiId and
                :FBoyaMiktari > ReceteTuruKimyasal.BaslangicMiktar and :FBoyaMiktari <= ReceteTuruKimyasal.BitisMiktar
          order by ReceteTuruKimyasal.ReceteAsamaId nulls last
          into :FMiktar do
        begin
          if (KayitYapildi = 0) then
          if (FMiktar is not null) then
          begin
            update LabReceteUrun set MiktarYuzde = :FMiktar where LabReceteUrun.LabReceteUrunId = :FLabReceteUrunId;
            KayitYapildi = 1;
          end
          else exception KayitHatasi 'Uygun Aralık Bulunamadığı için Kimyasallar Hesaplanamadı..2';
        end
      end
    end

    if (FBoyaTipi=0) then /* pamuksa */
    begin
      --select count(*)from LabReceteUrun,  Urun where  Urun.UrunId = LabReceteUrun.UrunId and LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId and Urun.KimyasalTipiId = :FTuzKimyasalTipiId into :FLabReceteTuzVarmi;
      --select count(*)from LabReceteUrun,  Urun where  Urun.UrunId = LabReceteUrun.UrunId and LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId and Urun.KimyasalTipiId = :FSulfatKimyasalTipiId into :FLabReceteSulfatVarmi;

      --if (FLabReceteTuzVarmi=0 and FLabReceteSulfatVarmi=0) then
      --begin
        for select Urun.UrunId, Urun.KimyasalTipiId, UrunBirim.UrunBirimId, (UrunFiyat.UretimeCikisFiyati * (select DovizKuru from EnYakinDovizKuruGetir(UrunFiyat.UretimeCikisFiyatiDovizID, :FLRDovizID, current_date)))
              from  Urun
              left join UrunBirim on UrunBirim.UrunId = Urun.UrunId and UrunBirim.ReceteDefaultBirim = 1
              left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
              where  ((Urun.KimyasalTipiId = :FTuzKimyasalTipiId) or ((Urun.KimyasalTipiId = :FSulfatKimyasalTipiId)))
        into :FUrunId, :FKimyasalTipiId, :FUrunBirimId, :FBirimFiyat do
        begin

          if (BoyaGrubaGoreKimyasalHesapla = 1) then
          begin
          /* ReceteTuruBoyaGrupKimyasal tablosundansa*/
            for select Miktar,Miktar2,Miktar3,Miktar4 from ReceteTuruBoyaGrupKimyasal
                  where ReceteTuruBoyaGrupKimyasal.ReceteTuruId = :FReceteTuruId and ReceteTuruBoyaGrupKimyasal.ReceteAsamaId = :FReceteAsamaId and
                  ((ReceteTuruBoyaGrupKimyasal.KimyasalTipiId = :FKimyasalTipiId)) and
                  ReceteTuruBoyaGrupKimyasal.BoyaGrupID = :BoyaGrupID and :FBoyaMiktari > BaslangicMiktar and :FBoyaMiktari <= BitisMiktar
            into :FMiktar, :FMiktar2, :FMiktar3, :FMiktar4 do
            begin
              if ( (FMiktar > 0) or (FMiktar2 > 0) or (FMiktar3 > 0) or (FMiktar4 > 0)) then
                delete from LabReceteUrun where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId and LabReceteUrun.UrunId = :FUrunId;
              if (FTuzSulfatBastan=1) then
              begin
              /*ReceteTuruBoyaGrupKimyasal  tuz sülfat bastan*/
                if (FMiktar4>0) then
                begin
                  FSira = 0;
                  update LabReceteUrun set sira = sira + 1 where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId;
                  insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                  values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar4,:FUrunBirimId,null,1,1);
                end
                if (FMiktar3>0) then
                begin
                  FSira = 0;
                  update LabReceteUrun set sira = sira + 1 where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId;
                  insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                  values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar3,:FUrunBirimId, null, 1, 1);
                end
                if (FMiktar2>0) then
                begin
                  FSira = 0;
                  update LabReceteUrun set sira = sira + 1 where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId;
                  insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                  values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar2,:FUrunBirimId ,null, 1, 1);
                end
                if (FMiktar>0) then
                begin
                  FSira = 0;
                  update LabReceteUrun set sira = sira + 1 where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId;
                  insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                  values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar,:FUrunBirimId, null, 1, 1);
                end

              end
              else
              begin
              /*ReceteTuruBoyaGrupKimyasal  tuz sülfat sondan*/
                Select Max(LabReceteUrun.Sira) + 1 from LabReceteUrun where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId into :FSira;
                if (FMiktar>0) then
                begin
                  if (FSira is null) then FSira = 0; else FSira = FSira +1;
                  insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                  values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar,:FUrunBirimId, null, 1, 1);
                end
                if (FMiktar2>0) then
                begin
                  if (FSira is null) then FSira = 0; else FSira = FSira + 1;
                  insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                  values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar2,:FUrunBirimId,null,1,1);
                end
                if (FMiktar3>0) then
                begin
                  if (:FSira is null) then FSira = 0; else FSira = FSira + 1;
                  insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                  values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar3,:FUrunBirimId,null,1,1);
                end
                if (FMiktar4>0) then
                begin
                  if (:FSira is null) then FSira = 0; else FSira = FSira + 1;
                  insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                  values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar4,:FUrunBirimId,null,1,1);
                end
              end
            end
        end
        else
        begin
        /* ReceteTuruKimyasal tablosundansa*/
          for select Miktar,Miktar2,Miktar3,Miktar4 from ReceteTuruKimyasal
                where ReceteTuruKimyasal.ReceteTuruId = :FReceteTuruId and ReceteTuruKimyasal.ReceteAsamaId = :FReceteAsamaId and
               (ReceteTuruKimyasal.KimyasalTipiId = :FKimyasalTipiId)  and :FBoyaMiktari > BaslangicMiktar and :FBoyaMiktari <= BitisMiktar
           into :FMiktar, :FMiktar2, :FMiktar3, :FMiktar4 do
           begin
              if ( (FMiktar > 0) or (FMiktar2 > 0) or (FMiktar3 > 0) or (FMiktar4 > 0)) then
                delete from LabReceteUrun where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId and LabReceteUrun.UrunId = :FUrunId;
             if (FTuzSulfatBastan=1) then
             begin
               /* ReceteTuruKimyasal tablosundansa tuz sulfat bastarn*/
               if (FMiktar4>0) then
               begin
                 FSira = 0;
                 update LabReceteUrun set sira = sira + 1 where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId;
                 insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                 values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar4,:FUrunBirimId,null,1,1);
               end
               if (FMiktar3>0) then
               begin
                 FSira = 0;
                 update LabReceteUrun set sira = sira + 1 where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId;
                 insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                 values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar3,:FUrunBirimId,null,1,1);
               end
               if (FMiktar2>0) then
               begin
                 FSira = 0;
                 update LabReceteUrun set sira = sira + 1 where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId;
                 insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                 values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar2,:FUrunBirimId,null,1,1);
               end
               if (FMiktar>0) then
               begin
                 FSira = 0;
                 update LabReceteUrun set sira = sira + 1 where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId;
                 insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                 values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar,:FUrunBirimId,null,1,1);
               end
             end
             else
             begin
             /* ReceteTuruKimyasal tablosundansa tuz sulfat sonradan*/
               Select Max(LabReceteUrun.Sira) + 1 from LabReceteUrun where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId into :FSira;
               if (FMiktar>0) then
               begin
                 if (FSira is null) then FSira = 0; else FSira = FSira +1;
                 insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                 values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar,:FUrunBirimId,null,1,1);
               end
               if (FMiktar2>0) then
               begin
                 if (FSira is null) then FSira = 0; else FSira = FSira + 1;
                 insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                 values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar2,:FUrunBirimId,null,1,1);
               end
               if (FMiktar3>0) then
               begin
                 if (:FSira is null) then FSira = 0; else FSira = FSira + 1;
                 insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                 values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar3,:FUrunBirimId,null,1,1);
               end
               if (FMiktar4>0) then
               begin
                 if (:FSira is null) then FSira = 0; else FSira = FSira + 1;
                 insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
                 values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar4,:FUrunBirimId,null,1,1);
               end
             end
           end
         end
       end
        /*Soda*/
     select Urun.UrunId, Urun.KimyasalTipiId, UrunBirim.UrunBirimId, (UrunFiyat.UretimeCikisFiyati * (select DovizKuru from EnYakinDovizKuruGetir(UrunFiyat.UretimeCikisFiyatiDovizID, :FLRDovizID, current_date)))
            from  Urun
               left join UrunBirim on UrunBirim.UrunId = Urun.UrunId and UrunBirim.ReceteDefaultBirim = 1
              left join UrunFiyat on UrunFiyat.UrunId = Urun.UrunId
              where  (Urun.KimyasalTipiId = :FSodaKimyasalTipiId)
     into :FUrunId, :FKimyasalTipiId, :FUrunBirimId, :FBirimFiyat;

      if (BoyaGrubaGoreKimyasalHesapla = 1) then
      begin
        for select Miktar,Miktar2,Miktar3,Miktar4 from ReceteTuruBoyaGrupKimyasal
                   where ReceteTuruBoyaGrupKimyasal.ReceteTuruId = :FReceteTuruId and ReceteTuruBoyaGrupKimyasal.ReceteAsamaId = :FReceteAsamaId and
                   (ReceteTuruBoyaGrupKimyasal.KimyasalTipiId = :FSodaKimyasalTipiId)  and :FBoyaMiktari > BaslangicMiktar and :FBoyaMiktari <= BitisMiktar
        into :FMiktar, :FMiktar2, :FMiktar3, :FMiktar4 do
        begin
        if ( (FMiktar > 0) or (FMiktar2 > 0) or (FMiktar3 > 0) or (FMiktar4 > 0)) then
            delete from LabReceteUrun where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId and LabReceteUrun.UrunId = :FUrunId;
          Select Max(LabReceteUrun.Sira)  from LabReceteUrun where LabReceteUrun.LabReceteAtisId = :FLabReceteAtisId into :FSira;
          if (FMiktar>0) then
          begin
            if (:FSira is null) then FSira = 0; else FSira = FSira +1;
            insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
            values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar,:FUrunBirimId,null,1,1);
          end
          if (FMiktar2>0) then
          begin
            if (:FSira is null) then FSira = 0; else FSira = FSira +1;
            insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
            values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar2,:FUrunBirimId,null,1,1);
          end
          if (FMiktar3>0) then
          begin
            if (:FSira is null) then FSira = 0; else FSira = FSira +1;
            insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId,Notu, InsertKullaniciId, KullaniciId)
            values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar3,:FUrunBirimId,null,1,1);
          end
          if (FMiktar4>0) then
          begin
            if (:FSira is null) then FSira = 0; else FSira = FSira +1;
            insert into LabReceteUrun(LabReceteAtisId, LabReceteId, Sira, UrunId, MiktarYuzde, UrunBirimId, Notu, InsertKullaniciId, KullaniciId)
            values(:ALabReceteAtisId,:ALabReceteId,:FSira,:FUrunId,:FMiktar4,:FUrunBirimId,null,1,1);
          end
        end
      end
      else
      begin
        for select Miktar,Miktar2,Miktar3,Miktar4 from ReceteTuruKimyasal
                   where ReceteTuruKimyasal.ReceteTuruId = :FReceteTuruId and ReceteTuruKimyasal.ReceteAsamaId = :FReceteAsamaId and
                   (ReceteTuruKimyasal.KimyasalTipiId = :FSodaKimyasalTipiId)  and :FBoyaMiktari > BaslangicMiktar and :FBoyaMiktari <= BitisMiktar
