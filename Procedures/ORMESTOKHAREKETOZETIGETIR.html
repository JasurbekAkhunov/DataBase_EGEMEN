CREATE PROCEDURE ORMESTOKHAREKETOZETIGETIR(
    AHAMID INTEGER)
RETURNS (
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    DEVIR DOUBLE PRECISION,
    ALISTOPLAMI DOUBLE PRECISION,
    SATISTOPLAMI DOUBLE PRECISION,
    BRUTSATISTOPLAMI DOUBLE PRECISION,
    ALISIADETOPLAMI DOUBLE PRECISION,
    SATISIADETOPLAMI DOUBLE PRECISION,
    FASONGIRISTOPLAMI DOUBLE PRECISION,
    FASONGIRISIADETOPLAMI DOUBLE PRECISION,
    FASONURETIMECIKISTOPLAMI DOUBLE PRECISION,
    FASONURETIMIADETOPLAMI DOUBLE PRECISION,
    URETIMECIKISTOPLAMI DOUBLE PRECISION,
    URETIMIADETOPLAMI DOUBLE PRECISION,
    FASONURETIMTOPLAMI DOUBLE PRECISION,
    URETIMTOPLAMI DOUBLE PRECISION,
    SAKATURETIMTOPLAMI DOUBLE PRECISION,
    GELENREKLAMASYONTOPLAMI DOUBLE PRECISION,
    GIDENREKLAMASYONTOPLAMI DOUBLE PRECISION,
    STOKKAYDIRMAGIRISTOPLAMI DOUBLE PRECISION,
    STOKKAYDIRMACIKISTOPLAMI DOUBLE PRECISION,
    SARFTOPLAMI DOUBLE PRECISION,
    STOKKAPADEDI INTEGER,
    STOKMEVCUDU DOUBLE PRECISION,
    SAKATKAPADEDI INTEGER,
    SAKATSTOKMEVCUDU DOUBLE PRECISION,
    NETSTOKKAPADEDI INTEGER,
    NETSTOKMEVCUDU DOUBLE PRECISION,
    HAREKETTIPI INTEGER,
    SONGIRISTARIHI TIMESTAMP,
    SONGIRISIRSALIYENO IRSALIYENOSIZE,
    SONGIRISFATURANO FATURANOSIZE,
    SONGIRISFIRMAID VARCHAR(20),
    SONGIRISFIRMAADI VARCHAR(100),
    SONGIRISFIYATI DOUBLE PRECISION,
    SONGIRISDOVIZID VARCHAR(5),
    KULLANILACAKMIKTAR DOUBLE PRECISION)
AS
declare variable shtDevir integer;
declare variable shtAlis integer;
declare variable shtSatis integer;
declare variable shtAlisIade integer;
declare variable shtSatisIade integer;
declare variable shtFasonAlis integer;
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable shtFasonUretim integer;
declare variable shtUretim integer;
declare variable ShtKaydirmaCikis integer;
declare variable ShtKaydirmaGiris integer;
declare variable ShtSarf integer;
declare variable shtReklamasyonGiris integer;
declare variable shtReklamasyonCikis integer;

declare variable FHareketTipi     integer;
declare variable FMiktar        Double Precision;
declare variable FBrutMiktar    Double Precision;
declare variable FKapAdedi      integer;
declare variable FHamIade       integer;
declare variable FGirisCikis    integer;
declare variable FSakat         integer;
begin
  select shtDevir, shtAlis, shtSatis, ShtAlisIade, ShtSatisIade, ShtFasonAlis, shtFasonUretimeCikis,  shtUretimeCikis, shtFasonUretim,
         ShtUretim, ShtKaydirmaCikis, ShtKaydirmaGiris, ShtSarf, shtReklamasyonGiris, shtReklamasyonCikis from OrmeStokHareketTipiGetir
   into :shtDevir, :shtAlis, :shtSatis, :ShtAlisIade, :ShtSatisIade, :ShtFasonAlis, :shtFasonUretimeCikis,  :shtUretimeCikis,
        :shtFasonUretim, :ShtUretim, :ShtKaydirmaCikis, :ShtKaydirmaGiris, :ShtSarf, :shtReklamasyonGiris, :shtReklamasyonCikis;

  Select Ham.HamId, Ham.Adi from Ham where Ham.HamId = :AHamId into :HamId, :HamAdi;

  Devir                         = 0;
  AlisToplami                   = 0;
  SatisToplami                  = 0;
  BrutSatisToplami              = 0;
  AlisIadeToplami               = 0;
  SatisIadeToplami              = 0;
  FasonGirisToplami             = 0;
  FasonGirisIadeToplami         = 0;
  FasonUretimeCikisToplami      = 0;
  FasonUretimIadeToplami        = 0;
  UretimeCikisToplami           = 0;
  UretimIadeToplami             = 0;
  FasonUretimToplami            = 0;
  UretimToplami                 = 0;
  SakatUretimToplami            = 0;
  GelenReklamasyonToplami       = 0;
  GidenReklamasyonToplami       = 0;
  SarfToplami                   = 0;
  StokKaydirmaGirisToplami      = 0;
  StokKaydirmaCikisToplami      = 0;

  StokMevcudu                   = 0;
  SakatStokMevcudu              = 0;
  StokKapAdedi                  = 0;
  SakatKapAdedi                 = 0;
  NetStokKapAdedi               = 0;

  For select StokHareket.HareketTipi, StokHareket.GirisCikis, StokHareket.HamIade, StokHareket.Sakat,
      sum(StokHareket.Miktar), sum(StokHareket.BrutMiktar), sum(StokHareket.KapAdedi) from stokhareket
    where stokhareket.HamID = :AHamId
    group by StokHareket.HareketTipi, StokHareket.GirisCikis, StokHareket.HamIade, StokHareket.Sakat
    into :FHareketTipi, :FGirisCikis, :FHamIade, :FMiktar, :FBrutMiktar, :FKapAdedi, :FSakat do
  begin
    if ((FMiktar is Null or FMiktar = 0) and FBrutMiktar > 0) then FMiktar = FBrutMiktar;

    if (FHareketTipi = shtDevir) then Devir = Devir + FMiktar;
    else if (FHareketTipi = shtAlis) then AlisToplami = AlisToplami + FMiktar;
    else if (FHareketTipi = shtSatis and FHamIade = 0) then
    begin
      SatisToplami = SatisToplami + FMiktar;
      BrutSatisToplami = BrutSatisToplami + FBrutMiktar;
    end
    else if (FHareketTipi = shtSatis and FHamIade = 1) then FasonGirisIadeToplami = FasonGirisIadeToplami + FMiktar;
    else if (FHareketTipi = shtAlisIade) then AlisIadeToplami = AlisIadeToplami + FMiktar;
    else if (FHareketTipi = shtSatisIade) then SatisIadeToplami = SatisIadeToplami + FMiktar;
    else if (FHareketTipi = shtFasonAlis) then FasonGirisToplami = FasonGirisToplami + FMiktar;
    else if (FHareketTipi = shtSarf) then SarfToplami = SarfToplami + FMiktar;
    else if (FHareketTipi = shtFasonUretimeCikis) then FasonUretimeCikisToplami = FasonUretimeCikisToplami + FMiktar;
    else if (FHareketTipi = shtUretimeCikis) then UretimeCikisToplami = UretimeCikisToplami + FMiktar;
    else if (FHareketTipi = shtFasonUretim and FHamIade = 0) then FasonUretimToplami = FasonUretimToplami + FMiktar;
    else if (FHareketTipi = shtFasonUretim and FHamIade = 1) then FasonUretimIadeToplami = FasonUretimIadeToplami + FMiktar;
    else if (FHareketTipi = shtUretim and FHamIade = 0 and FSakat = 0) then UretimToplami = UretimToplami + FMiktar;
    else if (FHareketTipi = shtUretim and FHamIade = 1 and FSakat = 0) then UretimIadeToplami = UretimIadeToplami + FMiktar;
    else if (FHareketTipi = shtUretim and FSakat = 1) then
    begin
      SakatUretimToplami = SakatUretimToplami + FMiktar;
      SakatKapAdedi = SakatKapAdedi + :FKapAdedi;
    end
    else if (FHareketTipi = shtReklamasyonGiris) then GelenReklamasyonToplami = GelenReklamasyonToplami + FMiktar;
    else if (FHareketTipi = shtReklamasyonCikis) then GidenReklamasyonToplami = GidenReklamasyonToplami + FMiktar;
    else if (FHareketTipi = ShtKaydirmaCikis) then StokKaydirmaCikisToplami = StokKaydirmaCikisToplami + FMiktar;
    else if (FHareketTipi = ShtKaydirmaGiris) then StokKaydirmaGirisToplami = StokKaydirmaGirisToplami + FMiktar;

    if (FGirisCikis = 1) then
    begin
      StokMevcudu = StokMevcudu + FMiktar;
      StokKapAdedi = StokKapAdedi + FKapAdedi;
    end
    else if (FGirisCikis = -1) then
    begin
      StokMevcudu  = StokMevcudu - FMiktar;
      StokKapAdedi = StokKapAdedi - FKapAdedi;
    end
  end

  select SonAlisTarihi, SonAlisIrsaliyeNo, SonAlisFaturaNo, SonAlisFirmaId, SonAlisFirmaAdi, SonAlisFiyati,
         DovizId from OrmeHamStokSonAlisBilgisi(:AHamID, 0)
         into :SonGirisTarihi, :SonGirisIrsaliyeNo, :SonGirisFaturaNo, :SonGirisFirmaId, :SonGirisFirmaAdi,
              :SonGirisFiyati, :SonGirisDovizId;

  suspend;
end
