CREATE PROCEDURE REZERVEDILMISHAMSTOKLAR_1(
    ASIPARISTIPIID INTEGER,
    AONGORUONAY INTEGER,
    AIMALATTIPIID INTEGER,
    AISLEMTIPIID INTEGER)
RETURNS (
    BOYASIPARISREZERVID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    BOYASIPARISSARJID INTEGER,
    HAMSTOKMASTERID INTEGER,
    HAMSTOKDETAYID INTEGER,
    HAMSTOKIRSALIYEID INTEGER,
    LOT VARCHAR(150),
    IRSALIYENO IRSALIYENOSIZE,
    IRSALIYETARIHI TIMESTAMP,
    EVRAKTIPI INTEGER,
    EVRAKTIPISTR VARCHAR(8),
    ORMECIADI VARCHAR(30),
    ORMECIMAKINANO VARCHAR(10),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    REFERANSFIRMAID VARCHAR(20),
    REFERANSFIRMAADI VARCHAR(100),
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    KALANADET INTEGER,
    REZERVKILO NUMERIC(12,3),
    REZERVMETRE NUMERIC(12,3),
    REZERVADET INTEGER,
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    PARTILENENADET INTEGER,
    YARIMAMULPARTINOSTR VARCHAR(20),
    RAFID RAFKODUSIZE,
    RAFYARIMAMULSTR VARCHAR(5))
AS
declare variable YariMamul EgeBoolean;
declare variable YariMamulPartiYil integer;
declare variable YariMamulPartiId  integer;
begin
  For select BoyaSiparisRezerv.BoyaSiparisRezervId, BoyaSiparisRezerv.BoyaSiparisKayitId, BoyaSiparisRezerv.BoyaSiparisDetayId, BoyaSiparisRezerv.BoyaSiparisSarjId,
             BoyaSiparisRezerv.HamStokDetayId, HamStokDetay.HamStokMasterId, HamStokDetay.HamStokIrsaliyeId, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             Ormeci.Adi, HamStokMaster.OrmeciMakinaNo, HamStokMaster.Lot,
             BoyaSiparisRezerv.Kilo, BoyaSiparisRezerv.Metre, BoyaSiparisRezerv.Adet,
             BoyaSiparisRezerv.KullanilanKilo, BoyaSiparisRezerv.KullanilanMetre, BoyaSiparisRezerv.KullanilanAdet,
             HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo, HamStokMaster.ReferansFirmaId, ReferansFirma.Adi,
             Max(Raf.RafId), Max(Raf.YariMamul), Parti.PartiYil, Parti.PartiId from BoyaSiparisRezerv
   left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisRezerv.BoyaSiparisKayitId
   left join BoyaSiparisDetayOngoru on BoyaSiparisDetayOngoru.BoyaSiparisDetayId = BoyaSiparisRezerv.BoyaSiparisDetayId and BoyaSiparisDetayOngoru.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
   left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = BoyaSiparisRezerv.BoyaSiparisDetayId
   left join HamStokDetay on HamStokDetay.HamStokDetayId = BoyaSiparisRezerv.HamStokDetayId
   left join HamStokMaster on HamStokMaster.HamStokMasterId = HamStokDetay.HamStokMasterId
   left join HamstokIrsaliye on hamstokIrsaliye.HamStokIrsaliyeid = HamstokDetay.HamStokIrsaliyeid
   left join Ormeci on Ormeci.OrmeciId = HamstokIrsaliye.OrmeciId
   left join Firma ReferansFirma on ReferansFirma.FirmaId = HamStokMaster.ReferansFirmaId
   left join PartiStok on PartiStok.PartiSinifId = HamStokMaster.PartiSinifId
   left join Parti on Parti.PartiKayitID = PartiStok.PartiKayitID
   left join RafDetay on RafDetay.HamStokDetayId = HamStokDetay.HamStokDetayId
   left join Raf on Raf.RafID = RafDetay.RafId
   where BoyaSiparisRezerv.RezervVar = 1 and
         (BoyaSiparisDetay.SiparisTipiId = :ASiparisTipiId or :ASiparisTipiId is null) and
         (BoyaSiparisDetay.IslemTipiId = :AIslemTipiId or :AIslemTipiId is null) and
         ((:AImalatTipiID = 0 and BoyaSiparisDetay.DesenID is Null) or (:AImalatTipiID = 1 and BoyaSiparisDetay.DesenID is not Null) or  (:AImalatTipiID is Null)) and
         BoyaSiparisDetayOngoru.OnayDurumu = :AOngoruOnay
   group by
             BoyaSiparisRezerv.BoyaSiparisRezervId, BoyaSiparisRezerv.BoyaSiparisKayitId, BoyaSiparisRezerv.BoyaSiparisDetayId, BoyaSiparisRezerv.BoyaSiparisSarjId,
             BoyaSiparisRezerv.HamStokDetayId, HamStokDetay.HamStokMasterId, HamStokDetay.HamStokIrsaliyeId, HamStokIrsaliye.IrsaliyeNo, HamStokIrsaliye.IrsaliyeTarihi, HamStokIrsaliye.EvrakTipi,
             Ormeci.Adi, HamStokMaster.OrmeciMakinaNo, HamStokMaster.Lot,
             BoyaSiparisRezerv.Kilo, BoyaSiparisRezerv.Metre, BoyaSiparisRezerv.Adet,
             BoyaSiparisRezerv.KullanilanKilo, BoyaSiparisRezerv.KullanilanMetre, BoyaSiparisRezerv.KullanilanAdet,
             HamStokMaster.MusteriSiparisNo, HamStokMaster.ReferansSiparisNo, HamStokMaster.ReferansFirmaId, ReferansFirma.Adi,
             Parti.PartiYil, Parti.PartiId
   order by BoyaSiparisRezerv.BoyaSiparisKayitId, BoyaSiparisRezerv.BoyaSiparisDetayId
   into :BoyaSiparisRezervId, :BoyaSiparisKayitId, :BoyaSiparisDetayId, :BoyaSiparisSarjId, :HamStokDetayId, :HamStokMasterId, :HamStokIrsaliyeId,
        :IrsaliyeNo, :IrsaliyeTarihi, :EvrakTipi, :OrmeciAdi, :OrmeciMakinaNo, :Lot, :RezervKilo, :RezervMetre, :RezervAdet,
        :PartilenenKilo, :PartilenenMetre, :PartilenenAdet,
        :MusteriSiparisNo, :ReferansSiparisNo, :ReferansFirmaId, :ReferansFirmaAdi, :RafId, :YariMamul, :YariMamulPartiYil, :YariMamulPartiId do
   begin
     YariMamulPartiNoStr = '';
     if (YariMamulPartiYil is not null) then
       YariMamulPartiNoStr = '['||YariMamulPartiYil||') '||YariMamulPartiId;

     RafYariMamulStr = '';
     if (YariMamul = 1) then
       RafYariMamulStr = 'Evet';

     KalanKilo = RezervKilo - PartilenenKilo;
     KalanMetre = RezervMetre - PartilenenMetre;
     KalanAdet = RezervAdet - PartilenenAdet;

     if (EvrakTipi = 0) then EvrakTipiStr = 'Fiş';
     else if (EvrakTipi = 1) then EvrakTipiStr = 'İrsaliye';
     else EvrakTipiStr = 'Hatalı';
     suspend;
   end
end
