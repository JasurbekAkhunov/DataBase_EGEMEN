CREATE PROCEDURE BOYASTOKENVANTERRAPORU_4(
    ABASLANGICGRUP1ID VARCHAR(4),
    ABITISGRUP1ID VARCHAR(4),
    ABASLANGICGRUP2ID VARCHAR(4),
    ABITISGRUP2ID VARCHAR(4),
    ABASLANGICGRUP3ID VARCHAR(4),
    ABITISGRUP3ID VARCHAR(4),
    ABASLANGICGRUP4ID VARCHAR(4),
    ABITISGRUP4ID VARCHAR(4),
    AHAMGRUP1ARALIKLI INTEGER,
    AHAMGRUP2ARALIKLI INTEGER,
    AHAMGRUP3ARALIKLI INTEGER,
    AHAMGRUP4ARALIKLI INTEGER,
    AKUMASCINSIID VARCHAR(15),
    AIPLIKCINSIID VARCHAR(15),
    AKALINLIKID VARCHAR(15),
    AKAPALI INTEGER,
    AHSTIPLIK INTEGER,
    AHSTORMEKUMAS INTEGER,
    AHSTDOKUMAKUMAS INTEGER,
    AHSTRIBANA INTEGER,
    AHSTAKSESUAR INTEGER,
    AHSTELYAF INTEGER,
    AHSTLYCRA INTEGER,
    AHSTTELEF INTEGER,
    AFASON INTEGER,
    ARAPORBASLANGICTARIHI TIMESTAMP,
    ARAPORBITISTARIHI TIMESTAMP,
    ASIFIROLANLARDOKULSUNMU INTEGER,
    AHAMID INTEGER,
    ADEVIRICINGIRDIK INTEGER,
    AMALIYETLERIHESAPLA INTEGER,
    HAMID INTEGER,
    RENKDERINLIGIID INTEGER)
RETURNS (
    HAMSATISIADE NUMERIC(12,3),
    RENKLISATISIADE NUMERIC(12,3),
    HAMSATISIADETL NUMERIC(12,3),
    RENKLISATISIADETL NUMERIC(12,3),
    HAMSATISIADEMETRETL NUMERIC(12,3),
    RENKLISATISIADEMETRETL NUMERIC(12,3),
    HAMSATISIADEMETRE NUMERIC(12,3),
    RENKLISATISIADEMETRE NUMERIC(12,3),
    DOVIZID VARCHAR(5))
AS
declare variable FIrsaliyeDovizId  varchar(5);
declare variable DovizKuru         Double Precision;
declare variable FMaliyet          Double Precision;
declare variable FMaliyetMetre     Double Precision;
begin
  Select SistemDovizId from Sistem where sistemid = 1 into :DovizId;

  HamSatisIade                  = 0;
  RenkliSatisIade               = 0;
  HamSatisIadeMetre                  = 0;
  RenkliSatisIadeMetre               = 0;
  HamSatisIadeTL                  = 0;
  RenkliSatisIadeTL               = 0;

  if (AFason is Null or AFason <> 1) then
  begin
    select sum(HamStokMaster.Kilo), sum(HamStokMaster.Metre) from HamStokMaster
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
      where HamStokIrsaliye.IrsaliyeTarihi >= f_stripTime(:ARaporBaslangicTarihi) and HamStokIrsaliye.IrsaliyeTarihi <= f_stripTime(:ARaporBitisTarihi) and
            HamStokIrsaliye.HamStokHareketTipiId = 1 and HamStokMaster.Renk is Null and HamStokMaster.HamId = :HamId
      into :HamSatisIade, :HamSatisIadeMetre;
    if (HamSatisIade is Null) then HamSatisIade = 0;
    if (HamSatisIadeMetre is Null) then HamSatisIadeMetre = 0;

    if (AMaliyetleriHesapla = 1) then
    begin
      HamSatisIadeTL = 0;
      HamSatisIadeMetreTL = 0;
      For Select HamStokIrsaliye.DovizId, Sum(HamStokMaster.Kilo * HamStokMaster.NetBirimFiyat), Sum(HamStokMaster.Metre * HamStokMaster.NetBirimFiyat) from HamStokMaster
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
        where HamStokIrsaliye.IrsaliyeTarihi >= f_stripTime(:ARaporBaslangicTarihi) and HamStokIrsaliye.IrsaliyeTarihi <= f_stripTime(:ARaporBitisTarihi) and
              HamStokIrsaliye.HamStokHareketTipiId = 1 and HamStokMaster.Renk is Null and HamStokMaster.HamId = :HamId
        Group By HamStokIrsaliye.DovizId
        into :FIrsaliyeDovizId, :FMaliyet, :FMaliyetMetre do
      begin
        if (FIrsaliyeDovizId <> DovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FIrsaliyeDovizId, :DovizId, Current_date) into :DovizKuru;
        else DovizKuru = 1;

        if (DovizKuru is null) then DovizKuru = 0;

        HamSatisIadeTL = HamSatisIadeTL + FMaliyet * DovizKuru;
        HamSatisIadeMetreTL = HamSatisIadeMetreTL + FMaliyetMetre * DovizKuru;
      end
    end
  end

  if (AFason is Null or AFason <> 1) then
  begin
    select sum(HamStokMaster.Kilo), sum(HamStokMaster.Metre) from HamStokMaster
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
      where HamStokIrsaliye.IrsaliyeTarihi >= f_stripTime(:ARaporBaslangicTarihi) and HamStokIrsaliye.IrsaliyeTarihi <= f_stripTime(:ARaporBitisTarihi) and
            HamStokIrsaliye.HamStokHareketTipiId = 1 and HamStokMaster.Renk is Not Null and HamStokMaster.HamId = :HamId
      into :RenkliSatisIade, :RenkliSatisIadeMetre;
    if (RenkliSatisIade is Null) then RenkliSatisIade = 0;
    if (RenkliSatisIadeMetre is Null) then RenkliSatisIadeMetre = 0;

    if (AMaliyetleriHesapla = 1) then
    begin
      RenkliSatisIadeTL = 0;
      RenkliSatisIadeMetreTL = 0;
      For Select HamStokIrsaliye.DovizId, Sum(HamStokMaster.Kilo * HamStokMaster.NetBirimFiyat), Sum(HamStokMaster.Metre * HamStokMaster.NetBirimFiyat) from HamStokMaster
      left join HamStokIrsaliye on HamStokIrsaliye.HamStokIrsaliyeId = HamStokMaster.HamStokIrsaliyeId
        where HamStokIrsaliye.IrsaliyeTarihi >= f_stripTime(:ARaporBaslangicTarihi) and HamStokIrsaliye.IrsaliyeTarihi <= f_stripTime(:ARaporBitisTarihi) and
              HamStokIrsaliye.HamStokHareketTipiId = 1 and HamStokMaster.Renk is Not Null and HamStokMaster.HamId = :HamId
        Group By HamStokIrsaliye.DovizId
        into :FIrsaliyeDovizId, :FMaliyet, :FMaliyetMetre do
      begin
        if (FIrsaliyeDovizId <> DovizId) then select DovizKuru from EnYakinDovizKuruGetir(:FIrsaliyeDovizId, :DovizId, Current_date) into :DovizKuru;
        else DovizKuru = 1;

        if (DovizKuru is null) then DovizKuru = 0;

        RenkliSatisIadeTL = RenkliSatisIadeTL + FMaliyet  * DovizKuru;
        RenkliSatisIadeMetreTL = RenkliSatisIadeMetreTL + FMaliyetMetre  * DovizKuru;
      end
    end
  end

  Suspend;
end
