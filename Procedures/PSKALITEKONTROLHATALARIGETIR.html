CREATE PROCEDURE PSKALITEKONTROLHATALARIGETIR(
    APARTIKAYITID INTEGER)
RETURNS (
    PSKALITEKONTROLASAMAID INTEGER,
    PSKALITEKONTROLHATALARIID INTEGER,
    PSKALITEKONTROLID INTEGER,
    HATATIPIID INTEGER,
    HATATIPIADI VARCHAR(20),
    HATATIPIDIGERADI1 VARCHAR(20),
    HATATIPIDIGERADI2 VARCHAR(20),
    ADET INTEGER,
    BASLANGICBITISMETRESTR VARCHAR(1024),
    TEYITGEREKTIRIR INTEGER,
    TEYITGEREKTIRIRSTR VARCHAR(5),
    HATAPUANI DOUBLE PRECISION,
    HATATIPIPUANI INTEGER,
    ESLESTIRMEKODU ADI10,
    INSERTKULLANICIADI VARCHAR(40),
    INSERTTARIHI TIMESTAMP,
    KULLANICIADI VARCHAR(40),
    DEGISIMTARIHI TIMESTAMP,
    TOPLAMHATAMETRESI INTEGER)
AS
declare variable NoktasalHata integer;
declare variable BaslangicMetre Double Precision;
declare variable BitisMetre     Double Precision;
declare variable HataYeri       integer;
declare variable KS integer;
begin
  For Select PSKaliteKontrol.PSKaliteKontrolAsamaId, PSKaliteKontrolHatalari.PSKaliteKontrolHatalariId, PSKaliteKontrol.PSKaliteKontrolId, PSKaliteKontrolHatalari.hataTipiId, HataTipi.Adi, HataTipi.DigerAdi1, HataTipi.DigerAdi2,
             PSKaliteKontrolHatalari.Adet, HataTipi.NoktasalHata, HataTipi.TeyitGerektirir, IK.Adi, PSKaliteKontrolHatalari.InsertTarihi, SK.Adi, PSKaliteKontrolHatalari.DegisimTarihi, HataTipi.HataTipiPuani,
             HataTipi.EslestirmeKodu from PSKaliteKontrol
    left join PSKaliteKontrolHatalari on PSKaliteKontrolHatalari.PSKaliteKontrolId  = PSKaliteKontrol.PSKaliteKontrolId
    left join HataTipi on HataTipi.HataTipiId  = PSKaliteKontrolHatalari.HataTipiId
    left join Kullanici IK on IK.KullaniciId  = PSKaliteKontrolHatalari.InsertKullaniciId
    left join Kullanici SK on SK.KullaniciId  = PSKaliteKontrolHatalari.KullaniciId
    where PSKaliteKontrol.PartiKayitId = :APartiKayitId
  into :PSKaliteKontrolAsamaId, :PSKaliteKontrolHatalariId, :PSKaliteKontrolId, :HataTipiId, :HataTipiAdi, :HataTipiDigerAdi1, :HataTipiDigerAdi2, :Adet,
       :NoktasalHata, :TeyitGerektirir, :InsertKullaniciAdi, :InsertTarihi,
       :KullaniciAdi, :DegisimTarihi, :HataTipiPuani, :EslestirmeKodu do
  begin
    if (TeyitGerektirir = 1) then TeyitGerektirirStr = 'Evet';
    else TeyitGerektirirStr = '';

    BaslangicBitisMetreStr = '';

    if (Adet > 0) then
    begin
      For Select PSKaliteKontrolHataMetre.HataYeri, PsKaliteKontrolHataMetre.BaslangicMetre, PsKaliteKontrolHataMetre.BitisMetre from PsKaliteKontrolHataMetre
        where PsKaliteKontrolHataMetre.PSKaliteKontrolId = :PSKaliteKontrolId and PsKaliteKontrolHataMetre.HataTipiId = :HataTipiId
        order by PsKaliteKontrolHataMetre.BaslangicMetre, PsKaliteKontrolHataMetre.BitisMetre
        into :HataYeri, :BaslangicMetre, :BitisMetre do
      begin
        BaslangicMetre = Yuvarla(BaslangicMetre, 100);
        BitisMetre = Yuvarla(BitisMetre, 100);

        if (Char_Length(BaslangicBitisMetreStr) < 900) then
        begin
          if (BaslangicBitisMetreStr <> '' ) then BaslangicBitisMetreStr = BaslangicBitisMetreStr || cast(', ' as varchar(2) character set UTF8);
            BaslangicBitisMetreStr = BaslangicBitisMetreStr ||  Round(:BaslangicMetre, 2);

          if (NoktasalHata = 0) then
            BaslangicBitisMetreStr = BaslangicBitisMetreStr || '-' || Round(:BitisMetre, 2);

          if (HataYeri = 1) then BaslangicBitisMetreStr = BaslangicBitisMetreStr || ' (L)';
          else if (HataYeri = 2) then BaslangicBitisMetreStr = BaslangicBitisMetreStr || ' (C)';
          else if (HataYeri = 3) then BaslangicBitisMetreStr = BaslangicBitisMetreStr || ' (R)';

        end
      end
    end

    if (NoktasalHata = 1) then
    begin
      ToplamHataMetresi = 0;

      for Select count(*) from PSKaliteKontrolHataMetre
        where PSKaliteKontrolHataMetre.PSKaliteKontrolId = :PSKaliteKontrolId and PSKaliteKontrolHataMetre.HataTipiId = :HataTipiId
        group by KusuratSil(PSKaliteKontrolHataMetre.BaslangicMetre, 1)
        into :KS do
      begin
        ToplamHataMetresi = ToplamHataMetresi + 1;
      end
    end
    else
    begin
      Select Sum(Yuvarla((PSKaliteKontrolHataMetre.BitisMetre - PSKaliteKontrolHataMetre.BaslangicMetre + 0.5), 1)) from PSKaliteKontrolHataMetre
        where PSKaliteKontrolHataMetre.PSKaliteKontrolId = :PSKaliteKontrolId and PSKaliteKontrolHataMetre.HataTipiId = :HataTipiId
        into :ToplamHataMetresi;
    end

    if (ToplamHataMetresi = 0) then ToplamHataMetresi = Null;

    Select Sum(HataPuani) from PSKaliteKontrolHataMetre
      where PsKaliteKontrolHataMetre.PSKaliteKontrolId = :PSKaliteKontrolId and PsKaliteKontrolHataMetre.HataTipiId = :HataTipiId
      into :HataPuani;

    suspend;
  end
end
