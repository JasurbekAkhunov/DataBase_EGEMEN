CREATE PROCEDURE DOKUMATALIMATRAPORURECETE(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    ASIPARISFIRMAID VARCHAR(20),
    ABASLANGICBOYASIPARISYIL INTEGER,
    ABASLANGICBOYASIPARISID INTEGER,
    ABITISBOYASIPARISYIL INTEGER,
    ABITISBOYASIPARISID INTEGER,
    ASIPARISGRUPID VARCHAR(4),
    ASIPARISGRUP2ID VARCHAR(4),
    AORMESIPARISSEKLIID INTEGER,
    APLANLAMAFIRMAID VARCHAR(20),
    APLANLAMAMAKINAID VARCHAR(10),
    AMAKINAGRUPID VARCHAR(4),
    ABITTI INTEGER,
    AKDVDAHIL INTEGER)
RETURNS (
    DOKUMAPLANLAMARECETEID INTEGER,
    DOKUMAPLANLAMAID INTEGER,
    HAMID INTEGER,
    HAMTIPIID INTEGER,
    HAMTIPIADI VARCHAR(20),
    HAMADI VARCHAR(255),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    URETICIID INTEGER,
    URETICIADI VARCHAR(40),
    LOT VARCHAR(30),
    RECETEFIREORANI DOUBLE PRECISION,
    KULLANIMORANI DOUBLE PRECISION,
    KULLANILACAKMIKTAR DOUBLE PRECISION,
    CIKISMIKTARI DOUBLE PRECISION,
    IADEMIKTARI DOUBLE PRECISION,
    ATKICOZGU SMALLINT,
    ATKICOZGUSTR VARCHAR(10),
    SIRANO SMALLINT,
    TAHMINIFIREORANI DOUBLE PRECISION,
    TELSAYISI INTEGER,
    UZAMA INTEGER,
    MMRACK INTEGER,
    TANSIYONAYARI DOUBLE PRECISION,
    DISKADI VARCHAR(100),
    DISKNO VARCHAR(100),
    ROTASYONU VARCHAR(255),
    TEMP VARCHAR(255),
    GAUGE VARCHAR(255))
AS
declare variable FTalimatReceteFireOrani Double Precision;
declare variable UrunHamId Integer;
declare variable OrmeTalimattaReceteFiresiYap Integer;
declare variable OtomatikUCikislardaReceteFire integer;
begin
  For select DokumaTalimatRaporu.DokumaPlanlamaId, DokumaTalimatRaporu.TahminiFireOrani, DokumaPlanlamaRecete.DokumaPlanlamaReceteId, DokumaPlanlamaRecete.HamId, Ham.HamTipiId, HamTipi.Adi,
             Ham.Adi, DokumaPlanlamaRecete.RenkId, Renk.Adi, DokumaPlanlamaRecete.Lot, DokumaPlanlamaRecete.KullanimOrani, DokumaPlanlamaRecete.CikisMiktari, DokumaPlanlamaRecete.IadeMiktari,
             DokumaTalimatRaporu.TalimatMiktari * DokumaPlanlamaRecete.KullanimOrani / 100, DokumaPlanlamaRecete.UreticiId,
             Uretici.Adi, DokumaPlanlamaRecete.TahminiFireOrani, DokumaPlanlamaRecete.AtkiCozgu, Decode(DokumaPlanlamaRecete.AtkiCozgu, 1, 'Atkı', 2, 'Çözgü', ''),
             DokumaPlanlamaRecete.SiraNo, DokumaPlanlamaRecete.TelSayisi, DokumaPlanlamaRecete.Uzama, DokumaPlanlamaRecete.MmRack,
             DokumaPlanlamaRecete.TansiyonAyari, DiskTanitim.Adi, DiskTanitim.DiskNo, DiskTanitim.Rotasyonu, DiskTanitim.Temp, DiskTanitim.Gauge
  from DokumaTalimatRaporu(:ARAPORTIPI, :ABaslangicTarihi, :ABitisTarihi, :ASiparisFirmaId, :ABaslangicBoyaSiparisYil, :ABaslangicBoyaSiparisId, :ABitisBoyaSiparisYil, :ABitisBoyaSiparisId, :ASiparisGrupId, :ASiparisGrup2Id, :AOrmeSiparisSekliId, :APlanlamaFirmaId, :APlanlamaMakinaId, :AMakinaGrupId, :ABitti, :AKdvDahil)
  left join DokumaPlanlamaRecete on DokumaPlanlamaRecete.DokumaPlanlamaId = DokumaTalimatRaporu.DokumaPlanlamaId
  left join Ham on Ham.HamId = DokumaPlanlamaRecete.HamId
  left join Renk on Renk.RenkId = DokumaPlanlamaRecete.RenkId
  left join HamTipi on HamTipi.HamTipiId = Ham.HamTipiId
  Left Join Kullanici IK on IK.KullaniciId = DokumaPlanlamaRecete.InsertKullaniciId
  Left Join Kullanici SK on SK.KullaniciId = DokumaPlanlamaRecete.KullaniciId
  left join Uretici on Uretici.UreticiId = DokumaPlanlamaRecete.UreticiId
  left join DiskTanitim on DiskTanitim.DiskTanitimId=DokumaPlanlamaRecete.DiskTanitimId
  where DokumaPlanlamaRecete.DokumaPlanlamaId is not null
  order by DokumaTalimatRaporu.DokumaPlanlamaId
  into :DokumaPlanlamaId, :TahminiFireOrani, :DokumaPlanlamaReceteId, :HamId, :hamTipiId, :HamTipiAdi,
       :HamAdi, :RenkId, :RenkAdi, :Lot, :KullanimOrani, :CikisMiktari, :IadeMiktari, :KullanilacakMiktar,
       :UreticiID, :UreticiAdi, :FTalimatReceteFireOrani, :AtkiCozgu, :AtkiCozguStr, :SiraNo, :TelSayisi, :Uzama, :MmRack, :TansiyonAyari, :DiskAdi, :DiskNo,
       :Rotasyonu, :Temp, :Gauge do
  begin

    if (HamTipiID = 0 or HamTipiID = 6) then
    begin
      select first 1 OrmeTalimattaReceteFiresiYap, OtomatikUCikislardaReceteFire from Sistem into :OrmeTalimattaReceteFiresiYap, :OtomatikUCikislardaReceteFire;
      select first 1 FireOrani from HamStokRecete where HamId = :UrunHamId and ReceteHamId = :HamId into :ReceteFireOrani;
      if (OtomatikUCikislardaReceteFire = 1) then
      begin
        KullanilacakMiktar = KullanilacakMiktar / ((100 - FTalimatReceteFireOrani) / 100);
        ReceteFireOrani    = FTalimatReceteFireOrani;
      end
      else if (OrmeTalimattaReceteFiresiYap <> 1) then
      begin
        KullanilacakMiktar = KullanilacakMiktar / ((100 - TahminiFireOrani) / 100);
        ReceteFireOrani    = TahminiFireOrani;
      end
      else
        KullanilacakMiktar = KullanilacakMiktar / ((100 - ReceteFireOrani) / 100);
    end
    suspend;
  end
end
