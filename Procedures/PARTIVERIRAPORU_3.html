CREATE PROCEDURE PARTIVERIRAPORU_3(
    APARTIYIL INTEGER,
    APARTIID INTEGER)
RETURNS (
    PARTIASAMASTOKDETAYID INTEGER,
    ARAVERMESEBEBIID INTEGER,
    ARAVERMESEBEBIADI VARCHAR(50),
    BASLAMATARIHI TIMESTAMP,
    BITISTARIHI TIMESTAMP)
AS
declare variable MakinaID     varchar(10);
declare variable GirisTarihi  TimeStamp;
declare variable CikisTarihi  TimeStamp;
begin
  for Select PartiAsamaStokDetayId, MakinaId, GirisTarihi, CikisTarihi from PartiVeriRaporu_1(:APartiYil, :APartiId) order by GirisTarihi
     into :PartiAsamaStokDetayId, :MakinaId, :GirisTarihi, :CikisTarihi do
  begin
    for select ImalataAraVerme.AraVermeSebebiId, AraVermeSebebi.Adi AraVermeSebebiAdi, MK.Tarih, Bmk.tarih  from MakinaDurumu MK
    left join Imalataaraverme on Imalataaraverme.BasladiMakinaDurumuId = MK.MakinaDurumuId
    left join AraVermeSebebi on AraVermeSebebi.AraVermeSebebiId = ImalataAraVerme.AraVermeSebebiId
    left join Makina on Makina.MakinaId = MK.MakinaId
    left join MakinaDurumu BMK on BMK.MakinaDurumuId = ImalataAraVerme.bittiMakinaDurumuId
    where MK.tarih >= :GirisTarihi and MK.tarih <= :CikisTarihi and MK.ImalataAraVerildi = 1 and
         (Mk.MakinaId = :MakinaId)
    into :AraVermeSebebiId, :AraVermeSebebiAdi, :BaslamaTarihi, :BitisTarihi do suspend;
  end
end
