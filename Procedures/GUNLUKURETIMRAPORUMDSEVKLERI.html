CREATE PROCEDURE GUNLUKURETIMRAPORUMDSEVKLERI(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AAYLIKBASLANGIC TIMESTAMP,
    AAYLIKBITIS TIMESTAMP)
RETURNS (
    UNITEID INTEGER,
    UNITEADI VARCHAR(30),
    FASONIPLIKSEVKTOPLAMI NUMERIC(12,3),
    SATISIPLIKSEVKTOPLAMI NUMERIC(12,3),
    FASONKUMASSEVKTOPLAMI NUMERIC(12,3),
    SATISKUMASSEVKTOPLAMI NUMERIC(12,3),
    UCRETLITAMIRSEVKTOPLAMI NUMERIC(12,3),
    UCRETSIZTAMIRSEVKTOPLAMI NUMERIC(12,3),
    AYLIKFASONIPLIKSEVKTOPLAMI NUMERIC(12,3),
    AYLIKSATISIPLIKSEVKTOPLAMI NUMERIC(12,3),
    AYLIKFASONKUMASSEVKTOPLAMI NUMERIC(12,3),
    AYLIKSATISKUMASSEVKTOPLAMI NUMERIC(12,3),
    AYLIKUCRETLITAMIRSEVKTOPLAMI NUMERIC(12,3),
    AYLIKUCRETSIZTAMIRSEVKTOPLAMI NUMERIC(12,3))
AS
declare variable FPartiTipiId       integer;
declare variable FFason             integer;
declare variable FIslemTipiId       integer;
declare variable FFaturaKesilecek   integer;
declare variable FFisTarihi         TimeStamp;
declare variable FNetKilo           Numeric(12,3);
declare variable FUniteId           integer;
declare variable FLastUniteId       integer;
begin
  FasonIplikSevkToplami    = 0;
  SatisIplikSevkToplami    = 0;
  FasonKumasSevkToplami    = 0;
  SatisKumasSevkToplami    = 0;
  UcretliTamirSevkToplami  = 0;
  UcretsizTamirSevkToplami = 0;

  AylikFasonIplikSevkToplami    = 0;
  AylikSatisIplikSevkToplami    = 0;
  AylikFasonKumasSevkToplami    = 0;
  AylikSatisKumasSevkToplami    = 0;
  AylikUcretliTamirSevkToplami  = 0;
  AylikUcretsizTamirSevkToplami = 0;

  FLastUniteId = -1;

  For Select GD.UniteId, MamulDepoTransferFisi.FisTarihi, Parti.PartiTipiId, Parti.Fason, Parti.IslemTipiId, Parti.FaturaKesilecek,  Sum(PartiStokTarti.Kilo) from MamulDepoTransferFisi
    left join Depo GD on GD.DepoID = MamulDepoTransferFisi.DepoId
    left join Depo AD on AD.DepoID = MamulDepoTransferFisi.TransferDepoId
    left join MamulDepoTransferFisiDetay on MamulDepoTransferFisiDetay.MamulDepoTransferFisiId = MamulDepoTransferFisi.MamulDepoTransferFisiId
    left join PartiStokTarti on PartiStokTarti.PartiStokTartiId = MamulDepoTransferFisiDetay.PartiStokTartiId
    left join Parti on Parti.PartiKayitId = PartiStokTarti.PartiKayitId
    where MamulDepoTransferFisi.FisTarihi >= :AAylikBaslangic and MamulDepoTransferFisi.FisTarihi <= :AAylikBitis and MamulDepoTransferFisi.SistemKaydi = 0 and
          Gd.UniteId <> Ad.UniteId
    group by GD.UniteId, MamulDepoTransferFisi.FisTarihi, Parti.PartiTipiId, Parti.Fason, Parti.IslemTipiId, Parti.FaturaKesilecek
    order by GD.UniteId
    into :FUniteId, :FFisTarihi, :FPartiTipiId, :FFason, :FIslemTipiId, :FFaturaKesilecek, :FNetKilo do
  begin
    if (FUniteId <> FLastUniteId) then
    begin
      if (FLastUniteId <> -1) then suspend;
      FLastUniteId = FUniteId;
      UniteId = FUniteId;

      Select Adi from Unite where UniteId = :UniteId into :UniteAdi;

      FasonIplikSevkToplami    = 0;
      SatisIplikSevkToplami    = 0;
      FasonKumasSevkToplami    = 0;
      SatisKumasSevkToplami    = 0;
      UcretliTamirSevkToplami  = 0;
      UcretsizTamirSevkToplami = 0;

      AylikFasonIplikSevkToplami    = 0;
      AylikSatisIplikSevkToplami    = 0;
      AylikFasonKumasSevkToplami    = 0;
      AylikSatisKumasSevkToplami    = 0;
      AylikUcretliTamirSevkToplami  = 0;
      AylikUcretsizTamirSevkToplami = 0;
    end

    if (FFisTarihi >= ABaslangicTarihi and FFisTarihi <= ABitisTarihi) then
    begin
      if (FPartiTipiId = 0) then
      begin
        if (FIslemTipiID = 3) then
        begin
          if (FFaturaKesilecek = 1) then UcretliTamirSevkToplami = UcretliTamirSevkToplami + FNetKilo;
          else UcretsizTamirSevkToplami = UcretsizTamirSevkToplami + FNetKilo;
        end
        else
        begin
          if (FFason = 1) then FasonIplikSevkToplami = FasonIplikSevkToplami + FNetKilo;
          else SatisIplikSevkToplami = SatisIplikSevkToplami + FNetKilo;
        end
      end
      else
      begin
        if (FIslemTipiID = 3) then
        begin
          if (FFaturaKesilecek = 1) then UcretliTamirSevkToplami = UcretliTamirSevkToplami + FNetKilo;
          else UcretsizTamirSevkToplami = UcretsizTamirSevkToplami + FNetKilo;
        end
        else
        begin
          if (FFason = 1) then FasonKumasSevkToplami = FasonKumasSevkToplami + FNetKilo;
          else SatisKumasSevkToplami = SatisKumasSevkToplami + FNetKilo;
        end
      end
    end

    if (FPartiTipiId is not null) then
    begin
      if (FIslemTipiID = 3) then
      begin
        if (FFaturaKesilecek = 1) then AylikUcretliTamirSevkToplami = AylikUcretliTamirSevkToplami + FNetKilo;
        else AylikUcretsizTamirSevkToplami = AylikUcretsizTamirSevkToplami + FNetKilo;
      end
      else
      begin
        if (FFason = 1) then AylikFasonIplikSevkToplami = AylikFasonIplikSevkToplami + FNetKilo;
        else AylikSatisIplikSevkToplami = AylikSatisIplikSevkToplami + FNetKilo;
      end
    end
    else
    begin
      if (FIslemTipiID = 3) then
      begin
        if (FFaturaKesilecek = 1) then AylikUcretliTamirSevkToplami = AylikUcretliTamirSevkToplami + FNetKilo;
        else AylikUcretsizTamirSevkToplami = AylikUcretsizTamirSevkToplami + FNetKilo;
      end
      else
      begin
        if (FFason = 1) then AylikFasonKumasSevkToplami = AylikFasonKumasSevkToplami + FNetKilo;
        else AylikSatisKumasSevkToplami = AylikSatisKumasSevkToplami + FNetKilo;
      end
    end
  end
  if (FLastUniteId <> -1) then suspend;
end
