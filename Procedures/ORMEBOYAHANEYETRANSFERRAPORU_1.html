CREATE PROCEDURE ORMEBOYAHANEYETRANSFERRAPORU_1
RETURNS (
    IDSTR VARCHAR(20),
    EVRAKTIPI VARCHAR(40),
    ORMEFISID INTEGER,
    ORMEIRSALIYEID INTEGER,
    IPLIKIRSALIYEID INTEGER,
    FISIRSALIYENO IRSALIYENOSIZE,
    FISIRSALIYETARIHI TIMESTAMP,
    FIILISEVKTARIHI TIMESTAMP,
    SINIF VARCHAR(1),
    DOKUMAHAREKETI EGEBOOLEAN,
    SATISTIPI INTEGER)
AS
declare variable shtFasonUretimeCikis integer;
declare variable shtUretimeCikis integer;
declare variable uitBoya integer;
declare variable FirmaKodumuz    varchar(20);
declare variable shtSatis integer;
begin
  Select shtFasonUretimeCikis,  shtUretimeCikis from OrmeStokHareketTipiGetir into :shtFasonUretimeCikis, :shtUretimeCikis;
  Select shtSatis from IplikStokHareketTipiGetir into :shtSatis;

  Select uitBoya from UretimIslemTipiGetir into :uitBoya;
  Select FirmaKodumuz from Sistem where SistemID = 1 into :FirmaKodumuz;

  for select Distinct OrmeFis.Fisno, OrmeFis.FisTarihi, OrmeFis.OrmeFisId, OrmeFis.Sinif, OrmeFis.DokumaHareketi from OrmeFis
  left join StokHareket On StokHareket.OrmeFisID = OrmeFis.OrmeFisId
  where OrmeFis.UretimIslemTipiID = :uitBoya and OrmeFis.HareketTipi = :shtUretimeCikis and
        OrmeFis.BoyahaneTransferEdebilir = 1 and StokHareket.HamStokIrsaliyeID is null and StokHareket.StokHareketID is Not Null
  into :FisIrsaliyeNo, :FisIrsaliyeTarihi, :OrmeFisId, :Sinif, :DokumaHareketi do
  begin
    OrmeIrsaliyeId = Null;
    IplikIrsaliyeId = Null;
    FiiliSevkTarihi = Null;
    SatisTipi = Null;
    EvrakTipi = 'Merkez Fiş';
    if (DokumaHareketi = 1) then
      EvrakTipi = 'Dokuma '||EvrakTipi;
    else
      EvrakTipi = 'Örme '||EvrakTipi;
    IDStr = 'F' || EgemenIntegerToStr(ormeFisID);
    suspend;
  end

  for Select Distinct OrmeIrsaliye.IrsaliyeNo, OrmeIrsaliye.IrsaliyeTarihi, OrmeIrsaliye.OrmeIrsaliyeId, OrmeIrsaliye.FiiliSevkTarihi, OrmeIrsaliye.Sinif, OrmeIrsaliye.DokumaHareketi, OrmeIrsaliye.SatisTipi from OrmeIrsaliye
  left join StokHareket On StokHareket.OrmeIrsaliyeId = OrmeIrsaliye.OrmeIrsaliyeID
  where (OrmeIrsaliye.UretimIslemTipiID = :uitBoya and OrmeIrsaliye.HareketTipi = :shtFasonUretimeCikis) and
        OrmeIrsaliye.FirmaId = :FirmaKodumuz and StokHareket.HamStokIrsaliyeID is null and
        OrmeIrsaliye.BoyahaneTransferEdebilir = 1 and StokHareket.StokHareketID is Not Null
  into :FisIrsaliyeNo, :FisIrsaliyeTarihi, :OrmeIrsaliyeId, :FiiliSevkTarihi, :Sinif, :DokumaHareketi, :SatisTipi do
  begin
    OrmeFisId = Null;
    IplikIrsaliyeId = Null;
    EvrakTipi = 'Merkez İrsaliye';
    if (DokumaHareketi = 1) then
      EvrakTipi = 'Dokuma '||EvrakTipi;
    else
      EvrakTipi = 'Örme '||EvrakTipi;
    IDStr = 'I' || EgemenIntegerToStr(OrmeIrsaliyeId);
    suspend;
  end

  for select Distinct IplikIrsaliye.IrsaliyeNo, IplikIrsaliye.IrsaliyeTarihi, IplikIrsaliye.IplikIrsaliyeId, IplikIrsaliye.FiiliSevkTarihi, IplikIrsaliye.Sinif from IplikIrsaliye
  left join IplikIrsaliyeDetay On IplikIrsaliyeDetay.IplikIrsaliyeId = IplikIrsaliye.IplikIrsaliyeID
  where IplikIrsaliye.HareketTipi = :shtSatis and IplikIrsaliye.FirmaId = :FirmaKodumuz and IplikIrsaliyeDetay.HamStokIrsaliyeID is null and IplikIrsaliyeDetay.IplikIrsaliyeDetayId is not Null
  into :FisIrsaliyeNo, :FisIrsaliyeTarihi, :IplikIrsaliyeId, :FiiliSevkTarihi, :Sinif do
  begin
    OrmeFisId = Null;
    OrmeIrsaliyeId = Null;
    SatisTipi = Null;
    EvrakTipi = 'İplik İrsaliye';
    IDStr = 'II' || EgemenIntegerToStr(IplikIrsaliyeId);
    suspend;
  end
end
