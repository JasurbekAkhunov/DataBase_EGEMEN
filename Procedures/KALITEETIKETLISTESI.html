CREATE PROCEDURE KALITEETIKETLISTESI(
    AKALITELER VARCHAR(1024),
    AKOPYASAYISI INTEGER)
RETURNS (
    KALITEID VARCHAR(20),
    KALITEADI ADI255,
    PARTISINIFID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    HAMADI VARCHAR(255),
    HAMADI2 VARCHAR(255),
    HAMADI3 VARCHAR(255),
    HAMADI4 VARCHAR(255),
    KUMASCINSIADI VARCHAR(40),
    KUMASCINSIADI2 VARCHAR(60),
    RECETEKODU VARCHAR(20),
    RECETEADI VARCHAR(40),
    DESENKODU VARCHAR(20),
    DESENADI VARCHAR(50),
    DESENVARYANTADI VARCHAR(30),
    DENYE INTEGER,
    SIPARISPROSESID INTEGER,
    SIPARISPROSESADI VARCHAR(255),
    ENISTENEN VARCHAR(10),
    GRAMAJISTENEN VARCHAR(10),
    CEKMEZLIKENISTENEN DOUBLE PRECISION,
    CEKMEZLIKBOYISTENEN DOUBLE PRECISION,
    DONMEISTENEN VARCHAR(10),
    SANSISTENEN VARCHAR(10),
    PUS INTEGER,
    FEINE INTEGER,
    DINLENMISGRAMAJ INTEGER,
    IPLIKBOYU1 VARCHAR(20),
    IPLIKBOYU2 VARCHAR(20),
    IPLIKBOYU3 VARCHAR(20),
    IPLIKBOYU4 VARCHAR(20),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    ENVERILEN VARCHAR(10),
    GRAMAJVERILEN VARCHAR(10),
    DONMEVERILEN VARCHAR(10),
    CEKMEZLIKENVERILEN DOUBLE PRECISION,
    CEKMEZLIKBOYVERILEN DOUBLE PRECISION,
    CEKMEZLIKENVERILEN2 DOUBLE PRECISION,
    CEKMEZLIKBOYVERILEN2 DOUBLE PRECISION,
    ACIKLAMA VARCHAR(100),
    NOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    METRAJNUMUNESIVAR INTEGER,
    METRAJNUMUNESIVARSTR VARCHAR(3),
    YIKAMATALIMATIADI VARCHAR(60),
    AGARTMATALIMATIADI VARCHAR(60),
    KURUTMATALIMATIADI VARCHAR(60),
    SIKMATALIMATIADI VARCHAR(60),
    UTUTALIMATIADI VARCHAR(60),
    KURUTEMIZLEMETALIMATIADI VARCHAR(60),
    YIKAMATALIMATIRESIM BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    AGARTMATALIMATIRESIM BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KURUTMATALIMATIRESIM BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    SIKMATALIMATIRESIM BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    UTUTALIMATIRESIM BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KURUTEMIZLEMETALIMATIRESIM BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    KARTELASTOKMEVCUDU INTEGER,
    ILAVEISLEMLER VARCHAR(255),
    COMPOSITION VARCHAR(255))
AS
declare variable VirgulPos  integer;
declare variable Str        varchar(20);
declare variable Sayac      integer;
begin
  VirgulPos = PosBul(AKaliteler, ',');

  While (VirgulPos > 1) do
  begin
    KaliteID = EgemenCopyStr(AKaliteler, 1, VirgulPos -1);
    KaliteAdi             = Null;
    PartiSinifId          = Null;
    PartiYil              = Null;
    PartiId               = Null;
    PartiIdStr            = Null;
    HamAdi                = Null;
    HamAdi2               = Null;
    HamAdi3               = Null;
    HamAdi4               = Null;
    KumasCinsiAdi         = Null;
    KumasCinsiAdi2        = Null;
    ReceteKodu            = Null;
    ReceteAdi             = Null;
    DesenKodu             = Null;
    DesenAdi              = Null;
    DesenVaryantAdi       = Null;
    Denye                 = Null;
    SiparisProsesId       = Null;
    SiparisProsesAdi      = Null;
    EnIstenen             = Null;
    GramajIstenen         = Null;
    CekmezlikEnIstenen    = Null;
    CekmezlikBoyIstenen   = Null;
    DonmeIstenen          = Null;
    SansIstenen           = Null;
    Pus                   = Null;
    Feine                 = Null;
    DinlenmisGramaj       = Null;
    IplikBoyu1            = Null;
    IplikBoyu2            = Null;
    IplikBoyu3            = Null;
    IplikBoyu4            = Null;
    TupAcikEnMayli        = Null;
    TupAcikEnMayliStr     = Null;
    EnVerilen             = Null;
    GramajVerilen         = Null;
    DonmeVerilen          = Null;
    CekmezlikEnVerilen    = Null;
    CekmezlikBoyVerilen   = Null;
    CekmezlikEnVerilen2   = Null;
    CekmezlikBoyVerilen2  = Null;
    Aciklama              = Null;
    Notu                  = Null;
    MetrajNumunesiVar            = Null;
    MetrajNumunesiVarStr         = Null;
    YikamaTalimatiAdi            = Null;
    AgartmaTalimatiAdi           = Null;
    KurutmaTalimatiAdi           = Null;
    SikmaTalimatiAdi             = Null;
    UtuTalimatiAdi               = Null;
    KuruTemizlemeTalimatiAdi     = Null;
    KartelaStokMevcudu           = Null;
    YikamaTalimatiResim = Null;
    AgartmaTalimatiResim = Null;
    SikmaTalimatiResim = Null;
    KurutmaTalimatiResim = Null;
    UtuTalimatiResim = Null;
    KuruTemizlemeTalimatiResim = Null;
    IlaveIslemler = Null;
    Composition = Null;

    Select Kalite.KaliteAdi, Kalite.PartiSinifId, Kalite.PartiYil, Kalite.PartiId, Kalite.HamAdi, KumasCinsi.Adi, KumasCinsi.Adi2,
           Kalite.ReceteKodu, Kalite.ReceteAdi, Kalite.Denye, Kalite.SiparisProsesId, SiparisProses.Adi, Kalite.EnIstenen, Kalite.GramajIstenen, Kalite.CekmezlikEnIstenen,
           Kalite.CekmezlikBoyIstenen, Kalite.DonmeIstenen, Kalite.SansIstenen, Kalite.Pus, Kalite.Feine, Kalite.DinlenmisGramaj, Kalite.IplikBoyu1,
           Kalite.IplikBoyu2, Kalite.IplikBoyu3, Kalite.IplikBoyu4, Kalite.TupAcikEnMayli,
           PartiStokFizikKontrol.EnVerilen, PartiStokFizikKontrol.GramajVerilen, PartiStokFizikKontrol.DonmeVerilen,
           PartiStokFizikKontrol.CekmezlikEnVerilen, PartiStokFizikKontrol.CekmezlikBoyVerilen, PartiStokFizikKontrol.CekmezlikEnVerilen2,
           PartiStokFizikKontrol.CekmezlikBoyVerilen2, Kalite.DesenKodu, Kalite.DesenAdi, Kalite.DesenVaryantAdi,
           Kalite.Aciklama, Kalite.Notu, Kalite.MetrajNumunesiVar, YikamaTalimati.Adi, AgartmaTalimati.Adi,
           SikmaTalimati.Adi, KurutmaTalimati.Adi, UtuTalimati.Adi, KuruTemizlemeTalimati.Adi,
           Kalite.KartelaStokMevcudu, YikamaTalimati.Resim, AgartmaTalimati.Resim,
           SikmaTalimati.Resim, KurutmaTalimati.Resim, UtuTalimati.Resim, KuruTemizlemeTalimati.Resim, Kalite.IlaveIslemler, Kalite.Composition, Ham.Adi2, Ham.Adi3, Ham.Adi4 from Kalite
      left join PartiStok on PartiStok.PartiSinifId = Kalite.PartiSinifId
      left join SiparisProses on SiparisProses.SiparisProsesId = Kalite.SiparisProsesId
      left join PartiStokFizikKontrol on PartiStokFizikKontrol.PartiSinifId = PartiStok.PartiSinifId and PartiStokFizikKontrol.SonKayit = 1
      left join Ham on Ham.HamId = Kalite.HamId
      left join KumasCinsi on KumasCinsi.KumasCinsiId = Ham.KumasCinsiId
      left join YikamaTalimati on YikamaTalimati.YikamaTalimatiId = Kalite.YikamaTalimatiId
      left join AgartmaTalimati on AgartmaTalimati.AgartmaTalimatiId = Kalite.AgartmaTalimatiId
      left join SikmaTalimati on SikmaTalimati.SikmaTalimatiId = Kalite.SikmaTalimatiId
      left join KurutmaTalimati on KurutmaTalimati.KurutmaTalimatiId = Kalite.KurutmaTalimatiId
      left join UtuTalimati on UtuTalimati.UtuTalimatiId = Kalite.UtuTalimatiId
      left join KuruTemizlemeTalimati on KuruTemizlemeTalimati.KuruTemizlemeTalimatiId = Kalite.KuruTemizlemeTalimatiId
      where Kalite.KaliteID = :KaliteID
      into :KaliteAdi, :PartiSinifId, :PartiYil, :PartiId, :HamAdi, :KumasCinsiAdi, :KumasCinsiAdi2, :ReceteKodu, :ReceteAdi,
           :Denye, :SiparisProsesId, :SiparisProsesAdi, :EnIstenen, :GramajIstenen, :CekmezlikEnIstenen,
           :CekmezlikBoyIstenen, :DonmeIstenen, :SansIstenen, :Pus, :Feine, :DinlenmisGramaj, :IplikBoyu1,
           :IplikBoyu2, :IplikBoyu3, :IplikBoyu4, :TupAcikEnMayli, :EnVerilen, :GramajVerilen, :DonmeVerilen,
           :CekmezlikEnVerilen, :CekmezlikBoyVerilen, :CekmezlikEnVerilen2, :CekmezlikBoyVerilen2,
           :DesenKodu, :DesenAdi, :DesenVaryantAdi, Aciklama, :Notu,
           :MetrajNumunesiVar, :YikamaTalimatiAdi, :AgartmaTalimatiAdi, :SikmaTalimatiAdi, :KurutmaTalimatiAdi, :UtuTalimatiAdi,
           :KuruTemizlemeTalimatiAdi, :KartelaStokMevcudu, :YikamaTalimatiResim, :AgartmaTalimatiResim,
           :SikmaTalimatiResim, :KurutmaTalimatiResim, :UtuTalimatiResim, :KuruTemizlemeTalimatiResim, :IlaveIslemler, :Composition, :HamAdi2, :HamAdi3, :HamAdi4;

    if (PartiYil is not null) then PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';
    else PartiIdStr = '';

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    if (MetrajNumunesiVar = 1) then MetrajNumunesiVarStr = 'Var';
    else MetrajNumunesiVarStr = 'Yok';

    Sayac = 1;

    While (Sayac <= AKopyaSayisi) do
    begin
      suspend;
      Sayac = Sayac + 1;
    end

    if (Char_Length(AKaliteler) > VirgulPos) then AKaliteler = EgemenCopyStr(AKaliteler, VirgulPos + 1, Char_Length(AKaliteler) - VirgulPos);
    else AKaliteler = '';

    VirgulPos = PosBul(AKaliteler, ',');
  end
end
