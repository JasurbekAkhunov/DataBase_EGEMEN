CREATE PROCEDURE PARTIYIBASKASIPARISETASI(
    APARTIKAYITID INTEGER,
    ABOYASIPARISKAYITID INTEGER,
    ASINIFBOYASIPARISDETAYID VARCHAR(255),
    AKULLANICIID INTEGER)
AS
declare variable PartiFirmaId   Varchar(20);
declare variable SiparisFirmaId   Varchar(20);
declare variable PartiFason            integer;
declare variable SiparisFason          integer;

declare variable PartiTipiId      integer;
declare variable PartiSinifId     integer;
declare variable BoyaSiparisDetayId integer;
declare variable BoyaSiparisSarjId  integer;
declare variable BoyaSiparisBaskiId  integer;
declare variable VirgulPos          integer;
declare variable KayitSayisi        integer;
declare variable Pos          integer;
declare variable Str                varchar(255);
declare variable FMsg               varchar(255);
declare variable ImalatTipiId integer;
begin
  Select FirmaId, Fason, PartiTipiId, ImalatTipiId from Parti where Parti.PartiKayitId = :APartiKayitId into :PartiFirmaId, :PartiFason, :PartiTipiId, :ImalatTipiId;

  Select BoyaSiparis.FirmaId, BoyaSiparis.Fason from BoyaSiparis
    where BoyaSiparis.BoyaSiparisKayitId = :ABoyaSiparisKayitId
    into :SiparisFirmaId, :SiparisFason;

  if (PartiFason <> SiparisFason) then exception SatisTipiUymuyorAktarilamaz;

  if (PartiFason = 1) then
  begin
    if (PartiFirmaId <> SiparisFirmaId) then exception FasonFirmaUymuyorAktarilamaz;
  end

  Update Parti Set BoyaSiparisKayitId = :ABoyaSiparisKayitId, FirmaId = :SiparisFirmaId, LinkIslemi = 1 where Parti.PartiKayitId = :APartiKayitId;

/*sinifid=boyasiparisdetayid,sinifid=boyasiparisdetayid,sinifid=boyasiparisdetayid, seklinde gelecek*/

  VirgulPos = PosBul(ASinifBoyaSiparisDetayId, ',');

  While (VirgulPos > 1) do
  begin
    BoyaSiparisDetayId = Null;
    BoyaSiparisSarjId = Null;
    Str = EgemenCopyStr(ASinifBoyaSiparisDetayId, 1, VirgulPos -1);

    Pos = PosBul(Str, '=');
    if (Pos = 0) then exception ParametreHatasi;
    else
    begin
      PartiSinifId = EgemenCopyStr(Str, 1, Pos - 1);

      Select Count(*) from PartiStok where PartiStok.PartiKayitId = :APartiKayitId and PartiStok.PartiSinifId = :PartiSinifId into :KayitSayisi;

      if (KayitSayisi = 0) then exception ParametreHatasi;

      if (ImalatTipiId = 0) then
      begin
        if (PartiTipiId = 0) then BoyaSiparisSarjId = EgemenCopyStr(Str, Pos+ 1, Char_Length(Str) - Pos);
        else BoyaSiparisDetayId = EgemenCopyStr(Str, Pos+ 1, Char_Length(Str) - Pos);

        if (PartiTipiId = 0 and (BoyaSiparisSarjId is null or BoyaSiparisSarjId = 0)) then exception ParametreHatasi;
        if (PartiTipiId <> 0 and (BoyaSiparisDetayId is null or BoyaSiparisDetayId = 0)) then exception ParametreHatasi;

      end
      else
      begin
        BoyaSiparisBaskiId = EgemenCopyStr(Str, Pos+ 1, Char_Length(Str) - Pos);
        if (BoyaSiparisBaskiId is null or BoyaSiparisBaskiId = 0) then exception ParametreHatasi;
      end
    end

    execute Procedure PartiStokBSDetayIdDegistir(:PartiSinifId, :BoyaSiparisDetayId, :BoyaSiparisSarjId, :BoyaSiparisBaskiId, :AKullaniciId);

    if (Char_Length(ASinifBoyaSiparisDetayId) > VirgulPos) then ASinifBoyaSiparisDetayId = EgemenCopyStr(ASinifBoyaSiparisDetayId, VirgulPos + 1, Char_Length(ASinifBoyaSiparisDetayId) - VirgulPos);
    else ASinifBoyaSiparisDetayId = '';

    VirgulPos = PosBul(ASinifBoyaSiparisDetayId, ',');
  end
end
