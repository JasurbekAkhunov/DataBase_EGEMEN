CREATE PROCEDURE DESENISTEKSARFSTOKTANDUS(
    ADESENISTEKLERIID INTEGER)
AS
declare variable FUrunId                varchar(20);
declare variable FUrunBirimId           integer;
declare variable FStokTabanDovizId      varchar(5);
declare variable FStokId                integer;
declare variable FMiktar                Double Precision;
declare variable KayitYapildi           integer;
declare variable FHareketTipi           integer;
declare variable FFisSeriNo             varchar(2);
declare variable FFisNo                 integer;
declare variable FTarih                 TimeStamp;
declare variable FSablonCekimmi          integer;
declare variable FSablonSilmemi          integer;
declare variable utSablon               integer;
declare variable FDepoId                varchar(10);
declare variable FLastDepoId            varchar(10);
declare variable FTransferDepoId        varchar(10);
declare variable FSablonSilmeDesenIstekTipiId varchar(5);
declare variable FDesenIstekTipiId            varchar(5);
declare variable StoktanDusuldu integer;
declare variable IlkGun  timeStamp;
declare variable SonGun  timeStamp;
declare variable FTamir Smallint;
declare variable FKirik Smallint;

begin
  FHareketTipi = 5;  /* depo transfer */
  FTarih =  current_date;

  IlkGun = TimeStampToAyIlkGun(FTarih);
  SonGun = FTarih;


  Select DesenIstekTipi.DesenIstekTipiId, DesenIstekTipi.SablonCekimmi, DesenIstekleri.StoktanDusuldu, DesenIstekTipi.DepoID, DesenIstekTipi.Tamir, DesenIstekTipi.Kirik from DesenIstekleri
    left join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiId = DesenIstekleri.DesenIstekTipiId
    where DesenIstekleri.DesenIstekleriID = :ADesenIstekleriID
    into :FDesenIstekTipiId, :FSablonCekimmi, :StoktanDusuldu, :FTransferDepoId, :FTamir, :FKirik;

  if (StoktanDusuldu = 1) then exit;

  Select STOKTABANDOVIZID, SablonSilmeDesenIstekTipiId from SISTEM where SISTEMID = 1 into :FStokTabanDovizId, :FSablonSilmeDesenIstekTipiId;

  Select DepoTransferFisSeriNo from Kullanici where KullaniciId = 1 into :FFisSeriNo;

  if (FFisSeriNo is null) then FFisSeriNo = '';


  if (FSablonCekimmi <> 1 and FTamir <> 1 and FKirik <> 1 and FDesenIstekTipiId <> FSablonSilmeDesenIstekTipiId) then
  begin
    exit;
  end


  if (FTransferDepoId is null) then exit;

  KayitYapildi = 0;

  FLastDepoId = '';

  for select DesenIstekSarfiyat.DepoId, DesenIstekSarfiyat.UrunId, DesenIstekSarfiyat.UrunBirimId, DesenIstekSarfiyat.Miktar from DesenIstekSarfiyat
    left join Urun on Urun.UrunId = DesenIstekSarfiyat.UrunId
    where DesenIstekSarfiyat.DesenIstekleriId = :ADesenIstekleriId and Urun.UrunTipiID = 6
    order by DesenIstekSarfiyat.DepoId, DesenIstekSarfiyat.UrunId
    into :FDepoId, :FUrunId, :FUrunBirimId, :FMiktar do
  begin
    if (FLastDepoID <> FDepoId) then
    begin
      FStokID = Null;

      Select Max(StokID) from Stok
        where Stok.HareketTipi = :FHareketTipi and Stok.FisTarihi >= :IlkGun and Stok.FisTarihi <= :SonGun and
              Stok.DesenSablonHareketi = 1 and
              Stok.DepoID = :FDepoId and Stok.TransferDepoID = :FTransferDepoID
        into :FStokID;

      if (FStokID is null) then
      begin
        Select FisNo from FisNumaratorAL(:FFisSeriNo) into :FFisNo;

        FStokId = gen_id(Stok_Id_Gen, 1);

        insert into Stok(StokId, HareketTipi, DesenSablonHareketi, SeriNo, FisNo, FisTarihi, DepoId, TransferDepoId, DovizId, TabanDovizKuru1, TabanDovizKuru2)
                    values(:FStokId, :FHareketTipi, 1, :FFisSeriNo, :FFisNo, :FTarih, :FDepoId, :FTransferDepoId, :FStokTabanDovizId, 1, 1);
      end

      FLastDepoID = FDepoId;
    end

    insert into StokDetay(Tarih, HareketTipi, StokId, DepoId, TransferDepoId, UrunId, UrunBirimId, Miktar, DesenIstekleriId, Otomasyon)
                values(:FTarih, :FHareketTipi, :FStokId, :FDepoId, :FTransferDepoId, :FUrunId, :FUrunBirimId, :FMiktar, :ADesenIstekleriId, 1);
    KayitYapildi = 1;
  end

  if (KayitYapildi = 1) then
    Update DesenIstekleri Set StoktanDusuldu = 1 where DesenIstekleri.DesenIstekleriId = :ADesenIstekleriId;
end
