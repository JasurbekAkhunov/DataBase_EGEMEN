CREATE PROCEDURE PARTIENERJISARFIYATIHESAPLA2(
    APARTIKAYITID INTEGER,
    APARTIASAMALARIID INTEGER,
    AMAKINAID VARCHAR(10))
RETURNS (
    DOGALGAZSARFIYATI DOUBLE PRECISION,
    ELEKTIRIKSARFIYATI DOUBLE PRECISION,
    OKUNANSUSARFIYATI DOUBLE PRECISION)
AS
declare variable ToplamSarjKilo  Double Precision;
declare variable PartiSarjKilo  Double Precision;
declare variable MakinaSarjID    integer;
declare variable Dogalgaz        Double Precision;
declare variable Elektirik       Double Precision;
declare variable OkunanSu              Double Precision;
declare variable DogalGazSarfiyatCarpani Double Precision;
declare variable ElektrikSarfiyatCarpani Double Precision;
declare variable OkunanSuSarfiyatCarpani       Double Precision;
begin
  DogalgazSarfiyati = 0;
  ElektirikSarfiyati = 0;
  OkunanSuSarfiyati = 0;

  For Select PartiAsamaStokDetay.MakinaSarjID,  MakinaSarj.ToplamKilo, MakinaSarj.Dogalgaz, MakinaSarj.Elektirik, MakinaSarj.OkunanSu, Makina.DogalGazSarfiyatCarpani, Makina.ElektrikSarfiyatCarpani, Makina.OkunanSuSarfiyatCarpani  from PartiAsamaStokDetay
    left join MakinaSarj on MakinaSarj.MakinaSarjID = PartiAsamaStokDetay.MakinaSarjID
    left join Makina on Makina.MakinaID = PartiAsamaStokDetay.MakinaID
    where PartiAsamaStokDetay.PartiKayitId = :APartiKayitId and PartiAsamaStokDetay.PartiAsamalariID = :APartiAsamalariID and PartiAsamaStokDetay.MakinaID = :AMakinaID and
         (MakinaSarj.Dogalgaz > 0 or MakinaSarj.Elektirik > 0)
    group by PartiAsamaStokDetay.MakinaSarjID, MakinaSarj.ToplamKilo, MakinaSarj.Dogalgaz, MakinaSarj.Elektirik, MakinaSarj.OkunanSu, Makina.DogalGazSarfiyatCarpani, Makina.ElektrikSarfiyatCarpani, Makina.OkunanSuSarfiyatCarpani
    into :MakinaSarjID, :ToplamSarjKilo, :Dogalgaz, :Elektirik, :OkunanSu, :DogalGazSarfiyatCarpani, :ElektrikSarfiyatCarpani, :OkunanSuSarfiyatCarpani do
  begin
    Select Sum(PartiAsamaStokDetay.Kilo) from PartiAsamaStokDetay
      where PartiAsamaStokDetay.PartiKayitId = :APartiKayitId and PartiAsamaStokDetay.PartiAsamalariID = :APartiAsamalariID and  PartiAsamaStokDetay.MakinaSarjID = :MakinaSarjID
      into :PartiSarjKilo;

    if (ToplamSarjKilo <> 0) then
    begin
      DogalgazSarfiyati  = DogalgazSarfiyati + Dogalgaz * PartiSarjKilo / ToplamSarjKilo * DogalGazSarfiyatCarpani;
      ElektirikSarfiyati = ElektirikSarfiyati + Elektirik * PartiSarjKilo / ToplamSarjKilo * ElektrikSarfiyatCarpani;
      OkunanSuSarfiyati  = OkunanSuSarfiyati + OkunanSu * PartiSarjKilo / ToplamSarjKilo * OkunanSuSarfiyatCarpani;
    end
  end

  suspend;
end
