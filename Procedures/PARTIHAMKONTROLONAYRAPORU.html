CREATE PROCEDURE PARTIHAMKONTROLONAYRAPORU(
    ARAPORTIPI INTEGER,
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    PARTIKAYITID INTEGER,
    PARTIYIL INTEGER,
    PARTIID INTEGER,
    PARTIIDSTR VARCHAR(30),
    BOYASIPARISKAYITID INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    ISLEMTIPIID INTEGER,
    ISLEMTIPIADI VARCHAR(20),
    TERMINTARIHI TIMESTAMP,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    PARTISINIFID INTEGER,
    BOYASIPARISDETAYID INTEGER,
    HAMADITOPLAMI VARCHAR(255),
    RECETEID VARCHAR(20),
    RENKADI VARCHAR(40),
    FASON INTEGER,
    FASONSTR VARCHAR(5),
    KILO NUMERIC(12,3),
    HAMKONTROLNOTU BLOB SUB_TYPE 0 SEGMENT SIZE 80,
    HAMKONTROLDURUMU INTEGER,
    HAMKONTROLDURUMUSTR VARCHAR(5),
    HAMKONTROLKULLANICIID INTEGER,
    HAMKONTROLKULLANICIADI VARCHAR(40),
    HAMKONTROLTARIHI TIMESTAMP,
    HAMKONTROLONAYDURUMU INTEGER,
    HAMKONTROLONAYDURUMUSTR VARCHAR(5),
    HAMKONTROLONAYKULLANICIID INTEGER,
    HAMKONTROLONAYKULLANICIADI VARCHAR(40),
    HAMKONTROLONAYTARIHI TIMESTAMP)
AS
begin
  if (ARaporTipi = 0) then
  begin
    for select PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil,
              BoyaSiparis.BoyaSiparisId, Parti.IslemTipiId, IslemTipi.Adi, Parti.TerminTarihi, Parti.FirmaId, Firma.Adi,
              PartiStok.PartiSinifId, PartiStok.BoyaSiparisDetayId, PartiStok.HamAdiToplami, Parti.ReceteId, Recete.Adi,
              Parti.Fason, PartiStok.Kilo, PartiStok.HamKontrolNotu, PartiStok.HamKontrolDurumu, PartiStok.HamKontrolKullaniciId,
              HKK.Adi, PartiStok.HamKontrolTarihi, PartiStok.HamKontrolOnayDurumu, PartiStok.HamKontrolONayKullaniciId, HKOK.Adi,
              PartiStok.HamKontrolOnaytarihi from PartiStok
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
        left join Firma On Firma.FirmaId = Parti.FirmaId
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
        left join Recete on Recete.ReceteId = Parti.ReceteId
        left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
        left join Kullanici HKK On HKK.KullaniciId = PartiStok.HamKontrolKullaniciId
        left join Kullanici HKOK On HKOK.KullaniciId = PartiStok.HamKontrolOnayKullaniciId
        where PartiStok.Durumu < 8 and Parti.PartiTipiID <> 0 and PartiStok.HamKontrolOnayKullaniciId is null
      into :PartiKayitId, :PartiYil, :PartiId, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :IslemTipiId, :IslemTipiAdi, :TerminTarihi,
           :FirmaId, :FirmaAdi, :PartiSinifId, :BoyaSiparisDetayId, :HamAdiToplami, :ReceteId, :RenkAdi, :Fason, :Kilo,
           :HamKontrolNotu, :HamKontrolDurumu, :HamKontrolKullaniciId, :HamKontrolKullaniciAdi, :HamKontrolTarihi, :HamKontrolOnayDurumu,
           :HamKontrolONayKullaniciId, :HamKontrolONayKullaniciAdi, :HamKontrolOnaytarihi do
    begin
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      else BoyaSiparisIdStr = '';

      if (HamKontrolDurumu = 1) then HamKontrolDurumuStr = 'Kabul';
      else if (HamKontrolDurumu = 2) then HamKontrolDurumuStr = 'Ret';
      else HamKontrolDurumuStr = '';

      if (HamKontrolOnayDurumu = 1) then HamKontrolOnayDurumuStr = 'Kabul';
      else if (HamKontrolOnayDurumu = 2) then HamKontrolOnayDurumuStr = 'Ret';
      else HamKontrolOnayDurumuStr = '';
      suspend;
    end
  end
  else if (ARaporTipi = 1 or ARaporTipi = 2) then
  begin
    for select PartiStok.PartiKayitId, Parti.PartiYil, Parti.PartiId, Parti.BoyaSiparisKayitId, BoyaSiparis.BoyaSiparisYil,
              BoyaSiparis.BoyaSiparisId, Parti.IslemTipiId, IslemTipi.Adi, Parti.TerminTarihi, Parti.FirmaId, Firma.Adi,
              PartiStok.PartiSinifId, PartiStok.BoyaSiparisDetayId, PartiStok.HamAdiToplami, Parti.ReceteId, Recete.Adi,
              Parti.Fason, PartiStok.Kilo, PartiStok.HamKontrolNotu, PartiStok.HamKontrolDurumu, PartiStok.HamKontrolKullaniciId,
              HKK.Adi, PartiStok.HamKontrolTarihi, PartiStok.HamKontrolOnayDurumu, PartiStok.HamKontrolONayKullaniciId, HKOK.Adi,
              PartiStok.HamKontrolOnaytarihi from PartiStok
        left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
        left join Firma On Firma.FirmaId = Parti.FirmaId
        left join BoyaSiparisDetay on BoyaSiparisDetay.BoyaSiparisDetayId = PartiStok.BoyaSiparisDetayId
        left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = BoyaSiparisDetay.BoyaSiparisKayitId
        left join Recete on Recete.ReceteId = Parti.ReceteId
        left join IslemTipi On IslemTipi.IslemTipiId = Parti.IslemTipiId
        left join Kullanici HKK On HKK.KullaniciId = PartiStok.HamKontrolKullaniciId
        left join Kullanici HKOK On HKOK.KullaniciId = PartiStok.HamKontrolOnayKullaniciId
        where PartiStok.Durumu < 8 and Parti.PartiTipiID <> 0 and PartiStok.HamKontrolOnayKullaniciId is not null and PartiStok.HamKontrolOnayDurumu = :ARaporTipi and
              (PartiStok.HamKontrolOnayTarihi >= :ABaslangicTarihi and PartiStok.HamKontrolOnayTarihi <= :ABitisTarihi)
      into :PartiKayitId, :PartiYil, :PartiId, :BoyaSiparisKayitId, :BoyaSiparisYil, :BoyaSiparisId, :IslemTipiId, :IslemTipiAdi, :TerminTarihi,
           :FirmaId, :FirmaAdi, :PartiSinifId, :BoyaSiparisDetayId, :HamAdiToplami, :ReceteId, :RenkAdi, :Fason, :Kilo,
           :HamKontrolNotu, :HamKontrolDurumu, :HamKontrolKullaniciId, :HamKontrolKullaniciAdi, :HamKontrolTarihi, :HamKontrolOnayDurumu,
           :HamKontrolONayKullaniciId, :HamKontrolONayKullaniciAdi, :HamKontrolOnaytarihi do
    begin
      if (Fason = 1) then FasonStr = 'Fason';
      else FasonStr = '';

      PartiIdStr = '(' || EgemenIntegerToStr(PartiYil)  || ') ' || EgemenIntegerToStr(PartiId) || ' ';

      if (BoyaSiparisId is not null) then BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil)  || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';
      else BoyaSiparisIdStr = '';

      if (HamKontrolDurumu = 1) then HamKontrolDurumuStr = 'Kabul';
      else if (HamKontrolDurumu = 2) then HamKontrolDurumuStr = 'Ret';
      else HamKontrolDurumuStr = '';

      if (HamKontrolOnayDurumu = 1) then HamKontrolOnayDurumuStr = 'Kabul';
      else if (HamKontrolOnayDurumu = 2) then HamKontrolOnayDurumuStr = 'Ret';
      else HamKontrolOnayDurumuStr = '';
      suspend;
    end
  end
end
