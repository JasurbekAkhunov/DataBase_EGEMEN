CREATE PROCEDURE CSFIRMAYACIROEDILMISLER(
    ACS VARCHAR(1),
    AFIRMAID VARCHAR(20))
RETURNS (
    CIROBORDROID INTEGER,
    CIROTARIHI TIMESTAMP,
    CIRODOVIZKURU1 DOUBLE PRECISION,
    CIRODOVIZKURU2 DOUBLE PRECISION,
    CIRODOVIZKURUSTR VARCHAR(40),
    CEKSENETID INTEGER,
    GENELKAYITID INTEGER,
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    ACIKLAMA VARCHAR(40),
    VADETARIHI TIMESTAMP,
    MEBLAG DOUBLE PRECISION,
    DOVIZID VARCHAR(5),
    CEKSENETNO VARCHAR(30),
    HESAPNO VARCHAR(40),
    ILKBORCLU VARCHAR(80),
    KESIDEYERI VARCHAR(20),
    AITOLDUGUBANKAADI VARCHAR(40),
    CEKSENETGRUP1 VARCHAR(5),
    CEKSENETGRUP2 VARCHAR(5),
    CEKSENETGRUP3 VARCHAR(5),
    ISLEMDOVIZID VARCHAR(5),
    ISLEMDOVIZKURU1 DOUBLE PRECISION,
    ISLEMDOVIZKURU2 DOUBLE PRECISION,
    ISLEMDOVIZKURUSTR VARCHAR(40))
AS
declare variable CiroHareketDetayId Integer;
declare variable SonHareketDetayId Integer;
declare variable FirmaDovizId varchar(5);
begin
  Select DovizId from Firma where Firma.FirmaId = :AFirmaId into :FirmaDovizId;

  for select ceksenethareket.bordroid,ceksenethareket.Tarih, CekSenetHareketDetay.DovizKuru1, CekSenetHareketDetay.DovizKuru2, ceksenetdetay.ceksenetid, CekSenetDetay.GenelKayitId,
   CekSenet.FirmaId, Firma.Adi, CekSenetDetay.Aciklama, CekSenetDetay.VadeTarihi, CekSenetDetay.Meblag, CekSenetDetay.DovizId,
   CekSenetDetay.CekSenetNo, CekSenetDetay.HesapNo, CekSenetDetay.IlkBorclu, CekSenetDetay.KesideYeri, Banka.Adi,
   CekSenetDetay.CekSenetGrup1, CekSenetDetay.CekSenetGrup2,CekSenetDetay.CekSenetGrup3, CekSenetHareketDetay.IslemDovizID, CekSenetHareketDetay.IslemDovizKuru1, CekSenetHareketDetay.IslemDovizKuru2
   from ceksenethareket
  left join ceksenethareketdetay on ceksenethareketdetay.Bordrotipi = 1 and CekSenetHareketDetay.Bordroid = ceksenethareket.BordroId
                                                                        and CekSenetHareketDetay.CS = ceksenethareket.Cs
  left join ceksenetdetay on ceksenetdetay.genelkayitid = ceksenethareketdetay.genelkayitid and ceksenetdetay.SonHareketId = ceksenethareketdetay.cshareketdetayid
  left join CekSenet on CekSenet.CsBordroKayitId = CekSenetDetay.CsBordroKayitId
  left join Firma on Firma.FirmaId = CekSenet.FirmaId
  left join Banka on Banka.BankaId = CekSenetDetay.AitOlduguBankaId
  where ceksenethareket.FirmaId = :AFirmaId and ceksenethareket.Bordrotipi = 1 and ceksenethareket.cs = :ACs and CekSenetDetay.GenelKayitId is not null
  into :CiroBordroId, :CiroTarihi, :CiroDovizKuru1, :CiroDovizKuru2, :CekSenetId, :GenelKayitId, :FirmaId, :FirmaAdi, :Aciklama, :Vadetarihi,
       :Meblag, :DovizId, :CekSenetNo, :HesapNo, :IlkBorclu, :KesideYeri, :AitOlduguBankaAdi, :CekSenetGrup1, :CekSenetGrup2,:CekSenetGrup3, :IslemDovizID, :IslemDovizKuru1, :IslemDovizKuru2 do
  begin

    if (FirmaDovizId <> DovizId) then
    begin
      CiroDovizKuruStr = Format_Float('', CiroDovizKuru1) || ' ' || DovizId || '-' || Format_Float('', CiroDovizKuru2) || ' ' || FirmaDovizId;
    end
    else CiroDovizKuruStr = '';

    if (IslemDovizID <> DovizId) then
    begin
      IslemDovizKuruStr = Format_Float('', IslemDovizKuru1) || ' ' || IslemDovizId || '-' || Format_Float('', IslemDovizKuru2) || ' ' || DovizId;
    end
    else IslemDovizKuruStr = '';

    suspend;
  end
end
