CREATE PROCEDURE HAMSTOKGIRISGUNLUKRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    AAYBASLANGICTARIHI TIMESTAMP,
    ARAPORHAMTIPI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    HAMKILO NUMERIC(12,3),
    TAMIRKILO NUMERIC(12,3),
    TOPLAMKILO NUMERIC(12,3),
    HAMORAN NUMERIC(12,3),
    TAMIRORAN NUMERIC(12,3),
    AYHAMKILO NUMERIC(12,3),
    AYTAMIRKILO NUMERIC(12,3),
    AYTOPLAMKILO NUMERIC(12,3),
    AYHAMORAN NUMERIC(12,3),
    AYTAMIRORAN NUMERIC(12,3))
AS
declare variable FFirmaId  varchar(20);
declare variable FFirmaAdi varchar(100);
declare variable Tamir integer;
declare variable Kilo Numeric(12,3);
declare variable AyKilo Numeric(12,3);
declare variable SonFirmaId  varchar(20);
declare variable hstIplik        integer;
declare variable hstOrmeKumas    integer;
declare variable hstDokumaKumas  integer;
declare variable hstRibana       integer;
declare variable hstAksesuar     integer;
declare variable hstElyaf        integer;
declare variable hstTelef        integer;
begin
  Select hstIplik, hstOrmeKumas, hstDokumaKumas, hstRibana, hstAksesuar, hstElyaf, hstTelef from HamTipiGetir into :hstIplik, :hstOrmeKumas, :hstDokumaKumas, :hstRibana, :hstAksesuar, :hstElyaf, :hstTelef;

  SonFirmaId = '';

  for select HS1.FirmaId, Firma.Adi, HSD1.Tamir, sum(HSD1.Kilo),
    (select sum(HSD2.Kilo) from HamStokIrsaliye HS2
     left join HamStokMaster HSD2 on HSD2.HamStokIrsaliyeId = HS2.HamStokIrsaliyeId
     left join Ham Ham2 on Ham2.HamId = HSD2.HamId
     where HSD2.PartiSinifId is null and HS2.Irsaliyetarihi = :ABaslangicTarihi and HS2.Fason = 0 and HS2.FirmaId = HS1.FirmaId and HSD2.Tamir = HSD1.Tamir and
        ( ( (Ham2.HamTipiId = :hstIplik or Ham2.HamTipiId = :hstElyaf or Ham2.HamTipiId = :hstTelef) and :ARaporHamTipi = 0) or
          (Ham2.HamTipiId <> :hstIplik and Ham2.HamTipiId <> :hstElyaf  and Ham2.HamTipiId <> :hstTelef and :ARaporHamTipi = 1) or
          ( (Ham2.HamTipiId = :hstOrmeKumas or Ham2.HamTipiId = :hstRibana or Ham2.HamTipiId = :hstAksesuar) and :ARaporHamTipi = 2) or
          (Ham2.HamTipiId = :hstDokumaKumas and :ARaporHamTipi = 3) or :ARaporHamTipi is null)) from hamstokIrsaliye HS1
   left join firma on firma.firmaid = HS1.firmaid
   left join HamStokMaster HSD1 on HSD1.HamStokIrsaliyeId = HS1.HamStokIrsaliyeId
   left join Ham Ham1 on Ham1.HamId = HSD1.HamId
   where HSD1.PartiSinifId is null and HS1.Irsaliyetarihi >= :AAyBaslangicTarihi and HS1.Irsaliyetarihi <= :ABaslangicTarihi and HSD1.Kilo > 0 and HS1.fason = 0 and
        ( ( (Ham1.HamTipiId = :hstIplik or Ham1.HamTipiId = :hstElyaf or Ham1.HamTipiId = :hstTelef) and :ARaporHamTipi = 0) or
          (Ham1.HamTipiId <> :hstIplik and Ham1.HamTipiId <> :hstElyaf and Ham1.HamTipiId <> :hstTelef and :ARaporHamTipi = 1) or
          ( (Ham1.HamTipiId = :hstOrmeKumas or Ham1.HamTipiId = :hstRibana or Ham1.HamTipiId = :hstAksesuar) and :ARaporHamTipi = 2) or
          (Ham1.HamTipiId = :hstDokumaKumas and :ARaporHamTipi = 3) or :ARaporHamTipi is null)
   group by HS1.FirmaId, HSD1.Tamir, Firma.Adi
   order by HS1.FirmaId
    into :FFirmaId, :FFirmaAdi, :Tamir, :AyKilo, :Kilo do
  begin
    if (SonFirmaId <> FFirmaId) then
    begin
      if (SonFirmaId <> '') then
      begin
        ToplamKilo = HamKilo + TamirKilo;
        HamOran = 0;
        TamirOran = 0;
        if (ToplamKilo > 0) then
        begin
          HamOran = HamKilo * 100 / ToplamKilo;
          TamirOran = TamirKilo * 100 / ToplamKilo;
        end
        AyToplamKilo = AyHamKilo + AyTamirKilo;
        AyHamOran = 0;
        AyTamirOran = 0;
        if (AyToplamKilo > 0) then
        begin
          AyHamOran = AyHamKilo * 100 / AyToplamKilo;
          AyTamirOran = AyTamirKilo * 100 / AyToplamKilo;
        end
        if (ToplamKilo > 0) then suspend;
      end
      SonFirmaId = FFirmaId;
      FirmaId = FFirmaId;
      FirmaAdi = FFirmaAdi;
      HamKilo = 0;
      TamirKilo = 0;
      ToplamKilo = 0;
      HamOran = 0;
      TamirOran = 0;

      AyHamKilo = 0;
      AyTamirKilo = 0;
      AyToplamKilo = 0;
      AyHamOran = 0;
      AyTamirOran = 0;
    end

    if (Kilo is null) then Kilo = 0;
    if (AyKilo is null) then AyKilo = 0;

    if (Tamir = 0) then
    begin
      AyHamKilo = AyKilo;
      HamKilo = Kilo;
    end
    else
    begin
      TamirKilo = Kilo;
      AyTamirKilo = AyKilo;
    end
  end

  if (SonFirmaId <> '') then
  begin
    ToplamKilo = HamKilo + TamirKilo;
    HamOran = 0;
    TamirOran = 0;
    if (ToplamKilo > 0) then
    begin
      HamOran = HamKilo * 100 / ToplamKilo;
      TamirOran = TamirKilo * 100 / ToplamKilo;
    end
    AyToplamKilo = AyHamKilo + AyTamirKilo;
    AyHamOran = 0;
    AyTamirOran = 0;
    if (AyToplamKilo > 0) then
    begin
      AyHamOran = AyHamKilo * 100 / AyToplamKilo;
      AyTamirOran = AyTamirKilo * 100 / AyToplamKilo;
    end

    if (ToplamKilo > 0) then suspend;
  end
end
