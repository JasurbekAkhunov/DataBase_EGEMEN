CREATE PROCEDURE ORMETOPKES(
    AMAKINAID MAKINAKODUSIZE,
    ATARIH TIMESTAMP,
    AORMEPLANLAMAID INTEGER,
    AISLEM SMALLINT)
RETURNS (
    STATUS INTEGER,
    MESAJ VARCHAR(255),
    ORMEKALITEKONTROLID INTEGER,
    YAZICIADI VARCHAR(150),
    PERSONELID INTEGER,
    PERSONELADI VARCHAR(40),
    URETIMPARTINO VARCHAR(30),
    HAMADI ADI255,
    ORMETOPNO INTEGER,
    FIRMAADI ADI255,
    LOT VARCHAR(255),
    MIKTAR DOUBLE PRECISION,
    SIPARISGRUP4ADI VARCHAR(20),
    SIRKETADI VARCHAR(150))
AS
declare variable OrmeVardiyaId       integer;
declare variable FOrmeMakinaDurumuId integer;
declare variable KS                          integer;
declare variable FBeforeOrmeMakinaDurumId    integer;
declare variable EgemenFirmaId       integer;
declare variable ToplamTur           double precision;
declare variable BinTurdaOrulecekKg  double precision;
declare variable FMiktar             double precision;
declare variable OrmeTopKilosu double precision;
declare variable FLastTopTesmeTarihi timestamp;
declare variable Ffark double precision;
declare variable FOrmeMinImalatBitirmeSuresi integer;
declare variable MelanjKodu Type of Column OrmeSiparisDetay.MelanjKodu;
begin
  if (AMakinaId is Null) then
  begin
    Status = 0;
    Mesaj = 'Makina Kodu zorunludur!';
  end
  else
  begin
    Select YaziciAdi from OrmeMakina where OrmeMakinaID = :AMakinaId into :YaziciAdi;
  end

  Select EgemenFirmaID, SirketAdi, OrmeMinImalatBitirmeSuresi from Sistem where SistemID = 1 into :EgemenFirmaID, :SirketAdi, :FOrmeMinImalatBitirmeSuresi;

  if (ATarih is Null) then ATarih = Current_TimeStamp(0);
  if (AOrmePlanlamaID is Null) then
  begin
    Select Max(OrmePlanlama.OrmePlanlamaID) from OrmePlanlama
      where OrmePlanlama.OrmeMakinaId = :AMakinaId and
            OrmePlanlama.Bitti = 0 and
            OrmePlanlama.Sira = (Select Min(sira) from OrmePlanlama where OrmePlanlama.Bitti = 0 and OrmePlanlama.OrmeMakinaId = :AMakinaId )
    into :AOrmePlanlamaID;
  end

  Select Max(OrmeMakinaDurumuId) from OrmeMakinaDurumu where OrmeMakinaID = :AMakinaId and OrmePlanlamaID = :AOrmePlanlamaID and TopKesme = 1 into :FOrmeMakinaDurumuId;

  if (AIslem = 1) then
  begin
    Select Max(OrmeKaliteKontrol.InsertTarihi) from OrmeKaliteKontrol where OrmeKaliteKontrol.OrmePlanlamaid = :AOrmePlanlamaID  into :FLastTopTesmeTarihi;
    Ffark = ATarih - FLastTopTesmeTarihi;
    if ((Ffark*1440) < FOrmeMinImalatBitirmeSuresi) then exception YakinZamandaTopKesilmis;

    if (EgemenFirmaID <> 39 or EgemenFirmaID <> 100 or EgemenFirmaID = 122 or EgemenFirmaID = 131 or  EgemenFirmaID = 120 or  EgemenFirmaID = 132 or  EgemenFirmaID = 146) then --baxt ,Chust, Global, Ünver , Sadırlar , MarForMell, Özsar
      Select Sum(Tur) from OrmeMakinaDurumu where (OrmeMakinaDurumuId > :FOrmeMakinaDurumuId or :FOrmeMakinaDurumuId is Null) and OrmeMakinaID = :AMakinaId and OrmePlanlamaID = :AOrmePlanlamaID into :ToplamTur;

    insert into OrmeMakinaDurumu(OrmeMakinaId, OrmePlanlamaID, Tarih, Tur, TopKesme)
                          values(:AMakinaId, :AOrmePlanlamaID, :ATarih, 0, 1);
    OrmeKaliteKontrolId = Null;
    if (ToplamTur > 0 or (EgemenFirmaID = 39 or EgemenFirmaID = 100 or EgemenFirmaID = 122 or EgemenFirmaID = 131 or  EgemenFirmaID = 120 or  EgemenFirmaID = 132 or  EgemenFirmaID = 146)) then
    begin
      Select OrmePlanlama.BinTurdaOrulecekKg, OrmePlanlama.UretimPartiNo, Ham.Adi ,Firma.Adi, OrmePlanlama.Lot, SiparisGrup4.Adi, Kalite.OrmeTopKilosu, OrmeSiparisDetay.MelanjKodu from OrmePlanlama
        Left Join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
        Left Join Ham on Ham.HamId = OrmeSiparisDetay.HamId
        Left Join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
        Left Join Firma on Firma.FirmaId = BoyaSiparis.FirmaId
        Left Join SiparisGrup SiparisGrup4 on SiparisGrup4.SiparisGrupId = BoyaSiparis.SiparisGrup4Id
        left join Kalite on Kalite.KaliteId = OrmeSiparisDetay.KaliteId
      where OrmePlanlama.OrmePlanlamaID = :AOrmePlanlamaId
      into :BinTurdaOrulecekKg, :UretimPartiNo, :HamAdi , :FirmaAdi,:Lot, :SiparisGrup4Adi, :OrmeTopKilosu, :MelanjKodu;

      if (BinTurdaOrulecekKg > 0 or (EgemenFirmaID = 39 or EgemenFirmaID = 100 or EgemenFirmaID = 122 or  EgemenFirmaID = 131 or  EgemenFirmaID = 120 or  EgemenFirmaID = 132 or  EgemenFirmaID = 146)) then
      begin
        if (EgemenFirmaID = 132) then
          FMiktar = OrmeTopKilosu;
        else if (EgemenFirmaID = 39 or EgemenFirmaID = 100 or EgemenFirmaID = 122) then --baxt, ChustChust, Global
          FMiktar = 24.5;
         else if (EgemenFirmaID = 131 or  EgemenFirmaID = 120 or  EgemenFirmaID = 146) then --Ünver,Sadırlar, Özsar
          FMiktar = 0.01;
        else
          FMiktar = (ToplamTur / 1000.0) * BinTurdaOrulecekKg;

        OrmeKaliteKontrolId = gen_id(OrmeKaliteKontrol_id_gen, 1) ;

        Select OrmeSonVardiyaIdGetir.OrmeVardiyaId, OrmeVardiyaPersonel.PersonelId, Personel.Adi from OrmeSonVardiyaIdGetir(:AMakinaID, null)
          left join OrmeVardiyaPersonel on OrmeVardiyaPersonel.OrmeVardiyaId = OrmeSonVardiyaIdGetir.OrmeVardiyaId and OrmeVardiyaPersonel.OrmeMakinaId = :AMakinaID
          left join Personel on Personel.PersonelId = OrmeVardiyaPersonel.PersonelId
        into :OrmeVardiyaId, :PersonelId, :PersonelAdi;

        insert into OrmeKaliteKontrol(OrmeKaliteKontrolId, OrmePlanlamaId, OrmeVardiyaId, Miktar, RafID, MelanjKodu)
          values(:OrmeKaliteKontrolId, :AOrmePlanlamaID, :OrmeVardiyaId, :FMiktar, 'KK', :MelanjKodu);
        Miktar=:FMiktar;
        select  OrmeKaliteKontrol.TopNo from OrmeKaliteKontrol   where   OrmeKaliteKontrol.OrmeKaliteKontrolId =:OrmeKaliteKontrolId into :OrmeTopNo ;

      end
    end

    Status = 1;
    if (OrmeKaliteKontrolId is Null) then  Mesaj = 'Kayıt yapıldı (Üretim miktarı yok!)';
    else Mesaj = 'Kayıt yapıldı';

  end
  else
  begin
    Select OrmeSonVardiyaIdGetir.OrmeVardiyaId from OrmeSonVardiyaIdGetir(:AMakinaID, null) into :OrmeVardiyaId;

    Delete from OrmeKaliteKontrol where OrmeKaliteKontrol.OrmeKaliteKontrolID = (Select Max(OrmeKaliteKontrolID) from OrmeKaliteKontrol where OrmeVardiyaId = :OrmeVardiyaId);
    select count(MakinaDurumuOzet.OrmeMakinaDurumuID) from  MakinaDurumuOzet where  MakinaDurumuOzet.OrmeMakinaDurumuID = :FOrmeMakinaDurumuId into :KS     ;
    if (KS>0) then
    begin
     select max(OrmeMakinaDurumu.OrmeMakinaDurumuID) from OrmeMakinaDurumu where OrmeMakinaDurumu.ormemakinaid= :AMakinaId and OrmeMakinaDurumu.OrmeMakinaDurumuID < :FOrmeMakinaDurumuId and OrmeMakinaDurumu.OrmeMakinaDurumuID > :FOrmeMakinaDurumuId-5 into :FBeforeOrmeMakinaDurumId  ;
     update MakinaDurumuOzet set  MakinaDurumuOzet.OrmeMakinaDurumuID=:FBeforeOrmeMakinaDurumId where  MakinaDurumuOzet.OrmeMakinaDurumuID = :FOrmeMakinaDurumuId;
    end
    Delete from OrmeMakinaDurumu where OrmeMakinaDurumu.OrmeMakinaDurumuID = :FOrmeMakinaDurumuId;

    Status = 1;
    Mesaj = 'Son top kesme bilgisi iptal edildi.';
  end

  suspend;
end
