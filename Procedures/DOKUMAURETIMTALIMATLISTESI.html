CREATE PROCEDURE DOKUMAURETIMTALIMATLISTESI(
    ADOKUMAMAKINAID VARCHAR(20),
    AFIRMAID VARCHAR(20))
RETURNS (
    RESULTID VARCHAR(255),
    DOKUMAPLANLAMAID INTEGER,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    URETIMPARTINO VARCHAR(30),
    RENKID INTEGER,
    RENKADI VARCHAR(50),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    SIPARISMIKTARI NUMERIC(12,3),
    SIPARISTOPLAMDOKUNAN NUMERIC(12,3),
    SIPARISTOPLAMSAKATDOKUNAN NUMERIC(12,3),
    SIPARISDOKUNANMIKTAR NUMERIC(12,3),
    PLANLANANMIKTAR NUMERIC(12,3),
    GERCEKLESENMIKTAR NUMERIC(12,3),
    TOPLAMSAKATDOKUNAN NUMERIC(12,3),
    KALANMIKTAR NUMERIC(12,3),
    MUSTERISIPARISNO VARCHAR(50),
    LOT VARCHAR(150),
    ORMESIPARISSEKLIID INTEGER,
    ORMESIPARISSEKLIADI VARCHAR(15),
    DOKUMASIPARISTIPIID INTEGER,
    DOKUMASIPARISTIPIADI VARCHAR(30))
AS
declare variable Bitti integer;
declare variable FHamStokBirimID        integer;
begin
  for Select DokumaPlanlama.DokumaPlanlamaId, DokumaPlanlama.BoyaSiparisKayitId, DokumaPlanlama.OrmeSiparisDetayId, OrmeSiparisDetay.SiparisSiraNo,
             BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId,
             DokumaPlanlama.UretimPartiNo, OrmeSiparisDetay.HamId, Ham.Adi, DokumaPlanlama.Bitti,
             OrmeSiparisDetay.HamMiktar, OrmeSiparisDetay.ToplamDokunan, OrmeSiparisDetay.ToplamSakatDokunan,
             (OrmeSiparisDetay.HamMiktar - OrmeSiparisDetay.ToplamDokunan - OrmeSiparisDetay.StoktanKullanilacakMiktar - OrmeSiparisDetay.SatinAlinacakMiktar + OrmeSiparisDetay.ToplamSakatDokunan),
             DokumaPlanlama.PlanlananMiktar, DokumaPlanlama.GerceklesenMiktar, DokumaPlanlama.ToplamSakatDokunan,
             OrmeSiparisDetay.MusteriSiparisNo,
             OrmeSiparisDetay.RenkId, Renk.Adi, DokumaPlanlama.Lot, BoyaSiparis.OrmeSiparisSekliId, OrmeSiparisSekli.Adi, BoyaSiparis.DokumaSiparisTipiId, DokumaSiparisTipi.Adi, OrmeSiparisDetay.HamStokBirimID from DokumaPlanlama
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = DokumaPlanlama.OrmeSiparisDetayId
    left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmeSiparisDetay.BoyaSiparisKayitId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join OrmeSiparisSekli On OrmeSiparisSekli.OrmeSiparisSekliId = BoyaSiparis.OrmeSiparisSekliId
    left join DokumaSiparisTipi On DokumaSiparisTipi.DokumaSiparisTipiId = BoyaSiparis.DokumaSiparisTipiId
  where (:ADokumaMakinaId is null and :AFirmaId is null) or
        (:ADokumaMakinaId is null and :AFirmaId is not null and DokumaPlanlama.FirmaId = :AFirmaId) or
        (:AFirmaId is null and :ADokumaMakinaId is not null and DokumaPlanlama.DokumaMakinaId = :ADokumaMakinaId) or
        (:AFirmaId is Not null and :ADokumaMakinaId is not null and DokumaPlanlama.DokumaMakinaId = :ADokumaMakinaId and DokumaPlanlama.FirmaId = :AFirmaId) and OrmeSiparisDetay.Kapali = 0
  into :DokumaPlanlamaId, :BoyaSiparisKayitId, :OrmeSiparisDetayId, :SiparisSiraNo, :BoyaSiparisYil, :BoyaSiparisId,
       :UretimPartiNo, :HamId, :HamAdi, :Bitti, :SiparisMiktari, :SiparisToplamDokunan, :SiparisToplamSakatDokunan, :SiparisDokunanMiktar,
       :PlanlananMiktar, :GerceklesenMiktar, :ToplamSakatDokunan, :MusteriSiparisNo, :RenkId, :RenkAdi, :Lot, :OrmeSiparisSekliId, :OrmeSiparisSekliAdi, :DokumaSiparisTipiId, :DokumaSiparisTipiAdi, :FHamStokBirimID do
  begin
    if (Bitti = 0) then
    begin
      ResultId = '"' || EgemenIntegerToStr(DokumaPlanlamaId) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisKayitId) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(OrmeSiparisDetayId) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisYil) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(BoyaSiparisId) || '",' ;
      ResultId = ResultId || '"' || EgemenIntegerToStr(DokumaSiparisTipiId) || '",' ;
      ResultId = ResultId || '"' || UretimPartiNo || '",' ;
      if (RenkId is null) then
        ResultId = ResultId || '"",' ;
      else
        ResultId = ResultId || '"' || RenkId || '",' ;

      ResultId = ResultId || '"' || EgemenIntegerToStr(HamId) || '",' ;

      if (Lot is not null) then
        ResultId = ResultId || '"' || Lot || '",' ;
      else
        ResultId = ResultId || '"",' ;

      ResultId = ResultId || '"' || EgemenIntegerToStr(FHamStokBirimID) || '"' ;

      KalanMiktar = PlanlananMiktar - GerceklesenMiktar + ToplamSakatDokunan;
      BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') ' || EgemenIntegerToStr(BoyaSiparisId) || ' ';

      suspend;
    end
  end
end
