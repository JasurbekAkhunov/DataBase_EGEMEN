CREATE PROCEDURE ISLETMETOPLAMFIRMADAGILIMI(
    AFASON INTEGER,
    APARTITIPIID INTEGER,
    ARAPORTIPI INTEGER)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    KILO DOUBLE PRECISION,
    METRE DOUBLE PRECISION)
AS
declare Variable FLastFirmaID       varchar(20);
declare Variable FFirmaID       varchar(20);
declare Variable FFirmaAdi      varchar(100);
declare variable IslemTipi      integer;
declare variable Bitti          integer;
declare variable FaturaKesilecek integer;
declare variable FKilo           Numeric(12,3);
declare variable FMetre          Numeric(12,3);
declare variable itTamir        integer;
declare variable Okey integer;
begin
  Select itTamir from IslemTipiGetir into :itTamir;

  FLastFirmaID = '';

  for select Parti.FirmaId, Firma.Adi, Parti.IslemTipiId, Parti.FaturaKesilecek, PartiTamirSinif.Bitti, sum(PartiStok.Kilo), sum(PartiStok.Metre) from PartiStok
    left join PartiTamirSinif on PartiTamirSinif.PartiSinifId = PartiStok.PartiSinifId and PartiTamirSinif.Bitti = 0
    left join Parti on Parti.PartiKayitId = PartiStok.PartiKayitId
    left join Firma on Firma.FirmaId = Parti.FirmaId
    where durumu < 7 and Parti.Fason = :AFason and Parti.PartiTipiId = :APartiTipiId
    group by Parti.FirmaId, Firma.Adi, Parti.IslemTipiId, Parti.FaturaKesilecek, PartiTamirSinif.Bitti
    order by Parti.FirmaId
    into :FFirmaId, :FFirmaAdi, :IslemTipi, :FaturaKesilecek, :Bitti, :FKilo, :FMetre do
  begin
    if (FLastFirmaID <> FFirmaID) then
    begin
      if (FLastFirmaID <> '' and (Kilo <> 0 or Metre <> 0)) then suspend;
      FLastFirmaID = FFirmaID;
      FirmaID = FFirmaId;
      FirmaAdi = FFirmaAdi;
      Kilo = 0;
      Metre = 0;
    end

    Okey = 1;

    if (ARaporTipi = 0 and (IslemTipi = itTamir or  Bitti is not null)) then Okey = 0;
    else if (ARaporTipi = 1 and (IslemTipi <> itTamir or FaturaKesilecek = 0)) then Okey = 0;
    else if (ARaporTipi = 2 and (IslemTipi <> itTamir or FaturaKesilecek = 1)) then Okey = 0;
    else if (ARaporTipi = 3 and (IslemTipi = itTamir or Bitti is null)) then Okey = 0;

    if (Okey = 1) then
    begin
      Kilo = Kilo + FKilo;
      Metre = Metre + FMetre;
    end
  end

  if (FLastFirmaID <> '' and (Kilo <> 0 or Metre <> 0)) then suspend;
end
