CREATE PROCEDURE SABLONBASKIPERFORMANSRAPORU(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP)
RETURNS (
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    SABLONCEKIMDESENSAYISI INTEGER,
    SABLONSAYISI INTEGER,
    BASKIVARYANTSAYISI INTEGER,
    BASKIMETRE NUMERIC(12,3))
AS
declare variable FFirmaId                         varchar(20);
declare variable FLastFirmaId                     varchar(20);
declare variable FFirmaAdi                        varchar(100);

declare variable FDesenvaryantId integer;
declare variable FLastDesenvaryantId integer;

declare variable FHamTipiID   integer;
declare variable FMetre       Numeric(12,3);
declare variable FCikisMetre  Numeric(12,3);
declare variable FDesenId     varchar(20);
declare variable FSablonSayisi integer;
begin
  SablonCekimDesenSayisi  = 0;
  SablonSayisi            = 0;
  BaskiVaryantSayisi      = 0;
  BaskiMetre              = 0;

  FLastFirmaId = '';

  For Select Firma.FirmaId, Firma.Adi, Desen.DesenId, Desen.SablonSayisi from DesenIstekleri
    left join DesenIstekTipi on DesenIstekTipi.DesenIstekTipiID = DesenIstekleri.DesenIstekTipiID
    left join Desen on Desen.DesenId = DesenIstekleri.DesenId
    left join Firma on Firma.FirmaID = Desen.FirmaID
    left join Kullanici On Kullanici.KullaniciID = DesenIstekleri.IstekYapanKullaniciId
    where DesenIstekleri.BaslamaTarihi >= :ABaslangicTarihi and DesenIstekleri.BaslamaTarihi < :ABitisTarihi + 1 and DesenIstekTipi.SablonCekimmi = 1
    ORDER BY Firma.FirmaId, Firma.Adi, Desen.DesenId
    into :FFirmaId, :FFirmaAdi, :FDesenId, :FSablonSayisi do
  begin
    if (FFirmaID <> FLastFirmaId) then
    begin
      if (FLastFirmaId <> '') then suspend;
      FirmaID = FFirmaId;
      FirmaAdi = FFirmaAdi;
      FLastFirmaID = FFirmaId;

      SablonCekimDesenSayisi  = 0;
      SablonSayisi            = 0;
      BaskiVaryantSayisi      = 0;
      BaskiMetre              = 0;
    end
    SablonCekimDesenSayisi  = SablonCekimDesenSayisi + 1;
    SablonSayisi            = SablonSayisi + FSablonSayisi;

    FLastDesenvaryantId= -1;

    For Select BoyaSiparisBaski.DesenVaryantId,  PartiStok.HamTipiId, PartiAsamaStokDetay.Metre, PartiAsamaStokDetay.CikisMetre from Desen
      left join BoyaSiparisBaski on BoyaSiparisBaski.DesenId = Desen.DesenId
      left join PartiStok on PartiStok.BoyaSiparisBaskiId = BoyaSiparisBaski.BoyaSiparisBaskiId
      left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiStok.PartiSinifId
      left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
      left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PartiAsamalariID
      left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariID = PartiAsamalari.PartiAsamalariID and PartiAsamaStokDetay.PartiSinifId = PartiStok.PartiSinifId
      where Desen.DesenId = :FDesenId and BoyaSiparisBaski.BoyaSiparisBaskiId is not null and partiStok.BoyaSiparisBaskiId is not null and Asama.BaskiAsamasimi = 1
      order by BoyaSiparisBaski.DesenVaryantId
      into :FDesenVaryantId, :FHamTipiId, :FMetre, :FCikisMetre do
    begin
      if (FLastDesenvaryantId <> FDesenvaryantId) then BaskiVaryantSayisi = BaskiVaryantSayisi + 1;

      if (FHamTipiID = 2) then BaskiMetre = BaskiMetre + :FMetre;
      else BaskiMetre = BaskiMetre + :FCikisMetre;
    end
  end

  if (FLastFirmaId <> '') then suspend;
end
