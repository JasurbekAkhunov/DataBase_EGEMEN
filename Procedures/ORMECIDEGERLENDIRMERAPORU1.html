CREATE PROCEDURE ORMECIDEGERLENDIRMERAPORU1(
    ABASLANGICTARIHI TIMESTAMP,
    ABITISTARIHI TIMESTAMP,
    AFIRMAID VARCHAR(20))
RETURNS (
    ORMEPLANLAMAID INTEGER,
    TARIH TIMESTAMP,
    BASLAMATARIHI TIMESTAMP,
    TERMINTARIHI TIMESTAMP,
    GERCEKLESENTESLIMTARIHI TIMESTAMP,
    BOYASIPARISKAYITID INTEGER,
    ORMESIPARISDETAYID INTEGER,
    SIPARISSIRANO INTEGER,
    URETIMPARTINO VARCHAR(30),
    FIRMAID VARCHAR(20),
    FIRMAADI VARCHAR(100),
    BOYASIPARISYIL INTEGER,
    BOYASIPARISID INTEGER,
    BOYASIPARISIDSTR VARCHAR(30),
    SIPARISFIRMAID VARCHAR(20),
    SIPARISFIRMAADI VARCHAR(100),
    HAMID INTEGER,
    HAMADI VARCHAR(255),
    RENKADI VARCHAR(50),
    MUSTERISIPARISNO VARCHAR(50),
    REFERANSSIPARISNO VARCHAR(30),
    ARTIKELNO VARCHAR(20),
    TUPACIKENMAYLI INTEGER,
    TUPACIKENMAYLISTR VARCHAR(10),
    SIPTUPACIKENMAYLI INTEGER,
    SIPTUPACIKENMAYLISTR VARCHAR(10),
    TALIMATMIKTARI DOUBLE PRECISION,
    GERCEKLESENMIKTAR DOUBLE PRECISION,
    KALANMIKTAR DOUBLE PRECISION,
    TERMIN INTEGER)
AS
declare variable FGerceklesenTeslimTarihi  Double precision;
declare variable shtFasonUretim            integer;
begin
  Select shtFasonUretim from OrmeStokHareketTipiGetir into :shtFasonUretim;

  for Select OrmePlanlama.OrmePlanlamaId, OrmePlanlama.Tarih, OrmePlanlama.BaslamaTarihi, OrmePlanlama.TerminTarihi, OrmePlanlama.UretimPartiNo, OrmePlanlama.FirmaId, OPF.Adi,
         OrmePlanlama.BoyaSiparisKayitId, OrmePlanlama.OrmeSiparisDetayId, BoyaSiparis.BoyaSiparisYil, BoyaSiparis.BoyaSiparisId, OrmeSiparisDetay.HamId, Ham.Adi, Renk.Adi,
         OrmePlanlama.PlanlananMiktar, OrmePlanlama.GerceklesenMiktar, OrmeSiparisDetay.MusteriSiparisNo, OrmeSiparisDetay.ReferansSiparisNo, BoyaSiparis.FirmaId, ASF.Adi,
         OrmeSiparisDetay.SiparisSiraNo, OrmeSiparisDetay.ArtikelNo, OrmePlanlama.TupAcikEnMayli, OrmeSiparisDetay.TupAcikEnMayli  from OrmePlanlama
    left join OrmeSiparisDetay on OrmeSiparisDetay.OrmeSiparisDetayId = OrmePlanlama.OrmeSiparisDetayId
    left join BoyaSiparis on BoyaSiparis.BoyaSiparisKayitId = OrmePlanlama.BoyaSiparisKayitId
    left join Ham on Ham.HamId = OrmeSiparisDetay.HamId
    left join Renk on Renk.RenkId = OrmeSiparisDetay.RenkId
    left join OrmeMakina on OrmeMakina.OrmeMakinaId = OrmePlanlama.OrmeMakinaId
    left join Firma OPF on OPF.FirmaId = OrmePlanlama.FirmaId
    left join Firma ASF on ASF.FirmaId = BoyaSiparis.FirmaId
    where OrmePlanlama.Tarih >= :ABaslangicTarihi and OrmePlanlama.Tarih <= :ABitisTarihi and OrmePlanlama.FirmaID is not null and
         (OrmePlanlama.FirmaID = :AFirmaID or :AFirmaID is null)
    into :OrmePlanlamaId, :Tarih, :BaslamaTarihi, :TerminTarihi, :UretimPartiNo, :FirmaId, :FirmaAdi,
         :BoyaSiparisKayitId, :OrmeSiparisDetayId, :BoyaSiparisYil, :BoyaSiparisId, :HamId, :HamAdi, :RenkAdi, :TalimatMiktari, :GerceklesenMiktar,
         :MusteriSiparisNo, :ReferansSiparisNo, :SiparisFirmaId, :SiparisFirmaAdi, :SiparisSiraNo,
         :ArtikelNo, :TupAcikEnMayli, :SipTupAcikEnMayli do
  begin
    BoyaSiparisIdStr = '(' || EgemenIntegerToStr(BoyaSiparisYil) || ') '  || EgemenIntegerToStr(BoyaSiparisId);

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:TupAcikenMayli) into :TupAcikEnMayliStr;

    Select TupAcikEnMayliStr from TupAcikEnMayliStrGetir(:SipTupAcikenMayli) into :SipTupAcikEnMayliStr;

    KalanMiktar = TalimatMiktari - GerceklesenMiktar;
    if (KalanMiktar < 0) then KalanMiktar = 0;

    Termin = 0;

    GerceklesenTeslimTarihi = Null;
    Termin = 0;

    if (GerceklesenMiktar > 0) then
    begin
      FGerceklesenTeslimTarihi = Null;

      Select Sum(TimeStampToDouble(StokHareket.Tarih) * StokHareket.Miktar) / Sum(StokHareket.Miktar) from StokHareket
         where StokHareket.OrmePlanlamaID = :OrmePlanlamaId and StokHareket.HareketTipi = :shtFasonUretim and StokHareket.hamIade = 0
         into :FGerceklesenTeslimTarihi;

      if (FGerceklesenTeslimTarihi is not null) then
      begin
        FGerceklesenTeslimTarihi = Yuvarla(FGerceklesenTeslimTarihi, 1);
        GerceklesenTeslimTarihi = DoubleToTimeStamp(FGerceklesenTeslimTarihi);

        if (GerceklesenTeslimTarihi is null) then Termin = -1;
        else if (TerminTarihi < GerceklesenTeslimTarihi) then Termin = -1;
        else Termin = 1;
      end
    end
    else if (TerminTarihi < Current_Date) then Termin = -1;
    suspend;
  end
end
