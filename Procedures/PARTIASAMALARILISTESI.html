CREATE PROCEDURE PARTIASAMALARILISTESI(
    APARTIKAYITID INTEGER)
RETURNS (
    PARTIASAMALARIID INTEGER,
    ASAMAADI VARCHAR(40),
    ASAMAID INTEGER,
    RECETEID VARCHAR(20),
    RECETEADI VARCHAR(40),
    SIRA INTEGER)
AS
begin
  for select PartiAsamalari.PartiAsamalariId, PartiAsamalari.Sira, PartiAsamalari.ReceteId, Recete.Adi, Asama.AsamaId, Asama.Adi from PartiAsamalari
  left join Recete on Recete.ReceteId = PartiAsamalari.ReceteId
  left join PartiAsamaStok on PartiAsamaStok.PartiAsamalariId = PartiAsamalari.PartiAsamalariId
  left join Asama on Asama.AsamaId = PartiAsamaStok.AsamaId
  where PartiAsamalari.PartiKayitId = :APartiKayitId and PartiAsamaStok.GirisTarihi is not null
  group by PartiAsamalari.PartiAsamalariId, PartiAsamalari.Sira, PartiAsamalari.ReceteId, Recete.Adi, Asama.AsamaId, Asama.Adi
  order by PartiAsamalari.sira
  into :PartiAsamalariId, :Sira, :ReceteId, :ReceteAdi, :AsamaId, :AsamaAdi do
  begin
    suspend;
  end
end
