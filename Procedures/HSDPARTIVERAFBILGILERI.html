CREATE PROCEDURE HSDPARTIVERAFBILGILERI(
    HAMSTOKDETAYID INTEGER)
RETURNS (
    PARTILENENKAPADEDI INTEGER,
    PARTILENENKILO NUMERIC(12,3),
    PARTILENENMETRE NUMERIC(12,3),
    PARTILENMISRAFTAKILO NUMERIC(12,3),
    PARTILENMISRAFTAMETRE NUMERIC(12,3),
    SARFFISIKILO NUMERIC(12,3),
    SARFFISIMETRE NUMERIC(12,3),
    AKTARMAYACIKANKILO NUMERIC(12,3),
    AKTARMADAKALANKILO NUMERIC(12,3),
    PARTIIPTALKILO NUMERIC(12,3),
    PARTIIPTALMETRE NUMERIC(12,3),
    PARTIIPTALCIKISKILO NUMERIC(12,3),
    PARTIIPTALRAFIDSTR VARCHAR(100),
    RAFSTR VARCHAR(50),
    RAFGRUPIDSTR VARCHAR(50),
    SATISKILO NUMERIC(12,3),
    SATISMETRE NUMERIC(12,3),
    STOKTANSATISKILO NUMERIC(12,3),
    STOKTANSATISMETRE NUMERIC(12,3),
    IADEEDILECEKKAPADEDI INTEGER,
    IADEEDILECEKKILO NUMERIC(12,3),
    IADEEDILECEKMETRE NUMERIC(12,3),
    IADEEDILENKAPADEDI INTEGER,
    IADEEDILENKILO NUMERIC(12,3),
    IADEEDILENMETRE NUMERIC(12,3),
    KALANKAPADEDI INTEGER,
    KALANKILO NUMERIC(12,3),
    KALANMETRE NUMERIC(12,3),
    REZERVEDILENKILO NUMERIC(12,3),
    REZERVEDILENMETRE NUMERIC(12,3),
    REZERVSONRASIKALANKILO NUMERIC(12,3),
    REZERVSONRASIKALANMETRE NUMERIC(12,3),
    PARTILENENADET INTEGER,
    IADEEDILECEKADET INTEGER,
    IADEEDILENADET INTEGER,
    KALANADET INTEGER,
    PARTILEMECIKISMIKTARI BOYAMIKTARSIZE,
    PARTILEMECIKISMETRE BOYAMIKTARSIZE,
    PARTILEMECIKISKILO BOYAMIKTARSIZE,
    SARFKAPADEDI INTEGER)
AS
declare variable AktarmadanKullanilanKilo Numeric(12,3);
declare variable FRafId     varchar(10);
declare variable FRafGrupId     varchar(4);
declare variable HamTipiId             integer;
declare variable KapAdedi Numeric(12,3);
declare variable Kilo Numeric(12,3);
declare variable Metre Numeric(12,3);
declare variable Adet Numeric(12,3);
declare variable StokCikisOran Double Precision;
declare variable FPartiSinifId integer;
declare variable FSira integer;
declare variable PartiSinifId integer;
declare variable Sira integer;
declare variable FPartilemeCikisMiktari BoyaMiktarSize;
declare variable FPartilemeCikisMetre   BoyaMiktarSize;
declare variable FPartilemeCikisKilo    BoyaMiktarSize;
declare variable PartiPartilemeCikisMiktari BoyaMiktarSize;
declare variable PartiPartilemeCikisMetre   BoyaMiktarSize;
declare variable PartiPartilemeCikisKilo    BoyaMiktarSize;
declare variable Oran Double Precision;
declare variable KapAdediSatis integer;
begin

  select HamStokDetay.RezervEdilenKilo, HamStokDetay.RezervEdilenMetre, Ham.HamTipiId,
    HamStokDetay.KapAdedi, HamStokDetay.Kilo, HamStokDetay.Metre, HamStokDetay.Adet from HamStokDetay
    left join Ham on Ham.HamId = HamStokDetay.HamId
    where HamStokDetay.HamStokDetayId = :HamStokDetayId
    into :RezervEdilenKilo, :RezervEdilenMetre, :HamTipiId,
    :KapAdedi, :Kilo, :Metre, :Adet;

  Select sum(PartiStokDetay.KapAdedi), sum(PartiStokDetay.Kilo), sum(PartiStokDetay.Metre), sum(PartiStokDetay.Adet) from RafDetay
  left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
  where RafDetay.HamStokDetayId = :HamStokDetayId
  into :PartilenenKapAdedi, :PartilenenKilo,  :PartilenenMetre, :PartilenenAdet;

  /*Hamdan üretilen partilerin partileme aşamasında çıkan miktarlar*/
  PartilemeCikisMiktari = 0;
  PartilemeCikisMetre = 0;
  PartilemeCikisKilo = 0;
  PartiSinifId = 0;
  Sira = 0;
  for select PartiAsamaStokDetay.CikisMiktar, PartiAsamaStokDetay.CikisMetre, PartiAsamaStokDetay.CikisKilo, PartiStokDetay.PartiSinifId, PartiAsamalari.Sira from RafDetay
    left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
    left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiStokDetay.PartiSinifId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PArtiAsamalariId
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifID = PartiStokDetay.PartiSinifID
    left join Asama on Asama.Asamaid = PartiAsamalari.AsamaId
    where RafDetay.HamStokDetayId = :HamStokDetayId and Asama.PartilemeMi = 1
    order by PartiStokDetay.PartiSinifId, PartiAsamalari.Sira into :FPartilemeCikisMiktari, :FPartilemeCikisMetre, :FPartilemeCikisKilo, :FPartiSinifId, :FSira do
  begin
    if (PartiSinifId <> 0 and FPartiSinifId <> PartiSinifId) then
    begin
        select iif(Sum(PartiStokDetay.Miktar) = 0, 0, iif(Sum(PartiStokDetay.Miktar) > 0, Sum(iif(RafDetay.HamStokDetayId = :HamStokDetayId, PartiStokDetay.Miktar, 0)) / Sum(PartiStokDetay.Miktar), 0))
        from PartiStokDetay
        left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
        where PartiStokDetay.PartiSinifId = :PartiSinifId into :Oran;

        PartilemeCikisMiktari = PartilemeCikisMiktari + (PartiPartilemeCikisMiktari * Oran);
        PartilemeCikisMetre = PartilemeCikisMetre + (PartiPartilemeCikisMetre * Oran);
        PartilemeCikisKilo = PartilemeCikisKilo + (PartiPartilemeCikisKilo * Oran);
    end

    if (PartiSinifId = 0 or FPartiSinifId <> PartiSinifId) then
    begin
      Sira = FSira;
      PartiSinifId = FPartiSinifId;

      PartiPartilemeCikisMiktari = 0;
      PartiPartilemeCikisMetre = 0;
      PartiPartilemeCikisKilo = 0;
    end

    if (FSira = Sira) then
    begin
      PartiPartilemeCikisMiktari = PartiPartilemeCikisMiktari + FPartilemeCikisMiktari;
      PartiPartilemeCikisMetre = PartiPartilemeCikisMetre + FPartilemeCikisMetre;
      PartiPartilemeCikisKilo = PartiPartilemeCikisKilo + FPartilemeCikisKilo;
    end
  end
  if (PartiSinifId <> 0) then
  begin
    select iif(Sum(PartiStokDetay.Miktar) = 0, 0, Sum(iif(RafDetay.HamStokDetayId = :HamStokDetayId, PartiStokDetay.Miktar, 0)) / Sum(PartiStokDetay.Miktar))
    from PartiStokDetay
    left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
    where PartiStokDetay.PartiSinifId = :PartiSinifId into :Oran;

    PartilemeCikisMiktari = PartilemeCikisMiktari + (PartiPartilemeCikisMiktari * Oran);
    PartilemeCikisMetre = PartilemeCikisMetre + (PartiPartilemeCikisMetre * Oran);
    PartilemeCikisKilo = PartilemeCikisKilo + (PartiPartilemeCikisKilo * Oran);
  end

  /* üste alindi
  select
  Sum(CikisMiktar * (select Sum(iif(RafDetay.HamStokDetayId = :HamStokDetayId, PartiStokDetay.Miktar, 0)) / Sum(PartiStokDetay.Miktar)
                      from PartiStokDetay
                      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
                      where PartiStokDetay.PartiSinifId = PartiSinifId_)),
  Sum(CikisMetre * (select Sum(iif(RafDetay.HamStokDetayId = :HamStokDetayId, PartiStokDetay.Miktar, 0)) / Sum(PartiStokDetay.Miktar)
                      from PartiStokDetay
                      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
                      where PartiStokDetay.PartiSinifId = PartiSinifId_)),
  Sum(CikisKilo * (select Sum(iif(RafDetay.HamStokDetayId = :HamStokDetayId, PartiStokDetay.Miktar, 0)) / Sum(PartiStokDetay.Miktar)
                      from PartiStokDetay
                      left join RafDetay on RafDetay.RafDetayId = PartiStokDetay.RafDetayId
                      where PartiStokDetay.PartiSinifId = PartiSinifId_))
  from (
    select Sum(PartiAsamaStokDetay.CikisMiktar) CikisMiktar, Sum(PartiAsamaStokDetay.CikisMetre) CikisMetre, Sum(PartiAsamaStokDetay.CikisKilo) CikisKilo, PartiStokDetay.PartiSinifId PartiSinifId_ from RafDetay
    left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
    left join PartiAsamaStok on PartiAsamaStok.PartiSinifId = PartiStokDetay.PartiSinifId
    left join PartiAsamalari on PartiAsamalari.PartiAsamalariId = PartiAsamaStok.PArtiAsamalariId
    left join PartiAsamaStokDetay on PartiAsamaStokDetay.PartiAsamalariId = PartiAsamalari.PartiAsamalariId and PartiAsamaStokDetay.PartiSinifID = PartiStokDetay.PartiSinifID
    left join Asama on Asama.Asamaid = PartiAsamalari.AsamaId
    where RafDetay.HamStokDetayId = :HamStokDetayId and Asama.PartilemeMi = 1
    group by PartiStokDetay.PartiSinifId
  )
  into :PartilemeCikisMiktari, :PartilemeCikisMetre, :PartilemeCikisKilo; */

/* partilenmiş ama ham depoda olanları hesaplar */
  Select Sum(PartiStokDetay.Kilo), Sum(PartiStokDetay.Metre) from RafDetay

  left join PartiStokDetay on PartiStokDetay.RafDetayId = RafDetay.RafDetayId
  left join PartiStok on PartiStok.PartiSinifId = PartiStokDetay.PartiSinifId
  where RafDetay.HamStokDetayId = :HamStokDetayId and PartiStokDetay.PartiSinifId is not null and PartiStok.PartiAsamalariId is null and PartiStok.partiStokIptalKayitId is null
  into :PartilenmisRaftaKilo, :PartilenmisRaftaMetre;

  select sum(HamStokSarfFisiDetay.Kilo), sum(HamStokSarfFisiDetay.Metre), sum(HamStokSarfFisiDetay.kapadedi) from RafDetay
  left join HamStokSarfFisiDetay on HamStokSarfFisiDetay.RafDetayId = RafDetay.RafDetayId
  where RafDetay.HamStokDetayId = :HamStokDetayId
  into :SarfFisiKilo, :SarfFisiMetre, :SarfKapAdedi;

  AktarmadanKullanilanKilo = 0;
  AktarmayaCikanKilo = 0;
  AktarmadaKalanKilo = 0;

  if (HamTipiId = 0) then
  begin
    Select Sum(Kilo - IadeEdilenKilo) from HAktarmaFisiDetay where HAktarmaFisiDetay.HamStokDetayId = :HamStokDetayId into :AktarmayaCikanKilo;
    Select Sum(RafDetay.IadeEdilenKilo + RafDetay.PartilenenKilo + RafDetay.SatisKilo + RafDetay.AktarmayaCikanKilo - RafDetay.AktarmadanIadeEdienKilo) from RafDetay where RafDetay.HamStokDetayID = :HamStokDetayId and RafDetay.Aktarma = 1 into :AktarmadanKullanilanKilo;
    AktarmadaKalanKilo = AktarmayaCikanKilo - AktarmadanKullanilanKilo;
  end

  select sum(RafDetay.Kilo), sum(RafDetay.Metre), Sum(PartiStok.SonCikisKilo) from RafDetay
  left join PartiStokIptal on PartiStokIptal.PartiStokIptalKayitId = RafDetay.PartiStokIptalKayitId
  left join PartiStok on PartiStok.PartiSinifId = PartiStokIptal.PartiSinifId
  where RafDetay.HamStokDetayId = :HamStokDetayId and RafDetay.PartiStokIptalKayitId is not null
  into :PartiIptalKilo, :PartiIptalMetre, :PartiIptalCikisKilo;

  PartiIptalRafIDStr = '';
  if (PartiIptalCikisKilo is not null) then
  begin
    for select PRD.RafId from RafDetay RD
    left join PartiStokIptal on PartiStokIptal.PartiStokIptalKayitId = RD.PartiStokIptalKayitId
    left join PartiStok on PartiStok.PartiSinifId = PartiStokIptal.PartiSinifId
    left join PartiStokDetay on PartiStokDetay.PartiSinifId = PartiStok.PartiSinifId
    left join RafDetay PRD on PRD.RafDetayId = PartiStokDetay.RafDetayId
    where RD.HamStokDetayId = :HamStokDetayId and RD.PartiStokIptalKayitId is not null
    group by PRD.RafId
    into :FRafID do
    begin
      if (Char_Length(PartiIptalRafIDStr) + Char_Length(FRafId) < 97) then
      begin
       if (PartiIptalRafIDStr <> '') then PartiIptalRafIDStr = PartiIptalRafIDStr || ',';
       PartiIptalRafIDStr = PartiIptalRafIDStr || FRafId;
      end
    end
  end

  RafStr = '';
  RafGrupIdStr = '';

  for select RafDetay.RafId from HamStokDetay
    left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId
    where HamStokDetay.HamStokDetayId = :HamStokDetayId and RafDetay.RafDetayID is not null and RafDetay.StokVar = 1
    Group By RafDetay.RafId
    into :FRafId do
  begin
    if (Char_Length(RafStr) + Char_Length(FRafId) < 48) then
    begin
     if (RafStr <> '' ) then RafStr = RafStr || cast(', ' as varchar(2) character set UTF8);
     RafStr = RafStr || FRafId;
    end
  end

  if (RafStr <> '') then
  begin
    for select Raf.RafGrupId from HamStokDetay
      left join RafDetay on RafDetay.HamStokDetayID = HamStokDetay.HamStokDetayId
      left join Raf on Raf.RafID = RafDetay.RafId
      where HamStokDetay.HamStokDetayId = :HamStokDetayId and RafDetay.RafDetayID is not null and RafDetay.StokVar = 1
      Group By Raf.RafGrupId
      into :FRafGrupId do
    begin
      if (Char_Length(RafGrupIdStr) + Char_Length(FRafGrupId) < 48) then
      begin
       if (RafGrupIdStr <> '' ) then RafGrupIdStr = RafGrupIdStr || cast(', ' as varchar(2) character set UTF8);
       RafGrupIdStr = RafGrupIdStr || FRafGrupId;
      end
    end
  end

  select sum(RafDetay.IadeEdilecekKapAdedi), sum(RafDetay.IadeEdilecekKilo), sum(RafDetay.IadeEdilecekMetre),
         sum(RafDetay.IadeEdilenKapAdedi), sum(RafDetay.IadeEdilenKilo), sum(RafDetay.IadeEdilenMetre),
         sum(RafDetay.IadeEdilecekAdet), sum(RafDetay.IadeEdilenAdet),
         Sum(RafDetay.SatisKilo), Sum(RafDetay.SatisMetre) from RafDetay
    where RafDetay.HamStokDetayId = :HamStokDetayId
    into :IadeEdilecekKapAdedi, :IadeEdilecekKilo, :IadeEdilecekMetre, :IadeEdilenKapAdedi, :IadeEdilenKilo,
         :IadeEdilenMetre, :IadeEdilecekAdet, :IadeEdilenAdet,
         :SatisKilo, :SatisMetre;

   select Sum(BoyaIrsaliyeDetay.KapAdedi) from RafDetay
      left join BoyaIrsaliyeDetay on BoyaIrsaliyeDetay.RafDetayID =  RafDetay.RafDetayID
      where RafDetay.HamStokDetayID = :HamStokDetayID and BoyaIrsaliyeDetay.HareketTipiID = 0 into :KapAdediSatis;

  if (PartilenenKapAdedi is null) then KalanKapAdedi = KapAdedi;
  else KalanKapAdedi = KapAdedi - PartilenenKapAdedi;
  if (IadeEdilecekKapAdedi is not null) then KalanKapAdedi = KalanKapAdedi - IadeEdilecekKapAdedi;
  if (IadeEdilenKapAdedi is not null) then KalanKapAdedi = KalanKapAdedi - IadeEdilenKapAdedi;
  if (KapAdediSatis is not null) then KalanKapAdedi = KalanKapAdedi - KapAdediSatis;
  if (sarfkapadedi is not null) then KalanKapAdedi = KalanKapAdedi - sarfkapadedi;

  if (PartilenenKilo is null) then KalanKilo = Kilo;
  else KalanKilo = Kilo - PartilenenKilo;
  if (IadeEdilecekKilo is not null) then KalanKilo = KalanKilo - IadeEdilecekKilo;
  if (IadeEdilenKilo is not null) then KalanKilo = KalanKilo - IadeEdilenKilo;

  if (PartilenenMetre is null) then KalanMetre = Metre;
  else KalanMetre = Metre - PartilenenMetre;
  if (IadeEdilecekMetre is not null) then KalanMetre = KalanMetre - IadeEdilecekMetre;
  if (IadeEdilenMetre is not null) then KalanMetre = KalanMetre - IadeEdilenMetre;

  if (PartiIptalKilo is null) then PartiIptalKilo = 0;
  if (PartiIptalMetre is null) then PartiIptalMetre = 0;

  if (SatisKilo is null) then SatisKilo = 0;
  if (SatisMetre is null) then SatisMetre = 0;

  if (SarfFisiKilo is null) then SarfFisiKilo = 0;
  if (SarfFisiMetre is null) then SarfFisiMetre = 0;

  KalanKilo = KalanKilo + PartiIptalKilo - SatisKilo - SarfFisiKilo;
  KalanMetre = KalanMetre + PartiIptalMetre - SatisMetre - SarfFisiMetre;

  if (PartilenenAdet is null) then KalanAdet = Adet;
  else KalanAdet = Adet - PartilenenAdet;
  if (IadeEdilecekAdet is not null) then KalanAdet = KalanAdet - IadeEdilecekAdet;
  if (IadeEdilenAdet is not null) then KalanAdet = KalanAdet - IadeEdilenAdet;

  if (KalanKilo is null) then KalanKilo = 0;
  if (KalanMetre is null) then KalanMetre = 0;

  RezervSonrasiKalanKilo = KalanKilo - RezervEdilenKilo;

  RezervSonrasiKalanMetre = RezervSonrasiKalanMetre  - RezervEdilenMetre;

  Suspend;
end
